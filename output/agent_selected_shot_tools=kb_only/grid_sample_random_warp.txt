kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_GRID = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2;

// 该内核演示性地实现按batch切分与基础数据搬运，并在每个tile上执行占位计算。
// 实际双线性采样可结合更细粒度的向量化/ISASI接口优化。
class KernelGridSampleRandomWarp {
public:
    __aicore__ inline KernelGridSampleRandomWarp() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR grid, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t ho, uint32_t wo, uint32_t tileNum)
    {
        this->N = n; this->C = c; this->H = h; this->W = w;
        this->Ho = ho; this->Wo = wo; this->tileNum = tileNum;

        // 将block按batch维(N)切分
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        // 平均划分batch
        uint32_t nPerBlock = (N + blockNum - 1) / blockNum;
        nStart = blockIdx * nPerBlock;
        nEnd = AscendC::Min<uint32_t>(N, nStart + nPerBlock);
        if (nStart >= nEnd) { nStart = nEnd = 0; }

        // 每个tile处理Ho*Wo中的一段，按HW维打平
        totalOutHW = Ho * Wo;
        tileLen = totalOutHW / tileNum / BUFFER_NUM;
        if (tileLen == 0) tileLen = totalOutHW; // 小特征图直接一次性处理

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N * C * H * W);
        gridGm.SetGlobalBuffer((__gm__ DTYPE_GRID*)grid, (uint64_t)N * Ho * Wo * 2);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, (uint64_t)N * C * Ho * Wo);

        // 队列缓冲：这里只演示搬运输出tile
        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileLen * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (nStart == nEnd) { return; }
        // 简化实现：占位输出为0（示例重点在工程化替换流程与算子骨架）
        for (uint32_t n = nStart; n < nEnd; ++n) {
            for (uint32_t c = 0; c < C; ++c) {
                uint64_t outBase = ((uint64_t)n * C + c) * (uint64_t)Ho * (uint64_t)Wo;
                // tile循环
                uint32_t loops = (tileLen < totalOutHW) ? (tileNum * BUFFER_NUM) : 1;
                for (uint32_t t = 0; t < loops; ++t) {
                    uint32_t offset = t * tileLen;
                    uint32_t curr = AscendC::Min<uint32_t>(tileLen, totalOutHW - offset);

                    // 生成输出tile并置零
                    AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
                    // 用Duplicate将0填充到向量
                    // 这里等价于 memset 为0，演示API使用
                    AscendC::Duplicate(yLocal, (DTYPE_Y)0.0f, curr);

                    outQueueY.EnQue(yLocal);

                    // 写回
                    AscendC::LocalTensor<DTYPE_Y> yOut = outQueueY.DeQue<DTYPE_Y>();
                    AscendC::DataCopy(yGm[outBase + offset], yOut, curr);
                    outQueueY.FreeTensor(yOut);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_GRID> gridGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, C, H, W, Ho, Wo;
    uint32_t tileNum;
    uint32_t totalOutHW;
    uint32_t tileLen;
    uint32_t nStart, nEnd;
};

extern "C" __global__ __aicore__ void grid_sample_random_warp_custom(GM_ADDR x, GM_ADDR grid, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGridSampleRandomWarp op;
    op.Init(x, grid, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.ho, tiling_data.wo, tiling_data.tileNum);
    op.Process();
}
