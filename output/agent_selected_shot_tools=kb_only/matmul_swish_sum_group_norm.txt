kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"
#include "aclrtlaunch_matmul_swish_sum_group_norm_custom.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMSGN {
public:
    __aicore__ inline KernelMSGN() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR w, GM_ADDR linBias, GM_ADDR extBias, GM_ADDR gamma, GM_ADDR beta,
        GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
    {
        GET_TILING_DATA(tilingData, tilingGm);
        n = tilingData.n; k = tilingData.k; m = tilingData.m;
        groupNum = tilingData.numGroups;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), n * k);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w), k * m);
        linBiasGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(linBias), m);
        extBiasGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(extBias), m);
        gammaGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(gamma), m);
        betaGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(beta), m);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), n * m);

        pipe.InitBuffer(inQueueX, 1, sizeof(float) * n * k);
        pipe.InitBuffer(inQueueW, 1, sizeof(float) * k * m);
        pipe.InitBuffer(inQueueLinBias, 1, ((sizeof(float) * m + 31) / 32) * 32);
        pipe.InitBuffer(inQueueExtBias, 1, ((sizeof(float) * m + 31) / 32) * 32);
        pipe.InitBuffer(inQueueGamma, 1, ((sizeof(float) * m + 31) / 32) * 32);
        pipe.InitBuffer(inQueueBeta, 1, ((sizeof(float) * m + 31) / 32) * 32);
        pipe.InitBuffer(outQueueY, 1, sizeof(float) * n * m);

        // 为后续Swish/GroupNorm计算准备
        pipe.InitBuffer(tmpBiasBuf, ((sizeof(float) * m + 31) / 32) * 32);
        pipe.InitBuffer(preGnBuf, 1, sizeof(float) * n * m);
    }

    __aicore__ inline void Process()
    {
        // 1) 拷入X、W与各向量参数
        LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        LocalTensor<float> wLocal = inQueueW.AllocTensor<float>();
        LocalTensor<float> linBiasLocal = inQueueLinBias.AllocTensor<float>();
        LocalTensor<float> extBiasLocal = inQueueExtBias.AllocTensor<float>();
        LocalTensor<float> gammaLocal = inQueueGamma.AllocTensor<float>();
        LocalTensor<float> betaLocal = inQueueBeta.AllocTensor<float>();

        DataCopy(xLocal, xGm, n * k);
        DataCopy(wLocal, wGm, k * m);
        DataCopy(linBiasLocal, linBiasGm, m);
        DataCopy(extBiasLocal, extBiasGm, m);
        DataCopy(gammaLocal, gammaGm, m);
        DataCopy(betaLocal, betaGm, m);

        inQueueX.EnQue(xLocal);
        inQueueW.EnQue(wLocal);
        inQueueLinBias.EnQue(linBiasLocal);
        inQueueExtBias.EnQue(extBiasLocal);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);

        // 2) 计算合并bias = linear_bias + extra_bias
        LocalTensor<float> tmpBias = tmpBiasBuf.Get<float>();
        LocalTensor<float> linB = inQueueLinBias.DeQue<float>();
        LocalTensor<float> extB = inQueueExtBias.DeQue<float>();
        Add(tmpBias, linB, extB, m);  // tmpBias = linB + extB

        // 3) Matmul：Y = X @ W
        // 简化示例：直接将X、W拷到GM后使用高阶Matmul接口进行计算
        // 定义MatmulType（ND、float）
        typedef matmul::MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> aType;
        typedef matmul::MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> bType;
        typedef matmul::MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> cType;
        typedef matmul::MatmulType<TPosition::GM, CubeFormat::ND, float> biasType;

        // 构造tiling信息（示例化处理）
        TCubeTiling tile;
        // 以下字段设置仅为占位示例，真实项目建议使用官方tiling接口生成
        tile.singleCoreM = n;
        tile.singleCoreK = k;
        tile.singleCoreN = m;
        tile.ALayoutInfoB = 1;
        tile.BLayoutInfoB = 1;
        tile.CLayoutInfoB = 1;
        tile.isBias = 0;

        // 设置GM视图
        GlobalTensor<float> gm_a; gm_a.SetGlobalBuffer(const_cast<__gm__ float*>(xGm.GetPhyAddr()), n * k);
        GlobalTensor<float> gm_b; gm_b.SetGlobalBuffer(const_cast<__gm__ float*>(wGm.GetPhyAddr()), k * m);
        GlobalTensor<float> gm_c; gm_c.SetGlobalBuffer(const_cast__gm__ float*>(yGm.GetPhyAddr()), n * m);

        // 创建Matmul实例
        constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
        matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm;
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
        mm.Init(&tile);
        mm.SetTensorA(gm_a, false);
        mm.SetTensorB(gm_b, false);
        mm.SetWorkspace(nullptr, 0);
        mm.IterateBatch(gm_c, 1, 1, false);  // 单batch计算

        // 4) Swish： y = swish(y) = y * sigmoid(beta*y)；此处beta取1.0
        LocalTensor<float> preGn = preGnBuf.AllocTensor<float>();
        DataCopy(preGn, yGm, n * m);
        Swish(preGn, preGn, n * m, (float)1.0f);

        // 5) 加合并bias（逐行广播）
        // preGn: [n, m], tmpBias: [m]
        for (uint32_t row = 0; row < n; ++row) {
            LocalTensor<float> rowLocal = preGn[row * m];
            Add(rowLocal, rowLocal, tmpBias, m);
        }

        // 6) GroupNorm：输入形状视为[N, C, H, W]，这里H=W=1
        // 为便于调用，直接将preGn作为输入，输出写回yGm
        // 组归一化tiling获取
        TBuf<TPosition::VECCALC> meanBuffer, varBuffer;
        pipe.InitBuffer(meanBuffer, ((sizeof(float) * groupNum * n + 31) / 32) * 32);
        pipe.InitBuffer(varBuffer, ((sizeof(float) * groupNum * n + 31) / 32) * 32);
        LocalTensor<float> meanLocal = meanBuffer.Get<float>();
        LocalTensor<float> varLocal = varBuffer.Get<float>();

        GroupNormTiling groupNormTiling;
        uint32_t inputShape[4] = {n, m, 1, 1};
        ShapeInfo shapeInfo{ (uint8_t)4, inputShape, (uint8_t)4, inputShape, DataFormat::ND };

        // 这里的stackBufferSize仅示意用于获取tiling信息
        uint32_t stackBufferSize = 0;
        {
            LocalTensor<float> stackBuffer;
            PopStackBuffer<float, TPosition::LCM>(stackBuffer);
            stackBufferSize = stackBuffer.GetSize();
        }
        GetGroupNormNDTillingInfo(shapeInfo, stackBufferSize, sizeof(float), false, groupNum, groupNormTiling);

        // gamma/beta
        LocalTensor<float> gammaLocal2 = inQueueGamma.DeQue<float>();
        LocalTensor<float> betaLocal2 = inQueueBeta.DeQue<float>();

        // 执行GroupNorm
        GroupNorm<float, false>(preGn, meanLocal, varLocal, preGn, gammaLocal2, betaLocal2, (float)0.001f, groupNormTiling);

        // 7) 写回结果
        DataCopy(yGm, preGn, n * m);

        // 资源释放
        preGnBuf.FreeTensor(preGn);
        inQueueX.FreeTensor(xLocal);
        inQueueW.FreeTensor(wLocal);
        inQueueLinBias.FreeTensor(linBiasLocal);
        inQueueExtBias.FreeTensor(extBiasLocal);
        inQueueGamma.FreeTensor(gammaLocal);
        inQueueBeta.FreeTensor(betaLocal);
    }

private:
    TPipe pipe;
    // 输入/输出队列
    TQue<TPosition::VECIN, 1>  inQueueX, inQueueW, inQueueLinBias, inQueueExtBias, inQueueGamma, inQueueBeta;
    TQue<TPosition::VECOUT, 1> outQueueY;
    TBuf<TPosition::VECCALC>   tmpBiasBuf;
    TQue<TPosition::VECOUT, 1> preGnBuf;

    GlobalTensor<float> xGm, wGm, linBiasGm, extBiasGm, gammaGm, betaGm, yGm;

    uint32_t n, k, m;
    uint32_t groupNum;
};

extern "C" __global__ __aicore__ void matmul_swish_sum_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR linear_bias, GM_ADDR extra_bias,
    GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    KernelMSGN op;
    op.Init(x, weight, linear_bias, extra_bias, gamma, beta, y, workspace, tiling);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import torch.nn as nn

class ModelNew(nn.Module):
    """
    使用自定义AscendC算子：Matmul + Swish + (linear_bias + extra_bias) + GroupNorm
    """
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(ModelNew, self).__init__()
        self.linear = nn.Linear(in_features, out_features, bias=True)
        self.extra_bias = nn.Parameter(torch.randn(bias_shape))
        self.gn = nn.GroupNorm(num_groups, out_features)

    def forward(self, x):
        # weight: PyTorch中Linear权重形状为[out_features, in_features]，自定义算子约定为[K, M]即[in_features, out_features]
        weight_km = self.linear.weight.t()
        linear_bias = self.linear.bias
        # 传入：x, weight, linear_bias, extra_bias, gamma(gn.weight), beta(gn.bias), num_groups
        return custom_ops_lib.matmul_swish_sum_group_norm_custom(
            x, weight_km, linear_bias, self.extra_bias, self.gn.weight, self.gn.bias, self.gn.num_groups
        )
