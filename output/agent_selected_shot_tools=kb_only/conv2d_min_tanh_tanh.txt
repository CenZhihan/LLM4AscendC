kernel_src="""
#include "kernel_operator.h"

// 本自定义核实现对输入特征图 x (N, C, H, W) 进行：沿通道C维取最小值(保留维度)-> Tanh -> Tanh
// 输出 y 的形状为 (N, 1, H, W)，数据类型为 half

constexpr int32_t BUFFER_NUM = 2;

class KernelMinTanhTanh {
public:
    __aicore__ inline KernelMinTanhTanh() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t N, uint32_t C, uint32_t H, uint32_t W, uint32_t tileNum)
    {
        this->N = N;
        this->C = C;
        this->H = H;
        this->W = W;
        this->tileNum = tileNum;

        planeSize = H * W;                    // 每个通道的像素数
        totalIn  = N * C * planeSize;         // 输入总元素
        totalOut = N * planeSize;             // 输出总元素

        xGm.SetGlobalBuffer((__gm__ half*)x, totalIn);
        yGm.SetGlobalBuffer((__gm__ half*)y, totalOut);

        // 每个tile处理的像素数量（按H*W平铺），最后一个tile可能不足
        tileLen = (planeSize + tileNum - 1) / tileNum;
        if (tileLen == 0) tileLen = planeSize;

        // 申请两个VECCALC本地缓冲，满足Tanh非原地约束及逐通道最小化
        pipe.InitBuffer(tmpQue, BUFFER_NUM, tileLen * sizeof(half));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t t = 0; t < tileNum; ++t) {
                uint32_t tileStart = t * tileLen;
                if (tileStart >= planeSize) {
                    break;
                }
                uint32_t curLen = planeSize - tileStart;
                if (curLen > tileLen) curLen = tileLen;

                // 分配用于运行最小值累积的缓冲A
                AscendC::LocalTensor<half> bufA = tmpQue.AllocTensor<half>();

                // 复制第0个通道当前tile的数据到bufA
                uint64_t base0 = (uint64_t)n * C * planeSize + (uint64_t)0 * planeSize + tileStart;
                AscendC::DataCopy(bufA, xGm[base0], curLen);

                // 逐通道取最小值，bufA保存运行中的最小值
                for (uint32_t c = 1; c < C; ++c) {
                    AscendC::LocalTensor<half> bufB = tmpQue.AllocTensor<half>();
                    uint64_t baseC = (uint64_t)n * C * planeSize + (uint64_t)c * planeSize + tileStart;
                    AscendC::DataCopy(bufB, xGm[baseC], curLen);
                    // 向量点对点最小值
                    AscendC::Min(bufA, bufA, bufB, static_cast<int32_t>(curLen));
                    tmpQue.FreeTensor(bufB);
                }

                // 连续两次Tanh：由于Tanh不允许原地，因此需要bufB做中间缓冲
                AscendC::LocalTensor<half> bufB2 = tmpQue.AllocTensor<half>();
                AscendC::Tanh(bufB2, bufA, curLen);
                AscendC::Tanh(bufA, bufB2, curLen);
                tmpQue.FreeTensor(bufB2);

                // 写回输出 (N,1,H,W) -> 展平为 N*H*W
                uint64_t yBase = (uint64_t)n * planeSize + tileStart;
                AscendC::DataCopy(yGm[yBase], bufA, curLen);

                tmpQue.FreeTensor(bufA);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQue;

    AscendC::GlobalTensor<half> xGm;
    AscendC::GlobalTensor<half> yGm;

    uint32_t N;
    uint32_t C;
    uint32_t H;
    uint32_t W;
    uint32_t planeSize;
    uint32_t totalIn;
    uint32_t totalOut;

    uint32_t tileNum;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMinTanhTanh op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.tileNum);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    先执行PyTorch的Conv2d，再用自定义AscendC算子完成：沿通道最小值 + Tanh + Tanh
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        x = self.conv(x)
        x = custom_ops_lib.conv2d_min_tanh_tanh_custom(x)
        return x
