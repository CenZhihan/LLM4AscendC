inline uint32_t out_dim(uint32_t in, uint32_t k, uint32_t s, uint32_t p, bool ceil_mode) {
    int64_t num = static_cast<int64_t>(in) + 2 * static_cast<int64_t>(p) - static_cast<int64_t>(k);
    if (ceil_mode) {
        return static_cast<uint32_t>(num >= 0 ? ( (num + s) - 1 ) / s + 1 : 0);
    } else {
        return static_cast<uint32_t>(num >= 0 ? num / s + 1 : 0);
    }
}

kernel_src="""
#include "kernel_operator.h"
#include <cstdint>

class KernelMaxPool3D {
public:
    __aicore__ inline KernelMaxPool3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t k, uint32_t s, uint32_t p,
                                uint32_t od, uint32_t oh, uint32_t ow,
                                uint64_t totalOut, uint32_t winElems)
    {
        this->n = n; this->c = c; this->d = d; this->h = h; this->w = w;
        this->k = k; this->s = s; this->p = p;
        this->od = od; this->oh = oh; this->ow = ow;
        this->totalOut = totalOut;
        this->winElems = winElems;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), static_cast<uint32_t>(n * c * d * h * w));
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), static_cast<uint32_t>(totalOut));

        // Per-block work range
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();
        uint64_t perBlock = totalOut / blocks;
        this->startIndex = perBlock * bid;
        this->endIndex = (bid == blocks - 1) ? totalOut : (perBlock * (bid + 1));

        // We compute one output element per iteration
        pipe.InitBuffer(outQueueY, 1, sizeof(float));
        // Window buffer and work buffer
        pipe.InitBuffer(tmpWinQueue, 1, winElems * sizeof(float));
        // Work buffer for ReduceMax: conservatively allocate 4x window size
        pipe.InitBuffer(tmpWorkQueue, 1, winElems * 4 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint64_t idx = startIndex; idx < endIndex; ++idx) {
            ComputeOne(idx);
            CopyOutOne(idx);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint64_t outLinearIdx)
    {
        // Decode linear index to (n,c,od,oh,ow)
        uint64_t tmp = outLinearIdx;
        uint32_t ow_ = static_cast<uint32_t>(tmp % ow); tmp /= ow;
        uint32_t oh_ = static_cast<uint32_t>(tmp % oh); tmp /= oh;
        uint32_t od_ = static_cast<uint32_t>(tmp % od); tmp /= od;
        uint32_t c_  = static_cast<uint32_t>(tmp % c);  tmp /= c;
        uint32_t n_  = static_cast<uint32_t>(tmp % n);

        // Allocate locals
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        AscendC::LocalTensor<float> winLocal = tmpWinQueue.AllocTensor<float>();
        AscendC::LocalTensor<float> workLocal = tmpWorkQueue.AllocTensor<float>();

        // Initialize window with very small numbers to emulate -inf
        float negInf = -1.0e30f;
        AscendC::Duplicate(winLocal, negInf, winElems);

        // compute input start indices
        int32_t d0 = static_cast<int32_t>(od_) * static_cast<int32_t>(s) - static_cast<int32_t>(p);
        int32_t h0 = static_cast<int32_t>(oh_) * static_cast<int32_t>(s) - static_cast<int32_t>(p);
        int32_t w0 = static_cast<int32_t>(ow_) * static_cast<int32_t>(s) - static_cast<int32_t>(p);

        // strides in GM
        uint64_t strideW = 1;
        uint64_t strideH = this->w;
        uint64_t strideD = static_cast<uint64_t>(this->h) * this->w;
        uint64_t strideC = static_cast<uint64_t>(this->d) * this->h * this->w;
        uint64_t strideN = static_cast<uint64_t>(this->c) * this->d * this->h * this->w;

        // Fill valid parts of the window
        for (uint32_t kd = 0; kd < k; ++kd) {
            int32_t id = d0 + static_cast<int32_t>(kd);
            if (id < 0 || id >= static_cast<int32_t>(d)) {
                continue;
            }
            for (uint32_t kh = 0; kh < k; ++kh) {
                int32_t ih = h0 + static_cast<int32_t>(kh);
                if (ih < 0 || ih >= static_cast<int32_t>(h)) {
                    continue;
                }
                int32_t wStart = w0;
                int32_t wEnd = w0 + static_cast<int32_t>(k);
                int32_t left = wStart < 0 ? 0 : wStart;
                int32_t right = wEnd > static_cast<int32_t>(w) ? static_cast<int32_t>(w) : wEnd;
                int32_t len = right - left;
                if (len <= 0) {
                    continue;
                }
                // destination offset inside window buffer
                uint32_t dstOff = kd * k * k + kh * k + static_cast<uint32_t>(left - wStart);
                // GM offset to read
                uint64_t gmOff = static_cast<uint64_t>(n_) * strideN +
                                 static_cast<uint64_t>(c_) * strideC +
                                 static_cast<uint64_t>(id) * strideD +
                                 static_cast<uint64_t>(ih) * strideH +
                                 static_cast<uint64_t>(left) * strideW;
                AscendC::DataCopy(winLocal[dstOff], xGm[gmOff], static_cast<uint32_t>(len));
            }
        }

        // ReduceMax over the window
        AscendC::ReduceMax(yLocal, winLocal, workLocal, static_cast<int32_t>(winElems), false);

        outQueueY.EnQue<float>(yLocal);
        tmpWinQueue.FreeTensor(winLocal);
        tmpWorkQueue.FreeTensor(workLocal);
    }

    __aicore__ inline void CopyOutOne(uint64_t outLinearIdx)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[outLinearIdx], yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpWinQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpWorkQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n, c, d, h, w;
    uint32_t k, s, p;
    uint32_t od, oh, ow;
    uint64_t totalOut;
    uint32_t winElems;
    uint64_t startIndex;
    uint64_t endIndex;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelMaxPool3D op;
    op.Init(x, y,
            tilingData.n, tilingData.c, tilingData.d, tilingData.h, tilingData.w,
            tilingData.k, tilingData.s, tilingData.p,
            tilingData.od, tilingData.oh, tilingData.ow,
            tilingData.totalOut, tilingData.winElems);
    op.Process();
}
