const uint32_t BLOCK_DIM = 8; // 针对示例N=8的场景，按batch划分到8个core
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    AddBiasFourDimBroadcastCustomTilingData tiling;

    uint32_t totalLength = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    uint32_t c = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());
    uint32_t n = BLOCK_DIM; // 本示例按N=8划分
    if (n == 0 || c == 0) {
        return ge::GRAPH_FAILED;
    }
    uint32_t hw = totalLength / (n * c);
    uint32_t tileHw = hw >= 256 ? 256 : hw; // 将H*W切成不超过256的tile

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_hw(hw);
    tiling.set_tileHw(tileHw);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // 无额外workspace
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_B = float;
using DTYPE_Z = float;

constexpr int32_t BUFFER_NUM = 2; // pipeline中每个队列的tensor个数

class KernelAddBiasFourDimBroadcastCustom {
public:
    __aicore__ inline KernelAddBiasFourDimBroadcastCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z, uint32_t n, uint32_t c, uint32_t hw, uint32_t tileHw)
    {
        this->n = n;
        this->c = c;
        this->hw = hw;
        this->tileHw = tileHw;

        // 以batch维按block划分：blockIdx对应一个batch
        uint32_t nIdx = AscendC::GetBlockIdx();
        uint64_t batchOffset = static_cast<uint64_t>(nIdx) * static_cast<uint64_t>(c) * static_cast<uint64_t>(hw);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + batchOffset, c * hw);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z + batchOffset, c * hw);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_B*)bias, c);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileHw * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileHw * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueueBiasScalar, 1, sizeof(DTYPE_B));          // 单元素bias
        pipe.InitBuffer(tmpQueueBiasTile, 1, this->tileHw * sizeof(DTYPE_B)); // 展开后的bias tile
    }

    __aicore__ inline void Process()
    {
        // 遍历该core承担的单个batch内所有通道与空间tile
        for (uint32_t ci = 0; ci < c; ++ci) {
            // 准备当前通道的bias标量到本地
            AscendC::LocalTensor<DTYPE_B> biasScalarLocal = tmpQueueBiasScalar.AllocTensor<DTYPE_B>();
            AscendC::DataCopy(biasScalarLocal, biasGm[ci], 1);

            for (uint32_t s = 0; s < hw; s += tileHw) {
                uint32_t cur = (s + tileHw <= hw) ? tileHw : (hw - s);

                // CopyIn
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[ci * hw + s], cur);

                // Compute: 将bias标量广播到tile长度，再执行逐元素相加
                AscendC::LocalTensor<DTYPE_B> biasTileLocal = tmpQueueBiasTile.AllocTensor<DTYPE_B>();
                uint32_t dstShape[1] = {cur};
                uint32_t srcShape[1] = {1};
                AscendC::BroadCast<DTYPE_B, 1, 0>(biasTileLocal, biasScalarLocal, dstShape, srcShape);

                AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
                AscendC::Add(zLocal, xLocal, biasTileLocal, cur);

                // CopyOut
                outQueueZ.EnQue<DTYPE_Z>(zLocal);
                AscendC::LocalTensor<DTYPE_Z> zOutLocal = outQueueZ.DeQue<DTYPE_Z>();
                AscendC::DataCopy(zGm[ci * hw + s], zOutLocal, cur);

                // 释放本次tile的本地tensor
                outQueueZ.FreeTensor(zOutLocal);
                inQueueX.FreeTensor(xLocal);
                tmpQueueBiasTile.FreeTensor(biasTileLocal);
            }
            tmpQueueBiasScalar.FreeTensor(biasScalarLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpQueueBiasScalar;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpQueueBiasTile;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_B> biasGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t n;
    uint32_t c;
    uint32_t hw;
    uint32_t tileHw;
};

extern "C" __global__ __aicore__ void add_bias_four_dim_broadcast_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAddBiasFourDimBroadcastCustom op;
    op.Init(x, bias, z, tiling_data.n, tiling_data.c, tiling_data.hw, tiling_data.tileHw);
    op.Process();
}
