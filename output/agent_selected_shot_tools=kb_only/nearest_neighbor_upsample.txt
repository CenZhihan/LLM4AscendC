kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1;

class KernelNearestNeighborUpsample {
public:
    __aicore__ inline KernelNearestNeighborUpsample() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t inH, uint32_t inW,
                                uint32_t scale, uint32_t outH, uint32_t outW)
    {
        this->n = n;
        this->c = c;
        this->inH = inH;
        this->inW = inW;
        this->scale = scale;
        this->outH = outH;
        this->outW = outW;

        // Partition by rows across blocks
        uint32_t totalRows = n * c * inH;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t baseRows = totalRows / blockNum;
        uint32_t extra = totalRows % blockNum;
        this->myRows = baseRows + ((blockIdx < extra) ? 1U : 0U);
        this->startRow = blockIdx * baseRows + ((blockIdx < extra) ? blockIdx : extra);

        // Set GM buffers to the block range
        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + this->startRow * inW, this->myRows * inW);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + this->startRow * outW * scale, this->myRows * outW * scale);

        // Queues: one input row, one expanded output row, one tmp repeat buffer of size 'scale'
        pipe.InitBuffer(inQueueRow, BUFFER_NUM, inW * sizeof(DTYPE));
        pipe.InitBuffer(outQueueRow, BUFFER_NUM, outW * sizeof(DTYPE));
        pipe.InitBuffer(tmpQueue, 1, scale * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < myRows; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE> srcLocal = inQueueRow.AllocTensor<DTYPE>();
        AscendC::DataCopy(srcLocal, xGm[r * inW], inW);
        inQueueRow.EnQue(srcLocal);
    }

    __aicore__ inline void Compute(uint32_t /*r*/)
    {
        AscendC::LocalTensor<DTYPE> srcLocal = inQueueRow.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> dstRowLocal = outQueueRow.AllocTensor<DTYPE>();

        // Horizontal expansion: each element repeats 'scale' times
        for (uint32_t j = 0; j < inW; ++j) {
            DTYPE val = srcLocal.GetValue(j);
            AscendC::LocalTensor<DTYPE> repLocal = tmpQueue.AllocTensor<DTYPE>();
            AscendC::Duplicate(repLocal, val, scale);
            AscendC::DataCopy(dstRowLocal[j * scale], repLocal, scale);
            tmpQueue.FreeTensor(repLocal);
        }

        outQueueRow.EnQue<DTYPE>(dstRowLocal);
        inQueueRow.FreeTensor(srcLocal);
    }

    __aicore__ inline void CopyOut(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE> dstRowLocal = outQueueRow.DeQue<DTYPE>();
        // Vertical replication: copy the horizontally expanded row 'scale' times
        for (uint32_t v = 0; v < scale; ++v) {
            AscendC::DataCopy(yGm[(r * scale + v) * outW], dstRowLocal, outW);
        }
        outQueueRow.FreeTensor(dstRowLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueRow;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueRow;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue;
    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t n, c, inH, inW, scale, outH, outW;
    uint32_t startRow;
    uint32_t myRows;
};

extern "C" __global__ __aicore__ void nearest_neighbor_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelNearestNeighborUpsample op;
    op.Init(x, y,
            tilingData.n(),
            tilingData.c(),
            tilingData.inH(),
            tilingData.inW(),
            tilingData.scale(),
            tilingData.outH(),
            tilingData.outW());
    op.Process();
}
