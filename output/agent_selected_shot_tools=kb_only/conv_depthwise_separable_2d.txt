kernel_src="""
#include "kernel_operator.h"

using dtype = float;

class KernelDepthwiseSeparable2D {
public:
    __aicore__ inline KernelDepthwiseSeparable2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR dw_w, GM_ADDR pw_w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(tilingData, tiling);
        this->n = tilingData.n();
        this->c = tilingData.c();
        this->h = tilingData.h();
        this->w = tilingData.w();
        this->outc = tilingData.outc();
        this->kh = tilingData.kh();
        this->kw = tilingData.kw();
        this->stride = tilingData.stride();
        this->padding = tilingData.padding();
        this->dilation = tilingData.dilation();
        this->hout = tilingData.hout();
        this->wout = tilingData.wout();

        xGm.SetGlobalBuffer((__gm__ dtype*)x, (uint64_t)n * c * h * w);
        dwWGm.SetGlobalBuffer((__gm__ dtype*)dw_w, (uint64_t)c * kh * kw);                       // [C,1,Kh,Kw] -> flatten per C:(Kh*Kw)
        pwWGm.SetGlobalBuffer((__gm__ dtype*)pw_w, (uint64_t)outc * c);                          // [OutC, C, 1, 1] -> flatten (OutC*C)
        yGm.SetGlobalBuffer((__gm__ dtype*)y, (uint64_t)n * outc * hout * wout);
        dwOutGm.SetGlobalBuffer((__gm__ dtype*)workspace, (uint64_t)n * c * hout * wout);        // intermediate

        // 为了简化，按plane(H*W)或(hout*wout)大小申请UB队列
        pipe.InitBuffer(inQueuePlane, 1, (uint32_t)(h * w * sizeof(dtype)));
        pipe.InitBuffer(inQueueDWK, 1, (uint32_t)(kh * kw * sizeof(dtype)));
        pipe.InitBuffer(outQueueDWPlane, 1, (uint32_t)(hout * wout * sizeof(dtype)));
        pipe.InitBuffer(tmpQueuePlane, 1, (uint32_t)(hout * wout * sizeof(dtype)));
        pipe.InitBuffer(outQueueYPlane, 1, (uint32_t)(hout * wout * sizeof(dtype)));
    }

    __aicore__ inline void Process()
    {
        // 1) Depthwise：对每个batch、每个通道做独立卷积，写入workspace
        for (uint32_t bn = 0; bn < n; ++bn) {
            for (uint32_t cc = 0; cc < c; ++cc) {
                // Load input plane [H*W]
                AscendC::LocalTensor<dtype> xLocal = inQueuePlane.AllocTensor<dtype>();
                AscendC::DataCopy(xLocal, xGm[(((uint64_t)bn * c + cc) * h) * w], h * w);

                // Load depthwise kernel [Kh*Kw] for this channel
                AscendC::LocalTensor<dtype> kLocal = inQueueDWK.AllocTensor<dtype>();
                AscendC::DataCopy(kLocal, dwWGm[((uint64_t)cc) * kh * kw], kh * kw);

                // Compute depthwise conv -> outLocal [hout*wout]
                AscendC::LocalTensor<dtype> outLocal = outQueueDWPlane.AllocTensor<dtype>();
                AscendC::Duplicate(outLocal, (dtype)0, hout * wout);

                for (uint32_t oh = 0; oh < hout; ++oh) {
                    for (uint32_t ow = 0; ow < wout; ++ow) {
                        float acc = 0.f;
                        for (uint32_t r = 0; r < kh; ++r) {
                            int32_t ih = (int32_t)oh * (int32_t)stride + (int32_t)r * (int32_t)dilation - (int32_t)padding;
                            if (ih < 0 || ih >= (int32_t)h) continue;
                            for (uint32_t s = 0; s < kw; ++s) {
                                int32_t iw = (int32_t)ow * (int32_t)stride + (int32_t)s * (int32_t)dilation - (int32_t)padding;
                                if (iw < 0 || iw >= (int32_t)w) continue;
                                uint32_t in_idx = (uint32_t)ih * w + (uint32_t)iw;
                                uint32_t k_idx = r * kw + s;
                                acc += xLocal.GetValue(in_idx) * kLocal.GetValue(k_idx);
                            }
                        }
                        outLocal.SetValue(oh * wout + ow, acc);
                    }
                }

                // write depthwise plane to workspace
                AscendC::DataCopy(dwOutGm[(((uint64_t)bn * c + cc) * hout) * wout], outLocal, hout * wout);

                inQueuePlane.FreeTensor(xLocal);
                inQueueDWK.FreeTensor(kLocal);
                outQueueDWPlane.EnQue(outLocal);
                AscendC::LocalTensor<dtype> outLocalDeque = outQueueDWPlane.DeQue<dtype>();
                outQueueDWPlane.FreeTensor(outLocalDeque);
            }
        }

        // 2) Pointwise：y[bn, oc, :, :] = sum_c ( dwOut[bn, c, :, :] * pwW[oc, c] )
        for (uint32_t bn = 0; bn < n; ++bn) {
            for (uint32_t oc = 0; oc < outc; ++oc) {
                // yLocal accumulator
                AscendC::LocalTensor<dtype> yLocal = outQueueYPlane.AllocTensor<dtype>();
                AscendC::Duplicate(yLocal, (dtype)0, hout * wout);

                for (uint32_t cc = 0; cc < c; ++cc) {
                    // load dwOut plane (hout*wout)
                    AscendC::LocalTensor<dtype> dwPlane = tmpQueuePlane.AllocTensor<dtype>();
                    AscendC::DataCopy(dwPlane, dwOutGm[(((uint64_t)bn * c + cc) * hout) * wout], hout * wout);

                    // load scalar pw weight[oc, cc]
                    // 将一行pw权重按C复制一次以便读取(简单做法：整行复制再读)
                    // 这里直接从GM按标量读取，通过临时LocalTensor承载长度1
                    AscendC::LocalTensor<dtype> pwScalarLocal = inQueueDWK.AllocTensor<dtype>(); // 重用队列容量，长度1
                    AscendC::DataCopy(pwScalarLocal, pwWGm[(uint64_t)oc * c + cc], 1);
                    float w_scalar = pwScalarLocal.GetValue(0);

                    // tmp = dwPlane * w_scalar
                    AscendC::LocalTensor<dtype> tmpMul = outQueueDWPlane.AllocTensor<dtype>();
                    AscendC::Muls(tmpMul, dwPlane, (dtype)w_scalar, hout * wout);

                    // yLocal += tmp
                    AscendC::Add(yLocal, yLocal, tmpMul, hout * wout);

                    tmpQueuePlane.FreeTensor(dwPlane);
                    inQueueDWK.FreeTensor(pwScalarLocal);
                    outQueueDWPlane.EnQue(tmpMul);
                    AscendC::LocalTensor<dtype> tmpDeque = outQueueDWPlane.DeQue<dtype>();
                    outQueueDWPlane.FreeTensor(tmpDeque);
                }

                // store yLocal -> yGm
                AscendC::DataCopy(yGm[(((uint64_t)bn * outc + oc) * hout) * wout], yLocal, hout * wout);

                outQueueYPlane.EnQue(yLocal);
                AscendC::LocalTensor<dtype> yDeque = outQueueYPlane.DeQue<dtype>();
                outQueueYPlane.FreeTensor(yDeque);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueuePlane;      // input plane
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueDWK;         // depthwise kernel / reuse for scalar
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueDWPlane;   // tmp planes
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueuePlane;      // reuse for tmp load
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueYPlane;    // y plane

    AscendC::GlobalTensor<dtype> xGm;
    AscendC::GlobalTensor<dtype> dwWGm;
    AscendC::GlobalTensor<dtype> pwWGm;
    AscendC::GlobalTensor<dtype> yGm;
    AscendC::GlobalTensor<dtype> dwOutGm;

    uint32_t n, c, h, w, outc, kh, kw, stride, padding, dilation, hout, wout;
};

extern "C" __global__ __aicore__ void conv_depthwise_separable_2d_custom(GM_ADDR x, GM_ADDR dw_weight, GM_ADDR pw_weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelDepthwiseSeparable2D op;
    op.Init(x, dw_weight, pw_weight, y, workspace, tiling);
    op.Process();
}
