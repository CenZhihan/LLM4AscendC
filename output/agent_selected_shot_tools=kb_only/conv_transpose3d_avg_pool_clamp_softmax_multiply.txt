kernel_src="""
#include "kernel_operator.h"

// 说明：此自定义核函数实现逐元素乘法，用于在PyTorch中完成avg_pool、conv_transpose3d、clamp与softmax后，
// 将结果与通道尺度scale进行逐元素相乘，从而替代模型中的最后一步乘法。

constexpr int32_t BUFFER_NUM = 2; // 每个队列的本地tensor数量

class KernelMul {
public:
    __aicore__ inline KernelMul() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        sGm.SetGlobalBuffer((__gm__ DTYPE_Y *)scale + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueS, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> sLocal = inQueueS.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(sLocal, sGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueS.EnQue(sLocal);
    }

    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> sLocal = inQueueS.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        AscendC::Mul(zLocal, xLocal, sLocal, this->tileLength);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueS.FreeTensor(sLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueS;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> sGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_avg_pool_clamp_softmax_multiply_custom(GM_ADDR x, GM_ADDR scale_full, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMul op;
    op.Init(x, scale_full, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版：保持AvgPool3d与ConvTranspose3d在PyTorch侧执行，
    将clamp与softmax后得到的张量与scale的逐元素乘法替换为AscendC自定义算子。
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, pool_kernel_size, clamp_min, clamp_max):
        super(ModelNew, self).__init__()
        self.avg_pool = torch.nn.AvgPool3d(pool_kernel_size)
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max
        self.scale = torch.nn.Parameter(torch.ones(1, out_channels, 1, 1, 1))

    def forward(self, x):
        # 1) 平均池化
        x = self.avg_pool(x)
        # 2) 反卷积
        x = self.conv_transpose(x)
        # 3) 截断
        x = torch.clamp(x, self.clamp_min, self.clamp_max)
        # 4) 空间维度softmax
        b, c, d, h, w = x.shape
        x = x.view(b, c, -1)
        x = torch.softmax(x, dim=2)
        x = x.view(b, c, d, h, w)
        # 5) 通道缩放，使用自定义AscendC算子（要求scale已按x形状展开）
        scale_full = self.scale.expand_as(x).contiguous()
        x = custom_ops_lib.conv_transpose3d_avg_pool_clamp_softmax_multiply_custom(x, scale_full)
        return x
'''
