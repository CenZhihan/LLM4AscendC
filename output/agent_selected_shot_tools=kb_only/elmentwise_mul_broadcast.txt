host_operator_src="""
#include "elmentwise_mul_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t MaxU32(uint32_t a, uint32_t b) { return a > b ? a : b; }
static inline bool IsBroadcastable(uint32_t a, uint32_t b) {
    return (a == b) || (a == 1) || (b == 1);
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;   // 固定使用8个AI Core
const uint32_t TILE_NUM = 8;    // 将最后一维分成8个tile

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ElmentwiseMulBroadcastCustomTilingData tiling;

    const gert::StorageShape *a_shape = context->GetInputShape(0);
    const gert::StorageShape *b_shape = context->GetInputShape(1);

    // 假设为3D ND张量 [D0, D1, D2]
    uint32_t a0 = a_shape->GetOriginShape().GetDim(0);
    uint32_t a1 = a_shape->GetOriginShape().GetDim(1);
    uint32_t a2 = a_shape->GetOriginShape().GetDim(2);

    uint32_t b0 = b_shape->GetOriginShape().GetDim(0);
    uint32_t b1 = b_shape->GetOriginShape().GetDim(1);
    uint32_t b2 = b_shape->GetOriginShape().GetDim(2);

    // 校验可广播性
    if (!IsBroadcastable(a0, b0) || !IsBroadcastable(a1, b1) || a2 != b2) {
        return ge::GRAPH_FAILED;
    }

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_a0(a0);
    tiling.set_a1(a1);
    tiling.set_a2(a2);
    tiling.set_b0(b0);
    tiling.set_b1(b1);
    tiling.set_b2(b2);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    // 仅处理3维情况
    if (a_shape->GetDimNum() != 3 || b_shape->GetDimNum() != 3) {
        return GRAPH_FAILED;
    }

    int64_t a0 = a_shape->GetDim(0);
    int64_t a1 = a_shape->GetDim(1);
    int64_t a2 = a_shape->GetDim(2);

    int64_t b0 = b_shape->GetDim(0);
    int64_t b1 = b_shape->GetDim(1);
    int64_t b2 = b_shape->GetDim(2);

    // 广播规则
    auto out0 = (a0 == b0) ? a0 : (a0 == 1 ? b0 : (b0 == 1 ? a0 : -1));
    auto out1 = (a1 == b1) ? a1 : (a1 == 1 ? b1 : (b1 == 1 ? a1 : -1));
    auto out2 = (a2 == b2) ? a2 : -1;

    if (out0 < 0 || out1 < 0 || out2 < 0) {
        return GRAPH_FAILED;
    }

    gert::Shape* z_shape = context->GetOutputShape(0);
    z_shape->SetDimNum(3);
    z_shape->SetDim(0, out0);
    z_shape->SetDim(1, out1);
    z_shape->SetDim(2, out2);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {

public:
    explicit ElmentwiseMulBroadcastCustom(const char* name) : OpDef(name)

        this->Input("a")

            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("z")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }

OP_ADD(ElmentwiseMulBroadcastCustom);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数量

class KernelMulBroadcast {
public:
    __aicore__ inline KernelMulBroadcast() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR z,
                                uint32_t a0In, uint32_t a1In, uint32_t a2In,
                                uint32_t b0In, uint32_t b1In, uint32_t b2In,
                                uint32_t tileNumIn)
    {
        a0 = a0In; a1 = a1In; a2 = a2In;
        b0 = b0In; b1 = b1In; b2 = b2In;
        // 输出维度
        o0 = a0 > b0 ? a0 : b0;
        o1 = a1 > b1 ? a1 : b1;
        inner = a2; // 已在tiling中校验 a2 == b2
        tileNum = tileNumIn;

        // 将最后一维分块，向上取整保证覆盖
        uint32_t totalTiles = tileNum * BUFFER_NUM;
        tileLength = (inner + totalTiles - 1) / totalTiles;

        aGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(a));
        bGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(b));
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(z));

        pipe.InitBuffer(inQueueA, BUFFER_NUM, tileLength * sizeof(float));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, tileLength * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, tileLength * sizeof(float));

        uint32_t pairCnt = o0 * o1;
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t pairsPerBlock = (pairCnt + blocks - 1) / blocks;
        startPair = pairsPerBlock * AscendC::GetBlockIdx();
        endPair = startPair + pairsPerBlock;
        if (endPair > pairCnt) endPair = pairCnt;
    }

    __aicore__ inline void Process()
    {
        for (uint32_t pair = startPair; pair < endPair; ++pair) {
            uint32_t i = pair / o1;
            uint32_t j = pair % o1;

            uint32_t ai = (a0 == 1) ? 0 : i;
            uint32_t aj = (a1 == 1) ? 0 : j;

            uint32_t bi = (b0 == 1) ? 0 : i;
            uint32_t bj = (b1 == 1) ? 0 : j;

            uint64_t aBase = (static_cast<uint64_t>(ai) * a1 + aj) * inner;
            uint64_t bBase = (static_cast<uint64_t>(bi) * b1 + bj) * inner;
            uint64_t zBase = (static_cast<uint64_t>(i) * o1 + j) * inner;

            int32_t loopCount = tileNum * BUFFER_NUM;
            for (int32_t t = 0; t < loopCount; ++t) {
                uint64_t offset = static_cast<uint64_t>(t) * tileLength;
                if (offset >= inner) break;
                uint32_t curLen = static_cast<uint32_t>(inner - offset > tileLength ? tileLength : inner - offset);

                // CopyIn
                AscendC::LocalTensor<float> aLocal = inQueueA.AllocTensor<float>();
                AscendC::LocalTensor<float> bLocal = inQueueB.AllocTensor<float>();
                AscendC::DataCopy(aLocal, aGm[aBase + offset], curLen);
                AscendC::DataCopy(bLocal, bGm[bBase + offset], curLen);
                inQueueA.EnQue(aLocal);
                inQueueB.EnQue(bLocal);

                // Compute
                AscendC::LocalTensor<float> aL = inQueueA.DeQue<float>();
                AscendC::LocalTensor<float> bL = inQueueB.DeQue<float>();
                AscendC::LocalTensor<float> zL = outQueueZ.AllocTensor<float>();
                AscendC::Mul(zL, aL, bL, curLen);
                outQueueZ.EnQue<float>(zL);
                inQueueA.FreeTensor(aL);
                inQueueB.FreeTensor(bL);

                // CopyOut
                AscendC::LocalTensor<float> zLDe = outQueueZ.DeQue<float>();
                AscendC::DataCopy(zGm[zBase + offset], zLDe, curLen);
                outQueueZ.FreeTensor(zLDe);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t a0, a1, a2;
    uint32_t b0, b1, b2;
    uint32_t o0, o1, inner;

    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t startPair, endPair;
};

extern "C" __global__ __aicore__ void elmentwise_mul_broadcast_custom(GM_ADDR a, GM_ADDR b, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMulBroadcast op;
    op.Init(a, b, z,
            tiling_data.a0, tiling_data.a1, tiling_data.a2,
            tiling_data.b0, tiling_data.b1, tiling_data.b2,
            tiling_data.tileNum);
    op.Process();
}
