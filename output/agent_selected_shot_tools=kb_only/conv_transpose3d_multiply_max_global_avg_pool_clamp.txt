kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数量

class KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClamp {
public:
    __aicore__ inline KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClamp() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, float scale, float clampMin, float clampMax)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        this->scale = scale;
        this->clampMin = clampMin;
        this->clampMax = clampMax;

        xGm.SetGlobalBuffer((__gm__ float *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ float *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        // y = clamp(x * scale, clampMin, clampMax)
        AscendC::Muls(yLocal, xLocal, (float)this->scale, this->tileLength);
        AscendC::Maxs(yLocal, yLocal, (float)this->clampMin, this->tileLength);
        AscendC::Mins(yLocal, yLocal, (float)this->clampMax, this->tileLength);

        outQueueY.EnQue<float>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    float scale;
    float clampMin;
    float clampMax;
};

extern "C" __global__ __aicore__ void conv_transpose3d_multiply_max_global_avg_pool_clamp_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.scale, tiling_data.clampMin, tiling_data.clampMax);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版：将全局平均池化后的缩放与夹紧融合为自定义AscendC算子
    保持计算等价：当scale>=0时，mean(s * z) = s * mean(z)，且clamp在缩放之后
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, scale, maxpool_kernel_size):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.scale = float(scale)
        self.maxpool = torch.nn.MaxPool3d(kernel_size=maxpool_kernel_size)
        self.global_avg_pool = torch.nn.AdaptiveAvgPool3d((1, 1, 1))
        self.clamp_min = 0.0
        self.clamp_max = 1.0

    def forward(self, x):
        x = self.conv_transpose(x)
        # 与原逻辑等价的重排：将缩放移至全局平均池化之后（scale>=0情形）
        x = self.maxpool(x)
        x = self.global_avg_pool(x)
        x = custom_ops_lib.conv_transpose3d_multiply_max_global_avg_pool_clamp_custom(x, self.scale, self.clamp_min, self.clamp_max)
        return x
