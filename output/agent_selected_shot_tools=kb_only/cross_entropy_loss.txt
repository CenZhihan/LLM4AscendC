kernel_src="""
#include "kernel_operator.h"
#include <cstdint>
#include <cfloat>

constexpr int32_t BUFFER_NUM = 1;

class KernelCrossEntropyLoss {
public:
    __aicore__ inline KernelCrossEntropyLoss() {}

    __aicore__ inline void Init(GM_ADDR predict, GM_ADDR label, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t tmpSize) {
        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");

        this->rows = rows;
        this->cols = cols;
        this->tmpSize = tmpSize;

        logitsGm.SetGlobalBuffer((__gm__ float*)predict, rows * cols);
        // 标签按int32读取（若实际为int64，需在上层做dtype转换）
        labelsGm.SetGlobalBuffer((__gm__ int32_t*)label, rows);
        outGm.SetGlobalBuffer((__gm__ float*)y, 32); // 输出标量

        // 行缓冲：一行logits
        pipe.InitBuffer(inQueueRow, BUFFER_NUM, this->cols * sizeof(float));
        // work buffer用于ReduceSum的临时空间（按cols大小分配，满足对齐要求）
        pipe.InitBuffer(workQueue, BUFFER_NUM, this->cols * sizeof(float));
        // exp/log等中间变量临时空间
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tmpSize);
        // 累积loss缓冲
        pipe.InitBuffer(accQueue, 1, 32);
        // 单元素缓冲存标签和标量运算
        pipe.InitBuffer(scalarQueue, 1, 64);
    }

    __aicore__ inline void Process() {
        // 累加器初始化为0
        AscendC::LocalTensor<float> lossAccum = accQueue.AllocTensor<float>();
        AscendC::Duplicate(lossAccum, 0.0f, 1);

        for (uint32_t i = 0; i < rows; ++i) {
            ComputeOneRow(i, lossAccum);
        }

        // 求平均
        AscendC::LocalTensor<float> denom = scalarQueue.AllocTensor<float>();
        denom.SetValue(0, static_cast<float>(rows));
        AscendC::Div(lossAccum, lossAccum, denom, 1);

        // 写回输出
        outGm.SetValue(0, lossAccum.GetValue(0));

        scalarQueue.FreeTensor(denom);
        accQueue.FreeTensor(lossAccum);
    }

private:
    __aicore__ inline void ComputeOneRow(uint32_t rowIdx, AscendC::LocalTensor<float>& lossAccum) {
        // 1. 拷贝该行logits到UB
        AscendC::LocalTensor<float> rowLocal = inQueueRow.AllocTensor<float>();
        AscendC::DataCopy(rowLocal, logitsGm[rowIdx * cols], cols);
        inQueueRow.EnQue(rowLocal);

        // 2. 计算该行最大值（标量循环）
        float maxVal = -FLT_MAX;
        for (uint32_t j = 0; j < cols; ++j) {
            float v = rowLocal.GetValue(j);
            maxVal = (v > maxVal) ? v : maxVal;
        }

        // 3. 计算exp(logits - max)向量
        AscendC::LocalTensor<float> workLocal = workQueue.AllocTensor<float>();
        AscendC::Duplicate(workLocal, maxVal, cols);
        // rowLocal = rowLocal - max
        AscendC::Sub(rowLocal, rowLocal, workLocal, cols);

        AscendC::LocalTensor<uint8_t> sharedTmp = tmpQueue.AllocTensor<uint8_t>();
        // 在整行上做exp（采用泰勒展开阶数15示例）
        AscendC::Exp<float, 15, false>(rowLocal, rowLocal, sharedTmp, cols);

        // 4. reduce sum 得到sum_exp
        // 将exp结果reduce到rowLocal[0]
        AscendC::ReduceSum<float>(rowLocal, rowLocal, workLocal, cols);
        float sumExp = rowLocal.GetValue(0);

        // 5. 读取对应label的logit（x_j）
        AscendC::LocalTensor<int32_t> labelLocal = scalarQueue.AllocTensor<int32_t>();
        AscendC::DataCopy(labelLocal, labelsGm[rowIdx], 1);
        int32_t targetIdx = labelLocal.GetValue(0);
        if (targetIdx < 0) { targetIdx = 0; }
        if ((uint32_t)targetIdx >= cols) { targetIdx = (int32_t)(cols - 1); }
        // 由于rowLocal当前是exp(logits - max)，需要x_j原值，需从GM再次读取该单点
        float xj = logitsGm.GetValue(rowIdx * cols + (uint32_t)targetIdx);

        // 6. 计算logsumexp = max + log(sumExp)
        AscendC::LocalTensor<float> sumLocal = scalarQueue.AllocTensor<float>();
        sumLocal.SetValue(0, sumExp);
        // 对单元素做log（示例使用Log接口）
        AscendC::Log(sumLocal, sumLocal, 1);
        float logsumexp = maxVal + sumLocal.GetValue(0);

        // 7. 累加loss：loss_i = logsumexp - xj
        float loss_i = logsumexp - xj;
        float accum = lossAccum.GetValue(0);
        lossAccum.SetValue(0, accum + loss_i);

        // 释放本次申请的本地tensor
        scalarQueue.FreeTensor(sumLocal);
        scalarQueue.FreeTensor(labelLocal);
        tmpQueue.FreeTensor(sharedTmp);
        workQueue.FreeTensor(workLocal);
        inQueueRow.FreeTensor(rowLocal);
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueRow;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> workQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> accQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> scalarQueue;

    AscendC::GlobalTensor<float> logitsGm;
    AscendC::GlobalTensor<int32_t> labelsGm;
    AscendC::GlobalTensor<float> outGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tmpSize;
};

extern "C" __global__ __aicore__ void cross_entropy_loss_custom(GM_ADDR predict, GM_ADDR label, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCrossEntropyLoss op;
    op.Init(predict, label, y, tiling_data.rows, tiling_data.cols, tiling_data.tmpSize);
    op.Process();
}
