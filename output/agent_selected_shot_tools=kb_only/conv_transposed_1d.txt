kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelConvTransposed1D {
public:
    __aicore__ inline KernelConvTransposed1D() : pipe(nullptr) {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t kernel, uint32_t lout,
                                AscendC::TPipe* pPipe)
    {
        this->n = n;
        this->cin = cin;
        this->cout = cout;
        this->lin = lin;
        this->kernel = kernel;
        this->lout = lout;
        this->pipe = pPipe;

        // 全量一维扁平化访问
        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(x), static_cast<uint32_t>(n * cin * lin));
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(w), static_cast<uint32_t>(cin * cout * kernel));
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(y), static_cast<uint32_t>(n * cout * lout));

        // 为每个block处理一个 (n_idx, j_idx) 对
        // 为简化，分配整条序列的本地缓冲
        pipe->InitBuffer(inQueueX, 1, lin * sizeof(DTYPE));   // 输入段缓冲
        pipe->InitBuffer(outQueueY, 1, lout * sizeof(DTYPE)); // 输出段缓冲
        pipe->InitBuffer(tmpQueueW, 1, sizeof(DTYPE));        // 权重标量缓冲
    }

    __aicore__ inline void Process()
    {
        uint32_t bIdx = AscendC::GetBlockIdx();
        uint32_t blocks = AscendC::GetBlockNum();
        uint64_t totalSeq = static_cast<uint64_t>(n) * static_cast<uint64_t>(cout);
        if (bIdx >= totalSeq) {
            return;
        }
        // block绑定到(n_idx, j_idx)
        uint32_t n_idx = bIdx / cout;
        uint32_t j_idx = bIdx % cout;

        // 分配输出本地缓冲并清零
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
        AscendC::Duplicate(yLocal, static_cast<DTYPE>(0), static_cast<int32_t>(lout));

        // 遍历输入通道cin与kernel
        for (uint32_t i_idx = 0; i_idx < cin; ++i_idx) {
            for (uint32_t k = 0; k < kernel; ++k) {
                // 假设 stride=1, padding=0, dilation=1
                // 偏移: offset = (kernel-1 - k)
                int32_t offset = static_cast<int32_t>(kernel - 1 - k);

                // 有效输出区间 [y_start, y_end)
                int32_t y_start = offset > 0 ? offset : 0;
                int32_t y_end = (offset + static_cast<int32_t>(lin)) < static_cast<int32_t>(lout)
                                    ? (offset + static_cast<int32_t>(lin))
                                    : static_cast<int32_t>(lout);
                int32_t len_valid = y_end - y_start;
                if (len_valid <= 0) {
                    continue;
                }

                // 复制输入段到本地
                AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
                // x 段起始对应输入的起点
                uint64_t x_base = (static_cast<uint64_t>(n_idx) * cin + i_idx) * lin;
                AscendC::DataCopy(xLocal, xGm[x_base], static_cast<uint32_t>(len_valid));

                // 从权重取出标量 w[i, j, k]
                AscendC::LocalTensor<DTYPE> wLocal = tmpQueueW.AllocTensor<DTYPE>();
                uint64_t w_index = (static_cast<uint64_t>(i_idx) * cout + j_idx) * kernel + k;
                AscendC::DataCopy(wLocal, wGm[w_index], 1);

                // 将标量读取到寄存器变量（假设可通过GetValue读取，或编译器优化成标量）
                DTYPE w_scalar = wLocal.GetValue(0);

                // yLocal[y_start : y_start+len_valid] += w_scalar * xLocal[0:len_valid]
                // 这里依赖LocalTensor支持偏移索引
                AscendC::Axpy(yLocal[y_start], xLocal, w_scalar, static_cast<uint32_t>(len_valid));

                inQueueX.FreeTensor(xLocal);
                tmpQueueW.FreeTensor(wLocal);
            }
        }

        // 输出回GM: y[n_idx, j_idx, :]
        uint64_t y_base = (static_cast<uint64_t>(n_idx) * cout + j_idx) * lout;
        AscendC::DataCopy(yGm[y_base], yLocal, lout);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe* pipe;
    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueueW;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    uint32_t n;
    uint32_t cin;
    uint32_t cout;
    uint32_t lin;
    uint32_t kernel;
    uint32_t lout;
};

extern "C" __global__ __aicore__ void conv_transposed_1d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    AscendC::TPipe pipe;
    KernelConvTransposed1D op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.kernel, tiling_data.lout,
            &pipe);
    op.Process();
}

#include "pytorch_npu_helper.hpp"

at::Tensor conv_transposed_1d_custom_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    int64_t n = x.size(0);
    int64_t cin = x.size(1);
    int64_t lin = x.size(2);
    int64_t w_cin = weight.size(0);
    int64_t cout = weight.size(1);

    int64_t lout = lin + kernel - 1;

    at::Tensor y = at::empty({n, cout, lout}, x.options());
    EXEC_NPU_CMD(aclnnConvTransposed1dCustom, x, weight, y);
    return y;

    m.impl("conv_transposed_1d_custom", &conv_transposed_1d_custom_impl_npu);

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    自定义AscendC实现的1D反卷积（转置卷积），假设 stride=1, padding=0, groups=1, 无bias。
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int):
        super().__init__()
        # PyTorch ConvTranspose1d的权重形状: (in_channels, out_channels, kernel_size)
        self.weight = torch.nn.Parameter(torch.empty(in_channels, out_channels, kernel_size))
        torch.nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_1d_custom(x, self.weight)
