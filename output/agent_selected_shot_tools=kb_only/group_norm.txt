host_operator_src="""
#include "group_norm_custom_tiling.h"
#include "register/op_def_registry.h"
#include "optiling/group_norm_tiling.h"

namespace optiling {
static const uint32_t BLOCK_DIM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context) {
  GroupNormCustomTilingData tiling;

  const gert::StorageShape* x_shape = context->GetInputShape(0);
  const gert::Shape shape = x_shape->GetStorageShape();
  if (shape.GetDimNum() != 4) {
    return ge::GRAPH_FAILED;
  }
  uint32_t n = static_cast<uint32_t>(shape.GetDim(0));
  uint32_t c = static_cast<uint32_t>(shape.GetDim(1));
  uint32_t h = static_cast<uint32_t>(shape.GetDim(2));
  uint32_t w = static_cast<uint32_t>(shape.GetDim(3));

  const int64_t* groupPtr = context->GetAttrs()->GetAttrPointer<int64_t>(0); // group_num
  const float* epsilonPtr = context->GetAttrs()->GetAttrPointer<float>(1);   // epsilon
  uint32_t groupNum = static_cast<uint32_t>(*groupPtr);
  (void)epsilonPtr; // epsilon used in kernel, no need for tiling here

  std::vector<int64_t> vec = { (int64_t)n, (int64_t)c, (int64_t)h, (int64_t)w };
  ge::Shape srcShape(vec);
  uint32_t maxSize = 0, minSize = 0;

  AscendC::GetGroupNormMaxMinTmpSize(srcShape, sizeof(float), false, groupNum, maxSize, minSize);
  // Choose maxSize within UB capability for better performance
  AscendC::GetGroupNormNDTilingInfo(srcShape, maxSize, sizeof(float), false, groupNum, tiling.groupNormTilingData);

  tiling.set_n(n);
  tiling.set_c(c);
  tiling.set_h(h);
  tiling.set_w(w);
  tiling.set_groupNum(groupNum);

  context->SetBlockDim(BLOCK_DIM);
  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
  context->SetTilingKey(1);

  return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context) {
  const gert::Shape* in_shape = context->GetInputShape(0);
  gert::Shape* out_shape = context->GetOutputShape(0);
  *out_shape = *in_shape;
  return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class GroupNormCustom : public OpDef {
 public:
  explicit GroupNormCustom(const char* name) : OpDef(name) {
    this->Input("x")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});
    this->Input("gamma")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});
    this->Input("beta")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});
    this->Output("res_out")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});

    this->Attr("group_num").AttrType(REQUIRED).Int(1);

    this->SetInferShape(ge::InferShape);
    this->AICore().SetTiling(optiling::TilingFunc);
    this->AICore().AddConfig("ascend910b");
  }

OP_ADD(GroupNormCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

class KernelGroupNorm {
 public:
  __aicore__ inline KernelGroupNorm() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR out, GM_ADDR tiling) {
    GET_TILING_DATA(td, tiling);
    n = td.n;
    c = td.c;
    h = td.h;
    w = td.w;
    g = td.groupNum;
    groupTiling = td.groupNormTilingData;

    // Set GM buffers
    uint32_t bshLength = n * c * h * w;
    xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), bshLength);
    outGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(out), bshLength);
    gammaGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(gamma), c);
    betaGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(beta), c);

    // UB buffer sizes
    uint32_t hwAlignSize = ((sizeof(float) * h * w + ONE_BLK_SIZE - 1) / ONE_BLK_SIZE) * ONE_BLK_SIZE / sizeof(float);

    pipe.InitBuffer(inQueueX, 1, sizeof(float) * n * c * hwAlignSize);
    pipe.InitBuffer(inQueueGamma, 1, ((sizeof(float) * c + 31) / 32) * 32);
    pipe.InitBuffer(inQueueBeta, 1, ((sizeof(float) * c + 31) / 32) * 32);
    pipe.InitBuffer(outQueue, 1, sizeof(float) * n * c * hwAlignSize);
    pipe.InitBuffer(meanBuffer, ((sizeof(float) * g * n + 31) / 32) * 32);
    pipe.InitBuffer(varBuffer, ((sizeof(float) * g * n + 31) / 32) * 32);
  }

  __aicore__ inline void Process(float epsilon) {
    CopyIn();
    Compute(epsilon);
    CopyOut();
  }

 private:
  __aicore__ inline void CopyIn() {
    LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
    LocalTensor<float> gammaLocal = inQueueGamma.AllocTensor<float>();
    LocalTensor<float> betaLocal = inQueueBeta.AllocTensor<float>();

    DataCopyParams copyParams{static_cast<uint16_t>(n * c),
                              static_cast<uint16_t>(h * w * sizeof(float)), 0, 0};
    DataCopyPadParams padParams{true, 0,
        static_cast<uint8_t>(((sizeof(float) * h * w + ONE_BLK_SIZE - 1) / ONE_BLK_SIZE * ONE_BLK_SIZE / sizeof(float)) - h * w), 0};
    DataCopyPad(xLocal, xGm, copyParams, padParams);

    DataCopyParams copyParamsGamma{1, static_cast<uint16_t>(c * sizeof(float)), 0, 0};
    DataCopyPadParams padParamsGamma{false, 0, 0, 0};
    DataCopyPad(gammaLocal, gammaGm, copyParamsGamma, padParamsGamma);
    DataCopyPad(betaLocal, betaGm, copyParamsGamma, padParamsGamma);

    inQueueX.EnQue(xLocal);
    inQueueGamma.EnQue(gammaLocal);
    inQueueBeta.EnQue(betaLocal);
  }

  __aicore__ inline void Compute(float epsilon) {
    LocalTensor<float> xLocal = inQueueX.DeQue<float>();
    LocalTensor<float> gammaLocal = inQueueGamma.DeQue<float>();
    LocalTensor<float> betaLocal = inQueueBeta.DeQue<float>();
    LocalTensor<float> outLocal = outQueue.AllocTensor<float>();
    LocalTensor<float> meanLocal = meanBuffer.Get<float>();
    LocalTensor<float> varianceLocal = varBuffer.Get<float>();

    GroupNorm<float, false>(outLocal, meanLocal, varianceLocal, xLocal,
                            gammaLocal, betaLocal, (float)epsilon, groupTiling);

    inQueueX.FreeTensor(xLocal);
    inQueueGamma.FreeTensor(gammaLocal);
    inQueueBeta.FreeTensor(betaLocal);
    outQueue.EnQue(outLocal);
  }

  __aicore__ inline void CopyOut() {
    LocalTensor<float> outLocal = outQueue.DeQue<float>();
    DataCopyParams copyParams{static_cast<uint16_t>(n * c),
                              static_cast<uint16_t>(h * w * sizeof(float)), 0, 0};
    DataCopyPadParams padParams{true, 0,
        static_cast<uint8_t>(((sizeof(float) * h * w + ONE_BLK_SIZE - 1) / ONE_BLK_SIZE * ONE_BLK_SIZE / sizeof(float)) - h * w), 0};
    DataCopyPad(outGm, outLocal, copyParams, padParams);
    outQueue.FreeTensor(outLocal);
  }

 private:
  TPipe pipe;
  TQue<QuePosition::VECIN, 1> inQueueX, inQueueGamma, inQueueBeta;
  TQue<QuePosition::VECOUT, 1> outQueue;
  TBuf<QuePosition::VECCALC> meanBuffer, varBuffer;

  GlobalTensor<float> xGm, gammaGm, betaGm, outGm;

  uint32_t n = 0, c = 0, h = 0, w = 0, g = 0;
  GroupNormTiling groupTiling;
};

extern "C" __global__ __aicore__ void group_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta,
                                                         GM_ADDR res_out, GM_ADDR workspace,
                                                         GM_ADDR tiling, float epsilon) {
  KernelGroupNorm op;
  op.Init(x, gamma, beta, res_out, tiling);
  if (TILING_KEY_IS(1)) {
    op.Process(epsilon);
  }
}

    m.def("group_norm_custom", &group_norm_impl_npu, "group norm");

model_src='''
import torch, torch_npu, custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_features: int, num_groups: int):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.group_num = num_groups
        self.epsilon = 1e-3
        self.gamma = torch.nn.Parameter(torch.ones(num_features))
        self.beta = torch.nn.Parameter(torch.zeros(num_features))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.group_norm_custom(x, self.gamma, self.beta, int(self.group_num), float(self.epsilon))
