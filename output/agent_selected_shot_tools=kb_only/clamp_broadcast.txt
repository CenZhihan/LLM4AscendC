#include "clamp_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>

namespace optiling {
const uint32_t MAX_BLOCK_DIM = 8;
const uint32_t TILE_ROWS = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ClampBroadcastCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    const auto& origin = in_shape->GetOriginShape();

    uint32_t totalLength = static_cast<uint32_t>(origin.GetShapeSize());
    uint32_t dimNum = static_cast<uint32_t>(origin.GetDimNum());
    uint32_t cols = 1;
    if (dimNum > 0) {
        cols = static_cast<uint32_t>(origin.GetDim(dimNum - 1));
    }
    if (cols == 0) {
        return ge::GRAPH_FAILED;
    }
    uint32_t rows = totalLength / cols;

    uint32_t blockDim = std::min(MAX_BLOCK_DIM, rows == 0 ? 1u : rows);
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileRows(TILE_ROWS);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* z_shape = context->GetOutputShape(0);
    *z_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

class KernelClampBroadcast {
public:
    __aicore__ inline KernelClampBroadcast() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR minv, GM_ADDR maxv, GM_ADDR z,
                                uint32_t rows, uint32_t cols, uint32_t tileRows)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileRows = tileRows > 0 ? tileRows : 1;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t baseRows = rows / blockNum;
        uint32_t tail = rows % blockNum;
        this->myRows = baseRows + (blockIdx < tail ? 1 : 0);
        this->startRow = baseRows * blockIdx + (blockIdx < tail ? blockIdx : tail);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)startRow * cols, (uint64_t)myRows * cols);
        minGm.SetGlobalBuffer((__gm__ DTYPE_X*)minv, cols);
        maxGm.SetGlobalBuffer((__gm__ DTYPE_X*)maxv, cols);
        zGm.SetGlobalBuffer((__gm__ DTYPE_X*)z + (uint64_t)startRow * cols, (uint64_t)myRows * cols);

        pipe.InitBuffer(inQueueX, 1, cols * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueMin, 1, cols * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueMax, 1, cols * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, 1, cols * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < myRows; ++r) {
            CopyInRow(r);
            ComputeRow();
            CopyOutRow(r);
        }
    }

private:
    __aicore__ inline void CopyInRow(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> minLocal = inQueueMin.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> maxLocal = inQueueMax.AllocTensor<DTYPE_X>();

        AscendC::DataCopy(xLocal, xGm[(uint64_t)r * cols], cols);
        AscendC::DataCopy(minLocal, minGm, cols);
        AscendC::DataCopy(maxLocal, maxGm, cols);

        inQueueX.EnQue(xLocal);
        inQueueMin.EnQue(minLocal);
        inQueueMax.EnQue(maxLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> minLocal = inQueueMin.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> maxLocal = inQueueMax.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> zLocal = outQueueZ.AllocTensor<DTYPE_X>();

        // z = max(x, min_val)
        AscendC::Max(zLocal, xLocal, minLocal, cols);
        // z = min(z, max_val)
        AscendC::Min(zLocal, zLocal, maxLocal, cols);

        outQueueZ.EnQue<DTYPE_X>(zLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueMin.FreeTensor(minLocal);
        inQueueMax.FreeTensor(maxLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_X> zLocal = outQueueZ.DeQue<DTYPE_X>();
        AscendC::DataCopy(zGm[(uint64_t)r * cols], zLocal, cols);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX, inQueueMin, inQueueMax;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> minGm;
    AscendC::GlobalTensor<DTYPE_X> maxGm;
    AscendC::GlobalTensor<DTYPE_X> zGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tileRows;
    uint32_t myRows;
    uint32_t startRow;
};

extern "C" __global__ __aicore__ void clamp_broadcast_custom(GM_ADDR x, GM_ADDR minv, GM_ADDR maxv, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelClampBroadcast op;
    op.Init(x, minv, maxv, z, tiling_data.rows, tiling_data.cols, tiling_data.tileRows);
    op.Process();
}
"""
