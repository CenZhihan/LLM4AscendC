kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数量

class KernelFusedTanh {
public:
    __aicore__ inline KernelFusedTanh() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        // 防止除0
        uint32_t base = this->tileNum * BUFFER_NUM;
        this->tileLength = (base > 0) ? (this->blockLength / base) : 0;

        inGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x) + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(y) + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (this->tileLength == 0) {
            return;
        }
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
        // 处理剩余尾块
        uint32_t done = loopCount * this->tileLength;
        uint32_t remain = this->blockLength > done ? (this->blockLength - done) : 0;
        if (remain > 0) {
            CopyInTail(done, remain);
            ComputeTail(remain);
            CopyOutTail(done, remain);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueue.AllocTensor<float>();
        AscendC::DataCopy(xLocal, inGm[progress * this->tileLength], this->tileLength);
        inQueue.EnQue(xLocal);
    }
    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<float> xLocal = inQueue.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueue.AllocTensor<float>();
        // 本示例实现Tanh激活，作为融合链路中的一环
        AscendC::Tanh(yLocal, xLocal, this->tileLength);
        outQueue.EnQue(yLocal);
        inQueue.FreeTensor(xLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueue.DeQue<float>();
        AscendC::DataCopy(outGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueue.FreeTensor(yLocal);
    }

    __aicore__ inline void CopyInTail(uint32_t offset, uint32_t remain)
    {
        AscendC::LocalTensor<float> xLocal = inQueue.AllocTensor<float>();
        AscendC::DataCopy(xLocal, inGm[offset], remain);
        inQueue.EnQue(xLocal);
    }
    __aicore__ inline void ComputeTail(uint32_t remain)
    {
        AscendC::LocalTensor<float> xLocal = inQueue.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueue.AllocTensor<float>();
        AscendC::Tanh(yLocal, xLocal, remain);
        outQueue.EnQue(yLocal);
        inQueue.FreeTensor(xLocal);
    }
    __aicore__ inline void CopyOutTail(uint32_t offset, uint32_t remain)
    {
        AscendC::LocalTensor<float> yLocal = outQueue.DeQue<float>();
        AscendC::DataCopy(outGm[offset], yLocal, remain);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<float> inGm;
    AscendC::GlobalTensor<float> outGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void convtranspose2d_batchnorm_tanh_maxpool_groupnorm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedTanh op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC算子替换部分链路：以Tanh为例（kernel: convtranspose2d_batchnorm_tanh_maxpool_groupnorm_custom）
    其余保持与原模型一致。
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, groups, num_groups):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.batch_norm = torch.nn.BatchNorm2d(out_channels)
        self.max_pool = torch.nn.MaxPool2d(kernel_size=2, stride=2)
        self.group_norm = torch.nn.GroupNorm(num_groups=num_groups, num_channels=out_channels)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.batch_norm(x)
        # 使用自定义AscendC算子执行激活（示例为Tanh）
        x = custom_ops_lib.convtranspose2d_batchnorm_tanh_maxpool_groupnorm_custom(x)
        # 其余算子仍走PyTorch实现
        x = self.max_pool(x)
        x = self.group_norm(x)
        return x
