kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

using DTYPE = float;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wavg, GM_ADDR cbias, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t tileCols)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileCols = tileCols;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = (rows + blockNum - 1) / blockNum;
        this->rowStart = blockIdx * this->rowsPerBlock;
        this->rowEnd = AscendC::Min(this->rowStart + this->rowsPerBlock, rows);

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x, (uint64_t)rows * cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y, (uint64_t)rows * cols);
        wavgGm.SetGlobalBuffer((__gm__ DTYPE *)wavg, cols);
        cbiasGm.SetGlobalBuffer((__gm__ DTYPE *)cbias, 1);

        pipe.InitBuffer(inQueueX, 1, this->tileCols * sizeof(DTYPE));
        pipe.InitBuffer(inQueueW, 1, this->tileCols * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, 1, this->tileCols * sizeof(DTYPE));
        pipe.InitBuffer(tmpVecBuf, 1, this->tileCols * sizeof(DTYPE));
        pipe.InitBuffer(scalarBuf, 1, sizeof(DTYPE)); // 存储单个标量
    }

    __aicore__ inline void Process()
    {
        // 读取 c_bias 标量
        DTYPE c_scalar = 0;
        {
            AscendC::LocalTensor<DTYPE> cLocal = scalarBuf.AllocTensor<DTYPE>();
            AscendC::DataCopy(cLocal, cbiasGm, 1);
            scalarBuf.EnQue(cLocal);
            AscendC::LocalTensor<DTYPE> cLocalDeq = scalarBuf.DeQue<DTYPE>();
            // 通过将标量拷回到寄存器变量
            c_scalar = *((DTYPE*)cLocalDeq.GetPhyAddr());
            scalarBuf.FreeTensor(cLocalDeq);
        }

        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            // 计算 s = x[r] dot w_avg + c_scalar
            DTYPE s = 0;
            uint64_t baseOffset = (uint64_t)r * cols;
            for (uint32_t c = 0; c < cols; c += tileCols) {
                uint32_t curTile = AscendC::Min(tileCols, cols - c);
                AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> wLocal = inQueueW.AllocTensor<DTYPE>();
                AscendC::DataCopy(xLocal, xGm[baseOffset + c], curTile);
                AscendC::DataCopy(wLocal, wavgGm[c], curTile);
                inQueueX.EnQue(xLocal);
                inQueueW.EnQue(wLocal);

                AscendC::LocalTensor<DTYPE> xDeq = inQueueX.DeQue<DTYPE>();
                AscendC::LocalTensor<DTYPE> wDeq = inQueueW.DeQue<DTYPE>();
                AscendC::LocalTensor<DTYPE> mulLocal = tmpVecBuf.AllocTensor<DTYPE>();
                AscendC::Mul(mulLocal, xDeq, wDeq, curTile);
                // 对 mulLocal 求和得到本 tile 的部分和
                DTYPE partial = 0;
                // 将局部向量累加为标量（逐元素求和）
                for (uint32_t i = 0; i < curTile; ++i) {
                    partial += *((DTYPE*)mulLocal.GetPhyAddr() + i);
                }
                s += partial;
                tmpVecBuf.FreeTensor(mulLocal);
                inQueueX.FreeTensor(xDeq);
                inQueueW.FreeTensor(wDeq);
            }
            s += c_scalar;

            // GELU(s)
            DTYPE s_gelu = 0;
            {
                AscendC::LocalTensor<DTYPE> inScalar = scalarBuf.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> outScalar = scalarBuf.AllocTensor<DTYPE>();
                // 将 s 写入本地缓冲
                *((DTYPE*)inScalar.GetPhyAddr()) = s;
                AscendC::Gelu(outScalar, inScalar, 1);
                s_gelu = *((DTYPE*)outScalar.GetPhyAddr());
                scalarBuf.FreeTensor(inScalar);
                scalarBuf.FreeTensor(outScalar);
            }

            // 残差加：y[r, :] = x[r, :] + s_gelu（广播到行）
            for (uint32_t c = 0; c < cols; c += tileCols) {
                uint32_t curTile = AscendC::Min(tileCols, cols - c);
                AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
                AscendC::DataCopy(xLocal, xGm[baseOffset + c], curTile);
                // 将 yLocal 初始化为标量 s_gelu
                for (uint32_t i = 0; i < curTile; ++i) {
                    *((DTYPE*)yLocal.GetPhyAddr() + i) = s_gelu;
                }
                // yLocal = xLocal + yLocal
                AscendC::Add(yLocal, xLocal, yLocal, curTile);
                outQueueY.EnQue<DTYPE>(yLocal);
                inQueueX.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE> yDeq = outQueueY.DeQue<DTYPE>();
                AscendC::LocalTensor<DTYPE> xDeq2 = inQueueX.DeQue<DTYPE>();
                AscendC::DataCopy(yGm[baseOffset + c], yDeq, curTile);
                outQueueY.FreeTensor(yDeq);
                inQueueX.FreeTensor(xDeq2);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX, inQueueW;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpVecBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> scalarBuf;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wavgGm;
    AscendC::GlobalTensor<DTYPE> cbiasGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t rows, cols, tileCols;
    uint32_t rowsPerBlock, rowStart, rowEnd;
};

extern "C" __global__ __aicore__ void gemm_subtract_global_avg_pool_log_sum_exp_gelu_residual_add_custom(
    GM_ADDR x, GM_ADDR wavg, GM_ADDR cbias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, wavg, cbias, y,
            tiling_data.rows, tiling_data.cols, tiling_data.tileCols);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused kernel: computes
      s = x @ mean(W, dim=0)^T + mean(b - subtract)
      s = GELU(s)
      y = x + s  (row-wise broadcast)
    which is algebraically equivalent to:
      y = x + GELU(LogSumExp(Mean((xW + b - subtract), dim=1, keepdim=True), dim=1, keepdim=True))
    because LogSumExp over a size-1 dim is identity.
    """
    def __init__(self, in_features, out_features, bias=True):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features, bias=bias)
        self.subtract = torch.nn.Parameter(torch.randn(out_features))
        self.bias_flag = bias

    def forward(self, x):
        # 预计算 w_avg 与 c_bias（可放到更上层缓存以减少重复计算）
        # w_avg: [in_features] = mean over out_features of W
        # PyTorch Linear 权重形状为 [out_features, in_features]
        w_avg = self.gemm.weight.mean(dim=0)
        if self.bias_flag and self.gemm.bias is not None:
            c_bias = (self.gemm.bias - self.subtract).mean()
        else:
            c_bias = (-self.subtract).mean()
        c_bias = c_bias.view(1)

        y = custom_ops_lib.gemm_subtract_global_avg_pool_log_sum_exp_gelu_residual_add_custom(x, w_avg, c_bias)
        return y

def get_inputs():
    batch_size = 2048
    in_features = 8192
    x = torch.rand(batch_size, in_features)
    return [x]

def get_init_inputs():
    batch_size = 2048
    in_features = 8192
    out_features = 8192
    return [in_features, out_features]
'''
