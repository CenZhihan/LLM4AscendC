kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// 简单的Mish实现：mish(x) = x * tanh(softplus(x)), softplus(x)=log(1+exp(x))
template<typename T>
__aicore__ inline void MishInplace(const AscendC::LocalTensor<T>& dst, const AscendC::LocalTensor<T>& src, int32_t len)
{
    // 分配中间局部tensor
    AscendC::LocalTensor<T> tmpExp = dst;        // 复用dst的空间
    AscendC::LocalTensor<T> tmpSp  = src;        // 复用src的空间
    AscendC::LocalTensor<T> tmpTanh = dst;       // 复用
    // e = exp(x)
    AscendC::Exp(tmpExp, src, len);
    // ones
    AscendC::LocalTensor<T> ones = dst;
    T oneV(1);
    AscendC::Duplicate<T>(ones, oneV, len);
    // sp = log(1 + e)
    AscendC::Add(tmpSp, tmpExp, ones, len);
    AscendC::Ln(tmpSp, tmpSp, len);
    // t = tanh(sp)
    AscendC::Tanh(tmpTanh, tmpSp, len);
    // y = x * t
    AscendC::Mul(dst, src, tmpTanh, len);
}

class KernelMatmulMishMish {
public:
    __aicore__ inline KernelMatmulMishMish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z, GM_ADDR tilingGM, GM_ADDR workspaceGM)
    {
        // 读取TCubeTiling
        auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
        auto tempTiling = (uint32_t*)&tiling_;
        for (int i = 0; i < (int)(sizeof(tiling_) / sizeof(int32_t)); ++i, ++tempTilingGM, ++tempTiling) {
            *tempTiling = *tempTilingGM;
        }

        // 初始化GM
        aGlobal.SetGlobalBuffer((__gm__ DTYPE_X*)x, tiling_.ALayoutInfoB * tiling_.singleCoreM * tiling_.singleCoreK);
        bGlobal.SetGlobalBuffer((__gm__ DTYPE_W*)w, tiling_.BLayoutInfoB * tiling_.singleCoreK * tiling_.singleCoreN);
        cGlobal.SetGlobalBuffer((__gm__ DTYPE_Z*)z, tiling_.CLayoutInfoB * tiling_.singleCoreM * tiling_.singleCoreN);
        biasGlobal.SetGlobalBuffer((__gm__ DTYPE_B*)b, tiling_.CLayoutInfoB * tiling_.singleCoreN);

        // Matmul实例
        constexpr auto MM_CFG = matmul::GetNormalConfig(false, false, false, matmul::BatchMode::BATCH_LESS_THAN_L1);
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm_);
        mm_.Init(&tiling_);
        mm_.SetWorkspace(workspaceGM, 0);

        // 设置Tensor A/B/偏置
        gm_a.SetGlobalBuffer(const_cast<__gm__ DTYPE_X*>(aGlobal[0].GetPhyAddr()), tiling_.singleCoreM * tiling_.singleCoreK);
        gm_b.SetGlobalBuffer(const_cast<__gm__ DTYPE_W*>(bGlobal[0].GetPhyAddr()), tiling_.singleCoreK * tiling_.singleCoreN);
        gm_c.SetGlobalBuffer(const_cast<__gm__ DTYPE_Z*>(cGlobal[0].GetPhyAddr()), tiling_.singleCoreM * tiling_.singleCoreN);
        gm_bias.SetGlobalBuffer(const_cast<__gm__ DTYPE_B*>(biasGlobal[0].GetPhyAddr()), tiling_.singleCoreN);

        mm_.SetTensorA(gm_a, false);
        mm_.SetTensorB(gm_b, false);
        mm_.SetBias(gm_bias);

        // UB缓冲为后续Mish双次激活准备
        totalMN_ = tiling_.singleCoreM * tiling_.singleCoreN;
        // 将整个C分块处理
        tileLen_ = totalMN_ / AscendC::GetBlockNum();
        tileLen_ = tileLen_ / BUFFER_NUM; // 每队列缓存大小
        if (tileLen_ == 0) {
            tileLen_ = totalMN_;
        }
        pipe.InitBuffer(inQueueC_, BUFFER_NUM, tileLen_ * sizeof(DTYPE_Z));
        pipe.InitBuffer(outQueueZ_, BUFFER_NUM, tileLen_ * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // 先进行Matmul：C = A * B + bias
        mm_.IterateBatch(gm_c, 1, 1, false);

        // 对C执行两次Mish并输出到Z（原地）
        int32_t loopCount = (totalMN_ + tileLen_ - 1) / tileLen_;
        for (int32_t i = 0; i < loopCount; i++) {
            int32_t cur = (i < loopCount - 1) ? tileLen_ : (totalMN_ - i * tileLen_);
            CopyIn(i, cur);
            Compute(i, cur);
            CopyOut(i, cur);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress, int32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_Z> cLocal = inQueueC_.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(cLocal, gm_c[progress * tileLen_], curLen);
        inQueueC_.EnQue(cLocal);
    }
    __aicore__ inline void Compute(int32_t progress, int32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_Z> cLocal = inQueueC_.DeQue<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ_.AllocTensor<DTYPE_Z>();

        // 第一次Mish
        MishInplace<DTYPE_Z>(zLocal, cLocal, curLen);
        // 第二次Mish：输入为上一步输出，输出仍写回zLocal
        MishInplace<DTYPE_Z>(zLocal, zLocal, curLen);

        outQueueZ_.EnQue<DTYPE_Z>(zLocal);
        inQueueC_.FreeTensor(cLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress, int32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ_.DeQue<DTYPE_Z>();
        AscendC::DataCopy(gm_c[progress * tileLen_], zLocal, curLen);
        outQueueZ_.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;

    // Matmul相关
    TCubeTiling tiling_;
    AscendC::GlobalTensor<DTYPE_X> aGlobal;
    AscendC::GlobalTensor<DTYPE_W> bGlobal;
    AscendC::GlobalTensor<DTYPE_Z> cGlobal;
    AscendC::GlobalTensor<DTYPE_B> biasGlobal;

    AscendC::GlobalTensor<DTYPE_X> gm_a;
    AscendC::GlobalTensor<DTYPE_W> gm_b;
    AscendC::GlobalTensor<DTYPE_Z> gm_c;
    AscendC::GlobalTensor<DTYPE_B> gm_bias;

    typedef matmul::Matmul<
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, DTYPE_X, false, LayoutMode::NORMAL>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, DTYPE_W, false, LayoutMode::NORMAL>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, DTYPE_Z, false, LayoutMode::NORMAL>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, DTYPE_B, false, LayoutMode::NORMAL>
    > MatmulT;
    MatmulT mm_;

    // Mish流水
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueC_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ_;
    int32_t totalMN_{0};
    int32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    KernelMatmulMishMish op;
    op.Init(x, w, b, z, tiling, workspace);
    op.Process();
}
"""
