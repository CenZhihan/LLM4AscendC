kernel_src="""
#include "kernel_operator.h"
#include "lib/softmax_intf.h"
#include "lib/dropout_intf.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

struct SoftmaxLocalBuf {
    // Buffers for SoftmaxFlashV2 and SimpleSoftMax
    // We'll allocate them from queues on demand.
};

class KernelDropoutSoftmax {
public:
    __aicore__ inline KernelDropoutSoftmax() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR keepProb, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t tileNum)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileNum = tileNum;

        // Per-block row partition
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = (rows + blockNum - 1) / blockNum;
        this->rowStart = blockIdx * rowsPerBlock;
        if (this->rowStart > rows) this->rowStart = rows;
        this->rowEnd = AscendC::Min(rows, this->rowStart + rowsPerBlock);

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, rows * cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, rows * cols);
        maskGm.SetGlobalBuffer((__gm__ uint8_t*)mask, rows * cols);
        keepProbGm.SetGlobalBuffer((__gm__ DTYPE*)keepProb, 1);

        // Tile config along columns
        uint32_t totalPerBlock = (this->rowEnd - this->rowStart) * this->cols;
        this->tileLength = AscendC::Max<uint32_t>(1, (this->cols / (this->tileNum * BUFFER_NUM)));
        this->tileLength = (this->tileLength + 15) / 16 * 16; // align

        // Queues for dropout streaming
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(inQueueMask, BUFFER_NUM, this->tileLength * sizeof(uint8_t));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
    }

    __aicore__ inline void Run()
    {
        // 1) Dropout (byte mask) and scale by 1/keepProb
        DTYPE kp = LoadKeepProb();
        DTYPE invKeep = (kp > 0) ? (DTYPE)(1.0f / kp) : (DTYPE)0.0f;
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            // process row by row in tiles
            uint32_t tiles = (cols + tileLength - 1) / tileLength;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = r * cols + t * tileLength;
                uint32_t len = AscendC::Min(tileLength, cols - t * tileLength);
                CopyInDropout(offset, len);
                ComputeDropout(len, invKeep);
                CopyOutDropout(offset, len);
            }
        }

        // 2) Softmax per row (use SoftmaxFlashV2 to compute max/sum, then SimpleSoftMax to normalize)
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            // allocate locals for one full row
            AscendC::LocalTensor<DTYPE> rowLocal = softInBuf.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> dstLocal  = softOutBuf.AllocTensor<DTYPE>();
            AscendC::DataCopy(rowLocal, yGm[r * cols], cols);

            // Prepare softmax tiling
            SoftMaxShapeInfo shapeInfo;
            shapeInfo.srcM = 1;
            shapeInfo.srcK = cols;
            shapeInfo.oriSrcM = 1;
            shapeInfo.oriSrcK = cols;

            // Temporary outputs for FlashV2
            AscendC::LocalTensor<DTYPE> expSumLocal = auxBuf1.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> maxLocal    = auxBuf2.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> expMaxLocal = auxBuf3.AllocTensor<DTYPE>(); // unused for update=false
            AscendC::LocalTensor<DTYPE> inExpSumLocal = auxBuf4.AllocTensor<DTYPE>(); // not used
            AscendC::LocalTensor<DTYPE> inMaxLocal    = auxBuf5.AllocTensor<DTYPE>(); // not used

            // Compute tiling inside kernel
            auto tiling = SoftMaxFlashV2TilingFunc(shapeInfo, sizeof(DTYPE), sizeof(DTYPE), 0, false, false, false, false);

            // First pass: get reducemax and reducesum
            SoftmaxFlashV2<DTYPE, false, false, false, false>(
                dstLocal, expSumLocal, maxLocal, rowLocal, expMaxLocal, inExpSumLocal, inMaxLocal, tiling, shapeInfo);

            // Second pass: normalized softmax using computed max/sum
            SimpleSoftMax<DTYPE, false, false, false>(
                dstLocal, expSumLocal, maxLocal, rowLocal, tiling, shapeInfo);

            AscendC::DataCopy(yGm[r * cols], dstLocal, cols);

            // free locals
            softInBuf.FreeTensor(rowLocal);
            softOutBuf.FreeTensor(dstLocal);
            auxBuf1.FreeTensor(expSumLocal);
            auxBuf2.FreeTensor(maxLocal);
            auxBuf3.FreeTensor(expMaxLocal);
            auxBuf4.FreeTensor(inExpSumLocal);
            auxBuf5.FreeTensor(inMaxLocal);
        }
    }

private:
    __aicore__ inline DTYPE LoadKeepProb()
    {
        AscendC::LocalTensor<DTYPE> tmp = scalarBuf.AllocTensor<DTYPE>();
        AscendC::DataCopy(tmp, keepProbGm[0], 1);
        DTYPE val = tmp.GetValue(0);
        scalarBuf.FreeTensor(tmp);
        return val;
    }

    __aicore__ inline void CopyInDropout(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::LocalTensor<uint8_t> mLocal = inQueueMask.AllocTensor<uint8_t>();
        AscendC::DataCopy(xLocal, xGm[offset], len);
        AscendC::DataCopy(mLocal, maskGm[offset], len);
        inQueueX.EnQue(xLocal);
        inQueueMask.EnQue(mLocal);
    }

    __aicore__ inline void ComputeDropout(uint32_t len, DTYPE invKeep)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<uint8_t> mLocal = inQueueMask.DeQue<uint8_t>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();

        // Byte-mode dropout: y = (mask != 0 ? x : 0) * (1/keep_prob)
        // Implement via element-wise multiply after casting mask to float
        // Allocate a temp for casted mask
        AscendC::LocalTensor<DTYPE> mCast = tempBuf.AllocTensor<DTYPE>();
        AscendC::Cast(mCast, mLocal, len);          // uint8 -> float
        AscendC::Mul(yLocal, xLocal, mCast, len);   // y = x * mask
        AscendC::Muls(yLocal, yLocal, invKeep, len);// scale

        outQueueY.EnQue(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueMask.FreeTensor(mLocal);
        tempBuf.FreeTensor(mCast);
    }

    __aicore__ inline void CopyOutDropout(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[offset], yLocal, len);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    // Queues for streaming
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueMask;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    // Extra buffers for softmax and casting/scalars
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  softInBuf;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> softOutBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  auxBuf1;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  auxBuf2;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  auxBuf3;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  auxBuf4;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  auxBuf5;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  tempBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  scalarBuf;

    AscendC::GlobalTensor<DTYPE>   xGm;
    AscendC::GlobalTensor<DTYPE>   yGm;
    AscendC::GlobalTensor<uint8_t> maskGm;
    AscendC::GlobalTensor<DTYPE>   keepProbGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t rowStart;
    uint32_t rowEnd;
    uint32_t rowsPerBlock;
};

extern "C" __global__ __aicore__ void matmul_dropout_mean_softmax_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR keep_prob, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelDropoutSoftmax op;
    op.Init(x, mask, keep_prob, y, tiling_data.rows, tiling_data.cols, tiling_data.tileNum);
    op.Run();
}

model_src='''
import torch
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused pipeline: Linear (PyTorch) + custom NPU kernel for Dropout + Softmax.
    """
    def __init__(self, in_features, out_features, dropout_p):
        super().__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.dropout_p = float(dropout_p)

    def forward(self, x):
        # Linear
        x = F.linear(x, self.matmul.weight, self.matmul.bias)
        # Prepare mask and keep_prob
        keep_prob = 1.0 - self.dropout_p
        if self.training:
            # byte-mode mask same shape as x
            mask = torch.bernoulli(torch.full_like(x, keep_prob)).to(torch.uint8)
        else:
            # eval: identity mask
            mask = torch.ones_like(x, dtype=torch.uint8)
        keep_prob_tensor = torch.tensor([keep_prob], device=x.device, dtype=x.dtype)
        # Fused Dropout + Softmax on NPU
        y = custom_ops_lib.matmul_dropout_mean_softmax_custom(x, mask, keep_prob_tensor)
        return y
