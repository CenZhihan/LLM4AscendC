kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelChannelSum {
public:
    __aicore__ inline KernelChannelSum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w)
    {
        this->n = n;
        this->c = c;
        this->d = d;
        this->h = h;
        this->w = w;
        this->plane = d * h * w;  // 每个通道的元素数
        uint64_t in_total = static_cast<uint64_t>(n) * c * plane;
        uint64_t out_total = static_cast<uint64_t>(n) * plane;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, in_total);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, out_total);

        // 以平面维进行分块，tileLength不超过plane
        // 选择适中的tile以便在UB中计算（可根据实际芯片UB大小调优）
        this->tileLength = plane;
        const uint32_t maxTile = 8192;  // 简单选择，避免占用过多UB
        if (this->tileLength > maxTile) {
            this->tileLength = maxTile;
        }

        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueDst, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // 将batch维分配到block上，尽量均匀划分
        uint32_t base = n / blockNum;
        uint32_t rem  = n % blockNum;
        uint32_t myN  = base + (blockIdx < rem ? 1 : 0);
        uint32_t startN = blockIdx * base + (blockIdx < rem ? blockIdx : rem);

        for (uint32_t ni = 0; ni < myN; ++ni) {
            uint32_t nIdx = startN + ni;
            // 每个n上按plane分块
            for (uint32_t off = 0; off < plane; off += tileLength) {
                uint32_t curLen = AscendC::min(tileLength, plane - off);

                // 累加buffer
                AscendC::LocalTensor<DTYPE_Y> dstLocal = outQueueDst.AllocTensor<DTYPE_Y>();
                // 先加载第0个通道数据到dstLocal
                uint64_t base0 = static_cast<uint64_t>(nIdx) * c * plane + off;
                AscendC::DataCopy(dstLocal, xGm[base0], curLen);

                // 其余通道累加
                for (uint32_t ci = 1; ci < c; ++ci) {
                    AscendC::LocalTensor<DTYPE_X> tmpLocal = tmpQueue.AllocTensor<DTYPE_X>();
                    uint64_t basec = static_cast<uint64_t>(nIdx) * c * plane + static_cast<uint64_t>(ci) * plane + off;
                    AscendC::DataCopy(tmpLocal, xGm[basec], curLen);
                    AscendC::Add(dstLocal, dstLocal, tmpLocal, curLen);
                    tmpQueue.FreeTensor(tmpLocal);
                }

                // 写回输出到对应n的平面位置
                uint64_t yBase = static_cast<uint64_t>(nIdx) * plane + off;
                AscendC::DataCopy(yGm[yBase], dstLocal, curLen);
                outQueueDst.FreeTensor(dstLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueDst;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, d, h, w;
    uint32_t plane;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_max_max_sum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelChannelSum op;
    op.Init(x, y, tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版模型：ConvTranspose3d与两次MaxPool3d仍使用PyTorch算子，
    最终沿通道维求和由自定义AscendC算子替换。
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.max_pool1 = torch.nn.MaxPool3d(kernel_size=2)
        self.max_pool2 = torch.nn.MaxPool3d(kernel_size=3)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.max_pool1(x)
        x = self.max_pool2(x)
        # 使用自定义AscendC算子：对通道维(1)求和, 保持维度
        x = custom_ops_lib.conv_transpose3d_max_max_sum_custom(x)
        return x
