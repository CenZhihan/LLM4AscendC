kernel_src="""
#include "kernel_operator.h"

using dtype = float;

class KernelGemmBnScaleSoftmax {
public:
    __aicore__ inline KernelGemmBnScaleSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight_t, GM_ADDR bias, GM_ADDR z,
                                uint32_t M, uint32_t K, uint32_t N, uint32_t axpyTmpSize) {
        this->M = M;
        this->K = K;
        this->N = N;
        xGm.SetGlobalBuffer((__gm__ dtype*)x, static_cast<uint64_t>(M) * K);
        wTGm.SetGlobalBuffer((__gm__ dtype*)weight_t, static_cast<uint64_t>(K) * N);
        bGm.SetGlobalBuffer((__gm__ dtype*)bias, N);
        zGm.SetGlobalBuffer((__gm__ dtype*)z, static_cast<uint64_t>(M) * N);

        // Queues for one row processing
        pipe.InitBuffer(inQueueXRow, 1, K * sizeof(dtype));
        pipe.InitBuffer(inQueueWRow, 1, N * sizeof(dtype));
        pipe.InitBuffer(outQueueZRow, 1, N * sizeof(dtype));
        if (axpyTmpSize > 0) {
            pipe.InitBuffer(tmpQue, 1, axpyTmpSize);
            hasTmpBuf = true;
        } else {
            hasTmpBuf = false;
        }
    }

    __aicore__ inline void Process() {
        for (uint32_t i = 0; i < M; ++i) {
            ComputeRow(i);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t rowIdx) {
        // Load x row [K]
        AscendC::LocalTensor<dtype> xRowLocal = inQueueXRow.AllocTensor<dtype>();
        AscendC::DataCopy(xRowLocal, xGm[rowIdx * K], K);
        inQueueXRow.EnQue(xRowLocal);

        // Allocate z row buffer and initialize with bias
        AscendC::LocalTensor<dtype> zRowLocal = outQueueZRow.AllocTensor<dtype>();
        AscendC::DataCopy(zRowLocal, bGm, N);

        // temp buffer for AXPY if available
        AscendC::LocalTensor<uint8_t> sharedTmp;
        if (hasTmpBuf) {
            sharedTmp = tmpQue.AllocTensor<uint8_t>();
        }

        // GEMM using AXPY: z_row += x[row,k] * weight_t[k,:]
        AscendC::LocalTensor<dtype> xRow = inQueueXRow.DeQue<dtype>();
        for (uint32_t k = 0; k < K; ++k) {
            dtype scalar = xRow.GetValue(k);
            AscendC::LocalTensor<dtype> wRowLocal = inQueueWRow.AllocTensor<dtype>();
            AscendC::DataCopy(wRowLocal, wTGm[k * N], N);
            if (hasTmpBuf) {
                AscendC::Axpy(zRowLocal, wRowLocal, scalar, sharedTmp, N);
            } else {
                // Fallback: zRowLocal += scalar * wRowLocal using two ops
                // zRowLocal = zRowLocal + scalar * wRowLocal
                AscendC::LocalTensor<dtype> tmpLocal = inQueueWRow.AllocTensor<dtype>();
                // Muls (vector * scalar) then Add. If Muls is unavailable, Axpy path should be used.
                AscendC::Muls(tmpLocal, wRowLocal, scalar, N);
                AscendC::Add(zRowLocal, zRowLocal, tmpLocal, N);
                inQueueWRow.FreeTensor(tmpLocal);
            }
            inQueueWRow.FreeTensor(wRowLocal);
        }
        inQueueXRow.FreeTensor(xRow);

        // Softmax along last axis for this row
        AscendC::SoftMax(zRowLocal, zRowLocal, softmaxTiling);

        // Write back
        outQueueZRow.EnQue<dtype>(zRowLocal);
        AscendC::LocalTensor<dtype> zLocal = outQueueZRow.DeQue<dtype>();
        AscendC::DataCopy(zGm[rowIdx * N], zLocal, N);
        outQueueZRow.FreeTensor(zLocal);

        if (hasTmpBuf) {
            tmpQue.FreeTensor(sharedTmp);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueXRow;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueWRow;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZRow;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQue;

    AscendC::GlobalTensor<dtype> xGm;
    AscendC::GlobalTensor<dtype> wTGm;
    AscendC::GlobalTensor<dtype> bGm;
    AscendC::GlobalTensor<dtype> zGm;

    AscendC::SoftMaxTiling softmaxTiling;
    uint32_t M = 0, K = 0, N = 0;
    bool hasTmpBuf = false;
};

extern "C" __global__ __aicore__ void gemm_batch_norm_scaling_softmax_custom(GM_ADDR x,
                                                                              GM_ADDR weight_t,
                                                                              GM_ADDR bias,
                                                                              GM_ADDR z,
                                                                              GM_ADDR workspace,
                                                                              GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBnScaleSoftmax op;
    op.Init(x, weight_t, bias, z,
            tiling_data.M(), tiling_data.K(), tiling_data.N(),
            tiling_data.axpyTmpSize());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused: GEMM (Linear) + BatchNorm (inference affine via running stats) + scale + Softmax.
    The BN and scaling are pre-fused into weight and bias on host; the custom kernel performs GEMM + Softmax.
    """
    def __init__(self, in_features, out_features, bn_eps=1e-5, bn_momentum=0.1, scale_shape=(1,)):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.bn = torch.nn.BatchNorm1d(out_features, eps=bn_eps, momentum=bn_momentum)
        self.scale = torch.nn.Parameter(torch.ones(scale_shape))
        self.softmax_dim = 1

    def _fuse_bn_and_scale(self):
        # Compute A and B for BN+scale: y = A * x + B, where x is pre-BN activation per feature.
        # A = scale * (gamma / sqrt(running_var + eps))
        # B = scale * (beta - running_mean * gamma / sqrt(running_var + eps))
        running_mean = self.bn.running_mean
        running_var = self.bn.running_var
        gamma = self.bn.weight
        beta = self.bn.bias
        eps = self.bn.eps

        inv_std = torch.rsqrt(running_var + eps)
        scale_val = self.scale
        if scale_val.numel() == 1:
            scale_val = scale_val.expand_as(gamma)
        scale_val = scale_val.reshape_as(gamma)

        A = scale_val * (gamma * inv_std)
        B = scale_val * (beta - running_mean * gamma * inv_std)

        # Fuse into Linear: y = A * (x W^T + b) + B = x (A * W)^T + (A * b + B)
        W = self.gemm.weight        # [out, in]
        b = self.gemm.bias          # [out]
        W_fused = W * A.unsqueeze(1)            # [out, in]
        b_fused = A * b + B                     # [out]
        # Kernel expects weight transposed to [K, N] with contiguous last dimension
        W_fused_t = W_fused.t().contiguous()    # [in, out]
        return W_fused_t, b_fused

    def forward(self, x):
        # Prepare fused parameters on host
        weight_t, bias = self._fuse_bn_and_scale()
        # Call fused custom op (GEMM + Softmax)
        z = custom_ops_lib.gemm_batch_norm_scaling_softmax_custom(x, weight_t, bias)
        return z

def get_inputs():
    batch_size = 1024
    in_features = 8192
    return [torch.rand(batch_size, in_features)]

def get_init_inputs():
    batch_size = 1024
    in_features = 8192
    out_features = 8192
    bn_eps = 1e-5
    bn_momentum = 0.1
    scale_shape = (1,)
    return [in_features, out_features, bn_eps, bn_momentum, scale_shape]
'''
