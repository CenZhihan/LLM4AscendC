kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE_X;
typedef float DTYPE_Y;

constexpr int32_t BUFFER_NUM = 2;

class KernelMishTanh {
public:
    __aicore__ inline KernelMishTanh() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        // Temporary buffers for intermediate tensors
        pipe.InitBuffer(tmpQueue1, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X)); // exp(x) or softplus work buffer
        pipe.InitBuffer(tmpQueue2, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X)); // softplus/x*tanh(softplus)
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        // DeQue input
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();

        // Allocate temporaries
        AscendC::LocalTensor<DTYPE_X> tmp1 = tmpQueue1.AllocTensor<DTYPE_X>(); // exp(x) -> log(1+exp(x))
        AscendC::LocalTensor<DTYPE_X> tmp2 = tmpQueue2.AllocTensor<DTYPE_X>(); // tanh(softplus) or final tanh input

        // 1) tmp1 = exp(x)
        AscendC::Exp(tmp1, xLocal, this->tileLength);

        // 2) tmp1 = 1 + exp(x)  (reuse tmp1)
        AscendC::Adds(tmp1, tmp1, (DTYPE_X)1.0f, this->tileLength);

        // 3) tmp2 = log(tmp1) = softplus(x)
        AscendC::Log(tmp2, tmp1, this->tileLength);

        // 4) tmp2 = tanh(softplus(x))
        AscendC::Tanh(tmp2, tmp2, this->tileLength);

        // 5) yLocal_partial = x * tanh(softplus(x)) = Mish(x)
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::Mul(yLocal, xLocal, tmp2, this->tileLength);

        // 6) Final activation: y = tanh(mish(x))
        AscendC::LocalTensor<DTYPE_Y> yLocalFinal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::Tanh(yLocalFinal, yLocal, this->tileLength);

        // Enqueue output
        outQueueY.EnQue<DTYPE_Y>(yLocalFinal);

        // Free temporaries and input
        outQueueY.FreeTensor(yLocal);
        tmpQueue1.FreeTensor(tmp1);
        tmpQueue2.FreeTensor(tmp2);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocalFinal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocalFinal, this->tileLength);
        outQueueY.FreeTensor(yLocalFinal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue1;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue2;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv3d_mish_tanh_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMishTanh op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: perform nn.Conv3d, then fused Mish+Tanh via custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)

    def forward(self, x):
        x = self.conv(x)
        x = custom_ops_lib.conv3d_mish_tanh_custom(x)
        return x
