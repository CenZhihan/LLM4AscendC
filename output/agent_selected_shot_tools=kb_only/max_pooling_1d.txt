#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaxPooling1dCustomTilingData tiling;

    // 获取输入形状 N,C,L
    const gert::Shape* in_shape = context->GetInputShape(0);
    const auto& ori = in_shape->GetOriginShape();
    uint32_t n = static_cast<uint32_t>(ori.GetDim(0));
    uint32_t c = static_cast<uint32_t>(ori.GetDim(1));
    uint32_t l = static_cast<uint32_t>(ori.GetDim(2));

    // 获取属性：kernel_size、stride、padding
    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    uint32_t k = *(attrs->GetAttrPointer<uint32_t>(0));      // kernel_size
    uint32_t s = *(attrs->GetAttrPointer<uint32_t>(1));      // stride
    uint32_t p = *(attrs->GetAttrPointer<uint32_t>(2));      // padding

    // 计算输出长度
    uint32_t outLen = (l + 2 * p >= k) ? ((l + 2 * p - k) / s + 1) : 0;

    // 设置BlockDim与每个block处理的行数（按N*C行划分）
    uint32_t totalRows = n * c;
    uint32_t rowsPerBlock = CeilDiv(totalRows, BLOCK_DIM);

    // 设置tiling数据
    context->SetBlockDim(BLOCK_DIM);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_l(l);
    tiling.set_k(k);
    tiling.set_stride(s);
    tiling.set_padding(p);
    tiling.set_outLen(outLen);
    tiling.set_rowsPerBlock(rowsPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}


namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // 输入形状
    const gert::Shape* in_shape = context->GetInputShape(0);
    const auto& ori = in_shape->GetOriginShape();
    uint64_t n = ori.GetDim(0);
    uint64_t c = ori.GetDim(1);
    uint64_t l = ori.GetDim(2);

    // 从属性获取kernel/stride/padding，推导输出长度
    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    uint64_t k = *(attrs->GetAttrPointer<uint32_t>(0));
    uint64_t s = *(attrs->GetAttrPointer<uint32_t>(1));
    uint64_t p = *(attrs->GetAttrPointer<uint32_t>(2));
    uint64_t outLen = (l + 2 * p >= k) ? ((l + 2 * p - k) / s + 1) : 0;

    out_shape->SetDimNum(3);
    out_shape->SetDim(0, n);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // 每个队列的张量个数（行级处理）

class KernelMaxPool1d {
public:
    __aicore__ inline KernelMaxPool1d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t l,
                                uint32_t k, uint32_t stride, uint32_t padding,
                                uint32_t outLen, uint32_t rowsPerBlock)
    {
        this->n = n;
        this->c = c;
        this->l = l;
        this->k = k;
        this->stride = stride;
        this->padding = padding;
        this->outLen = outLen;
        this->rowsPerBlock = rowsPerBlock;

        uint32_t totalRows = n * c;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > totalRows) endRow = totalRows;
        this->startRow = startRow;
        this->endRow = endRow;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y);

        // 每行局部缓冲长度（含padding）
        tileInLen = l + 2 * padding;
        tileOutLen = outLen;

        // 初始化UB队列
        pipe.InitBuffer(inQueue, BUFFER_NUM, tileInLen * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, tileOutLen * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t row = startRow; row < endRow; ++row) {
            CopyInRow(row);
            ComputeRow();
            CopyOutRow(row);
        }
    }

private:
    __aicore__ inline void CopyInRow(uint32_t rowIdx)
    {
        // 分配并清零（padding使用0）
        inLocal = inQueue.AllocTensor<DTYPE_X>();
        AscendC::Duplicate<DTYPE_X>(inLocal, DTYPE_X(0), tileInLen);

        // 将一行输入复制到中间位置 [padding, padding + l)
        uint64_t inOffset = static_cast<uint64_t>(rowIdx) * l;
        AscendC::DataCopy(inLocal[padding], xGm[inOffset], l);
        inQueue.EnQue(inLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<DTYPE_X> srcLocal = inQueue.DeQue<DTYPE_X>();
        outLocal = outQueue.AllocTensor<DTYPE_Y>();

        // 仅针对stride==1进行矢量化计算：
        // out[j] = max_{i in [0,k)} srcLocal[j + i], j in [0, outLen)
        if (stride == 1) {
            // 初始化为第0个窗口切片
            AscendC::DataCopy(outLocal, srcLocal[0], outLen);
            for (uint32_t i = 1; i < k; ++i) {
                AscendC::Max(outLocal, outLocal, srcLocal[i], outLen);
            }
        } else {
            // 简化实现：通用stride（效率较低但保证正确性）
            // 逐位置计算：out[j] = max(srcLocal[j*stride + t], t=0..k-1)
            // 为避免逐元素API，这里采用分段处理
            // 每次处理chunk元素，chunk大小选为ONE_REPEAT_BYTE_SIZE/sizeof(T)
            const uint32_t chunk = AscendC::ONE_REPEAT_BYTE_SIZE / sizeof(DTYPE_X);
            uint32_t done = 0;
            while (done < outLen) {
                uint32_t cur = (outLen - done) > chunk ? chunk : (outLen - done);
                // 将第一列拷贝到outLocal片段
                for (uint32_t idx = 0; idx < cur; ++idx) {
                    uint32_t pos = done + idx;
                    uint32_t base = pos * stride;
                    outLocal[pos] = srcLocal[base];
                }
                // 累积Max
                for (uint32_t i = 1; i < k; ++i) {
                    for (uint32_t idx = 0; idx < cur; ++idx) {
                        uint32_t pos = done + idx;
                        uint32_t base = pos * stride + i;
                        // 使用两步：临时拷贝至同一位置，然后逐元素比较更新
                        // 由于缺少标量API，这里直接利用LocalTensor的元素访问
                        DTYPE_X curv = srcLocal[base];
                        DTYPE_Y &dstv = outLocal[pos];
                        dstv = dstv > curv ? dstv : curv;
                    }
                }
                done += cur;
            }
        }

        outQueue.EnQue<DTYPE_Y>(outLocal);
        inQueue.FreeTensor(srcLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.DeQue<DTYPE_Y>();
        uint64_t outOffset = static_cast<uint64_t>(rowIdx) * outLen;
        AscendC::DataCopy(yGm[outOffset], yLocal, outLen);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    AscendC::LocalTensor<DTYPE_X> inLocal;
    AscendC::LocalTensor<DTYPE_Y> outLocal;

    uint32_t n, c, l;
    uint32_t k, stride, padding;
    uint32_t outLen;
    uint32_t rowsPerBlock;
    uint32_t startRow, endRow;

    uint32_t tileInLen;
    uint32_t tileOutLen;
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPool1d op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.l,
            tiling_data.k, tiling_data.stride, tiling_data.padding,
            tiling_data.outLen, tiling_data.rowsPerBlock);
    op.Process();
}
"""

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor max_pooling_1d_custom_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding, bool return_indices) {
    TORCH_CHECK(self.dim() == 3, \"input must be 3-D (N, C, L)\");
    TORCH_CHECK(!return_indices, \"return_indices is not supported in this custom kernel\");

    auto N = self.size(0);
    auto C = self.size(1);
    auto L = self.size(2);
    int64_t s = stride;
    if (s <= 0) {
        TORCH_CHECK(false, \"stride must be positive\");
    }
    int64_t outLen = (L + 2 * padding >= kernel_size) ? ((L + 2 * padding - kernel_size) / s + 1) : 0;
    at::Tensor result = at::empty({N, C, outLen}, self.options());

    EXEC_NPU_CMD(aclnnMaxPooling1dCustom, self, (int)kernel_size, (int)stride, (int)padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"max_pooling_1d_custom\", &max_pooling_1d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"max_pooling_1d_custom\", &max_pooling_1d_custom_impl_npu, \"MaxPool1d custom\");
