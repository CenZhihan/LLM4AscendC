kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数量

class KernelMean {
public:
    __aicore__ inline KernelMean() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        // 将totalLength视为N*C*D*H*W，按block数等分
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = AscendC::max(1u, (this->blockLength / tileNum / BUFFER_NUM));

        xGm.SetGlobalBuffer((__gm__ float*)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ float*)y + AscendC::GetBlockIdx(), 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueueWork, 1, 32 * sizeof(float)); // ReduceSum工作空间，最小64B
        pipe.InitBuffer(tmpQueueDst, BUFFER_NUM, 32);         // ReduceSum的dst占位（对acc取值用）
        pipe.InitBuffer(outQueueY, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t loopCount = this->tileNum * BUFFER_NUM;
        uint32_t consumed = 0;

        for (uint32_t i = 0; i < loopCount && consumed < this->blockLength; ++i) {
            uint32_t curLen = AscendC::min(this->tileLength, this->blockLength - consumed);
            CopyIn(i, consumed, curLen);
            Compute(i, curLen);
            consumed += curLen;
        }

        // 从累加器中取出总和
        float sum = AscendC::GetAccVal<float>();
        float mean = (this->blockLength > 0) ? (sum / static_cast<float>(this->blockLength)) : 0.0f;

        // 将mean写回到y（每个block对应一个样本/分块）
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        AscendC::Duplicate(yLocal, mean, 1);
        outQueueY.EnQue<float>(yLocal);
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress, uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[offset], len);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t progress, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> dstLocal = tmpQueueDst.AllocTensor<float>();    // 仅用于调用接口
        AscendC::LocalTensor<float> workLocal = tmpQueueWork.AllocTensor<float>();  // ReduceSum工作缓冲
        AscendC::ReduceSum(dstLocal, xLocal, workLocal, static_cast<int32_t>(len));
        tmpQueueDst.FreeTensor(dstLocal);
        inQueueX.FreeTensor(xLocal);
        tmpQueueWork.FreeTensor(workLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm, yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,   BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>          tmpQueueWork;
    AscendC::TQue<AscendC::TPosition::VECOUT,  BUFFER_NUM> tmpQueueDst;
    AscendC::TQue<AscendC::TPosition::VECOUT,  1>          outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv3d_group_norm_mean_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMean op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    3D卷积 + GroupNorm + 自定义AscendC核求每样本均值
    """
    def __init__(self, in_channels, out_channels, kernel_size, num_groups):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_channels)

    def forward(self, x):
        x = self.conv(x)
        x = self.group_norm(x)
        # 使用自定义AscendC算子计算在(C, D, H, W)维度上的均值，输出(batch,)
        y = custom_ops_lib.conv3d_group_norm_mean_custom(x)
        return y
