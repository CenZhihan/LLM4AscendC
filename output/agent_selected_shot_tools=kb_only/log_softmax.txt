host_operator_src="""
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);

    // 获取输入形状
    const gert::Shape *x_shape = context->GetInputShape(0);
    const ge::Shape origin = x_shape->GetOriginShape();
    const std::vector<int64_t> dims = origin.GetDims();
    const int64_t rank = static_cast<int64_t>(dims.size());

    int64_t dim = *dimAttr;
    if (dim < 0) dim += rank;
    // 计算 outter(inner 之外各轴乘积) 与 inner(目标维长度)
    uint64_t outter_u64 = 1;
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        outter_u64 *= static_cast<uint64_t>(dims[i]);
    }
    uint64_t inner_u64 = static_cast<uint64_t>(dims[dim]);

    uint32_t outter = static_cast<uint32_t>(outter_u64);
    uint32_t inner = static_cast<uint32_t>(inner_u64);

    // LogSoftMax高阶API按最后一维做归约，这里假设dim为最后一维使用
    // 若dim不是最后一维，建议在图优化阶段调整轴顺序；此处仍按[m,k]传入tiling
    std::vector<int64_t> shapeVec = {static_cast<int64_t>(outter), static_cast<int64_t>(inner)};
    ge::Shape srcShape(shapeVec);

    // 选择最小临时空间以确保功能正确
    const uint32_t tmpSize = AscendC::GetLogSoftMaxMinTmpSize(srcShape, sizeof(float), false);
    // 获取LogSoftMax的tiling参数
    AscendC::LogSoftMaxTilingFunc(srcShape, sizeof(float), tmpSize, tiling.logSoftmaxTilingData);

    tiling.set_m(outter);
    tiling.set_k(inner);
    tiling.set_tmpSize(tmpSize);
    tiling.set_dim(static_cast<int32_t>(dim));

    // 设置block数（可根据实际情况调整）
    context->SetBlockDim(8);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // 不需要GM侧workspace

    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore()
            .SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(LogSoftmaxCustom);
}

kernel_src="""
#include "kernel_operator.h"
constexpr int32_t BUFFER_NUM = 1; // 简化为单缓冲

class KernelLogSoftmax {
public:
    __aicore__ inline KernelLogSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t m, uint32_t k,
                                uint32_t tmpSize,
                                const optiling::LogSoftMaxTiling &tiling)
    {
        this->m = m;
        this->k = k;
        this->tmpSize = tmpSize;
        this->tiling = tiling;

        // 以行（M维）为单位在多核间划分
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerCore = (m + blockNum - 1) / blockNum;
        this->startRow = rowsPerCore * blockIdx;
        this->endRow = AscendC::Min(m, this->startRow + rowsPerCore);
        this->rowsThisCore = (this->endRow > this->startRow) ? (this->endRow - this->startRow) : 0;

        // 设定当前核的GM起始地址
        xGm.SetGlobalBuffer((__gm__ float*)x + static_cast<uint64_t>(this->startRow) * k,
                            static_cast<uint64_t>(this->rowsThisCore) * k);
        yGm.SetGlobalBuffer((__gm__ float*)y + static_cast<uint64_t>(this->startRow) * k,
                            static_cast<uint64_t>(this->rowsThisCore) * k);

        // 初始化本地缓冲
        // 每次处理一行（outter=1）
        pipe.InitBuffer(inQueueX, BUFFER_NUM, k * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, k * sizeof(float));
        // 临时空间（由Host侧LogSoftMaxTilingFunc给出大小）
        pipe.InitBuffer(tmpBuf, tmpSize);
        // max / sum中间结果缓冲（大小按一行上限配置）
        pipe.InitBuffer(maxBuf, k * sizeof(float));
        pipe.InitBuffer(sumBuf, k * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsThisCore; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        // 从GM拷贝一行到本地
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->k], this->k);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*rowIdx*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        AscendC::LocalTensor<uint8_t> tmpLocal = tmpBuf.Get<uint8_t>();
        AscendC::LocalTensor<float> maxLocal = maxBuf.Get<float>();
        AscendC::LocalTensor<float> sumLocal = sumBuf.Get<float>();

        AscendC::SoftMaxShapeInfo softmaxInfo = {1, this->k, 1, this->k};
        // 调用高阶LogSoftMax API
        AscendC::LogSoftMax<float, false>(yLocal, maxLocal, sumLocal, xLocal, tmpLocal, this->tiling, softmaxInfo);

        outQueueY.EnQue<float>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[rowIdx * this->k], yLocal, this->k);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TBuf<AscendC::QuePosition::VECCALC> tmpBuf;
    AscendC::TBuf<AscendC::QuePosition::VECCALC> maxBuf;
    AscendC::TBuf<AscendC::QuePosition::VECCALC> sumBuf;

    AscendC::GlobalTensor<float> xGm, yGm;

    uint32_t m;
    uint32_t k;
    uint32_t tmpSize;
    uint32_t startRow;
    uint32_t endRow;
    uint32_t rowsThisCore;
    optiling::LogSoftMaxTiling tiling;
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelLogSoftmax op;
    op.Init(x, y,
            tilingData.m, tilingData.k,
            tilingData.tmpSize,
            tilingData.logSoftmaxTilingData);
    op.Process();
}
"""
