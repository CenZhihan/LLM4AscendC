kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulSigmoidSum {
public:
    __aicore__ inline KernelMatmulSigmoidSum() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batch, uint32_t inK, uint32_t outN, uint32_t tileK)
    {
        this->batch = batch;
        this->inK = inK;
        this->outN = outN;
        this->tileK = tileK;

        // Partition rows by blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        this->rowsPerBlock = (batch + blocks - 1) / blocks; // ceil
        this->rowStart = idx * this->rowsPerBlock;
        uint32_t tmpEnd = this->rowStart + this->rowsPerBlock;
        this->rowEnd = tmpEnd > batch ? batch : tmpEnd;

        xGm.SetGlobalBuffer((__gm__ float*)x, batch * inK);
        wGm.SetGlobalBuffer((__gm__ float*)w, outN * inK);  // weight is [N, K], row-major
        bGm.SetGlobalBuffer((__gm__ float*)b, outN);
        yGm.SetGlobalBuffer((__gm__ float*)y, batch);       // output [B, 1]

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(workQueue, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(scalarQueueA, BUFFER_NUM, 1 * sizeof(float));
        pipe.InitBuffer(scalarQueueB, BUFFER_NUM, 1 * sizeof(float));
        pipe.InitBuffer(scalarQueueTmp, BUFFER_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t m = rowStart; m < rowEnd; ++m) {
            // sumAcc = 0
            AscendC::LocalTensor<float> sumAcc = scalarQueueA.AllocTensor<float>();
            AscendC::Duplicate(sumAcc, 0.0f, 1);

            for (uint32_t n = 0; n < outN; ++n) {
                // dotAcc = 0
                AscendC::LocalTensor<float> dotAcc = scalarQueueB.AllocTensor<float>();
                AscendC::Duplicate(dotAcc, 0.0f, 1);

                // K reduction with tiling
                for (uint32_t k = 0; k < inK; k += tileK) {
                    uint32_t kt = (k + tileK <= inK) ? tileK : (inK - k);

                    AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                    AscendC::LocalTensor<float> wLocal = inQueueW.AllocTensor<float>();
                    AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();
                    AscendC::LocalTensor<float> workLocal = workQueue.AllocTensor<float>();
                    AscendC::LocalTensor<float> partial = scalarQueueTmp.AllocTensor<float>();

                    // Load tiles
                    AscendC::DataCopy(xLocal, xGm[m * inK + k], kt);
                    AscendC::DataCopy(wLocal, wGm[n * inK + k], kt);

                    // tmp = x * w
                    AscendC::Mul(tmpLocal, xLocal, wLocal, kt);

                    // partial = ReduceSum(tmp)
                    AscendC::ReduceSum(partial, tmpLocal, workLocal, kt);

                    // dotAcc += partial
                    AscendC::Add(dotAcc, dotAcc, partial, 1);

                    // free locals
                    tmpQueue.FreeTensor(tmpLocal);
                    workQueue.FreeTensor(workLocal);
                    inQueueX.FreeTensor(xLocal);
                    inQueueW.FreeTensor(wLocal);
                    scalarQueueTmp.FreeTensor(partial);
                }

                // dotAcc += bias[n]
                AscendC::LocalTensor<float> bLocal = scalarQueueTmp.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm[n], 1);
                AscendC::Add(dotAcc, dotAcc, bLocal, 1);
                scalarQueueTmp.FreeTensor(bLocal);

                // sigmoid(dotAcc) in-place: dotAcc = 1 / (1 + exp(-dotAcc))
                AscendC::Muls(dotAcc, dotAcc, -1.0f, 1);
                AscendC::Exp(dotAcc, dotAcc, 1);
                AscendC::Adds(dotAcc, dotAcc, 1.0f, 1);
                AscendC::Reciprocal(dotAcc, dotAcc, 1);

                // sumAcc += sigmoid
                AscendC::Add(sumAcc, sumAcc, dotAcc, 1);

                scalarQueueB.FreeTensor(dotAcc);
            }

            // write out sumAcc -> y[m]
            AscendC::DataCopy(yGm[m], sumAcc, 1);
            scalarQueueA.FreeTensor(sumAcc);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueW;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> workQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> scalarQueueA, scalarQueueB, scalarQueueTmp;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t batch;
    uint32_t inK;
    uint32_t outN;
    uint32_t tileK;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void matmul_sigmoid_sum_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                                                 GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSigmoidSum op;
    op.Init(x, w, b, y,
            tiling_data.batch, tiling_data.inK, tiling_data.outN, tiling_data.tileK);
    op.Process();
}
