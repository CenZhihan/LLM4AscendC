kernel_src="""
#include "kernel_operator.h"

// 本自定义核针对张量完成两步融合：
// 1) 逐元素除以常量(divisor) -> 使用Muls乘以(1/divisor)
// 2) 逐元素LeakyReLU(negativeSlope)

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2; // VECIN/VECOUT队列双缓冲

class KernelConv2dDivideLeakyRelu {
public:
    __aicore__ inline KernelConv2dDivideLeakyRelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, float divisor, float negativeSlope)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        this->mulScalar = 1.0f / divisor;
        this->leakySlope = negativeSlope;

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(midBuf, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> tLocal = midBuf.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();

        // 除以常量 -> 乘以(1/divisor)
        AscendC::Muls(tLocal, xLocal, this->mulScalar, this->tileLength);

        // LeakyReLU(t) with slope = leakySlope
        AscendC::LeakyRelu(yLocal, tLocal, this->leakySlope, this->tileLength);

        outQueueY.EnQue<DTYPE>(yLocal);
        midBuf.FreeTensor(tLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> midBuf;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    DTYPE mulScalar;
    DTYPE leakySlope;
};

extern "C" __global__ __aicore__ void conv2d_divide_leaky_relu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dDivideLeakyRelu op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileNum,
            tiling_data.divisor,
            tiling_data.negativeSlope);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC算子将 x / divisor + LeakyReLU 融合加速，Conv2d保持原生以便复用已有权重/布局。
    """
    def __init__(self, in_channels, out_channels, kernel_size, divisor):
        super().__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.divisor = float(divisor)
        self.negative_slope = 0.01

    def forward(self, x):
        x = self.conv(x)
        # 调用自定义算子：完成除以常量与LeakyReLU
        x = custom_ops_lib.conv2d_divide_leaky_relu_custom(x, self.divisor, self.negative_slope)
        return x
