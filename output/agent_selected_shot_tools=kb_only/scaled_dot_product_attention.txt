namespace optiling {
static constexpr uint32_t MAX_BLOCKS = 64;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ScaledDotProductAttentionCustomTilingData tiling;

    // 期望输入形状为 [B, H, S, D]
    const gert::Shape* q_shape = context->GetInputShape(0);
    if (q_shape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }
    const int64_t B = q_shape->GetOriginShape().GetDim(0);
    const int64_t H = q_shape->GetOriginShape().GetDim(1);
    const int64_t S = q_shape->GetOriginShape().GetDim(2);
    const int64_t D = q_shape->GetOriginShape().GetDim(3);

    tiling.set_B(static_cast<uint32_t>(B));
    tiling.set_H(static_cast<uint32_t>(H));
    tiling.set_S(static_cast<uint32_t>(S));
    tiling.set_D(static_cast<uint32_t>(D));
    const float scale = 1.0f / std::sqrt(static_cast<float>(D));
    tiling.set_scale(scale);

    // 一个block处理一个 (b,h) 头
    uint32_t block_dim = static_cast<uint32_t>(B * H);
    if (block_dim == 0) {
        return ge::GRAPH_FAILED;
    }
    block_dim = block_dim > MAX_BLOCKS ? MAX_BLOCKS : block_dim;
    context->SetBlockDim(block_dim);

    // Workspace: 为每个 (b,h) 分配 S*S 的中间注意力矩阵 (float)
    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = static_cast<size_t>(B * H) * static_cast<size_t>(S) * static_cast<size_t>(S) * sizeof(float);

    // 保存tiling
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

// 简化实现：每个Block处理一个(b,h)头: Q[b,h,S,D], K[b,h,S,D], V[b,h,S,D]
// 计算: P = softmax((Q * K^T) * scale), 输出 O = P * V
// 注意：示例参考Matmul NORMAL布局与SoftMax高阶API用法

using dtype = float;

struct SDPAParams {
    uint32_t B;
    uint32_t H;
    uint32_t S;
    uint32_t D;
    float scale;
};

class KernelSDPA {
public:
    __aicore__ inline KernelSDPA() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(tiling_data, tiling);
        this->params.B = tiling_data.B;
        this->params.H = tiling_data.H;
        this->params.S = tiling_data.S;
        this->params.D = tiling_data.D;
        this->params.scale = tiling_data.scale;

        bhNum = params.B * params.H;
        bhIdx = AscendC::GetBlockIdx();

        // 设置指针到当前(b,h)
        uint64_t perHeadElems_QKD = static_cast<uint64_t>(params.S) * static_cast<uint64_t>(params.D);
        uint64_t perHeadElems_S2  = static_cast<uint64_t>(params.S) * static_cast<uint64_t>(params.S);

        qGm.SetGlobalBuffer((__gm__ dtype*)q + bhIdx * perHeadElems_QKD, perHeadElems_QKD);
        kGm.SetGlobalBuffer((__gm__ dtype*)k + bhIdx * perHeadElems_QKD, perHeadElems_QKD);
        vGm.SetGlobalBuffer((__gm__ dtype*)v + bhIdx * perHeadElems_QKD, perHeadElems_QKD);
        oGm.SetGlobalBuffer((__gm__ dtype*)o + bhIdx * perHeadElems_QKD, perHeadElems_QKD);

        // workspace为注意力矩阵 P 或中间 scores: [S, S]
        attnGm.SetGlobalBuffer((__gm__ dtype*)workspace + bhIdx * perHeadElems_S2, perHeadElems_S2);

        // 为SoftMax行缓冲初始化队列buffer（每次处理一行S元素）
        const uint32_t rowBytes = params.S * sizeof(dtype);
        pipe.InitBuffer(inRowQ, 1, rowBytes);  // 行缓冲: 输入/输出共享
        pipe.InitBuffer(inRowTmp, 1, rowBytes);
    }

    __aicore__ inline void Process()
    {
        // 1) 计算 scores = Q * K^T，保存到 attnGm[S,S]
        MatmulQKt();

        // 2) 对每一行做缩放和SoftMax: P = softmax(scores * scale)
        RowWiseSoftmax();

        // 3) 计算 O = P * V，输出到 oGm[S,D]
        MatmulPV();
    }

private:
    // 使用Matmul NORMAL布局计算 Q[S,D] * K[D,S]^T => [S,S]
    __aicore__ inline void MatmulQKt()
    {
        typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype, false, LayoutMode::NORMAL> aType;
        typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype, true,  LayoutMode::NORMAL> bType; // 允许转置
        typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype, false, LayoutMode::NORMAL> cType;
        typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype> biasType;

        TCubeTiling tiling;
        // 简化填充必要字段（NORMAL场景）
        tiling.singleCoreM = params.S;
        tiling.singleCoreK = params.D;
        tiling.singleCoreN = params.S;
        tiling.ALayoutInfoB = 1;
        tiling.BLayoutInfoB = 1;
        tiling.CLayoutInfoB = 1;
        tiling.isBias = 0;
        tiling.shareMode = 0;
        tiling.shareL1Size = 512 * 1024;
        tiling.shareL0CSize = 128 * 1024;
        tiling.shareUbSize = 0;

        AscendC::GlobalTensor<dtype> gm_a, gm_b, gm_c;
        gm_a.SetGlobalBuffer(qGm.GetPhyAddr(), params.S * params.D);
        gm_b.SetGlobalBuffer(kGm.GetPhyAddr(), params.S * params.D);
        gm_c.SetGlobalBuffer(attnGm.GetPhyAddr(), params.S * params.S);

        matmul::Matmul<aType, bType, cType, biasType> mm;
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
        mm.Init(&tiling);
        // A: [S,D], B: [S,D] with transpose enabled -> Treat B as [D,S]
        mm.SetTensorA(gm_a, 0);
        mm.SetTensorB(gm_b, 1);
        mm.SetWorkspace(nullptr, 0);
        // 单次批次
        mm.IterateBatch(gm_c, 1, 1, false);
    }

    // 对每一行 scores[i,:] 执行: scores *= scale; softmax(scores) => 写回 attnGm[i,:]
    __aicore__ inline void RowWiseSoftmax()
    {
        for (uint32_t row = 0; row < params.S; ++row) {
            // 读取一行到UB
            AscendC::LocalTensor<dtype> rowLocal = inRowQ.AllocTensor<dtype>();
            AscendC::DataCopy(rowLocal, attnGm[row * params.S], params.S);

            // 缩放
            AscendC::Muls(rowLocal, rowLocal, params.scale, params.S);

            // SoftMax: 行向量 [1, S]
            // 这里使用API框架分配临时空间路径（不传sharedTmpBuffer），并开启内部形状一致性检查
            SoftMaxTiling softTiling; // 使用默认，内部会根据形状检查并修正
            SoftMaxShapeInfo shapeInfo;
            shapeInfo.srcShape = {1, params.S}; // 伪代码：表示行向量形状（实现中仅作占位传递）
            AscendC::LocalTensor<dtype> rowOut = inRowTmp.AllocTensor<dtype>();
            SoftMax<dtype>(rowOut, rowLocal, softTiling, shapeInfo);

            // 写回
            AscendC::DataCopy(attnGm[row * params.S], rowOut, params.S);
            inRowQ.FreeTensor(rowLocal);
            inRowTmp.FreeTensor(rowOut);
        }
    }

    // 使用Matmul NORMAL布局计算 O = P[S,S] * V[S,D] => [S,D]
    __aicore__ inline void MatmulPV()
    {
        typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype, false, LayoutMode::NORMAL> aType;
        typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype, false, LayoutMode::NORMAL> bType;
        typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype, false, LayoutMode::NORMAL> cType;
        typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype> biasType;

        TCubeTiling tiling;
        tiling.singleCoreM = params.S;
        tiling.singleCoreK = params.S;
        tiling.singleCoreN = params.D;
        tiling.ALayoutInfoB = 1;
        tiling.BLayoutInfoB = 1;
        tiling.CLayoutInfoB = 1;
        tiling.isBias = 0;
        tiling.shareMode = 0;
        tiling.shareL1Size = 512 * 1024;
        tiling.shareL0CSize = 128 * 1024;
        tiling.shareUbSize = 0;

        AscendC::GlobalTensor<dtype> gm_a, gm_b, gm_c;
        gm_a.SetGlobalBuffer(attnGm.GetPhyAddr(), params.S * params.S);
        gm_b.SetGlobalBuffer(vGm.GetPhyAddr(), params.S * params.D);
        gm_c.SetGlobalBuffer(oGm.GetPhyAddr(), params.S * params.D);

        matmul::Matmul<aType, bType, cType, biasType> mm;
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
        mm.Init(&tiling);
        mm.SetTensorA(gm_a, 0);
        mm.SetTensorB(gm_b, 0);
        mm.SetWorkspace(nullptr, 0);
        mm.IterateBatch(gm_c, 1, 1, false);
    }

private:
    SDPAParams params;
    uint32_t bhNum;
    uint32_t bhIdx;

    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inRowQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> inRowTmp;

    AscendC::GlobalTensor<dtype> qGm;
    AscendC::GlobalTensor<dtype> kGm;
    AscendC::GlobalTensor<dtype> vGm;
    AscendC::GlobalTensor<dtype> oGm;
    AscendC::GlobalTensor<dtype> attnGm;
};

extern "C" __global__ __aicore__ void scaled_dot_product_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling) {
    KernelSDPA op;
    op.Init(q, k, v, o, workspace, tiling);
    op.Process();
}
