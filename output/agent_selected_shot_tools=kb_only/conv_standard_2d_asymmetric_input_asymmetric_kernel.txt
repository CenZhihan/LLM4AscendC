kernel_src="""
#include "kernel_operator.h"

using data_t = half;

class KernelConv2D {
public:
    __aicore__ inline KernelConv2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR z,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t wi,
                                uint32_t cout, uint32_t kh, uint32_t kw,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_l, uint32_t pad_r, uint32_t pad_t, uint32_t pad_b,
                                uint32_t dil_h, uint32_t dil_w)
    {
        this->n_ = n; this->c_ = c; this->h_ = h; this->w_ = wi;
        this->cout_ = cout; this->kh_ = kh; this->kw_ = kw;
        this->stride_h_ = stride_h; this->stride_w_ = stride_w;
        this->pad_l_ = pad_l; this->pad_r_ = pad_r; this->pad_t_ = pad_t; this->pad_b_ = pad_b;
        this->dil_h_ = dil_h; this->dil_w_ = dil_w;

        // 按 NCHW 输入、[Cout, Cin, Kh, Kw] 权重，输出 NCHW
        uint32_t x_elems = n_ * c_ * h_ * w_;
        uint32_t w_elems = cout_ * c_ * kh_ * kw_;

        // 计算输出尺寸（与Host相同公式）
        uint32_t ho = (h_ + pad_t_ + pad_b_ - dil_h_ * (kh_ - 1) - 1) / stride_h_ + 1;
        uint32_t wo = (w_ + pad_l_ + pad_r_ - dil_w_ * (kw_ - 1) - 1) / stride_w_ + 1;
        this->ho_ = ho;
        this->wo_ = wo;
        uint32_t y_elems = n_ * cout_ * ho_ * wo_;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ data_t *>(x), x_elems);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ data_t *>(w), w_elems);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ data_t *>(z), y_elems);

        // 简化：整体搬运到本地再计算（示例代码）
        pipe.InitBuffer(inQueueX, 1, x_elems * sizeof(data_t));
        pipe.InitBuffer(inQueueW, 1, w_elems * sizeof(data_t));
        pipe.InitBuffer(outQueueY, 1, y_elems * sizeof(data_t));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<data_t> xLocal = inQueueX.AllocTensor<data_t>();
        AscendC::LocalTensor<data_t> wLocal = inQueueW.AllocTensor<data_t>();
        AscendC::DataCopy(xLocal, xGm, n_ * c_ * h_ * w_);
        AscendC::DataCopy(wLocal, wGm, cout_ * c_ * kh_ * kw_);
        inQueueX.EnQue(xLocal);
        inQueueW.EnQue(wLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<data_t> xLocal = inQueueX.DeQue<data_t>();
        AscendC::LocalTensor<data_t> wLocal = inQueueW.DeQue<data_t>();
        AscendC::LocalTensor<data_t> yLocal = outQueueY.AllocTensor<data_t>();

        // 构造Conv2dParams并获取tilling
        AscendC::Conv2dParams params;
        params.imgShape[0] = h_;
        params.imgShape[1] = w_;
        params.kernelShape[0] = kh_;
        params.kernelShape[1] = kw_;
        params.stride[0] = stride_h_;
        params.stride[1] = stride_w_;
        params.cin = c_;
        params.cout = cout_;
        params.padList[0] = pad_l_;
        params.padList[1] = pad_r_;
        params.padList[2] = pad_t_;
        params.padList[3] = pad_b_;
        params.dilation[0] = dil_h_;
        params.dilation[1] = dil_w_;
        params.initY = 1;        // 覆盖写
        params.partialSum = 0;

        AscendC::Conv2dTilling tilling = AscendC::GetConv2dTiling<data_t>(params);

        // 调用Conv2D（示例使用，具体格式/对齐请按实际数据布局准备）
        AscendC::Conv2D<data_t, data_t>(yLocal, xLocal, wLocal, params, tilling);

        outQueueY.EnQue(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueW.FreeTensor(wLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<data_t> yLocal = outQueueY.DeQue<data_t>();
        AscendC::DataCopy(yGm, yLocal, n_ * cout_ * ho_ * wo_);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<data_t> xGm;
    AscendC::GlobalTensor<data_t> wGm;
    AscendC::GlobalTensor<data_t> yGm;

    uint32_t n_, c_, h_, w_;
    uint32_t cout_, kh_, kw_;
    uint32_t stride_h_, stride_w_;
    uint32_t pad_l_, pad_r_, pad_t_, pad_b_;
    uint32_t dil_h_, dil_w_;
    uint32_t ho_, wo_;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    KernelConv2D op;
    op.Init(x, w, z,
            tilingData.n(), tilingData.c(), tilingData.h(), tilingData.w(),
            tilingData.cout(), tilingData.kh(), tilingData.kw(),
            tilingData.stride_h(), tilingData.stride_w(),
            tilingData.pad_l(), tilingData.pad_r(), tilingData.pad_t(), tilingData.pad_b(),
            tilingData.dil_h(), tilingData.dil_w());
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义 AscendC 内核替换标准 Conv2d（假设 stride=(1,1), padding=(0,0), dilation=(1,1), groups=1，无偏置）。
    注意：算子当前以 float16 进行计算。
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: tuple = (1, 1), padding: tuple = (0, 0), dilation: tuple = (1, 1), groups: int = 1, bias: bool = False):
        super().__init__()
        assert bias is False, "当前自定义算子示例不支持 bias"
        assert groups == 1, "当前自定义算子示例仅支持 groups=1"
        # 仅用于存储权重（参数形状遵循 PyTorch Conv2d: [out_channels, in_channels, Kh, Kw]）
        Kh, Kw = kernel_size
        weight = torch.empty(out_channels, in_channels, Kh, Kw)
        torch.nn.init.kaiming_uniform_(weight, a=5**0.5)
        self.weight = torch.nn.Parameter(weight)

        # 记录参数（当前内核以默认 stride=(1,1), padding=(0,0), dilation=(1,1) 执行）
        self.kernel_size = kernel_size

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # 转为半精度以匹配内核与项目定义
        x_fp16 = x.to(torch.float16)
        w_fp16 = self.weight.to(torch.float16)
        y_fp16 = custom_ops_lib.conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(x_fp16, w_fp16)
        # 输出按需转回输入 dtype
        return y_fp16.to(x.dtype)
