namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 获取形状：x:[B, N]，idx:[B, K]，updates:[B, K]
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* idx_shape = context->GetInputShape(1);
    if (x_shape == nullptr || idx_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    // 假设形状为2D：[B, N] 和 [B, K]
    uint32_t rows = (x_shape->GetOriginShape().GetDimNum() >= 1) ? static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0)) : 1;
    uint32_t cols = (x_shape->GetOriginShape().GetDimNum() >= 2) ? static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1)) : static_cast<uint32_t>(x_shape->GetOriginShape().GetShapeSize());
    uint32_t updatesLen = (idx_shape->GetOriginShape().GetDimNum() >= 2) ? static_cast<uint32_t>(idx_shape->GetOriginShape().GetDim(1)) : static_cast<uint32_t>(idx_shape->GetOriginShape().GetShapeSize());

    // 设置block数为batch行数
    if (rows == 0) rows = 1;
    context->SetBlockDim(rows);

    // 设置每次处理的K分块大小（可根据UB容量调整）
    const uint32_t TILE_K = 256;

    ScatterAddCustomTilingData tiling;
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_updatesLen(updatesLen);
    tiling.set_tileK(TILE_K);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

class KernelScatterAdd {
public:
    __aicore__ inline KernelScatterAdd() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR updates, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t updatesLen, uint32_t tileK)
    {
        this->rows = rows;
        this->cols = cols;
        this->updatesLen = updatesLen;
        this->tileK = tileK;

        // 当前core处理第blockIdx行
        uint32_t rowId = AscendC::GetBlockIdx();
        if (rowId >= this->rows) {
            rowId = this->rows - 1; // 简单保护
        }

        xGm.SetGlobalBuffer((__gm__ float*)x + rowId * this->cols, this->cols);
        yGm.SetGlobalBuffer((__gm__ float*)y + rowId * this->cols, this->cols);
        idxGm.SetGlobalBuffer((__gm__ int32_t*)idx + rowId * this->updatesLen, this->updatesLen);
        updGm.SetGlobalBuffer((__gm__ float*)updates + rowId * this->updatesLen, this->updatesLen);

        // Buffer配置：一行输出 + 索引/更新的K分块
        pipe.InitBuffer(inQueueIdx, 1, this->tileK * sizeof(int32_t));
        pipe.InitBuffer(inQueueUpd, 1, this->tileK * sizeof(float));
        pipe.InitBuffer(outQueueY, 1, this->cols * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // 将x这一行拷至本地作为y的起点
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        AscendC::DataCopy(yLocal, xGm, this->cols);

        // 遍历K分块，执行scatter_add到yLocal
        uint32_t processed = 0;
        while (processed < this->updatesLen) {
            uint32_t curTile = this->tileK;
            if (processed + curTile > this->updatesLen) {
                curTile = this->updatesLen - processed;
            }
            // 加载索引与更新块
            AscendC::LocalTensor<int32_t> idxLocal = inQueueIdx.AllocTensor<int32_t>();
            AscendC::LocalTensor<float> updLocal = inQueueUpd.AllocTensor<float>();
            AscendC::DataCopy(idxLocal, idxGm[processed], curTile);
            AscendC::DataCopy(updLocal, updGm[processed], curTile);

            // 执行逐元素散射累加
            // 注意：频繁SetValue/GetValue会影响性能，但保证易用性与语义正确性
            for (uint32_t i = 0; i < curTile; ++i) {
                int32_t col = idxLocal.GetValue(i);
                if (col >= 0 && (uint32_t)col < this->cols) {
                    float oldv = yLocal.GetValue((uint32_t)col);
                    float addv = updLocal.GetValue(i);
                    yLocal.SetValue((uint32_t)col, oldv + addv);
                }
            }

            inQueueIdx.FreeTensor(idxLocal);
            inQueueUpd.FreeTensor(updLocal);
            processed += curTile;
        }

        // 写回到GM
        AscendC::DataCopy(yGm, yLocal, this->cols);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueUpd;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<int32_t> idxGm;
    AscendC::GlobalTensor<float> updGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t updatesLen;
    uint32_t tileK;
};

extern "C" __global__ __aicore__ void scatter_add_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR updates, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelScatterAdd op;
    op.Init(x, idx, updates, y,
            tiling_data.rows, tiling_data.cols, tiling_data.updatesLen, tiling_data.tileK);
    op.Process();
}

at::Tensor scatter_add_custom_impl_npu(const at::Tensor& x,
                                       const at::Tensor& idx,
                                       const at::Tensor& updates) {

    at::Tensor idx_c = (idx.dtype() == at::kInt) ? idx.contiguous() : idx.to(at::kInt).contiguous();
