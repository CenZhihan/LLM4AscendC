kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_M = float;
using DTYPE_S = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2; // 队列中tensor个数

class KernelConv3dMulINClampMulMax {
public:
    __aicore__ inline KernelConv3dMulINClampMulMax() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR multiplier,
                                GM_ADDR clampMin,
                                GM_ADDR clampMax,
                                GM_ADDR y,
                                uint32_t n,
                                uint32_t c,
                                uint32_t spatial,
                                uint32_t tileNum,
                                uint32_t nPerBlock)
    {
        this->n = n;
        this->c = c;
        this->spatial = spatial;
        this->tileNum = tileNum;
        this->nPerBlock = nPerBlock;
        this->tileLength = this->spatial / tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint32_t>(n) * c * spatial);
        mGm.SetGlobalBuffer((__gm__ DTYPE_M *)multiplier, c); // shape: [C,1,1,1],读取第0元素即可
        minGm.SetGlobalBuffer((__gm__ DTYPE_S *)clampMin, 1);
        maxGm.SetGlobalBuffer((__gm__ DTYPE_S *)clampMax, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, static_cast<uint32_t>(n) * spatial);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // 读取clamp标量
        DTYPE_S clampMinScalar = 0.0f;
        DTYPE_S clampMaxScalar = 0.0f;
        {
            AscendC::LocalTensor<DTYPE_S> tmin = tmpQueue.AllocTensor<DTYPE_S>();
            AscendC::LocalTensor<DTYPE_S> tmax = tmpQueue.AllocTensor<DTYPE_S>();
            AscendC::DataCopy(tmin, minGm[0], 1);
            AscendC::DataCopy(tmax, maxGm[0], 1);
            pipe.Barrier();
            // 通过局部内存读取标量
            // 这里简单使用一次DataCopy到长度1的LocalTensor，然后拷贝到寄存变量
            AscendC::DataCopy(&clampMinScalar, tmin, 1);
            AscendC::DataCopy(&clampMaxScalar, tmax, 1);
            tmpQueue.FreeTensor(tmin);
            tmpQueue.FreeTensor(tmax);
        }

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t nStart = blockIdx * this->nPerBlock;
        uint32_t nEnd = AscendC::Min(nStart + this->nPerBlock, this->n);

        for (uint32_t ni = nStart; ni < nEnd; ++ni) {
            for (uint32_t t = 0; t < tileNum; ++t) {
                // 初始化当前tile的最大值为clampMinScalar
                AscendC::LocalTensor<DTYPE_Y> maxLocal = outQueueY.AllocTensor<DTYPE_Y>();
                AscendC::Dup(maxLocal, clampMinScalar, this->tileLength);

                for (uint32_t ci = 0; ci < c; ++ci) {
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                    // 复制输入x的当前通道tile
                    uint64_t base = (static_cast<uint64_t>(ni) * c + ci) * spatial + static_cast<uint64_t>(t) * tileLength;
                    AscendC::DataCopy(xLocal, xGm[base], this->tileLength);

                    // clamp: x <- min(max(x, clampMin), clampMax)
                    AscendC::Max(xLocal, xLocal, clampMinScalar, this->tileLength);
                    AscendC::Min(xLocal, xLocal, clampMaxScalar, this->tileLength);

                    // 读取对应通道的multiplier标量
                    DTYPE_M mulScalar = 1.0f;
                    {
                        AscendC::LocalTensor<DTYPE_M> ml = tmpQueue.AllocTensor<DTYPE_M>();
                        AscendC::DataCopy(ml, mGm[ci], 1);
                        pipe.Barrier();
                        AscendC::DataCopy(&mulScalar, ml, 1);
                        tmpQueue.FreeTensor(ml);
                    }
                    // 乘以multiplier标量
                    AscendC::Mul(xLocal, xLocal, mulScalar, this->tileLength);

                    // 更新max: maxLocal = max(maxLocal, xLocal)
                    AscendC::Max(maxLocal, maxLocal, xLocal, this->tileLength);

                    inQueueX.FreeTensor(xLocal);
                }

                // 写回输出
                uint64_t yBase = static_cast<uint64_t>(ni) * spatial + static_cast<uint64_t>(t) * tileLength;
                AscendC::DataCopy(yGm[yBase], maxLocal, this->tileLength);
                outQueueY.FreeTensor(maxLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_M> mGm;
    AscendC::GlobalTensor<DTYPE_S> minGm;
    AscendC::GlobalTensor<DTYPE_S> maxGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t spatial;
    uint32_t tileNum;
    uint32_t nPerBlock;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv3d_multiply_instance_norm_clamp_multiply_max_custom(
    GM_ADDR x, GM_ADDR multiplier, GM_ADDR clamp_min, GM_ADDR clamp_max, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMulINClampMulMax op;
    op.Init(x, multiplier, clamp_min, clamp_max, y,
            tiling_data.n(), tiling_data.c(), tiling_data.spatial(),
            tiling_data.tileNum(), tiling_data.nPerBlock());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版本：保留Conv3d与InstanceNorm3d由框架执行，将 clamp -> 乘法 -> 按通道最大值 的序列融合为自定义AscendC算子。
    """
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape, clamp_min, clamp_max):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.multiplier = torch.nn.Parameter(torch.randn(multiplier_shape))
        self.instance_norm = torch.nn.InstanceNorm3d(out_channels)
        self.clamp_min = clamp_min
        self.clamp_max = clamp_max

    def forward(self, x):
        x = self.conv(x)                       # [N, C, D, H, W]
        x = x * self.multiplier                # [N, C, D, H, W]
        x = self.instance_norm(x)              # [N, C, D, H, W]
        clamp_min_t = torch.tensor(self.clamp_min, dtype=x.dtype, device=x.device)
        clamp_max_t = torch.tensor(self.clamp_max, dtype=x.dtype, device=x.device)
        # 执行: clamp -> 乘法 -> 按通道最大值
        y = custom_ops_lib.conv3d_multiply_instance_norm_clamp_multiply_max_custom(
            x, self.multiplier, clamp_min_t, clamp_max_t
        )  # [N, D, H, W]
        return y
