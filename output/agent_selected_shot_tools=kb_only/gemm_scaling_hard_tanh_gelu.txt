kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

constexpr int32_t BUFFER_NUM = 2;

typedef float A_T;
typedef float B_T;
typedef float C_T;
typedef float BiasT;

class KernelGemmScalingHardTanhGelu {
public:
    __aicore__ inline KernelGemmScalingHardTanhGelu() {}
    __aicore__ inline void Init(GM_ADDR xGM, GM_ADDR wGM, GM_ADDR bGM, GM_ADDR yGM,
                                GM_ADDR scaleGM, GM_ADDR minGM, GM_ADDR maxGM,
                                GM_ADDR tilingGM, GM_ADDR workspaceGM) {
        // 读取TCubeTiling
        for (int i = 0; i < sizeof(TCubeTiling) / sizeof(int32_t); ++i) {
            ((__gm__ int32_t*)(&tiling))[i] = ((__gm__ int32_t*)tilingGM)[i];
        }

        // 读取标量参数
        scale_val = *reinterpret_cast<__gm__ float*>(scaleGM);
        min_val   = *reinterpret_cast<__gm__ float*>(minGM);
        max_val   = *reinterpret_cast<__gm__ float*>(maxGM);

        // 初始化GM Tensors
        aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(xGM),
            tiling.singleCoreM * tiling.singleCoreK);
        bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(wGM),
            tiling.singleCoreK * tiling.singleCoreN);
        biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ BiasT*>(bGM),
            tiling.singleCoreN);
        // 中间C存放在workspace
        cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(workspaceGM),
            tiling.singleCoreM * tiling.singleCoreN);
        yGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(yGM),
            tiling.singleCoreM * tiling.singleCoreN);

        // 初始化向量管道缓冲
        totalLen = tiling.singleCoreM * tiling.singleCoreN;
        pipe.InitBuffer(inQueueC, BUFFER_NUM, (totalLen / BUFFER_NUM) * sizeof(C_T));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, (totalLen / BUFFER_NUM) * sizeof(C_T));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, (totalLen / BUFFER_NUM) * sizeof(C_T));
    }

    __aicore__ inline void DoMatmul() {
        // 定义Matmul类型（NORMAL布局）
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType; // 以B转置
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
        typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

        set_atomic_none();

        matmul::Matmul<aType, bType, cType, biasType> mm;
        AscendC::TPipe matmulPipe;
        g_cubeTPipePtr = &matmulPipe;
        REGIST_MATMUL_OBJ(&matmulPipe, GetSysWorkSpacePtr(), mm);
        mm.Init(&tiling);

        mm.SetTensorA(aGlobal, false);
        mm.SetTensorB(bGlobal, true);
        mm.SetWorkspace(reinterpret_cast<GM_ADDR>(cGlobal.GetPhyAddr()), 0);
        if (tiling.isBias) {
            mm.SetBias(biasGlobal);
        }

        // 单次批次计算，输出到cGlobal
        mm.IterateBatch(cGlobal, 1, 1, false);
    }

    __aicore__ inline void PostProcess() {
        // 将C进行缩放->HardTanh->GELU，输出到Y
        uint32_t tileLen = totalLen / BUFFER_NUM;
        for (uint32_t i = 0; i < BUFFER_NUM; ++i) {
            // CopyIn
            AscendC::LocalTensor<C_T> cLocal = inQueueC.AllocTensor<C_T>();
            AscendC::DataCopy(cLocal, cGlobal[i * tileLen], tileLen);
            inQueueC.EnQue(cLocal);

            // Compute
            AscendC::LocalTensor<C_T> cIn  = inQueueC.DeQue<C_T>();
            AscendC::LocalTensor<C_T> tmpL = tmpQueue.AllocTensor<C_T>();
            AscendC::LocalTensor<C_T> yLocal = outQueueY.AllocTensor<C_T>();

            // 缩放
            AscendC::Muls(tmpL, cIn, scale_val, tileLen);
            // hardtanh: clamp到[min_val, max_val]
            AscendC::Maxs(tmpL, tmpL, min_val, tileLen);
            AscendC::Mins(tmpL, tmpL, max_val, tileLen);
            // GELU
            AscendC::Gelu(yLocal, tmpL, tileLen);

            outQueueY.EnQue<C_T>(yLocal);
            tmpQueue.FreeTensor(tmpL);
            inQueueC.FreeTensor(cIn);

            // CopyOut
            AscendC::LocalTensor<C_T> yOut = outQueueY.DeQue<C_T>();
            AscendC::DataCopy(yGlobal[i * tileLen], yOut, tileLen);
            outQueueY.FreeTensor(yOut);
        }
    }

private:
    TCubeTiling tiling;
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueC;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<BiasT> biasGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;
    AscendC::GlobalTensor<C_T> yGlobal;

    uint32_t totalLen = 0;
    float scale_val = 1.0f;
    float min_val = -1.0f;
    float max_val = 1.0f;
};

extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scale, GM_ADDR min_val, GM_ADDR max_val,
    GM_ADDR y, GM_ADDR tiling, GM_ADDR workspace)
{
    KernelGemmScalingHardTanhGelu op;
    op.Init(x, weight, bias, y, scale, min_val, max_val, tiling, workspace);
    op.DoMatmul();
    op.PostProcess();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, scaling_factor, hardtanh_min, hardtanh_max):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.scaling_factor = float(scaling_factor)
        self.hardtanh_min = float(hardtanh_min)
        self.hardtanh_max = float(hardtanh_max)

    def forward(self, x):
        # 将标量参数包装为1元素Tensor并移动到与x一致的设备与dtype
        scale = torch.tensor(self.scaling_factor, dtype=x.dtype, device=x.device)
        min_val = torch.tensor(self.hardtanh_min, dtype=x.dtype, device=x.device)
        max_val = torch.tensor(self.hardtanh_max, dtype=x.dtype, device=x.device)
        # 权重与偏置
        weight = self.gemm.weight
        bias = self.gemm.bias if self.gemm.bias is not None else torch.zeros(weight.size(0), dtype=x.dtype, device=x.device)
        # 自定义算子：GEMM->缩放->HardTanh->GELU
        return custom_ops_lib.gemm_scaling_hard_tanh_gelu_custom(x, weight, bias, scale, min_val, max_val)
