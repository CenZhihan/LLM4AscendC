#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"
#include "register/tilingdata_base.h"

namespace optiling {

const uint32_t BLOCK_DIM = 8;
// 简化演示用：固定输出tile大小，实际可根据UB容量自适应
const uint32_t TILE_H = 8;
const uint32_t TILE_W = 128;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaxPooling2dCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    // 假设输入为 NCHW
    uint32_t dim = in_shape->GetOriginShape().GetDimNum();
    if (dim != 4) {
        return ge::GRAPH_FAILED;
    }
    uint32_t n = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(0));
    uint32_t c = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(1));
    uint32_t h = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(2));
    uint32_t w = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(3));

    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    // Attr顺序需与OpDef一致：kernel_size, stride, padding
    uint32_t kernel = *(attrs->GetAttrPointer<uint32_t>(0));
    uint32_t stride = *(attrs->GetAttrPointer<uint32_t>(1));
    uint32_t padding = *(attrs->GetAttrPointer<uint32_t>(2));

    uint32_t out_h = 0;
    uint32_t out_w = 0;
    if (stride == 0) return ge::GRAPH_FAILED;
    if (h + 2 * padding < kernel || w + 2 * padding < kernel) {
        // 输出为空，简单兜底
        out_h = 0;
        out_w = 0;
    } else {
        out_h = (h + 2 * padding - kernel) / stride + 1;
        out_w = (w + 2 * padding - kernel) / stride + 1;
    }

    uint32_t blockNum = BLOCK_DIM;
    context->SetBlockDim(blockNum);
    uint32_t totalNC = n * c;
    uint32_t ncPerBlock = CeilDiv(totalNC, blockNum);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_out_h(out_h);
    tiling.set_out_w(out_w);
    tiling.set_kernel(kernel);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_ncPerBlock(ncPerBlock);
    tiling.set_tileH(TILE_H);
    tiling.set_tileW(TILE_W);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);

    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    uint32_t kernel = *(attrs->GetAttrPointer<uint32_t>(0));
    uint32_t stride = *(attrs->GetAttrPointer<uint32_t>(1));
    uint32_t padding = *(attrs->GetAttrPointer<uint32_t>(2));

    if (in_shape->GetOriginShape().GetDimNum() != 4) {
        return GRAPH_FAILED;
    }

    int64_t n = in_shape->GetOriginShape().GetDim(0);
    int64_t c = in_shape->GetOriginShape().GetDim(1);
    int64_t h = in_shape->GetOriginShape().GetDim(2);
    int64_t w = in_shape->GetOriginShape().GetDim(3);

    int64_t out_h = (h + 2 * padding - kernel) / (stride == 0 ? 1 : stride) + 1;
    int64_t out_w = (w + 2 * padding - kernel) / (stride == 0 ? 1 : stride) + 1;

    gert::Shape new_shape;
    new_shape.SetDimNum(4);
    new_shape.SetDim(0, n);
    new_shape.SetDim(1, c);
    new_shape.SetDim(2, out_h);
    new_shape.SetDim(3, out_w);
    *out_shape = new_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include <cstdint>

constexpr int32_t BUFFER_NUM = 1; // 简化：每队列1个buffer

template<typename T>
__aicore__ inline T NegInfVal();
template<>
__aicore__ inline float NegInfVal<float>() { return -3.402823466e+38F; } // float负无穷近似

class KernelMaxPool2D {
public:
    __aicore__ inline KernelMaxPool2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t out_h, uint32_t out_w,
                                uint32_t kernel, uint32_t stride, uint32_t padding,
                                uint32_t ncPerBlock, uint32_t tileH, uint32_t tileW)
    {
        this->n = n; this->c = c; this->h = h; this->w = w;
        this->out_h = out_h; this->out_w = out_w;
        this->kernel = kernel; this->stride = stride; this->padding = padding;
        this->ncPerBlock = ncPerBlock;
        this->tileH = tileH;
        this->tileW = tileW;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y);

        // 行缓冲区长度：对于stride=1时，窗口覆盖列范围为 tileW + kernel - 1
        // 为保持通用，保守按 tileW*stride + kernel - 1 申请
        uint32_t maxRowBuf = tileW * (this->stride == 0 ? 1 : this->stride) + this->kernel - 1;
        if (maxRowBuf < this->kernel) maxRowBuf = this->kernel;

        // 输出tile与工作缓冲
        pipe.InitBuffer(inRowQueue, BUFFER_NUM, maxRowBuf * sizeof(DTYPE_X));
        pipe.InitBuffer(accQueue, BUFFER_NUM, tileH * tileW * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, tileH * tileW * sizeof(DTYPE_Y));

        // block划分到NC维
        this->blockIdx = AscendC::GetBlockIdx();
        this->blockNum = AscendC::GetBlockNum();
        this->ncStart = this->blockIdx * this->ncPerBlock;
        uint32_t totalNC = this->n * this->c;
        uint32_t end = this->ncStart + this->ncPerBlock;
        this->ncEnd = end > totalNC ? totalNC : end;
    }

    __aicore__ inline void Process()
    {
        if (out_h == 0 || out_w == 0) return;
        // 仅支持stride>=1，常见情形stride=1
        if (stride == 0) return;

        for (uint32_t ncIdx = ncStart; ncIdx < ncEnd; ++ncIdx) {
            uint32_t nn = ncIdx / c;
            uint32_t cc = ncIdx % c;

            for (uint32_t ohBase = 0; ohBase < out_h; ohBase += tileH) {
                uint32_t curTileH = (ohBase + tileH <= out_h) ? tileH : (out_h - ohBase);
                for (uint32_t owBase = 0; owBase < out_w; owBase += tileW) {
                    uint32_t curTileW = (owBase + tileW <= out_w) ? tileW : (out_w - owBase);

                    // 分配累积与输出tile
                    AscendC::LocalTensor<DTYPE_X> accLocal = accQueue.AllocTensor<DTYPE_X>();
                    AscendC::LocalTensor<DTYPE_Y> outLocal = outQueue.AllocTensor<DTYPE_Y>();

                    // 初始化为负无穷
                    DTYPE_X negInf = (DTYPE_X)NegInfVal<float>();
                    AscendC::Duplicate(accLocal, negInf, curTileH * curTileW);

                    // 对窗口进行堆叠最大化: dy, dx
                    for (uint32_t dy = 0; dy < kernel; ++dy) {
                        // 每个输出行独立处理
                        for (uint32_t tr = 0; tr < curTileH; ++tr) {
                            int32_t inRow = static_cast<int32_t>(ohBase * stride + tr * stride) - static_cast<int32_t>(padding) + static_cast<int32_t>(dy);
                            // 行缓冲
                            AscendC::LocalTensor<DTYPE_X> rowBuf = inRowQueue.AllocTensor<DTYPE_X>();
                            // 先填充为负无穷
                            AscendC::Duplicate(rowBuf, negInf, curTileW * stride + kernel - 1);

                            if (inRow >= 0 && inRow < static_cast<int32_t>(h)) {
                                // 基础列起点（未加dx）：owBase*stride - padding
                                int32_t baseCol = static_cast<int32_t>(owBase * stride) - static_cast<int32_t>(padding);
                                // 拷贝长度
                                int32_t segLen = static_cast<int32_t>(curTileW * stride + kernel - 1);
                                int32_t gmCopyStart = baseCol < 0 ? 0 : baseCol;
                                int32_t gmCopyEnd = baseCol + segLen;
                                if (gmCopyEnd > static_cast<int32_t>(w)) gmCopyEnd = static_cast<int32_t>(w);
                                int32_t validLen = gmCopyEnd - gmCopyStart;
                                if (validLen > 0) {
                                    int32_t localOffset = gmCopyStart - baseCol; // 放到rowBuf中的偏移
                                    // 计算GM偏移：( (n*C + c)*H + inRow )*W + gmCopyStart
                                    int64_t planeOff = ((int64_t)nn * c + cc) * (int64_t)h * (int64_t)w;
                                    int64_t rowOff = (int64_t)inRow * (int64_t)w + (int64_t)gmCopyStart;
                                    int64_t gmOffset = planeOff + rowOff;
                                    AscendC::DataCopy(rowBuf[localOffset], xGm[gmOffset], validLen);
                                }
                            }

                            // 针对当前行，遍历dx叠加最大值
                            // 输出行acc起点偏移
                            int64_t accRowOff = (int64_t)tr * (int64_t)curTileW;
                            // 当stride为1时，rowBuf[dx]是连续子段
                            for (uint32_t dx = 0; dx < kernel; ++dx) {
                                AscendC::LocalTensor<DTYPE_X> srcSeg = rowBuf[dx];
                                AscendC::LocalTensor<DTYPE_X> accSeg = accLocal[accRowOff];
                                // 使用逐元素最大化
                                // 注意：使用Maximum为两输入逐元素最大
                                AscendC::Maximum(accSeg, accSeg, srcSeg, curTileW);
                            }

                            inRowQueue.FreeTensor(rowBuf);
                        } // tr
                    } // dy

                    // 将accLocal写入outLocal（类型可能一致，直接拷贝）
                    AscendC::DataCopy(outLocal, accLocal, curTileH * curTileW);

                    // 回写到GM：按行写出
                    for (uint32_t tr = 0; tr < curTileH; ++tr) {
                        int64_t outRowOffLocal = (int64_t)tr * (int64_t)curTileW;
                        int64_t outRowIdx = (int64_t)ohBase + (int64_t)tr;
                        int64_t outColIdx = (int64_t)owBase;
                        int64_t planeOutOff = ((int64_t)nn * c + cc) * (int64_t)out_h * (int64_t)out_w;
                        int64_t gmOutOffset = planeOutOff + outRowIdx * (int64_t)out_w + outColIdx;
                        AscendC::DataCopy(yGm[gmOutOffset], outLocal[outRowOffLocal], curTileW);
                    }

                    outQueue.FreeTensor(outLocal);
                    accQueue.FreeTensor(accLocal);
                } // owBase
            } // ohBase
        } // ncIdx
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inRowQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> accQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, h, w;
    uint32_t out_h, out_w;
    uint32_t kernel, stride, padding;
    uint32_t ncPerBlock;
    uint32_t tileH, tileW;

    uint32_t blockIdx, blockNum;
    uint32_t ncStart, ncEnd;
};

extern "C" __global__ __aicore__ void max_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(td, tiling);
    KernelMaxPool2D op;
    op.Init(x, y,
            td.n, td.c, td.h, td.w,
            td.out_h, td.out_w,
            td.kernel, td.stride, td.padding,
            td.ncPerBlock, td.tileH, td.tileW);
    op.Process();
}
