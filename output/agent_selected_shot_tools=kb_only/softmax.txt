host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    // 读取输入形状 (假设为二维 [m, k])
    const gert::Shape *x_shape = context->GetInputShape(0);
    const ge::Shape &ori_shape = x_shape->GetOriginShape();
    int64_t dim0 = ori_shape.GetDim(0);
    int64_t dim1 = ori_shape.GetDim(1);
    uint32_t m = static_cast<uint32_t>(dim0);
    uint32_t k = static_cast<uint32_t>(dim1);

    // 读取属性 dim（仅支持最后一维/第二维）
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim_attr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dim_attr != nullptr) ? *dim_attr : 1; // 默认与 PyTorch 中一致

    // 设置并行核数（可根据需要调整）
    const uint32_t BLOCK_DIM = 8;
    context->SetBlockDim(BLOCK_DIM);

    // 计算 SoftMax 需要的最小临时空间，并生成 tiling
    std::vector<int64_t> vec = {dim0, dim1};
    ge::Shape srcShape(vec);
    const uint32_t localWorkSpaceSize = AscendC::GetSoftMaxMinTmpSize(srcShape, sizeof(float), false);
    AscendC::SoftMaxTilingFunc(srcShape, sizeof(float), localWorkSpaceSize, tiling.get_softmaxTilingData());

    // 保存 tiling 数据
    tiling.set_m(m);
    tiling.set_k(k);
    tiling.set_tmpBufferSize(localWorkSpaceSize);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // 无需 workspace
    size_t *workspace_sizes = context->GetWorkspaceSizes(1);
    workspace_sizes[0] = 0;

    context->SetTilingKey(1);
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_Q_DEPTH = 1;
constexpr int32_t OUT_Q_DEPTH = 1;

class KernelSoftmax {
public:
    __aicore__ inline KernelSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t m, uint32_t k, uint32_t tmpBufferSize,
                                const AscendC::SoftMaxTiling &tiling, int64_t dim)
    {
        this->m = m;
        this->k = k;
        this->tmpBufferSize = tmpBufferSize;
        this->tiling = tiling;
        this->dim = dim;

        // 每个核心处理的起始行与行数
        uint32_t block_num = AscendC::GetBlockNum();
        uint32_t block_idx = AscendC::GetBlockIdx();
        uint32_t rows_per_core = m / block_num;
        uint32_t remainder = m % block_num;
        startRow = block_idx * rows_per_core + (block_idx < remainder ? block_idx : remainder);
        rowsThisCore = rows_per_core + (block_idx < remainder ? 1 : 0);

        xGm.SetGlobalBuffer((__gm__ float *)x + startRow * k, rowsThisCore * k);
        yGm.SetGlobalBuffer((__gm__ float *)y + startRow * k, rowsThisCore * k);

        // 队列与缓冲区初始化（按行处理）
        pipe.InitBuffer(inQueue, IN_Q_DEPTH, k * sizeof(float));
        pipe.InitBuffer(outQueue, OUT_Q_DEPTH, k * sizeof(float));
        pipe.InitBuffer(tmpQueue, 1, tmpBufferSize);
    }

    __aicore__ inline void Process()
    {
        // 仅支持最后一维 softmax（dim == 1），与模型一致
        for (uint32_t r = 0; r < rowsThisCore; ++r) {
            CopyIn(r);
            Compute();
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueue.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * k], k);
        inQueue.EnQue<float>(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> xLocal = inQueue.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueue.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> sharedTmpBuffer = tmpQueue.AllocTensor<uint8_t>();

        // SoftMax 形状信息：按行处理，srcM = 1，srcK = k
        AscendC::SoftMaxShapeInfo shapeInfo;
        shapeInfo.srcM = 1;
        shapeInfo.srcK = k;
        shapeInfo.oriSrcM = 1;
        shapeInfo.oriSrcK = k;

        // 使用带 sharedTmpBuffer 的 SoftMax 高级 API
        AscendC::SoftMax<float, false, false>(yLocal, xLocal, sharedTmpBuffer, tiling, shapeInfo);

        outQueue.EnQue<float>(yLocal);
        inQueue.FreeTensor(xLocal);
        tmpQueue.FreeTensor(sharedTmpBuffer);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> yLocal = outQueue.DeQue<float>();
        AscendC::DataCopy(yGm[rowIdx * k], yLocal, k);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, IN_Q_DEPTH> inQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, OUT_Q_DEPTH> outQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t m;
    uint32_t k;
    uint32_t startRow;
    uint32_t rowsThisCore;
    uint32_t tmpBufferSize;
    AscendC::SoftMaxTiling tiling;
    int64_t dim;
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelSoftmax op;
    op.Init(x, y, tilingData.m, tilingData.k, tilingData.tmpBufferSize, tilingData.softmaxTilingData, tilingData.dim);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}
