host_operator_src="""
#include "max_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <vector>

namespace {
static inline void PadTo3(const gert::Shape &src, uint32_t (&dims)[3]) {
    size_t r = src.GetDimNum();
    // right align and pad leading with 1
    int out_idx = 2;
    for (int i = (int)r - 1; i >= 0 && out_idx >= 0; --i, --out_idx) {
        dims[out_idx] = static_cast<uint32_t>(src.GetDim(i));
    }
    for (; out_idx >= 0; --out_idx) {
        dims[out_idx] = 1U;
    }
}

static inline bool BroadcastDim(uint32_t a, uint32_t b, uint32_t &out) {
    if (a == b) { out = a; return true; }
    if (a == 1U) { out = b; return true; }
    if (b == 1U) { out = a; return true; }
    return false;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const gert::Shape* aShape = context->GetInputShape(0);
    const gert::Shape* bShape = context->GetInputShape(1);

    uint32_t aDims[3], bDims[3];
    PadTo3(*aShape, aDims);
    PadTo3(*bShape, bDims);

    uint32_t outDims[3];
    for (int i = 0; i < 3; ++i) {
        uint32_t od = 1;
        if (!BroadcastDim(aDims[i], bDims[i], od)) {
            return ge::GRAPH_FAILED;
        }
        outDims[i] = od;
    }

    uint64_t totalLength64 = static_cast<uint64_t>(outDims[0]) * outDims[1] * outDims[2];
    if (totalLength64 == 0) {
        return ge::GRAPH_FAILED;
    }

    // contiguous strides for inputs (row-major ND: s2=1, s1=d2, s0=d1*d2)
    uint32_t aS2 = 1U, aS1 = aDims[2], aS0 = aDims[1] * aDims[2];
    uint32_t bS2 = 1U, bS1 = bDims[2], bS0 = bDims[1] * bDims[2];

    MaxBroadcastCustomTilingData tiling;
    tiling.set_totalLength(static_cast<uint32_t>(totalLength64));
    tiling.set_outD0(outDims[0]);
    tiling.set_outD1(outDims[1]);
    tiling.set_outD2(outDims[2]);

    tiling.set_aD0(aDims[0]); tiling.set_aD1(aDims[1]); tiling.set_aD2(aDims[2]);
    tiling.set_bD0(bDims[0]); tiling.set_bD1(bDims[1]); tiling.set_bD2(bDims[2]);

    tiling.set_aS0(aS0); tiling.set_aS1(aS1); tiling.set_aS2(aS2);
    tiling.set_bS0(bS0); tiling.set_bS1(bS1); tiling.set_bS2(bS2);

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* aShape = context->GetInputShape(0);
    const gert::Shape* bShape = context->GetInputShape(1);

    uint32_t aDims[3], bDims[3];
    PadTo3(*aShape, aDims);
    PadTo3(*bShape, bDims);

    std::vector<uint64_t> outDims;
    for (int i = 0; i < 3; ++i) {
        uint32_t od = 1;
        if (!BroadcastDim(aDims[i], bDims[i], od)) {
            return GRAPH_FAILED;
        }
        outDims.push_back(od);
    }

    gert::Shape* yShape = context->GetOutputShape(0);
    gert::Shape tmp;
    for (auto d : outDims) { tmp.AppendDim(d); }
    *yShape = tmp;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

#define DTYPE_X float
#define DTYPE_Y float
#define DTYPE_Z float

class KernelMaxBroadcast {
public:
    __aicore__ inline KernelMaxBroadcast() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t totalLength,
                                uint32_t outD0, uint32_t outD1, uint32_t outD2,
                                uint32_t aD0, uint32_t aD1, uint32_t aD2,
                                uint32_t bD0, uint32_t bD1, uint32_t bD2,
                                uint32_t aS0, uint32_t aS1, uint32_t aS2,
                                uint32_t bS0, uint32_t bS1, uint32_t bS2)
    {
        this->totalLength = totalLength;
        this->outD0 = outD0; this->outD1 = outD1; this->outD2 = outD2;
        this->aD0 = aD0; this->aD1 = aD1; this->aD2 = aD2;
        this->bD0 = bD0; this->bD1 = bD1; this->bD2 = bD2;
        this->aS0 = aS0; this->aS1 = aS1; this->aS2 = aS2;
        this->bS0 = bS0; this->bS1 = bS1; this->bS2 = bS2;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, totalLength); // full pointer, we'll index with absolute offsets
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, totalLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z, totalLength);

        // element-wise processing, 1 element per iteration
        pipe.InitBuffer(inQueueX, 1, sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, 1, sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, 1, sizeof(DTYPE_Z));

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->blockStart = (totalLength / blockNum) * blockIdx;
        this->blockEnd = (blockIdx == blockNum - 1) ? totalLength : (totalLength / blockNum) * (blockIdx + 1);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t idx = blockStart; idx < blockEnd; ++idx) {
            CopyInComputeOut(idx);
        }
    }

private:
    __aicore__ inline void CopyInComputeOut(uint32_t absOutIdx)
    {
        // decode (i0,i1,i2) from absOutIdx with out dims
        uint32_t i0 = absOutIdx / (outD1 * outD2);
        uint32_t r  = absOutIdx % (outD1 * outD2);
        uint32_t i1 = r / outD2;
        uint32_t i2 = r % outD2;

        // map to a and b indices considering broadcasting
        uint32_t ax0 = (aD0 == 1U) ? 0U : i0;
        uint32_t ax1 = (aD1 == 1U) ? 0U : i1;
        uint32_t ax2 = (aD2 == 1U) ? 0U : i2;
        uint32_t bx0 = (bD0 == 1U) ? 0U : i0;
        uint32_t bx1 = (bD1 == 1U) ? 0U : i1;
        uint32_t bx2 = (bD2 == 1U) ? 0U : i2;

        uint32_t aIndex = ax0 * aS0 + ax1 * aS1 + ax2 * aS2;
        uint32_t bIndex = bx0 * bS0 + bx1 * bS1 + bx2 * bS2;

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        AscendC::DataCopy(xLocal, xGm[aIndex], 1);
        AscendC::DataCopy(yLocal, yGm[bIndex], 1);

        AscendC::Max(zLocal, xLocal, yLocal, 1);

        AscendC::DataCopy(zGm[absOutIdx], zLocal, 1);

        outQueueZ.FreeTensor(zLocal);
        inQueueY.FreeTensor(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t totalLength;
    uint32_t outD0, outD1, outD2;
    uint32_t aD0, aD1, aD2;
    uint32_t bD0, bD1, bD2;
    uint32_t aS0, aS1, aS2;
    uint32_t bS0, bS1, bS2;

    uint32_t blockStart;
    uint32_t blockEnd;
};

extern "C" __global__ __aicore__ void max_broadcast_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxBroadcast op;
    op.Init(x, y, z,
            tiling_data.totalLength(),
            tiling_data.outD0(), tiling_data.outD1(), tiling_data.outD2(),
            tiling_data.aD0(), tiling_data.aD1(), tiling_data.aD2(),
            tiling_data.bD0(), tiling_data.bD1(), tiling_data.bD2(),
            tiling_data.aS0(), tiling_data.aS1(), tiling_data.aS2(),
            tiling_data.bS0(), tiling_data.bS1(), tiling_data.bS2());
    op.Process();
}
