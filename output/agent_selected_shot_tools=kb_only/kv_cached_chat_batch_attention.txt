kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 1;

class KernelKvCachedChatBatchAttention {
public:
    __aicore__ inline KernelKvCachedChatBatchAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o,
                                GM_ADDR workspace,
                                uint32_t batch, uint32_t qLen, uint32_t kvLen, uint32_t dModel,
                                uint32_t tileD, uint32_t tileKV, float invSqrtD)
    {
        this->batch = batch;
        this->qLen = qLen;
        this->kvLen = kvLen;
        this->dModel = dModel;
        this->tileD = tileD;
        this->tileKV = tileKV;
        this->invSqrtD = invSqrtD;

        uint32_t bIdx = AscendC::GetBlockIdx();
        // Base offsets per batch
        uint64_t qOffset = static_cast<uint64_t>(bIdx) * qLen * dModel;
        uint64_t kOffset = static_cast<uint64_t>(bIdx) * kvLen * dModel;
        uint64_t vOffset = static_cast<uint64_t>(bIdx) * kvLen * dModel;
        uint64_t oOffset = static_cast<uint64_t>(bIdx) * qLen * dModel;

        qGm.SetGlobalBuffer((__gm__ DTYPE*)q + qOffset, qLen * dModel);
        kGm.SetGlobalBuffer((__gm__ DTYPE*)k + kOffset, kvLen * dModel);
        vGm.SetGlobalBuffer((__gm__ DTYPE*)v + vOffset, kvLen * dModel);
        oGm.SetGlobalBuffer((__gm__ DTYPE*)o + oOffset, qLen * dModel);
        scoreWorkspace.SetGlobalBuffer((__gm__ DTYPE*)workspace + static_cast<uint64_t>(bIdx) * kvLen, kvLen);

        // Queues
        pipe.InitBuffer(inQueueQ, BUFFER_NUM, dModel * sizeof(DTYPE));
        pipe.InitBuffer(inQueueK, BUFFER_NUM, dModel * sizeof(DTYPE));
        pipe.InitBuffer(inQueueV, BUFFER_NUM, dModel * sizeof(DTYPE));
        pipe.InitBuffer(outQueueO, BUFFER_NUM, dModel * sizeof(DTYPE));
        pipe.InitBuffer(workQueue, 1, 64 * sizeof(DTYPE)); // temporary workspace for reductions
    }

    __aicore__ inline void Process()
    {
        // Assume qLen == 1 for autoregressive decoding; general qLen supported by looping.
        for (uint32_t qIdx = 0; qIdx < qLen; ++qIdx) {
            ComputeOneQuery(qIdx);
        }
    }

private:
    __aicore__ inline void ComputeOneQuery(uint32_t qIdx)
    {
        // Load Q vector (length dModel)
        AscendC::LocalTensor<DTYPE> qLocal = inQueueQ.AllocTensor<DTYPE>();
        AscendC::DataCopy(qLocal, qGm[qIdx * dModel], dModel);
        inQueueQ.EnQue(qLocal);
        qLocal = inQueueQ.DeQue<DTYPE>();

        // First pass: compute exp(scores) and accumulate sumExp
        DTYPE sumExpScalar = static_cast<DTYPE>(0.0f);

        // Temporary local buffers
        AscendC::LocalTensor<DTYPE> kLocal = inQueueK.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> mulLocal = inQueueK.AllocTensor<DTYPE>(); // reuse queue memory for temp mul
        AscendC::LocalTensor<DTYPE> redWork = workQueue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> scalarBuf = outQueueO.AllocTensor<DTYPE>(); // 1-element buffer used as scalar
        // We will only touch first element of scalarBuf as scalar storage.

        for (uint32_t i = 0; i < kvLen; ++i) {
            // load K[i, :]
            AscendC::DataCopy(kLocal, kGm[i * dModel], dModel);
            // elementwise multiply
            AscendC::Mul(mulLocal, qLocal, kLocal, dModel);
            // reduce sum to scalar
            // ReduceSum(dst, src, workLocal, length, /*is_5d*/ false)
            AscendC::ReduceSum<DTYPE>(scalarBuf, mulLocal, redWork, dModel, false);
            // scale by 1/sqrt(d)
            scalarBuf.SetValue(0, scalarBuf.GetValue(0) * invSqrtD);
            // exp
            AscendC::Exp(scalarBuf, scalarBuf, 1);
            // write to workspace and accumulate sum
            DTYPE e = scalarBuf.GetValue(0);
            scoreWorkspace.SetValue(i, e);
            sumExpScalar += e;
        }

        // Second pass: output = sum_i softmax_i * V[i]
        AscendC::LocalTensor<DTYPE> oLocal = outQueueO.AllocTensor<DTYPE>();
        // zero init
        AscendC::Duplicate(oLocal, static_cast<DTYPE>(0.0f), dModel);

        for (uint32_t i = 0; i < kvLen; ++i) {
            // prob = exp(score_i) / sumExp
            DTYPE prob = scoreWorkspace.GetValue(i) / sumExpScalar;

            // load V[i, :]
            AscendC::LocalTensor<DTYPE> vLocal = inQueueV.AllocTensor<DTYPE>();
            AscendC::DataCopy(vLocal, vGm[i * dModel], dModel);

            // scale V by prob and accumulate
            // Create a local vector filled with prob
            AscendC::LocalTensor<DTYPE> probVec = inQueueV.AllocTensor<DTYPE>();
            AscendC::Duplicate(probVec, prob, dModel);

            AscendC::Mul(probVec, vLocal, probVec, dModel);
            AscendC::Add(oLocal, oLocal, probVec, dModel);

            inQueueV.FreeTensor(vLocal);
            inQueueV.FreeTensor(probVec);
        }

        // write output
        outQueueO.EnQue<DTYPE>(oLocal);
        AscendC::LocalTensor<DTYPE> oWrite = outQueueO.DeQue<DTYPE>();
        AscendC::DataCopy(oGm[qIdx * dModel], oWrite, dModel);
        outQueueO.FreeTensor(oWrite);

        // free
        inQueueQ.FreeTensor(qLocal);
        inQueueK.FreeTensor(kLocal);
        inQueueK.FreeTensor(mulLocal);
        workQueue.FreeTensor(redWork);
        outQueueO.FreeTensor(scalarBuf);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueK;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueV;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueO;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> workQueue;

    AscendC::GlobalTensor<DTYPE> qGm, kGm, vGm, oGm;
    AscendC::GlobalTensor<DTYPE> scoreWorkspace;

    uint32_t batch;
    uint32_t qLen;
    uint32_t kvLen;
    uint32_t dModel;
    uint32_t tileD;
    uint32_t tileKV;
    float invSqrtD;
};

extern "C" __global__ __aicore__ void kv_cached_chat_batch_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelKvCachedChatBatchAttention op;
    op.Init(q, k, v, o,
            workspace,
            tiling_data.batch, tiling_data.qLen, tiling_data.kvLen, tiling_data.dModel,
            tiling_data.tileD, tiling_data.tileKV, tiling_data.invSqrtD);
    op.Process();
}
