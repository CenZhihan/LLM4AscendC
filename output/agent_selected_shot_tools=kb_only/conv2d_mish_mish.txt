kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // queue内并行使用的tensor个数

// 该Kernel对输入逐元素执行：y = mish(mish(x))，其中 mish(x) = x * tanh(softplus(x)), softplus(x)=ln(1+exp(x))
// 注意：此Kernel假设数据类型为float，与project_json一致
class KernelConv2dMishMish {
public:
    __aicore__ inline KernelConv2dMishMish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, uint32_t tmpBufferSize)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        // 每个block处理blockLength，进一步切分为 tileNum * BUFFER_NUM 份
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ float *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ float *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
        // 中间向量(tmp1/tmp2)
        pipe.InitBuffer(tmpCalcQueue1, 1, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpCalcQueue2, 1, this->tileLength * sizeof(float));
        // 高阶数学API的共享临时缓冲
        pipe.InitBuffer(tmpSharedBufQueue, 1, tmpBufferSize);
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    // 计算：y = mish(mish(x))
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        AscendC::LocalTensor<float> tmp1 = tmpCalcQueue1.AllocTensor<float>();    // 存放中间exp/softplus结果
        AscendC::LocalTensor<float> tmp2 = tmpCalcQueue2.AllocTensor<float>();    // 存放log/tanh结果
        AscendC::LocalTensor<uint8_t> sharedTmpBuffer = tmpSharedBufQueue.AllocTensor<uint8_t>();

        // 第一次 Mish：x -> y1
        // tmp1 = exp(x)
        AscendC::Exp<float, 15, false>(tmp1, xLocal, sharedTmpBuffer, this->tileLength);
        // tmp1 = tmp1 + 1.0
        AscendC::Adds(tmp1, tmp1, 1.0f, this->tileLength);
        // tmp2 = log(tmp1)
        AscendC::Log(tmp2, tmp1, sharedTmpBuffer, this->tileLength);
        // tmp2 = tanh(tmp2)
        AscendC::Tanh(tmp2, tmp2, sharedTmpBuffer, this->tileLength);
        // yLocal = xLocal * tmp2
        AscendC::Mul(yLocal, xLocal, tmp2, this->tileLength);

        // 第二次 Mish：y1 -> y
        // tmp1 = exp(yLocal)
        AscendC::Exp<float, 15, false>(tmp1, yLocal, sharedTmpBuffer, this->tileLength);
        // tmp1 = tmp1 + 1.0
        AscendC::Adds(tmp1, tmp1, 1.0f, this->tileLength);
        // tmp2 = log(tmp1)
        AscendC::Log(tmp2, tmp1, sharedTmpBuffer, this->tileLength);
        // tmp2 = tanh(tmp2)
        AscendC::Tanh(tmp2, tmp2, sharedTmpBuffer, this->tileLength);
        // yLocal = yLocal * tmp2
        AscendC::Mul(yLocal, yLocal, tmp2, this->tileLength);

        outQueueY.EnQue<float>(yLocal);

        inQueueX.FreeTensor(xLocal);
        tmpCalcQueue1.FreeTensor(tmp1);
        tmpCalcQueue2.FreeTensor(tmp2);
        tmpSharedBufQueue.FreeTensor(sharedTmpBuffer);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpCalcQueue1;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpCalcQueue2;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSharedBufQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMishMish op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.tmpBufferSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用AscendC自定义算子将连续两次Mish融合为一次kernel调用。
    卷积仍使用PyTorch算子，随后由自定义算子完成 mish(mish(x))。
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        x = self.conv(x)
        x = custom_ops_lib.conv2d_mish_mish_custom(x)
        return x
