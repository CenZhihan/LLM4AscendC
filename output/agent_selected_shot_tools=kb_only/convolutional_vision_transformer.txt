kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的局部tensor数

class KernelCViTCat {
public:
    __aicore__ inline KernelCViTCat() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR cls, GM_ADDR z,
                                uint32_t batch, uint32_t embedDim, uint32_t tileNum)
    {
        this->batch = batch;
        this->embedDim = embedDim;
        this->tileNum = tileNum;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t bStart = (batch * blockIdx) / blocks;
        uint32_t bEnd = (batch * (blockIdx + 1)) / blocks;
        this->batchLocal = bEnd > bStart ? (bEnd - bStart) : 0;

        // 针对E维做分块
        uint32_t segs = this->tileNum * BUFFER_NUM;
        this->tileLen = (embedDim + segs - 1) / segs;
        if (this->tileLen == 0) { this->tileLen = 1; }

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x) + static_cast<int64_t>(bStart) * embedDim,
                            static_cast<int64_t>(this->batchLocal) * embedDim);
        clsGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cls), embedDim);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(z) + static_cast<int64_t>(bStart) * 2 * embedDim,
                            static_cast<int64_t>(this->batchLocal) * 2 * embedDim);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(float));
        pipe.InitBuffer(inQueueC, BUFFER_NUM, this->tileLen * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM * 2, this->tileLen * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (this->batchLocal == 0) { return; }
        uint32_t segs = (embedDim + tileLen - 1) / tileLen;

        for (uint32_t b = 0; b < this->batchLocal; ++b) {
            for (uint32_t i = 0; i < segs; ++i) {
                uint32_t start = i * tileLen;
                if (start >= embedDim) { break; }
                uint32_t len = AscendC::Min(static_cast<uint32_t>(tileLen), static_cast<uint32_t>(embedDim - start));
                CopyIn(b, start, len);
                Compute(len);
                CopyOut(b, start, len);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t bIdx, uint32_t start, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<float> cLocal = inQueueC.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[bIdx * embedDim + start], len);
        AscendC::DataCopy(cLocal, clsGm[start], len);
        inQueueX.EnQue(xLocal);
        inQueueC.EnQue(cLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> cLocal = inQueueC.DeQue<float>();

        AscendC::LocalTensor<float> zLocal0 = outQueueZ.AllocTensor<float>();
        AscendC::LocalTensor<float> zLocal1 = outQueueZ.AllocTensor<float>();

        AscendC::DataCopy(zLocal0, cLocal, len); // 第0个token: cls
        AscendC::DataCopy(zLocal1, xLocal, len); // 第1个token: x

        outQueueZ.EnQue<float>(zLocal0);
        outQueueZ.EnQue<float>(zLocal1);

        inQueueX.FreeTensor(xLocal);
        inQueueC.FreeTensor(cLocal);
    }

    __aicore__ inline void CopyOut(uint32_t bIdx, uint32_t start, uint32_t len)
    {
        AscendC::LocalTensor<float> zLocal0 = outQueueZ.DeQue<float>();
        AscendC::LocalTensor<float> zLocal1 = outQueueZ.DeQue<float>();

        // 输出形状 [B, 2, E]，本block处理的batch起点已在SetGlobalBuffer中平移
        AscendC::DataCopy(zGm[bIdx * 2 * embedDim + 0 * embedDim + start], zLocal0, len);
        AscendC::DataCopy(zGm[bIdx * 2 * embedDim + 1 * embedDim + start], zLocal1, len);

        outQueueZ.FreeTensor(zLocal0);
        outQueueZ.FreeTensor(zLocal1);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueC;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM * 2> outQueueZ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> clsGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t batch;
    uint32_t embedDim;
    uint32_t batchLocal;
    uint32_t tileNum;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void convolutional_vision_transformer_custom(GM_ADDR x, GM_ADDR cls, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCViTCat op;
    op.Init(x, cls, z, tiling_data.batch, tiling_data.embedDim, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import torch.nn as nn

class ModelNew(nn.Module):
    def __init__(self, num_classes, embed_dim=512, num_heads=8, num_layers=6, 
                 mlp_ratio=4.0, patch_size=4, in_channels=3, image_size=32):
        super(ModelNew, self).__init__()

        self.patch_size = patch_size
        self.image_size = image_size
        self.embed_dim = embed_dim

        self.conv1 = nn.Conv2d(in_channels, embed_dim, kernel_size=patch_size, stride=patch_size)
        num_patches = (image_size // patch_size) ** 2
        self.linear_proj = nn.Linear(embed_dim * num_patches, embed_dim)

        self.transformer_layers = nn.ModuleList([
            nn.TransformerEncoderLayer(
                d_model=embed_dim,
                nhead=num_heads,
                dim_feedforward=int(embed_dim * mlp_ratio),
                dropout=0.0,
                batch_first=True
            ) for _ in range(num_layers)
        ])

        self.cls_token = nn.Parameter(torch.zeros(1, 1, embed_dim))
        self.fc_out = nn.Linear(embed_dim, num_classes)

    def forward(self, x):
        B = x.size(0)
        x = self.conv1(x)                  # (B, embed_dim, H/patch_size, W/patch_size)
        x = x.flatten(start_dim=1)         # (B, embed_dim * num_patches)
        x = self.linear_proj(x)            # (B, embed_dim)

        # 使用自定义AscendC算子将cls_token与x拼接为(B, 2, E)
        cls_vec = self.cls_token.view(-1)  # (E)
        x = custom_ops_lib.convolutional_vision_transformer_custom(x, cls_vec)

        for layer in self.transformer_layers:
            x = layer(x)

        return self.fc_out(x[:, 0])
'''
