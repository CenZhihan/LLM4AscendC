#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose2dStub {
public:
    __aicore__ inline KernelConvTranspose2dStub() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cOut, uint32_t hOut, uint32_t wOut,
                                uint32_t hasBias, uint32_t tileNum)
    {
        (void)x; (void)w; (void)b; // not used in this stub implementation
        this->n = n;
        this->cOut = cOut;
        this->hOut = hOut;
        this->wOut = wOut;
        this->hasBias = hasBias;
        this->tileNum = tileNum;

        totalOutLen = static_cast<uint64_t>(n) * cOut * hOut * wOut;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Split by block
        this->blockLength = totalOutLen / blockNum;
        this->blockOffset = this->blockLength * blockIdx;
        // Handle tail by letting last block consume remainder
        if (blockIdx == blockNum - 1) {
            this->blockLength = totalOutLen - this->blockLength * (blockNum - 1);
        }

        yGm.SetGlobalBuffer((__gm__ float*)y + this->blockOffset, this->blockLength);

        // tile length per queue buffer
        this->tileLength = (this->blockLength / tileNum) / BUFFER_NUM;
        if (this->tileLength == 0) {
            this->tileLength = this->blockLength; // degenerate case
        }

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = (this->blockLength + this->tileLength - 1) / this->tileLength;
        for (int32_t i = 0; i < loopCount; ++i) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        uint64_t currLen = this->tileLength;
        // Last tile length adjust
        uint64_t proceeded = static_cast<uint64_t>(tileProgress) * this->tileLength;
        if (proceeded + currLen > this->blockLength) {
            currLen = this->blockLength - proceeded;
        }

        // Fill zeros as a stub
        float zero = 0.0f;
        AscendC::Duplicate(yLocal, zero, currLen);

        // (Optional) In a real implementation, add bias broadcasting if hasBias == 1.
        outQueueY.EnQue<float>(yLocal);
        ++tileProgress;
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        uint64_t proceeded = static_cast<uint64_t>(progress) * this->tileLength;
        uint64_t currLen = this->tileLength;
        if (proceeded + currLen > this->blockLength) {
            currLen = this->blockLength - proceeded;
        }
        AscendC::DataCopy(yGm[proceeded], yLocal, currLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n, cOut, hOut, wOut;
    uint32_t hasBias;
    uint32_t tileNum;
    uint64_t totalOutLen;
    uint64_t blockLength;
    uint64_t blockOffset;
    uint64_t tileLength;
    uint64_t tileProgress = 0;
};

// Kernel entry
extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    (void)workspace;
    GET_TILING_DATA(tilingData, tiling);
    KernelConvTranspose2dStub op;
    op.Init(x, weight, bias, y,
            tilingData.n, tilingData.cOut, tilingData.hOut, tilingData.wOut,
            tilingData.hasBias, tilingData.tileNum);
    op.Process();
}
