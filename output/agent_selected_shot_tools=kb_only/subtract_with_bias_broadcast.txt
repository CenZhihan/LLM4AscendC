kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的张量数

class KernelSubtractWithBiasBroadcast {
public:
    __aicore__ inline KernelSubtractWithBiasBroadcast() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t hLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum(); // 当前block处理的元素总数
        this->hLength = hLength;                                   // 每行长度（最后一维）
        this->rowsPerBlock = this->blockLength / this->hLength;    // 当前block处理的行数
        this->tileNum = tileNum;

        // 以最后一维为切片单位进行分块，保证不跨行，便于广播减法
        // 基础tile长度，尽量均分到tileNum个分块
        this->tileH = (this->hLength + this->tileNum - 1) / this->tileNum;
        if (this->tileH == 0) {
            this->tileH = this->hLength;
        }
        // 实际的tile循环次数（向上取整）
        this->tileLoopCount = (this->hLength + this->tileH - 1) / this->tileH;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias, this->hLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // 队列缓冲按照单个tile的最大长度分配
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileH * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueBias, BUFFER_NUM, this->tileH * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileH * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            for (uint32_t t = 0; t < this->tileLoopCount * BUFFER_NUM; ++t) {
                CopyIn(r, t);
                Compute(t);
                CopyOut(r, t);
            }
        }
    }

private:
    __aicore__ inline uint32_t GetTileOffset(uint32_t progress) const
    {
        uint32_t tileIdx = progress % this->tileLoopCount;
        return tileIdx * this->tileH;
    }

    __aicore__ inline uint32_t GetTileLen(uint32_t progress) const
    {
        uint32_t start = GetTileOffset(progress);
        uint32_t remain = (start < this->hLength) ? (this->hLength - start) : 0;
        uint32_t curLen = (remain > this->tileH) ? this->tileH : remain;
        return curLen;
    }

    __aicore__ inline void CopyIn(uint32_t row, uint32_t progress)
    {
        uint32_t hOffset = GetTileOffset(progress);
        uint32_t curLen = GetTileLen(progress);
        if (curLen == 0) { return; }

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> biasLocal = inQueueBias.AllocTensor<DTYPE_Y>();

        // x的偏移：按行对齐 + 行内偏移
        AscendC::DataCopy(xLocal, xGm[row * this->hLength + hOffset], curLen);
        // bias的偏移：直接取对应的行内段
        AscendC::DataCopy(biasLocal, biasGm[hOffset], curLen);

        inQueueX.EnQue(xLocal);
        inQueueBias.EnQue(biasLocal);
    }

    __aicore__ inline void Compute(uint32_t progress)
    {
        uint32_t curLen = GetTileLen(progress);
        if (curLen == 0) { return; }

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> biasLocal = inQueueBias.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // y = x - bias（按行广播，已按行内切片对齐）
        AscendC::Sub(yLocal, xLocal, biasLocal, curLen);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueBias.FreeTensor(biasLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row, uint32_t progress)
    {
        uint32_t hOffset = GetTileOffset(progress);
        uint32_t curLen = GetTileLen(progress);
        if (curLen == 0) { return; }

        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[row * this->hLength + hOffset], yLocal, curLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBias;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> biasGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t blockLength;
    uint32_t hLength;
    uint32_t rowsPerBlock;
    uint32_t tileNum;
    uint32_t tileH;
    uint32_t tileLoopCount;
};

extern "C" __global__ __aicore__ void subtract_with_bias_broadcast_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSubtractWithBiasBroadcast op;
    op.Init(x, bias, y, tiling_data.totalLength, tiling_data.hLength, tiling_data.tileNum);
    op.Process();
}
