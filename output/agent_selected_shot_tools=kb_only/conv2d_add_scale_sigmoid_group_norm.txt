kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelConv2dAddScaleSigmoidGroupNorm {
public:
    __aicore__ inline KernelConv2dAddScaleSigmoidGroupNorm() {}
    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR biasGm, GM_ADDR scaleGm, GM_ADDR gammaGm, GM_ADDR betaGm,
                                GM_ADDR yGm,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t groups)
    {
        this->n = n; this->c = c; this->h = h; this->w = w; this->groups = groups;
        bshLength = n * c * h * w;
        bsLength = groups * n;

        x_global.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(xGm), bshLength);
        bias_global.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(biasGm), bshLength);
        scale_global.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(scaleGm), bshLength);
        gamma_global.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(gammaGm), c);
        beta_global.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(betaGm), c);
        y_global.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(yGm), bshLength);

        uint32_t hwAlignSize = (sizeof(DTYPE) * h * w + ONE_BLK_SIZE - 1) / ONE_BLK_SIZE * ONE_BLK_SIZE / sizeof(DTYPE);
        this->hwAlignSize = hwAlignSize;

        pipe.InitBuffer(inQueueX, 1, sizeof(DTYPE) * n * c * hwAlignSize);
        pipe.InitBuffer(inQueueBias, 1, sizeof(DTYPE) * n * c * hwAlignSize);
        pipe.InitBuffer(inQueueScale, 1, sizeof(DTYPE) * n * c * hwAlignSize);
        pipe.InitBuffer(inQueueGamma, 1, (sizeof(DTYPE) * c + 31) / 32 * 32);
        pipe.InitBuffer(inQueueBeta, 1, (sizeof(DTYPE) * c + 31) / 32 * 32);
        pipe.InitBuffer(outQueue, 1, sizeof(DTYPE) * n * c * hwAlignSize);
        pipe.InitBuffer(meanBuffer, (sizeof(DTYPE) * groups * n + 31) / 32 * 32);
        pipe.InitBuffer(varBuffer, (sizeof(DTYPE) * groups * n + 31) / 32 * 32);
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        // Allocate locals
        x_local = inQueueX.AllocTensor<DTYPE>();
        bias_local = inQueueBias.AllocTensor<DTYPE>();
        scale_local = inQueueScale.AllocTensor<DTYPE>();
        gamma_local = inQueueGamma.AllocTensor<DTYPE>();
        beta_local = inQueueBeta.AllocTensor<DTYPE>();
        y_local = outQueue.AllocTensor<DTYPE>();
        mean_local = meanBuffer.Get<DTYPE>();
        var_local = varBuffer.Get<DTYPE>();

        // Copy with padding on HW to align to 32B
        AscendC::DataCopyParams xCopyParams{static_cast<uint16_t>(n * c), static_cast<uint16_t>(h * w * sizeof(DTYPE)), 0, 0};
        AscendC::DataCopyPadParams xPad{true, 0, static_cast<uint8_t>(hwAlignSize - h * w), 0};
        AscendC::DataCopyPad(x_local, x_global, xCopyParams, xPad);

        AscendC::DataCopyParams bCopyParams{static_cast<uint16_t>(n * c), static_cast<uint16_t>(h * w * sizeof(DTYPE)), 0, 0};
        AscendC::DataCopyPadParams bPad{true, 0, static_cast<uint8_t>(hwAlignSize - h * w), 0};
        AscendC::DataCopyPad(bias_local, bias_global, bCopyParams, bPad);

        AscendC::DataCopyParams sCopyParams{static_cast<uint16_t>(n * c), static_cast<uint16_t>(h * w * sizeof(DTYPE)), 0, 0};
        AscendC::DataCopyPadParams sPad{true, 0, static_cast<uint8_t>(hwAlignSize - h * w), 0};
        AscendC::DataCopyPad(scale_local, scale_global, sCopyParams, sPad);

        AscendC::DataCopyParams gCopyParams{1, static_cast<uint16_t>(c * sizeof(DTYPE)), 0, 0};
        AscendC::DataCopyPadParams gPad{false, 0, 0, 0};
        AscendC::DataCopyPad(gamma_local, gamma_global, gCopyParams, gPad);
        AscendC::DataCopyPad(beta_local, beta_global, gCopyParams, gPad);

        AscendC::PipeBarrier<PIPE_ALL>();
    }

    __aicore__ inline void Compute()
    {
        uint32_t vecSize = n * c * hwAlignSize;

        // element-wise: (x + bias) * scale
        AscendC::Add(x_local, x_local, bias_local, vecSize);
        AscendC::Mul(x_local, x_local, scale_local, vecSize);

        // Sigmoid(x) element-wise
        AscendC::Sigmoid(x_local, x_local, vecSize);

        // GroupNorm: prepare tiling
        uint32_t stackBufferSize = 0;
        {
            AscendC::LocalTensor<float> stackBuffer;
            AscendC::PopStackBuffer<float, AscendC::TPosition::LCM>(stackBuffer);
            stackBufferSize = stackBuffer.GetSize();
        }
        AscendC::GroupNormTiling groupNormTiling;
        uint32_t inputShape[4] = {n, c, h, w};
        AscendC::ShapeInfo shapeInfo{ (uint8_t)4, inputShape, (uint8_t)4, inputShape, AscendC::DataFormat::ND };
        AscendC::GetGroupNormNDTillingInfo(shapeInfo, stackBufferSize, sizeof(DTYPE), false, groups, groupNormTiling);

        // y_local as output; mean_local and var_local as buffers
        DTYPE epsilon = static_cast<DTYPE>(0.001f);
        AscendC::GroupNorm<DTYPE, false>(y_local, mean_local, var_local, x_local, gamma_local, beta_local, epsilon, groupNormTiling);

        AscendC::PipeBarrier<PIPE_ALL>();
        // Free intermediates not needed further (locals managed by queues/buffers)
        inQueueBias.FreeTensor(bias_local);
        inQueueScale.FreeTensor(scale_local);
    }

    __aicore__ inline void CopyOut()
    {
        // Copy back without HW padding
        AscendC::DataCopyParams yCopyParams{static_cast<uint16_t>(n * c), static_cast<uint16_t>(h * w * sizeof(DTYPE)), 0, 0};
        AscendC::DataCopyPadParams yPad{false, 0, 0, 0};
        AscendC::DataCopyPad(y_global, y_local, yCopyParams, yPad);

        // Free tensors
        inQueueX.FreeTensor(x_local);
        inQueueGamma.FreeTensor(gamma_local);
        inQueueBeta.FreeTensor(beta_local);
        outQueue.FreeTensor(y_local);

        AscendC::PipeBarrier<PIPE_ALL>();
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueBias;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueScale;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueGamma;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueBeta;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue;
    AscendC::TBuf<AscendC::QuePosition::VECCALC>   meanBuffer, varBuffer;

    AscendC::GlobalTensor<DTYPE> x_global, bias_global, scale_global, gamma_global, beta_global, y_global;

    AscendC::LocalTensor<DTYPE> x_local, bias_local, scale_local, gamma_local, beta_local, y_local;
    AscendC::LocalTensor<DTYPE> mean_local, var_local;

    uint32_t n, c, h, w, groups;
    uint32_t bshLength, bsLength;
    uint32_t hwAlignSize;
};

extern "C" __global__ __aicore__ void conv2d_add_scale_sigmoid_group_norm_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv2dAddScaleSigmoidGroupNorm op;
    op.Init(x, bias, scale, gamma, beta, y,
            tilingData.n(), tilingData.c(), tilingData.h(), tilingData.w(), tilingData.groups());
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Replace post-conv add/scale/sigmoid/groupnorm with custom AscendC fused kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, num_groups, bias_shape, scale_shape):
        super().__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.scale = nn.Parameter(torch.randn(scale_shape))
        self.group_norm = nn.GroupNorm(num_groups, out_channels)

    def forward(self, x):
        x = self.conv(x)
        # Broadcast bias and scale to input shape beforehand to simplify kernel broadcasting
        bias_broadcast = self.bias.expand_as(x)
        scale_broadcast = self.scale.expand_as(x)
        y = custom_ops_lib.conv2d_add_scale_sigmoid_group_norm_custom(
            x, bias_broadcast, scale_broadcast, self.group_norm.weight, self.group_norm.bias
        )
        return y
