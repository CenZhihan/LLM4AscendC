#include "inplace_update_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    InplaceUpdateCustomTilingData tiling;
    const auto* x_shape = context->GetInputShape(0);
    const auto* idx_shape = context->GetInputShape(1);
    const auto* val_shape = context->GetInputShape(2);

    // Assume x: [N, D], idx: [K], value: [K, D]
    uint32_t rows = (x_shape->GetOriginShape().GetDimNum() >= 1) ? x_shape->GetOriginShape().GetDim(0) : 0;
    uint32_t cols = (x_shape->GetOriginShape().GetDimNum() >= 2) ? x_shape->GetOriginShape().GetDim(1) : 1;
    uint32_t updates = (idx_shape->GetOriginShape().GetDimNum() >= 1) ? idx_shape->GetOriginShape().GetDim(0) : 0;

    uint32_t totalLength = rows * cols;

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_updates(updates);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {

    const gert::Shape* x_shape = context->GetInputShape(0);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelInplaceUpdate {
public:
    __aicore__ inline KernelInplaceUpdate() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR value, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t rows, uint32_t cols, uint32_t updates)
    {
        this->totalLength = totalLength;
        this->tileNum = tileNum;
        this->rows = rows;
        this->cols = cols;
        this->updates = updates;

        blockNum = AscendC::GetBlockNum();
        blockIdx = AscendC::GetBlockIdx();

        blockCopyLen = totalLength / blockNum;
        startCopyOffset = blockCopyLen * blockIdx;
        tileLen = blockCopyLen / tileNum / BUFFER_NUM;

        // Partition updates across blocks (nearly even)
        uint32_t base = updates / blockNum;
        uint32_t rem = updates % blockNum;
        if (blockIdx < rem) {
            kStart = blockIdx * (base + 1);
            kLen = base + 1;
        } else {
            kStart = rem * (base + 1) + (blockIdx - rem) * base;
            kLen = base;
        }

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x), totalLength);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(y), totalLength);
        idxGm.SetGlobalBuffer(reinterpret_cast<__gm__ int32_t *>(idx), updates);
        valGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(value), updates * cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, tileLen * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileLen * sizeof(float));
        pipe.InitBuffer(idxQueue, 1, (kLen > 0 ? kLen : 1) * sizeof(int32_t));
        pipe.InitBuffer(valRowQueue, 1, cols * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Step 1: copy x -> y for assigned block segment
        int32_t loopCount = tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyInBase(i);
            CopyOutBase(i);
        }

        // Step 2: apply updates to y with indices and values
        if (kLen > 0) {
            ApplyUpdates();
        }
    }

private:
    __aicore__ inline void CopyInBase(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        uint32_t offset = startCopyOffset + progress * tileLen;
        AscendC::DataCopy(xLocal, xGm[offset], tileLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void CopyOutBase(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        AscendC::DataCopy(yLocal, xLocal, tileLen);
        uint32_t offset = startCopyOffset + progress * tileLen;
        AscendC::DataCopy(yGm[offset], yLocal, tileLen);
        outQueueY.EnQue<float>(yLocal);
        inQueueX.FreeTensor(xLocal);
        AscendC::LocalTensor<float> yLocal2 = outQueueY.DeQue<float>();
        outQueueY.FreeTensor(yLocal2);
    }

    __aicore__ inline void ApplyUpdates()
    {
        // Load idx chunk for this block
        AscendC::LocalTensor<int32_t> idxLocal = idxQueue.AllocTensor<int32_t>();
        AscendC::DataCopy(idxLocal, idxGm[kStart], kLen);

        for (uint32_t j = 0; j < kLen; ++j) {
            int32_t rowIdx = idxLocal.GetValue(j);
            if (rowIdx < 0 || rowIdx >= static_cast<int32_t>(rows)) {
                continue; // skip invalid indices
            }
            AscendC::LocalTensor<float> rowLocal = valRowQueue.AllocTensor<float>();
            uint32_t srcOffset = (kStart + j) * cols;
            AscendC::DataCopy(rowLocal, valGm[srcOffset], cols);
            uint32_t dstOffset = static_cast<uint32_t>(rowIdx) * cols;
            AscendC::DataCopy(yGm[dstOffset], rowLocal, cols);
            valRowQueue.FreeTensor(rowLocal);
        }

        idxQueue.FreeTensor(idxLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> idxQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> valRowQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<int32_t> idxGm;
    AscendC::GlobalTensor<float> valGm;

    uint32_t totalLength{0};
    uint32_t tileNum{0};
    uint32_t rows{0};
    uint32_t cols{0};
    uint32_t updates{0};

    uint32_t blockNum{1};
    uint32_t blockIdx{0};
    uint32_t blockCopyLen{0};
    uint32_t startCopyOffset{0};
    uint32_t tileLen{0};

    uint32_t kStart{0};
    uint32_t kLen{0};
};

extern "C" __global__ __aicore__ void inplace_update_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR value,
                                                             GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelInplaceUpdate op;
    op.Init(x, idx, value, y,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.rows, tiling_data.cols, tiling_data.updates);
    op.Process();
}
"""
