using T = float;

class KernelConvtranspose3dMeanAddSoftmaxTanhScalingCustom {
public:
    __aicore__ inline KernelConvtranspose3dMeanAddSoftmaxTanhScalingCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR y,
                                uint32_t m, uint32_t k,
                                const SoftMaxTiling &softmaxTilingData)
    {
        this->m_ = m;
        this->k_ = k;
        this->tiling_ = softmaxTilingData;

        xGm_.SetGlobalBuffer((__gm__ T *)x);
        biasGm_.SetGlobalBuffer((__gm__ T *)bias);
        yGm_.SetGlobalBuffer((__gm__ T *)y);
        scaleGm_.SetGlobalBuffer((__gm__ T *)scale);

        // 队列/缓冲区
        pipe_.InitBuffer(inQueueX_, 1, k_ * sizeof(T));
        pipe_.InitBuffer(biasQueue_, 1, k_ * sizeof(T));
        pipe_.InitBuffer(sumQueue_, 1, elementNumPerBlk_ * sizeof(T));
        pipe_.InitBuffer(maxQueue_, 1, elementNumPerBlk_ * sizeof(T));
        pipe_.InitBuffer(tmpQueue_, 1, k_ * sizeof(T));   // 加偏置后的临时
        pipe_.InitBuffer(outQueueY_, 1, k_ * sizeof(T));  // softmax 行结果
    }

    __aicore__ inline void Process()
    {
        // 复制 bias 一次（NHWC 下为 [1,1,1,C] 展开为长度 k_ 向量）
        AscendC::LocalTensor<T> biasLocal = biasQueue_.AllocTensor<T>();
        AscendC::DataCopy(biasLocal, biasGm_, k_);
        biasQueue_.EnQue<T>(biasLocal);

        // scale 不在核中使用（后续在 Python 侧执行 tanh 和 scaling）
        // 主循环：按行（m_）处理
        for (uint32_t i = 0; i < m_; ++i) {
            CopyInRow(i);
            ComputeRow();
            CopyOutRow(i);
        }

        // 释放 bias
        biasLocal = biasQueue_.DeQue<T>();
        biasQueue_.FreeTensor(biasLocal);
    }

private:
    __aicore__ inline void CopyInRow(uint32_t rowIdx)
    {
        AscendC::LocalTensor<T> xLocal = inQueueX_.AllocTensor<T>();
        AscendC::DataCopy(xLocal, xGm_[rowIdx * k_], k_);
        inQueueX_.EnQue<T>(xLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX_.DeQue<T>();
        AscendC::LocalTensor<T> biasLocal = biasQueue_.DeQue<T>();
        AscendC::LocalTensor<T> tmpLocal = tmpQueue_.AllocTensor<T>();
        AscendC::LocalTensor<T> sumLocal = sumQueue_.AllocTensor<T>();
        AscendC::LocalTensor<T> maxLocal = maxQueue_.AllocTensor<T>();
        AscendC::LocalTensor<T> yLocal = outQueueY_.AllocTensor<T>();

        // 加偏置（按通道，最后一维为 C）
        AscendC::Add(tmpLocal, xLocal, biasLocal, k_);

        // SoftMax：沿最后一维（C）做归一化
        AscendC::SoftMaxShapeInfo shapeInfo = {1, k_, 1, k_};
        AscendC::SoftMax<T>(yLocal, sumLocal, maxLocal, tmpLocal, tiling_, shapeInfo);

        // 入队输出
        outQueueY_.EnQue<T>(yLocal);

        // 释放本次使用的本地张量
        tmpQueue_.FreeTensor(tmpLocal);
        sumQueue_.FreeTensor(sumLocal);
        maxQueue_.FreeTensor(maxLocal);
        inQueueX_.FreeTensor(xLocal);
        // biasLocal 放回队列供后续循环复用
        biasQueue_.EnQue<T>(biasLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t rowIdx)
    {
        AscendC::LocalTensor<T> yLocal = outQueueY_.DeQue<T>();
        AscendC::DataCopy(yGm_[rowIdx * k_], yLocal, k_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> biasQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> sumQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> maxQueue_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueue_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY_;

    AscendC::GlobalTensor<T> xGm_;
    AscendC::GlobalTensor<T> biasGm_;
    AscendC::GlobalTensor<T> yGm_;
    AscendC::GlobalTensor<T> scaleGm_;

    uint32_t m_ = 0;     // 行数 = N*H*W
    uint32_t k_ = 0;     // 通道数 C（最后一维）
    uint32_t elementNumPerBlk_ = 32 / sizeof(T);
    SoftMaxTiling tiling_;
};

extern \"C\" __global__ __aicore__ void convtranspose3d_mean_add_softmax_tanh_scaling_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版本：
    - 保留 ConvTranspose3d 由 PyTorch 执行
    - 将 mean(沿深度) 后的特征排列为 NHWC
    - 使用自定义 AscendC 算子完成：按通道加偏置 + softmax（沿通道维）
    - 随后在 PyTorch 侧执行 tanh 与缩放
    - 最终恢复到原始的 NCDHW 形状（深度为 1）
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.bias = torch.nn.Parameter(torch.randn(1, out_channels, 1, 1, 1))
        self.scaling_factor = scaling_factor

    def forward(self, x):
        # 1) 反卷积
        x = self.conv_transpose(x)                  # (B, C, D, H, W)
        # 2) 深度均值（去掉 D 维）
        x = x.mean(dim=2, keepdim=False)           # (B, C, H, W)

        # 准备 NHWC
        x_nhwc = x.permute(0, 2, 3, 1).contiguous()    # (B, H, W, C)

        # 准备 bias: (1, C, 1, 1, 1) -> (1, C, 1, 1) -> (1, 1, 1, C)
        bias_4d = self.bias.squeeze(2).squeeze(2).squeeze(2)  # (1, C, 1, 1)
        bias_nhwc = bias_4d.permute(0, 2, 3, 1).contiguous()   # (1, 1, 1, C)

        # scale 作为 1 元素张量传入
        scale_tensor = torch.tensor([self.scaling_factor], dtype=x_nhwc.dtype, device=x_nhwc.device)

        # 3)+4) 自定义核：bias add + softmax（沿通道：最后一维）
        y_nhwc = custom_ops_lib.convtranspose3d_mean_add_softmax_tanh_scaling_custom(x_nhwc, bias_nhwc, scale_tensor)

        # 5) tanh
        y_nhwc = torch.tanh(y_nhwc)
        # 6) scaling
        y_nhwc = y_nhwc * self.scaling_factor

        # 恢复到 NCHW 并补回深度维 = 1
        y = y_nhwc.permute(0, 3, 1, 2).contiguous()     # (B, C, H, W)
        y = y.unsqueeze(2)                              # (B, C, 1, H, W)
        return y
'''
