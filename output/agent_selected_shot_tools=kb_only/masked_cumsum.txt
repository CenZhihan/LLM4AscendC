kernel_src="""
#include "kernel_operator.h"

template <typename T>
class KernelMaskedCumsum {
public:
    __aicore__ inline KernelMaskedCumsum() {}
    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR mGm, GM_ADDR zGm, uint32_t outterVal, uint32_t innerVal)
    {
        outter = outterVal;
        inner  = innerVal;

        xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(xGm), outter * inner);
        mGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(mGm), outter * inner);
        zGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(zGm), outter * inner);

        pipe.InitBuffer(inQueueX, 1, outter * inner * sizeof(T));
        pipe.InitBuffer(inQueueM, 1, outter * inner * sizeof(T));
        pipe.InitBuffer(workQueue, 1, outter * inner * sizeof(T));
        pipe.InitBuffer(outQueueZ, 1, outter * inner * sizeof(T));
        pipe.InitBuffer(lastRowQueue, 1, inner * sizeof(T));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX.AllocTensor<T>();
        AscendC::LocalTensor<T> mLocal = inQueueM.AllocTensor<T>();
        AscendC::DataCopy(xLocal, xGlobal, outter * inner);
        AscendC::DataCopy(mLocal, mGlobal, outter * inner);
        inQueueX.EnQue(xLocal);
        inQueueM.EnQue(mLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX.DeQue<T>();
        AscendC::LocalTensor<T> mLocal = inQueueM.DeQue<T>();

        AscendC::LocalTensor<T> prodLocal = workQueue.AllocTensor<T>();
        AscendC::Mul(prodLocal, xLocal, mLocal, static_cast<int32_t>(outter * inner));

        AscendC::LocalTensor<T> zLocal = outQueueZ.AllocTensor<T>();
        AscendC::LocalTensor<T> lastRowLocal = lastRowQueue.AllocTensor<T>();

        static constexpr AscendC::CumSumConfig cumSumConfig{true, false, false}; // 沿最后一维累计
        const AscendC::CumSumInfo info{outter, inner};
        AscendC::CumSum<T, cumSumConfig>(zLocal, lastRowLocal, prodLocal, info);

        outQueueZ.EnQue<T>(zLocal);

        workQueue.FreeTensor(prodLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueM.FreeTensor(mLocal);
        lastRowQueue.FreeTensor(lastRowLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<T> zLocal = outQueueZ.DeQue<T>();
        AscendC::DataCopy(zGlobal, zLocal, outter * inner);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueM;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> workQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueZ;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> lastRowQueue;

    AscendC::GlobalTensor<T> xGlobal;
    AscendC::GlobalTensor<T> mGlobal;
    AscendC::GlobalTensor<T> zGlobal;

    uint32_t outter{1};
    uint32_t inner{1};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsum<DTYPE_X> op;
    op.Init(x, mask, z, tiling_data.outter, tiling_data.inner);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, dim):
        super().__init__()
        # 本实现固定沿最后一维进行累计（需 dim == -1 或与输入为二维时的最后一维一致）
        self.dim = dim

    def forward(self, x, mask):
        # 将布尔 mask 转为与 x 相同的数据类型（0/1），符合自定义算子输入约定
        mask_f = mask.to(dtype=x.dtype)
        return custom_ops_lib.masked_cumsum_custom(x, mask_f)
