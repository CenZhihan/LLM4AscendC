kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelBiasAddAfterPoolScaling {
public:
    __aicore__ inline KernelBiasAddAfterPoolScaling() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias_full, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias_full + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(bLocal, biasGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::Add(zLocal, xLocal, bLocal, this->tileLength);
        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> biasGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_scaling_avg_pool_bias_add_scaling_custom(GM_ADDR x, GM_ADDR bias_scaled_full, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {

}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep ConvTranspose3d and AvgPool3d in PyTorch,
    fuse trailing bias add with both scalings via a custom AscendC kernel.
    The fused kernel performs: out = x_scaled + bias_scaled_full,
    where x_scaled = x * (scale1 * scale2), bias_scaled_full = (bias * scale2) broadcasted to x shape.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, scale1, scale2, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.scale1 = torch.nn.Parameter(torch.tensor(scale1, dtype=torch.float32))
        self.avg_pool = torch.nn.AvgPool3d(kernel_size=2)
        # bias parameter: shape [C,1,1,1]
        self.bias = torch.nn.Parameter(torch.randn(bias_shape, dtype=torch.float32))
        self.scale2 = torch.nn.Parameter(torch.tensor(scale2, dtype=torch.float32))

    def forward(self, x):
        # ConvTranspose3d
        x = self.conv_transpose(x)
        # AvgPool3d
        x = self.avg_pool(x)
        # Pre-scale in PyTorch to keep kernel simple and fully vectorized
        scale12 = (self.scale1 * self.scale2).to(dtype=x.dtype, device=x.device)
        x_scaled = x * scale12
        # Prepare broadcasted bias scaled by scale2
        bias_scaled = (self.bias * self.scale2).to(dtype=x.dtype, device=x.device)
        # bias_scaled: [C,1,1,1] -> broadcast to [N,C,D,H,W]
        bias_scaled_full = bias_scaled.view(1, bias_scaled.shape[0], 1, 1, 1).expand_as(x_scaled).contiguous()
        # Custom fused bias add
        out = custom_ops_lib.conv_transpose3d_scaling_avg_pool_bias_add_scaling_custom(x_scaled, bias_scaled_full)
        return out
'''
