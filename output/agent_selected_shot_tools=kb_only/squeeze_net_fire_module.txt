kernel_src="""
#include "kernel_operator.h"

using T = float;

struct FireTilingData {
  uint32_t n;
  uint32_t cin;
  uint32_t h;
  uint32_t w;
  uint32_t cs;
  uint32_t ce1;
  uint32_t ce3;
};

class KernelFire {
public:
  __aicore__ inline KernelFire() {}

  __aicore__ inline void Init(GM_ADDR x,
                              GM_ADDR w_s, GM_ADDR b_s,
                              GM_ADDR w_e1, GM_ADDR b_e1,
                              GM_ADDR w_e3, GM_ADDR b_e3,
                              GM_ADDR y,
                              GM_ADDR workspace,
                              const FireTilingData &t) {
    xGm = reinterpret_cast<__gm__ T*>(x);
    wSqueezeGm = reinterpret_cast<__gm__ T*>(w_s);
    bSqueezeGm = reinterpret_cast<__gm__ T*>(b_s);
    wE1Gm = reinterpret_cast<__gm__ T*>(w_e1);
    bE1Gm = reinterpret_cast<__gm__ T*>(b_e1);
    wE3Gm = reinterpret_cast<__gm__ T*>(w_e3);
    bE3Gm = reinterpret_cast<__gm__ T*>(b_e3);
    yGm = reinterpret_cast<__gm__ T*>(y);
    uGm = reinterpret_cast<__gm__ T*>(workspace);

    n = t.n; cin = t.cin; h = t.h; w = t.w;
    cs = t.cs; ce1 = t.ce1; ce3 = t.ce3;
  }

  __aicore__ inline void Process() {
    // 先计算 squeeze: U [n, cs, h, w]
    ComputeSqueeze();
    // expand1x1 + expand3x3 + ReLU 并拼接到 y
    ComputeExpandsAndConcat();
  }

private:
  __aicore__ inline void ComputeSqueeze() {
    // wSqueeze: [cs, cin], bSqueeze: [cs]
    for (uint32_t ni = 0; ni < n; ++ni) {
      for (uint32_t hi = 0; hi < h; ++hi) {
        for (uint32_t wi = 0; wi < w; ++wi) {
          // 计算每个 squeeze 通道
          for (uint32_t cso = 0; cso < cs; ++cso) {
            float acc = bSqueezeGm[cso];
            const __gm__ T* wRow = wSqueezeGm + static_cast<size_t>(cso) * cin;
            for (uint32_t ci = 0; ci < cin; ++ci) {
              size_t xIndex = (((size_t)ni * cin + ci) * h + hi) * w + wi;
              acc += wRow[ci] * xGm[xIndex];
            }
            // ReLU
            if (acc < 0.0f) acc = 0.0f;
            size_t uIndex = (((size_t)ni * cs + cso) * h + hi) * w + wi;
            uGm[uIndex] = acc;
          }
        }
      }
    }
  }

  __aicore__ inline void ComputeExpandsAndConcat() {
    const uint32_t cout = ce1 + ce3;
    for (uint32_t ni = 0; ni < n; ++ni) {
      for (uint32_t hi = 0; hi < h; ++hi) {
        for (uint32_t wi = 0; wi < w; ++wi) {
          // expand 1x1: y[:ce1, :, :, :]
          for (uint32_t co = 0; co < ce1; ++co) {
            float acc = bE1Gm[co];
            const __gm__ T* wRow = wE1Gm + static_cast<size_t>(co) * cs;
            for (uint32_t csi = 0; csi < cs; ++csi) {
              size_t uIndex = (((size_t)ni * cs + csi) * h + hi) * w + wi;
              acc += wRow[csi] * uGm[uIndex];
            }
            if (acc < 0.0f) acc = 0.0f;
            size_t yIndex = (((size_t)ni * cout + co) * h + hi) * w + wi;
            yGm[yIndex] = acc;
          }
          // expand 3x3: y[ce1:ce1+ce3, :, :, :]，padding=1
          for (uint32_t co = 0; co < ce3; ++co) {
            float acc = bE3Gm[co];
            // 权重布局: [ce3, cs, 3, 3]
            for (uint32_t csi = 0; csi < cs; ++csi) {
              for (int32_t ky = -1; ky <= 1; ++ky) {
                int32_t ih = static_cast<int32_t>(hi) + ky;
                if (ih < 0 || ih >= static_cast<int32_t>(h)) {
                  continue;
                }
                for (int32_t kx = -1; kx <= 1; ++kx) {
                  int32_t iw = static_cast<int32_t>(wi) + kx;
                  if (iw < 0 || iw >= static_cast<int32_t>(w)) {
                    continue;
                  }
                  size_t uIndex = (((size_t)ni * cs + csi) * h + (uint32_t)ih) * w + (uint32_t)iw;
                  uint32_t kyIdx = (uint32_t)(ky + 1);
                  uint32_t kxIdx = (uint32_t)(kx + 1);
                  size_t wIndex = ((((size_t)co * cs + csi) * 3 + kyIdx) * 3 + kxIdx);
                  acc += wE3Gm[wIndex] * uGm[uIndex];
                }
              }
            }
            if (acc < 0.0f) acc = 0.0f;
            size_t yIndex = (((size_t)ni * cout + (ce1 + co)) * h + hi) * w + wi;
            yGm[yIndex] = acc;
          }
        }
      }
    }
  }

private:
  __gm__ T* xGm;
  __gm__ T* wSqueezeGm;
  __gm__ T* bSqueezeGm;
  __gm__ T* wE1Gm;
  __gm__ T* bE1Gm;
  __gm__ T* wE3Gm;
  __gm__ T* bE3Gm;
  __gm__ T* yGm;
  __gm__ T* uGm;

  uint32_t n, cin, h, w, cs, ce1, ce3;
};

extern "C" __global__ __aicore__ void squeeze_net_fire_module_custom(
    GM_ADDR x,
    GM_ADDR w_s, GM_ADDR b_s,
    GM_ADDR w_e1, GM_ADDR b_e1,
    GM_ADDR w_e3, GM_ADDR b_e3,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling) {

  GET_TILING_DATA(tiling_data_raw, tiling);
  FireTilingData t;
  // 与 host_tiling 中的结构字段对应
  t.n   = tiling_data_raw->n;
  t.cin = tiling_data_raw->cin;
  t.h   = tiling_data_raw->h;
  t.w   = tiling_data_raw->w;
  t.cs  = tiling_data_raw->cs;
  t.ce1 = tiling_data_raw->ce1;
  t.ce3 = tiling_data_raw->ce3;

  KernelFire op;
  op.Init(x, w_s, b_s, w_e1, b_e1, w_e3, b_e3, y, workspace, t);
  op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, squeeze_channels, expand1x1_channels, expand3x3_channels):
        super().__init__()
        self.squeeze = torch.nn.Conv2d(in_channels, squeeze_channels, kernel_size=1)
        self.squeeze_activation = torch.nn.ReLU(inplace=True)
        self.expand1x1 = torch.nn.Conv2d(squeeze_channels, expand1x1_channels, kernel_size=1)
        self.expand1x1_activation = torch.nn.ReLU(inplace=True)
        self.expand3x3 = torch.nn.Conv2d(squeeze_channels, expand3x3_channels, kernel_size=3, padding=1)
        self.expand3x3_activation = torch.nn.ReLU(inplace=True)

    def forward(self, x):
        # 使用自定义 AscendC 内核一次性完成 squeeze + ReLU, expand1x1 + ReLU, expand3x3 + ReLU 并拼接
        # 直接传入当前卷积层的权重与偏置
        return custom_ops_lib.squeeze_net_fire_module_custom(
            x,
            self.squeeze.weight, self.squeeze.bias,
            self.expand1x1.weight, self.expand1x1.bias,
            self.expand3x3.weight, self.expand3x3.bias
        )
