kernel_src="""
#include "kernel_operator.h"
#include <cstdint>

constexpr int32_t BUFFER_NUM = 1; // 简化为单缓冲

// 仅支持 stride=1, padding=0, dilation=1, output_padding=0, groups=1 的通用转置卷积（deconv）参考实现
class KernelDeconv2D {
public:
    __aicore__ inline KernelDeconv2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t kH, uint32_t kW,
                                uint32_t Hout, uint32_t Wout,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->N = N;
        this->Cin = Cin;
        this->Cout = Cout;
        this->H = H;
        this->W = W;
        this->kH = kH;
        this->kW = kW;
        this->Hout = Hout;
        this->Wout = Wout;
        this->totalLength = totalLength;

        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum > 0 ? tileNum : 1;
        this->tileLength = (this->blockLength / this->tileNum);
        if (this->tileLength == 0) {
            this->tileLength = this->blockLength;
        }

        xGmPtr = reinterpret_cast<__gm__ DTYPE_X *>(x);
        wGmPtr = reinterpret_cast<__gm__ DTYPE_Y *>(w);
        yGmPtr = reinterpret_cast<__gm__ DTYPE_Z *>(y);

        // 输出分块（每个block负责一段线性 y 上的区间）
        this->blockStart = this->blockLength * AscendC::GetBlockIdx();
        if (AscendC::GetBlockIdx() == (AscendC::GetBlockNum() - 1)) {
            // 最后一个block覆盖到末尾
            this->blockLength = totalLength - this->blockStart;
        }

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint32_t loops = (this->blockLength + this->tileLength - 1) / this->tileLength;
        for (uint32_t it = 0; it < loops; ++it) {
            uint32_t curTileBase = this->blockStart + it * this->tileLength;
            uint32_t curTileSize = AscendC::Min(this->tileLength, this->blockLength - it * this->tileLength);
            ComputeTile(curTileBase, curTileSize);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t tileBase, uint32_t tileSize)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // 逐元素计算该 tile 的输出（朴素实现）
        for (uint32_t idx = 0; idx < tileSize; ++idx) {
            uint32_t outLinear = tileBase + idx;

            // 线性 index -> (n, oc, oh, ow)
            uint32_t ow = outLinear % Wout;
            uint32_t tmp1 = outLinear / Wout;
            uint32_t oh = tmp1 % Hout;
            uint32_t tmp2 = tmp1 / Hout;
            uint32_t oc = tmp2 % Cout;
            uint32_t n  = tmp2 / Cout;

            float acc = 0.f;

            // deconv：y[n, oc, oh, ow] = sum_{ic, kh, kw} x[n, ic, ih, iw] * w[ic, oc, kh, kw]
            // 其中 ih = oh - (kH - 1 - kh), iw = ow - (kW - 1 - kw)
            for (uint32_t ic = 0; ic < Cin; ++ic) {
                for (uint32_t kh = 0; kh < kH; ++kh) {
                    int32_t ih = static_cast<int32_t>(oh) - static_cast<int32_t>(kH - 1 - kh);
                    if (ih < 0 || ih >= static_cast<int32_t>(H)) continue;
                    for (uint32_t kw = 0; kw < kW; ++kw) {
                        int32_t iw = static_cast<int32_t>(ow) - static_cast<int32_t>(kW - 1 - kw);
                        if (iw < 0 || iw >= static_cast<int32_t>(W)) continue;

                        // x index: [N, Cin, H, W]
                        uint32_t xIdx = (((n * Cin + ic) * H + static_cast<uint32_t>(ih)) * W + static_cast<uint32_t>(iw));
                        // w index: [Cin, Cout, kH, kW]
                        uint32_t wIdx = (((ic * Cout + oc) * kH + kh) * kW + kw);

                        float xv = static_cast<float>(xGmPtr[xIdx]);
                        float wv = static_cast<float>(wGmPtr[wIdx]);
                        acc += xv * wv;
                    }
                }
            }

            // 写入本地 y 缓冲
            // 这里直接使用向量方式写入单值到对应位置：利用 DataCopy 将标量写入（长度为1）
            // 为简化，先放入本地缓冲，再整体写回
            // 访问 LocalTensor 的线性地址（0-based）
            // 将 acc 写入到 yLocal[idx]
            // 由于 AscendC 基础示例未给出直接标量赋值API，这里通过一个1长度的临时缓冲规避
            // 但出于示例简化，我们假设可以通过 reinterpret_cast 写入
            ((__ubuf__ DTYPE_Z*)yLocal.GetBuf())[idx] = static_cast<DTYPE_Z>(acc);
        }

        // 将本地 tile 写回 GM
        AscendC::DataCopy(yGmPtr + tileBase, yLocal, tileSize);
        outQueueY.EnQue<DTYPE_Z>(yLocal);
        AscendC::LocalTensor<DTYPE_Z> yLocalOut = outQueueY.DeQue<DTYPE_Z>();
        outQueueY.FreeTensor(yLocalOut);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    __gm__ DTYPE_X *xGmPtr;
    __gm__ DTYPE_Y *wGmPtr;
    __gm__ DTYPE_Z *yGmPtr;

    uint32_t N, Cin, Cout, H, W, kH, kW, Hout, Wout;
    uint32_t totalLength;

    uint32_t blockStart;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    KernelDeconv2D op;
    op.Init(x, w, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.kH, tiling_data.kW,
            tiling_data.Hout, tiling_data.Wout,
            tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
