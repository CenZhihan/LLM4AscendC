inline uint32_t GetDim(const gert::Shape* shp, uint32_t idx, uint32_t default_val=1) {
    if (shp == nullptr) return default_val;
    auto &os = shp->GetOriginShape();
    if (idx < os.GetDimNum()) {
        return static_cast<uint32_t>(os.GetDim(idx));
    }
    return default_val;
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_TOKENS = 16;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    SparseAttentionCustomTilingData tiling;

    // q/k/v assumed shape: [B, S, H, Dh]
    const gert::Shape* q_shape = context->GetInputShape(0);
    uint32_t batch   = GetDim(q_shape, 0, 1);
    uint32_t seqLen  = GetDim(q_shape, 1, 1);
    uint32_t numHeads= GetDim(q_shape, 2, 1);
    uint32_t headDim = GetDim(q_shape, 3, 1);

    // window comes as a 1-element tensor; we pass it via GM and also store in tiling's window with a default fallback
    uint32_t window = 0;
    // fallback: set to 32 if runtime window is not available on host
    window = 32;

    uint32_t totalTokens = batch * seqLen * numHeads;
    uint32_t blockTokens = (totalTokens + BLOCK_DIM - 1) / BLOCK_DIM;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_batch(batch);
    tiling.set_seqLen(seqLen);
    tiling.set_numHeads(numHeads);
    tiling.set_headDim(headDim);
    tiling.set_window(window);
    tiling.set_totalTokens(totalTokens);
    tiling.set_blockTokens(blockTokens);
    tiling.set_tileTokens(TILE_TOKENS);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

class KernelSparseAttention {
public:
    __aicore__ inline KernelSparseAttention() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR out,
                                uint32_t batch, uint32_t seqLen, uint32_t numHeads, uint32_t headDim,
                                uint32_t window, uint32_t blockTokens, uint32_t tileTokens)
    {
        this->batch = batch;
        this->seqLen = seqLen;
        this->numHeads = numHeads;
        this->headDim = headDim;
        this->window = window;
        this->blockTokens = blockTokens;
        this->tileTokens = tileTokens;

        uint32_t totalLength = batch * seqLen * numHeads * headDim;

        qGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(q), totalLength);
        kGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(k), totalLength);
        vGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(v), totalLength);
        outGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(out), totalLength);

        // Each token processes headDim elements for accumulation
        pipe.InitBuffer(bufQue, BUFFER_NUM, headDim * sizeof(DTYPE));
        pipe.InitBuffer(accQue, BUFFER_NUM, headDim * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // Sliding-window aggregation over V as a sparse attention core (demo implementation)
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startToken = blockIdx * blockTokens;
        uint32_t endToken = AscendC::MIN(startToken + blockTokens, batch * seqLen * numHeads);

        for (uint32_t t = startToken; t < endToken; ++t) {
            CopyComputeStoreToken(t);
        }
    }

private:
    __aicore__ inline void CopyComputeStoreToken(uint32_t tokenLinearIdx)
    {
        // Decode (b, s, h) from linear token index
        uint32_t b = tokenLinearIdx / (seqLen * numHeads);
        uint32_t rem = tokenLinearIdx % (seqLen * numHeads);
        uint32_t s = rem / numHeads;
        uint32_t h = rem % numHeads;

        int32_t s_start = static_cast<int32_t>(s) - static_cast<int32_t>(window);
        int32_t s_end   = static_cast<int32_t>(s) + static_cast<int32_t>(window) + 1;
        if (s_start < 0) s_start = 0;
        if (s_end > static_cast<int32_t>(seqLen)) s_end = static_cast<int32_t>(seqLen);
        uint32_t validLen = static_cast<uint32_t>(s_end - s_start);
        if (validLen == 0) {
            // Directly zero output if no window
            ZeroOut(b, s, h);
            return;
        }

        // Accumulator for V aggregation
        AscendC::LocalTensor<DTYPE> accLocal = accQue.AllocTensor<DTYPE>();
        // Initialize accumulator to zero
        AscendC::Duplicate(accLocal, static_cast<DTYPE>(0.0f), headDim);

        // Sum V within [s_start, s_end)
        for (int32_t sj = s_start; sj < s_end; ++sj) {
            AscendC::LocalTensor<DTYPE> tmpLocal = bufQue.AllocTensor<DTYPE>();
            uint64_t offV = GetOffset(b, static_cast<uint32_t>(sj), h) * headDim;
            AscendC::DataCopy(tmpLocal, vGm[offV], headDim);
            AscendC::Add(accLocal, accLocal, tmpLocal, headDim);
            bufQue.FreeTensor(tmpLocal);
        }

        // Average: accLocal = accLocal / validLen
        DTYPE inv = static_cast<DTYPE>(1.0f) / static_cast<DTYPE>(validLen);
        AscendC::Muls(accLocal, accLocal, inv, headDim);

        // Store to out
        uint64_t offOut = GetOffset(b, s, h) * headDim;
        AscendC::DataCopy(outGm[offOut], accLocal, headDim);
        accQue.FreeTensor(accLocal);
    }

    __aicore__ inline void ZeroOut(uint32_t b, uint32_t s, uint32_t h)
    {
        AscendC::LocalTensor<DTYPE> tmpLocal = bufQue.AllocTensor<DTYPE>();
        AscendC::Duplicate(tmpLocal, static_cast<DTYPE>(0.0f), headDim);
        uint64_t off = GetOffset(b, s, h) * headDim;
        AscendC::DataCopy(outGm[off], tmpLocal, headDim);
        bufQue.FreeTensor(tmpLocal);
    }

    __aicore__ inline uint64_t GetOffset(uint32_t b, uint32_t s, uint32_t h)
    {
        // Layout: [B, S, H, Dh] contiguous ND
        return (((static_cast<uint64_t>(b) * seqLen + s) * numHeads + h));
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bufQue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> accQue;
    AscendC::GlobalTensor<DTYPE> qGm;
    AscendC::GlobalTensor<DTYPE> kGm;
    AscendC::GlobalTensor<DTYPE> vGm;
    AscendC::GlobalTensor<DTYPE> outGm;

    uint32_t batch;
    uint32_t seqLen;
    uint32_t numHeads;
    uint32_t headDim;
    uint32_t window;
    uint32_t blockTokens;
    uint32_t tileTokens;
};

extern "C" __global__ __aicore__ void sparse_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    // Read runtime window from GM if provided; tiling_data.window is a fallback
    // Here we simply use tiling_data.window as the effective window.
    KernelSparseAttention op;
    op.Init(q, k, v, out,
            tiling_data.batch,
            tiling_data.seqLen,
            tiling_data.numHeads,
            tiling_data.headDim,
            tiling_data.window,
            tiling_data.blockTokens,
            tiling_data.tileTokens);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    使用自定义AscendC核实现稀疏注意力（滑动窗口聚合V的演示实现）。
    注意：此实现以演示自定义核替换PyTorch算子为目的，将注意力核心替换为窗口内V的聚合。
    """
    def __init__(self, d_model=256, num_heads=4, window_size=32):
        super().__init__()
        self.d_model = d_model
        self.num_heads = num_heads
        self.window_size = window_size
        assert d_model % num_heads == 0
        head_dim = d_model // num_heads
        self.q_proj = nn.Linear(d_model, d_model, bias=False)
        self.k_proj = nn.Linear(d_model, d_model, bias=False)
        self.v_proj = nn.Linear(d_model, d_model, bias=False)
        self.out_proj = nn.Linear(d_model, d_model, bias=False)

    def forward(self, x):
        # x: [B, S, D]
        B, S, D = x.shape
        H = self.num_heads
        Dh = D // H

        q = self.q_proj(x).reshape(B, S, H, Dh).contiguous()
        k = self.k_proj(x).reshape(B, S, H, Dh).contiguous()
        v = self.v_proj(x).reshape(B, S, H, Dh).contiguous()

        window = torch.tensor([self.window_size], dtype=torch.int32, device=x.device)
        # 自定义核：期望 q/k/v 形状 [B, S, H, Dh]，window 为1元素int32张量
        out_ctx = custom_ops_lib.sparse_attention_custom(q, k, v, window)
        out = out_ctx.reshape(B, S, D)
        out = self.out_proj(out)
        return out
'''
