kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // use single-buffered queues for simplicity

class KernelConvTransposed1dDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed1dDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR params, GM_ADDR dims, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->totalLength = totalLength;
        this->tileNum = tileNum;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileLength = (this->blockLength / tileNum);
        // shift base pointer for current block
        xGm.SetGlobalBuffer((__gm__ float *)x, 0xFFFFFFFF); // x read-only, access by computed offsets
        wGm.SetGlobalBuffer((__gm__ float *)w, 0xFFFFFFFF);
        bGm.SetGlobalBuffer((__gm__ float *)b, 0xFFFFFFFF);
        paramsGm.SetGlobalBuffer((__gm__ int32_t *)params, 3);
        dimsGm.SetGlobalBuffer((__gm__ int32_t *)dims, 5);
        yGm.SetGlobalBuffer((__gm__ float *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // allocate buffers
        // weight and bias copied once
        pipe.InitBuffer(inQueueW, 1, sizeof(float) * MAX_W_SIZE);
        pipe.InitBuffer(inQueueB, 1, sizeof(float) * MAX_B_SIZE);
        pipe.InitBuffer(inQueueParams, 1, sizeof(int32_t) * 3);
        pipe.InitBuffer(inQueueDims, 1, sizeof(int32_t) * 5);
        pipe.InitBuffer(outQueueY, BUFFER_NUM, sizeof(float) * this->tileLength);
    }

    __aicore__ inline void Process()
    {
        // copy dims and params, weight, bias to UB once
        CopyMetaAndWeights();
        // split current block into tiles
        int32_t loopCount = 1 * BUFFER_NUM * tileNum;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyOutTileInit(i);
            Compute(i);
            CopyOut(i);
        }
        FreeMetaAndWeights();
    }

private:
    // maximum bounds for UB copies (guard values; actual sizes read from dims/params)
    static constexpr uint32_t MAX_W_SIZE = 1024 * 1024;  // upper bound placeholder
    static constexpr uint32_t MAX_B_SIZE = 65536;

    __aicore__ inline void CopyMetaAndWeights()
    {
        // dims: [N, Cin, Cout, Lin, K]
        // params: [stride, padding, dilation]
        dimsLocal = inQueueDims.AllocTensor<int32_t>();
        paramsLocal = inQueueParams.AllocTensor<int32_t>();
        AscendC::DataCopy(dimsLocal, dimsGm, 5);
        AscendC::DataCopy(paramsLocal, paramsGm, 3);
        inQueueDims.EnQue(dimsLocal);
        inQueueParams.EnQue(paramsLocal);

        // DeQue immediately for size info
        dimsLocal = inQueueDims.DeQue<int32_t>();
        paramsLocal = inQueueParams.DeQue<int32_t>();

        N = dimsLocal.GetValue(0);
        Cin = dimsLocal.GetValue(1);
        Cout = dimsLocal.GetValue(2);
        Lin = dimsLocal.GetValue(3);
        K = dimsLocal.GetValue(4);
        stride = paramsLocal.GetValue(0);
        padding = paramsLocal.GetValue(1);
        dilation = paramsLocal.GetValue(2);
        // compute Lout for stride=1 general formula (assume groups=1)
        Lout = Lin + (K - 1) * dilation - 2 * padding;
        if (stride > 1) {
            // basic extension for stride > 1 (transpose conv): Lout = (Lin - 1) * stride - 2*padding + (K - 1)*dilation + 1
            Lout = (Lin - 1) * stride - 2 * padding + (K - 1) * dilation + 1;
        }

        // copy weight [Cin, Cout, K]
        uint32_t wSize = Cin * Cout * K;
        wLocal = inQueueW.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm, wSize);
        inQueueW.EnQue(wLocal);
        // copy bias [Cout]
        uint32_t bSize = Cout;
        bLocal = inQueueB.AllocTensor<float>();
        AscendC::DataCopy(bLocal, bGm, bSize);
        inQueueB.EnQue(bLocal);

        // keep local tensors for compute
        wLocal = inQueueW.DeQue<float>();
        bLocal = inQueueB.DeQue<float>();
    }

    __aicore__ inline void FreeMetaAndWeights()
    {
        inQueueW.FreeTensor(wLocal);
        inQueueB.FreeTensor(bLocal);
        inQueueDims.FreeTensor(dimsLocal);
        inQueueParams.FreeTensor(paramsLocal);
    }

    __aicore__ inline void CopyOutTileInit(int32_t progress)
    {
        // Alloc output tile buffer and zero initialize
        yTileLocal = outQueueY.AllocTensor<float>();
        uint32_t len = tileLength;
        // Initialize to 0
        AscendC::SetValue(yTileLocal, 0.0f, len);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        // compute base index within the global output
        uint32_t base = this->blockLength * AscendC::GetBlockIdx() + progress * this->tileLength;

        for (uint32_t i = 0; i < this->tileLength; ++i) {
            uint32_t gIndex = base + i;               // flat index in y (N * Cout * Lout)
            // map to (n, oc, t)
            uint32_t t = gIndex % Lout;
            uint32_t tmp = gIndex / Lout;
            uint32_t oc = tmp % Cout;
            uint32_t n  = tmp / Cout;

            float acc = bLocal.GetValue(oc); // bias

            // sum over ic and k
            for (uint32_t ic = 0; ic < Cin; ++ic) {
                for (uint32_t kIdx = 0; kIdx < K; ++kIdx) {
                    // p = t + padding - k*dilation for stride=1
                    // For general stride: t - k*dilation must be divisible by stride, and p = (t + padding - k*dilation)/stride
                    int32_t numer = static_cast<int32_t>(t) + static_cast<int32_t>(padding) - static_cast<int32_t>(kIdx) * static_cast<int32_t>(dilation);
                    if (stride == 1) {
                        int32_t p = numer;
                        if (p >= 0 && p < static_cast<int32_t>(Lin)) {
                            // load x[n, ic, p]
                            uint64_t xOffset = ((uint64_t)n * Cin + ic) * Lin + static_cast<uint32_t>(p);
                            // copy single element
                            float xVal = LoadScalar(xOffset);
                            // w[ic, oc, k]
                            uint64_t wOffset = ((uint64_t)ic * Cout + oc) * K + kIdx;
                            float wVal = wLocal.GetValue(wOffset);
                            acc += xVal * wVal;
                        }
                    } else {
                        // stride > 1 case:
                        if (numer % stride == 0) {
                            int32_t p = numer / stride;
                            if (p >= 0 && p < static_cast<int32_t>(Lin)) {
                                uint64_t xOffset = ((uint64_t)n * Cin + ic) * Lin + static_cast<uint32_t>(p);
                                float xVal = LoadScalar(xOffset);
                                uint64_t wOffset = ((uint64_t)ic * Cout + oc) * K + kIdx;
                                float wVal = wLocal.GetValue(wOffset);
                                acc += xVal * wVal;
                            }
                        }
                    }
                }
            }
            yTileLocal.SetValue(i, acc);
        }
        outQueueY.EnQue<float>(yTileLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocalOut = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocalOut, this->tileLength);
        outQueueY.FreeTensor(yLocalOut);
    }

    __aicore__ inline float LoadScalar(uint64_t xOffset)
    {
        // allocate 1-element buffer for scalar load
        AscendC::LocalTensor<float> tmp = scalarBuf.AllocTensor<float>();
        AscendC::DataCopy(tmp, xGm[xOffset], 1);
        float val = tmp.GetValue(0);
        scalarBuf.FreeTensor(tmp);
        return val;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueB;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueParams;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueDims;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 2> scalarBuf; // temporary 1-element buffer queue

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<int32_t> paramsGm;
    AscendC::GlobalTensor<int32_t> dimsGm;
    AscendC::GlobalTensor<float> yGm;

    AscendC::LocalTensor<float> wLocal;
    AscendC::LocalTensor<float> bLocal;
    AscendC::LocalTensor<int32_t> paramsLocal;
    AscendC::LocalTensor<int32_t> dimsLocal;
    AscendC::LocalTensor<float> yTileLocal;

    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    // dimensions and params
    uint32_t N, Cin, Cout, Lin, K, Lout;
    uint32_t stride, padding, dilation;
};

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR params, GM_ADDR dims, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1dDilatedCustom op;
    op.Init(x, w, b, params, dims, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline at::Tensor ensure_bias(const at::Tensor& maybe_bias, int64_t Cout, const at::Tensor& like_tensor) {
    if (maybe_bias.defined() && maybe_bias.numel() == Cout) {
        return maybe_bias.contiguous();
    }
    // create zero bias on the same device and dtype
    return at::zeros({Cout}, like_tensor.options());
}

at::Tensor conv_transposed_1d_dilated_custom_impl_npu(const at::Tensor& x,
                                                      const at::Tensor& w,
                                                      c10::optional<at::Tensor> bias_opt,
                                                      int64_t stride,
                                                      int64_t padding,
                                                      int64_t dilation) {
    TORCH_CHECK(x.dim() == 3, "x should be 3D tensor (N, Cin, Lin)");
    TORCH_CHECK(w.dim() == 3, "w should be 3D tensor (Cin, Cout, K)");
    auto x_c = x.contiguous();
    auto w_c = w.contiguous();
    int64_t N = x_c.size(0);
    int64_t Cin = x_c.size(1);
    int64_t Lin = x_c.size(2);
    int64_t K = w_c.size(2);
    int64_t Cout = w_c.size(1);

    // Output length formula for transposed conv1d
    int64_t Lout = (Lin - 1) * stride - 2 * padding + (K - 1) * dilation + 1;
    at::Tensor y = at::empty({N, Cout, Lout}, x_c.options());

    // prepare params tensor [stride, padding, dilation]
    at::Tensor params = at::tensor({(int32_t)stride, (int32_t)padding, (int32_t)dilation}, at::TensorOptions().dtype(at::kInt).device(x.device()));
    // prepare dims tensor [N, Cin, Cout, Lin, K]
    at::Tensor dims = at::tensor({(int32_t)N, (int32_t)Cin, (int32_t)Cout, (int32_t)Lin, (int32_t)K}, at::TensorOptions().dtype(at::kInt).device(x.device()));
    at::Tensor b = ensure_bias(bias_opt.has_value() ? bias_opt.value() : at::Tensor(), Cout, x_c);

    EXEC_NPU_CMD(aclnnConvTransposed1dDilatedCustom, x_c, w_c, b, params, dims, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_transposed_1d_dilated_custom", &conv_transposed_1d_dilated_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_transposed_1d_dilated_custom", &conv_transposed_1d_dilated_custom_impl_npu, "ConvTranspose1d with dilation (custom AscendC)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom ConvTranspose1d (transposed 1D convolution) with dilation using AscendC kernel.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int,
                 stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        # Weight shape for PyTorch ConvTranspose1d: (in_channels, out_channels, kernel_size)
        self.weight = torch.nn.Parameter(torch.empty(in_channels, out_channels, kernel_size))
        if bias:
            self.bias = torch.nn.Parameter(torch.empty(out_channels))
        else:
            self.bias = None
        # Initialize similar to PyTorch default
        torch.nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)
        if self.bias is not None:
            fan_in = in_channels * kernel_size
            bound = 1 / (fan_in ** 0.5)
            torch.nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_1d_dilated_custom(
            x, self.weight, self.bias, self.stride, self.padding, self.dilation
        )
