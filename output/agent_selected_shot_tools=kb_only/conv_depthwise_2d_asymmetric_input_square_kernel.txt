kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_Q_NUM = 1;
constexpr int32_t WT_Q_NUM = 1;
constexpr int32_t OUT_Q_NUM = 1;

class KernelDepthwise2D {
public:
    __aicore__ inline KernelDepthwise2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR z,
                                uint32_t N, uint32_t C, uint32_t HIn, uint32_t WIn,
                                uint32_t K, uint32_t HOut, uint32_t WOut, uint32_t tileH)
    {
        this->N = N;
        this->C = C;
        this->HIn = HIn;
        this->WIn = WIn;
        this->K = K;
        this->HOut = HOut;
        this->WOut = WOut;
        this->tileH = tileH;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N * C * HIn * WIn);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)C * K * K);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z, (uint64_t)N * C * HOut * WOut);

        // Max rows needed for an output tile: rows = tileH*stride + (K-1), stride=1
        uint32_t rowsMax = (tileH == 0 ? 1 : tileH) + (K > 0 ? (K - 1) : 0);
        uint32_t inBufSize = rowsMax * WIn;
        uint32_t outBufSize = (tileH == 0 ? 1 : tileH) * WOut;
        uint32_t wtBufSize = K * K;

        pipe.InitBuffer(inQueueIn, IN_Q_NUM, inBufSize * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueWt, WT_Q_NUM, wtBufSize * sizeof(DTYPE_W));
        pipe.InitBuffer(outQueueOut, OUT_Q_NUM, outBufSize * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // Channel sharding among blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t base = C / (blockNum == 0 ? 1 : blockNum);
        uint32_t rem = C % (blockNum == 0 ? 1 : blockNum);
        uint32_t cBegin = base * blockIdx + (blockIdx < rem ? blockIdx : rem);
        uint32_t cLen = base + (blockIdx < rem ? 1 : 0);
        uint32_t cEnd = (cBegin + cLen > C) ? C : (cBegin + cLen);

        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t c = cBegin; c < cEnd; ++c) {
                // Load weight for channel c (KxK)
                AscendC::LocalTensor<DTYPE_W> wLocal = inQueueWt.AllocTensor<DTYPE_W>();
                uint64_t wOffset = (uint64_t)c * K * K;
                AscendC::DataCopy(wLocal, wGm[wOffset], K * K);
                inQueueWt.EnQue(wLocal);

                for (uint32_t hOutStart = 0; hOutStart < HOut; hOutStart += tileH) {
                    uint32_t th = (hOutStart + tileH <= HOut) ? tileH : (HOut - hOutStart);
                    uint32_t rowsNeeded = th + (K > 0 ? (K - 1) : 0); // stride=1

                    // Prepare input tile buffer (rowsNeeded x WIn)
                    AscendC::LocalTensor<DTYPE_X> inLocal = inQueueIn.AllocTensor<DTYPE_X>();
                    // Fill with zeros first for safety (borders)
                    for (uint32_t i = 0; i < rowsNeeded * WIn; ++i) {
                        inLocal.SetValue(i, (DTYPE_X)0);
                    }
                    // Compute the top input row index covered by this tile
                    int64_t hInMin = static_cast<int64_t>(hOutStart); // padding=0, stride=1
                    for (uint32_t r = 0; r < rowsNeeded; ++r) {
                        int64_t hIn = hInMin + static_cast<int64_t>(r);
                        if (hIn >= 0 && hIn < (int64_t)HIn) {
                            uint64_t xRowOffset = (((uint64_t)n * C + c) * HIn + (uint64_t)hIn) * WIn;
                            AscendC::DataCopy(inLocal[r * WIn], xGm[xRowOffset], WIn);
                        }
                    }
                    inQueueIn.EnQue(inLocal);

                    // Allocate output local buffer for current tile
                    AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.AllocTensor<DTYPE_Z>();
                    // Compute convolution for the tile
                    AscendC::LocalTensor<DTYPE_W> wLocalUsed = inQueueWt.DeQue<DTYPE_W>();
                    AscendC::LocalTensor<DTYPE_X> inLocalUsed = inQueueIn.DeQue<DTYPE_X>();

                    for (uint32_t thIdx = 0; thIdx < th; ++thIdx) {
                        for (uint32_t wOut = 0; wOut < WOut; ++wOut) {
                            float acc = 0.0f;
                            for (uint32_t kh = 0; kh < K; ++kh) {
                                int64_t hInIndex = (int64_t)thIdx + (int64_t)kh; // relative to inLocal start
                                if (hInIndex < 0 || hInIndex >= (int64_t)rowsNeeded) { continue; }
                                for (uint32_t kw = 0; kw < K; ++kw) {
                                    int64_t wInIndex = (int64_t)wOut + (int64_t)kw; // padding=0, stride=1
                                    if (wInIndex >= 0 && wInIndex < (int64_t)WIn) {
                                        DTYPE_X inVal = inLocalUsed.GetValue((uint32_t)hInIndex * WIn + (uint32_t)wInIndex);
                                        DTYPE_W wVal = wLocalUsed.GetValue(kh * K + kw);
                                        acc += (float)inVal * (float)wVal;
                                    }
                                }
                            }
                            outLocal.SetValue(thIdx * WOut + wOut, (DTYPE_Z)acc);
                        }
                    }

                    // Write out the tile
                    uint64_t zTileOffset = (((uint64_t)n * C + c) * HOut + hOutStart) * WOut;
                    AscendC::DataCopy(zGm[zTileOffset], outLocal, th * WOut);

                    // recycle local buffers
                    outQueueOut.EnQue(outLocal);
                    inQueueWt.FreeTensor(wLocalUsed);
                    inQueueIn.FreeTensor(inLocalUsed);

                    // Free queued tensors
                    outQueueOut.FreeTensor(outLocal);
                } // end hOutStart loop
            } // end channel loop
        } // end batch loop
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_Q_NUM> inQueueIn;
    AscendC::TQue<AscendC::TPosition::VECIN, WT_Q_NUM> inQueueWt;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_Q_NUM> outQueueOut;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t N;
    uint32_t C;
    uint32_t HIn;
    uint32_t WIn;
    uint32_t K;
    uint32_t HOut;
    uint32_t WOut;
    uint32_t tileH;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelDepthwise2D op;
    op.Init(x, weight, z,
            tilingData.N(),
            tilingData.C(),
            tilingData.HIn(),
            tilingData.WIn(),
            tilingData.K(),
            tilingData.HOut(),
            tilingData.WOut(),
            tilingData.tileH());
    op.Process();
}

import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        assert in_channels == out_channels, "Depthwise conv requires in_channels == out_channels"
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        # Depthwise weight [C, 1, K, K]
        w = torch.empty(out_channels, 1, kernel_size, kernel_size)
        torch.nn.init.kaiming_uniform_(w, a=math.sqrt(5))
        self.weight = torch.nn.Parameter(w)
        if bias:
            self.bias = torch.nn.Parameter(torch.zeros(out_channels))
        else:
            self.register_parameter("bias", None)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Custom kernel supports stride=1, padding=0; bias not applied in this kernel
        assert self.stride == 1 and self.padding == 0, "Custom kernel currently supports stride=1 and padding=0"
        y = custom_ops_lib.conv_depthwise_2d_asymmetric_input_square_kernel_custom(x, self.weight)
        if self.bias is not None:
            # Add bias per channel
            y = y + self.bias.view(1, -1, 1, 1)
        return y
