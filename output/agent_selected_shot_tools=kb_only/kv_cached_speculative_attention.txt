kernel_src="""
#include "kernel_operator.h"

using DTYPE_W = float;
using DTYPE_V = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2; // double buffer for better overlap

class KernelKvCachedSpeculativeAttention {
public:
    __aicore__ inline KernelKvCachedSpeculativeAttention() {}

    __aicore__ inline void Init(GM_ADDR weight, GM_ADDR value, GM_ADDR y,
                                uint32_t B, uint32_t Q, uint32_t KV, uint32_t D,
                                uint32_t tileKV, uint32_t tileD)
    {
        this->B = B;
        this->Q = Q;
        this->KV = KV;
        this->D = D;
        this->tileKV = tileKV;
        this->tileD = tileD;

        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)weight, (uint64_t)B * Q * KV);
        vGm.SetGlobalBuffer((__gm__ DTYPE_V*)value, (uint64_t)B * KV * D);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, (uint64_t)B * Q * D);

        // Initialize UB buffers
        pipe.InitBuffer(wQueue, BUFFER_NUM, this->tileKV * sizeof(DTYPE_W));
        pipe.InitBuffer(vQueue, BUFFER_NUM, this->tileD * sizeof(DTYPE_V));
        pipe.InitBuffer(yQueue, BUFFER_NUM, this->tileD * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileD * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Compute Y[b, q, d] = sum_{t=0..KV-1} weight[b,q,t] * V[b,t,d]
        for (uint32_t b = 0; b < B; ++b) {
            for (uint32_t q = 0; q < Q; ++q) {
                // iterate D in tiles
                for (uint32_t d0 = 0; d0 < D; d0 += tileD) {
                    uint32_t curD = AscendC::min((uint32_t)tileD, (uint32_t)(D - d0));

                    // yLocal init to 0
                    AscendC::LocalTensor<DTYPE_Y> yLocal = yQueue.AllocTensor<DTYPE_Y>();
                    AscendC::Duplicate(yLocal, (DTYPE_Y)0.0f, curD);
                    // loop over KV in tiles
                    for (uint32_t kv0 = 0; kv0 < KV; kv0 += tileKV) {
                        uint32_t curKV = AscendC::min((uint32_t)tileKV, (uint32_t)(KV - kv0));

                        // Load weight vector [curKV]
                        AscendC::LocalTensor<DTYPE_W> wLocal = wQueue.AllocTensor<DTYPE_W>();
                        uint64_t wOffset = ((uint64_t)b * Q + q) * (uint64_t)KV + kv0;
                        AscendC::DataCopy(wLocal, wGm[wOffset], curKV);
                        wQueue.EnQue(wLocal);

                        // For each t in this KV tile, accumulate w[t] * V[b, kv0+t, d0:d0+curD]
                        wLocal = wQueue.DeQue<DTYPE_W>();
                        for (uint32_t t = 0; t < curKV; ++t) {
                            // Load V vector chunk
                            AscendC::LocalTensor<DTYPE_V> vLocal = vQueue.AllocTensor<DTYPE_V>();
                            uint64_t vOffset = ((uint64_t)b * KV + (kv0 + t)) * (uint64_t)D + d0;
                            AscendC::DataCopy(vLocal, vGm[vOffset], curD);

                            // tmpLocal = vLocal * scalar_weight
                            AscendC::LocalTensor<DTYPE_Y> tmpLocal = tmpQueue.AllocTensor<DTYPE_Y>();
                            // Scalar multiply: vector <- vector * scalar
                            // Note: Muls is a two-operand scalar instruction from vector APIs
                            AscendC::Muls(tmpLocal, vLocal, *(reinterpret_cast<DTYPE_W*>(wLocal.GetPhyAddr())) + t, curD);

                            // yLocal += tmpLocal
                            AscendC::Add(yLocal, yLocal, tmpLocal, curD);

                            tmpQueue.FreeTensor(tmpLocal);
                            vQueue.FreeTensor(vLocal);
                        }
                        wQueue.FreeTensor(wLocal);
                    }

                    // Store yLocal -> GM
                    uint64_t yOffset = ((uint64_t)b * Q + q) * (uint64_t)D + d0;
                    AscendC::DataCopy(yGm[yOffset], yLocal, curD);
                    yQueue.FreeTensor(yLocal);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> vQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue;

    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_V> vGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t B;
    uint32_t Q;
    uint32_t KV;
    uint32_t D;
    uint32_t tileKV;
    uint32_t tileD;
};

extern "C" __global__ __aicore__ void kv_cached_speculative_attention_custom(GM_ADDR weight, GM_ADDR value, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelKvCachedSpeculativeAttention op;
    op.Init(weight, value, y,
            tiling_data.batch, tiling_data.q_len, tiling_data.kv_len, tiling_data.d_model,
            tiling_data.tileKV, tiling_data.tileD);
    op.Process();
}

model_src='''
import math
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self) -> None:
        super().__init__()

    def forward(self, Q, K_cache, V_cache):
        # Compute attention probabilities on PyTorch, apply custom kernel for weighted sum (V aggregation)
        scale = 1.0 / math.sqrt(Q.size(-1))
        scores = torch.matmul(Q, K_cache.transpose(-2, -1)) * scale
        prob = torch.softmax(scores, dim=-1)  # [B,Q,KV]
        # Custom AscendC kernel: Y = prob @ V_cache
        return custom_ops_lib.kv_cached_speculative_attention_custom(prob, V_cache)
