kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelSigmoidSum {
public:
    __aicore__ inline KernelSigmoidSum() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t batchN, uint32_t tileNum)
    {
        this->totalLen = totalLength;       // N*C*H*W
        this->batchN = batchN;              // N
        this->tileNum = tileNum;

        this->elemPerSample = this->totalLen / this->batchN; // C*H*W
        // Basic tiling along sample inner length
        this->tileLength = AscendC::Max((uint32_t)1, this->elemPerSample / (this->tileNum * BUFFER_NUM));

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x, this->totalLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y, this->batchN);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(calcQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(workQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE)); // workspace for ReduceSum
        pipe.InitBuffer(outQueueY, 1, sizeof(DTYPE)); // per-sample scalar
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t perBlockN = this->batchN / blockNum;
        uint32_t nStart = blockIdx * perBlockN;
        uint32_t nEnd = (blockIdx == blockNum - 1) ? this->batchN : (nStart + perBlockN);

        for (uint32_t n = nStart; n < nEnd; ++n) {
            ComputeOneSample(n);
        }
    }

private:
    __aicore__ inline void ComputeOneSample(uint32_t n)
    {
        // sum over all C*H*W after sigmoid
        DTYPE sumVal = (DTYPE)0;
        uint32_t base = n * this->elemPerSample;

        // number of tiles to cover elemPerSample
        uint32_t fullTileSize = this->tileLength;
        if (fullTileSize == 0) { return; }
        uint32_t tiles = (this->elemPerSample + fullTileSize - 1) / fullTileSize;

        for (uint32_t t = 0; t < tiles * BUFFER_NUM; ++t) {
            CopyIn(base, t, tiles);
            ComputeTile(t, tiles, sumVal);
        }

        // write sumVal to GM y[n]
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
        AscendC::Duplicate(yLocal, sumVal, 1);
        outQueueY.EnQue<DTYPE>(yLocal);
        AscendC::LocalTensor<DTYPE> yLocalDeq = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[n], yLocalDeq, 1);
        outQueueY.FreeTensor(yLocalDeq);
    }

    __aicore__ inline void CopyIn(uint32_t base, uint32_t progress, uint32_t tiles)
    {
        // progress loops through tiles * BUFFER_NUM; map to tile index
        uint32_t tileIdx = progress % tiles;
        uint32_t offset = tileIdx * this->tileLength;
        uint32_t remaining = (offset + this->tileLength <= this->elemPerSample) ? this->tileLength : (this->elemPerSample - offset);
        if (remaining == 0) {
            // allocate and enqueue dummy to keep pipeline consistent
            AscendC::LocalTensor<DTYPE> xLocalDummy = inQueueX.AllocTensor<DTYPE>();
            inQueueX.EnQue(xLocalDummy);
            return;
        }
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[base + offset], remaining);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeTile(uint32_t progress, uint32_t tiles, DTYPE &sumVal)
    {
        uint32_t tileIdx = progress % tiles;
        uint32_t offset = tileIdx * this->tileLength;
        uint32_t remaining = (offset + this->tileLength <= this->elemPerSample) ? this->tileLength : (this->elemPerSample - offset);
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        if (remaining > 0) {
            // Sigmoid
            AscendC::LocalTensor<DTYPE> calcLocal = calcQueue.AllocTensor<DTYPE>();
            AscendC::Sigmoid(calcLocal, xLocal, remaining);

            // ReduceSum over the tile
            AscendC::LocalTensor<DTYPE> dstLocal = calcQueue.AllocTensor<DTYPE>(); // temporary dst
            AscendC::LocalTensor<DTYPE> workLocal = workQueue.AllocTensor<DTYPE>(); // workspace
            AscendC::ReduceSum(dstLocal, calcLocal, workLocal, remaining);
            DTYPE tileSum = AscendC::GetAccVal<DTYPE>();
            sumVal += tileSum;

            calcQueue.FreeTensor(calcLocal);
            calcQueue.FreeTensor(dstLocal);
            workQueue.FreeTensor(workLocal);
        }
        inQueueX.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> calcQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> workQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t totalLen;
    uint32_t batchN;
    uint32_t elemPerSample;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_avg_pool_sigmoid_sum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSigmoidSum op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.batchN, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep Conv2d and AvgPool2d in PyTorch, fuse Sigmoid + Sum into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, pool_kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.avg_pool = torch.nn.AvgPool2d(pool_kernel_size)

    def forward(self, x):
        x = self.conv(x)
        x = self.avg_pool(x)
        # custom op computes sigmoid(x) and then sums over [1,2,3] -> [N]
        x = custom_ops_lib.conv2d_avg_pool_sigmoid_sum_custom(x)
        return x
