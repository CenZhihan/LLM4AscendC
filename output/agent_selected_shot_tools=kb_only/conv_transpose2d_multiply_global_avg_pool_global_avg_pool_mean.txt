kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

template <typename T, typename accT>
class KernelFusedGlobalMean {
public:
    __aicore__ inline KernelFusedGlobalMean() {}

    __aicore__ inline void Init(GM_ADDR x_gm, GM_ADDR y_gm, GM_ADDR workspace_gm,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t tileRows1, uint32_t tileRows2, float multiplier)
    {
        this->n = n; this->c = c; this->h = h; this->w = w;
        this->tileRows1 = tileRows1;
        this->tileRows2 = tileRows2;
        this->multiplier = (T)multiplier;

        // aligned inner sizes for Mean
        this->inner1 = ((w * sizeof(T) + 32 - 1) / 32) * 32 / sizeof(T);
        this->inner2 = ((h * sizeof(T) + 32 - 1) / 32) * 32 / sizeof(T);

        this->outter1 = n * c * h; // rows for reduce over W
        this->outter2 = n * c;     // rows for reduce over H

        xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(x_gm), (uint32_t)(n * c * h * w));
        yGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(y_gm), (uint32_t)(n * c));
        interGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(workspace_gm), (uint32_t)(outter1));

        // Queues and buffers
        // Stage 1: in size = tileRows1 * inner1, out size = tileRows1
        pipe.InitBuffer(inQ1, 1, tileRows1 * inner1 * sizeof(T));
        pipe.InitBuffer(outQ1, 1, tileRows1 * sizeof(T));

        // Stage 2: in size = tileRows2 * inner2, out size = tileRows2
        pipe.InitBuffer(inQ2, 1, tileRows2 * inner2 * sizeof(T));
        pipe.InitBuffer(outQ2, 1, tileRows2 * sizeof(T));
    }

    __aicore__ inline void Process()
    {
        // Pass 1: Mean over W
        for (uint32_t start = 0; start < outter1; start += tileRows1) {
            uint32_t rows = tileRows1;
            if (start + rows > outter1) rows = outter1 - start;

            // Copy in contiguous block of rows * w (if inner1 == w); If inner1 > w, zero padding is recommended by doc.
            LocalTensor<T> srcLocal = inQ1.AllocTensor<T>();
            // When inner1 == w, can copy directly
            if (inner1 == w) {
                DataCopy(srcLocal, xGlobal[start * w], rows * w);
            } else {
                // Fallback: zero initialize then row-wise copy and pad
                // Set to zero
                SetTensorValue(srcLocal, (T)0, rows * inner1);
                for (uint32_t r = 0; r < rows; ++r) {
                    DataCopy(srcLocal[r * inner1], xGlobal[(start + r) * w], w);
                }
            }
            inQ1.EnQue(srcLocal);

            // Compute mean along last axis (W)
            LocalTensor<T> srcLocalDeq = inQ1.DeQue<T>();
            LocalTensor<T> dstLocal = outQ1.AllocTensor<T>();
            MeanParams p1;
            p1.outter = rows;
            p1.inner = inner1;
            p1.n = w;
            AscendC::Mean<T, accT>(dstLocal, srcLocalDeq, p1);
            outQ1.EnQue<T>(dstLocal);
            inQ1.FreeTensor(srcLocalDeq);

            // Copy out to workspace
            LocalTensor<T> dstLocalDeq = outQ1.DeQue<T>();
            DataCopy(interGlobal[start], dstLocalDeq, rows);
            outQ1.FreeTensor(dstLocalDeq);
        }

        // Pass 2: Mean over H
        for (uint32_t start2 = 0; start2 < outter2; start2 += tileRows2) {
            uint32_t rows2 = tileRows2;
            if (start2 + rows2 > outter2) rows2 = outter2 - start2;

            LocalTensor<T> src2Local = inQ2.AllocTensor<T>();
            if (inner2 == h) {
                DataCopy(src2Local, interGlobal[start2 * h], rows2 * h);
            } else {
                SetTensorValue(src2Local, (T)0, rows2 * inner2);
                for (uint32_t r = 0; r < rows2; ++r) {
                    DataCopy(src2Local[r * inner2], interGlobal[(start2 + r) * h], h);
                }
            }
            inQ2.EnQue(src2Local);

            LocalTensor<T> src2LocalDeq = inQ2.DeQue<T>();
            LocalTensor<T> dst2Local = outQ2.AllocTensor<T>();
            MeanParams p2;
            p2.outter = rows2;
            p2.inner = inner2;
            p2.n = h;
            AscendC::Mean<T, accT>(dst2Local, src2LocalDeq, p2);

            // Multiply by scalar
            AscendC::Muls(dst2Local, dst2Local, multiplier, rows2);

            outQ2.EnQue<T>(dst2Local);
            inQ2.FreeTensor(src2LocalDeq);

            LocalTensor<T> dst2LocalDeq = outQ2.DeQue<T>();
            // Output is [N*C], which maps to [N, C, 1, 1]
            DataCopy(yGlobal[start2], dst2LocalDeq, rows2);
            outQ2.FreeTensor(dst2LocalDeq);
        }
    }

private:
    __aicore__ inline void SetTensorValue(LocalTensor<T>& t, T v, uint32_t cnt)
    {
        // simple fill by setting first and broadcast via vector ops if available; fallback loop
        // Using Muls after setting to zero for safety
        // Initialize to zero
        AscendC::SetVector(t, (uint8_t)0, cnt * sizeof(T));
        if (v != (T)0) {
            // create a temp buffer and use Adds with zero then Muls. For simplicity, do Muls with 0 -> 0 (skip)
            // Here we skip non-zero fill for brevity.
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQ1, inQ2;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQ1, outQ2;

    AscendC::GlobalTensor<T> xGlobal;
    AscendC::GlobalTensor<T> yGlobal;
    AscendC::GlobalTensor<T> interGlobal;

    uint32_t n, c, h, w;
    uint32_t outter1, inner1, outter2, inner2;
    uint32_t tileRows1, tileRows2;
    T multiplier;
};

extern "C" __global__ __aicore__ void conv_transpose2d_multiply_global_avg_pool_global_avg_pool_mean_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    uint32_t n = tilingData.n();
    uint32_t c = tilingData.c();
    uint32_t h = tilingData.h();
    uint32_t w = tilingData.w();
    uint32_t tileRows1 = tilingData.tileRows1();
    uint32_t tileRows2 = tilingData.tileRows2();
    float multiplier = tilingData.multiplier();

    if (TILING_KEY_IS(1)) {
        // float16 compute with float accumulation
        KernelFusedGlobalMean<half, float> op;
        op.Init(x, y, workspace, n, c, h, w, tileRows1, tileRows2, multiplier);
        op.Process();
    } else {
        KernelFusedGlobalMean<float, float> op;
        op.Init(x, y, workspace, n, c, h, w, tileRows1, tileRows2, multiplier);
        op.Process();
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused: x (after ConvTranspose2d) * multiplier -> global avg pool (H,W) -> global avg pool (H,W)
    The two global avg pools are equivalent to a single global avg pool when keepdim=True after the first.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, multiplier):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.multiplier = float(multiplier)

    def forward(self, x):
        x = self.conv_transpose(x)
        # Fused custom op handles multiply + global average pooling twice (equivalent to once)
        return custom_ops_lib.conv_transpose2d_multiply_global_avg_pool_global_avg_pool_mean_custom(x, self.multiplier)
