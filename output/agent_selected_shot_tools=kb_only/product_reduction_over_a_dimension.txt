host_operator_src="""
#include "product_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
constexpr uint32_t BLOCK_DIM = 1;
// Choose a tile length that balances UB usage and throughput.
constexpr uint32_t ONE_REPEAT_BYTES = 256;
constexpr uint32_t TILE_BYTES = ONE_REPEAT_BYTES * 4; // 1KB per tile
constexpr uint32_t TILE_FLOAT_ELEMS = TILE_BYTES / sizeof(float);

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TilingData tiling;
    // Expect 3D input: [B, M, N]
    auto shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t dimNum = shape.GetDimNum();
    if (dimNum < 2) {
        return ge::GRAPH_FAILED;
    }
    uint32_t B = 1, M = 1, N = 1;
    if (dimNum == 2) { // treat as [B, N], reduce over dim=1 not applicable; fallback
        B = shape.GetDim(0);
        M = 1;
        N = shape.GetDim(1);
    } else {
        B = shape.GetDim(0);
        M = shape.GetDim(1);
        N = shape.GetDim(2);
    }

    // Fix reduction over dim=1 as per provided model.
    uint32_t reduceDim = 1;
    uint32_t tileN = TILE_FLOAT_ELEMS;
    if (tileN == 0) tileN = 64;
    // Output shape is [B, N] when reducing over dim=1.
    uint32_t outLen = B * N;

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_batch(B);
    tiling.set_mLen(M);
    tiling.set_nLen(N);
    tiling.set_reduceDim(reduceDim);
    tiling.set_tileN(tileN);
    tiling.set_outLength(outLen);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static graphStatus InferShape(gert::InferShapeContext *context)
{
    // Input assumed to be [B, M, N]; reduce over dim=1 => [B, N]
    auto in_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t dimNum = in_shape.GetDimNum();
    if (dimNum < 2) {
        return GRAPH_FAILED;
    }
    uint64_t B = (dimNum >= 1) ? in_shape.GetDim(0) : 1;
    uint64_t N = (dimNum >= 3) ? in_shape.GetDim(2) : ((dimNum == 2) ? in_shape.GetDim(1) : 1);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = {B, N};
    return GRAPH_SUCCESS;
}

static graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

class KernelProdReduce {
public:
    __aicore__ inline KernelProdReduce() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t batch, uint32_t mLen, uint32_t nLen,
                                uint32_t reduceDim, uint32_t tileN, uint32_t outLength)
    {
        this->batch = batch;
        this->mLen = mLen;
        this->nLen = nLen;
        this->reduceDim = reduceDim; // currently only support reduce over dim=1
        this->tileN = tileN;
        this->outLength = outLength;

        xPtr = reinterpret_cast<__gm__ float *>(x);
        zPtr = reinterpret_cast<__gm__ float *>(z);

        // Prepare UB buffers: one tile for accumulator and one for current slice.
        uint32_t bytesPerTile = tileN * sizeof(float);
        pipe.InitBuffer(accBuf, 1, bytesPerTile);
        pipe.InitBuffer(curBuf, 1, bytesPerTile);
    }

    __aicore__ inline void Process()
    {
        // Reduce over dim=1 for input [B, M, N] -> output [B, N]
        for (uint32_t b = 0; b < batch; ++b) {
            for (uint32_t nStart = 0; nStart < nLen; nStart += tileN) {
                uint32_t curTile = AscendC::min(tileN, nLen - nStart);
                // Local tensors for accumulator and current slice
                AscendC::LocalTensor<float> accLocal = accBuf.Get<float>();
                AscendC::LocalTensor<float> curLocal = curBuf.Get<float>();

                // Initialize accumulator to 1.0
                float one = 1.0f;
                AscendC::Duplicate(accLocal, one, curTile);

                // Iterate over M slices and multiply element-wise
                for (uint32_t m = 0; m < mLen; ++m) {
                    size_t inOffset = static_cast<size_t>(b) * mLen * nLen
                                    + static_cast<size_t>(m) * nLen
                                    + nStart;
                    AscendC::GlobalTensor<float> xTileGm;
                    xTileGm.SetGlobalBuffer(xPtr + inOffset, curTile);
                    AscendC::DataCopy(curLocal, xTileGm, curTile);

                    // accLocal = accLocal * curLocal (first n elements)
                    AscendC::Mul(accLocal, accLocal, curLocal, static_cast<int32_t>(curTile));
                }

                // Write back accumulator to output
                size_t outOffset = static_cast<size_t>(b) * nLen + nStart;
                AscendC::GlobalTensor<float> zTileGm;
                zTileGm.SetGlobalBuffer(zPtr + outOffset, curTile);
                AscendC::DataCopy(zTileGm, accLocal, curTile);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::TPosition::VECCALC> accBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> curBuf;

    __gm__ float *xPtr;
    __gm__ float *zPtr;

    uint32_t batch;
    uint32_t mLen;
    uint32_t nLen;
    uint32_t reduceDim;
    uint32_t tileN;
    uint32_t outLength;
};

extern "C" __global__ __aicore__ void product_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelProdReduce op;
    op.Init(x, z,
            tiling_data.batch,
            tiling_data.mLen,
            tiling_data.nLen,
            tiling_data.reduceDim,
            tiling_data.tileN,
            tiling_data.outLength);
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor product_reduction_over_a_dimension_impl_npu(const at::Tensor& x, int64_t dim) {
    // Reduce over dim=1 -> output shape [B, N] for input [B, M, N]
    TORCH_CHECK(x.dim() >= 2, \"Input tensor rank must be >= 2\");
    at::Tensor result;
    if (x.dim() == 3 && dim == 1) {
        result = at::empty({x.size(0), x.size(2)}, x.options());
    } else if (x.dim() == 2 && dim == 1) {
        result = at::empty({x.size(0)}, x.options());
    } else {
        // Fallback shape inference; let backend decide if unsupported
        result = at::empty(x.sizes().vec(), x.options());
    }
    // Pass dim as an argument; backend may fix to dim=1 as per kernel.
    EXEC_NPU_CMD(aclnnProductReductionOverADimensionCustom, x, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"product_reduction_over_a_dimension_custom\", &product_reduction_over_a_dimension_impl_npu);
}
PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"product_reduction_over_a_dimension_custom\", &product_reduction_over_a_dimension_impl_npu, \"product reduction over a dimension\");
