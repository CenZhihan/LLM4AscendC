kernel_src="""
#include "kernel_operator.h"

#define DTYPE_X float
#define DTYPE_B float
#define DTYPE_Y float

class KernelConv3dDivideMaxGlobalAvgPoolBiasAddSum {
public:
    __aicore__ inline KernelConv3dDivideMaxGlobalAvgPoolBiasAddSum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t D, uint32_t H, uint32_t W,
                                uint32_t tileNum)
    {
        n_ = N;
        c_ = C;
        d_ = D;
        h_ = H;
        w_ = W;
        tileNum_ = tileNum;
        spatialSize_ = static_cast<uint32_t>(d_) * static_cast<uint32_t>(h_) * static_cast<uint32_t>(w_);
        tileLen_ = (spatialSize_ + tileNum_ - 1) / tileNum_;
        if (tileLen_ == 0) tileLen_ = 1;

        xGm_.SetGlobalBuffer((__gm__ DTYPE_X*)x);
        bGm_.SetGlobalBuffer((__gm__ DTYPE_B*)bias);
        yGm_.SetGlobalBuffer((__gm__ DTYPE_Y*)y);

        pipe_.InitBuffer(inQueueX_, 1, tileLen_ * sizeof(DTYPE_X));
        pipe_.InitBuffer(biasQueue_, 1, sizeof(DTYPE_B));
        pipe_.InitBuffer(outQueueY_, 1, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();

        uint32_t n_per_block = (n_ + blocks - 1) / blocks;
        uint32_t n_begin = idx * n_per_block;
        uint32_t n_end = n_begin + n_per_block;
        if (n_end > n_) n_end = n_;

        for (uint32_t n = n_begin; n < n_end; ++n) {
            float sum_c = 0.0f;
            for (uint32_t c = 0; c < c_; ++c) {
                uint64_t baseOffset = ((uint64_t)n * c_ + c) * (uint64_t)spatialSize_;
                float acc = 0.0f;
                uint32_t processed = 0;
                while (processed < spatialSize_) {
                    uint32_t curLen = tileLen_;
                    if (processed + curLen > spatialSize_) {
                        curLen = spatialSize_ - processed;
                    }
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX_.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(xLocal, xGm_[baseOffset + processed], curLen);
                    for (uint32_t i = 0; i < curLen; ++i) {
                        acc += xLocal[i];
                    }
                    inQueueX_.FreeTensor(xLocal);
                    processed += curLen;
                }
                float mean_val = acc / static_cast<float>(spatialSize_);

                AscendC::LocalTensor<DTYPE_B> bLocal = biasQueue_.AllocTensor<DTYPE_B>();
                AscendC::DataCopy(bLocal, bGm_[c], 1);
                float b = bLocal[0];
                biasQueue_.FreeTensor(bLocal);

                sum_c += (mean_val + b);
            }

            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY_.AllocTensor<DTYPE_Y>();
            yLocal[0] = sum_c;
            AscendC::DataCopy(yGm_[n], yLocal, 1);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::GlobalTensor<DTYPE_X> xGm_;
    AscendC::GlobalTensor<DTYPE_B> bGm_;
    AscendC::GlobalTensor<DTYPE_Y> yGm_;

    uint32_t n_{0}, c_{0}, d_{0}, h_{0}, w_{0};
    uint32_t spatialSize_{0};
    uint32_t tileNum_{1};
    uint32_t tileLen_{1};
};

extern "C" __global__ __aicore__ void conv3d_divide_max_global_avg_pool_bias_add_sum_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dDivideMaxGlobalAvgPoolBiasAddSum op;
    op.Init(x, bias, y,
            tiling_data.N, tiling_data.C, tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model:
    Keeps Conv3d, divide and MaxPool3d in PyTorch, and fuses the remaining
    GlobalAvgPool3d + BiasAdd + channel-wise Sum into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divisor, pool_size, bias_shape, sum_dim):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.divisor = divisor
        self.max_pool = torch.nn.MaxPool3d(pool_size)
        # bias parameter retained for fused op
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.sum_dim = sum_dim  # expected to be 1

    def forward(self, x):
        x = self.conv(x)
        x = x / self.divisor
        x = self.max_pool(x)
        # Fused: global average pooling + bias add + sum over channels
        y = custom_ops_lib.conv3d_divide_max_global_avg_pool_bias_add_sum_custom(x, self.bias)
        return y
