kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelConvTransposed3D {
public:
    __aicore__ inline KernelConvTransposed3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Din, uint32_t Hin, uint32_t Win,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw)
    {
        this->N = N;
        this->Cin = Cin;
        this->Cout = Cout;
        this->Din = Din;
        this->Hin = Hin;
        this->Win = Win;
        this->Kd = Kd;
        this->Kh = Kh;
        this->Kw = Kw;

        this->Dout = Din + Kd - 1;
        this->Hout = Hin + Kh - 1;
        this->Wout = Win + Kw - 1;

        uint64_t x_elems = (uint64_t)N * Cin * Din * Hin * Win;
        uint64_t w_elems = (uint64_t)Cin * Cout * Kd * Kh * Kw;
        uint64_t y_elems = (uint64_t)N * Cout * Dout * Hout * Wout;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(x), x_elems);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(w), w_elems);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(y), y_elems);

        // Queues for scalar operations implemented via 1-element vectors
        pipe.InitBuffer(inQueueX, 1, sizeof(DTYPE));
        pipe.InitBuffer(inQueueW, 1, sizeof(DTYPE));
        pipe.InitBuffer(tmpQueue, 1, sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, 1, sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        Compute();
    }

private:
    __aicore__ inline uint64_t XOffset(uint32_t b, uint32_t ic, uint32_t z, uint32_t y, uint32_t x) const
    {
        return (((((uint64_t)b * Cin + ic) * Din + z) * Hin + y) * Win + x);
    }

    __aicore__ inline uint64_t WOffset(uint32_t ic, uint32_t oc, uint32_t kz, uint32_t ky, uint32_t kx) const
    {
        return (((((uint64_t)ic * Cout + oc) * Kd + kz) * Kh + ky) * Kw + kx);
    }

    __aicore__ inline uint64_t YOffset(uint32_t b, uint32_t oc, uint32_t z, uint32_t y, uint32_t x) const
    {
        return (((((uint64_t)b * Cout + oc) * Dout + z) * Hout + y) * Wout + x);
    }

    __aicore__ inline void Compute()
    {
        // Prepare a zero local tensor for initializing accumulators
        AscendC::LocalTensor<DTYPE> zeroLocal = tmpQueue.AllocTensor<DTYPE>();
        // Load any one element then zero it by subtracting itself
        AscendC::DataCopy(zeroLocal, xGm[0], 1);
        AscendC::Sub(zeroLocal, zeroLocal, zeroLocal, 1);

        for (uint32_t b = 0; b < N; ++b) {
            for (uint32_t oc = 0; oc < Cout; ++oc) {
                for (uint32_t oz = 0; oz < Dout; ++oz) {
                    for (uint32_t oy = 0; oy < Hout; ++oy) {
                        for (uint32_t ox = 0; ox < Wout; ++ox) {
                            // sumLocal = 0
                            AscendC::LocalTensor<DTYPE> sumLocal = outQueueY.AllocTensor<DTYPE>();
                            AscendC::Sub(sumLocal, zeroLocal, zeroLocal, 1);

                            for (uint32_t ic = 0; ic < Cin; ++ic) {
                                // Loop over kernel
                                for (uint32_t kz = 0; kz < Kd; ++kz) {
                                    int32_t iz = (int32_t)oz - (int32_t)kz;
                                    if (iz < 0 || iz >= (int32_t)Din) continue;
                                    for (uint32_t ky = 0; ky < Kh; ++ky) {
                                        int32_t iy = (int32_t)oy - (int32_t)ky;
                                        if (iy < 0 || iy >= (int32_t)Hin) continue;
                                        for (uint32_t kx = 0; kx < Kw; ++kx) {
                                            int32_t ix = (int32_t)ox - (int32_t)kx;
                                            if (ix < 0 || ix >= (int32_t)Win) continue;

                                            uint64_t x_off = XOffset(b, ic, (uint32_t)iz, (uint32_t)iy, (uint32_t)ix);
                                            uint64_t w_off = WOffset(ic, oc, kz, ky, kx);

                                            AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
                                            AscendC::LocalTensor<DTYPE> wLocal = inQueueW.AllocTensor<DTYPE>();
                                            AscendC::LocalTensor<DTYPE> mulLocal = tmpQueue.AllocTensor<DTYPE>();

                                            AscendC::DataCopy(xLocal, xGm[x_off], 1);
                                            AscendC::DataCopy(wLocal, wGm[w_off], 1);
                                            AscendC::Mul(mulLocal, xLocal, wLocal, 1);
                                            AscendC::Add(sumLocal, sumLocal, mulLocal, 1);

                                            inQueueX.FreeTensor(xLocal);
                                            inQueueW.FreeTensor(wLocal);
                                            tmpQueue.FreeTensor(mulLocal);
                                        }
                                    }
                                }
                            }

                            uint64_t y_off = YOffset(b, oc, oz, oy, ox);
                            AscendC::DataCopy(yGm[y_off], sumLocal, 1);
                            outQueueY.FreeTensor(sumLocal);
                        }
                    }
                }
            }
        }

        tmpQueue.FreeTensor(zeroLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t N, Cin, Cout;
    uint32_t Din, Hin, Win;
    uint32_t Kd, Kh, Kw;
    uint32_t Dout, Hout, Wout;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvTransposed3D op;
    op.Init(x, w, y,
            tilingData.N(), tilingData.Cin(), tilingData.Cout(),
            tilingData.Din(), tilingData.Hin(), tilingData.Win(),
            tilingData.Kd(), tilingData.Kh(), tilingData.Kw());
    op.Process();
}

class ModelNew(torch.nn.Module):

    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: tuple = (1, 1, 1), padding: tuple = (0, 0, 0), output_padding: tuple = (0, 0, 0), groups: int = 1, bias: bool = False):
        super().__init__()
        assert stride == (1,1,1), "当前自定义算子仅支持 stride=(1,1,1)"
        assert padding == (0,0,0), "当前自定义算子仅支持 padding=(0,0,0)"
        assert output_padding == (0,0,0), "当前自定义算子仅支持 output_padding=(0,0,0)"
