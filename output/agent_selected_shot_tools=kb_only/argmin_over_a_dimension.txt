kernel_src="""
#include "kernel_operator.h"

// 输入float，输出int32：对最后一维做argmin（返回索引）
using DTYPE_X = float;
using DTYPE_Z = int32_t;

constexpr int32_t BUFFER_NUM = 1; // 每次搬运/计算的tile组数（简单起见设为1）

class KernelArgminLastDim {
public:
    __aicore__ inline KernelArgminLastDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, uint32_t totalOuter, uint32_t hLength, uint32_t tileNum)
    {
        this->hLength = hLength;
        this->tileNum = tileNum;

        // 每核处理的外部行数（即输出元素个数）
        this->blockOuter = totalOuter / AscendC::GetBlockNum();
        // 每tile处理的行数
        this->rowsPerTile = this->blockOuter / this->tileNum / BUFFER_NUM;
        if (this->rowsPerTile == 0) {
            this->rowsPerTile = 1; // 保底
        }

        // 为当前核设置全局指针区间
        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_X *>(x) + static_cast<uint64_t>(this->blockOuter) * this->hLength * AscendC::GetBlockIdx(),
                            static_cast<uint64_t>(this->blockOuter) * this->hLength);
        idxGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_Z *>(idx) + static_cast<uint64_t>(this->blockOuter) * AscendC::GetBlockIdx(),
                              this->blockOuter);

        // 队列buffer：输入tile为 rowsPerTile * hLength 的连续块；输出tile为 rowsPerTile 个int32
        pipe.InitBuffer(inQueueX, BUFFER_NUM, static_cast<uint64_t>(this->rowsPerTile) * this->hLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueIdx, BUFFER_NUM, static_cast<uint64_t>(this->rowsPerTile) * sizeof(DTYPE_Z));
        // ReduceMin需要的work空间，按文档示例，最小可取32个元素大小（以float计）
        pipe.InitBuffer(workQueue, BUFFER_NUM, 32 * sizeof(DTYPE_X));
        // ReduceMin的dst占位（仅用于触发计算；实际索引通过GetReduceMaxMinCount读取）
        pipe.InitBuffer(tmpDstQueue, BUFFER_NUM, 1 * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // 源数据是按 [outer, hLength] 连续排布（outer聚合了除最后一维外的维度）
        uint64_t gmOffset = static_cast<uint64_t>(progress) * this->rowsPerTile * this->hLength;
        AscendC::DataCopy(xLocal, xGm[gmOffset], static_cast<uint32_t>(this->rowsPerTile * this->hLength));
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> idxLocal = outQueueIdx.AllocTensor<DTYPE_Z>();

        // 对每一行（长度为hLength）做ReduceMin，获取最小值的索引
        for (uint32_t r = 0; r < rowsPerTile; ++r) {
            // 行起始offset
            uint64_t rowOff = static_cast<uint64_t>(r) * this->hLength;

            AscendC::LocalTensor<DTYPE_X> rowTensor = xLocal[rowOff];                // 行起始位置
            AscendC::LocalTensor<DTYPE_X> workLocal = workQueue.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> dstLocal = tmpDstQueue.AllocTensor<DTYPE_X>();

            // 调用ReduceMin，calIndex=true 以便后续读取索引
            AscendC::ReduceMin<DTYPE_X>(dstLocal, rowTensor, workLocal, this->hLength, true);

            // 从连续的ReduceMin场景读取最小值和索引
            DTYPE_X minVal = 0.0f;
            DTYPE_X idxToken = 0.0f;
            AscendC::GetReduceMaxMinCount<DTYPE_X>(minVal, idxToken);

            // 文档约束：索引以数据类型方式存储，需要reinterpret_cast为整型读取
            uint32_t &idxBits = reinterpret_cast<uint32_t &>(idxToken);
            idxLocal[r] = static_cast<DTYPE_Z>(idxBits);

            tmpDstQueue.FreeTensor(dstLocal);
            workQueue.FreeTensor(workLocal);
        }

        outQueueIdx.EnQue<DTYPE_Z>(idxLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> idxLocal = outQueueIdx.DeQue<DTYPE_Z>();
        uint64_t gmOffset = static_cast<uint64_t>(progress) * this->rowsPerTile;
        AscendC::DataCopy(idxGm[gmOffset], idxLocal, this->rowsPerTile);
        outQueueIdx.FreeTensor(idxLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueIdx;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> workQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> tmpDstQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> idxGm;

    uint32_t hLength;
    uint32_t tileNum;
    uint32_t blockOuter;
    uint32_t rowsPerTile;
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgminLastDim op;
    op.Init(x, idx, tiling_data.totalOuter, tiling_data.hLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC核进行加速：对指定维度执行argmin。
    内部通过维度变换将目标维度移动到最后一维，再调用自定义核（该核对最后一维做argmin）。
    """
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # 将需要argmin的维度移动到最后一维，便于核在连续内存上做归约
        red_dim = self.dim if self.dim >= 0 else (x.dim() + self.dim)
        if red_dim != x.dim() - 1:
            perm = [d for d in range(x.dim()) if d != red_dim] + [red_dim]
            x_t = x.permute(perm).contiguous()
        else:
            x_t = x

        # 调用自定义核（对最后一维求argmin），返回int32
        idx_int32 = custom_ops_lib.argmin_over_a_dimension_custom(x_t)

        # 为与PyTorch一致，转换为int64(Long)
        idx_long = idx_int32.to(torch.long)

        # 输出shape为去掉最后一维后的形状，已与期望一致
        return idx_long
'''
