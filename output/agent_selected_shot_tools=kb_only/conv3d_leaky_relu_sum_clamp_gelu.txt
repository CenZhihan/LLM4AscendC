kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConv3dLeakyReluSumClampGelu {
public:
    __aicore__ inline KernelConv3dLeakyReluSumClampGelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR sum_tensor, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)sum_tensor + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(bLocal, bGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // 1) LeakyReLU with negative_slope = 0.2
        const DTYPE_Z alpha = (DTYPE_Z)0.2f;
        AscendC::LeakyRelu(yLocal, xLocal, alpha, this->tileLength);

        // 2) Add sum tensor
        AscendC::Add(yLocal, yLocal, bLocal, this->tileLength);

        // 3) Clamp to [-1.0, 1.0] using scalar mins/maxs
        AscendC::Mins(yLocal, yLocal, (DTYPE_Z)1.0f, this->tileLength);   // y = min(y, 1.0)
        AscendC::Maxs(yLocal, yLocal, (DTYPE_Z)(-1.0f), this->tileLength); // y = max(y, -1.0)

        // 4) GELU activation (in-place: src and dst can reuse the same tensor)
        AscendC::Gelu(yLocal, yLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv3d_leaky_relu_sum_clamp_gelu_custom(GM_ADDR x, GM_ADDR sum_tensor, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dLeakyReluSumClampGelu op;
    op.Init(x, sum_tensor, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused post-conv kernel on AscendC: LeakyReLU (slope=0.2) + add sum_tensor + clamp[-1,1] + GELU.
    Conv3d remains as nn.Conv3d for weight handling; fused kernel handles element-wise stages efficiently.
    """
    def __init__(self, in_channels, out_channels, kernel_size, sum_tensor_shape):
        super().__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        # sum_tensor learned parameter of shape (C,1,1,1)
        self.sum_tensor = torch.nn.Parameter(torch.randn(sum_tensor_shape))

    def forward(self, x):
        y = self.conv(x)
        # Expand sum_tensor to match y (N,C,D,H,W) for the custom kernel
        C = y.shape[1]
        sum_broadcast = self.sum_tensor.view(1, C, 1, 1, 1).expand_as(y)
        return custom_ops_lib.conv3d_leaky_relu_sum_clamp_gelu_custom(y, sum_broadcast)
