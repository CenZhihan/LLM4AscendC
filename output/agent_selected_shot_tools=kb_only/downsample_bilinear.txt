#include "downsample_bilinear_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
constexpr uint32_t TARGET_H = 60;
constexpr uint32_t TARGET_W = 80;
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    DownsampleBilinearCustomTilingData tiling;

    // Input shape: NCHW
    const gert::Shape* in_shape = context->GetInputShape(0);
    uint64_t N = in_shape->GetOriginShape().GetDim(0);
    uint64_t C = in_shape->GetOriginShape().GetDim(1);
    uint64_t Hin = in_shape->GetOriginShape().GetDim(2);
    uint64_t Win = in_shape->GetOriginShape().GetDim(3);

    // Output shape (N, C, 60, 80)
    uint64_t Hout = TARGET_H;
    uint64_t Wout = TARGET_W;

    uint32_t totalNC = static_cast<uint32_t>(N * C);
    uint32_t blockNum = 8;
    if (blockNum > totalNC) blockNum = totalNC == 0 ? 1 : totalNC;
    context->SetBlockDim(blockNum);

    uint32_t ncPerBlock = (totalNC + blockNum - 1) / blockNum;

    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_c(static_cast<uint32_t>(C));
    tiling.set_hin(static_cast<uint32_t>(Hin));
    tiling.set_win(static_cast<uint32_t>(Win));
    tiling.set_hout(static_cast<uint32_t>(Hout));
    tiling.set_wout(static_cast<uint32_t>(Wout));
    tiling.set_ncPerBlock(ncPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    gert::Shape out_s = *x_shape;
    out_s.SetDim(2, TARGET_H);
    out_s.SetDim(3, TARGET_W);
    *y_shape = out_s;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    auto dt = context->GetInputDataType(0);
    context->SetOutputDataType(0, dt);
    return GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

class KernelDownsampleBilinear {
public:
    __aicore__ inline KernelDownsampleBilinear() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t hin, uint32_t win,
                                uint32_t hout, uint32_t wout,
                                uint32_t ncPerBlock)
    {
        this->n = n;
        this->c = c;
        this->hin = hin;
        this->win = win;
        this->hout = hout;
        this->wout = wout;
        this->ncPerBlock = ncPerBlock;

        this->inPlaneSize = hin * win;
        this->outPlaneSize = hout * wout;
        this->totalNC = n * c;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x), (uint64_t)totalNC * inPlaneSize);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(y), (uint64_t)totalNC * outPlaneSize);

        pipe.InitBuffer(inQueue, 1, inPlaneSize * sizeof(float));
        pipe.InitBuffer(outQueue, 1, outPlaneSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startNc = blockIdx * ncPerBlock;
        uint32_t endNc = startNc + ncPerBlock;
        if (endNc > totalNC) endNc = totalNC;

        for (uint32_t ncIdx = startNc; ncIdx < endNc; ++ncIdx) {
            CopyIn(ncIdx);
            Compute();
            CopyOut(ncIdx);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t ncIdx)
    {
        AscendC::LocalTensor<float> srcLocal = inQueue.AllocTensor<float>();
        AscendC::DataCopy(srcLocal, xGm[(uint64_t)ncIdx * inPlaneSize], inPlaneSize);
        inQueue.EnQue(srcLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> srcLocal = inQueue.DeQue<float>();
        AscendC::LocalTensor<float> dstLocal = outQueue.AllocTensor<float>();

        // 使用芯片内置的双线性插值能力（垂直、水平迭代）
        // 对应参考文档的 ISASI API: BilinearInterpolation
        // 目标尺寸: (hout, wout), 输入尺寸: (hin, win), align_corners = false
        AscendC::BilinearInterpolation(dstLocal, srcLocal, hin, win, hout, wout, false);

        outQueue.EnQue<float>(dstLocal);
        inQueue.FreeTensor(srcLocal);
    }

    __aicore__ inline void CopyOut(uint32_t ncIdx)
    {
        AscendC::LocalTensor<float> dstLocal = outQueue.DeQue<float>();
        AscendC::DataCopy(yGm[(uint64_t)ncIdx * outPlaneSize], dstLocal, outPlaneSize);
        outQueue.FreeTensor(dstLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n, c;
    uint32_t hin, win, hout, wout;
    uint32_t inPlaneSize, outPlaneSize;
    uint32_t totalNC;
    uint32_t ncPerBlock;
};

extern "C" __global__ __aicore__ void downsample_bilinear_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelDownsampleBilinear op;
    op.Init(x, y,
            tilingData.n, tilingData.c,
            tilingData.hin, tilingData.win,
            tilingData.hout, tilingData.wout,
            tilingData.ncPerBlock);
    op.Process();
}
"""
