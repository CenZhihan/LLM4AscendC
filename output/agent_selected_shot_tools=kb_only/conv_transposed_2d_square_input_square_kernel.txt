kernel_src="""
#include "kernel_operator.h"

using DTYPE_IN = float;
using DTYPE_W  = float;
using DTYPE_OUT= float;

class KernelConvTransposed2d {
public:
    __aicore__ inline KernelConvTransposed2d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t tileH, uint32_t tileW, uint32_t ocPerBlock)
    {
        this->N = N; this->Cin = Cin; this->Cout = Cout;
        this->H = H; this->W = W; this->K = K;
        this->tileH = tileH; this->tileW = tileW;
        this->Hout = H + K - 1; this->Wout = W + K - 1;
        this->ocPerBlock = ocPerBlock;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_IN *>(x));
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_W  *>(w));
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_OUT*>(y));

        // Buffers: xTile (tileH*tileW), yLocal ( (tileH+K-1)*(tileW+K-1) ), yAcc (same), wLocal (K*K)
        uint32_t maxTileOH = tileH + K - 1;
        uint32_t maxTileOW = tileW + K - 1;

        pipe.InitBuffer(xQueue, 1, tileH * tileW * sizeof(DTYPE_IN));
        pipe.InitBuffer(yQueue, 1, maxTileOH * maxTileOW * sizeof(DTYPE_OUT));
        pipe.InitBuffer(accQueue, 1, maxTileOH * maxTileOW * sizeof(DTYPE_OUT));
        pipe.InitBuffer(wQueue, 1, K * K * sizeof(DTYPE_W));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t ocStart = blockIdx * ocPerBlock;
        uint32_t ocEnd = ocStart + ocPerBlock;
        if (ocStart >= Cout) return;
        if (ocEnd > Cout) ocEnd = Cout;

        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oc = ocStart; oc < ocEnd; ++oc) {
                for (uint32_t ty = 0; ty < H; ty += tileH) {
                    uint32_t curTileH = AscendC::Min(tileH, H - ty);
                    uint32_t curTileOH = curTileH + K - 1;
                    for (uint32_t tx = 0; tx < W; tx += tileW) {
                        uint32_t curTileW = AscendC::Min(tileW, W - tx);
                        uint32_t curTileOW = curTileW + K - 1;

                        // Allocate locals
                        AscendC::LocalTensor<DTYPE_OUT> yLocal = yQueue.AllocTensor<DTYPE_OUT>();
                        AscendC::LocalTensor<DTYPE_OUT> yAccLocal = accQueue.AllocTensor<DTYPE_OUT>();
                        AscendC::LocalTensor<DTYPE_IN>  xLocal = xQueue.AllocTensor<DTYPE_IN>();
                        AscendC::LocalTensor<DTYPE_W>   wLocal = wQueue.AllocTensor<DTYPE_W>();

                        // zero yLocal
                        AscendC::Duplicate<DTYPE_OUT>(yLocal, static_cast<DTYPE_OUT>(0), curTileOH * curTileOW);

                        // Accumulate over input channels
                        for (uint32_t ic = 0; ic < Cin; ++ic) {
                            // Load x tile [curTileH x curTileW] from GM
                            for (uint32_t r = 0; r < curTileH; ++r) {
                                uint64_t xBase = ((((uint64_t)n * Cin + ic) * H + (ty + r)) * W + tx);
                                AscendC::DataCopy(xLocal[r * curTileW], xGm[xBase], curTileW);
                            }
                            // Load weight [K x K] for (ic, oc)
                            {
                                uint64_t wBase = (((uint64_t)ic * Cout + oc) * K) * K;
                                AscendC::DataCopy(wLocal, wGm[wBase], K * K);
                            }

                            // Compute yLocal += xLocal (*) wLocal (transposed-conv scatter add)
                            for (uint32_t r = 0; r < curTileH; ++r) {
                                for (uint32_t c = 0; c < curTileW; ++c) {
                                    DTYPE_IN xv = xLocal.GetValue(r * curTileW + c);
                                    if (xv == (DTYPE_IN)0) { continue; }
                                    for (uint32_t ky = 0; ky < K; ++ky) {
                                        for (uint32_t kx = 0; kx < K; ++kx) {
                                            DTYPE_W wv = wLocal.GetValue(ky * K + kx);
                                            uint32_t yr = r + ky;
                                            uint32_t yc = c + kx;
                                            uint32_t yidx = yr * curTileOW + yc;
                                            DTYPE_OUT acc = yLocal.GetValue(yidx);
                                            yLocal.SetValue(yidx, acc + xv * wv);
                                        }
                                    }
                                }
                            }
                        } // ic

                        // Load destination region to yAccLocal, accumulate, then write back
                        for (uint32_t rr = 0; rr < curTileOH; ++rr) {
                            uint64_t yBase = ((((uint64_t)n * Cout + oc) * Hout + (ty + rr)) * Wout + tx);
                            AscendC::DataCopy(yAccLocal[rr * curTileOW], yGm[yBase], curTileOW);
                        }
                        // yAccLocal = yAccLocal + yLocal
                        AscendC::Add(yAccLocal, yAccLocal, yLocal, curTileOH * curTileOW);

                        // Store back
                        for (uint32_t rr = 0; rr < curTileOH; ++rr) {
                            uint64_t yBase = ((((uint64_t)n * Cout + oc) * Hout + (ty + rr)) * Wout + tx);
                            AscendC::DataCopy(yGm[yBase], yAccLocal[rr * curTileOW], curTileOW);
                        }

                        // Free locals
                        yQueue.FreeTensor(yLocal);
                        accQueue.FreeTensor(yAccLocal);
                        xQueue.FreeTensor(xLocal);
                        wQueue.FreeTensor(wLocal);
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> xQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> wQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> yQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> accQueue;

    AscendC::GlobalTensor<DTYPE_IN>  xGm;
    AscendC::GlobalTensor<DTYPE_W>   wGm;
    AscendC::GlobalTensor<DTYPE_OUT> yGm;

    uint32_t N, Cin, Cout, H, W, K;
    uint32_t Hout, Wout;
    uint32_t tileH, tileW;
    uint32_t ocPerBlock;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvTransposed2d op;
    op.Init(x, weight, y,
            tilingData.N(),
            tilingData.Cin(),
            tilingData.Cout(),
            tilingData.H(),
            tilingData.W(),
            tilingData.K(),
            tilingData.tileH(),
            tilingData.tileW(),
            tilingData.ocPerBlock());
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom ConvTranspose2d for square input and square kernel on Ascend using AscendC kernel.
    Assumptions: stride=1, padding=0, output_padding=0, groups=1, bias=False.
    Weight layout: [in_channels, out_channels, k, k] to align with PyTorch ConvTranspose2d.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        # Parameter shape matches PyTorch ConvTranspose2d weight layout: [Cin, Cout, K, K]
        weight = torch.empty(in_channels, out_channels, kernel_size, kernel_size)
        torch.nn.init.kaiming_uniform_(weight, a=5 ** 0.5)
        self.weight = torch.nn.Parameter(weight)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_2d_square_input_square_kernel_custom(x, self.weight)
