    return (a + b - 1) / b;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ConvDepthwise2dSquareInputSquareKernelCustomTilingData tiling;

    const gert::Shape& xShape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape& wShape = context->GetInputShape(1)->GetOriginShape();

    uint32_t n = static_cast<uint32_t>(xShape.GetDim(0));
    uint32_t c = static_cast<uint32_t>(xShape.GetDim(1));
    uint32_t h = static_cast<uint32_t>(xShape.GetDim(2));
    uint32_t w = static_cast<uint32_t>(xShape.GetDim(3));

    // weight shape could be [C, 1, K, K]
    uint32_t k = static_cast<uint32_t>(wShape.GetDim(wShape.GetDimNum() - 1)); // assume square kernel

    uint32_t stride = 1;   // this custom kernel assumes stride=1
    uint32_t padding = 0;  // and padding=0

    uint32_t hout = (h + 2 * padding - k) / stride + 1;
    uint32_t wout = (w + 2 * padding - k) / stride + 1;

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_k(k);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_hout(hout);
    tiling.set_wout(wout);

    uint32_t totalTasks = n * c;  // depthwise: each (n,c) is an independent task
    const uint32_t MAX_BLOCKS = 32;
    uint32_t blockDim = totalTasks < MAX_BLOCKS ? totalTasks : MAX_BLOCKS;
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);
    uint32_t tasksPerBlock = CeilDiv(totalTasks, blockDim);
    tiling.set_tasksPerBlock(tasksPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

class KernelDepthwiseConv2D {
public:
    __aicore__ inline KernelDepthwiseConv2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t k, uint32_t stride, uint32_t padding,
                                uint32_t hout, uint32_t wout, uint32_t tasksPerBlock)
    {
        this->N = n;
        this->C = c;
        this->H = h;
        this->W = w_in;
        this->K = k;
        this->S = stride;
        this->P = padding;
        this->HOut = hout;
        this->WOut = wout;
        this->tasksPerBlock = tasksPerBlock;

        uint32_t xSize = N * C * H * W;
        uint32_t wSize = C * K * K;
        uint32_t ySize = N * C * HOut * WOut;

        xGm.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe.InitBuffer(inQueueSeg, 1, WOut * sizeof(float));
        pipe.InitBuffer(tmpQueue, 1, WOut * sizeof(float));
        pipe.InitBuffer(outRowQueue, 1, WOut * sizeof(float));
        pipe.InitBuffer(weightQueue, 1, K * K * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t totalTasks = N * C;
        uint32_t startTask = blockIdx * tasksPerBlock;
        uint32_t endTask = startTask + tasksPerBlock;
        if (endTask > totalTasks) {
            endTask = totalTasks;
        }
        if (startTask >= endTask) {
            return;
        }

        for (uint32_t task = startTask; task < endTask; ++task) {
            uint32_t n = task / C;
            uint32_t c = task % C;

            // load weights for channel c
            AscendC::LocalTensor<float> wLocal = weightQueue.AllocTensor<float>();
            uint32_t wOffset = c * K * K;
            AscendC::DataCopy(wLocal, wGm[wOffset], K * K);

            for (uint32_t oh = 0; oh < HOut; ++oh) {
                bool first = true;

                // allocate out row buffer
                AscendC::LocalTensor<float> outRowLocal = outRowQueue.AllocTensor<float>();

                for (uint32_t kr = 0; kr < K; ++kr) {
                    uint32_t inRow = oh * S + kr;
                    uint32_t baseInRowOffset = ((n * C + c) * H + inRow) * W;

                    for (uint32_t kc = 0; kc < K; ++kc) {
                        // segment starts at column kc, length WOut (stride = 1 assumed)
                        AscendC::LocalTensor<float> inSegLocal = inQueueSeg.AllocTensor<float>();
                        AscendC::DataCopy(inSegLocal, xGm[baseInRowOffset + kc], WOut);

                        AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();
                        // multiply by scalar weight
                        float wScalar = wLocal[kr * K + kc];
                        AscendC::Muls(tmpLocal, inSegLocal, wScalar, WOut);

                        if (first) {
                            AscendC::DataCopy(outRowLocal, tmpLocal, WOut);
                            first = false;
                        } else {
                            AscendC::Add(outRowLocal, outRowLocal, tmpLocal, WOut);
                        }

                        tmpQueue.FreeTensor(tmpLocal);
                        inQueueSeg.FreeTensor(inSegLocal);
                    }
                }

                // write back the out row
                uint32_t yOffset = ((n * C + c) * HOut + oh) * WOut;
                AscendC::DataCopy(yGm[yOffset], outRowLocal, WOut);
                outRowQueue.FreeTensor(outRowLocal);
            }

            weightQueue.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueSeg;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> weightQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outRowQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> tmpQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t N, C, H, W, K, S, P, HOut, WOut;
    uint32_t tasksPerBlock;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDepthwiseConv2D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.k, tiling_data.stride, tiling_data.padding,
            tiling_data.hout, tiling_data.wout, tiling_data.tasksPerBlock);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Depthwise Conv2d with custom AscendC kernel (square input and square kernel, stride=1, padding=0).
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        # Depthwise weight: [C, 1, K, K]
        self.weight = torch.nn.Parameter(torch.randn(in_channels, 1, kernel_size, kernel_size))
        if bias:
            # This custom op does not support bias. Keep it for API parity but not used.
            self.bias = torch.nn.Parameter(torch.zeros(in_channels))
        else:
            self.register_parameter("bias", None)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Only stride=1, padding=0 is supported
        return custom_ops_lib.conv_depthwise_2d_square_input_square_kernel_custom(x, self.weight)
