kernel_src="""
#include "kernel_operator.h"

// 本算子接收 x:[N, C, 1, 1] 和 bias:[C, 1, 1]，计算：y = x + bias（广播到N）
// 然后对每个样本n执行 logsumexp(y[n, :, 1, 1])，最终乘以10，输出 z:[N, 1]

constexpr int32_t BUFFER_NUM = 2;

using DTYPE = float;

class KernelConvBiasLogsumexpMul {
public:
    __aicore__ inline KernelConvBiasLogsumexpMul() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z, uint32_t N, uint32_t C)
    {
        this->N = N;
        this->C = C;
        this->blockN = (N + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        // 为了均衡，使用起始偏移
        uint32_t start_n = blockN * AscendC::GetBlockIdx();
        if (start_n > N) { start_n = N; }
        this->validN = (start_n + blockN <= N) ? blockN : (N - start_n);
        this->startN = start_n;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(x) + startN * C, validN * C);
        biasGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(bias), C);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(z) + startN, validN);

        // 队列初始化
        pipe.InitBuffer(inQueueX, BUFFER_NUM, C * sizeof(DTYPE));          // 每次处理一个样本的C向量
        pipe.InitBuffer(inQueueBias, 1, C * sizeof(DTYPE));                // bias向量
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, C * sizeof(DTYPE));          // 中间缓存
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, sizeof(DTYPE));             // 单元素输出
    }

    __aicore__ inline void Process()
    {
        // 先将bias搬入UB
        AscendC::LocalTensor<DTYPE> biasLocal = inQueueBias.AllocTensor<DTYPE>();
        AscendC::DataCopy(biasLocal, biasGm, C);
        inQueueBias.EnQue(biasLocal);

        for (uint32_t n = 0; n < validN; ++n) {
            CopyIn(n);
            Compute(n);
            CopyOut(n);
        }

        // 释放biasLocal
        AscendC::LocalTensor<DTYPE> biasLocalDeq = inQueueBias.DeQue<DTYPE>();
        inQueueBias.FreeTensor(biasLocalDeq);
    }

private:
    __aicore__ inline void CopyIn(uint32_t nIdx)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[nIdx * C], C);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t nIdx)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> biasLocal = inQueueBias.DeQue<DTYPE>(); // 不释放，复用
        AscendC::LocalTensor<DTYPE> yLocal = tmpQueue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> expLocal = tmpQueue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> sumLocal = outQueueZ.AllocTensor<DTYPE>(); // 用作中间1元素，后续再复用为输出

        // y = x + bias
        AscendC::Add(yLocal, xLocal, biasLocal, C);

        // exp(y)
        AscendC::Exp(expLocal, yLocal, C);

        // sum(exp(y))（在最后一维上求和）
        AscendC::Sum(sumLocal, expLocal, C);

        // ln(sum(exp(y)))
        AscendC::Ln(sumLocal, sumLocal, 1);

        // multiply by 10
        AscendC::Muls(sumLocal, sumLocal, static_cast<DTYPE>(10.0f), 1);

        // 将结果入队
        outQueueZ.EnQue(sumLocal);

        // 释放临时与输入
        tmpQueue.FreeTensor(yLocal);
        tmpQueue.FreeTensor(expLocal);
        inQueueX.FreeTensor(xLocal);
        // biasLocal不释放，继续使用
        inQueueBias.EnQue(biasLocal);
    }

    __aicore__ inline void CopyOut(uint32_t nIdx)
    {
        AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.DeQue<DTYPE>();
        AscendC::DataCopy(zGm[nIdx], zLocal, 1);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueBias;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> biasGm;
    AscendC::GlobalTensor<DTYPE> zGm;

    uint32_t N;
    uint32_t C;
    uint32_t blockN;
    uint32_t validN;
    uint32_t startN;
};

extern "C" __global__ __aicore__ void convtranspose2d_globalavgpool_biasadd_logsumexp_sum_multiply_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvBiasLogsumexpMul op;
    op.Init(x, bias, z, tiling_data.N, tiling_data.C);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版：保持 PyTorch 的转置卷积与全局平均池化，
    之后使用自定义 AscendC 算子融合 bias add + logsumexp + sum + multiply。
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        # 1) ConvTranspose2d
        x = self.conv_transpose(x)
        # 2) Global Average Pooling -> [N, C, 1, 1]
        x = torch.mean(x, dim=(2, 3), keepdim=True)
        # 3) 使用自定义算子完成：x + bias -> logsumexp(dim=1, keepdim=True) -> sum(spatial) -> *10
        #    输出形状为 [N, 1]
        return custom_ops_lib.convtranspose2d_globalavgpool_biasadd_logsumexp_sum_multiply_custom(x, self.bias)
