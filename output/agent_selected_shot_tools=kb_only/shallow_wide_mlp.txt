kernel_src="""
#include "kernel_operator.h"

namespace {
struct MlpTiling {
    uint32_t batchSize;
    uint32_t inputSize;
    uint32_t hidden1Size;
    uint32_t hidden2Size;
    uint32_t outputSize;
};
}

class KernelShallowWideMlp {
public:
    __aicore__ inline KernelShallowWideMlp() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w0, GM_ADDR b0, GM_ADDR w1, GM_ADDR b1, GM_ADDR w2, GM_ADDR b2, GM_ADDR z,
                                GM_ADDR workspace, const MlpTiling &t)
    {
        tiling = t;

        xPtr = reinterpret_cast<__gm__ float*>(x);
        w0Ptr = reinterpret_cast<__gm__ float*>(w0);
        b0Ptr = reinterpret_cast<__gm__ float*>(b0);
        w1Ptr = reinterpret_cast<__gm__ float*>(w1);
        b1Ptr = reinterpret_cast<__gm__ float*>(b1);
        w2Ptr = reinterpret_cast<__gm__ float*>(w2);
        b2Ptr = reinterpret_cast<__gm__ float*>(b2);
        zPtr  = reinterpret_cast<__gm__ float*>(z);

        // workspace layout: y0 [batch*hidden1], y1 [batch*hidden2]
        __gm__ float* ws = reinterpret_cast<__gm__ float*>(workspace);
        y0Ptr = ws;
        y1Ptr = ws + static_cast<uint32_t>(tiling.batchSize) * tiling.hidden1Size;
    }

    __aicore__ inline void Process()
    {
        // Layer 0: y0 = ReLU(x @ w0 + b0)
        MatMulAddBiasRelu(xPtr, w0Ptr, b0Ptr, y0Ptr, tiling.batchSize, tiling.inputSize, tiling.hidden1Size);

        // Layer 1: y1 = ReLU(y0 @ w1 + b1)
        MatMulAddBiasRelu(y0Ptr, w1Ptr, b1Ptr, y1Ptr, tiling.batchSize, tiling.hidden1Size, tiling.hidden2Size);

        // Layer 2: z = y1 @ w2 + b2
        MatMulAddBias(y1Ptr, w2Ptr, b2Ptr, zPtr, tiling.batchSize, tiling.hidden2Size, tiling.outputSize);
    }

private:
    __aicore__ inline void MatMulAddBiasRelu(__gm__ float* A, __gm__ float* B, __gm__ float* bias, __gm__ float* C,
                                             uint32_t M, uint32_t K, uint32_t N)
    {
        // Compute C[M,N] = A[M,K] x B[K,N] + bias[N], then ReLU
        for (uint32_t m = 0; m < M; ++m) {
            for (uint32_t n = 0; n < N; ++n) {
                float acc = 0.0f;
                const __gm__ float* aRow = A + m * K;
                const __gm__ float* bCol = B + n; // column-major access via stride N
                for (uint32_t k = 0; k < K; ++k) {
                    acc += aRow[k] * bCol[k * N];
                }
                acc += bias[n];
                // ReLU
                C[m * N + n] = acc > 0.0f ? acc : 0.0f;
            }
        }
    }

    __aicore__ inline void MatMulAddBias(__gm__ float* A, __gm__ float* B, __gm__ float* bias, __gm__ float* C,
                                         uint32_t M, uint32_t K, uint32_t N)
    {
        for (uint32_t m = 0; m < M; ++m) {
            for (uint32_t n = 0; n < N; ++n) {
                float acc = 0.0f;
                const __gm__ float* aRow = A + m * K;
                const __gm__ float* bCol = B + n; // column-major via stride N
                for (uint32_t k = 0; k < K; ++k) {
                    acc += aRow[k] * bCol[k * N];
                }
                acc += bias[n];
                C[m * N + n] = acc;
            }
        }
    }

private:
    MlpTiling tiling;
    __gm__ float* xPtr;
    __gm__ float* w0Ptr;
    __gm__ float* b0Ptr;
    __gm__ float* w1Ptr;
    __gm__ float* b1Ptr;
    __gm__ float* w2Ptr;
    __gm__ float* b2Ptr;
    __gm__ float* zPtr;
    __gm__ float* y0Ptr;
    __gm__ float* y1Ptr;
};

extern "C" __global__ __aicore__ void shallow_wide_mlp_custom(GM_ADDR x, GM_ADDR w0, GM_ADDR b0,
                                                               GM_ADDR w1, GM_ADDR b1,
                                                               GM_ADDR w2, GM_ADDR b2,
                                                               GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingDataRaw, tiling);
    MlpTiling t;
    t.batchSize = tilingDataRaw.batchSize;
    t.inputSize = tilingDataRaw.inputSize;
    t.hidden1Size = tilingDataRaw.hidden1Size;
    t.hidden2Size = tilingDataRaw.hidden2Size;
    t.outputSize = tilingDataRaw.outputSize;

    KernelShallowWideMlp op;
    op.Init(x, w0, b0, w1, b1, w2, b2, z, workspace, t);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_layer_sizes, output_size):
        super(ModelNew, self).__init__()
        layers = []
        current_input_size = input_size
        for hidden_size in hidden_layer_sizes:
            linear = torch.nn.Linear(current_input_size, hidden_size)
            layers.append(linear)
            layers.append(torch.nn.ReLU())
            current_input_size = hidden_size
        layers.append(torch.nn.Linear(current_input_size, output_size))
        self.network = torch.nn.Sequential(*layers)

    def forward(self, x):
        # Extract weights/biases and transpose weights to shape [in_features, out_features]
        w0 = self.network[0].weight.t().contiguous()
        b0 = self.network[0].bias.contiguous()
        w1 = self.network[2].weight.t().contiguous()
        b1 = self.network[2].bias.contiguous()
        w2 = self.network[4].weight.t().contiguous()
        b2 = self.network[4].bias.contiguous()
        return custom_ops_lib.shallow_wide_mlp_custom(x, w0, b0, w1, b1, w2, b2)
