#include "index_copy_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 1;
const uint32_t TILE_NUM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    IndexCopyCustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::StorageShape *idx_shape = context->GetInputShape(1);

    uint32_t totalRows = 1;
    uint32_t innerSize = 1;
    if (x_shape->GetStorageShape().GetDimNum() >= 2) {
        totalRows = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(0));
        innerSize = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(1));
    } else {
        totalRows = static_cast<uint32_t>(x_shape->GetStorageShape().GetShapeSize());
        innerSize = 1;
    }
    uint32_t indexSize = static_cast<uint32_t>(idx_shape->GetOriginShape().GetShapeSize());

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalRows(totalRows);
    tiling.set_innerSize(innerSize);
    tiling.set_indexSize(indexSize);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {

{
    const gert::Shape* x_shape = context->GetInputShape(0);

kernel_src="""
#include "kernel_operator.h"

using DTYPE_T = float;
using DTYPE_INDEX = uint32_t;

constexpr int32_t BUFFER_NUM = 1;

class KernelIndexCopy {
public:
    __aicore__ inline KernelIndexCopy() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, GM_ADDR src, GM_ADDR y,
                                uint32_t totalRows, uint32_t innerSize, uint32_t indexSize, uint32_t tileNum)
    {
        this->totalRows = totalRows;
        this->innerSize = innerSize;
        this->indexSize = indexSize;
        (void)tileNum;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_T*>(x));
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_T*>(y));
        srcGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_T*>(src));
        idxGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_INDEX*>(indices));

        pipe.InitBuffer(inQueueXRow, BUFFER_NUM, this->innerSize * sizeof(DTYPE_T));
        pipe.InitBuffer(inQueueSrcRow, BUFFER_NUM, this->innerSize * sizeof(DTYPE_T));
        pipe.InitBuffer(outQueueYRow, BUFFER_NUM, this->innerSize * sizeof(DTYPE_T));
        pipe.InitBuffer(inQueueIdx, 1, this->indexSize * sizeof(DTYPE_INDEX));
    }

    __aicore__ inline void Process()
    {
        // Load all indices into UB once
        AscendC::LocalTensor<DTYPE_INDEX> idxLocal = inQueueIdx.AllocTensor<DTYPE_INDEX>();
        AscendC::DataCopy(idxLocal, idxGm, indexSize);
        inQueueIdx.EnQue(idxLocal);

        // Step 1: y = x (row by row)
        for (uint32_t r = 0; r < totalRows; ++r) {
            CopyRowXIn(r);
            CopyRowXToY();
            CopyRowYOut(r);
        }

        // Step 2: overwrite rows specified by indices with src rows
        AscendC::LocalTensor<DTYPE_INDEX> idxLocalRd = inQueueIdx.DeQue<DTYPE_INDEX>();
        for (uint32_t t = 0; t < indexSize; ++t) {
            // 取出第 t 个索引（假设可通过下标读取）
            DTYPE_INDEX idxVal = idxLocalRd[t];

            // 读取 src 第 t 行
            AscendC::LocalTensor<DTYPE_T> srcLocal = inQueueSrcRow.AllocTensor<DTYPE_T>();
            AscendC::DataCopy(srcLocal, srcGm[t * innerSize], innerSize);
            inQueueSrcRow.EnQue(srcLocal);

            // 写入到 y 的 idxVal 行
            AscendC::LocalTensor<DTYPE_T> srcLocalRd = inQueueSrcRow.DeQue<DTYPE_T>();
            AscendC::LocalTensor<DTYPE_T> yLocal = outQueueYRow.AllocTensor<DTYPE_T>();
            AscendC::DataCopy(yLocal, srcLocalRd, innerSize);
            outQueueYRow.EnQue<DTYPE_T>(yLocal);
            inQueueSrcRow.FreeTensor(srcLocalRd);

            AscendC::LocalTensor<DTYPE_T> yLocalWr = outQueueYRow.DeQue<DTYPE_T>();
            AscendC::DataCopy(yGm[idxVal * innerSize], yLocalWr, innerSize);
            outQueueYRow.FreeTensor(yLocalWr);
        }
        inQueueIdx.FreeTensor(idxLocalRd);
    }

private:
    __aicore__ inline void CopyRowXIn(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_T> xLocal = inQueueXRow.AllocTensor<DTYPE_T>();
        AscendC::DataCopy(xLocal, xGm[row * innerSize], innerSize);
        inQueueXRow.EnQue(xLocal);
    }
    __aicore__ inline void CopyRowXToY()
    {
        AscendC::LocalTensor<DTYPE_T> xLocal = inQueueXRow.DeQue<DTYPE_T>();
        AscendC::LocalTensor<DTYPE_T> yLocal = outQueueYRow.AllocTensor<DTYPE_T>();
        AscendC::DataCopy(yLocal, xLocal, innerSize);
        outQueueYRow.EnQue<DTYPE_T>(yLocal);
        inQueueXRow.FreeTensor(xLocal);
    }
    __aicore__ inline void CopyRowYOut(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_T> yLocal = outQueueYRow.DeQue<DTYPE_T>();
        AscendC::DataCopy(yGm[row * innerSize], yLocal, innerSize);
        outQueueYRow.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueXRow;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueSrcRow;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueYRow;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueIdx;

    AscendC::GlobalTensor<DTYPE_T> xGm;
    AscendC::GlobalTensor<DTYPE_T> yGm;
    AscendC::GlobalTensor<DTYPE_T> srcGm;
    AscendC::GlobalTensor<DTYPE_INDEX> idxGm;

    uint32_t totalRows;
    uint32_t innerSize;
    uint32_t indexSize;
};

extern "C" __global__ __aicore__ void index_copy_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR src, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelIndexCopy op;
    op.Init(x, indices, src, y,
            tiling_data.totalRows, tiling_data.innerSize, tiling_data.indexSize, tiling_data.tileNum);
    op.Process();
}
