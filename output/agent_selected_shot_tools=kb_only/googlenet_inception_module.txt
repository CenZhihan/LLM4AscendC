#include "googlenet_inception_module_custom_tiling.h"

namespace {
inline uint64_t GetShapeSize(const gert::Shape* s) {
    return s->GetOriginShape().GetShapeSize();
}
// Try to get N, C, H, W from shape; fall back to 4D assumptions.
inline void GetNCHW(const gert::Shape* s, int64_t& n, int64_t& c, int64_t& h, int64_t& w) {
    auto origin = s->GetOriginShape();
    int32_t dimNum = origin.GetDimNum();
    if (dimNum >= 4) {
        n = origin.GetDim(0);
        c = origin.GetDim(1);
        h = origin.GetDim(2);
        w = origin.GetDim(3);
    } else {
        // Fallback: treat as 4D with leading dims as 1
        n = 1;
        c = origin.GetShapeSize();
        h = 1;
        w = 1;
    }
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 1;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GooglenetInceptionModuleCustomTilingData tiling;

    const gert::Shape* s0 = context->GetInputShape(0);
    const gert::Shape* s1 = context->GetInputShape(1);
    const gert::Shape* s2 = context->GetInputShape(2);
    const gert::Shape* s3 = context->GetInputShape(3);

    int64_t n0, c0, h0, w0;
    int64_t n1, c1, h1, w1;
    int64_t n2, c2, h2, w2;
    int64_t n3, c3, h3, w3;
    GetNCHW(s0, n0, c0, h0, w0);
    GetNCHW(s1, n1, c1, h1, w1);
    GetNCHW(s2, n2, c2, h2, w2);
    GetNCHW(s3, n3, c3, h3, w3);

    // Basic consistency check (N/H/W should match)
    // If not matched, runtime will report error by infer shape or kernel boundary checks.
    uint32_t n = static_cast<uint32_t>(n0);
    uint32_t ps1 = static_cast<uint32_t>(c0 * h0 * w0);
    uint32_t ps2 = static_cast<uint32_t>(c1 * h1 * w1);
    uint32_t ps3 = static_cast<uint32_t>(c2 * h2 * w2);
    uint32_t ps4 = static_cast<uint32_t>(c3 * h3 * w3);

    uint64_t totalPerSample = static_cast<uint64_t>(ps1) + ps2 + ps3 + ps4;
    uint64_t totalLength = totalPerSample * n;

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_n(n);
    tiling.set_ps1(ps1);
    tiling.set_ps2(ps2);
    tiling.set_ps3(ps3);
    tiling.set_ps4(ps4);
    tiling.set_totalLength(static_cast<uint32_t>(totalLength)); // assume fits
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* s0 = context->GetInputShape(0);
    const gert::Shape* s1 = context->GetInputShape(1);
    const gert::Shape* s2 = context->GetInputShape(2);
    const gert::Shape* s3 = context->GetInputShape(3);
    int64_t n0, c0, h0, w0;
    int64_t n1, c1, h1, w1;
    int64_t n2, c2, h2, w2;
    int64_t n3, c3, h3, w3;
    GetNCHW(s0, n0, c0, h0, w0);
    GetNCHW(s1, n1, c1, h1, w1);
    GetNCHW(s2, n2, c2, h2, w2);
    GetNCHW(s3, n3, c3, h3, w3);
    if (!(n0 == n1 && n0 == n2 && n0 == n3 && h0 == h1 && h0 == h2 && h0 == h3 && w0 == w1 && w0 == w2 && w0 == w3)) {

    y->GetOriginShape().SetDim(1, c0 + c1 + c2 + c3);
    y->GetOriginShape().SetDim(2, h0);
    y->GetOriginShape().SetDim(3, w0);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE;

constexpr int32_t BUFFER_NUM = 2;

class KernelConcatChannels {
public:
    __aicore__ inline KernelConcatChannels() {}
    __aicore__ inline void Init(GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR x4, GM_ADDR y,
                                uint32_t n, uint32_t ps1, uint32_t ps2, uint32_t ps3, uint32_t ps4, uint32_t totalLen, uint32_t tileNum)
    {
        this->n = n;
        this->ps1 = ps1;
        this->ps2 = ps2;
        this->ps3 = ps3;
        this->ps4 = ps4;
        this->perSampleOut = ps1 + ps2 + ps3 + ps4;
        this->totalLen = totalLen;
        this->tileNum = tileNum;

        x1Gm.SetGlobalBuffer((__gm__ DTYPE*)x1, totalLen); // totalLen is upper bound for addressing
        x2Gm.SetGlobalBuffer((__gm__ DTYPE*)x2, totalLen);
        x3Gm.SetGlobalBuffer((__gm__ DTYPE*)x3, totalLen);
        x4Gm.SetGlobalBuffer((__gm__ DTYPE*)x4, totalLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, totalLen);

        this->tileLength = totalLen / (tileNum * BUFFER_NUM);
        if (this->tileLength == 0) { this->tileLength = 1; }

        pipe.InitBuffer(inQueueSrc, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueDst, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
    }
    __aicore__ inline void Process()
    {
        uint32_t offset2 = ps1;
        uint32_t offset3 = ps1 + ps2;
        uint32_t offset4 = ps1 + ps2 + ps3;

        for (uint32_t b = 0; b < n; ++b) {
            uint32_t baseOut = b * perSampleOut;

            CopyBranch(x1Gm, b * ps1, baseOut + 0, ps1);
            CopyBranch(x2Gm, b * ps2, baseOut + offset2, ps2);
            CopyBranch(x3Gm, b * ps3, baseOut + offset3, ps3);
            CopyBranch(x4Gm, b * ps4, baseOut + offset4, ps4);
        }
    }

private:
    __aicore__ inline void CopyBranch(AscendC::GlobalTensor<DTYPE>& src, uint32_t srcBase,
                                      uint32_t dstBase, uint32_t length)
    {
        uint32_t copied = 0;
        while (copied < length) {
            uint32_t curr = this->tileLength;
            if (copied + curr > length) curr = length - copied;

            AscendC::LocalTensor<DTYPE> inLocal = inQueueSrc.AllocTensor<DTYPE>();
            AscendC::DataCopy(inLocal, src[srcBase + copied], curr);
            inQueueSrc.EnQue(inLocal);

            AscendC::LocalTensor<DTYPE> inLocalDeq = inQueueSrc.DeQue<DTYPE>();
            AscendC::LocalTensor<DTYPE> outLocal = outQueueDst.AllocTensor<DTYPE>();
            AscendC::DataCopy(outLocal, inLocalDeq, curr);
            outQueueDst.EnQue(outLocal);
            inQueueSrc.FreeTensor(inLocalDeq);

            AscendC::LocalTensor<DTYPE> outLocalDeq = outQueueDst.DeQue<DTYPE>();
            AscendC::DataCopy(yGm[dstBase + copied], outLocalDeq, curr);
            outQueueDst.FreeTensor(outLocalDeq);

            copied += curr;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueSrc;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueDst;

    AscendC::GlobalTensor<DTYPE> x1Gm;
    AscendC::GlobalTensor<DTYPE> x2Gm;
    AscendC::GlobalTensor<DTYPE> x3Gm;
    AscendC::GlobalTensor<DTYPE> x4Gm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t n;
    uint32_t ps1;
    uint32_t ps2;
    uint32_t ps3;
    uint32_t ps4;
    uint32_t perSampleOut;
    uint32_t totalLen;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void googlenet_inception_module_custom(GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR x4, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConcatChannels op;
    op.Init(x1, x2, x3, x4, y,
            tiling_data.n, tiling_data.ps1, tiling_data.ps2, tiling_data.ps3, tiling_data.ps4,
            tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_1x1, reduce_3x3, out_3x3, reduce_5x5, out_5x5, pool_proj):
        super(ModelNew, self).__init__()
        # 1x1 branch
        self.branch1x1 = nn.Conv2d(in_channels, out_1x1, kernel_size=1)
        # 3x3 branch
        self.branch3x3 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_3x3, kernel_size=1),
            nn.Conv2d(reduce_3x3, out_3x3, kernel_size=3, padding=1)
        )
        # 5x5 branch
        self.branch5x5 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_5x5, kernel_size=1),
            nn.Conv2d(reduce_5x5, out_5x5, kernel_size=5, padding=2)
        )
        # Pool + 1x1 branch
        self.branch_pool = nn.Sequential(
            nn.MaxPool2d(kernel_size=3, stride=1, padding=1),
            nn.Conv2d(in_channels, pool_proj, kernel_size=1)
        )

    def forward(self, x):
        b1 = self.branch1x1(x)
        b3 = self.branch3x3(x)
        b5 = self.branch5x5(x)
        bp = self.branch_pool(x)
        # 使用自定义AscendC内核完成通道维度拼接
        return custom_ops_lib.googlenet_inception_module_custom(b1, b3, b5, bp)
