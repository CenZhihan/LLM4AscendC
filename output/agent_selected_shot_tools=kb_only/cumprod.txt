#include "cumprod_custom_tiling.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CumprodCustomTilingData tiling;

    // 获取输入形状，总元素数
    auto inShape = context->GetInputShape(0)->GetOriginShape();
    uint64_t totalLength = inShape.GetShapeSize();

    // 尝试获取二维形状的两个维度
    uint32_t outer = 1;
    uint32_t inner = static_cast<uint32_t>(totalLength);
    // 假设输入为二维 (B, N)，优先使用前两个维度
    if (inShape.GetDimNum() >= 2) {
        outer = static_cast<uint32_t>(inShape.GetDim(0));
        inner = static_cast<uint32_t>(inShape.GetDim(1));
    }

    // 获取属性 dim，默认与最后一维一致（dim=1）
    int64_t dim = 1;
    if (context->GetAttrs() != nullptr && context->GetAttrs()->HasAttr("dim")) {
        dim = context->GetAttrs()->GetInt("dim");
    }

    uint32_t isLastAxis = (dim == 1) ? 1 : 0; // 仅支持沿最后一维
    tiling.set_outer(outer);
    tiling.set_inner(inner);
    tiling.set_isLastAxis(isLastAxis);

    // 配置 BlockDim
    const uint32_t BLOCK_DIM = 8;
    context->SetBlockDim(BLOCK_DIM);

    // 保存 tiling
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

// 当前实现假设输入为二维 [outer, inner]，且 dim=1（沿最后一维进行累计乘）
// 要求输入为正数（示例使用 torch.rand），实现为 cumprod(x) = exp(cumsum(ln(x))).

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1;

class KernelCumprod {
public:
    __aicore__ inline KernelCumprod() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outer, uint32_t inner, uint32_t isLastAxis)
    {
        this->outer = outer;
        this->inner = inner;
        this->isLastAxis = isLastAxis;

        // 基于行切分到不同Core
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (outer + blocks - 1) / blocks;
        this->rowStart = idx * rowsPerBlock;
        uint32_t rowEnd = (rowStart + rowsPerBlock > outer) ? outer : (rowStart + rowsPerBlock);
        this->rowCount = (rowEnd > rowStart) ? (rowEnd - rowStart) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + this->rowStart * this->inner, this->rowCount * this->inner);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + this->rowStart * this->inner, this->rowCount * this->inner);

        // 每次处理一整行
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->inner * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->inner * sizeof(DTYPE));
        // 用于中间缓冲：对数与前缀和
        pipe.InitBuffer(tmpLogQue, BUFFER_NUM, this->inner * sizeof(DTYPE));
        pipe.InitBuffer(tmpSumQue, BUFFER_NUM, this->inner * sizeof(DTYPE));
        // CumSum 要求 lastRow tensor（即便不输出也需要传参）
        pipe.InitBuffer(lastRowQue, BUFFER_NUM, this->inner * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        if (this->rowCount == 0) {
            return;
        }
        for (uint32_t r = 0; r < this->rowCount; r++) {
            CopyIn(r);
            ComputeRow();
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[r * this->inner], this->inner);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> logLocal = tmpLogQue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> sumLocal = tmpSumQue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> lastRowLocal = lastRowQue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();

        // ln(x)
        AscendC::Ln(logLocal, xLocal, (int32_t)this->inner);

        // 沿最后一维做 cumsum（当前仅支持最后一维）
        // 将 (1, inner) 作为输入给 CumSum
        static constexpr AscendC::CumSumConfig cfgLast{true, false, false};
        AscendC::CumSumInfo info;
        info.outter = 1;
        info.inner = this->inner;
        AscendC::CumSum<DTYPE, cfgLast>(sumLocal, lastRowLocal, logLocal, info);

        // y = exp(cumsum(ln(x)))
        AscendC::Exp(yLocal, sumLocal, (int32_t)this->inner);

        outQueueY.EnQue<DTYPE>(yLocal);
        lastRowQue.FreeTensor(lastRowLocal);
        tmpSumQue.FreeTensor(sumLocal);
        tmpLogQue.FreeTensor(logLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[r * this->inner], yLocal, this->inner);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpLogQue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpSumQue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> lastRowQue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t outer;
    uint32_t inner;
    uint32_t isLastAxis;
    uint32_t rowStart;
    uint32_t rowCount;
};

extern "C" __global__ __aicore__ void cumprod_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumprod op;
    op.Init(x, y, tiling_data.outer, tiling_data.inner, tiling_data.isLastAxis);
    op.Process();
}
"""
