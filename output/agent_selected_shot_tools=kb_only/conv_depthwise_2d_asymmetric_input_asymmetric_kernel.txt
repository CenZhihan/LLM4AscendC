namespace {
inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}
inline uint32_t ComputeOut(uint32_t in, uint32_t k, uint32_t s, uint32_t p, uint32_t d) {
    // out = floor((in + 2*pad - d*(k-1) - 1)/stride + 1)
    int64_t num = static_cast<int64_t>(in) + 2 * static_cast<int64_t>(p) - static_cast<int64_t>(d) * (static_cast<int64_t>(k) - 1) - 1;
    if (num < 0) num = 0;
    return static_cast<uint32_t>(num / static_cast<int64_t>(s) + 1);
}
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    // Read attrs for stride/pad/dilation
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t stride_h = 1, stride_w = 1, pad_h = 0, pad_w = 0, dil_h = 1, dil_w = 1;
    if (attrs && attrs->GetNumAttrs() >= 6) {
        stride_h = *(attrs->GetAttrPointer<uint32_t>(0));
        stride_w = *(attrs->GetAttrPointer<uint32_t>(1));
        pad_h    = *(attrs->GetAttrPointer<uint32_t>(2));
        pad_w    = *(attrs->GetAttrPointer<uint32_t>(3));
        dil_h    = *(attrs->GetAttrPointer<uint32_t>(4));
        dil_w    = *(attrs->GetAttrPointer<uint32_t>(5));
    }

    uint32_t n = (uint32_t)xShape->GetDim(0);
    uint32_t c = (uint32_t)xShape->GetDim(1);
    uint32_t h = (uint32_t)xShape->GetDim(2);
    uint32_t w = (uint32_t)xShape->GetDim(3);
    uint32_t kh = (uint32_t)wShape->GetDim(2);
    uint32_t kw = (uint32_t)wShape->GetDim(3);

    auto ComputeOut = [](uint32_t in, uint32_t k, uint32_t s, uint32_t p, uint32_t d) -> uint32_t {
        int64_t num = (int64_t)in + 2 * (int64_t)p - (int64_t)d * ((int64_t)k - 1) - 1;
        if (num < 0) num = 0;
        return (uint32_t)(num / (int64_t)s + 1);
    };

    uint32_t out_h = ComputeOut(h, kh, stride_h, pad_h, dil_h);
    uint32_t out_w = ComputeOut(w, kw, stride_w, pad_w, dil_w);

    gert::Shape *yShape = context->GetOutputShape(0);
    std::vector<int64_t> outDims = { (int64_t)n, (int64_t)c, (int64_t)out_h, (int64_t)out_w };
    yShape->SetDimNum(4);
    for (size_t i = 0; i < outDims.size(); ++i) {
        yShape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}

#include "kernel_operator.h"

class KernelDepthwiseConv2D {
public:
    __aicore__ inline KernelDepthwiseConv2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t kh, uint32_t kw,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t dil_h, uint32_t dil_w,
                                uint32_t out_h, uint32_t out_w,
                                uint32_t has_bias, uint32_t channelPerBlock)
    {
        this->n_ = n; this->c_ = c; this->h_ = h; this->w_ = w_in;
        this->kh_ = kh; this->kw_ = kw;
        this->sh_ = stride_h; this->sw_ = stride_w;
        this->ph_ = pad_h; this->pw_ = pad_w;
        this->dh_ = dil_h; this->dw_ = dil_w;
        this->oh_ = out_h; this->ow_ = out_w;
        this->has_bias_ = has_bias;
        this->channelPerBlock_ = channelPerBlock;

        xPtr_ = reinterpret_cast<__gm__ float *>(x);
        wPtr_ = reinterpret_cast<__gm__ float *>(w);
        bPtr_ = reinterpret_cast<__gm__ float *>(b);
        yPtr_ = reinterpret_cast<__gm__ float *>(y);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startC = blockIdx * channelPerBlock_;
        uint32_t endC = startC + channelPerBlock_;
        if (endC > c_) endC = c_;

        for (uint32_t ni = 0; ni < n_; ++ni) {
            for (uint32_t ci = startC; ci < endC; ++ci) {
                for (uint32_t oh = 0; oh < oh_; ++oh) {
                    for (uint32_t ow = 0; ow < ow_; ++ow) {
                        float acc = 0.0f;
                        for (uint32_t kh = 0; kh < kh_; ++kh) {
                            int32_t ih = static_cast<int32_t>(oh) * static_cast<int32_t>(sh_) - static_cast<int32_t>(ph_) + static_cast<int32_t>(kh) * static_cast<int32_t>(dh_);
                            if (ih < 0 || ih >= static_cast<int32_t>(h_)) continue;
                            for (uint32_t kw = 0; kw < kw_; ++kw) {
                                int32_t iw = static_cast<int32_t>(ow) * static_cast<int32_t>(sw_) - static_cast<int32_t>(pw_) + static_cast<int32_t>(kw) * static_cast<int32_t>(dw_);
                                if (iw < 0 || iw >= static_cast<int32_t>(w_)) continue;

                                uint64_t x_off = (((uint64_t)ni * c_ + ci) * h_ + (uint32_t)ih) * w_ + (uint32_t)iw;
                                uint64_t w_off = (((uint64_t)ci) * kh_ + kh) * kw_ + kw; // weight [C,1,KH,KW]
                                acc += xPtr_[x_off] * wPtr_[w_off];
                            }
                        }
                        if (has_bias_) {
                            acc += bPtr_[ci];
                        }
                        uint64_t y_off = (((uint64_t)ni * c_ + ci) * oh_ + oh) * ow_ + ow;
                        yPtr_[y_off] = acc;
                    }
                }
            }
        }
    }

private:
    __gm__ float *xPtr_;
    __gm__ float *wPtr_;
    __gm__ float *bPtr_;
    __gm__ float *yPtr_;

    uint32_t n_, c_, h_, w_;
    uint32_t kh_, kw_;
    uint32_t sh_, sw_;
    uint32_t ph_, pw_;
    uint32_t dh_, dw_;
    uint32_t oh_, ow_;
    uint32_t has_bias_;
    uint32_t channelPerBlock_;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelDepthwiseConv2D op;
    op.Init(x, w, b, y,
            tilingData.n(), tilingData.c(), tilingData.h(), tilingData.w(),
            tilingData.kh(), tilingData.kw(),
            tilingData.stride_h(), tilingData.stride_w(),
            tilingData.pad_h(), tilingData.pad_w(),
            tilingData.dil_h(), tilingData.dil_w(),
            tilingData.out_h(), tilingData.out_w(),
            tilingData.has_bias(), tilingData.channelPerBlock());
    op.Process();
}
