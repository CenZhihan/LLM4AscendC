kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1;

class KernelFusedBnGeluRelu {
public:
    __aicore__ inline KernelFusedBnGeluRelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR mean, GM_ADDR rstd, GM_ADDR z, uint32_t rows, uint32_t cols)
    {
        this->rows = rows;
        this->cols = cols;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Range-based row split
        uint64_t start_row = (static_cast<uint64_t>(rows) * blockIdx) / blockNum;
        uint64_t end_row   = (static_cast<uint64_t>(rows) * (blockIdx + 1)) / blockNum;
        this->rowStart = static_cast<uint32_t>(start_row);
        this->rowCount = static_cast<uint32_t>(end_row - start_row);

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(x) + this->rowStart * this->cols, this->rowCount * this->cols);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(z) + this->rowStart * this->cols, this->rowCount * this->cols);

        gammaGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(gamma), this->cols);
        betaGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(beta), this->cols);
        meanGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(mean), this->cols);
        rstdGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE*>(rstd), this->cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->cols * sizeof(DTYPE));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->cols * sizeof(DTYPE));

        // BN params local buffers (kept for reuse)
        pipe.InitBuffer(bufGamma, this->cols * sizeof(DTYPE));
        pipe.InitBuffer(bufBeta,  this->cols * sizeof(DTYPE));
        pipe.InitBuffer(bufMean,  this->cols * sizeof(DTYPE));
        pipe.InitBuffer(bufRstd,  this->cols * sizeof(DTYPE));

        // Temp buffers
        pipe.InitBuffer(tmpBuf, this->cols * sizeof(DTYPE) * 3);

        // Preload BN params once
        {
            AscendC::LocalTensor<DTYPE> gLocal = bufGamma.Get<DTYPE>();
            AscendC::LocalTensor<DTYPE> bLocal = bufBeta.Get<DTYPE>();
            AscendC::LocalTensor<DTYPE> mLocal = bufMean.Get<DTYPE>();
            AscendC::LocalTensor<DTYPE> rLocal = bufRstd.Get<DTYPE>();
            AscendC::DataCopy(gLocal, gammaGm, this->cols);
            AscendC::DataCopy(bLocal, betaGm,  this->cols);
            AscendC::DataCopy(mLocal, meanGm,  this->cols);
            AscendC::DataCopy(rLocal, rstdGm,  this->cols);
        }
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowCount; ++r) {
            CopyIn(r);
            Compute();
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->cols], this->cols);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.AllocTensor<DTYPE>();

        // BN params locals (already loaded)
        AscendC::LocalTensor<DTYPE> gLocal = bufGamma.Get<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = bufBeta.Get<DTYPE>();
        AscendC::LocalTensor<DTYPE> mLocal = bufMean.Get<DTYPE>();
        AscendC::LocalTensor<DTYPE> rLocal = bufRstd.Get<DTYPE>();

        // Temp locals
        AscendC::LocalTensor<DTYPE> t0 = tmpBuf.Get<DTYPE>();                 // x - mean
        AscendC::LocalTensor<DTYPE> t1 = tmpBuf.Get<DTYPE>() + this->cols;    // (x-mean)*rstd -> then *gamma -> then +beta
        AscendC::LocalTensor<DTYPE> t2 = tmpBuf.Get<DTYPE>() + this->cols*2;  // GELU intermediate

        // BN: (x - mean) * rstd * gamma + beta
        AscendC::Sub(t0, xLocal, mLocal, this->cols);
        AscendC::Mul(t1, t0, rLocal, this->cols);
        AscendC::Mul(t1, t1, gLocal, this->cols);
        AscendC::Add(t1, t1, bLocal, this->cols);

        // GELU
        AscendC::Gelu(t2, t1, this->cols);

        // ReLU
        AscendC::Relu(zLocal, t2, this->cols);

        outQueueZ.EnQue<DTYPE>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.DeQue<DTYPE>();
        AscendC::DataCopy(zGm[rowIdx * this->cols], zLocal, this->cols);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    uint32_t rows;
    uint32_t cols;
    uint32_t rowStart;
    uint32_t rowCount;

    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> zGm;

    AscendC::GlobalTensor<DTYPE> gammaGm;
    AscendC::GlobalTensor<DTYPE> betaGm;
    AscendC::GlobalTensor<DTYPE> meanGm;
    AscendC::GlobalTensor<DTYPE> rstdGm;

    AscendC::TBuf<AscendC::TPosition::VECCALC> bufGamma;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufBeta;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufMean;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufRstd;

    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf;
};

extern "C" __global__ __aicore__ void gemm_batch_norm_gelu_group_norm_mean_relu_custom(
    GM_ADDR x,
    GM_ADDR gamma,
    GM_ADDR beta,
    GM_ADDR mean,
    GM_ADDR rstd,
    GM_ADDR z,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedBnGeluRelu op;
    op.Init(x, gamma, beta, mean, rstd, z, tiling_data.rows, tiling_data.cols);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.batch_norm = torch.nn.BatchNorm1d(out_features)

    def forward(self, x):
        # GEMM
        x = self.gemm(x)

        # Prepare BN parameters and rstd on host
        gamma = self.batch_norm.weight
        beta = self.batch_norm.bias
        running_mean = self.batch_norm.running_mean
        running_var = self.batch_norm.running_var
        eps = self.batch_norm.eps

        rstd = 1.0 / torch.sqrt(running_var + eps)

        # Fused: BatchNorm -> GELU -> ReLU
        y = custom_ops_lib.gemm_batch_norm_gelu_group_norm_mean_relu_custom(
            x, gamma, beta, running_mean, rstd
        )
        return y
