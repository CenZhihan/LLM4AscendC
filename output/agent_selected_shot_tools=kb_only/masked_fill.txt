#include "masked_fill_custom_tiling.h"

#include <limits>

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaskedFillCustomTilingData tiling;

    // 展平形状，使用整个张量作为最后一维，满足对齐要求（示例化处理）
    uint64_t totalLen64 = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    uint32_t totalLength = static_cast<uint32_t>(totalLen64);

    // 形状信息（firstAxis x srcLastAxis = totalLength）
    uint32_t firstAxis    = 1;
    uint32_t srcLastAxis  = totalLength;
    uint32_t maskLastAxis = totalLength;

    // 临时缓冲大小（示例值，实际可用5.10.2接口计算）
    uint32_t tmpSize = 4096;

    // 选择标量：-inf
    float scalarValue = -std::numeric_limits<float>::infinity();

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_scalarValue(scalarValue);
    tiling.set_firstAxis(firstAxis);
    tiling.set_srcLastAxis(srcLastAxis);
    tiling.set_maskLastAxis(maskLastAxis);
    tiling.set_tmpSize(tmpSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* z_shape = context->GetOutputShape(0);

    return GRAPH_SUCCESS;

    const auto inputDataType = context->GetInputDataType(0);

kernel_src="""
#include "kernel_operator.h"

template <typename srcType, typename maskType>
class KernelMaskedFill {
public:
    __aicore__ inline KernelMaskedFill() {}
    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR maskGm, GM_ADDR zGm,
                                float scalarValue,
                                uint32_t firstAxis,
                                uint32_t srcLastAxis,
                                uint32_t maskLastAxis,
                                uint32_t tmpSize)
    {
        uint32_t srcSize  = firstAxis * srcLastAxis;
        uint32_t maskSize = firstAxis * maskLastAxis;

        xGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ srcType*>(xGm), srcSize);
        zGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ srcType*>(zGm), srcSize);
        maskGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ maskType*>(maskGm), maskSize);

        pipe.InitBuffer(inQueueX, 1, srcSize * sizeof(srcType));
        pipe.InitBuffer(maskQueue, 1, maskSize * sizeof(maskType));
        pipe.InitBuffer(tmpQueue, 1, tmpSize);

        bufferSize = srcSize;
        maskBufferSize = maskSize;

        info.firstAxis   = firstAxis;
        info.srcLastAxis = srcLastAxis;
        info.maskLastAxis = maskLastAxis;

        scalar = static_cast<srcType>(scalarValue);
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        xLocal = inQueueX.AllocTensor<srcType>();
        AscendC::DataCopy(xLocal, xGlobal, bufferSize);
        inQueueX.EnQue(xLocal);

        AscendC::LocalTensor<maskType> maskLocal = maskQueue.AllocTensor<maskType>();
        AscendC::DataCopy(maskLocal, maskGlobal, maskBufferSize);
        maskQueue.EnQue(maskLocal);
    }

    __aicore__ inline void Compute()
    {
        xLocal = inQueueX.DeQue<srcType>();
        AscendC::LocalTensor<maskType> maskLocal = maskQueue.DeQue<maskType>();
        AscendC::LocalTensor<uint8_t> tmpLocal = tmpQueue.AllocTensor<uint8_t>();

        AscendC::SelectWithBytesMask(xLocal, xLocal, scalar, maskLocal, tmpLocal, info);

        maskQueue.FreeTensor(maskLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::DataCopy(zGlobal, xLocal, bufferSize);
        inQueueX.FreeTensor(xLocal);
    }

private:
    AscendC::GlobalTensor<srcType> xGlobal;
    AscendC::GlobalTensor<srcType> zGlobal;
    AscendC::GlobalTensor<maskType> maskGlobal;

    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> maskQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueue;

    AscendC::SelectWithBytesMaskShapeInfo info;
    AscendC::LocalTensor<srcType> xLocal;

    uint32_t bufferSize = 0;
    uint32_t maskBufferSize = 0;
    srcType scalar = 0.0f;
};

extern "C" __global__ __aicore__ void masked_fill_custom(GM_ADDR xGm, GM_ADDR maskGm, GM_ADDR zGm, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelMaskedFill<float, bool> op;
    op.Init(xGm,
            maskGm,
            zGm,
            tilingData.scalarValue,
            tilingData.firstAxis,
            tilingData.srcLastAxis,
            tilingData.maskLastAxis,
            tilingData.tmpSize);
    op.Process();
}
