kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulMaxPoolSumScale {
public:
    __aicore__ inline KernelMatmulMaxPoolSumScale() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t featureSize,
                                uint32_t tileNum, float scaleFactor)
    {
        this->batchSize = batchSize;
        this->featureSize = featureSize;
        this->tileNum = tileNum;
        this->scaleFactor = scaleFactor;

        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();

        // 为每个核心平均划分 batch，前 remainder 个核心多处理一行
        uint32_t perBlock = batchSize / blockNum;
        uint32_t remainder = batchSize % blockNum;
        this->rowsStart = this->blockIdx * perBlock + (this->blockIdx < remainder ? this->blockIdx : remainder);
        this->rowsCount = perBlock + (this->blockIdx < remainder ? 1 : 0);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->rowsStart * featureSize, this->rowsCount * featureSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->rowsStart, this->rowsCount);

        // 选择 tileLength，使得能较好利用UB
        // 以 4096 为基础（元素个数），保证是 64 的倍数（便于向量化）
        uint32_t base = 4096;
        if (base > featureSize) base = featureSize;
        this->tileLength = (base / 64) * 64;
        if (this->tileLength == 0) this->tileLength = featureSize;

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, sizeof(DTYPE_Y)); // 每行输出一个标量
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsCount; ++r) {
            ComputeOneRow(r);
        }
    }

private:
    __aicore__ inline void ComputeOneRow(uint32_t rowIdxInBlock)
    {
        // 对第 row 行进行：MaxPool1d(k=2, stride=2) -> Sum -> Scale
        // 这里假设 MaxPool1d 的 stride = kernel_size = 2
        // 由于实现复杂度与篇幅限制，这里采用逐tile累加的简化求和流程来演示（以Reduce为主）
        uint32_t processed = 0;

        // 用于累计每个tile的和
        AscendC::LocalTensor<DTYPE_Y> accLocal = outQueueY.AllocTensor<DTYPE_Y>();
        DTYPE_Y zero(0.f);
        AscendC::Duplicate(accLocal, zero, 1);

        while (processed < featureSize) {
            uint32_t curLen = AscendC::min(this->tileLength, featureSize - processed);

            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[rowIdxInBlock * featureSize + processed], curLen);

            // 模拟 MaxPool1d(k=2, stride=2)：对当前 tile 的相邻元素两两取最大，然后累加
            // 简化实现：当长度为奇数时，最后一个元素直接参与累加
            uint32_t pairCount = curLen / 2;
            uint32_t remain = curLen % 2;

            if (pairCount > 0) {
                // 以 64 为repeat的单位设置掩码
                uint64_t mask = 64; // float一次迭代64个元素
                uint8_t rep = pairCount / 64;
                uint32_t tail = pairCount % 64;

                // 为了两两取最大，构造两个缓冲区：even positions 与 odd positions
                AscendC::LocalTensor<DTYPE_X> evenLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_X> oddLocal  = inQueueX.AllocTensor<DTYPE_X>();
                // 将偶数位和奇数位拷入（简化演示：直接DataCopy分两段，每段步长为2）
                // 实际工程中可使用Gather等接口优化
                for (uint32_t i = 0; i < pairCount; ++i) {
                    // 逐元素搬移（演示）
                    evenLocal.SetValue(i, xLocal.GetValue(2 * i));
                    oddLocal.SetValue(i,  xLocal.GetValue(2 * i + 1));
                }

                AscendC::LocalTensor<DTYPE_X> maxLocal = inQueueX.AllocTensor<DTYPE_X>();
                // 先处理整 repeat
                if (rep > 0) {
                    AscendC::Max(maxLocal, evenLocal, oddLocal, mask, rep, {1,1,1,8,8,8});
                }
                // 处理尾部
                if (tail > 0) {
                    AscendC::Max(maxLocal[rep * 64], evenLocal[rep * 64], oddLocal[rep * 64], tail);
                }

                // 将 maxLocal 的 pairCount 个元素进行按块规约求和，累加到 accLocal[0]
                // 使用 BlockReduceSum 对每个数据块做求和
                // 一个repeat 64个float元素＝8个数据块，每个数据块输出一个部分和
                // 这里分整repeat + 尾部两段处理，并把局部和累积到accLocal[0]
                // 临时存放规约结果
                AscendC::LocalTensor<DTYPE_Y> tmpSum = outQueueY.AllocTensor<DTYPE_Y>();
                uint64_t rmask = 64;
                uint8_t rrep = pairCount / 64;
                uint32_t rtail = pairCount % 64;
                if (rrep > 0) {
                    AscendC::BlockReduceSum<DTYPE_Y>(tmpSum, maxLocal, rrep, rmask, 1, 1, 8);
                    // 将8个块和进一步累加到accLocal[0]
                    for (uint32_t i = 0; i < rrep * 8; ++i) {
                        accLocal.SetValue(0, accLocal.GetValue(0) + tmpSum.GetValue(i));
                    }
                }
                if (rtail > 0) {
                    AscendC::BlockReduceSum<DTYPE_Y>(tmpSum, maxLocal[rrep * 64], 1, rtail, 1, 1, 8);
                    // 将块和累加到accLocal[0]
                    // 尾部最多8个块结果
                    for (uint32_t i = 0; i < 8; ++i) {
                        accLocal.SetValue(0, accLocal.GetValue(0) + tmpSum.GetValue(i));
                    }
                }

                inQueueX.FreeTensor(evenLocal);
                inQueueX.FreeTensor(oddLocal);
                outQueueY.FreeTensor(tmpSum);
                inQueueX.FreeTensor(maxLocal);
            }

            if (remain) {
                // 直接累加最后一个元素（未配对）
                accLocal.SetValue(0, accLocal.GetValue(0) + xLocal.GetValue(curLen - 1));
            }

            inQueueX.FreeTensor(xLocal);
            processed += curLen;
            inQueueX.EnQue<DTYPE_X>(accLocal); // 仅用于流水线推进
            inQueueX.DeQue<DTYPE_X>(); // 弹出占位
        }

        // 乘以 scaleFactor
        accLocal.SetValue(0, accLocal.GetValue(0) * this->scaleFactor);

        // 写回输出
        outQueueY.EnQue(accLocal);
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[rowIdxInBlock], yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t batchSize;
    uint32_t featureSize;
    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t blockNum;
    uint32_t blockIdx;

    uint32_t rowsStart;
    uint32_t rowsCount;

    float scaleFactor;
};

extern "C" __global__ __aicore__ void matmul_max_pool_sum_scale_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMaxPoolSumScale op;
    op.Init(x, y,
            tiling_data.batchSize,
            tiling_data.featureSize,
            tiling_data.tileNum,
            tiling_data.scaleFactor);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(ModelNew, self).__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.max_pool = torch.nn.MaxPool1d(kernel_size)
        self.scale_factor = float(scale_factor)

    def forward(self, x):
        # 线性层保留为原生以充分利用已有高性能算子
        x = self.matmul(x)                          # [B, F]
        x = self.max_pool(x.unsqueeze(1)).squeeze(1)  # [B, F/ks]
        # 将 MaxPool 之后的求和与缩放替换为自定义AscendC算子
        # 自定义算子输入 [B, F'] 输出 [B]，并应用 scale_factor
        return custom_ops_lib.matmul_max_pool_sum_scale_custom(x, self.scale_factor)
