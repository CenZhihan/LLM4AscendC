const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)

    SgdCustomTilingData tiling;
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);

    tiling.set_lr(LR);

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}


namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);
    *out_shape = *in_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)

    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelSgd {
public:
    __aicore__ inline KernelSgd() {}
    __aicore__ inline void Init(GM_ADDR param, GM_ADDR grad, GM_ADDR velocity, GM_ADDR out,
                                float momentum, float lr, uint32_t totalLength, uint32_t tileNum)
    {
        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        ASSERT(tileNum != 0 && "tile num can not be zero!");
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->momentum = momentum;
        this->lr = lr;

        paramGm.SetGlobalBuffer((__gm__ float*)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gradGm.SetGlobalBuffer((__gm__ float*)grad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        velGm.SetGlobalBuffer((__gm__ float*)velocity + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ float*)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueParam, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueGrad, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueVel, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueOut, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> pLocal = inQueueParam.AllocTensor<float>();
        AscendC::LocalTensor<float> gLocal = inQueueGrad.AllocTensor<float>();
        AscendC::LocalTensor<float> vLocal = inQueueVel.AllocTensor<float>();
        AscendC::DataCopy(pLocal, paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gLocal, gradGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(vLocal, velGm[progress * this->tileLength], this->tileLength);
        inQueueParam.EnQue(pLocal);
        inQueueGrad.EnQue(gLocal);
        inQueueVel.EnQue(vLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> pLocal = inQueueParam.DeQue<float>();
        AscendC::LocalTensor<float> gLocal = inQueueGrad.DeQue<float>();
        AscendC::LocalTensor<float> vLocal = inQueueVel.DeQue<float>();
        AscendC::LocalTensor<float> outLocal = outQueueOut.AllocTensor<float>();

        // outLocal 用作中间缓存:
        // outLocal = momentum * vLocal + gLocal
        AscendC::Muls(outLocal, vLocal, (float)momentum, this->tileLength);
        AscendC::Add(outLocal, outLocal, gLocal, this->tileLength);
        // outLocal = pLocal - lr * outLocal
        AscendC::Muls(outLocal, outLocal, (float)lr, this->tileLength);
        AscendC::Sub(outLocal, pLocal, outLocal, this->tileLength);

        outQueueOut.EnQue<float>(outLocal);
        inQueueParam.FreeTensor(pLocal);
        inQueueGrad.FreeTensor(gLocal);
        inQueueVel.FreeTensor(vLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> outLocal = outQueueOut.DeQue<float>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueParam, inQueueGrad, inQueueVel;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut;

    AscendC::GlobalTensor<float> paramGm;
    AscendC::GlobalTensor<float> gradGm;
    AscendC::GlobalTensor<float> velGm;
    AscendC::GlobalTensor<float> outGm;

    float momentum;
    float lr;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void sgd_custom(GM_ADDR param, GM_ADDR grad, GM_ADDR velocity, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelSgd op;
    op.Init(param, grad, velocity, out, tilingData.momentum, tilingData.lr, tilingData.totalLength, tilingData.tileNum);
    op.Process();
}
