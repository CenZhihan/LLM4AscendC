static ge::graphStatus TilingFunc(gert::TilingContext* context)

    ScatterCustomTilingData tiling;

    const auto* idx_shape = context->GetInputShape(1);

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Z = float;
using DTYPE_UPD = float;
using DTYPE_IDX = int32_t;

class KernelScatter {
public:
    __aicore__ inline KernelScatter() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR updates, GM_ADDR z,
                                uint32_t rowSize, uint32_t indexCount)
    {
        this->rowSize = rowSize;
        this->indexCount = indexCount;
        uint32_t rowId = AscendC::GetBlockIdx();

        xPtr = reinterpret_cast<__gm__ DTYPE_X*>(x) + static_cast<uint64_t>(rowId) * rowSize;
        zPtr = reinterpret_cast<__gm__ DTYPE_Z*>(z) + static_cast<uint64_t>(rowId) * rowSize;
        updPtr = reinterpret_cast<__gm__ DTYPE_UPD*>(updates) + static_cast<uint64_t>(rowId) * indexCount;
        idxPtr32 = reinterpret_cast<__gm__ DTYPE_IDX*>(idx) + static_cast<uint64_t>(rowId) * indexCount;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_X*>(x) + static_cast<uint64_t>(rowId) * rowSize, rowSize);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_Z*>(z) + static_cast<uint64_t>(rowId) * rowSize, rowSize);
    }

    __aicore__ inline void Process()
    {
        // 先将整行复制到输出
        AscendC::DataCopy(zGm, xGm, rowSize);

        // 将updates根据idx写入到对应位置（赋值语义）
        // 注意：若同一位置出现重复索引，写入次序未定义，行为与并行赋值一致
        for (uint32_t j = 0; j < indexCount; ++j) {
            DTYPE_IDX col = idxPtr32[j];
            if (col >= 0 && col < static_cast<DTYPE_IDX>(rowSize)) {
                zPtr[col] = updPtr[j];
            }
        }
    }

private:
    uint32_t rowSize;
    uint32_t indexCount;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_Z* zPtr;
    __gm__ DTYPE_UPD* updPtr;
    __gm__ DTYPE_IDX* idxPtr32;
};

extern "C" __global__ __aicore__ void scatter_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR updates, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelScatter op;
    op.Init(x, idx, updates, z, tilingData.rowSize(), tilingData.indexCount());
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static at::Tensor scatter_custom_impl_npu(const at::Tensor& x, const at::Tensor& idx_in, const at::Tensor& updates) {
    // 保证 idx 为 int32，且三者在 NPU 上连续
    at::Tensor idx = idx_in.dtype() == at::kInt ? idx_in : idx_in.to(at::kInt);
    at::Tensor x_c = x.contiguous();
    at::Tensor idx_c = idx.contiguous();
    at::Tensor upd_c = updates.contiguous();
    at::Tensor z = at::empty_like(x_c);

    EXEC_NPU_CMD(aclnnScatterCustom, x_c, idx_c, upd_c, z);
    return z;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("scatter_custom", &scatter_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("scatter_custom", &scatter_custom_impl_npu, "scatter along dim=1 with src assignment");
}
