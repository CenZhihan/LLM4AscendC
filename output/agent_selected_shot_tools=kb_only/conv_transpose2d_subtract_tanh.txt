static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ConvTranspose2dSubtractTanhCustomTilingData tiling;

    // Input 0 shape (same as output)
    uint32_t totalLength = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Compute tile length per core
    uint32_t blockNum = BLOCK_DIM;
    uint32_t blockLength = totalLength / blockNum;
    uint32_t tileNum = TILE_NUM;
    uint32_t tileLength = blockLength / tileNum / BUFFER_NUM;
    if (tileLength == 0) {
        tileLength = 1;
    }

    // Query min tmp buffer size for Axpy and Tanh on this tile length
    uint32_t tanhMax = 0, tanhMin = 0;
    uint32_t axpyMax = 0, axpyMin = 0;
    {
        std::vector<int64_t> shape_vec = {static_cast<int64_t>(tileLength)};
        ge::Shape shape(shape_vec);
        // float -> 4 bytes
        AscendC::GetTanhMaxMinTmpSize(shape, 4, false, tanhMax, tanhMin);
    }
    {
        std::vector<int64_t> shape_vec = {static_cast<int64_t>(tileLength)};
        ge::Shape shape(shape_vec);
        // float -> 4 bytes
        AscendC::GetAxpyMaxMinTmpSize(shape, 4, false, axpyMax, axpyMin);
    }
    uint32_t bufferSize = std::max(tanhMin, axpyMin);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(tileNum);
    tiling.set_bufferSize(bufferSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue
 
class KernelSubTanh {
public:
    __aicore__ inline KernelSubTanh() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, uint32_t bufferSize)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        // Guard
        if (this->tileLength == 0) {
            this->tileLength = 1;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpVecQue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));

        // Allocate shared tmp buffer for Axpy/Tanh once
        pipe.InitBuffer(tmpBufQue, 1, bufferSize);
        sharedTmpBuffer = tmpBufQue.AllocTensor<uint8_t>();
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(bLocal, bGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.DeQue<DTYPE_Y>();
        // tmp = x - bias (elementwise)
        AscendC::LocalTensor<DTYPE_Z> tmpLocal = tmpVecQue.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(tmpLocal, xLocal, this->tileLength);
        AscendC::Axpy(tmpLocal, bLocal, static_cast<DTYPE_Y>(-1.0f), sharedTmpBuffer, this->tileLength);

        // y = tanh(tmp)
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::Tanh(yLocal, tmpLocal, sharedTmpBuffer, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        tmpVecQue.FreeTensor(tmpLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpVecQue;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBufQue;
    AscendC::LocalTensor<uint8_t> sharedTmpBuffer;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose2d_subtract_tanh_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSubTanh op;
    op.Init(x, bias, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.bufferSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused path: PyTorch ConvTranspose2d + custom (subtract bias + tanh) AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape, stride=2, padding=1, output_padding=1):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        # Keep original bias parameter shape (C,1,1)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        x = self.conv_transpose(x)
        # Expand bias to match x for elementwise subtraction inside the custom kernel
        bias_expanded = self.bias.view(1, -1, 1, 1).expand_as(x).contiguous()
        x = custom_ops_lib.conv_transpose2d_subtract_tanh_custom(x, bias_expanded)
        return x
