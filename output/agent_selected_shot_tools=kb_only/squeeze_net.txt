kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelSqueezeNetCustom {
public:
    __aicore__ inline KernelSqueezeNetCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR ones, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        onesGm.SetGlobalBuffer((__gm__ DTYPE_X *)ones + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueOnes, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> onesLocal = inQueueOnes.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(onesLocal, onesGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueOnes.EnQue(onesLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> onesLocal = inQueueOnes.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        // Use ReGlu to compute ReLU(x) * ones = ReLU(x)
        AscendC::ReGlu<DTYPE_Y, false>(yLocal, xLocal, onesLocal, this->tileLength);
        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueOnes.FreeTensor(onesLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueOnes;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> onesGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void squeeze_net_custom(GM_ADDR x, GM_ADDR ones, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSqueezeNetCustom op;
    op.Init(x, ones, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib
import torch.nn as nn
import torch.nn.functional as F

def relu_custom(x):
    return custom_ops_lib.squeeze_net_custom(x)

class FireModuleNew(nn.Module):
    def __init__(self, in_channels, squeeze_channels, expand1x1_channels, expand3x3_channels):
        super(FireModuleNew, self).__init__()
        self.squeeze = nn.Conv2d(in_channels, squeeze_channels, kernel_size=1)
        self.expand1x1 = nn.Conv2d(squeeze_channels, expand1x1_channels, kernel_size=1)
        self.expand3x3 = nn.Conv2d(squeeze_channels, expand3x3_channels, kernel_size=3, padding=1)

    def forward(self, x):
        x = relu_custom(self.squeeze(x))
        out1 = relu_custom(self.expand1x1(x))
        out3 = relu_custom(self.expand3x3(x))
        return torch.cat([out1, out3], dim=1)

class ModelNew(nn.Module):
    def __init__(self, num_classes=1000):
        super(ModelNew, self).__init__()
        # First conv + relu + pool
        self.conv1 = nn.Conv2d(3, 96, kernel_size=7, stride=2)
        self.pool1 = nn.MaxPool2d(kernel_size=3, stride=2, ceil_mode=True)

        # Fire modules and pools (ReLU replaced by custom op inside FireModuleNew)
        self.fire2 = FireModuleNew(96, 16, 64, 64)
        self.fire3 = FireModuleNew(128, 16, 64, 64)
        self.fire4 = FireModuleNew(128, 32, 128, 128)
        self.pool4 = nn.MaxPool2d(kernel_size=3, stride=2, ceil_mode=True)
        self.fire5 = FireModuleNew(256, 32, 128, 128)
        self.fire6 = FireModuleNew(256, 48, 192, 192)
        self.fire7 = FireModuleNew(384, 48, 192, 192)
        self.fire8 = FireModuleNew(384, 64, 256, 256)
        self.pool8 = nn.MaxPool2d(kernel_size=3, stride=2, ceil_mode=True)
        self.fire9 = FireModuleNew(512, 64, 256, 256)

        # Classifier (replace ReLU with custom op)
        self.drop = nn.Dropout(p=0.0)
        self.conv_cls = nn.Conv2d(512, num_classes, kernel_size=1)
        self.avg = nn.AdaptiveAvgPool2d((1, 1))

    def forward(self, x):
        x = relu_custom(self.conv1(x))
        x = self.pool1(x)

        x = self.fire2(x)
        x = self.fire3(x)
        x = self.fire4(x)
        x = self.pool4(x)
        x = self.fire5(x)
        x = self.fire6(x)
        x = self.fire7(x)
        x = self.fire8(x)
        x = self.pool8(x)
        x = self.fire9(x)

        x = self.drop(x)
        x = relu_custom(self.conv_cls(x))
        x = self.avg(x)
        return torch.flatten(x, 1)
'''
