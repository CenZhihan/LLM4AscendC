kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUF_NUM = 2;

class KernelGemmScaleBN {
public:
    __aicore__ inline KernelGemmScaleBN() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR alpha, GM_ADDR beta2,
                                GM_ADDR y, uint32_t M, uint32_t N, uint32_t K, uint32_t tileN)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileN = tileN;

        // Block partition on M dimension
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockId = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (M + blocks - 1) / blocks;
        rowStart = blockId * rowsPerBlock;
        rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > M) rowEnd = M;

        xGm.SetGlobalBuffer((__gm__ float*)x + rowStart * K, (rowEnd - rowStart) * K);
        // weight is [N, K]
        wGm.SetGlobalBuffer((__gm__ float*)weight, N * K);
        alphaGm.SetGlobalBuffer((__gm__ float*)alpha, N);
        beta2Gm.SetGlobalBuffer((__gm__ float*)beta2, N);
        yGm.SetGlobalBuffer((__gm__ float*)y + rowStart * N, (rowEnd - rowStart) * N);

        // init pipe buffers
        // row slice of y and accumulators
        pipe.InitBuffer(bufY, BUF_NUM, tileN * sizeof(float));
        pipe.InitBuffer(bufTmp, BUF_NUM, tileN * sizeof(float));
        pipe.InitBuffer(bufW, BUF_NUM, tileN * sizeof(float));
        pipe.InitBuffer(bufAlpha, BUF_NUM, tileN * sizeof(float));
        pipe.InitBuffer(bufBeta2, BUF_NUM, tileN * sizeof(float));
        // x row buffer
        pipe.InitBuffer(bufXRow, 1, K * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // For each assigned row, compute y_row = (x_row @ W^T) * alpha + beta2
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            // Load x row [K]
            AscendC::LocalTensor<float> xRowLocal = bufXRow.AllocTensor<float>();
            AscendC::DataCopy(xRowLocal, xGm[(r - rowStart) * K], K);

            // Iterate over N in tiles
            uint32_t loops = (N + tileN - 1) / tileN;
            for (uint32_t t = 0; t < loops; ++t) {
                uint32_t colStart = t * tileN;
                uint32_t currTile = ((colStart + tileN) <= N) ? tileN : (N - colStart);

                // Alloc locals
                AscendC::LocalTensor<float> yLocal = bufY.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpLocal = bufTmp.AllocTensor<float>();
                AscendC::LocalTensor<float> wLocal = bufW.AllocTensor<float>();
                AscendC::LocalTensor<float> alphaLocal = bufAlpha.AllocTensor<float>();
                AscendC::LocalTensor<float> beta2Local = bufBeta2.AllocTensor<float>();

                // zero initialize accumulator
                AscendC::Muls(tmpLocal, tmpLocal, 0.0f, currTile);

                // GEMM accumulation over K:
                // tmpLocal[j] += xRow[k] * weight[colStart + j, k]
                for (uint32_t k = 0; k < K; ++k) {
                    // Load weight tile for this k: contiguous over N dimension
                    AscendC::DataCopy(wLocal, wGm[(colStart * K) + k], currTile, K); // stride is K
                    // Get scalar xRow[k] by duplicating 1 element to a scalar via Muls
                    // Use vector multiply by scalar
                    AscendC::Muls(yLocal, wLocal, *((float*)xRowLocal.GetPhyAddr() + k), currTile);
                    AscendC::Add(tmpLocal, tmpLocal, yLocal, currTile);
                }

                // Post: tmp = tmp * alpha + beta2 (per-column)
                AscendC::DataCopy(alphaLocal, alphaGm[colStart], currTile);
                AscendC::DataCopy(beta2Local, beta2Gm[colStart], currTile);
                AscendC::Mul(yLocal, tmpLocal, alphaLocal, currTile);
                AscendC::Add(yLocal, yLocal, beta2Local, currTile);

                // Store back y row tile
                AscendC::DataCopy(yGm[(r - rowStart) * N + colStart], yLocal, currTile);

                // Free locals
                bufY.FreeTensor(yLocal);
                bufTmp.FreeTensor(tmpLocal);
                bufW.FreeTensor(wLocal);
                bufAlpha.FreeTensor(alphaLocal);
                bufBeta2.FreeTensor(beta2Local);
            }

            bufXRow.FreeTensor(xRowLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_NUM>  bufW;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>        bufXRow;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_NUM>  bufAlpha;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_NUM>  bufBeta2;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUF_NUM> bufY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUF_NUM> bufTmp;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> alphaGm;
    AscendC::GlobalTensor<float> beta2Gm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileN;

    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void gemm_scale_batch_norm_custom(GM_ADDR x,
                                                                   GM_ADDR weight,
                                                                   GM_ADDR alpha,
                                                                   GM_ADDR beta2,
                                                                   GM_ADDR y,
                                                                   GM_ADDR workspace,
                                                                   GM_ADDR tilingData)
{
    // Read simple tiling (M,N,K,tileN) from host
    GET_TILING_DATA(tiling, tilingData);
    KernelGemmScaleBN op;
    op.Init(x, weight, alpha, beta2, y,
            tiling.M, tiling.N, tiling.K, tiling.tileN);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.scale = torch.nn.Parameter(torch.randn(scale_shape))
        self.bn = torch.nn.BatchNorm1d(out_features, eps=eps, momentum=momentum)

    def forward(self, x):
        # Prepare fused vectors:
        # alpha = scale * gamma / sqrt(var + eps)
        # beta2 = beta - mean * gamma / sqrt(var + eps) + bias * alpha
        gamma = self.bn.weight
        beta = self.bn.bias
        mean = self.bn.running_mean
        var = self.bn.running_var
        eps = self.bn.eps
        inv_std = torch.rsqrt(var + eps)
        alpha = self.scale * gamma * inv_std
        if self.gemm.bias is not None:
            beta2 = beta - mean * gamma * inv_std + self.gemm.bias * alpha
        else:
            beta2 = beta - mean * gamma * inv_std

        # Call fused kernel: y = (x @ W^T) * alpha + beta2
        return custom_ops_lib.gemm_scale_batch_norm_custom(x, self.gemm.weight, alpha, beta2)
