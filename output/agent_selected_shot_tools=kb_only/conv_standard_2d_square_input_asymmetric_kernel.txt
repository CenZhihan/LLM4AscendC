kernel_src="""
#include "kernel_operator.h"

using dtype = float;

class KernelConv2D {
public:
    __aicore__ inline KernelConv2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, uint32_t n, uint32_t c_in, uint32_t h, uint32_t w_in,
                                uint32_t c_out, uint32_t kH, uint32_t kW, uint32_t outH, uint32_t outW, uint32_t ocPerBlock)
    {
        this->N = n;
        this->Cin = c_in;
        this->Hin = h;
        this->Win = w_in;
        this->Cout = c_out;
        this->KH = kH;
        this->KW = kW;
        this->Hout = outH;
        this->Wout = outW;
        this->ocPerBlock = ocPerBlock;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ dtype *>(x), static_cast<int64_t>(N) * Cin * Hin * Win);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ dtype *>(w), static_cast<int64_t>(Cout) * Cin * KH * KW);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ dtype *>(y), static_cast<int64_t>(N) * Cout * Hout * Wout);

        pipe.InitBuffer(inRowQ, 1, KW * sizeof(dtype));
        pipe.InitBuffer(wRowQ, 1, KW * sizeof(dtype));
        pipe.InitBuffer(tmpRowQ, 1, KW * sizeof(dtype));
        pipe.InitBuffer(yScalarQ, 1, sizeof(dtype));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t ocStart = blockIdx * ocPerBlock;
        uint32_t ocEnd = ocStart + ocPerBlock;
        if (ocEnd > Cout) ocEnd = Cout;

        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oc = ocStart; oc < ocEnd; ++oc) {
                for (uint32_t oh = 0; oh < Hout; ++oh) {
                    for (uint32_t ow = 0; ow < Wout; ++ow) {
                        dtype acc = 0.0f;
                        for (uint32_t ic = 0; ic < Cin; ++ic) {
                            for (uint32_t kh = 0; kh < KH; ++kh) {
                                // Copy input row window and weight row
                                AscendC::LocalTensor<dtype> inRowLocal = inRowQ.AllocTensor<dtype>();
                                AscendC::LocalTensor<dtype> wRowLocal  = wRowQ.AllocTensor<dtype>();

                                int64_t in_off = ((static_cast<int64_t>(n) * Cin + ic) * Hin + (oh + kh)) * Win + ow;
                                int64_t w_off  = (((static_cast<int64_t>(oc) * Cin + ic) * KH + kh) * KW + 0);

                                AscendC::DataCopy(inRowLocal, xGm[in_off], KW);
                                AscendC::DataCopy(wRowLocal,  wGm[w_off],  KW);

                                inRowQ.EnQue(inRowLocal);
                                wRowQ.EnQue(wRowLocal);

                                AscendC::LocalTensor<dtype> inRow = inRowQ.DeQue<dtype>();
                                AscendC::LocalTensor<dtype> wRow  = wRowQ.DeQue<dtype>();
                                AscendC::LocalTensor<dtype> tmpRow = tmpRowQ.AllocTensor<dtype>();

                                AscendC::Mul(tmpRow, inRow, wRow, KW);

                                // Reduce sum of tmpRow elements to scalar
                                dtype rowSum = 0.0f;
                                for (uint32_t i = 0; i < KW; ++i) {
                                    rowSum += tmpRow.GetValue(i);
                                }
                                acc += rowSum;

                                tmpRowQ.FreeTensor(tmpRow);
                                inRowQ.FreeTensor(inRow);
                                wRowQ.FreeTensor(wRow);
                            }
                        }
                        // Write out scalar result
                        AscendC::LocalTensor<dtype> yScalar = yScalarQ.AllocTensor<dtype>();
                        yScalar.SetValue(0, acc);
                        int64_t y_off = ((static_cast<int64_t>(n) * Cout + oc) * Hout + oh) * Wout + ow;
                        AscendC::DataCopy(yGm[y_off], yScalar, 1);
                        yScalarQ.FreeTensor(yScalar);
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inRowQ;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> wRowQ;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> tmpRowQ;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> yScalarQ;

    AscendC::GlobalTensor<dtype> xGm;
    AscendC::GlobalTensor<dtype> wGm;
    AscendC::GlobalTensor<dtype> yGm;

    uint32_t N, Cin, Hin, Win, Cout, KH, KW, Hout, Wout, ocPerBlock;
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2D op;
    op.Init(x, weight, y,
            tiling_data.n(), tiling_data.c_in(), tiling_data.h(), tiling_data.w(),
            tiling_data.c_out(), tiling_data.kH(), tiling_data.kW(),
            tiling_data.outH(), tiling_data.outW(), tiling_data.ocPerBlock());
    op.Process();
}
