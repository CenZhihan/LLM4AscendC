kernel_src="""
#include "kernel_operator.h"

class KernelConvTransposed3D {
public:
    __aicore__ inline KernelConvTransposed3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t B, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t K, uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t Do, uint32_t Ho, uint32_t Wo)
    {
        this->B=B; this->Cin=Cin; this->Cout=Cout;
        this->D=D; this->H=H; this->W=W;
        this->K=K; this->stride=stride; this->padding=padding; this->dilation=dilation;
        this->Do=Do; this->Ho=Ho; this->Wo=Wo;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)B*Cin*D*H*W);
        wGm.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cin*Cout*K*K*K);
        if (bias != nullptr) {
            biasGm.SetGlobalBuffer((__gm__ float*)bias, Cout);
            hasBias = true;
        } else {
            hasBias = false;
        }
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint64_t)B*Cout*Do*Ho*Wo);

        const uint32_t zeroTile = 1024;
        pipe.InitBuffer(tmpQueueA, 1, zeroTile * sizeof(float)); // zero buffer
        pipe.InitBuffer(tmpQueueB, 1, sizeof(float)); // scalar temp buffers
        pipe.InitBuffer(tmpQueueC, 1, sizeof(float));
        pipe.InitBuffer(tmpQueueD, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        ZeroOutput();
        ComputeDeconv();
        if (hasBias) {
            AddBias();
        }
    }

private:
    __aicore__ inline uint64_t XIndex(uint32_t b, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const {
        return (((uint64_t)b * Cin + c) * D + d) * H * W + ((uint64_t)h * W + w);
    }
    __aicore__ inline uint64_t WIndex(uint32_t cIn, uint32_t cOut, uint32_t kd, uint32_t kh, uint32_t kw) const {
        return (((uint64_t)cIn * Cout + cOut) * K + kd) * K * K + ((uint64_t)kh * K + kw);
    }
    __aicore__ inline uint64_t YIndex(uint32_t b, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const {
        return (((uint64_t)b * Cout + c) * Do + d) * Ho * Wo + ((uint64_t)h * Wo + w);
    }

    __aicore__ inline void ZeroOutput()
    {
        const uint64_t total = (uint64_t)B * Cout * Do * Ho * Wo;
        const uint32_t tile = 1024;
        AscendC::LocalTensor<float> zeroLocal = tmpQueueA.AllocTensor<float>();
        AscendC::Duplicate(zeroLocal, 0.0f, tile);
        uint64_t done = 0;
        while (done < total) {
            uint64_t len = (total - done) > tile ? tile : (total - done);
            AscendC::DataCopy(yGm[done], zeroLocal, (uint32_t)len);
            done += len;
        }
        tmpQueueA.FreeTensor(zeroLocal);
        AscendC::PipeBarrier<PIPE_ALL>();
    }

    __aicore__ inline void ComputeDeconv()
    {
        // Local scalars
        AscendC::LocalTensor<float> xLocal = tmpQueueB.AllocTensor<float>(); // 1 elem
        AscendC::LocalTensor<float> wLocal = tmpQueueC.AllocTensor<float>(); // 1 elem
        AscendC::LocalTensor<float> prodLocal = tmpQueueD.AllocTensor<float>(); // 1 elem

        for (uint32_t b = 0; b < B; ++b) {
            for (uint32_t cIn = 0; cIn < Cin; ++cIn) {
                for (uint32_t id = 0; id < D; ++id) {
                    for (uint32_t ih = 0; ih < H; ++ih) {
                        for (uint32_t iw = 0; iw < W; ++iw) {
                            // Load input scalar
                            AscendC::DataCopy(xLocal, xGm[XIndex(b, cIn, id, ih, iw)], 1);

                            for (uint32_t kd = 0; kd < K; ++kd) {
                                int64_t od = (int64_t)id * (int64_t)stride - (int64_t)padding + (int64_t)kd * (int64_t)dilation;
                                if (od < 0 || od >= (int64_t)Do) continue;
                                for (uint32_t kh = 0; kh < K; ++kh) {
                                    int64_t oh = (int64_t)ih * (int64_t)stride - (int64_t)padding + (int64_t)kh * (int64_t)dilation;
                                    if (oh < 0 || oh >= (int64_t)Ho) continue;
                                    for (uint32_t kw = 0; kw < K; ++kw) {
                                        int64_t ow = (int64_t)iw * (int64_t)stride - (int64_t)padding + (int64_t)kw * (int64_t)dilation;
                                        if (ow < 0 || ow >= (int64_t)Wo) continue;

                                        for (uint32_t cOut = 0; cOut < Cout; ++cOut) {
                                            // w[cIn, cOut, kd, kh, kw]
                                            AscendC::DataCopy(wLocal, wGm[WIndex(cIn, cOut, kd, kh, kw)], 1);
                                            AscendC::Mul(prodLocal, xLocal, wLocal, 1);
                                            // y += prod
                                            // Load y
                                            AscendC::LocalTensor<float> yLocal = tmpQueueA.AllocTensor<float>(); // reuse big queue for scalar
                                            AscendC::DataCopy(yLocal, yGm[YIndex(b, cOut, (uint32_t)od, (uint32_t)oh, (uint32_t)ow)], 1);
                                            AscendC::Add(yLocal, yLocal, prodLocal, 1);
                                            AscendC::DataCopy(yGm[YIndex(b, cOut, (uint32_t)od, (uint32_t)oh, (uint32_t)ow)], yLocal, 1);
                                            tmpQueueA.FreeTensor(yLocal);
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
        tmpQueueB.FreeTensor(xLocal);
        tmpQueueC.FreeTensor(wLocal);
        tmpQueueD.FreeTensor(prodLocal);
        AscendC::PipeBarrier<PIPE_ALL>();
    }

    __aicore__ inline void AddBias()
    {
        // Add bias per channel: y[b, co, :, :, :] += bias[co]
        AscendC::LocalTensor<float> biasLocal = tmpQueueB.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = tmpQueueC.AllocTensor<float>();
        AscendC::LocalTensor<float> addLocal = tmpQueueD.AllocTensor<float>();

        for (uint32_t co = 0; co < Cout; ++co) {
            AscendC::DataCopy(biasLocal, biasGm[co], 1);
            for (uint32_t b = 0; b < B; ++b) {
                for (uint32_t od = 0; od < Do; ++od) {
                    for (uint32_t oh = 0; oh < Ho; ++oh) {
                        for (uint32_t ow = 0; ow < Wo; ++ow) {
                            AscendC::DataCopy(yLocal, yGm[YIndex(b, co, od, oh, ow)], 1);
                            AscendC::Add(addLocal, yLocal, biasLocal, 1);
                            AscendC::DataCopy(yGm[YIndex(b, co, od, oh, ow)], addLocal, 1);
                        }
                    }
                }
            }
        }
        tmpQueueB.FreeTensor(biasLocal);
        tmpQueueC.FreeTensor(yLocal);
        tmpQueueD.FreeTensor(addLocal);
        AscendC::PipeBarrier<PIPE_ALL>();
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueueA;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueueB;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueueC;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> tmpQueueD;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> biasGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t B, Cin, Cout;
    uint32_t D, H, W;
    uint32_t K, stride, padding, dilation;
    uint32_t Do, Ho, Wo;
    bool hasBias = false;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_padded_dilated_strided_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3D op;
    op.Init(x, weight, bias, y,
            tiling_data.B(), tiling_data.Cin(), tiling_data.Cout(),
            tiling_data.D(), tiling_data.H(), tiling_data.W(),
            tiling_data.K(), tiling_data.stride(), tiling_data.padding(), tiling_data.dilation(),
            tiling_data.Do(), tiling_data.Ho(), tiling_data.Wo());
    op.Process();
}
