kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConv2dSubtractSubtractMish {
public:
    __aicore__ inline KernelConv2dSubtractSubtractMish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR s1, GM_ADDR s2, GM_ADDR ones, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        s1Gm.SetGlobalBuffer((__gm__ DTYPE_Y *)s1 + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        s2Gm.SetGlobalBuffer((__gm__ DTYPE_Z *)s2 + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        oneGm.SetGlobalBuffer((__gm__ DTYPE_W *)ones + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_V *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueS1, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueS2, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(inQueueOne, BUFFER_NUM, this->tileLength * sizeof(DTYPE_W));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_V));
        pipe.InitBuffer(tmpQue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_V));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> s1Local = inQueueS1.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> s2Local = inQueueS2.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_W> oneLocal = inQueueOne.AllocTensor<DTYPE_W>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(s1Local, s1Gm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(s2Local, s2Gm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(oneLocal, oneGm[progress * this->tileLength], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueS1.EnQue(s1Local);
        inQueueS2.EnQue(s2Local);
        inQueueOne.EnQue(oneLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> s1Local = inQueueS1.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> s2Local = inQueueS2.DeQue<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_W> oneLocal = inQueueOne.DeQue<DTYPE_W>();

        AscendC::LocalTensor<DTYPE_V> yLocal = outQueueY.AllocTensor<DTYPE_V>();
        AscendC::LocalTensor<DTYPE_V> tmpLocal1 = tmpQue.AllocTensor<DTYPE_V>();
        AscendC::LocalTensor<DTYPE_V> tmpLocal2 = tmpQue.AllocTensor<DTYPE_V>();

        // step1: z = x + s1 (s1 is negative-filled tensor to perform subtraction)
        AscendC::Add(yLocal, xLocal, s1Local, this->tileLength);
        // step2: z = z + s2 (s2 is negative-filled tensor)
        AscendC::Add(yLocal, yLocal, s2Local, this->tileLength);

        // step3: compute softplus(yLocal) = log(1 + exp(yLocal))
        // tmpLocal1 = exp(yLocal)
        AscendC::Exp(tmpLocal1, yLocal, this->tileLength);
        // tmpLocal2 = tmpLocal1 + oneLocal  (add 1 element-wise)
        AscendC::Add(tmpLocal2, tmpLocal1, oneLocal, this->tileLength);
        // tmpLocal1 = log(tmpLocal2)
        AscendC::Log(tmpLocal1, tmpLocal2, this->tileLength);

        // step4: tanh_softplus = tanh(tmpLocal1)
        AscendC::Tanh(tmpLocal2, tmpLocal1, this->tileLength);

        // step5: y = yLocal * tanh_softplus
        AscendC::Mul(yLocal, yLocal, tmpLocal2, this->tileLength);

        outQueueY.EnQue<DTYPE_V>(yLocal);

        tmpQue.FreeTensor(tmpLocal2);
        tmpQue.FreeTensor(tmpLocal1);
        inQueueOne.FreeTensor(oneLocal);
        inQueueS2.FreeTensor(s2Local);
        inQueueS1.FreeTensor(s1Local);
        inQueueX.FreeTensor(xLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_V> yLocal = outQueueY.DeQue<DTYPE_V>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueS1, inQueueS2, inQueueOne;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> s1Gm;
    AscendC::GlobalTensor<DTYPE_Z> s2Gm;
    AscendC::GlobalTensor<DTYPE_W> oneGm;
    AscendC::GlobalTensor<DTYPE_V> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_subtract_subtract_mish_custom(GM_ADDR x, GM_ADDR s1, GM_ADDR s2, GM_ADDR ones, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractSubtractMish op;
    op.Init(x, s1, s2, ones, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused kernel after conv: (x - v1) - v2, then Mish activation inside custom kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract_value_1, subtract_value_2):
        super().__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract_value_1 = subtract_value_1
        self.subtract_value_2 = subtract_value_2

    def forward(self, x):
        x = self.conv(x)
        # 构造与x形状一致的张量：-v1、-v2 和 全1
        s1 = torch.full_like(x, -self.subtract_value_1)
        s2 = torch.full_like(x, -self.subtract_value_2)
        ones = torch.ones_like(x)
        # 自定义AscendC算子：结果为 Mish((x + s1) + s2) = Mish(x - v1 - v2)
        return custom_ops_lib.conv2d_subtract_subtract_mish_custom(x, s1, s2, ones)
'''
