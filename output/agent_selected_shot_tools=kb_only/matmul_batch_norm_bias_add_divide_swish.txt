#include "kernel_operator.h"
#include "lib/matmul_intf.h"

// Data types for Matmul
using A_T = half;   // x
using B_T = half;   // weight
using C_T = float;  // output and post-process

constexpr int32_t BUFFER_NUM = 2;

struct ExtraTiling {
    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileN;
};

class KernelMatmulBnBiasDivSwish {
public:
    __aicore__ inline KernelMatmulBnBiasDivSwish() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR biasLinear,
                                GM_ADDR bnScale, GM_ADDR bnShift,
                                GM_ADDR biasAdd, GM_ADDR divideValue,
                                GM_ADDR y, GM_ADDR tilingGM)
    {
        // Load TCubeTiling then Extra
        // Copy raw tiling buffer into local structs
        // First TCubeTiling
        uint32_t tcSize = sizeof(TCubeTiling);
        __gm__ uint8_t* tgm = reinterpret_cast<__gm__ uint8_t*>(tilingGM);

        // Copy TCubeTiling
        uint8_t* tcLocal = reinterpret_cast<uint8_t*>(&cubeTiling_);
        for (uint32_t i = 0; i < tcSize; ++i) {
            tcLocal[i] = tgm[i];
        }
        // Copy Extra
        ExtraTiling extraTmp = {0};
        uint8_t* extraLocal = reinterpret_cast<uint8_t*>(&extraTmp);
        uint32_t extraSize = sizeof(ExtraTiling);
        for (uint32_t i = 0; i < extraSize; ++i) {
            extraLocal[i] = tgm[tcSize + i];
        }
        M_ = extraTmp.M;
        N_ = extraTmp.N;
        K_ = extraTmp.K;
        tileN_ = extraTmp.tileN;

        // Set GM pointers
        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(x), static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(w), static_cast<uint64_t>(N_) * K_);
        biasLinearGm_.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(biasLinear), N_);
        scaleGm_.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(bnScale), N_);
        shiftGm_.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(bnShift), N_);
        biasAddGm_.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(biasAdd), 1);
        divGm_.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(divideValue), 1);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(y), static_cast<uint64_t>(M_) * N_);

        // Init pipe and queues for post-process
        pipe_.InitBuffer(inQueueC_, BUFFER_NUM, tileN_ * sizeof(C_T));
        pipe_.InitBuffer(inQueueScale_, BUFFER_NUM, tileN_ * sizeof(C_T));
        pipe_.InitBuffer(inQueueShift_, BUFFER_NUM, tileN_ * sizeof(C_T));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileN_ * sizeof(C_T));

        // Register Matmul object
        g_cubeTPipePtr = &pipe_;
        REGIST_MATMUL_OBJ(&pipe_, GetSysWorkSpacePtr(), mm_);
        mm_.Init(&cubeTiling_);
        // Set inputs for Matmul
        mm_.SetTensorA(xGm_);
        // weight is [N, K], treat as transposed so that Matmul sees [K, N]
        mm_.SetTensorB(wGm_, true);
        mm_.SetBias(biasLinearGm_);
    }

    __aicore__ inline void Run()
    {
        // Matmul: C = A x B + bias_linear
        mm_.IterateAll(yGm_);

        // Read scalar biasAdd and divide
        C_T biasAddScalar = 0.f;
        C_T divideScalar = 1.f;
        {
            // Load 1 element to local
            AscendC::LocalTensor<C_T> tmp = inQueueC_.AllocTensor<C_T>();
            AscendC::DataCopy(tmp, biasAddGm_, 1);
            biasAddScalar = tmp.GetValue(0);
            inQueueC_.FreeTensor(tmp);
        }
        {
            AscendC::LocalTensor<C_T> tmp = inQueueC_.AllocTensor<C_T>();
            AscendC::DataCopy(tmp, divGm_, 1);
            divideScalar = tmp.GetValue(0);
            inQueueC_.FreeTensor(tmp);
        }

        // Post-process per row, chunk along N
        for (uint32_t m = 0; m < M_; ++m) {
            for (uint32_t nOff = 0; nOff < N_; nOff += tileN_) {
                uint32_t curN = AscendC::min(tileN_, N_ - nOff);

                // CopyIn
                CopyIn(m, nOff, curN);

                // Compute fused: y = ((y * scale) + shift + biasAddScalar) / divideScalar ; y = y * sigmoid(y)
                Compute(curN, biasAddScalar, divideScalar);

                // CopyOut
                CopyOut(m, nOff, curN);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t m, uint32_t nOff, uint32_t curN)
    {
        uint64_t base = static_cast<uint64_t>(m) * N_ + nOff;

        AscendC::LocalTensor<C_T> cLocal = inQueueC_.AllocTensor<C_T>();
        AscendC::LocalTensor<C_T> sLocal = inQueueScale_.AllocTensor<C_T>();
        AscendC::LocalTensor<C_T> tLocal = inQueueShift_.AllocTensor<C_T>();

        AscendC::DataCopy(cLocal, yGm_[base], curN);
        AscendC::DataCopy(sLocal, scaleGm_[nOff], curN);
        AscendC::DataCopy(tLocal, shiftGm_[nOff], curN);

        inQueueC_.EnQue(cLocal);
        inQueueScale_.EnQue(sLocal);
        inQueueShift_.EnQue(tLocal);
    }

    __aicore__ inline void Compute(uint32_t curN, C_T biasAddScalar, C_T divideScalar)
    {
        AscendC::LocalTensor<C_T> cLocal = inQueueC_.DeQue<C_T>();
        AscendC::LocalTensor<C_T> sLocal = inQueueScale_.DeQue<C_T>();
        AscendC::LocalTensor<C_T> tLocal = inQueueShift_.DeQue<C_T>();
        AscendC::LocalTensor<C_T> yLocal = outQueueY_.AllocTensor<C_T>();
        AscendC::LocalTensor<C_T> tmpLocal = inQueueC_.AllocTensor<C_T>(); // temp for sigmoid

        // y = c * scale
        AscendC::Mul(yLocal, cLocal, sLocal, curN);
        // y = y + shift
        AscendC::Add(yLocal, yLocal, tLocal, curN);
        // y = y + biasAddScalar
        AscendC::Adds(yLocal, yLocal, biasAddScalar, curN);
        // y = y / divideScalar
        AscendC::Divs(yLocal, yLocal, divideScalar, curN);
        // tmp = sigmoid(y)
        AscendC::Sigmoid(tmpLocal, yLocal, curN);
        // y = y * tmp
        AscendC::Mul(yLocal, yLocal, tmpLocal, curN);

        outQueueY_.EnQue<C_T>(yLocal);

        // free inputs
        inQueueC_.FreeTensor(cLocal);
        inQueueScale_.FreeTensor(sLocal);
        inQueueShift_.FreeTensor(tLocal);
        inQueueC_.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(uint32_t m, uint32_t nOff, uint32_t curN)
    {
        uint64_t base = static_cast<uint64_t>(m) * N_ + nOff;

        AscendC::LocalTensor<C_T> yLocal = outQueueY_.DeQue<C_T>();
        AscendC::DataCopy(yGm_[base], yLocal, curN);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    // GM tensors
    AscendC::GlobalTensor<A_T> xGm_;
    AscendC::GlobalTensor<B_T> wGm_;
    AscendC::GlobalTensor<C_T> biasLinearGm_;
    AscendC::GlobalTensor<C_T> scaleGm_;
    AscendC::GlobalTensor<C_T> shiftGm_;
    AscendC::GlobalTensor<C_T> biasAddGm_;
    AscendC::GlobalTensor<C_T> divGm_;
    AscendC::GlobalTensor<C_T> yGm_;

    // Pipe and queues
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueC_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueScale_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueShift_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    // Matmul
    matmul::Matmul<
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, A_T, false>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, B_T, true>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T>
    > mm_;
    TCubeTiling cubeTiling_;

    // Shapes/tiles
    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileN_{0};
};

extern "C" __global__ __aicore__
void matmul_batch_norm_bias_add_divide_swish_custom(GM_ADDR x,
                                                    GM_ADDR weight,
                                                    GM_ADDR bias_linear,
                                                    GM_ADDR bn_scale,
                                                    GM_ADDR bn_shift,
                                                    GM_ADDR bias_add,
                                                    GM_ADDR divide_value,
                                                    GM_ADDR y,
                                                    GM_ADDR workspace,
                                                    GM_ADDR tiling)
{
    KernelMatmulBnBiasDivSwish op;
    op.Init(x, weight, bias_linear, bn_scale, bn_shift, bias_add, divide_value, y, tiling);
    op.Run();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom AscendC kernel: Matmul -> BatchNorm (inference-style) -> BiasAdd -> Divide -> Swish
    Note:
      - The kernel expects x and weight in float16 for Matmul efficiency.
      - The BN is applied with precomputed scale/shift:
         scale = gamma * rsqrt(running_var + eps)
         shift = beta - running_mean * scale
    """
    def __init__(self, in_features, out_features, bn_eps=1e-5, bn_momentum=0.1, bias_shape=(1,), divide_value=1.0):
        super().__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.bn = torch.nn.BatchNorm1d(out_features, eps=bn_eps, momentum=bn_momentum)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.divide_value = divide_value
        self.eps = bn_eps

    def forward(self, x: torch.Tensor):
        # Prepare inputs for fused kernel
        # Precompute scale and shift for BN
        # Use running stats to emulate inference-style BN inside fused op
        running_mean = self.bn.running_mean
        running_var = self.bn.running_var
        gamma = self.bn.weight
        beta = self.bn.bias

        inv_std = torch.rsqrt(running_var + self.eps)
        scale = gamma * inv_std
        shift = beta - running_mean * scale

        # Extract Linear weights/bias
        weight = self.matmul.weight
        bias_linear = self.matmul.bias

        # Prepare additional tensors
        # bias_add is scalar tensor (shape [1]) and divide_value as tensor [1]
        bias_add = self.bias.reshape(1).to(bias_linear.dtype)
        divide_value = torch.tensor([self.divide_value], dtype=bias_linear.dtype, device=x.device)

        # Cast x and weight to float16 for Matmul; others remain float32
        x_half = x.to(torch.float16)
        w_half = weight.to(torch.float16)

        # Call custom fused kernel
        y = custom_ops_lib.matmul_batch_norm_bias_add_divide_swish_custom(
            x_half, w_half, bias_linear, scale, shift, bias_add, divide_value
        )
        return y
'''
