kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensors per queue

class KernelDivisionBroadcast {
public:
    __aicore__ inline KernelDivisionBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR z, uint32_t n, uint32_t c, uint32_t hw, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->hw = hw;
        this->tileNum = tileNum;

        uint32_t totalSlices = n * c; // each slice is one (n,c) with length hw
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t baseSlices = totalSlices / blockNum;

        this->sliceStart = baseSlices * blockIdx;
        this->sliceCount = baseSlices;

        this->loopCount = this->tileNum * BUFFER_NUM;
        this->tileLength = (this->hw / this->loopCount);
        if (this->tileLength == 0) {
            this->tileLength = this->hw; // degenerate case, one tile
            this->loopCount = 1;
        }

        uint32_t baseElemOffset = this->sliceStart * this->hw;

        xGm.SetGlobalBuffer((__gm__ float *)x + baseElemOffset, this->sliceCount * this->hw);
        zGm.SetGlobalBuffer((__gm__ float *)z + baseElemOffset, this->sliceCount * this->hw);
        scaleGm.SetGlobalBuffer((__gm__ float *)scale, this->c);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueScale, 1, this->c * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Load full scale vector [C]
        AscendC::LocalTensor<float> scaleLocal = inQueueScale.AllocTensor<float>();
        AscendC::DataCopy(scaleLocal, scaleGm, c);
        inQueueScale.EnQue(scaleLocal);
        AscendC::LocalTensor<float> scaleBuf = inQueueScale.DeQue<float>();

        for (uint32_t s = 0; s < this->sliceCount; ++s) {
            // compute channel index for this slice
            uint32_t globalSlice = this->sliceStart + s;
            uint32_t cIdx = globalSlice % this->c;

            for (uint32_t t = 0; t < this->loopCount; ++t) {
                uint32_t processed = t * this->tileLength;
                uint32_t remain = this->hw - processed;
                uint32_t copyLen = (t == this->loopCount - 1) ? remain : this->tileLength;

                CopyIn(s, t, copyLen);
                Compute(scaleBuf, cIdx, copyLen);
                CopyOut(s, t, copyLen);
            }
        }

        inQueueScale.FreeTensor(scaleBuf);
    }

private:
    __aicore__ inline void CopyIn(uint32_t s, uint32_t t, uint32_t copyLen)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        uint32_t offset = s * this->hw + t * this->tileLength;
        AscendC::DataCopy(xLocal, xGm[offset], copyLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(const AscendC::LocalTensor<float>& scaleBuf, uint32_t cIdx, uint32_t copyLen)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
        float sVal = scaleBuf.GetValue(cIdx);
        float inv = 1.0f / sVal;
        AscendC::Muls(zLocal, xLocal, inv, copyLen);
        outQueueZ.EnQue<float>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t s, uint32_t t, uint32_t copyLen)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        uint32_t offset = s * this->hw + t * this->tileLength;
        AscendC::DataCopy(zGm[offset], zLocal, copyLen);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueScale;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;
    AscendC::GlobalTensor<float> scaleGm;

    uint32_t n;
    uint32_t c;
    uint32_t hw;
    uint32_t tileNum;

    uint32_t sliceStart;
    uint32_t sliceCount;
    uint32_t loopCount;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void division_broadcast_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDivisionBroadcast op;
    op.Init(x, scale, z, tiling_data.n, tiling_data.c, tiling_data.hw, tiling_data.tileNum);
    op.Process();
}
