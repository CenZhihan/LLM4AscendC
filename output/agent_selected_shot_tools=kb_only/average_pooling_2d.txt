host_operator_src="""
#include "average_pooling_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1) / b);
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();

    uint32_t N = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t C = static_cast<uint32_t>(origin.GetDim(1));
    uint32_t H = static_cast<uint32_t>(origin.GetDim(2));
    uint32_t W = static_cast<uint32_t>(origin.GetDim(3));

    int64_t ksize_attr = 1;
    int64_t stride_attr = 0;
    int64_t padding_attr = 0;
    (void)context->GetAttr<int64_t>("ksize", ksize_attr);
    (void)context->GetAttr<int64_t>("stride", stride_attr);
    (void)context->GetAttr<int64_t>("padding", padding_attr);

    uint32_t K = static_cast<uint32_t>(ksize_attr);
    uint32_t S = static_cast<uint32_t>(stride_attr == 0 ? ksize_attr : stride_attr);
    uint32_t P = static_cast<uint32_t>(padding_attr);

    // out = floor((H + 2P - K)/S) + 1
    uint32_t out_h = static_cast<uint32_t>((static_cast<int64_t>(H) + 2 * static_cast<int64_t>(P) - static_cast<int64_t>(K)) / static_cast<int64_t>(S) + 1);
    uint32_t out_w = static_cast<uint32_t>((static_cast<int64_t>(W) + 2 * static_cast<int64_t>(P) - static_cast<int64_t>(K)) / static_cast<int64_t>(S) + 1);

    uint64_t totalElems64 = static_cast<uint64_t>(N) * C * out_h * out_w;
    uint32_t totalElems = static_cast<uint32_t>(totalElems64);
    uint32_t elemsPerBlock = CeilDiv(totalElems64, BLOCK_DIM);

    context->SetBlockDim(BLOCK_DIM);

    AveragePooling2dCustomTilingData tiling;
    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_ksize(K);
    tiling.set_stride(S);
    tiling.set_padding(P);
    tiling.set_out_h(out_h);
    tiling.set_out_w(out_w);
    tiling.set_elemsPerBlock(elemsPerBlock);
    tiling.set_totalElems(totalElems);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();

    int64_t N = origin.GetDim(0);
    int64_t C = origin.GetDim(1);
    int64_t H = origin.GetDim(2);
    int64_t W = origin.GetDim(3);

    int64_t ksize_attr = 1;
    int64_t stride_attr = 0;
    int64_t padding_attr = 0;
    (void)context->GetAttr<int64_t>("ksize", ksize_attr);
    (void)context->GetAttr<int64_t>("stride", stride_attr);
    (void)context->GetAttr<int64_t>("padding", padding_attr);

    int64_t S = (stride_attr == 0) ? ksize_attr : stride_attr;
    int64_t P = padding_attr;

    int64_t out_h = (H + 2 * P - ksize_attr) / S + 1;
    int64_t out_w = (W + 2 * P - ksize_attr) / S + 1;

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, out_h);
    y_shape->SetDim(3, out_w);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

class KernelAvgPool2D {
public:
    __aicore__ inline KernelAvgPool2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t K, uint32_t S, uint32_t P,
                                uint32_t OH, uint32_t OW,
                                uint32_t elemsPerBlock, uint32_t totalElems)
    {
        this->N = N; this->C = C; this->H = H; this->W = W;
        this->K = K; this->S = S; this->P = P;
        this->OH = OH; this->OW = OW;
        this->elemsPerBlock = elemsPerBlock;
        this->totalElems = totalElems;

        uint32_t base = elemsPerBlock * AscendC::GetBlockIdx();
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint32_t)(N * C * H * W));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, (uint32_t)(N * C * OH * OW));
        baseOffset = base;
        pipe.InitBuffer(tmpQue, 1, 2 * sizeof(DTYPE_X)); // for accum and temp
    }

    __aicore__ inline void Process()
    {
        // Allocate 1-element temporaries in UB
        AscendC::LocalTensor<DTYPE_X> accum = tmpQue.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> temp  = tmpQue.AllocTensor<DTYPE_X>();

        DTYPE_X zero = static_cast<DTYPE_X>(0.0f);
        DTYPE_X invDiv = static_cast<DTYPE_X>(1.0f / (float)(K * K));

        uint32_t start = baseOffset;
        uint32_t end = AscendC::MIN(baseOffset + elemsPerBlock, totalElems);

        for (uint32_t idx = start; idx < end; ++idx) {
            // map linear index to (n, c, oh, ow)
            uint32_t rem = idx;
            uint32_t nc = OH * OW;
            uint32_t n = rem / (C * nc);
            rem = rem % (C * nc);
            uint32_t c = rem / nc;
            rem = rem % nc;
            uint32_t oh = rem / OW;
            uint32_t ow = rem % OW;

            // Initialize accumulator
            AscendC::Duplicate(accum, zero, 1);

            // window top-left (with padding)
            int32_t h_start = (int32_t)oh * (int32_t)S - (int32_t)P;
            int32_t w_start = (int32_t)ow * (int32_t)S - (int32_t)P;

            // Sum over KxK window (count_include_pad = true)
            for (uint32_t kh = 0; kh < K; ++kh) {
                int32_t h = h_start + (int32_t)kh;
                if (h < 0 || h >= (int32_t)H) {
                    continue; // outside input -> treated as zero
                }
                for (uint32_t kw = 0; kw < K; ++kw) {
                    int32_t w = w_start + (int32_t)kw;
                    if (w < 0 || w >= (int32_t)W) {
                        continue; // outside input -> treated as zero
                    }
                    uint32_t x_off = ((n * C + c) * H + (uint32_t)h) * W + (uint32_t)w;
                    AscendC::DataCopy(temp, xGm[x_off], 1);
                    AscendC::Add(accum, accum, temp, 1);
                }
            }

            // Multiply by reciprocal of (K*K)
            // temp <- invDiv
            AscendC::Duplicate(temp, invDiv, 1);
            AscendC::Mul(accum, accum, temp, 1);

            // Write back to y
            uint32_t y_off = ((n * C + c) * OH + oh) * OW + ow;
            AscendC::DataCopy(yGm[y_off], accum, 1);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, C, H, W;
    uint32_t K, S, P;
    uint32_t OH, OW;
    uint32_t elemsPerBlock, totalElems, baseOffset;
};

extern "C" __global__ __aicore__ void average_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAvgPool2D op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.ksize, tiling_data.stride, tiling_data.padding,
            tiling_data.out_h, tiling_data.out_w,
            tiling_data.elemsPerBlock, tiling_data.totalElems);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include \"pytorch_npu_helper.hpp\"
#include <torch/extension.h>

static inline int64_t div_floor(int64_t a, int64_t b) {
    return a / b;
}

at::Tensor average_pooling_2d_custom_impl_npu(const at::Tensor& x, int64_t ksize, int64_t stride, int64_t padding) {
    TORCH_CHECK(x.dim() == 4, \"Input must be 4D NCHW\");
    int64_t N = x.size(0), C = x.size(1), H = x.size(2), W = x.size(3);
    if (stride <= 0) stride = ksize;
    int64_t out_h = div_floor(H + 2 * padding - ksize, stride) + 1;
    int64_t out_w = div_floor(W + 2 * padding - ksize, stride) + 1;
    TORCH_CHECK(out_h > 0 && out_w > 0, \"Invalid output shape computed\");
    at::Tensor y = at::empty({N, C, out_h, out_w}, x.options());
    EXEC_NPU_CMD(aclnnAveragePooling2dCustom, x, (int)ksize, (int)stride, (int)padding, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"average_pooling_2d_custom\", &average_pooling_2d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"average_pooling_2d_custom\", &average_pooling_2d_custom_impl_npu, \"AveragePool2D custom\");
}
