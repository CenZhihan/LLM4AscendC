kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelConvTranspose2dMinSumGeluAdd {
public:
    __aicore__ inline KernelConvTranspose2dMinSumGeluAdd() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z, uint32_t n, uint32_t c, uint32_t h, uint32_t w)
    {
        this->N = n;
        this->C = c;
        this->H = h;
        this->W = w;
        this->planeSize = h * w;
        this->nStride = c * h * w;
        this->cStride = h * w;
        this->outStride = w;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, static_cast<uint32_t>(N * nStride));
        biasGm.SetGlobalBuffer((__gm__ DTYPE*)bias, static_cast<uint32_t>(W));
        zGm.SetGlobalBuffer((__gm__ DTYPE*)z, static_cast<uint32_t>(N * outStride));

        // 队列与缓冲区
        pipe.InitBuffer(inQueuePlane, 1, planeSize * sizeof(DTYPE));     // 每次读入单通道平面 [H*W]
        pipe.InitBuffer(tmpMinQ, 1, planeSize * sizeof(DTYPE));           // 维护当前min平面
        pipe.InitBuffer(tmpRowQ, 1, W * sizeof(DTYPE));                   // 每行缓存 [W]
        pipe.InitBuffer(tmpSumQ, 1, W * sizeof(DTYPE));                   // 按宽度聚合的和 [W]
        pipe.InitBuffer(biasQ, 1, W * sizeof(DTYPE));                     // bias向量 [W]
        pipe.InitBuffer(outQueueZ, 1, W * sizeof(DTYPE));                 // 输出 [W]
    }

    __aicore__ inline void Process()
    {
        // 预取bias向量（长度为W），由上层保证bias已扩展到W
        AscendC::LocalTensor<DTYPE> biasLocal = biasQ.AllocTensor<DTYPE>();
        AscendC::DataCopy(biasLocal, biasGm, W);
        biasQ.EnQue(biasLocal);

        for (uint32_t n = 0; n < N; ++n) {
            // min over C -> [H*W]
            AscendC::LocalTensor<DTYPE> minLocal = tmpMinQ.AllocTensor<DTYPE>();
            // c=0
            AscendC::DataCopy(minLocal, xGm[n * nStride + 0 * cStride], planeSize);
            for (uint32_t c = 1; c < C; ++c) {
                AscendC::LocalTensor<DTYPE> planeLocal = inQueuePlane.AllocTensor<DTYPE>();
                AscendC::DataCopy(planeLocal, xGm[n * nStride + c * cStride], planeSize);
                AscendC::Min(minLocal, minLocal, planeLocal, planeSize);
                inQueuePlane.FreeTensor(planeLocal);
            }

            // sum over H -> [W]
            AscendC::LocalTensor<DTYPE> sumLocal = tmpSumQ.AllocTensor<DTYPE>();
            AscendC::Duplicate(sumLocal, static_cast<DTYPE>(0.0f), W);
            for (uint32_t hh = 0; hh < H; ++hh) {
                AscendC::LocalTensor<DTYPE> rowLocal = tmpRowQ.AllocTensor<DTYPE>();
                // 从minLocal拷贝一行到rowLocal
                AscendC::Copy(rowLocal, minLocal[hh * W], W);
                AscendC::Add(sumLocal, sumLocal, rowLocal, W);
                tmpRowQ.FreeTensor(rowLocal);
            }

            // GELU + Add(bias)
            AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.AllocTensor<DTYPE>();
            AscendC::Gelu(zLocal, sumLocal, W);

            AscendC::LocalTensor<DTYPE> biasLocalUse = biasQ.DeQue<DTYPE>();
            AscendC::Add(zLocal, zLocal, biasLocalUse, W);
            biasQ.EnQue(biasLocalUse);

            // 写回 [W]
            AscendC::DataCopy(zGm[n * outStride], zLocal, W);

            // 释放本轮缓存
            outQueueZ.FreeTensor(zLocal);
            tmpSumQ.FreeTensor(sumLocal);
            tmpMinQ.FreeTensor(minLocal);
        }

        // 释放bias
        AscendC::LocalTensor<DTYPE> biasLocalEnd = biasQ.DeQue<DTYPE>();
        biasQ.FreeTensor(biasLocalEnd);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueuePlane;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpMinQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpRowQ;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpSumQ;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> biasQ;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueZ;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> biasGm;
    AscendC::GlobalTensor<DTYPE> zGm;

    uint32_t N, C, H, W;
    uint32_t planeSize;   // H*W
    uint32_t nStride;     // C*H*W
    uint32_t cStride;     // H*W
    uint32_t outStride;   // W
};

extern "C" __global__ __aicore__ void conv_transpose2d_min_sum_gelu_add_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose2dMinSumGeluAdd op;
    op.Init(x, bias, z, tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    替换Min/Sum/GELU/Add阶段为自定义AscendC算子；反卷积仍使用PyTorch原生实现。
    输入: x [N,C,H,W]
    输出: [N,1,1,W]
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride, padding, output_padding)
        # 训练形状为(1,1,1)的标量bias
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        # 1) 反卷积
        y = self.conv_transpose(x)  # [N, C_out, H_out, W_out]
        # 2) 自定义算子: 沿通道取最小 -> 沿高度求和 -> GELU -> 加bias
        #    自定义算子期望第二输入为长度为W_out的一维bias向量
        W = y.size(-1)
        bias_vec = self.bias.view(-1).expand(W).contiguous()
        return custom_ops_lib.conv_transpose2d_min_sum_gelu_add_custom(y, bias_vec)

def get_inputs():
    batch_size = 16
    in_channels = 64
    height, width = 128, 128
    return [torch.rand(batch_size, in_channels, height, width)]

def get_init_inputs():
    batch_size = 16
    in_channels = 64
    out_channels = 128
    height, width = 128, 128
    kernel_size = 3
    stride = 2
    padding = 1
    output_padding = 1
    bias_shape = (1, 1, 1)
    return [in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape]
