kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor个数

// 计算：z = x + add； h = hardswish(z) = z * clamp(z + 3, 0, 6) / 6； y = z * h
class KernelConvTranspose3dAddHardSwish {
public:
    __aicore__ inline KernelConvTranspose3dAddHardSwish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        addGm.SetGlobalBuffer((__gm__ DTYPE_Y *)add + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueAdd, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));   // 临时缓冲
        pipe.InitBuffer(tmpQueue2, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));  // 第二个临时缓冲
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> addLocal = inQueueAdd.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(addLocal, addGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueAdd.EnQue(addLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> addLocal = inQueueAdd.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> tmpLocal = tmpQueue.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> tmpLocal2 = tmpQueue2.AllocTensor<DTYPE_Z>();

        // z = x + add
        AscendC::Add(tmpLocal, xLocal, addLocal, this->tileLength);

        // t = z + 3
        AscendC::Adds(tmpLocal2, tmpLocal, static_cast<DTYPE_Z>(3.0f), this->tileLength);
        // t = clamp(t, 0, 6) = min(max(t, 0), 6)
        AscendC::Maxs(tmpLocal2, tmpLocal2, static_cast<DTYPE_Z>(0.0f), this->tileLength);
        AscendC::Mins(tmpLocal2, tmpLocal2, static_cast<DTYPE_Z>(6.0f), this->tileLength);
        // h = z * (t / 6) = z * t * (1/6)
        AscendC::Muls(tmpLocal2, tmpLocal2, static_cast<DTYPE_Z>(1.0f / 6.0f), this->tileLength);
        AscendC::Mul(tmpLocal2, tmpLocal, tmpLocal2, this->tileLength);
        // y = z * h
        AscendC::Mul(yLocal, tmpLocal, tmpLocal2, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);

        tmpQueue.FreeTensor(tmpLocal);
        tmpQueue2.FreeTensor(tmpLocal2);
        inQueueX.FreeTensor(xLocal);
        inQueueAdd.FreeTensor(addLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueAdd;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue, tmpQueue2;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> addGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_add_hard_swish_custom(GM_ADDR x, GM_ADDR add, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dAddHardSwish op;
    op.Init(x, add, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版：保留 ConvTranspose3d，由自定义 AscendC kernel 融合完成 add + HardSwish 与额外乘法 (x * HardSwish(x))。
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding
        )
        # 原模型中存在但未在 forward 显式使用的 bias，这里保持以对齐参数初始化接口
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x, add_input):
        # 先由 PyTorch 完成 ConvTranspose3d
        z = self.conv_transpose(x)
        # 由自定义 AscendC kernel 完成： y = (z + add) * hardswish(z + add)
        y = custom_ops_lib.conv_transpose3d_add_hard_swish_custom(z, add_input)
        return y
