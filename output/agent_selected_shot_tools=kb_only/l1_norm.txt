host_operator_src="""
#include "l1_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 48;

static ge::graphStatus TilingFunc(gert::TilingContext* context) {
  L1NormCustomTilingData tiling;
  const gert::StorageShape* x_shape = context->GetInputShape(0);
  const gert::Shape shape = x_shape->GetStorageShape();
  auto totalRow = static_cast<uint32_t>(shape.GetDim(0));
  auto colNum = static_cast<uint32_t>(shape.GetDim(1));
  auto coreRowNum = totalRow / BLOCK_DIM;

  tiling.set_rowNum(coreRowNum);
  tiling.set_rowNumSp(coreRowNum + 1);
  tiling.set_rowLength(colNum);
  tiling.set_blockPivot(totalRow - coreRowNum * BLOCK_DIM);

  // 为了兼容超大宽度，按行处理，tileLoop=1，避免UB压力
  uint32_t tileLoop = 1;
  tiling.set_tileLoop(tileLoop);
  tiling.set_tileLength(tileLoop * colNum);
  tiling.set_loopCount(coreRowNum / tileLoop);

  tiling.set_factor(1.0f / static_cast<float>(colNum));

  context->SetBlockDim(BLOCK_DIM);
  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
  context->SetTilingKey(1);

  return ge::GRAPH_SUCCESS;
}
}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context) {
  const gert::Shape* x_shape = context->GetInputShape(0);
  gert::Shape* y_shape = context->GetOutputShape(0);
  *y_shape = *x_shape;
  return GRAPH_SUCCESS;
}
}  // namespace ge

namespace ops {
class L1NormCustom : public OpDef {
 public:
  explicit L1NormCustom(const char* name) : OpDef(name) {
    this->Input("x")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});
    this->Output("res_out")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});

    this->SetInferShape(ge::InferShape);

    this->AICore().SetTiling(optiling::TilingFunc);
    this->AICore().AddConfig("ascend910b");
  }
};

OP_ADD(L1NormCustom);
}

kernel_src="""
#include "kernel_operator.h"
constexpr int32_t BUFFER_NUM = 1;

class KernelL1Norm {
 public:
  __aicore__ inline KernelL1Norm() {}
  __aicore__ inline void InitTiling(GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    rowNum = tiling_data.rowNum;
    rowNumSp = tiling_data.rowNumSp;
    rowLength = tiling_data.rowLength;
    blockPivot = tiling_data.blockPivot;
    tileLoop = tiling_data.tileLoop;
    tileLength = tiling_data.tileLength;
    loopCount = tiling_data.loopCount;
    factor = tiling_data.factor;
  }

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, GM_ADDR tiling) {
    InitTiling(tiling);

    this->leftRow = this->rowNum % this->tileLoop;
    if (AscendC::GetBlockIdx() < this->blockPivot) {
      this->rowNum = this->rowNumSp;
      this->leftRow += 1;
    }

    this->blockLength = this->rowNum * this->rowLength;
    uint32_t offset = 0;
    if (AscendC::GetBlockIdx() < this->blockPivot) {
      offset = this->blockLength * AscendC::GetBlockIdx();
    } else {
      offset = this->blockLength * AscendC::GetBlockIdx() +
               this->rowLength * this->blockPivot;
    }

    xGm.SetGlobalBuffer((__gm__ float *)x + offset, this->blockLength);
    zGm.SetGlobalBuffer((__gm__ float *)z + offset, this->blockLength);

    pipe.InitBuffer(queueX, BUFFER_NUM, this->tileLength * sizeof(float));
    pipe.InitBuffer(queueZ, BUFFER_NUM, this->tileLength * sizeof(float));

    // 用于ReduceSum的临时缓冲
    pipe.InitBuffer(tmpBufferReduce, 64 * sizeof(float));
    // 用于存放每一行的(和/均值/倒数)标量，长度为tileLoop
    pipe.InitBuffer(rowScalarBuf, this->tileLoop * sizeof(float));
  }

  __aicore__ inline void Process() {
    for (int32_t i = 0; i < this->loopCount; i++) {
      CopyIn(i, this->tileLoop);
      Compute(i, this->tileLoop);
      CopyOut(i, this->tileLoop);
    }
    if (this->leftRow > 0) {
      CopyIn(this->loopCount, this->leftRow);
      Compute(this->loopCount, this->leftRow);
      CopyOut(this->loopCount, this->leftRow);
    }
  }

 private:
  __aicore__ inline void CopyIn(int32_t progress, int32_t rows) {
    AscendC::LocalTensor<float> xLocal = queueX.AllocTensor<float>();
    AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->rowLength * rows);
    queueX.EnQue(xLocal);
  }

  __aicore__ inline void Compute(int32_t progress, int32_t rows) {
    AscendC::LocalTensor<float> xLocal = queueX.DeQue<float>();
    AscendC::LocalTensor<float> zLocal = queueZ.AllocTensor<float>();
    AscendC::LocalTensor<float> tmpReduce = tmpBufferReduce.Get<float>();
    AscendC::LocalTensor<float> rowScalars = rowScalarBuf.Get<float>();

    // 先将原始x复制到输出缓冲，后续按行缩放得到最终结果
    AscendC::DataCopy(zLocal, xLocal, this->rowLength * rows);

    // 逐行计算abs和 -> 均值 -> 倒数
    for (uint32_t j = 0; j < (uint32_t)rows; ++j) {
      uint32_t buffIndex = j * this->rowLength;

      // 对当前行取绝对值（就地计算在xLocal上）
      AscendC::Abs(xLocal[buffIndex], xLocal[buffIndex], this->rowLength);

      // 对abs(x)进行ReduceSum，结果存放到rowScalars[j]
      AscendC::ReduceSum<float>(rowScalars[j], xLocal[buffIndex], tmpReduce, this->rowLength);
    }

    // 计算均值：mean = sum / rowLength
    AscendC::Muls(rowScalars, rowScalars, this->factor, rows);
    // 求倒数：inv_mean = 1 / mean
    AscendC::Reciprocal(rowScalars, rowScalars, rows);

    // 用对应行的标量倒数对原始拷贝的zLocal逐行缩放
    for (uint32_t j = 0; j < (uint32_t)rows; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      AscendC::Muls(zLocal[buffIndex], zLocal[buffIndex], rowScalars.GetValue(j), this->rowLength);
    }

    queueZ.EnQue<float>(zLocal);
    queueX.FreeTensor(xLocal);
  }

  __aicore__ inline void CopyOut(int32_t progress, int32_t rows) {
    AscendC::LocalTensor<float> zLocal = queueZ.DeQue<float>();
    AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, rows * this->rowLength);
    queueZ.FreeTensor(zLocal);
  }

 private:
  AscendC::TPipe pipe;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> tmpBufferReduce;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> rowScalarBuf;
  AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> queueX;
  AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> queueZ;
  AscendC::GlobalTensor<float> xGm;
  AscendC::GlobalTensor<float> zGm;

  uint32_t blockLength = 0;
  uint32_t leftRow = 0;
  uint32_t rowNum = 1;
  uint32_t rowNumSp = 2;
  uint32_t rowLength = 1024;
  uint32_t blockPivot = 0;
  uint32_t tileLoop = 1;
  uint32_t tileLength = 1024;
  uint32_t loopCount = 0;
  float factor = 1.0f;
};

extern "C" __global__ __aicore__ void l1_norm_custom(
    GM_ADDR x, GM_ADDR res_out, GM_ADDR workspace, GM_ADDR tiling) {
  KernelL1Norm op;
  op.Init(x, res_out, tiling);
  if (TILING_KEY_IS(1)) {
    op.Process();
  }
}
"""
