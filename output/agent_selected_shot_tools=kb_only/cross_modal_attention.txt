kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

class KernelCrossModalAttention {
public:
    __aicore__ inline KernelCrossModalAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR kv, GM_ADDR out, 
                                uint32_t B, uint32_t H, uint32_t Tq, uint32_t Tk, uint32_t Dh)
    {
        this->B = B;
        this->H = H;
        this->Tq = Tq;
        this->Tk = Tk;
        this->Dh = Dh;

        uint32_t groups = B * H;
        uint32_t elemsQPerGroup = Tq * Dh;
        uint32_t elemsKvPerGroup = Tk * Dh;

        // Each block handles one (b, h) group indexed by blockIdx
        uint32_t gid = AscendC::GetBlockIdx();
        uint32_t qOffset = gid * elemsQPerGroup;
        uint32_t kvOffset = gid * elemsKvPerGroup;

        qGm.SetGlobalBuffer((__gm__ DTYPE *)q + qOffset, elemsQPerGroup);
        kvGm.SetGlobalBuffer((__gm__ DTYPE *)kv + kvOffset, elemsKvPerGroup);
        outGm.SetGlobalBuffer((__gm__ DTYPE *)out + qOffset, elemsQPerGroup);

        // Allocate queues for full-group tiles (simple version)
        // Local buffers:
        // qLocal: [Tq, Dh], kvLocal: [Tk, Dh], scores: [Tq, Tk], probs: [Tq, Tk], outLocal: [Tq, Dh]
        uint32_t qBytes  = elemsQPerGroup * sizeof(DTYPE);
        uint32_t kvBytes = elemsKvPerGroup * sizeof(DTYPE);
        uint32_t outBytes = elemsQPerGroup * sizeof(DTYPE);

        // For simplicity, use queues for q and out; kv handled similarly
        AscendC::GetTPipePtr()->InitBuffer(inQueueQ, 1, qBytes);
        AscendC::GetTPipePtr()->InitBuffer(inQueueKV, 1, kvBytes);
        AscendC::GetTPipePtr()->InitBuffer(outQueueO, 1, outBytes);
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        // Q
        AscendC::LocalTensor<DTYPE> qLocal = inQueueQ.AllocTensor<DTYPE>();
        AscendC::DataCopy(qLocal, qGm, Tq * Dh);
        inQueueQ.EnQue(qLocal);

        // KV (used as both K and V)
        AscendC::LocalTensor<DTYPE> kvLocal = inQueueKV.AllocTensor<DTYPE>();
        AscendC::DataCopy(kvLocal, kvGm, Tk * Dh);
        inQueueKV.EnQue(kvLocal);
    }

    __aicore__ inline void Compute()
    {
        // Dequeue inputs
        AscendC::LocalTensor<DTYPE> qLocal = inQueueQ.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> kvLocal = inQueueKV.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> outLocal = outQueueO.AllocTensor<DTYPE>();

        // Allocate temporaries in local L1/L0 by using queue-style temps
        // scores: [Tq, Tk]
        AscendC::LocalTensor<DTYPE> scoresLocal = AscendC::AllocLocalTensor<DTYPE>(Tq * Tk);
        AscendC::LocalTensor<DTYPE> probsLocal = AscendC::AllocLocalTensor<DTYPE>(Tq * Tk);

        // scores = (q @ k^T) / sqrt(Dh)
        // q: [Tq, Dh], k: [Tk, Dh] -> k^T: [Dh, Tk]
        // MatMul: (Tq x Dh) * (Dh x Tk) = (Tq x Tk)
        // Note: API signatures are conceptual; rely on AscendC high-level matmul
        AscendC::MatMul(scoresLocal, qLocal, kvLocal, Tq, Tk, Dh, false, true);

        // scale
        DTYPE scale = (DTYPE)(1.0f / sqrtf((float)Dh));
        AscendC::Muls(scoresLocal, scoresLocal, scale, Tq * Tk);

        // softmax along last dim (Tk) for each of Tq rows
        AscendC::Softmax(probsLocal, scoresLocal, Tq, Tk);

        // out = probs @ v, where v == kvLocal: [Tk, Dh]
        // (Tq x Tk) * (Tk x Dh) = (Tq x Dh)
        AscendC::MatMul(outLocal, probsLocal, kvLocal, Tq, Dh, Tk, false, false);

        // release inputs
        inQueueQ.FreeTensor(qLocal);
        inQueueKV.FreeTensor(kvLocal);

        // enqueue output
        outQueueO.EnQue(outLocal);

        // free temporaries
        AscendC::FreeLocalTensor(probsLocal);
        AscendC::FreeLocalTensor(scoresLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<DTYPE> outLocal = outQueueO.DeQue<DTYPE>();
        AscendC::DataCopy(outGm, outLocal, Tq * Dh);
        outQueueO.FreeTensor(outLocal);
    }

private:
    // Queues and global tensors
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueQ;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueKV;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueO;

    AscendC::GlobalTensor<DTYPE> qGm;
    AscendC::GlobalTensor<DTYPE> kvGm;
    AscendC::GlobalTensor<DTYPE> outGm;

    // dims per (b,h) group
    uint32_t B, H, Tq, Tk, Dh;
};

extern "C" __global__ __aicore__ void cross_modal_attention_custom(GM_ADDR q, GM_ADDR kv, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    uint32_t B  = tiling_data.batch;
    uint32_t H  = tiling_data.heads;
    uint32_t Tq = tiling_data.lenQ;
    uint32_t Tk = tiling_data.lenK;
    uint32_t Dh = tiling_data.headDim;

    // One TPipe per kernel entry
    AscendC::TPipe pipe;

    KernelCrossModalAttention op;
    op.Init(q, kv, out, B, H, Tq, Tk, Dh);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Cross-modal attention using custom AscendC kernel for scaled dot-product attention.
    Expect kernel input shapes:
      q:  [B, H, Tq, Dh]
      kv: [B, H, Tk, Dh]
    Output:
      out: [B, H, Tq, Dh] -> merged to [B, Tq, D]
    """
    def __init__(self, d_model_text=256, d_model_img=512, num_heads=8):
        super().__init__()
        self.d_model_text = d_model_text
        self.d_model_img = d_model_img
        self.num_heads = num_heads
        self.head_dim = d_model_text // num_heads
        assert self.d_model_text % self.num_heads == 0, "d_model_text must be divisible by num_heads"

        self.q_proj = torch.nn.Linear(d_model_text, d_model_text, bias=True)
        self.kv_proj = torch.nn.Linear(d_model_img, d_model_text, bias=True)

    def forward(self, text, img):
        # text: [B, Tq, Dt], img: [B, Tk, Di]
        B, Tq, _ = text.shape
        Tk = img.shape[1]

        q = self.q_proj(text)    # [B, Tq, Dt]
        kv = self.kv_proj(img)   # [B, Tk, Dt]

        H = self.num_heads
        Dh = self.head_dim

        # [B, Tq, H, Dh] -> [B, H, Tq, Dh]
        q = q.view(B, Tq, H, Dh).permute(0, 2, 1, 3).contiguous()
        # [B, Tk, H, Dh] -> [B, H, Tk, Dh]
        kv = kv.view(B, Tk, H, Dh).permute(0, 2, 1, 3).contiguous()

        out_heads = custom_ops_lib.cross_modal_attention_custom(q, kv)  # [B, H, Tq, Dh]

        # merge heads: [B, Tq, D]
        out = out_heads.permute(0, 2, 1, 3).contiguous().view(B, Tq, H * Dh)
        return out
'''
