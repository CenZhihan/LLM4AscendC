host_operator_src="""
#include "matmul_gelu_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

#include "platform_ascendc.h"
#include "matmul_tiling.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto xShape = context->GetInputShape(0)->GetOriginShape();
    auto wShape = context->GetInputShape(1)->GetOriginShape();
    // x: [M, K], weight: [N, K] (PyTorch Linear stores [N, K]), we will set transposeB=true in kernel
    const uint64_t M = xShape.GetDim(0);
    const uint64_t K = xShape.GetDim(1);
    const uint64_t N = wShape.GetDim(0);

    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);
    tiling.SetDim(1); // let tiling compute the final core num
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetShape(static_cast<uint32_t>(M), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    tiling.SetSingleShape(static_cast<uint32_t>(M), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    tiling.SetOrgShape(static_cast<uint32_t>(M), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    tiling.SetBias(true);
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret == -1) {
        return ge::GRAPH_FAILED;
    }

    // set block dim from core num after tiling
    uint32_t blockDim = tiling.GetCoreNum();
    if (blockDim == 0) {
        blockDim = 1;
    }
    context->SetBlockDim(blockDim);

    // save raw tiling bytes
    auto *raw = context->GetRawTilingData();
    size_t need = sizeof(optiling::TCubeTiling);
    if (raw->GetCapacity() < need) {
        return ge::GRAPH_FAILED;
    }
    std::memcpy(raw->GetData(), &tilingData, need);
    raw->SetDataSize(need);

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* xShape = context->GetInputShape(0);   // [M, K]
    const gert::Shape* wShape = context->GetInputShape(1);   // [N, K]
    gert::Shape* yShape = context->GetOutputShape(0);        // [M, N]
    if (xShape->GetDimNum() != 2 || wShape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    std::vector<int64_t> outDims(2);
    outDims[0] = xShape->GetDim(0);
    outDims[1] = wShape->GetDim(0);
    yShape->SetDimNum(2);
    yShape->SetDim(0, outDims[0]);
    yShape->SetDim(1, outDims[1]);
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = float;
using B_T = float;
using C_T = float;
using BiasT = float;

constexpr int32_t GELU_TILE_ELEMS = 4096;

extern "C" __global__ __aicore__ void matmul_gelu_softmax_custom(GM_ADDR xGM, GM_ADDR wGM, GM_ADDR bGM,
                                                                 GM_ADDR yGM, GM_ADDR workspaceGM, GM_ADDR tilingGM) {
    // 1) 读取Matmul tiling数据
    TCubeTiling tiling;
    auto tGM = (__gm__ uint32_t*)tilingGM;
    auto tLocal = (uint32_t*)&tiling;
    for (int i = 0; i < (int)(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tGM, ++tLocal) {
        *tLocal = *tGM;
    }

    // 2) 准备GM张量
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;
    AscendC::GlobalTensor<BiasT> biasGlobal;

    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(C_T);
    int32_t sizeBias = tiling.CLayoutInfoB * tiling.singleCoreN * sizeof(BiasT);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(xGM), sizeA / sizeof(A_T));
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(wGM), sizeB / sizeof(B_T));
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(yGM), sizeC / sizeof(C_T));
    biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ BiasT*>(bGM), sizeBias / sizeof(BiasT));

    // 3) Matmul实例（NORMAL布局），B按需转置
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType;  // 转置B
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

    tiling.shareMode = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize = 0;

    AscendC::GlobalTensor<A_T> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal[0].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreK);
    AscendC::GlobalTensor<B_T> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal[0].GetPhyAddr()),
                         tiling.singleCoreK * tiling.singleCoreN);
    AscendC::GlobalTensor<C_T> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal[0].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreN);
    AscendC::GlobalTensor<BiasT> gm_bias;
    gm_bias.SetGlobalBuffer(const_cast<__gm__ BiasT*>(biasGlobal[0].GetPhyAddr()),
                            tiling.singleCoreN);

    AscendC::TPipe pipe;
    matmul::Matmul<aType, bType, cType, biasType> mm1;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);
    mm1.Init(&tiling);
    mm1.SetTensorA(gm_a, false);
    mm1.SetTensorB(gm_b, true);
    mm1.SetWorkspace(workspaceGM, 0);
    if (tiling.isBias) {
        mm1.SetBias(gm_bias);
    }
    // 4) 执行GEMM写回到gm_c (单batch)
    mm1.IterateBatch(gm_c, 1, 1, false);

    // 5) 对结果执行GELU：按1D分块处理，避免本地缓冲区过大
    const uint32_t totalElems = tiling.singleCoreM * tiling.singleCoreN;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQ;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQ;
    pipe.InitBuffer(inQ, 1, GELU_TILE_ELEMS * sizeof(C_T));
    pipe.InitBuffer(outQ, 1, GELU_TILE_ELEMS * sizeof(C_T));

    uint32_t offset = 0;
    while (offset < totalElems) {
        uint32_t step = AscendC::min((uint32_t)GELU_TILE_ELEMS, totalElems - offset);
        AscendC::LocalTensor<C_T> srcLocal = inQ.AllocTensor<C_T>();
        AscendC::DataCopy(srcLocal, cGlobal[offset], step);
        inQ.EnQue(srcLocal);

        AscendC::LocalTensor<C_T> dstLocal = outQ.AllocTensor<C_T>();
        AscendC::LocalTensor<C_T> sLocal   = inQ.DeQue<C_T>();
        AscendC::Gelu(dstLocal, sLocal, step);
        outQ.EnQue<C_T>(dstLocal);
        inQ.FreeTensor(sLocal);

        AscendC::LocalTensor<C_T> dLocal = outQ.DeQue<C_T>();
        AscendC::DataCopy(cGlobal[offset], dLocal, step);
        outQ.FreeTensor(dLocal);

        offset += step;
    }

    // Softmax留在上层Python完成（dim=1）
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features, bias=True)

    def forward(self, x):
        # 自定义AscendC内核: 计算 Y = GELU(X @ W^T + b)
        # Softmax(dim=1) 在Python侧完成，避免GPU/CPU与NPU数据反复迁移，同时保持数值稳定性
        y = custom_ops_lib.matmul_gelu_softmax_custom(x, self.linear.weight, self.linear.bias)
        y = torch.nn.functional.softmax(y, dim=1)
        return y
