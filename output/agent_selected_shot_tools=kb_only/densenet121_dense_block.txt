kernel_src="""
#include "kernel_operator.h"

constexpr uint32_t TILE_ELEMS = 4096;

class KernelDenseCat {
public:
    __aicore__ inline KernelDenseCat() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t N, uint32_t C1, uint32_t C2, uint32_t H, uint32_t W)
    {
        this->N = N;
        this->C1 = C1;
        this->C2 = C2;
        this->H = H;
        this->W = W;

        this->aStride = C1 * H * W;
        this->bStride = C2 * H * W;
        this->zStride = (C1 + C2) * H * W;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N * aStride);
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N * bStride);
        zGm.SetGlobalBuffer((__gm__ float*)z, (uint64_t)N * zStride);

        pipe.InitBuffer(tmpBuf, TILE_ELEMS * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockNum == 0) { return; }
        uint32_t batchPerBlock = (N + blockNum - 1) / blockNum;
        uint32_t startBatch = blockIdx * batchPerBlock;
        uint32_t endBatch = startBatch + batchPerBlock;
        if (startBatch >= N) { return; }
        if (endBatch > N) { endBatch = N; }

        for (uint32_t b = startBatch; b < endBatch; ++b) {
            uint64_t aStart = (uint64_t)b * aStride;
            uint64_t bStart = (uint64_t)b * bStride;
            uint64_t zStart = (uint64_t)b * zStride;

            CopyRange(xGm, aStart, zGm, zStart, aStride);
            CopyRange(yGm, bStart, zGm, zStart + aStride, bStride);
        }
    }

private:
    __aicore__ inline void CopyRange(AscendC::GlobalTensor<float>& src, uint64_t srcOffset,
                                     AscendC::GlobalTensor<float>& dst, uint64_t dstOffset,
                                     uint32_t length)
    {
        uint32_t pos = 0;
        while (pos < length) {
            uint32_t cur = (length - pos) < TILE_ELEMS ? (length - pos) : TILE_ELEMS;
            AscendC::LocalTensor<float> tmpLocal = tmpBuf.Get<float>();
            AscendC::DataCopy(tmpLocal, src[srcOffset + pos], cur);
            AscendC::DataCopy(dst[dstOffset + pos], tmpLocal, cur);
            pos += cur;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t N;
    uint32_t C1;
    uint32_t C2;
    uint32_t H;
    uint32_t W;

    uint32_t aStride;
    uint32_t bStride;
    uint32_t zStride;
};

extern "C" __global__ __aicore__ void densenet121_dense_block_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelDenseCat op;
    op.Init(x, y, z, tiling_data.N, tiling_data.C1, tiling_data.C2, tiling_data.H, tiling_data.W);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        super().__init__()
        layers = []
        for i in range(num_layers):
            layers.append(self._make_layer(num_input_features + i * growth_rate, growth_rate))
        self.layers = torch.nn.ModuleList(layers)

    def _make_layer(self, in_features: int, growth_rate: int):
        return torch.nn.Sequential(
            torch.nn.BatchNorm2d(in_features),
            torch.nn.ReLU(inplace=True),
            torch.nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
            torch.nn.Dropout(0.0)
        )

    def forward(self, x):
        for layer in self.layers:
            new_feature = layer(x)
            # Use custom AscendC kernel to concatenate along channel dimension
            x = custom_ops_lib.densenet121_dense_block_custom(x, new_feature)
        return x
'''
