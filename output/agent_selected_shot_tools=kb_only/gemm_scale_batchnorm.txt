static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GemmScaleBatchnormCustomTilingData tiling;

    // 输入0: x，形状为 [B, C]，totalLength = B*C
    uint32_t totalLength = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    // 输入1: scale，形状为 [C]，featureDim = C
    uint32_t featureDim = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());

    // 设置并行块数
    context->SetBlockDim(BLOCK_DIM);

    // 行数 rows = totalLength / featureDim
    uint32_t rows = (featureDim == 0) ? 0 : (totalLength / featureDim);
    // 每个block处理的行数（向下取整）
    uint32_t rowsPerBlock = (BLOCK_DIM == 0) ? 0 : (rows / BLOCK_DIM);
    // 每个block内的tile个数，按双缓冲拆分
    uint32_t tileNum = (BUFFER_NUM == 0) ? 0 : (rowsPerBlock / BUFFER_NUM);

    tiling.set_totalLength(totalLength);
    tiling.set_featureDim(featureDim);
    tiling.set_tileNum(tileNum);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0; // 无需额外workspace
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

class KernelGemmScaleBatchnorm {
public:
    __aicore__ inline KernelGemmScaleBatchnorm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR alpha, GM_ADDR beta2, GM_ADDR z,
                                uint32_t totalLength, uint32_t featureDim, uint32_t /*tileNumIn*/)
    {
        this->featureDim = featureDim;
        // 每个block处理的元素数
        this->blockLength = totalLength / AscendC::GetBlockNum();
        // 处理的起始偏移（按元素）
        uint32_t startOffset = this->blockLength * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + startOffset, this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE*)z + startOffset, this->blockLength);

        // 这三个是按通道参数，长度为 featureDim
        scaleGm.SetGlobalBuffer((__gm__ DTYPE*)scale, featureDim);
        alphaGm.SetGlobalBuffer((__gm__ DTYPE*)alpha, featureDim);
        beta2Gm.SetGlobalBuffer((__gm__ DTYPE*)beta2, featureDim);

        // 每个tile处理一整行（featureDim）
        this->tileLength = featureDim;

        // 队列初始化
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(inQueueS, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // 以行（featureDim）为单位处理
        uint32_t rowsThisBlock = (featureDim == 0) ? 0 : (this->blockLength / this->featureDim);
        uint32_t loopCount = rowsThisBlock * BUFFER_NUM; // 保持与模板一致的循环结构
        for (uint32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress)
    {
        // 当前进度对应的行索引（双缓冲下的线性进度）
        uint32_t rowIdx = progress % ((blockLength / featureDim));
        // 分配本地tensor
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> sLocal = inQueueS.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> aLocal = inQueueA.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = inQueueB.AllocTensor<DTYPE>();

        // 从GM拷贝一整行x
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->tileLength], this->tileLength);
        // 按通道参数，每次直接整矢量拷贝
        AscendC::DataCopy(sLocal, scaleGm, this->tileLength);
        AscendC::DataCopy(aLocal, alphaGm, this->tileLength);
        AscendC::DataCopy(bLocal, beta2Gm, this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueS.EnQue(sLocal);
        inQueueA.EnQue(aLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> sLocal = inQueueS.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> aLocal = inQueueA.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = inQueueB.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.AllocTensor<DTYPE>();

        // y = ((x * scale) * alpha) + beta2
        AscendC::LocalTensor<DTYPE> tmpLocal1 = outQueueZ.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> tmpLocal2 = outQueueZ.AllocTensor<DTYPE>();
        AscendC::Mul(tmpLocal1, xLocal, sLocal, this->tileLength);
        AscendC::Mul(tmpLocal2, tmpLocal1, aLocal, this->tileLength);
        AscendC::Add(zLocal, tmpLocal2, bLocal, this->tileLength);

        // 回填到队列
        outQueueZ.EnQue<DTYPE>(zLocal);

        // 释放输入本地tensor
        inQueueX.FreeTensor(xLocal);
        inQueueS.FreeTensor(sLocal);
        inQueueA.FreeTensor(aLocal);
        inQueueB.FreeTensor(bLocal);
        // 释放临时本地tensor
        outQueueZ.FreeTensor(tmpLocal1);
        outQueueZ.FreeTensor(tmpLocal2);
    }

    __aicore__ inline void CopyOut(uint32_t progress)
    {
        uint32_t rowIdx = progress % ((blockLength / featureDim));
        AscendC::LocalTensor<DTYPE> zLocal = outQueueZ.DeQue<DTYPE>();
        AscendC::DataCopy(zGm[rowIdx * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueS;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> scaleGm;
    AscendC::GlobalTensor<DTYPE> alphaGm;
    AscendC::GlobalTensor<DTYPE> beta2Gm;
    AscendC::GlobalTensor<DTYPE> zGm;

    uint32_t blockLength;
    uint32_t featureDim;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gemm_scale_batchnorm_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR alpha, GM_ADDR beta2,
                                                                  GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScaleBatchnorm op;
    op.Init(x, scale, alpha, beta2, z, tiling_data.totalLength, tiling_data.featureDim, tiling_data.tileNum);
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static at::Tensor compute_alpha(const at::Tensor& gamma, const at::Tensor& running_var, double eps) {
    // alpha = gamma / sqrt(running_var + eps)
    return gamma / (running_var + eps).sqrt();
}

static at::Tensor compute_beta2(const at::Tensor& beta, const at::Tensor& running_mean, const at::Tensor& alpha) {
    // beta2 = beta - running_mean * alpha
    return beta - running_mean * alpha;
}

at::Tensor gemm_scale_batchnorm_custom_impl_npu(const at::Tensor& x,
                                                const at::Tensor& scale,
                                                const at::Tensor& bn_gamma,
                                                const at::Tensor& bn_beta,
                                                const at::Tensor& running_mean,
                                                const at::Tensor& running_var,
                                                double eps) {
    // 计算BN融合参数 alpha 与 beta2
    at::Tensor alpha = compute_alpha(bn_gamma, running_var, eps);
    at::Tensor beta2 = compute_beta2(bn_beta, running_mean, alpha);

    // 输出
    at::Tensor result = at::empty_like(x);
    // 执行自定义算子：y = ((x * scale) * alpha) + beta2
    EXEC_NPU_CMD(aclnnGemmScaleBatchnormCustom, x, scale, alpha, beta2, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("gemm_scale_batchnorm_custom", &gemm_scale_batchnorm_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("gemm_scale_batchnorm_custom", &gemm_scale_batchnorm_custom_impl_npu, "GEMM+Scale+BN fused (post-GEMM) operator");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    以自定义AscendC算子替换线性层后的缩放与BatchNorm计算。
    GEMM仍由nn.Linear执行，随后使用融合算子完成 y = ((x * scale) * alpha) + beta2，
    其中 alpha = gamma / sqrt(running_var + eps), beta2 = beta - mean * alpha。
    """
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.scale = torch.nn.Parameter(torch.rand(scale_shape))
        self.bn = torch.nn.BatchNorm1d(out_features, eps=eps, momentum=momentum)

    def forward(self, x):
        x = self.gemm(x)  # [B, C]
        # 使用自定义AscendC融合算子完成 scale 与 BN（基于running stats）的推理计算
        x = custom_ops_lib.gemm_scale_batchnorm_custom(
            x,
            self.scale,
            self.bn.weight,
            self.bn.bias,
            self.bn.running_mean,
            self.bn.running_var,
            float(self.bn.eps),
        )
        return x
