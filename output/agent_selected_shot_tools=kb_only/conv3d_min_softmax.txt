#include "kernel_operator.h"

// 说明：本kernel为示例骨架，演示如何组织tile、加载/保存数据与预留SoftMax相关tiling。
// 实际算法应在Compute中完成：
// 1) 沿dim=2(深度D)做reduce-min，得到(N, C, H, W)
// 2) 沿通道C(维度1)做softmax。可结合tiling数据调用SoftMax高阶API。
// 这里简化为将输入分片搬运后原样拷贝（留出注释位置进行功能扩展）。

constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor个数

using DTYPE_IN  = float;
using DTYPE_OUT = float;

class KernelConv3dMinSoftmax {
public:
    __aicore__ inline KernelConv3dMinSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t tmpBufSize,
                                uint32_t mDim, uint32_t kDim,
                                const optiling::SoftMaxTiling &softmaxTilingDataIn)
    {
        // 简单按block划分
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        // 简化分片
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_IN *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_OUT *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_IN));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_OUT));

        this->tmpBufSize = tmpBufSize;
        this->mDim = mDim;
        this->kDim = kDim;
        this->softmaxTilingData = softmaxTilingDataIn;
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_IN> xLocal = inQueueX.AllocTensor<DTYPE_IN>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        // 取出输入局部tile
        AscendC::LocalTensor<DTYPE_IN> xLocal = inQueueX.DeQue<DTYPE_IN>();
        AscendC::LocalTensor<DTYPE_OUT> yLocal = outQueueY.AllocTensor<DTYPE_OUT>();

        // TODO: 在此实现
        // Step1: reduce-min 沿D维（dim=2） -> 生成(N,C,H,W)的子块
        // Step2: softmax 沿C维（dim=1），可结合softmaxTilingData与tmp buffer调用SoftMax高阶API
        // 这里先做直拷（示例），请在实际工程中替换为真正计算逻辑
        AscendC::DataCopy(yLocal, xLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_OUT>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_OUT> yLocal = outQueueY.DeQue<DTYPE_OUT>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_IN>  xGm;
    AscendC::GlobalTensor<DTYPE_OUT> yGm;

    uint32_t blockLength{0};
    uint32_t tileNum{1};
    uint32_t tileLength{0};

    uint32_t tmpBufSize{0};
    uint32_t mDim{0};
    uint32_t kDim{0};
    optiling::SoftMaxTiling softmaxTilingData;
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv3dMinSoftmax op;
    op.Init(x, y,
            tilingData.totalLength,
            tilingData.tileNum,
            tilingData.tmpBufSize,
            tilingData.mDim,
            tilingData.kDim,
            tilingData.softmaxTilingData);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC算子替换 PyTorch 中的 min(dim=2) + softmax(dim=1) 组合。
    Conv3d 仍由 nn.Conv3d 执行，随后调用 conv3d_min_softmax_custom 完成 reduce-min 与softmax。
    """
    def __init__(self, in_channels, out_channels, kernel_size, dim):
        super(ModelNew, self).__init__()
        assert dim == 2, "本示例kernel固定在dim=2上执行min归约"
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.dim = dim

    def forward(self, x):
        # x: (N, C_in, D, H, W)
        x = self.conv(x)  # (N, C_out, D', H', W')
        # 自定义算子：内部完成 min(dim=2) -> (N, C_out, H', W')，然后 softmax(dim=1)
        y = custom_ops_lib.conv3d_min_softmax_custom(x)
