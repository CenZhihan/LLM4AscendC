    // Standard conv output formula: floor((in + 2*pad - dilation*(k-1) - 1) / stride + 1)

kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1; // use scalar/very small tiles per iteration

class KernelConv3D {
public:
    __aicore__ inline KernelConv3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c_in, uint32_t d, uint32_t h, uint32_t wid,
                                uint32_t c_out, uint32_t k,
                                uint32_t stride, uint32_t pad, uint32_t dilation,
                                uint32_t d_out, uint32_t h_out, uint32_t w_out, uint32_t tileNum) {
        this->n = n; this->c_in = c_in; this->d = d; this->h = h; this->wid = wid;
        this->c_out = c_out; this->k = k;
        this->stride = stride; this->pad = pad; this->dilation = dilation;
        this->d_out = d_out; this->h_out = h_out; this->w_out = w_out;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, (uint64_t)n * c_in * d * h * wid);
        wGm.SetGlobalBuffer((__gm__ DTYPE*)w, (uint64_t)c_out * c_in * k * k * k);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, (uint64_t)n * c_out * d_out * h_out * w_out);

        // Scalar-sized queues
        pipe.InitBuffer(inQueueX, BUFFER_NUM, sizeof(DTYPE));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, sizeof(DTYPE));
    }

    __aicore__ inline void Process() {
        uint64_t totalY = (uint64_t)n * c_out * d_out * h_out * w_out;
        uint64_t blockNum = AscendC::GetBlockNum();
        uint64_t blockIdx = AscendC::GetBlockIdx();
        // Simple 1D partition across output elements
        uint64_t perBlock = (totalY + blockNum - 1) / blockNum;
        uint64_t start = blockIdx * perBlock;
        uint64_t end = start + perBlock;
        if (end > totalY) end = totalY;

        for (uint64_t outIndex = start; outIndex < end; ++outIndex) {
            computeOne(outIndex);
        }
    }

private:
    __aicore__ inline void computeOne(uint64_t outIndex) {
        // Decode outIndex -> (n, co, od, oh, ow)
        uint64_t ow = outIndex % w_out;
        uint64_t tmp1 = outIndex / w_out;
        uint64_t oh = tmp1 % h_out;
        uint64_t tmp2 = tmp1 / h_out;
        uint64_t od = tmp2 % d_out;
        uint64_t tmp3 = tmp2 / d_out;
        uint64_t co = tmp3 % c_out;
        uint64_t nn = tmp3 / c_out;

        // Local scalar accumulator
        AscendC::LocalTensor<DTYPE> accLocal = outQueueY.AllocTensor<DTYPE>();
        AscendC::Duplicate<DTYPE>(accLocal, static_cast<DTYPE>(0), 1);

        for (uint32_t ci = 0; ci < c_in; ++ci) {
            for (uint32_t kz = 0; kz < k; ++kz) {
                int64_t in_d = static_cast<int64_t>(od) * stride - pad + static_cast<int64_t>(kz) * dilation;
                if (in_d < 0 || in_d >= d) continue;
                for (uint32_t ky = 0; ky < k; ++ky) {
                    int64_t in_h = static_cast<int64_t>(oh) * stride - pad + static_cast<int64_t>(ky) * dilation;
                    if (in_h < 0 || in_h >= h) continue;
                    for (uint32_t kx = 0; kx < k; ++kx) {
                        int64_t in_w = static_cast<int64_t>(ow) * stride - pad + static_cast<int64_t>(kx) * dilation;
                        if (in_w < 0 || in_w >= wid) continue;

                        // Load x scalar
                        uint64_t xOffset = (((nn * c_in + ci) * d + (uint32_t)in_d) * h + (uint32_t)in_h) * wid + (uint32_t)in_w;
                        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
                        AscendC::DataCopy(xLocal, xGm[xOffset], 1);
                        inQueueX.EnQue(xLocal);
                        AscendC::LocalTensor<DTYPE> xVal = inQueueX.DeQue<DTYPE>();

                        // Load w scalar
                        uint64_t wOffset = ((((co * c_in) + ci) * k + kz) * k + ky) * k + kx;
                        AscendC::LocalTensor<DTYPE> wLocal = inQueueW.AllocTensor<DTYPE>();
                        AscendC::DataCopy(wLocal, wGm[wOffset], 1);
                        inQueueW.EnQue(wLocal);
                        AscendC::LocalTensor<DTYPE> wVal = inQueueW.DeQue<DTYPE>();

                        // temp = x * w
                        AscendC::LocalTensor<DTYPE> tmpLocal = inQueueX.AllocTensor<DTYPE>();
                        AscendC::Mul(tmpLocal, xVal, wVal, 1);
                        // acc += temp
                        AscendC::Add(accLocal, accLocal, tmpLocal, 1);

                        inQueueX.FreeTensor(tmpLocal);
                        inQueueW.FreeTensor(wVal);
                        inQueueX.FreeTensor(xVal);
                    }
                }
            }
        }

        // Store acc to y
        uint64_t yOffset = (((nn * c_out + co) * d_out + (uint32_t)od) * h_out + (uint32_t)oh) * w_out + (uint32_t)ow;
        AscendC::DataCopy(yGm[yOffset], accLocal, 1);
        outQueueY.FreeTensor(accLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX, inQueueW;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t n, c_in, d, h, wid;
    uint32_t c_out, k;
    uint32_t stride, pad, dilation;
    uint32_t d_out, h_out, w_out;
    uint32_t tileNum;
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(t, tiling);
    KernelConv3D op;
    op.Init(x, w, y,
            t.n(), t.c_in(), t.d(), t.h(), t.w(),
            t.c_out(), t.k(),
            t.stride(), t.pad(), t.dilation(),
            t.d_out(), t.h_out(), t.w_out(), t.tileNum());
    op.Process();
}
