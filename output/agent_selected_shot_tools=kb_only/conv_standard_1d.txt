#include "conv_standard_1d_custom_tiling.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // 仅支持 stride=1, padding=0, dilation=1, groups=1, 无bias 的标准1D卷积
    ConvStandard1dCustomTilingData tiling;

    // 输入 x: [B, Cin, Lin]
    const gert::Shape* x_shape = context->GetInputShape(0);
    // 权重 w: [Cout, Cin, K]
    const gert::Shape* w_shape = context->GetInputShape(1);

    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    uint32_t B = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t Lin = static_cast<uint32_t>(x_shape->GetDim(2));

    uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    uint32_t Cin_w = static_cast<uint32_t>(w_shape->GetDim(1));
    uint32_t K = static_cast<uint32_t>(w_shape->GetDim(2));

    if (Cin != Cin_w) {
        return ge::GRAPH_FAILED;
    }

    // Lout = Lin - K + 1  (stride=1, pad=0, dilation=1)
    if (Lin < K) {
        return ge::GRAPH_FAILED;
    }
    uint32_t Lout = Lin - K + 1;

    // 设置并行度（按 (B*Cout) 维度切分），默认8
    const uint32_t BLOCK_DIM = 8;
    context->SetBlockDim(BLOCK_DIM);

    // 简单设定tileLen，按长度维度分块（可根据UB空间调优）
    // 这里取 2048 的保守值
    uint32_t tileLen = 2048;
    if (tileLen > Lout) tileLen = Lout;

    uint64_t totalPairs = static_cast<uint64_t>(B) * static_cast<uint64_t>(Cout);
    uint32_t blockDim = BLOCK_DIM;
    uint32_t pairsPerCore = static_cast<uint32_t>((totalPairs + blockDim - 1) / blockDim);

    tiling.set_B(B);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_K(K);
    tiling.set_Lin(Lin);
    tiling.set_Lout(Lout);
    tiling.set_tileLen(tileLen);
    tiling.set_pairsPerCore(pairsPerCore);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // workspace 需求为0
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // x: [B, Cin, Lin], w: [Cout, Cin, K] -> y: [B, Cout, Lout]
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;
    }
    int64_t B = x_shape->GetDim(0);
    int64_t Cin = x_shape->GetDim(1);
    int64_t Lin = x_shape->GetDim(2);
    int64_t Cout = w_shape->GetDim(0);
    int64_t Cin_w = w_shape->GetDim(1);
    int64_t K = w_shape->GetDim(2);
    if (Cin != Cin_w || Lin < K) {
        return GRAPH_FAILED;
    }
    int64_t Lout = Lin - K + 1;

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, B);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Lout);
    return GRAPH_SUCCESS;
}

    // 输出dtype跟随输入x

    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelConv1D {
public:
    __aicore__ inline KernelConv1D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t B, uint32_t Cin, uint32_t Cout,
                                uint32_t K, uint32_t Lin, uint32_t Lout,
                                uint32_t tileLen, uint32_t pairsPerCore)
    {
        this->B = B;
        this->Cin = Cin;
        this->Cout = Cout;
        this->K = K;
        this->Lin = Lin;
        this->Lout = Lout;
        this->tileLen = tileLen;
        this->pairsPerCore = pairsPerCore;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(B * Cin * Lin));
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint32_t)(Cout * Cin * K));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint32_t)(B * Cout * Lout));

        // 为长度维度的tile分配本地缓存
        pipe.InitBuffer(xTileQueue, BUFFER_NUM, tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(yTileQueue, BUFFER_NUM, tileLen * sizeof(DTYPE_Y));
        // 临时缓冲（用于中间累加）
        pipe.InitBuffer(tmpAccQueue, BUFFER_NUM, tileLen * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // 将 (B * Cout) 平铺为一维对，按Core划分
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint64_t totalPairs = (uint64_t)B * (uint64_t)Cout;
        uint32_t startPair = blockIdx * pairsPerCore;
        uint32_t endPair = startPair + pairsPerCore;
        if (endPair > totalPairs) endPair = (uint32_t)totalPairs;

        for (uint32_t pairIdx = startPair; pairIdx < endPair; ++pairIdx) {
            uint32_t b = pairIdx / Cout;
            uint32_t co = pairIdx % Cout;

            // 沿输出长度维度分块
            for (uint32_t loBase = 0; loBase < Lout; loBase += tileLen) {
                uint32_t curTileLen = (loBase + tileLen <= Lout) ? tileLen : (Lout - loBase);

                // 分配accumulator并清零
                AscendC::LocalTensor<DTYPE_Y> accLocal = tmpAccQueue.AllocTensor<DTYPE_Y>();
                // 置0
                AscendC::Duplicate(accLocal, (DTYPE_Y)0, curTileLen);

                // 遍历 Cin 和 K，进行累加
                for (uint32_t ci = 0; ci < Cin; ++ci) {
                    // 权重基址 (co, ci, 0)
                    uint64_t wBase = ((uint64_t)co * Cin + ci) * K;

                    for (uint32_t k = 0; k < K; ++k) {
                        // 取输入窗口 x[b, ci, loBase + k : loBase + k + curTileLen]
                        uint64_t xOffset = ((uint64_t)b * Cin + ci) * Lin + (loBase + k);
                        AscendC::LocalTensor<DTYPE_X> xLocal = xTileQueue.AllocTensor<DTYPE_X>();
                        AscendC::DataCopy(xLocal, xGm[xOffset], curTileLen);

                        // 将权重标量扩展为向量，然后进行 y += x * w
                        // 读权重标量 w[co, ci, k]
                        DTYPE_W wScalar;
                        {
                            // 简单方式：从GM地址计算并读取（用于演示）
                            __gm__ DTYPE_W* wPtr = (__gm__ DTYPE_W*)wGm.GetPhyAddr();
                            wScalar = wPtr[wBase + k];
                        }
                        // accLocal = accLocal + xLocal * wScalar
                        AscendC::Axpy(accLocal, xLocal, wScalar, curTileLen);

                        xTileQueue.FreeTensor(xLocal);
                    }
                }

                // 写回 y[b, co, loBase : loBase + curTileLen]
                uint64_t yOffset = ((uint64_t)b * Cout + co) * Lout + loBase;
                AscendC::LocalTensor<DTYPE_Y> yLocal = yTileQueue.AllocTensor<DTYPE_Y>();
                AscendC::DataCopy(yLocal, accLocal, curTileLen);
                AscendC::DataCopy(yGm[yOffset], yLocal, curTileLen);

                yTileQueue.FreeTensor(yLocal);
                tmpAccQueue.FreeTensor(accLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xTileQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yTileQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpAccQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t B;
    uint32_t Cin;
    uint32_t Cout;
    uint32_t K;
    uint32_t Lin;
    uint32_t Lout;
    uint32_t tileLen;
    uint32_t pairsPerCore;
};

extern "C" __global__ __aicore__ void conv_standard_1d_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelConv1D op;
    op.Init(x, w, y,
            tilingData.B, tilingData.Cin, tilingData.Cout,
            tilingData.K, tilingData.Lin, tilingData.Lout,
            tilingData.tileLen, tilingData.pairsPerCore);
    op.Process();
}

at::Tensor conv_standard_1d_custom_impl_npu(const at::Tensor& x, const at::Tensor& w) {
    // 输出形状: [B, Cout, Lout]，由后端InferShape完成
    // 这里直接创建与InferShape一致的输出张量
    // 实际运行时可使用 empty_like 修改为输出shape推断后的张量

    auto Cin = x.size(1);
    auto Lin = x.size(2);
    auto Cout = w.size(0);
    auto K = w.size(2);
    auto Lout = Lin - K + 1;
    at::Tensor y = at::empty({B, Cout, Lout}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard1dCustom, x, w, y);
    return y;
}

    m.impl("conv_standard_1d_custom", &conv_standard_1d_custom_impl_npu);

    m.def("conv_standard_1d_custom", &conv_standard_1d_custom_impl_npu, "Standard 1D Convolution (custom AscendC)");

    """
    使用自定义AscendC核替换标准1D卷积（仅支持 stride=1, padding=0, dilation=1, groups=1, 无bias）。
    """

        torch.nn.init.kaiming_uniform_(self.weight, a=5**0.5)
