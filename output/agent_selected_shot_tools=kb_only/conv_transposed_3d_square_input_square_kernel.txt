    uint32_t D   = static_cast<uint32_t>(GetDim(x_shape, 2));
    uint32_t H   = static_cast<uint32_t>(GetDim(x_shape, 3));

    uint32_t KCin  = static_cast<uint32_t>(GetDim(w_shape, 0));
    uint32_t KCout = static_cast<uint32_t>(GetDim(w_shape, 1));
    uint32_t Kd    = static_cast<uint32_t>(GetDim(w_shape, 2));
    uint32_t Kh    = static_cast<uint32_t>(GetDim(w_shape, 3));
    uint32_t Kw    = static_cast<uint32_t>(GetDim(w_shape, 4));

    // assume square kernel Kd==Kh==Kw
    uint32_t K = Kd;

    // Basic constraints assumed: stride=1, padding=0, output_padding=0, groups=1, bias=False
    // Output shape: [N, Cout, D+K-1, H+K-1, W+K-1]

    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_d(D);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_k(K);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0); // [N, Cin, D, H, W]
    const gert::Shape* w_shape = context->GetInputShape(1); // [Cin, Cout, K, K, K]
    gert::Shape* y_shape = context->GetOutputShape(0);
    if (x_shape == nullptr || w_shape == nullptr || y_shape == nullptr) {
        return GRAPH_FAILED;
    }

    int64_t N   = x_shape->GetOriginShape().GetDim(0);
    int64_t Cin = x_shape->GetOriginShape().GetDim(1);
    int64_t D   = x_shape->GetOriginShape().GetDim(2);
    int64_t H   = x_shape->GetOriginShape().GetDim(3);
    int64_t W   = x_shape->GetOriginShape().GetDim(4);

    int64_t KCin  = w_shape->GetOriginShape().GetDim(0);
    int64_t Cout  = w_shape->GetOriginShape().GetDim(1);
    int64_t Kd    = w_shape->GetOriginShape().GetDim(2);
    int64_t Kh    = w_shape->GetOriginShape().GetDim(3);
    int64_t Kw    = w_shape->GetOriginShape().GetDim(4);

    if (!(KCin == Cin && Kd == Kh && Kh == Kw)) {
        return GRAPH_FAILED;
    }

    int64_t K = Kd;
    int64_t Do = D + K - 1;
    int64_t Ho = H + K - 1;
    int64_t Wo = W + K - 1;

    gert::Shape out;
    out.SetDimNum(5);
    out.SetDim(0, N);
    out.SetDim(1, Cout);
    out.SetDim(2, Do);
    out.SetDim(3, Ho);
    out.SetDim(4, Wo);
    *y_shape = out;
    return GRAPH_SUCCESS;
}

    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

// 简化假设：stride=1, padding=0, output_padding=0, groups=1, 无bias
// 数据布局：NCDHW，权重布局：[Cin, Cout, K, K, K]

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);

    const uint32_t N   = t.n;
    const uint32_t Cin = t.cin;
    const uint32_t Cout= t.cout;
    const uint32_t D   = t.d;
    const uint32_t H   = t.h;
    const uint32_t W   = t.w;
    const uint32_t K   = t.k;

    const uint32_t Do = D + K - 1;
    const uint32_t Ho = H + K - 1;
    const uint32_t Wo = W + K - 1;

    __gm__ float* xPtr = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* wPtr = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* yPtr = reinterpret_cast<__gm__ float*>(y);

    // 为简单清晰，按块划分 batch 维度
    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t nPerBlock = (N + blockNum - 1) / blockNum;
    const uint32_t nStart = blockIdx * nPerBlock;
    const uint32_t nEnd = (nStart + nPerBlock) > N ? N : (nStart + nPerBlock);

    auto x_index = [=](uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w)->uint64_t {
        return (((static_cast<uint64_t>(n) * Cin + c) * D + d) * H + h) * W + w;
    };
    auto w_index = [=](uint32_t ci, uint32_t co, uint32_t kd, uint32_t kh, uint32_t kw)->uint64_t {
        return (((static_cast<uint64_t>(ci) * Cout + co) * K + kd) * K + kh) * K + kw;
    };
    auto y_index = [=](uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w)->uint64_t {
        return (((static_cast<uint64_t>(n) * Cout + c) * Do + d) * Ho + h) * Wo + w;
    };

    // 初始化输出为0
    for (uint32_t n = nStart; n < nEnd; ++n) {
        for (uint32_t co = 0; co < Cout; ++co) {
            for (uint32_t od = 0; od < Do; ++od) {
                for (uint32_t oh = 0; oh < Ho; ++oh) {
                    for (uint32_t ow = 0; ow < Wo; ++ow) {
                        yPtr[y_index(n, co, od, oh, ow)] = 0.0f;
                    }
                }
            }
        }
    }

    // 直接计算反卷积（stride=1, padding=0）的朴素实现
    for (uint32_t n = nStart; n < nEnd; ++n) {
        for (uint32_t co = 0; co < Cout; ++co) {
            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t kd = 0; kd < K; ++kd) {
                    for (uint32_t kh = 0; kh < K; ++kh) {
                        for (uint32_t kw = 0; kw < K; ++kw) {
                            float wval = wPtr[w_index(ci, co, kd, kh, kw)];
                            // 输出坐标 od = id + kd, oh = ih + kh, ow = iw + kw
                            for (uint32_t id = 0; id < D; ++id) {
                                uint32_t od = id + kd;
                                for (uint32_t ih = 0; ih < H; ++ih) {
                                    uint32_t oh = ih + kh;
                                    // 线性遍历 iw 提升局部性
                                    for (uint32_t iw_ = 0; iw_ < W; ++iw_) {
                                        uint32_t ow_ = iw_ + kw;
                                        float xval = xPtr[x_index(n, ci, id, ih, iw_)];
                                        yPtr[y_index(n, co, od, oh, ow_)] += xval * wval;
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
    }
}

    // x: [N, Cin, D, H, W], w: [Cin, Cout, K, K, K]
    auto N   = x.size(0);
    auto Cin = x.size(1);
    auto D   = x.size(2);
    auto H   = x.size(3);
    auto W   = x.size(4);

    TORCH_CHECK(w.dim() == 5, "weight must be 5D");
    TORCH_CHECK(w.size(0) == Cin, "weight Cin mismatch");
    TORCH_CHECK(w.size(2) == w.size(3) && w.size(3) == w.size(4), "kernel must be cubic");
    auto Cout = w.size(1);
    auto K    = w.size(2);

    auto Do = D + K - 1;
    auto Ho = H + K - 1;
    auto Wo = W + K - 1;

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    AscendC 自定义核实现：stride=1, padding=0, output_padding=0, groups=1, 无bias 的 3D 反卷积（转置卷积）。
    权重张量形状: [Cin, Cout, K, K, K]
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0, groups: int = 1, bias: bool = False):
        super().__init__()
        # 仍然复用 PyTorch 的参数管理与权重初始化
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.output_padding = output_padding
        self.groups = groups
        self.bias_flag = bias

        # 仅支持 stride=1 padding=0 output_padding=0 groups=1 无bias
        assert stride == 1 and padding == 0 and output_padding == 0 and groups == 1 and (bias is False or bias is None), \
            "ModelNew 仅支持 stride=1, padding=0, output_padding=0, groups=1, bias=False 的情况"

        # 创建与 nn.ConvTranspose3d 对应的权重，形状 [Cin, Cout, K, K, K]
        # 注意：ConvTranspose3d 的权重布局是 [in_channels, out_channels // groups, kD, kH, kW]
        k = kernel_size
        self.weight = torch.nn.Parameter(torch.empty(in_channels, out_channels, k, k, k))
        torch.nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # 调用自定义 AscendC 算子
        return custom_ops_lib.conv_transposed_3d_square_input_square_kernel_custom(x, self.weight)
