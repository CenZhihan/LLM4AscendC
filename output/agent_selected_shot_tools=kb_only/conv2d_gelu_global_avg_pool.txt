kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // buffers per queue

class KernelConv2dGeluGlobalAvgPoolCustom {
public:
    __aicore__ inline KernelConv2dGeluGlobalAvgPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outter, uint32_t inner, uint32_t /*tileNum*/)
    {
        this->outter = outter;   // groups = B*C
        this->inner = inner;     // H*W
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Split groups across blocks
        uint32_t base = outter / blockNum;
        uint32_t rem  = outter % blockNum;
        this->blockOutter = base + (blockIdx < rem ? 1 : 0);
        this->blockStart = base * blockIdx + (blockIdx < rem ? blockIdx : rem);

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x) + static_cast<uint64_t>(this->blockStart) * this->inner,
                            static_cast<uint64_t>(this->blockOutter) * this->inner);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(y) + this->blockStart, this->blockOutter);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->inner * sizeof(float));
        pipe.InitBuffer(midQueue, BUFFER_NUM, this->inner * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t g = 0; g < this->blockOutter; ++g) {
            CopyIn(g);
            Compute(g);
            CopyOut(g);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t groupIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[groupIdx * this->inner], this->inner);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*groupIdx*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> geluLocal = midQueue.AllocTensor<float>();

        // GELU over H*W
        AscendC::Gelu(geluLocal, xLocal, this->inner);

        // Mean over H*W -> single value
        AscendC::LocalTensor<float> meanLocal = outQueueY.AllocTensor<float>();
        AscendC::MeanParams mp;
        mp.outter = 1;
        mp.inner = this->inner; // inner*sizeof(T) must be 32-byte aligned; ensured by tiling
        mp.n = this->inner;
        AscendC::Mean<float>(meanLocal, geluLocal, mp);

        outQueueY.EnQue<float>(meanLocal);
        inQueueX.FreeTensor(xLocal);
        midQueue.FreeTensor(geluLocal);
    }

    __aicore__ inline void CopyOut(uint32_t groupIdx)
    {
        AscendC::LocalTensor<float> meanLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[groupIdx], meanLocal, 1);
        outQueueY.FreeTensor(meanLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> midQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t outter;       // groups (B*C)
    uint32_t inner;        // H*W
    uint32_t blockOutter;  // groups handled by this core
    uint32_t blockStart;   // starting group index
};

extern "C" __global__ __aicore__ void conv2d_gelu_global_avg_pool_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dGeluGlobalAvgPoolCustom op;
    op.Init(x, y, tiling_data.outter, tiling_data.inner, tiling_data.tileNum);
    op.Process();
}

class ModelNew(torch.nn.Module):

        x = self.conv(x)  # [B, C, H_out, W_out]
