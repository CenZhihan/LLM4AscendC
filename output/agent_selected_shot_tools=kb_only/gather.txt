#include "gather_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GatherCustomTilingData tiling;

    const auto* xShape = context->GetInputShape(0);
    const auto* idxShape = context->GetInputShape(1);
    uint32_t rows = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t srcCols = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    uint32_t idxCols = static_cast<uint32_t>(idxShape->GetOriginShape().GetDim(1));

    // 设置块数
    context->SetBlockDim(BLOCK_DIM);

    // 设置合理的tile大小，适配UB；这里固定为1024，若idxCols更小则取idxCols
    uint32_t tileSize = 1024;
    if (idxCols < tileSize) tileSize = idxCols;

    tiling.set_rows(rows);
    tiling.set_srcCols(srcCols);
    tiling.set_idxCols(idxCols);
    tiling.set_tileSize(tileSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // 输出形状与idx相同（按dim=1 gather）
    const gert::Shape* idx_shape = context->GetInputShape(1);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *idx_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // 输出dtype与x相同
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Y = float;
using DTYPE_INDEX = uint32_t;

constexpr int32_t BUFFER_NUM_TILE = 2;

class KernelGather {
public:
    __aicore__ inline KernelGather() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR y,
                                uint32_t rows, uint32_t srcCols, uint32_t idxCols, uint32_t tileSize)
    {
        this->rows = rows;
        this->srcCols = srcCols;
        this->idxCols = idxCols;
        this->tileSize = tileSize;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // 简单均分，假定rows能整除blockNum（与Tiling BLOCK_DIM匹配）
        this->rowsPerBlock = rows / blockNum;
        this->rowStart = this->rowsPerBlock * blockIdx;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)rowStart * srcCols, (uint64_t)rowsPerBlock * srcCols);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_INDEX*)idx + (uint64_t)rowStart * idxCols, (uint64_t)rowsPerBlock * idxCols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + (uint64_t)rowStart * idxCols, (uint64_t)rowsPerBlock * idxCols);

        // 队列与缓冲区
        // 每行需要整行x作为源
        pipe.InitBuffer(inQueueX, 1, (uint64_t)srcCols * sizeof(DTYPE_X));
        // 按tile处理idx与输出
        pipe.InitBuffer(inQueueIdx, BUFFER_NUM_TILE, (uint64_t)tileSize * sizeof(DTYPE_INDEX));
        pipe.InitBuffer(inQueueOffset, BUFFER_NUM_TILE, (uint64_t)tileSize * sizeof(uint32_t));
        pipe.InitBuffer(outQueueY, BUFFER_NUM_TILE, (uint64_t)tileSize * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsPerBlock; ++r) {
            // 拷入当前行的x
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[(uint64_t)r * srcCols], srcCols);
            inQueueX.EnQue(xLocal);
            xLocal = inQueueX.DeQue<DTYPE_X>(); // 取出用于后续多次使用

            uint32_t numTiles = (idxCols + tileSize - 1) / tileSize;
            for (uint32_t t = 0; t < numTiles; ++t) {
                uint32_t baseCol = t * tileSize;
                uint32_t curTile = (baseCol + tileSize <= idxCols) ? tileSize : (idxCols - baseCol);

                // 1) 读入本tile的idx
                AscendC::LocalTensor<DTYPE_INDEX> idxLocal = inQueueIdx.AllocTensor<DTYPE_INDEX>();
                AscendC::DataCopy(idxLocal, idxGm[(uint64_t)r * idxCols + baseCol], curTile);
                inQueueIdx.EnQue(idxLocal);

                // 2) 计算offset（字节偏移） = idx * sizeof(DTYPE_X)
                idxLocal = inQueueIdx.DeQue<DTYPE_INDEX>();
                AscendC::LocalTensor<uint32_t> offsetLocal = inQueueOffset.AllocTensor<uint32_t>();
                // 将uint32索引放大为字节偏移，Gather需要字节偏移
                AscendC::Muls(offsetLocal, idxLocal, (uint32_t)sizeof(DTYPE_X), curTile);

                // 3) Gather
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
                AscendC::Gather(yLocal, xLocal, offsetLocal, (uint32_t)0, curTile);

                // 4) 写回
                outQueueY.EnQue<DTYPE_Y>(yLocal);
                AscendC::LocalTensor<DTYPE_Y> yLocalOut = outQueueY.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[(uint64_t)r * idxCols + baseCol], yLocalOut, curTile);

                // 5) 释放本tile的缓冲
                outQueueY.FreeTensor(yLocalOut);
                inQueueIdx.FreeTensor(idxLocal);
                inQueueOffset.FreeTensor(offsetLocal);
            }

            // 释放当前行的x
            inQueueX.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM_TILE> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM_TILE> inQueueOffset;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM_TILE> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_INDEX> idxGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows;
    uint32_t srcCols;
    uint32_t idxCols;
    uint32_t tileSize;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
};

extern "C" __global__ __aicore__ void gather_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGather op;
    op.Init(x, idx, y,
            tiling_data.rows,
            tiling_data.srcCols,
            tiling_data.idxCols,
            tiling_data.tileSize);
    op.Process();
}
