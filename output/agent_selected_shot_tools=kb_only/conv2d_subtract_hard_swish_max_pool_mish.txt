kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelHSwitchSub {
public:
    __aicore__ inline KernelHSwitchSub() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR sub, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ float*)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        subGm.SetGlobalBuffer((__gm__ float*)sub + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ float*)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueSub, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<float> subLocal = inQueueSub.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(subLocal, subGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueSub.EnQue(subLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> subLocal = inQueueSub.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();

        // tmpLocal = x - sub
        AscendC::Sub(tmpLocal, xLocal, subLocal, this->tileLength);

        // HardSwish: hswish(z) = z * clamp(z + 3, 0, 6) / 6, where z = x - sub
        AscendC::Adds(yLocal, tmpLocal, 3.0f, this->tileLength);            // yLocal = z + 3
        AscendC::Maxs(yLocal, yLocal, 0.0f, this->tileLength);              // yLocal = max(z+3, 0)
        AscendC::Mins(yLocal, yLocal, 6.0f, this->tileLength);              // yLocal = min(yLocal, 6)
        AscendC::Mul(yLocal, yLocal, tmpLocal, this->tileLength);           // yLocal = yLocal * z
        AscendC::Muls(yLocal, yLocal, 1.0f/6.0f, this->tileLength);         // yLocal = yLocal / 6

        outQueueY.EnQue<float>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueSub.FreeTensor(subLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueSub;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> subGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(GM_ADDR x, GM_ADDR sub, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelHSwitchSub op;
    op.Init(x, sub, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static at::Tensor mish_activate(const at::Tensor& x) {
    // mish(x) = x * tanh(softplus(x)) = x * tanh(log(1 + exp(x)))
    auto softplus = at::log1p(at::exp(x));
    return x * at::tanh(softplus);
}

at::Tensor conv2d_subtract_hard_swish_max_pool_mish_custom_impl_npu(const at::Tensor& x, double subtract_value, int64_t pool_kernel_size) {
    // 1) Prepare element-wise subtract tensor (same shape as x)
    at::Tensor sub_full = at::full_like(x, static_cast<float>(subtract_value));
    // 2) Run custom AscendC kernel: y0 = HardSwish(x - subtract_value)
    at::Tensor y0 = at::empty_like(x);
    EXEC_NPU_CMD(aclnnConv2dSubtractHardSwishMaxPoolMishCustom, x, sub_full, y0);
    // 3) MaxPool2d with kernel_size and stride equal to pool_kernel_size
    at::Tensor y1 = at::max_pool2d(y0, {pool_kernel_size, pool_kernel_size});
    // 4) Mish activation
    at::Tensor y2 = mish_activate(y1);
    return y2;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_subtract_hard_swish_max_pool_mish_custom", &conv2d_subtract_hard_swish_max_pool_mish_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_subtract_hard_swish_max_pool_mish_custom", &conv2d_subtract_hard_swish_max_pool_mish_custom_impl_npu, "conv2d -> subtract -> hardswish -> maxpool -> mish (fused call)");
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep nn.Conv2d, fuse subtract + HardSwish + MaxPool + Mish into a custom op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract_value, pool_kernel_size):
        super().__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract_value = float(subtract_value)
        self.pool_kernel_size = int(pool_kernel_size)

    def forward(self, x):
        x = self.conv(x)
        # Fused: subtract -> HardSwish -> MaxPool -> Mish
        x = custom_ops_lib.conv2d_subtract_hard_swish_max_pool_mish_custom(x, self.subtract_value, self.pool_kernel_size)
        return x
