kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTakeAlongDim {
public:
    __aicore__ inline KernelTakeAlongDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR y,
                                uint32_t b, uint32_t n, uint32_t m, uint32_t tileNum)
    {
        this->B = b;
        this->N = n;
        this->M = m;
        this->tileNum = tileNum;

        // 每个core处理的行数
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = (B + blockNum - 1) / blockNum;  // 向上取整
        this->rowStart = blockIdx * this->rowsPerBlock;
        this->rowEnd = AscendC::Min(this->rowStart + this->rowsPerBlock, B);

        // GM缓冲区设置：x:[B*N], idx:[B*M], y:[B*M]
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, B * N);
        idxGm.SetGlobalBuffer((__gm__ int32_t *)idx, B * M);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, B * M);

        // 列方向分块
        this->tileLen = AscendC::Max((uint32_t)1, (M / tileNum / BUFFER_NUM));
        if (this->tileLen == 0) { this->tileLen = 1; }

        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, this->tileLen * sizeof(int32_t));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = this->rowStart; r < this->rowEnd; ++r) {
            uint32_t tilesPerRow = (M + this->tileLen - 1) / this->tileLen;
            int32_t loopCount = tilesPerRow * BUFFER_NUM;
            for (int32_t p = 0; p < loopCount; ++p) {
                if (p < loopCount) { CopyIn(r, p); }
                if (p > 0) { Compute(r, p - 1); }
                if (p > 1) { CopyOut(r, p - 2); }
            }
            // 处理尾部两次未落盘的阶段
            if (loopCount >= 2) {
                Compute(r, loopCount - 1);
                CopyOut(r, loopCount - 2);
                CopyOut(r, loopCount - 1);
            } else if (loopCount == 1) {
                Compute(r, 0);
                CopyOut(r, 0);
            }
        }
    }

private:
    __aicore__ inline uint32_t TileStart(uint32_t tileIdx) const
    {
        return tileIdx * this->tileLen;
    }

    __aicore__ inline uint32_t TileLen(uint32_t tileIdx) const
    {
        uint32_t start = TileStart(tileIdx);
        return start + this->tileLen <= M ? this->tileLen : (M - start);
    }

    __aicore__ inline void CopyIn(uint32_t row, int32_t progress)
    {
        uint32_t tileIdx = progress % ((M + this->tileLen - 1) / this->tileLen);
        uint32_t curLen = TileLen(tileIdx);
        if (curLen == 0) return;

        AscendC::LocalTensor<int32_t> idxLocal = inQueueIdx.AllocTensor<int32_t>();
        uint32_t gmOffsetIdx = row * M + TileStart(tileIdx);
        AscendC::DataCopy(idxLocal, idxGm[gmOffsetIdx], curLen);
        inQueueIdx.EnQue(idxLocal);
    }

    __aicore__ inline void Compute(uint32_t row, int32_t progress)
    {
        uint32_t tileIdx = progress % ((M + this->tileLen - 1) / this->tileLen);
        uint32_t curLen = TileLen(tileIdx);
        if (curLen == 0) return;

        AscendC::LocalTensor<int32_t> idxLocal = inQueueIdx.DeQue<int32_t>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // 对当前tile内的每个位置执行按列gather
        // y[row, j] = x[row, idx[row, j]]
        uint32_t rowBaseX = row * N;
        for (uint32_t i = 0; i < curLen; ++i) {
            // 从UB读取索引值
            int32_t col = idxLocal.GetValue(i);
            // 边界保护
            if (col < 0) col = 0;
            if ((uint32_t)col >= N) col = (int32_t)(N - 1);

            // 将x的单元素复制到y的对应位置（均在UB内）
            AscendC::DataCopy(yLocal[i], xGm[rowBaseX + (uint32_t)col], 1);
        }

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueIdx.FreeTensor(idxLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row, int32_t progress)
    {
        uint32_t tileIdx = progress % ((M + this->tileLen - 1) / this->tileLen);
        uint32_t curLen = TileLen(tileIdx);
        if (curLen == 0) return;

        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        uint32_t gmOffsetY = row * M + TileStart(tileIdx);
        AscendC::DataCopy(yGm[gmOffsetY], yLocal, curLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<int32_t> idxGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t B, N, M;
    uint32_t tileNum;
    uint32_t tileLen;
    uint32_t rowsPerBlock;
    uint32_t rowStart, rowEnd;
};

extern "C" __global__ __aicore__ void take_along_dim_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTakeAlongDim op;
    op.Init(x, idx, y, tiling_data.b, tiling_data.n, tiling_data.m, tiling_data.tileNum);
    op.Process();
}
"""
