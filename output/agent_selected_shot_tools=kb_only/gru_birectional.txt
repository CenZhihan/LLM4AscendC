const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // x: [T, B, I]
    const gert::Shape* x_shape = context->GetInputShape(0);
    if (x_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }
    uint64_t dims = x_shape->GetDimNum();
    if (dims < 1) {
        return ge::GRAPH_FAILED;
    }

    uint64_t T = 1, B = 1, I = 1;
    if (dims == 1) {
        I = x_shape->GetDim(0);
    } else if (dims == 2) {
        T = x_shape->GetDim(0);
        I = x_shape->GetDim(1);
    } else {
        // typical GRU input: [seq_len, batch, input_size]
        T = x_shape->GetDim(0);
        B = x_shape->GetDim(1);
        I = x_shape->GetDim(2);
    }

    GruBirectionalCustomTilingData tiling;
    uint64_t totalLength = T * B * I;
    if (totalLength == 0) {
        return ge::GRAPH_FAILED;
    }

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_inputSize(static_cast<uint32_t>(I));
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // no workspace

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling


namespace ge {

// Infer output shape as [T, B, 2 * I]
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0); // [T, B, I]
    if (x_shape == nullptr) {
        return GRAPH_FAILED;
    }
    gert::Shape* y_shape = context->GetOutputShape(0);

    std::vector<int64_t> out_dims;
    out_dims.reserve(x_shape->GetDimNum());
    for (uint64_t i = 0; i < x_shape->GetDimNum(); ++i) {
        out_dims.push_back(x_shape->GetDim(i));
    }
    if (out_dims.size() == 0) {
        // scalar not supported
        return GRAPH_FAILED;
    }
    // double the last dim: 2 * I
    out_dims.back() = out_dims.back() * 2;

    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// 该核实现：对输入 x 按最后一维进行双向拼接 y = concat(x, x)，形状从 [..., I] -> [..., 2*I]
// 不使用 h0，仅做占位，便于框架替换
class KernelGruBidirCat {
public:
    __aicore__ inline KernelGruBidirCat() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR y, uint32_t totalLength, uint32_t inputSize, uint32_t tileNum)
    {
        (void)h0; // 未使用
        this->inputSize = inputSize;           // I
        this->unitSizeIn  = inputSize;         // I
        this->unitSizeOut = inputSize * 2;     // 2*I
        this->unitsTotal  = totalLength / inputSize;  // T*B
        this->blockUnits  = unitsTotal / AscendC::GetBlockNum(); // 每个计算块处理多少个 unit
        this->blockInLen  = this->blockUnits * this->unitSizeIn;  // 本 block 读入元素数
        this->blockOutLen = this->blockUnits * this->unitSizeOut; // 本 block 写出元素数

        // 为了保证每个 tile 是若干完整 unit，tileLength 取 unitSizeIn 的倍数
        this->tileNum  = tileNum;
        uint32_t rawTile = this->blockInLen / tileNum / BUFFER_NUM;
        uint32_t unitsPerTile = AscendC::Max(1u, rawTile / this->unitSizeIn);
        this->tileInLen  = unitsPerTile * this->unitSizeIn;    // 输入 tile 长度
        this->tileOutLen = unitsPerTile * this->unitSizeOut;   // 输出 tile 长度

        // 计算 GM 起始偏移
        uint32_t blockStartInOffset  = this->blockInLen * AscendC::GetBlockIdx();
        uint32_t blockStartUnitIdx   = blockStartInOffset / this->unitSizeIn;
        uint32_t blockStartOutOffset = blockStartUnitIdx * this->unitSizeOut;

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x + blockStartInOffset,  this->blockInLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y + blockStartOutOffset, this->blockOutLen);

        // 队列与本地缓存
        pipe.InitBuffer(inQueueX,  BUFFER_NUM, this->tileInLen  * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileOutLen * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = (this->blockInLen + this->tileInLen - 1) / this->tileInLen;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        // 计算本次 tile 实际长度（末尾可能不足整 tile）
        uint32_t remainIn = this->blockInLen - progress * this->tileInLen;
        uint32_t curInLen = remainIn > this->tileInLen ? this->tileInLen : remainIn;
        // 保证按 unit 对齐
        curInLen = (curInLen / this->unitSizeIn) * this->unitSizeIn;

        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileInLen], curInLen);
        curTileInLen = curInLen; // 保存供后续阶段使用
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        (void)progress;
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();

        // xLocal:  [unitsPerTile * I]
        // yLocal:  [unitsPerTile * 2I] = concat(x, x) 按 unit 复制两份
        uint32_t unitsPerTile = curTileInLen / this->unitSizeIn;
        for (uint32_t u = 0; u < unitsPerTile; ++u) {
            uint32_t srcOff  = u * this->unitSizeIn;
            uint32_t dstOff1 = u * this->unitSizeOut;
            uint32_t dstOff2 = dstOff1 + this->unitSizeIn;
            AscendC::DataCopy(yLocal[dstOff1], xLocal[srcOff], this->unitSizeIn);
            AscendC::DataCopy(yLocal[dstOff2], xLocal[srcOff], this->unitSizeIn);
        }

        outQueueY.EnQue<DTYPE>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();

        uint32_t unitsPerTile = curTileInLen / this->unitSizeIn;
        uint32_t curOutLen = unitsPerTile * this->unitSizeOut;

        AscendC::DataCopy(yGm[progress * this->tileOutLen], yLocal, curOutLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t inputSize;
    uint32_t unitSizeIn;
    uint32_t unitSizeOut;
    uint32_t unitsTotal;

    uint32_t blockUnits;
    uint32_t blockInLen;
    uint32_t blockOutLen;

    uint32_t tileNum;
    uint32_t tileInLen;
    uint32_t tileOutLen;

    uint32_t curTileInLen; // 动态记录当前 tile 输入长度
};

extern "C" __global__ __aicore__ void gru_birectional_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelGruBidirCat op;
    op.Init(x, h0, y, tiling_data.totalLength, tiling_data.inputSize, tiling_data.tileNum);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib
import torch.nn.functional as F

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_size, num_layers=3, bias=True, batch_first=False):
        super().__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.bias = bias
        self.batch_first = batch_first

    def forward(self, x, h0):
        # 自定义核：输出 [..., 2*input_size]，将 x 在最后一维拼接自身，作为双向占位
        y_cat = custom_ops_lib.gru_birectional_custom(x, h0)  # [T, B, 2*I]（或 [B, T, 2*I] 取决于输入）
        # 若期望输出为 2*hidden_size，则做维度对齐：
        in_feat = self.input_size
        out_feat = self.hidden_size
        if out_feat > in_feat:
            # 末尾补零到 2*hidden_size
            pad = (0, 2 * (out_feat - in_feat))  # 填充最后一维
            y = F.pad(y_cat, pad, mode="constant", value=0.0)
        elif out_feat < in_feat:
            # 截断到 2*hidden_size
            y = y_cat[..., : 2 * out_feat]
        else:
            y = y_cat
        return y
'''
