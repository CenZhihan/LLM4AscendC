kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

class KernelGemmGroupNormHardtanh {
public:
    __aicore__ inline KernelGemmGroupNormHardtanh() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR gamma, GM_ADDR beta,
                                GM_ADDR groups, GM_ADDR minval, GM_ADDR maxval, GM_ADDR y,
                                uint32_t n, uint32_t k, uint32_t m, uint32_t tileNum)
    {
        this->n = n;
        this->k = k;
        this->m = m;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x, n * k);
        wGm.SetGlobalBuffer((__gm__ DTYPE *)w, m * k);
        bGm.SetGlobalBuffer((__gm__ DTYPE *)b, m);
        gammaGm.SetGlobalBuffer((__gm__ DTYPE *)gamma, m);
        betaGm.SetGlobalBuffer((__gm__ DTYPE *)beta, m);
        groupsGm.SetGlobalBuffer((__gm__ int32_t *)groups, 1);
        minvalGm.SetGlobalBuffer((__gm__ DTYPE *)minval, 1);
        maxvalGm.SetGlobalBuffer((__gm__ DTYPE *)maxval, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y, n * m);

        // simple tiling on rows
        this->rowsPerBlock = (n + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        this->rowStart = this->rowsPerBlock * AscendC::GetBlockIdx();
        this->rowEnd = (this->rowStart + this->rowsPerBlock > n) ? n : (this->rowStart + this->rowsPerBlock);

        // setup pipe buffers for one row and full output row
        pipe.InitBuffer(inQueueX, 1, k * sizeof(DTYPE));
        pipe.InitBuffer(inQueueB, 1, m * sizeof(DTYPE));
        pipe.InitBuffer(inQueueW, 1, m * k * sizeof(DTYPE));  // naive copy for sample
        pipe.InitBuffer(inQueueGamma, 1, m * sizeof(DTYPE));
        pipe.InitBuffer(inQueueBeta, 1, m * sizeof(DTYPE));
        pipe.InitBuffer(inQueueScalars, 1, 3 * sizeof(DTYPE)); // min, max, groups as float buffer
        pipe.InitBuffer(outQueueY, 1, m * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // Load shared parameters once
        AscendC::LocalTensor<DTYPE> bLocal = inQueueB.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> gammaLocal = inQueueGamma.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> betaLocal = inQueueBeta.AllocTensor<DTYPE>();
        AscendC::DataCopy(bLocal, bGm, m);
        AscendC::DataCopy(gammaLocal, gammaGm, m);
        AscendC::DataCopy(betaLocal, betaGm, m);
        inQueueB.EnQue(bLocal);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);

        // scalar parameters
        AscendC::LocalTensor<DTYPE> scalarsLocal = inQueueScalars.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> minLocal(reinterpret_cast<__ubuf__ DTYPE *>(scalarsLocal.GetPhyAddr()), 1);
        AscendC::LocalTensor<DTYPE> maxLocal(reinterpret_cast<__ubuf__ DTYPE *>(scalarsLocal.GetPhyAddr()) + 1, 1);
        AscendC::DataCopy(minLocal, minvalGm, 1);
        AscendC::DataCopy(maxLocal, maxvalGm, 1);
        inQueueScalars.EnQue(scalarsLocal);

        // naive load of weight
        AscendC::LocalTensor<DTYPE> wLocal = inQueueW.AllocTensor<DTYPE>();
        AscendC::DataCopy(wLocal, wGm, m * k);
        inQueueW.EnQue(wLocal);

        // process rows assigned to this block
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            CopyInRow(r);
            ComputeRow();
            CopyOutRow(r);
        }

        // free shared
        inQueueB.FreeTensor(inQueueB.DeQue<DTYPE>());
        inQueueGamma.FreeTensor(inQueueGamma.DeQue<DTYPE>());
        inQueueBeta.FreeTensor(inQueueBeta.DeQue<DTYPE>());
        inQueueW.FreeTensor(inQueueW.DeQue<DTYPE>());
        inQueueScalars.FreeTensor(inQueueScalars.DeQue<DTYPE>());
    }

private:
    __aicore__ inline void CopyInRow(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[r * k], k);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        // tensors
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> wLocal = inQueueW.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = inQueueB.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> gammaLocal = inQueueGamma.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> betaLocal = inQueueBeta.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> scalarsLocal = inQueueScalars.DeQue<DTYPE>();
        DTYPE minv = scalarsLocal.GetValue(0);
        DTYPE maxv = scalarsLocal.GetValue(1);

        // initialize yLocal with bias
        // duplicate bias to yLocal and then will add GEMM result
        AscendC::Duplicate(yLocal, bLocal, m);

        // naive GEMM: y_i = sum_k x[k] * W[i, k] + b[i]
        for (uint32_t i = 0; i < m; ++i) {
            // compute dot(x, W_i)
            // get pointer to W row i
            AscendC::LocalTensor<DTYPE> wi(reinterpret_cast<__ubuf__ DTYPE *>(wLocal.GetPhyAddr()) + i * k, k);
            AscendC::LocalTensor<DTYPE> prod(reinterpret_cast<__ubuf__ DTYPE *>(xLocal.GetPhyAddr()), k); // reuse buffer
            AscendC::Mul(prod, xLocal, wi, k);
            // sum reduction (assume ReduceSum writes one element back to prod[0])
            AscendC::ReduceSum(prod, prod, k);
            DTYPE acc = prod.GetValue(0);
            // y[i] = acc + b[i]
            DTYPE yv = acc + bLocal.GetValue(i);
            yLocal.SetValue(i, yv);
        }

        // elementwise affine (gamma, beta)
        AscendC::Mul(yLocal, yLocal, gammaLocal, m);
        AscendC::Add(yLocal, yLocal, betaLocal, m);

        // HardTanh: clamp to [minv, maxv]
        AscendC::Maxs(yLocal, yLocal, minv, m);
        AscendC::Mins(yLocal, yLocal, maxv, m);

        outQueueY.EnQue(yLocal);

        // re-enqueue shared tensors for next row
        inQueueW.EnQue(wLocal);
        inQueueB.EnQue(bLocal);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);
        inQueueScalars.EnQue(scalarsLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[r * m], yLocal, m);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX, inQueueB, inQueueW, inQueueGamma, inQueueBeta, inQueueScalars;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> gammaGm;
    AscendC::GlobalTensor<DTYPE> betaGm;
    AscendC::GlobalTensor<int32_t> groupsGm;
    AscendC::GlobalTensor<DTYPE> minvalGm;
    AscendC::GlobalTensor<DTYPE> maxvalGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t n;
    uint32_t k;
    uint32_t m;
    uint32_t tileNum;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void gemm_group_norm_hardtanh_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                      GM_ADDR gamma, GM_ADDR beta, GM_ADDR groups,
                                                                      GM_ADDR minval, GM_ADDR maxval,
                                                                      GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmGroupNormHardtanh op;
    op.Init(x, weight, bias, gamma, beta, groups, minval, maxval, y,
            tiling_data.n, tiling_data.k, tiling_data.m, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, num_groups, hardtanh_min, hardtanh_max):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features)
        self.hardtanh_min = hardtanh_min
        self.hardtanh_max = hardtanh_max
        self.num_groups = num_groups

    def forward(self, x):
        # 调用自定义融合算子：输入为 x、权重/偏置、GN缩放/平移以及硬阈值，输出为 y
        groups_t = torch.tensor([self.num_groups], dtype=torch.int32, device=x.device)
        min_t = torch.tensor([self.hardtanh_min], dtype=x.dtype, device=x.device)
        max_t = torch.tensor([self.hardtanh_max], dtype=x.dtype, device=x.device)
        return custom_ops_lib.gemm_group_norm_hardtanh_custom(
            x, self.gemm.weight, self.gemm.bias,
            self.group_norm.weight, self.group_norm.bias,
            groups_t, min_t, max_t
        )
