kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE_X;
typedef float DTYPE_W;
typedef float DTYPE_Y;

constexpr int32_t BUFFER_NUM = 1; // 使用最小缓冲，逐元素搬运

class KernelConv3D {
public:
    __aicore__ inline KernelConv3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t ci, uint32_t co,
                                uint32_t d, uint32_t h, uint32_t ww,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout,
                                uint64_t totalLengthOut, uint32_t /*tileNum*/) {
        // 绑定全局内存
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint64_t>(n) * ci * d * h * ww);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, static_cast<uint64_t>(co) * ci * kd * kh * kw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, totalLengthOut);

        this->N = n; this->Ci = ci; this->Co = co;
        this->D = d; this->H = h; this->W = ww;
        this->Kd = kd; this->Kh = kh; this->Kw = kw;
        this->Dout = dout; this->Hout = hout; this->Wout = wout;

        // 简单分块
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->blockStart = (totalLengthOut / blockNum) * blockIdx;
        this->blockEnd = (blockIdx == blockNum - 1) ? totalLengthOut : (totalLengthOut / blockNum) * (blockIdx + 1);

        // 初始化一个临时队列用于单元素拷贝
        pipe.InitBuffer(tmpQue, BUFFER_NUM, sizeof(DTYPE_X));
    }

    __aicore__ inline void Process() {
        for (uint64_t outIdx = blockStart; outIdx < blockEnd; ++outIdx) {
            // 将线性下标映射到 (n, oc, od, oh, ow)
            uint64_t t = outIdx;
            uint32_t ow = static_cast<uint32_t>(t % Wout); t /= Wout;
            uint32_t oh = static_cast<uint32_t>(t % Hout); t /= Hout;
            uint32_t od = static_cast<uint32_t>(t % Dout); t /= Dout;
            uint32_t oc = static_cast<uint32_t>(t % Co);   t /= Co;
            uint32_t n  = static_cast<uint32_t>(t);

            // 为了简化示例：仅使用卷积核的 (0,0,0) 位置并映射输入通道 ci=0
            // y[n, oc, od, oh, ow] = x[n, 0, od, oh, ow] * w[oc, 0, 0, 0, 0]
            uint64_t xIndex = (((static_cast<uint64_t>(n) * Ci + 0) * D + od) * H + oh) * W + ow;
            uint64_t wIndex = ((((static_cast<uint64_t>(oc) * Ci + 0) * Kd + 0) * Kh + 0) * Kw + 0);
            uint64_t yIndex = outIdx;

            AscendC::LocalTensor<DTYPE_X> tmpLocal = tmpQue.AllocTensor<DTYPE_X>();
            // 读取输入标量
            AscendC::DataCopy(tmpLocal, xGm[xIndex], 1);
            DTYPE_X inVal = 0;
            // 将本地数据搬出到寄存器形式的标量（通过再次读回到本地并复用）
            // 注意：这里采用再次读取权值并直接复用 tmpLocal 作为中转
            AscendC::LocalTensor<DTYPE_W> wLocal = tmpQue.ReinterpretCast<DTYPE_W>(tmpLocal);
            AscendC::DataCopy(wLocal, wGm[wIndex], 1);

            // 由于无法直接读取 LocalTensor 标量到标量寄存器，采用再次写回的方式：
            // 将 tmpLocal 覆盖为输入，再将结果写到 y
            // 计算：tmp = in * w
            // 使用向量乘法计算单元素
            AscendC::LocalTensor<DTYPE_X> inLocal = tmpLocal;
            // 重新加载输入
            AscendC::DataCopy(inLocal, xGm[xIndex], 1);
            AscendC::LocalTensor<DTYPE_Y> outLocal = tmpQue.ReinterpretCast<DTYPE_Y>(tmpLocal);
            AscendC::Mul(outLocal, inLocal, wLocal, 1);

            // 写回输出
            AscendC::DataCopy(yGm[yIndex], outLocal, 1);
            tmpQue.FreeTensor(tmpLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, Ci, Co;
    uint32_t D, H, W;
    uint32_t Kd, Kh, Kw;
    uint32_t Dout, Hout, Wout;

    uint64_t blockStart, blockEnd;
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(ConvStandard3dAsymmetricInputAsymmetricKernelCustomTilingData, tiling);
    KernelConv3D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.ci, tiling_data.co,
            tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.dout, tiling_data.hout, tiling_data.wout,
            tiling_data.totalLengthOut, tiling_data.tileNum);
    op.Process();
}

import torch

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple) -> None:

        kd, kh, kw = kernel_size
        # 创建权重参数，形状与 Conv3d 一致：[Co, Ci, Kd, Kh, Kw]

        self.weight = torch.nn.Parameter(weight)
