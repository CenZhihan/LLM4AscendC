        (void)attrs->GetInt("pad_bottom", pad_bottom);
        (void)attrs->GetInt("pad_left", pad_left);
        (void)attrs->GetInt("pad_right", pad_right);
        (void)attrs->GetInt("dilation_h", dilation_h);
        (void)attrs->GetInt("dilation_w", dilation_w);
    }

    if (cin != cin_w) {
        return GRAPH_FAILED;
    }

    int64_t ho = (h + pad_top + pad_bottom - dilation_h * (kh - 1) - 1) / stride_h + 1;

    yShape->SetDim(3, wo);
    return GRAPH_SUCCESS;

#include "kernel_operator.h"

using uint64 = unsigned long long;

class KernelConv2DNaive {
public:
    __aicore__ inline KernelConv2DNaive() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t ww, uint32_t kh, uint32_t kw,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_top, uint32_t pad_bottom,
                                uint32_t pad_left, uint32_t pad_right,
                                uint32_t dilation_h, uint32_t dilation_w,
                                uint32_t ho, uint32_t wo, uint64 totalOut)
    {
        this->n = n; this->cin = cin; this->cout = cout;
        this->h = h; this->w = ww; this->kh = kh; this->kw = kw;
        this->stride_h = stride_h; this->stride_w = stride_w;
        this->pad_top = pad_top; this->pad_bottom = pad_bottom;
        this->pad_left = pad_left; this->pad_right = pad_right;
        this->dilation_h = dilation_h; this->dilation_w = dilation_w;
        this->ho = ho; this->wo = wo;
        this->totalOut = totalOut;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64)n * cin * h * ww);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint64)cout * cin * kh * kw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint64)n * cout * ho * wo);

        // 每个Block负责的输出元素区间
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint64 base = totalOut / blockNum;
        uint64 rem = totalOut % blockNum;
        this->startIdx = base * blockIdx + (blockIdx < rem ? blockIdx : rem);
        this->workLen = base + (blockIdx < rem ? 1 : 0);

        // 为标量运算准备的最小本地缓冲队列
        pipe.InitBuffer(scalarQ, 4, 1 * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint64 i = 0; i < workLen; ++i) {
            uint64 outLinear = startIdx + i;

            // 反解出 (b, oc, oh, ow)
            uint64 ow_idx = outLinear % wo;
            uint64 tmp1 = outLinear / wo;
            uint64 oh_idx = tmp1 % ho;
            uint64 tmp2 = tmp1 / ho;
            uint64 oc_idx = tmp2 % cout;
            uint64 b_idx  = tmp2 / cout;

            // UB中构建一个标量累加器 acc = 0
            AscendC::LocalTensor<DTYPE_Y> acc = scalarQ.AllocTensor<DTYPE_Y>();
            AscendC::Duplicate(acc, (DTYPE_Y)0, 1);

            // 遍历卷积核和输入通道
            for (uint32_t ic = 0; ic < cin; ++ic) {
                for (uint32_t r = 0; r < kh; ++r) {
                    int64_t ih = (int64_t)oh_idx * (int64_t)stride_h - (int64_t)pad_top + (int64_t)r * (int64_t)dilation_h;
                    if (ih < 0 || ih >= (int64_t)h) { continue; }
                    for (uint32_t c = 0; c < kw; ++c) {
                        int64_t iw = (int64_t)ow_idx * (int64_t)stride_w - (int64_t)pad_left + (int64_t)c * (int64_t)dilation_w;
                        if (iw < 0 || iw >= (int64_t)w) { continue; }

                        // 计算x, w线性偏移
                        uint64 xOff = (((uint64)b_idx * cin + ic) * h + (uint64)ih) * w + (uint64)iw;
                        uint64 wOff = ((((uint64)oc_idx * cin) + ic) * kh + r) * kw + c;

                        // 从GM中取标量 x 与 w
                        AscendC::LocalTensor<DTYPE_X> xScalar = scalarQ.AllocTensor<DTYPE_X>();
                        AscendC::LocalTensor<DTYPE_W> wScalar = scalarQ.AllocTensor<DTYPE_W>();
                        AscendC::DataCopy(xScalar, xGm[xOff], 1);
                        AscendC::DataCopy(wScalar, wGm[wOff], 1);

                        // 计算 tmp = x * w
                        AscendC::LocalTensor<DTYPE_Y> tmp = scalarQ.AllocTensor<DTYPE_Y>();
                        AscendC::Mul(tmp, xScalar, wScalar, 1);

                        // acc += tmp
                        AscendC::Add(acc, acc, tmp, 1);

                        scalarQ.FreeTensor(xScalar);
                        scalarQ.FreeTensor(wScalar);
                        scalarQ.FreeTensor(tmp);
                    }
                }
            }

            // 将结果写回GM
            uint64 yOff = (((uint64)b_idx * cout + oc_idx) * ho + oh_idx) * wo + ow_idx;
            AscendC::DataCopy(yGm[yOff], acc, 1);
            scalarQ.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 4> scalarQ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, cin, cout, h, w, kh, kw;
    uint32_t stride_h, stride_w;
    uint32_t pad_top, pad_bottom, pad_left, pad_right;
    uint32_t dilation_h, dilation_w;
    uint32_t ho, wo;
    uint64 totalOut;
    uint64 startIdx;
    uint64 workLen;
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2DNaive op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w, tiling_data.kh, tiling_data.kw,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_top, tiling_data.pad_bottom,
            tiling_data.pad_left, tiling_data.pad_right,
            tiling_data.dilation_h, tiling_data.dilation_w,
            tiling_data.ho, tiling_data.wo, tiling_data.totalOut);
    op.Process();
}
