kernel_src="""
#include "kernel_operator.h"

// 简化实现示例：对输入进行 Swish -> GroupNorm -> HardSwish 的融合。
// 输入张量视为 NCDHW（5D），在 GroupNorm 中将 D 与 N 展平为 N' = N*D 以适配 4D 接口。

template<typename T>
__aicore__ inline void HardSwish(AscendC::LocalTensor<T>& dst, const AscendC::LocalTensor<T>& src, uint32_t count)
{
    // hswish(x) = x * relu6(x + 3) / 6
    AscendC::LocalTensor<T> tmp1 = dst; // 复用 dst 作中间缓冲
    AscendC::Adds(tmp1, src, (T)3.0f, count);         // tmp1 = x + 3
    // 需要 Max/Min 饱和，若没有对应 API，可用 Clip 的组合，示例中使用 Mins/Maxs
    AscendC::LocalTensor<T> tmp2 = dst;               // 复用
    AscendC::Maxs(tmp2, tmp1, (T)0.0f, count);        // tmp2 = max(tmp1, 0)
    AscendC::Mins(tmp1, tmp2, (T)6.0f, count);        // tmp1 = min(tmp2, 6)
    AscendC::Mul(dst, src, tmp1, count);              // dst = x * tmp1
    AscendC::Muls(dst, dst, (T)(1.0f/6.0f), count);   // dst = dst / 6
}

class KernelGNFused {
public:
    __aicore__ inline KernelGNFused() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t groups, float eps, uint32_t tileNum)
    {
        this->n = n; this->c = c; this->d = d; this->h = h; this->w = w;
        this->groups = groups; this->eps = eps;
        this->tileNum = tileNum;

        this->n_flat = n * d;
        this->total = n * c * d * h * w;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), total);
        gammaGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(gamma), c);
        betaGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(beta), c);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), total);

        // 简化：一次性分配整块缓冲
        pipe.InitBuffer(inQueueX, 1, sizeof(float) * total);
        pipe.InitBuffer(inQueueGamma, 1, sizeof(float) * c);
        pipe.InitBuffer(inQueueBeta, 1, sizeof(float) * c);
        pipe.InitBuffer(outQueueY, 1, sizeof(float) * total);

        // mean/var 缓冲（GroupNorm 输出均值/方差），大小 [N', groups]
        pipe.InitBuffer(meanBuffer, (sizeof(float) * n_flat * groups + 31) / 32 * 32);
        pipe.InitBuffer(varBuffer, (sizeof(float) * n_flat * groups + 31) / 32 * 32);
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma.AllocTensor<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueBeta.AllocTensor<float>();

        AscendC::DataCopy(xLocal, xGm, total);
        AscendC::DataCopy(gammaLocal, gammaGm, c);
        AscendC::DataCopy(betaLocal, betaGm, c);

        inQueueX.EnQue(xLocal);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma.DeQue<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueBeta.DeQue<float>();

        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        AscendC::LocalTensor<float> meanLocal = meanBuffer.Get<float>();
        AscendC::LocalTensor<float> varianceLocal = varBuffer.Get<float>();

        // Swish: x * sigmoid(x) = x / (1 + exp(-x))
        AscendC::LocalTensor<float> tmp1 = yLocal; // 复用 yLocal 作为中间缓冲
        AscendC::Muls(tmp1, xLocal, -1.0f, total);    // tmp1 = -x
        AscendC::Exp(tmp1, tmp1, total);              // tmp1 = exp(-x)
        AscendC::Adds(tmp1, tmp1, 1.0f, total);       // tmp1 = 1 + exp(-x)
        // dst = x / tmp1
        AscendC::Div(yLocal, xLocal, tmp1, total);    // yLocal = x * sigmoid(x)

        // GroupNorm（将 D 合并到 N 维：N' = N*D）
        // 形状信息与 tiling
        uint32_t n_flatten = n_flat;
        DataFormat dataFormat = DataFormat::ND;

        // 准备 stack buffer 大小
        uint32_t stackBufferSize = 0;
        {
            AscendC::LocalTensor<float> stackBuffer;
            bool ok = AscendC::PopStackBuffer<float, AscendC::TPosition::LCM>(stackBuffer);
            (void)ok;
            stackBufferSize = stackBuffer.GetSize();
        }

        AscendC::GroupNormTiling groupNormTiling;
        uint32_t inputShape[4] = { n_flatten, c, h, w };
        AscendC::ShapeInfo shapeInfo{ (uint8_t)4, inputShape, (uint8_t)4, inputShape, dataFormat };
        AscendC::GetGroupNormNDTillingInfo(shapeInfo, stackBufferSize, sizeof(float), false, groups, groupNormTiling);

        // yLocal = GN( Swish(x) )
        AscendC::GroupNorm<float, false>(yLocal, meanLocal, varianceLocal, yLocal, gammaLocal, betaLocal, (float)eps, groupNormTiling);

        // HardSwish 后处理
        HardSwish<float>(yLocal, yLocal, total);

        outQueueY.EnQue(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueGamma.FreeTensor(gammaLocal);
        inQueueBeta.FreeTensor(betaLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm, yLocal, total);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  inQueueGamma;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  inQueueBeta;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;
    AscendC::TBuf<AscendC::TPosition::VECCALC> meanBuffer, varBuffer;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> gammaGm;
    AscendC::GlobalTensor<float> betaGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n, c, d, h, w;
    uint32_t n_flat;
    uint32_t groups;
    float eps;
    uint32_t total;
    uint32_t tileNum;
};

extern "C" __global__ __aicore__ void conv_transpose3d_swish_group_norm_hard_swish_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelGNFused op;
    op.Init(x, gamma, beta, y,
            tilingData.n(), tilingData.c(), tilingData.d(), tilingData.h(), tilingData.w(),
            tilingData.groups(), tilingData.eps(), tilingData.tileNum());
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版本：保持 PyTorch 的 ConvTranspose3d，实现 Swish + GroupNorm + HardSwish 的自定义 AscendC 融合算子
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, groups, eps, bias=True):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        self.group_norm = torch.nn.GroupNorm(num_groups=groups, num_channels=out_channels, eps=eps)

    def forward(self, x):
        x = self.conv_transpose(x)
        # 将 Swish + GroupNorm + HardSwish 融合到自定义 OP
        x = custom_ops_lib.conv_transpose3d_swish_group_norm_hard_swish_custom(x, self.group_norm.weight, self.group_norm.bias)
        return x
