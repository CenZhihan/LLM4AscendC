kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2; // queue depth for output tiles

class KernelConv2DDirect {
public:
    __aicore__ inline KernelConv2DDirect() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t Hout, uint32_t Wout,
                                uint32_t totalOut, uint32_t tileNum)
    {
        this->N = N; this->Cin = Cin; this->Cout = Cout;
        this->H = H; this->W = W; this->K = K;
        this->Hout = Hout; this->Wout = Wout;
        this->totalOut = totalOut;

        // Split output across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        this->blockLength = (blocks > 0) ? (totalOut / blocks) : totalOut;
        this->tileNum = tileNum;
        this->tileLength = (this->tileNum > 0 && BUFFER_NUM > 0) ? (this->blockLength / (this->tileNum * BUFFER_NUM)) : this->blockLength;

        // Set GM buffers; x and w use full buffers, y is sharded by block
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, N * Cin * H * W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, Cout * Cin * K * K);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // Output tile buffer
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));

        // Scalar workspace buffer for single element transfers
        pipe.InitBuffer(workScalarQueue, 1, 1 * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // Fill the output tile by direct convolution (stride=1, pad=0, dilation=1)
        for (uint32_t t = 0; t < this->tileLength; ++t) {
            uint32_t gidx = progress * this->tileLength + t; // index within current block
            DTYPE_Y acc = (DTYPE_Y)0.0f;

            // Map to global output index
            uint32_t globalIdx = this->blockLength * AscendC::GetBlockIdx() + gidx;
            if (globalIdx >= this->totalOut) {
                yLocal.SetValue(t, (DTYPE_Y)0.0f);
                continue;
            }

            uint32_t ow = globalIdx % Wout;
            uint32_t tmp1 = globalIdx / Wout;
            uint32_t oh = tmp1 % Hout;
            uint32_t tmp2 = tmp1 / Hout;
            uint32_t oc = tmp2 % Cout;
            uint32_t n  = tmp2 / Cout;

            // Convolution accumulation
            for (uint32_t ic = 0; ic < Cin; ++ic) {
                for (uint32_t kh = 0; kh < K; ++kh) {
                    uint32_t ih = oh + kh; // pad=0, stride=1
                    for (uint32_t kw = 0; kw < K; ++kw) {
                        uint32_t iw = ow + kw;

                        // compute linear indices
                        uint32_t x_off = (((n * Cin + ic) * H) + ih) * W + iw;
                        uint32_t w_off = ((((oc * Cin) + ic) * K) + kh) * K + kw;

                        // load x value (scalar) from GM
                        DTYPE_X xv = (DTYPE_X)0.0f;
                        {
                            AscendC::LocalTensor<DTYPE_X> tmp = workScalarQueue.AllocTensor<DTYPE_X>();
                            AscendC::DataCopy(tmp, xGm[x_off], 1);
                            xv = tmp.GetValue(0);
                            workScalarQueue.FreeTensor(tmp);
                        }

                        // load w value (scalar) from GM
                        DTYPE_W wv = (DTYPE_W)0.0f;
                        {
                            AscendC::LocalTensor<DTYPE_W> tmp = workScalarQueue.AllocTensor<DTYPE_W>();
                            AscendC::DataCopy(tmp, wGm[w_off], 1);
                            wv = tmp.GetValue(0);
                            workScalarQueue.FreeTensor(tmp);
                        }

                        acc += (DTYPE_Y)(xv * wv);
                    }
                }
            }

            yLocal.SetValue(t, acc);
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> workScalarQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, Cin, Cout, H, W, K, Hout, Wout;
    uint32_t totalOut;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2DDirect op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w, tiling_data.k,
            tiling_data.hout, tiling_data.wout,
            tiling_data.totalOut, tiling_data.tileNum);
    op.Process();
}

    auto Wout = W - K + 1;

    at::Tensor y = at::empty({N, Cout, Hout, Wout}, x.options());

    def forward(self, x: torch.Tensor) -> torch.Tensor:
