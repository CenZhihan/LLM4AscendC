kernel_src="""
#include "kernel_operator.h"

using DTYPE_A = bool;
using DTYPE_B = bool;
using DTYPE_OUT = bool;
using U16 = uint16_t;

class KernelLogicAndBroadcast {
public:
    __aicore__ inline KernelLogicAndBroadcast() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR out,
                                uint32_t n, uint32_t c, uint32_t w, uint32_t rowsPerTile)
    {
        this->N = n;
        this->C = c;
        this->W = w;
        this->rowsPerTile = rowsPerTile;

        uint32_t totalRows = N * C;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = totalRows / blockNum;
        this->rowStart = blockIdx * this->rowsPerBlock;

        aGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_A*>(a) + static_cast<uint64_t>(rowStart) * W,
                            static_cast<uint64_t>(rowsPerBlock) * W);
        outGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_OUT*>(out) + static_cast<uint64_t>(rowStart) * W,
                              static_cast<uint64_t>(rowsPerBlock) * W);

        // b的有效长度为C(形状[1,C,1])
        bGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_B*>(b), C);

        // 队列与缓冲初始化: 单行处理
        pipe.InitBuffer(inQueueA, 1, W * sizeof(DTYPE_A));
        pipe.InitBuffer(outQueueOut, 1, W * sizeof(DTYPE_OUT));
        pipe.InitBuffer(tmpQue, 2, W * sizeof(U16)); // 临时缓冲用于u16中间结果和广播行
    }

    __aicore__ inline void Process()
    {
        uint32_t tiles = (rowsPerBlock + rowsPerTile - 1) / rowsPerTile;
        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t tileRowStart = t * rowsPerTile;
            uint32_t rowsThisTile = AscendC::Min(rowsPerTile, rowsPerBlock - tileRowStart);
            for (uint32_t r = 0; r < rowsThisTile; ++r) {
                uint32_t globalRow = rowStart + tileRowStart + r;
                RowCompute(globalRow);
            }
        }
    }

private:
    __aicore__ inline void RowCompute(uint32_t globalRow)
    {
        // 计算c索引：按(N,C,W)展开，行序是c最快
        uint32_t cIdx = globalRow % C;

        // 1) 读入一整行a (bool)
        AscendC::LocalTensor<DTYPE_A> aLocal = inQueueA.AllocTensor<DTYPE_A>();
        AscendC::DataCopy(aLocal, aGm[(globalRow - rowStart) * W], W);
        inQueueA.EnQue(aLocal);

        // 2) 从GM取b[cIdx]标量(bool)
        AscendC::LocalTensor<DTYPE_B> bScalarLocal = tmpQue.AllocTensor<DTYPE_B>();
        AscendC::DataCopy(bScalarLocal, bGm[cIdx], 1);
        bool bVal = bScalarLocal.GetValue(0);
        tmpQue.FreeTensor(bScalarLocal);
        U16 bU16Val = bVal ? static_cast<U16>(1) : static_cast<U16>(0);

        // 3) DeQue a，转换为u16
        aLocal = inQueueA.DeQue<DTYPE_A>();
        AscendC::LocalTensor<U16> aU16Local = tmpQue.AllocTensor<U16>();
        AscendC::Cast(aU16Local, aLocal, W);

        // 4) 构造广播行bRow(u16)
        AscendC::LocalTensor<U16> bU16Row = tmpQue.AllocTensor<U16>();
        AscendC::Duplicate(bU16Row, bU16Val, W);

        // 5) 计算按位与: zU16 = aU16 & bU16Row
        AscendC::LocalTensor<U16> zU16Local = tmpQue.AllocTensor<U16>();
        AscendC::And(zU16Local, aU16Local, bU16Row, W);

        // 6) 转回bool并写回
        AscendC::LocalTensor<DTYPE_OUT> zLocal = outQueueOut.AllocTensor<DTYPE_OUT>();
        AscendC::Cast(zLocal, zU16Local, W);
        outQueueOut.EnQue<DTYPE_OUT>(zLocal);

        // 7) 释放临时
        tmpQue.FreeTensor(aU16Local);
        tmpQue.FreeTensor(bU16Row);
        tmpQue.FreeTensor(zU16Local);
        inQueueA.FreeTensor(aLocal);

        // 8) 拷出
        zLocal = outQueueOut.DeQue<DTYPE_OUT>();
        AscendC::DataCopy(outGm[(globalRow - rowStart) * W], zLocal, W);
        outQueueOut.FreeTensor(zLocal);
    }

private:
    uint32_t N, C, W;
    uint32_t rowsPerTile;
    uint32_t rowsPerBlock;
    uint32_t rowStart;

    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueA;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueOut;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQue;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_OUT> outGm;
};

extern "C" __global__ __aicore__ void logic_and_broadcast_custom(GM_ADDR a, GM_ADDR b, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogicAndBroadcast op;
    op.Init(a, b, out,
            tiling_data.n(),
            tiling_data.c(),
            tiling_data.w(),
            tiling_data.rowsPerTile());
    op.Process();
}
