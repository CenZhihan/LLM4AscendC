kernel_src="""
#include "kernel_operator.h"
#include <cmath>

typedef float DTYPE_X;
typedef float DTYPE_Y;
typedef float DTYPE_Z;

class KernelCrossAttention {
public:
    __aicore__ inline KernelCrossAttention() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t batch, uint32_t m, uint32_t k, uint32_t n)
    {
        xGm = reinterpret_cast<__gm__ DTYPE_X*>(x);
        yGm = reinterpret_cast<__gm__ DTYPE_Y*>(y);
        zGm = reinterpret_cast<__gm__ DTYPE_Z*>(z);
        this->batch = batch;
        this->M = m;
        this->K = k;
        this->N = n;
        this->invSqrtK = 1.0f / sqrtf(static_cast<float>(K));
    }

    __aicore__ inline void Process()
    {
        // Split work by batch among blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();

        uint32_t batchPerBlock = (batch + blocks - 1) / blocks;
        uint32_t bStart = bid * batchPerBlock;
        uint32_t bEnd = AscendC::Min(batch, (bid + 1) * batchPerBlock);

        for (uint32_t b = bStart; b < bEnd; ++b) {
            __gm__ DTYPE_X* qPtr = xGm + static_cast<uint64_t>(b) * M * K;       // [M, K]
            __gm__ DTYPE_Y* kvPtr = yGm + static_cast<uint64_t>(b) * N * K;      // [N, K]  (K and V share same tensor)
            __gm__ DTYPE_Z* outPtr = zGm + static_cast<uint64_t>(b) * M * K;     // [M, K]

            for (uint32_t qi = 0; qi < M; ++qi) {
                // 1) Find max score for numerical stability
                float maxScore = -INFINITY;
                for (uint32_t j = 0; j < N; ++j) {
                    float dot = 0.0f;
                    for (uint32_t kk = 0; kk < K; ++kk) {
                        float qv = static_cast<float>(qPtr[qi * K + kk]);
                        float kv = static_cast<float>(kvPtr[j * K + kk]); // K row
                        dot += qv * kv;
                    }
                    float score = dot * invSqrtK;
                    if (score > maxScore) maxScore = score;
                }

                // 2) Compute sum of exp(score - max)
                float sumExp = 0.0f;
                for (uint32_t j = 0; j < N; ++j) {
                    float dot = 0.0f;
                    for (uint32_t kk = 0; kk < K; ++kk) {
                        float qv = static_cast<float>(qPtr[qi * K + kk]);
                        float kv = static_cast<float>(kvPtr[j * K + kk]);
                        dot += qv * kv;
                    }
                    float score = dot * invSqrtK;
                    sumExp += expf(score - maxScore);
                }
                float norm = (sumExp > 0.0f) ? (1.0f / sumExp) : 0.0f;

                // 3) Compute output vector: out[qi, :] = sum_j softmax(score)[j] * V[j, :]
                for (uint32_t kk = 0; kk < K; ++kk) {
                    outPtr[qi * K + kk] = 0.0f;
                }
                for (uint32_t j = 0; j < N; ++j) {
                    float dot = 0.0f;
                    for (uint32_t kk = 0; kk < K; ++kk) {
                        float qv = static_cast<float>(qPtr[qi * K + kk]);
                        float kv = static_cast<float>(kvPtr[j * K + kk]);
                        dot += qv * kv;
                    }
                    float score = dot * invSqrtK;
                    float att = expf(score - maxScore) * norm;
                    for (uint32_t kk = 0; kk < K; ++kk) {
                        // V shares storage with K here: take V[j, kk] as kvPtr[j*K + kk]
                        outPtr[qi * K + kk] += static_cast<DTYPE_Z>(att * static_cast<float>(kvPtr[j * K + kk]));
                    }
                }
            }
        }
    }

private:
    __gm__ DTYPE_X* xGm;
    __gm__ DTYPE_Y* yGm;
    __gm__ DTYPE_Z* zGm;
    uint32_t batch;
    uint32_t M;
    uint32_t K;
    uint32_t N;
    float invSqrtK;
};

extern "C" __global__ __aicore__ void cross_attention_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCrossAttention op;
    op.Init(x, y, z, tiling_data.batch, tiling_data.m, tiling_data.k, tiling_data.n);
    op.Process();
}
