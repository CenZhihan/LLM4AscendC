kernel_src="""
#include "kernel_operator.h"

// 数据类型设定
using DTYPE = float;
constexpr int32_t INOUT_BUFFER_NUM = 1;

class KernelSoftmaxSigmoid {
public:
    __aicore__ inline KernelSoftmaxSigmoid() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalRows, uint32_t width)
    {
        this->totalRows = totalRows;
        this->width = width;

        // 基于行并行的 block 划分
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t baseRows = totalRows / blockNum;
        uint32_t tail = totalRows % blockNum;

        this->myRows = baseRows + (blockIdx < tail ? 1 : 0);
        this->rowOffset = baseRows * blockIdx + (blockIdx < tail ? blockIdx : tail);

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + (uint64_t)rowOffset * width, (uint64_t)myRows * width);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + (uint64_t)rowOffset * width, (uint64_t)myRows * width);

        // 每次处理 1 行（长度为 width）
        pipe.InitBuffer(inQueueX, INOUT_BUFFER_NUM, width * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, INOUT_BUFFER_NUM, width * sizeof(DTYPE));

        // sum/max 的中间结果 buffer（最后轴定长为 32 字节 => 32/sizeof(DTYPE)）
        elementNumPerBlk = 32 / sizeof(DTYPE);
        pipe.InitBuffer(sumQueue, INOUT_BUFFER_NUM, elementNumPerBlk * sizeof(DTYPE));
        pipe.InitBuffer(maxQueue, INOUT_BUFFER_NUM, elementNumPerBlk * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < myRows; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progressRow)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[(uint64_t)progressRow * width], width);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progressRow*/)
    {
        // 取入队的输入行
        AscendC::LocalTensor<DTYPE> srcLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> sumLocal = sumQueue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> maxLocal = maxQueue.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> dstLocal = outQueueY.AllocTensor<DTYPE>();

        // SoftMax: 形状 [m=1, n=width]，沿最后一轴（width）归约
        AscendC::SoftMaxShapeInfo shapeInfo = {1, width, 1, width};
        AscendC::SoftMaxTiling smTiling; // 保持默认，内部根据形状检查/计算
        AscendC::SoftMax<DTYPE>(dstLocal, sumLocal, maxLocal, srcLocal, smTiling, shapeInfo);

        // Sigmoid: 对 SoftMax 结果逐元素激活
        AscendC::Sigmoid(dstLocal, dstLocal, width);

        // 出队
        outQueueY.EnQue<DTYPE>(dstLocal);
        inQueueX.FreeTensor(srcLocal);
        sumQueue.FreeTensor(sumLocal);
        maxQueue.FreeTensor(maxLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progressRow)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[(uint64_t)progressRow * width], yLocal, width);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, INOUT_BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, INOUT_BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECIN, INOUT_BUFFER_NUM> sumQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, INOUT_BUFFER_NUM> maxQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t totalRows = 0;
    uint32_t width = 0;
    uint32_t myRows = 0;
    uint32_t rowOffset = 0;
    uint32_t elementNumPerBlk = 0;
};

extern "C" __global__ __aicore__ void conv_transpose3d_softmax_sigmoid_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelSoftmaxSigmoid op;
    op.Init(x, y, tilingData.totalRows, tilingData.width);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义 AscendC 内核：对 ConvTranspose3d 输出执行 Softmax(沿通道维) + Sigmoid
    要求：在调用自定义算子前，将张量转为 channels_last_3d 连续内存，以便在最后一维上做 softmax。
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias=True):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels, out_channels, kernel_size,
            stride=stride, padding=padding, output_padding=output_padding, bias=bias
        )

    def forward(self, x):
        x = self.conv_transpose(x)
        # 转为 channels_last_3d 连续布局，使通道维成为内存最后一维
        x = x.to(memory_format=torch.channels_last_3d).contiguous()
        # 自定义内核在最后一维做 softmax，再逐元素 sigmoid
        x = custom_ops_lib.conv_transpose3d_softmax_sigmoid_custom(x)
        # 返回为标准连续格式（NCDHW）
        x = x.contiguous(memory_format=torch.contiguous_format)
        return x
