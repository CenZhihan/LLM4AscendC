    tiling.set_inputSize(static_cast<uint32_t>(inputSize));
    tiling.set_hiddenSize(static_cast<uint32_t>(hiddenSize));
    tiling.set_segLen(SEG_LEN);
    uint32_t rowsPerBlock = static_cast<uint32_t>(batchSize / BLOCK_DIM);
    if (rowsPerBlock == 0) rowsPerBlock = 1;
    tiling.set_rowsPerBlock(rowsPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0); // x: [batch, input]
    gert::Shape* y_shape = context->GetOutputShape(0);
    if (x_shape == nullptr || y_shape == nullptr) {
        return GRAPH_FAILED;
    }

    std::vector<int64_t> out_dims;
    if (x_shape->GetDimNum() >= 1) {
        out_dims.push_back(x_shape->GetDim(0)); // batch
    } else {
        out_dims.push_back(1);
    }
    out_dims.push_back(1); // keepdim=True -> (batch, 1)

    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelGemmDivideSumScaling {
public:
    __aicore__ inline KernelGemmDivideSumScaling() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR scale,
                                uint32_t batchSize, uint32_t inputSize, uint32_t hiddenSize,
                                uint32_t segLen, uint32_t rowsPerBlock)
    {
        this->batchSize = batchSize;
        this->inputSize = inputSize;
        this->hiddenSize = hiddenSize;
        this->segLen = segLen;
        this->rowsPerBlock = rowsPerBlock;

        uint32_t blockIdx  = AscendC::GetBlockIdx();
        uint32_t blockNum  = AscendC::GetBlockNum();
        this->blockRowStart = blockIdx * rowsPerBlock;
        uint32_t rowsForThisBlock = rowsPerBlock;
        if (blockIdx == blockNum - 1) {
            uint32_t assigned = rowsPerBlock * blockNum;
            if (batchSize > assigned) {
                rowsForThisBlock += (batchSize - assigned);
            }
        }
        this->rowsThisBlock = rowsForThisBlock;

        xGm.SetGlobalBuffer((__gm__ float*)x + this->blockRowStart * this->inputSize, this->rowsThisBlock * this->inputSize);
        wGm.SetGlobalBuffer((__gm__ float*)w, this->hiddenSize * this->inputSize);
        yGm.SetGlobalBuffer((__gm__ float*)y + this->blockRowStart, this->rowsThisBlock);
        sGm.SetGlobalBuffer((__gm__ float*)scale, 1);

        // queues and buffers
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->segLen * sizeof(float));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, this->segLen * sizeof(float));
        pipe.InitBuffer(tmpQueueVec, BUFFER_NUM, this->segLen * sizeof(float));   // vSeg / temp / prod
        pipe.InitBuffer(tmpQueueProd, BUFFER_NUM, this->segLen * sizeof(float));
        pipe.InitBuffer(sumDstQueue, BUFFER_NUM, 32);                             // 1 float aligned
        pipe.InitBuffer(sumWorkQueue, BUFFER_NUM, this->segLen * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, 32);
        pipe.InitBuffer(scaleQueue, BUFFER_NUM, 32);
    }

    __aicore__ inline void Process()
    {
        // Initialize y partials to 0
        float zeroVal = 0.0f;
        AscendC::LocalTensor<float> zeroLocal = outQueueY.AllocTensor<float>();
        AscendC::Duplicate(zeroLocal, zeroVal, 1);
        for (uint32_t r = 0; r < this->rowsThisBlock; ++r) {
            AscendC::DataCopy(yGm[r], zeroLocal, 1);
        }
        outQueueY.FreeTensor(zeroLocal);

        // Load scale factor into local
        AscendC::LocalTensor<float> scaleLocal = scaleQueue.AllocTensor<float>();
        AscendC::DataCopy(scaleLocal, sGm, 1);

        // Main computation:
        // For each input-segment k0..k0+seg-1:
        for (uint32_t k0 = 0; k0 < this->inputSize; k0 += this->segLen) {
            uint32_t len = AscendC::min(this->segLen, this->inputSize - k0);

            // vSeg = sum over hidden rows of weight[:, k0:k0+len]
            AscendC::LocalTensor<float> vSeg = tmpQueueVec.AllocTensor<float>();
            float zero = 0.0f;
            AscendC::Duplicate(vSeg, zero, len);

            for (uint32_t j = 0; j < this->hiddenSize; ++j) {
                AscendC::LocalTensor<float> wSeg = inQueueW.AllocTensor<float>();
                AscendC::DataCopy(wSeg, wGm[j * this->inputSize + k0], len);
                AscendC::Add(vSeg, vSeg, wSeg, len);
                inQueueW.FreeTensor(wSeg);
            }

            // For each row r, compute partial dot with vSeg and accumulate to y[r]
            for (uint32_t r = 0; r < this->rowsThisBlock; ++r) {
                AscendC::LocalTensor<float> xSeg = inQueueX.AllocTensor<float>();
                AscendC::DataCopy(xSeg, xGm[r * this->inputSize + k0], len);

                AscendC::LocalTensor<float> prodSeg = tmpQueueProd.AllocTensor<float>();
                AscendC::Mul(prodSeg, xSeg, vSeg, len);

                AscendC::LocalTensor<float> sumDst = sumDstQueue.AllocTensor<float>();
                AscendC::LocalTensor<float> sumWork = sumWorkQueue.AllocTensor<float>();
                AscendC::ReduceSum(sumDst, prodSeg, sumWork, len);
                float segVal = AscendC::GetAccVal<float>();

                // y[r] += segVal
                AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
                AscendC::DataCopy(yLocal, yGm[r], 1);
                AscendC::LocalTensor<float> segLocal = outQueueY.AllocTensor<float>();
                AscendC::Duplicate(segLocal, segVal, 1);
                AscendC::Add(yLocal, yLocal, segLocal, 1);
                AscendC::DataCopy(yGm[r], yLocal, 1);
                outQueueY.FreeTensor(yLocal);
                outQueueY.FreeTensor(segLocal);

                inQueueX.FreeTensor(xSeg);
                tmpQueueProd.FreeTensor(prodSeg);
                sumDstQueue.FreeTensor(sumDst);
                sumWorkQueue.FreeTensor(sumWork);
            }

            tmpQueueVec.FreeTensor(vSeg);
        }

        // Apply scaling and divide by 2: y = y * scale * 0.5
        AscendC::LocalTensor<float> halfLocal = outQueueY.AllocTensor<float>();
        float halfVal = 0.5f;
        AscendC::Duplicate(halfLocal, halfVal, 1);

        for (uint32_t r = 0; r < this->rowsThisBlock; ++r) {
            AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
            AscendC::DataCopy(yLocal, yGm[r], 1);
            AscendC::Mul(yLocal, yLocal, scaleLocal, 1);
            AscendC::Mul(yLocal, yLocal, halfLocal, 1);
            AscendC::DataCopy(yGm[r], yLocal, 1);
            outQueueY.FreeTensor(yLocal);
        }

        outQueueY.FreeTensor(halfLocal);
        scaleQueue.FreeTensor(scaleLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueW;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueVec;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueProd;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> sumDstQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> sumWorkQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> scaleQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<float> sGm;

    uint32_t batchSize;
    uint32_t inputSize;
    uint32_t hiddenSize;
    uint32_t segLen;
    uint32_t rowsPerBlock;

    uint32_t blockRowStart;
    uint32_t rowsThisBlock;
};

extern "C" __global__ __aicore__ void gemm_divide_sum_scaling_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR scale, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmDivideSumScaling op;
    op.Init(x, weight, y, scale,
            tiling_data.batchSize, tiling_data.inputSize, tiling_data.hiddenSize,
            tiling_data.segLen, tiling_data.rowsPerBlock);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom AscendC op: (x @ weight.T) / 2 -> sum(dim=1, keepdim=True) -> * scaling_factor
    """
    def __init__(self, input_size, hidden_size, scaling_factor):
        super().__init__()
        self.weight = torch.nn.Parameter(torch.randn(hidden_size, input_size))
        self.scaling_factor = float(scaling_factor)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        scale = torch.tensor([self.scaling_factor], dtype=x.dtype, device=x.device)
        return custom_ops_lib.gemm_divide_sum_scaling_custom(x, self.weight, scale)
