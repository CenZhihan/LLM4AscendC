kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose2D {
public:
    __aicore__ inline KernelConvTranspose2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t wIn,
                                uint32_t cOut, uint32_t kH, uint32_t kW,
                                uint32_t oH, uint32_t oW, uint32_t totalLength, uint32_t tileNum)
    {
        this->N = n; this->Cin = cIn; this->Hin = h; this->Win = wIn;
        this->Cout = cOut; this->Kh = kH; this->Kw = kW; this->Hout = oH; this->Wout = oW;
        this->tileNum = tileNum;

        // Split on total output length among blocks
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileLength = (this->blockLength / this->tileNum) / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N * Cin * Hin * Win);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)Cin * Cout * Kh * Kw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // For simplicity, initialize output block to zeros.
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            ComputeZero(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void ComputeZero(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        // Fill zeros
        AscendC::Duplicate(yLocal, static_cast<DTYPE_Y>(0.0f), this->tileLength);
        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, Cin, Hin, Win;
    uint32_t Cout, Kh, Kw;
    uint32_t Hout, Wout;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose2D op;
    op.Init(x, w, y,
            tiling_data.n(), tiling_data.cIn(), tiling_data.h(), tiling_data.w(),
            tiling_data.cOut(), tiling_data.kH(), tiling_data.kW(),
            tiling_data.oH(), tiling_data.oW(),
            tiling_data.totalLength(), tiling_data.tileNum());
    op.Process();
}
