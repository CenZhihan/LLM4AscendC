inline int64_t FloorDiv(int64_t a, int64_t b) {
    return a / b;
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    AveragePooling3dCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    ge::Shape origin = in_shape->GetOriginShape();
    std::vector<int64_t> in_dims = origin.GetDimNum() > 0 ? origin.GetDims() : std::vector<int64_t>{};

    if (in_dims.size() != 5) {
        return ge::GRAPH_FAILED;
    }

    int64_t n = in_dims[0];
    int64_t c = in_dims[1];
    int64_t d = in_dims[2];
    int64_t h = in_dims[3];
    int64_t w = in_dims[4];

    int64_t ksize = 1;
    int64_t stride = 1;
    int64_t padding = 0;

    // Get attributes
    (void)context->GetAttrInt("kernel_size", ksize);
    (void)context->GetAttrInt("stride", stride);
    (void)context->GetAttrInt("padding", padding);

    if (stride <= 0) {
        stride = ksize; // default as PyTorch when stride=None
    }

    // Output dims (count_include_pad = true behavior: denominator is ksize^3)
    int64_t od = FloorDiv((d + 2 * padding - ksize), stride) + 1;
    int64_t oh = FloorDiv((h + 2 * padding - ksize), stride) + 1;
    int64_t ow = FloorDiv((w + 2 * padding - ksize), stride) + 1;

    if (od < 0) od = 0;
    if (oh < 0) oh = 0;
    if (ow < 0) ow = 0;

    uint64_t totalLength = static_cast<uint64_t>(n) * static_cast<uint64_t>(c) * static_cast<uint64_t>(od) *
                           static_cast<uint64_t>(oh) * static_cast<uint64_t>(ow);

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_tileNum(TILE_NUM);

    tiling.set_n(static_cast<uint32_t>(n));
    tiling.set_c(static_cast<uint32_t>(c));
    tiling.set_d(static_cast<uint32_t>(d));
    tiling.set_h(static_cast<uint32_t>(h));
    tiling.set_w(static_cast<uint32_t>(w));

    tiling.set_od(static_cast<uint32_t>(od));
    tiling.set_oh(static_cast<uint32_t>(oh));
    tiling.set_ow(static_cast<uint32_t>(ow));

    tiling.set_kernelSize(static_cast<uint32_t>(ksize));
    tiling.set_stride(static_cast<uint32_t>(stride));
    tiling.set_padding(static_cast<uint32_t>(padding));
    tiling.set_windowVol(static_cast<uint32_t>(ksize * ksize * ksize));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // no workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    context->SetTilingKey(1);
    return ge::GRAPH_SUCCESS;
}

    const gert::Shape* in_shape = context->GetInputShape(0);

    std::vector<int64_t> in_dims = origin.GetDims();

    int64_t c = in_dims[1];
    int64_t d = in_dims[2];
    int64_t h = in_dims[3];
    int64_t w = in_dims[4];

    int64_t stride = 1;
    int64_t padding = 0;
    (void)context->GetAttrInt("kernel_size", ksize);
    (void)context->GetAttrInt("stride", stride);
    (void)context->GetAttrInt("padding", padding);

kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1;

class KernelAvgPool3d {
public:
    __aicore__ inline KernelAvgPool3d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength_,
                                uint32_t tileNum_,
                                uint32_t n_, uint32_t c_,
                                uint32_t d_, uint32_t h_, uint32_t w_,
                                uint32_t od_, uint32_t oh_, uint32_t ow_,
                                uint32_t ksize_, uint32_t stride_, uint32_t padding_, uint32_t windowVol_)
    {
        totalLength = totalLength_;
        tileNum = tileNum_;

        n = n_; c = c_; d = d_; h = h_; w = w_;
        od = od_; oh = oh_; ow = ow_;
        ksize = ksize_; stride = stride_; padding = padding_; windowVol = windowVol_;

        blockNum = AscendC::GetBlockNum();
        blockIdx = AscendC::GetBlockIdx();

        // balanced range
        uint32_t base = totalLength / blockNum;
        uint32_t rem = totalLength % blockNum;
        blockStart = blockIdx * base + (blockIdx < rem ? blockIdx : rem);
        uint32_t blkLen = base + (blockIdx < rem ? 1 : 0);
        blockEnd = blockStart + blkLen;

        // full input range for reading
        uint32_t inputLength = n * c * d * h * w;
        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, inputLength);
        // output range only for this block
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + blockStart, blkLen);

        // queues for 1-element operations
        pipe.InitBuffer(inQueueVal, BUFFER_NUM, 1 * sizeof(DTYPE));
        pipe.InitBuffer(outQueueAcc, BUFFER_NUM, 1 * sizeof(DTYPE));
        pipe.InitBuffer(outQueueTmp, BUFFER_NUM, 1 * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t idx = blockStart; idx < blockEnd; ++idx) {
            ComputeOne(idx - blockStart, idx);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t localOutIdx, uint32_t globalOutIdx)
    {
        // map linear index to (n,c,od,oh,ow)
        uint32_t owStride = 1;
        uint32_t ohStride = ow;
        uint32_t odStride = oh * ow;
        uint32_t cStride  = od * oh * ow;
        uint32_t nStride  = c * od * oh * ow;

        uint32_t tmp = globalOutIdx;
        uint32_t nIdx = tmp / nStride; tmp -= nIdx * nStride;
        uint32_t cIdx = tmp / cStride; tmp -= cIdx * cStride;
        uint32_t odIdx = tmp / odStride; tmp -= odIdx * odStride;
        uint32_t ohIdx = tmp / ohStride; tmp -= ohIdx * ohStride;
        uint32_t owIdx = tmp / owStride;

        // accumulation buffer
        AscendC::LocalTensor<DTYPE> acc = outQueueAcc.AllocTensor<DTYPE>();
        bool hasInit = false;

        // iterate over window
        for (uint32_t kd = 0; kd < ksize; ++kd) {
            int32_t id = static_cast<int32_t>(odIdx) * static_cast<int32_t>(stride) - static_cast<int32_t>(padding) + static_cast<int32_t>(kd);
            if (id < 0 || id >= static_cast<int32_t>(d)) continue;
            for (uint32_t kh = 0; kh < ksize; ++kh) {
                int32_t ih = static_cast<int32_t>(ohIdx) * static_cast<int32_t>(stride) - static_cast<int32_t>(padding) + static_cast<int32_t>(kh);
                if (ih < 0 || ih >= static_cast<int32_t>(h)) continue;
                for (uint32_t kw = 0; kw < ksize; ++kw) {
                    int32_t iw = static_cast<int32_t>(owIdx) * static_cast<int32_t>(stride) - static_cast<int32_t>(padding) + static_cast<int32_t>(kw);
                    if (iw < 0 || iw >= static_cast<int32_t>(w)) continue;

                    uint32_t inLinear = (((nIdx * c + cIdx) * d + static_cast<uint32_t>(id)) * h + static_cast<uint32_t>(ih)) * w + static_cast<uint32_t>(iw);

                    AscendC::LocalTensor<DTYPE> val = inQueueVal.AllocTensor<DTYPE>();
                    AscendC::DataCopy(val, xGm[inLinear], 1);

                    if (!hasInit) {
                        AscendC::DataCopy(acc, val, 1);
                        hasInit = true;
                    } else {
                        AscendC::Add(acc, acc, val, 1);
                    }

                    inQueueVal.FreeTensor(val);
                }
            }
        }

        // if no valid elements (shouldn't happen in typical padding settings), set 0
        if (!hasInit) {
            // set acc to 0 by scaling an uninitialized value to 0 using Axpy with alpha=0
            AscendC::LocalTensor<DTYPE> tmpLocal = outQueueTmp.AllocTensor<DTYPE>();
            AscendC::Axpy(tmpLocal, acc, (DTYPE)0.0f, 1);
            AscendC::DataCopy(acc, tmpLocal, 1);
            outQueueTmp.FreeTensor(tmpLocal);
        }

        // scale by window volume (count_include_pad = true)
        DTYPE scale = (DTYPE)(1.0f / static_cast<float>(windowVol));
        AscendC::LocalTensor<DTYPE> yLocal = outQueueTmp.AllocTensor<DTYPE>();
        AscendC::Axpy(yLocal, acc, scale, 1);

        // write to output
        AscendC::DataCopy(yGm[localOutIdx], yLocal, 1);

        outQueueTmp.FreeTensor(yLocal);
        outQueueAcc.FreeTensor(acc);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> inQueueVal;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueAcc;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueTmp;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t totalLength, tileNum;
    uint32_t n, c, d, h, w;
    uint32_t od, oh, ow;
    uint32_t ksize, stride, padding, windowVol;

    uint32_t blockNum, blockIdx;
    uint32_t blockStart, blockEnd;
};

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAvgPool3d op;
    op.Init(
        x, y,
        tiling_data.totalLength(),
        tiling_data.tileNum(),
        tiling_data.n(), tiling_data.c(),
        tiling_data.d(), tiling_data.h(), tiling_data.w(),
        tiling_data.od(), tiling_data.oh(), tiling_data.ow(),
        tiling_data.kernelSize(), tiling_data.stride(), tiling_data.padding(),
        tiling_data.windowVol()
    );
    op.Process();
}

static inline std::vector<int64_t> infer_avgpool3d_out_shape(const at::Tensor& x,
                                                             int64_t kernel_size,
                                                             int64_t stride,
                                                             int64_t padding) {
    auto sizes = x.sizes();
    int64_t n = sizes[0];
    int64_t c = sizes[1];
    int64_t d = sizes[2];
    int64_t h = sizes[3];
    int64_t w = sizes[4];

    int64_t oh = (h + 2 * padding - kernel_size) / stride + 1;

    if (od < 0) od = 0;
    if (oh < 0) oh = 0;
    if (ow < 0) ow = 0;
    return {n, c, od, oh, ow};
}

                                              int64_t kernel_size,
                                              int64_t stride,
