#include "index_select_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    IndexSelectCustomTilingData tiling;

    uint32_t rows = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(0));
    uint32_t srcCols = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));
    uint32_t dstCols = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());

    context->SetBlockDim(BLOCK_DIM);

    // choose a reasonable tile size on dstCols
    uint32_t baseTile = (dstCols + TILE_NUM - 1) / TILE_NUM;
    uint32_t maxTile = 1024U;
    uint32_t tileK = baseTile < maxTile ? baseTile : maxTile;
    if (tileK == 0) tileK = 1;

    tiling.set_rows(rows);
    tiling.set_srcCols(srcCols);
    tiling.set_dstCols(dstCols);
    tiling.set_tileK(tileK);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* idx_shape = context->GetInputShape(1);

    uint64_t rows = x_shape->GetOriginShape().GetDim(0);
    uint64_t k = idx_shape->GetOriginShape().GetShapeSize();

    gert::Shape out;
    out.SetDimNum(2);
    out.SetDim(0, rows);
    out.SetDim(1, k);

    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = out;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Y = float;
using DTYPE_INDEX = int32_t;

class KernelIndexSelect {
public:
    __aicore__ inline KernelIndexSelect() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, GM_ADDR y,
                                uint32_t rows, uint32_t srcCols, uint32_t dstCols, uint32_t tileK)
    {
        this->rows = rows;
        this->srcCols = srcCols;
        this->dstCols = dstCols;
        this->tileK = tileK;

        uint32_t blockId = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t baseRows = rows / blockNum;
        uint32_t remain = rows % blockNum;
        this->rowStart = blockId * baseRows + (blockId < remain ? blockId : remain);
        this->myRows = baseRows + (blockId < remain ? 1 : 0);

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_X *>(x) + static_cast<uint64_t>(rowStart) * srcCols,
                            static_cast<uint64_t>(myRows) * srcCols);
        idxGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_INDEX *>(indices), dstCols);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_Y *>(y) + static_cast<uint64_t>(rowStart) * dstCols,
                            static_cast<uint64_t>(myRows) * dstCols);

        pipe.InitBuffer(idxBuf, tileK * sizeof(DTYPE_INDEX));
        pipe.InitBuffer(yTmpBuf, tileK * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t k0 = 0; k0 < dstCols; k0 += tileK) {
            uint32_t curTileK = (k0 + tileK <= dstCols) ? tileK : (dstCols - k0);

            AscendC::LocalTensor<DTYPE_INDEX> idxLocal = idxBuf.Get<DTYPE_INDEX>(curTileK);
            AscendC::DataCopy(idxLocal, idxGm[k0], curTileK);

            for (uint32_t r = 0; r < myRows; ++r) {
                AscendC::LocalTensor<DTYPE_Y> yLocal = yTmpBuf.Get<DTYPE_Y>(curTileK);

                // For each index in the tile, gather one element from the row.
                for (uint32_t t = 0; t < curTileK; ++t) {
                    DTYPE_INDEX col = idxLocal.GetValue(t);
                    // bounds check is omitted; assume inputs are valid
                    AscendC::DataCopy(yLocal[t], xGm[static_cast<uint64_t>(r) * srcCols + static_cast<uint32_t>(col)], 1);
                }

                AscendC::DataCopy(yGm[static_cast<uint64_t>(r) * dstCols + k0], yLocal, curTileK);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::TPosition::VECCALC> idxBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> yTmpBuf;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_INDEX> idxGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows;
    uint32_t srcCols;
    uint32_t dstCols;
    uint32_t tileK;

    uint32_t rowStart;
    uint32_t myRows;
};

extern "C" __global__ __aicore__ void index_select_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    KernelIndexSelect op;
    op.Init(x, indices, y,
            tilingData.rows(),
            tilingData.srcCols(),
            tilingData.dstCols(),
            tilingData.tileK());
    op.Process();
}

at::Tensor index_select_custom_impl_npu(const at::Tensor& x, const at::Tensor& indices) {
    // Ensure indices are int32 on NPU

        indices_i32 = indices.to(at::kInt);
    }
    at::Tensor result = at::empty({x.size(0), indices.size(0)}, x.options());
    EXEC_NPU_CMD(aclnnIndexSelectCustom, x, indices_i32, result);
    return result;
