kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的Tensor数量

class KernelMatmulSubMulRelu {
public:
    __aicore__ inline KernelMatmulSubMulRelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR sub, GM_ADDR mul, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ float *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        subGm.SetGlobalBuffer((__gm__ float *)sub + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mulGm.SetGlobalBuffer((__gm__ float *)mul + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ float *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueSub, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueMul, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<float> subLocal = inQueueSub.AllocTensor<float>();
        AscendC::LocalTensor<float> mulLocal = inQueueMul.AllocTensor<float>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(subLocal, subGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(mulLocal, mulGm[progress * this->tileLength], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueSub.EnQue(subLocal);
        inQueueMul.EnQue(mulLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> subLocal = inQueueSub.DeQue<float>();
        AscendC::LocalTensor<float> mulLocal = inQueueMul.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

        // z = relu( (x - sub) * mul )
        AscendC::Sub(zLocal, xLocal, subLocal, this->tileLength);
        AscendC::Mul(zLocal, zLocal, mulLocal, this->tileLength);
        AscendC::Relu(zLocal, zLocal, this->tileLength);

        outQueueZ.EnQue<float>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueSub.FreeTensor(subLocal);
        inQueueMul.FreeTensor(mulLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueSub, inQueueMul;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> subGm;
    AscendC::GlobalTensor<float> mulGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void matmul_subtract_multiply_relu_custom(GM_ADDR x, GM_ADDR sub, GM_ADDR mul, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSubMulRelu op;
    op.Init(x, sub, mul, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    计算流程：
    y = Linear(x)
    z = ReLU( (y - subtract_value) * multiply_value )
    其中减法/乘法/激活由自定义AscendC算子完成。
    """
    def __init__(self, in_features, out_features, subtract_value, multiply_value) -> None:
        super().__init__()
        self.linear = torch.nn.Linear(in_features, out_features)
        self.subtract_value = float(subtract_value)
        self.multiply_value = float(multiply_value)

    def forward(self, x):
        y = self.linear(x)
        # 为了与自定义算子接口对齐，将标量广播成与y同形状的张量，避免在Kernel侧处理标量广播
        sub = torch.full_like(y, self.subtract_value)
        mul = torch.full_like(y, self.multiply_value)
        return custom_ops_lib.matmul_subtract_multiply_relu_custom(y, sub, mul)
'''
