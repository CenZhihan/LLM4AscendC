kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Y = float;

constexpr int32_t IN_BUFFER_NUM = 1;
constexpr int32_t OUT_BUFFER_NUM = 1;

class KernelSubtractSpatialMean3D {
public:
    __aicore__ inline KernelSubtractSpatialMean3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t n, uint32_t c,
                                uint32_t d, uint32_t h, uint32_t w,
                                uint32_t groupSize,
                                uint32_t tileNum,
                                uint32_t tileLen)
    {
        this->n = n;
        this->c = c;
        this->d = d;
        this->h = h;
        this->w = w;
        this->groupSize = groupSize;
        this->groups = n * c;
        this->tileNum = tileNum;
        this->tileLen = tileLen;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blk = AscendC::GetBlockIdx();
        uint32_t begin = (uint64_t)this->groups * blk / blockNum;
        uint32_t end = (uint64_t)this->groups * (blk + 1) / blockNum;
        this->groupBegin = begin;
        this->groupEnd = end;
        this->groupsThisBlock = (end > begin) ? (end - begin) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)groupBegin * groupSize,
                            (uint64_t)groupsThisBlock * groupSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + (uint64_t)groupBegin * groupSize,
                            (uint64_t)groupsThisBlock * groupSize);

        // 缓冲区大小按tileLen配置
        uint32_t bufBytes = this->tileLen * sizeof(DTYPE_X);
        pipe.InitBuffer(inQ, IN_BUFFER_NUM, bufBytes);
        pipe.InitBuffer(outQ, OUT_BUFFER_NUM, bufBytes);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t g = 0; g < groupsThisBlock; ++g) {
            // 第一遍：计算该(N,C)组的总和
            float sum = 0.0f;
            uint32_t tiles = (groupSize + tileLen - 1) / tileLen;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = g * groupSize + t * tileLen;
                uint32_t remain = groupSize > t * tileLen ? (groupSize - t * tileLen) : 0;
                uint32_t copyLen = remain > tileLen ? tileLen : remain;
                if (copyLen == 0) break;

                AscendC::LocalTensor<DTYPE_X> xLocal = inQ.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[offset], copyLen);
                inQ.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_X> xBuf = inQ.DeQue<DTYPE_X>();
                for (uint32_t i = 0; i < copyLen; ++i) {
                    sum += static_cast<float>(xBuf.GetValue(i));
                }
                inQ.FreeTensor(xBuf);
            }
            float mean = sum / static_cast<float>(groupSize);

            // 第二遍：减去均值并写回
            for (uint32_t t = 0; t < (groupSize + tileLen - 1) / tileLen; ++t) {
                uint32_t offset = g * groupSize + t * tileLen;
                uint32_t remain = groupSize > t * tileLen ? (groupSize - t * tileLen) : 0;
                uint32_t copyLen = remain > tileLen ? tileLen : remain;
                if (copyLen == 0) break;

                AscendC::LocalTensor<DTYPE_X> xLocal = inQ.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[offset], copyLen);
                inQ.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_X> xBuf = inQ.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQ.AllocTensor<DTYPE_Y>();
                for (uint32_t i = 0; i < copyLen; ++i) {
                    float v = static_cast<float>(xBuf.GetValue(i));
                    yLocal.SetValue(i, static_cast<DTYPE_Y>(v - mean));
                }
                outQ.EnQue(yLocal);
                inQ.FreeTensor(xBuf);

                AscendC::LocalTensor<DTYPE_Y> yBuf = outQ.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[offset], yBuf, copyLen);
                outQ.FreeTensor(yBuf);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, IN_BUFFER_NUM> inQ;
    AscendC::TQue<AscendC::QuePosition::VECOUT, OUT_BUFFER_NUM> outQ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, d, h, w;
    uint32_t groupSize;
    uint32_t groups;
    uint32_t tileNum;
    uint32_t tileLen;

    uint32_t groupBegin;
    uint32_t groupEnd;
    uint32_t groupsThisBlock;
};

extern "C" __global__ __aicore__ void conv_transpose3d_batch_norm_subtract_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSubtractSpatialMean3D op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.groupSize,
            tiling_data.tileNum,
            tiling_data.tileLen);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    ConvTranspose3d + BatchNorm3d 后，使用自定义AscendC算子进行空间维度均值相减
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, bias=True):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, bias=bias)
        self.batch_norm = torch.nn.BatchNorm3d(out_channels)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.batch_norm(x)
        # 自定义AscendC核完成沿(D,H,W)的均值计算并相减
        x = custom_ops_lib.conv_transpose3d_batch_norm_subtract_custom(x)
        return x
