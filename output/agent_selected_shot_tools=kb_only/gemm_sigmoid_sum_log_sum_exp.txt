kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

// Fused kernel: y = logsumexp( (Sigmoid(x)) @ W2^T + b2, dim=1 )
// Inputs:
//   x  : [B, H]
//   w2 : [O, H]
//   b2 : [O]
// Output:
//   y  : [B]
constexpr int32_t IN_Q_NUM = 1;
constexpr int32_t OUT_Q_NUM = 1;

struct TilingData {
    uint32_t batch;
    uint32_t hidden;
    uint32_t outdim;
    uint32_t rowsPerBlock;
    uint32_t tileK;
    uint32_t tileO;
};

class KernelGemmSigmoidSumLogSumExp {
public:
    __aicore__ inline KernelGemmSigmoidSumLogSumExp() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w2, GM_ADDR b2, GM_ADDR y,
                                const TilingData &t)
    {
        batch_  = t.batch;
        hidden_ = t.hidden;
        out_    = t.outdim;
        rowsPerBlock_ = t.rowsPerBlock;
        tileK_  = t.tileK;
        tileO_  = t.tileO;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // row range per block (ceil split, clamp end)
        startRow_ = blockIdx * rowsPerBlock_;
        endRow_   = startRow_ + rowsPerBlock_;
        if (endRow_ > batch_) endRow_ = batch_;
        if (startRow_ > batch_) startRow_ = batch_;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(x), (size_t)batch_ * hidden_);
        wGm_.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(w2), (size_t)out_ * hidden_);
        bGm_.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(b2), (size_t)out_);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(y), (size_t)batch_);

        // UB buffers
        // one row x: hidden_
        pipe_.InitBuffer(inQX_, IN_Q_NUM, hidden_ * sizeof(DTYPE));
        // sigmoid(x): hidden_
        pipe_.InitBuffer(bufSig_, 1, hidden_ * sizeof(DTYPE));
        // weight row chunk and mul temp
        pipe_.InitBuffer(bufW_, 1, tileK_ * sizeof(DTYPE));
        pipe_.InitBuffer(bufMul_, 1, tileK_ * sizeof(DTYPE));
        // scalar holders
        pipe_.InitBuffer(bufScalar_, 1, sizeof(DTYPE) * 4);
        // bias chunk
        pipe_.InitBuffer(bufBias_, 1, sizeof(DTYPE));
        // not storing full output row; use streaming logsumexp
        pipe_.InitBuffer(outQY_, OUT_Q_NUM, sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = startRow_; r < endRow_; ++r) {
            // Load x row
            AscendC::LocalTensor<DTYPE> xLocal = inQX_.AllocTensor<DTYPE>();
            AscendC::DataCopy(xLocal, xGm_[r * hidden_], hidden_);
            inQX_.EnQue(xLocal);

            // Sigmoid(x)
            AscendC::LocalTensor<DTYPE> xRow = inQX_.DeQue<DTYPE>();
            AscendC::LocalTensor<DTYPE> sigRow = bufSig_.Get<DTYPE>();
            AscendC::Sigmoid(sigRow, xRow, hidden_);
            inQX_.FreeTensor(xRow);

            // Streaming log-sum-exp over outputs after matvec
            // m = -inf, s = 0 initially; we do first element specially
            bool first = true;
            AscendC::LocalTensor<DTYPE> tmpScalar0 = bufScalar_.Get<DTYPE>(); // single scalar used variably
            AscendC::LocalTensor<DTYPE> tmpScalar1 = bufScalar_.Get<DTYPE>(); // alias ok, will use length 1 semantics

            // Hold m and s as 1-length tensors
            AscendC::LocalTensor<DTYPE> mLoc = bufScalar_.Get<DTYPE>();
            AscendC::LocalTensor<DTYPE> sLoc = bufScalar_.Get<DTYPE>();

            for (uint32_t j = 0; j < out_; ++j) {
                // Compute v_j = dot(w2[j,:], sigRow) + b2[j]
                // Accumulate over hidden in chunks of tileK_
                // sum_j as scalar
                AscendC::LocalTensor<DTYPE> sumLoc = bufScalar_.Get<DTYPE>();
                // initialize sum to 0
                // Use Duplicate to fill zeros to one element
                AscendC::Duplicate(sumLoc, static_cast<DTYPE>(0), 1);

                uint32_t kBase = 0;
                while (kBase < hidden_) {
                    uint32_t kChunk = tileK_;
                    if (kBase + kChunk > hidden_) kChunk = hidden_ - kBase;

                    // load weight chunk w2[j, kBase:kBase+kChunk]
                    AscendC::LocalTensor<DTYPE> wChunk = bufW_.Get<DTYPE>();
                    AscendC::DataCopy(wChunk, wGm_[j * hidden_ + kBase], kChunk);

                    // element-wise mul temp = wChunk * sigRow[kBase:...]
                    AscendC::LocalTensor<DTYPE> mulChunk = bufMul_.Get<DTYPE>();
                    AscendC::LocalTensor<DTYPE> sigSub = sigRow[kBase];
                    AscendC::Mul(mulChunk, wChunk, sigSub, kChunk);

                    // Reduce sum of mulChunk into tmpScalar0
                    AscendC::ReduceSum(tmpScalar0, mulChunk, kChunk);
                    // sumLoc += tmpScalar0
                    AscendC::Add(sumLoc, sumLoc, tmpScalar0, 1);

                    kBase += kChunk;
                }

                // add bias
                AscendC::LocalTensor<DTYPE> bLoc = bufBias_.Get<DTYPE>();
                AscendC::DataCopy(bLoc, bGm_[j], 1);
                AscendC::Add(sumLoc, sumLoc, bLoc, 1); // now sumLoc holds v_j

                if (first) {
                    // m = v_j, s = 1
                    AscendC::DataCopy(mLoc, sumLoc, 1);
                    AscendC::Duplicate(sLoc, static_cast<DTYPE>(1), 1);
                    first = false;
                } else {
                    // new_m = max(m, v)
                    AscendC::LocalTensor<DTYPE> newM = bufScalar_.Get<DTYPE>();
                    AscendC::Max(newM, mLoc, sumLoc, 1);

                    // s = s * exp(m - new_m) + exp(v - new_m)
                    AscendC::LocalTensor<DTYPE> t1 = bufScalar_.Get<DTYPE>();
                    AscendC::LocalTensor<DTYPE> t2 = bufScalar_.Get<DTYPE>();
                    AscendC::Sub(t1, mLoc, newM, 1);
                    AscendC::Sub(t2, sumLoc, newM, 1);

                    AscendC::Exp(t1, t1, 1);
                    AscendC::Exp(t2, t2, 1);

                    // s * t1
                    AscendC::Mul(sLoc, sLoc, t1, 1);
                    // s = s + t2
                    AscendC::Add(sLoc, sLoc, t2, 1);

                    // m = new_m
                    AscendC::DataCopy(mLoc, newM, 1);
                }
            }

            // lse = m + ln(s)
            AscendC::LocalTensor<DTYPE> lseLoc = outQY_.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> lnS = bufScalar_.Get<DTYPE>();
            AscendC::Ln(lnS, sLoc, 1);
            AscendC::Add(lseLoc, mLoc, lnS, 1);

            // write result
            AscendC::DataCopy(yGm_[r], lseLoc, 1);

            outQY_.EnQue<DTYPE>(lseLoc);
            AscendC::LocalTensor<DTYPE> yTmp = outQY_.DeQue<DTYPE>();
            outQY_.FreeTensor(yTmp);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_Q_NUM> inQX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_Q_NUM> outQY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufSig_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufW_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufMul_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufScalar_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufBias_;

    AscendC::GlobalTensor<DTYPE> xGm_;
    AscendC::GlobalTensor<DTYPE> wGm_;
    AscendC::GlobalTensor<DTYPE> bGm_;
    AscendC::GlobalTensor<DTYPE> yGm_;

    uint32_t batch_;
    uint32_t hidden_;
    uint32_t out_;
    uint32_t rowsPerBlock_;
    uint32_t tileK_;
    uint32_t tileO_;
    uint32_t startRow_;
    uint32_t endRow_;
};

extern "C" __global__ __aicore__ void gemm_sigmoid_sum_log_sum_exp_custom(GM_ADDR x, GM_ADDR w2, GM_ADDR b2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    TilingData t;
    GET_TILING_DATA(t, tiling);
    KernelGemmSigmoidSumLogSumExp op;
    op.Init(x, w2, b2, y, t);
    op.Process();
}
