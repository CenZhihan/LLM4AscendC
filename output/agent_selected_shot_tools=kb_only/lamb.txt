kernel_src="""
#include "kernel_operator.h"

// 简化实现：在核内两遍处理
// 第1遍：计算 r = m / (sqrt(v) + eps)，并归约得到 param 的 L2 范数与 r 的 L2 范数；同时将 r 写入workspace
// 第2遍：根据 trust_ratio = ||param||_2 / (||r||_2 + eps)，执行 param - lr * trust_ratio * r，写回输出

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2; // 每个队列的tensor数量

// 默认超参：如需自定义可扩展为属性/额外输入
__aicore__ inline DTYPE DefaultLR() { return static_cast<DTYPE>(1e-3f); }
__aicore__ inline DTYPE DefaultEPS() { return static_cast<DTYPE>(1e-6f); }

class KernelLambCustom {
public:
    __aicore__ inline KernelLambCustom() {}
    __aicore__ inline void Init(GM_ADDR param, GM_ADDR m, GM_ADDR v, GM_ADDR out,
                                GM_ADDR workspace, uint32_t totalLength, uint32_t tileNum)
    {
        this->totalLength = totalLength;
        this->blockLength = totalLength; // 单block处理全量
        this->tileNum = tileNum > 0 ? tileNum : 1;
        this->tileLength = (this->blockLength + (this->tileNum * BUFFER_NUM - 1)) / (this->tileNum * BUFFER_NUM);

        paramGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(param), this->blockLength);
        mGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(m), this->blockLength);
        vGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(v), this->blockLength);
        outGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(out), this->blockLength);
        rGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(workspace), this->blockLength);

        pipe.InitBuffer(inQueueParam, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(inQueueM, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(inQueueV, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE));   // 临时/复用
        pipe.InitBuffer(outQueueR, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueOut, BUFFER_NUM, this->tileLength * sizeof(DTYPE));

        // 归约标量缓冲，按框架对齐要求分配
        pipe.InitBuffer(reduceBufParam, 1, 32); // 存放一个标量(对齐到32B)
        pipe.InitBuffer(reduceBufR, 1, 32);
    }

    __aicore__ inline void Process()
    {
        // 超参
        DTYPE lr = DefaultLR();
        DTYPE eps = DefaultEPS();

        // 第一遍：计算r与范数
        DTYPE sumParam2 = static_cast<DTYPE>(0.0f);
        DTYPE sumR2 = static_cast<DTYPE>(0.0f);

        int32_t loopCount = this->tileNum * BUFFER_NUM;
        uint32_t processed = 0;
        for (int32_t i = 0; i < loopCount && processed < this->blockLength; i++) {
            uint32_t curLen = GetCurTileLen(i);
            CopyInPass1(i, curLen);
            sumParam2 += ComputePass1(i, curLen, eps);
            CopyOutPass1(i, curLen);
            processed += curLen;
        }

        DTYPE normParam = SqrtScalar(sumParam2);
        DTYPE normR = SqrtScalar(sumR2 = sumR2 + static_cast<DTYPE>(0.0f)); // 保持接口
        // 为了稳健，可以再次使用sumR2值（已在ComputePass1内部累加）

        // 由于sumR2是累加的，我们需要实际赋值。上面写法保留接口一致性。
        normR = SqrtScalar(sumR2);

        DTYPE trust_ratio = normParam / (normR + eps);
        DTYPE scale = lr * trust_ratio;

        // 第二遍：根据r更新param
        processed = 0;
        for (int32_t i = 0; i < loopCount && processed < this->blockLength; i++) {
            uint32_t curLen = GetCurTileLen(i);
            CopyInPass2(i, curLen);
            ComputePass2(i, curLen, scale);
            CopyOutPass2(i, curLen);
            processed += curLen;
        }
    }

private:
    __aicore__ inline uint32_t GetCurTileLen(int32_t progress) const
    {
        uint32_t start = static_cast<uint32_t>(progress) * this->tileLength;
        if (start >= this->blockLength) return 0;
        uint32_t remain = this->blockLength - start;
        return remain < this->tileLength ? remain : this->tileLength;
    }

    // sqrt标量
    __aicore__ inline DTYPE SqrtScalar(DTYPE x) const
    {
        // 简单牛顿迭代近似或依赖设备sqrt，可由升腾内置实现；这里直接调用设备向量sqrt接口的标量近似
        // 实际实现中可使用AscendC的Rsqrt/Sqrt对应标量或向量API。
        // 为示例，简单近似：
        DTYPE r = x;
        if (r <= 0) return 0;
        DTYPE g = r;
        for (int k = 0; k < 6; ++k) {
            g = static_cast<DTYPE>(0.5f) * (g + r / g);
        }
        return g;
    }

    // 第一遍：拷入param/m/v
    __aicore__ inline void CopyInPass1(int32_t progress, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE> paramLocal = inQueueParam.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> mLocal = inQueueM.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> vLocal = inQueueV.AllocTensor<DTYPE>();
        AscendC::DataCopy(paramLocal, paramGm[progress * this->tileLength], curLen);
        AscendC::DataCopy(mLocal, mGm[progress * this->tileLength], curLen);
        AscendC::DataCopy(vLocal, vGm[progress * this->tileLength], curLen);
        inQueueParam.EnQue(paramLocal);
        inQueueM.EnQue(mLocal);
        inQueueV.EnQue(vLocal);
    }

    // 第一遍：计算r，写workspace，并归约param^2和r^2
    __aicore__ inline DTYPE ComputePass1(int32_t progress, uint32_t curLen, DTYPE eps)
    {
        AscendC::LocalTensor<DTYPE> paramLocal = inQueueParam.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> mLocal = inQueueM.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> vLocal = inQueueV.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> rLocal = outQueueR.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> tmpLocal = tmpQueue.AllocTensor<DTYPE>();

        // denom = sqrt(v) + eps
        AscendC::Sqrt(tmpLocal, vLocal, curLen);
        AscendC::Adds(tmpLocal, tmpLocal, eps, curLen);

        // r = m / denom
        AscendC::Div(rLocal, mLocal, tmpLocal, curLen);

        // sumParam2 += sum(param^2)
        AscendC::Mul(tmpLocal, paramLocal, paramLocal, curLen);
        AscendC::LocalTensor<DTYPE> redParam = reduceBufParam.AllocTensor<DTYPE>();
        AscendC::ReduceSum(redParam, tmpLocal, curLen);
        DTYPE sumParam2_tile = redParam.GetValue(0); // 假设LocalTensor支持获取标量

        // sumR2 += sum(r^2)
        AscendC::Mul(tmpLocal, rLocal, rLocal, curLen);
        AscendC::LocalTensor<DTYPE> redR = reduceBufR.AllocTensor<DTYPE>();
        AscendC::ReduceSum(redR, tmpLocal, curLen);
        DTYPE sumR2_tile = redR.GetValue(0);

        // 将r写回workspace
        outQueueR.EnQue<DTYPE>(rLocal);

        // 释放
        inQueueParam.FreeTensor(paramLocal);
        inQueueM.FreeTensor(mLocal);
        inQueueV.FreeTensor(vLocal);
        tmpQueue.FreeTensor(tmpLocal);
        reduceBufParam.FreeTensor(redParam);
        reduceBufR.FreeTensor(redR);

        // 累加到成员变量sumR2（通过返回值和成员累加）
        this->sumR2_acc += sumR2_tile;
        return sumParam2_tile;
    }

    // 第一遍：拷出r到workspace
    __aicore__ inline void CopyOutPass1(int32_t progress, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE> rLocal = outQueueR.DeQue<DTYPE>();
        AscendC::DataCopy(rGm[progress * this->tileLength], rLocal, curLen);
        outQueueR.FreeTensor(rLocal);
    }

    // 第二遍：从workspace读r，并更新param -> out
    __aicore__ inline void CopyInPass2(int32_t progress, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE> paramLocal = inQueueParam.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> rLocal = inQueueM.AllocTensor<DTYPE>(); // 复用队列
        AscendC::DataCopy(paramLocal, paramGm[progress * this->tileLength], curLen);
        AscendC::DataCopy(rLocal, rGm[progress * this->tileLength], curLen);
        inQueueParam.EnQue(paramLocal);
        inQueueM.EnQue(rLocal);
    }

    __aicore__ inline void ComputePass2(int32_t progress, uint32_t curLen, DTYPE scale)
    {
        AscendC::LocalTensor<DTYPE> paramLocal = inQueueParam.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> rLocal = inQueueM.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> outLocal = outQueueOut.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> tmpLocal = tmpQueue.AllocTensor<DTYPE>();

        // tmp = scale * r
        AscendC::Muls(tmpLocal, rLocal, scale, curLen);
        // out = param - tmp
        AscendC::Sub(outLocal, paramLocal, tmpLocal, curLen);

        outQueueOut.EnQue<DTYPE>(outLocal);
        inQueueParam.FreeTensor(paramLocal);
        inQueueM.FreeTensor(rLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOutPass2(int32_t progress, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE> outLocal = outQueueOut.DeQue<DTYPE>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, curLen);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueParam, inQueueM, inQueueV;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueR, outQueueOut;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> reduceBufParam, reduceBufR;

    AscendC::GlobalTensor<DTYPE> paramGm;
    AscendC::GlobalTensor<DTYPE> mGm;
    AscendC::GlobalTensor<DTYPE> vGm;
    AscendC::GlobalTensor<DTYPE> outGm;
    AscendC::GlobalTensor<DTYPE> rGm;

    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    DTYPE sumR2_acc = static_cast<DTYPE>(0.0f);
};

extern "C" __global__ __aicore__ void lamb_custom(GM_ADDR param, GM_ADDR m, GM_ADDR v, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLambCustom op;
    op.Init(param, m, v, out, workspace, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
