static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    KlDivLossCustomTilingData tiling;
    uint32_t sizeofdatatype = 4; // float
    uint32_t totalLengthAligned;
    uint32_t totalLength = context->GetInputShape(0)->GetStorageShape().GetShapeSize();

    // 取batch维度用于batchmean
    uint32_t batchSize = 1;
    {
        auto dims = context->GetInputShape(0)->GetStorageShape().GetDims();
        if (!dims.empty()) {
            batchSize = static_cast<uint32_t>(dims[0]);
        }
    }

    uint32_t ALIGN_NUM = BLOCK_SIZE / sizeofdatatype; // 32B对齐单元中的元素个数
    uint32_t ub_block_num = 1024;  // 每次tile处理的对齐块个数(示例值)
    uint32_t tile_num = 0;

    if (ub_block_num % 2 != 0) {
        ub_block_num = ub_block_num - 1;
    }

    if (totalLength % ALIGN_NUM != 0) {
        totalLengthAligned = ((totalLength + ALIGN_NUM - 1) / ALIGN_NUM) * ALIGN_NUM;
    } else {
        totalLengthAligned = totalLength;
    }

    tiling.set_totalLength(totalLength);

    auto block_dim = 1;
    context->SetBlockDim(block_dim);

    uint32_t blockLength = 0;
    uint32_t tileLength = 0;
    uint32_t lasttileLength = 0;

    blockLength = totalLengthAligned / block_dim;
    tile_num = blockLength / ALIGN_NUM / ub_block_num;

    if ((totalLengthAligned / block_dim / ALIGN_NUM) % ub_block_num == 0 || tile_num == 0) {
        if (tile_num == 0) {
            tile_num = 1;
        }
        if (blockLength < ub_block_num * ALIGN_NUM) {
            tileLength = ((blockLength / ALIGN_NUM) + 1) / 2 * 2 * ALIGN_NUM;
            lasttileLength = tileLength;
        } else {
            tileLength = ub_block_num * ALIGN_NUM;
            lasttileLength = tileLength;
        }
    } else {
        tile_num = tile_num + 1;
        tileLength = ub_block_num * ALIGN_NUM;
        lasttileLength = blockLength - (tile_num - 1) * tileLength;
    }

    tiling.set_blockLength(blockLength);
    tiling.set_tileNum(tile_num);
    tiling.set_tileLength(tileLength);
    tiling.set_lasttileLength(lasttileLength);
    tiling.set_batchSize(batchSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

// 自定义KL散度(batchmean)核：输入p(预测概率)，q(目标概率)
// loss = sum(q * (ln(q) - ln(p))) / batch_size

constexpr int32_t BUFFER_NUM = 1;

class KernelKlDivLoss {
public:
    __aicore__ inline KernelKlDivLoss() {}
    __aicore__ inline void Init(GM_ADDR predict, GM_ADDR label, GM_ADDR y,
                                uint32_t totalLength, uint32_t blockLength,
                                uint32_t tileNum, uint32_t tileLength,
                                uint32_t lasttileLength, uint32_t batchSize) {
        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");

        this->totalLength = static_cast<int32_t>(totalLength);
        this->blockLength = blockLength;
        this->tileNum = tileNum;
        this->tileLength = tileLength / BUFFER_NUM;
        this->lasttileLength = lasttileLength;
        this->batchSize = batchSize;

        pGm.SetGlobalBuffer((__gm__ float*)predict + this->blockLength * AscendC::GetBlockIdx(),
                            this->blockLength);
        qGm.SetGlobalBuffer((__gm__ float*)label + this->blockLength * AscendC::GetBlockIdx(),
                            this->blockLength);
        outGm.SetGlobalBuffer((__gm__ float*)y + this->blockLength * AscendC::GetBlockIdx(), 32);

        this->reduce_num = this->tileNum * BUFFER_NUM;
        uint32_t reduce_align = (this->reduce_num + 31) / 32 * 32;

        // 输入双缓冲(p,q) + 局部计算缓冲(ln(q)或中间结果) + 最终reduce的部分和缓冲
        pipe.InitBuffer(inQueueIN, BUFFER_NUM, this->tileLength * 2 * sizeof(float));
        pipe.InitBuffer(calcBuf, this->tileLength * sizeof(float));
        pipe.InitBuffer(tempBuf, reduce_align * sizeof(float));
    }

    __aicore__ inline void Process() {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
        }
        // 对所有tile的部分和再做一次ReduceSum
        AscendC::LocalTensor<float> partSums = tempBuf.Get<float>();
        AscendC::LocalTensor<float> tmpVec = inQueueIN.AllocTensor<float>();
        AscendC::Duplicate(tmpVec, (float)0, this->tileLength);
        AscendC::ReduceSum<float>(partSums, partSums, tmpVec, this->reduce_num);

        // batchmean：除以batchSize
        tmpVec.SetValue(0, static_cast<float>(this->batchSize));
        AscendC::Div(partSums, partSums, tmpVec, 1);

        outGm.SetValue(0, partSums.GetValue(0));
        inQueueIN.FreeTensor(tmpVec);
    }

private:
    __aicore__ inline void CopyIn(int32_t progress) {
        AscendC::LocalTensor<float> inLocal = inQueueIN.AllocTensor<float>();
        // 将p和q依次拷入inLocal[0 : tileLength) 和 inLocal[tileLength : 2*tileLength)
        if (progress == this->tileNum - 1) {
            if (progress == 0) {
                AscendC::DataCopy(inLocal[0], pGm[0], this->tileLength);
                AscendC::DataCopy(inLocal[this->tileLength], qGm[0], this->tileLength);
            } else {
                AscendC::DataCopy(
                    inLocal[0],
                    pGm[(progress - 1) * this->tileLength + this->lasttileLength],
                    this->tileLength);
                AscendC::DataCopy(
                    inLocal[this->tileLength],
                    qGm[(progress - 1) * this->tileLength + this->lasttileLength],
                    this->tileLength);
            }
        } else {
            AscendC::DataCopy(inLocal[0], pGm[progress * this->tileLength], this->tileLength);
            AscendC::DataCopy(inLocal[this->tileLength], qGm[progress * this->tileLength], this->tileLength);
        }
        inQueueIN.EnQue(inLocal);
    }

    __aicore__ inline void Compute(int32_t progress) {
        AscendC::LocalTensor<float> inLocal = inQueueIN.DeQue<float>();
        AscendC::LocalTensor<float> pLocal = inLocal;                         // 概率p
        AscendC::LocalTensor<float> qLocal = inLocal[this->tileLength];       // 概率q
        AscendC::LocalTensor<float> lnqLocal = calcBuf.Get<float>();          // 存放ln(q)
        AscendC::LocalTensor<float> partSums = tempBuf.Get<float>();          // 部分和存放处

        // ln(p)写回pLocal
        AscendC::Ln(pLocal, pLocal, this->tileLength);
        // ln(q)写入lnqLocal，保留qLocal原值
        AscendC::Ln(lnqLocal, qLocal, this->tileLength);
        // ln(q) - ln(p)
        AscendC::Sub(lnqLocal, lnqLocal, pLocal, this->tileLength);
        // q * (ln(q) - ln(p))
        AscendC::Mul(lnqLocal, lnqLocal, qLocal, this->tileLength);
        // 对当前tile求和，使用pLocal作为workLocal
        AscendC::ReduceSum<float>(lnqLocal, lnqLocal, pLocal, this->tileLength);

        // 保存当前tile的部分和到tempBuf
        partSums.SetValue(progress, lnqLocal.GetValue(0));

        inQueueIN.FreeTensor(inLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueIN;
    AscendC::TBuf<> calcBuf;
    AscendC::TBuf<> tempBuf;

    AscendC::GlobalTensor<float> pGm;
    AscendC::GlobalTensor<float> qGm;
    AscendC::GlobalTensor<float> outGm;

    int32_t totalLength;
    uint32_t reduce_num;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t lasttileLength;
    uint32_t batchSize;
};

extern "C" __global__ __aicore__ void kl_div_loss_custom(GM_ADDR predict, GM_ADDR label, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelKlDivLoss op;
    op.Init(predict, label, y,
            tiling_data.totalLength, tiling_data.blockLength,
            tiling_data.tileNum, tiling_data.tileLength,
            tiling_data.lasttileLength, tiling_data.batchSize);
    op.Process();
}
