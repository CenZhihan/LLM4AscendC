kernel_src="""
#include "kernel_operator.h"

using DTYPE_T = float;

constexpr int32_t IN_Q_DEPTH = 1;

class KernelNetVladNoGhostClusters {
public:
    __aicore__ inline KernelNetVladNoGhostClusters() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR assignment, GM_ADDR clusters2, GM_ADDR z,
                                uint32_t B, uint32_t N, uint32_t D, uint32_t K) {
        this->B = B;
        this->N = N;
        this->D = D;
        this->K = K;

        // block split on batch dimension
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        uint32_t perBlock = (B + blocks - 1) / blocks;
        bStart = perBlock * idx;
        bEnd = bStart + perBlock;
        if (bEnd > B) bEnd = B;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_T *>(x) + bStart * (N * D), (bEnd - bStart) * (N * D));
        aGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_T *>(assignment) + bStart * (N * K), (bEnd - bStart) * (N * K));
        c2Gm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_T *>(clusters2), D * K);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_T *>(z) + bStart * (D * K), (bEnd - bStart) * (D * K));

        // Queues
        pipe.InitBuffer(inQX, IN_Q_DEPTH, D * sizeof(DTYPE_T));          // x row: [D]
        pipe.InitBuffer(inQA, IN_Q_DEPTH, K * sizeof(DTYPE_T));          // a row: [K]
        pipe.InitBuffer(tmpQD, IN_Q_DEPTH, D * sizeof(DTYPE_T));         // tmp D
        pipe.InitBuffer(tmpQK, IN_Q_DEPTH, K * sizeof(DTYPE_T));         // a_sum [K]
        pipe.InitBuffer(workQC2, IN_Q_DEPTH, D * K * sizeof(DTYPE_T));   // clusters2 [D*K]
        pipe.InitBuffer(outQZ, IN_Q_DEPTH, D * K * sizeof(DTYPE_T));     // vlad [D*K]
    }

    __aicore__ inline void Process() {
        // Load clusters2 once
        AscendC::LocalTensor<DTYPE_T> c2Local = workQC2.AllocTensor<DTYPE_T>();
        AscendC::DataCopy(c2Local, c2Gm, D * K);
        workQC2.EnQue(c2Local);

        for (uint32_t b = bStart; b < bEnd; ++b) {
            ComputeSample(b);
        }

        c2Local = workQC2.DeQue<DTYPE_T>();
        workQC2.FreeTensor(c2Local);
    }

private:
    __aicore__ inline void ComputeSample(uint32_t b) {
        // Allocate buffers
        AscendC::LocalTensor<DTYPE_T> vladLocal = outQZ.AllocTensor<DTYPE_T>(); // [D*K]
        AscendC::LocalTensor<DTYPE_T> aSumLocal  = tmpQK.AllocTensor<DTYPE_T>(); // [K]
        AscendC::LocalTensor<DTYPE_T> c2Local = workQC2.DeQue<DTYPE_T>(); // [D*K]
        // init zeros
        DTYPE_T zero = static_cast<DTYPE_T>(0);
        AscendC::Duplicate<DTYPE_T>(vladLocal, zero, D * K);
        AscendC::Duplicate<DTYPE_T>(aSumLocal, zero, K);

        // iterate over N, accumulate vlad and a_sum
        for (uint32_t n = 0; n < N; ++n) {
            AscendC::LocalTensor<DTYPE_T> xRow = inQX.AllocTensor<DTYPE_T>();
            AscendC::LocalTensor<DTYPE_T> aRow = inQA.AllocTensor<DTYPE_T>();

            // x[b, n, :]
            uint64_t xOffset = static_cast<uint64_t>(b - bStart) * (N * D) + static_cast<uint64_t>(n) * D;
            AscendC::DataCopy(xRow, xGm[xOffset], D);
            // assignment[b, n, :]
            uint64_t aOffset = static_cast<uint64_t>(b - bStart) * (N * K) + static_cast<uint64_t>(n) * K;
            AscendC::DataCopy(aRow, aGm[aOffset], K);

            inQX.EnQue(xRow);
            inQA.EnQue(aRow);

            // update a_sum
            xRow = inQX.DeQue<DTYPE_T>();
            aRow = inQA.DeQue<DTYPE_T>();
            AscendC::Add(aSumLocal, aSumLocal, aRow, K);

            // outer product accumulate: for each k, vlad[:,k] += aRow[k] * xRow[:]
            AscendC::LocalTensor<DTYPE_T> tmpD = tmpQD.AllocTensor<DTYPE_T>();
            for (uint32_t k = 0; k < K; ++k) {
                // scalar assignment value
                // assume we can read scalar from aRow[k]
                DTYPE_T coef = static_cast<DTYPE_T>(0); // placeholder for scalar load
                // NOTE: platform-specific intrinsic to read scalar may differ; pseudo-code:
                // coef = aRow.GetValue(k);
                // use Muls to compute tmpD = xRow * coef
                AscendC::Muls(tmpD, xRow, coef, D);
                // accumulate to vladLocal segment
                AscendC::Add(vladLocal[k * D], vladLocal[k * D], tmpD, D);
            }
            tmpQD.FreeTensor(tmpD);
            inQX.FreeTensor(xRow);
            inQA.FreeTensor(aRow);
        }

        // subtract a = a_sum * clusters2, for each k: vlad[:,k] -= aSum[k] * clusters2[:,k]
        AscendC::LocalTensor<DTYPE_T> tmpD2 = tmpQD.AllocTensor<DTYPE_T>();
        for (uint32_t k = 0; k < K; ++k) {
            DTYPE_T coef = static_cast<DTYPE_T>(0); // placeholder for scalar load
            // coef = aSumLocal.GetValue(k);
            AscendC::Muls(tmpD2, c2Local[k * D], coef, D);
            AscendC::Sub(vladLocal[k * D], vladLocal[k * D], tmpD2, D);
        }
        tmpQD.FreeTensor(tmpD2);

        // store output flattened [D*K]
        outQZ.EnQue<DTYPE_T>(vladLocal);
        AscendC::LocalTensor<DTYPE_T> vladOut = outQZ.DeQue<DTYPE_T>();
        uint64_t zOffset = static_cast<uint64_t>(b - bStart) * (D * K);
        AscendC::DataCopy(zGm[zOffset], vladOut, D * K);
        outQZ.FreeTensor(vladOut);

        tmpQK.FreeTensor(aSumLocal);
        workQC2.EnQue(c2Local);
    }

private:
    AscendC::TPipe pipe;
    // Queues
    AscendC::TQue<AscendC::QuePosition::VECIN, IN_Q_DEPTH> inQX;     // x row
    AscendC::TQue<AscendC::QuePosition::VECIN, IN_Q_DEPTH> inQA;     // assignment row
    AscendC::TQue<AscendC::QuePosition::VECCALC, IN_Q_DEPTH> tmpQD;  // temp D vec
    AscendC::TQue<AscendC::QuePosition::VECCALC, IN_Q_DEPTH> tmpQK;  // a_sum K vec
    AscendC::TQue<AscendC::QuePosition::VECIN, IN_Q_DEPTH> workQC2;  // clusters2
    AscendC::TQue<AscendC::QuePosition::VECOUT, IN_Q_DEPTH> outQZ;   // output vlad

    // GM tensors
    AscendC::GlobalTensor<DTYPE_T> xGm;
    AscendC::GlobalTensor<DTYPE_T> aGm;
    AscendC::GlobalTensor<DTYPE_T> c2Gm;
    AscendC::GlobalTensor<DTYPE_T> zGm;

    // dims and block range
    uint32_t B, N, D, K;
    uint32_t bStart, bEnd;
};

extern "C" __global__ __aicore__ void net_vlad_no_ghost_clusters_custom(GM_ADDR x, GM_ADDR assignment, GM_ADDR clusters2, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelNetVladNoGhostClusters op;
    op.Init(x, assignment, clusters2, z, tiling_data.B, tiling_data.N, tiling_data.D, tiling_data.K);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import torch.nn.functional as F
import math

class ModelNew(torch.nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(ModelNew, self).__init__()
        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        # (w,b) folded into BatchNorm weights; keep as in original for logits
        self.clusters = torch.nn.Parameter(init_sc * torch.randn(feature_size, clusters))
        self.batch_norm = torch.nn.BatchNorm1d(clusters)
        # visual words c_k
        self.clusters2 = torch.nn.Parameter(init_sc * torch.randn(1, feature_size, cluster_size))

    def forward(self, x, mask=None):
        # x: [B, N, D]
        B, N, D = x.shape
        x_flat = x.reshape(-1, self.feature_size)  # [B*N, D]

        # assignment: BN x (K+G)
        assignment = torch.matmul(x_flat, self.clusters)            # [B*N, K+G]
        assignment = self.batch_norm(assignment)                    # BN x (K+G)
        assignment = F.softmax(assignment, dim=1)                   # BN x (K+G)

        # remove ghost assignments -> BN x K
        if self.ghost_clusters > 0:
            assignment = assignment[:, :self.cluster_size]
        assignment = assignment.reshape(B, N, self.cluster_size)    # B x N x K

        # clusters2: [1, D, K] -> [D, K]
        c2 = self.clusters2.squeeze(0)

        # call fused custom op to compute VLAD core (without L2 norms)
        vlad = custom_ops_lib.net_vlad_no_ghost_clusters_custom(x, assignment, c2)  # [B, D*K]

        # L2 intra norm: reshape back to [B, D, K], normalize on dim=1 (D)
        vlad = vlad.view(B, self.feature_size, self.cluster_size)
        vlad = F.normalize(vlad, p=2.0, dim=1)

        # flatten + L2 norm over DK
        vlad = vlad.reshape(B, self.cluster_size * self.feature_size)
        vlad = F.normalize(vlad, p=2.0, dim=1)
        return vlad
'''
