kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_B = float;
using DTYPE_Z = float;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConvTranspose3dLogSumExpHardSwishSubtractClampMax {
public:
    __aicore__ inline KernelConvTranspose3dLogSumExpHardSwishSubtractClampMax() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)bias + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(bLocal, bGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }
    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.DeQue<DTYPE_B>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> tmpLocal = tmpQueue.AllocTensor<DTYPE_Z>();

        // tmp = x + 3
        AscendC::Copy(tmpLocal, xLocal, this->tileLength);
        AscendC::Adds(tmpLocal, tmpLocal, static_cast<DTYPE_Z>(3.0f), this->tileLength);
        // tmp = min(max(tmp, 0), 6)  -> ReLU6(x+3)
        AscendC::Maxs(tmpLocal, tmpLocal, static_cast<DTYPE_Z>(0.0f), this->tileLength);
        AscendC::Mins(tmpLocal, tmpLocal, static_cast<DTYPE_Z>(6.0f), this->tileLength);
        // y = x * tmp
        AscendC::Mul(yLocal, xLocal, tmpLocal, this->tileLength);
        // y = y * (1/6)
        AscendC::Muls(yLocal, yLocal, static_cast<DTYPE_Z>(1.0f/6.0f), this->tileLength);
        // y = y - bias (bias is expanded to the same shape as x on host side)
        AscendC::Sub(yLocal, yLocal, bLocal, this->tileLength);
        // clamp to [-1, 1]
        AscendC::Maxs(yLocal, yLocal, static_cast<DTYPE_Z>(-1.0f), this->tileLength);
        AscendC::Mins(yLocal, yLocal, static_cast<DTYPE_Z>(1.0f), this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_log_sum_exp_hard_swish_subtract_clamp_max_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dLogSumExpHardSwishSubtractClampMax op;
    op.Init(x, bias, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keeps ConvTranspose3d + LogSumExp in PyTorch,
    fuses HardSwish + subtract + clamp into a custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, bias_shape):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        # Keep the same bias parameter definition as original
        self.bias = torch.nn.Parameter(torch.randn(*bias_shape))

    def forward(self, x):
        x = self.conv_transpose(x)
        x = torch.logsumexp(x, dim=1, keepdim=True)
        # Expand bias to match x for vectorized subtraction in kernel
        if self.bias.dim() == 4:
            bias_expanded = self.bias.reshape(1, 1, 1, 1, 1).expand_as(x)
        else:
            bias_expanded = self.bias.expand_as(x)
        x = custom_ops_lib.conv_transpose3d_log_sum_exp_hard_swish_subtract_clamp_max_custom(x, bias_expanded)
        return x
