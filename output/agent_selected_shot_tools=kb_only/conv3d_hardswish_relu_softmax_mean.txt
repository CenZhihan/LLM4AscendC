kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

// 计算 HardSwish: x * clamp(x+3, 0, 6) / 6
__aicore__ inline void Hardswish(AscendC::LocalTensor<DTYPE>& dst,
                                 const AscendC::LocalTensor<DTYPE>& src,
                                 uint32_t calCount) {
    AscendC::LocalTensor<DTYPE> tmp = dst; // 复用 dst 作为中间空间
    // tmp = src + 3
    AscendC::Adds(tmp, src, static_cast<DTYPE>(3.0f), calCount);
    // tmp = max(tmp, 0)
    AscendC::Maxs(tmp, tmp, static_cast<DTYPE>(0.0f), calCount);
    // tmp = min(tmp, 6)
    AscendC::Mins(tmp, tmp, static_cast<DTYPE>(6.0f), calCount);
    // tmp = tmp / 6
    AscendC::Muls(tmp, tmp, static_cast<DTYPE>(1.0f/6.0f), calCount);
    // dst = src * tmp
    AscendC::Mul(dst, src, tmp, calCount);
}

// Kernel 实现：对输入 x（B,C,D,H,W）
// 1) 对每个切片 (N' = B*D) 做 HardSwish -> GroupNorm（组数由 attr 提供）
// 2) 对空间 D,H,W 求平均（输出 [B,C]）
class KernelConv3dHardswishReluSoftmaxMeanCustom {
public:
    __aicore__ inline KernelConv3dHardswishReluSoftmaxMeanCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t B, uint32_t C, uint32_t D, uint32_t H, uint32_t W,
                                uint32_t group, float epsilon,
                                const optiling::Conv3dHardswishReluSoftmaxMeanCustomTiling &tld)
    {
        this->B = B; this->C = C; this->D = D; this->H = H; this->W = W;
        this->group = group; this->epsilon = epsilon;

        uint32_t Np = B * D;
        this->Np = Np;
        this->HW = H * W;
        this->BStride = C * D * H * W;
        this->NSliceStride = C * H * W;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(x), static_cast<uint32_t>(B*C*D*H*W));
        gammaGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(gamma), C);
        betaGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(beta), C);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(y), B * C);

        // 队列与缓冲初始化
        // 单次处理 1 个 N' 切片，局部缓冲大小：C*H*W
        tileLength = C * H * W;
        pipe.InitBuffer(inQueueX, BUFFER_NUM, tileLength * sizeof(DTYPE));
        pipe.InitBuffer(inQueueGamma, 1, C * sizeof(DTYPE));
        pipe.InitBuffer(inQueueBeta, 1, C * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, 1, B * C * sizeof(DTYPE)); // 最终输出累加缓冲
        // 输出累加缓冲初始化为 0
        outAccum = outQueueY.AllocTensor<DTYPE>();
        AscendC::InitGlobalMemory(outAccum, outAccum.GetSize(), static_cast<DTYPE>(0.0f));
        outQueueY.EnQue(outAccum);
    }

    __aicore__ inline void Process()
    {
        // 拷贝 gamma/beta
        AscendC::LocalTensor<DTYPE> gammaLocal = inQueueGamma.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> betaLocal = inQueueBeta.AllocTensor<DTYPE>();
        AscendC::DataCopy(gammaLocal, gammaGm, C);
        AscendC::DataCopy(betaLocal, betaGm, C);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);

        // 取输出累加缓冲
        AscendC::LocalTensor<DTYPE> outLocal = outQueueY.DeQue<DTYPE>();

        // 遍历 N' = B*D 切片
        for (uint32_t nIdx = 0; nIdx < Np; ++nIdx) {
            // 计算当前切片对应 batch 索引
            uint32_t bIdx = nIdx / D;

            // 读入当前 N' 切片 (C,H,W)
            AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
            AscendC::DataCopy(xLocal, xGm[nIdx * NSliceStride], tileLength);

            // HardSwish
            Hardswish(xLocal, xLocal, tileLength);

            // 取 gamma/beta
            AscendC::LocalTensor<DTYPE> gammaL = inQueueGamma.DeQue<DTYPE>();
            AscendC::LocalTensor<DTYPE> betaL = inQueueBeta.DeQue<DTYPE>();

            // GroupNorm（N=1, C, H, W）
            // 为简化，直接使用 Host 传入的 tiling（适配 N' 维度分片）
            // 输出与中间 mean/var 缓冲
            AscendC::LocalTensor<DTYPE> gnOut = inQueueX.AllocTensor<DTYPE>(); // 复用队列空间
            AscendC::LocalTensor<DTYPE> gnMean = inQueueX.AllocTensor<DTYPE>(); // 大小按 (groupNum * N=1)
            AscendC::LocalTensor<DTYPE> gnVar = inQueueX.AllocTensor<DTYPE>();
            AscendC::GroupNorm<DTYPE, false>(
                gnOut, gnMean, gnVar,
                xLocal, gammaL, betaL,
                static_cast<DTYPE>(epsilon),
                tilingData.groupNormTilingData
            );

            // 将 gnOut 在 H*W 上求和并累加到 outLocal[bIdx, :]
            // 逐通道累加
            for (uint32_t cIdx = 0; cIdx < C; ++cIdx) {
                // 通道 c 的偏移
                uint32_t base = cIdx * HW;
                // 使用矢量 Reduce 对当前通道块求和
                // 简化：分段加法累加（按 tile），每段大小取 HW
                // 这里直接循环累加（示例），实际可用向量化 ReduceSum 优化
                DTYPE sumVal = 0.0f;
                for (uint32_t hw = 0; hw < HW; ++hw) {
                    sumVal += gnOut.GetValue(base + hw);
                }
                // 求平均: / (D*H*W)，并累加到 outLocal[bIdx*C + cIdx]
                DTYPE meanVal = sumVal / static_cast<DTYPE>(D * H * W);
                DTYPE oldVal = outLocal.GetValue(bIdx * C + cIdx);
                outLocal.SetValue(bIdx * C + cIdx, oldVal + meanVal);
            }

            // 释放当前切片局部张量
            inQueueX.FreeTensor(gnVar);
            inQueueX.FreeTensor(gnMean);
            inQueueX.FreeTensor(gnOut);
            inQueueX.FreeTensor(xLocal);

            // 归还 gamma/beta 以便后续复用
            inQueueGamma.EnQue(gammaL);
            inQueueBeta.EnQue(betaL);
        }

        // 写回输出
        AscendC::DataCopy(yGm, outLocal, B * C);

        // 释放
        outQueueY.FreeTensor(outLocal);
        AscendC::LocalTensor<DTYPE> gL = inQueueGamma.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> bL = inQueueBeta.DeQue<DTYPE>();
        inQueueGamma.FreeTensor(gL);
        inQueueBeta.FreeTensor(bL);
    }

    optiling::Conv3dHardswishReluSoftmaxMeanCustomTiling tilingData;

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueGamma;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueBeta;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> gammaGm;
    AscendC::GlobalTensor<DTYPE> betaGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t B, C, D, H, W;
    uint32_t Np, HW;
    uint32_t BStride, NSliceStride;
    uint32_t group;
    float epsilon;

    uint32_t tileLength;
    AscendC::LocalTensor<DTYPE> outAccum;
};

extern "C" __global__ __aicore__ void conv3d_hardswish_relu_softmax_mean_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    KernelConv3dHardswishReluSoftmaxMeanCustom op;
    op.tilingData = tilingData;

    op.Init(
        x, gamma, beta, y,
        tilingData.n(), tilingData.c(), tilingData.d(), tilingData.h(), tilingData.w(),
        tilingData.group(), tilingData.epsilon(),
        tilingData
    );
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版：保留 Conv3D 使用原生实现；融合 GroupNorm + 空间均值，以及在核内完成 HardSwish。
    """
    def __init__(self, in_channels, out_channels, kernel_size, num_groups=4, bias=True):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size, bias=bias)
        self.num_groups = num_groups
        # 保持与原模型一致的 GroupNorm 参数（作为 gamma/beta 输入给自定义算子）
        self.group_norm = torch.nn.GroupNorm(num_groups, out_channels)

    def forward(self, x):
        # Conv3D (原生)
        x = self.conv(x)  # [B,C,D,H,W]
        # 直接调用自定义 AscendC 融合核：核内执行 HardSwish -> GroupNorm -> Mean([2,3,4])
        y = custom_ops_lib.conv3d_hardswish_relu_softmax_mean_custom(
            x, self.group_norm.weight, self.group_norm.bias, self.num_groups, 1e-5
        )  # [B,C]
        return y
