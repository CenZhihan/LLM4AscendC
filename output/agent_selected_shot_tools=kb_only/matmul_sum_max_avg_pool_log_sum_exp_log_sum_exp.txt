kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1; // simple single-buffer pipeline

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wsum, GM_ADDR bsum, GM_ADDR y, uint32_t rowLength, uint32_t totalRows, uint32_t tileRows)
    {
        this->rowLength = rowLength;
        this->totalRows = totalRows;
        this->tileRows = tileRows;

        uint32_t blockRows = totalRows / AscendC::GetBlockNum(); // assume divisible
        this->blockRows = blockRows;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + (uint64_t)blockRows * this->rowLength * AscendC::GetBlockIdx(), (uint64_t)blockRows * this->rowLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + (uint64_t)blockRows * AscendC::GetBlockIdx(), blockRows);
        wsumGm.SetGlobalBuffer((__gm__ DTYPE*)wsum, this->rowLength);
        bsumGm.SetGlobalBuffer((__gm__ DTYPE*)bsum, 1);

        // Buffers
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->rowLength * sizeof(DTYPE));      // one row
        pipe.InitBuffer(inQueueWSum, 1, this->rowLength * sizeof(DTYPE));            // wsum vector
        pipe.InitBuffer(tmpProdQueue, BUFFER_NUM, this->rowLength * sizeof(DTYPE));  // product per row
        pipe.InitBuffer(workQueue, 1, 2048);                                         // temporary work space for ReduceSum
        pipe.InitBuffer(inQueueBSum, 1, sizeof(DTYPE));                              // scalar bias sum
        pipe.InitBuffer(outQueueY, BUFFER_NUM, sizeof(DTYPE));                       // one scalar per row
    }

    __aicore__ inline void Process()
    {
        // Load wsum and bsum once
        AscendC::LocalTensor<DTYPE> wLocal = inQueueWSum.AllocTensor<DTYPE>();
        AscendC::DataCopy(wLocal, wsumGm, this->rowLength);
        inQueueWSum.EnQue(wLocal);

        AscendC::LocalTensor<DTYPE> bLocal = inQueueBSum.AllocTensor<DTYPE>();
        AscendC::DataCopy(bLocal, bsumGm, 1);
        inQueueBSum.EnQue(bLocal);

        for (uint32_t r = 0; r < this->blockRows; ++r) {
            CopyIn(r);
            ComputeRow();
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        // Copy one full row
        AscendC::DataCopy(xLocal, xGm[(uint64_t)rowIdx * this->rowLength], this->rowLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> wLocal = inQueueWSum.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = inQueueBSum.DeQue<DTYPE>();

        AscendC::LocalTensor<DTYPE> prodLocal = tmpProdQueue.AllocTensor<DTYPE>();
        // Elementwise product
        AscendC::Mul(prodLocal, xLocal, wLocal, this->rowLength);

        AscendC::LocalTensor<DTYPE> sumLocal = outQueueY.AllocTensor<DTYPE>(); // 1 element to hold sum
        AscendC::LocalTensor<DTYPE> workLocal = workQueue.AllocTensor<DTYPE>();

        // ReduceSum over rowLength elements -> scalar
        AscendC::ReduceSum(sumLocal, prodLocal, workLocal, this->rowLength);

        // y = sum + bsum (1-element tensors)
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
        AscendC::Add(yLocal, sumLocal, bLocal, 1);

        outQueueY.EnQue<DTYPE>(yLocal);

        // recycle queues
        tmpProdQueue.FreeTensor(prodLocal);
        outQueueY.FreeTensor(sumLocal);
        workQueue.FreeTensor(workLocal);

        // Keep wsum and bsum persistent by enqueuing back
        inQueueWSum.EnQue(wLocal);
        inQueueBSum.EnQue(bLocal);

        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[rowIdx], yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueWSum;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueBSum;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpProdQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wsumGm;
    AscendC::GlobalTensor<DTYPE> bsumGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t rowLength;
    uint32_t totalRows;
    uint32_t blockRows;
    uint32_t tileRows;
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(GM_ADDR x, GM_ADDR wsum, GM_ADDR bsum, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, wsum, bsum, y, tiling_data.rowLength, tiling_data.totalRows, tiling_data.tileRows);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused kernel replaces:
      Linear -> sum(dim=1, keepdim=True) -> max/mean/logsumexp/logsumexp
    Note: After summation to shape (B,1), the subsequent ops over dim=1 with keepdim=True
    are no-ops. So the fused computation reduces to: y = x @ (sum(weight, dim=0)) + sum(bias).
    """
    def __init__(self, in_features, out_features):
        super().__init__()
        self.linear = torch.nn.Linear(in_features, out_features, bias=True)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Precompute fused parameters on-the-fly from the Linear weights/bias
        # wsum: sum over out_features -> shape [in_features]
        wsum = self.linear.weight.sum(dim=0)
        # bsum: scalar
        bsum = self.linear.bias.sum() if self.linear.bias is not None else torch.tensor(0.0, dtype=x.dtype, device=x.device)
        # Call fused custom op
        return custom_ops_lib.matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(x, wsum, bsum.view(1))
