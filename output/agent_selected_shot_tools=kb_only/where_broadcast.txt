static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // x,y 的形状一致，输出形状与 x 相同；cond 形状可广播为 [d0,d1,d2]，其中 cond 提供 [1,d1,1]
    const gert::Shape* x_shape = context->GetInputShape(1);
    auto x_origin = x_shape->GetOriginShape();
    uint32_t d0 = static_cast<uint32_t>(x_origin.GetDim(0));
    uint32_t d1 = static_cast<uint32_t>(x_origin.GetDim(1));
    uint32_t d2 = static_cast<uint32_t>(x_origin.GetDim(2));

    // 每个 Core 处理一个 d0 切片，确保广播维度处理简单
    context->SetBlockDim(d0);

    // 选择沿最后一维的 tileK，保证 32B 对齐与 16 的倍数，优先使用完整最后一维
    uint32_t tileK = d2;
    // 可根据 UB 空间适当缩小 tileK，这里保持简单直接使用 d2

    WhereBroadcastCustomTilingData tiling;
    tiling.set_d0(d0);
    tiling.set_d1(d1);
    tiling.set_d2(d2);
    tiling.set_tileK(tileK);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // 无需 workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    context->SetTilingKey(1);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // 每个队列的 tensor 数

class KernelWhereBroadcast {
public:
    __aicore__ inline KernelWhereBroadcast() {}

    __aicore__ inline void Init(GM_ADDR cond, GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t d0, uint32_t d1, uint32_t d2, uint32_t tileK)
    {
        this->d0 = d0;
        this->d1 = d1;
        this->d2 = d2;
        this->tileK = tileK;

        // 每个 Core 处理一个 i 切片
        uint32_t i_idx = AscendC::GetBlockIdx();
        if (i_idx >= d0) {
            // 安全保护
            this->valid = false;
            return;
        }
        this->valid = true;
        this->i_idx = i_idx;

        // cond 视为长度为 d1 的一维张量（cond[0, j, 0]）
        condGm.SetGlobalBuffer(reinterpret_cast<__gm__ bool *>(cond), d1);

        // x/y/z 的起始偏移为 i_idx * (d1 * d2)
        uint32_t rowSize = d1 * d2;
        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(x) + rowSize * i_idx, rowSize);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(y) + rowSize * i_idx, rowSize);
        zGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(z) + rowSize * i_idx, rowSize);

        // UB 队列初始化：按最后一维 tileK 处理
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileK * sizeof(float));
        pipe.InitBuffer(condQueue, 1, sizeof(bool)); // 每个 j 只读一个 cond 标量
    }

    __aicore__ inline void Process()
    {
        if (!valid) return;

        // 逐 j（中间维）遍历，每次 j 的 cond 值相同，在最后一维上做整段拷贝
        for (uint32_t j = 0; j < d1; ++j) {
            // 读取 cond[0, j, 0] 标量
            AscendC::LocalTensor<bool> condLocal = condQueue.AllocTensor<bool>();
            AscendC::DataCopy(condLocal, condGm[j], 1);
            bool flag = condLocal.GetValue(0);
            condQueue.FreeTensor(condLocal);

            // 在最后一维上分段处理
            for (uint32_t k = 0; k < d2; k += tileK) {
                uint32_t curLen = AscendC::min(tileK, d2 - k);

                // CopyIn
                AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                AscendC::LocalTensor<float> yLocal = inQueueY.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm[j * d2 + k], curLen);
                AscendC::DataCopy(yLocal, yGm[j * d2 + k], curLen);
                inQueueX.EnQue(xLocal);
                inQueueY.EnQue(yLocal);

                // Compute
                AscendC::LocalTensor<float> xLocal2 = inQueueX.DeQue<float>();
                AscendC::LocalTensor<float> yLocal2 = inQueueY.DeQue<float>();
                AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
                if (flag) {
                    AscendC::DataCopy(zLocal, xLocal2, curLen);
                } else {
                    AscendC::DataCopy(zLocal, yLocal2, curLen);
                }
                outQueueZ.EnQue<float>(zLocal);
                inQueueX.FreeTensor(xLocal2);
                inQueueY.FreeTensor(yLocal2);

                // CopyOut
                AscendC::LocalTensor<float> zLocal2 = outQueueZ.DeQue<float>();
                AscendC::DataCopy(zGm[j * d2 + k], zLocal2, curLen);
                outQueueZ.FreeTensor(zLocal2);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX, inQueueY;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> condQueue;

    AscendC::GlobalTensor<bool> condGm;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> yGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t d0 = 0, d1 = 0, d2 = 0, tileK = 0;
    uint32_t i_idx = 0;
    bool valid = false;
};

extern "C" __global__ __aicore__ void where_broadcast_custom(GM_ADDR cond, GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelWhereBroadcast op;
    op.Init(cond, x, y, z, tiling_data.d0, tiling_data.d1, tiling_data.d2, tiling_data.tileK);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}
