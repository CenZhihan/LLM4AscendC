kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 1;

class KernelConv3dMaxLogSumExpRelu {
public:
    __aicore__ inline KernelConv3dMaxLogSumExpRelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t D, uint32_t H, uint32_t W,
                                uint64_t outerSize, uint32_t tileOuter)
    {
        this->N = N; this->C = C; this->D = D; this->H = H; this->W = W;
        this->outerSize = outerSize;
        this->tileOuter = tileOuter;

        // 分块
        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();
        this->blockOuter = static_cast<uint64_t>((outerSize + blockNum - 1) / blockNum);
        uint64_t begin = this->blockOuter * this->blockIdx;
        if (begin > outerSize) begin = outerSize;
        uint64_t end = this->blockOuter * (this->blockIdx + 1);
        if (end > outerSize) end = outerSize;
        this->workOuter = (end > begin) ? (end - begin) : 0;
        this->startOuter = begin;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, static_cast<uint64_t>(N) * C * D * H * W);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, outerSize);

        // UB 分配：每次聚合一条通道向量（长度 C），以及若干临时缓冲
        pipe.InitBuffer(inQueue, BUFFER_NUM, C * sizeof(DTYPE));            // x_line
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, C * sizeof(DTYPE) * 4);       // 临时缓冲: work, exp, etc.
        pipe.InitBuffer(outQueue, BUFFER_NUM, sizeof(DTYPE));               // 输出标量
    }

    __aicore__ inline void Process()
    {
        for (uint64_t pos = 0; pos < workOuter; ++pos) {
            CopyIn(pos);
            Compute(pos);
            CopyOut(pos);
        }
    }

private:
    // 线性化假设：按最后一维为通道连续处理（示例代码，需与上层布局保持一致）
    // 对于形状 [N, C, D, H, W]，此处简化为按某一空间位置聚合 C 个通道
    __aicore__ inline void CopyIn(uint64_t progress)
    {
        AscendC::LocalTensor<DTYPE> xLine = inQueue.AllocTensor<DTYPE>();
        // 起始外层位置
        uint64_t outerIndex = startOuter + progress;
        // 简化：假设通道作为最后一维连续存储（NDHWC风格）
        // 若为 NCDHW 连续布局，上层可在调用前调整布局或以实际实现替换
        uint64_t base = outerIndex * C;
        AscendC::DataCopy(xLine, xGm[base], C);
        inQueue.EnQue(xLine);
    }

    __aicore__ inline void Compute(uint64_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE> xLine = inQueue.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> tmp0 = tmpQueue.AllocTensor<DTYPE>(); // workLocal / 中间缓冲
        AscendC::LocalTensor<DTYPE> tmp1 = tmpQueue.AllocTensor<DTYPE>(); // exp buffer
        AscendC::LocalTensor<DTYPE> outScalar = outQueue.AllocTensor<DTYPE>(); // 输出标量（1元素）

        // 1) 计算通道维最大值 m
        // ReduceMax 返回 [max, index]，此处作为示例，dstMax 长度设为 2
        AscendC::LocalTensor<DTYPE> dstMax = tmpQueue.AllocTensor<DTYPE>();
        // 为了简化，假定 tmp0 作为 workLocal 使用
        AscendC::ReduceMax(dstMax, xLine, tmp0, static_cast<int32_t>(C), false);
        DTYPE mVal = dstMax.GetValue(0);

        // 2) 计算 y = exp(x - m)
        // 使用 Adds 实现向量与标量相减：y = x + (-mVal)
        AscendC::Adds(tmp1, xLine, -mVal, C);
        AscendC::Exp(tmp1, tmp1, C);

        // 3) 计算 s = sum(exp(x - m))
        AscendC::LocalTensor<DTYPE> dstSum = tmpQueue.AllocTensor<DTYPE>();
        // 简化调用形式：ReduceSum(dst, src, count)
        AscendC::ReduceSum(dstSum, tmp1, static_cast<int32_t>(C));

        // 4) 计算 logsumexp = m + ln(s)
        DTYPE sVal = dstSum.GetValue(0);
        AscendC::LocalTensor<DTYPE> oneVal = tmpQueue.AllocTensor<DTYPE>();
        // 将 sVal 放入 oneVal[0]
        // 利用 Duplicate 将标量写入 oneVal（长度1）
        AscendC::Duplicate(oneVal, sVal, 1);
        AscendC::Ln(oneVal, oneVal, 1);
        DTYPE lnS = oneVal.GetValue(0);
        DTYPE lse = mVal + lnS;

        // 5) ReLU
        // 写入 outScalar，执行 ReLU（长度1）
        AscendC::Duplicate(outScalar, lse, 1);
        AscendC::Relu(outScalar, outScalar, 1);

        // 入队并释放临时
        outQueue.EnQue(outScalar);
        inQueue.FreeTensor(xLine);
        tmpQueue.FreeTensor(tmp0);
        tmpQueue.FreeTensor(tmp1);
        tmpQueue.FreeTensor(dstMax);
        tmpQueue.FreeTensor(dstSum);
        tmpQueue.FreeTensor(oneVal);
    }

    __aicore__ inline void CopyOut(uint64_t progress)
    {
        AscendC::LocalTensor<DTYPE> outScalar = outQueue.DeQue<DTYPE>();
        uint64_t outerIndex = startOuter + progress;
        AscendC::DataCopy(yGm[outerIndex], outScalar, 1);
        outQueue.FreeTensor(outScalar);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t N, C, D, H, W;
    uint64_t outerSize, workOuter, startOuter;
    uint32_t tileOuter;
    uint32_t blockNum, blockIdx;
    uint64_t blockOuter;
};

extern "C" __global__ __aicore__ void conv3d_max_log_sum_exp_relu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv3dMaxLogSumExpRelu op;
    op.Init(x, y,
            tilingData.N, tilingData.C, tilingData.D, tilingData.H, tilingData.W,
            tilingData.outerSize, tilingData.tileOuter);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    以 PyTorch 完成 Conv3d 与 MaxPool3d，随后用自定义 AscendC Kernel 完成 logsumexp(dim=1, keepdim=True)+ReLU
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super().__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.max_pool = torch.nn.MaxPool3d(kernel_size=2, stride=2)

    def forward(self, x):
        x = self.conv(x)
        x = self.max_pool(x)
        # 自定义AscendC算子：基于通道维的logsumexp并ReLU
        x = custom_ops_lib.conv3d_max_log_sum_exp_relu_custom(x)
        return x
