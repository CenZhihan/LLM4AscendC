#include "bilinear_upsample_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    BilinearUpsampleCustomTilingData tiling;
    const gert::Shape* in_shape = context->GetInputShape(0);
    const gert::Shape* out_shape = context->GetOutputShape(0);

    uint32_t n  = (in_shape->GetOriginShape().GetDimNum() > 0) ? (uint32_t)in_shape->GetOriginShape().GetDim(0) : 1;
    uint32_t c  = (in_shape->GetOriginShape().GetDimNum() > 1) ? (uint32_t)in_shape->GetOriginShape().GetDim(1) : 1;
    uint32_t h  = (in_shape->GetOriginShape().GetDimNum() > 2) ? (uint32_t)in_shape->GetOriginShape().GetDim(2) : 1;
    uint32_t w  = (in_shape->GetOriginShape().GetDimNum() > 3) ? (uint32_t)in_shape->GetOriginShape().GetDim(3) : 1;

    uint32_t out_h = (out_shape->GetOriginShape().GetDimNum() > 2) ? (uint32_t)out_shape->GetOriginShape().GetDim(2) : (h * 2);
    uint32_t out_w = (out_shape->GetOriginShape().GetDimNum() > 3) ? (uint32_t)out_shape->GetOriginShape().GetDim(3) : (w * 2);

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_out_h(out_h);
    tiling.set_out_w(out_w);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    // 这里按外部构图给定的输出形状为准（一般为 H、W 维度放大 2 倍）
    // 若无，则默认将 H、W 放大 2 倍
    if (x_shape->GetOriginShape().GetDimNum() >= 4) {
        gert::Shape::DataShape out;
        out.Resize(4);
        out.SetDim(0, x_shape->GetOriginShape().GetDim(0));
        out.SetDim(1, x_shape->GetOriginShape().GetDim(1));
        out.SetDim(2, x_shape->GetOriginShape().GetDim(2) * 2);
        out.SetDim(3, x_shape->GetOriginShape().GetDim(3) * 2);
        y_shape->SetOriginShape(out);
    } else {
        *y_shape = *x_shape;
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class BilinearUpsampleCustom : public OpDef {
public:
    explicit BilinearUpsampleCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

    }

OP_ADD(BilinearUpsampleCustom);

kernel_src="""
#include "kernel_operator.h"

class KernelBilinearUpsample {
public:
    __aicore__ inline KernelBilinearUpsample() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t out_h, uint32_t out_w, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->out_h = out_h;
        this->out_w = out_w;
        this->tileNum = tileNum;

        // 将平面按 N*C 维进行分块
        uint32_t planes = n * c;
        uint32_t planesPerBlock = (planes + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t startPlane = planesPerBlock * AscendC::GetBlockIdx();
        uint32_t endPlane = startPlane + planesPerBlock;
        if (endPlane > planes) endPlane = planes;

        this->startPlane = startPlane;
        this->planeCount = (endPlane > startPlane) ? (endPlane - startPlane) : 0;

        uint64_t inCount  = (uint64_t)this->planeCount * (uint64_t)h * (uint64_t)w;
        uint64_t outCount = (uint64_t)this->planeCount * (uint64_t)out_h * (uint64_t)out_w;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)startPlane * (uint64_t)h * (uint64_t)w, inCount);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + (uint64_t)startPlane * (uint64_t)out_h * (uint64_t)out_w, outCount);

        // 为简化示例，按平面整体搬运（真实工程中建议按行或小块分片）
        pipe.InitBuffer(inQueueX, 1, (uint32_t)inCount * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, 1, (uint32_t)outCount * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (planeCount == 0) return;
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm, (uint32_t)(planeCount * h * w));
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // 使用硬件插值指令完成双线性上采样（纵向与横向迭代，align_corners=false）
        // 具体实现由底层指令完成，示例仅展示调用流程
        // 注意：实际项目中可按行块分片，减少UB占用
        // AscendC::BilinearInterpolation 执行 H、W 两方向的插值
        // 目标尺寸 out_h/out_w，源尺寸 h/w，align_corners=false
        // 以下为示意调用
        AscendC::BilinearInterpolation(yLocal, xLocal, h, w, out_h, out_w, false);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm, yLocal, (uint32_t)(planeCount * out_h * out_w));
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, h, w, out_h, out_w;
    uint32_t tileNum;

    uint32_t startPlane;
    uint32_t planeCount;
};

extern "C" __global__ __aicore__ void bilinear_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelBilinearUpsample op;
    op.Init(x, y,
            tilingData.n, tilingData.c, tilingData.h, tilingData.w,
            tilingData.out_h, tilingData.out_w, tilingData.tileNum);
    op.Process();
}
