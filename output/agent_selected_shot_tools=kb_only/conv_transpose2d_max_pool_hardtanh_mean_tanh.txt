kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelConvTranspose2dMaxPoolHardtanhMeanTanh {
public:
    __aicore__ inline KernelConvTranspose2dMaxPoolHardtanhMeanTanh() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR minGm, GM_ADDR maxGm, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t planesPerBlock)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->planeSize = h * w;
        this->totalPlanes = n * c;
        this->planesPerBlock = planesPerBlock;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startPlane = blockIdx * planesPerBlock;
        if (startPlane >= totalPlanes) {
            startPlane = totalPlanes;
        }
        this->startPlane = startPlane;
        uint32_t endPlane = startPlane + planesPerBlock;
        if (endPlane > totalPlanes) {
            endPlane = totalPlanes;
        }
        this->actualPlanes = (endPlane > startPlane) ? (endPlane - startPlane) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + (uint64_t)startPlane * planeSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + (uint64_t)startPlane * 1); // 每个平面输出1个标量
        minGmGlobal.SetGlobalBuffer((__gm__ DTYPE*)minGm);
        maxGmGlobal.SetGlobalBuffer((__gm__ DTYPE*)maxGm);

        // UB缓冲区：一整个平面 + 输出标量 + 标量缓存
        pipe.InitBuffer(inQueuePlane, 1, planeSize * sizeof(DTYPE));
        pipe.InitBuffer(outQueueScalar, 1, 32); // 标量占位，按框架对齐
        pipe.InitBuffer(scalarBufQueue, 1, 32); // 存放min/max
    }

    __aicore__ inline void Process()
    {
        // 读入 min/max 标量
        DTYPE minVal = 0.f, maxVal = 0.f;
        {
            AscendC::LocalTensor<DTYPE> tmpLocal = scalarBufQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(tmpLocal, minGmGlobal, 1);
            minVal = tmpLocal.GetValue(0);
            AscendC::DataCopy(tmpLocal, maxGmGlobal, 1);
            maxVal = tmpLocal.GetValue(0);
            scalarBufQueue.FreeTensor(tmpLocal);
        }

        for (uint32_t p = 0; p < actualPlanes; ++p) {
            // 复制整个平面到UB
            AscendC::LocalTensor<DTYPE> planeLocal = inQueuePlane.AllocTensor<DTYPE>();
            AscendC::DataCopy(planeLocal, xGm[p * planeSize], planeSize);

            // Hardtanh: clamp到[minVal, maxVal]
            AscendC::Maxs(planeLocal, planeLocal, minVal, planeSize);
            AscendC::Mins(planeLocal, planeLocal, maxVal, planeSize);

            // 计算 HxW 平均值（按最后一维，outter=1，inner=planeSize，对齐满足32字节）
            AscendC::LocalTensor<DTYPE> meanLocal = outQueueScalar.AllocTensor<DTYPE>();
            AscendC::MeanParams mp;
            mp.outter = 1;
            mp.inner = planeSize; // 由于 float，planeSize*4=65536，为32字节对齐
            mp.n = planeSize;
            AscendC::Mean<DTYPE, DTYPE, false, false, -1>(meanLocal, planeLocal, mp);

            // tanh 激活
            AscendC::Tanh(meanLocal, meanLocal, 1);

            // 写回结果
            AscendC::DataCopy(yGm[p], meanLocal, 1);

            outQueueScalar.FreeTensor(meanLocal);
            inQueuePlane.FreeTensor(planeLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueuePlane;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueScalar;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> scalarBufQueue;

    AscendC::GlobalTensor<DTYPE> xGm, yGm;
    AscendC::GlobalTensor<DTYPE> minGmGlobal, maxGmGlobal;

    uint32_t n{0}, c{0}, h{0}, w{0};
    uint32_t planeSize{0};
    uint32_t totalPlanes{0};
    uint32_t planesPerBlock{0};
    uint32_t startPlane{0};
    uint32_t actualPlanes{0};
};

extern "C" __global__ __aicore__ void conv_transpose2d_max_pool_hardtanh_mean_tanh_custom(
    GM_ADDR x, GM_ADDR minGm, GM_ADDR maxGm, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    uint32_t n  = tilingData.n;
    uint32_t c  = tilingData.c;
    uint32_t h  = tilingData.h;
    uint32_t w  = tilingData.w;
    uint32_t planesPerBlock = tilingData.planesPerBlock;

    KernelConvTranspose2dMaxPoolHardtanhMeanTanh op;
    op.Init(x, minGm, maxGm, y, n, c, h, w, planesPerBlock);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    优化版本：
    - 保留原始的 ConvTranspose2d 和 MaxPool2d（利用框架高效实现）
    - 将 Hardtanh + mean(沿HW, keepdim=True) + tanh 融合为自定义 AscendC 算子
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding,
                 maxpool_kernel_size, maxpool_stride, hardtanh_min, hardtanh_max):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size,
                                                       stride=stride, padding=padding)
        self.maxpool = torch.nn.MaxPool2d(kernel_size=maxpool_kernel_size, stride=maxpool_stride)
        self.hardtanh_min = hardtanh_min
        self.hardtanh_max = hardtanh_max

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.maxpool(x)
        # 自定义AscendC算子：对池化后的特征执行 Hardtanh + Mean(HW, keepdim=True) + Tanh
        x = custom_ops_lib.conv_transpose2d_max_pool_hardtanh_mean_tanh_custom(
            x, float(self.hardtanh_min), float(self.hardtanh_max)
        )
        return x
