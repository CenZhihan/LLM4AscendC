kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using DTYPE = float;

__aicore__ inline float sigmoidf(float x) {
    return 1.0f / (1.0f + expf(-x));
}

extern "C" __global__ __aicore__ void gemm_group_norm_swish_multiply_swish_custom(
    GM_ADDR x,           // [N, K]
    GM_ADDR w,           // [M, K] (row-major, W[o, k])
    GM_ADDR b,           // [M]
    GM_ADDR gn_weight,   // [M]
    GM_ADDR gn_bias,     // [M]
    GM_ADDR mul_w,       // [M]
    GM_ADDR groupsGM,    // [1] int32
    GM_ADDR epsGM,       // [1] float
    GM_ADDR z,           // [N, M]
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.N;
    const uint32_t K = tiling_data.K;
    const uint32_t M = tiling_data.M;

    __gm__ DTYPE* xPtr = reinterpret_cast<__gm__ DTYPE*>(x);
    __gm__ DTYPE* wPtr = reinterpret_cast<__gm__ DTYPE*>(w);
    __gm__ DTYPE* bPtr = reinterpret_cast<__gm__ DTYPE*>(b);
    __gm__ DTYPE* gammaPtr = reinterpret_cast<__gm__ DTYPE*>(gn_weight);
    __gm__ DTYPE* betaPtr  = reinterpret_cast<__gm__ DTYPE*>(gn_bias);
    __gm__ DTYPE* mulPtr   = reinterpret_cast<__gm__ DTYPE*>(mul_w);
    __gm__ int32_t* gPtr   = reinterpret_cast<__gm__ int32_t*>(groupsGM);
    __gm__ DTYPE* epsPtr   = reinterpret_cast<__gm__ DTYPE*>(epsGM);
    __gm__ DTYPE* zPtr     = reinterpret_cast<__gm__ DTYPE*>(z);

    // Read scalar groups and eps
    int32_t groups = *gPtr;
    DTYPE eps = *epsPtr;
    if (groups <= 0) {
        groups = 1;
    }
    uint32_t groupSize = M / static_cast<uint32_t>(groups);
    if (groupSize == 0) {
        groupSize = M; // fallback
    }

    // Partition groups across blocks to keep group boundaries intact
    uint32_t numGroups = M / groupSize;
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t groupsPerBlock = (numGroups + blockNum - 1) / blockNum;
    uint32_t gStart = blockIdx * groupsPerBlock;
    uint32_t gEnd = gStart + groupsPerBlock;
    if (gStart > numGroups) gStart = numGroups;
    if (gEnd > numGroups) gEnd = numGroups;

    // Allocate local buffer for one group's pre-activation values
    AscendC::TPipe pipe;
    constexpr int BUF_NUM = 1;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_NUM> bufQ;
    pipe.InitBuffer(bufQ, BUF_NUM, groupSize * sizeof(DTYPE));

    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t g = gStart; g < gEnd; ++g) {
            uint32_t mBase = g * groupSize;

            // buffer to hold y before normalization
            AscendC::LocalTensor<DTYPE> yLocal = bufQ.AllocTensor<DTYPE>();

            // First pass: GEMM for this (n, group) and accumulate sum, sqsum
            DTYPE sum = 0.0f;
            DTYPE sqsum = 0.0f;
            for (uint32_t i = 0; i < groupSize && (mBase + i) < M; ++i) {
                uint32_t mIdx = mBase + i;
                DTYPE acc = 0.0f;
                // Dot(x[n, :], W[mIdx, :])
                __gm__ DTYPE* wRow = wPtr + mIdx * K;
                __gm__ DTYPE* xRow = xPtr + n * K;
                for (uint32_t k = 0; k < K; ++k) {
                    acc += xRow[k] * wRow[k];
                }
                acc += bPtr[mIdx];
                yLocal.GetPhyAddr()[i] = acc;
                sum += acc;
                sqsum += acc * acc;
            }
            uint32_t valid = (mBase + groupSize <= M) ? groupSize : (M - mBase);
            DTYPE mean = sum / static_cast<DTYPE>(valid);
            DTYPE var = (sqsum / static_cast<DTYPE>(valid)) - mean * mean;
            if (var < 0.0f) var = 0.0f;
            DTYPE rstd = 1.0f / sqrtf(var + eps);

            // Second pass: GroupNorm affine + Swish + Multiply + Swish, write to GM
            for (uint32_t i = 0; i < valid; ++i) {
                uint32_t mIdx = mBase + i;
                DTYPE v = yLocal.GetPhyAddr()[i];
                // GroupNorm
                v = (v - mean) * rstd;
                v = v * gammaPtr[mIdx] + betaPtr[mIdx];
                // Swish
                DTYPE s1 = sigmoidf(v);
                v = v * s1;
                // Multiply
                v = v * mulPtr[mIdx];
                // Swish again
                DTYPE s2 = sigmoidf(v);
                v = v * s2;
                zPtr[n * M + mIdx] = v;
            }

            bufQ.FreeTensor(yLocal);
        }
    }
}

    const at::Tensor& w,

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized fused model: GEMM -> GroupNorm -> Swish -> Multiply -> Swish
    implemented by custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, num_groups, multiply_weight_shape):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features)
        self.multiply_weight = torch.nn.Parameter(torch.randn(multiply_weight_shape))

    def forward(self, x):
        return custom_ops_lib.gemm_group_norm_swish_multiply_swish_custom(
            x,
            self.gemm.weight,       # [out_features, in_features]
            self.gemm.bias,         # [out_features]
            self.group_norm.weight, # [out_features]
            self.group_norm.bias,   # [out_features]
            self.multiply_weight,   # [out_features]
            self.group_norm.num_groups,
            self.group_norm.eps
        )
