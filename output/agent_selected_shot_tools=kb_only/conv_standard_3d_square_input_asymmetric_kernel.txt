kernel_src="""
#include "kernel_operator.h"

class KernelConv3D {
public:
    __aicore__ inline KernelConv3D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                 uint32_t B, uint32_t Cin, uint32_t Cout,
                                 uint32_t W, uint32_t H, uint32_t D,
                                 uint32_t kW, uint32_t kH, uint32_t kD,
                                 uint32_t Wo, uint32_t Ho, uint32_t Do)
    {
        xPtr = reinterpret_cast<__gm__ float*>(x);
        wPtr = reinterpret_cast<__gm__ float*>(w);
        yPtr = reinterpret_cast<__gm__ float*>(y);

        this->B = B; this->Cin = Cin; this->Cout = Cout;
        this->W = W; this->H = H; this->D = D;
        this->kW = kW; this->kH = kH; this->kD = kD;
        this->Wo = Wo; this->Ho = Ho; this->Do = Do;

        totalOut = static_cast<uint64_t>(B) * Cout * Wo * Ho * Do;

        uint64_t blocks = static_cast<uint64_t>(AscendC::GetBlockNum());
        perBlock = (totalOut + blocks - 1ull) / blocks;
        startIdx = perBlock * static_cast<uint64_t>(AscendC::GetBlockIdx());
        endIdx = startIdx + perBlock;
        if (endIdx > totalOut) endIdx = totalOut;
    }

    __aicore__ inline void Process()
    {
        for (uint64_t idx = startIdx; idx < endIdx; ++idx) {
            // Decode linear index into [b, oc, wo, ho, do]
            uint64_t t = idx;

            uint32_t do_idx = static_cast<uint32_t>(t % Do);  t /= Do;
            uint32_t ho_idx = static_cast<uint32_t>(t % Ho);  t /= Ho;
            uint32_t wo_idx = static_cast<uint32_t>(t % Wo);  t /= Wo;
            uint32_t oc     = static_cast<uint32_t>(t % Cout); t /= Cout;
            uint32_t b      = static_cast<uint32_t>(t);

            float acc = 0.0f;

            for (uint32_t c = 0; c < Cin; ++c) {
                for (uint32_t kz = 0; kz < kD; ++kz) {
                    for (uint32_t ky = 0; ky < kH; ++ky) {
                        for (uint32_t kx = 0; kx < kW; ++kx) {
                            uint32_t in_w = wo_idx + kx;
                            uint32_t in_h = ho_idx + ky;
                            uint32_t in_d = do_idx + kz;

                            uint64_t x_off = (((((uint64_t)b * Cin + c) * W + in_w) * H + in_h) * D + in_d);
                            uint64_t w_off = (((((uint64_t)oc * Cin + c) * kW + kx) * kH + ky) * kD + kz);

                            float xv = xPtr[x_off];
                            float wv = wPtr[w_off];
                            acc += xv * wv;
                        }
                    }
                }
            }

            uint64_t y_off = (((((uint64_t)b * Cout + oc) * Wo + wo_idx) * Ho + ho_idx) * Do + do_idx);
            yPtr[y_off] = acc;
        }
    }

private:
    __gm__ float* xPtr;
    __gm__ float* wPtr;
    __gm__ float* yPtr;

    uint32_t B, Cin, Cout;
    uint32_t W, H, D;
    uint32_t kW, kH, kD;
    uint32_t Wo, Ho, Do;

    uint64_t totalOut;
    uint64_t perBlock;
    uint64_t startIdx;
    uint64_t endIdx;
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3D op;
    op.Init(x, w, y,
            tiling_data.B(),
            tiling_data.Cin(),
            tiling_data.Cout(),
            tiling_data.W(),
            tiling_data.H(),
            tiling_data.D(),
            tiling_data.kW(),
            tiling_data.kH(),
            tiling_data.kD(),
            tiling_data.Wo(),
            tiling_data.Ho(),
            tiling_data.Do());
    op.Process();
}
