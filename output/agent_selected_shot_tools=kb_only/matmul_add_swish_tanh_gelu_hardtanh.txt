kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelMatmulAddSwishTanhGeluHardtanh {
public:
    __aicore__ inline KernelMatmulAddSwishTanhGeluHardtanh() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        addGm.SetGlobalBuffer((__gm__ DTYPE *)add + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(inQueueAdd, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> addLocal = inQueueAdd.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(addLocal, addGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueAdd.EnQue(addLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE> xLocal  = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> addLocal = inQueueAdd.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal  = outQueueY.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> tmpLocal = tmpQueue.AllocTensor<DTYPE>();

        // y = x + add
        AscendC::Add(yLocal, xLocal, addLocal, this->tileLength);

        // Swish approximation: y = y * sigmoid(y)
        AscendC::Sigmoid(tmpLocal, yLocal, this->tileLength);
        AscendC::Mul(yLocal, yLocal, tmpLocal, this->tileLength);

        // Tanh
        AscendC::Tanh(tmpLocal, yLocal, this->tileLength);

        // GELU
        AscendC::Gelu(yLocal, tmpLocal, this->tileLength);

        // Hardtanh clamp to [-1, 1]
        AscendC::Maxs(tmpLocal, yLocal, (DTYPE)(-1.0f), this->tileLength);
        AscendC::Mins(yLocal, tmpLocal, (DTYPE)(1.0f), this->tileLength);

        outQueueY.EnQue<DTYPE>(yLocal);
        tmpQueue.FreeTensor(tmpLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueAdd.FreeTensor(addLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX, inQueueAdd;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY, tmpQueue;
    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> addGm;
    AscendC::GlobalTensor<DTYPE> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void matmul_add_swish_tanh_gelu_hardtanh_custom(GM_ADDR x, GM_ADDR add, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulAddSwishTanhGeluHardtanh op;
    op.Init(x, add, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom AscendC kernel to optimize post-linear pipeline: (x + add) -> Swish -> Tanh -> GELU -> Hardtanh.
    The Linear (matmul + bias) is kept from PyTorch for weight handling, then fused kernel applies the rest.
    """
    def __init__(self, in_features, out_features, add_value_shape):
        super(ModelNew, self).__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.add_value = torch.nn.Parameter(torch.randn(add_value_shape))

    def forward(self, x):
        # Matmul + bias via nn.Linear
        x = self.matmul(x)
        # Expand add_value to match x shape for element-wise fused kernel
        add_expanded = self.add_value.unsqueeze(0).expand_as(x).contiguous()
        # Fused element-wise: (x + add) -> Swish -> Tanh -> GELU -> Hardtanh
        return custom_ops_lib.matmul_add_swish_tanh_gelu_hardtanh_custom(x, add_expanded)
