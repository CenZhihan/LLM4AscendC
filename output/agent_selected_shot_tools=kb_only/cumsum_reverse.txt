kernel_src="""
#include "kernel_operator.h"

template <typename T>
class KernelCumsumReverse {
public:
    __aicore__ inline KernelCumsumReverse() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outter, uint32_t inner)
    {
        this->outter = outter;
        this->inner = inner;

        srcGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(x), outter * inner);
        dstGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ T *>(y), outter * inner);

        pipe.InitBuffer(inQueueX, 1, outter * inner * sizeof(T));
        pipe.InitBuffer(outQueueY, 1, outter * inner * sizeof(T));
        // tmp for intermediate cumsum result
        pipe.InitBuffer(tmpQueue, 1, outter * inner * sizeof(T));
        // lastRow buffer required by CumSum API even if not used (set to inner length)
        pipe.InitBuffer(lastRowQueue, 1, inner * sizeof(T));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<T> srcLocal = inQueueX.AllocTensor<T>();
        AscendC::DataCopy(srcLocal, srcGlobal, outter * inner);
        inQueueX.EnQue(srcLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<T> srcLocal = inQueueX.DeQue<T>();
        AscendC::LocalTensor<T> dstLocal = outQueueY.AllocTensor<T>();          // used as reversed input and final output
        AscendC::LocalTensor<T> tmpCum   = tmpQueue.AllocTensor<T>();           // cumsum(intermediate)
        AscendC::LocalTensor<T> lastRow  = lastRowQueue.AllocTensor<T>();       // dummy buffer for CumSum

        // Reverse along last axis (dim = 1 in 2D [outter, inner])
        for (uint32_t i = 0; i < outter; ++i) {
            uint32_t rowBase = i * inner;
            for (uint32_t j = 0; j < inner; ++j) {
                dstLocal[rowBase + j] = srcLocal[rowBase + (inner - 1 - j)];
            }
        }

        // CumSum along last axis on reversed data
        static constexpr AscendC::CumSumConfig cfg{true, false, false}; // isLastAxis=true, outputLastRow=false
        const AscendC::CumSumInfo info{outter, inner};
        AscendC::CumSum<T, cfg>(tmpCum, lastRow, dstLocal, info);

        // Reverse back to obtain reverse-cumsum result
        for (uint32_t i = 0; i < outter; ++i) {
            uint32_t rowBase = i * inner;
            for (uint32_t j = 0; j < inner; ++j) {
                dstLocal[rowBase + j] = tmpCum[rowBase + (inner - 1 - j)];
            }
        }

        outQueueY.EnQue<T>(dstLocal);
        inQueueX.FreeTensor(srcLocal);
        tmpQueue.FreeTensor(tmpCum);
        lastRowQueue.FreeTensor(lastRow);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<T> dstLocal = outQueueY.DeQue<T>();
        AscendC::DataCopy(dstGlobal, dstLocal, outter * inner);
        outQueueY.FreeTensor(dstLocal);
    }

private:
    AscendC::GlobalTensor<T> srcGlobal;
    AscendC::GlobalTensor<T> dstGlobal;

    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1>  inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> lastRowQueue;

    uint32_t outter{1};
    uint32_t inner{1};
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverse<float> op;
    op.Init(x, y, tiling_data.outter, tiling_data.inner);
    op.Process();
}
"""
