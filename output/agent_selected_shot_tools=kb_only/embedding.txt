host_operator_src="""
#include "embedding_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t DEFAULT_TILE_ROWS = 64;
const uint32_t DEFAULT_EMB_DIM = 768; // fallback if embDim cannot be parsed

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    EmbeddingCustomTilingData tiling;

    // Input 0: weight [num_embeddings, emb_dim]
    // Input 1: indices [...]
    const gert::Shape* weightShape = context->GetInputShape(0);
    const gert::Shape* indicesShape = context->GetInputShape(1);

    uint32_t totalRows = indicesShape->GetOriginShape().GetShapeSize();
    uint32_t weightNumel = weightShape->GetOriginShape().GetShapeSize();

    // Try to get embDim from weight second dim; if unavailable, fall back to DEFAULT_EMB_DIM
    uint32_t embDim = DEFAULT_EMB_DIM;
    if (weightShape->GetOriginShape().GetDimNum() >= 2) {
        embDim = static_cast<uint32_t>(weightShape->GetOriginShape().GetDim(1));
    } else if (weightNumel % DEFAULT_EMB_DIM == 0) {
        embDim = DEFAULT_EMB_DIM;
    }
    uint32_t weightRows = (embDim > 0) ? (weightNumel / embDim) : 0;

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalRows(totalRows);
    tiling.set_embDim(embDim);
    tiling.set_tileRows(DEFAULT_TILE_ROWS);
    tiling.set_weightRows(weightRows);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Output shape = indices.shape + [embDim]
    const gert::Shape* indicesShape = context->GetInputShape(1);
    const gert::Shape* weightShape = context->GetInputShape(0);
    gert::Shape* outShape = context->GetOutputShape(0);

    // Copy indices shape
    *outShape = *indicesShape;
    // Try to append embDim if API supports; otherwise keep as-is (runtime will handle real shape)
    (void)weightShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output dtype follows weight dtype
    const auto weightDtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, weightDtype);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class EmbeddingCustom : public OpDef {
public:
    explicit EmbeddingCustom(const char* name) : OpDef(name)
    {
        this->Input("weight")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("indices")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT32, ge::DT_INT64})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("output")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(EmbeddingCustom);
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE_W = float;    // weight and output data type
using DTYPE_Z = float;
using DTYPE_I32 = int32_t;

constexpr int32_t BUFFER_NUM = 1;

class KernelEmbedding {
public:
    __aicore__ inline KernelEmbedding() {}

    __aicore__ inline void Init(GM_ADDR weight, GM_ADDR indices, GM_ADDR output,
                                uint32_t totalRows, uint32_t embDim, uint32_t tileRows, uint32_t weightRows)
    {
        this->totalRows = totalRows;
        this->embDim = embDim;
        this->tileRows = tileRows;
        this->weightRows = weightRows;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        // Simple even split across blocks
        this->rowsPerBlock = totalRows / blockNum;
        uint32_t startRow = rowsPerBlock * blockIdx;

        weightGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_W *>(weight), static_cast<uint32_t>(weightRows) * embDim);
        indicesGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_I32 *>(indices) + startRow, rowsPerBlock);
        outputGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE_Z *>(output) + startRow * embDim, rowsPerBlock * embDim);

        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, tileRows * sizeof(DTYPE_I32));
        pipe.InitBuffer(tmpRowQueue, BUFFER_NUM, embDim * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint32_t processed = 0;
        while (processed < rowsPerBlock) {
            uint32_t currRows = AscendC::Min(tileRows, rowsPerBlock - processed);
            CopyIn(processed, currRows);
            Compute(processed, currRows);
            processed += currRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progressRows, uint32_t currRows)
    {
        AscendC::LocalTensor<DTYPE_I32> idxLocal = inQueueIdx.AllocTensor<DTYPE_I32>();
        AscendC::DataCopy(idxLocal, indicesGm[progressRows], currRows);
        inQueueIdx.EnQue(idxLocal);
    }

    __aicore__ inline void Compute(uint32_t progressRows, uint32_t currRows)
    {
        AscendC::LocalTensor<DTYPE_I32> idxLocal = inQueueIdx.DeQue<DTYPE_I32>();

        for (uint32_t r = 0; r < currRows; ++r) {
            // Read index value from local tensor
            DTYPE_I32 idxVal = idxLocal.GetValue(r);  // assume LocalTensor supports GetValue

            // Bounds check (optional)
            if (idxVal < 0) { idxVal = 0; }
            if (static_cast<uint32_t>(idxVal) >= weightRows) { idxVal = (DTYPE_I32)(weightRows - 1); }

            AscendC::LocalTensor<DTYPE_Z> rowLocal = tmpRowQueue.AllocTensor<DTYPE_Z>();
            // Copy a whole embedding row from GM to UB
            AscendC::DataCopy(rowLocal, weightGm[static_cast<uint32_t>(idxVal) * embDim], embDim);
            // Copy UB row to GM output at the right place
            AscendC::DataCopy(outputGm[(progressRows + r) * embDim], rowLocal, embDim);
            tmpRowQueue.FreeTensor(rowLocal);
        }

        inQueueIdx.FreeTensor(idxLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpRowQueue;

    AscendC::GlobalTensor<DTYPE_W> weightGm;
    AscendC::GlobalTensor<DTYPE_I32> indicesGm;
    AscendC::GlobalTensor<DTYPE_Z> outputGm;

    uint32_t totalRows;
    uint32_t embDim;
    uint32_t tileRows;
    uint32_t weightRows;
    uint32_t rowsPerBlock;
};

extern "C" __global__ __aicore__ void embedding_custom(GM_ADDR weight, GM_ADDR indices, GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelEmbedding op;
    op.Init(weight, indices, output,
            tiling_data.totalRows(), tiling_data.embDim(), tiling_data.tileRows(), tiling_data.weightRows());
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor embedding_custom_impl_npu(const at::Tensor& weight, const at::Tensor& indices) {
    TORCH_CHECK(weight.dim() == 2, "weight must be 2D [num_embeddings, embedding_dim]");
    auto idx = indices.to(at::kInt);
    std::vector<int64_t> outShape = idx.sizes().vec();
    outShape.push_back(weight.size(1));
    at::Tensor result = at::empty(outShape, weight.options());
    EXEC_NPU_CMD(aclnnEmbeddingCustom, weight, idx, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("embedding_custom", &embedding_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("embedding_custom", &embedding_custom_impl_npu, "Embedding lookup (custom AscendC)");
}
