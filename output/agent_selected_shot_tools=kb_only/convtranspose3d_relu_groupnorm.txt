kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvtranspose3dReluGroupnorm {
public:
    __aicore__ inline KernelConvtranspose3dReluGroupnorm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mean, GM_ADDR rstd, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        int32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ float*)x + blockOffset, this->blockLength);
        meanGm.SetGlobalBuffer((__gm__ float*)mean + blockOffset, this->blockLength);
        rstdGm.SetGlobalBuffer((__gm__ float*)rstd + blockOffset, this->blockLength);
        gammaGm.SetGlobalBuffer((__gm__ float*)gamma + blockOffset, this->blockLength);
        betaGm.SetGlobalBuffer((__gm__ float*)beta + blockOffset, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ float*)y + blockOffset, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueMean, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueRstd, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueGamma, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueBeta, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<float> meanLocal = inQueueMean.AllocTensor<float>();
        AscendC::LocalTensor<float> rstdLocal = inQueueRstd.AllocTensor<float>();
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma.AllocTensor<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueBeta.AllocTensor<float>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(meanLocal, meanGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(rstdLocal, rstdGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gammaLocal, gammaGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(betaLocal, betaGm[progress * this->tileLength], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueMean.EnQue(meanLocal);
        inQueueRstd.EnQue(rstdLocal);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> meanLocal = inQueueMean.DeQue<float>();
        AscendC::LocalTensor<float> rstdLocal = inQueueRstd.DeQue<float>();
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma.DeQue<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueBeta.DeQue<float>();

        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        // yLocal = ReLU(xLocal)
        AscendC::Relu(yLocal, xLocal, this->tileLength);
        // yLocal = yLocal - mean
        AscendC::Sub(yLocal, yLocal, meanLocal, this->tileLength);
        // yLocal = yLocal * rstd
        AscendC::Mul(yLocal, yLocal, rstdLocal, this->tileLength);
        // yLocal = yLocal * gamma
        AscendC::Mul(yLocal, yLocal, gammaLocal, this->tileLength);
        // yLocal = yLocal + beta
        AscendC::Add(yLocal, yLocal, betaLocal, this->tileLength);

        outQueueY.EnQue<float>(yLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueMean.FreeTensor(meanLocal);
        inQueueRstd.FreeTensor(rstdLocal);
        inQueueGamma.FreeTensor(gammaLocal);
        inQueueBeta.FreeTensor(betaLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMean;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueRstd;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueGamma;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBeta;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> meanGm;
    AscendC::GlobalTensor<float> rstdGm;
    AscendC::GlobalTensor<float> gammaGm;
    AscendC::GlobalTensor<float> betaGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void convtranspose3d_relu_groupnorm_custom(
    GM_ADDR x, GM_ADDR mean, GM_ADDR rstd, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvtranspose3dReluGroupnorm op;
    op.Init(x, mean, rstd, gamma, beta, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    使用自定义AscendC核将 ReLU + GroupNorm 融合，ConvTranspose3d 仍复用原生实现。
    自定义核完成: y = ((ReLU(x) - mean) * rstd) * gamma + beta
    其中 mean 与 rstd 为按批次/分组统计并广播到与x相同形状的张量，gamma/beta 为每通道仿射参数同样广播。
    """
    def __init__(self, in_channels, out_channels, kernel_size, groups, bias=False):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, bias=bias)
        self.groups = groups
        # 直接复用GroupNorm的可学习参数
        self.weight = torch.nn.Parameter(torch.ones(out_channels))  # gamma
        self.bias = torch.nn.Parameter(torch.zeros(out_channels))   # beta

    def forward(self, x):
        # 先做反卷积
        y = self.conv_transpose(x)  # [N, C, D, H, W]
        N, C, D, H, W = y.shape
        G = self.groups
        eps = 1e-5

        # 计算每个样本、每个分组的均值与rstd，并广播到与y相同的形状
        y_reshaped = y.view(N, G, C // G, D, H, W)
        mean = y_reshaped.mean(dim=(2, 3, 4, 5), keepdim=True)                        # [N, G, 1, 1, 1, 1]
        var = ((y_reshaped - mean) ** 2).mean(dim=(2, 3, 4, 5), keepdim=True)         # [N, G, 1, 1, 1, 1]
        rstd = torch.rsqrt(var + eps)                                                 # [N, G, 1, 1, 1, 1]

        # 广播到 [N, C, D, H, W]
        mean_expanded = mean.expand(N, G, C // G, D, H, W).contiguous().view(N, C, D, H, W)
        rstd_expanded = rstd.expand(N, G, C // G, D, H, W).contiguous().view(N, C, D, H, W)

        # gamma/beta 按通道广播到 [N, C, D, H, W]
        gamma_expanded = self.weight.view(1, C, 1, 1, 1).expand(N, C, D, H, W).contiguous()
        beta_expanded = self.bias.view(1, C, 1, 1, 1).expand(N, C, D, H, W).contiguous()

        # 调用自定义AscendC融合核完成 ReLU + GroupNorm + affine
        out = custom_ops_lib.convtranspose3d_relu_groupnorm_custom(y.contiguous(),
                                                                   mean_expanded,
                                                                   rstd_expanded,
                                                                   gamma_expanded,
                                                                   beta_expanded)
        return out
'''
