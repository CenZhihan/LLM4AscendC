kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelConvTransposed2D {
public:
    __aicore__ inline KernelConvTransposed2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t hin, uint32_t win,
                                uint32_t hout, uint32_t wout,
                                uint32_t k, uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t cPerCore) {
        this->N = n; this->Cin = cin; this->Cout = cout;
        this->Hin = hin; this->Win = win;
        this->Hout = hout; this->Wout = wout;
        this->K = k; this->Stride = stride; this->Padding = padding; this->Dilation = dilation;
        this->CPerCore = cPerCore;

        uint32_t xSize = n * cin * hin * win;
        uint32_t wSize = cin * cout * k * k;
        uint32_t ySize = n * cout * hout * wout;

        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(x), xSize);
        wGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(w), wSize);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(y), ySize);

        pipe.InitBuffer(inQueueX, 1, sizeof(DTYPE));
        pipe.InitBuffer(inQueueW, 1, sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, 1, sizeof(DTYPE));
        pipe.InitBuffer(tmpQueue, 1, sizeof(DTYPE));
    }

    __aicore__ inline void Process() {
        uint32_t coreIdx = AscendC::GetBlockIdx();
        uint32_t startCout = coreIdx * CPerCore;
        uint32_t endCout = startCout + CPerCore;
        if (startCout >= Cout) { return; }
        if (endCout > Cout) endCout = Cout;

        for (uint32_t nIdx = 0; nIdx < N; ++nIdx) {
            for (uint32_t co = startCout; co < endCout; ++co) {
                for (uint32_t oh = 0; oh < Hout; ++oh) {
                    for (uint32_t ow = 0; ow < Wout; ++ow) {
                        // acc = 0
                        AscendC::LocalTensor<DTYPE> accLocal = tmpQueue.AllocTensor<DTYPE>();
                        // 将accLocal置零：用乘加的方式累加（首次累加前创建一个0值）
                        // 这里采用将0从全局偏移0处复制的方式，要求调用方保证内存安全；若不可行，可使用算子Duplicate替代。
                        // 简化：先写入0，再按需累加
                        SetLocalZero(accLocal);

                        bool has_valid = false;
                        // 遍历 cin, kh, kw，计算有效 (hin, win)
                        for (uint32_t ci = 0; ci < Cin; ++ci) {
                            for (uint32_t kh = 0; kh < K; ++kh) {
                                int32_t tmp_h = static_cast<int32_t>(oh) + static_cast<int32_t>(Padding) - static_cast<int32_t>(kh * Dilation);
                                if (tmp_h % static_cast<int32_t>(Stride) != 0) continue;
                                int32_t ih = tmp_h / static_cast<int32_t>(Stride);
                                if (ih < 0 || ih >= static_cast<int32_t>(Hin)) continue;

                                for (uint32_t kw = 0; kw < K; ++kw) {
                                    int32_t tmp_w = static_cast<int32_t>(ow) + static_cast<int32_t>(Padding) - static_cast<int32_t>(kw * Dilation);
                                    if (tmp_w % static_cast<int32_t>(Stride) != 0) continue;
                                    int32_t iw = tmp_w / static_cast<int32_t>(Stride);
                                    if (iw < 0 || iw >= static_cast<int32_t>(Win)) continue;

                                    // 取 x[n, ci, ih, iw]
                                    uint32_t xOff = ((nIdx * Cin + ci) * Hin + static_cast<uint32_t>(ih)) * Win + static_cast<uint32_t>(iw);
                                    AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
                                    AscendC::DataCopy(xLocal, xGm[xOff], 1);

                                    // 取 w[ci, co, kh, kw]
                                    uint32_t wOff = ((ci * Cout + co) * K + kh) * K + kw;
                                    AscendC::LocalTensor<DTYPE> wLocal = inQueueW.AllocTensor<DTYPE>();
                                    AscendC::DataCopy(wLocal, wGm[wOff], 1);

                                    // temp = x * w
                                    AscendC::LocalTensor<DTYPE> tmpLocal = tmpQueue.AllocTensor<DTYPE>();
                                    AscendC::Mul(tmpLocal, xLocal, wLocal, 1);

                                    if (!has_valid) {
                                        // acc = temp
                                        AscendC::DataCopy(accLocal, tmpLocal, 1);
                                        has_valid = true;
                                    } else {
                                        // acc += temp
                                        AscendC::Add(accLocal, accLocal, tmpLocal, 1);
                                    }

                                    inQueueX.FreeTensor(xLocal);
                                    inQueueW.FreeTensor(wLocal);
                                    tmpQueue.FreeTensor(tmpLocal);
                                }
                            }
                        }

                        // 写回 y[n, co, oh, ow]
                        uint32_t yOff = ((nIdx * Cout + co) * Hout + oh) * Wout + ow;
                        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
                        if (has_valid) {
                            AscendC::DataCopy(yLocal, accLocal, 1);
                        } else {
                            // 无有效贡献则写0
                            SetLocalZero(yLocal);
                        }
                        AscendC::DataCopy(yGm[yOff], yLocal, 1);
                        outQueueY.FreeTensor(yLocal);
                        tmpQueue.FreeTensor(accLocal);
                    }
                }
            }
        }
    }

private:
    __aicore__ inline void SetLocalZero(AscendC::LocalTensor<DTYPE>& t) {
        // 使用加法将局部张量清零：t = t - t
        // 先分配一个临时零张量为t，再进行一次减法，确保为0
        AscendC::LocalTensor<DTYPE> zeroLocal = tmpQueue.AllocTensor<DTYPE>();
        // 采用自身相减得到0：zeroLocal = zeroLocal - zeroLocal
        AscendC::Sub(zeroLocal, zeroLocal, zeroLocal, 1);
        AscendC::DataCopy(t, zeroLocal, 1);
        tmpQueue.FreeTensor(zeroLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  inQueueW;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t N, Cin, Cout;
    uint32_t Hin, Win, Hout, Wout;
    uint32_t K, Stride, Padding, Dilation;
    uint32_t CPerCore;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(ConvTransposed2dAsymmetricInputSquareKernelDilatedPaddedStridedCustomTilingData, tiling);
    KernelConvTransposed2D op;
    op.Init(x, w, y,
            tiling_data.n(), tiling_data.cin(), tiling_data.cout(),
            tiling_data.hin(), tiling_data.win(),
            tiling_data.hout(), tiling_data.wout(),
            tiling_data.k(), tiling_data.stride(), tiling_data.padding(), tiling_data.dilation(),
            tiling_data.cPerCore());
    op.Process();
}

class ModelNew(torch.nn.Module):
    """
    使用自定义 AscendC 算子执行 2D 反卷积（转置卷积），支持方形卷积核、stride / padding / dilation。
    权重作为本模块参数保存，无偏置。

        self.in_channels = in_channels

        w = torch.empty(in_channels, out_channels, kernel_size, kernel_size)
        torch.nn.init.kaiming_uniform_(w, a=5**0.5)
        self.weight = torch.nn.Parameter(w)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom(
