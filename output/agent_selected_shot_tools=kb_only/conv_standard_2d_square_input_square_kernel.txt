kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelConv2dDirect {
public:
    __aicore__ inline KernelConv2dDirect() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t OC, uint32_t KH, uint32_t KW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t OH, uint32_t OW)
    {
        this->N = N; this->C = C; this->H = H; this->W = W;
        this->OC = OC; this->KH = KH; this->KW = KW;
        this->strideH = strideH; this->strideW = strideW;
        this->padH = padH; this->padW = padW;
        this->OH = OH; this->OW = OW;

        // set global buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, (uint64_t)N*C*H*W);
        wGm.SetGlobalBuffer((__gm__ DTYPE*)w, (uint64_t)OC*C*KH*KW);
        bGm.SetGlobalBuffer((__gm__ DTYPE*)b, (uint64_t)OC);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, (uint64_t)N*OC*OH*OW);

        // local buffers: one kernel slice and one scalar tmp for load/store
        uint32_t kernelSize = C * KH * KW;
        pipe.InitBuffer(inQueueKernel, 1, kernelSize * sizeof(DTYPE));
        pipe.InitBuffer(inQueueScalar, 1, 1 * sizeof(DTYPE));
        pipe.InitBuffer(outQueueScalar, 1, 1 * sizeof(DTYPE));
        this->kernelSize = kernelSize;
    }

    __aicore__ inline void Process()
    {
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oc = 0; oc < OC; ++oc) {
                // Load weight kernel for this oc into local
                AscendC::LocalTensor<DTYPE> wLocal = inQueueKernel.AllocTensor<DTYPE>();
                uint64_t wOffset = (uint64_t)oc * kernelSize;
                AscendC::DataCopy(wLocal, wGm[wOffset], kernelSize);

                // Load bias for this oc
                DTYPE biasVal = 0.0f;
                {
                    AscendC::LocalTensor<DTYPE> tmpB = inQueueScalar.AllocTensor<DTYPE>();
                    AscendC::DataCopy(tmpB, bGm[oc], 1);
                    biasVal = tmpB.GetValue(0);
                    inQueueScalar.FreeTensor(tmpB);
                }

                for (uint32_t oh = 0; oh < OH; ++oh) {
                    int64_t ihBase = (int64_t)oh * (int64_t)strideH - (int64_t)padH;
                    for (uint32_t ow = 0; ow < OW; ++ow) {
                        int64_t iwBase = (int64_t)ow * (int64_t)strideW - (int64_t)padW;
                        DTYPE sumVal = biasVal;

                        // Direct convolution accumulation
                        uint32_t wIdxBaseC = 0;
                        for (uint32_t c = 0; c < C; ++c) {
                            uint32_t wIdxBaseR = wIdxBaseC;
                            for (uint32_t kh = 0; kh < KH; ++kh) {
                                int64_t ih = ihBase + (int64_t)kh;
                                uint32_t wIdxBaseS = wIdxBaseR;
                                for (uint32_t kw = 0; kw < KW; ++kw) {
                                    int64_t iw = iwBase + (int64_t)kw;
                                    DTYPE inVal = 0.0f;
                                    if (ih >= 0 && ih < (int64_t)H && iw >= 0 && iw < (int64_t)W) {
                                        uint64_t xOffset = (((uint64_t)n * C + c) * H + (uint64_t)ih) * W + (uint64_t)iw;
                                        AscendC::LocalTensor<DTYPE> tmpIn = inQueueScalar.AllocTensor<DTYPE>();
                                        AscendC::DataCopy(tmpIn, xGm[xOffset], 1);
                                        inVal = tmpIn.GetValue(0);
                                        inQueueScalar.FreeTensor(tmpIn);
                                    }
                                    DTYPE wVal = wLocal.GetValue(wIdxBaseS);
                                    sumVal += inVal * wVal;
                                    ++wIdxBaseS;
                                }
                                wIdxBaseR += KW;
                            }
                            wIdxBaseC += KH * KW;
                        }

                        // store output
                        uint64_t yOffset = ((((uint64_t)n) * OC + oc) * OH + oh) * OW + ow;
                        AscendC::LocalTensor<DTYPE> tmpOut = outQueueScalar.AllocTensor<DTYPE>();
                        tmpOut.SetValue(0, sumVal);
                        AscendC::DataCopy(yGm[yOffset], tmpOut, 1);
                        outQueueScalar.FreeTensor(tmpOut);
                    }
                }

                inQueueKernel.FreeTensor(wLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueKernel;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueScalar;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueScalar;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t N, C, H, W;
    uint32_t OC, KH, KW;
    uint32_t strideH, strideW;
    uint32_t padH, padW;
    uint32_t OH, OW;
    uint32_t kernelSize;
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv2dDirect op;
    op.Init(x, w, b, y,
            tilingData.N, tilingData.C, tilingData.H, tilingData.W,
            tilingData.OC, tilingData.KH, tilingData.KW,
            tilingData.strideH, tilingData.strideW,
            tilingData.padH, tilingData.padW,
            tilingData.OH, tilingData.OW);
    op.Process();
}
