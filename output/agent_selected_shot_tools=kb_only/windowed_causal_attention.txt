kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 1;

template <typename T>
__aicore__ inline uint32_t AlignUpTo(const T x, const T a) {
    return static_cast<uint32_t>(((x + a - 1) / a) * a);
}

class KernelWindowedCausalAttention {
public:
    __aicore__ inline KernelWindowedCausalAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR w, GM_ADDR out,
                                uint32_t B, uint32_t H, uint32_t N, uint32_t Dh,
                                uint32_t tileNum)
    {
        this->B = B; this->H = H; this->N = N; this->Dh = Dh;
        this->tileNum = tileNum;
        this->tokensPerBlock = (static_cast<uint64_t>(B) * H * N) / AscendC::GetBlockNum();

        // Set GM buffers; element count = B*H*N*Dh per block
        uint64_t blockTokenBase = this->tokensPerBlock * AscendC::GetBlockIdx();
        uint64_t blockElemBase = blockTokenBase * Dh;

        qGm.SetGlobalBuffer((__gm__ float*)q + blockElemBase, this->tokensPerBlock * Dh);
        kGm.SetGlobalBuffer((__gm__ float*)k + blockElemBase, this->tokensPerBlock * Dh);
        vGm.SetGlobalBuffer((__gm__ float*)v + blockElemBase, this->tokensPerBlock * Dh);
        outGm.SetGlobalBuffer((__gm__ float*)out + blockElemBase, this->tokensPerBlock * Dh);

        // Read window size (int32 scalar) from GM
        pipe.InitBuffer(intBuf, 1, sizeof(int32_t));
        AscendC::LocalTensor<int32_t> wLocal = intBuf.AllocTensor<int32_t>();
        AscendC::GlobalTensor<int32_t> wGm;
        wGm.SetGlobalBuffer((__gm__ int32_t*)w, 1);
        AscendC::DataCopy(wLocal, wGm, 1);
        intBuf.EnQue(wLocal);
        AscendC::LocalTensor<int32_t> wLocal2 = intBuf.DeQue<int32_t>();
        this->W = static_cast<uint32_t>(wLocal2.GetValue(0));
        intBuf.FreeTensor(wLocal2);

        // UB buffers
        pipe.InitBuffer(qBuf, BUFFER_NUM, this->Dh * sizeof(float));
        pipe.InitBuffer(kBuf, BUFFER_NUM, this->Dh * sizeof(float));
        pipe.InitBuffer(vBuf, BUFFER_NUM, this->Dh * sizeof(float));
        pipe.InitBuffer(outBuf, BUFFER_NUM, this->Dh * sizeof(float));

        // Max window length is W+1; pad last axis to 32-byte (8 floats) alignment
        uint32_t maxL = this->W + 1;
        paddedL = AlignUpTo<uint32_t>(maxL, 8);
        pipe.InitBuffer(scoresBuf, BUFFER_NUM, paddedL * sizeof(float));
        pipe.InitBuffer(workBuf, BUFFER_NUM, paddedL * sizeof(float)); // for exp/weights temp
    }

    __aicore__ inline void Process()
    {
        float invSqrtDh = 1.0f / sqrtf(static_cast<float>(this->Dh));

        for (uint64_t p = 0; p < this->tokensPerBlock; ++p) {
            // Map linear p in this block to (b,h,i)
            uint64_t g = p + this->tokensPerBlock * AscendC::GetBlockIdx();
            uint32_t tokensPerBH = this->N;
            uint32_t bh = static_cast<uint32_t>(g % (static_cast<uint64_t>(this->B) * this->H));
            uint32_t i = static_cast<uint32_t>(g % this->N);
            uint32_t h = static_cast<uint32_t>(bh % this->H);
            uint32_t b = static_cast<uint32_t>(bh / this->H);

            uint64_t baseBH = (static_cast<uint64_t>(b) * this->H + h) * this->N;

            // Load Q[b,h,i,:]
            AscendC::LocalTensor<float> qLocal = qBuf.AllocTensor<float>();
            uint64_t qOff = (baseBH + i) * this->Dh;
            AscendC::DataCopy(qLocal, qGm[qOff], this->Dh);
            qBuf.EnQue(qLocal);
            AscendC::LocalTensor<float> qVec = qBuf.DeQue<float>();

            // Prepare out accumulator
            AscendC::LocalTensor<float> outLocal = outBuf.AllocTensor<float>();
            float zero = 0.0f;
            AscendC::Duplicate(outLocal, zero, this->Dh);

            // Scores buffer
            AscendC::LocalTensor<float> scoresLocal = scoresBuf.AllocTensor<float>();
            scoresLocal.SetSize(paddedL); // in elements

            // Compute start and effective length
            uint32_t start = (i > this->W) ? (i - this->W) : 0U;
            uint32_t L = i - start + 1U; // inclusive of i

            // Compute scaled dot-product scores
            float neg_large = -1.0e9f;
            for (uint32_t t = 0; t < paddedL; ++t) {
                scoresLocal.SetValue(t, neg_large);
            }

            for (uint32_t t = 0; t < L; ++t) {
                uint32_t j = start + t; // key index
                // Load K[b,h,j,:]
                AscendC::LocalTensor<float> kLocal = kBuf.AllocTensor<float>();
                uint64_t kOff = (baseBH + j) * this->Dh;
                AscendC::DataCopy(kLocal, kGm[kOff], this->Dh);
                kBuf.EnQue(kLocal);
                AscendC::LocalTensor<float> kVec = kBuf.DeQue<float>();

                // dot = sum(q * k)
                float dot = 0.0f;
                for (uint32_t d = 0; d < this->Dh; ++d) {
                    float qv = qVec.GetValue(d);
                    float kv = kVec.GetValue(d);
                    dot += qv * kv;
                }
                dot *= invSqrtDh;
                scoresLocal.SetValue(t, dot);

                kBuf.FreeTensor(kVec);
            }

            // Softmax over first L elements (scoresLocal[0..L-1])
            // Find max
            float smax = -1.0e30f;
            for (uint32_t t = 0; t < L; ++t) {
                float v = scoresLocal.GetValue(t);
                if (v > smax) smax = v;
            }

            // Subtract max and exponentiate
            AscendC::LocalTensor<float> expLocal = workBuf.AllocTensor<float>();
            expLocal.SetSize(paddedL);
            for (uint32_t t = 0; t < L; ++t) {
                float v = scoresLocal.GetValue(t) - smax;
                // elementwise exp for scalar
                float ev = expf(v);
                expLocal.SetValue(t, ev);
            }
            for (uint32_t t = L; t < paddedL; ++t) {
                expLocal.SetValue(t, 0.0f);
            }

            // Sum exp
            float ssum = 0.0f;
            for (uint32_t t = 0; t < L; ++t) {
                ssum += expLocal.GetValue(t);
            }
            float invSum = ssum > 0.0f ? (1.0f / ssum) : 0.0f;

            // Compute weighted sum: out = sum_t weights[t] * V[b,h,start+t,:]
            for (uint32_t t = 0; t < L; ++t) {
                float w = expLocal.GetValue(t) * invSum;

                // Load V[b,h,j,:]
                AscendC::LocalTensor<float> vLocal = vBuf.AllocTensor<float>();
                uint64_t vOff = (baseBH + (start + t)) * this->Dh;
                AscendC::DataCopy(vLocal, vGm[vOff], this->Dh);
                vBuf.EnQue(vLocal);
                AscendC::LocalTensor<float> vVec = vBuf.DeQue<float>();

                // out += w * vVec
                for (uint32_t d = 0; d < this->Dh; ++d) {
                    float acc = outLocal.GetValue(d);
                    float vv  = vVec.GetValue(d);
                    acc += w * vv;
                    outLocal.SetValue(d, acc);
                }

                vBuf.FreeTensor(vVec);
            }

            // Write out
            AscendC::DataCopy(outGm[qOff], outLocal, this->Dh);

            // free locals
            workBuf.FreeTensor(expLocal);
            scoresBuf.FreeTensor(scoresLocal);
            outBuf.FreeTensor(outLocal);
            qBuf.FreeTensor(qVec);
        }
    }

private:
    AscendC::TPipe pipe;
    // GM
    AscendC::GlobalTensor<float> qGm, kGm, vGm, outGm;
    // Queues
    AscendC::TQue<AscendC::QuePosition::VECIN,  BUFFER_NUM> qBuf, kBuf, vBuf;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outBuf;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> scoresBuf, workBuf;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> intBuf;

    uint32_t B, H, N, Dh, W;
    uint32_t tileNum;
    uint64_t tokensPerBlock;
    uint32_t paddedL;
};

extern "C" __global__ __aicore__ void windowed_causal_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR w, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelWindowedCausalAttention op;
    op.Init(q, k, v, w, out,
            tiling_data.B, tiling_data.H, tiling_data.N, tiling_data.Dh,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import math
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    使用自定义AscendC核实现滑动窗口因果注意力（局部注意力）。
    接口期望输入Q/K/V为[B, H, N, d_h]布局，因此在调用前进行维度重排。
    """
    def __init__(self, d_model=1024, num_heads=16, window_size=256):
        super().__init__()
        self.q_proj = nn.Linear(d_model, d_model)
        self.k_proj = nn.Linear(d_model, d_model)
        self.v_proj = nn.Linear(d_model, d_model)
        self.out_proj = nn.Linear(d_model, d_model)
        self.num_heads = num_heads
        self.window_size = window_size

    def forward(self, x):
        B, N, D = x.shape
        H = self.num_heads
        d_h = D // H

        # 生成Q/K/V并转为[B, H, N, d_h]，以匹配自定义算子预期布局
        Q = self.q_proj(x).view(B, N, H, d_h).permute(0, 2, 1, 3).contiguous()
        K = self.k_proj(x).view(B, N, H, d_h).permute(0, 2, 1, 3).contiguous()
        V = self.v_proj(x).view(B, N, H, d_h).permute(0, 2, 1, 3).contiguous()

        # 调用自定义窗口注意力核：输出仍为[B, H, N, d_h]
        attn_out_bhnd = custom_ops_lib.windowed_causal_attention_custom(Q, K, V, self.window_size)

        # 还原回[B, N, D]
        attn_out = attn_out_bhnd.permute(0, 2, 1, 3).contiguous().view(B, N, D)
        return self.out_proj(attn_out)
'''
