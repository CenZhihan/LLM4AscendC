static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    AveragePooling1dCustomTilingData tiling;
    const gert::StorageShape* src_shape = context->GetInputShape(0);
    auto inShape = src_shape->GetOriginShape();
    // Expect shape [N, C, L]
    uint32_t N = static_cast<uint32_t>(inShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(inShape.GetDim(1));
    uint32_t L = static_cast<uint32_t>(inShape.GetDim(2));
    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    uint32_t kernel = *(attrs->GetAttrPointer<uint32_t>(0));
    uint32_t stride = *(attrs->GetAttrPointer<uint32_t>(1));
    uint32_t padding = *(attrs->GetAttrPointer<uint32_t>(2));

    // Output length = floor((L + 2*p - k)/s) + 1
    int64_t outLenCalc = (static_cast<int64_t>(L) + static_cast<int64_t>(2 * padding) - static_cast<int64_t>(kernel));
    int64_t outLen = outLenCalc >= 0 ? (outLenCalc / static_cast<int64_t>(stride)) + 1 : 0;
    if (outLen < 0) outLen = 0;

    uint32_t bc = N * C;
    tiling.set_bc(bc);
    tiling.set_inLength(L);
    tiling.set_outLength(static_cast<uint32_t>(outLen));
    tiling.set_kernelSize(kernel);
    tiling.set_stride(stride);
    tiling.set_padding(padding);

    // One group per block
    context->SetBlockDim(bc);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelAveragePooling1d {
public:
    __aicore__ inline KernelAveragePooling1d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t bc, uint32_t inLength, uint32_t outLength,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding)
    {
        this->bc = bc;
        this->inLength = inLength;
        this->outLength = outLength;
        this->kernelSize = kernelSize;
        this->stride = stride;
        this->padding = padding;

        uint32_t groupIdx = AscendC::GetBlockIdx();
        if (groupIdx >= bc) {
            return;
        }
        // Flatten [N, C, L] into bc groups along (N*C), each group length = inLength
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + groupIdx * inLength, inLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + groupIdx * outLength, outLength);
        pipe.InitBuffer(inQueue, BUFFER_NUM, inLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, outLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        uint32_t groupIdx = AscendC::GetBlockIdx();
        if (groupIdx >= bc) {
            return;
        }
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm, inLength);
        inQueue.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.AllocTensor<DTYPE_Y>();

        // Sliding window average with zero padding
        for (uint32_t j = 0; j < outLength; ++j) {
            float sum = 0.0f;
            int32_t start = static_cast<int32_t>(j * stride) - static_cast<int32_t>(padding);
            for (uint32_t t = 0; t < kernelSize; ++t) {
                int32_t idx = start + static_cast<int32_t>(t);
                float v = 0.0f;
                if (idx >= 0 && idx < static_cast<int32_t>(inLength)) {
                    v = static_cast<float>(xLocal[idx]);
                }
                sum += v;
            }
            yLocal[j] = static_cast<DTYPE_Y>(sum / static_cast<float>(kernelSize));
        }

        outQueue.EnQue<DTYPE_Y>(yLocal);
        inQueue.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm, yLocal, outLength);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t bc;
    uint32_t inLength;
    uint32_t outLength;
    uint32_t kernelSize;
    uint32_t stride;
    uint32_t padding;
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling1d op;
    op.Init(x, y,
            tiling_data.bc, tiling_data.inLength, tiling_data.outLength,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor average_pooling_1d_custom_impl_npu(const at::Tensor& input, int64_t kernel_size, int64_t stride, int64_t padding) {
    // Infer output shape in PyTorch side to allocate output tensor
    auto sizes = input.sizes();
    int64_t N = sizes[0];
    int64_t C = sizes[1];
    int64_t L = sizes[2];
    int64_t outLenCalc = (L + 2 * padding - kernel_size);
    int64_t outLen = outLenCalc >= 0 ? (outLenCalc / stride) + 1 : 0;
    if (outLen < 0) outLen = 0;

    at::TensorOptions opts = input.options();
    at::Tensor output = at::empty({N, C, outLen}, opts);

    EXEC_NPU_CMD(aclnnAveragePooling1dCustom, input, kernel_size, stride, padding, output);
    return output;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("average_pooling_1d_custom", &average_pooling_1d_custom_impl_npu);

    m.def("average_pooling_1d_custom", &average_pooling_1d_custom_impl_npu, "AvgPool1d custom AscendC");
