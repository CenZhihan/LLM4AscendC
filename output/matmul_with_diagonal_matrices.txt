kernel_src="""
#include "kernel_operator.h"

namespace {
struct MatmulWithDiagonalMatricesCustomTilingData {
    uint32_t rows;
    uint32_t cols;
    uint32_t tileLength;
};
}

extern "C" __global__ __aicore__ void matmul_with_diagonal_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace,
                                                                           GM_ADDR tiling)
{
    GET_TILING_DATA(MatmulWithDiagonalMatricesCustomTilingData, tilingData, tiling);
    uint32_t totalRows = tilingData.rows;
    uint32_t cols = tilingData.cols;
    uint32_t tileLength = tilingData.tileLength;

    uint32_t blockIdx = GetBlockIdx();
    uint32_t blockDim = GetBlockNum();
    
    uint32_t rowsPerBlock = totalRows / blockDim;
    uint32_t remain = totalRows % blockDim;
    uint32_t startRow, endRow;
    
    if (blockIdx < remain) {
        startRow = blockIdx * (rowsPerBlock + 1);
        endRow = startRow + (rowsPerBlock + 1);
    } else {
        startRow = remain * (rowsPerBlock + 1) + (blockIdx - remain) * rowsPerBlock;
        endRow = startRow + rowsPerBlock;
    }
    
    uint32_t currentRows = endRow - startRow;
    if (currentRows == 0) return;

    AscendC::TPipe pipe;
    // Buffer for A: holds currentRows floats. 
    // Max rows typically < 10000 per core. Allocating logical max.
    // 4096 rows total. Max 4096 elements ~ 16KB.
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueA;
    pipe.InitBuffer(inQueueA, 1, (currentRows + 32) * sizeof(float)); // +32 for padding/alignment safety

    // Double buffering for B and C
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueB;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueC;
    pipe.InitBuffer(inQueueB, 2, tileLength * sizeof(float));
    pipe.InitBuffer(outQueueC, 2, tileLength * sizeof(float));

    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> cGm;
    aGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(A), totalRows);
    bGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(B), totalRows * cols);
    cGm.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(C), totalRows * cols);

    // Pre-load A
    AscendC::LocalTensor<float> aLocal = inQueueA.AllocTensor<float>();
    AscendC::DataCopy(aLocal, aGm[startRow], currentRows);
    inQueueA.EnQue(aLocal);
    
    aLocal = inQueueA.DeQue<float>();

    for (uint32_t i = 0; i < currentRows; ++i) {
        float scalarA = aLocal.GetValue(i);
        uint32_t globalRow = startRow + i;
        uint32_t rowOffset = globalRow * cols;

        for (uint32_t j = 0; j < cols; j += tileLength) {
             uint32_t currentTile = (j + tileLength > cols) ? (cols - j) : tileLength;
             
             AscendC::LocalTensor<float> bLocal = inQueueB.AllocTensor<float>();
             AscendC::DataCopy(bLocal, bGm[rowOffset + j], currentTile);
             inQueueB.EnQue(bLocal);

             bLocal = inQueueB.DeQue<float>();
             AscendC::LocalTensor<float> cLocal = outQueueC.AllocTensor<float>();
             AscendC::Muls(cLocal, bLocal, scalarA, currentTile);
             outQueueC.EnQue(cLocal);
             inQueueB.FreeTensor(bLocal);

             cLocal = outQueueC.DeQue<float>();
             AscendC::DataCopy(cGm[rowOffset + j], cLocal, currentTile);
             outQueueC.FreeTensor(cLocal);
        }
    }
    inQueueA.FreeTensor(aLocal);
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>
#include <iostream>

at::Tensor matmul_with_diagonal_matrices_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Keep execution on the custom ACLNN kernel path.
    // Kernel expects A as a 1D diagonal vector and computes row scaling.
    auto A_vec = A.contiguous();
    auto B_contig = B.contiguous();
    auto A_sizes = A_vec.sizes();
    auto B_sizes = B_contig.sizes();
    TORCH_CHECK(A_sizes.size() == 1 && B_sizes.size() == 2, "A must be 1D and B must be 2D.");
    TORCH_CHECK(A_sizes[0] == B_sizes[0], "A length must match B rows.");
    at::Tensor result = at::empty({A_sizes[0], B_sizes[1]}, B.options());
    EXEC_NPU_CMD(aclnnMatmulWithDiagonalMatricesCustom, A_vec, B_contig, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_with_diagonal_matrices_custom", &matmul_with_diagonal_matrices_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_with_diagonal_matrices_custom", &matmul_with_diagonal_matrices_impl_npu, "matmul with diagonal matrix");
}
