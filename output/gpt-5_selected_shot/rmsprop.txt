kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelRmsprop {
public:
    __aicore__ inline KernelRmsprop() {}
    __aicore__ inline void Init(
        GM_ADDR param, GM_ADDR grad, GM_ADDR v,
        GM_ADDR lr, GM_ADDR alpha, GM_ADDR eps,
        GM_ADDR out,
        uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        paramGm.SetGlobalBuffer((__gm__ float *)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gradGm.SetGlobalBuffer((__gm__ float *)grad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ float *)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        lrGm.SetGlobalBuffer((__gm__ float *)lr + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        alphaGm.SetGlobalBuffer((__gm__ float *)alpha + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        epsGm.SetGlobalBuffer((__gm__ float *)eps + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        outGm.SetGlobalBuffer((__gm__ float *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inParamQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inGradQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inVQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inLRQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inAlphaQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inEpsQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQ, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> paramLocal = inParamQ.AllocTensor<float>();
        AscendC::LocalTensor<float> gradLocal = inGradQ.AllocTensor<float>();
        AscendC::LocalTensor<float> vLocal    = inVQ.AllocTensor<float>();
        AscendC::LocalTensor<float> lrLocal   = inLRQ.AllocTensor<float>();
        AscendC::LocalTensor<float> alphaLocal= inAlphaQ.AllocTensor<float>();
        AscendC::LocalTensor<float> epsLocal  = inEpsQ.AllocTensor<float>();

        AscendC::DataCopy(paramLocal, paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gradLocal,  gradGm[progress * this->tileLength],  this->tileLength);
        AscendC::DataCopy(vLocal,     vGm[progress * this->tileLength],     this->tileLength);
        AscendC::DataCopy(lrLocal,    lrGm[progress * this->tileLength],    this->tileLength);
        AscendC::DataCopy(alphaLocal, alphaGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(epsLocal,   epsGm[progress * this->tileLength],   this->tileLength);

        inParamQ.EnQue(paramLocal);
        inGradQ.EnQue(gradLocal);
        inVQ.EnQue(vLocal);
        inLRQ.EnQue(lrLocal);
        inAlphaQ.EnQue(alphaLocal);
        inEpsQ.EnQue(epsLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<float> paramLocal = inParamQ.DeQue<float>();
        AscendC::LocalTensor<float> gradLocal  = inGradQ.DeQue<float>();
        AscendC::LocalTensor<float> vLocal     = inVQ.DeQue<float>();
        AscendC::LocalTensor<float> lrLocal    = inLRQ.DeQue<float>();
        AscendC::LocalTensor<float> alphaLocal = inAlphaQ.DeQue<float>();
        AscendC::LocalTensor<float> epsLocal   = inEpsQ.DeQue<float>();
        AscendC::LocalTensor<float> outLocal   = outQ.AllocTensor<float>();

        // grad2 -> outLocal = grad * grad
        AscendC::Mul(outLocal, gradLocal, gradLocal, this->tileLength);

        // vLocal = v - grad2
        AscendC::Sub(vLocal, vLocal, outLocal, this->tileLength);

        // vLocal = alpha * (v - grad2)
        AscendC::Mul(vLocal, alphaLocal, vLocal, this->tileLength);

        // vLocal = vLocal + grad2 => v_new
        AscendC::Add(vLocal, vLocal, outLocal, this->tileLength);

        // vLocal = sqrt(v_new)
        AscendC::Sqrt(vLocal, vLocal, this->tileLength);

        // vLocal = vLocal + eps
        AscendC::Add(vLocal, vLocal, epsLocal, this->tileLength);

        // outLocal = lr * grad
        AscendC::Mul(outLocal, lrLocal, gradLocal, this->tileLength);

        // outLocal = (lr * grad) / (sqrt(v_new) + eps)
        AscendC::Div(outLocal, outLocal, vLocal, this->tileLength);

        // outLocal = param - update
        AscendC::Sub(outLocal, paramLocal, outLocal, this->tileLength);

        outQ.EnQue<float>(outLocal);

        inParamQ.FreeTensor(paramLocal);
        inGradQ.FreeTensor(gradLocal);
        inVQ.FreeTensor(vLocal);
        inLRQ.FreeTensor(lrLocal);
        inAlphaQ.FreeTensor(alphaLocal);
        inEpsQ.FreeTensor(epsLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> outLocal = outQ.DeQue<float>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQ.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inParamQ, inGradQ, inVQ, inLRQ, inAlphaQ, inEpsQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQ;

    AscendC::GlobalTensor<float> paramGm;
    AscendC::GlobalTensor<float> gradGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> lrGm;
    AscendC::GlobalTensor<float> alphaGm;
    AscendC::GlobalTensor<float> epsGm;
    AscendC::GlobalTensor<float> outGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void rmsprop_custom(
    GM_ADDR param, GM_ADDR grad, GM_ADDR v,
    GM_ADDR lr, GM_ADDR alpha, GM_ADDR eps,
    GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsprop op;
    op.Init(param, grad, v, lr, alpha, eps, out, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, lr=1e-3, alpha=0.99, eps=1e-8):
        super().__init__()
        self.lr = lr
        self.alpha = alpha
        self.eps = eps

    def forward(self, param, grad, v):
        lr_t = torch.full_like(param, self.lr)
        alpha_t = torch.full_like(param, self.alpha)
        eps_t = torch.full_like(param, self.eps)
        return custom_ops_lib.rmsprop_custom(param, grad, v, lr_t, alpha_t, eps_t)
