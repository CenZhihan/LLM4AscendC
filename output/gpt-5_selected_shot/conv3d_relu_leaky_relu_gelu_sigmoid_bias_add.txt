kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// This kernel applies: ReLU -> LeakyReLU(0.01) -> GELU(approx) -> Sigmoid -> + bias (broadcast over N,D,H,W)
class KernelPostActivations {
public:
    __aicore__ inline KernelPostActivations() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        this->n = n; this->c = c; this->d = d; this->h = h; this->w = w;
        this->spatialSize = d * h * w;
        this->blockBaseOffset = this->blockLength * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockBaseOffset, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockBaseOffset, this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)bias, this->c);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // Constants
        const float negSlope = 0.01f;
        const float k0 = 0.5f;
        const float k1 = 0.7978845608028654f;   // sqrt(2/pi)
        const float k2 = 0.044715f;

        uint32_t base = progress * this->tileLength;

        for (uint32_t i = 0; i < this->tileLength; ++i) {
            uint32_t idx = base + i; // index within this block
            uint32_t globalIdx = this->blockBaseOffset + idx;

            // Channel index from flattened [N, C, D, H, W]
            // linear = ((((n * C) + c) * D + d) * H + h) * W + w
            // cIdx = (linear / (D*H*W)) % C
            uint32_t cIdx = (globalIdx / this->spatialSize) % this->c;

            // Load input value
            float v = static_cast<float>(xLocal.GetValue(i));

            // ReLU
            float relu_v = v > 0.f ? v : 0.f;

            // LeakyReLU with slope on negative side; since relu_v >= 0, it stays same. Keep generic form:
            float lrelu_v = relu_v >= 0.f ? relu_v : (negSlope * relu_v);

            // GELU (approx with tanh)
            float x3 = lrelu_v * lrelu_v * lrelu_v;
            float gelu_v = k0 * lrelu_v * (1.f + tanhf(k1 * (lrelu_v + k2 * x3)));

            // Sigmoid
            float sig_v = 1.f / (1.f + expf(-gelu_v));

            // Bias add (broadcast over N,D,H,W)
            // Read bias channel value
            // Note: using direct GM read for illustration; in practice, prefetch bias to local buffer for reuse.
            float bval = static_cast<float>(*(__gm__ DTYPE_B*)(bGm.GetPhyAddr(cIdx)));

            float out_v = sig_v + bval;
            yLocal.SetValue(i, (DTYPE_Y)out_v);
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t blockBaseOffset;

    uint32_t n, c, d, h, w;
    uint32_t spatialSize;
};

extern "C" __global__ __aicore__ void conv3d_relu_leaky_relu_gelu_sigmoid_bias_add_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelPostActivations op;
    op.Init(x, bias, y,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: Conv3d in PyTorch, followed by fused custom AscendC kernel
    that applies ReLU -> LeakyReLU(0.01) -> GELU -> Sigmoid -> BiasAdd.
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size, bias=False)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        x = self.conv(x)
        x = custom_ops_lib.conv3d_relu_leaky_relu_gelu_sigmoid_bias_add_custom(x, self.bias)
        return x
