host_operator_src="""
#include "conv_standard_1d_dilated_strided_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t ComputeLout(uint32_t lin, uint32_t k, uint32_t stride, uint32_t dilation) {
    // no padding
    int64_t effective = static_cast<int64_t>(lin) - static_cast<int64_t>(dilation) * static_cast<int64_t>(k - 1);
    if (effective <= 0) return 0;
    return static_cast<uint32_t>((effective - 1) / stride + 1);
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ConvStandard1dDilatedStridedCustomTilingData tiling;

    // Input shapes
    auto in_shape = context->GetInputShape(0)->GetOriginShape(); // x: [N, Cin, Lin]
    auto w_shape  = context->GetInputShape(1)->GetOriginShape(); // w: [Cout, Cin, K]

    uint32_t n    = static_cast<uint32_t>(in_shape.GetDim(0));
    uint32_t cin  = static_cast<uint32_t>(in_shape.GetDim(1));
    uint32_t lin  = static_cast<uint32_t>(in_shape.GetDim(2));
    uint32_t cout = static_cast<uint32_t>(w_shape.GetDim(0));
    uint32_t k    = static_cast<uint32_t>(w_shape.GetDim(2));

    // Attributes
    int64_t stride_attr = 1;
    int64_t dilation_attr = 1;
    (void)context->TryGetAttr("stride", stride_attr);
    (void)context->TryGetAttr("dilation", dilation_attr);
    uint32_t stride = static_cast<uint32_t>(stride_attr);
    uint32_t dilation = static_cast<uint32_t>(dilation_attr);

    uint32_t lout = ComputeLout(lin, k, stride, dilation);

    uint64_t totalOutElems64 = static_cast<uint64_t>(n) * static_cast<uint64_t>(cout) * static_cast<uint64_t>(lout);
    uint32_t totalOutElems = static_cast<uint32_t>(totalOutElems64);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_cin(cin);
    tiling.set_cout(cout);
    tiling.set_lin(lin);
    tiling.set_lout(lout);
    tiling.set_ksize(k);
    tiling.set_stride(stride);
    tiling.set_dilation(dilation);
    tiling.set_totalOutElems(totalOutElems);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0); // [N, Cin, Lin]
    const gert::Shape* w_shape = context->GetInputShape(1); // [Cout, Cin, K]
    int64_t stride = 1;
    int64_t dilation = 1;
    (void)context->TryGetAttr("stride", stride);
    (void)context->TryGetAttr("dilation", dilation);

    int64_t N    = x_shape->GetDim(0);
    int64_t Cin  = x_shape->GetDim(1);
    int64_t Lin  = x_shape->GetDim(2);
    int64_t Cout = w_shape->GetDim(0);
    int64_t K    = w_shape->GetDim(2);
    int64_t eff  = Lin - dilation * (K - 1);
    int64_t Lout = eff > 0 ? ( (eff - 1) / stride + 1 ) : 0;

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Lout);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_Y = float;

class KernelConv1D {
public:
    __aicore__ inline KernelConv1D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t lout, uint32_t k,
                                uint32_t stride, uint32_t dilation,
                                uint32_t totalOutElems, uint32_t tileNum)
    {
        this->xPtr = (__gm__ DTYPE_X *)x;
        this->wPtr = (__gm__ DTYPE_W *)w;
        this->yPtr = (__gm__ DTYPE_Y *)y;
        this->N = n;
        this->Cin = cin;
        this->Cout = cout;
        this->Lin = lin;
        this->Lout = lout;
        this->K = k;
        this->stride = stride;
        this->dilation = dilation;
        this->totalOutElems = totalOutElems;
        this->tileNum = tileNum;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->blockLen = (totalOutElems + blockNum - 1) / blockNum; // round up
        this->startIdx = blockIdx * this->blockLen;
        uint32_t endCandidate = this->startIdx + this->blockLen;
        this->endIdx = endCandidate > totalOutElems ? totalOutElems : endCandidate;
    }

    __aicore__ inline void Process()
    {
        for (uint32_t t = this->startIdx; t < this->endIdx; ++t) {
            // decode linear index t -> (n, oc, lo)
            uint32_t nc_out = this->Cout * this->Lout;
            uint32_t n = t / nc_out;
            uint32_t rem = t % nc_out;
            uint32_t oc = rem / this->Lout;
            uint32_t lo = rem % this->Lout;

            float acc = 0.0f;
            uint32_t lin0 = lo * this->stride;

            for (uint32_t ci = 0; ci < this->Cin; ++ci) {
                uint32_t xBase = ((n * this->Cin + ci) * this->Lin);
                uint32_t wBase = ((oc * this->Cin + ci) * this->K);
                #pragma unroll
                for (uint32_t kk = 0; kk <  /* max kernel size unknown at compile */ 1; ++kk) {
                    // The pragma won't unroll effectively; keep general loop below.
                }
                for (uint32_t kpos = 0; kpos < this->K; ++kpos) {
                    uint32_t lin = lin0 + kpos * this->dilation;
                    if (lin < this->Lin) {
                        DTYPE_X xv = this->xPtr[xBase + lin];
                        DTYPE_W wv = this->wPtr[wBase + kpos];
                        acc += xv * wv;
                    }
                }
            }

            uint32_t yIndex = ((n * this->Cout + oc) * this->Lout + lo);
            this->yPtr[yIndex] = acc;
        }
    }

private:
    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_W* wPtr;
    __gm__ DTYPE_Y* yPtr;
    uint32_t N, Cin, Cout, Lin, Lout, K, stride, dilation;
    uint32_t totalOutElems;
    uint32_t tileNum;
    uint32_t blockLen;
    uint32_t startIdx;
    uint32_t endIdx;
};

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv1D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.lout, tiling_data.ksize,
            tiling_data.stride, tiling_data.dilation,
            tiling_data.totalOutElems, tiling_data.tileNum);
    op.Process();
}

#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv_standard_1d_dilated_strided_custom_impl_npu(const at::Tensor& x,
                                                            const at::Tensor& w,
                                                            int64_t stride,
                                                            int64_t dilation) {
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto Lin = x.size(2);
    auto Cout = w.size(0);
    auto K = w.size(2);
    // compute output length (no padding)
    int64_t eff = Lin - dilation * (K - 1);
    int64_t Lout = eff > 0 ? ((eff - 1) / stride + 1) : 0;

    EXEC_NPU_CMD(aclnnConvStandard1dDilatedStridedCustom, x, w, stride, dilation, y);

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized 1D convolution using custom AscendC kernel.
    Matches nn.Conv1d behavior for no-padding, given stride and dilation.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, dilation: int = 1, bias: bool = False):
        super().__init__()
        self.conv1d = torch.nn.Conv1d(in_channels, out_channels, kernel_size, stride=stride, dilation=dilation, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        y = custom_ops_lib.conv_standard_1d_dilated_strided_custom(
            x,
            self.conv1d.weight,
            int(self.conv1d.stride[0]),
            int(self.conv1d.dilation[0])
        )
        if self.conv1d.bias is not None:
            y = y + self.conv1d.bias.view(1, -1, 1)
        return y
