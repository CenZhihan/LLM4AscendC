kernel_src="""
#include "kernel_operator.h"
#include <cmath>

class KernelFused {
public:
    __aicore__ inline KernelFused() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t groups, uint32_t cpg, uint32_t dhw, uint32_t tileDhw, float eps,
                                GM_ADDR workspace)
    {
        this->N = n; this->C = c; this->D = d; this->H = h; this->W = w;
        this->G = groups; this->CPG = cpg; this->DHW = dhw; this->tileDHW = tileDhw; this->eps = eps;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n * c * dhw);
        gammaGm.SetGlobalBuffer((__gm__ float*)gamma, c);
        betaGm.SetGlobalBuffer((__gm__ float*)beta, c);
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n * c * dhw);

        // workspace layout: [mean (N*G), rstd (N*G)]
        meanGm.SetGlobalBuffer((__gm__ float*)workspace, (uint64_t)n * groups);
        rstdGm.SetGlobalBuffer((__gm__ float*)workspace + (uint64_t)n * groups, (uint64_t)n * groups);

        pipe.InitBuffer(inQueueX, 2, this->tileDHW * sizeof(float));
        pipe.InitBuffer(outQueueY, 2, this->tileDHW * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Phase 1: compute per-(n,g) mean and rstd over Swish(x)
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t g = 0; g < G; ++g) {
                float sum = 0.f;
                float sumsq = 0.f;
                uint64_t elements = (uint64_t)CPG * DHW;

                uint32_t cStart = g * CPG;
                uint32_t cEnd = cStart + CPG;

                for (uint32_t c = cStart; c < cEnd; ++c) {
                    uint64_t base = offset(n, c);
                    for (uint32_t t = 0; t < DHW; t += tileDHW) {
                        uint32_t len = AscendC::min(this->tileDHW, DHW - t);
                        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                        AscendC::DataCopy(xLocal, xGm[base + t], len);
                        // Swish: s = x * sigmoid(x)
                        swish_inplace(xLocal, len);
                        // reduce
                        float localSum = 0.f;
                        float localSumSq = 0.f;
                        for (uint32_t i = 0; i < len; ++i) {
                            float v = xLocal.GetValue(i);
                            localSum += v;
                            localSumSq += v * v;
                        }
                        sum += localSum;
                        sumsq += localSumSq;
                        inQueueX.FreeTensor(xLocal);
                    }
                }
                float mean = sum / static_cast<float>(elements);
                float var = sumsq / static_cast<float>(elements) - mean * mean;
                float rstd = 1.0f / AscendC::sqrt(var + eps);

                meanGm.SetValue(n * G + g, mean);
                rstdGm.SetValue(n * G + g, rstd);
            }
        }

        // Phase 2: normalize, affine, then HardSwish
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t g = 0; g < G; ++g) {
                float mean = meanGm.GetValue(n * G + g);
                float rstd = rstdGm.GetValue(n * G + g);
                uint32_t cStart = g * CPG;
                uint32_t cEnd = cStart + CPG;
                for (uint32_t c = cStart; c < cEnd; ++c) {
                    float gamma = gammaGm.GetValue(c);
                    float beta = betaGm.GetValue(c);
                    uint64_t base = offset(n, c);
                    for (uint32_t t = 0; t < DHW; t += tileDHW) {
                        uint32_t len = AscendC::min(this->tileDHW, DHW - t);
                        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

                        AscendC::DataCopy(xLocal, xGm[base + t], len);
                        // Swish
                        swish_inplace(xLocal, len);
                        // GroupNorm affine: ((x - mean) * rstd) * gamma + beta
                        for (uint32_t i = 0; i < len; ++i) {
                            float v = xLocal.GetValue(i);
                            float nrm = (v - mean) * rstd;
                            float aff = nrm * gamma + beta;
                            yLocal.SetValue(i, aff);
                        }
                        // HardSwish: y * clip(y + 3, 0, 6) / 6
                        hardswish_inplace(yLocal, len);

                        AscendC::DataCopy(yGm[base + t], yLocal, len);
                        inQueueX.FreeTensor(xLocal);
                        outQueueY.FreeTensor(yLocal);
                    }
                }
            }
        }
    }

private:
    __aicore__ inline uint64_t offset(uint32_t n, uint32_t c) const {
        return ((uint64_t)n * C + c) * DHW;
    }

    __aicore__ inline void swish_inplace(AscendC::LocalTensor<float>& t, uint32_t len) {
        // t = t * sigmoid(t)
        for (uint32_t i = 0; i < len; ++i) {
            float v = t.GetValue(i);
            float s = 1.0f / (1.0f + expf(-v));
            t.SetValue(i, v * s);
        }
    }

    __aicore__ inline void hardswish_inplace(AscendC::LocalTensor<float>& t, uint32_t len) {
        for (uint32_t i = 0; i < len; ++i) {
            float v = t.GetValue(i);
            float u = v + 3.0f;
            if (u < 0.f) u = 0.f;
            if (u > 6.f) u = 6.f;
            float out = v * (u / 6.0f);
            t.SetValue(i, out);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> gammaGm;
    AscendC::GlobalTensor<float> betaGm;
    AscendC::GlobalTensor<float> yGm;

    AscendC::GlobalTensor<float> meanGm;
    AscendC::GlobalTensor<float> rstdGm;

    uint32_t N, C, D, H, W;
    uint32_t G, CPG, DHW, tileDHW;
    float eps;
};

extern "C" __global__ __aicore__ void conv_transpose3d_swish_group_norm_hard_swish_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, gamma, beta, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.groups, tiling_data.cpg, tiling_data.dhw, tiling_data.tileDhw, tiling_data.eps,
            workspace);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: ConvTranspose3d -> fused [Swish + GroupNorm + HardSwish] custom AscendC kernel
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, groups, eps, bias=True):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            bias=bias
        )
        # keep GroupNorm module only to hold parameters (gamma, beta, groups, eps)
        self.group_norm = torch.nn.GroupNorm(num_groups=groups, num_channels=out_channels, eps=eps)

    def forward(self, x):
        x = self.conv_transpose(x)
        # Fused kernel performs: Swish -> GroupNorm -> HardSwish
        x = custom_ops_lib.conv_transpose3d_swish_group_norm_hard_swish_custom(
            x,
            self.group_norm.weight,
            self.group_norm.bias,
            self.group_norm.num_groups,
            float(self.group_norm.eps)
        )
        return x
