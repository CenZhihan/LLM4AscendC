kernel_src="""
#include \"kernel_operator.h\"
#include <cmath>

extern \"C\" __global__ __aicore__ void windowed_causal_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t B  = tiling_data.B;
    const uint32_t N  = tiling_data.N;
    const uint32_t H  = tiling_data.H;
    const uint32_t Dh = tiling_data.Dh;
    const uint32_t W  = tiling_data.windowSize; // window size (inclusive of current token)

    const int32_t blockIdx = AscendC::GetBlockIdx();
    const int32_t totalBH  = static_cast<int32_t>(B * H);
    if (blockIdx >= totalBH) {
        return;
    }
    const uint32_t b = static_cast<uint32_t>(blockIdx) / H;
    const uint32_t h = static_cast<uint32_t>(blockIdx) % H;

    __gm__ float* qG = reinterpret_cast<__gm__ float*>(q);
    __gm__ float* kG = reinterpret_cast<__gm__ float*>(k);
    __gm__ float* vG = reinterpret_cast<__gm__ float*>(v);
    __gm__ float* zG = reinterpret_cast<__gm__ float*>(z);

    auto index = [&](uint32_t bb, uint32_t nn, uint32_t hh, uint32_t dd) -> size_t {
        // Layout [B, N, H, Dh] with Dh as innermost
        return (((static_cast<size_t>(bb) * N + nn) * H + hh) * Dh + dd);
    };

    const float scale = 1.0f / std::sqrt(static_cast<float>(Dh));

    for (uint32_t i = 0; i < N; ++i) {
        uint32_t start = (i > W) ? (i - W) : 0;
        // First pass: compute max logit for numerical stability
        float maxLogit = -1.0e30f;
        for (uint32_t kk = start; kk <= i; ++kk) {
            float dot = 0.0f;
            for (uint32_t d = 0; d < Dh; ++d) {
                float qv = qG[index(b, i, h, d)];
                float kv = kG[index(b, kk, h, d)];
                dot += qv * kv;
            }
            float logit = dot * scale;
            if (logit > maxLogit) {
                maxLogit = logit;
            }
        }

        // Second pass: compute denominator (sum of exp)
        float denom = 0.0f;
        for (uint32_t kk = start; kk <= i; ++kk) {
            float dot = 0.0f;
            for (uint32_t d = 0; d < Dh; ++d) {
                float qv = qG[index(b, i, h, d)];
                float kv = kG[index(b, kk, h, d)];
                dot += qv * kv;
            }
            float logit = dot * scale;
            denom += expf(logit - maxLogit);
        }
        float invDenom = denom > 0.0f ? (1.0f / denom) : 0.0f;

        // Third pass: accumulate weighted V into output z
        for (uint32_t d = 0; d < Dh; ++d) {
            float acc = 0.0f;
            for (uint32_t kk = start; kk <= i; ++kk) {
                float dot = 0.0f;
                for (uint32_t td = 0; td < Dh; ++td) {
                    float qv = qG[index(b, i, h, td)];
                    float kv = kG[index(b, kk, h, td)];
                    dot += qv * kv;
                }
                float logit = dot * scale;
                float w = expf(logit - maxLogit) * invDenom;
                float vv = vG[index(b, kk, h, d)];
                acc += w * vv;
            }
            zG[index(b, i, h, d)] = acc;
        }
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized sliding-window causal attention using a custom AscendC kernel.
    """
    def __init__(self, d_model=1024, num_heads=16, window_size=256):
        super().__init__()
        self.q_proj = torch.nn.Linear(d_model, d_model)
        self.k_proj = torch.nn.Linear(d_model, d_model)
        self.v_proj = torch.nn.Linear(d_model, d_model)
        self.out_proj = torch.nn.Linear(d_model, d_model)
        self.num_heads = num_heads
        self.window_size = window_size

    def forward(self, x):
        B, N, D = x.shape
        H = self.num_heads
        d_h = D // H

        # Projections
        Q = self.q_proj(x).view(B, N, H, d_h)
        K = self.k_proj(x).view(B, N, H, d_h)
        V = self.v_proj(x).view(B, N, H, d_h)

        # Custom windowed causal attention op
        attn_out = custom_ops_lib.windowed_causal_attention_custom(Q, K, V, self.window_size)

        return self.out_proj(attn_out.reshape(B, N, D))
