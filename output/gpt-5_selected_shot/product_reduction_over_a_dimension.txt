kernel_src="""
#include "kernel_operator.h"

class KernelProdReduceAxis1 {
public:
    __aicore__ inline KernelProdReduceAxis1() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t n, uint32_t d, uint32_t m)
    {
        this->n = n;
        this->d = d;
        this->m = m;

        xGm.SetGlobalBuffer((__gm__ float *)x, static_cast<uint64_t>(n) * d * m);
        zGm.SetGlobalBuffer((__gm__ float *)z, static_cast<uint64_t>(n) * m);

        // Buffers for one row of length m
        pipe.InitBuffer(inQueueX, 1, m * sizeof(float));
        pipe.InitBuffer(outQueueZ, 1, m * sizeof(float));
        pipe.InitBuffer(accBuf, m * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // For each batch n, compute product across axis-1 (d) for every position along m
        for (uint32_t bn = 0; bn < n; ++bn) {
            AscendC::LocalTensor<float> accLocal = accBuf.Get<float>();
            // Initialize accumulator to 1.0
            AscendC::Dup<float>(accLocal, 1.0f, m);

            for (uint32_t rd = 0; rd < d; ++rd) {
                // Load slice x[bn, rd, 0:m]
                AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                uint64_t offset = (static_cast<uint64_t>(bn) * d + rd) * m;
                AscendC::DataCopy(xLocal, xGm[offset], m);
                inQueueX.EnQue(xLocal);

                // Multiply accumulator by current slice
                AscendC::LocalTensor<float> xLocalDeq = inQueueX.DeQue<float>();
                AscendC::Mul<float>(accLocal, accLocal, xLocalDeq, m);
                inQueueX.FreeTensor(xLocalDeq);
            }

            // Write result z[bn, 0:m]
            AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
            AscendC::DataCopy(zLocal, accLocal, m);
            outQueueZ.EnQue(zLocal);

            AscendC::LocalTensor<float> zLocalDeq = outQueueZ.DeQue<float>();
            uint64_t zOffset = static_cast<uint64_t>(bn) * m;
            AscendC::DataCopy(zGm[zOffset], zLocalDeq, m);
            outQueueZ.FreeTensor(zLocalDeq);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;
    AscendC::TBuf<AscendC::TPosition::VECCALC> accBuf;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t n;
    uint32_t d;
    uint32_t m;
};

extern "C" __global__ __aicore__ void product_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelProdReduceAxis1 op;
    op.Init(x, z, tiling_data.n, tiling_data.d, tiling_data.m);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor product_reduction_over_a_dimension_impl_npu(const at::Tensor& x, int64_t dim) {
    // Compute output shape by removing 'dim'
    int64_t ndim = x.dim();
    if (dim < 0) dim += ndim;
    TORCH_CHECK(ndim >= 1, "Input must have at least 1 dimension");
    TORCH_CHECK(dim >= 0 && dim < ndim, "dim out of range");

    std::vector<int64_t> out_sizes;
    out_sizes.reserve(ndim > 0 ? static_cast<size_t>(ndim - 1) : 0);
    for (int64_t i = 0; i < ndim; ++i) {
        if (i == dim) continue;
        out_sizes.push_back(x.size(i));
    }
    if (out_sizes.empty()) {
        out_sizes.push_back(1); // at least 1D tensor
    }

    at::Tensor result = at::empty(out_sizes, x.options());
    // For this kernel, dim==1 on 3D inputs is expected; backend reduces axis-1.
    EXEC_NPU_CMD(aclnnProductReductionOverADimensionCustom, x, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("product_reduction_over_a_dimension_custom", &product_reduction_over_a_dimension_impl_npu);
}
PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("product_reduction_over_a_dimension_custom", &product_reduction_over_a_dimension_impl_npu, "product reduction over a dimension");
