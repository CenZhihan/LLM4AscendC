kernel_src="""
#include "kernel_operator.h"

extern \"C\" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_strided_padded_grouped_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    // Types
    using DTYPE = float;

    // Dimensions
    const int N      = static_cast<int>(tiling_data.n);
    const int Cin    = static_cast<int>(tiling_data.cin);
    const int Cout   = static_cast<int>(tiling_data.cout);
    const int Din    = static_cast<int>(tiling_data.din);
    const int Hin    = static_cast<int>(tiling_data.hin);
    const int Win    = static_cast<int>(tiling_data.win);
    const int K      = static_cast<int>(tiling_data.k);
    const int S      = static_cast<int>(tiling_data.stride);
    const int P      = static_cast<int>(tiling_data.padding);
    const int G      = static_cast<int>(tiling_data.groups);
    const int Dout   = static_cast<int>(tiling_data.dout);
    const int Hout   = static_cast<int>(tiling_data.hout);
    const int Wout   = static_cast<int>(tiling_data.wout);
    const long long total = static_cast<long long>(tiling_data.totalLength);

    const int CoutPerG = (G > 0) ? (Cout / G) : Cout;
    const int CinPerG  = (G > 0) ? (Cin / G) : Cin;

    DTYPE* x_ptr = reinterpret_cast<DTYPE*>(x);
    DTYPE* w_ptr = reinterpret_cast<DTYPE*>(w);
    DTYPE* y_ptr = reinterpret_cast<DTYPE*>(y);

    // Parallelize by blocks over output elements
    int block_num = AscendC::GetBlockNum();
    int block_idx = AscendC::GetBlockIdx();
    long long block_len = (total + block_num - 1) / block_num;
    long long start = static_cast<long long>(block_idx) * block_len;
    long long end = start + block_len;
    if (end > total) end = total;

    // Strides for indexing
    const long long x_c_stride = static_cast<long long>(Din) * Hin * Win;
    const long long x_n_stride = static_cast<long long>(Cin) * x_c_stride;

    const long long y_c_stride = static_cast<long long>(Dout) * Hout * Wout;
    const long long y_n_stride = static_cast<long long>(Cout) * y_c_stride;

    const long long w_co_stride = static_cast<long long>(K) * K * K;
    const long long w_ci_stride = static_cast<long long>(CoutPerG) * w_co_stride;

    for (long long idx = start; idx < end; ++idx) {
        // Decode linear index to (n, co, od, oh, ow)
        long long tmp = idx;
        int ow = static_cast<int>(tmp % Wout); tmp /= Wout;
        int oh = static_cast<int>(tmp % Hout); tmp /= Hout;
        int od = static_cast<int>(tmp % Dout); tmp /= Dout;
        int co = static_cast<int>(tmp % Cout); tmp /= Cout;
        int n  = static_cast<int>(tmp);

        int g = (CoutPerG > 0) ? (co / CoutPerG) : 0;
        int co_in_group = (CoutPerG > 0) ? (co % CoutPerG) : co;

        DTYPE acc = static_cast<DTYPE>(0);

        // Gather-based computation to avoid atomic writes
        // For each input channel in the same group
        int ci_start = g * CinPerG;
        int ci_end   = ci_start + CinPerG;

        // Precompute padded coordinates
        int td = od + P;
        int th = oh + P;
        int tw = ow + P;

        for (int ci = ci_start; ci < ci_end; ++ci) {
            // For each kernel position
            for (int kd = 0; kd < K; ++kd) {
                int id_t = td - kd;
                if (id_t < 0) continue;
                if (S != 0 && (id_t % S) != 0) continue;
                int id = (S == 0) ? 0 : (id_t / S);
                if (id < 0 || id >= Din) continue;

                for (int kh = 0; kh < K; ++kh) {
                    int ih_t = th - kh;
                    if (ih_t < 0) continue;
                    if (S != 0 && (ih_t % S) != 0) continue;
                    int ih = (S == 0) ? 0 : (ih_t / S);
                    if (ih < 0 || ih >= Hin) continue;

                    for (int kw = 0; kw < K; ++kw) {
                        int iw_t = tw - kw;
                        if (iw_t < 0) continue;
                        if (S != 0 && (iw_t % S) != 0) continue;
                        int iw = (S == 0) ? 0 : (iw_t / S);
                        if (iw < 0 || iw >= Win) continue;

                        long long x_offset = static_cast<long long>(n) * x_n_stride
                                           + static_cast<long long>(ci) * x_c_stride
                                           + static_cast<long long>(id) * (Hin * Win)
                                           + static_cast<long long>(ih) * Win
                                           + static_cast<long long>(iw);

                        long long w_offset = static_cast<long long>(ci) * w_ci_stride
                                           + static_cast<long long>(co_in_group) * w_co_stride
                                           + static_cast<long long>(kd) * (K * K)
                                           + static_cast<long long>(kh) * K
                                           + static_cast<long long>(kw);

                        acc += x_ptr[x_offset] * w_ptr[w_offset];
                    }
                }
            }
        }

        long long y_offset = static_cast<long long>(n) * y_n_stride
                           + static_cast<long long>(co) * y_c_stride
                           + static_cast<long long>(od) * (Hout * Wout)
                           + static_cast<long long>(oh) * Wout
                           + static_cast<long long>(ow);

        y_ptr[y_offset] = acc;
    }
}

static inline std::vector<int64_t> infer_deconv3d_output_shape(const at::Tensor& x, const at::Tensor& w, int64_t stride, int64_t padding, int64_t groups) {
    // x: [N, Cin, Din, Hin, Win]
    // w: [Cin, Cout_per_group, K, K, K]
    TORCH_CHECK(x.dim() == 5, \"x must be 5D\");
    TORCH_CHECK(w.dim() == 5, \"w must be 5D\");
    int64_t N = x.size(0);
    int64_t Cin = x.size(1);
    int64_t Din = x.size(2);
    int64_t Hin = x.size(3);
    int64_t Win = x.size(4);
    int64_t K = w.size(2);
    int64_t CoutPerG = w.size(1);
    TORCH_CHECK(groups > 0, \"groups must be > 0\");
    TORCH_CHECK(Cin % groups == 0, \"Cin must be divisible by groups\");
    int64_t Cout = CoutPerG * groups;

    int64_t Dout = (Din - 1) * stride - 2 * padding + K;
    int64_t Hout = (Hin - 1) * stride - 2 * padding + K;
    int64_t Wout = (Win - 1) * stride - 2 * padding + K;
    return {N, Cout, Dout, Hout, Wout};
}

import torch

import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom 3D transposed convolution using AscendC kernel.
    Weight layout follows PyTorch ConvTranspose3d: [Cin, Cout_per_group, K, K, K]
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, groups: int = 1, bias: bool = False):
        super().__init__()
        assert out_channels % groups == 0, "out_channels must be divisible by groups"
        assert in_channels % groups == 0, "in_channels must be divisible by groups"
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = int(stride)
        self.padding = int(padding)
        self.groups = int(groups)
        self.bias = None
        # Weight shape: [Cin, Cout_per_group, K, K, K]
        cout_per_g = out_channels // groups
        self.weight = torch.nn.Parameter(torch.empty(in_channels, cout_per_g, kernel_size, kernel_size, kernel_size))
        torch.nn.init.kaiming_uniform_(self.weight, a=5**0.5)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_3d_asymmetric_input_square_kernel_strided_padded_grouped_custom(
