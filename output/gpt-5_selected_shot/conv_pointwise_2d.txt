kernel_src="""
#include "kernel_operator.h"

// Kernel for pointwise (1x1) conv: y[n, cout, h, w] = sum_{cin} x[n, cin, h, w] * w[cout, cin]
constexpr int32_t BUFFER_NUM = 2;

class KernelPointwiseConv1x1 {
public:
    __aicore__ inline KernelPointwiseConv1x1() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalNHW, uint32_t cIn, uint32_t cOut, uint32_t tileNum) {
        this->totalNHW = totalNHW;
        this->cIn = cIn;
        this->cOut = cOut;
        this->tileNum = tileNum;

        // Distribute NHW across blocks
        this->blockNHW = totalNHW / AscendC::GetBlockNum();
        this->tileNHW = (this->blockNHW / tileNum);
        // GM base pointers for this block
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)AscendC::GetBlockIdx() * (uint64_t)blockNHW * (uint64_t)cIn,
                            (uint64_t)blockNHW * (uint64_t)cIn);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)cOut * (uint64_t)cIn);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + (uint64_t)AscendC::GetBlockIdx() * (uint64_t)blockNHW * (uint64_t)cOut,
                            (uint64_t)blockNHW * (uint64_t)cOut);

        // Allocate local buffers per tile
        pipe.InitBuffer(inQueueX, BUFFER_NUM, (uint64_t)tileNHW * (uint64_t)cIn * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, (uint64_t)tileNHW * (uint64_t)cOut * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process() {
        int32_t loopCount = tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress) {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        uint64_t offset = (uint64_t)progress * (uint64_t)tileNHW * (uint64_t)cIn;
        AscendC::DataCopy(xLocal, xGm[offset], (uint64_t)tileNHW * (uint64_t)cIn);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/) {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // Naive compute: for each spatial position in tile, compute Cout outputs
        // yLocal[m, cout] = sum_{cin} xLocal[m, cin] * wGm[cout, cin]
        for (uint32_t m = 0; m < tileNHW; ++m) {
            for (uint32_t co = 0; co < cOut; ++co) {
                float acc = 0.f;
                for (uint32_t ci = 0; ci < cIn; ++ci) {
                    // Flattened indexing
                    DTYPE_X xv = xLocal[m * cIn + ci];
                    DTYPE_W wv = wGm[co * cIn + ci];
                    acc += (float)xv * (float)wv;
                }
                yLocal[m * cOut + co] = (DTYPE_Y)acc;
            }
        }

        outQueueY.EnQue(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress) {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        uint64_t offset = (uint64_t)progress * (uint64_t)tileNHW * (uint64_t)cOut;
        AscendC::DataCopy(yGm[offset], yLocal, (uint64_t)tileNHW * (uint64_t)cOut);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalNHW;
    uint32_t cIn;
    uint32_t cOut;
    uint32_t blockNHW;
    uint32_t tileNum;
    uint32_t tileNHW;
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelPointwiseConv1x1 op;
    op.Init(x, w, y, tiling_data.totalNHW, tiling_data.cIn, tiling_data.cOut, tiling_data.tileNum);
    op.Process();
}

at::Tensor conv_pointwise_2d_custom_impl_npu(const at::Tensor& x, const at::Tensor& w) {
    // x: (N, C_in, H, W), w: (C_out, C_in)
    auto N = x.size(0);

    auto W = x.size(3);
    auto Cout = w.size(0);

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized 1x1 Conv2D using custom AscendC kernel.
    """
    def __init__(self, in_channels: int, out_channels: int, bias: bool = False):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        # Weight in shape (C_out, C_in) for pointwise conv
        self.weight = torch.nn.Parameter(torch.randn(out_channels, in_channels) * (1.0 / (in_channels ** 0.5)))
        self.bias = None
        if bias:
            self.bias = torch.nn.Parameter(torch.zeros(out_channels))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x: (N, C_in, H, W)
        y = custom_ops_lib.conv_pointwise_2d_custom(x, self.weight)
        if self.bias is not None:
            # Add bias per output channel
            y = y + self.bias.view(1, -1, 1, 1)
        return y
