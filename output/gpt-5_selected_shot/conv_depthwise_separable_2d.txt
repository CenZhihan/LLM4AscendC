kernel_src="""
#include "kernel_operator.h"

extern \"C\" __global__ __aicore__ void conv_depthwise_separable_2d_custom(
    GM_ADDR x,
    GM_ADDR dw_weight,
    GM_ADDR pw_weight,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);

    const uint32_t N = t.n;
    const uint32_t C_in = t.c_in;
    const uint32_t C_out = t.c_out;
    const uint32_t H = t.h;
    const uint32_t W = t.w;
    const uint32_t K_h = t.k_h;
    const uint32_t K_w = t.k_w;
    const uint32_t stride_h = t.stride_h;
    const uint32_t stride_w = t.stride_w;
    const uint32_t pad_h = t.pad_h;
    const uint32_t pad_w = t.pad_w;
    const uint32_t dil_h = t.dil_h;
    const uint32_t dil_w = t.dil_w;
    const uint32_t OH = t.out_h;
    const uint32_t OW = t.out_w;

    // Cast GM addresses
    const __gm__ float* x_ptr = reinterpret_cast<const __gm__ float*>(x);
    const __gm__ float* dw_w_ptr = reinterpret_cast<const __gm__ float*>(dw_weight);
    const __gm__ float* pw_w_ptr = reinterpret_cast<const __gm__ float*>(pw_weight);
    __gm__ float* y_ptr = reinterpret_cast<__gm__ float*>(y);

    // Distribute work across blocks by batch dimension
    uint32_t block_num = AscendC::GetBlockNum();
    uint32_t block_idx = AscendC::GetBlockIdx();
    uint32_t n_per_block = (N + block_num - 1) / block_num;
    uint32_t n_start = block_idx * n_per_block;
    uint32_t n_end = AscendC::Min(static_cast<uint32_t>(N), n_start + n_per_block);

    // Strides for indexing
    uint64_t x_n_stride = static_cast<uint64_t>(C_in) * H * W;
    uint64_t x_c_stride = static_cast<uint64_t>(H) * W;
    uint64_t x_h_stride = static_cast<uint64_t>(W);

    uint64_t dw_c_stride = static_cast<uint64_t>(K_h) * K_w; // (in_c, 1, kh, kw)
    uint64_t dw_kh_stride = static_cast<uint64_t>(K_w);

    uint64_t pw_co_stride = static_cast<uint64_t>(C_in); // (co, ci, 1, 1)

    uint64_t y_n_stride = static_cast<uint64_t>(C_out) * OH * OW;
    uint64_t y_c_stride = static_cast<uint64_t>(OH) * OW;
    uint64_t y_h_stride = static_cast<uint64_t>(OW);

    // Compute fused depthwise + pointwise
    for (uint32_t n = n_start; n < n_end; ++n) {
        uint64_t x_n_off = static_cast<uint64_t>(n) * x_n_stride;
        uint64_t y_n_off = static_cast<uint64_t>(n) * y_n_stride;
        for (uint32_t co = 0; co < C_out; ++co) {
            uint64_t pw_co_off = static_cast<uint64_t>(co) * pw_co_stride;
            for (uint32_t oh = 0; oh < OH; ++oh) {
                int64_t base_ih = static_cast<int64_t>(oh) * static_cast<int64_t>(stride_h) - static_cast<int64_t>(pad_h);
                for (uint32_t ow = 0; ow < OW; ++ow) {
                    int64_t base_iw = static_cast<int64_t>(ow) * static_cast<int64_t>(stride_w) - static_cast<int64_t>(pad_w);
                    float acc = 0.0f;
                    for (uint32_t ci = 0; ci < C_in; ++ci) {
                        uint64_t x_ci_off = x_n_off + static_cast<uint64_t>(ci) * x_c_stride;
                        uint64_t dw_ci_off = static_cast<uint64_t>(ci) * dw_c_stride;
                        // Depthwise conv for (n, ci, oh, ow)
                        float dw_val = 0.0f;
                        for (uint32_t kh = 0; kh < K_h; ++kh) {
                            int64_t ih = base_ih + static_cast<int64_t>(kh) * static_cast<int64_t>(dil_h);
                            if (ih < 0 || ih >= static_cast<int64_t>(H)) continue;
                            for (uint32_t kw = 0; kw < K_w; ++kw) {
                                int64_t iw = base_iw + static_cast<int64_t>(kw) * static_cast<int64_t>(dil_w);
                                if (iw < 0 || iw >= static_cast<int64_t>(W)) continue;
                                uint64_t x_off = x_ci_off + static_cast<uint64_t>(ih) * x_h_stride + static_cast<uint64_t>(iw);
                                uint64_t w_off = dw_ci_off + static_cast<uint64_t>(kh) * dw_kh_stride + static_cast<uint64_t>(kw);
                                dw_val += x_ptr[x_off] * dw_w_ptr[w_off];
                            }
                        }
                        // Pointwise 1x1: multiply dw_val with pw weight at (co, ci)
                        uint64_t pw_off = pw_co_off + static_cast<uint64_t>(ci);
                        acc += dw_val * pw_w_ptr[pw_off];
                    }
                    uint64_t y_off = y_n_off + static_cast<uint64_t>(co) * y_c_stride + static_cast<uint64_t>(oh) * y_h_stride + static_cast<uint64_t>(ow);
                    y_ptr[y_off] = acc;
                }
            }
        }
    }
}

#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv_depthwise_separable_2d_custom_impl_npu(
    const at::Tensor& x,
    const at::Tensor& dw_weight,
    const at::Tensor& pw_weight,
    int64_t stride,
    int64_t padding,
    int64_t dilation)
{
    // Convert scalar ints to 2D attributes (H, W)
    std::vector<int64_t> stride_vec{stride, stride};
    std::vector<int64_t> padding_vec{padding, padding};
    std::vector<int64_t> dilation_vec{dilation, dilation};

    // Output tensor shape is inferred by the runtime/operator
    at::Tensor y = at::empty({x.size(0), pw_weight.size(0),
                              (x.size(2) + 2 * padding - dilation * (dw_weight.size(2) - 1) - 1) / stride + 1,
                              (x.size(3) + 2 * padding - dilation * (dw_weight.size(3) - 1) - 1) / stride + 1},
                              x.options());

    EXEC_NPU_CMD(aclnnConvDepthwiseSeparable2dCustom, x, dw_weight, pw_weight, stride_vec, padding_vec, dilation_vec, y);
    return y;

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized depthwise-separable 2D convolution using a fused custom AscendC kernel.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

        # Depthwise weight: (in_channels, 1, k, k)
        self.dw_weight = torch.nn.Parameter(torch.randn(in_channels, 1, kernel_size, kernel_size))
        # Pointwise weight: (out_channels, in_channels, 1, 1)
        self.pw_weight = torch.nn.Parameter(torch.randn(out_channels, in_channels, 1, 1))

        # Bias not fused in this custom kernel (optional in original), omit or handle externally.
        if bias:
            raise NotImplementedError("Bias handling is not implemented in the fused custom kernel.")

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_depthwise_separable_2d_custom(
            x, self.dw_weight, self.pw_weight, self.stride, self.padding, self.dilation
        )
