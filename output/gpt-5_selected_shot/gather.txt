#include "gather_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GatherCustomTilingData tiling;

    auto* x_shape = context->GetInputShape(0);
    auto* idx_shape = context->GetInputShape(1);

    uint32_t batch = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t inWidth = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t outWidth = static_cast<uint32_t>(idx_shape->GetOriginShape().GetDim(1));

    uint32_t totalLength = batch * outWidth;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_batch(batch);
    tiling.set_inWidth(inWidth);
    tiling.set_outWidth(outWidth);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* idx_shape = context->GetInputShape(1);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *idx_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto xType = context->GetInputDataType(0);
    context->SetOutputDataType(0, xType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class GatherCustom : public OpDef {
public:
    explicit GatherCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("idx")
            .ParamType(REQUIRED)

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // double buffering

class KernelGather {
public:
    __aicore__ inline KernelGather() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR z,
                                uint32_t batch, uint32_t inWidth, uint32_t outWidth)
    {
        this->batch = batch;
        this->inWidth = inWidth;
        this->outWidth = outWidth;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Distribute rows to blocks as evenly as possible
        uint32_t baseRowsPerBlock = batch / blockNum;
        uint32_t remain = batch % blockNum;
        this->rowsThisBlock = baseRowsPerBlock + (blockIdx < remain ? 1 : 0);
        uint32_t startRow = baseRowsPerBlock * blockIdx + AscendC::Min(blockIdx, remain);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + static_cast<uint64_t>(startRow) * inWidth,
                            static_cast<uint64_t>(rowsThisBlock) * inWidth);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_Y*)idx + static_cast<uint64_t>(startRow) * outWidth,
                              static_cast<uint64_t>(rowsThisBlock) * outWidth);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z + static_cast<uint64_t>(startRow) * outWidth,
                            static_cast<uint64_t>(rowsThisBlock) * outWidth);

        // Buffers sized per single row
        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, outWidth * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueXRow, BUFFER_NUM, inWidth * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, outWidth * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsThisBlock; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row)
    {
        // Load idx row
        AscendC::LocalTensor<DTYPE_Y> idxLocal = inQueueIdx.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(idxLocal, idxGm[row * outWidth], outWidth);
        inQueueIdx.EnQue(idxLocal);

        // Load x row
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueXRow.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[row * inWidth], inWidth);
        inQueueXRow.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*row*/)
    {
        AscendC::LocalTensor<DTYPE_Y> idxLocal = inQueueIdx.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueXRow.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // Gather along the 1st dimension (within-row gather)
        AscendC::Gather(zLocal, xLocal, idxLocal, outWidth);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueIdx.FreeTensor(idxLocal);
        inQueueXRow.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[row * outWidth], zLocal, outWidth);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueXRow;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> idxGm; // int32 indices
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t batch;
    uint32_t inWidth;
    uint32_t outWidth;
    uint32_t rowsThisBlock;
};

extern "C" __global__ __aicore__ void gather_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGather op;
    op.Init(x, idx, z, tiling_data.batch, tiling_data.inWidth, tiling_data.outWidth);
    op.Process();
}
"""

#include <torch/library.h>

#include <torch/extension.h>

at::Tensor gather_custom_impl_npu(const at::Tensor& x, const at::Tensor& idx) {
    TORCH_CHECK(x.dim() == 2 && idx.dim() == 2, "Expect 2D tensors");
    TORCH_CHECK(x.size(0) == idx.size(0), "Batch dimension must match");
    at::Tensor result = at::empty(idx.sizes(), x.options());
    EXEC_NPU_CMD(aclnnGatherCustom, x, idx, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("gather_custom", &gather_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("gather_custom", &gather_custom_impl_npu, "gather along dim=1");
}
