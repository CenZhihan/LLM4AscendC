#include "batch_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 48;

static ge::graphStatus TilingFunc(gert::TilingContext* context) {
  BatchNormCustomTilingData tiling;
  const gert::StorageShape* x_shape = context->GetInputShape(0);
  const gert::Shape shape = x_shape->GetStorageShape();
  auto n = static_cast<uint32_t>(shape.GetDim(0));
  auto c = static_cast<uint32_t>(shape.GetDim(1));
  auto h = static_cast<uint32_t>(shape.GetDim(2));
  auto w = static_cast<uint32_t>(shape.GetDim(3));

  uint64_t rowsTotal64 = static_cast<uint64_t>(n) * static_cast<uint64_t>(c) * static_cast<uint64_t>(h);
  uint32_t rowsTotal = static_cast<uint32_t>(rowsTotal64);
  auto coreRowNum = rowsTotal / BLOCK_DIM;

  const float* epsilonAttr = context->GetAttrs()->GetAttrPointer<float>(0);
  tiling.set_eps(*epsilonAttr);
  tiling.set_rowNum(coreRowNum);
  tiling.set_rowNumSp(coreRowNum + 1);
  tiling.set_rowLength(w);
  tiling.set_blockPivot(rowsTotal - coreRowNum * BLOCK_DIM);

  uint32_t tileLoop = 32;  // rows per tile
  tiling.set_tileLoop(tileLoop);
  tiling.set_tileLength(tileLoop * w);
  tiling.set_loopCount(coreRowNum / tileLoop);

  tiling.set_channels(c);
  tiling.set_height(h);

  context->SetBlockDim(BLOCK_DIM);
  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
  context->SetTilingKey(1);
  return ge::GRAPH_SUCCESS;
}
}  // namespace optiling

kernel_src="""
#include "kernel_operator.h"
constexpr int32_t BUFFER_NUM = 1;

class KernelBatchNorm {
 public:
  __aicore__ inline KernelBatchNorm() {}
  __aicore__ inline void InitTiling(GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    baseRowNum = tiling_data.rowNum;
    rowNumSp = tiling_data.rowNumSp;
    rowLength = tiling_data.rowLength;   // W
    blockPivot = tiling_data.blockPivot;
    tileLoop = tiling_data.tileLoop;
    tileLength = tiling_data.tileLength; // tileLoop * W
    loopCount = tiling_data.loopCount;
    channels = tiling_data.channels;     // C
    height = tiling_data.height;         // H
    eps = tiling_data.eps;
  }
  __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR mean, GM_ADDR var, GM_ADDR z,
                              GM_ADDR tiling) {
    InitTiling(tiling);

    // rows assigned to this block
    rowNum = baseRowNum;
    leftRow = rowNum % tileLoop;
    if (AscendC::GetBlockIdx() < blockPivot) {
      rowNum += 1;
      leftRow = (leftRow + 1) % tileLoop;
    }

    blockLength = rowNum * rowLength;
    uint32_t offset = 0;
    if (AscendC::GetBlockIdx() < blockPivot) {
      offset = blockLength * AscendC::GetBlockIdx();
    } else {
      offset = blockLength * AscendC::GetBlockIdx() + rowLength * blockPivot;
    }

    xGm.SetGlobalBuffer((__gm__ float *)x + offset, blockLength);
    zGm.SetGlobalBuffer((__gm__ float *)z + offset, blockLength);

    gammaGm.SetGlobalBuffer((__gm__ float *)gamma, channels);
    betaGm.SetGlobalBuffer((__gm__ float *)beta, channels);
    meanGm.SetGlobalBuffer((__gm__ float *)mean, channels);
    varGm.SetGlobalBuffer((__gm__ float *)var, channels);

    pipe.InitBuffer(queueX, BUFFER_NUM, tileLength * sizeof(float));
    pipe.InitBuffer(queueZ, BUFFER_NUM, tileLength * sizeof(float));

    // buffers for per-channel affine parameters
    pipe.InitBuffer(alphaBuf, channels * sizeof(float));
    pipe.InitBuffer(beta2Buf, channels * sizeof(float));
    pipe.InitBuffer(tmpGammaBuf, channels * sizeof(float));
    pipe.InitBuffer(tmpBetaBuf, channels * sizeof(float));
    pipe.InitBuffer(tmpMeanBuf, channels * sizeof(float));
    pipe.InitBuffer(tmpVarBuf, channels * sizeof(float));
    pipe.InitBuffer(onesBuf, channels * sizeof(float));

    // Precompute alpha = gamma / sqrt(var + eps)
    // and beta2 = beta - mean * alpha
    AscendC::LocalTensor<float> gammaLocal = tmpGammaBuf.Get<float>();
    AscendC::LocalTensor<float> betaLocal = tmpBetaBuf.Get<float>();
    AscendC::LocalTensor<float> meanLocal = tmpMeanBuf.Get<float>();
    AscendC::LocalTensor<float> varLocal  = tmpVarBuf.Get<float>();
    AscendC::LocalTensor<float> alphaLocal = alphaBuf.Get<float>();
    AscendC::LocalTensor<float> beta2Local = beta2Buf.Get<float>();
    AscendC::LocalTensor<float> onesLocal  = onesBuf.Get<float>();

    AscendC::DataCopy(gammaLocal, gammaGm[0], channels);
    AscendC::DataCopy(betaLocal,  betaGm[0],  channels);
    AscendC::DataCopy(meanLocal,  meanGm[0],  channels);
    AscendC::DataCopy(varLocal,   varGm[0],   channels);
    AscendC::Duplicate<float>(onesLocal, 1.0f, channels);

    AscendC::Adds(varLocal, varLocal, eps, channels);         // var + eps
    AscendC::Sqrt(varLocal, varLocal, channels);              // sqrt(var + eps)
    AscendC::Div(varLocal, onesLocal, varLocal, channels);    // 1 / sqrt(...)
    AscendC::Mul(alphaLocal, gammaLocal, varLocal, channels); // alpha
    AscendC::Mul(meanLocal, meanLocal, alphaLocal, channels); // mean * alpha
    AscendC::Muls(meanLocal, meanLocal, -1.0f, channels);     // -(mean * alpha)
    AscendC::Add(beta2Local, betaLocal, meanLocal, channels); // beta2
  }

  __aicore__ inline void Process() {
    for (uint32_t i = 0; i < loopCount; ++i) {
      CopyIn(i, tileLoop);
      Compute(i, tileLoop);
      CopyOut(i, tileLoop);
    }
    uint32_t processed = loopCount * tileLoop;
    uint32_t remaining = rowNum > processed ? (rowNum - processed) : 0;
    if (remaining > 0) {
      CopyIn(loopCount, remaining);
      Compute(loopCount, remaining);
      CopyOut(loopCount, remaining);
    }
  }

 private:
  __aicore__ inline void CopyIn(uint32_t progress, uint32_t rowsThisTile) {
    AscendC::LocalTensor<float> xLocal = queueX.AllocTensor<float>();
    AscendC::DataCopy(xLocal, xGm[progress * tileLength], rowsThisTile * rowLength);
    queueX.EnQue(xLocal);
  }

  __aicore__ inline void Compute(uint32_t progress, uint32_t rowsThisTile) {
    AscendC::LocalTensor<float> xLocal = queueX.DeQue<float>();
    AscendC::LocalTensor<float> zLocal = queueZ.AllocTensor<float>();
    AscendC::LocalTensor<float> alphaLocal = alphaBuf.Get<float>();
    AscendC::LocalTensor<float> beta2Local = beta2Buf.Get<float>();

    uint32_t startRowBase = (AscendC::GetBlockIdx() < blockPivot)
                                ? (rowNumSp * AscendC::GetBlockIdx())
                                : (baseRowNum * AscendC::GetBlockIdx() + blockPivot);
    uint32_t baseRow = startRowBase + progress * tileLoop;

    for (uint32_t j = 0; j < rowsThisTile; ++j) {
      uint32_t buffIndex = j * rowLength;
      uint32_t globalRow = baseRow + j;
      uint32_t cIdx = (globalRow / height) % channels;

      float alphaVal = alphaLocal.GetValue(cIdx);
      float beta2Val = beta2Local.GetValue(cIdx);

      AscendC::Muls(zLocal[buffIndex], xLocal[buffIndex], alphaVal, rowLength);
      AscendC::Adds(zLocal[buffIndex], zLocal[buffIndex], beta2Val, rowLength);
    }

    queueZ.EnQue<float>(zLocal);
    queueX.FreeTensor(xLocal);
  }

  __aicore__ inline void CopyOut(uint32_t progress, uint32_t rowsThisTile) {
    AscendC::LocalTensor<float> zLocal = queueZ.DeQue<float>();
    AscendC::DataCopy(zGm[progress * tileLength], zLocal, rowsThisTile * rowLength);
    queueZ.FreeTensor(zLocal);
  }

 private:
  AscendC::TPipe pipe;
  AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> queueX;
  AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> queueZ;

  AscendC::TBuf<AscendC::QuePosition::VECCALC> alphaBuf, beta2Buf;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> tmpGammaBuf, tmpBetaBuf, tmpMeanBuf, tmpVarBuf, onesBuf;

  AscendC::GlobalTensor<float> xGm;
  AscendC::GlobalTensor<float> gammaGm;
  AscendC::GlobalTensor<float> betaGm;
  AscendC::GlobalTensor<float> meanGm;
  AscendC::GlobalTensor<float> varGm;
  AscendC::GlobalTensor<float> zGm;

  uint32_t blockLength = 0;
  uint32_t baseRowNum = 0;    // rows per block (base)
  uint32_t rowNum = 0;        // rows assigned to this block
  uint32_t rowNumSp = 0;
  uint32_t rowLength = 0;     // W
  uint32_t blockPivot = 0;
  uint32_t tileLoop = 0;
  uint32_t tileLength = 0;
  uint32_t loopCount = 0;
  uint32_t channels = 0;      // C
  uint32_t height = 0;        // H
  uint32_t leftRow = 0;
  float eps = 1e-5f;
};

extern "C" __global__ __aicore__ void batch_norm_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR mean, GM_ADDR var, GM_ADDR res_out, GM_ADDR workspace,
    GM_ADDR tiling) {
  KernelBatchNorm op;
  op.Init(x, gamma, beta, mean, var, res_out, tiling);
  if (TILING_KEY_IS(1)) {
    op.Process();
  }
}
"""

model_src='''
import torch, torch_npu, custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_features: int):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.epsilon = 1e-5
        self.gamma = torch.nn.Parameter(torch.ones(num_features))
        self.beta = torch.nn.Parameter(torch.zeros(num_features))
        self.register_buffer("running_mean", torch.zeros(num_features))
        self.register_buffer("running_var", torch.ones(num_features))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Inference-style BatchNorm using provided running statistics
        return custom_ops_lib.batch_norm_custom(x, self.gamma, self.beta, self.running_mean, self.running_var, self.epsilon)
