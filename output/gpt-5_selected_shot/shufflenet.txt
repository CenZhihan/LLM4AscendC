kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelShuffle {
public:
    __aicore__ inline KernelShuffle() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t groups, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->groups = groups;
        this->tileNum = tileNum;

        this->sliceSize = h * w;                  // elements per channel slice (H*W)
        this->blockN = n / AscendC::GetBlockNum();// split by batch among blocks
        this->blockSlices = this->blockN * c;     // number of (n,c) slices this block handles
        this->startN = this->blockN * AscendC::GetBlockIdx();

        uint64_t startOffset = (uint64_t)startN * c * sliceSize;
        uint64_t blockElemLen = (uint64_t)blockSlices * sliceSize;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + startOffset, blockElemLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + startOffset, blockElemLen);

        // each queue tensor holds exactly one H*W slice
        pipe.InitBuffer(inQueue,  BUFFER_NUM, this->sliceSize * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->sliceSize * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t s = 0; s < this->blockSlices; ++s) {
            CopyIn(s);
            Compute(s);
            CopyOut(s);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t s)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.AllocTensor<DTYPE_X>();
        // source is contiguous slice s
        AscendC::DataCopy(xLocal, xGm[(uint64_t)s * sliceSize], sliceSize);
        inQueue.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*s*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueue.AllocTensor<DTYPE_Z>();
        // local-to-local copy
        AscendC::DataCopy(yLocal, xLocal, sliceSize);
        outQueue.EnQue<DTYPE_Z>(yLocal);
        inQueue.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t s)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueue.DeQue<DTYPE_Z>();

        // compute destination slice index under channel shuffle
        // within this block's local [blockN, C] domain
        uint32_t n_idx_local = s / c;
        uint32_t c_idx = s % c;

        uint32_t channels_per_group = c / groups;
        uint32_t new_c = (c_idx % channels_per_group) * groups + (c_idx / channels_per_group);

        uint32_t dest_slice_local = n_idx_local * c + new_c;

        AscendC::DataCopy(yGm[(uint64_t)dest_slice_local * sliceSize], yLocal, sliceSize);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t h;
    uint32_t w;
    uint32_t groups;

    uint32_t tileNum;
    uint32_t sliceSize;
    uint32_t blockN;
    uint32_t blockSlices;
    uint32_t startN;
};

extern "C" __global__ __aicore__ void shufflenet_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelShuffle op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.groups, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ChannelShuffle(nn.Module):
    def __init__(self, groups):
        super(ChannelShuffle, self).__init__()
        self.groups = groups
    def forward(self, x):
        # Use custom AscendC kernel
        return custom_ops_lib.shufflenet_custom(x, int(self.groups))

class ShuffleNetUnit(nn.Module):
    def __init__(self, in_channels, out_channels, groups=3):
        super(ShuffleNetUnit, self).__init__()
        assert out_channels % 4 == 0
        mid_channels = out_channels // 4

        self.conv1 = nn.Conv2d(in_channels, mid_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn1 = nn.BatchNorm2d(mid_channels)

        self.conv2 = nn.Conv2d(mid_channels, mid_channels, kernel_size=3, stride=1, padding=1, groups=mid_channels, bias=False)
        self.bn2 = nn.BatchNorm2d(mid_channels)

        self.conv3 = nn.Conv2d(mid_channels, out_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn3 = nn.BatchNorm2d(out_channels)

        self.shuffle = ChannelShuffle(groups)

        if in_channels == out_channels:
            self.shortcut = nn.Sequential()
        else:
            self.shortcut = nn.Sequential(
                nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=False),
                nn.BatchNorm2d(out_channels)
            )

    def forward(self, x):
        out = F.relu(self.bn1(self.conv1(x)))
        out = self.bn2(self.conv2(out))
        out = self.shuffle(out)
        out = F.relu(self.bn3(self.conv3(out)))
        out += self.shortcut(x)
        return out

class ModelNew(nn.Module):
    def __init__(self, num_classes=1000, groups=3, stages_repeats=[3, 7, 3], stages_out_channels=[24, 240, 480, 960]):
        super(ModelNew, self).__init__()
        self.conv1 = nn.Conv2d(3, stages_out_channels[0], kernel_size=3, stride=2, padding=1, bias=False)
        self.bn1 = nn.BatchNorm2d(stages_out_channels[0])
        self.maxpool = nn.MaxPool2d(kernel_size=3, stride=2, padding=1)

        self.stage2 = self._make_stage(stages_out_channels[0], stages_out_channels[1], stages_repeats[0], groups)
        self.stage3 = self._make_stage(stages_out_channels[1], stages_out_channels[2], stages_repeats[1], groups)
        self.stage4 = self._make_stage(stages_out_channels[2], stages_out_channels[3], stages_repeats[2], groups)

        self.conv5 = nn.Conv2d(stages_out_channels[3], 1024, kernel_size=1, stride=1, padding=0, bias=False)
        self.bn5 = nn.BatchNorm2d(1024)
        self.fc = nn.Linear(1024, num_classes)

    def _make_stage(self, in_channels, out_channels, repeats, groups):
        layers = []
        layers.append(ShuffleNetUnit(in_channels, out_channels, groups))
        for _ in range(1, repeats):
            layers.append(ShuffleNetUnit(out_channels, out_channels, groups))
        return nn.Sequential(*layers)

    def forward(self, x):
        x = F.relu(self.bn1(self.conv1(x)))
        x = self.maxpool(x)
        x = self.stage2(x)
        x = self.stage3(x)
        x = self.stage4(x)
        x = F.relu(self.bn5(self.conv5(x)))
        x = F.adaptive_avg_pool2d(x, (1, 1))
        x = x.view(x.size(0), -1)
        x = self.fc(x)
        return x
'''
