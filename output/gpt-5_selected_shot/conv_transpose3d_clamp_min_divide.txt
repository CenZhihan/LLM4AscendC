kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelClampMinDiv {
public:
    __aicore__ inline KernelClampMinDiv() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR min_t, GM_ADDR div_t, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        minGm.SetGlobalBuffer((__gm__ DTYPE_X *)min_t + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        divGm.SetGlobalBuffer((__gm__ DTYPE_X *)div_t + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_X *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueMin, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueDiv, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> minLocal = inQueueMin.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> divLocal = inQueueDiv.AllocTensor<DTYPE_X>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(minLocal, minGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(divLocal, divGm[progress * this->tileLength], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueMin.EnQue(minLocal);
        inQueueDiv.EnQue(divLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> minLocal = inQueueMin.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> divLocal = inQueueDiv.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> yLocal = outQueueY.AllocTensor<DTYPE_X>();

        AscendC::Maximum(yLocal, xLocal, minLocal, this->tileLength);
        AscendC::Div(yLocal, yLocal, divLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_X>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueMin.FreeTensor(minLocal);
        inQueueDiv.FreeTensor(divLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> yLocal = outQueueY.DeQue<DTYPE_X>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueMin, inQueueDiv;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> minGm;
    AscendC::GlobalTensor<DTYPE_X> divGm;
    AscendC::GlobalTensor<DTYPE_X> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_clamp_min_divide_custom(GM_ADDR x, GM_ADDR min_t, GM_ADDR div_t, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelClampMinDiv op;
    op.Init(x, min_t, div_t, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Transposed 3D convolution followed by fused clamp_min and divide using custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, min_value, divisor):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.min_value = float(min_value)
        self.divisor = float(divisor)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = custom_ops_lib.conv_transpose3d_clamp_min_divide_custom(x, self.min_value, self.divisor)
        return x
