    SgdCustomTilingData tiling;
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();

    float lr = 1e-2f;

    // Try to fetch attributes if available (API style may vary by toolkit version)

    if (context->HasAttr("momentum")) {
        momentum = static_cast<float>(context->GetAttr("momentum")->GetFloat());

    if (context->HasAttr("lr")) {
        lr = static_cast<float>(context->GetAttr("lr")->GetFloat());

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_momentum(momentum);
    tiling.set_lr(lr);

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor buffers per queue

class KernelSgd {
public:
    __aicore__ inline KernelSgd() {}
    __aicore__ inline void Init(GM_ADDR param, GM_ADDR grad, GM_ADDR velocity, GM_ADDR out,
                                uint32_t totalLength, uint32_t tileNum,
                                float momentum, float lr)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->momentum = momentum;
        this->lr = lr;

        paramGm.SetGlobalBuffer((__gm__ DTYPE_X *)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gradGm.SetGlobalBuffer((__gm__ DTYPE_Y *)grad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        velGm.SetGlobalBuffer((__gm__ DTYPE_Z *)velocity + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueParam, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueGrad, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueVel, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(outQueueOut,  BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> paramLocal = inQueueParam.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> gradLocal  = inQueueGrad.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> velLocal   = inQueueVel.AllocTensor<DTYPE_Z>();

        AscendC::DataCopy(paramLocal, paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gradLocal,  gradGm[progress * this->tileLength],  this->tileLength);
        AscendC::DataCopy(velLocal,   velGm[progress * this->tileLength],   this->tileLength);

        inQueueParam.EnQue(paramLocal);
        inQueueGrad.EnQue(gradLocal);
        inQueueVel.EnQue(velLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> paramLocal = inQueueParam.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> gradLocal  = inQueueGrad.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> velLocal   = inQueueVel.DeQue<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> outLocal   = outQueueOut.AllocTensor<DTYPE_Z>();

        // velocity_new = momentum * velocity + grad
        AscendC::Muls(outLocal, velLocal, this->momentum, this->tileLength); // outLocal = momentum * vel
        AscendC::Add(outLocal, outLocal, gradLocal, this->tileLength);       // outLocal = outLocal + grad

        // param_out = param - lr * velocity_new
        AscendC::Muls(outLocal, outLocal, this->lr, this->tileLength);       // outLocal = lr * velocity_new
        AscendC::Sub(outLocal, paramLocal, outLocal, this->tileLength);      // outLocal = param - outLocal

        outQueueOut.EnQue<DTYPE_Z>(outLocal);

        inQueueParam.FreeTensor(paramLocal);
        inQueueGrad.FreeTensor(gradLocal);
        inQueueVel.FreeTensor(velLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.DeQue<DTYPE_Z>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueParam, inQueueGrad, inQueueVel;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut;
    AscendC::GlobalTensor<DTYPE_X> paramGm;
    AscendC::GlobalTensor<DTYPE_Y> gradGm;
    AscendC::GlobalTensor<DTYPE_Z> velGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    float momentum;
    float lr;
};

extern "C" __global__ __aicore__ void sgd_custom(GM_ADDR param, GM_ADDR grad, GM_ADDR velocity, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSgd op;
    op.Init(param, grad, velocity, out,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.momentum, tiling_data.lr);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, momentum=0.9, lr=1e-2):
        super().__init__()
        self.momentum = momentum
        self.lr = lr

    def forward(self, param, grad, velocity):
        # momentum and lr are passed as scalar attributes to the custom kernel
        return custom_ops_lib.sgd_custom(param, grad, velocity, float(self.momentum), float(self.lr))
