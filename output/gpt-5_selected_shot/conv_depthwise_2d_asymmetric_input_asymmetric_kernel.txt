static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;

    auto wi = w_shape->GetOriginShape();
    if (xi.GetDimNum() != 4 || wi.GetDimNum() != 4) {
        return GRAPH_FAILED;
    }

    int64_t N = xi.GetDim(0);
    int64_t C = xi.GetDim(1);
    int64_t H = xi.GetDim(2);
    int64_t W = xi.GetDim(3);

    int64_t KH = wi.GetDim(2);
    int64_t KW = wi.GetDim(3);

    // Read attributes with defaults
    int64_t stride_h = 1, stride_w = 1, padding_h = 0, padding_w = 0, dilation_h = 1, dilation_w = 1;
    auto attrs = context->GetAttrs();
    if (attrs != nullptr) {
        attrs->GetInt("stride_h", stride_h);
        attrs->GetInt("stride_w", stride_w);
        attrs->GetInt("padding_h", padding_h);
        attrs->GetInt("padding_w", padding_w);
        attrs->GetInt("dilation_h", dilation_h);
        attrs->GetInt("dilation_w", dilation_w);
    }

    int64_t kernel_extent_h = dilation_h * (KH - 1) + 1;
    int64_t kernel_extent_w = dilation_w * (KW - 1) + 1;

    int64_t OH = (H + 2 * padding_h - kernel_extent_h) / stride_h + 1;
    int64_t OW = (W + 2 * padding_w - kernel_extent_w) / stride_w + 1;
    if (OH < 0) OH = 0;
    if (OW < 0) OW = 0;

    gert::Shape* y_shape = context->GetOutputShape(0);
    gert::Shape out;
    out.SetDimNum(4);
    out.SetDim(0, N);
    out.SetDim(1, C);
    out.SetDim(2, OH);
    out.SetDim(3, OW);
    *y_shape = out;
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

#ifndef DTYPE_X
typedef float DTYPE_X;
#endif
#ifndef DTYPE_W
typedef float DTYPE_W;
#endif
#ifndef DTYPE_B
typedef float DTYPE_B;
#endif
#ifndef DTYPE_Y
typedef float DTYPE_Y;
#endif

class KernelConvDepthwise2D {
public:
    __aicore__ inline KernelConvDepthwise2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t KH, uint32_t KW,
                                uint32_t SH, uint32_t SW,
                                uint32_t PH, uint32_t PW,
                                uint32_t DH, uint32_t DW,
                                uint32_t hasBias)
    {
        this->N = N; this->C = C; this->H = H; this->W = W;
        this->KH = KH; this->KW = KW;
        this->SH = SH; this->SW = SW;
        this->PH = PH; this->PW = PW;
        this->DH = DH; this->DW = DW;
        this->hasBias = hasBias != 0;

        // Output spatial sizes
        int32_t kernel_extent_h = (int32_t)DH * ((int32_t)KH - 1) + 1;
        int32_t kernel_extent_w = (int32_t)DW * ((int32_t)KW - 1) + 1;
        this->OH = (H + 2 * PH - kernel_extent_h) / SH + 1;
        this->OW = (W + 2 * PW - kernel_extent_w) / SW + 1;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N * C * H * W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)C * KH * KW);
        if (hasBias) {
            bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, (uint64_t)C);
        }
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, (uint64_t)N * C * this->OH * this->OW);

        // Determine channel range for this block
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockId = AscendC::GetBlockIdx();
        uint32_t cPerBlock = (C + blocks - 1) / blocks;
        cStart = cPerBlock * blockId;
        cEnd = cStart + cPerBlock;
        if (cEnd > C) cEnd = C;
    }

    __aicore__ inline void Process()
    {
        // Naive depthwise convolution over assigned channels
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t c = cStart; c < cEnd; ++c) {
                for (uint32_t oh = 0; oh < OH; ++oh) {
                    for (uint32_t ow = 0; ow < OW; ++ow) {
                        DTYPE_Y acc = (DTYPE_Y)0;
                        if (hasBias) {
                            acc = bGm.GetValue(c);
                        }
                        int32_t base_h = (int32_t)oh * (int32_t)SH - (int32_t)PH;
                        int32_t base_w = (int32_t)ow * (int32_t)SW - (int32_t)PW;
                        for (uint32_t kh = 0; kh < KH; ++kh) {
                            int32_t ih = base_h + (int32_t)kh * (int32_t)DH;
                            if (ih < 0 || ih >= (int32_t)H) continue;
                            for (uint32_t kw = 0; kw < KW; ++kw) {
                                int32_t iw = base_w + (int32_t)kw * (int32_t)DW;
                                if (iw < 0 || iw >= (int32_t)W) continue;

                                uint64_t x_idx = ((uint64_t)n * C + c) * (uint64_t)H * W
                                               + (uint64_t)ih * W + (uint64_t)iw;
                                uint64_t w_idx = (uint64_t)c * KH * KW
                                               + (uint64_t)kh * KW + (uint64_t)kw;

                                DTYPE_X xv = xGm.GetValue(x_idx);
                                DTYPE_W wv = wGm.GetValue(w_idx);
                                acc += (DTYPE_Y)(xv * wv);
                            }
                        }
                        uint64_t y_idx = ((uint64_t)n * C + c) * (uint64_t)OH * OW
                                       + (uint64_t)oh * OW + (uint64_t)ow;
                        yGm.SetValue(y_idx, acc);
                    }
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, C, H, W;
    uint32_t KH, KW;
    uint32_t SH, SW;
    uint32_t PH, PW;
    uint32_t DH, DW;
    uint32_t OH, OW;
    bool hasBias;

    uint32_t cStart;
    uint32_t cEnd;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2D op;
    op.Init(x, w, b, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.KH, tiling_data.KW,
            tiling_data.SH, tiling_data.SW,
            tiling_data.PH, tiling_data.PW,
            tiling_data.DH, tiling_data.DW,
            tiling_data.hasBias);
    op.Process();
}

#include <torch/extension.h>

static inline int64_t infer_out_size(int64_t in, int64_t pad, int64_t dil, int64_t k, int64_t stride) {
    int64_t kernel_extent = dil * (k - 1) + 1;
    return (in + 2 * pad - kernel_extent) / stride + 1;
}

at::Tensor conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom_impl_npu(
    const at::Tensor& x,
    const at::Tensor& w,
    c10::optional<at::Tensor> b_opt,
    int64_t stride_h, int64_t stride_w,
    int64_t padding_h, int64_t padding_w,
    int64_t dilation_h, int64_t dilation_w,
    int64_t groups)
{
    TORCH_CHECK(x.dim() == 4, \"x must be 4D NCHW\");
    TORCH_CHECK(w.dim() == 4, \"w must be 4D [C,1,KH,KW] for depthwise\");
    auto N = x.size(0);
    auto C = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);
    auto KH = w.size(2);
    auto KW = w.size(3);
    int64_t OH = infer_out_size(H, padding_h, dilation_h, KH, stride_h);
    int64_t OW = infer_out_size(W, padding_w, dilation_w, KW, stride_w);
    TORCH_CHECK(groups == C, \"depthwise conv expects groups == in_channels\");

    at::Tensor result = at::empty({N, C, OH, OW}, x.options());
    if (b_opt.has_value()) {
        const at::Tensor& b = b_opt.value();
        EXEC_NPU_CMD(aclnnConvDepthwise2dAsymmetricInputAsymmetricKernelCustom,
                     x, w, b, result,
                     stride_h, stride_w,
                     padding_h, padding_w,
                     dilation_h, dilation_w,
                     groups);
    } else {
        // Pass an undefined tensor as bias
        at::Tensor b_undef;
        EXEC_NPU_CMD(aclnnConvDepthwise2dAsymmetricInputAsymmetricKernelCustom,
                     x, w, b_undef, result,
                     stride_h, stride_w,
                     padding_h, padding_w,
                     dilation_h, dilation_w,
                     groups);
    }
    return result;
}
