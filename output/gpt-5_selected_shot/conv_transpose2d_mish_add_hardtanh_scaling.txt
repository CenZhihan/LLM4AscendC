kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add, GM_ADDR scale, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        addGm.SetGlobalBuffer((__gm__ DTYPE_ADD *)add + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_SCALE *)scale + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueAdd, BUFFER_NUM, this->tileLength * sizeof(DTYPE_ADD));
        pipe.InitBuffer(inQueueScale, BUFFER_NUM, this->tileLength * sizeof(DTYPE_SCALE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_ADD> addLocal = inQueueAdd.AllocTensor<DTYPE_ADD>();
        AscendC::LocalTensor<DTYPE_SCALE> scaleLocal = inQueueScale.AllocTensor<DTYPE_SCALE>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(addLocal, addGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(scaleLocal, scaleGm[progress * this->tileLength], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueAdd.EnQue(addLocal);
        inQueueScale.EnQue(scaleLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_ADD> addLocal = inQueueAdd.DeQue<DTYPE_ADD>();
        AscendC::LocalTensor<DTYPE_SCALE> scaleLocal = inQueueScale.DeQue<DTYPE_SCALE>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // Mish: x * tanh(ln(1 + exp(x)))
        AscendC::Exp(yLocal, xLocal, this->tileLength);          // y = exp(x)
        AscendC::Adds(yLocal, yLocal, static_cast<DTYPE_Y>(1.0f), this->tileLength); // y = y + 1
        AscendC::Ln(yLocal, yLocal, this->tileLength);           // y = ln(y)
        AscendC::Tanh(yLocal, yLocal, this->tileLength);         // y = tanh(ln(1+exp(x)))
        AscendC::Mul(yLocal, yLocal, xLocal, this->tileLength);  // y = x * tanh(...)

        // Add constant (vector form) and Hardtanh clamp to [-1, 1]
        AscendC::Add(yLocal, yLocal, addLocal, this->tileLength); // y = y + add
        AscendC::Maxs(yLocal, yLocal, static_cast<DTYPE_Y>(-1.0f), this->tileLength); // y = max(y, -1)
        AscendC::Mins(yLocal, yLocal, static_cast<DTYPE_Y>(1.0f), this->tileLength);  // y = min(y, 1)

        // Scale (vector form)
        AscendC::Mul(yLocal, yLocal, scaleLocal, this->tileLength); // y = y * scale

        outQueueY.EnQue<DTYPE_Y>(yLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueAdd.FreeTensor(addLocal);
        inQueueScale.FreeTensor(scaleLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueAdd, inQueueScale;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_ADD> addGm;
    AscendC::GlobalTensor<DTYPE_SCALE> scaleGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose2d_mish_add_hardtanh_scaling_custom(GM_ADDR x, GM_ADDR add, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, add, scale, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Optimized model: ConvTranspose2d followed by fused Mish + Add + Hardtanh + Scaling custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, add_value, scale):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride, padding, output_padding)
        self.add_value = float(add_value)
        self.scale = float(scale)

    def forward(self, x):
        x = self.conv_transpose(x)
        add_t = torch.full_like(x, self.add_value)
        scale_t = torch.full_like(x, self.scale)
        x = custom_ops_lib.conv_transpose2d_mish_add_hardtanh_scaling_custom(x, add_t, scale_t)
        return x
