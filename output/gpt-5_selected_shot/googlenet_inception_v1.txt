kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConcatChn4 {
public:
    __aicore__ inline KernelConcatChn4() {}

    __aicore__ inline void Init(GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR x4, GM_ADDR y,
                                uint32_t n, uint32_t h, uint32_t w,
                                uint32_t c1, uint32_t c2, uint32_t c3, uint32_t c4,
                                uint32_t tileNum) {
        this->n = n;
        this->h = h;
        this->w = w;
        this->c1 = c1;
        this->c2 = c2;
        this->c3 = c3;
        this->c4 = c4;
        this->tileNum = tileNum;

        this->hw = h * w;
        this->b1 = this->c1 * this->hw;
        this->b2 = this->c2 * this->hw;
        this->b3 = this->c3 * this->hw;
        this->b4 = this->c4 * this->hw;
        this->bout = this->b1 + this->b2 + this->b3 + this->b4;

        uint32_t maxSeg = this->b1;
        if (this->b2 > maxSeg) maxSeg = this->b2;
        if (this->b3 > maxSeg) maxSeg = this->b3;
        if (this->b4 > maxSeg) maxSeg = this->b4;

        // Define a reasonable tile length, at least 1
        this->tileLength = maxSeg / (this->tileNum * BUFFER_NUM);
        if (this->tileLength == 0) this->tileLength = 1;

        x1Gm.SetGlobalBuffer((__gm__ float*)x1, (uint64_t)this->n * this->b1);
        x2Gm.SetGlobalBuffer((__gm__ float*)x2, (uint64_t)this->n * this->b2);
        x3Gm.SetGlobalBuffer((__gm__ float*)x3, (uint64_t)this->n * this->b3);
        x4Gm.SetGlobalBuffer((__gm__ float*)x4, (uint64_t)this->n * this->b4);
        yGm.SetGlobalBuffer((__gm__ float*)y,  (uint64_t)this->n * this->bout);

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t perBlock = (this->n + blockNum - 1) / blockNum;
        uint32_t startBatch = perBlock * blockIdx;
        uint32_t endBatch = startBatch + perBlock;
        if (endBatch > this->n) endBatch = this->n;

        for (uint32_t b = startBatch; b < endBatch; ++b) {
            uint64_t yBase = (uint64_t)b * this->bout;

            // x1 -> y
            CopySegment(x1Gm, yGm, (uint64_t)b * this->b1, yBase + 0, this->b1);
            // x2 -> y
            CopySegment(x2Gm, yGm, (uint64_t)b * this->b2, yBase + this->b1, this->b2);
            // x3 -> y
            CopySegment(x3Gm, yGm, (uint64_t)b * this->b3, yBase + this->b1 + this->b2, this->b3);
            // x4 -> y
            CopySegment(x4Gm, yGm, (uint64_t)b * this->b4, yBase + this->b1 + this->b2 + this->b3, this->b4);
        }
    }

private:
    __aicore__ inline void CopySegment(AscendC::GlobalTensor<float> &src,
                                       AscendC::GlobalTensor<float> &dst,
                                       uint64_t srcBase, uint64_t dstBase, uint32_t len) {
        uint32_t remain = len;
        uint64_t srcOff = 0;
        uint64_t dstOff = 0;

        while (remain > 0) {
            uint32_t cur = remain > this->tileLength ? this->tileLength : remain;

            AscendC::LocalTensor<float> local = inQueue.AllocTensor<float>();
            AscendC::DataCopy(local, src[srcBase + srcOff], cur);
            inQueue.EnQue(local);

            AscendC::LocalTensor<float> l = inQueue.DeQue<float>();
            AscendC::DataCopy(dst[dstBase + dstOff], l, cur);
            inQueue.FreeTensor(l);

            remain -= cur;
            srcOff += cur;
            dstOff += cur;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;

    AscendC::GlobalTensor<float> x1Gm;
    AscendC::GlobalTensor<float> x2Gm;
    AscendC::GlobalTensor<float> x3Gm;
    AscendC::GlobalTensor<float> x4Gm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t n, h, w;
    uint32_t c1, c2, c3, c4;
    uint32_t hw;
    uint32_t b1, b2, b3, b4;
    uint32_t bout;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void googlenet_inception_v1_custom(GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR x4, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConcatChn4 op;
    op.Init(x1, x2, x3, x4, y,
            tiling_data.n, tiling_data.h, tiling_data.w,
            tiling_data.c1, tiling_data.c2, tiling_data.c3, tiling_data.c4,
            tiling_data.tileNum);
    op.Process();
}
"""

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class InceptionModule(nn.Module):
    def __init__(self, in_channels, out_1x1, reduce_3x3, out_3x3, reduce_5x5, out_5x5, pool_proj):
        super(InceptionModule, self).__init__()
        self.branch1x1 = nn.Conv2d(in_channels, out_1x1, kernel_size=1)

        self.branch3x3 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_3x3, kernel_size=1),
            nn.Conv2d(reduce_3x3, out_3x3, kernel_size=3, padding=1)
        )

        self.branch5x5 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_5x5, kernel_size=1),
            nn.Conv2d(reduce_5x5, out_5x5, kernel_size=5, padding=2)
        )

        self.branch_pool = nn.Sequential(
            nn.MaxPool2d(kernel_size=3, stride=1, padding=1),
            nn.Conv2d(in_channels, pool_proj, kernel_size=1)
        )

    def forward(self, x):
        branch1x1 = self.branch1x1(x)
        branch3x3 = self.branch3x3(x)
        branch5x5 = self.branch5x5(x)
        branch_pool = self.branch_pool(x)
        # Replace torch.cat with custom AscendC kernel
        return custom_ops_lib.googlenet_inception_v1_custom(branch1x1, branch3x3, branch5x5, branch_pool)

class ModelNew(nn.Module):
    def __init__(self, num_classes=1000):
        super(ModelNew, self).__init__()
        self.conv1 = nn.Conv2d(3, 64, kernel_size=7, stride=2, padding=3)
        self.maxpool1 = nn.MaxPool2d(3, stride=2, padding=1)
        self.conv2 = nn.Conv2d(64, 64, kernel_size=1)
        self.conv3 = nn.Conv2d(64, 192, kernel_size=3, padding=1)
        self.maxpool2 = nn.MaxPool2d(3, stride=2, padding=1)

        self.inception3a = InceptionModule(192, 64, 96, 128, 16, 32, 32)
        self.inception3b = InceptionModule(256, 128, 128, 192, 32, 96, 64)
        self.maxpool3 = nn.MaxPool2d(3, stride=2, padding=1)

        self.inception4a = InceptionModule(480, 192, 96, 208, 16, 48, 64)
        self.inception4b = InceptionModule(512, 160, 112, 224, 24, 64, 64)
        self.inception4c = InceptionModule(512, 128, 128, 256, 24, 64, 64)
        self.inception4d = InceptionModule(512, 112, 144, 288, 32, 64, 64)
        self.inception4e = InceptionModule(528, 256, 160, 320, 32, 128, 128)
        self.maxpool4 = nn.MaxPool2d(3, stride=2, padding=1)

        self.inception5a = InceptionModule(832, 256, 160, 320, 32, 128, 128)
        self.inception5b = InceptionModule(832, 384, 192, 384, 48, 128, 128)

        self.avgpool = nn.AdaptiveAvgPool2d((1, 1))
        self.dropout = nn.Dropout(0.0)
        self.fc = nn.Linear(1024, num_classes)

    def forward(self, x):
        x = self.maxpool1(F.relu(self.conv1(x)))
        x = F.relu(self.conv2(x))
        x = self.maxpool2(F.relu(self.conv3(x)))

        x = self.inception3a(x)
        x = self.inception3b(x)
        x = self.maxpool3(x)

        x = self.inception4a(x)
        x = self.inception4b(x)
        x = self.inception4c(x)
        x = self.inception4d(x)
        x = self.inception4e(x)
        x = self.maxpool4(x)

        x = self.inception5a(x)
        x = self.inception5b(x)

        x = self.avgpool(x)
        x = torch.flatten(x, 1)
        x = self.dropout(x)
        x = self.fc(x)
        return x
'''
