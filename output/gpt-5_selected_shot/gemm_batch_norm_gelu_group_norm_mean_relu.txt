kernel_src="""
#include "kernel_operator.h"

// Fused GEMM + BatchNorm (inference) + GELU + ReLU
// x: [M, K], w: [N, K] (row-major, equals PyTorch Linear.weight), b: [N]
// bn_*: [N], eps: scalar tensor
// y: [M, N]
constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBnGeluRelu {
public:
    __aicore__ inline KernelGemmBnGeluRelu() {}
    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR bnw, GM_ADDR bnb, GM_ADDR bnm, GM_ADDR bnv, GM_ADDR eps,
        GM_ADDR y, uint32_t m, uint32_t n, uint32_t k, uint32_t tileNum)
    {
        this->m = m;
        this->n = n;
        this->k = k;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint32_t)((uint64_t)m * k));
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint32_t)((uint64_t)n * k));
        bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, n);
        bnwGm.SetGlobalBuffer((__gm__ DTYPE_BNW*)bnw, n);
        bnbGm.SetGlobalBuffer((__gm__ DTYPE_BNB*)bnb, n);
        bnmGm.SetGlobalBuffer((__gm__ DTYPE_BNM*)bnm, n);
        bnvGm.SetGlobalBuffer((__gm__ DTYPE_BNV*)bnv, n);
        epsGm.SetGlobalBuffer((__gm__ DTYPE_EPS*)eps, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, (uint32_t)((uint64_t)m * n));

        // Simple tiling along output N dimension
        this->tileN = (n + tileNum - 1) / tileNum;
        this->tileN = (this->tileN + 15) / 16 * 16; // align for vectorization
        if (this->tileN > n) this->tileN = n;

        // Queue buffers
        pipe.InitBuffer(inQueueX, BUFFER_NUM, k * sizeof(DTYPE_X));                // one row of X
        pipe.InitBuffer(inQueueW, BUFFER_NUM, this->tileN * k * sizeof(DTYPE_W));  // W tile [tileN, K]
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileN * sizeof(DTYPE_B));
        pipe.InitBuffer(inQueueBNW, BUFFER_NUM, this->tileN * sizeof(DTYPE_BNW));
        pipe.InitBuffer(inQueueBNB, BUFFER_NUM, this->tileN * sizeof(DTYPE_BNB));
        pipe.InitBuffer(inQueueBNM, BUFFER_NUM, this->tileN * sizeof(DTYPE_BNM));
        pipe.InitBuffer(inQueueBNV, BUFFER_NUM, this->tileN * sizeof(DTYPE_BNV));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileN * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Load eps scalar
        DTYPE_EPS epsLocal;
        AscendC::DataCopy(&epsLocal, epsGm.GetPhyAddr(), 1);

        // Loop over output column tiles
        for (uint32_t tn = 0; tn < tileNum; ++tn) {
            uint32_t colStart = tn * tileN;
            if (colStart >= n) break;
            uint32_t curTileN = AscendC::min(tileN, n - colStart);

            // Load current W/B/BN params tile to local
            AscendC::LocalTensor<DTYPE_W> wTile = inQueueW.AllocTensor<DTYPE_W>();
            AscendC::LocalTensor<DTYPE_B> bTile = inQueueB.AllocTensor<DTYPE_B>();
            AscendC::LocalTensor<DTYPE_BNW> bnwTile = inQueueBNW.AllocTensor<DTYPE_BNW>();
            AscendC::LocalTensor<DTYPE_BNB> bnbTile = inQueueBNB.AllocTensor<DTYPE_BNB>();
            AscendC::LocalTensor<DTYPE_BNM> bnmTile = inQueueBNM.AllocTensor<DTYPE_BNM>();
            AscendC::LocalTensor<DTYPE_BNV> bnvTile = inQueueBNV.AllocTensor<DTYPE_BNV>();

            AscendC::DataCopy(wTile, wGm[colStart * k], curTileN * k);
            AscendC::DataCopy(bTile, bGm[colStart], curTileN);
            AscendC::DataCopy(bnwTile, bnwGm[colStart], curTileN);
            AscendC::DataCopy(bnbTile, bnbGm[colStart], curTileN);
            AscendC::DataCopy(bnmTile, bnmGm[colStart], curTileN);
            AscendC::DataCopy(bnvTile, bnvGm[colStart], curTileN);

            inQueueW.EnQue(wTile);
            inQueueB.EnQue(bTile);
            inQueueBNW.EnQue(bnwTile);
            inQueueBNB.EnQue(bnbTile);
            inQueueBNM.EnQue(bnmTile);
            inQueueBNV.EnQue(bnvTile);

            // Loop over rows of X
            for (uint32_t i = 0; i < m; ++i) {
                // Load x row
                AscendC::LocalTensor<DTYPE_X> xRow = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xRow, xGm[i * k], k);
                inQueueX.EnQue(xRow);

                // Dequeue params and compute
                AscendC::LocalTensor<DTYPE_X> xRowLocal = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.DeQue<DTYPE_W>();
                AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.DeQue<DTYPE_B>();
                AscendC::LocalTensor<DTYPE_BNW> bnwLocal = inQueueBNW.DeQue<DTYPE_BNW>();
                AscendC::LocalTensor<DTYPE_BNB> bnbLocal = inQueueBNB.DeQue<DTYPE_BNB>();
                AscendC::LocalTensor<DTYPE_BNM> bnmLocal = inQueueBNM.DeQue<DTYPE_BNM>();
                AscendC::LocalTensor<DTYPE_BNV> bnvLocal = inQueueBNV.DeQue<DTYPE_BNV>();

                AscendC::LocalTensor<DTYPE_Y> yTile = outQueueY.AllocTensor<DTYPE_Y>();

                // GEMM partial: y = xRow @ wLocal^T + b
                for (uint32_t j = 0; j < curTileN; ++j) {
                    // dot(xRow, wLocal[j, :])
                    DTYPE_ACC acc = (DTYPE_ACC)0;
                    for (uint32_t kk = 0; kk < k; ++kk) {
                        acc += (DTYPE_ACC)xRowLocal.GetValue(kk) * (DTYPE_ACC)wLocal.GetValue(j * k + kk);
                    }
                    DTYPE_ACC val = acc + (DTYPE_ACC)bLocal.GetValue(j);

                    // BatchNorm inference fusion: scale = gamma / sqrt(var + eps)
                    DTYPE_ACC scale = (DTYPE_ACC)bnwLocal.GetValue(j) / AscendC::Sqrt((DTYPE_ACC)bnvLocal.GetValue(j) + (DTYPE_ACC)epsLocal);
                    DTYPE_ACC shift = (DTYPE_ACC)bnbLocal.GetValue(j) - (DTYPE_ACC)bnmLocal.GetValue(j) * scale;
                    DTYPE_ACC bn_out = val * scale + shift;

                    // GELU (approx): 0.5*x*(1+tanh(0.79788456*(x+0.044715*x^3)))
                    DTYPE_ACC x1 = bn_out;
                    DTYPE_ACC x3 = x1 * x1 * x1;
                    DTYPE_ACC gelu_inner = (DTYPE_ACC)0.79788456f * (x1 + (DTYPE_ACC)0.044715f * x3);
                    DTYPE_ACC gelu_val = (DTYPE_ACC)0.5f * x1 * ((DTYPE_ACC)1.0f + AscendC::Tanh(gelu_inner));

                    // ReLU
                    DTYPE_ACC relu_val = gelu_val > (DTYPE_ACC)0 ? gelu_val : (DTYPE_ACC)0;

                    yTile.SetValue(j, (DTYPE_Y)relu_val);
                }

                outQueueY.EnQue(yTile);

                // Write back
                AscendC::LocalTensor<DTYPE_Y> yOut = outQueueY.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[i * n + colStart], yOut, curTileN);
                outQueueY.FreeTensor(yOut);

                // Free local tensors
                inQueueX.FreeTensor(xRowLocal);
                inQueueW.EnQue(wLocal);
                inQueueB.EnQue(bLocal);
                inQueueBNW.EnQue(bnwLocal);
                inQueueBNB.EnQue(bnbLocal);
                inQueueBNM.EnQue(bnmLocal);
                inQueueBNV.EnQue(bnvLocal);
            }

            // Free param tiles
            AscendC::LocalTensor<DTYPE_W> wLocalF = inQueueW.DeQue<DTYPE_W>();
            AscendC::LocalTensor<DTYPE_B> bLocalF = inQueueB.DeQue<DTYPE_B>();
            AscendC::LocalTensor<DTYPE_BNW> bnwLocalF = inQueueBNW.DeQue<DTYPE_BNW>();
            AscendC::LocalTensor<DTYPE_BNB> bnbLocalF = inQueueBNB.DeQue<DTYPE_BNB>();
            AscendC::LocalTensor<DTYPE_BNM> bnmLocalF = inQueueBNM.DeQue<DTYPE_BNM>();
            AscendC::LocalTensor<DTYPE_BNV> bnvLocalF = inQueueBNV.DeQue<DTYPE_BNV>();

            inQueueW.FreeTensor(wLocalF);
            inQueueB.FreeTensor(bLocalF);
            inQueueBNW.FreeTensor(bnwLocalF);
            inQueueBNB.FreeTensor(bnbLocalF);
            inQueueBNM.FreeTensor(bnmLocalF);
            inQueueBNV.FreeTensor(bnvLocalF);
        }
    }

private:
    AscendC::TPipe pipe;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBNW;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBNB;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBNM;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBNV;

    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_BNW> bnwGm;
    AscendC::GlobalTensor<DTYPE_BNB> bnbGm;
    AscendC::GlobalTensor<DTYPE_BNM> bnmGm;
    AscendC::GlobalTensor<DTYPE_BNV> bnvGm;
    AscendC::GlobalTensor<DTYPE_EPS> epsGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t m, n, k;
    uint32_t tileNum;
    uint32_t tileN;
};

extern "C" __global__ __aicore__ void gemm_batch_norm_gelu_group_norm_mean_relu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR bn_weight, GM_ADDR bn_bias, GM_ADDR bn_mean, GM_ADDR bn_var, GM_ADDR eps,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBnGeluRelu op;
    op.Init(x, w, b, bn_weight, bn_bias, bn_mean, bn_var, eps, y,
            tiling_data.m, tiling_data.n, tiling_data.k, tiling_data.tileNum);
    op.Process();
}
