kernel_src="""
#include "kernel_operator.h"

// Naive direct-convolution kernel for stride=(1,1), padding=(0,0), dilation=(1,1), groups=1
// x: [N, C, H, W]  (DTYPE_X)
// w: [OC, C, KH, KW] (DTYPE_Y)
// y: [N, OC, Hout, Wout] (DTYPE_Z)

class KernelConv2D {
public:
    __aicore__ inline KernelConv2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t wd,
                                uint32_t oc, uint32_t kh, uint32_t kw,
                                uint32_t hout, uint32_t wout,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t dil_h, uint32_t dil_w,
                                uint32_t totalLength)
    {
        this->N = n; this->C = c; this->H = h; this->W = wd;
        this->OC = oc; this->KH = kh; this->KW = kw;
        this->Hout = hout; this->Wout = wout;
        this->stride_h = stride_h; this->stride_w = stride_w;
        this->pad_h = pad_h; this->pad_w = pad_w;
        this->dil_h = dil_h; this->dil_w = dil_w;
        this->totalLen = totalLength;

        xPtr = (__gm__ DTYPE_X*)x;
        wPtr = (__gm__ DTYPE_Y*)w;
        yPtr = (__gm__ DTYPE_Z*)y;

        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();

        // Divide work evenly across blocks
        this->basePerBlock = totalLen / blockNum;
        this->rem = totalLen % blockNum;
        // Distribute remainder to last block
        this->start = blockIdx * basePerBlock;
        this->end = (blockIdx == (blockNum - 1)) ? totalLen : (start + basePerBlock);
    }

    __aicore__ inline void Process()
    {
        // For each output element assigned to this block, compute convolution
        for (uint32_t pos = start; pos < end; ++pos) {
            // Decode linear index into (n, oc, oh, ow)
            uint32_t tmp = pos;
            uint32_t ow = tmp % Wout; tmp /= Wout;
            uint32_t oh = tmp % Hout; tmp /= Hout;
            uint32_t oc = tmp % OC;   tmp /= OC;
            uint32_t n  = tmp;

            float acc = 0.0f;

            // Compute input origin for this output pixel
            int32_t ih_base = (int32_t)oh * (int32_t)stride_h - (int32_t)pad_h;
            int32_t iw_base = (int32_t)ow * (int32_t)stride_w - (int32_t)pad_w;

            // Accumulate over all input channels and kernel elements
            for (uint32_t ci = 0; ci < C; ++ci) {
                for (uint32_t kh = 0; kh < KH; ++kh) {
                    int32_t ih = ih_base + (int32_t)kh * (int32_t)dil_h;
                    if (ih < 0 || ih >= (int32_t)H) continue;
                    for (uint32_t kw = 0; kw < KW; ++kw) {
                        int32_t iw = iw_base + (int32_t)kw * (int32_t)dil_w;
                        if (iw < 0 || iw >= (int32_t)W) continue;

                        uint64_t x_index = ((uint64_t)n * C + ci) * (uint64_t)H * (uint64_t)W
                                         + (uint64_t)ih * (uint64_t)W + (uint64_t)iw;
                        uint64_t w_index = ((uint64_t)oc * C + ci) * (uint64_t)KH * (uint64_t)KW
                                         + (uint64_t)kh * (uint64_t)KW + (uint64_t)kw;

                        float xv = (float)xPtr[x_index];
                        float wv = (float)wPtr[w_index];
                        acc += xv * wv;
                    }
                }
            }

            uint64_t y_index = ((uint64_t)n * OC + oc) * (uint64_t)Hout * (uint64_t)Wout
                             + (uint64_t)oh * (uint64_t)Wout + (uint64_t)ow;
            yPtr[y_index] = (DTYPE_Z)acc;
        }
    }

private:
    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_Y* wPtr;
    __gm__ DTYPE_Z* yPtr;

    uint32_t N, C, H, W;
    uint32_t OC, KH, KW;
    uint32_t Hout, Wout;
    uint32_t stride_h, stride_w;
    uint32_t pad_h, pad_w;
    uint32_t dil_h, dil_w;

    uint32_t totalLen;
    uint32_t blockNum;
    uint32_t blockIdx;
    uint32_t basePerBlock;
    uint32_t rem;
    uint32_t start;
    uint32_t end;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.oc, tiling_data.kh, tiling_data.kw,
            tiling_data.hout, tiling_data.wout,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.dil_h, tiling_data.dil_w,
            tiling_data.totalLength);
    op.Process();
}

import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using custom AscendC kernel for standard 2D convolution.
    Constraints in this custom kernel:
      - stride = (1, 1)
      - padding = (0, 0)
      - dilation = (1, 1)
      - groups = 1
      - bias is not supported
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple):
        super().__init__()
        if isinstance(kernel_size, int):
            kh, kw = kernel_size, kernel_size
        else:
            kh, kw = kernel_size
        # Initialize weights similar to nn.Conv2d defaults
        conv_ref = nn.Conv2d(in_channels, out_channels, (kh, kw), bias=False)
        self.weight = nn.Parameter(conv_ref.weight.detach().clone())
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = (kh, kw)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(x, self.weight)
