kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float EPSILON_IN = 1e-5f;

class KernelConv3dMNClampMax {
public:
    __aicore__ inline KernelConv3dMNClampMax() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR multiplier, GM_ADDR clamp_min, GM_ADDR clamp_max, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w, uint32_t dhw, uint32_t tileSize)
    {
        this->N = n;
        this->C = c;
        this->D = d;
        this->H = h;
        this->W = w;
        this->DHW = dhw;
        this->tileSize = tileSize;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)n * c * dhw);
        multGm.SetGlobalBuffer((__gm__ DTYPE_X *)multiplier, (uint64_t)c);
        minGm.SetGlobalBuffer((__gm__ DTYPE_X *)clamp_min, 1);
        maxGm.SetGlobalBuffer((__gm__ DTYPE_X *)clamp_max, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint64_t)n * dhw);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileSize * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueZ, BUFFER_NUM, this->tileSize * sizeof(DTYPE_Z));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileSize * sizeof(DTYPE_Z));

        // Load clamp scalars once
        AscendC::LocalTensor<DTYPE_X> tmpMin = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> tmpMax = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(tmpMin, minGm[0], 1);
        AscendC::DataCopy(tmpMax, maxGm[0], 1);
        this->cmin = tmpMin.GetValue(0);
        this->cmax = tmpMax.GetValue(0);
        inQueueX.FreeTensor(tmpMin);
        inQueueX.FreeTensor(tmpMax);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t n = 0; n < N; ++n) {
            bool firstChannel = true;
            for (uint32_t c = 0; c < C; ++c) {
                // Load multiplier value m[c]
                float mval = LoadMultiplier(c);

                // First pass: compute mean and variance over DHW
                float sum = 0.0f;
                float sumsq = 0.0f;
                uint32_t processed = 0;
                while (processed < DHW) {
                    uint32_t curTile = AscendC::Min(tileSize, DHW - processed);
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(xLocal, xGm[n * (C * DHW) + c * DHW + processed], curTile);
                    // accumulate
                    for (uint32_t i = 0; i < curTile; ++i) {
                        float xv = static_cast<float>(xLocal.GetValue(i)) * mval;
                        sum += xv;
                        sumsq += xv * xv;
                    }
                    inQueueX.FreeTensor(xLocal);
                    processed += curTile;
                }
                float mean = sum / static_cast<float>(DHW);
                float var = sumsq / static_cast<float>(DHW) - mean * mean;
                float invStd = AscendC::Rsqrt(var + EPSILON_IN);

                // Second pass: normalize, clamp, scale, reduce max over channel
                processed = 0;
                while (processed < DHW) {
                    uint32_t curTile = AscendC::Min(tileSize, DHW - processed);
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(xLocal, xGm[n * (C * DHW) + c * DHW + processed], curTile);

                    AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
                    if (!firstChannel) {
                        // load existing output to compute max
                        AscendC::LocalTensor<DTYPE_Z> zIn = inQueueZ.AllocTensor<DTYPE_Z>();
                        AscendC::DataCopy(zIn, yGm[n * DHW + processed], curTile);
                        for (uint32_t i = 0; i < curTile; ++i) {
                            float xv = static_cast<float>(xLocal.GetValue(i)) * mval;
                            float norm = (xv - mean) * invStd;
                            float clamped = AscendC::MinFloat(AscendC::MaxFloat(norm, cmin), cmax);
                            float val = clamped * mval;
                            float prev = static_cast<float>(zIn.GetValue(i));
                            float outv = (val > prev) ? val : prev;
                            zLocal.SetValue(i, static_cast<DTYPE_Z>(outv));
                        }
                        inQueueZ.FreeTensor(zIn);
                    } else {
                        // initialize output with first channel
                        for (uint32_t i = 0; i < curTile; ++i) {
                            float xv = static_cast<float>(xLocal.GetValue(i)) * mval;
                            float norm = (xv - mean) * invStd;
                            float clamped = AscendC::MinFloat(AscendC::MaxFloat(norm, cmin), cmax);
                            float val = clamped * mval;
                            zLocal.SetValue(i, static_cast<DTYPE_Z>(val));
                        }
                    }

                    // store output tile
                    AscendC::DataCopy(yGm[n * DHW + processed], zLocal, curTile);

                    inQueueX.FreeTensor(xLocal);
                    outQueueZ.FreeTensor(zLocal);
                    processed += curTile;
                }
                firstChannel = false;
            }
        }
    }

private:
    __aicore__ inline float LoadMultiplier(uint32_t cIndex)
    {
        AscendC::LocalTensor<DTYPE_X> mLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(mLocal, multGm[cIndex], 1);
        float mv = static_cast<float>(mLocal.GetValue(0));
        inQueueX.FreeTensor(mLocal);
        return mv;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueZ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> multGm;
    AscendC::GlobalTensor<DTYPE_X> minGm;
    AscendC::GlobalTensor<DTYPE_X> maxGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N, C, D, H, W, DHW, tileSize;
    float cmin, cmax;
};

extern "C" __global__ __aicore__ void conv3d_multiply_instance_norm_clamp_multiply_max_custom(GM_ADDR x, GM_ADDR multiplier, GM_ADDR clamp_min, GM_ADDR clamp_max, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMNClampMax op;
    op.Init(x, multiplier, clamp_min, clamp_max, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.dhw, tiling_data.tileSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Conv3d followed by fused multiply -> instance norm -> clamp -> multiply -> max over channel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape, clamp_min, clamp_max):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        # trainable per-channel multiplier [C,1,1,1]
        self.multiplier = torch.nn.Parameter(torch.randn(multiplier_shape))
        self.clamp_min = float(clamp_min)
        self.clamp_max = float(clamp_max)

    def forward(self, x):
        x = self.conv(x)
        # Fused custom kernel on NPU performs:
        # x = x * multiplier -> instance_norm -> clamp -> x * multiplier -> max over channel
        y = custom_ops_lib.conv3d_multiply_instance_norm_clamp_multiply_max_custom(x, self.multiplier, self.clamp_min, self.clamp_max)
        return y
