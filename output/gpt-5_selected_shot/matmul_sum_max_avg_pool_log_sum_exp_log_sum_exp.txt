kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__
void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t batch = tiling_data.batch;
    const uint32_t inFeat = tiling_data.inFeat;
    const uint32_t outFeat = tiling_data.outFeat;

    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t blockNum = AscendC::GetBlockNum();

    uint32_t rowsPerBlock = (batch + blockNum - 1) / blockNum;
    uint32_t rowStart = blockIdx * rowsPerBlock;
    uint32_t rowEnd = rowStart + rowsPerBlock;
    if (rowEnd > batch) rowEnd = batch;

    __gm__ float* x_g = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_g = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* b_g = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* z_g = reinterpret_cast<__gm__ float*>(z);

    // Compute sum of bias terms once per block
    float b_sum = 0.f;
    for (uint32_t j = 0; j < outFeat; ++j) {
        b_sum += b_g[j];
    }

    for (uint32_t i = rowStart; i < rowEnd; ++i) {
        float out_val = 0.f;

        // Accumulate sum over outputs of (x_i dot W_j^T)
        for (uint32_t j = 0; j < outFeat; ++j) {
            float dot_val = 0.f;
            const uint32_t w_row_offset = j * inFeat;
            const uint32_t x_row_offset = i * inFeat;
            for (uint32_t k = 0; k < inFeat; ++k) {
                dot_val += x_g[x_row_offset + k] * w_g[w_row_offset + k];
            }
            out_val += dot_val;
        }

        // Following ops (max/mean/logsumexp/logsumexp along dim=1) are identities for a single value,
        // so final result is simply out_val + b_sum.
        out_val += b_sum;

        z_g[i] = out_val;
    }
}
