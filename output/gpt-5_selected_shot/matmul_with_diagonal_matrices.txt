host_operator_src="""
#include "matmul_with_diagonal_matrices_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"
#include "tiling/platform/platform_ascendc.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    auto shapeA = context->GetInputTensor(0)->GetOriginShape();
    auto shapeB = context->GetInputTensor(1)->GetOriginShape();

    int32_t N = static_cast<int32_t>(shapeA.GetDim(0));
    int32_t M = static_cast<int32_t>(shapeB.GetDim(1));

    // Simple row-based tiling
    int32_t rowsPerCore = 128;
    int32_t blocks = (N + rowsPerCore - 1) / rowsPerCore;
    if (blocks <= 0) {
        blocks = 1;
    }

    MatmulWithDiagonalMatricesCustomTilingData tiling;
    tiling.set_N(N);
    tiling.set_M(M);
    tiling.set_rowsPerCore(rowsPerCore);

    context->SetBlockDim(blocks);
    context->SetTilingKey(1);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;

    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ops {
class MatmulWithDiagonalMatricesCustom : public OpDef {
public:
    explicit MatmulWithDiagonalMatricesCustom(const char *name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND});
        this->Input("B")
            .ParamType(REQUIRED)

} // namespace ops

kernel_src="""
#include "kernel_operator.h"
#include "matmul_with_diagonal_matrices_custom_tiling.h"

extern "C" __global__ __aicore__ void matmul_with_diagonal_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    int32_t N = tilingData.get_N();
    int32_t M = tilingData.get_M();
    int32_t rowsPerCore = tilingData.get_rowsPerCore();

    int32_t blockIdx = GetBlockIdx();
    int32_t rowStart = blockIdx * rowsPerCore;
    int32_t rowEnd = rowStart + rowsPerCore;
    if (rowEnd > N) {
        rowEnd = N;
    }

    __gm__ float* aPtr = reinterpret_cast<__gm__ float*>(A);
    __gm__ float* bPtr = reinterpret_cast<__gm__ float*>(B);
    __gm__ float* cPtr = reinterpret_cast<__gm__ float*>(C);

    for (int32_t i = rowStart; i < rowEnd; ++i) {
        float scale = aPtr[i];
        int64_t base = static_cast<int64_t>(i) * static_cast<int64_t>(M);
        for (int32_t j = 0; j < M; ++j) {
            cPtr[base + j] = scale * bPtr[base + j];
        }
    }
}
"""
