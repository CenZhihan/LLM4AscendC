kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x,
                                                                   GM_ADDR weight,
                                                                   GM_ADDR bias,
                                                                   GM_ADDR add_bias,
                                                                   GM_ADDR z,
                                                                   GM_ADDR workspace,
                                                                   GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N   = tiling_data.N;
    const uint32_t Cin = tiling_data.Cin;
    const uint32_t H   = tiling_data.H;
    const uint32_t W   = tiling_data.W;
    const uint32_t Cout= tiling_data.Cout;
    const uint32_t kH  = tiling_data.kH;
    const uint32_t kW  = tiling_data.kW;
    const uint32_t Hout= tiling_data.Hout;
    const uint32_t Wout= tiling_data.Wout;
    const uint64_t totalOut = tiling_data.totalOut;

    // Linearize output space and split by blocks
    const uint32_t numBlocks = AscendC::GetBlockNum();
    const uint32_t blockId   = AscendC::GetBlockIdx();
    const uint64_t blockLen  = (totalOut + numBlocks - 1) / numBlocks;
    const uint64_t startIdx  = blockLen * blockId;
    const uint64_t endIdx    = (startIdx + blockLen) < totalOut ? (startIdx + blockLen) : totalOut;

    const float* __restrict__ xPtr   = reinterpret_cast<const float*>(x);
    const float* __restrict__ wPtr   = reinterpret_cast<const float*>(weight);
    const float* __restrict__ bPtr   = reinterpret_cast<const float*>(bias);
    const float* __restrict__ addPtr = reinterpret_cast<const float*>(add_bias);
    float* __restrict__ zPtr         = reinterpret_cast<float*>(z);

    for (uint64_t idx = startIdx; idx < endIdx; ++idx) {
        uint64_t t = idx;
        const uint32_t ow  = static_cast<uint32_t>(t % Wout); t /= Wout;
        const uint32_t oh  = static_cast<uint32_t>(t % Hout); t /= Hout;
        const uint32_t co  = static_cast<uint32_t>(t % Cout); t /= Cout;
        const uint32_t n   = static_cast<uint32_t>(t);

        float acc = bPtr ? bPtr[co] : 0.0f;

        for (uint32_t ci = 0; ci < Cin; ++ci) {
            for (uint32_t kh = 0; kh < kH; ++kh) {
                const uint32_t ih = oh + kh; // stride=1, pad=0
                for (uint32_t kw = 0; kw < kW; ++kw) {
                    const uint32_t iw = ow + kw;

                    const uint64_t inIdx =
                        (((uint64_t)n * Cin + ci) * H + ih) * W + iw;
                    const uint64_t wIdx =
                        ((((uint64_t)co * Cin + ci) * kH + kh) * kW + kw);

                    acc += xPtr[inIdx] * wPtr[wIdx];
                }
            }
        }

        // ReLU
        if (acc < 0.0f) acc = 0.0f;
        // Add broadcast bias (Cout,1,1)
        if (addPtr) acc += addPtr[co];

        const uint64_t outIdx =
            (((uint64_t)n * Cout + co) * Hout + oh) * Wout + ow;
        zPtr[outIdx] = acc;
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv2d_relu_bias_add_custom_impl_npu(const at::Tensor& x,
                                                const at::Tensor& weight,
                                                const at::Tensor& bias,
                                                const at::Tensor& add_bias) {
    TORCH_CHECK(x.dim() == 4, "Input x must be NCHW");
    TORCH_CHECK(weight.dim() == 4, "Weight must be [Cout, Cin, kH, kW]");
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);

    auto Cout = weight.size(0);
    auto kH = weight.size(2);
    auto kW = weight.size(3);

    TORCH_CHECK(weight.size(1) == Cin, "Cin mismatch between x and weight");

    // stride=1, padding=0
    auto Hout = H - kH + 1;
    auto Wout = W - kW + 1;
    TORCH_CHECK(Hout >= 0 && Wout >= 0, "Invalid output size, consider padding");

    at::Tensor result = at::empty({N, Cout, Hout, Wout}, x.options());
    EXEC_NPU_CMD(aclnnConv2dReluBiasAddCustom, x, weight, bias, add_bias, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_relu_bias_add_custom", &conv2d_relu_bias_add_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_relu_bias_add_custom", &conv2d_relu_bias_add_custom_impl_npu, "Conv2D + ReLU + BiasAdd fused");
