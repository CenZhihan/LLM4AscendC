kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv_transposed_1d_asymmetric_input_square_kernel_padded_strided_dilated_custom(
    GM_ADDR x,
    GM_ADDR weight,
    GM_ADDR bias,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    // Retrieve tiling parameters
    const int32_t N   = static_cast<int32_t>(tiling_data.n);
    const int32_t Cin = static_cast<int32_t>(tiling_data.cin);
    const int32_t Cout= static_cast<int32_t>(tiling_data.cout);
    const int32_t Lin = static_cast<int32_t>(tiling_data.lin);
    const int32_t Lout= static_cast<int32_t>(tiling_data.lout);
    const int32_t K   = static_cast<int32_t>(tiling_data.k);
    const int32_t S   = static_cast<int32_t>(tiling_data.stride);
    const int32_t P   = static_cast<int32_t>(tiling_data.padding);
    const int32_t D   = static_cast<int32_t>(tiling_data.dilation);
    const int32_t hasBias = static_cast<int32_t>(tiling_data.hasBias);

    // Process only by one block to avoid overlap in this simple kernel implementation
    if (AscendC::GetBlockIdx() != 0) {
        return;
    }

    __gm__ float* x_gm  = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_gm  = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* b_gm  = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* y_gm  = reinterpret_cast<__gm__ float*>(y);

    // Initialize output to zero (or bias if provided)
    for (int32_t n = 0; n < N; ++n) {
        for (int32_t oc = 0; oc < Cout; ++oc) {
            for (int32_t p = 0; p < Lout; ++p) {
                int64_t y_off = static_cast<int64_t>(n) * Cout * Lout + static_cast<int64_t>(oc) * Lout + p;
                float val = 0.0f;
                if (hasBias) {
                    val = b_gm[oc];
                }
                y_gm[y_off] = val;
            }
        }
    }

    // Deconvolution (transposed conv) accumulation
    // weight layout: [Cin, Cout, K]
    for (int32_t n = 0; n < N; ++n) {
        for (int32_t ic = 0; ic < Cin; ++ic) {
            for (int32_t i = 0; i < Lin; ++i) {
                float xval = x_gm[ static_cast<int64_t>(n) * Cin * Lin + static_cast<int64_t>(ic) * Lin + i ];
                // Contribute to output positions
                for (int32_t k = 0; k < K; ++k) {
                    int32_t out_pos = i * S - P + k * D;
                    if (out_pos < 0 || out_pos >= Lout) continue;
                    // for all oc
                    for (int32_t oc = 0; oc < Cout; ++oc) {
                        float wval = w_gm[ static_cast<int64_t>(ic) * Cout * K + static_cast<int64_t>(oc) * K + k ];
                        int64_t y_off = static_cast<int64_t>(n) * Cout * Lout + static_cast<int64_t>(oc) * Lout + out_pos;
                        y_gm[y_off] += xval * wval;
                    }
                }
            }
        }
    }
}
