kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelIndexCopy {
public:
    __aicore__ inline KernelIndexCopy() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, GM_ADDR src, GM_ADDR z,
                                uint32_t totalLength, uint32_t numRows, uint32_t rowSize,
                                uint32_t numUpdates, uint32_t tileNum)
    {
        this->totalLength = totalLength;
        this->numRows     = numRows;
        this->rowSize     = rowSize;
        this->numUpdates  = numUpdates;
        this->tileNum     = tileNum;

        // For simplicity, use single block
        this->blockLength = totalLength; // all elements handled by block 0
        this->tileLength  = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, this->blockLength);
        srcGm.SetGlobalBuffer((__gm__ DTYPE_X *)src, (uint32_t)(numUpdates * rowSize));
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, this->blockLength);
        idxGm.SetGlobalBuffer((__gm__ int32_t *)indices, this->numUpdates);

        pipe.InitBuffer(copyInQ,  BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(copyOutQ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));

        // buffers for update rows and indices
        pipe.InitBuffer(updateInQ,  1, this->rowSize * sizeof(DTYPE_X));
        pipe.InitBuffer(updateOutQ, 1, this->rowSize * sizeof(DTYPE_Z));
        pipe.InitBuffer(idxQ,       1, sizeof(int32_t));
    }

    __aicore__ inline void Process()
    {
        // Phase 1: copy x -> z entirely (single block for determinism)
        if (AscendC::GetBlockIdx() == 0) {
            int32_t loopCount = this->tileNum * BUFFER_NUM;
            for (int32_t i = 0; i < loopCount; i++) {
                CopyIn(i);
                ComputeCopy(i);
                CopyOut(i);
            }
        }
        AscendC::PipeBarrier(); // ensure copy finished before updates

        // Phase 2: apply index_copy updates in order (single block)
        if (AscendC::GetBlockIdx() == 0) {
            for (uint32_t i = 0; i < this->numUpdates; ++i) {
                // load index scalar
                AscendC::LocalTensor<int32_t> idxLocal = idxQ.AllocTensor<int32_t>();
                AscendC::DataCopy(idxLocal, idxGm[i], 1);
                idxQ.EnQue(idxLocal);
                AscendC::LocalTensor<int32_t> idxValLocal = idxQ.DeQue<int32_t>();
                int32_t rowIndex = idxValLocal.GetValue(0);
                idxQ.FreeTensor(idxValLocal);

                // load src row
                AscendC::LocalTensor<DTYPE_X> rowLocalIn = updateInQ.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(rowLocalIn, srcGm[(uint32_t)i * this->rowSize], this->rowSize);
                updateInQ.EnQue(rowLocalIn);

                // prepare local output row and copy to GM position
                AscendC::LocalTensor<DTYPE_X> rowIn2 = updateInQ.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Z> rowLocalOut = updateOutQ.AllocTensor<DTYPE_Z>();
                AscendC::DataCopy(rowLocalOut, rowIn2, this->rowSize);
                updateOutQ.EnQue(rowLocalOut);
                updateInQ.FreeTensor(rowIn2);

                AscendC::LocalTensor<DTYPE_Z> rowOut2 = updateOutQ.DeQue<DTYPE_Z>();
                AscendC::DataCopy(zGm[(uint32_t)rowIndex * this->rowSize], rowOut2, this->rowSize);
                updateOutQ.FreeTensor(rowOut2);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = copyInQ.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[(uint32_t)progress * this->tileLength], this->tileLength);
        copyInQ.EnQue(xLocal);
    }

    __aicore__ inline void ComputeCopy(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = copyInQ.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = copyOutQ.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(zLocal, xLocal, this->tileLength);
        copyOutQ.EnQue<DTYPE_Z>(zLocal);
        copyInQ.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = copyOutQ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[(uint32_t)progress * this->tileLength], zLocal, this->tileLength);
        copyOutQ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    // queues for full copy
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> copyInQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> copyOutQ;
    // queues for update rows and indices
    AscendC::TQue<AscendC::TPosition::VECIN,  1> updateInQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> updateOutQ;
    AscendC::TQue<AscendC::TPosition::VECIN,  1> idxQ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> srcGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    AscendC::GlobalTensor<int32_t> idxGm;

    uint32_t totalLength;
    uint32_t numRows;
    uint32_t rowSize;
    uint32_t numUpdates;
    uint32_t tileNum;
    uint32_t blockLength;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void index_copy_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR src, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelIndexCopy op;
    op.Init(x, indices, src, z,
            tiling_data.totalLength,
            tiling_data.numRows,
            tiling_data.rowSize,
            tiling_data.numUpdates,
            tiling_data.tileNum);
    op.Process();
}
"""
