kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE_X;
typedef float DTYPE_B;
typedef float DTYPE_S;
typedef float DTYPE_Z;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR b, GM_ADDR y, GM_ADDR s, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        sGm.SetGlobalBuffer((__gm__ DTYPE_S *)s, 1); // scalar scale

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_B));
        pipe.InitBuffer(inQueueS, 1, sizeof(DTYPE_S)); // only one scalar buffer needed
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));

        // preload scale once
        AscendC::LocalTensor<DTYPE_S> sLocal = inQueueS.AllocTensor<DTYPE_S>();
        AscendC::DataCopy(sLocal, sGm, 1);
        inQueueS.EnQue(sLocal);
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(bLocal, bGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.DeQue<DTYPE_B>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_S> sLocal = inQueueS.DeQue<DTYPE_S>();

        // y = clamp(clamp(x + b, 0,1) * s, 0,1) / s
        AscendC::Add(yLocal, xLocal, bLocal, this->tileLength);

        // clamp to [0,1]
        AscendC::Maxs(yLocal, yLocal, (DTYPE_Z)0.0f, this->tileLength);
        AscendC::Mins(yLocal, yLocal, (DTYPE_Z)1.0f, this->tileLength);

        // multiply by s
        AscendC::Muls(yLocal, yLocal, (DTYPE_Z)sLocal.GetValue(0), this->tileLength);

        // clamp to [0,1] again
        AscendC::Maxs(yLocal, yLocal, (DTYPE_Z)0.0f, this->tileLength);
        AscendC::Mins(yLocal, yLocal, (DTYPE_Z)1.0f, this->tileLength);

        // divide by s
        AscendC::Divs(yLocal, yLocal, (DTYPE_Z)sLocal.GetValue(0), this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
        // keep sLocal in queue for reuse
        inQueueS.EnQue(sLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>        inQueueS;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    AscendC::GlobalTensor<DTYPE_S> sGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose2d_bias_add_clamp_scaling_clamp_divide_custom(GM_ADDR x, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling, GM_ADDR s) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, b, y, s, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
