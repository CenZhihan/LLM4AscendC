namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CrossModalAttentionCustomTilingData tiling;

    auto q_shape = context->GetInputShape(0)->GetOriginShape();
    auto k_shape = context->GetInputShape(1)->GetOriginShape();

    uint32_t batch = (q_shape.GetDimNum() >= 3) ? static_cast<uint32_t>(q_shape.GetDim(0)) : 1;
    uint32_t m = (q_shape.GetDimNum() >= 3) ? static_cast<uint32_t>(q_shape.GetDim(1)) : 1;
    uint32_t kdim = (q_shape.GetDimNum() >= 3) ? static_cast<uint32_t>(q_shape.GetDim(2)) : 1;
    uint32_t n = (k_shape.GetDimNum() >= 3) ? static_cast<uint32_t>(k_shape.GetDim(1)) : 1;

    uint32_t blockDim = batch < 8 ? batch : 8;
    if (blockDim == 0) blockDim = 1;

    tiling.set_batch(batch);
    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_kdim(kdim);

    context->SetBlockDim(blockDim);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* q_shape = context->GetInputShape(0);
    const gert::Shape* k_shape = context->GetInputShape(1);
    gert::Shape* out_shape = context->GetOutputShape(0);

    if (q_shape->GetDimNum() >= 3 && k_shape->GetDimNum() >= 3) {
        // [B*H, Lq, Dh] x [B*H, Lk, Dh] -> [B*H, Lq, Lk]
        out_shape->SetDimNum(3);
        out_shape->SetDim(0, q_shape->GetDim(0));
        out_shape->SetDim(1, q_shape->GetDim(1));
        out_shape->SetDim(2, k_shape->GetDim(1));
    } else {
        *out_shape = *q_shape;
    }
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void cross_modal_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR attn_logits, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    uint32_t batch = tiling_data.batch;   // B*H
    uint32_t M = tiling_data.m;          // Lq
    uint32_t N = tiling_data.n;          // Lk
    uint32_t K = tiling_data.kdim;       // Dh

    __gm__ DTYPE_X* qPtr = reinterpret_cast<__gm__ DTYPE_X*>(q);
    __gm__ DTYPE_Y* kPtr = reinterpret_cast<__gm__ DTYPE_Y*>(k);
    __gm__ DTYPE_Z* oPtr = reinterpret_cast<__gm__ DTYPE_Z*>(attn_logits);

    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t blockNum = AscendC::GetBlockNum();

    for (uint32_t b = blockIdx; b < batch; b += (blockNum == 0 ? 1 : blockNum)) {
        uint64_t q_base = static_cast<uint64_t>(b) * M * K;
        uint64_t k_base = static_cast<uint64_t>(b) * N * K;
        uint64_t o_base = static_cast<uint64_t>(b) * M * N;

        for (uint32_t i = 0; i < M; ++i) {
            uint64_t q_row_off = q_base + static_cast<uint64_t>(i) * K;
            for (uint32_t j = 0; j < N; ++j) {
                uint64_t k_row_off = k_base + static_cast<uint64_t>(j) * K;

                float acc = 0.0f;
                for (uint32_t t = 0; t < K; ++t) {
                    float qv = static_cast<float>(qPtr[q_row_off + t]);
                    float kv = static_cast<float>(kPtr[k_row_off + t]);
                    acc += qv * kv;
                }
                oPtr[o_base + static_cast<uint64_t>(i) * N + j] = static_cast<DTYPE_Z>(acc);
            }
        }
    }
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Cross-modal attention with custom kernel for QK^T logits.
    """
    def __init__(self, d_model_text=256, d_model_img=512, num_heads=8):
        super().__init__()
        self.d_model_text = d_model_text
        self.d_model_img = d_model_img
        self.num_heads = num_heads
        self.q_proj = torch.nn.Linear(d_model_text, d_model_text)
        self.kv_proj = torch.nn.Linear(d_model_img, d_model_text)

    def forward(self, text, img):
        # Projections
        q = self.q_proj(text)   # [B, Lq, D]
        kv = self.kv_proj(img)  # [B, Lk, D] used as both K and V

        B, Lq, D = q.shape
        _, Lk, _ = kv.shape
        H = self.num_heads
        Dh = D // H

        # Reshape to multi-head: [B*H, L, Dh]
        qh = q.view(B, Lq, H, Dh).permute(0, 2, 1, 3).reshape(B * H, Lq, Dh).contiguous()
        kh = kv.view(B, Lk, H, Dh).permute(0, 2, 1, 3).reshape(B * H, Lk, Dh).contiguous()
        vh = kh  # value == key as per original architecture

        # Custom kernel computes attention logits: [B*H, Lq, Lk]
        attn_logits = custom_ops_lib.cross_modal_attention_custom(qh, kh)

        # Scaled softmax and value aggregation
        scale = (Dh ** -0.5)
        attn = torch.softmax(attn_logits * scale, dim=-1)
        oh = torch.bmm(attn, vh)  # [B*H, Lq, Dh]

        # Merge heads back: [B, Lq, D]
        out = oh.view(B, H, Lq, Dh).permute(0, 2, 1, 3).reshape(B, Lq, D).contiguous()
        return out
'''
