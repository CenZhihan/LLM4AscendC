#include "kernel_operator.h"
#include <cmath>

using DTYPE = float;

class KernelMatmulScaleResidualAddClampLogSumExpMish {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scale, GM_ADDR cmin, GM_ADDR cmax,
                                GM_ADDR z, uint32_t batch, uint32_t inDim, uint32_t outDim, uint32_t /*tileNum*/)
    {
        this->batch = batch;
        this->inDim = inDim;
        this->outDim = outDim;

        xPtr = reinterpret_cast<__gm__ DTYPE*>(x);
        wPtr = reinterpret_cast<__gm__ DTYPE*>(w);     // [outDim, inDim]
        bPtr = reinterpret_cast<__gm__ DTYPE*>(b);     // [outDim]
        sPtr = reinterpret_cast<__gm__ DTYPE*>(scale); // [1]
        mnPtr = reinterpret_cast<__gm__ DTYPE*>(cmin); // [1]
        mxPtr = reinterpret_cast<__gm__ DTYPE*>(cmax); // [1]
        zPtr = reinterpret_cast<__gm__ DTYPE*>(z);     // [batch,1]

        rowsPerBlock = batch / AscendC::GetBlockNum();
        startRow = rowsPerBlock * AscendC::GetBlockIdx();
    }

    __aicore__ inline void Process()
    {
        DTYPE scale = sPtr[0];
        DTYPE cmin = mnPtr[0];
        DTYPE cmax = mxPtr[0];

        for (uint32_t r = 0; r < rowsPerBlock; ++r) {
            uint32_t rowIdx = startRow + r;
            const __gm__ DTYPE* xRow = xPtr + static_cast<uint64_t>(rowIdx) * inDim;

            // Pass 1: find max for log-sum-exp
            DTYPE maxv = -1.0e30f;
            for (uint32_t j = 0; j < outDim; ++j) {
                const __gm__ DTYPE* wRow = wPtr + static_cast<uint64_t>(j) * inDim;
                DTYPE acc = 0.f;
                for (uint32_t k = 0; k < inDim; ++k) {
                    acc += xRow[k] * wRow[k];
                }
                acc += bPtr[j];
                // scale then residual add with itself: x = x * scale; x = x + x -> 2 * scale * acc
                DTYPE v = (acc * scale) + (acc * scale);
                // clamp
                if (v < cmin) v = cmin;
                if (v > cmax) v = cmax;
                if (v > maxv) maxv = v;
            }

            // Pass 2: accumulate exp
            DTYPE sumExp = 0.f;
            for (uint32_t j = 0; j < outDim; ++j) {
                const __gm__ DTYPE* wRow = wPtr + static_cast<uint64_t>(j) * inDim;
                DTYPE acc = 0.f;
                for (uint32_t k = 0; k < inDim; ++k) {
                    acc += xRow[k] * wRow[k];
                }
                acc += bPtr[j];
                DTYPE v = (acc * scale) + (acc * scale);
                if (v < cmin) v = cmin;
                if (v > cmax) v = cmax;
                sumExp += expf(v - maxv);
            }
            DTYPE lse = logf(sumExp) + maxv;

            // mish(lse) = lse * tanh(softplus(lse)), softplus(x)=log(1+exp(x))
            DTYPE sp = logf(1.0f + expf(lse));
            DTYPE mish = lse * tanhf(sp);
            DTYPE outv = lse * mish;

            zPtr[rowIdx] = outv; // store at (rowIdx, 0)
        }
    }

private:
    __gm__ DTYPE* xPtr;
    __gm__ DTYPE* wPtr;
    __gm__ DTYPE* bPtr;
    __gm__ DTYPE* sPtr;
    __gm__ DTYPE* mnPtr;
    __gm__ DTYPE* mxPtr;
    __gm__ DTYPE* zPtr;

    uint32_t batch;
    uint32_t inDim;
    uint32_t outDim;

    uint32_t rowsPerBlock;
    uint32_t startRow;
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scale, GM_ADDR clamp_min, GM_ADDR clamp_max,
    GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScaleResidualAddClampLogSumExpMish op;
    op.Init(x, weight, bias, scale, clamp_min, clamp_max, z,
            tiling_data.batch, tiling_data.inDim, tiling_data.outDim, tiling_data.tileNum);
    op.Process();
}
