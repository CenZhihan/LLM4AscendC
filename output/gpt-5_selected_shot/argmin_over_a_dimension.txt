#include "argmin_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint64_t ProductExcept(const std::vector<int64_t>& dims, size_t begin, size_t end)
{
    uint64_t p = 1;
    for (size_t i = begin; i < end; ++i) p *= static_cast<uint64_t>(dims[i]);
    return p;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ArgminOverADimensionCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();
    size_t rank = origin.GetDimNum();
    std::vector<int64_t> dims(rank);
    for (size_t i = 0; i < rank; ++i) {
        dims[i] = static_cast<int64_t>(origin.GetDim(i));
    }

    int64_t dim = 0;
    // Try to fetch dim attribute (fallback to 0 if unavailable)
    if (context->GetAttrs() != nullptr) {
        (void)context->GetAttrs()->GetInt("dim", dim);
    }

    if (dim < 0) dim += static_cast<int64_t>(rank);
    if (dim < 0) dim = 0;
    if (dim >= static_cast<int64_t>(rank)) dim = static_cast<int64_t>(rank) - 1;

    uint64_t reduceLen = static_cast<uint64_t>(dims[dim]);
    uint64_t stride = ProductExcept(dims, dim + 1, rank);
    uint64_t totalSize = 1;
    for (size_t i = 0; i < rank; ++i) totalSize *= static_cast<uint64_t>(dims[i]);
    uint64_t outerSize = (reduceLen == 0) ? 0 : (totalSize / reduceLen);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_outerSize(outerSize);
    tiling.set_stride(stride);
    tiling.set_reduceLen(static_cast<uint32_t>(reduceLen));
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    auto origin = x_shape->GetOriginShape();
    int64_t rank = static_cast<int64_t>(origin.GetDimNum());
    std::vector<int64_t> dims(rank);
    for (int64_t i = 0; i < rank; ++i) {
        dims[i] = static_cast<int64_t>(origin.GetDim(i));
    }

    int64_t dim = 0;
    if (context->GetAttrs() != nullptr) {
        (void)context->GetAttrs()->GetInt("dim", dim);
    }
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    std::vector<int64_t> out_dims;
    out_dims.reserve(std::max<int64_t>(1, rank - 1));
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out_dims.push_back(dims[i]);
    }
    if (out_dims.empty()) out_dims.push_back(1); // scalar -> 1D size 1 for safety

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, static_cast<uint64_t>(out_dims[i]));
    }

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output indices dtype is int64
    context->SetOutputDataType(0, ge::DT_INT64);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

// This kernel computes argmin along a specified dimension using tiling over outer elements.
// Tiling data provides: outerSize (# of output elements), reduceLen (length along reduced dim),
// stride (distance between consecutive elements along reduced dim in the flat memory),
// tileNum (tiles per block).

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmin {
public:
    __aicore__ inline KernelArgmin() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, uint64_t outerSize, uint32_t reduceLen, uint64_t stride, uint32_t tileNum)
    {
        this->outerSize = outerSize;
        this->reduceLen = reduceLen;
        this->stride = stride;
        this->tileNum = tileNum;

        this->blockOuter = outerSize / AscendC::GetBlockNum();
        this->tileOuter = (this->blockOuter == 0) ? 0 : (this->blockOuter / tileNum / BUFFER_NUM);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->blockOuter * reduceLen * AscendC::GetBlockIdx(), this->blockOuter * reduceLen);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_Z*)indices + this->blockOuter * AscendC::GetBlockIdx(), this->blockOuter);

        // Minimal local buffers for scalar reads/writes
        pipe.InitBuffer(inQueueX, BUFFER_NUM, sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueIdx, BUFFER_NUM, sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (this->blockOuter == 0 || this->tileOuter == 0) {
            // Fallback: process all outer elements sequentially in this block
            SequentialProcess();
            return;
        }

        int32_t loopCount = this->tileNum * BUFFER_NUM;
        uint64_t baseOuter = 0;
        for (int32_t i = 0; i < loopCount; ++i) {
            uint64_t startOuter = i * this->tileOuter;
            uint64_t endOuter = startOuter + this->tileOuter;
            if (endOuter > this->blockOuter) endOuter = this->blockOuter;
            ComputeRange(startOuter, endOuter);
        }
    }

private:
    __aicore__ inline void SequentialProcess()
    {
        for (uint64_t o = 0; o < this->blockOuter; ++o) {
            ComputeOne(o);
        }
    }

    __aicore__ inline void ComputeRange(uint64_t startOuter, uint64_t endOuter)
    {
        for (uint64_t o = startOuter; o < endOuter; ++o) {
            ComputeOne(o);
        }
    }

    __aicore__ inline void ComputeOne(uint64_t o)
    {
        // Compute argmin for outer index o across reduceLen with stride.
        // Flat base offset of first element in this outer slice relative to xGm buffer.
        uint64_t base = o * this->reduceLen * ((this->stride == 0) ? 1 : 1); // stride accounted in indexing
        // Initialize min
        DTYPE_X minVal;
        // Read first element
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[base + 0 * this->stride], 1);
        inQueueX.EnQue(xLocal);
        xLocal = inQueueX.DeQue<DTYPE_X>();
        minVal = xLocal.GetValue(0);
        inQueueX.FreeTensor(xLocal);
        int64_t minIdx = 0;

        for (uint32_t k = 1; k < this->reduceLen; ++k) {
            AscendC::LocalTensor<DTYPE_X> xTmp = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xTmp, xGm[base + k * this->stride], 1);
            inQueueX.EnQue(xTmp);
            xTmp = inQueueX.DeQue<DTYPE_X>();
            DTYPE_X v = xTmp.GetValue(0);
            inQueueX.FreeTensor(xTmp);
            if (v < minVal) {
                minVal = v;
                minIdx = static_cast<int64_t>(k);
            }
        }

        AscendC::LocalTensor<DTYPE_Z> idxLocal = outQueueIdx.AllocTensor<DTYPE_Z>();
        // Store index
        idxLocal.SetValue(0, static_cast<DTYPE_Z>(minIdx));
        outQueueIdx.EnQue(idxLocal);
        AscendC::LocalTensor<DTYPE_Z> idxLocalOut = outQueueIdx.DeQue<DTYPE_Z>();
        AscendC::DataCopy(idxGm[o], idxLocalOut, 1);
        outQueueIdx.FreeTensor(idxLocalOut);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueIdx;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> idxGm;

    uint64_t outerSize;
    uint64_t blockOuter;
    uint64_t tileOuter;
    uint64_t stride;
    uint32_t reduceLen;
    uint32_t tileNum;
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmin op;
    op.Init(x, indices, tiling_data.outerSize, tiling_data.reduceLen, tiling_data.stride, tiling_data.tileNum);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline std::vector<int64_t> infer_argmin_output_shape(const at::Tensor& self, int64_t dim) {
    auto sizes = self.sizes().vec();
    int64_t rank = static_cast<int64_t>(sizes.size());
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;
    std::vector<int64_t> out;
    out.reserve(rank > 0 ? rank - 1 : 1);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out.push_back(sizes[i]);
    }
    if (out.empty()) out.push_back(1);
    return out;
}

at::Tensor argmin_over_a_dimension_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    auto out_sizes = infer_argmin_output_shape(self, dim);
    auto options = self.options().dtype(at::kLong);
    at::Tensor result = at::empty(out_sizes, options);
    EXEC_NPU_CMD(aclnnArgminOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmin_over_a_dimension_custom", &argmin_over_a_dimension_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmin_over_a_dimension_custom", &argmin_over_a_dimension_custom_impl_npu, "argmin over a dimension");
}
