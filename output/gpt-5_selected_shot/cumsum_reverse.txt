    CumsumReverseCustomTilingData tiling;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // buffers per queue

class KernelCumsumReverse {
public:
    __aicore__ inline KernelCumsumReverse() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outerRows, uint32_t lastDim, uint32_t /*tileNum*/) {
        this->outerRows = outerRows;
        this->lastDim = lastDim;

        // Initialize global memory tensors
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint32_t>(outerRows * lastDim));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, static_cast<uint32_t>(outerRows * lastDim));

        // Buffer size per row
        uint32_t rowBytes = this->lastDim * sizeof(DTYPE_X);
        pipe.InitBuffer(inQueue, BUFFER_NUM, rowBytes);
        pipe.InitBuffer(outQueue, BUFFER_NUM, rowBytes);
    }

    __aicore__ inline void Process() {
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();

        uint32_t rowsPerBlock = (outerRows + blocks - 1) / blocks;
        uint32_t startRow = rowsPerBlock * bid;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > outerRows) endRow = outerRows;

        for (uint32_t r = startRow; r < endRow; ++r) {
            CopyInRow(r);
            ComputeRow();
            CopyOutRow(r);
        }
    }

private:
    __aicore__ inline void CopyInRow(uint32_t rowIdx) {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->lastDim], this->lastDim);
        inQueue.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRow() {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.AllocTensor<DTYPE_Y>();

        DTYPE_Y acc = static_cast<DTYPE_Y>(0);
        // Reverse cumulative sum along the last dimension
        for (int64_t i = static_cast<int64_t>(this->lastDim) - 1; i >= 0; --i) {
            DTYPE_Y val = static_cast<DTYPE_Y>(xLocal[i]);
            acc = acc + val;
            yLocal[i] = acc;
        }

        outQueue.EnQue<DTYPE_Y>(yLocal);
        inQueue.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t rowIdx) {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[rowIdx * this->lastDim], yLocal, this->lastDim);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t outerRows;
    uint32_t lastDim;
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverse op;
    op.Init(x, y, tiling_data.outerRows, tiling_data.lastDim, tiling_data.tileNum);
    op.Process();
}
"""
