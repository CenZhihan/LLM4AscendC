host_operator_src="""
#include "frobenius_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context) {
  FrobeniusNormCustomTilingData tiling;
  const gert::StorageShape* x_shape_ptr = context->GetInputShape(0);
  const gert::Shape shape = x_shape_ptr->GetStorageShape();

  uint32_t total = 1;
  for (int i = 0; i < shape.GetDimNum(); ++i) {
    total *= static_cast<uint32_t>(shape.GetDim(i));
  }

  const float* epsilonAttr = context->GetAttrs()->GetAttrPointer<float>(0);
  tiling.set_eps(*epsilonAttr);

  // Choose a safe tile length for local buffer usage
  uint32_t tileLength = 4096; // elements per tile
  uint32_t loopCount = total / tileLength;
  uint32_t leftLength = total % tileLength;

  tiling.set_totalLength(total);
  tiling.set_tileLength(tileLength);
  tiling.set_loopCount(loopCount);
  tiling.set_leftLength(leftLength);

  context->SetBlockDim(BLOCK_DIM);
  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
  context->SetTilingKey(1);
  return ge::GRAPH_SUCCESS;
}
}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context) {
  const gert::Shape* x_shape = context->GetInputShape(0);
  gert::Shape* y_shape = context->GetOutputShape(0);
  *y_shape = *x_shape;
  return GRAPH_SUCCESS;
}
}  // namespace ge

namespace ops {
class FrobeniusNormCustom : public OpDef {
 public:
  explicit FrobeniusNormCustom(const char* name) : OpDef(name) {
    this->Input("x")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});
    this->Output("res_out")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});
    this->Attr("epsilon").AttrType(OPTIONAL).Float(1e-12);

    this->SetInferShape(ge::InferShape);

    this->AICore().SetTiling(optiling::TilingFunc);
    this->AICore().AddConfig("ascend910b");
  }
};

OP_ADD(FrobeniusNormCustom);
}

kernel_src="""
#include "kernel_operator.h"
constexpr int32_t BUFFER_NUM = 1;

class KernelFrobeniusNorm {
 public:
  __aicore__ inline KernelFrobeniusNorm() {}
  __aicore__ inline void InitTiling(GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    totalLength = tiling_data.totalLength;
    tileLength = tiling_data.tileLength;
    loopCount = tiling_data.loopCount;
    leftLength = tiling_data.leftLength;
    eps = tiling_data.eps;
  }

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, GM_ADDR tiling) {
    InitTiling(tiling);
    xGm.SetGlobalBuffer((__gm__ float*)x, totalLength);
    zGm.SetGlobalBuffer((__gm__ float*)z, totalLength);

    pipe.InitBuffer(xBuffer, tileLength * sizeof(float));
    pipe.InitBuffer(tmpReduceBuffer1, 64 * sizeof(float));
    pipe.InitBuffer(tmpReduceBuffer2, 1 * sizeof(float));
    pipe.InitBuffer(oneBuffer, 1 * sizeof(float));
  }

  __aicore__ inline void Process() {
    float sumsq = 0.0f;

    // First pass: compute sum of squares
    uint32_t offset = 0;
    for (uint32_t i = 0; i < loopCount; ++i) {
      AccumulateTile(offset, tileLength, sumsq);
      offset += tileLength;
    }
    if (leftLength > 0) {
      AccumulateTile(offset, leftLength, sumsq);
      offset += leftLength;
    }

    // Compute inverse norm: 1 / (sqrt(sumsq) + eps)
    AscendC::LocalTensor<float> tmp2 = tmpReduceBuffer2.Get<float>();
    AscendC::Duplicate<float>(tmp2, sumsq, 1);
    AscendC::Sqrt(tmp2, tmp2, 1);
    AscendC::Adds(tmp2, tmp2, eps, 1);
    AscendC::LocalTensor<float> ones = oneBuffer.Get<float>();
    AscendC::Duplicate<float>(ones, 1.0f, 1);
    AscendC::Div(tmp2, ones, tmp2, 1);
    float inv_norm = tmp2.GetValue(0);

    // Second pass: normalize
    offset = 0;
    for (uint32_t i = 0; i < loopCount; ++i) {
      NormalizeTile(offset, tileLength, inv_norm);
      offset += tileLength;
    }
    if (leftLength > 0) {
      NormalizeTile(offset, leftLength, inv_norm);
      offset += leftLength;
    }
  }

 private:
  __aicore__ inline void AccumulateTile(uint32_t offset, uint32_t curLen, float& sumsq) {
    AscendC::LocalTensor<float> xLocal = xBuffer.Get<float>();
    AscendC::DataCopy(xLocal, xGm[offset], curLen);
    // Square in place: xLocal = xLocal * xLocal
    AscendC::Mul(xLocal, xLocal, xLocal, curLen);

    AscendC::LocalTensor<float> tmp1 = tmpReduceBuffer1.Get<float>();
    AscendC::LocalTensor<float> tmp2 = tmpReduceBuffer2.Get<float>();
    AscendC::ReduceSum<float>(tmp2[0], xLocal, tmp1, curLen);
    sumsq += tmp2.GetValue(0);
  }

  __aicore__ inline void NormalizeTile(uint32_t offset, uint32_t curLen, float inv_norm) {
    AscendC::LocalTensor<float> xLocal = xBuffer.Get<float>();
    AscendC::DataCopy(xLocal, xGm[offset], curLen);
    AscendC::Muls(xLocal, xLocal, inv_norm, curLen);
    AscendC::DataCopy(zGm[offset], xLocal, curLen);
  }

 private:
  AscendC::TPipe pipe;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> xBuffer;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> tmpReduceBuffer1;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> tmpReduceBuffer2;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> oneBuffer;

  AscendC::GlobalTensor<float> xGm;
  AscendC::GlobalTensor<float> zGm;

  uint32_t totalLength = 0;
  uint32_t tileLength = 0;
  uint32_t loopCount = 0;
  uint32_t leftLength = 0;
  float eps = 1e-12f;
};

extern "C" __global__ __aicore__ void frobenius_norm_custom(
    GM_ADDR x, GM_ADDR res_out, GM_ADDR workspace, GM_ADDR tiling) {
  KernelFrobeniusNorm op;
  op.Init(x, res_out, tiling);
  if (TILING_KEY_IS(1)) {
    op.Process();
  }
}
"""
