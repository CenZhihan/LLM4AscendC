    int64_t s = 1;

    if (k <= 0) k = 1;
    if (p < 0) p = 0;

    int64_t tmp = static_cast<int64_t>(L) + 2 * p - k;
    uint64_t outL = tmp >= 0 ? static_cast<uint64_t>(tmp / s + 1) : 0;

    tiling.set_rows(static_cast<uint32_t>(N * C));

    tiling.set_outLength(static_cast<uint32_t>(outL));

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();
    if (origin.GetDimNum() < 3) {
        return GRAPH_FAILED;
    }

    int64_t k = 0;
    int64_t s = 1;
    int64_t p = 0;
    (void)context->GetAttrInt("kernel_size", k);
    (void)context->GetAttrInt("stride", s);
    (void)context->GetAttrInt("padding", p);
    if (s <= 0) s = 1;
    if (k <= 0) k = 1;
    if (p < 0) p = 0;

    int64_t L = origin.GetDim(2);
    int64_t tmp = L + 2 * p - k;
    int64_t outL = tmp >= 0 ? (tmp / s + 1) : 0;

    gert::Shape* out_shape = context->GetOutputShape(0);
    out_shape->SetDimNum(3);
    out_shape->SetDim(0, origin.GetDim(0));
    out_shape->SetDim(1, origin.GetDim(1));
    out_shape->SetDim(2, outL);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelAvgPool1D {
public:
    __aicore__ inline KernelAvgPool1D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalElems, uint32_t rows, uint32_t rowLength, uint32_t outLength,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding, uint32_t tileNum)
    {
        this->rows = rows;
        this->rowLength = rowLength;
        this->outLength = outLength;
        this->kernelSize = kernelSize;
        this->stride = stride;
        this->padding = padding;

        this->blockRows = rows / AscendC::GetBlockNum();
        this->rowOffset = this->blockRows * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)this->rowOffset * this->rowLength,
                            (uint64_t)this->blockRows * this->rowLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + (uint64_t)this->rowOffset * this->outLength,
                            (uint64_t)this->blockRows * this->outLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->rowLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->outLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->blockRows; ++r) {
            CopyInRow(r);
            ComputeRow(r);
            CopyOutRow(r);
        }
    }

private:
    __aicore__ inline void CopyInRow(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[(uint64_t)r * this->rowLength], this->rowLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRow(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        for (uint32_t j = 0; j < this->outLength; ++j) {
            float sumVal = 0.0f;
            int32_t start = (int32_t)j * (int32_t)this->stride - (int32_t)this->padding;
            for (uint32_t t = 0; t < this->kernelSize; ++t) {
                int32_t idx = start + (int32_t)t;
                if (idx >= 0 && (uint32_t)idx < this->rowLength) {
                    sumVal += (float)xLocal.GetValue((uint32_t)idx);
                }
            }
            float avg = sumVal / (float)this->kernelSize;
            yLocal.SetValue(j, (DTYPE_Y)avg);
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[(uint64_t)r * this->outLength], yLocal, this->outLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows;
    uint32_t rowLength;
    uint32_t outLength;
    uint32_t kernelSize;
    uint32_t stride;
    uint32_t padding;

    uint32_t blockRows;
    uint32_t rowOffset;
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAvgPool1D op;
    op.Init(x, y,
            tiling_data.totalElems,
            tiling_data.rows,
            tiling_data.rowLength,
            tiling_data.outLength,
            tiling_data.kernelSize,
            tiling_data.stride,
            tiling_data.padding,
            tiling_data.tileNum);
    op.Process();
}

at::Tensor average_pooling_1d_custom_impl_npu(const at::Tensor& x, int64_t kernel_size, int64_t stride, int64_t padding) {

                                  (x.size(2) + 2 * padding - kernel_size) / (stride <= 0 ? 1 : stride) + 1},

    return result;
