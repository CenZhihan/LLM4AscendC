namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    AdamCustomTilingData tiling;

    uint32_t totalLength = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // read attributes with fallbacks
    float beta1 = get_attr_f(context, "beta1", 0.9f);
    float beta2 = get_attr_f(context, "beta2", 0.999f);
    float lr    = get_attr_f(context, "lr", 1e-3f);
    float eps   = get_attr_f(context, "eps", 1e-8f);
    int64_t step = get_attr_i(context, "step", 1);

    // bias correction denominators inverse
    float inv_mcorr = 1.0f / (1.0f - std::pow(beta1, static_cast<float>(step)));
    float inv_vcorr = 1.0f / (1.0f - std::pow(beta2, static_cast<float>(step)));

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_beta1(beta1);
    tiling.set_beta2(beta2);
    tiling.set_lr(lr);
    tiling.set_eps(eps);
    tiling.set_inv_mcorr(inv_mcorr);
    tiling.set_inv_vcorr(inv_vcorr);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor buffers per queue

class KernelAdam {
public:
    __aicore__ inline KernelAdam() {}

    __aicore__ inline void Init(GM_ADDR param, GM_ADDR grad, GM_ADDR m, GM_ADDR v, GM_ADDR out,
                                uint32_t totalLength, uint32_t tileNum,
                                float beta1, float beta2, float lr, float eps,
                                float inv_mcorr, float inv_vcorr)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        paramGm.SetGlobalBuffer((__gm__ float *)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gradGm.SetGlobalBuffer((__gm__ float *)grad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mGm.SetGlobalBuffer((__gm__ float *)m + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ float *)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ float *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inParamQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inGradQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inMQ,    BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inVQ,    BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQ,    BUFFER_NUM, this->tileLength * sizeof(float));
        // temp queues for intermediate tensors
        pipe.InitBuffer(tmpQ1,   BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQ2,   BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQ3,   BUFFER_NUM, this->tileLength * sizeof(float));

        this->beta1 = beta1;
        this->beta2 = beta2;
        this->one_minus_b1 = 1.0f - beta1;
        this->one_minus_b2 = 1.0f - beta2;
        this->lr = lr;
        this->eps = eps;
        this->inv_mcorr = inv_mcorr;
        this->inv_vcorr = inv_vcorr;
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> pLocal = inParamQ.AllocTensor<float>();
        AscendC::LocalTensor<float> gLocal = inGradQ.AllocTensor<float>();
        AscendC::LocalTensor<float> mLocal = inMQ.AllocTensor<float>();
        AscendC::LocalTensor<float> vLocal = inVQ.AllocTensor<float>();

        AscendC::DataCopy(pLocal, paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gLocal, gradGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(mLocal, mGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(vLocal, vGm[progress * this->tileLength], this->tileLength);

        inParamQ.EnQue(pLocal);
        inGradQ.EnQue(gLocal);
        inMQ.EnQue(mLocal);
        inVQ.EnQue(vLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> pLocal = inParamQ.DeQue<float>();
        AscendC::LocalTensor<float> gLocal = inGradQ.DeQue<float>();
        AscendC::LocalTensor<float> mLocal = inMQ.DeQue<float>();
        AscendC::LocalTensor<float> vLocal = inVQ.DeQue<float>();
        AscendC::LocalTensor<float> outLocal = outQ.AllocTensor<float>();

        // allocate temporaries
        AscendC::LocalTensor<float> t1 = tmpQ1.AllocTensor<float>();
        AscendC::LocalTensor<float> t2 = tmpQ2.AllocTensor<float>();
        AscendC::LocalTensor<float> t3 = tmpQ3.AllocTensor<float>();

        // Create scalar vectors
        AscendC::LocalTensor<float> b1Vec  = tmpQ1.AllocTensor<float>();
        AscendC::LocalTensor<float> ob1Vec = tmpQ2.AllocTensor<float>();
        AscendC::LocalTensor<float> b2Vec  = tmpQ3.AllocTensor<float>();
        AscendC::LocalTensor<float> ob2Vec = tmpQ1.AllocTensor<float>();
        AscendC::LocalTensor<float> lrVec  = tmpQ2.AllocTensor<float>();
        AscendC::LocalTensor<float> epsVec = tmpQ3.AllocTensor<float>();
        AscendC::LocalTensor<float> imcVec = tmpQ1.AllocTensor<float>();
        AscendC::LocalTensor<float> ivcVec = tmpQ2.AllocTensor<float>();

        AscendC::Dup(b1Vec,  this->beta1, this->tileLength);
        AscendC::Dup(ob1Vec, this->one_minus_b1, this->tileLength);
        AscendC::Dup(b2Vec,  this->beta2, this->tileLength);
        AscendC::Dup(ob2Vec, this->one_minus_b2, this->tileLength);
        AscendC::Dup(lrVec,  this->lr, this->tileLength);
        AscendC::Dup(epsVec, this->eps, this->tileLength);
        AscendC::Dup(imcVec, this->inv_mcorr, this->tileLength);
        AscendC::Dup(ivcVec, this->inv_vcorr, this->tileLength);

        // m_new = beta1 * m + (1 - beta1) * grad
        AscendC::Mul(t1, mLocal, b1Vec, this->tileLength);
        AscendC::Mul(t2, gLocal, ob1Vec, this->tileLength);
        AscendC::Add(t3, t1, t2, this->tileLength); // t3 = m_new

        // v_new = beta2 * v + (1 - beta2) * grad^2
        AscendC::Mul(t1, gLocal, gLocal, this->tileLength); // t1 = grad^2
        AscendC::Mul(t2, vLocal, b2Vec, this->tileLength);  // t2 = beta2 * v
        AscendC::Mul(t1, t1, ob2Vec, this->tileLength);     // t1 = (1-beta2) * grad^2
        AscendC::Add(t1, t2, t1, this->tileLength);         // t1 = v_new

        // m_hat = m_new * inv_mcorr
        AscendC::Mul(t2, t3, imcVec, this->tileLength);     // t2 = m_hat

        // v_hat = v_new * inv_vcorr
        AscendC::Mul(t3, t1, ivcVec, this->tileLength);     // t3 = v_hat

        // denom = sqrt(v_hat) + eps
        AscendC::Sqrt(t1, t3, this->tileLength);            // t1 = sqrt(v_hat)
        AscendC::Add(t1, t1, epsVec, this->tileLength);     // t1 = sqrt(v_hat) + eps

        // update = m_hat / denom
        AscendC::Div(t2, t2, t1, this->tileLength);         // t2 = update

        // out = param - lr * update
        AscendC::Mul(t2, t2, lrVec, this->tileLength);      // t2 = lr * update
        AscendC::Sub(outLocal, pLocal, t2, this->tileLength);

        outQ.EnQue<float>(outLocal);

        // free local tensors
        inParamQ.FreeTensor(pLocal);
        inGradQ.FreeTensor(gLocal);
        inMQ.FreeTensor(mLocal);
        inVQ.FreeTensor(vLocal);

        tmpQ1.FreeTensor(t1);
        tmpQ2.FreeTensor(t2);
        tmpQ3.FreeTensor(t3);

        tmpQ1.FreeTensor(b1Vec);
        tmpQ2.FreeTensor(ob1Vec);
        tmpQ3.FreeTensor(b2Vec);
        tmpQ1.FreeTensor(ob2Vec);
        tmpQ2.FreeTensor(lrVec);
        tmpQ3.FreeTensor(epsVec);
        tmpQ1.FreeTensor(imcVec);
        tmpQ2.FreeTensor(ivcVec);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> outLocal = outQ.DeQue<float>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQ.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inParamQ, inGradQ, inMQ, inVQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQ;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> tmpQ1, tmpQ2, tmpQ3;

    AscendC::GlobalTensor<float> paramGm;
    AscendC::GlobalTensor<float> gradGm;
    AscendC::GlobalTensor<float> mGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> outGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    float beta1;
    float beta2;
    float one_minus_b1;
    float one_minus_b2;
    float lr;
    float eps;
    float inv_mcorr;
    float inv_vcorr;
};

extern "C" __global__ __aicore__ void adam_custom(GM_ADDR param, GM_ADDR grad, GM_ADDR m, GM_ADDR v, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAdam op;
    op.Init(param, grad, m, v, out,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.beta1, tiling_data.beta2, tiling_data.lr, tiling_data.eps,
            tiling_data.inv_mcorr, tiling_data.inv_vcorr);
    op.Process();
}
