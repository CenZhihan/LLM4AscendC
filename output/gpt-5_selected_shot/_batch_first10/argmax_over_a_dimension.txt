}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    // Set output shape as input shape with the last dimension removed.
    auto x_origin = x_shape->GetOriginShape();
    std::vector<int64_t> dims;

        dims.push_back(x_origin.GetDim(i));

    for (size_t i = 0; i < dims.size(); ++i) {
        out.SetDim(i, dims[i]);

    y_shape->SetOriginShape(out);
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // number of rows processed per queue entry

class KernelArgmaxLastDim {
public:
    __aicore__ inline KernelArgmaxLastDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t outerSize, uint32_t reduceSize, uint32_t tileNum)
    {
        this->outerSize = outerSize;      // number of rows
        this->reduceSize = reduceSize;    // length of last dimension to reduce
        this->tileNum = tileNum;

        // Distribute rows across blocks
        this->blockOuter = (outerSize + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t startRow = this->blockOuter * AscendC::GetBlockIdx();
        uint32_t endRow = AscendC::GetBlockIdx() == AscendC::GetBlockNum() - 1 ? outerSize : startRow + this->blockOuter;
        this->rowsThisBlock = endRow > startRow ? (endRow - startRow) : 0;
        this->blockStartRow = startRow;

        // GM buffers: x is contiguous rows of length reduceSize; z is 1 index per row
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + static_cast<uint64_t>(this->blockStartRow) * this->reduceSize,
                            static_cast<uint64_t>(this->rowsThisBlock) * this->reduceSize);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockStartRow, this->rowsThisBlock);

        // Local buffers sized for one row of input and one index output per iteration
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->reduceSize * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsThisBlock; ++r) {
            CopyIn(r);
            Compute();
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // Copy a full row (contiguous along last dimension)
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->reduceSize], this->reduceSize);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // Find argmax over the row.
        // Initialize with the first element.
        DTYPE_X maxVal = xLocal.GetValue(0);
        int64_t maxIdx = 0;
        for (uint32_t i = 1; i < this->reduceSize; ++i) {
            DTYPE_X v = xLocal.GetValue(i);
            if (v > maxVal) {
                maxVal = v;
                maxIdx = static_cast<int64_t>(i);
            }
        }
        // Store index (output dtype is int64 via DTYPE_Z)
        zLocal.SetValue(0, static_cast<DTYPE_Z>(maxIdx));

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[rowIdx], zLocal, 1);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t outerSize;
    uint32_t reduceSize;

    uint32_t tileNum;
    uint32_t blockOuter;
    uint32_t rowsThisBlock;
    uint32_t blockStartRow;
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxLastDim op;
    op.Init(x, z, tiling_data.outerSize, tiling_data.reduceSize, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Model that performs Argmax over a specified dimension using a custom AscendC kernel.
    The custom kernel computes argmax over the last dimension. If the desired dimension
    is not the last, the input tensor is permuted so that the target dimension becomes last.
    """
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        dim = self.dim
        if dim < 0:
            dim = x.dim() + dim
        # Move the target dimension to the last position if necessary
        if dim != x.dim() - 1:
            perm = [i for i in range(x.dim()) if i != dim] + [dim]
            x = x.permute(perm).contiguous()
        # Call the custom kernel that reduces the last dimension
        return custom_ops_lib.argmax_over_a_dimension_custom(x)
