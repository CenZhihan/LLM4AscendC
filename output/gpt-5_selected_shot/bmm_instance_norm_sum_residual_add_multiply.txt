kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;
constexpr float EPSILON = 1e-5f;

class KernelBmmInstanceNormSumResidualAddMultiply {
public:
    __aicore__ inline KernelBmmInstanceNormSumResidualAddMultiply() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR w, GM_ADDR b, GM_ADDR z,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures, uint32_t tileNum)
    {
        this->batchSize = batchSize;
        this->inFeatures = inFeatures;
        this->outFeatures = outFeatures;
        this->tileNum = tileNum;

        // Set GM buffers (no block partition for simplicity; each block processes a share of rows)
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)((uint64_t)batchSize * inFeatures));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint32_t)((uint64_t)batchSize * outFeatures));
        wGm.SetGlobalBuffer((__gm__ DTYPE_WEIGHT *)w, (uint32_t)((uint64_t)outFeatures * inFeatures));
        bGm.SetGlobalBuffer((__gm__ DTYPE_BIAS *)b, outFeatures);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, (uint32_t)((uint64_t)batchSize * outFeatures));

        // Allocate local buffers per row
        pipe.InitBuffer(rowXQueue, BUFFER_NUM, this->inFeatures * sizeof(DTYPE_X));
        pipe.InitBuffer(rowYQueue, BUFFER_NUM, this->outFeatures * sizeof(DTYPE_Y));
        pipe.InitBuffer(rowZQueue, BUFFER_NUM, this->outFeatures * sizeof(DTYPE_Z));
        pipe.InitBuffer(biasQueue, BUFFER_NUM, this->outFeatures * sizeof(DTYPE_BIAS));
        pipe.InitBuffer(weightRowQueue, BUFFER_NUM, this->inFeatures * sizeof(DTYPE_WEIGHT));

        // Preload bias once to L1 for reuse
        AscendC::LocalTensor<DTYPE_BIAS> bLocal = biasQueue.AllocTensor<DTYPE_BIAS>();
        AscendC::DataCopy(bLocal, bGm[0], this->outFeatures);
        biasQueue.EnQue(bLocal);

        // Compute row range for this block
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (batchSize + blockNum - 1) / blockNum;
        startRow = blockIdx * rowsPerBlock;
        endRow = startRow + rowsPerBlock;
        if (endRow > batchSize) endRow = batchSize;
    }

    __aicore__ inline void Process()
    {
        for (uint32_t row = startRow; row < endRow; ++row) {
            LoadRow(row);
            ComputeRow();
            StoreRow(row);
        }
    }

private:
    __aicore__ inline void LoadRow(uint32_t row)
    {
        // Load x row
        AscendC::LocalTensor<DTYPE_X> xLocal = rowXQueue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[row * this->inFeatures], this->inFeatures);
        rowXQueue.EnQue(xLocal);

        // Load y row
        AscendC::LocalTensor<DTYPE_Y> yLocal = rowYQueue.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(yLocal, yGm[row * this->outFeatures], this->outFeatures);
        rowYQueue.EnQue(yLocal);

        // Prepare z buffer
        AscendC::LocalTensor<DTYPE_Z> zLocal = rowZQueue.AllocTensor<DTYPE_Z>();
        rowZQueue.EnQue(zLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = rowXQueue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = rowYQueue.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = rowZQueue.DeQue<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_BIAS> bLocal = biasQueue.DeQue<DTYPE_BIAS>();

        // GEMV: compute zLocal[j] = dot(xLocal, w[j]) + b[j]
        for (uint32_t j = 0; j < outFeatures; ++j) {
            AscendC::LocalTensor<DTYPE_WEIGHT> wRowLocal = weightRowQueue.AllocTensor<DTYPE_WEIGHT>();
            AscendC::DataCopy(wRowLocal, wGm[j * this->inFeatures], this->inFeatures);
            DTYPE_Z acc = static_cast<DTYPE_Z>(0);
            for (uint32_t k = 0; k < inFeatures; ++k) {
                acc += static_cast<DTYPE_Z>(xLocal.GetValue(k)) * static_cast<DTYPE_Z>(wRowLocal.GetValue(k));
            }
            acc += static_cast<DTYPE_Z>(bLocal.GetValue(j));
            zLocal.SetValue(j, acc);
            weightRowQueue.FreeTensor(wRowLocal);
        }

        // Instance-like normalization over feature dimension (per row)
        // mean
        DTYPE_Z mean = static_cast<DTYPE_Z>(0);
        for (uint32_t j = 0; j < outFeatures; ++j) {
            mean += zLocal.GetValue(j);
        }
        mean = mean / static_cast<DTYPE_Z>(outFeatures);
        // variance
        DTYPE_Z var = static_cast<DTYPE_Z>(0);
        for (uint32_t j = 0; j < outFeatures; ++j) {
            DTYPE_Z diff = zLocal.GetValue(j) - mean;
            var += diff * diff;
        }
        var = var / static_cast<DTYPE_Z>(outFeatures);
        DTYPE_Z invStd = AscendC::Rsqrt(var + static_cast<DTYPE_Z>(EPSILON));

        // normalize, add residual y, then multiply by y
        for (uint32_t j = 0; j < outFeatures; ++j) {
            DTYPE_Z normed = (zLocal.GetValue(j) - mean) * invStd;
            DTYPE_Z yv = static_cast<DTYPE_Z>(yLocal.GetValue(j));
            DTYPE_Z outv = (normed + yv) * yv;
            zLocal.SetValue(j, outv);
        }

        rowXQueue.FreeTensor(xLocal);
        rowYQueue.FreeTensor(yLocal);
        biasQueue.EnQue(bLocal); // keep bias resident
        rowZQueue.EnQue(zLocal);
    }

    __aicore__ inline void StoreRow(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = rowZQueue.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[row * this->outFeatures], zLocal, this->outFeatures);
        rowZQueue.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> rowXQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> rowYQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> biasQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> weightRowQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> rowZQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_WEIGHT> wGm;
    AscendC::GlobalTensor<DTYPE_BIAS> bGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t batchSize;
    uint32_t inFeatures;
    uint32_t outFeatures;
    uint32_t tileNum;
    uint32_t startRow;
    uint32_t endRow;
};

extern "C" __global__ __aicore__ void bmm_instance_norm_sum_residual_add_multiply_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBmmInstanceNormSumResidualAddMultiply op;
    op.Init(x, y, weight, bias, z, tiling_data.batchSize, tiling_data.inFeatures, tiling_data.outFeatures, tiling_data.tileNum);
    op.Process();
}

import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused model: Linear (x @ W^T + b) -> per-row normalization -> + y -> * y
    """
    def __init__(self, in_features, out_features, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()

        # Keep instance norm module to hold hyper-params; computation is fused in custom kernel
        self.instance_norm = torch.nn.InstanceNorm2d(out_features, eps=eps, momentum=momentum)

    def forward(self, x, y):
        # Custom kernel consumes x, y, weight, bias
