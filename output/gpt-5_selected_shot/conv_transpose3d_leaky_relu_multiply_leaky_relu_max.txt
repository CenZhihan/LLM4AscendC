kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR m, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, float alpha)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->alpha = alpha;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mGm.SetGlobalBuffer((__gm__ DTYPE_Y *)m + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueM, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> mLocal = inQueueM.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(mLocal, mGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueM.EnQue(mLocal);
    }
    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> mLocal = inQueueM.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Fused sequence:
        // t1 = LeakyReLU(x, alpha)
        // t2 = t1 * multiplier
        // y  = LeakyReLU(t2, alpha)
        // Note: pooling is handled outside this kernel in the model for simplicity.

        // First LeakyReLU
        AscendC::LeakyRelu(yLocal, xLocal, this->alpha, this->tileLength);
        // Multiply by per-element multiplier (should be pre-broadcasted to match x shape)
        AscendC::Mul(yLocal, yLocal, mLocal, this->tileLength);
        // Second LeakyReLU
        AscendC::LeakyRelu(yLocal, yLocal, this->alpha, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueM.FreeTensor(mLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueM;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> mGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    float alpha;
};

extern "C" __global__ __aicore__ void conv_transpose3d_leaky_relu_multiply_leaky_relu_max_custom(GM_ADDR x, GM_ADDR multiplier, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, multiplier, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.alpha);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, multiplier_shape):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding
        )
        # Learnable per-channel multiplier (expects shape [C, 1, 1, 1])
        self.multiplier = torch.nn.Parameter(torch.randn(multiplier_shape))
        self.max_pool = torch.nn.MaxPool3d(kernel_size=2)
        self.negative_slope = 0.2

    def forward(self, x):
        # 1) ConvTranspose3d
        x = self.conv_transpose(x)
        # 2) Fused: LeakyReLU -> Multiply -> LeakyReLU
        # Broadcast multiplier to match x: [N, C, D, H, W]
        mult = self.multiplier.view(1, -1, 1, 1, 1).expand_as(x)
        x = custom_ops_lib.conv_transpose3d_leaky_relu_multiply_leaky_relu_max_custom(x, mult, self.negative_slope)
        # 3) MaxPool3d (kept in PyTorch for simplicity)
        x = self.max_pool(x)
        return x
