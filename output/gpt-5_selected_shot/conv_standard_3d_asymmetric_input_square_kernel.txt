const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ConvStandard3dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    auto xOrigin = x_shape->GetOriginShape();
    auto wOrigin = w_shape->GetOriginShape();

    // Expect x: [N, Cin, H, W, D], weight: [Cout, Cin, 1, K, K]
    uint32_t N   = static_cast<uint32_t>(xOrigin.GetDim(0));
    uint32_t Cin = static_cast<uint32_t>(xOrigin.GetDim(1));
    uint32_t H   = static_cast<uint32_t>(xOrigin.GetDim(2));
    uint32_t W   = static_cast<uint32_t>(xOrigin.GetDim(3));
    uint32_t D   = static_cast<uint32_t>(xOrigin.GetDim(4));

    uint32_t Cout = static_cast<uint32_t>(wOrigin.GetDim(0));
    uint32_t Kd   = static_cast<uint32_t>(wOrigin.GetDim(2));
    uint32_t K    = static_cast<uint32_t>(wOrigin.GetDim(3)); // square kernel: kh == kw == K

    // Only support Kd == 1, stride=1, padding=0, dilation=1
    if (Kd != 1) {
        return ge::GRAPH_FAILED;
    }

    uint32_t Hout = (H >= K) ? (H - K + 1) : 0;
    uint32_t Wout = (W >= K) ? (W - K + 1) : 0;
    uint32_t Dout = D;

    uint64_t totalOut = static_cast<uint64_t>(N) * Cout * Hout * Wout * Dout;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_inH(H);
    tiling.set_inW(W);
    tiling.set_inD(D);
    tiling.set_k(K);
    tiling.set_hout(Hout);
    tiling.set_wout(Wout);
    tiling.set_dout(Dout);
    tiling.set_totalLength(totalOut);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelConv3dSquare2DSpatial {
public:
    __aicore__ inline KernelConv3dSquare2DSpatial() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t inH, uint32_t inW, uint32_t inD,
                                uint32_t k, uint32_t hout, uint32_t wout, uint32_t dout,
                                uint64_t totalOut)
    {
        this->N = n;
        this->Cin = cin;
        this->Cout = cout;
        this->H = inH;
        this->W = inW;
        this->D = inD;
        this->K = k;
        this->Hout = hout;
        this->Wout = wout;
        this->Dout = dout;
        this->totalOut = totalOut;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)n * cin * inH * inW * inD);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)w, (uint64_t)cout * cin * 1 * k * k);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, totalOut);

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->start = (totalOut * blockIdx) / blockNum;
        this->end   = (totalOut * (blockIdx + 1)) / blockNum;
    }

    __aicore__ inline void Process()
    {
        for (uint64_t idx = start; idx < end; ++idx) {
            uint64_t tmp = idx;
            uint32_t d = (uint32_t)(tmp % Dout); tmp /= Dout;
            uint32_t w = (uint32_t)(tmp % Wout); tmp /= Wout;
            uint32_t h = (uint32_t)(tmp % Hout); tmp /= Hout;
            uint32_t co = (uint32_t)(tmp % Cout); tmp /= Cout;
            uint32_t n  = (uint32_t)tmp;

            DTYPE_Z acc = (DTYPE_Z)0;
            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t kh = 0; kh < K; ++kh) {
                    uint32_t ih = h + kh;
                    for (uint32_t kw = 0; kw < K; ++kw) {
                        uint32_t iw = w + kw;

                        uint64_t in_index =
                            (((((uint64_t)n) * Cin + ci) * H + ih) * W + iw) * D + d;

                        uint64_t w_index =
                            (((((uint64_t)co) * Cin + ci) * 1 + 0) * (uint64_t)K * K) + (uint64_t)kh * K + kw;

                        acc += (DTYPE_Z)(xGm[in_index]) * (DTYPE_Z)(wGm[w_index]);
                    }
                }
            }
            yGm[idx] = acc;
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N, Cin, Cout, H, W, D, K, Hout, Wout, Dout;
    uint64_t totalOut;
    uint64_t start, end;
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dSquare2DSpatial op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.inH, tiling_data.inW, tiling_data.inD,
            tiling_data.k, tiling_data.hout, tiling_data.wout, tiling_data.dout,
            tiling_data.totalLength);
    op.Process();
}

    int64_t Wout = W - K + 1;
    int64_t Dout = D;

    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):

        assert stride == 1 and padding == 0 and dilation == 1 and groups == 1 and not bias, "This custom kernel supports stride=1, padding=0, dilation=1, groups=1, bias=False."
        self.in_channels = in_channels
