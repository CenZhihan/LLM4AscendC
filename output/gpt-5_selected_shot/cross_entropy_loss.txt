kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelCrossEntropyLoss {
public:
    __aicore__ inline KernelCrossEntropyLoss() {}
    __aicore__ inline void Init(GM_ADDR logits, GM_ADDR target, GM_ADDR y,
                                uint32_t batchSize, uint32_t numClasses,
                                uint32_t tileNum, uint32_t tileBatch, uint32_t lastTileBatch) {
        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");

        this->batchSize = batchSize;
        this->numClasses = numClasses;
        this->tileNum = tileNum;
        this->tileBatch = tileBatch;
        this->lastTileBatch = lastTileBatch;

        logitsGm.SetGlobalBuffer((__gm__ float*)logits, (uint32_t)(batchSize * numClasses));
        targetGm.SetGlobalBuffer((__gm__ int32_t*)target, batchSize);
        outGm.SetGlobalBuffer((__gm__ float*)y, 32);

        uint32_t reduce_align = (this->tileNum + 31) / 32 * 32;
        pipe.InitBuffer(inQueue, BUFFER_NUM, this->numClasses * sizeof(float));
        pipe.InitBuffer(tmpVecBuf, this->numClasses * sizeof(float));
        pipe.InitBuffer(redWsVecBuf, this->numClasses * sizeof(float));
        pipe.InitBuffer(lossBuf, reduce_align * sizeof(float));
        pipe.InitBuffer(globalRedWsBuf, reduce_align * sizeof(float));
        pipe.InitBuffer(labelBuf, sizeof(int32_t));
    }

    __aicore__ inline void Process() {
        AscendC::LocalTensor<float> lossLocal = lossBuf.Get<float>();
        // Accumulate per-row loss into lossLocal[progress]
        int32_t loopCount = this->tileNum;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i, lossLocal);
        }
        // Reduce sum over all rows
        AscendC::LocalTensor<float> redWs = globalRedWsBuf.Get<float>();
        uint32_t reduce_align = (this->tileNum + 31) / 32 * 32;
        AscendC::Duplicate(redWs, (float)0, reduce_align);
        AscendC::ReduceSum<float>(lossLocal, lossLocal, redWs, this->tileNum);

        // Mean over batch
        AscendC::LocalTensor<float> denom = inQueue.AllocTensor<float>();
        AscendC::Duplicate(denom, static_cast<float>(this->batchSize), 1);
        AscendC::Div(lossLocal, lossLocal, denom, 1);

        outGm.SetValue(0, lossLocal.GetValue(0));
        inQueue.FreeTensor(denom);
    }

private:
    __aicore__ inline void CopyIn(int32_t rowIdx) {
        AscendC::LocalTensor<float> rowLocal = inQueue.AllocTensor<float>();
        uint32_t offset = (uint32_t)rowIdx * this->numClasses;
        AscendC::DataCopy(rowLocal[0], logitsGm[offset], this->numClasses);
        inQueue.EnQue(rowLocal);
    }

    __aicore__ inline void Compute(int32_t rowIdx, AscendC::LocalTensor<float>& lossLocal) {
        AscendC::LocalTensor<float> row = inQueue.DeQue<float>();
        AscendC::LocalTensor<float> tmp = tmpVecBuf.Get<float>();
        AscendC::LocalTensor<float> redWs = redWsVecBuf.Get<float>();

        // Copy row to tmp for ReduceMax
        AscendC::DataCopy(tmp, row, this->numClasses);
        // Reduce max over classes for numeric stability
        AscendC::ReduceMax<float>(tmp, tmp, redWs, this->numClasses);
        float maxVal = tmp.GetValue(0);
        // Shift by max
        AscendC::Duplicate(redWs, maxVal, this->numClasses);
        AscendC::Sub(row, row, redWs, this->numClasses);

        // Load target index
        AscendC::LocalTensor<int32_t> labelLocal = labelBuf.Get<int32_t>();
        AscendC::DataCopy(labelLocal, targetGm[rowIdx], 1);
        int32_t tgtIdx = labelLocal.GetValue(0);
        // Get shifted logit for target class
        float tgtShifted = row.GetValue(tgtIdx);

        // Compute logsumexp
        AscendC::Exp(row, row, this->numClasses);
        AscendC::ReduceSum<float>(row, row, redWs, this->numClasses);
        AscendC::Log(row, row, 1); // row[0] = log(sum(exp(shifted)))

        float l = row.GetValue(0) - tgtShifted;
        lossLocal.SetValue(rowIdx, l);

        inQueue.FreeTensor(row);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TBuf<> tmpVecBuf;
    AscendC::TBuf<> redWsVecBuf;
    AscendC::TBuf<> lossBuf;
    AscendC::TBuf<> globalRedWsBuf;
    AscendC::TBuf<> labelBuf;

    AscendC::GlobalTensor<float> logitsGm;
    AscendC::GlobalTensor<int32_t> targetGm;
    AscendC::GlobalTensor<float> outGm;

    uint32_t batchSize;
    uint32_t numClasses;
    uint32_t tileNum;
    uint32_t tileBatch;
    uint32_t lastTileBatch;
};

extern "C" __global__ __aicore__ void cross_entropy_loss_custom(GM_ADDR logits, GM_ADDR target, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCrossEntropyLoss op;
    op.Init(logits, target, y,
            tiling_data.batchSize, tiling_data.numClasses,
            tiling_data.tileNum, tiling_data.tileBatch, tiling_data.lastTileBatch);
    op.Process();
}
