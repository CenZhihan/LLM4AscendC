host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *axis_ptr = attrs->GetAttrPointer<int64_t>(0);

    int64_t axis = (axis_ptr == nullptr) ? 1 : *axis_ptr;

    // Assume 2D input for this kernel (N, M) and softmax along dim=1 by default
    uint32_t rows = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(0));
    uint32_t cols = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(1));

    if (axis == 0) {
        // If softmax along dim 0, swap roles logically: we still process per-row,
        // but host maps rows/cols accordingly. For simplicity, keep block mapping by rows,
        // kernel will treat each 'row' as the reduction dimension.
        // Here, we just keep rows/cols as is; a transpose-based handling would be needed
        // for generality. This example targets axis==1 usage.
    }

    uint32_t maxBlocks = 32;
    uint32_t blockDim = rows < maxBlocks ? rows : maxBlocks;

    context->SetBlockDim(blockDim);

    uint32_t rowsPerBlock = (rows + blockDim - 1) / blockDim;

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_axis(static_cast<int32_t>(axis));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0; // no workspace
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("axis").AttrType(OPTIONAL).Int(1);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // double-buffering for pipeline

class KernelSoftmax {
public:
    __aicore__ inline KernelSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t rowsPerBlock, int32_t axis)
    {
        this->rows = rows;
        this->cols = cols;
        this->rowsPerBlock = rowsPerBlock;
        this->axis = axis;

        uint32_t startRow = AscendC::GetBlockIdx() * rowsPerBlock;
        uint32_t maxRow = (startRow + rowsPerBlock) > rows ? rows : (startRow + rowsPerBlock);
        this->rowStart = startRow;
        this->rowCount = (maxRow > startRow) ? (maxRow - startRow) : 0;

        xGm.SetGlobalBuffer((__gm__ float *)x + (uint64_t)startRow * cols, (uint64_t)rowCount * cols);
        yGm.SetGlobalBuffer((__gm__ float *)y + (uint64_t)startRow * cols, (uint64_t)rowCount * cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, cols * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, cols * sizeof(float));
        pipe.InitBuffer(expBuf, cols * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowCount; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t r)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[(uint64_t)r * cols], cols);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t r)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
        AscendC::LocalTensor<float> eLocal = expBuf.Get<float>();

        // Numerically stable softmax along the row
        float maxVal = AscendC::ReduceMax(xLocal, cols);
        AscendC::Subs(eLocal, xLocal, maxVal, cols);   // eLocal = x - max
        AscendC::Exp(eLocal, eLocal, cols);            // eLocal = exp(x - max)
        float sumVal = AscendC::ReduceSum(eLocal, cols);
        float invSum = 1.0f / sumVal;
        AscendC::Muls(yLocal, eLocal, invSum, cols);   // y = exp / sum

        outQueueY.EnQue<float>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t r)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[(uint64_t)r * cols], yLocal, cols);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::QuePosition::VECCALC> expBuf;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<float> xGm, yGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowCount;
    int32_t axis;
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmax op;
    op.Init(x, y,
            tiling_data.rows,
            tiling_data.cols,
            tiling_data.rowsPerBlock,
            tiling_data.axis);
    op.Process();
}
"""
