#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t calc_out_len(uint32_t l_in, uint32_t k, uint32_t s, uint32_t p) {
    // Same as PyTorch: floor((L + 2P - K)/S) + 1
    if (s == 0) s = k;
    int64_t numer = static_cast<int64_t>(l_in) + 2LL * static_cast<int64_t>(p) - static_cast<int64_t>(k);
    if (numer < 0) numer = 0;
    return static_cast<uint32_t>(numer / static_cast<int64_t>(s) + 1);
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaxPooling1dCustomTilingData tiling;

    const auto* in_shape = context->GetInputShape(0);
    auto shape = in_shape->GetOriginShape();
    uint32_t n = (shape.GetDimNum() > 0) ? static_cast<uint32_t>(shape.GetDim(0)) : 1;
    uint32_t c = (shape.GetDimNum() > 1) ? static_cast<uint32_t>(shape.GetDim(1)) : 1;
    uint32_t l_in = (shape.GetDimNum() > 2) ? static_cast<uint32_t>(shape.GetDim(2)) : 1;

    int64_t k_attr = 1;
    (void)context->GetAttr("kernel_size", k_attr);
    int64_t s_attr = 0;
    (void)context->GetAttr("stride", s_attr);
    int64_t p_attr = 0;
    (void)context->GetAttr("padding", p_attr);
    // return_indices ignored in kernel tiling

    uint32_t k = static_cast<uint32_t>(k_attr);
    uint32_t s = static_cast<uint32_t>(s_attr == 0 ? k_attr : s_attr);
    uint32_t p = static_cast<uint32_t>(p_attr);
    uint32_t l_out = calc_out_len(l_in, k, s, p);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_l_in(l_in);
    tiling.set_l_out(l_out);
    tiling.set_kernel_size(k);
    tiling.set_stride(s);
    tiling.set_padding(p);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    auto in_shape = x_shape->GetOriginShape();

    int64_t k_attr = 1;
    (void)context->GetAttr("kernel_size", k_attr);
    int64_t s_attr = 0;
    (void)context->GetAttr("stride", s_attr);
    int64_t p_attr = 0;
    (void)context->GetAttr("padding", p_attr);

    uint32_t n = (in_shape.GetDimNum() > 0) ? static_cast<uint32_t>(in_shape.GetDim(0)) : 1;
    uint32_t c = (in_shape.GetDimNum() > 1) ? static_cast<uint32_t>(in_shape.GetDim(1)) : 1;
    uint32_t l_in = (in_shape.GetDimNum() > 2) ? static_cast<uint32_t>(in_shape.GetDim(2)) : 1;

    uint32_t k = static_cast<uint32_t>(k_attr);
    uint32_t s = static_cast<uint32_t>(s_attr == 0 ? k_attr : s_attr);
    uint32_t p = static_cast<uint32_t>(p_attr);
    uint32_t l_out = (l_in + 2 * p - k) / s + 1;

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, c);
    y_shape->SetDim(2, l_out);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto dt = context->GetInputDataType(0);
    context->SetOutputDataType(0, dt);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#include <cfloat>

class KernelMaxPool1D {
public:
    __aicore__ inline KernelMaxPool1D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t l_in, uint32_t l_out,
                                uint32_t kernel_size, uint32_t stride, uint32_t padding) {
        x_ptr = (__gm__ DTYPE_X*)x;
        y_ptr = (__gm__ DTYPE_Y*)y;
        this->n = n;
        this->c = c;
        this->l_in = l_in;
        this->l_out = l_out;
        this->k = kernel_size;
        this->s = stride == 0 ? kernel_size : stride;
        this->p = padding;

        total_channels = n * c;
        block_num = AscendC::GetBlockNum();
        block_idx = AscendC::GetBlockIdx();
        // Split channels across blocks
        ch_per_block = (total_channels + block_num - 1) / block_num;
        ch_start = block_idx * ch_per_block;
        ch_end = ch_start + ch_per_block;
        if (ch_end > total_channels) ch_end = total_channels;
    }

    __aicore__ inline void Process() {
        // For each (n,c) channel assigned to this block, do 1D max pooling along last dim
        for (uint32_t ch = ch_start; ch < ch_end; ++ch) {
            uint64_t in_base = static_cast<uint64_t>(ch) * static_cast<uint64_t>(l_in);
            uint64_t out_base = static_cast<uint64_t>(ch) * static_cast<uint64_t>(l_out);
            for (uint32_t out_idx = 0; out_idx < l_out; ++out_idx) {
                // window [start, start + k)
                int64_t start = static_cast<int64_t>(out_idx) * static_cast<int64_t>(s) - static_cast<int64_t>(p);
                int64_t end = start + static_cast<int64_t>(k);
                float m = -FLT_MAX;
                for (int64_t t = start; t < end; ++t) {
                    if (t >= 0 && t < static_cast<int64_t>(l_in)) {
                        float v = static_cast<float>(x_ptr[in_base + static_cast<uint64_t>(t)]);
                        m = (v > m) ? v : m;
                    } else {
                        // padded with -inf for max pool
                        // m unchanged
                    }
                }
                y_ptr[out_base + out_idx] = static_cast<DTYPE_Y>(m);
            }
        }
    }

private:
    __gm__ DTYPE_X* x_ptr;
    __gm__ DTYPE_Y* y_ptr;

    uint32_t n, c, l_in, l_out;
    uint32_t k, s, p;

    uint32_t total_channels;
    uint32_t block_num;
    uint32_t block_idx;
    uint32_t ch_per_block;
    uint32_t ch_start, ch_end;
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPool1D op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.l_in, tiling_data.l_out,
            tiling_data.kernel_size, tiling_data.stride, tiling_data.padding);
    op.Process();
}
"""
