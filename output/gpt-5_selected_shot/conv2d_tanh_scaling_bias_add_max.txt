kernel_src="""
#include "kernel_operator.h"
#include <cmath>

#ifndef DTYPE_X
#define DTYPE_X float
#endif
#ifndef DTYPE_W
#define DTYPE_W float
#endif
#ifndef DTYPE_B
#define DTYPE_B float
#endif
#ifndef DTYPE_Z
#define DTYPE_Z float
#endif

extern "C" __global__ __aicore__ void conv2d_tanh_scaling_bias_add_max_custom(
    GM_ADDR x,           // [N, C_in, H, W]
    GM_ADDR weight,      // [C_out, C_in, kH, kW]
    GM_ADDR conv_bias,   // [C_out]
    GM_ADDR add_bias,    // [C_out, 1, 1] -> [C_out]
    GM_ADDR scale,       // [1]
    GM_ADDR z,           // [N, C_out, P_H, P_W]
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N    = tiling_data.n;
    const uint32_t Ci   = tiling_data.cIn;
    const uint32_t H    = tiling_data.h;
    const uint32_t W    = tiling_data.w;
    const uint32_t Co   = tiling_data.cOut;
    const uint32_t kH   = tiling_data.kH;
    const uint32_t kW   = tiling_data.kW;
    const uint32_t outH = tiling_data.outH; // (H - kH + 1) for stride=1,pad=0
    const uint32_t outW = tiling_data.outW;
    const uint32_t pKH  = tiling_data.poolKH;
    const uint32_t pKW  = tiling_data.poolKW;

    const uint32_t pH = (pKH == 0) ? 0 : (outH / pKH);
    const uint32_t pW = (pKW == 0) ? 0 : (outW / pKW);

    const DTYPE_X* x_ptr    = reinterpret_cast<const DTYPE_X*>(x);
    const DTYPE_W* w_ptr    = reinterpret_cast<const DTYPE_W*>(weight);
    const DTYPE_B* cb_ptr   = reinterpret_cast<const DTYPE_B*>(conv_bias);
    const DTYPE_B* ab_ptr   = reinterpret_cast<const DTYPE_B*>(add_bias);
    const DTYPE_B* sc_ptr   = reinterpret_cast<const DTYPE_B*>(scale);
    DTYPE_Z*       z_ptr    = reinterpret_cast<DTYPE_Z*>(z);

    const DTYPE_B sVal = sc_ptr[0];

    const uint64_t in_hw   = static_cast<uint64_t>(H) * W;
    const uint64_t in_chw  = static_cast<uint64_t>(Ci) * in_hw;
    const uint64_t w_khw   = static_cast<uint64_t>(kH) * kW;
    const uint64_t w_ckhw  = static_cast<uint64_t>(Ci) * w_khw;
    const uint64_t out_hw  = static_cast<uint64_t>(outH) * outW;
    const uint64_t pool_hw = static_cast<uint64_t>(pH) * pW;

    for (uint32_t n = 0; n < N; ++n) {
        uint64_t x_n_off = static_cast<uint64_t>(n) * in_chw;
        uint64_t z_n_off = static_cast<uint64_t>(n) * static_cast<uint64_t>(Co) * pool_hw;

        for (uint32_t co = 0; co < Co; ++co) {
            uint64_t w_co_off = static_cast<uint64_t>(co) * w_ckhw;
            DTYPE_B conv_b = cb_ptr[co];
            DTYPE_B add_b  = ab_ptr[co]; // add_bias broadcast

            for (uint32_t ph = 0; ph < pH; ++ph) {
                for (uint32_t pw_ = 0; pw_ < pW; ++pw_) {
                    // pool window origin in conv output
                    uint32_t oh0 = ph * pKH;
                    uint32_t ow0 = pw_ * pKW;

                    DTYPE_Z max_val = -INFINITY;

                    for (uint32_t kh_pool = 0; kh_pool < pKH; ++kh_pool) {
                        for (uint32_t kw_pool = 0; kw_pool < pKW; ++kw_pool) {
                            uint32_t oh = oh0 + kh_pool;
                            uint32_t ow = ow0 + kw_pool;
                            if (oh >= outH || ow >= outW) continue;

                            // compute conv at (oh, ow, co)
                            DTYPE_Z acc = static_cast<DTYPE_Z>(0);
                            for (uint32_t ci = 0; ci < Ci; ++ci) {
                                uint64_t x_ci_off = x_n_off + static_cast<uint64_t>(ci) * in_hw;
                                uint64_t w_ci_off = w_co_off + static_cast<uint64_t>(ci) * w_khw;

                                for (uint32_t kh = 0; kh < kH; ++kh) {
                                    uint32_t ih = oh + kh; // stride=1, pad=0
                                    const uint64_t x_row_off = x_ci_off + static_cast<uint64_t>(ih) * W;

                                    for (uint32_t kw = 0; kw < kW; ++kw) {
                                        uint32_t iw = ow + kw;
                                        DTYPE_X xv = x_ptr[x_row_off + iw];
                                        DTYPE_W wv = w_ptr[w_ci_off + static_cast<uint64_t>(kh) * kW + kw];
                                        acc += static_cast<DTYPE_Z>(xv) * static_cast<DTYPE_Z>(wv);
                                    }
                                }
                            }

                            // post ops: +conv_bias -> tanh -> *scale -> +add_bias
                            acc += static_cast<DTYPE_Z>(conv_b);
                            acc = static_cast<DTYPE_Z>(tanhf(static_cast<float>(acc)));
                            acc *= static_cast<DTYPE_Z>(sVal);
                            acc += static_cast<DTYPE_Z>(add_b);

                            if (acc > max_val) { max_val = acc; }
                        }
                    }

                    uint64_t out_index = z_n_off + static_cast<uint64_t>(co) * pool_hw
                                       + static_cast<uint64_t>(ph) * pW + pw_;
                    z_ptr[out_index] = max_val;
                }
            }
        }
    }
}

    int64_t pool_kw)

    // x: [N, C_in, H, W], weight: [C_out, C_in, kH, kW], stride=1, padding=0

    int64_t outW = (W >= kW) ? (W - kW + 1) : 0;

    int64_t pW = pool_kw > 0 ? outW / pool_kw : 0;

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused: Conv2d -> Tanh -> Scale -> BiasAdd -> MaxPool2d
    Backed by custom AscendC operator conv2d_tanh_scaling_bias_add_max_custom.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super().__init__()
        # Keep Conv2d module to own weight/bias tensors; stride=1, padding=0 as in original.
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size, stride=1, padding=0, bias=True)
        self.scaling_factor = scaling_factor
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self._pool_kernel_size = pool_kernel_size if isinstance(pool_kernel_size, tuple) else (pool_kernel_size, pool_kernel_size)

    def forward(self, x):
        scale_tensor = torch.tensor([self.scaling_factor], dtype=x.dtype, device=x.device)
        kh, kw = self._pool_kernel_size
        return custom_ops_lib.conv2d_tanh_scaling_bias_add_max_custom(
            x,
            self.conv.weight,
            self.conv.bias if self.conv.bias is not None else torch.zeros(self.conv.out_channels, dtype=x.dtype, device=x.device),
            self.bias,
            scale_tensor,
            int(kh),
            int(kw)
        )
