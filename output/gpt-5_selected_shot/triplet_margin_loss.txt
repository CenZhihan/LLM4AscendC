kernel_src="""
#include "kernel_operator.h"
constexpr int32_t BUFFER_NUM = 1;

class KernelTripletMarginLoss {
public:
    __aicore__ inline KernelTripletMarginLoss() {}
    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR margin, GM_ADDR y,
                                uint32_t totalLength, uint32_t blockLength,
                                uint32_t tileNum, uint32_t tileLength,
                                uint32_t lasttileLength) {
        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");

        this->totalLength = static_cast<int32_t>(totalLength);
        this->blockLength = blockLength;
        this->tileNum = tileNum;
        this->tileLength = tileLength;
        this->lasttileLength = lasttileLength;

        aGm.SetGlobalBuffer((__gm__ float*)anchor + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pGm.SetGlobalBuffer((__gm__ float*)positive + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        nGm.SetGlobalBuffer((__gm__ float*)negative + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        marginGm.SetGlobalBuffer((__gm__ float*)margin, 1);
        outGm.SetGlobalBuffer((__gm__ float*)y + this->blockLength * AscendC::GetBlockIdx(), 32);

        this->reduce_num = this->tileNum;

        // Buffers: one IN queue for three vectors (anchor, positive, negative) per tile.
        pipe.InitBuffer(inQueueIN, BUFFER_NUM, this->tileLength * 3 * sizeof(float));
        // Buffer to store per-sample loss values for final reduction
        uint32_t reduce_align = (this->reduce_num + 31) / 32 * 32;
        pipe.InitBuffer(partialBuf, reduce_align * sizeof(float));
        // Buffer to hold scalar margin on device
        pipe.InitBuffer(marginBuf, sizeof(float));

        // Load margin scalar from GM to local buffer
        AscendC::LocalTensor<float> mLocal = marginBuf.Get<float>();
        AscendC::DataCopy(mLocal[0], marginGm[0], 1);
    }

    __aicore__ inline void Process() {
        int32_t loopCount = this->tileNum;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
        }

        AscendC::LocalTensor<float> partials = partialBuf.Get<float>();
        AscendC::LocalTensor<float> tmpWorkspace = inQueueIN.AllocTensor<float>();

        AscendC::ReduceSum<float>(partials, partials, tmpWorkspace, this->reduce_num);

        // Divide by batch (tileNum) to get mean
        tmpWorkspace.SetValue(0, static_cast<float>(this->tileNum));
        AscendC::Div(partials, partials, tmpWorkspace, 1);

        outGm.SetValue(0, partials.GetValue(0));
        inQueueIN.FreeTensor(tmpWorkspace);
    }

private:
    __aicore__ inline void CopyIn(int32_t progress) {
        AscendC::LocalTensor<float> inLocal = inQueueIN.AllocTensor<float>();

        uint32_t offset = static_cast<uint32_t>(progress) * this->tileLength;
        AscendC::DataCopy(inLocal[0], aGm[offset], this->tileLength);
        AscendC::DataCopy(inLocal[this->tileLength], pGm[offset], this->tileLength);
        AscendC::DataCopy(inLocal[this->tileLength * 2], nGm[offset], this->tileLength);

        inQueueIN.EnQue(inLocal);
    }

    __aicore__ inline void Compute(int32_t progress) {
        AscendC::LocalTensor<float> inLocal = inQueueIN.DeQue<float>();
        AscendC::LocalTensor<float> aLocal = inLocal;
        AscendC::LocalTensor<float> pLocal = inLocal[this->tileLength];
        AscendC::LocalTensor<float> nLocal = inLocal[this->tileLength * 2];
        AscendC::LocalTensor<float> partials = partialBuf.Get<float>();
        AscendC::LocalTensor<float> mLocal = marginBuf.Get<float>();

        // d_ap = sqrt(sum((a - p)^2))
        AscendC::Sub(pLocal, aLocal, pLocal, this->tileLength);
        AscendC::Mul(pLocal, pLocal, pLocal, this->tileLength);
        AscendC::ReduceSum<float>(pLocal, pLocal, aLocal, this->tileLength);
        AscendC::Sqrt(pLocal, pLocal, 1);

        // d_an = sqrt(sum((a - n)^2))
        AscendC::Sub(nLocal, aLocal, nLocal, this->tileLength);
        AscendC::Mul(nLocal, nLocal, nLocal, this->tileLength);
        AscendC::ReduceSum<float>(nLocal, nLocal, aLocal, this->tileLength);
        AscendC::Sqrt(nLocal, nLocal, 1);

        // l_i = max(d_ap - d_an + margin, 0)
        AscendC::Sub(pLocal, pLocal, nLocal, 1);
        AscendC::Add(pLocal, pLocal, mLocal, 1);
        AscendC::Duplicate(aLocal, (float)0.0f, 1);
        AscendC::Max(pLocal, pLocal, aLocal, 1);

        partials.SetValue(progress, pLocal.GetValue(0));
        inQueueIN.FreeTensor(inLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueIN;
    AscendC::TBuf<> partialBuf;
    AscendC::TBuf<> marginBuf;

    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> pGm;
    AscendC::GlobalTensor<float> nGm;
    AscendC::GlobalTensor<float> marginGm;
    AscendC::GlobalTensor<float> outGm;

    int32_t totalLength;
    uint32_t reduce_num;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t lasttileLength;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR margin, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLoss op;
    op.Init(anchor, positive, negative, margin, y,
            tiling_data.totalLength, tiling_data.blockLength,
            tiling_data.tileNum, tiling_data.tileLength, tiling_data.lasttileLength);
    op.Process();
}
