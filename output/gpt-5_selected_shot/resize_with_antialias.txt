inline uint32_t safe_cast_i64_to_u32(int64_t v) {
    return static_cast<uint32_t>(v);
}

namespace optiling {
const uint32_t MAX_BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ResizeWithAntialiasCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    const gert::Shape* out_shape = context->GetOutputShape(0);

    // Assume NCHW
    uint32_t n = safe_cast_i64_to_u32(in_shape->GetOriginShape().GetDim(0));
    uint32_t c = safe_cast_i64_to_u32(in_shape->GetOriginShape().GetDim(1));
    uint32_t h = safe_cast_i64_to_u32(in_shape->GetOriginShape().GetDim(2));
    uint32_t w = safe_cast_i64_to_u32(in_shape->GetOriginShape().GetDim(3));

    uint32_t oh = safe_cast_i64_to_u32(out_shape->GetOriginShape().GetDim(2));
    uint32_t ow = safe_cast_i64_to_u32(out_shape->GetOriginShape().GetDim(3));

    uint32_t totalPlanes = n * c;
    uint32_t blockDim = totalPlanes < MAX_BLOCK_DIM ? (totalPlanes == 0 ? 1 : totalPlanes) : MAX_BLOCK_DIM;
    context->SetBlockDim(blockDim);
    uint32_t planesPerBlock = (totalPlanes + blockDim - 1) / blockDim;

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_oh(oh);
    tiling.set_ow(ow);
    tiling.set_totalPlanes(totalPlanes);
    tiling.set_planesPerBlock(planesPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

    // If possible, overwrite spatial dims to 128x128 (Model uses 128x128)

    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;  // row buffers
constexpr int32_t OUT_BUF_NUM = 1; // one output row buffer

class KernelResizeAA {
public:
    __aicore__ inline KernelResizeAA() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z,
                                uint32_t n, uint32_t c,
                                uint32_t h, uint32_t w,
                                uint32_t oh, uint32_t ow,
                                uint32_t totalPlanes, uint32_t planesPerBlock)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->oh = oh;
        this->ow = ow;
        this->totalPlanes = totalPlanes;
        this->planesPerBlock = planesPerBlock;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startPlane = blockIdx * planesPerBlock;
        this->startPlane = startPlane;
        uint32_t endPlane = startPlane + planesPerBlock;
        if (endPlane > totalPlanes) {
            endPlane = totalPlanes;
        }
        this->endPlane = endPlane;

        uint64_t inPlaneSize = (uint64_t)h * (uint64_t)w;
        uint64_t outPlaneSize = (uint64_t)oh * (uint64_t)ow;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + startPlane * inPlaneSize, (endPlane - startPlane) * inPlaneSize);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + startPlane * outPlaneSize, (endPlane - startPlane) * outPlaneSize);

        // One row = w elements for input; output row = ow elements
        pipe.InitBuffer(inRowQueue0, IN_BUF_NUM, w * sizeof(DTYPE_X));
        pipe.InitBuffer(inRowQueue1, IN_BUF_NUM, w * sizeof(DTYPE_X));
        pipe.InitBuffer(outRowQueue, OUT_BUF_NUM, ow * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t p = startPlane; p < endPlane; ++p) {
            uint32_t localPlaneIdx = p - startPlane;
            // base offsets for this (n,c) plane
            uint64_t inBase = (uint64_t)localPlaneIdx * (uint64_t)h * (uint64_t)w;
            uint64_t outBase = (uint64_t)localPlaneIdx * (uint64_t)oh * (uint64_t)ow;

            for (uint32_t r = 0; r < oh; ++r) {
                CopyInRows(inBase, r);
                ComputeRow();
                CopyOutRow(outBase, r);
            }
        }
    }

private:
    __aicore__ inline void CopyInRows(uint64_t inBase, uint32_t outRowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> row0 = inRowQueue0.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> row1 = inRowQueue1.AllocTensor<DTYPE_X>();

        uint32_t inRow0 = outRowIdx << 1;                // 2 * r
        uint32_t inRow1 = (inRow0 + 1) < h ? (inRow0 + 1) : (h - 1);

        AscendC::DataCopy(row0, xGm[inBase + (uint64_t)inRow0 * (uint64_t)w], w);
        AscendC::DataCopy(row1, xGm[inBase + (uint64_t)inRow1 * (uint64_t)w], w);

        inRowQueue0.EnQue(row0);
        inRowQueue1.EnQue(row1);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<DTYPE_X> row0 = inRowQueue0.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> row1 = inRowQueue1.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> outRow = outRowQueue.AllocTensor<DTYPE_Z>();

        // Simple 2x2 box filter downsampling (anti-aliasing) for 2x scale:
        // out[r, c] = 0.25 * (row0[2c] + row0[2c+1] + row1[2c] + row1[2c+1])
        // Assumes even input size; clamps on boundaries.
        for (uint32_t col = 0; col < ow; ++col) {
            uint32_t c0 = col << 1;
            uint32_t c1 = (c0 + 1) < w ? (c0 + 1) : (w - 1);
            // load
            DTYPE_X a = row0.GetValue(c0);
            DTYPE_X b = row0.GetValue(c1);
            DTYPE_X d = row1.GetValue(c0);
            DTYPE_X e = row1.GetValue(c1);
            DTYPE_Z v = static_cast<DTYPE_Z>((a + b + d + e) * (DTYPE_X)0.25f);
            outRow.SetValue(col, v);
        }

        outRowQueue.EnQue(outRow);
        inRowQueue0.FreeTensor(row0);
        inRowQueue1.FreeTensor(row1);
    }

    __aicore__ inline void CopyOutRow(uint64_t outBase, uint32_t outRowIdx)
    {
        AscendC::LocalTensor<DTYPE_Z> outRow = outRowQueue.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[outBase + (uint64_t)outRowIdx * (uint64_t)ow], outRow, ow);
        outRowQueue.FreeTensor(outRow);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inRowQueue0;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inRowQueue1;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outRowQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t n, c, h, w, oh, ow;
    uint32_t totalPlanes;
    uint32_t planesPerBlock;
    uint32_t startPlane;
    uint32_t endPlane;
};

// Kernel entry
extern "C" __global__ __aicore__ void resize_with_antialias_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelResizeAA op;
    op.Init(x, z,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.oh, tiling_data.ow,
            tiling_data.totalPlanes, tiling_data.planesPerBlock);
    op.Process();
}

at::Tensor resize_with_antialias_custom_impl_npu(const at::Tensor& self) {
    TORCH_CHECK(self.dim() == 4, "Input must be NCHW 4D tensor");
    auto sizes = self.sizes();
    const int64_t N = sizes[0];
    const int64_t C = sizes[1];
    const int64_t OH = 128;
    const int64_t OW = 128;
    at::Tensor result = at::empty({N, C, OH, OW}, self.options());
    EXEC_NPU_CMD(aclnnResizeWithAntialiasCustom, self, result);
    return result;
}

TORCH_LIBRARY(myops, m) {
