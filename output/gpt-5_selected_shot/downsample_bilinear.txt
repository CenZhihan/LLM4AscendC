kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUFFER_NUM = 2;  // buffers for two input rows
constexpr int32_t OUT_BUFFER_NUM = 1; // buffer for one output row

class KernelDownsampleBilinearCustom {
public:
    __aicore__ inline KernelDownsampleBilinearCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t outH, uint32_t outW)
    {
        this->N = n;
        this->C = c;
        this->H = h;
        this->W = w;
        this->OH = outH;
        this->OW = outW;

        uint32_t totalNC = N * C;
        uint32_t coreNum = AscendC::GetBlockNum();
        uint32_t coreId = AscendC::GetBlockIdx();
        uint32_t ncPerCore = (totalNC + coreNum - 1) / coreNum;
        this->ncStart = coreId * ncPerCore;
        this->ncEnd = this->ncStart + ncPerCore;
        if (this->ncEnd > totalNC) this->ncEnd = totalNC;
        if (this->ncStart > this->ncEnd) this->ncStart = this->ncEnd;

        uint64_t inOffset = static_cast<uint64_t>(this->ncStart) * H * W;
        uint64_t outOffset = static_cast<uint64_t>(this->ncStart) * OH * OW;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + inOffset, (this->ncEnd - this->ncStart) * H * W);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + outOffset, (this->ncEnd - this->ncStart) * OH * OW);

        pipe.InitBuffer(inRow0Q, IN_BUFFER_NUM, this->W * sizeof(DTYPE_X));
        pipe.InitBuffer(inRow1Q, IN_BUFFER_NUM, this->W * sizeof(DTYPE_X));
        pipe.InitBuffer(outRowQ, OUT_BUFFER_NUM, this->OW * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (ncStart >= ncEnd) {
            return;
        }

        float scaleH = static_cast<float>(H) / static_cast<float>(OH);
        float scaleW = static_cast<float>(W) / static_cast<float>(OW);

        for (uint32_t ncRel = 0; ncRel < (ncEnd - ncStart); ++ncRel) {
            for (uint32_t yo = 0; yo < OH; ++yo) {
                // Compute source y indices per align_corners=False
                float inY = (static_cast<float>(yo) + 0.5f) * scaleH - 0.5f;
                int32_t y0 = static_cast<int32_t>(inY);
                if (y0 < 0) y0 = 0;
                if (y0 > static_cast<int32_t>(H) - 2) y0 = static_cast<int32_t>(H) - 2;
                int32_t y1 = y0 + 1;

                CopyInRows(ncRel, static_cast<uint32_t>(y0), static_cast<uint32_t>(y1));
                ComputeRow(scaleW, yo);
                CopyOutRow(ncRel, yo);
            }
        }
    }

private:
    __aicore__ inline void CopyInRows(uint32_t ncRel, uint32_t y0, uint32_t y1)
    {
        AscendC::LocalTensor<DTYPE_X> row0Local = inRow0Q.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> row1Local = inRow1Q.AllocTensor<DTYPE_X>();
        uint64_t baseIn = static_cast<uint64_t>(ncRel) * H * W;
        AscendC::DataCopy(row0Local, xGm[baseIn + static_cast<uint64_t>(y0) * W], W);
        AscendC::DataCopy(row1Local, xGm[baseIn + static_cast<uint64_t>(y1) * W], W);
        inRow0Q.EnQue(row0Local);
        inRow1Q.EnQue(row1Local);
    }

    __aicore__ inline void ComputeRow(float scaleW, uint32_t yo)
    {
        AscendC::LocalTensor<DTYPE_X> row0Local = inRow0Q.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> row1Local = inRow1Q.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> outLocal = outRowQ.AllocTensor<DTYPE_Z>();

        for (uint32_t xo = 0; xo < OW; ++xo) {
            float inX = (static_cast<float>(xo) + 0.5f) * scaleW - 0.5f;
            int32_t x0 = static_cast<int32_t>(inX);
            if (x0 < 0) x0 = 0;
            if (x0 > static_cast<int32_t>(W) - 2) x0 = static_cast<int32_t>(W) - 2;
            int32_t x1 = x0 + 1;

            float lx = inX - static_cast<float>(x0);
            float ly = (static_cast<float>(yo) + 0.5f) * (static_cast<float>(H) / static_cast<float>(OH)) - 0.5f;
            // recompute ly consistently with y0 used above
            int32_t yy0 = static_cast<int32_t>(ly);
            if (yy0 < 0) yy0 = 0;
            if (yy0 > static_cast<int32_t>(H) - 2) yy0 = static_cast<int32_t>(H) - 2;
            float wy = ly - static_cast<float>(yy0);

            float wx0 = 1.0f - lx;
            float wx1 = lx;
            float wy0 = 1.0f - wy;
            float wy1 = wy;

            float v00 = static_cast<float>(row0Local[x0]);
            float v01 = static_cast<float>(row0Local[x1]);
            float v10 = static_cast<float>(row1Local[x0]);
            float v11 = static_cast<float>(row1Local[x1]);

            float top = wx0 * v00 + wx1 * v01;
            float bottom = wx0 * v10 + wx1 * v11;
            float val = wy0 * top + wy1 * bottom;

            outLocal[xo] = static_cast<DTYPE_Z>(val);
        }

        outRowQ.EnQue(outLocal);
        inRow0Q.FreeTensor(row0Local);
        inRow1Q.FreeTensor(row1Local);
    }

    __aicore__ inline void CopyOutRow(uint32_t ncRel, uint32_t yo)
    {
        AscendC::LocalTensor<DTYPE_Z> outLocal = outRowQ.DeQue<DTYPE_Z>();
        uint64_t baseOut = static_cast<uint64_t>(ncRel) * OH * OW;
        AscendC::DataCopy(yGm[baseOut + static_cast<uint64_t>(yo) * OW], outLocal, OW);
        outRowQ.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inRow0Q;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inRow1Q;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outRowQ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N;
    uint32_t C;
    uint32_t H;
    uint32_t W;
    uint32_t OH;
    uint32_t OW;

    uint32_t ncStart;
    uint32_t ncEnd;
};

extern "C" __global__ __aicore__ void downsample_bilinear_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    KernelDownsampleBilinearCustom op;
    op.Init(x, y,
            tiling_data.n,
            tiling_data.c,
            tiling_data.h,
            tiling_data.w,
            tiling_data.outH,
            tiling_data.outW);
    op.Process();
}
"""
