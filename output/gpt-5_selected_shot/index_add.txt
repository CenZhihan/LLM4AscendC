    IndexAddCustomTilingData tiling;

    uint32_t updates = static_cast<uint32_t>(idx_shape->GetOriginShape().GetShapeSize());

    uint32_t tileCols = cols / (TILE_NUM * BUFFER_NUM);

    uint32_t copyTileElems = static_cast<uint32_t>(totalElems / (TILE_NUM * BUFFER_NUM));
    if (copyTileElems == 0) copyTileElems = 1;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_updates(updates);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_tileCols(tileCols);
    tiling.set_copyTileElems(copyTileElems);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());

kernel_src="""
#include "kernel_operator.h"

// We implement: z = x; for i in [0, updates): z[index[i], :] += source[i, :]
constexpr int32_t BUFFER_NUM = 2;

class KernelIndexAdd {
public:
    __aicore__ inline KernelIndexAdd() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR indices,
                                GM_ADDR source,
                                GM_ADDR z,
                                uint32_t rows,
                                uint32_t cols,
                                uint32_t updates,
                                uint32_t tileNum,
                                uint32_t tileCols,
                                uint32_t copyTileElems)
    {
        this->rows = rows;
        this->cols = cols;
        this->updates = updates;
        this->tileNum = tileNum;
        this->tileCols = tileCols;
        this->copyTileElems = copyTileElems;

        blockNum = AscendC::GetBlockNum();
        blockIdx = AscendC::GetBlockIdx();

        // Partition copy of x->z by contiguous elements
        uint64_t totalElems = static_cast<uint64_t>(rows) * static_cast<uint64_t>(cols);
        uint64_t basePerBlock = totalElems / blockNum;
        uint64_t remain = totalElems % blockNum;
        uint64_t startElem = basePerBlock * blockIdx + (blockIdx < remain ? blockIdx : remain);
        uint64_t lenElems = basePerBlock + (blockIdx < remain ? 1 : 0);
        this->copyStart = static_cast<uint32_t>(startElem);
        this->copyLen = static_cast<uint32_t>(lenElems);

        xGm.SetGlobalBuffer((__gm__ float *)x + copyStart, copyLen);
        zGm.SetGlobalBuffer((__gm__ float *)z + copyStart, copyLen);

        idxGm.SetGlobalBuffer((__gm__ int32_t *)indices, updates);
        srcGm.SetGlobalBuffer((__gm__ float *)source, static_cast<uint32_t>(updates * cols));

        pipe.InitBuffer(copyInQ, BUFFER_NUM, copyTileElems * sizeof(float));
        pipe.InitBuffer(copyOutQ, BUFFER_NUM, copyTileElems * sizeof(float));
        pipe.InitBuffer(valQ, BUFFER_NUM, tileCols * sizeof(float));
    }

    __aicore__ inline void CopyXtoZ()
    {
        uint32_t processed = 0;
        while (processed < copyLen) {
            uint32_t cur = AscendC::min(copyTileElems, copyLen - processed);

            AscendC::LocalTensor<float> xLocal = copyInQ.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm[processed], cur);
            copyInQ.EnQue(xLocal);

            AscendC::LocalTensor<float> xLocal2 = copyInQ.DeQue<float>();
            AscendC::LocalTensor<float> zLocal = copyOutQ.AllocTensor<float>();
            // Simple pass-through
            AscendC::DataCopy(zLocal, xLocal2, cur);
            copyOutQ.EnQue(zLocal);
            copyInQ.FreeTensor(xLocal2);

            AscendC::LocalTensor<float> zOut = copyOutQ.DeQue<float>();
            AscendC::DataCopy(zGm[processed], zOut, cur);
            copyOutQ.FreeTensor(zOut);

            processed += cur;
        }
    }

    __aicore__ inline void ProcessUpdates()
    {
        // Partition updates among blocks
        uint32_t basePerBlock = updates / blockNum;
        uint32_t remain = updates % blockNum;
        uint32_t start = basePerBlock * blockIdx + (blockIdx < remain ? blockIdx : remain);
        uint32_t count = basePerBlock + (blockIdx < remain ? 1 : 0);
        uint32_t end = start + count;

        for (uint32_t i = start; i < end; ++i) {
            // load index i
            int32_t rowIdx = 0;
            {
                // Read one index element
                AscendC::LocalTensor<int32_t> idxLocal;
                // allocate a tiny temp buffer via valQ for reuse (avoids dedicating another queue)
                // Note: allocate at least 1 float sized buffer and reinterpret for index to avoid extra queue.
                // Safer approach: use a single-element local tensor created from pipe's temp buffer.
                // Here we emulate by allocating from valQ if space allows. Alternatively rely on compiler provided local variable.
                // Instead, perform direct GM scalar read by staging through a 1-length LocalTensor.
                AscendC::LocalTensor<int32_t> tmp = pipe.AllocLocal<int32_t>(1);
                AscendC::DataCopy(tmp, idxGm[i], 1);
                rowIdx = tmp.GetValue(0);
                pipe.FreeLocal(tmp);
            }
            if (rowIdx < 0) continue; // guard

            uint32_t base = static_cast<uint32_t>(rowIdx) * cols;

            // iterate columns in tiles
            uint32_t col = 0;
            while (col < cols) {
                uint32_t curCols = AscendC::min(tileCols, cols - col);

                AscendC::LocalTensor<float> vLocal = valQ.AllocTensor<float>();
                AscendC::DataCopy(vLocal, srcGm[i * cols + col], curCols);
                valQ.EnQue(vLocal);

                AscendC::LocalTensor<float> vIn = valQ.DeQue<float>();
                // atomic add each element into z at (rowIdx, col:col+curCols)
                for (uint32_t k = 0; k < curCols; ++k) {
                    float val = vIn.GetValue(k);
                    // Atomic add into global memory
                    AscendC::AtomicAdd(zAllGm[base + col + k], val);
                }
                valQ.FreeTensor(vIn);

                col += curCols;
            }
        }
    }

    __aicore__ inline void Process()
    {
        CopyXtoZ();
        // Prepare a global tensor handle that spans the whole z (for atomic addressing)
        zAllGm.SetGlobalBuffer((__gm__ float *)zGm.GetGMAddr(0) - copyStart, rows * cols);
        ProcessUpdates();
    }

private:
    AscendC::TPipe pipe;

    // Queues for copy x->z
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> copyInQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> copyOutQ;

    // Queue for values tiles
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> valQ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;     // block slice
    AscendC::GlobalTensor<float> zAllGm;  // full z for atomic addressing
    AscendC::GlobalTensor<int32_t> idxGm;
    AscendC::GlobalTensor<float> srcGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t updates;
    uint32_t tileNum;
    uint32_t tileCols;
    uint32_t copyTileElems;

    uint32_t blockNum;
    uint32_t blockIdx;

    uint32_t copyStart;
    uint32_t copyLen;
};

extern "C" __global__ __aicore__ void index_add_custom(GM_ADDR x, GM_ADDR index, GM_ADDR source, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelIndexAdd op;
    op.Init(x,
            index,
            source,
            z,
            tiling_data.rows,
            tiling_data.cols,
            tiling_data.updates,
            tiling_data.tileNum,
            tiling_data.tileCols,
            tiling_data.copyTileElems);
    op.Process();
}
