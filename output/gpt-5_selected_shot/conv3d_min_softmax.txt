kernel_src="""
#include "kernel_operator.h"
#include <cmath>

typedef float DTYPE_X;
typedef float DTYPE_Y;

class KernelConv3dMinSoftmax {
public:
    __aicore__ inline KernelConv3dMinSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                int32_t reduceDim)
    {
        xGm = (__gm__ DTYPE_X*)x;
        yGm = (__gm__ DTYPE_Y*)y;

        this->N = n; this->C = c; this->D = d; this->H = h; this->W = w;
        this->reduceDim = reduceDim;

        // Remaining dims after reduction
        if (reduceDim == 2) { // reduce D -> remain H,W
            rem1 = H; rem2 = W; R = D;
            strideD = H * W;
            strideH = W;
            strideW = 1;
        } else if (reduceDim == 3) { // reduce H -> remain D,W
            rem1 = D; rem2 = W; R = H;
            strideD = H * W;
            strideH = W;
            strideW = 1;
        } else { // reduceDim == 4, reduce W -> remain D,H
            rem1 = D; rem2 = H; R = W;
            strideD = H * W;
            strideH = W;
            strideW = 1;
        }

        groupCount = N * rem1 * rem2;                 // groups for softmax (per (n, rem1, rem2))
        outCount   = N * C * rem1 * rem2;             // number of elements in reduced tensor
    }

    __aicore__ inline void ReduceMinPass()
    {
        // Distribute work across blocks by rows of output (N*C*rem1*rem2)
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint64_t itemsPerBlock = outCount / blockNum;
        uint64_t residue = outCount % blockNum;
        uint64_t start = blockIdx * itemsPerBlock + AscendC::Min(blockIdx, residue);
        uint64_t len   = itemsPerBlock + (blockIdx < residue ? 1 : 0);
        uint64_t end   = start + len;

        for (uint64_t t = start; t < end; ++t) {
            // Decode t -> (n, c, r1, r2)
            uint64_t tmp = t;
            uint32_t r2 = tmp % rem2; tmp /= rem2;
            uint32_t r1 = tmp % rem1; tmp /= rem1;
            uint32_t c  = tmp % C;    tmp /= C;
            uint32_t n  = tmp % N;

            // Map (n,c,r1,r2) to base indices in (D,H,W)
            // Determine fixed coordinates for reduced dimension
            uint32_t d0=0, h0=0, w0=0;
            if (reduceDim == 2) { // reduce over D, keep H=r1, W=r2
                h0 = r1; w0 = r2;
            } else if (reduceDim == 3) { // reduce over H, keep D=r1, W=r2
                d0 = r1; w0 = r2;
            } else { // reduce over W, keep D=r1, H=r2
                d0 = r1; h0 = r2;
            }

            // Compute base offset to first element along reduction axis
            uint64_t base = ((uint64_t)n * C + c) * (uint64_t)(D * H * W);
            uint64_t offsetFixed = 0;
            if (reduceDim == 2) {
                offsetFixed = (uint64_t)h0 * W + w0;
            } else if (reduceDim == 3) {
                offsetFixed = (uint64_t)d0 * (H * W) + w0;
            } else { // reduce W
                offsetFixed = (uint64_t)d0 * (H * W) + (uint64_t)h0 * W;
            }

            DTYPE_Y minVal = (DTYPE_Y)INFINITY;
            for (uint32_t r = 0; r < R; ++r) {
                uint64_t off = 0;
                if (reduceDim == 2) { // varying d
                    off = (uint64_t)r * (H * W) + offsetFixed;
                } else if (reduceDim == 3) { // varying h
                    off = (uint64_t)d0 * (H * W) + (uint64_t)r * W + w0;
                } else { // varying w
                    off = (uint64_t)d0 * (H * W) + (uint64_t)h0 * W + r;
                }
                DTYPE_X v = xGm[base + off];
                minVal = v < minVal ? v : minVal;
            }

            // Store to y at linearized (n,c,r1,r2)
            uint64_t yOff = (((uint64_t)n * C + c) * rem1 + r1) * rem2 + r2;
            yGm[yOff] = minVal;
        }
    }

    __aicore__ inline void SoftmaxPass()
    {
        // Distribute work across blocks by groups (n, rem1, rem2), compute softmax across C
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint64_t groupsPerBlock = groupCount / blockNum;
        uint64_t residue = groupCount % blockNum;
        uint64_t start = blockIdx * groupsPerBlock + AscendC::Min(blockIdx, residue);
        uint64_t len   = groupsPerBlock + (blockIdx < residue ? 1 : 0);
        uint64_t end   = start + len;

        for (uint64_t g = start; g < end; ++g) {
            uint64_t tmp = g;
            uint32_t r2 = tmp % rem2; tmp /= rem2;
            uint32_t r1 = tmp % rem1; tmp /= rem1;
            uint32_t n  = tmp % N;

            // Compute max for numerical stability
            DTYPE_Y maxVal = -((DTYPE_Y)INFINITY);
            for (uint32_t c = 0; c < C; ++c) {
                uint64_t yOff = (((uint64_t)n * C + c) * rem1 + r1) * rem2 + r2;
                DTYPE_Y v = yGm[yOff];
                maxVal = v > maxVal ? v : maxVal;
            }
            // Compute exp and sum
            DTYPE_Y sumVal = (DTYPE_Y)0;
            for (uint32_t c = 0; c < C; ++c) {
                uint64_t yOff = (((uint64_t)n * C + c) * rem1 + r1) * rem2 + r2;
                DTYPE_Y e = (DTYPE_Y)expf((float)(yGm[yOff] - maxVal));
                yGm[yOff] = e; // store temporary exp
                sumVal += e;
            }
            // Normalize
            DTYPE_Y invSum = sumVal > 0 ? ((DTYPE_Y)1 / sumVal) : (DTYPE_Y)0;
            for (uint32_t c = 0; c < C; ++c) {
                uint64_t yOff = (((uint64_t)n * C + c) * rem1 + r1) * rem2 + r2;
                yGm[yOff] = yGm[yOff] * invSum;
            }
        }
    }

    __aicore__ inline void Process()
    {
        ReduceMinPass();
        AscendC::EventRecord(); // barrier hint between passes
        SoftmaxPass();
    }

private:
    __gm__ DTYPE_X* xGm;
    __gm__ DTYPE_Y* yGm;
    uint32_t N, C, D, H, W;
    uint32_t rem1, rem2, R;
    int32_t reduceDim;
    uint64_t groupCount;
    uint64_t outCount;
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMinSoftmax op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.reduceDim);
    op.Process();
}

at::Tensor conv3d_min_softmax_custom_impl_npu(const at::Tensor& x, int64_t reduce_dim) {

    at::Tensor result;

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model:
    - Keeps the Conv3d layer on PyTorch/NPU
    - Replaces min along dim and softmax over channels with a fused custom AscendC kernel
    """
    def __init__(self, in_channels, out_channels, kernel_size, dim):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.dim = dim  # expected to be 2, 3, or 4

    def forward(self, x):
        x = self.conv(x)  # (N, C, D, H, W)
        # Fused op: min along self.dim, then softmax along channel dim=1
        x = custom_ops_lib.conv3d_min_softmax_custom(x, int(self.dim))
        return x
