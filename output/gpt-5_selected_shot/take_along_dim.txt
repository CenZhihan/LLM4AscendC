#include "take_along_dim_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    TakeAlongDimCustomTilingData tiling;

    // x shape: [N, M], idx shape: [N, K], output: [N, K]
    const auto* x_shape = context->GetInputShape(0);
    const auto* idx_shape = context->GetInputShape(1);

    uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t K = static_cast<uint32_t>(idx_shape->GetOriginShape().GetDim(1));

    uint32_t tileK = K < 1024 ? K : 1024;
    uint32_t blockRows = N / BLOCK_DIM;
    if (blockRows == 0) { blockRows = 1; }

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_N(N);
    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_blockRows(blockRows);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // output shape = idx shape
    const gert::Shape* idx_shape = context->GetInputShape(1);
    gert::Shape* out_shape = context->GetOutputShape(0);
    *out_shape = *idx_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // output dtype same as x dtype
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // double-buffering for streaming tiles

class KernelTakeAlongDim {
public:
    __aicore__ inline KernelTakeAlongDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR z,
                                uint32_t N, uint32_t M, uint32_t K,
                                uint32_t tileK, uint32_t blockRows)
    {
        this->N = N;
        this->M = M;
        this->K = K;
        this->tileK = tileK;
        this->blockRows = blockRows;
        this->rowStart = blockRows * AscendC::GetBlockIdx();

        // Set global buffers to the slice handled by this block
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (size_t)rowStart * M, (size_t)blockRows * M);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_Y *)idx + (size_t)rowStart * K, (size_t)blockRows * K);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + (size_t)rowStart * K, (size_t)blockRows * K);

        // Initialize pipe buffers
        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, this->tileK * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileK * sizeof(DTYPE_Z));
        pipe.InitBuffer(rowXQueue, 1, this->M * sizeof(DTYPE_X)); // one whole row of x
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < blockRows; ++r) {
            // Copy one entire row of x into local buffer
            AscendC::LocalTensor<DTYPE_X> xRowLocal = rowXQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xRowLocal, xGm[(size_t)r * M], M);
            rowXQueue.EnQue(xRowLocal);
            xRowLocal = rowXQueue.DeQue<DTYPE_X>();

            // Stream over K in tiles
            for (uint32_t segStart = 0; segStart < K; segStart += tileK) {
                uint32_t curLen = (segStart + tileK <= K) ? tileK : (K - segStart);
                CopyIn(r, segStart, curLen);
                Compute(curLen, xRowLocal);
                CopyOut(r, segStart, curLen);
            }

            // Free x row local buffer
            rowXQueue.FreeTensor(xRowLocal);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t r, uint32_t segStart, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_Y> idxLocal = inQueueIdx.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(idxLocal, idxGm[(size_t)r * K + segStart], curLen);
        inQueueIdx.EnQue(idxLocal);
    }

    __aicore__ inline void Compute(uint32_t curLen, const AscendC::LocalTensor<DTYPE_X>& xRowLocal)
    {
        AscendC::LocalTensor<DTYPE_Y> idxLocal = inQueueIdx.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::Gather(zLocal, xRowLocal, idxLocal, curLen);
        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueIdx.FreeTensor(idxLocal);
    }

    __aicore__ inline void CopyOut(uint32_t r, uint32_t segStart, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[(size_t)r * K + segStart], zLocal, curLen);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> rowXQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> idxGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t N;
    uint32_t M;
    uint32_t K;
    uint32_t tileK;
    uint32_t blockRows;
    uint32_t rowStart;
};

extern "C" __global__ __aicore__ void take_along_dim_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTakeAlongDim op;
    op.Init(x, idx, z,
            tiling_data.N, tiling_data.M, tiling_data.K,
            tiling_data.tileK, tiling_data.blockRows);
    op.Process();
}
"""
