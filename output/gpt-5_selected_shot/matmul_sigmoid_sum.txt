kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulSigmoidSum {
public:
    __aicore__ inline KernelMatmulSigmoidSum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batch, uint32_t inputSize, uint32_t hiddenSize, uint32_t tileNum)
    {
        this->batch = batch;
        this->inputSize = inputSize;
        this->hiddenSize = hiddenSize;
        this->tileNum = tileNum;

        // Rows per block
        this->rowsPerBlock = (batch + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        this->rowStart = AscendC::GetBlockIdx() * this->rowsPerBlock;
        if (this->rowStart + this->rowsPerBlock > this->batch) {
            this->rowsPerBlock = (this->batch > this->rowStart) ? (this->batch - this->rowStart) : 0;
        }

        // Simple tiling across hidden dimension
        this->hiddenTile = (this->hiddenSize + (tileNum * BUFFER_NUM) - 1) / (tileNum * BUFFER_NUM);
        if (this->hiddenTile == 0) this->hiddenTile = 1;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint32_t)(batch * inputSize));
        wGm.SetGlobalBuffer((__gm__ float*)w, (uint32_t)(hiddenSize * inputSize));
        bGm.SetGlobalBuffer((__gm__ float*)b, (uint32_t)(hiddenSize));
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint32_t)(batch));

        pipe.InitBuffer(xQueue, BUFFER_NUM, inputSize * sizeof(float));
        pipe.InitBuffer(wQueue, BUFFER_NUM, hiddenTile * inputSize * sizeof(float));
        pipe.InitBuffer(bQueue, BUFFER_NUM, hiddenTile * sizeof(float));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, hiddenTile * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rowsPerBlock == 0) {
            return;
        }
        // Process each row assigned to this block
        for (uint32_t r = 0; r < rowsPerBlock; ++r) {
            uint32_t globalRow = rowStart + r;
            if (globalRow >= batch) break;

            // Copy in one row of X
            AscendC::LocalTensor<float> xLocal = xQueue.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm[globalRow * inputSize], inputSize);
            xQueue.EnQue(xLocal);

            // Accumulator for the final sum over hidden dimension
            float accSum = 0.0f;

            uint32_t tiles = (hiddenSize + hiddenTile - 1) / hiddenTile;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t colStart = t * hiddenTile;
                uint32_t thisTile = (colStart + hiddenTile <= hiddenSize) ? hiddenTile : (hiddenSize - colStart);

                // Copy weight tile [thisTile, inputSize] and bias tile [thisTile]
                AscendC::LocalTensor<float> wLocal = wQueue.AllocTensor<float>();
                AscendC::LocalTensor<float> bLocal = bQueue.AllocTensor<float>();
                AscendC::DataCopy(wLocal, wGm[colStart * inputSize], thisTile * inputSize);
                AscendC::DataCopy(bLocal, bGm[colStart], thisTile);
                wQueue.EnQue(wLocal);
                bQueue.EnQue(bLocal);

                // Dequeue inputs
                AscendC::LocalTensor<float> xRow = xQueue.DeQue<float>();
                AscendC::LocalTensor<float> wTile = wQueue.DeQue<float>();
                AscendC::LocalTensor<float> bTile = bQueue.DeQue<float>();

                // Temporary vector for tile outputs
                AscendC::LocalTensor<float> tmpOut = tmpQueue.AllocTensor<float>();

                // Pseudo-compute: for each out neuron in tile, compute dot(xRow, w_i) + b_i, then sigmoid
                // Note: Below loop is illustrative; platform-provided vectorized matmul would be used in practice.
                for (uint32_t i = 0; i < thisTile; ++i) {
                    float dotv = 0.0f;
                    // compute dot product over inputSize
                    for (uint32_t k = 0; k < inputSize; ++k) {
                        // wTile is laid out as row-major [thisTile, inputSize]
                        dotv += wTile.GetValue(i * inputSize + k) * xRow.GetValue(k);
                    }
                    dotv += bTile.GetValue(i);
                    // sigmoid
                    float s = 1.0f / (1.0f + AscendC::Exp(-dotv));
                    tmpOut.SetValue(i, s);
                }

                // Accumulate sum over this tile
                for (uint32_t i = 0; i < thisTile; ++i) {
                    accSum += tmpOut.GetValue(i);
                }

                // Release local tensors for this tile
                tmpQueue.EnQue(tmpOut);
                tmpQueue.FreeTensor(tmpOut);
                wQueue.FreeTensor(wTile);
                bQueue.FreeTensor(bTile);
                // keep xRow for next tiles (no Free)
                if (t == tiles - 1) {
                    xQueue.FreeTensor(xRow);
                } else {
                    // re-enqueue xRow for reuse
                    xQueue.EnQue(xRow);
                }
            }

            // Write result scalar accSum to y[globalRow]
            AscendC::LocalTensor<float> yLocal = tmpQueue.AllocTensor<float>();
            yLocal.SetValue(0, accSum);
            AscendC::DataCopy(yGm[globalRow], yLocal, 1);
            tmpQueue.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<float> xGm;  // [B, K]
    AscendC::GlobalTensor<float> wGm;  // [N, K]
    AscendC::GlobalTensor<float> bGm;  // [N]
    AscendC::GlobalTensor<float> yGm;  // [B]

    uint32_t batch;
    uint32_t inputSize;
    uint32_t hiddenSize;
    uint32_t tileNum;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t hiddenTile;
};

extern "C" __global__ __aicore__ void matmul_sigmoid_sum_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSigmoidSum op;
    op.Init(x, weight, bias, y,
            tiling_data.batch, tiling_data.inputSize, tiling_data.hiddenSize, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: fuses linear matmul + sigmoid + sum into a custom AscendC op.
    """
    def __init__(self, input_size, hidden_size):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(input_size, hidden_size)

    def forward(self, x):
        # x: [B, K], weight: [N, K], bias: [N]; result: [B, 1]
        return custom_ops_lib.matmul_sigmoid_sum_custom(x, self.linear.weight, self.linear.bias)
