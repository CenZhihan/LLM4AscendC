kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t TMP_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 1;

class KernelConv2dMinTanhTanh {
public:
    __aicore__ inline KernelConv2dMinTanhTanh() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t tileHW)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->hw = h * w;
        this->tileHW = tileHW > 0 ? (tileHW <= this->hw ? tileHW : this->hw) : this->hw;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)n * c * this->hw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint64_t)n * this->hw);

        pipe.InitBuffer(inBuf, IN_BUF_NUM, this->tileHW * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpBuf, TMP_BUF_NUM, this->tileHW * sizeof(DTYPE_X));
        pipe.InitBuffer(outBuf, OUT_BUF_NUM, this->tileHW * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // split batch among blocks
        uint32_t blkNum = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();
        uint32_t base = n / blkNum;
        uint32_t rem = n % blkNum;
        uint32_t nStart = blkIdx < rem ? blkIdx * (base + 1) : (rem * (base + 1) + (blkIdx - rem) * base);
        uint32_t nCount = blkIdx < rem ? (base + 1) : base;

        for (uint32_t ni = 0; ni < nCount; ++ni) {
            uint32_t nCur = nStart + ni;
            uint64_t xBatchBase = (uint64_t)nCur * c * hw;
            uint64_t yBatchBase = (uint64_t)nCur * hw;

            for (uint32_t hwStart = 0; hwStart < hw; hwStart += tileHW) {
                uint32_t len = AscendC::Min(tileHW, hw - hwStart);

                AscendC::LocalTensor<DTYPE_X> minLocal = inBuf.AllocTensor<DTYPE_X>();
                // load first channel slice to initialize min
                AscendC::DataCopy(minLocal, xGm[xBatchBase + /*c=0*/ hwStart], len);

                for (uint32_t cc = 1; cc < c; ++cc) {
                    AscendC::LocalTensor<DTYPE_X> curLocal = tmpBuf.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(curLocal, xGm[xBatchBase + (uint64_t)cc * hw + hwStart], len);
                    AscendC::Minimum(minLocal, minLocal, curLocal, len);
                    tmpBuf.FreeTensor(curLocal);
                }

                AscendC::LocalTensor<DTYPE_Y> yLocal = outBuf.AllocTensor<DTYPE_Y>();
                AscendC::Tanh(yLocal, minLocal, len);
                AscendC::Tanh(yLocal, yLocal, len);

                AscendC::DataCopy(yGm[yBatchBase + hwStart], yLocal, len);

                outBuf.FreeTensor(yLocal);
                inBuf.FreeTensor(minLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, TMP_BUF_NUM> tmpBuf;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outBuf;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t h;
    uint32_t w;
    uint32_t hw;
    uint32_t tileHW;
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanh op;
    op.Init(x, y, tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w, tiling_data.tileHW);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that keeps the Conv2d in PyTorch and replaces: min over channel + tanh + tanh
    with a fused custom AscendC operator.
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        x = self.conv(x)
        x = custom_ops_lib.conv2d_min_tanh_tanh_custom(x)
        return x
