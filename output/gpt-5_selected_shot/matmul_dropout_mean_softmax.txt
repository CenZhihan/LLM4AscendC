kernel_src="""
#include "kernel_operator.h"
#include <float.h>

class KernelMatmulDropoutMeanSoftmax {
public:
    __aicore__ inline KernelMatmulDropoutMeanSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR p, GM_ADDR z,
                                uint32_t m, uint32_t k, uint32_t n, uint32_t tileN)
    {
        xGm = (__gm__ float*)x;
        wGm = (__gm__ float*)w;
        bGm = (__gm__ float*)b;
        pGm = (__gm__ float*)p;
        zGm = (__gm__ float*)z;

        this->M = m;
        this->K = k;
        this->N = n;
        this->tileN = tileN;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();
        rowsPerBlock = (M + blocks - 1) / blocks;
        rowStart = bid * rowsPerBlock;
        rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > M) rowEnd = M;
    }

    __aicore__ inline void Process()
    {
        float dropout_p = (pGm != nullptr) ? pGm[0] : 0.0f;
        float scale = 1.0f - dropout_p;

        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            float row_max = -FLT_MAX;

            // Pass 1: compute matmul + bias, apply dropout mean scaling; store provisional outputs and track max
            for (uint32_t j0 = 0; j0 < N; j0 += tileN) {
                uint32_t curN = (j0 + tileN <= N) ? tileN : (N - j0);
                for (uint32_t jj = 0; jj < curN; ++jj) {
                    uint32_t j = j0 + jj;
                    float acc = bGm[j];
                    // dot product of x[r, :] and w[j, :]
                    const __gm__ float* xRow = xGm + r * K;
                    const __gm__ float* wRow = wGm + j * K; // weight stored as [N, K]
                    for (uint32_t kk = 0; kk < K; ++kk) {
                        acc += xRow[kk] * wRow[kk];
                    }
                    acc *= scale;
                    zGm[r * N + j] = acc; // store provisional
                    row_max = (acc > row_max) ? acc : row_max;
                }
            }

            // Pass 2: compute sum of exp(v - max)
            float row_sum = 0.0f;
            for (uint32_t j0 = 0; j0 < N; j0 += tileN) {
                uint32_t curN = (j0 + tileN <= N) ? tileN : (N - j0);
                for (uint32_t jj = 0; jj < curN; ++jj) {
                    uint32_t j = j0 + jj;
                    float v = zGm[r * N + j];
                    float e = AscendC::Exp(v - row_max);
                    zGm[r * N + j] = e; // temporarily store exp
                    row_sum += e;
                }
            }

            // Pass 3: normalize
            float inv_sum = 1.0f / row_sum;
            for (uint32_t j0 = 0; j0 < N; j0 += tileN) {
                uint32_t curN = (j0 + tileN <= N) ? tileN : (N - j0);
                for (uint32_t jj = 0; jj < curN; ++jj) {
                    uint32_t j = j0 + jj;
                    float e = zGm[r * N + j];
                    zGm[r * N + j] = e * inv_sum;
                }
            }
        }
    }

private:
    __gm__ float* xGm;
    __gm__ float* wGm;
    __gm__ float* bGm;
    __gm__ float* pGm;
    __gm__ float* zGm;

    uint32_t M, K, N;
    uint32_t tileN;
    uint32_t rowsPerBlock;
    uint32_t rowStart, rowEnd;
};

extern "C" __global__ __aicore__ void matmul_dropout_mean_softmax_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR p, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulDropoutMeanSoftmax op;
    op.Init(x, w, b, p, z, tiling_data.m, tiling_data.k, tiling_data.n, tiling_data.tileN);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom AscendC op: linear (matmul + bias) -> dropout mean scaling -> softmax.
    Dropout is applied by scaling with (1 - p) to reflect expected value (mean) during training.
    """
    def __init__(self, in_features, out_features, dropout_p):
        super(ModelNew, self).__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.dropout = torch.nn.Dropout(dropout_p)

    def forward(self, x):
        # Fused custom op expects: x [B, K], weight [N, K], bias [N], p scalar tensor
        p_tensor = torch.tensor([self.dropout.p], dtype=x.dtype, device=x.device)
        return custom_ops_lib.matmul_dropout_mean_softmax_custom(x, self.matmul.weight, self.matmul.bias, p_tensor)
