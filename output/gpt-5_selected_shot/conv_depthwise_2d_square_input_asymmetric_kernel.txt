#include "kernel_operator.h"

class KernelDWConvVert {
public:
    __aicore__ inline KernelDWConvVert() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t KH, uint32_t KW,
                                uint32_t SH, uint32_t SW,
                                uint32_t PH, uint32_t PW,
                                uint32_t DH, uint32_t DW,
                                uint32_t HO, uint32_t WO,
                                uint32_t totalLength) {
        xGm = reinterpret_cast<__gm__ DTYPE_X*>(x);
        wGm = reinterpret_cast<__gm__ DTYPE_Y*>(w);
        yGm = reinterpret_cast<__gm__ DTYPE_Z*>(y);

        this->N = N; this->C = C; this->H = H; this->W = W;
        this->KH = KH; this->KW = KW;
        this->SH = SH; this->SW = SW;
        this->PH = PH; this->PW = PW;
        this->DH = DH; this->DW = DW;
        this->HO = HO; this->WO = WO;

        this->totalLength = totalLength;
        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();
        this->blockLength = totalLength / this->blockNum;
        this->startIndex = this->blockLength * this->blockIdx;
    }

    __aicore__ inline void Process() {
        // Flattened output index loop for this block
        for (uint32_t i = 0; i < blockLength; ++i) {
            uint32_t outIndex = startIndex + i;

            // Recover n, c, oh, ow from flattened output index in NCHW
            uint32_t ow = outIndex % WO;
            uint32_t tmp1 = outIndex / WO;
            uint32_t oh = tmp1 % HO;
            uint32_t tmp2 = tmp1 / HO;
            uint32_t c  = tmp2 % C;
            uint32_t n  = tmp2 / C;

            // Compute input base indices with padding/stride
            int32_t h_base = static_cast<int32_t>(oh) * static_cast<int32_t>(SH) - static_cast<int32_t>(PH);
            int32_t w_base = static_cast<int32_t>(ow) * static_cast<int32_t>(SW) - static_cast<int32_t>(PW);

            // Accumulate over KH (KW is expected to be 1)
            float acc = 0.0f;

            if (w_base >= 0 && w_base < static_cast<int32_t>(W)) {
                for (uint32_t kh = 0; kh < KH; ++kh) {
                    int32_t h_in = h_base + static_cast<int32_t>(kh) * static_cast<int32_t>(DH);
                    if (h_in < 0 || h_in >= static_cast<int32_t>(H)) {
                        continue;
                    }

                    // input index: (((n*C + c)*H + h_in)*W + w_base)
                    uint32_t x_idx = (((n * C + c) * H + static_cast<uint32_t>(h_in)) * W + static_cast<uint32_t>(w_base));
                    // weight index: (c * KH + kh) as KW == 1 and groups == C, channels multiplier = 1
                    uint32_t w_idx = c * (KH * KW) + kh * KW + 0U;

                    acc += static_cast<float>(xGm[x_idx]) * static_cast<float>(wGm[w_idx]);
                }
            }

            // output index equals outIndex for contiguous NCHW
            yGm[outIndex] = static_cast<DTYPE_Z>(acc);
        }
    }

private:
    __gm__ DTYPE_X* xGm;
    __gm__ DTYPE_Y* wGm;
    __gm__ DTYPE_Z* yGm;

    uint32_t N, C, H, W;
    uint32_t KH, KW;
    uint32_t SH, SW;
    uint32_t PH, PW;
    uint32_t DH, DW;
    uint32_t HO, WO;

    uint32_t totalLength;
    uint32_t blockNum;
    uint32_t blockIdx;
    uint32_t blockLength;
    uint32_t startIndex;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDWConvVert op;
    op.Init(x, w, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.KH, tiling_data.KW,
            tiling_data.SH, tiling_data.SW,
            tiling_data.PH, tiling_data.PW,
            tiling_data.DH, tiling_data.DW,
            tiling_data.HO, tiling_data.WO,
            tiling_data.totalLength);
    op.Process();
}

    const at::Tensor& x,
    const at::Tensor& w,

    TORCH_CHECK(x.device().type() == c10::kPrivateUse1, "Input must be on NPU (PrivateUse1)");
    TORCH_CHECK(w.device().type() == c10::kPrivateUse1, "Weight must be on NPU (PrivateUse1)");

    auto H = x.size(2);
    auto W = x.size(3);

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized depthwise 2D convolution with a square input and an asymmetric (K, 1) kernel
    implemented via a custom AscendC kernel.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        assert not bias, "This custom kernel currently does not support bias."
        self.in_channels = in_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        # Depthwise weight: (C, 1, KH, KW=1)
        weight = torch.randn(in_channels, 1, kernel_size, 1, dtype=torch.float32)
        # Kaiming-like scaling for stability
        weight = weight * (1.0 / (kernel_size ** 0.5))
        self.weight = torch.nn.Parameter(weight)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_depthwise_2d_square_input_asymmetric_kernel_custom(
            x, self.weight, self.stride, self.padding, self.dilation
