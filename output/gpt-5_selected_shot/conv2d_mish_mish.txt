kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // simple pipeline for fused op

// Fused Conv2D + Mish + Mish kernel (stride=1, pad=0, dilation=1, groups=1)
class KernelConv2dMishMish {
public:
    __aicore__ inline KernelConv2dMishMish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t OC, uint32_t KH, uint32_t KW,
                                uint32_t OH, uint32_t OW,
                                uint32_t totalOutSize, uint32_t tileNum)
    {
        this->N = N; this->C = C; this->H = H; this->W = W;
        this->OC = OC; this->KH = KH; this->KW = KW;
        this->OH = OH; this->OW = OW;
        this->totalOutSize = totalOutSize;

        this->blockLength = (totalOutSize + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        this->startIndex = this->blockLength * AscendC::GetBlockIdx();
        if (this->startIndex + this->blockLength > totalOutSize) {
            this->blockLength = totalOutSize - this->startIndex;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N*C*H*W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y*)w, (uint64_t)OC*C*KH*KW);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y*)b, (uint64_t)OC);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z, (uint64_t)N*OC*OH*OW);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < this->blockLength; ++i) {
            uint32_t idx = this->startIndex + i;
            if (idx >= this->totalOutSize) break;

            // Unravel linear index to (n, oc, oh, ow)
            uint32_t ocStride = OH * OW;
            uint32_t nchwStride = OC * ocStride;
            uint32_t n  = idx / (OC * ocStride);
            uint32_t rem = idx % (OC * ocStride);
            uint32_t oc = rem / ocStride;
            uint32_t rem2 = rem % ocStride;
            uint32_t oh = rem2 / OW;
            uint32_t ow = rem2 % OW;

            // Convolution accumulation
            float acc = 0.0f;
            // bias
            acc += static_cast<float>(bGm.GetValue(oc));

            for (uint32_t ic = 0; ic < C; ++ic) {
                for (uint32_t kh = 0; kh < KH; ++kh) {
                    int32_t ih = static_cast<int32_t>(oh) + static_cast<int32_t>(kh);
                    if (ih < 0 || ih >= static_cast<int32_t>(H)) continue;
                    for (uint32_t kw = 0; kw < KW; ++kw) {
                        int32_t iw = static_cast<int32_t>(ow) + static_cast<int32_t>(kw);
                        if (iw < 0 || iw >= static_cast<int32_t>(W)) continue;

                        uint64_t x_off = (((uint64_t)n * C + ic) * H + ih) * W + iw;
                        uint64_t w_off = (((uint64_t)oc * C + ic) * KH + kh) * KW + kw;

                        float xv = static_cast<float>(xGm.GetValue(x_off));
                        float wv = static_cast<float>(wGm.GetValue(w_off));
                        acc += xv * wv;
                    }
                }
            }

            // Mish activation twice: y = x * tanh(softplus(x)), softplus(x) = log(1 + exp(x))
            float y1 = acc;
            // simple implementation of mish using device-compatible intrinsics is assumed
            float sp1 = logf(1.0f + expf(y1));
            float t1 = tanhf(sp1);
            float mish1 = y1 * t1;

            float sp2 = logf(1.0f + expf(mish1));
            float t2 = tanhf(sp2);
            float mish2 = mish1 * t2;

            uint64_t z_off = (((uint64_t)n * OC + oc) * OH + oh) * OW + ow;
            zGm.SetValue(z_off, static_cast<DTYPE_Z>(mish2));
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t N, C, H, W, OC, KH, KW, OH, OW;
    uint32_t totalOutSize;
    uint32_t blockLength;
    uint32_t startIndex;
};

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMishMish op;
    op.Init(x, w, b, z,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.OC, tiling_data.KH, tiling_data.KW,
            tiling_data.OH, tiling_data.OW,
            tiling_data.totalOutSize, tiling_data.tileNum);
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv2d_mish_mish_custom_impl_npu(const at::Tensor& x, const at::Tensor& w, const at::Tensor& b) {
    TORCH_CHECK(x.dim() == 4, "Input x must be 4D NCHW");
    TORCH_CHECK(w.dim() == 4, "Weight w must be 4D OIHW");
    TORCH_CHECK(b.dim() == 1, "Bias b must be 1D with size equal to out_channels");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "Only float is supported");
    TORCH_CHECK(w.scalar_type() == at::kFloat && b.scalar_type() == at::kFloat, "Only float is supported");

    int64_t N = x.size(0);
    int64_t C = x.size(1);
    int64_t H = x.size(2);
    int64_t W = x.size(3);

    int64_t OC = w.size(0);
    int64_t KH = w.size(2);
    int64_t KW = w.size(3);

    // Defaults: stride=1, pad=0
    int64_t OH = (H - KH) + 1;
    int64_t OW = (W - KW) + 1;

    TORCH_CHECK(OH > 0 && OW > 0, "Invalid output size, check kernel size vs input size");
    at::Tensor result = at::empty({N, OC, OH, OW}, x.options());
    EXEC_NPU_CMD(aclnnConv2dMishMishCustom, x, w, b, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_mish_mish_custom", &conv2d_mish_mish_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_mish_mish_custom", &conv2d_mish_mish_custom_impl_npu, "Fused Conv2D + Mish + Mish");
