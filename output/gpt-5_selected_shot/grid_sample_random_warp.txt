kernel_src="""
#include "kernel_operator.h"
#include <cmath>

typedef float DTYPE_X;
typedef float DTYPE_GRID;
typedef float DTYPE_Y;

class KernelGridSampleRandomWarp {
public:
    __aicore__ inline KernelGridSampleRandomWarp() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR grid, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t Ho, uint32_t Wo, uint32_t totalLength)
    {
        this->N = N; this->C = C; this->H = H; this->W = W;
        this->Ho = Ho; this->Wo = Wo;
        this->totalLength = totalLength;

        xPtr = (__gm__ DTYPE_X*)x;
        gridPtr = (__gm__ DTYPE_GRID*)grid;
        yPtr = (__gm__ DTYPE_Y*)y;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        this->blockStart = (totalLength / blocks) * blockIdx;
        this->blockEnd = (blockIdx == (blocks - 1)) ? totalLength : ((totalLength / blocks) * (blockIdx + 1));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t idx = this->blockStart; idx < this->blockEnd; ++idx) {
            // Map 1D idx -> (n, c, h_out, w_out)
            uint32_t spatial = Ho * Wo;
            uint32_t per_n = C * spatial;

            uint32_t n = idx / per_n;
            uint32_t rem1 = idx % per_n;
            uint32_t c = rem1 / spatial;
            uint32_t rem2 = rem1 % spatial;
            uint32_t h_out = rem2 / Wo;
            uint32_t w_out = rem2 % Wo;

            // Read grid coords (normalized [-1, 1], align_corners=False)
            uint32_t gridBase = (n * Ho * Wo + h_out * Wo + w_out) * 2;
            float gx = gridPtr[gridBase + 0];
            float gy = gridPtr[gridBase + 1];

            // Denormalize to source coordinates (align_corners=False)
            float x_src = ((gx + 1.0f) * static_cast<float>(W) - 1.0f) * 0.5f;
            float y_src = ((gy + 1.0f) * static_cast<float>(H) - 1.0f) * 0.5f;

            // Bilinear neighbors
            int32_t x0 = static_cast<int32_t>(floorf(x_src));
            int32_t y0 = static_cast<int32_t>(floorf(y_src));
            int32_t x1 = x0 + 1;
            int32_t y1 = y0 + 1;

            float wx1 = x_src - static_cast<float>(x0);
            float wy1 = y_src - static_cast<float>(y0);
            float wx0 = 1.0f - wx1;
            float wy0 = 1.0f - wy1;

            float v00 = 0.0f, v01 = 0.0f, v10 = 0.0f, v11 = 0.0f;

            // Helper to check bounds and read
            auto in_bounds = [&](int32_t yy, int32_t xx) -> bool {
                return (yy >= 0 && yy < static_cast<int32_t>(H) && xx >= 0 && xx < static_cast<int32_t>(W));
            };

            uint64_t xBase = (uint64_t)n * C * H * W + (uint64_t)c * H * W;

            if (in_bounds(y0, x0)) {
                v00 = xPtr[xBase + (uint64_t)y0 * W + (uint64_t)x0];
            }
            if (in_bounds(y0, x1)) {
                v01 = xPtr[xBase + (uint64_t)y0 * W + (uint64_t)x1];
            }
            if (in_bounds(y1, x0)) {
                v10 = xPtr[xBase + (uint64_t)y1 * W + (uint64_t)x0];
            }
            if (in_bounds(y1, x1)) {
                v11 = xPtr[xBase + (uint64_t)y1 * W + (uint64_t)x1];
            }

            float val = v00 * (wx0 * wy0) +
                        v01 * (wx1 * wy0) +
                        v10 * (wx0 * wy1) +
                        v11 * (wx1 * wy1);

            // Write output
            uint64_t yIndex = (uint64_t)n * C * Ho * Wo + (uint64_t)c * Ho * Wo + (uint64_t)h_out * Wo + (uint64_t)w_out;
            yPtr[yIndex] = val;
        }
    }

private:
    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_GRID* gridPtr;
    __gm__ DTYPE_Y* yPtr;

    uint32_t N, C, H, W, Ho, Wo;
    uint32_t totalLength;
    uint32_t blockStart;
    uint32_t blockEnd;
};

extern "C" __global__ __aicore__ void grid_sample_random_warp_custom(GM_ADDR x, GM_ADDR grid, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGridSampleRandomWarp op;
    op.Init(x, grid, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.Ho, tiling_data.Wo, tiling_data.totalLength);
    op.Process();
}
