kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConvActBn {
public:
    __aicore__ inline KernelConvActBn() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR running_mean, GM_ADDR running_var, GM_ADDR eps, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gammaGm.SetGlobalBuffer((__gm__ DTYPE_X *)gamma + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        betaGm.SetGlobalBuffer((__gm__ DTYPE_X *)beta + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        meanGm.SetGlobalBuffer((__gm__ DTYPE_X *)running_mean + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        varGm.SetGlobalBuffer((__gm__ DTYPE_X *)running_var + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        epsGm.SetGlobalBuffer((__gm__ DTYPE_X *)eps + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_X *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueGamma, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueBeta, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueMean, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueVar, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueEps, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> gLocal = inQueueGamma.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> bLocal = inQueueBeta.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> mLocal = inQueueMean.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> vLocal = inQueueVar.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> eLocal = inQueueEps.AllocTensor<DTYPE_X>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gLocal, gammaGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(bLocal, betaGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(mLocal, meanGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(vLocal, varGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(eLocal, epsGm[progress * this->tileLength], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueGamma.EnQue(gLocal);
        inQueueBeta.EnQue(bLocal);
        inQueueMean.EnQue(mLocal);
        inQueueVar.EnQue(vLocal);
        inQueueEps.EnQue(eLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> gLocal = inQueueGamma.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> bLocal = inQueueBeta.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> mLocal = inQueueMean.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> vLocal = inQueueVar.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> eLocal = inQueueEps.DeQue<DTYPE_X>();

        AscendC::LocalTensor<DTYPE_X> yLocal = outQueueY.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> tmpLocal = tmpQueue.AllocTensor<DTYPE_X>();

        // Activation: x * tanh(softplus(x))
        AscendC::Softplus(yLocal, xLocal, this->tileLength);
        AscendC::Tanh(yLocal, yLocal, this->tileLength);
        AscendC::Mul(yLocal, yLocal, xLocal, this->tileLength);

        // BatchNorm (inference): y = (y - mean) / sqrt(var + eps) * gamma + beta
        AscendC::Sub(yLocal, yLocal, mLocal, this->tileLength);
        AscendC::Add(tmpLocal, vLocal, eLocal, this->tileLength);
        AscendC::Sqrt(tmpLocal, tmpLocal, this->tileLength);
        AscendC::Div(yLocal, yLocal, tmpLocal, this->tileLength);
        AscendC::Mul(yLocal, yLocal, gLocal, this->tileLength);
        AscendC::Add(yLocal, yLocal, bLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_X>(yLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueGamma.FreeTensor(gLocal);
        inQueueBeta.FreeTensor(bLocal);
        inQueueMean.FreeTensor(mLocal);
        inQueueVar.FreeTensor(vLocal);
        inQueueEps.FreeTensor(eLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> yLocal = outQueueY.DeQue<DTYPE_X>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueGamma, inQueueBeta, inQueueMean, inQueueVar, inQueueEps;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY, tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> gammaGm;
    AscendC::GlobalTensor<DTYPE_X> betaGm;
    AscendC::GlobalTensor<DTYPE_X> meanGm;
    AscendC::GlobalTensor<DTYPE_X> varGm;
    AscendC::GlobalTensor<DTYPE_X> epsGm;
    AscendC::GlobalTensor<DTYPE_X> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_activation_batch_norm_custom(GM_ADDR x,
                                                                           GM_ADDR gamma,
                                                                           GM_ADDR beta,
                                                                           GM_ADDR running_mean,
                                                                           GM_ADDR running_var,
                                                                           GM_ADDR eps,
                                                                           GM_ADDR y,
                                                                           GM_ADDR workspace,
                                                                           GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvActBn op;
    op.Init(x, gamma, beta, running_mean, running_var, eps, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a custom AscendC kernel to fuse activation and batch normalization after convolution.
    The custom op expects broadcasted tensors for BN parameters to match the input feature map shape.
    """
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = torch.nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum)

    def forward(self, x):
        x = self.conv(x)
        # Prepare broadcasted BN parameters to match x's shape (N, C, H, W)
        gamma = self.bn.weight.view(1, -1, 1, 1).expand_as(x).contiguous()
        beta = self.bn.bias.view(1, -1, 1, 1).expand_as(x).contiguous()
        running_mean = self.bn.running_mean.view(1, -1, 1, 1).expand_as(x).contiguous()
        running_var = self.bn.running_var.view(1, -1, 1, 1).expand_as(x).contiguous()
        eps = torch.full_like(x, float(self.bn.eps))
        # Fused activation (x * tanh(softplus(x))) and batch norm using custom kernel
        x = custom_ops_lib.conv2d_activation_batch_norm_custom(x, gamma, beta, running_mean, running_var, eps)
        return x
'''
