kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    // Assume float data type
    __gm__ float* x_ptr = reinterpret_cast<__gm__ float*>(x); // [N, Cin, Hin, Win]
    __gm__ float* w_ptr = reinterpret_cast<__gm__ float*>(w); // [Cin, Cout, K, K]
    __gm__ float* y_ptr = reinterpret_cast<__gm__ float*>(y); // [N, Cout, Hout, Wout]

    uint32_t N = tiling_data.n;
    uint32_t Cin = tiling_data.cin;
    uint32_t Cout = tiling_data.cout;
    uint32_t Hin = tiling_data.hin;
    uint32_t Win = tiling_data.win;
    uint32_t K = tiling_data.k;
    uint32_t Hout = tiling_data.hout;
    uint32_t Wout = tiling_data.wout;

    uint64_t totalOut = tiling_data.totalOut;

    // Strides
    uint64_t x_stride_n = (uint64_t)Cin * Hin * Win;
    uint64_t x_stride_c = (uint64_t)Hin * Win;
    uint64_t x_stride_h = (uint64_t)Win;

    uint64_t y_stride_n = (uint64_t)Cout * Hout * Wout;
    uint64_t y_stride_c = (uint64_t)Hout * Wout;
    uint64_t y_stride_h = (uint64_t)Wout;

    uint64_t w_stride_ic = (uint64_t)Cout * K * K;
    uint64_t w_stride_oc = (uint64_t)K * K;
    uint64_t w_stride_kh = (uint64_t)K;

    // Split work by block: each block computes a contiguous slice of output elements
    uint32_t block_num = AscendC::GetBlockNum();
    uint32_t block_idx = AscendC::GetBlockIdx();
    uint64_t blk_len = (totalOut + block_num - 1) / block_num;
    uint64_t start = blk_len * block_idx;
    uint64_t end = start + blk_len;
    if (end > totalOut) end = totalOut;

    for (uint64_t out_idx = start; out_idx < end; ++out_idx) {
        // decode linear index to [n, oc, oh, ow]
        uint64_t tmp = out_idx;
        uint32_t ow = (uint32_t)(tmp % Wout);
        tmp /= Wout;
        uint32_t oh = (uint32_t)(tmp % Hout);
        tmp /= Hout;
        uint32_t oc = (uint32_t)(tmp % Cout);
        uint32_t n = (uint32_t)(tmp / Cout);

        float acc = 0.0f;

        for (uint32_t ic = 0; ic < Cin; ++ic) {
            // Iterate over kernel window
            for (uint32_t kh = 0; kh < K; ++kh) {
                // Corresponding input h index: oh - kh
                int32_t ih = (int32_t)oh - (int32_t)kh;
                if (ih < 0 || ih >= (int32_t)Hin) continue;
                for (uint32_t kw = 0; kw < K; ++kw) {
                    int32_t iw = (int32_t)ow - (int32_t)kw;
                    if (iw < 0 || iw >= (int32_t)Win) continue;

                    uint64_t x_off = (uint64_t)n * x_stride_n + (uint64_t)ic * x_stride_c +
                                     (uint64_t)ih * x_stride_h + (uint64_t)iw;
                    uint64_t w_off = (uint64_t)ic * w_stride_ic + (uint64_t)oc * w_stride_oc +
                                     (uint64_t)kh * w_stride_kh + (uint64_t)kw;

                    acc += x_ptr[x_off] * w_ptr[w_off];
                }
            }
        }

        uint64_t y_off = (uint64_t)n * y_stride_n + (uint64_t)oc * y_stride_c +
                         (uint64_t)oh * y_stride_h + (uint64_t)ow;
        y_ptr[y_off] = acc;
    }
}

    TORCH_CHECK(x.scalar_type() == at::kFloat && w.scalar_type() == at::kFloat, "Only float32 is supported");
    TORCH_CHECK(x.size(1) == w.size(0), "Cin mismatch between x and w");
    TORCH_CHECK(w.size(2) == w.size(3), "Kernel must be square");

    auto N = x.size(0);
    auto Cin = x.size(1);
    auto Hin = x.size(2);
    auto Win = x.size(3);
    auto Cout = w.size(1);
    auto K = w.size(2);

    auto Hout = Hin + K - 1;
    auto Wout = Win + K - 1;

    at::Tensor result = at::empty({N, Cout, Hout, Wout}, x.options());
    EXEC_NPU_CMD(aclnnConvTransposed2dAsymmetricInputSquareKernelCustom, x, w, result);
