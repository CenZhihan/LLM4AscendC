kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // number of tensors for each queue

// This kernel computes y[n,c,0,0] = multiplier * mean_{h,w}(x[n,c,h,w])
class KernelGAPMul {
public:
    __aicore__ inline KernelGAPMul() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR multiplier, GM_ADDR y,
                                uint32_t outerLength, uint32_t innerSize, uint32_t tileNum)
    {
        this->outerLength = outerLength; // N*C
        this->innerSize = innerSize;     // H*W
        this->blockOuter = (outerLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        // Align block start and end for current block
        this->blockStart = this->blockOuter * AscendC::GetBlockIdx();
        this->blockEnd = AscendC::Min(this->blockStart + this->blockOuter, this->outerLength);
        this->tileNum = tileNum == 0 ? 1 : tileNum;
        this->tileOuter = AscendC::Max<uint32_t>(1, (this->blockEnd - this->blockStart) / this->tileNum);
        if (this->tileOuter == 0) {
            this->tileOuter = this->blockEnd > this->blockStart ? (this->blockEnd - this->blockStart) : 1;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockStart * this->innerSize,
                            (this->blockEnd - this->blockStart) * this->innerSize);
        mGm.SetGlobalBuffer((__gm__ DTYPE_Y *)multiplier, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockStart, (this->blockEnd - this->blockStart));

        // local buffers: input tile = tileOuter * innerSize, output tile = tileOuter
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileOuter * this->innerSize * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileOuter * sizeof(DTYPE_Z));
        pipe.InitBuffer(scaleQueue, BUFFER_NUM, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // load multiplier once
        LoadScale();

        int32_t totalTiles = 0;
        if (this->blockEnd > this->blockStart) {
            uint32_t elems = this->blockEnd - this->blockStart;
            totalTiles = (elems + this->tileOuter - 1) / this->tileOuter;
        }

        for (int32_t t = 0; t < totalTiles; ++t) {
            uint32_t curOuter = this->tileOuter;
            uint32_t offsetOuter = t * this->tileOuter;
            if (offsetOuter + curOuter > (this->blockEnd - this->blockStart)) {
                curOuter = (this->blockEnd - this->blockStart) - offsetOuter;
            }
            CopyIn(offsetOuter, curOuter);
            Compute(curOuter);
            CopyOut(offsetOuter, curOuter);
        }
    }

private:
    __aicore__ inline void LoadScale()
    {
        AscendC::LocalTensor<DTYPE_Y> sLocal = scaleQueue.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(sLocal, mGm[0], 1);
        scaleQueue.EnQue(sLocal);
    }

    __aicore__ inline void CopyIn(uint32_t offsetOuter, uint32_t curOuter)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal,
                          xGm[offsetOuter * this->innerSize],
                          curOuter * this->innerSize);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t curOuter)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Y> sLocal = scaleQueue.DeQue<DTYPE_Y>();

        // For each (n,c) in the tile, reduce over innerSize to a scalar
        for (uint32_t i = 0; i < curOuter; ++i) {
            AscendC::LocalTensor<DTYPE_Z> tmp = yLocal[i]; // placeholder view
            // Reduce sum over innerSize elements
            AscendC::ReduceSum(tmp, xLocal[i * this->innerSize], this->innerSize);
            // Compute average
            DTYPE_Y invInner = static_cast<DTYPE_Y>(1.0f) / static_cast<DTYPE_Y>(this->innerSize);
            AscendC::Muls(tmp, tmp, invInner, 1);
            // Multiply by scalar multiplier
            AscendC::Muls(tmp, tmp, sLocal.GetValue(0), 1);
        }

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        scaleQueue.FreeTensor(sLocal);
    }

    __aicore__ inline void CopyOut(uint32_t offsetOuter, uint32_t curOuter)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[offsetOuter], yLocal, curOuter);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> scaleQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> mGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t outerLength;
    uint32_t innerSize;
    uint32_t blockStart;
    uint32_t blockEnd;
    uint32_t blockOuter;
    uint32_t tileNum;
    uint32_t tileOuter;
};

extern "C" __global__ __aicore__ void conv_transpose2d_multiply_global_avg_pool_global_avg_pool_mean_custom(GM_ADDR x, GM_ADDR multiplier, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGAPMul op;
    op.Init(x, multiplier, y, tiling_data.outerLength, tiling_data.innerSize, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep ConvTranspose2d on PyTorch, fuse multiply + two global average pools into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, multiplier):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.multiplier = multiplier

    def forward(self, x):
        x = self.conv_transpose(x)
        # fused: multiply by scalar and two global average pools
        scale = torch.tensor(self.multiplier, dtype=x.dtype, device=x.device).view(1)
        y = custom_ops_lib.conv_transpose2d_multiply_global_avg_pool_global_avg_pool_mean_custom(x, scale)
        return y
