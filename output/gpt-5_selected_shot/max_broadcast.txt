host_operator_src="""
#include "max_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaxBroadcastCustomTilingData tiling;

    auto* in0 = context->GetInputShape(0);
    auto* in1 = context->GetInputShape(1);

    // Assume 3D inputs (N, C, W) with broadcasting-compatible shapes.
    // x: [d0, 1, d2], y: [1, d1, 1] -> z: [d0, d1, d2]
    const auto& s0 = in0->GetOriginShape();
    const auto& s1 = in1->GetOriginShape();

    uint32_t dims0 = static_cast<uint32_t>(s0.GetDimNum());
    uint32_t dims1 = static_cast<uint32_t>(s1.GetDimNum());
    // Fallback to using last 3 dims
    uint32_t a0 = static_cast<uint32_t>(s0.GetDim(dims0 - 3));
    uint32_t a1 = static_cast<uint32_t>(s0.GetDim(dims0 - 2));
    uint32_t a2 = static_cast<uint32_t>(s0.GetDim(dims0 - 1));
    uint32_t b0 = static_cast<uint32_t>(s1.GetDim(dims1 - 3));
    uint32_t b1 = static_cast<uint32_t>(s1.GetDim(dims1 - 2));
    uint32_t b2 = static_cast<uint32_t>(s1.GetDim(dims1 - 1));

    uint32_t d0 = a0 > b0 ? a0 : b0;
    uint32_t d1 = a1 > b1 ? a1 : b1;
    uint32_t d2 = a2 > b2 ? a2 : b2;

    // We iterate segments along (d0 * d1), and tile along d2
    uint32_t totalSegments = d0 * d1;
    uint32_t segmentsPerBlock = CeilDiv(totalSegments, BLOCK_DIM);

    // Choose tileLen along d2
    uint32_t tileLen = CeilDiv(d2, TILE_NUM * 2); // considering double-buffering
    if (tileLen == 0) tileLen = d2;
    uint32_t tilesPerSeg = CeilDiv(d2, tileLen);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_d0(d0);
    tiling.set_d1(d1);
    tiling.set_d2(d2);
    tiling.set_segmentsPerBlock(segmentsPerBlock);
    tiling.set_tileLen(tileLen);
    tiling.set_tilesPerSeg(tilesPerSeg);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* wk = context->GetWorkspaceSizes(1);
    wk[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* s0 = context->GetInputShape(0);
    const gert::Shape* s1 = context->GetInputShape(1);

    const auto& a = s0->GetOriginShape();
    const auto& b = s1->GetOriginShape();

    // Broadcast infer for last 3 dims
    uint32_t dimsA = static_cast<uint32_t>(a.GetDimNum());
    uint32_t dimsB = static_cast<uint32_t>(b.GetDimNum());
    int64_t a0 = a.GetDim(dimsA - 3), a1 = a.GetDim(dimsA - 2), a2 = a.GetDim(dimsA - 1);
    int64_t b0 = b.GetDim(dimsB - 3), b1 = b.GetDim(dimsB - 2), b2 = b.GetDim(dimsB - 1);

    int64_t d0 = std::max(a0, b0);
    int64_t d1 = std::max(a1, b1);
    int64_t d2 = std::max(a2, b2);

    std::vector<int64_t> outDims = {d0, d1, d2};
    gert::Shape* out = context->GetOutputShape(0);
    out->SetDimNum(3);
    for (int i = 0; i < 3; ++i) out->SetDim(i, outDims[i]);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto dt0 = context->GetInputDataType(0);
    context->SetOutputDataType(0, dt0);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // double buffering

class KernelMaxBroadcast {
public:
    __aicore__ inline KernelMaxBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t d0, uint32_t d1, uint32_t d2,
                                uint32_t segmentsPerBlock, uint32_t tileLen, uint32_t tilesPerSeg) {
        this->d0 = d0;
        this->d1 = d1;
        this->d2 = d2;
        this->segmentsPerBlock = segmentsPerBlock;
        this->tileLen = tileLen;
        this->tilesPerSeg = tilesPerSeg;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, d0 * d2);              // x shape [d0,1,d2] => flatten length d0*d2
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, d1);                    // y shape [1,d1,1] => flatten length d1
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z, d0 * d1 * d2);          // z shape [d0,d1,d2]

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Y));  // reused for broadcast buffer
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process() {
        // Each block handles a continuous range of (d0*d1) segments
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockStartSeg = blockIdx * segmentsPerBlock;
        uint32_t totalSegments = d0 * d1;

        for (uint32_t seg = 0; seg < segmentsPerBlock; ++seg) {
            uint32_t segGlobal = blockStartSeg + seg;
            if (segGlobal >= totalSegments) break;

            uint32_t i0 = segGlobal / d1;       // along d0
            uint32_t i1 = segGlobal % d1;       // along d1

            // Base offsets
            uint32_t xBase = i0 * d2;                    // x[i0,0,:]
            uint32_t yIndex = i1;                        // y[0,i1,0]
            uint32_t zBase = i0 * (d1 * d2) + i1 * d2;   // z[i0,i1,:]

            for (uint32_t t = 0; t < tilesPerSeg; ++t) {
                uint32_t currLen = AscendC::min(this->tileLen, d2 - t * this->tileLen);
                if (currLen == 0) break;
                CopyIn(xBase, yIndex, t, currLen);
                Compute(currLen);
                CopyOut(zBase, t, currLen);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t xBase, uint32_t yIndex, uint32_t t, uint32_t currLen) {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();

        // Load x segment
        AscendC::DataCopy(xLocal, xGm[xBase + t * this->tileLen], currLen);

        // Load scalar y and broadcast into yLocal
        // Read 1 element into the first slot
        AscendC::DataCopy(yLocal, yGm[yIndex], 1);
        // Duplicate the first scalar across vector
        AscendC::Duplicate(yLocal, yLocal.GetValue(0), currLen);

        inQueueX.EnQue<DTYPE_X>(xLocal);
        inQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void Compute(uint32_t currLen) {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        AscendC::Maximum(zLocal, xLocal, yLocal, currLen);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueY.FreeTensor(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t zBase, uint32_t t, uint32_t currLen) {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[zBase + t * this->tileLen], zLocal, currLen);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t d0;
    uint32_t d1;
    uint32_t d2;
    uint32_t segmentsPerBlock;
    uint32_t tileLen;
    uint32_t tilesPerSeg;
};

extern "C" __global__ __aicore__ void max_broadcast_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxBroadcast op;
    op.Init(x, y, z,
            tiling_data.d0, tiling_data.d1, tiling_data.d2,
            tiling_data.segmentsPerBlock, tiling_data.tileLen, tiling_data.tilesPerSeg);
    op.Process();
}
