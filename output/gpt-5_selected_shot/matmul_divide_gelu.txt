kernel_src="""
#include "kernel_operator.h"
#include <cmath>

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR d, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t B = tiling_data.batch;
    const uint32_t I = tiling_data.inSize;
    const uint32_t O = tiling_data.outSize;
    const uint32_t total = tiling_data.total;

    __gm__ DTYPE_X* xGm = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_W* wGm = reinterpret_cast<__gm__ DTYPE_W*>(w);
    __gm__ DTYPE_B* bGm = reinterpret_cast<__gm__ DTYPE_B*>(b);
    __gm__ DTYPE_D* dGm = reinterpret_cast<__gm__ DTYPE_D*>(d);
    __gm__ DTYPE_Z* yGm = reinterpret_cast<__gm__ DTYPE_Z*>(y);

    if (B == 0 || I == 0 || O == 0) {
        return;
    }

    DTYPE_D divisor_val = dGm[0];

    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t work = (total + blockNum - 1) / blockNum;
    uint32_t start = work * blockIdx;
    uint32_t end = start + work;
    if (end > total) end = total;

    for (uint32_t idx = start; idx < end; ++idx) {
        uint32_t bIdx = idx / O;
        uint32_t oIdx = idx % O;

        float acc = 0.0f;
        uint64_t x_base = static_cast<uint64_t>(bIdx) * I;
        uint64_t w_base = static_cast<uint64_t>(oIdx) * I;

        for (uint32_t k = 0; k < I; ++k) {
            float xv = static_cast<float>(xGm[x_base + k]);
            float wv = static_cast<float>(wGm[w_base + k]);
            acc += xv * wv;
        }
        acc += static_cast<float>(bGm[oIdx]);
        acc = acc / static_cast<float>(divisor_val);

        // GELU approximate using tanh formulation
        float x3 = acc * acc * acc;
        float inner = 0.7978845608028654f * (acc + 0.044715f * x3);
        float gelu = 0.5f * acc * (1.0f + tanhf(inner));

        yGm[static_cast<uint64_t>(bIdx) * O + oIdx] = static_cast<DTYPE_Z>(gelu);
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused MatMul + bias + divide + GELU using custom AscendC kernel.
    """
    def __init__(self, input_size, output_size, divisor):
        super().__init__()
        self.linear = torch.nn.Linear(input_size, output_size)
        self.divisor = float(divisor)

    def forward(self, x):
        # Prepare inputs for fused kernel:
        # x: [B, I], w: [O, I], b: [O], d: scalar as 1-element tensor
        d = torch.tensor([self.divisor], dtype=x.dtype, device=x.device)
        return custom_ops_lib.matmul_divide_gelu_custom(x, self.linear.weight, self.linear.bias, d)
