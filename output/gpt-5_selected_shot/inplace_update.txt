kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // buffer count for pipeline

class KernelInplaceUpdate {
public:
    __aicore__ inline KernelInplaceUpdate() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR value, GM_ADDR z,
                                uint32_t rowsX, uint32_t cols, uint32_t idxLen, uint32_t tileNum)
    {
        this->rows = rowsX;
        this->cols = cols;
        this->idxLen = idxLen;
        this->tileNum = tileNum;

        // Partition the index list across blocks to avoid write races
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->blockStartIdx = (uint64_t)(idxLen) * blockIdx / blockNum;
        uint64_t blockEndIdx = (uint64_t)(idxLen) * (blockIdx + 1) / blockNum;
        this->blockIdxLen = blockEndIdx - this->blockStartIdx;

        idxGm.SetGlobalBuffer((__gm__ int64_t*)idx + this->blockStartIdx, this->blockIdxLen);
        valGm.SetGlobalBuffer((__gm__ float*)value + this->blockStartIdx * (uint64_t)this->cols, this->blockIdxLen * (uint64_t)this->cols);
        zGm.SetGlobalBuffer((__gm__ float*)z, (uint64_t)this->rows * (uint64_t)this->cols);

        tileIdxPerBuf = AscendC::Max((uint32_t)1, (uint32_t)(this->blockIdxLen / this->tileNum / BUFFER_NUM));
        tileValLength = tileIdxPerBuf * this->cols;

        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, tileIdxPerBuf * sizeof(int64_t));
        pipe.InitBuffer(inQueueVal, BUFFER_NUM, tileValLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (this->blockIdxLen == 0) {
            return;
        }
        uint64_t loopCount = (this->blockIdxLen + tileIdxPerBuf - 1) / tileIdxPerBuf;
        for (uint64_t i = 0; i < loopCount; ++i) {
            uint32_t tileCount = (uint32_t)AscendC::Min((uint64_t)tileIdxPerBuf, this->blockIdxLen - i * tileIdxPerBuf);
            CopyIn(i, tileCount);
            Compute(tileCount);
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t progress, uint32_t tileCount)
    {
        AscendC::LocalTensor<int64_t> idxLocal = inQueueIdx.AllocTensor<int64_t>();
        AscendC::LocalTensor<float> valLocal = inQueueVal.AllocTensor<float>();

        // Copy indices
        AscendC::DataCopy(idxLocal, idxGm[progress * tileIdxPerBuf], tileCount);
        // Copy values corresponding to indices
        AscendC::DataCopy(valLocal, valGm[progress * tileIdxPerBuf * this->cols], tileCount * this->cols);

        inQueueIdx.EnQue(idxLocal);
        inQueueVal.EnQue(valLocal);
    }

    __aicore__ inline void Compute(uint32_t tileCount)
    {
        AscendC::LocalTensor<int64_t> idxLocal = inQueueIdx.DeQue<int64_t>();
        AscendC::LocalTensor<float> valLocal = inQueueVal.DeQue<float>();

        // For each index in the tile, write the corresponding row from value into z
        for (uint32_t j = 0; j < tileCount; ++j) {
            int64_t rowIndex = idxLocal[j];
            // Write row j of valLocal to row 'rowIndex' of zGm
            AscendC::DataCopy(zGm[(uint64_t)rowIndex * this->cols], valLocal[j * this->cols], this->cols);
        }

        inQueueIdx.FreeTensor(idxLocal);
        inQueueVal.FreeTensor(valLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdx, inQueueVal;

    AscendC::GlobalTensor<int64_t> idxGm;
    AscendC::GlobalTensor<float> valGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tileNum;

    uint64_t blockStartIdx;
    uint64_t blockIdxLen;

    uint32_t tileIdxPerBuf;
    uint32_t tileValLength;
};

extern "C" __global__ __aicore__ void inplace_update_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR value, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelInplaceUpdate op;
    op.Init(x, idx, value, z, tiling_data.rowsX, tiling_data.cols, tiling_data.idxLen, tiling_data.tileNum);
    op.Process();
}
"""
