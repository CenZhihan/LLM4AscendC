#include "average_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t compute_out_dim(uint32_t in, uint32_t k, uint32_t s, uint32_t p) {
    return static_cast<uint32_t>((static_cast<int64_t>(in) + 2 * static_cast<int64_t>(p) - static_cast<int64_t>(k)) / static_cast<int64_t>(s) + 1);
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    AveragePooling3dCustomTilingData tiling;

    const auto* inShape = context->GetInputShape(0);
    auto originShape = inShape->GetOriginShape();
    // Expect 5D: N, C, D, H, W
    uint32_t n = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t c = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t d = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t h = static_cast<uint32_t>(originShape.GetDim(3));
    uint32_t w = static_cast<uint32_t>(originShape.GetDim(4));

    // Read attributes
    int64_t k_attr = 1;
    int64_t s_attr = 1;
    int64_t p_attr = 0;
    if (context->HaveAttr("kernel_size")) {
        k_attr = context->GetAttrInt("kernel_size");
    }
    if (context->HaveAttr("stride")) {
        s_attr = context->GetAttrInt("stride");
    }
    if (context->HaveAttr("padding")) {
        p_attr = context->GetAttrInt("padding");
    }
    uint32_t k = static_cast<uint32_t>(k_attr);
    uint32_t s = static_cast<uint32_t>(s_attr);
    uint32_t p = static_cast<uint32_t>(p_attr);

    uint32_t out_d = compute_out_dim(d, k, s, p);
    uint32_t out_h = compute_out_dim(h, k, s, p);
    uint32_t out_w = compute_out_dim(w, k, s, p);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_d(d);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_out_d(out_d);
    tiling.set_out_h(out_h);
    tiling.set_out_w(out_w);
    tiling.set_k(k);
    tiling.set_s(s);
    tiling.set_p(p);

    uint64_t totalOut = static_cast<uint64_t>(n) * c * out_d * out_h * out_w;
    tiling.set_totalOut(totalOut);

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}


namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    auto inShape = x_shape->GetOriginShape();
    // Attributes
    int64_t k = 1, s = 1, p = 0;
    if (context->HaveAttr("kernel_size")) k = context->GetAttrInt("kernel_size");
    if (context->HaveAttr("stride"))      s = context->GetAttrInt("stride");
    if (context->HaveAttr("padding"))     p = context->GetAttrInt("padding");

    uint32_t n = static_cast<uint32_t>(inShape.GetDim(0));
    uint32_t c = static_cast<uint32_t>(inShape.GetDim(1));
    uint32_t d = static_cast<uint32_t>(inShape.GetDim(2));
    uint32_t h = static_cast<uint32_t>(inShape.GetDim(3));
    uint32_t w = static_cast<uint32_t>(inShape.GetDim(4));

    auto outD = static_cast<uint32_t>((static_cast<int64_t>(d) + 2 * p - k) / s + 1);
    auto outH = static_cast<uint32_t>((static_cast<int64_t>(h) + 2 * p - k) / s + 1);
    auto outW = static_cast<uint32_t>((static_cast<int64_t>(w) + 2 * p - k) / s + 1);

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, c);
    y_shape->SetDim(2, outD);
    y_shape->SetDim(3, outH);
    y_shape->SetDim(4, outW);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    // Assume float data type by default
    __gm__ const DTYPE_X* x_ptr = reinterpret_cast<__gm__ const DTYPE_X*>(x);
    __gm__ DTYPE_Y* y_ptr = reinterpret_cast<__gm__ DTYPE_Y*>(y);

    uint32_t N = tiling_data.n;
    uint32_t C = tiling_data.c;
    uint32_t D = tiling_data.d;
    uint32_t H = tiling_data.h;
    uint32_t W = tiling_data.w;

    uint32_t OD = tiling_data.out_d;
    uint32_t OH = tiling_data.out_h;
    uint32_t OW = tiling_data.out_w;

    uint32_t K = tiling_data.k;
    uint32_t S = tiling_data.s;
    uint32_t P = tiling_data.p;

    uint64_t totalOut = tiling_data.totalOut;

    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();

    uint64_t perBlock = (totalOut + blockNum - 1) / blockNum;
    uint64_t startIdx = perBlock * blockIdx;
    uint64_t endIdx = startIdx + perBlock;
    if (endIdx > totalOut) endIdx = totalOut;

    auto calc_input_offset = [=](__ubuf__ uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) -> uint64_t {
        return (((static_cast<uint64_t>(n) * C + c) * D + d) * H + h) * W + w;
    };
    auto calc_output_offset = [=](__ubuf__ uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) -> uint64_t {
        return (((static_cast<uint64_t>(n) * C + c) * OD + d) * OH + h) * OW + w;
    };

    for (uint64_t idx = startIdx; idx < endIdx; ++idx) {
        uint64_t t = idx;
        uint32_t ow = static_cast<uint32_t>(t % OW); t /= OW;
        uint32_t oh = static_cast<uint32_t>(t % OH); t /= OH;
        uint32_t od = static_cast<uint32_t>(t % OD); t /= OD;
        uint32_t c  = static_cast<uint32_t>(t % C);  t /= C;
        uint32_t n  = static_cast<uint32_t>(t);

        int32_t in_d0 = static_cast<int32_t>(od) * static_cast<int32_t>(S) - static_cast<int32_t>(P);
        int32_t in_h0 = static_cast<int32_t>(oh) * static_cast<int32_t>(S) - static_cast<int32_t>(P);
        int32_t in_w0 = static_cast<int32_t>(ow) * static_cast<int32_t>(S) - static_cast<int32_t>(P);

        float sumVal = 0.0f;
        uint32_t count = 0;

        for (uint32_t kd = 0; kd < K; ++kd) {
            int32_t id = in_d0 + static_cast<int32_t>(kd);
            if (id < 0 || id >= static_cast<int32_t>(D)) continue;
            for (uint32_t kh = 0; kh < K; ++kh) {
                int32_t ih = in_h0 + static_cast<int32_t>(kh);
                if (ih < 0 || ih >= static_cast<int32_t>(H)) continue;

                int32_t iw_start = in_w0;
                int32_t iw_end = in_w0 + static_cast<int32_t>(K);

                int32_t iw_s = iw_start < 0 ? 0 : iw_start;
                int32_t iw_e = iw_end > static_cast<int32_t>(W) ? static_cast<int32_t>(W) : iw_end;

                for (int32_t iw = iw_s; iw < iw_e; ++iw) {
                    uint64_t in_off = calc_input_offset(n, c, static_cast<uint32_t>(id), static_cast<uint32_t>(ih), static_cast<uint32_t>(iw));
                    sumVal += static_cast<float>(x_ptr[in_off]);
                    ++count;
                }
            }
        }

        float outVal = (count > 0) ? (sumVal / static_cast<float>(count)) : 0.0f;
        uint64_t out_off = calc_output_offset(n, c, od, oh, ow);
        y_ptr[out_off] = static_cast<DTYPE_Y>(outVal);
    }
}
