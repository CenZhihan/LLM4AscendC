kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvTranspose2D {
public:
    __aicore__ inline KernelConvTranspose2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t Hout, uint32_t Wout,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->N = N;
        this->Cin = Cin;
        this->Cout = Cout;
        this->H = H;
        this->W = W;
        this->K = K;
        this->Hout = Hout;
        this->Wout = Wout;
        this->totalLength = totalLength;

        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();
        this->blockLength = totalLength / this->blockNum;
        this->startOffset = this->blockLength * this->blockIdx;

        // Set global buffers: read-only x/w from start, y starts at block offset
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, N * Cin * H * W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)w, Cin * Cout * K * K);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->startOffset, this->blockLength);

        // Minimal pipeline buffers for output tiles if needed
        pipe.InitBuffer(outQueueY, BUFFER_NUM, 1 * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // Naive per-element computation for each output element owned by this block
        for (uint32_t idx = 0; idx < this->blockLength; ++idx) {
            uint64_t gidx = static_cast<uint64_t>(this->startOffset) + idx;

            // Decode N, Cout, Hout, Wout from linear index
            uint32_t wout = gidx % Wout;
            uint64_t tmp1 = gidx / Wout;
            uint32_t hout = tmp1 % Hout;
            uint64_t tmp2 = tmp1 / Hout;
            uint32_t cout = tmp2 % Cout;
            uint32_t n = tmp2 / Cout;

            float acc = 0.0f;

            // Deconvolution (transpose conv) for stride=1, padding=0
            for (uint32_t cin = 0; cin < Cin; ++cin) {
                for (uint32_t kh = 0; kh < K; ++kh) {
                    for (uint32_t kw = 0; kw < K; ++kw) {
                        // Map to input coordinate
                        int32_t hi = static_cast<int32_t>(hout) - static_cast<int32_t>(K - 1 - kh);
                        int32_t wi = static_cast<int32_t>(wout) - static_cast<int32_t>(K - 1 - kw);
                        if (hi >= 0 && hi < static_cast<int32_t>(H) && wi >= 0 && wi < static_cast<int32_t>(W)) {
                            // x index: (((n * Cin + cin) * H + hi) * W + wi)
                            uint64_t xIndex = (((static_cast<uint64_t>(n) * Cin + cin) * H + static_cast<uint32_t>(hi)) * W + static_cast<uint32_t>(wi));
                            // w index: (((cin * Cout + cout) * K + kh) * K + kw)
                            uint64_t wIndex = (((static_cast<uint64_t>(cin) * Cout + cout) * K + kh) * K + kw);

                            DTYPE_X xval = xGm[xIndex];
                            DTYPE_Y wval = wGm[wIndex];
                            acc += static_cast<float>(xval) * static_cast<float>(wval);
                        }
                    }
                }
            }
            yGm[idx] = static_cast<DTYPE_Z>(acc);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N, Cin, Cout, H, W, K, Hout, Wout;
    uint32_t totalLength;
    uint32_t blockNum;
    uint32_t blockIdx;
    uint32_t blockLength;
    uint32_t startOffset;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose2D op;
    op.Init(x, w, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.K,
            tiling_data.Hout, tiling_data.Wout,
            tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized transposed 2D convolution using custom AscendC kernel.
    Assumes square input and square kernel, stride=1, padding=0, groups=1, bias=False.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0, groups: int = 1, bias: bool = False):
        super().__init__()
        assert stride == 1, "Custom kernel expects stride=1"
        assert padding == 0, "Custom kernel expects padding=0"
        assert groups == 1, "Custom kernel expects groups=1"
        assert bias is False, "Custom kernel expects bias=False"
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        # ConvTranspose2d weight shape: (in_channels, out_channels, kH, kW)
        w = torch.empty(in_channels, out_channels, kernel_size, kernel_size)
        torch.nn.init.kaiming_uniform_(w, a=5 ** 0.5)
        self.weight = torch.nn.Parameter(w)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_2d_square_input_square_kernel_custom(x, self.weight)
