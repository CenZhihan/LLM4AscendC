kernel_src="""
#include "kernel_operator.h"

// A simple reference implementation for transposed 3D convolution (stride=1, pad=0, dilation=1, groups=1, no bias).
// Weight layout matches PyTorch ConvTranspose3d: [Cin, Cout, K, K, K].

class KernelConvTranspose3D {
public:
    __aicore__ inline KernelConvTranspose3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t d, uint32_t h, uint32_t in_w,
                                uint32_t k,
                                uint32_t do_, uint32_t ho, uint32_t wo,
                                uint32_t totalLength)
    {
        xPtr = (__gm__ DTYPE_X*)x;
        wPtr = (__gm__ DTYPE_Y*)w;
        yPtr = (__gm__ DTYPE_Z*)y;

        this->N = n; this->Cin = cin; this->Cout = cout;
        this->D = d; this->H = h; this->W = in_w;
        this->K = k;
        this->Do = do_; this->Ho = ho; this->Wo = wo;
        this->Total = totalLength;

        // Define workload range for this block
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t baseLen = totalLength / blockNum;
        uint32_t remain = totalLength % blockNum;
        // Distribute remainder to last block
        start = baseLen * blockIdx;
        end = start + baseLen;
        if (blockIdx == blockNum - 1) {
            end += remain;
        }
    }

    __aicore__ inline void Process()
    {
        // Initialize output slice to zero before accumulation
        for (uint32_t idx = start; idx < end; ++idx) {
            yPtr[idx] = static_cast<DTYPE_Z>(0);
        }

        // For each output element in this block's range, compute accumulation
        for (uint32_t idx = start; idx < end; ++idx) {
            // Decode linear index into (n, co, oz, oy, ox)
            uint32_t tmp = idx;

            uint32_t ox = tmp % Wo; tmp /= Wo;
            uint32_t oy = tmp % Ho; tmp /= Ho;
            uint32_t oz = tmp % Do; tmp /= Do;
            uint32_t co = tmp % Cout; tmp /= Cout;
            uint32_t n  = tmp; // 0..N-1

            float acc = 0.f;

            // Iterate over kernel and input channels
            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t kz = 0; kz < K; ++kz) {
                    int32_t iz = (int32_t)oz - (int32_t)kz;
                    if (iz < 0 || iz >= (int32_t)D) continue;
                    for (uint32_t ky = 0; ky < K; ++ky) {
                        int32_t iy = (int32_t)oy - (int32_t)ky;
                        if (iy < 0 || iy >= (int32_t)H) continue;
                        for (uint32_t kx = 0; kx < K; ++kx) {
                            int32_t ix = (int32_t)ox - (int32_t)kx;
                            if (ix < 0 || ix >= (int32_t)W) continue;

                            // x offset: [n, ci, iz, iy, ix]
                            uint64_t xOff = (((uint64_t)n * Cin + ci) * D + (uint32_t)iz);
                            xOff = (xOff * H + (uint32_t)iy) * W + (uint32_t)ix;

                            // w offset: [ci, co, kz, ky, kx] with shape [Cin, Cout, K, K, K]
                            uint64_t wOff = (((uint64_t)ci * Cout + co) * K + kz);
                            wOff = (wOff * K + ky) * K + kx;

                            acc += static_cast<float>(xPtr[xOff]) * static_cast<float>(wPtr[wOff]);
                        }
                    }
                }
            }

            yPtr[idx] = static_cast<DTYPE_Z>(acc);
        }
    }

private:
    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_Y* wPtr;
    __gm__ DTYPE_Z* yPtr;

    uint32_t N, Cin, Cout;
    uint32_t D, H, W;
    uint32_t K;
    uint32_t Do, Ho, Wo;
    uint32_t Total;

    uint32_t start;
    uint32_t end;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.k,
            tiling_data.do_, tiling_data.ho, tiling_data.wo,
            tiling_data.totalLength);
    op.Process();
}

at::Tensor conv_transposed_3d_asymmetric_input_square_kernel_custom_impl_npu(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, Cin, D, H, W]
    // w: [Cin, Cout, K, K, K]
    TORCH_CHECK(x.dim() == 5, "x must be 5D NCDHW");
    TORCH_CHECK(w.dim() == 5, "w must be 5D [Cin, Cout, K, K, K]");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 supported");
    TORCH_CHECK(w.scalar_type() == at::kFloat, "only float32 supported");
    TORCH_CHECK(x.size(1) == w.size(0), "Cin mismatch between x and w");
    TORCH_CHECK(w.size(2) == w.size(3) && w.size(3) == w.size(4), "Kernel must be square KxKxK");

    auto N = x.size(0);
    auto Cin = x.size(1);
    auto D = x.size(2);
    auto H = x.size(3);
    auto WW = x.size(4);

    auto Cout = w.size(1);
    auto K = w.size(2);

    // stride=1, padding=0, dilation=1
    auto Do = D + K - 1;
    auto Ho = H + K - 1;
    auto Wo = WW + K - 1;

    at::Tensor y = at::empty({N, Cout, Do, Ho, Wo}, x.options());

    EXEC_NPU_CMD(aclnnConvTransposed3dAsymmetricInputSquareKernelCustom, x, w, y);
    return y;
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom ConvTranspose3d using an AscendC kernel (stride=1, padding=0, dilation=1, groups=1, bias=False).
    Weight layout: [in_channels, out_channels, K, K, K]
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        # Weight matches PyTorch ConvTranspose3d layout
        w = torch.empty(in_channels, out_channels, kernel_size, kernel_size, kernel_size)
        torch.nn.init.kaiming_uniform_(w, a=5**0.5)
        self.weight = torch.nn.Parameter(w)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x: [N, Cin, D, H, W]
        return custom_ops_lib.conv_transposed_3d_asymmetric_input_square_kernel_custom(x, self.weight)
