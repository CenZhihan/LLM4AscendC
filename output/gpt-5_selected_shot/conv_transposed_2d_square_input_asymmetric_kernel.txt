kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv_transposed_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    // Read tiling parameters
    const uint32_t n    = tiling_data.n;
    const uint32_t cin  = tiling_data.cin;
    const uint32_t cout = tiling_data.cout;
    const uint32_t H    = tiling_data.h;
    const uint32_t W    = tiling_data.w;
    const uint32_t kH   = tiling_data.kh;
    const uint32_t kW   = tiling_data.kw;
    const uint32_t outH = tiling_data.outH;
    const uint32_t outW = tiling_data.outW;

    __gm__ DTYPE_X* xg = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_Y* wg = reinterpret_cast<__gm__ DTYPE_Y*>(w);
    __gm__ DTYPE_Z* yg = reinterpret_cast<__gm__ DTYPE_Z*>(y);

    const uint32_t blocks = AscendC::GetBlockNum();
    const uint32_t block_id = AscendC::GetBlockIdx();

    const uint32_t total_groups = n * cout; // each (n, oc) pair is independent
    const uint32_t chunk = (total_groups + blocks - 1) / blocks;
    const uint32_t start = block_id * chunk;
    const uint32_t end = (start + chunk > total_groups) ? total_groups : (start + chunk);

    // Strides for indexing
    const uint64_t x_stride_n = static_cast<uint64_t>(cin) * H * W;
    const uint64_t x_stride_c = static_cast<uint64_t>(H) * W;
    const uint64_t x_stride_h = static_cast<uint64_t>(W);

    const uint64_t w_stride_ic = static_cast<uint64_t>(cout) * kH * kW;
    const uint64_t w_stride_oc = static_cast<uint64_t>(kH) * kW;
    const uint64_t w_stride_kh = static_cast<uint64_t>(kW);

    const uint64_t y_stride_n = static_cast<uint64_t>(cout) * outH * outW;
    const uint64_t y_stride_c = static_cast<uint64_t>(outH) * outW;
    const uint64_t y_stride_h = static_cast<uint64_t>(outW);

    for (uint32_t idx = start; idx < end; ++idx) {
        uint32_t n_idx = idx / cout;
        uint32_t oc = idx % cout;

        // zero initialize output slice for (n_idx, oc)
        uint64_t y_base = static_cast<uint64_t>(n_idx) * y_stride_n + static_cast<uint64_t>(oc) * y_stride_c;
        for (uint32_t oh = 0; oh < outH; ++oh) {
            uint64_t row_off = y_base + static_cast<uint64_t>(oh) * y_stride_h;
            for (uint32_t ow = 0; ow < outW; ++ow) {
                yg[row_off + ow] = (DTYPE_Z)0;
            }
        }

        // accumulate contributions
        for (uint32_t ic = 0; ic < cin; ++ic) {
            uint64_t x_base = static_cast<uint64_t>(n_idx) * x_stride_n + static_cast<uint64_t>(ic) * x_stride_c;
            uint64_t w_base = static_cast<uint64_t>(ic) * w_stride_ic + static_cast<uint64_t>(oc) * w_stride_oc;

            for (uint32_t ih = 0; ih < H; ++ih) {
                uint64_t x_row = x_base + static_cast<uint64_t>(ih) * x_stride_h;
                for (uint32_t iw = 0; iw < W; ++iw) {
                    DTYPE_X xv = xg[x_row + iw];

                    // scatter-add kernel window
                    for (uint32_t kh = 0; kh < kH; ++kh) {
                        uint32_t oh = ih + kh;
                        uint64_t y_row_off = y_base + static_cast<uint64_t>(oh) * y_stride_h;
                        uint64_t w_row = w_base + static_cast<uint64_t>(kh) * w_stride_kh;

                        for (uint32_t kw = 0; kw < kW; ++kw) {
                            uint32_t ow = iw + kw;
                            DTYPE_Y wv = wg[w_row + kw];
                            yg[y_row_off + ow] = yg[y_row_off + ow] + (DTYPE_Z)(xv * wv);
                        }
                    }
                }
            }
        }
    }
}
