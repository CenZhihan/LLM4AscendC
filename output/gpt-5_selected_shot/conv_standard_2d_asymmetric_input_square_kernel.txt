kernel_src="""
#include "kernel_operator.h"

class KernelConv2d {
public:
    __aicore__ inline KernelConv2d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t hin, uint32_t win,
                                uint32_t cout, uint32_t k, uint32_t hout, uint32_t wout,
                                uint64_t totalOut)
    {
        this->n = n;
        this->cin = cin;
        this->hin = hin;
        this->win = win;
        this->cout = cout;
        this->k = k;
        this->hout = hout;
        this->wout = wout;
        this->totalOut = totalOut;

        xPtr = reinterpret_cast<__gm__ float*>(x);
        wPtr = reinterpret_cast<__gm__ float*>(w);
        yPtr = reinterpret_cast<__gm__ float*>(y);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint64_t blkLen = (totalOut + blockNum - 1) / blockNum;
        uint64_t start = (uint64_t)blockIdx * blkLen;
        uint64_t end = start + blkLen;
        if (end > totalOut) end = totalOut;

        for (uint64_t outIndex = start; outIndex < end; ++outIndex) {
            // Decode y index: [n, cout, hout, wout] with row-major
            uint64_t t = outIndex;
            uint32_t wo = (uint32_t)(t % wout); t /= wout;
            uint32_t ho = (uint32_t)(t % hout); t /= hout;
            uint32_t co = (uint32_t)(t % cout); t /= cout;
            uint32_t bn = (uint32_t)t;

            float acc = 0.f;
            // Convolution: stride=1, pad=0, dilation=1
            for (uint32_t ci = 0; ci < cin; ++ci) {
                for (uint32_t ky = 0; ky < k; ++ky) {
                    uint32_t in_h = ho + ky;
                    for (uint32_t kx = 0; kx < k; ++kx) {
                        uint32_t in_w = wo + kx;

                        uint64_t x_index = ((((uint64_t)bn * cin + ci) * hin + in_h) * win + in_w);
                        uint64_t w_index = ((((uint64_t)co * cin + ci) * k + ky) * k + kx);

                        float xv = xPtr[x_index];
                        float wv = wPtr[w_index];
                        acc += xv * wv;
                    }
                }
            }
            yPtr[outIndex] = acc;
        }
    }

private:
    __gm__ float* xPtr;
    __gm__ float* wPtr;
    __gm__ float* yPtr;

    uint32_t n, cin, hin, win;
    uint32_t cout, k, hout, wout;
    uint64_t totalOut;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2d op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.hin, tiling_data.win,
            tiling_data.cout, tiling_data.k, tiling_data.hout, tiling_data.wout,
            tiling_data.totalOut);
    op.Process();
}

static inline std::vector<int64_t> infer_conv2d_out_shape(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, Cin, H, W]; w: [Cout, Cin, K, K]
    auto N  = x.size(0);
    auto H  = x.size(2);
    auto W  = x.size(3);
    auto Cout = w.size(0);
    auto K  = w.size(2);

    // stride=1, padding=0, dilation=1
    int64_t Hout = H - K + 1;
    int64_t Wout = W - K + 1;
    return {N, Cout, Hout, Wout};
}
