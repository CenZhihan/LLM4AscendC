kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelInceptionConcat {
public:
    __aicore__ inline KernelInceptionConcat() {}

    __aicore__ inline void Init(GM_ADDR in1, GM_ADDR in2, GM_ADDR in3, GM_ADDR in4, GM_ADDR out,
                                uint32_t n, uint32_t c1, uint32_t c2, uint32_t c3, uint32_t c4,
                                uint32_t h, uint32_t w)
    {
        this->n = n;
        this->c1 = c1;
        this->c2 = c2;
        this->c3 = c3;
        this->c4 = c4;
        this->h = h;
        this->w = w;
        this->p = h * w;

        in1Gm.SetGlobalBuffer((__gm__ float*)in1, (uint64_t)n * c1 * this->p);
        in2Gm.SetGlobalBuffer((__gm__ float*)in2, (uint64_t)n * c2 * this->p);
        in3Gm.SetGlobalBuffer((__gm__ float*)in3, (uint64_t)n * c3 * this->p);
        in4Gm.SetGlobalBuffer((__gm__ float*)in4, (uint64_t)n * c4 * this->p);
        outGm.SetGlobalBuffer((__gm__ float*)out, (uint64_t)n * (c1 + c2 + c3 + c4) * this->p);

        // tile along HW plane
        this->tileLen = 4096;
        if (this->tileLen > this->p) {
            this->tileLen = this->p;
        }
        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLen * sizeof(float));
    }

    __aicore__ inline void CopySegment(AscendC::GlobalTensor<float>& src, uint32_t srcC, uint32_t dstCOffset, uint32_t batchIdx)
    {
        uint32_t baseSrc = batchIdx * srcC * p;
        uint32_t baseDst = batchIdx * (c1 + c2 + c3 + c4) * p + dstCOffset * p;

        for (uint32_t cc = 0; cc < srcC; ++cc) {
            uint32_t srcOffset = baseSrc + cc * p;
            uint32_t dstOffset = baseDst + cc * p;
            uint32_t moved = 0;
            while (moved < p) {
                uint32_t len = (p - moved) < tileLen ? (p - moved) : tileLen;
                AscendC::LocalTensor<float> buf = inQueue.AllocTensor<float>();
                AscendC::DataCopy(buf, src[srcOffset + moved], len);
                AscendC::DataCopy(outGm[dstOffset + moved], buf, len);
                inQueue.FreeTensor(buf);
                moved += len;
            }
        }
    }

    __aicore__ inline void Process()
    {
        // Single block handles all batches
        for (uint32_t b = 0; b < n; ++b) {
            uint32_t cOfs = 0;
            CopySegment(in1Gm, c1, cOfs, b); cOfs += c1;
            CopySegment(in2Gm, c2, cOfs, b); cOfs += c2;
            CopySegment(in3Gm, c3, cOfs, b); cOfs += c3;
            CopySegment(in4Gm, c4, cOfs, b);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;

    AscendC::GlobalTensor<float> in1Gm;
    AscendC::GlobalTensor<float> in2Gm;
    AscendC::GlobalTensor<float> in3Gm;
    AscendC::GlobalTensor<float> in4Gm;
    AscendC::GlobalTensor<float> outGm;

    uint32_t n, c1, c2, c3, c4, h, w, p;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void googlenet_inception_module_custom(GM_ADDR branch1x1,
                                                                         GM_ADDR branch3x3,
                                                                         GM_ADDR branch5x5,
                                                                         GM_ADDR branch_pool,
                                                                         GM_ADDR out,
                                                                         GM_ADDR workspace,
                                                                         GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelInceptionConcat op;
    op.Init(branch1x1, branch3x3, branch5x5, branch_pool, out,
            tiling_data.n, tiling_data.c1, tiling_data.c2, tiling_data.c3, tiling_data.c4,
            tiling_data.h, tiling_data.w);
    op.Process();
}

model_src='''
import torch
import torch_npu
import torch.nn as nn
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_1x1, reduce_3x3, out_3x3, reduce_5x5, out_5x5, pool_proj):
        super(ModelNew, self).__init__()
        # 1x1 convolution branch
        self.branch1x1 = nn.Conv2d(in_channels, out_1x1, kernel_size=1)

        # 3x3 convolution branch
        self.branch3x3 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_3x3, kernel_size=1),
            nn.Conv2d(reduce_3x3, out_3x3, kernel_size=3, padding=1)
        )

        # 5x5 convolution branch
        self.branch5x5 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_5x5, kernel_size=1),
            nn.Conv2d(reduce_5x5, out_5x5, kernel_size=5, padding=2)
        )

        # Max pooling branch
        self.branch_pool = nn.Sequential(
            nn.MaxPool2d(kernel_size=3, stride=1, padding=1),
            nn.Conv2d(in_channels, pool_proj, kernel_size=1)
        )

    def forward(self, x):
        branch1x1 = self.branch1x1(x)
        branch3x3 = self.branch3x3(x)
        branch5x5 = self.branch5x5(x)
        branch_pool = self.branch_pool(x)
        # Replace torch.cat with custom AscendC kernel op
        return custom_ops_lib.googlenet_inception_module_custom(branch1x1, branch3x3, branch5x5, branch_pool)
'''
