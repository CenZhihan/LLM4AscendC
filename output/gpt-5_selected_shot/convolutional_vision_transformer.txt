namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* cls_shape = context->GetInputShape(1);

    gert::Shape* z_shape = context->GetOutputShape(0);

    // Expect x: (B, E) or (B, 1, E) flattens to (B, E)
    // cls: (1, 1, E) or (1, E) -> use last dim as E
    auto x_origin = x_shape->GetOriginShape();
    auto cls_origin = cls_shape->GetOriginShape();

    int64_t x_dim_num = x_origin.GetDimNum();
    int64_t cls_dim_num = cls_origin.GetDimNum();

    int64_t B = (x_dim_num >= 1) ? x_origin.GetDim(0) : 1;
    int64_t E = (x_dim_num >= 2) ? x_origin.GetDim(x_dim_num - 1) : x_origin.GetShapeSize();
    int64_t E_cls = (cls_dim_num >= 1) ? cls_origin.GetDim(cls_dim_num - 1) : cls_origin.GetShapeSize();

    if (E_cls > 0) {
        E = E_cls; // prefer cls last dim if provided
    }

    // Set output to (B, 2, E)
    z_shape->SetDimNum(3);
    z_shape->SetDim(0, B);
    z_shape->SetDim(1, 2);
    z_shape->SetDim(2, E);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // Using a single on-chip buffer

class KernelCViTConcat {
public:
    __aicore__ inline KernelCViTConcat() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR cls, GM_ADDR z,
                                uint32_t batch, uint32_t embedDim)
    {
        this->batch = batch;
        this->embedDim = embedDim;

        // x: (B, E)
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint32_t>(batch * embedDim));
        // cls: (1, 1, E) (or (1, E)) treated as E
        clsGm.SetGlobalBuffer((__gm__ DTYPE_Y *)cls, static_cast<uint32_t>(embedDim));
        // z: (B, 2, E)
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, static_cast<uint32_t>(batch * 2 * embedDim));

        pipe.InitBuffer(bufQ, BUFFER_NUM, this->embedDim * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        // Load cls token once to local buffer
        AscendC::LocalTensor<DTYPE_Y> clsLocal = bufQ.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(clsLocal, clsGm[0], this->embedDim);
        bufQ.EnQue(clsLocal);
        AscendC::LocalTensor<DTYPE_Y> clsCached = bufQ.DeQue<DTYPE_Y>();

        for (uint32_t b = 0; b < this->batch; ++b) {
            // Load x[b, :]
            AscendC::LocalTensor<DTYPE_X> xLocal = bufQ.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[b * this->embedDim], this->embedDim);

            // Write z[b, 0, :] = cls
            AscendC::DataCopy(zGm[b * 2 * this->embedDim], clsCached, this->embedDim);
            // Write z[b, 1, :] = x[b, :]
            AscendC::DataCopy(zGm[b * 2 * this->embedDim + this->embedDim], xLocal, this->embedDim);

            bufQ.FreeTensor(xLocal);
        }

        bufQ.FreeTensor(clsCached);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bufQ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> clsGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    uint32_t batch;
    uint32_t embedDim;
};

extern "C" __global__ __aicore__ void convolutional_vision_transformer_custom(GM_ADDR x, GM_ADDR cls, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCViTConcat op;
    op.Init(x, cls, z, tiling_data.batch, tiling_data.embedDim);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_classes, embed_dim=512, num_heads=8, num_layers=6, 
                 mlp_ratio=4.0, patch_size=4, in_channels=3, image_size=32):
        super(ModelNew, self).__init__()

        self.patch_size = patch_size
        self.image_size = image_size
        self.embed_dim = embed_dim

        self.conv1 = torch.nn.Conv2d(in_channels, embed_dim, kernel_size=patch_size, stride=patch_size)
        num_patches = (image_size // patch_size) ** 2
        self.linear_proj = torch.nn.Linear(embed_dim * num_patches, embed_dim)

        self.transformer_layers = torch.nn.ModuleList([
            torch.nn.TransformerEncoderLayer(
                d_model=embed_dim,
                nhead=num_heads,
                dim_feedforward=int(embed_dim * mlp_ratio),
                dropout=0.0,
                batch_first=True
            ) for _ in range(num_layers)
        ])

        self.cls_token = torch.nn.Parameter(torch.zeros(1, 1, embed_dim))
        self.fc_out = torch.nn.Linear(embed_dim, num_classes)

    def forward(self, x):
        B = x.size(0)
        x = self.conv1(x)                  # (B, embed_dim, H/patch_size, W/patch_size)
        x = x.flatten(start_dim=1)         # (B, embed_dim * num_patches)
        x = self.linear_proj(x)            # (B, embed_dim)

        # Replace expand + unsqueeze + cat with custom AscendC kernel
        tokens = custom_ops_lib.convolutional_vision_transformer_custom(x, self.cls_token)  # (B, 2, embed_dim)

        for layer in self.transformer_layers:
            tokens = layer(tokens)

        return self.fc_out(tokens[:, 0])
'''
