#include "grid_sample_affine_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GridSampleAffineCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const auto& os = x_shape->GetOriginShape();

    uint32_t n = (os.GetDimNum() > 0) ? os.GetDim(0) : 1;
    uint32_t c = (os.GetDimNum() > 1) ? os.GetDim(1) : 1;
    uint32_t h = (os.GetDimNum() > 2) ? os.GetDim(2) : 1;
    uint32_t w = (os.GetDimNum() > 3) ? os.GetDim(3) : 1;

    uint32_t totalLength = n * c * h * w;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {

    const gert::Shape* x_shape = context->GetInputShape(0);

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

extern "C" __global__ __aicore__ void grid_sample_affine_custom(GM_ADDR x, GM_ADDR theta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.n;
    const uint32_t C = tiling_data.c;
    const uint32_t H = tiling_data.h;
    const uint32_t W = tiling_data.w;
    const uint32_t totalLength = tiling_data.totalLength;

    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t blockLength = totalLength / blockNum;
    const uint32_t start = blockIdx * blockLength;

    __gm__ DTYPE_X* xPtr = (__gm__ DTYPE_X*)x;
    __gm__ DTYPE_Y* thPtr = (__gm__ DTYPE_Y*)theta;
    __gm__ DTYPE_Z* yPtr = (__gm__ DTYPE_Z*)y;

    for (uint32_t i = 0; i < blockLength; ++i) {
        uint32_t idx = start + i;

        uint32_t tmp = idx;
        uint32_t wIdx = tmp % W; tmp /= W;
        uint32_t hIdx = tmp % H; tmp /= H;
        uint32_t cIdx = tmp % C; tmp /= C;
        uint32_t nIdx = tmp;

        float gx_t = ((2.0f * static_cast<float>(wIdx) + 1.0f) - static_cast<float>(W)) / static_cast<float>(W);
        float gy_t = ((2.0f * static_cast<float>(hIdx) + 1.0f) - static_cast<float>(H)) / static_cast<float>(H);

        uint32_t tBase = nIdx * 6;
        float t00 = static_cast<float>(thPtr[tBase + 0]);
        float t01 = static_cast<float>(thPtr[tBase + 1]);
        float t02 = static_cast<float>(thPtr[tBase + 2]);
        float t10 = static_cast<float>(thPtr[tBase + 3]);
        float t11 = static_cast<float>(thPtr[tBase + 4]);
        float t12 = static_cast<float>(thPtr[tBase + 5]);

        float sx = t00 * gx_t + t01 * gy_t + t02;
        float sy = t10 * gx_t + t11 * gy_t + t12;

        float x_src = ((sx + 1.0f) * static_cast<float>(W) - 1.0f) * 0.5f;
        float y_src = ((sy + 1.0f) * static_cast<float>(H) - 1.0f) * 0.5f;

        int x0 = static_cast<int>(floorf(x_src));
        int y0 = static_cast<int>(floorf(y_src));
        int x1 = x0 + 1;
        int y1 = y0 + 1;

        float dx = x_src - static_cast<float>(x0);
        float dy = y_src - static_cast<float>(y0);

        float w00 = (1.0f - dx) * (1.0f - dy);
        float w01 = dx * (1.0f - dy);
        float w10 = (1.0f - dx) * dy;
        float w11 = dx * dy;

        auto in_bounds = [](int yy, int xx, int H, int W)->bool {
            return (yy >= 0 && yy < H && xx >= 0 && xx < W);
        };

        float out_val = 0.0f;

        if (in_bounds(y0, x0, (int)H, (int)W)) {
            uint64_t off = (((uint64_t)nIdx * C + cIdx) * H + (uint32_t)y0) * W + (uint32_t)x0;
            out_val += w00 * static_cast<float>(xPtr[off]);
        }
        if (in_bounds(y0, x1, (int)H, (int)W)) {
            uint64_t off = (((uint64_t)nIdx * C + cIdx) * H + (uint32_t)y0) * W + (uint32_t)x1;
            out_val += w01 * static_cast<float>(xPtr[off]);
        }
        if (in_bounds(y1, x0, (int)H, (int)W)) {
            uint64_t off = (((uint64_t)nIdx * C + cIdx) * H + (uint32_t)y1) * W + (uint32_t)x0;
            out_val += w10 * static_cast<float>(xPtr[off]);
        }
        if (in_bounds(y1, x1, (int)H, (int)W)) {
            uint64_t off = (((uint64_t)nIdx * C + cIdx) * H + (uint32_t)y1) * W + (uint32_t)x1;
            out_val += w11 * static_cast<float>(xPtr[off]);
        }

        yPtr[idx] = static_cast<DTYPE_Z>(out_val);
    }
}
