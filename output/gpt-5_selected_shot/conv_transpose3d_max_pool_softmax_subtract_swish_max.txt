kernel_src="""
#include "kernel_operator.h"

// This kernel expects input in NDHWC layout for contiguous channel access.
// It computes, per spatial position (N,D,H,W):
// y = max_c( swish( softmax(x[..., c]) - sub[c] ) )
// where swish(t) = sigmoid(t) * t

constexpr int32_t BUFFER_NUM = 2;

class KernelCTSMSM {
public:
    __aicore__ inline KernelCTSMSM() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR sub, GM_ADDR y, uint32_t positions, uint32_t channels)
    {
        this->channels = channels;
        this->blockPositions = positions / AscendC::GetBlockNum();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)this->blockPositions * this->channels * AscendC::GetBlockIdx(),
                            this->blockPositions * this->channels);
        subGm.SetGlobalBuffer((__gm__ DTYPE_Y *)sub, this->channels);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + (uint64_t)this->blockPositions * AscendC::GetBlockIdx(),
                            this->blockPositions);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->channels * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueS, BUFFER_NUM, this->channels * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->channels * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t pos = 0; pos < this->blockPositions; ++pos) {
            CopyIn(pos);
            Compute(pos);
            CopyOut(pos);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t pos)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> sLocal = inQueueS.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[pos * this->channels], this->channels);
        AscendC::DataCopy(sLocal, subGm[0], this->channels);
        inQueueX.EnQue(xLocal);
        inQueueS.EnQue(sLocal);
    }

    __aicore__ inline void Compute(uint32_t /*pos*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> sLocal = inQueueS.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Softmax across channels
        // 1) shift by max to improve numeric stability
        DTYPE_Z maxVal = AscendC::ReduceMax(xLocal, this->channels);
        AscendC::Sub(zLocal, xLocal, maxVal, this->channels);
        AscendC::Exp(zLocal, zLocal, this->channels);
        DTYPE_Z sumVal = AscendC::ReduceSum(zLocal, this->channels);
        AscendC::Div(zLocal, zLocal, sumVal, this->channels);

        // Subtract channel-wise vector
        AscendC::Sub(zLocal, zLocal, sLocal, this->channels);

        // Swish: sigmoid(z) * z
        AscendC::LocalTensor<DTYPE_Z> sigLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::Sigmoid(sigLocal, zLocal, this->channels);
        AscendC::Mul(zLocal, zLocal, sigLocal, this->channels);
        outQueueY.FreeTensor(sigLocal);

        // Max across channels -> scalar result; store at index 0
        DTYPE_Z outScalar = AscendC::ReduceMax(zLocal, this->channels);
        // Fill first element with outScalar; other values ignored
        AscendC::Fill(zLocal, outScalar, 1);

        outQueueY.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueS.FreeTensor(sLocal);
    }

    __aicore__ inline void CopyOut(uint32_t pos)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[pos], zLocal, 1);
        outQueueY.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueS;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> subGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t channels;
    uint32_t blockPositions;
};

extern "C" __global__ __aicore__ void conv_transpose3d_max_pool_softmax_subtract_swish_max_custom(
    GM_ADDR x, GM_ADDR sub, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCTSMSM op;
    op.Init(x, sub, y, tiling_data.positions, tiling_data.channels);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model:
      - ConvTranspose3d (PyTorch)
      - MaxPool3d (PyTorch)
      - Custom AscendC op: Softmax -> Subtract -> Swish -> Max over channels
        The custom op expects NDHWC input for efficient channel-wise vectorization.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, pool_kernel_size, pool_stride, pool_padding):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.max_pool = torch.nn.MaxPool3d(kernel_size=pool_kernel_size, stride=pool_stride, padding=pool_padding)
        self.subtract = torch.nn.Parameter(torch.randn(out_channels))

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.max_pool(x)
        # Convert to NDHWC for contiguous channel-last layout
        x = x.permute(0, 2, 3, 4, 1).contiguous()
        # Custom fused op: softmax (over C) -> subtract (channel-wise) -> swish -> max (over C)
        x = custom_ops_lib.conv_transpose3d_max_pool_softmax_subtract_swish_max_custom(x, self.subtract)
        return x
