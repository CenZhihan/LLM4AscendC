#include "mean_reduction_over_a_dimension_custom_tiling.h"

namespace {
inline uint64_t Prod(const std::vector<int64_t> &dims, size_t start, size_t end) {
    uint64_t p = 1;
    for (size_t i = start; i < end; ++i) {
        p *= static_cast<uint64_t>(dims[i] < 0 ? 0 : dims[i]);
    }
    return p;
}
} // namespace

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TilingData tiling;

    // Get input shape dims
    auto xShape = context->GetInputShape(0)->GetOriginShape();
    std::vector<int64_t> dims;
    dims.reserve(xShape.GetDimNum());
    for (int i = 0; i < xShape.GetDimNum(); ++i) {
        dims.emplace_back(xShape.GetDim(i));
    }

    // Get dim attribute
    int64_t rank = static_cast<int64_t>(dims.size());
    int64_t dimAttr = 0;
    (void)context->GetAttr("dim", dimAttr);
    if (dimAttr < 0) dimAttr += rank;
    if (dimAttr < 0) dimAttr = 0;
    if (dimAttr >= rank) dimAttr = rank - 1;

    uint64_t outer = Prod(dims, 0, static_cast<size_t>(dimAttr));
    uint64_t reduce = (dimAttr >= 0 && dimAttr < rank) ? static_cast<uint64_t>(dims[dimAttr]) : 1ULL;
    uint64_t inner = Prod(dims, static_cast<size_t>(dimAttr + 1), dims.size());

    uint64_t totalLength = outer * reduce * inner;
    uint64_t outSize = outer * inner;

    context->SetBlockDim(1);
    tiling.set_outer(static_cast<uint32_t>(outer));
    tiling.set_reduce(static_cast<uint32_t>(reduce));
    tiling.set_inner(static_cast<uint32_t>(inner));
    tiling.set_outSize(static_cast<uint32_t>(outSize));
    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_dim(static_cast<int32_t>(dimAttr));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static graphStatus InferShape(gert::InferShapeContext *context)
{
    auto xShape = context->GetInputShape(0)->GetOriginShape();
    int64_t rank = xShape.GetDimNum();
    int64_t dimAttr = 0;
    (void)context->GetAttr("dim", dimAttr);
    if (dimAttr < 0) dimAttr += rank;
    if (dimAttr < 0) dimAttr = 0;
    if (dimAttr >= rank) dimAttr = rank - 1;

    std::vector<int64_t> outDims;
    outDims.reserve(rank > 0 ? static_cast<size_t>(rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dimAttr) continue;
        outDims.emplace_back(xShape.GetDim(i));
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    if (outDims.empty()) {
        // If all dims reduced (rank==1), produce scalar {1}
        *y_shape = {1};
    } else {
        y_shape->SetDimNum(outDims.size());
        for (size_t i = 0; i < outDims.size(); ++i) {
            y_shape->SetDim(i, outDims[i]);
        }
    }
    return GRAPH_SUCCESS;
}

static graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

class KernelMeanReduce {
public:
    __aicore__ inline KernelMeanReduce() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t outer, uint32_t reduce, uint32_t inner, uint32_t outSize, uint32_t totalLength)
    {
        this->outer = outer;
        this->reduce = reduce;
        this->inner = inner;
        this->outSize = outSize;
        this->totalLength = totalLength;

        xGm.SetGlobalBuffer((__gm__ float *)x, totalLength);
        zGm.SetGlobalBuffer((__gm__ float *)z, outSize);

        // Allocate one big local buffer for input and output
        pipe.InitBuffer(inQueueX, 1, totalLength * sizeof(float));
        pipe.InitBuffer(outQueueZ, 1, outSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm, totalLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

        // Initialize output to 0
        for (uint32_t i = 0; i < outSize; ++i) {
            zLocal.SetValue(i, 0.0f);
        }

        // Accumulate sums along reduction axis
        // Treat x as [outer, reduce, inner], output z as [outer, inner]
        for (uint32_t o = 0; o < outer; ++o) {
            for (uint32_t i = 0; i < inner; ++i) {
                float acc = 0.0f;
                uint32_t base = o * reduce * inner + i;
                for (uint32_t r = 0; r < reduce; ++r) {
                    acc += xLocal.GetValue(base + r * inner);
                }
                // mean
                zLocal.SetValue(o * inner + i, acc / static_cast<float>(reduce));
            }
        }

        outQueueZ.EnQue<float>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm, zLocal, outSize);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t outer;
    uint32_t reduce;
    uint32_t inner;
    uint32_t outSize;
    uint32_t totalLength;
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReduce op;
    op.Init(x, z,
            tiling_data.outer,
            tiling_data.reduce,
            tiling_data.inner,
            tiling_data.outSize,
            tiling_data.totalLength);
    op.Process();
}
"""

at::Tensor mean_reduction_over_a_dimension_impl_npu(const at::Tensor& x, int64_t dim) {
    auto sizes = x.sizes().vec();
    int64_t rank = static_cast<int64_t>(sizes.size());
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    std::vector<int64_t> out_sizes;
    out_sizes.reserve(rank > 0 ? static_cast<size_t>(rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out_sizes.push_back(sizes[i]);
    }
    if (out_sizes.empty()) {
        out_sizes.push_back(1); // scalar-like tensor
    }
    at::Tensor result = at::empty(out_sizes, x.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, x, result, dim);
    return result;
}

    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_over_a_dimension_impl_npu);
