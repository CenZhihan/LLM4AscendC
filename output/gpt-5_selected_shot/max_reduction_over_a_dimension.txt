host_operator_src="""
#include "max_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#define REDUCE_MAX_TILING_0 1
#define REDUCE_MAX_TILING_1 2
#define REDUCE_MAX_TILING_2 3

namespace optiling {
constexpr uint32_t BLOCK_DIM = 1;
constexpr uint32_t ONE_REPEAT_LEN = 256;  // bytes
constexpr uint32_t ONE_BLOCK_LEN = 32;    // repeats
constexpr uint32_t FLOATS_PER_REPEAT = ONE_REPEAT_LEN / sizeof(float);
constexpr uint32_t FLOAT_THRESHOLD0 = FLOATS_PER_REPEAT;                         // 256B
constexpr uint32_t FLOAT_THRESHOLD1 = FLOATS_PER_REPEAT * (ONE_BLOCK_LEN / 4);   // heuristic
constexpr uint32_t FLOAT_THRESHOLD2 = FLOATS_PER_REPEAT * ONE_BLOCK_LEN;         // ~2KB

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TilingData tiling;

    const auto &inShape = context->GetInputShape(0)->GetOriginShape();
    std::vector<int64_t> dims = inShape.GetDims();
    if (dims.empty()) {
        return ge::GRAPH_FAILED;
    }

    uint32_t totalLength = static_cast<uint32_t>(inShape.GetShapeSize());
    uint32_t innerLen = static_cast<uint32_t>(dims.back());
    uint32_t outerLen = totalLength / innerLen;

    auto inputDtype = context->GetInputTensor(0)->GetDataType();

    if (innerLen <= FLOAT_THRESHOLD0 && inputDtype == ge::DT_FLOAT) {
        context->SetTilingKey(REDUCE_MAX_TILING_0);
    } else if (innerLen <= FLOAT_THRESHOLD1 && inputDtype == ge::DT_FLOAT) {
        context->SetTilingKey(REDUCE_MAX_TILING_1);
    } else if (innerLen <= FLOAT_THRESHOLD2 && inputDtype == ge::DT_FLOAT) {
        context->SetTilingKey(REDUCE_MAX_TILING_2);
    } else {
        // Fallback
        context->SetTilingKey(REDUCE_MAX_TILING_2);
    }

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_innerLen(innerLen);
    tiling.set_outerLen(outerLen);
    tiling.set_outLength(outerLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static graphStatus InferShape(gert::InferShapeContext *context)
{
    const auto inShape = context->GetInputShape(0)->GetOriginShape();
    std::vector<int64_t> inDims = inShape.GetDims();
    if (inDims.empty()) {
        return GRAPH_FAILED;
    }
    inDims.pop_back(); // reduce over the last dimension
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = inDims;
    return GRAPH_SUCCESS;
}

static graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

kernel_src="""
#include "kernel_operator.h"
#define REDUCE_MAX_TILING_0 1
#define REDUCE_MAX_TILING_1 2
#define REDUCE_MAX_TILING_2 3

class KernelReduceMaxLastDim {
static constexpr uint32_t DEFAULT_BLK_STRIDE = 1;
static constexpr uint32_t DEFAULT_REP_STRIDE = 8;
static constexpr uint32_t REP_LEN = 256; // bytes
static constexpr uint32_t BLK_LEN = 32;  // repeats
public:
    __aicore__ inline KernelReduceMaxLastDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t outerLen, uint32_t innerLen, uint32_t outLength)
    {
        this->outerLen = outerLen;
        this->innerLen = innerLen;
        this->totalLength = outerLen * innerLen;
        this->outLength = outLength;

        xGm.SetGlobalBuffer((__gm__ float *)x, totalLength);
        zGm.SetGlobalBuffer((__gm__ float *)z, outLength);

        // Input local buffer allocates entire input for simplicity in this sample
        pipe.InitBuffer(inQueueX, 1, totalLength * sizeof(float));
        // Output per-row maximums
        pipe.InitBuffer(outQueueZ, 1, outLength * sizeof(float));
    }

    __aicore__ inline void Process1()
    {
        CopyIn();
        Compute1();
        CopyOut();
    }
    __aicore__ inline void Process2()
    {
        CopyIn();
        Compute2();
        CopyOut();
    }
    __aicore__ inline void Process3()
    {
        CopyIn();
        Compute3();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm, totalLength);
        inQueueX.EnQue(xLocal);
    }

    // Case 1: innerLen <= 256B; use WholeReduceMax per row
    __aicore__ inline void Compute1()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();
        constexpr int64_t maskLen = REP_LEN / sizeof(float);

        for (uint32_t row = 0; row < outerLen; ++row) {
            // compute max across last dimension for this row
            // Assume WholeReduceMax writes a single max value to the output position
            AscendC::SetVectorMask<float>(0, innerLen);
            AscendC::WholeReduceMax<float>(zLocal[row], xLocal[row * innerLen], maskLen, 1,
                DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
            AscendC::PipeBarrier<PIPE_V>();
        }
        AscendC::SetMaskNorm();
        outQueueZ.EnQue<float>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    // Case 2: innerLen in (256B, ~2KB]; combine BlockReduceMax then WholeReduceMax
    __aicore__ inline void Compute2()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

        pipe.InitBuffer(calcBuf, innerLen * sizeof(float));
        AscendC::LocalTensor<float> tempTensor1 = calcBuf.Get<float>();

        const uint32_t c0Count = BLK_LEN / sizeof(float);
        for (uint32_t row = 0; row < outerLen; ++row) {
            const uint32_t offset = row * innerLen;
            const uint32_t blockNum0 = (innerLen + c0Count - 1) / c0Count;

            AscendC::SetMaskCount();
            AscendC::SetVectorMask<float>(0, innerLen);
            AscendC::BlockReduceMax<float, false>(tempTensor1, xLocal[offset], AscendC::MASK_PLACEHOLDER, 1,
                DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
            AscendC::PipeBarrier<PIPE_V>();

            AscendC::SetVectorMask<float>(0, blockNum0);
            AscendC::WholeReduceMax<float, false>(zLocal[row], tempTensor1, AscendC::MASK_PLACEHOLDER, 1,
                DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
            AscendC::PipeBarrier<PIPE_V>();
            AscendC::SetMaskNorm();
        }

        outQueueZ.EnQue<float>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    // Case 3: innerLen in (~2KB, ~8KB]; two-stage WholeReduceMax
    __aicore__ inline void Compute3()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

        pipe.InitBuffer(calcBuf, innerLen * sizeof(float));
        AscendC::LocalTensor<float> tempTensor1 = calcBuf.Get<float>();

        const uint32_t repeatNum = (innerLen * sizeof(float) + REP_LEN - 1) / REP_LEN;

        for (uint32_t row = 0; row < outerLen; ++row) {
            const uint32_t offset = row * innerLen;

            AscendC::SetMaskCount();
            AscendC::SetVectorMask<float>(0, innerLen);
            AscendC::WholeReduceMax<float, false>(tempTensor1, xLocal[offset], AscendC::MASK_PLACEHOLDER, 1,
                DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
            AscendC::PipeBarrier<PIPE_V>();

            AscendC::SetVectorMask<float>(0, repeatNum);
            AscendC::WholeReduceMax<float, false>(zLocal[row], tempTensor1, AscendC::MASK_PLACEHOLDER, 1,
                DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
            AscendC::PipeBarrier<PIPE_V>();
            AscendC::SetMaskNorm();
        }

        outQueueZ.EnQue<float>(zLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm, zLocal, this->outLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBuf;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t totalLength;
    uint32_t innerLen;
    uint32_t outerLen;
    uint32_t outLength;
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelReduceMaxLastDim op;
    op.Init(x, z, tiling_data.outerLen, tiling_data.innerLen, tiling_data.outLength);
    if (TILING_KEY_IS(REDUCE_MAX_TILING_0)) {
        op.Process1();
    } else if (TILING_KEY_IS(REDUCE_MAX_TILING_1)) {
        op.Process2();
    } else if (TILING_KEY_IS(REDUCE_MAX_TILING_2)) {
        op.Process3();
    }
}

at::Tensor max_reduction_over_a_dimension_impl_npu(const at::Tensor& x_lastdim) {
    // x_lastdim is expected to have the dimension to be reduced as the last dim
    TORCH_CHECK(x_lastdim.dim() >= 1, \\"Input tensor must have at least 1 dimension\\");
    std::vector<int64_t> out_sizes = x_lastdim.sizes().vec();

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: performs Max reduction over a specific dimension using custom AscendC kernel.
    The custom kernel reduces over the last dimension, so we move the target dimension to the last.
    """
    def __init__(self, dim: int):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Move the target dimension to the last, ensure contiguous for kernel consumption
        y = torch.movedim(x, self.dim, -1).contiguous()
        # Custom op reduces over the last dimension
        return custom_ops_lib.max_reduction_over_a_dimension_custom(y)
