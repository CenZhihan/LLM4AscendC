namespace {
inline uint32_t GetDimSafe(const gert::Shape* s, size_t idx, uint32_t default_val = 1) {
    if (s == nullptr) return default_val;
    auto origin = s->GetOriginShape();
    if (idx < origin.GetDimNum()) {
        return static_cast<uint32_t>(origin.GetDim(idx));
    }
    return default_val;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    const gert::Shape* w_shape  = context->GetInputShape(1);
    // Expect shapes: x [N, Cin, H, W], weight [Cout, Cin, Kh, Kw]
    uint32_t N  = GetDimSafe(in_shape, 0, 1);
    uint32_t Cin= GetDimSafe(in_shape, 1, 1);
    uint32_t H  = GetDimSafe(in_shape, 2, 1);
    uint32_t W  = GetDimSafe(in_shape, 3, 1);

    uint32_t Cout= GetDimSafe(w_shape, 0, 1);
    uint32_t CinW= GetDimSafe(w_shape, 1, 1);
    uint32_t Kh  = GetDimSafe(w_shape, 2, 1);
    uint32_t Kw  = GetDimSafe(w_shape, 3, 1);

    // Simple check
    if (CinW != Cin) {
        return ge::GRAPH_FAILED;
    }

    // stride=1, padding=0
    uint32_t Hout = (H >= Kh) ? (H - Kh + 1) : 0;
    uint32_t Wout = (W >= Kw) ? (W - Kw + 1) : 0;

    Conv2dBatchNormScalingCustomTilingData tiling;
    tiling.set_n(N);
    tiling.set_cIn(Cin);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_cOut(Cout);
    tiling.set_kH(Kh);
    tiling.set_kW(Kw);
    tiling.set_hOut(Hout);
    tiling.set_wOut(Wout);
    uint32_t totalOutLen = N * Cout * Hout * Wout;
    tiling.set_totalOutLen(totalOutLen);
    tiling.set_tileNum(TILE_NUM);

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;
    }
    auto x0 = x_shape->GetShape();
    auto w0 = w_shape->GetShape();
    if (x0.size() != 4 || w0.size() != 4) {
        return GRAPH_FAILED;
    }
    int64_t N   = x0[0];
    int64_t Cin = x0[1];
    int64_t H   = x0[2];
    int64_t W   = x0[3];
    int64_t Cout= w0[0];
    int64_t CinW= w0[1];
    int64_t Kh  = w0[2];
    int64_t Kw  = w0[3];
    if (CinW != Cin) {
        return GRAPH_FAILED;
    }
    // stride=1, padding=0
    int64_t Hout = H - Kh + 1;
    int64_t Wout = W - Kw + 1;
    if (Hout < 0 || Wout < 0) {
        return GRAPH_FAILED;

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Hout);
    y_shape->SetDim(3, Wout);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv2d_batch_norm_scaling_custom(
    GM_ADDR x,
    GM_ADDR weight,
    GM_ADDR bias,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N    = tiling_data.n;
    const uint32_t Cin  = tiling_data.cIn;
    const uint32_t H    = tiling_data.h;
    const uint32_t W    = tiling_data.w;
    const uint32_t Cout = tiling_data.cOut;
    const uint32_t Kh   = tiling_data.kH;
    const uint32_t Kw   = tiling_data.kW;
    const uint32_t Hout = tiling_data.hOut;
    const uint32_t Wout = tiling_data.wOut;
    const uint32_t totalOutLen = tiling_data.totalOutLen;

    __gm__ float* x_ptr = (__gm__ float*)x;
    __gm__ float* w_ptr = (__gm__ float*)weight;
    __gm__ float* b_ptr = (__gm__ float*)bias;
    __gm__ float* y_ptr = (__gm__ float*)y;

    // 1D block splitting on output elements
    uint32_t block_num = AscendC::GetBlockNum();
    uint32_t block_idx = AscendC::GetBlockIdx();

    uint32_t base = (totalOutLen / block_num) * block_idx;
    uint32_t len  = (block_idx == block_num - 1) ? (totalOutLen - base) : (totalOutLen / block_num);

    for (uint32_t idx = 0; idx < len; ++idx) {
        uint32_t out_index = base + idx;

        // Map linear idx -> (n, co, ho, wo) in NCHW with contiguous layout (W fastest)
        uint32_t wo = out_index % Wout;
        uint32_t tmp = out_index / Wout;
        uint32_t ho = tmp % Hout;
        tmp /= Hout;
        uint32_t co = tmp % Cout;
        uint32_t n  = tmp / Cout;

        float acc = b_ptr[co];

        // Convolution sum over Cin, Kh, Kw with stride=1, no padding
        for (uint32_t ci = 0; ci < Cin; ++ci) {
            for (uint32_t kh = 0; kh < Kh; ++kh) {
                uint32_t ih = ho + kh;
                for (uint32_t kw = 0; kw < Kw; ++kw) {
                    uint32_t iw = wo + kw;
                    // x index: ((n*Cin + ci)*H + ih)*W + iw
                    uint32_t x_idx = (((n * Cin + ci) * H + ih) * W + iw);
                    // w index: (((co*Cin + ci)*Kh + kh)*Kw + kw)
                    uint32_t w_idx = ((((co * Cin) + ci) * Kh + kh) * Kw + kw);
                    acc += x_ptr[x_idx] * w_ptr[w_idx];
                }
            }
        }

        // y index equals out_index under contiguous NCHW with sizes [N, Cout, Hout, Wout]
        y_ptr[out_index] = acc;
    }
}

    TORCH_CHECK(weight.dim() == 4, \\"weight must be 4D [Cout, Cin, Kh, Kw]\\");
    TORCH_CHECK(bias.dim() == 1, \\"bias must be 1D [Cout]\\");

    auto N   = x.size(0);
    auto Cin = x.size(1);
    auto H   = x.size(2);
    auto W   = x.size(3);
    auto Cout= weight.size(0);
    auto CinW= weight.size(1);
    auto Kh  = weight.size(2);
    auto Kw  = weight.size(3);
    TORCH_CHECK(Cin == CinW, \\"Cin mismatch\\");

    auto Hout = H - Kh + 1;
    auto Wout = W - Kw + 1;
    TORCH_CHECK(Hout >= 0 && Wout >= 0, \\"Invalid output size\\");

    at::Tensor y = at::empty({N, Cout, Hout, Wout}, x.options());

    EXEC_NPU_CMD(aclnnConv2dBatchNormScalingCustom, x, weight, bias, y);
    return y;

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused Conv2d + BatchNorm (using running stats) + scaling via a custom AscendC operator.
    BN and scaling are folded into conv weights/bias on host; device kernel performs conv only.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.bn = torch.nn.BatchNorm2d(out_channels)
        self.scaling_factor = float(scaling_factor)

    def forward(self, x):
        # Fold BatchNorm and scaling into conv params: W' and b'
        W = self.conv.weight
        b = self.conv.bias if self.conv.bias is not None else torch.zeros(W.size(0), device=W.device, dtype=W.dtype)

        running_mean = self.bn.running_mean
        running_var = self.bn.running_var
        gamma = self.bn.weight
        beta = self.bn.bias
        eps = self.bn.eps

        # Compute per-channel alpha and fused params
        denom = torch.sqrt(running_var + eps)
        alpha = (self.scaling_factor * gamma) / denom  # [Cout]
        W_fused = W * alpha.view(-1, 1, 1, 1)
        b_fused = self.scaling_factor * (beta + (b - running_mean) * (gamma / denom))

        return custom_ops_lib.conv2d_batch_norm_scaling_custom(x, W_fused, b_fused)
'''
