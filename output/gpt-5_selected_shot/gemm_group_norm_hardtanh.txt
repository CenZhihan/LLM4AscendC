kernel_src="""
#include "kernel_operator.h"

extern \"C\" __global__ __aicore__ void gemm_group_norm_hardtanh_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR gn_weight, GM_ADDR gn_bias,
    GM_ADDR ht_min, GM_ADDR ht_max, GM_ADDR groups,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t M = tiling_data.m;
    const uint32_t K = tiling_data.k;
    const uint32_t N = tiling_data.n;

    // Cast global memory pointers
    __gm__ float* x_gm = (__gm__ float*)x;                // [M, K]
    __gm__ float* w_gm = (__gm__ float*)weight;           // [N, K]
    __gm__ float* b_gm = (__gm__ float*)bias;             // [N]
    __gm__ float* gamma_gm = (__gm__ float*)gn_weight;    // [N]
    __gm__ float* beta_gm  = (__gm__ float*)gn_bias;      // [N]
    __gm__ float* y_gm = (__gm__ float*)y;                // [M, N]

    const float ht_min_val = ((__gm__ float*)ht_min)[0];
    const float ht_max_val = ((__gm__ float*)ht_max)[0];
    const int32_t groups_val = ((__gm__ int32_t*)groups)[0];

    const float eps = 1e-5f;

    uint32_t block_num = AscendC::GetBlockNum();
    uint32_t block_idx = AscendC::GetBlockIdx();

    // Split M across blocks
    uint32_t rows_per_block = (M + block_num - 1) / block_num;
    uint32_t row_start = block_idx * rows_per_block;
    uint32_t row_end = row_start + rows_per_block;
    if (row_end > M) row_end = M;

    if (row_start >= row_end) {
        return;
    }

    // Guard groups_val
    int32_t G = groups_val > 0 ? groups_val : 1;
    uint32_t group_size = (G > 0) ? (N / (uint32_t)G) : N;
    if (group_size == 0) group_size = N;

    for (uint32_t m = row_start; m < row_end; ++m) {
        // Process per group to limit local buffer size
        for (int32_t g = 0; g < G; ++g) {
            uint32_t start_o = (uint32_t)g * group_size;
            uint32_t end_o = (g == G - 1) ? N : (start_o + group_size);
            uint32_t cur_group_size = end_o - start_o;

            // Temporary buffer in register-backed local array (size bounded by group)
            // Note: This is a simplified reference implementation.
            // On real hardware, consider tiling K and using vector instructions.
            for (uint32_t j = 0; j < cur_group_size; ++j) {
                uint32_t o = start_o + j;
                float acc = b_gm[o];
                // GEMM: dot(x[m,:], w[o,:])
                for (uint32_t k = 0; k < K; ++k) {
                    acc += x_gm[m * K + k] * w_gm[o * K + k];
                }
                // Overwrite y_gm temporarily to hold pre-norm result
                y_gm[m * N + o] = acc;
            }

            // Compute mean
            float sum = 0.f;
            for (uint32_t j = 0; j < cur_group_size; ++j) {
                uint32_t o = start_o + j;
                sum += y_gm[m * N + o];
            }
            float mean = sum / static_cast<float>(cur_group_size);

            // Compute variance
            float vsum = 0.f;
            for (uint32_t j = 0; j < cur_group_size; ++j) {
                uint32_t o = start_o + j;
                float d = y_gm[m * N + o] - mean;
                vsum += d * d;
            }
            float var = vsum / static_cast<float>(cur_group_size);
            float inv_std = AscendC::vsqrt(1.0f / (var + eps)); // approximate inverse std

            // Normalize, affine, hardtanh, and write out
            for (uint32_t j = 0; j < cur_group_size; ++j) {
                uint32_t o = start_o + j;
                float v = y_gm[m * N + o];
                float norm = (v - mean) * inv_std;
                float affine = norm * gamma_gm[o] + beta_gm[o];
                // HardTanh
                float clamped = affine;
                if (clamped < ht_min_val) clamped = ht_min_val;
                if (clamped > ht_max_val) clamped = ht_max_val;
                y_gm[m * N + o] = clamped;
            }
        }
    }
}

at::Tensor gemm_group_norm_hardtanh_custom_impl_npu(const at::Tensor& x,
                                                    const at::Tensor& weight,
                                                    const at::Tensor& bias,
                                                    const at::Tensor& gn_weight,
                                                    const at::Tensor& gn_bias,
                                                    double ht_min,
                                                    double ht_max,
                                                    int64_t groups) {
    TORCH_CHECK(x.is_cuda() || x.device().type() == at::kPrivateUse1, "Input must be on NPU");
    auto options_f = x.options().dtype(at::kFloat);
    auto options_i = x.options().dtype(at::kInt);

    at::Tensor y = at::empty({x.size(0), weight.size(0)}, options_f);

    at::Tensor ht_min_t = at::empty({1}, options_f);
    at::Tensor ht_max_t = at::empty({1}, options_f);
    at::Tensor groups_t = at::empty({1}, options_i);

    ht_min_t.fill_(ht_min);
    ht_max_t.fill_(ht_max);
    groups_t.fill_(static_cast<int>(groups));

    EXEC_NPU_CMD(aclnnGemmGroupNormHardtanhCustom,
                 x, weight, bias, gn_weight, gn_bias, ht_min_t, ht_max_t, groups_t, y);
    return y;
}

    m.impl("gemm_group_norm_hardtanh_custom", &gemm_group_norm_hardtanh_custom_impl_npu);

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Optimized model with fused custom AscendC operator:
    Y = HardTanh(GroupNorm(GEMM(X, W^T) + b))
    """
    def __init__(self, in_features, out_features, num_groups, hardtanh_min, hardtanh_max):
        super(ModelNew, self).__init__()
        self.gemm = nn.Linear(in_features, out_features)
        self.group_norm = nn.GroupNorm(num_groups, out_features)
        self.hardtanh = nn.Hardtanh(min_val=hardtanh_min, max_val=hardtanh_max)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Use fused custom op with module parameters
        return custom_ops_lib.gemm_group_norm_hardtanh_custom(
            x,
            self.gemm.weight,
            self.gemm.bias,
            self.group_norm.weight,
            self.group_norm.bias,
            float(self.hardtanh.min_val),
            float(self.hardtanh.max_val),
            int(self.group_norm.num_groups)
        )
