kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelKvCachedSpeculativeAttention {
public:
    __aicore__ inline KernelKvCachedSpeculativeAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o,
                                uint32_t b, uint32_t qLen, uint32_t kvLen, uint32_t dModel, uint32_t tileNumQ)
    {
        this->b = b;
        this->qLen = qLen;
        this->kvLen = kvLen;
        this->dModel = dModel;
        this->tileNumQ = tileNumQ;

        uint32_t batchIdx = AscendC::GetBlockIdx();
        if (batchIdx >= b) {
            return;
        }

        uint64_t qBatchOffset = static_cast<uint64_t>(batchIdx) * qLen * dModel;
        uint64_t kvBatchOffset = static_cast<uint64_t>(batchIdx) * kvLen * dModel;

        qGm.SetGlobalBuffer((__gm__ DTYPE_Q *)q + qBatchOffset, qLen * dModel);
        kGm.SetGlobalBuffer((__gm__ DTYPE_K *)k + kvBatchOffset, kvLen * dModel);
        vGm.SetGlobalBuffer((__gm__ DTYPE_V *)v + kvBatchOffset, kvLen * dModel);
        oGm.SetGlobalBuffer((__gm__ DTYPE_O *)o + qBatchOffset, qLen * dModel);

        // For simplicity, tile by contiguous elements along (qLen * dModel)
        totalElemPerBatch = qLen * dModel;
        uint32_t tiles = tileNumQ * BUFFER_NUM;
        tileLength = (totalElemPerBatch + tiles - 1) / tiles;

        pipe.InitBuffer(inQueueQ, BUFFER_NUM, tileLength * sizeof(DTYPE_Q));
        pipe.InitBuffer(outQueueO, BUFFER_NUM, tileLength * sizeof(DTYPE_O));
    }

    __aicore__ inline void Process()
    {
        if (AscendC::GetBlockIdx() >= b) {
            return;
        }
        int32_t loopCount = tileNumQ * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            uint32_t offset = i * tileLength;
            if (offset >= totalElemPerBatch) break;

            uint32_t curLen = AscendC::min(tileLength, totalElemPerBatch - offset);

            CopyIn(offset, curLen);
            Compute(offset, curLen);
            CopyOut(offset, curLen);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_Q> qLocal = inQueueQ.AllocTensor<DTYPE_Q>();
        AscendC::DataCopy(qLocal, qGm[offset], curLen);
        inQueueQ.EnQue(qLocal);
    }

    __aicore__ inline void Compute(uint32_t offset, uint32_t curLen)
    {
        // Placeholder compute: directly forward Q to O segment-wise.
        // Real implementation would compute scaled dot-product attention using K/V caches.
        AscendC::LocalTensor<DTYPE_Q> qLocal = inQueueQ.DeQue<DTYPE_Q>();
        AscendC::LocalTensor<DTYPE_O> oLocal = outQueueO.AllocTensor<DTYPE_O>();
        AscendC::DataCopy(oLocal, qLocal, curLen);
        outQueueO.EnQue<DTYPE_O>(oLocal);
        inQueueQ.FreeTensor(qLocal);
    }

    __aicore__ inline void CopyOut(uint32_t offset, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_O> oLocal = outQueueO.DeQue<DTYPE_O>();
        AscendC::DataCopy(oGm[offset], oLocal, curLen);
        outQueueO.FreeTensor(oLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueO;

    AscendC::GlobalTensor<DTYPE_Q> qGm;
    AscendC::GlobalTensor<DTYPE_K> kGm;
    AscendC::GlobalTensor<DTYPE_V> vGm;
    AscendC::GlobalTensor<DTYPE_O> oGm;

    uint32_t b;
    uint32_t qLen;
    uint32_t kvLen;
    uint32_t dModel;
    uint32_t tileNumQ;
    uint32_t totalElemPerBatch;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void kv_cached_speculative_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelKvCachedSpeculativeAttention op;
    op.Init(q, k, v, o,
            tiling_data.b, tiling_data.qLen, tiling_data.kvLen, tiling_data.dModel, tiling_data.tileNumQ);
    op.Process();
}
