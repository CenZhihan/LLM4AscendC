kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor buffers per queue

class KernelChannelMin {
public:
    __aicore__ inline KernelChannelMin() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t S, uint32_t C, uint32_t tileNum)
    {
        this->S = S;            // number of positions per sample across N,H,W
        this->C = C;            // channel count
        this->blockLength = S / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        uint32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();
        // x tensor spans channels; ensure capacity covers access to all channels for this block
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + blockOffset, this->blockLength + (this->C - 1) * this->S);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + blockOffset, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            ComputeTile(i);
            StoreTile(i);
        }
    }

private:
    __aicore__ inline void ComputeTile(int32_t progress)
    {
        uint32_t offset = progress * this->tileLength;

        // Initialize output local buffer with first channel slice
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_X> xLocal0 = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal0, xGm[offset], this->tileLength);
        AscendC::DataCopy(yLocal, xLocal0, this->tileLength);
        inQueueX.FreeTensor(xLocal0);

        // Iterate over remaining channels and reduce by min
        for (uint32_t c = 1; c < this->C; ++c) {
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[c * this->S + offset], this->tileLength);
            AscendC::Min(yLocal, yLocal, xLocal, this->tileLength);
            inQueueX.FreeTensor(xLocal);
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void StoreTile(int32_t progress)
    {
        uint32_t offset = progress * this->tileLength;
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[offset], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t S;
    uint32_t C;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelChannelMin op;
    op.Init(x, y, tiling_data.S, tiling_data.C, tiling_data.tileNum);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: retains Conv2d in PyTorch, fuses channel-wise min into a custom AscendC op.
    Scaling is applied in PyTorch prior to reduction.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scale_factor = scale_factor

    def forward(self, x):
        x = self.conv(x)
        x = x * self.scale_factor
        # Custom kernel performs min over channel dimension (dim=1, keepdim=True)
        x = custom_ops_lib.conv2d_scaling_min_custom(x)
        return x
'''
