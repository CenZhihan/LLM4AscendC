kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t SUB_TILE_D = 256; // tile along d_model dimension

class KernelKvCachedAttentionInference {
public:
    __aicore__ inline KernelKvCachedAttentionInference() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o,
                                uint32_t batch, uint32_t qLen, uint32_t kvLen,
                                uint32_t dModel, uint32_t tileK)
    {
        this->batch = batch;
        this->qLen = qLen;
        this->kvLen = kvLen;
        this->dModel = dModel;
        this->tileK = tileK;

        // map block to (b, q_idx)
        uint32_t pairIdx = AscendC::GetBlockIdx();
        this->b = pairIdx / qLen;
        this->qIdx = pairIdx % qLen;

        // set GM buffers
        qGm.SetGlobalBuffer((__gm__ float *)q, batch * qLen * dModel);
        kGm.SetGlobalBuffer((__gm__ float *)k, batch * kvLen * dModel);
        vGm.SetGlobalBuffer((__gm__ float *)v, batch * kvLen * dModel);
        oGm.SetGlobalBuffer((__gm__ float *)o, batch * qLen * dModel);

        // init local buffers
        pipe.InitBuffer(qQueue, 1, dModel * sizeof(float));
        pipe.InitBuffer(oQueue, 1, dModel * sizeof(float));
        pipe.InitBuffer(logitQueue, 1, tileK * sizeof(float));
        pipe.InitBuffer(chunkQueue, 1, SUB_TILE_D * sizeof(float));
        pipe.InitBuffer(tileChunkQueue, 1, SUB_TILE_D * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Load Q[b, qIdx, :]
        AscendC::LocalTensor<float> qLocal = qQueue.AllocTensor<float>();
        uint64_t qOffset = ((uint64_t)b * qLen + qIdx) * dModel;
        AscendC::DataCopy(qLocal, qGm[qOffset], dModel);
        qQueue.EnQue(qLocal);

        // First pass: compute global max of scaled logits
        float scale = 1.0f / AscendC::Sqrt((float)dModel);
        float globalMax = -3.402823466e+38F; // -FLT_MAX

        int32_t kvTiles = (kvLen + tileK - 1) / tileK;
        for (int32_t t = 0; t < kvTiles; ++t) {
            int32_t curTileLen = AscendC::Min(tileK, (int32_t)(kvLen - t * tileK));

            AscendC::LocalTensor<float> logits = logitQueue.AllocTensor<float>();
            // zero logits
            for (int32_t i = 0; i < curTileLen; ++i) {
                logits.GetValue(i) = 0.0f;
            }

            // accumulate dot products over dModel
            for (int32_t d0 = 0; d0 < (int32_t)dModel; d0 += SUB_TILE_D) {
                int32_t curD = AscendC::Min(SUB_TILE_D, (int32_t)(dModel - d0));
                AscendC::LocalTensor<float> qChunk = chunkQueue.AllocTensor<float>();
                AscendC::DataCopy(qChunk, qLocal[d0], curD);

                for (int32_t i = 0; i < curTileLen; ++i) {
                    uint64_t kRowOffset = ((uint64_t)b * kvLen + (t * tileK + i)) * dModel + d0;
                    AscendC::LocalTensor<float> kChunk = tileChunkQueue.AllocTensor<float>();
                    AscendC::DataCopy(kChunk, kGm[kRowOffset], curD);

                    // compute partial dot
                    float sum = 0.0f;
                    for (int32_t j = 0; j < curD; ++j) {
                        sum += qChunk.GetValue(j) * kChunk.GetValue(j);
                    }
                    logits.GetValue(i) += sum;

                    tileChunkQueue.FreeTensor(kChunk);
                }
                chunkQueue.FreeTensor(qChunk);
            }

            // scale and max
            for (int32_t i = 0; i < curTileLen; ++i) {
                float v = logits.GetValue(i) * scale;
                logits.GetValue(i) = v;
                globalMax = AscendC::Max(globalMax, v);
            }
            logitQueue.FreeTensor(logits);
        }

        // Second pass: accumulate sumExp and output vector
        float sumExp = 0.0f;
        AscendC::LocalTensor<float> oLocal = oQueue.AllocTensor<float>();
        for (int32_t d = 0; d < (int32_t)dModel; ++d) {
            oLocal.GetValue(d) = 0.0f;
        }

        for (int32_t t = 0; t < kvTiles; ++t) {
            int32_t curTileLen = AscendC::Min(tileK, (int32_t)(kvLen - t * tileK));

            AscendC::LocalTensor<float> logits = logitQueue.AllocTensor<float>();
            // recompute logits for this tile
            for (int32_t i = 0; i < curTileLen; ++i) {
                logits.GetValue(i) = 0.0f;
            }
            for (int32_t d0 = 0; d0 < (int32_t)dModel; d0 += SUB_TILE_D) {
                int32_t curD = AscendC::Min(SUB_TILE_D, (int32_t)(dModel - d0));
                AscendC::LocalTensor<float> qChunk = chunkQueue.AllocTensor<float>();
                AscendC::DataCopy(qChunk, qLocal[d0], curD);

                for (int32_t i = 0; i < curTileLen; ++i) {
                    uint64_t kRowOffset = ((uint64_t)b * kvLen + (t * tileK + i)) * dModel + d0;
                    AscendC::LocalTensor<float> kChunk = tileChunkQueue.AllocTensor<float>();
                    AscendC::DataCopy(kChunk, kGm[kRowOffset], curD);

                    float sum = 0.0f;
                    for (int32_t j = 0; j < curD; ++j) {
                        sum += qChunk.GetValue(j) * kChunk.GetValue(j);
                    }
                    logits.GetValue(i) += sum;
                    tileChunkQueue.FreeTensor(kChunk);
                }
                chunkQueue.FreeTensor(qChunk);
            }

            // scale and compute weights, then accumulate oLocal += w * V
            for (int32_t i = 0; i < curTileLen; ++i) {
                float logit = logits.GetValue(i) * scale;
                float w = AscendC::Exp(logit - globalMax);
                sumExp += w;

                // accumulate o with w * V[b, t*tileK+i, :]
                for (int32_t d0 = 0; d0 < (int32_t)dModel; d0 += SUB_TILE_D) {
                    int32_t curD = AscendC::Min(SUB_TILE_D, (int32_t)(dModel - d0));
                    uint64_t vRowOffset = ((uint64_t)b * kvLen + (t * tileK + i)) * dModel + d0;
                    AscendC::LocalTensor<float> vChunk = tileChunkQueue.AllocTensor<float>();
                    AscendC::DataCopy(vChunk, vGm[vRowOffset], curD);
                    for (int32_t j = 0; j < curD; ++j) {
                        float newv = oLocal.GetValue(d0 + j) + w * vChunk.GetValue(j);
                        oLocal.SetValue(d0 + j, newv);
                    }
                    tileChunkQueue.FreeTensor(vChunk);
                }
            }
            logitQueue.FreeTensor(logits);
        }

        // normalize by sumExp
        float invSum = 1.0f / sumExp;
        for (int32_t d = 0; d < (int32_t)dModel; ++d) {
            oLocal.SetValue(d, oLocal.GetValue(d) * invSum);
        }

        // write output
        uint64_t oOffset = ((uint64_t)b * qLen + qIdx) * dModel;
        AscendC::DataCopy(oGm[oOffset], oLocal, dModel);

        // free locals
        qQueue.FreeTensor(qLocal);
        oQueue.FreeTensor(oLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> oQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> logitQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> chunkQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tileChunkQueue;

    AscendC::GlobalTensor<float> qGm;
    AscendC::GlobalTensor<float> kGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> oGm;

    uint32_t batch;
    uint32_t qLen;
    uint32_t kvLen;
    uint32_t dModel;
    uint32_t tileK;

    uint32_t b;
    uint32_t qIdx;
};

extern "C" __global__ __aicore__ void kv_cached_attention_inference_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelKvCachedAttentionInference op;
    op.Init(q, k, v, o,
            tiling_data.batch,
            tiling_data.qLen,
            tiling_data.kvLen,
            tiling_data.dModel,
            tiling_data.tileK);
    op.Process();
}
