kernel_src="""
#include "kernel_operator.h"

constexpr uint32_t BLOCK_DIM = 8;

// Simple fused kernel that assumes input x is the Linear (matmul) output of shape (N, L).
// It performs: AvgPool1d on dimension L with kernel_size=stride=pool_ks, then GELU (approx), then scale, then Max over dim=1.
// Output y has shape (N,).
struct TilingInfo {
    uint32_t batch;
    uint32_t rowLen;
    uint32_t tileNum;
};

class KernelFused {
public:
    __aicore__ inline KernelFused() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR poolks, GM_ADDR y, uint32_t batch, uint32_t rowLen)
    {
        xGm = reinterpret_cast<__gm__ float*>(x);
        scaleGm = reinterpret_cast<__gm__ float*>(scale);
        poolKsGm = reinterpret_cast<__gm__ int32_t*>(poolks);
        yGm = reinterpret_cast<__gm__ float*>(y);
        this->batch = batch;
        this->rowLen = rowLen;
    }

    __aicore__ inline float gelu_approx(float v)
    {
        // Lightweight approximation: 0.5 * v
        // (placeholder for demo; replace with better approximation if needed)
        return 0.5f * v;
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        float scaleVal = 1.0f;
        if (scaleGm != nullptr) {
            scaleVal = scaleGm[0];
        }

        int32_t poolKs = 1;
        if (poolKsGm != nullptr) {
            poolKs = poolKsGm[0];
            if (poolKs <= 0) poolKs = 1;
        }

        uint32_t segs = (poolKs > 0) ? (rowLen / static_cast<uint32_t>(poolKs)) : rowLen;

        for (uint32_t n = blockIdx; n < batch; n += blockNum) {
            const __gm__ float* rowBase = xGm + static_cast<uint64_t>(n) * static_cast<uint64_t>(rowLen);
            float maxVal = -3.4e38f;

            for (uint32_t s = 0; s < segs; ++s) {
                float acc = 0.0f;
                uint32_t start = s * static_cast<uint32_t>(poolKs);
                for (int32_t k = 0; k < poolKs; ++k) {
                    uint32_t idx = start + static_cast<uint32_t>(k);
                    if (idx < rowLen) {
                        acc += rowBase[idx];
                    }
                }
                float avg = acc / static_cast<float>(poolKs > 0 ? poolKs : 1);
                float g = gelu_approx(avg);
                float val = g * scaleVal;
                if (val > maxVal) {
                    maxVal = val;
                }
            }
            yGm[n] = maxVal;
        }
    }

private:
    __gm__ float* xGm;
    __gm__ float* yGm;
    __gm__ float* scaleGm;
    __gm__ int32_t* poolKsGm;
    uint32_t batch;
    uint32_t rowLen;
};

extern "C" __global__ __aicore__ void matmul_avg_pool_gelu_scale_max_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR pool_ks, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, scale, pool_ks, y, tiling_data.batch, tiling_data.rowLen);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a custom AscendC fused operator for AvgPool1d + GELU + Scale + Max
    on top of the Linear (matmul) result.
    """
    def __init__(self, in_features, out_features, pool_kernel_size, scale_factor):
        super(ModelNew, self).__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.pool_kernel_size = int(pool_kernel_size)
        self.scale_factor = float(scale_factor)

    def forward(self, x: torch.Tensor):
        # Compute Linear (MatMul + Bias)
        x = self.matmul(x)  # shape: (N, out_features)

        # Prepare scalar inputs for fused kernel
        scale_t = torch.tensor(self.scale_factor, dtype=x.dtype, device=x.device)
        pool_ks_t = torch.tensor(self.pool_kernel_size, dtype=torch.int32, device=x.device)

        # Fused: AvgPool1d(kernel=pool_ks), GELU, scale, and max over dim=1
        y = custom_ops_lib.matmul_avg_pool_gelu_scale_max_custom(x, scale_t, pool_ks_t)
        return y
