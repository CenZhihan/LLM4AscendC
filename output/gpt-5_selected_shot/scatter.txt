#include "scatter_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ScatterCustomTilingData tiling;

    const auto* x_shape = context->GetInputShape(0);
    const auto* idx_shape = context->GetInputShape(1);

    uint32_t rows = 1;
    uint32_t colsX = 1;
    uint32_t colsIdx = 1;

    if (x_shape->GetOriginShape().GetDimNum() >= 2) {
        rows = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
        colsX = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    } else {
        rows = static_cast<uint32_t>(x_shape->GetOriginShape().GetShapeSize());
        colsX = 1;
    }
    if (idx_shape->GetOriginShape().GetDimNum() >= 2) {
        colsIdx = static_cast<uint32_t>(idx_shape->GetOriginShape().GetDim(1));
    } else if (idx_shape->GetOriginShape().GetDimNum() == 1) {
        colsIdx = static_cast<uint32_t>(idx_shape->GetOriginShape().GetDim(0));
    }

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_rows(rows);
    tiling.set_colsX(colsX);
    tiling.set_colsIdx(colsIdx);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* z_shape = context->GetOutputShape(0);
    *z_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto xType = context->GetInputDataType(0);
    context->SetOutputDataType(0, xType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // buffer count per queue

class KernelScatter {
public:
    __aicore__ inline KernelScatter() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR index, GM_ADDR updates, GM_ADDR z,
                                uint32_t rows, uint32_t colsX, uint32_t colsIdx, uint32_t /*tileNum*/)
    {
        this->rows = rows;
        this->colsX = colsX;
        this->colsIdx = colsIdx;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Split rows across blocks
        uint32_t rowsPerBlock = rows / blockNum;
        uint32_t left = rows % blockNum;
        // Distribute remainder across first 'left' blocks
        this->startRow = blockIdx * rowsPerBlock + AscendC::Min(blockIdx, left);
        this->myRows = rowsPerBlock + (blockIdx < left ? 1 : 0);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->startRow * this->colsX, this->myRows * this->colsX);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_Y*)index + this->startRow * this->colsIdx, this->myRows * this->colsIdx);
        updGm.SetGlobalBuffer((__gm__ DTYPE_X*)updates + this->startRow * this->colsIdx, this->myRows * this->colsIdx);
        zGm.SetGlobalBuffer((__gm__ DTYPE_X*)z + this->startRow * this->colsX, this->myRows * this->colsX);

        pipe.InitBuffer(inQueueXRow, BUFFER_NUM, this->colsX * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueIdxRow, BUFFER_NUM, this->colsIdx * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueUpdRow, BUFFER_NUM, this->colsIdx * sizeof(DTYPE_X));
        // We'll write back directly; no separate out queue required
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->myRows; ++r) {
            // Copy one row of x
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueXRow.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[r * this->colsX], this->colsX);
            inQueueXRow.EnQue(xLocal);

            // Copy one row of indices
            AscendC::LocalTensor<DTYPE_Y> idxLocal = inQueueIdxRow.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(idxLocal, idxGm[r * this->colsIdx], this->colsIdx);
            inQueueIdxRow.EnQue(idxLocal);

            // Copy one row of updates
            AscendC::LocalTensor<DTYPE_X> updLocal = inQueueUpdRow.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(updLocal, updGm[r * this->colsIdx], this->colsIdx);
            inQueueUpdRow.EnQue(updLocal);

            // Dequeue for compute
            AscendC::LocalTensor<DTYPE_X> xRow = inQueueXRow.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Y> idxRow = inQueueIdxRow.DeQue<DTYPE_Y>();
            AscendC::LocalTensor<DTYPE_X> updRow = inQueueUpdRow.DeQue<DTYPE_X>();

            // Perform scatter along dim=1: z[row, idx[row,j]] = updates[row,j]
            for (uint32_t j = 0; j < this->colsIdx; ++j) {
                int64_t col = static_cast<int64_t>(idxRow.GetValue(j)); // supports int32/int64
                if (col >= 0 && col < static_cast<int64_t>(this->colsX)) {
                    xRow.SetValue(static_cast<uint32_t>(col), updRow.GetValue(j));
                }
            }

            // Write back the updated row to z
            AscendC::DataCopy(zGm[r * this->colsX], xRow, this->colsX);

            // Free local tensors
            inQueueXRow.FreeTensor(xRow);
            inQueueIdxRow.FreeTensor(idxRow);
            inQueueUpdRow.FreeTensor(updRow);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueXRow;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdxRow;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueUpdRow;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> idxGm;
    AscendC::GlobalTensor<DTYPE_X> updGm;
    AscendC::GlobalTensor<DTYPE_X> zGm;

    uint32_t rows;
    uint32_t colsX;
    uint32_t colsIdx;
    uint32_t startRow;
    uint32_t myRows;
};

extern "C" __global__ __aicore__ void scatter_custom(GM_ADDR x, GM_ADDR index, GM_ADDR updates, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelScatter op;
    op.Init(x, index, updates, z, tiling_data.rows, tiling_data.colsX, tiling_data.colsIdx, tiling_data.tileNum);
    op.Process();
}
"""

at::Tensor scatter_custom_impl_npu(const at::Tensor& x, const at::Tensor& index, const at::Tensor& updates) {
    // Expect dim=1 scatter behavior; output same shape as x
    at::Tensor result = at::empty_like(x);
    EXEC_NPU_CMD(aclnnScatterCustom, x, index, updates, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("scatter_custom", &scatter_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("scatter_custom", &scatter_custom_impl_npu, "scatter along dim=1");
}
