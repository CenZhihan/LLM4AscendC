kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

extern \"C\" __global__ __aicore__ void matmul_group_norm_leaky_relu_sum_custom(
    GM_ADDR x,           // [N, K]
    GM_ADDR weight,      // [M, K] (Linear weight)
    GM_ADDR bias,        // [M]
    GM_ADDR gn_weight,   // [M]
    GM_ADDR gn_bias,     // [M]
    GM_ADDR y,           // [N, M]
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);

    const uint32_t N = t.batch;
    const uint32_t K = t.inSize;
    const uint32_t M = t.outSize;
    const uint32_t groups = t.numGroups > 0 ? t.numGroups : 1;
    const uint32_t groupSize = M / groups;
    const float eps = t.eps;
    const float negSlope = t.negativeSlope;

    __gm__ DTYPE* xGm = reinterpret_cast<__gm__ DTYPE*>(x);
    __gm__ DTYPE* wGm = reinterpret_cast<__gm__ DTYPE*>(weight);
    __gm__ DTYPE* bGm = reinterpret_cast<__gm__ DTYPE*>(bias);
    __gm__ DTYPE* gwGm = reinterpret_cast<__gm__ DTYPE*>(gn_weight);
    __gm__ DTYPE* gbGm = reinterpret_cast<__gm__ DTYPE*>(gn_bias);
    __gm__ DTYPE* yGm = reinterpret_cast<__gm__ DTYPE*>(y);

    // Simple parallel split across blocks by rows (N)
    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t rowsPerBlock = (N + blockNum - 1) / blockNum;
    uint32_t rowStart = blockIdx * rowsPerBlock;
    uint32_t rowEnd = AscendC::Min(rowStart + rowsPerBlock, N);

    // Allocate small local buffers per row processing
    for (uint32_t n = rowStart; n < rowEnd; ++n) {
        // 1) GEMM: y[n, :] = x[n, :] @ W^T + b
        // Compute into L1 via vector add/mul in a naive loop
        // We'll compute output vector yrow[m]
        // To avoid large local memory, compute in chunks along M dimension
        const uint32_t tileM = 256; // chunk of output features
        for (uint32_t mBase = 0; mBase < M; mBase += tileM) {
            uint32_t mLen = AscendC::Min(tileM, M - mBase);

            // Initialize accumulation buffer
            // Using software loop with GM reads
            for (uint32_t m = 0; m < mLen; ++m) {
                DTYPE acc = 0.f;
                // Dot product over K
                for (uint32_t k = 0; k < K; ++k) {
                    DTYPE xv = xGm[n * K + k];
                    DTYPE wv = wGm[(mBase + m) * K + k];
                    acc += xv * wv;
                }
                // add bias
                acc += bGm[mBase + m];
                // store temporary to yGm (we will normalize in-place)
                yGm[n * M + (mBase + m)] = acc;
            }
        }

        // 2) GroupNorm over channel dimension M for this row n
        // For each group, compute mean and variance, then normalize
        for (uint32_t g = 0; g < groups; ++g) {
            uint32_t cStart = g * groupSize;
            uint32_t cEnd = cStart + groupSize;

            // mean
            DTYPE mean = 0.f;
            for (uint32_t c = cStart; c < cEnd; ++c) {
                mean += yGm[n * M + c];
            }
            mean /= static_cast<DTYPE>(groupSize);

            // variance
            DTYPE var = 0.f;
            for (uint32_t c = cStart; c < cEnd; ++c) {
                DTYPE v = yGm[n * M + c] - mean;
                var += v * v;
            }
            var /= static_cast<DTYPE>(groupSize);
            DTYPE invStd = 1.0f / AscendC::Sqrt(var + static_cast<DTYPE>(eps));

            // normalize, affine, leaky relu, and sum with itself (x + x)
            for (uint32_t c = cStart; c < cEnd; ++c) {
                DTYPE v = yGm[n * M + c];
                DTYPE norm = (v - mean) * invStd;
                DTYPE affine = norm * gwGm[c] + gbGm[c];
                // leaky relu
                DTYPE act = affine >= 0.f ? affine : affine * static_cast<DTYPE>(negSlope);
                // final sum x + x
                yGm[n * M + c] = act + act;
            }
        }
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using fused custom AscendC kernel:
    y = (LeakyReLU(GroupNorm(Linear(x)))) + (LeakyReLU(GroupNorm(Linear(x))))
    """
    def __init__(self, input_size, hidden_size, num_groups, eps=1e-5, negative_slope=0.01):
        super().__init__()
        self.fc = torch.nn.Linear(input_size, hidden_size, bias=True)
        self.gn = torch.nn.GroupNorm(num_groups=num_groups, num_channels=hidden_size, eps=eps, affine=True)
        self.negative_slope = float(negative_slope)
        self.num_groups = int(num_groups)
        self.eps = float(eps)

    def forward(self, x):
        # Pass raw tensors and attributes to the fused custom kernel.
        # Linear: weight [out, in], bias [out]
        # GroupNorm affine: weight [out], bias [out]
        return custom_ops_lib.matmul_group_norm_leaky_relu_sum_custom(
            x,
            self.fc.weight,
            self.fc.bias,
            self.gn.weight,
            self.gn.bias,
            self.num_groups,
            self.eps,
            self.negative_slope
        )
