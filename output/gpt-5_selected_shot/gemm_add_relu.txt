    if (x_shape->GetOriginShape().GetDimNum() >= 2 && w_shape->GetOriginShape().GetDimNum() >= 2) {

class KernelGemmAddRelu {
public:
    __aicore__ inline KernelGemmAddRelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batch, uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        this->B = batch;
        this->M = m; // in_features
        this->N = n; // out_features
        this->K = k; // equals M for GEMM x:[B,M], w:[M,N]
        this->tileM = tileM;
        this->tileN = tileN;
        this->tileK = tileK;

        // Split batch across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        this->blockBatch = (B + blocks - 1) / blocks; // ceil-div
        uint32_t bStart = this->blockBatch * AscendC::GetBlockIdx();
        if (bStart + this->blockBatch > B) {
            this->blockBatch = B > bStart ? (B - bStart) : 0;
        }
        this->batchStart = bStart;

        xGm.SetGlobalBuffer((__gm__ float*)x + bStart * M, this->blockBatch * M);
        wGm.SetGlobalBuffer((__gm__ float*)w, K * N);
        bGm.SetGlobalBuffer((__gm__ float*)b, N);
        yGm.SetGlobalBuffer((__gm__ float*)y + bStart * N, this->blockBatch * N);

        // Buffers for tiles
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileM * this->tileK * sizeof(float));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, this->tileK * this->tileN * sizeof(float));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileN * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileM * this->tileN * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Iterate over batch rows in this block
        for (uint32_t bm = 0; bm < this->blockBatch; bm += this->tileM) {
            uint32_t curTileM = (bm + tileM <= blockBatch) ? tileM : (blockBatch - bm);
            for (uint32_t bn = 0; bn < this->N; bn += this->tileN) {
                uint32_t curTileN = (bn + tileN <= N) ? tileN : (N - bn);

                // Initialize output tile to zeros
                AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();
                ZeroTile(yLocal, curTileM, curTileN);

                // Accumulate GEMM over K
                for (uint32_t bk = 0; bk < this->K; bk += this->tileK) {
                    uint32_t curTileK = (bk + tileK <= K) ? tileK : (K - bk);

                    // Copy X tile [curTileM, curTileK]
                    AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                    {
                        // Copy contiguous region per tile (row-major)
                        for (uint32_t i = 0; i < curTileM; ++i) {
                            uint32_t xOffsetGm = (bm + i) * M + bk;
                            AscendC::DataCopy(xLocal[i * curTileK], xGm[xOffsetGm], curTileK);
                        }
                    }
                    inQueueX.EnQue(xLocal);

                    // Copy W tile [curTileK, curTileN]
                    AscendC::LocalTensor<float> wLocal = inQueueW.AllocTensor<float>();
                    {
                        for (uint32_t kk = 0; kk < curTileK; ++kk) {
                            uint32_t wOffsetGm = (bk + kk) * N + bn;
                            AscendC::DataCopy(wLocal[kk * curTileN], wGm[wOffsetGm], curTileN);
                        }
                    }
                    inQueueW.EnQue(wLocal);

                    // Compute partial GEMM: yLocal += xLocal @ wLocal
                    ComputeGemmTile(yLocal, xLocal, wLocal, curTileM, curTileN, curTileK);

                    inQueueX.FreeTensor(xLocal);
                    inQueueW.FreeTensor(wLocal);
                }

                // Add bias and apply ReLU
                AscendC::LocalTensor<float> bLocal = inQueueB.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm[bn], curTileN);
                inQueueB.EnQue(bLocal);

                AddBiasRelu(yLocal, bLocal, curTileM, curTileN);

                inQueueB.FreeTensor(bLocal);

                // Write back Y tile
                for (uint32_t i = 0; i < curTileM; ++i) {
                    uint32_t yOffsetGm = (bm + i) * N + bn;
                    AscendC::DataCopy(yGm[yOffsetGm], yLocal[i * curTileN], curTileN);
                }
                outQueueY.FreeTensor(yLocal);
            }
        }
    }

private:

    {

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused GEMM + Bias + ReLU using custom AscendC kernel.
    x: [B, in_features], w: [in_features, out_features], b: [out_features]
    """
    def __init__(self, in_features, out_features, bias_shape):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features, bias=False)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):

        return custom_ops_lib.gemm_add_relu_custom(x, self.gemm.weight.t(), self.bias)
