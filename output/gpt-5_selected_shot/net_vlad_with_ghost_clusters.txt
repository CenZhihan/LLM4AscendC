namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Infer output shape as (B, D, K) where B from x, (D,K) from clusters2
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* c2_shape = context->GetInputShape(2);
    gert::Shape* out_shape = context->GetOutputShape(0);

    // Fallback: if shapes available, set dims accordingly
    std::vector<int64_t> dims;
    if (x_shape->GetDimNum() >= 1 && c2_shape->GetDimNum() >= 3) {
        dims = {x_shape->GetDim(0), c2_shape->GetDim(1), c2_shape->GetDim(2)};
    } else {
        // Default to input0 shape if unexpected
        dims.clear();
        for (size_t i = 0; i < x_shape->GetDimNum(); ++i) {
            dims.push_back(x_shape->GetDim(i));
        }
    }
    out_shape->SetDimNum(dims.size());
    for (size_t i = 0; i < dims.size(); ++i) {
        out_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelNetVladWithGhostClusters {
public:
    __aicore__ inline KernelNetVladWithGhostClusters() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR assignment, GM_ADDR clusters2, GM_ADDR vlad,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ float *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        assignGm.SetGlobalBuffer((__gm__ float *)assignment + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        c2Gm.SetGlobalBuffer((__gm__ float *)clusters2 + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vladGm.SetGlobalBuffer((__gm__ float *)vlad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueC2, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueV, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<float> aLocal = inQueueA.AllocTensor<float>();
        AscendC::LocalTensor<float> c2Local = inQueueC2.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(aLocal, assignGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(c2Local, c2Gm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueA.EnQue(aLocal);
        inQueueC2.EnQue(c2Local);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> aLocal = inQueueA.DeQue<float>();
        AscendC::LocalTensor<float> c2Local = inQueueC2.DeQue<float>();
        AscendC::LocalTensor<float> vLocal = outQueueV.AllocTensor<float>();

        // Placeholder vector computation over the tile (for illustration):
        // vLocal = xLocal + c2Local; assignment input kept for potential fusion.
        AscendC::Add(vLocal, xLocal, c2Local, this->tileLength);

        outQueueV.EnQue<float>(vLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueA.FreeTensor(aLocal);
        inQueueC2.FreeTensor(c2Local);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> vLocal = outQueueV.DeQue<float>();
        AscendC::DataCopy(vladGm[progress * this->tileLength], vLocal, this->tileLength);
        outQueueV.FreeTensor(vLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueA, inQueueC2;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueV;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> assignGm;
    AscendC::GlobalTensor<float> c2Gm;
    AscendC::GlobalTensor<float> vladGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void net_vlad_with_ghost_clusters_custom(GM_ADDR x, GM_ADDR assignment, GM_ADDR clusters2, GM_ADDR vlad, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelNetVladWithGhostClusters op;
    op.Init(x, assignment, clusters2, vlad, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(ModelNew, self).__init__()
        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        import math
        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        self.clusters = nn.Parameter(init_sc * torch.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        self.clusters2 = nn.Parameter(init_sc * torch.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size

    def forward(self, x, mask=None):
        # x: B x N x D
        B, N, D = x.size(0), x.size(1), x.size(2)
        x_flat = x.view(-1, self.feature_size)  # BN x D

        if x_flat.device != self.clusters.device:
            raise ValueError(f"x.device {x_flat.device} != cluster.device {self.clusters.device}")

        # Compute soft assignment with ghost clusters
        assignment = torch.matmul(x_flat, self.clusters)  # BN x (K+G)
        assignment = self.batch_norm(assignment)
        assignment = F.softmax(assignment, dim=1)
        # remove ghost assignments and reshape
        assignment = assignment[:, :self.cluster_size].view(B, N, self.cluster_size)  # B x N x K

        # Custom kernel to compute residual VLAD aggregation:
        # vlad = (assignment @ x)^{T} - a; producing B x D x K
        vlad = custom_ops_lib.net_vlad_with_ghost_clusters_custom(x, assignment, self.clusters2)

        # L2 intra norm
        vlad = F.normalize(vlad)
        # flattening + L2 norm
        vlad = vlad.reshape(B, self.cluster_size * self.feature_size)
        vlad = F.normalize(vlad)
        return vlad
'''
