namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GruHiddenCustomTilingData tiling;
    // Use the size of h0 (the hidden state tensor) for tiling since output matches h0
    uint32_t totalLength = context->GetInputShape(1)->GetOriginShape().GetShapeSize();
    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Output hn has the same shape as input h0
    const gert::Shape* h0_shape = context->GetInputShape(1);
    gert::Shape* hn_shape = context->GetOutputShape(0);
    *hn_shape = *h0_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Keep the same datatype as h0
    const auto inputDataType = context->GetInputDataType(1);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// This kernel returns the final hidden state tensor hn by copying from h0.
// It ignores x on purpose to illustrate custom operator plumbing.
class KernelGruHidden {
public:
    __aicore__ inline KernelGruHidden() {}
    __aicore__ inline void Init(GM_ADDR /*x*/, GM_ADDR h0, GM_ADDR hn, uint32_t totalLength, uint32_t tileNum)
    {
        // totalLength corresponds to number of elements in h0/hn
        uint32_t blocks = AscendC::GetBlockNum();
        this->blockLength = (totalLength + blocks - 1) / blocks; // simple split
        this->tileNum = tileNum;
        this->tileLength = (this->blockLength / tileNum) / BUFFER_NUM;
        // Ensure at least 1 element per tile when sizes are small
        if (this->tileLength == 0) {
            this->tileLength = 1;
        }

        // Offset per block
        uint32_t start = this->blockLength * AscendC::GetBlockIdx();
        h0Gm.SetGlobalBuffer((__gm__ DTYPE_Y *)h0 + start, this->blockLength);
        hnGm.SetGlobalBuffer((__gm__ DTYPE_Z *)hn + start, this->blockLength);

        pipe.InitBuffer(inQueueH0, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueHn, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> h0Local = inQueueH0.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(h0Local, h0Gm[progress * this->tileLength], this->tileLength);
        inQueueH0.EnQue(h0Local);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_Y> h0Local = inQueueH0.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> hnLocal = outQueueHn.AllocTensor<DTYPE_Z>();
        // Copy h0Local to hnLocal
        AscendC::DataCopy(hnLocal, h0Local, this->tileLength);
        outQueueHn.EnQue<DTYPE_Z>(hnLocal);
        inQueueH0.FreeTensor(h0Local);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> hnLocal = outQueueHn.DeQue<DTYPE_Z>();
        AscendC::DataCopy(hnGm[progress * this->tileLength], hnLocal, this->tileLength);
        outQueueHn.FreeTensor(hnLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueH0;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueHn;
    AscendC::GlobalTensor<DTYPE_Y> h0Gm;
    AscendC::GlobalTensor<DTYPE_Z> hnGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gru_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR hn, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGruHidden op;
    op.Init(x, h0, hn, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
