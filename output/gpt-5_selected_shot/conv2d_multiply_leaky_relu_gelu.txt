kernel_src="""
#include "kernel_operator.h"

__aicore__ inline float gelu_approx(float x) {
    // tanh-based approximation
    const float kAlpha = 0.7978845608028654f; // sqrt(2/pi)
    const float kCoeff = 0.044715f;
    float x3 = x * x * x;
    float inner = kAlpha * (x + kCoeff * x3);
    // AscendC provides vector tanh, but here scalar for simplicity
    float t = AscendC::Tanh(inner);
    return 0.5f * x * (1.0f + t);
}

extern \"C\" __global__ __aicore__ void conv2d_multiply_leaky_relu_gelu_custom(
    GM_ADDR x,
    GM_ADDR weight,
    GM_ADDR bias,
    GM_ADDR multiplier,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const int32_t block_idx = AscendC::GetBlockIdx();
    const int32_t block_num = AscendC::GetBlockNum();

    // Read dims from tiling
    const int32_t N    = (int32_t)tiling_data.n;
    const int32_t Cin  = (int32_t)tiling_data.cin;
    const int32_t Hin  = (int32_t)tiling_data.hin;
    const int32_t Win  = (int32_t)tiling_data.win;
    const int32_t Cout = (int32_t)tiling_data.cout;
    const int32_t Kh   = (int32_t)tiling_data.kh;
    const int32_t Kw   = (int32_t)tiling_data.kw;
    const int32_t Hout = (int32_t)tiling_data.hout;
    const int32_t Wout = (int32_t)tiling_data.wout;
    const float negSlope = tiling_data.negativeSlope;

    // Cast GM pointers
    __gm__ float* x_gm  = (__gm__ float*)x;        // [N, Cin, Hin, Win]
    __gm__ float* w_gm  = (__gm__ float*)weight;   // [Cout, Cin, Kh, Kw]
    __gm__ float* b_gm  = (__gm__ float*)bias;     // [Cout]
    __gm__ float* m_gm  = (__gm__ float*)multiplier; // [Cout, 1, 1] broadcast-able
    __gm__ float* y_gm  = (__gm__ float*)y;        // [N, Cout, Hout, Wout]

    // Simple block split over output channels
    int32_t oc_per_block = (Cout + block_num - 1) / block_num;
    int32_t oc_begin = block_idx * oc_per_block;
    int32_t oc_end = AscendC::Min(oc_begin + oc_per_block, Cout);

    for (int32_t n = 0; n < N; ++n) {
        for (int32_t oc = oc_begin; oc < oc_end; ++oc) {
            float bias_val = b_gm[oc];
            float mult_val = m_gm[oc]; // assume [Cout] or [Cout,1,1] contiguous
            for (int32_t oh = 0; oh < Hout; ++oh) {
                for (int32_t ow = 0; ow < Wout; ++ow) {
                    float acc = 0.0f;
                    // conv accumulation
                    for (int32_t ic = 0; ic < Cin; ++ic) {
                        for (int32_t kh = 0; kh < Kh; ++kh) {
                            int32_t ih = oh + kh;
                            for (int32_t kw = 0; kw < Kw; ++kw) {
                                int32_t iw = ow + kw;
                                int64_t x_idx = (((int64_t)n * Cin + ic) * Hin + ih) * Win + iw;
                                int64_t w_idx = (((int64_t)oc * Cin + ic) * Kh + kh) * Kw + kw;
                                acc += x_gm[x_idx] * w_gm[w_idx];
                            }
                        }
                    }
                    // bias
                    acc += bias_val;
                    // multiply by learnable scalar
                    acc *= mult_val;
                    // LeakyReLU
                    float lrelu = (acc >= 0.0f) ? acc : (acc * negSlope);
                    // GELU
                    float gelu = gelu_approx(lrelu);

                    int64_t y_idx = (((int64_t)n * Cout + oc) * Hout + oh) * Wout + ow;
                    y_gm[y_idx] = gelu;
                }
            }
        }
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv2d_multiply_leaky_relu_gelu_custom_impl_npu(const at::Tensor& x,
                                                           const at::Tensor& weight,
                                                           const at::Tensor& bias,
                                                           const at::Tensor& multiplier) {
    TORCH_CHECK(x.device().type() == at::kPrivateUse1, "Input must be on NPU");
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto Hin = x.size(2);
    auto Win = x.size(3);

    auto Cout = weight.size(0);
    auto Kh = weight.size(2);
    auto Kw = weight.size(3);

    // default conv params: stride=1, padding=0, dilation=1
    auto Hout = Hin - Kh + 1;
    auto Wout = Win - Kw + 1;
    TORCH_CHECK(Hout > 0 && Wout > 0, "Invalid output size");

    at::Tensor y = at::empty({N, Cout, Hout, Wout}, x.options());

    EXEC_NPU_CMD(aclnnConv2dMultiplyLeakyReluGeluCustom, x, weight, bias, multiplier, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_multiply_leaky_relu_gelu_custom", &conv2d_multiply_leaky_relu_gelu_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom kernel: Conv2d -> multiply by learnable scalar -> LeakyReLU -> GELU
    Assumptions: stride=1, padding=0, dilation=1, groups=1 (match original usage)
    """
    def __init__(self, in_channels, out_channels, kernel_size, multiplier_shape):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)
        self.multiplier = torch.nn.Parameter(torch.randn(multiplier_shape))
        self.leaky_relu = torch.nn.LeakyReLU()  # kept for parity in attributes (slope), not called directly

    def forward(self, x):
        return custom_ops_lib.conv2d_multiply_leaky_relu_gelu_custom(
            x, self.conv.weight, self.conv.bias, self.multiplier
        )
