kernel_src="""
#include "kernel_operator.h"

// Simple 1D transposed convolution (stride=1, padding=0, dilation=1, groups=1).
// This kernel maps each block to a (batch, channel) pair on the diagonal (cin==cout path).
// For general shapes, only min(cin, cout) diagonal channels are computed.

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed1D {
public:
    __aicore__ inline KernelConvTransposed1D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout, uint32_t ccommon,
                                uint32_t lin, uint32_t lout, uint32_t ksize, uint32_t tileNum)
    {
        this->n = n;
        this->cin = cin;
        this->cout = cout;
        this->ccommon = ccommon;
        this->lin = lin;
        this->lout = lout;
        this->ksize = ksize;

        uint32_t blk = AscendC::GetBlockIdx();
        uint32_t chPerBatch = (ccommon == 0) ? 1 : ccommon;
        this->nIdx = blk / chPerBatch;
        this->cIdx = blk % chPerBatch;

        if (this->nIdx >= this->n) {
            this->nIdx = 0;
            this->cIdx = 0;
        }

        // Offsets for (nIdx, cIdx) diagonal channel
        uint32_t xOffset = (this->nIdx * this->cin + this->cIdx) * this->lin;
        uint32_t yOffset = (this->nIdx * this->cout + this->cIdx) * this->lout;
        uint32_t wOffset = (this->cIdx * this->cout + this->cIdx) * this->ksize;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + xOffset, this->lin);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w + wOffset, this->ksize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + yOffset, this->lout);

        this->tileNum = tileNum;
        this->tileLength = (this->lin + (tileNum * BUFFER_NUM) - 1) / (tileNum * BUFFER_NUM);
        if (this->tileLength == 0) this->tileLength = this->lin;
        this->tileOutLength = this->tileLength + this->ksize - 1;

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, 1, this->ksize * sizeof(DTYPE_W));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileOutLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, 1, this->tileOutLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Initialize Y to zeros for this (n,c) plane by tiles to avoid race
        // We'll zero-write lazily per tile when computing.

        int32_t loopCount = AscendC::CeilDiv((int32_t)this->lin, (int32_t)this->tileLength);
        // Preload weights each iteration (small)
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint32_t start = (uint32_t)progress * this->tileLength;
        uint32_t validLen = this->lin > start ? AscendC::Min(this->tileLength, this->lin - start) : 0;

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        if (validLen > 0) {
            AscendC::DataCopy(xLocal, xGm[start], validLen);
            if (validLen < this->tileLength) {
                // pad remaining with zeros
                AscendC::LocalTensor<DTYPE_X> xPad = xLocal[validLen];
                AscendC::Duplicate(xPad, (DTYPE_X)0, this->tileLength - validLen);
            }
        } else {
            AscendC::Duplicate(xLocal, (DTYPE_X)0, this->tileLength);
        }
        inQueueX.EnQue(xLocal);

        AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.AllocTensor<DTYPE_W>();
        AscendC::DataCopy(wLocal, wGm, this->ksize);
        inQueueW.EnQue(wLocal);

        // Prepare output tile buffer by reading existing values (to accumulate)
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        uint32_t yStart = start;
        uint32_t yValid = AscendC::Min(this->tileOutLength, this->lout - yStart);
        if (yValid > 0) {
            AscendC::DataCopy(yLocal, yGm[yStart], yValid);
            if (yValid < this->tileOutLength) {
                AscendC::LocalTensor<DTYPE_Y> yPad = yLocal[yValid];
                AscendC::Duplicate(yPad, (DTYPE_Y)0, this->tileOutLength - yValid);
            }
        } else {
            AscendC::Duplicate(yLocal, (DTYPE_Y)0, this->tileOutLength);
        }
        outQueueY.EnQue(yLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        // Very simple diagonal-channel transposed conv:
        // y_tile[t + k] += x_tile[t] * w[k] for k in [0, ksize)
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.DeQue<DTYPE_W>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();

        // Compute per-element accumulation (naive UB loops)
        // Note: Using scalar-like access; actual AscendC may optimize differently.
        for (uint32_t t = 0; t < this->tileLength; ++t) {
            // load scalar x
            DTYPE_X xv = xLocal.GetValue(t);
            for (uint32_t k = 0; k < this->ksize; ++k) {
                DTYPE_W wk = wLocal.GetValue(k);
                DTYPE_Y acc = yLocal.GetValue(t + k);
                acc = acc + (DTYPE_Y)(xv * wk);
                yLocal.SetValue(t + k, acc);
            }
        }

        // recycle queues
        outQueueY.EnQue(yLocal);
        inQueueW.FreeTensor(wLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        uint32_t start = (uint32_t)progress * this->tileLength;
        uint32_t yValid = AscendC::Min(this->tileOutLength, this->lout - start);
        if (yValid > 0) {
            AscendC::DataCopy(yGm[start], yLocal, yValid);
        }
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, cin, cout, ccommon;
    uint32_t lin, lout, ksize;
    uint32_t nIdx, cIdx;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t tileOutLength;
};

extern "C" __global__ __aicore__ void conv_transposed_1d_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout, tiling_data.ccommon,
            tiling_data.lin, tiling_data.lout, tiling_data.ksize, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom implementation of transposed 1D convolution using AscendC kernel.
    Assumptions for the custom kernel:
      - stride=1, padding=0, dilation=1, groups=1
      - bias is not supported
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0, groups: int = 1, bias: bool = False):
        super().__init__()
        assert stride == 1, "Custom kernel currently supports stride=1"
        assert padding == 0, "Custom kernel currently supports padding=0"
        assert output_padding == 0, "Custom kernel currently supports output_padding=0"
        assert groups == 1, "Custom kernel currently supports groups=1"
        assert bias is False, "Custom kernel currently does not support bias"
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size

        # Weight layout matches PyTorch ConvTranspose1d: [Cin, Cout, K]
        w = torch.randn(in_channels, out_channels, kernel_size, dtype=torch.float32)
        self.weight = torch.nn.Parameter(w)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_1d_custom(x, self.weight)
