        return GRAPH_PARAM_INVALID;

    int64_t h = dims[2];
    int64_t w = dims[3];

    if (context->HasIntAttr("padding")) {

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // buffers per queue

// This kernel demonstrates a tiled average pooling operator using AscendC APIs.
// For simplicity, it treats the output as a flattened vector of length totalOutLength and
// performs a naive per-element computation within each tile.
class KernelAvgPool2D {
public:
    __aicore__ inline KernelAvgPool2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t out_h, uint32_t out_w,
                                uint32_t kernel_h, uint32_t kernel_w,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t totalOutLength, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->out_h = out_h;
        this->out_w = out_w;
        this->kernel_h = kernel_h;
        this->kernel_w = kernel_w;
        this->stride_h = stride_h;
        this->stride_w = stride_w;
        this->pad_h = pad_h;
        this->pad_w = pad_w;

        this->blockOutLength = totalOutLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileOutLength = this->blockOutLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, n * c * h * w);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockOutLength * AscendC::GetBlockIdx(), this->blockOutLength);

        // Initialize small buffers:
        // - outQueue for a tile of outputs
        // - inQueueRow for a single row segment fetch (up to kernel_w elements)
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileOutLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueRow, BUFFER_NUM, this->kernel_w * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            ComputeTile(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void ComputeTile(int32_t progress)
    {
        // Allocate a local buffer for tile outputs
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        uint32_t tileBase = progress * this->tileOutLength;

        for (uint32_t t = 0; t < this->tileOutLength; ++t) {
            uint32_t outLinear = tileBase + t + this->blockOutLength * AscendC::GetBlockIdx();
            if (outLinear >= n * c * out_h * out_w) {
                // Out-of-range guard
                yLocal.SetValue(t, (DTYPE_Y)0);
                continue;
            }

            // Decode output indices: n_idx, c_idx, oh, ow
            uint32_t ow = outLinear % out_w;
            uint32_t tmp1 = outLinear / out_w;
            uint32_t oh = tmp1 % out_h;
            uint32_t tmp2 = tmp1 / out_h;
            uint32_t c_idx = tmp2 % c;
            uint32_t n_idx = tmp2 / c;

            // Corresponding input top-left
            int32_t ih0 = static_cast<int32_t>(oh) * static_cast<int32_t>(stride_h) - static_cast<int32_t>(pad_h);
            int32_t iw0 = static_cast<int32_t>(ow) * static_cast<int32_t>(stride_w) - static_cast<int32_t>(pad_w);

            // Accumulate within the pooling window (with padding handling)
            float acc = 0.0f;
            uint32_t count = 0;

            for (uint32_t kh = 0; kh < kernel_h; ++kh) {
                int32_t ih = ih0 + static_cast<int32_t>(kh);
                if (ih < 0 || ih >= static_cast<int32_t>(h)) continue;

                // compute valid w-range for this row
                int32_t w_start = iw0;
                int32_t w_end = iw0 + static_cast<int32_t>(kernel_w);
                int32_t valid_w_start = w_start < 0 ? 0 : w_start;
                int32_t valid_w_end = w_end > static_cast<int32_t>(w) ? static_cast<int32_t>(w) : w_end;
                int32_t valid_len = valid_w_end - valid_w_start;
                if (valid_len <= 0) continue;

                // Allocate row-local buffer and copy contiguous segment
                AscendC::LocalTensor<DTYPE_X> rowLocal = inQueueRow.AllocTensor<DTYPE_X>();
                uint64_t baseIndex = (((uint64_t)n_idx * c + c_idx) * h + static_cast<uint32_t>(ih)) * w + static_cast<uint32_t>(valid_w_start);
                AscendC::DataCopy(rowLocal, xGm[baseIndex], static_cast<uint32_t>(valid_len));

                // Sum the row segment
                for (int32_t j = 0; j < valid_len; ++j) {
                    acc += static_cast<float>(rowLocal.GetValue(j));
                }
                count += static_cast<uint32_t>(valid_len);
                inQueueRow.FreeTensor(rowLocal);
            }

            float avg = count > 0 ? (acc / static_cast<float>(count)) : 0.0f;
            yLocal.SetValue(t, static_cast<DTYPE_Y>(avg));
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileOutLength], yLocal, this->tileOutLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueRow;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, h, w;
    uint32_t out_h, out_w;
    uint32_t kernel_h, kernel_w;
    uint32_t stride_h, stride_w;
    uint32_t pad_h, pad_w;

    uint32_t blockOutLength;
    uint32_t tileNum;
    uint32_t tileOutLength;
};

extern "C" __global__ __aicore__ void average_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAvgPool2D op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.out_h, tiling_data.out_w,
            tiling_data.kernel_h, tiling_data.kernel_w,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.totalOutLength, tiling_data.tileNum);
    op.Process();
}
