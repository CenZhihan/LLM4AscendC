kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using DTYPE = float;

constexpr float EPS = 1e-5f;

class KernelGemmBiasAddHardtanhMishGroupNorm {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNorm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias_linear, GM_ADDR bias_extra,
                                GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t M, uint32_t K, uint32_t O,
                                uint32_t numGroups, uint32_t channelsPerGroup,
                                uint32_t rowsPerBlock, float minVal, float maxVal)
    {
        this->M = M;
        this->K = K;
        this->O = O;
        this->numGroups = numGroups;
        this->channelsPerGroup = channelsPerGroup;
        this->rowsPerBlock = rowsPerBlock;
        this->minVal = minVal;
        this->maxVal = maxVal;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowCount = rowsPerBlock;
        if (rowStart + rowCount > M) {
            rowCount = (M > rowStart) ? (M - rowStart) : 0;
        }
        this->rowStart = rowStart;
        this->rowCount = rowCount;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x + (uint64_t)rowStart * K, (uint64_t)rowCount * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE*)w, (uint64_t)O * K);
        bias0Gm.SetGlobalBuffer((__gm__ DTYPE*)bias_linear, O);
        bias1Gm.SetGlobalBuffer((__gm__ DTYPE*)bias_extra, O);
        gammaGm.SetGlobalBuffer((__gm__ DTYPE*)gamma, O);
        betaGm.SetGlobalBuffer((__gm__ DTYPE*)beta, O);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + (uint64_t)rowStart * O, (uint64_t)rowCount * O);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowCount; ++r) {
            ComputeRow(r);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t rLocal)
    {
        // local scratch buffers in UB (conceptual)
        // Load x row r
        // For simplicity, load element-wise from GM during dot product

        // Temporary store for post-activation vector before groupnorm
        // We'll write out to GM after GroupNorm
        for (uint32_t o = 0; o < O; ++o) {
            // GEMM: y_val = dot(x_row, w[o, :])
            DTYPE acc = 0.0f;
            for (uint32_t k = 0; k < K; ++k) {
                DTYPE xv = xGm[rLocal * K + k];
                DTYPE wv = wGm[o * K + k]; // w shape (O, K)
                acc += xv * wv;
            }
            // Bias add (linear bias + extra bias)
            acc += bias0Gm[o];
            acc += bias1Gm[o];

            // Hardtanh
            if (acc < minVal) acc = minVal;
            if (acc > maxVal) acc = maxVal;

            // Mish: x * tanh(softplus(x)), softplus(x)=log(1+exp(x))
            DTYPE sp = log1pf(expf(acc));
            DTYPE mish = acc * tanhf(sp);

            // temporarily store back into yGm buffer; we'll normalize in-place
            yGm[rLocal * O + o] = mish;
        }

        // GroupNorm per row
        uint32_t G = numGroups;
        uint32_t C = O;
        uint32_t Cpg = channelsPerGroup;
        if (Cpg * G > C) {
            Cpg = (C + G - 1) / G; // ensure coverage
        }

        for (uint32_t g = 0; g < G; ++g) {
            uint32_t cStart = g * Cpg;
            if (cStart >= C) break;
            uint32_t cEnd = cStart + Cpg;
            if (cEnd > C) cEnd = C;
            uint32_t count = cEnd - cStart;

            // mean
            DTYPE mean = 0.0f;
            for (uint32_t c = cStart; c < cEnd; ++c) {
                mean += yGm[rLocal * O + c];
            }
            mean /= (DTYPE)count;

            // var
            DTYPE var = 0.0f;
            for (uint32_t c = cStart; c < cEnd; ++c) {
                DTYPE v = yGm[rLocal * O + c] - mean;
                var += v * v;
            }
            var /= (DTYPE)count;
            DTYPE invStd = rsqrtf(var + EPS);

            // normalize and affine
            for (uint32_t c = cStart; c < cEnd; ++c) {
                DTYPE v = yGm[rLocal * O + c];
                DTYPE norm = (v - mean) * invStd;
                DTYPE outv = norm * gammaGm[c] + betaGm[c];
                yGm[rLocal * O + c] = outv;
            }
        }
    }

private:
    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> bias0Gm;
    AscendC::GlobalTensor<DTYPE> bias1Gm;
    AscendC::GlobalTensor<DTYPE> gammaGm;
    AscendC::GlobalTensor<DTYPE> betaGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t M;
    uint32_t K;
    uint32_t O;
    uint32_t numGroups;
    uint32_t channelsPerGroup;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowCount;
    float minVal;
    float maxVal;
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR bias_linear, GM_ADDR bias_extra,
    GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNorm op;
    op.Init(x, w, bias_linear, bias_extra, gamma, beta, y,
            tiling_data.M, tiling_data.K, tiling_data.O,
            tiling_data.numGroups, tiling_data.channelsPerGroup,
            tiling_data.rowsPerBlock, tiling_data.minVal, tiling_data.maxVal);
    op.Process();
}
