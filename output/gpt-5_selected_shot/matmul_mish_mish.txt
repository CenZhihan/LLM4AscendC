kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// Fused MatMul + Bias + Mish + Mish
class KernelMatmulMishMish {
public:
    __aicore__ inline KernelMatmulMishMish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK, uint32_t blockM)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileM = tileM;
        this->tileN = tileN;
        this->tileK = tileK;
        this->rowsPerBlock = blockM;

        // shard rows by block
        uint32_t rowStart = AscendC::GetBlockIdx() * rowsPerBlock;
        if (rowStart > M) rowStart = M;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > M) rowEnd = M;
        this->rowStart = rowStart;
        this->rowEnd = rowEnd;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, M * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, N * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b, N);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, M * N);

        // initialize queue buffers
        // A tiles: tileM x tileK, B tiles: tileN x tileK, C tiles: tileM x tileN, Bias: tileN
        pipe.InitBuffer(inQueueA, BUFFER_NUM, tileM * tileK * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, tileN * tileK * sizeof(DTYPE_W));
        pipe.InitBuffer(inQueueBias, BUFFER_NUM, tileN * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, tileM * tileN * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t m0 = rowStart; m0 < rowEnd; m0 += tileM) {
            uint32_t curTileM = (m0 + tileM <= rowEnd) ? tileM : (rowEnd - m0);
            for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
                uint32_t curTileN = (n0 + tileN <= N) ? tileN : (N - n0);
                // Allocate C tile local buffer and zero init
                AscendC::LocalTensor<DTYPE_Z> cLocal = outQueueC.AllocTensor<DTYPE_Z>();
                // Zero initialization
                AscendC::Fill(cLocal, static_cast<DTYPE_Z>(0), curTileM * curTileN);

                // GEMM loop over K
                for (uint32_t k0 = 0; k0 < K; k0 += tileK) {
                    uint32_t curTileK = (k0 + tileK <= K) ? tileK : (K - k0);

                    // Load A tile (curTileM x curTileK)
                    AscendC::LocalTensor<DTYPE_X> aLocal = inQueueA.AllocTensor<DTYPE_X>();
                    // Load B tile (curTileN x curTileK)
                    AscendC::LocalTensor<DTYPE_W> bLocal = inQueueB.AllocTensor<DTYPE_W>();

                    // Copy A: rows m0..m0+curTileM, cols k0..k0+curTileK
                    for (uint32_t rm = 0; rm < curTileM; ++rm) {
                        AscendC::DataCopy(aLocal[rm * curTileK],
                                          xGm[(m0 + rm) * K + k0],
                                          curTileK);
                    }
                    // Copy B: rows n0..n0+curTileN, cols k0..k0+curTileK
                    for (uint32_t rn = 0; rn < curTileN; ++rn) {
                        AscendC::DataCopy(bLocal[rn * curTileK],
                                          wGm[(n0 + rn) * K + k0],
                                          curTileK);
                    }
                    inQueueA.EnQue(aLocal);
                    inQueueB.EnQue(bLocal);

                    // Compute partial C += A * B^T for this K tile
                    AscendC::LocalTensor<DTYPE_X> aTile = inQueueA.DeQue<DTYPE_X>();
                    AscendC::LocalTensor<DTYPE_W> bTile = inQueueB.DeQue<DTYPE_W>();

                    // Naive multiply-accumulate over curTileK
                    for (uint32_t rm = 0; rm < curTileM; ++rm) {
                        for (uint32_t rn = 0; rn < curTileN; ++rn) {
                            DTYPE_Z acc = 0;
                            for (uint32_t rk = 0; rk < curTileK; ++rk) {
                                acc += static_cast<DTYPE_Z>(aTile[rm * curTileK + rk]) *
                                       static_cast<DTYPE_Z>(bTile[rn * curTileK + rk]);
                            }
                            cLocal[rm * curTileN + rn] += acc;
                        }
                    }

                    inQueueA.FreeTensor(aTile);
                    inQueueB.FreeTensor(bTile);
                }

                // Add bias and apply Mish twice on cLocal
                // Load bias tile
                AscendC::LocalTensor<DTYPE_B> biasLocal = inQueueBias.AllocTensor<DTYPE_B>();
                AscendC::DataCopy(biasLocal, bGm[n0], curTileN);
                inQueueBias.EnQue(biasLocal);
                AscendC::LocalTensor<DTYPE_B> bTile = inQueueBias.DeQue<DTYPE_B>();

                // C += bias (broadcast over rows)
                for (uint32_t rm = 0; rm < curTileM; ++rm) {
                    for (uint32_t rn = 0; rn < curTileN; ++rn) {
                        cLocal[rm * curTileN + rn] =
                            static_cast<DTYPE_Z>(cLocal[rm * curTileN + rn] + static_cast<DTYPE_Z>(bTile[rn]));
                    }
                }

                inQueueBias.FreeTensor(bTile);

                // Mish: y = x * tanh(ln(1 + exp(x))) applied twice
                ApplyMish(cLocal, curTileM * curTileN);
                ApplyMish(cLocal, curTileM * curTileN);

                // Store C tile to GM
                for (uint32_t rm = 0; rm < curTileM; ++rm) {
                    AscendC::DataCopy(zGm[(m0 + rm) * N + n0],
                                      cLocal[rm * curTileN],
                                      curTileN);
                }
                outQueueC.FreeTensor(cLocal);
            }
        }
    }

private:
    __aicore__ inline void ApplyMish(AscendC::LocalTensor<DTYPE_Z>& x, uint32_t len)
    {
        // temp buffers
        // Allocate temporary local tensors via pipe if necessary; here we reuse a simple sequence with inplace ops
        // Compute s = tanh(ln(1 + exp(x)))
        for (uint32_t i = 0; i < len; ++i) {
            DTYPE_Z v = x[i];
            DTYPE_Z e = (DTYPE_Z)AscendC::Exp((DTYPE_Z)v);
            DTYPE_Z sp = (DTYPE_Z)AscendC::Ln((DTYPE_Z)(e + (DTYPE_Z)1.0f));
            DTYPE_Z t = (DTYPE_Z)AscendC::Tanh(sp);
            x[i] = v * t;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB, inQueueBias;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t M, N, K;
    uint32_t tileM, tileN, tileK;
    uint32_t rowsPerBlock;
    uint32_t rowStart, rowEnd;
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMishMish op;
    op.Init(x, w, b, z,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.blockM);
    op.Process();
}

class ModelNew(torch.nn.Module):

    """
