kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmSigmoidSumLogSumExp {
public:
    __aicore__ inline KernelGemmSigmoidSumLogSumExp() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w1, GM_ADDR b1, GM_ADDR w2, GM_ADDR b2, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t hiddenSize, uint32_t outFeatures,
                                uint32_t rowsPerBlock, uint32_t tileNum, uint32_t rowTile)
    {
        this->batchSize   = batchSize;
        this->inFeatures  = inFeatures;
        this->hiddenSize  = hiddenSize;
        this->outFeatures = outFeatures;
        this->rowsPerBlock= rowsPerBlock;
        this->tileNum     = tileNum;
        this->rowTile     = rowTile;

        uint32_t blockStartRow = AscendC::GetBlockIdx() * rowsPerBlock;
        xGm.SetGlobalBuffer((__gm__ DTYPE*)x  + (uint64_t)blockStartRow * inFeatures,  (uint64_t)rowsPerBlock * inFeatures);
        w1Gm.SetGlobalBuffer((__gm__ DTYPE*)w1, (uint64_t)hiddenSize * inFeatures);
        b1Gm.SetGlobalBuffer((__gm__ DTYPE*)b1, (uint64_t)hiddenSize);
        w2Gm.SetGlobalBuffer((__gm__ DTYPE*)w2, (uint64_t)outFeatures * hiddenSize);
        b2Gm.SetGlobalBuffer((__gm__ DTYPE*)b2, (uint64_t)outFeatures);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y  + (uint64_t)blockStartRow, rowsPerBlock);

        // Buffers for a tile of rows
        uint32_t xTileSize   = rowTile * inFeatures;
        uint32_t hiddenTile  = rowTile * hiddenSize;
        uint32_t outTileSize = rowTile * outFeatures;
        pipe.InitBuffer(inQueueX, BUFFER_NUM, xTileSize * sizeof(DTYPE));
        pipe.InitBuffer(tmpHiddenQ, BUFFER_NUM, hiddenTile * sizeof(DTYPE));
        pipe.InitBuffer(tmpOutQ, BUFFER_NUM, outTileSize * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, rowTile * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        int32_t loops = tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loops; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint64_t xOffsetRows = (uint64_t)(progress * rowTile);
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        uint64_t copyElems = (uint64_t)rowTile * inFeatures;
        if (xOffsetRows * inFeatures + copyElems > xGm.GetSize()) {
            copyElems = (xGm.GetSize() > xOffsetRows * inFeatures) ? (xGm.GetSize() - xOffsetRows * inFeatures) : 0;
        }
        AscendC::DataCopy(xLocal, xGm[xOffsetRows * inFeatures], copyElems);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        // Dequeue X tile
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();

        // Allocate temporaries
        AscendC::LocalTensor<DTYPE> hiddenLocal = tmpHiddenQ.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> outLocal    = tmpOutQ.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal      = outQueueY.AllocTensor<DTYPE>();

        // Dimensions
        uint32_t rows = rowTile;
        uint32_t M = rows;
        uint32_t K1 = inFeatures;
        uint32_t N1 = hiddenSize;
        uint32_t K2 = hiddenSize;
        uint32_t N2 = outFeatures;

        // Pseudo high-level compute steps with AscendC helper interfaces.
        // 1) GEMM1: [M,K1] x [N1,K1]^T -> [M,N1]
        // Assuming w1 is [N1,K1] contiguous, compute: hidden = xLocal * w1^T
        AscendC::MatMul(hiddenLocal, xLocal, w1Gm.GetGlobalData(), M, K1, N1, false, true);

        // 2) Add bias b1 (broadcast) and apply Sigmoid
        AscendC::BroadcastAdd(hiddenLocal, b1Gm.GetGlobalData(), hiddenLocal, M, N1);
        AscendC::Sigmoid(hiddenLocal, hiddenLocal, (uint64_t)M * N1);

        // 3) GEMM2: [M,K2] x [N2,K2]^T -> [M,N2]
        AscendC::MatMul(outLocal, hiddenLocal, w2Gm.GetGlobalData(), M, K2, N2, false, true);

        // 4) Add bias b2 (broadcast)
        AscendC::BroadcastAdd(outLocal, b2Gm.GetGlobalData(), outLocal, M, N2);

        // 5) LogSumExp along feature dimension (axis=1)
        // y_i = log(sum_j exp(outLocal[i,j]))
        // Use a numerically-stable variant via max-trick
        AscendC::ReduceLogSumExpRowWise(yLocal, outLocal, M, N2);

        // Enqueue outputs and free temps
        outQueueY.EnQue(yLocal);
        tmpOutQ.EnQue(outLocal);
        tmpHiddenQ.EnQue(hiddenLocal);

        inQueueX.FreeTensor(xLocal);
        tmpOutQ.FreeTensor(outLocal);
        tmpHiddenQ.FreeTensor(hiddenLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        uint64_t yOffset = (uint64_t)(progress * rowTile);
        uint64_t copyElems = rowTile;
        if (yOffset + copyElems > yGm.GetSize()) {
            copyElems = (yGm.GetSize() > yOffset) ? (yGm.GetSize() - yOffset) : 0;
        }
        AscendC::DataCopy(yGm[yOffset], yLocal, copyElems);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpHiddenQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpOutQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> w1Gm;
    AscendC::GlobalTensor<DTYPE> b1Gm;
    AscendC::GlobalTensor<DTYPE> w2Gm;
    AscendC::GlobalTensor<DTYPE> b2Gm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t batchSize;
    uint32_t inFeatures;
    uint32_t hiddenSize;
    uint32_t outFeatures;
    uint32_t rowsPerBlock;
    uint32_t tileNum;
    uint32_t rowTile;
};

extern "C" __global__ __aicore__ void gemm_sigmoid_sum_log_sum_exp_custom(GM_ADDR x, GM_ADDR w1, GM_ADDR b1, GM_ADDR w2, GM_ADDR b2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSigmoidSumLogSumExp op;
    op.Init(x, w1, b1, w2, b2, y,
            tiling_data.batchSize, tiling_data.inFeatures, tiling_data.hiddenSize, tiling_data.outFeatures,
            tiling_data.rowsPerBlock, tiling_data.tileNum, tiling_data.rowTile);
    op.Process();
}

class ModelNew(torch.nn.Module):

    """

        super(ModelNew, self).__init__()
        self.linear1 = torch.nn.Linear(input_size, hidden_size)
        self.linear2 = torch.nn.Linear(hidden_size, output_size)

        # Fused call: consumes x and both layer parameters, returns per-sample LogSumExp
        return custom_ops_lib.gemm_sigmoid_sum_log_sum_exp_custom(
            x,
            self.linear1.weight, self.linear1.bias,
