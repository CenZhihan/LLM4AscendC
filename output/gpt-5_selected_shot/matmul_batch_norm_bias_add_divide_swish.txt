#include "kernel_operator.h"
#include <cmath>

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(
        GM_ADDR x,
        GM_ADDR weight,
        GM_ADDR bias_linear,
        GM_ADDR bn_running_mean,
        GM_ADDR bn_running_var,
        GM_ADDR bn_gamma,
        GM_ADDR bn_beta,
        GM_ADDR custom_bias,
        GM_ADDR divide_value,
        GM_ADDR bn_eps,
        GM_ADDR y,
        uint32_t M, uint32_t N, uint32_t K, uint32_t tileNum)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileNum = tileNum;

        uint32_t total = M * N;
        uint32_t blocks = AscendC::GetBlockNum();
        this->blockStart = (total / blocks) * AscendC::GetBlockIdx();
        uint32_t baseLen = total / blocks;
        uint32_t rem = total % blocks;
        if (AscendC::GetBlockIdx() < rem) {
            this->blockStart += AscendC::GetBlockIdx();
            this->blockLen = baseLen + 1;
        } else {
            this->blockStart += rem;
            this->blockLen = baseLen;
        }

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint32_t)(M * K));
        wGm.SetGlobalBuffer((__gm__ float*)weight, (uint32_t)(N * K));
        bLinGm.SetGlobalBuffer((__gm__ float*)bias_linear, (uint32_t)(N));
        meanGm.SetGlobalBuffer((__gm__ float*)bn_running_mean, (uint32_t)(N));
        varGm.SetGlobalBuffer((__gm__ float*)bn_running_var, (uint32_t)(N));
        gammaGm.SetGlobalBuffer((__gm__ float*)bn_gamma, (uint32_t)(N));
        betaGm.SetGlobalBuffer((__gm__ float*)bn_beta, (uint32_t)(N));
        cBiasGm.SetGlobalBuffer((__gm__ float*)custom_bias, (uint32_t)(1));
        divValGm.SetGlobalBuffer((__gm__ float*)divide_value, (uint32_t)(1));
        epsGm.SetGlobalBuffer((__gm__ float*)bn_eps, (uint32_t)(1));
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint32_t)(M * N));
    }

    __aicore__ inline void Process()
    {
        if (this->blockLen == 0) {
            return;
        }
        float div_val = divValGm.GetValue(0);
        float eps_val = epsGm.GetValue(0);
        float c_bias = cBiasGm.GetValue(0);

        uint32_t start = this->blockStart;
        uint32_t end = start + this->blockLen;

        for (uint32_t idx = start; idx < end; ++idx) {
            uint32_t row = idx / N;
            uint32_t col = idx % N;

            // Matmul: (x @ W^T)[row, col] where weight is [N, K]
            float acc = 0.0f;
            uint32_t x_row_off = row * K;
            uint32_t w_row_off = col * K; // weight[col, :]
            for (uint32_t k = 0; k < K; ++k) {
                float xv = xGm.GetValue(x_row_off + k);
                float wv = wGm.GetValue(w_row_off + k);
                acc += xv * wv;
            }
            // Add linear bias
            acc += bLinGm.GetValue(col);

            // BatchNorm (inference): y = gamma * (acc - mean) / sqrt(var + eps) + beta
            float mean = meanGm.GetValue(col);
            float var = varGm.GetValue(col);
            float gamma = gammaGm.GetValue(col);
            float beta = betaGm.GetValue(col);
            float norm = (acc - mean) / sqrtf(var + eps_val);
            float bn_out = norm * gamma + beta;

            // Bias add + divide
            float v = (bn_out + c_bias) / div_val;

            // Swish: v * sigmoid(v)
            float sig = 1.0f / (1.0f + expf(-v));
            float outv = v * sig;

            yGm.SetValue(idx, outv);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bLinGm;
    AscendC::GlobalTensor<float> meanGm;
    AscendC::GlobalTensor<float> varGm;
    AscendC::GlobalTensor<float> gammaGm;
    AscendC::GlobalTensor<float> betaGm;
    AscendC::GlobalTensor<float> cBiasGm;
    AscendC::GlobalTensor<float> divValGm;
    AscendC::GlobalTensor<float> epsGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileNum;
    uint32_t blockStart;
    uint32_t blockLen;
};

extern "C" __global__ __aicore__ void matmul_batch_norm_bias_add_divide_swish_custom(
    GM_ADDR x,
    GM_ADDR weight,
    GM_ADDR bias_linear,
    GM_ADDR bn_running_mean,
    GM_ADDR bn_running_var,
    GM_ADDR bn_gamma,
    GM_ADDR bn_beta,
    GM_ADDR custom_bias,
    GM_ADDR divide_value,
    GM_ADDR bn_eps,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(
        x, weight, bias_linear, bn_running_mean, bn_running_var, bn_gamma, bn_beta, custom_bias, divide_value, bn_eps, y,
        tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused kernel: MatMul -> BatchNorm (inference) -> Bias Add -> Divide -> Swish
    """
    def __init__(self, in_features, out_features, bn_eps=1e-5, bn_momentum=0.1, bias_shape=(1,), divide_value=1.0):
        super().__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.bn = torch.nn.BatchNorm1d(out_features, eps=bn_eps, momentum=bn_momentum)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.divide_value = divide_value

    def forward(self, x):
        # Prepare scalar tensors for divide_value and bn_eps for the custom kernel
        divide_value_t = torch.tensor(self.divide_value, dtype=x.dtype, device=x.device)
        bn_eps_t = torch.tensor(self.bn.eps, dtype=x.dtype, device=x.device)

        # Call fused custom op
        return custom_ops_lib.matmul_batch_norm_bias_add_divide_swish_custom(
            x,
            self.matmul.weight,
            self.matmul.bias if self.matmul.bias is not None else torch.zeros(self.matmul.weight.size(0), dtype=x.dtype, device=x.device),
            self.bn.running_mean,
            self.bn.running_var,
            self.bn.weight,
            self.bn.bias,
            self.bias,
            divide_value_t,
            bn_eps_t
        )
