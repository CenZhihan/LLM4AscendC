kernel_src="""
#include "kernel_operator.h"

class KernelConvTransposed2D {
public:
    __aicore__ inline KernelConvTransposed2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t N, uint32_t IC, uint32_t OC,
                                uint32_t H, uint32_t W,
                                uint32_t KH, uint32_t KW,
                                uint32_t OH, uint32_t OW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW) {
        this->N = N; this->IC = IC; this->OC = OC;
        this->H = H; this->W = W;
        this->KH = KH; this->KW = KW;
        this->OH = OH; this->OW = OW;
        this->strideH = strideH; this->strideW = strideW;
        this->padH = padH; this->padW = padW;

        this->total = totalLength;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint64_t perBlock = total / blockNum;
        this->start = perBlock * blockIdx;
        this->end = (blockIdx == blockNum - 1) ? total : (this->start + perBlock);

        xPtr = (__gm__ DTYPE_X*)x;
        wPtr = (__gm__ DTYPE_W*)w;
        yPtr = (__gm__ DTYPE_Y*)y;
    }

    __aicore__ inline void Process() {
        for (uint64_t idx = start; idx < end; ++idx) {
            // decode flat index to (n, oc, oh, ow)
            uint64_t t = idx;
            uint32_t ow = t % OW; t /= OW;
            uint32_t oh = t % OH; t /= OH;
            uint32_t oc = t % OC; t /= OC;
            uint32_t n  = t % N;

            float acc = 0.0f;
            for (uint32_t ic = 0; ic < IC; ++ic) {
                for (uint32_t kh = 0; kh < KH; ++kh) {
                    int64_t num_h = (int64_t)oh + (int64_t)padH - (int64_t)kh;
                    if (num_h < 0) continue;
                    if (num_h % (int64_t)strideH != 0) continue;
                    int64_t ih = num_h / (int64_t)strideH;
                    if (ih < 0 || ih >= (int64_t)H) continue;

                    for (uint32_t kw = 0; kw < KW; ++kw) {
                        int64_t num_w = (int64_t)ow + (int64_t)padW - (int64_t)kw;
                        if (num_w < 0) continue;
                        if (num_w % (int64_t)strideW != 0) continue;
                        int64_t iw = num_w / (int64_t)strideW;
                        if (iw < 0 || iw >= (int64_t)W) continue;

                        uint64_t x_index = (((uint64_t)n * IC + ic) * H + (uint32_t)ih) * W + (uint32_t)iw;
                        uint64_t w_index = (((uint64_t)ic * OC + oc) * KH + kh) * KW + kw;

                        float xv = (float)xPtr[x_index];
                        float wv = (float)wPtr[w_index];
                        acc += xv * wv;
                    }
                }
            }
            yPtr[idx] = (DTYPE_Y)acc;
        }
    }

private:
    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_W* wPtr;
    __gm__ DTYPE_Y* yPtr;

    uint32_t N, IC, OC, H, W, KH, KW, OH, OW;
    uint32_t strideH, strideW, padH, padW;
    uint64_t total, start, end;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_padded_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2D op;
    op.Init(x, w, y,
            tiling_data.totalLength(),
            tiling_data.N(), tiling_data.IC(), tiling_data.OC(),
            tiling_data.H(), tiling_data.W(),
            tiling_data.KH(), tiling_data.KW(),
            tiling_data.OH(), tiling_data.OW(),
            tiling_data.strideH(), tiling_data.strideW(),
            tiling_data.padH(), tiling_data.padW());
    op.Process();
}

                                                               int64_t stride_h, int64_t stride_w,
                                                               int64_t pad_h, int64_t pad_w) {
    // x: [N, IC, H, W], w: [IC, OC, KH, KW]
    auto N = x.size(0);
    auto IC = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);
    auto ICw = w.size(0);
    auto OC = w.size(1);
    auto KH = w.size(2);
    auto KW = w.size(3);

    int64_t OH = (H - 1) * stride_h - 2 * pad_h + KH;
    int64_t OW = (W - 1) * stride_w - 2 * pad_w + KW;
    return {N, OC, OH, OW};
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom Model using AscendC kernel ConvTransposed2dAsymmetricInputAsymmetricKernelPaddedCustom
    to replace torch.nn.ConvTranspose2d for performance optimization.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: tuple = (1, 1), padding: tuple = (0, 0), bias: bool = False):
        super().__init__()
        assert bias is False, "Bias is not supported in this custom kernel example."
        if isinstance(kernel_size, int):
            kernel_size = (kernel_size, kernel_size)
        if isinstance(stride, int):
            stride = (stride, stride)
        if isinstance(padding, int):
            padding = (padding, padding)

        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding

        # Weight layout for ConvTranspose2d: [in_channels, out_channels, kH, kW]
        kH, kW = kernel_size
        w = torch.empty(in_channels, out_channels, kH, kW)
        torch.nn.init.kaiming_uniform_(w, a=5 ** 0.5)
        self.weight = torch.nn.Parameter(w)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        sh, sw = self.stride
        ph, pw = self.padding
        return custom_ops_lib.conv_transposed_2d_asymmetric_input_asymmetric_kernel_padded_custom(
            x, self.weight, int(sh), int(sw), int(ph), int(pw)
