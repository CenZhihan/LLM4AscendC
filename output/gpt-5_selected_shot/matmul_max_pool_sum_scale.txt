kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelRowReduceScale {
public:
    __aicore__ inline KernelRowReduceScale() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, uint32_t rowLength)
    {
        this->tileNum = tileNum;
        this->rowLength = rowLength;
        // one block processes one row
        this->blockLength = rowLength;
        // split into tiles across double-buffer
        this->tileLength = (rowLength + tileNum * BUFFER_NUM - 1) / (tileNum * BUFFER_NUM);
        this->tileLength = this->tileLength > 0 ? this->tileLength : 1;

        // Set GM buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)AscendC::GetBlockIdx() * (uint64_t)this->blockLength, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + AscendC::GetBlockIdx(), 1);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_SCALE *)scale, 1);

        // init queues
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));

        // load scale once
        AscendC::LocalTensor<DTYPE_SCALE> sLocal = pipe.AllocWorkspace<DTYPE_SCALE>(1);
        AscendC::DataCopy(sLocal, scaleGm[0], 1);
        scaleVal = sLocal.GetValue(0);
        pipe.FreeWorkspace(sLocal);

        acc = static_cast<DTYPE_ACC>(0);
    }

    __aicore__ inline void Process()
    {
        uint32_t processed = 0;
        int32_t iter = 0;
        while (processed < this->blockLength) {
            CopyIn(iter, processed);
            Compute(iter, processed);
            iter++;
        }
        // write result
        DTYPE_Y outVal = static_cast<DTYPE_Y>(acc * static_cast<DTYPE_ACC>(scaleVal));
        AscendC::LocalTensor<DTYPE_Y> yLocal = pipe.AllocWorkspace<DTYPE_Y>(1);
        yLocal.SetValue(0, outVal);
        AscendC::DataCopy(yGm[0], yLocal, 1);
        pipe.FreeWorkspace(yLocal);
    }

private:
    __aicore__ inline void CopyIn(int32_t progress, uint32_t processed)
    {
        uint32_t currLen = AscendC::Min(this->tileLength, this->blockLength - processed);
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[processed], currLen);
        xLocal.SetSize(currLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress, uint32_t &processed)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        uint32_t currLen = xLocal.GetSize();
        // sum reduction on the tile
        for (uint32_t i = 0; i < currLen; ++i) {
            acc += static_cast<DTYPE_ACC>(xLocal.GetValue(i));
        }
        inQueueX.FreeTensor(xLocal);
        processed += currLen;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_SCALE> scaleGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t rowLength;

    DTYPE_SCALE scaleVal;
    DTYPE_ACC acc;
};

// DTYPE definitions
using DTYPE_X = float;
using DTYPE_Y = float;
using DTYPE_SCALE = float;
using DTYPE_ACC = float;

extern "C" __global__ __aicore__ void matmul_max_pool_sum_scale_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelRowReduceScale op;
    op.Init(x, scale, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.rowLength);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keeps Linear and MaxPool on PyTorch,
    replaces sum + scaling with custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(ModelNew, self).__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.max_pool = torch.nn.MaxPool1d(kernel_size)
        self.scale_factor = scale_factor
        self.kernel_size = kernel_size

    def forward(self, x):
        x = self.matmul(x)                                  # (B, O)
        x = self.max_pool(x.unsqueeze(1)).squeeze(1)        # (B, O//k)
        scale = torch.tensor(self.scale_factor, device=x.device, dtype=x.dtype)
        return custom_ops_lib.matmul_max_pool_sum_scale_custom(x, scale)
