kernel_src="""
#include "kernel_operator.h"

using DTYPE_Q = float;
using DTYPE_K = float;
using DTYPE_V = float;
using DTYPE_O = float;

constexpr int32_t BUFFER_NUM = 2;
constexpr float EPS = 1e-6f;

class KernelLinearAttention {
public:
    __aicore__ inline KernelLinearAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR out, GM_ADDR workspace,
                                uint32_t batch, uint32_t qLen, uint32_t kLen, uint32_t dModel, uint32_t tileD)
    {
        this->batch = batch;
        this->qLen = qLen;
        this->kLen = kLen;
        this->dModel = dModel;
        this->tileD = tileD;

        // one block per batch
        uint32_t bIdx = AscendC::GetBlockIdx();
        uint64_t qStride = static_cast<uint64_t>(qLen) * dModel;
        uint64_t kStride = static_cast<uint64_t>(kLen) * dModel;
        uint64_t vStride = static_cast<uint64_t>(kLen) * dModel;
        uint64_t oStride = static_cast<uint64_t>(qLen) * dModel;
        uint64_t sStride = static_cast<uint64_t>(dModel) * dModel; // workspace (D x V)

        qGm.SetGlobalBuffer((__gm__ DTYPE_Q *)q + bIdx * qStride, qStride);
        kGm.SetGlobalBuffer((__gm__ DTYPE_K *)k + bIdx * kStride, kStride);
        vGm.SetGlobalBuffer((__gm__ DTYPE_V *)v + bIdx * vStride, vStride);
        oGm.SetGlobalBuffer((__gm__ DTYPE_O *)out + bIdx * oStride, oStride);
        sGm.SetGlobalBuffer((__gm__ DTYPE_O *)workspace + bIdx * sStride, sStride); // KV buffer

        // simple buffers for tileD
        pipe.InitBuffer(kQueue, BUFFER_NUM, this->tileD * sizeof(DTYPE_K));
        pipe.InitBuffer(qQueue, BUFFER_NUM, this->tileD * sizeof(DTYPE_Q));
        pipe.InitBuffer(sQueue, BUFFER_NUM, this->tileD * sizeof(DTYPE_O));
    }

    __aicore__ inline void Process()
    {
        // Phase 1: S = sum_k relu(K)+eps (outer) V -> shape [D, V]
        ComputeKV();

        // Phase 2: out = (relu(Q)+eps) @ S   -> [Q, V]
        ComputeQS();
    }

private:
    __aicore__ inline void ComputeKV()
    {
        // For each v in [0, dModel)
        for (uint32_t vcol = 0; vcol < dModel; ++vcol) {
            // for d in tiles of tileD
            for (uint32_t d0 = 0; d0 < dModel; d0 += tileD) {
                uint32_t cur = AscendC::min(tileD, dModel - d0);
                AscendC::LocalTensor<DTYPE_O> accLocal = sQueue.AllocTensor<DTYPE_O>();
                // zero init accumulator
                for (uint32_t i = 0; i < cur; ++i) {
                    accLocal.GetBuffer()[i] = 0.0f;
                }

                // accumulate over k dimension
                for (uint32_t kidx = 0; kidx < kLen; ++kidx) {
                    // load K tile
                    AscendC::LocalTensor<DTYPE_K> kLocal = kQueue.AllocTensor<DTYPE_K>();
                    AscendC::DataCopy(kLocal, kGm[kidx * dModel + d0], cur);

                    // apply relu + eps to K
                    for (uint32_t i = 0; i < cur; ++i) {
                        DTYPE_K kv = kLocal.GetBuffer()[i];
                        if (kv < (DTYPE_K)0) kv = 0;
                        kLocal.GetBuffer()[i] = kv + (DTYPE_K)EPS;
                    }

                    // read scalar V[kidx, vcol]
                    DTYPE_V v_scalar = vGm.GetValue(kidx * dModel + vcol);

                    // acc += kLocal * v_scalar
                    for (uint32_t i = 0; i < cur; ++i) {
                        accLocal.GetBuffer()[i] += (DTYPE_O)(kLocal.GetBuffer()[i] * (DTYPE_O)v_scalar);
                    }
                    kQueue.FreeTensor(kLocal);
                }

                // write back S[d0:d0+cur, vcol]
                for (uint32_t i = 0; i < cur; ++i) {
                    sGm.SetValue((d0 + i) * dModel + vcol, accLocal.GetBuffer()[i]);
                }
                sQueue.FreeTensor(accLocal);
            }
        }
    }

    __aicore__ inline void ComputeQS()
    {
        // For each query position q in [0, qLen)
        for (uint32_t qidx = 0; qidx < qLen; ++qidx) {
            // For each v column, compute dot over d
            for (uint32_t vcol = 0; vcol < dModel; ++vcol) {
                DTYPE_O sum = 0.0f;
                for (uint32_t d0 = 0; d0 < dModel; d0 += tileD) {
                    uint32_t cur = AscendC::min(tileD, dModel - d0);

                    // load Q tile
                    AscendC::LocalTensor<DTYPE_Q> qLocal = qQueue.AllocTensor<DTYPE_Q>();
                    AscendC::DataCopy(qLocal, qGm[qidx * dModel + d0], cur);

                    // apply relu + eps to Q
                    for (uint32_t i = 0; i < cur; ++i) {
                        DTYPE_Q qv = qLocal.GetBuffer()[i];
                        if (qv < (DTYPE_Q)0) qv = 0;
                        qLocal.GetBuffer()[i] = qv + (DTYPE_Q)EPS;
                    }

                    // load S tile column for vcol
                    AscendC::LocalTensor<DTYPE_O> sLocal = sQueue.AllocTensor<DTYPE_O>();
                    for (uint32_t i = 0; i < cur; ++i) {
                        sLocal.GetBuffer()[i] = sGm.GetValue((d0 + i) * dModel + vcol);
                    }

                    // accumulate dot product
                    for (uint32_t i = 0; i < cur; ++i) {
                        sum += (DTYPE_O)qLocal.GetBuffer()[i] * sLocal.GetBuffer()[i];
                    }
                    qQueue.FreeTensor(qLocal);
                    sQueue.FreeTensor(sLocal);
                }
                oGm.SetValue(qidx * dModel + vcol, sum);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> kQueue, qQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> sQueue;

    AscendC::GlobalTensor<DTYPE_Q> qGm;
    AscendC::GlobalTensor<DTYPE_K> kGm;
    AscendC::GlobalTensor<DTYPE_V> vGm;
    AscendC::GlobalTensor<DTYPE_O> oGm;
    AscendC::GlobalTensor<DTYPE_O> sGm; // workspace: [D, V]

    uint32_t batch;
    uint32_t qLen;
    uint32_t kLen;
    uint32_t dModel;
    uint32_t tileD;
};

extern "C" __global__ __aicore__ void linear_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLinearAttention op;
    op.Init(q, k, v, out, workspace,
            tiling_data.batch, tiling_data.qLen, tiling_data.kLen, tiling_data.dModel, tiling_data.tileD);
    op.Process();
}
