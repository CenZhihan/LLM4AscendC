kernel_src="""
#include "kernel_operator.h"
#include <cmath>

extern "C" __global__ __aicore__ void conv2d_gelu_global_avg_pool_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    // Dimensions from tiling
    const uint32_t N    = tiling_data.N;
    const uint32_t Cin  = tiling_data.Cin;
    const uint32_t H    = tiling_data.H;
    const uint32_t W    = tiling_data.W;
    const uint32_t Cout = tiling_data.Cout;
    const uint32_t Kh   = tiling_data.Kh;
    const uint32_t Kw   = tiling_data.Kw;
    const uint32_t strideH = tiling_data.strideH;
    const uint32_t strideW = tiling_data.strideW;
    const uint32_t padH = tiling_data.padH;
    const uint32_t padW = tiling_data.padW;

    const uint32_t H_out = (H + 2 * padH - Kh) / strideH + 1;
    const uint32_t W_out = (W + 2 * padW - Kw) / strideW + 1;
    const uint32_t HW_out = H_out * W_out;

    // total (N * Cout) work-items
    const uint32_t totalLength = tiling_data.totalLength;
    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();

    const uint32_t workPerBlock = (totalLength + blockNum - 1) / blockNum;
    const uint32_t startIdx = blockIdx * workPerBlock;
    const uint32_t endIdx = AscendC::Min<uint32_t>(startIdx + workPerBlock, totalLength);

    __gm__ float* x_gm = reinterpret_cast<__gm__ float*>(x); // [N, Cin, H, W]
    __gm__ float* w_gm = reinterpret_cast<__gm__ float*>(w); // [Cout, Cin, Kh, Kw]
    __gm__ float* b_gm = reinterpret_cast<__gm__ float*>(b); // [Cout]
    __gm__ float* y_gm = reinterpret_cast<__gm__ float*>(y); // [N, Cout]

    auto idx_input = [Cin, H, W](uint32_t n, uint32_t c, uint32_t h, uint32_t wi) -> uint64_t {
        return (((uint64_t)n * Cin + c) * H + h) * W + wi;
    };
    auto idx_weight = [Cin, Kh, Kw](uint32_t co, uint32_t ci, uint32_t kh, uint32_t kw) -> uint64_t {
        return (((uint64_t)co * Cin + ci) * Kh + kh) * Kw + kw;
    };
    auto idx_output = [Cout](uint32_t n, uint32_t co) -> uint64_t {
        return (uint64_t)n * Cout + co;
    };

    for (uint32_t idx = startIdx; idx < endIdx; ++idx) {
        uint32_t n = idx / Cout;
        uint32_t co = idx % Cout;

        float accum = 0.0f;
        const float bias_val = b_gm[co];

        for (uint32_t oh = 0; oh < H_out; ++oh) {
            int32_t in_h_base = (int32_t)oh * (int32_t)strideH - (int32_t)padH;
            for (uint32_t ow = 0; ow < W_out; ++ow) {
                int32_t in_w_base = (int32_t)ow * (int32_t)strideW - (int32_t)padW;
                float conv_val = 0.0f;

                // Convolution sum
                for (uint32_t ci = 0; ci < Cin; ++ci) {
                    for (uint32_t kh = 0; kh < Kh; ++kh) {
                        int32_t ih = in_h_base + (int32_t)kh;
                        if (ih < 0 || ih >= (int32_t)H) continue;
                        for (uint32_t kw = 0; kw < Kw; ++kw) {
                            int32_t iw = in_w_base + (int32_t)kw;
                            if (iw < 0 || iw >= (int32_t)W) continue;
                            float xv = x_gm[idx_input(n, ci, (uint32_t)ih, (uint32_t)iw)];
                            float ww = w_gm[idx_weight(co, ci, kh, kw)];
                            conv_val += xv * ww;
                        }
                    }
                }
                conv_val += bias_val;

                // GELU (approx) using tanh approximation
                const float kAlpha = 0.7978845608f;     // sqrt(2/pi)
                const float kBeta  = 0.044715f;
                float x3 = conv_val * conv_val * conv_val;
                float gelu_val = 0.5f * conv_val * (1.0f + tanhf(kAlpha * (conv_val + kBeta * x3)));

                accum += gelu_val;
            }
        }

        float avg_val = accum / static_cast<float>(HW_out);
        y_gm[idx_output(n, co)] = avg_val;
    }
}
