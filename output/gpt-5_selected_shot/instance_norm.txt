host_operator_src="""
#include "instance_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 48;

static ge::graphStatus TilingFunc(gert::TilingContext* context) {
  InstanceNormCustomTilingData tiling;
  const gert::StorageShape* x1_shape = context->GetInputShape(0);
  const gert::Shape shape = x1_shape->GetStorageShape();

  uint64_t n = 1, c = 1, h = 1, w = 1;
  if (shape.GetDimNum() >= 4) {
    n = static_cast<uint64_t>(shape.GetDim(0));
    c = static_cast<uint64_t>(shape.GetDim(1));
    h = static_cast<uint64_t>(shape.GetDim(2));
    w = static_cast<uint64_t>(shape.GetDim(3));
  } else {
    // Fallback for non-4D shapes: treat first dim as rows, rest as cols
    uint64_t rows = static_cast<uint64_t>(shape.GetDim(0));
    uint64_t cols = 1;
    for (size_t i = 1; i < shape.GetDimNum(); ++i) cols *= static_cast<uint64_t>(shape.GetDim(i));
    n = rows; c = 1; h = cols; w = 1;
  }

  uint64_t rowAll = n * c;
  uint64_t colNum = h * w;
  uint32_t coreRowNum = static_cast<uint32_t>(rowAll / BLOCK_DIM);

  const float* epsilonAttr = context->GetAttrs()->GetAttrPointer<float>(0);
  tiling.set_eps(*epsilonAttr);
  tiling.set_rowNum(coreRowNum);
  tiling.set_rowNumSp(coreRowNum + 1);
  tiling.set_rowLength(static_cast<uint32_t>(colNum));
  tiling.set_blockPivot(static_cast<uint32_t>(rowAll - static_cast<uint64_t>(coreRowNum) * BLOCK_DIM));
  uint32_t tileLoop = 32; // must be <= temporary buffer length
  tiling.set_tileLoop(tileLoop);
  tiling.set_tileLength(tileLoop * static_cast<uint32_t>(colNum));
  tiling.set_loopCount(coreRowNum / tileLoop);
  tiling.set_factor(1.0f / static_cast<float>(colNum));
  tiling.set_mfactor(-1.0f / static_cast<float>(colNum));
  tiling.set_channels(static_cast<uint32_t>(c));

  context->SetBlockDim(BLOCK_DIM);
  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
  context->SetTilingKey(1);

  return ge::GRAPH_SUCCESS;
}
}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context) {
  const gert::Shape* x1_shape = context->GetInputShape(0);
  gert::Shape* y_shape = context->GetOutputShape(0);
  *y_shape = *x1_shape;
  return GRAPH_SUCCESS;
}
}  // namespace ge

namespace ops {
class InstanceNormCustom : public OpDef {

  explicit InstanceNormCustom(const char* name) : OpDef(name) {
    this->Input("x")

    this->Input("beta")
        .ParamType(REQUIRED)

        .UnknownShapeFormat({ge::FORMAT_ND});
    this->Output("res_out")

        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})

kernel_src="""
#include "kernel_operator.h"
constexpr int32_t BUFFER_NUM = 1;

class KernelInstanceNorm {
 public:
  __aicore__ inline KernelInstanceNorm() {}
  __aicore__ inline void InitTiling(GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    rowNum = tiling_data.rowNum;
    rowNumSp = tiling_data.rowNumSp;
    rowLength = tiling_data.rowLength;
    blockPivot = tiling_data.blockPivot;
    tileLoop = tiling_data.tileLoop;
    tileLength = tiling_data.tileLength;
    loopCount = tiling_data.loopCount;
    factor = tiling_data.factor;
    mfactor = tiling_data.mfactor;
    eps = tiling_data.eps;
    channels = tiling_data.channels;
  }
  __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR z,
                              GM_ADDR tiling) {
    InitTiling(tiling);

    this->leftRow = this->rowNum % this->tileLoop;
    if (AscendC::GetBlockIdx() < this->blockPivot) {
      this->rowNum = this->rowNumSp;
      this->leftRow += 1;
    }

    // compute base row index (number of (n,c) rows before this block)
    if (AscendC::GetBlockIdx() < this->blockPivot) {
      baseRow = AscendC::GetBlockIdx() * this->rowNumSp;
    } else {
      baseRow = this->blockPivot * this->rowNumSp +
                (AscendC::GetBlockIdx() - this->blockPivot) * (this->rowNum);
    }

    this->blockLength = this->rowNum * this->rowLength;
    uint32_t offset = 0;
    if (AscendC::GetBlockIdx() < this->blockPivot) {
      offset = this->blockLength * AscendC::GetBlockIdx();
    } else {
      offset = this->blockLength * AscendC::GetBlockIdx() +
               this->rowLength * this->blockPivot;
    }

    xGm.SetGlobalBuffer((__gm__ float *)x + offset, this->blockLength);
    zGm.SetGlobalBuffer((__gm__ float *)z + offset, this->blockLength);

    gammaGm.SetGlobalBuffer((__gm__ float *)gamma, this->channels);
    betaGm.SetGlobalBuffer((__gm__ float *)beta, this->channels);

    pipe.InitBuffer(queueX, BUFFER_NUM, this->tileLength * sizeof(float));
    pipe.InitBuffer(queueZ, BUFFER_NUM, this->tileLength * sizeof(float));

    pipe.InitBuffer(tmpBuffer1, 64 * sizeof(float));
    pipe.InitBuffer(tmpBuffer2, 64 * sizeof(float));
    pipe.InitBuffer(onesBuffer, 64 * sizeof(float));

    pipe.InitBuffer(queueGamma, 1, this->channels * sizeof(float));
    pipe.InitBuffer(queueBeta, 1, this->channels * sizeof(float));
  }
  __aicore__ inline void Process() {
    for (int32_t i = 0; i < this->loopCount; i++) {
      CopyIn(i, this->tileLoop);
      Compute(i, this->tileLoop);
      CopyOut(i, this->tileLoop);
    }
    if (this->leftRow > 0) {
      CopyIn(this->loopCount, this->leftRow);
      Compute(this->loopCount, this->leftRow);
      CopyOut(this->loopCount, this->leftRow);
    }
  }

 private:
  __aicore__ inline void CopyIn(int32_t progress, int32_t rowNumTile) {
    AscendC::LocalTensor<float> xLocal = queueX.AllocTensor<float>();
    AscendC::LocalTensor<float> gammaLocal = queueGamma.AllocTensor<float>();
    AscendC::LocalTensor<float> betaLocal = queueBeta.AllocTensor<float>();
    AscendC::DataCopy(xLocal, xGm[progress * this->tileLength],
             this->rowLength * rowNumTile);
    AscendC::DataCopy(gammaLocal, gammaGm[0], this->channels);
    AscendC::DataCopy(betaLocal, betaGm[0], this->channels);
    queueX.EnQue(xLocal);
    queueGamma.EnQue(gammaLocal);
    queueBeta.EnQue(betaLocal);
  }

  __aicore__ inline void Compute(int32_t progress, int32_t rowNumTile) {
    AscendC::LocalTensor<float> xLocal = queueX.DeQue<float>();
    AscendC::LocalTensor<float> gammaLocal = queueGamma.DeQue<float>();
    AscendC::LocalTensor<float> betaLocal = queueBeta.DeQue<float>();

    AscendC::LocalTensor<float> tmpTensor1 = tmpBuffer1.Get<float>();
    AscendC::LocalTensor<float> tmpTensor2 = tmpBuffer2.Get<float>();
    AscendC::LocalTensor<float> onesLocal = onesBuffer.Get<float>();
    AscendC::LocalTensor<float> zLocal = queueZ.AllocTensor<float>();
    AscendC::Duplicate<float>(onesLocal, 1.0f, this->tileLoop);

    // compute per-row mean
    for (uint32_t j = 0; j < (uint32_t)rowNumTile; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      AscendC::ReduceSum<float>(tmpTensor2[j], xLocal[buffIndex], tmpTensor1,
                       this->rowLength);
    }
    AscendC::Muls(zLocal, tmpTensor2, this->mfactor, rowNumTile);

    // center
    for (uint32_t j = 0; j < (uint32_t)rowNumTile; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      AscendC::Adds(xLocal[buffIndex], xLocal[buffIndex], zLocal.GetValue(j),
           this->rowLength);
    }

    // variance
    for (uint32_t j = 0; j < (uint32_t)rowNumTile; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      AscendC::Mul(zLocal[buffIndex], xLocal[buffIndex], xLocal[buffIndex],
          this->rowLength);
    }
    for (uint32_t j = 0; j < (uint32_t)rowNumTile; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      AscendC::ReduceSum<float>(tmpTensor2[j], zLocal[buffIndex], tmpTensor1,
                       this->rowLength);
    }
    AscendC::Muls(tmpTensor2, tmpTensor2, this->factor, rowNumTile);
    AscendC::Adds(tmpTensor2, tmpTensor2, this->eps, rowNumTile);
    AscendC::Sqrt(tmpTensor2, tmpTensor2, rowNumTile);
    AscendC::Div(tmpTensor2, onesLocal, tmpTensor2, rowNumTile); // inverse std

    // normalize and affine per-channel
    for (uint32_t j = 0; j < (uint32_t)rowNumTile; ++j) {
      uint32_t buffIndex = j * this->rowLength;
      // normalize
      AscendC::Muls(zLocal[buffIndex], xLocal[buffIndex], tmpTensor2.GetValue(j),
           this->rowLength);

      // compute channel index for this row
      uint32_t rowGlobal = this->baseRow + (uint32_t)progress * this->tileLoop + j;
      uint32_t ch = (this->channels == 0) ? 0 : (rowGlobal % this->channels);

      // apply gamma and beta scalars
      AscendC::Muls(zLocal[buffIndex], zLocal[buffIndex], gammaLocal.GetValue(ch),
           this->rowLength);
      AscendC::Adds(zLocal[buffIndex], zLocal[buffIndex], betaLocal.GetValue(ch),
           this->rowLength);
    }

    queueZ.EnQue<float>(zLocal);
    queueGamma.FreeTensor(gammaLocal);
    queueBeta.FreeTensor(betaLocal);
    queueX.FreeTensor(xLocal);
  }

  __aicore__ inline void CopyOut(int32_t progress, int32_t rowNumTile) {
    AscendC::LocalTensor<float> zLocal = queueZ.DeQue<float>();

    AscendC::DataCopy(zGm[progress * this->tileLength], zLocal,
             rowNumTile * this->rowLength);

    queueZ.FreeTensor(zLocal);
  }

 private:
  AscendC::TPipe pipe;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> tmpBuffer1, tmpBuffer2, onesBuffer;
  AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> queueX;
  AscendC::TQue<AscendC::QuePosition::VECIN, 1> queueGamma, queueBeta;
  AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> queueZ;
  AscendC::GlobalTensor<float> xGm;
  AscendC::GlobalTensor<float> gammaGm;
  AscendC::GlobalTensor<float> betaGm;
  AscendC::GlobalTensor<float> zGm;

  uint32_t blockLength = 0;
  uint32_t leftRow = 0;
  uint32_t rowNum = 0;
  uint32_t rowNumSp = 0;
  uint32_t rowLength = 0;
  uint32_t blockPivot = 0;
  uint32_t tileLoop = 0;
  uint32_t tileLength = 0;
  uint32_t loopCount = 0;
  uint32_t channels = 1;
  uint32_t baseRow = 0;
  float factor = 0.f;
  float mfactor = 0.f;
  float eps = 1e-5f;
};
extern "C" __global__ __aicore__ void instance_norm_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR res_out, GM_ADDR workspace,
    GM_ADDR tiling) {
  KernelInstanceNorm op;
  op.Init(x, gamma, beta, res_out, tiling);
  if (TILING_KEY_IS(1)) {
    op.Process();
  }
}
