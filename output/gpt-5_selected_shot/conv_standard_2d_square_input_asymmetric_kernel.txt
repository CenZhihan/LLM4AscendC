kernel_src="""
#include "kernel_operator.h"

// Naive direct 2D convolution kernel (NCHW, groups=1) for illustration.
// Defaults: stride=1, pad=0, dilation=1. Bias not supported in this example.

class KernelConv2D {
public:
    __aicore__ inline KernelConv2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t outH, uint32_t outW)
    {
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)N*C*H*W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)w, (uint64_t)Cout*C*Kh*Kw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint64_t)N*Cout*outH*outW);

        this->N = N; this->C = C; this->H = H; this->W = W;
        this->Cout = Cout; this->Kh = Kh; this->Kw = Kw;
        this->strideH = strideH; this->strideW = strideW;
        this->padH = padH; this->padW = padW;
        this->dilationH = dilationH; this->dilationW = dilationW;
        this->outH = outH; this->outW = outW;
    }

    __aicore__ inline void Process()
    {
        // Simple single-block execution for correctness illustration
        if (AscendC::GetBlockIdx() != 0) {
            return;
        }

        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oc = 0; oc < Cout; ++oc) {
                for (uint32_t oh = 0; oh < outH; ++oh) {
                    for (uint32_t ow = 0; ow < outW; ++ow) {
                        float acc = 0.f;
                        // Convolution sum over input channels and kernel window
                        for (uint32_t ic = 0; ic < C; ++ic) {
                            for (uint32_t kh = 0; kh < Kh; ++kh) {
                                for (uint32_t kw = 0; kw < Kw; ++kw) {
                                    int32_t ih = static_cast<int32_t>(oh) * static_cast<int32_t>(strideH)
                                               - static_cast<int32_t>(padH)
                                               + static_cast<int32_t>(kh) * static_cast<int32_t>(dilationH);
                                    int32_t iw = static_cast<int32_t>(ow) * static_cast<int32_t>(strideW)
                                               - static_cast<int32_t>(padW)
                                               + static_cast<int32_t>(kw) * static_cast<int32_t>(dilationW);

                                    if (ih >= 0 && iw >= 0 && ih < static_cast<int32_t>(H) && iw < static_cast<int32_t>(W)) {
                                        uint64_t xIdx = ((uint64_t)n * C + ic) * (uint64_t)H * (uint64_t)W
                                                      + (uint64_t)ih * (uint64_t)W + (uint64_t)iw;
                                        uint64_t wIdx = ((uint64_t)oc * C + ic) * (uint64_t)Kh * (uint64_t)Kw
                                                      + (uint64_t)kh * (uint64_t)Kw + (uint64_t)kw;

                                        float xv = static_cast<float>(xGm.GetValue(xIdx));
                                        float wv = static_cast<float>(wGm.GetValue(wIdx));
                                        acc += xv * wv;
                                    }
                                }
                            }
                        }
                        uint64_t yIdx = ((uint64_t)n * Cout + oc) * (uint64_t)outH * (uint64_t)outW
                                      + (uint64_t)oh * (uint64_t)outW + (uint64_t)ow;
                        yGm.SetValue(yIdx, static_cast<DTYPE_Z>(acc));
                    }
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N, C, H, W;
    uint32_t Cout, Kh, Kw;
    uint32_t strideH, strideW, padH, padW, dilationH, dilationW;
    uint32_t outH, outW;
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2D op;
    op.Init(x, w, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.Cout, tiling_data.Kh, tiling_data.Kw,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.outH, tiling_data.outW);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super().__init__()
        # Use nn.Conv2d to manage parameters (weight/bias) and shapes
        self.conv2d = torch.nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, groups=groups, bias=bias)
        if bias:
            torch.nn.init.zeros_(self.conv2d.bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Extract convolution parameters
        stride_h, stride_w = (self.conv2d.stride if isinstance(self.conv2d.stride, tuple) else (self.conv2d.stride, self.conv2d.stride))
        if isinstance(self.conv2d.padding, tuple):
            pad_h, pad_w = self.conv2d.padding
        else:
            pad_h = pad_w = self.conv2d.padding
        dilation_h, dilation_w = (self.conv2d.dilation if isinstance(self.conv2d.dilation, tuple) else (self.conv2d.dilation, self.conv2d.dilation))

        # Call the custom kernel (bias not supported in this example)
        y = custom_ops_lib.conv_standard_2d_square_input_asymmetric_kernel_custom(
            x, self.conv2d.weight, stride_h, stride_w, pad_h, pad_w, dilation_h, dilation_w, groups
