kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConvTranspose3dAddHardSwish {
public:
    __aicore__ inline KernelConvTranspose3dAddHardSwish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        addGm.SetGlobalBuffer((__gm__ DTYPE_Y *)add + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueAdd, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> addLocal = inQueueAdd.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(addLocal, addGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueAdd.EnQue(addLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> addLocal = inQueueAdd.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // y = x + add
        AscendC::Add(zLocal, xLocal, addLocal, this->tileLength);

        // Reuse xLocal and addLocal as temporaries after their original usage:
        // xLocal = y (copy)
        AscendC::DataCopy(xLocal, zLocal, this->tileLength);
        // addLocal = y (copy)
        AscendC::DataCopy(addLocal, zLocal, this->tileLength);

        // xLocal = y * y
        AscendC::Mul(xLocal, xLocal, xLocal, this->tileLength);

        // addLocal = clamp(y + 3, 0, 6)
        AscendC::Adds(addLocal, addLocal, static_cast<DTYPE_Y>(3.0f), this->tileLength);
        AscendC::Maxs(addLocal, addLocal, static_cast<DTYPE_Y>(0.0f), this->tileLength);
        AscendC::Mins(addLocal, addLocal, static_cast<DTYPE_Y>(6.0f), this->tileLength);

        // zLocal = (y*y) * clamp / 6
        AscendC::Mul(zLocal, xLocal, addLocal, this->tileLength);
        AscendC::Muls(zLocal, zLocal, static_cast<DTYPE_Z>(1.0f / 6.0f), this->tileLength);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueAdd.FreeTensor(addLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueAdd;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> addGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_add_hard_swish_custom(GM_ADDR x, GM_ADDR add, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dAddHardSwish op;
    op.Init(x, add, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: uses PyTorch ConvTranspose3d followed by a custom fused AscendC op that
    computes (conv_out + add_input) * hardswish(conv_out + add_input).
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels, out_channels, kernel_size,
            stride=stride, padding=padding, output_padding=output_padding
        )
        # Keep bias parameter to align with original signature, though not used in forward here.
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x, add_input):
        conv_out = self.conv_transpose(x)
