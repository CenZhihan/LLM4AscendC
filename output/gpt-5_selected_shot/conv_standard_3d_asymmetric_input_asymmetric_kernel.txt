kernel_src="""
#include "kernel_operator.h"

// Simple illustrative kernel for standard 3D convolution with stride=1, pad=0, dilation=1, groups=1.
// It tiles across the flattened output domain.

constexpr int32_t BUFFER_NUM = 2;

class KernelConv3D {
public:
    __aicore__ inline KernelConv3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t din, uint32_t hin, uint32_t win,
                                uint32_t cout, uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout, uint32_t tileNum) {
        this->n = n;
        this->cin = cin;
        this->din = din;
        this->hin = hin;
        this->win = win;
        this->cout = cout;
        this->kd = kd;
        this->kh = kh;
        this->kw = kw;
        this->dout = dout;
        this->hout = hout;
        this->wout = wout;

        this->tileNum = tileNum;

        uint64_t totalOut = (uint64_t)n * cout * dout * hout * wout;
        this->blockLength = totalOut / AscendC::GetBlockNum();
        this->tileLength = (this->blockLength == 0 ? 0 : this->blockLength / tileNum / BUFFER_NUM);
        if (this->tileLength == 0 && totalOut > 0) {
            this->tileLength = 1;
            this->blockLength = totalOut / AscendC::GetBlockNum();
            if (this->blockLength == 0) this->blockLength = totalOut;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)n * cin * din * hin * win);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)cout * cin * kd * kh * kw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process() {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress) {
        uint64_t startIndex = (uint64_t)AscendC::GetBlockIdx() * this->blockLength + (uint64_t)progress * this->tileLength;
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        for (uint32_t t = 0; t < this->tileLength; ++t) {
            uint64_t outIndex = startIndex + t;
            if (outIndex >= (uint64_t)n * cout * dout * hout * wout) {
                break;
            }

            // Unflatten outIndex -> (n_o, co, do, ho, wo) in row-major order
            uint64_t remaining = outIndex;
            uint32_t wo = remaining % wout; remaining /= wout;
            uint32_t ho = remaining % hout; remaining /= hout;
            uint32_t do_ = remaining % dout; remaining /= dout;
            uint32_t co = remaining % cout; remaining /= cout;
            uint32_t no = remaining;

            // Accumulator
            DTYPE_Y acc = (DTYPE_Y)0;

            // Compute convolution for stride=1, pad=0, dilation=1
            for (uint32_t ci = 0; ci < cin; ++ci) {
                for (uint32_t kd_i = 0; kd_i < kd; ++kd_i) {
                    uint32_t di = do_ + kd_i;
                    for (uint32_t kh_i = 0; kh_i < kh; ++kh_i) {
                        uint32_t hi = ho + kh_i;
                        for (uint32_t kw_i = 0; kw_i < kw; ++kw_i) {
                            uint32_t wi = wo + kw_i;

                            // Compute flattened indices for x and w in ND contiguous layout
                            uint64_t xIdx = ((((uint64_t)no * cin + ci) * din + di) * hin + hi) * win + wi;
                            uint64_t wIdx = ((((uint64_t)co * cin + ci) * kd + kd_i) * kh + kh_i) * kw + kw_i;

                            // Read and accumulate: using GlobalTensor operator[] for illustrative purposes
                            DTYPE_X xv = xGm[xIdx];
                            DTYPE_W wv = wGm[wIdx];
                            acc = acc + (DTYPE_Y)(xv * wv);
                        }
                    }
                }
            }
            yLocal.SetValue(t, acc);
        }
        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress) {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, cin, din, hin, win;
    uint32_t cout, kd, kh, kw;
    uint32_t dout, hout, wout;

    uint32_t tileNum;
    uint64_t blockLength;
    uint64_t tileLength;
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3D op;
    op.Init(
        x, w, y,
        tiling_data.n, tiling_data.cin, tiling_data.din, tiling_data.hin, tiling_data.win,
        tiling_data.cout, tiling_data.kd, tiling_data.kh, tiling_data.kw,
        tiling_data.dout, tiling_data.hout, tiling_data.wout,
        tiling_data.tileNum
    );
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized 3D convolution using a custom AscendC kernel.
    Assumptions in this optimized path:
      - stride=(1,1,1), padding=(0,0,0), dilation=(1,1,1), groups=1
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: tuple = (1, 1, 1), padding: tuple = (0, 0, 0), dilation: tuple = (1, 1, 1), groups: int = 1, bias: bool = False):
        super().__init__()
        # Keep a Conv3d module only to manage parameters (weight/bias) and shapes.
        self.conv3d = torch.nn.Conv3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, groups=groups, bias=bias)

        # Validate the supported parameter set for this custom kernel
        if stride != (1, 1, 1) or padding != (0, 0, 0) or dilation != (1, 1, 1) or groups != 1:
            raise ValueError("ModelNew custom kernel currently supports only stride=(1,1,1), padding=(0,0,0), dilation=(1,1,1), groups=1.")

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Invoke custom 3D convolution operator with input and learned weights
        y = custom_ops_lib.conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(x, self.conv3d.weight)
        # Add bias if present
        if self.conv3d.bias is not None:
            y = y + self.conv3d.bias.view(1, -1, 1, 1, 1)
        return y
