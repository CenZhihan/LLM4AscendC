#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
    GM_ADDR x,
    GM_ADDR w,
    GM_ADDR b,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    // Shapes and attributes
    const int32_t N    = static_cast<int32_t>(tiling_data.n);
    const int32_t Cin  = static_cast<int32_t>(tiling_data.cIn);
    const int32_t H    = static_cast<int32_t>(tiling_data.h);
    const int32_t W    = static_cast<int32_t>(tiling_data.w);
    const int32_t Cout = static_cast<int32_t>(tiling_data.cOut);
    const int32_t kH   = static_cast<int32_t>(tiling_data.kH);
    const int32_t kW   = static_cast<int32_t>(tiling_data.kW);
    const int32_t sH   = static_cast<int32_t>(tiling_data.strideH);
    const int32_t sW   = static_cast<int32_t>(tiling_data.strideW);
    const int32_t pH   = static_cast<int32_t>(tiling_data.padH);
    const int32_t pW   = static_cast<int32_t>(tiling_data.padW);
    const int32_t dH   = static_cast<int32_t>(tiling_data.dilH);
    const int32_t dW   = static_cast<int32_t>(tiling_data.dilW);
    const int32_t Hout = static_cast<int32_t>(tiling_data.hOut);
    const int32_t Wout = static_cast<int32_t>(tiling_data.wOut);

    float* x_ptr = reinterpret_cast<float*>(x);
    float* w_ptr = reinterpret_cast<float*>(w);
    float* b_ptr = reinterpret_cast<float*>(b);
    float* y_ptr = reinterpret_cast<float*>(y);

    int64_t total_output = static_cast<int64_t>(N) * Cout * Hout * Wout;
    int32_t block_num = AscendC::GetBlockNum();
    int32_t block_idx = AscendC::GetBlockIdx();

    int64_t work_per_block = (total_output + block_num - 1) / block_num;
    int64_t start_idx = work_per_block * block_idx;
    int64_t end_idx = start_idx + work_per_block;
    if (end_idx > total_output) end_idx = total_output;

    for (int64_t idx = start_idx; idx < end_idx; ++idx) {
        int32_t ow = static_cast<int32_t>(idx % Wout);
        int64_t t1 = idx / Wout;
        int32_t oh = static_cast<int32_t>(t1 % Hout);
        int64_t t2 = t1 / Hout;
        int32_t co = static_cast<int32_t>(t2 % Cout);
        int32_t n  = static_cast<int32_t>(t2 / Cout);

        float acc = b_ptr != nullptr ? b_ptr[co] : 0.0f;

        int32_t ih_base = oh * sH - pH;
        int32_t iw_base = ow * sW - pW;

        for (int32_t ci = 0; ci < Cin; ++ci) {
            for (int32_t kh = 0; kh < kH; ++kh) {
                int32_t ih = ih_base + kh * dH;
                if (ih < 0 || ih >= H) continue;
                for (int32_t kw = 0; kw < kW; ++kw) {
                    int32_t iw = iw_base + kw * dW;
                    if (iw < 0 || iw >= W) continue;

                    int64_t x_index = (((static_cast<int64_t>(n) * Cin + ci) * H + ih) * W + iw);
                    int64_t w_index = (((static_cast<int64_t>(co) * Cin + ci) * kH + kh) * kW + kw);

                    acc += x_ptr[x_index] * w_ptr[w_index];
                }
            }
        }

        int64_t y_index = (((static_cast<int64_t>(n) * Cout + co) * Hout + oh) * Wout + ow);
        y_ptr[y_index] = acc;
    }
}

import custom_ops_lib

class ModelNew(torch.nn.Module):

    Optimized 2D convolution using custom AscendC kernel:
    conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom
    """

        super().__init__()
        kh, kw = kernel_size
        # Parameters following PyTorch Conv2d layout: [C_out, C_in, kH, kW]

        self.dil_w = int(dilation[1])
