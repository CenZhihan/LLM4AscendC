kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dReluHardSwish {
public:
    __aicore__ inline KernelConv2dReluHardSwish() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t N, uint32_t CIn, uint32_t COut,
                                uint32_t H, uint32_t W, uint32_t KH, uint32_t KW,
                                uint32_t outH, uint32_t outW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW) {
        // store pointers
        xPtr = reinterpret_cast<__gm__ float*>(x);
        wPtr = reinterpret_cast<__gm__ float*>(w);
        bPtr = reinterpret_cast<__gm__ float*>(b);
        yPtr = reinterpret_cast<__gm__ float*>(y);

        this->totalLength = totalLength;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        this->N = N;
        this->CIn = CIn;
        this->COut = COut;
        this->H = H;
        this->W = W;
        this->KH = KH;
        this->KW = KW;
        this->outH = outH;
        this->outW = outW;
        this->strideH = strideH;
        this->strideW = strideW;
        this->padH = padH;
        this->padW = padW;
    }

    __aicore__ inline void Process() {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        uint32_t blockStart = this->blockLength * AscendC::GetBlockIdx();
        for (int32_t p = 0; p < loopCount; ++p) {
            uint32_t start = blockStart + p * this->tileLength;
            uint32_t end = start + this->tileLength;
            if (end > blockStart + this->blockLength) {
                end = blockStart + this->blockLength;
            }
            for (uint32_t idx = start; idx < end; ++idx) {
                // map linear idx to (n, oc, oh, ow)
                uint32_t ow = idx % outW;
                uint32_t tmp1 = idx / outW;
                uint32_t oh = tmp1 % outH;
                uint32_t tmp2 = tmp1 / outH;
                uint32_t oc = tmp2 % COut;
                uint32_t n  = tmp2 / COut;

                float acc = 0.f;
                for (uint32_t ic = 0; ic < CIn; ++ic) {
                    for (uint32_t kh = 0; kh < KH; ++kh) {
                        for (uint32_t kw = 0; kw < KW; ++kw) {
                            int32_t ih = static_cast<int32_t>(oh) * static_cast<int32_t>(strideH) - static_cast<int32_t>(padH) + static_cast<int32_t>(kh);
                            int32_t iw = static_cast<int32_t>(ow) * static_cast<int32_t>(strideW) - static_cast<int32_t>(padW) + static_cast<int32_t>(kw);
                            // Bounds guaranteed with pad=0, stride=1 in our usage
                            uint64_t xIndex = (((uint64_t)n * CIn + ic) * H + ih) * W + iw;
                            uint64_t wIndex = (((uint64_t)oc * CIn + ic) * KH + kh) * KW + kw;
                            acc += xPtr[xIndex] * wPtr[wIndex];
                        }
                    }
                }
                acc += bPtr[oc];                 // bias
                float relu = acc > 0.f ? acc : 0.f;            // ReLU
                float hs = (relu + 3.f) / 6.f;                 // HardSwish component
                if (hs < 0.f) hs = 0.f;
                if (hs > 1.f) hs = 1.f;
                float outv = relu * hs;
                yPtr[idx] = outv;
            }
        }
    }

private:
    __gm__ float* xPtr;
    __gm__ float* wPtr;
    __gm__ float* bPtr;
    __gm__ float* yPtr;

    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t N, CIn, COut, H, W, KH, KW, outH, outW;
    uint32_t strideH, strideW, padH, padW;
};

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluHardSwish op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.N, tiling_data.CIn, tiling_data.COut,
            tiling_data.H, tiling_data.W, tiling_data.KH, tiling_data.KW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW);
    op.Process();
}
