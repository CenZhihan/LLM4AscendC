kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmGroupNormMinBiasAdd {
public:
    __aicore__ inline KernelGemmGroupNormMinBiasAdd() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b_gemm, GM_ADDR gamma, GM_ADDR beta, GM_ADDR bias, GM_ADDR groups, GM_ADDR z,
                                uint32_t N, uint32_t I, uint32_t C, uint32_t tileNum) {
        this->N = N;
        this->I = I;
        this->C = C;
        this->tileNum = tileNum;
        this->rowsPerBlock = N / AscendC::GetBlockNum();
        this->rowStart = this->rowsPerBlock * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ float *)x + this->rowStart * this->I, this->rowsPerBlock * this->I);
        wGm.SetGlobalBuffer((__gm__ float *)w, this->C * this->I);
        bGemmGm.SetGlobalBuffer((__gm__ float *)b_gemm, this->C);
        gammaGm.SetGlobalBuffer((__gm__ float *)gamma, this->C);
        betaGm.SetGlobalBuffer((__gm__ float *)beta, this->C);
        biasGm.SetGlobalBuffer((__gm__ float *)bias, this->C); // assume bias stored as [1,C,1,1] contiguous
        groupsGm.SetGlobalBuffer((__gm__ int32_t *)groups, 1);
        zGm.SetGlobalBuffer((__gm__ float *)z + this->rowStart * this->C, this->rowsPerBlock * this->C);
    }

    __aicore__ inline void Process() {
        int32_t groups = 1;
        // Read groups value from GM
        AscendC::LocalTensor<int32_t> groupsLocal;
        groupsLocal.SetSize(1);
        AscendC::DataCopy(groupsLocal, groupsGm, 1);
        groups = groupsLocal.GetValue(0);
        if (groups <= 0) { groups = 1; }
        int32_t groupSize = this->C / groups;
        float eps = 1e-5f;

        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            // Load x row
            AscendC::LocalTensor<float> xRow;
            xRow.SetSize(this->I);
            AscendC::DataCopy(xRow, xGm[r * this->I], this->I);

            // Allocate buffers
            AscendC::LocalTensor<float> yRow;
            AscendC::LocalTensor<float> normRow;
            yRow.SetSize(this->C);
            normRow.SetSize(this->C);

            // GEMM: yRow[c] = dot(xRow, w[c]) + b_gemm[c]
            for (uint32_t c = 0; c < this->C; ++c) {
                // Load weight vector w[c,:]
                float acc = 0.0f;
                for (uint32_t k = 0; k < this->I; ++k) {
                    float xv = xRow.GetValue(k);
                    float wv = wGm[c * this->I + k];
                    acc += xv * wv;
                }
                float bg = (bGemmGm.GetSize() > 0) ? bGemmGm[c] : 0.0f;
                yRow.SetValue(c, acc + bg);
            }

            // GroupNorm: compute per-group mean/var on channels for this row
            // Accumulate sums per group
            AscendC::LocalTensor<float> sumG;
            AscendC::LocalTensor<float> sumSqG;
            sumG.SetSize(groups);
            sumSqG.SetSize(groups);
            for (int32_t g = 0; g < groups; ++g) {
                sumG.SetValue(g, 0.0f);
                sumSqG.SetValue(g, 0.0f);
            }
            for (uint32_t c = 0; c < this->C; ++c) {
                int32_t g = static_cast<int32_t>(c / groupSize);
                float v = yRow.GetValue(c);
                sumG.SetValue(g, sumG.GetValue(g) + v);
                sumSqG.SetValue(g, sumSqG.GetValue(g) + v * v);
            }
            // Normalize and apply affine
            for (uint32_t c = 0; c < this->C; ++c) {
                int32_t g = static_cast<int32_t>(c / groupSize);
                float mean = sumG.GetValue(g) / static_cast<float>(groupSize);
                float var = sumSqG.GetValue(g) / static_cast<float>(groupSize) - mean * mean;
                float invStd = 1.0f / AscendC::sqrt(var + eps);
                float v = yRow.GetValue(c);
                float gamma = (gammaGm.GetSize() > 0) ? gammaGm[c] : 1.0f;
                float beta = (betaGm.GetSize() > 0) ? betaGm[c] : 0.0f;
                float normalized = (v - mean) * invStd;
                float affine = normalized * gamma + beta;
                normRow.SetValue(c, affine);
            }

            // Min across channels
            float minVal = normRow.GetValue(0);
            for (uint32_t c = 1; c < this->C; ++c) {
                float v = normRow.GetValue(c);
                minVal = v < minVal ? v : minVal;
            }

            // Add external bias and broadcast to [N, C, 1, 1] (contiguous as N*C)
            for (uint32_t c = 0; c < this->C; ++c) {
                float bv = biasGm[c];
                zGm[r * this->C + c] = minVal + bv;
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGemmGm;
    AscendC::GlobalTensor<float> gammaGm;
    AscendC::GlobalTensor<float> betaGm;
    AscendC::GlobalTensor<float> biasGm;
    AscendC::GlobalTensor<int32_t> groupsGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t N;
    uint32_t I;
    uint32_t C;
    uint32_t tileNum;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
};

extern "C" __global__ __aicore__ void gemm_group_norm_min_bias_add_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b_gemm, GM_ADDR gamma, GM_ADDR beta, GM_ADDR bias, GM_ADDR groups, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmGroupNormMinBiasAdd op;
    op.Init(x, w, b_gemm, gamma, beta, bias, groups, z, tiling_data.N, tiling_data.I, tiling_data.C, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a custom AscendC kernel:
    Performs GEMM, GroupNorm, Min (over channel dim), and Bias addition.
    Output shape: [N, C, 1, 1]
    """
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        groups_tensor = torch.tensor(self.group_norm.num_groups, dtype=torch.int32, device=x.device)
        return custom_ops_lib.gemm_group_norm_min_bias_add_custom(
            x,
            self.gemm.weight,
            self.gemm.bias,
            self.group_norm.weight,
            self.group_norm.bias,
            self.bias,
            groups_tensor
        )
