kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelScatterAdd {
public:
    __aicore__ inline KernelScatterAdd() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR updates, GM_ADDR z,
                                uint32_t rows, uint32_t cols, uint32_t updates_len)
    {
        this->rows = rows;
        this->cols = cols;
        this->updates_len = updates_len;

        uint32_t row_id = AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)row_id * cols, cols);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z + (uint64_t)row_id * cols, cols);
        idxGm.SetGlobalBuffer((__gm__ int32_t*)idx + (uint64_t)row_id * updates_len, updates_len);
        updGm.SetGlobalBuffer((__gm__ DTYPE_X*)updates + (uint64_t)row_id * updates_len, updates_len);

        pipe.InitBuffer(rowBufQ, BUFFER_NUM, cols * sizeof(DTYPE_Z));
        pipe.InitBuffer(idxBufQ, BUFFER_NUM, updates_len * sizeof(int32_t));
        pipe.InitBuffer(updBufQ, BUFFER_NUM, updates_len * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        // Copy base row from x to local buffer
        AscendC::LocalTensor<DTYPE_Z> rowLocal = rowBufQ.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(rowLocal, xGm, cols);
        rowBufQ.EnQue(rowLocal);

        // Copy indices and updates to local buffers
        AscendC::LocalTensor<int32_t> idxLocal = idxBufQ.AllocTensor<int32_t>();
        AscendC::DataCopy(idxLocal, idxGm, updates_len);
        idxBufQ.EnQue(idxLocal);

        AscendC::LocalTensor<DTYPE_X> updLocal = updBufQ.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(updLocal, updGm, updates_len);
        updBufQ.EnQue(updLocal);

        // Apply scatter add on local row buffer
        AscendC::LocalTensor<DTYPE_Z> rowL = rowBufQ.DeQue<DTYPE_Z>();
        AscendC::LocalTensor<int32_t> idxL = idxBufQ.DeQue<int32_t>();
        AscendC::LocalTensor<DTYPE_X> updL = updBufQ.DeQue<DTYPE_X>();

        for (uint32_t i = 0; i < updates_len; ++i) {
            int32_t pos = idxL.GetValue(i); // position within the row
            if (pos >= 0 && (uint32_t)pos < cols) {
                DTYPE_Z cur = rowL.GetValue(pos);
                DTYPE_Z addv = (DTYPE_Z)updL.GetValue(i);
                cur = cur + addv;
                rowL.SetValue(pos, cur);
            }
        }

        rowBufQ.EnQue(rowL);
        idxBufQ.FreeTensor(idxL);
        updBufQ.FreeTensor(updL);

        // Write back updated row to output
        AscendC::LocalTensor<DTYPE_Z> rowOut = rowBufQ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm, rowOut, cols);
        rowBufQ.FreeTensor(rowOut);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> rowBufQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> idxBufQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> updBufQ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    AscendC::GlobalTensor<int32_t> idxGm;
    AscendC::GlobalTensor<DTYPE_X> updGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t updates_len;
};

extern "C" __global__ __aicore__ void scatter_add_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR updates, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelScatterAdd op;
    op.Init(x, idx, updates, z, tiling_data.rows, tiling_data.cols, tiling_data.updates_len);
    op.Process();
}
"""
