kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelDenseBlockConcat {
public:
    __aicore__ inline KernelDenseBlockConcat() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z, uint32_t totalLenX, uint32_t totalLenY, uint32_t tileNum)
    {
        this->blockLenX = totalLenX / AscendC::GetBlockNum();
        this->blockLenY = totalLenY / AscendC::GetBlockNum();
        this->tileNum = tileNum;

        // Avoid zero tile length
        this->tileLenX = AscendC::Max((uint32_t)1, (this->blockLenX / tileNum / BUFFER_NUM));
        this->tileLenY = AscendC::Max((uint32_t)1, (this->blockLenY / tileNum / BUFFER_NUM));

        int32_t blockIdx = AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->blockLenX * blockIdx, this->blockLenX);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->blockLenY * blockIdx, this->blockLenY);

        // Output is concatenation along channel dim -> z = [x_flat, y_flat]
        // X portion of z starts at offset blockLenX * blockIdx
        zGmX.SetGlobalBuffer((__gm__ DTYPE_Z*)z + this->blockLenX * blockIdx, this->blockLenX);
        // Y portion of z starts at global offset totalLenX + blockLenY * blockIdx
        zGmY.SetGlobalBuffer((__gm__ DTYPE_Z*)z + totalLenX + this->blockLenY * blockIdx, this->blockLenY);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLenX * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileLenY * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Copy X part
        int32_t loopCountX = this->CalcLoopCount(this->blockLenX, this->tileLenX);
        for (int32_t i = 0; i < loopCountX; ++i) {
            int32_t len = this->GetTileLen(this->blockLenX, this->tileLenX, i);
            CopyInX(i, len);
            CopyOutX(i, len);
        }

        // Copy Y part
        int32_t loopCountY = this->CalcLoopCount(this->blockLenY, this->tileLenY);
        for (int32_t i = 0; i < loopCountY; ++i) {
            int32_t len = this->GetTileLen(this->blockLenY, this->tileLenY, i);
            CopyInY(i, len);
            CopyOutY(i, len);
        }
    }

private:
    __aicore__ inline int32_t CalcLoopCount(uint32_t blockLen, uint32_t tileLen)
    {
        uint32_t tilesPerBufferRound = this->tileNum * BUFFER_NUM;
        if (tileLen == 0) return 0;
        uint32_t totalTiles = (blockLen + tileLen - 1) / tileLen;
        // Ensure at least one loop
        return totalTiles > 0 ? totalTiles : 1;
    }

    __aicore__ inline int32_t GetTileLen(uint32_t blockLen, uint32_t tileLen, int32_t progress)
    {
        uint32_t offset = progress * tileLen;
        uint32_t remain = (offset < blockLen) ? (blockLen - offset) : 0;
        return remain > tileLen ? tileLen : remain;
    }

    __aicore__ inline void CopyInX(int32_t progress, int32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLenX], len);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void CopyOutX(int32_t progress, int32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::DataCopy(zGmX[progress * this->tileLenX], xLocal, len);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyInY(int32_t progress, int32_t len)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(yLocal, yGm[progress * this->tileLenY], len);
        inQueueY.EnQue(yLocal);
    }

    __aicore__ inline void CopyOutY(int32_t progress, int32_t len)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(zGmY[progress * this->tileLenY], yLocal, len);
        inQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGmX;
    AscendC::GlobalTensor<DTYPE_Z> zGmY;

    uint32_t blockLenX;
    uint32_t blockLenY;
    uint32_t tileNum;
    uint32_t tileLenX;
    uint32_t tileLenY;
};

extern "C" __global__ __aicore__ void densenet121_dense_block_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDenseBlockConcat op;
    op.Init(x, y, z, tiling_data.totalLenX, tiling_data.totalLenY, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        super(ModelNew, self).__init__()
        layers = []
        for i in range(num_layers):
            layers.append(self._make_layer(num_input_features + i * growth_rate, growth_rate))
        self.layers = torch.nn.ModuleList(layers)

    def _make_layer(self, in_features: int, growth_rate: int):
        return torch.nn.Sequential(
            torch.nn.BatchNorm2d(in_features),
            torch.nn.ReLU(inplace=True),
            torch.nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
            torch.nn.Dropout(0.0)
        )

    def forward(self, x):
        # Start with initial features; progressively concatenate new features along channel dim via custom op
        for layer in self.layers:
            new_feature = layer(x)
            x = custom_ops_lib.densenet121_dense_block_custom(x, new_feature)
        return x
'''
