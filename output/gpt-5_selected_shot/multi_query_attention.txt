kernel_src="""
#include "kernel_operator.h"

// The kernel computes: out[b, n, h, d] = sum_k attn[b, h, n, k] * v[b, k, d]
// Inputs:
//  - attn: [B, H, N, K]   (DTYPE_X)
//  - v:    [B, K, Dh]     (DTYPE_Y)
// Output:
//  - out:  [B, N, H, Dh]  (DTYPE_Z)

class KernelMQA {
public:
    __aicore__ inline KernelMQA() {}
    __aicore__ inline void Init(GM_ADDR attn, GM_ADDR v, GM_ADDR out,
                                uint32_t B, uint32_t H, uint32_t N, uint32_t K, uint32_t Dh)
    {
        this->B = B;
        this->H = H;
        this->N = N;
        this->K = K;
        this->Dh = Dh;

        attnGm.SetGlobalBuffer((__gm__ DTYPE_X*)attn, (uint32_t)((uint64_t)B * H * N * K));
        vGm.SetGlobalBuffer((__gm__ DTYPE_Y*)v, (uint32_t)((uint64_t)B * K * Dh));
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z*)out, (uint32_t)((uint64_t)B * N * H * Dh));

        // Initialize buffers for Dh-sized vectors
        uint32_t vecSizeBytes = Dh * sizeof(DTYPE_Z);
        pipe.InitBuffer(vRowQueue, 1, vecSizeBytes);
        pipe.InitBuffer(accQueue, 1, vecSizeBytes);
    }

    __aicore__ inline void Process()
    {
        // Map blockIdx to (b, h) pair
        uint32_t totalBH = this->B * this->H;
        uint32_t blk = AscendC::GetBlockIdx();
        if (totalBH == 0) {
            return;
        }
        uint32_t bh = blk % totalBH;
        uint32_t b = bh / this->H;
        uint32_t h = bh % this->H;

        for (uint32_t n = 0; n < this->N; ++n) {
            // accumulator for one output vector of length Dh
            AscendC::LocalTensor<DTYPE_Z> accLocal = accQueue.AllocTensor<DTYPE_Z>();
            // Initialize to zero
            Zero(accLocal, this->Dh);

            for (uint32_t k = 0; k < this->K; ++k) {
                // Load attn scalar alpha = attn[b,h,n,k]
                DTYPE_X alpha;
                LoadAttnScalar(b, h, n, k, alpha);

                // Load v row: v[b, k, :]
                AscendC::LocalTensor<DTYPE_Y> vRowLocal = vRowQueue.AllocTensor<DTYPE_Y>();
                uint64_t vOffset = ((uint64_t)b * this->K + k) * this->Dh;
                AscendC::DataCopy(vRowLocal, vGm[vOffset], this->Dh);

                // acc += alpha * vRow
                Fma(accLocal, vRowLocal, alpha, this->Dh);

                vRowQueue.FreeTensor(vRowLocal);
            }

            // Store out[b, n, h, :]
            uint64_t outOffset = (((uint64_t)b * this->N + n) * this->H + h) * this->Dh;
            AscendC::DataCopy(outGm[outOffset], accLocal, this->Dh);

            accQueue.FreeTensor(accLocal);
        }
    }

private:
    __aicore__ inline void Zero(AscendC::LocalTensor<DTYPE_Z>& t, uint32_t len)
    {
        // Set vector to 0.0
        // Use a simple loop to set zeros
        for (uint32_t i = 0; i < len; ++i) {
            t.SetValue(i, (DTYPE_Z)0.0f);
        }
    }

    __aicore__ inline void Fma(AscendC::LocalTensor<DTYPE_Z>& acc,
                               AscendC::LocalTensor<DTYPE_Y>& vrow,
                               DTYPE_X alpha, uint32_t len)
    {
        // acc[i] += alpha * vrow[i]
        for (uint32_t i = 0; i < len; ++i) {
            DTYPE_Z prev = acc.GetValue(i);
            DTYPE_Z val = (DTYPE_Z)((DTYPE_Z)alpha * (DTYPE_Z)vrow.GetValue(i));
            acc.SetValue(i, prev + val);
        }
    }

    __aicore__ inline void LoadAttnScalar(uint32_t b, uint32_t h, uint32_t n, uint32_t k, DTYPE_X& outAlpha)
    {
        uint64_t idx = (((uint64_t)b * this->H + h) * this->N + n) * this->K + k;
        outAlpha = attnGm.GetValue(idx);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> vRowQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> accQueue;

    AscendC::GlobalTensor<DTYPE_X> attnGm;
    AscendC::GlobalTensor<DTYPE_Y> vGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;

    uint32_t B;
    uint32_t H;
    uint32_t N;
    uint32_t K;
    uint32_t Dh;
};

extern "C" __global__ __aicore__ void multi_query_attention_custom(GM_ADDR attn, GM_ADDR v, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMQA op;
    op.Init(attn, v, out, tiling_data.B, tiling_data.H, tiling_data.N, tiling_data.K, tiling_data.Dh);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import torch.nn as nn
import math

class ModelNew(torch.nn.Module):
    """
    Multi-Query Attention (MQA) optimized: fuse context computation (attn @ V) into custom AscendC kernel.
    """
    def __init__(self, d_model=4096, num_heads=32):
        super().__init__()
        self.q_proj = nn.Linear(d_model, d_model)
        self.k_proj = nn.Linear(d_model, d_model // num_heads)  # shared across heads
        self.v_proj = nn.Linear(d_model, d_model // num_heads)
        self.out_proj = nn.Linear(d_model, d_model)
        self.num_heads = num_heads

    def forward(self, x):
        B, N, D = x.shape
        H = self.num_heads
        d_h = D // H

        # Projections
        Q = self.q_proj(x).view(B, N, H, d_h)            # [B,N,H,Dh]
        K = self.k_proj(x).unsqueeze(2)                  # [B,N,1,Dh] shared across heads
        V = self.v_proj(x).unsqueeze(2)                  # [B,N,1,Dh]

        # Scaled dot-product attention scores per head: [B,H,N,K]
        attn = torch.einsum("bnhd,bkhd->bhnk", Q, K) / math.sqrt(d_h)
        attn = torch.softmax(attn, dim=-1)

        # Context via custom kernel: out[b,n,h,d] = sum_k attn[b,h,n,k] * V[b,k,d]
        out = custom_ops_lib.multi_query_attention_custom(attn, V.squeeze(2))  # [B,N,H,Dh]

        out = out.reshape(B, N, D)
        return self.out_proj(out)
'''
