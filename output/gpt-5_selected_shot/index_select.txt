static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    IndexSelectCustomTilingData tiling;
    const auto* x_shape = context->GetInputShape(0);
    const auto* idx_shape = context->GetInputShape(1);

    uint32_t rows = 1;
    uint32_t cols = 1;
    if (x_shape->GetOriginShape().GetDimNum() >= 2) {
        rows = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
        cols = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    } else if (x_shape->GetOriginShape().GetDimNum() == 1) {
        rows = 1;
        cols = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    }

    uint32_t k = static_cast<uint32_t>(idx_shape->GetOriginShape().GetShapeSize());

    // Set an appropriate block dim, not exceeding rows
    uint32_t blockDim = rows < 8 ? rows : 8;
    if (blockDim == 0) {
        blockDim = 1;
    }
    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_k(k);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* idx_shape = context->GetInputShape(1);
    gert::Shape* y_shape = context->GetOutputShape(0);

    int64_t rows = 1;
    if (x_shape->GetOriginShape().GetDimNum() >= 2) {
        rows = x_shape->GetOriginShape().GetDim(0);

    int64_t k = idx_shape->GetOriginShape().GetShapeSize();

    std::vector<int64_t> out_dims = {rows, k};
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelIndexSelect {
public:
    __aicore__ inline KernelIndexSelect() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t k)
    {
        this->rows = rows;
        this->cols = cols;
        this->k = k;

        // Partition rows among blocks
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        this->rowStart = (rows * blockIdx) / blockNum;
        this->rowEnd   = (rows * (blockIdx + 1)) / blockNum;
        this->rowsInBlock = this->rowEnd - this->rowStart;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + static_cast<uint64_t>(rowStart) * cols,
                            static_cast<uint64_t>(rowsInBlock) * cols);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_INDICES*)indices, k);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + static_cast<uint64_t>(rowStart) * k,
                            static_cast<uint64_t>(rowsInBlock) * k);

        // Buffers: one full row of x, full indices, and one output row
        pipe.InitBuffer(inQueueX, BUFFER_NUM, cols * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueIdx, 1, k * sizeof(DTYPE_INDICES));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, k * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // Load indices once
        AscendC::LocalTensor<DTYPE_INDICES> idxLocal = inQueueIdx.AllocTensor<DTYPE_INDICES>();
        AscendC::DataCopy(idxLocal, idxGm, k);

        for (uint32_t r = 0; r < rowsInBlock; ++r) {
            // Copy in one row of x
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[r * cols], cols);
            inQueueX.EnQue(xLocal);

            // Compute gather for this row
            AscendC::LocalTensor<DTYPE_X> xRow = inQueueX.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

            // Perform gather: yLocal[t] = xRow[ idxLocal[t] ], selecting along dim=1 of the original 2D
            AscendC::Gather(yLocal, xRow, idxLocal, k);

            outQueueY.EnQue<DTYPE_Z>(yLocal);
            inQueueX.FreeTensor(xRow);

            // Copy out one row of y
            AscendC::LocalTensor<DTYPE_Z> yRow = outQueueY.DeQue<DTYPE_Z>();
            AscendC::DataCopy(yGm[r * k], yRow, k);
            outQueueY.FreeTensor(yRow);
        }
        inQueueIdx.FreeTensor(idxLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_INDICES> idxGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t k;
    uint32_t rowStart;
    uint32_t rowEnd;
    uint32_t rowsInBlock;
};

extern "C" __global__ __aicore__ void index_select_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelIndexSelect op;
    op.Init(x, indices, y, tiling_data.rows, tiling_data.cols, tiling_data.k);
    op.Process();
}
"""

at::Tensor index_select_custom_impl_npu(const at::Tensor& self, const at::Tensor& indices) {

    at::IntArrayRef out_sizes = {self.size(0), indices.numel()};

    return result;

    m.impl(\"index_select_custom\", &index_select_custom_impl_npu);
