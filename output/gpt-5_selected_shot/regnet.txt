namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    RegnetCustomTilingData tiling;

    auto* inShape = context->GetInputShape(0);
    auto originShape = inShape->GetOriginShape();
    uint32_t dimNum = originShape.GetDimNum();
    uint32_t n = dimNum >= 1 ? (uint32_t)originShape.GetDim(0) : 1;
    uint32_t c = dimNum >= 2 ? (uint32_t)originShape.GetDim(1) : 1;
    uint32_t h = dimNum >= 3 ? (uint32_t)originShape.GetDim(2) : 1;
    uint32_t w = dimNum >= 4 ? (uint32_t)originShape.GetDim(3) : 1;
    uint32_t hw = h * w;
    uint32_t features = n * c;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_hw(hw);
    tiling.set_features(features);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);
    // Default to copying and then adjusting to (N, C)
    *out_shape = *in_shape;
    auto originShape = in_shape->GetOriginShape();
    uint64_t n = originShape.GetDimNum() >= 1 ? originShape.GetDim(0) : 1;
    uint64_t c = originShape.GetDimNum() >= 2 ? originShape.GetDim(1) : 1;

    gert::Shape newShape;
    newShape.SetDimNum(2);
    newShape.SetDim(0, n);
    newShape.SetDim(1, c);
    *out_shape = newShape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue
 
class KernelGAP {
public:
    __aicore__ inline KernelGAP() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t hw, uint32_t features, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->hw = hw;
        this->features = features;
        this->tileNum = tileNum;

        this->blockFeatureNum = (features + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t startFeat = this->blockFeatureNum * AscendC::GetBlockIdx();
        uint32_t endFeat = AscendC::Min(features, startFeat + this->blockFeatureNum);
        this->blockFeatureNum = endFeat - startFeat;

        // Each feature has 'hw' elements in memory; layout is [features, hw] in contiguous sense
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + startFeat * hw, this->blockFeatureNum * hw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + startFeat, this->blockFeatureNum);

        // Tile by features
        this->tileFeat = AscendC::Max((uint32_t)1, (this->blockFeatureNum / (tileNum * BUFFER_NUM)));
        if (this->tileFeat * tileNum * BUFFER_NUM < this->blockFeatureNum) {
            this->tileFeat += 1;
        }

        // Choose a pixel tile for reduction along hw
        this->pixelTile = AscendC::Min(hw, (uint32_t)2048);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileFeat * this->pixelTile * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileFeat * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = (this->blockFeatureNum + this->tileFeat - 1) / this->tileFeat;
        loopCount = AscendC::Max(loopCount, (int32_t)(this->tileNum * BUFFER_NUM));

        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint32_t featOffset = progress * this->tileFeat;
        if (featOffset >= this->blockFeatureNum) {
            // enqueue empty to keep pipeline aligned
            AscendC::LocalTensor<DTYPE_X> dummy = inQueueX.AllocTensor<DTYPE_X>();
            inQueueX.EnQue(dummy);
            return;
        }
        uint32_t curTileFeat = AscendC::Min(this->tileFeat, this->blockFeatureNum - featOffset);
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // copy first chunk of pixelTile for all features in this tile
        uint32_t copyElems = curTileFeat * AscendC::Min(this->pixelTile, this->hw);
        AscendC::DataCopy(xLocal, xGm[featOffset * this->hw], copyElems);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        uint32_t featOffset = progress * this->tileFeat;
        if (featOffset >= this->blockFeatureNum) {
            outQueueY.EnQue<DTYPE_Y>(yLocal);
            inQueueX.FreeTensor(xLocal);
            return;
        }
        uint32_t curTileFeat = AscendC::Min(this->tileFeat, this->blockFeatureNum - featOffset);

        // Initialize outputs to zero
        for (uint32_t i = 0; i < curTileFeat; ++i) {
            yLocal.SetValue(i, (DTYPE_Y)0);
        }

        // Accumulate over all pixels in chunks of pixelTile
        uint32_t remaining = this->hw;
        uint32_t pixelBase = 0;
        while (remaining > 0) {
            uint32_t step = AscendC::Min(remaining, this->pixelTile);
            // xLocal currently holds the chunk starting at pixelBase for all features in the tile
            // Accumulate
            for (uint32_t f = 0; f < curTileFeat; ++f) {
                DTYPE_Y acc = yLocal.GetValue(f);
                for (uint32_t p = 0; p < step; ++p) {
                    acc += (DTYPE_Y)xLocal.GetValue(f * step + p);
                }
                yLocal.SetValue(f, acc);
            }
            pixelBase += step;
            remaining -= step;
            if (remaining > 0) {
                // load next chunk
                uint32_t copyElems = curTileFeat * AscendC::Min(remaining, this->pixelTile);
                AscendC::DataCopy(xLocal, xGm[featOffset * this->hw + pixelBase], copyElems);
            }
        }

        // Divide by hw to get mean
        DTYPE_Y inv_hw = (DTYPE_Y)(1.0f / (float)this->hw);
        for (uint32_t i = 0; i < curTileFeat; ++i) {
            DTYPE_Y val = yLocal.GetValue(i) * inv_hw;
            yLocal.SetValue(i, val);
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        uint32_t featOffset = progress * this->tileFeat;
        if (featOffset < this->blockFeatureNum) {
            uint32_t curTileFeat = AscendC::Min(this->tileFeat, this->blockFeatureNum - featOffset);
            AscendC::DataCopy(yGm[featOffset], yLocal, curTileFeat);
        }
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t h;
    uint32_t w;
    uint32_t hw;
    uint32_t features;

    uint32_t blockFeatureNum;
    uint32_t tileNum;
    uint32_t tileFeat;
    uint32_t pixelTile;
};

extern "C" __global__ __aicore__ void regnet_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGAP op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.hw, tiling_data.features, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_channels, stages, block_widths, output_classes):
        super(ModelNew, self).__init__()

        self.stages = stages
        self.block_widths = block_widths
        
        layers = []
        current_channels = input_channels
        
        # Construct the stages with their respective blocks
        for i in range(stages):
            layers.append(self._make_stage(current_channels, block_widths[i]))
            current_channels = block_widths[i]
        
        self.feature_extractor = torch.nn.Sequential(*layers)
        
        # Final fully connected layer for classification
        self.fc = torch.nn.Linear(block_widths[-1], output_classes)
    
    def _make_stage(self, in_channels, out_channels):
        return torch.nn.Sequential(
            torch.nn.Conv2d(in_channels, out_channels, kernel_size=3, padding=1),
            torch.nn.BatchNorm2d(out_channels),
            torch.nn.ReLU(),
            torch.nn.Conv2d(out_channels, out_channels, kernel_size=3, padding=1),
            torch.nn.BatchNorm2d(out_channels),
            torch.nn.ReLU(),
            torch.nn.MaxPool2d(kernel_size=2, stride=2)
        )

    def forward(self, x):
        # Feature extraction stages
        x = self.feature_extractor(x)
        # Replace global average pooling with custom AscendC kernel
        x = custom_ops_lib.regnet_custom(x)
        # Fully connected classification layer
        x = self.fc(x)
        return x
'''
