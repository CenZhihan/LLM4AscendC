kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    // Dimensions from tiling
    const uint32_t N  = tiling_data.n;
    const uint32_t C  = tiling_data.c;
    const uint32_t H  = tiling_data.h;
    const uint32_t W  = tiling_data.w;
    const uint32_t OC = tiling_data.oc;
    const uint32_t KH = tiling_data.kh;
    const uint32_t KW = tiling_data.kw;
    const uint32_t SH = tiling_data.sh;
    const uint32_t SW = tiling_data.sw;
    const uint32_t PH = tiling_data.ph;
    const uint32_t PW = tiling_data.pw;
    const uint32_t OH = tiling_data.oh;
    const uint32_t OW = tiling_data.ow;
    const uint64_t TOTAL = tiling_data.totalOut;

    __gm__ float* xPtr = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* wPtr = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* bPtr = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* yPtr = reinterpret_cast<__gm__ float*>(y);

    const uint64_t inStrideN = static_cast<uint64_t>(C) * H * W;
    const uint64_t inStrideC = static_cast<uint64_t>(H) * W;

    const uint64_t outStrideN = static_cast<uint64_t>(OC) * OH * OW;
    const uint64_t outStrideC = static_cast<uint64_t>(OH) * OW;

    const uint64_t wStrideOC = static_cast<uint64_t>(C) * KH * KW;
    const uint64_t wStrideC  = static_cast<uint64_t>(KH) * KW;
    const uint64_t wStrideKH = static_cast<uint64_t>(KW);

    const uint32_t blocks = AscendC::GetBlockNum();
    const uint32_t bid    = AscendC::GetBlockIdx();

    uint64_t basePerBlock = TOTAL / blocks;
    uint64_t remain = TOTAL % blocks;
    uint64_t start = bid * basePerBlock + AscendC::Min(static_cast<uint32_t>(bid), static_cast<uint32_t>(remain));
    uint64_t len = basePerBlock + (bid < remain ? 1 : 0);
    uint64_t end = start + len;

    for (uint64_t idx = start; idx < end; ++idx) {
        // Map linear index to (n, oc, oh, ow)
        uint64_t tmp = idx;
        uint32_t n  = static_cast<uint32_t>(tmp / outStrideN); tmp %= outStrideN;
        uint32_t oc = static_cast<uint32_t>(tmp / outStrideC); tmp %= outStrideC;
        uint32_t oh = static_cast<uint32_t>(tmp / OW);
        uint32_t ow = static_cast<uint32_t>(tmp % OW);

        float acc = 0.0f;
        if (bPtr != nullptr) {
            acc = bPtr[oc];
        }

        // Convolution accumulation
        for (uint32_t ci = 0; ci < C; ++ci) {
            for (uint32_t kh = 0; kh < KH; ++kh) {
                int32_t ih = static_cast<int32_t>(oh) * static_cast<int32_t>(SH) - static_cast<int32_t>(PH) + static_cast<int32_t>(kh);
                if (ih < 0 || ih >= static_cast<int32_t>(H)) continue;
                for (uint32_t kw = 0; kw < KW; ++kw) {
                    int32_t iw = static_cast<int32_t>(ow) * static_cast<int32_t>(SW) - static_cast<int32_t>(PW) + static_cast<int32_t>(kw);
                    if (iw < 0 || iw >= static_cast<int32_t>(W)) continue;

                    uint64_t inOffset  = static_cast<uint64_t>(n) * inStrideN + static_cast<uint64_t>(ci) * inStrideC + static_cast<uint64_t>(ih) * W + static_cast<uint64_t>(iw);
                    uint64_t wOffset   = static_cast<uint64_t>(oc) * wStrideOC + static_cast<uint64_t>(ci) * wStrideC + static_cast<uint64_t>(kh) * wStrideKH + static_cast<uint64_t>(kw);

                    acc += xPtr[inOffset] * wPtr[wOffset];
                }
            }
        }

        uint64_t outOffset = static_cast<uint64_t>(n) * outStrideN + static_cast<uint64_t>(oc) * outStrideC + static_cast<uint64_t>(oh) * OW + static_cast<uint64_t>(ow);
        yPtr[outOffset] = acc;
    }
}

at::Tensor conv_standard_2d_square_input_square_kernel_custom_impl_npu(const at::Tensor& x, const at::Tensor& weight, const at::Tensor& bias) {
    TORCH_CHECK(x.dim() == 4, "Input x must be 4D NCHW");
    TORCH_CHECK(weight.dim() == 4, "Weight must be 4D [OC, C, KH, KW]");
    TORCH_CHECK(x.size(1) == weight.size(1), "Input channels mismatch with weight");
    TORCH_CHECK(weight.size(2) == 11 && weight.size(3) == 11, "This custom kernel supports kernel_size=11 only");
    // Fixed stride/padding for this specialized kernel
    const int64_t sh = 4, sw = 4, ph = 2, pw = 2;
    const int64_t N = x.size(0);
    const int64_t C = x.size(1);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);
    const int64_t OC = weight.size(0);
    const int64_t OH = (H + 2 * ph - 11) / sh + 1;
    const int64_t OW = (W + 2 * pw - 11) / sw + 1;

    at::Tensor y = at::empty({N, OC, OH, OW}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard2dSquareInputSquareKernelCustom, x, weight, bias, y);
    return y;
}
