host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline int64_t NormalizeDim(int64_t dim, int64_t rank) {
    if (dim < 0) dim += rank;
    return dim;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const gert::OriShape &ori = in_shape->GetOriginShape();

    int64_t rank = ori.GetDimNum();
    uint32_t totalLength = static_cast<uint32_t>(ori.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim_ptr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim_attr = (dim_ptr != nullptr) ? *dim_ptr : 1;
    int64_t norm_dim = NormalizeDim(dim_attr, rank);

    // Fallback: if dim out of range, clamp to last dim
    if (norm_dim < 0 || norm_dim >= rank) {
        norm_dim = rank - 1;
    }

    uint32_t axisLen = static_cast<uint32_t>(ori.GetDim(norm_dim));
    uint32_t outerNum = (axisLen == 0) ? 0 : (totalLength / axisLen);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_outerNum(outerNum);
    tiling.set_axisLen(axisLen);
    tiling.set_dim(static_cast<int32_t>(norm_dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace_sizes = context->GetWorkspaceSizes(1);
    workspace_sizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"
constexpr int32_t BUFFER_NUM = 1;

class KernelLogSoftmax {
public:
    __aicore__ inline KernelLogSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerNum, uint32_t axisLen, int32_t dim)
    {
        this->outerNum = outerNum;
        this->axisLen = axisLen;
        this->dim = dim;

        xGm.SetGlobalBuffer((__gm__ float *)x, outerNum * axisLen);
        yGm.SetGlobalBuffer((__gm__ float *)y, outerNum * axisLen);

        // Allocate buffers: one full row per buffer to keep logic simple
        pipe.InitBuffer(xBuf, axisLen * sizeof(float));
        pipe.InitBuffer(yBuf, axisLen * sizeof(float));
        pipe.InitBuffer(tmpBuf, axisLen * sizeof(float));
        pipe.InitBuffer(scalarBuf, sizeof(float));  // single element buffer
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Split rows across blocks as evenly as possible
        uint32_t baseRows = outerNum / blockNum;
        uint32_t rem = outerNum % blockNum;
        uint32_t rowsForThisBlock = baseRows + (blockIdx < rem ? 1 : 0);
        if (rowsForThisBlock == 0) {
            return;
        }
        uint32_t startRow = baseRows * blockIdx + (blockIdx < rem ? blockIdx : rem);

        AscendC::LocalTensor<float> xLocal = xBuf.Get<float>();
        AscendC::LocalTensor<float> yLocal = yBuf.Get<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpBuf.Get<float>();
        AscendC::LocalTensor<float> sLocal = scalarBuf.Get<float>(); // length 1

        for (uint32_t r = 0; r < rowsForThisBlock; ++r) {
            uint32_t rowIndex = startRow + r;
            uint32_t offset = rowIndex * axisLen;

            // Load full row
            AscendC::DataCopy(xLocal, xGm[offset], axisLen);

            // rowMax = max(x)
            AscendC::ReduceMax(sLocal, xLocal, axisLen);          // sLocal[0] = max
            // x_centered = x - rowMax
            AscendC::Sub(yLocal, xLocal, sLocal, axisLen);        // broadcast subtract scalar

            // tmp = exp(x_centered)
            AscendC::Exp(tmpLocal, yLocal, axisLen);

            // sumExp = sum(tmp)
            AscendC::ReduceSum(sLocal, tmpLocal, axisLen);        // sLocal[0] = sumExp
            // sLocal = log(sumExp)
            AscendC::Log(sLocal, sLocal, 1);

            // y = x_centered - logsumexp
            AscendC::Sub(yLocal, yLocal, sLocal, axisLen);

            // Store result row
            AscendC::DataCopy(yGm[offset], yLocal, axisLen);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::QuePosition::VECCALC> xBuf, yBuf, tmpBuf, scalarBuf;
    AscendC::GlobalTensor<float> xGm, yGm;
    uint32_t outerNum;
    uint32_t axisLen;
    int32_t dim;
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmax op;
    op.Init(x, y, tiling_data.outerNum, tiling_data.axisLen, tiling_data.dim);
    op.Process();
}
"""
