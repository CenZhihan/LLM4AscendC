kernel_src="""
#include "kernel_operator.h"

class KernelDWConv2d {
public:
    __aicore__ inline KernelDWConv2d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t wi,
                                uint32_t k, uint32_t stride, uint32_t padding,
                                uint32_t hout, uint32_t wout)
    {
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)n * c * h * wi);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint64_t)c * k * k);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint64_t)n * c * hout * wout);

        this->N = n;
        this->C = c;
        this->H = h;
        this->W = wi;
        this->K = k;
        this->S = stride;
        this->P = padding;
        this->Ho = hout;
        this->Wo = wout;

        // Work partition: split over N*C across blocks
        this->totalNC = n * c;
        this->blkNum = AscendC::GetBlockNum();
        this->blkIdx = AscendC::GetBlockIdx();
        this->ncPerBlock = (totalNC + blkNum - 1) / blkNum;
        this->ncStart = blkIdx * ncPerBlock;
        this->ncEnd = AscendC::min<uint32_t>(totalNC, ncStart + ncPerBlock);
    }

    __aicore__ inline void Process()
    {
        // Currently supports stride=1 and padding=0
        for (uint32_t nc = ncStart; nc < ncEnd; ++nc) {
            uint32_t n = nc / C;
            uint32_t cidx = nc % C;

            // Base offsets
            uint64_t xBase = (((uint64_t)n * C + cidx) * H) * W;
            uint64_t yBase = (((uint64_t)n * C + cidx) * Ho) * Wo;
            uint64_t wBase = ((uint64_t)cidx) * K * K;

            for (uint32_t oh = 0; oh < Ho; ++oh) {
                for (uint32_t ow = 0; ow < Wo; ++ow) {
                    float acc = 0.0f;
                    // Depthwise kernel accumulation
                    for (uint32_t kh = 0; kh < K; ++kh) {
                        uint32_t ih = oh * S + kh;
                        for (uint32_t kw = 0; kw < K; ++kw) {
                            uint32_t iw = ow * S + kw;
                            float xv = ((const __gm__ DTYPE_X*)xGm.GetPhyAddr())[xBase + (uint64_t)ih * W + iw];
                            float wv = ((const __gm__ DTYPE_W*)wGm.GetPhyAddr())[wBase + (uint64_t)kh * K + kw];
                            acc += xv * wv;
                        }
                    }
                    ((__gm__ DTYPE_Y*)yGm.GetPhyAddr())[yBase + (uint64_t)oh * Wo + ow] = (DTYPE_Y)acc;
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, C, H, W, K, S, P, Ho, Wo;
    uint32_t totalNC;
    uint32_t blkNum;
    uint32_t blkIdx;
    uint32_t ncPerBlock;
    uint32_t ncStart;
    uint32_t ncEnd;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDWConv2d op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.k, tiling_data.stride, tiling_data.padding,
            tiling_data.hout, tiling_data.wout);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Performs a depthwise 2D convolution operation with square input and square kernel (optimized with custom AscendC kernel).
    Only stride=1 and padding=0 are supported in this optimized path.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super(ModelNew, self).__init__()
        # Keep Conv2d to own parameters and shapes; groups=in_channels for depthwise
        self.conv2d = torch.nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding, groups=in_channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Use custom kernel (no bias, stride=1, padding=0)
        assert self.conv2d.bias is None, "Custom kernel path currently supports bias=False."
        assert self.conv2d.stride == (1, 1), "Custom kernel path supports stride=1 only."
        assert self.conv2d.padding == (0, 0), "Custom kernel path supports padding=0 only."
        return custom_ops_lib.conv_depthwise_2d_square_input_square_kernel_custom(x, self.conv2d.weight)
