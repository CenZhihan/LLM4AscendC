#include "min_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

#define REDUCE_TILING_0 1
#define REDUCE_TILING_1 2
#define REDUCE_TILING_2 3

namespace optiling {
constexpr uint32_t BLOCK_DIM = 1;
constexpr uint32_t ONE_REPEAT_BYTES = 256;
constexpr uint32_t ONE_BLOCK_BYTES = 32 * ONE_REPEAT_BYTES;
constexpr uint32_t TWO_BLOCK_BYTES = 16 * ONE_REPEAT_BYTES; // 4KB
constexpr uint32_t SIXTEEN_KB_BYTES = 16 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TilingData tiling;

    // Get input shape
    auto inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rank = static_cast<uint32_t>(inShape.GetDimNum());
    uint64_t totalLen = 1;
    uint32_t d0 = 1, d1 = 1, d2 = 1;

    if (rank >= 1) { d0 = static_cast<uint32_t>(inShape.GetDim(0)); }
    if (rank >= 2) { d1 = static_cast<uint32_t>(inShape.GetDim(1)); }
    if (rank >= 3) { d2 = static_cast<uint32_t>(inShape.GetDim(2)); }

    for (uint32_t i = 0; i < rank; ++i) {
        totalLen *= static_cast<uint64_t>(inShape.GetDim(i));
    }

    // Get attribute dim (default last dim if not provided)
    int64_t reduceDim = (rank == 0 ? 0 : static_cast<int64_t>(rank - 1));
    // Try get dim attribute if available
    // API may vary by version; this is indicative usage.
    auto attrs = context->GetAttrs();
    if (attrs != nullptr) {
        (void)attrs->GetInt("dim", reduceDim);
    }
    if (reduceDim < 0) {
        reduceDim += static_cast<int64_t>(rank);
    }
    if (reduceDim < 0 || reduceDim >= static_cast<int64_t>(rank)) {
        // Fallback to last dimension
        reduceDim = (rank == 0 ? 0 : static_cast<int64_t>(rank - 1));
    }

    // Compute outLength (flattened) as product of dims excluding reduceDim
    uint64_t outLen = 1;
    for (uint32_t i = 0; i < rank; ++i) {
        if (static_cast<int64_t>(i) == reduceDim) continue;
        outLen *= static_cast<uint64_t>(inShape.GetDim(i));
    }
    if (rank == 0) {
        outLen = 1;
    }

    // Set tiling key heuristics based on size per reduced row (bytes)
    uint64_t rowBytes = 0;
    if (rank == 0) {
        rowBytes = sizeof(float);
    } else if (reduceDim == 0) {
        rowBytes = static_cast<uint64_t>(d0) * sizeof(float);
    } else if (reduceDim == 1) {
        rowBytes = static_cast<uint64_t>(d1) * sizeof(float);
    } else { // reduceDim == 2 or higher
        rowBytes = static_cast<uint64_t>(d2) * sizeof(float);
    }

    if (rowBytes <= ONE_REPEAT_BYTES) {
        context->SetTilingKey(REDUCE_TILING_0);
    } else if (rowBytes <= TWO_BLOCK_BYTES) {
        context->SetTilingKey(REDUCE_TILING_1);
    } else if (rowBytes <= SIXTEEN_KB_BYTES) {
        context->SetTilingKey(REDUCE_TILING_2);
    } else {
        // default
        context->SetTilingKey(REDUCE_TILING_2);
    }

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_d0(d0);
    tiling.set_d1(d1);
    tiling.set_d2(d2);
    tiling.set_reduceDim(static_cast<uint32_t>(reduceDim));
    tiling.set_totalLength(static_cast<uint32_t>(totalLen));
    tiling.set_outLength(static_cast<uint32_t>(outLen));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static graphStatus InferShape(gert::InferShapeContext *context)
{
    auto inShape = context->GetInputShape(0)->GetOriginShape();
    int64_t rank = inShape.GetDimNum();
    // Get dim attribute
    int64_t reduceDim = rank > 0 ? rank - 1 : 0;
    auto attrs = context->GetAttrs();
    if (attrs != nullptr) {
        (void)attrs->GetInt("dim", reduceDim);
    }
    if (reduceDim < 0) {
        reduceDim += rank;
    }
    if (reduceDim < 0 || reduceDim >= rank) {
        reduceDim = rank > 0 ? rank - 1 : 0;
    }

    std::vector<int64_t> outDims;
    outDims.reserve(rank > 0 ? static_cast<size_t>(rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == reduceDim) continue;
        outDims.emplace_back(inShape.GetDim(i));
    }
    if (rank == 0) {
        outDims.push_back(1);
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = outDims;
    return GRAPH_SUCCESS;
}

static graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

#define REDUCE_TILING_0 1
#define REDUCE_TILING_1 2
#define REDUCE_TILING_2 3

class KernelMinReduce {
static constexpr uint32_t DEFAULT_BLK_STRIDE = 1;
static constexpr uint32_t DEFAULT_REP_STRIDE = 8;
static constexpr uint32_t REP_BYTES = 256;
public:
    __aicore__ inline KernelMinReduce() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z,
                                uint32_t d0, uint32_t d1, uint32_t d2,
                                uint32_t reduceDim, uint32_t outLength)
    {
        this->d0 = d0;
        this->d1 = d1;
        this->d2 = d2;
        this->reduceDim = reduceDim;
        this->outLength = outLength;

        xGm.SetGlobalBuffer((__gm__ float *)x, static_cast<uint64_t>(d0) * d1 * d2);
        zGm.SetGlobalBuffer((__gm__ float *)z, outLength);

        pipe.InitBuffer(outQueueZ, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process(uint32_t tilingKey)
    {
        // Currently implement reduction across the last dimension efficiently; other dims use generic path.
        if (reduceDim == 2 || (d0 == 1 && d1 == 1)) {
            ReduceLastDim(tilingKey);
        } else if (reduceDim == 1) {
            ReduceMiddleDim(tilingKey);
        } else {
            ReduceFirstDim(tilingKey);
        }
    }

private:
    __aicore__ inline void ReduceRowToScalar(AscendC::LocalTensor<float> &outLocal,
                                             AscendC::LocalTensor<float> &rowLocal,
                                             uint32_t rowLen, uint32_t tilingKey)
    {
        // Use vector reduction intrinsics based on size
        if (tilingKey == REDUCE_TILING_0) {
            constexpr int64_t maskLen = REP_BYTES / sizeof(float);
            AscendC::WholeReduceMin<float>(outLocal, rowLocal, maskLen, 1,
                DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
        } else if (tilingKey == REDUCE_TILING_1) {
            // First reduce inside blocks, then whole reduce
            pipe.InitBuffer(calcBuf, rowLen * sizeof(float));
            AscendC::LocalTensor<float> tmp = calcBuf.Get<float>();
            AscendC::SetMaskCount();
            AscendC::SetVectorMask<float>(0, rowLen);
            AscendC::BlockReduceMin<float, false>(tmp, rowLocal, AscendC::MASK_PLACEHOLDER, 1,
                DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
            AscendC::PipeBarrier<PIPE_V>();
            uint32_t c0Count = 32 / sizeof(float);
            uint32_t blockNum0 = (rowLen + c0Count - 1) / c0Count;
            AscendC::SetVectorMask<float>(0, blockNum0);
            AscendC::WholeReduceMin<float, false>(outLocal, tmp, AscendC::MASK_PLACEHOLDER, 1,
                DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
            AscendC::PipeBarrier<PIPE_V>();
            AscendC::SetMaskNorm();
        } else { // REDUCE_TILING_2 or default
            pipe.InitBuffer(calcBuf, rowLen * sizeof(float));
            AscendC::LocalTensor<float> tmp = calcBuf.Get<float>();
            uint32_t repeatNum = (rowLen * sizeof(float) + REP_BYTES - 1) / REP_BYTES;
            AscendC::SetMaskCount();
            AscendC::SetVectorMask<float>(0, rowLen);
            AscendC::WholeReduceMin<float, false>(tmp, rowLocal, AscendC::MASK_PLACEHOLDER, 1,
                DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
            AscendC::PipeBarrier<PIPE_V>();
            AscendC::SetVectorMask<float>(0, repeatNum);
            AscendC::WholeReduceMin<float, false>(outLocal, tmp, AscendC::MASK_PLACEHOLDER, 1,
                DEFAULT_BLK_STRIDE, DEFAULT_BLK_STRIDE, DEFAULT_REP_STRIDE);
            AscendC::PipeBarrier<PIPE_V>();
            AscendC::SetMaskNorm();
        }
    }

    __aicore__ inline void ReduceLastDim(uint32_t tilingKey)
    {
        // For each [d0, d1] row reduce over contiguous d2
        uint64_t outer = static_cast<uint64_t>(d0) * d1;
        // allocate once max rowLen
        pipe.InitBuffer(rowBuf, static_cast<uint32_t>(d2) * sizeof(float));
        for (uint64_t i = 0; i < outer; ++i) {
            uint64_t xOffset = i * d2;
            AscendC::LocalTensor<float> rowLocal = rowBuf.Get<float>();
            AscendC::DataCopy(rowLocal, xGm[xOffset], d2);

            AscendC::LocalTensor<float> outLocal = outQueueZ.AllocTensor<float>();
            ReduceRowToScalar(outLocal, rowLocal, d2, tilingKey);

            // Write out a scalar
            AscendC::DataCopy(zGm[i], outLocal, 1);
            outQueueZ.FreeTensor(outLocal);
        }
    }

    __aicore__ inline void ReduceMiddleDim(uint32_t tilingKey)
    {
        // Reduce over d1: output shape [d0, d2], iterate i0 in d0, i2 in d2
        pipe.InitBuffer(rowBuf, d1 * sizeof(float));
        for (uint32_t i0 = 0; i0 < d0; ++i0) {
            for (uint32_t i2 = 0; i2 < d2; ++i2) {
                AscendC::LocalTensor<float> rowLocal = rowBuf.Get<float>();
                // Gather along middle dim
                for (uint32_t j = 0; j < d1; ++j) {
                    uint64_t xIndex = static_cast<uint64_t>(i0) * d1 * d2 + j * d2 + i2;
                    // Place into rowLocal[j]
                    AscendC::DataCopy(rowLocal[j], xGm[xIndex], 1);
                }
                AscendC::LocalTensor<float> outLocal = outQueueZ.AllocTensor<float>();
                ReduceRowToScalar(outLocal, rowLocal, d1, tilingKey);

                uint64_t outIndex = static_cast<uint64_t>(i0) * d2 + i2;
                AscendC::DataCopy(zGm[outIndex], outLocal, 1);
                outQueueZ.FreeTensor(outLocal);
            }
        }
    }

    __aicore__ inline void ReduceFirstDim(uint32_t tilingKey)
    {
        // Reduce over d0: output shape [d1, d2], iterate i1 in d1, i2 in d2
        pipe.InitBuffer(rowBuf, d0 * sizeof(float));
        for (uint32_t i1 = 0; i1 < d1; ++i1) {
            for (uint32_t i2 = 0; i2 < d2; ++i2) {
                AscendC::LocalTensor<float> rowLocal = rowBuf.Get<float>();
                // Gather along first dim
                for (uint32_t j = 0; j < d0; ++j) {
                    uint64_t xIndex = static_cast<uint64_t>(j) * d1 * d2 + static_cast<uint64_t>(i1) * d2 + i2;
                    AscendC::DataCopy(rowLocal[j], xGm[xIndex], 1);
                }
                AscendC::LocalTensor<float> outLocal = outQueueZ.AllocTensor<float>();
                ReduceRowToScalar(outLocal, rowLocal, d0, tilingKey);

                uint64_t outIndex = static_cast<uint64_t>(i1) * d2 + i2;
                AscendC::DataCopy(zGm[outIndex], outLocal, 1);
                outQueueZ.FreeTensor(outLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBuf;
    AscendC::TBuf<AscendC::TPosition::VECIN> rowBuf;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t d0;
    uint32_t d1;
    uint32_t d2;
    uint32_t reduceDim;
    uint32_t outLength;
};

extern "C" __global__ __aicore__ void min_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMinReduce op;
    op.Init(x, z,
            tiling_data.d0, tiling_data.d1, tiling_data.d2,
            tiling_data.reduceDim, tiling_data.outLength);
    if (TILING_KEY_IS(REDUCE_TILING_0)) {
        op.Process(REDUCE_TILING_0);
    } else if (TILING_KEY_IS(REDUCE_TILING_1)) {
        op.Process(REDUCE_TILING_1);
    } else if (TILING_KEY_IS(REDUCE_TILING_2)) {
        op.Process(REDUCE_TILING_2);
    } else {
        op.Process(REDUCE_TILING_2);
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline std::vector<int64_t> compute_min_reduce_out_sizes(const at::IntArrayRef& in_sizes, int64_t dim) {
    std::vector<int64_t> out_sizes;
    int64_t rank = in_sizes.size();
    int64_t d = dim;
    if (d < 0) d += rank;
    for (int64_t i = 0; i < rank; ++i) {
        if (i == d) continue;
        out_sizes.push_back(in_sizes[i]);
    }
    if (rank == 0) {
        out_sizes.push_back(1);
    }
    return out_sizes;
}

at::Tensor min_reduction_over_a_dimension_impl_npu(const at::Tensor& x, int64_t dim) {
    auto out_sizes = compute_min_reduce_out_sizes(x.sizes(), dim);
    at::Tensor result = at::empty(out_sizes, x.options());
    EXEC_NPU_CMD(aclnnMinReductionOverADimensionCustom, x, dim, result);
    return result;
}

TORCH_LIBRARY(myops, m) {
    m.def("min_reduction_over_a_dimension_custom(Tensor x, int dim) -> Tensor");
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("min_reduction_over_a_dimension_custom", &min_reduction_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("min_reduction_over_a_dimension_custom", &min_reduction_over_a_dimension_impl_npu, "min reduction over a dimension");
}
