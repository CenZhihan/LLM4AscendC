#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_padded_dilated_strided_custom(
    GM_ADDR x,
    GM_ADDR w,
    GM_ADDR b,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N   = tiling_data.N;
    const uint32_t Cin = tiling_data.Cin;
    const uint32_t Cout= tiling_data.Cout;
    const uint32_t Din = tiling_data.Din;
    const uint32_t Hin = tiling_data.Hin;
    const uint32_t Win = tiling_data.Win;
    const uint32_t K   = tiling_data.K;
    const uint32_t Stride   = tiling_data.Stride;
    const uint32_t Padding  = tiling_data.Padding;
    const uint32_t Dilation = tiling_data.Dilation;
    const uint32_t Dout = tiling_data.Dout;
    const uint32_t Hout = tiling_data.Hout;
    const uint32_t Wout = tiling_data.Wout;

    __gm__ float* x_gm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_gm = reinterpret_cast<__gm__ float*>(w);
    __gm__ float* b_gm = reinterpret_cast<__gm__ float*>(b);
    __gm__ float* y_gm = reinterpret_cast<__gm__ float*>(y);

    // Partition work across blocks by (N * Cout)
    const uint32_t totalOC = N * Cout;
    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blkIdx = AscendC::GetBlockIdx();
    const uint32_t ocPerBlock = (totalOC + blockNum - 1) / blockNum;
    const uint32_t ocBegin = blkIdx * ocPerBlock;
    const uint32_t ocEnd = AscendC::Min(totalOC, ocBegin + ocPerBlock);

    for (uint32_t oc = ocBegin; oc < ocEnd; ++oc) {
        const uint32_t n = oc / Cout;
        const uint32_t cout = oc % Cout;

        for (uint32_t d = 0; d < Dout; ++d) {
            for (uint32_t h = 0; h < Hout; ++h) {
                for (uint32_t wout = 0; wout < Wout; ++wout) {
                    float acc = 0.0f;

                    // Iterate over inputs and kernel
                    for (uint32_t cin = 0; cin < Cin; ++cin) {
                        for (uint32_t kd = 0; kd < K; ++kd) {
                            int64_t din_num = static_cast<int64_t>(d) + static_cast<int64_t>(Padding) - static_cast<int64_t>(kd) * static_cast<int64_t>(Dilation);
                            if (din_num < 0) continue;
                            if (din_num % static_cast<int64_t>(Stride) != 0) continue;
                            uint32_t din = static_cast<uint32_t>(din_num / static_cast<int64_t>(Stride));
                            if (din >= Din) continue;

                            for (uint32_t kh = 0; kh < K; ++kh) {
                                int64_t hin_num = static_cast<int64_t>(h) + static_cast<int64_t>(Padding) - static_cast<int64_t>(kh) * static_cast<int64_t>(Dilation);
                                if (hin_num < 0) continue;
                                if (hin_num % static_cast<int64_t>(Stride) != 0) continue;
                                uint32_t hin = static_cast<uint32_t>(hin_num / static_cast<int64_t>(Stride));
                                if (hin >= Hin) continue;

                                for (uint32_t kw = 0; kw < K; ++kw) {
                                    int64_t win_num = static_cast<int64_t>(wout) + static_cast<int64_t>(Padding) - static_cast<int64_t>(kw) * static_cast<int64_t>(Dilation);
                                    if (win_num < 0) continue;
                                    if (win_num % static_cast<int64_t>(Stride) != 0) continue;
                                    uint32_t win = static_cast<uint32_t>(win_num / static_cast<int64_t>(Stride));
                                    if (win >= Win) continue;

                                    // x index: [n, cin, din, hin, win]
                                    uint64_t x_idx = (((static_cast<uint64_t>(n) * Cin + cin) * Din + din) * Hin + hin) * Win + win;
                                    // w index: [cin, cout, kd, kh, kw]
                                    uint64_t w_idx = ((((static_cast<uint64_t>(cin) * Cout) + cout) * K + kd) * K + kh) * K + kw;

                                    acc += x_gm[x_idx] * w_gm[w_idx];
                                }
                            }
                        }
                    }

                    if (b_gm != nullptr) {
                        acc += b_gm[cout];
                    }

                    // y index: [n, cout, d, h, w]
                    uint64_t y_idx = (((static_cast<uint64_t>(n) * Cout + cout) * Dout + d) * Hout + h) * Wout + wout;
                    y_gm[y_idx] = acc;
                }
            }
        }
    }
}

static inline std::vector<int64_t> compute_output_shape_3d_transpose(
    const at::Tensor& x, const at::Tensor& weight,
    int64_t stride, int64_t padding, int64_t dilation) {

    // x: [N, Cin, D, H, W]
    // weight: [Cin, Cout, K, K, K] (square kernel)
    int64_t N = x.size(0);
    int64_t Cin = x.size(1);
    int64_t D = x.size(2);
    int64_t H = x.size(3);
    int64_t W = x.size(4);

    int64_t Cout = weight.size(1);
    int64_t K = weight.size(2);

    int64_t Dout = (D - 1) * stride - 2 * padding + dilation * (K - 1) + 1;
    int64_t Hout = (H - 1) * stride - 2 * padding + dilation * (K - 1) + 1;
    int64_t Wout = (W - 1) * stride - 2 * padding + dilation * (K - 1) + 1;

    return {N, Cout, Dout, Hout, Wout};
}
