kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConcatC {
public:
    __aicore__ inline KernelConcatC() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z, uint32_t firstLength, uint32_t secondLength, uint32_t tileNum)
    {
        this->blockLenX = firstLength / AscendC::GetBlockNum();
        this->blockLenY = secondLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLenX = this->blockLenX / (tileNum * BUFFER_NUM);
        this->tileLenY = this->blockLenY / (tileNum * BUFFER_NUM);

        // Guard against very small tensors
        if (this->tileLenX == 0) this->tileLenX = this->blockLenX;
        if (this->tileLenY == 0) this->tileLenY = this->blockLenY;

        uint32_t blockIdx = AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLenX * blockIdx, this->blockLenX);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLenY * blockIdx, this->blockLenY);
        // z base covers whole output to allow offset addressing
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, firstLength + secondLength);

        uint32_t maxTile = this->tileLenX > this->tileLenY ? this->tileLenX : this->tileLenY;
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, maxTile * sizeof(DTYPE_Z));

        this->zBaseX = this->blockLenX * blockIdx;
        this->zBaseY = firstLength + this->blockLenY * blockIdx;
    }

    __aicore__ inline void Process()
    {
        // Copy X branch into front channels
        int32_t loopX = AscendC::DivCeil(this->blockLenX, this->tileLenX);
        for (int32_t i = 0; i < loopX; ++i) {
            uint32_t curLen = ((i + 1) * this->tileLenX <= this->blockLenX) ? this->tileLenX : (this->blockLenX - i * this->tileLenX);
            CopyBlock(xGm, zBaseX, i, curLen);
        }

        // Copy Y branch into tail channels
        int32_t loopY = AscendC::DivCeil(this->blockLenY, this->tileLenY);
        for (int32_t i = 0; i < loopY; ++i) {
            uint32_t curLen = ((i + 1) * this->tileLenY <= this->blockLenY) ? this->tileLenY : (this->blockLenY - i * this->tileLenY);
            CopyBlock(yGm, zBaseY, i, curLen);
        }
    }

private:
    template<typename GmTensorT>
    __aicore__ inline void CopyBlock(GmTensorT& srcGm, uint32_t zBase, int32_t idx, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_Z> tmpLocal = tmpQueue.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(tmpLocal, srcGm[idx * ((GmTensorT*)&srcGm == &xGm ? this->tileLenX : this->tileLenY)], curLen);
        tmpQueue.EnQue(tmpLocal);
        AscendC::LocalTensor<DTYPE_Z> outLocal = tmpQueue.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[zBase + idx * ((GmTensorT*)&srcGm == &xGm ? this->tileLenX : this->tileLenY)], outLocal, curLen);
        tmpQueue.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    uint32_t blockLenX;
    uint32_t blockLenY;
    uint32_t tileNum;
    uint32_t tileLenX;
    uint32_t tileLenY;
    uint32_t zBaseX;
    uint32_t zBaseY;
};

extern "C" __global__ __aicore__ void squeeze_net_fire_module_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConcatC op;
    op.Init(x, y, z, tiling_data.firstLength, tiling_data.secondLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, squeeze_channels, expand1x1_channels, expand3x3_channels):
        super().__init__()
        self.squeeze = torch.nn.Conv2d(in_channels, squeeze_channels, kernel_size=1)
        self.squeeze_activation = torch.nn.ReLU(inplace=True)

        self.expand1x1 = torch.nn.Conv2d(squeeze_channels, expand1x1_channels, kernel_size=1)
        self.expand1x1_activation = torch.nn.ReLU(inplace=True)

        self.expand3x3 = torch.nn.Conv2d(squeeze_channels, expand3x3_channels, kernel_size=3, padding=1)
        self.expand3x3_activation = torch.nn.ReLU(inplace=True)

    def forward(self, x):
        x = self.squeeze_activation(self.squeeze(x))
        out1 = self.expand1x1_activation(self.expand1x1(x))
        out3 = self.expand3x3_activation(self.expand3x3(x))
        return custom_ops_lib.squeeze_net_fire_module_custom(out1, out3)
'''
