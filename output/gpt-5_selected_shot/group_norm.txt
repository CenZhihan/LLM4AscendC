#include "group_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 48;

static ge::graphStatus TilingFunc(gert::TilingContext* context) {
  GroupNormCustomTilingData tiling;

  // Input shape
  const gert::StorageShape* x_shape = context->GetInputShape(0);
  const gert::Shape shape = x_shape->GetStorageShape();
  uint32_t dimNum = shape.GetDimNum();
  if (dimNum < 2) {
    return ge::GRAPH_FAILED;
  }
  uint64_t N = static_cast<uint64_t>(shape.GetDim(0));
  uint64_t C = static_cast<uint64_t>(shape.GetDim(1));
  uint64_t inner = 1;
  for (uint32_t i = 2; i < dimNum; ++i) {
    inner *= static_cast<uint64_t>(shape.GetDim(i));
  }

  // Attributes
  const int32_t* numGroupsAttr = context->GetAttrs()->GetAttrPointer<int32_t>(0); // index 0: num_groups
  const float* epsilonAttr = context->GetAttrs()->GetAttrPointer<float>(1);        // index 1: epsilon

  uint32_t G = static_cast<uint32_t>(*numGroupsAttr);
  if (G == 0) {
    return ge::GRAPH_FAILED;
  }
  uint32_t channelsPerGroup = static_cast<uint32_t>(C / G);
  uint64_t rowsTotal = N * G;
  uint64_t rowLength = channelsPerGroup * inner;

  // Distribute rows across blocks
  uint32_t coreRowNum = static_cast<uint32_t>(rowsTotal / BLOCK_DIM);
  uint32_t blockPivot = static_cast<uint32_t>(rowsTotal - static_cast<uint64_t>(coreRowNum) * BLOCK_DIM);

  tiling.set_eps(*epsilonAttr);
  tiling.set_rowNum(coreRowNum);
  tiling.set_rowNumSp(coreRowNum + 1);
  tiling.set_rowLength(static_cast<uint32_t>(rowLength));
  tiling.set_blockPivot(blockPivot);

  uint32_t tileLoop = 8; // rows per tile
  tiling.set_tileLoop(tileLoop);
  tiling.set_tileLength(static_cast<uint32_t>(tileLoop * rowLength));
  tiling.set_loopCount(coreRowNum / tileLoop);

  float factor = 1.0f / static_cast<float>(rowLength);
  tiling.set_factor(factor);
  tiling.set_mfactor(-factor);

  tiling.set_numGroups(G);
  tiling.set_channelsPerGroup(channelsPerGroup);
  tiling.set_innerSize(static_cast<uint32_t>(inner));
  tiling.set_channelCount(static_cast<uint32_t>(C));

  context->SetBlockDim(BLOCK_DIM);
  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
  context->SetTilingKey(1);

  return ge::GRAPH_SUCCESS;
}
}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context) {
  const gert::Shape* x_shape = context->GetInputShape(0);
  gert::Shape* y_shape = context->GetOutputShape(0);
  *y_shape = *x_shape;
  return GRAPH_SUCCESS;
}
}  // namespace ge

kernel_src="""
#include "kernel_operator.h"
constexpr int32_t BUFFER_NUM = 1;

class KernelGroupNorm {
 public:
  __aicore__ inline KernelGroupNorm() {}
  __aicore__ inline void InitTiling(GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    rowNumCore = tiling_data.rowNum;
    rowNum = tiling_data.rowNum;
    rowNumSp = tiling_data.rowNumSp;
    rowLength = tiling_data.rowLength;
    blockPivot = tiling_data.blockPivot;
    tileLoop = tiling_data.tileLoop;
    tileLength = tiling_data.tileLength;
    loopCount = tiling_data.loopCount;
    factor = tiling_data.factor;
    mfactor = tiling_data.mfactor;
    eps = tiling_data.eps;
    numGroups = tiling_data.numGroups;
    channelsPerGroup = tiling_data.channelsPerGroup;
    innerSize = tiling_data.innerSize;
    channelCount = tiling_data.channelCount;
  }
  __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR z,
                              GM_ADDR tiling) {
    InitTiling(tiling);

    leftRow = rowNum % tileLoop;
    if (AscendC::GetBlockIdx() < blockPivot) {
      rowNum = rowNumSp;
      leftRow += 1;
    }

    blockLength = rowNum * rowLength;

    // Compute start row index for this block
    uint32_t blockStartRow = 0;
    if (AscendC::GetBlockIdx() < blockPivot) {
      blockStartRow = AscendC::GetBlockIdx() * rowNumSp;
    } else {
      blockStartRow = blockPivot * rowNumSp + (AscendC::GetBlockIdx() - blockPivot) * rowNumCore;
    }
    this->baseRowIndex = blockStartRow;

    uint64_t elementOffset = static_cast<uint64_t>(blockStartRow) * static_cast<uint64_t>(rowLength);

    xGm.SetGlobalBuffer((__gm__ float *)x + elementOffset, blockLength);
    zGm.SetGlobalBuffer((__gm__ float *)z + elementOffset, blockLength);

    gammaGm.SetGlobalBuffer((__gm__ float *)gamma, channelCount);
    betaGm.SetGlobalBuffer((__gm__ float *)beta, channelCount);

    pipe.InitBuffer(queueX, BUFFER_NUM, tileLength * sizeof(float));
    pipe.InitBuffer(queueZ, BUFFER_NUM, tileLength * sizeof(float));

    pipe.InitBuffer(tmpBuffer1, 64 * sizeof(float));
    pipe.InitBuffer(tmpBuffer2, 64 * sizeof(float));
    pipe.InitBuffer(onesBuffer, 64 * sizeof(float));

    pipe.InitBuffer(queueGamma, 1, channelCount * sizeof(float));
    pipe.InitBuffer(queueBeta, 1, channelCount * sizeof(float));
  }
  __aicore__ inline void Process() {
    for (uint32_t i = 0; i < loopCount; i++) {
      CopyIn(i, tileLoop);
      Compute(i, tileLoop);
      CopyOut(i, tileLoop);
    }
    if (leftRow > 0) {
      CopyIn(loopCount, leftRow);
      Compute(loopCount, leftRow);
      CopyOut(loopCount, leftRow);
    }
  }

 private:
  __aicore__ inline void CopyIn(uint32_t progress, uint32_t rows) {
    AscendC::LocalTensor<float> xLocal = queueX.AllocTensor<float>();
    AscendC::LocalTensor<float> gammaLocal = queueGamma.AllocTensor<float>();
    AscendC::LocalTensor<float> betaLocal = queueBeta.AllocTensor<float>();

    AscendC::DataCopy(xLocal, xGm[progress * tileLength], rowLength * rows);
    AscendC::DataCopy(gammaLocal, gammaGm[0], channelCount);
    AscendC::DataCopy(betaLocal, betaGm[0], channelCount);

    queueX.EnQue(xLocal);
    queueGamma.EnQue(gammaLocal);
    queueBeta.EnQue(betaLocal);
  }

  __aicore__ inline void Compute(uint32_t progress, uint32_t rows) {
    AscendC::LocalTensor<float> xLocal = queueX.DeQue<float>();
    AscendC::LocalTensor<float> gammaLocal = queueGamma.DeQue<float>();
    AscendC::LocalTensor<float> betaLocal = queueBeta.DeQue<float>();

    AscendC::LocalTensor<float> tmpTensor1 = tmpBuffer1.Get<float>();
    AscendC::LocalTensor<float> tmpTensor2 = tmpBuffer2.Get<float>();
    AscendC::LocalTensor<float> onesLocal = onesBuffer.Get<float>();
    AscendC::LocalTensor<float> zLocal = queueZ.AllocTensor<float>();

    AscendC::Duplicate<float>(onesLocal, 1.0f, tileLoop);

    // Compute mean per row
    for (uint32_t j = 0; j < rows; ++j) {
      uint32_t buffIndex = j * rowLength;
      AscendC::ReduceSum<float>(tmpTensor2[j], xLocal[buffIndex], tmpTensor1, rowLength);
    }

    // Center x by mean: x = x + (-mean)
    AscendC::Muls(zLocal, tmpTensor2, mfactor, rows);
    for (uint32_t j = 0; j < rows; ++j) {
      uint32_t buffIndex = j * rowLength;
      AscendC::Adds(xLocal[buffIndex], xLocal[buffIndex], zLocal.GetValue(j), rowLength);
    }

    // Variance: mean of squared centered values
    for (uint32_t j = 0; j < rows; ++j) {
      uint32_t buffIndex = j * rowLength;
      AscendC::Mul(zLocal[buffIndex], xLocal[buffIndex], xLocal[buffIndex], rowLength);
    }
    for (uint32_t j = 0; j < rows; ++j) {
      uint32_t buffIndex = j * rowLength;
      AscendC::ReduceSum<float>(tmpTensor2[j], zLocal[buffIndex], tmpTensor1, rowLength);
    }

    AscendC::Muls(tmpTensor2, tmpTensor2, factor, rows);  // variance
    AscendC::Adds(tmpTensor2, tmpTensor2, eps, rows);     // variance + eps
    AscendC::Sqrt(tmpTensor2, tmpTensor2, rows);          // std
    AscendC::Div(tmpTensor2, onesLocal, tmpTensor2, rows); // inv_std

    // Normalize: z = x * inv_std
    for (uint32_t j = 0; j < rows; ++j) {
      uint32_t buffIndex = j * rowLength;
      AscendC::Muls(zLocal[buffIndex], xLocal[buffIndex], tmpTensor2.GetValue(j), rowLength);
    }

    // Apply per-channel affine within each group
    for (uint32_t j = 0; j < rows; ++j) {
      uint32_t buffIndex = j * rowLength;
      // Compute global row index
      uint32_t globalRow = baseRowIndex + progress * tileLoop + j;
      uint32_t groupIndex = (numGroups > 0) ? (globalRow % numGroups) : 0;
      uint32_t groupStartChannel = groupIndex * channelsPerGroup;

      for (uint32_t c = 0; c < channelsPerGroup; ++c) {
        uint32_t chIndex = groupStartChannel + c;
        float g = gammaLocal.GetValue(chIndex);
        float b = betaLocal.GetValue(chIndex);
        uint32_t chOffset = buffIndex + c * innerSize;

        AscendC::Muls(zLocal[chOffset], zLocal[chOffset], g, innerSize);
        AscendC::Adds(zLocal[chOffset], zLocal[chOffset], b, innerSize);
      }
    }

    queueZ.EnQue<float>(zLocal);
    queueGamma.FreeTensor(gammaLocal);
    queueBeta.FreeTensor(betaLocal);
    queueX.FreeTensor(xLocal);
  }

  __aicore__ inline void CopyOut(uint32_t progress, uint32_t rows) {
    AscendC::LocalTensor<float> zLocal = queueZ.DeQue<float>();
    AscendC::DataCopy(zGm[progress * tileLength], zLocal, rows * rowLength);
    queueZ.FreeTensor(zLocal);
  }

 private:
  AscendC::TPipe pipe;
  AscendC::TBuf<AscendC::QuePosition::VECCALC> tmpBuffer1, tmpBuffer2, onesBuffer;
  AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> queueX;
  AscendC::TQue<AscendC::QuePosition::VECIN, 1> queueGamma, queueBeta;
  AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> queueZ;

  AscendC::GlobalTensor<float> xGm;
  AscendC::GlobalTensor<float> gammaGm;
  AscendC::GlobalTensor<float> betaGm;
  AscendC::GlobalTensor<float> zGm;

  uint32_t blockLength = 0;
  uint32_t leftRow = 0;
  uint32_t rowNum = 0;         // rows assigned to this block
  uint32_t rowNumCore = 0;     // base rows per block before pivot
  uint32_t rowNumSp = 0;
  uint32_t rowLength = 0;
  uint32_t blockPivot = 0;
  uint32_t tileLoop = 0;
  uint32_t tileLength = 0;
  uint32_t loopCount = 0;
  float factor = 0.0f;
  float mfactor = 0.0f;
  float eps = 1e-5f;

  uint32_t numGroups = 1;
  uint32_t channelsPerGroup = 1;
  uint32_t innerSize = 1;
  uint32_t channelCount = 1;
  uint32_t baseRowIndex = 0;
};

extern "C" __global__ __aicore__ void group_norm_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR res_out, GM_ADDR workspace,
    GM_ADDR tiling) {
  KernelGroupNorm op;
  op.Init(x, gamma, beta, res_out, tiling);
  if (TILING_KEY_IS(1)) {
    op.Process();
  }
}
