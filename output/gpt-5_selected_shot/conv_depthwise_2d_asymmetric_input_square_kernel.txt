kernel_src="""
#include "kernel_operator.h"

extern \"C\" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.n;
    const uint32_t C_IN = tiling_data.c_in;
    const uint32_t C_OUT = tiling_data.c_out; // for depthwise multiplier=1, C_OUT == C_IN
    const uint32_t H_IN = tiling_data.h_in;
    const uint32_t W_IN = tiling_data.w_in;
    const uint32_t K = tiling_data.k;
    const uint32_t STRIDE = tiling_data.stride;   // assumed 1 per tiling
    const uint32_t PADDING = tiling_data.padding; // assumed 0 per tiling
    const uint32_t H_OUT = tiling_data.h_out;
    const uint32_t W_OUT = tiling_data.w_out;

    __gm__ float* x_gm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_gm = reinterpret_cast<__gm__ float*>(w);
    __gm__ float* y_gm = reinterpret_cast<__gm__ float*>(y);

    uint32_t block_num = AscendC::GetBlockNum();
    uint32_t block_idx = AscendC::GetBlockIdx();

    // Split channels across blocks
    uint32_t channels_per_block = (C_OUT + block_num - 1) / block_num;
    uint32_t c_begin = block_idx * channels_per_block;
    uint32_t c_end = c_begin + channels_per_block;
    if (c_end > C_OUT) c_end = C_OUT;

    // Offsets helpers
    auto x_index = [&](uint32_t n, uint32_t c, uint32_t h, uint32_t w) -> uint64_t {
        return (((uint64_t)n * C_IN + c) * H_IN + h) * W_IN + w;
    };
    auto y_index = [&](uint32_t n, uint32_t c, uint32_t h, uint32_t w) -> uint64_t {
        return (((uint64_t)n * C_OUT + c) * H_OUT + h) * W_OUT + w;
    };
    auto w_index = [&](uint32_t c, uint32_t kh, uint32_t kw) -> uint64_t {
        // weight shape [C_OUT, 1, K, K]
        return ((uint64_t)c * K + kh) * K + kw;
    };

    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t c = c_begin; c < c_end; ++c) {
            for (uint32_t ho = 0; ho < H_OUT; ++ho) {
                for (uint32_t wo = 0; wo < W_OUT; ++wo) {
                    float acc = 0.0f;
                    // Compute convolution for this output
                    for (uint32_t kh = 0; kh < K; ++kh) {
                        int32_t hi = static_cast<int32_t>(ho) * static_cast<int32_t>(STRIDE) - static_cast<int32_t>(PADDING) + static_cast<int32_t>(kh);
                        if (hi < 0 || hi >= static_cast<int32_t>(H_IN)) continue;
                        for (uint32_t kw = 0; kw < K; ++kw) {
                            int32_t wi = static_cast<int32_t>(wo) * static_cast<int32_t>(STRIDE) - static_cast<int32_t>(PADDING) + static_cast<int32_t>(kw);
                            if (wi < 0 || wi >= static_cast<int32_t>(W_IN)) continue;
                            float xv = x_gm[x_index(n, c, static_cast<uint32_t>(hi), static_cast<uint32_t>(wi))];
                            float fv = w_gm[w_index(c, kh, kw)];
                            acc += xv * fv;
                        }
                    }
                    y_gm[y_index(n, c, ho, wo)] = acc;
                }
            }
        }
    }
}

static inline int64_t calc_out(int64_t in, int64_t k, int64_t stride, int64_t padding) {
    return (in + 2 * padding - k) / stride + 1;
}

at::Tensor conv_depthwise_2d_asymmetric_input_square_kernel_custom_impl_npu(
    const at::Tensor& x,
    const at::Tensor& weight,
    int64_t stride,
    int64_t padding) {

    TORCH_CHECK(x.dim() == 4, "x must be NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be [C_out, 1, K, K] for depthwise");
    TORCH_CHECK(weight.size(1) == 1, "depthwise weight channel multiplier 1 expected");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "only float32 supported");
    TORCH_CHECK(weight.scalar_type() == at::kFloat, "only float32 supported");

    int64_t N = x.size(0);
    int64_t C_in = x.size(1);
    int64_t H_in = x.size(2);
    int64_t W_in = x.size(3);
    int64_t C_out = weight.size(0);
    int64_t K = weight.size(2);

    int64_t H_out = calc_out(H_in, K, stride, padding);
    int64_t W_out = calc_out(W_in, K, stride, padding);

    auto opts = x.options();
    at::Tensor result = at::empty({N, C_out, H_out, W_out}, opts);

    // Current kernel assumes stride=1, padding=0; Ensure runtime consistency for now.
    TORCH_CHECK(stride == 1 && padding == 0, "custom kernel currently supports stride=1 and padding=0");

    EXEC_NPU_CMD(aclnnConvDepthwise2dAsymmetricInputSquareKernelCustom, x, weight, result);
    return result;

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized depthwise 2D convolution using a custom AscendC kernel.
    Assumes square kernel, stride=1, padding=0, and depthwise groups=in_channels with multiplier=1.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        # Keep nn.Conv2d only for weight/bias storage and shape convenience
        self.conv2d = torch.nn.Conv2d(in_channels, out_channels, kernel_size=(kernel_size, kernel_size), stride=stride, padding=padding, groups=in_channels, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        s = self.conv2d.stride[0] if isinstance(self.conv2d.stride, tuple) else int(self.conv2d.stride)
        p = self.conv2d.padding[0] if isinstance(self.conv2d.padding, tuple) else int(self.conv2d.padding)
        y = custom_ops_lib.conv_depthwise_2d_asymmetric_input_square_kernel_custom(x, self.conv2d.weight, s, p)
        if self.conv2d.bias is not None:
            y = y + self.conv2d.bias.view(1, -1, 1, 1)
        return y
