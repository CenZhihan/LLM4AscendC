kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConv2dMinAddMultiply {
public:
    __aicore__ inline KernelConv2dMinAddMultiply() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR cval, GM_ADDR scale, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_BIAS *)bias + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        cvalGm.SetGlobalBuffer((__gm__ DTYPE_CVAL *)cval + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_SCALE *)scale + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueBias, BUFFER_NUM, this->tileLength * sizeof(DTYPE_BIAS));
        pipe.InitBuffer(inQueueCval, BUFFER_NUM, this->tileLength * sizeof(DTYPE_CVAL));
        pipe.InitBuffer(inQueueScale, BUFFER_NUM, this->tileLength * sizeof(DTYPE_SCALE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_BIAS> biasLocal = inQueueBias.AllocTensor<DTYPE_BIAS>();
        AscendC::LocalTensor<DTYPE_CVAL> cvalLocal = inQueueCval.AllocTensor<DTYPE_CVAL>();
        AscendC::LocalTensor<DTYPE_SCALE> scaleLocal = inQueueScale.AllocTensor<DTYPE_SCALE>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(biasLocal, biasGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(cvalLocal, cvalGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(scaleLocal, scaleGm[progress * this->tileLength], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueBias.EnQue(biasLocal);
        inQueueCval.EnQue(cvalLocal);
        inQueueScale.EnQue(scaleLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_BIAS> biasLocal = inQueueBias.DeQue<DTYPE_BIAS>();
        AscendC::LocalTensor<DTYPE_CVAL> cvalLocal = inQueueCval.DeQue<DTYPE_CVAL>();
        AscendC::LocalTensor<DTYPE_SCALE> scaleLocal = inQueueScale.DeQue<DTYPE_SCALE>();

        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // y = (min(x, cval) + bias) * scale
        AscendC::Min(yLocal, xLocal, cvalLocal, this->tileLength);
        AscendC::Add(yLocal, yLocal, biasLocal, this->tileLength);
        AscendC::Mul(yLocal, yLocal, scaleLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Y>(yLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueBias.FreeTensor(biasLocal);
        inQueueCval.FreeTensor(cvalLocal);
        inQueueScale.FreeTensor(scaleLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueBias, inQueueCval, inQueueScale;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_BIAS> biasGm;
    AscendC::GlobalTensor<DTYPE_CVAL> cvalGm;
    AscendC::GlobalTensor<DTYPE_SCALE> scaleGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR cval, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinAddMultiply op;
    op.Init(x, bias, cval, scale, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Optimized model that keeps Conv2d in PyTorch/NPU and fuses min-with-const, add-bias and multiply-scale
    into a single custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, constant_value, bias_shape, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv = nn.Conv2d(in_channels, out_channels, kernel_size)
        self.constant_value = constant_value
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.scaling_factor = scaling_factor

    def forward(self, x):
        x = self.conv(x)  # [N, C, H, W]
        # Prepare broadcasted tensors for the fused custom op
        # bias: per-channel broadcast to [N, C, H, W]
        bias_broadcasted = self.bias.view(1, -1, 1, 1).expand_as(x).contiguous()
        # cval and scale: scalar broadcast to [N, C, H, W]
        cval_broadcasted = torch.full_like(x, self.constant_value)
        scale_broadcasted = torch.full_like(x, self.scaling_factor)
        # Custom fused op: y = (min(x, cval) + bias) * scale
        x = custom_ops_lib.conv2d_min_add_multiply_custom(x, bias_broadcasted, cval_broadcasted, scale_broadcasted)
        return x
'''
