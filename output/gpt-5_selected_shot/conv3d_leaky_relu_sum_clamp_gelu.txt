kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR s, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        sGm.SetGlobalBuffer((__gm__ DTYPE_Y *)s + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueS, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> sLocal = inQueueS.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(sLocal, sGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueS.EnQue(sLocal);
    }

    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> sLocal = inQueueS.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // yLocal = min(x, 0)  -> negative part
        AscendC::Minimums(yLocal, xLocal, (DTYPE_Z)0.0f, this->tileLength);
        // yLocal = -0.8 * yLocal
        AscendC::Muls(yLocal, yLocal, (DTYPE_Z)(-0.8f), this->tileLength);
        // yLocal = yLocal + xLocal => leaky_relu with alpha=0.2
        AscendC::Add(yLocal, yLocal, xLocal, this->tileLength);

        // yLocal = yLocal + sLocal
        AscendC::Add(yLocal, yLocal, sLocal, this->tileLength);

        // clamp to [-1.0, 1.0]
        AscendC::Minimums(yLocal, yLocal, (DTYPE_Z)1.0f, this->tileLength);
        AscendC::Maximums(yLocal, yLocal, (DTYPE_Z)(-1.0f), this->tileLength);

        // GELU approximate: 0.5 * x * (1 + tanh(0.7978845608 * (x + 0.044715 * x^3)))
        // reuse sLocal as temp buffer
        // sLocal = yLocal * yLocal
        AscendC::Mul(sLocal, yLocal, yLocal, this->tileLength);
        // sLocal = sLocal * yLocal = yLocal^3
        AscendC::Mul(sLocal, sLocal, yLocal, this->tileLength);
        // sLocal = 0.044715 * sLocal
        AscendC::Muls(sLocal, sLocal, (DTYPE_Y)0.044715f, this->tileLength);
        // sLocal = yLocal + sLocal
        AscendC::Add(sLocal, yLocal, sLocal, this->tileLength);
        // sLocal = 0.7978845608 * sLocal
        AscendC::Muls(sLocal, sLocal, (DTYPE_Y)0.7978845608f, this->tileLength);
        // sLocal = tanh(sLocal)
        AscendC::Tanh(sLocal, sLocal, this->tileLength);
        // sLocal = 1 + sLocal
        AscendC::Adds(sLocal, sLocal, (DTYPE_Y)1.0f, this->tileLength);
        // sLocal = 0.5 * sLocal
        AscendC::Muls(sLocal, sLocal, (DTYPE_Y)0.5f, this->tileLength);
        // yLocal = yLocal * sLocal
        AscendC::Mul(yLocal, yLocal, sLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueS.FreeTensor(sLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueS;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> sGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv3d_leaky_relu_sum_clamp_gelu_custom(GM_ADDR x, GM_ADDR s, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, s, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model where the post-convolution sequence (LeakyReLU -> +sum -> clamp -> GELU)
    is replaced with a fused custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, sum_tensor_shape):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.sum_tensor = torch.nn.Parameter(torch.randn(sum_tensor_shape))

    def forward(self, x):
        x = self.conv(x)
        # Prepare broadcasted sum tensor to match x's shape for efficient elementwise fusion
        s = self.sum_tensor.view(1, -1, 1, 1, 1).expand_as(x).contiguous()
        x = custom_ops_lib.conv3d_leaky_relu_sum_clamp_gelu_custom(x, s)
        return x
