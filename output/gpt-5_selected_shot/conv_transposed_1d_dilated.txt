        return ge::GRAPH_FAILED;

    tiling.set_cout(cout);

    tiling.set_kw(kw);
    tiling.set_lout(lout);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""
#include "kernel_operator.h"

// This kernel computes 1D transposed convolution with stride=1, padding=0.
// Dilation should be pre-applied on weights from host/Python, i.e., w has length Kw_eff = (K-1)*dilation+1.
// x: [N, Cin, Lin], w: [Cin, Cout, Kw_eff], y: [N, Cout, Lin + Kw_eff - 1]

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    uint32_t N    = tiling_data.n;
    uint32_t Cin  = tiling_data.cin;
    uint32_t Cout = tiling_data.cout;
    uint32_t Lin  = tiling_data.lin;
    uint32_t Kw   = tiling_data.kw;
    uint32_t Lout = tiling_data.lout;

    uint32_t total = tiling_data.totalLength;
    uint32_t blocks = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();

    uint32_t start = (total / blocks) * blockIdx;
    uint32_t end = (blockIdx == blocks - 1) ? total : (total / blocks) * (blockIdx + 1);

    // Cast global pointers
    __gm__ DTYPE_X* x_gm = reinterpret_cast<__gm__ DTYPE_X*>(x); // x
    __gm__ DTYPE_Y* w_gm = reinterpret_cast<__gm__ DTYPE_Y*>(w); // w
    __gm__ DTYPE_Z* y_gm = reinterpret_cast<__gm__ DTYPE_Z*>(y); // y

    // Flattened indexing helpers:
    // x_off(n, ci, t)   = ((n * Cin + ci) * Lin) + t
    // w_off(ci, co, k)  = ((ci * Cout + co) * Kw) + k
    // y_off(n, co, t)   = ((n * Cout + co) * Lout) + t

    for (uint32_t idx = start; idx < end; ++idx) {
        uint32_t t = idx % Lout;
        uint32_t tmp = idx / Lout;
        uint32_t co = tmp % Cout;
        uint32_t n  = tmp / Cout;

        DTYPE_Z acc = static_cast<DTYPE_Z>(0);
        // Accumulate over Cin and Kw
        for (uint32_t ci = 0; ci < Cin; ++ci) {
            for (uint32_t k = 0; k < Kw; ++k) {
                // Input position corresponding to this output t and kernel index k
                int32_t tin = static_cast<int32_t>(t) - static_cast<int32_t>(k);
                if (tin >= 0 && tin < static_cast<int32_t>(Lin)) {
                    uint32_t x_index = ((n * Cin + ci) * Lin) + static_cast<uint32_t>(tin);
                    uint32_t w_index = ((ci * Cout + co) * Kw) + k;
                    acc += static_cast<DTYPE_Z>(x_gm[x_index]) * static_cast<DTYPE_Z>(w_gm[w_index]);
                }
            }
        }
        uint32_t y_index = ((n * Cout + co) * Lout) + t;
        y_gm[y_index] = acc;
    }
}
"""

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static at::Tensor dilated_conv_transpose1d_shape_out(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, Cin, Lin], w: [Cin, Cout, Kw_eff]
    TORCH_CHECK(x.dim() == 3, \"x must be 3D [N, Cin, Lin]\");
    TORCH_CHECK(w.dim() == 3, \"w must be 3D [Cin, Cout, Kw_eff]\");
    TORCH_CHECK(x.size(1) == w.size(0), \"Cin of x must match Cin of w\");
    int64_t N = x.size(0);
    int64_t Cout = w.size(1);
    int64_t Lin = x.size(2);
    int64_t Kw = w.size(2);
    int64_t Lout = Lin + Kw - 1;
    auto out_sizes = at::IntArrayRef({N, Cout, Lout});
    return at::empty(out_sizes, x.options());
}

at::Tensor conv_transposed_1d_dilated_custom_impl_npu(const at::Tensor& x, const at::Tensor& w) {
    // Expect w to already be dilation-expanded along last dim: Kw_eff = (K-1)*dilation+1
    at::Tensor result = dilated_conv_transpose1d_shape_out(x, w);
    EXEC_NPU_CMD(aclnnConvTransposed1dDilatedCustom, x, w, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv_transposed_1d_dilated_custom\", &conv_transposed_1d_dilated_custom_impl_npu);
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

def _expand_dilated_weight_transpose1d(weight: torch.Tensor, dilation: int) -> torch.Tensor:
    # weight: [Cin, Cout, K]
    if dilation == 1:
        return weight
    cin, cout, k = weight.shape
    keff = (k - 1) * dilation + 1
    w_expanded = weight.new_zeros(cin, cout, keff)
    idx = torch.arange(k, device=weight.device) * dilation
    w_expanded.index_copy_(2, idx, weight)
    return w_expanded

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False) -> None:
        super().__init__()
        # Keep nn layer to manage parameters; will call custom op in forward.
        self.conv1d_transpose = nn.ConvTranspose1d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Only supporting stride=1 and padding=0 in custom op. Dilation handled by expanding weights.
        stride = self.conv1d_transpose.stride[0]
        padding = self.conv1d_transpose.padding[0]
        dilation = self.conv1d_transpose.dilation[0]
        if stride != 1 or padding != 0:
            # Fallback to PyTorch if unsupported attributes are used
            return self.conv1d_transpose(x)
        w = self.conv1d_transpose.weight
        w_expanded = _expand_dilated_weight_transpose1d(w, dilation).contiguous()
        y = custom_ops_lib.conv_transposed_1d_dilated_custom(x.contiguous(), w_expanded)
        if self.conv1d_transpose.bias is not None:
            y = y + self.conv1d_transpose.bias.view(1, -1, 1)
        return y
'''
