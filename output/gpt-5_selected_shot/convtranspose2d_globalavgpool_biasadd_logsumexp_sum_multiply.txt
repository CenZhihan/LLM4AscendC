kernel_src="""
#include "kernel_operator.h"
#include <cmath>

extern "C" __global__ __aicore__ void convtranspose2d_globalavgpool_biasadd_logsumexp_sum_multiply_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    const int32_t n = static_cast<int32_t>(tiling_data.n);
    const int32_t c = static_cast<int32_t>(tiling_data.c);
    const int32_t bIdx = AscendC::GetBlockIdx();

    if (bIdx >= n) {
        return;
    }

    __gm__ DTYPE_X* xPtr = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_Y* bPtr = reinterpret_cast<__gm__ DTYPE_Y*>(bias);
    __gm__ DTYPE_Z* yPtr = reinterpret_cast<__gm__ DTYPE_Z*>(y);

    // x is expected to be [N, C, 1, 1] (already globally averaged over H and W)
    // Compute: y[b] = 10 * logsumexp(x[b, :] + bias[:]) over channel dimension
    int64_t base = static_cast<int64_t>(bIdx) * static_cast<int64_t>(c);

    float m = -3.402823466e+38F; // -FLT_MAX
    for (int i = 0; i < c; ++i) {
        float val = static_cast<float>(xPtr[base + i]) + static_cast<float>(bPtr[i]);
        m = val > m ? val : m;
    }
    float sumExp = 0.0f;
    for (int i = 0; i < c; ++i) {
        float val = static_cast<float>(xPtr[base + i]) + static_cast<float>(bPtr[i]);
        sumExp += expf(val - m);
    }
    float out = (m + logf(sumExp)) * 10.0f;
    yPtr[bIdx] = static_cast<DTYPE_Z>(out);
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a custom AscendC operator to fuse:
    bias add + logsumexp (over channel) + sum (over spatial dims) + multiply.
    The ConvTranspose2d and GlobalAvgPool remain in PyTorch for flexibility.
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        x = self.conv_transpose(x)
        x = torch.mean(x, dim=(2, 3), keepdim=True)  # Global average pooling -> [N, C, 1, 1]
        # Custom fused op: (x + bias) -> logsumexp over channel -> sum over H,W (trivial 1x1) -> *10
        y = custom_ops_lib.convtranspose2d_globalavgpool_biasadd_logsumexp_sum_multiply_custom(x, self.bias)
