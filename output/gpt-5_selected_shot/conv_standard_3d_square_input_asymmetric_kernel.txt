kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConv3d {
public:
    __aicore__ inline KernelConv3d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Dout, uint32_t Hout, uint32_t Wout,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->N = N; this->Cin = Cin; this->Cout = Cout;
        this->D = D; this->H = H; this->W = W;
        this->Kd = Kd; this->Kh = Kh; this->Kw = Kw;
        this->Dout = Dout; this->Hout = Hout; this->Wout = Wout;

        this->totalLength = totalLength;
        this->blockLength = (totalLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        this->startIdx = this->blockLength * AscendC::GetBlockIdx();
        if (this->startIdx + this->blockLength > totalLength) {
            this->blockLength = (totalLength > this->startIdx) ? (totalLength - this->startIdx) : 0;
        }
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / ((tileNum > 0) ? tileNum : 1);
        if (this->tileLength == 0) this->tileLength = this->blockLength;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(N * Cin * D * H * W));
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)w, (uint32_t)(Cout * Cin * Kd * Kh * Kw));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint32_t)(N * Cout * Dout * Hout * Wout));
    }

    __aicore__ inline void Process()
    {
        if (blockLength == 0) return;
        uint32_t processed = 0;
        while (processed < blockLength) {
            uint32_t cur = AscendC::min(blockLength - processed, tileLength);
            ComputeRange(startIdx + processed, cur);
            processed += cur;
        }
    }

private:
    __aicore__ inline void ComputeRange(uint32_t globalStart, uint32_t len)
    {
        // Convert linear index over output tensor [N, Cout, Dout, Hout, Wout]
        for (uint32_t idx = 0; idx < len; ++idx) {
            uint32_t g = globalStart + idx;
            if (g >= totalLength) break;

            // unravel index
            uint32_t Wp = Wout;
            uint32_t Hp = Hout * Wp;
            uint32_t Dp = Dout * Hp;
            uint32_t Cp = Cout * Dp;

            uint32_t n = g / Cp; g -= n * Cp;
            uint32_t co = g / Dp; g -= co * Dp;
            uint32_t od = g / Hp; g -= od * Hp;
            uint32_t oh = g / Wp; g -= oh * Wp;
            uint32_t ow = g;

            float acc = 0.0f;
            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t kd = 0; kd < Kd; ++kd) {
                    uint32_t id = od + kd; // stride=1, pad=0
                    for (uint32_t kh = 0; kh < Kh; ++kh) {
                        uint32_t ih = oh + kh;
                        for (uint32_t kw = 0; kw < Kw; ++kw) {
                            uint32_t iw = ow + kw;

                            uint32_t x_off =
                                (((n * Cin + ci) * D + id) * H + ih) * W + iw;

                            uint32_t w_off =
                                ((((co) * Cin + ci) * Kd + kd) * Kh + kh) * Kw + kw;

                            float xv = static_cast<float>(xGm.GetValue(x_off));
                            float wv = static_cast<float>(wGm.GetValue(w_off));
                            acc += xv * wv;
                        }
                    }
                }
            }

            uint32_t y_off =
                (((n * Cout + co) * Dout + od) * Hout + oh) * Wout + ow;
            yGm.SetValue(y_off, static_cast<DTYPE_Z>(acc));
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N, Cin, Cout;
    uint32_t D, H, W;
    uint32_t Kd, Kh, Kw;
    uint32_t Dout, Hout, Wout;

    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t startIdx;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3d op;
    op.Init(x, w, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.Kd, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Dout, tiling_data.Hout, tiling_data.Wout,
            tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch

import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super().__init__()
        assert stride == 1 and padding == 0 and dilation == 1 and groups == 1, "This custom kernel supports stride=1, padding=0, dilation=1, groups=1"
        assert not bias, "This custom kernel currently does not support bias"
        kd, kh, kw = kernel_size[2] if len(kernel_size) == 3 else kernel_size[0], kernel_size[1] if len(kernel_size) == 3 else kernel_size[0], kernel_size[0] if isinstance(kernel_size, int) else kernel_size[-1]
        # Normalize kernel_size to (Kd, Kh, Kw) assuming PyTorch's (D,H,W)
        if isinstance(kernel_size, tuple) and len(kernel_size) == 3:
            kd, kh, kw = kernel_size
        elif isinstance(kernel_size, int):
            kd = kh = kw = kernel_size
        else:
            raise ValueError("kernel_size must be int or tuple of length 3")
        self.weight = torch.nn.Parameter(torch.randn(out_channels, in_channels, kd, kh, kw))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_standard_3d_square_input_asymmetric_kernel_custom(x, self.weight)
