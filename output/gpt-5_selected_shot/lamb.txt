kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLamb {
public:
    __aicore__ inline KernelLamb() : blockLength(0), tileNum(0), tileLength(0), lrVal(0.f), epsVal(0.f) {}
    __aicore__ inline void Init(GM_ADDR param, GM_ADDR m, GM_ADDR v, GM_ADDR lr, GM_ADDR eps, GM_ADDR out, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        paramGm.SetGlobalBuffer((__gm__ float *)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mGm.SetGlobalBuffer((__gm__ float *)m + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ float *)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ float *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // Scalars are 1-element tensors in GM space
        lrGm.SetGlobalBuffer((__gm__ float *)lr, 1);
        epsGm.SetGlobalBuffer((__gm__ float *)eps, 1);

        pipe.InitBuffer(inParamQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inMQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inVQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(scalarQ, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Load scalars lr and eps
        {
            AscendC::LocalTensor<float> sLocal = scalarQ.AllocTensor<float>();
            AscendC::DataCopy(sLocal, lrGm, 1);
            scalarQ.FreeTensor(sLocal);
        }
        {
            AscendC::LocalTensor<float> sLocal2 = scalarQ.AllocTensor<float>();
            AscendC::DataCopy(sLocal2, epsGm, 1);
            scalarQ.FreeTensor(sLocal2);
        }

        // First pass: compute r = m / (sqrt(v) + eps), store r to outGm, and accumulate norms
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        float sumParam2 = 0.f;
        float sumR2 = 0.f;

        for (int32_t i = 0; i < loopCount; ++i) {
            // CopyIn
            AscendC::LocalTensor<float> pLocal = inParamQ.AllocTensor<float>();
            AscendC::LocalTensor<float> mLocal = inMQ.AllocTensor<float>();
            AscendC::LocalTensor<float> vLocal = inVQ.AllocTensor<float>();
            AscendC::DataCopy(pLocal, paramGm[i * this->tileLength], this->tileLength);
            AscendC::DataCopy(mLocal, mGm[i * this->tileLength], this->tileLength);
            AscendC::DataCopy(vLocal, vGm[i * this->tileLength], this->tileLength);
            inParamQ.EnQue(pLocal);
            inMQ.EnQue(mLocal);
            inVQ.EnQue(vLocal);

            // Compute r and partial sums
            pLocal = inParamQ.DeQue<float>();
            mLocal = inMQ.DeQue<float>();
            vLocal = inVQ.DeQue<float>();
            AscendC::LocalTensor<float> rLocal = outQ.AllocTensor<float>();
            // vLocal = sqrt(vLocal)
            AscendC::Sqrt(vLocal, vLocal, this->tileLength);
            // vLocal = vLocal + eps
            AscendC::Adds(vLocal, vLocal, epsVal, this->tileLength);
            // rLocal = mLocal / vLocal
            AscendC::Div(rLocal, mLocal, vLocal, this->tileLength);

            // tmpLocal = pLocal * pLocal, sum them
            AscendC::LocalTensor<float> tmpLocal = tmpQ.AllocTensor<float>();
            AscendC::Mul(tmpLocal, pLocal, pLocal, this->tileLength);
            float partialP2 = AscendC::ReduceSum(tmpLocal, this->tileLength);
            tmpQ.FreeTensor(tmpLocal);

            // tmpLocal2 = rLocal * rLocal, sum them
            AscendC::LocalTensor<float> tmpLocal2 = tmpQ.AllocTensor<float>();
            AscendC::Mul(tmpLocal2, rLocal, rLocal, this->tileLength);
            float partialR2 = AscendC::ReduceSum(tmpLocal2, this->tileLength);
            tmpQ.FreeTensor(tmpLocal2);

            sumParam2 += partialP2;
            sumR2 += partialR2;

            outQ.EnQue(rLocal);
            inParamQ.FreeTensor(pLocal);
            inMQ.FreeTensor(mLocal);
            inVQ.FreeTensor(vLocal);

            // CopyOut r to GM (temporary)
            rLocal = outQ.DeQue<float>();
            AscendC::DataCopy(outGm[i * this->tileLength], rLocal, this->tileLength);
            outQ.FreeTensor(rLocal);
        }

        float normParam = AscendC::SqrtScalar(sumParam2);
        float normR = AscendC::SqrtScalar(sumR2);
        float trustRatio = normParam / (normR + epsVal);
        float alpha = lrVal * trustRatio;

        // Second pass: param_new = param - alpha * r (r stored in outGm)
        for (int32_t i = 0; i < loopCount; ++i) {
            AscendC::LocalTensor<float> pLocal = inParamQ.AllocTensor<float>();
            AscendC::LocalTensor<float> rLocal = inMQ.AllocTensor<float>();
            AscendC::DataCopy(pLocal, paramGm[i * this->tileLength], this->tileLength);
            AscendC::DataCopy(rLocal, outGm[i * this->tileLength], this->tileLength);
            inParamQ.EnQue(pLocal);
            inMQ.EnQue(rLocal);

            pLocal = inParamQ.DeQue<float>();
            rLocal = inMQ.DeQue<float>();
            AscendC::LocalTensor<float> outLocal = outQ.AllocTensor<float>();

            // outLocal = pLocal - alpha * rLocal
            AscendC::Muls(outLocal, rLocal, alpha, this->tileLength);
            AscendC::Sub(outLocal, pLocal, outLocal, this->tileLength);

            outQ.EnQue(outLocal);
            inParamQ.FreeTensor(pLocal);
            inMQ.FreeTensor(rLocal);

            outLocal = outQ.DeQue<float>();
            AscendC::DataCopy(outGm[i * this->tileLength], outLocal, this->tileLength);
            outQ.FreeTensor(outLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inParamQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inMQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inVQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQ;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scalarQ;

    AscendC::GlobalTensor<float> paramGm;
    AscendC::GlobalTensor<float> mGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> lrGm;
    AscendC::GlobalTensor<float> epsGm;
    AscendC::GlobalTensor<float> outGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    float lrVal;
    float epsVal;
};

extern "C" __global__ __aicore__ void lamb_custom(GM_ADDR param, GM_ADDR m, GM_ADDR v, GM_ADDR lr, GM_ADDR eps, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLamb op;
    op.Init(param, m, v, lr, eps, out, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
