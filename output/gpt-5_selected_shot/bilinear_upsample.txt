kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM_IN = 2;
constexpr int32_t BUFFER_NUM_OUT = 1;

__aicore__ inline int32_t clamp_int(int32_t v, int32_t lo, int32_t hi) {
    return v < lo ? lo : (v > hi ? hi : v);
}

class KernelBilinearUpsample {
public:
    __aicore__ inline KernelBilinearUpsample() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, uint32_t N, uint32_t C, uint32_t H, uint32_t W, uint32_t tileNum)
    {
        this->N = N;
        this->C = C;
        this->H = H;
        this->W = W;
        this->OH = H * 2;
        this->OW = W * 2;

        this->rowsIn = static_cast<uint64_t>(N) * C * H;
        this->rowsOut = static_cast<uint64_t>(N) * C * this->OH;

        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint64_t>(N) * C * H * W);
        tmpGm.SetGlobalBuffer((__gm__ DTYPE_X *)workspace, static_cast<uint64_t>(N) * C * H * this->OW);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, static_cast<uint64_t>(N) * C * this->OH * this->OW);

        // Allocate UB buffers large enough to hold one expanded row (2W)
        pipe.InitBuffer(inQueue, BUFFER_NUM_IN, this->OW * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM_OUT, this->OW * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        HorizontalPass();
        VerticalPass();
    }

private:
    __aicore__ inline void HorizontalPass()
    {
        // Distribute input rows across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t bidx = AscendC::GetBlockIdx();

        uint64_t rowsPerBlock = (rowsIn + blockNum - 1) / blockNum;
        uint64_t rowStart = static_cast<uint64_t>(bidx) * rowsPerBlock;
        uint64_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > rowsIn) rowEnd = rowsIn;

        for (uint64_t row = rowStart; row < rowEnd; ++row) {
            // Load input row (length W)
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[row * W], W);

            // Temporary local for expanded horizontal row (length 2W)
            AscendC::LocalTensor<DTYPE_X> tmpLocal = inQueue.AllocTensor<DTYPE_X>();

            // Compute horizontal linear interpolation with align_corners=False and scale=2
            for (uint32_t ow = 0; ow < OW; ++ow) {
                float in_w = (static_cast<float>(ow) - 0.5f) * 0.5f; // (ow - 0.5)/2
                int32_t w0 = static_cast<int32_t>(AscendC::Floor(in_w));
                float w1_alpha = in_w - static_cast<float>(w0);
                float w0_alpha = 1.0f - w1_alpha;
                int32_t iw0 = clamp_int(w0, 0, static_cast<int32_t>(W) - 1);
                int32_t iw1 = clamp_int(w0 + 1, 0, static_cast<int32_t>(W) - 1);
                // Read two pixels and blend
                float v0 = static_cast<float>(xLocal[iw0]);
                float v1 = static_cast<float>(xLocal[iw1]);
                tmpLocal[ow] = static_cast<DTYPE_X>(v0 * w0_alpha + v1 * w1_alpha);
            }

            // Store expanded row to workspace
            AscendC::DataCopy(tmpGm[row * OW], tmpLocal, OW);

            inQueue.FreeTensor(xLocal);
            inQueue.FreeTensor(tmpLocal);
        }
    }

    __aicore__ inline void VerticalPass()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t bidx = AscendC::GetBlockIdx();

        uint64_t rowsPerBlock = (rowsOut + blockNum - 1) / blockNum;
        uint64_t rowStart = static_cast<uint64_t>(bidx) * rowsPerBlock;
        uint64_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > rowsOut) rowEnd = rowsOut;

        for (uint64_t orow = rowStart; orow < rowEnd; ++orow) {
            // Compute corresponding input row indices for vertical interpolation
            // Decompose orow into (n,c,oh)
            uint64_t rowsPerNC_out = static_cast<uint64_t>(OH);
            uint64_t nc = orow / rowsPerNC_out;
            uint32_t oh = static_cast<uint32_t>(orow % rowsPerNC_out);

            float in_h = (static_cast<float>(oh) - 0.5f) * 0.5f; // (oh - 0.5)/2
            int32_t h0 = static_cast<int32_t>(AscendC::Floor(in_h));
            float h1_alpha = in_h - static_cast<float>(h0);
            float h0_alpha = 1.0f - h1_alpha;
            int32_t ih0 = clamp_int(h0, 0, static_cast<int32_t>(H) - 1);
            int32_t ih1 = clamp_int(h0 + 1, 0, static_cast<int32_t>(H) - 1);

            uint64_t baseInRow0 = nc * static_cast<uint64_t>(H) + static_cast<uint64_t>(ih0);
            uint64_t baseInRow1 = nc * static_cast<uint64_t>(H) + static_cast<uint64_t>(ih1);

            // Load two horizontally-expanded rows from workspace
            AscendC::LocalTensor<DTYPE_X> tmp0 = inQueue.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> tmp1 = inQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(tmp0, tmpGm[baseInRow0 * OW], OW);
            AscendC::DataCopy(tmp1, tmpGm[baseInRow1 * OW], OW);

            // Output local row
            AscendC::LocalTensor<DTYPE_Z> outLocal = outQueue.AllocTensor<DTYPE_Z>();

            for (uint32_t ow = 0; ow < OW; ++ow) {
                float v0 = static_cast<float>(tmp0[ow]);
                float v1 = static_cast<float>(tmp1[ow]);
                outLocal[ow] = static_cast<DTYPE_Z>(v0 * h0_alpha + v1 * h1_alpha);
            }

            // Store final row
            AscendC::DataCopy(zGm[orow * OW], outLocal, OW);

            inQueue.FreeTensor(tmp0);
            inQueue.FreeTensor(tmp1);
            outQueue.FreeTensor(outLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM_IN> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM_OUT> outQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> tmpGm; // workspace
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t N, C, H, W, OH, OW;
    uint32_t tileNum;
    uint64_t rowsIn;
    uint64_t rowsOut;
};

extern "C" __global__ __aicore__ void bilinear_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    // Note: interface keeps (x, [unused], z). 'y' is not used for this 1-input op.
    GET_TILING_DATA(tiling_data, tiling);
    KernelBilinearUpsample op;
    op.Init(x, z, workspace, tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W, tiling_data.tileNum);
    op.Process();
}
