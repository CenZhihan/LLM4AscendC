const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)

    PowerBroadcastCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    auto origin = x_shape->GetOriginShape();
    uint32_t totalLength = origin.GetShapeSize();
    uint32_t dimNum = origin.GetDimNum();
    uint32_t colSize = dimNum > 0 ? static_cast<uint32_t>(origin.GetDim(dimNum - 1)) : 1;

    tiling.set_totalLength(totalLength);
    tiling.set_colSize(colSize);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    return GRAPH_SUCCESS;

    const auto inputDataType = context->GetInputDataType(0);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelPowBroadcast {
public:
    __aicore__ inline KernelPowBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR exponent, GM_ADDR z, uint32_t totalLength, uint32_t colSize, uint32_t /*tileNum*/)
    {
        this->colSize = colSize > 0 ? colSize : 1;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->rowsPerBlock = this->blockLength / this->colSize;
        this->tileLength = this->colSize; // process one full row per tile

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        eGm.SetGlobalBuffer((__gm__ DTYPE_Y *)exponent, this->colSize);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // init local buffers: x and z use row length; exponent uses colSize
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueE, BUFFER_NUM, this->colSize * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> eLocal = inQueueE.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->colSize], this->tileLength);
        // load the exponent vector once per row
        AscendC::DataCopy(eLocal, eGm, this->colSize);
        inQueueX.EnQue(xLocal);
        inQueueE.EnQue(eLocal);
    }
    __aicore__ inline void Compute(uint32_t /*rowIdx*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> eLocal = inQueueE.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::Pow(zLocal, xLocal, eLocal, this->tileLength);
        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueE.FreeTensor(eLocal);
    }
    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[rowIdx * this->colSize], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueE;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> eGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    uint32_t blockLength;
    uint32_t colSize;
    uint32_t rowsPerBlock;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void power_broadcast_custom(GM_ADDR x, GM_ADDR exponent, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelPowBroadcast op;
    op.Init(x, exponent, z, tiling_data.totalLength, tiling_data.colSize, tiling_data.tileNum);
    op.Process();
}
