inline uint32_t GetDimSafe(const gert::Shape* s, int idx) {

    return static_cast<uint32_t>(os.GetDim(idx));

kernel_src="""
#include "kernel_operator.h"

// Naive transposed 2D convolution kernel for stride=1, padding=0, dilation=1, groups=1, no bias.
// x: [N, Cin, Hin, Win]
// w: [Cin, Cout, Kh, Kw]
// y: [N, Cout, Hout, Wout] with Hout=Hin+Kh-1, Wout=Win+Kw-1

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    // Read tiling data (dimensions)
    const uint32_t N    = tiling_data.n;
    const uint32_t Cin  = tiling_data.cin;
    const uint32_t Cout = tiling_data.cout;
    const uint32_t Hin  = tiling_data.hin;
    const uint32_t Win  = tiling_data.win;
    const uint32_t Kh   = tiling_data.kh;
    const uint32_t Kw   = tiling_data.kw;
    const uint32_t Hout = tiling_data.hout;
    const uint32_t Wout = tiling_data.wout;

    const uint64_t total = static_cast<uint64_t>(N) * Cout * Hout * Wout;
    const uint32_t blocks = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();

    uint64_t start = (total / blocks) * blockIdx;
    uint64_t end   = (blockIdx == blocks - 1) ? total : (total / blocks) * (blockIdx + 1);

    __gm__ const float* xPtr = reinterpret_cast<__gm__ const float*>(x);
    __gm__ const float* wPtr = reinterpret_cast<__gm__ const float*>(w);
    __gm__ float* yPtr = reinterpret_cast<__gm__ float*>(y);

    // Strides
    const uint64_t xStrideN = static_cast<uint64_t>(Cin) * Hin * Win;
    const uint64_t xStrideC = static_cast<uint64_t>(Hin) * Win;
    const uint64_t xStrideH = static_cast<uint64_t>(Win);
    const uint64_t xStrideW = 1ULL;

    const uint64_t wStrideCin  = static_cast<uint64_t>(Cout) * Kh * Kw;
    const uint64_t wStrideCout = static_cast<uint64_t>(Kh) * Kw;
    const uint64_t wStrideKh   = static_cast<uint64_t>(Kw);
    const uint64_t wStrideKw   = 1ULL;

    const uint64_t yStrideN = static_cast<uint64_t>(Cout) * Hout * Wout;
    const uint64_t yStrideC = static_cast<uint64_t>(Hout) * Wout;
    const uint64_t yStrideH = static_cast<uint64_t>(Wout);
    const uint64_t yStrideW = 1ULL;

    for (uint64_t idx = start; idx < end; ++idx) {
        // Map linear index to (n, cout, h, w)
        uint64_t rem = idx;
        uint32_t n = static_cast<uint32_t>(rem / yStrideN); rem %= yStrideN;
        uint32_t co = static_cast<uint32_t>(rem / yStrideC); rem %= yStrideC;
        uint32_t ho = static_cast<uint32_t>(rem / yStrideH); rem %= yStrideH;
        uint32_t wo = static_cast<uint32_t>(rem / yStrideW);

        float acc = 0.f;
        // Deconvolution accumulation
        for (uint32_t ci = 0; ci < Cin; ++ci) {
            for (uint32_t kh = 0; kh < Kh; ++kh) {
                int32_t hi = static_cast<int32_t>(ho) - static_cast<int32_t>(kh);
                if (hi < 0 || hi >= static_cast<int32_t>(Hin)) continue;
                for (uint32_t kw = 0; kw < Kw; ++kw) {
                    int32_t wi = static_cast<int32_t>(wo) - static_cast<int32_t>(kw);
                    if (wi < 0 || wi >= static_cast<int32_t>(Win)) continue;
                    uint64_t xOff = static_cast<uint64_t>(n) * xStrideN
                                  + static_cast<uint64_t>(ci) * xStrideC
                                  + static_cast<uint64_t>(hi) * xStrideH
                                  + static_cast<uint64_t>(wi) * xStrideW;
                    uint64_t wOff = static_cast<uint64_t>(ci) * wStrideCin
                                  + static_cast<uint64_t>(co) * wStrideCout
                                  + static_cast<uint64_t>(kh) * wStrideKh
                                  + static_cast<uint64_t>(kw) * wStrideKw;
                    acc += xPtr[xOff] * wPtr[wOff];
                }
            }
        }
        uint64_t yOff = static_cast<uint64_t>(n) * yStrideN
                      + static_cast<uint64_t>(co) * yStrideC
                      + static_cast<uint64_t>(ho) * yStrideH
                      + static_cast<uint64_t>(wo) * yStrideW;
        yPtr[yOff] = acc;
    }
}
