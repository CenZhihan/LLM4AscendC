kernel_src="""
#include "kernel_operator.h"
#include <cmath>

// Simple fused kernel: Y = GELU(hardtanh((X * W^T + b) * scale))
// Shapes:
//   X: [M, K]
//   W: [N, K]  (row-major, corresponds to PyTorch nn.Linear weight [out_features, in_features])
//   b: [N]
//   scale: [1]
//   ht_min: [1]
//   ht_max: [1]
//   Y: [M, N]

extern \"C\" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scale, GM_ADDR ht_min, GM_ADDR ht_max,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t M = tiling_data.M;
    const uint32_t N = tiling_data.N;
    const uint32_t K = tiling_data.K;

    // Partition rows by block
    const uint32_t blk_num = AscendC::GetBlockNum();
    const uint32_t blk_idx = AscendC::GetBlockIdx();
    const uint32_t rows_per_blk = (M + blk_num - 1) / blk_num;
    const uint32_t row_start = blk_idx * rows_per_blk;
    const uint32_t row_end = (row_start + rows_per_blk) < M ? (row_start + rows_per_blk) : M;

    // GM pointers
    __gm__ float* X = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* W = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* B = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* Y = reinterpret_cast<__gm__ float*>(y);

    const float s = reinterpret_cast<__gm__ float*>(scale)[0];
    const float htmin = reinterpret_cast<__gm__ float*>(ht_min)[0];
    const float htmax = reinterpret_cast<__gm__ float*>(ht_max)[0];

    auto gelu_approx = [](float v) __aicore__ -> float {
        const float k0 = 0.7978845608028654f; // sqrt(2/pi)
        const float k1 = 0.044715f;
        float t = k0 * (v + k1 * v * v * v);
        float th = tanh(t);
        return 0.5f * v * (1.0f + th);
    };

    for (uint32_t m = row_start; m < row_end; ++m) {
        const uint32_t x_row_off = m * K;
        const uint32_t y_row_off = m * N;

        for (uint32_t n = 0; n < N; ++n) {
            // Dot(X[m, :], W[n, :])
            const uint32_t w_row_off = n * K;
            float acc = 0.0f;
            // Unrolled accumulation (simple)
            for (uint32_t k = 0; k < K; ++k) {
                acc += X[x_row_off + k] * W[w_row_off + k];
            }

            // + bias
            acc += B[n];
            // scale
            acc *= s;
            // hardtanh clamp
            acc = acc < htmin ? htmin : acc;
            acc = acc > htmax ? htmax : acc;
            // gelu
            acc = gelu_approx(acc);

            Y[y_row_off + n] = acc;
        }
    }
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, scaling_factor, hardtanh_min, hardtanh_max):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.scaling_factor = float(scaling_factor)
        self.hardtanh_min = float(hardtanh_min)
        self.hardtanh_max = float(hardtanh_max)

    def forward(self, x):
        # Prepare scalar parameters as 1-element tensors on the same device/dtype as x
        dev = x.device
        dtype = x.dtype
        scale_t = torch.tensor([self.scaling_factor], device=dev, dtype=dtype)
        ht_min_t = torch.tensor([self.hardtanh_min], device=dev, dtype=dtype)
        ht_max_t = torch.tensor([self.hardtanh_max], device=dev, dtype=dtype)
        bias = self.gemm.bias
        if bias is None:
            bias = torch.zeros(self.gemm.weight.size(0), device=dev, dtype=dtype)
        # Fused call: Y = GELU(hardtanh((X @ W^T + b) * scale))
        return custom_ops_lib.gemm_scaling_hard_tanh_gelu_custom(
            x, self.gemm.weight, bias, scale_t, ht_min_t, ht_max_t
        )
