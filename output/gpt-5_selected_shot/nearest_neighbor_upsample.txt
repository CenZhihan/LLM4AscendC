kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // number of local buffers for pipeline

class KernelNearestNeighborUpsample {
public:
    __aicore__ inline KernelNearestNeighborUpsample() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t scale, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->scale = scale;
        this->hout = h * scale;
        this->wout = w * scale;

        uint64_t totalOutLength = (uint64_t)n * c * this->hout * this->wout;
        uint64_t totalInLength  = (uint64_t)n * c * h * w;

        this->blockOutLength = totalOutLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockOutLength / tileNum / BUFFER_NUM;

        // Input is read-only; set base to the beginning of the tensor
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, totalInLength);
        // Output is partitioned by block
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockOutLength * AscendC::GetBlockIdx(), this->blockOutLength);

        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            ComputeAndStore(i);
        }
    }

private:
    __aicore__ inline void ComputeAndStore(int32_t progress)
    {
        uint64_t tileBase = (uint64_t)progress * this->tileLength;
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        uint64_t blockBase = (uint64_t)this->blockOutLength * AscendC::GetBlockIdx();

        for (uint64_t j = 0; j < this->tileLength; ++j) {
            uint64_t outIdxGlobal = blockBase + tileBase + j;

            uint64_t wOutIdx = outIdxGlobal % this->wout;
            uint64_t tmp0 = outIdxGlobal / this->wout;
            uint64_t hOutIdx = tmp0 % this->hout;
            uint64_t tmp1 = tmp0 / this->hout;
            uint64_t cIdx = tmp1 % this->c;
            uint64_t nIdx = tmp1 / this->c;

            uint64_t hInIdx = hOutIdx / this->scale;
            uint64_t wInIdx = wOutIdx / this->scale;

            uint64_t inIdx = (((nIdx * (uint64_t)this->c + cIdx) * (uint64_t)this->h + hInIdx) * (uint64_t)this->w + wInIdx);

            AscendC::DataCopy(zLocal[j], xGm[inIdx], 1);
        }

        outQueueZ.EnQue<DTYPE_Z>(zLocal);

        AscendC::LocalTensor<DTYPE_Z> zLocalOut = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[tileBase], zLocalOut, this->tileLength);
        outQueueZ.FreeTensor(zLocalOut);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t h;
    uint32_t w;
    uint32_t scale;
    uint32_t hout;
    uint32_t wout;

    uint64_t blockOutLength;
    uint32_t tileNum;
    uint64_t tileLength;
};

extern "C" __global__ __aicore__ void nearest_neighbor_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelNearestNeighborUpsample op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.scale, tiling_data.tileNum);
    op.Process();
}
"""
