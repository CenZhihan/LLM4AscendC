    uint32_t groups = 3;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelShuffle {
public:
    __aicore__ inline KernelShuffle() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t groups)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->groups = groups;
        this->inner = h * w;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(n * c * h * w));
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, (uint32_t)(n * c * h * w));

        // Tile one channel (H*W) per buffer slot
        tileLength = this->inner;
        pipe.InitBuffer(inQueueX, BUFFER_NUM, tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Distribute channels across blocks
        uint32_t cStart = (uint64_t)c * blockIdx / blockNum;
        uint32_t cEnd   = (uint64_t)c * (blockIdx + 1) / blockNum;

        uint32_t K = c / groups;

        for (uint32_t ni = 0; ni < n; ++ni) {
            for (uint32_t ci = cStart; ci < cEnd; ++ci) {
                uint32_t g = ci / K;
                uint32_t k = ci % K;
                uint32_t cNew = k * groups + g;

                uint64_t srcOff = ((uint64_t)ni * c + ci) * inner;
                uint64_t dstOff = ((uint64_t)ni * c + cNew) * inner;

                // CopyIn
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[srcOff], tileLength);
                inQueueX.EnQue(xLocal);

                // Compute (pass-through)
                AscendC::LocalTensor<DTYPE_X> xLocal2 = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
                AscendC::DataCopy(zLocal, xLocal2, tileLength);
                outQueueZ.EnQue(zLocal);
                inQueueX.FreeTensor(xLocal2);

                // CopyOut
                AscendC::LocalTensor<DTYPE_Z> zLocal2 = outQueueZ.DeQue<DTYPE_Z>();
                AscendC::DataCopy(zGm[dstOff], zLocal2, tileLength);
                outQueueZ.FreeTensor(zLocal2);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t n;
    uint32_t c;
    uint32_t h;
    uint32_t w;
    uint32_t groups;
    uint32_t inner;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void shufflenet_unit_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelShuffle op;
    op.Init(x, z, tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w, tiling_data.groups);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ChannelShuffle(torch.nn.Module):
    def __init__(self, groups):
        super(ChannelShuffle, self).__init__()
        self.groups = groups
    def forward(self, x):
        return custom_ops_lib.shufflenet_unit_custom(x)

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, groups=3):
        super(ModelNew, self).__init__()
        assert out_channels % 4 == 0
        mid_channels = out_channels // 4

        self.conv1 = torch.nn.Conv2d(in_channels, mid_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn1 = torch.nn.BatchNorm2d(mid_channels)

        self.conv2 = torch.nn.Conv2d(mid_channels, mid_channels, kernel_size=3, stride=1, padding=1, groups=mid_channels, bias=False)
        self.bn2 = torch.nn.BatchNorm2d(mid_channels)

        self.conv3 = torch.nn.Conv2d(mid_channels, out_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn3 = torch.nn.BatchNorm2d(out_channels)

        self.shuffle = ChannelShuffle(groups)

        if in_channels == out_channels:
            self.shortcut = torch.nn.Sequential()
        else:
            self.shortcut = torch.nn.Sequential(
                torch.nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=False),
                torch.nn.BatchNorm2d(out_channels)
            )

    def forward(self, x):
        out = torch.nn.functional.relu(self.bn1(self.conv1(x)))
        out = self.bn2(self.conv2(out))
        out = self.shuffle(out)
        out = torch.nn.functional.relu(self.bn3(self.conv3(out)))
        out += self.shortcut(x)
        return out
'''
