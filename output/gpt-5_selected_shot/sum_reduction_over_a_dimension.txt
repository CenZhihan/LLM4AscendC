host_operator_src="""
#include "sum_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
constexpr uint32_t BLOCK_DIM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TilingData tiling;
    auto *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();
    uint32_t dims = origin.GetDimNum();
    uint64_t total = 1;
    for (uint32_t i = 0; i < dims; ++i) {
        total *= static_cast<uint64_t>(origin.GetDim(i));
    }
    // Reduce last dimension only for kernel (model will permute if needed)
    uint32_t lastDimSize = dims > 0 ? static_cast<uint32_t>(origin.GetDim(dims - 1)) : 1;
    uint32_t outerSize = lastDimSize == 0 ? 0 : static_cast<uint32_t>(total / lastDimSize);
    uint32_t outLength = outerSize; // keepdim=True -> output is (...,1), flattened size = outerSize

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_outerSize(outerSize);
    tiling.set_reduceSize(lastDimSize);
    tiling.set_totalLength(static_cast<uint32_t>(total));
    tiling.set_outLength(outLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &x_origin = x_shape->GetOriginShape();
    uint32_t dims = x_origin.GetDimNum();
    if (dims == 0) {
        // scalar stays scalar with keepdim=True -> shape [1]
        gert::Shape *y_shape = context->GetOutputShape(0);
        *y_shape = {1};
        return GRAPH_SUCCESS;
    }
    std::vector<int64_t> out_dims;
    out_dims.reserve(dims);
    for (uint32_t i = 0; i < dims - 1; ++i) {
        out_dims.push_back(x_origin.GetDim(i));
    }
    out_dims.push_back(1); // keepdim=True for last dimension
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = out_dims;
    return GRAPH_SUCCESS;
}

    const auto inputDataType = context->GetInputDataType(0);

kernel_src="""
#include "kernel_operator.h"

class KernelReduceLastDim {
public:
    __aicore__ inline KernelReduceLastDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t totalLength, uint32_t outerSize, uint32_t reduceSize, uint32_t outLength)
    {
        this->totalLength = totalLength;
        this->outerSize = outerSize;
        this->reduceSize = reduceSize;
        this->outLength = outLength;

        xGm.SetGlobalBuffer((__gm__ float *)x, totalLength);
        zGm.SetGlobalBuffer((__gm__ float *)z, outLength);
        // Buffers
        pipe.InitBuffer(inQueueX, 1, totalLength * sizeof(float));
        pipe.InitBuffer(outQueueZ, 1, outLength * sizeof(float));
        // local scratch for one row accumulation if needed
        pipe.InitBuffer(calcBuf, reduceSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Copy entire input
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm, totalLength);
        inQueueX.EnQue(xLocal);

        // Prepare output local
        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

        // Compute: reduce over last dimension (contiguous). x is shaped [outerSize, reduceSize].
        AscendC::LocalTensor<float> xBuf = inQueueX.DeQue<float>();
        // Simple loop accumulation per row
        for (uint32_t o = 0; o < outerSize; ++o) {
            // Accumulate sum over reduceSize elements for this row
            float acc = 0.0f;
            uint32_t base = o * reduceSize;
            // Accumulate in scalar
            for (uint32_t r = 0; r < reduceSize; ++r) {
                // Pseudo element access; conceptual for illustration
                acc += xBuf.GetValue(base + r);
            }
            zLocal.SetValue(o, acc);
        }

        // Enqueue and free
        outQueueZ.EnQue<float>(zLocal);
        inQueueX.FreeTensor(xBuf);

        // Copy out
        AscendC::LocalTensor<float> zLocalOut = outQueueZ.DeQue<float>();
        AscendC::DataCopy(zGm, zLocalOut, this->outLength);
        outQueueZ.FreeTensor(zLocalOut);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueZ;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBuf;
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;
    uint32_t totalLength;
    uint32_t outerSize;
    uint32_t reduceSize;
    uint32_t outLength;
};

extern "C" __global__ __aicore__ void sum_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelReduceLastDim op;
    op.Init(x, z,
            tiling_data.totalLength,
            tiling_data.outerSize,
            tiling_data.reduceSize,
            tiling_data.outLength);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that reduces over an arbitrary dimension by:
    - Permuting the target dim to the last dim
    - Calling a custom AscendC kernel that reduces over the last dim with keepdim=True
    - Permuting back to original layout
    """
    def __init__(self, dim: int):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        dim = self.dim
        rank = x.dim()
        if dim < 0:
            dim += rank
        assert 0 <= dim < rank, "Invalid reduction dimension"

        if rank == 0:
            # Scalar: just call kernel (treat as last-dim)
            return custom_ops_lib.sum_reduction_over_a_dimension_custom(x)

        # Permute target dim to the last dim
        perm = [d for d in range(rank) if d != dim] + [dim]
        x_perm = x.permute(perm).contiguous()

        # Apply custom kernel: reduces last dim with keepdim=True
        y_perm = custom_ops_lib.sum_reduction_over_a_dimension_custom(x_perm)

        # Inverse permutation to restore original order
        inv_perm = [0] * rank
        for i, p in enumerate(perm):
            inv_perm[p] = i
        y = y_perm.permute(inv_perm).contiguous()
        return y
'''
