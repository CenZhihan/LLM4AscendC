kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelGNSwishMulSwish {
public:
    __aicore__ inline KernelGNSwishMulSwish() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR mw, GM_ADDR groups,
                                GM_ADDR y, uint32_t n, uint32_t c, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->tileNum = tileNum;

        xGm = (__gm__ float*)x;
        gammaGm = (__gm__ float*)gamma;
        betaGm = (__gm__ float*)beta;
        mwGm = (__gm__ float*)mw;
        groupsGm = (__gm__ int32_t*)groups;
        yGm = (__gm__ float*)y;

        // Partition by batch across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        rowsPerBlock = (n + blocks - 1) / blocks; // ceil-div
        startRow = AscendC::GetBlockIdx() * rowsPerBlock;
        if (startRow + rowsPerBlock > n) {
            rowsPerBlock = (n > startRow) ? (n - startRow) : 0;
        }
    }

    __aicore__ inline void Process()
    {
        if (rowsPerBlock == 0) {
            return;
        }
        int32_t groups = 1;
        // Read groups value from GM scalar
        groups = groupsGm[0] <= 0 ? 1 : groupsGm[0];
        uint32_t groupSize = c / (uint32_t)groups;
        if (groupSize == 0) {
            groupSize = c; // fallback no grouping
            groups = 1;
        }

        for (uint32_t r = 0; r < rowsPerBlock; ++r) {
            uint32_t globalRow = startRow + r;
            if (globalRow >= n) break;

            // For each group, compute mean/var then normalize and apply fused activations
            for (int32_t g = 0; g < groups; ++g) {
                uint32_t start = (uint32_t)g * groupSize;
                uint32_t end = (g == groups - 1) ? c : (start + groupSize);
                float sum = 0.f;
                float sqsum = 0.f;

                // compute mean/var
                for (uint32_t idx = start; idx < end; ++idx) {
                    float v = xGm[globalRow * c + idx];
                    sum += v;
                    sqsum += v * v;
                }
                uint32_t len = end - start;
                float mean = sum / (float)len;
                float var = sqsum / (float)len - mean * mean;
                float invStd = 1.0f / sqrtf(var + 1e-5f);

                // normalize + affine + swish + multiply + swish
                for (uint32_t idx = start; idx < end; ++idx) {
                    float v = xGm[globalRow * c + idx];
                    float gVal = gammaGm[idx];
                    float bVal = betaGm[idx];
                    float mVal = mwGm[idx];

                    float norm = (v - mean) * invStd;
                    float affine = norm * gVal + bVal;

                    float sig1 = 1.0f / (1.0f + expf(-affine));
                    float sw1 = affine * sig1;

                    float mul = sw1 * mVal;

                    float sig2 = 1.0f / (1.0f + expf(-mul));
                    float sw2 = mul * sig2;

                    yGm[globalRow * c + idx] = sw2;
                }
            }
        }
    }

private:
    __gm__ float* xGm;
    __gm__ float* gammaGm;
    __gm__ float* betaGm;
    __gm__ float* mwGm;
    __gm__ int32_t* groupsGm;
    __gm__ float* yGm;

    uint32_t n;
    uint32_t c;
    uint32_t tileNum;

    uint32_t rowsPerBlock;
    uint32_t startRow;
};

extern "C" __global__ __aicore__ void gemm_group_norm_swish_multiply_swish_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR mw, GM_ADDR groups, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGNSwishMulSwish op;
    op.Init(x, gamma, beta, mw, groups, y,
            tiling_data.n, tiling_data.c, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using custom AscendC kernel for GroupNorm + Swish + Multiply + Swish
    after GEMM. GEMM (Linear) is kept as nn.Linear for weight management and matmul,
    while the normalization and activations are fused in a single custom op.
    """
    def __init__(self, in_features, out_features, num_groups, multiply_weight_shape):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features)
        self.multiply_weight = torch.nn.Parameter(torch.randn(multiply_weight_shape))

    def forward(self, x):
        # GEMM
        x = self.gemm(x)
        # Fused GroupNorm + Swish + Multiply + Swish
        groups_tensor = torch.tensor([self.group_norm.num_groups], dtype=torch.int32, device=x.device)
        x = custom_ops_lib.gemm_group_norm_swish_multiply_swish_custom(
            x, self.group_norm.weight, self.group_norm.bias, self.multiply_weight, groups_tensor
        )
        return x
