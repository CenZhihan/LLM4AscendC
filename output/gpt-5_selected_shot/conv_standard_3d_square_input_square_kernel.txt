inline int64_t GetDimSafe(const gert::Shape* s, size_t i) {

    // Input 0: x, shape [N, C_in, D, H, W]
    const gert::Shape* x_shape = context->GetInputShape(0);
    // Input 1: w, shape [C_out, C_in, K, K, K]
    const gert::Shape* w_shape = context->GetInputShape(1);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // number of buffers for output queue

// A very simple tiling: split the flat output work evenly by block
class KernelConv3D {
public:
    __aicore__ inline KernelConv3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t d, uint32_t h, uint32_t wi,
                                uint32_t k, uint32_t out_d, uint32_t out_h, uint32_t out_w,
                                uint64_t totalWork, uint32_t tileNum) {
        this->N = n;
        this->Cin = cin;
        this->Cout = cout;
        this->D = d;
        this->H = h;
        this->W = wi;
        this->K = k;
        this->OutD = out_d;
        this->OutH = out_h;
        this->OutW = out_w;
        this->tileNum = tileNum;

        // Work partition
        uint32_t blockNum = AscendC::GetBlockNum();
        uint64_t base = (totalWork / blockNum) * AscendC::GetBlockIdx();
        uint64_t len  = (totalWork / blockNum);

        // Set GM buffers; x and w are shared across blocks; y is offset by base
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)N * Cin * D * H * W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)w, (uint64_t)Cout * Cin * K * K * K);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + base, len);

        // Tile length
        this->blockWork = len;
        this->tileLen = (uint32_t)((len / tileNum) > 0 ? (len / tileNum) : len);

        // Queues: only output is queued; intermediate buffers are computed on-the-fly
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process() {
        // Simple loop over tiles
        uint32_t loops = (this->blockWork + this->tileLen - 1) / this->tileLen;
        for (uint32_t t = 0; t < loops; ++t) {
            uint32_t curLen = (t == loops - 1) ? (uint32_t)(this->blockWork - (uint64_t)t * this->tileLen) : this->tileLen;
            ComputeTile(t, curLen);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t tileIdx, uint32_t curLen) {
        // Allocate local output buffer
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        // Initialize local buffer to zero
        for (uint32_t i = 0; i < curLen; ++i) {
            yLocal.SetValue(i, (DTYPE_Z)0.f);
        }

        // For each element in the tile, compute the convolution result
        uint64_t globalStart = (uint64_t)tileIdx * this->tileLen;
        for (uint32_t i = 0; i < curLen; ++i) {
            uint64_t outIndexLinear = globalStart + i;

            // map linear index to (n, co, od, oh, ow)
            uint32_t ow = (uint32_t)(outIndexLinear % OutW);
            uint64_t tmp1 = outIndexLinear / OutW;
            uint32_t oh = (uint32_t)(tmp1 % OutH);
            uint64_t tmp2 = tmp1 / OutH;
            uint32_t od = (uint32_t)(tmp2 % OutD);
            uint64_t tmp3 = tmp2 / OutD;
            uint32_t co = (uint32_t)(tmp3 % Cout);
            uint32_t n  = (uint32_t)(tmp3 / Cout);

            float acc = 0.0f;

            // Accumulate over input channels and kernel window
            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t kd = 0; kd < K; ++kd) {
                    for (uint32_t kh = 0; kh < K; ++kh) {
                        for (uint32_t kw = 0; kw < K; ++kw) {
                            // input indices (stride=1, padding=0)
                            uint32_t id = od + kd;
                            uint32_t ih = oh + kh;
                            uint32_t iw = ow + kw;

                            // compute flat indices
                            uint64_t xIdx = (((uint64_t)n * Cin + ci) * D + id) * H * W + (uint64_t)ih * W + iw;
                            uint64_t wIdx = (((uint64_t)co * Cin + ci) * K + kd) * K * K + (uint64_t)kh * K + kw;

                            // Load scalar from GM into registers via a 1-element local buffer
                            AscendC::LocalTensor<DTYPE_X> xScalar;
                            AscendC::LocalTensor<DTYPE_Y> wScalar;
                            xScalar.SetSize(1);
                            wScalar.SetSize(1);
                            AscendC::DataCopy(xScalar, xGm[xIdx], 1);
                            AscendC::DataCopy(wScalar, wGm[wIdx], 1);
                            float xv = static_cast<float>(xScalar.GetValue(0));
                            float wv = static_cast<float>(wScalar.GetValue(0));
                            acc += xv * wv;
                        }
                    }
                }
            }
            yLocal.SetValue(i, (DTYPE_Z)acc);
        }

        outQueueY.EnQue(yLocal);
        // Write back to GM
        AscendC::LocalTensor<DTYPE_Z> yLocalOut = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[(uint64_t)tileIdx * this->tileLen], yLocalOut, curLen);
        outQueueY.FreeTensor(yLocalOut);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N, Cin, Cout;
    uint32_t D, H, W;
    uint32_t K;
    uint32_t OutD, OutH, OutW;
    uint32_t tileNum;
    uint64_t blockWork;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3D op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.k, tiling_data.out_d, tiling_data.out_h, tiling_data.out_w,
            tiling_data.totalWork, tiling_data.tileNum);
    op.Process();
}
