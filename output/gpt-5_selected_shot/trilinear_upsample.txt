#include "trilinear_upsample_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    TrilinearUpsampleCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    // Expecting 5D: [N, C, D, H, W]
    uint32_t n  = (in_shape->GetDimNum() >= 1) ? static_cast<uint32_t>(in_shape->GetDim(0)) : 1;
    uint32_t c  = (in_shape->GetDimNum() >= 2) ? static_cast<uint32_t>(in_shape->GetDim(1)) : 1;
    uint32_t inD = (in_shape->GetDimNum() >= 3) ? static_cast<uint32_t>(in_shape->GetDim(2)) : 1;
    uint32_t inH = (in_shape->GetDimNum() >= 4) ? static_cast<uint32_t>(in_shape->GetDim(3)) : 1;
    uint32_t inW = (in_shape->GetDimNum() >= 5) ? static_cast<uint32_t>(in_shape->GetDim(4)) : 1;

    // align_corners=True, scale_factor=2 -> out = (in-1)*2 + 1
    auto out_align_corners_2 = [](uint32_t in_size) -> uint32_t {
        if (in_size <= 1) return in_size;
        return (in_size - 1) * 2 + 1;
    };
    uint32_t outD = out_align_corners_2(inD);
    uint32_t outH = out_align_corners_2(inH);
    uint32_t outW = out_align_corners_2(inW);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inD(inD);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    uint64_t totalOut = static_cast<uint64_t>(n) * c * outD * outH * outW;
    tiling.set_totalOut(totalOut);

    context->SetBlockDim(BLOCK_DIM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    // Default pass-through if not 5D
    if (x_shape->GetDimNum() != 5) {
        *y_shape = *x_shape;
        return GRAPH_SUCCESS;
    }

    int64_t n  = x_shape->GetDim(0);
    int64_t c  = x_shape->GetDim(1);
    int64_t inD = x_shape->GetDim(2);
    int64_t inH = x_shape->GetDim(3);
    int64_t inW = x_shape->GetDim(4);

    auto out_align_corners_2 = [](int64_t in_size) -> int64_t {
        if (in_size <= 1) return in_size;
        return (in_size - 1) * 2 + 1;
    };
    int64_t outD = out_align_corners_2(inD);
    int64_t outH = out_align_corners_2(inH);
    int64_t outW = out_align_corners_2(inW);

    std::vector<int64_t> out_dims = {n, c, outD, outH, outW};
    y_shape->SetDim(out_dims.data(), out_dims.size());
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using DTYPE = float;

extern \"C\" __global__ __aicore__ void trilinear_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const int64_t N   = static_cast<int64_t>(tiling_data.n);
    const int64_t C   = static_cast<int64_t>(tiling_data.c);
    const int64_t inD = static_cast<int64_t>(tiling_data.inD);
    const int64_t inH = static_cast<int64_t>(tiling_data.inH);
    const int64_t inW = static_cast<int64_t>(tiling_data.inW);
    const int64_t outD = static_cast<int64_t>(tiling_data.outD);
    const int64_t outH = static_cast<int64_t>(tiling_data.outH);
    const int64_t outW = static_cast<int64_t>(tiling_data.outW);
    const int64_t totalOut = static_cast<int64_t>(tiling_data.totalOut);

    __gm__ DTYPE* x_ptr = reinterpret_cast<__gm__ DTYPE*>(x);
    __gm__ DTYPE* y_ptr = reinterpret_cast<__gm__ DTYPE*>(y);

    int64_t block_num = AscendC::GetBlockNum();
    int64_t block_idx = AscendC::GetBlockIdx();
    int64_t work_per_block = (totalOut + block_num - 1) / block_num;
    int64_t start = block_idx * work_per_block;
    int64_t end = start + work_per_block;
    if (end > totalOut) end = totalOut;

    const float sd = (outD > 1 && inD > 1) ? static_cast<float>(inD - 1) / static_cast<float>(outD - 1) : 0.0f;
    const float sh = (outH > 1 && inH > 1) ? static_cast<float>(inH - 1) / static_cast<float>(outH - 1) : 0.0f;
    const float sw = (outW > 1 && inW > 1) ? static_cast<float>(inW - 1) / static_cast<float>(outW - 1) : 0.0f;

    for (int64_t outIndex = start; outIndex < end; ++outIndex) {
        int64_t tmp = outIndex;

        int64_t ow = tmp % outW; tmp /= outW;
        int64_t oh = tmp % outH; tmp /= outH;
        int64_t od = tmp % outD; tmp /= outD;
        int64_t c  = tmp % C;    tmp /= C;
        int64_t n  = tmp;

        float in_d_f = sd * static_cast<float>(od);
        float in_h_f = sh * static_cast<float>(oh);
        float in_w_f = sw * static_cast<float>(ow);

        int64_t id0 = static_cast<int64_t>(floorf(in_d_f));
        int64_t ih0 = static_cast<int64_t>(floorf(in_h_f));
        int64_t iw0 = static_cast<int64_t>(floorf(in_w_f));

        int64_t id1 = (id0 + 1 < inD) ? (id0 + 1) : (inD - 1);
        int64_t ih1 = (ih0 + 1 < inH) ? (ih0 + 1) : (inH - 1);
        int64_t iw1 = (iw0 + 1 < inW) ? (iw0 + 1) : (inW - 1);

        float wd = in_d_f - static_cast<float>(id0);
        float wh = in_h_f - static_cast<float>(ih0);
        float ww = in_w_f - static_cast<float>(iw0);

        float w000 = (1.0f - wd) * (1.0f - wh) * (1.0f - ww);
        float w001 = (1.0f - wd) * (1.0f - wh) * (        ww);
        float w010 = (1.0f - wd) * (        wh) * (1.0f - ww);
        float w011 = (1.0f - wd) * (        wh) * (        ww);
        float w100 = (        wd) * (1.0f - wh) * (1.0f - ww);
        float w101 = (        wd) * (1.0f - wh) * (        ww);
        float w110 = (        wd) * (        wh) * (1.0f - ww);
        float w111 = (        wd) * (        wh) * (        ww);

        int64_t base_in_nc = ((n * C) + c) * inD * inH * inW;

        auto off = [&](int64_t d, int64_t h, int64_t w) -> int64_t {
            return base_in_nc + (((d * inH) + h) * inW + w);
        };

        float v000 = x_ptr[off(id0, ih0, iw0)];
        float v001 = x_ptr[off(id0, ih0, iw1)];
        float v010 = x_ptr[off(id0, ih1, iw0)];
        float v011 = x_ptr[off(id0, ih1, iw1)];
        float v100 = x_ptr[off(id1, ih0, iw0)];
        float v101 = x_ptr[off(id1, ih0, iw1)];
        float v110 = x_ptr[off(id1, ih1, iw0)];
        float v111 = x_ptr[off(id1, ih1, iw1)];

        float out_val =
            v000 * w000 + v001 * w001 + v010 * w010 + v011 * w011 +
            v100 * w100 + v101 * w101 + v110 * w110 + v111 * w111;

        y_ptr[outIndex] = out_val;
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline at::IntArrayRef infer_out_shape_align_corners_2(const at::Tensor& x, std::array<int64_t, 5>& out_shape) {
    auto in_sizes = x.sizes();
    // Expect 5D: N,C,D,H,W
    int64_t N = in_sizes[0];
    int64_t C = in_sizes[1];
    int64_t inD = in_sizes[2];
    int64_t inH = in_sizes[3];
    int64_t inW = in_sizes[4];
    auto out_dim = [](int64_t in_sz) -> int64_t {
        if (in_sz <= 1) return in_sz;
        return (in_sz - 1) * 2 + 1;
    };
    out_shape = {N, C, out_dim(inD), out_dim(inH), out_dim(inW)};
    return at::IntArrayRef(out_shape.data(), out_shape.size());
}

at::Tensor trilinear_upsample_custom_impl_npu(const at::Tensor& x) {
    std::array<int64_t, 5> out_shape;
    auto sizes_ref = infer_out_shape_align_corners_2(x, out_shape);
    at::Tensor y = at::empty(sizes_ref, x.options());
    EXEC_NPU_CMD(aclnnTrilinearUpsampleCustom, x, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("trilinear_upsample_custom", &trilinear_upsample_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("trilinear_upsample_custom", &trilinear_upsample_custom_impl_npu, "Trilinear upsample x2 with align_corners=True");
}
