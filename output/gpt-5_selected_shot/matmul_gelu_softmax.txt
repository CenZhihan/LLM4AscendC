kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulGeluSoftmax {
public:
    __aicore__ inline KernelMatmulGeluSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN, uint32_t tileK, uint32_t rowsPerBlock) {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileN = tileN;
        this->tileK = tileK;

        uint32_t blockId = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t startRow = rowsPerBlock * blockId;
        uint32_t endRow = AscendC::Min(startRow + rowsPerBlock, M);
        this->rowStart = startRow;
        this->rowEnd = endRow;
        this->rowsPerBlockActual = (endRow > startRow) ? (endRow - startRow) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (uint64_t)startRow * K, (uint64_t)this->rowsPerBlockActual * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint64_t)N * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b, (uint64_t)N);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + (uint64_t)startRow * N, (uint64_t)this->rowsPerBlockActual * N);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileK * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, this->tileN * this->tileK * sizeof(DTYPE_W));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileN * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileN * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileN * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process() {
        for (uint32_t r = 0; r < this->rowsPerBlockActual; ++r) {
            ComputeRow(r);
        }
    }

private:
    __aicore__ inline void CopyInX(uint32_t kkOffset) {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        const uint64_t base = (uint64_t)kkOffset;
        AscendC::DataCopy(xLocal, xGm[rowXOffset + base], this->tileK);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void CopyInW(uint32_t tnOffset, uint32_t kkOffset) {
        AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.AllocTensor<DTYPE_W>();
        // copy tileN rows of length tileK from wGm
        for (uint32_t i = 0; i < this->tileN; ++i) {
            uint64_t wBase = ((uint64_t)(tnOffset + i)) * K + kkOffset;
            AscendC::DataCopy(wLocal[i * this->tileK], wGm[wBase], this->tileK);
        }
        inQueueW.EnQue(wLocal);
    }

    __aicore__ inline void CopyInB(uint32_t tnOffset) {
        AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
        AscendC::DataCopy(bLocal, bGm[tnOffset], this->tileN);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void ComputeRow(uint32_t r) {
        // base offsets for this row in x and y
        rowXOffset = (uint64_t)r * K;
        uint64_t rowYBase = (uint64_t)r * N;

        // 1) Linear + GELU: y = GELU(x * W^T + b)
        for (uint32_t tn = 0; tn < N; tn += this->tileN) {
            uint32_t currentTileN = AscendC::Min(this->tileN, N - tn);

            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
            // init accum to 0
            for (uint32_t i = 0; i < currentTileN; ++i) {
                yLocal[i] = (DTYPE_Y)0.0f;
            }

            for (uint32_t kk = 0; kk < K; kk += this->tileK) {
                uint32_t currentTileK = AscendC::Min(this->tileK, K - kk);

                // load x tile
                CopyInX(kk);
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();

                // load w tile (tileN x tileK)
                CopyInW(tn, kk);
                AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.DeQue<DTYPE_W>();

                // compute yLocal += wLocal * xLocal
                for (uint32_t i = 0; i < currentTileN; ++i) {
                    DTYPE_Y acc = (DTYPE_Y)0.0f;
                    for (uint32_t k = 0; k < currentTileK; ++k) {
                        acc += (DTYPE_Y)(wLocal[i * this->tileK + k] * xLocal[k]);
                    }
                    yLocal[i] += acc;
                }

                inQueueX.FreeTensor(xLocal);
                inQueueW.FreeTensor(wLocal);
            }

            // add bias
            CopyInB(tn);
            AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.DeQue<DTYPE_B>();
            for (uint32_t i = 0; i < currentTileN; ++i) {
                yLocal[i] += (DTYPE_Y)bLocal[i];
            }
            inQueueB.FreeTensor(bLocal);

            // apply GELU
            AscendC::Gelu(yLocal, yLocal, currentTileN);

            // write logits to GM
            AscendC::DataCopy(yGm[rowYBase + tn], yLocal, currentTileN);
            outQueueY.FreeTensor(yLocal);
        }

        // 2) Row-wise Softmax over N
        // compute max
        DTYPE_Y rowMax = (DTYPE_Y)-1e30f;
        for (uint32_t tn = 0; tn < N; tn += this->tileN) {
            uint32_t currentTileN = AscendC::Min(this->tileN, N - tn);
            AscendC::LocalTensor<DTYPE_Y> tmp = tmpQueue.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(tmp, yGm[rowYBase + tn], currentTileN);
            for (uint32_t i = 0; i < currentTileN; ++i) {
                if (tmp[i] > rowMax) rowMax = tmp[i];
            }
            tmpQueue.FreeTensor(tmp);
        }

        // compute exp(x - max) and sum
        DTYPE_Y rowSum = (DTYPE_Y)0.0f;
        for (uint32_t tn = 0; tn < N; tn += this->tileN) {
            uint32_t currentTileN = AscendC::Min(this->tileN, N - tn);
            AscendC::LocalTensor<DTYPE_Y> tmp = tmpQueue.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(tmp, yGm[rowYBase + tn], currentTileN);
            for (uint32_t i = 0; i < currentTileN; ++i) {
                tmp[i] = (DTYPE_Y)AscendC::Exp(tmp[i] - rowMax);
                rowSum += tmp[i];
            }
            AscendC::DataCopy(yGm[rowYBase + tn], tmp, currentTileN); // store exp values
            tmpQueue.FreeTensor(tmp);
        }

        // normalize
        for (uint32_t tn = 0; tn < N; tn += this->tileN) {
            uint32_t currentTileN = AscendC::Min(this->tileN, N - tn);
            AscendC::LocalTensor<DTYPE_Y> tmp = tmpQueue.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(tmp, yGm[rowYBase + tn], currentTileN);
            for (uint32_t i = 0; i < currentTileN; ++i) {
                tmp[i] = tmp[i] / rowSum;
            }
            AscendC::DataCopy(yGm[rowYBase + tn], tmp, currentTileN);
            tmpQueue.FreeTensor(tmp);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileN;
    uint32_t tileK;

    uint32_t rowStart;
    uint32_t rowEnd;
    uint32_t rowsPerBlockActual;
    uint64_t rowXOffset;
};

extern "C" __global__ __aicore__ void matmul_gelu_softmax_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulGeluSoftmax op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileN, tiling_data.tileK, tiling_data.rowsPerBlock);
    op.Process();
}
