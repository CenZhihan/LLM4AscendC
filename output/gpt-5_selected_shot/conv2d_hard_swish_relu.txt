kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv2d_hard_swish_relu_custom(GM_ADDR x,
                                                                     GM_ADDR weight,
                                                                     GM_ADDR bias,
                                                                     GM_ADDR y,
                                                                     GM_ADDR workspace,
                                                                     GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    uint32_t N  = tiling_data.n;
    uint32_t IC = tiling_data.ic;
    uint32_t IH = tiling_data.ih;
    uint32_t IW = tiling_data.iw;
    uint32_t OC = tiling_data.oc;
    uint32_t KH = tiling_data.kh;
    uint32_t KW = tiling_data.kw;
    uint32_t OH = tiling_data.oh;
    uint32_t OW = tiling_data.ow;
    uint32_t strideH = tiling_data.strideH;
    uint32_t strideW = tiling_data.strideW;
    uint32_t padH = tiling_data.padH;
    uint32_t padW = tiling_data.padW;

    __gm__ float* x_ptr = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_ptr = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* b_ptr = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* y_ptr = reinterpret_cast<__gm__ float*>(y);

    uint32_t total = N * OC * OH * OW;
    uint32_t blocks = AscendC::GetBlockNum();
    uint32_t idx = AscendC::GetBlockIdx();

    uint32_t base = total / blocks;
    uint32_t extras = total % blocks;
    uint32_t myLen = base + (idx < extras ? 1U : 0U);
    uint32_t start = idx * base + (idx < extras ? idx : extras);

    for (uint32_t p = 0; p < myLen; ++p) {
        uint32_t linear = start + p;

        uint32_t ow_i = linear % OW; linear /= OW;
        uint32_t oh_i = linear % OH; linear /= OH;
        uint32_t oc_i = linear % OC; linear /= OC;
        uint32_t n_i  = linear;

        float acc = b_ptr != nullptr ? b_ptr[oc_i] : 0.0f;

        for (uint32_t ic_i = 0; ic_i < IC; ++ic_i) {
            for (uint32_t kh_i = 0; kh_i < KH; ++kh_i) {
                int ih_i = static_cast<int>(oh_i * strideH) + static_cast<int>(kh_i) - static_cast<int>(padH);
                if (ih_i < 0 || ih_i >= static_cast<int>(IH)) continue;
                for (uint32_t kw_i = 0; kw_i < KW; ++kw_i) {
                    int iw_i = static_cast<int>(ow_i * strideW) + static_cast<int>(kw_i) - static_cast<int>(padW);
                    if (iw_i < 0 || iw_i >= static_cast<int>(IW)) continue;

                    uint32_t x_off = (((n_i * IC + ic_i) * IH + static_cast<uint32_t>(ih_i)) * IW + static_cast<uint32_t>(iw_i));
                    uint32_t w_off = (((oc_i * IC + ic_i) * KH + kh_i) * KW + kw_i);

                    acc += x_ptr[x_off] * w_ptr[w_off];
                }
            }
        }

        // HardSwish: x * clamp(x + 3, 0, 6) / 6
        float t = acc + 3.0f;
        if (t < 0.0f) t = 0.0f;
        if (t > 6.0f) t = 6.0f;
        float hs = acc * (t / 6.0f);
        // ReLU
        float out = hs > 0.0f ? hs : 0.0f;

        uint32_t y_off = (((n_i * OC + oc_i) * OH + oh_i) * OW + ow_i);
        y_ptr[y_off] = out;
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv2d_hard_swish_relu_custom_impl_npu(const at::Tensor& x,
                                                  const at::Tensor& weight,
                                                  const at::Tensor& bias) {
    TORCH_CHECK(x.dim() == 4, \"Input x must be 4D NCHW\");
    TORCH_CHECK(weight.dim() == 4, \"Weight must be 4D [OC, IC, KH, KW]\");
    TORCH_CHECK(bias.dim() == 1, \"Bias must be 1D [OC]\");
    auto N = x.size(0);
    auto IC = x.size(1);
    auto IH = x.size(2);
    auto IW = x.size(3);
    auto OC = weight.size(0);
    auto ICw = weight.size(1);
    auto KH = weight.size(2);
    auto KW = weight.size(3);

    TORCH_CHECK(IC == ICw, \"Input channels mismatch between x and weight\");

    // stride=1, padding=0
    auto OH = IH - KH + 1;
    auto OW = IW - KW + 1;
    TORCH_CHECK(OH > 0 && OW > 0, \"Invalid output spatial size\");
    at::Tensor result = at::empty({N, OC, OH, OW}, x.options());

    EXEC_NPU_CMD(aclnnConv2dHardSwishReluCustom, x, weight, bias, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv2d_hard_swish_relu_custom\", &conv2d_hard_swish_relu_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"conv2d_hard_swish_relu_custom\", &conv2d_hard_swish_relu_custom_impl_npu, \"conv2d + HardSwish + ReLU fused\");

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that fuses Conv2d + HardSwish + ReLU into a single custom AscendC kernel.
    Assumes stride=1 and padding=0 to match the provided baseline.
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)

    def forward(self, x):
        return custom_ops_lib.conv2d_hard_swish_relu_custom(x, self.conv.weight, self.conv.bias)
