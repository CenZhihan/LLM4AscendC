kernel_src="""
#include "kernel_operator.h"
#include <cmath>

extern "C" __global__ __aicore__ void convtranspose3d_mean_add_softmax_tanh_scaling_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    // Dimensions
    const uint32_t N = tiling_data.n;
    const uint32_t C = tiling_data.c;
    const uint32_t D = tiling_data.d;
    const uint32_t H = tiling_data.h;
    const uint32_t W = tiling_data.w;

    __gm__ float* xg = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* bg = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* sg = reinterpret_cast<__gm__ float*>(scale);
    __gm__ float* yg = reinterpret_cast<__gm__ float*>(y);

    float alpha = sg[0];

    // Partition work across blocks by (N * H * W)
    uint64_t NHW = static_cast<uint64_t>(N) * H * W;
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();

    uint64_t start = (NHW * blockIdx) / blockNum;
    uint64_t end   = (NHW * (blockIdx + 1)) / blockNum;

    for (uint64_t s = start; s < end; ++s) {
        uint32_t n = static_cast<uint32_t>(s / (H * W));
        uint32_t rem = static_cast<uint32_t>(s % (H * W));
        uint32_t h = rem / W;
        uint32_t w = rem % W;

        // First pass: find max over channels for numerical stability
        float maxv = -1.0e30f;
        for (uint32_t c = 0; c < C; ++c) {
            // Compute mean over depth for (n, c, h, w)
            float sum = 0.0f;
            // Base index for d=0
            uint64_t base = (((static_cast<uint64_t>(n) * C + c) * D) * H + h) * W + w;
            for (uint32_t d = 0; d < D; ++d) {
                uint64_t idx = base + static_cast<uint64_t>(d) * H * W;
                sum += xg[idx];
            }
            float mean_val = sum / static_cast<float>(D);
            float val = mean_val + bg[c]; // broadcast bias per channel
            if (val > maxv) {
                maxv = val;
            }
        }

        // Second pass: compute sum of exp
        float sumexp = 0.0f;
        for (uint32_t c = 0; c < C; ++c) {
            float sum = 0.0f;
            uint64_t base = (((static_cast<uint64_t>(n) * C + c) * D) * H + h) * W + w;
            for (uint32_t d = 0; d < D; ++d) {
                uint64_t idx = base + static_cast<uint64_t>(d) * H * W;
                sum += xg[idx];
            }
            float mean_val = sum / static_cast<float>(D);
            float val = mean_val + bg[c];
            float e = expf(val - maxv);
            sumexp += e;
        }

        // Third pass: write output after softmax, tanh, and scaling
        for (uint32_t c = 0; c < C; ++c) {
            float sum = 0.0f;
            uint64_t base = (((static_cast<uint64_t>(n) * C + c) * D) * H + h) * W + w;
            for (uint32_t d = 0; d < D; ++d) {
                uint64_t idx = base + static_cast<uint64_t>(d) * H * W;
                sum += xg[idx];
            }
            float mean_val = sum / static_cast<float>(D);
            float val = mean_val + bg[c];
            float soft = expf(val - maxv) / sumexp;
            float t = tanhf(soft);
            float outv = t * alpha;

            // Output index for (n, c, d'=1, h, w) with D'=1
            uint64_t out_idx = (((static_cast<uint64_t>(n) * C + c) * 1) * H + h) * W + w;
            yg[out_idx] = outv;
        }
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: runs ConvTranspose3d using PyTorch, then fused custom AscendC kernel for:
    mean over depth, bias add, softmax over channels, tanh, and scaling.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.bias = torch.nn.Parameter(torch.randn(1, out_channels, 1, 1, 1))
        self.scaling_factor = scaling_factor

    def forward(self, x):
        x = self.conv_transpose(x)  # (B, C, D, H, W)
        scale = torch.tensor(self.scaling_factor, dtype=x.dtype, device=x.device).view(1)
        y = custom_ops_lib.convtranspose3d_mean_add_softmax_tanh_scaling_custom(x, self.bias, scale)
