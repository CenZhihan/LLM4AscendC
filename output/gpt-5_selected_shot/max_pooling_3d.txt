inline int64_t compute_out_len(int64_t in, int64_t k, int64_t s, int64_t p, bool ceil_mode) {
    const int64_t num = in + 2 * p - k;
    if (ceil_mode) {
        return (num + s) / s + 1 - ((num + s) % s == 0 ? 0 : 0);
    }
    return num / s + 1;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// Naive 3D max pooling kernel over NCDHW layout
class KernelMaxPool3d {
public:
    __aicore__ inline KernelMaxPool3d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t sd, uint32_t sh, uint32_t sw,
                                uint32_t pd, uint32_t ph, uint32_t pw,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint64_t totalOut, uint32_t tileNum)
    {
        this->N = n; this->C = c; this->D = d; this->H = h; this->W = w;
        this->kD = kd; this->kH = kh; this->kW = kw;
        this->sD = sd; this->sH = sh; this->sW = sw;
        this->pD = pd; this->pH = ph; this->pW = pw;
        this->outD = outD; this->outH = outH; this->outW = outW;
        this->tileNum = tileNum;

        this->blockOut = totalOut / AscendC::GetBlockNum();
        this->tileLength = (uint32_t)(this->blockOut / tileNum / BUFFER_NUM);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)n * c * d * h * w);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockOut * AscendC::GetBlockIdx(), this->blockOut);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        uint64_t start = (uint64_t)progress * this->tileLength;
        for (uint32_t t = 0; t < this->tileLength; ++t) {
            uint64_t idx = start + t + this->blockOut * AscendC::GetBlockIdx();
            if (idx >= this->N * (uint64_t)this->C * this->outD * this->outH * this->outW) {
                continue;
            }
            // Unravel index into N, C, od, oh, ow
            uint64_t tmp = idx;
            uint32_t ow = (uint32_t)(tmp % this->outW); tmp /= this->outW;
            uint32_t oh = (uint32_t)(tmp % this->outH); tmp /= this->outH;
            uint32_t od = (uint32_t)(tmp % this->outD); tmp /= this->outD;
            uint32_t c = (uint32_t)(tmp % this->C); tmp /= this->C;
            uint32_t n = (uint32_t)tmp;

            int32_t id0 = (int32_t)od * (int32_t)this->sD - (int32_t)this->pD;
            int32_t ih0 = (int32_t)oh * (int32_t)this->sH - (int32_t)this->pH;
            int32_t iw0 = (int32_t)ow * (int32_t)this->sW - (int32_t)this->pW;

            bool inited = false;
            DTYPE_Y maxv = (DTYPE_Y)0;

            for (uint32_t kd = 0; kd < this->kD; ++kd) {
                int32_t id = id0 + (int32_t)kd;
                if (id < 0 || id >= (int32_t)this->D) continue;
                for (uint32_t kh = 0; kh < this->kH; ++kh) {
                    int32_t ih = ih0 + (int32_t)kh;
                    if (ih < 0 || ih >= (int32_t)this->H) continue;
                    for (uint32_t kw = 0; kw < this->kW; ++kw) {
                        int32_t iw = iw0 + (int32_t)kw;
                        if (iw < 0 || iw >= (int32_t)this->W) continue;

                        uint64_t inIndex =
                            (((((uint64_t)n * this->C + c) * this->D + (uint32_t)id) * this->H + (uint32_t)ih) * this->W + (uint32_t)iw);

                        // Load single element from GM to local scalar
                        // Allocate a 1-element local tensor and copy
                        AscendC::LocalTensor<DTYPE_X> local1;
                        // Using pipe to allocate a temporary small buffer is optional; here we read via a temporary variable
                        // DataCopy with length 1
                        DTYPE_X val;
                        AscendC::DataCopy(&val, xGm[inIndex], 1);
                        if (!inited) {
                            maxv = (DTYPE_Y)val;
                            inited = true;
                        } else {
                            maxv = (val > maxv) ? (DTYPE_Y)val : maxv;
                        }
                    }
                }
            }
            yLocal.SetValue(t, maxv);
        }
        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, C, D, H, W;
    uint32_t kD, kH, kW;
    uint32_t sD, sH, sW;
    uint32_t pD, pH, pW;
    uint32_t outD, outH, outW;
    uint32_t tileNum;
    uint64_t blockOut;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPool3d op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.sd, tiling_data.sh, tiling_data.sw,
            tiling_data.pd, tiling_data.ph, tiling_data.pw,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.totalOut, tiling_data.tileNum);
    op.Process();
}
