const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto* in_shape = context->GetInputShape(0);
    auto* w_shape  = context->GetInputShape(1);

    uint32_t n  = GetDimSafe(in_shape, 0, 1);
    uint32_t ci = GetDimSafe(in_shape, 1, 1);
    uint32_t hi = GetDimSafe(in_shape, 2, 1);
    uint32_t wi = GetDimSafe(in_shape, 3, 1);

    uint32_t k  = GetDimSafe(w_shape, 2, 1); // square kernel (k x k)
    uint32_t co = GetDimSafe(w_shape, 1, 1);

    // Get attributes
    int64_t stride_attr   = 1;
    int64_t padding_attr  = 0;
    int64_t dilation_attr = 1;
    // Best-effort attribute access (APIs may differ by version)
    if (context->HasAttr("stride"))   { stride_attr   = context->GetAttrInt("stride"); }
    if (context->HasAttr("padding"))  { padding_attr  = context->GetAttrInt("padding"); }
    if (context->HasAttr("dilation")) { dilation_attr = context->GetAttrInt("dilation"); }

    uint32_t stride   = static_cast<uint32_t>(stride_attr);
    uint32_t padding  = static_cast<uint32_t>(padding_attr);
    uint32_t dilation = static_cast<uint32_t>(dilation_attr);

    // Output spatial size for transposed conv: H_out = (H_in - 1) * stride - 2*padding + dilation * (k - 1) + 1
    uint32_t ho = (hi - 1) * stride - 2 * padding + dilation * (k - 1) + 1;
    uint32_t wo = (wi - 1) * stride - 2 * padding + dilation * (k - 1) + 1;

    context->SetBlockDim(BLOCK_DIM);

    ConvTransposed2dAsymmetricInputSquareKernelDilatedPaddedStridedCustomTilingData tiling;
    tiling.set_n(n);
    tiling.set_ci(ci);
    tiling.set_co(co);
    tiling.set_hi(hi);
    tiling.set_wi(wi);
    tiling.set_k(k);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_dilation(dilation);
    tiling.set_ho(ho);
    tiling.set_wo(wo);
    tiling.set_totalOutLen(n * co * ho * wo);
    tiling.set_tileNum(TILE_NUM);
    // bias present if third input exists
    uint32_t hasBias = (context->GetInputShapeNum() >= 3 && context->GetInputShape(2) != nullptr) ? 1U : 0U;

    return ge::GRAPH_SUCCESS;

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0); // N, Ci, Hi, Wi
    const gert::Shape* w_shape = context->GetInputShape(1); // Ci, Co, K, K
    if (x_shape == nullptr || w_shape == nullptr) {
        return GRAPH_FAILED;
    }
    // Attributes
    int64_t stride   = 1;
    int64_t padding  = 0;
    int64_t dilation = 1;
    if (context->HasAttr("stride"))   { stride   = context->GetAttrInt("stride"); }
    if (context->HasAttr("padding"))  { padding  = context->GetAttrInt("padding"); }
    if (context->HasAttr("dilation")) { dilation = context->GetAttrInt("dilation"); }

    int64_t N  = x_shape->GetDim(0);
    int64_t Ci = x_shape->GetDim(1);
    int64_t Hi = x_shape->GetDim(2);
    int64_t Wi = x_shape->GetDim(3);
    int64_t K  = w_shape->GetDim(2);
    int64_t Co = w_shape->GetDim(1);

    int64_t Ho = (Hi - 1) * stride - 2 * padding + dilation * (K - 1) + 1;
    int64_t Wo = (Wi - 1) * stride - 2 * padding + dilation * (K - 1) + 1;

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Co);
    y_shape->SetDim(2, Ho);
    y_shape->SetDim(3, Wo);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_B = float;
using DTYPE_Y = float;

extern \"C\" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N  = tiling_data.n;
    const uint32_t Ci = tiling_data.ci;
    const uint32_t Co = tiling_data.co;
    const uint32_t Hi = tiling_data.hi;
    const uint32_t Wi = tiling_data.wi;
    const uint32_t K  = tiling_data.k;
    const uint32_t Stride   = tiling_data.stride;
    const uint32_t Padding  = tiling_data.padding;
    const uint32_t Dilation = tiling_data.dilation;
    const uint32_t Ho = tiling_data.ho;
    const uint32_t Wo = tiling_data.wo;
    const uint32_t totalOutLen = tiling_data.totalOutLen;
    const uint32_t hasBias = tiling_data.hasBias;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    // Set buffers
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();

    uint32_t blockLen = (totalOutLen + blockNum - 1) / blockNum;  // ceiling
    uint32_t startIdx = blockIdx * blockLen;
    uint32_t endIdx = startIdx + blockLen;
    if (endIdx > totalOutLen) endIdx = totalOutLen;

    xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, N * Ci * Hi * Wi);
    wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, Ci * Co * K * K);
    yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, N * Co * Ho * Wo);
    if (hasBias) {
        bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, Co);
    }

    // Strides for indexing
    const uint32_t x_stride_n = Ci * Hi * Wi;
    const uint32_t x_stride_c = Hi * Wi;
    const uint32_t x_stride_h = Wi;
    const uint32_t x_stride_w = 1;

    const uint32_t w_stride_ci = Co * K * K;
    const uint32_t w_stride_co = K * K;
    const uint32_t w_stride_kh = K;
    const uint32_t w_stride_kw = 1;

    const uint32_t y_stride_n = Co * Ho * Wo;
    const uint32_t y_stride_c = Ho * Wo;
    const uint32_t y_stride_h = Wo;
    const uint32_t y_stride_w = 1;

    for (uint32_t outIndex = startIdx; outIndex < endIdx; ++outIndex) {
        // Map flat index to (n, co, ho, wo)
        uint32_t tmp = outIndex;
        uint32_t n = tmp / y_stride_n; tmp -= n * y_stride_n;
        uint32_t co = tmp / y_stride_c; tmp -= co * y_stride_c;
        uint32_t ho = tmp / y_stride_h; tmp -= ho * y_stride_h;
        uint32_t wo = tmp; // remaining

        float acc = 0.0f;

        // Backward mapping: ho + padding = hi*stride + dilation*kh
        //                   wo + padding = wi*stride + dilation*kw
        const int32_t th = static_cast<int32_t>(ho) + static_cast<int32_t>(Padding);
        const int32_t tw = static_cast<int32_t>(wo) + static_cast<int32_t>(Padding);

        for (uint32_t ci = 0; ci < Ci; ++ci) {
            for (uint32_t kh = 0; kh < K; ++kh) {
                int32_t h_nom = th - static_cast<int32_t>(Dilation) * static_cast<int32_t>(kh);
                if (h_nom < 0) continue;
                if (h_nom % static_cast<int32_t>(Stride) != 0) continue;
                int32_t hi = h_nom / static_cast<int32_t>(Stride);
                if (hi < 0 || hi >= static_cast<int32_t>(Hi)) continue;

                for (uint32_t kw = 0; kw < K; ++kw) {
                    int32_t w_nom = tw - static_cast<int32_t>(Dilation) * static_cast<int32_t>(kw);
                    if (w_nom < 0) continue;
                    if (w_nom % static_cast<int32_t>(Stride) != 0) continue;
                    int32_t wi = w_nom / static_cast<int32_t>(Stride);
                    if (wi < 0 || wi >= static_cast<int32_t>(Wi)) continue;

                    uint32_t x_off = n * x_stride_n + ci * x_stride_c + static_cast<uint32_t>(hi) * x_stride_h + static_cast<uint32_t>(wi) * x_stride_w;
                    uint32_t w_off = ci * w_stride_ci + co * w_stride_co + kh * w_stride_kh + kw * w_stride_kw;

                    float xv = xGm[x_off];
                    float wv = wGm[w_off];
                    acc += xv * wv;
                }
            }
        }

        if (hasBias) {
            acc += bGm[co];
        }

        uint32_t y_off = n * y_stride_n + co * y_stride_c + ho * y_stride_h + wo * y_stride_w;
        yGm[y_off] = acc;
    }
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Optimized 2D transposed convolution using custom AscendC kernel.
    Supports asymmetric input, square kernel, dilation, padding, and stride.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        # Weight layout follows PyTorch ConvTranspose2d: (in_channels, out_channels, kH, kW)
        weight = torch.empty(in_channels, out_channels, kernel_size, kernel_size)
        nn.init.kaiming_uniform_(weight, a=5**0.5)
        self.weight = nn.Parameter(weight)
        if bias:
            self.bias = nn.Parameter(torch.zeros(out_channels))
        else:
            self.register_parameter("bias", None)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom(
            x, self.weight, self.bias, self.stride, self.padding, self.dilation
        )
