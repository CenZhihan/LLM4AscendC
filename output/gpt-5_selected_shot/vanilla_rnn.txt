kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnn {
public:
    __aicore__ inline KernelVanillaRnn() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR h,
        GM_ADDR w_i2h, GM_ADDR b_i2h,
        GM_ADDR w_h2o, GM_ADDR b_h2o,
        GM_ADDR h_new, GM_ADDR y,
        uint32_t batchSize, uint32_t inputSize, uint32_t hiddenSize, uint32_t outputSize, uint32_t tileNum)
    {
        this->batchSize = batchSize;
        this->inputSize = inputSize;
        this->hiddenSize = hiddenSize;
        this->outputSize = outputSize;
        this->tileNum = tileNum;

        this->rowsPerBlock = batchSize / AscendC::GetBlockNum();
        // Tile over rows
        this->tileRows = AscendC::Max(1U, rowsPerBlock / tileNum / BUFFER_NUM);

        xGm.SetGlobalBuffer((__gm__ float *)x + rowsPerBlock * inputSize * AscendC::GetBlockIdx(), rowsPerBlock * inputSize);
        hGm.SetGlobalBuffer((__gm__ float *)h + rowsPerBlock * hiddenSize * AscendC::GetBlockIdx(), rowsPerBlock * hiddenSize);
        wI2hGm.SetGlobalBuffer((__gm__ float *)w_i2h, hiddenSize * (inputSize + hiddenSize));
        bI2hGm.SetGlobalBuffer((__gm__ float *)b_i2h, hiddenSize);
        wH2oGm.SetGlobalBuffer((__gm__ float *)w_h2o, outputSize * hiddenSize);
        bH2oGm.SetGlobalBuffer((__gm__ float *)b_h2o, outputSize);
        hNewGm.SetGlobalBuffer((__gm__ float *)h_new + rowsPerBlock * hiddenSize * AscendC::GetBlockIdx(), rowsPerBlock * hiddenSize);
        yGm.SetGlobalBuffer((__gm__ float *)y + rowsPerBlock * outputSize * AscendC::GetBlockIdx(), rowsPerBlock * outputSize);

        // Buffers for a tile of rows
        pipe.InitBuffer(inQueueX, BUFFER_NUM, tileRows * inputSize * sizeof(float));
        pipe.InitBuffer(inQueueH, BUFFER_NUM, tileRows * hiddenSize * sizeof(float));
        pipe.InitBuffer(outQueueHNew, BUFFER_NUM, tileRows * hiddenSize * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileRows * outputSize * sizeof(float));

        // Optional: preload biases (read-only small tensors)
        AscendC::LocalTensor<float> b_i2h_local;
        b_i2h_local.SetSize(hiddenSize);
        AscendC::DataCopy(b_i2h_local, bI2hGm, hiddenSize);
        biasI2hLocal = b_i2h_local;

        AscendC::LocalTensor<float> b_h2o_local;
        b_h2o_local.SetSize(outputSize);
        AscendC::DataCopy(b_h2o_local, bH2oGm, outputSize);
        biasH2oLocal = b_h2o_local;
    }

    __aicore__ inline void Process()
    {
        int32_t totalTiles = AscendC::Ceil(rowsPerBlock, tileRows);
        for (int32_t t = 0; t < totalTiles; ++t) {
            int32_t rowsThisTile = AscendC::Min((uint32_t)tileRows, rowsPerBlock - t * tileRows);
            CopyIn(t, rowsThisTile);
            Compute(t, rowsThisTile);
            CopyOut(t, rowsThisTile);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t tileIdx, int32_t rowsThisTile)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>(rowsThisTile * inputSize);
        AscendC::LocalTensor<float> hLocal = inQueueH.AllocTensor<float>(rowsThisTile * hiddenSize);

        uint32_t xOffset = tileIdx * tileRows * inputSize;
        uint32_t hOffset = tileIdx * tileRows * hiddenSize;

        AscendC::DataCopy(xLocal, xGm[xOffset], rowsThisTile * inputSize);
        AscendC::DataCopy(hLocal, hGm[hOffset], rowsThisTile * hiddenSize);

        inQueueX.EnQue(xLocal);
        inQueueH.EnQue(hLocal);
    }

    __aicore__ inline void Compute(int32_t tileIdx, int32_t rowsThisTile)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> hLocal = inQueueH.DeQue<float>();

        AscendC::LocalTensor<float> hNewLocal = outQueueHNew.AllocTensor<float>(rowsThisTile * hiddenSize);
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>(rowsThisTile * outputSize);

        // Preload weights into local once per compute call (can be cached across tiles depending on size)
        AscendC::LocalTensor<float> wI2hLocal;
        wI2hLocal.SetSize(hiddenSize * (inputSize + hiddenSize));
        AscendC::DataCopy(wI2hLocal, wI2hGm, hiddenSize * (inputSize + hiddenSize));

        AscendC::LocalTensor<float> wH2oLocal;
        wH2oLocal.SetSize(outputSize * hiddenSize);
        AscendC::DataCopy(wH2oLocal, wH2oGm, outputSize * hiddenSize);

        // For each row in tile: combined = concat(x_row, h_row)
        for (int r = 0; r < rowsThisTile; ++r) {
            // Views
            AscendC::LocalTensor<float> xRow = xLocal.Slice(r * inputSize, (r + 1) * inputSize);
            AscendC::LocalTensor<float> hRow = hLocal.Slice(r * hiddenSize, (r + 1) * hiddenSize);
            AscendC::LocalTensor<float> hNewRow = hNewLocal.Slice(r * hiddenSize, (r + 1) * hiddenSize);
            AscendC::LocalTensor<float> yRow = yLocal.Slice(r * outputSize, (r + 1) * outputSize);

            // Build combined row
            AscendC::LocalTensor<float> combinedRow;
            combinedRow.SetSize(inputSize + hiddenSize);
            AscendC::DataCopy(combinedRow.Slice(0, inputSize), xRow, inputSize);
            AscendC::DataCopy(combinedRow.Slice(inputSize, inputSize + hiddenSize), hRow, hiddenSize);

            // h_new_row = tanh(combinedRow @ w_i2h^T + b_i2h)
            // MatMul: (1 x (I+H)) * ((I+H) x H) => (1 x H)
            AscendC::MatMul(hNewRow, combinedRow, wI2hLocal, 1, hiddenSize, inputSize + hiddenSize);
            AscendC::Add(hNewRow, hNewRow, biasI2hLocal, hiddenSize);
            AscendC::Tanh(hNewRow, hNewRow, hiddenSize);

            // y_row = h_new_row @ w_h2o^T + b_h2o
            // MatMul: (1 x H) * (H x O) => (1 x O)
            AscendC::MatMul(yRow, hNewRow, wH2oLocal, 1, outputSize, hiddenSize);
            AscendC::Add(yRow, yRow, biasH2oLocal, outputSize);
        }

        outQueueHNew.EnQue(hNewLocal);
        outQueueY.EnQue(yLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueH.FreeTensor(hLocal);
    }

    __aicore__ inline void CopyOut(int32_t tileIdx, int32_t rowsThisTile)
    {
        AscendC::LocalTensor<float> hNewLocal = outQueueHNew.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();

        uint32_t hNewOffset = tileIdx * tileRows * hiddenSize;
        uint32_t yOffset = tileIdx * tileRows * outputSize;

        AscendC::DataCopy(hNewGm[hNewOffset], hNewLocal, rowsThisTile * hiddenSize);
        AscendC::DataCopy(yGm[yOffset], yLocal, rowsThisTile * outputSize);

        outQueueHNew.FreeTensor(hNewLocal);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueH;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueHNew, outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> hGm;
    AscendC::GlobalTensor<float> wI2hGm;
    AscendC::GlobalTensor<float> bI2hGm;
    AscendC::GlobalTensor<float> wH2oGm;
    AscendC::GlobalTensor<float> bH2oGm;
    AscendC::GlobalTensor<float> hNewGm;
    AscendC::GlobalTensor<float> yGm;

    AscendC::LocalTensor<float> biasI2hLocal;
    AscendC::LocalTensor<float> biasH2oLocal;

    uint32_t batchSize;
    uint32_t inputSize;
    uint32_t hiddenSize;
    uint32_t outputSize;
    uint32_t tileNum;
    uint32_t rowsPerBlock;
    uint32_t tileRows;
};

extern "C" __global__ __aicore__ void vanilla_rnn_custom(
    GM_ADDR x, GM_ADDR h,
    GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o,
    GM_ADDR h_new, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnn op;
    op.Init(
        x, h, w_i2h, b_i2h, w_h2o, b_h2o, h_new, y,
        tiling_data.batchSize, tiling_data.inputSize, tiling_data.hiddenSize, tiling_data.outputSize, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size: int, hidden_size: int, output_size: int):
        super(ModelNew, self).__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size

        # Parameters same as original Model
        self.i2h = torch.nn.Linear(input_size + hidden_size, hidden_size)
        self.h2o = torch.nn.Linear(hidden_size, output_size)
        self.tanh = torch.nn.Tanh()

        # Hidden state buffer
        self.hidden = torch.randn((256, hidden_size))  # default batch_size=256 as in original script

    def forward(self, x: torch.Tensor, initial_hidden=None) -> torch.Tensor:
        if initial_hidden is not None:
            self.hidden.copy_(initial_hidden)
        self.hidden = self.hidden.to(x.device)

        # Fused custom operator: computes new hidden and output
        h_new, y = custom_ops_lib.vanilla_rnn_custom(
            x,
            self.hidden,
            self.i2h.weight,   # [H, I+H]
            self.i2h.bias,     # [H]
            self.h2o.weight,   # [O, H]
            self.h2o.bias      # [O]
        )

        self.hidden.copy_(h_new)
        return y
'''
