static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GroupQueryAttentionCustomTilingData tiling;
    const auto* inShape = context->GetInputShape(0);
    // Expect input shape as [B, H, L, Hd] or [BH, L, Hd]; we use origin dims
    auto originShape = inShape->GetOriginShape();
    uint32_t dims = originShape.GetDimNum();
    uint32_t batch = 1;
    uint32_t heads = 1;
    uint32_t seqLen = 1;
    uint32_t headDim = 1;
    if (dims == 4) {
        batch = originShape.GetDim(0);
        heads = originShape.GetDim(1);
        seqLen = originShape.GetDim(2);
        headDim = originShape.GetDim(3);
    } else if (dims == 3) {
        // collapsed BH case
        batch = originShape.GetDim(0); // actually BH
        heads = 1;
        seqLen = originShape.GetDim(1);
        headDim = originShape.GetDim(2);
    } else {
        // Fallback to size-based split
        uint64_t size = originShape.GetShapeSize();
        headDim = 1;
        seqLen = 1;
        heads = 1;
        batch = static_cast<uint32_t>(size);
    }

    // Choose blockDim along (batch * heads) dimension
    uint64_t bh = static_cast<uint64_t>(batch) * static_cast<uint64_t>(heads);
    uint32_t blockDim = static_cast<uint32_t>(bh < 16 ? bh : 16); // cap blocks

    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_seqLen(seqLen);
    tiling.set_numHeads(heads);
    tiling.set_headDim(headDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelGQA {
public:
    __aicore__ inline KernelGQA() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR z,
                                uint32_t batch, uint32_t seqLen, uint32_t numHeads, uint32_t headDim)
    {
        this->batch = batch;
        this->seqLen = seqLen;
        this->numHeads = numHeads;
        this->headDim = headDim;

        // Each block handles one (b,h) slice. If input is [BH, L, Hd], we treat batch=BH and numHeads=1
        uint32_t bhTotal = batch * numHeads;
        uint32_t bhIndex = AscendC::GetBlockIdx();
        if (bhIndex >= bhTotal) {
            return;
        }

        // Slice base offset: assume input layout is [BH, L, Hd] contiguous
        uint64_t sliceSize = static_cast<uint64_t>(seqLen) * static_cast<uint64_t>(headDim);
        qGm.SetGlobalBuffer((__gm__ float*)q + sliceSize * bhIndex, sliceSize);
        kGm.SetGlobalBuffer((__gm__ float*)k + sliceSize * bhIndex, sliceSize);
        vGm.SetGlobalBuffer((__gm__ float*)v + sliceSize * bhIndex, sliceSize);
        zGm.SetGlobalBuffer((__gm__ float*)z + sliceSize * bhIndex, sliceSize);

        // Initialize simple buffers for row-wise operations
        pipe.InitBuffer(bufQ, 1, headDim * sizeof(float));
        pipe.InitBuffer(bufK, 1, headDim * sizeof(float));
        pipe.InitBuffer(bufV, 1, headDim * sizeof(float));
        pipe.InitBuffer(bufZ, 1, headDim * sizeof(float));
        pipe.InitBuffer(bufScores, 1, seqLen * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // NOTE: This is a simplified, illustrative kernel. For brevity, it demonstrates
        // the structure and memory movement. Replace with optimized vectorized math as needed.

        float invSqrtHd = 1.0f / AscendC::sqrtf(static_cast<float>(headDim));

        for (uint32_t l = 0; l < seqLen; ++l) {
            // Load Q[l, :]
            AscendC::LocalTensor<float> qRow = bufQ.AllocTensor<float>();
            AscendC::DataCopy(qRow, qGm[l * headDim], headDim);

            // First pass: compute raw scores = dot(Q[l], K[m]) * invSqrtHd
            AscendC::LocalTensor<float> scores = bufScores.AllocTensor<float>();
            // Initialize scores to zero
            for (uint32_t m = 0; m < seqLen; ++m) {
                scores.SetValue(m, 0.0f);
            }

            float maxScore = -1e30f;
            for (uint32_t m = 0; m < seqLen; ++m) {
                AscendC::LocalTensor<float> kRow = bufK.AllocTensor<float>();
                AscendC::DataCopy(kRow, kGm[m * headDim], headDim);
                float dot = 0.0f;
                for (uint32_t d = 0; d < headDim; ++d) {
                    dot += qRow.GetValue(d) * kRow.GetValue(d);
                }
                dot *= invSqrtHd;
                scores.SetValue(m, dot);
                if (dot > maxScore) {
                    maxScore = dot;
                }
                bufK.FreeTensor(kRow);
            }

            // Softmax over m dimension: stable softmax using max subtraction
            float sumExp = 0.0f;
            for (uint32_t m = 0; m < seqLen; ++m) {
                float val = scores.GetValue(m) - maxScore;
                float e = AscendC::expf(val);
                scores.SetValue(m, e);
                sumExp += e;
            }
            float invSumExp = 1.0f / sumExp;
            for (uint32_t m = 0; m < seqLen; ++m) {
                scores.SetValue(m, scores.GetValue(m) * invSumExp);
            }

            // Weighted sum: Z[l, :] = sum_m scores[m] * V[m, :]
            AscendC::LocalTensor<float> zRow = bufZ.AllocTensor<float>();
            for (uint32_t d = 0; d < headDim; ++d) {
                zRow.SetValue(d, 0.0f);
            }
            for (uint32_t m = 0; m < seqLen; ++m) {
                float w = scores.GetValue(m);
                AscendC::LocalTensor<float> vRow = bufV.AllocTensor<float>();
                AscendC::DataCopy(vRow, vGm[m * headDim], headDim);
                for (uint32_t d = 0; d < headDim; ++d) {
                    float acc = zRow.GetValue(d) + w * vRow.GetValue(d);
                    zRow.SetValue(d, acc);
                }
                bufV.FreeTensor(vRow);
            }

            // Store Z[l, :]
            AscendC::DataCopy(zGm[l * headDim], zRow, headDim);

            // Free buffers
            bufQ.FreeTensor(qRow);
            bufZ.FreeTensor(zRow);
            bufScores.FreeTensor(scores);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  bufQ, bufK, bufV;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> bufZ;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  bufScores;

    AscendC::GlobalTensor<float> qGm;
    AscendC::GlobalTensor<float> kGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t batch;
    uint32_t seqLen;
    uint32_t numHeads;
    uint32_t headDim;
};

extern "C" __global__ __aicore__ void group_query_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGQA op;
    op.Init(q, k, v, z,
            tiling_data.batch,
            tiling_data.seqLen,
            tiling_data.numHeads,
            tiling_data.headDim);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import math
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Optimized Grouped-Query Attention using custom AscendC kernel for the attention core.
    Linear projections remain in PyTorch; attention (Q,K,V -> output) is offloaded.
    """
    def __init__(self, d_model=1024, num_heads=16, num_kv_heads=4):
        super().__init__()
        assert num_heads % num_kv_heads == 0, "num_heads must be divisible by num_kv_heads"
        self.num_heads = num_heads
        self.num_kv_heads = num_kv_heads
        self.group_size = num_heads // num_kv_heads

        self.q_proj = nn.Linear(d_model, d_model)
        self.k_proj = nn.Linear(d_model, d_model * num_kv_heads // num_heads)
        self.v_proj = nn.Linear(d_model, d_model * num_kv_heads // num_heads)
        self.out_proj = nn.Linear(d_model, d_model)

    def forward(self, x):
        B, L, D = x.shape
        H, H_kv = self.num_heads, self.num_kv_heads
        head_dim = D // H

        # projections
        Q = self.q_proj(x).view(B, L, H, head_dim)            # [B, L, H, Hd]
        K = self.k_proj(x).view(B, L, H_kv, head_dim)         # [B, L, H_kv, Hd]
        V = self.v_proj(x).view(B, L, H_kv, head_dim)         # [B, L, H_kv, Hd]

        # Expand K/V to match query groups
        if self.group_size > 1:
            K = K.repeat_interleave(self.group_size, dim=2)   # [B, L, H, Hd]
            V = V.repeat_interleave(self.group_size, dim=2)   # [B, L, H, Hd]

        # Layout transform to make [BH, L, Hd] contiguous per (b,h) slice for kernel
        Q_ = Q.permute(0, 2, 1, 3).contiguous()               # [B, H, L, Hd]
        K_ = K.permute(0, 2, 1, 3).contiguous()               # [B, H, L, Hd]
        V_ = V.permute(0, 2, 1, 3).contiguous()               # [B, H, L, Hd]

        # Collapse batch-head for kernel convenience
        Q_flat = Q_.reshape(B * H, L, head_dim)               # [BH, L, Hd]
        K_flat = K_.reshape(B * H, L, head_dim)               # [BH, L, Hd]
        V_flat = V_.reshape(B * H, L, head_dim)               # [BH, L, Hd]

        # Custom attention kernel: returns [BH, L, Hd]
        out_flat = custom_ops_lib.group_query_attention_custom(Q_flat, K_flat, V_flat)

        # Restore to [B, L, H, Hd] then to [B, L, D]
        out = out_flat.view(B, H, L, head_dim).permute(0, 2, 1, 3).contiguous().view(B, L, D)
        return self.out_proj(out)
'''
