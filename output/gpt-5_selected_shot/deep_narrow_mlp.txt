#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelDeepNarrowMlp {
public:
    __aicore__ inline KernelDeepNarrowMlp() {}
    __aicore__ inline void Init(
        GM_ADDR x,
        GM_ADDR w1, GM_ADDR b1,
        GM_ADDR w2, GM_ADDR b2,
        GM_ADDR w3, GM_ADDR b3,
        GM_ADDR w4, GM_ADDR b4,
        GM_ADDR w5, GM_ADDR b5,
        GM_ADDR w6, GM_ADDR b6,
        GM_ADDR w7, GM_ADDR b7,
        GM_ADDR w8, GM_ADDR b8,
        GM_ADDR w9, GM_ADDR b9,
        GM_ADDR w10, GM_ADDR b10,
        GM_ADDR w11, GM_ADDR b11,
        GM_ADDR w12, GM_ADDR b12,
        GM_ADDR w13, GM_ADDR b13,
        GM_ADDR w14, GM_ADDR b14,
        GM_ADDR w15, GM_ADDR b15,
        GM_ADDR w16, GM_ADDR b16,
        GM_ADDR w17, GM_ADDR b17,
        GM_ADDR y,
        uint32_t totalBatch, uint32_t inputSize, uint32_t hiddenSize, uint32_t numHiddenLayers, uint32_t outputSize, uint32_t tileNum)
    {
        this->batch = totalBatch;
        this->inputSize = inputSize;
        this->hiddenSize = hiddenSize;
        this->numHiddenLayers = numHiddenLayers;
        this->outputSize = outputSize;
        this->tileNum = tileNum;

        this->rowsPerBlock = batch / AscendC::GetBlockNum();
        if (this->rowsPerBlock == 0) { this->rowsPerBlock = batch; }
        this->tileRows = (this->rowsPerBlock / tileNum / BUFFER_NUM);
        if (this->tileRows == 0) { this->tileRows = 1; }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + AscendC::GetBlockIdx() * this->rowsPerBlock * this->inputSize, this->rowsPerBlock * this->inputSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_X*)y + AscendC::GetBlockIdx() * this->rowsPerBlock * this->outputSize, this->rowsPerBlock * this->outputSize);

        w1Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w1, this->inputSize * this->hiddenSize);
        b1Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b1, this->hiddenSize);
        w2Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w2, this->hiddenSize * this->hiddenSize);
        b2Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b2, this->hiddenSize);
        w3Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w3, this->hiddenSize * this->hiddenSize);
        b3Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b3, this->hiddenSize);
        w4Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w4, this->hiddenSize * this->hiddenSize);
        b4Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b4, this->hiddenSize);
        w5Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w5, this->hiddenSize * this->hiddenSize);
        b5Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b5, this->hiddenSize);
        w6Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w6, this->hiddenSize * this->hiddenSize);
        b6Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b6, this->hiddenSize);
        w7Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w7, this->hiddenSize * this->hiddenSize);
        b7Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b7, this->hiddenSize);
        w8Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w8, this->hiddenSize * this->hiddenSize);
        b8Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b8, this->hiddenSize);
        w9Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w9, this->hiddenSize * this->hiddenSize);
        b9Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b9, this->hiddenSize);
        w10Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w10, this->hiddenSize * this->hiddenSize);
        b10Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b10, this->hiddenSize);
        w11Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w11, this->hiddenSize * this->hiddenSize);
        b11Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b11, this->hiddenSize);
        w12Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w12, this->hiddenSize * this->hiddenSize);
        b12Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b12, this->hiddenSize);
        w13Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w13, this->hiddenSize * this->hiddenSize);
        b13Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b13, this->hiddenSize);
        w14Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w14, this->hiddenSize * this->hiddenSize);
        b14Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b14, this->hiddenSize);
        w15Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w15, this->hiddenSize * this->hiddenSize);
        b15Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b15, this->hiddenSize);
        w16Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w16, this->hiddenSize * this->hiddenSize);
        b16Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b16, this->hiddenSize);
        w17Gm.SetGlobalBuffer((__gm__ DTYPE_W*)w17, this->hiddenSize * this->outputSize);
        b17Gm.SetGlobalBuffer((__gm__ DTYPE_B*)b17, this->outputSize);

        // local buffers for vectors and biases
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->inputSize * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, AscendC::Max(this->hiddenSize, this->outputSize) * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, AscendC::Max(this->hiddenSize, this->outputSize) * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        uint32_t rowsTotal = this->rowsPerBlock;
        for (uint32_t r = 0; r < rowsTotal; ++r) {
            ComputeRow(r);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> curLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(curLocal, xGm[rowIdx * this->inputSize], this->inputSize);

        // Layer 1: inputSize -> hiddenSize with ReLU
        AscendC::LocalTensor<DTYPE_B> bLocal = inQueueY.AllocTensor<DTYPE_B>();
        AscendC::DataCopy(bLocal, b1Gm, this->hiddenSize);
        AscendC::LocalTensor<DTYPE_X> nextLocal = outQueueZ.AllocTensor<DTYPE_X>();
        for (uint32_t j = 0; j < this->hiddenSize; ++j) {
            DTYPE_X sum = (DTYPE_X)bLocal.GetValue(j);
            for (uint32_t k = 0; k < this->inputSize; ++k) {
                sum += curLocal.GetValue(k) * w1Gm[k * this->hiddenSize + j];
            }
            if (sum < (DTYPE_X)0) sum = (DTYPE_X)0;
            nextLocal.SetValue(j, sum);
        }
        inQueueY.FreeTensor(bLocal);
        inQueueX.FreeTensor(curLocal);

        // Hidden layers 2..16 with ReLU
        for (uint32_t l = 2; l <= this->numHiddenLayers; ++l) {
            AscendC::LocalTensor<DTYPE_X> curHidden = nextLocal;
            AscendC::LocalTensor<DTYPE_X> outHidden = outQueueZ.AllocTensor<DTYPE_X>();

            // select w and b
            __gm__ DTYPE_W* wGmPtr = nullptr;
            __gm__ DTYPE_B* bGmPtr = nullptr;
            if (l == 2) { wGmPtr = w2Gm; bGmPtr = b2Gm; }
            if (l == 3) { wGmPtr = w3Gm; bGmPtr = b3Gm; }
            if (l == 4) { wGmPtr = w4Gm; bGmPtr = b4Gm; }
            if (l == 5) { wGmPtr = w5Gm; bGmPtr = b5Gm; }
            if (l == 6) { wGmPtr = w6Gm; bGmPtr = b6Gm; }
            if (l == 7) { wGmPtr = w7Gm; bGmPtr = b7Gm; }
            if (l == 8) { wGmPtr = w8Gm; bGmPtr = b8Gm; }
            if (l == 9) { wGmPtr = w9Gm; bGmPtr = b9Gm; }
            if (l == 10) { wGmPtr = w10Gm; bGmPtr = b10Gm; }
            if (l == 11) { wGmPtr = w11Gm; bGmPtr = b11Gm; }
            if (l == 12) { wGmPtr = w12Gm; bGmPtr = b12Gm; }
            if (l == 13) { wGmPtr = w13Gm; bGmPtr = b13Gm; }
            if (l == 14) { wGmPtr = w14Gm; bGmPtr = b14Gm; }
            if (l == 15) { wGmPtr = w15Gm; bGmPtr = b15Gm; }
            if (l == 16) { wGmPtr = w16Gm; bGmPtr = b16Gm; }

            AscendC::LocalTensor<DTYPE_B> bLocalH = inQueueY.AllocTensor<DTYPE_B>();
            AscendC::DataCopy(bLocalH, bGmPtr, this->hiddenSize);

            for (uint32_t j = 0; j < this->hiddenSize; ++j) {
                DTYPE_X sum = (DTYPE_X)bLocalH.GetValue(j);
                for (uint32_t k = 0; k < this->hiddenSize; ++k) {
                    sum += curHidden.GetValue(k) * wGmPtr[k * this->hiddenSize + j];
                }
                if (sum < (DTYPE_X)0) sum = (DTYPE_X)0;
                outHidden.SetValue(j, sum);
            }

            inQueueY.FreeTensor(bLocalH);
            outQueueZ.FreeTensor(curHidden);
            nextLocal = outHidden;
        }

        // Final layer: hiddenSize -> outputSize (no ReLU)
        AscendC::LocalTensor<DTYPE_X> outFinal = outQueueZ.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_B> bLocalOut = inQueueY.AllocTensor<DTYPE_B>();
        AscendC::DataCopy(bLocalOut, b17Gm, this->outputSize);
        for (uint32_t j = 0; j < this->outputSize; ++j) {
            DTYPE_X sum = (DTYPE_X)bLocalOut.GetValue(j);
            for (uint32_t k = 0; k < this->hiddenSize; ++k) {
                sum += nextLocal.GetValue(k) * w17Gm[k * this->outputSize + j];
            }
            outFinal.SetValue(j, sum);
        }
        inQueueY.FreeTensor(bLocalOut);
        outQueueZ.FreeTensor(nextLocal);

        AscendC::DataCopy(yGm[rowIdx * this->outputSize], outFinal, this->outputSize);
        outQueueZ.FreeTensor(outFinal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> yGm;

    AscendC::GlobalTensor<DTYPE_W> w1Gm; AscendC::GlobalTensor<DTYPE_B> b1Gm;
    AscendC::GlobalTensor<DTYPE_W> w2Gm; AscendC::GlobalTensor<DTYPE_B> b2Gm;
    AscendC::GlobalTensor<DTYPE_W> w3Gm; AscendC::GlobalTensor<DTYPE_B> b3Gm;
    AscendC::GlobalTensor<DTYPE_W> w4Gm; AscendC::GlobalTensor<DTYPE_B> b4Gm;
    AscendC::GlobalTensor<DTYPE_W> w5Gm; AscendC::GlobalTensor<DTYPE_B> b5Gm;
    AscendC::GlobalTensor<DTYPE_W> w6Gm; AscendC::GlobalTensor<DTYPE_B> b6Gm;
    AscendC::GlobalTensor<DTYPE_W> w7Gm; AscendC::GlobalTensor<DTYPE_B> b7Gm;
    AscendC::GlobalTensor<DTYPE_W> w8Gm; AscendC::GlobalTensor<DTYPE_B> b8Gm;
    AscendC::GlobalTensor<DTYPE_W> w9Gm; AscendC::GlobalTensor<DTYPE_B> b9Gm;
    AscendC::GlobalTensor<DTYPE_W> w10Gm; AscendC::GlobalTensor<DTYPE_B> b10Gm;
    AscendC::GlobalTensor<DTYPE_W> w11Gm; AscendC::GlobalTensor<DTYPE_B> b11Gm;
    AscendC::GlobalTensor<DTYPE_W> w12Gm; AscendC::GlobalTensor<DTYPE_B> b12Gm;
    AscendC::GlobalTensor<DTYPE_W> w13Gm; AscendC::GlobalTensor<DTYPE_B> b13Gm;
    AscendC::GlobalTensor<DTYPE_W> w14Gm; AscendC::GlobalTensor<DTYPE_B> b14Gm;
    AscendC::GlobalTensor<DTYPE_W> w15Gm; AscendC::GlobalTensor<DTYPE_B> b15Gm;
    AscendC::GlobalTensor<DTYPE_W> w16Gm; AscendC::GlobalTensor<DTYPE_B> b16Gm;
    AscendC::GlobalTensor<DTYPE_W> w17Gm; AscendC::GlobalTensor<DTYPE_B> b17Gm;

    uint32_t batch;
