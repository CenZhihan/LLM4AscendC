kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMambaReturnFinalState {
public:
    __aicore__ inline KernelMambaReturnFinalState() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t rows, uint32_t channels, uint32_t cols, uint32_t tileCols)
    {
        this->rows = rows;
        this->channels = channels;
        this->cols = cols;
        this->tileCols = tileCols;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)rows * channels * cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint64_t)rows * cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileCols * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // Distribute rows across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (rows + blockNum - 1) / blockNum;
        uint32_t rowStart = rowsPerBlock * blockIdx;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > rows) rowEnd = rows;

        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            for (uint32_t c0 = 0; c0 < cols; c0 += tileCols) {
                uint32_t currTile = (c0 + tileCols <= cols) ? tileCols : (cols - c0);

                // Accumulator buffer
                AscendC::LocalTensor<DTYPE_Z> accLocal = outQueueY.AllocTensor<DTYPE_Z>();

                // Initialize accumulator with first channel
                {
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                    uint64_t offset = ((uint64_t)r * channels * cols) + (uint64_t)0 * cols + c0;
                    AscendC::DataCopy(xLocal, xGm[offset], currTile);
                    // Cast or copy to accumulator
                    AscendC::DataCopy(accLocal, xLocal, currTile);
                    inQueueX.EnQue(xLocal);
                    inQueueX.FreeTensor(xLocal);
                }

                // Accumulate remaining channels
                for (uint32_t ch = 1; ch < channels; ++ch) {
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                    uint64_t offset = ((uint64_t)r * channels * cols) + (uint64_t)ch * cols + c0;
                    AscendC::DataCopy(xLocal, xGm[offset], currTile);
                    inQueueX.EnQue(xLocal);

                    AscendC::LocalTensor<DTYPE_X> xLocalDeq = inQueueX.DeQue<DTYPE_X>();
                    // accLocal = accLocal + xLocalDeq
                    AscendC::Add(accLocal, accLocal, xLocalDeq, currTile);
                    inQueueX.FreeTensor(xLocalDeq);
                }

                // Write out
                uint64_t yoff = (uint64_t)r * cols + c0;
                AscendC::DataCopy(yGm[yoff], accLocal, currTile);
                outQueueY.EnQue<DTYPE_Z>(accLocal);
                outQueueY.FreeTensor(accLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t rows;
    uint32_t channels;
    uint32_t cols;
    uint32_t tileCols;
};

extern "C" __global__ __aicore__ void mamba_return_final_state_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMambaReturnFinalState op;
    op.Init(x, y, tiling_data.rows, tiling_data.channels, tiling_data.cols, tiling_data.tileCols);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
from einops import rearrange

class ModelNew(torch.nn.Module):
    def __init__(self, batch_size, seq_length, n_heads, d_head, d_state, block_len=64):
        super(ModelNew, self).__init__()
        assert seq_length % block_len == 0, "Sequence length must be divisible by block length"
        self.batch_size = batch_size
        self.seq_length = seq_length
        self.n_heads = n_heads
        self.d_head = d_head
        self.d_state = d_state
        self.block_len = block_len

        self.A = torch.nn.Parameter(torch.randn(batch_size, seq_length, n_heads))
        self.B = torch.nn.Parameter(torch.randn(batch_size, seq_length, n_heads, d_state))
        self.C = torch.nn.Parameter(torch.randn(batch_size, seq_length, n_heads, d_state))

    def segsum(self, x):
        T = x.size(-1)
        x_cumsum = torch.cumsum(x, dim=-1)
        x_segsum = x_cumsum[..., :, None] - x_cumsum[..., None, :]
        mask = torch.tril(torch.ones(T, T, device=x.device, dtype=bool), diagonal=0)
        x_segsum = x_segsum.masked_fill(~mask, -torch.inf)
        return x_segsum

    def forward(self, X, initial_states=None):
        # Rearrange into blocks/chunks
        X_blocks, A_blocks, B_blocks, C_blocks = [
            rearrange(x, "b (c l) ... -> b c l ...", l=self.block_len)
            for x in (X, self.A, self.B, self.C)
        ]

        A_blocks = rearrange(A_blocks, "b c l h -> b h c l")
        A_cumsum = torch.cumsum(A_blocks, dim=-1)

        # 1. Compute diagonal block outputs
        L = torch.exp(self.segsum(A_blocks))
        Y_diag = torch.einsum("bclhn,bcshn,bhcls,bcshp->bclhp",
                              C_blocks, B_blocks, L, X_blocks)

        # 2. Compute intra-chunk states
        decay_states = torch.exp((A_cumsum[:, :, :, -1:] - A_cumsum))
        states = torch.einsum("bclhn,bhcl,bclhp->bchpn",
                              B_blocks, decay_states, X_blocks)

        # 3. Compute inter-chunk recurrence
        if initial_states is None:
            initial_states = torch.zeros_like(states[:, :1])
        states = torch.cat([initial_states, states], dim=1)  # b (c+1) h p n

        decay_chunk = torch.exp(self.segsum(torch.nn.functional.pad(A_cumsum[:, :, :, -1], (1, 0))))
        # Take the last z-slice of decay_chunk for weighted sum over c
        decay_last = decay_chunk[:, :, -1, :]  # b h (c+1)

        # Prepare tensors to call custom kernel:
        # We compute weighted states per (b,h) over c to get final state (p,n)
        # states: b (c+1) h p n -> b h (c+1) p n
        states_bh_cpn = states.permute(0, 2, 1, 3, 4).contiguous()
        # Broadcast multiply by decay_last over c
        weighted_states = states_bh_cpn * decay_last[:, :, :, None, None]
        b, h, c_len, p, n = weighted_states.shape

        # Reshape to [rows=B*H, channels=C, cols=P*N]
        rows = b * h
        cols = p * n
        x3d = weighted_states.reshape(rows, c_len, cols).contiguous()

        # Call custom NPU kernel that reduces over channel (dim=1) to produce [rows, cols]
        out2d = custom_ops_lib.mamba_return_final_state_custom(x3d)

        # Reshape back to [b, h, p, n] and return
        final_state = out2d.reshape(b, h, p, n).contiguous()
        return final_state
'''
