host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t MinU32(uint32_t a, uint32_t b) { return a < b ? a : b; }
}

// UB model (conservative):
//   inQueueX: 2 * tileRows * cols
//   outQueueY: 2 * tileRows * cols
//   tmpSumQueue: 1 * tileRows * cols   (over-provisioned for simplicity)
//   tmpMaxQueue: 1 * tileRows * cols   (over-provisioned for simplicity)
// Total floats = 6 * tileRows * cols => bytes = 24 * tileRows * cols
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative UB
static constexpr uint32_t MAX_BLOCKS = 24;
static constexpr uint32_t MIN_BLOCKS = 8;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> dims = x_shape->GetOriginShape().GetDims();

    // Fallback for non-2D: flatten leading dims into rows, last dim as cols
    uint32_t rows = 1;
    uint32_t cols = 1;
    if (!dims.empty()) {
        cols = static_cast<uint32_t>(dims.back());
        int64_t lead = 1;
        for (size_t i = 0; i + 1 < dims.size(); ++i) { lead *= dims[i]; }
        rows = static_cast<uint32_t>(lead);
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimAttr != nullptr) ? *dimAttr : 1;

    // Only support softmax along last dimension for now (dim = -1 or dim = last axis)
    // For the given model, this is dim=1 on 2D input.
    if (!(dim == -1 || (dims.size() >= 1 && dim == static_cast<int64_t>(dims.size() - 1)))) {
        // Best-effort: if 2D and dim==1, treat as last-dim
        if (!(dims.size() == 2 && dim == 1)) {
            // Unsupported dim; still proceed assuming last dim to avoid hard fail.
            dim = (dims.empty() ? 0 : (int64_t)dims.size() - 1);
        }
    }

    // Derive tileRows from UB budget.
    // bytes_per_tile = 24 * tileRows * cols
    uint64_t denom = 24ull * cols;
    uint32_t tileRows = 1;
    if (denom > 0) {
        tileRows = static_cast<uint32_t>(UB_BUDGET_BYTES / denom);
        if (tileRows == 0) tileRows = 1;
    }
    if (tileRows > rows) tileRows = rows;

    uint32_t numTiles = CeilDiv(rows, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace required
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"
#include "softmax_high_perf.h"  // Ensure SoftMax high-level API is available in include path

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t tileRows, int64_t dim)
    {
        rows_ = rows;
        cols_ = cols;
        tileRows_ = tileRows > 0 ? tileRows : 1;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)rows_ * cols_);

        const uint32_t tileElems = tileRows_ * cols_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileElems * sizeof(float));
        // Over-provisioned temporaries for safety across implementations
        pipe_.InitBuffer(sumQueue_, 1, tileElems * sizeof(float));
        pipe_.InitBuffer(maxQueue_, 1, tileElems * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) { return; }

        // We support softmax along last dimension only; for typical 2D input this is dim=1.
        // When dim is different, treat as last-dim softmax in the kernel.
        const uint32_t tileElems = tileRows_ * cols_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride by tiles of rows
        const uint32_t strideTiles = blockNum;
        uint32_t tileIdx = blockIdx;

        // Prefetch first tile if exists
        if (tileIdx * tileRows_ >= rows_) { return; }
        uint32_t curRowStart = tileIdx * tileRows_;
        uint32_t curRows = rows_ - curRowStart;
        if (curRows > tileRows_) curRows = tileRows_;
        CopyIn(curRowStart, curRows);

        while (true) {
            uint32_t nextTileIdx = tileIdx + strideTiles;
            uint32_t nextRowStart = nextTileIdx * tileRows_;
            uint32_t nextRows = 0;
            if (nextRowStart < rows_) {
                nextRows = rows_ - nextRowStart;
                if (nextRows > tileRows_) nextRows = tileRows_;
                CopyIn(nextRowStart, nextRows);
            }

            Compute(curRows);

            CopyOut(curRowStart, curRows);

            if (nextRows == 0) break;
            tileIdx = nextTileIdx;
            curRowStart = nextRowStart;
            curRows = nextRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowStart, uint32_t numRows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        const uint32_t len = numRows * cols_;
        const uint64_t gmOffset = (uint64_t)rowStart * cols_;
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t numRows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = maxQueue_.AllocTensor<float>();

        // For each row in the tile, apply high-level SoftMax along last axis
        for (uint32_t r = 0; r < numRows; ++r) {
            const uint32_t rowOff = r * cols_;
            AscendC::LocalTensor<float> xRow = xLocal[rowOff];
            AscendC::LocalTensor<float> yRow = yLocal[rowOff];

            // Construct tiling for a single row [1, cols_]
            AscendC::SoftMaxTiling smTiling;
            // Use default shape info (ND last-axis reduction). Many implementations infer from cols.
            // If required, user can fill smTiling with cols via helper.
            smTiling.SetN(cols_);  // set last-axis length
            smTiling.SetM(1);      // single row

            // Run SoftMax. Using overload with explicit sum/max temporaries.
            AscendC::SoftMax<float>(yRow, sumLocal, maxLocal, xRow, smTiling);
        }

        maxQueue_.FreeTensor(maxLocal);
        sumQueue_.FreeTensor(sumLocal);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowStart, uint32_t numRows)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        const uint32_t len = numRows * cols_;
        const uint64_t gmOffset = (uint64_t)rowStart * cols_;
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maxQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileRows_ {1};
    int64_t dim_ {1};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.rows, tiling_data.cols,
            tiling_data.tileRows, tiling_data.dim);
    op.Process();
}
"""
