host_operator_src="""

#include "gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model (approx):
//   inQueueX:  2 * tileLen * 4
//   outQueueY: 2 * tileLen * 4
//   tmpFloat:  1 * tileLen * 4
// Total bytes ≈ 5 * tileLen * 4 = 20 * tileLen
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // prefer aligned DMA
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap per-iter length

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GeluCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Tile length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    if (maxTile == 0u) maxTile = 1u;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    if (maxTile >= ALIGN_ELEMS) maxTile = AlignDown(maxTile, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, (tileLen >= ALIGN_ELEMS) ? ALIGN_ELEMS : 1u);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }
    if (tileLen == 0) tileLen = 1;

    // Block dim selection
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_tmpErfBytesPerTile(0); // not used in kernel

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class GeluCustom : public OpDef {
public:
    explicit GeluCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(GeluCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float INV_SQRT2PI = 0.7978845608028654f; // sqrt(2/pi)
constexpr float TANH_PADE_A = 27.0f;               // Pade coeffs for tanh approx
constexpr float TANH_PADE_B = 9.0f;                // tanh(z) ≈ z*(A + z^2)/(A + B*z^2)

class KernelGeluCustom {
public:
    __aicore__ inline KernelGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t /*tmpErfBytesPerTile*/)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpFloatQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tFloat = tmpFloatQueue_.AllocTensor<float>();

        // Approximate GELU using tanh-based formula with Pade approx for tanh:
        // z = sqrt(2/pi) * (x + 0.044715 * x^3)
        // tanh(z) ≈ z * (A + z^2) / (A + B * z^2), A=27, B=9
        // y = 0.5 * x * (1 + tanh(z))

        // tFloat = x^2
        AscendC::Mul(tFloat, xLocal, xLocal, len);
        // tFloat = x^3
        AscendC::Mul(tFloat, tFloat, xLocal, len);
        // tFloat = 0.044715 * x^3
        AscendC::Muls(tFloat, tFloat, 0.044715f, len);
        // tFloat = x + 0.044715 * x^3
        AscendC::Add(tFloat, tFloat, xLocal, len);
        // tFloat = z = sqrt(2/pi) * (x + 0.044715 * x^3)
        AscendC::Muls(tFloat, tFloat, INV_SQRT2PI, len);

        // yLocal = z^2
        AscendC::Mul(yLocal, tFloat, tFloat, len);
        // yLocal = (A + z^2)
        AscendC::Adds(yLocal, yLocal, TANH_PADE_A, len);
        // tFloat = numerator = z * (A + z^2)
        AscendC::Mul(tFloat, tFloat, yLocal, len);

        // yLocal = z^2
        AscendC::Mul(yLocal, yLocal, yLocal, 0); // dummy to avoid warnings, will overwrite next
        AscendC::Mul(yLocal, tFloat, tFloat, 0); // no-op when len==0
        // recompute yLocal = z^2
        AscendC::Mul(yLocal, outQueueY_.GetLocalBuffer<float>(), outQueueY_.GetLocalBuffer<float>(), 0); // safeguard
        AscendC::Mul(yLocal, tFloat, tFloat, 0); // safeguard

        // Recompute z^2 into yLocal properly (tFloat currently holds numerator)
        // We still have z in numerator's left operand lost; recalc z^2 by:
        // Get z back by dividing numerator by (A + z^2) is expensive; easier: recalc z^2:
        // We can recompute z into yLocal via following steps:
        // Use tFloat as temp holder: restore z by dividing numerator by (A + z^2),
        // but we already overwrote (A + z^2). To avoid this, compute denom directly from z:
        // Instead, compute denom via z reusing the fact that we still have z in numerator is not accessible.
        // Simpler approach: recompute z2 = (sqrt(2/pi) * (x + 0.044715 x^3))^2.
        // Recompute (x + 0.044715 x^3) into yLocal:

        // yLocal = x^2
        AscendC::Mul(yLocal, xLocal, xLocal, len);
        // yLocal = x^3
        AscendC::Mul(yLocal, yLocal, xLocal, len);
        // yLocal = 0.044715 * x^3
        AscendC::Muls(yLocal, yLocal, 0.044715f, len);
        // yLocal = x + 0.044715 * x^3
        AscendC::Add(yLocal, yLocal, xLocal, len);
        // yLocal = z^2 = (sqrt(2/pi) * yLocal)^2 = (2/pi) * yLocal^2
        AscendC::Mul(yLocal, yLocal, yLocal, len);
        AscendC::Muls(yLocal, yLocal, (INV_SQRT2PI * INV_SQRT2PI), len);

        // denom = A + B * z^2
        AscendC::Muls(yLocal, yLocal, TANH_PADE_B, len);
        AscendC::Adds(yLocal, yLocal, TANH_PADE_A, len);

        // tFloat currently holds numerator = z * (A + z^2)
        // tFloat = tanh_approx(z) = numerator / denom
        AscendC::Div(tFloat, tFloat, yLocal, len);

        // tFloat = 1 + tanh_approx(z)
        AscendC::Adds(tFloat, tFloat, 1.0f, len);
        // yLocal = 0.5 * x
        AscendC::Muls(yLocal, xLocal, 0.5f, len);
        // yLocal = y = (0.5 * x) * (1 + tanh_approx(z))
        AscendC::Mul(yLocal, yLocal, tFloat, len);

        tmpFloatQueue_.FreeTensor(tFloat);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpFloatQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void gelu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGeluCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.tmpErfBytesPerTile);
    op.Process();
}
