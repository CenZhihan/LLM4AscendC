host_operator_src="""

#include "min_gpt_new_gelu_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/utils/type_utils.h"
#include <vector>

// AscendC high-level API for tmp-size query
#include "ascendc.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency
static constexpr uint32_t BUFFER_NUM = 2;
static constexpr uint32_t TYPE_SIZE = 4; // float32

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MinGptNewGeluCustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // UB consumption model (bytes): inQueueX: 2*tile, outQueueY: 2*tile, tmpWork: 1*tile
    // => baseBytes = 5 * tileLen * TYPE_SIZE
    // plus Tanh tmp buffer in bytes for this tile
    uint32_t tileLen = 0;
    uint32_t tmpMinBytes = 0;

    // initial guess ignoring tanh tmp
    uint32_t maxTileByBase = UB_BUDGET_BYTES / (5u * TYPE_SIZE);
    maxTileByBase = AlignDown(maxTileByBase, ALIGN_ELEMS);
    if (maxTileByBase == 0) maxTileByBase = ALIGN_ELEMS;
    if (maxTileByBase > HARD_CAP_ELEMS) maxTileByBase = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    tileLen = (totalLength < maxTileByBase) ? totalLength : maxTileByBase;
    if (tileLen >= ALIGN_ELEMS) tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0 && totalLength > 0) tileLen = totalLength;

    // refine with Tanh tmp size; choose minimal tmp to save UB
    auto compute_tmp_min = [](uint32_t elems) -> uint32_t {
        std::vector<int64_t> shape_vec = { static_cast<int64_t>(elems) };
        ge::Shape shape(shape_vec);
        uint32_t maxV = 0, minV = 0;
        AscendC::GetTanhMaxMinTmpSize(shape, TYPE_SIZE, false, maxV, minV);
        return minV;
    };

    // ensure fit; iterate a couple of times for safety
    for (int i = 0; i < 3; ++i) {
        tmpMinBytes = compute_tmp_min(tileLen);
        uint64_t baseBytes = static_cast<uint64_t>(5u) * tileLen * TYPE_SIZE;
        if (baseBytes + tmpMinBytes <= UB_BUDGET_BYTES) break;
        if (UB_BUDGET_BYTES <= tmpMinBytes + 16) {
            // pathological; force tiny tile
            tileLen = (totalLength < ALIGN_ELEMS) ? totalLength : ALIGN_ELEMS;
        } else {
            uint32_t maxByFit = static_cast<uint32_t>((UB_BUDGET_BYTES - tmpMinBytes) / (5u * TYPE_SIZE));
            maxByFit = (maxByFit > 0) ? AlignDown(maxByFit, ALIGN_ELEMS) : maxByFit;
            if (maxByFit == 0) maxByFit = (totalLength < ALIGN_ELEMS) ? totalLength : ALIGN_ELEMS;
            if (maxByFit > HARD_CAP_ELEMS) maxByFit = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
            tileLen = (tileLen > maxByFit) ? maxByFit : tileLen;
            if (tileLen == 0) tileLen = (totalLength > 0) ? 1u : 0u;
        }
    }

    if (tileLen == 0 && totalLength > 0) {
        tileLen = (totalLength < ALIGN_ELEMS) ? totalLength : ALIGN_ELEMS;
        tmpMinBytes = compute_tmp_min(tileLen);
    }
    if (tileLen > totalLength) tileLen = totalLength;

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_tmpBufBytes(tmpMinBytes);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    gert::Shape *y_shape = context->GetOutputShape(0);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMinGptNewGeluCustom {
public:
    __aicore__ inline KernelMinGptNewGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t tmpBufBytes)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        tmpBufBytes_ = tmpBufBytes;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpWorkQueue_, 1, tileLen_ * sizeof(float));
        // shared tmp buffer for Tanh
        uint32_t allocBytes = tmpBufBytes_;
        if (allocBytes == 0) {
            // fallback small buffer to satisfy allocator; Tanh may not need it
            allocBytes = 64;
        }
        pipe_.InitBuffer(tmpTanhQueue_, 1, allocBytes);

        // constants
        cubicCoeff_ = 0.044715f;
        scale_ = 0.7978845608028654f; // sqrt(2/pi)
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpWorkQueue_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tanhTmp = tmpTanhQueue_.AllocTensor<uint8_t>();

        // tLocal = x * x
        AscendC::Mul(tLocal, xLocal, xLocal, len);
        // tLocal = tLocal * x => x^3
        AscendC::Mul(tLocal, tLocal, xLocal, len);
        // tLocal = 0.044715 * tLocal
        AscendC::Muls(tLocal, tLocal, cubicCoeff_, len);
        // tLocal = tLocal + x
        AscendC::Add(tLocal, tLocal, xLocal, len);
        // tLocal = scale_ * tLocal
        AscendC::Muls(tLocal, tLocal, scale_, len);

        // yLocal = tanh(tLocal) using shared tmp buffer
        AscendC::Tanh(yLocal, tLocal, tanhTmp, len);

        // yLocal = yLocal + 1.0
        AscendC::Adds(yLocal, yLocal, 1.0f, len);
        // yLocal = yLocal * x
        AscendC::Mul(yLocal, yLocal, xLocal, len);
        // yLocal = 0.5 * yLocal
        AscendC::Muls(yLocal, yLocal, 0.5f, len);

        tmpTanhQueue_.FreeTensor(tanhTmp);
        tmpWorkQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpWorkQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpTanhQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t tmpBufBytes_ {0};

    float cubicCoeff_ {0.044715f};
    float scale_ {0.7978845608028654f};
};

extern "C" __global__ __aicore__ void min_gpt_new_gelu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMinGptNewGeluCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.tmpBufBytes);
    op.Process();
}
