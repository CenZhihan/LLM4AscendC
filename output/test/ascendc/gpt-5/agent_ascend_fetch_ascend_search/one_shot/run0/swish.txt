host_operator_src="""

#include <vector>
#include <cstdint>
#include "swish_custom_tiling.h"
#include "register/op_def_registry.h"

// Sigmoid tmp size helper
#include "graph/shape.h"
// Use AscendC high-level API query for sigmoid tmp size
namespace AscendC {
void GetSigmoidMaxMinTmpSize(const ge::Shape &srcShape, const uint32_t typeSize, bool isReuseSource,
                             uint32_t &maxValue, uint32_t &minValue);
}

namespace optiling {

// UB model per core for this kernel (float):
//   inQueueX:     2 * tile
//   outQueueY:    2 * tile
//   sigLocal:     1 * tile
//   sigmoidTmp:   variable (queried via GetSigmoidMaxMinTmpSize)
// Total float bytes (without sigmoidTmp) = (2 + 2 + 1) * 4 * tile = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core UB budget
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor large DMA alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid excessive per-iter latency
static constexpr uint32_t SAFETY_MARGIN = 1024;          // small cushion

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SwishCustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Initial tile length estimate ignoring sigmoid temp
    uint32_t maxTile = (UB_BUDGET_BYTES > SAFETY_MARGIN) ? (UB_BUDGET_BYTES - SAFETY_MARGIN) / 20u : 1024u;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // Fit considering Sigmoid temporary UB requirement (choose minValue for correctness)
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0u) ? aligned : totalLength;
    }

    // Iteratively adjust tileLen to fit UB budget with Sigmoid tmp
    uint32_t chosenTmpBytes = 0;
    for (int iter = 0; iter < 8; ++iter) {
        if (tileLen == 0) { tileLen = 1; }
        std::vector<int64_t> shape_vec = { static_cast<int64_t>(tileLen) };
        ge::Shape shape(shape_vec);
        uint32_t maxV = 0, minV = 0;
        AscendC::GetSigmoidMaxMinTmpSize(shape, /*float32*/4, /*isReuseSource*/false, maxV, minV);
        // Use minimal required tmp for correctness
        const uint32_t tmpBytes = minV;
        const uint64_t bytesNoTmp = static_cast<uint64_t>(20u) * static_cast<uint64_t>(tileLen);
        const uint64_t totalBytes = bytesNoTmp + static_cast<uint64_t>(tmpBytes) + SAFETY_MARGIN;
        if (totalBytes <= UB_BUDGET_BYTES) {
            chosenTmpBytes = tmpBytes;
            break;
        }
        // Reduce tileLen and retry
        uint32_t next = AlignDown(tileLen / 2u, ALIGN_ELEMS);
        if (next == 0u && tileLen > 1u) next = 1u;  // allow very small tails
        tileLen = next;
    }

    if (tileLen == 0u) {
        tileLen = (totalLength == 0u) ? 0u : (totalLength < ALIGN_ELEMS ? totalLength : ALIGN_ELEMS);
        // Recompute tmpBytes for final fallback
        std::vector<int64_t> shape_vec = { static_cast<int64_t>(tileLen) };
        ge::Shape shape(shape_vec);
        uint32_t maxV = 0, minV = 0;
        AscendC::GetSigmoidMaxMinTmpSize(shape, 4, false, maxV, minV);
        chosenTmpBytes = minV;
    }

    // Determine blockDim based on number of tiles
    const uint32_t numTiles = (tileLen == 0u) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_sigmoidTmpBytes(chosenTmpBytes);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t INOUT_BUFFER_NUM = 2;

class KernelSwishCustom {
public:
    __aicore__ inline KernelSwishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t sigmoidTmpBytes)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        sigmoidTmpBytes_ = sigmoidTmpBytes;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, INOUT_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, INOUT_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sigLocalQueue_, 1, tileLen_ * sizeof(float));
        if (sigmoidTmpBytes_ > 0) {
            pipe_.InitBuffer(sigmoidTmpQueue_, 1, sigmoidTmpBytes_);
        }
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> sigLocal = sigLocalQueue_.AllocTensor<float>();

        if (sigmoidTmpBytes_ > 0) {
            AscendC::LocalTensor<uint8_t> tmpBuf = sigmoidTmpQueue_.AllocTensor<uint8_t>();
            AscendC::Sigmoid<float>(sigLocal, xLocal, tmpBuf, len);
            sigmoidTmpQueue_.FreeTensor(tmpBuf);
        } else {
            AscendC::Sigmoid<float>(sigLocal, xLocal, len);
        }

        AscendC::Mul(yLocal, xLocal, sigLocal, len);

        sigLocalQueue_.FreeTensor(sigLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INOUT_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, INOUT_BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sigLocalQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sigmoidTmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t sigmoidTmpBytes_ {0};
};

extern "C" __global__ __aicore__ void swish_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSwishCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.sigmoidTmpBytes);
    op.Process();
}
