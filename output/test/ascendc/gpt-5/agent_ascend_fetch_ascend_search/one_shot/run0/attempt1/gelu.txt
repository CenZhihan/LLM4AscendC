host_operator_src="""

#include "gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX:    2 * tileLen * typeSize
//   outQueueY:   2 * tileLen * typeSize
//   tmpFloat:    1 * tileLen * typeSize
//   tmpErfU8:    1 * (tileLen * typeSize * maxLiveNodeCount + extraBuf)
// Total bytes = (5 + maxLiveNodeCount) * tileLen * typeSize + extraBuf
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // prefer aligned DMA
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid long per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GeluCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Erf tmp buffer factor: size = tileLen * typeSize * maxLiveNodeCount + extraBuf
    // type is float32 => typeSize = 4
    uint32_t maxLiveNodeCount = 0;
    uint32_t extraBuf = 0;
    AscendC::GetErfTmpBufferFactorSize(4, maxLiveNodeCount, extraBuf);

    // Solve for tileLen within UB budget
    uint64_t usable = (UB_BUDGET_BYTES > extraBuf) ? (UB_BUDGET_BYTES - extraBuf) : 0u;
    uint64_t denom = static_cast<uint64_t>(4u) * static_cast<uint64_t>(5u + maxLiveNodeCount);
    uint32_t maxTile = (denom == 0 || usable == 0) ? ALIGN_ELEMS
                                                   : static_cast<uint32_t>(usable / denom);

    if (maxTile == 0u) maxTile = 1u;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    // Prefer alignment when possible
    if (maxTile >= ALIGN_ELEMS) {
        maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    }

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, (tileLen >= ALIGN_ELEMS) ? ALIGN_ELEMS : 1u);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }
    if (tileLen == 0) tileLen = 1;

    // Compute tmpErfBytesPerTile from chosen tileLen
    uint64_t tmpErfBytes = static_cast<uint64_t>(tileLen) * 4u * maxLiveNodeCount + extraBuf;
    if (tmpErfBytes > UB_BUDGET_BYTES) {
        // Fallback: shrink tileLen to fit within UB (safety guard)
        uint64_t fitUsable = (UB_BUDGET_BYTES > extraBuf) ? (UB_BUDGET_BYTES - extraBuf) : 0u;
        uint64_t fitTile = (denom == 0 || fitUsable == 0) ? 1u : (fitUsable / denom);
        if (fitTile == 0) fitTile = 1;
        tileLen = static_cast<uint32_t>(fitTile);
        if (tileLen >= ALIGN_ELEMS) tileLen = AlignDown(tileLen, ALIGN_ELEMS);
        tmpErfBytes = static_cast<uint64_t>(tileLen) * 4u * maxLiveNodeCount + extraBuf;
    }

    // Block dim selection
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_tmpErfBytesPerTile(static_cast<uint32_t>(tmpErfBytes));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class GeluCustom : public OpDef {
public:
    explicit GeluCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(GeluCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float INV_SQRT2 = 0.7071067811865475244008443621048490f;  // 1/sqrt(2)

class KernelGeluCustom {
public:
    __aicore__ inline KernelGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t tmpErfBytesPerTile)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        tmpErfBytesPerTile_ = tmpErfBytesPerTile;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpFloatQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpErfU8Queue_, 1, tmpErfBytesPerTile_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tFloat = tmpFloatQueue_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tmpBuf = tmpErfU8Queue_.AllocTensor<uint8_t>(tmpErfBytesPerTile_);

        // tFloat = x / sqrt(2)
        AscendC::Muls(tFloat, xLocal, INV_SQRT2, len);

        // tFloat = erf(tFloat), using shared tmp buffer
        AscendC::Erf(tFloat, tFloat, tmpBuf, len);

        // tFloat = 0.5 * (1 + tFloat)
        AscendC::Adds(tFloat, tFloat, 1.0f, len);
        AscendC::Muls(tFloat, tFloat, 0.5f, len);

        // y = x * tFloat
        AscendC::Mul(yLocal, xLocal, tFloat, len);

        tmpErfU8Queue_.FreeTensor(tmpBuf);
        tmpFloatQueue_.FreeTensor(tFloat);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpFloatQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpErfU8Queue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t tmpErfBytesPerTile_ {0};
};

extern "C" __global__ __aicore__ void gelu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGeluCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.tmpErfBytesPerTile);
    op.Process();
}
"""
