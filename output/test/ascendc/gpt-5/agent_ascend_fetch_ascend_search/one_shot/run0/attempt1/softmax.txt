host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

// UB model for this kernel:
// Per-core UB usage per tile:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue: 2 * tile
//   scalarQueue: small
// Total floats ~= 6 * tile => bytes ~= 24 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin = x_shape->GetOriginShape();
    const uint32_t totalLength = static_cast<uint32_t>(origin.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *axisPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t axis = (axisPtr != nullptr) ? *axisPtr : 1;

    // Guard axis with rank
    int32_t rank = static_cast<int32_t>(origin.GetDimNum());
    if (rank <= 0) rank = 1;
    if (axis < 0) axis += rank;
    if (axis < 0) axis = 0;
    if (axis >= rank) axis = rank - 1;

    const uint32_t rowLength = static_cast<uint32_t>(origin.GetDim(axis));
    const uint32_t rows = (rowLength == 0) ? 0u : (totalLength / rowLength);

    // tileLen from UB budget: bytes ≈ 24 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 24;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        const uint32_t aligned = AlignDown(rowLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : rowLength;
    }
    if (tileLen == 0 && rowLength > 0) tileLen = rowLength;

    // blockDim: one block handles one row grid-stride
    uint32_t blockDim = (rows == 0) ? 1u : rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_axis(axis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("axis").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t rowLength,
                                uint32_t tileLength, int32_t axis)
    {
        rows_ = rows;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        axis_ = axis;

        totalLen_ = rows_ * rowLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 2, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes rows in a grid-stride loop
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            RowSoftmax(row);
        }
    }

private:
    __aicore__ inline void CopyInTile(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOutTile(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

    __aicore__ inline void RowSoftmax(uint32_t row)
    {
        const uint32_t base = row * rowLen_;

        // Pass 1: compute row max
        float rowMax = -3.402823466e+38F; // -FLT_MAX
        uint32_t processed = 0;
        while (processed < rowLen_) {
            const uint32_t len = AscendC::min(tileLen_, rowLen_ - processed);
            CopyInTile(base + processed, len);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

            AscendC::LocalTensor<float> tileMaxLocal = scalarQueue_.AllocTensor<float>();
            // Reduce max over tile
            AscendC::ReduceMax(tileMaxLocal, xLocal, tmpLocal, len);

            // Read back 1-element max to scalar
            float tileMax;
            AscendC::DataCopy(&tileMax, tileMaxLocal, 1);

            rowMax = (rowMax > tileMax) ? rowMax : tileMax;

            scalarQueue_.FreeTensor(tileMaxLocal);
            tmpQueue_.FreeTensor(tmpLocal);
            inQueueX_.FreeTensor(xLocal);
            processed += len;
        }

        // Pass 2: compute sum of exp(x - rowMax)
        float rowSum = 0.0f;
        processed = 0;
        while (processed < rowLen_) {
            const uint32_t len = AscendC::min(tileLen_, rowLen_ - processed);
            CopyInTile(base + processed, len);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> shiftLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();

            // shift: x - rowMax
            AscendC::Adds(shiftLocal, xLocal, -rowMax, len);
            // exp(shift)
            AscendC::Exp(expLocal, shiftLocal, shiftLocal, len);

            AscendC::LocalTensor<float> tileSumLocal = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceSum(tileSumLocal, expLocal, shiftLocal, len);

            float tileSum;
            AscendC::DataCopy(&tileSum, tileSumLocal, 1);
            rowSum += tileSum;

            scalarQueue_.FreeTensor(tileSumLocal);
            tmpQueue_.FreeTensor(expLocal);
            tmpQueue_.FreeTensor(shiftLocal);
            inQueueX_.FreeTensor(xLocal);
            processed += len;
        }

        // Pass 3: write normalized y = exp(x - rowMax) / rowSum
        const float invSum = 1.0f / rowSum;
        processed = 0;
        while (processed < rowLen_) {
            const uint32_t len = AscendC::min(tileLen_, rowLen_ - processed);
            CopyInTile(base + processed, len);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

            AscendC::Adds(tmpLocal, xLocal, -rowMax, len);     // shift
            AscendC::Exp(yLocal, tmpLocal, tmpLocal, len);     // exp
            AscendC::Muls(yLocal, yLocal, invSum, len);        // divide by sum

            tmpQueue_.FreeTensor(tmpLocal);
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);

            CopyOutTile(base + processed, len);
            processed += len;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    int32_t axis_ {1};
    uint32_t totalLen_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.rows, tiling_data.rowLength, tiling_data.tileLength, tiling_data.axis);
    op.Process();
}
"""
