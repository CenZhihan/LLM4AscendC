static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dimAttr = -1;
    if (attrs != nullptr && attrs->GetNumArgs() > 0) {
        const int64_t *pDim = attrs->GetAttrPointer<int64_t>(0);
        if (pDim != nullptr) dimAttr = *pDim;
    }

    const gert::Shape *in_shape = context->GetInputShape(0);
    const gert::Shape &origin = in_shape->GetOriginShape();

    // Compute rowLen as the size along reduction axis (last axis by default)
    uint32_t rank = static_cast<uint32_t>(origin.GetDimNum());
    int64_t reduceAxis = (dimAttr == -1 ? static_cast<int64_t>(rank) - 1 : dimAttr);
    if (reduceAxis < 0) {
        reduceAxis += rank; // support negative indexing
    }
    if (reduceAxis < 0) reduceAxis = static_cast<int64_t>(rank) - 1;
    if (reduceAxis >= static_cast<int64_t>(rank)) reduceAxis = static_cast<int64_t>(rank) - 1;

    uint64_t rowLen64 = 1;
    if (rank > 0) {
        rowLen64 = static_cast<uint64_t>(origin.GetDim(reduceAxis));
    }
    uint64_t totalSize = origin.GetShapeSize();
    uint64_t rows64 = (rowLen64 == 0) ? 0ULL : (totalSize / rowLen64);

    uint32_t rowLen = static_cast<uint32_t>(rowLen64);
    uint32_t totalRows = static_cast<uint32_t>(rows64);

    // BlockDim heuristic: one block per 64 rows up to 24
    uint32_t blockDim = totalRows;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 24u) {
        blockDim = 24u;
    }
    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_rowLen(rowLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0; // no workspace
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLen)
    {
        totalRows_ = totalRows;
        rowLen_ = rowLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(totalRows_) * rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(totalRows_) * rowLen_);

        // UB buffers for one row processing
        uint32_t rowBytes = rowLen_ * sizeof(float);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, rowBytes);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, rowBytes);
        pipe_.InitBuffer(tmpQueueVec1_, 1, rowBytes); // for exp or ln or broadcast
        pipe_.InitBuffer(tmpQueueVec2_, 1, rowBytes); // for shifted, broadcast, etc.
        pipe_.InitBuffer(tmpQueueScalar_, 2, sizeof(float)); // scalars for reductions
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride loop over rows
        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            // Copy in one row
            CopyIn(row);

            // Compute log_softmax for one row
            ComputeRow();

            // Copy out
            CopyOut(row);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        const uint64_t offset = static_cast<uint64_t>(rowIdx) * rowLen_;
        AscendC::DataCopy(xLocal, xGm_[offset], rowLen_);
        inQueueX_.EnQue<float>(xLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> vec1 = tmpQueueVec1_.AllocTensor<float>();
        AscendC::LocalTensor<float> vec2 = tmpQueueVec2_.AllocTensor<float>();

        // Temporary scalar tensors
        AscendC::LocalTensor<float> maxLocal = tmpQueueScalar_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = tmpQueueScalar_.AllocTensor<float>();

        // 1) Compute max along row -> maxLocal[0]
        AscendC::ReduceMax(maxLocal, xLocal, rowLen_);

        // 2) Shift: vec1 = x - max
        AscendC::Dup(vec2, maxLocal.GetValue(0), rowLen_);   // broadcast max
        AscendC::Sub(vec1, xLocal, vec2, rowLen_);

        // 3) exp of shifted: vec2 = exp(vec1)
        AscendC::Exp(vec2, vec1, rowLen_);

        // 4) sum of exp -> sumLocal[0]
        AscendC::ReduceSum(sumLocal, vec2, rowLen_);

        // 5) ln(sum) broadcast
        AscendC::Ln(sumLocal, sumLocal, 1);
        AscendC::Dup(vec2, sumLocal.GetValue(0), rowLen_);

        // 6) ln(exp(shift)) = shift; so y = shift - ln_sum
        //    We can recompute y as: yLocal = vec1 - vec2
        AscendC::Sub(yLocal, vec1, vec2, rowLen_);

        // Release temps and queues
        tmpQueueScalar_.FreeTensor(sumLocal);
        tmpQueueScalar_.FreeTensor(maxLocal);
        tmpQueueVec2_.FreeTensor(vec2);
        tmpQueueVec1_.FreeTensor(vec1);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        const uint64_t offset = static_cast<uint64_t>(rowIdx) * rowLen_;
        AscendC::DataCopy(yGm_[offset], yLocal, rowLen_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueVec1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueVec2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueueScalar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_{0};
    uint32_t rowLen_{0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.rowLen);
    op.Process();
}
