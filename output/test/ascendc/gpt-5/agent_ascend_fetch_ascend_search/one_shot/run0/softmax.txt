host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1ull) / b);
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    const gert::Shape &shape = inShape->GetOriginShape();
    const size_t rank = shape.GetDimNum();

    // Default axis from attrs
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *axisPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t axis = (axisPtr != nullptr) ? *axisPtr : 1;

    // Normalize axis to last dimension (softmax along last axis)
    int32_t normAxis = axis;
    if (normAxis < 0) normAxis += static_cast<int32_t>(rank);
    if (normAxis < 0) normAxis = static_cast<int32_t>(rank) - 1;
    if (normAxis >= static_cast<int32_t>(rank)) normAxis = static_cast<int32_t>(rank) - 1;

    // Flatten to [m, n] with n as size along 'normAxis'
    uint64_t m = 1;
    uint64_t n = 1;
    for (size_t i = 0; i < rank; ++i) {
        const uint64_t d = static_cast<uint64_t>(shape.GetDim(i));
        if (i < static_cast<size_t>(normAxis)) {
            m *= d;
        } else if (i == static_cast<size_t>(normAxis)) {
            n = d;
        } else {
            // axes after 'normAxis' are collapsed into n as last-axis
            n *= d;
        }
    }

    if (m == 0 || n == 0) {
        context->SetBlockDim(1);
        tiling.set_m(static_cast<uint32_t>(m));
        tiling.set_n(static_cast<uint32_t>(n));
        tiling.set_tileRows(0);
        tiling.set_blockDim(1);
        tiling.set_axis(normAxis);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *workspace = context->GetWorkspaceSizes(1);
        workspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    // Row-tiling: process 1 row per core iteration for simplicity/stability
    uint32_t tileRows = 1;

    // Determine blockDim based on number of row tiles
    uint32_t numTiles = CeilDiv(m, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_m(static_cast<uint32_t>(m));
    tiling.set_n(static_cast<uint32_t>(n));
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);
    tiling.set_axis(normAxis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t m, uint32_t n,
                                uint32_t tileRows)
    {
        m_ = m;
        n_ = n;
        tileRows_ = (tileRows == 0) ? 1u : tileRows;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(m_) * n_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(m_) * n_);

        uint32_t rowElems = n_;
        uint32_t rowBytes = rowElems * sizeof(float);
        // in/out queues hold one row worth of data per buffer
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, rowBytes);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, rowBytes);
        // tmp holds max and sum buffers for SoftMax
        pipe_.InitBuffer(tmpQueue_, 2, rowBytes);
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || n_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        // Each core processes grid-stride rows
        uint32_t row = blockIdx;
        if (row >= m_) return;

        // Preload first row for overlap
        CopyInRow(row);

        while (true) {
            const uint32_t nextRow = row + blockNum;
            bool hasNext = (nextRow < m_);
            if (hasNext) {
                CopyInRow(nextRow);
            }

            ComputeRow(n_);
            CopyOutRow(row);

            if (!hasNext) break;
            row = nextRow;
        }
    }

private:
    __aicore__ inline void CopyInRow(uint32_t rowIdx)
    {
        const uint64_t gmOffset = static_cast<uint64_t>(rowIdx) * n_;
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        DataCopy(xLocal, xGm_[gmOffset], n_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRow(uint32_t len)
    {
        LocalTensor<float> srcLocal = inQueueX_.DeQue<float>();
        LocalTensor<float> dstLocal = outQueueY_.AllocTensor<float>();
        LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
        LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();

        // Prepare SoftMax shape info for a single row [1, n_]
        SoftMaxShapeInfo shapeInfo;
        shapeInfo.SetOriSrcM(1);
        shapeInfo.SetOriSrcK(len);
        shapeInfo.SetIsLastDim(true); // compute along last axis

        // Let API auto-check/recompute tiling if needed
        SoftMaxTiling tiling;
        // Use default config with isCheckTiling = true
        SoftmaxConfig cfg;

        (void)cfg; // suppress unused warning if cfg not directly used by compiler

        // Execute SoftMax on this row
        SoftMax<float, false, false, false>(dstLocal, sumLocal, maxLocal, srcLocal, tiling, shapeInfo);

        tmpQueue_.FreeTensor(sumLocal);
        tmpQueue_.FreeTensor(maxLocal);
        outQueueY_.EnQue<float>(dstLocal);
        inQueueX_.FreeTensor(srcLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t rowIdx)
    {
        const uint64_t gmOffset = static_cast<uint64_t>(rowIdx) * n_;
        LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        DataCopy(yGm_[gmOffset], yLocal, n_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 2> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t m_{0};
    uint32_t n_{0};
    uint32_t tileRows_{1};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.m, tiling_data.n, tiling_data.tileRows);
    op.Process();
}
"""
