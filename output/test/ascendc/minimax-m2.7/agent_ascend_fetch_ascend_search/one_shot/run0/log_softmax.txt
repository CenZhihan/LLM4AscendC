host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 1 * tile (logsoftmax result)
//   maxQueue: 1 * tile (for row max reduction)
//   expQueue: 1 * tile (for exp results)
//   sumQueue: 1 * tile (for row sum reduction)
// Total: 6 * tile floats => 24 * tile bytes
static constexpr uint32_t UB_BUDGET_BYTES = 128 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::Shape *originShape = context->GetOriginShape(0);
    
    uint64_t totalLength = static_cast<uint64_t>(originShape->GetShapeSize());
    uint64_t rowLength = static_cast<uint64_t>(inputShape->GetShapeVector()[inputShape->GetShapeVector().size() - 1]);
    uint64_t batchSize = totalLength / rowLength;
    
    // Tiling based on UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / (6 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(totalLength), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(totalLength);
    }
    
    // Block dim calculation
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(static_cast<uint32_t>(totalLength), tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;
    
    context->SetBlockDim(blockDim);
    
    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_batchSize(batchSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint64_t totalLength, uint64_t rowLength,
                                uint64_t batchSize, uint32_t tileLength)
    {
        totalLen_ = static_cast<uint32_t>(totalLength);
        rowLen_ = static_cast<uint32_t>(rowLength);
        batchSize_ = static_cast<uint32_t>(batchSize);
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Buffer sizes: inQueueX(2*tile), maxQueue(1*tile for reduction),
        // expQueue(2*tile), sumQueue(1*tile for reduction), outQueueY(1*tile)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(maxQueue_, 1, rowLen_ * sizeof(float));
        pipe_.InitBuffer(expQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sumQueue_, 1, rowLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Process batch rows with block distribution
        const uint32_t rowsPerBlock = (batchSize_ + blockNum - 1) / blockNum;
        const uint32_t startRow = blockIdx * rowsPerBlock;
        const uint32_t endRow = (startRow + rowsPerBlock > batchSize_) ? batchSize_ : startRow + rowsPerBlock;

        for (uint32_t row = startRow; row < endRow; ++row) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        const uint32_t rowOffset = rowIdx * rowLen_;

        // Step 1: Load entire row and compute row max
        AscendC::LocalTensor<float> maxLocal = maxQueue_.AllocTensor<float>();
        
        // Load first tile of row
        uint32_t remaining = rowLen_;
        uint32_t offset = rowOffset;
        
        // Copy first portion to maxLocal
        AscendC::LocalTensor<float> xFirst = inQueueX_.AllocTensor<float>();
        uint32_t copyLen = (remaining > tileLen_) ? tileLen_ : remaining;
        AscendC::DataCopy(xFirst, xGm_[offset], copyLen);
        inQueueX_.EnQue(xFirst);
        
        // Initialize max with first tile
        AscendC::ReduceMax(maxLocal, xFirst, 0, 1, copyLen, rowLen_);
        
        // Process remaining tiles
        offset += copyLen;
        remaining -= copyLen;
        
        while (remaining > 0) {
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            inQueueX_.FreeTensor(xLocal);
            
            AscendC::LocalTensor<float> xNext = inQueueX_.AllocTensor<float>();
            copyLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::DataCopy(xNext, xGm_[offset], copyLen);
            inQueueX_.EnQue(xNext);
            
            // Update max
            AscendC::ReduceMax(maxLocal, xNext, 0, 1, copyLen, rowLen_);
            
            offset += copyLen;
            remaining -= copyLen;
        }
        
        // Deque last tile and free
        AscendC::LocalTensor<float> xLast = inQueueX_.DeQue<float>();
        inQueueX_.FreeTensor(xLast);
        
        // Broadcast max to entire row length in expQueue
        AscendC::LocalTensor<float> expBuf = expQueue_.AllocTensor<float>();
        AscendC::Broadcast(expBuf, maxLocal, rowLen_);
        maxQueue_.FreeTensor(maxLocal);
        
        // Step 2: Subtract max and compute exp, then reduce sum
        remaining = rowLen_;
        offset = rowOffset;
        
        // Process first tile: sub, exp
        AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
        copyLen = (remaining > tileLen_) ? tileLen_ : remaining;
        AscendC::DataCopy(xTile, xGm_[offset], copyLen);
        inQueueX_.EnQue(xTile);
        
        AscendC::LocalTensor<float> subLocal = expQueue_.AllocTensor<float>();
        AscendC::Sub(subLocal, xTile, expBuf, copyLen);
        
        AscendC::LocalTensor<float> expLocal = expQueue_.AllocTensor<float>();
        AscendC::Exp(expLocal, subLocal, copyLen);
        
        inQueueX_.FreeTensor(xTile);
        
        // Reduce sum for first portion
        AscendC::ReduceSum(sumQueue_.AllocTensor<float>(), expLocal, 0, 1, copyLen, rowLen_);
        expQueue_.FreeTensor(expLocal);
        
        offset += copyLen;
        remaining -= copyLen;
        
        // Process remaining tiles
        while (remaining > 0) {
            xTile = inQueueX_.AllocTensor<float>();
            copyLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::DataCopy(xTile, xGm_[offset], copyLen);
            inQueueX_.EnQue(xTile);
            
            subLocal = expQueue_.AllocTensor<float>();
            AscendC::Sub(subLocal, xTile, expBuf, copyLen);
            
            expLocal = expQueue_.AllocTensor<float>();
            AscendC::Exp(expLocal, subLocal, copyLen);
            
            inQueueX_.FreeTensor(xTile);
            expQueue_.FreeTensor(subLocal);
            
            // Continue reduction sum
            AscendC::LocalTensor<float> sumTensor = sumQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumTensor, expLocal, 0, 1, copyLen, rowLen_);
            expQueue_.FreeTensor(expLocal);
            
            offset += copyLen;
            remaining -= copyLen;
        }
        
        // Get sum result
        AscendC::LocalTensor<float> sumLocal = sumQueue_.DeQue<float>();
        
        // Broadcast sum to expBuf
        AscendC::Broadcast(expBuf, sumLocal, rowLen_);
        sumQueue_.FreeTensor(sumLocal);
        
        // Step 3: Div and log, then copy out
        remaining = rowLen_;
        offset = rowOffset;
        
        // First tile
        AscendC::LocalTensor<float> xTileOut = inQueueX_.AllocTensor<float>();
        copyLen = (remaining > tileLen_) ? tileLen_ : remaining;
        AscendC::DataCopy(xTileOut, xGm_[offset], copyLen);
        inQueueX_.EnQue(xTileOut);
        
        AscendC::LocalTensor<float> subOut = expQueue_.AllocTensor<float>();
        AscendC::Sub(subOut, xTileOut, expBuf, copyLen);
        
        AscendC::LocalTensor<float> expOut = expQueue_.AllocTensor<float>();
        AscendC::Exp(expOut, subOut, copyLen);
        
        inQueueX_.FreeTensor(xTileOut);
        expQueue_.FreeTensor(subOut);
        
        AscendC::LocalTensor<float> divOut = outQueueY_.AllocTensor<float>();
        AscendC::Div(divOut, expOut, expBuf, copyLen);
        expQueue_.FreeTensor(expOut);
        
        AscendC::LocalTensor<float> logOut = outQueueY_.AllocTensor<float>();
        AscendC::Log(divOut, logOut, copyLen);
        outQueueY_.FreeTensor(divOut);
        
        // Copy out first portion
        AscendC::DataCopy(yGm_[offset], logOut, copyLen);
        outQueueY_.FreeTensor(logOut);
        
        offset += copyLen;
        remaining -= copyLen;
        
        // Process remaining tiles
        while (remaining > 0) {
            xTileOut = inQueueX_.AllocTensor<float>();
            copyLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::DataCopy(xTileOut, xGm_[offset], copyLen);
            inQueueX_.EnQue(xTileOut);
            
            subOut = expQueue_.AllocTensor<float>();
            AscendC::Sub(subOut, xTileOut, expBuf, copyLen);
            
            expOut = expQueue_.AllocTensor<float>();
            AscendC::Exp(expOut, subOut, copyLen);
            
            inQueueX_.FreeTensor(xTileOut);
            expQueue_.FreeTensor(subOut);
            
            divOut = outQueueY_.AllocTensor<float>();
            AscendC::Div(divOut, expOut, expBuf, copyLen);
            expQueue_.FreeTensor(expOut);
            
            logOut = outQueueY_.AllocTensor<float>();
            AscendC::Log(divOut, logOut, copyLen);
            outQueueY_.FreeTensor(divOut);
            
            AscendC::DataCopy(yGm_[offset], logOut, copyLen);
            outQueueY_.FreeTensor(logOut);
            
            offset += copyLen;
            remaining -= copyLen;
        }
        
        expQueue_.FreeTensor(expBuf);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maxQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> expQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLength, tiling_data.batchSize, tiling_data.tileLength);
    op.Process();
}
