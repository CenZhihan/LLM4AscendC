host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t ELEMS_PER_BLOCK = 4096;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const uint32_t batchSize = static_cast<uint32_t>(inputShape->GetShapeDim(0));
    const uint32_t featureSize = static_cast<uint32_t>(inputShape->GetShapeDim(1));

    uint32_t tileBatch = ELEMS_PER_BLOCK;
    if (tileBatch > batchSize) {
        tileBatch = batchSize;
    }

    uint32_t blockDim = CeilDiv(batchSize, tileBatch);
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 64) blockDim = 64;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_featureSize(featureSize);
    tiling.set_tileBatch(tileBatch);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::graphStatus::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return ge::graphStatus::GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return ge::graphStatus::GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(ge::ParamType::REQUIRED)
            .DataType({ge::DataType::DT_FLOAT})
            .Format({ge::Format::FORMAT_ND})
            .UnknownShapeFormat({ge::Format::FORMAT_ND});

        this->Output("y")
            .ParamType(ge::ParamType::REQUIRED)
            .DataType({ge::DataType::DT_FLOAT})
            .Format({ge::Format::FORMAT_ND})
            .UnknownShapeFormat({ge::Format::FORMAT_ND});

        this->SetInferShape(ge::InferShape);
        this->SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_FEATURE_SIZE = 32768;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t featureSize,
                                uint32_t tileBatch)
    {
        batchSize_ = batchSize;
        featureSize_ = featureSize;
        tileBatch_ = tileBatch;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * featureSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * featureSize);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileBatch_ * featureSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileBatch_ * featureSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueMax_, 1, tileBatch_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueSum_, 1, tileBatch_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || featureSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t startBatch = blockIdx * tileBatch_;
        if (startBatch >= batchSize_) return;

        uint32_t remainBatch = batchSize_ - startBatch;
        uint32_t curBatch = (remainBatch < tileBatch_) ? remainBatch : tileBatch_;

        CopyIn(startBatch, curBatch);

        while (true) {
            Compute(curBatch);
            CopyOut(startBatch, curBatch);

            uint32_t nextStartBatch = startBatch + blockNum * tileBatch_;
            if (nextStartBatch >= batchSize_) break;

            uint32_t nextRemainBatch = batchSize_ - nextStartBatch;
            uint32_t nextCurBatch = (nextRemainBatch < tileBatch_) ? nextRemainBatch : tileBatch_;

            CopyIn(nextStartBatch, nextCurBatch);
            startBatch = nextStartBatch;
            curBatch = nextCurBatch;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t batchOffset, uint32_t batchCount)
    {
        uint32_t copySize = batchCount * featureSize_;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[batchOffset * featureSize_], copySize);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t batchCount)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = tmpQueueMax_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = tmpQueueSum_.AllocTensor<float>();

        uint32_t rowLength = featureSize_;

        AscendC::ReduceMax<float>(maxLocal, xLocal, true, AscendC::KernelCoords(), rowLength, batchCount);

        AscendC::SubScalar(yLocal, xLocal, maxLocal, batchCount, rowLength, rowLength);

        AscendC::Exp(yLocal, yLocal, batchCount * rowLength);

        AscendC::ReduceSum<float>(sumLocal, yLocal, true, AscendC::KernelCoords(), rowLength, batchCount);

        AscendC::Div(yLocal, yLocal, sumLocal, batchCount, rowLength, rowLength);

        tmpQueueMax_.FreeTensor(maxLocal);
        tmpQueueSum_.FreeTensor(sumLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t batchOffset, uint32_t batchCount)
    {
        uint32_t copySize = batchCount * featureSize_;
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[batchOffset * featureSize_], yLocal, copySize);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueSum_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t featureSize_ {0};
    uint32_t tileBatch_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.batchSize, tiling_data.featureSize, tiling_data.tileBatch);
    op.Process();
}
