#include "layer_norm_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0)->GetOriginShape();
    // x shape: (*, normalized_shape). Treat as (totalRows, rowLen)
    const uint32_t ndim = xShape->GetDimNum();
    uint32_t totalRows = 1;
    uint32_t rowLen = 1;
    // For simplicity assume normalized_shape is the last dimension only,
    // but actually it can be multiple dims. We need to parse from gamma shape.
    const gert::Shape *gammaShape = context->GetInputShape(1)->GetOriginShape();
    uint32_t gammaLen = static_cast<uint32_t>(gammaShape->GetShapeSize());
    // rowLen = gammaLen, totalRows = totalElements / rowLen
    uint32_t totalElems = static_cast<uint32_t>(xShape->GetShapeSize());
    totalRows = totalElems / gammaLen;
    rowLen = gammaLen;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float epsilon = attrs->GetAttrPointer<float>(0)[0];

    // Set blockDim: simple heuristic using a fixed value
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_rowLen(rowLen);
    tiling.set_blockDim(blockDim);
    tiling.set_epsilon(epsilon);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src='''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLen,
                                float epsilon)
    {
        totalRows_ = totalRows;
        rowLen_ = rowLen;
        epsilon_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * rowLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, rowLen_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * rowLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, rowLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueGamma_, 1, rowLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueBeta_, 1, rowLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, rowLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, rowLen_ * sizeof(float));
        // Extra workspace for ReduceSum (can reuse tmpQueue but separate for clarity)
        // Use one extra element for scalar means/variance
        pipe_.InitBuffer(workspaceQueue_, 1, sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 2, sizeof(float) * 2); // store mean and variance scalars
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowLen_ == 0) return;

        // Load gamma and beta once (constant across rows)
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma_.AllocTensor<float>();
        AscendC::LocalTensor<float> betaLocal  = inQueueBeta_.AllocTensor<float>();
        AscendC::DataCopy(gammaLocal, gammaGm_, rowLen_);
        AscendC::DataCopy(betaLocal,  betaGm_,  rowLen_);
        inQueueGamma_.EnQue(gammaLocal);
        inQueueBeta_.EnQue(betaLocal);

        // Grid-stride over rows
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideRows = blockNum;
        uint32_t row = blockIdx;
        while (row < totalRows_) {
            ProcessRow(row);
            row += strideRows;
        }

        // Free gamma/beta tensors
        (void)inQueueGamma_.DeQue<float>();
        inQueueGamma_.FreeTensor(gammaLocal);
        (void)inQueueBeta_.DeQue<float>();
        inQueueBeta_.FreeTensor(betaLocal);
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        // Copy x row to local
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[row * rowLen_], rowLen_);

        // Allocate workspace for reduction
        AscendC::LocalTensor<float> reduceWorkspace = workspaceQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> scalar = scalarQueue_.AllocTensor<float>(); // at least 2 elements

        // Compute mean = sum(x) / rowLen
        AscendC::ReduceSum(scalar, xLocal, reduceWorkspace, rowLen_);
        float meanVal = scalar.GetValue(0) / static_cast<float>(rowLen_);

        // Compute x_centered = x - mean; Used for variance and later normalization
        // We'll reuse xLocal for x_centered (since we no longer need original x)
        // But we need x for final normalize, so better use tmp queue.
        AscendC::LocalTensor<float> centered = tmpQueue_.AllocTensor<float>();
        // centering: Adds(xLocal, -meanVal)
        AscendC::Adds(centered, xLocal, -meanVal, rowLen_);

        // Compute variance = sum((x-mean)^2) / rowLen
        AscendC::LocalTensor<float> sq = tmpQueue_.AllocTensor<float>(); // reuse? Need separate since centered is in use
        // Actually we can reuse centered buffer after computing variance? But we need centered later for normalize.
        // Let's allocate a separate tensor for squared values.
        AscendC::LocalTensor<float> sqTmp = workspaceQueue_.AllocTensor<float>(); // reuse reduceWorkspace? Normally workspace is separate. We'll allocate from pipe extra.
        // Instead, we can compute variance using centered and then reuse it for multiply with gamma.
        // Use in-place: Mul(sq, centered, centered) into same buffer? But centered needed later.
        // Simpler: allocate another local tensor from tmpQueue. Since tmpQueue has 1 tile, we need to manage.
        // We'll compute variance in a two-step: first accumulate sum of squares using centered,
        // then store variance in scalar.
        // We'll compute sum of squares using ReduceSum on elementwise squared values.
        // To avoid extra buffer, compute squared values into centered (overwrite) but then we lose centered for normalization.
        // So we need two temporary buffers: one for centered, one for squared.
        // Since rowLen could be large, we cannot have too many buffers. Use a single temporary and recompute? Or use two tiles from queue.
        // We'll allocate another tile from inQueueX? No, that's used for loading. We'll allocate from tmpQueue with larger size? But we only have 1 tile in tmpQueue.
        // Given the constraints, we'll recompute x-mean later after computing variance. That's fine: we already have xLocal and mean.
        // So compute sum of squares directly: create a temporary local tensor for (x-mean)^2, then reduce.
        // Use centered2 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> centered2 = tmpQueue_.AllocTensor<float>();
        AscendC::Sub(centered2, xLocal, scalar, rowLen_); // scalar has (mean, mean, ...) Actually need a broadcast of mean.
        // Better: use Adds as before. Already computed centered via Adds. So we can use centered.
        // But we overwrote centered? No, we computed centered earlier and left it. We'll keep centered.
        // Compute squares: Mul(sq, centered, centered)
        AscendC::LocalTensor<float> sqLocal = tmpQueue_.AllocTensor<float>();
        AscendC::Mul(sqLocal, centered, centered, rowLen_);
        // Reduce sum of squares
        AscendC::ReduceSum(scalar + sizeof(float), sqLocal, reduceWorkspace, rowLen_); // use second half of scalar for var
        float varVal = (scalar.GetValue(1) / static_cast<float>(rowLen_)) + epsilon_;
        float invStd = 1.0f / sqrt(varVal);

        // Normalize: y = centered * invStd * gamma + beta
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        AscendC::Muls(tmp, centered, invStd, rowLen_);
        AscendC::Mul(tmp, tmp, gammaLocal, rowLen_);
        AscendC::Add(tmp, tmp, betaLocal, rowLen_);

        // Copy out
        AscendC::DataCopy(yGm_[row * rowLen_], tmp, rowLen_);

        // Free all local tensors
        inQueueX_.FreeTensor(xLocal);
        tmpQueue_.FreeTensor(centered);
        tmpQueue_.FreeTensor(sqLocal);
        tmpQueue_.FreeTensor(centered2); // not used
        tmpQueue_.FreeTensor(tmp);
        workspaceQueue_.FreeTensor(reduceWorkspace);
        scalarQueue_.FreeTensor(scalar);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueGamma_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueBeta_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workspaceQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t rowLen_ {0};
    float epsilon_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y, tiling_data.totalRows, tiling_data.rowLen, tiling_data.epsilon);
    op.Process();
}
'''
