host_operator_src='''
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t ALIGN_ROW = 8;
static constexpr uint32_t MAX_BLOCK = 24;
static constexpr uint32_t SHARED_TMP_BYTES = 60 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    // Use StorageShape pointer for GetOriginShape()
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(inputShape->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    // For int attr, use GetAttrNum (returns int64_t)
    int64_t dim = attrs->GetAttrNum(0);

    // Assume input is 2D: [batch, features]
    uint32_t columnLength = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(dim));
    uint32_t rowLength = totalLength / columnLength;

    uint32_t usedNumBlocks = (rowLength < MAX_BLOCK) ? rowLength : MAX_BLOCK;
    context->SetBlockDim(usedNumBlocks);

    uint32_t coreRowNum = rowLength / usedNumBlocks;
    uint32_t tailRow = rowLength % usedNumBlocks;
    uint32_t singleLoopCoreRowNum = coreRowNum;
    if (singleLoopCoreRowNum == 0) singleLoopCoreRowNum = 1;

    uint32_t singleCoreLoopCount = 1;
    uint32_t leftRow = 0;
    uint32_t tailCoreSingleLoopCoreRowNum = coreRowNum;
    uint32_t tailCoreSingleCoreLoopCount = 1;
    uint32_t tailCoreSingleCoreLoopTail = 0;

    if (tailRow != 0) {
        tailCoreSingleLoopCoreRowNum = coreRowNum + 1;
        tailCoreSingleCoreLoopCount = 1;
        tailCoreSingleCoreLoopTail = 0;
    }

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_columnLength(columnLength);
    tiling.set_usedNumBlocks(usedNumBlocks);
    tiling.set_coreRowNum(coreRowNum);
    tiling.set_singleLoopCoreRowNum(singleLoopCoreRowNum);
    tiling.set_singleCoreLoopCount(singleCoreLoopCount);
    tiling.set_leftRow(leftRow);
    tiling.set_tailCoreSingleLoopCoreRowNum(tailCoreSingleLoopCoreRowNum);
    tiling.set_tailCoreSingleCoreLoopCount(tailCoreSingleCoreLoopCount);
    tiling.set_tailCoreSingleCoreLoopTail(tailCoreSingleCoreLoopTail);
    tiling.set_sharedTmpBufferSize(SHARED_TMP_BYTES);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src='''
#include "kernel_operator.h"
#include "adv_api/activation/logsoftmax.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z,
                                uint32_t totalLength, uint32_t rowLength, uint32_t columnLength,
                                uint32_t usedNumBlocks, uint32_t coreRowNum,
                                uint32_t singleLoopCoreRowNum, uint32_t singleCoreLoopCount,
                                uint32_t leftRow, uint32_t tailCoreSingleLoopCoreRowNum,
                                uint32_t tailCoreSingleCoreLoopCount, uint32_t tailCoreSingleCoreLoopTail,
                                uint32_t sharedTmpBufferSize)
    {
        totalLength_ = totalLength;
        rowLength_ = rowLength;
        columnLength_ = columnLength;
        usedNumBlocks_ = usedNumBlocks;
        coreRowNum_ = coreRowNum;
        singleLoopCoreRowNum_ = singleLoopCoreRowNum;
        singleCoreLoopCount_ = singleCoreLoopCount;
        leftRow_ = leftRow;
        tailCoreSingleLoopCoreRowNum_ = tailCoreSingleLoopCoreRowNum;
        tailCoreSingleCoreLoopCount_ = tailCoreSingleCoreLoopCount;
        tailCoreSingleCoreLoopTail_ = tailCoreSingleCoreLoopTail;
        sharedTmpBufferSize_ = sharedTmpBufferSize;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t myRows = coreRowNum_;
        if (blockIdx == usedNumBlocks_ - 1 && leftRow_ > 0) {
            myRows = tailCoreSingleLoopCoreRowNum_;
        }

        uint32_t blockLength = myRows * columnLength_;
        uint32_t offset = coreRowNum_ * columnLength_ * blockIdx;
        xGm_.SetGlobalBuffer((__gm__ float*)x + offset, blockLength);
        zGm_.SetGlobalBuffer((__gm__ float*)z + offset, blockLength);

        uint32_t tileLength = singleLoopCoreRowNum_ * columnLength_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength * sizeof(float));
        pipe_.InitBuffer(outQueueZ_, BUFFER_NUM, tileLength * sizeof(float));
        pipe_.InitBuffer(queueMax_, 1, singleLoopCoreRowNum_ * sizeof(float));
        pipe_.InitBuffer(queueSum_, 1, singleLoopCoreRowNum_ * sizeof(float));
        pipe_.InitBuffer(sharedTmpBuffer_, sharedTmpBufferSize_);
    }

    __aicore__ inline void Process()
    {
        if (rowLength_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= usedNumBlocks_) return;

        uint32_t myRows = coreRowNum_;
        if (blockIdx == usedNumBlocks_ - 1 && leftRow_ > 0) {
            myRows = tailCoreSingleLoopCoreRowNum_;
        }

        uint32_t tileRowNum = myRows;
        uint32_t tileLen = tileRowNum * columnLength_;
        CopyIn(tileLen);
        Compute(tileRowNum, tileLen);
        CopyOut(tileLen);
    }

private:
    __aicore__ inline void CopyIn(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_, len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rowNum, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = queueMax_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = queueSum_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tmpBuf = sharedTmpBuffer_.Get<uint8_t>();

        AscendC::SoftMaxShapeInfo shape = {rowNum, columnLength_, rowNum, columnLength_};
        AscendC::LogSoftMaxTiling logSoftMaxTiling;  // default tiling

        AscendC::LogSoftMax(zLocal, sumLocal, maxLocal, xLocal, tmpBuf, logSoftMaxTiling, shape);

        queueMax_.FreeTensor(maxLocal);
        queueSum_.FreeTensor(sumLocal);
        outQueueZ_.EnQue<float>(zLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t len)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ_.DeQue<float>();
        AscendC::DataCopy(zGm_, zLocal, len);
        outQueueZ_.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> queueMax_, queueSum_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> sharedTmpBuffer_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> zGm_;

    uint32_t totalLength_;
    uint32_t rowLength_;
    uint32_t columnLength_;
    uint32_t usedNumBlocks_;
    uint32_t coreRowNum_;
    uint32_t singleLoopCoreRowNum_;
    uint32_t singleCoreLoopCount_;
    uint32_t leftRow_;
    uint32_t tailCoreSingleLoopCoreRowNum_;
    uint32_t tailCoreSingleCoreLoopCount_;
    uint32_t tailCoreSingleCoreLoopTail_;
    uint32_t sharedTmpBufferSize_;
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, z,
            tilingData.totalLength, tilingData.rowLength, tilingData.columnLength,
            tilingData.usedNumBlocks, tilingData.coreRowNum,
            tilingData.singleLoopCoreRowNum, tilingData.singleCoreLoopCount,
            tilingData.leftRow, tilingData.tailCoreSingleLoopCoreRowNum,
            tilingData.tailCoreSingleCoreLoopCount, tilingData.tailCoreSingleCoreLoopTail,
            tilingData.sharedTmpBufferSize);
    op.Process();
}
'''
