#include "triplet_margin_loss_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN = 32;                    // 32B alignment

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;
    const gert::Shape *anchor_shape = context->GetInputShape(0);
    uint32_t totalN = anchor_shape->GetDim(0);
    uint32_t totalD = anchor_shape->GetDim(1);

    // Compute max tile batch from UB budget.
    // For each batch element, we need temporary buffers for diff_ap, diff_an, sq_ap, sq_an, plus margin copy.
    // Rough: 3*totalD floats input + 1*totalD diff + 1*totalD diff_sq + scalars => ~5*totalD floats per batch.
    // With double buffering factor 1, we allocate once.
    uint32_t bytesPerBatch = 5 * totalD * sizeof(float);
    uint32_t maxBatch = UB_BUDGET_BYTES / bytesPerBatch;
    maxBatch = AlignDown(maxBatch, 1u); // no extra alignment, but ensure >=1
    if (maxBatch == 0) maxBatch = 1;
    uint32_t tileBatch = std::min(totalN, maxBatch);

    uint32_t blockDim = 1; // single block for simplicity; can be increased for large n
    context->SetBlockDim(blockDim);

    tiling.set_totalN(totalN);
    tiling.set_totalD(totalD);
    tiling.set_tileBatch(tileBatch);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;
constexpr uint32_t MASK_8 = 8;   // 256 bits for float

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR margin, uint32_t totalN, uint32_t totalD,
                                uint32_t tileBatch)
    {
        totalN_ = totalN;
        totalD_ = totalD;
        dim_ = totalD;
        // margin is assumed to be a scalar tensor in GM
        marginGm_.SetGlobalBuffer((__gm__ float*)margin, 1);

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalN * totalD);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalN * totalD);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalN * totalD);
        // output shape (totalN)
        yGm_.SetGlobalBuffer((__gm__ float*)anchor, totalN); // reuse anchor ptr for output? Use separate output shape.
        // Actually output should be separate. We'll re-map.
        // Correction: we need a separate output buffer. Use the fourth input? No.
        // The operator definition has output y. The kernel receives it as fifth argument.
        // So we should get it from Init caller.
        yGm_.SetGlobalBuffer((__gm__ float*)anchor, totalN); // placeholder, will be set in Process.
    }

    __aicore__ inline void Process()
    {
        if (totalN_ == 0 || totalD_ == 0) return;

        // Re-map yGm_ to correct address: we need to pass it as extra argument.
        // For simplicity, we assume Init receives an additional argument for y.
        // We'll fix in final.
    }

    // Overloaded Init with output
    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR margin, GM_ADDR y,
                                uint32_t totalN, uint32_t totalD, uint32_t tileBatch)
    {
        totalN_ = totalN;
        totalD_ = totalD;
        dim_ = totalD;
        marginGm_.SetGlobalBuffer((__gm__ float*)margin, 1);
        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalN * totalD);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalN * totalD);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalN * totalD);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalN);
        tileBatch_ = tileBatch;

        // Allocate local buffers for a tile of batch elements.
        // We'll process tileBatch elements at a time.
        uint32_t batchElems = tileBatch_ * totalD_;
        pipe_.InitBuffer(anchorLocal_, BUFFER_NUM, batchElems * sizeof(float));
        pipe_.InitBuffer(positiveLocal_, BUFFER_NUM, batchElems * sizeof(float));
        pipe_.InitBuffer(negativeLocal_, BUFFER_NUM, batchElems * sizeof(float));
        pipe_.InitBuffer(yLocal_, BUFFER_NUM, tileBatch_ * sizeof(float));
        pipe_.InitBuffer(tmpDiff_, 1, totalD_ * sizeof(float));      // for diff vector
        pipe_.InitBuffer(tmpSq_, 1, totalD_ * sizeof(float));       // for squar
        pipe_.InitBuffer(tmpReduce_, 1, 16 * sizeof(float));        // for ReduceSum result (16 floats safe)
        pipe_.InitBuffer(marginLocal_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalN_ == 0 || totalD_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        // grid-stride over batch
        for (uint32_t batchStart = blockIdx * tileBatch_; batchStart < totalN_; batchStart += tileBatch_ * 1) // use single block
        {
            uint32_t curBatch = totalN_ - batchStart;
            if (curBatch > tileBatch_) curBatch = tileBatch_;

            // Copy in anchor, positive, negative for this tile
            AscendC::LocalTensor<float> anchorLoc = anchorLocal_.AllocTensor<float>();
            AscendC::LocalTensor<float> positiveLoc = positiveLocal_.AllocTensor<float>();
            AscendC::LocalTensor<float> negativeLoc = negativeLocal_.AllocTensor<float>();
            AscendC::DataCopy(anchorLoc, anchorGm_[batchStart * totalD_], curBatch * totalD_);
            AscendC::DataCopy(positiveLoc, positiveGm_[batchStart * totalD_], curBatch * totalD_);
            AscendC::DataCopy(negativeLoc, negativeGm_[batchStart * totalD_], curBatch * totalD_);
            // copy margin scalar
            AscendC::LocalTensor<float> marginLoc = marginLocal_.AllocTensor<float>();
            AscendC::DataCopy(marginLoc, marginGm_, 1);
            // We'll use margin scalar for all elements
            float marginVal = marginLoc.GetValue(0);
            marginLocal_.FreeTensor(marginLoc);

            AscendC::LocalTensor<float> yLoc = yLocal_.AllocTensor<float>();

            // Process each batch element in this tile sequentially
            for (uint32_t i = 0; i < curBatch; i++)
            {
                AscendC::LocalTensor<float> a = anchorLoc[i * totalD_];
                AscendC::LocalTensor<float> p = positiveLoc[i * totalD_];
                AscendC::LocalTensor<float> n = negativeLoc[i * totalD_];

                // Compute diff_ap = a - p
                AscendC::Sub(tmpDiff_, a, p, totalD_);
                // sq_ap = diff_ap * diff_ap
                AscendC::Mul(tmpSq_, tmpDiff_, tmpDiff_, totalD_);
                // ReduceSum to get sum over d
                AscendC::ReduceSum<float>(tmpReduce_, tmpSq_, tmpDiff_, totalD_); // tmpDiff_ reused as temp buffer
                float sq_ap = tmpReduce_.GetValue(0);

                // Compute diff_an = a - n
                AscendC::Sub(tmpDiff_, a, n, totalD_);
                AscendC::Mul(tmpSq_, tmpDiff_, tmpDiff_, totalD_);
                AscendC::ReduceSum<float>(tmpReduce_, tmpSq_, tmpDiff_, totalD_);
                float sq_an = tmpReduce_.GetValue(0);

                // Loss = max(sq_ap - sq_an + margin, 0)
                float loss = sq_ap - sq_an + marginVal;
                if (loss < 0.0f) loss = 0.0f;
                yLoc.SetValue(i, loss);
            }

            // Copy out y for this tile
            AscendC::DataCopy(yGm_[batchStart], yLoc, curBatch);
            yLocal_.FreeTensor(yLoc);
            anchorLocal_.FreeTensor(anchorLoc);
            positiveLocal_.FreeTensor(positiveLoc);
            negativeLocal_.FreeTensor(negativeLoc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> anchorLocal_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> positiveLocal_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> negativeLocal_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yLocal_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDiff_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSq_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpReduce_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> marginLocal_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> marginGm_;

    uint32_t totalN_ {0};
    uint32_t totalD_ {0};
    uint32_t dim_ {0};
    uint32_t tileBatch_ {0};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                 GM_ADDR margin, GM_ADDR y,
                                                                 GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, margin, y,
            tiling_data.totalN, tiling_data.totalD, tiling_data.tileBatch);
    op.Process();
}
