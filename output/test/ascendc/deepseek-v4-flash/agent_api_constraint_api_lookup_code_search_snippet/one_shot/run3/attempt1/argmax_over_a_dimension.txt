#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model:
//   inQueueX: 2 * tileRows * numCols * sizeof(float)
//   outQueueY: 2 * tileRows * sizeof(int32_t)
//   tmpQueue:  1 * numCols * sizeof(float)   (plus alignment)
//   extra workspace for ReduceMax (inside tmp)
//   total bytes = 2*tilesize*4 + 2*tileRows*4 + 1*numCols*4 ≈ 8*tilesize + 8*tileRows + 4*numCols
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ALIGN_FLOATS = 64;              // 256B alignment in floats

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const auto &input_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rank = input_shape.GetDimNum();
    const int *dims = input_shape.GetDims();

    // parse attr dim
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += rank;
    uint32_t reduceDim = static_cast<uint32_t>(dim);

    // compute numRows (product of all dims except reduceDim) and numCols (size of reduceDim)
    uint32_t numRows = 1;
    uint32_t numCols = 1;
    for (uint32_t i = 0; i < rank; ++i) {
        uint32_t sz = static_cast<uint32_t>(dims[i]);
        if (i == reduceDim)
            numCols = sz;
        else
            numRows *= sz;
    }

    const uint32_t totalLength = numRows * numCols;

    // compute tileRows from UB budget
    // Let tilesize = tileRows * numCols
    // bytes needed for double-buffered input: 2 * tilesize * sizeof(float) = 8 * tilesize
    // double-buffered output: 2 * tileRows * sizeof(int32_t) = 8 * tileRows
    // tmp buffer: numCols * sizeof(float) = 4 * numCols
    // we ignore extra alignment for simplicity; add a small overhead
    uint32_t bytesPerRowComponent = 8 * numCols + 8;   // input + output per row (double)
    uint32_t fixedBytes = 4 * numCols;                  // tmp buffer
    uint32_t maxRows = (UB_BUDGET_BYTES - fixedBytes) / bytesPerRowComponent;
    if (maxRows == 0) maxRows = 1;
    uint32_t tileRows = AlignDown(maxRows, 8u);  // favor DMA alignment
    if (tileRows < 1) tileRows = 1;
    // also ensure tileRows * numCols is a reasonable size
    if (tileRows > 256) tileRows = 256;  // safety cap

    // blockDim = numRows / tileRows rounded up
    uint32_t blockDim = CeilDiv(numRows, tileRows);
    // clamp blockDim to reasonable range
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_numRows(numRows);
    tiling.set_numCols(numCols);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    // attrs to get dim
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    uint32_t rank = x_shape->GetDimNum();
    if (dim < 0) dim += rank;
    uint32_t reduceDim = static_cast<uint32_t>(dim);

    // output shape: same as input but with reduceDim removed
    std::vector<int64_t> new_dims;
    for (uint32_t i = 0; i < rank; ++i) {
        if (i != reduceDim) {
            new_dims.push_back(x_shape->GetDim(i));
        }
    }
    y_shape->SetDimNum(new_dims.size());
    for (uint32_t i = 0; i < new_dims.size(); ++i) {
        y_shape->SetDim(i, new_dims[i]);
    }
    return GRAPH_SUCCESS;
}
