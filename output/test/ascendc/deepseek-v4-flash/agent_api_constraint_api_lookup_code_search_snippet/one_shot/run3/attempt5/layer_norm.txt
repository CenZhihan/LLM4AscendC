kernel_src='''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLen,
                                float invRowLen, float epsilon)
    {
        totalRows_ = totalRows;
        rowLen_ = rowLen;
        invRowLen_ = invRowLen;
        epsilon_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * rowLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, rowLen_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * rowLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, rowLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueGamma_, 1, rowLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueBeta_, 1, rowLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, rowLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, rowLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, 2 * sizeof(float));
        // workspace for ReduceSum: need at least rowLen_ elements (actual requirement may be smaller)
        pipe_.InitBuffer(wsQueue_, 1, rowLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowLen_ == 0) return;

        // Load gamma and beta once
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma_.AllocTensor<float>();
        AscendC::LocalTensor<float> betaLocal  = inQueueBeta_.AllocTensor<float>();
        AscendC::DataCopy(gammaLocal, gammaGm_, rowLen_);
        AscendC::DataCopy(betaLocal,  betaGm_,  rowLen_);
        inQueueGamma_.EnQue(gammaLocal);
        inQueueBeta_.EnQue(betaLocal);

        // Grid-stride over rows
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideRows = blockNum;
        uint32_t row = blockIdx;
        while (row < totalRows_) {
            ProcessRow(row, gammaLocal, betaLocal);
            row += strideRows;
        }

        inQueueGamma_.FreeTensor(gammaLocal);
        inQueueBeta_.FreeTensor(betaLocal);
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row,
                                      const AscendC::LocalTensor<float>& gammaLocal,
                                      const AscendC::LocalTensor<float>& betaLocal)
    {
        // Copy x row to local
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[row * rowLen_], rowLen_);

        // Scalar tensor for mean and variance (2 floats)
        AscendC::LocalTensor<float> scalar = scalarQueue_.AllocTensor<float>();

        // Workspace tensor for ReduceSum
        AscendC::LocalTensor<float> ws = wsQueue_.AllocTensor<float>();

        // 1. Compute mean = sum(x) * invRowLen
        AscendC::ReduceSum(scalar, xLocal, ws, rowLen_);
        float meanVal = scalar.GetValue(0) * invRowLen_;

        // 2. Compute centered = x - mean (use tmp queue)
        AscendC::LocalTensor<float> centered = tmpQueue_.AllocTensor<float>();
        AscendC::Adds(centered, xLocal, -meanVal, rowLen_);

        // 3. Compute variance = sum(centered^2) * invRowLen + epsilon
        AscendC::Mul(centered, centered, centered, rowLen_);   // centered becomes squared
        AscendC::ReduceSum(scalar[1], centered, ws, rowLen_);  // store variance sum in scalar[1]
        float varVal = (scalar.GetValue(1) * invRowLen_) + epsilon_;
        float invStd = 1.0f / sqrt(varVal);

        // 4. Normalize: y = (x - mean) * invStd * gamma + beta
        // Recompute centered from xLocal
        AscendC::Adds(centered, xLocal, -meanVal, rowLen_);
        AscendC::Muls(centered, centered, invStd, rowLen_);
        AscendC::Mul(centered, centered, gammaLocal, rowLen_);
        AscendC::Add(centered, centered, betaLocal, rowLen_);

        // 5. Copy out
        AscendC::DataCopy(yGm_[row * rowLen_], centered, rowLen_);

        // Free local tensors
        inQueueX_.FreeTensor(xLocal);
        tmpQueue_.FreeTensor(centered);
        scalarQueue_.FreeTensor(scalar);
        wsQueue_.FreeTensor(ws);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueGamma_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueBeta_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wsQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t rowLen_ {0};
    float invRowLen_ {0.0f};
    float epsilon_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y, tiling_data.totalRows, tiling_data.rowLen, tiling_data.invRowLen, tiling_data.epsilon);
    op.Process();
}
