#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core
static constexpr uint32_t ALIGN = 2048;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    auto *x_storage_shape = context->GetInputShape(0);
    const auto &x_origin_shape = x_storage_shape->GetOriginShape();
    const uint32_t ndim = x_origin_shape.GetDimNum();
    uint32_t batch = 1, dim1 = 1, dim2 = 1;
    if (ndim >= 1) batch = static_cast<uint32_t>(x_origin_shape.GetDim(0));
    if (ndim >= 2) dim1 = static_cast<uint32_t>(x_origin_shape.GetDim(1));
    if (ndim >= 3) dim2 = static_cast<uint32_t>(x_origin_shape.GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *attr_dim = attrs->GetAttrPointer<int32_t>(0);
    (void)attr_dim; // kernel assumes reduce on dim=1

    uint32_t totalOutput = batch * dim2;

    uint32_t blockDim = CeilDiv(totalOutput, ALIGN);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    uint32_t perBlock = CeilDiv(totalOutput, blockDim);
    perBlock = ((perBlock + ALIGN - 1) / ALIGN) * ALIGN;
    if (perBlock > 4096) perBlock = 4096;
    if (perBlock == 0) perBlock = 1;

    tiling.set_batch(batch);
    tiling.set_dim1(dim1);
    tiling.set_dim2(dim2);
    tiling.set_totalOutput(totalOutput);
    tiling.set_tileLen(perBlock);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t ndim = x_shape->GetDimNum();
    uint32_t outRank = ndim;
    if (ndim > 0 && ndim <= 3) {
        outRank = ndim - 1;
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(outRank);
    uint32_t outIdx = 0;
    for (uint32_t i = 0; i < ndim; ++i) {
        if (i != 1) {
            y_shape->SetDim(outIdx, x_shape->GetDim(i));
            ++outIdx;
        }
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t dim1, uint32_t dim2,
                                uint32_t totalOutput, uint32_t tileLen)
    {
        batch_  = batch;
        dim1_   = dim1;
        dim2_   = dim2;
        totalOutput_ = totalOutput;
        tileLen_ = tileLen;
        int32_t dim1Signed = static_cast<int32_t>(dim1_);
        invDim1_ = 1.0f / static_cast<float>(dim1Signed);

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        if (totalOutput_ == 0 || dim1_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        for (uint32_t i = blockIdx; i < totalOutput_; i += blockNum) {
            uint32_t n = i / dim2_;
            uint32_t h = i % dim2_;

            float sum = 0.0f;
            for (uint32_t c = 0; c < dim1_; c++) {
                uint32_t index = n * dim1_ * dim2_ + c * dim2_ + h;
                sum += xGm_.GetValue(index);
            }
            float mean = sum * invDim1_;
            yGm_.SetValue(i, mean);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t batch_   {0};
    uint32_t dim1_    {0};
    uint32_t dim2_    {0};
    uint32_t totalOutput_ {0};
    uint32_t tileLen_     {0};
    float    invDim1_     {0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y, td.batch, td.dim1, td.dim2, td.totalOutput, td.tileLen);
    op.Process();
}

#include <torch/library.h>

    int64_t ndim = self.dim();
    TORCH_CHECK(dim >= 0 && dim < ndim, "dim out of range");
    std::vector<int64_t> out_shape;

        if (i != dim) out_shape.push_back(self.size(i));

    at::Tensor result = at::empty(out_shape, self.options());
