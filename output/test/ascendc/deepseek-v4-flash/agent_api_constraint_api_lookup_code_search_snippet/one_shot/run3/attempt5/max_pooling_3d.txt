host_operator_src = """

#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;

    // ---- read input shape using StorageShape (TilingContext) ----
    const auto *inputStorageShape = context->GetInputShape(0);
    const auto &originShape = inputStorageShape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t D = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(3));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(4));

    // ---- read attributes ----
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kSize = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(2);

    uint32_t kd = static_cast<uint32_t>(kSize[0]);
    uint32_t kh = static_cast<uint32_t>(kSize[1]);
    uint32_t kw = static_cast<uint32_t>(kSize[2]);
    uint32_t sd = static_cast<uint32_t>(stride[0]);
    uint32_t sh = static_cast<uint32_t>(stride[1]);
    uint32_t sw = static_cast<uint32_t>(stride[2]);
    uint32_t pd = static_cast<uint32_t>(padding[0]);
    uint32_t ph = static_cast<uint32_t>(padding[1]);
    uint32_t pw = static_cast<uint32_t>(padding[2]);

    // ---- compute output dimensions (floor mode, ignoring ceil_mode for simplicity) ----
    uint32_t outD = (D + 2 * pd - kd) / sd + 1;
    uint32_t outH = (H + 2 * ph - kh) / sh + 1;
    uint32_t outW = (W + 2 * pw - kw) / sw + 1;

    uint32_t totalLength = N * C * outD * outH * outW;

    // ---- tile length based on UB budget ----
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;

    // ---- block dim ----
    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = (numTiles < 8) ? 8 : (numTiles > 24 ? 24 : numTiles);
    context->SetBlockDim(blockDim);

    // ---- fill tiling data ----
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_D(D);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_kd(kd);
    tiling.set_kh(kh);
    tiling.set_kw(kw);
    tiling.set_sd(sd);
    tiling.set_sh(sh);
    tiling.set_sw(sw);
    tiling.set_pd(pd);
    tiling.set_ph(ph);
    tiling.set_pw(pw);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const auto *attrs = context->GetAttrs();
    const int64_t *kSize = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(2);

    uint32_t D = static_cast<uint32_t>(xShape->GetDim(2));
    uint32_t H = static_cast<uint32_t>(xShape->GetDim(3));
    uint32_t W = static_cast<uint32_t>(xShape->GetDim(4));
    uint32_t kd = static_cast<uint32_t>(kSize[0]);
    uint32_t kh = static_cast<uint32_t>(kSize[1]);
    uint32_t kw = static_cast<uint32_t>(kSize[2]);
    uint32_t sd = static_cast<uint32_t>(stride[0]);
    uint32_t sh = static_cast<uint32_t>(stride[1]);
    uint32_t sw = static_cast<uint32_t>(stride[2]);
    uint32_t pd = static_cast<uint32_t>(padding[0]);
    uint32_t ph = static_cast<uint32_t>(padding[1]);
    uint32_t pw = static_cast<uint32_t>(padding[2]);

    uint32_t outD = (D + 2 * pd - kd) / sd + 1;
    uint32_t outH = (H + 2 * ph - kh) / sh + 1;
    uint32_t outW = (W + 2 * pw - kw) / sw + 1;

    gert::Shape *yShape = context->GetOutputShape(0);
    yShape->SetDim(0, xShape->GetDim(0));
    yShape->SetDim(1, xShape->GetDim(1));
    yShape->SetDim(2, static_cast<int64_t>(outD));
    yShape->SetDim(3, static_cast<int64_t>(outH));
    yShape->SetDim(4, static_cast<int64_t>(outW));
    return GRAPH_SUCCESS;
}

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t sd, uint32_t sh, uint32_t sw,
                                uint32_t pd, uint32_t ph, uint32_t pw)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N; C_ = C; D_ = D; H_ = H; W_ = W;
        outD_ = outD; outH_ = outH; outW_ = outW;
        kd_ = kd; kh_ = kh; kw_ = kw;
        sd_ = sd; sh_ = sh; sw_ = sw;
        pd_ = pd; ph_ = ph; pw_ = pw;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
            ComputeBlock(base, curLen, yLocal);
            outQueue_.EnQue<float>(yLocal);

            AscendC::LocalTensor<float> yOut = outQueue_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yOut, curLen);
            outQueue_.FreeTensor(yOut);

            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeBlock(uint32_t outOffset, uint32_t len,
                                         AscendC::LocalTensor<float>& yLocal)
    {
        for (uint32_t i = 0; i < len; i++) {
            uint32_t linearIdx = outOffset + i;
            uint32_t ow = linearIdx % outW_;
            uint32_t oh = (linearIdx / outW_) % outH_;
            uint32_t od = (linearIdx / (outW_ * outH_)) % outD_;
            uint32_t c = (linearIdx / (outW_ * outH_ * outD_)) % C_;
            uint32_t n = linearIdx / (C_ * outD_ * outH_ * outW_);

            int32_t dStart = static_cast<int32_t>(od * sd_) - static_cast<int32_t>(pd_);
            int32_t hStart = static_cast<int32_t>(oh * sh_) - static_cast<int32_t>(ph_);
            int32_t wStart = static_cast<int32_t>(ow * sw_) - static_cast<int32_t>(pw_);

            float maxVal = -__FLT_MAX__;
            for (uint32_t kd = 0; kd < kd_; kd++) {
                int32_t d = dStart + static_cast<int32_t>(kd);
                if (d < 0 || d >= static_cast<int32_t>(D_)) continue;
                for (uint32_t kh = 0; kh < kh_; kh++) {
                    int32_t h = hStart + static_cast<int32_t>(kh);
                    if (h < 0 || h >= static_cast<int32_t>(H_)) continue;
                    for (uint32_t kw = 0; kw < kw_; kw++) {
                        int32_t w = wStart + static_cast<int32_t>(kw);
                        if (w < 0 || w >= static_cast<int32_t>(W_)) continue;

                        uint32_t inIdx = ((n * C_ + c) * D_ + d) * H_ * W_ + h * W_ + w;
                        float val = xGm_.GetValue(inIdx);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }
            yLocal.SetValue(i, maxVal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t N_, C_, D_, H_, W_;
    uint32_t outD_, outH_, outW_;
    uint32_t kd_, kh_, kw_;
    uint32_t sd_, sh_, sw_;
    uint32_t pd_, ph_, pw_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.sd, tiling_data.sh, tiling_data.sw,
            tiling_data.pd, tiling_data.ph, tiling_data.pw);
    op.Process();
}
"""

# ===== model_src =====
model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size, stride=None, padding=0, return_indices=False, ceil_mode=False):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size if isinstance(kernel_size, tuple) else (kernel_size, kernel_size, kernel_size)
        self.stride = stride if stride is not None else self.kernel_size
        if isinstance(self.stride, int):
            self.stride = (self.stride, self.stride, self.stride)
        self.padding = padding if isinstance(padding, tuple) else (padding, padding, padding)
        self.return_indices = return_indices
        self.ceil_mode = ceil_mode

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding, self.return_indices, self.ceil_mode)
'''
