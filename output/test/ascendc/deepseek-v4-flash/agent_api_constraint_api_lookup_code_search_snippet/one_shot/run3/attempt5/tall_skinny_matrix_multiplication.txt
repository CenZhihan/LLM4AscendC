host_operator_src="""

#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::StorageShape *aStorageShape = context->GetInputShape(0);
    const gert::StorageShape *bStorageShape = context->GetInputShape(1);

    uint32_t M = static_cast<uint32_t>(aStorageShape->GetOriginShape().GetDim(0));
    uint32_t K = static_cast<uint32_t>(aStorageShape->GetOriginShape().GetDim(1));
    uint32_t N = static_cast<uint32_t>(bStorageShape->GetOriginShape().GetDim(1));

    // conservative block dim
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    // tiling parameters for float ND matmul
    uint32_t baseM = (M > 128) ? 128 : M;
    uint32_t baseN = (N > 128) ? 128 : N;
    uint32_t baseK = (K > 128) ? 128 : K;

    TallSkinnyMatrixMultiplicationCustomTilingData tiling;
    tiling.set_baseM(baseM);
    tiling.set_baseN(baseN);
    tiling.set_baseK(baseK);
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_aStride(K);
    tiling.set_bStride(N);
    tiling.set_cStride(N);
    tiling.set_aBlockStride(K * baseM);
    tiling.set_bBlockStride(N * baseK);
    tiling.set_cBlockStride(N * baseM);
    tiling.set_hasBias(false);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    uint32_t M = a_shape->GetDim(0);
    uint32_t N = b_shape->GetDim(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class TallSkinnyMatmulKernel {
public:
    __aicore__ inline TallSkinnyMatmulKernel() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, TCubeTiling tiling) {
        aGm_.SetGlobalBuffer((__gm__ float*)a);
        bGm_.SetGlobalBuffer((__gm__ float*)b);
        cGm_.SetGlobalBuffer((__gm__ float*)c);
        tiling_ = tiling;
    }
    __aicore__ inline void Process(AscendC::TPipe *pipe) {
        REGIST_MATMUL_OBJ(pipe, GetSysWorkSpacePtr(), matmulObj_, &tiling_);
        matmulObj_.Init(&tiling_);
        matmulObj_.SetTensorA(aGm_);
        matmulObj_.SetTensorB(bGm_);
        matmulObj_.SetTensorC(cGm_);
        matmulObj_.IterateAll();
    }
private:
    GlobalTensor<float> aGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> cGm_;
    TCubeTiling tiling_;
    Matmul<MatmulType<TPosition::GM, CubeFormat::ND, float, true>,
           MatmulType<TPosition::GM, CubeFormat::ND, float, true>,
           MatmulType<TPosition::GM, CubeFormat::ND, float>,
           MatmulType<TPosition::GM, CubeFormat::ND, float>> matmulObj_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                                                                GM_ADDR workspace, GM_ADDR tilingGm)
{
    if (ASCEND_IS_AIV) return;
    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);
    AscendC::TPipe pipe;
    TallSkinnyMatmulKernel matmulKernel;
    matmulKernel.Init(a, b, c, tiling);
    matmulKernel.Process(&pipe);
}

#include <torch/library.h>

at::Tensor tall_skinny_matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    int64_t M = A.size(0);
    int64_t N = B.size(1);
    at::Tensor result = at::empty({M, N}, A.options());
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, A, B, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu, "Tall-skinny matrix multiplication (TallSkinnyMatrixMultiplicationCustom on NPU)");
