host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_FLOATS = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const auto &input_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rank = input_shape.GetDimNum();
    std::vector<int64_t> dims(rank);
    for (uint32_t i = 0; i < rank; ++i) {
        dims[i] = input_shape.GetDim(i);
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += rank;
    uint32_t reduceDim = static_cast<uint32_t>(dim);

    uint32_t numRows = 1;
    uint32_t numCols = 1;
    for (uint32_t i = 0; i < rank; ++i) {
        uint32_t sz = static_cast<uint32_t>(dims[i]);
        if (i == reduceDim)
            numCols = sz;
        else
            numRows *= sz;
    }

    const uint32_t totalLength = numRows * numCols;

    // UB budget: double buffer input: 2*tileRows*numCols*4 ; double buffer output: 2*tileRows*4 ; tmp: numCols*4
    uint32_t bytesPerRowComponent = 8 * numCols + 8;
    uint32_t fixedBytes = 4 * numCols;
    uint32_t maxRows = (UB_BUDGET_BYTES - fixedBytes) / bytesPerRowComponent;
    if (maxRows == 0) maxRows = 1;
    uint32_t tileRows = AlignDown(maxRows, 8u);
    if (tileRows < 1) tileRows = 1;
    if (tileRows > 256) tileRows = 256;

    uint32_t blockDim = CeilDiv(numRows, tileRows);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_numRows(numRows);
    tiling.set_numCols(numCols);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    uint32_t rank = x_shape->GetDimNum();
    if (dim < 0) dim += rank;
    uint32_t reduceDim = static_cast<uint32_t>(dim);

    std::vector<int64_t> new_dims;
    for (uint32_t i = 0; i < rank; ++i) {
        if (i != reduceDim) {
            new_dims.push_back(x_shape->GetDim(i));
        }
    }
    y_shape->SetDimNum(new_dims.size());
    for (uint32_t i = 0; i < new_dims.size(); ++i) {
        y_shape->SetDim(i, new_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT32);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class ArgmaxOverADimensionCustom : public OpDef {
public:
    explicit ArgmaxOverADimensionCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT32})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);

    }
};

OP_ADD(ArgmaxOverADimensionCustom);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t numRows, uint32_t numCols, uint32_t tileRows)
    {
        numRows_ = numRows;
        numCols_ = numCols;
        tileRows_ = tileRows;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ int32_t*)y);

        uint32_t tileSizeFloats = tileRows_ * numCols_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileSizeFloats * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * sizeof(int32_t));
        pipe_.InitBuffer(tmpQueue_, 1, numCols_ * sizeof(float));
        pipe_.InitBuffer(smallResultQueue_, 1, 2 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (numRows_ == 0 || numCols_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t startRow = blockIdx * tileRows_;
        if (startRow >= numRows_) return;

        while (startRow < numRows_) {
            uint32_t remaining = numRows_ - startRow;
            uint32_t curRows = (remaining > tileRows_) ? tileRows_ : remaining;
            CopyIn(startRow, curRows);
            Compute(curRows, startRow);
            CopyOut(startRow, curRows);
            startRow += blockNum * tileRows_;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowBase, uint32_t rows)
    {
        uint32_t lenFloats = rows * numCols_;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowBase * numCols_], lenFloats);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rows, uint32_t rowBase)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<int32_t> yLocal = outQueueY_.AllocTensor<int32_t>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> smallResult = smallResultQueue_.AllocTensor<float>();

        for (uint32_t r = 0; r < rows; ++r) {
            AscendC::LocalTensor<float> rowLocal = xLocal[r * numCols_];
            // reduce with index: outputs {max_value, index as float}
            AscendC::ReduceMax(smallResult, rowLocal, tmpLocal, numCols_, true);
            float idxFloat = smallResult.GetValue(1);
            yLocal.SetValue(r, static_cast<int32_t>(idxFloat));
        }

        smallResultQueue_.FreeTensor(smallResult);
        tmpQueue_.FreeTensor(tmpLocal);
        outQueueY_.EnQue<int32_t>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowBase, uint32_t rows)
    {
        AscendC::LocalTensor<int32_t> yLocal = outQueueY_.DeQue<int32_t>();
        AscendC::DataCopy(yGm_[rowBase], yLocal, rows);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> smallResultQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int32_t> yGm_;
    uint32_t numRows_ {0};
    uint32_t numCols_ {0};
    uint32_t tileRows_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.numRows, tiling_data.numCols, tiling_data.tileRows);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include <vector>
#include "pytorch_npu_helper.hpp"

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    int64_t ndim = self.dim();
    if (dim < 0) dim += ndim;
    std::vector<int64_t> new_shape;
    for (int64_t i = 0; i < ndim; ++i) {
        if (i != dim) new_shape.push_back(self.size(i));
    }
    at::Tensor result = at::empty(new_shape, self.options().dtype(at::kInt));
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu, "Argmax over a dimension (ArgmaxOverADimensionCustom on NPU)");
