#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN = 32;                    // 32B alignment

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;
    const gert::StorageShape *anchor_shape = context->GetInputShape(0);
    uint32_t totalN = anchor_shape->GetOriginShape().GetDim(0);
    uint32_t totalD = anchor_shape->GetOriginShape().GetDim(1);

    // For each row we need temporary buffers: 4 rows (anchor, pos, neg, diff) + 1 sq buffer + 1 reduce sum buffer
    // Using local memory sized for one row at a time.
    uint32_t bytesPerRow = 5 * totalD * sizeof(float) + 16 * sizeof(float);
    uint32_t maxBatch = UB_BUDGET_BYTES / bytesPerRow;
    maxBatch = AlignDown(maxBatch, 1u);
    if (maxBatch == 0) maxBatch = 1;
    uint32_t tileBatch = std::min(totalN, maxBatch);

    uint32_t blockDim = 1; // single block for simplicity; can be increased for large n
    context->SetBlockDim(blockDim);

    tiling.set_totalN(totalN);
    tiling.set_totalD(totalD);
    tiling.set_tileBatch(tileBatch);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR margin, GM_ADDR y,
                                uint32_t totalN, uint32_t totalD, uint32_t tileBatch)
    {
        totalN_ = totalN;
        totalD_ = totalD;
        tileBatch_ = tileBatch;

        marginGm_.SetGlobalBuffer((__gm__ float*)margin, 1);
        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalN * totalD);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalN * totalD);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalN * totalD);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalN);

        // Allocate local buffers for one row of each input and temporary vectors
        pipe_.InitBuffer(anchorLoc_, BUFFER_NUM, totalD_ * sizeof(float));
        pipe_.InitBuffer(positiveLoc_, BUFFER_NUM, totalD_ * sizeof(float));
        pipe_.InitBuffer(negativeLoc_, BUFFER_NUM, totalD_ * sizeof(float));
        pipe_.InitBuffer(diffLoc_, BUFFER_NUM, totalD_ * sizeof(float));
        pipe_.InitBuffer(sqLoc_, BUFFER_NUM, totalD_ * sizeof(float));
        pipe_.InitBuffer(reduceLoc_, BUFFER_NUM, 16 * sizeof(float));  // 16 floats for reduction result (aligned)
        pipe_.InitBuffer(marginLoc_, BUFFER_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalN_ == 0 || totalD_ == 0) return;

        // Process all rows sequentially
        for (uint32_t row = 0; row < totalN_; row++) {
            // Copy input row from global to local
            AscendC::LocalTensor<float> anchorRow = anchorLoc_.AllocTensor<float>();
            AscendC::DataCopy(anchorRow, anchorGm_[row * totalD_], totalD_);
            anchorLoc_.EnQue(anchorRow);

            AscendC::LocalTensor<float> posRow = positiveLoc_.AllocTensor<float>();
            AscendC::DataCopy(posRow, positiveGm_[row * totalD_], totalD_);
            positiveLoc_.EnQue(posRow);

            AscendC::LocalTensor<float> negRow = negativeLoc_.AllocTensor<float>();
            AscendC::DataCopy(negRow, negativeGm_[row * totalD_], totalD_);
            negativeLoc_.EnQue(negRow);

            // Copy margin scalar (only once, but we do for each row for simplicity)
            AscendC::LocalTensor<float> marginVal = marginLoc_.AllocTensor<float>();
            AscendC::DataCopy(marginVal, marginGm_, 1);
            float m = marginVal.GetValue(0);
            marginLoc_.FreeTensor(marginVal);

            // Compute sq_ap = sum((anchor - positive)^2)
            AscendC::LocalTensor<float> a = anchorLoc_.DeQue<float>();
            AscendC::LocalTensor<float> p = positiveLoc_.DeQue<float>();
            AscendC::LocalTensor<float> diff = diffLoc_.AllocTensor<float>();
            AscendC::Sub(diff, a, p, totalD_);
            AscendC::LocalTensor<float> sq = sqLoc_.AllocTensor<float>();
            AscendC::Mul(sq, diff, diff, totalD_);
            AscendC::LocalTensor<float> red = reduceLoc_.AllocTensor<float>();
            AscendC::ReduceSum<float>(red, sq, diff, totalD_);
            float sq_ap = red.GetValue(0);
            reduceLoc_.FreeTensor(red);
            sqLoc_.FreeTensor(sq);
            diffLoc_.FreeTensor(diff);
            anchorLoc_.FreeTensor(a);
            positiveLoc_.FreeTensor(p);

            // Compute sq_an = sum((anchor - negative)^2)
            AscendC::LocalTensor<float> n = negativeLoc_.DeQue<float>();
            AscendC::LocalTensor<float> a2 = anchorLoc_.AllocTensor<float>();
            AscendC::DataCopy(a2, anchorGm_[row * totalD_], totalD_);  // re-copy anchor row
            AscendC::LocalTensor<float> diff2 = diffLoc_.AllocTensor<float>();
            AscendC::Sub(diff2, a2, n, totalD_);
            AscendC::LocalTensor<float> sq2 = sqLoc_.AllocTensor<float>();
            AscendC::Mul(sq2, diff2, diff2, totalD_);
            AscendC::LocalTensor<float> red2 = reduceLoc_.AllocTensor<float>();
            AscendC::ReduceSum<float>(red2, sq2, diff2, totalD_);
            float sq_an = red2.GetValue(0);
            reduceLoc_.FreeTensor(red2);
            sqLoc_.FreeTensor(sq2);
            diffLoc_.FreeTensor(diff2);
            anchorLoc_.FreeTensor(a2);
            negativeLoc_.FreeTensor(n);

            // Compute loss = max(sq_ap - sq_an + m, 0)
            float loss = sq_ap - sq_an + m;
            if (loss < 0.0f) loss = 0.0f;

            // Store to output global memory
            yGm_.SetValue(row, loss);
        }
    }

private:
    AscendC::TPipe pipe_;
    // Local tensor queues (size 1 for single buffering)
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> anchorLoc_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> positiveLoc_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> negativeLoc_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> diffLoc_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> sqLoc_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> reduceLoc_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> marginLoc_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> marginGm_;

    uint32_t totalN_ {0};
    uint32_t totalD_ {0};
    uint32_t tileBatch_ {0};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                 GM_ADDR margin, GM_ADDR y,
                                                                 GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, margin, y,
            tiling_data.totalN, tiling_data.totalD, tiling_data.tileBatch);
    op.Process();
}
