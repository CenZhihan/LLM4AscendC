kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class TallSkinnyMatmulKernel {
public:
    __aicore__ inline TallSkinnyMatmulKernel() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t blockDim)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        blockDim_ = blockDim;

        aGm_.SetGlobalBuffer((__gm__ float*)a, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, M_ * N_);

        // Compute cube tiling parameters from shape
        // For tall-skinny, base sizes limited by small dimensions
        uint32_t baseM = (M_ > 128) ? 128 : M_;
        uint32_t baseN = (N_ > 128) ? 128 : N_;
        uint32_t baseK = (K_ > 128) ? 128 : K_;

        tiling_.set_baseM(baseM);
        tiling_.set_baseN(baseN);
        tiling_.set_baseK(baseK);
        tiling_.set_M(M_);
        tiling_.set_N(N_);
        tiling_.set_K(K_);
        tiling_.set_aStride(K_);
        tiling_.set_bStride(N_);
        tiling_.set_cStride(N_);
        tiling_.set_aBlockStride(K_ * baseM);
        tiling_.set_bBlockStride(N_ * baseK);
        tiling_.set_cBlockStride(N_ * baseM);
        tiling_.set_hasBias(false);

        // Initialize matmul object
        REGIST_MATMUL_OBJ(&pipe_, GetSysWorkSpacePtr(), matmulObj_, &tiling_);
    }

    __aicore__ inline void Process()
    {
        matmulObj_.Init(&tiling_);
        matmulObj_.SetTensorA(aGm_);
        matmulObj_.SetTensorB(bGm_);
        matmulObj_.SetTensorC(cGm_);
        matmulObj_.Process();
    }

private:
    TPipe pipe_;
    GlobalTensor<float> aGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> cGm_;
    TCubeTiling tiling_;
    Matmul<float, float, float, float> matmulObj_;
    uint32_t M_, N_, K_, blockDim_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    TallSkinnyMatmulKernel op;
    op.Init(a, b, c, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.blockDim);
    op.Process();
}

    at::Tensor result = at::empty({M, N}, A.options());

    return result;
