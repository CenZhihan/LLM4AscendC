#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape *origShape = &(context->GetInputShape(0)->GetOriginShape());
    uint32_t dimNum = origShape->GetDimNum();
    uint32_t outer = 1;
    uint32_t inner = 1;
    if (dimNum >= 2) {
        inner = static_cast<uint32_t>(origShape->GetDim(dimNum - 1));
        for (uint32_t i = 0; i < dimNum - 1; ++i) {
            outer *= static_cast<uint32_t>(origShape->GetDim(i));
        }
    } else if (dimNum == 1) {
        inner = static_cast<uint32_t>(origShape->GetDim(0));
        outer = 1;
    } else {
        uint32_t total = static_cast<uint32_t>(origShape->GetShapeSize());
        inner = total;
        outer = 1;
    }
    uint32_t totalLength = outer * inner;

    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (inner < tileLen) {
        const uint32_t aligned = AlignDown(inner, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : inner;
    }

    uint32_t blockDim = outer;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 4u) blockDim = 4u;
    if (blockDim > outer) blockDim = outer;
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outer(outer);
    tiling.set_inner(inner);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outer, uint32_t inner,
                                uint32_t tileLength)
    {
        totalLen_ = totalLength;
        outer_ = outer;
        inner_ = inner;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(workspaceQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t rowsPerBlock = (outer_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > outer_) rowEnd = outer_;
        if (rowStart >= outer_) return;

        uint32_t numRowsLocal = rowEnd - rowStart;
        if (numRowsLocal == 0) return;

        AscendC::LocalTensor<float> rowMaxLocal;
        AscendC::LocalTensor<float> rowSumLocal;
        uint32_t rowBufferBytes = numRowsLocal * sizeof(float);
        pipe_.InitBuffer(rowMaxBuf_, rowBufferBytes);
        pipe_.InitBuffer(rowSumBuf_, rowBufferBytes);
        rowMaxLocal = rowMaxBuf_.AllocTensor<float>();
        rowSumLocal = rowSumBuf_.AllocTensor<float>();

        // Pass 1: compute row max
        for (uint32_t r = 0; r < numRowsLocal; ++r) {
            uint32_t rowIdx = rowStart + r;
            uint32_t rowBase = rowIdx * inner_;
            float rowMaxVal = -1e38f;

            uint32_t tiles = (inner_ + tileLen_ - 1) / tileLen_;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = rowBase + t * tileLen_;
                uint32_t curLen = (inner_ - t * tileLen_ < tileLen_) ? (inner_ - t * tileLen_) : tileLen_;
                if (curLen == 0) break;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curLen);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> workspace = workspaceQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> maxScalar = tmpQueue_.AllocTensor<float>();

                AscendC::ReduceMax(maxScalar, xTile, workspace, curLen);
                float tileMax = maxScalar(0);
                if (tileMax > rowMaxVal) rowMaxVal = tileMax;

                tmpQueue_.FreeTensor(maxScalar);
                workspaceQueue_.FreeTensor(workspace);
                inQueueX_.FreeTensor(xTile);
            }
            rowMaxLocal(r) = rowMaxVal;
        }

        // Pass 2: compute sum of exp(x - max)
        for (uint32_t r = 0; r < numRowsLocal; ++r) {
            uint32_t rowIdx = rowStart + r;
            uint32_t rowBase = rowIdx * inner_;
            float rowMax = rowMaxLocal(r);
            float rowSumVal = 0.0f;

            uint32_t tiles = (inner_ + tileLen_ - 1) / tileLen_;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = rowBase + t * tileLen_;
                uint32_t curLen = (inner_ - t * tileLen_ < tileLen_) ? (inner_ - t * tileLen_) : tileLen_;
                if (curLen == 0) break;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curLen);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> workspace = workspaceQueue_.AllocTensor<float>();

                // x - rowMax → tmp
                AscendC::Adds(tmp, xTile, -rowMax, curLen);
                AscendC::Exp(tmp, tmp, curLen);

                AscendC::LocalTensor<float> sumScalar = outQueueY_.AllocTensor<float>();
                AscendC::ReduceSum(sumScalar, tmp, workspace, curLen);
                rowSumVal += sumScalar(0);

                outQueueY_.FreeTensor(sumScalar);
                workspaceQueue_.FreeTensor(workspace);
                tmpQueue_.FreeTensor(tmp);
                inQueueX_.FreeTensor(xTile);
            }
            rowSumLocal(r) = rowSumVal;
        }

        // Pass 3: compute log_softmax output
        for (uint32_t r = 0; r < numRowsLocal; ++r) {
            uint32_t rowIdx = rowStart + r;
            uint32_t rowBase = rowIdx * inner_;
            float rowMax = rowMaxLocal(r);
            float rowSum = rowSumLocal(r);
            // Compute log(rowSum) using AscendC::Log on a 1‑element tensor
            AscendC::LocalTensor<float> logScalar = tmpQueue_.AllocTensor<float>();
            logScalar(0) = rowSum;
            AscendC::Log(logScalar, logScalar, 1);
            float logSum = logScalar(0);
            tmpQueue_.FreeTensor(logScalar);

            uint32_t tiles = (inner_ + tileLen_ - 1) / tileLen_;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = rowBase + t * tileLen_;
                uint32_t curLen = (inner_ - t * tileLen_ < tileLen_) ? (inner_ - t * tileLen_) : tileLen_;
                if (curLen == 0) break;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curLen);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // yLocal = xTile - rowMax
                AscendC::Adds(yLocal, xTile, -rowMax, curLen);
                // yLocal = yLocal - logSum
                AscendC::Adds(yLocal, yLocal, -logSum, curLen);

                AscendC::DataCopy(yGm_[offset], yLocal, curLen);

                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xTile);
            }
        }

        rowMaxBuf_.FreeTensor(rowMaxLocal);
        rowSumBuf_.FreeTensor(rowSumLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workspaceQueue_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> rowMaxBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> rowSumBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outer_ {0};
    uint32_t inner_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.outer, tiling_data.inner, tiling_data.tileLength);
    op.Process();
}
