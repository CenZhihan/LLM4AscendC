#include "tall_skinny_matrix_multiplication_custom_tiling.h"

namespace optiling {

static constexpr uint32_t FULL_L1_SIZE = 0xFFFFFFFF;
static constexpr uint32_t FULL_L0C_SIZE = 0xFFFFFFFF;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const auto *shapeA = context->GetInputShape(0);
    const auto *shapeB = context->GetInputShape(1);
    uint32_t M = static_cast<uint32_t>(shapeA->GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA->GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB->GetDim(1));

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_shareMode(0);
    tiling.set_shareL1Size(FULL_L1_SIZE);
    tiling.set_shareL0CSize(FULL_L0C_SIZE);
    tiling.set_shareUbSize(0);

    // heuristic blockDim: limit to e.g. 8 for simplicity
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)

    shapeC->SetDim(1, shapeB->GetDim(1));

    const ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}

#include "kernel_operator.h"

using namespace AscendC;

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                                                                GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling_data, tilingGm);

    TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulObj, nullptr);

    // Set matmul tiling using a single API call if available; for simplicity use constant parameters.
    // For production, consider using MultiCoreMatmulTiling.
    using A_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float>;
    using B_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float>;
    using C_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float>;
    using BIAS_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float>;
    constexpr auto CFG = GetCustomConstantCFG<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE>();
    Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE, CFG> matmulObj;

    matmulObj.Init(a, b, static_cast<__gm__ float*>(nullptr), c, tilingGm);
    matmulObj.Process(&pipe);
}
