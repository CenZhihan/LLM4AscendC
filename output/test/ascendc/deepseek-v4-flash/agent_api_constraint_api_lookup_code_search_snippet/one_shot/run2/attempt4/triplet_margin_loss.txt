kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
                                uint32_t batchSize, uint32_t dim, float margin, float batchSizeFloat)
    {
        batchSize_ = batchSize;
        dim_ = dim;
        margin_ = margin;
        batchSizeFloat_ = batchSizeFloat;
        totalElems_ = batchSize * dim;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalElems_);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalElems_);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalElems_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inQueueAnchor_, BUFFER_NUM, dim_ * sizeof(float));
        pipe_.InitBuffer(inQueuePositive_, BUFFER_NUM, dim_ * sizeof(float));
        pipe_.InitBuffer(inQueueNegative_, BUFFER_NUM, dim_ * sizeof(float));

        // Workspace for per-row computation
        pipe_.InitBuffer(tmpBufDiff_, 1, dim_ * sizeof(float));
        pipe_.InitBuffer(tmpBufSum_, 1, sizeof(float));        // scalar sum
        pipe_.InitBuffer(tmpBufWs_, 1, dim_ * sizeof(float)); // workspace for ReduceSum
        pipe_.InitBuffer(scalarBuf_, 1, sizeof(float));
        pipe_.InitBuffer(totalLossBuf_, 1, sizeof(float));    // persistent accumulator (TBuf)
    }

    __aicore__ inline void Process()
    {
        // Initialize totalLoss to 0 using TBuf
        AscendC::LocalTensor<float> totalLoss = totalLossBuf_.AllocTensor<float>();
        totalLoss.SetValue(0, 0.0f);

        for (uint32_t i = 0; i < batchSize_; ++i) {
            uint32_t offset = i * dim_;

            // --- Load anchor, positive, negative ---
            AscendC::LocalTensor<float> anchorLocal = inQueueAnchor_.AllocTensor<float>();
            AscendC::DataCopy(anchorLocal, anchorGm_[offset], dim_);
            AscendC::LocalTensor<float> positiveLocal = inQueuePositive_.AllocTensor<float>();
            AscendC::DataCopy(positiveLocal, positiveGm_[offset], dim_);
            AscendC::LocalTensor<float> negativeLocal = inQueueNegative_.AllocTensor<float>();
            AscendC::DataCopy(negativeLocal, negativeGm_[offset], dim_);

            // --- Compute distPos = sqrt(sum((anchor-positive)^2)) ---
            AscendC::LocalTensor<float> diffPos = tmpBufDiff_.AllocTensor<float>();
            AscendC::Sub(diffPos, anchorLocal, positiveLocal, dim_);
            AscendC::Mul(diffPos, diffPos, diffPos, dim_);

            AscendC::LocalTensor<float> sumPos = tmpBufSum_.AllocTensor<float>();
            AscendC::LocalTensor<float> wsPos = tmpBufWs_.AllocTensor<float>();
            AscendC::ReduceSum(sumPos, diffPos, wsPos, dim_ / 8);
            AscendC::Sqrt(sumPos, sumPos, 1);
            // Free diff and ws, but keep sumPos alive
            tmpBufDiff_.FreeTensor(diffPos);
            tmpBufWs_.FreeTensor(wsPos);

            // --- Compute distNeg = sqrt(sum((anchor-negative)^2)) ---
            AscendC::LocalTensor<float> diffNeg = tmpBufDiff_.AllocTensor<float>();
            AscendC::Sub(diffNeg, anchorLocal, negativeLocal, dim_);
            AscendC::Mul(diffNeg, diffNeg, diffNeg, dim_);

            AscendC::LocalTensor<float> sumNeg = tmpBufSum_.AllocTensor<float>();
            AscendC::LocalTensor<float> wsNeg = tmpBufWs_.AllocTensor<float>();
            AscendC::ReduceSum(sumNeg, diffNeg, wsNeg, dim_ / 8);
            AscendC::Sqrt(sumNeg, sumNeg, 1);
            tmpBufDiff_.FreeTensor(diffNeg);
            tmpBufWs_.FreeTensor(wsNeg);

            // --- loss = max(distPos - distNeg + margin, 0) ---
            AscendC::LocalTensor<float> lossVal = totalLossBuf_.AllocTensor<float>(); // reuse same buffer? No, use scalarBuf_
            // Actually we need a separate tensor for lossVal; use scalarBuf_ temporarily
            AscendC::LocalTensor<float> lossVal = scalarBuf_.AllocTensor<float>();
            AscendC::Sub(lossVal, sumPos, sumNeg, 1);  // sumPos and sumNeg are scalar (length 1)
            // Add margin
            AscendC::LocalTensor<float> marginTensor = scalarBuf_.AllocTensor<float>();
            marginTensor.SetValue(0, margin_);
            AscendC::Add(lossVal, lossVal, marginTensor, 1);
            // Max with 0
            AscendC::LocalTensor<float> zeroTensor = scalarBuf_.AllocTensor<float>();
            zeroTensor.SetValue(0, 0.0f);
            AscendC::Max(lossVal, lossVal, zeroTensor, 1);
            // Accumulate totalLoss
            AscendC::Add(totalLoss, totalLoss, lossVal, 1);

            // Free all per-row tensors
            scalarBuf_.FreeTensor(zeroTensor);
            scalarBuf_.FreeTensor(marginTensor);
            scalarBuf_.FreeTensor(lossVal);
            tmpBufSum_.FreeTensor(sumPos);
            tmpBufSum_.FreeTensor(sumNeg);
            inQueueAnchor_.FreeTensor(anchorLocal);
            inQueuePositive_.FreeTensor(positiveLocal);
            inQueueNegative_.FreeTensor(negativeLocal);
        }

        // Compute mean loss = totalLoss / batchSize
        // Use a scalar tensor with precomputed batchSizeFloat (no cast in kernel)
        AscendC::LocalTensor<float> batchSizeTensor = scalarBuf_.AllocTensor<float>();
        batchSizeTensor.SetValue(0, batchSizeFloat_);
        AscendC::Div(totalLoss, totalLoss, batchSizeTensor, 1);
        scalarBuf_.FreeTensor(batchSizeTensor);

        // Write result to global memory
        AscendC::DataCopy(lossGm_[0], totalLoss, 1);

        // Free persistent totalLoss tensor
        totalLossBuf_.FreeTensor(totalLoss);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePositive_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNegative_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBufDiff_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBufSum_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBufWs_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> scalarBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> totalLossBuf_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t batchSize_;
    uint32_t dim_;
    uint32_t totalElems_;
    float margin_;
    float batchSizeFloat_;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.batchSize, tiling_data.dim, tiling_data.margin, tiling_data.batchSizeFloat);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'triplet_margin_loss_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, *inputs):
        fn = getattr(custom_ops_lib, _FN)
        anchor, positive, negative, margin = inputs
        margin_scalar = float(margin.reshape(-1)[0].item()) if margin.numel() else 0.0
        return fn(anchor, positive, negative, margin_scalar)
