#include "layer_norm_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const auto& xShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t aLength = static_cast<uint32_t>(xShape.GetDim(0));
    const uint32_t rLength = static_cast<uint32_t>(xShape.GetDim(1));

    constexpr uint32_t typeSize = sizeof(float);
    int32_t alignElems = 32 / typeSize;
    uint32_t rLengthWithPadding = (rLength + alignElems - 1) / alignElems * alignElems;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilon = attrs->GetAttrPointer<float>(0);

    LayerNormCustomTilingData tiling;
    tiling.set_aLength(aLength);
    tiling.set_rLength(rLength);
    tiling.set_rLengthWithPadding(rLengthWithPadding);
    tiling.set_epsilon(*epsilon);

    context->SetBlockDim(1);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"
#include "layer_norm_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t aLength, uint32_t rLength,
                                uint32_t rLengthWithPadding, float epsilon,
                                optiling::LayerNormCustomTilingData *tilingData,
                                AscendC::TPipe *pipe)
    {
        pipe_ = pipe;
        aLength_ = aLength;
        rLength_ = rLength;
        rLengthWithPadding_ = rLengthWithPadding;
        epsilon_ = epsilon;
        tilingData_ = tilingData;

        blockLength_ = aLength_;
        meanVarLength_ = aLength_;

        inputXGlobal.SetGlobalBuffer((__gm__ float*)x, aLength_ * rLength_);
        gammaGlobal.SetGlobalBuffer((__gm__ float*)gamma, rLength_);
        betaGlobal.SetGlobalBuffer((__gm__ float*)beta, rLength_);
        outputGlobal.SetGlobalBuffer((__gm__ float*)y, aLength_ * rLength_);

        pipe_->InitBuffer(inQueueX, BUFFER_NUM, rLength_ * sizeof(float));
        pipe_->InitBuffer(inQueueGamma, 1, rLength_ * sizeof(float));
        pipe_->InitBuffer(inQueueBeta, 1, rLength_ * sizeof(float));
        pipe_->InitBuffer(outQueue, BUFFER_NUM, rLength_ * sizeof(float));
        pipe_->InitBuffer(outQueueMean, 1, meanVarLength_ * sizeof(float));
        pipe_->InitBuffer(outQueueRstd, 1, meanVarLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma.AllocTensor<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueBeta.AllocTensor<float>();
        AscendC::DataCopy(gammaLocal, gammaGlobal, rLength_);
        AscendC::DataCopy(betaLocal, betaGlobal, rLength_);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);

        for (uint32_t i = 0; i < aLength_; ++i)
        {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }

        inQueueGamma.FreeTensor(gammaLocal);
        inQueueBeta.FreeTensor(betaLocal);
    }

private:
    __aicore__ inline void CopyIn(uint32_t batchIdx)
    {
        uint32_t offset = batchIdx * rLength_;
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::DataCopy(xLocal, inputXGlobal[offset], rLength_);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t batchIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma.DeQue<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueBeta.DeQue<float>();
        AscendC::LocalTensor<float> outputLocal = outQueue.AllocTensor<float>();
        AscendC::LocalTensor<float> meanLocal = outQueueMean.AllocTensor<float>();
        AscendC::LocalTensor<float> rstdLocal = outQueueRstd.AllocTensor<float>();

        AscendC::LayerNorm<float>(outputLocal, meanLocal, rstdLocal,
                                   xLocal, gammaLocal, betaLocal,
                                   epsilon_, tilingData_);

        outQueue.EnQue<float>(outputLocal);
        outQueueMean.FreeTensor(meanLocal);
        outQueueRstd.FreeTensor(rstdLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueGamma.FreeTensor(gammaLocal);
        inQueueBeta.FreeTensor(betaLocal);
    }

    __aicore__ inline void CopyOut(uint32_t batchIdx)
    {
        uint32_t offset = batchIdx * rLength_;
        AscendC::LocalTensor<float> outputLocal = outQueue.DeQue<float>();
        AscendC::DataCopy(outputGlobal[offset], outputLocal, rLength_);
        outQueue.FreeTensor(outputLocal);
    }

private:
    AscendC::TPipe* pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueGamma;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueBeta;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueMean;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueRstd;

    AscendC::GlobalTensor<float> inputXGlobal;
    AscendC::GlobalTensor<float> gammaGlobal;
    AscendC::GlobalTensor<float> betaGlobal;
    AscendC::GlobalTensor<float> outputGlobal;

    uint32_t aLength_;
    uint32_t rLength_;
    uint32_t rLengthWithPadding_;
    float epsilon_;
    uint32_t blockLength_;
    uint32_t meanVarLength_{0};
    optiling::LayerNormCustomTilingData *tilingData_;
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta,
                                                        GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    AscendC::TPipe pipe;
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.aLength,
            tiling_data.rLength,
            tiling_data.rLengthWithPadding,
            tiling_data.epsilon,
            &tiling_data,
            &pipe);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor layer_norm_impl_npu(const at::Tensor& self,
                                const at::Tensor& gamma,
                                const at::Tensor& beta,
                                double epsilon = 1e-5) {
    auto self_sizes = self.sizes();
    int64_t ndim = self_sizes.size();
    TORCH_CHECK(ndim >= 1, "LayerNorm requires at least 1D input");
    int64_t batch = self_sizes[0];
    int64_t feat = 1;
    for (int64_t i = 1; i < ndim; ++i) feat *= self_sizes[i];

    at::Tensor self_flat = self.contiguous().view({batch, feat});
    at::Tensor gamma_flat = gamma.contiguous().view({feat});
    at::Tensor beta_flat = beta.contiguous().view({feat});
    at::Tensor result = at::empty_like(self_flat);

    EXEC_NPU_CMD(aclnnLayerNormCustom, self_flat, gamma_flat, beta_flat, epsilon, result);

    return result.view(self.sizes());
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("layer_norm_custom", &layer_norm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("layer_norm_custom", &layer_norm_impl_npu, "Layer Normalization (LayerNormCustom on NPU)");
}

import torch
import torch.nn as nn

import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        self.weight = nn.Parameter(torch.ones(normalized_shape, dtype=torch.float32))
        self.bias = nn.Parameter(torch.zeros(normalized_shape, dtype=torch.float32))
        self.epsilon = 1e-5

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias, self.epsilon)
