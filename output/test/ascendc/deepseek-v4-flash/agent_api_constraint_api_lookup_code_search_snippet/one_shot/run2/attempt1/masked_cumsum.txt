#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;   // alignment for tileOuter
static constexpr uint32_t MAX_BLOCK_DIM = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaskedCumsumCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t outer = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t inner = static_cast<uint32_t>(xShape->GetDim(1));
    uint32_t totalLength = outer * inner;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimPtr != nullptr) ? *dimPtr : -1;

    // UB budget per row for x, mask, y, sharedTmp, lastRow (approximate)
    // sharedTmp ~ 2*inner bytes, lastRow ~ outer*sizeof(float) (small)
    uint32_t bytesPerRow = inner * sizeof(float) * 3 + inner * 2 + outer * sizeof(float);
    uint32_t maxRows = (UB_BUDGET_BYTES / bytesPerRow);
    if (maxRows == 0) maxRows = 1;
    uint32_t tileOuter = (maxRows > outer) ? outer : AlignDown(maxRows, 1);
    if (tileOuter < 1) tileOuter = 1;

    uint32_t blockDim = CeilDiv(outer, tileOuter);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outer(outer);
    tiling.set_inner(inner);
    tiling.set_tileOuter(tileOuter);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    return GRAPH_SUCCESS;

    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);

}

kernel_src = '''

#include "kernel_operator.h"
#include "adv_api/math/cumsum.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t outer, uint32_t inner,
                                uint32_t tileOuter, uint32_t blockDim, int32_t dim)
    {
        totalLen_ = totalLength;
        outer_ = outer;
        inner_ = inner;
        tileOuter_ = tileOuter;
        gmOuter_ = outer;
        gmInner_ = inner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Shared buffer for CumSum (size heuristic: 2*inner floats)
        uint32_t sharedTmpBytes = ((inner_ * 2 * sizeof(float)) + 31) & ~31;
        pipe_.InitBuffer(inQueueX_, 1, tileOuter_ * inner_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, 1, tileOuter_ * inner_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileOuter_ * inner_ * sizeof(float));
        pipe_.InitBuffer(sharedTmpQueue_, 1, sharedTmpBytes);
        pipe_.InitBuffer(lastRowQueue_, 1, tileOuter_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t strideRows = blockNum * tileOuter_;
        uint32_t startRow = blockIdx * tileOuter_;
        if (startRow >= outer_) return;

        while (true) {
            uint32_t tileRows = tileOuter_;
            if (startRow + tileRows > outer_)
                tileRows = outer_ - startRow;

            LoadTile(startRow, tileRows);
            ComputeTile(tileRows);
            StoreTile(startRow, tileRows);

            startRow += strideRows;
            if (startRow >= outer_) break;
        }
    }

private:
    __aicore__ inline void LoadTile(uint32_t startRow, uint32_t tileRows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
        for (uint32_t r = 0; r < tileRows; r++) {
            uint32_t gmOffset = (startRow + r) * gmInner_;
            uint32_t locOffset = r * inner_;
            AscendC::DataCopyPad(xLocal[locOffset], xGm_[gmOffset],
                                 {1, static_cast<uint16_t>(gmInner_ * sizeof(float)), 0, 0});
            AscendC::DataCopyPad(maskLocal[locOffset], maskGm_[gmOffset],
                                 {1, static_cast<uint16_t>(gmInner_ * sizeof(float)), 0, 0});
        }
        inQueueX_.EnQue(xLocal);
        inQueueMask_.EnQue(maskLocal);
    }

    __aicore__ inline void ComputeTile(uint32_t tileRows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maskLocal = inQueueMask_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> sharedTmp = sharedTmpQueue_.AllocTensor<uint8_t>();
        AscendC::LocalTensor<float> lastRowLocal = lastRowQueue_.AllocTensor<float>();

        // Multiply x * mask
        AscendC::Mul(yLocal, xLocal, maskLocal, tileRows * inner_);

        // Cumulative sum along last dimension (inner)
        AscendC::CumSumInfo cumSumInfo{tileRows, inner_};
        AscendC::CumSum<float>(yLocal, lastRowLocal, yLocal, sharedTmp, cumSumInfo);

        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueMask_.FreeTensor(maskLocal);
        sharedTmpQueue_.FreeTensor(sharedTmp);
        lastRowQueue_.FreeTensor(lastRowLocal);
    }

    __aicore__ inline void StoreTile(uint32_t startRow, uint32_t tileRows)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        for (uint32_t r = 0; r < tileRows; r++) {
            uint32_t gmOffset = (startRow + r) * gmInner_;
            uint32_t locOffset = r * inner_;
            AscendC::DataCopyPad(yGm_[gmOffset], yLocal[locOffset],
                                 {1, static_cast<uint16_t>(gmInner_ * sizeof(float)), 0, 0});
        }
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueMask_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> sharedTmpQueue_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> lastRowQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outer_ {0};
    uint32_t inner_ {0};
    uint32_t tileOuter_ {0};
    uint32_t gmOuter_ {0};
    uint32_t gmInner_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.totalLength,
            tiling_data.outer,
            tiling_data.inner,
            tiling_data.tileOuter,
            tiling_data.blockDim,
            tiling_data.dim);
    op.Process();
}
'''
