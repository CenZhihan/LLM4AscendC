kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
                                uint32_t batchSize, uint32_t dim, float margin)
    {
        batchSize_ = batchSize;
        dim_ = dim;
        margin_ = margin;
        totalElems_ = batchSize * dim;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalElems_);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalElems_);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalElems_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        // Allocate ping-pong buffers for each input row
        pipe_.InitBuffer(inQueueAnchor_, BUFFER_NUM, dim_ * sizeof(float));
        pipe_.InitBuffer(inQueuePositive_, BUFFER_NUM, dim_ * sizeof(float));
        pipe_.InitBuffer(inQueueNegative_, BUFFER_NUM, dim_ * sizeof(float));

        // Workspace for distance computation per row
        pipe_.InitBuffer(tmpBuf1_, 1, dim_ * sizeof(float)); // diff = anchor - positive
        pipe_.InitBuffer(tmpBuf2_, 1, dim_ * sizeof(float)); // diff_sq = diff * diff
        // Reuse tmpBuf2 for negative branch
        pipe_.InitBuffer(tmpBuf3_, 1, 1 * sizeof(float));   // reduce sum result
        pipe_.InitBuffer(tmpBuf4_, 1, dim_ * sizeof(float)); // diff_neg
        pipe_.InitBuffer(tmpBuf5_, 1, dim_ * sizeof(float)); // diff_neg_sq
        pipe_.InitBuffer(outBuf_, 1, sizeof(float));          // final loss scalar
    }

    __aicore__ inline void Process()
    {
        float totalLoss = 0.0f;

        for (uint32_t i = 0; i < batchSize_; ++i) {
            uint32_t offset = i * dim_;

            // Load anchor, positive, negative rows
            AscendC::LocalTensor<float> anchorLocal = inQueueAnchor_.AllocTensor<float>();
            AscendC::DataCopy(anchorLocal, anchorGm_[offset], dim_);
            inQueueAnchor_.EnQue(anchorLocal);

            AscendC::LocalTensor<float> positiveLocal = inQueuePositive_.AllocTensor<float>();
            AscendC::DataCopy(positiveLocal, positiveGm_[offset], dim_);
            inQueuePositive_.EnQue(positiveLocal);

            AscendC::LocalTensor<float> negativeLocal = inQueueNegative_.AllocTensor<float>();
            AscendC::DataCopy(negativeLocal, negativeGm_[offset], dim_);
            inQueueNegative_.EnQue(negativeLocal);

            // Compute distance pos = sqrt( sum((anchor-positive)^2) )
            AscendC::LocalTensor<float> diffPos = tmpBuf1_.Get<float>();
            AscendC::Sub(diffPos, anchorLocal, positiveLocal, dim_);  // diff = anchor - positive
            AscendC::Mul(diffPos, diffPos, diffPos, dim_);            // diff_sq = diff * diff
            AscendC::LocalTensor<float> sumPos = tmpBuf3_.Get<float>();
            AscendC::ReduceSum(sumPos, diffPos, tmpBuf3_, dim_ / 8);  // reduce, note: workspace = tmpBuf3_ reused
            AscendC::Sqrt(sumPos, sumPos, 1);                         // dist_pos = sqrt(sum)

            // Compute distance neg = sqrt( sum((anchor-negative)^2) )
            AscendC::LocalTensor<float> diffNeg = tmpBuf4_.Get<float>();
            AscendC::Sub(diffNeg, anchorLocal, negativeLocal, dim_);
            AscendC::Mul(diffNeg, diffNeg, diffNeg, dim_);
            AscendC::LocalTensor<float> sumNeg = tmpBuf3_.Get<float>(); // reuse workspace for second reduce
            AscendC::ReduceSum(sumNeg, diffNeg, tmpBuf3_, dim_ / 8);
            AscendC::Sqrt(sumNeg, sumNeg, 1);

            // loss = max(dist_pos - dist_neg + margin, 0)
            AscendC::LocalTensor<float> lossVal = outBuf_.Get<float>();
            AscendC::Sub(lossVal, sumPos, sumNeg, 1);         // dist_pos - dist_neg
            AscendC::Add(lossVal, lossVal, margin_, 1);        // + margin
            AscendC::Max(lossVal, lossVal, 0.0f, 1);          // max(., 0)

            // Accumulate to totalLoss
            // Copy lossVal scalar to a local variable; using DataCopy to a scalar cast? Use SetValue
            float localLoss;
            AscendC::DataCopy((__local__ float*)&localLoss, lossVal, 1);
            totalLoss += localLoss;

            // Free tensors
            inQueueAnchor_.FreeTensor(anchorLocal);
            inQueuePositive_.FreeTensor(positiveLocal);
            inQueueNegative_.FreeTensor(negativeLocal);
        }

        // Output mean loss
        float meanLoss = totalLoss / static_cast<float>(batchSize_);
        AscendC::LocalTensor<float> outScalar = outBuf_.Get<float>();
        outScalar.SetValue(0, meanLoss);
        AscendC::DataCopy(lossGm_[0], outScalar, 1);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePositive_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNegative_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf3_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf4_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf5_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outBuf_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t batchSize_;
    uint32_t dim_;
    uint32_t totalElems_;
    float margin_;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.batchSize, tiling_data.dim, tiling_data.margin);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'triplet_margin_loss_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, *inputs):
        fn = getattr(custom_ops_lib, _FN)
        # inputs: anchor, positive, negative, margin (scalar tensor)
        anchor, positive, negative, margin = inputs
        margin_scalar = float(margin.reshape(-1)[0].item()) if margin.numel() else 0.0
        return fn(anchor, positive, negative, margin_scalar)
