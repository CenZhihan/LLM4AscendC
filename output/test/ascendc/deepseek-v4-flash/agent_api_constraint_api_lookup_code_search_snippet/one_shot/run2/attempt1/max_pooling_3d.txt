host_operator_src="""

#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;
    // Input shape: [N, C, D, H, W]
    const gert::StorageShape *shape = context->GetInputShape(0);
    const auto &origin = shape->GetOriginShape();
    const uint32_t n = origin.GetDim(0);
    const uint32_t c = origin.GetDim(1);
    const uint32_t id = origin.GetDim(2);
    const uint32_t ih = origin.GetDim(3);
    const uint32_t iw = origin.GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(2);

    int32_t kd = static_cast<int32_t>(kernel_size[0]);
    int32_t kh = static_cast<int32_t>(kernel_size[1]);
    int32_t kw = static_cast<int32_t>(kernel_size[2]);
    int32_t sd = static_cast<int32_t>(stride[0]);
    int32_t sh = static_cast<int32_t>(stride[1]);
    int32_t sw = static_cast<int32_t>(stride[2]);
    int32_t pd = static_cast<int32_t>(padding[0]);
    int32_t ph = static_cast<int32_t>(padding[1]);
    int32_t pw = static_cast<int32_t>(padding[2]);

    // output dimensions (ceil_mode=False -> floor)
    uint32_t od = (id + 2 * pd - kd) / sd + 1;
    uint32_t oh = (ih + 2 * ph - kh) / sh + 1;
    uint32_t ow = (iw + 2 * pw - kw) / sw + 1;

    uint32_t totalElements = n * c * od * oh * ow;
    uint32_t spatialPlane = id * ih * iw;

    // UB budget for one tile: we need buffers for input window (max 27 floats) and output tile
    // For simplicity, we compute per-plane tile
    uint32_t windowElems = static_cast<uint32_t>(kd * kh * kw);
    // Allocate per plane: input plane (id*ih*iw) is too large, so we tile across output spatial
    // Use simple: tileLength = number of output elements to process per iteration
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * (windowElems + 1)); // window + 1 output
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = maxTile > totalElements ? totalElements : maxTile;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = totalElements;

    uint32_t numTiles = CeilDiv(totalElements, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_id(id);
    tiling.set_ih(ih);
    tiling.set_iw(iw);
    tiling.set_od(od);
    tiling.set_oh(oh);
    tiling.set_ow(ow);
    tiling.set_kd(kd);
    tiling.set_kh(kh);
    tiling.set_kw(kw);
    tiling.set_sd(sd);
    tiling.set_sh(sh);
    tiling.set_sw(sw);
    tiling.set_pd(pd);
    tiling.set_ph(ph);
    tiling.set_pw(pw);
    tiling.set_totalLength(totalElements);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_WINDOW = 27; // 3*3*3

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                const MaxPooling3dCustomTilingData &tiling)
    {
        n_ = tiling.n;
        c_ = tiling.c;
        id_ = tiling.id;
        ih_ = tiling.ih;
        iw_ = tiling.iw;
        od_ = tiling.od;
        oh_ = tiling.oh;
        ow_ = tiling.ow;
        kd_ = tiling.kd;
        kh_ = tiling.kh;
        kw_ = tiling.kw;
        sd_ = tiling.sd;
        sh_ = tiling.sh;
        sw_ = tiling.sw;
        pd_ = tiling.pd;
        ph_ = tiling.ph;
        pw_ = tiling.pw;
        totalLen_ = tiling.totalLength;
        tileLen_ = tiling.tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // UB buffers: one input window, one output tile, one for max accumulator
        pipe_.InitBuffer(inWindowBuf_, 1, MAX_WINDOW * sizeof(float));
        pipe_.InitBuffer(outTileBuf_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t outBase = blockIdx * tileLen_;
        if (outBase >= totalLen_) return;

        // Spatial stride in output space
        uint32_t outSpatialSize = od_ * oh_ * ow_;

        while (true) {
            uint32_t remaining = totalLen_ - outBase;
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            if (curLen == 0) break;

            // Compute for each output element in this tile
            ComputeTile(outBase, curLen);

            outBase += strideElems;
            if (outBase >= totalLen_) break;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t outBase, uint32_t len)
    {
        AscendC::LocalTensor<float> outTile = outTileBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>();

        // Output elements are linear across N, C, D, H, W
        // For each element, compute its spatial coordinates
        for (uint32_t i = 0; i < len; i++) {
            uint32_t idx = outBase + i;
            uint32_t b = idx / (c_ * od_ * oh_ * ow_);
            uint32_t rem1 = idx % (c_ * od_ * oh_ * ow_);
            uint32_t ch = rem1 / (od_ * oh_ * ow_);
            uint32_t rem2 = rem1 % (od_ * oh_ * ow_);
            uint32_t od = rem2 / (oh_ * ow_);
            uint32_t rem3 = rem2 % (oh_ * ow_);
            uint32_t oh = rem3 / ow_;
            uint32_t ow = rem3 % ow_;

            // Input coordinates with padding
            int32_t idStart = static_cast<int32_t>(od * sd_) - pd_;
            int32_t ihStart = static_cast<int32_t>(oh * sh_) - ph_;
            int32_t iwStart = static_cast<int32_t>(ow * sw_) - pw_;

            float maxVal = -std::numeric_limits<float>::infinity();

            // Iterate over kernel window
            for (int32_t kd = 0; kd < kd_; kd++) {
                int32_t d = idStart + kd;
                if (d < 0 || d >= static_cast<int32_t>(id_)) continue;
                for (int32_t kh = 0; kh < kh_; kh++) {
                    int32_t h = ihStart + kh;
                    if (h < 0 || h >= static_cast<int32_t>(ih_)) continue;
                    for (int32_t kw = 0; kw < kw_; kw++) {
                        int32_t w = iwStart + kw;
                        if (w < 0 || w >= static_cast<int32_t>(iw_)) continue;
                        uint32_t inIdx = (b * c_ + ch) * (id_ * ih_ * iw_) + d * ih_ * iw_ + h * iw_ + w;
                        float val = xGm_.GetValue(inIdx);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }
            // Write to output
            outTile.SetValue(i, maxVal);
        }

        // Copy tile output to global
        AscendC::DataCopy(yGm_[outBase], outTile, len);

        outTileBuf_.FreeTensor(outTile);
        tmpBuf_.FreeTensor(tmp);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> inWindowBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outTileBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_, c_, id_, ih_, iw_;
    uint32_t od_, oh_, ow_;
    int32_t kd_, kh_, kw_;
    int32_t sd_, sh_, sw_;
    int32_t pd_, ph_, pw_;
    uint32_t totalLen_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y, tiling_data);
    op.Process();
}

import torch

_FN = 'max_pooling_3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int = 3, stride: int = 2, padding: int = 1,
                 return_indices: bool = False, ceil_mode: bool = False):
        super(ModelNew, self).__init__()
        self.kernel_size = [kernel_size, kernel_size, kernel_size]
        self.stride = [stride, stride, stride]
        self.padding = [padding, padding, padding]
        # Ignore return_indices and ceil_mode for custom kernel
        if return_indices or ceil_mode:
            print("Warning: custom kernel does not support return_indices or ceil_mode, ignoring.")
        # Optionally convert to half for performance (not shown)
    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding)
