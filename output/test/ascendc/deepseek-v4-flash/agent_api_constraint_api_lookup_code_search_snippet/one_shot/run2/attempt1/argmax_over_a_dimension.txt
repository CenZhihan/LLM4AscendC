host_operator_src = '''

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape &inputShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(inputShape.GetShapeSize());
    int32_t ndim = static_cast<int32_t>(inputShape.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;
    tiling.set_dim(dim);

    // last dimension size (contiguous axis)
    uint32_t lastDim = static_cast<uint32_t>(inputShape.GetDim(ndim - 1));
    uint32_t numRows = totalLength / lastDim;

    // tileLen per row: we will tile across rows; each row is lastDim elements
    uint32_t rowBytes = lastDim * sizeof(float);
    uint32_t maxTileRows = UB_BUDGET_BYTES / (rowBytes * 3 + 2 * sizeof(float) + 256); // conservative
    if (maxTileRows == 0) maxTileRows = 1;
    uint32_t tileRows = (numRows < 32) ? numRows : 32; // small tile for double buffering
    if (tileRows > numRows) tileRows = numRows;

    uint32_t tileLength = lastDim * tileRows; // elements per input tile

    // align to 32B
    tileLength = AlignDown(tileLength, ALIGN_ELEMS);
    if (tileLength == 0) tileLength = lastDim; // at least one row

    uint32_t blockDim = CeilDiv(numRows, tileRows);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_lastDim(lastDim);
    tiling.set_numRows(numRows);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    int32_t ndim = x_shape->GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;

    std::vector<int64_t> outDims;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i != dim) {
            outDims.push_back(x_shape->GetDim(i));
        }
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(static_cast<int32_t>(outDims.size()));
    for (size_t i = 0; i < outDims.size(); ++i) {
        y_shape->SetDim(static_cast<int32_t>(i), outDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class ArgmaxOverADimensionCustom : public OpDef {
public:
    explicit ArgmaxOverADimensionCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT64})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(ArgmaxOverADimensionCustom);
} // namespace ops

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t lastDim,
                                uint32_t numRows, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        lastDim_ = lastDim;
        numRows_ = numRows;
        tileLen_ = tileLength;

        // Input: flattened [numRows, lastDim]
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        // Output: [numRows] indices
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, numRows_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 2 * sizeof(float)); // value+indexindexindex
