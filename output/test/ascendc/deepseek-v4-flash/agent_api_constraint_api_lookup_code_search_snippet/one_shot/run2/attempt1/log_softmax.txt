host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

// For each tile we need: inQueueX (2*tile), outQueueY (2*tile), tmpQueue (1*tile for workspace)
// Also pass1: workspace for ReduceMax (tile floats per inner?), but we can reuse.
// Conservative: 5*tile floats = 20*tile bytes
static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const std::vector<int64_t> &shape = inputShape->GetOriginShape().GetDims();
    uint32_t outer = 1;
    uint32_t inner = 1;
    // Assume 2D shape: (outer, inner). If more dims, treat as outer * rest / inner.
    if (shape.size() == 2) {
        outer = static_cast<uint32_t>(shape[0]);
        inner = static_cast<uint32_t>(shape[1]);
    } else {
        // fallback: flatten all but last dim
        if (shape.size() > 1) {
            for (size_t i = 0; i < shape.size() - 1; ++i) outer *= static_cast<uint32_t>(shape[i]);
            inner = static_cast<uint32_t>(shape.back());
        } else {
            outer = 1;
            inner = static_cast<uint32_t>(shape[0]);
        }
    }
    uint32_t totalLength = outer * inner;

    // tileLen from UB budget: bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (inner < tileLen) {
        const uint32_t aligned = AlignDown(inner, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : inner;
    }

    // blockDim: number of rows per block? We assign blocks to outer dimension.
    // Each block processes a contiguous chunk of rows.
    // We want each block to have a workable number of rows.
    const uint32_t numRows = outer;
    uint32_t blockDim = numRows;  // ideal: one block per row
    if (blockDim > 24u) blockDim = 24u; // cap
    if (blockDim < 4u) blockDim = 4u;

    // Ensure blockDim does not exceed outer
    if (blockDim > outer) blockDim = outer;
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outer(outer);
    tiling.set_inner(inner);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outer, uint32_t inner,
                                uint32_t tileLength)
    {
        totalLen_ = totalLength;
        outer_ = outer;
        inner_ = inner;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // queues for each pass can be reused; we allocate maximum needed.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(workspaceQueue_, 1, tileLen_ * sizeof(float)); // for ReduceMax/Sum workspace
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a set of rows (outer dimension)
        uint32_t rowsPerBlock = (outer_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > outer_) rowEnd = outer_;
        if (rowStart >= outer_) return;

        // Allocate local buffers for per-row max and sum (max 2 floats per row)
        uint32_t numRowsLocal = rowEnd - rowStart;
        if (numRowsLocal == 0) return;
        // Use local tensors from workspace or pipe? We'll allocate from pipe as TBuf.
        // For simplicity, use a fixed-size array (max 170 rows, 170*8=1360 bytes < UB)
        // We'll allocate dynamically from temporary buffer.
        AscendC::LocalTensor<float> rowMaxLocal;
        AscendC::LocalTensor<float> rowSumLocal;
        // We need numRowsLocal * sizeof(float) bytes for each
        uint32_t rowBufferBytes = numRowsLocal * sizeof(float);
        pipe_.InitBuffer(rowMaxBuf_, rowBufferBytes);
        pipe_.InitBuffer(rowSumBuf_, rowBufferBytes);
        rowMaxLocal = rowMaxBuf_.AllocTensor<float>();
        rowSumLocal = rowSumBuf_.AllocTensor<float>();

        // Row offset in global memory
        // For each row, we process tiles
        for (uint32_t r = 0; r < numRowsLocal; ++r) {
            uint32_t rowIdx = rowStart + r;
            uint32_t rowBase = rowIdx * inner_;
            float rowMaxVal = -1e38f; // conservative negative infinity
            float rowSumVal = 0.0f;

            // ---------- Pass 1: compute row max ----------
            uint32_t tiles = (inner_ + tileLen_ - 1) / tileLen_;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = rowBase + t * tileLen_;
                uint32_t curLen = (inner_ - t * tileLen_ < tileLen_) ? (inner_ - t * tileLen_) : tileLen_;
                if (curLen == 0) break;

                // CopyIn tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curLen);
                inQueueX_.EnQue(xLocal);

                // Dequeue for compute
                AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> workspace = workspaceQueue_.AllocTensor<float>();

                // ReduceMax returns scalar (size 1 tensor)
                AscendC::LocalTensor<float> maxScalar = tmpQueue_.AllocTensor<float>(); // size 1
                AscendC::ReduceMax(maxScalar, xTile, workspace, curLen);
                // Compare with running max
                float tileMax = maxScalar(0);
                if (tileMax > rowMaxVal) rowMaxVal = tileMax;

                tmpQueue_.FreeTensor(maxScalar);
                workspaceQueue_.FreeTensor(workspace);
                inQueueX_.FreeTensor(xTile);
            }
            rowMaxLocal(r) = rowMaxVal;

            // ---------- Pass 2: compute sum of exp(x - max) ----------
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = rowBase + t * tileLen_;
                uint32_t curLen = (inner_ - t * tileLen_ < tileLen_) ? (inner_ - t * tileLen_) : tileLen_;
                if (curLen == 0) break;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curLen);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> workspace = workspaceQueue_.AllocTensor<float>();

                // tmp = x - max
                AscendC::Sub(tmp, xTile, rowMaxVal, curLen);
                // exp(tmp)
                AscendC::Exp(tmp, tmp, curLen);
                // sum over tile
                AscendC::LocalTensor<float> sumScalar = outQueueY_.AllocTensor<float>(); // reuse out queue as temporary
                AscendC::ReduceSum(sumScalar, tmp, workspace, curLen);
                rowSumVal += sumScalar(0);

                outQueueY_.FreeTensor(sumScalar);
                workspaceQueue_.FreeTensor(workspace);
                tmpQueue_.FreeTensor(tmp);
                inQueueX_.FreeTensor(xTile);
            }
            rowSumLocal(r) = rowSumVal;
        }

        // ---------- Pass 3: compute log_softmax output ----------
        // We have rowMax and rowSum for each local row.
        // Re-read input and write output.
        for (uint32_t r = 0; r < numRowsLocal; ++r) {
            uint32_t rowIdx = rowStart + r;
            uint32_t rowBase = rowIdx * inner_;
            float rowMax = rowMaxLocal(r);
            float rowSum = rowSumLocal(r);
            // logSum = log(rowSum)
            AscendC::LocalTensor<float> logSumTensor = tmpQueue_.AllocTensor<float>(); // size 1
            AscendC::Log(logSumTensor, rowSum, 1);
            float logSum = logSumTensor(0);
            tmpQueue_.FreeTensor(logSumTensor);

            uint32_t tiles = (inner_ + tileLen_ - 1) / tileLen_;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = rowBase + t * tileLen_;
                uint32_t curLen = (inner_ - t * tileLen_ < tileLen_) ? (inner_ - t * tileLen_) : tileLen_;
                if (curLen == 0) break;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curLen);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

                // tmp = x - max
                AscendC::Sub(tmp, xTile, rowMax, curLen);
                // y = tmp - logSum
                AscendC::Sub(yLocal, tmp, logSum, curLen);

                // CopyOut
                AscendC::DataCopy(yGm_[offset], yLocal, curLen);

                tmpQueue_.FreeTensor(tmp);
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xTile);
            }
        }

        rowMaxBuf_.FreeTensor(rowMaxLocal);
        rowSumBuf_.FreeTensor(rowSumLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workspaceQueue_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> rowMaxBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> rowSumBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outer_ {0};
    uint32_t inner_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.outer, tiling_data.inner, tiling_data.tileLength);
    op.Process();
}
