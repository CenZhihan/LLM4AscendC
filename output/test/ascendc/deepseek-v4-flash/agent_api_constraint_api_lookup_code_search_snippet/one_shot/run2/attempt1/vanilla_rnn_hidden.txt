// We assume A and B are row-major; first matmul needs B transposed.
// For simplicity, we use two separate Matmul objects.
// The tiling data (from host) gives dimensions; we use a simple 1-block configuration.

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(
        GM_ADDR x_t, GM_ADDR h, GM_ADDR w_i2h, GM_ADDR b_i2h,
        GM_ADDR w_h2o, GM_ADDR b_h2o,
        GM_ADDR h_next, GM_ADDR y_t,
        const optiling::VanillaRnnHiddenCustomTilingData &tiling)
    {
        batch_      = tiling.batch;
        inputDim_   = tiling.inputDim;
        hiddenDim_  = tiling.hiddenDim;
        outputDim_  = tiling.outputDim;
        concatDim_  = tiling.concatDim;

        x_tGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x_t), batch_ * inputDim_);
        hGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(h), batch_ * hiddenDim_);
        w_i2hGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w_i2h), hiddenDim_ * concatDim_);
        b_i2hGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b_i2h), hiddenDim_);
        w_h2oGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w_h2o), outputDim_ * hiddenDim_);
        b_h2oGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b_h2o), outputDim_);
        h_nextGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(h_next), batch_ * hiddenDim_);
        y_tGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y_t), batch_ * outputDim_);

        // Allocate one pipe and two matmul objects (each matmul uses its own object)
        // We'll use REGIST_MATMUL_OBJ as in the preload example.
        // Define a single pipe for both matmuls (they are sequential).
        pipe_.InitBuffer(inQ1, 1, batch_ * concatDim_ * sizeof(float));   // concatenated input
        pipe_.InitBuffer(inQ2, 1, batch_ * hiddenDim_ * sizeof(float));   // hidden after tanh
        pipe_.InitBuffer(outQ1, 1, batch_ * hiddenDim_ * sizeof(float));  // pre (before tanh)
        pipe_.InitBuffer(outQ2, 1, batch_ * outputDim_ * sizeof(float));  // output y
        // Weights are treated as constant; we can store permanently in UB if small enough.
        // For simplicity here we assume they fit; we'll copy them once.
        pipe_.InitBuffer(w1Q, 1, hiddenDim_ * concatDim_ * sizeof(float));
        pipe_.InitBuffer(w2Q, 1, outputDim_ * hiddenDim_ * sizeof(float));

        // Copy weights and biases into UB (they are constant across batch steps)
        w1Local_ = w1Q.AllocTensor<float>();
        AscendC::DataCopy(w1Local_, w_i2hGm_, hiddenDim_ * concatDim_);
        w2Local_ = w2Q.AllocTensor<float>();
        AscendC::DataCopy(w2Local_, w_h2oGm_, outputDim_ * hiddenDim_);
    }

    __aicore__ inline void Process()
    {
        // 1. Copy in x_t and h, concatenate them into a [batch, concatDim] tensor
        CopyIn();
        // 2. First matmul: (concat) * (w_i2h)^T  => [batch, hiddenDim]
        //    Since w_i2h is [hiddenDim, concatDim], we need to perform C = A * B^T
        //    Using Matmul with transB = true.
        ComputeMatmul(w1Local_, true);
        // 3. Add bias and apply tanh
        ApplyTanhAndBias();
        // 4. Second matmul: (h_next) * (w_h2o)^T  => [batch, outputDim]
        ComputeMatmul(w2Local_, true);
        // 5. Add bias
        AddBiasOutput();
        // 6. Copy out results
        CopyOut();

    }

    {

        // For demonstration, we use AscendC::Matmul with a simple tiling.

        inQ1.FreeTensor(A);
        outQ1.EnQue(C);
    }

    __aicore__ inline void ApplyTanhAndBias()
    {
        // Dequeue pre result, add bias, apply tanh
        AscendC::LocalTensor<float> pre = outQ1.DeQue<float>();
        AscendC::LocalTensor<float> biasLocal = inQ1.AllocTensor<float>(); // reuse inQ1
