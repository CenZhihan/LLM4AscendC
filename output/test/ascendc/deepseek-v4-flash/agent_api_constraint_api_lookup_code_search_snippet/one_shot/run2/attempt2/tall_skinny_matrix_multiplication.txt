static constexpr uint32_t FULL_L1_SIZE = 0xFFFFFFFF;
static constexpr uint32_t FULL_L0C_SIZE = 0xFFFFFFFF;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::StorageShape *storageShapeA = context->GetInputShape(0);
    const gert::StorageShape *storageShapeB = context->GetInputShape(1);
    uint32_t M = static_cast<uint32_t>(storageShapeA->GetStorageShape().GetDim(0));
    uint32_t K = static_cast<uint32_t>(storageShapeA->GetStorageShape().GetDim(1));
    uint32_t N = static_cast<uint32_t>(storageShapeB->GetStorageShape().GetDim(1));

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_shareMode(0);
    tiling.set_shareL1Size(FULL_L1_SIZE);
    tiling.set_shareL0CSize(FULL_L0C_SIZE);
    tiling.set_shareUbSize(0);

    // heuristic blockDim: limit to e.g. 8 for simplicity
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *storageShapeA = context->GetInputShape(0);
    const gert::StorageShape *storageShapeB = context->GetInputShape(1);
    gert::Shape *shapeC = context->GetOutputShape(0);
    shapeC->SetDimNum(2);
    shapeC->SetDim(0, storageShapeA->GetStorageShape().GetDim(0));
    shapeC->SetDim(1, storageShapeB->GetStorageShape().GetDim(1));
    return GRAPH_SUCCESS;
}

{
    const ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatrixMul {
public:
    __aicore__ inline KernelTallSkinnyMatrixMul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm)
    {
        // Copy tiling data from GM to local struct
        uint32_t *tiling = reinterpret_cast<uint32_t *>(&tilingData_);
        uint32_t *tilingGm32 = reinterpret_cast<__gm__ uint32_t *>(tilingGm);
        for (int i = 0; i < sizeof(TallSkinnyMatrixMultiplicationCustomTilingData) / sizeof(uint32_t); ++i) {
            tiling[i] = tilingGm32[i];
        }

        M_ = tilingData_.M;
        N_ = tilingData_.N;
        K_ = tilingData_.K;

        aGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(a), M_ * K_);
        bGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(b), K_ * N_);
        cGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float *>(c), M_ * N_);

        if (GetBlockIdx() >= tilingData_.blockDim) {
            return;
        }

        // Set up matmul with a fixed tile size (can be tuned)
        // Use the Cube unit directly via Matmul API
        matmul_.SetOrgShape(M_, N_, K_);
    }

    __aicore__ inline void Process()
    {
        if (GetBlockIdx() >= tilingData_.blockDim) {
            return;
        }

        // Simple single-iteration matmul for demonstration; real implementation should tile
        REGIST_MATMUL_OBJ(&pipe_, GetSysWorkSpacePtr(), matmulObj_, (TCubeTiling *)nullptr);
        matmul_.SetTensorA(aGlobal_, false);
        matmul_.SetTensorB(bGlobal_, false);
        matmul_.IterateAll(cGlobal_);
        matmul_.End();
    }

private:
    TallSkinnyMatrixMultiplicationCustomTilingData tilingData_;
    uint32_t M_, N_, K_;
    GlobalTensor<float> aGlobal_;
    GlobalTensor<float> bGlobal_;
    GlobalTensor<float> cGlobal_;
    TPipe pipe_;
    Matmul<MatmulType<TPosition::GM, CubeFormat::ND, float>,
           MatmulType<TPosition::GM, CubeFormat::ND, float>,
           MatmulType<TPosition::GM, CubeFormat::ND, float>,
           MatmulType<TPosition::GM, CubeFormat::ND, float>,
           DefaultMatmulCfg> matmulObj_;
    Matmul<MatmulType<TPosition::GM, CubeFormat::ND, float>,
           MatmulType<TPosition::GM, CubeFormat::ND, float>,
           MatmulType<TPosition::GM, CubeFormat::ND, float>,
           MatmulType<TPosition::GM, CubeFormat::ND, float>,
           DefaultMatmulCfg> matmul_; // redundant, kept for clarity
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                                                                GM_ADDR workspace, GM_ADDR tilingGm)
{
    KernelTallSkinnyMatrixMul op;
    op.Init(a, b, c, workspace, tilingGm);
    op.Process();
}
