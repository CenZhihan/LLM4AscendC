host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>

namespace optiling {

// UB model:
//   inQueueX     : 2 * tileLen * reduceLen * sizeof(float)
//   outQueueY    : 2 * tileLen * sizeof(float)
//   tmpQueue     : 1 * reduceLen * sizeof(float)
//   conservatively target total 96KB per processing block.
static constexpr uint32_t UB_BUDGET_BYTES = 96 * 1024;
static constexpr uint32_t MIN_TILE_ROWS = 4;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &inputShape = storageShape->GetOriginShape();
    uint32_t rank = static_cast<uint32_t>(inputShape.GetDimNum());

    // Get dim attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;
    // Handle negative dim
    if (dim < 0) {
        dim += static_cast<int64_t>(rank);
    }
    if (dim < 0 || dim >= static_cast<int64_t>(rank)) {
        dim = 0; // fallback
    }

    // Compute outerLen = product of dims before and after reduce axis
    uint32_t outer_before = 1;
    for (int64_t i = 0; i < dim; ++i) {
        outer_before *= static_cast<uint32_t>(inputShape.GetDim(static_cast<uint32_t>(i)));
    }
    uint32_t reduceLen = static_cast<uint32_t>(inputShape.GetDim(static_cast<uint32_t>(dim)));
    uint32_t outer_after = 1;
    for (int64_t i = dim + 1; i < static_cast<int64_t>(rank); ++i) {
        outer_after *= static_cast<uint32_t>(inputShape.GetDim(static_cast<uint32_t>(i)));
    }
    uint32_t outerLen = outer_before * outer_after;

    // Calculate tileLen based on UB budget
    // bytes = 2 * tileLen * reduceLen * sizeof(float)   inQueueX
    //       + 2 * tileLen * sizeof(float)               outQueueY
    //       + reduceLen * sizeof(float)                 tmp
    // Solve: tileLen <= (UB_BUDGET - 4*reduceLen) / (8*reduceLen + 8)
    uint32_t maxTile = (UB_BUDGET_BYTES - 4 * reduceLen) / (8 * reduceLen + 8);
    if (maxTile < MIN_TILE_ROWS) maxTile = MIN_TILE_ROWS;
    if (maxTile > outerLen) maxTile = outerLen;
    constexpr uint32_t ALIGN = 32;
    uint32_t tileLen = AlignDown(maxTile, ALIGN);
    if (tileLen < MIN_TILE_ROWS) tileLen = MIN_TILE_ROWS;

    uint32_t numTiles = (outerLen + tileLen - 1) / tileLen;
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_outerLen(outerLen);
    tiling.set_reduceLen(reduceLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &x_shape = storageShape->GetOriginShape();
    uint32_t rank = static_cast<uint32_t>(x_shape.GetDimNum());

    // Get dim attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;
    if (dim < 0) dim += static_cast<int64_t>(rank);
    if (dim < 0 || dim >= static_cast<int64_t>(rank)) dim = 0;

    // Output shape: remove the reduced dimension
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(rank - 1);
    uint32_t idx = 0;
    for (uint32_t d = 0; d < rank; ++d) {
        if (static_cast<int64_t>(d) != dim) {
            y_shape->SetDim(idx, x_shape.GetDim(d));
            ++idx;
        }
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "reduce/adv_api/reduce.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerLen, uint32_t reduceLen, uint32_t tileLen)
    {
        outerLen_ = outerLen;
        reduceLen_ = reduceLen;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, outerLen_ * reduceLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outerLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * reduceLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, reduceLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outerLen_ == 0 || reduceLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideRows = blockNum * tileLen_;
        uint32_t rowBase = blockIdx * tileLen_;
        if (rowBase >= outerLen_) return;

        uint32_t curRows = outerLen_ - rowBase;
        if (curRows > tileLen_) curRows = tileLen_;
        CopyIn(rowBase, curRows);

        while (true) {
            const uint32_t nextRowBase = rowBase + strideRows;
            uint32_t nextRows = 0;
            if (nextRowBase < outerLen_) {
                nextRows = outerLen_ - nextRowBase;
                if (nextRows > tileLen_) nextRows = tileLen_;
                CopyIn(nextRowBase, nextRows);
            }

            Compute(curRows);
            CopyOut(rowBase, curRows);

            if (nextRows == 0) break;
            rowBase = nextRowBase;
            curRows = nextRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowOffset, uint32_t rows)
    {
        uint32_t copyLen = rows * reduceLen_ * sizeof(float);
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopyExtParams copyParams;
        copyParams.blockCount = rows;
        copyParams.blockLen = reduceLen_ * sizeof(float);
        copyParams.srcStride = 0;
        copyParams.dstStride = 0;
        AscendC::DataCopyPad(xLocal, xGm_[rowOffset * reduceLen_], copyParams);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tmpBuf = tmpQueue_.AllocTensor<uint8_t>();

        uint32_t srcShape[2] = {rows, reduceLen_};
        constexpr bool srcInnerPad = false;
        AscendC::ReduceMean<float, AscendC::Pattern::Reduce::AR, false>(yLocal, xLocal, tmpBuf, srcShape, srcInnerPad);

        tmpQueue_.FreeTensor(tmpBuf);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowOffset, uint32_t rows)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[rowOffset], yLocal, rows);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outerLen_;
    uint32_t reduceLen_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y, tiling_data.outerLen, tiling_data.reduceLen, tiling_data.tileLen);
    op.Process();
}

at::Tensor mean_reduction_impl_npu(const at::Tensor& self, int64_t dim) {
    // Compute output shape: same as input but with dimension dim removed
    auto input_shape = self.sizes();
    std::vector<int64_t> out_shape;
    for (size_t i = 0; i < input_shape.size(); ++i) {
        if (static_cast<int64_t>(i) != dim) {
            out_shape.push_back(input_shape[i]);
        }
    }
    at::Tensor result = at::empty(out_shape, self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, dim, result);
    return result;
