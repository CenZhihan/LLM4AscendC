host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: inQueueX 2*tile + outQueueY 2*tile + tmpMax 1*tile + tmpSum 1*tile + tmpExp 1*tile + tmpLog 1*tile ≈ 8*tile floats => 32*tile bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;
static constexpr uint32_t FLOAT_NUM_OF_SINGLE_BLOCK = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    auto dimAttr = context->GetAttrs()->GetAttrPointer<int64_t>(0);
    int32_t dim = static_cast<int32_t>(dimAttr ? *dimAttr : 1);

    // Assume input is 2D: rows x cols
    const auto& shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rowLength = (shape.GetDimNum() == 0) ? 1u : static_cast<uint32_t>(shape.GetDimSize(0));
    uint32_t columnLength = (shape.GetDimNum() < 2) ? 1u : static_cast<uint32_t>(shape.GetDimSize(1));
    if (dim != 1) {
        rowLength = (shape.GetDimNum() < 2) ? 1u : static_cast<uint32_t>(shape.GetDimSize(1));
        columnLength = (shape.GetDimNum() == 0) ? 1u : static_cast<uint32_t>(shape.GetDimSize(0));
    }

    // UB budget: we need rowMax and rowSum per chunk (64 floats per chunk used in softmax reference)
    // For simplicity, use the same per-row approach as softmax kernel.
    uint32_t ubFloats = UB_BUDGET_BYTES / sizeof(float);
    uint32_t tileFloats = AlignDown(ubFloats / 8, ALIGN_ELEMS);  // 8 copies of tile (x, y, max, sum, exp, log...)
    if (tileFloats < ALIGN_ELEMS) tileFloats = ALIGN_ELEMS;
    if (tileFloats > HARD_CAP_ELEMS) tileFloats = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // Tile length per iteration (columns)
    uint32_t tileLen = columnLength;
    if (tileLen > tileFloats) tileLen = AlignDown(tileFloats, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = tileFloats;  // fallback

    // Number of rows per block (distribute)
    uint32_t rowsPerBlock = CeilDiv(rowLength, context->GetBlockDim());
    if (rowsPerBlock == 0) rowsPerBlock = 1;

    // Total elements per block
    uint32_t blockLength = rowsPerBlock * columnLength;
    uint32_t numTiles = CeilDiv(blockLength, tileLen);
    if (numTiles < 1) numTiles = 1;

    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_rowLength(rowLength);
    tiling.set_columnLength(columnLength);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    *context->GetOutputShape(0) = *context->GetInputShape(0);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t columnLength, uint32_t rowsPerBlock)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        colLen_ = columnLength;
        rowsPerBlock_ = rowsPerBlock;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // Temporary buffers for row max and row sum (one per row)
        pipe_.InitBuffer(tmpMax_, 1, colLen_ * sizeof(float));
        pipe_.InitBuffer(tmpSum_, 1, colLen_ * sizeof(float));
        pipe_.InitBuffer(tmpExp_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Distribute rows across blocks
        uint32_t startRow = blockIdx * rowsPerBlock_;
        uint32_t endRow = startRow + rowsPerBlock_;
        if (startRow >= colLen_ ? totalLen_/colLen_ : totalLen_/colLen_) {} // placeholder, safe anyway
        uint32_t numRows = (endRow > (totalLen_ / colLen_)) ? (totalLen_ / colLen_) - startRow : rowsPerBlock_;

        for (uint32_t r = 0; r < numRows; r++) {
            uint32_t rowOffset = (startRow + r) * colLen_;
            uint32_t rowLen = colLen_;
            ProcessRow(rowOffset, rowLen);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t gmOffset, uint32_t len)
    {
        // Step 1: Find row max
        copyInRow(gmOffset, len);
        computeRowMax(len);
        // Step 2: Subtract max, compute exp, sum
        computeExpSum(len);
        // Step 3: Compute log sum, final y = (x - max) - log(sum)
        computeFinal(len);
        copyOutRow(gmOffset, len);
    }

    __aicore__ inline void copyInRow(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void computeRowMax(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maxLocal = tmpMax_.AllocTensor<float>();
        // ReduceMax requires a workspace tensor; we use tmpExp_ as workspace
        AscendC::ReduceMax(maxLocal, xLocal, tmpExp_, len);
        // Broadcast max to full length (first element contains max)
        AscendC::LocalTensor<float> bcastMax = tmpExp_;
        AscendC::LocalTensor<float> scalar = tmpMax_;  // first element
        AscendC::Duplicate(bcastMax, scalar, len);      // fill entire tmpExp_ with scalar
        tmpMax_.FreeTensor(maxLocal);
        // enqueue bcastMax and xLocal for next stage
        tmpMax_.EnQue(bcastMax);  // reused as bcastMax
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void computeExpSum(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> bcastMax = tmpMax_.DeQue<float>();
        AscendC::LocalTensor<float> sub = outQueueY_.AllocTensor<float>();  // reuse output queue as temp
        // x - max
        AscendC::Sub(sub, xLocal, bcastMax, len);
        // exp
        AscendC::Exp(sub, sub, len);
        // sum
        AscendC::LocalTensor<float> sumLocal = tmpSum_.AllocTensor<float>();
        AscendC::ReduceSum(sumLocal, sub, tmpExp_, len);  // workspace tmpExp_ (now free)
        // Broadcast sum
        AscendC::LocalTensor<float> bcastSum = tmpExp_;
        AscendC::LocalTensor<float> scalar = sumLocal;
        AscendC::Duplicate(bcastSum, scalar, len);
        tmpSum_.FreeTensor(sumLocal);
        // enqueue for next stage: xLocal (original x), sub (exp), bcastSum (row sum broadcast)
        tmpSum_.EnQue(bcastSum);  // reused as bcastSum
        outQueueY_.EnQue(sub);   // sub now holds exp(x - max)
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void computeFinal(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> expLocal = outQueueY_.DeQue<float>();  // exp from previous step
        AscendC::LocalTensor<float> bcastSum = tmpSum_.DeQue<float>();
        AscendC::LocalTensor<float> bcastMax = tmpMax_.AllocTensor<float>();  // will be used as bcastMax again
        // Log of sum
        AscendC::LocalTensor<float> logSum = tmpMax_;  // reuse as logSum
        AscendC::Log(logSum, bcastSum, len);
        // Compute final y = (x - max) - log(sum) = (x - max) - logsum
        AscendC::Sub(bcastMax, xLocal, bcastMax, len);  // bcastMax now holds (x - max)
        AscendC::Sub(bcastMax, bcastMax, logSum, len);
        // Copy to output queue
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Copy(yLocal, bcastMax, len);
        outQueueY_.EnQue(yLocal);
        // Free others
        inQueueX_.FreeTensor(xLocal);
        outQueueY_.FreeTensor(expLocal);
        tmpSum_.FreeTensor(bcastSum);
        tmpMax_.FreeTensor(bcastMax);  // same as logSum
    }

    __aicore__ inline void copyOutRow(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSum_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpExp_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t colLen_ {0};
    uint32_t rowsPerBlock_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.columnLength, tiling_data.rowsPerBlock);
    op.Process();
}
