host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t DivUp(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const gert::Shape &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rowLength = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t columnLength = static_cast<uint32_t>(shape.GetDim(1));
    uint32_t totalLength = rowLength * columnLength;

    // We process one row at a time per core, splitting columns into tiles.
    // Each tile needs: x (tileColNum), y (tileColNum), maxLocal, sumLocal, tmpBuffer.
    // Safe per-tile UB bytes = 5 * tileColNum * sizeof(float) + 60KB (tmp for SoftMax)
    // We reserve 60KB for tmp, rest for data.
    uint32_t tmpBytes = 60 * 1024;
    uint32_t dataBytesPerTile = UB_BUDGET_BYTES - tmpBytes;
    uint32_t maxColsPerTile = dataBytesPerTile / (5 * sizeof(float));  // 5 * col floats
    maxColsPerTile = (maxColsPerTile / ALIGN_ELEMS) * ALIGN_ELEMS;
    if (maxColsPerTile < ALIGN_ELEMS) maxColsPerTile = ALIGN_ELEMS;
    uint32_t tileColNum = maxColsPerTile;
    if (tileColNum > columnLength) tileColNum = columnLength;

    uint32_t coreRowNum = 1;  // one row per block per iteration (grid stride may repeat)
    // blockDim: use a moderate number of blocks
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_columnLength(columnLength);
    tiling.set_tileColNum(tileColNum);
    tiling.set_blockDim(blockDim);
    tiling.set_coreRowNum(coreRowNum);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t FLOAT_NUM_OF_SINGLE_BLOCK = 8;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rowLength, uint32_t columnLength,
                                uint32_t tileColNum, uint32_t coreRowNum)
    {
        rowLen_ = rowLength;
        colLen_ = columnLength;
        tileColNum_ = tileColNum;
        coreRowNum_ = coreRowNum;

        xGm_.SetGlobalBuffer((__gm__ float*)x, rowLen_ * colLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, rowLen_ * colLen_);

        // Each tile: x + y = 2 * tileColNum floats, plus small local max/sum/tmp
        uint32_t tileBytes = tileColNum_ * sizeof(float);
        uint32_t msBytes = coreRowNum_ * FLOAT_NUM_OF_SINGLE_BLOCK * sizeof(float);
        uint32_t tmpBytes = 60 * 1024;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileBytes);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileBytes);
        pipe_.InitBuffer(maxQueue_, 1, msBytes);
        pipe_.InitBuffer(sumQueue_, 1, msBytes);
        pipe_.InitBuffer(sharedTmpBuffer_, tmpBytes);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Grid stride over rows
        for (uint32_t r = blockIdx; r < rowLen_; r += blockNum) {
            // Row r: compute softmax across columns in tiles
            // Pass 1: find max
            float rowMax = -std::numeric_limits<float>::infinity();
            uint32_t colOffset = 0;
            while (colOffset < colLen_) {
                uint32_t curCols = (colOffset + tileColNum_ <= colLen_) ? tileColNum_ : (colLen_ - colOffset);
                CopyIn(r, colOffset, curCols);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                // Compute local max
                AscendC::LocalTensor<float> maxLocal = maxQueue_.AllocTensor<float>();
                AscendC::ReduceMax(maxLocal, xLocal, curCols);
                float localMax = maxLocal.GetValue(0);
                maxQueue_.FreeTensor(maxLocal);
                if (localMax > rowMax) rowMax = localMax;
                inQueueX_.FreeTensor(xLocal);
                colOffset += curCols;
            }

            // Pass 2: compute sum of exp(x - rowMax)
            float rowSum = 0.0f;
            colOffset = 0;
            while (colOffset < colLen_) {
                uint32_t curCols = (colOffset + tileColNum_ <= colLen_) ? tileColNum_ : (colLen_ - colOffset);
                CopyIn(r, colOffset, curCols);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                // y = exp(x - rowMax)
                AscendC::Subs(yLocal, xLocal, rowMax, curCols);
                AscendC::Exp(yLocal, yLocal, curCols);
                // sum y
                AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
                AscendC::ReduceSum(sumLocal, yLocal, curCols);
                float localSum = sumLocal.GetValue(0);
                sumQueue_.FreeTensor(sumLocal);
                rowSum += localSum;
                // store y back for later normalization? Or store in separate buffer?
                // We'll keep y in queue for final normalization pass.
                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xLocal);
                colOffset += curCols;
            }

            // Pass 3: normalize
            float invSum = 1.0f / rowSum;
            colOffset = 0;
            while (colOffset < colLen_) {
                uint32_t curCols = (colOffset + tileColNum_ <= colLen_) ? tileColNum_ : (colLen_ - colOffset);
                // We need to dequeue y from previous pass? Actually we stored them in outQueueY_.
                AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
                AscendC::Muls(yLocal, yLocal, invSum, curCols);
                CopyOut(r, colOffset, curCols, yLocal);
                outQueueY_.FreeTensor(yLocal);
                colOffset += curCols;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row, uint32_t colStart, uint32_t len)
    {
        uint32_t gmIdx = row * colLen_ + colStart;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmIdx], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row, uint32_t colStart, uint32_t len,
                                    AscendC::LocalTensor<float>& yLocal)
    {
        uint32_t gmIdx = row * colLen_ + colStart;
        AscendC::DataCopy(yGm_[gmIdx], yLocal, len);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> maxQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> sumQueue_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> sharedTmpBuffer_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rowLen_ {0};
    uint32_t colLen_ {0};
    uint32_t tileColNum_ {0};
    uint32_t coreRowNum_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.rowLength,
            tiling_data.columnLength,
            tiling_data.tileColNum,
            tiling_data.coreRowNum);
    op.Process();
}
