#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: inQueueX 2*tile + outQueueY 2*tile + tmpWork 1*tile + tmpScalar 1 (small) ≈ 5*tile floats => 20*tile bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const uint32_t totalLength = static_cast<uint32_t>(inputShape->GetOriginShape().GetShapeSize());

    // Assume 2D input: last dimension is the softmax dimension.
    uint32_t dimNum = inputShape->GetOriginShape().GetDimNum();
    uint32_t colLength = (dimNum >= 2) ? static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(dimNum - 1)) : totalLength;

    // Tile size from UB budget: 5 * tileLen floats
    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (colLength < tileLen) {
        tileLen = AlignDown(colLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = colLength;
    }

    // Block dim: one block per row group (large), but keep conservative
    uint32_t blockDim = 24u; // fixed for simplicity
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_colLength(colLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    *context->GetOutputShape(0) = *context->GetInputShape(0);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -3.402823e+38f;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t colLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        colLen_ = colLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpWorkspace_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpScalar_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || colLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Distribute rows across blocks
        uint32_t rowLen = totalLen_ / colLen_;
        uint32_t rowsPerBlock = (rowLen + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = (startRow + rowsPerBlock > rowLen) ? rowLen : startRow + rowsPerBlock;

        for (uint32_t r = startRow; r < endRow; r++) {
            ProcessRow(r * colLen_);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t baseOffset)
    {
        uint32_t numTiles = (colLen_ + tileLen_ - 1) / tileLen_;

        // ---------- Pass 1: compute row max ----------
        AscendC::LocalTensor<float> maxScalar = tmpScalar_.AllocTensor<float>();
        maxScalar.SetValue(0, NEG_INF);

        for (uint32_t t = 0; t < numTiles; t++) {
            uint32_t offset = baseOffset + t * tileLen_;
            uint32_t len = (t == numTiles - 1) ? colLen_ - t * tileLen_ : tileLen_;
            if (len == 0) break;

            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // Compute tile max (scalar)
            AscendC::LocalTensor<float> tileMax = tmpScalar_.AllocTensor<float>();
            AscendC::LocalTensor<float> work = tmpWorkspace_.AllocTensor<float>();
            AscendC::ReduceMax(tileMax, xLocal, work, len);
            tmpWorkspace_.FreeTensor(work);

            // Compare with global max
            AscendC::LocalTensor<float> tempScalar = tmpScalar_.AllocTensor<float>();
            tempScalar.SetValue(0, tileMax.GetValue(0));
            AscendC::Max(maxScalar, maxScalar, tempScalar, 1u);
            tmpScalar_.FreeTensor(tempScalar);
            tmpScalar_.FreeTensor(tileMax);
            inQueueX_.FreeTensor(xLocal);
        }

        float rowMax = maxScalar.GetValue(0);
        tmpScalar_.FreeTensor(maxScalar);

        // ---------- Pass 2: compute exp sum ----------
        AscendC::LocalTensor<float> sumScalar = tmpScalar_.AllocTensor<float>();
        sumScalar.SetValue(0, 0.0f);

        for (uint32_t t = 0; t < numTiles; t++) {
            uint32_t offset = baseOffset + t * tileLen_;
            uint32_t len = (t == numTiles - 1) ? colLen_ - t * tileLen_ : tileLen_;
            if (len == 0) break;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // (x - max)
            AscendC::LocalTensor<float> sub = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> bcastMax = tmpWorkspace_.AllocTensor<float>();
            AscendC::Duplicate(bcastMax, rowMax, static_cast<int32_t>(len));
            AscendC::Sub(sub, xLocal, bcastMax, len);
            tmpWorkspace_.FreeTensor(bcastMax);

            // exp
            AscendC::Exp(sub, sub, len);

            // sum tile (scalar)
            AscendC::LocalTensor<float> tileSum = tmpScalar_.AllocTensor<float>();
            AscendC::LocalTensor<float> work = tmpWorkspace_.AllocTensor<float>();
            AscendC::ReduceSum(tileSum, sub, work, len);
            tmpWorkspace_.FreeTensor(work);

            // Accumulate sum
            AscendC::LocalTensor<float> tempScalar = tmpScalar_.AllocTensor<float>();
            tempScalar.SetValue(0, tileSum.GetValue(0));
            AscendC::Add(sumScalar, sumScalar, tempScalar, 1u);
            tmpScalar_.FreeTensor(tempScalar);
            tmpScalar_.FreeTensor(tileSum);
            outQueueY_.FreeTensor(sub);
            inQueueX_.FreeTensor(xLocal);
        }

        float rowSum = sumScalar.GetValue(0);

        // Compute log(sum) using AscendC::Log on a scalar tensor
        AscendC::LocalTensor<float> logSumTensor = tmpScalar_.AllocTensor<float>();
        logSumTensor.SetValue(0, rowSum);
        AscendC::Log(logSumTensor, logSumTensor, 1u);
        float logSum = logSumTensor.GetValue(0);
        tmpScalar_.FreeTensor(logSumTensor);

        // ---------- Pass 3: compute final output ----------
        for (uint32_t t = 0; t < numTiles; t++) {
            uint32_t offset = baseOffset + t * tileLen_;
            uint32_t len = (t == numTiles - 1) ? colLen_ - t * tileLen_ : tileLen_;
            if (len == 0) break;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // y = x - max
            AscendC::LocalTensor<float> bcastMax = tmpWorkspace_.AllocTensor<float>();
            AscendC::Duplicate(bcastMax, rowMax, static_cast<int32_t>(len));
            AscendC::Sub(yLocal, xLocal, bcastMax, len);
            tmpWorkspace_.FreeTensor(bcastMax);

            // y = y - logSum
            AscendC::LocalTensor<float> bcastLogSum = tmpWorkspace_.AllocTensor<float>();
            AscendC::Duplicate(bcastLogSum, logSum, static_cast<int32_t>(len));
            AscendC::Sub(yLocal, yLocal, bcastLogSum, len);
            tmpWorkspace_.FreeTensor(bcastLogSum);

            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);

            // Copy out
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yLocal, len);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpWorkspace_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpScalar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t colLen_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.colLength);
    op.Process();
}
