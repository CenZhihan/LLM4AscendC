host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_SIZE = 32; // 32B alignment for efficient DMA

static inline uint32_t RoundUp(uint32_t x, uint32_t a)
{
    return (x + a - 1u) / a * a;
}

static inline uint32_t DivUp(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rowLength = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t columnLength = static_cast<uint32_t>(shape.GetDim(1));
    uint32_t totalLength = rowLength * columnLength;

    // For each row we need roughly: x tile, max, sum, tmp buffer, output => 5 * col elements
    // plus overhead; safe factor = 8 * col * sizeof(float)
    uint32_t rowCostBytes = columnLength * sizeof(float) * 8;
    uint32_t maxRowsPerTile = (UB_BUDGET_BYTES / rowCostBytes);
    if (maxRowsPerTile == 0) maxRowsPerTile = 1;
    uint32_t coreRowNum = maxRowsPerTile;
    if (coreRowNum > rowLength) coreRowNum = rowLength;

    // tileLength = number of floats per tile (multiple of columnLength)
    uint32_t tileLength = coreRowNum * columnLength;

    // number of tiles for all rows
    uint32_t numTiles = DivUp(rowLength, coreRowNum);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // sharedTmpBufferSize: roughly 60KB as typical, but we compute minimal:
    // needed for SoftMax internal workspace: ~ 2 * columnLength * sizeof(float)
    // aligned to 32B
    uint32_t tmpBufSize = RoundUp(2 * columnLength * sizeof(float), ALIGN_SIZE);
    if (tmpBufSize < 60 * 1024) tmpBufSize = 60 * 1024; // keep safe margin

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_rowLength(rowLength);
    tiling.set_columnLength(columnLength);
    tiling.set_coreRowNum(coreRowNum);
    tiling.set_usedNumBlocks(numTiles);  // actual used blocks (<= blockDim)
    tiling.set_sharedTmpBufferSize(tmpBufSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t FLOAT_NUM_OF_SINGLE_BLOCK = 8; // not used directly, keep for reference

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t rowLength, uint32_t columnLength,
                                uint32_t coreRowNum, uint32_t usedNumBlocks,
                                uint32_t sharedTmpBufferSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        rowLen_ = rowLength;
        colLen_ = columnLength;
        coreRowNum_ = coreRowNum;       // rows this block processes per iteration
        usedNumBlocks_ = usedNumBlocks; // number of blocks actually used

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        // each block processes a contiguous stripe of rows
        uint32_t rowsPerBlock = coreRowNum_;
        uint32_t startRow = blockIdx * rowsPerBlock;
        if (startRow >= rowLen_) startRow = rowLen_; // will return early

        // initialize queues
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(maxQueue_, 1, coreRowNum_ * sizeof(float)); // one per row
        pipe_.InitBuffer(sumQueue_, 1, coreRowNum_ * sizeof(float)); // one per row
        pipe_.InitBuffer(sharedTmpBuffer_, sharedTmpBufferSize);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t startRow = blockIdx * coreRowNum_;
        if (startRow >= rowLen_) return;

        uint32_t rowsRemaining = rowLen_ - startRow;
        uint32_t rowsThisBlock = (rowsRemaining > coreRowNum_) ? coreRowNum_ : rowsRemaining;

        while (rowsThisBlock > 0) {
            uint32_t offset = startRow * colLen_;
            uint32_t len = rowsThisBlock * colLen_;

            CopyIn(offset, len);
            Compute(rowsThisBlock);
            CopyOut(offset, len);

            startRow += coreRowNum_ * static_cast<uint32_t>(AscendC::GetBlockNum()); // grid-stride
            if (startRow >= rowLen_) break;
            rowsRemaining = rowLen_ - startRow;
            rowsThisBlock = (rowsRemaining > coreRowNum_) ? coreRowNum_ : rowsRemaining;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = maxQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tmpBuffer = sharedTmpBuffer_.Get<uint8_t>();

        // Fix 1: SoftMaxShapeInfo does not require template parameter
        AscendC::SoftMaxShapeInfo srcShape = {rows, colLen_, rows * colLen_, rows * colLen_};
        // Fix 2: SoftMaxTiling is a type in the global namespace (no AscendC::)
        SoftMaxTiling softmaxTiling;
        // Use safe version (no alignment optimizations)
        AscendC::SoftMax<float, true>(yLocal, sumLocal, maxLocal, xLocal, tmpBuffer,
                                      softmaxTiling, srcShape);

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        maxQueue_.FreeTensor(maxLocal);
        sumQueue_.FreeTensor(sumLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> maxQueue_;    // per-row max
    AscendC::TQue<AscendC::TPosition::VECIN, 1> sumQueue_;    // per-row sum
    AscendC::TBuf<AscendC::TPosition::VECCALC> sharedTmpBuffer_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t colLen_ {0};
    uint32_t coreRowNum_ {0};
    uint32_t usedNumBlocks_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.rowLength,
            tiling_data.columnLength,
            tiling_data.coreRowNum,
            tiling_data.usedNumBlocks,
            tiling_data.sharedTmpBufferSize);
    op.Process();
}
