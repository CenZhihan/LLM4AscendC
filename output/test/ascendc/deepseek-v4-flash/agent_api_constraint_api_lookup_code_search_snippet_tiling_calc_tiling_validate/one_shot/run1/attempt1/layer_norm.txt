host_tiling_src = '''

#include "register/tilingdata_base.h"
#include "impl/normalization/layernorm/layernorm_tiling.h"

namespace optiling {

struct LayerNormCustomTilingData {
    uint32_t aLength;
    uint32_t rLength;
    uint32_t rLengthWithPadding;
    float epsilon;
    AscendC::LayerNormSeparateTiling layernormTilingData;
};

} // namespace optiling

extern "C" ge::graphStatus TilingFunc(gert::TilingContext *context) {
    auto *tiling = reinterpret_cast<optiling::LayerNormCustomTilingData *>(
        context->GetRawTilingData()->GetData());

    const gert::Shape *xShape = context->GetInputShape(0);
    int64_t *dims = const_cast<int64_t *>(xShape->GetDim());

    // Input shape: (batch, features, dim1, dim2) -> flatten to (aLength, rLength)
    uint32_t aLength = static_cast<uint32_t>(dims[0]);
    uint32_t rLength = 1;
    for (size_t i = 1; i < xShape->GetDimNum(); ++i) {
        rLength *= static_cast<uint32_t>(dims[i]);
    }

    uint32_t typeSize = sizeof(float);
    int32_t alignSize = 32 / typeSize;
    uint32_t rLengthWithPadding = (rLength + alignSize - 1) / alignSize * alignSize;

    // Get eps from attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    float epsilon = epsPtr ? *epsPtr : 1e-5f;

    // Compute tiling using API
    ge::Shape geShape({ static_cast<int64_t>(aLength), static_cast<int64_t>(rLength) });
    uint32_t maxTmpSize = 0;
    uint32_t minTmpSize = 0;
    bool isReuseSource = false;
    AscendC::GetLayerNormMaxMinTmpSize(geShape, typeSize, isReuseSource, true, false, maxTmpSize, minTmpSize);
    uint32_t localWorkspaceSize = minTmpSize;

    AscendC::GetLayerNormNDTilingInfo(geShape, localWorkspaceSize, typeSize, isReuseSource, true,
                                      tiling->layernormTilingData);

    tiling->aLength = aLength;
    tiling->rLength = rLength;
    tiling->rLengthWithPadding = rLengthWithPadding;
    tiling->epsilon = epsilon;

    context->SetBlockDim(1); // Simplify: one block for demo; can be tuned.

    context->GetRawTilingData()->SetDataSize(sizeof(optiling::LayerNormCustomTilingData));
    return ge::GRAPH_SUCCESS;
}

#include "layer_norm_custom_tiling.h"   // This is the host_tiling_src content (included as file)
#include "register/op_def_registry.h"

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

class LayerNormCustom : public OpDef {
public:
    explicit LayerNormCustom(const char *name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("gamma")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("beta")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t aLength, uint32_t rLength,
                                uint32_t rLengthWithPadding, float epsilon,
                                const AscendC::LayerNormSeparateTiling &tiling) {
        aLength_ = aLength;
        rLength_ = rLength;
        rLengthWithPadding_ = rLengthWithPadding;
        epsilon_ = epsilon;
        tiling_ = tiling;

        uint32_t totalSize = aLength * rLengthWithPadding;

        xGm_.SetGlobalBuffer((__gm__ float *)x, totalSize);
        gammaGm_.SetGlobalBuffer((__gm__ float *)gamma, rLengthWithPadding);
        betaGm_.SetGlobalBuffer((__gm__ float *)beta, rLengthWithPadding);
        yGm_.SetGlobalBuffer((__gm__ float *)y, totalSize);

        // Single buffer for simplicity; using the library API acts as a 'whole' copy
        pipe_.InitBuffer(inQueueX_, 1, totalSize * sizeof(float));
        pipe_.InitBuffer(inQueueGamma_, 1, rLengthWithPadding * sizeof(float));
        pipe_.InitBuffer(inQueueBeta_, 1, rLengthWithPadding * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, totalSize * sizeof(float));
        pipe_.InitBuffer(outQueueMean_, 1, aLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueRstd_, 1, aLength_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn() {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma_.AllocTensor<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueBeta_.AllocTensor<float>();

        AscendC::DataCopy(xLocal, xGm_, aLength_ * rLengthWithPadding_);
        AscendC::DataCopy(gammaLocal, gammaGm_, rLengthWithPadding_);
        AscendC::DataCopy(betaLocal, betaGm_, rLengthWithPadding_);

        inQueueX_.EnQue(xLocal);
        inQueueGamma_.EnQue(gammaLocal);
        inQueueBeta_.EnQue(betaLocal);
    }

    __aicore__ inline void Compute() {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma_.DeQue<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueBeta_.DeQue<float>();

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> meanLocal = outQueueMean_.AllocTensor<float>();
        AscendC::LocalTensor<float> rstdLocal = outQueueRstd_.AllocTensor<float>();

        const AscendC::LayerNormPara para = {aLength_, rLength_, rLengthWithPadding_};
        static constexpr AscendC::LayerNormConfig config = AscendC::GetConfig();
        AscendC::LayerNorm<float, float, false, config>(yLocal, meanLocal, rstdLocal,
                                                        xLocal, gammaLocal, betaLocal,
                                                        epsilon_, para, tiling_);

        outQueueY_.EnQue<float>(yLocal);
        outQueueMean_.EnQue<float>(meanLocal);
        outQueueRstd_.EnQue<float>(rstdLocal);

        inQueueX_.FreeTensor(xLocal);
        inQueueGamma_.FreeTensor(gammaLocal);
        inQueueBeta_.FreeTensor(betaLocal);
    }

    __aicore__ inline void CopyOut() {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        // Mean and rstd are not needed for output
        outQueueMean_.DeQue<float>();
        outQueueRstd_.DeQue<float>();

        AscendC::DataCopy(yGm_, yLocal, aLength_ * rLengthWithPadding_);

        outQueueY_.FreeTensor(yLocal);
        outQueueMean_.FreeTensor(outQueueMean_.DeQue<float>());
        outQueueRstd_.FreeTensor(outQueueRstd_.DeQue<float>());
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueGamma_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueBeta_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueMean_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueRstd_;

    uint32_t aLength_;
    uint32_t rLength_;
    uint32_t rLengthWithPadding_;
    float epsilon_;
    AscendC::LayerNormSeparateTiling tiling_;
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta,
                                                         GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.aLength,
            tiling_data.rLength,
            tiling_data.rLengthWithPadding,
            tiling_data.epsilon,
            tiling_data.layernormTilingData);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor layer_norm_impl_npu(const at::Tensor &x,
                                 const at::Tensor &gamma,
                                 const at::Tensor &beta,
                                 double epsilon) {
    at::Tensor result = at::empty_like(x);
    EXEC_NPU_CMD(aclnnLayerNormCustom, x, gamma, beta, epsilon, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("layer_norm_custom", &layer_norm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("layer_norm_custom", &layer_norm_impl_npu, "LayerNormalization (LayerNormCustom on NPU)");
}

model_src = '''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        self.ln = nn.LayerNorm(normalized_shape=normalized_shape)

    def forward(self, x):
        # Flatten gamma and beta to 1D for the custom op
        gamma = self.ln.weight.view(-1)
        beta = self.ln.bias.view(-1)
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, gamma, beta, self.ln.eps)
'''
