#include "layer_norm_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    const gert::StorageShape *xStorageShape = context->GetInputShape(0);
    const gert::Shape &xOriginShape = xStorageShape->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(xOriginShape.GetShapeSize());

    uint32_t dimNum = xOriginShape.GetDimNum();
    uint32_t normElements = 1;
    for (size_t i = 1; i < dimNum; ++i) {
        normElements *= static_cast<uint32_t>(xOriginShape.GetDim(i));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    float epsilon = epsPtr ? *epsPtr : 1e-5f;

    uint32_t totalBatch = totalLength / normElements;
    uint32_t blockDim = totalBatch;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 32) blockDim = 32;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_normElements(normElements);
    tiling.set_epsilon(epsilon);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t normElements, float epsilon)
    {
        totalLen_ = totalLength;
        normElems_ = normElements;
        epsilon_ = epsilon;
        invNormElems_ = 1.0f / normElems_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, normElems_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, normElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, normElems_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 5, normElems_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 2, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t numBatches = totalLen_ / normElems_;

        if (numBatches == 0) return;

        // Load gamma and beta once
        AscendC::LocalTensor<float> gammaLocal = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(gammaLocal, gammaGm_, normElems_);
        AscendC::LocalTensor<float> betaLocal = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(betaLocal, betaGm_, normElems_);

        // Workspace for ReduceSum (size normElems_)
        AscendC::LocalTensor<float> ws = tmpQueue_.AllocTensor<float>();

        // Persistent temporary tensors
        AscendC::LocalTensor<float> diffLocal = tmpQueue_.AllocTensor<float>();

        // Scalars
        AscendC::LocalTensor<float> scalarSum = scalarQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> scalarSumSq = scalarQueue_.AllocTensor<float>();

        // Process each batch element assigned to this block
        for (uint32_t batchIdx = blockIdx; batchIdx < numBatches; batchIdx += blockNum) {
            uint32_t gmOffset = batchIdx * normElems_;

            // Load x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], normElems_);

            // --- mean ---
            AscendC::ReduceSum(scalarSum, xLocal, ws, normElems_);
            AscendC::Muls(scalarSum, scalarSum, invNormElems_, 1);
            float meanVal = scalarSum.GetValue(0);

            // --- variance ---
            AscendC::Subs(diffLocal, xLocal, meanVal, normElems_);
            AscendC::Mul(diffLocal, diffLocal, diffLocal, normElems_);  // (x-mean)^2
            AscendC::ReduceSum(scalarSumSq, diffLocal, ws, normElems_);
            AscendC::Muls(scalarSumSq, scalarSumSq, invNormElems_, 1); // variance
            AscendC::Adds(scalarSumSq, scalarSumSq, epsilon_, 1);      // variance + epsilon
            AscendC::Sqrt(scalarSumSq, scalarSumSq, 1);                // std
            float stdVal = scalarSumSq.GetValue(0);

            // --- normalize ---
            AscendC::Subs(diffLocal, xLocal, meanVal, normElems_);      // recompute diff
            AscendC::LocalTensor<float> normLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Divs(normLocal, diffLocal, stdVal, normElems_);

            // --- gamma * norm + beta ---
            AscendC::Mul(normLocal, normLocal, gammaLocal, normElems_);
            AscendC::Add(normLocal, normLocal, betaLocal, normElems_);

            // --- write output ---
            AscendC::DataCopy(yGm_[gmOffset], normLocal, normElems_);

            // Free per-batch tensors
            tmpQueue_.FreeTensor(normLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        // Free persistent tensors
        tmpQueue_.FreeTensor(gammaLocal);
        tmpQueue_.FreeTensor(betaLocal);
        tmpQueue_.FreeTensor(ws);
        tmpQueue_.FreeTensor(diffLocal);
        scalarQueue_.FreeTensor(scalarSum);
        scalarQueue_.FreeTensor(scalarSumSq);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 5> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t normElems_;
    float epsilon_;
    float invNormElems_;
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta,
                                                         GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength,
            tiling_data.normElements,
            tiling_data.epsilon);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor layer_norm_impl_npu(const at::Tensor &x,
                                 const at::Tensor &gamma,
                                 const at::Tensor &beta,
                                 double epsilon) {
    at::Tensor result = at::empty_like(x);
    EXEC_NPU_CMD(aclnnLayerNormCustom, x, gamma, beta, epsilon, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("layer_norm_custom", &layer_norm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("layer_norm_custom", &layer_norm_impl_npu, "LayerNormalization (LayerNormCustom on NPU)");
}

model_src = '''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        self.ln = nn.LayerNorm(normalized_shape=normalized_shape)

    def forward(self, x):
        gamma = self.ln.weight.view(-1)
        beta = self.ln.bias.view(-1)
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, gamma, beta, self.ln.eps)
'''
