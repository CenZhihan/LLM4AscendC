#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(xShape->GetOriginShape().GetShapeSize());

    // normalized_shape is the product of the last (dimNum-1) dimensions
    uint32_t dimNum = xShape->GetOriginShape().GetDimNum();
    uint32_t normElements = 1;
    for (size_t i = 1; i < dimNum; ++i) {
        normElements *= static_cast<uint32_t>(xShape->GetOriginShape().GetDim(i));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    float epsilon = epsPtr ? *epsPtr : 1e-5f;

    // Simple block assignment: one block per batch element, capped at 32
    uint32_t totalBatch = totalLength / normElements;
    uint32_t blockDim = totalBatch;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 32) blockDim = 32;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_normElements(normElements);
    tiling.set_epsilon(epsilon);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)

    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    return GRAPH_SUCCESS;

    const ge::DataType x_dtype = context->GetInputDataType(0);

    return GRAPH_SUCCESS;
}

kernel_src = '''

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t normElements, float epsilon)
    {
        totalLen_ = totalLength;
        normElems_ = normElements;
        epsilon_ = epsilon;
        invNormElems_ = 1.0f / static_cast<float>(normElems_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, normElems_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, normElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // For each batch element we need temporary buffers for mean, variance, and normalized slice.
        // We'll process one batch element at a time.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, normElems_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, normElems_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, normElems_ * sizeof(float));  // mean, variance
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t numBatches = totalLen_ / normElems_;

        if (numBatches == 0) return;

        // Load gamma and beta once (they are the same for all batch elements)
        AscendC::LocalTensor<float> gammaLocal = tmpQueue_.AllocTensor<float>(); // reuse tmp
        AscendC::DataCopy(gammaLocal, gammaGm_, normElems_);
        AscendC::LocalTensor<float> betaLocal = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(betaLocal, betaGm_, normElems_);

        // Distribute batch elements across blocks
        for (uint32_t batchIdx = blockIdx; batchIdx < numBatches; batchIdx += blockNum) {
            uint32_t gmOffset = batchIdx * normElems_;

            // Copy x slice
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], normElems_);

            // Compute mean = sum(x) / normElems
            AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceAdd(sumLocal, xLocal, 1.0f, normElems_);
            AscendC::LocalTensor<float> meanLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Broadcast(meanLocal, sumLocal, normElems_);
            AscendC::Muls(meanLocal, meanLocal, invNormElems_, normElems_);

            // Compute variance = sum((x - mean)^2) / normElems
            AscendC::LocalTensor<float> diffLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(diffLocal, xLocal, meanLocal, normElems_);
            AscendC::Mul(diffLocal, diffLocal, diffLocal, normElems_);
            AscendC::LocalTensor<float> sumSqLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceAdd(sumSqLocal, diffLocal, 1.0f, normElems_);
            AscendC::Broadcast(diffLocal, sumSqLocal, normElems_); // reuse diff as variance
            AscendC::Muls(diffLocal, diffLocal, invNormElems_, normElems_);

            // stddev = sqrt(variance + epsilon)
            AscendC::Adds(diffLocal, diffLocal, epsilon_, normElems_);
            AscendC::Sqrt(diffLocal, diffLocal, normElems_);

            // normalized = (x - mean) / stddev
            // reuse xLocal as normalized (x - mean already in diff now? careful)
            // Recompute (x - mean) or use stored diff? We still have xLocal and meanLocal.
            AscendC::Sub(diffLocal, xLocal, meanLocal, normElems_); // diff = x - mean
            AscendC::Div(diffLocal, diffLocal, diffLocal, normElems_); // error: need stddev
            // Actually we need to divide by stddev. Use a different temp.
            // Let's use outQueueY tensor for temporary.
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Sub(yLocal, xLocal, meanLocal, normElems_); // yLocal = x - mean
            AscendC::Div(yLocal, yLocal, diffLocal, normElems_); // diffLocal currently holds stddev? Wait we overwrote diffLocal with recomputed (x-mean). We lost stddev.
            // Redesign: keep stddev in a separate tensor.
            // Better to recompute properly.
            // Since we have enough tmp, let's allocate another one.

            // Cleaner approach: recompute (x - mean) and divide by stddev using separate tensors.
            // We'll use a new local tensor for stddev.
            AscendC::LocalTensor<float> stddevLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Copy(stddevLocal, diffLocal, normElems_); // copy stddev from diffLocal before overwriting
            // Now compute normalized
            AscendC::Sub(yLocal, xLocal, meanLocal, normElems_);
            AscendC::Div(yLocal, yLocal, stddevLocal, normElems_);

            // y = gamma * normalized + beta
            AscendC::Mul(yLocal, yLocal, gammaLocal, normElems_);
            AscendC::Add(yLocal, yLocal, betaLocal, normElems_);

            // Copy out
            AscendC::DataCopy(yGm_[gmOffset], yLocal, normElems_);

            // Free used tensors
            tmpQueue_.FreeTensor(sumLocal);
            tmpQueue_.FreeTensor(meanLocal);
            tmpQueue_.FreeTensor(diffLocal);
            tmpQueue_.FreeTensor(sumSqLocal);
            tmpQueue_.FreeTensor(stddevLocal);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        tmpQueue_.FreeTensor(gammaLocal);
        tmpQueue_.FreeTensor(betaLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t normElems_;
    float epsilon_;
    float invNormElems_;
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta,
                                                         GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength,
            tiling_data.normElements,
            tiling_data.epsilon);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor layer_norm_impl_npu(const at::Tensor &x,
                                 const at::Tensor &gamma,
                                 const at::Tensor &beta,
                                 double epsilon) {
    at::Tensor result = at::empty_like(x);
    EXEC_NPU_CMD(aclnnLayerNormCustom, x, gamma, beta, epsilon, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("layer_norm_custom", &layer_norm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("layer_norm_custom", &layer_norm_impl_npu, "LayerNormalization (LayerNormCustom on NPU)");
}

model_src = '''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        self.ln = nn.LayerNorm(normalized_shape=normalized_shape)

    def forward(self, x):
        gamma = self.ln.weight.view(-1)
        beta = self.ln.bias.view(-1)
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, gamma, beta, self.ln.eps)
'''
