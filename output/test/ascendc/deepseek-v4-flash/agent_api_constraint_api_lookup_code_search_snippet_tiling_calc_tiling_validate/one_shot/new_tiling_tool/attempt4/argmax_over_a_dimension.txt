  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int64_t axis = attrs->GetAttrPointer<int64_t>(0)[0];
  if (axis < 0) axis += ndim;
  if (axis < 0 || axis >= ndim) axis = ndim - 1;

  gert::Shape outShape;
  uint32_t outRank = 0;
  for (int32_t i = 0; i < ndim; ++i) {

  uint32_t outIdx = 0;
  for (int32_t i = 0; i < ndim; ++i) {

      outShape.SetDim(outIdx, inShape->GetDim(i));
      outIdx++;
    }
  }
  *context->GetOutputShape(0) = outShape;
  return GRAPH_SUCCESS;

kernel_src='''
#include "kernel_operator.h"
#include <limits>

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
  __aicore__ inline KernelArgmaxOverADimensionCustom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                              uint32_t totalLength,
                              uint32_t outerSize,
                              uint32_t dimSize,
                              uint32_t innerSize,
                              int32_t axis) {
    totalLen_ = totalLength;
    outerSize_ = outerSize;
    dimSize_ = dimSize;
    innerSize_ = innerSize;
    axis_ = axis;

    xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
    yGm_.SetGlobalBuffer((__gm__ int64_t*)y, outerSize_ * innerSize_);

    // Allocate local buffer for one slice (dimSize floats, aligned to 32 bytes)
    alignedDimSize_ = (dimSize_ + 7) / 8 * 8;
    pipe_.InitBuffer(inQueueSrc_, BUFFER_NUM, alignedDimSize_ * sizeof(float));
    pipe_.InitBuffer(outQueueDst_, BUFFER_NUM, 1 * sizeof(int64_t));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0) return;

    uint32_t blockDim = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t numSlices = outerSize_ * innerSize_;
    if (numSlices == 0) return;

    // Block-level grid stride over slices
    uint32_t slicesPerBlock = (numSlices + blockDim - 1) / blockDim;
    uint32_t startSlice = blockIdx * slicesPerBlock;
    uint32_t endSlice = startSlice + slicesPerBlock;
    if (startSlice >= numSlices) return;
    if (endSlice > numSlices) endSlice = numSlices;

    for (uint32_t sl = startSlice; sl < endSlice; ++sl) {
      uint32_t outerIdx = sl / innerSize_;
      uint32_t innerIdx = sl % innerSize_;

      // Compute base offset for the first element in this slice along the reduce axis.
      // Memory layout: outerIdx * dimSize * innerSize + innerIdx (innermost offset)
      uint32_t base = outerIdx * dimSize_ * innerSize_ + innerIdx;

      // Copy the strided slice into contiguous local buffer.
      // Use DataCopyPad with row-by-row copy: each row = 1 float, stride = innerSize_ * sizeof(float)
      AscendC::LocalTensor<float> srcLocal = inQueueSrc_.AllocTensor<float>();
      AscendC::DataCopyExtParams copyParams = {
        static_cast<uint16_t>(dimSize_),      // rowNum
        sizeof(float),                        // colBytes (bytes per row)
        0,                                    // srcStride (not used when colBytes == total row size)
        0,                                    // dstStride
        0                                     // reserved
      };
      // For strided source, we need to use the overload with global tensor and stride
      // Actually we need the overload: DataCopyPad(dst, src, copyParams, padParams)
      // where copyParams has srcStride = (innerSize_ - 1) * sizeof(float) because each row is 1 float,
      // and the gap between rows = innerSize_ * sizeof(float) - sizeof(float)
      int16_t srcStride = static_cast<int16_t>((innerSize_ - 1) * sizeof(float));
      copyParams.srcStride = srcStride;
      AscendC::DataCopyPadExtParams<float> padParams = {false, 0, 0, 0};
      AscendC::DataCopyPad<float>(srcLocal, xGm_[base], copyParams, padParams);
      inQueueSrc_.EnQue(srcLocal);

      // Reduce: find argmax by iterating over elements
      AscendC::LocalTensor<float> src = inQueueSrc_.DeQue<float>();
      float globalMax = -std::numeric_limits<float>::infinity();
      int64_t globalIdx = 0;
      for (uint32_t i = 0; i < dimSize_; ++i) {
        float val = src.GetValue(i);
        if (val > globalMax) {
          globalMax = val;
          globalIdx = static_cast<int64_t>(i);
        }
      }

      AscendC::LocalTensor<int64_t> dst = outQueueDst_.AllocTensor<int64_t>();
      dst.SetValue(0, globalIdx);
      outQueueDst_.EnQue<int64_t>(dst);
      inQueueSrc_.FreeTensor(src);

      // Copy out
      AscendC::LocalTensor<int64_t> dstOut = outQueueDst_.DeQue<int64_t>();
      yGm_.SetValue(sl, dstOut.GetValue(0));
      outQueueDst_.FreeTensor(dstOut);
    }
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueSrc_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueDst_;
  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<int64_t> yGm_;
  uint32_t totalLen_ {0};
  uint32_t outerSize_ {0};
  uint32_t dimSize_ {0};
  uint32_t innerSize_ {0};
  uint32_t alignedDimSize_ {0};
  int32_t axis_ {-1};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelArgmaxOverADimensionCustom op;
  op.Init(x, y,
          tiling_data.totalLength,
          tiling_data.outerSize,
          tiling_data.dimSize,
          tiling_data.innerSize,
          tiling_data.axis);
  op.Process();
}

#include <torch/library.h>

#include "pytorch_npu_helper.hpp"

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor &self, int64_t dim) {
  auto shape = self.sizes();
  std::vector<int64_t> outShape;
  for (int64_t i = 0; i < (int64_t)shape.size(); ++i) {
    if (i != dim) outShape.push_back(shape[i]);
  }
  at::Tensor result = at::empty(outShape, self.options().dtype(at::kLong));
  EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result);
  return result;
}

  m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu);
}
