host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t FLOAT_NUM_OF_SINGLE_BLOCK = 8;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const auto &originShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rowLength = originShape.GetDim(0);
    uint32_t columnLength = originShape.GetDim(1);

    // UB budget: two queues (2*tile) + intermediate temps (3*tile) ~ 5*tile floats
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLength = maxTile;
    if (columnLength < tileLength) {
        tileLength = AlignDown(columnLength, ALIGN_ELEMS);
        if (tileLength == 0) tileLength = columnLength;
    }

    uint32_t usedNumBlocks = CeilDiv(columnLength, tileLength);
    if (usedNumBlocks < 8) usedNumBlocks = 8;
    if (usedNumBlocks > 24) usedNumBlocks = 24;

    uint32_t coreRowNum = CeilDiv(rowLength, usedNumBlocks);
    if (coreRowNum == 0) coreRowNum = 1;
    uint32_t singleLoopCoreRowNum = coreRowNum;
    if (singleLoopCoreRowNum > 8) singleLoopCoreRowNum = 8;
    uint32_t singleCoreLoopCount = CeilDiv(coreRowNum, singleLoopCoreRowNum);
    uint32_t leftRow = coreRowNum % singleLoopCoreRowNum;
    uint32_t tailCoreSingleLoopCoreRowNum = (leftRow == 0) ? singleLoopCoreRowNum : leftRow;
    uint32_t tailCoreSingleCoreLoopCount = CeilDiv(coreRowNum, tailCoreSingleLoopCoreRowNum);
    uint32_t tailCoreSingleCoreLoopTail = coreRowNum % tailCoreSingleLoopCoreRowNum;

    // sharedTmpBuffer for LogSoftMax temporary workspace (60KB typical)
    uint32_t sharedTmpBufferSize = 60 * 1024;

    context->SetBlockDim(usedNumBlocks);

    tiling.set_rowLength(rowLength);
    tiling.set_columnLength(columnLength);
    tiling.set_sharedTmpBufferSize(sharedTmpBufferSize);
    tiling.set_usedNumBlocks(usedNumBlocks);
    tiling.set_coreRowNum(coreRowNum);
    tiling.set_singleLoopCoreRowNum(singleLoopCoreRowNum);
    tiling.set_singleCoreLoopCount(singleCoreLoopCount);
    tiling.set_leftRow(leftRow);
    tiling.set_tailCoreSingleLoopCoreRowNum(tailCoreSingleLoopCoreRowNum);
    tiling.set_tailCoreSingleCoreLoopCount(tailCoreSingleCoreLoopCount);
    tiling.set_tailCoreSingleCoreLoopTail(tailCoreSingleCoreLoopTail);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t FLOAT_NUM_OF_SINGLE_BLOCK = 8;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                const LogSoftmaxCustomTilingData& tilingData)
    {
        rowLength_ = tilingData.rowLength;
        columnLength_ = tilingData.columnLength;
        usedNumBlocks_ = tilingData.usedNumBlocks;
        coreRowNum_ = tilingData.coreRowNum;
        singleLoopCoreRowNum_ = tilingData.singleLoopCoreRowNum;
        singleCoreLoopCount_ = tilingData.singleCoreLoopCount;
        leftRow_ = tilingData.leftRow;
        tailCoreSingleLoopCoreRowNum_ = tilingData.tailCoreSingleLoopCoreRowNum;
        tailCoreSingleCoreLoopCount_ = tilingData.tailCoreSingleCoreLoopCount;
        tailCoreSingleCoreLoopTail_ = tilingData.tailCoreSingleCoreLoopTail;
        sharedTmpBufferSize_ = tilingData.sharedTmpBufferSize;

        if (AscendC::GetBlockIdx() == this->usedNumBlocks_) {
            this->singleLoopCoreRowNum_ = this->tailCoreSingleLoopCoreRowNum_;
            this->singleCoreLoopCount_ = this->tailCoreSingleCoreLoopCount_;
            this->leftRow_ = this->tailCoreSingleCoreLoopTail_;
        }

        this->blockLength_ = this->coreRowNum_ * this->columnLength_;
        this->msLength_ = this->coreRowNum_ * FLOAT_NUM_OF_SINGLE_BLOCK;

        uint32_t offset = this->blockLength_ * AscendC::GetBlockIdx();
        xGm_.SetGlobalBuffer((__gm__ float*)x + offset, this->blockLength_);
        yGm_.SetGlobalBuffer((__gm__ float*)y + offset, this->blockLength_);

        this->tileLength_ = this->singleLoopCoreRowNum_ * this->columnLength_;
        this->msTileLength_ = this->singleLoopCoreRowNum_ * FLOAT_NUM_OF_SINGLE_BLOCK;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, this->tileLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, this->tileLength_ * sizeof(float));
        pipe_.InitBuffer(sharedTmpBuffer_, sharedTmpBufferSize_);
    }

    __aicore__ inline void Process()
    {
        if (AscendC::GetBlockIdx() > this->usedNumBlocks_) {
            return;
        }
        for (int32_t i = 0; i < this->singleCoreLoopCount_; i++) {
            CopyIn(i, this->singleLoopCoreRowNum_);
            Compute(i, this->singleLoopCoreRowNum_);
            CopyOut(i, this->singleLoopCoreRowNum_);
        }
        if (this->leftRow_ > 0) {
            CopyIn(this->singleCoreLoopCount_, this->leftRow_);
            Compute(this->singleCoreLoopCount_, this->leftRow_);
            CopyOut(this->singleCoreLoopCount_, this->leftRow_);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress, uint32_t rowNum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[progress * this->tileLength_], rowNum * this->columnLength_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progressm, uint32_t rowNum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maxLocal = sharedTmpBuffer_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = sharedTmpBuffer_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tmpBuffer = sharedTmpBuffer_.Get<uint8_t>();

        AscendC::SoftMaxShapeInfo srcShape = {rowNum, this->columnLength_, rowNum, this->columnLength_};
        AscendC::LogSoftMax<float>(xLocal, sumLocal, maxLocal, xLocal, tmpBuffer,
                                   this->softmaxTiling_, srcShape);

        sharedTmpBuffer_.FreeTensor(maxLocal);
        sharedTmpBuffer_.FreeTensor(sumLocal);
        outQueueY_.EnQue<float>(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress, uint32_t rowNum)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[progress * this->tileLength_], yLocal, rowNum * this->columnLength_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> sharedTmpBuffer_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    LogSoftMaxTiling softmaxTiling_;

    uint32_t rowLength_ = 0;
    uint32_t columnLength_ = 0;
    uint32_t usedNumBlocks_ = 0;
    uint32_t coreRowNum_ = 0;
    uint32_t blockLength_ = 0;
    uint32_t msLength_ = 0;
    uint32_t tileLength_ = 0;
    uint32_t msTileLength_ = 0;
    uint32_t sharedTmpBufferSize_ = 0;
    uint32_t singleLoopCoreRowNum_ = 0;
    uint32_t singleCoreLoopCount_ = 0;
    uint32_t leftRow_ = 0;
    uint32_t tailCoreSingleLoopCoreRowNum_ = 0;
    uint32_t tailCoreSingleCoreLoopCount_ = 0;
    uint32_t tailCoreSingleCoreLoopTail_ = 0;
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data);
    op.Process();
}
