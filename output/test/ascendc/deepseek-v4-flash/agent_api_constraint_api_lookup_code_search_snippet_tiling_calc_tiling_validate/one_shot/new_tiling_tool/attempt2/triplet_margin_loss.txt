#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: 4 local tensors (anchor, positive, negative, diff) each of tileDimF floats
// Additional small workspace for scalar reduction (negligible)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;   // 32 bytes align for float

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;
    // All inputs have same shape: (N, D). Get N from dim 0.
    uint32_t totalTriplets = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(0));
    uint32_t featureDim = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // Compute tileDimF from UB budget (4 buffers * tileDimF * 4 bytes)
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * static_cast<uint32_t>(sizeof(float)));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileDimF = maxTile;
    if (featureDim < tileDimF) tileDimF = featureDim;

    // Single block for simplicity (no cross-block synchronization needed)
    uint32_t blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalTriplets(totalTriplets);
    tiling.set_featureDim(featureDim);
    tiling.set_tileDimF(tileDimF);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
                                uint32_t totalTriplets, uint32_t featureDim,
                                uint32_t tileDimF, float margin)
    {
        totalTriplets_ = totalTriplets;
        featureDim_ = featureDim;
        tileDimF_ = tileDimF;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalTriplets_ * featureDim_);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalTriplets_ * featureDim_);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalTriplets_ * featureDim_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        // Buffers for one tile of feature dimension
        pipe_.InitBuffer(anchorQueue_, BUFFER_NUM, tileDimF_ * sizeof(float));
        pipe_.InitBuffer(positiveQueue_, BUFFER_NUM, tileDimF_ * sizeof(float));
        pipe_.InitBuffer(negativeQueue_, BUFFER_NUM, tileDimF_ * sizeof(float));
        pipe_.InitBuffer(diffQueue_, BUFFER_NUM, tileDimF_ * sizeof(float));
        // Workspace for ReduceSum
        pipe_.InitBuffer(reduceWq_, BUFFER_NUM, tileDimF_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalTriplets_ == 0 || featureDim_ == 0) return;

        float totalLoss = 0.0f;

        // Iterate over each triplet
        for (uint32_t t = 0; t < totalTriplets_; t++) {
            float sumApSq = 0.0f;
            float sumAnSq = 0.0f;

            // Process feature dimension in tiles
            for (uint32_t f = 0; f < featureDim_; f += tileDimF_) {
                uint32_t curTileLen = featureDim_ - f;
                if (curTileLen > tileDimF_) curTileLen = tileDimF_;

                // Load anchor, positive, negative tiles
                AscendC::LocalTensor<float> aLocal = anchorQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> pLocal = positiveQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> nLocal = negativeQueue_.AllocTensor<float>();

                uint32_t baseOffset = t * featureDim_ + f;
                AscendC::DataCopy(aLocal, anchorGm_[baseOffset], curTileLen);
                AscendC::DataCopy(pLocal, positiveGm_[baseOffset], curTileLen);
                AscendC::DataCopy(nLocal, negativeGm_[baseOffset], curTileLen);

                // Compute squared Euclidean distance for anchor-positive
                AscendC::LocalTensor<float> diffAp = diffQueue_.AllocTensor<float>();
                AscendC::Sub(diffAp, aLocal, pLocal, curTileLen);
                AscendC::Mul(diffAp, diffAp, diffAp, curTileLen);
                // ReduceSum
                AscendC::LocalTensor<float> wsAp = reduceWq_.AllocTensor<float>();
                AscendC::ReduceSum(diffAp, diffAp, wsAp, curTileLen);  // result in diffAp[0]
                reduceWq_.FreeTensor(wsAp);
                sumApSq += diffAp.GetValue(0);

                // Compute squared Euclidean distance for anchor-negative
                AscendC::LocalTensor<float> diffAn = diffQueue_.AllocTensor<float>();
                AscendC::Sub(diffAn, aLocal, nLocal, curTileLen);
                AscendC::Mul(diffAn, diffAn, diffAn, curTileLen);
                AscendC::LocalTensor<float> wsAn = reduceWq_.AllocTensor<float>();
                AscendC::ReduceSum(diffAn, diffAn, wsAn, curTileLen);
                reduceWq_.FreeTensor(wsAn);
                sumAnSq += diffAn.GetValue(0);

                // Free tensors
                anchorQueue_.FreeTensor(aLocal);
                positiveQueue_.FreeTensor(pLocal);
                negativeQueue_.FreeTensor(nLocal);
                diffQueue_.FreeTensor(diffAp);
                diffQueue_.FreeTensor(diffAn);
            }

            // Compute Euclidean distances using vector sqrt on scalars
            float normAp, normAn;
            {
                AscendC::LocalTensor<float> tmp = diffQueue_.AllocTensor<float>();
                tmp.SetValue(0, sumApSq);
                AscendC::Sqrt(tmp, tmp, 1);
                normAp = tmp.GetValue(0);
                diffQueue_.FreeTensor(tmp);
            }
            {
                AscendC::LocalTensor<float> tmp = diffQueue_.AllocTensor<float>();
                tmp.SetValue(0, sumAnSq);
                AscendC::Sqrt(tmp, tmp, 1);
                normAn = tmp.GetValue(0);
                diffQueue_.FreeTensor(tmp);
            }

            // Compute loss per triplet: max(normAp - normAn + margin, 0)
            float lossPer = normAp - normAn + margin_;
            if (lossPer < 0.0f) lossPer = 0.0f;
            totalLoss += lossPer;
        }

        // Mean loss
        totalLoss = totalLoss / (float)totalTriplets_;

        // Write scalar output
        AscendC::LocalTensor<float> lossLocal = diffQueue_.AllocTensor<float>();
        lossLocal.SetValue(0, totalLoss);
        AscendC::DataCopy(lossGm_, lossLocal, 1);
        diffQueue_.FreeTensor(lossLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> anchorQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> positiveQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> negativeQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> diffQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> reduceWq_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalTriplets_;
    uint32_t featureDim_;
    uint32_t tileDimF_;
    float margin_;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive,
                                                  GM_ADDR negative, GM_ADDR loss,
                                                  GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalTriplets, tiling_data.featureDim,
            tiling_data.tileDimF, tiling_data.margin);
    op.Process();
}
