static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape *inShape = context->GetInputShape(0);
  int32_t ndim = inShape->GetDimNum();

  // Read axis attribute (same as tiling)
  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int64_t axis = attrs->GetAttrPointer<int64_t>(0)[0];  // attr dim
  if (axis < 0) axis += ndim;
  if (axis < 0 || axis >= ndim) axis = ndim - 1; // fallback

  // Build output shape: remove the axis dimension
  gert::Shape outShape;
  uint32_t outRank = 0;
  for (int32_t i = 0; i < ndim; ++i) {
    if (i != axis) {
      outRank++;
    }
  }
  outShape.SetDimNum(outRank);
  uint32_t outIdx = 0;
  for (int32_t i = 0; i < ndim; ++i) {
    if (i != axis) {
      outShape.SetDim(outIdx, inShape->GetDim(i));
      outIdx++;
    }
  }
  *context->GetOutputShape(0) = outShape;
  return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MASK_FLOAT = 64;   // float mask size (elements in one 256-bit vector)

class KernelArgmaxOverADimensionCustom {
public:
  __aicore__ inline KernelArgmaxOverADimensionCustom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                              uint32_t totalLength,
                              uint32_t outerSize,
                              uint32_t dimSize,
                              uint32_t innerSize,
                              int32_t axis) {
    totalLen_ = totalLength;
    outerSize_ = outerSize;
    dimSize_ = dimSize;
    innerSize_ = innerSize;
    axis_ = axis;

    xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
    yGm_.SetGlobalBuffer((__gm__ int64_t*)y, outerSize_ * innerSize_);

    // Allocate double-buffered tiles for input slices
    pipe_.InitBuffer(inQueueSrc_, BUFFER_NUM, dimSize_ * sizeof(float));
    pipe_.InitBuffer(outQueueDst_, BUFFER_NUM, 1 * sizeof(int64_t)); // one index per slice
    pipe_.InitBuffer(tmpQueue_, 1, (dimSize_ / MASK_FLOAT + 1) * 2 * sizeof(float)); // max and index storage
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0) return;

    uint32_t blockDim = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t numSlices = outerSize_ * innerSize_;
    if (numSlices == 0) return;

    // Block-level grid stride over slices
    uint32_t slicesPerBlock = (numSlices + blockDim - 1) / blockDim;
    uint32_t startSlice = blockIdx * slicesPerBlock;
    uint32_t endSlice = startSlice + slicesPerBlock;
    if (startSlice >= numSlices) return;
    if (endSlice > numSlices) endSlice = numSlices;

    for (uint32_t sl = startSlice; sl < endSlice; ++sl) {
      // Convert slice index to outer_idx and inner_idx
      uint32_t outerIdx = sl / innerSize_;
      uint32_t innerIdx = sl % innerSize_;

      // Compute global offset of the slice elements along the reduction axis
      // For a 3D tensor (outer, dim, inner) in row-major:
      // base = outerIdx * dimSize * innerSize_ + innerIdx
      // elements at positions base + i*innerSize_ for i=0..dimSize-1 (if axis is middle)
      // But we assume axis is the innermost for contiguous access.
      // If axis is not innermost, we handle it generically using strides.
      // Here we implement only the case axis == 2 (innermost) for ndim=3.
      // For general axis, we would need to compute stride = innerSize (if axis is middle).
      // For simplicity we assume axis is the last dimension (innermost).
      uint32_t stride = 1; // contiguous
      uint32_t baseOffset = (axis_ == 2) ? (outerIdx * dimSize_ * innerSize_ + innerIdx) :
                                          (outerIdx * dimSize_ * innerSize_ + innerIdx * stride);

      // Copy the slice data into local buffer
      AscendC::LocalTensor<float> srcLocal = inQueueSrc_.AllocTensor<float>();
      // Use DataCopyExtParams to copy one contiguous block of dimSize_ float32 elements
      AscendC::DataCopyExtParams copyParams = {1, dimSize_ * sizeof(float), 0, 0, 0};
      AscendC::DataCopyPad<float>(srcLocal, xGm_[baseOffset], copyParams);
      inQueueSrc_.EnQue(srcLocal);

      // Compute
      AscendC::LocalTensor<float> src = inQueueSrc_.DeQue<float>();
      AscendC::LocalTensor<int64_t> dst = outQueueDst_.AllocTensor<int64_t>();
      AscendC::LocalTensor<float> work = tmpQueue_.AllocTensor<float>();

      // Use WholeReduceMax with returning index
      uint32_t repeat = (dimSize_ + MASK_FLOAT - 1) / MASK_FLOAT;
      uint32_t mask = (repeat == 1) ? dimSize_ : MASK_FLOAT; // last repeat may be partial? Handle with padding
      // To handle non-power-of-two dimSize, we pad with -inf? Instead use mask[2] bitmask
      uint64_t maskBits[2] = {0, 0};
      if (dimSize_ <= 64) {
        maskBits[0] = (1ULL << dimSize_) - 1;
      } else if (dimSize_ <= 128) {
        maskBits[0] = 0xFFFFFFFFFFFFFFFF;
        maskBits[1] = (1ULL << (dimSize_ - 64)) - 1;
      } else {
        maskBits[0] = 0xFFFFFFFFFFFFFFFF;
        maskBits[1] = 0xFFFFFFFFFFFFFFFF;
        // dimSize >128 not supported in this simple implementation
      }

      // Perform WholeReduceMax; destination will contain (value, index) interleaved per repeat
      AscendC::WholeReduceMax<float>(work, src, maskBits, repeat, 1, 1, 8);
      AscendC::PipeBarrier<PIPE_V>();

      // Combine results from multiple repeats using GetReduceRepeatMaxMinSpr
      float maxVal = 0.0f;
      int64_t globalIdx = 0;
      AscendC::GetReduceRepeatMaxMinSpr<float>(maxVal, globalIdx);
      AscendC::PipeBarrier<PIPE_V>();

      // The returned globalIdx is the linear index within the whole slice (0..dimSize_-1)
      dst.SetValue(0, globalIdx);

      outQueueDst_.EnQue<int64_t>(dst);
      inQueueSrc_.FreeTensor(src);
      tmpQueue_.FreeTensor(work);

      // Copy out
      AscendC::LocalTensor<int64_t> dstOut = outQueueDst_.DeQue<int64_t>();
      // Output is 1 int64 per slice; store at position sl
      yGm_.SetValue(sl, dstOut.GetValue(0));
      outQueueDst_.FreeTensor(dstOut);
    }
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueSrc_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueDst_;
  AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<int64_t> yGm_;
  uint32_t totalLen_ {0};
  uint32_t outerSize_ {0};
  uint32_t dimSize_ {0};
  uint32_t innerSize_ {0};
  int32_t axis_ {-1};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelArgmaxOverADimensionCustom op;
  op.Init(x, y,
          tiling_data.totalLength,
          tiling_data.outerSize,
          tiling_data.dimSize,
          tiling_data.innerSize,
          tiling_data.axis);
  op.Process();
}

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor &self, int64_t dim) {

  std::vector<int64_t> outShape;
