host_operator_src="""

#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const auto &a_shape = context->GetInputShape(0)->GetOriginShape();
    const auto &b_shape = context->GetInputShape(1)->GetOriginShape();
    int64_t M = a_shape.GetDim(0);
    int64_t K = a_shape.GetDim(1);
    int64_t N = b_shape.GetDim(1);

    context->SetBlockDim(8);

    TCubeTiling cubeTiling;
    cubeTiling.M = M;
    cubeTiling.N = N;
    cubeTiling.K = K;
    cubeTiling.baseM = 128;
    cubeTiling.baseN = 128;
    cubeTiling.baseK = 128;
    cubeTiling.aType = ge::DT_FLOAT;
    cubeTiling.bType = ge::DT_FLOAT;
    cubeTiling.cType = ge::DT_FLOAT;
    cubeTiling.biasType = ge::DT_FLOAT;
    cubeTiling.formatA = ge::FORMAT_ND;
    cubeTiling.formatB = ge::FORMAT_ND;
    cubeTiling.formatC = ge::FORMAT_ND;
    cubeTiling.hasBias = false;

    tiling.set_cubeTiling(cubeTiling);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    int64_t M = a_shape->GetDim(0);
    int64_t N = b_shape->GetDim(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class TallSkinnyMatrixMultiplicationCustom : public OpDef {
public:
    explicit TallSkinnyMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("a")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("b")
            .ParamType(REQUIRED)

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, const TCubeTiling& tiling)
    {
        using A_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
        using B_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
        using C_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
        using BIAS_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float>;

        matmulObj.Init(a, b, nullptr, c, workspace, tiling);
    }

    __aicore__ inline void Process(AscendC::TPipe* pipe)
    {
        matmulObj.Process(pipe);
    }

    AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> matmulObj;

private:
    using A_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using B_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using C_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float>;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm)
{
    if (ASCEND_IS_AIV) {
        return;
    }

    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), kernel.matmulObj, &tiling);

    KernelTallSkinnyMatMul kernel;
    kernel.Init(a, b, c, workspace, tiling);
    kernel.Process(&pipe);
}
"""

#include <torch/library.h>

at::Tensor tall_skinny_matmul_impl_npu(const at::Tensor& a, const at::Tensor& b) {
    at::Tensor result = at::empty({a.size(0), b.size(1)}, a.options());
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, a, b, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu,
          "Matrix multiplication for tall-skinny matrices (TallSkinnyMatrixMultiplicationCustom on NPU)");
}
