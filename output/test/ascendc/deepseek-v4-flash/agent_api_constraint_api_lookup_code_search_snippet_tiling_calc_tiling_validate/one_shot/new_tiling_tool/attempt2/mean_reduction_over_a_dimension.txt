    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape *inputShape = context->GetInputShape(0);
    std::vector<int64_t> shape = inputShape->GetOriginShape().GetDims();
    uint32_t totalLength = 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    // Normalize dim
    int32_t ndim = static_cast<int32_t>(shape.size());
    if (dim < 0) dim += ndim;

    // Reduction size = shape[dim]
    uint32_t reductionSize = static_cast<uint32_t>(shape[dim]);

    // Output size = totalLength / reductionSize
    uint32_t outputSize = totalLength / reductionSize;

    // Simple block dim based on output tiles
    uint32_t blockDim = outputSize;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_reductionSize(reductionSize);
    tiling.set_outputSize(outputSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    std::vector<int64_t> dims = x_shape->GetOriginShape().GetDims();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = (attrs != nullptr) ? attrs->GetAttrPointer<int64_t>(0)[0] : 0;
    int32_t ndim = static_cast<int32_t>(dims.size());
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= ndim) dim = ndim - 1;

    std::vector<int64_t> outDims;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i != dim) outDims.push_back(dims[i]);
    }
    if (outDims.empty()) outDims.push_back(1); // scalar case

    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = gert::Shape(outDims);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t reductionSize,
                                uint32_t outputSize)
    {
        totalLen_ = totalLength;
        redSize_ = reductionSize;
        outSize_ = outputSize;
        rowCount_ = outSize_; // rows = output elements

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outSize_);

        // Allocate enough UB for one reduction row + workspace
        pipe_.InitBuffer(inQueueX_, 1, redSize_ * sizeof(float));
        pipe_.InitBuffer(workspaceBuf_, 1, redSize_ * sizeof(float));
        // No out queue, we write directly
    }

    __aicore__ inline void Process()
    {
        if (rowCount_ == 0 || redSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t startRow = blockIdx;
        while (startRow < rowCount_) {
            uint32_t gmOffset = startRow * redSize_;
            // Copy a row from GM to local
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], redSize_);

            // Allocate workspace
            AscendC::LocalTensor<float> wLocal = workspaceBuf_.AllocTensor<float>();

            // Reduce sum
            AscendC::LocalTensor<float> sumLocal;
            AscendC::ReduceSum(sumLocal, xLocal, wLocal, redSize_);
            // sumLocal is a scalar (1 element). Divide by redSize_
            float invRedSize = 1.0f / static_cast<float>(redSize_);
            AscendC::Muls(sumLocal, sumLocal, invRedSize, 1);

            // Write scalar to output
            AscendC::DataCopy(yGm_[startRow], sumLocal, 1);

            workspaceBuf_.FreeTensor(wLocal);
            inQueueX_.FreeTensor(xLocal);

            startRow += blockNum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workspaceBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t redSize_ {0};
    uint32_t outSize_ {0};
    uint32_t rowCount_ {0};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.reductionSize, tiling_data.outputSize);
    op.Process();
}
