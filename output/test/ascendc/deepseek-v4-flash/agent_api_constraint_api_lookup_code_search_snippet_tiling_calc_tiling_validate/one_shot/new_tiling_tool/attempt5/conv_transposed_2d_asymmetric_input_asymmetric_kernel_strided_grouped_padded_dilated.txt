namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    const int64_t *stride = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *output_padding = attrs->GetAttrPointer<int64_t>(2);
    int64_t groups = *attrs->GetAttrPointer<int64_t>(3);
    const int64_t *dilation = attrs->GetAttrPointer<int64_t>(4);

    int64_t N = x_shape->GetDim(0);
    int64_t C_in = x_shape->GetDim(1);
    int64_t H_in = x_shape->GetDim(2);
    int64_t W_in = x_shape->GetDim(3);
    int64_t C_out = w_shape->GetDim(1) * groups;
    int64_t kH = w_shape->GetDim(2);
    int64_t kW = w_shape->GetDim(3);

    int64_t H_out = (H_in - 1) * stride[0] - 2 * padding[0] + dilation[0] * (kH - 1) + output_padding[0] + 1;
    int64_t W_out = (W_in - 1) * stride[1] - 2 * padding[1] + dilation[1] * (kW - 1) + output_padding[1] + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C_out);
    y_shape->SetDim(2, H_out);
    y_shape->SetDim(3, W_out);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
