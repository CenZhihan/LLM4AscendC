static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    VanillaRnnHiddenCustomTilingData tiling;

    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::StorageShape *wI2hShape = context->GetInputShape(2);
    const gert::StorageShape *wH2oShape = context->GetInputShape(4);

    const gert::Shape &xOrigin = xShape->GetOriginShape();
    const gert::Shape &wI2hOrigin = wI2hShape->GetOriginShape();
    const gert::Shape &wH2oOrigin = wH2oShape->GetOriginShape();

    uint32_t T = xOrigin.GetDim(0);
    uint32_t B = xOrigin.GetDim(1);
    uint32_t I = xOrigin.GetDim(2);
    uint32_t H = wI2hOrigin.GetDim(0);
    uint32_t O = wH2oOrigin.GetDim(0);
    uint32_t K = I + H;

    tiling.set_totalSteps(T);
    tiling.set_batchSize(B);
    tiling.set_hiddenSize(H);
    tiling.set_inputSize(I);
    tiling.set_outputSize(O);
    tiling.set_concatSize(K);

    uint32_t tileK = 512;
    uint32_t tileH = 256;
    uint32_t tileO = 128;
    tileK = (tileK / ALIGN) * ALIGN;
    tileH = (tileH / ALIGN) * ALIGN;
    tileO = (tileO / ALIGN) * ALIGN;

    tiling.set_tileK(tileK);
    tiling.set_tileH(tileH);
    tiling.set_tileO(tileO);

    uint32_t blockDim = T;
    if (blockDim > 24) blockDim = 24;

kernel_src = '''
#include "kernel_operator.h"
#include "lib/matmul.h"
#include "adv_api/math/tanh.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                VanillaRnnHiddenCustomTilingData tiling)
    {
        totalSteps_ = tiling.totalSteps;
        batchSize_ = tiling.batchSize;
        hiddenSize_ = tiling.hiddenSize;
        inputSize_ = tiling.inputSize;
        outputSize_ = tiling.outputSize;
        concatSize_ = tiling.concatSize;
        tileK_ = tiling.tileK;
        tileH_ = tiling.tileH;
        tileO_ = tiling.tileO;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalSteps_ * batchSize_ * inputSize_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, batchSize_ * hiddenSize_);
        wI2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, hiddenSize_ * concatSize_);
        bI2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, hiddenSize_);
        wH2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, outputSize_ * hiddenSize_);
        bH2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, outputSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalSteps_ * batchSize_ * outputSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, batchSize_ * inputSize_ * sizeof(float));
        pipe_.InitBuffer(localH_, 1, batchSize_ * hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(localY_, 1, batchSize_ * outputSize_ * sizeof(float));
        pipe_.InitBuffer(tmp1_, 1, batchSize_ * hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(tmp2_, 1, batchSize_ * outputSize_ * sizeof(float));
        pipe_.InitBuffer(weightTileI2h_, 1, tileK_ * tileH_ * sizeof(float));
        pipe_.InitBuffer(weightTileH2o_, 1, tileH_ * tileO_ * sizeof(float));
        pipe_.InitBuffer(sharedTmp_, 1, batchSize_ * hiddenSize_ * sizeof(uint8_t));
    }

    __aicore__ inline void Process()
    {
        if (totalSteps_ == 0 || batchSize_ == 0) return;

        const uint32_t blockIdx = GetBlockIdx();
        const uint32_t blockNum = GetBlockNum();

        for (uint32_t t = blockIdx; t < totalSteps_; t += blockNum)
        {
            if (t == 0)
            {
                DataCopy(localH_, h0Gm_, batchSize_ * hiddenSize_);
            }

            auto xLocal = inQueueX_.AllocTensor<float>();
            uint32_t offsetX = t * batchSize_ * inputSize_;
            DataCopy(xLocal, xGm_[offsetX], batchSize_ * inputSize_);
            inQueueX_.EnQue(xLocal);

            SetZero(tmp1_, batchSize_ * hiddenSize_);

            uint32_t I_tiles = (inputSize_ + tileK_ - 1) / tileK_;
            for (uint32_t i = 0; i < I_tiles; ++i)
            {
                uint32_t ki = i * tileK_;
                uint32_t curK = min(tileK_, inputSize_ - ki);
                ComputeI2hPart(xLocal, localH_, ki, curK, wI2hGm_, bI2hGm_, tmp1_);
            }

            uint32_t H_tiles = (hiddenSize_ + tileK_ - 1) / tileK_;
            for (uint32_t j = 0; j < H_tiles; ++j)
            {
                uint32_t kj = j * tileK_;
                uint32_t curK = min(tileK_, hiddenSize_ - kj);
                ComputeI2hPartH(localH_, inputSize_ + kj, curK, wI2hGm_, bI2hGm_, tmp1_);
            }

            Add(tmp1_, tmp1_, bI2hGm_, batchSize_ * hiddenSize_);

            Tanh(localH_, tmp1_, sharedTmp_);

            SetZero(tmp2_, batchSize_ * outputSize_);
            for (uint32_t j = 0; j < H_tiles; ++j)
            {
                uint32_t hj = j * tileH_;
                uint32_t curH = min(tileH_, hiddenSize_ - hj);
                ComputeH2oPart(localH_, hj, curH, wH2oGm_, bH2oGm_, tmp2_);
            }
            Add(tmp2_, tmp2_, bH2oGm_, batchSize_ * outputSize_);

            uint32_t offsetY = t * batchSize_ * outputSize_;
            DataCopy(yGm_[offsetY], tmp2_, batchSize_ * outputSize_);

            auto xLocal2 = inQueueX_.DeQue<float>();
            inQueueX_.FreeTensor(xLocal2);
        }
    }

private:
    __aicore__ inline void ComputeI2hPart(const LocalTensor<float>& xLocal,
                                          const LocalTensor<float>& hLocal,
                                          uint32_t kStart, uint32_t kLen,
                                          const GlobalTensor<float>& wGm,
                                          const GlobalTensor<float>& bGm,
                                          LocalTensor<float>& out)
    {
        for (uint32_t b = 0; b < batchSize_; ++b)
        {
            for (uint32_t h = 0; h < hiddenSize_; ++h)
            {
                float sum = 0.0f;
                for (uint32_t k = 0; k < kLen; ++k)
                {
                    uint32_t idxW = h * concatSize_ + kStart + k;
                    float w = wGm.GetValue(idxW);
                    float inp_val = (kStart + k < inputSize_) ?
                        xLocal[b * inputSize_ + k] :
                        hLocal[b * hiddenSize_ + (kStart + k - inputSize_)];
                    sum += w * inp_val;
                }
                out[b * hiddenSize_ + h] += sum;
            }
        }
    }

    __aicore__ inline void ComputeI2hPartH(const LocalTensor<float>& hLocal,
                                           uint32_t kStart, uint32_t kLen,
                                           const GlobalTensor<float>& wGm,
                                           const GlobalTensor<float>& bGm,
                                           LocalTensor<float>& out)
    {
        ComputeI2hPart(hLocal, hLocal, kStart, kLen, wGm, bGm, out);
    }

    __aicore__ inline void ComputeH2oPart(const LocalTensor<float>& hLocal,
                                          uint32_t hStart, uint32_t hLen,
                                          const GlobalTensor<float>& wGm,
                                          const GlobalTensor<float>& bGm,
                                          LocalTensor<float>& out)
    {
        for (uint32_t b = 0; b < batchSize_; ++b)
        {
            for (uint32_t o = 0; o < outputSize_; ++o)
            {
                float sum = 0.0f;
                for (uint32_t h = 0; h < hLen; ++h)
                {
                    float w = wGm.GetValue(o * hiddenSize_ + hStart + h);
                    sum += w * hLocal[b * hiddenSize_ + hStart + h];
                }
                out[b * outputSize_ + o] += sum;
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    LocalTensor<float> localH_;
    LocalTensor<float> localY_;
    LocalTensor<float> tmp1_, tmp2_;
    LocalTensor<float> weightTileI2h_, weightTileH2o_;
    LocalTensor<uint8_t> sharedTmp_;

    GlobalTensor<float> xGm_, h0Gm_, wI2hGm_, bI2hGm_, wH2oGm_, bH2oGm_, yGm_;

    uint32_t totalSteps_, batchSize_, hiddenSize_, inputSize_, outputSize_, concatSize_;
    uint32_t tileK_, tileH_, tileO_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(
    GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y, tiling_data);
    op.Process();
}
'''

#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor vanilla_rnn_hidden_impl_npu(
    const at::Tensor& x,
    const at::Tensor& h0,
    const at::Tensor& w_i2h,
    const at::Tensor& b_i2h,
    const at::Tensor& w_h2o,
    const at::Tensor& b_h2o)

    auto T = x.size(0);
    auto B = x.size(1);
    auto O = w_h2o.size(0);
    at::Tensor result = at::empty({T, B, O}, x.options());
    EXEC_NPU_CMD(aclnnVanillaRnnHiddenCustom, x, h0, w_i2h, b_i2h, w_h2o, b_h2o, result);
    return result;

}
