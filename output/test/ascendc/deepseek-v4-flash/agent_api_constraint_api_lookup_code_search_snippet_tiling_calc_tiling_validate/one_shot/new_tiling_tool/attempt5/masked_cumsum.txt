#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MAX_TILE_ROWS = 128;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalLength = static_cast<uint32_t>(shape.GetShapeSize());

    // dim is the third input as a scalar tensor
    const int32_t *dimPtr = context->GetInputTensor(2)->GetData<int32_t>();
    int32_t axis = *dimPtr;
    int32_t ndim = static_cast<int32_t>(shape.GetDimNum());
    if (axis < 0) axis += ndim;
    uint32_t uaxis = static_cast<uint32_t>(axis);
    uint32_t dimSize = static_cast<uint32_t>(shape.GetDim(uaxis));
    uint32_t numRows = totalLength / dimSize;

    // Compute tileRows from UB budget
    uint32_t bytesPerRow = dimSize * sizeof(float) * 5;
    uint32_t maxRows = (bytesPerRow == 0) ? 1 : (UB_BUDGET_BYTES / bytesPerRow);
    if (maxRows > MAX_TILE_ROWS) maxRows = MAX_TILE_ROWS;
    if (maxRows > numRows) maxRows = numRows;
    if (maxRows == 0) maxRows = 1;

    uint32_t tileRows = AlignDown(maxRows, 1u);

    uint32_t blockDim = CeilDiv(numRows, tileRows);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_dimSize(dimSize);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t dimSize, uint32_t tileRows)
    {
        totalLen_ = totalLength;
        dimSize_ = dimSize;
        tileRows_ = tileRows;
        numRows_ = totalLength / dimSize;
        rowStride_ = dimSize * sizeof(float);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ uint8_t*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        uint32_t bufSize = tileRows_ * dimSize_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, bufSize * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, bufSize * sizeof(uint8_t));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, bufSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, bufSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileRows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideRows = blockNum * tileRows_;
        uint32_t rowBase = blockIdx * tileRows_;
        if (rowBase >= numRows_) return;

        while (true) {
            uint32_t curRows = numRows_ - rowBase;
            if (curRows > tileRows_) curRows = tileRows_;

            uint32_t nextRowBase = rowBase + strideRows;
            uint32_t nextRows = 0;
            if (nextRowBase < numRows_) {
                nextRows = numRows_ - nextRowBase;
                if (nextRows > tileRows_) nextRows = tileRows_;
                CopyIn(nextRowBase * dimSize_, nextRows);
            }

            CopyIn(rowBase * dimSize_, curRows);
            Compute(curRows);
            CopyOut(rowBase * dimSize_, curRows);

            if (nextRows == 0) break;
            rowBase = nextRowBase;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t elemOffset, uint32_t rows)
    {
        uint32_t len = rows * dimSize_;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> maskLocal = inQueueMask_.AllocTensor<uint8_t>();
        AscendC::DataCopy(xLocal, xGm_[elemOffset], len);
        AscendC::DataCopy(maskLocal, maskGm_[elemOffset], len);
        inQueueX_.EnQue(xLocal);
        inQueueMask_.EnQue(maskLocal);
    }

    __aicore__ inline void Compute(uint32_t rows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<uint8_t> maskLocal = inQueueMask_.DeQue<uint8_t>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

        uint32_t totalElems = rows * dimSize_;

        // convert mask from bool (uint8) to float
        AscendC::Cast(tmpLocal, maskLocal, totalElems, AscendC::CAST_ROUND);
        AscendC::Mul(yLocal, xLocal, tmpLocal, totalElems);

        // manual cumulative sum per row (along dimSize axis)
        for (uint32_t r = 0; r < rows; ++r) {
            uint32_t rowOffset = r * dimSize_;
            // first element remains
            for (uint32_t c = 1; c < dimSize_; ++c) {
                uint32_t idx = rowOffset + c;
                float prev = yLocal.GetValue(idx - 1);
                float curr = yLocal.GetValue(idx);
                yLocal.SetValue(idx, prev + curr);
            }
        }

        tmpQueue_.FreeTensor(tmpLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueMask_.FreeTensor(maskLocal);
    }

    __aicore__ inline void CopyOut(uint32_t elemOffset, uint32_t rows)
    {
        uint32_t len = rows * dimSize_;
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[elemOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<uint8_t> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t dimSize_ {0};
    uint32_t tileRows_ {0};
    uint32_t numRows_ {0};
    uint32_t rowStride_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR dim_gm, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.totalLength, tiling_data.dimSize, tiling_data.tileRows);
    op.Process();
}
