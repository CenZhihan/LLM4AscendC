static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const auto &origin = inputShape->GetOriginShape();
    int32_t ndim = static_cast<int32_t>(origin.GetDimNum());
    if (ndim != 5) {
        return ge::GRAPH_FAILED;
    }

    int32_t inN = static_cast<int32_t>(origin.GetDim(0));
    int32_t inC = static_cast<int32_t>(origin.GetDim(1));
    int32_t inD = static_cast<int32_t>(origin.GetDim(2));
    int32_t inH = static_cast<int32_t>(origin.GetDim(3));
    int32_t inW = static_cast<int32_t>(origin.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    // Attr order: 0: kernel_size, 1: strides, 2: pads, 3: dilation, 4: ceil_mode
    const int64_t *ksize = attrs->GetAttrPointer<int64_t>(0);
    int32_t kernelD = static_cast<int32_t>(ksize[0]);
    int32_t kernelH = static_cast<int32_t>(ksize[1]);
    int32_t kernelW = static_cast<int32_t>(ksize[2]);

    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    int32_t strideD = static_cast<int32_t>(stride[0]);
    int32_t strideH = static_cast<int32_t>(stride[1]);
    int32_t strideW = static_cast<int32_t>(stride[2]);

    const int64_t *pads = attrs->GetAttrPointer<int64_t>(2);
    int32_t padD = static_cast<int32_t>(pads[0]);
    int32_t padH = static_cast<int32_t>(pads[1]);
    int32_t padW = static_cast<int32_t>(pads[2]);

    const int64_t *dil = attrs->GetAttrPointer<int64_t>(3);
    int32_t dilationD = static_cast<int32_t>(dil[0]);
    int32_t dilationH = static_cast<int32_t>(dil[1]);
    int32_t dilationW = static_cast<int32_t>(dil[2]);

    bool ceilMode = attrs->GetAttrPointer<bool>(4)[0];

    auto outSize = [&](int32_t inSize, int32_t pad, int32_t kSize, int32_t stride, int32_t dilation) -> int32_t {
        int32_t d = (inSize + 2 * pad - dilation * (kSize - 1) - 1);
        if (ceilMode) {
            return (d + stride - 1) / stride + 1;
        } else {
            return d / stride + 1;
        }
    };

    int32_t outD = outSize(inD, padD, kernelD, strideD, dilationD);
    int32_t outH = outSize(inH, padH, kernelH, strideH, dilationH);
    int32_t outW = outSize(inW, padW, kernelW, strideW, dilationW);

    uint32_t totalElements = static_cast<uint32_t>(inN) * static_cast<uint32_t>(inC)

    uint32_t blockDim = 32;
    uint32_t tileElements = (totalElements + blockDim - 1) / blockDim;
    tileElements = AlignDown(tileElements, ALIGN_ELEMS);
    if (tileElements == 0) tileElements = ALIGN_ELEMS;
    if (tileElements > 8192) tileElements = AlignDown(8192, ALIGN_ELEMS);

    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_kernelD(kernelD);
    tiling.set_kernelH(kernelH);
    tiling.set_kernelW(kernelW);
    tiling.set_strideD(strideD);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padD(padD);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_dilationD(dilationD);
    tiling.set_dilationH(dilationH);
    tiling.set_dilationW(dilationW);
    tiling.set_ceilMode(ceilMode ? 1 : 0);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());

    currentWorkspace[0] = 0;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalElements, uint32_t tileElements,
                                uint32_t blockDim,
                                int32_t inN, int32_t inC, int32_t inD, int32_t inH, int32_t inW,
                                int32_t outD, int32_t outH, int32_t outW,
                                int32_t kernelD, int32_t kernelH, int32_t kernelW,
                                int32_t strideD, int32_t strideH, int32_t strideW,
                                int32_t padD, int32_t padH, int32_t padW,
                                int32_t dilationD, int32_t dilationH, int32_t dilationW,
                                uint8_t ceilMode)
    {
        totalElements_ = totalElements;
        tileElements_ = tileElements;
        blockDim_ = blockDim;
        inN_ = inN; inC_ = inC; inD_ = inD; inH_ = inH; inW_ = inW;
        outD_ = outD; outH_ = outH; outW_ = outW;
        kernelD_ = kernelD; kernelH_ = kernelH; kernelW_ = kernelW;
        strideD_ = strideD; strideH_ = strideH; strideW_ = strideW;
        padD_ = padD; padH_ = padH; padW_ = padW;
        dilationD_ = dilationD; dilationH_ = dilationH; dilationW_ = dilationW;
        ceilMode_ = ceilMode;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElements_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElements_);

        constexpr uint32_t maxKernelElems = 27;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, maxKernelElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileElements_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalElements_ == 0 || tileElements_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startIdx = blockIdx * tileElements_;
        uint32_t endIdx = startIdx + tileElements_;
        if (startIdx >= totalElements_) return;
        if (endIdx > totalElements_) endIdx = totalElements_;

        uint32_t cnt = endIdx - startIdx;
        if (cnt == 0) return;

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < cnt; i++) {
            uint32_t linearIdx = startIdx + i;
            int32_t ow = linearIdx % outW_;
            int32_t oh = (linearIdx / outW_) % outH_;
            int32_t od = (linearIdx / (outW_ * outH_)) % outD_;
            int32_t c = (linearIdx / (outW_ * outH_ * outD_)) % inC_;
            int32_t n = linearIdx / (outW_ * outH_ * outD_ * inC_);

            int32_t startD = od * strideD_ - padD_;
            int32_t startH = oh * strideH_ - padH_;
            int32_t startW = ow * strideW_ - padW_;

            float maxVal = -__builtin_huge_valf();
            for (int32_t kd = 0; kd < kernelD_; kd++) {
                int32_t id = startD + kd * dilationD_;
                if (id < 0 || id >= inD_) continue;
                for (int32_t kh = 0; kh < kernelH_; kh++) {
                    int32_t ih = startH + kh * dilationH_;
                    if (ih < 0 || ih >= inH_) continue;
                    for (int32_t kw = 0; kw < kernelW_; kw++) {
                        int32_t iw = startW + kw * dilationW_;
                        if (iw < 0 || iw >= inW_) continue;

                        uint32_t inputOffset = ((n * inC_ + c) * inD_ + id) * inH_ * inW_ + ih * inW_ + iw;
                        float val = xGm_.GetValue(inputOffset);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }
            yLocal.SetValue(i, maxVal);
        }

        AscendC::DataCopy(yGm_[startIdx], yLocal, cnt);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalElements_ {0};
    uint32_t tileElements_ {0};
    uint32_t blockDim_ {0};
    int32_t inN_, inC_, inD_, inH_, inW_;
    int32_t outD_, outH_, outW_;
    int32_t kernelD_, kernelH_, kernelW_;
    int32_t strideD_, strideH_, strideW_;
    int32_t padD_, padH_, padW_;
    int32_t dilationD_, dilationH_, dilationW_;
    uint8_t ceilMode_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.totalElements, tiling_data.tileElements, tiling_data.blockDim,
            tiling_data.inN, tiling_data.inC, tiling_data.inD, tiling_data.inH, tiling_data.inW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW,
            tiling_data.dilationD, tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.ceilMode);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size=3, stride=2, padding=1, ceil_mode=False):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.ceil_mode = ceil_mode

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        if isinstance(self.kernel_size, int):
            ks = [self.kernel_size] * 3
        else:
            ks = list(self.kernel_size)
        if isinstance(self.stride, int):
            st = [self.stride] * 3
        else:
            st = list(self.stride)
        if isinstance(self.padding, int):
            pd = [self.padding] * 3
        else:
            pd = list(self.padding)
        dil = [1, 1, 1]
        return fn(x, ks, st, pd, dil, self.ceil_mode)
'''
