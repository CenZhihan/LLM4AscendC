#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t ROWS_PER_TILE = 8;

static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return (x + a - 1) / a * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LayerNormCustomTilingData tiling;
    const uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    const uint32_t gammaLength = context->GetInputShape(1)->GetOriginShape().GetShapeSize();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilonPtr = attrs->GetAttrPointer<float>(0);  // epsilon is attr index 0

    uint32_t rLength = gammaLength;  // gamma shape equals normalized_shape
    uint32_t aLength = totalLength / rLength;

    uint32_t rLengthAligned = AlignUp(rLength, ALIGN_ELEMS);

    uint32_t maxRowsInUB = UB_BUDGET_BYTES / (24 * rLengthAligned * sizeof(float));
    if (maxRowsInUB > ROWS_PER_TILE) maxRowsInUB = ROWS_PER_TILE;
    if (maxRowsInUB < 1) maxRowsInUB = 1;
    uint32_t tileARows = maxRowsInUB;

    tiling.set_aLength(aLength);
    tiling.set_rLength(rLength);
    tiling.set_rLengthAligned(rLengthAligned);
    tiling.set_tileARows(tileARows);
    tiling.set_totalLength(totalLength);
    tiling.set_blockDim(8);
    tiling.set_epsilon(*epsilonPtr);

    context->SetBlockDim(8);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;

    const ge::DataType x_dtype = context->GetInputDataType(0);

kernel_src = """
#include "kernel_operator.h"
#include "adv_api/normalization/layernorm.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t ALIGN = 64;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t aLength, uint32_t rLength, uint32_t rLengthAligned,
                                uint32_t tileARows, float epsilon) {
        aLen_ = aLength;
        rLen_ = rLength;
        rLenAligned_ = rLengthAligned;
        tileRows_ = tileARows;
        epsilon_ = epsilon;
        rLenFloat_ = static_cast<float>(rLen_);
        totalRows_ = aLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, aLen_ * rLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, rLen_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, rLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, aLen_ * rLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileRows_ * rLenAligned_ * sizeof(float));
        pipe_.InitBuffer(inQueueG_, 1, rLenAligned_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, rLenAligned_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * rLenAligned_ * sizeof(float));
        // temporary buffers for mean, variance, and shared workspace (uint8_t)
        uint32_t meanVarSize = Align(tileRows_, ALIGN);
        uint32_t sharedSize = tileRows_ * rLenAligned_ * sizeof(float);
        pipe_.InitBuffer(tmpQueue_, 1, (meanVarSize * 2 + sharedSize) * sizeof(float));
        pipe_.InitBuffer(sharedQueue_, 1, sharedSize);  // for uint8_t shared buffer
    }

    __aicore__ inline void Process() {
        if (totalRows_ == 0) return;
        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();

        CopyGammaBeta();

        uint32_t rowStart = blockIdx * tileRows_;
        while (rowStart < totalRows_) {
            uint32_t tileRows = (rowStart + tileRows_ <= totalRows_) ? tileRows_ : (totalRows_ - rowStart);
            CopyInX(rowStart, tileRows);
            Compute(rowStart, tileRows);
            CopyOutY(rowStart, tileRows);
            rowStart += blockNum * tileRows_;
        }
    }

private:
    __aicore__ inline void CopyGammaBeta() {
        AscendC::LocalTensor<float> gammaLocal = inQueueG_.AllocTensor<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueB_.AllocTensor<float>();
        AscendC::DataCopy(gammaLocal, gammaGm_, rLen_);
        AscendC::DataCopy(betaLocal, betaGm_, rLen_);
        if (rLenAligned_ > rLen_) {
            float zero = 0.0f;
            AscendC::Muls(gammaLocal[rLen_], gammaLocal[rLen_], zero, rLenAligned_ - rLen_);
            AscendC::Muls(betaLocal[rLen_], betaLocal[rLen_], zero, rLenAligned_ - rLen_);
        }
        inQueueG_.EnQue(gammaLocal);
        inQueueB_.EnQue(betaLocal);
    }

    __aicore__ inline void CopyInX(uint32_t rowOffset, uint32_t rows) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        uint32_t offset = rowOffset * rLen_;
        for (uint32_t r = 0; r < rows; r++) {
            AscendC::DataCopy(xLocal[r * rLenAligned_], xGm_[offset + r * rLen_], rLen_);
            if (rLenAligned_ > rLen_) {
                float zero = 0.0f;
                AscendC::Muls(xLocal[r * rLenAligned_ + rLen_], xLocal[r * rLenAligned_ + rLen_], zero, rLenAligned_ - rLen_);
            }
        }
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rowOffset, uint32_t rows) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> gammaLocal = inQueueG_.DeQue<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueB_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

        uint32_t meanVarSize = Align(rows, ALIGN);
        AscendC::LocalTensor<float> mean = tmp[0];
        AscendC::LocalTensor<float> var = tmp[meanVarSize];
        AscendC::LocalTensor<uint8_t> sharedBuf = sharedQueue_.AllocTensor<uint8_t>();

        AscendC::LayerNormTiling layerNormTiling;
        layerNormTiling.bLength = rows;
        layerNormTiling.sLength = 1;
        layerNormTiling.hLength = rLen_;

        AscendC::LayerNorm(yLocal, mean, var, xLocal, gammaLocal, betaLocal, sharedBuf, epsilon_, layerNormTiling);

        sharedQueue_.FreeTensor(sharedBuf);
        tmpQueue_.FreeTensor(tmp);
        inQueueG_.FreeTensor(gammaLocal);
        inQueueB_.FreeTensor(betaLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutY(uint32_t rowOffset, uint32_t rows) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint32_t offset = rowOffset * rLen_;
        for (uint32_t r = 0; r < rows; r++) {
            AscendC::DataCopy(yGm_[offset + r * rLen_], yLocal[r * rLenAligned_], rLen_);
        }
        outQueueY_.FreeTensor(yLocal);
    }

    static __aicore__ inline uint32_t Align(uint32_t x, uint32_t a) {
        return (x + a - 1) / a * a;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueG_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sharedQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t aLen_;
    uint32_t rLen_;
    uint32_t rLenAligned_;
    uint32_t tileRows_;
    float epsilon_;
    float rLenFloat_;
    uint32_t totalRows_;
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                                         GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.aLength,
            tiling_data.rLength,
            tiling_data.rLengthAligned,
            tiling_data.tileARows,
            tiling_data.epsilon);
    op.Process();
}
"""

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
