host_operator_src="""
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    auto a_shape = context->GetInputShape(0)->GetOriginShape();
    auto b_shape = context->GetInputShape(1)->GetOriginShape();
    int64_t M = a_shape.GetDim(0);
    int64_t K = a_shape.GetDim(1);
    int64_t N = b_shape.GetDim(1);

    // For tall-skinny, use simple blocking: 1 block, total elements = M*N
    uint32_t totalLength = static_cast<uint32_t>(M * N);
    uint32_t tileLength = 4096;  // small tile to fit UB
    if (tileLength > totalLength) tileLength = totalLength;
    // Align to 32 for float accesses
    tileLength = (tileLength / 32) * 32;
    if (tileLength == 0) tileLength = totalLength;
    uint32_t blockDim = 1;  // Single block multiplication using cube -> one core

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    int64_t M = a_shape->GetDim(0);
    int64_t N = b_shape->GetDim(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class TallSkinnyMatrixMultiplicationCustom : public OpDef {
public:
    explicit TallSkinnyMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("a")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("b")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("c")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(TallSkinnyMatrixMultiplicationCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"
#include "matmul.h"

using namespace AscendC;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                GM_ADDR workspace, GM_ADDR tilingGm)
    {
        TCubeTiling tiling;
        CopyTiling(&tiling, tilingGm);

        matmulObj.Init(a, b, nullptr, c, workspace, tiling);
    }

    __aicore__ inline void Process(AscendC::TPipe* pipe)
    {
        matmulObj.Process(pipe);
    }

    AscendC::Matmul<
        MatmulType<TPosition::GM, CubeFormat::ND, float, false>,
        MatmulType<TPosition::GM, CubeFormat::ND, float, false>,
        MatmulType<TPosition::GM, CubeFormat::ND, float, false>,
        MatmulType<TPosition::GM, CubeFormat::ND, float>
    > matmulObj;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm)
{
    if (ASCEND_IS_AIV) {
        return;
    }

    AscendC::TPipe pipe;
    KernelTallSkinnyMatMul kernel;
    kernel.Init(a, b, c, workspace, tilingGm);
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), kernel.matmulObj, tilingGm);
    kernel.Process(&pipe);
}
