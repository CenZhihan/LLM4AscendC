host_operator_src = '''
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // per core
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;
static constexpr uint32_t FLOAT_NUM_OF_SINGLE_BLOCK = 64; // for max/sum intermediate

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t Min(uint32_t a, uint32_t b) { return a < b ? a : b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::StorageType st = context->GetInputShape(0)->GetStorageType();
    const gert::Shape &xShape = (st == gert::STORAGE_TYPE_ORIGIN)
                                    ? context->GetInputShape(0)->GetOriginShape()
                                    : context->GetInputShape(0)->GetStorageShape();
    const uint32_t totalLength = static_cast<uint32_t>(xShape.GetShapeSize());
    const uint32_t ndim = static_cast<uint32_t>(xShape.GetDimNum());
    uint32_t dim = 1;
    if (ndim >= 2) {
        const gert::RuntimeAttrs *attrs = context->GetAttrs();
        dim = (ndim > 0 && attrs != nullptr) ? static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(0)[0]) : 1;
    }
    // Only support 2D logsoftmax along last dimension (dim=1 or ndim-1)
    uint32_t rowLength = xShape.GetDim(0);
    uint32_t columnLength = (ndim >= 2) ? xShape.GetDim(1) : 1;

    // Estimate ub usage per tile: We need: xLocal, yLocal, tmpBuffer (for muls/add etc), plus sum/max locals.
    // Use same budget as softmax example: ~ 20*elems bytes.
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // We tile by rows: each tile processes tileRowNum rows.
    uint32_t tileRowNum = maxTile / columnLength;
    if (tileRowNum == 0) tileRowNum = 1;
    if (tileRowNum > rowLength) tileRowNum = rowLength;

    uint32_t tileLength = tileRowNum * columnLength; // elements per tile
    uint32_t coreRowNum = tileRowNum;                // default per block per loop
    uint32_t numTiles = CeilDiv(rowLength, tileRowNum);

    // Simple block distribution: each block handles some rows.
    uint32_t blockDim = Min(numTiles, 24);
    if (blockDim < 8) blockDim = 8;
    context->SetBlockDim(blockDim);

    uint32_t rowsPerBlock = CeilDiv(rowLength, blockDim);
    uint32_t usedNumBlocks = CeilDiv(rowLength, rowsPerBlock);
    uint32_t coreRowNumFull = rowsPerBlock;
    uint32_t singleLoopCoreRowNum = coreRowNumFull;
    uint32_t singleCoreLoopCount = 1;  // all rows in one iteration if possible
    uint32_t leftRow = 0;
    // For simplicity, handle only a single loop per block; if rows exceed tile, break into multiple loops.
    // For now, assume all rows fit in one tile per block (i.e., tileRowNum >= rowsPerBlock)
    // If not, we need multi-loop logic. Adjust:
    if (rowsPerBlock > tileRowNum) {
        singleLoopCoreRowNum = tileRowNum;
        singleCoreLoopCount = rowsPerBlock / tileRowNum;
        leftRow = rowsPerBlock % tileRowNum;
    }

    // Tail core adjustments (if using fewer blocks than total)
    uint32_t tailCoreSingleLoopCoreRowNum = singleLoopCoreRowNum;
    uint32_t tailCoreSingleCoreLoopCount = singleCoreLoopCount;
    uint32_t tailCoreSingleCoreLoopTail = leftRow;
    if (usedNumBlocks < blockDim) {
        uint32_t tailRows = rowLength % rowsPerBlock;
        if (tailRows > 0) {
            tailCoreSingleLoopCoreRowNum = tailRows > tileRowNum ? tileRowNum : tailRows;
            tailCoreSingleCoreLoopCount = (tailRows > tileRowNum) ? (tailRows / tileRowNum) : 1;
            tailCoreSingleCoreLoopTail = tailRows % tileRowNum;
        }
    }

    uint32_t sharedTmpBufferSize = 61440; // 60KB

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_columnLength(columnLength);
    tiling.set_dim(dim);
    tiling.set_coreRowNum(coreRowNum);
    tiling.set_tileRowNum(tileRowNum);
    tiling.set_loopCount(singleCoreLoopCount);
    tiling.set_tileLength(tileLength);
    tiling.set_usedNumBlocks(usedNumBlocks);
    tiling.set_sharedTmpBufferSize(sharedTmpBufferSize);
    tiling.set_singleLoopCoreRowNum(singleLoopCoreRowNum);
    tiling.set_singleCoreLoopCount(singleCoreLoopCount);
    tiling.set_leftRow(leftRow);
    tiling.set_tailCoreSingleLoopCoreRowNum(tailCoreSingleLoopCoreRowNum);
    tiling.set_tailCoreSingleCoreLoopCount(tailCoreSingleCoreLoopCount);
    tiling.set_tailCoreSingleCoreLoopTail(tailCoreSingleCoreLoopTail);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''
#include "kernel_operator.h"
#include "adv_api/activation/logsoftmax.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t FLOAT_NUM_OF_SINGLE_BLOCK = 64;  // for max/sum intermediate (not used globally)

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, const optiling::LogSoftmaxCustomTilingData& tilingData,
                                AscendC::TPipe* pipeIn)
    {
        pipe = pipeIn;
        totalLength = tilingData.totalLength;
        rowLength = tilingData.rowLength;
        columnLength = tilingData.columnLength;
        usedNumBlocks = tilingData.usedNumBlocks;
        coreRowNum = tilingData.coreRowNum;
        tileRowNum = tilingData.tileRowNum;
        singleLoopCoreRowNum = tilingData.singleLoopCoreRowNum;
        singleCoreLoopCount = tilingData.singleCoreLoopCount;
        leftRow = tilingData.leftRow;
        tailCoreSingleLoopCoreRowNum = tilingData.tailCoreSingleLoopCoreRowNum;
        tailCoreSingleCoreLoopCount = tilingData.tailCoreSingleCoreLoopCount;
        tailCoreSingleCoreLoopTail = tilingData.tailCoreSingleCoreLoopTail;
        sharedTmpBufferSize = tilingData.sharedTmpBufferSize;

        ASSERT(AscendC::GetBlockNum() != 0 && "block dim can not be zero!");

        // Adjust for tail core if needed
        if (AscendC::GetBlockIdx() == this->usedNumBlocks) {
            this->singleLoopCoreRowNum = this->tailCoreSingleLoopCoreRowNum;
            this->singleCoreLoopCount = this->tailCoreSingleCoreLoopCount;
            this->leftRow = this->tailCoreSingleCoreLoopTail;
        }

        this->blockLength = this->coreRowNum * this->columnLength;
        this->tileLength = this->tileRowNum * this->columnLength;

        xGm.SetGlobalBuffer((__gm__ float*)x, totalLength);
        zGm.SetGlobalBuffer((__gm__ float*)y, totalLength);

        // Use pipeline: queueX for input, queueY for output, queueMax/Sum for intermediate
        // (max/sum are not written to global)
        pipe->InitBuffer(queueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe->InitBuffer(queueY, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe->InitBuffer(queueMax, 1, this->tileRowNum * FLOAT_NUM_OF_SINGLE_BLOCK * sizeof(float));
        pipe->InitBuffer(queueSum, 1, this->tileRowNum * FLOAT_NUM_OF_SINGLE_BLOCK * sizeof(float));
        pipe->InitBuffer(sharedTmpBuffer, sharedTmpBufferSize);
    }

    __aicore__ inline void Process()
    {
        if (AscendC::GetBlockIdx() > this->usedNumBlocks) {
            return;
        }

        uint32_t progress = 0;
        // Loop over full row tiles
        for (uint32_t i = 0; i < this->singleCoreLoopCount; i++) {
            CopyIn(i, this->singleLoopCoreRowNum);
            Compute(i, this->singleLoopCoreRowNum);
            CopyOut(i, this->singleLoopCoreRowNum);
            progress = i + 1;
        }
        // Handle leftover rows
        if (this->leftRow > 0) {
            CopyIn(progress, this->leftRow);
            Compute(progress, this->leftRow);
            CopyOut(progress, this->leftRow);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress, uint32_t rowNum)
    {
        AscendC::LocalTensor<float> xLocal = queueX.AllocTensor<float>();
        uint32_t offset = progress * this->tileLength;
        AscendC::DataCopy(xLocal, xGm[offset], rowNum * this->columnLength);
        queueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress, uint32_t rowNum)
    {
        AscendC::LocalTensor<float> xLocal = queueX.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = queueY.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = queueMax.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = queueSum.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tmpBuffer = sharedTmpBuffer.Get<uint8_t>();

        // Prepare shape info: rows, cols
        AscendC::SoftMaxShapeInfo srcShape = {rowNum, this->columnLength, rowNum, this->columnLength};
        // Use LogSoftMax API (default tiling, adjust if needed)
        // For simplicity, use a default constructed LogSoftMaxTiling
        AscendC::LogSoftMaxTiling tiling;  // default all zeros should work

        if (rowNum % 2 == 0 && this->columnLength % 64 == 0 && this->columnLength < 4096) {
            AscendC::LogSoftMax<float, true, true>(yLocal, sumLocal, maxLocal, xLocal, tmpBuffer, tiling, srcShape);
        } else {
            AscendC::LogSoftMax<float, true>(yLocal, sumLocal, maxLocal, xLocal, tmpBuffer, tiling, srcShape);
        }

        queueY.EnQue<float>(yLocal);
        queueMax.FreeTensor(maxLocal);
        queueSum.FreeTensor(sumLocal);
        queueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress, uint32_t rowNum)
    {
        AscendC::LocalTensor<float> yLocal = queueY.DeQue<float>();
        uint32_t offset = progress * this->tileLength;
        AscendC::DataCopy(zGm[offset], yLocal, rowNum * this->columnLength);
        queueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe* pipe;
    AscendC::TBuf<AscendC::TPosition::VECCALC> sharedTmpBuffer;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> queueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> queueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> queueMax, queueSum;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t totalLength = 0;
    uint32_t rowLength = 0;
    uint32_t columnLength = 0;
    uint32_t coreRowNum = 0;
    uint32_t usedNumBlocks = 0;
    uint32_t tileRowNum = 0;
    uint32_t tileLength = 0;
    uint32_t blockLength = 0;
    uint32_t sharedTmpBufferSize = 0;
    uint32_t singleLoopCoreRowNum = 0;
    uint32_t singleCoreLoopCount = 0;
    uint32_t leftRow = 0;
    uint32_t tailCoreSingleLoopCoreRowNum = 0;
    uint32_t tailCoreSingleCoreLoopCount = 0;
    uint32_t tailCoreSingleCoreLoopTail = 0;
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    AscendC::TPipe pipe;
    op.Init(x, y, tiling_data, &pipe);
    op.Process();
}
'''
