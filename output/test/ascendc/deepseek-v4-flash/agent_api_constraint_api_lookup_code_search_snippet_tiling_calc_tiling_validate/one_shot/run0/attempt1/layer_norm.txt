#include "register/tilingdata_base.h"

namespace optiling {

// Structure matching the kernel's VecTiling (used by CopyTiling)
struct LayerNormVecTiling {
  uint32_t aLength;
  uint32_t rLength;
  uint32_t rLengthWithPadding;
  float epsilon;
  uint32_t layernormTilingData[21]; // opaque for LayerNormSeparateTiling
};

BEGIN_TILING_DATA_DEF(LayerNormCustomTilingData)
  TILING_DATA_FIELD_DEF(uint32_t, aLength);
  TILING_DATA_FIELD_DEF(uint32_t, rLength);
  TILING_DATA_FIELD_DEF(uint32_t, rLengthWithPadding);
  TILING_DATA_FIELD_DEF(float, epsilon);
  TILING_DATA_FIELD_DEF(uint32_t, layernormTilingData[21]);
END_TILING_DATA_DEF;

REGISTER_TILING_DATA_CLASS(LayerNormCustom, LayerNormCustomTilingData)
}  // namespace optiling

namespace {

static constexpr uint32_t BLOCK_SIZE = 32;
static constexpr uint32_t SIZEOF_FLOAT = 4;
constexpr uint32_t LAYERNORM_ONE_BLK_SIZE = BLOCK_SIZE;
constexpr uint32_t LAYERNORM_ONE_NUMBER = 1;
constexpr uint32_t LAYERNORM_ZERO_NUMBER = 0;
constexpr uint32_t LAYERNORM_TWO_TIMES = 2;

void GetLayerNormNDTilingInfo(const std::vector<int64_t>& shapeDims,
                              const uint32_t stackBufferSize,
                              const uint32_t typeSize,
                              const bool isReuseSource,
                              const bool isComputeRstd,
                              uint32_t* layernormTilingData)
{
    (void)isReuseSource;
    (void)isComputeRstd;
    const uint32_t aLength = static_cast<uint32_t>(shapeDims[0]);
    const uint32_t rLength = static_cast<uint32_t>(shapeDims[1]);
    int32_t align = BLOCK_SIZE / typeSize;
    uint32_t rLengthWithPadding = (rLength + align - 1) / align * align;

    const uint32_t inputXSize = aLength * rLengthWithPadding;
    const uint32_t meanVarSize = aLength;
    uint32_t numberOfTmpBuf = LAYERNORM_TWO_TIMES;
    constexpr uint32_t oneBlockNum = LAYERNORM_ONE_BLK_SIZE / SIZEOF_FLOAT;
    constexpr uint32_t varianceTmpTensorPos = LAYERNORM_ZERO_NUMBER;
    const uint32_t varianceTmpTensorSize = (meanVarSize + oneBlockNum - LAYERNORM_ONE_NUMBER) / oneBlockNum * oneBlockNum;
    const uint32_t tmpBufSize = stackBufferSize / SIZEOF_FLOAT;
    uint32_t oneTmpSize = (tmpBufSize - varianceTmpTensorSize) / numberOfTmpBuf;
    oneTmpSize = oneTmpSize / rLengthWithPadding * rLengthWithPadding;
    if (oneTmpSize > inputXSize) oneTmpSize = inputXSize;
    if (oneTmpSize == LAYERNORM_ZERO_NUMBER) return;

    const uint32_t firstTmpStartPos = varianceTmpTensorSize;
    const uint32_t secondTmpStartPos = firstTmpStartPos + oneTmpSize;
    const uint32_t loopRound = inputXSize / oneTmpSize;
    const uint32_t inputRoundSize = oneTmpSize;
    const uint32_t inputTailSize = inputXSize % oneTmpSize;
    const uint32_t inputTailPos = inputXSize - inputTailSize;
    const uint32_t meanVarRoundSize = inputRoundSize / rLengthWithPadding;
    const uint32_t meanVarTailSize = inputTailSize / rLengthWithPadding;
    const uint32_t meanVarTailPos = meanVarSize - meanVarTailSize;
    const uint32_t arCurLength = inputRoundSize;
    const uint32_t aCurLength = meanVarRoundSize;
    const float rValueBack = 1.0f / static_cast<float>(rLength);

    // Fill the opaque tiling data (21 uint32_t fields)
    layernormTilingData[0] = aLength;
    layernormTilingData[1] = rLength;
    layernormTilingData[2] = inputXSize;
    layernormTilingData[3] = meanVarSize;
    layernormTilingData[4] = numberOfTmpBuf;
    layernormTilingData[5] = varianceTmpTensorPos;
    layernormTilingData[6] = varianceTmpTensorSize;
    layernormTilingData[7] = tmpBufSize;
    layernormTilingData[8] = oneTmpSize;
    layernormTilingData[9] = firstTmpStartPos;
    layernormTilingData[10] = secondTmpStartPos;
    layernormTilingData[11] = loopRound;
    layernormTilingData[12] = inputRoundSize;
    layernormTilingData[13] = inputTailSize;
    layernormTilingData[14] = inputTailPos;
    layernormTilingData[15] = meanVarRoundSize;
    layernormTilingData[16] = meanVarTailSize;
    layernormTilingData[17] = meanVarTailPos;
    layernormTilingData[18] = arCurLength;
    layernormTilingData[19] = aCurLength;
    memcpy(layernormTilingData + 20, &rValueBack, sizeof(float));
}
} // unnamed namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> shapeDims = x_shape->GetOriginShape().GetDims();
    uint32_t totalLength = 1;
    for (auto d : shapeDims) totalLength *= static_cast<uint32_t>(d);
    // Flatten non-normalized dims into aLength, and normalized dims into rLength
    // For normalized_shape = (features, dim1, dim2), the last rank-1 dimensions are normalized.
    // We assume the last (rank-1) dimensions are the normalized dimensions.
    // Get the rank from input.
    size_t rank = shapeDims.size();
    // The last N-1 dimensions are normalized? Actually nn.LayerNorm normalizes over the last N dimensions.
    // For simplicity, we use the last dimension as the normalized dimension? But the example has 4D: (batch, features, dim1, dim2) and normalized_shape=(features, dim1, dim2) => normalize over last 3 dims.
    // We need to handle arbitrary normalized_shape. We'll set aLength = product of first (rank - normalized_rank) dims, rLength = product of last normalized_rank dims.
    // We don't have normalized_rank explicitly, but we can get it from the shape of gamma: it should be exactly the normalized_shape.
    const gert::Shape *gamma_shape = context->GetInputShape(1);
    uint32_t normalized_elems = 1;
    for (uint32_t i = 0; i < gamma_shape->GetOriginShape().GetDimNum(); ++i)
        normalized_elems *= gamma_shape->GetOriginShape().GetDim(i);
    uint32_t aLength = totalLength / normalized_elems;
    uint32_t rLength = normalized_elems;

    float epsilon = 1e-5f;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs && attrs->GetAttrNum() > 0) {
        epsilon = *attrs->GetAttrPointer<float>(0);
    }

    // UB budget heuristic: 192KB per core, reserve for local tensors
    constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    uint32_t typeSize = sizeof(float);
    uint32_t stackBufferSize = UB_BUDGET_BYTES; // simplified

    LayerNormCustomTilingData tiling;
    tiling.set_aLength(aLength);
    tiling.set_rLength(rLength);
    tiling.set_epsilon(epsilon);
    // Compute rLengthWithPadding and fill layernormTilingData
    uint32_t layernormTilingData[21] = {0};
    std::vector<int64_t> dims = {static_cast<int64_t>(aLength), static_cast<int64_t>(rLength)};
    GetLayerNormNDTilingInfo(dims, stackBufferSize, typeSize, false, true, layernormTilingData);
    tiling.set_rLengthWithPadding(layernormTilingData[1] ? layernormTilingData[1] : (rLength + 7)/8*8); // fallback align
    for (int i = 0; i < 21; ++i) {
        tiling.SetField(layernormTilingData[i], i);
    }
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    uint32_t blockDim = (aLength > 8) ? 8 : aLength;
    context->SetBlockDim(blockDim);

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    // optional mean/rstd shapes: same as batch dim
    if (context->GetOutputShapeCount() > 1) {
        gert::Shape *mean_shape = context->GetOutputShape(1);
        // mean shape is (batch,) where batch = product of non-normalized dims
        uint32_t total = 1;
        const auto &dims = x_shape->GetOriginShape().GetDims();
        for (size_t i = 0; i < dims.size() - (context->GetInputShape(1)->GetOriginShape().GetDimNum()); ++i)
            total *= dims[i];
        mean_shape->SetDimNum(1);
        mean_shape->SetDim(0, total);
    }
    if (context->GetOutputShapeCount() > 2) {
        gert::Shape *rstd_shape = context->GetOutputShape(2);
        *rstd_shape = *context->GetOutputShape(1);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    if (context->GetOutputDataTypeCount() > 1) context->SetOutputDataType(1, x_dtype);
    if (context->GetOutputDataTypeCount() > 2) context->SetOutputDataType(2, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''
#include "kernel_operator.h"
#include "lib/layernorm_v2.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                GM_ADDR mean, GM_ADDR rstd, VecTiling tilingData)
    {
        pipe_ = new TPipe();
        epsilon_ = tilingData.epsilon;
        aLength_ = tilingData.aLength;
        rLength_ = tilingData.rLength;
        rLengthWithPadding_ = tilingData.rLengthWithPadding;
        arLength_ = aLength_ * rLengthWithPadding_;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), arLength_);
        gammaGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(gamma), rLengthWithPadding_);
        betaGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(beta), rLengthWithPadding_);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), arLength_);
        if (mean != nullptr)
            meanGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(mean), aLength_);
        if (rstd != nullptr)
            rstdGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(rstd), aLength_);

        pipe_->InitBuffer(inQueueX_, BUFFER_NUM, sizeof(float) * arLength_);
        pipe_->InitBuffer(inQueueGamma_, 1, sizeof(float) * rLengthWithPadding_);
        pipe_->InitBuffer(inQueueBeta_, 1, sizeof(float) * rLengthWithPadding_);
        pipe_->InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float) * arLength_);
        pipe_->InitBuffer(outQueueMean_, 1, sizeof(float) * aLength_);
        pipe_->InitBuffer(outQueueRstd_, 1, sizeof(float) * aLength_);

        layernormTiling_ = tilingData.layernormTilingData;
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        LocalTensor<float> gammaLocal = inQueueGamma_.AllocTensor<float>();
        LocalTensor<float> betaLocal = inQueueBeta_.AllocTensor<float>();

        DataCopy(xLocal, xGm_, arLength_);
        DataCopy(gammaLocal, gammaGm_, rLengthWithPadding_);
        DataCopy(betaLocal, betaGm_, rLengthWithPadding_);

        inQueueX_.EnQue(xLocal);
        inQueueGamma_.EnQue(gammaLocal);
        inQueueBeta_.EnQue(betaLocal);
    }

    __aicore__ inline void Compute()
    {
        LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        LocalTensor<float> gammaLocal = inQueueGamma_.DeQue<float>();
        LocalTensor<float> betaLocal = inQueueBeta_.DeQue<float>();

        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        LocalTensor<float> meanLocal = outQueueMean_.AllocTensor<float>();
        LocalTensor<float> rstdLocal = outQueueRstd_.AllocTensor<float>();

        LayerNormPara para = {aLength_, rLength_, rLengthWithPadding_};
        static constexpr LayerNormConfig config = GetConfig();
        LayerNorm<float, float, false, config>(yLocal, meanLocal, rstdLocal, xLocal,
                                                gammaLocal, betaLocal, epsilon_, para, layernormTiling_);

        outQueueY_.EnQue(yLocal);
        outQueueMean_.EnQue(meanLocal);
        outQueueRstd_.EnQue(rstdLocal);

        inQueueX_.FreeTensor(xLocal);
        inQueueGamma_.FreeTensor(gammaLocal);
        inQueueBeta_.FreeTensor(betaLocal);
    }

    __aicore__ inline void CopyOut()
    {
        LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        LocalTensor<float> meanLocal = outQueueMean_.DeQue<float>();
        LocalTensor<float> rstdLocal = outQueueRstd_.DeQue<float>();

        DataCopy(yGm_, yLocal, arLength_);
        if (meanGm_.GetSize() > 0)
            DataCopy(meanGm_, meanLocal, aLength_);
        if (rstdGm_.GetSize() > 0)
            DataCopy(rstdGm_, rstdLocal, aLength_);

        outQueueY_.FreeTensor(yLocal);
        outQueueMean_.FreeTensor(meanLocal);
        outQueueRstd_.FreeTensor(rstdLocal);
    }

private:
    TPipe* pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECIN, 1> inQueueGamma_;
    TQue<TPosition::VECIN, 1> inQueueBeta_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECOUT, 1> outQueueMean_;
    TQue<TPosition::VECOUT, 1> outQueueRstd_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> gammaGm_;
    GlobalTensor<float> betaGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> meanGm_;
    GlobalTensor<float> rstdGm_;

    uint32_t aLength_;
    uint32_t rLength_;
    uint32_t rLengthWithPadding_;
    uint32_t arLength_;
    float epsilon_;
    LayerNormSeparateTiling layernormTiling_;
};

extern "C" __global__ __aicore__ void layer_norm_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
    GM_ADDR mean, GM_ADDR rstd, GM_ADDR workspace, GM_ADDR tiling)
{
    VecTiling tilingData;
    CopyTiling(&tilingData, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y, mean, rstd, tilingData);
    op.Process();
}
'''

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        self.weight = torch.nn.Parameter(torch.ones(normalized_shape))
        self.bias = torch.nn.Parameter(torch.zeros(normalized_shape))

    def forward(self, x):
        # The custom op expects gamma and beta as separate inputs
        gamma = self.weight.float()
        beta = self.bias.float()
        result, _, _ = getattr(custom_ops_lib, _FN)(x, gamma, beta)
        return result
