#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: 192 KiB per core.
// Per-tile requirement: x(float) + mask(int8) + y(float) = 3 * tileLen * 4 ≈ 12 * tileLen bytes.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaskedCumsumCustomTilingData tiling;
    const auto &xShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batchSize = static_cast<uint32_t>(xShape.GetDim(0));
    uint32_t rowSize = static_cast<uint32_t>(xShape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t axis = *attrs->GetAttrPointer<int64_t>(0);
    if (axis != 1) {
        return ge::GRAPH_FAILED;
    }

    // Compute tile length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 12; // approx 3 * sizeof(float) per element
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowSize < tileLen) {
        tileLen = AlignDown(rowSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowSize;
    }

    // Block dimension: one per row, capped to 24
    uint32_t blockDim = batchSize;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_rowSize(rowSize);
    tiling.set_axis(static_cast<uint32_t>(axis));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:
    explicit MaskedCumsumCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t batchSize, uint32_t rowSize,
                                uint32_t tileLength)
    {
        batchSize_ = batchSize;
        rowSize_ = rowSize;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        maskGm_.SetGlobalBuffer((__gm__ int8_t*)mask);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(int8_t));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || rowSize_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= batchSize_) return;

        uint32_t rowBase = blockIdx * rowSize_;
        float carry = 0.0f;

        uint32_t remaining = rowSize_;
        uint32_t tileOff = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            uint32_t curOff = rowBase + tileOff;

            CopyIn(curOff, curLen);
            Compute(curLen, carry);
            CopyOut(curOff, curLen);

            tileOff += curLen;
            remaining -= curLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOff, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<int8_t> maskLocal = inQueueMask_.AllocTensor<int8_t>();

        AscendC::DataCopy(xLocal, xGm_[gmOff], len);
        AscendC::DataCopy(maskLocal, maskGm_[gmOff], len);

        inQueueX_.EnQue(xLocal);
        inQueueMask_.EnQue(maskLocal);
    }

    __aicore__ inline void Compute(uint32_t len, float &carry)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<int8_t> maskLocal = inQueueMask_.DeQue<int8_t>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Scalar loop: y[i] = (mask[i] != 0) ? carry + x[i] : 0
        for (uint32_t i = 0; i < len; i++) {
            int8_t m = maskLocal(i);
            float xv = xLocal(i);
            if (m != 0) {
                carry += xv;
            } else {
                carry = 0.0f;
            }
            yLocal(i) = carry;
        }

        inQueueMask_.FreeTensor(maskLocal);
        inQueueX_.FreeTensor(xLocal);
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOff, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOff], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int8_t> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t rowSize_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask,
                                                           GM_ADDR y,
                                                           GM_ADDR workspace,
                                                           GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.batchSize, tiling_data.rowSize,
            tiling_data.tileLength);
    op.Process();
}
