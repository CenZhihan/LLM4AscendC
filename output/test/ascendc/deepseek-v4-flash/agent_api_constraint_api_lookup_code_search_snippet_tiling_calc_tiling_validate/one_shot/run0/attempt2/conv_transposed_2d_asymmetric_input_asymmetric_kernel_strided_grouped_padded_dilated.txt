static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // Compute output shape for transposed convolution
    const gert::StorageShape *x = context->GetInputShape(0);
    const gert::StorageShape *w = context->GetInputShape(1);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *stride = attrs->GetListInt(0)->GetData();
    const int64_t *padding = attrs->GetListInt(1)->GetData();
    const int64_t *output_padding = attrs->GetListInt(2)->GetData();
    const int64_t *dilation = attrs->GetListInt(4)->GetData();
    int64_t groups = *attrs->GetInt(3);

    int64_t n = x->GetDim(0);
    int64_t h_in = x->GetDim(2);
    int64_t w_in = x->GetDim(3);
    int64_t c_out_per_group = w->GetDim(1);
    int64_t k_h = w->GetDim(2);
    int64_t k_w = w->GetDim(3);
    int64_t c_out = c_out_per_group * groups;

    int64_t h_out = (h_in - 1) * stride[0] - 2 * padding[0] + dilation[0] * (k_h - 1) + output_padding[0] + 1;
    int64_t w_out = (w_in - 1) * stride[1] - 2 * padding[1] + dilation[1] * (k_w - 1) + output_padding[1] + 1;

    gert::Shape *y = context->GetOutputShape(0);
    y->SetDimNum(4);
    y->SetDim(0, n);
    y->SetDim(1, c_out);
    y->SetDim(2, h_out);
    y->SetDim(3, w_out);
    return GRAPH_SUCCESS;
}

#include "kernel_operator.h"

using namespace AscendC;

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t n, uint32_t c_in, uint32_t h_in, uint32_t w_in,
                                uint32_t c_out_per_group, uint32_t k_h, uint32_t k_w,
                                uint32_t groups,
                                int32_t stride_h, int32_t stride_w,
                                int32_t pad_h, int32_t pad_w,
                                int32_t output_pad_h, int32_t output_pad_w,
                                int32_t dilation_h, int32_t dilation_w)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        n_ = n; c_in_ = c_in; h_in_ = h_in; w_in_ = w_in;
        c_out_per_group_ = c_out_per_group;
        k_h_ = k_h; k_w_ = k_w;
        groups_ = groups;
        stride_h_ = stride_h; stride_w_ = stride_w;
        pad_h_ = pad_h; pad_w_ = pad_w;
        output_pad_h_ = output_pad_h; output_pad_w_ = output_pad_w;
        dilation_h_ = dilation_h; dilation_w_ = dilation_w;

        // Compute output dimensions
        h_out_ = (h_in_ - 1) * stride_h_ - 2 * pad_h_ + dilation_h_ * (k_h_ - 1) + output_pad_h_ + 1;
        w_out_ = (w_in_ - 1) * stride_w_ - 2 * pad_w_ + dilation_w_ * (k_w_ - 1) + output_pad_w_ + 1;
        c_out_ = c_out_per_group_ * groups_;
    }

    __aicore__ inline void Process()
    {
        // Simple naive implementation: single-core, single-block loop
        // For each batch, output channel, output height, output width,
        // compute the dot product over input channel and kernel positions.
        // This is a placeholder; a production kernel would use tiling and vectorization.
        for (uint32_t bn = 0; bn < n_; bn++) {
            for (uint32_t oc = 0; oc < c_out_; oc++) {
                uint32_t g = oc / c_out_per_group_;
                uint32_t ic_start = g * (c_in_ / groups_);
                uint32_t ic_end = (g + 1) * (c_in_ / groups_);
                for (int32_t oh = 0; oh < h_out_; oh++) {
                    for (int32_t ow = 0; ow < w_out_; ow++) {
                        float sum = 0.0f;
                        for (uint32_t ic = ic_start; ic < ic_end; ic++) {
                            for (uint32_t kh = 0; kh < k_h_; kh++) {
                                for (uint32_t kw = 0; kw < k_w_; kw++) {
                                    // Determine input position
                                    int32_t ih = (oh + pad_h_ - dilation_h_ * kh) / stride_h_;
                                    int32_t iw = (ow + pad_w_ - dilation_w_ * kw) / stride_w_;
                                    // Check bounds
                                    if ((oh + pad_h_ - dilation_h_ * kh) % stride_h_ == 0 &&
                                        (ow + pad_w_ - dilation_w_ * kw) % stride_w_ == 0 &&
                                        ih >= 0 && ih < h_in_ && iw >= 0 && iw < w_in_) {
                                        uint32_t w_offset = oc * (c_in_ / groups_) * k_h_ * k_w_
                                                          + (ic - ic_start) * k_h_ * k_w_
                                                          + kh * k_w_ + kw;
                                        uint32_t x_offset = bn * c_in_ * h_in_ * w_in_
                                                          + ic * h_in_ * w_in_
                                                          + ih * w_in_ + iw;
                                        sum += xGm_.GetValue(x_offset) * wGm_.GetValue(w_offset);
                                    }
                                }
                            }
                        }
                        uint32_t y_offset = bn * c_out_ * h_out_ * w_out_
                                          + oc * h_out_ * w_out_
                                          + oh * w_out_ + ow;
                        yGm_.SetValue(y_offset, sum);
                    }
                }
            }
        }
    }

private:
    GlobalTensor<float> xGm_, wGm_, yGm_;
    uint32_t n_, c_in_, h_in_, w_in_;
    uint32_t c_out_per_group_, k_h_, k_w_, groups_;
    int32_t stride_h_, stride_w_, pad_h_, pad_w_;
    int32_t output_pad_h_, output_pad_w_, dilation_h_, dilation_w_;
    int32_t h_out_, w_out_;
    uint32_t c_out_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom op;
    op.Init(x, weight, y,
            td.n, td.c_in, td.h_in, td.w_in,
            td.c_out_per_group, td.k_h, td.k_w,
            td.groups,
            td.stride_h, td.stride_w,
            td.pad_h, td.pad_w,
            td.output_pad_h, td.output_pad_w,
            td.dilation_h, td.dilation_w);
    op.Process();
}
