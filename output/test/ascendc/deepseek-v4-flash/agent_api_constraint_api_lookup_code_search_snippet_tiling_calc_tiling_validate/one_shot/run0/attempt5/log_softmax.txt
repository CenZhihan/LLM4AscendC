host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // per core
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t Min(uint32_t a, uint32_t b) { return a < b ? a : b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData *tiling = context->GetTilingData<LogSoftmaxCustomTilingData>();
    const gert::Shape &xShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalLength = static_cast<uint32_t>(xShape.GetShapeSize());
    const uint32_t ndim = static_cast<uint32_t>(xShape.GetDimNum());
    uint32_t dim = 1;
    if (ndim >= 2) {
        const gert::RuntimeAttrs *attrs = context->GetAttrs();
        if (attrs != nullptr) {
            dim = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(0)[0]);
        }
    }

    uint32_t rowLength = xShape.GetDim(0);
    uint32_t columnLength = (ndim >= 2) ? xShape.GetDim(1) : 1;

    // UB budget for three passes: each pass processes tileRowNum * columnLength floats
    // We also need per‑row max/sum buffers (small)
    // conservative: assume 5 * tileLength floats total
    uint32_t maxTile = UB_BUDGET_BYTES / 20; // 5 * sizeof(float) per element
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileRowNum = maxTile / columnLength;
    if (tileRowNum == 0) tileRowNum = 1;
    if (tileRowNum > rowLength) tileRowNum = rowLength;

    uint32_t tileLength = tileRowNum * columnLength;
    uint32_t numTiles = CeilDiv(rowLength, tileRowNum);

    uint32_t blockDim = Min(numTiles, 24);
    if (blockDim < 8) blockDim = 8;
    context->SetBlockDim(blockDim);

    uint32_t rowsPerBlock = CeilDiv(rowLength, blockDim);
    uint32_t usedNumBlocks = CeilDiv(rowLength, rowsPerBlock);
    uint32_t coreRowNum = rowsPerBlock;
    uint32_t singleLoopCoreRowNum = coreRowNum;
    uint32_t singleCoreLoopCount = 1;
    uint32_t leftRow = 0;
    if (rowsPerBlock > tileRowNum) {
        singleLoopCoreRowNum = tileRowNum;
        singleCoreLoopCount = rowsPerBlock / tileRowNum;
        leftRow = rowsPerBlock % tileRowNum;
    }

    uint32_t tailCoreSingleLoopCoreRowNum = singleLoopCoreRowNum;
    uint32_t tailCoreSingleCoreLoopCount = singleCoreLoopCount;
    uint32_t tailCoreSingleCoreLoopTail = leftRow;
    if (usedNumBlocks < blockDim) {
        uint32_t tailRows = rowLength % rowsPerBlock;
        if (tailRows > 0) {
            tailCoreSingleLoopCoreRowNum = tailRows > tileRowNum ? tileRowNum : tailRows;
            tailCoreSingleCoreLoopCount = (tailRows > tileRowNum) ? (tailRows / tileRowNum) : 1;
            tailCoreSingleCoreLoopTail = tailRows % tileRowNum;
        }
    }

    uint32_t sharedTmpBufferSize = 61440;

    tiling->set_totalLength(totalLength);
    tiling->set_rowLength(rowLength);
    tiling->set_columnLength(columnLength);
    tiling->set_dim(dim);
    tiling->set_coreRowNum(coreRowNum);
    tiling->set_tileRowNum(tileRowNum);
    tiling->set_loopCount(singleCoreLoopCount);
    tiling->set_tileLength(tileLength);
    tiling->set_usedNumBlocks(usedNumBlocks);
    tiling->set_sharedTmpBufferSize(sharedTmpBufferSize);
    tiling->set_singleLoopCoreRowNum(singleLoopCoreRowNum);
    tiling->set_singleCoreLoopCount(singleCoreLoopCount);
    tiling->set_leftRow(leftRow);
    tiling->set_tailCoreSingleLoopCoreRowNum(tailCoreSingleLoopCoreRowNum);
    tiling->set_tailCoreSingleCoreLoopCount(tailCoreSingleCoreLoopCount);
    tiling->set_tailCoreSingleCoreLoopTail(tailCoreSingleCoreLoopTail);

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t CHUNK = 1024;  // process columns in chunks for UB economy

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t columnLength, uint32_t tileRowNum,
                                uint32_t tileLength, uint32_t coreRowNum,
                                uint32_t blockIdx, uint32_t blockNum)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        colLen_ = columnLength;
        tileRowNum_ = tileRowNum;
        tileLen_ = tileLength;
        coreRowNum_ = coreRowNum;
        blockIdx_ = blockIdx;
        blockNum_ = blockNum;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // UB buffers: one tile of input, output, and a working chunk
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpChunk_, CHUNK * sizeof(float));  // TBuf, not TQue
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        // Determine rows assigned to this block
        uint32_t startRow = blockIdx_ * coreRowNum_;
        uint32_t rowsThisBlock = coreRowNum_;
        if (startRow >= rowLen_) return;
        if (startRow + rowsThisBlock > rowLen_)
            rowsThisBlock = rowLen_ - startRow;

        // Process rows in tiles of size tileRowNum
        uint32_t processed = 0;
        while (processed < rowsThisBlock) {
            uint32_t currentTileRows = tileRowNum_;
            if (processed + currentTileRows > rowsThisBlock)
                currentTileRows = rowsThisBlock - processed;

            CopyIn(startRow + processed, currentTileRows);
            Compute(currentTileRows);
            CopyOut(startRow + processed, currentTileRows);
            processed += currentTileRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowBase, uint32_t rows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        uint32_t offset = rowBase * colLen_;
        AscendC::DataCopy(xLocal, xGm_[offset], rows * colLen_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rows)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> chunk = tmpChunk_.Get<float>();

        // For each row in this tile
        for (uint32_t r = 0; r < rows; r++) {
            uint32_t rowOffset = r * colLen_;

            // ---- Step 1: find max over the row ----
            float rowMax = -3.40282e+38f;  // -FLT_MAX
            for (uint32_t c = 0; c < colLen_; c += CHUNK) {
                uint32_t curChunk = CHUNK;
                if (c + curChunk > colLen_) curChunk = colLen_ - c;
                AscendC::DataCopy(chunk, xLocal[rowOffset + c], curChunk);
                for (uint32_t i = 0; i < curChunk; i++) {
                    float val = chunk.GetValue(i);
                    if (val > rowMax) rowMax = val;
                }
            }

            // ---- Step 2: compute exp(x - max) and sum ----
            float rowSum = 0.0f;
            for (uint32_t c = 0; c < colLen_; c += CHUNK) {
                uint32_t curChunk = CHUNK;
                if (c + curChunk > colLen_) curChunk = colLen_ - c;
                AscendC::DataCopy(chunk, xLocal[rowOffset + c], curChunk);
                // subtract rowMax
                AscendC::Subs(chunk, chunk, rowMax, curChunk);  // x - rowMax
                AscendC::Exp(chunk, chunk, curChunk);
                for (uint32_t i = 0; i < curChunk; i++) {
                    rowSum += chunk.GetValue(i);
                }
            }

            // ---- Step 3: compute log(sum) ----
            float logSum = 0.0f;
            if (rowSum > 0.0f) {
                AscendC::LocalTensor<float> scalar = tmpChunk_.Get<float>();
                scalar.SetValue(0, rowSum);
                AscendC::Log(scalar, scalar, 1);
                logSum = scalar.GetValue(0);
            }

            // ---- Step 4: final output y = (x - rowMax) - logSum ----
            for (uint32_t c = 0; c < colLen_; c += CHUNK) {
                uint32_t curChunk = CHUNK;
                if (c + curChunk > colLen_) curChunk = colLen_ - c;
                AscendC::DataCopy(chunk, xLocal[rowOffset + c], curChunk);
                AscendC::Subs(chunk, chunk, rowMax, curChunk);   // x - rowMax
                AscendC::Subs(chunk, chunk, logSum, curChunk);   // (x - rowMax) - logSum
                AscendC::DataCopy(yLocal[rowOffset + c], chunk, curChunk);
            }
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowBase, uint32_t rows)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint32_t offset = rowBase * colLen_;
        AscendC::DataCopy(yGm_[offset], yLocal, rows * colLen_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpChunk_;   // reused for various temporary vectors

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t colLen_ {0};
    uint32_t tileRowNum_ {0};
    uint32_t tileLen_ {0};
    uint32_t coreRowNum_ {0};
    uint32_t blockIdx_ {0};
    uint32_t blockNum_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();

    // Compute per-block row assignment from tiling fields
    // We reuse coreRowNum as rows per block (already computed in tiling)
    uint32_t coreRowNum = tiling_data.coreRowNum;
    // blockIdx determines which rows it handles: sequential assignment
    uint32_t rowsPerBlock = coreRowNum;
    uint32_t startRow = blockIdx * rowsPerBlock;
    if (startRow >= tiling_data.rowLength) return;
    uint32_t rowsThisBlock = rowsPerBlock;
    if (startRow + rowsThisBlock > tiling_data.rowLength)
        rowsThisBlock = tiling_data.rowLength - startRow;

    KernelLogSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.rowLength,
            tiling_data.columnLength,
            tiling_data.tileRowNum,
            tiling_data.tileLength,
            rowsThisBlock,
            blockIdx,
            blockNum);
    op.Process();
}
