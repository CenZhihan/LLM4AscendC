kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y,
                                uint32_t batchSize, uint32_t dim, float margin)
    {
        batchSize_ = batchSize;
        dim_ = dim;
        margin_ = margin;
        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, batchSize * dim);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, batchSize * dim);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, batchSize * dim);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, dim_ * sizeof(float));
        pipe_.InitBuffer(inQueueP_, BUFFER_NUM, dim_ * sizeof(float));
        pipe_.InitBuffer(inQueueN_, BUFFER_NUM, dim_ * sizeof(float));

        pipe_.InitBuffer(tmpDiffPos_, 1, dim_ * sizeof(float));
        pipe_.InitBuffer(tmpDiffNeg_, 1, dim_ * sizeof(float));
        pipe_.InitBuffer(tmpWorkspace_, 1, dim_ * sizeof(float));

        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        float totalLoss = 0.0f;
        for (uint32_t i = 0; i < batchSize_; ++i) {
            uint32_t offset = i * dim_;

            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            AscendC::DataCopy(aLocal, anchorGm_[offset], dim_);
            inQueueA_.EnQue(aLocal);

            AscendC::LocalTensor<float> pLocal = inQueueP_.AllocTensor<float>();
            AscendC::DataCopy(pLocal, positiveGm_[offset], dim_);
            inQueueP_.EnQue(pLocal);

            AscendC::LocalTensor<float> nLocal = inQueueN_.AllocTensor<float>();
            AscendC::DataCopy(nLocal, negativeGm_[offset], dim_);
            inQueueN_.EnQue(nLocal);

            AscendC::LocalTensor<float> a = inQueueA_.DeQue<float>();
            AscendC::LocalTensor<float> p = inQueueP_.DeQue<float>();
            AscendC::LocalTensor<float> n = inQueueN_.DeQue<float>();

            AscendC::LocalTensor<float> diffPos = tmpDiffPos_.AllocTensor<float>();
            AscendC::Sub(diffPos, a, p, dim_);
            AscendC::Mul(diffPos, diffPos, diffPos, dim_);
            AscendC::LocalTensor<float> workspace = tmpWorkspace_.AllocTensor<float>();
            AscendC::LocalTensor<float> posDistScalar = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceSum(posDistScalar, diffPos, workspace, dim_);
            float posDist = posDistScalar(0);
            scalarQueue_.FreeTensor(posDistScalar);

            AscendC::LocalTensor<float> diffNeg = tmpDiffNeg_.AllocTensor<float>();
            AscendC::Sub(diffNeg, a, n, dim_);
            AscendC::Mul(diffNeg, diffNeg, diffNeg, dim_);
            AscendC::LocalTensor<float> negDistScalar = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceSum(negDistScalar, diffNeg, workspace, dim_);
            float negDist = negDistScalar(0);
            scalarQueue_.FreeTensor(negDistScalar);

            tmpWorkspace_.FreeTensor(workspace);
            tmpDiffPos_.FreeTensor(diffPos);
            tmpDiffNeg_.FreeTensor(diffNeg);

            inQueueA_.FreeTensor(a);
            inQueueP_.FreeTensor(p);
            inQueueN_.FreeTensor(n);

            float sampleLoss = posDist - negDist + margin_;
            if (sampleLoss < 0.0f) sampleLoss = 0.0f;
            totalLoss += sampleLoss;
        }

        AscendC::LocalTensor<float> yLocal = scalarQueue_.AllocTensor<float>();
        yLocal(0) = totalLoss;
        AscendC::DataCopy(yGm_, yLocal, 1);
        scalarQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueP_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueN_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDiffPos_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDiffNeg_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpWorkspace_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> scalarQueue_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t dim_ {0};
    float margin_ {0.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(
    GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y,
            tiling_data.batchSize, tiling_data.dim, tiling_data.margin);
    op.Process();
}
