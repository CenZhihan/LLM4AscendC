__aicore__ inline void Process()
{
    if (batchSize_ == 0 || rowSize_ == 0 || tileLen_ == 0) return;

    uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
    uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

    for (uint32_t r = blockIdx; r < batchSize_; r += blockNum) {
        uint32_t rowBase = r * rowSize_;
        float carry = 0.0f;
        uint32_t remaining = rowSize_;
        uint32_t tileOff = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            uint32_t curOff = rowBase + tileOff;
            CopyIn(curOff, curLen);
            Compute(curLen, carry);
            CopyOut(curOff, curLen);
            tileOff += curLen;
            remaining -= curLen;
        }
    }
}
