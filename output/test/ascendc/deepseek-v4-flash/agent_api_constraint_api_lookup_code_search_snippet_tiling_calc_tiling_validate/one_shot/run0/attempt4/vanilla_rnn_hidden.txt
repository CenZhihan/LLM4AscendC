static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    VanillaRnnHiddenCustomTilingData tiling;

    auto getDim = [&](uint32_t idx, uint32_t dim) -> uint32_t {
        return static_cast<uint32_t>(context->GetInputShape(idx)->GetOriginShape().GetDim(dim));
    };
    uint32_t T = getDim(0, 0);
    uint32_t B = getDim(0, 1);
    uint32_t I = getDim(0, 2);
    uint32_t H = getDim(1, 1);
    uint32_t K = getDim(2, 1);
    uint32_t O = getDim(4, 0);

    tiling.set_T(T);
    tiling.set_B(B);
    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_K(K);

    // Use 16 blocks for parallelism
    uint32_t blockDim = 16;
    context->SetBlockDim(blockDim);
    tiling.set_blockDim(blockDim);

    // Prepare matmul tiling for first matmul: M=B, N=H, K=K (A cat, B w_i2h transposed)
    // we use TCubeTiling::SaveToBuffer for each
    TCubeTiling cubeTiling0;
    cubeTiling0.SetDim(M_DIM, B);
    cubeTiling0.SetDim(N_DIM, H);
    cubeTiling0.SetDim(K_DIM, K);
    cubeTiling0.SetAts(A_MATRIX, false, false, false); // A is row-major, no transpose
    cubeTiling0.SetBts(B_MATRIX, true, false, false); // B is column-major (transposed)
    cubeTiling0.SetBias(false); // we add bias separately
    cubeTiling0.SaveToBuffer(tiling.matmul0TilingData, tiling.matmul0TilingSize);
    tiling.set_matmul0TilingSize(tiling.matmul0TilingSize);

    // Second matmul: M=B, N=O, K=H (A h, B w_h2o transposed)
    TCubeTiling cubeTiling1;
    cubeTiling1.SetDim(M_DIM, B);
    cubeTiling1.SetDim(N_DIM, O);
    cubeTiling1.SetDim(K_DIM, H);
    cubeTiling1.SetAts(A_MATRIX, false, false, false);
    cubeTiling1.SetBts(B_MATRIX, true, false, false);
    cubeTiling1.SetBias(false);
    cubeTiling1.SaveToBuffer(tiling.matmul1TilingData, tiling.matmul1TilingSize);
    tiling.set_matmul1TilingSize(tiling.matmul1TilingSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"
#include "adv_api/matmul/matmul_intf.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                GM_ADDR workspace, GM_ADDR tilingGm) {
        GET_TILING_DATA(td, tilingGm);
        T_ = td.T; B_ = td.B; I_ = td.I;
        H_ = td.H; O_ = td.O; K_ = td.K;
        blockDim_ = td.blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T_ * B_ * I_);
        hGm_.SetGlobalBuffer((__gm__ float*)h0, B_ * H_);
        wI2HGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H_ * K_);
        bI2HGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);
        wH2OGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O_ * H_);
        bH2OGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, T_ * B_ * O_);

        // Load matmul tilings from host
        memcpy(&cubeTiling0_, td.matmul0TilingData, td.matmul0TilingSize);
        memcpy(&cubeTiling1_, td.matmul1TilingData, td.matmul1TilingSize);

        // Init pipe and buffers
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, B_ * I_ * sizeof(float));
        pipe_.InitBuffer(inQueueH_, BUFFER_NUM, B_ * H_ * sizeof(float));
        pipe_.InitBuffer(inQueueWI2H_, 1, H_ * K_ * sizeof(float));
        pipe_.InitBuffer(inQueueBI2H_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(inQueueWH2O_, 1, O_ * H_ * sizeof(float));
        pipe_.InitBuffer(inQueueBH2O_, 1, O_ * sizeof(float));
        pipe_.InitBuffer(tmpCat_, 1, B_ * K_ * sizeof(float));
        pipe_.InitBuffer(tmpPre_, 1, B_ * H_ * sizeof(float));
        pipe_.InitBuffer(tmpTanh_, 1, B_ * H_ * sizeof(float));
        pipe_.InitBuffer(tmpY_, 1, B_ * O_ * sizeof(float));
        pipe_.InitBuffer(outQueueH_, BUFFER_NUM, B_ * H_ * sizeof(float));

        // Load weights and biases (w_i2h is HxK, w_h2o is OxH)
        {
            LocalTensor<float> wI2HLocal = inQueueWI2H_.AllocTensor<float>();
            DataCopy(wI2HLocal, wI2HGm_, H_ * K_);
            inQueueWI2H_.EnQue(wI2HLocal);
        }
        {
            LocalTensor<float> biasI2HLocal = inQueueBI2H_.AllocTensor<float>();
            DataCopy(biasI2HLocal, bI2HGm_, H_);
            inQueueBI2H_.EnQue(biasI2HLocal);
        }
        {
            LocalTensor<float> wH2OLocal = inQueueWH2O_.AllocTensor<float>();
            DataCopy(wH2OLocal, wH2OGm_, O_ * H_);
            inQueueWH2O_.EnQue(wH2OLocal);
        }
        {
            LocalTensor<float> biasH2OLocal = inQueueBH2O_.AllocTensor<float>();
            DataCopy(biasH2OLocal, bH2OGm_, O_);
            inQueueBH2O_.EnQue(biasH2OLocal);
        }
        // Load initial h
        {
            LocalTensor<float> hLocal = inQueueH_.AllocTensor<float>();
            DataCopy(hLocal, hGm_, B_ * H_);
            inQueueH_.EnQue(hLocal);
        }
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = blockDim_;
        uint32_t tStart = blockIdx * (T_ / blockNum);
        uint32_t tEnd = (blockIdx + 1) * (T_ / blockNum);
        if (blockIdx == blockNum - 1) tEnd = T_;

        for (uint32_t t = tStart; t < tEnd; ++t) {
            // Load x[t]
            uint32_t xOff = t * B_ * I_;
            {
                LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                DataCopy(xLocal, xGm_[xOff], B_ * I_);
                inQueueX_.EnQue(xLocal);
            }

            LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            LocalTensor<float> hLocal = inQueueH_.DeQue<float>();

            // Concatenate x[t] and h -> catBuf (BxK)
            LocalTensor<float> catBuf = tmpCat_.AllocTensor<float>();
            DataCopy(catBuf, xLocal, B_ * I_);
            DataCopy(catBuf[B_ * I_], hLocal, B_ * H_);

            // First matmul: catBuf (BxK) * w_i2h^T (KxH) -> pre (BxH)
            LocalTensor<float> wI2HLocal = inQueueWI2H_.DeQue<float>();
            LocalTensor<float> biasI2HLocal = inQueueBI2H_.DeQue<float>();
            LocalTensor<float> preBuf = tmpPre_.AllocTensor<float>();

            // Use Matmul API: Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE, MM_CFG, MM_CB, MATMUL_POLICY>
            // We use float for all types and default config
            using Matmul0 = Matmul<float, float, float, float, CubeFormat::NZ, CubeFormat::ND, false>;
            Matmul0 matmul0;
            matmul0.SetA(catBuf, 1, B_ * K_);
            matmul0.SetB(wI2HLocal, 1, H_ * K_);
            matmul0.SetC(preBuf, 1, B_ * H_);
            matmul0.SetBias(biasI2HLocal);
            matmul0.SetTiling(cubeTiling0_);
            matmul0.IterateAll();

            inQueueWI2H_.EnQue(wI2HLocal); // reuse for next iteration
            inQueueBI2H_.EnQue(biasI2HLocal);

            // Tanh
            LocalTensor<float> tanhBuf = tmpTanh_.AllocTensor<float>();
            Tanh(tanhBuf, preBuf, B_ * H_);

            // Second matmul: tanhBuf (BxH) * w_h2o^T (HxO) -> y (BxO)
            LocalTensor<float> wH2OLocal = inQueueWH2O_.DeQue<float>();
            LocalTensor<float> biasH2OLocal = inQueueBH2O_.DeQue<float>();
            LocalTensor<float> yBuf = tmpY_.AllocTensor<float>();

            using Matmul1 = Matmul<float, float, float, float, CubeFormat::NZ, CubeFormat::ND, false>;
            Matmul1 matmul1;
            matmul1.SetA(tanhBuf, 1, B_ * H_);
            matmul1.SetB(wH2OLocal, 1, O_ * H_);
            matmul1.SetC(yBuf, 1, B_ * O_);
            matmul1.SetBias(biasH2OLocal);
            matmul1.SetTiling(cubeTiling1_);
            matmul1.IterateAll();

            // Store y and update hidden
            uint32_t yOff = t * B_ * O_;
            DataCopy(yGm_[yOff], yBuf, B_ * O_);

            // Compute next hidden: tanhBuf (we need to keep it for next timestep)
            // Enqueue tanhBuf as next h
            LocalTensor<float> nextH = outQueueH_.AllocTensor<float>();
            DataCopy(nextH, tanhBuf, B_ * H_);
            inQueueH_.EnQue(nextH);

            // Free local tensors
            inQueueX_.FreeTensor(xLocal);
            inQueueH_.FreeTensor(hLocal); // current h consumed
            tmpCat_.FreeTensor(catBuf);
            tmpPre_.FreeTensor(preBuf);
            tmpTanh_.FreeTensor(tanhBuf);
            tmpY_.FreeTensor(yBuf);
            inQueueWH2O_.EnQue(wH2OLocal);
            inQueueBH2O_.EnQue(biasH2OLocal);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueH_;
    TQue<TPosition::VECIN, 1> inQueueWI2H_;
    TQue<TPosition::VECIN, 1> inQueueBI2H_;
    TQue<TPosition::VECIN, 1> inQueueWH2O_;
    TQue<TPosition::VECIN, 1> inQueueBH2O_;
    TQue<TPosition::VECCALC, 1> tmpCat_;
    TQue<TPosition::VECCALC, 1> tmpPre_;
    TQue<TPosition::VECCALC, 1> tmpTanh_;
    TQue<TPosition::VECCALC, 1> tmpY_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueH_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> hGm_;
    GlobalTensor<float> wI2HGm_;
    GlobalTensor<float> bI2HGm_;
    GlobalTensor<float> wH2OGm_;
    GlobalTensor<float> bH2OGm_;
    GlobalTensor<float> yGm_;

    TCubeTiling cubeTiling0_;
    TCubeTiling cubeTiling1_;

    uint32_t T_, B_, I_, H_, O_, K_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(
    GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tilingGm)
{
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y, workspace, tilingGm);
    op.Process();
}
