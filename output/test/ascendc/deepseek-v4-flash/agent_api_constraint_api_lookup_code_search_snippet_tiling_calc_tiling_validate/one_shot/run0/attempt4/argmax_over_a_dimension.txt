#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    int32_t ndim = inputShape->GetDimNum();
    int64_t dim = context->GetAttrs()->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += ndim;
    uint32_t totalRows = 1;
    uint32_t rowLen = 1;
    for (int32_t i = 0; i < ndim; ++i) {
        uint32_t s = inputShape->GetDim(i);
        if (i == dim) {
            rowLen = s;
        } else {
            totalRows *= s;
        }
    }

    uint32_t bytesPerRow = rowLen * sizeof(float) * 2 + sizeof(int64_t);
    uint32_t maxRowsInTile = UB_BUDGET_BYTES / bytesPerRow;
    if (maxRowsInTile < 1) maxRowsInTile = 1;
    if (maxRowsInTile > 256) maxRowsInTile = 256;

    uint32_t tileRows = (totalRows < maxRowsInTile) ? totalRows : maxRowsInTile;

    uint32_t blockDim = CeilDiv(totalRows, tileRows);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);
    tiling.set_totalRows(totalRows);
    tiling.set_rowLen(rowLen);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *inputShape = context->GetInputShape(0);
    int32_t ndim = inputShape->GetDimNum();
    int64_t dim = context->GetAttrs()->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += ndim;

    gert::Shape *yShape = context->GetOutputShape(0);
    yShape->SetDimNum(ndim - 1);
    int32_t outIdx = 0;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i != dim) {
            yShape->SetDim(outIdx++, inputShape->GetDim(i));
        }
    }
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLen,
                                uint32_t tileRows) {
        totalRows_ = totalRows;
        rowLen_ = rowLen;
        tileRows_ = tileRows;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y);

        uint32_t ipB = tileRows_ * rowLen_ * sizeof(float);
        uint32_t opB = tileRows_ * sizeof(int64_t);
        // temporary buffer for ReduceMax: 2 floats for output + some scratch
        uint32_t tmpB = 64 * sizeof(float); // conservative

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, ipB);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, opB);
        pipe_.InitBuffer(tmpQueue_, 1, tmpB);
    }

    __aicore__ inline void Process() {
        if (totalRows_ == 0 || rowLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t strideRows = blockNum * tileRows_;

        uint32_t base = blockIdx * tileRows_;
        if (base >= totalRows_) return;

        uint32_t curRows = totalRows_ - base;
        if (curRows > tileRows_) curRows = tileRows_;

        CopyIn(base, curRows);
        while (true) {
            uint32_t nextBase = base + strideRows;
            uint32_t nextRows = 0;
            if (nextBase < totalRows_) {
                nextRows = totalRows_ - nextBase;
                if (nextRows > tileRows_) nextRows = tileRows_;
                CopyIn(nextBase, nextRows);
            }

            Compute(curRows);
            CopyOut(base, curRows);

            if (nextRows == 0) break;
            base = nextBase;
            curRows = nextRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowOffset, uint32_t rows) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowOffset * rowLen_], rows * rowLen_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rows) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.AllocTensor<int64_t>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

        for (uint32_t r = 0; r < rows; ++r) {
            AscendC::LocalTensor<float> row = xLocal[r * rowLen_];
            // Use ReduceMax with index; tmp provides two float slots for result
            AscendC::ReduceMax<float>(tmp, row, tmp + 2, rowLen_, true);
            AscendC::PipeBarrier<PIPE_V>();
            float idxFloat = tmp.GetValue(1);
            int64_t idx = static_cast<int64_t>(idxFloat);
            yLocal.SetValue(r, idx);
        }

        tmpQueue_.FreeTensor(tmp);
        outQueueY_.EnQue<int64_t>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowOffset, uint32_t rows) {
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.DeQue<int64_t>();
        AscendC::DataCopy(yGm_[rowOffset], yLocal, rows);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;
    uint32_t totalRows_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileRows_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.rowLen, tiling_data.tileRows);
    op.Process();
}
