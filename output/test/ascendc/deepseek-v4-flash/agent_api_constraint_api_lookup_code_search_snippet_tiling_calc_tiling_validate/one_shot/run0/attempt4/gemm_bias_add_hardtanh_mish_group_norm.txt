kernel_src="""

#include "kernel_operator.h"
#include "math.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t BUFFER_NUM_Y = 1;  // Output does not need double buffering
constexpr float HARD_TANH_MIN = -1.0f;
constexpr float HARD_TANH_MAX = 1.0f;
constexpr float GN_EPSILON = 1e-5f;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
                                GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t numGroups, uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                float groupSizeReciprocal)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        tileM_ = tileM;  // =1
        tileN_ = tileN;  // =outFeatures
        tileK_ = tileK;  // =inFeatures

        groupSizeReciprocal_ = groupSizeReciprocal;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * inFeatures_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures_ * inFeatures_);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures_);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures_);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * outFeatures_);

        // Allocate queues: use single buffer for output to reduce UB pressure
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileK_ * sizeof(float));         // row of input (double buffered)
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM_Y, tileN_ * sizeof(float));      // output row (single buffer)
        pipe_.InitBuffer(tmpQueue1_, 1, tileN_ * sizeof(float));                 // for mish intermediate
        pipe_.InitBuffer(tmpQueue2_, 1, tileN_ * sizeof(float));                 // for mish intermediate
        pipe_.InitBuffer(tmpQueueMean_, 1, numGroups_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueVar_, 1, numGroups_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || outFeatures_ == 0 || inFeatures_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        for (uint32_t row = blockIdx; row < batchSize_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        // Load input row
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        DataCopy(xLocal, xGm_[row * inFeatures_], inFeatures_);

        // Output tile (single buffer)
        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // ----- Matmul (manual dot product) -----
        for (uint32_t j = 0; j < outFeatures_; j++) {
            float sum = 0.0f;
            for (uint32_t k = 0; k < inFeatures_; k++) {
                sum += xLocal(k) * weightGm_.GetValue(j * inFeatures_ + k);
            }
            sum += linBiasGm_.GetValue(j);
            yLocal(j) = sum;
        }

        // ----- Add bias -----
        for (uint32_t j = 0; j < outFeatures_; j++) {
            yLocal(j) += biasGm_.GetValue(j);
        }

        // ----- Hardtanh -----
        Mins(yLocal, yLocal, HARD_TANH_MAX, outFeatures_);
        Maxs(yLocal, yLocal, HARD_TANH_MIN, outFeatures_);

        // ----- Mish: x * tanh(softplus(x)) -----
        LocalTensor<float> tmp1 = tmpQueue1_.AllocTensor<float>();
        LocalTensor<float> tmp2 = tmpQueue2_.AllocTensor<float>();
        Exp(tmp1, yLocal, outFeatures_);               // tmp1 = exp(x)
        Adds(tmp1, tmp1, 1.0f, outFeatures_);           // tmp1 = 1 + exp(x)
        Log(tmp2, tmp1, outFeatures_);                 // tmp2 = log(1+exp(x)) = softplus
        Tanh(tmp2, tmp2, outFeatures_);                // tmp2 = tanh(softplus)
        Mul(yLocal, yLocal, tmp2, outFeatures_);       // y = x * tanh(softplus)
        tmpQueue1_.FreeTensor(tmp1);
        tmpQueue2_.FreeTensor(tmp2);

        // ----- Group Norm -----
        LocalTensor<float> groupMean = tmpQueueMean_.AllocTensor<float>();
        LocalTensor<float> groupVar = tmpQueueVar_.AllocTensor<float>();
        uint32_t groupSize = outFeatures_ / numGroups_;

        for (uint32_t g = 0; g < numGroups_; g++) {
            float mean = 0.0f;
            float var = 0.0f;
            uint32_t start = g * groupSize;
            for (uint32_t c = 0; c < groupSize; c++) {
                float val = yLocal(start + c);
                mean += val;
                var += val * val;
            }
            mean = mean * groupSizeReciprocal_;
            var = var * groupSizeReciprocal_ - mean * mean;
            groupMean(g) = mean;
            groupVar(g) = var;
        }

        // Apply gamma and beta
        for (uint32_t g = 0; g < numGroups_; g++) {
            float invStd = 1.0f / sqrt(groupVar(g) + GN_EPSILON);
            float mean = groupMean(g);
            uint32_t start = g * groupSize;
            for (uint32_t c = 0; c < groupSize; c++) {
                float normalized = (yLocal(start + c) - mean) * invStd;
                yLocal(start + c) = normalized * gnGammaGm_.GetValue(start + c) + gnBetaGm_.GetValue(start + c);
            }
        }

        tmpQueueMean_.FreeTensor(groupMean);
        tmpQueueVar_.FreeTensor(groupVar);

        // Store output
        DataCopy(yGm_[row * outFeatures_], yLocal, outFeatures_);

        // Free local tensors
        inQueueX_.FreeTensor(xLocal);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM_Y> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMean_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueVar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_;
    uint32_t inFeatures_;
    uint32_t outFeatures_;
    uint32_t numGroups_;
    uint32_t tileM_;
    uint32_t tileN_;
    uint32_t tileK_;
    float groupSizeReciprocal_;
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.batchSize, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.numGroups, tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.groupSizeReciprocal);
    op.Process();
}
