static uint32_t CalcOutSize(uint32_t inputSize, uint32_t kernelSize,
                            uint32_t stride, uint32_t padding,
                            uint32_t dilation, bool ceilMode)
{
    uint32_t dilatedSize = kernelSize + (kernelSize - 1) * (dilation - 1);
    if (ceilMode) {
        return static_cast<uint32_t>(std::ceil(static_cast<float>(inputSize + 2 * padding - dilatedSize) / stride)) + 1;
    } else {
        return static_cast<uint32_t>(std::floor(static_cast<float>(inputSize + 2 * padding - dilatedSize) / stride)) + 1;
    }
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *inputShape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(inputShape->GetDim(0));
    uint32_t C = static_cast<uint32_t>(inputShape->GetDim(1));
    uint32_t D = static_cast<uint32_t>(inputShape->GetDim(2));
    uint32_t H = static_cast<uint32_t>(inputShape->GetDim(3));
    uint32_t W = static_cast<uint32_t>(inputShape->GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const std::vector<int64_t>* kernelSizePtr = attrs->GetAttrPointer<std::vector<int64_t>>(0);
    const std::vector<int64_t>* stridePtr = attrs->GetAttrPointer<std::vector<int64_t>>(1);
    const std::vector<int64_t>* paddingPtr = attrs->GetAttrPointer<std::vector<int64_t>>(2);
    std::vector<int64_t> dilation = {1,1,1};
    bool ceilMode = false;
    if (attrs->GetAttrNum() > 3) {
        const std::vector<int64_t>* dilationPtr = attrs->GetAttrPointer<std::vector<int64_t>>(3);
        dilation = *dilationPtr;
    }
    if (attrs->GetAttrNum() > 4) {
        ceilMode = *(attrs->GetAttrPointer<bool>(4));
    }

    auto calc = [](uint32_t in, uint32_t k, uint32_t s, uint32_t p, uint32_t d, bool ceil) -> uint32_t {
        uint32_t dilated = k + (k-1)*(d-1);
        if (ceil) return static_cast<uint32_t>(std::ceil((in + 2*p - dilated)/s) + 1);
        else return static_cast<uint32_t>(std::floor((in + 2*p - dilated)/s) + 1);
    };

    uint32_t kD = static_cast<uint32_t>((*kernelSizePtr)[0]);
    uint32_t kH = static_cast<uint32_t>((*kernelSizePtr)[1]);
    uint32_t kW = static_cast<uint32_t>((*kernelSizePtr)[2]);
    uint32_t sD = static_cast<uint32_t>((*stridePtr)[0]);
    uint32_t sH = static_cast<uint32_t>((*stridePtr)[1]);
    uint32_t sW = static_cast<uint32_t>((*stridePtr)[2]);
    uint32_t pD = static_cast<uint32_t>((*paddingPtr)[0]);
    uint32_t pH = static_cast<uint32_t>((*paddingPtr)[1]);
    uint32_t pW = static_cast<uint32_t>((*paddingPtr)[2]);
    uint32_t dD = static_cast<uint32_t>(dilation[0]);
    uint32_t dH = static_cast<uint32_t>(dilation[1]);
    uint32_t dW = static_cast<uint32_t>(dilation[2]);

    uint32_t oD = calc(D, kD, sD, pD, dD, ceilMode);
    uint32_t oH = calc(H, kH, sH, pH, dH, ceilMode);
    uint32_t oW = calc(W, kW, sW, pW, dW, ceilMode);

    gert::Shape *outShape = context->GetOutputShape(0);
    outShape->SetDimNum(5);
    outShape->SetDim(0, static_cast<int64_t>(N));
    outShape->SetDim(1, static_cast<int64_t>(C));
    outShape->SetDim(2, static_cast<int64_t>(oD));
    outShape->SetDim(3, static_cast<int64_t>(oH));
    outShape->SetDim(4, static_cast<int64_t>(oW));
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include <cfloat>

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t sD, uint32_t sH, uint32_t sW,
                                uint32_t pD, uint32_t pH, uint32_t pW,
                                uint32_t dD, uint32_t dH, uint32_t dW,
                                uint32_t oD, uint32_t oH, uint32_t oW,
                                uint32_t ceilMode,
                                uint32_t totalOutputElems,
                                uint32_t tileElems)
    {
        N_ = n;
        C_ = c;
        D_ = d;
        H_ = h;
        W_ = w;
        kD_ = kD; kH_ = kH; kW_ = kW;
        sD_ = sD; sH_ = sH; sW_ = sW;
        pD_ = pD; pH_ = pH; pW_ = pW;
        dD_ = dD; dH_ = dH; dW_ = dW;
        oD_ = oD; oH_ = oH; oW_ = oW;
        ceilMode_ = ceilMode;

        totalOutputElems_ = totalOutputElems;
        tileElems_ = tileElems;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t startIdx = blockIdx * tileElems_;
        uint32_t endIdx = startIdx + tileElems_;
        if (startIdx >= totalOutputElems_) return;
        if (endIdx > totalOutputElems_) endIdx = totalOutputElems_;

        for (uint32_t idx = startIdx; idx < endIdx; ++idx) {
            uint32_t ow = idx % oW_;
            uint32_t oh = (idx / oW_) % oH_;
            uint32_t od = (idx / (oW_ * oH_)) % oD_;
            uint32_t c_ = (idx / (oW_ * oH_ * oD_)) % C_;
            uint32_t n_ = idx / (oW_ * oH_ * oD_ * C_);

            int32_t startD = static_cast<int32_t>(od) * sD_ - pD_;
            int32_t startH = static_cast<int32_t>(oh) * sH_ - pH_;
            int32_t startW = static_cast<int32_t>(ow) * sW_ - pW_;

            float maxVal = -FLT_MAX;
            for (uint32_t kd = 0; kd < kD_; ++kd) {
                int32_t id = startD + kd * dD_;
                if (id < 0 || id >= static_cast<int32_t>(D_)) continue;
                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    int32_t ih = startH + kh * dH_;
                    if (ih < 0 || ih >= static_cast<int32_t>(H_)) continue;
                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        int32_t iw = startW + kw * dW_;
                        if (iw < 0 || iw >= static_cast<int32_t>(W_)) continue;
                        uint32_t offset = n_ * C_ * D_ * H_ * W_ +
                                          c_ * D_ * H_ * W_ +
                                          static_cast<uint32_t>(id) * H_ * W_ +
                                          static_cast<uint32_t>(ih) * W_ +
                                          static_cast<uint32_t>(iw);
                        float val = xGm_.GetValue(offset);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }
            yGm_.SetValue(idx, maxVal);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_, yGm_;
    uint32_t N_, C_, D_, H_, W_;
    uint32_t kD_, kH_, kW_;
    uint32_t sD_, sH_, sW_;
    uint32_t pD_, pH_, pW_;
    uint32_t dD_, dH_, dW_;
    uint32_t oD_, oH_, oW_;
    uint32_t ceilMode_;
    uint32_t totalOutputElems_;
    uint32_t tileElems_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y,
                                                            GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);

    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tilingData.n, tilingData.c, tilingData.d, tilingData.h, tilingData.w,
            tilingData.kD, tilingData.kH, tilingData.kW,
            tilingData.sD, tilingData.sH, tilingData.sW,
            tilingData.pD, tilingData.pH, tilingData.pW,
            tilingData.dD, tilingData.dH, tilingData.dW,
            tilingData.oD, tilingData.oH, tilingData.oW,
            tilingData.ceilMode,
            tilingData.totalOutputElems, tilingData.tileElems);
    op.Process();
}
