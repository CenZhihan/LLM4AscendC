    return (a + b - 1u) / b;

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MeanReductionOverADimensionCustomTilingData tiling;

    // Get shape directly (no GetOriginShape)
    const gert::Shape *xShape = context->GetInputShape(0);
    const uint32_t ndim = static_cast<uint32_t>(xShape->GetDimNum());

    // Get attribute 'dim' and normalize to non-negative
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    int ridx = static_cast<int>(*dimAttr);
    if (ridx < 0) ridx += static_cast<int>(ndim);
    tiling.set_dim(static_cast<int32_t>(ridx));

    // Copy input shape into tiling data (up to 4 dimensions)
    uint32_t shape[4] = {1, 1, 1, 1};
    for (uint32_t i = 0; i < ndim && i < 4; ++i) {
        shape[i] = static_cast<uint32_t>(xShape->GetDim(i));
    }
    tiling.set_shape0(shape[0]);
    tiling.set_shape1(shape[1]);
    tiling.set_shape2(shape[2]);
    tiling.set_shape3(shape[3]);

    uint32_t reduceSize = shape[ridx];
    uint32_t totalIn = static_cast<uint32_t>(xShape->GetShapeSize());
    uint32_t outLength = totalIn / reduceSize;

    tiling.set_outLength(outLength);

    if (tileLen > outLength) tileLen = outLength;
    tiling.set_tileLength(tileLen);

    if (blockDim < 8) blockDim = 8;

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *xShape = context->GetInputShape(0);
    const uint32_t ndim = static_cast<uint32_t>(xShape->GetDimNum());

    // Get attribute 'dim' and normalize
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    int ridx = static_cast<int>(*dimAttr);
    if (ridx < 0) ridx += static_cast<int>(ndim);

    // Build output shape by removing the reduced dimension
    uint32_t newNdim = ndim - 1;
    int64_t newDims[4];
    uint32_t idx = 0;
    for (uint32_t i = 0; i < ndim; ++i) {
        if (static_cast<int>(i) != ridx) {
            newDims[idx++] = xShape->GetDim(i);
        }
    }

    gert::Shape *outShape = context->GetOutputShape(0);
    outShape->SetDimNum(newNdim);
    for (uint32_t i = 0; i < newNdim; ++i) {
        outShape->SetDim(i, newDims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t VECTOR_LEN = 2048;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outLength, uint32_t reduceSize,
                                uint32_t tileLength, int32_t dim,
                                uint32_t shape0, uint32_t shape1,
                                uint32_t shape2, uint32_t shape3)
    {
        outLen_ = outLength;
        reduceSize_ = reduceSize;
        tileLen_ = tileLength;
        dim_ = dim;

        // store shape
        shapes_[0] = shape0;
        shapes_[1] = shape1;
        shapes_[2] = shape2;
        shapes_[3] = shape3;
        // infer ndim (trailing 1's)
        ndim_ = 4;
        while (ndim_ > 1 && shapes_[ndim_-1] == 1) ndim_--;

        // Normalize dim
        if (dim_ < 0) dim_ += ndim_;

        // Compute input strides (row-major)
        inputStrides_[ndim_-1] = 1;
        for (int32_t i = ndim_-2; i >= 0; --i) {
            inputStrides_[i] = inputStrides_[i+1] * shapes_[i+1];
        }

        // Compute output shape and output strides
        uint32_t outShapeIdx = 0;
        for (int32_t i = 0; i < ndim_; ++i) {
            if (i != dim_) {
                outShape_[outShapeIdx] = shapes_[i];
                outShapeIdx++;
            }
        }
        outNdim_ = ndim_ - 1;
        if (outNdim_ == 0) outNdim_ = 1; // keep at least 1 for scalar output
        outStrides_[outNdim_-1] = 1;
        for (int32_t i = outNdim_-2; i >= 0; --i) {
            outStrides_[i] = outStrides_[i+1] * outShape_[i+1];
        }

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, reduceSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outLen_ == 0 || reduceSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= outLen_) return;

        // Process tiles
        uint32_t curTile = outLen_ - base;
        if (curTile > tileLen_) curTile = tileLen_;

        for (uint32_t i = 0; i < curTile; ++i) {
            uint32_t outIdx = base + i;
            uint32_t inputBase = getInputBase(outIdx);

            float sum = 0.0f;
            uint32_t remReduce = reduceSize_;
            uint32_t offset = 0;
            while (remReduce > 0) {
                uint32_t chunk = remReduce > VECTOR_LEN ? VECTOR_LEN : remReduce;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[inputBase + offset*inputStrides_[dim_]], chunk);
                // Scalar accumulation (simplified)
                for (uint32_t j = 0; j < chunk; ++j) {
                    sum += xLocal(j);
                }
                inQueueX_.FreeTensor(xLocal);
                offset += chunk;
                remReduce -= chunk;
            }
            yGm_[outIdx] = sum / static_cast<float>(reduceSize_);
        }
    }

private:
    __aicore__ inline uint32_t getInputBase(uint32_t outLinear) {
        // Convert output linear index to output coordinates, then to input coordinates
        // by inserting 0 at the reduced dimension, then compute input linear index.
        uint32_t coords[4] = {0};
        uint32_t tmp = outLinear;
        uint32_t outDims = outNdim_;
        for (int32_t i = 0; i < static_cast<int32_t>(outDims); ++i) {
            uint32_t dimSize = outShape_[i];
            if (dimSize > 0) {
                coords[i] = tmp / outStrides_[i];
                tmp = tmp % outStrides_[i];
            }
        }

        // Now we have output coordinates; create input coordinates by inserting 0 at dim_
        uint32_t inCoords[4] = {0};
        uint32_t outIdx = 0;
        for (int32_t i = 0; i < ndim_; ++i) {
            if (i == dim_) {
                inCoords[i] = 0; // reduce dimension
            } else {
                inCoords[i] = coords[outIdx];
                outIdx++;
            }
        }

        // Compute linear index from input coordinates
        uint32_t linear = 0;
        for (int32_t i = 0; i < ndim_; ++i) {
            linear += inCoords[i] * inputStrides_[i];
        }
        return linear;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outLen_ {0};
    uint32_t reduceSize_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ {0};
    int32_t ndim_ {4};
    int32_t outNdim_ {3};
    uint32_t shapes_[4] {1,1,1,1};
    uint32_t outShape_[4] {1,1,1,1};
    uint32_t inputStrides_[4] {0};
    uint32_t outStrides_[4] {0};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.outLength, tiling_data.reduceSize,
            tiling_data.tileLength, tiling_data.dim,
            tiling_data.shape0, tiling_data.shape1,
            tiling_data.shape2, tiling_data.shape3);
    op.Process();
}
