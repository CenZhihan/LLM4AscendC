host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per core
static constexpr uint32_t ALIGN_ELEMS = 2048;            // DMA alignment

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)   { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const auto &inShape = context->GetInputShape(0)->GetOriginShape();
    int64_t ndim = inShape.GetDimNum();
    // Assume last dimension is the reduction axis (dim=1)
    uint32_t rowLength = static_cast<uint32_t>(inShape.GetDim(ndim - 1));
    uint32_t totalElements = static_cast<uint32_t>(inShape.GetShapeSize());
    uint32_t totalRows = totalElements / rowLength;

    // Compute tileLength from UB budget: we need buffers for x, exp, and one temporary per tile.
    // vectors: inX (1), outY (1), tmpExp (1) -> 3 * tileLen * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    // also tileLength should not exceed rowLength
    uint32_t tileLength = (maxTile > rowLength) ? rowLength : maxTile;
    // Ensure tileLength is multiple of (some alignment) but at least 1
    if (tileLength == 0) tileLength = rowLength;

    // Number of tiles per row
    uint32_t tilePerRow = CeilDiv(rowLength, tileLength);

    // Number of blocks: choose as totalRows (rows) but capped by hardware (e.g. 40)
    // We'll set blockDim = (totalRows > 40) ? 40 : totalRows; but keep at least 1
    uint32_t blockDim = (totalRows > 40) ? 40 : totalRows;
    if (blockDim < 1) blockDim = 1;

    // Rows per block (floor)
    uint32_t rowsPerBlock = totalRows / blockDim;
    if (rowsPerBlock == 0) rowsPerBlock = 1;

    // Workspace size: each block stores per-row max and sum (2 floats per row)
    // plus some margin for alignment
    uint32_t workspaceBytes = rowsPerBlock * 2 * sizeof(float);
    // align to 32 bytes (DMA alignment)
    workspaceBytes = (workspaceBytes + 31) & ~31;

    context->SetBlockDim(blockDim);

    tiling.set_totalElements(totalElements);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_tilePerRow(tilePerRow);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = workspaceBytes;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace,
                                uint32_t totalElements, uint32_t rowLength,
                                uint32_t tileLength, uint32_t blockDim,
                                uint32_t rowsPerBlock, uint32_t tilePerRow)
    {
        totalElements_ = totalElements;
        rowLength_ = rowLength;
        tileLength_ = tileLength;
        blockDim_ = blockDim;
        rowsPerBlock_ = rowsPerBlock;
        tilePerRow_ = tilePerRow;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElements_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElements_);
        // workspace is used to store per-row max and sum for the rows assigned to this block
        // size = rowsPerBlock_ * 2 floats
        wsGm_.SetGlobalBuffer((__gm__ float*)workspace, rowsPerBlock_ * 2);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLength_ * sizeof(float));
        // small buffer for scalar results (1 float)
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalElements_ == 0 || rowLength_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = blockDim_;

        // Determine the range of rows this block processes
        uint32_t rowStart = blockIdx * rowsPerBlock_;
        uint32_t rowEnd = (blockIdx == blockNum - 1) ? (totalElements_ / rowLength_) : (rowStart + rowsPerBlock_);
        if (rowStart >= (totalElements_ / rowLength_)) return; // no rows for this block

        uint32_t localRowCount = rowEnd - rowStart;

        // -------- PASS 1: compute per-row max and sum (online softmax) ---------
        // For simplicity, we compute rowMax and rowSum for each row and store in workspace
        // Workspace layout: row k (local index) -> max at ws[k*2], sum at ws[k*2+1]
        // We'll use local scalar variables and write only after each row.
        // To avoid many global writes, we accumulate in local and write at end?
        // We'll write after each row (max and sum are scalars, write is fine).
        for (uint32_t localRow = 0; localRow < localRowCount; ++localRow)
        {
            uint32_t globalRow = rowStart + localRow;
            float rowMax = -FLT_MAX;
            float rowSum = 0.0f;

            // iterate over tiles of this row
            for (uint32_t tileIdx = 0; tileIdx < tilePerRow_; ++tileIdx)
            {
                uint32_t base = globalRow * rowLength_ + tileIdx * tileLength_;
                uint32_t curLen = (tileIdx == tilePerRow_ - 1) ? (rowLength_ - tileIdx * tileLength_) : tileLength_;

                // Load tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base], curLen);
                inQueueX_.EnQue(xLocal);
                xLocal = inQueueX_.DeQue<float>();

                // Compute tile max
                AscendC::LocalTensor<float> tileMaxLocal = scalarQueue_.AllocTensor<float>();
                AscendC::VecReduceMax(tileMaxLocal, xLocal, curLen); // result is a 1-element local tensor
                float tileMax = *((float*)tileMaxLocal.GetAddr());   // dereference scalar
                scalarQueue_.FreeTensor(tileMaxLocal);

                // Compute exp(x - newMax) and sum for this tile
                AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
                float newMax = fmaxf(rowMax, tileMax);
                // Subtract: x - newMax
                AscendC::Sub(expLocal, xLocal, newMax, curLen);
                // Exp
                AscendC::Exp(expLocal, expLocal, curLen);
                // Update rowSum: rowSum = rowSum * exp(rowMax - newMax) + sum_exp_tile
                // compute scalar factor
                float factor = expf(rowMax - newMax);
                rowSum *= factor;
                // sum of expLocal for this tile
                AscendC::LocalTensor<float> sumLocal = scalarQueue_.AllocTensor<float>();
                AscendC::VecReduceSum(sumLocal, expLocal, curLen);
                float tileSum = *((float*)sumLocal.GetAddr());
                scalarQueue_.FreeTensor(sumLocal);
                rowSum += tileSum;

                rowMax = newMax;

                tmpQueue_.FreeTensor(expLocal);
                inQueueX_.FreeTensor(xLocal);

                // Early break if we have processed all elements (curLen < tileLen_ for last tile? no, row may not fill)
                // Not needed as tilePerRow_ rounds up.
            }

            // Store rowMax and rowSum in workspace
            wsGm_[localRow * 2] = rowMax;
            wsGm_[localRow * 2 + 1] = rowSum;
        }

        // -------- PASS 2: normalize each row ---------
        for (uint32_t localRow = 0; localRow < localRowCount; ++localRow)
        {
            uint32_t globalRow = rowStart + localRow;
            float rowMax = wsGm_[localRow * 2];
            float rowSum = wsGm_[localRow * 2 + 1];
            float invSum = 1.0f / rowSum; // use multiplication instead of division

            for (uint32_t tileIdx = 0; tileIdx < tilePerRow_; ++tileIdx)
            {
                uint32_t base = globalRow * rowLength_ + tileIdx * tileLength_;
                uint32_t curLen = (tileIdx == tilePerRow_ - 1) ? (rowLength_ - tileIdx * tileLength_) : tileLength_;

                // Load x
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base], curLen);
                inQueueX_.EnQue(xLocal);
                xLocal = inQueueX_.DeQue<float>();

                // Compute y = exp(x - rowMax) / rowSum
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpExp = tmpQueue_.AllocTensor<float>();
                // sub
                AscendC::Sub(tmpExp, xLocal, rowMax, curLen);
                // exp
                AscendC::Exp(tmpExp, tmpExp, curLen);
                // multiply by invSum
                AscendC::Mul(yLocal, tmpExp, invSum, curLen);

                // Output
                outQueueY_.EnQue(yLocal);
                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[base], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
                tmpQueue_.FreeTensor(tmpExp);
                inQueueX_.FreeTensor(xLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;   // for scalar results

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wsGm_;

    uint32_t totalElements_ {0};
    uint32_t rowLength_ {0};
    uint32_t tileLength_ {0};
    uint32_t blockDim_ {1};
    uint32_t rowsPerBlock_ {1};
    uint32_t tilePerRow_ {1};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, workspace,
            tiling_data.totalElements, tiling_data.rowLength,
            tiling_data.tileLength, tiling_data.blockDim,
            tiling_data.rowsPerBlock, tiling_data.tilePerRow);
    op.Process();
}
