```cpp
    // Compute tileLen based on UB budget for two buffers (in and out)
    uint32_t maxTile = UB_BUDGET_BYTES / (2u * sizeof(float)); // 2 floats per element for in and out
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (inner < tileLen) {
        tileLen = inner;
    }

    tiling.set_tileLen(tileLen);
    tiling.set_totalLength(totalLength);

    // Block dim: try to assign each tile of a row to a block if possible
    // For simplicity, set blockDim to outer (one block per row) capped by max blocks
    uint32_t blockDim = outer;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;
    context->SetBlockDim(blockDim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
```

```cpp
kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outer, uint32_t inner,
                                uint32_t tileLen, int32_t dim)
    {
        totalLen_ = totalLength;
        outer_ = outer;
        inner_ = inner;
        tileLen_ = tileLen;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float)); // for exp temporary?
        pipe_.InitBuffer(scratchQueue_, 1, 2 * sizeof(float)); // for max and sum scalars
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || outer_ == 0 || inner_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Distribute rows to blocks (each row is independent)
        uint32_t rowsPerBlock = CeilDiv(outer_, blockNum);
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > outer_) rowEnd = outer_;

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        // Get max for this row
        float maxVal = -FLT_MAX;
        for (uint32_t off = 0; off < inner_; off += tileLen_) {
            uint32_t curLen = (off + tileLen_ <= inner_) ? tileLen_ : inner_ - off;
            CopyInRow(row, off, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            // compute local max
            for (uint32_t i = 0; i < curLen; ++i) {
                if (xLocal(i) > maxVal) maxVal = xLocal(i);
            }
            inQueueX_.FreeTensor(xLocal);
        }

        // Compute sum of exp(x - max) for this row
        float sumVal = 0.0f;
        for (uint32_t off = 0; off < inner_; off += tileLen_) {
            uint32_t curLen = (off + tileLen_ <= inner_) ? tileLen_ : inner_ - off;
            CopyInRow(row, off, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(tmp, xLocal, maxVal, curLen);
            AscendC::Exp(tmp, tmp, curLen);
            for (uint32_t i = 0; i < curLen; ++i) {
                sumVal += tmp(i);
            }
            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xLocal);
        }

        // Compute final log_softmax for this row
        float logSum = logf(sumVal);
        for (uint32_t off = 0; off < inner_; off += tileLen_) {
            uint32_t curLen = (off + tileLen_ <= inner_) ? tileLen_ : inner_ - off;
            CopyInRow(row, off, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // y = x - max - logSum
            AscendC::Sub(yLocal, xLocal, maxVal, curLen);
            AscendC::Sub(yLocal, yLocal, logSum, curLen);
            CopyOutRow(row, off, curLen, yLocal);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

    __aicore__ inline void CopyInRow(uint32_t row, uint32_t colOff, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        uint32_t gmOffset = row * inner_ + colOff;
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t row, uint32_t colOff, uint32_t len,
                                      AscendC::LocalTensor<float> yLocal)
    {
        uint32_t gmOffset = row * inner_ + colOff;
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scratchQueue_; // not used in this version

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outer_ {0};
    uint32_t inner_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.outer, tiling_data.inner,
            tiling_data.tileLen, tiling_data.dim);
    op.Process();
}
"""
```
