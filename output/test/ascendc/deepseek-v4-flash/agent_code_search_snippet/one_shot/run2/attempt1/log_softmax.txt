host_operator_src='''
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;
    uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = (attrs->GetAttrPointer<int64_t>(0)) ? *attrs->GetAttrPointer<int64_t>(0) : 1;

    // Compute row length = size along dim
    // Input shape is known; we assume ND format with dimensions up to 8
    const gert::Shape *shape = context->GetInputShape(0);
    uint32_t rowLen = 1;
    int32_t rank = shape->GetDimNum();
    // Normalize dim to positive index
    int32_t normalizedDim = dim;
    if (normalizedDim < 0) normalizedDim += rank;
    rowLen = (normalizedDim >= 0 && normalizedDim < rank) ? shape->GetDim(normalizedDim) : 1;

    // UB budget for three temporary tensors + input/output (2 tiles)
    // Each tile: input (1), output (1), maxBuf (1), expBuf (1), sumBuf (1) -> 5*tileLen floats
    // Actually kernel uses 2 input tiles, 2 output tiles, and some temporaries, let's budget 6 tiles
    uint32_t tileFloats = UB_BUDGET_BYTES / (6 * sizeof(float));
    tileFloats = AlignDown(tileFloats, ALIGN_ELEMS);
    if (tileFloats < ALIGN_ELEMS) tileFloats = ALIGN_ELEMS;
    if (tileFloats > HARD_CAP_ELEMS) tileFloats = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = tileFloats;
    // Ensure tile length divides row length? Not required but good for alignment
    // Simplify: tileLen is the number of elements per tile (along row)
    // We may restrict to be <= rowLen
    if (rowLen < tileLen) {
        tileLen = AlignDown(rowLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowLen;
    }

    uint32_t numTilesPerRow = (tileLen == 0) ? 1 : CeilDiv(rowLen, tileLen);
    uint32_t numRows = (rowLen == 0) ? 0 : totalLength / rowLen;
    uint32_t totalTiles = numRows * numTilesPerRow;

    // Block dim: aim for 8-24 cores, use number of rows (each row processed by one block)
    uint32_t blockDim = totalTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_rowLen(rowLen);
    tiling.set_dim(static_cast<int32_t>(dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t rowLen, int32_t dim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        rowLen_ = rowLen;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(maxBuf_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(expBuf_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sumBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || rowLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a set of rows (grid-stride over rows)
        uint32_t numRows = totalLen_ / rowLen_;
        uint32_t strideRows = blockNum;
        for (uint32_t r = blockIdx; r < numRows; r += strideRows) {
            ProcessRow(r);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        uint32_t rowStart = rowIdx * rowLen_;
        uint32_t tileNum = (tileLen_ == 0) ? 1 : (rowLen_ + tileLen_ - 1) / tileLen_;

        // ---------- Pass 1: compute row max ----------
        float rowMax = -FLT_MAX;
        for (uint32_t t = 0; t < tileNum; ++t) {
            uint32_t offset = rowStart + t * tileLen_;
            uint32_t curLen = (rowLen_ - t * tileLen_ < tileLen_) ? rowLen_ - t * tileLen_ : tileLen_;
            if (curLen == 0) continue;

            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            // Compute max of tile using vector min? Actually use a scalar loop for simplicity
            float tileMax = -FLT_MAX;
            for (uint32_t i = 0; i < curLen; ++i) {
                float val = xLocal(i);
                if (val > tileMax) tileMax = val;
            }
            inQueueX_.FreeTensor(xLocal);
            if (tileMax > rowMax) rowMax = tileMax;
        }

        // ---------- Pass 2: compute sum of exp(x - max) ----------
        float rowSum = 0.0f;
        for (uint32_t t = 0; t < tileNum; ++t) {
            uint32_t offset = rowStart + t * tileLen_;
            uint32_t curLen = (rowLen_ - t * tileLen_ < tileLen_) ? rowLen_ - t * tileLen_ : tileLen_;
            if (curLen == 0) continue;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);

            // exp(x - max)
            AscendC::LocalTensor<float> expLocal = expBuf_.AllocTensor<float>();
            AscendC::Sub(expLocal, xLocal, rowMax, curLen);   // x - max
            AscendC::Exp(expLocal, expLocal, curLen);         // exp(x - max)

            // sum the tile
            float tileSum = 0.0f;
            for (uint32_t i = 0; i < curLen; ++i) {
                tileSum += expLocal(i);
            }
            rowSum += tileSum;

            expBuf_.FreeTensor(expLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        // log(sum)
        float logSum = logf(rowSum);

        // ---------- Pass 3: compute output = x - max - log(sum) ----------
        for (uint32_t t = 0; t < tileNum; ++t) {
            uint32_t offset = rowStart + t * tileLen_;
            uint32_t curLen = (rowLen_ - t * tileLen_ < tileLen_) ? rowLen_ - t * tileLen_ : tileLen_;
            if (curLen == 0) continue;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            // y = x - max - logSum
            AscendC::Sub(yLocal, xLocal, rowMax, curLen);
            AscendC::Sub(yLocal, yLocal, logSum, curLen);

            // Write out
            AscendC::DataCopy(yGm_[offset], yLocal, curLen);

            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maxBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> expBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t rowLen_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.rowLen, tiling_data.dim);
    op.Process();
}
