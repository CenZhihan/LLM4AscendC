host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SoftmaxCustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalRows = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t colLength = static_cast<uint32_t>(shape.GetDim(1));
    uint32_t totalElements = totalRows * colLength;

    // UB budget: temporary buffers needed:
    // inQueueX: 2 tiles (double buffer)
    // outQueueY: 2 tiles
    // tmpLocalMax: 1 colTile
    // tmpLocalSum: 1 colTile
    // tmpLocalExp: 1 colTile
    // Total floats per inner loop tile: 2*colTile (in/out) + 3*colTile (tmp) = 5*colTile
    // But we also need to hold rowsPerLoop rows simultaneously.
    // Assume rowsPerLoop = 1 for simplicity; then colTileSize bounded by UB/5.
    uint32_t maxColTile = UB_BUDGET_BYTES / (5 * sizeof(float));  // elements
    maxColTile = AlignDown(maxColTile, ALIGN_ELEMS);
    if (maxColTile < ALIGN_ELEMS) maxColTile = ALIGN_ELEMS;
    // Ensure tile doesn't exceed column length
    uint32_t colTileSize = (colLength < maxColTile) ? colLength : maxColTile;
    if (colTileSize > totalElements) colTileSize = totalElements;  // safety

    // Block dimension: each block processes an equal number of rows
    // Use all available cores (max 40 on 910B)
    uint32_t blockCount = context->GetBlockNum();
    if (blockCount == 0) blockCount = 1;
    uint32_t rowsPerBlock = CeilDiv(totalRows, blockCount);
    uint32_t blockDim = blockCount;
    context->SetBlockDim(blockDim);

    // Rows per inner loop: we process one row at a time internally (to simplify)
    uint32_t rowsPerLoop = 1;

    tiling.set_totalRows(totalRows);
    tiling.set_colLength(colLength);
    tiling.set_blockDim(blockDim);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_rowsPerLoop(rowsPerLoop);
    tiling.set_colTileSize(colTileSize);
    tiling.set_ubBudgetFloat(UB_BUDGET_BYTES / sizeof(float));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;  // no workspace needed

    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, 
                                uint32_t totalRows, uint32_t colLength,
                                uint32_t rowsPerBlock, uint32_t rowsPerLoop,
                                uint32_t colTileSize)
    {
        totalRows_ = totalRows;
        colLength_ = colLength;
        rowsPerBlock_ = rowsPerBlock;
        rowsPerLoop_ = rowsPerLoop;  // always 1
        colTileSize_ = colTileSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows * colLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows * colLength);

        // Double buffers for input and output
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, colTileSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, colTileSize_ * sizeof(float));
        // Temporary buffers for max, sum, exp (single use each)
        pipe_.InitBuffer(tmpMax_, 1, colTileSize_ * sizeof(float));
        pipe_.InitBuffer(tmpSum_, 1, colTileSize_ * sizeof(float));
        pipe_.InitBuffer(tmpExp_, 1, colTileSize_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowStart = blockIdx * rowsPerBlock_;
        uint32_t rowEnd = rowStart + rowsPerBlock_;
        if (rowEnd > totalRows_) rowEnd = totalRows_;

        // Iterate over rows assigned to this block
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            ProcessOneRow(r);
        }
    }

private:
    __aicore__ inline void ProcessOneRow(uint32_t rowIdx) {
        // Process one row in column tiles
        uint32_t colOffset = rowIdx * colLength_;
        uint32_t tileCount = (colLength_ + colTileSize_ - 1) / colTileSize_;
        float rowMax = -FLT_MAX;
        float rowSum = 0.0f;

        // First pass: compute global max and subtract, exp, sum
        // We need two passes: first to find max, second to compute exp and sum.
        // To avoid storing all tiles, we do a two-pass approach with double buffering reuse.

        // Pass 1: find max across all tiles
        for (uint32_t t = 0; t < tileCount; ++t) {
            uint32_t tileLen = (t == tileCount-1) ? (colLength_ - t * colTileSize_) : colTileSize_;
            uint32_t gmOffset = colOffset + t * colTileSize_;

            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], tileLen);
            inQueueX_.EnQue(xLocal);

            // Deque and compute local max
            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> maxLocal = tmpMax_.Get<float>();
            AscendC::ReduceMax(maxLocal, xLocal, tileLen);
            float currentMax = maxLocal.GetValue(0);  // scalar
            inQueueX_.FreeTensor(xLocal);

            if (currentMax > rowMax) rowMax = currentMax;
        }

        // Pass 2: compute exp(x - max) and sum
        for (uint32_t t = 0; t < tileCount; ++t) {
            uint32_t tileLen = (t == tileCount-1) ? (colLength_ - t * colTileSize_) : colTileSize_;
            uint32_t gmOffset = colOffset + t * colTileSize_;

            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], tileLen);
            inQueueX_.EnQue(xLocal);

            // Deque
            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> expLocal = tmpExp_.Get<float>();
            AscendC::LocalTensor<float> sumLocal = tmpSum_.Get<float>();

            // exp(x - rowMax)
            AscendC::Sub(expLocal, xLocal, rowMax, tileLen);
            AscendC::Exp(expLocal, expLocal, tileLen);

            // Accumulate sum (need to reduce across tiles)
            AscendC::ReduceSum(sumLocal, expLocal, tileLen);
            float tileSum = sumLocal.GetValue(0);  // scalar
            rowSum += tileSum;

            // Free xLocal, but we still need expLocal for the output; we will reuse it later.
            // Actually we need to keep expLocal until second pass normalizes.
            // To avoid extra buffer, we can store exp back to the same tile? Not possible because we need x again for normalization.
            // Better: in the same pass, after computing exp, we write to output directly? But output needs to be normalized.
            // This two-pass approach with double total memory is not ideal.
            // We'll redo the exp computation in the normalization pass to save memory (recompute).
            inQueueX_.FreeTensor(xLocal);
        }

        // Now rowSum and rowMax are known.
        // Second pass: compute normalized output
        for (uint32_t t = 0; t < tileCount; ++t) {
            uint32_t tileLen = (t == tileCount-1) ? (colLength_ - t * colTileSize_) : colTileSize_;
            uint32_t gmOffset = colOffset + t * colTileSize_;

            // Load x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], tileLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> expLocal = tmpExp_.Get<float>();

            // exp(x - rowMax)
            AscendC::Sub(expLocal, xLocal, rowMax, tileLen);
            AscendC::Exp(expLocal, expLocal, tileLen);
            // Normalize: y = exp / rowSum
            AscendC::Div(yLocal, expLocal, rowSum, tileLen);

            // Enqueue output
            outQueueY_.EnQue<float>(yLocal);
            inQueueX_.FreeTensor(xLocal);

            // Copy out
            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[gmOffset], yOut, tileLen);
            outQueueY_.FreeTensor(yOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSum_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpExp_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalRows_ {0};
    uint32_t colLength_ {0};
    uint32_t rowsPerBlock_ {0};
    uint32_t rowsPerLoop_ {1};
    uint32_t colTileSize_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalRows, tiling_data.colLength,
            tiling_data.rowsPerBlock, tiling_data.rowsPerLoop,
            tiling_data.colTileSize);
    op.Process();
}
