#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SoftmaxCustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalRows = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t colLength = static_cast<uint32_t>(shape.GetDim(1));

    // UB budget: 5 * colTileSize floats (2 in/out + 3 tmp)
    uint32_t maxColTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxColTile = AlignDown(maxColTile, ALIGN_ELEMS);
    if (maxColTile < ALIGN_ELEMS) maxColTile = ALIGN_ELEMS;
    uint32_t colTileSize = (colLength < maxColTile) ? colLength : maxColTile;

    uint32_t blockDim = (totalRows < 24) ? totalRows : 24;
    uint32_t rowsPerBlock = CeilDiv(totalRows, blockDim);
    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_colLength(colLength);
    tiling.set_blockDim(blockDim);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_colTileSize(colTileSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {

    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    const ge::DataType x_dtype = context->GetInputDataType(0);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t colLength,
                                uint32_t rowsPerBlock, uint32_t colTileSize)
    {
        totalRows_ = totalRows;
        colLength_ = colLength;
        rowsPerBlock_ = rowsPerBlock;
        colTileSize_ = colTileSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows * colLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows * colLength);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, colTileSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, colTileSize_ * sizeof(float));
        pipe_.InitBuffer(tmpMax_, colTileSize_ * sizeof(float));
        pipe_.InitBuffer(tmpSum_, colTileSize_ * sizeof(float));
        pipe_.InitBuffer(tmpExp_, colTileSize_ * sizeof(float));
        pipe_.InitBuffer(scalarBuf_, 1 * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowStart = blockIdx * rowsPerBlock_;
        uint32_t rowEnd = rowStart + rowsPerBlock_;
        if (rowEnd > totalRows_) rowEnd = totalRows_;

        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            ProcessOneRow(r);
        }
    }

private:
    __aicore__ inline void ProcessOneRow(uint32_t rowIdx) {
        uint32_t colOffset = rowIdx * colLength_;
        uint32_t tileCount = (colLength_ + colTileSize_ - 1) / colTileSize_;
        float rowMax = -1e30f;

        // First pass: compute global max across tiles
        for (uint32_t t = 0; t < tileCount; ++t) {
            uint32_t tileLen = (t == tileCount-1) ? (colLength_ - t * colTileSize_) : colTileSize_;
            uint32_t gmOffset = colOffset + t * colTileSize_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], tileLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> maxLocal = tmpMax_.Get<float>();
            AscendC::ReduceMax(maxLocal, xLocal, tileLen, 1);
            float currentMax = maxLocal.GetValue(0);
            inQueueX_.FreeTensor(xLocal);

            if (currentMax > rowMax) rowMax = currentMax;
        }

        // Second pass: compute exp(x - max), sum, then normalize
        float rowSum = 0.0f;
        // We need to store per-tile sum for later normalization (since rowSum accumulates)
        // But we will compute normalized output per tile using accumulated sum so far? That is incorrect because full sum needed. So we need two sub-passes: first compute exp(x-max) and sum across all tiles, then second sub-pass normalize.
        // To avoid three passes, we can store exp values in global memory or local tiles. Simpler: two passes: first compute exp and sum, second normalize.
        // Let's do a separate pass for exp and sum, then a third pass for normalization.
        // Given UB constraints, we can store exp temporarily in local tile but we need to keep all tiles? Not feasible. So we'll use global memory for intermediate exp? That adds complexity.
        // Instead, adopt standard approach: compute max, then compute exp(x-max) and accumulate sum, then compute norm in same pass? But norm requires full sum. So we must either store exp or recompute. We'll recompute: third pass re-reads x and uses rowMax and rowSum to compute y.
        // So we need a third pass.
        // First compute exp(x-max) and sum.
        for (uint32_t t = 0; t < tileCount; ++t) {
            uint32_t tileLen = (t == tileCount-1) ? (colLength_ - t * colTileSize_) : colTileSize_;
            uint32_t gmOffset = colOffset + t * colTileSize_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], tileLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> expLocal = tmpExp_.Get<float>();
            AscendC::LocalTensor<float> sumLocal = tmpSum_.Get<float>();

            // exp(x - rowMax) = exp(x + (-rowMax))
            AscendC::Adds(expLocal, xLocal, -rowMax, tileLen);
            AscendC::Exp(expLocal, expLocal, tileLen);

            AscendC::ReduceSum(sumLocal, expLocal, tileLen, 1);
            float tileSum = sumLocal.GetValue(0);
            rowSum += tileSum;

            inQueueX_.FreeTensor(xLocal);
        }

        // Third pass: normalize and write output
        float recipSum = 1.0f / rowSum;
        for (uint32_t t = 0; t < tileCount; ++t) {
            uint32_t tileLen = (t == tileCount-1) ? (colLength_ - t * colTileSize_) : colTileSize_;
            uint32_t gmOffset = colOffset + t * colTileSize_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], tileLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> expLocal = tmpExp_.Get<float>();
            AscendC::Adds(expLocal, xLocal, -rowMax, tileLen);
            AscendC::Exp(expLocal, expLocal, tileLen);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Muls(yLocal, expLocal, recipSum, tileLen);
            outQueueY_.EnQue<float>(yLocal);

            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[gmOffset], yOut, tileLen);
            outQueueY_.FreeTensor(yOut);

            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpMax_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpSum_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpExp_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> scalarBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalRows_ {0};
    uint32_t colLength_ {0};
    uint32_t rowsPerBlock_ {0};
    uint32_t colTileSize_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalRows, tiling_data.colLength,
            tiling_data.rowsPerBlock, tiling_data.colTileSize);
    op.Process();
}
