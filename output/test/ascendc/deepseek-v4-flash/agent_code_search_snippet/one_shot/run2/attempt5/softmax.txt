host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include "kernel_operator.h"

namespace optiling {
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SoftmaxCustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalRows = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t colLength = static_cast<uint32_t>(shape.GetDim(1));

    // Determine colTileSize from UB budget:
    // We need inQueueX: 2 * colTileSize floats, outQueueY: 2 * colTileSize,
    // tmpMax: rowsPerTile, tmpSum: rowsPerTile, sharedTmpBuffer: ~2 * colTileSize (softmax workspace)
    // Total ~ (4 + 2) * colTileSize = 6 * colTileSize floats.
    uint32_t maxColTile = UB_BUDGET_BYTES / (6 * sizeof(float));
    maxColTile = AlignDown(maxColTile, ALIGN_ELEMS);
    if (maxColTile < ALIGN_ELEMS) maxColTile = ALIGN_ELEMS;
    uint32_t colTileSize = (colLength < maxColTile) ? colLength : maxColTile;

    // Block dimension: up to 24 cores
    uint32_t blockDim = (totalRows < 24) ? totalRows : 24;
    uint32_t rowsPerBlock = CeilDiv(totalRows, blockDim);

    // Shared tmp buffer size for softmax: empirically ~ 2 * colTileSize * sizeof(float)
    uint32_t sharedTmpBufferSize = 2 * colTileSize * sizeof(float);
    if (sharedTmpBufferSize < 61440) sharedTmpBufferSize = 61440;  // min 60 KB

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_colLength(colLength);
    tiling.set_blockDim(blockDim);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_colTileSize(colTileSize);
    tiling.set_sharedTmpBufferSize(sharedTmpBufferSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t FLOAT_NUM_OF_SINGLE_BLOCK = 1;  // per row max/sum is scalar

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t colLength,
                                uint32_t rowsPerBlock, uint32_t colTileSize,
                                uint32_t sharedTmpBufferSize)
    {
        totalRows_ = totalRows;
        colLength_ = colLength;
        rowsPerBlock_ = rowsPerBlock;
        colTileSize_ = colTileSize;
        sharedTmpBufferSize_ = sharedTmpBufferSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows * colLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows * colLength);

        // Each block processes `rowsPerBlock` rows per iteration
        uint32_t blockOffset = blockIdx_ * rowsPerBlock_ * colLength_;
        blockEnd_ = (blockIdx_ + 1) * rowsPerBlock_ * colLength_;
        if (blockEnd_ > totalRows_ * colLength_)
            blockEnd_ = totalRows_ * colLength_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, colTileSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, colTileSize_ * sizeof(float));
        pipe_.InitBuffer(queueMax_, 1, rowsPerBlock_ * FLOAT_NUM_OF_SINGLE_BLOCK * sizeof(float));
        pipe_.InitBuffer(queueSum_, 1, rowsPerBlock_ * FLOAT_NUM_OF_SINGLE_BLOCK * sizeof(float));
        pipe_.InitBuffer(sharedTmpBuffer_, sharedTmpBufferSize_);
        pipe_.InitBuffer(softmaxTilingBuf_, sizeof(AscendC::SoftMaxTiling));
    }

    __aicore__ inline void Process()
    {
        blockIdx_ = AscendC::GetBlockIdx();
        if (blockIdx_ >= (totalRows_ + rowsPerBlock_ - 1) / rowsPerBlock_)
            return;

        // Iterate over rows in this block, tiling by columns
        for (uint32_t rowStart = blockIdx_ * rowsPerBlock_; rowStart < totalRows_; rowStart += rowsPerBlock_) {
            uint32_t batchRows = rowsPerBlock_;
            if (rowStart + batchRows > totalRows_)
                batchRows = totalRows_ - rowStart;

            // Process all columns for these rows
            for (uint32_t colStart = 0; colStart < colLength_; colStart += colTileSize_) {
                uint32_t tileCols = colTileSize_;
                if (colStart + tileCols > colLength_)
                    tileCols = colLength_ - colStart;

                // Copy input tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowStart * colLength_ + colStart], batchRows * tileCols);
                inQueueX_.EnQue(xLocal);

                xLocal = inQueueX_.DeQue<float>();

                // Allocate output tile, max, sum, workspace
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> maxLocal = queueMax_.AllocTensor<float>();
                AscendC::LocalTensor<float> sumLocal = queueSum_.AllocTensor<float>();
                AscendC::LocalTensor<uint8_t> tmpBuffer = sharedTmpBuffer_.Get<uint8_t>();
                AscendC::LocalTensor<AscendC::SoftMaxTiling> tilingLocal = softmaxTilingBuf_.Get<AscendC::SoftMaxTiling>();

                // Initialize SoftMax tiling (default is sufficient for float32)
                AscendC::SetSoftMaxTiling(tilingLocal, AscendC::SoftMaxParam<float>());

                // SoftMax shape: batchRows rows, tileCols columns
                AscendC::SoftMaxShapeInfo srcShape = {batchRows, tileCols, batchRows, tileCols};

                // Compute softmax
                AscendC::SoftMax<float, true>(yLocal, sumLocal, maxLocal, xLocal, tmpBuffer, tilingLocal, srcShape);

                // Write output to global
                AscendC::DataCopy(yGm_[rowStart * colLength_ + colStart], yLocal, batchRows * tileCols);

                // Free tensors
                outQueueY_.FreeTensor(yLocal);
                queueMax_.FreeTensor(maxLocal);
                queueSum_.FreeTensor(sumLocal);
                inQueueX_.FreeTensor(xLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> queueMax_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> queueSum_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> sharedTmpBuffer_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> softmaxTilingBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t colLength_ {0};
    uint32_t rowsPerBlock_ {0};
    uint32_t colTileSize_ {0};
    uint32_t sharedTmpBufferSize_ {0};
    uint32_t blockIdx_ {0};
    uint32_t blockEnd_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalRows, tiling_data.colLength,
            tiling_data.rowsPerBlock, tiling_data.colTileSize,
            tiling_data.sharedTmpBufferSize);
    op.Process();
}
