host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // per‑core UB share
static constexpr uint32_t ALIGN_ELEMS   = 128;
static constexpr uint32_t MAX_COL_ELEMS = 1024;           // assume each row fits in a tile

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)   { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    // input shape: 2D (batch, cols)
    auto input_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = input_shape.GetShapeSize();
    uint32_t columnLength = input_shape.GetDim(1);   // dim 1 is the reduction axis
    uint32_t rowLength = columnLength; // all rows share the same length
    uint32_t totalRows = totalLength / columnLength;

    // Assume each row fits in UB -> columnLength <= MAX_COL_ELEMS
    // If not, the host tiling should split rows further; here we keep it simple.
    uint32_t tileLen = columnLength;   // one tile = one full row

    // Determine block dimension: distribute rows across cores
    uint32_t blockDim = totalRows;
    if (blockDim < 8)  blockDim = 8;
    if (blockDim > 24) blockDim = 24;  // keep conservative

    // Tail handling: cores 0..usedNumBlocks-1 get an equal share
    uint32_t usedNumBlocks = blockDim;
    uint32_t rowsPerCore = totalRows / usedNumBlocks;
    uint32_t tailRows = totalRows % usedNumBlocks;

    uint32_t coreRowNum = rowsPerCore + (tailRows > 0 ? 1 : 0);
    uint32_t singleLoopCoreRowNum = coreRowNum;  // no further splitting
    uint32_t leftRow = 0;                        // no tail within core
    uint32_t singleCoreLoopCount = (coreRowNum > 0) ? 1 : 0;
    uint32_t tailCoreSingleLoopCoreRowNum = (tailRows > 0) ? tailRows : 0;
    uint32_t tailCoreSingleCoreLoopCount = (tailRows > 0) ? 1 : 0;
    uint32_t tailCoreSingleCoreLoopTail = 0;

    // Shared temporary buffer (used by library internally)
    uint32_t sharedTmpBufferSize = 60 * 1024; // 60 KB

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_columnLength(columnLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_usedNumBlocks(usedNumBlocks);
    tiling.set_coreRowNum(coreRowNum);
    tiling.set_singleLoopCoreRowNum(singleLoopCoreRowNum);
    tiling.set_singleCoreLoopCount(singleCoreLoopCount);
    tiling.set_leftRow(leftRow);
    tiling.set_tailCoreSingleLoopCoreRowNum(tailCoreSingleLoopCoreRowNum);
    tiling.set_tailCoreSingleCoreLoopCount(tailCoreSingleCoreLoopCount);
    tiling.set_tailCoreSingleCoreLoopTail(tailCoreSingleCoreLoopTail);
    tiling.set_sharedTmpBufferSize(sharedTmpBufferSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    *context->GetOutputShape(0) = *context->GetInputShape(0);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength, uint32_t columnLength,
                                uint32_t tileLength, uint32_t blockDim,
                                uint32_t usedNumBlocks, uint32_t coreRowNum,
                                uint32_t singleLoopCoreRowNum, uint32_t singleCoreLoopCount,
                                uint32_t leftRow, uint32_t tailCoreSingleLoopCoreRowNum,
                                uint32_t tailCoreSingleCoreLoopCount, uint32_t tailCoreSingleCoreLoopTail,
                                uint32_t sharedTmpBufferSize)
    {
        totalLen_ = totalLength;
        columnLen_ = columnLength;
        tileLen_ = tileLength;
        usedNumBlocks_ = usedNumBlocks;
        coreRowNum_ = coreRowNum;
        singleLoopCoreRowNum_ = singleLoopCoreRowNum;
        singleCoreLoopCount_ = singleCoreLoopCount;
        leftRow_ = leftRow;
        tailCoreSingleLoopCoreRowNum_ = tailCoreSingleLoopCoreRowNum;
        tailCoreSingleCoreLoopCount_ = tailCoreSingleCoreLoopCount;
        tailCoreSingleCoreLoopTail_ = tailCoreSingleCoreLoopTail;
        sharedTmpBufferSize_ = sharedTmpBufferSize;

        // Tail core adjustment
        if (AscendC::GetBlockIdx() == usedNumBlocks_) {
            singleLoopCoreRowNum_ = tailCoreSingleLoopCoreRowNum_;
            singleCoreLoopCount_ = tailCoreSingleCoreLoopCount_;
            leftRow_ = tailCoreSingleCoreLoopTail_;
        }

        blockLength_ = coreRowNum_ * columnLen_;
        uint32_t offset = blockLength_ * AscendC::GetBlockIdx();
        xGm_.SetGlobalBuffer((__gm__ float*)x + offset, blockLength_);
        yGm_.SetGlobalBuffer((__gm__ float*)y + offset, blockLength_);

        // Queues
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));

        // Internal max/sum queues (only used by LogSoftMax library)
        uint32_t msTile = singleLoopCoreRowNum_ * 64;   // 64 floats per block (safe)
        pipe_.InitBuffer(queueMax_, 1, msTile * sizeof(float));
        pipe_.InitBuffer(queueSum_, 1, msTile * sizeof(float));
        pipe_.InitBuffer(tmpBuffer_, sharedTmpBufferSize_);
    }

    __aicore__ inline void Process()
    {
        if (AscendC::GetBlockIdx() > usedNumBlocks_)
            return;

        for (int32_t i = 0; i < singleCoreLoopCount_; i++) {
            CopyIn(i, singleLoopCoreRowNum_);
            Compute(i, singleLoopCoreRowNum_);
            CopyOut(i, singleLoopCoreRowNum_);
        }
        if (leftRow_ > 0) {
            CopyIn(singleCoreLoopCount_, leftRow_);
            Compute(singleCoreLoopCount_, leftRow_);
            CopyOut(singleCoreLoopCount_, leftRow_);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress, uint32_t rowNum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[progress * tileLen_], rowNum * columnLen_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress, uint32_t rowNum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::LocalTensor<float> maxLocal = queueMax_.AllocTensor<float>();
        AscendC::LocalTensor<float> sumLocal = queueSum_.AllocTensor<float>();
        AscendC::LocalTensor<uint8_t> tmp = tmpBuffer_.Get<uint8_t>();

        AscendC::SoftMaxShapeInfo shape = {rowNum, columnLen_, rowNum, columnLen_};

        // Use the library LogSoftMax (assumes the same interface as SoftMax)
        AscendC::LogSoftMax<float, true>(yLocal, sumLocal, maxLocal, xLocal, tmp, shape);

        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
        queueMax_.FreeTensor(maxLocal);
        queueSum_.FreeTensor(sumLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress, uint32_t rowNum)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[progress * tileLen_], yLocal, rowNum * columnLen_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> queueMax_, queueSum_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuffer_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t columnLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t usedNumBlocks_ {0};
    uint32_t coreRowNum_ {0};
    uint32_t blockLength_ {0};
    uint32_t singleLoopCoreRowNum_ {0};
    uint32_t singleCoreLoopCount_ {0};
    uint32_t leftRow_ {0};
    uint32_t tailCoreSingleLoopCoreRowNum_ {0};
    uint32_t tailCoreSingleCoreLoopCount_ {0};
    uint32_t tailCoreSingleCoreLoopTail_ {0};
    uint32_t sharedTmpBufferSize_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y,
            td.totalLength, td.rowLength, td.columnLength,
            td.tileLength, td.blockDim,
            td.usedNumBlocks, td.coreRowNum,
            td.singleLoopCoreRowNum, td.singleCoreLoopCount,
            td.leftRow, td.tailCoreSingleLoopCoreRowNum,
            td.tailCoreSingleCoreLoopCount, td.tailCoreSingleCoreLoopTail,
            td.sharedTmpBufferSize);
    op.Process();
}
"""
