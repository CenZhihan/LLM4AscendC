host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: 5 * rowLength floats per row (x, y, tmp) * 2 buffered = 10 * rowLength floats => 40 * rowLength bytes
// We also need workspace for reduction. But we keep conservative.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalLength = static_cast<uint32_t>(inputShape->GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dimVal = *dimPtr;

    // Get dimension sizes
    uint32_t numDims = static_cast<uint32_t>(inputShape->GetDimNum());
    // Handle negative dim
    if (dimVal < 0) dimVal += static_cast<int32_t>(numDims);
    // Clamp
    if (dimVal < 0) dimVal = 0;
    if (dimVal >= static_cast<int32_t>(numDims)) dimVal = static_cast<int32_t>(numDims) - 1;

    uint32_t rowLength = static_cast<uint32_t>(inputShape->GetDim(dimVal));
    uint32_t numRows = totalLength / rowLength;

    // Compute tileLen: number of elements per tile (must be multiple of rowLength)
    // UB budget per tile: we need x (tile) + y (tile) + tmp (tile) = 3*tile floats, buffered x2 => 6*tile floats => 24*tile bytes
    // plus reduction workspace (could be same as tmp). So take 30*tile bytes.
    uint32_t maxTile = UB_BUDGET_BYTES / 30; // number of floats
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < rowLength) maxTile = rowLength;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, rowLength);

    // Ensure tileLen is multiple of rowLength
    uint32_t tileRows = maxTile / rowLength;
    if (tileRows < 1) tileRows = 1;
    uint32_t tileLen = tileRows * rowLength;
    if (tileLen > totalLength) {
        tileLen = AlignDown(totalLength, rowLength);
        if (tileLen == 0) tileLen = totalLength;
    }

    // Block dim: one tile per block, but limit
    uint32_t numTiles = CeilDiv(numRows, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dimVal);
    tiling.set_rowLength(rowLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops
"""

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t dim, uint32_t rowLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        dim_ = dim;
        rowLen_ = rowLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, rowLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, rowLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, rowLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || rowLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalRows = totalLen_ / rowLen_;
        const uint32_t tileRows = tileLen_ / rowLen_;
        const uint32_t rowStride = blockNum * tileRows;

        uint32_t curRow = blockIdx * tileRows;
        if (curRow >= totalRows) return;

        uint32_t curOffset = curRow * rowLen_;
        CopyIn(curOffset);

        while (true) {
            uint32_t nextRow = curRow + rowStride;
            uint32_t nextOffset = nextRow * rowLen_;
            bool hasNext = (nextRow < totalRows);
            if (hasNext) {
                CopyIn(nextOffset);
            }

            Compute(curOffset);
            CopyOut(curOffset);

            if (!hasNext) break;
            curRow = nextRow;
            curOffset = nextOffset;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], rowLen_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t gmOffset)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // tLocal = xLocal
        AscendC::DataCopy(tLocal, xLocal, rowLen_);

        // maxVal = ReduceMax(tLocal)
        float maxVal = AscendC::ReduceMax(tLocal, rowLen_);

        // xLocal = xLocal - maxVal
        AscendC::Subs(xLocal, xLocal, maxVal, rowLen_);

        // tLocal = exp(xLocal)
        AscendC::Exp(tLocal, xLocal, rowLen_);

        // sumVal = ReduceSum(tLocal)
        float sumVal = AscendC::ReduceSum(tLocal, rowLen_);

        // logSum = log(sumVal)
        float logSum = AscendC::Log(sumVal);  // scalar log

        // yLocal = xLocal - (maxVal + logSum)
        float addVal = maxVal + logSum;
        AscendC::Subs(yLocal, xLocal, addVal, rowLen_);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, rowLen_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ {1};
    uint32_t rowLen_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.dim, tiling_data.rowLength);
    op.Process();
}
"""
