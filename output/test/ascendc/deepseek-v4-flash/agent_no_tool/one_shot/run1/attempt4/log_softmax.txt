host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cstdint>

namespace optiling {

// UB model: need 5 * tileLen floats (row, exp, max, sum, tmp) for reduction phases
// We also need a few scalars, but we account with a safe multiple.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;            // vector alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;      // cap per iteration latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    uint32_t outerSize = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(0));
    uint32_t innerSize = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(1));
    uint32_t totalLength = outerSize * innerSize;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim = attrs->GetAttrPointer<int64_t>(0);
    uint32_t dimVal = static_cast<uint32_t>(*dim);

    // Determine tileLength from UB budget: need at least 5 * tileLen floats
    uint32_t maxElem = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxElem = AlignDown(maxElem, ALIGN_ELEMS);
    if (maxElem < ALIGN_ELEMS) maxElem = ALIGN_ELEMS;
    if (maxElem > HARD_CAP_ELEMS) maxElem = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // For log-softmax we process whole rows, but if innerSize is too large we must tile inner dim
    uint32_t tileLen = (innerSize < maxElem) ? innerSize : maxElem;
    // Ensure tileLen divides innerSize (or at least aligns)
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = ALIGN_ELEMS;
    if (tileLen > innerSize) tileLen = innerSize;

    // BlockDim based on number of rows per core
    uint32_t numRowsPerTile = 1; // each core processes rows independently
    uint32_t blockDim = (outerSize + 7) / 8;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dimVal);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLen, uint32_t outerSize,
                                uint32_t innerSize, uint32_t tileLen,
                                uint32_t dim)
    {
        totalLen_ = totalLen;
        outerSize_ = outerSize;
        innerSize_ = innerSize;
        tileLen_ = tileLen;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // Two temporary buffers for reduction (max, exp, sum, etc.)
        pipe_.InitBuffer(tmpQueue_, 2, tileLen_ * sizeof(float));
        // Scalar queue for per‑tile max/sum (1 element)
        pipe_.InitBuffer(scalarQueue_, 2, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid‑stride over rows
        uint32_t rowIdx = blockIdx;
        if (rowIdx >= outerSize_) return;

        while (true) {
            // Compute one row
            // Since innerSize may exceed UB, we tile the inner dimension
            // We perform a two‑pass reduction: first gather max and sum over tiles,
            // then compute logsoftmax per tile.

            uint32_t globalOffset = rowIdx * innerSize_;

            // Pass 1: compute row max and sum of exp(x - max)
            float rowMax = -INFINITY;
            float rowSumExp = 0.0f;
            for (uint32_t tileStart = 0; tileStart < innerSize_; tileStart += tileLen_) {
                uint32_t curLen = innerSize_ - tileStart;
                if (curLen > tileLen_) curLen = tileLen_;

                // Load tile into local
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[globalOffset + tileStart], curLen);
                inQueueX_.EnQue(xLocal);
                xLocal = inQueueX_.DeQue<float>();

                // Find local max
                AscendC::LocalTensor<float> t0 = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceMax(t0, xLocal, curLen);    // t0 holds scalar in first element
                float localMax = t0(0);
                // Update global max
                if (localMax > rowMax) rowMax = localMax;

                // Compute exp(x - rowMax) for this tile using current rowMax (approx)
                // We will correct after final max. For now use temporary max = localMax
                AscendC::Sub(t0, xLocal, localMax, curLen);
                AscendC::Exp(t0, t0, curLen);
                // Sum exp values in tile
                AscendC::LocalTensor<float> t1 = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceSum(t1, t0, curLen);
                float tileSum = t1(0);
                rowSumExp += tileSum;

                tmpQueue_.FreeTensor(t1);
                tmpQueue_.FreeTensor(t0);
                inQueueX_.FreeTensor(xLocal);
            }

            // Now we have exact rowMax and rowSumExp.
            // Pass 2: recompute exp(x - rowMax) and accumulate final result: y = x - (log(rowSumExp) + rowMax)
            float logSumExp = logf(rowSumExp) + rowMax;
            AscendC::LocalTensor<float> logSumScalar = scalarQueue_.AllocTensor<float>();
            logSumScalar(0) = logSumExp;

            for (uint32_t tileStart = 0; tileStart < innerSize_; tileStart += tileLen_) {
                uint32_t curLen = innerSize_ - tileStart;
                if (curLen > tileLen_) curLen = tileLen_;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[globalOffset + tileStart], curLen);
                inQueueX_.EnQue(xLocal);
                xLocal = inQueueX_.DeQue<float>();

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                // y = x - log_sum
                AscendC::Sub(yLocal, xLocal, logSumScalar, curLen);

                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xLocal);

                // Copy result to global
                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[globalOffset + tileStart], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
            }

            scalarQueue_.FreeTensor(logSumScalar);

            // Move to next row
            rowIdx += blockNum;
            if (rowIdx >= outerSize_) break;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outerSize_ {0};
    uint32_t innerSize_ {0};
    uint32_t tileLen_ {0};
    uint32_t dim_ {1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.outerSize,
            tiling_data.innerSize,
            tiling_data.tileLength,
            tiling_data.dim);
    op.Process();
}
