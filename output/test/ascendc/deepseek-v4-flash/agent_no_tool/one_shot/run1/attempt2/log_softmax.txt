static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const std::vector<int64_t> dims = inputShape->GetOriginShape().GetDims();
    // Assume 2D for simplicity; for general case we would compute from dim.
    uint32_t outerSize = static_cast<uint32_t>(dims[0]);
    uint32_t innerSize = static_cast<uint32_t>(dims[1]);
    uint32_t totalLength = outerSize * innerSize;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim = attrs->GetAttrPointer<int64_t>(0);
    uint32_t dimVal = static_cast<uint32_t>(*dim);

    // UB budget: each row needs 5 * innerSize floats (row, exp, tmp, max scalars simplify)
    // We'll process rows sequentially, one at a time. tileLength = innerSize.
    uint32_t tileLen = innerSize;
    // BlockDim = min(outerSize, 24) to utilize cores.
    uint32_t blockDim = outerSize < 24 ? outerSize : 24;
    if (blockDim < 8) blockDim = 8;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dimVal);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLen, uint32_t outerSize,
                                uint32_t innerSize, uint32_t tileLen,
                                uint32_t dim)
    {
        totalLen_ = totalLen;
        outerSize_ = outerSize;
        innerSize_ = innerSize;
        tileLen_ = tileLen;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // tmp queue: two buffers of size innerSize for max/exp/sum operations
        pipe_.InitBuffer(tmpQueue_, 2, tileLen_ * sizeof(float));
        // scalar buffers for max and sum (1 element each)
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        uint32_t rowIdx = blockIdx;
        if (rowIdx >= outerSize_) return;

        while (true) {
            // Prefetch next row for overlap
            uint32_t nextRow = rowIdx + blockNum;
            if (nextRow < outerSize_) {
                uint32_t gmOffset = nextRow * innerSize_;
                CopyIn(gmOffset, innerSize_);
            }

            // Compute current row
            uint32_t gmOffsetCur = rowIdx * innerSize_;
            CopyIn(gmOffsetCur, innerSize_);
            Compute(innerSize_);
            CopyOut(gmOffsetCur, innerSize_);

            rowIdx = nextRow;
            if (rowIdx >= outerSize_) break;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> t0 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> t1 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2 = tmpQueue_.AllocTensor<float>();

        // Compute max over row
        AscendC::LocalTensor<float> maxScalar = scalarQueue_.AllocTensor<float>();
        // Use vector max reduction: iterate in chunks
        uint32_t chunk = 64;  // vector length
        uint32_t i = 0;
        AscendC::Muls(maxScalar, xLocal, 1.0f, 1);  // initialize with first element
        for (; i + chunk <= len; i += chunk) {
            AscendC::Max(t0, xLocal[i], maxScalar, chunk);
            AscendC::Muls(maxScalar, t0, 1.0f, 1);
        }
        if (i < len) {
            uint32_t rem = len - i;
            AscendC::Max(t0, xLocal[i], maxScalar, rem);
            AscendC::Muls(maxScalar, t0, 1.0f, 1);
        }

        // Compute exp(x - max)
        AscendC::Sub(t0, xLocal, maxScalar, len);
        AscendC::Exp(t1, t0, len);

        // Compute sum of exp
        AscendC::LocalTensor<float> sumScalar = scalarQueue_.AllocTensor<float>();
        i = 0;
        AscendC::Muls(sumScalar, t1, 1.0f, 1);  // initialize with first element
        for (; i + chunk <= len; i += chunk) {
            AscendC::Add(t2, sumScalar, t1[i], chunk);
            AscendC::Muls(sumScalar, t2, 1.0f, 1);
        }
        if (i < len) {
            uint32_t rem = len - i;
            AscendC::Add(t2, sumScalar, t1[i], rem);
            AscendC::Muls(sumScalar, t2, 1.0f, 1);
        }

        // Compute log_sum = log(sum) + max
        AscendC::Ln(t0, sumScalar, 1);
        AscendC::Add(t0, t0, maxScalar, 1);

        // y = x - log_sum
        AscendC::Sub(yLocal, xLocal, t0, len);

        // Free temps
        scalarQueue_.FreeTensor(maxScalar);
        scalarQueue_.FreeTensor(sumScalar);
        tmpQueue_.FreeTensor(t0);
        tmpQueue_.FreeTensor(t1);
        tmpQueue_.FreeTensor(t2);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outerSize_ {0};
    uint32_t innerSize_ {0};
    uint32_t tileLen_ {0};
    uint32_t dim_ {1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.outerSize,
            tiling_data.innerSize,
            tiling_data.tileLength,
            tiling_data.dim);
    op.Process();
}
"""
