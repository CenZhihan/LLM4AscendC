host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model:
// inQueueX: 2 * tileLen
// outQueueY: 2 * tileLen
// tmpQueue: 1 * tileLen
// total floats = 5 * tileLen => bytes = 20 * tileLen
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SoftmaxCustomTilingData tiling;
    const gert::Shape *xShape = &context->GetInputShape(0)->GetOriginShape();
    int32_t ndim = xShape->GetDimNum();
    // Assume ND input with at least 2 dims; softmax along dim=1
    uint32_t totalRows = 1;
    for (int32_t i = 0; i < ndim - 1; ++i) {
        totalRows *= static_cast<uint32_t>(xShape->GetDim(i));
    }
    uint32_t rowLen = static_cast<uint32_t>(xShape->GetDim(ndim - 1));
    uint32_t totalElements = totalRows * rowLen;

    // Tile length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 20; // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLen < tileLen) {
        tileLen = AlignDown(rowLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowLen;
    }

    // Block dimension: each block handles one or more rows
    uint32_t blockDim = totalRows;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_rowLen(rowLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLen,
                                uint32_t tileLen)
    {
        totalRows_ = totalRows;
        rowLen_ = rowLen;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * rowLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t baseIdx = row * rowLen_;
        float rowMax = -1e38f;  // sufficiently small
        float rowSum = 0.0f;

        uint32_t remaining = rowLen_;
        uint32_t offset = 0;

        // Pass 1: compute row max
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            float localMax = -1e38f;
            LoadTile(baseIdx + offset, curLen, localMax);
            if (localMax > rowMax) rowMax = localMax;
            remaining -= curLen;
            offset += curLen;
        }

        // Pass 2: compute sum of exp(x - rowMax)
        remaining = rowLen_;
        offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            float localSum = 0.0f;
            SumExpTile(baseIdx + offset, curLen, rowMax, localSum);
            rowSum += localSum;
            remaining -= curLen;
            offset += curLen;
        }

        // Pass 3: compute output y = exp(x - rowMax) / rowSum
        remaining = rowLen_;
        offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            ComputeOutputTile(baseIdx + offset, curLen, rowMax, rowSum);
            remaining -= curLen;
            offset += curLen;
        }
    }

    __aicore__ inline void LoadTile(uint32_t gmOffset, uint32_t len, float& localMax)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);

        // Compute max
        AscendC::LocalTensor<float> tmpMax = tmpQueue_.AllocTensor<float>();
        AscendC::ReduceMax(xLocal, xLocal, len);
        localMax = xLocal.GetValue(0);

        tmpQueue_.FreeTensor(tmpMax);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void SumExpTile(uint32_t gmOffset, uint32_t len, float rowMax, float& localSum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);

        AscendC::LocalTensor<float> expTmp = tmpQueue_.AllocTensor<float>();
        // Subtract rowMax
        AscendC::Subs(expTmp, xLocal, rowMax, len);
        // Compute exp
        AscendC::Exp(expTmp, expTmp, len);
        // Reduce sum
        AscendC::ReduceSum(expTmp, expTmp, len);
        localSum = expTmp.GetValue(0);

        tmpQueue_.FreeTensor(expTmp);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeOutputTile(uint32_t gmOffset, uint32_t len, float rowMax, float rowSum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

        // Compute exp(x - rowMax)
        AscendC::Subs(tmp, xLocal, rowMax, len);
        AscendC::Exp(yLocal, tmp, len);
        // Divide by rowSum
        AscendC::Muls(yLocal, yLocal, 1.0f / rowSum, len);

        // Store to global
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);

        tmpQueue_.FreeTensor(tmp);
        outQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.rowLen, tiling_data.tileLen);
    op.Process();
}
