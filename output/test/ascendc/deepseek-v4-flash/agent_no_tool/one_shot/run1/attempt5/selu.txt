host_operator_src = """
#include "selu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SeluCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Budget: input tile, output tile, one temp tile => 3 * tileLen * 4 bytes = 12 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SeluCustom : public OpDef {
public:
    explicit SeluCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SeluCustom);
} // namespace ops

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float SELU_ALPHA = 1.6732632423543772848170429916717f;
constexpr float SELU_SCALE = 1.0507009873554804934193349852946f;

class KernelSeluCustom {
public:
    __aicore__ inline KernelSeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // SELU: y = scale * (x > 0 ? x : alpha * (exp(x) - 1))
        // We compute exp(x) first, then subtract 1, multiply by alpha,
        // then use max(x,0) and min(0, alpha*(exp(x)-1)), combine and multiply by scale.
        // Simplified: compute t = alpha * (exp(x) - 1); then y = scale * (max(x,0) + min(x,0) * (alpha*exp(x)-alpha)/x ?? No, we'll do direct.
        // Better: compute expVal = exp(x); subOne = expVal - 1; mulAlpha = alpha * subOne; then where x>0 use x else mulAlpha; then multiply by scale.
        // Use pipeline: compute pos = max(x,0); compute negPart = exp(x) - 1; mul by alpha; then y = scale * (pos + (negPart if x<0 else 0)? Actually min(x,0) gives negative part, but we already computed full expression.
        // We'll use three steps: 
        // 1. t1 = exp(x)
        // 2. t2 = t1 - 1.0f
        // 3. t3 = SELU_ALPHA * t2   (for all elements, will be replaced for x>=0 later)
        // 4. t4 = max(x, 0.0f)  // positive part
        // 5. t5 = min(x, 0.0f)  // negative part (zero for positive, original for negative)
        // 6. t6 = (t5 != 0.0f) ? t3 : 0.0f? Actually we need to combine: for negative elements, use t3; for positive, use x.
        // Use AscendC::Sel (select) with condition x > 0.
        // We'll compute:
        //   pos = max(x,0);
        //   exp_x = exp(x);
        //   neg = SELU_ALPHA * (exp_x - 1.0f);
        //   y = SELU_SCALE * (pos + (x <= 0 ? neg : 0)); // but careful: for x>0, neg is computed but we ignore.
        // Better: compute y = SELU_SCALE * ( (x>0 ? x : neg) )
        // Use Sel: y = Sel(x > 0, x, neg) then multiply by scale.

        AscendC::LocalTensor<float> pos = tLocal; // reuse tLocal for pos
        AscendC::Maxs(pos, xLocal, 0.0f, len);  // pos = max(x,0)

        AscendC::LocalTensor<float> exp_x = yLocal; // reuse yLocal temporarily? No, we need yLocal later. We'll use tmpQueue for additional temp.
        // But we only have one tmpQueue. We'll reuse tLocal for exp, but we already used it for pos. So we need a separate buffer.
        // We have only two local tensors aside from input/output (tLocal). We can compute exp into tLocal and then use another local? Not enough.
        // Let's restructure: use yLocal as work buffer for exp, then compute neg into tLocal, then combine.
        // Steps:
        // 1. Compute exp_x into yLocal (overwrites output? but we will compute final output later into yLocal).
        // 2. Compute subOne = exp_x - 1.0f into tLocal (reuse after pos? we didn't compute pos yet).
        // 3. Compute neg = SELU_ALPHA * subOne into tLocal (overwrites)
        // 4. Compute pos = max(xLocal, 0.0f) into yLocal (overwrites exp_x)
        // 5. Compute cond = xLocal > 0.0f into ? we need a mask. Use AscendC::Sel with condition.
        // Actually AscendC::Sel takes three tensors: condition, true_val, false_val. We need a condition tensor. We can compute it into tLocal after we are done with neg? Or use another local from tmpQueue. Let's allocate a second temp from tmpQueue? tmpQueue has only 1 buffer of tileLen. We can't allocate another.
        // Instead, we can compute condition as a mask: compare x>0, store result in tLocal (as float, 1.0f for true, 0.0f for false). Then use Mul and Add to blend.
        // Approach:
        //   bool_mask = x > 0 (store as float 1.0 or 0.0 in tLocal)
        //   pos_only = bool_mask * x   (positive part)
        //   neg_only = (1-bool_mask) * (SELU_ALPHA * (exp(x)-1))
        //   y = SELU_SCALE * (pos_only + neg_only)
        // We'll compute step by step using tLocal and yLocal.

        // Step 1: compute exp(x) into yLocal
        AscendC::Exp(yLocal, xLocal, len);
        // Step 2: sub 1.0f, result in yLocal
        AscendC::Subs(yLocal, yLocal, 1.0f, len);
        // Step 3: multiply by alpha, result in yLocal (this is the negative branch value)
        AscendC::Muls(yLocal, yLocal, SELU_ALPHA, len);
        // Now yLocal holds neg_val = alpha*(exp(x)-1)
        // Step 4: compute bool_mask = (x > 0) into tLocal
        AscendC::Compare(tLocal, xLocal, 0.0f, AscendC::CompareMode::GREATER, len);  // returns 1.0f where true, 0.0f else
        // Step 5: compute pos_part = bool_mask * xLocal
        AscendC::Mul(pos, tLocal, xLocal, len); // pos = bool_mask * x (positive elements, zero elsewhere)
        // Step 6: compute one_minus_mask = 1 - bool_mask
        AscendC::Subs(tLocal, 1.0f, tLocal, len); // now tLocal = 1 - bool_mask
        // Step 7: compute neg_part = one_minus_mask * yLocal (the neg_val computed earlier)
        AscendC::Mul(tLocal, tLocal, yLocal, len); // tLocal now holds neg_part (only negative elements)
        // Step 8: combine pos_part + neg_part into yLocal
        AscendC::Add(yLocal, pos, tLocal, len);
        // Step 9: multiply by scale
        AscendC::Muls(yLocal, yLocal, SELU_SCALE, len);

        // Free temp resources
        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void selu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSeluCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
