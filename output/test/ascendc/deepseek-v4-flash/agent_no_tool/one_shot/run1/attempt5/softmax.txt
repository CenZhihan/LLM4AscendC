#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: 1 tile input + 1 tile output + 3 scalar tensors + 1 broadcast tensor + 1 exp tensor
//   ≈ 3 * tileLen floats => 12 * tileLen bytes → safe max 16384 with 192KB
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;      // 32 floats = 128 bytes
static constexpr uint32_t HARD_CAP_ELEMS = 4096;  // per-tile limit

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SoftmaxCustomTilingData tiling;
    const gert::Shape *xShape = &context->GetInputShape(0)->GetOriginShape();
    int32_t ndim = xShape->GetDimNum();
    uint32_t totalRows = 1;
    for (int32_t i = 0; i < ndim - 1; ++i) {
        totalRows *= static_cast<uint32_t>(xShape->GetDim(i));
    }
    uint32_t rowLen = static_cast<uint32_t>(xShape->GetDim(ndim - 1));
    uint32_t totalElements = totalRows * rowLen;

    // Compute safe tileLen: use at most 3*sizeof(float) * tileLen <= UB_BUDGET_BYTES
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));  // 3 floats per element peak usage
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLen < tileLen) {
        tileLen = AlignDown(rowLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowLen;
    }

    // BlockDim: parallelize over rows
    uint32_t blockDim = totalRows;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_rowLen(rowLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

// Proper kernel with correct API calls and three passes:

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLen,
                                uint32_t tileLen)
    {
        totalRows_ = totalRows;
        rowLen_ = rowLen;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * rowLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 4, tileLen_ * sizeof(float)); // enough for all temporaries
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t base = row * rowLen_;
        uint32_t numTiles = (rowLen_ + tileLen_ - 1) / tileLen_;

        // Scalar tensors for row max and sum
        AscendC::LocalTensor<float> rowMax = tmpQueue_.AllocTensor<float>(); // size 1
        AscendC::LocalTensor<float> rowSum = tmpQueue_.AllocTensor<float>(); // size 1

        // Pass 1: compute max
        rowMax.SetValue(0, -3.402823466e+38f); // -FLT_MAX
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = t * tileLen_;
            uint32_t curLen = (rowLen_ - offset > tileLen_) ? tileLen_ : (rowLen_ - offset);
            uint32_t gmOffset = base + offset;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], curLen);

            AscendC::LocalTensor<float> tileMax = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceMax(tileMax, xLocal, curLen, 0); // pattern 0 -> reduce all to scalar

            AscendC::Max(rowMax, rowMax, tileMax, 1);

            tmpQueue_.FreeTensor(tileMax);
            inQueueX_.FreeTensor(xLocal);
        }

        // Pass 2: compute sum of exp(x - max) and output normalized values directly
        rowSum.SetValue(0, 0.0f);
        // We'll reuse pass 2 to also store output? No, we need two passes because sum is needed for normalization.
        // So separate pass 2 and pass 3.

        // Pass 2: compute sum
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = t * tileLen_;
            uint32_t curLen = (rowLen_ - offset > tileLen_) ? tileLen_ : (rowLen_ - offset);
            uint32_t gmOffset = base + offset;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], curLen);

            AscendC::LocalTensor<float> rowMaxVec = tmpQueue_.AllocTensor<float>();
            AscendC::Broadcast(rowMaxVec, rowMax, 1, curLen, 0);

            AscendC::LocalTensor<float> expTmp = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(expTmp, xLocal, rowMaxVec, curLen);
            AscendC::Exp(expTmp, expTmp, curLen);

            AscendC::LocalTensor<float> tileSum = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(tileSum, expTmp, curLen, 0);
            AscendC::Add(rowSum, rowSum, tileSum, 1);

            tmpQueue_.FreeTensor(tileSum);
            tmpQueue_.FreeTensor(expTmp);
            tmpQueue_.FreeTensor(rowMaxVec);
            inQueueX_.FreeTensor(xLocal);
        }

        // Pass 3: output normalized values
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = t * tileLen_;
            uint32_t curLen = (rowLen_ - offset > tileLen_) ? tileLen_ : (rowLen_ - offset);
            uint32_t gmOffset = base + offset;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmOffset], curLen);

            AscendC::LocalTensor<float> rowMaxVec = tmpQueue_.AllocTensor<float>();
            AscendC::Broadcast(rowMaxVec, rowMax, 1, curLen, 0);
            AscendC::LocalTensor<float> rowSumVec = tmpQueue_.AllocTensor<float>();
            AscendC::Broadcast(rowSumVec, rowSum, 1, curLen, 0);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Sub(yLocal, xLocal, rowMaxVec, curLen);
            AscendC::Exp(yLocal, yLocal, curLen);
            AscendC::Div(yLocal, yLocal, rowSumVec, curLen);

            // Write back to global
            AscendC::DataCopy(yGm_[gmOffset], yLocal, curLen);

            outQueueY_.FreeTensor(yLocal);
            tmpQueue_.FreeTensor(rowSumVec);
            tmpQueue_.FreeTensor(rowMaxVec);
            inQueueX_.FreeTensor(xLocal);
        }

        tmpQueue_.FreeTensor(rowSum);
        tmpQueue_.FreeTensor(rowMax);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{

    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.rowLen, tiling_data.tileLen);
    op.Process();
}
