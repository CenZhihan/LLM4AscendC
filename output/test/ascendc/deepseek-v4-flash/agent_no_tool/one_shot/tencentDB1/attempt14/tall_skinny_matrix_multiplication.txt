    uint32_t tileK = (K > MAX_TILE_K) ? MAX_TILE_K : K;

    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);

        uint32_t numRowTiles = (M + tileM - 1) / tileM;
        uint32_t numColTiles = (N + tileN - 1) / tileN;

        if (blockDim < 1) blockDim = 1;

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatmul {
public:
    __aicore__ inline KernelTallSkinnyMatmul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; K_ = K; N_ = N;
        tileM_ = tileM; tileN_ = tileN; tileK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)a, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, M_ * N_);

        // Local buffers for tiles
        pipe_.InitBuffer(aBuf_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(bBuf_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(cBuf_, 1, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        uint32_t numRowTiles = (M_ + tileM_ - 1) / tileM_;
        uint32_t numColTiles = (N_ + tileN_ - 1) / tileN_;

        for (uint32_t id = blockIdx; id < numRowTiles * numColTiles; id += blockNum) {
            uint32_t tileRow = id / numColTiles;
            uint32_t tileCol = id % numColTiles;

            uint32_t curTileM = (tileRow == numRowTiles - 1) ? (M_ - tileRow * tileM_) : tileM_;
            uint32_t curTileN = (tileCol == numColTiles - 1) ? (N_ - tileCol * tileN_) : tileN_;

            // Allocate accumulator tensor and zero it
            LocalTensor<float> cTile = cBuf_.Get<float>();
            AscendC::DupTensor(cTile, 0.0f, tileM_ * tileN_);

            // Accumulate over K in tile chunks
            for (uint32_t kk = 0; kk < K_; kk += tileK_) {
                uint32_t curTileK = (kk + tileK_ <= K_) ? tileK_ : (K_ - kk);

                // Load A tile
                uint32_t aOffset = tileRow * tileM_ * K_ + kk;
                LocalTensor<float> aTile = aBuf_.AllocTensor<float>();
                AscendC::DataCopy(aTile, aGm_[aOffset], curTileM * curTileK);
                aBuf_.EnQue(aTile);

                // Load B tile
                uint32_t bOffset = kk * N_ + tileCol * tileN_;
                LocalTensor<float> bTile = bBuf_.AllocTensor<float>();
                AscendC::DataCopy(bTile, bGm_[bOffset], curTileK * curTileN);
                bBuf_.EnQue(bTile);

                aTile = aBuf_.DeQue<float>();
                bTile = bBuf_.DeQue<float>();

                // Matmul: C += A * B
                Matmul<MatmulType<AscendC::mm, float, float, float>> mm;
                mm.SetTensorA(aTile);
                mm.SetTensorB(bTile);
                mm.SetTensorC(cTile);
                mm.SetMMSize(curTileM, curTileN, curTileK);
                mm.IterateAll();

                aBuf_.FreeTensor(aTile);
                bBuf_.FreeTensor(bTile);
            }

            // Write back C tile
            uint32_t cOffset = tileRow * tileM_ * N_ + tileCol * tileN_;
            AscendC::DataCopy(cGm_[cOffset], cTile, curTileM * curTileN);
            cBuf_.FreeTensor(cTile);
        }
    }

private:
    TPipe pipe_;
    TBuf<TPosition::VECIN> aBuf_;
    TBuf<TPosition::VECIN> bBuf_;
    TBuf<TPosition::VECOUT> cBuf_;

    GlobalTensor<float> aGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {128};
    uint32_t tileN_ {32};
    uint32_t tileK_ {128};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatmul op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
