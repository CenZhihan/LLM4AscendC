kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dCustom {
public:
    __aicore__ inline KernelConvTransposed2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                const ConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustomTilingData &tiling)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        tiling_ = tiling;
        // Precompute strides in elements
        inStrideN_ = tiling.inChannels * tiling.inH * tiling.inW;
        inStrideC_ = tiling.inH * tiling.inW;
        inStrideH_ = tiling.inW;
        outStrideN_ = tiling.outChannels * tiling.outH * tiling.outW;
        outStrideC_ = tiling.outH * tiling.outW;
        outStrideH_ = tiling.outW;
        wStrideCIn_ = (tiling.outChannels / tiling.groups) * tiling.kernelH * tiling.kernelW; // per in-channel group
        wStrideCOut_ = tiling.kernelH * tiling.kernelW; // per out-channel (within group)
        wStrideH_ = tiling.kernelW;

        // UB buffer sizes per tile
        uint32_t tileOutSize = tiling.tileCOut * tiling.tileHOut * tiling.tileWOut;
        uint32_t tileWSize = (tiling.inChannels / tiling.groups) * tiling.kernelH * tiling.kernelW;
        // We will allocate ping-pong for output and input/weight temp
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileOutSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOutSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, std::max(tileOutSize, tileWSize) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t batch = tiling_.batch;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Each block processes one or more (b, ocTile, ohTile, owTile) tiles
        // Global tile index: linear across batch, outC, outH, outW tiles
        uint32_t numOutCTiles = CeilDiv(tiling_.outChannels, tiling_.tileCOut);
        uint32_t numHTiles = CeilDiv(tiling_.outH, tiling_.tileHOut);
        uint32_t numWTiles = CeilDiv(tiling_.outW, tiling_.tileWOut);
        uint32_t tilesPerBatch = numOutCTiles * numHTiles * numWTiles;
        uint32_t totalTiles = batch * tilesPerBatch;

        // Grid-stride over tiles
        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            uint32_t b = tileIdx / tilesPerBatch;
            uint32_t tileInBatch = tileIdx % tilesPerBatch;
            uint32_t ocTileIdx = tileInBatch / (numHTiles * numWTiles);
            uint32_t hTileIdx = (tileInBatch % (numHTiles * numWTiles)) / numWTiles;
            uint32_t wTileIdx = tileInBatch % numWTiles;

            uint32_t ocStart = ocTileIdx * tiling_.tileCOut;
            uint32_t ocEnd = std::min(ocStart + tiling_.tileCOut, tiling_.outChannels);
            uint32_t ohStart = hTileIdx * tiling_.tileHOut;
            uint32_t ohEnd = std::min(ohStart + tiling_.tileHOut, tiling_.outH);
            uint32_t owStart = wTileIdx * tiling_.tileWOut;
            uint32_t owEnd = std::min(owStart + tiling_.tileWOut, tiling_.outW);

            uint32_t tileCOut = ocEnd - ocStart;
            uint32_t tileHOut = ohEnd - ohStart;
            uint32_t tileWOut = owEnd - owStart;

            // For each output channel in this tile, we need to accumulate over input channel group
            AscendC::LocalTensor<float> outTile = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> zero = tmpQueue_.AllocTensor<float>();
            AscendC::Duplicate<float>(zero, 0.0f, tileCOut * tileHOut * tileWOut);
            AscendC::DataCopy(outTile, zero, tileCOut * tileHOut * tileWOut);
            tmpQueue_.FreeTensor(zero);

            // For each group (each input channel group projects to output channels)
            for (uint32_t g = 0; g < tiling_.groups; g++) {
                uint32_t outCPerGroup = tiling_.outChannels / tiling_.groups;
                uint32_t outCStartGroup = g * outCPerGroup;
                uint32_t outCEndGroup = outCStartGroup + outCPerGroup;

                uint32_t localOcStart = (ocStart >= outCStartGroup) ? ocStart : outCStartGroup;
                uint32_t localOcEnd   = (ocEnd <= outCEndGroup) ? ocEnd : outCEndGroup;
                if (localOcStart >= localOcEnd) continue;

                uint32_t localOcOffset = localOcStart - outCStartGroup; // within group

                // For each input channel in this group
                for (uint32_t ic = 0; ic < tiling_.tileCIn; ic++) {
                    uint32_t globalIc = g * tiling_.tileCIn + ic;

                    AscendC::LocalTensor<float> wLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::LocalTensor<float> inLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::LocalTensor<float> accLocal = outTile;

                    uint32_t wBase = globalIc * (outCPerGroup * tiling_.kernelH * tiling_.kernelW);
                    uint32_t wLoadSize = outCPerGroup * tiling_.kernelH * tiling_.kernelW;
                    AscendC::DataCopy(wLocal, wGm_[wBase], wLoadSize);

                    for (uint32_t oh = 0; oh < tileHOut; oh++) {
                        uint32_t globalOh = ohStart + oh;
                        for (uint32_t ow = 0; ow < tileWOut; ow++) {
                            uint32_t globalOw = owStart + ow;
                            for (uint32_t kh = 0; kh < tiling_.kernelH; kh++) {
                                for (uint32_t kw = 0; kw < tiling_.kernelW; kw++) {
                                    int32_t ih = (int32_t)globalOh + (int32_t)tiling_.paddingH - (int32_t)(kh * tiling_.dilationH);
                                    int32_t iw = (int32_t)globalOw + (int32_t)tiling_.paddingW - (int32_t)(kw * tiling_.dilationW);
                                    if (ih % (int32_t)tiling_.strideH != 0 || iw % (int32_t)tiling_.strideW != 0) continue;
                                    ih = ih / (int32_t)tiling_.strideH;
                                    iw = iw / (int32_t)tiling_.strideW;
                                    if (ih < 0 || (uint32_t)ih >= tiling_.inH || iw < 0 || (uint32_t)iw >= tiling_.inW) continue;

                                    float inputVal = xGm_[b * inStrideN_ + globalIc * inStrideC_ + ih * inStrideH_ + iw];

                                    for (uint32_t locOc = 0; locOc < tileCOut; locOc++) {
                                        uint32_t globalOc = ocStart + locOc;
                                        if (globalOc < outCStartGroup || globalOc >= outCEndGroup) continue;
                                        uint32_t groupLocalOc = globalOc - outCStartGroup;
                                        uint32_t wIdx = wBase + groupLocalOc * tiling_.kernelH * tiling_.kernelW + kh * tiling_.kernelW + kw;
                                        float wVal = wGm_[wIdx];
                                        uint32_t outIdx = locOc * tileHOut * tileWOut + oh * tileWOut + ow;
                                        // Accumulate (scalar, not vectorized)
                                        accLocal[outIdx] += inputVal * wVal;
                                    }
                                }
                            }
                        }
                    }
                    tmpQueue_.FreeTensor(wLocal);
                    tmpQueue_.FreeTensor(inLocal);
                }
            }

            // Write out the output tile to global memory
            AscendC::LocalTensor<float> outSlice = outTile;
            for (uint32_t locOc = 0; locOc < tileCOut; locOc++) {
                uint32_t globalOc = ocStart + locOc;
                uint32_t outOffset = b * outStrideN_ + globalOc * outStrideC_ + ohStart * outStrideH_ + owStart;
                uint32_t localOffset = locOc * tileHOut * tileWOut;
                outSlice.SetBase(outTile.GetBase() + localOffset);
                AscendC::DataCopy(yGm_[outOffset], outSlice, tileHOut * tileWOut);
            }
            outQueueY_.FreeTensor(outTile);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    ConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustomTilingData tiling_;

    uint32_t inStrideN_, inStrideC_, inStrideH_;
    uint32_t outStrideN_, outStrideC_, outStrideH_;
    uint32_t wStrideCIn_, wStrideCOut_, wStrideH_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dCustom op;
    op.Init(x, weight, y, tiling_data);
    op.Process();
}
