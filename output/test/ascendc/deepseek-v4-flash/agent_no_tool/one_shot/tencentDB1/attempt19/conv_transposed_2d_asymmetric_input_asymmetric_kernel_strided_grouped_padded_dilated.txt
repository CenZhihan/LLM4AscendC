#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dCustom {
public:
    __aicore__ inline KernelConvTransposed2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength, uint32_t blockDim,
                                uint32_t N, uint32_t Cout, uint32_t Hout, uint32_t Wout,
                                uint32_t Cin, uint32_t Hin, uint32_t Win,
                                uint32_t kH, uint32_t kW, uint32_t groups,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilH, uint32_t dilW)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N; Cout_ = Cout; Hout_ = Hout; Wout_ = Wout;
        Cin_ = Cin; Hin_ = Hin; Win_ = Win;
        kH_ = kH; kW_ = kW; groups_ = groups;
        strideH_ = strideH; strideW_ = strideW;
        padH_ = padH; padW_ = padW;
        dilH_ = dilH; dilW_ = dilW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * Cin_ * Hin_ * Win_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, Cin_ * Cout_ * kH_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * Cout_ * Hout_ * Wout_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // grid-stride over flattened output elements
        uint32_t stride = blockNum * tileLen_;
        uint32_t start = blockIdx * tileLen_;

        for (uint32_t base = start; base < totalLen_; base += stride) {
            uint32_t len = totalLen_ - base;
            if (len > tileLen_) len = tileLen_;

            for (uint32_t i = 0; i < len; i++) {
                uint32_t linear = base + i;
                uint32_t n = linear / (Cout_ * Hout_ * Wout_);
                uint32_t rem1 = linear % (Cout_ * Hout_ * Wout_);
                uint32_t go = rem1 / (Hout_ * Wout_);
                uint32_t rem2 = rem1 % (Hout_ * Wout_);
                uint32_t hout = rem2 / Wout_;
                uint32_t wout = rem2 % Wout_;

                // Determine group and per-group indices
                uint32_t group = go / (Cout_ / groups_);
                uint32_t cout_per_group = go - group * (Cout_ / groups_);

                float sum = 0.0f;
                uint32_t cin_per_group = Cin_ / groups_;
                for (uint32_t cg = 0; cg < cin_per_group; cg++) {
                    uint32_t gin = group * cin_per_group + cg;
                    for (uint32_t kh = 0; kh < kH_; kh++) {
                        for (uint32_t kw = 0; kw < kW_; kw++) {
                            int32_t hin = (static_cast<int32_t>(hout) + static_cast<int32_t>(padH_)
                                          - static_cast<int32_t>(dilH_ * kh));
                            if (hin < 0 || hin % static_cast<int32_t>(strideH_) != 0) continue;
                            hin /= static_cast<int32_t>(strideH_);
                            if (hin >= static_cast<int32_t>(Hin_)) continue;

                            int32_t win = (static_cast<int32_t>(wout) + static_cast<int32_t>(padW_)
                                          - static_cast<int32_t>(dilW_ * kw));
                            if (win < 0 || win % static_cast<int32_t>(strideW_) != 0) continue;
                            win /= static_cast<int32_t>(strideW_);
                            if (win >= static_cast<int32_t>(Win_)) continue;

                            uint32_t x_idx = n * (Cin_ * Hin_ * Win_) + gin * (Hin_ * Win_) + hin * Win_ + win;
                            uint32_t w_idx = gin * (Cout_ * kH_ * kW_) + go * (kH_ * kW_) + kh * kW_ + kw;
                            sum += xGm_[x_idx] * wGm_[w_idx];
                        }
                    }
                }
                yGm_[linear] = sum;
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_ {0}, Cout_ {0}, Hout_ {0}, Wout_ {0};
    uint32_t Cin_ {0}, Hin_ {0}, Win_ {0};
    uint32_t kH_ {0}, kW_ {0}, groups_ {1};
    uint32_t strideH_ {1}, strideW_ {1};
    uint32_t padH_ {0}, padW_ {0};
    uint32_t dilH_ {1}, dilW_ {1};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dCustom op;
    op.Init(x, weight, y,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim,
            tiling_data.N, tiling_data.C_out, tiling_data.H_out, tiling_data.W_out,
            tiling_data.C_in, tiling_data.H_in, tiling_data.W_in,
            tiling_data.kH, tiling_data.kW, tiling_data.groups,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilH, tiling_data.dilW);
    op.Process();
}
