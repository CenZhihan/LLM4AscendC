kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                uint32_t totalT, uint32_t B, uint32_t H, uint32_t I, uint32_t O, uint32_t K,
                                uint32_t batchChunk, uint32_t tileH, uint32_t tileO)
    {
        totalT_ = totalT;
        B_ = B;
        H_ = H;
        I_ = I;
        O_ = O;
        K_ = K;
        batchChunk_ = batchChunk;
        tileH_ = tileH;
        tileO_ = tileO;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalT_ * B_ * I_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B_ * H_);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H_ * K_);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O_ * H_);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalT_ * B_ * O_);

        pipe_.InitBuffer(inpBuf_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(hBuf_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(preBuf_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(ytBuf_, 1, O_ * sizeof(float));
        pipe_.InitBuffer(b_i2hBuf_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(b_h2oBuf_, 1, O_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, (H_ > O_ ? H_ : O_) * sizeof(float) > K_ * sizeof(float) ? (H_ > O_ ? H_ : O_) * sizeof(float) : K_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalT_ == 0 || B_ == 0 || H_ == 0 || I_ == 0 || O_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startBatch = blockIdx * batchChunk_;
        uint32_t endBatch = startBatch + batchChunk_;
        if (endBatch > B_) endBatch = B_;

        // Load biases to local
        AscendC::LocalTensor<float> b_i2hLocal = b_i2hBuf_.Get<float>();
        AscendC::LocalTensor<float> b_h2oLocal = b_h2oBuf_.Get<float>();
        AscendC::DataCopy(b_i2hLocal, b_i2hGm_[0], H_);
        AscendC::DataCopy(b_h2oLocal, b_h2oGm_[0], O_);

        for (uint32_t b = startBatch; b < endBatch; b++) {
            AscendC::LocalTensor<float> hLocal = hBuf_.Get<float>();
            AscendC::DataCopy(hLocal, h0Gm_[b * H_], H_);

            for (uint32_t t = 0; t < totalT_; t++) {
                // Concatenate x[t,b,:] and h to form inp (K)
                AscendC::LocalTensor<float> inpLocal = inpBuf_.Get<float>();
                const uint32_t xOffset = (t * B_ + b) * I_;
                AscendC::DataCopy(inpLocal, xGm_[xOffset], I_);
                AscendC::DataCopy(inpLocal[I_], hLocal, H_);

                // pre = inp * w_i2h.T + b_i2h
                AscendC::LocalTensor<float> preLocal = preBuf_.Get<float>();
                for (uint32_t j = 0; j < H_; j += tileH_) {
                    uint32_t actualTile = (j + tileH_ > H_) ? H_ - j : tileH_;
                    AscendC::SetZero(preLocal[j], actualTile);
                    for (uint32_t j0 = 0; j0 < actualTile; j0++) {
                        float sum = 0.0f;
                        for (uint32_t k = 0; k < K_; k++) {
                            float inpVal = inpLocal.GetValue(k);
                            float wVal = w_i2hGm_.GetValue((j + j0) * K_ + k);
                            sum += inpVal * wVal;
                        }
                        preLocal[j + j0] = sum + b_i2hLocal[j + j0];
                    }
                }

                // h = tanh(pre)
                AscendC::Tanh(hLocal, preLocal, H_);

                // y_t = h * w_h2o.T + b_h2o
                AscendC::LocalTensor<float> ytLocal = ytBuf_.Get<float>();
                for (uint32_t o = 0; o < O_; o += tileO_) {
                    uint32_t actualTile = (o + tileO_ > O_) ? O_ - o : tileO_;
                    AscendC::SetZero(ytLocal[o], actualTile);
                    for (uint32_t o0 = 0; o0 < actualTile; o0++) {
                        float sum = 0.0f;
                        for (uint32_t hIdx = 0; hIdx < H_; hIdx++) {
                            float hVal = hLocal.GetValue(hIdx);
                            float wVal = w_h2oGm_.GetValue((o + o0) * H_ + hIdx);
                            sum += hVal * wVal;
                        }
                        ytLocal[o + o0] = sum + b_h2oLocal[o + o0];
                    }
                }

                // Store y_t
                AscendC::DataCopy(yGm_[(t * B_ + b) * O_], ytLocal, O_);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> inpBuf_, hBuf_, preBuf_, ytBuf_, b_i2hBuf_, b_h2oBuf_, tmpBuf_;

    AscendC::GlobalTensor<float> xGm_, h0Gm_, w_i2hGm_, b_i2hGm_, w_h2oGm_, b_h2oGm_, yGm_;

    uint32_t totalT_, B_, H_, I_, O_, K_;
    uint32_t batchChunk_, tileH_, tileO_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.totalT, tiling_data.B, tiling_data.H, tiling_data.I, tiling_data.O, tiling_data.K,
            tiling_data.batchChunk, tiling_data.tileH, tiling_data.tileO);
    op.Process();
}
"""
