namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TripletMarginLossCustomTilingData tiling;
    auto shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalBatch = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t featureDim = static_cast<uint32_t>(shape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float margin = *attrs->GetAttrPointer<float>(0);

    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < 1) maxTile = 1;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = maxTile;
    if (featureDim < tileLen) tileLen = featureDim;
    if (tileLen == 0) tileLen = 1;

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalBatch(totalBatch);
    tiling.set_featureDim(featureDim);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);
    tiling.set_invBatch(1.0f / static_cast<float>(totalBatch));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
                                uint32_t totalBatch, uint32_t featureDim, uint32_t tileLen,
                                float margin, float invBatch) {
        totalBatch_ = totalBatch;
        featureDim_ = featureDim;
        tileLen_ = tileLen;
        margin_ = margin;
        invBatch_ = invBatch;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalBatch * featureDim);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalBatch * featureDim);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalBatch * featureDim);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inAnchorQ_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inPositiveQ_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inNegativeQ_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpDiffQ_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpSqQ_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpSumBuf_, sizeof(float));
        pipe_.InitBuffer(tmpScalarBuf_, sizeof(float));
        pipe_.InitBuffer(tmpAccumBuf_, sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalBatch_ == 0 || featureDim_ == 0) return;

        float totalLoss = 0.0f;

        for (uint32_t b = 0; b < totalBatch_; b++) {
            float distAp = 0.0f;
            float distAn = 0.0f;
            uint32_t remaining = featureDim_;
            uint32_t base = b * featureDim_;

            while (remaining > 0) {
                uint32_t curTile = (remaining < tileLen_) ? remaining : tileLen_;
                uint32_t offset = base + featureDim_ - remaining;

                // Copy tiles
                AscendC::LocalTensor<float> aLocal = inAnchorQ_.AllocTensor<float>();
                AscendC::DataCopy(aLocal, anchorGm_[offset], curTile);
                inAnchorQ_.EnQue(aLocal);

                AscendC::LocalTensor<float> pLocal = inPositiveQ_.AllocTensor<float>();
                AscendC::DataCopy(pLocal, positiveGm_[offset], curTile);
                inPositiveQ_.EnQue(pLocal);

                AscendC::LocalTensor<float> nLocal = inNegativeQ_.AllocTensor<float>();
                AscendC::DataCopy(nLocal, negativeGm_[offset], curTile);
                inNegativeQ_.EnQue(nLocal);

                // Dequeue and compute squared distances
                AscendC::LocalTensor<float> a = inAnchorQ_.DeQue<float>();
                AscendC::LocalTensor<float> p = inPositiveQ_.DeQue<float>();
                AscendC::LocalTensor<float> n = inNegativeQ_.DeQue<float>();
                AscendC::LocalTensor<float> diff = tmpDiffQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> sq = tmpSqQ_.AllocTensor<float>();

                // diff_ap = a - p
                AscendC::Sub(diff, a, p, curTile);
                AscendC::Mul(sq, diff, diff, curTile);
                // ReduceSum with proper pattern and repeatTimes=1
                AscendC::LocalTensor<float> sumLocal = tmpSumBuf_.Get<float>();
                AscendC::ReduceSum<AscendC::ReducePattern::SUM>(sumLocal, sq, curTile, 1);
                float partialAp = sumLocal.GetValue(0);
                distAp += partialAp;

                // diff_an = a - n
                AscendC::Sub(diff, a, n, curTile);
                AscendC::Mul(sq, diff, diff, curTile);
                AscendC::ReduceSum<AscendC::ReducePattern::SUM>(sumLocal, sq, curTile, 1);
                float partialAn = sumLocal.GetValue(0);
                distAn += partialAn;

                // Free tensors
                inAnchorQ_.FreeTensor(a);
                inPositiveQ_.FreeTensor(p);
                inNegativeQ_.FreeTensor(n);
                tmpDiffQ_.FreeTensor(diff);
                tmpSqQ_.FreeTensor(sq);

                remaining -= curTile;
            }

            // Compute sqrt and margin loss for this sample
            AscendC::LocalTensor<float> sqrtTensor = tmpScalarBuf_.Get<float>();
            sqrtTensor.SetValue(0, distAp);
            AscendC::Sqrt(sqrtTensor, sqrtTensor, 1);
            float sqrtAp = sqrtTensor.GetValue(0);

            sqrtTensor.SetValue(0, distAn);
            AscendC::Sqrt(sqrtTensor, sqrtTensor, 1);
            float sqrtAn = sqrtTensor.GetValue(0);

            float diffLoss = sqrtAp - sqrtAn + margin_;
            if (diffLoss > 0.0f) totalLoss += diffLoss;
        }

        // Average over batch using precomputed invBatch
        AscendC::LocalTensor<float> accumTensor = tmpAccumBuf_.Get<float>();
        accumTensor.SetValue(0, totalLoss);
        AscendC::LocalTensor<float> invBatchTensor = tmpScalarBuf_.Get<float>();
        invBatchTensor.SetValue(0, invBatch_);
        AscendC::Mul(accumTensor, accumTensor, invBatchTensor, 1);
        float avgLoss = accumTensor.GetValue(0);

        lossGm_.SetValue(0, avgLoss);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inAnchorQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inPositiveQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inNegativeQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDiffQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSqQ_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpSumBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpScalarBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpAccumBuf_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalBatch_ {0};
    uint32_t featureDim_ {0};
    uint32_t tileLen_ {0};
    float margin_ {0.0f};
    float invBatch_ {0.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalBatch, tiling_data.featureDim, tiling_data.tileLen,
            tiling_data.margin, tiling_data.invBatch);
    op.Process();
}
