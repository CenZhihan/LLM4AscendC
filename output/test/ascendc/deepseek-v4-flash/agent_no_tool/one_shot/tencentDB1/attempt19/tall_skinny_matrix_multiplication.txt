    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    uint32_t tileN = (N > MAX_TILE_N) ? MAX_TILE_N : N;
    uint32_t tileK = (K > MAX_TILE_K) ? MAX_TILE_K : K;

    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);

    uint32_t blockDim = 1;
    if (tileM > 0 && tileN > 0) {
        uint32_t numRowTiles = (M + tileM - 1) / tileM;
        uint32_t numColTiles = (N + tileN - 1) / tileN;
        blockDim = numRowTiles * numColTiles;
        if (blockDim > 24) blockDim = 24;
        if (blockDim < 1) blockDim = 1;

    tiling.set_blockDim(blockDim);
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatmul {
public:
    __aicore__ inline KernelTallSkinnyMatmul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; K_ = K; N_ = N;
        tileM_ = tileM; tileN_ = tileN; tileK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)a, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, M_ * N_);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        uint32_t numRowTiles = (M_ + tileM_ - 1) / tileM_;
        uint32_t numColTiles = (N_ + tileN_ - 1) / tileN_;

        for (uint32_t id = blockIdx; id < numRowTiles * numColTiles; id += blockNum) {
            uint32_t tileRow = id / numColTiles;
            uint32_t tileCol = id % numColTiles;

            uint32_t curTileM = (tileRow == numRowTiles - 1) ? (M_ - tileRow * tileM_) : tileM_;
            uint32_t curTileN = (tileCol == numColTiles - 1) ? (N_ - tileCol * tileN_) : tileN_;

            // Allocate output tile and initialize to zero
            pipe_.InitBuffer(cQue_, 1, curTileM * curTileN * sizeof(float));
            LocalTensor<float> cTile = cQue_.AllocTensor<float>();
            Duplicate(cTile, 0.0f, curTileM * curTileN);

            // Accumulate over K in tile chunks
            for (uint32_t kk = 0; kk < K_; kk += tileK_) {
                uint32_t curTileK = (kk + tileK_ <= K_) ? tileK_ : (K_ - kk);

                // Load A tile: contiguous in row-major
                uint32_t aOffset = tileRow * tileM_ * K_ + kk;
                pipe_.InitBuffer(aQue_, 1, curTileM * curTileK * sizeof(float));
                LocalTensor<float> aTile = aQue_.AllocTensor<float>();
                DataCopy(aTile, aGm_[aOffset], curTileM * curTileK);

                // Load B tile: contiguous in row-major (K rows, N columns)
                uint32_t bOffset = kk * N_ + tileCol * tileN_;
                pipe_.InitBuffer(bQue_, 1, curTileK * curTileN * sizeof(float));
                LocalTensor<float> bTile = bQue_.AllocTensor<float>();
                DataCopy(bTile, bGm_[bOffset], curTileK * curTileN);

                // Manual matrix multiply: cTile += aTile * bTile
                // Since tile sizes are small, use scalar loops (will be vectorized by compiler)
                for (uint32_t i = 0; i < curTileM; i++) {
                    for (uint32_t j = 0; j < curTileN; j++) {
                        float sum = 0.0f;
                        for (uint32_t k = 0; k < curTileK; k++) {
                            sum += aTile[i * curTileK + k] * bTile[k * curTileN + j];
                        }
                        cTile[i * curTileN + j] += sum;
                    }
                }

                aQue_.FreeTensor(aTile);
                bQue_.FreeTensor(bTile);
            }

            // Write out final C tile
            uint32_t cOffset = tileRow * tileM_ * N_ + tileCol * tileN_;
            DataCopy(cGm_[cOffset], cTile, curTileM * curTileN);
            cQue_.FreeTensor(cTile);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bQue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> cQue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
    uint32_t tileK_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatmul op;
    op.Init(a, b, c, tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
