kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                uint32_t totalT, uint32_t B, uint32_t H, uint32_t I,
                                uint32_t O, uint32_t K, uint32_t blockDim,
                                uint32_t batchChunk, uint32_t tileH, uint32_t tileO)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalT * B * I);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B * H);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H * K);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O * H);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalT * B * O);

        totalT_ = totalT;
        B_ = B;
        H_ = H;
        I_ = I;
        O_ = O;
        K_ = K;
        blockDim_ = blockDim;
        batchChunk_ = batchChunk;
        tileH_ = tileH;
        tileO_ = tileO;

        uint32_t maxRow = (tileH_ > tileO_) ? tileH_ : tileO_;
        pipe_.InitBuffer(inBufX_, BUFFER_NUM, I_ * sizeof(float));
        pipe_.InitBuffer(inBufH_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(inBufCat_, BUFFER_NUM, K_ * sizeof(float));
        pipe_.InitBuffer(inBufWih_, BUFFER_NUM, tileH_ * sizeof(float));
        pipe_.InitBuffer(inBufWho_, BUFFER_NUM, tileO_ * sizeof(float));
        pipe_.InitBuffer(outBufPre_, BUFFER_NUM, tileH_ * sizeof(float));
        pipe_.InitBuffer(outBufYt_, BUFFER_NUM, tileO_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, K_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blkIdx = AscendC::GetBlockIdx();
        uint32_t batchStart = blkIdx * batchChunk_;
        uint32_t batchEnd = batchStart + batchChunk_;
        if (batchEnd > B_) batchEnd = B_;
        if (batchStart >= B_) return;

        uint32_t b = batchStart;
        if (b >= B_) return;

        AscendC::LocalTensor<float> hLocal = inBufH_.AllocTensor<float>();
        AscendC::DataCopy(hLocal, h0Gm_[b * H_], H_);

        for (uint32_t t = 0; t < totalT_; t++) {
            AscendC::LocalTensor<float> xLocal = inBufX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[(t * B_ + b) * I_], I_);

            AscendC::LocalTensor<float> catLocal = inBufCat_.AllocTensor<float>();
            AscendC::DataCopy(catLocal, xLocal, I_);
            AscendC::DataCopy(catLocal[I_], hLocal, H_);

            AscendC::LocalTensor<float> preLocal = outBufPre_.AllocTensor<float>();
            for (uint32_t hOff = 0; hOff < H_; hOff += tileH_) {
                uint32_t hTile = (hOff + tileH_ > H_) ? (H_ - hOff) : tileH_;
                for (uint32_t hi = 0; hi < hTile; hi++) {
                    uint32_t rowIdx = hOff + hi;
                    float sum = 0.0f;
                    for (uint32_t k = 0; k < K_; k++) {
                        sum += catLocal(k) * w_i2hGm_[rowIdx * K_ + k];
                    }
                    sum += b_i2hGm_[rowIdx];
                    preLocal(hi) = sum;
                }
            }

            AscendC::Tanh(preLocal, preLocal, H_);
            AscendC::DataCopy(hLocal, preLocal, H_);

            AscendC::LocalTensor<float> ytLocal = outBufYt_.AllocTensor<float>();
            for (uint32_t oOff = 0; oOff < O_; oOff += tileO_) {
                uint32_t oTile = (oOff + tileO_ > O_) ? (O_ - oOff) : tileO_;
                for (uint32_t oi = 0; oi < oTile; oi++) {
                    uint32_t colIdx = oOff + oi;
                    float sum = 0.0f;
                    for (uint32_t h = 0; h < H_; h++) {
                        sum += hLocal(h) * w_h2oGm_[colIdx * H_ + h];
                    }
                    sum += b_h2oGm_[colIdx];
                    ytLocal(oi) = sum;
                }
            }

            AscendC::DataCopy(yGm_[(t * B_ + b) * O_], ytLocal, O_);

            inBufX_.FreeTensor(xLocal);
            inBufCat_.FreeTensor(catLocal);
            outBufPre_.FreeTensor(preLocal);
            outBufYt_.FreeTensor(ytLocal);
        }

        inBufH_.FreeTensor(hLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufH_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufCat_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufWih_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufWho_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outBufPre_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outBufYt_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> w_i2hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalT_, B_, H_, I_, O_, K_;
    uint32_t blockDim_, batchChunk_;
    uint32_t tileH_, tileO_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.totalT, tiling_data.B, tiling_data.H, tiling_data.I,
            tiling_data.O, tiling_data.K, tiling_data.blockDim,
            tiling_data.batchChunk, tiling_data.tileH, tiling_data.tileO);
    op.Process();
}
