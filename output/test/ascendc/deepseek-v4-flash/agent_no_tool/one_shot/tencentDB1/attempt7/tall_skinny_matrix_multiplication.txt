#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t MIN_TILE_ELEMS = 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    uint32_t M = a_shape->GetDim(0);
    uint32_t K = a_shape->GetDim(1);
    uint32_t N = b_shape->GetDim(1);

    // UB budget: A tile (tileM * K) + B tile (K * tileN) + C tile (tileM * tileN) + temp (tileN)
    // Approximate: 3 * tileM * K + tileM * tileN
    uint32_t maxFloats = UB_BUDGET_BYTES / sizeof(float) * 0.9f;
    uint32_t tileM = static_cast<uint32_t>(std::sqrt(static_cast<double>(maxFloats / 3)));
    uint32_t tileN = tileM;
    if (tileM > M) tileM = M;
    if (tileN > N) tileN = N;
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    tileN = AlignDown(tileN, ALIGN_ELEMS);
    if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;
    if (tileN < ALIGN_ELEMS) tileN = ALIGN_ELEMS;
    if (tileM > M) tileM = M;
    if (tileN > N) tileN = N;

    uint32_t numTilesRow = CeilDiv(M, tileM);
    uint32_t numTilesCol = CeilDiv(N, tileN);
    uint32_t totalTiles = numTilesRow * numTilesCol;
    uint32_t blockDim = totalTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);

}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileN)
    {
        M_ = M;
        K_ = K;
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;
        aGm_.SetGlobalBuffer((__gm__ float*)A, M * K);
        bGm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        cGm_.SetGlobalBuffer((__gm__ float*)C, M * N);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM * K * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, K * tileN * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileM * tileN * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileN * sizeof(float)); // for scaled B row
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t tilesRow = (M_ + tileM_ - 1) / tileM_;
        uint32_t tilesCol = (N_ + tileN_ - 1) / tileN_;
        uint32_t totalTiles = tilesRow * tilesCol;

        // Grid-stride loop over output tiles
        uint32_t tileIdx = blockIdx;
        while (tileIdx < totalTiles) {
            uint32_t tileRow = tileIdx / tilesCol;
            uint32_t tileCol = tileIdx % tilesCol;

            uint32_t rowStart = tileRow * tileM_;
            uint32_t colStart = tileCol * tileN_;
            uint32_t curTileM = (rowStart + tileM_ <= M_) ? tileM_ : (M_ - rowStart);
            uint32_t curTileN = (colStart + tileN_ <= N_) ? tileN_ : (N_ - colStart);

            ComputeTile(rowStart, colStart, curTileM, curTileN);

            tileIdx += blockNum;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t rowStart, uint32_t colStart,
                                       uint32_t tileM, uint32_t tileN)
    {
        AscendC::LocalTensor<float> aTile = inQueueA_.AllocTensor<float>();
        AscendC::LocalTensor<float> bTile = inQueueB_.AllocTensor<float>();
        AscendC::LocalTensor<float> cTile = outQueueC_.AllocTensor<float>();

        AscendC::SetZero(cTile, tileM * tileN);

        // Load A tile
        for (uint32_t r = 0; r < tileM; r++) {
            uint32_t globalRow = rowStart + r;
            AscendC::DataCopy(aTile + r * K_, aGm_[globalRow * K_], K_);
        }

        // Load B tile
        for (uint32_t k = 0; k < K_; k++) {
            uint32_t globalOffset = k * N_ + colStart;
            AscendC::DataCopy(bTile + k * tileN, bGm_[globalOffset], tileN);
        }

        // Compute C = A * B for this tile using a temporary buffer for scaling
        for (uint32_t i = 0; i < tileM; i++) {
            AscendC::LocalTensor<float> cRow = cTile + i * tileN;
            for (uint32_t k = 0; k < K_; k++) {
                float scalar = aTile[i * K_ + k];
                AscendC::LocalTensor<float> bRow = bTile + k * tileN;
                AscendC::LocalTensor<float> bScaled = tmpQueue_.AllocTensor<float>();
                AscendC::Muls(bScaled, bRow, scalar, tileN);
                AscendC::Add(cRow, cRow, bScaled, tileN);
                tmpQueue_.FreeTensor(bScaled);
            }
        }

        // Store C tile to global
        for (uint32_t r = 0; r < tileM; r++) {
            uint32_t globalRow = rowStart + r;
            AscendC::DataCopy(cGm_[globalRow * N_ + colStart], cTile + r * tileN, tileN);
        }

        outQueueC_.FreeTensor(cTile);
        inQueueB_.FreeTensor(bTile);
        inQueueA_.FreeTensor(aTile);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(a, b, c, tiling_data.M, tiling_data.K, tiling_data.N, tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
