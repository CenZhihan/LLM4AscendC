#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelBiasAddHardtanhMishCustom {
public:
    __aicore__ inline KernelBiasAddHardtanhMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueBias_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmp2Queue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Load x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base], curLen);

            // Load bias
            AscendC::LocalTensor<float> biasLocal = inQueueBias_.AllocTensor<float>();
            AscendC::DataCopy(biasLocal, biasGm_[base], curLen);

            // y = x + bias
            AscendC::Add(xLocal, xLocal, biasLocal, curLen);
            inQueueBias_.FreeTensor(biasLocal);

            // Hardtanh: clamp to [-1, 1]
            AscendC::Mins(xLocal, xLocal, 1.0f, curLen);
            AscendC::Maxs(xLocal, xLocal, -1.0f, curLen);

            // Mish: x * tanh(softplus(x))
            AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> softplusLocal = tmp2Queue_.AllocTensor<float>();
            AscendC::Exp(expLocal, xLocal, curLen);
            AscendC::Adds(softplusLocal, expLocal, 1.0f, curLen);
            AscendC::Log(softplusLocal, softplusLocal, curLen);
            AscendC::Tanh(softplusLocal, softplusLocal, curLen);
            AscendC::Mul(xLocal, xLocal, softplusLocal, curLen);
            tmpQueue_.FreeTensor(expLocal);
            tmp2Queue_.FreeTensor(softplusLocal);

            // Write output
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, xLocal, curLen);
            AscendC::DataCopy(yGm_[base], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);

            base += stride;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBias_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp2Queue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
};

class KernelGroupNormCustom {
public:
    __aicore__ inline KernelGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t numGroups,
                                uint32_t groupSize, float epsilon) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        numGroups_ = numGroups;
        groupSize_ = groupSize;
        epsilon_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, batchSize_ * numGroups_ * groupSize_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, batchSize_ * numGroups_ * groupSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueGamma_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueBeta_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmp2Queue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t totalGroups = batchSize_ * numGroups_;
        const uint32_t stride = blockNum;
        uint32_t gid = blockIdx;

        while (gid < totalGroups) {
            uint32_t sample = gid / numGroups_;
            uint32_t group = gid % numGroups_;
            uint32_t elemStart = sample * groupSize_ * numGroups_ + group * groupSize_;
            uint32_t elemLen = groupSize_;

            if (elemStart + elemLen > totalLen_) break;

            // Load group data
            AscendC::LocalTensor<float> groupLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(groupLocal, xGm_[elemStart], elemLen);

            // Compute mean
            AscendC::LocalTensor<float> meanLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(meanLocal, groupLocal, elemLen, 0);
            float meanVal = meanLocal.GetValue(0) / static_cast<float>(elemLen);
            FillTensor(meanLocal, 1, meanVal);

            // Compute variance
            AscendC::LocalTensor<float> diffLocal = tmp2Queue_.AllocTensor<float>();
            AscendC::Sub(diffLocal, groupLocal, meanLocal, elemLen);
            AscendC::Mul(diffLocal, diffLocal, diffLocal, elemLen);
            AscendC::ReduceSum(meanLocal, diffLocal, elemLen, 0);
            float varVal = meanLocal.GetValue(0) / static_cast<float>(elemLen) + epsilon_;
            float invStd = 1.0f / sqrt(varVal);

            // Normalize
            AscendC::Sub(diffLocal, groupLocal, meanLocal, elemLen);
            AscendC::Muls(diffLocal, diffLocal, invStd, elemLen);

            // Scale and shift
            AscendC::LocalTensor<float> gammaLocal = inQueueGamma_.AllocTensor<float>();
            AscendC::DataCopy(gammaLocal, gammaGm_[elemStart], elemLen);
            AscendC::LocalTensor<float> betaLocal = inQueueBeta_.AllocTensor<float>();
            AscendC::DataCopy(betaLocal, betaGm_[elemStart], elemLen);
            AscendC::Mul(diffLocal, diffLocal, gammaLocal, elemLen);
            AscendC::Add(diffLocal, diffLocal, betaLocal, elemLen);

            // Write output
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, diffLocal, elemLen);
            AscendC::DataCopy(yGm_[elemStart], yLocal, elemLen);

            inQueueX_.FreeTensor(groupLocal);
            tmpQueue_.FreeTensor(meanLocal);
            tmp2Queue_.FreeTensor(diffLocal);
            inQueueGamma_.FreeTensor(gammaLocal);
            inQueueBeta_.FreeTensor(betaLocal);
            outQueueY_.FreeTensor(yLocal);

            gid += stride;
        }
    }

private:
    __aicore__ inline void FillTensor(AscendC::LocalTensor<float>& t, uint32_t len, float val) {
        for (uint32_t i = 0; i < len; ++i) {
            t(i) = val;
        }
    }

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueGamma_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp2Queue_;
    // Note: inQueueBeta_ is not used with buffer; we load directly into tmp2Queue_ via alloc
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBeta_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t batchSize_{0};
    uint32_t numGroups_{0};
    uint32_t groupSize_{0};
    float epsilon_{1e-5f};
};

extern "C" __global__ __aicore__ void bias_add_hardtanh_mish_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(td, tiling);

    op.Process();
}

    KernelGroupNormCustom op;
    op.Init(x, gamma, beta, y, td.totalLength, td.tileLength, td.batchSize, td.numGroups, td.groupSize, td.epsilon);
    op.Process();
}

import torch
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

_FN1 = 'bias_add_hardtanh_mish_custom'
_FN2 = 'group_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, bias_shape, num_groups):
        super(ModelNew, self).__init__()
        self.num_groups = num_groups
        self._in_features = in_features
        self._out_features = out_features

    def forward(self, x, weight, lin_bias, bias, gn_gamma, gn_beta):
        # linear with bias
        x = F.linear(x, weight, lin_bias)
        # fuse bias add + hardtanh + mish
        fn1 = getattr(custom_ops_lib, _FN1)
        x = fn1(x, bias)
        # group norm
        fn2 = getattr(custom_ops_lib, _FN2)
        x = fn2(x, gn_gamma, gn_beta, num_groups=self.num_groups, epsilon=1e-5)
        return x
