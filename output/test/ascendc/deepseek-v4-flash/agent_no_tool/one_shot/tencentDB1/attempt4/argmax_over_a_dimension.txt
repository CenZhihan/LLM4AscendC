#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }

static constexpr uint32_t TILE_ELEMS = 4096;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    gert::Shape origin = inputShape->GetOriginShape();
    int32_t ndim = origin.GetDimNum();

    // For simplicity assume 3D; could extend to arbitrary dims with shape array.
    int32_t d0 = (ndim >= 1) ? origin.GetDim(0) : 1;
    int32_t d1 = (ndim >= 2) ? origin.GetDim(1) : 1;
    int32_t d2 = (ndim >= 3) ? origin.GetDim(2) : 1;
    int32_t totalInput = d0 * d1 * d2;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dimAttr = *attrs->GetAttrPointer<int64_t>(0);
    if (dimAttr < 0) dimAttr += ndim;

    // Compute output size
    uint32_t totalOutput = totalInput;
    if (dimAttr == 0) totalOutput = d1 * d2;
    else if (dimAttr == 1) totalOutput = d0 * d2;
    else if (dimAttr == 2) totalOutput = d0 * d1;
    // For ndim > 3, compute accordingly (simplified for this use case)

    uint32_t tileLen = (totalOutput < TILE_ELEMS) ? totalOutput : TILE_ELEMS;
    if (tileLen == 0) tileLen = 1;
    uint32_t numTiles = CeilDiv(totalOutput, tileLen);
    uint32_t blockDim = numTiles > 24 ? 24 : (numTiles < 8 ? 8 : numTiles);
    context->SetBlockDim(blockDim);

    tiling.set_totalOutputLength(totalOutput);
    tiling.set_tileLength(tileLen);
    tiling.set_d0(d0);
    tiling.set_d1(d1);
    tiling.set_d2(d2);
    tiling.set_dim(static_cast<int32_t>(dimAttr));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    int ndim = x_shape->GetDimNum();
    int64_t dim = *context->GetAttrs()->GetAttrPointer<int64_t>(0);
    if (dim < 0) dim += ndim;

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> dims;
    for (int i = 0; i < ndim; ++i) {
        if (i != dim) dims.push_back(x_shape->GetDim(i));
    }
    y_shape->SetDimNum(dims.size());
    for (size_t i = 0; i < dims.size(); ++i) {
        y_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    (void)context->GetInputDataType(0);
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
}  // namespace ge

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputLength,
                                uint32_t tileLength,
                                int32_t d0, int32_t d1, int32_t d2,
                                int32_t dim)
    {
        totalOutputLen_ = totalOutputLength;
        tileLen_ = tileLength;
        d0_ = d0;
        d1_ = d1;
        d2_ = d2;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, d0_ * d1_ * d2_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalOutputLen_);

        // Precompute strides for 3D
        stride_[0] = d1_ * d2_;
        stride_[1] = d2_;
        stride_[2] = 1;

        // Compute output shape and strides for the two non-reduced dimensions
        int idxOut = 0;
        for (int d = 0; d < 3; ++d) {
            if (d != dim_) {
                outShape_[idxOut] = (d == 0) ? d0_ : ((d == 1) ? d1_ : d2_);
                outStride_[idxOut] = stride_[d];
                idxOut++;
            }
        }
    }

    __aicore__ inline void Process()
    {
        if (totalOutputLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutputLen_) return;

        uint32_t end = base + tileLen_;
        if (end > totalOutputLen_) end = totalOutputLen_;

        int32_t reduceLen = (dim_ == 0) ? d0_ : ((dim_ == 1) ? d1_ : d2_);
        if (reduceLen <= 0) reduceLen = 1;

        for (uint32_t outIdx = base; outIdx < end; ++outIdx) {
            // Decompose linear output index into coordinates of non-reduced dims
            int32_t coord[2];
            uint32_t tmp = outIdx;
            for (int d = 1; d >= 0; --d) {
                coord[d] = static_cast<int32_t>(tmp % outShape_[d]);
                tmp /= outShape_[d];
            }

            float maxVal = -1e38f;
            int64_t argmax = 0;

            // Loop over reduction dimension
            for (int32_t r = 0; r < reduceLen; ++r) {
                int32_t linearIdx = 0;
                int coordIdx = 0;
                for (int d = 0; d < 3; ++d) {
                    if (d == dim_) {
                        linearIdx += r * stride_[d];
                    } else {
                        linearIdx += coord[coordIdx] * stride_[d];
                        coordIdx++;
                    }
                }
                float val = xGm_.GetValue(linearIdx);
                if (val > maxVal) {
                    maxVal = val;
                    argmax = static_cast<int64_t>(r);
                }
            }
            yGm_.SetValue(outIdx, argmax);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalOutputLen_ {0};
    uint32_t tileLen_ {0};
    int32_t d0_ {0}, d1_ {0}, d2_ {0}, dim_ {0};
    int32_t stride_[3];
    int32_t outShape_[2];
    int32_t outStride_[2];
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalOutputLength,
            tiling_data.tileLength,
            tiling_data.d0,
            tiling_data.d1,
            tiling_data.d2,
            tiling_data.dim);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    // Output shape: same as input but without the reduced dimension
    auto input_shape = self.sizes();
    std::vector<int64_t> out_shape;
    for (size_t i = 0; i < input_shape.size(); ++i) {
        if (static_cast<int64_t>(i) != dim)
            out_shape.push_back(input_shape[i]);
    }
    at::Tensor result = at::empty(out_shape, self.options().dtype(at::kLong));
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu,
