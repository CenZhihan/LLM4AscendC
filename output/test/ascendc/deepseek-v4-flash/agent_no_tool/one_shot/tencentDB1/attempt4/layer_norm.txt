#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // 192KB UB per core
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t MAX_TILE_ELEMS = 12288;        // 4 * 12288 = 49152 floats -> 196608B < 192KB

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LayerNormCustomTilingData tiling;
    const gert::Shape &xShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(xShape.GetShapeSize());
    uint32_t ndim = xShape.GetDimNum();
    uint32_t normalizedSize = 1;
    for (uint32_t i = 1; i < ndim; i++) {
        normalizedSize *= static_cast<uint32_t>(xShape.GetDim(i));
    }
    uint32_t batchSize = (ndim > 0) ? static_cast<uint32_t>(xShape.GetDim(0)) : 1;

    // tileLength: fit within UB (4 buffers: x, y, weight, bias)
    uint32_t maxTile = std::min(
        AlignDown(UB_BUDGET_BYTES / (4 * sizeof(float)), ALIGN_ELEMS),
        MAX_TILE_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLength = std::min(maxTile, normalizedSize);
    if (tileLength > ALIGN_ELEMS)
        tileLength = AlignDown(tileLength, ALIGN_ELEMS);

    // Block dimension: one block per sample
    uint32_t blockDim = std::min(batchSize, 24u);
    context->SetBlockDim(blockDim);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    float eps = (epsPtr != nullptr) ? *epsPtr : 1e-5f;

    tiling.set_totalLength(totalLength);
    tiling.set_normalizedSize(normalizedSize);
    tiling.set_batchSize(batchSize);
    tiling.set_tileLength(tileLength);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {

    gert::Shape *y_shape = context->GetOutputShape(0);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t normalizedSize,
                                uint32_t batchSize,
                                uint32_t tileLength,
                                float eps) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, normalizedSize);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, normalizedSize);
        totalLen_ = totalLength;
        normalizedSize_ = normalizedSize;
        batchSize_ = batchSize;
        tileLen_ = tileLength;
        eps_ = eps;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(weightBuf_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(biasBuf_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || normalizedSize_ == 0) return;

        // Load weight and bias once (must fit in tileLen, else tiling ensures)
        AscendC::LocalTensor<float> wLocal = weightBuf_.Get<float>();
        AscendC::DataCopy(wLocal, weightGm_, normalizedSize_);
        AscendC::LocalTensor<float> bLocal = biasBuf_.Get<float>();
        AscendC::DataCopy(bLocal, biasGm_, normalizedSize_);

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Process samples assigned to this block
        for (uint32_t s = blockIdx; s < batchSize_; s += blockNum) {
            uint32_t sampleOffset = s * normalizedSize_;
            ProcessSample(sampleOffset, wLocal, bLocal);
        }
    }

private:
    __aicore__ inline void ProcessSample(uint32_t startOffset,
                                         AscendC::LocalTensor<float> &wLocal,
                                         AscendC::LocalTensor<float> &bLocal) {
        // Phase 1: compute sum_x and sum_x2 (tiled)
        float sumX = 0.0f;
        float sumX2 = 0.0f;
        uint32_t remaining = normalizedSize_;
        uint32_t offset = startOffset;

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);

            // Compute partial sums using vector reduction
            AscendC::LocalTensor<float> tmp = tmpBuf_.Get<float>();
            AscendC::LocalTensor<float> sumBuf = tmpBuf_.Get<float>(); // reuse
            // Sum of elements
            AscendC::ReduceSum(sumBuf, xLocal, curLen);
            sumX += sumBuf(0);
            // Sum of squares
            AscendC::Mul(tmp, xLocal, xLocal, curLen);
            AscendC::ReduceSum(sumBuf, tmp, curLen);
            sumX2 += sumBuf(0);

            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }

        // Compute mean and variance
        float mean = sumX / static_cast<float>(normalizedSize_);
        float mean2 = sumX2 / static_cast<float>(normalizedSize_);
        float var = mean2 - mean * mean;
        float invStd = 1.0f / sqrtf(var + eps_);

        // Phase 2: normalize and write output
        remaining = normalizedSize_;
        offset = startOffset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // x tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // y = (x - mean) * invStd * w + b
            AscendC::LocalTensor<float> tmp = tmpBuf_.Get<float>();
            AscendC::Sub(tmp, xLocal, mean, curLen);
            AscendC::Muls(tmp, tmp, invStd, curLen);
            AscendC::Mul(tmp, tmp, wLocal, curLen);
            AscendC::Add(yLocal, tmp, bLocal, curLen);

            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);

            // Copy out
            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yOut, curLen);
            outQueueY_.FreeTensor(yOut);

            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> weightBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> biasBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalLen_ {0};
    uint32_t normalizedSize_ {0};
    uint32_t batchSize_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength,
            tiling_data.normalizedSize,
            tiling_data.batchSize,
            tiling_data.tileLength,
            tiling_data.eps);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

// Operator expects: x, optional weight, optional bias, eps attribute
at::Tensor layer_norm_impl_npu(const at::Tensor& self,
                               const c10::optional<at::Tensor>& weight_opt,
                               const c10::optional<at::Tensor>& bias_opt,
                               double eps) {
    at::Tensor result = at::empty_like(self);
    if (weight_opt.has_value() && bias_opt.has_value()) {
        EXEC_NPU_CMD(aclnnLayerNormCustom, self, weight_opt.value(), bias_opt.value(), eps, result);
    } else if (weight_opt.has_value()) {
        EXEC_NPU_CMD(aclnnLayerNormCustom, self, weight_opt.value(), at::Tensor(), eps, result);
    } else if (bias_opt.has_value()) {
        EXEC_NPU_CMD(aclnnLayerNormCustom, self, at::Tensor(), bias_opt.value(), eps, result);
    } else {
        EXEC_NPU_CMD(aclnnLayerNormCustom, self, at::Tensor(), at::Tensor(), eps, result);
    }
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    // Matches PyTorch's aten::layer_norm signature? We'll create a custom op.
    m.impl("layer_norm_custom", &layer_norm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("layer_norm_custom", &layer_norm_impl_npu,
          "Layer Normalization (LayerNormCustom on NPU)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        self.normalized_shape = normalized_shape
        self.weight = torch.nn.Parameter(torch.ones(normalized_shape))
        self.bias = torch.nn.Parameter(torch.zeros(normalized_shape))
        self.eps = 1e-5

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        # Squeeze batch dimension for weight/bias? weight/bias shape = normalized_shape
        weight = self.weight.view(1, *self.normalized_shape).expand(x.shape[0], *self.normalized_shape)
        bias = self.bias.view(1, *self.normalized_shape).expand(x.shape[0], *self.normalized_shape)
        return fn(x, weight, bias, self.eps)
'''
