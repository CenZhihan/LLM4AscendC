#include "vanilla_rnn_hidden_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    VanillaRnnHiddenCustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::StorageShape *h0_shape = context->GetInputShape(1);
    const gert::StorageShape *w_i2h_shape = context->GetInputShape(2);
    const gert::StorageShape *w_h2o_shape = context->GetInputShape(4);

    uint32_t totalT = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t B     = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t I     = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    uint32_t H     = static_cast<uint32_t>(h0_shape->GetOriginShape().GetDim(1));
    uint32_t K     = static_cast<uint32_t>(w_i2h_shape->GetOriginShape().GetDim(1));
    uint32_t O     = static_cast<uint32_t>(w_h2o_shape->GetOriginShape().GetDim(0));

    uint32_t tileH = 32;
    uint32_t tileO = 64;

    uint32_t blockDim = B;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;
    uint32_t batchChunk = (B + blockDim - 1) / blockDim;
    if (batchChunk < 1) batchChunk = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalT(totalT);
    tiling.set_B(B);
    tiling.set_H(H);
    tiling.set_I(I);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_blockDim(blockDim);
    tiling.set_batchChunk(batchChunk);
    tiling.set_tileH(tileH);
    tiling.set_tileO(tileO);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::StorageShape *w_h2o_shape = context->GetInputShape(4);
    int64_t T = x_shape->GetOriginShape().GetDim(0);
    int64_t B = x_shape->GetOriginShape().GetDim(1);
    int64_t O = w_h2o_shape->GetOriginShape().GetDim(0);

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDim(0, T);
    y_shape->SetDim(1, B);
    y_shape->SetDim(2, O);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge
