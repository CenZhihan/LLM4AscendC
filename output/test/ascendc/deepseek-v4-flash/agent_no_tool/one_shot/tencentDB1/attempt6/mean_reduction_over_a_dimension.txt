namespace optiling {

static constexpr uint32_t MAX_RANK = 8;
static constexpr uint32_t ALIGN_ELEMS = 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;
    const gert::Shape& input_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rank = static_cast<uint32_t>(input_shape.GetDimNum());
    if (rank > MAX_RANK) return ge::GRAPH_FAILED;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim_raw = attrs->GetAttrPointer<int64_t>(0)[0];
    uint32_t dim = (dim_raw < 0) ? static_cast<uint32_t>(dim_raw + rank) : static_cast<uint32_t>(dim_raw);
    if (dim >= rank) dim = rank - 1;

    const uint32_t shapeVals[MAX_RANK] = {
        static_cast<uint32_t>(input_shape.GetDim(0)),
        static_cast<uint32_t>(input_shape.GetDim(1)),
        static_cast<uint32_t>(input_shape.GetDim(2)),
        static_cast<uint32_t>(input_shape.GetDim(3)),
        static_cast<uint32_t>(input_shape.GetDim(4)),
        static_cast<uint32_t>(input_shape.GetDim(5)),
        static_cast<uint32_t>(input_shape.GetDim(6)),
        static_cast<uint32_t>(input_shape.GetDim(7))
    };
    tiling.set_shape0(shapeVals[0]);
    tiling.set_shape1(shapeVals[1]);
    tiling.set_shape2(shapeVals[2]);
    tiling.set_shape3(shapeVals[3]);
    tiling.set_shape4(shapeVals[4]);
    tiling.set_shape5(shapeVals[5]);
    tiling.set_shape6(shapeVals[6]);
    tiling.set_shape7(shapeVals[7]);

    uint32_t totalSize = 1;
    uint32_t strideDim = 1;
    for (uint32_t i = dim + 1; i < rank; ++i) strideDim *= shapeVals[i];
    uint32_t outerSize = 1;
    for (uint32_t i = 0; i < rank; ++i) {
        totalSize *= shapeVals[i];
        if (i != dim) outerSize *= shapeVals[i];
    }
    uint32_t reduceLen = shapeVals[dim];

    float invReduceLen = (reduceLen > 0) ? (1.0f / static_cast<float>(reduceLen)) : 0.0f;

    // Process all outer elements in a single tile on a single core for simplicity.
    // This avoids fragmentation and ensures correctness.
    uint32_t tileLen = outerSize;
    uint32_t blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_rank(rank);
    tiling.set_dim(dim);
    tiling.set_totalLength(totalSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_outerSize(outerSize);
    tiling.set_reduceLen(reduceLen);
    tiling.set_invReduceLen(invReduceLen);
    tiling.set_strideDim(strideDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    currentWorkspace[0] = 0;

#include "kernel_operator.h"

constexpr int32_t MAX_RANK = 8;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rank, uint32_t dim,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim,
                                uint32_t outerSize, uint32_t reduceLen,
                                float invReduceLen,
                                uint32_t strideDim,
                                uint32_t shape0, uint32_t shape1, uint32_t shape2,
                                uint32_t shape3, uint32_t shape4, uint32_t shape5,
                                uint32_t shape6, uint32_t shape7)
    {
        rank_ = rank;
        dim_ = dim;
        outerSize_ = outerSize;
        reduceLen_ = reduceLen;
        invReduceLen_ = invReduceLen;
        strideDim_ = strideDim;
        tileLen_ = tileLength;
        uint32_t shape[MAX_RANK] = {shape0, shape1, shape2, shape3, shape4, shape5, shape6, shape7};
        for (uint32_t i = 0; i < rank; ++i) inputShape_[i] = shape[i];

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outerSize);

        uint32_t stride = 1;
        for (int32_t i = rank_ - 1; i >= 0; --i) {
            inputStride_[i] = stride;
            stride *= inputShape_[i];
        }

        uint32_t outRank = rank_ - 1;
        uint32_t outIdx = 0;
        for (uint32_t i = 0; i < rank_; ++i) {
            if (i != dim_) {
                outputShape_[outIdx] = inputShape_[i];
                outIdx++;
            }
        }
        uint32_t outStride = 1;
        for (int32_t i = outRank - 1; i >= 0; --i) {
            outputStride_[i] = outStride;
            outStride *= outputShape_[i];
        }

        blockOffset_ = AscendC::GetBlockIdx() * tileLen_;
        if (blockOffset_ >= outerSize_) localTileLen_ = 0;
        else {
            localTileLen_ = tileLen_;
            if (blockOffset_ + tileLen_ > outerSize_) localTileLen_ = outerSize_ - blockOffset_;
        }
    }

    __aicore__ inline void Process()
    {
        if (localTileLen_ == 0) return;

        for (uint32_t idx = 0; idx < localTileLen_; ++idx) {
            uint32_t outerIdx = blockOffset_ + idx;

            uint32_t outCoord[MAX_RANK];
            uint32_t temp = outerIdx;
            uint32_t outRank = rank_ - 1;
            for (uint32_t i = 0; i < outRank; ++i) {
                outCoord[i] = temp / outputStride_[i];
                temp = temp % outputStride_[i];
            }

            uint32_t base = 0;
            uint32_t outPos = 0;
            for (uint32_t i = 0; i < rank_; ++i) {
                if (i == dim_) continue;
                base += outCoord[outPos] * inputStride_[i];
                outPos++;
            }

            float sum = 0.0f;
            for (uint32_t r = 0; r < reduceLen_; ++r) {
                sum += xGm_.GetValue(base + r * strideDim_);
            }
            yGm_.SetValue(outerIdx, sum * invReduceLen_);
        }
    }

private:
    uint32_t rank_;
    uint32_t dim_;
    uint32_t outerSize_;
    uint32_t reduceLen_;
    float invReduceLen_;
    uint32_t strideDim_;
    uint32_t tileLen_;
    uint32_t blockOffset_;
    uint32_t localTileLen_;

    uint32_t inputShape_[MAX_RANK];
    uint32_t inputStride_[MAX_RANK];
    uint32_t outputShape_[MAX_RANK];
    uint32_t outputStride_[MAX_RANK];

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.rank, tiling_data.dim,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.blockDim,
            tiling_data.outerSize, tiling_data.reduceLen,
            tiling_data.invReduceLen,
            tiling_data.strideDim,
            tiling_data.shape0, tiling_data.shape1, tiling_data.shape2,
            tiling_data.shape3, tiling_data.shape4, tiling_data.shape5,
            tiling_data.shape6, tiling_data.shape7);
    op.Process();
}
