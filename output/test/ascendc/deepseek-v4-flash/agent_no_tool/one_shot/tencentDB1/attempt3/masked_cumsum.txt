host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model per block:
//   xLocal:         1 * tileLength floats  (4 bytes each)
//   maskLocal:      1 * tileLength uint8    (1 byte each)
//   yLocal:         1 * tileLength floats   (4 bytes each)
//   productLocal:   1 * tileLength floats   (4 bytes each)
//   maskFloatLocal: 1 * tileLength floats   (4 bytes each)
//   scratchLocal:   1 * tileLength floats   (4 bytes each)  // cumsum temporary
// Total bytes = (4+1+4+4+4+4)*tileLength = 21 * tileLength
// Conservative: 200 KB UB (per core) => maxTile ~ 200*1024/21 ≈ 9763
static constexpr uint32_t UB_BUDGET_BYTES = 200 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaskedCumsumCustomTilingData tiling;

    auto shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(shape.GetShapeSize());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dimVal = *attrs->GetAttrPointer<int32_t>(0);

    // Compute dimSize and outerSize for a generic ND tensor
    int32_t ndim = shape.GetDimNum();
    // Normalize dim
    if (dimVal < 0) dimVal += ndim;
    if (dimVal < 0) dimVal = 0;
    if (dimVal >= ndim) dimVal = ndim - 1;

    uint32_t dimSize = 1;
    uint32_t outerSize = 1;
    for (int32_t i = 0; i < ndim; ++i) {
        uint32_t d = static_cast<uint32_t>(shape.GetDim(i));
        if (i < dimVal) {
            outerSize *= d;
        } else if (i == dimVal) {
            dimSize = d;
        }
    }

    // tiling length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 21; // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (dimSize < tileLen) {
        uint32_t aligned = AlignDown(dimSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : dimSize;
    }

    // blockDim: distribute rows across blocks, cap at 24
    uint32_t numRows = outerSize;
    uint32_t blockDim = numRows;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_dimSize(dimSize);
    tiling.set_outerSize(outerSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    context->GetWorkspaceSizes(1)[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MaskedCumsumCustom : public OpDef {
public:
    explicit MaskedCumsumCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("mask")
            .ParamType(REQUIRED)
            .DataType({ge::DT_BOOL})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(MaskedCumsumCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t dimSize,
                                uint32_t outerSize, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        dimSize_  = dimSize;
        outerSize_ = outerSize;
        tileLen_  = tileLength;
        rowStride_ = dimSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ uint8_t*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < outerSize_; row += blockNum) {
            uint32_t rowBase = row * rowStride_;
            float runningSum = 0.0f;

            for (uint32_t col = 0; col < dimSize_; col += tileLen_) {
                uint32_t curLen = tileLen_;
                if (col + curLen > dimSize_) curLen = dimSize_ - col;
                uint32_t gmOffset = rowBase + col;

                // Compute masked cumulative sum element by element using global tensor access
                for (uint32_t i = 0; i < curLen; ++i) {
                    float xVal = xGm_[gmOffset + i];
                    uint8_t maskVal = maskGm_[gmOffset + i];
                    float product = xVal * (maskVal ? 1.0f : 0.0f);
                    float cumVal = product + runningSum;
                    yGm_[gmOffset + i] = cumVal;
                    runningSum = cumVal;
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<uint8_t> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t dimSize_  {0};
    uint32_t outerSize_ {0};
    uint32_t tileLen_  {0};
    uint32_t rowStride_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                                            GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.totalLength, tiling_data.dimSize,
            tiling_data.outerSize, tiling_data.tileLength);
    op.Process();
}
