namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t MIN_TILE_ELEMS = 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::StorageShape *a_shape = context->GetInputShape(0);
    const gert::StorageShape *b_shape = context->GetInputShape(1);
    uint32_t M = a_shape->GetOriginShape().GetDim(0);
    uint32_t K = a_shape->GetOriginShape().GetDim(1);
    uint32_t N = b_shape->GetOriginShape().GetDim(1);

    uint32_t maxBytes = UB_BUDGET_BYTES * 0.8;
    uint32_t maxFloats = maxBytes / sizeof(float);
    uint32_t perTileFloats = maxFloats / 3;
    uint32_t tileM = static_cast<uint32_t>(std::sqrt(static_cast<double>(perTileFloats)));
    uint32_t tileN = tileM;
    if (tileM > M) tileM = M;
    if (tileN > N) tileN = N;
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    tileN = AlignDown(tileN, ALIGN_ELEMS);
    if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;
    if (tileN < ALIGN_ELEMS) tileN = ALIGN_ELEMS;
    if (tileM > M) tileM = M;
    if (tileN > N) tileN = N;

    uint32_t numTilesRow = CeilDiv(M, tileM);
    uint32_t numTilesCol = CeilDiv(N, tileN);
    uint32_t totalTiles = numTilesRow * numTilesCol;
    uint32_t blockDim = totalTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileN)
    {
        M_ = M;
        K_ = K;
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;
        aGm_.SetGlobalBuffer((__gm__ float*)A, M * K);
        bGm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        cGm_.SetGlobalBuffer((__gm__ float*)C, M * N);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM * K * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, K * tileN * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileM * tileN * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t tilesRow = (M_ + tileM_ - 1) / tileM_;
        uint32_t tilesCol = (N_ + tileN_ - 1) / tileN_;
        uint32_t totalTiles = tilesRow * tilesCol;

        // Grid-stride loop over output tiles
        uint32_t tileIdx = blockIdx;
        while (tileIdx < totalTiles) {
            uint32_t tileRow = tileIdx / tilesCol;
            uint32_t tileCol = tileIdx % tilesCol;

            uint32_t rowStart = tileRow * tileM_;
            uint32_t colStart = tileCol * tileN_;
            uint32_t curTileM = (rowStart + tileM_ <= M_) ? tileM_ : (M_ - rowStart);
            uint32_t curTileN = (colStart + tileN_ <= N_) ? tileN_ : (N_ - colStart);

            // Compute C[rowStart:rowStart+curTileM, colStart:colStart+curTileN] = A[rowStart:rowStart+curTileM, :] * B[:, colStart:colStart+curTileN]
            ComputeTile(rowStart, colStart, curTileM, curTileN);

            tileIdx += blockNum;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t rowStart, uint32_t colStart,
                                       uint32_t tileM, uint32_t tileN)
    {
        // We need to accumulate along the K dimension.
        // Load A tile (tileM x K) and B tile (K x tileN) into local memory, then do outer product.
        // For simplicity, use a naive loop over K with dot products.
        // Since K is small (e.g., 32), this is acceptable.

        // Allocate local tensors for A tile and B tile and accumulate C tile
        AscendC::LocalTensor<float> aTile = inQueueA_.AllocTensor<float>();
        AscendC::LocalTensor<float> bTile = inQueueB_.AllocTensor<float>();
        AscendC::LocalTensor<float> cTile = outQueueC_.AllocTensor<float>();

        // Initialize cTile to zero
        AscendC::SetZero(cTile, tileM * tileN);

        // Load full A tile (tileM rows of A, each of length K)
        for (uint32_t r = 0; r < tileM; r++) {
            uint32_t globalRow = rowStart + r;
            AscendC::DataCopy(aTile + r * K_, aGm_[globalRow * K_], K_);
        }

        // Load full B tile (K columns of B, each of length tileN)
        for (uint32_t c = 0; c < tileN; c++) {
            uint32_t globalCol = colStart + c;
            for (uint32_t k = 0; k < K_; k++) {
                // B is stored column-major? Actually B is row-major, so B[k][globalCol] is at offset k*N_ + globalCol
                // We need to load a scalar, but we can do vector copy of a row of B? B row is K x N, so each row of B is N elements.
                // Better: Load a row of B (length N_) into a temporary, then extract needed column.
                // Since K is small, we can just load the whole K x tileN block as a flat array.
                // For simplicity, we load the needed B elements column by column.
                AscendC::LocalTensor<float> temp = pipe_.TmpBuf->AllocTensor<float>(N_);
                AscendC::DataCopy(temp, bGm_[k * N_], N_); // load one row of B
                bTile[c * K_ + k] = temp[globalCol];
                pipe_.TmpBuf->FreeTensor(temp);
            }
        }

        // Now we have aTile [tileM x K] and bTile [K x tileN] (bTile stored as K rows, tileN columns, row-major)
        // Compute C = aTile * bTile using matrix multiply
        // Use AscendC's MatMul? Not available from user kernel? We'll implement with inner loops.
        for (uint32_t r = 0; r < tileM; r++) {
            for (uint32_t c = 0; c < tileN; c++) {
                float sum = 0.0f;
                for (uint32_t k = 0; k < K_; k++) {
                    sum += aTile[r * K_ + k] * bTile[k * tileN + c];
                }
                cTile[r * tileN + c] = sum;
            }
        }

        // Write back C tile to global memory
        for (uint32_t r = 0; r < tileM; r++) {
            uint32_t globalRow = rowStart + r;
            AscendC::DataCopy(cGm_[globalRow * N_ + colStart], cTile + r * tileN, tileN);
        }

        outQueueC_.EnQue(cTile);
        inQueueA_.FreeTensor(aTile);
        inQueueB_.FreeTensor(bTile);
        outQueueC_.DeQue<float>(); // discard
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(A, B, C, tiling_data.M, tiling_data.K, tiling_data.N, tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
