#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget estimation: conservative, 192KB
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;
    const gert::Shape* anchorShape = context->GetInputShape(0);
    const uint32_t batchSize = anchorShape->GetDim(0);
    const uint32_t featureDim = anchorShape->GetDim(1);

    const gert::RuntimeAttrs* attrs = context->GetAttrs();
    const float* margin = attrs->GetAttrPointer<float>(0);

    // Simple blockDim based on batchSize
    uint32_t blockDim = 8;
    if (batchSize < 8) blockDim = batchSize;
    else if (batchSize > 64) blockDim = 16;
    else blockDim = batchSize > 32 ? 16 : 8;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_featureDim(featureDim);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *anchor_shape = context->GetInputShape(0);
    const uint32_t batchSize = anchor_shape->GetDim(0);
    gert::Shape *loss_shape = context->GetOutputShape(0);
    loss_shape->SetDimNum(1);
    loss_shape->SetDim(0, batchSize);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class TripletMarginLossCustom : public OpDef {
public:
    explicit TripletMarginLossCustom(const char* name) : OpDef(name)
    {
        this->Input("anchor")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("positive")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("negative")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("loss_per_sample")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
