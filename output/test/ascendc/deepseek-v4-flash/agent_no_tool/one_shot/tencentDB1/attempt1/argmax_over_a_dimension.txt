namespace optiling {

static constexpr uint32_t TILE_SIZE = 4096;      // number of output elements per iteration (tuned)

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::StorageDesc *inputDesc = context->GetInputDesc(0);
    const gert::Shape *inputShape = context->GetInputShape(0);
    const ge::Shape &originShape = inputShape->GetOriginShape();
    int32_t ndim = originShape.GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dimAttr = *attrs->GetAttrPointer<int64_t>(0);
    if (dimAttr < 0) dimAttr += ndim; // support negative dim

    // Assume 3D for simplicity; but we handle generic by passing d0,d1,d2 and dim
    uint32_t d0 = 1, d1 = 1, d2 = 1;
    if (ndim >= 1) d0 = originShape.GetDim(0);
    if (ndim >= 2) d1 = originShape.GetDim(1);
    if (ndim >= 3) d2 = originShape.GetDim(2);
    // For higher ndim, we could generalize; but for the model it's 3D.

    uint32_t totalInputLength = d0 * d1 * d2;
    uint32_t totalOutputLength = d0 * (dimAttr == 0 ? d1 * d2 : (dimAttr == 1 ? d0 * d2 : d0 * d1));
    // Actually for dim=1 reduce d1: output = d0 * d2
    // For dim=0 reduce d0: output = d1 * d2
    // For dim=2 reduce d2: output = d0 * d1
    // We'll compute accordingly.
    uint32_t outLen;
    if (dimAttr == 0) outLen = d1 * d2;
    else if (dimAttr == 1) outLen = d0 * d2;
    else if (dimAttr == 2) outLen = d0 * d1;
    else outLen = totalInputLength; // fallback

    // Tiling over output elements
    uint32_t tileLen = TILE_SIZE;
    if (tileLen > outLen) tileLen = outLen;
    if (tileLen == 0) tileLen = 1;

    uint32_t numTiles = CeilDiv(outLen, tileLen);
    uint32_t blockDim = numTiles > 24 ? 24 : (numTiles < 8 ? 8 : numTiles);
    context->SetBlockDim(blockDim);

    tiling.set_totalInputLength(totalInputLength);
    tiling.set_totalOutputLength(outLen);
    tiling.set_tileLength(tileLen);
    tiling.set_d0(d0);
    tiling.set_d1(d1);
    tiling.set_d2(d2);
    tiling.set_dim(static_cast<int32_t>(dimAttr));
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const ge::Shape &origin = x_shape->GetOriginShape();
    int ndim = origin.GetDimNum();
    int64_t dim = *context->GetAttrs()->GetAttrPointer<int64_t>(0);
    if (dim < 0) dim += ndim;

    ge::Shape y_shape;
    if (ndim == 0) {
        y_shape = origin;
    } else {
        std::vector<int64_t> dims;
        for (int i = 0; i < ndim; ++i) {
            if (i != dim) dims.push_back(origin.GetDim(i));
        }
        y_shape = ge::Shape(dims);
    }
    context->SetOutputShape(0, y_shape);
    return GRAPH_SUCCESS;
}
