kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss_per_sample,
                                uint32_t batchSize, uint32_t featureDim, float margin)
    {
        batchSize_ = batchSize;
        featureDim_ = featureDim;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, batchSize * featureDim);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, batchSize * featureDim);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, batchSize * featureDim);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss_per_sample, batchSize);

        // Allocate local buffers: one tile for each input and one for output and temporaries
        uint32_t tileLen = featureDim; // process one sample at a time
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileLen * sizeof(float));
        pipe_.InitBuffer(inQueueP_, BUFFER_NUM, tileLen * sizeof(float));
        pipe_.InitBuffer(inQueueN_, BUFFER_NUM, tileLen * sizeof(float));
        pipe_.InitBuffer(outQueueL_, BUFFER_NUM, sizeof(float));  // one float per sample
        pipe_.InitBuffer(tmpQueue_, 1, tileLen * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || featureDim_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each core processes a contiguous chunk of samples
        uint32_t samplesPerBlock = (batchSize_ + blockNum - 1) / blockNum;
        uint32_t startSample = blockIdx * samplesPerBlock;
        uint32_t endSample = startSample + samplesPerBlock;
        if (startSample >= batchSize_) return;
        if (endSample > batchSize_) endSample = batchSize_;

        for (uint32_t sampleIdx = startSample; sampleIdx < endSample; ++sampleIdx) {
            // Load anchor, positive, negative for this sample
            uint32_t baseOffset = sampleIdx * featureDim_;
            copyIn(anchorGm_, baseOffset, featureDim_, inQueueA_);
            copyIn(positiveGm_, baseOffset, featureDim_, inQueueP_);
            copyIn(negativeGm_, baseOffset, featureDim_, inQueueN_);

            // Compute distances
            AscendC::LocalTensor<float> aLocal = inQueueA_.DeQue<float>();
            AscendC::LocalTensor<float> pLocal = inQueueP_.DeQue<float>();
            AscendC::LocalTensor<float> nLocal = inQueueN_.DeQue<float>();
            AscendC::LocalTensor<float> diff = tmpQueue_.AllocTensor<float>();

            // Compute squared distance anchor-positive
            AscendC::Sub(diff, aLocal, pLocal, featureDim_);
            AscendC::Mul(diff, diff, diff, featureDim_);
            float sumAp = reduceSum(diff, featureDim_);
            float dAp = sqrtf(sumAp);

            // Compute squared distance anchor-negative
            AscendC::Sub(diff, aLocal, nLocal, featureDim_);
            AscendC::Mul(diff, diff, diff, featureDim_);
            float sumAn = reduceSum(diff, featureDim_);
            float dAn = sqrtf(sumAn);

            // Loss: max(0, dAp - dAn + margin)
            float lossVal = dAp - dAn + margin_;
            if (lossVal < 0.0f) lossVal = 0.0f;

            // Free input tensors
            inQueueA_.FreeTensor(aLocal);
            inQueueP_.FreeTensor(pLocal);
            inQueueN_.FreeTensor(nLocal);
            tmpQueue_.FreeTensor(diff);

            // Write loss to global memory
            AscendC::LocalTensor<float> lossLocal = outQueueL_.AllocTensor<float>();
            lossLocal.SetValue(0, lossVal);
            AscendC::DataCopy(lossGm_[sampleIdx], lossLocal, 1);
            outQueueL_.FreeTensor(lossLocal);
        }
    }

private:
    __aicore__ inline void copyIn(AscendC::GlobalTensor<float>& srcGm,
                                  uint32_t offset, uint32_t len,
                                  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>& queue)
    {
        AscendC::LocalTensor<float> local = queue.AllocTensor<float>();
        AscendC::DataCopy(local, srcGm[offset], len);
        queue.EnQue(local);
    }

    __aicore__ inline float reduceSum(AscendC::LocalTensor<float>& vec, uint32_t len)
    {
        // Simple scalar reduction using a loop over segments
        float sum = 0.0f;
        for (uint32_t i = 0; i < len; ++i) {
            sum += vec.GetValue(i);
        }
        return sum;
    }

    __aicore__ inline float sqrtf(float val)
    {
        return sqrtf(val);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueP_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueN_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueL_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t batchSize_ {0};
    uint32_t featureDim_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive,
                                                                  GM_ADDR negative, GM_ADDR loss_per_sample,
                                                                  GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss_per_sample,
            tiling_data.batchSize, tiling_data.featureDim, tiling_data.margin);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'triplet_margin_loss_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, *inputs):
        # inputs: anchor, positive, negative, margin_tensor
        anchor, positive, negative, margin_tensor = inputs
        margin = float(margin_tensor.reshape(-1)[0].item()) if margin_tensor.numel() else 0.0
        loss_per_sample = getattr(custom_ops_lib, _FN)(anchor, positive, negative, margin)
        # Reduce to scalar (mean reduction as in original)
        return loss_per_sample.mean()
