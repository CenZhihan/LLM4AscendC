host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model per block:
//   xLocal:         1 * tileLength floats  (4 bytes each)
//   maskLocal:      1 * tileLength uint8    (1 byte each)
//   yLocal:         1 * tileLength floats   (4 bytes each)
//   productLocal:   1 * tileLength floats   (4 bytes each)
//   scratchLocal:   1 * tileLength floats   (4 bytes each)  // cumsum temporary
// Total bytes = (4+1+4+4+4)*tileLength = 17 * tileLength
// Conservative: 200 KB UB (per core) => maxTile ~ 200*1024/17 ≈ 12000
static constexpr uint32_t UB_BUDGET_BYTES = 200 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 12288;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaskedCumsumCustomTilingData tiling;

    auto shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(shape.GetShapeSize());
    auto dim = context->GetAttrs()->GetAttrPointer<int32_t>(0);
    int32_t dimVal = *dim;

    // Compute dimSize and outerSize for a generic ND tensor
    int32_t ndim = shape.GetDimNum();
    uint32_t dimSize = 1;
    uint32_t outerSize = 1;
    for (int32_t i = 0; i < ndim; ++i) {
        uint32_t d = static_cast<uint32_t>(shape.GetDim(i));
        if (i < dimVal) {
            outerSize *= d;
        } else if (i == dimVal) {
            dimSize = d;
        }
    }

    // tiling length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 17; // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (dimSize < tileLen) {
        tileLen = AlignDown(dimSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = dimSize;
    }

    // blockDim: number of rows (outerSize) distributed across blocks, cap at 24
    uint32_t numRows = outerSize;
    uint32_t blockDim = numRows;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_dimSize(dimSize);
    tiling.set_outerSize(outerSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    context->GetWorkspaceSizes(1)[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:
    explicit MaskedCumsumCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)

            .ParamType(REQUIRED)

            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int(0);

        this->AICore().SetTiling(optiling::TilingFunc);

};
OP_ADD(MaskedCumsumCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t dimSize,
                                uint32_t outerSize, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        dimSize_  = dimSize;
        outerSize_ = outerSize;
        tileLen_  = tileLength;
        // Stride to next row: dimSize
        rowStride_ = dimSize_;
        // Index offsets start at blockIdx * ? we'll handle grid stride

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ uint8_t*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(uint8_t));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float)); // for product + scratch
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows (outerSize)
        for (uint32_t row = blockIdx; row < outerSize_; row += blockNum) {
            uint32_t rowBase = row * rowStride_;
            float runningSum = 0.0f;

            for (uint32_t col = 0; col < dimSize_; col += tileLen_) {
                uint32_t curLen = tileLen_;
                if (col + curLen > dimSize_) curLen = dimSize_ - col;
                uint32_t gmOffset = rowBase + col;

                // Prefetch next tile? For simplicitly use blocking without double buffering
                // However, we can perform a simple loop with single buffer.
                // To overlap compute with copy we would need double buffering, but for clarity we keep it simple.
                // One can enhance later.

                // Copy in x and mask
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<uint8_t> maskLocal = inQueueMask_.AllocTensor<uint8_t>();
                AscendC::DataCopy(xLocal, xGm_[gmOffset], curLen);
                AscendC::DataCopy(maskLocal, maskGm_[gmOffset], curLen);

                // Compute product = x * mask (float)
                AscendC::LocalTensor<float> productLocal = tmpQueue_.AllocTensor<float>();
                AscendC::Mul(productLocal, xLocal, maskLocal, curLen); // Mul supports float and uint8? Must cast.
                // Actually Mul requires same type; we can cast mask to float.
                // Better: convert mask to float using Mul with 1.0f? Or use Muls? We'll convert via Cast.
                // Let's do: cast mask to float temporary.
                // We'll reuse xLocal after product? Not needed.
                // To simplify, we first convert mask to float in a temp, then multiply.
                AscendC::LocalTensor<float> maskFloatLocal = inQueueMask_.AllocTensor<float>(); // reuse mask queue? No, mask is uint8.
                // We'll allocate another buffer.
                // Actually we can do: product = x * (mask ? 1.0 : 0.0) using conditional? Simpler: use AscendC::Mul with maskLocal as float? No.
                // Instead, we can use AscendC::Select? Or just compute product by:
                // for i in range: if maskLocal[i] > 0 then product[i]=x[i] else product[i]=0.
                // AscendC::Mul does not support mixed types. We can use AscendC::Casts.
                // We'll cast mask to float using AscendC::Cast.
                AscendC::LocalTensor<float> maskFloat = tmpQueue_.AllocTensor<float>(); // need additional temp
                AscendC::Cast(maskFloat, maskLocal, AscendC::RoundMode::CAST_NONE, curLen);
                AscendC::Mul(productLocal, xLocal, maskFloat, curLen);

                // Compute cumulative sum on productLocal into yLocal
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Cumsum(yLocal, productLocal, curLen);

                // Add runningSum to all elements
                AscendC::Adds(yLocal, yLocal, runningSum, curLen);

                // Update runningSum with the last element of the cumsum result
                runningSum = yLocal[curLen - 1];

                // Copy out y
                AscendC::DataCopy(yGm_[gmOffset], yLocal, curLen);

                // Free tensors
                outQueueY_.FreeTensor(yLocal);
                tmpQueue_.FreeTensor(maskFloat);
                tmpQueue_.FreeTensor(productLocal);
                inQueueMask_.FreeTensor(maskLocal);
                inQueueX_.FreeTensor(xLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;   // uint8
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<uint8_t> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t dimSize_  {0};
    uint32_t outerSize_ {0};
    uint32_t tileLen_  {0};
    uint32_t rowStride_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                                            GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.totalLength, tiling_data.dimSize,
            tiling_data.outerSize, tiling_data.tileLength);
    op.Process();
}
