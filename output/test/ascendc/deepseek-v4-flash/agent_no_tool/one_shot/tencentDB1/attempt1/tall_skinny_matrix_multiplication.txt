namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128; // for vectorized loads
static constexpr uint32_t MIN_TILE_ELEMS = 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    // Extract shapes
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    uint32_t M = a_shape->GetOriginShape().GetDim(0);
    uint32_t K = a_shape->GetOriginShape().GetDim(1);
    uint32_t N = b_shape->GetOriginShape().GetDim(1);

    // For tall-skinny case, M and N are large, K is small (e.g., 32)
    // UB budget per block: we need space for A_tile (tileM * K), B_tile (K * tileN), C_tile (tileM * tileN)
    // and workspace for intermediate results. We'll use a conservative estimate:
    // total floats = tileM*K + K*tileN + tileM*tileN  (all float)
    // To keep it simple, we allocate for max tile sizes that fit.
    // We'll set tileM = tileN so that output tile is square. But we can also optimize for shape.
    // Attempt to use available UB: let's assume each tile uses 3 * (tileM * tileN) floats approx.
    // Since K is small, tileM*K and K*tileN are less significant. We'll solve for tileSize.
    uint32_t maxBytes = UB_BUDGET_BYTES * 0.8; // leave margin
    uint32_t maxFloats = maxBytes / sizeof(float);
    // We need room for 3 tile-sized buffers -> max per tile floats = maxFloats / 3.
    uint32_t perTileFloats = maxFloats / 3;
    // Approximate tileM * tileN = perTileFloats (ignoring K terms for small K)
    uint32_t tileM = static_cast<uint32_t>(sqrt(static_cast<double>(perTileFloats)));
    uint32_t tileN = tileM;
    // Clip to shape dimensions
    if (tileM > M) tileM = M;
    if (tileN > N) tileN = N;
    // Align down to ALIGN_ELEMS for vectorization
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    tileN = AlignDown(tileN, ALIGN_ELEMS);
    if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;
    if (tileN < ALIGN_ELEMS) tileN = ALIGN_ELEMS;
    if (tileM > M) tileM = M;
    if (tileN > N) tileN = N;

    // Compute number of tiles and blockDim
    uint32_t numTilesRow = CeilDiv(M, tileM);
    uint32_t numTilesCol = CeilDiv(N, tileN);
    uint32_t totalTiles = numTilesRow * numTilesCol;
    uint32_t blockDim = totalTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileN)
    {
        M_ = M;
        K_ = K;
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;
        // Global tensor wrappers
        aGm_.SetGlobalBuffer((__gm__ float*)A, M * K);
        bGm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        cGm_.SetGlobalBuffer((__gm__ float*)C, M * N);

        // Pipe buffers: two for A tile, two for B tile, and one for C result
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM * K * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, K * tileN * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileM * tileN * sizeof(float));
        // Temporary workspace for accumulation? We'll compute directly from local tensors.
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a set of output tiles in a grid-stride pattern
        // Number of tiles in row/col direction
        uint32_t tilesRow = (M_ + tileM_ - 1) / tileM_;
        uint32_t tilesCol = (N_ + tileN_ - 1) / tileN_;
        uint32_t totalTiles = tilesRow * tilesCol;

        // Grid-stride over tiles
        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            uint32_t tileRow = tileIdx / tilesCol;
            uint32_t tileCol = tileIdx % tilesCol;

            uint32_t rowStart = tileRow * tileM_;
            uint32_t rowEnd = rowStart + tileM_;
            if (rowEnd > M_) rowEnd = M_;
            uint32_t curTileM = rowEnd - rowStart;

            uint32_t colStart = tileCol * tileN_;
            uint32_t colEnd = colStart + tileN_;
            if (colEnd > N_) colEnd = N_;
            uint32_t curTileN = colEnd - colStart;

            // We'll process the tile by accumulating over K dimension.
            // We'll double buffer A tiles (rows) and B tiles (columns) to overlap compute with data copy.
            // Since K might be small, we can load entire K dimension per tile? Actually we need to accumulate over K.
            // For simplicity, we load entire A block (curTileM x K) and B block (K x curTileN) and then compute.
            // This does not pipeline over K, but we can loop over K sub-tiles if needed. Assume K is small (e.g., 32).
            // We'll just do a single load for A and B.

            LoadAB(rowStart, curTileM, colStart, curTileN, tileRow, tileCol);
            Compute(curTileM, curTileN);
            StoreC(rowStart, curTileM, colStart, curTileN);
        }
    }

private:
    __aicore__ inline void LoadAB(uint32_t rowStart, uint32_t curTileM,
                                  uint32_t colStart, uint32_t curTileN,
                                  uint32_t tileRow, uint32_t tileCol)
    {
        // Allocate local tensors
        AscendC::LocalTensor<float> aTile = inQueueA_.AllocTensor<float>();
        AscendC::LocalTensor<float> bTile = inQueueB_.AllocTensor<float>();

        // Load A tile: shape (curTileM, K)
        // A is stored row-major: aGm_[rowStart * K_] to (rowStart+curTileM-1)*K_+K_-1
        for (uint32_t i = 0; i < curTileM; i++) {
            uint32_t srcOffset = (rowStart + i) * K_;
            uint32_t dstOffset = i * K_;
            AscendC::DataCopy(aTile[dstOffset], aGm_[srcOffset], K_);
        }

        // Load B tile: shape (K, curTileN)
        // B is stored row-major: bGm_[colStart] to ? We need each row of B.
        // B has K rows, each of length N. So B element (k, j) is at k*N + j.
        // We need columns colStart..colStart+curTileN-1 for each row. We'll transpose? For easier access, we load the block.
        // Since K may be small, we can read row by row.
        for (uint32_t k = 0; k < K_; k++) {
            uint32_t srcOffset = k * N_ + colStart;
            uint32_t dstOffset = k * curTileN;
            AscendC::DataCopy(bTile[dstOffset], bGm_[srcOffset], curTileN);
        }

        // Enqueue the tiles for compute
        inQueueA_.EnQue(aTile);
        inQueueB_.EnQue(bTile);
    }

    __aicore__ inline void Compute(uint32_t curTileM, uint32_t curTileN)
    {
        AscendC::LocalTensor<float> aTile = inQueueA_.DeQue<float>();
        AscendC::LocalTensor<float> bTile = inQueueB_.DeQue<float>();
        AscendC::LocalTensor<float> cTile = outQueueC_.AllocTensor<float>();

        // Zero out cTile
        AscendC::SetZero(cTile, curTileM * curTileN);

        // Compute C = A * B (outer product accumulation)
        // A: (curTileM, K) row-major, B: (K, curTileN) row-major
        // We'll do triple nested loop: for i in 0..curTileM-1, for j in 0..curTileN-1, for k in 0..K-1
        // But vectorized approach: use vector multiply-add. Since K is small, we can unroll.
        // For each k, we do: cTile[i][j] += aTile[i*K_ + k] * bTile[k*curTileN + j] for all i,j.
        // This is a rank-1 update. AscendC provides vector outer product? We can use VecMulAdd.
        // We'll loop over k and use vector operations on rows/columns.
        for (uint32_t k = 0; k < K_; k++) {
            // Create a scalar from A column k:
            // For each i, a_scl = aTile[i*K_ + k]
            // Then multiply bTile row k (length curTileN) and add to corresponding row of cTile.
            // We can use VecMuls and VecAdd.
            for (uint32_t i = 0; i < curTileM; i++) {
                float aVal = aTile[i * K_ + k];
                AscendC::LocalTensor<float> cRow = cTile + i * curTileN;
                AscendC::LocalTensor<float> bRow = bTile + k * curTileN;
                AscendC::VecMulAdd(cRow, bRow, aVal, cRow, curTileN);
            }
        }

        outQueueC_.EnQue(cTile);
        inQueueA_.FreeTensor(aTile);
        inQueueB_.FreeTensor(bTile);
    }

    __aicore__ inline void StoreC(uint32_t rowStart, uint32_t curTileM,
                                  uint32_t colStart, uint32_t curTileN)
    {
        AscendC::LocalTensor<float> cTile = outQueueC_.DeQue<float>();
        // Write back to global: C is row-major (M x N)
        for (uint32_t i = 0; i < curTileM; i++) {
            uint32_t srcOffset = i * curTileN;
            uint32_t dstOffset = (rowStart + i) * N_ + colStart;
            AscendC::DataCopy(cGm_[dstOffset], cTile[srcOffset], curTileN);
        }
        outQueueC_.FreeTensor(cTile);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
