static constexpr uint32_t MAX_RANK = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;
    const gert::StorageFormat storage_format = context->GetInputShape(0)->GetStorageFormat();
    (void)storage_format;
    const gert::Shape& input_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rank = static_cast<uint32_t>(input_shape.GetDimNum());
    if (rank > MAX_RANK) return ge::GRAPH_FAILED;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim_raw = atts->GetAttrPointer<int64_t>(0)[0];
    uint32_t dim = (dim_raw < 0) ? static_cast<uint32_t>(dim_raw + rank) : static_cast<uint32_t>(dim_raw);
    if (dim >= rank) dim = rank - 1;

    // Fill shape and compute total size
    uint32_t totalSize = 1;
    for (uint32_t i = 0; i < rank; ++i) {
        tiling.inputShape[i] = static_cast<uint32_t>(input_shape.GetDim(i));
        totalSize *= tiling.inputShape[i];
    }
    uint32_t reduceLen = tiling.inputShape[dim];
    uint32_t outerSize = totalSize / reduceLen;  // number of output elements

    // Determine tile length (output elements per block)
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * reduceLen + 16);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > outerSize) maxTile = AlignDown(outerSize, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = 1;

    // Block dimension: divide outer elements into groups
    uint32_t numTiles = CeilDiv(outerSize, maxTile);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_rank(rank);
    tiling.set_dim(dim);
    tiling.set_totalLength(totalSize);
    tiling.set_tileLength(maxTile);
    tiling.set_blockDim(blockDim);
    tiling.set_reduceLen(reduceLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
