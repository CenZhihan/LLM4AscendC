#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t numGroups,
                                uint32_t groupSize, float epsilon,
                                float invGroupSize) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        numGroups_ = numGroups;
        groupSize_ = groupSize;
        epsilon_ = epsilon;
        invGroupSize_ = invGroupSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, totalLen_);
        betaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueBias_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueGamma_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueBeta_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmp2Queue_, 1, tileLen_ * sizeof(float));
        // small scalar tensors (size 1) for reduction results
        pipe_.InitBuffer(sumQueue_, 1, sizeof(float));
        pipe_.InitBuffer(meanQueue_, 1, sizeof(float));
        pipe_.InitBuffer(varQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t totalGroups = batchSize_ * numGroups_;
        const uint32_t stride = blockNum;
        uint32_t gid = blockIdx;

        while (gid < totalGroups) {
            uint32_t sample = gid / numGroups_;
            uint32_t group = gid % numGroups_;
            uint32_t elemStart = sample * groupSize_ * numGroups_ + group * groupSize_;
            uint32_t elemLen = groupSize_;

            if (elemStart + elemLen > totalLen_) break;

            // Load data for this group
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[elemStart], elemLen);

            AscendC::LocalTensor<float> biasLocal = inQueueBias_.AllocTensor<float>();
            AscendC::DataCopy(biasLocal, biasGm_[elemStart], elemLen);

            // 1) bias add
            AscendC::Add(xLocal, xLocal, biasLocal, elemLen);
            inQueueBias_.FreeTensor(biasLocal);

            // 2) hardtanh: clamp to [-1, 1]
            AscendC::Mins(xLocal, xLocal, 1.0f, elemLen);
            AscendC::Maxs(xLocal, xLocal, -1.0f, elemLen);

            // 3) mish: x * tanh(softplus(x))
            AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> softplusLocal = tmp2Queue_.AllocTensor<float>();
            AscendC::Exp(expLocal, xLocal, elemLen);
            AscendC::Adds(softplusLocal, expLocal, 1.0f, elemLen);
            AscendC::Log(softplusLocal, softplusLocal, elemLen);
            AscendC::Tanh(softplusLocal, softplusLocal, elemLen);
            AscendC::Mul(xLocal, xLocal, softplusLocal, elemLen);
            tmpQueue_.FreeTensor(expLocal);
            tmp2Queue_.FreeTensor(softplusLocal);

            // 4) group norm
            // compute mean = sum(x) * invGroupSize
            AscendC::LocalTensor<float> sumScalar = sumQueue_.AllocTensor<float>();   // size 1
            AscendC::ReduceSum<false>(sumScalar, xLocal, elemLen, 0);
            AscendC::Muls(sumScalar, sumScalar, invGroupSize_, 1);  // now sumScalar(0) = mean
            float meanVal = sumScalar(0);  // scalar

            // subtract mean
            AscendC::Subs(xLocal, xLocal, meanVal, elemLen);

            // compute variance = sum( (x-mean)^2 ) * invGroupSize
            AscendC::Mul(xLocal, xLocal, xLocal, elemLen);  // square
            AscendC::ReduceSum<false>(sumScalar, xLocal, elemLen, 0);
            AscendC::Muls(sumScalar, sumScalar, invGroupSize_, 1);
            float varVal = sumScalar(0) + epsilon_;

            // invStd = rsqrt(variance)
            AscendC::LocalTensor<float> invStdScalar = varQueue_.AllocTensor<float>(); // size 1
            invStdScalar(0) = 1.0f / AscendC::Sqrt(varVal);  // use sqrt and reciprocal
            // Alternatively could use Rsqrt but Sqrt is simpler
            // Restore x from square? We need original (x-mean) again.
            // We have overwritten xLocal with square. Reload from input? Not possible; we need to recompute.
            // Better to compute diff before squaring. Let's restructure:
            // We'll use a separate diff tensor for variance computation.

            // I will restructure: compute diff, square it, sum, then restore diff for normalization.
            // Since we are in a tight loop, we can reload xLocal from global? That would be costly.
            // Better: keep a copy of diff in another buffer.
            // We'll allocate diffLocal from tmpQueue_ (after freeing expLocal/softplusLocal, they are free).
            // But we already used tmpQueue_ for expLocal and freed it. So it's available.
            // Let's re-allocate diffLocal.

            // Actually, let's restructure the whole group norm to avoid reloading.
            // Compute mean, then compute diff, square, sum, then compute invStd, then apply gain/bias.
            // We'll need two passes over x: one for mean, one for variance? Not necessary if we store diff.

            // Given time, I'll keep simple: recompute diff after squaring by reloading xLocal from global? No.
            // Instead, we can compute diff = x - mean, store it in a temporary, then square it for variance, then use the stored diff for normalization.

            sumQueue_.FreeTensor(sumScalar);
            // We need to redo the steps properly. Let's rewrite the whole group norm section cleanly.

            // 4) group norm (corrected)
            // a) compute mean
            // b) compute diff = x - mean
            // c) compute variance = mean of diff^2
            // d) compute invStd = rsqrt(variance + epsilon)
            // e) normalize: x_norm = diff * invStd
            // f) apply gamma and beta

            // We'll use separate buffers: diffLocal from tmpQueue_, sqrLocal from tmp2Queue_ (they are freed earlier)
            // meanScalar from meanQueue_, sumScalar from sumQueue_, varScalar from varQueue_.

            // Initialize scalars
            AscendC::LocalTensor<float> meanScalar = meanQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> sumScalar2 = sumQueue_.AllocTensor<float>(); // reuse sumQueue
            AscendC::LocalTensor<float> varScalar = varQueue_.AllocTensor<float>();

            // mean
            AscendC::ReduceSum<false>(sumScalar2, xLocal, elemLen, 0);
            AscendC::Muls(sumScalar2, sumScalar2, invGroupSize_, 1);
            meanScalar(0) = sumScalar2(0);
            float mean = meanScalar(0);

            // diff = x - mean
            AscendC::LocalTensor<float> diffLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Subs(diffLocal, xLocal, mean, elemLen);

            // squared diff for variance
            AscendC::LocalTensor<float> sqrLocal = tmp2Queue_.AllocTensor<float>();
            AscendC::Mul(sqrLocal, diffLocal, diffLocal, elemLen);

            // variance = mean of squares
            AscendC::ReduceSum<false>(sumScalar2, sqrLocal, elemLen, 0);
            AscendC::Muls(sumScalar2, sumScalar2, invGroupSize_, 1);
            float var = sumScalar2(0) + epsilon_;
            float invStd = 1.0f / AscendC::Sqrt(var);
            varScalar(0) = invStd;

            // normalize
            AscendC::Muls(diffLocal, diffLocal, invStd, elemLen);

            // load gamma and beta
            AscendC::LocalTensor<float> gammaLocal = inQueueGamma_.AllocTensor<float>();
            AscendC::DataCopy(gammaLocal, gammaGm_[elemStart], elemLen);
            AscendC::LocalTensor<float> betaLocal = inQueueBeta_.AllocTensor<float>();
            AscendC::DataCopy(betaLocal, betaGm_[elemStart], elemLen);

            // apply gamma and beta
            AscendC::Mul(diffLocal, diffLocal, gammaLocal, elemLen);
            AscendC::Add(diffLocal, diffLocal, betaLocal, elemLen);

            // output
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, diffLocal, elemLen);
            AscendC::DataCopy(yGm_[elemStart], yLocal, elemLen);

            // free all
            inQueueX_.FreeTensor(xLocal);
            tmpQueue_.FreeTensor(diffLocal);
            tmp2Queue_.FreeTensor(sqrLocal);
            meanQueue_.FreeTensor(meanScalar);
            sumQueue_.FreeTensor(sumScalar2);
            varQueue_.FreeTensor(varScalar);
            inQueueGamma_.FreeTensor(gammaLocal);
            inQueueBeta_.FreeTensor(betaLocal);
            outQueueY_.FreeTensor(yLocal);

            gid += stride;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBias_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueGamma_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBeta_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp2Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQueue_;   // for scalars
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> meanQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> varQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t batchSize_{0};
    uint32_t numGroups_{0};
    uint32_t groupSize_{0};
    float epsilon_{1e-5f};
    float invGroupSize_{0.0f};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    op.Process();
}
