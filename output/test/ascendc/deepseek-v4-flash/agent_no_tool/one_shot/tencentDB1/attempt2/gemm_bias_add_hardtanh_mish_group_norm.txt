#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR linBias, GM_ADDR bias,
                                GM_ADDR gnGamma, GM_ADDR gnBeta, GM_ADDR y, GM_ADDR workspace,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t numGroups, float epsilon) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        epsilon_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inFeatures);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures * inFeatures);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)linBias, outFeatures);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gnGamma, outFeatures);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gnBeta, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        wsGm_.SetGlobalBuffer((__gm__ float*)workspace, totalLength);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmp2Queue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        Phase1MatmulWithPostOps();
        Phase2GroupNorm();
    }

private:
    __aicore__ inline void Phase1MatmulWithPostOps() {
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        using Matmul = AscendC::Matmul<AscendC::MatmulType<AscendC::ROMMATRIX_IN_VALUE, AscendC::ROMMATRIX_IN_VALUE, AscendC::ROMMATRIX_OUT_VALUE>>;
        Matmul matmul;
        matmul.SetTensorA(xGm_);
        matmul.SetTensorB(wGm_);
        matmul.SetTransB(true);
        matmul.SetBias(linBiasGm_);
        matmul.SetTensorC(wsGm_);
        matmul.IterateAll();

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> wsLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(wsLocal, wsGm_[base], curLen);

            uint32_t biasBase = base % outFeatures_;
            uint32_t biasLen = curLen;
            if (biasBase + biasLen > outFeatures_) biasLen = outFeatures_ - biasBase;
            AscendC::LocalTensor<float> biasLocal = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(biasLocal, biasGm_[biasBase], biasLen);
            AscendC::Add(wsLocal, wsLocal, biasLocal, curLen);
            tmpQueue_.FreeTensor(biasLocal);

            AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> minLocal = tmp2Queue_.AllocTensor<float>();
            AscendC::Mins(maxLocal, wsLocal, 1.0f, curLen);
            AscendC::Maxs(wsLocal, maxLocal, -1.0f, curLen);
            tmpQueue_.FreeTensor(maxLocal);
            tmp2Queue_.FreeTensor(minLocal);

            AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> softplusLocal = tmp2Queue_.AllocTensor<float>();
            AscendC::Exp(expLocal, wsLocal, curLen);
            AscendC::Add(softplusLocal, expLocal, 1.0f, curLen);
            AscendC::Log(softplusLocal, softplusLocal, curLen);
            AscendC::Tanh(softplusLocal, softplusLocal, curLen);
            AscendC::Mul(wsLocal, wsLocal, softplusLocal, curLen);
            tmpQueue_.FreeTensor(expLocal);
            tmp2Queue_.FreeTensor(softplusLocal);

            AscendC::DataCopy(wsGm_[base], wsLocal, curLen);
            inQueueX_.FreeTensor(wsLocal);

            base += stride;
        }
    }

    __aicore__ inline void Phase2GroupNorm() {
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t groupSize = outFeatures_ / numGroups_;
        const uint32_t totalGroups = batchSize_ * numGroups_;
        const uint32_t stride = blockNum;
        uint32_t gid = blockIdx;
        while (gid < totalGroups) {
            uint32_t sample = gid / numGroups_;
            uint32_t group = gid % numGroups_;
            uint32_t elemStart = sample * outFeatures_ + group * groupSize;
            uint32_t elemLen = groupSize;

            AscendC::LocalTensor<float> groupLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(groupLocal, wsGm_[elemStart], elemLen);

            AscendC::LocalTensor<float> meanLocal = tmpQueue_.AllocTensor<float>();
            AscendC::VecReduceSum(meanLocal, groupLocal, elemLen);
            float meanScalar = meanLocal.GetValue(0) / elemLen;
            AscendC::BroadcastVar(meanLocal, meanScalar, elemLen);

            AscendC::LocalTensor<float> diffLocal = tmp2Queue_.AllocTensor<float>();
            AscendC::Sub(diffLocal, groupLocal, meanLocal, elemLen);
            AscendC::Mul(diffLocal, diffLocal, diffLocal, elemLen);
            AscendC::VecReduceSum(meanLocal, diffLocal, elemLen);
            float varScalar = meanLocal.GetValue(0) / elemLen + epsilon_;
            tmp2Queue_.FreeTensor(diffLocal);

            float sqrtVar = sqrt(varScalar);
            AscendC::LocalTensor<float> normLocal = outQueueY_.AllocTensor<float>();
            AscendC::Sub(normLocal, groupLocal, meanLocal, elemLen);
            AscendC::Muls(normLocal, normLocal, 1.0f / sqrtVar, elemLen);

            AscendC::LocalTensor<float> gammaLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> betaLocal = tmp2Queue_.AllocTensor<float>();
            AscendC::DataCopy(gammaLocal, gnGammaGm_[group * groupSize], elemLen);
            AscendC::DataCopy(betaLocal, gnBetaGm_[group * groupSize], elemLen);
            AscendC::Mul(normLocal, normLocal, gammaLocal, elemLen);
            AscendC::Add(normLocal, normLocal, betaLocal, elemLen);

            AscendC::DataCopy(yGm_[elemStart], normLocal, elemLen);

            inQueueX_.FreeTensor(groupLocal);
            tmpQueue_.FreeTensor(meanLocal);
            outQueueY_.FreeTensor(normLocal);
            tmp2Queue_.FreeTensor(betaLocal);

            gid += stride;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp2Queue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wsGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t batchSize_{0};
    uint32_t inFeatures_{0};
    uint32_t outFeatures_{0};
    uint32_t numGroups_{0};
    float epsilon_{1e-5f};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR linBias, GM_ADDR bias,
    GM_ADDR gnGamma, GM_ADDR gnBeta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(td, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, linBias, bias, gnGamma, gnBeta, y, workspace,
            td.totalLength, td.tileLength, td.batchSize, td.inFeatures,
            td.outFeatures, td.numGroups, td.epsilon);
    op.Process();
}
