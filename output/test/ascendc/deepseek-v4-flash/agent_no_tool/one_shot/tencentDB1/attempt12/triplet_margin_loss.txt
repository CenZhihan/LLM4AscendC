#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1;  // no alignment requirement for reduction
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TripletMarginLossCustomTilingData tiling;
    const gert::Shape *anchorShape = context->GetInputShape(0);
    auto &shape = anchorShape->GetOriginShape();
    uint32_t totalBatch = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t featureDim = static_cast<uint32_t>(shape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float margin = *attrs->GetAttrPointer<float>(0);

    // UB budget: 3 input tiles (anchor, positive, negative) + 2 intermediate (diff and sq, reused) + 1 output scalar
    // Each tile uses featureDim elements, but we'll use tileLen for pipeline
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float)); // 3 tensors simultaneously
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < 1) maxTile = 1;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = maxTile;
    if (featureDim < tileLen) tileLen = featureDim;
    // Ensure tileLen > 0
    if (tileLen == 0) tileLen = 1;

    // Single block for reduction (simplest)
    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalBatch(totalBatch);
    tiling.set_featureDim(featureDim);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t totalBatch, uint32_t featureDim,
                                uint32_t tileLen, float margin) {
        totalBatch_ = totalBatch;
        featureDim_ = featureDim;
        tileLen_ = tileLen;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalBatch * featureDim);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalBatch * featureDim);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalBatch * featureDim);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueP_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueN_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(diffQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sqQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalBatch_ == 0 || featureDim_ == 0) return;

        float totalLoss = 0.0f;

        for (uint32_t sample = 0; sample < totalBatch_; ++sample) {
            uint32_t base = sample * featureDim_;
            // Reset per-sample accumulators
            float sqAP = 0.0f;
            float sqAN = 0.0f;
            uint32_t remaining = featureDim_;
            uint32_t offset = 0;

            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

                // Load anchor, positive, negative
                AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
                AscendC::DataCopy(aLocal, anchorGm_[base + offset], curLen);
                AscendC::LocalTensor<float> pLocal = inQueueP_.AllocTensor<float>();
                AscendC::DataCopy(pLocal, positiveGm_[base + offset], curLen);
                AscendC::LocalTensor<float> nLocal = inQueueN_.AllocTensor<float>();
                AscendC::DataCopy(nLocal, negativeGm_[base + offset], curLen);

                // diffAP = a - p
                AscendC::LocalTensor<float> diff = diffQueue_.AllocTensor<float>();
                AscendC::Sub(diff, aLocal, pLocal, curLen);
                // sqAP = diff * diff
                AscendC::LocalTensor<float> sq = sqQueue_.AllocTensor<float>();
                AscendC::Mul(sq, diff, diff, curLen);
                // reduce sum to scalar
                float partialSqAP = AscendC::ReduceSum(sq, curLen);
                sqAP += partialSqAP;

                // diffAN = a - n (reuse diff tensor)
                AscendC::Sub(diff, aLocal, nLocal, curLen);
                AscendC::Mul(sq, diff, diff, curLen);
                float partialSqAN = AscendC::ReduceSum(sq, curLen);
                sqAN += partialSqAN;

                // free tensors
                diffQueue_.FreeTensor(diff);
                sqQueue_.FreeTensor(sq);
                inQueueA_.FreeTensor(aLocal);
                inQueueP_.FreeTensor(pLocal);
                inQueueN_.FreeTensor(nLocal);

                remaining -= curLen;
                offset += curLen;
            }

            // loss per sample = max(0, sqAP - sqAN + margin)
            float sampleLoss = (sqAP - sqAN + margin_);
            if (sampleLoss < 0.0f) sampleLoss = 0.0f;
            totalLoss += sampleLoss;
        }

        float meanLoss = totalLoss / (float)totalBatch_;

        // Write result
        AscendC::LocalTensor<float> outLocal;
        outLocal.SetValue(0, meanLoss);
        AscendC::DataCopy(lossGm_, outLocal, 1);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueP_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueN_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> diffQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sqQueue_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalBatch_;
    uint32_t featureDim_;
    uint32_t tileLen_;
    float margin_;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                  GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalBatch, tiling_data.featureDim,
            tiling_data.tileLen, tiling_data.margin);
    op.Process();
}

import torch

class ModelNew(torch.nn.Module):
