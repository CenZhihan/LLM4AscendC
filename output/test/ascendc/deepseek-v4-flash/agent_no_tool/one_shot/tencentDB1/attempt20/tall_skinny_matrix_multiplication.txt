host_operator_src="""

#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t MAX_TILE_M = 128;
static constexpr uint32_t MAX_TILE_N = 32;
static constexpr uint32_t MAX_TILE_K = 128;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    uint32_t M = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(0));
    uint32_t K = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));
    uint32_t N = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetDim(1));

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);

    uint32_t tileM = (M > MAX_TILE_M) ? MAX_TILE_M : M;
    uint32_t tileN = (N > MAX_TILE_N) ? MAX_TILE_N : N;
    uint32_t tileK = (K > MAX_TILE_K) ? MAX_TILE_K : K;

    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);

    uint32_t blockDim = 1;
    if (tileM > 0 && tileN > 0) {
        uint32_t numRowTiles = (M + tileM - 1) / tileM;
        uint32_t numColTiles = (N + tileN - 1) / tileN;
        blockDim = numRowTiles * numColTiles;
        if (blockDim > 24) blockDim = 24;
        if (blockDim < 1) blockDim = 1;
    }
    tiling.set_blockDim(blockDim);
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    int64_t M = a_shape->GetDim(0);
    int64_t N = b_shape->GetDim(1);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

kernel_src="""

#include "kernel_operator.h"
#include "matmul_tiling.h"

using namespace AscendC;

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);

    uint32_t M = td.M;
    uint32_t K = td.K;
    uint32_t N = td.N;
    uint32_t tileM = td.tileM;
    uint32_t tileN = td.tileN;
    uint32_t tileK = td.tileK;

    // Set global tensors
    GlobalTensor<float> aGm;
    GlobalTensor<float> bGm;
    GlobalTensor<float> cGm;
    aGm.SetGlobalBuffer((__gm__ float*)a, M * K);
    bGm.SetGlobalBuffer((__gm__ float*)b, K * N);
    cGm.SetGlobalBuffer((__gm__ float*)c, M * N);

    // Matmul tiling configuration
    MatmulTiling mmTiling;
    mmTiling.SetM(tileM);
    mmTiling.SetN(tileN);
    mmTiling.SetK(tileK);
    mmTiling.SetOrigM(M);
    mmTiling.SetOrigN(N);
    mmTiling.SetOrigK(K);
    mmTiling.SetAStride(K, 1);
    mmTiling.SetBStride(N, 1);
    mmTiling.SetCStride(N, 1);
    mmTiling.SetBaseA(aGm);
    mmTiling.SetBaseB(bGm);
    mmTiling.SetBaseC(cGm);

    // Matmul object
    Matmul<MatmulTraits<float, float, float>> matmul;
    matmul.SetTiling(mmTiling);
    matmul.IterateAll();
}
