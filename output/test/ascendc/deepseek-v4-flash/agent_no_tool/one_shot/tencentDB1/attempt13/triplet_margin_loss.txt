static constexpr uint32_t ALIGN_ELEMS = 1;  // no alignment requirement for reduction
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TripletMarginLossCustomTilingData tiling;
    auto shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalBatch = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t featureDim = static_cast<uint32_t>(shape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float margin = *attrs->GetAttrPointer<float>(0);

    // UB budget: 3 input tiles (anchor, positive, negative) + 2 intermediate (diff and sq, reused) + 1 output scalar
    // Each tile uses featureDim elements, but we'll use tileLen for pipeline
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float)); // 3 tensors simultaneously
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < 1) maxTile = 1;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = maxTile;
    if (featureDim < tileLen) tileLen = featureDim;
    // Ensure tileLen > 0
    if (tileLen == 0) tileLen = 1;

    // Single block for reduction (simplest)
    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalBatch(totalBatch);
    tiling.set_featureDim(featureDim);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
