host_operator_src="""

#include "max_pooling3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static inline uint32_t OutputDim(uint32_t in, uint32_t kernel, uint32_t stride, uint32_t pad, bool ceil) {
    if (ceil)
        return CeilDiv(in + 2 * pad - (kernel - 1) - 1, stride) + 1;
    else
        return (in + 2 * pad - (kernel - 1) - 1) / stride + 1;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaxPooling3dCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const gert::Shape &x_shape_ref = context->GetInputShape(0)->GetOriginShape();
    uint32_t n = static_cast<uint32_t>(x_shape_ref.GetDim(0));
    uint32_t c = static_cast<uint32_t>(x_shape_ref.GetDim(1));
    uint32_t d = static_cast<uint32_t>(x_shape_ref.GetDim(2));
    uint32_t h = static_cast<uint32_t>(x_shape_ref.GetDim(3));
    uint32_t w = static_cast<uint32_t>(x_shape_ref.GetDim(4));
    uint32_t totalElements = n * c * d * h * w;

    uint32_t kernel_size = static_cast<uint32_t>(attrs->GetAttrPointer<int>(0)[0]);
    uint32_t stride = static_cast<uint32_t>(attrs->GetAttrPointer<int>(1)[0]);
    uint32_t padding = static_cast<uint32_t>(attrs->GetAttrPointer<int>(2)[0]);
    bool ceil_mode = attrs->GetAttrPointer<bool>(4)[0];

    uint32_t od = OutputDim(d, kernel_size, stride, padding, ceil_mode);
    uint32_t oh = OutputDim(h, kernel_size, stride, padding, ceil_mode);
    uint32_t ow = OutputDim(w, kernel_size, stride, padding, ceil_mode);

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalElements < tileLen) {
        tileLen = AlignDown(totalElements, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalElements;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalElements, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalElements);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_d(d);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_kernel_size(kernel_size);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_ceil_mode(ceil_mode ? 1u : 0u);
    tiling.set_od(od);
    tiling.set_oh(oh);
    tiling.set_ow(ow);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t kernel_size = static_cast<uint32_t>(attrs->GetAttrPointer<int>(0)[0]);
    uint32_t stride = static_cast<uint32_t>(attrs->GetAttrPointer<int>(1)[0]);
    uint32_t padding = static_cast<uint32_t>(attrs->GetAttrPointer<int>(2)[0]);
    bool ceil_mode = attrs->GetAttrPointer<bool>(4)[0];

    auto *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, x_shape->GetDim(0));
    y_shape->SetDim(1, x_shape->GetDim(1));
    uint32_t d = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t h = static_cast<uint32_t>(x_shape->GetDim(3));
    uint32_t w = static_cast<uint32_t>(x_shape->GetDim(4));

    auto OutputDim = [&](uint32_t in, uint32_t k, uint32_t s, uint32_t p) -> uint32_t {
        if (ceil_mode)
            return (in + 2 * p - (k - 1) - 1 + s - 1) / s + 1;
        else
            return (in + 2 * p - (k - 1) - 1) / s + 1;
    };
    y_shape->SetDim(2, static_cast<int64_t>(OutputDim(d, kernel_size, stride, padding)));
    y_shape->SetDim(3, static_cast<int64_t>(OutputDim(h, kernel_size, stride, padding)));
    y_shape->SetDim(4, static_cast<int64_t>(OutputDim(w, kernel_size, stride, padding)));
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t kernel_size, uint32_t stride, uint32_t padding,
                                uint32_t ceil_mode,
                                uint32_t od, uint32_t oh, uint32_t ow) {
        n_ = n; c_ = c; d_ = d; h_ = h; w_ = w;
        k_ = kernel_size; s_ = stride; p_ = padding;
        ceil_ = ceil_mode;
        od_ = od; oh_ = oh; ow_ = ow;
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, od_ * oh_ * ow_ * c_ * n_);
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t totalOutput = n_ * c_ * od_ * oh_ * ow_;
        uint32_t chunkSize = (totalOutput + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * chunkSize;
        uint32_t end = start + chunkSize;
        if (end > totalOutput) end = totalOutput;

        for (uint32_t idx = start; idx < end; ++idx) {
            uint32_t ow_idx = idx % ow_;
            uint32_t temp = idx / ow_;
            uint32_t oh_idx = temp % oh_;
            temp /= oh_;
            uint32_t od_idx = temp % od_;
            temp /= od_;
            uint32_t c_idx = temp % c_;
            uint32_t n_idx = temp / c_;

            int32_t in_d = static_cast<int32_t>(od_idx * s_) - static_cast<int32_t>(p_);
            int32_t in_h = static_cast<int32_t>(oh_idx * s_) - static_cast<int32_t>(p_);
            int32_t in_w = static_cast<int32_t>(ow_idx * s_) - static_cast<int32_t>(p_);

            float max_val = -3.4028235e+38f;

            for (uint32_t kd = 0; kd < k_; ++kd) {
                int32_t id = in_d + static_cast<int32_t>(kd);
                if (id < 0 || id >= static_cast<int32_t>(d_)) continue;
                for (uint32_t kh = 0; kh < k_; ++kh) {
                    int32_t ih = in_h + static_cast<int32_t>(kh);
                    if (ih < 0 || ih >= static_cast<int32_t>(h_)) continue;
                    for (uint32_t kw = 0; kw < k_; ++kw) {
                        int32_t iw = in_w + static_cast<int32_t>(kw);
                        if (iw < 0 || iw >= static_cast<int32_t>(w_)) continue;

                        uint32_t offset = ((n_idx * c_ + c_idx) * d_ + static_cast<uint32_t>(id)) * h_ * w_
                                        + static_cast<uint32_t>(ih) * w_ + static_cast<uint32_t>(iw);
                        float val = xGm_.GetValue(offset);
                        if (val > max_val) max_val = val;
                    }
                }
            }

            uint32_t output_offset = ((n_idx * c_ + c_idx) * od_ + od_idx) * oh_ * ow_
                                    + oh_idx * ow_ + ow_idx;
            yGm_.SetValue(output_offset, max_val);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t n_, c_, d_, h_, w_;
    uint32_t k_, s_, p_;
    uint32_t ceil_;
    uint32_t od_, oh_, ow_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void max_pooling3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.kernel_size, tiling_data.stride, tiling_data.padding,
            tiling_data.ceil_mode,
            tiling_data.od, tiling_data.oh, tiling_data.ow);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0, return_indices: bool = False, ceil_mode: bool = False):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride if stride is not None else kernel_size
        self.padding = padding
        self.ceil_mode = ceil_mode

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding, self.ceil_mode)
