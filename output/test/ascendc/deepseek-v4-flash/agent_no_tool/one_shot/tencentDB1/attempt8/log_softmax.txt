host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (three passes, each uses 2*in + 2*out + 1*tmp = 5*tile floats)
// plus scratch for reduction (small). So keep 5*tile bytes = 20*tile.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    // dim attribute: note shape is 2D, we interpret dim as the reduction axis
    // For simplicity we enforce dim == 1 (last dim) here; the attribute is stored but not used in tiling beyond sanity.
    // The actual dim is used in kernel to know row length = totalLength / innerDim (batchsize).
    // We'll compute batchSize = totalLength / dim; dim must divide totalLength.
    int64_t dimAttr = 1;
    if (attrs && attrs->GetAttrNum() > 0) {
        dimAttr = attrs->GetAttrPointer<int64_t>(0)[0];
    }
    // For this simple implementation, we assume dim == 1 (last dimension).
    // The tiling just computes tileLen based on UB budget.
    (void)dimAttr; // avoid warning

    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    // Total length is product of all dims. We need to pass the inner dimension (the one we softmax over).
    // For ND input, we assume the last dimension is the softmax dimension.
    // We extract it from input shape.
    // Since the operator is defined for ND, we take the last dimension size.
    const gert::StorageShape &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t ndim = shape.GetDimNum();
    uint32_t innerDim = 1;
    if (ndim > 0) {
        innerDim = static_cast<uint32_t>(shape.GetDim(ndim-1));
    }
    // Ensure totalLength is divisible by innerDim
    uint32_t batchSize = (innerDim == 0) ? 1 : (totalLength / innerDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(innerDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t dim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        dim_ = dim;
        batchSize_ = (dim_ == 0) ? 1 : (totalLen_ / dim_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(redQueue_, 1, sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || batchSize_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows (each row is a softmax instance)
        for (uint32_t row = blockIdx; row < batchSize_; row += blockDim_) {
            uint32_t rowStart = row * dim_;

            // ---- Pass 1: compute row max ----
            float maxVal = -INFINITY;
            uint32_t remaining = dim_;
            uint32_t offset = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                LoadTile(rowStart + offset, curLen);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                float tileMax = TileMax(xLocal, curLen);
                if (tileMax > maxVal) maxVal = tileMax;
                inQueueX_.FreeTensor(xLocal);
                remaining -= curLen;
                offset += curLen;
            }

            // ---- Pass 2: compute sum of exp(x - maxVal) ----
            float sum = 0.0f;
            remaining = dim_;
            offset = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                LoadTile(rowStart + offset, curLen);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tmp1 = tmpQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmp2 = tmpQueue_.AllocTensor<float>();
                // tmp2 = x - maxVal
                AscendC::SubScalar(tmp2, xLocal, maxVal, curLen);
                // tmp1 = exp(x - maxVal)
                AscendC::Exp(tmp1, tmp2, curLen);
                // update sum
                sum += TileSum(tmp1, curLen);
                tmpQueue_.FreeTensor(tmp1);
                tmpQueue_.FreeTensor(tmp2);
                inQueueX_.FreeTensor(xLocal);
                remaining -= curLen;
                offset += curLen;
            }

            // log(sum)
            float logSum = logf(sum);  // use math library

            // offsetVal = maxVal + log(sum)
            float offsetVal = maxVal + logSum;

            // ---- Pass 3: compute y = x - offsetVal ----
            remaining = dim_;
            offset = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                LoadTile(rowStart + offset, curLen);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                // y = x - offsetVal
                AscendC::SubScalar(yLocal, xLocal, offsetVal, curLen);
                inQueueX_.FreeTensor(xLocal);
                outQueueY_.EnQue<float>(yLocal);
                StoreTile(rowStart + offset, curLen);
                remaining -= curLen;
                offset += curLen;
            }
        }
    }

private:
    __aicore__ inline float TileMax(const AscendC::LocalTensor<float>& src, uint32_t len)
    {
        AscendC::LocalTensor<float> dst = redQueue_.AllocTensor<float>();
        AscendC::ReduceMax(dst, src, len, 1);  // reduce along 1D
        float val = dst.GetValue(0);
        redQueue_.FreeTensor(dst);
        return val;
    }

    __aicore__ inline float TileSum(const AscendC::LocalTensor<float>& src, uint32_t len)
    {
        AscendC::LocalTensor<float> dst = redQueue_.AllocTensor<float>();
        AscendC::ReduceSum(dst, src, len, 1);
        float val = dst.GetValue(0);
        redQueue_.FreeTensor(dst);
        return val;
    }

    __aicore__ inline void LoadTile(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void StoreTile(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> redQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
    uint32_t dim_ {0};
    uint32_t batchSize_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim, tiling_data.dim);
    op.Process();
}
