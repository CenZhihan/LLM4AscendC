static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MAX_BLOCK_DIM = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TripletMarginLossCustomTilingData tiling;

    const gert::StorageShape *anchorStorageShape = context->GetInputShape(0);
    uint32_t batchSize = anchorStorageShape->GetOriginShape().GetDim(0);
    uint32_t featureDim = anchorStorageShape->GetOriginShape().GetDim(1);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = featureDim;
    if (tileLen > maxTile) tileLen = AlignDown(maxTile, ALIGN_ELEMS);
    else tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = ALIGN_ELEMS;

    uint32_t blockDim = batchSize;
    if (blockDim > MAX_BLOCK_DIM) blockDim = AlignDown(MAX_BLOCK_DIM, 1);
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_featureDim(featureDim);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss, uint32_t batchSize, uint32_t featureDim,
                                uint32_t tileLen, float margin) {
        batchSize_ = batchSize;
        featureDim_ = featureDim;
        tileLen_ = tileLen;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, batchSize * featureDim);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, batchSize * featureDim);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, batchSize * featureDim);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, batchSize);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueP_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueN_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(diffQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sqQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process() {
        if (batchSize_ == 0 || featureDim_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        for (uint32_t sample = blockIdx; sample < batchSize_; sample += blockNum) {
            ProcessSample(sample);
        }
    }

private:
    __aicore__ inline void ProcessSample(uint32_t sample) {
        uint32_t baseOffset = sample * featureDim_;

        float sum1 = 0.0f;
        float sum2 = 0.0f;

        uint32_t remaining = featureDim_;
        uint32_t offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            AscendC::DataCopy(aLocal, anchorGm_[baseOffset + offset], curLen);
            AscendC::LocalTensor<float> pLocal = inQueueP_.AllocTensor<float>();
            AscendC::DataCopy(pLocal, positiveGm_[baseOffset + offset], curLen);
            AscendC::LocalTensor<float> nLocal = inQueueN_.AllocTensor<float>();
            AscendC::DataCopy(nLocal, negativeGm_[baseOffset + offset], curLen);

            // diff1 = a - p
            AscendC::LocalTensor<float> diff = diffQueue_.AllocTensor<float>();
            AscendC::Sub(diff, aLocal, pLocal, curLen);
            // sq1 = diff * diff
            AscendC::LocalTensor<float> sq = sqQueue_.AllocTensor<float>();
            AscendC::Mul(sq, diff, diff, curLen);
            // reduce sum1 (manual loop)
            for (uint32_t i = 0; i < curLen; ++i) {
                sum1 += sq(i);
            }

            // diff2 = a - n
            AscendC::Sub(diff, aLocal, nLocal, curLen);
            AscendC::Mul(sq, diff, diff, curLen);
            for (uint32_t i = 0; i < curLen; ++i) {
                sum2 += sq(i);
            }

            diffQueue_.FreeTensor(diff);
            sqQueue_.FreeTensor(sq);
            inQueueA_.FreeTensor(aLocal);
            inQueueP_.FreeTensor(pLocal);
            inQueueN_.FreeTensor(nLocal);

            remaining -= curLen;
            offset += curLen;
        }

        // loss = max(sum1 - sum2 + margin, 0)
        float loss = sum1 - sum2 + margin_;
        if (loss < 0.0f) loss = 0.0f;

        // Write output
        AscendC::LocalTensor<float> outLocal = outQueue_.AllocTensor<float>();
        outLocal(0) = loss;
        AscendC::DataCopy(lossGm_[sample], outLocal, 1);
        outQueue_.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueP_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueN_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> diffQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sqQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t batchSize_ {0};
    uint32_t featureDim_ {0};
    uint32_t tileLen_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(
    GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
    GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.batchSize, tiling_data.featureDim,
            tiling_data.tileLen, tiling_data.margin);
    op.Process();
}
