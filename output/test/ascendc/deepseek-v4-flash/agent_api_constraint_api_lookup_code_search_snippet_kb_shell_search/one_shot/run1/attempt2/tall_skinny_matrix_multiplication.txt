#include "tall_skinny_matrix_multiplication_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  TallSkinnyMatrixMultiplicationCustomTilingData tiling;

  // Fix: GetInputShape returns const StorageShape*, use GetOriginShape() to get const Shape&
  const gert::Shape &shapeA = context->GetInputShape(0)->GetOriginShape();
  const gert::Shape &shapeB = context->GetInputShape(1)->GetOriginShape();
  uint32_t M = static_cast<uint32_t>(shapeA.GetDim(0));
  uint32_t K = static_cast<uint32_t>(shapeA.GetDim(1));
  uint32_t N = static_cast<uint32_t>(shapeB.GetDim(1));

  // Determine the tall dimension and split cores accordingly.
  // For typical tall-skinny, M >> N. We split along M.
  const uint32_t totalCM = M;
  const uint32_t totalCN = N;

  // Heuristic block dimensions: aim for 8-24 cores.
  uint32_t singleCoreM = M;
  uint32_t singleCoreN = N;
  uint32_t blockDim = 1;
  if (M > 8 * N) {
    // Split M across cores; each core handles a chunk of rows.
    blockDim = 8;
    singleCoreM = (M + blockDim - 1) / blockDim;
  } else if (N > 8 * M) {
    blockDim = 8;
    singleCoreN = (N + blockDim - 1) / blockDim;
  } else {
    // Balanced: split both dimensions as needed.
    blockDim = 8;
    singleCoreM = (M + blockDim - 1) / blockDim;
    singleCoreN = 1;
  }
  // Cap block dimension to a reasonable max.
  if (blockDim > 24) blockDim = 24;

  // Recompute block dim if singleCoreM or singleCoreN becomes zero.
  if (singleCoreM == 0) singleCoreM = M;
  if (singleCoreN == 0) singleCoreN = N;

  context->SetBlockDim(blockDim);

  tiling.set_M(M);
  tiling.set_K(K);
  tiling.set_N(N);
  tiling.set_singleCoreM(singleCoreM);
  tiling.set_singleCoreN(singleCoreN);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;  // System workspace handles memory.
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context) {

  const gert::Shape &shapeB = context->GetInputShape(1)->GetOriginShape();

#include "kernel_operator.h"

using namespace AscendC;

template<typename aType, typename bType, typename cType, typename biasType>
class TallSkinnyMatmulKernel {
public:
  __aicore__ inline TallSkinnyMatmulKernel() {}
  __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                              const TCubeTiling &tiling) {
    this->tiling_ = tiling;
    aGm_.SetGlobalBuffer(reinterpret_cast<__gm__ aType *>(a), tiling.M * tiling.K);
    bGm_.SetGlobalBuffer(reinterpret_cast<__gm__ bType *>(b), tiling.K * tiling.N);
    cGm_.SetGlobalBuffer(reinterpret_cast<__gm__ cType *>(c), tiling.M * tiling.N);

    // Compute offset for this block (rows split along M)
    int32_t blockIdx = GetBlockIdx();
    int32_t singleM = tiling.singleCoreM;
    int32_t baseM = blockIdx * singleM;
    if (baseM >= tiling.M) {
      // Block out of range, set pointers to zero-length
      aGm_ = aGm_[0:0];
      bGm_ = bGm_[0:0];
      cGm_ = cGm_[0:0];
      return;
    }
    int32_t mRemain = tiling.M - baseM;
    if (mRemain > singleM) mRemain = singleM;
    aGm_ = aGm_[baseM * tiling.K];
    cGm_ = cGm_[baseM * tiling.N];
    // B matrix is fully visible (no column split)
    bGm_ = bGm_[0];
    localM_ = mRemain;
  }

  __aicore__ inline void Process(TPipe *pipe) {
    if (localM_ == 0) return;
    // Register matmul object and perform computation
    REGIST_MATMUL_OBJ(pipe, GetSysWorkSpacePtr(), matmulObj_, (TCubeTiling*)nullptr);
    matmulObj_.SetTensorA(aGm_);
    matmulObj_.SetTensorB(bGm_);
    matmulObj_.SetBias(bGm_[0]); // dummy bias, ignored because bias pointer is same as b but will not be used if IterateAll is called without bias; safer: use a zero tensor? Actually we can avoid bias by using overload without bias.
    // However, Matmul requires bias. We'll provide a dummy zero vector of size N. For simplicity, we use a local zero buffer, but that requires extra init. Instead, we set bias to a global buffer of zeros allocated in workspace? To avoid complexity, we use the existing bGm_ as bias but it is not correct. A proper solution: define a zero tensor in Init from workspace. Since we do not have workspace for bias, we can pass a null pointer and ensure Matmul object is configured to not use bias. The standard approach: if bias is nullptr, Matmul ignores it. We'll set biasGlobal to a zero-length buffer. In current API, we must provide a tensor. We'll set it to a zero tensor of size N using a separate global buffer of zeros? But that requires extra memory. Instead, we can use the fact that the Matmul implementation can tolerate bias being non-null but not used if we call IterateAll without bias. Let's check the sample: they call SetBias and then IterateAll. The bias parameter in IterateAll is optional? In the API, IterateAll(C) without bias argument uses the previously set bias. So we must set a valid bias. To keep it simple, we allocate a small global buffer in workspace? The host tiling sets workspace to 0, so we cannot rely on that. Another approach: set bias to a global buffer that is null, but the API expects a GlobalTensor. We can create a null GlobalTensor by not calling SetBias. But the constructor requires bias type. We'll set bias to a dummy tensor that points to a valid memory area. Since we are not actually using bias in the computation (matmul without bias in many cases), we can cheat by setting bias to bGm_ and the kernel will still compute correctly because bias is added after matmul? Actually, if bias is present, the matmul output C = A*B + bias. To avoid that, we need bias to be all zeros. We can allocate a zero vector in UB and use it. Let's do that: in Process, allocate a local tensor of zeros and set bias to it. That is efficient.

    // Allocate a zero bias vector in UB
    TBuf<> tmpBuf;
    pipe->InitBuffer(tmpBuf, tiling_.N * sizeof(float));
    LocalTensor<float> biasLocal = tmpBuf.Get<float>(tiling_.N);
    for (int i = 0; i < tiling_.N; i++) {
      biasLocal.SetValue(i, 0.0f);
    }
    matmulObj_.SetBias(biasLocal);

    matmulObj_.IterateAll(cGm_);
    matmulObj_.End();
  }

private:
  Matmul<MatmulType<TPosition::GM, CubeFormat::ND, aType>,
         MatmulType<TPosition::GM, CubeFormat::ND, bType>,
         MatmulType<TPosition::GM, CubeFormat::ND, cType>,
         MatmulType<TPosition::GM, CubeFormat::ND, biasType>>
      matmulObj_;

  GlobalTensor<aType> aGm_;
  GlobalTensor<bType> bGm_;
  GlobalTensor<cType> cGm_;
  TCubeTiling tiling_;
  int32_t localM_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm) {
  if (ASCEND_IS_AIV) return;

  CopyTiling(&tiling, tilingGm);
  TPipe pipe;

  TallSkinnyMatmulKernel<float, float, float, float> kernel;

}
