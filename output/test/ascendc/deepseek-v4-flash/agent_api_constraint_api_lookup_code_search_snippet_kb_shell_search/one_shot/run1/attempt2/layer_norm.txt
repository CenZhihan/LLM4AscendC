    LayerNormCustomTilingData tiling;

    uint32_t totalRows = totalElems / hiddenSize;

    float epsilon = 1e-5f;
    if (attrs != nullptr) {
        const float *epsAttr = attrs->GetAttrPointer<float>(0);
        if (epsAttr != nullptr) epsilon = *epsAttr;

    tiling.set_totalRows(totalRows);
    tiling.set_hiddenSize(hiddenSize);
    tiling.set_epsilon(epsilon);
    tiling.set_invHiddenSize(1.0f / static_cast<float>(hiddenSize));

    // Block dim: one block per row, capped
    uint32_t blockDim = totalRows < 8 ? totalRows : 8;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalRows, uint32_t hiddenSize, float epsilon, float invHiddenSize) {
        totalRows_ = totalRows;
        hiddenSize_ = hiddenSize;
        epsilon_ = epsilon;
        invHiddenSize_ = invHiddenSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows * hiddenSize);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, hiddenSize);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, hiddenSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows * hiddenSize);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, 1, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(tmp1_, 1, hiddenSize_ * sizeof(float)); // diff
        pipe_.InitBuffer(tmp2_, 1, hiddenSize_ * sizeof(float)); // sqr
        pipe_.InitBuffer(scalar_, 1, sizeof(float)); // scalar for mean/variance
    }

    __aicore__ inline void Process() {
        if (totalRows_ == 0 || hiddenSize_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (totalRows_ + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > totalRows_) endRow = totalRows_;

        // Load gamma (weight) and beta (bias) once
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, hiddenSize_);
        inQueueW_.EnQue(wLocal);
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        AscendC::DataCopy(bLocal, bGm_, hiddenSize_);
        inQueueB_.EnQue(bLocal);

        for (uint32_t row = startRow; row < endRow; ++row) {
            uint32_t offset = row * hiddenSize_;

            // Load x row (double buffering)
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], hiddenSize_);

            // Allocate output buffer
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // Temporary tensors
            AscendC::LocalTensor<float> diffLocal = tmp1_.AllocTensor<float>();
            AscendC::LocalTensor<float> sqrLocal = tmp2_.AllocTensor<float>();
            AscendC::LocalTensor<float> scalarLocal = scalar_.AllocTensor<float>();

            // Compute mean = sum(x) / N
            AscendC::ReduceSum(scalarLocal, xLocal, hiddenSize_);
            float mean_val = scalarLocal.GetValue(0) * invHiddenSize_;

            // Compute variance = sum((x - mean)^2) / N
            AscendC::Sub(diffLocal, xLocal, mean_val, hiddenSize_);
            AscendC::Mul(sqrLocal, diffLocal, diffLocal, hiddenSize_);
            AscendC::ReduceSum(scalarLocal, sqrLocal, hiddenSize_);
            float variance = scalarLocal.GetValue(0) * invHiddenSize_;

            // Standard deviation inverse
            float stdInv = 1.0f / sqrt(variance + epsilon_);

            // Normalize: y = (x - mean) * stdInv * weight + bias
            AscendC::Muls(diffLocal, diffLocal, stdInv, hiddenSize_); // (x - mean)/std
            AscendC::Mul(yLocal, diffLocal, wLocal, hiddenSize_);
            AscendC::Add(yLocal, yLocal, bLocal, hiddenSize_);

            // Copy output to global memory
            AscendC::DataCopy(yGm_[offset], yLocal, hiddenSize_);

            // Free tensors
            scalar_.FreeTensor(scalarLocal);
            tmp2_.FreeTensor(sqrLocal);
            tmp1_.FreeTensor(diffLocal);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        // Free gamma & beta tensors
        AscendC::LocalTensor<float> wDeque = inQueueW_.DeQue<float>();
        inQueueW_.FreeTensor(wDeque);
        AscendC::LocalTensor<float> bDeque = inQueueB_.DeQue<float>();
        inQueueB_.FreeTensor(bDeque);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_;
    uint32_t hiddenSize_;
    float epsilon_;
    float invHiddenSize_;
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                        GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalRows, tiling_data.hiddenSize,
            tiling_data.epsilon, tiling_data.invHiddenSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        # Create weight and bias parameters matching the original nn.LayerNorm
        self.weight = torch.nn.Parameter(torch.ones(normalized_shape, dtype=torch.float32))
        self.bias = torch.nn.Parameter(torch.zeros(normalized_shape, dtype=torch.float32))
        self.epsilon = 1e-5

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias, self.epsilon)
