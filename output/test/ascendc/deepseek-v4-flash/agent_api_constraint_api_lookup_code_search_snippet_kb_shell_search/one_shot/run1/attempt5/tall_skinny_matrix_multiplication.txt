#include "tall_skinny_matrix_multiplication_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  TallSkinnyMatrixMultiplicationCustomTilingData tiling;

  // Fix: GetInputShape returns const gert::Shape*, use GetDim directly
  const gert::Shape *shapeA = context->GetInputShape(0);
  const gert::Shape *shapeB = context->GetInputShape(1);
  uint32_t M = static_cast<uint32_t>(shapeA->GetDim(0));
  uint32_t K = static_cast<uint32_t>(shapeA->GetDim(1));
  uint32_t N = static_cast<uint32_t>(shapeB->GetDim(1));

  const uint32_t totalCM = M;
  const uint32_t totalCN = N;

  uint32_t singleCoreM = M;
  uint32_t singleCoreN = N;
  uint32_t blockDim = 1;
  if (M > 8 * N) {
    blockDim = 8;
    singleCoreM = (M + blockDim - 1) / blockDim;
  } else if (N > 8 * M) {
    blockDim = 8;
    singleCoreN = (N + blockDim - 1) / blockDim;
  } else {
    blockDim = 8;
    singleCoreM = (M + blockDim - 1) / blockDim;
    singleCoreN = 1;
  }
  if (blockDim > 24) blockDim = 24;

  if (singleCoreM == 0) singleCoreM = M;
  if (singleCoreN == 0) singleCoreN = N;

  context->SetBlockDim(blockDim);

  tiling.set_M(M);
  tiling.set_K(K);
  tiling.set_N(N);
  tiling.set_singleCoreM(singleCoreM);
  tiling.set_singleCoreN(singleCoreN);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"
#include "matmul_tiling.h"

using namespace AscendC;

template<typename aType, typename bType, typename cType, typename biasType>
class TallSkinnyMatmulKernel {
public:
  __aicore__ inline TallSkinnyMatmulKernel() {}
  __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                              const TCubeTiling &tiling) {
    this->tiling_ = tiling;
    aGm_.SetGlobalBuffer(reinterpret_cast<__gm__ aType *>(a), tiling.M * tiling.K);
    bGm_.SetGlobalBuffer(reinterpret_cast<__gm__ bType *>(b), tiling.K * tiling.N);
    cGm_.SetGlobalBuffer(reinterpret_cast<__gm__ cType *>(c), tiling.M * tiling.N);

    // Compute offset for this block (rows split along M)
    int32_t blockIdx = GetBlockIdx();
    int32_t singleM = tiling.singleCoreM;
    int32_t baseM = blockIdx * singleM;
    if (baseM >= tiling.M) {
      aGm_ = aGm_[0:0];
      bGm_ = bGm_[0:0];
      cGm_ = cGm_[0:0];
      return;
    }
    int32_t mRemain = tiling.M - baseM;
    if (mRemain > singleM) mRemain = singleM;
    aGm_ = aGm_[baseM * tiling.K];
    cGm_ = cGm_[baseM * tiling.N];
    bGm_ = bGm_[0];
    localM_ = mRemain;
  }

  __aicore__ inline void Process(TPipe *pipe) {
    if (localM_ == 0) return;
    REGIST_MATMUL_OBJ(pipe, GetSysWorkSpacePtr(), matmulObj_, (TCubeTiling*)nullptr);
    matmulObj_.SetTensorA(aGm_);
    matmulObj_.SetTensorB(bGm_);
    // Bias: allocate zero local buffer
    TBuf<> tmpBuf;
    pipe->InitBuffer(tmpBuf, tiling_.N * sizeof(float));
    LocalTensor<float> biasLocal = tmpBuf.Get<float>(tiling_.N);
    for (int i = 0; i < tiling_.N; i++) {
      biasLocal.SetValue(i, 0.0f);
    }
    matmulObj_.SetBias(biasLocal);

    matmulObj_.IterateAll(cGm_);
    matmulObj_.End();
  }

private:
  Matmul<MatmulType<TPosition::GM, CubeFormat::ND, aType>,
         MatmulType<TPosition::GM, CubeFormat::ND, bType>,
         MatmulType<TPosition::GM, CubeFormat::ND, cType>,
         MatmulType<TPosition::GM, CubeFormat::ND, biasType>>
      matmulObj_;

  GlobalTensor<aType> aGm_;
  GlobalTensor<bType> bGm_;
  GlobalTensor<cType> cGm_;
  TCubeTiling tiling_;
  int32_t localM_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm) {
  if (ASCEND_IS_AIV) return;

  TCubeTiling tiling;
  CopyTiling(&tiling, tilingGm);
  TPipe pipe;

  TallSkinnyMatmulKernel<float, float, float, float> kernel;
  kernel.Init(a, b, c, tiling);
  kernel.Process(&pipe);
}
