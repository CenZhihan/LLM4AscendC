static ge::graphStatus TilingFunc(gert::TilingContext *context)

    // Retrieve shapes: A (M, K), B (K, N)
    const uint32_t M = context->GetInputShape(0)->GetOriginShape().GetDim(0);
    const uint32_t K = context->GetInputShape(0)->GetOriginShape().GetDim(1);
    const uint32_t N = context->GetInputShape(1)->GetOriginShape().GetDim(1);

    // Set up MultiCoreMatmulTiling for float32
    matmul_tiling::MultiCoreMatmulTiling matmulTiling;
    matmulTiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    matmulTiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    matmulTiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    matmulTiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);

    matmulTiling.SetOrgShape(M, K, N);
    matmulTiling.SetShape(M, K, N);

    context->SetWorkSpaceSizes(1, &workspaceSize);

    return ge::GRAPH_SUCCESS;
}
