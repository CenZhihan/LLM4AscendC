host_operator_src="""

#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include "matmul_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const uint32_t M = context->GetInputShape(0)->GetOriginShape().GetDim(0);
    const uint32_t K = context->GetInputShape(0)->GetOriginShape().GetDim(1);
    const uint32_t N = context->GetInputShape(1)->GetOriginShape().GetDim(1);

    matmul_tiling::MultiCoreMatmulTiling matmulTiling;
    matmulTiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    matmulTiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    matmulTiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    matmulTiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);

    matmulTiling.SetOrgShape(M, K, N);
    matmulTiling.SetShape(M, K, N);
    matmulTiling.SetBias(false);
    matmulTiling.SetBufferSpace(-1, -1, -1);

    TCubeTiling tiling;
    if (matmulTiling.GetTiling(tiling) == -1) {
        return ge::GRAPH_FAILED;
    }

    context->SetBlockDim(tiling.coreNum);

    uint8_t* tilingData = context->GetRawTilingData()->GetData();
    std::memcpy(tilingData, &tiling, sizeof(TCubeTiling));
    context->GetRawTilingData()->SetDataSize(sizeof(TCubeTiling));

    size_t workspaceSize = matmulTiling.GetWorkspaceSize();
    context->SetWorkSpaceSizes(1, &workspaceSize);

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    uint32_t M = a_shape->GetDim(0);
    uint32_t N = b_shape->GetDim(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType in_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, in_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class TallSkinnyMatrixMultiplicationCustom : public OpDef {
public:
    explicit TallSkinnyMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("a")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("b")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("c")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(TallSkinnyMatrixMultiplicationCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include "matmul.h"

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm)
{
    if ASCEND_IS_AIV {
        return;
    }

    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    AscendC::TPipe pipe;

    MatmulKernel<float, float, float, float> matmulKernel;
    REGIST_MATMUL_OBJ(&pipe, (void*)workspace, matmulKernel.matmulObj, &tiling);

    matmulKernel.Init(a, b, nullptr, c, tiling);
    matmulKernel.Process(&pipe);
}

#include <torch/library.h>

at::Tensor tall_skinny_matmul_impl_npu(const at::Tensor& a, const at::Tensor& b) {
    auto M = a.size(0);
    auto N = b.size(1);
    at::Tensor result = at::empty({M, N}, a.options());
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, a, b, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
