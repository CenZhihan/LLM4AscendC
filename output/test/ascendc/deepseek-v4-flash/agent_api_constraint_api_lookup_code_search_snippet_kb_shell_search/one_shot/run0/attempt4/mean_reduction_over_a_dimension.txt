#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: accBuf, rowBuf each of tileLen floats -> 2 * tileLen * 4 bytes = 8 * tileLen
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape &inputShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t ndim = static_cast<uint32_t>(inputShape.GetDimNum());
    // We assume 3D for this implementation; axis=1 is hardcoded.
    if (ndim != 3) return ge::GRAPH_FAILED;

    uint32_t d0 = static_cast<uint32_t>(inputShape.GetDim(0));
    uint32_t d1 = static_cast<uint32_t>(inputShape.GetDim(1));
    uint32_t d2 = static_cast<uint32_t>(inputShape.GetDim(2));
    uint32_t totalLength = d0 * d1 * d2;

    // Read attribute dim (unused, but we enforce axis=1)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;
    // Ensure it's 1 for this kernel
    if (dim != 1) return ge::GRAPH_FAILED;

    // Compute tile length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 8; // 8 bytes per element (2 floats)
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (d2 < tileLen) {
        uint32_t aligned = AlignDown(d2, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : d2;
    }

    // blockDim = number of batches (each block processes one batch)
    // No clamping; the runtime handles serialization if blockDim > physical cores
    uint32_t blockDim = (d0 > 0) ? d0 : 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_batchSize(d0);
    tiling.set_dim1(d1);
    tiling.set_dim2(d2);
    tiling.set_tileLen(tileLen);
    tiling.set_invDim1(1.0f / static_cast<float>(d1));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    // Get dim attribute
    int64_t dim = 1; // default
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs != nullptr) {
        const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
        if (dimPtr != nullptr) dim = *dimPtr;
    }
    uint32_t ndim = static_cast<uint32_t>(x_shape->GetDimNum());
    // Handle negative dimension
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= static_cast<int64_t>(ndim)) return ge::GRAPH_FAILED;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(ndim - 1);
    uint32_t outIdx = 0;
    for (uint32_t i = 0; i < ndim; ++i) {
        if (static_cast<int64_t>(i) == dim) continue;
        y_shape->SetDim(outIdx++, x_shape->GetDim(i));
    }
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{

    context->SetOutputDataType(0, x_dtype);
    return ge::GRAPH_SUCCESS;
}
