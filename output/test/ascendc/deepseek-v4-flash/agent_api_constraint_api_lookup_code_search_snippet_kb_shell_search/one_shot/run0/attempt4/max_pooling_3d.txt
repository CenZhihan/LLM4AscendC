static inline int64_t calc_output_size(int64_t input, int64_t kernel, int64_t stride, int64_t padding, bool ceil_mode)
{
    // floor mode: (input + 2*padding - kernel) / stride + 1
    // ceil mode:  ceil((input + 2*padding - kernel) / stride) + 1
    int64_t numerator = input + 2 * padding - kernel;
    if (ceil_mode) {
        return (numerator + stride - 1) / stride + 1;
    } else {
        return numerator / stride + 1;
    }
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    if (xShape == nullptr) return GRAPH_FAILED;

    auto ndim = xShape->GetDimNum();
    int64_t N = (ndim >= 1) ? xShape->GetDim(0) : 1;
    int64_t C = (ndim >= 2) ? xShape->GetDim(1) : 1;
    int64_t D = (ndim >= 3) ? xShape->GetDim(2) : 1;
    int64_t H = (ndim >= 4) ? xShape->GetDim(3) : 1;
    int64_t W = (ndim >= 5) ? xShape->GetDim(4) : 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    auto getAttrListInt = [&](int idx) -> const int64_t* {
        const gert::TypedContinuousVector<int64_t>* vec = attrs->GetListInt(idx);
        if (vec == nullptr || vec->GetSize() == 0) return nullptr;
        return vec->GetData();
    };

    const int64_t* ksize = getAttrListInt(0);
    int64_t kD = (ksize != nullptr) ? ksize[0] : 3;
    int64_t kH = (ksize != nullptr) ? (ksize[1] ? ksize[1] : kD) : kD;
    int64_t kW = (ksize != nullptr) ? (ksize[2] ? ksize[2] : kD) : kD;

    const int64_t* sAttr = getAttrListInt(1);
    int64_t sD = (sAttr != nullptr) ? sAttr[0] : kD;
    int64_t sH = (sAttr != nullptr) ? (sAttr[1] ? sAttr[1] : sD) : sD;
    int64_t sW = (sAttr != nullptr) ? (sAttr[2] ? sAttr[2] : sD) : sD;

    const int64_t* pAttr = getAttrListInt(2);
    int64_t pD = (pAttr != nullptr) ? pAttr[0] : 0;
    int64_t pH = (pAttr != nullptr) ? (pAttr[1] ? pAttr[1] : pD) : pD;
    int64_t pW = (pAttr != nullptr) ? (pAttr[2] ? pAttr[2] : pD) : pD;

    bool ceil_mode = false;

    auto calc = [](int64_t input, int64_t kernel, int64_t stride, int64_t padding, bool ceil_mode) -> int64_t {
        int64_t numerator = input + 2 * padding - kernel;
        if (ceil_mode) {
            return (numerator + stride - 1) / stride + 1;
        } else {
            return numerator / stride + 1;
        }
    };

    int64_t oD = calc(D, kD, sD, pD, ceil_mode);
    int64_t oH = calc(H, kH, sH, pH, ceil_mode);
    int64_t oW = calc(W, kW, sW, pW, ceil_mode);

    gert::Shape yShape{static_cast<int64_t>(N), static_cast<int64_t>(C), oD, oH, oW};
    context->SetOutputShape(0, yShape);

    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t channels,
                                uint32_t inputD, uint32_t inputH, uint32_t inputW,
                                uint32_t outputD, uint32_t outputH, uint32_t outputW,
                                uint32_t kernelD, uint32_t kernelH, uint32_t kernelW,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t padD, uint32_t padH, uint32_t padW)
    {
        batch_ = batch;
        channels_ = channels;
        inputD_ = inputD;
        inputH_ = inputH;
        inputW_ = inputW;
        outputD_ = outputD;
        outputH_ = outputH;
        outputW_ = outputW;
        kernelD_ = kernelD;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        padD_ = padD;
        padH_ = padH;
        padW_ = padW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        // Grid-stride over output elements (each block handles one output pixel)
        uint32_t totalOut = batch_ * channels_ * outputD_ * outputH_ * outputW_;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t stride = blockNum;
        for (uint32_t idx = blockIdx; idx < totalOut; idx += stride) {
            ComputeOne(idx);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t outIdx)
    {
        // Decode output index to (n, c, od, oh, ow)
        uint32_t ow = outIdx % outputW_;
        uint32_t rem = outIdx / outputW_;
        uint32_t oh = rem % outputH_;
        rem /= outputH_;
        uint32_t od = rem % outputD_;
        rem /= outputD_;
        uint32_t c = rem % channels_;
        uint32_t n = rem / channels_;

        // Compute input window start (with padding)
        int32_t startD = static_cast<int32_t>(od * strideD_) - static_cast<int32_t>(padD_);
        int32_t startH = static_cast<int32_t>(oh * strideH_) - static_cast<int32_t>(padH_);
        int32_t startW = static_cast<int32_t>(ow * strideW_) - static_cast<int32_t>(padW_);

        float maxVal = -FLT_MAX;

        // Iterate over kernel window
        for (uint32_t kd = 0; kd < kernelD_; ++kd) {
            int32_t id = startD + static_cast<int32_t>(kd);
            if (id < 0 || id >= static_cast<int32_t>(inputD_)) continue;
            for (uint32_t kh = 0; kh < kernelH_; ++kh) {
                int32_t ih = startH + static_cast<int32_t>(kh);
                if (ih < 0 || ih >= static_cast<int32_t>(inputH_)) continue;
                for (uint32_t kw = 0; kw < kernelW_; ++kw) {
                    int32_t iw = startW + static_cast<int32_t>(kw);
                    if (iw < 0 || iw >= static_cast<int32_t>(inputW_)) continue;
                    uint32_t inIdx = n * channels_ * inputD_ * inputH_ * inputW_
                                   + c * inputD_ * inputH_ * inputW_
                                   + id * inputH_ * inputW_
                                   + ih * inputW_
                                   + iw;
                    float val = xGm_.GetValue(inIdx);
                    if (val > maxVal) maxVal = val;
                }
            }
        }

        // Write output
        uint32_t outOffset = n * channels_ * outputD_ * outputH_ * outputW_
                           + c * outputD_ * outputH_ * outputW_
                           + od * outputH_ * outputW_
                           + oh * outputW_
                           + ow;
        yGm_.SetValue(outOffset, maxVal);
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t channels_;
    uint32_t inputD_, inputH_, inputW_;
    uint32_t outputD_, outputH_, outputW_;
    uint32_t kernelD_, kernelH_, kernelW_;
    uint32_t strideD_, strideH_, strideW_;
    uint32_t padD_, padH_, padW_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.batch, tiling_data.channels,
            tiling_data.inputD, tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputD, tiling_data.outputH, tiling_data.outputW,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW);
    op.Process();
}
