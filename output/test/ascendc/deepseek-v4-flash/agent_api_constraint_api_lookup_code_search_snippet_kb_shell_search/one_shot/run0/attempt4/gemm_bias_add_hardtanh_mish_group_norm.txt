#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR linBias, GM_ADDR bias,
                                GM_ADDR gnGamma, GM_ADDR gnBeta, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batch, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t numGroups, float epsilon)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batch_ = batch;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        epsilon_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_ * inFeatures_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures_ * inFeatures_);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)linBias, outFeatures_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gnGamma, outFeatures_);
        betaGm_.SetGlobalBuffer((__gm__ float*)gnBeta, outFeatures_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        uint32_t bufSize = tileLen_ / batch_ * outFeatures_; // per sample outFeatures
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inFeatures_ * sizeof(float));
        pipe_.InitBuffer(inQueueWeight_, BUFFER_NUM, outFeatures_ * inFeatures_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, bufSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, bufSize * sizeof(float));
        pipe_.InitBuffer(gnTmpQueue_, 1, 4 * bufSize * sizeof(uint8_t));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            CopyIn(base, curLen);
            Compute(curLen);
            CopyOut(base, curLen);

            base += strideElems;
            if (base >= totalLen_) break;

            uint32_t nextLen = totalLen_ - base;
            if (nextLen > tileLen_) nextLen = tileLen_;
            if (nextLen > 0) CopyIn(base, nextLen);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        uint32_t numSamples = len / outFeatures_;
        uint32_t baseSample = gmOffset / outFeatures_;
        for (uint32_t s = 0; s < numSamples; ++s) {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            uint32_t sampleIdx = baseSample + s;
            AscendC::DataCopy(xLocal, xGm_[sampleIdx * inFeatures_], inFeatures_);
            inQueueX_.EnQue(xLocal);
        }
        if (!weightCopied_) {
            AscendC::LocalTensor<float> wLocal = inQueueWeight_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_, outFeatures_ * inFeatures_);
            inQueueWeight_.EnQue(wLocal);
            weightCopied_ = true;
        }
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueWeight_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Placeholder: copy input to output (real implementation would perform matmul etc.)
        AscendC::DataCopy(yLocal, xLocal, outFeatures_);

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueWeight_.FreeTensor(wLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWeight_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> gnTmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t batch_{0};
    uint32_t inFeatures_{0};
    uint32_t outFeatures_{0};
    uint32_t numGroups_{0};
    float epsilon_{1e-5f};
    bool weightCopied_{false};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR linBias, GM_ADDR bias,
    GM_ADDR gnGamma, GM_ADDR gnBeta, GM_ADDR y,

{
    GET_TILING_DATA(tiling_data, tiling);

            tiling_data.numGroups, tiling_data.epsilon);
    op.Process();
}
