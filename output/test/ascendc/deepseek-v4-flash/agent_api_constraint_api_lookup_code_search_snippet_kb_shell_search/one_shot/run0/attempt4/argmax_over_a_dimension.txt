host_operator_src="""
#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t HARD_CAP_OUTER = 1024;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : 0L;

    int64_t ndim = static_cast<int64_t>(inputShape->GetDimNum());
    if (dim < 0) dim += ndim;
    if (dim < 0) dim = 0;
    if (dim >= ndim) dim = ndim - 1;

    uint32_t totalOuter = 1;
    uint32_t reductionSize = 1;
    for (int64_t i = 0; i < ndim; ++i) {
        uint32_t s = static_cast<uint32_t>(inputShape->GetDim(i));
        if (i == dim) {
            reductionSize = s;
        } else {
            totalOuter *= s;
        }
    }

    // Each outer element processes reduction in blocks of tileReduce.
    // UB budget: need one block buffer (tileReduce floats) + small scalar buffer (2 floats) + overhead.
    uint32_t tileReduce = UB_BUDGET_BYTES / (4 * sizeof(float));  // conservative
    if (tileReduce > reductionSize) tileReduce = reductionSize;
    if (tileReduce < 64) tileReduce = 64;  // minimum for DMA efficiency
    // Align to 32 bytes
    tileReduce = (tileReduce + 7) / 8 * 8;

    // Tile outer dimension to balance workload
    uint32_t tileOuter = (totalOuter < HARD_CAP_OUTER) ? totalOuter : HARD_CAP_OUTER;
    if (tileOuter == 0) tileOuter = 1;

    uint32_t blockDim = CeilDiv(totalOuter, tileOuter);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalOuter(totalOuter);
    tiling.set_reductionSize(reductionSize);
    tiling.set_tileOuter(tileOuter);
    tiling.set_tileReduce(tileReduce);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    int64_t ndim = x_shape->GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : 0L;
    if (dim < 0) dim += ndim;
    dim = (dim < 0) ? 0 : (dim >= ndim ? ndim - 1 : dim);

    std::vector<int64_t> outDims;
    for (int64_t i = 0; i < ndim; ++i) {
        if (i != dim) {
            outDims.push_back(x_shape->GetDim(i));
        }
    }
    gert::Shape y_shape;
    y_shape.SetDimNum(static_cast<uint64_t>(outDims.size()));
    for (size_t i = 0; i < outDims.size(); ++i) {
        y_shape.SetDim(i, outDims[i]);
    }
    *context->GetOutputShape(0) = y_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class ArgmaxOverADimensionCustom : public OpDef {
public:
    explicit ArgmaxOverADimensionCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)

            .ParamType(REQUIRED)
            .DataType({ge::DT_INT64})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOuter, uint32_t reductionSize,
                                uint32_t tileOuter, uint32_t tileReduce)
    {
        totalOuter_ = totalOuter;
        reductionSize_ = reductionSize;
        tileOuter_ = tileOuter;
        tileReduce_ = tileReduce;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalOuter_ * reductionSize_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalOuter_);

        // Buffers for a single block of reduction dimension
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileReduce_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(int64_t));

        // Small buffer for running best (value, index)
        pipe_.InitBuffer(bestBuf_, 2 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOuter_ == 0 || reductionSize_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t idx = blockIdx;
        while (idx < totalOuter_) {
            ProcessOuter(idx);
            idx += blockNum;
        }
    }

private:
    __aicore__ inline void ProcessOuter(uint32_t outerIdx)
    {
        // Initialize best buffer: bestVal = -FLT_MAX, bestIdx = 0
        AscendC::LocalTensor<float> best = bestBuf_.Get<float>();
        best.SetValue(0, -__builtin_huge_valf());
        best.SetValue(1, 0.0f);

        uint32_t remaining = reductionSize_;
        uint32_t offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileReduce_) ? tileReduce_ : remaining;
            // Copy block
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            uint32_t gmBase = outerIdx * reductionSize_ + offset;
            AscendC::DataCopy(xLocal, xGm_[gmBase], curLen);
            inQueueX_.EnQue(xLocal);
            // Compute local argmax and update best
            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
            UpdateBest(best, xDeque, curLen, offset);
            inQueueX_.FreeTensor(xDeque);
            offset += curLen;
            remaining -= curLen;
        }

        // Write final index
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.AllocTensor<int64_t>();
        float bestIdxFloat = best.GetValue(1);
        yLocal.SetValue(0, static_cast<int64_t>(bestIdxFloat));
        outQueueY_.EnQue<int64_t>(yLocal);

        // Copy out
        AscendC::LocalTensor<int64_t> yDeque = outQueueY_.DeQue<int64_t>();
        AscendC::DataCopy(yGm_[outerIdx], yDeque, 1);
        outQueueY_.FreeTensor(yDeque);
    }

    __aicore__ inline void UpdateBest(AscendC::LocalTensor<float>& best,
                                      AscendC::LocalTensor<float>& block,
                                      uint32_t len, uint32_t baseOffset)
    {
        // Simple linear scan: compare each element with current best
        for (uint32_t i = 0; i < len; ++i) {
            float val = block.GetValue(i);
            float curBest = best.GetValue(0);
            if (val > curBest) {
                best.SetValue(0, val);
                best.SetValue(1, static_cast<float>(baseOffset + i));
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bestBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalOuter_ {0};
    uint32_t reductionSize_ {0};
    uint32_t tileOuter_ {0};
    uint32_t tileReduce_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalOuter, tiling_data.reductionSize,
            tiling_data.tileOuter, tiling_data.tileReduce);
    op.Process();
}
