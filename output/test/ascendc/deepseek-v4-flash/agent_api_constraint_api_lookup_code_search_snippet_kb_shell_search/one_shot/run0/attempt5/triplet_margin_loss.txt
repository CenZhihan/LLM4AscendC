kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t n_, uint32_t d_, float margin_, float invN_)
    {
        n = n_;
        d = d_;
        margin = margin_;
        invN = invN_;
        totalElems = n * d;

        anchorGm.SetGlobalBuffer((__gm__ float*)anchor, totalElems);
        positiveGm.SetGlobalBuffer((__gm__ float*)positive, totalElems);
        negativeGm.SetGlobalBuffer((__gm__ float*)negative, totalElems);
        lossGm.SetGlobalBuffer((__gm__ float*)loss, 1);

        // Initialize UB buffers: we process one dimension (d) at a time.
        pipe.InitBuffer(inQueueAnchor, BUFFER_NUM, d * sizeof(float));
        pipe.InitBuffer(inQueuePositive, BUFFER_NUM, d * sizeof(float));
        pipe.InitBuffer(inQueueNegative, BUFFER_NUM, d * sizeof(float));
        pipe.InitBuffer(tmpQueue, 1, d * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n == 0 || d == 0) return;

        float totalLoss = 0.0f;

        for (uint32_t i = 0; i < n; ++i) {
            // Copy one triplet
            AscendC::LocalTensor<float> anchorLocal = inQueueAnchor.AllocTensor<float>();
            AscendC::LocalTensor<float> positiveLocal = inQueuePositive.AllocTensor<float>();
            AscendC::LocalTensor<float> negativeLocal = inQueueNegative.AllocTensor<float>();
            AscendC::DataCopy(anchorLocal, anchorGm[i * d], d);
            AscendC::DataCopy(positiveLocal, positiveGm[i * d], d);
            AscendC::DataCopy(negativeLocal, negativeGm[i * d], d);

            AscendC::LocalTensor<float> tmp = tmpQueue.AllocTensor<float>();

            // dap = sum((anchor - positive)^2)
            AscendC::Sub(tmp, anchorLocal, positiveLocal, d);
            AscendC::Mul(tmp, tmp, tmp, d);
            AscendC::SetMaskCount();
            AscendC::SetVectorMask<float>(0, d);
            AscendC::WholeReduceSum<float, false>(tmp, tmp, AscendC::MASK_PLACEHOLDER, 1, 1, 1, 8);
            AscendC::PipeBarrier<PIPE_V>();
            AscendC::ResetMask();
            AscendC::SetMaskNorm();
            float dap = tmp.GetValue(0);

            // dan = sum((anchor - negative)^2)
            AscendC::Sub(tmp, anchorLocal, negativeLocal, d);
            AscendC::Mul(tmp, tmp, tmp, d);
            AscendC::SetMaskCount();
            AscendC::SetVectorMask<float>(0, d);
            AscendC::WholeReduceSum<float, false>(tmp, tmp, AscendC::MASK_PLACEHOLDER, 1, 1, 1, 8);
            AscendC::PipeBarrier<PIPE_V>();
            AscendC::ResetMask();
            AscendC::SetMaskNorm();
            float dan = tmp.GetValue(0);

            float loss_i = dap - dan + margin;
            if (loss_i < 0.0f) loss_i = 0.0f;
            totalLoss += loss_i;

            tmpQueue.FreeTensor(tmp);
            inQueueAnchor.FreeTensor(anchorLocal);
            inQueuePositive.FreeTensor(positiveLocal);
            inQueueNegative.FreeTensor(negativeLocal);
        }

        float meanLoss = totalLoss * invN;

        // Write result using single-element DataCopy
        AscendC::LocalTensor<float> outLocal = tmpQueue.AllocTensor<float>();
        outLocal.SetValue(0, meanLoss);
        AscendC::DataCopy(lossGm, outLocal, 1);
        tmpQueue.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePositive;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNegative;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue;

    AscendC::GlobalTensor<float> anchorGm;
    AscendC::GlobalTensor<float> positiveGm;
    AscendC::GlobalTensor<float> negativeGm;
    AscendC::GlobalTensor<float> lossGm;

    uint32_t n;
    uint32_t d;
    uint32_t totalElems;
    float margin;
    float invN;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                  GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.n, tiling_data.d, tiling_data.margin, tiling_data.invN);
    op.Process();
}
