kernel_src="""

#include "kernel_operator.h"
#include "adv_api/math/tanh.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class VanillaRnnHiddenCustom {
public:
    __aicore__ inline VanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x_t, GM_ADDR h_in, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                 GM_ADDR w_h2o, GM_ADDR b_h2o,
                                 GM_ADDR h_out, GM_ADDR y_t,
                                 const VanillaRnnHiddenCustomTilingData &tiling)
    {
        B_ = tiling.B; I_ = tiling.I; H_ = tiling.H; O_ = tiling.O; K_ = tiling.K;

        xGm_.SetGlobalBuffer((__gm__ float*)x_t, B_ * I_);
        hGmIn_.SetGlobalBuffer((__gm__ float*)h_in, B_ * H_);
        w1Gm_.SetGlobalBuffer((__gm__ float*)w_i2h, H_ * K_);
        b1Gm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);
        w2Gm_.SetGlobalBuffer((__gm__ float*)w_h2o, O_ * H_);
        b2Gm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);
        hOutGm_.SetGlobalBuffer((__gm__ float*)h_out, B_ * H_);
        yGm_.SetGlobalBuffer((__gm__ float*)y_t, B_ * O_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, I_ * sizeof(float));
        pipe_.InitBuffer(inQueueH_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, (K_ > H_ ? K_ : H_) * sizeof(float));
        pipe_.InitBuffer(outQueueH_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, O_ * sizeof(float));
        pipe_.InitBuffer(tanhBuf_, H_ * sizeof(uint8_t) * 2);
    }

    __aicore__ inline void Process()
    {
        if (B_ == 0) return;

        for (uint32_t b = 0; b < B_; b++) {
            LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            LocalTensor<float> hLocal = inQueueH_.AllocTensor<float>();
            DataCopy(xLocal, xGm_[b * I_], I_);
            DataCopy(hLocal, hGmIn_[b * H_], H_);
            inQueueX_.EnQue(xLocal);
            inQueueH_.EnQue(hLocal);

            LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
            LocalTensor<float> hDeq = inQueueH_.DeQue<float>();
            LocalTensor<float> inp = tmpQueue_.AllocTensor<float>();
            DataCopy(inp, xDeq, I_);
            DataCopy(inp[I_], hDeq, H_);
            inQueueX_.FreeTensor(xDeq);
            inQueueH_.FreeTensor(hDeq);

            LocalTensor<float> pre = outQueueH_.AllocTensor<float>();
            for (uint32_t i = 0; i < H_; i++) {
                LocalTensor<float> wRow = tmpQueue_.AllocTensor<float>();
                DataCopy(wRow, w1Gm_[i * K_], K_);
                float dot = 0.0f;
                for (uint32_t j = 0; j < K_; j++) {
                    dot += inp[j] * wRow[j];
                }
                pre[i] = dot + b1Gm_[i];
                tmpQueue_.FreeTensor(wRow);
            }
            tmpQueue_.FreeTensor(inp);
            outQueueH_.EnQue<float>(pre);

            LocalTensor<float> hNewLocal = outQueueH_.DeQue<float>();
            LocalTensor<uint8_t> tanhTmp = tanhBuf_.Get<uint8_t>();
            Tanh(hNewLocal, hNewLocal, tanhTmp);
            DataCopy(hOutGm_[b * H_], hNewLocal, H_);

            LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            for (uint32_t o = 0; o < O_; o++) {
                LocalTensor<float> wRow = tmpQueue_.AllocTensor<float>();
                DataCopy(wRow, w2Gm_[o * H_], H_);
                float dot = 0.0f;
                for (uint32_t j = 0; j < H_; j++) {
                    dot += hNewLocal[j] * wRow[j];
                }
                yLocal[o] = dot + b2Gm_[o];
                tmpQueue_.FreeTensor(wRow);
            }
            outQueueY_.EnQue<float>(yLocal);
            outQueueH_.FreeTensor(hNewLocal);

            LocalTensor<float> yDeq = outQueueY_.DeQue<float>();
            DataCopy(yGm_[b * O_], yDeq, O_);
            outQueueY_.FreeTensor(yDeq);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueH_;
    TQue<TPosition::VECIN, 1> tmpQueue_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueH_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TBuf<TPosition::VECCALC> tanhBuf_;

    GlobalTensor<float> xGm_, hGmIn_, w1Gm_, b1Gm_, w2Gm_, b2Gm_, hOutGm_, yGm_;
    uint32_t B_, I_, H_, O_, K_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x_t, GM_ADDR h, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                 GM_ADDR w_h2o, GM_ADDR b_h2o,
                                                                 GM_ADDR h_out, GM_ADDR y_t,
                                                                 GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    VanillaRnnHiddenCustom op;
    op.Init(x_t, h, w_i2h, b_i2h, w_h2o, b_h2o, h_out, y_t, tiling_data);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'vanilla_rnn_hidden_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, x, h0, w_i2h, b_i2h, w_h2o, b_h2o):
        T, B, I = x.shape
        H = h0.shape[1]
        O = w_h2o.shape[0]
        h = h0
        outs = []
        custom_fn = getattr(custom_ops_lib, _FN)
        for t in range(T):
            x_t = x[t]  # (B, I)
            h, y_t = custom_fn(x_t, h, w_i2h, b_i2h, w_h2o, b_h2o)
            outs.append(y_t)
        return torch.stack(outs, dim=0)
'''
