kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr float INF_NEG = -1e38f;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t height, uint32_t width) {
        height_ = height;
        width_ = width;

        xGm_.SetGlobalBuffer((__gm__ float*)x, height_ * width_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, height_ * width_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, width_ * sizeof(float));
        pipe_.InitBuffer(workspaceQueue_, 1, width_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, width_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        for (uint32_t row = blockIdx; row < height_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row) {
        uint32_t base = row * width_;
        const uint32_t tileSize = (192 * 1024) / (2 * sizeof(float));
        uint32_t numTiles = (width_ + tileSize - 1) / tileSize;

        float globalMax = INF_NEG;
        // Pass 1: compute global max
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = t * tileSize;
            uint32_t len = (offset + tileSize <= width_) ? tileSize : (width_ - offset);

            LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            DataCopy(xLocal, xGm_[base + offset], len);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            LocalTensor<float> workspace = workspaceQueue_.AllocTensor<float>();
            LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
            ReduceMax(maxLocal, xLocal, workspace, len);
            float localMax = maxLocal.GetValue(0);
            if (localMax > globalMax) globalMax = localMax;

            tmpQueue_.FreeTensor(maxLocal);
            workspaceQueue_.FreeTensor(workspace);
            inQueueX_.FreeTensor(xLocal);
        }

        // Pass 2: compute sum of exp(x - globalMax)
        float sum = 0.0f;
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = t * tileSize;
            uint32_t len = (offset + tileSize <= width_) ? tileSize : (width_ - offset);

            LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            DataCopy(xLocal, xGm_[base + offset], len);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            LocalTensor<float> workspace = workspaceQueue_.AllocTensor<float>();
            LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
            Sub(workspace, xLocal, globalMax, len);
            Exp(expLocal, workspace, len);

            LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
            ReduceSum(sumLocal, expLocal, workspace, len);
            sum += sumLocal.GetValue(0);

            tmpQueue_.FreeTensor(sumLocal);
            tmpQueue_.FreeTensor(expLocal);
            workspaceQueue_.FreeTensor(workspace);
            inQueueX_.FreeTensor(xLocal);
        }

        float logSum = logf(sum);

        // Pass 3: compute y = (x - globalMax) - logSum and copy out
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = t * tileSize;
            uint32_t len = (offset + tileSize <= width_) ? tileSize : (width_ - offset);

            LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            DataCopy(xLocal, xGm_[base + offset], len);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            LocalTensor<float> yLocal = tmpQueue_.AllocTensor<float>();
            Sub(yLocal, xLocal, globalMax, len);
            Sub(yLocal, yLocal, logSum, len);

            DataCopy(yGm_[base + offset], yLocal, len);

            tmpQueue_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    TPipe pipe_;
    TQue<QuePosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<QuePosition::VECIN, 1> workspaceQueue_;
    TQue<QuePosition::VECIN, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t height_ {0};
    uint32_t width_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.height, tiling_data.width);
    op.Process();
}
