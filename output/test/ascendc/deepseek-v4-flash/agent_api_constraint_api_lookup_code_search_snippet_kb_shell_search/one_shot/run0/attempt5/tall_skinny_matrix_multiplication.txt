host_operator_src="""
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/matmul/matmul_tiling.h"
#include <cstdint>

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Get input shapes (dimensions are logical)
    auto a_shape = context->GetInputShape(0);
    auto b_shape = context->GetInputShape(1);
    uint32_t M = static_cast<uint32_t>(a_shape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(a_shape->GetDim(1));
    uint32_t N = static_cast<uint32_t>(b_shape->GetDim(1));

    // Determine blockDim based on problem size (tall-skinny heuristics)
    uint32_t blockDim = 1;
    if (M >= 8192 && N <= 64) {
        blockDim = 8;
    } else if (M >= 4096) {
        blockDim = 4;
    } else if (M >= 1024) {
        blockDim = 2;
    }
    context->SetBlockDim(blockDim);

    // Generate matmul tiling using the official API
    matmul_tiling::MatmulApiTiling tilingGen;
    tilingGen.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT, false);
    tilingGen.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT, false);
    tilingGen.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);
    tilingGen.SetOrgShape(M, N, K);
    tilingGen.SetShape(M, N, K);
    tilingGen.SetBias(false);
    tilingGen.SetBufferSpace(-1, -1, -1);

    // Get the base tiling data
    optiling::TCubeTiling cubeTiling;
    if (tilingGen.GetTiling(cubeTiling) != 0) {
        return ge::GRAPH_FAILED;
    }

    // Save to raw buffer
    auto rawTiling = context->GetRawTilingData();
    uint8_t *data = static_cast<uint8_t*>(rawTiling->GetData());
    cubeTiling.SaveToBuffer(data, rawTiling->GetCapacity());
    rawTiling->SetDataSize(cubeTiling.GetDataSize());

    // No workspace needed for this matmul (workspace handled by kernel)
    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    auto a_shape = context->GetInputShape(0);
    auto b_shape = context->GetInputShape(1);
    uint32_t M = static_cast<uint32_t>(a_shape->GetDim(0));
    uint32_t N = static_cast<uint32_t>(b_shape->GetDim(1));
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class TallSkinnyMatrixMultiplicationCustom : public OpDef {
public:
    explicit TallSkinnyMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("a")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("b")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("c")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(TallSkinnyMatrixMultiplicationCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace,
                                const TCubeTiling &tiling)
    {
        matmulObj.Init(a, b, nullptr, c, workspace, tiling);
    }

    __aicore__ inline void Process()
    {
        matmulObj.Process(&pipe_);
    }

private:
    TPipe pipe_;
    MatmulKernel<float, float, float, float> matmulObj;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c,
    GM_ADDR workspace, GM_ADDR tilingGm)
{
    if (ASCEND_IS_AIV) {
        return;
    }

    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);
    KernelTallSkinnyMatMul op;
    op.Init(a, b, c, workspace, tiling);
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor tall_skinny_matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    int64_t M = A.size(0);
    int64_t N = B.size(1);
    at::Tensor result = at::empty({M, N}, A.options());
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, A, B, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu,
          "Tall-skinny matrix multiplication (A x B) on NPU");
}
