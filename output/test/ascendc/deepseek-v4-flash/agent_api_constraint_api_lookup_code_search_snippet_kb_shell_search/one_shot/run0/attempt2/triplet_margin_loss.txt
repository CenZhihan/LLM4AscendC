kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t n_, uint32_t d_, float margin_)
    {
        n = n_;
        d = d_;
        margin = margin_;
        totalElems = n * d;

        anchorGm.SetGlobalBuffer((__gm__ float*)anchor, totalElems);
        positiveGm.SetGlobalBuffer((__gm__ float*)positive, totalElems);
        negativeGm.SetGlobalBuffer((__gm__ float*)negative, totalElems);

        // output: single float
        lossGm.SetGlobalBuffer((__gm__ float*)loss, 1);

        // Double buffer for input chunks
        pipe.InitBuffer(inQueueAnchor, BUFFER_NUM, totalElems * sizeof(float));
        pipe.InitBuffer(inQueuePositive, BUFFER_NUM, totalElems * sizeof(float));
        pipe.InitBuffer(inQueueNegative, BUFFER_NUM, totalElems * sizeof(float));
        // Temporary for per-sample squared distances (d floats)
        pipe.InitBuffer(tmpQueue, 1, d * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n == 0 || d == 0) return;

        // Copy all inputs (single tile, fits UB)
        CopyIn();

        float totalLoss = 0.0f;

        // Local tensors for the three inputs
        AscendC::LocalTensor<float> anchorLocal = inQueueAnchor.DeQue<float>();
        AscendC::LocalTensor<float> positiveLocal = inQueuePositive.DeQue<float>();
        AscendC::LocalTensor<float> negativeLocal = inQueueNegative.DeQue<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue.AllocTensor<float>();

        // Process each sample
        for (uint32_t i = 0; i < n; ++i) {
            // Compute squared distance: anchor - positive
            AscendC::Sub(tmp, anchorLocal[i * d], positiveLocal[i * d], d);
            AscendC::Mul(tmp, tmp, tmp, d);
            // Reduce sum to get dist_ap^2
            AscendC::LocalTensor<float> distAp = tmp;  // reuse first element as reduction destination
            // We need workspace for ReduceSum; use tmp as workspace (it's large enough)
            AscendC::ReduceSum<float>(distAp, tmp, tmp, d);  // reduces tmp of length d -> first element
            // Now distAp[0] holds sum of squares
            AscendC::LocalTensor<float> distAn = tmp;  // reuse
            // Anchor - negative
            AscendC::Sub(tmp, anchorLocal[i * d], negativeLocal[i * d], d);
            AscendC::Mul(tmp, tmp, tmp, d);
            // Wait for previous reduction to complete? We need to ensure distAp value is captured before overwriting tmp.
            // Use a scalar variable
            float dap = distAp.GetValue(0);
            // Now perform reduction for dist_an after reusing tmp
            AscendC::ReduceSum<float>(distAn, tmp, tmp, d);
            float dan = distAn.GetValue(0);

            float loss_i = dap - dan + margin;
            if (loss_i < 0.0f) loss_i = 0.0f;
            totalLoss += loss_i;
        }

        // Compute mean
        float meanLoss = totalLoss / static_cast<float>(n);

        // Write to global output
        AscendC::LocalTensor<float> outLocal = tmp;  // reuse tmp for output
        outLocal.SetValue(0, meanLoss);
        AscendC::DataCopy(lossGm, outLocal, 1);

        // Free resources
        tmpQueue.FreeTensor(tmp);
        inQueueAnchor.FreeTensor(anchorLocal);
        inQueuePositive.FreeTensor(positiveLocal);
        inQueueNegative.FreeTensor(negativeLocal);
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<float> aLocal = inQueueAnchor.AllocTensor<float>();
        AscendC::LocalTensor<float> pLocal = inQueuePositive.AllocTensor<float>();
        AscendC::LocalTensor<float> nLocal = inQueueNegative.AllocTensor<float>();
        AscendC::DataCopy(aLocal, anchorGm, totalElems);
        AscendC::DataCopy(pLocal, positiveGm, totalElems);
        AscendC::DataCopy(nLocal, negativeGm, totalElems);
        inQueueAnchor.EnQue(aLocal);
        inQueuePositive.EnQue(pLocal);
        inQueueNegative.EnQue(nLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePositive;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNegative;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue;

    AscendC::GlobalTensor<float> anchorGm;
    AscendC::GlobalTensor<float> positiveGm;
    AscendC::GlobalTensor<float> negativeGm;
    AscendC::GlobalTensor<float> lossGm;

    uint32_t n;
    uint32_t d;
    uint32_t totalElems;
    float margin;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                  GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.n, tiling_data.d, tiling_data.margin);
    op.Process();
}
