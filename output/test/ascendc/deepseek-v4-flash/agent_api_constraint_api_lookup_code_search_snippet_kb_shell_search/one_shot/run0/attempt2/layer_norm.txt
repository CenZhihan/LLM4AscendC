#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: 2 input tiles (x and weight+bias combined?) Actually we need local buffers:
// For first pass: xLocal, sumLocal (1 float per tile? We accumulate in registers, so minimal)
// For second pass: xLocal, yLocal, weightLocal, biasLocal, plus temporary.
// Estimate: 4 tiles * tileLen * sizeof(float) = 4 * tileLen * 4 = 16*tileLen bytes.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    const gert::StorageShape *shape = context->GetInputShape(0);
    const gert::Shape &originShape = shape->GetOriginShape();
    // Input x: shape = [batch_size, features, dim1, dim2]
    // The normalized_shape is [features, dim1, dim2]
    uint32_t batchSize = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t cols = 1;
    for (size_t i = 1; i < originShape.GetDimNum(); ++i) {
        cols *= static_cast<uint32_t>(originShape.GetDim(i));
    }
    uint32_t totalLength = batchSize * cols;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    float eps = (epsPtr != nullptr) ? *epsPtr : 1e-5f;

    // tileLen from UB budget: maximize but fit 4 tiles of float
    // We need at least cols to be covered, but we divide into tiles.
    // Budget: 4 * tileLen * 4 = 16*tileLen bytes.
    uint32_t maxTile = UB_BUDGET_BYTES / 16; // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (cols < maxTile) ? cols : maxTile;
    // Ensure tileLen divides cols? Not necessary but helps alignment.
    // We'll just use tileLen as is.

    // blockDim adaptive
    uint32_t blockDim = batchSize; // ideally one row per block
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24; // limit

    context->SetBlockDim(blockDim);

    tiling.set_rows(batchSize);
    tiling.set_cols(cols);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // no workspace needed
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t rows, uint32_t cols, uint32_t tileLen, float eps)
    {
        rows_ = rows;
        cols_ = cols;
        tileLen_ = tileLen;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, rows_ * cols_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, cols_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, cols_);

        // Buffers: x, weight, bias, y, temporary
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes rows in a grid-stride loop
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        // First pass: compute mean and variance for this row
        float sum = 0.0f;
        float sumSq = 0.0f;
        uint32_t offset = row * cols_;

        uint32_t remaining = cols_;
        uint32_t base = 0;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

        while (remaining > 0) {
            uint32_t len = (remaining > tileLen_) ? tileLen_ : remaining;

            // Load x tile
            AscendC::DataCopy(xLocal, xGm_[offset + base], len);

            // Manual sum and sumSq using scalar loop (avoids ReduceSum complexity)
            float tileSum = 0.0f;
            float tileSumSq = 0.0f;
            for (uint32_t i = 0; i < len; i++) {
                float val = xLocal(i);
                tileSum += val;
                tileSumSq += val * val;
            }
            sum += tileSum;
            sumSq += tileSumSq;

            base += len;
            remaining -= len;
        }

        float cols_f = cols_;
        float mean = sum / cols_f;
        float var = sumSq / cols_f - mean * mean;
        float invStd = 1.0f / sqrt(var + eps_);

        // Second pass: normalize and apply weight and bias
        remaining = cols_;
        base = 0;
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        while (remaining > 0) {
            uint32_t len = (remaining > tileLen_) ? tileLen_ : remaining;

            // Load x, weight, bias
            AscendC::DataCopy(xLocal, xGm_[offset + base], len);
            AscendC::DataCopy(wLocal, weightGm_[base], len);
            AscendC::DataCopy(bLocal, biasGm_[base], len);

            // y = (x - mean) * invStd * weight + bias
            AscendC::Adds(tmp, xLocal, -mean, len);          // x - mean
            AscendC::Muls(tmp, tmp, invStd, len);            // (x-mean)/std
            AscendC::Mul(tmp, tmp, wLocal, len);             // * weight
            AscendC::Add(yLocal, tmp, bLocal, len);          // + bias

            AscendC::DataCopy(yGm_[offset + base], yLocal, len);

            base += len;
            remaining -= len;
        }

        // Free all allocated tensors for this row
        inQueueX_.FreeTensor(xLocal);
        inQueueW_.FreeTensor(wLocal);
        inQueueB_.FreeTensor(bLocal);
        outQueueY_.FreeTensor(yLocal);
        tmpQueue_.FreeTensor(tmp);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, y, weight, bias, tiling_data.rows, tiling_data.cols, tiling_data.tileLen, tiling_data.eps);
    op.Process();
}
