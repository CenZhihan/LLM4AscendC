#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const uint32_t M = context->GetInputShape(0)->GetOriginShape().GetDim(0);
    const uint32_t K = context->GetInputShape(0)->GetOriginShape().GetDim(1);
    const uint32_t N = context->GetInputShape(1)->GetOriginShape().GetDim(1);

    // For tall-skinny matmul, we tile over the M dimension.
    // UB budget: we need space for a tile of A (tileM x K) and the full B matrix (K x N)
    // Both are floats: tileM * K + K * N floats. Rough budget: 192KB -> 49152 floats.
    // K = N (since B is N x M, but N here is the second dim of B = M) -> careful.
    // Actually, B shape is (K, N_in) where K = B.shape[0] = A.shape[1] = K, and N_in = B.shape[1] = M.
    // So we store B as K x M in UB, that is K*M floats.
    // Tile of A: tileM * K floats.
    // Total: (tileM + M) * K floats <= ~49152.
    // Since M ~32768, K ~32, we have (tileM + 32768)*32 <= 49152 => tileM <= (49152/32)-32768 = 1536-32768 negative.
    // So we cannot fit entire B in UB for this huge M. We need to tile both A and B over the outer dimension.
    // We'll adopt a 2D tiling: tile over M and over N (which is M as well), so we use small tiles.
    // To simplify, use a fixed tileM = 128, tileN = 128, and process C in blocks.
    // We'll modify the kernel accordingly.

    // For simplicity, we set tileM = 128, tileN = 128, and blockDim based on total work.
    constexpr uint32_t TILE_M = 128;
    constexpr uint32_t TILE_N = 128;
    uint32_t blockDim = 8;  // fixed small number of cores

    context->SetBlockDim(blockDim);

    TallSkinnyMatrixMultiplicationCustomTilingData tiling;
    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(TILE_M);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class TallSkinnyMatmulKernel {
public:
    __aicore__ inline TallSkinnyMatmulKernel() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM)
    {
        M_ = M;
        K_ = K;
        N_ = N;  // N = M (output second dimension)
        tileM_ = tileM;

        aGm_.SetGlobalBuffer((__gm__ float*)a, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, K_ * N_);   // B is K x N
        cGm_.SetGlobalBuffer((__gm__ float*)c, M_ * N_);

        // UB buffers: for a tile of A (tileM x K) and B submatrix (K x tileN)
        // We also need accumulator for output tile (tileM x tileN)
        // But we'll compute C row by row to simplify.
        // Allocate for one row of A (K floats) and one column of B (K floats)
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, K_ * sizeof(float));     // one row of A
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, K_ * sizeof(float));     // one column of B
        pipe_.InitBuffer(accQueue_, 1, sizeof(float));                   // accumulator
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();

        // Each block processes rows in a strided loop
        const uint32_t stride = blockNum * 1; // one row per iteration
        for (uint32_t row = blockIdx; row < M_; row += stride) {
            // Load row of A: A[row][k]
            AscendC::LocalTensor<float> aRow = inQueueA_.AllocTensor<float>();
            AscendC::DataCopy(aRow, aGm_[row * K_], K_);
            inQueueA_.EnQue(aRow);

            // For each column of C (output), compute dot product
            for (uint32_t col = 0; col < N_; col++) {
                // Load column of B: B[k][col]
                AscendC::LocalTensor<float> bCol = inQueueB_.AllocTensor<float>();
                // B is stored as K x N, so column col corresponds to B[0][col] at offset col, B[1][col] at offset N + col, etc.
                // Use a strided copy? Not directly. We'll load each element individually.
                // Instead, we can preload the entire B into a local tensor? That would be too large.
                // To keep it simple, we load one element at a time and do multiply-add.
                // This is inefficient but works for small K (32).
                // We'll use a manual loop with DataCopy for each element? That's terrible.
                // Better: we can load the entire column of B by iterating over k and using DataCopy for each scalar.
                // Since K is 32, we'll just do a scalar loop.
                float sum = 0.0f;
                for (uint32_t k = 0; k < K_; k++) {
                    float aVal = aRow(k);  // scalar read from local tensor
                    float bVal = bGm_[k * N_ + col]; // global read
                    sum += aVal * bVal;
                }
                // Write result
                cGm_[row * N_ + col] = sum;
            }

            // Free row buffer
            AscendC::LocalTensor<float> aRowDeque = inQueueA_.DeQue<float>();
            inQueueA_.FreeTensor(aRowDeque);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    TallSkinnyMatmulKernel op;
    op.Init(a, b, c, tiling_data.M, tiling_data.K, tiling_data.N, tiling_data.tileM);
    op.Process();
}
