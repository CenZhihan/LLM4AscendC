host_operator_src="""
#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_32B = 32;
static constexpr uint32_t HARD_CAP_OUTER = 1024;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::Shape &inputShapeRef = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape *inputShape = &inputShapeRef;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : 0L;

    int64_t ndim = static_cast<int64_t>(inputShape->GetDimNum());
    if (dim < 0) dim += ndim;
    if (dim < 0) dim = 0;
    if (dim >= ndim) dim = ndim - 1;

    uint32_t totalOuter = 1;
    uint32_t reductionSize = 1;
    for (int64_t i = 0; i < ndim; ++i) {
        uint32_t s = static_cast<uint32_t>(inputShape->GetDim(i));
        if (i == dim) {
            reductionSize = s;
        } else {
            totalOuter *= s;
        }
    }

    // per-outer-element work: we need a local buffer of reductionSize floats for the slice,
    // plus a work buffer of 2 floats (for max and index). Total bytes = (reductionSize + 2) * sizeof(float)
    // UB budget fraction; we also need to pipeline.
    uint32_t bytesPerOuter = (reductionSize + 2) * sizeof(float);
    uint32_t maxOuter = UB_BUDGET_BYTES / bytesPerOuter;
    if (maxOuter < 1) maxOuter = 1;
    if (maxOuter > HARD_CAP_OUTER) maxOuter = HARD_CAP_OUTER;

    uint32_t tileOuter = (totalOuter < maxOuter) ? totalOuter : maxOuter;
    // ensure tileOuter > 0
    if (tileOuter == 0) tileOuter = 1;

    uint32_t blockDim = CeilDiv(totalOuter, tileOuter);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalOuter(totalOuter);
    tiling.set_reductionSize(reductionSize);
    tiling.set_tileOuter(tileOuter);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    int64_t ndim = x_shape->GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : 0L;
    if (dim < 0) dim += ndim;
    dim = (dim < 0) ? 0 : (dim >= ndim ? ndim - 1 : dim);

    std::vector<int64_t> outDims;
    for (int64_t i = 0; i < ndim; ++i) {
        if (i != dim) {
            outDims.push_back(x_shape->GetDim(i));
        }
    }
    gert::Shape y_shape;
    y_shape.SetDimNum(static_cast<uint64_t>(outDims.size()));
    for (size_t i = 0; i < outDims.size(); ++i) {
        y_shape.SetDim(i, outDims[i]);
    }
    *context->GetOutputShape(0) = y_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:

            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(0);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOuter, uint32_t reductionSize,
                                uint32_t tileOuter, float /*negativeSlope unused*/)
    {
        totalOuter_ = totalOuter;
        reductionSize_ = reductionSize;
        tileOuter_ = tileOuter;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalOuter_ * reductionSize_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalOuter_);

        // Each tile: tileOuter_ outer elements, each of reductionSize_ floats
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileOuter_ * reductionSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOuter_ * sizeof(int64_t));
        // Work buffer for per-slice ReduceMax (temporary)
        pipe_.InitBuffer(tmpMaxBuf_, 1, 2 * sizeof(float)); // max + index as float
    }

    __aicore__ inline void Process()
    {
        if (totalOuter_ == 0 || reductionSize_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Grid-stride loop over outer elements, one per iteration
        uint32_t idx = blockIdx;
        while (idx < totalOuter_) {
            // copy one outer slice
            CopyIn(idx);
            Compute(idx);
            CopyOut(idx);
            idx += blockNum;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t outerIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        uint32_t gmOffset = outerIdx * reductionSize_;
        AscendC::DataCopy(xLocal, xGm_[gmOffset], reductionSize_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*outerIdx*/)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.AllocTensor<int64_t>();
        AscendC::LocalTensor<float> tmpMax = tmpMaxBuf_.AllocTensor<float>();

        // ReduceMax with index: result[0] = max, result[1] = index (as float)
        AscendC::ReduceMax<float>(tmpMax, xLocal, tmpMax, reductionSize_, true);
        // Extract index and convert to int64
        float idxFloat = tmpMax.GetValue(1);
        int64_t idxInt = static_cast<int64_t>(idxFloat);
        yLocal.SetValue(0, idxInt);

        outQueueY_.EnQue<int64_t>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        tmpMaxBuf_.FreeTensor(tmpMax);
    }

    __aicore__ inline void CopyOut(uint32_t outerIdx)
    {
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.DeQue<int64_t>();
        AscendC::DataCopy(yGm_[outerIdx], yLocal, 1);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMaxBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalOuter_ {0};
    uint32_t reductionSize_ {0};
    uint32_t tileOuter_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalOuter, tiling_data.reductionSize, tiling_data.tileOuter, 0.0f);
    op.Process();
}
