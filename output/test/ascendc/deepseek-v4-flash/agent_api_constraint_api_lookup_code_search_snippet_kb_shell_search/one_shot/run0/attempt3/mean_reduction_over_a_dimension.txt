#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: accBuf, rowBuf each of tileLen floats -> 2 * tileLen * 4 bytes = 8 * tileLen
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape &inputShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t ndim = static_cast<uint32_t>(inputShape.GetDimNum());
    // We assume 3D for this implementation; axis=1 is hardcoded.
    if (ndim != 3) return ge::GRAPH_FAILED;

    uint32_t d0 = static_cast<uint32_t>(inputShape.GetDim(0));
    uint32_t d1 = static_cast<uint32_t>(inputShape.GetDim(1));
    uint32_t d2 = static_cast<uint32_t>(inputShape.GetDim(2));
    uint32_t totalLength = d0 * d1 * d2;

    // Read attribute dim (unused, but we enforce axis=1)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;
    // Ensure it's 1 for this kernel
    if (dim != 1) return ge::GRAPH_FAILED;

    // Compute tile length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 8; // 8 bytes per element (2 floats)
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (d2 < tileLen) {
        uint32_t aligned = AlignDown(d2, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : d2;
    }

    // blockDim: parallelize over batches
    uint32_t blockDim = d0;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_batchSize(d0);
    tiling.set_dim1(d1);
    tiling.set_dim2(d2);
    tiling.set_tileLen(tileLen);
    tiling.set_invDim1(1.0f / static_cast<float>(d1));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    // Get dim attribute
    int64_t dim = 1; // default
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs != nullptr) {
        const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
        if (dimPtr != nullptr) dim = *dimPtr;
    }
    uint32_t ndim = static_cast<uint32_t>(x_shape->GetDimNum());
    // Handle negative dimension
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= static_cast<int64_t>(ndim)) return ge::GRAPH_FAILED;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(ndim - 1);
    uint32_t outIdx = 0;
    for (uint32_t i = 0; i < ndim; ++i) {
        if (static_cast<int64_t>(i) == dim) continue;
        y_shape->SetDim(outIdx++, x_shape->GetDim(i));
    }
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return ge::GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:
    explicit MeanReductionOverADimensionCustom(const char* name) : OpDef(name)

kernel_src="""

#include "kernel_operator.h"

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t dim1, uint32_t dim2,
                                uint32_t tileLen, float invDim1)
    {
        batchSize_ = batchSize;
        dim1_ = dim1;
        dim2_ = dim2;
        tileLen_ = tileLen;
        invDim1_ = invDim1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * dim1_ * dim2_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * dim2_);

        pipe_.InitBuffer(accBuf_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(rowBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || dim1_ == 0 || dim2_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= batchSize_) return;

        // Process single batch
        uint32_t batchOffset = blockIdx * dim1_ * dim2_;

        // Tile over columns (dim2)
        for (uint32_t colBase = 0; colBase < dim2_; colBase += tileLen_) {
            uint32_t curTileLen = (colBase + tileLen_ <= dim2_) ? tileLen_ : (dim2_ - colBase);
            if (curTileLen == 0) break;

            AscendC::LocalTensor<float> accLocal = accBuf_.AllocTensor<float>();
            AscendC::LocalTensor<float> rowLocal = rowBuf_.AllocTensor<float>();

            // Initialize accumulator to zero
            AscendC::Duplicate<float>(accLocal, 0.0f, curTileLen);

            // Accumulate over dim1
            for (uint32_t j = 0; j < dim1_; ++j) {
                uint32_t readOffset = batchOffset + j * dim2_ + colBase;
                AscendC::DataCopy(rowLocal, xGm_[readOffset], curTileLen);
                AscendC::Add(accLocal, accLocal, rowLocal, curTileLen);
            }

            // Scale by inverse of dim1
            AscendC::Muls(accLocal, accLocal, invDim1_, curTileLen);

            // Write output
            uint32_t writeOffset = blockIdx * dim2_ + colBase;
            AscendC::DataCopy(yGm_[writeOffset], accLocal, curTileLen);

            rowBuf_.FreeTensor(rowLocal);
            accBuf_.FreeTensor(accLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> rowBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t dim1_ {0};
    uint32_t dim2_ {0};
    uint32_t tileLen_ {0};
    float invDim1_ {0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.batchSize,
            tiling_data.dim1,
            tiling_data.dim2,
            tiling_data.tileLen,
            tiling_data.invDim1);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor mean_reduction_impl_npu(const at::Tensor& self, int64_t dim) {
    // dim expected to be 1 for this kernel; enforce
    TORCH_CHECK(dim == 1, "This custom kernel only supports dim=1");
    at::Tensor result = at::empty_like(self);
    // Remove the reduced dimension from shape
    std::vector<int64_t> outShape;
    for (int64_t i = 0; i < self.dim(); ++i) {
        if (i != dim) outShape.push_back(self.size(i));
    }
    result = at::empty(outShape, self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("mean_reduction_over_a_dimension_custom", &mean_reduction_impl_npu,
          "Mean reduction over a specific dimension (only dim=1 supported)");
