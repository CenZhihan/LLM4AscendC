static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::StorageShape *h_shape = context->GetInputShape(1);
    const gert::StorageShape *wo_shape = context->GetInputShape(4);
    uint32_t B = x_shape->GetDim(0);
    uint32_t H = h_shape->GetDim(1);

    h_new_shape->SetDimNum(2);

    y_shape->SetDimNum(2);
    y_shape->SetDim(0, B);
    y_shape->SetDim(1, O);

    return GRAPH_SUCCESS;
}
