static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

// Return floor division (non-negative)
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t batch   = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t channels= static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t dimD    = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t dimH    = static_cast<uint32_t>(x_shape->GetDim(3));
    uint32_t dimW    = static_cast<uint32_t>(x_shape->GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t kernel_size = static_cast<uint32_t>(attrs->GetAttrInt(0));
    uint32_t stride      = static_cast<uint32_t>(attrs->GetAttrInt(1));
    uint32_t padding     = static_cast<uint32_t>(attrs->GetAttrInt(2));

    // compute output dimensions (ceil mode false, dilation=1)
    auto outDim = [&](uint32_t inDim) -> uint32_t {
        int32_t numerator = static_cast<int32_t>(inDim) + 2 * static_cast<int32_t>(padding) - static_cast<int32_t>(kernel_size);
        if (numerator < 0) return 0;
        return static_cast<uint32_t>(numerator) / stride + 1;
    };
    uint32_t outD = outDim(dimD);
    uint32_t outH = outDim(dimH);
    uint32_t outW = outDim(dimW);

    // Total output elements per batch-channel slice
    uint32_t totalOutPerSlice = outD * outH * outW;
    // For each output element we need a window of size kernel_size^3
    // Each window read requires kernel_size^3 floats -> we need to load a tile of input.
    // Conservatively, process one output element at a time using a small local buffer.
    // We'll allocate UB for:
    //   - one input window: kernel_size*kernel_size*kernel_size floats
    //   - one output element: 1 float
    //   - temporary for comparison
    // So UB budget per thread (block) must fit.
    uint32_t windowSize = kernel_size * kernel_size * kernel_size;
    uint32_t neededFloats = windowSize + 2; // +2 for max value and index
    uint32_t neededBytes = neededFloats * sizeof(float);
    // We'll use a fixed small tile length for output elements to balance.
    // For simplicity, we process a single output element per iteration.
    // But to avoid too many iterations we can process multiple output elements per thread.
    // Here we choose to process a block of output elements such that the input window overlaps.

    // For a simple functional kernel, we'll set blockDim = number of output elements / some factor.
    // We'll just use a fixed blockDim = 8 as a safe start.
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_inputD(dimD);
    tiling.set_inputH(dimH);
    tiling.set_inputW(dimW);
    tiling.set_outputD(outD);
    tiling.set_outputH(outH);
    tiling.set_outputW(outW);
    tiling.set_kernelSize(kernel_size);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batch, uint32_t channels,
                                uint32_t inputD, uint32_t inputH, uint32_t inputW,
                                uint32_t outputD, uint32_t outputH, uint32_t outputW,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding)
    {
        batch_ = batch;
        channels_ = channels;
        inD_ = inputD;
        inH_ = inputH;
        inW_ = inputW;
        outD_ = outputD;
        outH_ = outputH;
        outW_ = outputW;
        ksize_ = kernelSize;
        stride_ = stride;
        pad_ = padding;

        // Total elements in input
        totalInElems_ = batch_ * channels_ * inD_ * inH_ * inW_;
        totalOutElems_ = batch_ * channels_ * outD_ * outH_ * outW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutElems_);

        // For simplicity, allocate a small local buffer for one window and one output value.
        uint32_t windowSize = ksize_ * ksize_ * ksize_;
        pipe_.InitBuffer(inQueue_, BUFFER_NUM, windowSize * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float) + sizeof(float)); // maxVal and one temp
    }

    __aicore__ inline void Process()
    {
        if (totalOutElems_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a subset of output elements.
        // We'll assign each block a range of output elements by linear index.
        // Simple grid-stride over totalOutElems
        uint32_t start = blockIdx;
        uint32_t step = blockNum;

        for (uint32_t idx = start; idx < totalOutElems_; idx += step) {
            // Convert linear output index to (b, c, od, oh, ow)
            uint32_t tmp = idx;
            uint32_t ow = tmp % outW_; tmp /= outW_;
            uint32_t oh = tmp % outH_; tmp /= outH_;
            uint32_t od = tmp % outD_; tmp /= outD_;
            uint32_t c  = tmp % channels_; tmp /= channels_;
            uint32_t b  = tmp;

            // Compute input window bounds with padding (zero-padding with -inf)
            int32_t dStart = static_cast<int32_t>(od) * static_cast<int32_t>(stride_) - static_cast<int32_t>(pad_);
            int32_t hStart = static_cast<int32_t>(oh) * static_cast<int32_t>(stride_) - static_cast<int32_t>(pad_);
            int32_t wStart = static_cast<int32_t>(ow) * static_cast<int32_t>(stride_) - static_cast<int32_t>(pad_);

            float maxVal = -std::numeric_limits<float>::infinity();

            // Loop over kernel window
            for (uint32_t kd = 0; kd < ksize_; ++kd) {
                int32_t id = dStart + static_cast<int32_t>(kd);
                if (id < 0 || id >= static_cast<int32_t>(inD_)) continue;
                for (uint32_t kh = 0; kh < ksize_; ++kh) {
                    int32_t ih = hStart + static_cast<int32_t>(kh);
                    if (ih < 0 || ih >= static_cast<int32_t>(inH_)) continue;
                    for (uint32_t kw = 0; kw < ksize_; ++kw) {
                        int32_t iw = wStart + static_cast<int32_t>(kw);
                        if (iw < 0 || iw >= static_cast<int32_t>(inW_)) continue;

                        // Compute global input index
                        uint32_t inIdx = ((b * channels_ + c) * inD_ + static_cast<uint32_t>(id)) * inH_ + static_cast<uint32_t>(ih);
                        inIdx = inIdx * inW_ + static_cast<uint32_t>(iw);
                        float val = xGm_.GetValue(inIdx);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }

            // Write output
            yGm_.SetValue(idx, maxVal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t channels_;
    uint32_t inD_, inH_, inW_;
    uint32_t outD_, outH_, outW_;
    uint32_t ksize_;
    uint32_t stride_;
    uint32_t pad_;
    uint32_t totalInElems_;
    uint32_t totalOutElems_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.batch, tiling_data.channels,
            tiling_data.inputD, tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputD, tiling_data.outputH, tiling_data.outputW,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding);
    op.Process();
}
