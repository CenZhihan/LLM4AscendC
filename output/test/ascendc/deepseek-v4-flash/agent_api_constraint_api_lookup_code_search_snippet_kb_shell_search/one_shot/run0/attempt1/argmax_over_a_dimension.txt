host_operator_src = """

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_32B = 32;
static constexpr uint32_t HARD_CAP_OUTER = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::Shape *inputShape = context->GetInputShape(0)->GetOriginShape();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : 0L;

    // Convert negative dim
    int64_t ndim = static_cast<int64_t>(inputShape->GetDimNum());
    if (dim < 0) dim += ndim;
    dim = (dim < 0) ? 0 : (dim >= ndim ? ndim - 1 : dim);

    // Compute totalOuter and reductionSize
    uint32_t totalOuter = 1;
    uint32_t reductionSize = 1;
    for (int64_t i = 0; i < ndim; ++i) {
        uint32_t s = static_cast<uint32_t>(inputShape->GetDim(i));
        if (i == dim) {
            reductionSize = s;
        } else {
            totalOuter *= s;
        }
    }

    // Tiling on outer dimension
    uint32_t bytesPerOuter = (reductionSize * 2 + 1) * sizeof(float);  // approx: src + tmp + dst per outer elem
    uint32_t maxOuterPerTile = UB_BUDGET_BYTES / bytesPerOuter;
    maxOuterPerTile = AlignDown(maxOuterPerTile, 32);  // align to multiple of 32
    if (maxOuterPerTile < 32) maxOuterPerTile = 32;
    if (maxOuterPerTile > HARD_CAP_OUTER) maxOuterPerTile = AlignDown(HARD_CAP_OUTER, 32);

    uint32_t tileOuter = (totalOuter < maxOuterPerTile) ? totalOuter : maxOuterPerTile;

    uint32_t blockDim = CeilDiv(totalOuter, tileOuter);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalOuter(totalOuter);
    tiling.set_reductionSize(reductionSize);
    tiling.set_tileOuter(tileOuter);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    int64_t ndim = x_shape->GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : 0L;
    if (dim < 0) dim += ndim;
    dim = (dim < 0) ? 0 : (dim >= ndim ? ndim - 1 : dim);

    std::vector<int64_t> outDims;
    for (int64_t i = 0; i < ndim; ++i) {
        if (i != dim) {
            outDims.push_back(x_shape->GetDim(i));
        }
    }
    gert::Shape y_shape(outDims);
    *context->GetOutputShape(0) = y_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class ArgmaxOverADimensionCustom : public OpDef {
public:
    explicit ArgmaxOverADimensionCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT64})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(ArgmaxOverADimensionCustom);
} // namespace ops

kernel_src = """

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
        GET_TILING_DATA(tiling_data, tiling);
        totalOuter_ = tiling_data.totalOuter;
        reductionSize_ = tiling_data.reductionSize;
        tileOuter_ = tiling_data.tileOuter;
        dim_ = tiling_data.dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalOuter_ * reductionSize_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalOuter_);

        // Buffers: one tile of input (tileOuter_ * reductionSize_), and per-element tmp
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileOuter_ * reductionSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOuter_ * sizeof(int64_t));
        pipe_.InitBuffer(tmpQueue_, 1, tileOuter_ * sizeof(float) * 2);  // each ReduceMax needs 2*reductionSize? but we do per-element
        // Actually we'll do per-element inside Compute, so buffer smaller
    }

    __aicore__ inline void Process() {
        if (totalOuter_ == 0 || reductionSize_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Each block handles a contiguous range of outer elements
        const uint32_t outerPerBlock = CeilDiv(totalOuter_, blockNum);
        const uint32_t start = blockIdx * outerPerBlock;
        const uint32_t end = (start + outerPerBlock > totalOuter_) ? totalOuter_ : start + outerPerBlock;

        uint32_t cur = start;
        while (cur < end) {
            uint32_t remaining = end - cur;
            uint32_t len = (remaining > tileOuter_) ? tileOuter_ : remaining;
            CopyIn(cur, len);
            Compute(len);
            CopyOut(cur, len);
            cur += len;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t outerOffset, uint32_t len) {
        // Each outer element occupies reductionSize_ contiguous floats
        uint32_t byteOffset = outerOffset * reductionSize_ * sizeof(float);
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[byteOffset / sizeof(float)], len * reductionSize_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.AllocTensor<int64_t>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

        // For each outer element, reduce over the reduction axis
        for (uint32_t i = 0; i < len; ++i) {
            AscendC::LocalTensor<float> slice = xLocal[i * reductionSize_];
            AscendC::LocalTensor<float> result = tmpLocal[i * 2];     // place for [max, idx] as float
            // Use ReduceMax with index flag: result[0]=max, result[1]=idx (as float)
            AscendC::ReduceMax<float>(result, slice, tmpLocal[0], reductionSize_, true);
            // Extract index and cast to int64
            float idxFloat = result.GetValue(1);
            int64_t idxInt = static_cast<int64_t>(idxFloat);
            yLocal.SetValue(i, idxInt);
        }

        outQueueY_.EnQue<int64_t>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        tmpQueue_.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(uint32_t outerOffset, uint32_t len) {
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.DeQue<int64_t>();
        AscendC::DataCopy(yGm_[outerOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

    static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
        return (a + b - 1) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;
    uint32_t totalOuter_ {0};
    uint32_t reductionSize_ {0};
    uint32_t tileOuter_ {0};
    int32_t dim_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}
"""

python_bind_src = """

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    at::Tensor result = at::empty_like(self, self.options().dtype(at::kLong));
    // Remove the reduced dimension from shape for result
    auto shape = self.sizes().vec();
    shape.erase(shape.begin() + dim);
    result = result.resize_(shape);
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu, "Argmax over a dimension (ArgmaxOverADimensionCustom on NPU)");
