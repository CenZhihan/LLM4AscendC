#include "register/tilingdata_base.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const uint32_t M = context->GetInputShape(0)->GetOriginShape().GetDim(0);
    const uint32_t N = context->GetInputShape(1)->GetOriginShape().GetDim(1);
    const uint32_t K = context->GetInputShape(0)->GetOriginShape().GetDim(1);

    // Use MatmulApiTiling to compute optimal tiling parameters
    auto ascendPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling matmulTiling(ascendPlatform);
    matmulTiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    matmulTiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    matmulTiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    matmulTiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);

    matmulTiling.SetOrgShape(M, N, K);
    matmulTiling.SetShape(M, N, K);
    matmulTiling.SetBias(false);
    matmulTiling.SetBufferSpace(-1, -1, -1);

    TCubeTiling tilingData;
    if (matmulTiling.GetTiling(tilingData) == -1) {
        return ge::GRAPH_FAILED;
    }

    // Set block dimension to number of cores (can be adjusted)
    size_t coreNum = ascendPlatform.GetCoreNumAic();
    context->SetBlockDim(coreNum);

    // Write tiling data to raw buffer
    tilingData.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tilingData.GetDataSize());

    // No workspace needed
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    uint32_t M = a_shape->GetDim(0);

    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

// Kernel matrix types (all float)
using A_MATRIX_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float, false>;
using B_MATRIX_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float, true>;
using C_MATRIX_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float, false>;
using BIAS_MATRIX_TYPE = MatmulType<TPosition::GM, CubeFormat::ND, float>;

class TallSkinnyMatmulKernel {
public:
    __aicore__ inline TallSkinnyMatmulKernel() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, TCubeTiling &tiling)
    {
        aGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(a), tiling.M * tiling.Ka);
        bGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b), tiling.Kb * tiling.N);
        cGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(c), tiling.M * tiling.N);

        // Bias not used, but set to a dummy (nullptr) — SetBias will be skipped
        biasGlobal_.SetGlobalBuffer(nullptr, 1);

        int32_t offsetA = 0, offsetB = 0, offsetC = 0;
        CalcOffset(GetBlockIdx(), tiling, offsetA, offsetB, offsetC);

        aGlobal_ = aGlobal_[offsetA];
        bGlobal_ = bGlobal_[offsetB];
        cGlobal_ = cGlobal_[offsetC];
    }

    __aicore__ inline void Process(AscendC::TPipe *pipe)
    {
        if (GetBlockIdx() >= 1) {
            return;
        }

        matmulObj_.SetTensorA(aGlobal_);
        matmulObj_.SetTensorB(bGlobal_);
        // No bias used
        matmulObj_.IterateAll(cGlobal_);
        matmulObj_.End();
    }

private:
    __aicore__ inline void CalcOffset(int32_t blockIdx, const TCubeTiling &tiling,
                                      int32_t &offsetA, int32_t &offsetB, int32_t &offsetC)
    {
        int32_t mSingleBlocks = Ceiling(tiling.M, tiling.singleCoreM);
        int32_t mCoreIndx = blockIdx % mSingleBlocks;
        int32_t nCoreIndx = blockIdx / mSingleBlocks;

        offsetA = mCoreIndx * tiling.Ka * tiling.singleCoreM;
        offsetB = nCoreIndx * tiling.singleCoreN;
        offsetC = mCoreIndx * tiling.N * tiling.singleCoreM + nCoreIndx * tiling.singleCoreN;
    }

private:
    Matmul<A_MATRIX_TYPE, B_MATRIX_TYPE, C_MATRIX_TYPE, BIAS_MATRIX_TYPE> matmulObj_;
    GlobalTensor<float> aGlobal_;
    GlobalTensor<float> bGlobal_;
    GlobalTensor<float> cGlobal_;
    GlobalTensor<float> biasGlobal_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm)
{
    if ASCEND_IS_AIV {
        return;
    }

    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulObj_, &tiling);

    TallSkinnyMatmulKernel kernel;
    kernel.Init(a, b, c, tiling);
    kernel.Process(&pipe);
}
