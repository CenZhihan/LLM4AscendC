using C_TYPE = AscendC::MatmulType<TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;

public:
    __aicore__ inline VanillaRnnHiddenCustom() {}

import torch
import torch_npu
import custom_ops_lib

_FN = 'vanilla_rnn_hidden_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, x, h0, w_i2h, b_i2h, w_h2o, b_h2o):
        T, B, I = x.shape
        H = h0.shape[1]
        O = w_h2o.shape[0]
        h = h0
        outs = []
        custom_fn = getattr(custom_ops_lib, _FN)
        for t in range(T):
            x_t = x[t]  # (B, I)
            h, y_t = custom_fn(x_t, h, w_i2h, b_i2h, w_h2o, b_h2o)
