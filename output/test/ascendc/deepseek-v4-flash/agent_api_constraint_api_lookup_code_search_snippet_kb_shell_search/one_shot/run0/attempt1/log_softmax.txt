host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative per-core UB share
static constexpr uint32_t ALIGN_WIDTH = 2;                 // subTileWidth must be even for ReduceMax
static constexpr uint32_t MIN_SUB_TILE = 256;               // minimum sub‑tile width

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    uint32_t height = static_cast<uint32_t>(shape->GetOriginShape().GetDim(0));
    uint32_t width  = static_cast<uint32_t>(shape->GetOriginShape().GetDim(1));
    uint32_t totalLength = height * width;

    // compute subTileWidth: we need to load xLocal, yLocal, expLocal, tmpLocal (4 floats per element)
    // plus one element for max, one for sum. Roughly: 4 * subTileWidth * sizeof(float) = 16 * subTileWidth bytes
    uint32_t maxSubTile = UB_BUDGET_BYTES / 16;
    maxSubTile = AlignDown(maxSubTile, ALIGN_WIDTH);
    if (maxSubTile < MIN_SUB_TILE) maxSubTile = MIN_SUB_TILE;

    uint32_t subTileWidth = (width > maxSubTile) ? maxSubTile : width;

    // Each kernel processes one row at a time (height tile = 1)
    uint32_t tileHeight = 1;
    uint32_t tileLength = tileHeight * subTileWidth;   // elements per sub‑tile load (actually per row we load one sub‑tile at a time)

    // blockDim: number of rows divided equally among cores
    uint32_t blockDim = height;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_height(height);
    tiling.set_width(width);
    tiling.set_subTileWidth(subTileWidth);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

public:

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;   // single buffer, no double buffering for simplicity
constexpr float INF_NEG = -1e38f;   // small negative value for init max

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileLength,
                                uint32_t height, uint32_t width, uint32_t subTileWidth) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;   // subTileWidth elements (since tileHeight=1)
        height_ = height;
        width_ = width;
        subTileWidth_ = subTileWidth;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // buffers: xLocal, yLocal, expLocal, tmpLocal (all of subTileWidth)
        // maxLocal and sumLocal are scalars, stored in separate tensors.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, subTileWidth_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, subTileWidth_ * sizeof(float));
        pipe_.InitBuffer(expQueue_, BUFFER_NUM, subTileWidth_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, subTileWidth_ * sizeof(float));
        pipe_.InitBuffer(maxQueue_, 1, sizeof(float));
        pipe_.InitBuffer(sumQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        // grid‑stride over rows
        const uint32_t blockNum = GetBlockNum();
        const uint32_t blockIdx = GetBlockIdx();

        for (uint32_t row = blockIdx; row < height_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row) {
        uint32_t baseX = row * width_;
        uint32_t baseY = row * width_;

        // ---------------------------------------------
        // Pass 1: compute global max over width
        // ---------------------------------------------
        float globalMax = INF_NEG;
        uint32_t numSubTiles = (width_ + subTileWidth_ - 1) / subTileWidth_;

        for (uint32_t st = 0; st < numSubTiles; ++st) {
            uint32_t offset = st * subTileWidth_;
            uint32_t curLen = (offset + subTileWidth_ <= width_) ? subTileWidth_ : (width_ - offset);

            // load one sub‑tile into local
            LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            DataCopy(xLocal, xGm_[baseX + offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // compute local max (store in maxQueue)
            LocalTensor<float> maxLocal = maxQueue_.AllocTensor<float>();
            ReduceMax(maxLocal, xLocal, tmpQueue_.Get<float>(), curLen);
            float localMax = maxLocal.GetValue(0);
            globalMax = fmaxf(globalMax, localMax);

            maxQueue_.FreeTensor(maxLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        // ---------------------------------------------
        // Pass 2: compute sum of exp(x ‑ globalMax)
        // ---------------------------------------------
        float sum = 0.0f;
        for (uint32_t st = 0; st < numSubTiles; ++st) {
            uint32_t offset = st * subTileWidth_;
            uint32_t curLen = (offset + subTileWidth_ <= width_) ? subTileWidth_ : (width_ - offset);

            LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            DataCopy(xLocal, xGm_[baseX + offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            LocalTensor<float> expLocal = expQueue_.AllocTensor<float>();
            LocalTensor<float> tmpLocal = tmpQueue_.Get<float>();

            // tmp = x ‑ globalMax
            Sub(tmpLocal, xLocal, globalMax, curLen);
            // exp = exp(tmp)
            Exp(expLocal, tmpLocal, curLen);

            // reduce add: sum of exp
            LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
            ReduceAdd(sumLocal, expLocal, tmpLocal, curLen);
            sum += sumLocal.GetValue(0);

            inQueueX_.FreeTensor(xLocal);
            expQueue_.FreeTensor(expLocal);
            sumQueue_.FreeTensor(sumLocal);
        }

        // log of sum (clamp to avoid log(0))
        float logSum = logf(fmaxf(sum, 1e-38f));

        // ---------------------------------------------
        // Pass 3: y = (x ‑ globalMax) ‑ logSum
        // ---------------------------------------------
        for (uint32_t st = 0; st < numSubTiles; ++st) {
            uint32_t offset = st * subTileWidth_;
            uint32_t curLen = (offset + subTileWidth_ <= width_) ? subTileWidth_ : (width_ - offset);

            LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            DataCopy(xLocal, xGm_[baseX + offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            LocalTensor<float> tmpLocal = tmpQueue_.Get<float>();

            // tmp = x ‑ globalMax
            Sub(tmpLocal, xLocal, globalMax, curLen);
            // y = tmp ‑ logSum
            Sub(yLocal, tmpLocal, logSum, curLen);

            // store output
            DataCopy(yGm_[baseY + offset], yLocal, curLen);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    TPipe pipe_;
    TQue<QuePosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<QuePosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<QuePosition::VECCALC, BUFFER_NUM> expQueue_;
    TQue<QuePosition::VECCALC, 1> tmpQueue_;
    TQue<QuePosition::VECCALC, 1> maxQueue_;
    TQue<QuePosition::VECCALC, 1> sumQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    uint32_t subTileWidth_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.height, tiling_data.width, tiling_data.subTileWidth);
    op.Process();
}
