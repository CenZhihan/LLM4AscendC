host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile (input)
//   outQueueY: 2 * tile (output)
//   tmpQueue: 1 * tile (reduction temp)
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const gert::StorageShape *inShape = context->GetInputShape(0);
    const uint32_t totalLength =
        static_cast<uint32_t>(inShape->GetOriginShape().GetShapeSize());
    // Assume dim=1 (row‑wise softmax)
    const uint32_t rowSize =
        static_cast<uint32_t>(inShape->GetOriginShape().GetDim(1));

    // tileLen from UB budget: bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // Adaptive blockDim; use up to 24 to improve parallelism
    const uint32_t numRows = totalLength / rowSize;
    uint32_t blockDim = numRows;   // coarse guess
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_rowSize(rowSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // Workspace for intermediate exp values (float per element)
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = totalLength * sizeof(float);
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})

            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t rowSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        rowSize_ = rowSize;
        numRows_ = totalLen_ / rowSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        wkGm_.SetGlobalBuffer((__gm__ float*)workspace, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Row‑stride grid‑stride: each block processes rows rowIdx = blockIdx, blockIdx+blockNum, ...
        for (uint32_t row = blockIdx; row < numRows_; row += blockNum) {
            uint32_t rowBase = row * rowSize_;

            // ---------- First pass: compute row max ----------
            float rowMax = -FLT_MAX;
            for (uint32_t col = 0; col < rowSize_; col += tileLen_) {
                uint32_t curLen = rowSize_ - col;
                if (curLen > tileLen_) curLen = tileLen_;
                uint32_t gmOffset = rowBase + col;

                // Load input tile
                AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xTile, xGm_[gmOffset], curLen);
                inQueueX_.EnQue(xTile);

                // Compute local max
                xTile = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceMax(maxLocal, xTile, curLen);   // single element
                float localMaxVal = maxLocal.GetValue(0);
                tmpQueue_.FreeTensor(maxLocal);
                inQueueX_.FreeTensor(xTile);

                if (localMaxVal > rowMax) rowMax = localMaxVal;
            }

            // ---------- Second pass: compute exp(x - rowMax) and store in workspace,
            //             accumulate row sum ----------
            float rowSum = 0.0f;
            for (uint32_t col = 0; col < rowSize_; col += tileLen_) {
                uint32_t curLen = rowSize_ - col;
                if (curLen > tileLen_) curLen = tileLen_;
                uint32_t gmOffset = rowBase + col;

                // Load input tile
                AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xTile, xGm_[gmOffset], curLen);
                inQueueX_.EnQue(xTile);

                xTile = inQueueX_.DeQue<float>();
                // Compute exp(x - rowMax) and store to workspace
                AscendC::LocalTensor<float> expTile = tmpQueue_.AllocTensor<float>();
                AscendC::Sub(expTile, xTile, rowMax, curLen);
                AscendC::Exp(expTile, expTile, curLen);
                // Write to workspace
                AscendC::DataCopy(wkGm_[gmOffset], expTile, curLen);
                // Accumulate sum of this tile
                AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>(); // reuse? careful
                AscendC::ReduceSum(sumLocal, expTile, curLen);
                float tileSum = sumLocal.GetValue(0);
                tmpQueue_.FreeTensor(sumLocal); // free the reduction temp
                tmpQueue_.FreeTensor(expTile);
                inQueueX_.FreeTensor(xTile);

                rowSum += tileSum;
            }

            // ---------- Third pass: load from workspace, divide by row sum, write output ----------
            for (uint32_t col = 0; col < rowSize_; col += tileLen_) {
                uint32_t curLen = rowSize_ - col;
                if (curLen > tileLen_) curLen = tileLen_;
                uint32_t gmOffset = rowBase + col;

                // Load workspace tile
                AscendC::LocalTensor<float> expTile = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(expTile, wkGm_[gmOffset], curLen);
                inQueueX_.EnQue(expTile);

                expTile = inQueueX_.DeQue<float>();
                // Divide by rowSum
                AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();
                AscendC::Div(yTile, expTile, rowSum, curLen);
                // Write output
                outQueueY_.EnQue(yTile);
                AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[gmOffset], yOut, curLen);
                outQueueY_.FreeTensor(yOut);
                inQueueX_.FreeTensor(expTile);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wkGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t rowSize_ {0};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y,
                                                     GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, workspace, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.rowSize);
    op.Process();
}
"""
