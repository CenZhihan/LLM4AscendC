#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MAX_TILE_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const auto& shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t innerDim = static_cast<uint32_t>(shape.GetDim(shape.GetDimNum() - 1));

    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // 3 * tileLen * sizeof(float)
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (maxTile < innerDim) ? maxTile : innerDim;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = innerDim;

    const uint32_t totalRows = totalLength / innerDim;
    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_innerDim(innerDim);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:
    explicit SoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -1e38f;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t innerDim,
                                uint32_t tileLength) {
        totalLen_ = totalLength;
        innerDim_ = innerDim;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float *)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float *)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(expQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || innerDim_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalRows = totalLen_ / innerDim_;
        const uint32_t rowsPerBlock = (totalRows + blockNum - 1) / blockNum;
        const uint32_t rowStart = blockIdx * rowsPerBlock;
        const uint32_t rowEnd = (rowStart + rowsPerBlock > totalRows) ? totalRows : rowStart + rowsPerBlock;

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            const uint32_t rowOffset = row * innerDim_;
            ProcessRow(rowOffset);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowOffset) {
        // ---------- Pass 1: compute row max ----------
        float rowMax = NEG_INF;
        uint32_t remaining = innerDim_;
        uint32_t offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            LoadTile(rowOffset + offset, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            // Use vector Max to compute per-tile max into tmpQueue (tile length)
            AscendC::LocalTensor<float> maxTensor = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceMax(maxTensor, xLocal, curLen, curLen);
            // maxTensor[0] now contains the max of this tile
            float tileMax = maxTensor.GetValue(0);
            tmpQueue_.FreeTensor(maxTensor);
            inQueueX_.FreeTensor(xLocal);
            if (tileMax > rowMax) rowMax = tileMax;
            offset += curLen;
            remaining -= curLen;
        }

        // ---------- Pass 2: compute exp(x - max) and sum ----------
        float rowSum = 0.0f;
        remaining = innerDim_;
        offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            LoadTile(rowOffset + offset, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> expLocal = expQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            // exp(x - rowMax) implemented as:
            //   t = x * (-1.0f) + rowMax   ->  rowMax - x
            //   then exp(-t) = exp(x - rowMax)
            // Actually we want exp(x - rowMax). Compute:
            //   t = x + (-rowMax)  (with scalar -rowMax via Muls + Add)
            AscendC::Muls(tLocal, xLocal, 1.0f, curLen);
            AscendC::Add(tLocal, tLocal, -rowMax, curLen);  // t = x - rowMax
            AscendC::Exp(expLocal, tLocal, curLen);
            // Accumulate sum: use ReduceSum into a scalar
            AscendC::LocalTensor<float> sumTensor = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumTensor, expLocal, curLen, curLen);
            float partSum = sumTensor.GetValue(0);
            tmpQueue_.FreeTensor(sumTensor);
            rowSum += partSum;
            // enqueue exp for pass 3
            expQueue_.EnQue(expLocal);
            tmpQueue_.FreeTensor(tLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }

        // ---------- Pass 3: normalize and write output ----------
        remaining = innerDim_;
        offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> expLocal = expQueue_.DeQue<float>();
            AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            // divide exp by rowSum: compute reciprocal of rowSum
            AscendC::Muls(tLocal, expLocal, 1.0f, curLen);
            AscendC::Muls(tLocal, tLocal, 1.0f / rowSum, curLen);  // multiply by scalar reciprocal
            AscendC::Copy(outLocal, tLocal, curLen);  // or directly use tLocal
            AscendC::DataCopy(yGm_[rowOffset + offset], outLocal, curLen);
            outQueueY_.FreeTensor(outLocal);
            expQueue_.FreeTensor(expLocal);
            tmpQueue_.FreeTensor(tLocal);
            offset += curLen;
            remaining -= curLen;
        }
    }

    __aicore__ inline void LoadTile(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> expQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t innerDim_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.innerDim, tiling_data.tileLength);
    op.Process();
}
