static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MAX_ROWS_PER_BLOCK = 16;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape& xShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(xShape.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = *attrs->GetAttrPointer<int32_t>(0);

    uint32_t rowLength = static_cast<uint32_t>(xShape.GetDim(1));
    uint32_t numRows = static_cast<uint32_t>(xShape.GetDim(0));

    // Compute tile length based on UB budget:
    // We need 6 * tileLen floats for temporaries (exp, broadcastMax, broadcastSum, logSum, plus in/out? in/out are separate queues)
    // Conservatively allocate for 6*tileLen floats => tileLen <= UB_BUDGET_BYTES / (6 * sizeof(float))
    uint32_t maxTile = UB_BUDGET_BYTES / (6 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = rowLength;
    if (tileLen > maxTile) tileLen = maxTile;
    // also need to consider whole row: if rowLength <= maxTile, we do whole row; else tile.

    uint32_t rowsPerTile = 1; // We process one row per tile; if needed can tile within row.
    // For simplicity, we assume tileLen = rowLength (if fits) or tileLen < rowLength, then we need to iterate over tiles per row.
    // The kernel expects to process a single tile per row. So we set tileLen = rowLength (full row) for now.
    // To handle large rows correctly, one would need to tile within the row, but that is beyond the current fix.
    tileLen = rowLength; // Keep full row for now; will work if row fits in UB.

    uint32_t numTiles = CeilDiv(numRows, rowsPerTile);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t tileLength, int32_t dim)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        dim_ = dim;
        numRows_ = totalLen_ / rowLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpExp_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBroadcast1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBroadcast2_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBroadcast3_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpScalar1_, 1, sizeof(float));
        pipe_.InitBuffer(tmpScalar2_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        for (uint32_t row = blockIdx; row < numRows_; row += blockNum)
        {
            uint32_t rowOffset = row * rowLen_;
            LogSoftmaxRow(rowOffset);
        }
    }

private:
    __aicore__ inline void LogSoftmaxRow(uint32_t rowOffset)
    {
        // Step 1: Load input
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowOffset], tileLen_);

        // Step 2: ReduceMax -> scalar tensor
        AscendC::LocalTensor<float> maxScalar = tmpScalar1_.AllocTensor<float>();
        AscendC::ReduceMax(maxScalar, xLocal, tileLen_, 0);

        // Step 3: Broadcast max
        AscendC::LocalTensor<float> maxBroadcast = tmpBroadcast1_.AllocTensor<float>();
        AscendC::BroadCast(maxBroadcast, maxScalar, tileLen_);
        tmpScalar1_.FreeTensor(maxScalar);  // scalar no longer needed

        // Step 4: exp = x - max
        AscendC::LocalTensor<float> expLocal = tmpExp_.AllocTensor<float>();
        AscendC::Sub(expLocal, xLocal, maxBroadcast, tileLen_);
        AscendC::Exp(expLocal, expLocal, tileLen_);

        // Step 5: ReduceSum -> scalar
        AscendC::LocalTensor<float> sumScalar = tmpScalar2_.AllocTensor<float>();
        AscendC::ReduceSum(sumScalar, expLocal, tileLen_, 0);

        // Step 6: Broadcast sum
        AscendC::LocalTensor<float> sumBroadcast = tmpBroadcast2_.AllocTensor<float>();
        AscendC::BroadCast(sumBroadcast, sumScalar, tileLen_);
        tmpScalar2_.FreeTensor(sumScalar);

        // Step 7: logSum = log(sumBroadcast) (element-wise, all identical)
        AscendC::LocalTensor<float> logSumBroadcast = tmpBroadcast3_.AllocTensor<float>();
        AscendC::Log(logSumBroadcast, sumBroadcast, tileLen_);
        tmpBroadcast2_.FreeTensor(sumBroadcast);  // sum broadcast no longer needed

        // Step 8: compute y = x - max - logSum
        // Reuse expLocal as temporary for (max + logSum)
        AscendC::Add(expLocal, maxBroadcast, logSumBroadcast, tileLen_);
        tmpBroadcast1_.FreeTensor(maxBroadcast);
        tmpBroadcast3_.FreeTensor(logSumBroadcast);

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Sub(yLocal, xLocal, expLocal, tileLen_);
        // At this point expLocal holds (max+logSum). We can free it after use.
        tmpExp_.FreeTensor(expLocal);
        inQueueX_.FreeTensor(xLocal);

        // Step 9: Store output
        AscendC::DataCopy(yGm_[rowOffset], yLocal, tileLen_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpExp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBroadcast1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBroadcast2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBroadcast3_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpScalar1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpScalar2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numRows_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLength, tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
'''
