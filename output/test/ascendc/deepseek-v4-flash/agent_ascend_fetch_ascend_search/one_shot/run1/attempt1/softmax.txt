host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: three buffers of tileLength floats (for input, temp, output)
// Also one extra for reduction workspace.
// Total floats ~ 4 * tileLength => bytes = 16 * tileLength
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MAX_TILE_ELEMS = 4096;  // per tile size

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    // Assume the last dimension is to be reduced (softmax along dim=1 for 2D)
    const auto& shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t innerDim = static_cast<uint32_t>(shape.GetDim(shape.GetDimNum() - 1));

    // Compute tileLength: multiple of ALIGN_ELEMS, not exceeding MAX_TILE_ELEMS
    uint32_t maxTile = UB_BUDGET_BYTES / 16;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (maxTile < innerDim) ? maxTile : innerDim;
    // Ensure tileLen does not exceed innerDim
    if (tileLen > innerDim) tileLen = innerDim;
    // Round down to ALIGN_ELEMS
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = innerDim;  // fallback

    const uint32_t totalRows = totalLength / innerDim;
    // Assign each block a row, but limit block count
    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_innerDim(innerDim);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
