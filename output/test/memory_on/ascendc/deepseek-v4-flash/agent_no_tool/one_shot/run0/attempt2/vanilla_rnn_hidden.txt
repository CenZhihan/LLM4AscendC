#include "vanilla_rnn_hidden_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    VanillaRnnHiddenCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *h0_shape = context->GetInputShape(1);
    const gert::Shape *w_i2h_shape = context->GetInputShape(2);
    const gert::Shape *w_h2o_shape = context->GetInputShape(4);

    uint32_t T = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t B = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t I = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    uint32_t H = static_cast<uint32_t>(w_i2h_shape->GetOriginShape().GetDim(0));
    uint32_t O = static_cast<uint32_t>(w_h2o_shape->GetOriginShape().GetDim(0));
    uint32_t K = I + H;

    // Estimate per-batch-element memory per time step: input concat (K), weight i2h (H*K),
    // weight h2o (O*H), biases (H+O), pre (H), y (O). Rough: 2*K + H*K + O*H + H + O floats per batch per step.
    // To be conservative, we assume a single time step's worth per batch element.
    uint32_t est_floats_per_batch = K + H + O + I + H;  // approximate
    uint32_t bytes_per_batch = est_floats_per_batch * 4;
    uint32_t max_tileB = UB_BUDGET_BYTES / bytes_per_batch;
    if (max_tileB > HARD_CAP_ELEMS) max_tileB = HARD_CAP_ELEMS;
    max_tileB = AlignDown(max_tileB, ALIGN_ELEMS);
    if (max_tileB < 1) max_tileB = 1;
    uint32_t tileB = (B < max_tileB) ? B : max_tileB;

    uint32_t numTiles = CeilDiv(B, tileB);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_T(T);
    tiling.set_B(B);
    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_tileB(tileB);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h_gm, GM_ADDR b_i2h_gm,
                                GM_ADDR w_h2o_gm, GM_ADDR b_h2o_gm, GM_ADDR y_gm, GM_ADDR h_new_gm,
                                uint32_t T, uint32_t B, uint32_t I, uint32_t H, uint32_t O, uint32_t K, uint32_t tileB)
    {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O; K_ = K; tileB_ = tileB;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T_ * B_ * I_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B_ * H_);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h_gm, H_ * K_);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h_gm, H_);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o_gm, O_ * H_);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o_gm, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y_gm, T_ * B_ * O_);
        hNewGm_.SetGlobalBuffer((__gm__ float*)h_new_gm, B_ * H_);

        // Allocate scratch buffers: we need for each batch tile:
        // input concat [tileB, K], w_i2h [H, K], b_i2h [H], pre [tileB, H],
        // h [tileB, H], w_h2o [O, H], b_h2o [O], y [tileB, O].
        uint32_t maxTile = tileB_;
        uint32_t scratchElems = maxTile * K_ + H_ * K_ + H_ + maxTile * H_ + O_ * H_ + O_ + maxTile * O_;
        pipe_.InitBuffer(scratchBuf_, 1, scratchElems * sizeof(float));
        scratchLocal_ = scratchBuf_.Get<float>(0);
    }

    __aicore__ inline void Process()
    {
        if (tileB_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t startB = blockIdx * tileB_;
        if (startB >= B_) return;

        // Load weights and biases once (shared across all time steps for this batch tile)
        AscendC::LocalTensor<float> w_i2hLocal = scratchLocal_[0];
        AscendC::DataCopy(w_i2hLocal, w_i2hGm_, H_ * K_);
        AscendC::LocalTensor<float> w_h2oLocal = scratchLocal_[H_ * K_];
        AscendC::DataCopy(w_h2oLocal, w_h2oGm_, O_ * H_);
        AscendC::LocalTensor<float> b_i2hLocal = scratchLocal_[H_ * K_ + O_ * H_];
        AscendC::DataCopy(b_i2hLocal, b_i2hGm_, H_);
        AscendC::LocalTensor<float> b_h2oLocal = scratchLocal_[H_ * K_ + O_ * H_ + H_];
        AscendC::DataCopy(b_h2oLocal, b_h2oGm_, O_);

        // Initialize h for each batch element from h0
        AscendC::LocalTensor<float> hLocal = scratchLocal_[H_ * K_ + O_ * H_ + H_ + O_]; // reuse later
        AscendC::DataCopy(hLocal, h0Gm_[startB * H_], tileB_ * H_);

        // Process each time step
        for (uint32_t t = 0; t < T_; ++t) {
            // Load x[t, startB:startB+tileB, :] 
            AscendC::LocalTensor<float> xLocal = scratchLocal_[H_ * K_ + O_ * H_ + H_ + O_ + tileB_ * H_]; // after h
            AscendC::DataCopy(xLocal, xGm_[t * B_ * I_ + startB * I_], tileB_ * I_);

            // Concatenate x and h into input buffer (reuse xLocal area for input concat)
            AscendC::LocalTensor<float> inputLocal = scratchLocal_[H_ * K_ + O_ * H_ + H_ + O_ + tileB_ * H_]; // same as xLocal start
            for (uint32_t b = 0; b < tileB_; ++b) {
                // copy x part
                AscendC::DataCopy(inputLocal[b * K_], xLocal[b * I_], I_);
                // copy h part
                AscendC::DataCopy(inputLocal[b * K_ + I_], hLocal[b * H_], H_);
            }

            // Compute pre = input @ w_i2h^T + b_i2h
            AscendC::LocalTensor<float> preLocal = scratchLocal_[H_ * K_ + O_ * H_ + H_ + O_ + tileB_ * H_ + tileB_ * K_]; // after input
            AscendC::MatMul(preLocal, inputLocal, w_i2hLocal, false, true,
                            tileB_, H_, K_, K_, K_, K_);
            // Add bias
            for (uint32_t b = 0; b < tileB_; ++b) {
                AscendC::Add(preLocal[b * H_], preLocal[b * H_], b_i2hLocal, H_);
            }

            // tanh activation -> update h
            AscendC::Tanh(hLocal, preLocal, tileB_ * H_); // store back to hLocal

            // Compute y = h @ w_h2o^T + b_h2o
            AscendC::LocalTensor<float> yLocal = scratchLocal_[H_ * K_ + O_ * H_ + H_ + O_ + tileB_ * H_ + tileB_ * K_ + tileB_ * H_]; // after pre
            AscendC::MatMul(yLocal, hLocal, w_h2oLocal, false, true,
                            tileB_, O_, H_, H_, H_, H_);
            // Add bias
            for (uint32_t b = 0; b < tileB_; ++b) {
                AscendC::Add(yLocal[b * O_], yLocal[b * O_], b_h2oLocal, O_);
            }

            // Store y to global memory
            AscendC::DataCopy(yGm_[t * B_ * O_ + startB * O_], yLocal, tileB_ * O_);
        }

        // Store final h_new
        AscendC::DataCopy(hNewGm_[startB * H_], hLocal, tileB_ * H_);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> scratchBuf_;
    AscendC::LocalTensor<float> scratchLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> w_i2hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> hNewGm_;

    uint32_t T_, B_, I_, H_, O_, K_, tileB_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h,
                                                                 GM_ADDR b_i2h, GM_ADDR w_h2o, GM_ADDR b_h2o,
                                                                 GM_ADDR y, GM_ADDR h_new, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y, h_new,
            tiling_data.T, tiling_data.B, tiling_data.I, tiling_data.H, tiling_data.O, tiling_data.K, tiling_data.tileB);
    op.Process();
}
