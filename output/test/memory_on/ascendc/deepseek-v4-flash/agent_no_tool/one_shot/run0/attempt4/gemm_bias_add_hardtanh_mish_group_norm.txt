#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
                                GM_ADDR bias_act, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures,
                                uint32_t outFeatures, uint32_t numGroups,
                                uint32_t groupSize, uint32_t tileOut,
                                uint32_t tileIn) {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        groupSize_ = groupSize;
        tileOut_ = tileOut;
        tileIn_ = tileIn;

        // Set global buffers
        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * inFeatures_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures_ * inFeatures_);
        if (lin_bias != nullptr) {
            linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures_);
            hasLinBias_ = true;
        } else {
            hasLinBias_ = false;
        }
        biasGm_.SetGlobalBuffer((__gm__ float*)bias_act, outFeatures_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures_);
        betaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * outFeatures_);

        // Precompute constants for mish
        // No extra init needed

        // Init pipe
        // UB layout (conceptual):
        //   rowBuf:  outFeatures floats
        //   yTile:   tileOut
        //   xChunk:  tileIn
        //   wTile:   tileOut * tileIn
        //   tmpBuf:  tileOut
        //   groupSumBuf: numGroups*2 floats (mean, var)
        uint32_t rowBufSz   = outFeatures_ * sizeof(float);
        uint32_t yTileSz    = tileOut_ * sizeof(float);
        uint32_t xChunkSz   = tileIn_ * sizeof(float);
        uint32_t wTileSz    = tileOut_ * tileIn_ * sizeof(float);
        uint32_t tmpBufSz   = tileOut_ * sizeof(float);
        uint32_t groupBufSz = numGroups_ * 2 * sizeof(float);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, xChunkSz);
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, wTileSz);
        pipe_.InitBuffer(inQueueLB_, BUFFER_NUM, yTileSz);  // for lin_bias tile
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, yTileSz);
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tmpBufSz);
        pipe_.InitBuffer(rowBufQueue_, 1, rowBufSz);
        pipe_.InitBuffer(groupSumQueue_, 1, groupBufSz);
    }

    __aicore__ inline void Process() {
        if (batchSize_ == 0 || inFeatures_ == 0 || outFeatures_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Each block processes a contiguous set of rows (batch elements)
        uint32_t rowsPerBlock = (batchSize_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > batchSize_) rowEnd = batchSize_;

        // Allocate persistent UB buffers (row buf and group sums)
        AscendC::LocalTensor<float> rowBuf = rowBufQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> groupSumBuf = groupSumQueue_.AllocTensor<float>();

        for (uint32_t b = rowStart; b < rowEnd; b++) {
            ProcessOneRow(b, rowBuf, groupSumBuf);
        }

        groupSumQueue_.FreeTensor(groupSumBuf);
        rowBufQueue_.FreeTensor(rowBuf);
    }

private:
    __aicore__ inline void ProcessOneRow(uint32_t rowIdx,
                                         AscendC::LocalTensor<float>& rowBuf,
                                         AscendC::LocalTensor<float>& groupSumBuf) {
        // Step 1: Compute linear + bias_add + hardtanh + mish
        //         and store full row in rowBuf -> then write to global temp? No, keep in UB.
        // We'll compute tile by tile, accumulate in rowBuf.
        // rowBuf size = outFeatures.

        uint32_t outFeat = outFeatures_;
        uint32_t inFeat  = inFeatures_;
        uint32_t tileO   = tileOut_;
        uint32_t tileI   = tileIn_;

        // Full row base in GM for x
        uint32_t xBase = rowIdx * inFeat;

        // Outer loop over output feature tiles
        for (uint32_t oStart = 0; oStart < outFeat; oStart += tileO) {
            uint32_t curTileO = (oStart + tileO <= outFeat) ? tileO : (outFeat - oStart);
            // Accumulator for current output tile (yTile)
            AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpBuf = tmpQueue_.AllocTensor<float>();

            // Initialize yTile with lin_bias (if present) and bias
            if (hasLinBias_) {
                AscendC::LocalTensor<float> linBiasTile = inQueueLB_.AllocTensor<float>();
                AscendC::DataCopy(linBiasTile, linBiasGm_[oStart], curTileO);
                AscendC::Add(yTile, linBiasTile, biasGm_[oStart], curTileO);
                inQueueLB_.FreeTensor(linBiasTile);
            } else {
                AscendC::DataCopy(yTile, biasGm_[oStart], curTileO);
            }

            // Inner loop over input feature tiles for matmul accumulation
            for (uint32_t iStart = 0; iStart < inFeat; iStart += tileI) {
                uint32_t curTileI = (iStart + tileI <= inFeat) ? tileI : (inFeat - iStart);

                // Load x chunk
                AscendC::LocalTensor<float> xChunk = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xChunk, xGm_[xBase + iStart], curTileI);

                // Load weight tile [oStart:oStart+curTileO, iStart:iStart+curTileI]
                AscendC::LocalTensor<float> wTile = inQueueW_.AllocTensor<float>();
                // weight is stored row-major: weight[o][i]
                uint32_t wBase = oStart * inFeat + iStart;
                AscendC::DataCopy(wTile, weightGm_[wBase], curTileO * curTileI);

                // Compute: yTile[t] += sum_{i} xChunk[i] * wTile[t * curTileI + i]
                // for each output index t in [0, curTileO)
                for (uint32_t t = 0; t < curTileO; t++) {
                    float acc = 0.0f;
                    AscendC::LocalTensor<float> wRow = wTile[t * curTileI];
                    // Use vector multiply-reduce:
                    // tmpBuf = xChunk[i] * wRow[i]
                    AscendC::Mul(tmpBuf, xChunk, wRow, curTileI);
                    // Reduction sum -> scalar
                    float sumVal = 0.0f;
                    AscendC::VecReduceSum(sumVal, tmpBuf, curTileI);
                    // Add to yTile[t]
                    AscendC::LocalTensor<float> yPtr = yTile[t];
                    float cur = yPtr.GetValue(0);
                    cur += sumVal;
                    yPtr.SetValue(0, cur);
                }

                inQueueW_.FreeTensor(wTile);
                inQueueX_.FreeTensor(xChunk);
            }

            // After matmul accumulation, apply hardtanh then mish
            // hardtanh: clamp to [-1, 1]
            AscendC::Maximum(tmpBuf, yTile, -1.0f, curTileO);
            AscendC::Minimum(yTile, tmpBuf, 1.0f, curTileO);

            // mish: x * tanh(log(1 + exp(x)))
            // Compute exp(x)
            AscendC::Exp(tmpBuf, yTile, curTileO);
            // 1 + exp(x)
            AscendC::Adds(tmpBuf, tmpBuf, 1.0f, curTileO);
            // log(1 + exp(x))
            AscendC::Log(tmpBuf, tmpBuf, curTileO);
            // tanh(log(...))
            AscendC::Tanh(tmpBuf, tmpBuf, curTileO);
            // x * tanh(...)
            AscendC::Mul(yTile, yTile, tmpBuf, curTileO);

            // Store completed tile into rowBuf
            rowBuf[oStart].SetValue(yTile, curTileO);

            tmpQueue_.FreeTensor(tmpBuf);
            outQueueY_.FreeTensor(yTile);
        }

        // Step 2: Compute group_norm on rowBuf
        // reshape rowBuf to [numGroups, groupSize]
        // For each group g:
        //   mean = sum(rowBuf[g*groupSize:(g+1)*groupSize]) / groupSize
        //   var = sum((rowBuf[...] - mean)^2) / groupSize
        //   normalized = (val - mean) / sqrt(var + eps)
        //   output = normalized * gamma + beta
        float eps = 1e-5f;
        AscendC::LocalTensor<float> tmpBuf2 = tmpQueue_.AllocTensor<float>();

        for (uint32_t g = 0; g < numGroups_; g++) {
            uint32_t gStart = g * groupSize_;
            uint32_t gSize = groupSize_;

            // Compute mean
            float sumVal = 0.0f;
            AscendC::VecReduceSum(sumVal, rowBuf[gStart], gSize);
            float mean = sumVal / static_cast<float>(gSize);

            // Compute sum of (x - mean)^2
            // tmpBuf2 = rowBuf[gStart] - mean
            AscendC::Subs(tmpBuf2, rowBuf[gStart], mean, gSize);
            // square
            AscendC::Mul(tmpBuf2, tmpBuf2, tmpBuf2, gSize);
            float sqSum = 0.0f;
            AscendC::VecReduceSum(sqSum, tmpBuf2, gSize);
            float var = sqSum / static_cast<float>(gSize);
            float denom = 1.0f / sqrtf(var + eps);

            // Normalize and apply gamma, beta
            for (uint32_t i = 0; i < gSize; i++) {
                uint32_t idx = gStart + i;
                float val = rowBuf[idx].GetValue(0);
                float normalized = (val - mean) * denom;
                float outVal = normalized * gammaGm_[idx].GetValue(0) + betaGm_[idx].GetValue(0);
                rowBuf[idx].SetValue(0, outVal);
            }
        }

        tmpQueue_.FreeTensor(tmpBuf2);

        // Step 3: Write rowBuf to global memory
        AscendC::DataCopy(yGm_[rowIdx * outFeat], rowBuf, outFeat);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueLB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> rowBufQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> groupSumQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_{0};
    uint32_t inFeatures_{0};
    uint32_t outFeatures_{0};
    uint32_t numGroups_{0};
    uint32_t groupSize_{0};
    uint32_t tileOut_{0};
    uint32_t tileIn_{0};
    bool hasLinBias_{false};
};

}
