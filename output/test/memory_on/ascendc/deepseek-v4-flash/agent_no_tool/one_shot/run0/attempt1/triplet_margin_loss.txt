namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per core UB budget
static constexpr uint32_t ALIGN_ELEMS = 16;               // vectorization alignment
static constexpr uint32_t HARD_CAP_PAIRS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;
    const gert::Shape *shape = context->GetInputShape(0); // anchor shape
    const auto &origin = shape->GetOriginShape();
    uint32_t totalPairs = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t d = static_cast<uint32_t>(origin.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float margin = 0.0f;
    if (attrs->GetAttrNum() > 0) {
        const float *p = attrs->GetAttrPointer<float>(0);
        if (p) margin = *p;
    }

    // Per-pair memory: anchor (d), positive (d), negative (d), diff/sqr temp (d) -> 4*d floats
    uint32_t perPairBytes = 4 * d * sizeof(float);
    uint32_t maxPairs = perPairBytes > 0 ? UB_BUDGET_BYTES / perPairBytes : 1;
    maxPairs = AlignDown(maxPairs, ALIGN_ELEMS);
    if (maxPairs < 1) maxPairs = 1;
    if (maxPairs > HARD_CAP_PAIRS) maxPairs = AlignDown(HARD_CAP_PAIRS, ALIGN_ELEMS);

    uint32_t tileLen = std::min(maxPairs, totalPairs);
    if (tileLen == 0) tileLen = 1;

    uint32_t blockDim = 1; // single block to avoid reduction complexity

    context->SetBlockDim(blockDim);

    tiling.set_totalPairs(totalPairs);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_d(d);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // for double buffering

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t totalPairs, uint32_t tileLen,
                                uint32_t blockDim, uint32_t d, float margin)
    {
        totalPairs_ = totalPairs;
        tileLen_ = tileLen;
        blockDim_ = blockDim;
        d_ = d;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        // Allocate double buffering for three input vectors and one temp vector
        pipe_.InitBuffer(inAnchorQ_, BUFFER_NUM, d_ * sizeof(float));
        pipe_.InitBuffer(inPositiveQ_, BUFFER_NUM, d_ * sizeof(float));
        pipe_.InitBuffer(inNegativeQ_, BUFFER_NUM, d_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, d_ * sizeof(float));       // for diff/sqr reuse
        pipe_.InitBuffer(tmpOutQ_, 1, 1 * sizeof(float));     // output scalar
    }

    __aicore__ inline void Process()
    {
        if (totalPairs_ == 0) return;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx != 0) return; // blockDim==1, only block 0 works

        float sumLoss = 0.0f;
        uint32_t processed = 0;
        while (processed < totalPairs_) {
            uint32_t batch = tileLen_;
            if (batch > totalPairs_ - processed) batch = totalPairs_ - processed;

            // Double buffering load of first pair
            uint32_t pairIdx = processed;
            AscendC::LocalTensor<float> a0 = inAnchorQ_.AllocTensor<float>();
            AscendC::DataCopy(a0, anchorGm_[pairIdx * d_], d_);
            inAnchorQ_.EnQue(a0);
            AscendC::LocalTensor<float> p0 = inPositiveQ_.AllocTensor<float>();
            AscendC::DataCopy(p0, positiveGm_[pairIdx * d_], d_);
            inPositiveQ_.EnQue(p0);
            AscendC::LocalTensor<float> n0 = inNegativeQ_.AllocTensor<float>();
            AscendC::DataCopy(n0, negativeGm_[pairIdx * d_], d_);
            inNegativeQ_.EnQue(n0);

            uint32_t next = (batch > 1) ? 1 : 0;
            // Prefetch second pair if any
            if (next < batch) {
                pairIdx = processed + 1;
                AscendC::LocalTensor<float> a1 = inAnchorQ_.AllocTensor<float>();
                AscendC::DataCopy(a1, anchorGm_[pairIdx * d_], d_);
                inAnchorQ_.EnQue(a1);
                AscendC::LocalTensor<float> p1 = inPositiveQ_.AllocTensor<float>();
                AscendC::DataCopy(p1, positiveGm_[pairIdx * d_], d_);
                inPositiveQ_.EnQue(p1);
                AscendC::LocalTensor<float> n1 = inNegativeQ_.AllocTensor<float>();
                AscendC::DataCopy(n1, negativeGm_[pairIdx * d_], d_);
                inNegativeQ_.EnQue(n1);
            }

            // Process batch with double buffering
            uint32_t current = 0;
            while (current < batch) {
                // Dequeue current pair
                AscendC::LocalTensor<float> aCur = inAnchorQ_.DeQue<float>();
                AscendC::LocalTensor<float> pCur = inPositiveQ_.DeQue<float>();
                AscendC::LocalTensor<float> nCur = inNegativeQ_.DeQue<float>();

                // Compute loss for this pair
                float loss = computePairLoss(aCur, pCur, nCur);
                sumLoss += loss;

                // Free current pair buffers
                inAnchorQ_.FreeTensor(aCur);
                inPositiveQ_.FreeTensor(pCur);
                inNegativeQ_.FreeTensor(nCur);

                current++;

                // Prefetch next pair
                uint32_t nextIdx = processed + current + 1;
                if (nextIdx < processed + batch) {
                    AscendC::LocalTensor<float> aNext = inAnchorQ_.AllocTensor<float>();
                    AscendC::DataCopy(aNext, anchorGm_[nextIdx * d_], d_);
                    inAnchorQ_.EnQue(aNext);
                    AscendC::LocalTensor<float> pNext = inPositiveQ_.AllocTensor<float>();
                    AscendC::DataCopy(pNext, positiveGm_[nextIdx * d_], d_);
                    inPositiveQ_.EnQue(pNext);
                    AscendC::LocalTensor<float> nNext = inNegativeQ_.AllocTensor<float>();
                    AscendC::DataCopy(nNext, negativeGm_[nextIdx * d_], d_);
                    inNegativeQ_.EnQue(nNext);
                }
            }
            processed += batch;
        }

        // Compute mean loss
        float meanLoss = sumLoss / static_cast<float>(totalPairs_);

        // Write output
        AscendC::LocalTensor<float> outLocal = tmpOutQ_.AllocTensor<float>();
        outLocal(0) = meanLoss;
        AscendC::DataCopy(lossGm_, outLocal, 1);
        tmpOutQ_.FreeTensor(outLocal);
    }

private:
    __aicore__ inline float computePairLoss(
        const AscendC::LocalTensor<float>& a,
        const AscendC::LocalTensor<float>& p,
        const AscendC::LocalTensor<float>& n)
    {
        // d(anchor, positive) = sqrt( sum( (a-p)^2 ) )
        // d(anchor, negative) = sqrt( sum( (a-n)^2 ) )
        // loss = max(d_ap - d_an + margin, 0)
        AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>(); // reuse for diff & sqr

        // 1) a - p
        AscendC::Sub(tmp, a, p, d_);
        // square
        AscendC::Mul(tmp, tmp, tmp, d_);
        float sumAp = scalarReduce(tmp);

        // 2) a - n
        AscendC::Sub(tmp, a, n, d_);
        AscendC::Mul(tmp, tmp, tmp, d_);
        float sumAn = scalarReduce(tmp);

        tmpQ_.FreeTensor(tmp);

        float dAp = AscendC::Sqrt(sumAp);
        float dAn = AscendC::Sqrt(sumAn);
        float raw = dAp - dAn + margin_;
        return (raw > 0.0f) ? raw : 0.0f;
    }

    __aicore__ inline float scalarReduce(const AscendC::LocalTensor<float>& vec)
    {
        // Manual reduction: sum over d_ elements
        float sum = 0.0f;
        for (uint32_t i = 0; i < d_; i++) {
            sum += vec(i);
        }
        return sum;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inAnchorQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inPositiveQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inNegativeQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> tmpOutQ_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalPairs_{0};
    uint32_t tileLen_{1};
    uint32_t blockDim_{1};
    uint32_t d_{1};
    float margin_{0.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(
    GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
    GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalPairs, tiling_data.tileLen,
            tiling_data.blockDim, tiling_data.d, tiling_data.margin);
    op.Process();
}

class ModelNew(torch.nn.Module):

        # Extract scalar margin from tensor
        margin = float(margin_tensor.reshape(-1)[0].item()) if margin_tensor.numel() else 0.0
        fn = getattr(custom_ops_lib, _FN)
