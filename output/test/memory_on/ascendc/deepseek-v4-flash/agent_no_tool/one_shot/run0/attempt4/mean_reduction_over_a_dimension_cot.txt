static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    uint32_t ndim = inputShape->GetOriginShape().GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t reduceDim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (reduceDim < 0) reduceDim += ndim;
    uint32_t redIdx = static_cast<uint32_t>(reduceDim);

    uint32_t totalInputLength = 1;
    uint32_t outputLength = 1;
    uint32_t reduceLength = 1;
    uint32_t strideAfter = 1;
    for (uint32_t i = 0; i < ndim; ++i) {
        uint32_t dimSize = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(i));
        totalInputLength *= dimSize;
        if (i < redIdx) {
            outputLength *= dimSize;
        } else if (i == redIdx) {
            reduceLength = dimSize;
        } else {
            strideAfter *= dimSize;
        }
    }
    if (reduceLength == 0) {
        return ge::GRAPH_FAILED;
    }
    outputLength *= strideAfter;

    tiling.set_totalInputLength(totalInputLength);
    tiling.set_outputLength(outputLength);
