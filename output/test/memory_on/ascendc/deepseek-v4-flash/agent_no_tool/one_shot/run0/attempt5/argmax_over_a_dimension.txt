host_operator_src = """
#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::StorageShape *xShape = context->GetInputShape(0);
    const auto &shapeVec = xShape->GetOriginShape().GetDims();
    uint32_t totalElements = 1;
    for (auto d : shapeVec) totalElements *= d;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dimRaw = static_cast<int64_t>(attrs->GetAttrNum(0));
    int32_t dim = static_cast<int32_t>(dimRaw);
    int32_t ndim = static_cast<int32_t>(shapeVec.size());
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= ndim) dim = 0;

    uint32_t rows = 1;
    uint32_t cols = shapeVec[dim];
    for (int32_t i = 0; i < ndim; i++) {
        if (i == dim) continue;
        rows *= shapeVec[i];
    }

    uint32_t maxFloatsPerTile = UB_BUDGET_BYTES / sizeof(float);
    uint32_t tileRows = maxFloatsPerTile / cols;
    if (tileRows < 1) tileRows = 1;
    if (tileRows > rows) tileRows = rows;

    uint32_t blockDim = (rows + tileRows - 1) / tileRows;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalElements(totalElements);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileRows(tileRows);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
