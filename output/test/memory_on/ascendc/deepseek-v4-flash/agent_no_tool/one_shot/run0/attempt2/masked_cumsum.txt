host_operator_src='''
#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: 2 input tiles (x, mask) + 1 output tile = 12 bytes per element
static constexpr uint32_t UB_BUDGET_BYTES = 256 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
  return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  MaskedCumsumCustomTilingData tiling;

  const auto *shapeX = context->GetInputShape(0);
  const uint32_t totalLength =
      static_cast<uint32_t>(shapeX->GetOriginShape().GetShapeSize());

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
  int64_t dim = *dimPtr;

  const auto &originShape = shapeX->GetOriginShape();
  int64_t ndim = originShape.GetDimNum();

  // Normalise dim
  if (dim < 0) dim += ndim;
  if (dim < 0 || dim >= ndim) dim = 0;  // fallback

  // innerSize = shape[dim]
  uint32_t innerSize = 1;
  if (ndim > 0) {
    for (int64_t i = 0; i < ndim; i++) {
      uint32_t d = static_cast<uint32_t>(originShape.GetDim(i));
      if (i == dim) innerSize = d;
    }
  }
  if (innerSize == 0) innerSize = 1;  // avoid division by zero

  // Max elements that fit in UB (2 input tiles in flight)
  uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

  // Ensure tile length is a multiple of innerSize
  uint32_t tileLen = maxTile;
  if (totalLength < tileLen) {
    tileLen = AlignDown(totalLength, innerSize);
    if (tileLen == 0) tileLen = innerSize;
  } else {
    uint32_t numRows = tileLen / innerSize;
    if (numRows == 0) numRows = 1;
    tileLen = numRows * innerSize;
  }

  // Block dimension: number of rows per block (each row independent)
  uint32_t totalRows = CeilDiv(totalLength, innerSize);
  uint32_t rowsPerBlock = CeilDiv(tileLen, innerSize);
  uint32_t blockDim = CeilDiv(totalRows, rowsPerBlock);
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_totalLength(totalLength);
  tiling.set_tileLength(tileLen);
  tiling.set_innerSize(innerSize);
  tiling.set_blockDim(blockDim);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *workspace = context->GetWorkspaceSizes(1);
  workspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape *x_shape = context->GetInputShape(0);
  gert::Shape *y_shape = context->GetOutputShape(0);
  *y_shape = *x_shape;
  return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
  const ge::DataType x_dtype = context->GetInputDataType(0);
  context->SetOutputDataType(0, x_dtype);
  return GRAPH_SUCCESS;
}
}  // namespace ge

namespace ops {
class MaskedCumsumCustom : public OpDef {
public:
  explicit MaskedCumsumCustom(const char* name) : OpDef(name) {
    this->Input("x")
        .ParamType(REQUIRED)

        .Format({ge::FORMAT_ND})

    this->Input("mask")
        .ParamType(REQUIRED)

    this->Output("y")
        .ParamType(REQUIRED)

    this->Attr("dim").AttrType(OPTIONAL).Int(0);

    this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
    this->AICore().SetTiling(optiling::TilingFunc);

  }
};

OP_ADD(MaskedCumsumCustom);
}  // namespace ops

kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsum {
public:
  __aicore__ inline KernelMaskedCumsum() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                              uint32_t totalLength, uint32_t tileLength,
                              uint32_t innerSize) {
    totalLen_ = totalLength;
    tileLen_ = tileLength;
    innerSize_ = innerSize;

    xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
    maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    // Grid stride over rows (each row = innerSize elements)
    const uint32_t rowStride = blockNum * (tileLen_ / innerSize_);
    uint32_t rowBase = blockIdx * (tileLen_ / innerSize_);  // row index

    // Load first tile
    uint32_t startElem = rowBase * innerSize_;
    if (startElem >= totalLen_) return;
    uint32_t curLen = totalLen_ - startElem;
    if (curLen > tileLen_) curLen = tileLen_;
    CopyIn(startElem, curLen);

    while (true) {
      uint32_t nextRowBase = rowBase + rowStride;
      uint32_t nextStartElem = nextRowBase * innerSize_;
      uint32_t nextLen = 0;
      if (nextStartElem < totalLen_) {
        nextLen = totalLen_ - nextStartElem;
        if (nextLen > tileLen_) nextLen = tileLen_;
        // Prefetch next tile (enqueue only after current is dequeued)
        // We'll enqueue after Compute/CopyOut of current.
      }

      Compute(curLen);
      CopyOut(startElem, curLen);

      if (nextLen == 0) break;

      // Now enqueue next tile
      CopyIn(nextStartElem, nextLen);
      rowBase = nextRowBase;
      startElem = nextStartElem;
      curLen = nextLen;
    }
  }

private:
  __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
    AscendC::DataCopy(xLocal, xGm_[gmOffset], len);

    AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
    AscendC::DataCopy(maskLocal, maskGm_[gmOffset], len);

    inQueueX_.EnQue(xLocal);
    inQueueMask_.EnQue(maskLocal);
  }

  __aicore__ inline void Compute(uint32_t len) {
    AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
    AscendC::LocalTensor<float> maskLocal = inQueueMask_.DeQue<float>();

    // y = x * mask
    AscendC::Mul(xLocal, xLocal, maskLocal, len);
    inQueueMask_.FreeTensor(maskLocal);

    // Per-row cumulative sum
    uint32_t numRows = len / innerSize_;
    for (uint32_t r = 0; r < numRows; ++r) {
      uint32_t rowStart = r * innerSize_;
      for (uint32_t i = 1u; i < innerSize_; ++i) {
        float prev = xLocal.GetValue(rowStart + i - 1);
        float curr = xLocal.GetValue(rowStart + i);
        xLocal.SetValue(rowStart + i, curr + prev);
      }
    }

    outQueueY_.EnQue(xLocal);
  }

  __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
    AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
    AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
    outQueueY_.FreeTensor(yLocal);
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> maskGm_;
  AscendC::GlobalTensor<float> yGm_;

  uint32_t totalLen_ {0};
  uint32_t tileLen_ {0};
  uint32_t innerSize_ {1};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x,
                                                           GM_ADDR mask,
                                                           GM_ADDR y,
                                                           GM_ADDR workspace,
                                                           GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelMaskedCumsum op;
  op.Init(x, mask, y,
          tiling_data.totalLength,
          tiling_data.tileLength,
          tiling_data.innerSize);
  op.Process();
}
'''
