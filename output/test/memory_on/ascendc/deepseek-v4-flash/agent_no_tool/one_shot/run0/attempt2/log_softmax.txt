host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    // Fix: use StorageShape and GetOriginShape properly
    const gert::StorageShape *x_storage_shape = context->GetInputShape(0);
    const gert::Shape &x_origin_shape = x_storage_shape->GetOriginShape();
    const uint32_t totalLength =
        static_cast<uint32_t>(x_origin_shape.GetShapeSize());
    const uint32_t ndim = x_origin_shape.GetDimNum();
    // Assume ND input with ndim >= 2, reduce last dim
    const uint32_t rowLength = static_cast<uint32_t>(x_origin_shape.GetDim(ndim - 1));
    const uint32_t numRows = totalLength / rowLength;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = static_cast<int32_t>(
        attrs->GetAttrPointer<int32_t>(0)[0]);  // attr dim

    // tileLen: we need 3 tile buffers (in, out, tmp) -> 3*tileLen floats = 12*tileLen bytes
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        uint32_t aligned = AlignDown(rowLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : rowLength;
    }

    // blockDim: assign rows to blocks
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_numRows(numRows);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

public:

            .UnknownShapeFormat({ge::FORMAT_ND});

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t tileLength, uint32_t numRows,
                                int32_t dim)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        numRows_ = numRows;
        dim_ = dim;  // not used, assume reduction on last axis

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || numRows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride over rows
        for (uint32_t row = blockIdx; row < numRows_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        uint32_t baseOffset = rowIdx * rowLen_;

        // ---------- Pass 1: Find global maximum ----------
        float maxVal = -__builtin_huge_valf(); // -inf
        uint32_t offset = baseOffset;
        uint32_t remaining = rowLen_;
        uint32_t curTileLen = 0;

        while (remaining > 0) {
            curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curTileLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // Compute tile max
            AscendC::LocalTensor<float> tileMaxLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceMax(tileMaxLocal, xLocal, curTileLen);
            float tileMax = tileMaxLocal(0);  // scalar
            tmpQueue_.FreeTensor(tileMaxLocal);

            if (tileMax > maxVal) maxVal = tileMax;

            inQueueX_.FreeTensor(xLocal);
            offset += curTileLen;
            remaining -= curTileLen;
        }

        // ---------- Pass 2: Compute sum of exp(x - max) ----------
        float sumExp = 0.0f;
        offset = baseOffset;
        remaining = rowLen_;

        while (remaining > 0) {
            curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curTileLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // x_tmp = x - maxVal
            AscendC::LocalTensor<float> xTmp = tmpQueue_.AllocTensor<float>();
            AscendC::Subs(xTmp, xLocal, maxVal, curTileLen);

            // exp(x_tmp)
            AscendC::Exp(xTmp, xTmp, curTileLen);

            // sum over tile
            AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumLocal, xTmp, curTileLen);
            sumExp += sumLocal(0);

            tmpQueue_.FreeTensor(xTmp);
            tmpQueue_.FreeTensor(sumLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curTileLen;
            remaining -= curTileLen;
        }

        // log_sum = log(sumExp)
        float logSum = logf(sumExp);

        // ---------- Pass 3: Compute output ----------
        offset = baseOffset;
        remaining = rowLen_;

        while (remaining > 0) {
            curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // Load x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curTileLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // y = x - maxVal - logSum
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Subs(yLocal, xLocal, maxVal, curTileLen);
            AscendC::Subs(yLocal, yLocal, logSum, curTileLen);

            // Store y
            AscendC::DataCopy(yGm_[offset], yLocal, curTileLen);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            offset += curTileLen;
            remaining -= curTileLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numRows_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLength,
            tiling_data.tileLength, tiling_data.numRows, tiling_data.dim);
    op.Process();
}
