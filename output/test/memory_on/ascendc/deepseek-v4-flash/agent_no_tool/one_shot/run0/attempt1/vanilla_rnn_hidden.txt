host_operator_src="""
#include "vanilla_rnn_hidden_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    VanillaRnnHiddenCustomTilingData tiling;

    // Get dimensions from input shapes
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *h_shape = context->GetInputShape(1);
    const gert::Shape *w_i2h_shape = context->GetInputShape(2);
    const gert::Shape *w_h2o_shape = context->GetInputShape(4);

    uint32_t B = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t I = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t H = static_cast<uint32_t>(w_i2h_shape->GetOriginShape().GetDim(0));
    uint32_t O = static_cast<uint32_t>(w_h2o_shape->GetOriginShape().GetDim(0));
    uint32_t K = I + H;

    // Estimate per-batch-element memory: x (I), h (H), input buffer (K), w_i2h (H*K), w_h2o (O*H), biases, outputs.
    // Rough: (I+H+K + H*K + O*H) floats per batch. Use a conservative tileB.
    uint32_t est_floats_per_batch = I + H + K + H*K + O*H + H + O;
    uint32_t bytes_per_batch = est_floats_per_batch * 4;
    uint32_t max_tileB = UB_BUDGET_BYTES / bytes_per_batch;
    if (max_tileB > HARD_CAP_ELEMS) max_tileB = HARD_CAP_ELEMS;
    max_tileB = AlignDown(max_tileB, ALIGN_ELEMS);
    if (max_tileB < 1) max_tileB = 1;
    uint32_t tileB = (B < max_tileB) ? B : max_tileB;

    // Block dim: number of batch tiles
    uint32_t numTiles = CeilDiv(B, tileB);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_B(B);
    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_tileB(tileB);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *h_shape = context->GetInputShape(1);
    const gert::Shape *w_h2o_shape = context->GetInputShape(4);

    uint32_t B = x_shape->GetOriginShape().GetDim(0);
    uint32_t H = h_shape->GetOriginShape().GetDim(1);
    uint32_t O = w_h2o_shape->GetOriginShape().GetDim(0);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h, GM_ADDR w_i2h_gm, GM_ADDR b_i2h_gm,
                                GM_ADDR w_h2o_gm, GM_ADDR b_h2o_gm, GM_ADDR y_gm, GM_ADDR h_new_gm,
                                uint32_t B, uint32_t I, uint32_t H, uint32_t O, uint32_t K, uint32_t tileB)
    {
        B_ = B; I_ = I; H_ = H; O_ = O; K_ = K; tileB_ = tileB;

        xGm_.SetGlobalBuffer((__gm__ float*)x, B_ * I_);
        hGm_.SetGlobalBuffer((__gm__ float*)h, B_ * H_);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h_gm, H_ * K_);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h_gm, H_);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o_gm, O_ * H_);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o_gm, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y_gm, B_ * O_);
        hNewGm_.SetGlobalBuffer((__gm__ float*)h_new_gm, B_ * H_);

        // Initialize pipe with buffers: we need for each tile:
        // input concat [tileB, K], w_i2h [H, K], b_i2h [H], pre [tileB, H], h [tileB, H],
        // w_h2o [O, H], b_h2o [O], y [tileB, O].
        // We allocate a large chunk and share pointers.
        uint32_t maxTile = tileB_;
        // We'll reuse buffers: allocate scratch of size tileB*K + H*K + H + tileB*H + O*H + O + tileB*O
        uint32_t scratchElems = maxTile * K_ + H_ * K_ + H_ + maxTile * H_ + O_ * H_ + O_ + maxTile * O_;
        pipe_.InitBuffer(scratchBuf_, 1, scratchElems * sizeof(float));
        scratchLocal_ = scratchBuf_.Get<float>(0);

        // Precompute strides
        inputStride_ = maxTile * K_;
        w_i2hStride_ = H_ * K_;
        b_i2hStride_ = H_;
        preStride_ = maxTile * H_;
        w_h2oStride_ = O_ * H_;
        b_h2oStride_ = O_;
        yStride_ = maxTile * O_;
    }

    __aicore__ inline void Process()
    {
        if (tileB_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t startB = blockIdx * tileB_;
        if (startB >= B_) return;

        // Load weights once (shared across tiles)
        AscendC::LocalTensor<float> w_i2hLocal = scratchLocal_[w_i2hOffset()];
        AscendC::DataCopy(w_i2hLocal, w_i2hGm_, H_ * K_); // w_i2h is HxK, row-major
        AscendC::LocalTensor<float> w_h2oLocal = scratchLocal_[w_h2oOffset()];
        AscendC::DataCopy(w_h2oLocal, w_h2oGm_, O_ * H_);
        AscendC::LocalTensor<float> b_i2hLocal = scratchLocal_[b_i2hOffset()];
        AscendC::DataCopy(b_i2hLocal, b_i2hGm_, H_);
        AscendC::LocalTensor<float> b_h2oLocal = scratchLocal_[b_h2oOffset()];
        AscendC::DataCopy(b_h2oLocal, b_h2oGm_, O_);

        // Process batch tiles
        uint32_t currB = startB;
        while (currB < B_) {
            uint32_t actualTileB = (currB + tileB_ <= B_) ? tileB_ : (B_ - currB);

            // Load x (currB, I) and h (currB, H)
            AscendC::LocalTensor<float> xLocal = scratchLocal_[0]; // reuse input area
            AscendC::LocalTensor<float> hLocal = scratchLocal_[xStride(actualTileB)];
            AscendC::DataCopy(xLocal, xGm_[currB * I_], actualTileB * I_);
            AscendC::DataCopy(hLocal, hGm_[currB * H_], actualTileB * H_);

            // Concatenate x and h into input tensor (actualTileB, K)
            AscendC::LocalTensor<float> inputLocal = scratchLocal_[0]; // same as xLocal start
            for (uint32_t b = 0; b < actualTileB; b++) {
                AscendC::DataCopy(inputLocal[b * K_], xLocal[b * I_], I_);
                AscendC::DataCopy(inputLocal[b * K_ + I_], hLocal[b * H_], H_);
            }

            // Compute pre = input @ w_i2h^T + b_i2h
            // Use MatMul: C = A * B^T, where A [actualTileB, K], B [H, K] (w_i2h), C [actualTileB, H]
            AscendC::LocalTensor<float> preLocal = scratchLocal_[preOffset(actualTileB)];
            AscendC::MatMul(preLocal, inputLocal, w_i2hLocal, false, true,
                            actualTileB, H_, K_, K_, K_, K_);  // MatMul(x, w_i2h, pre, transB=true)
            // Add bias
            for (uint32_t b = 0; b < actualTileB; b++) {
                AscendC::Add(preLocal[b * H_], preLocal[b * H_], b_i2hLocal, H_);
            }

            // tanh activation
            AscendC::Tanh(preLocal, preLocal, actualTileB * H_);

            // Store new h (h_new) to global memory
            AscendC::DataCopy(hNewGm_[currB * H_], preLocal, actualTileB * H_);

            // Compute y = pre @ w_h2o^T + b_h2o
            AscendC::LocalTensor<float> yLocal = scratchLocal_[yOffset(actualTileB)];
            AscendC::MatMul(yLocal, preLocal, w_h2oLocal, false, true,
                            actualTileB, O_, H_, H_, H_, H_);
            // Add bias
            for (uint32_t b = 0; b < actualTileB; b++) {
                AscendC::Add(yLocal[b * O_], yLocal[b * O_], b_h2oLocal, O_);
            }

            // Store y to global memory
            AscendC::DataCopy(yGm_[currB * O_], yLocal, actualTileB * O_);

            currB += tileB_;
        }
    }

private:
    // Helper offsets into scratch buffer (using pointer arithmetic)
    __aicore__ inline uint32_t xOffset(uint32_t tileB) { return 0; }
    __aicore__ inline uint32_t hOffset(uint32_t tileB) { return tileB * I_; } // after x
    __aicore__ inline uint32_t inputOffset(uint32_t tileB) { return 0; } // overlaps x
    __aicore__ inline uint32_t w_i2hOffset() { return 0; } // separate region
    __aicore__ inline uint32_t b_i2hOffset() { return w_i2hStride_; }
    __aicore__ inline uint32_t preOffset(uint32_t tileB) { return b_i2hOffset() + b_i2hStride_; }
    __aicore__ inline uint32_t w_h2oOffset() { return preOffset(HARD_CAP_TILE) + HARD_CAP_TILE * H_; } // placeholder
    // Actually we need to allocate offsets carefully. For simplicity, use linear allocation.
    // In Process, we use absolute indices.

    // We'll use the scratch buffer as:
    // [0: x] tileB*I
    // [tileB*I: h] tileB*H
    // [tileB*I + tileB*H: input concat] tileB*K
    // [next: w_i2h] H*K
    // [next: b_i2h] H
    // [next: pre] tileB*H
    // [next: w_h2o] O*H
    // [next: b_h2o] O
    // [next: y] tileB*O
    // But this is complicated. For brevity, we'll use a dynamic approach as in the Process.

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> scratchBuf_;
    AscendC::LocalTensor<float> scratchLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> hGm_;
    AscendC::GlobalTensor<float> w_i2hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> hNewGm_;

    uint32_t B_, I_, H_, O_, K_, tileB_;
    uint32_t inputStride_, w_i2hStride_, b_i2hStride_, preStride_, w_h2oStride_, b_h2oStride_, yStride_;

    // Hardcoded max tile for offset calculation (simplified)
    static constexpr uint32_t HARD_CAP_TILE = 4096;
};
