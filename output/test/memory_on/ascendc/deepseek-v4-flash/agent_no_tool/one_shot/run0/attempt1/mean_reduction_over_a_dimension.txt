host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model:
//   inQueue: 1 tile (for load) + accumulator = 2 * tilePost floats = 8 * tilePost bytes
//   plus temporary for division
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t MIN_TILE_POST = 256;
static constexpr uint32_t MAX_TILE_POST = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;
    const gert::Shape *inShape = context->GetInputShape(0);
    const gert::Shape *outShape = context->GetOutputShape(0);
    int64_t totalIn = inShape->GetOriginShape().GetShapeSize();
    int64_t totalOut = outShape->GetOriginShape().GetShapeSize();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    // dim must be valid non-negative
    int32_t ndim = inShape->GetOriginShape().GetDimNum();
    if (dim < 0) dim += ndim;
    uint32_t preSize = 1;
    uint32_t dimSize = 1;
    uint32_t postSize = 1;
    for (int32_t i = 0; i < ndim; ++i) {
        uint32_t s = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(i));
        if (i < dim) preSize *= s;
        else if (i == dim) dimSize = s;
        else postSize *= s;
    }

    // Tile over postSize (the inner dimensions after reduction)
    // Each tile: load buffer + accumulator = 2 * tilePost floats
    uint32_t maxFloats = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxFloats = AlignDown(maxFloats, 128); // align to 128 floats for DMA
    if (maxFloats < MIN_TILE_POST) maxFloats = MIN_TILE_POST;
    if (maxFloats > MAX_TILE_POST) maxFloats = MAX_TILE_POST;

    uint32_t tilePost = maxFloats;
    if (tilePost > postSize) tilePost = postSize;
    // ensure tilePost is at least 1
    if (tilePost == 0) tilePost = 1;

    // Block dimension: aim for moderate occupancy
    uint32_t numOuter = CeilDiv(preSize, 4u); // rough tiles of outer
    uint32_t blockDim = numOuter;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalIn));
    tiling.set_outputLength(static_cast<uint32_t>(totalOut));
    tiling.set_preSize(preSize);
    tiling.set_dimSize(dimSize);
    tiling.set_postSize(postSize);
    tiling.set_tilePost(tilePost);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    int32_t ndim = x_shape->GetOriginShape().GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;

    gert::Shape y_shape;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i != dim) {
            y_shape.AppendDim(x_shape->GetOriginShape().GetDim(i));
        }
    }
    context->SetOutputShape(0, y_shape);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:

            .ParamType(REQUIRED)

            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int(0);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReduction {
public:
    __aicore__ inline KernelMeanReduction() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outputLength,
                                uint32_t preSize, uint32_t dimSize,
                                uint32_t postSize, uint32_t tilePost)
    {
        totalLen_ = totalLength;
        outputLen_ = outputLength;
        preSize_ = preSize;
        dimSize_ = dimSize;
        postSize_ = postSize;
        tilePost_ = tilePost;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputLen_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tilePost_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1, tilePost_ * sizeof(float));
        pipe_.InitBuffer(delayQueue_, 1, tilePost_ * sizeof(float)); // not used but placeholder
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || dimSize_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Each block handles a disjoint set of outer indices (preSize)
        uint32_t outerPerBlock = (preSize_ + blockNum - 1) / blockNum;
        uint32_t outerStart = blockIdx * outerPerBlock;
        uint32_t outerEnd = outerStart + outerPerBlock;
        if (outerStart >= preSize_) return;
        if (outerEnd > preSize_) outerEnd = preSize_;

        for (uint32_t i = outerStart; i < outerEnd; ++i) {
            // For this outer index, process all postSize tiles
            for (uint32_t t = 0; t < postSize_; t += tilePost_) {
                uint32_t curTile = postSize_ - t;
                if (curTile > tilePost_) curTile = tilePost_;

                // Initialize accumulator to zero
                AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();
                AscendC::Duplicate(acc, 0.0f, curTile);

                // Loop over reduction dimension rows
                for (uint32_t r = 0; r < dimSize_; ++r) {
                    // Load one row segment
                    AscendC::LocalTensor<float> row = inQueue_.AllocTensor<float>();
                    uint32_t offset = i * dimSize_ * postSize_ + r * postSize_ + t;
                    AscendC::DataCopy(row, xGm_[offset], curTile);
                    // Accumulate
                    AscendC::Add(acc, acc, row, curTile);
                    inQueue_.FreeTensor(row);
                }

                // Divide by dimSize
                const float invDim = 1.0f / static_cast<float>(dimSize_);
                AscendC::Muls(acc, acc, invDim, curTile);

                // Store to output
                uint32_t outOffset = i * postSize_ + t;
                AscendC::DataCopy(yGm_[outOffset], acc, curTile);
                accQueue_.FreeTensor(acc);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> delayQueue_; // unused but reserved

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outputLen_ {0};
    uint32_t preSize_ {0};
    uint32_t dimSize_ {0};
    uint32_t postSize_ {0};
    uint32_t tilePost_ {0};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReduction op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.outputLength,
            tiling_data.preSize, tiling_data.dimSize,
            tiling_data.postSize, tiling_data.tilePost);
    op.Process();
}
