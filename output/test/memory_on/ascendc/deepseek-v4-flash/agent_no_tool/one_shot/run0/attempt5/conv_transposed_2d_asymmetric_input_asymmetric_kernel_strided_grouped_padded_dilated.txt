kernel_src="""

#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    uint32_t n = td.n;
    uint32_t cIn = td.cIn;
    uint32_t hIn = td.hIn;
    uint32_t wIn = td.wIn;
    uint32_t cOutPerGroup = td.cOutPerGroup;
    uint32_t kH = td.kH;
    uint32_t kW = td.kW;
    uint32_t strideH = td.strideH;
    uint32_t strideW = td.strideW;
    uint32_t paddingH = td.paddingH;
    uint32_t paddingW = td.paddingW;
    uint32_t dilationH = td.dilationH;
    uint32_t dilationW = td.dilationW;
    uint32_t groups = td.groups;
    uint32_t outH = td.outH;
    uint32_t outW = td.outW;
    uint32_t outC = td.outC;

    float* x_ptr = (float*)x;
    float* w_ptr = (float*)weight;
    float* y_ptr = (float*)y;

    for (uint32_t b = 0; b < n; ++b) {
        for (uint32_t co = 0; co < outC; ++co) {
            uint32_t g = co / cOutPerGroup;
            uint32_t coInGroup = co % cOutPerGroup;
            uint32_t ciStart = g * (cIn / groups);
            for (uint32_t oh = 0; oh < outH; ++oh) {
                for (uint32_t ow = 0; ow < outW; ++ow) {
                    float sum = 0.0f;
                    for (uint32_t ci = 0; ci < cIn / groups; ++ci) {
                        uint32_t input_channel = ciStart + ci;
                        for (uint32_t kh = 0; kh < kH; ++kh) {
                            for (uint32_t kw = 0; kw < kW; ++kw) {
                                int32_t ih = (int32_t)oh + (int32_t)paddingH - (int32_t)(dilationH * kh);
                                int32_t iw = (int32_t)ow + (int32_t)paddingW - (int32_t)(dilationW * kw);
                                if (ih % (int32_t)strideH == 0 && iw % (int32_t)strideW == 0) {
                                    ih /= (int32_t)strideH;
                                    iw /= (int32_t)strideW;
                                    if (ih >= 0 && ih < (int32_t)hIn && iw >= 0 && iw < (int32_t)wIn) {
                                        float x_val = x_ptr[(b * cIn + input_channel) * (hIn * wIn) + ih * wIn + iw];
                                        float w_val = w_ptr[(input_channel * cOutPerGroup + coInGroup) * (kH * kW) + kh * kW + kw];
                                        sum += x_val * w_val;
                                    }
                                }
                            }
                        }
                    }
                    y_ptr[(b * outC + co) * (outH * outW) + oh * outW + ow] = sum;
                }
            }
        }
    }
}
