static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // Compute output shape: N, Cout, H_out, W_out
    // H_out = (inH-1)*strideH - 2*paddingH + dilationH*(kernelH-1) + outputPaddingH + 1
    // W_out similarly
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stride = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *padding = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *outputPadding = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *dilation = attrs->GetAttrPointer<int32_t>(4);

    int64_t N = x_shape->GetDim(0);
    int64_t Cout = w_shape->GetDim(1) * attrs->GetAttrValueInt(3); // groups * per group out channels
    int64_t inH = x_shape->GetDim(2);
    int64_t inW = x_shape->GetDim(3);
    int64_t kernelH = w_shape->GetDim(2);
    int64_t kernelW = w_shape->GetDim(3);

    int64_t H_out = (inH - 1) * stride[0] - 2 * padding[0] + dilation[0] * (kernelH - 1) + outputPadding[0] + 1;
    int64_t W_out = (inW - 1) * stride[1] - 2 * padding[1] + dilation[1] * (kernelW - 1) + outputPadding[1] + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, H_out);
    y_shape->SetDim(3, W_out);
    return GRAPH_SUCCESS;
}
