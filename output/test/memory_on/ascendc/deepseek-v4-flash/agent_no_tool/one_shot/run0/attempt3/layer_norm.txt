#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: need storage for x, gamma, beta, meanFull, varFull, y.
// Use 6*tile floats = 24*tile bytes.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps_ptr = attrs->GetAttrPointer<float>(0);
    const std::vector<int64_t> *norm_shape = attrs->GetAttrPointer<std::vector<int64_t>>(1);
    uint32_t normalizedSize = 1;
    for (auto d : *norm_shape) normalizedSize *= static_cast<uint32_t>(d);

    // tileLen: each tile processes a contiguous block of rows (1D slice)
    uint32_t maxTile = UB_BUDGET_BYTES / (6u * sizeof(float));  // 6 buffers
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    // tile length must be multiple of normalizedSize for correctness
    uint32_t rem = maxTile % normalizedSize;
    if (rem != 0) maxTile = maxTile - rem;
    if (maxTile < normalizedSize) maxTile = normalizedSize;

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, normalizedSize);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // blockDim: based on number of tiles, keep moderate
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_normalizedSize(normalizedSize);
    tiling.set_eps(*eps_ptr);
    tiling.set_invNormSize(1.0f / static_cast<float>(normalizedSize));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t normalizedSize, float eps, float invNormSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        normSize_ = normalizedSize;
        eps_ = eps;
        invNormSize_ = invNormSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, normSize_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, normSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueGamma_, BUFFER_NUM, normSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueBeta_, BUFFER_NUM, normSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(meanFull_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(varFull_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarBuf_, 1, sizeof(float));   // for reduction results
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over normalized groups
        uint32_t groupsTotal = totalLen_ / normSize_;
        uint32_t groupsPerBlock = groupsTotal / blockNum;
        uint32_t extra = groupsTotal % blockNum;
        uint32_t start = blockIdx * groupsPerBlock + (blockIdx < extra ? blockIdx : extra);
        uint32_t end = start + groupsPerBlock + (blockIdx < extra ? 1 : 0);
        if (start >= groupsTotal) return;

        // Load gamma and beta once
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma_.AllocTensor<float>();
        AscendC::LocalTensor<float> betaLocal  = inQueueBeta_.AllocTensor<float>();
        AscendC::DataCopy(gammaLocal, gammaGm_, normSize_);
        AscendC::DataCopy(betaLocal, betaGm_, normSize_);
        inQueueGamma_.EnQue(gammaLocal);
        inQueueBeta_.EnQue(betaLocal);

        for (uint32_t g = start; g < end; g++) {
            uint32_t base = g * normSize_;
            // Each tile exactly one group (normSize_ elements)
            CopyIn(base, normSize_);
            Compute(gammaLocal, betaLocal, normSize_);
            CopyOut(base, normSize_);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(AscendC::LocalTensor<float> &gammaLocal,
                                   AscendC::LocalTensor<float> &betaLocal,
                                   uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> meanBuf = meanFull_.AllocTensor<float>();
        AscendC::LocalTensor<float> varBuf  = varFull_.AllocTensor<float>();
        AscendC::LocalTensor<float> scalar  = scalarBuf_.AllocTensor<float>();

        // Compute mean = sum(x) / len
        int32_t axes[1] = {0}; // reduce along first axis (entire 1D tensor)
        AscendC::ReduceSum(scalar, xLocal, 1, axes, false);
        AscendC::Muls(meanBuf, scalar, invNormSize_, 1);
        // Broadcast mean to full vector
        AscendC::Broadcast(meanBuf, meanBuf, 0, len);

        // Compute variance = sum((x - mean)^2) / len
        AscendC::Sub(yLocal, xLocal, meanBuf, len);
        AscendC::Mul(yLocal, yLocal, yLocal, len);
        AscendC::ReduceSum(scalar, yLocal, 1, axes, false);
        AscendC::Muls(varBuf, scalar, invNormSize_, 1);
        // Broadcast var to full vector
        AscendC::Broadcast(varBuf, varBuf, 0, len);

        // Normalize: y = (x - mean) / sqrt(var + eps)
        AscendC::Adds(varBuf, varBuf, eps_, len);
        AscendC::Sqrt(varBuf, varBuf, len);
        // Recompute (x - mean) using meanBuf and xLocal
        AscendC::Sub(yLocal, xLocal, meanBuf, len);
        AscendC::Div(yLocal, yLocal, varBuf, len);

        // Scale and shift: y = y * gamma + beta
        AscendC::Mul(yLocal, yLocal, gammaLocal, len);
        AscendC::Add(yLocal, yLocal, betaLocal, len);

        meanFull_.FreeTensor(meanBuf);
        varFull_.FreeTensor(varBuf);
        scalarBuf_.FreeTensor(scalar);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueGamma_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBeta_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> meanFull_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> varFull_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t normSize_ {1};
    float eps_ {1e-5f};
    float invNormSize_ {1.0f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                                        GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.normalizedSize, tiling_data.eps, tiling_data.invNormSize);
    op.Process();
}
