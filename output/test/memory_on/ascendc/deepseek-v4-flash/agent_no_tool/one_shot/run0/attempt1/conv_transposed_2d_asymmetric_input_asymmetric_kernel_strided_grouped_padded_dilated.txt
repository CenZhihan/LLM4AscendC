// UB model: we store weight (cIn * cOutPerGroup * kH * kW floats) plus a small output tile.

static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{

    uint32_t strideW = stride[1];

    uint32_t paddingW = padding[1];

    uint32_t dilationW = dilation[1];

    uint32_t outH = (hIn - 1) * strideH - 2 * paddingH + dilationH * (kH - 1) + output_padding[0] + 1;

    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {

        tileLen = (aligned > 0) ? aligned : totalLength;

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);

    auto wIn = x_shape->GetDim(3);

    auto kW  = w_shape->GetDim(3);

    int64_t outH = (hIn - 1) * stride[0] - 2 * padding[0] + dilation[0] * (kH - 1) + output_padding[0] + 1;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t n, uint32_t cIn, uint32_t hIn, uint32_t wIn,
                                uint32_t cOutPerGroup, uint32_t kH, uint32_t kW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t paddingH, uint32_t paddingW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups, uint32_t outH, uint32_t outW)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        n_ = n;
        cIn_ = cIn;
        hIn_ = hIn;
        wIn_ = wIn;
        cOutPerGroup_ = cOutPerGroup;
        kH_ = kH;
        kW_ = kW;
        strideH_ = strideH;
        strideW_ = strideW;
        paddingH_ = paddingH;
        paddingW_ = paddingW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;
        outH_ = outH;
        outW_ = outW;
        cOut_ = groups * cOutPerGroup;
        cInPerGroup_ = cIn / groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n * cIn * hIn * wIn);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, cIn * cOutPerGroup * kH * kW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Allocate local memory for weight (once)
        uint32_t weightSize = cIn * cOutPerGroup * kH * kW;
        pipe_.InitBuffer(weightLocal_, 1, weightSize * sizeof(float));
        AscendC::LocalTensor<float> wLocal = weightLocal_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, weightGm_, weightSize);
        weightLocal_.EnQue(wLocal);

        // Output tile buffer (for later use if we tile output)
        // For simplicity we compute directly to global with grid-stride, no double buffering.
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output elements
        uint32_t start = blockIdx * tileLen_;
        if (start >= totalLen_) return;
        uint32_t end = start + tileLen_;
        if (end > totalLen_) end = totalLen_;

        // Get weight local tensor
        AscendC::LocalTensor<float> wLocal = weightLocal_.DeQue<float>();

        // Compute output chunk
        for (uint32_t idx = start; idx < end; ++idx) {
            Compute(idx, wLocal);
        }

        weightLocal_.EnQue(wLocal);  // put back for next Process call (not used again)
    }

private:
    __aicore__ inline void Compute(uint32_t linearIdx, AscendC::LocalTensor<float>& wLocal)
    {
        // Decompose linear index into (n, c, h, w)
        uint32_t n = linearIdx / (cOut_ * outH_ * outW_);
        uint32_t rem = linearIdx % (cOut_ * outH_ * outW_);
        uint32_t c = rem / (outH_ * outW_);
        uint32_t rem2 = rem % (outH_ * outW_);
        uint32_t oh = rem2 / outW_;
        uint32_t ow = rem2 % outW_;

        // Group and channel calculations
        uint32_t group = c / cOutPerGroup_;
        uint32_t cOutLocal = c % cOutPerGroup_;
        uint32_t cInStart = group * cInPerGroup_;
        uint32_t cInEnd = cInStart + cInPerGroup_;

        float sum = 0.0f;

        for (uint32_t kh = 0; kh < kH_; ++kh) {
            int32_t ih_numer = static_cast<int32_t>(oh) + static_cast<int32_t>(paddingH_) -
                               static_cast<int32_t>(dilationH_ * kh);
            // Check divisibility
            if (ih_numer % strideH_ != 0) continue;
            int32_t ih = ih_numer / strideH_;
            if (ih < 0 || ih >= static_cast<int32_t>(hIn_)) continue;

            for (uint32_t kw = 0; kw < kW_; ++kw) {
                int32_t iw_numer = static_cast<int32_t>(ow) + static_cast<int32_t>(paddingW_) -
                                   static_cast<int32_t>(dilationW_ * kw);
                if (iw_numer % strideW_ != 0) continue;
                int32_t iw = iw_numer / strideW_;
                if (iw < 0 || iw >= static_cast<int32_t>(wIn_)) continue;

                // Loop over input channels in this group
                for (uint32_t ci = cInStart; ci < cInEnd; ++ci) {
                    // Input value
                    uint32_t xIdx = n * cIn_ * hIn_ * wIn_ + ci * hIn_ * wIn_ + ih * wIn_ + iw;
                    float xVal = xGm_.GetValue(xIdx);

                    // Weight value: weight[ci][cOutLocal][kh][kw]
                    uint32_t wIdx = ci * cOutPerGroup_ * kH_ * kW_ + cOutLocal * kH_ * kW_ + kh * kW_ + kw;
                    float wVal = wLocal.GetValue(wIdx);

                    sum += xVal * wVal;
                }
            }
        }

        yGm_.SetValue(linearIdx, sum);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> localQueue_;  // not used in this simple version
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t n_ {0};
    uint32_t cIn_ {0};
    uint32_t hIn_ {0};
    uint32_t wIn_ {0};
    uint32_t cOutPerGroup_ {0};
    uint32_t kH_ {0};
    uint32_t kW_ {0};
    uint32_t strideH_ {2};
    uint32_t strideW_ {3};
    uint32_t paddingH_ {1};
    uint32_t paddingW_ {2};
    uint32_t dilationH_ {2};
    uint32_t dilationW_ {1};
    uint32_t groups_ {4};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
    uint32_t cOut_ {0};
    uint32_t cInPerGroup_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom op;
    op.Init(x, weight, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.cIn, tiling_data.hIn, tiling_data.wIn,
            tiling_data.cOutPerGroup, tiling_data.kH, tiling_data.kW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.paddingH, tiling_data.paddingW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups, tiling_data.outH, tiling_data.outW);
    op.Process();
}
