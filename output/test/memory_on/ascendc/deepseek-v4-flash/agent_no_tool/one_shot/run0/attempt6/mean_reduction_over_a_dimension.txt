host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t MIN_TILE_POST = 256;
static constexpr uint32_t MAX_TILE_POST = 24576;  // 192K / 8

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    // Correct API: GetInputShape returns const gert::StorageShape*
    const gert::StorageShape *inShape = context->GetInputShape(0);
    const gert::StorageShape *outShape = context->GetOutputShape(0);
    uint32_t totalIn = static_cast<uint32_t>(inShape->GetOriginShape().GetShapeSize());
    uint32_t totalOut = static_cast<uint32_t>(outShape->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t ndim = inShape->GetDimNum();
    if (dim < 0) dim += ndim;

    uint32_t preSize = 1;
    uint32_t dimSize = 1;
    uint32_t postSize = 1;
    for (int32_t i = 0; i < ndim; ++i) {
        uint32_t s = static_cast<uint32_t>(inShape->GetDim(i));
        if (i < dim) preSize *= s;
        else if (i == dim) dimSize = s;
        else postSize *= s;
    }

    // UB buffer: one input tile + one accumulator (each tilePost floats)
    uint32_t maxFloats = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxFloats = AlignDown(maxFloats, 128);
    if (maxFloats < MIN_TILE_POST) maxFloats = MIN_TILE_POST;
    if (maxFloats > MAX_TILE_POST) maxFloats = MAX_TILE_POST;

    uint32_t tilePost = maxFloats;
    if (tilePost > postSize) tilePost = postSize;
    if (tilePost == 0) tilePost = 1;

    // Block count: at most preSize blocks
    uint32_t blockDim = CeilDiv(preSize, 4u);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    float invDimSize = 1.0f / static_cast<float>(dimSize);

    tiling.set_totalLength(totalIn);
    tiling.set_outputLength(totalOut);
    tiling.set_preSize(preSize);
    tiling.set_dimSize(dimSize);
    tiling.set_postSize(postSize);
    tiling.set_tilePost(tilePost);
    tiling.set_blockDim(blockDim);
    tiling.set_invDimSize(invDimSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    int32_t ndim = x_shape->GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(ndim - 1);
    int32_t outIdx = 0;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i != dim) {
            y_shape->SetDim(outIdx, x_shape->GetDim(i));
            ++outIdx;
        }
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MeanReductionOverADimensionCustom : public OpDef {
public:
    explicit MeanReductionOverADimensionCustom(const char* name) : OpDef(name)
    {
        this->Input("x")

            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReduction {
public:
    __aicore__ inline KernelMeanReduction() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outputLength,
                                uint32_t preSize, uint32_t dimSize,
                                uint32_t postSize, uint32_t tilePost,
                                float invDimSize)
    {
        totalLen_ = totalLength;
        outputLen_ = outputLength;
        preSize_ = preSize;
        dimSize_ = dimSize;
        postSize_ = postSize;
        tilePost_ = tilePost;
        invDimSize_ = invDimSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputLen_);

        pipe_.InitBuffer(inQueueX_, 1, tilePost_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1, tilePost_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || dimSize_ == 0 || postSize_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Each block processes a contiguous range of outer indices (preSize dimension)
        uint32_t outerPerBlock = (preSize_ + blockNum - 1) / blockNum;
        uint32_t outerStart = blockIdx * outerPerBlock;
        uint32_t outerEnd = outerStart + outerPerBlock;
        if (outerStart >= preSize_) return;
        if (outerEnd > preSize_) outerEnd = preSize_;

        uint32_t dimSize = dimSize_;
        uint32_t postSize = postSize_;
        uint32_t tilePost = tilePost_;
        float invDim = invDimSize_;

        for (uint32_t i = outerStart; i < outerEnd; ++i) {
            for (uint32_t t = 0; t < postSize; t += tilePost) {
                uint32_t curTile = postSize - t;
                if (curTile > tilePost) curTile = tilePost;

                AscendC::LocalTensor<float> accLocal = accQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();

                // Initialize accumulator to zero
                AscendC::SetZero(accLocal, curTile);

                uint32_t baseIdx = i * dimSize * postSize; // start of this outer index

                for (uint32_t j = 0; j < dimSize; ++j) {
                    uint32_t srcOffset = baseIdx + j * postSize + t;
                    AscendC::DataCopy(xLocal, xGm_[srcOffset], curTile);
                    AscendC::Add(accLocal, accLocal, xLocal, curTile);
                }

                // Multiply by 1/dimSize to get mean
                AscendC::Muls(accLocal, accLocal, invDim, curTile);

                // Store to output
                uint32_t dstOffset = i * postSize + t;
                AscendC::DataCopy(yGm_[dstOffset], accLocal, curTile);

                inQueueX_.FreeTensor(xLocal);
                accQueue_.FreeTensor(accLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outputLen_ {0};
    uint32_t preSize_ {0};
    uint32_t dimSize_ {0};
    uint32_t postSize_ {0};
    uint32_t tilePost_ {0};
    float invDimSize_ {0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReduction op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.outputLength,
            tiling_data.preSize, tiling_data.dimSize,
            tiling_data.postSize, tiling_data.tilePost,
            tiling_data.invDimSize);
    op.Process();
}
