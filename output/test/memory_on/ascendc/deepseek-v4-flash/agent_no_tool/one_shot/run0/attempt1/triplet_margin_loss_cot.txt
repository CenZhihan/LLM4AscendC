class KernelTripletMarginLossCustom {
public:
    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y, uint32_t totalPairs, uint32_t tileLen, float margin, uint32_t d) {
        totalPairs_ = totalPairs;
        tileLen_ = tileLen;
        margin_ = margin;
        d_ = d;
        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);
        // Init pipes: each input queue with 2 buffers of d floats
        pipe_.InitBuffer(inAnchorQ_, BUFFER_NUM, d_ * sizeof(float));
        pipe_.InitBuffer(inPositiveQ_, BUFFER_NUM, d_ * sizeof(float));
        pipe_.InitBuffer(inNegativeQ_, BUFFER_NUM, d_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, d_ * sizeof(float)); // for squares, diff
        pipe_.InitBuffer(tmpSumQ_, 1, sizeof(float));   // but we can use scalar
    }
    __aicore__ inline void Process() {
        if (totalPairs_ == 0) return;
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        // Only block 0 does work if blockDim=1. Use grid-stride.
        // For simplicity, we process all pairs in block0.
        if (blockIdx != 0) return;
        float sum_loss = 0.0f;
        uint32_t start = 0;
        while (start < totalPairs_) {
            uint32_t batch = min(tileLen_, totalPairs_ - start);
            // Prefetch first pair
            LoadPair(start, d_);
            if (batch > 1) LoadPair(start+1, d_);
            for (uint32_t i = 0; i < batch; i++) {
                // Compute on current pair (from queue)
                float loss = ComputePair(d_);
                sum_loss += loss;
                // Free current pair's buffers
                FreeBuffers();
                // Prefetch next pair
                uint32_t next = start + i + 2;
                if (next < start + batch) {
                    LoadPair(next, d_);
                }
            }
            start += batch;
        }
        float mean = sum_loss / totalPairs_;
        // Write mean to output
        AscendC::LocalTensor<float> outLocal = outYQ_.AllocTensor<float>();
        outLocal.SetValue(0, mean);
        AscendC::DataCopy(yGm_, outLocal, 1);
        outYQ_.FreeTensor(outLocal);
    }
private:
    // ... implementation of LoadPair, ComputePair, FreeBuffers
};
