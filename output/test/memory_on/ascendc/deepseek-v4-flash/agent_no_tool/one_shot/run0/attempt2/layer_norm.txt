#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"
#include <numeric>
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t MIN_TILE_ELEMS = 4096;
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const std::vector<int64_t> *normShapePtr = attrs->GetAttrPointer<std::vector<int64_t>>(0);
    const float *epsilon = attrs->GetAttrPointer<float>(1);

    uint32_t rowSize = 1;
    if (normShapePtr && !normShapePtr->empty()) {
        for (auto s : *normShapePtr) {
            rowSize *= static_cast<uint32_t>(s);
        }
    } else {
        return ge::GRAPH_FAILED;
    }

    uint32_t totalElements = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    uint32_t totalRows = totalElements / rowSize;
    if (totalRows == 0 || rowSize == 0) return ge::GRAPH_FAILED;

    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < MIN_TILE_ELEMS) maxTile = MIN_TILE_ELEMS;

    uint32_t tileLen = (rowSize < maxTile) ? AlignDown(rowSize, ALIGN_ELEMS) : maxTile;
    if (tileLen < MIN_TILE_ELEMS) tileLen = rowSize;

    uint32_t blockDim = totalRows;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalElements(totalElements);
    tiling.set_rowSize(rowSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_epsilon(*epsilon);
    tiling.set_hasWeight(context->GetInputShape(1) != nullptr);
    tiling.set_hasBias(context->GetInputShape(2) != nullptr);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalElements, uint32_t rowSize, uint32_t tileLength,
                                float epsilon, bool hasWeight, bool hasBias)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        if (hasWeight) weightGm_.SetGlobalBuffer((__gm__ float*)weight, rowSize);
        if (hasBias)   biasGm_.SetGlobalBuffer((__gm__ float*)bias, rowSize);

        rowSize_ = rowSize;
        tileLen_ = tileLength;
        epsilon_ = epsilon;
        rowF_ = static_cast<float>(rowSize_);   // avoid cast in aicore later
        totalRows_ = totalElements / rowSize_;
        hasWeight_ = hasWeight;
        hasBias_ = hasBias;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t rowsPerBlock = (totalRows_ + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = (blockIdx + 1) * rowsPerBlock;
        if (endRow > totalRows_) endRow = totalRows_;

        for (uint32_t r = startRow; r < endRow; r++) {
            ProcessRow(r);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        uint32_t rowOffset = rowIdx * rowSize_;
        float sum = 0.0f;
        float sumSq = 0.0f;
        uint32_t remaining = rowSize_;
        uint32_t offset = rowOffset;

        // First pass: compute sum and sum of squares
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);

            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

            // Reduce sum
            AscendC::ReduceSum(tmp, xLocal, curLen);
            float tileSum = tmp.GetValue(0);
            sum += tileSum;

            // Reduce sum of squares: tmp = x^2, then reduce sum
            AscendC::Mul(tmp, xLocal, xLocal, curLen);
            AscendC::ReduceSum(tmp, tmp, curLen);
            float tileSumSq = tmp.GetValue(0);
            sumSq += tileSumSq;

            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xLocal);

            remaining -= curLen;
            offset += curLen;
        }

        // Compute mean and variance (use rowF_ to avoid cast in aicore)
        float mean = sum / rowF_;
        float var = sumSq / rowF_ - mean * mean;
        if (var < 0.0f) var = 0.0f;
        float invStd = 1.0f / sqrt(var + epsilon_);

        // Second pass: normalize and apply affine
        remaining = rowSize_;
        offset = rowOffset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // y = (x - mean) * invStd   (use scalar versions)
            AscendC::Subs(yLocal, xLocal, mean, curLen);
            AscendC::Muls(yLocal, yLocal, invStd, curLen);

            // Apply weight and bias if present
            if (hasWeight_ || hasBias_) {
                if (hasWeight_) {
                    AscendC::LocalTensor<float> wChunk = tmpQueue_.AllocTensor<float>();
                    AscendC::DataCopy(wChunk, weightGm_[offset % rowSize_], curLen);
                    AscendC::Mul(yLocal, yLocal, wChunk, curLen);
                    tmpQueue_.FreeTensor(wChunk);
                }
                if (hasBias_) {
                    AscendC::LocalTensor<float> bChunk = tmpQueue_.AllocTensor<float>();
                    AscendC::DataCopy(bChunk, biasGm_[offset % rowSize_], curLen);
                    AscendC::Add(yLocal, yLocal, bChunk, curLen);
                    tmpQueue_.FreeTensor(bChunk);
                }
            }

            // Write output
            AscendC::DataCopy(yGm_[offset], yLocal, curLen);

            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);

            remaining -= curLen;
            offset += curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t rowSize_ {0};
    uint32_t tileLen_ {0};
    uint32_t totalRows_ {0};
    float rowF_ {0.0f};            // float version of rowSize_
    float epsilon_ {1e-5f};
    bool hasWeight_ {false};
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, y, weight, bias,
            tiling_data.totalElements,
            tiling_data.rowSize,
            tiling_data.tileLength,
            tiling_data.epsilon,
            tiling_data.hasWeight,
            tiling_data.hasBias);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape, eps=1e-5, elementwise_affine=True):
        super(ModelNew, self).__init__()
        self.normalized_shape = normalized_shape
        self.eps = eps
        self.elementwise_affine = elementwise_affine
        if elementwise_affine:
            self.weight = torch.nn.Parameter(torch.ones(normalized_shape))
            self.bias = torch.nn.Parameter(torch.zeros(normalized_shape))
        else:
            self.register_parameter('weight', None)
            self.register_parameter('bias', None)

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        norm_shape = list(self.normalized_shape)
        weight_ = self.weight.view(-1) if self.weight is not None else None
        bias_ = self.bias.view(-1) if self.bias is not None else None
        return fn(x, weight_, bias_, norm_shape, self.eps)
'''
