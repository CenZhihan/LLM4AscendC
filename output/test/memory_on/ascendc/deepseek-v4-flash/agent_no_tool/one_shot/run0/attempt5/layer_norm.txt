host_operator_src = """
#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: store one row of x, y, plus temporary for mean, var, etc. (approx 8 floats per element)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // per-core
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MIN_TILE_ROWS = 1;
static constexpr uint32_t MAX_TILE_ROWS = 256;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LayerNormCustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t totalElements = xShape->GetOriginShape().GetShapeSize();

    // Get normalized_shape from attributes
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *normalizedShape = attrs->GetAttrPointer<int64_t>(0);
    uint32_t numDims = attrs->GetAttrNum(0);

    // Compute rowSize = product of last numDims dimensions
    uint32_t rowSize = 1;
    uint32_t shapeDims = xShape->GetOriginShape().GetDimNum();
    for (uint32_t i = shapeDims - numDims; i < shapeDims; ++i) {
        rowSize *= xShape->GetOriginShape().GetDim(i);
    }

    uint32_t totalRows = (rowSize > 0) ? totalElements / rowSize : 0;

    // Tiling over rows: compute tileRows based on UB budget and rowSize
    // Needed local floats per tile: rowSize (x) + rowSize (y) + rowSize (mean+var temporary) ~ 3*rowSize
    uint32_t floatsPerTile = 3 * rowSize;
    uint32_t maxTileRows = UB_BUDGET_BYTES / (floatsPerTile * sizeof(float));
    if (maxTileRows > MAX_TILE_ROWS) maxTileRows = MAX_TILE_ROWS;
    if (maxTileRows < MIN_TILE_ROWS) maxTileRows = MIN_TILE_ROWS;

    // Align tileRows to ALIGN_ELEMS? Not needed for rows count. Just ensure at least 1.
    uint32_t tileRows = maxTileRows;
    if (totalRows < tileRows) {
        tileRows = totalRows;
    }

    // Block dim: use number of tiles
    uint32_t numTiles = (tileRows == 0) ? 1 : CeilDiv(totalRows, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    // Get epsilon from attributes (second attr)
    const float *epsilon = attrs->GetAttrPointer<float>(1);

    tiling.set_totalElements(totalElements);
    tiling.set_rowSize(rowSize);
    tiling.set_tileRows(tileRows);
    tiling.set_epsilon(*epsilon);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LayerNormCustom : public OpDef {
public:
    explicit LayerNormCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("weight")
            .ParamType(OPTIONAL)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("bias")
            .ParamType(OPTIONAL)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("normalized_shape").AttrType(OPTIONAL).ListInt({});
        this->Attr("epsilon").AttrType(OPTIONAL).Float(1e-5);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LayerNormCustom);
} // namespace ops

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalElements, uint32_t rowSize,
                                uint32_t tileRows, float epsilon) {
        totalElements_ = totalElements;
        rowSize_ = rowSize;
        tileRows_ = tileRows;
        epsilon_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElements_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElements_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, rowSize_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, rowSize_);

        // Local buffers: 2 rows of x, 2 rows of y, temporary for mean/var (one row)
        uint32_t tileElems = tileRows_ * rowSize_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileElems * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileElems * sizeof(float));  // reuse for mean/var
    }

    __aicore__ inline void Process() {
        if (totalElements_ == 0 || rowSize_ == 0 || tileRows_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Grid-stride over rows
        uint32_t rowsPerBlock = tileRows_;
        uint32_t totalRows = totalElements_ / rowSize_;
        uint32_t startRow = blockIdx * rowsPerBlock;
        if (startRow >= totalRows) return;

        while (startRow < totalRows) {
            uint32_t curRows = (startRow + rowsPerBlock <= totalRows) ? rowsPerBlock : (totalRows - startRow);
            uint32_t curElems = curRows * rowSize_;

            CopyIn(startRow, curRows, curElems);
            Compute(startRow, curRows, curElems);
            CopyOut(startRow, curRows, curElems);

            startRow += blockNum * rowsPerBlock;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t startRow, uint32_t rows, uint32_t elems) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[startRow * rowSize_], elems);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t startRow, uint32_t rows, uint32_t elems) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

        // Process each row independently
        for (uint32_t r = 0; r < rows; r++) {
            uint32_t rowOffset = r * rowSize_;
            // Compute mean
            float sum = 0.0f;
            for (uint32_t i = 0; i < rowSize_; i++) {
                sum += xLocal[rowOffset + i];
            }
            float mean = sum / rowSize_;
            // Compute variance (mean of squared differences)
            float var = 0.0f;
            for (uint32_t i = 0; i < rowSize_; i++) {
                float diff = xLocal[rowOffset + i] - mean;
                var += diff * diff;
            }
            var = var / rowSize_ + epsilon_;
            float invStd = 1.0f / sqrt(var);
            // Normalize and apply weight/bias
            for (uint32_t i = 0; i < rowSize_; i++) {
                float val = (xLocal[rowOffset + i] - mean) * invStd;
                if (weightGm_.GetValue(rowOffset + i) != 0) { // placeholder; actual weight is from GM per row? weight is per-dim, so index i
                    val = val * weightGm_.GetValue(i) + biasGm_.GetValue(i);
                }
                yLocal[rowOffset + i] = val;
            }
        }

        tmpQueue_.FreeTensor(tmp);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t startRow, uint32_t rows, uint32_t elems) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[startRow * rowSize_], yLocal, elems);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalElements_ {0};
    uint32_t rowSize_ {0};
    uint32_t tileRows_ {0};
    float epsilon_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, y, weight, bias, tiling_data.totalElements, tiling_data.rowSize, tiling_data.tileRows, tiling_data.epsilon);
    op.Process();
}
"""
