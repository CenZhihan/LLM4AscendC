kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dCustom {
public:
    __aicore__ inline KernelConvTransposed2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C_in, uint32_t H_in, uint32_t W_in,
                                uint32_t C_out, uint32_t H_out, uint32_t W_out,
                                uint32_t K_H, uint32_t K_W,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t dilation_h, uint32_t dilation_w,
                                uint32_t groups)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N;
        C_in_ = C_in;
        H_in_ = H_in;
        W_in_ = W_in;
        C_out_ = C_out;
        H_out_ = H_out;
        W_out_ = W_out;
        K_H_ = K_H;
        K_W_ = K_W;
        stride_h_ = stride_h;
        stride_w_ = stride_w;
        pad_h_ = pad_h;
        pad_w_ = pad_w;
        dilation_h_ = dilation_h;
        dilation_w_ = dilation_w;
        groups_ = groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_in_ * H_in_ * W_in_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, C_in_ * (C_out_ / groups_) * K_H_ * K_W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inBuf_, 1, sizeof(float));
        pipe_.InitBuffer(accBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t base = blockIdx * tileLen_;
        uint32_t end = base + tileLen_;
        if (base >= totalLen_) return;
        if (end > totalLen_) end = totalLen_;

        for (uint32_t idx = base; idx < end; idx++) {
            uint32_t remainder = idx;
            uint32_t w_out = remainder % W_out_;
            remainder /= W_out_;
            uint32_t h_out = remainder % H_out_;
            remainder /= H_out_;
            uint32_t c_out = remainder % C_out_;
            uint32_t n = remainder / C_out_;

            uint32_t g = c_out / (C_out_ / groups_);
            uint32_t c_out_g = c_out % (C_out_ / groups_);
            uint32_t c_in_start = g * (C_in_ / groups_);

            float sum = 0.0f;
            for (uint32_t kh = 0; kh < K_H_; kh++) {
                for (uint32_t kw = 0; kw < K_W_; kw++) {
                    int32_t h_in = static_cast<int32_t>(h_out) + static_cast<int32_t>(pad_h_)
                                 - static_cast<int32_t>(kh * dilation_h_);
                    int32_t w_in = static_cast<int32_t>(w_out) + static_cast<int32_t>(pad_w_)
                                 - static_cast<int32_t>(kw * dilation_w_);

                    if (h_in < 0 || static_cast<uint32_t>(h_in) >= H_in_) continue;
                    if (w_in < 0 || static_cast<uint32_t>(w_in) >= W_in_) continue;
                    if (h_in % static_cast<int32_t>(stride_h_) != 0) continue;
                    if (w_in % static_cast<int32_t>(stride_w_) != 0) continue;

                    uint32_t h_in_valid = static_cast<uint32_t>(h_in) / stride_h_;
                    uint32_t w_in_valid = static_cast<uint32_t>(w_in) / stride_w_;

                    for (uint32_t ic = 0; ic < C_in_ / groups_; ic++) {
                        uint32_t global_c_in = c_in_start + ic;
                        float x_val = xGm_[n * C_in_ * H_in_ * W_in_ 
                                          + global_c_in * H_in_ * W_in_ 
                                          + h_in_valid * W_in_ 
                                          + w_in_valid];
                        float w_val = wGm_[global_c_in * (C_out_ / groups_) * K_H_ * K_W_ 
                                          + c_out_g * K_H_ * K_W_ 
                                          + kh * K_W_ 
                                          + kw];
                        sum += x_val * w_val;
                    }
                }
            }

            yGm_[idx] = sum;
        }
    }
    
private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> accBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_, C_in_, H_in_, W_in_, C_out_, H_out_, W_out_;
    uint32_t K_H_, K_W_;
    uint32_t stride_h_, stride_w_, pad_h_, pad_w_, dilation_h_, dilation_w_, groups_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dCustom op;
    op.Init(x, weight, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C_in, tiling_data.H_in, tiling_data.W_in,
            tiling_data.C_out, tiling_data.H_out, tiling_data.W_out,
            tiling_data.K_H, tiling_data.K_W,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.dilation_h, tiling_data.dilation_w,
            tiling_data.groups);
    op.Process();
}
