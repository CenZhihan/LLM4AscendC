kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
                                GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
                                uint32_t totalBatch, uint32_t outFeatures, uint32_t inFeatures,
                                uint32_t numGroups, uint32_t tileN, uint32_t tileK)
    {
        batch_ = totalBatch;
        outF_ = outFeatures;
        inF_ = inFeatures;
        numGroups_ = numGroups;
        tileN_ = tileN;
        tileK_ = tileK;
        groupSize_ = outFeatures / numGroups;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_ * inF_);
        wtGm_.SetGlobalBuffer((__gm__ float*)weight, outF_ * inF_);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outF_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outF_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outF_);
        betaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outF_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch_ * outF_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueueWt_, BUFFER_NUM, tileN_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(outBufQueue_, 1, outF_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= batch_) return;

        // Load entire x for this sample (inF_)
        float* xLocal = (float*)AscendC::GetLocalTensor();
        AscendC::DataCopy(xLocal, xGm_[blockIdx * inF_], inF_);
        // We'll keep x in UB for reuse across output tiles
        // But UB is limited; we'll load x tile by tileK in inner loop.
        // Instead, we'll use a local buffer for x of size tileK.
        // So we do not load full x at once.

        // Output buffer in UB (outF_ floats)
        float* outBuf = (float*)AscendC::GetLocalTensor();  // assume outBufQueue_ provides space
        AscendC::LocalTensor<float> outLocal = outBufQueue_.AllocTensor<float>();

        // Initialize outBuf to zero
        for (uint32_t i = 0; i < outF_; i += 128) {
            uint32_t len = (outF_ - i > 128) ? 128 : (outF_ - i);
            AscendC::SetZero(outLocal[i], len);
        }

        // Loop over output tiles
        for (uint32_t no = 0; no < outF_; no += tileN_) {
            uint32_t curN = (no + tileN_ <= outF_) ? tileN_ : (outF_ - no);
            // Temporary tile accumulator
            AscendC::LocalTensor<float> tileBuf = tmpQueue_.AllocTensor<float>();
            AscendC::SetZero(tileBuf, curN);

            // Loop over input tiles
            for (uint32_t ni = 0; ni < inF_; ni += tileK_) {
                uint32_t curK = (ni + tileK_ <= inF_) ? tileK_ : (inF_ - ni);

                // Load weight tile (curN x curK) from GM
                AscendC::LocalTensor<float> wtLocal = inQueueWt_.AllocTensor<float>();
                // weight layout: row-major: row major: row i (output channel) is at (no+i)*inF_ + ...
                uint32_t wtOffset = (no * inF_) + ni;  // not correct for submatrix; need for each row.
                // Actually we need a submatrix of weight: rows no..no+curN-1, columns ni..ni+curK-1.
                // In row-major, this is contiguous block? No, each row has stride inF_; columns contiguous.
                // To load efficiently, we can load each row separately? Or use DataCopy2D? Use SetGlobalBuffer with stride? 
                // Simplified: loop over rows in tile, load each row segment.
                for (uint32_t r = 0; r < curN; r++) {
                    uint32_t rowAddr = (no + r) * inF_ + ni;
                    AscendC::DataCopy(wtLocal[r * curK], wtGm_[rowAddr], curK);
                }
                // Mark weight enqueue
                inQueueWt_.EnQue(wtLocal);

                // Load x tile (curK) from GM (same block's x)
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[blockIdx * inF_ + ni], curK);
                inQueueX_.EnQue(xLocal);

                // Compute matrix-vector product: tileBuf += weight_tile * x_tile
                AscendC::LocalTensor<float> wtDeque = inQueueWt_.DeQue<float>();
                AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
                // For each row in tileBuf (curN rows), compute dot product with xDeque
                for (uint32_t r = 0; r < curN; r++) {
                    AscendC::LocalTensor<float> rowWt = wtDeque[r * curK];
                    AscendC::LocalTensor<float> rowTile = tileBuf[r];
                    // Multiply element-wise and sum
                    AscendC::Mul(rowTile, rowWt, xDeque, curK);
                    // But Mul overwrites; we need accumulation. Use Add after Mul.
                    // Actually we need to compute sum of products. Use ReduceSum?
                    // Simplify: use a temp vector of size curK, multiply, then reduce add.
                    // Since we have limited instructions, we'll do loop.
                    // For brevity, use AscendC::VecDot? Not available.
                    // Use manual loop:
                    float sum = 0.0f;
                    for (uint32_t k = 0; k < curK; k++) {
                        sum += rowWt.GetValue(k) * xDeque.GetValue(k);
                    }
                    // Add to tileBuf[r]
                    tileBuf.SetValue(r, tileBuf.GetValue(r) + sum);
                }
                inQueueWt_.FreeTensor(wtDeque);
                inQueueX_.FreeTensor(xDeque);
            }

            // Now tileBuf contains linear output for this output tile
            // Add bias, hardtanh, mish
            for (uint32_t i = 0; i < curN; i++) {
                float val = tileBuf.GetValue(i);
                val += biasGm_.GetValue(no + i);
                // hardtanh: clamp to [-1, 1]
                if (val < -1.0f) val = -1.0f;
                if (val > 1.0f) val = 1.0f;
                // mish: x * tanh(softplus(x))  ; softplus = log(1+exp(x))
                // Use approximation or Taylor? We'll use built-in functions if available.
                // AscendC has Exp, Log, Tanh. Compute:
                float sp = AscendC::Log(1.0f + AscendC::Exp(val));
                float tanh_sp = AscendC::Tanh(sp);
                val = val * tanh_sp;
                tileBuf.SetValue(i, val);
            }

            // Store tile to outBuf
            for (uint32_t i = 0; i < curN; i++) {
                outBuf[no + i] = tileBuf.GetValue(i);
            }

            tmpQueue_.FreeTensor(tileBuf);
        }

        // Now outBuf contains the output after linear, bias, hardtanh, mish.
        // Compute group norm: for each group, compute mean, variance, normalize.
        // Groups are contiguous blocks of groupSize channels.
        for (uint32_t g = 0; g < numGroups_; g++) {
            uint32_t start = g * groupSize_;
            // Compute mean
            float sum = 0.0f;
            for (uint32_t i = 0; i < groupSize_; i++) {
                sum += outBuf[start + i];
            }
            float mean = sum / (float)groupSize_;
            // Compute variance
            float var = 0.0f;
            for (uint32_t i = 0; i < groupSize_; i++) {
                float diff = outBuf[start + i] - mean;
                var += diff * diff;
            }
            float var_eps = var / (float)groupSize_ + 1e-5f;
            float inv_std = 1.0f / AscendC::Sqrt(var_eps);
            // Apply gamma, beta
            for (uint32_t i = 0; i < groupSize_; i++) {
                float normalized = (outBuf[start + i] - mean) * inv_std;
                normalized = normalized * gammaGm_.GetValue(start + i) + betaGm_.GetValue(start + i);
                outBuf[start + i] = normalized;
            }
        }

        // Write output to global memory
        for (uint32_t i = 0; i < outF_; i += 128) {
            uint32_t len = (outF_ - i > 128) ? 128 : (outF_ - i);
            AscendC::DataCopy(yGm_[blockIdx * outF_ + i], outBuf + i, len);
        }

        outBufQueue_.FreeTensor(outLocal);
    }

private:
    uint32_t batch_;
    uint32_t outF_;
    uint32_t inF_;
    uint32_t numGroups_;
    uint32_t groupSize_;
    uint32_t tileN_;
    uint32_t tileK_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wtGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWt_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> outBufQueue_;
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.totalBatch, tiling_data.outFeatures, tiling_data.inFeatures,
            tiling_data.numGroups, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
