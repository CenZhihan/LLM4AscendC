#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: 192 KB = 49152 floats.
// Double buffer for A tile: 2 * tileM * K
// Double buffer for B tile: 2 * K * tileN
// Accumulator for C tile: tileM * tileN
// Total ~ 2*tileM*K + 2*K*tileN + tileM*tileN + margin
// Choose fixed tile sizes that fit with K <= 32 typical
static constexpr uint32_t MAX_TILE_SIZE = 128;  // safe for K <= 32
static constexpr uint32_t ALIGN = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    auto a_shape = context->GetInputShape(0)->GetOriginShape();
    auto b_shape = context->GetInputShape(1)->GetOriginShape();

    uint32_t M = static_cast<uint32_t>(a_shape.GetDim(0));
    uint32_t K = static_cast<uint32_t>(a_shape.GetDim(1));
    uint32_t N = static_cast<uint32_t>(b_shape.GetDim(1));

    // Choose tileM and tileN: min(MAX_TILE_SIZE, dimension), aligned to 32
    uint32_t tileM = (M < MAX_TILE_SIZE) ? AlignDown(M, ALIGN) : AlignDown(MAX_TILE_SIZE, ALIGN);
    if (tileM < ALIGN) tileM = ALIGN;
    if (tileM > M) tileM = M;

    uint32_t tileN = (N < MAX_TILE_SIZE) ? AlignDown(N, ALIGN) : AlignDown(MAX_TILE_SIZE, ALIGN);
    if (tileN < ALIGN) tileN = ALIGN;
    if (tileN > N) tileN = N;

    // Compute blockDim based on number of row tiles, clamped to [8,24]
    const uint32_t numRowTiles = CeilDiv(M, tileM);
    uint32_t blockDim = numRowTiles * CeilDiv(N, tileN);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, a_shape->GetDim(0));
    c_shape->SetDim(1, b_shape->GetDim(1));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, M_ * N_);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM_ * K_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, K_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, 1, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0 || tileM_ == 0 || tileN_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Distribute row and column tiles across blocks
        const uint32_t numRowTiles = (M_ + tileM_ - 1) / tileM_;
        const uint32_t numColTiles = (N_ + tileN_ - 1) / tileN_;
        const uint32_t totalTiles = numRowTiles * numColTiles;

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            uint32_t rowTile = tileId / numColTiles;
            uint32_t colTile = tileId % numColTiles;
            uint32_t mStart = rowTile * tileM_;
            uint32_t nStart = colTile * tileN_;
            uint32_t curM = (mStart + tileM_ <= M_) ? tileM_ : (M_ - mStart);
            uint32_t curN = (nStart + tileN_ <= N_) ? tileN_ : (N_ - nStart);

            // Copy B tile row by row (since columns are not contiguous)
            uint32_t bTileSize = K_ * curN;
            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
            for (uint32_t k = 0; k < K_; ++k) {
                AscendC::DataCopy(bLocal[k * curN], bGm_[k * N_ + nStart], curN);
            }
            inQueueB_.EnQue(bLocal);

            // Copy A tile (contiguous)
            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            uint32_t aTileSize = curM * K_;
            AscendC::DataCopy(aLocal, aGm_[mStart * K_], aTileSize);
            inQueueA_.EnQue(aLocal);

            // Compute
            AscendC::LocalTensor<float> aLoc = inQueueA_.DeQue<float>();
            AscendC::LocalTensor<float> bLoc = inQueueB_.DeQue<float>();
            AscendC::LocalTensor<float> cLoc = outQueueC_.AllocTensor<float>();

            for (uint32_t i = 0; i < curM; ++i) {
                for (uint32_t j = 0; j < curN; ++j) {
                    float sum = 0.0f;
                    for (uint32_t k = 0; k < K_; ++k) {
                        float aVal = aLoc.GetValue(i * K_ + k);
                        float bVal = bLoc.GetValue(k * curN + j);
                        sum += aVal * bVal;
                    }
                    cLoc.SetValue(i * curN + j, sum);
                }
            }

            inQueueA_.FreeTensor(aLoc);
            inQueueB_.FreeTensor(bLoc);
            outQueueC_.EnQue(cLoc);

            // Copy output tile row by row
            AscendC::LocalTensor<float> cOut = outQueueC_.DeQue<float>();
            for (uint32_t i = 0; i < curM; ++i) {
                AscendC::DataCopy(cGm_[(mStart + i) * N_ + nStart], cOut[i * curN], curN);
            }
            outQueueC_.FreeTensor(cOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
