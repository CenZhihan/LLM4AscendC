host_operator_src="""
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 32;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t compute_tileM(uint32_t M, uint32_t K, uint32_t N, uint32_t tileK, uint32_t tileN) {
    uint32_t max_floats = UB_BUDGET_BYTES / 4 / 2;
    uint32_t numerator = max_floats - tileK * tileN;
    uint32_t denominator = tileK + tileN;
    if (denominator == 0) return 1;
    uint32_t tileM = numerator / denominator;
    tileM = AlignDown(tileM, ALIGN);
    if (tileM == 0) tileM = ALIGN;
    if (tileM > M) tileM = AlignDown(M, ALIGN);
    if (tileM == 0) tileM = M;
    return tileM;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    // Use context->GetInputShape(0) directly, no GetOriginShape()
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    uint32_t M = static_cast<uint32_t>(a_shape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(a_shape->GetDim(1));
    uint32_t N = static_cast<uint32_t>(b_shape->GetDim(1));

    uint32_t tileK = 16;
    uint32_t tileN = 16;
    uint32_t tileM = compute_tileM(M, K, N, tileK, tileN);

    uint32_t numTileM = CeilDiv(M, tileM);
    uint32_t blockDim = numTileM;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    outShape->SetDimNum(2);

kernel_src="""
#include "kernel_operator.h"
#include "matmul_intf.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN,
                                uint32_t blockDim) {
        M_ = M;
        K_ = K;
        N_ = N;
        tileM_ = tileM;
        tileK_ = tileK;
        tileN_ = tileN;
        blockDim_ = blockDim;

        aGm_.SetGlobalBuffer((__gm__ float*)a, M * K);
        bGm_.SetGlobalBuffer((__gm__ float*)b, K * N);
        cGm_.SetGlobalBuffer((__gm__ float*)c, M * N);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t rowStart = blockIdx * tileM_;
        if (rowStart >= M_) return;

        uint32_t rowEnd = rowStart + tileM_;
        if (rowEnd > M_) rowEnd = M_;
        uint32_t rowsThisBlock = rowEnd - rowStart;

        for (uint32_t r = 0; r < rowsThisBlock; r += tileM_) {
            uint32_t curRows = (r + tileM_ <= rowsThisBlock) ? tileM_ : (rowsThisBlock - r);
            ProcessRowChunk(rowStart + r, curRows);
        }
    }

private:
    __aicore__ inline void ProcessRowChunk(uint32_t rowOffset, uint32_t rowCount) {
        constexpr uint32_t CUBE_M = 16;
        constexpr uint32_t CUBE_K = 16;
        constexpr uint32_t CUBE_N = 16;

        uint32_t colStart = 0;
        while (colStart < K_) {
            uint32_t curK = (colStart + tileK_ <= K_) ? tileK_ : (K_ - colStart);
            break;  // Placeholder; full matmul logic would iterate over K
        }

        LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
        LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();

        for (uint32_t i = 0; i < rowCount; i++) {
            DataCopy(aLocal[i * K_], aGm_[(rowOffset + i) * K_], K_);
        }
        DataCopy(bLocal, bGm_, K_ * N_);
        for (uint32_t i = 0; i < rowCount; i++) {
            for (uint32_t j = 0; j < N_; j++) {
                float sum = 0.0f;
                for (uint32_t k = 0; k < K_; k++) {
                    sum += aLocal[i * K_ + k] * bLocal[k * N_ + j];
                }
                cLocal[i * N_ + j] = sum;
            }
        }
        for (uint32_t i = 0; i < rowCount; i++) {
            DataCopy(cGm_[(rowOffset + i) * N_], cLocal[i * N_], N_);
        }

        inQueueA_.FreeTensor(aLocal);
        inQueueB_.FreeTensor(bLocal);
        outQueueC_.FreeTensor(cLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueA_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueB_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueC_;

    GlobalTensor<float> aGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> cGm_;

    uint32_t M_;
    uint32_t K_;
    uint32_t N_;
    uint32_t tileM_;
    uint32_t tileK_;
    uint32_t tileN_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN,
            tiling_data.blockDim);
    op.Process();
}
