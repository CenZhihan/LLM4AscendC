kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                uint32_t T, uint32_t B, uint32_t I, uint32_t H,
                                uint32_t O, uint32_t K)
    {
        T_ = T;
        B_ = B;
        I_ = I;
        H_ = H;
        O_ = O;
        K_ = K;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T_ * B_ * I_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B_ * H_);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H_ * K_);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O_ * H_);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, T_ * B_ * O_);

        // Local buffers for one batch time step
        pipe_.InitBuffer(hBuf_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(xBuf_, BUFFER_NUM, I_ * sizeof(float));
        pipe_.InitBuffer(preBuf_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(yBuf_, BUFFER_NUM, O_ * sizeof(float));
        pipe_.InitBuffer(biasBuf_, 1, (H_ > O_ ? H_ : O_) * sizeof(float)); // reuse for either bias
    }

    __aicore__ inline void Process()
    {
        if (T_ == 0 || B_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= B_) return;

        // Initialize h local from h0 for this batch
        AscendC::LocalTensor<float> hLocal = hBuf_.AllocTensor<float>();
        AscendC::DataCopy(hLocal, h0Gm_[blockIdx * H_], H_);
        hBuf_.EnQue(hLocal);

        // Preload b_i2h into local (constant per batch)
        AscendC::LocalTensor<float> b_i2hLocal = biasBuf_.AllocTensor<float>();
        AscendC::DataCopy(b_i2hLocal, b_i2hGm_, H_);
        biasBuf_.EnQue(b_i2hLocal);

        // Preload b_h2o into local (constant per batch)
        AscendC::LocalTensor<float> b_h2oLocal = biasBuf_.AllocTensor<float>();
        AscendC::DataCopy(b_h2oLocal, b_h2oGm_, O_);
        biasBuf_.EnQue(b_h2oLocal);

        // Time loop
        for (uint32_t t = 0; t < T_; ++t) {
            // Dequeue h from previous step (or initial)
            AscendC::LocalTensor<float> hPrev = hBuf_.DeQue<float>();

            // Load x[t] for this batch
            AscendC::LocalTensor<float> xLocal = xBuf_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[t * B_ * I_ + blockIdx * I_], I_);

            // Compute pre = inp @ w_i2h^T + b_i2h
            AscendC::LocalTensor<float> preLocal = preBuf_.AllocTensor<float>();
            // For each hidden unit h, compute dot product
            for (uint32_t h = 0; h < H_; ++h) {
                float dot = 0.0f;
                // dot over I part
                for (uint32_t k = 0; k < I_; ++k) {
                    dot += xLocal(k) * w_i2hGm_(h * K_ + k);
                }
                // dot over H part
                for (uint32_t k = 0; k < H_; ++k) {
                    dot += hPrev(k) * w_i2hGm_(h * K_ + I_ + k);
                }
                dot += b_i2hLocal(h);
                preLocal(h) = dot;
            }

            // tanh activation -> new h (in-place on preLocal)
            AscendC::Tanh(preLocal, preLocal, H_);
            // preLocal now holds new h

            // Compute y_t = new_h @ w_h2o^T + b_h2o
            AscendC::LocalTensor<float> yLocal = yBuf_.AllocTensor<float>();
            for (uint32_t o = 0; o < O_; ++o) {
                float dot = 0.0f;
                for (uint32_t k = 0; k < H_; ++k) {
                    dot += preLocal(k) * w_h2oGm_(o * H_ + k);
                }
                dot += b_h2oLocal(o);
                yLocal(o) = dot;
            }

            // Write y_t to global
            AscendC::DataCopy(yGm_[t * B_ * O_ + blockIdx * O_], yLocal, O_);

            // Enqueue new h for next iteration
            hBuf_.EnQue(preLocal);

            // Free x and y locals
            xBuf_.FreeTensor(xLocal);
            yBuf_.FreeTensor(yLocal);
        }

        // All buffers are freed automatically (pipeline cleanup)
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> hBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> preBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> biasBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> w_i2hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t T_ {0};
    uint32_t B_ {0};
    uint32_t I_ {0};
    uint32_t H_ {0};
    uint32_t O_ {0};
    uint32_t K_ {0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(
    GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I, tiling_data.H,
            tiling_data.O, tiling_data.K);
    op.Process();
}
