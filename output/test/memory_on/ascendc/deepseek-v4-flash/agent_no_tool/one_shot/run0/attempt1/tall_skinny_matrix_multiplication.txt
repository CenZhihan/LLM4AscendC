#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: 192 KB.
// We double-buffer tileA (tileM * K) and tileC (tileM * N), and hold B (K * N) once.
// Total floats = 2 * tileM * K + K * N + 2 * tileM * N.
// Budget in floats = 192 * 1024 / 4 = 49152.
static constexpr uint32_t UB_FLOAT_BUDGET = 49152;
static constexpr uint32_t ALIGN = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape *b_shape = context->GetInputShape(1)->GetOriginShape();

    // We assume ND format, shape = [M, K] for A, [K, N] for B.
    uint32_t M = static_cast<uint32_t>(a_shape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(a_shape->GetDim(1));
    uint32_t N = static_cast<uint32_t>(b_shape->GetDim(1));

    // Compute tileM based on UB budget.
    uint32_t tileM = 1;
    uint32_t needed = 0;
    // Solve: 2*tileM*K + K*N + 2*tileM*N <= UB_FLOAT_BUDGET
    // => tileM * (2*K + 2*N) <= UB_FLOAT_BUDGET - K*N
    uint32_t numerator = UB_FLOAT_BUDGET - K * N;
    uint32_t denominator = 2 * (K + N);
    if (denominator > 0) {
        tileM = numerator / denominator;
    }
    // Ensure minimum tileM is 1 and aligned to ALIGN
    if (tileM < 1) tileM = 1;
    tileM = AlignDown(tileM, ALIGN);
    if (tileM > M) tileM = M;    // Cap to M.

    // BlockDim: adjust to number of tiles or cores. Use 8 as baseline.
    const uint32_t numTiles = CeilDiv(M, tileM);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_K = 128;   // Safe upper bound for inner dimension (small)
constexpr int32_t MAX_N = 128;   // Safe upper bound for N (small)

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileM)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;

        // Set up global tensors with strides.
        // A: [M, K] row-major, strideK = K
        // B: [K, N] row-major, strideN = N
        // C: [M, N] row-major, strideN = N
        aGm_.SetGlobalBuffer((__gm__ float*)A, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, M_ * N_);

        // Allocate local buffers (UB).
        // B is loaded entirely into a single buffer (assuming K*N fits).
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM_ * K_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, K_ * N_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileM_ * N_ * sizeof(float));
        // Temporary buffer for row of A (vec operations)
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * K_ * sizeof(float));  // not used in this simple version
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0 || tileM_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over tiles of M dimension.
        const uint32_t stride = blockNum * tileM_;
        uint32_t mBase = blockIdx * tileM_;

        // Pre-load B into UB (only once).
        {
            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
            AscendC::DataCopy(bLocal, bGm_, K_ * N_);
            inQueueB_.EnQue(bLocal);
        }

        while (mBase < M_) {
            uint32_t curTileM = (M_ - mBase) > tileM_ ? tileM_ : (M_ - mBase);

            // Load tile of A (curTileM x K)
            CopyInA(mBase, curTileM);

            // Compute C = A_tile * B
            Compute(curTileM);

            // Store tile of C
            CopyOutC(mBase, curTileM);

            mBase += stride;
        }
    }

private:
    __aicore__ inline void CopyInA(uint32_t rowOffset, uint32_t rows)
    {
        AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
        // We must copy rows * K elements; use DataCopy with appropriate stride.
        // Since A is row-major contiguous, we can copy a rectangle by multiple copies?
        // For simplicity, copy entire tile as a contiguous block if possible.
        // The tile is rows x K, which is contiguous in memory.
        AscendC::DataCopy(aLocal, aGm_[rowOffset * K_], rows * K_);
        inQueueA_.EnQue(aLocal);
    }

    __aicore__ inline void Compute(uint32_t rows)
    {
        // Deque A and B, allocate C.
        AscendC::LocalTensor<float> aLocal = inQueueA_.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB_.DeQue<float>();
        AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();

        // The tile of A has shape (rows, K) in aLocal (rows*K elements).
        // B has shape (K, N) in bLocal (K*N elements).
        // Compute C = A * B => (rows, N).
        // We'll do a simple triple loop over i (row), j (col), k (inner).
        // But use vector operations: for each i, for each j, compute dot product.
        // Given small N and K, a scalar loop may be acceptable.
        // Alternatively, use Mad over k dimension with broadcast.

        // We'll use local variables for indices.
        for (uint32_t i = 0; i < rows; i++) {
            for (uint32_t j = 0; j < N_; j++) {
                float sum = 0.0f;
                for (uint32_t k = 0; k < K_; k++) {
                    sum += aLocal[i * K_ + k] * bLocal[k * N_ + j];
                }
                cLocal[i * N_ + j] = sum;
            }
        }

        // Re-enqueue B for reuse by subsequent tiles (since it's read-only).
        inQueueB_.EnQue(bLocal);
        outQueueC_.EnQue(cLocal);
        inQueueA_.FreeTensor(aLocal);
    }

    __aicore__ inline void CopyOutC(uint32_t rowOffset, uint32_t rows)
    {
        AscendC::LocalTensor<float> cLocal = outQueueC_.DeQue<float>();
        AscendC::DataCopy(cGm_[rowOffset * N_], cLocal, rows * N_);
        outQueueC_.FreeTensor(cLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;   // B loaded once
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(A, B, C, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileM);
    op.Process();
}
"""
