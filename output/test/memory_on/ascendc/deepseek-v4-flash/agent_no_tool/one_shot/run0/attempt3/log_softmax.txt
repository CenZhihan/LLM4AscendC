host_operator_src="""
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const auto &originShape = inputShape->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(originShape.GetShapeSize());

    uint32_t batchSize = 1;
    uint32_t rowLength = totalLength;
    if (originShape.GetDimCount() == 2) {
        batchSize = static_cast<uint32_t>(originShape.GetDim(0));
        rowLength = static_cast<uint32_t>(originShape.GetDim(1));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int reductionDim = (dimPtr != nullptr) ? static_cast<int>(*dimPtr) : 1;

    // Tile size: we need at least 3 floats per element (x, exp_tmp, out_tmp)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) tileLen = rowLength;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = rowLength;

    // Block dim: limited by number of rows.
    uint32_t numBlocks = CeilDiv(batchSize, 8u); // at least 8 rows per block
    uint32_t blockDim = numBlocks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_batchSize(batchSize);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_reductionDim(reductionDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // Workspace: batchSize floats for row max, batchSize floats for row sumexp
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = batchSize * 2 * sizeof(float);
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name) {
        this->Input("x").ParamType(REQUIRED).DataType({ge::DT_FLOAT}).Format({ge::FORMAT_ND}).UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y").ParamType(REQUIRED).DataType({ge::DT_FLOAT}).Format({ge::FORMAT_ND}).UnknownShapeFormat({ge::FORMAT_ND});
        this->Attr("dim").AttrType(OPTIONAL).Int(1);
        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace,
                                uint32_t totalLength, uint32_t batchSize,
                                uint32_t rowLength, uint32_t tileLength,
                                int reductionDim) {
        totalLen_ = totalLength;
        batchSize_ = batchSize;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        reducDim_ = reductionDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        rowMaxGm_.SetGlobalBuffer((__gm__ float*)workspace, batchSize_);
        rowSumGm_.SetGlobalBuffer((__gm__ float*)workspace + batchSize_, batchSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));      // exp temporary
        pipe_.InitBuffer(redQueue_, 1, sizeof(float));                // scalar for reduction
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Each block processes rows cyclically
        uint32_t rowsPerBlock = CeilDiv(batchSize_, blockNum);
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (startRow >= batchSize_) return;
        if (endRow > batchSize_) endRow = batchSize_;

        for (uint32_t r = startRow; r < endRow; ++r) {
            // ---------- Pass 1: find max ----------
            float rowMax = -FLT_MAX;
            uint32_t rowOffset = r * rowLen_;
            uint32_t remaining = rowLen_;
            uint32_t base = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                // copy tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + base], curLen);
                inQueueX_.EnQue(xLocal);

                // compute local max
                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> scalar = redQueue_.AllocTensor<float>();
                AscendC::ReduceMax(tmp, xLocal, curLen);
                AscendC::MaxValue(scalar, tmp);          // get the scalar max
                float localMax;
                AscendC::BufferToScalar(localMax, scalar);
                if (localMax > rowMax) rowMax = localMax;

                tmpQueue_.FreeTensor(tmp);
                redQueue_.FreeTensor(scalar);
                inQueueX_.FreeTensor(xLocal);

                base += curLen;
                remaining -= curLen;
            }
            // Store row max to workspace
            AscendC::LocalTensor<float> maxScalar = redQueue_.AllocTensor<float>();
            AscendC::ScalarTensor(maxScalar, rowMax);
            AscendC::DataCopy(rowMaxGm_[r], maxScalar, 1);
            redQueue_.FreeTensor(maxScalar);

            // ---------- Pass 2: compute sum of exp(x - max) ----------
            float rowSum = 0.0f;
            remaining = rowLen_;
            base = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                // copy tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + base], curLen);
                inQueueX_.EnQue(xLocal);

                // compute exp(x - max)
                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> t = tmpQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> e = tmpQueue_.AllocTensor<float>(); // reuse?
                AscendC::Sub(t, xLocal, rowMax, curLen);
                AscendC::Exp(e, t, curLen);      // no in-place? assume ok
                // accumulate sum
                AscendC::LocalTensor<float> sumScalar = redQueue_.AllocTensor<float>();
                AscendC::ReduceSum(sumScalar, e, curLen);
                float partial;
                AscendC::BufferToScalar(partial, sumScalar);
                rowSum += partial;

                tmpQueue_.FreeTensor(t);
                tmpQueue_.FreeTensor(e);
                redQueue_.FreeTensor(sumScalar);
                inQueueX_.FreeTensor(xLocal);

                base += curLen;
                remaining -= curLen;
            }
            // Store row sum to workspace
            AscendC::LocalTensor<float> sumScalar = redQueue_.AllocTensor<float>();
            AscendC::ScalarTensor(sumScalar, rowSum);
            AscendC::DataCopy(rowSumGm_[r], sumScalar, 1);
            redQueue_.FreeTensor(sumScalar);

            // ---------- Pass 3: compute output y = (x - max) - log(sum) ----------
            float logSum = logf(rowSum);
            remaining = rowLen_;
            base = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                // Prefetch next tile? Simple double buffering may be omitted for clarity
                // copy input tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + base], curLen);
                inQueueX_.EnQue(xLocal);

                // compute y = (x - max) - logSum
                xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Sub(yLocal, xLocal, rowMax, curLen);
                AscendC::Adds(yLocal, yLocal, -logSum, curLen);   // subtract logSum
                outQueueY_.EnQue(yLocal);

                // copy out
                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[rowOffset + base], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);

                base += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> redQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> rowMaxGm_;
    AscendC::GlobalTensor<float> rowSumGm_;

    uint32_t totalLen_;
    uint32_t batchSize_;
    uint32_t rowLen_;
    uint32_t tileLen_;
    int reducDim_;
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, workspace,
            tiling_data.totalLength, tiling_data.batchSize,
            tiling_data.rowLength, tiling_data.tileLength,
            tiling_data.reductionDim);
    op.Process();
}
"""
