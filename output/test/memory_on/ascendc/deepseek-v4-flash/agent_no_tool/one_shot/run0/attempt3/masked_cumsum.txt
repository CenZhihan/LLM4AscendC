host_operator_src='''
#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
  return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  MaskedCumsumCustomTilingData tiling;

  const auto *shapeX = context->GetInputShape(0);
  const uint32_t totalLength =
      static_cast<uint32_t>(shapeX->GetOriginShape().GetShapeSize());

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
  int64_t dim = *dimPtr;

  const auto &originShape = shapeX->GetOriginShape();
  int64_t ndim = originShape.GetDimNum();
  if (dim < 0) dim += ndim;
  if (dim < 0 || dim >= ndim) dim = 0;

  uint32_t innerSize = 1;
  if (ndim > 0) {
    for (int64_t i = 0; i < ndim; i++) {
      uint32_t d = static_cast<uint32_t>(originShape.GetDim(i));
      if (i == dim) innerSize = d;
    }
  }
  if (innerSize == 0) innerSize = 1;

  // Max elements that fit in UB (3 buffers: x, mask, y)
  uint32_t maxTile = UB_BUDGET_BYTES / (3u * sizeof(float));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = maxTile;
  if (totalLength < tileLen) {
    tileLen = AlignDown(totalLength, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = totalLength;
  }

  // blockDim = number of rows, each block processes one or more entire rows
  uint32_t totalRows = CeilDiv(totalLength, innerSize);
  uint32_t blockDim = totalRows;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_totalLength(totalLength);
  tiling.set_tileLength(tileLen);
  tiling.set_innerSize(innerSize);
  tiling.set_blockDim(blockDim);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *workspace = context->GetWorkspaceSizes(1);
  workspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape *x_shape = context->GetInputShape(0);
  gert::Shape *y_shape = context->GetOutputShape(0);
  *y_shape = *x_shape;
  return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
  const ge::DataType x_dtype = context->GetInputDataType(0);
  context->SetOutputDataType(0, x_dtype);
  return GRAPH_SUCCESS;
}
}  // namespace ge

kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;   // single buffer, all resources freed per segment

class KernelMaskedCumsum {
public:
  __aicore__ inline KernelMaskedCumsum() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                              uint32_t totalLength, uint32_t tileLength,
                              uint32_t innerSize) {
    totalLen_ = totalLength;
    tileLen_ = tileLength;
    innerSize_ = innerSize;

    xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
    maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

    pipe_.InitBuffer(inX_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(inMask_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(outY_, BUFFER_NUM, tileLen_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    // Each block processes a distinct set of rows (cyclic for load balance)
    uint32_t totalRows = totalLen_ / innerSize_;
    for (uint32_t row = blockIdx; row < totalRows; row += blockNum) {
      uint32_t rowOffset = row * innerSize_;
      float carry = 0.0f;
      uint32_t remaining = innerSize_;
      uint32_t segStart = 0;
      while (remaining > 0) {
        uint32_t segLen = remaining;
        if (segLen > tileLen_) segLen = tileLen_;

        // Allocate and load x, mask
        AscendC::LocalTensor<float> xLocal = inX_.AllocTensor<float>();
        AscendC::LocalTensor<float> maskLocal = inMask_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowOffset + segStart], segLen);
        AscendC::DataCopy(maskLocal, maskGm_[rowOffset + segStart], segLen);

        // Compute x * mask
        AscendC::Mul(xLocal, xLocal, maskLocal, segLen);
        inMask_.FreeTensor(maskLocal);

        // In-place prefix sum on the segment (axis=0)
        AscendC::Cumsum(xLocal, xLocal, 0, segLen);

        // Add previous carry to all elements
        AscendC::Adds(xLocal, xLocal, carry, segLen);

        // Save the last element as new carry for this row
        carry = xLocal.GetValue(segLen - 1);

        // Store result to y
        AscendC::DataCopy(yGm_[rowOffset + segStart], xLocal, segLen);
        inX_.FreeTensor(xLocal);

        remaining -= segLen;
        segStart += segLen;
      }
    }
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inX_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inMask_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outY_;

  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> maskGm_;
  AscendC::GlobalTensor<float> yGm_;

  uint32_t totalLen_ {0};
  uint32_t tileLen_ {0};
  uint32_t innerSize_ {1};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x,
                                                           GM_ADDR mask,
                                                           GM_ADDR y,
                                                           GM_ADDR workspace,
                                                           GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelMaskedCumsum op;
  op.Init(x, mask, y,
          tiling_data.totalLength,
          tiling_data.tileLength,
          tiling_data.innerSize);
  op.Process();
}
'''
