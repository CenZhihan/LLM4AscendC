#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ROWS = 1;
static constexpr uint32_t HARD_CAP_ROWS = 4096;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::StorageShape* xStorageShape = context->GetInputShape(0);
    const gert::Shape& xShape = xStorageShape->GetOriginShape();
    uint32_t numDims = xShape.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = *attrs->GetAttrPointer<int32_t>(0);
    // Normalize dim
    if (dim < 0) dim += numDims;
    if (dim < 0 || dim >= static_cast<int32_t>(numDims)) {
        dim = 0;
    }

    uint32_t totalElements = 1;
    for (uint32_t i = 0; i < numDims; ++i) totalElements *= xShape.GetDim(i);
    uint32_t cols = xShape.GetDim(dim);
    uint32_t rows = totalElements / cols;

    // Tiling: each core processes a chunk of rows
    uint32_t rowsPerTile = UB_BUDGET_BYTES / (cols * sizeof(float) + sizeof(int64_t) * CeilDiv(cols, 32));
    if (rowsPerTile < 1) rowsPerTile = 1;
    if (rowsPerTile > HARD_CAP_ROWS) rowsPerTile = HARD_CAP_ROWS;

    uint32_t numTiles = CeilDiv(rows, rowsPerTile);
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);
    tiling.set_totalRows(rows);
    tiling.set_cols(cols);
    tiling.set_tileRows(rowsPerTile);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::StorageShape* xStorageShape = context->GetInputShape(0);
    const gert::Shape& xShape = xStorageShape->GetOriginShape();
    uint32_t numDims = xShape.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = *attrs->GetAttrPointer<int32_t>(0);
    if (dim < 0) dim += numDims;
    if (dim < 0 || dim >= static_cast<int32_t>(numDims)) dim = 0;

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> yDims;
    for (uint32_t i = 0; i < numDims; ++i) {
        if (static_cast<int32_t>(i) != dim) {
            yDims.push_back(xShape.GetDim(i));
        }
    }
    y_shape->SetDimNum(yDims.size());
    for (size_t i = 0; i < yDims.size(); ++i) {
        y_shape->SetDim(i, yDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t cols,
                                uint32_t tileRows, int32_t dim)
    {
        totalRows_ = totalRows;
        cols_ = cols;
        tileRows_ = tileRows;
        dim_ = dim;
        // For simplicity, assume input is flattened as (totalRows, cols) in row-major order.
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalRows_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileRows_ * cols_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * sizeof(int64_t));
        pipe_.InitBuffer(tmpQueue_, 1, cols_ * sizeof(float));
        pipe_.InitBuffer(tmpIdxQueue_, 1, sizeof(int64_t) * (cols_ + 31) / 32); // not used
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || cols_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid stride over rows
        uint32_t startRow = blockIdx * tileRows_;
        if (startRow >= totalRows_) return;

        while (true) {
            uint32_t rowsThisTile = tileRows_;
            if (startRow + rowsThisTile > totalRows_)
                rowsThisTile = totalRows_ - startRow;
            if (rowsThisTile == 0) break;

            // Load tile rows
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[startRow * cols_], rowsThisTile * cols_);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<int64_t> yLocal = outQueueY_.AllocTensor<int64_t>();

            // Compute argmax for each row
            for (uint32_t i = 0; i < rowsThisTile; ++i) {
                uint32_t offset = i * cols_;
                AscendC::LocalTensor<float> row = xDeq[offset];
                // Use a simple loop to find max index (sequential, not vectorized)
                float maxVal = row(0);
                int64_t maxIdx = 0;
                for (uint32_t j = 1; j < cols_; ++j) {
                    float val = row(j);
                    if (val > maxVal) {
                        maxVal = val;
                        maxIdx = j;
                    }
                }
                yLocal(i) = maxIdx;
            }

            outQueueY_.EnQue<int64_t>(yLocal);
            inQueueX_.FreeTensor(xDeq);

            // Copy out
            AscendC::LocalTensor<int64_t> yDeq = outQueueY_.DeQue<int64_t>();
            AscendC::DataCopy(yGm_[startRow], yDeq, rowsThisTile);
            outQueueY_.FreeTensor(yDeq);

            startRow += blockNum * tileRows_;
            if (startRow >= totalRows_) break;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpIdxQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalRows_ {0};
    uint32_t cols_ {0};
    uint32_t tileRows_ {0};
    int32_t dim_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.cols, tiling_data.tileRows, tiling_data.dim);
    op.Process();
}
