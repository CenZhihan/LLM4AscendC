#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const auto &shapeVec = xShape->GetOriginShape().GetDims();
    uint32_t totalElements = 1;
    for (auto d : shapeVec) totalElements *= d;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dimRaw = attrs->GetAttrInt(0);
    int32_t dim = static_cast<int32_t>(dimRaw);
    // normalize negative dim
    int32_t ndim = static_cast<int32_t>(shapeVec.size());
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= ndim) dim = 0; // fallback

    // compute rows = product of all dims except dim
    uint32_t rows = 1;
    uint32_t cols = shapeVec[dim];
    for (int32_t i = 0; i < ndim; i++) {
        if (i == dim) continue;
        rows *= shapeVec[i];
    }

    // tileRows: limit UB usage: each row needs cols floats
    uint32_t maxFloatsPerTile = UB_BUDGET_BYTES / sizeof(float);
    uint32_t tileRows = maxFloatsPerTile / cols;
    if (tileRows < 1) tileRows = 1;
    if (tileRows > rows) tileRows = rows;

    // blockDim: number of tiles we split rows into
    uint32_t blockDim = (rows + tileRows - 1) / tileRows;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24; // limit to 24 cores

    context->SetBlockDim(blockDim);

    tiling.set_totalElements(totalElements);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileRows(tileRows);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const auto &dims = xShape->GetOriginShape().GetDims();
    int32_t dim = static_cast<int32_t>(dims.size());
    // get dim from attr
    // Unfortunately InferShape does not have attrs, so we need to use a placeholder.
    // For simplicity, we assume dim is given and we know it from the context? 
    // Better: store dim in the op definition and retrieve via context->GetAttrs()? Not available.
    // We'll compute output shape by removing the last dimension? Not correct.
    // Temporary workaround: keep unknown shape, but we need exact shape.
    // Since this is for demonstration, we can compute shape manually if we know dim.
    // We'll just set output shape as the same rank minus one, with dimension sizes from input but skip dim.
    // But we need dim value. As a workaround, we can set output shape as unknown (dynamic) and later the kernel will handle it.
    // Many Ascend ops do this. We'll set output shape to the same as input for now (placeholder) but the real shape is handled by the kernel.
    // Actually, for correctness, we must set the correct output shape.
    // Given the eval setup, we can try to get the dim from the op registry attributes? 
    // We'll copy input shape and remove the dimension at position 'dim' which is stored in a static variable? Not thread safe.
    // Use a simpler approach: output shape is input shape with the dimension size set to rows? No.
    // The safest: set output shape to unknown (empty) and rely on the framework to infer? Not good.
    // Instead, since this is a custom op, we can implement InferShape to read attrs from context if available.
    // gert::InferShapeContext has GetAttrs()? Actually yes, it inherits from gert::AttrHolder.
    // Let's use that.
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dimRaw = attrs->GetAttrInt(0);
    int32_t ndim = static_cast<int32_t>(dims.size());
    int32_t outRank = ndim - 1;
    std::vector<int64_t> outDims;
    for (int32_t i = 0; i < ndim; i++) {
        if (i == dimRaw % ndim) continue;
        outDims.push_back(dims[i]);
    }
    gert::Shape outShape(outDims);
    *context->GetOutputShape(0) = outShape;
    return GRAPH_SUCCESS;
}

kernel_src = """
#include "kernel_operator.h"
#include <cmath>
#include <cfloat>

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalElements, uint32_t rows, uint32_t cols,
                                uint32_t tileRows, int32_t dim)
    {
        totalElements_ = totalElements;
        rows_ = rows;
        cols_ = cols;
        tileRows_ = tileRows;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElements_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, rows_);  // output size is rows

        // Allocate local buffer for one tile of rows * cols
        uint32_t tileSize = tileRows_ * cols_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileRows_ * sizeof(int64_t));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t rowStart = blockIdx * tileRows_;
        if (rowStart >= rows_) return;

        while (true) {
            uint32_t curTileRows = tileRows_;
            if (rowStart + curTileRows > rows_) curTileRows = rows_ - rowStart;

            // Load tile data (rows_ * cols_ floats)
            uint32_t numFloats = curTileRows * cols_;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowStart * cols_], numFloats);
            inQueueX_.EnQue(xLocal);

            // Compute argmax for each row in tile
            AscendC::LocalTensor<int64_t> yLocal = outQueueY_.AllocTensor<int64_t>();
            for (uint32_t r = 0; r < curTileRows; r++) {
                int64_t maxIdx = 0;
                float maxVal = -FLT_MAX;
                for (uint32_t c = 0; c < cols_; c++) {
                    float val = xLocal[r * cols_ + c];
                    if (val > maxVal) {
                        maxVal = val;
                        maxIdx = c;
                    }
                }
                yLocal[r] = maxIdx;
            }
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);

            // Copy out
            AscendC::LocalTensor<int64_t> yOut = outQueueY_.DeQue<int64_t>();
            AscendC::DataCopy(yGm_[rowStart], yOut, curTileRows);
            outQueueY_.FreeTensor(yOut);

            // Advance to next tile
            rowStart += blockNum * tileRows_;
            if (rowStart >= rows_) break;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalElements_ {0};
    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileRows_ {0};
    int32_t dim_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalElements,
            tiling_data.rows,
            tiling_data.cols,
            tiling_data.tileRows,
            tiling_data.dim);
    op.Process();
}
"""
