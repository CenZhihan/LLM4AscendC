#include "softplus_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: 2 * tile for in, 2 * tile for out, 1 * tile for tmp -> 5 * tile floats -> 20 * tile bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftplusCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *beta = attrs->GetAttrPointer<float>(0);
    const float *threshold = attrs->GetAttrPointer<float>(1);

    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_beta(*beta);
    tiling.set_threshold(*threshold);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftplusCustom {
public:
    __aicore__ inline KernelSoftplusCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                float beta, float threshold)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        beta_ = beta;
        threshold_ = threshold;
        invBeta_ = 1.0f / beta;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // t = beta * x
        AscendC::Muls(tLocal, xLocal, beta_, len);
        // mask = (t > threshold) ? 0.0f : 1.0f
        AscendC::LocalTensor<float> mask = tmpQueue_.AllocTensor<float>();  // reuse from tmpQueue? We already allocated tLocal; need another temp.
        // Actually we can reuse tLocal for mask after computing exp? Let's use the same buffer carefully.
        // Better: use a separate allocation from tmpQueue? tmpQueue has 1 * tile, we already used tLocal. We can use outQueueY's buffer? But we need yLocal for output.
        // Since we have only one tmp tile, we can repurpose tLocal after using it for exp.
        // Step: compute t = beta*x, then compute exp(t) into tLocal, then log1p, then multiply by invBeta.
        // We'll need a boolean mask for threshold. Use two temp buffers: tLocal and one from outQueueY? But outQueueY is for output. We can use yLocal as intermediate if careful.
        // Let's restructure: compute exp(beta*x) into tLocal, then compute log(1+exp(...)) into yLocal, then multiply by invBeta.
        // For threshold, we can compute x * (t > threshold) + (1/beta)*log(1+exp(beta*x)) * (t <= threshold). But that complicates.
        // Simpler: compute y = (1/beta) * log(1+exp(beta*x)), then for elements where beta*x > threshold, replace y with x.
        // We'll compute y in yLocal, then compute a mask and blend.
        // Steps:
        // 1. tLocal = beta * x
        // 2. exp(tLocal) -> tLocal (since we don't need t after)
        // 3. yLocal = 1.0f + tLocal
        // 4. AscendC::Log(yLocal, yLocal, len)
        // 5. yLocal = yLocal * invBeta
        // 6. mask = (tLocal > threshold)? Actually we need original t = beta*x, but tLocal now has exp. So we need to recompute t or store before exp. Better to compute t into a separate buffer.
        // Since we have only one tmp buffer, we can use the outQueueY buffer for output and compute step-by-step.
        // Let's allocate another local tensor from pipe? But pipe only has in, out, tmp. We can use a part of outQueueY? We have two buffers in outQueueY, we can use one for intermediate? But it's for output.
        // Simpler: use a separate local tensor from tmpQueue but we already allocated tLocal. We can free and reallocate? Not allowed in the same compute call.
        // Alternative: compute without threshold optimization? Softplus can be computed directly as log(1+exp(beta*x))/beta, but exp might overflow. However, for float, exp(88) ~ 2.6e38, so if beta*x > 88, it overflows. The threshold 20 is conservative. So we need threshold handling.
        // We'll implement with two tmp buffers by using the outQueueY buffer for one intermediate (since we don't need output until later). We'll store output in yLocal, but we can reuse it for intermediate.
        // Plan:
        // 1. Compute t = beta * x into tLocal.
        // 2. Compute mask = (t > threshold) as float 1.0 or 0.0 into yLocal (since we will overwrite it later).
        // 3. Compute exp(t) into tLocal.
        // 4. Compute log(1.0 + tLocal) into tLocal.
        // 5. Multiply tLocal by invBeta (into tLocal).
        // 6. Blend: y = mask * x + (1.0 - mask) * tLocal
        // But we need to store mask and xLocal. xLocal is still in the input queue, we haven't freed it yet. So we can use xLocal for x.
        // We'll keep xLocal until after blending.
        // Use yLocal as scratch for mask.
        // Compute mask first.
        // Then compute exp, log, mul into tLocal.
        // Then blend:
        //   AscendC::Mul(mask, mask, xLocal, len);
        //   AscendC::LocalTensor<float> oneMinusMask = tmpQueue_.AllocTensor<float>(); // but no more space.
        // Instead, we can compute (1.0 - mask) and then fma. But we have only one tmp buffer.
        // Maybe we can compute the final result directly using conditional select? AscendC has Select? Not sure.
        // Another approach: use the formula without threshold, but clamp beta*x to avoid overflow: beta*x = min(beta*x, 88.0f). Then compute exp.
        // That is safe and simpler. Softplus with beta* x clamped to 88 gives nearly x for large x. The threshold 20 is just to avoid overflow.
        // So we can clamp t = min(beta*x, 88.0f). Then y = (1/beta) * log(1+exp(t)). This is numerically stable and avoids conditionals.
        // We'll implement that.
        AscendC::Muls(tLocal, xLocal, beta_, len);
        AscendC::LocalTensor<float> clamps = tmpQueue_.AllocTensor<float>(); // reuse? Actually tLocal is from tmpQueue, we can use it.
        // But we already have tLocal. Let's set clamp value: 88.0f
        AscendC::Mins(tLocal, tLocal, 88.0f, len);  // now t = min(beta*x, 88)
        AscendC::Exp(tLocal, tLocal, len);           // exp(t)
        AscendC::Adds(tLocal, tLocal, 1.0f, len);    // 1 + exp(t)
        AscendC::Log(tLocal, tLocal, len);            // log(1+exp(t))
        AscendC::Muls(yLocal, tLocal, invBeta_, len); // y = (1/beta) * log(1+exp(clamped))

        outQueueY_.EnQue<float>(yLocal);
        tmpQueue_.FreeTensor(tLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    float beta_ {1.0f};
    float threshold_ {20.0f};
    float invBeta_ {1.0f};
};

extern "C" __global__ __aicore__ void softplus_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftplusCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.beta, tiling_data.threshold);
    op.Process();
}
