host_operator_src="""

#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>

namespace optiling {

static uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static uint32_t ComputeOutSize(uint32_t inSize, uint32_t kernelSize, uint32_t stride, uint32_t padding) {
    return (inSize + 2 * padding - kernelSize) / stride + 1;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaxPooling3dCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t batch = xShape->GetDim(0);
    uint32_t channels = xShape->GetDim(1);
    uint32_t depth = xShape->GetDim(2);
    uint32_t height = xShape->GetDim(3);
    uint32_t width = xShape->GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int kernel_size = attrs->GetAttrPointer<int>(0)[0];
    int stride = attrs->GetAttrPointer<int>(1)[0];
    int padding = attrs->GetAttrPointer<int>(2)[0];

    uint32_t depthOut = ComputeOutSize(depth, kernel_size, stride, padding);
    uint32_t heightOut = ComputeOutSize(height, kernel_size, stride, padding);
    uint32_t widthOut = ComputeOutSize(width, kernel_size, stride, padding);
    uint32_t totalOutElems = batch * channels * depthOut * heightOut * widthOut;

    // Conservative block dim: 8-24 based on batch*channels
    uint32_t numPlanes = batch * channels;
    uint32_t blockDim = std::min(numPlanes, (uint32_t)24);
    if (blockDim < 8) blockDim = 8;

    // Each block processes a contiguous chunk of output elements
    uint32_t tileLength = CeilDiv(totalOutElems, blockDim);
    // Align to 32 for vector access (optional)
    tileLength = (tileLength + 31) & ~31;
    if (tileLength > totalOutElems) tileLength = totalOutElems;

    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_depth(depth);
    tiling.set_height(height);
    tiling.set_width(width);
    tiling.set_kernelSize(kernel_size);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_depthOut(depthOut);
    tiling.set_heightOut(heightOut);
    tiling.set_widthOut(widthOut);
    tiling.set_totalOutElems(totalOutElems);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *xShape = context->GetInputShape(0);
    // Compute output shape from attributes
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int kernel_size = attrs->GetAttrPointer<int>(0)[0];
    int stride = attrs->GetAttrPointer<int>(1)[0];
    int padding = attrs->GetAttrPointer<int>(2)[0];

    uint32_t depthIn = xShape->GetDim(2);
    uint32_t heightIn = xShape->GetDim(3);
    uint32_t widthIn = xShape->GetDim(4);
    uint32_t depthOut = (depthIn + 2 * padding - kernel_size) / stride + 1;
    uint32_t heightOut = (heightIn + 2 * padding - kernel_size) / stride + 1;
    uint32_t widthOut = (widthIn + 2 * padding - kernel_size) / stride + 1;

    gert::Shape *yShape = context->GetOutputShape(0);
    *yShape = gert::Shape({(int64_t)xShape->GetDim(0),
                           (int64_t)xShape->GetDim(1),
                           (int64_t)depthOut,
                           (int64_t)heightOut,
                           (int64_t)widthOut});
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
        GET_TILING_DATA(td, tiling);
        batch_ = td.batch;
        channels_ = td.channels;
        depth_ = td.depth;
        height_ = td.height;
        width_ = td.width;
        kernelSize_ = td.kernelSize;
        stride_ = td.stride;
        padding_ = td.padding;
        depthOut_ = td.depthOut;
        heightOut_ = td.heightOut;
        widthOut_ = td.widthOut;
        totalOutElems_ = td.totalOutElems;
        tileLength_ = td.tileLength;
        blockDim_ = td.blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Input strides (assuming contiguous 5D): N*C*D*H*W
        strideC_ = depth_ * height_ * width_;
        strideD_ = height_ * width_;
        strideH_ = width_;
        // Output strides
        strideCout_ = depthOut_ * heightOut_ * widthOut_;
        strideDout_ = heightOut_ * widthOut_;
        strideHout_ = widthOut_;
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = blockDim_;

        // Each block processes a contiguous range of output elements
        uint32_t startIdx = blockIdx * tileLength_;
        uint32_t endIdx = startIdx + tileLength_;
        if (endIdx > totalOutElems_) endIdx = totalOutElems_;

        for (uint32_t idx = startIdx; idx < endIdx; idx++) {
            // Decode linear index to (n, c, d, h, w)
            uint32_t n = idx / (channels_ * strideCout_);
            uint32_t rem = idx % (channels_ * strideCout_);
            uint32_t c = rem / strideCout_;
            rem = rem % strideCout_;
            uint32_t d = rem / strideDout_;
            rem = rem % strideDout_;
            uint32_t h = rem / strideHout_;
            uint32_t w = rem % strideHout_;

            // Compute input window start
            int dStart = static_cast<int>(d) * static_cast<int>(stride_) - static_cast<int>(padding_);
            int hStart = static_cast<int>(h) * static_cast<int>(stride_) - static_cast<int>(padding_);
            int wStart = static_cast<int>(w) * static_cast<int>(stride_) - static_cast<int>(padding_);

            float maxVal = -1000000.0f; // small negative initial

            for (uint32_t kd = 0; kd < kernelSize_; kd++) {
                int id = dStart + static_cast<int>(kd);
                if (id < 0 || id >= static_cast<int>(depth_)) continue;
                for (uint32_t kh = 0; kh < kernelSize_; kh++) {
                    int ih = hStart + static_cast<int>(kh);
                    if (ih < 0 || ih >= static_cast<int>(height_)) continue;
                    for (uint32_t kw = 0; kw < kernelSize_; kw++) {
                        int iw = wStart + static_cast<int>(kw);
                        if (iw < 0 || iw >= static_cast<int>(width_)) continue;

                        // Compute input index
                        uint32_t inputIdx = n * strideC_ * channels_ + c * strideC_ + id * strideD_ + ih * strideH_ + iw;
                        float val = xGm_.GetValue(inputIdx);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }

            yGm_.SetValue(idx, maxVal);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t channels_;
    uint32_t depth_;
    uint32_t height_;
    uint32_t width_;
    uint32_t kernelSize_;
    uint32_t stride_;
    uint32_t padding_;
    uint32_t depthOut_;
    uint32_t heightOut_;
    uint32_t widthOut_;
    uint32_t totalOutElems_;
    uint32_t tileLength_;
    uint32_t blockDim_;

    // Strides for input and output (C contiguous layout assumed)
    uint32_t strideC_;
    uint32_t strideD_;
    uint32_t strideH_;
    uint32_t strideCout_;
    uint32_t strideDout_;
    uint32_t strideHout_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    KernelMaxPooling3dCustom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}

class ModelNew(torch.nn.Module):

        self.kernel_size = kernel_size
        if stride is None:

        self.stride = stride
        self.padding = padding

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
