kernel_src="""

#include "kernel_operator.h"

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
                                uint32_t totalSamples, uint32_t featureDim, float margin)
    {
        totalSamples_ = totalSamples;
        featureDim_   = featureDim;
        margin_       = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalSamples * featureDim);
        posGm_.SetGlobalBuffer((__gm__ float*)positive, totalSamples * featureDim);
        negGm_.SetGlobalBuffer((__gm__ float*)negative, totalSamples * featureDim);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        // Initialize local buffers for per-sample operations
        pipe_.InitBuffer(aBuf_, featureDim_ * sizeof(float));
        pipe_.InitBuffer(pBuf_, featureDim_ * sizeof(float));
        pipe_.InitBuffer(nBuf_, featureDim_ * sizeof(float));
        pipe_.InitBuffer(tBuf_, featureDim_ * sizeof(float));
        pipe_.InitBuffer(rBuf_, 1 * sizeof(float));
        pipe_.InitBuffer(outBuf_, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalSamples_ == 0 || featureDim_ == 0) return;

        float totalLoss = 0.0f;
        uint32_t len = featureDim_;

        for (uint32_t i = 0; i < totalSamples_; ++i) {
            uint32_t offset = i * featureDim_;

            // Allocate local tensors (size determined by InitBuffer)
            auto a = aBuf_.AllocTensor<float>();
            auto p = pBuf_.AllocTensor<float>();
            auto n = nBuf_.AllocTensor<float>();
            auto t = tBuf_.AllocTensor<float>();
            auto r = rBuf_.AllocTensor<float>();

            // Load row data from global
            AscendC::DataCopy(a, anchorGm_[offset], len);
            AscendC::DataCopy(p, posGm_[offset],   len);
            AscendC::DataCopy(n, negGm_[offset],   len);

            // dist_ap = sum((a-p)^2)
            AscendC::Sub(t, a, p, len);
            AscendC::Mul(t, t, t, len);
            AscendC::ReduceSum<AscendC::ReductionPattern::SUM>(r, t, len, 1);
            float distAp = r(0);

            // dist_an = sum((a-n)^2)
            AscendC::Sub(t, a, n, len);
            AscendC::Mul(t, t, t, len);
            AscendC::ReduceSum<AscendC::ReductionPattern::SUM>(r, t, len, 1);
            float distAn = r(0);

            // Hinge loss: max(0, margin + sqrt(dist_ap) - sqrt(dist_an))
            float loss = margin_ + sqrt(distAp) - sqrt(distAn);
            if (loss < 0.0f) loss = 0.0f;
            totalLoss += loss;

            // Free temporary tensors
            aBuf_.FreeTensor(a);
            pBuf_.FreeTensor(p);
            nBuf_.FreeTensor(n);
            tBuf_.FreeTensor(t);
            rBuf_.FreeTensor(r);
        }

        float meanLoss = totalLoss / totalSamples_;
        auto out = outBuf_.AllocTensor<float>();
        out(0) = meanLoss;
        AscendC::DataCopy(lossGm_, out, 1);
        outBuf_.FreeTensor(out);
    }

private:
    AscendC::TPipe pipe_;

    // Local buffers for per-sample computation
    AscendC::TBuf<AscendC::TPosition::VECCALC> aBuf_, pBuf_, nBuf_, tBuf_, rBuf_;
    AscendC::TBuf<AscendC::TPosition::VECOUT> outBuf_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> posGm_;
    AscendC::GlobalTensor<float> negGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalSamples_ {0};
    uint32_t featureDim_   {0};
    float    margin_       {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(
    GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalSamples, tiling_data.featureDim, tiling_data.margin);
    op.Process();
}
