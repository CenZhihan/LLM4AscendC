kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t T, uint32_t B, uint32_t I, uint32_t H, uint32_t O, uint32_t K)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        T_ = T;
        B_ = B;
        I_ = I;
        H_ = H;
        O_ = O;
        K_ = K;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T * B * I);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B * H);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H * K);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O * H);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // allocate local buffers for one batch per core
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, I_ * sizeof(float));  // x[t] for one batch
        pipe_.InitBuffer(inQueueH_, BUFFER_NUM, H_ * sizeof(float));  // h state for one batch
        pipe_.InitBuffer(inQueueBiasH_ , BUFFER_NUM, H_ * sizeof(float)); // b_i2h
        pipe_.InitBuffer(inQueueWH2O_ , BUFFER_NUM, O_ * H_ * sizeof(float)); // w_h2o full matrix
        pipe_.InitBuffer(inQueueBH2O_ , BUFFER_NUM, O_ * sizeof(float)); // b_h2o
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, O_ * sizeof(float)); // y_t for one batch
        pipe_.InitBuffer(tmpQueue_, 1, (I_ + H_ > O_ ? I_ + H_ : O_) * sizeof(float)); // general temp
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= B_) return;

        // Initialize h state from h0 for this batch
        AscendC::LocalTensor<float> hLocal = inQueueH_.AllocTensor<float>();
        AscendC::DataCopy(hLocal, h0Gm_[blockIdx * H_], H_);
        inQueueH_.EnQue(hLocal);

        // Pre-load w_h2o and biases (they fit in UB)
        AscendC::LocalTensor<float> w_h2oLocal = inQueueWH2O_.AllocTensor<float>();
        AscendC::DataCopy(w_h2oLocal, w_h2oGm_, O_ * H_);
        inQueueWH2O_.EnQue(w_h2oLocal);

        AscendC::LocalTensor<float> b_h2oLocal = inQueueBH2O_.AllocTensor<float>();
        AscendC::DataCopy(b_h2oLocal, b_h2oGm_, O_);
        inQueueBH2O_.EnQue(b_h2oLocal);

        AscendC::LocalTensor<float> b_i2hLocal = inQueueBiasH_.AllocTensor<float>();
        AscendC::DataCopy(b_i2hLocal, b_i2hGm_, H_);
        inQueueBiasH_.EnQue(b_i2hLocal);

        // Main time loop
        for (uint32_t t = 0; t < T_; ++t) {
            // Dequeue h (from previous step)
            AscendC::LocalTensor<float> hPrev = inQueueH_.DeQue<float>();

            // Load x[t] for this batch
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[t * B_ * I_ + blockIdx * I_], I_);

            // Concatenate x and h into one buffer for dot products
            // tmp = [x: I, h: H]
            AscendC::LocalTensor<float> catLocal = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(catLocal, xLocal, I_);
            AscendC::DataCopy(catLocal[I_], hPrev, H_);

            // Compute pre = inp @ w_i2h^T + b_i2h
            AscendC::LocalTensor<float> preLocal = tmpQueue_.AllocTensor<float>();
            for (uint32_t h = 0; h < H_; ++h) {
                float dot = 0.0f;
                // dot over I part
                for (uint32_t k = 0; k < I_; ++k) {
                    dot += catLocal[k] * w_i2hGm_[h * K_ + k];
                }
                // dot over H part
                for (uint32_t k = 0; k < H_; ++k) {
                    dot += catLocal[I_ + k] * w_i2hGm_[h * K_ + I_ + k];
                }
                dot += b_i2hLocal[h];
                preLocal[h] = dot;
            }

            // tanh activation -> new h
            AscendC::Tanh(hPrev, preLocal, H_);

            // Compute y_t = h @ w_h2o^T + b_h2o
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            for (uint32_t o = 0; o < O_; ++o)))
