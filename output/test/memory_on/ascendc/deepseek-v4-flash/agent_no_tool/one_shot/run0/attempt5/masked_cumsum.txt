host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;
    const gert::StorageShape *inStorageShape = context->GetInputShape(0);
    const gert::Shape &inShape = inStorageShape->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(inShape.GetShapeSize());
    uint32_t ndim = static_cast<uint32_t>(inShape.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim = attrs->GetAttrPointer<int64_t>(0);
    int64_t dimVal = *dim;
    if (dimVal < 0) dimVal += ndim;

    // Compute rows and cols: rows = product of dims before dim, cols = dim size
    uint32_t rows = 1, cols = 1;
    for (uint32_t i = 0; i < ndim; i++) {
        uint32_t d = static_cast<uint32_t>(inShape.GetDim(i));
        if (i < static_cast<uint32_t>(dimVal)) rows *= d;
        else if (i == static_cast<uint32_t>(dimVal)) cols = d;
        else rows *= d; // after dim, treat as part of rows? Actually we want to flatten all dims except dim into rows.
    }
    // Actually for generic ND, we want to collapse all dimensions except dim into rows.
    // So rows = totalLength / cols; (since cols = shape[dim])
    uint32_t cols_from_shape = static_cast<uint32_t>(inShape.GetDim(dimVal));
    rows = totalLength / cols_from_shape;
    cols = cols_from_shape;

    // tiling parameters
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float)); // x, mask, y, plus temp -> approx 4 tiles, use 3 for safety
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = (cols < maxTile) ? ( (AlignDown(cols, ALIGN_ELEMS) > 0) ? AlignDown(cols, ALIGN_ELEMS) : cols ) : maxTile;

    uint32_t numTilesPerRow = CeilDiv(cols, tileLen);
    uint32_t totalTiles = rows * numTilesPerRow;
    uint32_t blockDim = totalTiles < 24u ? (totalTiles < 8u ? 8u : totalTiles) : 24u;
    blockDim = blockDim > 24u ? 24u : blockDim;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MaskedCumsumCustom : public OpDef {
public:
    explicit MaskedCumsumCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("mask")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MaskedCumsumCustom);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t rows,
                                uint32_t cols, uint32_t tileLen,
                                uint32_t blockDim)
    {
        totalLen_ = totalLength;
        rows_ = rows;
        cols_ = cols;
        tileLen_ = tileLen;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rows_ == 0 || cols_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Each block handles rows in a round-robin manner
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            float runningSum = 0.0f;
            const uint32_t rowOffset = row * cols_;
            for (uint32_t offset = 0; offset < cols_; offset += tileLen_) {
                uint32_t len = (offset + tileLen_ <= cols_) ? tileLen_ : (cols_ - offset);
                // Copy in x and mask for this tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + offset], len);
                AscendC::DataCopy(maskLocal, maskGm_[rowOffset + offset], len);
                inQueueX_.EnQue(xLocal);
                inQueueMask_.EnQue(maskLocal);

                // Compute and enqueue output
                AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> maskDeq = inQueueMask_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

                // t = x * mask
                AscendC::Mul(tLocal, xDeq, maskDeq, len);

                // inclusive prefix sum on tLocal with carry from previous tile
                for (uint32_t i = 0; i < len; i++) {
                    if (i == 0) {
                        tLocal[i] = runningSum + tLocal[i];
                    } else {
                        tLocal[i] = tLocal[i - 1] + tLocal[i];
                    }
                }
                runningSum = tLocal[len - 1];

                // Copy tLocal to yLocal and enqueue
                AscendC::Copy(yLocal, tLocal, len);
                tmpQueue_.FreeTensor(tLocal);
                outQueueY_.EnQue<float>(yLocal);

                // Free input tensors
