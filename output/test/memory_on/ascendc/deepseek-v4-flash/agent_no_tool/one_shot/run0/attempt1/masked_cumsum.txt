host_operator_src='''
#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: 2 in-flights + 1 out = 12 bytes per element (3 tiles of 4B each)
// We use a conservative 256 KB per core.
static constexpr uint32_t UB_BUDGET_BYTES = 256 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
  return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  MaskedCumsumCustomTilingData tiling;

  // Use StorageShape* (fix from memory-1)
  const auto *shapeX = context->GetInputShape(0);
  const uint32_t totalLength =
      static_cast<uint32_t>(shapeX->GetOriginShape().GetShapeSize());

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
  const int64_t dim = *dimPtr;

  // Compute inner_size = shape[dim]
  const auto &originShape = shapeX->GetOriginShape();
  int64_t innerSize = 1;
  if (dim >= 0 && dim < originShape.GetDimNum()) {
    innerSize = originShape.GetDim(dim);
  }

  // Max elements that fit in UB (2 in‑flight tiles)
  uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

  // Make tile length a multiple of inner size whenever possible
  uint32_t tileLen = maxTile;
  if (totalLength < tileLen) {
    tileLen = AlignDown(totalLength, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = totalLength;
  } else if (innerSize <= tileLen) {
    // Ensure whole slices per tile for correct cumulative sum
    uint32_t alignedSlices = AlignDown(tileLen / innerSize, 1u);
    if (alignedSlices == 0) alignedSlices = 1;
    tileLen = alignedSlices * innerSize;
  }

  // Block dimension: enough to cover all tiles
  uint32_t numTiles = CeilDiv(totalLength, tileLen);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_totalLength(totalLength);
  tiling.set_tileLength(tileLen);
  tiling.set_blockDim(blockDim);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *workspace = context->GetWorkspaceSizes(1);
  workspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape *x_shape = context->GetInputShape(0);
  gert::Shape *y_shape = context->GetOutputShape(0);
  *y_shape = *x_shape;
  return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
  const ge::DataType x_dtype = context->GetInputDataType(0);
  context->SetOutputDataType(0, x_dtype);
  return GRAPH_SUCCESS;
}
}  // namespace ge

namespace ops {
class MaskedCumsumCustom : public OpDef {
public:
  explicit MaskedCumsumCustom(const char* name) : OpDef(name) {
    this->Input("x")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});

    this->Input("mask")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});

    this->Output("y")
        .ParamType(REQUIRED)
        .DataType({ge::DT_FLOAT})
        .Format({ge::FORMAT_ND})
        .UnknownShapeFormat({ge::FORMAT_ND});

    this->Attr("dim").AttrType(OPTIONAL).Int(0);

    this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
    this->AICore().SetTiling(optiling::TilingFunc);
    this->AICore().AddConfig("ascend910b");
  }
};

OP_ADD(MaskedCumsumCustom);
}  // namespace ops

kernel_src='''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsum {
public:
  __aicore__ inline KernelMaskedCumsum() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                              uint32_t totalLength, uint32_t tileLength) {
    totalLen_ = totalLength;
    tileLen_ = tileLength;

    xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
    maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

    // Two input queues (x and mask), one output queue (y)
    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    // Grid‑stride over tiles (each tile is a chunk of contiguous elements)
    uint32_t base = blockIdx * tileLen_;
    if (base >= totalLen_) return;

    while (true) {
      uint32_t curLen = totalLen_ - base;
      if (curLen > tileLen_) curLen = tileLen_;

      // Prefetch next tile early if available
      uint32_t nextBase = base + blockNum * tileLen_;
      uint32_t nextLen = 0;
      if (nextBase < totalLen_) {
        nextLen = totalLen_ - nextBase;
        if (nextLen > tileLen_) nextLen = tileLen_;
        CopyIn(nextBase, nextLen);
      }

      CopyIn(base, curLen);
      Compute(curLen);
      CopyOut(base, curLen);

      if (nextLen == 0) break;
      base = nextBase;
      curLen = nextLen;
    }
  }

private:
  __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len) {
    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
    AscendC::DataCopy(xLocal, xGm_[gmOffset], len);

    AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
    AscendC::DataCopy(maskLocal, maskGm_[gmOffset], len);

    inQueueX_.EnQue(xLocal);
    inQueueMask_.EnQue(maskLocal);
  }

  __aicore__ inline void Compute(uint32_t len) {
    // Dequeue x and mask
    AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
    AscendC::LocalTensor<float> maskLocal = inQueueMask_.DeQue<float>();

    // y = x * mask   (elementwise multiplication)
    AscendC::Mul(xLocal, xLocal, maskLocal, len);

    // Free mask – we no longer need it
    inQueueMask_.FreeTensor(maskLocal);

    // Now compute cumulative sum (in‑place on xLocal)
    // Uses a simple sequential loop – sufficient for tile sizes up to ~32k
    for (uint32_t i = 1u; i < len; ++i) {
      xLocal.SetValue(i, xLocal.GetValue(i) + xLocal.GetValue(i - 1));
    }

    // Enqueue result as output
    outQueueY_.EnQue(xLocal);
  }

  __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len) {
    AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
    AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
    outQueueY_.FreeTensor(yLocal);
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> maskGm_;
  AscendC::GlobalTensor<float> yGm_;

  uint32_t totalLen_ {0};
  uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x,
                                                           GM_ADDR mask,
                                                           GM_ADDR y,
                                                           GM_ADDR workspace,
                                                           GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelMaskedCumsum op;
  op.Init(x, mask, y,
          tiling_data.totalLength,
          tiling_data.tileLength);
  op.Process();
}
'''

at::Tensor masked_cumsum_impl_npu(const at::Tensor& self,
                                   const at::Tensor& mask,
                                   int64_t dim) {

    // The operator "MaskedCumsumCustom" is registered in the custom ops library
    EXEC_NPU_CMD(aclnnMaskedCumsumCustom, self, mask, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("masked_cumsum_custom", &masked_cumsum_impl_npu);
