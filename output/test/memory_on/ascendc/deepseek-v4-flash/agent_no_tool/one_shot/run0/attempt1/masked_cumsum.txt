#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;
    const gert::Shape *inShape = context->GetInputShape(0);
    const gert::Shape *maskShape = context->GetInputShape(1);
    uint32_t totalLength = static_cast<uint32_t>(inShape->GetOriginShape().GetShapeSize());

    // Infer rows and cols from the dim attribute.
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim = attrs->GetAttrPointer<int64_t>(0);
    int64_t dimVal = *dim;
    uint32_t ndim = static_cast<uint32_t>(inShape->GetOriginShape().GetDimNum());
    if (dimVal < 0) dimVal += ndim;
    uint32_t rows = 1, cols = totalLength;
    if (ndim == 2) {
        uint32_t d0 = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(0));
        uint32_t d1 = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(1));
        if (dimVal == 0) { rows = d1; cols = d0; }
        else { rows = d0; cols = d1; }
    } else if (ndim == 1) {
        rows = 1; cols = totalLength;
    }

    // Compute tileLen: need x, mask, y, tmp (3+1=4 tensors), double buffer for inX/inMask/outY => 3*2+1=7 tensors
    // But we can use single buffer for tmp, and double for others.
    // UB bytes needed: 2*tileLen*4 (inX) + 2*tileLen*4 (inMask) + 2*tileLen*4 (outY) + 1*tileLen*4 (tmp) = 7*tileLen*4 = 28*tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 28;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = (maxTile > cols) ? cols : maxTile;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = cols;

    // Block dim: assign rows to blocks, cap at 24
    uint32_t blockDim = rows;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t tileLen,
                                uint32_t blockDim)
    {
        rows_ = rows;
        cols_ = cols;
        tileLen_ = tileLen;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, rows_ * cols_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, rows_ * cols_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = blockDim_;

        // Each block processes rows in stride
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        uint32_t col = 0;
        float carry = 0.0f;

        while (col < cols_) {
            uint32_t len = (col + tileLen_ <= cols_) ? tileLen_ : (cols_ - col);

            // Prefetch next tile
            bool hasNext = (col + tileLen_ < cols_);
            if (hasNext) {
                uint32_t nextLen = (col + 2*tileLen_ <= cols_) ? tileLen_ : (cols_ - col - tileLen_);
                CopyIn(rowIdx, col + tileLen_, nextLen); // next tile
            }

            // Compute current tile with carry
            CopyIn(rowIdx, col, len);
            ComputeWithCarry(len, carry);

            // After compute, carry holds the last cumulative sum of this tile
            carry = ComputeCarry(len, carry);

            CopyOut(rowIdx, col, len);

            col += tileLen_;
        }
    }

    __aicore__ inline void CopyIn(uint32_t row, uint32_t col, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
        uint32_t offset = row * cols_ + col;
        AscendC::DataCopy(xLocal, xGm_[offset], len);
        AscendC::DataCopy(maskLocal, maskGm_[offset], len);
        inQueueX_.EnQue(xLocal);
        inQueueMask_.EnQue(maskLocal);
    }

    __aicore__ inline void ComputeWithCarry(uint32_t len, float carry)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maskLocal = inQueueMask_.DeQue<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // tmp = x * mask
        AscendC::Mul(tmp, xLocal, maskLocal, len);

        // Compute prefix sum with carry
        for (uint32_t i = 0; i < len; i++) {
            float val = tmp(i);
            if (i == 0) {
                yLocal(i) = carry + val;
            } else {
                yLocal(i) = yLocal(i-1) + val;
            }
        }

        tmpQueue_.FreeTensor(tmp);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueMask_.FreeTensor(maskLocal);
    }

    __aicore__ inline float ComputeCarry(uint32_t len, float prevCarry)
    {
        // Return the last cumulative sum of this tile
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        float last = yLocal(len - 1);
        outQueueY_.EnQue(yLocal);
        return last;
    }

    __aicore__ inline void CopyOut(uint32_t row, uint32_t col, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint32_t offset = row * cols_ + col;
        AscendC::DataCopy(yGm_[offset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.rows, tiling_data.cols, tiling_data.tileLen, tiling_data.blockDim);
    op.Process();
}
