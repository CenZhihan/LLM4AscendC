host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: each pass needs 1 input tile + 1 output tile + 1 temp tile (max 3 tiles)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = 1;
    if (attrs->GetAttrPointer<int32_t>(0) != nullptr) {
        dim = *attrs->GetAttrPointer<int32_t>(0);
    }

    // Infer rowLength from shape (assume dim is the last axis)
    const gert::Shape *shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t rowLength = 1;
    for (int32_t i = dim; i < shape->GetDimNum(); ++i) {
        rowLength *= shape->GetDim(i);
    }
    uint32_t numRows = totalLength / rowLength;

    // Maximum tile size based on UB budget (3 tiles in flight)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));  // 3 * 4 = 12 bytes per element
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        tileLen = AlignDown(rowLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowLength;
    }

    // Block dim: make each block handle at least one row
    uint32_t blockDim = numRows;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_numRows(numRows);
    tiling.set_dim(dim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

        this->AICore().SetTiling(optiling::TilingFunc);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t tileLength, uint32_t numRows,
                                int32_t dim)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        numRows_ = numRows;
        (void)dim;  // dim is implicitly used to compute rowLen; we assume last dim

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || numRows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < numRows_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        uint32_t baseOffset = rowIdx * rowLen_;

        // Pass 1: Find maximum
        float maxVal = -__builtin_huge_valf();
        uint32_t offset = baseOffset;
        uint32_t remaining = rowLen_;
        uint32_t curTileLen = 0;

        while (remaining > 0) {
            curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curTileLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> tileMaxLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceMax(tileMaxLocal, xLocal, curTileLen);
            float tileMax = tileMaxLocal(0);
            tmpQueue_.FreeTensor(tileMaxLocal);

            if (tileMax > maxVal) maxVal = tileMax;

            inQueueX_.FreeTensor(xLocal);
            offset += curTileLen;
            remaining -= curTileLen;
        }

        // Pass 2: Compute sum of exp(x - max)
        float sumExp = 0.0f;
        offset = baseOffset;
        remaining = rowLen_;

        while (remaining > 0) {
            curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curTileLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> xTmp = tmpQueue_.AllocTensor<float>();
            // xTmp = x - maxVal
            AscendC::Adds(xTmp, xLocal, -maxVal, curTileLen);
            AscendC::Exp(xTmp, xTmp, curTileLen);

            AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumLocal, xTmp, curTileLen);
            sumExp += sumLocal(0);

            tmpQueue_.FreeTensor(xTmp);
            tmpQueue_.FreeTensor(sumLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curTileLen;
            remaining -= curTileLen;
        }

        // Compute log(sumExp) using a scalar tensor
        AscendC::LocalTensor<float> scalarLocal = scalarQueue_.AllocTensor<float>();
        scalarLocal(0) = sumExp;
        AscendC::Log(scalarLocal, scalarLocal, 1);
        float logSum = scalarLocal(0);
        scalarQueue_.FreeTensor(scalarLocal);

        // Pass 3: Compute output y = x - maxVal - logSum
        offset = baseOffset;
        remaining = rowLen_;

        while (remaining > 0) {
            curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curTileLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Adds(yLocal, xLocal, -maxVal, curTileLen);
            AscendC::Adds(yLocal, yLocal, -logSum, curTileLen);

            AscendC::DataCopy(yGm_[offset], yLocal, curTileLen);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            offset += curTileLen;
            remaining -= curTileLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLength,
            tiling_data.tileLength, tiling_data.numRows, tiling_data.dim);
    op.Process();
}
