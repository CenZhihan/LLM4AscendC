#include <cmath>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static constexpr uint32_t ALIGN_ELEMS = 16;

static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {

  return (x / a) * a;

}

static inline int32_t OutputDim(int32_t in, int32_t k, int32_t s, int32_t p) {

  return static_cast<int32_t>(std::floor((in + 2 * p - k) / s)) + 1;

}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {

  MaxPooling3dCustomTilingData tiling;

  const gert::Shape *xShape = context->GetInputShape(0)->GetOriginShape();

  int32_t N = static_cast<int32_t>(xShape->GetDim(0));

  int32_t C = static_cast<int32_t>(xShape->GetDim(1));

  int32_t D = static_cast<int32_t>(xShape->GetDim(2));

  int32_t H = static_cast<int32_t>(xShape->GetDim(3));

  int32_t W = static_cast<int32_t>(xShape->GetDim(4));

  const gert::RuntimeAttrs *attrs = context->GetAttrs();

  int32_t kernel_size = attrs->GetAttrPointer<int32_t>(0)[0];

  int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];

  int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];

  int32_t D_out = OutputDim(D, kernel_size, stride, padding);

  int32_t H_out = OutputDim(H, kernel_size, stride, padding);

  int32_t W_out = OutputDim(W, kernel_size, stride, padding);

  uint32_t totalOutputElements = static_cast<uint32_t>(N) * static_cast<uint32_t>(C) *

                                 static_cast<uint32_t>(D_out) * static_cast<uint32_t>(H_out) *

                                 static_cast<uint32_t>(W_out);

  // Compute tile length: each output element requires reading about kernel_size^3 floats

  // plus one output float -> roughly (kernel_size*kernel_size*kernel_size + 1) * 4 bytes

  uint32_t windowSize = static_cast<uint32_t>(kernel_size * kernel_size * kernel_size);

  uint32_t bytesPerElement = (windowSize + 1) * sizeof(float);

  uint32_t maxTile = UB_BUDGET_BYTES / bytesPerElement;

  maxTile = AlignDown(maxTile, ALIGN_ELEMS);

  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = maxTile;

  if (totalOutputElements < tileLen) {

    uint32_t aligned = AlignDown(totalOutputElements, ALIGN_ELEMS);

    tileLen = (aligned > 0) ? aligned : totalOutputElements;

  }

  // blockDim: number of tiles, clamped

  uint32_t numTiles = (tileLen == 0) ? 1u : (totalOutputElements + tileLen - 1) / tileLen;

  uint32_t blockDim = numTiles;

  if (blockDim < 8u) blockDim = 8u;

  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_N(N);

  tiling.set_C(C);

  tiling.set_D(D);

  tiling.set_H(H);

  tiling.set_W(W);

  tiling.set_D_out(D_out);

  tiling.set_H_out(H_out);

  tiling.set_W_out(W_out);

  tiling.set_kernel_size(kernel_size);

  tiling.set_stride(stride);

  tiling.set_padding(padding);

  tiling.set_totalOutputElements(totalOutputElements);

  tiling.set_tileLength(tileLen);

  tiling.set_blockDim(blockDim);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

                      context->GetRawTilingData()->GetCapacity());

  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);

  currentWorkspace[0] = 0;

  return ge::GRAPH_SUCCESS;

}

}  // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context) {

  // Input shape: N, C, D, H, W

  const gert::Shape *x_shape = context->GetInputShape(0);

  int32_t N = static_cast<int32_t>(x_shape->GetDim(0));

  int32_t C = static_cast<int32_t>(x_shape->GetDim(1));

  int32_t D = static_cast<int32_t>(x_shape->GetDim(2));

  int32_t H = static_cast<int32_t>(x_shape->GetDim(3));

  int32_t W = static_cast<int32_t>(x_shape->GetDim(4));

  // Retrieve attributes from context

  const gert::RuntimeAttrs *attrs = context->GetAttrs();

  int32_t kernel_size = attrs->GetAttrPointer<int32_t>(0)[0];

  int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];

  int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];

  int32_t D_out = (D + 2 * padding - kernel_size) / stride + 1;

  int32_t H_out = (H + 2 * padding - kernel_size) / stride + 1;

  int32_t W_out = (W + 2 * padding - kernel_size) / stride + 1;

  gert::Shape *y_shape = context->GetOutputShape(0);

  y_shape->SetDim(0, N);

  y_shape->SetDim(1, C);

  y_shape->SetDim(2, D_out);

  y_shape->SetDim(3, H_out);

  y_shape->SetDim(4, W_out);

  return GRAPH_SUCCESS;

}

constexpr int32_t BUFFER_NUM = 2;

    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

        GET_TILING_DATA(tiling_data, tiling);

        N_ = tiling_data.N;

        C_ = tiling_data.C;

        D_ = tiling_data.D;

        H_ = tiling_data.H;

        W_ = tiling_data.W;

        D_out_ = tiling_data.D_out;

        H_out_ = tiling_data.H_out;

        W_out_ = tiling_data.W_out;

        kernel_size_ = tiling_data.kernel_size;

        stride_ = tiling_data.stride;

        padding_ = tiling_data.padding;

        totalOutputElements_ = tiling_data.totalOutputElements;

        tileLen_ = tiling_data.tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x);

        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Buffer for input window and the output element

        // max window size for kernel_size up to 7 (343 floats) safe

        windowSize_ = kernel_size_ * kernel_size_ * kernel_size_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, windowSize_ * sizeof(float));

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float)); // one output per tile element

        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float)); // temp for max

    }

    __aicore__ inline void Process() {

        if (totalOutputElements_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output elements

        const uint32_t strideElems = blockNum * tileLen_;

        uint32_t base = blockIdx * tileLen_;

        if (base >= totalOutputElements_) return;

        while (true) {

            uint32_t curLen = totalOutputElements_ - base;

            if (curLen > tileLen_) curLen = tileLen_;

            // Process tile: each element independently

            for (uint32_t i = 0; i < curLen; i++) {

                uint32_t globalIdx = base + i;

                int32_t ow = static_cast<int32_t>(globalIdx % W_out_);

                int32_t oh = static_cast<int32_t>((globalIdx / W_out_) % H_out_);

                int32_t od = static_cast<int32_t>((globalIdx / (W_out_ * H_out_)) % D_out_);

                int32_t c = static_cast<int32_t>((globalIdx / (W_out_ * H_out_ * D_out_)) % C_);

                int32_t n = static_cast<int32_t>(globalIdx / (W_out_ * H_out_ * D_out_ * C_));

                // Compute input window bounds

                int32_t in_d_start = od * stride_ - padding_;

                int32_t in_h_start = oh * stride_ - padding_;

                int32_t in_w_start = ow * stride_ - padding_;

                float maxVal = -FLT_MAX;

                // Load window and compute max

                for (int32_t kd = 0; kd < kernel_size_; kd++) {

                    int32_t id = in_d_start + kd;

                    if (id < 0 || id >= D_) continue;

                    for (int32_t kh = 0; kh < kernel_size_; kh++) {

                        int32_t ih = in_h_start + kh;

                        if (ih < 0 || ih >= H_) continue;

                        for (int32_t kw = 0; kw < kernel_size_; kw++) {

                            int32_t iw = in_w_start + kw;

                            if (iw < 0 || iw >= W_) continue;

                            uint32_t srcIdx = n * C_ * D_ * H_ * W_ +

                                              c * D_ * H_ * W_ +

                                              id * H_ * W_ +

                                              ih * W_ +

                                              iw;

                            float val = xGm_.GetValue(srcIdx);

                            if (val > maxVal) maxVal = val;

                        }

                    }

                }

                // Write output

                uint32_t dstIdx = n * C_ * D_out_ * H_out_ * W_out_ +

                                  c * D_out_ * H_out_ * W_out_ +

                                  od * H_out_ * W_out_ +

                                  oh * W_out_ +

                                  ow;

                yGm_.SetValue(dstIdx, maxVal);

            }

            // next tile

            base += strideElems;

            if (base >= totalOutputElements_) break;

        }

    }
