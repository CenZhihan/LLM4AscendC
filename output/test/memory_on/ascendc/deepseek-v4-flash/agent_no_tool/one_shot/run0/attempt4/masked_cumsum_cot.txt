#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t tileLen,
                                uint32_t blockDim)
    {
        rows_ = rows;
        cols_ = cols;
        tileLen_ = tileLen;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, rows_ * cols_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, rows_ * cols_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = blockDim_;

        // Each block processes rows in stride
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        float carry = 0.0f;
        for (uint32_t tileStart = 0; tileStart < cols_; tileStart += tileLen_) {
            uint32_t len = (tileStart + tileLen_ <= cols_) ? tileLen_ : (cols_ - tileStart);

            // Copy x and mask for current tile
            uint32_t offset = rowIdx * cols_ + tileStart;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            AscendC::DataCopy(maskLocal, maskGm_[offset], len);
            inQueueX_.EnQue(xLocal);
            inQueueMask_.EnQue(maskLocal);

            // Compute y = cumsum(x * mask) + carry
            AscendC::LocalTensor<float> xt = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> mt = inQueueMask_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // y = x * mask
            AscendC::Mul(yLocal, xt, mt, len);
            // inclusive prefix sum on y
            AscendC::CumSum(yLocal, yLocal, len);  // inclusive by default
            // add carry from previous tile
            AscendC::Adds(yLocal, yLocal, carry, len);

            // update carry from last element
            carry = yLocal(len - 1);

            // Enqueue result
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xt);
            inQueueMask_.FreeTensor(mt);

            // Copy out
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yLocal, len);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.rows, tiling_data.cols, tiling_data.tileLen, tiling_data.blockDim);
    op.Process();
}
