host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>
#include <cstdint>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape *inputShape = context->GetInputShape(0);
    uint32_t ndim = inputShape->GetOriginShape().GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t reduceDim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (reduceDim < 0) reduceDim += ndim;
    uint32_t redIdx = static_cast<uint32_t>(reduceDim);

    uint32_t totalInputLength = 1;
    uint32_t outputLength = 1;
    uint32_t reduceLength = 1;
    uint32_t strideAfter = 1;
    for (uint32_t i = 0; i < ndim; ++i) {
        uint32_t dimSize = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(i));
        totalInputLength *= dimSize;
        if (i < redIdx) {
            outputLength *= dimSize;
        } else if (i == redIdx) {
            reduceLength = dimSize;
        } else {
            strideAfter *= dimSize;
        }
    }
    if (reduceLength == 0) {
        return ge::GRAPH_FAILED;
    }
    outputLength *= strideAfter;

    tiling.set_totalInputLength(totalInputLength);
    tiling.set_outputLength(outputLength);
    tiling.set_reduceLength(reduceLength);
    tiling.set_strideAfter(strideAfter);
    tiling.set_invReduceLength(1.0f / static_cast<float>(reduceLength));

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (outputLength < tileLen) {
        uint32_t aligned = AlignDown(outputLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : outputLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(outputLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *inputShape = context->GetInputShape(0);
    uint32_t ndim = inputShape->GetOriginShape().GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t reduceDim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (reduceDim < 0) reduceDim += ndim;
    uint32_t redIdx = static_cast<uint32_t>(reduceDim);

    std::vector<int64_t> outDims;
    for (uint32_t i = 0; i < ndim; ++i) {
        if (i != redIdx) {
            outDims.push_back(inputShape->GetOriginShape().GetDim(i));
        }
    }

    gert::Shape *outputShape = context->GetOutputShape(0);
    outputShape->SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) {
        outputShape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputLength,
                                uint32_t outputLength,
                                uint32_t reduceLength,
                                uint32_t strideAfter,
                                float invReduceLength)
    {
        totalInputLen_ = totalInputLength;
        outputLen_ = outputLength;
        reduceLen_ = reduceLength;
        strideAfter_ = strideAfter;
        invReduceLen_ = invReduceLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputLen_);
    }

    __aicore__ inline void Process()
    {
        if (outputLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t perBlock = (outputLen_ + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * perBlock;
        uint32_t end = (blockIdx + 1) * perBlock;
        if (start >= outputLen_) return;
        if (end > outputLen_) end = outputLen_;

        for (uint32_t i = start; i < end; i++)
        {
            uint32_t beforeGroup = i / strideAfter_;
            uint32_t afterOffset = i % strideAfter_;
            uint32_t baseIdx = beforeGroup * (reduceLen_ * strideAfter_) + afterOffset;

            float sum = 0.0f;
            for (uint32_t k = 0; k < reduceLen_; k++)
            {
                uint32_t idx = baseIdx + k * strideAfter_;
                sum += xGm_.GetValue(idx);
            }
            yGm_.SetValue(i, sum * invReduceLen_);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputLen_ {0};
    uint32_t outputLen_ {0};
    uint32_t reduceLen_ {0};
    uint32_t strideAfter_ {0};
    float invReduceLen_ {0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalInputLength,
            tiling_data.outputLength,
            tiling_data.reduceLength,
            tiling_data.strideAfter,
            tiling_data.invReduceLength);
    op.Process();
}
