    return (x / a) * a;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const uint32_t rowCostFloats = 5 * d;
    const uint32_t maxRows = UB_BUDGET_BYTES / (rowCostFloats * sizeof(float));

    if (blockDim > 24) blockDim = 24;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t VEC_SIZE = 128;  // enough for typical d <= 128

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t totalRows, uint32_t dim,
                                uint32_t tileBatch, float margin)
    {
        totalRows_ = totalRows;
        dim_ = dim;
        tileBatch_ = tileBatch;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalRows_ * dim_);
        posGm_.SetGlobalBuffer((__gm__ float*)positive, totalRows_ * dim_);
        negGm_.SetGlobalBuffer((__gm__ float*)negative, totalRows_ * dim_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inQueueAnchor_, BUFFER_NUM, tileBatch_ * dim_ * sizeof(float));
        pipe_.InitBuffer(inQueuePos_, BUFFER_NUM, tileBatch_ * dim_ * sizeof(float));
        pipe_.InitBuffer(inQueueNeg_, BUFFER_NUM, tileBatch_ * dim_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileBatch_ * dim_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileBatch_ * dim_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || dim_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t baseRow = blockIdx * tileBatch_;
        if (baseRow >= totalRows_) return;

        float localSum = 0.0f;

        while (baseRow < totalRows_) {
            uint32_t curBatch = totalRows_ - baseRow;
            if (curBatch > tileBatch_) curBatch = tileBatch_;

            // Load tile data
            CopyIn(baseRow, curBatch);
            localSum += Compute(curBatch);
            CopyOut(baseRow, curBatch);

            baseRow += blockNum * tileBatch_;
        }

        // Atomic add to global loss scalar
        AscendC::AtomicAdd<float>(lossGm_, localSum);
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowOffset, uint32_t batchSize)
    {
        uint32_t offset = rowOffset * dim_;
        uint32_t len = batchSize * dim_;

        {
            auto xLocal = inQueueAnchor_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, anchorGm_[offset], len);
            inQueueAnchor_.EnQue(xLocal);
        }
        {
            auto xLocal = inQueuePos_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, posGm_[offset], len);
            inQueuePos_.EnQue(xLocal);
        }
        {
            auto xLocal = inQueueNeg_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, negGm_[offset], len);
            inQueueNeg_.EnQue(xLocal);
        }
    }

    __aicore__ inline float Compute(uint32_t batchSize)
    {
        auto anchorLocal = inQueueAnchor_.DeQue<float>();
        auto posLocal = inQueuePos_.DeQue<float>();
        auto negLocal = inQueueNeg_.DeQue<float>();

        auto diffAp = tmpQueue1_.AllocTensor<float>();
        auto diffAn = tmpQueue2_.AllocTensor<float>();

        AscendC::LocalTensor<float> sumAp, sumAn;

        // Process each row in the batch
        float rowSum = 0.0f;
        for (uint32_t r = 0; r < batchSize; ++r) {
            uint32_t rowOff = r * dim_;

            // diff_ap = anchor - positive
            AscendC::VSub(diffAp, anchorLocal[rowOff], posLocal[rowOff], dim_);
            // sq_ap = diff_ap * diff_ap
            AscendC::VMul(diffAp, diffAp, diffAp, dim_);
            // sum_ap = sum(sq_ap)
            sumAp.SetValue(0.0f);
            for (uint32_t i = 0; i < dim_; ++i) sumAp.SetValue(sumAp.GetValue() + diffAp.GetValue(i));

            // diff_an = anchor - negative
            AscendC::VSub(diffAn, anchorLocal[rowOff], negLocal[rowOff], dim_);
            AscendC::VMul(diffAn, diffAn, diffAn, dim_);
            sumAn.SetValue(0.0f);
            for (uint32_t i = 0; i < dim_; ++i) sumAn.SetValue(sumAn.GetValue() + diffAn.GetValue(i));

            float dAp = AscendC::Sqrt(sumAp.GetValue());
            float dAn = AscendC::Sqrt(sumAn.GetValue());

            float lossVal = dAp - dAn + margin_;
            if (lossVal < 0.0f) lossVal = 0.0f;
            rowSum += lossVal;
        }

        tmpQueue1_.FreeTensor(diffAp);
        tmpQueue2_.FreeTensor(diffAn);
        inQueueAnchor_.FreeTensor(anchorLocal);
        inQueuePos_.FreeTensor(posLocal);
        inQueueNeg_.FreeTensor(negLocal);

        return rowSum;
    }

    __aicore__ inline void CopyOut(uint32_t rowOffset, uint32_t batchSize)
    {
        // no output per tile, only atomic accumulation
        (void)rowOffset;
        (void)batchSize;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePos_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNeg_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> posGm_;
    AscendC::GlobalTensor<float> negGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalRows_ {0};
    uint32_t dim_ {0};
    uint32_t tileBatch_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                  GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalRows, tiling_data.dim, tiling_data.tileBatch, tiling_data.margin);
    op.Process();
}
"""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor triplet_margin_loss_npu(const at::Tensor& anchor,
                                    const at::Tensor& positive,
                                    const at::Tensor& negative,
                                    double margin) {
    // Create output scalar initialized to 0
    at::Tensor loss = at::zeros({}, anchor.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnTripletMarginLossCustom, anchor, positive, negative, margin, loss);

    // The kernel outputs sum, divide by batch size to get mean
    int64_t n = anchor.size(0);
    loss = loss / static_cast<float>(n);
    return loss;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("triplet_margin_loss_custom", &triplet_margin_loss_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("triplet_margin_loss_custom", &triplet_margin_loss_npu, "Triplet Margin Loss (TripletMarginLossCustom on NPU)");

import torch
import torch_npu
import custom_ops_lib

_FN = 'triplet_margin_loss_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, *inputs):
        anchor, positive, negative, margin_tensor = inputs
        # convert margin to float (scalar)
        margin = float(margin_tensor.reshape(-1)[0].item())
        fn = getattr(custom_ops_lib, _FN)
        # The custom kernel expects a scalar margin; the python binding also expects double.
        # We pass margin as a float; the operator attribute receives it.
        return fn(anchor, positive, negative, margin)
