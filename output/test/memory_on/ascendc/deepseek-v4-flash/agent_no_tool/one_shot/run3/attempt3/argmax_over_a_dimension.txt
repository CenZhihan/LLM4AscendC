host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    // GetInputShape returns const gert::StorageShape*
    const gert::StorageShape *input_storage = context->GetInputShape(0);
    auto input_dims = input_storage->GetOriginShape().GetDims(); // vector<int64_t>
    size_t ndim = input_dims.size();
    if (ndim == 0) {
        return ge::GRAPH_FAILED;
    }

    uint32_t total_elems = 1;
    for (size_t i = 0; i < ndim; ++i) total_elems *= static_cast<uint32_t>(input_dims[i]);

    uint32_t reduce_size = static_cast<uint32_t>(input_dims[ndim - 1]);
    uint32_t total_rows = (reduce_size > 0) ? total_elems / reduce_size : 0;

    uint32_t block_dim = 8;
    if (total_rows > 24) block_dim = 24;
    else if (total_rows > 8) block_dim = total_rows;

    context->SetBlockDim(block_dim);

    tiling.set_totalRows(total_rows);
    tiling.set_reduceSize(reduce_size);
    tiling.set_blockDim(block_dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *input_storage = context->GetInputShape(0);
    auto dims = input_storage->GetOriginShape().GetDims();
    if (dims.empty()) {
        return GRAPH_FAILED;
    }
    // Output shape: all dims except last
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(dims.size() - 1);
    for (size_t i = 0; i < dims.size() - 1; ++i) {
        y_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {
class ArgmaxOverADimensionCustom : public OpDef {

    explicit ArgmaxOverADimensionCustom(const char* name) : OpDef(name)

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t reduceSize)
    {
        totalRows_ = totalRows;
        reduceSize_ = reduceSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * reduceSize_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalRows_);
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || reduceSize_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            uint32_t base = row * reduceSize_;
            float maxVal = xGm_[base];
            int64_t maxIdx = 0;

            for (uint32_t i = 1; i < reduceSize_; ++i) {
                float curVal = xGm_[base + i];
                if (curVal > maxVal) {
                    maxVal = curVal;
                    maxIdx = static_cast<int64_t>(i);
                }
            }

            yGm_[row] = maxIdx;
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<int64_t> yGm_;

    uint32_t totalRows_ {0};
    uint32_t reduceSize_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.reduceSize);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor permute_to_last_dim(const at::Tensor& self, int64_t dim) {
    int64_t ndim = self.dim();
    dim = (dim + ndim) % ndim;
    std::vector<int64_t> perm;
    for (int64_t i = 0; i < ndim; ++i) {
        if (i != dim) perm.push_back(i);
    }
    perm.push_back(dim);
    return self.permute(perm);
}

static at::Tensor permute_back(const at::Tensor& self, int64_t dim, int64_t ndim) {
    std::vector<int64_t> perm;
    for (int64_t i = 0; i < ndim - 1; ++i) {
        if (i < dim) perm.push_back(i);
        else perm.push_back(i + 1);
    }
    perm.insert(perm.begin() + dim, ndim - 1);
    return self.permute(perm);
}

at::Tensor argmax_impl_npu(const at::Tensor& self, int64_t dim) {
    at::Tensor permuted = permute_to_last_dim(self, dim);
    std::vector<int64_t> out_shape;
    int64_t ndim = permuted.dim();
    for (int64_t i = 0; i < ndim - 1; ++i) {
        out_shape.push_back(permuted.size(i));
    }
    at::Tensor result = at::empty(out_shape, permuted.options().dtype(at::kLong));
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, permuted, result);
    int64_t orig_ndim = self.dim();
    return permute_back(result, dim, orig_ndim);
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_impl_npu, "Argmax over a dimension (AscendC custom)");
}
