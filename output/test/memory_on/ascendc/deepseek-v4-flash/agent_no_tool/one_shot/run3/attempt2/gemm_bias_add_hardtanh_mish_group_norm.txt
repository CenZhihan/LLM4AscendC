#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
                                GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures,
                                uint32_t outFeatures, uint32_t tileOutFeatures,
                                uint32_t numGroups)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        tileOutF_ = tileOutFeatures;
        numGroups_ = numGroups;
        groupSize_ = outFeatures / numGroups;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inFeatures);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, inFeatures * outFeatures);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * outFeatures);

        // Buffers: localA (tile of input row), localB (tile of weight column),
        // localC (tile of output), localTemp for misc ops, localGN for group norm buffers.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileOutF_ * sizeof(float));
        pipe_.InitBuffer(inQueueWeight_, BUFFER_NUM, inFeatures_ * tileOutF_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOutF_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileOutF_ * sizeof(float));
        // Group norm reduction buffers: for mean/variance per group per batch.
        // We'll compute group norm after all tiles are done; here we need to accumulate.
        // For simplicity, we process one row at a time, do full outFeatures per row? Not efficient with tiling.
        // Instead, we tile along outFeatures, compute partial GN stats per group? That's complex.
        // For a simplified but working kernel, we assume tileOutF_ = outFeatures (no tiling along output dim).
        // However, to follow tiling logic, we use a single-tile approach per block.
        // Given constraints, we set tileOutF_ = outFeatures in tiling to avoid cross-tile GN.
        // So we process entire rows in one tile.
        // So buffers are sized for full outFeatures.
    }

    __aicore__ inline void Process()
    {
        if (outFeatures_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Grid stride over rows
        for (uint32_t b = blockIdx; b < batchSize_; b += blockNum) {
            ProcessRow(b);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        // Load input row from x (1 x inFeatures)
        AscendC::LocalTensor<float> xRow = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(xRow, xGm_[rowIdx * inFeatures_], inFeatures_);

        // Allocate output tile
        AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();

        // MatMul: y = x * weight^T + lin_bias (using MatMul)
        // weight is [inFeatures, outFeatures] in row-major? Actually weight is [outFeatures, inFeatures] from PyTorch.
        // Our weight GM layout: inFeatures_ x outFeatures_ (column-major for matmul). But we need to be careful.
        // For simplicity, we assume weight is transposed. Let's use MatMul API.
        // MatMul expects A (MxK), B (KxN). Here: M=1, K=inFeatures, N=outFeatures.
        // xRow: 1 x K, weight: K x N (if weight is stored as KxN). We'll store weight as [inFeatures, outFeatures] (KxN).
        // So we need to ensure the weight layout matches.
        // We'll do: MatMul(yTile, xRow, weightGm_, 1, inFeatures_, outFeatures_, false, false);
        // But MatMul on AscendC uses LocalTensor for output, need to manage memory.
        // We'll allocate a temporary for result.
        AscendC::LocalTensor<float> matmulResult = tmpQueue_.AllocTensor<float>(); // reuse?
        AscendC::MatMul(matmulResult, xRow, weightGm_, 1, inFeatures_, outFeatures_, false, false);
        // Add lin_bias
        AscendC::Add(matmulResult, matmulResult, linBiasGm_, outFeatures_);
        // Add bias (element-wise)
        AscendC::Add(matmulResult, matmulResult, biasGm_, outFeatures_);
        // Hardtanh: clamp to [-1, 1]
        AscendC::Clamp(matmulResult, matmulResult, -1.0f, 1.0f, outFeatures_);
        // Mish: x * tanh(softplus(x)). We'll implement softplus = ln(1+exp(x)), then tanh, then mul.
        // Use local temp.
        AscendC::LocalTensor<float> temp = tmpQueue_.AllocTensor<float>();
        AscendC::Exp(temp, matmulResult, outFeatures_); // exp(x)
        AscendC::Adds(temp, temp, 1.0f, outFeatures_); // 1+exp(x)
        AscendC::Log(temp, temp, outFeatures_); // ln(1+exp(x))
        AscendC::Tanh(temp, temp, outFeatures_); // tanh(softplus)
        AscendC::Mul(matmulResult, matmulResult, temp, outFeatures_); // x * tanh(softplus)
        tmpQueue_.FreeTensor(temp);

        // Group norm: compute per group mean and variance
        // We have matmulResult of size [outFeatures]. numGroups divides outFeatures.
        // We'll compute mean and variance per group using reduction.
        // We'll allocate a local buffer for group sums.
        AscendC::LocalTensor<float> groupMean = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> groupVar = tmpQueue_.AllocTensor<float>();
        // ReduceSum over each group
        for (uint32_t g = 0; g < numGroups_; ++g) {
            uint32_t start = g * groupSize_;
            AscendC::LocalTensor<float> groupSlice = matmulResult[start];
            // Compute sum in groupVar
            AscendC::ReduceSum(groupVar[g], groupSlice, groupSize_, 0); // reduce along dimension
        }
        // Compute mean = sum / groupSize_
        for (uint32_t g = 0; g < numGroups_; ++g) {
            groupMean[g] = groupVar[g] / static_cast<float>(groupSize_);
        }
        // Compute variance: sum((x - mean)^2) / groupSize_
        for (uint32_t g = 0; g < numGroups_; ++g) {
            AscendC::LocalTensor<float> groupSlice = matmulResult[g * groupSize_];
            AscendC::Subs(groupVar[g], groupSlice[0], groupMean[g], groupSize_);
            AscendC::Mul(groupVar[g], groupVar[g], groupVar[g], groupSize_);
            AscendC::ReduceSum(groupVar[g], groupVar[g], groupSize_, 0);
            groupVar[g] /= static_cast<float>(groupSize_);
            groupVar[g] += 1e-5f; // epsilon
        }
        // Normalize: (x - mean) / sqrt(var) * gamma + beta
        for (uint32_t g = 0; g < numGroups_; ++g) {
            AscendC::LocalTensor<float> groupSlice = matmulResult[g * groupSize_];
            AscendC::Subs(groupSlice, groupSlice, groupMean[g], groupSize_);
            AscendC::Sqrt(groupVar[g], groupVar[g]); // sqrt(var)
            AscendC::Mul(groupVar[g], groupVar[g], 1.0f / groupVar[g]); // reciprocal? Actually we want 1/sqrt(var)
            // AscendC provides Div, but we can do reciprocal via Rsqrt? Use Rsqrt.
            AscendC::Rsqrt(groupVar[g], groupVar[g]);
            AscendC::Mul(groupSlice, groupSlice, groupVar[g], groupSize_);
            // Multiply by gamma and add beta per element
            AscendC::Mul(groupSlice, groupSlice, gnGammaGm_[g * groupSize_], groupSize_);
            AscendC::Add(groupSlice, groupSlice, gnBetaGm_[g * groupSize_], groupSize_);
        }

        // Copy result to global memory
        AscendC::DataCopy(yGm_[rowIdx * outFeatures_], matmulResult, outFeatures_);

        // Free all local tensors
        tmpQueue_.FreeTensor(xRow);
        tmpQueue_.FreeTensor(matmulResult);
        tmpQueue_.FreeTensor(groupMean);
        tmpQueue_.FreeTensor(groupVar);
        outQueueY_.FreeTensor(yTile);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWeight_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_;
    uint32_t inFeatures_;
    uint32_t outFeatures_;
    uint32_t tileOutF_;
    uint32_t numGroups_;
    uint32_t groupSize_;
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
    GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.batchSize, tiling_data.inFeatures,
            tiling_data.outFeatures, tiling_data.tileOutFeatures,
            tiling_data.numGroups);
    op.Process();
}
