#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t cols)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        cols_ = cols;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || cols_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t rows = totalLen_ / cols_;

        // grid-stride loop over rows
        for (uint32_t rowIdx = blockIdx; rowIdx < rows; rowIdx += blockNum) {
            ProcessRow(rowIdx);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        const uint32_t rowBase = rowIdx * cols_;
        const uint32_t rowLen = cols_;

        // ---------- Pass1: compute max of the row ----------
        float rowMax = -1.0e38f;
        for (uint32_t offset = 0; offset < rowLen; offset += tileLen_) {
            uint32_t curLen = (rowLen - offset) > tileLen_ ? tileLen_ : (rowLen - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowBase + offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue1_.AllocTensor<float>();
            // Use AllReduce pattern to reduce entire tile to scalar
            AscendC::ReduceMax<AscendC::ReduceMode::AllReduce>(tmp, xLocal, curLen);
            float tileMax = tmp(0);
            tmpQueue1_.FreeTensor(tmp);

            if (tileMax > rowMax) rowMax = tileMax;

            inQueueX_.FreeTensor(xLocal);
        }

        // ---------- Pass2: compute sum of exp(x - rowMax) ----------
        float rowSum = 0.0f;
        for (uint32_t offset = 0; offset < rowLen; offset += tileLen_) {
            uint32_t curLen = (rowLen - offset) > tileLen_ ? tileLen_ : (rowLen - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowBase + offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            // sub = x - rowMax  (overwrite xLocal)
            AscendC::Adds(xLocal, xLocal, -rowMax, curLen);
            // exp = exp(sub) into tmpQueue1_
            AscendC::LocalTensor<float> expLocal = tmpQueue1_.AllocTensor<float>();
            AscendC::Exp(expLocal, xLocal, curLen);
            // sum over tile using ReduceSum, result into tmpQueue2_
            AscendC::LocalTensor<float> sumLocal = tmpQueue2_.AllocTensor<float>();
            AscendC::ReduceSum<AscendC::ReduceMode::AllReduce>(sumLocal, expLocal, curLen);
            float tileSum = sumLocal(0);
            rowSum += tileSum;

            inQueueX_.FreeTensor(xLocal);
            tmpQueue1_.FreeTensor(expLocal);
            tmpQueue2_.FreeTensor(sumLocal);
        }

        // ---------- Compute log(rowSum) ----------
        AscendC::LocalTensor<float> logTensor = tmpQueue1_.AllocTensor<float>();
        logTensor(0) = rowSum;
        AscendC::Log(logTensor, logTensor, 1);
        float logRowSum = logTensor(0);
        tmpQueue1_.FreeTensor(logTensor);

        float offsetVal = rowMax + logRowSum;

        // ---------- Pass3: compute output = x - offsetVal ----------
        for (uint32_t offset = 0; offset < rowLen; offset += tileLen_) {
            uint32_t curLen = (rowLen - offset) > tileLen_ ? tileLen_ : (rowLen - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowBase + offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Adds(yLocal, xLocal, -offsetVal, curLen);

            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        // Copy out all tiles
        for (uint32_t offset = 0; offset < rowLen; offset += tileLen_) {
            uint32_t curLen = (rowLen - offset) > tileLen_ ? tileLen_ : (rowLen - offset);
            AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[rowBase + offset], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t cols_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.cols);
    op.Process();
}
