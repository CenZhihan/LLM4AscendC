#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;
static constexpr uint32_t FLOAT_SIZE = 4;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::StorageShape *a_storage_shape = context->GetInputShape(0);
    const gert::Shape &a_shape = a_storage_shape->GetOriginShape();
    const gert::StorageShape *b_storage_shape = context->GetInputShape(1);
    const gert::Shape &b_shape = b_storage_shape->GetOriginShape();
    uint32_t M = static_cast<uint32_t>(a_shape.GetDim(0));
    uint32_t K = static_cast<uint32_t>(a_shape.GetDim(1));
    uint32_t N = static_cast<uint32_t>(b_shape.GetDim(1));

    uint32_t maxFloats = UB_BUDGET_BYTES / FLOAT_SIZE;
    if (maxFloats > HARD_CAP_ELEMS) maxFloats = HARD_CAP_ELEMS;
    maxFloats = AlignDown(maxFloats, ALIGN_ELEMS);
    if (maxFloats < ALIGN_ELEMS) maxFloats = ALIGN_ELEMS;

    uint32_t tileM, tileN;
    if (K <= 128) {
        uint32_t maxTileN = 32;
        if (maxTileN > N) maxTileN = AlignDown(N, ALIGN_ELEMS);
        if (maxTileN == 0) maxTileN = ALIGN_ELEMS;
        uint32_t numerator = maxFloats - K * maxTileN;
        tileM = numerator / (K + maxTileN);
        tileM = AlignDown(tileM, ALIGN_ELEMS);
        if (tileM == 0) tileM = ALIGN_ELEMS;
        tileN = maxTileN;
    } else {
        uint32_t approx = static_cast<uint32_t>(sqrt(maxFloats / 3.0f));
        approx = AlignDown(approx, ALIGN_ELEMS);
        if (approx < ALIGN_ELEMS) approx = ALIGN_ELEMS;
        tileM = approx;
        tileN = approx;
        while (tileM > ALIGN_ELEMS && (tileM * K + K * tileN + tileM * tileN) > maxFloats) {
            tileM -= ALIGN_ELEMS;
        }
        while (tileN > ALIGN_ELEMS && (tileM * K + K * tileN + tileM * tileN) > maxFloats) {
            tileN -= ALIGN_ELEMS;
        }
        if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;
        if (tileN < ALIGN_ELEMS) tileN = ALIGN_ELEMS;
    }

    if (tileM > M) tileM = AlignDown(M, ALIGN_ELEMS);
    if (tileM == 0) tileM = M;
    if (tileN > N) tileN = AlignDown(N, ALIGN_ELEMS);
    if (tileN == 0) tileN = N;

    uint32_t numTilesM = CeilDiv(M, tileM);
    uint32_t numTilesN = CeilDiv(N, tileN);
    uint32_t totalTiles = numTilesM * numTilesN;
    uint32_t blockDim = totalTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileN) {
        M_ = M; K_ = K; N_ = N;
        tileM_ = tileM; tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, M * K);
        bGm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        cGm_.SetGlobalBuffer((__gm__ float*)C, M * N);

        uint32_t aSize = tileM_ * K_;
        uint32_t bSize = K_ * tileN_;
        uint32_t cSize = tileM_ * tileN_;
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, aSize * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, bSize * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, cSize * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t numTilesM = (M_ + tileM_ - 1) / tileM_;
        uint32_t numTilesN = (N_ + tileN_ - 1) / tileN_;
        uint32_t totalTiles = numTilesM * numTilesN;

        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            uint32_t tileRow = t / numTilesN;
            uint32_t tileCol = t % numTilesN;

            uint32_t rowStart = tileRow * tileM_;
            uint32_t rowEnd = rowStart + tileM_;
            if (rowEnd > M_) rowEnd = M_;
            uint32_t effectiveM = rowEnd - rowStart;

            uint32_t colStart = tileCol * tileN_;
            uint32_t colEnd = colStart + tileN_;
            if (colEnd > N_) colEnd = N_;
            uint32_t effectiveN = colEnd - colStart;

            // Load A tile (all K columns)
            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            for (uint32_t i = 0; i < effectiveM; i++) {
                uint32_t gmOffset = (rowStart + i) * K_;
                uint32_t localOffset = i * K_;
                AscendC::DataCopy(aLocal[localOffset], aGm_[gmOffset], K_);
            }

            // Load B tile (all K rows)
            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
            for (uint32_t k = 0; k < K_; k++) {
                uint32_t gmOffset = k * N_ + colStart;
                uint32_t localOffset = k * effectiveN;
                AscendC::DataCopy(bLocal[localOffset], bGm_[gmOffset], effectiveN);
            }

            // Compute C tile
            AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
            for (uint32_t i = 0; i < effectiveM; i++) {
                for (uint32_t j = 0; j < effectiveN; j++) {
                    float sum = 0.0f;
                    for (uint32_t k = 0; k < K_; k++) {
                        float aVal = aLocal.GetValue(i * K_ + k);
                        float bVal = bLocal.GetValue(k * effectiveN + j);
                        sum += aVal * bVal;
                    }
                    cLocal.SetValue(i * effectiveN + j, sum);
                }
            }

            // Store C tile
            for (uint32_t i = 0; i < effectiveM; i++) {
                uint32_t gmOffset = (rowStart + i) * N_ + colStart;
                uint32_t localOffset = i * effectiveN;
                AscendC::DataCopy(cGm_[gmOffset], cLocal[localOffset], effectiveN);
            }

            inQueueA_.FreeTensor(aLocal);
            inQueueB_.FreeTensor(bLocal);
            outQueueC_.FreeTensor(cLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
