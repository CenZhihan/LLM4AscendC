host_operator_src="""

#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: 3 tiles (input, sq, output) aside from small scalars
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const uint32_t totalLength = static_cast<uint32_t>(xShape->GetOriginShape().GetShapeSize());

    // Compute batchSize and innerSize from the shape
    // Assume normalized_shape is all dimensions except the first one.
    uint32_t batchSize = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t innerSize = totalLength / batchSize;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilon = attrs->GetAttrPointer<float>(0);

    // Compute tile length from UB budget (3 tiles * tileLen * sizeof(float))
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float)); // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (innerSize < tileLen) {
        tileLen = AlignDown(innerSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = innerSize;
    }

    // Block count: one block per batch element, capped by hardware limit
    uint32_t blockDim = batchSize;
    if (blockDim > 24) blockDim = 24; // typical limit for ascend910b
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_batchSize(batchSize);
    tiling.set_innerSize(innerSize);
    tiling.set_innerSizeFloat(static_cast<float>(innerSize));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_epsilon(*epsilon);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LayerNormCustom : public OpDef {
public:
    explicit LayerNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("epsilon").AttrType(OPTIONAL).Float(1e-5);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LayerNormCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t batchSize,
                                uint32_t innerSize, float innerSizeFloat,
                                uint32_t tileLength,
                                float epsilon)
    {
        totalLen_ = totalLength;
        batchSize_ = batchSize;
        innerSize_ = innerSize;
        innerSizeFloat_ = innerSizeFloat;
        tileLen_ = tileLength;
        epsilon_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(sqQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerSize_ == 0 || tileLen_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= batchSize_) return; // extra cores idle

        // Offset for this batch element
        const uint32_t baseOffset = blockIdx * innerSize_;
        const uint32_t curInnerSize = innerSize_;
        const uint32_t tileLen = tileLen_;

        // ----- First pass: compute sum and sum_sq for this batch element -----
        float sum = 0.0f;
        float sumSq = 0.0f;

        uint32_t remaining = curInnerSize;
        uint32_t offset = baseOffset;
        while (remaining > 0) {
            uint32_t curTile = (remaining < tileLen) ? remaining : tileLen;
            // Load tile directly
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curTile);
            // Compute squares
            AscendC::LocalTensor<float> sqLocal = sqQueue_.AllocTensor<float>();
            AscendC::Mul(sqLocal, xLocal, xLocal, curTile);

            // Reduce sum and sumSq
            float tileSum = 0.0f;
            float tileSqSum = 0.0f;
            for (uint32_t i = 0; i < curTile; ++i) {
                tileSum += xLocal(i);
                tileSqSum += sqLocal(i);
            }
            sum += tileSum;
            sumSq += tileSqSum;

            inQueueX_.FreeTensor(xLocal);
            sqQueue_.FreeTensor(sqLocal);
            offset += curTile;
            remaining -= curTile;
        }

        // ----- Compute mean, var, inv_std -----
        float mean = sum / innerSizeFloat_;
        float variance = sumSq / innerSizeFloat_ - mean * mean;
        float std = sqrt(variance + epsilon_);
        float invStd = 1.0f / std;

        // ----- Second pass: normalize this batch element -----
        remaining = curInnerSize;
        offset = baseOffset;
        while (remaining > 0) {
            uint32_t curTile = (remaining < tileLen) ? remaining : tileLen;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curTile);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // y = (x - mean) * invStd  = x * invStd - mean * invStd
            AscendC::Muls(yLocal, xLocal, invStd, curTile);
            AscendC::Adds(yLocal, yLocal, -mean * invStd, curTile);

            // Copy out
            AscendC::DataCopy(yGm_[offset], yLocal, curTile);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            offset += curTile;
            remaining -= curTile;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        // Placeholder; not used in this simplified pipeline
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> sqQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t innerSize_ {0};
    float innerSizeFloat_{0.0f};
    uint32_t tileLen_ {0};
    float epsilon_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.batchSize,
            tiling_data.innerSize,
            tiling_data.innerSizeFloat,
            tiling_data.tileLength,
            tiling_data.epsilon);
    op.Process();
}
