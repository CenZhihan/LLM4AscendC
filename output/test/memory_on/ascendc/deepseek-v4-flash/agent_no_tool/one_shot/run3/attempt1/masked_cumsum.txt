host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: we need three buffers per tile (x, mask, y) → 12*tileLen bytes
// Using conservative constants from repair memory #3
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;

    // Use StorageShape* as per repair memory #1
    auto *inShape  = context->GetInputShape(0);
    auto originShape = inShape->GetOriginShape();
    uint32_t ndim = static_cast<uint32_t>(originShape.GetDimNum());
    std::vector<int64_t> dimVec;
    for (uint32_t i = 0; i < ndim; ++i) {
        dimVec.push_back(originShape.GetDim(i));
    }

    uint32_t totalLength = 1;
    for (auto d : dimVec) {
        totalLength *= static_cast<uint32_t>(d);
    }

    // Assume cumulative sum is along the last dimension
    uint32_t cols = (ndim > 0) ? static_cast<uint32_t>(dimVec[ndim - 1]) : 1u;
    uint32_t rows = (cols > 0) ? totalLength / cols : 1u;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dimVal = (dimPtr != nullptr) ? *dimPtr : -1;

    // Compute tile length using HB budget formula from repair memory #3
    uint32_t maxTile = UB_BUDGET_BYTES / 28;   // conservative divisor
    if (maxTile > HARD_CAP_ELEMS) {
        maxTile = HARD_CAP_ELEMS;
    }
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) {
        maxTile = ALIGN_ELEMS;
    }

    uint32_t tileLen = (cols < maxTile) ? cols : maxTile;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) {
        tileLen = ALIGN_ELEMS;  // at least one element
    }

    // Block dimension: at least 1, at most 24, but we can limit to number of rows
    uint32_t blockDim = (rows < 24u) ? rows : 24u;
    if (blockDim == 0) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dimVal);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MaskedCumsumCustom : public OpDef {
public:
    explicit MaskedCumsumCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("mask")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(-1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MaskedCumsumCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t rows,
                                uint32_t cols, uint32_t tileLength,
                                int32_t dim)
    {
        totalLen_ = totalLength;
        rows_ = rows;
        cols_ = cols;
        tileLen_ = tileLength;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // not using tmpQueue_ but we declare it to satisfy the pattern
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rows_ == 0 || cols_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        for (uint32_t r = blockIdx; r < rows_; r += blockNum) {
            // Process one row
            float rowCumSum = 0.0f;
            uint32_t rowOffset = r * cols_;

            // Iterate over column tiles
            for (uint32_t tileStart = 0; tileStart < cols_; tileStart += tileLen_) {
                uint32_t curTileLen = tileLen_;
                if (tileStart + curTileLen > cols_) {
                    curTileLen = cols_ - tileStart;
                }

                // Load x and mask segments
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curTileLen);
                AscendC::DataCopy(maskLocal, maskGm_[rowOffset + tileStart], curTileLen);

                // Compute cumulative sum for this tile, updating rowCumSum
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                ComputeTile(xLocal, maskLocal, yLocal, curTileLen, rowCumSum);

                // Store output
                AscendC::DataCopy(yGm_[rowOffset + tileStart], yLocal, curTileLen);

                // Free local tensors
                inQueueX_.FreeTensor(xLocal);
                inQueueMask_.FreeTensor(maskLocal);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline void ComputeTile(AscendC::LocalTensor<float>& xLocal,
                                       AscendC::LocalTensor<float>& maskLocal,
                                       AscendC::LocalTensor<float>& yLocal,
                                       uint32_t len,
                                       float& carry)
    {
        // Loop over elements using GetValue/SetValue (repair memory #2)
        for (uint32_t i = 0; i < len; ++i) {
            float xVal = xLocal.GetValue(i);
            float maskVal = maskLocal.GetValue(i);
            float prod = xVal * maskVal;
            carry += prod;
            yLocal.SetValue(i, carry);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ {-1};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                                           GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.totalLength, tiling_data.rows,
            tiling_data.cols, tiling_data.tileLength, tiling_data.dim);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor masked_cumsum_impl_npu(const at::Tensor& self, const at::Tensor& mask,
                                  int dim) {
    at::Tensor result = at::empty_like(self);
    // Convert mask to float if it is boolean (PyTorch often uses bool,
    // but our op expects float; we convert here)
    at::Tensor mask_float = mask.to(at::kFloat);
    EXEC_NPU_CMD(aclnnMaskedCumsumCustom, self, mask_float, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("masked_cumsum_custom", &masked_cumsum_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("masked_cumsum_custom", &masked_cumsum_impl_npu,
          "Masked cumulative sum along dim (MaskedCumsumCustom on NPU)");
}
