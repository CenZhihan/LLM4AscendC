kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
                                GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures,
                                uint32_t outFeatures, uint32_t tileOutFeatures,
                                uint32_t numGroups)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        groupSize_ = outFeatures / numGroups;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inFeatures);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, inFeatures * outFeatures);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * outFeatures);

        // Allocate local buffers: one row of input, one row of output, temporary for reductions
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inFeatures_ * sizeof(float));
        pipe_.InitBuffer(inQueueWeight_, BUFFER_NUM, inFeatures_ * outFeatures_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(reduceQueue_, BUFFER_NUM, outFeatures_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outFeatures_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        for (uint32_t b = blockIdx; b < batchSize_; b += blockNum) {
            ProcessRow(b);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        // Load x row
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowIdx * inFeatures_], inFeatures_);

        // Load weight (full matrix, but we only need one row's worth? Actually we need entire matrix for matmul)
        // For simplicity, we load the entire weight matrix to local (might exceed UB, but tileOutF=outFeatures)
        AscendC::LocalTensor<float> weightLocal = inQueueWeight_.AllocTensor<float>();
        AscendC::DataCopy(weightLocal, weightGm_, inFeatures_ * outFeatures_);

        // Allocate output
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Perform matmul: y = x * weight^T  (weight is stored as inFeatures x outFeatures)
        // Use Matmul API: dst, src0, src1, M, K, N, transA, transB
        // Here: M=1, K=inFeatures, N=outFeatures, transA=false, transB=false (weight is KxN)
        AscendC::Matmul(yLocal, xLocal, weightLocal, 1, inFeatures_, outFeatures_, false, false);

        // Add lin_bias
        AscendC::LocalTensor<float> tmp1 = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(tmp1, linBiasGm_, outFeatures_);
        AscendC::Add(yLocal, yLocal, tmp1, outFeatures_);
        tmpQueue_.FreeTensor(tmp1);

        // Add bias
        AscendC::LocalTensor<float> tmp2 = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(tmp2, biasGm_, outFeatures_);
        AscendC::Add(yLocal, yLocal, tmp2, outFeatures_);
        tmpQueue_.FreeTensor(tmp2);

        // Hardtanh: clamp to [-1, 1] using Mins and Maxs
        AscendC::Mins(yLocal, yLocal, 1.0f, outFeatures_);
        AscendC::Maxs(yLocal, yLocal, -1.0f, outFeatures_);

        // Mish: x * tanh(softplus(x))
        // softplus = ln(1+exp(x))
        AscendC::LocalTensor<float> tmp3 = tmpQueue_.AllocTensor<float>();
        AscendC::Exp(tmp3, yLocal, outFeatures_);             // exp(x)
        AscendC::Adds(tmp3, tmp3, 1.0f, outFeatures_);        // 1+exp(x)
        AscendC::Log(tmp3, tmp3, outFeatures_);               // ln(1+exp(x))
        AscendC::Tanh(tmp3, tmp3, outFeatures_);              // tanh(softplus)
        AscendC::Mul(yLocal, yLocal, tmp3, outFeatures_);     // x * tanh(softplus)
        tmpQueue_.FreeTensor(tmp3);

        // Group norm: compute mean and variance per group
        // We'll use a reduction buffer for per-group sums
        AscendC::LocalTensor<float> groupSum = reduceQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> groupSum2 = reduceQueue_.AllocTensor<float>(); // for sum of squares
        // Initialize per-group accumulators (we treat groups as consecutive blocks)
        // Since numGroups_ is small (<=outFeatures), we can do manual loops.
        // For each group, compute sum and sum of squares using vector reductions.
        // We'll use ReduceSum with pattern = 0 (reduce along dimension)
        for (uint32_t g = 0; g < numGroups_; ++g) {
            uint32_t start = g * groupSize_;
            AscendC::LocalTensor<float> groupSlice = yLocal[start];
            // Compute sum
            AscendC::ReduceSum(groupSum[g], groupSlice, groupSize_, 0);
            // Compute sum of squares
            AscendC::Mul(groupSum2[g], groupSlice, groupSlice, groupSize_);
            AscendC::ReduceSum(groupSum2[g], groupSum2[g], groupSize_, 0);
        }
        // Compute mean and variance
        for (uint32_t g = 0; g < numGroups_; ++g) {
            float mean = groupSum[g] / (float)groupSize_;
            float var = groupSum2[g] / (float)groupSize_ - mean * mean;
            var = var + 1e-5f; // epsilon
            float invStd = 1.0f / AscendC::Sqrt(var); // use scalar sqrt
            // Apply normalization: (x - mean) * invStd * gamma + beta
            uint32_t start = g * groupSize_;
            AscendC::LocalTensor<float> groupSlice = yLocal[start];
            AscendC::Subs(groupSlice, groupSlice, mean, groupSize_);
            AscendC::Muls(groupSlice, groupSlice, invStd, groupSize_);
            // Multiply by gamma and add beta (element-wise)
            AscendC::LocalTensor<float> gammaSlice = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(gammaSlice, gnGammaGm_[start], groupSize_);
            AscendC::Mul(groupSlice, groupSlice, gammaSlice, groupSize_);
            AscendC::LocalTensor<float> betaSlice = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(betaSlice, gnBetaGm_[start], groupSize_);
            AscendC::Add(groupSlice, groupSlice, betaSlice, groupSize_);
            tmpQueue_.FreeTensor(gammaSlice);
            tmpQueue_.FreeTensor(betaSlice);
        }
        reduceQueue_.FreeTensor(groupSum);
        reduceQueue_.FreeTensor(groupSum2);

        // Copy result to global
        AscendC::DataCopy(yGm_[rowIdx * outFeatures_], yLocal, outFeatures_);

        // Free local tensors
        inQueueX_.FreeTensor(xLocal);
        inQueueWeight_.FreeTensor(weightLocal);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWeight_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> reduceQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_;
    uint32_t inFeatures_;
    uint32_t outFeatures_;
    uint32_t numGroups_;
    uint32_t groupSize_;
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
    GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.batchSize, tiling_data.inFeatures,
            tiling_data.outFeatures, tiling_data.tileOutFeatures,
            tiling_data.numGroups);
    op.Process();
}
