host_operator_src = '''
#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;
    const gert::Shape &inputShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(inputShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(inputShape.GetDim(1));
    uint32_t D_in = static_cast<uint32_t>(inputShape.GetDim(2));
    uint32_t H_in = static_cast<uint32_t>(inputShape.GetDim(3));
    uint32_t W_in = static_cast<uint32_t>(inputShape.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernel_size = attrs->GetInt(0);
    int32_t stride = attrs->GetInt(1);
    int32_t padding = attrs->GetInt(2);
    bool ceil_mode = attrs->GetBool(3);

    if (stride == 0) stride = kernel_size;

    uint32_t kD = static_cast<uint32_t>(kernel_size);
    uint32_t kH = static_cast<uint32_t>(kernel_size);
    uint32_t kW = static_cast<uint32_t>(kernel_size);
    uint32_t sD = static_cast<uint32_t>(stride);
    uint32_t sH = static_cast<uint32_t>(stride);
    uint32_t sW = static_cast<uint32_t>(stride);
    uint32_t pD = static_cast<uint32_t>(padding);
    uint32_t pH = static_cast<uint32_t>(padding);
    uint32_t pW = static_cast<uint32_t>(padding);

    auto computeOutSize = [&](uint32_t dim, uint32_t k, uint32_t s, uint32_t p) -> uint32_t {
        if (ceil_mode) {
            return static_cast<uint32_t>(std::ceil((dim + 2.0 * p - k) / static_cast<double>(s)) + 1);
        } else {
            return static_cast<uint32_t>(std::floor((dim + 2.0 * p - k) / static_cast<double>(s)) + 1);
        }
    };

    uint32_t D_out = computeOutSize(D_in, kD, sD, pD);
    uint32_t H_out = computeOutSize(H_in, kH, sH, pH);
    uint32_t W_out = computeOutSize(W_in, kW, sW, pW);

    uint32_t totalInputElements = N * C * D_in * H_in * W_in;
    uint32_t totalOutputElements = N * C * D_out * H_out * W_out;

    uint32_t windowSize = kD * kH * kW;
    uint32_t maxTileOutput = (UB_BUDGET_BYTES / sizeof(float)) / windowSize;
    maxTileOutput = AlignDown(maxTileOutput, ALIGN_ELEMS);
    if (maxTileOutput < ALIGN_ELEMS) maxTileOutput = ALIGN_ELEMS;
    if (maxTileOutput > HARD_CAP_ELEMS) maxTileOutput = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLength = maxTileOutput;
    if (totalOutputElements < tileLength) {
        tileLength = totalOutputElements;
    }

    uint32_t numTiles = CeilDiv(totalOutputElements, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalInputElements(totalInputElements);
    tiling.set_totalOutputElements(totalOutputElements);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_D_in(D_in);
    tiling.set_H_in(H_in);
    tiling.set_W_in(W_in);
    tiling.set_D_out(D_out);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_kernelD(kD);
    tiling.set_kernelH(kH);
    tiling.set_kernelW(kW);
    tiling.set_strideD(sD);
    tiling.set_strideH(sH);
    tiling.set_strideW(sW);
    tiling.set_padD(pD);
    tiling.set_padH(pH);
    tiling.set_padW(pW);
    tiling.set_ceilMode(ceil_mode ? 1 : 0);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t C = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t D_in = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t H_in = static_cast<uint32_t>(x_shape->GetDim(3));
    uint32_t W_in = static_cast<uint32_t>(x_shape->GetDim(4));

    // Read attributes
    const auto *attrs = context->GetAttrs();
    int32_t kernel_size = attrs->GetInt(0);
    int32_t stride = attrs->GetInt(1);
    int32_t padding = attrs->GetInt(2);
    bool ceil_mode = attrs->GetBool(3);
    if (stride == 0) stride = kernel_size;

    uint32_t kD = kernel_size, kH = kernel_size, kW = kernel_size;
    uint32_t sD = stride, sH = stride, sW = stride;
    uint32_t pD = padding, pH = padding, pW = padding;

    auto computeOutSize = [&](uint32_t dim, uint32_t k, uint32_t s, uint32_t p) -> uint32_t {
        if (ceil_mode) {
            return static_cast<uint32_t>(std::ceil((dim + 2.0 * p - k) / static_cast<double>(s)) + 1);
        } else {
            return static_cast<uint32_t>(std::floor((dim + 2.0 * p - k) / static_cast<double>(s)) + 1);
        }
    };

    uint32_t D_out = computeOutSize(D_in, kD, sD, pD);
    uint32_t H_out = computeOutSize(H_in, kH, sH, pH);
    uint32_t W_out = computeOutSize(W_in, kW, sW, pW);

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, D_out);
    y_shape->SetDim(3, H_out);
    y_shape->SetDim(4, W_out);
    return GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        totalInputElements_ = td.totalInputElements;
        totalOutputElements_ = td.totalOutputElements;
        N_ = td.N;
        C_ = td.C;
        D_in_ = td.D_in;
        H_in_ = td.H_in;
        W_in_ = td.W_in;
        D_out_ = td.D_out;
        H_out_ = td.H_out;
        W_out_ = td.W_out;
        kernelD_ = td.kernelD;
        kernelH_ = td.kernelH;
        kernelW_ = td.kernelW;
        strideD_ = td.strideD;
        strideH_ = td.strideH;
        strideW_ = td.strideW;
        padD_ = td.padD;
        padH_ = td.padH;
        padW_ = td.padW;
        ceilMode_ = td.ceilMode;
        tileLen_ = td.tileLength;
        blockDim_ = td.blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputElements_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElements_);

        uint32_t windowSize = kernelD_ * kernelH_ * kernelW_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, windowSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, windowSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElements_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutputElements_) return;

        uint32_t curLen = (totalOutputElements_ - base > tileLen_) ? tileLen_ : totalOutputElements_ - base;

        while (true) {
            ComputeTile(base, curLen);
            base += strideElems;
            if (base >= totalOutputElements_) break;
            curLen = (totalOutputElements_ - base > tileLen_) ? tileLen_ : totalOutputElements_ - base;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t outOffset, uint32_t count)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        for (uint32_t i = 0; i < count; i++) {
            uint32_t linearIdx = outOffset + i;
            uint32_t n = linearIdx / (C_ * D_out_ * H_out_ * W_out_);
            uint32_t rem1 = linearIdx % (C_ * D_out_ * H_out_ * W_out_);
            uint32_t c = rem1 / (D_out_ * H_out_ * W_out_);
            uint32_t rem2 = rem1 % (D_out_ * H_out_ * W_out_);
            uint32_t d = rem2 / (H_out_ * W_out_);
            uint32_t rem3 = rem2 % (H_out_ * W_out_);
            uint32_t h = rem3 / W_out_;
            uint32_t w = rem3 % W_out_;

            int32_t startD = static_cast<int32_t>(d) * static_cast<int32_t>(strideD_) - static_cast<int32_t>(padD_);
            int32_t startH = static_cast<int32_t>(h) * static_cast<int32_t>(strideH_) - static_cast<int32_t>(padH_);
            int32_t startW = static_cast<int32_t>(w) * static_cast<int32_t>(strideW_) - static_cast<int32_t>(padW_);
            int32_t endD = startD + static_cast<int32_t>(kernelD_);
            int32_t endH = startH + static_cast<int32_t>(kernelH_);
            int32_t endW = startW + static_cast<int32_t>(kernelW_);

            float maxVal = -std::numeric_limits<float>::infinity();
            for (int32_t kd = startD; kd < endD; kd++) {
                for (int32_t kh = startH; kh < endH; kh++) {
                    for (int32_t kw = startW; kw < endW; kw++) {
                        if (kd >= 0 && kd < static_cast<int32_t>(D_in_) &&
                            kh >= 0 && kh < static_cast<int32_t>(H_in_) &&
                            kw >= 0 && kw < static_cast<int32_t>(W_in_)) {
                            uint32_t inputIdx = n * C_ * D_in_ * H_in_ * W_in_
                                               + c * D_in_ * H_in_ * W_in_
                                               + kd * H_in_ * W_in_
                                               + kh * W_in_
                                               + kw;
                            float val = xGm_.GetValue(inputIdx);
                            if (val > maxVal) maxVal = val;
                        }
                    }
                }
            }
            yLocal.SetValue(i, maxVal);
        }
        outQueueY_.EnQue<float>(yLocal);
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[outOffset], yOut, count);
        outQueueY_.FreeTensor(yOut);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputElements_;
    uint32_t totalOutputElements_;
    uint32_t N_, C_, D_in_, H_in_, W_in_;
    uint32_t D_out_, H_out_, W_out_;
    uint32_t kernelD_, kernelH_, kernelW_;
    uint32_t strideD_, strideH_, strideW_;
    uint32_t padD_, padH_, padW_;
    uint32_t ceilMode_;
    uint32_t tileLen_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelMaxPooling3dCustom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}
'''
