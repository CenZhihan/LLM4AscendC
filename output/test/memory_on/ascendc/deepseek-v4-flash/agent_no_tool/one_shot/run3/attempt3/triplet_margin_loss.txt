#include "triplet_margin_loss_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;
static constexpr uint32_t HARD_CAP_ROWS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    // Use gert::Shape* directly (Memory 2)
    const gert::Shape *shape = context->GetInputShape(0);
    const uint32_t n = static_cast<uint32_t>(shape->GetOriginShape().GetDim(0));
    const uint32_t d = static_cast<uint32_t>(shape->GetOriginShape().GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    const uint32_t rowCostFloats = 5 * d;
    const uint32_t maxRows = UB_BUDGET_BYTES / (rowCostFloats * sizeof(float));
    uint32_t tileBatch = maxRows;
    if (tileBatch < 1) tileBatch = 1;
    if (tileBatch > HARD_CAP_ROWS) tileBatch = HARD_CAP_ROWS;
    if (tileBatch > n) tileBatch = n;

    tileBatch = AlignDown(tileBatch, ALIGN_ELEMS);
    if (tileBatch < 1) tileBatch = 1;

    uint32_t blockDim = CeilDiv(n, tileBatch);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(n);
    tiling.set_dim(d);
    tiling.set_tileBatch(tileBatch);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);

    // Set output shape to 1D with size blockDim
    gert::Shape outShape;
    outShape.SetDimNum(1);
    outShape.SetDim(0, static_cast<int64_t>(blockDim));
    context->SetOutputShape(0, outShape);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss_partial,
                                uint32_t totalRows, uint32_t dim,
                                uint32_t tileBatch, uint32_t blockDim, float margin)
    {
        totalRows_ = totalRows;
        dim_ = dim;
        tileBatch_ = tileBatch;
        blockDim_ = blockDim;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalRows_ * dim_);
        posGm_.SetGlobalBuffer((__gm__ float*)positive, totalRows_ * dim_);
        negGm_.SetGlobalBuffer((__gm__ float*)negative, totalRows_ * dim_);
        // output is a 1D array of size blockDim_
        lossPartialGm_.SetGlobalBuffer((__gm__ float*)loss_partial, blockDim_);

        pipe_.InitBuffer(inQueueAnchor_, BUFFER_NUM, tileBatch_ * dim_ * sizeof(float));
        pipe_.InitBuffer(inQueuePos_, BUFFER_NUM, tileBatch_ * dim_ * sizeof(float));
        pipe_.InitBuffer(inQueueNeg_, BUFFER_NUM, tileBatch_ * dim_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileBatch_ * dim_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileBatch_ * dim_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || dim_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t baseRow = blockIdx * tileBatch_;
        if (baseRow >= totalRows_) return;

        float localSum = 0.0f;

        while (baseRow < totalRows_) {
            uint32_t curBatch = totalRows_ - baseRow;
            if (curBatch > tileBatch_) curBatch = tileBatch_;

            CopyIn(baseRow, curBatch);
            localSum += Compute(curBatch);
            FreeTensors();

            const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
            baseRow += blockNum * tileBatch_;
        }

        // Write per-block partial sum to global output array
        lossPartialGm_.SetValue(blockIdx, localSum);
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowOffset, uint32_t batchSize)
    {
        uint32_t offset = rowOffset * dim_;
        uint32_t len = batchSize * dim_;

        {
            auto xLocal = inQueueAnchor_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, anchorGm_[offset], len);
            inQueueAnchor_.EnQue(xLocal);
        }
        {
            auto xLocal = inQueuePos_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, posGm_[offset], len);
            inQueuePos_.EnQue(xLocal);
        }
        {
            auto xLocal = inQueueNeg_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, negGm_[offset], len);
            inQueueNeg_.EnQue(xLocal);
        }
    }

    __aicore__ inline float Compute(uint32_t batchSize)
    {
        auto anchorLocal = inQueueAnchor_.DeQue<float>();
        auto posLocal = inQueuePos_.DeQue<float>();
        auto negLocal = inQueueNeg_.DeQue<float>();

        auto diffAp = tmpQueue1_.AllocTensor<float>();
        auto diffAn = tmpQueue2_.AllocTensor<float>();

        float rowSum = 0.0f;
        for (uint32_t r = 0; r < batchSize; ++r) {
            uint32_t rowOff = r * dim_;

            // diff_ap = anchor - positive
            AscendC::Sub(diffAp, anchorLocal[rowOff], posLocal[rowOff], dim_);
            // sq_ap = diff_ap * diff_ap
            AscendC::Mul(diffAp, diffAp, diffAp, dim_);

            // diff_an = anchor - negative
            AscendC::Sub(diffAn, anchorLocal[rowOff], negLocal[rowOff], dim_);
            AscendC::Mul(diffAn, diffAn, diffAn, dim_);

            // Sum of squares via scalar loop (simple fallback)
            float sumAp = 0.0f;
            float sumAn = 0.0f;
            for (uint32_t i = 0; i < dim_; ++i) {
                sumAp += diffAp.GetValue(i);
                sumAn += diffAn.GetValue(i);
            }

            float dAp = std::sqrt(sumAp);
            float dAn = std::sqrt(sumAn);

            float lossVal = dAp - dAn + margin_;
            if (lossVal < 0.0f) lossVal = 0.0f;
            rowSum += lossVal;
        }

        inQueueAnchor_.FreeTensor(anchorLocal);
        inQueuePos_.FreeTensor(posLocal);
        inQueueNeg_.FreeTensor(negLocal);
        tmpQueue1_.FreeTensor(diffAp);
        tmpQueue2_.FreeTensor(diffAn);

        return rowSum;
    }

    __aicore__ inline void FreeTensors()
    {
        // Tensors already freed in Compute; placeholder for structure
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePos_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNeg_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> posGm_;
    AscendC::GlobalTensor<float> negGm_;
    AscendC::GlobalTensor<float> lossPartialGm_;

    uint32_t totalRows_ {0};
    uint32_t dim_ {0};
    uint32_t tileBatch_ {0};
    uint32_t blockDim_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                  GM_ADDR loss_partial, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss_partial,
            tiling_data.totalRows, tiling_data.dim, tiling_data.tileBatch, tiling_data.blockDim, tiling_data.margin);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor triplet_margin_loss_npu(const at::Tensor& anchor,
                                    const at::Tensor& positive,
                                    const at::Tensor& negative,
                                    double margin) {
    // The output is a 1D tensor of size blockDim (set dynamically by tiling)
    // We create an empty tensor of type float; the framework will allocate correct size
    at::Tensor loss_partial = at::empty({0}, anchor.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnTripletMarginLossCustom, anchor, positive, negative, margin, loss_partial);

    // Sum all per-block contributions to get total loss, then average
    int64_t n = anchor.size(0);
    at::Tensor total_loss = loss_partial.sum();
    total_loss = total_loss / static_cast<float>(n);
    return total_loss;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("triplet_margin_loss_custom", &triplet_margin_loss_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("triplet_margin_loss_custom", &triplet_margin_loss_npu, "Triplet Margin Loss (TripletMarginLossCustom on NPU)");
}
