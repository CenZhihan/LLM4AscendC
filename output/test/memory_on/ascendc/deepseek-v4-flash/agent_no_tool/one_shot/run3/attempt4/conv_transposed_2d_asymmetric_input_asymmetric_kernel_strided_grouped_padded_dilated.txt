#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2d {
public:
    __aicore__ inline KernelConvTransposed2d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t strideH, int32_t strideW,
                                int32_t padH, int32_t padW,
                                int32_t dilationH, int32_t dilationW,
                                uint32_t group, uint32_t inChPerGroup,
                                uint32_t outChPerGroup, uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW,
                                uint32_t kH, uint32_t kW)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        group_ = group;
        inChPerGroup_ = inChPerGroup;
        outChPerGroup_ = outChPerGroup;
        inH_ = inH;
        inW_ = inW;
        outH_ = outH;
        outW_ = outW;
        kH_ = kH;
        kW_ = kW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, 0);
        wGm_.SetGlobalBuffer((__gm__ float*)w, 0);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            ProcessTile(base, curLen);

            base += strideElems;
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t outOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        for (uint32_t i = 0; i < len; i++) {
            uint32_t linearIdx = outOffset + i;
            uint32_t n = linearIdx / (outChPerGroup_ * group_ * outH_ * outW_);
            uint32_t rem = linearIdx % (outChPerGroup_ * group_ * outH_ * outW_);
            uint32_t oc = rem / (outH_ * outW_);
            uint32_t rem2 = rem % (outH_ * outW_);
            uint32_t oh = rem2 / outW_;
            uint32_t ow = rem2 % outW_;

            uint32_t g = oc / outChPerGroup_;
            uint32_t inC_start = g * inChPerGroup_;
            uint32_t oc_local = oc - g * outChPerGroup_;

            float acc = 0.0f;
            for (uint32_t ic_local = 0; ic_local < inChPerGroup_; ic_local++) {
                uint32_t ic = inC_start + ic_local;
                for (uint32_t kh = 0; kh < kH_; kh++) {
                    for (uint32_t kw = 0; kw < kW_; kw++) {
                        int32_t input_h = oh + padH_ - kh * dilationH_;
                        if (input_h < 0 || input_h % strideH_ != 0) continue;
                        input_h /= strideH_;
                        if ((uint32_t)input_h >= inH_) continue;

                        int32_t input_w = ow + padW_ - kw * dilationW_;
                        if (input_w < 0 || input_w % strideW_ != 0) continue;
                        input_w /= strideW_;
                        if ((uint32_t)input_w >= inW_) continue;

                        uint32_t xIdx = n * (inChPerGroup_ * group_ * inH_ * inW_)
                                        + ic * inH_ * inW_
                                        + (uint32_t)input_h * inW_
                                        + (uint32_t)input_w;
                        uint32_t wIdx = ic * outChPerGroup_ * kH_ * kW_
                                        + oc_local * kH_ * kW_
                                        + kh * kW_ + kw;

                        float x_val = xGm_.GetValue(xIdx);
                        float w_val = wGm_.GetValue(wIdx);
                        acc += x_val * w_val;
                    }
                }
            }
            yLocal.SetValue(i, acc);
        }

        AscendC::DataCopy(yGm_[outOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t strideH_, strideW_;
    int32_t padH_, padW_;
    int32_t dilationH_, dilationW_;
    uint32_t group_, inChPerGroup_, outChPerGroup_;
    uint32_t inH_, inW_, outH_, outW_, kH_, kW_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2d op;
    op.Init(x, w, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.group, tiling_data.inChannelsPerGroup,
            tiling_data.outChannelsPerGroup,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.kH, tiling_data.kW);
    op.Process();
}
