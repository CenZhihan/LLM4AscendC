host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cstdint>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t TILE_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    uint32_t inputDimNum = inputShape->GetOriginShape().GetDimNum();
    const int64_t *dims = inputShape->GetOriginShape().GetDims();

    // Get dim attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += inputDimNum;
    if (dim < 0 || dim >= (int64_t)inputDimNum) {
        return ge::GRAPH_FAILED;
    }

    // Compute leftSize, reductionSize, rightSize
    uint32_t leftSize = 1;
    for (uint32_t i = 0; i < (uint32_t)dim; ++i) {
        leftSize *= static_cast<uint32_t>(dims[i]);
    }
    uint32_t reductionSize = static_cast<uint32_t>(dims[dim]);
    uint32_t rightSize = 1;
    for (uint32_t i = (uint32_t)dim + 1; i < inputDimNum; ++i) {
        rightSize *= static_cast<uint32_t>(dims[i]);
    }

    uint32_t totalInputElems = leftSize * reductionSize * rightSize;
    uint32_t outputElems = leftSize * rightSize;

    // Determine tile length (number of output elements per block per iteration)
    // Each reduction loops over reductionSize elements; memory usage is tiny
    uint32_t tileLen = (outputElems < TILE_CAP_ELEMS) ? outputElems : TILE_CAP_ELEMS;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen < ALIGN_ELEMS) tileLen = outputElems;  // fallback

    // Adaptive blockDim
    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(outputElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalInputElems(totalInputElems);
    tiling.set_outputElems(outputElems);
    tiling.set_reductionSize(reductionSize);
    tiling.set_rightSize(rightSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    uint32_t dimNum = x_shape->GetOriginShape().GetDimNum();
    const int64_t *dims = x_shape->GetOriginShape().GetDims();

    // Get dim attribute (same as above)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += dimNum;
    if (dim < 0 || dim >= (int64_t)dimNum) {
        return GRAPH_FAILED;
    }

    // Output shape: remove dimension 'dim'
    uint32_t outDimNum = dimNum - 1;
    std::vector<int64_t> outDims(outDimNum);
    uint32_t outIdx = 0;
    for (uint32_t i = 0; i < dimNum; ++i) {
        if (i != (uint32_t)dim) {
            outDims[outIdx++] = dims[i];
        }
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(outDimNum);
    for (uint32_t i = 0; i < outDimNum; ++i) {
        y_shape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputElems,
                                uint32_t outputElems,
                                uint32_t reductionSize,
                                uint32_t rightSize,
                                uint32_t tileLength,
                                uint32_t blockDim)
    {
        totalInputElems_ = totalInputElems;
        outputElems_ = outputElems;
        reductionSize_ = reductionSize;
        rightSize_ = rightSize;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputElems_);
    }

    __aicore__ inline void Process()
    {
        if (outputElems_ == 0 || reductionSize_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = blockDim_;
        uint32_t stride = blockNum * tileLen_;

        for (uint32_t start = blockIdx * tileLen_; start < outputElems_; start += stride) {
            uint32_t end = start + tileLen_;
            if (end > outputElems_) end = outputElems_;

            for (uint32_t idx = start; idx < end; ++idx) {
                // Compute base index of the first element in the reduction vector
                uint32_t base = (idx / rightSize_) * (reductionSize_ * rightSize_) + (idx % rightSize_);
                float sum = 0.0f;
                for (uint32_t r = 0; r < reductionSize_; ++r) {
                    sum += xGm_[base + r * rightSize_];
                }
                yGm_[idx] = sum / static_cast<float>(reductionSize_);
            }
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalInputElems_;
    uint32_t outputElems_;
    uint32_t reductionSize_;
    uint32_t rightSize_;
    uint32_t tileLen_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalInputElems,
            tiling_data.outputElems,
            tiling_data.reductionSize,
            tiling_data.rightSize,
            tiling_data.tileLength,
            tiling_data.blockDim);
    op.Process();
}
