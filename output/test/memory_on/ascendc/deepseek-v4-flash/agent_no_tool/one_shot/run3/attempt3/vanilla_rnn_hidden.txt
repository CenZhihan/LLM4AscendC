#include "vanilla_rnn_hidden_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (conservative):
//   w_i2h: K*H floats (K = I+H)
//   w_h2o: H*O floats
//   x_tile: tileBatch*I
//   h_prev_tile: tileBatch*H
//   inp_tile: tileBatch*K
//   pre_tile: tileBatch*H
//   h_new_tile: tileBatch*H
//   y_tile: tileBatch*O
//   temporary buffers for matmul/activation
// We keep a large budget and tile accordingly.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;  // favor DMA

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    VanillaRnnHiddenCustomTilingData tiling;
    // Use Shape* (not StorageShape*) to access origin shape
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *h0_shape = context->GetInputShape(1);
    const gert::Shape *w_i2h_shape = context->GetInputShape(2);
    const gert::Shape *w_h2o_shape = context->GetInputShape(4);

    uint32_t B = x_shape->GetOriginShape().GetDim(0);
    uint32_t I = x_shape->GetOriginShape().GetDim(1);
    uint32_t H = h0_shape->GetOriginShape().GetDim(1);
    // w_h2o shape: (O, H) -> first dim O, second H
    uint32_t O = w_h2o_shape->GetOriginShape().GetDim(0);

    // Compute K directly to avoid storing it
    uint32_t K = I + H;

    // Compute tileBatch: how many samples per block
    // Estimate memory per sample: I + H (inputs) + K (inp) + H (pre) + H (h_new) + O (y) = I + 3H + K + O = I + 3H + (I+H) + O = 2I + 4H + O
    uint32_t bytesPerSample = (2*I + 4*H + O) * sizeof(float);
    // also need weight matrices: w_i2h (K*H floats) and w_h2o (H*O floats). We assume they are loaded once per block.
    uint32_t weightBytes = (K*H + H*O) * sizeof(float);
    // Some overhead for matmul workspace -> factor 2 safety
    uint32_t budget = UB_BUDGET_BYTES - weightBytes - 4096; // reserve for misc
    uint32_t maxTileBatch = budget / bytesPerSample;
    if (maxTileBatch < 1) maxTileBatch = 1;
    uint32_t tileBatch = AlignDown(maxTileBatch, 1);  // not alignment critical
    if (tileBatch < 1) tileBatch = 1;
    if (tileBatch > B) tileBatch = B;  // clamp

    // blockDim: number of tiles needed
    uint32_t blockDim = CeilDiv(B, tileBatch);
    // cap blockDim to reasonable number (e.g., 24 cores typical)
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_totalBatch(B);
    tiling.set_inputDim(I);
    tiling.set_hiddenDim(H);
    tiling.set_outputDim(O);
    tiling.set_tileBatch(tileBatch);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "matmul.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHidden {
public:
    __aicore__ inline KernelVanillaRnnHidden() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0,
                                GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o,
                                GM_ADDR h_new, GM_ADDR y,
                                uint32_t totalBatch, uint32_t inputDim,
                                uint32_t hiddenDim, uint32_t outputDim,
                                uint32_t tileBatch)
    {
        totalBatch_ = totalBatch;
        inputDim_ = inputDim;
        hiddenDim_ = hiddenDim;
        outputDim_ = outputDim;
        K_ = inputDim + hiddenDim;
        tileBatch_ = tileBatch;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalBatch_ * inputDim_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, totalBatch_ * hiddenDim_);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, K_ * hiddenDim_);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, hiddenDim_);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, hiddenDim_ * outputDim_);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, outputDim_);
        hNewGm_.SetGlobalBuffer((__gm__ float*)h_new, totalBatch_ * hiddenDim_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalBatch_ * outputDim_);

        // Allocate local buffers – assume they fit within UB after tiling
        pipe_.InitBuffer(inQueueLocal_, BUFFER_NUM, tileBatch_ * inputDim_ * sizeof(float));
        pipe_.InitBuffer(inQueueH0_, BUFFER_NUM, tileBatch_ * hiddenDim_ * sizeof(float));
        pipe_.InitBuffer(inQueueWi2h_, BUFFER_NUM, K_ * hiddenDim_ * sizeof(float));
        pipe_.InitBuffer(inQueueBi2h_, BUFFER_NUM, hiddenDim_ * sizeof(float));
        pipe_.InitBuffer(inQueueWh2o_, BUFFER_NUM, hiddenDim_ * outputDim_ * sizeof(float));
        pipe_.InitBuffer(inQueueBh2o_, BUFFER_NUM, outputDim_ * sizeof(float));
        pipe_.InitBuffer(outQueueHNew_, BUFFER_NUM, tileBatch_ * hiddenDim_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileBatch_ * outputDim_ * sizeof(float));

        // Temporary buffers for matmul: need a large VecCalc buffer for intermediate results
        pipe_.InitBuffer(tmpQueueInp_, BUFFER_NUM, tileBatch_ * K_ * sizeof(float));
        pipe_.InitBuffer(tmpQueuePre_, BUFFER_NUM, tileBatch_ * hiddenDim_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueH_, BUFFER_NUM, tileBatch_ * hiddenDim_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueY_, BUFFER_NUM, tileBatch_ * outputDim_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalBatch_ == 0) return;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t startSample = blockIdx * tileBatch_;
        if (startSample >= totalBatch_) return;
        uint32_t curBatch = totalBatch_ - startSample;
        if (curBatch > tileBatch_) curBatch = tileBatch_;

        // Load weights once (shared across samples in this block)
        // w_i2h
        AscendC::LocalTensor<float> w_i2hLocal = inQueueWi2h_.AllocTensor<float>();
        AscendC::DataCopy(w_i2hLocal, w_i2hGm_, K_ * hiddenDim_);
        inQueueWi2h_.EnQue(w_i2hLocal);

        // b_i2h
        AscendC::LocalTensor<float> b_i2hLocal = inQueueBi2h_.AllocTensor<float>();
        AscendC::DataCopy(b_i2hLocal, b_i2hGm_, hiddenDim_);
        inQueueBi2h_.EnQue(b_i2hLocal);

        // w_h2o
        AscendC::LocalTensor<float> w_h2oLocal = inQueueWh2o_.AllocTensor<float>();
        AscendC::DataCopy(w_h2oLocal, w_h2oGm_, hiddenDim_ * outputDim_);
        inQueueWh2o_.EnQue(w_h2oLocal);

        // b_h2o
        AscendC::LocalTensor<float> b_h2oLocal = inQueueBh2o_.AllocTensor<float>();
        AscendC::DataCopy(b_h2oLocal, b_h2oGm_, outputDim_);
        inQueueBh2o_.EnQue(b_h2oLocal);

        // Load x and h0 for these samples
        uint32_t xOffset = startSample * inputDim_;
        uint32_t h0Offset = startSample * hiddenDim_;

        AscendC::LocalTensor<float> xLocal = inQueueLocal_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[xOffset], curBatch * inputDim_);
        inQueueLocal_.EnQue(xLocal);

        AscendC::LocalTensor<float> h0Local = inQueueH0_.AllocTensor<float>();
        AscendC::DataCopy(h0Local, h0Gm_[h0Offset], curBatch * hiddenDim_);
        inQueueH0_.EnQue(h0Local);

        // Allocate temporary buffers for concatenation and matmul results
        AscendC::LocalTensor<float> inp = tmpQueueInp_.AllocTensor<float>();
        AscendC::LocalTensor<float> pre = tmpQueuePre_.AllocTensor<float>();
        AscendC::LocalTensor<float> hNew = tmpQueueH_.AllocTensor<float>();
        AscendC::LocalTensor<float> yOut = tmpQueueY_.AllocTensor<float>();

        // Dequeue x and h0
        AscendC::LocalTensor<float> xDeq = inQueueLocal_.DeQue<float>();
        AscendC::LocalTensor<float> h0Deq = inQueueH0_.DeQue<float>();

        // Concatenate: inp = [x, h0] along last dimension (K = I + H)
        // Copy x part (first inputDim_ elements per sample)
        AscendC::DataCopy(inp, xDeq, curBatch * inputDim_);
        // Copy h0 part (next hiddenDim_ elements per sample)
        for (uint32_t s = 0; s < curBatch; ++s) {
            uint32_t inpOffset = s * K_ + inputDim_;
            uint32_t h0Offset_L = s * hiddenDim_;
            AscendC::DataCopy(inp[inpOffset], h0Deq[h0Offset_L], hiddenDim_);
        }

        // Free x and h0 local after concat
        inQueueLocal_.FreeTensor(xDeq);
        inQueueH0_.FreeTensor(h0Deq);

        // Dequeue w_i2h
        AscendC::LocalTensor<float> w_i2hDeq = inQueueWi2h_.DeQue<float>();

        // Compute pre = inp @ w_i2h^T (matmul A: curBatch x K, B: K x H, result: curBatch x H)
        // Use aicore matmul API with transposeB = true (since w_i2h is stored as KxH, we want A*B^T)
        // For simplicity, use Matmul object
        AscendC::Matmul matmulPre;
        matmulPre.SetTensorA(inp);
        matmulPre.SetTensorB(w_i2hDeq);
        matmulPre.SetBias(b_i2hLocal); // we will add bias later
        matmulPre.SetTransposeB(true);
        matmulPre.IterateAll(pre);

        // Add bias: pre += b_i2h
        // b_i2h is (H,), broadcast across batch
        // Use Vector Add with broadcasting
        AscendC::Add(pre, pre, b_i2hLocal, curBatch * hiddenDim_);

        // tanh activation
        AscendC::Tanh(pre, pre, curBatch * hiddenDim_);

        // Now compute y = pre @ w_h2o (no transpose) + b_h2o
        AscendC::LocalTensor<float> w_h2oDeq = inQueueWh2o_.DeQue<float>();

        AscendC::Matmul matmulY;
        matmulY.SetTensorA(pre);
        matmulY.SetTensorB(w_h2oDeq);
        matmulY.SetBias(b_h2oLocal);
        matmulY.SetTransposeB(false);
        matmulY.IterateAll(yOut);

        // Add b_h2o
        AscendC::Add(yOut, yOut, b_h2oLocal, curBatch * outputDim_);

        // Copy results to global memory
        uint32_t hNewOffset = startSample * hiddenDim_;
        uint32_t yOffset = startSample * outputDim_;
        AscendC::DataCopy(hNewGm_[hNewOffset], pre, curBatch * hiddenDim_);
        AscendC::DataCopy(yGm_[yOffset], yOut, curBatch * outputDim_);

        // Free temporary tensors
        tmpQueueInp_.FreeTensor(inp);
        tmpQueuePre_.FreeTensor(pre);
        tmpQueueH_.FreeTensor(hNew);   // not used finally; we used pre directly
        tmpQueueY_.FreeTensor(yOut);
        inQueueWi2h_.FreeTensor(w_i2hDeq);
        inQueueWh2o_.FreeTensor(w_h2oDeq);
        inQueueBi2h_.FreeTensor(b_i2hLocal);
        inQueueBh2o_.FreeTensor(b_h2oLocal);
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> w_i2hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> hNewGm_;
    AscendC::GlobalTensor<float> yGm_;

    AscendC::TPipe pipe_;

    // Local queues for input data (buffered)
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueLocal_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueH0_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWi2h_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBi2h_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWh2o_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBh2o_;

    // Local queues for output
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueHNew_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    // Temporary buffers for intermediate results (VecCalc)
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueInp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueuePre_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueH_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueY_;

    uint32_t totalBatch_ {0};
    uint32_t inputDim_ {0};
    uint32_t hiddenDim_ {0};
    uint32_t outputDim_ {0};
    uint32_t K_ {0};
    uint32_t tileBatch_ {0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(
    GM_ADDR x, GM_ADDR h0,
    GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o,
    GM_ADDR h_new, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHidden op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, h_new, y,
            tiling_data.totalBatch, tiling_data.inputDim,
            tiling_data.hiddenDim, tiling_data.outputDim,
            tiling_data.tileBatch);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'vanilla_rnn_hidden_custom'

class ModelNew(torch.nn.Module):
    """
    Optimized version: loop T times calling custom kernel each step.
    We pre-transpose w_i2h and w_h2o to match kernel expectations.
    """
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, x, h0, w_i2h, b_i2h, w_h2o, b_h2o):
        T, B, I = x.shape
        H = h0.shape[1]
        O = w_h2o.shape[0]   # original shape (O, H)
        K = I + H

        # Transpose weights for kernel: w_i2h: (H,K) -> (K,H), w_h2o: (O,H) -> (H,O)
        w_i2h_t = w_i2h.transpose(0,1).contiguous()   # (K,H)
        w_h2o_t = w_h2o.transpose(0,1).contiguous()   # (H,O)

        h = h0
        outs = []
        fn = getattr(custom_ops_lib, _FN)
        for t in range(T):
            h, y_t = fn(x[t], h, w_i2h_t, b_i2h, w_h2o_t, b_h2o)
            outs.append(y_t)
        return torch.stack(outs, dim=0)
'''
