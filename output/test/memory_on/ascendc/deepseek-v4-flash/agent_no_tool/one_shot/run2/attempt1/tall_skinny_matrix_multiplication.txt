host_operator_src="""

#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB usage estimate:
// - B (full): K * N floats, here K and N <= 1024, typical <= 1024
// - A tile: tileM * K floats
// - C tile: tileM * N floats
// - Row temp: N floats
// Max per core: 192 KB = 49152 floats
static constexpr uint32_t UB_BUDGET_FLOATS = 49152;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t MAX_TILE_M = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const auto& shapeA = context->GetInputShape(0)->GetOriginShape();
    const auto& shapeB = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = shapeA.GetDim(0);
    uint32_t K = shapeA.GetDim(1);
    uint32_t N = shapeB.GetDim(1);  // B is (K, N)

    // Free space after storing B fully (K*N floats)
    uint32_t bFloats = K * N;
    uint32_t freeFloats = (bFloats < UB_BUDGET_FLOATS) ? (UB_BUDGET_FLOATS - bFloats) : 0;

    // For A tile and C tile and temp: need (tileM * K) + (tileM * N) + N floats
    // Solve for tileM: freeFloats >= tileM * (K + N) + N
    uint32_t maxTile = 0;
    if (K + N > 0) {
        maxTile = (freeFloats - N) / (K + N);
        maxTile = AlignDown(maxTile, ALIGN_ELEMS);
        if (maxTile == 0) maxTile = ALIGN_ELEMS;
        if (maxTile > MAX_TILE_M) maxTile = AlignDown(MAX_TILE_M, ALIGN_ELEMS);
    }

    uint32_t tileM = (maxTile > 0) ? maxTile : 1;
    if (tileM > M) tileM = AlignDown(M, ALIGN_ELEMS);
    if (tileM == 0) tileM = M;
    tileN = N;  // B fits fully, so we can process whole N

    uint32_t numTiles = (M + tileM - 1) / tileM;
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const auto& shapeA = context->GetInputShape(0)->GetOriginShape();
    const auto& shapeB = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = shapeA.GetDim(0);
    uint32_t N = shapeB.GetDim(1);
    gert::Shape yShape(std::vector<int64_t>({static_cast<int64_t>(M), static_cast<int64_t>(N)}));
    context->SetOutputShape(0, yShape);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class TallSkinnyMatrixMultiplicationCustom : public OpDef {
public:
    explicit TallSkinnyMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")

            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

};

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, M_ * N_);

        uint32_t maxTileMN = (tileM_ > N_) ? tileM_ : N_;
        uint32_t bLocalSize = K_ * N_;
        uint32_t aLocalSize = tileM_ * K_;
        uint32_t cLocalSize = tileM_ * N_;
        uint32_t tempRowSize = N_;

        pipe_.InitBuffer(inQueueB_, 1, bLocalSize * sizeof(float));
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, aLocalSize * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, cLocalSize * sizeof(float));
        pipe_.InitBuffer(tmpRow_, 1, tempRowSize * sizeof(float));
        pipe_.InitBuffer(tmpScalar_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Load B once (it's small) into local buffer
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        AscendC::DataCopy(bLocal, bGm_[0], K_ * N_);
        inQueueB_.EnQue(bLocal);

        uint32_t strideTiles = blockNum * tileM_;
        uint32_t baseRow = blockIdx * tileM_;
        if (baseRow >= M_) return;

        while (baseRow < M_) {
            uint32_t curTileM = (M_ - baseRow) < tileM_ ? (M_ - baseRow) : tileM_;
            // Copy A tile
            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            AscendC::DataCopy(aLocal, aGm_[baseRow * K_], curTileM * K_);
            inQueueA_.EnQue(aLocal);

            // Compute C tile = A_tile * B
            AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
            AscendC::LocalTensor<float> tempRow = tmpRow_.AllocTensor<float>();
            AscendC::LocalTensor<float> scalars = tmpScalar_.AllocTensor<float>();

            // Initialize C tile to zeros
            AscendC::SetZero(cLocal, curTileM * N_);

            // Retrieve B from queue (first time) or already dequeued? We'll use a loop outer prod
            // For each row i in A_tile, for each k, add A[i][k]*B[k][:] to C[i][:]
            // We can vectorize using broadcasting: for each k, we have a scalar A[i][k] and vector B[k][:].
            // We'll iterate over k in a loop, using repeat patterns.
            // Since K is small, loop is acceptable.
            for (uint32_t i = 0; i < curTileM; i++) {
                // load row of A tile into scalars? We'll load each element separately
                AscendC::LocalTensor<float> aRow = aLocal[i * K_]; // not correct indexing; but we can use byte offset
                // Instead, we can load the whole A tile into a linear region and index.
                // Let's use pointer arithmetic: we have aLocal as float*, we can index by i*K_+k.
                // For kernel code we can't use pointer, but we can use tensor offsets.
                // AscendC provides AscendC::Tensor<float>::operator() for indexing? Actually LocalTensor has [] operator.
                // We'll do nested loops:
                for (uint32_t k = 0; k < K_; k++) {
                    float aVal = aLocal(i * K_ + k);
                    // Broadcast aVal and multiply with B row k, add to C row i
                    AscendC::Dup(scalars, aVal, 1);
                    AscendC::LocalTensor<float> bRow = bLocal(k * N_);
                    AscendC::Mul(tempRow, bRow, scalars, N_);
                    AscendC::LocalTensor<float> cRow = cLocal(i * N_);
                    AscendC::Add(cRow, cRow, tempRow, N_);
                }
            }
            // Write result back
            AscendC::DataCopy(cGm_[baseRow * N_], cLocal, curTileM * N_);

            // Free tensors
            tmpRow_.FreeTensor(tempRow);
            tmpScalar_.FreeTensor(scalars);
            outQueueC_.EnQue(cLocal);
            // Dequeue and free A and C
            inQueueA_.FreeTensor(aLocal);
            AscendC::LocalTensor<float> cLocalOut = outQueueC_.DeQue<float>();
            outQueueC_.FreeTensor(cLocalOut);

            baseRow += strideTiles;
        }
        // Free B (only once, but since process is per-core and B is reused, we can free here)
        AscendC::LocalTensor<float> bLocalFree = inQueueB_.DeQue<float>();
        inQueueB_.FreeTensor(bLocalFree);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpRow_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpScalar_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(A, B, C, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
