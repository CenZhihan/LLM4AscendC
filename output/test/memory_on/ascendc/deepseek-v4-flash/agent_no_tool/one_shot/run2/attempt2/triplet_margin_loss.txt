#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: each sample needs 2*d floats for diff buffers + small overhead
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t HARD_CAP_SAMPLES = 4096;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TripletMarginLossCustomTilingData tiling;
    const auto &originShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalSamples = static_cast<uint32_t>(originShape.GetDim(0));
    const uint32_t sampleDim   = static_cast<uint32_t>(originShape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *marginPtr = attrs->GetAttrPointer<float>(0);
    float margin = marginPtr ? *marginPtr : 1.0f;

    // Determine tileSamples: each sample needs 2 * sampleDim floats for diffs
    uint32_t bytesPerSample = 2 * sampleDim * sizeof(float);
    uint32_t maxSamples = (bytesPerSample > 0) ? (UB_BUDGET_BYTES / bytesPerSample) : 1;
    if (maxSamples > HARD_CAP_SAMPLES) maxSamples = HARD_CAP_SAMPLES;
    if (maxSamples < 1) maxSamples = 1;
    uint32_t tileSamples = (totalSamples < maxSamples) ? totalSamples : maxSamples;
    if (tileSamples == 0) tileSamples = 1;

    // blockDim: number of tiles, capped at 24
    uint32_t numTiles = CeilDiv(totalSamples, tileSamples);
    uint32_t blockDim = (numTiles > 24) ? 24 : (numTiles < 1 ? 1 : numTiles);

    context->SetBlockDim(blockDim);

    tiling.set_totalSamples(totalSamples);
    tiling.set_sampleDim(sampleDim);
    tiling.set_tileSamples(tileSamples);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    // Input shapes: all [n, d], output [n]
    const gert::Shape *anchor_shape = context->GetInputShape(0);
    gert::Shape *loss_shape = context->GetOutputShape(0);
    loss_shape->SetDimNum(1);
    loss_shape->SetDim(0, anchor_shape->GetDim(0));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class TripletMarginLossCustom : public OpDef {
public:
    explicit TripletMarginLossCustom(const char* name) : OpDef(name) {
        this->Input("anchor")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("positive")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("negative")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("loss")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("margin").AttrType(OPTIONAL).Float(1.0f);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(TripletMarginLossCustom);
} // namespace ops
