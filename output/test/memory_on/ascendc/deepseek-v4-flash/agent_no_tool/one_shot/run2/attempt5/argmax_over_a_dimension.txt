host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape &inputShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t ndim = inputShape.GetDimNum();

    std::vector<int64_t> dims(ndim);
    for (uint32_t i = 0; i < ndim; ++i) {
        dims[i] = inputShape.GetDim(i);
    }

    uint32_t totalLength = 1;
    for (size_t i = 0; i < ndim; ++i) totalLength *= dims[i];
    tiling.set_totalInputLength(totalLength);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim = attrs->GetAttrPointer<int32_t>(0);
    int32_t redDim = *dim;
    if (redDim < 0) redDim += ndim;
    tiling.set_dim(redDim);

    uint32_t innerDim = dims[redDim];
    uint32_t pre = 1, post = 1;
    for (size_t i = 0; i < redDim; ++i) pre *= dims[i];
    for (size_t i = redDim + 1; i < ndim; ++i) post *= dims[i];
    tiling.set_innerDim(innerDim);
    tiling.set_outerDimPre(pre);
    tiling.set_outerDimPost(post);

    uint32_t maxTile = UB_BUDGET_BYTES / 24;  // per element bytes (6 tensors of different types)
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > post) maxTile = AlignDown(post, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = post;

    uint32_t outputLen = pre * post;
    uint32_t numTiles = (maxTile == 0) ? 1 : CeilDiv(outputLen, maxTile);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    tiling.set_tileLen(maxTile);
    tiling.set_blockDim(blockDim);

    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const uint32_t ndim = x_shape->GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim = attrs->GetAttrPointer<int32_t>(0);
    int32_t redDim = *dim;
    if (redDim < 0) redDim += ndim;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(ndim - 1);
    uint32_t outIdx = 0;
    for (uint32_t i = 0; i < ndim; ++i) {
        if ((int32_t)i != redDim) {
            y_shape->SetDim(outIdx, x_shape->GetDim(i));
            ++outIdx;
        }
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class ArgmaxOverADimensionCustom : public OpDef {
public:
    explicit ArgmaxOverADimensionCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT64})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(ArgmaxOverADimensionCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
using namespace AscendC;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputLength, int32_t dim,
                                uint32_t innerDim, uint32_t outerDimPre,
                                uint32_t outerDimPost, uint32_t tileLen)
    {
        totalInputLength_ = totalInputLength;
        dim_ = dim;
        innerDim_ = innerDim;
        outerDimPre_ = outerDimPre;
        outerDimPost_ = outerDimPost;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLength);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, outerDimPre * outerDimPost);

        pipe_.InitBuffer(inRowQue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(maxValsQue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(maxIdxQue_, 1, tileLen_ * sizeof(int32_t));
        pipe_.InitBuffer(jVecQue_, 1, tileLen_ * sizeof(int32_t));
        pipe_.InitBuffer(maskQue_, 1, tileLen_ * sizeof(uint8_t));
        pipe_.InitBuffer(outQue_, 1, tileLen_ * sizeof(int64_t));
    }

    __aicore__ inline void Process()
    {
        if (tileLen_ == 0 || outerDimPre_ == 0 || outerDimPost_ == 0) return;

        const uint32_t blockNum = GetBlockNum();
        const uint32_t blockIdx = GetBlockIdx();

        for (uint32_t b = blockIdx; b < outerDimPre_; b += blockNum) {
            for (uint32_t kStart = 0; kStart < outerDimPost_; kStart += tileLen_) {
                uint32_t curTileLen = tileLen_;
                if (kStart + curTileLen > outerDimPost_) {
                    curTileLen = outerDimPost_ - kStart;
                }

                // Allocate temporary buffers for this tile
                LocalTensor<float> inRow = inRowQue_.AllocTensor<float>();
                LocalTensor<float> maxVals = maxValsQue_.AllocTensor<float>();
                LocalTensor<int32_t> maxIdx = maxIdxQue_.AllocTensor<int32_t>();

                // Initialize maxVals to -inf and maxIdx to 0
                Duplicate(maxVals, -3.4028235e+38f, curTileLen);
                Duplicate(maxIdx, (int32_t)0, curTileLen);

                for (uint32_t j = 0; j < innerDim_; ++j) {
                    // Copy row from global memory
                    uint64_t offset = (uint64_t)b * innerDim_ * outerDimPost_ + (uint64_t)j * outerDimPost_ + kStart;
                    DataCopy(inRow, xGm_[offset], curTileLen);

                    // Allocate jVec and mask for this iteration
                    LocalTensor<int32_t> jVec = jVecQue_.AllocTensor<int32_t>();
                    LocalTensor<uint8_t> mask = maskQue_.AllocTensor<uint8_t>();
                    Duplicate(jVec, (int32_t)j, curTileLen);

                    // Compare row > maxVals and select accordingly
                    Compare(mask, inRow, maxVals, CMPMODE::GT, curTileLen);
                    Select(maxVals, inRow, maxVals, mask, curTileLen);
                    Select(maxIdx, jVec, maxIdx, mask, curTileLen);

                    // Release temporary buffers
                    jVecQue_.FreeTensor(jVec);
                    maskQue_.FreeTensor(mask);
                }

                // Convert maxIdx (int32) to int64 for output
                LocalTensor<int64_t> out = outQue_.AllocTensor<int64_t>();
                Cast(out, maxIdx, CastMode::CAST, curTileLen);

                // Write to global output
                uint64_t outOffset = (uint64_t)b * outerDimPost_ + kStart;
                DataCopy(yGm_[outOffset], out, curTileLen);

                // Free all buffers for this tile
                outQue_.FreeTensor(out);
                maxValsQue_.FreeTensor(maxVals);
                maxIdxQue_.FreeTensor(maxIdx);
                inRowQue_.FreeTensor(inRow);
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, 1> inRowQue_;
    TQue<TPosition::VECCALC, 1> maxValsQue_;
    TQue<TPosition::VECCALC, 1> maxIdxQue_;
    TQue<TPosition::VECCALC, 1> jVecQue_;
    TQue<TPosition::VECCALC, 1> maskQue_;
    TQue<TPosition::VECOUT, 1> outQue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<int64_t> yGm_;

    uint32_t totalInputLength_ {0};
    int32_t dim_ {-1};
    uint32_t innerDim_ {0};
    uint32_t outerDimPre_ {0};
    uint32_t outerDimPost_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalInputLength, tiling_data.dim,
            tiling_data.innerDim, tiling_data.outerDimPre, tiling_data.outerDimPost,
            tiling_data.tileLen);
    op.Process();
}
