kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;  // no double buffering needed for sequential time steps

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                uint32_t T, uint32_t B, uint32_t I, uint32_t H, uint32_t O,
                                uint32_t batchIdx)
    {
        T_ = T;
        B_ = B;
        I_ = I;
        H_ = H;
        O_ = O;
        batchIdx_ = batchIdx;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T * B * I);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B * H);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H * (I + H));
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O * H);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O);
        yGm_.SetGlobalBuffer((__gm__ float*)y, T * B * O);

        // Allocate UB buffers: each of size max(I,H,O) or K = I+H
        const uint32_t K = I + H;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, I_ * sizeof(float));
        pipe_.InitBuffer(inQueueH_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(inQueueWRow_, BUFFER_NUM, K * sizeof(float));  // reuse for w_i2h and w_h2o rows
        pipe_.InitBuffer(tmpQueuePre_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueY_, 1, O_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueDot_, 1, K * sizeof(float));  // for dot product
    }

    __aicore__ inline void Process()
    {
        // Load initial hidden state for this batch row
        for (uint32_t i = 0; i < H_; i++) {
            hLocal_(i) = h0Gm_[batchIdx_ * H_ + i];
        }

        // Loop over time steps
        for (uint32_t t = 0; t < T_; t++) {
            // Load x[t, batchIdx, :]
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[t * B_ * I_ + batchIdx_ * I_], I_);
            // Concat xLocal and hLocal into a temporary tensor (size K)
            AscendC::LocalTensor<float> concatLocal = tmpQueueDot_.Get();
            AscendC::DataCopy(concatLocal, xLocal, I_);
            AscendC::DataCopy(concatLocal[I_], hLocal_, H_);
            inQueueX_.FreeTensor(xLocal);

            // Compute pre = (concat @ w_i2h.T) + b_i2h
            // Process each hidden neuron j
            for (uint32_t j = 0; j < H_; j++) {
                // Load row j of w_i2h (size K)
                AscendC::LocalTensor<float> wRow = inQueueWRow_.AllocTensor<float>();
                AscendC::DataCopy(wRow, w_i2hGm_[j * (I_ + H_)], I_ + H_);
                // Elementwise mul: concat * wRow
                AscendC::LocalTensor<float> dotTmp = tmpQueueDot_.Get();
                AscendC::Mul(dotTmp, concatLocal, wRow, I_ + H_);
                // Reduce sum (all elements) -> scalar
                float sum = 0.0f;
                for (uint32_t k = 0; k < I_ + H_; k++) {
                    sum += dotTmp(k);
                }
                sum += b_i2hGm_[j];
                // Store in pre temp buffer
                // We'll reuse pre buffer stored in hLocal? Actually we need separate pre buffer.
                // Use tmpQueuePre_ for pre values.
                AscendC::LocalTensor<float> preLocal = tmpQueuePre_.Get();
                preLocal(j) = sum;
            }
            inQueueWRow_.FreeTensor(wRow);  // freed inside loop? Actually we allocate each iteration, need to free after use.

            // Compute h = tanh(pre)
            AscendC::LocalTensor<float> preLocal = tmpQueuePre_.Get();
            AscendC::Tanh(hLocal_, preLocal, H_);

            // Compute y_t = h @ w_h2o.T + b_h2o
            for (uint32_t k = 0; k < O_; k++) {
                AscendC::LocalTensor<float> wRow = inQueueWRow_.AllocTensor<float>();
                AscendC::DataCopy(wRow, w_h2oGm_[k * H_], H_);
                AscendC::LocalTensor<float> dotTmp = tmpQueueDot_.Get();
                AscendC::Mul(dotTmp, hLocal_, wRow, H_);
                float sum = 0.0f;
                for (uint32_t i = 0; i < H_; i++) {
                    sum += dotTmp(i);
                }
                sum += b_h2oGm_[k];
                // Store in y_local buffer
                AscendC::LocalTensor<float> yLocal = tmpQueueY_.Get();
                yLocal(k) = sum;
            }
            // Write y[t, batchIdx, :] to global
            AscendC::DataCopy(yGm_[t * B_ * O_ + batchIdx_ * O_], yLocal, O_);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueH_;  // not used as queue, but buffer for h
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWRow_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueuePre_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueDot_;

    AscendC::LocalTensor<float> hLocal_;  // persistent across time

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> w_i2hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t T_ {0};
    uint32_t B_ {0};
    uint32_t I_ {0};
    uint32_t H_ {0};
    uint32_t O_ {0};
    uint32_t batchIdx_ {0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h,
                                                                GM_ADDR b_i2h, GM_ADDR w_h2o,
                                                                GM_ADDR b_h2o, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    uint32_t batchIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I, tiling_data.H, tiling_data.O,
            batchIdx);
    op.Process();
}
