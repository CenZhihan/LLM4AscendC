host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/utils/op_desc_utils.h"
#include "graph/types.h"

namespace optiling {

// UB budget for one tile:
//   input_row:       tileLen * 4 bytes
//   max_vals:        tileLen * 4 bytes
//   max_idx:         tileLen * 4 bytes
//   output:          tileLen * 8 bytes
//   jVec (tmp):      tileLen * 4 bytes
//   conversion temp: tileLen * 8 bytes (can reuse output buffer after convert)
//   total ~ 28 * tileLen bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 32;              // DMA alignment

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape &inputShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t ndim = inputShape.GetDimNum();
    const int32_t *inputDims = inputShape.GetDims();

    // Compute total length
    uint32_t totalLength = 1;
    for (size_t i = 0; i < ndim; ++i) totalLength *= inputDims[i];
    tiling.set_totalInputLength(totalLength);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim = attrs->GetAttrPointer<int32_t>(0);
    int32_t redDim = *dim;
    if (redDim < 0) redDim += ndim;
    tiling.set_dim(redDim);

    // Compute innerDim (size of dimension to reduce), outerDimPre (product before), outerDimPost (product after)
    uint32_t innerDim = inputDims[redDim];
    uint32_t pre = 1, post = 1;
    for (size_t i = 0; i < redDim; ++i) pre *= inputDims[i];
    for (size_t i = redDim + 1; i < ndim; ++i) post *= inputDims[i];
    tiling.set_innerDim(innerDim);
    tiling.set_outerDimPre(pre);
    tiling.set_outerDimPost(post);

    // tileLen from UB budget: bytes = 28 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 28;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    // Do not exceed post (to keep each tile within one batch)
    if (maxTile > post) maxTile = AlignDown(post, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = post; // if post < ALIGN_ELEMS
    uint32_t tileLen = maxTile;

    uint32_t outputLen = pre * post;
    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(outputLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const uint32_t ndim = x_shape->GetDimNum();
    const int32_t *dims = x_shape->GetDims();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim = attrs->GetAttrPointer<int32_t>(0);
    int32_t redDim = *dim;
    if (redDim < 0) redDim += ndim;

    // Output shape = input shape with redDim removed
    std::vector<int32_t> outDims;
    for (size_t i = 0; i < ndim; ++i) {
        if ((int32_t)i != redDim) {
            outDims.push_back(dims[i]);
        }
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    ge::Shape geShape(outDims);
    *y_shape = gert::Shape(geShape);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"
using namespace AscendC;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputLength, int32_t dim,
                                uint32_t innerDim, uint32_t outerDimPre,
                                uint32_t outerDimPost, uint32_t tileLen)
    {
        totalInputLength_ = totalInputLength;
        dim_ = dim;
        innerDim_ = innerDim;
        outerDimPre_ = outerDimPre;
        outerDimPost_ = outerDimPost;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLength);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, outerDimPre * outerDimPost);

        // Allocate local buffers
        pipe_.InitBuffer(inRowBuf_, 1, tileLen_ * sizeof(float));       // row input
        pipe_.InitBuffer(maxValsBuf_, 1, tileLen_ * sizeof(float));     // max values
        pipe_.InitBuffer(maxIdxBuf_, 1, tileLen_ * sizeof(int32_t));    // max indices (int32)
        pipe_.InitBuffer(outBuf_, 1, tileLen_ * sizeof(int64_t));       // output (int64)
        pipe_.InitBuffer(jVecBuf_, 1, tileLen_ * sizeof(int32_t));      // temporary for j vector
    }

    __aicore__ inline void Process()
    {
        if (tileLen_ == 0 || outerDimPre_ == 0 || outerDimPost_ == 0) return;

        const uint32_t blockNum = GetBlockNum();
        const uint32_t blockIdx = GetBlockIdx();

        // Iterate over batches in grid-stride manner
        for (uint32_t b = blockIdx; b < outerDimPre_; b += blockNum) {
            // Process all k start positions within this batch
            for (uint32_t kStart = 0; kStart < outerDimPost_; kStart += tileLen_) {
                uint32_t curTileLen = tileLen_;
                if (kStart + curTileLen > outerDimPost_) {
                    curTileLen = outerDimPost_ - kStart;
                }

                // Initialize max values to -inf, indices to 0
                InitMax(curTileLen);

                // Load rows and update argmax
                for (uint32_t j = 0; j < innerDim_; ++j) {
                    // Load one row of curTileLen floats from global memory
                    uint64_t offset = (uint64_t)b * innerDim_ * outerDimPost_ + (uint64_t)j * outerDimPost_ + kStart;
                    LocalTensor<float> row = inRowBuf_.Get<float>();
                    DataCopy(row, xGm_[offset], curTileLen);

                    // Compare and update
                    UpdateArgmax(row, j, curTileLen);
                }

                // Convert int32 indices to int64 and write out
                ConvertAndOutput(b, kStart, curTileLen);
            }
        }
    }

private:
    __aicore__ inline void InitMax(uint32_t len)
    {
        LocalTensor<float> maxVals = maxValsBuf_.Get<float>();
        LocalTensor<int32_t> maxIdx = maxIdxBuf_.Get<int32_t>();

        // Set max values to -FLT_MAX (use a large negative number)
        Duplicate(maxVals, -3.4028235e+38f, len);
        Duplicate(maxIdx, 0, len);
    }

    __aicore__ inline void UpdateArgmax(LocalTensor<float>& row, uint32_t j, uint32_t len)
    {
        LocalTensor<float> maxVals = maxValsBuf_.Get<float>();
        LocalTensor<int32_t> maxIdx = maxIdxBuf_.Get<int32_t>();
        LocalTensor<int32_t> jVec = jVecBuf_.Get<int32_t>();

        // Create a vector filled with current j
        Duplicate(jVec, (int32_t)j, len);

        // Compare: mask = (row > maxVals)
        LocalTensor<uint8_t> mask = pipe_.TmpAllocTensor<uint8_t>(len);
        VCMP(mask, row, maxVals, CMP_GT, len);

        // Select new max values
        VSEL(maxVals, maxVals, row, mask, SEL_VSEL, len);
        // Select new indices
        VSEL(maxIdx, maxIdx, jVec, mask, SEL_VSEL, len);

        pipe_.TmpFreeTensor(mask);
    }

    __aicore__ inline void ConvertAndOutput(uint32_t b, uint32_t kStart, uint32_t len)
    {
        LocalTensor<int32_t> maxIdx = maxIdxBuf_.Get<int32_t>();
        LocalTensor<int64_t> out = outBuf_.Get<int64_t>();

        // Convert int32 to int64
        if (len > 0) {
            // Use ConvertPrecision for type conversion
            ConvertPrecision(out, maxIdx, len);
        }

        // Write to global memory
        uint64_t outOffset = (uint64_t)b * outerDimPost_ + kStart;
        DataCopy(yGm_[outOffset], out, len);
    }

private:
    TPipe pipe_;
    TBuf<TPosition::VECIN> inRowBuf_;      // input row
    TBuf<TPosition::VECIN> maxValsBuf_;    // max values
    TBuf<TPosition::VECIN> maxIdxBuf_;     // max indices (int32)
    TBuf<TPosition::VECOUT> outBuf_;       // output (int64)
    TBuf<TPosition::VECCALC> jVecBuf_;     // temporary j vector

    GlobalTensor<float> xGm_;
    GlobalTensor<int64_t> yGm_;

    uint32_t totalInputLength_ {0};
    int32_t dim_ {-1};
    uint32_t innerDim_ {0};
    uint32_t outerDimPre_ {0};
    uint32_t outerDimPost_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalInputLength, tiling_data.dim,
            tiling_data.innerDim, tiling_data.outerDimPre, tiling_data.outerDimPost,
            tiling_data.tileLen);
    op.Process();
}
"""

import torch
import torch_npu
import custom_ops_lib

_FN = 'argmax_over_a_dimension_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        if len(init_inputs) > 0:
            self.dim = init_inputs[0]
        else:
            self.dim = 1  # default

    def forward(self, *inputs):
        x = inputs[0]
        fn = getattr(custom_ops_lib, _FN)
