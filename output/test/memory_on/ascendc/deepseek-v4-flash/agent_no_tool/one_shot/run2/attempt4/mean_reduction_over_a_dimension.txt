host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: inQueueX holds tileLen * reductionSize floats (double buffered)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;
    const auto *storageShape = context->GetInputShape(0);
    const gert::Shape &originShape = storageShape->GetOriginShape();
    const uint32_t totalInputLength = static_cast<uint32_t>(originShape.GetShapeSize());
    const uint32_t dimNum = static_cast<uint32_t>(originShape.GetDimNum());
    const uint32_t reductionSize = static_cast<uint32_t>(originShape.GetDim(dimNum - 1));
    const uint32_t outputSize = totalInputLength / reductionSize;

    // Compute tileLen (number of output elements per iteration)
    // Each output element requires loading reductionSize floats, double buffered => 2 * reductionSize * 4 bytes
    uint32_t maxOutput = UB_BUDGET_BYTES / (2 * reductionSize * 4);
    maxOutput = AlignDown(maxOutput, ALIGN_ELEMS);
    if (maxOutput < 1) maxOutput = 1;
    if (maxOutput > HARD_CAP_ELEMS) maxOutput = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxOutput;
    if (outputSize < tileLen) {
        uint32_t aligned = AlignDown(outputSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : outputSize;
    }

    // Block dimension
    const uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(outputSize, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalInputLength(totalInputLength);
    tiling.set_reductionSize(reductionSize);
    tiling.set_outputSize(outputSize);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t dimNum = x_shape->GetDimNum();
    // The kernel reduces the last dimension, so output has one fewer dimension.
    gert::Shape y_shape;
    y_shape.SetDimNum(dimNum - 1);
    for (uint32_t i = 0; i < dimNum - 1; ++i) {
        y_shape.SetDim(i, x_shape->GetDim(i));
    }
    *context->GetOutputShape(0) = y_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReduction {
public:
    __aicore__ inline KernelMeanReduction() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputLength, uint32_t reductionSize,
                                uint32_t outputSize, uint32_t tileLen)
    {
        totalInputLen_ = totalInputLength;
        reductionSize_ = reductionSize;
        outputSize_ = outputSize;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * reductionSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalInputLen_ == 0 || reductionSize_ == 0 || outputSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideOutput = blockNum * tileLen_;
        uint32_t outIdx = blockIdx * tileLen_;
        if (outIdx >= outputSize_) return;

        while (outIdx < outputSize_) {
            uint32_t curTile = outputSize_ - outIdx;
            if (curTile > tileLen_) curTile = tileLen_;

            for (uint32_t i = 0; i < curTile; ++i) {
                uint32_t inputOffset = (outIdx + i) * reductionSize_;

                // Load one segment
                AscendC::LocalTensor<float> seg = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(seg, xGm_[inputOffset], reductionSize_);

                // Compute sum manually
                float sum = 0.0f;
                for (uint32_t j = 0; j < reductionSize_; ++j) {
                    sum += seg(j);
                }
                float mean = sum / static_cast<float>(reductionSize_);

                yGm_.SetValue(outIdx + i, mean);
                inQueueX_.FreeTensor(seg);
            }
            outIdx += strideOutput;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputLen_;
    uint32_t reductionSize_;
    uint32_t outputSize_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReduction op;
    op.Init(x, y, tiling_data.totalInputLength, tiling_data.reductionSize, tiling_data.outputSize, tiling_data.tileLen);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor mean_reduction_impl_npu(const at::Tensor& self) {
    // The input tensor has the reduction dimension as the last dimension.
    // Compute output shape by removing the last dimension.
    auto sizes = self.sizes().vec();
    sizes.pop_back();
    at::Tensor result = at::empty(sizes, self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("mean_reduction_over_a_dimension_custom", &mean_reduction_impl_npu, "Mean reduction over last dimension (MeanReductionOverADimensionCustom on NPU)");

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'mean_reduction_over_a_dimension_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, dim):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x):
        # Permute so that the reduction dimension is last
        ndim = x.dim()
        dim = self.dim if self.dim >= 0 else ndim + self.dim
        perm = [i for i in range(ndim) if i != dim] + [dim]
        x_perm = x.permute(perm).contiguous()
        fn = getattr(custom_ops_lib, _FN)
        return fn(x_perm)
'''
