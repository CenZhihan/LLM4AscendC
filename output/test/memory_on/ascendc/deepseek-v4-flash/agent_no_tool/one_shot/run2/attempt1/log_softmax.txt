host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX:  2 tiles
//   outQueueY: 2 tiles
//   tmpQueue:  1 tile
//   expQueue:  1 tile
//   Total floats = 6 * tile  => bytes = 24 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Extract the inner dimension (last dim)
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t innerDim = static_cast<uint32_t>(shape.GetDim(shape.GetDimNum() - 1));

    // Tile length from UB budget: 24 * tileLen < UB_BUDGET
    uint32_t maxTile = UB_BUDGET_BYTES / 24;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;
    // Ensure tileLen is a divisor of innerDim? Not required, but helps alignment.
    // We'll keep tileLen as is; kernel handles variable-length tiles.

    // Adaptive blockDim based on rows
    uint32_t numRows = totalLength / innerDim;
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_innerDim(innerDim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t innerDim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        innerDim_ = innerDim;
        numRows_ = totalLen_ / innerDim_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(expQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t rowsPerBlock = (numRows_ + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > numRows_) endRow = numRows_;
        if (startRow >= numRows_) return;

        for (uint32_t r = startRow; r < endRow; ++r) {
            uint32_t rowOffset = r * innerDim_;

            // Pass 1: compute row maximum
            float rowMax = computeRowMax(rowOffset);

            // Pass 2: compute sum(exp(x - rowMax))
            float sumExp = computeRowSumExp(rowOffset, rowMax);

            // Compute log(sum(exp)) + max
            float logSumExp = logf(sumExp) + rowMax;

            // Pass 3: compute y = x - logSumExp
            computeRowOutput(rowOffset, logSumExp);
        }
    }

private:
    __aicore__ float computeRowMax(uint32_t rowOffset)
    {
        float maxVal = -FLT_MAX;
        uint32_t remaining = innerDim_;
        uint32_t offset = rowOffset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            float tileMax = AscendC::ReduceMax(xLocal, curLen);
            if (tileMax > maxVal) maxVal = tileMax;

            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }
        return maxVal;
    }

    __aicore__ float computeRowSumExp(uint32_t rowOffset, float maxVal)
    {
        float sumExp = 0.0f;
        uint32_t remaining = innerDim_;
        uint32_t offset = rowOffset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> expLocal = expQueue_.AllocTensor<float>();
            AscendC::Exp(expLocal, xLocal, curLen);
            float tileSum = AscendC::ReduceSum(expLocal, curLen);
            sumExp += tileSum;

            expQueue_.FreeTensor(expLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }
        return sumExp;
    }

    __aicore__ void computeRowOutput(uint32_t rowOffset, float logSumExp)
    {
        uint32_t remaining = innerDim_;
        uint32_t offset = rowOffset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // Compute y = x - logSumExp
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Sub(yLocal, xLocal, logSumExp, curLen);

            // Write back
            outQueueY_.EnQue(yLocal);
            AscendC::LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yDeque, curLen);
            outQueueY_.FreeTensor(yDeque);
            inQueueX_.FreeTensor(xLocal);

            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> expQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t innerDim_ {0};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.innerDim);
    op.Process();
}
"""
