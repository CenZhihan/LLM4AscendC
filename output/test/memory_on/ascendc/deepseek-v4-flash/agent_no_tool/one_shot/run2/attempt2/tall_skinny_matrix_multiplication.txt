#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB usage estimate:
// - B tile: K * tileN floats (since tileN may be less than N)
// - A tile: tileM * K floats
// - C tile: tileM * tileN floats
// - Row temp: tileN floats
// Max per core: 192 KB = 49152 floats
static constexpr uint32_t UB_BUDGET_FLOATS = 49152;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t MAX_TILE_M = 4096;
static constexpr uint32_t MAX_TILE_N = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);
    uint32_t M = shapeA->GetDim(0);
    uint32_t K = shapeA->GetDim(1);
    uint32_t N = shapeB->GetDim(1);  // B is (K, N)

    // Compute tile sizes based on UB budget
    // Variables: tileM (rows of A), tileN (cols of B)
    // We require: tileM*K + K*tileN + tileM*tileN + tileN <= UB_BUDGET_FLOATS
    // Use heuristic: try tileN = N (if fits) else shrink
    uint32_t tileN = N;
    uint32_t tileM = 0;
    uint32_t needed = tileM * K + K * tileN + tileM * tileN + tileN;
    if (needed > UB_BUDGET_FLOATS) {
        // halve tileN iteratively until equation has solution
        while (tileN > 1) {
            tileN = AlignDown(tileN / 2, ALIGN_ELEMS);
            if (tileN == 0) tileN = ALIGN_ELEMS;
            // solve for max tileM:
            // (tileM * K) + (tileM * tileN) <= UB_BUDGET_FLOATS - K*tileN - tileN
            uint32_t remaining = UB_BUDGET_FLOATS - (K * tileN + tileN);
            if (K + tileN > 0) {
                tileM = remaining / (K + tileN);
                tileM = AlignDown(tileM, ALIGN_ELEMS);
                if (tileM > MAX_TILE_M) tileM = AlignDown(MAX_TILE_M, ALIGN_ELEMS);
            }
            if (tileM > 0) break;
        }
    } else {
        // tiledN fits, compute tileM
        uint32_t remaining = UB_BUDGET_FLOATS - (K * tileN + tileN);
        if (K + tileN > 0) {
            tileM = remaining / (K + tileN);
            tileM = AlignDown(tileM, ALIGN_ELEMS);
            if (tileM > MAX_TILE_M) tileM = AlignDown(MAX_TILE_M, ALIGN_ELEMS);
        }
    }

    // Ensure at least 1 element per tile
    if (tileM == 0) tileM = 1;
    if (tileN == 0) tileN = 1;
    if (tileM > M) tileM = AlignDown(M, ALIGN_ELEMS);
    if (tileM == 0) tileM = M;
    if (tileN > N) tileN = AlignDown(N, ALIGN_ELEMS);
    if (tileN == 0) tileN = N;

    uint32_t numRowTiles = (M + tileM - 1) / tileM;
    uint32_t numColTiles = (N + tileN - 1) / tileN;
    uint32_t totalTiles = numRowTiles * numColTiles;
    uint32_t blockDim = totalTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, M_ * N_);

        uint32_t bLocalSize = K_ * tileN_;   // one B tile
        uint32_t aLocalSize = tileM_ * K_;
        uint32_t cLocalSize = tileM_ * tileN_;
        uint32_t tempRowSize = tileN_;

        pipe_.InitBuffer(inQueueB_, 1, bLocalSize * sizeof(float));
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, aLocalSize * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, cLocalSize * sizeof(float));
        pipe_.InitBuffer(tmpRow_, 1, tempRowSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t numRowTiles = (M_ + tileM_ - 1) / tileM_;
        uint32_t numColTiles = (N_ + tileN_ - 1) / tileN_;
        uint32_t totalTiles = numRowTiles * numColTiles;

        // Each block handles consecutive tiles
        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            uint32_t rowTileIdx = tileIdx / numColTiles;
            uint32_t colTileIdx = tileIdx % numColTiles;
            uint32_t rowStart = rowTileIdx * tileM_;
            uint32_t colStart = colTileIdx * tileN_;
            uint32_t curTileM = (M_ - rowStart) < tileM_ ? (M_ - rowStart) : tileM_;
            uint32_t curTileN = (N_ - colStart) < tileN_ ? (N_ - colStart) : tileN_;

            // Copy B tile (K x tileN) — B is stored row-major: for each k, copy N elements starting at colStart
            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
            for (uint32_t k = 0; k < K_; k++) {
                uint32_t srcOffset = k * N_ + colStart;
                AscendC::DataCopy(bLocal[k * curTileN_], bGm_[srcOffset], curTileN);  // need to use correct offsets
                // Actually DataCopy expects contiguous. We'll copy element by element using SetValue? This is messy.
                // Instead, use a temporary global buffer? Since N is tall/skinny, tileN is small.
            }
            inQueueB_.EnQue(bLocal);

            // Copy A tile (tileM x K)
            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            AscendC::DataCopy(aLocal, aGm_[rowStart * K_], curTileM * K_);
            inQueueA_.EnQue(aLocal);

            // Compute C tile = A_tile * B_tile
            AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
            AscendC::SetZero(cLocal, curTileM * curTileN);

            // Dequeue B tile once
            AscendC::LocalTensor<float> bLocalIn = inQueueB_.DeQue<float>();
            // Iterate over rows of A tile
            for (uint32_t i = 0; i < curTileM; i++) {
                for (uint32_t k = 0; k < K_; k++) {
                    float aVal = aLocal.GetValue(i * K_ + k);
                    AscendC::LocalTensor<float> bRow = bLocalIn[k * curTileN_]; // need sub-tensor view
                    // Actually we cannot slice; we need to copy B row into tmpRow.
                    // For simplicity, copy B row into tempRow.
                    AscendC::LocalTensor<float> tempRow = tmpRow_.AllocTensor<float>();
                    for (uint32_t j = 0; j < curTileN; j++) {
                        tempRow.SetValue(j, bLocalIn.GetValue(k * curTileN + j));
                    }
                    // Multiply-add
                    for (uint32_t j = 0; j < curTileN; j++) {
                        float old = cLocal.GetValue(i * curTileN + j);
                        cLocal.SetValue(i * curTileN + j, old + aVal * tempRow.GetValue(j));
                    }
                    tmpRow_.FreeTensor(tempRow);
                }
            }
            inQueueB_.FreeTensor(bLocalIn);

            // Write C tile back
            // Need to handle column offset: output starts at rowStart*N_ + colStart
            // Write row by row because columns are not contiguous
            for (uint32_t i = 0; i < curTileM; i++) {
                uint32_t dstOffset = (rowStart + i) * N_ + colStart;
                for (uint32_t j = 0; j < curTileN; j++) {
                    float val = cLocal.GetValue(i * curTileN + j);
                    cGm_.SetValue(dstOffset + j, val);
                }
            }

            // Free tensors
            inQueueA_.FreeTensor(aLocal);
            outQueueC_.FreeTensor(cLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpRow_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(A, B, C, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
