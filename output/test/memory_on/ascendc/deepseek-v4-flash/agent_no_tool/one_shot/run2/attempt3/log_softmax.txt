host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model:
//   inQueueX:  2 tiles
//   outQueueY: 2 tiles
//   tmpQueue:  1 tile
//   expQueue:  1 tile
//   scalarQueue: 1 elem
//   broadcastBuf_: 1 tile
// Total approx 7 * tile floats -> 28 * tile bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength = 
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t innerDim = static_cast<uint32_t>(shape.GetDim(shape.GetDimNum() - 1));

    // Tile length from UB budget: 28 * tileLen < UB_BUDGET
    uint32_t maxTile = UB_BUDGET_BYTES / 28;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;

    uint32_t numRows = totalLength / innerDim;
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_innerDim(innerDim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -1.0e38f;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t innerDim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        innerDim_ = innerDim;
        numRows_ = totalLen_ / innerDim_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(expQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));               // 1 element for scalars
        pipe_.InitBuffer(reduceBuf_, 1, sizeof(float));                 // 1 element for reduction results
        pipe_.InitBuffer(broadcastBuf_, 1, tileLen_ * sizeof(float));   // for broadcasting scalar to tile
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t rowsPerBlock = (numRows_ + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > numRows_) endRow = numRows_;
        if (startRow >= numRows_) return;

        for (uint32_t r = startRow; r < endRow; ++r) {
            uint32_t rowOffset = r * innerDim_;

            // Pass 1: compute row maximum
            float rowMax = computeRowMax(rowOffset);

            // Pass 2: compute sum(exp(x - rowMax))
            float sumExp = computeRowSumExp(rowOffset, rowMax);

            // compute log(sumExp) + rowMax
            AscendC::LocalTensor<float> logVal = reduceBuf_.AllocTensor<float>();
            // logVal[0] = sumExp;  // not allowed, use SetValue
            logVal.SetValue(0, sumExp);
            AscendC::Log(logVal, logVal, 1);
            float logSumExp = logVal.GetValue(0) + rowMax;
            reduceBuf_.FreeTensor(logVal);

            // Pass 3: compute y = x - logSumExp
            computeRowOutput(rowOffset, logSumExp);
        }
    }

private:
    __aicore__ float computeRowMax(uint32_t rowOffset)
    {
        float maxVal = NEG_INF;
        uint32_t remaining = innerDim_;
        uint32_t offset = rowOffset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> maxBuf = reduceBuf_.AllocTensor<float>();
            AscendC::ReduceMax(maxBuf, xLocal, curLen, 1);  // 4 args: dst, src, lastAxisLength, repeatTimes
            float tileMax = maxBuf.GetValue(0);
            reduceBuf_.FreeTensor(maxBuf);
            inQueueX_.FreeTensor(xLocal);

            if (tileMax > maxVal) maxVal = tileMax;

            offset += curLen;
            remaining -= curLen;
        }
        return maxVal;
    }

    __aicore__ float computeRowSumExp(uint32_t rowOffset, float maxVal)
    {
        float sumExp = 0.0f;
        uint32_t remaining = innerDim_;
        uint32_t offset = rowOffset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // create a 1-element scalar tensor filled with maxVal
            AscendC::LocalTensor<float> scalarMax = scalarQueue_.AllocTensor<float>();
            scalarMax.SetValue(0, maxVal);
            // broadcast scalarMax to tile length
            AscendC::LocalTensor<float> maxBroadcast = broadcastBuf_.AllocTensor<float>();
            AscendC::BroadCast(maxBroadcast, scalarMax, curLen);  // Broadcast scalar to tensor of length curLen

            // compute shifted = x - maxVal
            AscendC::LocalTensor<float> shifted = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(shifted, xLocal, maxBroadcast, curLen);

            // exp(shifted)
            AscendC::LocalTensor<float> expVal = expQueue_.AllocTensor<float>();
            AscendC::Exp(expVal, shifted, curLen);

            // reduce sum
            AscendC::LocalTensor<float> sumBuf = reduceBuf_.AllocTensor<float>();
            AscendC::ReduceSum(sumBuf, expVal, curLen, 1);
            float tileSum = sumBuf.GetValue(0);

            // free tensors
            reduceBuf_.FreeTensor(sumBuf);
            expQueue_.FreeTensor(expVal);
            tmpQueue_.FreeTensor(shifted);
            broadcastBuf_.FreeTensor(maxBroadcast);
            scalarQueue_.FreeTensor(scalarMax);
            inQueueX_.FreeTensor(xLocal);

            sumExp += tileSum;

            offset += curLen;
            remaining -= curLen;
        }
        return sumExp;
    }

    __aicore__ void computeRowOutput(uint32_t rowOffset, float logSumExp)
    {
        uint32_t remaining = innerDim_;
        uint32_t offset = rowOffset;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // broadcast logSumExp
            AscendC::LocalTensor<float> scalarLog = scalarQueue_.AllocTensor<float>();
            scalarLog.SetValue(0, logSumExp);
            AscendC::LocalTensor<float> logBroadcast = broadcastBuf_.AllocTensor<float>();
            AscendC::BroadCast(logBroadcast, scalarLog, curLen);

            // y = x - logSumExp
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Sub(yLocal, xLocal, logBroadcast, curLen);

            // write back
            outQueueY_.EnQue(yLocal);
            AscendC::LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yDeque, curLen);
            outQueueY_.FreeTensor(yDeque);

            broadcastBuf_.FreeTensor(logBroadcast);
            scalarQueue_.FreeTensor(scalarLog);
            inQueueX_.FreeTensor(xLocal);

            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> expQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;   // 1-element scalar
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> reduceBuf_;     // 1-element reduction result
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> broadcastBuf_;  // tile-sized broadcast

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t innerDim_ {0};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.innerDim);
    op.Process();
}
