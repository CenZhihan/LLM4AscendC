#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    // Use StorageShape to access GetOriginShape
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const gert::StorageShape *weightShape = context->GetInputShape(1);
    uint32_t totalDim = static_cast<uint32_t>(inputShape->GetOriginShape().GetShapeSize());
    uint32_t innerSize = static_cast<uint32_t>(weightShape->GetOriginShape().GetShapeSize());
    uint32_t totalBatchSize = totalDim / innerSize;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = 1.0f;
    if (attrs->GetAttrNum() > 0) {
        eps = *attrs->GetAttrPointer<float>(0);
    }

    // UB usage per batch item: x (input), y (output), diff, mean, var, weight, bias
    // This is roughly 6 * innerSize floats. Compute conservative tile size.
    uint32_t maxInner = UB_BUDGET_BYTES / (6 * sizeof(float));
    maxInner = AlignDown(maxInner, ALIGN_ELEMS);
    if (maxInner < ALIGN_ELEMS) maxInner = ALIGN_ELEMS;
    if (maxInner > HARD_CAP_ELEMS) maxInner = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // For simplicity, assume innerSize fits in UB (otherwise more complex tiling needed)
    // Use tileBatchSize = 1
    uint32_t tileBatchSize = 1;

    uint32_t blockDim = CeilDiv(totalBatchSize, tileBatchSize);
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalBatchSize(totalBatchSize);
    tiling.set_innerSize(innerSize);
    tiling.set_tileBatchSize(tileBatchSize);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalBatchSize, uint32_t innerSize,
                                uint32_t tileBatchSize, float eps)
    {
        totalBatchSize_ = totalBatchSize;
        innerSize_ = innerSize;
        tileBatchSize_ = tileBatchSize;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalBatchSize * innerSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalBatchSize * innerSize);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, innerSize);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, innerSize);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, innerSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, innerSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, innerSize * sizeof(float));      // diff buffer
        pipe_.InitBuffer(meanQueue_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(varQueue_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(weightQueue_, 1, innerSize * sizeof(float));
        pipe_.InitBuffer(biasQueue_, 1, innerSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalBatchSize_ == 0 || innerSize_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Load weight and bias once
        AscendC::LocalTensor<float> weightLocal = weightQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> biasLocal = biasQueue_.AllocTensor<float>();
        AscendC::DataCopy(weightLocal, weightGm_, innerSize_);
        AscendC::DataCopy(biasLocal, biasGm_, innerSize_);
        weightQueue_.EnQue(weightLocal);
        biasQueue_.EnQue(biasLocal);
        weightLocal = weightQueue_.DeQue<float>();
        biasLocal = biasQueue_.DeQue<float>();

        uint32_t startBatch = blockIdx * tileBatchSize_;
        for (uint32_t b = startBatch; b < totalBatchSize_; b += blockNum * tileBatchSize_) {
            uint32_t batchItems = tileBatchSize_;
            if (b + batchItems > totalBatchSize_)
                batchItems = totalBatchSize_ - b;

            for (uint32_t i = 0; i < batchItems; i++) {
                uint32_t curBatchIdx = b + i;
                uint32_t gmOffset = curBatchIdx * innerSize_;

                // Load x
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[gmOffset], innerSize_);
                inQueueX_.EnQue(xLocal);
                xLocal = inQueueX_.DeQue<float>();

                // --- mean ---
                AscendC::LocalTensor<float> meanLocal = meanQueue_.AllocTensor<float>();
                AscendC::ReduceSum(meanLocal, xLocal, 0.0f, innerSize_);
                AscendC::Muls(meanLocal, meanLocal, 1.0f / static_cast<float>(innerSize_), 1);
                meanQueue_.EnQue(meanLocal);
                meanLocal = meanQueue_.DeQue<float>();

                // --- diff = x - mean (scalar) ---
                AscendC::LocalTensor<float> diffLocal = tmpQueue_.AllocTensor<float>();
                // Subtract scalar mean from each element
                AscendC::Subs(diffLocal, xLocal, meanLocal[0], innerSize_);

                // --- variance = sum(diff^2) / innerSize ---
                AscendC::Mul(diffLocal, diffLocal, diffLocal, innerSize_);
                AscendC::LocalTensor<float> varLocal = varQueue_.AllocTensor<float>();
                AscendC::ReduceSum(varLocal, diffLocal, 0.0f, innerSize_);
                AscendC::Muls(varLocal, varLocal, 1.0f / static_cast<float>(innerSize_), 1);
                AscendC::Adds(varLocal, varLocal, eps_, 1);
                varQueue_.EnQue(varLocal);
                varLocal = varQueue_.DeQue<float>();

                // --- normalized = diff / sqrt(var) --- (reuse diffLocal for normalized)
                AscendC::Sqrt(varLocal, varLocal, 1);
                AscendC::Reciprocal(varLocal, varLocal, 1);
                AscendC::Mul(diffLocal, diffLocal, varLocal, innerSize_);

                // --- y = weight * normalized + bias ---
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Mul(yLocal, weightLocal, diffLocal, innerSize_);
                AscendC::Add(yLocal, yLocal, biasLocal, innerSize_);

                tmpQueue_.FreeTensor(diffLocal);
                meanQueue_.FreeTensor(meanLocal);
                varQueue_.FreeTensor(varLocal);

                outQueueY_.EnQue<float>(yLocal);
                yLocal = outQueueY_.DeQue<float>();

                AscendC::DataCopy(yGm_[gmOffset], yLocal, innerSize_);
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
            }
        }

        weightQueue_.FreeTensor(weightLocal);
        biasQueue_.FreeTensor(biasLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;      // diff buffer
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> meanQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> varQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> weightQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalBatchSize_ {0};
    uint32_t innerSize_ {0};
    uint32_t tileBatchSize_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, y, weight, bias,
            tiling_data.totalBatchSize, tiling_data.innerSize,
            tiling_data.tileBatchSize, tiling_data.eps);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        self.normalized_shape = normalized_shape
        self.weight = torch.nn.Parameter(torch.randn(normalized_shape))
        self.bias = torch.nn.Parameter(torch.zeros(normalized_shape))
        self.eps = 1e-5

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias, self.eps)
