host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;

    const auto &inputShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalElements = static_cast<uint32_t>(inputShape.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    uint32_t rowLength = static_cast<uint32_t>(inputShape.GetDim(dim));
    uint32_t totalRows = (rowLength > 0) ? (totalElements / rowLength) : 1;

    uint32_t tileLen = rowLength;
    if (tileLen > HARD_CAP_ELEMS) {
        tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    }

    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // 3 * float + bool (estimated 20 bytes per element)
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (tileLen > maxTile) tileLen = maxTile;

    uint32_t tilesPerRow = CeilDiv(rowLength, tileLen);
    uint32_t totalTiles = totalRows * tilesPerRow;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalElements(totalElements);
    tiling.set_rowLength(rowLength);
    tiling.set_totalRows(totalRows);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalElements, uint32_t rowLength,
                                uint32_t totalRows, int32_t dim)
    {
        totalElems_ = totalElements;
        rowLength_ = rowLength;
        totalRows_ = totalRows;
        dim_ = dim;   // not used, assume last dimension

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElems_);
        maskGm_.SetGlobalBuffer((__gm__ bool*)mask, totalElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElems_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, rowLength_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, rowLength_ * sizeof(bool));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, rowLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, rowLength_ * sizeof(float));  // for mask as float
        pipe_.InitBuffer(tmpQueue2_, 1, rowLength_ * sizeof(float)); // for scanned result (optional)
    }

    __aicore__ inline void Process()
    {
        if (totalElems_ == 0 || rowLength_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid stride over rows
        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            uint32_t base = row * rowLength_;
            CopyInRow(base);
            ComputeRow();
            CopyOutRow(base);
        }
    }

private:
    __aicore__ inline void CopyInRow(uint32_t base)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<bool> maskLocal = inQueueMask_.AllocTensor<bool>();
        AscendC::DataCopy(xLocal, xGm_[base], rowLength_);
        AscendC::DataCopy(maskLocal, maskGm_[base], rowLength_);
        inQueueX_.EnQue(xLocal);
        inQueueMask_.EnQue(maskLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<bool> maskLocal = inQueueMask_.DeQue<bool>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> maskFloat = tmpQueue_.AllocTensor<float>();

        // Cast bool mask to float
        AscendC::Cast(maskFloat, maskLocal, AscendC::RoundMode::CAST_ROUND, rowLength_);

        // Multiply x by maskFloat
        AscendC::Mul(yLocal, xLocal, maskFloat, rowLength_);

        // Compute prefix sum manually: y[i] = sum_{j=0..i} y[j]
        // Use a simple sequential loop with local scalar (not efficient but correct)
        // We'll use a temporary local tensor for accumulation.
        AscendC::LocalTensor<float> work = tmpQueue2_.AllocTensor<float>();
        // Copy first element
        AscendC::DataCopy(work, yLocal[0], 1); // work[0] = y[0]
        for (uint32_t i = 1; i < rowLength_; i++) {
            float prev;
            AscendC::GetValue<float, float>(prev, work, i - 1);
            float cur;
            AscendC::GetValue<float, float>(cur, yLocal, i);
            float sum = prev + cur;
            AscendC::SetValue<float, float>(work, i, sum);
        }
        // Copy result back to yLocal
        AscendC::DataCopy(yLocal, work, rowLength_);

        tmpQueue2_.FreeTensor(work);
        tmpQueue_.FreeTensor(maskFloat);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueMask_.FreeTensor(maskLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t base)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[base], yLocal, rowLength_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;   // for mask float
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;  // for prefix sum work

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<bool> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalElems_ {0};
    uint32_t rowLength_ {0};
    uint32_t totalRows_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.totalElements, tiling_data.rowLength, tiling_data.totalRows, tiling_data.dim);
    op.Process();
}
"""

#include <torch/library.h>

at::Tensor masked_cumsum_impl_npu(const at::Tensor& self, const at::Tensor& mask, int64_t dim) {
    // mask is bool; kernel accepts bool mask directly
    at::Tensor result = at::empty_like(self);
    EXEC_NPU_CMD(aclnnMaskedCumsumCustom, self, mask, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("masked_cumsum_custom", &masked_cumsum_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("masked_cumsum_custom", &masked_cumsum_impl_npu, "Masked cumulative sum (MaskedCumsumCustom on NPU)");
