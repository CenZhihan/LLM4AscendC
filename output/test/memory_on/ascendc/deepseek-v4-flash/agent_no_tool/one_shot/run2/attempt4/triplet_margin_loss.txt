// Single block processes all samples; tileSamples based on UB capacity.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t HARD_CAP_SAMPLES = 4096;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TripletMarginLossCustomTilingData tiling;
    const auto &originShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalSamples = static_cast<uint32_t>(originShape.GetDim(0));
    const uint32_t sampleDim   = static_cast<uint32_t>(originShape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *marginPtr = attrs->GetAttrPointer<float>(0);
    float margin = marginPtr ? *marginPtr : 1.0f;

    // UB usage per sample: 5 * sampleDim floats
    uint32_t bytesPerSample = 5 * sampleDim * sizeof(float);
    uint32_t maxSamples = (bytesPerSample > 0) ? (UB_BUDGET_BYTES / bytesPerSample) : 1;
    if (maxSamples > HARD_CAP_SAMPLES) maxSamples = HARD_CAP_SAMPLES;
    if (maxSamples < 1) maxSamples = 1;
    uint32_t tileSamples = (totalSamples < maxSamples) ? totalSamples : maxSamples;
    if (tileSamples == 0) tileSamples = 1;

    context->SetBlockDim(1);

    tiling.set_totalSamples(totalSamples);
    tiling.set_sampleDim(sampleDim);
    tiling.set_tileSamples(tileSamples);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelTripletMarginLoss {
public:
    __aicore__ inline KernelTripletMarginLoss() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t totalSamples, uint32_t sampleDim,
                                uint32_t tileSamples, float margin)
    {
        totalSamples_ = totalSamples;
        sampleDim_ = sampleDim;
        tileSamples_ = tileSamples;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalSamples_ * sampleDim_);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalSamples_ * sampleDim_);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalSamples_ * sampleDim_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inQueAnchor_, 1, tileSamples_ * sampleDim_ * sizeof(float));
        pipe_.InitBuffer(inQuePositive_, 1, tileSamples_ * sampleDim_ * sizeof(float));
        pipe_.InitBuffer(inQueNegative_, 1, tileSamples_ * sampleDim_ * sizeof(float));
        pipe_.InitBuffer(tmpQue1_, 1, tileSamples_ * sampleDim_ * sizeof(float));
        pipe_.InitBuffer(tmpQue2_, 1, tileSamples_ * sampleDim_ * sizeof(float));
        pipe_.InitBuffer(tmpQue3_, 1, tileSamples_ * sizeof(float));
        pipe_.InitBuffer(tmpQue4_, 1, tileSamples_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalSamples_ == 0 || sampleDim_ == 0) return;

        float totalLoss = 0.0f;
        uint32_t processed = 0;

        while (processed < totalSamples_) {
            uint32_t curTileSamples = (processed + tileSamples_ <= totalSamples_) ?
                                      tileSamples_ : totalSamples_ - processed;
            uint32_t curTileSize = curTileSamples * sampleDim_;

            AscendC::LocalTensor<float> anchorLocal = inQueAnchor_.AllocTensor<float>();
            AscendC::DataCopy(anchorLocal, anchorGm_[processed * sampleDim_], curTileSize);
            AscendC::LocalTensor<float> positiveLocal = inQuePositive_.AllocTensor<float>();
            AscendC::DataCopy(positiveLocal, positiveGm_[processed * sampleDim_], curTileSize);
            AscendC::LocalTensor<float> negativeLocal = inQueNegative_.AllocTensor<float>();
            AscendC::DataCopy(negativeLocal, negativeGm_[processed * sampleDim_], curTileSize);

            AscendC::LocalTensor<float> diffPos = tmpQue1_.AllocTensor<float>();
            AscendC::Sub(diffPos, anchorLocal, positiveLocal, curTileSize);
            AscendC::LocalTensor<float> diffNeg = tmpQue2_.AllocTensor<float>();
            AscendC::Sub(diffNeg, anchorLocal, negativeLocal, curTileSize);

            AscendC::Mul(diffPos, diffPos, diffPos, curTileSize);
            AscendC::Mul(diffNeg, diffNeg, diffNeg, curTileSize);

            AscendC::LocalTensor<float> posDist = tmpQue3_.AllocTensor<float>();
            AscendC::LocalTensor<float> negDist = tmpQue4_.AllocTensor<float>();

            for (uint32_t s = 0; s < curTileSamples; ++s) {
                float posSum = 0.0f, negSum = 0.0f;
                for (uint32_t d = 0; d < sampleDim_; ++d) {
                    uint32_t idx = s * sampleDim_ + d;
                    posSum += diffPos(idx);
                    negSum += diffNeg(idx);
                }
                float sampleLoss = posSum - negSum + margin_;
                if (sampleLoss < 0.0f) sampleLoss = 0.0f;
                totalLoss += sampleLoss;
            }

            inQueAnchor_.FreeTensor(anchorLocal);
            inQuePositive_.FreeTensor(positiveLocal);
            inQueNegative_.FreeTensor(negativeLocal);
            tmpQue1_.FreeTensor(diffPos);
            tmpQue2_.FreeTensor(diffNeg);
            tmpQue3_.FreeTensor(posDist);
            tmpQue4_.FreeTensor(negDist);

            processed += curTileSamples;
        }

        // Fix: avoid direct cast from uint32_t to float
        float totalSamplesFloat = totalSamples_;
        float meanLoss = totalLoss / totalSamplesFloat;
        lossGm_.SetValue(0, meanLoss);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQuePositive_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueNegative_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQue3_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQue4_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalSamples_;
    uint32_t sampleDim_;
    uint32_t tileSamples_;
    float margin_;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive,
                                                                  GM_ADDR negative, GM_ADDR loss,
                                                                  GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLoss op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalSamples, tiling_data.sampleDim,
            tiling_data.tileSamples, tiling_data.margin);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'triplet_margin_loss_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, anchor, positive, negative, margin):
        # Convert margin tensor to float
        m = float(margin.reshape(-1)[0].item()) if margin.numel() else 0.0
        fn = getattr(custom_ops_lib, _FN)
        loss = fn(anchor, positive, negative, m)
        return loss.squeeze()
