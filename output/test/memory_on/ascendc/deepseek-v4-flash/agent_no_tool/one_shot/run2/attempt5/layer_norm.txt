#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>
#include <cstdint>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const std::vector<int64_t> *normShapeVec = attrs->GetAttrPointer<std::vector<int64_t>>(0);
    const float eps = attrs->GetAttrPointer<float>(1)[0];

    uint32_t normalizedSize = 1;
    for (auto dim : *normShapeVec) {
        normalizedSize *= static_cast<uint32_t>(dim);
    }

    // tileLen: UB budget = 1 tile input + 1 tile output + 1 tile for reduction temp = 3 * tileLen * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // Prefer tileLen <= normalizedSize to keep per-row tiling reasonable
    uint32_t tileLen = (maxTile < normalizedSize) ? maxTile : normalizedSize;
    if (tileLen > totalLength) tileLen = AlignDown(totalLength, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = totalLength;

    // Block dim: number of independent rows processed concurrently
    // each block processes rows in grid-stride, so only need few blocks to hide latency
    uint32_t blockDim = 16;  // typical for Ascend 910B
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_normalizedSize(normalizedSize);
    tiling.set_epsilon(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"
#include "reduce_traits.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t normalizedSize, float epsilon)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        normSize_ = normalizedSize;
        eps_ = epsilon;
        rowStride_ = normSize_;  // each row has normSize_ elements
        numRows_ = (normSize_ > 0) ? totalLen_ / normSize_ : 0;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, 1, tileLen_ * sizeof(float));         // only one buffer, reused
        pipe_.InitBuffer(outQueueY_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpReduce_, 1, sizeof(float));                   // for reduction result (scalar)
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || normSize_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();

        // grid-stride over rows
        for (uint32_t row = blockIdx; row < numRows_; row += blockNum) {
            uint32_t baseOffset = row * rowStride_;

            // ---- Phase 1: compute mean and variance ----
            float sum = 0.0f;
            float sumSq = 0.0f;
            uint32_t offset = baseOffset;
            uint32_t remain = normSize_;

            while (remain > 0) {
                uint32_t curLen = (remain > tileLen_) ? tileLen_ : remain;
                // copy in
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curLen);

                // reduce sum
                AscendC::LocalTensor<float> reduceBuf = tmpReduce_.AllocTensor<float>();
                AscendC::VecReduceSum(reduceBuf, xLocal, curLen);
                sum += reduceBuf(0);

                // reduce sum of squares
                AscendC::Mul(reduceBuf, xLocal, xLocal, curLen);
                AscendC::VecReduceSum(reduceBuf, reduceBuf, curLen);
                sumSq += reduceBuf(0);

                tmpReduce_.FreeTensor(reduceBuf);
                inQueueX_.FreeTensor(xLocal);

                offset += curLen;
                remain -= curLen;
            }

            float mean = sum / (float)normSize_;
            float variance = sumSq / (float)normSize_ - mean * mean;
            float invStd = 1.0f / sqrtf(variance + eps_);

            // ---- Phase 2: normalize ----
            offset = baseOffset;
            remain = normSize_;
            while (remain > 0) {
                uint32_t curLen = (remain > tileLen_) ? tileLen_ : remain;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[offset], curLen);

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // y = (x - mean) * invStd
                AscendC::Sub(yLocal, xLocal, mean, curLen);
                AscendC::Muls(yLocal, yLocal, invStd, curLen);

                AscendC::DataCopy(yGm_[offset], yLocal, curLen);

                inQueueX_.FreeTensor(xLocal);
                outQueueY_.FreeTensor(yLocal);

                offset += curLen;
                remain -= curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpReduce_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t normSize_ {0};
    uint32_t rowStride_ {0};
    uint32_t numRows_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.normalizedSize,
            tiling_data.epsilon);
    op.Process();
}
