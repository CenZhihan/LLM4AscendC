#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>

namespace optiling {

// UB model: each row load x and mask, plus output y, plus temporary work buffer (1 row)
// Each row: bytes = rowLength * (sizeof(float) + sizeof(bool) + sizeof(float)) + rowLength * sizeof(float) for tmp
// Simplification: assume float for x and y, bool for mask. Total per row = 3 * sizeof(float) * rowLength + 1 * sizeof(bool) * rowLength ~ 13 * rowLength bytes if bool stored as byte.
// But bool is 1 byte, so total ~ 9 * rowLength bytes.
// To be conservative, use 12 * rowLength bytes.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;
    const auto &xShape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalElements = static_cast<uint32_t>(xShape.GetShapeSize());
    const int32_t dim = static_cast<int32_t>(context->GetAttrs()->GetAttrPointer<int32_t>(0)[0]);

    // Compute rowLength (size of dim dimension) and totalRows (product of other dims)
    const int ndim = xShape.GetDimNum();
    uint32_t rowLength = 1;
    uint32_t totalRows = 1;
    for (int i = 0; i < ndim; ++i) {
        uint32_t d = static_cast<uint32_t>(xShape.GetDim(i));
        if (i == dim) {
            rowLength = d;
        } else {
            totalRows *= d;
        }
    }

    // Determine rows per block based on UB budget
    uint32_t bytesPerRow = rowLength * (sizeof(float) + sizeof(bool) + sizeof(float)) + rowLength * sizeof(float); // x, mask, y, tmp
    // Ensure at least one row per block
    uint32_t maxRows = UB_BUDGET_BYTES / bytesPerRow;
    if (maxRows == 0) maxRows = 1;

    // Align rows to keep things simple
    uint32_t rowsPerBlock = std::min(maxRows, totalRows);
    rowsPerBlock = AlignDown(rowsPerBlock, 1); // no special alignment needed

    uint32_t blockDim = CeilDiv(totalRows, rowsPerBlock);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalElements(totalElements);
    tiling.set_rowLength(rowLength);
    tiling.set_totalRows(totalRows);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
