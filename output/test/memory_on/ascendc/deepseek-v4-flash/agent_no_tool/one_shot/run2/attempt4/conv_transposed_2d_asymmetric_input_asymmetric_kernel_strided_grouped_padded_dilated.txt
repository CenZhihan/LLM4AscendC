kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTransposedConv2D {
public:
    __aicore__ inline KernelTransposedConv2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t n, uint32_t c_in, uint32_t h_in, uint32_t w_in,
                                uint32_t c_out, uint32_t k_h, uint32_t k_w,
                                uint32_t h_out, uint32_t w_out,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t padding_h, uint32_t padding_w,
                                uint32_t groups, uint32_t dilation_h, uint32_t dilation_w,
                                uint32_t totalLen, uint32_t tileLen)
    {
        n_ = n;
        c_in_ = c_in;
        h_in_ = h_in;
        w_in_ = w_in;
        c_out_ = c_out;
        k_h_ = k_h;
        k_w_ = k_w;
        h_out_ = h_out;
        w_out_ = w_out;
        stride_h_ = stride_h;
        stride_w_ = stride_w;
        padding_h_ = padding_h;
        padding_w_ = padding_w;
        groups_ = groups;
        dilation_h_ = dilation_h;
        dilation_w_ = dilation_w;
        totalLen_ = totalLen;
        tileLen_ = tileLen;

        c_in_per_group_ = c_in_ / groups_;
        c_out_per_group_ = c_out_ / groups_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * c_in_ * h_in_ * w_in_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, c_out_ * c_in_per_group_ * k_h_ * k_w_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        uint32_t tileOutElems = tileLen_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileOutElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOutElems * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileOutElems * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t chunkSize = tileLen_;
        uint32_t startElem = blockIdx * chunkSize;
        uint32_t endElem = startElem + chunkSize;
        if (endElem > totalLen_) endElem = totalLen_;
        if (startElem >= totalLen_) return;

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        AscendC::Duplicate<float>(yLocal, 0.0f, chunkSize);

        // Simple nested-loop convolution per output element
        for (uint32_t i = 0; i < chunkSize; i++) {
            uint32_t globalIdx = startElem + i;
            uint32_t outIdxN = globalIdx / (c_out_ * h_out_ * w_out_);
            uint32_t rem = globalIdx % (c_out_ * h_out_ * w_out_);
            uint32_t outIdxC = rem / (h_out_ * w_out_);
            rem = rem % (h_out_ * w_out_);
            uint32_t outIdxH = rem / w_out_;
            uint32_t outIdxW = rem % w_out_;

            uint32_t group = outIdxC / c_out_per_group_;
            uint32_t outCInGroup = outIdxC % c_out_per_group_;

            for (uint32_t cInIdx = 0; cInIdx < c_in_per_group_; cInIdx++) {
                uint32_t inputC = group * c_in_per_group_ + cInIdx;
                for (uint32_t kh = 0; kh < k_h_; kh++) {
                    for (uint32_t kw = 0; kw < k_w_; kw++) {
                        int32_t inH = (static_cast<int32_t>(outIdxH) - static_cast<int32_t>(dilation_h_ * kh) + static_cast<int32_t>(padding_h_));
                        int32_t inW = (static_cast<int32_t>(outIdxW) - static_cast<int32_t>(dilation_w_ * kw) + static_cast<int32_t>(padding_w_));
                        if (inH % static_cast<int32_t>(stride_h_) != 0 || inW % static_cast<int32_t>(stride_w_) != 0) continue;
                        inH /= static_cast<int32_t>(stride_h_);
                        inW /= static_cast<int32_t>(stride_w_);
                        if (inH < 0 || inH >= static_cast<int32_t>(h_in_) || inW < 0 || inW >= static_cast<int32_t>(w_in_)) continue;

                        uint32_t inputOffset = (outIdxN * c_in_ + inputC) * h_in_ * w_in_ + inH * w_in_ + inW;
                        uint32_t weightOffset = (outIdxC * c_in_per_group_ + cInIdx) * k_h_ * k_w_ + kh * k_w_ + kw;

                        float xVal = xGm_.GetValue(inputOffset);
                        float wVal = wGm_.GetValue(weightOffset);
                        yLocal.SetValue(i, yLocal.GetValue(i) + xVal * wVal);
                    }
                }
            }
        }

        AscendC::DataCopy(yGm_[startElem], yLocal, chunkSize);
        tmpQueue_.FreeTensor(tmpLocal);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    uint32_t n_, c_in_, h_in_, w_in_;
    uint32_t c_out_, k_h_, k_w_, h_out_, w_out_;
    uint32_t stride_h_, stride_w_, padding_h_, padding_w_;
    uint32_t groups_, dilation_h_, dilation_w_;
    uint32_t c_in_per_group_, c_out_per_group_;
    uint32_t totalLen_, tileLen_;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTransposedConv2D op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c_in, tiling_data.h_in, tiling_data.w_in,
            tiling_data.c_out, tiling_data.k_h, tiling_data.k_w,
            tiling_data.h_out, tiling_data.w_out,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.padding_h, tiling_data.padding_w,
            tiling_data.groups, tiling_data.dilation_h, tiling_data.dilation_w,
            tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
