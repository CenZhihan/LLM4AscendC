host_operator_src = """
#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
  return (a + b - 1u) / b;
}

static inline uint32_t ComputeOutSize(uint32_t inSize, uint32_t kernel, uint32_t stride,
                                      uint32_t pad, bool ceilMode) {
  // dilation = 1
  int32_t out = static_cast<int32_t>(inSize + 2 * pad - kernel) / static_cast<int32_t>(stride) + 1;
  if (ceilMode) {
    if ((out - 1) * stride >= inSize + pad) {
      out--;
    }
  }
  // Ensure out >= 1
  if (out < 1) out = 1;
  return static_cast<uint32_t>(out);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  MaxPooling3dCustomTilingData tiling;

  const gert::Shape* inShape = context->GetInputShape(0);
  const auto& originShape = inShape->GetOriginShape();
  size_t dimNum = originShape.GetDimNum();
  // Expect 5D (N, C, D, H, W)
  if (dimNum != 5) {
    return ge::GRAPH_FAILED;
  }
  uint32_t n = static_cast<uint32_t>(originShape.GetDim(0));
  uint32_t c = static_cast<uint32_t>(originShape.GetDim(1));
  uint32_t inD = static_cast<uint32_t>(originShape.GetDim(2));
  uint32_t inH = static_cast<uint32_t>(originShape.GetDim(3));
  uint32_t inW = static_cast<uint32_t>(originShape.GetDim(4));

  const gert::RuntimeAttrs* attrs = context->GetAttrs();
  uint32_t kernel_size = attrs->GetAttrPointer<uint32_t>(0)[0];   // assume scalar
  uint32_t stride = attrs->GetAttrPointer<uint32_t>(1)[0];
  uint32_t padding = attrs->GetAttrPointer<uint32_t>(2)[0];
  bool returnIndices = attrs->GetAttrPointer<bool>(3)[0];
  bool ceilMode = attrs->GetAttrPointer<bool>(4)[0];

  uint32_t outD = ComputeOutSize(inD, kernel_size, stride, padding, ceilMode);
  uint32_t outH = ComputeOutSize(inH, kernel_size, stride, padding, ceilMode);
  uint32_t outW = ComputeOutSize(inW, kernel_size, stride, padding, ceilMode);

  uint32_t outputLength = n * c * outD * outH * outW;
  uint32_t inputLength = n * c * inD * inH * inW;

  // Tile: each block processes tileLength output elements.
  // Set tileLength based on UB budget: we need local space for one output tile.
  // Each output element we compute a window of up to kernel_size^3 floats.
  // We store localY of tileLength floats.
  // Also we may need temp for window? We'll compute via scalar loads, so only output tile.
  // UB budget: reserve 192KB = 49152 floats. Use a fraction for safety.
  constexpr uint32_t UB_FLOATS = 49152;
  uint32_t tileLength = UB_FLOATS / 2;  // e.g., 24576
  if (tileLength > outputLength) tileLength = outputLength;
  // Align to 64?
  tileLength = (tileLength / 64) * 64;
  if (tileLength == 0) tileLength = 1;

  // Adaptive blockDim
  uint32_t numTiles = (tileLength == 0) ? 1 : CeilDiv(outputLength, tileLength);
  uint32_t blockDim = numTiles;
  if (blockDim < 8) blockDim = 8;
  if (blockDim > 24) blockDim = 24;

  context->SetBlockDim(blockDim);

  tiling.set_inputLength(inputLength);
  tiling.set_outputLength(outputLength);
  tiling.set_blockDim(blockDim);
  tiling.set_tileLength(tileLength);
  tiling.set_n(n);
  tiling.set_c(c);
  tiling.set_inD(inD);
  tiling.set_inH(inH);
  tiling.set_inW(inW);
  tiling.set_outD(outD);
  tiling.set_outH(outH);
  tiling.set_outW(outW);
  tiling.set_kd(kernel_size);
  tiling.set_kh(kernel_size);
  tiling.set_kw(kernel_size);
  tiling.set_stride(stride);
  tiling.set_pad(padding);
  tiling.set_returnIndices(returnIndices ? 1u : 0u);
  tiling.set_ceilMode(ceilMode ? 1u : 0u);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;

  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape* inShape = context->GetInputShape(0);
  size_t dimNum = inShape->GetOriginShape().GetDimNum();
  if (dimNum != 5) {
    return GRAPH_FAILED;
  }
  const auto& origin = inShape->GetOriginShape();
  uint32_t n = static_cast<uint32_t>(origin.GetDim(0));
  uint32_t c = static_cast<uint32_t>(origin.GetDim(1));
  uint32_t inD = static_cast<uint32_t>(origin.GetDim(2));
  uint32_t inH = static_cast<uint32_t>(origin.GetDim(3));
  uint32_t inW = static_cast<uint32_t>(origin.GetDim(4));

  const gert::RuntimeAttrs* attrs = context->GetAttrs();
  uint32_t kernel_size = attrs->GetAttrPointer<uint32_t>(0)[0];
  uint32_t stride = attrs->GetAttrPointer<uint32_t>(1)[0];
  uint32_t padding = attrs->GetAttrPointer<uint32_t>(2)[0];
  bool ceilMode = attrs->GetAttrPointer<bool>(4)[0];

  uint32_t outD = (inD + 2 * padding - kernel_size) / stride + 1;
  if (ceilMode && (static_cast<int32_t>(outD - 1) * stride >= static_cast<int32_t>(inD) + static_cast<int32_t>(padding)) ) {
      outD--;
  }
  uint32_t outH = (inH + 2 * padding - kernel_size) / stride + 1;
  if (ceilMode && (static_cast<int32_t>(outH - 1) * stride >= static_cast<int32_t>(inH) + static_cast<int32_t>(padding)) ) {
      outH--;
  }
  uint32_t outW = (inW + 2 * padding - kernel_size) / stride + 1;
  if (ceilMode && (static_cast<int32_t>(outW - 1) * stride >= static_cast<int32_t>(inW) + static_cast<int32_t>(padding)) ) {
      outW--;
  }
  // ensure minimum size 1
  if (outD < 1) outD = 1;
  if (outH < 1) outH = 1;
  if (outW < 1) outW = 1;

  gert::Shape* outShape = context->GetOutputShape(0);
  outShape->SetDimNum(5);
  outShape->SetDim(0, n);
  outShape->SetDim(1, c);
  outShape->SetDim(2, outD);
  outShape->SetDim(3, outH);
  outShape->SetDim(4, outW);
  return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
  const ge::DataType x_dtype = context->GetInputDataType(0);
  context->SetOutputDataType(0, x_dtype);
  return GRAPH_SUCCESS;
}

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
  __aicore__ inline KernelMaxPooling3dCustom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tilingData, tiling);
    xGm_.SetGlobalBuffer((__gm__ float*)x, tilingData.inputLength);
    yGm_.SetGlobalBuffer((__gm__ float*)y, tilingData.outputLength);

    n_ = tilingData.n;
    c_ = tilingData.c;
    inD_ = tilingData.inD;
    inH_ = tilingData.inH;
    inW_ = tilingData.inW;
    outD_ = tilingData.outD;
    outH_ = tilingData.outH;
    outW_ = tilingData.outW;
    kd_ = tilingData.kd;
    kh_ = tilingData.kh;
    kw_ = tilingData.kw;
    stride_ = tilingData.stride;
    pad_ = tilingData.pad;
    returnIndices_ = tilingData.returnIndices;
    ceilMode_ = tilingData.ceilMode;

    outputLength_ = tilingData.outputLength;
    tileLength_ = tilingData.tileLength;

    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
  }

    uint32_t blockIdx = GetBlockIdx();

      // Determine current tile length
      uint32_t curLen = outputLength_ - curStart;
      if (curLen > tileLength_) curLen = tileLength_;

      // Allocate output local tensor
      AscendC::LocalTensor<float> localY = outQueueY_.AllocTensor<float>();

      // Compute tile
      for (uint32_t i = 0; i < curLen; i++) {
 {
 {
