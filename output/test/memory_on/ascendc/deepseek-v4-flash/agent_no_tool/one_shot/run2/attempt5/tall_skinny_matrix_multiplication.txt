host_operator_src = '''
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: assume 192 KB per core = 49152 floats
// For each output tile (tileM x tileN):
//   A tile: tileM * K floats
//   B tile: K * tileN floats
//   C tile: tileM * tileN floats
//   Temp for partial sums: tileM * tileN (we can reuse C)
// total <= 3 * max(tileM*K, K*tileN, tileM*tileN) ≈ tileM*K + K*tileN + tileM*tileN
static constexpr uint32_t UB_BUDGET_FLOATS = 49152;
static constexpr uint32_t ALIGN = 256;
static constexpr uint32_t MAX_TILE = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape* shapeA = context->GetInputShape(0);
    const gert::Shape* shapeB = context->GetInputShape(1);
    uint32_t M = shapeA->GetOriginShape().GetDim(0);
    uint32_t K = shapeA->GetOriginShape().GetDim(1);
    uint32_t N = shapeB->GetOriginShape().GetDim(1);

    // Simple heuristic: try tileN = N (if small), then compute tileM
    uint32_t tileN = N;
    uint32_t tileM = 0;

    // need fit: tileM*K + K*tileN + tileM*tileN <= UB_BUDGET_FLOATS
    uint32_t needed = (tileM * K) + (K * tileN) + (tileM * tileN);
    if (needed > UB_BUDGET_FLOATS) {
        // reduce tileN
        while (tileN > 1) {
            tileN = AlignDown(tileN / 2, ALIGN);
            if (tileN == 0) tileN = ALIGN;
            // solve for max tileM: tileM*(K+tileN) <= UB_BUDGET_FLOATS - K*tileN
            if (K + tileN == 0) continue;
            tileM = (UB_BUDGET_FLOATS - K * tileN) / (K + tileN);
            tileM = AlignDown(tileM, ALIGN);
            if (tileM > MAX_TILE) tileM = AlignDown(MAX_TILE, ALIGN);
            if (tileM > 0) break;
        }
    } else {
        tileM = (UB_BUDGET_FLOATS - K * tileN) / (K + tileN);
        tileM = AlignDown(tileM, ALIGN);
        if (tileM > MAX_TILE) tileM = AlignDown(MAX_TILE, ALIGN);
    }
    if (tileM == 0) tileM = 1;
    if (tileN == 0) tileN = 1;
    if (tileM > M) tileM = AlignDown(M, ALIGN);
    if (tileM == 0) tileM = M;
    if (tileN > N) tileN = N;

    uint32_t numTilesM = (M + tileM - 1) / tileM;
    uint32_t numTilesN = (N + tileN - 1) / tileN;
    uint32_t totalTiles = numTilesM * numTilesN;
    uint32_t blockDim = totalTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    // C = (M, K) * (K, N) -> (M, N)
    uint32_t M = a_shape->GetOriginShape().GetDim(0);
    uint32_t N = b_shape->GetOriginShape().GetDim(1);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
}  // namespace ge

namespace ops {
class TallSkinnyMatrixMultiplicationCustom : public OpDef {
public:
    explicit TallSkinnyMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(TallSkinnyMatrixMultiplicationCustom);
}  // namespace ops

kernel_src = '''
#include "kernel_operator.h"

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)a, M * K);
        bGm_.SetGlobalBuffer((__gm__ float*)b, K * N);
        cGm_.SetGlobalBuffer((__gm__ float*)c, M * N);

        pipe_.InitBuffer(inQueueA_, 1, tileM_ * K_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, K_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, 1, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = GetBlockIdx();
        const uint32_t blockNum = GetBlockNum();

        uint32_t numTilesM = (M_ + tileM_ - 1) / tileM_;
        uint32_t numTilesN = (N_ + tileN_ - 1) / tileN_;
        uint32_t totalTiles = numTilesM * numTilesN;

        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            uint32_t tileRowIdx = t / numTilesN;
            uint32_t tileColIdx = t % numTilesN;

            uint32_t rowStart = tileRowIdx * tileM_;
            uint32_t rowEnd = rowStart + tileM_;
            if (rowEnd > M_) rowEnd = M_;
            uint32_t actualM = rowEnd - rowStart;

            uint32_t colStart = tileColIdx * tileN_;
            uint32_t colEnd = colStart + tileN_;
            if (colEnd > N_) colEnd = N_;
            uint32_t actualN = colEnd - colStart;

            // Load A tile [rowStart:rowEnd, 0:K)
            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            for (uint32_t r = 0; r < actualM; r++) {
                uint32_t srcOffset = (rowStart + r) * K_;
                 (aGm_[srcOffset]).CopyTo(aLocal[r * K_], K_);
            }
            inQueueA_.EnQue(aLocal);

            // Load B tile [0:K, colStart:colEnd)
            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
            for (uint32_t k = 0; k < K_; k++) {
                uint32_t srcOffset = k * N_ + colStart; // B is row-major (K*N)
                for (uint32_t c = 0; c < actualN; c++) {
                    bLocal[k * actualN + c] = bGm_[srcOffset + c];
                }
            }
            inQueueB_.EnQue(bLocal);

            // Compute C tile
            AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
            for (uint32_t r = 0; r < actualM; r++) {
                for (uint32_t c = 0; c < actualN; c++) {
                    float sum = 0.0f;
                    for (uint32_t k = 0; k < K_; k++) {
                        sum += aLocal[r * K_ + k] * bLocal[k * actualN + c];
                    }
                    cLocal[r * actualN + c] = sum;
                }
            }

            // Write back C tile
            for (uint32_t r = 0; r < actualM; r++) {
                uint32_t dstOffset = (rowStart + r) * N_ + colStart;
                for (uint32_t c = 0; c < actualN; c++) {
                    cGm_[dstOffset + c] = cLocal[r * actualN + c];
                }
            }

            outQueueC_.FreeTensor(cLocal);
            inQueueB_.FreeTensor(bLocal);
            inQueueA_.FreeTensor(aLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
'''
