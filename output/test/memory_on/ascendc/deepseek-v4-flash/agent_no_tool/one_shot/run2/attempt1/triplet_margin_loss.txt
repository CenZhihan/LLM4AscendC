// UB model per sample: approx 2*d floats for diff buffers + temp
// Let's assume d <= 512, so safe.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_SAMPLES = 1;  // no alignment needed for samples
static constexpr uint32_t HARD_CAP_SAMPLES = 4096;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TripletMarginLossCustomTilingData tiling;
    const gert::StorageType sType = context->GetInputShape(0)->GetStorageType();
    const uint32_t totalSamples = sType == ge::SHAPE_STORAGE_ND
        ? context->GetInputShape(0)->GetOriginShape().GetDim(0)
        : 0;
    const uint32_t sampleDim = sType == ge::SHAPE_STORAGE_ND
        ? context->GetInputShape(0)->GetOriginShape().GetDim(1)
        : 0;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *marginPtr = attrs->GetAttrPointer<float>(0);
    float margin = marginPtr ? *marginPtr : 1.0f;

    // Compute tileSamples: each sample needs 2*d floats for diff (simplified)
    uint32_t bufNeededPerSample = 2 * sampleDim * sizeof(float);
    uint32_t maxSamples = (bufNeededPerSample > 0) ? (UB_BUDGET_BYTES / bufNeededPerSample) : 1;
    if (maxSamples > HARD_CAP_SAMPLES) maxSamples = HARD_CAP_SAMPLES;
    if (maxSamples < 1) maxSamples = 1;
    uint32_t tileSamples = maxSamples;
    if (totalSamples < tileSamples) {
        tileSamples = totalSamples;
    }

    // blockDim: number of blocks to process all samples
    uint32_t numTiles = CeilDiv(totalSamples, tileSamples);
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;  // conservative

    context->SetBlockDim(blockDim);

    tiling.set_totalSamples(totalSamples);
    tiling.set_sampleDim(sampleDim);
    tiling.set_tileSamples(tileSamples);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLoss {
public:
    __aicore__ inline KernelTripletMarginLoss() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
                                uint32_t totalSamples, uint32_t sampleDim,
                                uint32_t tileSamples, float margin) {
        totalSamples_ = totalSamples;
        sampleDim_ = sampleDim;
        tileSamples_ = tileSamples;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalSamples_ * sampleDim_);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalSamples_ * sampleDim_);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalSamples_ * sampleDim_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, totalSamples_);

        // Buffer for diff (anchor - positive) and (anchor - negative)
        pipe_.InitBuffer(bufDiffPos_, 1, sampleDim_ * sizeof(float));
        pipe_.InitBuffer(bufDiffNeg_, 1, sampleDim_ * sizeof(float));
        // Temporary queue for results (d1, d2) per sample? We'll just use scalar temps.
        // No large queued buffers needed.
    }

    __aicore__ inline void Process() {
        if (totalSamples_ == 0 || sampleDim_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over sample groups
        uint32_t stride = blockNum * tileSamples_;
        uint32_t baseSample = blockIdx * tileSamples_;
        if (baseSample >= totalSamples_) return;

        while (baseSample < totalSamples_) {
            uint32_t samplesThisTile = tileSamples_;
            if (baseSample + samplesThisTile > totalSamples_) {
                samplesThisTile = totalSamples_ - baseSample;
            }

            // Process each sample in tile sequentially
            for (uint32_t s = 0; s < samplesThisTile; ++s) {
                uint32_t sampleIdx = baseSample + s;
                ComputeOneSample(sampleIdx);
            }

            baseSample += stride;
        }
    }

private:
    __aicore__ inline void ComputeOneSample(uint32_t sampleIdx) {
        // Get pointers to anchor, positive, negative for this sample
        uint32_t offset = sampleIdx * sampleDim_;
        AscendC::LocalTensor<float> diffPos = bufDiffPos_.AllocTensor<float>();
        AscendC::LocalTensor<float> diffNeg = bufDiffNeg_.AllocTensor<float>();

        // diffPos = anchor - positive
        AscendC::Sub(diffPos, anchorGm_[offset], positiveGm_[offset], sampleDim_);
        // diffNeg = anchor - negative
        AscendC::Sub(diffNeg, anchorGm_[offset], negativeGm_[offset], sampleDim_);

        // Compute squared sum for diffPos
        // We'll use vector square and then reduce add. AscendC has ReduceAdd but we need to do element-wise square first.
        AscendC::LocalTensor<float> sqPos;  // reuse diffPos buffer for square? We'll allocate separate.
        // Actually we can square in-place on diffPos, but we need original? Not anymore.
        AscendC::Mul(diffPos, diffPos, diffPos, sampleDim_); // now diffPos = (a-p)^2
        float d1 = AscendC::ReduceAdd(diffPos, sampleDim_);
        // sqrt
        d1 = AscendC::Sqrt(d1);

        // Same for diffNeg
        AscendC::Mul(diffNeg, diffNeg, diffNeg, sampleDim_);
        float d2 = AscendC::ReduceAdd(diffNeg, sampleDim_);
        d2 = AscendC::Sqrt(d2);

        // loss = max(0, d1 - d2 + margin)
        float lossVal = d1 - d2 + margin_;
        if (lossVal < 0.0f) lossVal = 0.0f;

        // Write to global memory
        AscendC::LocalScalar<float> lossScalar;
        lossScalar.SetValue(lossVal);
        AscendC::DataCopy(lossGm_[sampleIdx], lossScalar, 1);

        // Free temporary tensors
        bufDiffPos_.FreeTensor(diffPos);
        bufDiffNeg_.FreeTensor(diffNeg);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECIN> bufDiffPos_;   // temporary for diffPos
    AscendC::TBuf<AscendC::TPosition::VECIN> bufDiffNeg_;   // temporary for diffNeg

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalSamples_ {0};
    uint32_t sampleDim_ {0};
    uint32_t tileSamples_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLoss op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalSamples, tiling_data.sampleDim,
            tiling_data.tileSamples, tiling_data.margin);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'triplet_margin_loss_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, *inputs):
        anchor, positive, negative, margin_tensor = inputs
        # Convert margin tensor to float
        margin = float(margin_tensor.reshape(-1)[0].item()) if margin_tensor.numel() else 0.0
        fn = getattr(custom_ops_lib, _FN)
        per_sample_loss = fn(anchor, positive, negative, margin)
        # Return mean (to match original loss)
        return per_sample_loss.mean()
