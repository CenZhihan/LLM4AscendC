host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0);
    int32_t xDimNum = xShape->GetDimNum();
    const int64_t *xDim = xShape->GetDims();
    int64_t total = 1;
    for (int32_t i = 0; i < xDimNum; ++i) total *= xDim[i];

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    // Normalize dim
    if (dim < 0) dim += xDimNum;
    if (dim < 0 || dim >= xDimNum) dim = 0;

    // Compute outer, reduce, inner sizes
    uint32_t outer = 1;
    for (int32_t i = 0; i < dim; ++i) outer *= xDim[i];
    uint32_t reduce = xDim[dim];
    uint32_t inner = 1;
    for (int32_t i = dim + 1; i < xDimNum; ++i) inner *= xDim[i];
    uint32_t outputSize = total / reduce;   // number of output elements

    // Tiling based on output elements
    // Each output element requires iterating over reduce dimension (float reads)
    // We process output elements; for each we need a local buffer to hold reduce dim values
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * reduce + sizeof(int64_t));
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = ALIGN_ELEMS;
    if (outputSize < maxTile) {
        uint32_t aligned = AlignDown(outputSize, ALIGN_ELEMS);
        maxTile = (aligned > 0) ? aligned : outputSize;
    }

    uint32_t tileLen = maxTile;
    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(outputSize, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);
    tiling.set_outerDim(outer);
    tiling.set_reduceDim(reduce);
    tiling.set_innerDim(inner);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    int32_t dimNum = x_shape->GetDimNum();
    const int64_t *dims = x_shape->GetDims();
    // Read dim attribute from context (not directly available; use tiling data? Actually we can get from attrs)
    // Since we don't have attrs here, we rely on tiling having correct info; but for shape we need dim.
    // Alternative: use the same dim computation. We'll parse from context->GetAttrs()
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += dimNum;
    if (dim < 0 || dim >= dimNum) dim = 0;

    std::vector<int64_t> yDims;
    for (int32_t i = 0; i < dimNum; ++i) {
        if (i != dim) yDims.push_back(dims[i]);
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(yDims.size());
    for (size_t i = 0; i < yDims.size(); ++i) {
        y_shape->SetDim(i, yDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerDim, uint32_t reduceDim,
                                uint32_t innerDim, uint32_t blockDim)
    {
        outerDim_ = outerDim;
        reduceDim_ = reduceDim;
        innerDim_ = innerDim;
        blockDim_ = blockDim;
        outputLen_ = outerDim * innerDim;    // total output elements

        xGm_.SetGlobalBuffer((__gm__ float*)x, outerDim * reduceDim * innerDim);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, outputLen_);

        // We only need a small local buffer for reading reduce dim values (float) and one int64 for result
        pipe_.InitBuffer(inBuf_, 1, reduceDim_ * sizeof(float));
        pipe_.InitBuffer(outBuf_, 1, sizeof(int64_t));
    }

    __aicore__ inline void Process()
    {
        if (outputLen_ == 0 || reduceDim_ == 0) return;

        uint32_t blockId = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Grid-stride loop over output elements
        for (uint32_t outIdx = blockId; outIdx < outputLen_; outIdx += blockNum) {
            // Compute input base for this output element
            uint32_t outerIdx = outIdx / innerDim_;
            uint32_t innerIdx = outIdx % innerDim_;
            uint32_t base = outerIdx * (reduceDim_ * innerDim_) + innerIdx;

            // Load reduce dim values into local buffer
            AscendC::LocalTensor<float> xLocal = inBuf_.AllocTensor<float>();
            for (uint32_t k = 0; k < reduceDim_; ++k) {
                uint32_t gmIdx = base + k * innerDim_;
                xLocal(k) = xGm_.GetValue(gmIdx);
            }

            // Find argmax
            float maxVal = xLocal(0);
            int64_t maxIdx = 0;
            for (uint32_t k = 1; k < reduceDim_; ++k) {
                if (xLocal(k) > maxVal) {
                    maxVal = xLocal(k);
                    maxIdx = static_cast<int64_t>(k);
                }
            }
            inBuf_.FreeTensor(xLocal);

            // Write output
            yGm_.SetValue(outIdx, maxIdx);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t outerDim_ {0};
    uint32_t reduceDim_ {0};
    uint32_t innerDim_ {0};
    uint32_t blockDim_ {0};
    uint32_t outputLen_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.outerDim, tiling_data.reduceDim, tiling_data.innerDim, tiling_data.blockDim);
    op.Process();
}
