host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &xShape = storageShape->GetOriginShape();
    int32_t xDimNum = xShape.GetDimNum();
    int64_t total = 1;
    for (int32_t i = 0; i < xDimNum; ++i) total *= xShape.GetDim(i);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    // Normalize dim
    if (dim < 0) dim += xDimNum;
    if (dim < 0 || dim >= xDimNum) dim = 0;

    // Compute outer, reduce, inner sizes
    uint32_t outer = 1;
    for (int32_t i = 0; i < dim; ++i) outer *= xShape.GetDim(i);
    uint32_t reduce = xShape.GetDim(dim);
    uint32_t inner = 1;
    for (int32_t i = dim + 1; i < xDimNum; ++i) inner *= xShape.GetDim(i);
    uint32_t outputSize = total / reduce;   // number of output elements

    // Tiling based on output elements
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * reduce + sizeof(int64_t));
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = ALIGN_ELEMS;
    if (outputSize < maxTile) {
        uint32_t aligned = AlignDown(outputSize, ALIGN_ELEMS);
        maxTile = (aligned > 0) ? aligned : outputSize;
    }

    uint32_t tileLen = maxTile;
    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(outputSize, tileLen);
    uint32_t blockDim = numTiles;
