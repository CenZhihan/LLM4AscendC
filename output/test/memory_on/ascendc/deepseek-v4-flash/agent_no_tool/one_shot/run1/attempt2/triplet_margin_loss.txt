host_operator_src="""

#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget analysis:
// For each sample: need 3*d floats (anchor, positive, negative) for input,
// plus temporary for squared differences (2*d) => total 5*d floats per sample.
// Budget: 192*1024 bytes = 49152 floats.
// maxSamples = 49152 / (5*dim). For dim=128 -> ~76 samples.
static constexpr uint32_t UB_BUDGET_FLOATS = 49152;
static constexpr uint32_t ALIGN_SAMPLES = 8;
static constexpr uint32_t MAX_BLOCK_DIM = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const auto* shape = context->GetInputShape(0);
    uint32_t batch = static_cast<uint32_t>(shape->GetDim(0));
    uint32_t dim   = static_cast<uint32_t>(shape->GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // Compute tileBatch from UB budget
    uint32_t maxSamples = UB_BUDGET_FLOATS / (5 * dim);
    maxSamples = (maxSamples / ALIGN_SAMPLES) * ALIGN_SAMPLES;
    if (maxSamples < ALIGN_SAMPLES) maxSamples = ALIGN_SAMPLES;
    uint32_t tileBatch = maxSamples;
    if (batch < tileBatch) {
        tileBatch = batch;
        tileBatch = (tileBatch / ALIGN_SAMPLES) * ALIGN_SAMPLES;
        if (tileBatch == 0) tileBatch = batch;
    }

    uint32_t numTiles = (batch + tileBatch - 1) / tileBatch;
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;

    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_dim(dim);
    tiling.set_tileBatch(tileBatch);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    auto* shape = context->GetInputShape(0);
    uint32_t batch = shape->GetDim(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(1);
    y_shape->SetDim(0, batch);
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
                                uint32_t batch, uint32_t dim, uint32_t tileBatch, float margin)
    {
        batch_ = batch;
        dim_ = dim;
        tileBatch_ = tileBatch;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, batch * dim);
        posGm_.SetGlobalBuffer((__gm__ float*)positive, batch * dim);
        negGm_.SetGlobalBuffer((__gm__ float*)negative, batch * dim);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, batch);

        uint32_t tileBytes = tileBatch * dim * sizeof(float);
        pipe_.InitBuffer(inAnchorQ_, BUFFER_NUM, tileBytes);
        pipe_.InitBuffer(inPosQ_, BUFFER_NUM, tileBytes);
        pipe_.InitBuffer(inNegQ_, BUFFER_NUM, tileBytes);
        pipe_.InitBuffer(tmpQ_, 1, tileBytes);
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || dim_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t start = blockIdx * tileBatch_;
        uint32_t stride = blockNum * tileBatch_;

        while (start < batch_) {
            uint32_t curBatch = batch_ - start;
            if (curBatch > tileBatch_) curBatch = tileBatch_;
            ProcessTile(start, curBatch);
            start += stride;
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t start, uint32_t batchCount)
    {
        AscendC::LocalTensor<float> anchorLocal = inAnchorQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> posLocal   = inPosQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> negLocal   = inNegQ_.AllocTensor<float>();
        AscendC::DataCopy(anchorLocal, anchorGm_[start * dim_], batchCount * dim_);
        AscendC::DataCopy(posLocal,   posGm_[start * dim_],   batchCount * dim_);
        AscendC::DataCopy(negLocal,   negGm_[start * dim_],   batchCount * dim_);

        AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();

        for (uint32_t s = 0; s < batchCount; s++) {
            uint32_t offset = s * dim_;
            auto anchorRow = anchorLocal[offset];
            auto posRow    = posLocal[offset];
            auto negRow    = negLocal[offset];

            AscendC::Sub(tmp, anchorRow, posRow, dim_);
            AscendC::Mul(tmp, tmp, tmp, dim_);
            float posDist = AscendC::ReduceSum(tmp, dim_);

            AscendC::Sub(tmp, anchorRow, negRow, dim_);
            AscendC::Mul(tmp, tmp, tmp, dim_);
            float negDist = AscendC::ReduceSum(tmp, dim_);

            float lossVal = posDist - negDist + margin_;
            if (lossVal < 0.0f) lossVal = 0.0f;

            lossGm_[start + s] = lossVal;
        }

        tmpQ_.FreeTensor(tmp);
        inAnchorQ_.FreeTensor(anchorLocal);
        inPosQ_.FreeTensor(posLocal);
        inNegQ_.FreeTensor(negLocal);
    }

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inAnchorQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inPosQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inNegQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> posGm_;
    AscendC::GlobalTensor<float> negGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t batch_ {0};
    uint32_t dim_ {0};
    uint32_t tileBatch_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                  GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.batch, tiling_data.dim, tiling_data.tileBatch, tiling_data.margin);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'triplet_margin_loss_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, *inputs):
        anchor, positive, negative, margin_tensor = inputs
        margin = float(margin_tensor.reshape(-1)[0].item()) if margin_tensor.numel() else 0.0
        loss_per_sample = getattr(custom_ops_lib, _FN)(anchor, positive, negative, margin)
        return loss_per_sample.mean()
