host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * innerTileLen
//   inQueueMask: 2 * innerTileLen
//   outQueueY: 2 * innerTileLen
//   tmp: 0 (no extra queue besides output)
// Total floats = 6 * innerTileLen => bytes = 24 * innerTileLen
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;              // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_INNER_TILE = 8192;     // avoid too-large per-iteration latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaskedCumsumCustomTilingData tiling;

    // Use StorageShape and GetOriginShape to avoid conversion error
    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::Shape &x_shape = x_storage->GetOriginShape();
    const uint32_t ndims = x_shape.GetDimNum();
    // Read shape dimensions
    uint32_t totalRows = 1;
    uint32_t rowLen = 0;

    // We only support at most 2D tensors for simplicity.
    // If 1D, treat as a single row.
    if (ndims == 1) {
        totalRows = 1;
        rowLen = x_shape.GetDim(0);
    } else if (ndims == 2) {
        // Assume dim is last dimension
        totalRows = x_shape.GetDim(0);
        rowLen = x_shape.GetDim(1);
    } else {
        // fallback – flatten to 2D? For now return error.
        return ge::GRAPH_FAILED;
    }

    // Attribute dim (for completeness, but we ignore for 2D case)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    int32_t dimVal = static_cast<int32_t>(*dimAttr);
    if (ndims == 2 && dimVal != 1) {
        // For this kernel, we only support dim = last dimension (1 for 2D).
        // In a production operator, you'd handle arbitrary dim.
        return ge::GRAPH_FAILED;
    }

    // Compute innerTileLen from UB budget: bytes = 24 * innerTileLen
    uint32_t maxInner = UB_BUDGET_BYTES / 24;
    maxInner = AlignDown(maxInner, ALIGN_ELEMS);
    if (maxInner < ALIGN_ELEMS) maxInner = ALIGN_ELEMS;
    if (maxInner > HARD_CAP_INNER_TILE) maxInner = AlignDown(HARD_CAP_INNER_TILE, ALIGN_ELEMS);

    uint32_t innerTileLen = maxInner;
    if (rowLen < innerTileLen) {
        uint32_t aligned = AlignDown(rowLen, ALIGN_ELEMS);
        innerTileLen = (aligned > 0) ? aligned : rowLen;
    }

    // BlockDim distribution: parallel over rows
    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_rowLen(rowLen);
    tiling.set_innerTileLen(innerTileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dimVal);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"
using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLen,
                                uint32_t innerTileLen, int32_t dim)
    {
        totalRows_ = totalRows;
        rowLen_ = rowLen;
        innerTileLen_ = innerTileLen;
        dim_ = dim; // unused beyond sanity

        // Assume 2D tensor: x and y have shape [totalRows, rowLen]
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * rowLen_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalRows_ * rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * rowLen_);

        // Allocate double buffers for input x, input mask, and output y
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, innerTileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, innerTileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, innerTileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalRows_ == 0 || rowLen_ == 0 || innerTileLen_ == 0)
            return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        // Distribute rows among blocks cyclically for load balance
        for (uint32_t r = blockIdx; r < totalRows_; r += blockNum) {
            ProcessOneRow(r);
        }
    }

private:
    __aicore__ inline void ProcessOneRow(uint32_t rowIdx) {
        const uint32_t rowOffset = rowIdx * rowLen_;
        float runningSum = 0.0f;

        for (uint32_t tileStart = 0; tileStart < rowLen_; tileStart += innerTileLen_) {
            uint32_t curLen = (tileStart + innerTileLen_ <= rowLen_) ? innerTileLen_ : (rowLen_ - tileStart);

            // Allocate local tensors
            LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
            LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // Copy in
            DataCopy(xLocal, xGm_[rowOffset + tileStart], curLen);
            DataCopy(maskLocal, maskGm_[rowOffset + tileStart], curLen);

            // Compute cumulative sum for this tile
            for (uint32_t i = 0; i < curLen; ++i) {
                float xVal = xLocal.GetValue(i);
                float mVal = maskLocal.GetValue(i);
                runningSum = xVal * mVal + runningSum;
                yLocal.SetValue(i, runningSum);
            }

            // Copy out
            DataCopy(yGm_[rowOffset + tileStart], yLocal, curLen);

            // Free local tensors
            inQueueX_.FreeTensor(xLocal);
            inQueueMask_.FreeTensor(maskLocal);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> maskGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t rowLen_ {0};
    uint32_t innerTileLen_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.totalRows, tiling_data.rowLen, tiling_data.innerTileLen, tiling_data.dim);
    op.Process();
}
