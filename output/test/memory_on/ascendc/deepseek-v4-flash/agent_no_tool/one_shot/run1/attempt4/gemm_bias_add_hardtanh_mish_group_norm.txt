static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmBiasAddHardtanhMishGroupNormCustomTilingData tiling;

    // Input shapes: x[N, K] where K=outFeatures, bias[K], gn_gamma[K], gn_beta[K]
    const auto &x_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batch = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t outFeatures = static_cast<uint32_t>(x_shape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t numGroups = attrs->GetAttrPointer<int32_t>(0)[0];
    float minVal = attrs->GetAttrPointer<float>(1)[0];
    float maxVal = attrs->GetAttrPointer<float>(2)[0];
    float eps = attrs->GetAttrPointer<float>(3)[0];

    uint32_t groupSize = outFeatures / static_cast<uint32_t>(numGroups);
    if (outFeatures != groupSize * static_cast<uint32_t>(numGroups)) {
        return ge::GRAPH_FAILED; // Not divisible
    }

    // UB budget: we need three local buffers: x tile (groupSize), temp (groupSize), and group norm buffers (groupSize each)
    // Actually we need: input tile (groupSize), bias/gamma/beta tiles, and temp buffers.
    // We'll process one group at a time, so max local memory = 4 * groupSize * sizeof(float)
    uint32_t maxGroupSize = (UB_BUDGET_BYTES / 4) / 4; // 4 buffers of groupSize
    uint32_t tileGroupSize = groupSize;
    if (tileGroupSize > maxGroupSize) {
        // If group is too large, we need to tile further, but for simplicity assume it fits.
        // If not, we could set blockDim to 1 and rely on single block processing.
        tileGroupSize = AlignDown(maxGroupSize, ALIGN_ELEMS);
        if (tileGroupSize == 0) tileGroupSize = ALIGN_ELEMS;
    }

    // Block dimension: distribute rows among cores
    uint32_t blockDim = batch;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(batch * outFeatures);
    tiling.set_batch(batch);
    tiling.set_outFeatures(outFeatures);
    tiling.set_numGroups(static_cast<uint32_t>(numGroups));
    tiling.set_groupSize(groupSize);
    tiling.set_invGroupSize(1.0f / static_cast<float>(groupSize));
    tiling.set_blockDim(blockDim);
    tiling.set_minVal(minVal);
    tiling.set_maxVal(maxVal);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR gn_gamma,
                                GM_ADDR gn_beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t batch,
                                uint32_t outFeatures,
                                uint32_t numGroups, uint32_t groupSize,
                                float invGroupSize, float minVal, float maxVal, float eps)
    {
        totalLen_ = totalLength;
        batch_ = batch;
        outFeat_ = outFeatures;
        numGroups_ = numGroups;
        groupSize_ = groupSize;
        invGroupSize_ = invGroupSize;
        minVal_ = minVal;
        maxVal_ = maxVal;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures);
        betaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);

        // Allocate local buffers (size = groupSize)
        pipe_.InitBuffer(bufXLocal_, BUFFER_NUM, groupSize_ * sizeof(float));
        pipe_.InitBuffer(bufBiasLocal_, 1, groupSize_ * sizeof(float));
        pipe_.InitBuffer(bufGammaLocal_, 1, groupSize_ * sizeof(float));
        pipe_.InitBuffer(bufBetaLocal_, 1, groupSize_ * sizeof(float));
        pipe_.InitBuffer(bufTempLocal_, 1, groupSize_ * sizeof(float));
        // Buffers for scalar results of reduction
        pipe_.InitBuffer(bufSum_, 1, sizeof(float));
        pipe_.InitBuffer(bufVarSum_, 1, sizeof(float));
        // Buffer for rstd scalar
        pipe_.InitBuffer(bufRstd_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride over rows
        uint32_t row = blockIdx;
        while (row < batch_) {
            ProcessRow(row);
            row += blockNum;
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        for (uint32_t g = 0; g < numGroups_; ++g) {
            uint32_t channelStart = g * groupSize_;
            uint32_t groupOffset = row * outFeat_ + channelStart;

            // Load x slice (one group)
            AscendC::LocalTensor<float> xLocal = bufXLocal_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[groupOffset], groupSize_);

            // Load bias slice
            AscendC::LocalTensor<float> biasLocal = bufBiasLocal_.AllocTensor<float>();
            AscendC::DataCopy(biasLocal, biasGm_[channelStart], groupSize_);

            // x = x + bias
            AscendC::Add(xLocal, xLocal, biasLocal, groupSize_);
            bufBiasLocal_.FreeTensor(biasLocal);

            // hardtanh: clamp
            AscendC::Mins(xLocal, xLocal, maxVal_, groupSize_);
            AscendC::Maxs(xLocal, xLocal, minVal_, groupSize_);

            // mish: y = x * tanh(softplus(x))
            // softplus: log(1+exp(x)), but limit x to [-20,20] for numerical stability
            AscendC::LocalTensor<float> tempLocal = bufTempLocal_.AllocTensor<float>();
            AscendC::Mins(tempLocal, xLocal, 20.0f, groupSize_);
            AscendC::Maxs(tempLocal, tempLocal, -20.0f, groupSize_);
            // exp(temp)
            AscendC::Exp(tempLocal, tempLocal, groupSize_);
            // add 1
            AscendC::Adds(tempLocal, tempLocal, 1.0f, groupSize_);
            // log
            AscendC::Log(tempLocal, tempLocal, groupSize_);
            // tanh
            AscendC::Tanh(tempLocal, tempLocal, groupSize_);
            // multiply with original x
            AscendC::Mul(xLocal, xLocal, tempLocal, groupSize_);

            // group norm: compute mean and variance
            // Use vector reduction to compute sum
            AscendC::LocalTensor<float> sumTensor = bufSum_.AllocTensor<float>();
            AscendC::LocalTensor<float> varSumTensor = bufVarSum_.AllocTensor<float>();
            // ReduceSum with mask 0 (reduce last dimension)
            AscendC::ReduceSum(sumTensor, xLocal, groupSize_, 0);
            float groupSum = sumTensor(0);
            float mean = groupSum * invGroupSize_;

            // variance: sum((x - mean)^2) / groupSize
            // Use Adds to subtract scalar mean: x - mean = Adds(x, -mean)
            AscendC::Adds(tempLocal, xLocal, -mean, groupSize_);
            AscendC::Mul(tempLocal, tempLocal, tempLocal, groupSize_);
            AscendC::ReduceSum(varSumTensor, tempLocal, groupSize_, 0);
            float var = varSumTensor(0) * invGroupSize_;
            bufSum_.FreeTensor(sumTensor);
            bufVarSum_.FreeTensor(varSumTensor);

            // Compute rstd = 1/sqrt(var + eps)
            AscendC::LocalTensor<float> rstdTensor = bufRstd_.AllocTensor<float>();
            rstdTensor(0) = var + eps_;
            AscendC::Rsqrt(rstdTensor, rstdTensor, 1);
            float rstd = rstdTensor(0);
            bufRstd_.FreeTensor(rstdTensor);

            // normalize: (x - mean) * rstd
            AscendC::Adds(tempLocal, xLocal, -mean, groupSize_);
            AscendC::Muls(tempLocal, tempLocal, rstd, groupSize_);

            // load gamma and beta
            AscendC::LocalTensor<float> gammaLocal = bufGammaLocal_.AllocTensor<float>();
            AscendC::LocalTensor<float> betaLocal = bufBetaLocal_.AllocTensor<float>();
            AscendC::DataCopy(gammaLocal, gammaGm_[channelStart], groupSize_);
            AscendC::DataCopy(betaLocal, betaGm_[channelStart], groupSize_);

            // apply gamma and beta: out = temp * gamma + beta
            AscendC::Mul(tempLocal, tempLocal, gammaLocal, groupSize_);
            AscendC::Add(tempLocal, tempLocal, betaLocal, groupSize_);

            bufGammaLocal_.FreeTensor(gammaLocal);
            bufBetaLocal_.FreeTensor(betaLocal);

            // store to global y
            AscendC::DataCopy(yGm_[groupOffset], tempLocal, groupSize_);

            bufTempLocal_.FreeTensor(tempLocal);
            bufXLocal_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bufXLocal_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bufBiasLocal_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bufGammaLocal_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bufBetaLocal_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufTempLocal_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufSum_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufVarSum_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufRstd_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t batch_ {0};
    uint32_t outFeat_ {0};
    uint32_t numGroups_ {0};
    uint32_t groupSize_ {0};
    float invGroupSize_ {0.0f};
    float minVal_ {-1.0f};
    float maxVal_ {1.0f};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, bias, gn_gamma, gn_beta, y,
            tiling_data.totalLength, tiling_data.batch,
            tiling_data.outFeatures,
            tiling_data.numGroups, tiling_data.groupSize,
            tiling_data.invGroupSize, tiling_data.minVal, tiling_data.maxVal, tiling_data.eps);
    op.Process();
}

model_src='''
import torch
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

_FN = 'gemm_bias_add_hardtanh_mish_group_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        # init_inputs: [in_features, out_features, bias_shape, num_groups]
        self.num_groups = init_inputs[3] if len(init_inputs) > 3 else 256
        # Note: weight, lin_bias, bias, gn_gamma, gn_beta are provided in forward inputs

    def forward(self, *inputs):
        # inputs: x, weight, lin_bias, bias, gn_gamma, gn_beta
        x, weight, lin_bias, bias, gn_gamma, gn_beta = inputs
        # Linear (GEMM) done in PyTorch
        x = F.linear(x, weight, lin_bias)
        # Fused custom op: bias_add, hardtanh, mish, group_norm
        fn = getattr(custom_ops_lib, _FN)
        y = fn(x, bias, gn_gamma, gn_beta, num_groups=self.num_groups,
               min_val=-1.0, max_val=1.0, eps=1e-5)
        return y
'''
