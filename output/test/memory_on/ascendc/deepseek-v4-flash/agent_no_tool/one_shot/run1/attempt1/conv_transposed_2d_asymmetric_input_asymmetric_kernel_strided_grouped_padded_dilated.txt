    const int32_t *dilation_attr = attrs->GetAttrPointer<int32_t>(4);

    tiling.stride_w = static_cast<uint32_t>((stride_attr[1] > 0) ? stride_attr[1] : 1);
    tiling.pad_h = static_cast<uint32_t>(padding_attr[0]);
    tiling.pad_w = static_cast<uint32_t>(padding_attr[1]);
    tiling.output_pad_h = static_cast<uint32_t>(output_padding_attr[0]);
    tiling.output_pad_w = static_cast<uint32_t>(output_padding_attr[1]);
    tiling.dilation_h = static_cast<uint32_t>((dilation_attr[0] > 0) ? dilation_attr[0] : 1);
    tiling.dilation_w = static_cast<uint32_t>((dilation_attr[1] > 0) ? dilation_attr[1] : 1);

    // Compute output shape
    tiling.H_out = (tiling.H_in - 1) * tiling.stride_h - 2 * tiling.pad_h
                   + tiling.dilation_h * (tiling.kH - 1) + tiling.output_pad_h + 1;
    tiling.W_out = (tiling.W_in - 1) * tiling.stride_w - 2 * tiling.pad_w
                   + tiling.dilation_w * (tiling.kW - 1) + tiling.output_pad_w + 1;

    context->SetOutputShape(0, {tiling.N, tiling.C_out, tiling.H_out, tiling.W_out});

    // Determine tiling parameters

    uint32_t tileH = 8;

    if (tiling.W_out < tileW) tileW = tiling.W_out;

    uint32_t blockDim = totalTiles;

    if (blockDim > 24) blockDim = 24;
    tiling.blockDim = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
