#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose2dCustom {
public:
    __aicore__ inline KernelConvTranspose2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        N_ = td.N;
        C_in_ = td.C_in;
        H_in_ = td.H_in;
        W_in_ = td.W_in;
        C_out_ = td.C_out;
        kH_ = td.kH;
        kW_ = td.kW;
        groups_ = td.groups;
        stride_h_ = td.stride_h;
        stride_w_ = td.stride_w;
        pad_h_ = td.pad_h;
        pad_w_ = td.pad_w;
        output_pad_h_ = td.output_pad_h;
        output_pad_w_ = td.output_pad_w;
        dilation_h_ = td.dilation_h;
        dilation_w_ = td.dilation_w;
        H_out_ = td.H_out;
        W_out_ = td.W_out;
        tileH_ = td.tileH;
        tileW_ = td.tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)w);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        if (H_out_ == 0 || W_out_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockDim = blockDim_;  // from tiling, but stored manually

        // Each block handles one output channel group (all output channels for a given group index?)
        // Actually we parallelize over output channel groups and spatial tiles.
        // For simplicity, blockIdx corresponds to a specific output channel group and tile.
        // We'll assign: blockIdx = g * (numTilesH * numTilesW) + h * numTilesW + w
        // But we stored tileH_ and tileW_ as 1, so each tile is a single output element.
        // So each block processes one output element (n, c, h, w) with n=0 for this test.
        // We'll loop over output elements assigned to this block via grid-stride.

        // Since tile size is 1, we can treat each block as handling one output element per iteration.
        // But to support multiple output elements per block, we use grid-stride over total output elements.
        uint32_t totalOutElems = N_ * C_out_ * H_out_ * W_out_;
        if (totalOutElems == 0) return;

        // Compute the number of elements per block (blockDim is total tiles? We have one tile per block)
        // Instead, we use blockIdx to process element index directly.
        uint32_t elemIdx = blockIdx;
        const uint32_t stride = blockDim;

        while (elemIdx < totalOutElems) {
            // Decompose element index into (n, c, h, w)
            uint32_t n, c, h, w;
            uint32_t tmp = elemIdx;
            w = tmp % W_out_;
            tmp /= W_out_;
            h = tmp % H_out_;
            tmp /= H_out_;
            c = tmp % C_out_;
            n = tmp / C_out_;

            // Compute the output value at this element using conv transpose
            float out_val = 0.0f;

            // For each input channel group (g) and within-group input channels
            uint32_t group_idx = c / (C_out_ / groups_);  // which group this output channel belongs to
            uint32_t c_out_in_group = c % (C_out_ / groups_);
            // weight: shape (C_in, C_out/groups, kH, kW)
            // weight offset for a group: we need to index weight by input channels that belong to the same group.
            // For conv transpose, the weight is grouped: the input channels are partitioned into groups.
            // Input channel group = group_idx. So the weight sub-tensor for this group is:
            // weight[ group_idx * (C_in/groups) : (group_idx+1)*(C_in/groups), :, :, : ]
            // Actually weight first dim is C_in, so we need to iterate over input channels in this group.
            uint32_t C_in_per_group = C_in_ / groups_;
            uint32_t c_in_start = group_idx * C_in_per_group;
            uint32_t c_in_end = c_in_start + C_in_per_group;

            // Kernel iteration: for each kernel index (i, j)
            for (uint32_t i = 0; i < kH_; i++) {
                for (uint32_t j = 0; j < kW_; j++) {
                    // For conv transpose, the output position (h, w) corresponds to input position (h_in, w_in) such that
                    // h = stride_h * h_in + dilation_h * (kH - 1 - i) - pad_h? Wait the formula for conv2d transpose:
                    // output[h][w] += input[h_in][w_in] * weight[kh-1-i][kw-1-j] when (h - (dilation_h*(kh-1-i) - pad_h)) % stride_h == 0 ...
                    // Actually easier: compute the mapping from kernel index to input coordinate.
                    // For each kernel element (i, j) with dilation, the output is influenced by input at:
                    // h_in = (h + pad_h - dilation_h * i) / stride_h  (must be integer)
                    // w_in = (w + pad_w - dilation_w * j) / stride_w
                    // And the weight index for this kernel element is (kh-1-i, kw-1-j) because convolution transpose rotates the kernel.
                    // But the weight is provided as is (not rotated), so we use weight[ input_channel ][ output_channel_in_group ][ i ][ j ] directly.
                    // The standard conv_transpose2d formula:
                    // output[n][c_out][h][w] = sum_{c_in} sum_{i,j} input[n][c_in][h_in][w_in] * weight[c_in][c_out_per_group][i][j]
                    // where h_in = (h + pad_h - i * dilation_h) / stride_h (integer division), similarly for w.
                    // And we need to check that the division is exact.

                    int32_t numerator_h = static_cast<int32_t>(h) + static_cast<int32_t>(pad_h_) - static_cast<int32_t>(i * dilation_h_);
                    if (numerator_h < 0) continue;
                    if (numerator_h % static_cast<int32_t>(stride_h_) != 0) continue;
                    uint32_t h_in = static_cast<uint32_t>(numerator_h) / stride_h_;
                    if (h_in >= H_in_) continue;

                    int32_t numerator_w = static_cast<int32_t>(w) + static_cast<int32_t>(pad_w_) - static_cast<int32_t>(j * dilation_w_);
                    if (numerator_w < 0) continue;
                    if (numerator_w % static_cast<int32_t>(stride_w_) != 0) continue;
                    uint32_t w_in = static_cast<uint32_t>(numerator_w) / stride_w_;
                    if (w_in >= W_in_) continue;

                    // Now we have valid input position. Sum over input channels in this group.
                    float kernel_weight_sum = 0.0f;
                    // For each input channel in the group
                    // weight index: input_channel = c_in, output_channel_in_group = c_out_in_group, i, j
                    uint32_t w_offset_base = c_out_in_group * kH_ * kW_ + i * kW_ + j;
                    for (uint32_t c_in = 0; c_in < C_in_per_group; c_in++) {
                        uint32_t global_c_in = c_in_start + c_in;
                        float x_val = xGm_[n * (C_in_ * H_in_ * W_in_) + global_c_in * (H_in_ * W_in_) + h_in * W_in_ + w_in];
                        // weight index: (global_c_in, c_out_in_group, i, j)
                        uint32_t w_index = global_c_in * ( (C_out_/groups_) * kH_ * kW_ ) + w_offset_base;
                        float w_val = wGm_[w_index];
                        kernel_weight_sum += x_val * w_val;
                    }
                    out_val += kernel_weight_sum;
                }
            }

            // Write output
            uint32_t y_offset = n * (C_out_ * H_out_ * W_out_) + c * (H_out_ * W_out_) + h * W_out_ + w;
            yGm_[y_offset] = out_val;

            elemIdx += stride;
        }
    }

private:
    // All tiling data stored as members
    // (We need blockDim_ from tiling, but GET_TILING_DATA gave local, we can store it)
    // Actually blockDim_ is not stored; we can recompute or store. Let's add a member.
    // We'll get it from tiling data in Init and store.
    uint32_t blockDim_;

    // Other members
    uint32_t N_, C_in_, H_in_, W_in_;
    uint32_t C_out_, kH_, kW_, groups_;
    uint32_t stride_h_, stride_w_, pad_h_, pad_w_;
    uint32_t output_pad_h_, output_pad_w_;
    uint32_t dilation_h_, dilation_w_;
    uint32_t H_out_, W_out_;
    uint32_t tileH_, tileW_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    KernelConvTranspose2dCustom op;
    // We must store blockDim from tiling into the object.
    // Since the class doesn't have Init with blockDim, we extend.
    op.blockDim_ = td.blockDim;  // add this line after Init or adjust Init.
    // But Init currently does not set blockDim_. We'll modify Init to take all tiling data.
    // To keep minimal changes, we can store blockDim_ after Init.
    op.Init(x, w, y, workspace, tiling);  // This will read tiling again, but ok.
    // Store blockDim after Init:
    op.blockDim_ = td.blockDim;
    op.Process();
}
