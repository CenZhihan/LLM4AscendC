#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose2dCustom {
public:
    __aicore__ inline KernelConvTranspose2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        N_ = td.N;
        C_in_ = td.C_in;
        H_in_ = td.H_in;
        W_in_ = td.W_in;
        C_out_ = td.C_out;
        kH_ = td.kH;
        kW_ = td.kW;
        groups_ = td.groups;
        stride_h_ = td.stride_h;
        stride_w_ = td.stride_w;
        pad_h_ = td.pad_h;
        pad_w_ = td.pad_w;
        output_pad_h_ = td.output_pad_h;
        output_pad_w_ = td.output_pad_w;
        dilation_h_ = td.dilation_h;
        dilation_w_ = td.dilation_w;
        H_out_ = td.H_out;
        W_out_ = td.W_out;
        tileH_ = td.tileH;
        tileW_ = td.tileW;
        blockDim_ = td.blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)w);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        if (H_out_ == 0 || W_out_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t stride = blockDim_;

        uint32_t totalOutElems = N_ * C_out_ * H_out_ * W_out_;
        if (totalOutElems == 0) return;

        uint32_t elemIdx = blockIdx;
        while (elemIdx < totalOutElems) {
            uint32_t n, c, h, w;
            uint32_t tmp = elemIdx;
            w = tmp % W_out_;
            tmp /= W_out_;
            h = tmp % H_out_;
            tmp /= H_out_;
            c = tmp % C_out_;
            n = tmp / C_out_;

            float out_val = 0.0f;
            uint32_t group_idx = c / (C_out_ / groups_);
            uint32_t c_out_in_group = c % (C_out_ / groups_);
            uint32_t C_in_per_group = C_in_ / groups_;
            uint32_t c_in_start = group_idx * C_in_per_group;
            uint32_t c_in_end = c_in_start + C_in_per_group;

            for (uint32_t i = 0; i < kH_; i++) {
                for (uint32_t j = 0; j < kW_; j++) {
                    int32_t numerator_h = static_cast<int32_t>(h) + static_cast<int32_t>(pad_h_) - static_cast<int32_t>(i * dilation_h_);
                    if (numerator_h < 0) continue;
                    if (numerator_h % static_cast<int32_t>(stride_h_) != 0) continue;
                    uint32_t h_in = static_cast<uint32_t>(numerator_h) / stride_h_;
                    if (h_in >= H_in_) continue;

                    int32_t numerator_w = static_cast<int32_t>(w) + static_cast<int32_t>(pad_w_) - static_cast<int32_t>(j * dilation_w_);
                    if (numerator_w < 0) continue;
                    if (numerator_w % static_cast<int32_t>(stride_w_) != 0) continue;
                    uint32_t w_in = static_cast<uint32_t>(numerator_w) / stride_w_;
                    if (w_in >= W_in_) continue;

                    float kernel_weight_sum = 0.0f;
                    uint32_t w_offset_base = c_out_in_group * kH_ * kW_ + i * kW_ + j;
                    for (uint32_t c_in = 0; c_in < C_in_per_group; c_in++) {
                        uint32_t global_c_in = c_in_start + c_in;
                        float x_val = xGm_[n * (C_in_ * H_in_ * W_in_) + global_c_in * (H_in_ * W_in_) + h_in * W_in_ + w_in];
                        uint32_t w_index = global_c_in * ( (C_out_/groups_) * kH_ * kW_ ) + w_offset_base;
                        float w_val = wGm_[w_index];
                        kernel_weight_sum += x_val * w_val;
                    }
                    out_val += kernel_weight_sum;
                }
            }

            uint32_t y_offset = n * (C_out_ * H_out_ * W_out_) + c * (H_out_ * W_out_) + h * W_out_ + w;
            yGm_[y_offset] = out_val;

            elemIdx += stride;
        }
    }

private:
    uint32_t blockDim_;
    uint32_t N_, C_in_, H_in_, W_in_;
    uint32_t C_out_, kH_, kW_, groups_;
    uint32_t stride_h_, stride_w_, pad_h_, pad_w_;
    uint32_t output_pad_h_, output_pad_w_;
    uint32_t dilation_h_, dilation_w_;
    uint32_t H_out_, W_out_;
    uint32_t tileH_, tileW_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelConvTranspose2dCustom op;
    op.Init(x, w, y, workspace, tiling);
    op.Process();
}
