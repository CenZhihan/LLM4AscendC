host_operator_src="""

#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: we use a simple single-buffer approach for output tile.
// Each thread processes tileLength output elements, each requiring temporary storage
// for intermediate maxima. Budget: 1 tile of output (float) plus small overhead.
// We aim for tileLength <= 4096 to keep UB usage modest.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;   // align to 256 floats (1KB)
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;
    const gert::Shape *inShape = context->GetInputShape(0);
    uint32_t inNDim = inShape->GetOriginShape().GetDimNum();
    const int64_t *inDims = inShape->GetOriginShape().GetDims();
    int32_t inN = static_cast<int32_t>(inDims[0]);
    int32_t inC = static_cast<int32_t>(inDims[1]);
    int32_t inD = static_cast<int32_t>(inDims[2]);
    int32_t inH = static_cast<int32_t>(inDims[3]);
    int32_t inW = static_cast<int32_t>(inDims[4]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t stride    = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding   = attrs->GetAttrPointer<int32_t>(2)[0];

    // Compute output dimensions (ceil_mode=False)
    int32_t outD = (inD + 2 * padding - kernelSize) / stride + 1;
    int32_t outH = (inH + 2 * padding - kernelSize) / stride + 1;
    int32_t outW = (inW + 2 * padding - kernelSize) / stride + 1;
    uint32_t totalOutputElems = static_cast<uint32_t>(inN) * static_cast<uint32_t>(inC)
                                * static_cast<uint32_t>(outD) * static_cast<uint32_t>(outH)
                                * static_cast<uint32_t>(outW);

    // Determine tile length (number of output elements per compute iteration)
    // We want each thread to process a contiguous chunk of output elements.
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float); // elements that fit in UB (single buffer)
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLength = maxTile;
    if (totalOutputElems < tileLength) {
        tileLength = AlignDown(totalOutputElems, ALIGN_ELEMS);
        if (tileLength == 0) tileLength = totalOutputElems;
    }

    // blockDim: number of parallel blocks = number of tiles
    uint32_t numTiles = (tileLength == 0) ? 1 : CeilDiv(totalOutputElems, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_inN(inN);
    tiling.set_inC(inC);
    tiling.set_inD(inD);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElems, uint32_t tileLength,
                                int32_t kernelSize, int32_t stride, int32_t padding,
                                int32_t inN, int32_t inC, int32_t inD, int32_t inH, int32_t inW,
                                int32_t outD, int32_t outH, int32_t outW)
    {
        totalOutputElems_ = totalOutputElems;
        tileLength_ = tileLength;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        inN_ = inN; inC_ = inC; inD_ = inD; inH_ = inH; inW_ = inW;
        outD_ = outD; outH_ = outH; outW_ = outW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Single output tile buffer: store float values per output element
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        // Temporary buffer for one pixel's max (reuse for each element)
        pipe_.InitBuffer(tmpBuf_, 1, sizeof(float) * (kernelSize_ * kernelSize_ * kernelSize_));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0 || tileLength_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output elements
        uint32_t start = blockIdx * tileLength_;
        if (start >= totalOutputElems_) return;

        // We'll process one tile per block per iteration; use double buffering
        // For simplicity, we process one output element at a time and pack results.
        // We'll fill a local tensor of tileLength_ floats and then copy out.

        // Allocate output buffer (tile)
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        uint32_t remaining = totalOutputElems_ - start;
        uint32_t curTileLen = (remaining < tileLength_) ? remaining : tileLength_;

        // For each output element in the tile, compute max
        for (uint32_t i = 0; i < curTileLen; ++i) {
            uint32_t flatIndex = start + i;
            // Unflatten to (n, c, od, oh, ow)
            int32_t n = flatIndex / (inC_ * outD_ * outH_ * outW_);
            int32_t tmp = flatIndex % (inC_ * outD_ * outH_ * outW_);
            int32_t c = tmp / (outD_ * outH_ * outW_);
            tmp = tmp % (outD_ * outH_ * outW_);
            int32_t od = tmp / (outH_ * outW_);
            tmp = tmp % (outH_ * outW_);
            int32_t oh = tmp / outW_;
            int32_t ow = tmp % outW_;

            // Compute input window start indices
            int32_t idStart = od * stride_ - padding_;
            int32_t ihStart = oh * stride_ - padding_;
            int32_t iwStart = ow * stride_ - padding_;

            float maxVal = -__builtin_huge_valf(); // -inf

            // Loop over kernel
            for (int32_t kd = 0; kd < kernelSize_; ++kd) {
                int32_t id = idStart + kd;
                if (id < 0 || id >= inD_) continue;
                for (int32_t kh = 0; kh < kernelSize_; ++kh) {
                    int32_t ih = ihStart + kh;
                    if (ih < 0 || ih >= inH_) continue;
                    for (int32_t kw = 0; kw < kernelSize_; ++kw) {
                        int32_t iw = iwStart + kw;
                        if (iw < 0 || iw >= inW_) continue;
                        // Compute flat input index: (n, c, id, ih, iw)
                        uint32_t inFlat = ((n * inC_ + c) * inD_ + id) * inH_ + ih;
                        inFlat = inFlat * inW_ + iw;
                        float val = xGm_.GetValue(inFlat);
                        if (val > maxVal) maxVal = val;
                    }
                }
            }
            // Write to local buffer
            yLocal.SetValue(i, maxVal);
        }

        // Copy out the tile
        uint32_t base = start;
        AscendC::DataCopy(yGm_[base], yLocal, curTileLen);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutputElems_{0};
    uint32_t tileLength_{0};
    int32_t kernelSize_{3};
    int32_t stride_{2};
    int32_t padding_{0};
    int32_t inN_, inC_, inD_, inH_, inW_;
    int32_t outD_, outH_, outW_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.totalOutputElems,
            tiling_data.tileLength,
            tiling_data.kernelSize,
            tiling_data.stride,
            tiling_data.padding,
            tiling_data.inN,
            tiling_data.inC,
            tiling_data.inD,
            tiling_data.inH,
            tiling_data.inW,
            tiling_data.outD,
            tiling_data.outH,
            tiling_data.outW);
    op.Process();
}
