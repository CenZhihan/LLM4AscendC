host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &xShape = storageShape->GetOriginShape();
    int32_t xDimNum = xShape.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += xDimNum;
    if (dim < 0 || dim >= xDimNum) dim = 0;

    uint32_t outer = 1;
    for (int32_t i = 0; i < dim; ++i) outer *= xShape.GetDim(i);
    uint32_t reduce = xShape.GetDim(dim);
    uint32_t inner = 1;
    for (int32_t i = dim + 1; i < xDimNum; ++i) inner *= xShape.GetDim(i);

    uint32_t outputSize = outer * inner;   // total number of output elements

    // Estimate tile length: number of output elements processed per iteration.
    // For each output element, we need to compare 'reduce' floats. We keep local buffer for one float comparison.
    // Conservative: process at most 256 output elements per tile.
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * reduce + sizeof(int64_t));
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = ALIGN_ELEMS;
    if (outputSize < maxTile) {
        uint32_t aligned = AlignDown(outputSize, ALIGN_ELEMS);
        maxTile = (aligned > 0) ? aligned : outputSize;
    }
    uint32_t tileLen = maxTile;

    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(outputSize, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_outerDim(outer);
    tiling.set_reduceDim(reduce);
    tiling.set_innerDim(inner);
    tiling.set_blockDim(blockDim);
    tiling.set_tileLen(tileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    // Output shape: same as input but with reduced dimension removed.
    int32_t xDimNum = x_shape->GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += xDimNum;
    if (dim < 0 || dim >= xDimNum) dim = 0;

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> newDims;
    for (int32_t i = 0; i < xDimNum; ++i) {
        if (i != dim) newDims.push_back(x_shape->GetDim(i));
    }
    if (newDims.empty()) newDims.push_back(1);  // scalar output
    y_shape->SetDimNum(newDims.size());
    for (size_t i = 0; i < newDims.size(); ++i) {
        y_shape->SetDim(i, newDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class ArgmaxOverADimensionCustom : public OpDef {
public:
    explicit ArgmaxOverADimensionCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT64})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(ArgmaxOverADimensionCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t reduce, uint32_t inner,
                                uint32_t tileLen)
    {
        outer_ = outer;
        reduce_ = reduce;
        inner_ = inner;
        tileLen_ = tileLen;
        outputSize_ = outer * inner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, outer * reduce * inner);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, outputSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, reduce_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(int64_t));
        // tmpBuf for one element comparison
        pipe_.InitBuffer(tmpBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outputSize_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride loop over output elements
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= outputSize_) return;

        while (true)
        {
            uint32_t curLen = outputSize_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // For each output element in this tile, compute argmax
            AscendC::LocalTensor<int64_t> yLocal = outQueueY_.AllocTensor<int64_t>();
            for (uint32_t i = 0; i < curLen; ++i)
            {
                uint32_t pos = base + i;
                uint32_t outerIdx = pos / inner_;
                uint32_t innerIdx = pos % inner_;
                // Input base address for this reduction:
                // base_addr = (outerIdx * reduce * inner) + innerIdx
                // stride = inner
                uint32_t inBase = outerIdx * reduce_ * inner_ + innerIdx;
                int64_t maxIdx = 0;
                float maxVal = -__builtin_huge_valf();
                for (uint32_t k = 0; k < reduce_; ++k)
                {
                    // Load one float from global memory
                    AscendC::LocalTensor<float> tmp = tmpBuf_.Get<float>();
                    AscendC::DataCopy(tmp, xGm_[inBase + k * inner_], 1);
                    float val = tmp.GetValue(0);
                    if (val > maxVal) {
                        maxVal = val;
                        maxIdx = static_cast<int64_t>(k);
                    }
                }
                yLocal.SetValue(i, maxIdx);
            }

            outQueueY_.EnQue(yLocal);

            // Copy out
            AscendC::LocalTensor<int64_t> out = outQueueY_.DeQue<int64_t>();
            AscendC::DataCopy(yGm_[base], out, curLen);
            outQueueY_.FreeTensor(out);

            base += strideElems;
            if (base >= outputSize_) break;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t outer_ {0};
    uint32_t reduce_ {0};
    uint32_t inner_ {0};
    uint32_t tileLen_ {0};
    uint32_t outputSize_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.outerDim, tiling_data.reduceDim, tiling_data.innerDim, tiling_data.tileLen);
    op.Process();
}
