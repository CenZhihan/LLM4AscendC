static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t HARD_CAP_TILEK = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmBiasAddHardtanhMishGroupNormCustomTilingData tiling;

    // Input shapes: x[N, C], weight[K, C] with N=batch, C=inFeatures, K=outFeatures
    const auto &x_shape = context->GetInputShape(0)->GetOriginShape();
    const auto &w_shape = context->GetInputShape(1)->GetOriginShape();
    uint32_t batch = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t inFeatures = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t outFeatures = static_cast<uint32_t>(w_shape.GetDim(0));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t numGroups = attrs->GetAttrPointer<int32_t>(0)[0];
    float minVal = attrs->GetAttrPointer<float>(1)[0];
    float maxVal = attrs->GetAttrPointer<float>(2)[0];
    float eps = attrs->GetAttrPointer<float>(3)[0];

    uint32_t groupSize = outFeatures / static_cast<uint32_t>(numGroups);
    // Ensure divisibility
    if (outFeatures != groupSize * static_cast<uint32_t>(numGroups)) {
        return ge::GRAPH_FAILED; // Not divisible, should not happen
    }

    // Compute tileK from UB budget
    // Per row and per group we need:
    //    weight tile: groupSize * tileK floats
    //    x tile     : tileK floats
    //    group buffer: groupSize floats
    //    temp for mish: groupSize floats (can reuse)
    // Total floats = groupSize * tileK + tileK + 2*groupSize
    // Total bytes = 4 * (groupSize * tileK + tileK + 2*groupSize)
    // Solve for tileK:
    uint32_t maxTileK = (UB_BUDGET_BYTES / 4 - 2 * groupSize) / (groupSize + 1);
    maxTileK = AlignDown(maxTileK, ALIGN_ELEMS);
    if (maxTileK == 0) maxTileK = ALIGN_ELEMS;
    if (maxTileK > HARD_CAP_TILEK) maxTileK = AlignDown(HARD_CAP_TILEK, ALIGN_ELEMS);
    uint32_t tileK = (maxTileK > inFeatures) ? AlignDown(inFeatures, ALIGN_ELEMS) : maxTileK;
    if (tileK == 0) tileK = inFeatures; // fallback

    // Block dimension: distribute rows among cores, each core processes multiple rows
    uint32_t blockDim = batch;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(batch * outFeatures);
    tiling.set_batch(batch);
    tiling.set_inFeatures(inFeatures);
    tiling.set_outFeatures(outFeatures);
    tiling.set_numGroups(static_cast<uint32_t>(numGroups));
    tiling.set_groupSize(groupSize);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);
    tiling.set_minVal(minVal);
    tiling.set_maxVal(maxVal);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
                                GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t totalLength, uint32_t batch,
                                uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t numGroups, uint32_t groupSize,
                                uint32_t tileK, float minVal, float maxVal, float eps)
    {
        totalLen_ = totalLength;
        batch_ = batch;
        inFeat_ = inFeatures;
        outFeat_ = outFeatures;
        numGroups_ = numGroups;
        groupSize_ = groupSize;
        tileK_ = tileK;
        minVal_ = minVal;
        maxVal_ = maxVal;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures * inFeatures);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures);
        betaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        // Temporary buffers
        pipe_.InitBuffer(xLocal_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(wLocal_, 1, groupSize_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(accLocal_, 1, groupSize_ * sizeof(float));
        pipe_.InitBuffer(tmpLocal_, 1, groupSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || groupSize_ == 0) return;
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes rows with grid stride
        uint32_t startRow = blockIdx;
        while (startRow < batch_) {
            ProcessRow(startRow);
            startRow += blockNum;
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        for (uint32_t g = 0; g < numGroups_; ++g) {
            uint32_t channelStart = g * groupSize_;
            uint32_t groupOffset = row * outFeat_ + channelStart;

            // Initialize accLocal with (lin_bias + bias) for this group
            AscendC::LocalTensor<float> acc = accLocal_.Get();
            AscendC::LocalTensor<float> tmp = tmpLocal_.Get();
            AscendC::DataCopy(acc, linBiasGm_[channelStart], groupSize_);
            AscendC::Add(acc, acc, biasGm_[channelStart], groupSize_);

            // Accumulate over inFeatures tiles
            for (uint32_t k = 0; k < inFeat_; k += tileK_) {
                uint32_t curK = (k + tileK_ <= inFeat_) ? tileK_ : (inFeat_ - k);
                // Load x tile
                AscendC::LocalTensor<float> xTile = xLocal_.Get();
                AscendC::DataCopy(xTile, xGm_[row * inFeat_ + k], curK);
                // Load weight tile for this group: weight[channelStart:channelStart+groupSize][k:k+curK]
                AscendC::LocalTensor<float> wTile = wLocal_.Get();
                // Weight is stored as (outFeat_, inFeat_) in row-major.
                // For each channel i in group, copy row i's segment from k to k+curK.
                // We'll copy in a loop over groupSize (small). Could optimize but acceptable.
                for (uint32_t i = 0; i < groupSize_; ++i) {
                    uint32_t wRow = channelStart + i;
                    float* wRowGm = wGm_.GetPtr(wRow * inFeat_ + k);
                    AscendC::DataCopy(wTile[i * curK], wRowGm, curK);
                }
                // Vectorized outer product: acc += xTile * wTile^T (but wTile is groupSize x curK)
                // Use AscendC vector multiply-add: vmadd(xTile, wTile_rowwise)
                // We'll do manual loop over i: acc[i] += sum over curK of xTile[k] * wTile[i][k]
                for (uint32_t i = 0; i < groupSize_; ++i) {
                    AscendC::LocalTensor<float> wRow = wTile[i * curK];
                    AscendC::Mul(tmp, xTile, wRow, curK);
                    float sum = tmp.GetValue(0);
                    for (uint32_t j = 1; j < curK; ++j) {
                        sum += tmp.GetValue(j);
                    }
                    acc.SetValue(i, acc.GetValue(i) + sum);
                }
            }

            // hardtanh: clamp acc between minVal and maxVal
            AscendC::Mins(acc, acc, maxVal_, groupSize_);
            AscendC::Maxs(acc, acc, minVal_, groupSize_);

            // mish: y = x * tanh(softplus(x))
            for (uint32_t i = 0; i < groupSize_; ++i) {
                float xv = acc.GetValue(i);
                float softplus = (xv > 20.0f) ? xv : (xv < -20.0f) ? 0.0f : log1pf(expf(xv));
                float tanh_sp = tanhf(softplus);
                acc.SetValue(i, xv * tanh_sp);
            }

            // group norm: compute mean and variance for this group
            float sum = 0.0f, sumSq = 0.0f;
            for (uint32_t i = 0; i < groupSize_; ++i) {
                float v = acc.GetValue(i);
                sum += v;
                sumSq += v * v;
            }
            float mean = sum / static_cast<float>(groupSize_);
            float var = sumSq / static_cast<float>(groupSize_) - mean * mean;
            float rstd = 1.0f / sqrtf(var + eps_);

            // normalize and apply gamma, beta
            for (uint32_t i = 0; i < groupSize_; ++i) {
                float norm = (acc.GetValue(i) - mean) * rstd;
                float out = norm * gammaGm_.GetValue(channelStart + i) + betaGm_.GetValue(channelStart + i);
                // store to global output
                yGm_.SetValue(groupOffset + i, out);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xLocal_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wLocal_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accLocal_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t batch_ {0};
    uint32_t inFeat_ {0};
    uint32_t outFeat_ {0};
    uint32_t numGroups_ {0};
    uint32_t groupSize_ {0};
    uint32_t tileK_ {0};
    float minVal_ {-1.0f};
    float maxVal_ {1.0f};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
    GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.totalLength, tiling_data.batch,
            tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.numGroups, tiling_data.groupSize,
            tiling_data.tileK, tiling_data.minVal, tiling_data.maxVal, tiling_data.eps);
    op.Process();
}
