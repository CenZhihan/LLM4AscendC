kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelVanillaRnnHidden {
public:
    __aicore__ inline KernelVanillaRnnHidden() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                uint32_t T, uint32_t B, uint32_t I,
                                uint32_t H, uint32_t O)
    {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T_ * B_ * I_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B_ * H_);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H_ * (I_ + H_));
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O_ * H_);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, T_ * B_ * O_);

        // Initialize local buffers for persistent and temporary tensors
        pipe_.InitBuffer(hBuf_, H_ * sizeof(float));
        pipe_.InitBuffer(xBuf_, I_ * sizeof(float));
        pipe_.InitBuffer(preBuf_, H_ * sizeof(float));
        pipe_.InitBuffer(hNewBuf_, H_ * sizeof(float));
        pipe_.InitBuffer(yBuf_, O_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();
        uint32_t batchPerBlock = (B_ + blockNum - 1) / blockNum;
        uint32_t batchStart = blockIdx * batchPerBlock;
        uint32_t batchEnd = batchStart + batchPerBlock;
        if (batchEnd > B_) batchEnd = B_;

        // Allocate local tensors (reused across time steps)
        LocalTensor<float> hLocal = hBuf_.Get<float>();
        LocalTensor<float> xLocal = xBuf_.Get<float>();
        LocalTensor<float> pre = preBuf_.Get<float>();
        LocalTensor<float> hNew = hNewBuf_.Get<float>();
        LocalTensor<float> yLocal = yBuf_.Get<float>();

        for (uint32_t b = batchStart; b < batchEnd; b++) {
            // Copy initial hidden state h0[b] to hLocal
            DataCopy(hLocal, h0Gm_[b * H_], H_);

            for (uint32_t t = 0; t < T_; t++) {
                // Load x[t, b]
                uint32_t xOffset = t * B_ * I_ + b * I_;
                DataCopy(xLocal, xGm_[xOffset], I_);

                // Compute pre = b_i2h + x * w_i2h_x + h * w_i2h_h
                // w_i2h is stored as (H, I+H) in row-major order
                // w_i2h_x = first I columns, w_i2h_h = last H columns
                for (uint32_t j = 0; j < H_; j++) {
                    float sum = b_i2hGm_.GetValue(j);
                    uint32_t rowBase = j * (I_ + H_);
                    // contribution from x
                    for (uint32_t i = 0; i < I_; i++) {
                        sum += xLocal(i) * w_i2hGm_.GetValue(rowBase + i);
                    }
                    // contribution from h
                    for (uint32_t hIdx = 0; hIdx < H_; hIdx++) {
                        sum += hLocal(hIdx) * w_i2hGm_.GetValue(rowBase + I_ + hIdx);
                    }
                    pre(j) = sum;
                }

                // Tanh activation
                Tanh(hNew, pre, H_);

                // Copy hNew back to hLocal for next time step
                DataCopy(hLocal, hNew, H_);

                // Compute y = b_h2o + hNew * w_h2o (w_h2o stored as (O, H))
                for (uint32_t j = 0; j < O_; j++) {
                    float sum = b_h2oGm_.GetValue(j);
                    uint32_t rowBase = j * H_;
                    for (uint32_t k = 0; k < H_; k++) {
                        sum += hNew(k) * w_h2oGm_.GetValue(rowBase + k);
                    }
                    yLocal(j) = sum;
                }

                // Store output y[t, b]
                uint32_t yOffset = t * B_ * O_ + b * O_;
                DataCopy(yGm_[yOffset], yLocal, O_);
            }
        }
    }

private:
    TPipe pipe_;
    TBuf<TPosition::VECCALC> hBuf_;
    TBuf<TPosition::VECCALC> xBuf_;
    TBuf<TPosition::VECCALC> preBuf_;
    TBuf<TPosition::VECCALC> hNewBuf_;
    TBuf<TPosition::VECCALC> yBuf_;
    GlobalTensor<float> xGm_, h0Gm_, w_i2hGm_, b_i2hGm_, w_h2oGm_, b_h2oGm_, yGm_;
    uint32_t T_, B_, I_, H_, O_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHidden op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I,
            tiling_data.H, tiling_data.O);
    op.Process();
}
