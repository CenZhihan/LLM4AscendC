#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   anchor: 1 * tile, positive: 1 * tile, negative: 1 * tile
//   diff_ap/diff_an + squares: 2 * tile, accumulators: few
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;
    const gert::Shape *anchor_shape = context->GetInputShape(0);
    const auto &origin_shape = anchor_shape->GetOriginShape();
    uint32_t totalBatch = static_cast<uint32_t>(origin_shape.GetDim(0));
    uint32_t dim = static_cast<uint32_t>(origin_shape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // tileLen for feature dimension from UB budget: bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (dim < tileLen) {
        const uint32_t aligned = AlignDown(dim, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : dim;
    }

    // blockDim: use multiple cores up to 24
    uint32_t blockDim = totalBatch;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_totalBatch(totalBatch);
    tiling.set_dim(dim);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
                                uint32_t totalBatch, uint32_t dim, uint32_t tileLen,
                                float margin)
    {
        batchSize_ = totalBatch;
        dim_ = dim;
        tileLen_ = tileLen;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, batchSize_ * dim_);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, batchSize_ * dim_);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, batchSize_ * dim_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, batchSize_);

        pipe_.InitBuffer(inQueueAnchor_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueuePositive_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueNegative_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueLoss_, BUFFER_NUM, sizeof(float));   // scalar per sample
        // temporary for diff_sq accumulation (used inside ComputeOneSample)
        pipe_.InitBuffer(tmpDiffAp_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpDiffAn_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || dim_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride over batch indices
        const uint32_t stride = blockNum;
        for (uint32_t idx = blockIdx; idx < batchSize_; idx += stride) {
            ComputeOneSample(idx);
        }
    }

private:
    __aicore__ inline void ComputeOneSample(uint32_t sampleIdx)
    {
        float sumAp = 0.0f;
        float sumAn = 0.0f;
        const uint32_t baseOffset = sampleIdx * dim_;

        for (uint32_t offset = 0; offset < dim_; offset += tileLen_) {
            uint32_t len = tileLen_;
            if (offset + len > dim_) len = dim_ - offset;

            // Load anchor, positive, negative segments
            uint32_t anchorOffset = baseOffset + offset;
            uint32_t posOffset = baseOffset + offset;
            uint32_t negOffset = baseOffset + offset;

            AscendC::LocalTensor<float> anchorLocal = inQueueAnchor_.AllocTensor<float>();
            AscendC::LocalTensor<float> positiveLocal = inQueuePositive_.AllocTensor<float>();
            AscendC::LocalTensor<float> negativeLocal = inQueueNegative_.AllocTensor<float>();

            AscendC::DataCopy(anchorLocal, anchorGm_[anchorOffset], len);
            AscendC::DataCopy(positiveLocal, positiveGm_[posOffset], len);
            AscendC::DataCopy(negativeLocal, negativeGm_[negOffset], len);

            // Compute diff_ap = anchor - positive, squared accumulate
            AscendC::LocalTensor<float> diffAp = tmpDiffAp_.Get<float>();
            AscendC::Sub(diffAp, anchorLocal, positiveLocal, len);
            AscendC::Mul(diffAp, diffAp, diffAp, len);  // square
            for (uint32_t i = 0; i < len; ++i) {
                sumAp += diffAp.GetValue(i);
            }

            // Compute diff_an = anchor - negative, squared accumulate
            AscendC::LocalTensor<float> diffAn = tmpDiffAn_.Get<float>();
            AscendC::Sub(diffAn, anchorLocal, negativeLocal, len);
            AscendC::Mul(diffAn, diffAn, diffAn, len);
            for (uint32_t i = 0; i < len; ++i) {
                sumAn += diffAn.GetValue(i);
            }

            // Free allocated local tensors
            inQueueAnchor_.FreeTensor(anchorLocal);
            inQueuePositive_.FreeTensor(positiveLocal);
            inQueueNegative_.FreeTensor(negativeLocal);
        }

        // Compute distances and loss
        float dAp = sqrt(sumAp);
        float dAn = sqrt(sumAn);
        float loss = dAp - dAn + margin_;
        if (loss < 0.0f) loss = 0.0f;

        // Store loss for this sample
        AscendC::LocalTensor<float> lossLocal = outQueueLoss_.AllocTensor<float>();
        lossLocal.SetValue(0, loss);
        AscendC::DataCopy(lossGm_[sampleIdx], lossLocal, 1);
        outQueueLoss_.FreeTensor(lossLocal);
    }

private:
    AscendC::TPipe pipe_;
    // input queues for anchor, positive, negative (each tile)
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePositive_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNegative_;
    // output queue for scalar loss per sample
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueLoss_;
    // temporary tensors for diff and square (reused)
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpDiffAp_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpDiffAn_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t batchSize_{0};
    uint32_t dim_{0};
    uint32_t tileLen_{0};
    float margin_{1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalBatch, tiling_data.dim, tiling_data.tileLen,
            tiling_data.margin);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'triplet_margin_loss_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, anchor, positive, negative, margin_tensor):
        # margin is a single-element tensor in original forward; extract scalar
        margin = float(margin_tensor.reshape(-1)[0].item()) if margin_tensor.numel() else 0.0
        # custom kernel returns per-sample loss; then reduce mean to get scalar
        loss_vec = getattr(custom_ops_lib, _FN)(anchor, positive, negative, margin)
        return torch.mean(loss_vec)
