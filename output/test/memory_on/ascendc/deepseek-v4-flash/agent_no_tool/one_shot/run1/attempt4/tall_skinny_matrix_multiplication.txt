host_operator_src="""

#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 448 * 1024; // Cube L0A+L0B+L0C approx
static constexpr uint32_t CUBE_ALIGN = 16;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    // Fix: GetInputShape returns StorageShape*, use GetOriginShape()
    const gert::StorageShape *a_storage = context->GetInputShape(0);
    const gert::StorageShape *b_storage = context->GetInputShape(1);
    const gert::Shape &a_shape = a_storage->GetOriginShape();
    const gert::Shape &b_shape = b_storage->GetOriginShape();
    uint32_t m = a_shape.GetDim(0);
    uint32_t k = a_shape.GetDim(1);
    uint32_t n = b_shape.GetDim(1);

    // Align to cube units, but keep at least 1
    if (m >= CUBE_ALIGN) m = AlignDown(m, CUBE_ALIGN);
    if (k >= CUBE_ALIGN) k = AlignDown(k, CUBE_ALIGN);
    if (n >= CUBE_ALIGN) n = AlignDown(n, CUBE_ALIGN);

    // Heuristic tile sizes
    uint32_t tileM = 128;
    uint32_t tileK = 64;
    uint32_t tileN = 128;

    // Ensure tile sizes fit UB budget (4 bytes per float)
    uint32_t bytesPerTile = (tileM * tileK + tileK * tileN + tileM * tileN) * 4;
    while (bytesPerTile > UB_BUDGET_BYTES && tileM > 16) {
        tileM >>= 1;
        tileN >>= 1;
        bytesPerTile = (tileM * tileK + tileK * tileN + tileM * tileN) * 4;
    }

    if (m < tileM) tileM = m;
    if (k < tileK) tileK = k;
    if (n < tileN) tileN = n;

    uint32_t numTilesM = CeilDiv(m, tileM);
    uint32_t blockDim = numTilesM;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(m * n);
    tiling.set_m(m);
    tiling.set_k(k);
    tiling.set_n(n);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *a_storage = context->GetInputShape(0);
    const gert::StorageShape *b_storage = context->GetInputShape(1);
    const gert::Shape &a_shape = a_storage->GetOriginShape();
    const gert::Shape &b_shape = b_storage->GetOriginShape();
    uint32_t m = a_shape.GetDim(0);
    uint32_t n = b_shape.GetDim(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    *c_shape = gert::Shape({m, n});
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:
    explicit TallSkinnyMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("a")

kernel_src="""

#include "kernel_operator.h"
#include "matmul_api.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t m, uint32_t k, uint32_t n,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        m_ = m; k_ = k; n_ = n;
        tileM_ = tileM; tileK_ = tileK; tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)a, m * k);
        bGm_.SetGlobalBuffer((__gm__ float*)b, k * n);
        cGm_.SetGlobalBuffer((__gm__ float*)c, m * n);
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || k_ == 0 || n_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Each block processes a contiguous chunk of M rows
        uint32_t rowsPerBlock = (m_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = (rowStart + rowsPerBlock > m_) ? m_ : rowStart + rowsPerBlock;
        if (rowStart >= m_) return;

        // Prepare cube computation
        AscendC::Matmul<AscendC::CubeFormat::ND, AscendC::CubeFormat::ND,
                        AscendC::CubeFormat::ND> mm;
        AscendC::CubeTensor<float> aLocal, bLocal, cLocal;

        // Loop over rows in tileM steps
        for (uint32_t i = rowStart; i < rowEnd; i += tileM_) {
            uint32_t curTileM = (i + tileM_ > rowEnd) ? rowEnd - i : tileM_;
            for (uint32_t j = 0; j < n_; j += tileN_) {
                uint32_t curTileN = (j + tileN_ > n_) ? n_ - j : tileN_;
                // For matrix multiplication: C[i:i+curTileM, j:j+curTileN] = A[i:i+curTileM, :] * B[:, j:j+curTileN]
                // Use cube to compute one tile
                aLocal.SetGlobalBuffer((__gm__ float*)aGm_.GetPhyAddr() + i * k_, curTileM * k_);
                bLocal.SetGlobalBuffer((__gm__ float*)bGm_.GetPhyAddr() + j, k_ * curTileN);
                cLocal.SetGlobalBuffer((__gm__ float*)cGm_.GetPhyAddr() + i * n_ + j, curTileM * curTileN);

                mm.SetOrgShape(AscendC::CUBE_M, curTileM);
                mm.SetOrgShape(AscendC::CUBE_K, k_);
                mm.SetOrgShape(AscendC::CUBE_N, curTileN);
                mm.SetIsFormatA(false); // ND
                mm.SetIsFormatB(false);
                mm.SetIsFormatC(false);

                // For simplicity, we do a single cube call. In practice tiling inside cube is needed.
                // Here we assume cube handles tiling internally with provided tile sizes.
                mm.template Compute<AscendC::CCEComputeType::CCT_DEFAULT>(aLocal, bLocal, cLocal);
            }
        }
    }

private:
    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;
    uint32_t m_, k_, n_;
    uint32_t tileM_, tileK_, tileN_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(a, b, c,
            tiling_data.m, tiling_data.k, tiling_data.n,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
