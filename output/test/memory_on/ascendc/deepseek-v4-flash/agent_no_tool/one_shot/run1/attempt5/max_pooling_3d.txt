#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;
    const gert::StorageShape *inShape = context->GetInputShape(0);
    const auto &originShape = inShape->GetOriginShape();
    int32_t inN = static_cast<int32_t>(originShape.GetDim(0));
    int32_t inC = static_cast<int32_t>(originShape.GetDim(1));
    int32_t inD = static_cast<int32_t>(originShape.GetDim(2));
    int32_t inH = static_cast<int32_t>(originShape.GetDim(3));
    int32_t inW = static_cast<int32_t>(originShape.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];
    bool ceilMode = attrs->GetAttrPointer<bool>(3)[0];

    // Compute output dimensions using PyTorch formula (floor default)
    int32_t outD = (inD + 2 * padding - (kernelSize - 1) - 1) / stride + 1;
    if (ceilMode && ((outD - 1) * stride + kernelSize - 2 * padding < inD))
        outD += 1;
    int32_t outH = (inH + 2 * padding - (kernelSize - 1) - 1) / stride + 1;
    if (ceilMode && ((outH - 1) * stride + kernelSize - 2 * padding < inH))
        outH += 1;
    int32_t outW = (inW + 2 * padding - (kernelSize - 1) - 1) / stride + 1;
    if (ceilMode && ((outW - 1) * stride + kernelSize - 2 * padding < inW))
        outW += 1;

    uint32_t totalOutputElems = static_cast<uint32_t>(inN) * inC * outD * outH * outW;

    // Compute tile length from UB budget: we need space for input tile? but we will
    // read directly from GM for each output element, so UB usage is small.
    // Use a conservative tile size based on total output elements and block limit.
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // per element overhead
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOutputElems < tileLen) {
        uint32_t aligned = AlignDown(totalOutputElems, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutputElems;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutputElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_ceilMode(ceilMode);
    tiling.set_inN(inN);
    tiling.set_inC(inC);
    tiling.set_inD(inD);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElems, uint32_t tileLength,
                                int32_t kernelSize, int32_t stride, int32_t padding, bool ceilMode,
                                int32_t inN, int32_t inC, int32_t inD, int32_t inH, int32_t inW,
                                int32_t outD, int32_t outH, int32_t outW)
    {
        totalOutputElems_ = totalOutputElems;
        tileLen_ = tileLength;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        ceilMode_ = ceilMode;
        inN_ = inN;
        inC_ = inC;
        inD_ = inD;
        inH_ = inH;
        inW_ = inW;
        outD_ = outD;
        outH_ = outH;
        outW_ = outW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, inN_ * inC_ * inD_ * inH_ * inW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElems_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // No temp queue needed for elementwise max
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride loop over output elements
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutputElems_) return;

        while (base < totalOutputElems_) {
            uint32_t curLen = totalOutputElems_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Compute output elements for this tile
            AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();
            for (uint32_t i = 0; i < curLen; ++i) {
                uint32_t globalIdx = base + i;
                // Decompose global output index into (n, c, d, h, w)
                int32_t n = globalIdx / (inC_ * outD_ * outH_ * outW_);
                int32_t rem = globalIdx % (inC_ * outD_ * outH_ * outW_);
                int32_t c = rem / (outD_ * outH_ * outW_);
                rem %= (outD_ * outH_ * outW_);
                int32_t d = rem / (outH_ * outW_);
                rem %= (outH_ * outW_);
                int32_t h = rem / outW_;
                int32_t w = rem % outW_;

                // Compute window bounds
                int32_t dStart = d * stride_ - padding_;
                int32_t hStart = h * stride_ - padding_;
                int32_t wStart = w * stride_ - padding_;
                int32_t dEnd = dStart + kernelSize_;
                int32_t hEnd = hStart + kernelSize_;
                int32_t wEnd = wStart + kernelSize_;

                float maxVal = -FLT_MAX;
                for (int32_t kd = dStart; kd < dEnd; ++kd) {
                    if (kd < 0 || kd >= inD_) continue;
                    for (int32_t kh = hStart; kh < hEnd; ++kh) {
                        if (kh < 0 || kh >= inH_) continue;
                        for (int32_t kw = wStart; kw < wEnd; ++kw) {
                            if (kw < 0 || kw >= inW_) continue;
                            int64_t inIdx = ((int64_t)n * inC_ + c) * inD_ * inH_ * inW_
                                           + kd * inH_ * inW_ + kh * inW_ + kw;
                            float val = xGm_.GetValue(inIdx);
                            if (val > maxVal) maxVal = val;
                        }
                    }
                }
                outLocal.SetValue(i, maxVal);
            }

            outQueueY_.EnQue<float>(outLocal);
            AscendC::LocalTensor<float> outLocalDeque = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], outLocalDeque, curLen);
            outQueueY_.FreeTensor(outLocalDeque);

            base += strideElems;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutputElems_ {0};
    uint32_t tileLen_ {0};
    int32_t kernelSize_ {3};
    int32_t stride_ {2};
    int32_t padding_ {0};
    bool ceilMode_ {false};
    int32_t inN_ {0}, inC_ {0}, inD_ {0}, inH_ {0}, inW_ {0};
    int32_t outD_ {0}, outH_ {0}, outW_ {0};
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.totalOutputElems, tiling_data.tileLength,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding, tiling_data.ceilMode,
            tiling_data.inN, tiling_data.inC, tiling_data.inD, tiling_data.inH, tiling_data.inW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW);
    op.Process();
}
