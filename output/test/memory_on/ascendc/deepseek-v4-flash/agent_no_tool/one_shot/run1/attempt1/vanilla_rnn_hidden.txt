kernel_src="""

#include "kernel_operator.h"
#include "matmul_api.h"

using namespace AscendC;

class KernelVanillaRnnHidden {
public:
    __aicore__ inline KernelVanillaRnnHidden() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                uint32_t T, uint32_t B, uint32_t I,
                                uint32_t H, uint32_t O, uint32_t K)
    {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O; K_ = K;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T * B * I);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B * H);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H * K);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O * H);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O);
        yGm_.SetGlobalBuffer((__gm__ float*)y, T * B * O);

        // Allocate ping-pong buffers for intermediate hidden and projected states
        // We need temporary storage for each batch processing: hidden state H, input+hidden concatenated (K), output O
        // We use local tensors (size based on max dimension), but for simplicity we allocate global workspace or use smaller tiles.
        // Instead, we reuse global memory for workspace (not used). We'll do matmul tile-wise.

        // We'll use vectorized operations and local matmul tiling.
        // For simplicity, we assume dimensions fit within UB after tiling.
        // We'll process one batch at a time within each block, using grid stride over batches.
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();
        // Distribute batches across blocks
        uint32_t batchPerBlock = (B_ + blockNum - 1) / blockNum;
        uint32_t batchStart = blockIdx * batchPerBlock;
        uint32_t batchEnd = batchStart + batchPerBlock;
        if (batchEnd > B_) batchEnd = B_;

        // For each batch in this block
        for (uint32_t b = batchStart; b < batchEnd; b++) {
            // Local workspace for this batch
            // We'll allocate local tensors inside the loop; might be heavy but acceptable for demo.
            // In practice, allocate once outside and reuse, but for clarity we keep it simple.
            // Use TQue for intermediate states? To avoid reallocation, we manage manually.

            // Initialize hidden state for this batch
            // We'll keep h in global memory? Actually we need to read h0 and update h for each time step.
            // We'll store h in a local tensor of length H.
            LocalTensor<float> hLocal = pipe_.AllocTensor<float>(H_);
            // Copy h0 for this batch
            DataCopy(hLocal, h0Gm_[b * H_], H_);

            // Output buffer for this batch: y_t (O) per time step
            LocalTensor<float> yLocal = pipe_.AllocTensor<float>(O_);
            // Workspace for concatenated input (K)
            LocalTensor<float> inLocal = pipe_.AllocTensor<float>(K_);
            // Workspace for pre-activation (H)
            LocalTensor<float> preLocal = pipe_.AllocTensor<float>(H_);
            // Workspace for tanh output (H)
            LocalTensor<float> hNewLocal = pipe_.AllocTensor<float>(H_);

            // Matmul objects for i2h and h2o
            Matmul<MatmulType<TPosition::VECCALC, TPosition::VECCALC>> mm_i2h;
            Matmul<MatmulType<TPosition::VECCALC, TPosition::VECCALC>> mm_h2o;

            // Setup matmul parameters: we need to tile matrices. For simplicity, we do full matmul.
            // We'll set up fragment sizes based on dimensions.
            // Use default tiling strategy (auto)
            // In practice, need to set up MatmulTiling, but we skip for brevity.
            // Assume dimensions are small enough to fit in UB.

            for (uint32_t t = 0; t < T_; t++) {
                // Build concatenated input: [x[t, b, :], hLocal]
                // First copy x[t,b,:] of length I into inLocal
                uint32_t xOffset = t * B_ * I_ + b * I_;
                DataCopy(inLocal, xGm_[xOffset], I_);
                // Then copy hLocal (H) after I
                DataCopy(inLocal[I_], hLocal, H_);
                // Now inLocal contains the concatenated vector of length K

                // Compute pre = inLocal * w_i2h^T + b_i2h? Wait: original code does inp @ w_i2h.t() + b_i2h
                // w_i2h shape is (H, K). Actually w_i2h.t() is (K, H). So pre = inp (1xK) * w_i2h.t() (KxH) -> (1xH) + b_i2h (H)
                // So we need to multiply inLocal (K) by w_i2h transposed. w_i2h in memory is HxK, we treat it as (H, K) but we need (K, H).
                // We'll compute as pre = inLocal * (w_i2h^T) = (1xK) * (KxH) => we can use Matmul with left matrix (1,K) and right matrix (K,H) stored as (H,K) but need transpose.
                // Alternatively, we can store w_i2h in column-major? Simpler: do gemv-like: for each output element j, dot product of inLocal with column j of w_i2h^T which is row j of w_i2h.
                // We'll implement manually using vectorized dot product: for j in H: pre[j] = sum over k: inLocal[k] * w_i2h[j][k] + b_i2h[j]
                // This is inefficient but works. For performance, we should use Matmul with proper transposition.
                // We'll use Matmul with A=(1, K), B=(K, H) but B is stored as (H, K) in row-major. To get B in (K,H) we need to transpose.
                // Instead, we can store w_i2h transposed in global memory? Not possible.
                // For simplicity, we'll implement a loop over H to compute dot products.

                for (uint32_t j = 0; j < H_; j++) {
                    float sum = b_i2hGm_.GetValue(j);
                    for (uint32_t k = 0; k < K_; k++) {
                        sum += inLocal(k) * w_i2hGm_.GetValue(j * K_ + k);
                    }
                    preLocal(j) = sum;
                }

                // Apply tanh
                Tanh(hNewLocal, preLocal, H_);

                // Copy hNewLocal to hLocal for next time step
                DataCopy(hLocal, hNewLocal, H_);

                // Output projection: y_t = hNewLocal * w_h2o.t() + b_h2o
                // w_h2o is (O, H), transposed is (H, O). y_t (1xO) = hNewLocal (1xH) * w_h2o.t() (HxO)
                // Again loop over O for dot products.
                for (uint32_t j = 0; j < O_; j++) {
                    float sum = b_h2oGm_.GetValue(j);
                    for (uint32_t k = 0; k < H_; k++) {
                        sum += hNewLocal(k) * w_h2oGm_.GetValue(j * H_ + k);
                    }
                    yLocal(j) = sum;
                }

                // Write y_t to global output
                uint32_t yOffset = t * B_ * O_ + b * O_;
                DataCopy(yGm_[yOffset], yLocal, O_);
            }

            // Free local tensors for this batch
            pipe_.FreeTensor(hLocal);
            pipe_.FreeTensor(yLocal);
            pipe_.FreeTensor(inLocal);
            pipe_.FreeTensor(preLocal);
            pipe_.FreeTensor(hNewLocal);
        }
    }

private:
    TPipe pipe_;
    GlobalTensor<float> xGm_, h0Gm_, w_i2hGm_, b_i2hGm_, w_h2oGm_, b_h2oGm_, yGm_;
    uint32_t T_, B_, I_, H_, O_, K_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHidden op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I,
            tiling_data.H, tiling_data.O, tiling_data.K);
    op.Process();
}
