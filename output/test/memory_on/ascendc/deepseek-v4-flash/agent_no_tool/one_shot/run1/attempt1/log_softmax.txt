host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: each row needs: x, exp, sum, max, one buffer for log-sum, output.
// Roughly 5 * tile elements. tile = cols.
// bytes = 20 * cols. Also need workspace for reduction ops.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t MAX_COLS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(shape.GetShapeSize());

    // Assume 2D shape: (batch_size, feature_dim)
    uint32_t rows = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t cols = 1;
    if (shape.GetDimNum() >= 2) {
        cols = static_cast<uint32_t>(shape.GetDim(1));
    } else {
        // fallback for 1D
        rows = 1;
        cols = totalLength;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = 1;
    if (attrs->GetAttrNum() > 0) {
        dim = *attrs->GetAttrPointer<int32_t>(0);
    }

    // tileLength = cols (process one row at a time)
    uint32_t maxTile = UB_BUDGET_BYTES / 20; // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > MAX_COLS) maxTile = AlignDown(MAX_COLS, ALIGN_ELEMS);

    uint32_t tileLen = (cols <= maxTile) ? cols : maxTile;
    // If cols is larger than UB, we need to split row into multiple tiles (more complex).
    // For simplicity, we assume cols <= maxTile or we adjust blockDim.
    // But here we set tileLen = cols (assuming it fits), else use maxTile and handle in kernel.
    if (cols <= maxTile) {
        tileLen = cols;
    } else {
        tileLen = maxTile;
    }

    // Adaptive blockDim: each block processes a group of rows.
    uint32_t rowsPerBlock = 1;
    uint32_t blockDim = rows;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    // Ensure blockDim <= rows to have work
    if (blockDim > rows) blockDim = rows;
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t rows, uint32_t cols, int32_t dim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;   // = cols (or part of cols if split)
        rows_ = rows;
        cols_ = cols;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Buffers for one row: x (tileLen), exp, max (scalar but use tile for reduction), sum (scalar)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpMax_, 1, tileLen_ * sizeof(float));   // for max temp
        pipe_.InitBuffer(tmpExp_, 1, tileLen_ * sizeof(float));   // for exp temp
        pipe_.InitBuffer(tmpSum_, 1, tileLen_ * sizeof(float));    // for sum temp (actually scalar but use tile for reduce)
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a contiguous set of rows
        uint32_t rowsPerBlock = (rows_ + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = (startRow + rowsPerBlock > rows_) ? rows_ : startRow + rowsPerBlock;

        for (uint32_t r = startRow; r < endRow; r++) {
            processRow(r);
        }
    }

private:
    __aicore__ inline void processRow(uint32_t rowIdx)
    {
        uint32_t offset = rowIdx * cols_;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offset], cols_);
        inQueueX_.EnQue(xLocal);

        // ---- Step 1: compute max over row ----
        xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maxLocal = tmpMax_.AllocTensor<float>();
        // ReduceMax on last dim (cols) -> produces scalar in first element
        AscendC::ReduceMax(maxLocal, xLocal, 1, 0, false, 1, cols_, false); // axis=1? Actually for vector, we use last-axis reduction
        float rowMax = maxLocal.GetValue(0);
        tmpMax_.FreeTensor(maxLocal);

        // ---- Step 2: compute exp(x - max) and sum ----
        AscendC::LocalTensor<float> expLocal = tmpExp_.AllocTensor<float>();
        // x - max
        AscendC::Subs(expLocal, xLocal, rowMax, cols_);
        // exp
        AscendC::Exp(expLocal, expLocal, cols_);
        // sum of exp
        AscendC::LocalTensor<float> sumLocal = tmpSum_.AllocTensor<float>();
        AscendC::ReduceSum(sumLocal, expLocal, 1, 0, false, 1, cols_, false);
        float logSum = logf(sumLocal.GetValue(0));
        tmpSum_.FreeTensor(sumLocal);

        // ---- Step 3: output = x - max - logSum ----
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        // y = (x - max) - logSum
        AscendC::Subs(yLocal, expLocal, rowMax, cols_);   // reuse expLocal as (x-max)
        AscendC::Subs(yLocal, yLocal, logSum, cols_);
        outQueueY_.EnQue(yLocal);
        tmpExp_.FreeTensor(expLocal);
        inQueueX_.FreeTensor(xLocal);

        // ---- Copy out ----
        yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], yLocal, cols_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpExp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSum_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t rows_ {0};
    uint32_t cols_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.rows, tiling_data.cols, tiling_data.dim);
    op.Process();
}
