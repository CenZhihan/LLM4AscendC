#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_MULT = 2048;
static constexpr uint32_t MAX_TILE_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)   { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const uint32_t rowSize =
        static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);

    uint32_t maxTile = UB_BUDGET_BYTES / (6 * sizeof(float));
    maxTile = AlignDown(maxTile, rowSize);
    if (maxTile < rowSize) maxTile = rowSize;
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, rowSize);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, rowSize);
        if (tileLen == 0) tileLen = totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    float invRowSize = 1.0f / static_cast<float>(rowSize);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_rowSize(rowSize);
    tiling.set_eps(*eps);
    tiling.set_invRowSize(invRowSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t rowSize, float eps, float invRowSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        rowSize_ = rowSize;
        eps_ = eps;
        invRowSize_ = invRowSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, rowSize_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta,  rowSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(tmpRowBuf_,    rowSize_ * sizeof(float));
        pipe_.InitBuffer(sqBuf_,        rowSize_ * sizeof(float));
        pipe_.InitBuffer(gammaBuf_,     rowSize_ * sizeof(float));
        pipe_.InitBuffer(betaBuf_,      rowSize_ * sizeof(float));
        pipe_.InitBuffer(sumBuf_,       sizeof(float));
        pipe_.InitBuffer(sumsqBuf_,     sizeof(float));

        pipe_.InitBuffer(inQueueX_,  BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));

        auto gammaLocal = gammaBuf_.Get<float>();
        AscendC::DataCopy(gammaLocal, gammaGm_, rowSize_);
        auto betaLocal = betaBuf_.Get<float>();
        AscendC::DataCopy(betaLocal, betaGm_, rowSize_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }
            Compute(curLen);
            CopyOut(base, curLen);
            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        auto xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        auto xLocal = inQueueX_.DeQue<float>();
        auto yLocal = outQueueY_.AllocTensor<float>();

        auto tmpRow   = tmpRowBuf_.Get<float>();
        auto tmpRowSq = sqBuf_.Get<float>();
        auto gammaLocal = gammaBuf_.Get<float>();
        auto betaLocal  = betaBuf_.Get<float>();
        auto sumScalar  = sumBuf_.Get<float>();
        auto sumsqScalar = sumsqBuf_.Get<float>();

        const uint32_t numRows = len / rowSize_;
        for (uint32_t r = 0; r < numRows; ++r)
        {
            uint32_t offset = r * rowSize_;

            // Copy row from xLocal to tmpRow (vecin -> veccalc)
            AscendC::DataCopy(tmpRow, xLocal, rowSize_, offset, 0);

            // Compute sum
            AscendC::ReduceSum(sumScalar, tmpRow, rowSize_, 1);
            float sumVal = sumScalar(0);

            // Compute sum of squares
            AscendC::Mul(tmpRowSq, tmpRow, tmpRow, rowSize_);
            AscendC::ReduceSum(sumsqScalar, tmpRowSq, rowSize_, 1);
            float sumsqVal = sumsqScalar(0);

            // Mean and variance
            float mean  = sumVal * invRowSize_;
            float var   = sumsqVal * invRowSize_ - mean * mean;
            float invStd = 1.0f / sqrt(var + eps_);

            // Normalize and apply gamma/beta
            AscendC::Adds(tmpRow, tmpRow, -mean, rowSize_);       // x - mean
            AscendC::Muls(tmpRow, tmpRow, invStd, rowSize_);      // (x - mean) * invStd
            AscendC::Mul(tmpRow, tmpRow, gammaLocal, rowSize_);   // * gamma
            AscendC::Add(tmpRow, tmpRow, betaLocal, rowSize_);    // + beta

            // Write result to yLocal at offset
            AscendC::DataCopy(yLocal, tmpRow, rowSize_, 0, offset);
        }

        inQueueX_.FreeTensor(xLocal);
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        auto yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpRowBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> sqBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> gammaBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> betaBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> sumBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> sumsqBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t rowSize_ {0};
    float eps_ {1e-5f};
    float invRowSize_ {0.0f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta,
                                                         GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.rowSize, tiling_data.eps, tiling_data.invRowSize);
    op.Process();
}
"""
