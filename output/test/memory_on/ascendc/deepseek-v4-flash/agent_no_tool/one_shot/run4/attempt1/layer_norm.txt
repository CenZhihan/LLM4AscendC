#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 128;             // satisfy alignment requirements
static constexpr uint32_t MAX_BLOCK_DIM = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LayerNormCustomTilingData tiling;

    const gert::StorageShape* inputShape = context->GetInputShape(0);
    const gert::StorageShape* gammaShape = context->GetInputShape(2);  // gamma index 2? actually order: x, gamma, beta -> indices 0,1,2
    const gert::StorageShape* betaShape = context->GetInputShape(2);
    (void)betaShape; // shape same as gamma

    uint32_t totalLength = static_cast<uint32_t>(inputShape->GetOriginShape().GetShapeSize());
    uint32_t innerSize = static_cast<uint32_t>(gammaShape->GetOriginShape().GetShapeSize());
    uint32_t outerSize = totalLength / innerSize;

    // Determine tile length from UB budget: we need 5 tile buffers (x, y, gamma, beta, tmp)
    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    // Tile length should not exceed innerSize, but we may need to tile inner dimension
    uint32_t tileLen = (innerSize < maxTile) ? innerSize : maxTile;
    // If innerSize is not a multiple of tileLen, we may have a small last tile, but that is okay.

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilon = attrs->GetAttrPointer<float>(0);

    uint32_t blockDim = outerSize;  // one block per row ideally
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_epsilon(*epsilon);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t outerSize, uint32_t innerSize,
                                uint32_t tileLength, float epsilon) {
        totalLen_ = totalLength;
        outerSize_ = outerSize;
        innerSize_ = innerSize;
        tileLen_ = tileLength;
        epsilon_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, innerSize_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, innerSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Allocate local queues for double buffering on x and y; others single buffer.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(gammaQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(betaQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (outerSize_ == 0 || innerSize_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Distribute rows to blocks
        uint32_t rowsPerBlock = (outerSize_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = (blockIdx + 1) * rowsPerBlock;
        if (rowEnd > outerSize_) rowEnd = outerSize_;
        if (rowStart >= outerSize_) return;

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            uint32_t rowOffset = row * innerSize_;
            ProcessOneRow(rowOffset);
        }
    }

private:
    __aicore__ inline void ProcessOneRow(uint32_t rowOffset) {
        // First pass: compute sum and sum of squares across tiles of this row
        float sum = 0.0f;
        float sum2 = 0.0f;

        uint32_t tiles = (innerSize_ + tileLen_ - 1) / tileLen_;
        uint32_t remaining = innerSize_;

        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;
            if (curTileLen == 0) break;

            // Copy x tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + t * tileLen_], curTileLen);
            inQueueX_.EnQue(xLocal);

            // Dequeue for computation
            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();

            // Compute sum of x
            AscendC::LocalTensor<float> sumScalarLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceAdd(sumScalarLocal, xDeque, curTileLen);
            float sumTile;
            AscendC::LocalTensor<float> sumHost = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(sumHost, sumScalarLocal, 1);
            // Actually ReduceAdd writes to a scalar local; we can copy to scalar variable via builtin:
            sum += sumScalarLocal.GetValue(0);  // pseudo-code; actual AscendC API: use tensor element
            // For simplicity, we will use a different approach: accumulate via vector addition
            // Instead, we can compute sum and sum2 using vector operations:
            // We'll compute sum = sum + reduce_add(x); but we need to extract scalar.
            // We'll use a workaround: maintain sum as a local tensor of 1 element and add.
            AscendC::FreeTensor(sumScalarLocal);
            AscendC::FreeTensor(sumHost);

            // Compute sum of squares
            AscendC::LocalTensor<float> xSq = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(xSq, xDeque, xDeque, curTileLen);
            AscendC::LocalTensor<float> sum2Scalar = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceAdd(sum2Scalar, xSq, curTileLen);
            sum2 += sum2Scalar.GetValue(0);
            AscendC::FreeTensor(xSq);
            AscendC::FreeTensor(sum2Scalar);

            inQueueX_.FreeTensor(xDeque);

            remaining -= curTileLen;
        }

        float mean = sum / static_cast<float>(innerSize_);
        float var = sum2 / static_cast<float>(innerSize_) - mean * mean;
        float invStd = 1.0f / sqrt(var + epsilon_);

        // Second pass: apply normalization
        remaining = innerSize_;
        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;
            if (curTileLen == 0) break;
            uint32_t tileOffset = t * tileLen_;

            // Copy x tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + tileOffset], curTileLen);
            inQueueX_.EnQue(xLocal);

            // Copy gamma tile
            AscendC::LocalTensor<float> gammaLocal = gammaQueue_.AllocTensor<float>();
            AscendC::DataCopy(gammaLocal, gammaGm_[tileOffset], curTileLen);
            gammaQueue_.EnQue(gammaLocal);

            // Copy beta tile
            AscendC::LocalTensor<float> betaLocal = betaQueue_.AllocTensor<float>();
            AscendC::DataCopy(betaLocal, betaGm_[tileOffset], curTileLen);
            betaQueue_.EnQue(betaLocal);

            // Dequeue all
            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> gammaDeque = gammaQueue_.DeQue<float>();
            AscendC::LocalTensor<float> betaDeque = betaQueue_.DeQue<float>();

            // Compute y = gamma * (x - mean) * invStd + beta
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(tmp, xDeque, mean, curTileLen);
            AscendC::Mul(tmp, tmp, invStd, curTileLen);
            AscendC::Mul(tmp, tmp, gammaDeque, curTileLen);
            AscendC::Add(tmp, tmp, betaDeque, curTileLen);

            // Write to y global
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, tmp, curTileLen);  // copy from tmp to yLocal
            outQueueY_.EnQue(yLocal);

            // Dequeue y and write to global
            AscendC::LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[rowOffset + tileOffset], yDeque, curTileLen);

            // Free all
            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xDeque);
            gammaQueue_.FreeTensor(gammaDeque);
            betaQueue_.FreeTensor(betaDeque);
            outQueueY_.FreeTensor(yDeque);

            remaining -= curTileLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> gammaQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> betaQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outerSize_ {0};
    uint32_t innerSize_ {0};
    uint32_t tileLen_ {0};
    float epsilon_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength, tiling_data.outerSize, tiling_data.innerSize,
            tiling_data.tileLength, tiling_data.epsilon);
    op.Process();
}

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):

        super(ModelNew, self).__init__()
        # Create learnable gamma and beta parameters
        self.gamma = torch.nn.Parameter(torch.ones(normalized_shape, dtype=torch.float32))
        self.beta = torch.nn.Parameter(torch.zeros(normalized_shape, dtype=torch.float32))

        fn = getattr(custom_ops_lib, _FN)
