static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t MAX_BLOCK_DIM = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LayerNormCustomTilingData tiling;

    const gert::StorageShape* inputShape = context->GetInputShape(0);
    const gert::StorageShape* gammaShape = context->GetInputShape(1);
    const gert::StorageShape* betaShape = context->GetInputShape(2);
    (void)betaShape;

    uint32_t totalLength = static_cast<uint32_t>(inputShape->GetOriginShape().GetShapeSize());
    uint32_t innerSize = static_cast<uint32_t>(gammaShape->GetOriginShape().GetShapeSize());
    uint32_t outerSize = totalLength / innerSize;

    // UB: x, gamma, beta, tmp (for broadcast) -> 4 tile buffers each tileLen floats
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = (innerSize < maxTile) ? innerSize : maxTile;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilon = attrs->GetAttrPointer<float>(0);

    uint32_t blockDim = outerSize;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_epsilon(*epsilon);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t outerSize, uint32_t innerSize,
                                uint32_t tileLength, float epsilon) {
        totalLen_ = totalLength;
        outerSize_ = outerSize;
        innerSize_ = innerSize;
        innerSizeFloat_ = static_cast<float>(innerSize);
        tileLen_ = tileLength;
        epsilon_ = epsilon;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, innerSize_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, innerSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Queues: x input double buffered, gamma/beta single, tmp for reduction/broadcast
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(gammaQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(betaQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(reduceQueue_, 1, 1 * sizeof(float));   // single element for reduction result
    }

    __aicore__ inline void Process() {
        if (outerSize_ == 0 || innerSize_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t rowsPerBlock = (outerSize_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = (blockIdx + 1) * rowsPerBlock;
        if (rowEnd > outerSize_) rowEnd = outerSize_;
        if (rowStart >= outerSize_) return;

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            uint32_t rowOffset = row * innerSize_;
            ProcessOneRow(rowOffset);
        }
    }

private:
    __aicore__ inline void ProcessOneRow(uint32_t rowOffset) {
        // First pass: compute sum and sum of squares across tiles
        float sum = 0.0f;
        float sum2 = 0.0f;

        uint32_t tiles = (innerSize_ + tileLen_ - 1) / tileLen_;
        uint32_t remaining = innerSize_;

        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;
            if (curTileLen == 0) break;

            // Copy x tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + t * tileLen_], curTileLen);
            inQueueX_.EnQue(xLocal);

            // Dequeue for computation
            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();

            // Compute sum of x using ReduceSum (writes to single element tensor)
            AscendC::LocalTensor<float> sumScalar = reduceQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumScalar, xDeque, curTileLen);
            sum += sumScalar.GetValue(0);
            reduceQueue_.FreeTensor(sumScalar);

            // Compute sum of squares: x * x -> ReduceSum
            AscendC::LocalTensor<float> xSq = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(xSq, xDeque, xDeque, curTileLen);
            AscendC::LocalTensor<float> sum2Scalar = reduceQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sum2Scalar, xSq, curTileLen);
            sum2 += sum2Scalar.GetValue(0);
            reduceQueue_.FreeTensor(sum2Scalar);
            tmpQueue_.FreeTensor(xSq);

            inQueueX_.FreeTensor(xDeque);

            remaining -= curTileLen;
        }

        // Compute mean and invStd using floating-point arithmetic (no explicit cast)
        float mean = sum / innerSizeFloat_;
        float var = sum2 / innerSizeFloat_ - mean * mean;
        float invStd = 1.0f / sqrt(var + epsilon_);

        // Second pass: apply normalization
        remaining = innerSize_;
        for (uint32_t t = 0; t < tiles; ++t) {
            uint32_t curTileLen = (remaining > tileLen_) ? tileLen_ : remaining;
            if (curTileLen == 0) break;
            uint32_t tileOffset = t * tileLen_;

            // Copy x tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + tileOffset], curTileLen);
            inQueueX_.EnQue(xLocal);

            // Copy gamma and beta tiles
            AscendC::LocalTensor<float> gammaLocal = gammaQueue_.AllocTensor<float>();
            AscendC::DataCopy(gammaLocal, gammaGm_[tileOffset], curTileLen);
            gammaQueue_.EnQue(gammaLocal);

            AscendC::LocalTensor<float> betaLocal = betaQueue_.AllocTensor<float>();
            AscendC::DataCopy(betaLocal, betaGm_[tileOffset], curTileLen);
            betaQueue_.EnQue(betaLocal);

            // Dequeues
            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> gammaDeque = gammaQueue_.DeQue<float>();
            AscendC::LocalTensor<float> betaDeque = betaQueue_.DeQue<float>();

            // Broadcast mean and invStd to tensors of length curTileLen
            AscendC::LocalTensor<float> meanBroadcast = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> invStdBroadcast = tmpQueue_.AllocTensor<float>(); // reuse? need separate
            // Actually we can reuse same buffer for mean and invStd broadcast after Sub? Simpler: use two buffers from tmpQueue.
            // But tmpQueue has only 1 buffer allocated. We'll allocate two separate queues for broadcast values.
            // Since we already have gammaQueue and betaQueue idle after dequeue, we can reuse them for broadcast (they are single-buffer queues).
            // But we need to be careful about overwriting. Better to allocate additional queues.
            // For simplicity, create two temporary queues for scalar broadcasts.
            // We'll modify Init to include more queues.
            // Workaround: Use the gammaQueue and betaQueue buffers after dequeue for broadcast, but they are already used.
            // Since gamma and beta are only used once per tile, we can reuse them after dequeue.
            // After dequeue gammaDeque and betaDeque, we can reuse their memory for broadcast.
            // Actually we can overwrite gammaDeque with mean and betaDeque with invStd, as they are no longer needed.
            // So we will use gammaDeque as meanBroadcast and betaDeque as invStdBroadcast.
            AscendC::LocalTensor<float> meanBroadcast = gammaDeque;  // reuse buffer
            AscendC::LocalTensor<float> invStdBroadcast = betaDeque; // reuse buffer
            AscendC::Broadcast(meanBroadcast, mean, curTileLen);
            AscendC::Broadcast(invStdBroadcast, invStd, curTileLen);

            // Compute y = gamma * ((x - mean) * invStd) + beta
            // tmp = x - mean
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(tmp, xDeque, meanBroadcast, curTileLen);
            // tmp = tmp * invStd
            AscendC::Mul(tmp, tmp, invStdBroadcast, curTileLen);
            // tmp = tmp * gamma
            AscendC::Mul(tmp, tmp, gammaDeque, curTileLen);
            // tmp = tmp + beta
            AscendC::Add(tmp, tmp, betaDeque, curTileLen);

            // Write result to global memory
            AscendC::DataCopy(yGm_[rowOffset + tileOffset], tmp, curTileLen);

            // Free tensors
            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xDeque);
            // gammaDeque and betaDeque are reused as broadcast, they will be freed by the queue? Actually they were dequeued from gammaQueue and betaQueue.
            // We need to free them from their queues. But we used them as broadcast buffers; after finishing, we must still free them.
            // Since they are dequeued, we must call gammaQueue_.FreeTensor(gammaDeque) etc. But we reused their pointers; the content is changed but memory belongs to the queue.
            // So we need to free them using their respective queues.
            gammaQueue_.FreeTensor(gammaDeque);
            betaQueue_.FreeTensor(betaDeque);

            remaining -= curTileLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> gammaQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> betaQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> reduceQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outerSize_ {0};
    uint32_t innerSize_ {0};
    float innerSizeFloat_ {0.0f};
    uint32_t tileLen_ {0};
    float epsilon_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength, tiling_data.outerSize, tiling_data.innerSize,
            tiling_data.tileLength, tiling_data.epsilon);
    op.Process();
}
