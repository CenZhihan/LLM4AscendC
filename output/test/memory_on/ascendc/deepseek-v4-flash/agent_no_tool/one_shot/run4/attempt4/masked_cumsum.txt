#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t HARD_CAP_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaskedCumsumCustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::Shape &originShape = xShape->GetStorageShape();
    const uint32_t totalLength = static_cast<uint32_t>(originShape.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;

    uint32_t ndim = originShape.GetDimNum();
    if (dim < 0) dim += ndim;
    uint32_t rowLen = static_cast<uint32_t>(originShape.GetDim(dim));

    uint32_t maxTile = UB_BUDGET_BYTES / 24;
    if (rowLen < maxTile) maxTile = rowLen;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = maxTile;
    if (tileLen > rowLen) tileLen = AlignDown(rowLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = rowLen;

    uint32_t numRows = (rowLen == 0) ? 0 : totalLength / rowLen;
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLen(rowLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLen,
                                uint32_t tileLen, uint32_t blockDim) {
        totalLen_ = totalLength;
        rowLen_ = rowLen;
        tileLen_ = tileLen;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        uint32_t numRows = totalLen_ / rowLen_;
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        for (uint32_t rowIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
             rowIdx < numRows; rowIdx += blockNum) {
            uint32_t rowBase = rowIdx * rowLen_;
            float carry = 0.0f;

            uint32_t tiles = (rowLen_ + tileLen_ - 1) / tileLen_;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = t * tileLen_;
                uint32_t len = rowLen_ - offset;
                if (len > tileLen_) len = tileLen_;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + offset], len);
                AscendC::DataCopy(maskLocal, maskGm_[rowBase + offset], len);
                inQueueX_.EnQue(xLocal);
                inQueueMask_.EnQue(maskLocal);

                xLocal = inQueueX_.DeQue<float>();
                maskLocal = inQueueMask_.DeQue<float>();

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                for (uint32_t i = 0; i < len; ++i) {
                    float val = xLocal(i) * maskLocal(i) + carry;
                    yLocal(i) = val;
                    carry = val;
                }

                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xLocal);
                inQueueMask_.FreeTensor(maskLocal);

                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[rowBase + offset], yLocal, len);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.totalLength, tiling_data.rowLen, tiling_data.tileLen, tiling_data.blockDim);
    op.Process();
}
