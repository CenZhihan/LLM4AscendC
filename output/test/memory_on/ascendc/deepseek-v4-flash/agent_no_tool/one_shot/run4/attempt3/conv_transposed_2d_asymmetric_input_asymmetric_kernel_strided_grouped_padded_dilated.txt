#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_TILE = 1024;

class KernelConvTransposed2d {
public:
    __aicore__ inline KernelConvTransposed2d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t hIn, uint32_t wIn,
                                uint32_t cOut, uint32_t hOut, uint32_t wOut,
                                uint32_t kH, uint32_t kW,
                                uint32_t groups,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t outputPadH, uint32_t outputPadW,
                                uint32_t blockDim,
                                uint32_t tileN, uint32_t tileH, uint32_t tileW) {
        n_ = n; cIn_ = cIn; hIn_ = hIn; wIn_ = wIn;
        cOut_ = cOut; hOut_ = hOut; wOut_ = wOut;
        kH_ = kH; kW_ = kW; groups_ = groups;
        strideH_ = strideH; strideW_ = strideW;
        padH_ = padH; padW_ = padW;
        dilationH_ = dilationH; dilationW_ = dilationW;
        outputPadH_ = outputPadH; outputPadW_ = outputPadW;
        tileN_ = tileN; tileH_ = tileH; tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n * cIn * hIn * wIn);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, cIn * (cOut/groups) * kH * kW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n * cOut * hOut * wOut);
    }

    __aicore__ inline void Process() {
        for (uint32_t i = 0; i < n_ * cOut_ * hOut_ * wOut_; i++) {
            yGm_.SetValue(i, 0.0f);
        }
        for (uint32_t b = 0; b < n_; b++) {
            for (uint32_t g = 0; g < groups_; g++) {
                uint32_t cInPerGroup = cIn_ / groups_;
                uint32_t cOutPerGroup = cOut_ / groups_;
                for (uint32_t ci = 0; ci < cInPerGroup; ci++) {
                    uint32_t weightOffset = (g * cInPerGroup + ci) * cOutPerGroup * kH_ * kW_;
                    for (uint32_t hi = 0; hi < hIn_; hi++) {
                        for (uint32_t wi = 0; wi < wIn_; wi++) {
                            float xVal = xGm_.GetValue(b * cIn_ * hIn_ * wIn_ +
                                                        (g * cInPerGroup + ci) * hIn_ * wIn_ +
                                                        hi * wIn_ + wi);
                            for (uint32_t kh = 0; kh < kH_; kh++) {
                                for (uint32_t kw = 0; kw < kW_; kw++) {
                                    int ho = (int)hi * (int)strideH_ - (int)padH_ + (int)kh * (int)dilationH_;
                                    int wo = (int)wi * (int)strideW_ - (int)padW_ + (int)kw * (int)dilationW_;
                                    if (ho >= 0 && ho < (int)hOut_ && wo >= 0 && wo < (int)wOut_) {
                                        float wVal = wGm_.GetValue(weightOffset +
                                                                   kh * kW_ + kw);
                                        for (uint32_t co = 0; co < cOutPerGroup; co++) {
                                            float yVal = yGm_.GetValue(b * cOut_ * hOut_ * wOut_ +
                                                                       (g * cOutPerGroup + co) * hOut_ * wOut_ +
                                                                       ho * wOut_ + wo);
                                            yVal += xVal * wGm_.GetValue(weightOffset + co * kH_ * kW_ + kh * kW_ + kw);
                                            yGm_.SetValue(b * cOut_ * hOut_ * wOut_ +
                                                          (g * cOutPerGroup + co) * hOut_ * wOut_ +
                                                          ho * wOut_ + wo, yVal);
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;
    uint32_t n_, cIn_, hIn_, wIn_;
    uint32_t cOut_, hOut_, wOut_;
    uint32_t kH_, kW_, groups_;
    uint32_t strideH_, strideW_, padH_, padW_, dilationH_, dilationW_, outputPadH_, outputPadW_;
    uint32_t tileN_, tileH_, tileW_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(td, tiling);
    KernelConvTransposed2d op;
    op.Init(x, weight, y,
            td.n, td.cIn, td.hIn, td.wIn,
            td.cOut, td.hOut, td.wOut,
            td.kH, td.kW,
            td.groups,
            td.strideH, td.strideW,
            td.padH, td.padW,
            td.dilationH, td.dilationW,
            td.outputPadH, td.outputPadW,
            td.blockDim,
            td.tileN, td.tileH, td.tileW);
    op.Process();
}
