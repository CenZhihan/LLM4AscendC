host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    uint32_t ndim = inputShape->GetDimNum();
    int64_t dim = -1;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs->GetAttrNum() > 0) {
        dim = attrs->GetAttrPointer<int64_t>(0)[0];
    }
    // Normalize negative dimension
    if (dim < 0) dim += ndim;
    // compute shaping info
    uint32_t totalLength = 1;
    uint32_t outerDim = 1;
    uint32_t reduceDim = 1;
    uint32_t innerDim = 1;
    for (uint32_t i = 0; i < ndim; ++i) {
        uint32_t s = static_cast<uint32_t>(inputShape->GetDim(i));
        totalLength *= s;
        if (i < static_cast<uint32_t>(dim)) outerDim *= s;
        else if (i == static_cast<uint32_t>(dim)) reduceDim = s;
        else innerDim *= s;
    }
    uint32_t outputTotalLength = totalLength / reduceDim;

    // Per-tile output elements: set small to avoid complex indexing
    uint32_t tileLen = 128; // conservative, can be tuned
    if (outputTotalLength < tileLen) tileLen = outputTotalLength;

    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dimAttr(dim);
    tiling.set_outerDim(outerDim);
    tiling.set_reduceDim(reduceDim);
    tiling.set_innerDim(innerDim);
    tiling.set_outputTotalLength(outputTotalLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
