#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative
static constexpr uint32_t ALIGN_ELEMS = 16;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;
    
    // Input shapes: A [M, K], B [K, N]
    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);
    
    if (shapeA->GetDimNum() != 2 || shapeB->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    
    uint32_t M = static_cast<uint32_t>(shapeA->GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA->GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB->GetDim(1));
    
    // Check compatibility
    if (static_cast<uint32_t>(shapeB->GetDim(0)) != K) {
        return ge::GRAPH_FAILED;
    }
    
    // Determine tile sizes
    // We will tile along M (rows) and optionally along K (inner dimension)
    // Assume B fits in UB if K*N <= some limit, otherwise tile K as well.
    // For tall-skinny, K and N are relatively small; try to fit entire B.
    uint32_t sizeB = K * N * sizeof(float);
    uint32_t sizeATile = ???
    
    // Heuristic: Use as much UB as possible for vector operations.
    // Need space for: tile of A (tileM * K floats), B (K*N floats), C tile (tileM * N floats), temporary.
    // Let tileM be derived from remaining UB after B and some overhead.
    const uint32_t overhead = 8192; // safety
    uint32_t ubForB = sizeB;
    uint32_t remaining = UB_BUDGET_BYTES - ubForB - overhead;
    
    uint32_t tileM = 1;
    if (sizeB < remaining) {
        uint32_t bytesPerRow = (K + N) * sizeof(float); // A row + C row
        tileM = remaining / bytesPerRow;
        if (tileM > M) tileM = M;
        tileM = AlignDown(tileM, ALIGN_ELEMS);
        if (tileM < 1) tileM = 1;
    } else {
        // B is too large; need to tile K as well, fallback to simple tiling
        tileM = 128; // fixed small tile
    }
    
    uint32_t tileK = K; // no inner tiling by default
    // If B is too large, we could tile K, but for simplicity assume it fits.
    
    // Block dimension: number of tiles along M
    uint32_t numTilesM = CeilDiv(M, tileM);
    uint32_t blockDim = numTilesM;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 16) blockDim = 16; // limit cores used
    
    context->SetBlockDim(blockDim);
    
    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileK,
                                uint32_t blockDim) {
        A_gm_.SetGlobalBuffer((__gm__ float*)A, M * K);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, M * N);
        M_ = M; K_ = K; N_ = N;
        tileM_ = tileM; tileK_ = tileK;
        blockDim_ = blockDim;
        
        // Buffer sizes
        uint32_t sizeATile = tileM_ * K_;
        uint32_t sizeB = K_ * N_;
        uint32_t sizeCTile = tileM_ * N_;
        
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, sizeATile * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, sizeB * sizeof(float)); // B loaded once if fits
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, sizeCTile * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * sizeof(float)); // for row dot product temp
    }

    __aicore__ inline void Process() {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;
        
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t startRow = blockIdx * tileM_;
        if (startRow >= M_) return;
        
        // Load entire B into local memory (assume it fits)
        AscendC::LocalTensor<float> B_local = inQueueB_.AllocTensor<float>();
        AscendC::DataCopy(B_local, B_gm_, K_ * N_);
        inQueueB_.EnQue(B_local); // No need to dequeue later, reuse.
        
        for (uint32_t row = startRow; row < M_; row += blockNum * tileM_) {
            uint32_t curTileM = (row + tileM_ <= M_) ? tileM_ : M_ - row;
            
            // Load tile of A
            AscendC::LocalTensor<float> A_local = inQueueA_.AllocTensor<float>();
            uint32_t offsetA = row * K_;
            AscendC::DataCopy(A_local, A_gm_[offsetA], curTileM * K_);
            inQueueA_.EnQue(A_local);
            
            // Compute C tile
            AscendC::LocalTensor<float> C_local = outQueueC_.AllocTensor<float>();
            AscendC::LocalTensor<float> B_local_deq = inQueueB_.DeQue<float>();
            
            // For each row in tile
            for (uint32_t r = 0; r < curTileM; r++) {
                // Dot product with each column of B
                for (uint32_t n = 0; n < N_; n++) {
                    AscendC::LocalTensor<float> aRow = A_local[r * K_];
                    AscendC::LocalTensor<float> bCol = B_local_deq[n * K_]; // column-major? Actually B is K x N, so column n is offset n*K
                    // Compute sum
                    AscendC::LocalTensor<float> temp = tmpQueue_.AllocTensor<float>();
                    AscendC::Mul(temp, aRow, bCol, K_);
                    float sum = 0.0f;
                    for (uint32_t i = 0; i < K_; i++) {
                        sum += temp(i);
                    }
                    C_local[r * N_ + n] = sum;
                    tmpQueue_.FreeTensor(temp);
                }
            }
            
            inQueueB_.EnQue(B_local_deq); // return B for next iteration
            outQueueC_.EnQue(C_local);
            
            // Copy out C tile
            AscendC::LocalTensor<float> C_local_out = outQueueC_.DeQue<float>();
            uint32_t offsetC = row * N_;
            AscendC::DataCopy(C_gm_[offsetC], C_local_out, curTileM * N_);
            outQueueC_.FreeTensor(C_local_out);
            
            // Free A
            AscendC::LocalTensor<float> A_local_deq = inQueueA_.DeQue<float>();
            inQueueA_.FreeTensor(A_local_deq);
        }
        
        // Free B
        AscendC::LocalTensor<float> B_local_final = inQueueB_.DeQue<float>();
        inQueueB_.FreeTensor(B_local_final);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_; // single buffer for B
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    
    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;
    
    uint32_t M_, K_, N_;
    uint32_t tileM_, tileK_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileK,
            tiling_data.blockDim);
    op.Process();
}
