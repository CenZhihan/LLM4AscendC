#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const gert::Shape &originShape = inputShape->GetOriginShape();
    uint32_t ndim = originShape.GetDimNum();
    int64_t dim = -1;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs->GetAttrNum() > 0) {
        dim = attrs->GetAttrPointer<int64_t>(0)[0];
    }
    // Normalize negative dimension
    if (dim < 0) dim += ndim;
    // compute shaping info
    uint32_t totalLength = 1;
    uint32_t outerDim = 1;
    uint32_t reduceDim = 1;
    uint32_t innerDim = 1;
    for (uint32_t i = 0; i < ndim; ++i) {
        uint32_t s = static_cast<uint32_t>(originShape.GetDim(i));
        totalLength *= s;
        if (i < static_cast<uint32_t>(dim)) outerDim *= s;
        else if (i == static_cast<uint32_t>(dim)) reduceDim = s;
        else innerDim *= s;
    }
    uint32_t outputTotalLength = totalLength / reduceDim;

    // Per-tile output elements: set small to avoid complex indexing
    uint32_t tileLen = 128; // conservative, can be tuned
    if (outputTotalLength < tileLen) tileLen = outputTotalLength;

    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dimAttr(dim);
    tiling.set_outerDim(outerDim);
    tiling.set_reduceDim(reduceDim);
    tiling.set_innerDim(innerDim);
    tiling.set_outputTotalLength(outputTotalLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::Shape &originShape = x_shape->GetOriginShape();
    uint32_t ndim = originShape.GetDimNum();
    int64_t dim = -1;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs->GetAttrNum() > 0) {
        dim = attrs->GetAttrPointer<int64_t>(0)[0];
    }
    if (dim < 0) dim += ndim;
    gert::Shape *y_shape = context->GetOutputShape(0);
    if (ndim == 1) {
        y_shape->SetDimNum(1);
        y_shape->SetDim(0, 1);
    } else {
        y_shape->SetDimNum(ndim - 1);
        uint32_t idx = 0;
        for (uint32_t i = 0; i < ndim; ++i) {
            if (i != static_cast<uint32_t>(dim)) {
                y_shape->SetDim(idx, originShape.GetDim(i));
                ++idx;
            }
        }
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimension {
public:
    __aicore__ inline KernelArgmaxOverADimension() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int64_t dimAttr,
                                uint32_t outerDim, uint32_t reduceDim, uint32_t innerDim,
                                uint32_t outputTotalLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        dimAttr_ = dimAttr;
        outerDim_ = outerDim;
        reduceDim_ = reduceDim;
        innerDim_ = innerDim;
        outputTotalLen_ = outputTotalLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, outputTotalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(int64_t));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outputTotalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= outputTotalLen_) return;

        while (true) {
            uint32_t curTile = outputTotalLen_ - base;
            if (curTile > tileLen_) curTile = tileLen_;

            // For each output element, we will compute argmax directly using global memory access
            // This is inefficient but functional; tileLen should be kept small (e.g., 128)
            for (uint32_t i = 0; i < curTile; ++i) {
                uint32_t outIdx = base + i;
                // Map output index to multi-dimensional indices
                uint32_t outerIdx = outIdx / innerDim_;
                uint32_t innerIdx = outIdx % innerDim_;
                uint32_t sliceStride = innerDim_;
                uint32_t baseOffset = outerIdx * (reduceDim_ * innerDim_) + innerIdx;

                float maxVal = xGm_.GetValue(baseOffset);
                int64_t maxIdx = 0;
                for (uint32_t r = 1; r < reduceDim_; ++r) {
                    uint32_t offset = baseOffset + r * sliceStride;
                    float val = xGm_.GetValue(offset);
                    if (val > maxVal) {
                        maxVal = val;
                        maxIdx = r;
                    }
                }
                yGm_.SetValue(base + i, maxIdx);
            }

            uint32_t nextBase = base + strideElems;
            if (nextBase >= outputTotalLen_) break;
            base = nextBase;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int64_t dimAttr_ {-1};
    uint32_t outerDim_ {0};
    uint32_t reduceDim_ {0};
    uint32_t innerDim_ {0};
    uint32_t outputTotalLen_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimension op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.dimAttr,
            tiling_data.outerDim, tiling_data.reduceDim, tiling_data.innerDim,
            tiling_data.outputTotalLength);
    op.Process();
}
