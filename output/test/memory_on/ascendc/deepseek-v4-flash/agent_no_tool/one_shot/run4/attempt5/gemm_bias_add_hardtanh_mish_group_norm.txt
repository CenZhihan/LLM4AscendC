kernel_src="""
#include "kernel_operator.h"
#include <math.h>

constexpr int32_t BUFFER_NUM = 2;
constexpr float EPSILON = 1e-5f;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
                                GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures,
                                uint32_t outFeatures, uint32_t numGroups,
                                uint32_t tileLenOut, uint32_t tileLenIn)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        tileLenOut_ = tileLenOut;
        tileLenIn_ = tileLenIn;
        perGroupSize_ = outFeatures_ / numGroups_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inFeatures);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures * inFeatures);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * outFeatures);

        uint32_t xBufSize = batchSize_ * inFeatures_;
        uint32_t wBufSize = outFeatures_ * inFeatures_;
        uint32_t yBufSize = batchSize_ * outFeatures_;
        uint32_t biasBufSize = outFeatures_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, xBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueWeight_, BUFFER_NUM, wBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueLinBias_, BUFFER_NUM, biasBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueBias_, BUFFER_NUM, biasBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueGnGamma_, BUFFER_NUM, biasBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueGnBeta_, BUFFER_NUM, biasBufSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, yBufSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, (xBufSize + yBufSize + biasBufSize) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t xSize = batchSize_ * inFeatures_;
        const uint32_t wSize = outFeatures_ * inFeatures_;
        const uint32_t ySize = batchSize_ * outFeatures_;

        // Prefetch all constant inputs
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[0], xSize);
        inQueueX_.EnQue(xLocal);

        AscendC::LocalTensor<float> wLocal = inQueueWeight_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, weightGm_[0], wSize);
        inQueueWeight_.EnQue(wLocal);

        AscendC::LocalTensor<float> linB = inQueueLinBias_.AllocTensor<float>();
        AscendC::DataCopy(linB, linBiasGm_[0], outFeatures_);
        inQueueLinBias_.EnQue(linB);

        AscendC::LocalTensor<float> b = inQueueBias_.AllocTensor<float>();
        AscendC::DataCopy(b, biasGm_[0], outFeatures_);
        inQueueBias_.EnQue(b);

        AscendC::LocalTensor<float> gG = inQueueGnGamma_.AllocTensor<float>();
        AscendC::DataCopy(gG, gnGammaGm_[0], outFeatures_);
        inQueueGnGamma_.EnQue(gG);

        AscendC::LocalTensor<float> gB = inQueueGnBeta_.AllocTensor<float>();
        AscendC::DataCopy(gB, gnBetaGm_[0], outFeatures_);
        inQueueGnBeta_.EnQue(gB);

        // Dequeue all for compute
        AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wTile = inQueueWeight_.DeQue<float>();
        AscendC::LocalTensor<float> linBias = inQueueLinBias_.DeQue<float>();
        AscendC::LocalTensor<float> bias = inQueueBias_.DeQue<float>();
        AscendC::LocalTensor<float> gamma = inQueueGnGamma_.DeQue<float>();
        AscendC::LocalTensor<float> beta = inQueueGnBeta_.DeQue<float>();

        AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

        // For each batch, compute linear output manually using scalar loops (since no matmul API)
        for (uint32_t n = 0; n < batchSize_; n++) {
            for (uint32_t o = 0; o < outFeatures_; o++) {
                float sum = linBias.GetValue(o);
                for (uint32_t i = 0; i < inFeatures_; i++) {
                    sum += xTile.GetValue(n * inFeatures_ + i) * wTile.GetValue(o * inFeatures_ + i);
                }
                sum += bias.GetValue(o);
                // Hardtanh clamp [-1,1]
                if (sum < -1.0f) sum = -1.0f;
                else if (sum > 1.0f) sum = 1.0f;
                // Mish: x * tanh(log(1+exp(x)))  (softplus approximation)
                float sp = logf(1.0f + expf(sum));
                float tanhsp = tanhf(sp);
                sum = sum * tanhsp;
                yTile.SetValue(n * outFeatures_ + o, sum);
            }
        }

        // Group normalization on the output tile
        for (uint32_t n = 0; n < batchSize_; n++) {
            for (uint32_t g = 0; g < numGroups_; g++) {
                float mean = 0.0f, var = 0.0f;
                uint32_t base = n * outFeatures_ + g * perGroupSize_;
                for (uint32_t k = 0; k < perGroupSize_; k++) {
                    mean += yTile.GetValue(base + k);
                }
                mean /= perGroupSize_;
                for (uint32_t k = 0; k < perGroupSize_; k++) {
                    float diff = yTile.GetValue(base + k) - mean;
                    var += diff * diff;
                }
                var /= perGroupSize_;
                float invStd = 1.0f / sqrtf(var + EPSILON);
                for (uint32_t k = 0; k < perGroupSize_; k++) {
                    uint32_t idx = base + k;
                    float val = (yTile.GetValue(idx) - mean) * invStd * gamma.GetValue(g * perGroupSize_ + k) + beta.GetValue(g * perGroupSize_ + k);
                    yTile.SetValue(idx, val);
                }
            }
        }

        outQueueY_.EnQue<float>(yTile);
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[0], yOut, ySize);
        outQueueY_.FreeTensor(yOut);

        // Free resources
        inQueueX_.FreeTensor(xTile);
        inQueueWeight_.FreeTensor(wTile);
        inQueueLinBias_.FreeTensor(linBias);
        inQueueBias_.FreeTensor(bias);
        inQueueGnGamma_.FreeTensor(gamma);
        inQueueGnBeta_.FreeTensor(beta);
        tmpQueue_.FreeTensor(tmp);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWeight_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueLinBias_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBias_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueGnGamma_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueGnBeta_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_;
    uint32_t inFeatures_;
    uint32_t outFeatures_;
    uint32_t numGroups_;
    uint32_t tileLenOut_;
    uint32_t tileLenIn_;
    uint32_t perGroupSize_;
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
    GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.batchSize, tiling_data.inFeatures,
            tiling_data.outFeatures, tiling_data.numGroups,
            tiling_data.tileLenOut, tiling_data.tileLenIn);
    op.Process();
}
