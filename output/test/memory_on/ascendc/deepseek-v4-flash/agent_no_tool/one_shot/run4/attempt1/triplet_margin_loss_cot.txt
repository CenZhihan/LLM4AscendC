class KernelTripletMarginLoss {
public:
    __aicore__ inline Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR output, 
                           uint32_t totalSamples, uint32_t totalDim, uint32_t tileDim, float margin) {
        totalSamples_ = totalSamples;
        totalDim_ = totalDim;
        tileDim_ = tileDim;
        margin_ = margin;
        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalSamples * totalDim);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalSamples * totalDim);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalSamples * totalDim);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalSamples);
        // init queues
        pipe_.InitBuffer(inQX_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(inQP_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(inQN_, BUFFER_NUM, tileDim_ * sizeof(float));
        // no temporary queue needed, just local accumulators
    }

    __aicore__ inline Process() {
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        for (uint32_t sampleIdx = blockIdx; sampleIdx < totalSamples_; sampleIdx += blockNum) {
            float sumP = 0.0f;
            float sumN = 0.0f;
            uint32_t dimBase = sampleIdx * totalDim_;
            uint32_t remaining = totalDim_;
            uint32_t dimOffset = 0;
            while (remaining > 0) {
                uint32_t curDim = (remaining > tileDim_) ? tileDim_ : remaining;
                // copy in anchor, positive, negative for this dim chunk
                AscendC::LocalTensor<float> xLocal = inQX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, anchorGm_[dimBase + dimOffset], curDim);
                inQX_.EnQue(xLocal);
                AscendC::LocalTensor<float> pLocal = inQP_.AllocTensor<float>();
                AscendC::DataCopy(pLocal, positiveGm_[dimBase + dimOffset], curDim);
                inQP_.EnQue(pLocal);
                AscendC::LocalTensor<float> nLocal = inQN_.AllocTensor<float>();
                AscendC::DataCopy(nLocal, negativeGm_[dimBase + dimOffset], curDim);
                inQN_.EnQue(nLocal);

                // dequeue
                xLocal = inQX_.DeQue<float>();
                pLocal = inQP_.DeQue<float>();
                nLocal = inQN_.DeQue<float>();

                // compute diff and square for pos and neg
                // Use local tensors for diff?
                // We can compute in place: (x - p)^2
                // We'll create temporary local tensors? We can reuse one of the input tensors.
                // To avoid extra allocation, we can use a temporary queue? Let's use a single tmp queue for calculations.

                AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                // diff for pos: tmp = x - p
                AscendC::Sub(tmp, xLocal, pLocal, curDim);
                // square: tmp = tmp * tmp
                AscendC::Mul(tmp, tmp, tmp, curDim);

private:
