#include "argmax_over_a_dimension_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1;        // no special alignment needed
static constexpr uint32_t HARD_CAP_ELEMS = 4096;  // per-iteration output elements

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    uint32_t ndim = inputShape->GetOriginShape().GetDimNum();
    int64_t dim = -1;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs->GetAttrNum() > 0) {
        dim = attrs->GetAttrPointer<int64_t>(0)[0];
    }

    // compute shaping info
    uint32_t totalLength = 1;
    uint32_t outerDim = 1;
    uint32_t reduceDim = 1;
    uint32_t innerDim = 1;
    // we assume dim is valid (0 <= dim < ndim) and ndim >=1
    for (uint32_t i = 0; i < ndim; ++i) {
        uint32_t s = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(i));
        totalLength *= s;
        if (i < dim) outerDim *= s;
        else if (i == dim) reduceDim = s;
        else innerDim *= s;
    }
    uint32_t outputTotalLength = totalLength / reduceDim;  // assuming contiguous

    // per-tile output elements: we want to process multiple output elements per block
    // We will store per output element the max index, intermediate registers: we need to hold
    //   for each output element: current max value and index -> 2 floats? Actually index is uint32.
    //   But we can use local memory for scans.
    // Simplify: each block processes a single row of output (outerDim innerDim). Not efficient but works.
    // Use UB budget to determine tileLength (number of output elements processed per iteration)
    // We'll assume each output element needs 2 registers (value+index) => 8 bytes per element under reduction.
    // Also we need to read input slice. We'll just set tileLength to a small number to avoid overflow.
    uint32_t tileLen = 128; // conservative
    if (outputTotalLength < tileLen) tileLen = outputTotalLength;

    uint32_t blockDim = 8; // fixed for simplicity
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dimAttr(dim);
    tiling.set_outerDim(outerDim);
    tiling.set_reduceDim(reduceDim);
    tiling.set_innerDim(innerDim);
    tiling.set_outputTotalLength(outputTotalLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    int64_t dim = -1;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs->GetAttrNum() > 0) {
        dim = attrs->GetAttrPointer<int64_t>(0)[0];
    }
    uint32_t ndim = x_shape->GetOriginShape().GetDimNum();
    gert::Shape *y_shape = context->GetOutputShape(0);
    // output shape: remove the dim dimension
    if (ndim == 1) {
        // output is scalar, set shape to 1-dim with size 1
        y_shape->SetDimNum(1);
        y_shape->SetDim(0, 1);

        y_shape->SetDimNum(ndim - 1);
        uint32_t idx = 0;

            if (i != dim) {
                y_shape->SetDim(idx, x_shape->GetOriginShape().GetDim(i));
                ++idx;
            }
        }
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimension {
public:
    __aicore__ inline KernelArgmaxOverADimension() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int64_t dimAttr,
                                uint32_t outerDim, uint32_t reduceDim, uint32_t innerDim,
                                uint32_t outputTotalLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength; // per-iteration number of output elements this block processes
        dimAttr_ = dimAttr;
        outerDim_ = outerDim;
        reduceDim_ = reduceDim;
        innerDim_ = innerDim;
        outputTotalLen_ = outputTotalLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, outputTotalLen_);

        // A pipe for reading input tiles: we need to read a chunk of input that contains
        // all the reduction dimension elements for a set of output elements.
        // The input tile size = tileLen * reduceDim  (if each output corresponds to reduceDim elements)
        // But we process output elements evenly; each output element corresponds to 1 slice.
        // We will read the whole slice per output element sequentially. Since we process tileLen output elements,
        // the input tile size = tileLen * reduceDim.
        // We'll allocate a buffer for the input slice, and another for the indices.
        uint32_t inputBufSize = tileLen * reduceDim_ * sizeof(float);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputBufSize);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen * sizeof(int64_t));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen * sizeof(float)); // for max values
    }

    __aicore__ inline void Process()
    {
        if (outputTotalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output elements
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= outputTotalLen_) return;

        while (true) {
            uint32_t curTile = totalLen_ - base;  // remaining output elements
            if (curTile > tileLen_) curTile = tileLen_;

            // Prefetch next tile of output indices
            uint32_t nextBase = base + strideElems;
            uint32_t nextTile = 0;
            if (nextBase < outputTotalLen_) {
                nextTile = outputTotalLen_ - nextBase;
                if (nextTile > tileLen_) nextTile = tileLen_;
                // Prefetch input data for next tile
                for (uint32_t i = 0; i < nextTile; ++i) {
                    // We will prefetch in the loop? Not efficient. Instead, prefetch chunk of input.
                    // Actually we need to read input in a contiguous block for multiple output elements?
                    // For ndim>1, the input for each output is not contiguous across output elements.
                    // Simpler: each block processes 1 output element at a time (tileLen=1) to avoid complex indexing.
                    // But the tiling set earlier used tileLen=128 and single block processes multiple output.
                    // However, the input data for multiple output elements is not contiguous. For simplicity,
                    // we set tileLen=1 in tiling? But we already set a larger tileLen above.
                    // To keep code correct, we set tileLen=1 for this kernel. But tiling function already set tileLen=128,
                    // we override here? The tiling passed is fixed. We need to adapt tiling to set tileLen=1.
                    // Let's adjust tiling function instead: set tileLen=1 always.
                }
            }

            Compute(base, curTile);

            if (nextTile == 0) break;
            base = nextBase;
        }
    }

private:
    __aicore__ inline void CopyInSlice(uint32_t outputIdx, uint32_t len)
    {
        // Output index outputIdx corresponds to a specific slice of input.
        // For each output element, we need to read reduceDim contiguous floats from global memory,
        // not necessarily contiguous across output indices if ndim>2.
        // Since we cannot guarantee contiguity, we will read element by element in Compute using scalar loads.
        // This is inefficient but works for correctness. We'll skip using queue for slices.
        // Instead, we will directly access global memory inside Compute.
    }

    __aicore__ inline void Compute(uint32_t base, uint32_t len)
    {
        // For each output element idx in [base, base+len), compute argmax along dim.
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t outIdx = base + i;
            // Convert output index to multi-dimensional indices of input (without reduction dim)
            uint32_t outerIdx = outIdx / innerDim_;  // assuming innerDim != 0
            uint32_t innerIdx = outIdx % innerDim_;

            // The start offset in input for each slice along reduction dim:
            // input[outerIdx, 0, innerIdx] -> offset = outerIdx * reduceDim * innerDim + reduceDim * innerIdx? Actually:
            // Input shape: (outerDim, reduceDim, innerDim)
            // offset for a given outerIdx and innerIdx: outerIdx * (reduceDim * innerDim) + reduceDim * innerIdx? Wait:
            // For dim=1: For output at (outerIdx, innerIdx), the corresponding input slice is:
            //   input[outerIdx, :, innerIdx] which is contiguous? If the input is row-major:
            //   input[outerIdx, reduceIdx, innerIdx] has offset = outerIdx * (reduceDim * innerDim) + reduceIdx * innerDim + innerIdx.
            // So the stride for the reduce dimension is innerDim.
            uint32_t sliceStride = innerDim_; // number of elements between successive reduceDim elements for the same outer/inner
            uint32_t baseOffset = outerIdx * (reduceDim_ * innerDim_) + innerIdx; // offset of first element of slice

            // Now find max value and its index
            float maxVal = xGm_.GetValue(baseOffset);
            int64_t maxIdx = 0;
            for (uint32_t r = 1; r < reduceDim_; ++r) {
                uint32_t offset = baseOffset + r * sliceStride;
                float val = xGm_.GetValue(offset);
                if (val > maxVal) {
                    maxVal = val;
                    maxIdx = r;
                }
            }
            yGm_.SetValue(base + i, maxIdx);
        }
    }

    __aicore__ inline void CopyOut(uint32_t base, uint32_t len)
    {
        // We already wrote directly in Compute
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int64_t dimAttr_ {-1};
    uint32_t outerDim_ {0};
    uint32_t reduceDim_ {0};
    uint32_t innerDim_ {0};
    uint32_t outputTotalLen_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimension op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.dimAttr,
            tiling_data.outerDim, tiling_data.reduceDim, tiling_data.innerDim,
            tiling_data.outputTotalLength);
    op.Process();
}
