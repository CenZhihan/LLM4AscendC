kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNorm {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNorm() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
                                GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t numGroups, uint32_t tileLenOut, uint32_t tileLenIn)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        tileLenOut_ = tileLenOut;
        tileLenIn_ = tileLenIn;
        groupSize_ = outFeatures_ / numGroups_;
        eps_ = 1e-5f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * inFeatures_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures_ * inFeatures_);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures_);
        betaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * outFeatures_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLenIn_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLenOut_ * tileLenIn_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLenOut_ * sizeof(float));
        pipe_.InitBuffer(tmp1Queue_, 1, tileLenOut_ * sizeof(float));
        pipe_.InitBuffer(tmp2Queue_, 1, tileLenOut_ * sizeof(float));
        pipe_.InitBuffer(tanhQueue_, 1, tileLenOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || inFeatures_ == 0 || outFeatures_ == 0) return;
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= batchSize_) return;
        uint32_t sampleIdx = blockIdx;
        ProcessSample(sampleIdx);
    }

private:
    __aicore__ inline void ProcessSample(uint32_t sampleIdx)
    {
        pipe_.InitBuffer(fullOutQueue_, 1, outFeatures_ * sizeof(float));
        AscendC::LocalTensor<float> fullOut = fullOutQueue_.AllocTensor<float>();

        for (uint32_t i = 0; i < outFeatures_; i++) {
            fullOut.SetValue(i, 0.0f);
        }

        uint32_t inOffset = sampleIdx * inFeatures_;
        uint32_t outOffset = sampleIdx * outFeatures_;

        uint32_t numOutTiles = (outFeatures_ + tileLenOut_ - 1) / tileLenOut_;
        for (uint32_t ot = 0; ot < numOutTiles; ot++) {
            uint32_t outBase = ot * tileLenOut_;
            uint32_t outRem = outFeatures_ - outBase;
            uint32_t curOut = (outRem < tileLenOut_) ? outRem : tileLenOut_;

            AscendC::LocalTensor<float> outAcc = outQueueY_.AllocTensor<float>();
            for (uint32_t i = 0; i < curOut; i++) {
                outAcc.SetValue(i, 0.0f);
            }

            uint32_t numInTiles = (inFeatures_ + tileLenIn_ - 1) / tileLenIn_;
            for (uint32_t it = 0; it < numInTiles; it++) {
                uint32_t inBase = it * tileLenIn_;
                uint32_t inRem = inFeatures_ - inBase;
                uint32_t curIn = (inRem < tileLenIn_) ? inRem : tileLenIn_;

                AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xTile, xGm_[inOffset + inBase], curIn);

                AscendC::LocalTensor<float> wTile = inQueueW_.AllocTensor<float>();
                for (uint32_t row = 0; row < curOut; row++) {
                    uint32_t wRowOffset = (outBase + row) * inFeatures_ + inBase;
                    AscendC::DataCopy(wTile[row * curIn], wGm_[wRowOffset], curIn);
                }

                for (uint32_t row = 0; row < curOut; row++) {
                    float sum = 0.0f;
                    for (uint32_t col = 0; col < curIn; col++) {
                        sum += xTile.GetValue(col) * wTile.GetValue(row * curIn + col);
                    }
                    outAcc.SetValue(row, outAcc.GetValue(row) + sum);
                }

                inQueueX_.FreeTensor(xTile);
                inQueueW_.FreeTensor(wTile);
            }

            for (uint32_t i = 0; i < curOut; i++) {
                float val = outAcc.GetValue(i);
                val += linBiasGm_.GetValue(outBase + i);
                val += biasGm_.GetValue(outBase + i);
                outAcc.SetValue(i, val);
            }

            for (uint32_t i = 0; i < curOut; i++) {
                fullOut.SetValue(outBase + i, outAcc.GetValue(i));
            }
            outQueueY_.FreeTensor(outAcc);
        }

        // HardTanh
        {
            for (uint32_t i = 0; i < outFeatures_; i++) {
                float val = fullOut.GetValue(i);
                if (val < -1.0f) val = -1.0f;
                else if (val > 1.0f) val = 1.0f;
                fullOut.SetValue(i, val);
            }
        }

        // Mish
        {
            for (uint32_t i = 0; i < outFeatures_; i++) {
                float x_val = fullOut.GetValue(i);
                float sp = std::log(1.0f + std::exp(x_val));
                float th = std::tanh(sp);
                fullOut.SetValue(i, x_val * th);
            }
        }

        // Group Normalization
        {
            for (uint32_t g = 0; g < numGroups_; g++) {
                uint32_t gStart = g * groupSize_;
                float sum = 0.0f;
                for (uint32_t i = 0; i < groupSize_; i++) {
                    sum += fullOut.GetValue(gStart + i);
                }
                float mean = sum / (float)groupSize_;
                float varSum = 0.0f;
                for (uint32_t i = 0; i < groupSize_; i++) {
                    float d = fullOut.GetValue(gStart + i) - mean;
                    varSum += d * d;
                }
                float var = varSum / (float)groupSize_;
                float invStd = 1.0f / std::sqrt(var + eps_);
                for (uint32_t i = 0; i < groupSize_; i++) {
                    float x_hat = (fullOut.GetValue(gStart + i) - mean) * invStd;
                    float gn_out = x_hat * gammaGm_.GetValue(gStart + i) + betaGm_.GetValue(gStart + i);
                    fullOut.SetValue(gStart + i, gn_out);
                }
            }
        }

        // Copy to global
        for (uint32_t i = 0; i < outFeatures_; i++) {
            yGm_.SetValue(outOffset + i, fullOut.GetValue(i));
        }
        fullOutQueue_.FreeTensor(fullOut);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp1Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp2Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tanhQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> fullOutQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t inFeatures_ {0};
    uint32_t outFeatures_ {0};
    uint32_t numGroups_ {0};
    uint32_t tileLenOut_ {0};
    uint32_t tileLenIn_ {0};
    uint32_t groupSize_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
    GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNorm op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.batchSize, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.numGroups, tiling_data.tileLenOut, tiling_data.tileLenIn);
    op.Process();
}
