#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t HARD_CAP_GROUPS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MeanReductionOverADimensionCustomTilingData tiling;
    const gert::Shape shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t ndim = static_cast<uint32_t>(shape.GetDimNum());
    if (ndim == 0) return ge::GRAPH_PARAM_INVALID;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;
    if (dim < 0 || static_cast<uint32_t>(dim) >= ndim) return ge::GRAPH_PARAM_INVALID;

    uint32_t preProduct = 1;
    for (uint32_t i = 0; i < static_cast<uint32_t>(dim); ++i) {
        preProduct *= static_cast<uint32_t>(shape.GetDim(i));
    }
    uint32_t reductionDimSize = static_cast<uint32_t>(shape.GetDim(dim));
    uint32_t postProduct = 1;
    for (uint32_t i = static_cast<uint32_t>(dim) + 1; i < ndim; ++i) {
        postProduct *= static_cast<uint32_t>(shape.GetDim(i));
    }

    uint32_t totalInputLength = preProduct * reductionDimSize * postProduct;
    uint32_t totalOutputLength = preProduct * postProduct;

    uint32_t groupsPerTile = 1024;
    if (groupsPerTile > totalOutputLength) groupsPerTile = totalOutputLength;
    groupsPerTile = AlignDown(groupsPerTile, ALIGN_ELEMS);
    if (groupsPerTile < ALIGN_ELEMS) groupsPerTile = ALIGN_ELEMS;
    if (groupsPerTile > HARD_CAP_GROUPS) groupsPerTile = AlignDown(HARD_CAP_GROUPS, ALIGN_ELEMS);

    uint32_t numTiles = CeilDiv(totalOutputLength, groupsPerTile);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalInputLength(totalInputLength);
    tiling.set_totalOutputLength(totalOutputLength);
    tiling.set_reductionDimSize(reductionDimSize);
    tiling.set_reduceStride(postProduct);
    tiling.set_tileLength(groupsPerTile);
    tiling.set_blockDim(blockDim);
    tiling.set_invReductionDimSize(1.0f / static_cast<float>(reductionDimSize));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputLength,
                                uint32_t totalOutputLength,
                                uint32_t reductionDimSize,
                                uint32_t reduceStride,
                                uint32_t tileLength,
                                float invReductionDimSize)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputLength);
        totalInputLen_ = totalInputLength;
        totalOutputLen_ = totalOutputLength;
        reductionDimSize_ = reductionDimSize;
        reduceStride_ = reduceStride;
        tileLen_ = tileLength;
        invReductionDimSize_ = invReductionDimSize;

        useVectorized_ = (reduceStride_ == 1);
        if (useVectorized_) {
            pipe_.InitBuffer(inQueueX_, BUFFER_NUM, reductionDimSize_ * sizeof(float));
            pipe_.InitBuffer(tmpQueue_, 1, 1 * sizeof(float));  // scalar sum
        }
    }

    __aicore__ inline void Process() {
        if (totalOutputLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideGroups = blockNum * tileLen_;
        uint32_t groupStart = blockIdx * tileLen_;
        if (groupStart >= totalOutputLen_) return;

        while (true) {
            uint32_t groupsToProcess = totalOutputLen_ - groupStart;
            if (groupsToProcess > tileLen_) groupsToProcess = tileLen_;
            ProcessGroups(groupStart, groupsToProcess);
            groupStart += strideGroups;
            if (groupStart >= totalOutputLen_) break;
        }
    }

private:
    __aicore__ inline void ProcessGroups(uint32_t groupBase, uint32_t numGroups) {
        if (useVectorized_) {
            ProcessGroupsVectorized(groupBase, numGroups);
        } else {
            ProcessGroupsScalar(groupBase, numGroups);
        }
    }

    // Vectorized case when reduceStride == 1
    __aicore__ inline void ProcessGroupsVectorized(uint32_t groupBase, uint32_t numGroups) {
        const uint32_t rowLen = reductionDimSize_;
        const float invLen = invReductionDimSize_;

        for (uint32_t i = 0; i < numGroups; ++i) {
            uint32_t outputIdx = groupBase + i;
            uint32_t inputBase = outputIdx * rowLen;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[inputBase], rowLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();

            // ReduceSum with 4 arguments: dst, src, len, isReduceLastDim = true
            AscendC::ReduceSum(sumLocal, xDeque, rowLen, true);
            float sum = sumLocal.GetValue(0);
            float meanVal = sum * invLen;
            yGm_.SetValue(outputIdx, meanVal);

            inQueueX_.FreeTensor(xDeque);
            tmpQueue_.FreeTensor(sumLocal);
        }
    }

    // Scalar case for arbitrary stride
    __aicore__ inline void ProcessGroupsScalar(uint32_t groupBase, uint32_t numGroups) {
        const uint32_t R = reductionDimSize_;
        const uint32_t S = reduceStride_;
        const float invR = invReductionDimSize_;

        for (uint32_t i = 0; i < numGroups; ++i) {
            uint32_t outputIdx = groupBase + i;
            uint32_t batchBefore = outputIdx / S;
            uint32_t batchAfter = outputIdx % S;
            uint32_t base = batchBefore * (R * S) + batchAfter;

            float sum = 0.0f;
            for (uint32_t r = 0; r < R; ++r) {
                sum += xGm_.GetValue(base + r * S);
            }
            float meanVal = sum * invR;
            yGm_.SetValue(outputIdx, meanVal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputLen_ {0};
    uint32_t totalOutputLen_ {0};
    uint32_t reductionDimSize_ {0};
    uint32_t reduceStride_ {0};
    uint32_t tileLen_ {0};
    float invReductionDimSize_ {0.0f};
    bool useVectorized_ {false};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalInputLength,
            tiling_data.totalOutputLength,
            tiling_data.reductionDimSize,
            tiling_data.reduceStride,
            tiling_data.tileLength,
            tiling_data.invReductionDimSize);
    op.Process();
}
