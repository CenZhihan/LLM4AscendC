static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape &originShape = xShape->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(originShape.GetDim(1));
    uint32_t D = static_cast<uint32_t>(originShape.GetDim(2));
    uint32_t H = static_cast<uint32_t>(originShape.GetDim(3));
    uint32_t W = static_cast<uint32_t>(originShape.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];

    uint32_t kD = static_cast<uint32_t>(kernelSize);
    uint32_t kH = kD;
    uint32_t kW = kD;
    uint32_t sD = static_cast<uint32_t>(stride);
    uint32_t sH = sD;
    uint32_t sW = sD;
    uint32_t pD = static_cast<uint32_t>(padding);
    uint32_t pH = pD;
    uint32_t pW = pD;

    uint32_t outD = static_cast<uint32_t>(std::floor((D + 2 * pD - kD) / sD + 1));
    uint32_t outH = static_cast<uint32_t>(std::floor((H + 2 * pH - kH) / sH + 1));
    uint32_t outW = static_cast<uint32_t>(std::floor((W + 2 * pW - kW) / sW + 1));

    uint32_t totalInput = N * C * D * H * W;
    uint32_t totalOutput = N * C * outD * outH * outW;

    // Tiling: each block processes one or more (N, C) planes.
    // Use blockDim = N*C, but cap to reasonable number.
    uint32_t blockDim = N * C;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalInput);
    tiling.set_outputLength(totalOutput);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelD(kD);
    tiling.set_kernelH(kH);
    tiling.set_kernelW(kW);
    tiling.set_strideD(sD);
    tiling.set_strideH(sH);
    tiling.set_strideW(sW);
    tiling.set_padD(pD);
    tiling.set_padH(pH);
    tiling.set_padW(pW);
    tiling.set_inputN(N);
    tiling.set_inputC(C);
    tiling.set_inputD(D);
    tiling.set_inputH(H);
    tiling.set_inputW(W);
    tiling.set_outputD(outD);
    tiling.set_outputH(outH);
    tiling.set_outputW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape &originShape = x_shape->GetOriginShape();

    int64_t N = originShape.GetDim(0);
    int64_t C = originShape.GetDim(1);
    int64_t D = originShape.GetDim(2);
    int64_t H = originShape.GetDim(3);
    int64_t W = originShape.GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];

    int64_t outD = (D + 2 * padding - kernelSize) / stride + 1;
    int64_t outH = (H + 2 * padding - kernelSize) / stride + 1;
    int64_t outW = (W + 2 * padding - kernelSize) / stride + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);

    y_shape->SetDim(1, C);
    y_shape->SetDim(2, outD);
    y_shape->SetDim(3, outH);
    y_shape->SetDim(4, outW);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outputLength,
                                uint32_t kernelD, uint32_t kernelH, uint32_t kernelW,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t padD, uint32_t padH, uint32_t padW,
                                uint32_t inputN, uint32_t inputC,
                                uint32_t inputD, uint32_t inputH, uint32_t inputW,
                                uint32_t outputD, uint32_t outputH, uint32_t outputW)
    {
        totalLen_ = totalLength;
        outputLen_ = outputLength;
        kD_ = kernelD; kH_ = kernelH; kW_ = kernelW;
        sD_ = strideD; sH_ = strideH; sW_ = strideW;
        pD_ = padD; pH_ = padH; pW_ = padW;
        iN_ = inputN; iC_ = inputC;
        iD_ = inputD; iH_ = inputH; iW_ = inputW;
        oD_ = outputD; oH_ = outputH; oW_ = outputW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputLen_);

        // Each tile processes a single (N, C, z) slice? For simplicity, process each output element individually.
        // Use a buffer large enough for one kernel window: kW*kH*kD floats.
        uint32_t windowSize = kD_ * kH_ * kW_;
        // Align up for efficient DMA
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, windowSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));  // single output value
        pipe_.InitBuffer(tmpQueue_, 1, windowSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outputLen_ == 0) return;

        // Grid-stride over output elements: each block handles a contiguous chunk of indices
        // outputLen_ is total number of output elements (N*C*oD*oH*oW)
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Simple linear partition: each block processes a range of output indices
        uint32_t perBlock = (outputLen_ + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * perBlock;
        uint32_t end = (blockIdx + 1) * perBlock;
        if (start >= outputLen_) return;
        if (end > outputLen_) end = outputLen_;

        // Temporary workspace for window values
        AscendC::LocalTensor<float> windowBuf = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxVal = outQueueY_.AllocTensor<float>();

        for (uint32_t idx = start; idx < end; ++idx) {
            // Convert linear index to (n, c, od, oh, ow)
            uint32_t ow = idx % oW_;
            uint32_t oh = (idx / oW_) % oH_;
            uint32_t od = (idx / (oW_ * oH_)) % oD_;
            uint32_t c = (idx / (oW_ * oH_ * oD_)) % iC_;
            uint32_t n = idx / (oW_ * oH_ * oD_ * iC_);

            // Input start positions
            int32_t idStart = static_cast<int32_t>(od) * static_cast<int32_t>(sD_) - static_cast<int32_t>(pD_);
            int32_t ihStart = static_cast<int32_t>(oh) * static_cast<int32_t>(sH_) - static_cast<int32_t>(pH_);
            int32_t iwStart = static_cast<int32_t>(ow) * static_cast<int32_t>(sW_) - static_cast<int32_t>(pW_);

            // Load window values into windowBuf and find max
            float maxValF = -1e30f; // very large negative
            bool anyValid = false;

            for (uint32_t kz = 0; kz < kD_; ++kz) {
                int32_t iz = idStart + static_cast<int32_t>(kz);
                if (iz < 0 || iz >= static_cast<int32_t>(iD_)) continue;
                for (uint32_t ky = 0; ky < kH_; ++ky) {
                    int32_t iy = ihStart + static_cast<int32_t>(ky);
                    if (iy < 0 || iy >= static_cast<int32_t>(iH_)) continue;
                    for (uint32_t kx = 0; kx < kW_; ++kx) {
                        int32_t ix = iwStart + static_cast<int32_t>(kx);
                        if (ix < 0 || ix >= static_cast<int32_t>(iW_)) continue;
                        // Valid input position: compute linear index and load
                        uint32_t inputIdx = ((n*iC_ + c) * iD_ + static_cast<uint32_t>(iz)) * iH_ * iW_
                                            + static_cast<uint32_t>(iy) * iW_ + static_cast<uint32_t>(ix);
                        float val = xGm_.GetValue(inputIdx);
                        anyValid = true;
                        if (val > maxValF) maxValF = val;
                    }
                }
            }

            // If no valid element (should not happen if padding is small), set to 0
            if (!anyValid) maxValF = 0.0f;

            // Store output
            yGm_.SetValue(idx, maxValF);
        }

        tmpQueue_.FreeTensor(windowBuf);
        outQueueY_.FreeTensor(maxVal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outputLen_ {0};
    uint32_t kD_ {0}, kH_ {0}, kW_ {0};
    uint32_t sD_ {0}, sH_ {0}, sW_ {0};
    uint32_t pD_ {0}, pH_ {0}, pW_ {0};
    uint32_t iN_ {0}, iC_ {0}, iD_ {0}, iH_ {0}, iW_ {0};
    uint32_t oD_ {0}, oH_ {0}, oW_ {0};
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.outputLength,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW,
            tiling_data.inputN, tiling_data.inputC,
            tiling_data.inputD, tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputD, tiling_data.outputH, tiling_data.outputW);
    op.Process();
}
