host_operator_src="""
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: three passes each require input buffer, plus one output & one extra temp
// Max simultaneous tensor memory:
//   Phase1: 1 input tile (4B*T)
//   Phase2: 1 input + 1 exp temp = 8B*T
//   Phase3: 1 input + 1 output = 8B*T
// Worst case: Phase2/3 occupy 8B*T. Also scalar reduction tensors (~16B) negligible.
// So total float count = 2 * tileLen (input+output/exp) => 8*4 = 32B per element? Wait:
//   Phase2: input + exp (2 tiles) = 2*4*T = 8*T bytes.
//   Phase3: input + output (2 tiles) = 8*T bytes.
// We use double buffering so each queue uses 2 buffers: total in use = 2*2 = 4 tiles? Actually:
//   inQueueX: 2 buffers, outQueueY: 2 buffers, tmpQueue: 1 buffer for exp (not double buffered).
//   Max simultaneous: 2 (input) + 2 (output) + 1 (tmp) = 5 tiles => 20*T bytes.
// Set UB budget accordingly.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative share per core
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t MAX_TILE_ELEMS = 32768;        // reasonable cap

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape *shape = context->GetInputShape(0);
    const int64_t *dims = shape->GetOriginShape().GetDims();
    int32_t ndim = shape->GetOriginShape().GetDimNum();
    // Assume dim is internal dimension; we only support 2D for simplicity
    uint32_t numRows = static_cast<uint32_t>(dims[0]);
    uint32_t rowLen = static_cast<uint32_t>(dims[1]);   // last dim
    uint32_t totalLength = numRows * rowLen;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    int32_t dim = static_cast<int32_t>(*dimAttr);
    if (dim < 0) dim += ndim;
    // For now, we only support dim == ndim-1 (last axis)
    // Otherwise, fallback to native implementation.

    // Compute tile length from UB budget (20*T = 20*tileLen <= UB_BUDGET)
    uint32_t maxTile = UB_BUDGET_BYTES / 20;   // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (rowLen < maxTile) ? rowLen : maxTile;
    // Align down to ensure full tiles
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = rowLen; // extremely small row

    // Block dim: assign rows to blocks, but cap at 24
    uint32_t blockDim = numRows;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 8) blockDim = 8;
    // Ensure blockDim <= numRows
    if (blockDim > numRows) blockDim = numRows;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLen);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t tileLength, float /*unused dim*/) {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        numRows_ = totalLen_ / rowLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block handles rows in a grid-stride fashion
        for (uint32_t r = blockIdx; r < numRows_; r += blockNum) {
            uint32_t rowOffset = r * rowLen_;

            // ---------- Phase 1: find max of row ----------
            float rowMax = -FLT_MAX;
            uint32_t tileStart = 0;
            // Prefetch first tile
            uint32_t curTileLen = (rowLen_ < tileLen_) ? rowLen_ : tileLen_;
            uint32_t gmOffset = rowOffset;
            CopyInPhase1(gmOffset, curTileLen);
            while (true) {
                // next tile prefetch (overlap)
                uint32_t nextOffset = gmOffset + tileLen_;
                uint32_t nextLen = 0;
                if (nextOffset < rowOffset + rowLen_) {
                    nextLen = (rowOffset + rowLen_ - nextOffset);
                    if (nextLen > tileLen_) nextLen = tileLen_;
                    CopyInPhase1(nextOffset, nextLen);
                }

                // compute local max for current tile
                ComputePhase1(curTileLen, rowMax);

                // advance
                gmOffset = nextOffset;
                curTileLen = nextLen;
                if (curTileLen == 0) break;
            }

            // ---------- Phase 2: compute sum exp(x - max) ----------
            float sumExp = 0.0f;
            gmOffset = rowOffset;
            curTileLen = (rowLen_ < tileLen_) ? rowLen_ : tileLen_;
            CopyInPhase2(gmOffset, curTileLen);
            while (true) {
                uint32_t nextOffset = gmOffset + tileLen_;
                uint32_t nextLen = 0;
                if (nextOffset < rowOffset + rowLen_) {
                    nextLen = (rowOffset + rowLen_ - nextOffset);
                    if (nextLen > tileLen_) nextLen = tileLen_;
                    CopyInPhase2(nextOffset, nextLen);
                }

                ComputePhase2(curTileLen, rowMax, sumExp);

                gmOffset = nextOffset;
                curTileLen = nextLen;
                if (curTileLen == 0) break;
            }

            // compute log(sumExp)
            AscendC::LocalTensor<float> logSumTmp = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> sumTmp = tmpQueue_.AllocTensor<float>(); // reuse later
            sumTmp.SetValue(0, sumExp);  // store scalar in local tensor of size 1
            AscendC::Log(logSumTmp, sumTmp, 1);
            float logSumExp = rowMax + logSumTmp.GetValue(0);
            tmpQueue_.FreeTensor(logSumTmp);
            tmpQueue_.FreeTensor(sumTmp);

            // ---------- Phase 3: compute output y = x - logSumExp ----------
            gmOffset = rowOffset;
            curTileLen = (rowLen_ < tileLen_) ? rowLen_ : tileLen_;
            CopyInPhase3(gmOffset, curTileLen);
            while (true) {
                uint32_t nextOffset = gmOffset + tileLen_;
                uint32_t nextLen = 0;
                if (nextOffset < rowOffset + rowLen_) {
                    nextLen = (rowOffset + rowLen_ - nextOffset);
                    if (nextLen > tileLen_) nextLen = tileLen_;
                    CopyInPhase3(nextOffset, nextLen);
                }

                ComputePhase3(curTileLen, logSumExp);
                CopyOutPhase3(gmOffset, curTileLen);

                gmOffset = nextOffset;
                curTileLen = nextLen;
                if (curTileLen == 0) break;
            }
        }
    }

private:
    // Phase 1 copy: load tile into inQueueX
    __aicore__ inline void CopyInPhase1(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    // Phase 1 compute: update rowMax = max(rowMax, localMax)
    __aicore__ inline void ComputePhase1(uint32_t len, float &rowMax) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        // Use ReduceMax to get local maximum
        AscendC::LocalTensor<float> localMax = tmpQueue_.AllocTensor<float>();
        AscendC::ReduceMax(localMax, xLocal, AscendC::ReductionType::REDUCE_AND, len);
        float localMaxVal = localMax.GetValue(0);
        rowMax = (localMaxVal > rowMax) ? localMaxVal : rowMax;
        tmpQueue_.FreeTensor(localMax);
        inQueueX_.FreeTensor(xLocal);
    }

    // Phase 2 copy: load tile into inQueueX
    __aicore__ inline void CopyInPhase2(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    // Phase 2 compute: compute exp(x - rowMax) and sum
    __aicore__ inline void ComputePhase2(uint32_t len, float rowMax, float &sumExp) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> expTmp = tmpQueue_.AllocTensor<float>();
        // x - rowMax
        AscendC::Sub(expTmp, xLocal, rowMax, len);
        // exp
        AscendC::Exp(expTmp, expTmp, len);
        // sum
        AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
        AscendC::ReduceAdd(sumLocal, expTmp, AscendC::ReductionType::REDUCE_AND, len);
        sumExp += sumLocal.GetValue(0);
        tmpQueue_.FreeTensor(sumLocal);
        tmpQueue_.FreeTensor(expTmp);
        inQueueX_.FreeTensor(xLocal);
    }

    // Phase 3 copy: load tile into inQueueX
    __aicore__ inline void CopyInPhase3(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    // Phase 3 compute: y = x - logSumExp
    __aicore__ inline void ComputePhase3(uint32_t len, float logSumExp) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Sub(yLocal, xLocal, logSumExp, len);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    // Phase 3 output: write tile back to global
    __aicore__ inline void CopyOutPhase3(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_; // reused across phases

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t rowLen_{0};
    uint32_t tileLen_{0};
    uint32_t numRows_{0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLength, tiling_data.tileLength, static_cast<float>(tiling_data.dim));
    op.Process();
}
