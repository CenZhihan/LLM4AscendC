#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: 4 vectors (anchor, positive, negative, temp) each of tileDim floats
// Total floats = 4 * tileDim => bytes = 16 * tileDim
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TripletMarginLossCustomTilingData tiling;
    // Use StorageShape* and GetStorageShape() to access dims
    const gert::StorageShape *anchorShape = context->GetInputShape(0);
    const auto &storageShape = anchorShape->GetStorageShape();
    const uint32_t totalSamples = static_cast<uint32_t>(storageShape.GetDim(0));
    const uint32_t totalDim     = static_cast<uint32_t>(storageShape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // Compute tileDim from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 16;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    uint32_t tileDim = maxTile;
    if (totalDim < tileDim) {
        uint32_t aligned = AlignDown(totalDim, ALIGN_ELEMS);
        tileDim = (aligned > 0) ? aligned : totalDim;
    }

    // blockDim: fixed to ensure enough parallelism
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_totalSamples(totalSamples);
    tiling.set_totalDim(totalDim);
    tiling.set_tileDim(tileDim);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLoss {
public:
    __aicore__ inline KernelTripletMarginLoss() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive,
                                GM_ADDR negative, GM_ADDR output,
                                uint32_t totalSamples, uint32_t totalDim,
                                uint32_t tileDim, float margin) {
        totalSamples_ = totalSamples;
        totalDim_ = totalDim;
        tileDim_ = tileDim;
        margin_ = margin;
        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalSamples * totalDim);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalSamples * totalDim);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalSamples * totalDim);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalSamples);

        pipe_.InitBuffer(inQAnchor_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(inQPositive_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(inQNegative_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tileDim_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalSamples_ == 0 || totalDim_ == 0 || tileDim_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over samples
        for (uint32_t sampleIdx = blockIdx; sampleIdx < totalSamples_; sampleIdx += blockNum) {
            float sumP = 0.0f;
            float sumN = 0.0f;
            uint32_t base = sampleIdx * totalDim_;
            uint32_t remaining = totalDim_;
            uint32_t offset = 0;

            while (remaining > 0) {
                uint32_t curDim = (remaining > tileDim_) ? tileDim_ : remaining;

                // Prefetch three input tiles in parallel
                AscendC::LocalTensor<float> aLocal = inQAnchor_.AllocTensor<float>();
                AscendC::DataCopy(aLocal, anchorGm_[base + offset], curDim);
                inQAnchor_.EnQue(aLocal);

                AscendC::LocalTensor<float> pLocal = inQPositive_.AllocTensor<float>();
                AscendC::DataCopy(pLocal, positiveGm_[base + offset], curDim);
                inQPositive_.EnQue(pLocal);

                AscendC::LocalTensor<float> nLocal = inQNegative_.AllocTensor<float>();
                AscendC::DataCopy(nLocal, negativeGm_[base + offset], curDim);
                inQNegative_.EnQue(nLocal);

                // Dequeue all three
                aLocal = inQAnchor_.DeQue<float>();
                pLocal = inQPositive_.DeQue<float>();
                nLocal = inQNegative_.DeQue<float>();

                // Compute sumP: (a - p)^2
                AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                AscendC::Sub(tmp, aLocal, pLocal, curDim);
                AscendC::Mul(tmp, tmp, tmp, curDim);
                float partialP = AscendC::VecReduceAdd(tmp, curDim);
                sumP += partialP;

                // Compute sumN: (a - n)^2
                AscendC::Sub(tmp, aLocal, nLocal, curDim);
                AscendC::Mul(tmp, tmp, tmp, curDim);
                float partialN = AscendC::VecReduceAdd(tmp, curDim);
                sumN += partialN;

                tmpQ_.FreeTensor(tmp);
                inQAnchor_.FreeTensor(aLocal);
                inQPositive_.FreeTensor(pLocal);
                inQNegative_.FreeTensor(nLocal);

                offset += curDim;
                remaining -= curDim;
            }

            // loss = max(sumP - sumN + margin, 0)
            float loss = sumP - sumN + margin_;
            loss = (loss > 0.0f) ? loss : 0.0f;
            outputGm_.SetValue(sampleIdx, loss);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQPositive_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQNegative_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t totalSamples_ {0};
    uint32_t totalDim_ {0};
    uint32_t tileDim_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor,
    GM_ADDR positive, GM_ADDR negative, GM_ADDR output,
    GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLoss op;
    op.Init(anchor, positive, negative, output,
            tiling_data.totalSamples, tiling_data.totalDim,
            tiling_data.tileDim, tiling_data.margin);
    op.Process();
}
