#include "kernel_operator.h"
#include "math.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float EPS = 1e-5f;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
                                GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures,
                                uint32_t outFeatures, uint32_t numGroups,
                                uint32_t tileLenOut, uint32_t tileLenIn)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        tileLenOut_ = tileLenOut;
        tileLenIn_ = tileLenIn;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inFeatures);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures * inFeatures);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * outFeatures);

        // Allocate UB buffers for one tile
        uint32_t wBufSize = tileLenOut_ * inFeatures_;
        uint32_t xBufSize = batchSize_ * inFeatures_;
        uint32_t yBufSize = batchSize_ * tileLenOut_;  // partial output
        uint32_t biasBufSize = tileLenOut_;
        uint32_t miscBufSize = tileLenOut_; // for sum/count etc

        pipe_.InitBuffer(inQueueWeight_, BUFFER_NUM, wBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, xBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueLinBias_, BUFFER_NUM, biasBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueBias_, BUFFER_NUM, biasBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueGnGamma_, BUFFER_NUM, biasBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueGnBeta_, BUFFER_NUM, biasBufSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, yBufSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, (xBufSize + yBufSize + miscBufSize) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Copy constant weights/biases once (they are same for all tiles)
        // For simplicity, assume tileLenOut == outFeatures (no tiling along out dim)
        // If tiling needed, would iterate over outFeatures.
        // Here we implement single tile covering entire outFeatures.
        uint32_t wSize = outFeatures_ * inFeatures_;
        uint32_t xSize = batchSize_ * inFeatures_;
        uint32_t ySize = batchSize_ * outFeatures_;

        // Prefetch all inputs
        AscendC::LocalTensor<float> weightLocal = inQueueWeight_.AllocTensor<float>();
        AscendC::DataCopy(weightLocal, weightGm_[0], wSize);
        inQueueWeight_.EnQue(weightLocal);

        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[0], xSize);
        inQueueX_.EnQue(xLocal);

        AscendC::LocalTensor<float> linBiasLocal = inQueueLinBias_.AllocTensor<float>();
        AscendC::DataCopy(linBiasLocal, linBiasGm_[0], outFeatures_);
        inQueueLinBias_.EnQue(linBiasLocal);

        AscendC::LocalTensor<float> biasLocal = inQueueBias_.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, biasGm_[0], outFeatures_);
        inQueueBias_.EnQue(biasLocal);

        AscendC::LocalTensor<float> gnGammaLocal = inQueueGnGamma_.AllocTensor<float>();
        AscendC::DataCopy(gnGammaLocal, gnGammaGm_[0], outFeatures_);
        inQueueGnGamma_.EnQue(gnGammaLocal);

        AscendC::LocalTensor<float> gnBetaLocal = inQueueGnBeta_.AllocTensor<float>();
        AscendC::DataCopy(gnBetaLocal, gnBetaGm_[0], outFeatures_);
        inQueueGnBeta_.EnQue(gnBetaLocal);

        // Compute: for each batch, do linear, add bias, hardtanh, mish, groupnorm
        // Use local tensor for partial results
        AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wTile = inQueueWeight_.DeQue<float>();
        AscendC::LocalTensor<float> linB = inQueueLinBias_.DeQue<float>();
        AscendC::LocalTensor<float> b = inQueueBias_.DeQue<float>();
        AscendC::LocalTensor<float> gG = inQueueGnGamma_.DeQue<float>();
        AscendC::LocalTensor<float> gB = inQueueGnBeta_.DeQue<float>();

        // Allocate temporary for output buffer
        AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

        // We'll batch process samples in chunks using grid stride for simplicity
        const uint32_t stride = 1; // one block
        for (uint32_t n = 0; n < batchSize_; n++) {
            // Linear: y[n,:] = x[n,:] * weight^T + lin_bias
            // weight matrix is outFeatures x inFeatures, stored row-major
            // We need to do matrix-vector multiplication for each n
            // Using AscendC matmul: MatMul(x[n,:], weight.T) but weight is in row-major.
            // For simplicity, compute manually via loops using local functions.
            // Since AscendC provides Mmad for matrix multiply, we'll use Muls and Adds.
            // Actually, we can use MatMul: MatMul(x[n:1,:], weight:1xoutFeatures, result).
            // But we have to reshape. We'll do elementwise: for each out feature, compute dot product.
            // Use VecMul and ReduceAdd? Too complex. Use AscendC's Mmad if available.
            // Alternative: use the TBE built-in MatMul? Not directly.
            // For this solution, we assume outFeatures is small enough to fit.
            // We'll implement a simple loop:
            for (uint32_t i = 0; i < outFeatures_; i++) {
                float sum = linB.GetValue(i);
                for (uint32_t j = 0; j < inFeatures_; j++) {
                    sum += xTile.GetValue(n*inFeatures_ + j) * wTile.GetValue(i*inFeatures_ + j);
                }
                yTile.SetValue(n*outFeatures_ + i, sum + b.GetValue(i)); // add bias
            }
        }

        // Now apply hardtanh, mish, groupnorm on yTile
        // Hardtanh: clamp [-1,1]
        for (uint32_t i = 0; i < batchSize_ * outFeatures_; i++) {
            float val = yTile.GetValue(i);
            if (val < -1.0f) val = -1.0f;
            else if (val > 1.0f) val = 1.0f;
            yTile.SetValue(i, val);
        }

        // Mish: y * tanh(log(1+exp(y)))
        // Use AscendC functions: Exp, Log, Tanh
        for (uint32_t i = 0; i < batchSize_ * outFeatures_; i++) {
            float v = yTile.GetValue(i);
            float sp = log(1.0f + exp(v)); // softplus
            float tanhsp = tanh(sp);
            yTile.SetValue(i, v * tanhsp);
        }

        // Group norm: For each sample, compute per-group mean and variance
        uint32_t perGroupSize = outFeatures_ / numGroups_;
        for (uint32_t n = 0; n < batchSize_; n++) {
            for (uint32_t g = 0; g < numGroups_; g++) {
                float sum = 0.0f;
                for (uint32_t k = 0; k < perGroupSize; k++) {
                    sum += yTile.GetValue(n*outFeatures_ + g*perGroupSize + k);
                }
                float mean = sum / perGroupSize;
                float var = 0.0f;
                for (uint32_t k = 0; k < perGroupSize; k++) {
                    float diff = yTile.GetValue(n*outFeatures_ + g*perGroupSize + k) - mean;
                    var += diff * diff;
                }
                var /= perGroupSize;
                float invStd = 1.0f / sqrt(var + EPS);
                for (uint32_t k = 0; k < perGroupSize; k++) {
                    uint32_t idx = n*outFeatures_ + g*perGroupSize + k;
                    float val = (yTile.GetValue(idx) - mean) * invStd * gG.GetValue(g*perGroupSize + k) + gB.GetValue(g*perGroupSize + k);
                    yTile.SetValue(idx, val);
                }
            }
        }

        outQueueY_.EnQue<float>(yTile);
        // Copy out
        AscendC::LocalTensor<float> yFinal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[0], yFinal, batchSize_ * outFeatures_);
        outQueueY_.FreeTensor(yFinal);

        // Free other tensors
        inQueueX_.FreeTensor(xTile);
        inQueueWeight_.FreeTensor(wTile);
        inQueueLinBias_.FreeTensor(linB);
        inQueueBias_.FreeTensor(b);
        inQueueGnGamma_.FreeTensor(gG);
        inQueueGnBeta_.FreeTensor(gB);
        tmpQueue_.FreeTensor(tmp);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWeight_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueLinBias_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBias_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueGnGamma_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueGnBeta_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_;
    uint32_t inFeatures_;
    uint32_t outFeatures_;
    uint32_t numGroups_;
    uint32_t tileLenOut_;
    uint32_t tileLenIn_;
};
