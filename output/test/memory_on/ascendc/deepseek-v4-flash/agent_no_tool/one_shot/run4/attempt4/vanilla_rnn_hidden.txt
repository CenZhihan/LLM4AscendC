kernel_src="""

#include "kernel_operator.h"
#include "matmul.h"
using namespace AscendC;

class KernelVanillaRnnHidden {
public:
    __aicore__ inline KernelVanillaRnnHidden() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR h, GM_ADDR w_i2h, GM_ADDR b_i2h,
        GM_ADDR w_h2o, GM_ADDR b_h2o,
        GM_ADDR y, GM_ADDR h_new,
        uint32_t batchSize, uint32_t inputSize,
        uint32_t hiddenSize, uint32_t outputSize)
    {
        batchSize_ = batchSize;
        inputSize_ = inputSize;
        hiddenSize_ = hiddenSize;
        outputSize_ = outputSize;
        K_ = inputSize_ + hiddenSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        hGm_.SetGlobalBuffer((__gm__ float*)h);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        h_newGm_.SetGlobalBuffer((__gm__ float*)h_new);

        pipe_.InitBuffer(bufInp_, K_ * sizeof(float));
        pipe_.InitBuffer(bufH_, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(bufPre_, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(bufHnew_, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(bufY_, outputSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // First matmul: inp [1, K] * w_i2h [K, H] + bias [H] -> pre [1, H]
        matmul1_.SetM(1);
        matmul1_.SetN(hiddenSize_);
        matmul1_.SetK(K_);
        matmul1_.SetTransA(false);
        matmul1_.SetTransB(false);
        matmul1_.SetBias(b_i2hGm_);

        // Second matmul: hnew [1, H] * w_h2o [H, O] + bias [O] -> y [1, O]
        matmul2_.SetM(1);
        matmul2_.SetN(outputSize_);
        matmul2_.SetK(hiddenSize_);
        matmul2_.SetTransA(false);
        matmul2_.SetTransB(false);
        matmul2_.SetBias(b_h2oGm_);

        for (uint32_t b = 0; b < batchSize_; b++) {
            LocalTensor<float> inpLocal = bufInp_.Get<float>();
            LocalTensor<float> hLocal = bufH_.Get<float>();
            DataCopy(inpLocal, xGm_[b * inputSize_], inputSize_);
            DataCopy(hLocal, hGm_[b * hiddenSize_], hiddenSize_);
            DataCopy(inpLocal[inputSize_], hLocal, hiddenSize_);

            // First matmul
            matmul1_.SetTensorA(inpLocal);
            matmul1_.SetTensorB(w_i2hGm_);
            LocalTensor<float> preLocal = bufPre_.Get<float>();
            matmul1_.SetTensorC(preLocal);
            matmul1_.IterateAll();

            // Tanh
            LocalTensor<float> hnewLocal = bufHnew_.Get<float>();
            Tanh(hnewLocal, preLocal, hiddenSize_);

            DataCopy(h_newGm_[b * hiddenSize_], hnewLocal, hiddenSize_);

            // Second matmul
            matmul2_.SetTensorA(hnewLocal);
            matmul2_.SetTensorB(w_h2oGm_);
            LocalTensor<float> yLocal = bufY_.Get<float>();
            matmul2_.SetTensorC(yLocal);
            matmul2_.IterateAll();

            DataCopy(yGm_[b * outputSize_], yLocal, outputSize_);
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> hGm_;
    GlobalTensor<float> w_i2hGm_;
    GlobalTensor<float> b_i2hGm_;
    GlobalTensor<float> w_h2oGm_;
    GlobalTensor<float> b_h2oGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> h_newGm_;

    TPipe pipe_;
    TBuf<TPosition::VECCALC> bufInp_;
    TBuf<TPosition::VECCALC> bufH_;
    TBuf<TPosition::VECCALC> bufPre_;
    TBuf<TPosition::VECCALC> bufHnew_;
    TBuf<TPosition::VECCALC> bufY_;

    Matmul matmul1_;
    Matmul matmul2_;

    uint32_t batchSize_;
    uint32_t inputSize_;
    uint32_t hiddenSize_;
    uint32_t outputSize_;
    uint32_t K_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(
    GM_ADDR x, GM_ADDR h, GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o,
    GM_ADDR y, GM_ADDR h_new,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHidden op;
    op.Init(x, h, w_i2h, b_i2h, w_h2o, b_h2o, y, h_new,
            tiling_data.batchSize, tiling_data.inputSize,
            tiling_data.hiddenSize, tiling_data.outputSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'vanilla_rnn_hidden_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, x, h0, w_i2h, b_i2h, w_h2o, b_h2o):
        T, B, I = x.shape
        H = w_i2h.size(0)
        O = w_h2o.size(0)
        # Transpose weights: kernel expects [K, H] and [H, O]
        w_i2h_t = w_i2h.t().contiguous()  # [H, K] -> [K, H]
        w_h2o_t = w_h2o.t().contiguous()  # [O, H] -> [H, O]
        h = h0
        outs = []
        fn = getattr(custom_ops_lib, _FN)
        for t in range(T):
            xt = x[t]  # [B, I]
            y_t, h_new = fn(xt, h, w_i2h_t, b_i2h, w_h2o_t, b_h2o)
            outs.append(y_t)
            h = h_new
        return torch.stack(outs, dim=0)
'''
