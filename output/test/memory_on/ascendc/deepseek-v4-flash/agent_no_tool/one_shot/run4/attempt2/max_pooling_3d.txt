#include "register/op_def_registry.h"
#include <cmath>

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static inline uint32_t AlignUp(uint32_t x, uint32_t a)
{
    return ((x + a - 1u) / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const std::vector<int64_t> xdims = xShape->GetOriginShape().GetDims();
    if (xdims.size() != 5) return ge::GRAPH_FAILED;

    uint32_t N = static_cast<uint32_t>(xdims[0]);
    uint32_t C = static_cast<uint32_t>(xdims[1]);
    uint32_t D = static_cast<uint32_t>(xdims[2]);
    uint32_t H = static_cast<uint32_t>(xdims[3]);
    uint32_t W = static_cast<uint32_t>(xdims[4]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];

    uint32_t kD = static_cast<uint32_t>(kernelSize);
    uint32_t kH = kD;
    uint32_t kW = kD;
    uint32_t sD = static_cast<uint32_t>(stride);
    uint32_t sH = sD;
    uint32_t sW = sD;
    uint32_t pD = static_cast<uint32_t>(padding);
    uint32_t pH = pD;
    uint32_t pW = pD;

    uint32_t outD = static_cast<uint32_t>(std::floor((D + 2 * pD - kD) / sD + 1));
    uint32_t outH = static_cast<uint32_t>(std::floor((H + 2 * pH - kH) / sH + 1));
    uint32_t outW = static_cast<uint32_t>(std::floor((W + 2 * pW - kW) / sW + 1));

    uint32_t totalInput = N * C * D * H * W;
    uint32_t totalOutput = N * C * outD * outH * outW;

    // Tiling: each block processes one or more batches/channels
    // For simplicity, use blockDim = N*C (number of independent planes)
    uint32_t blockDim = N * C;
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalInput);
    tiling.set_outputLength(totalOutput);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelD(kD);
    tiling.set_kernelH(kH);
    tiling.set_kernelW(kW);

    tiling.set_strideH(sH);
    tiling.set_strideW(sW);
