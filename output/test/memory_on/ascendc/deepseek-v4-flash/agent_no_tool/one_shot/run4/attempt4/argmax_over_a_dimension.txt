static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    uint32_t ndim = inputShape->GetDimNum();
    int64_t dim = -1;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs->GetAttrNum() > 0) {
        dim = attrs->GetAttrPointer<int64_t>(0)[0];
    }
    if (dim < 0) dim += ndim;
    uint32_t totalLength = 1;
    uint32_t outerDim = 1;
    uint32_t reduceDim = 1;
    uint32_t innerDim = 1;
    for (uint32_t i = 0; i < ndim; ++i) {
        uint32_t s = static_cast<uint32_t>(inputShape->GetDim(i));
        totalLength *= s;
        if (i < static_cast<uint32_t>(dim)) outerDim *= s;
        else if (i == static_cast<uint32_t>(dim)) reduceDim = s;
        else innerDim *= s;
    }
    uint32_t outputTotalLength = totalLength / reduceDim;

    // tileLength: number of output elements per tile
    // UB usage: for each output element we need to read reduceDim floats (temporal) and output one int64.
    // Simplification: process one output element per cycle; tileLength controls granularity.
    uint32_t tileLen = 128;
    if (outputTotalLength < tileLen) tileLen = outputTotalLength;

    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dimAttr(dim);
    tiling.set_outerDim(outerDim);
    tiling.set_reduceDim(reduceDim);
    tiling.set_innerDim(innerDim);
    tiling.set_outputTotalLength(outputTotalLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int64_t dim, uint32_t outerDim, uint32_t reduceDim,
                                uint32_t innerDim, uint32_t outputTotalLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        dimAttr_ = dim;
        outerDim_ = outerDim;
        reduceDim_ = reduceDim;
        innerDim_ = innerDim;
        outputTotalLen_ = outputTotalLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, outputTotalLen_);

        // We need two local queues for input values (float) and output indices (int64)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, reduceDim_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(int64_t));
    }

    __aicore__ inline void Process()
    {
        if (outputTotalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride on output elements
        const uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= outputTotalLen_) return;

        while (true) {
            uint32_t curTile = outputTotalLen_ - base;
            if (curTile > tileLen_) curTile = tileLen_;

            // For each output element in this tile, compute argmax
            // We'll process one by one, loading the necessary slice of input.
            for (uint32_t i = 0; i < curTile; ++i) {
                uint32_t outIdx = base + i;
                // Convert output linear index to outer and inner indices
                uint32_t innerIdx = outIdx % innerDim_;
                uint32_t outerIdx = outIdx / innerDim_;
                // Compute global starting offset in input for this output element
                uint32_t inputBase = outerIdx * reduceDim_ * innerDim_ + innerIdx;
                // Load reduceDim elements from input (stride = innerDim)
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                for (uint32_t j = 0; j < reduceDim_; ++j) {
                    xLocal[j] = xGm_[inputBase + j * innerDim_];
                }
                // Find argmax (index of maximum)
                float maxVal = xLocal[0];
                int64_t maxIdx = 0;
                for (uint32_t j = 1; j < reduceDim_; ++j) {
                    if (xLocal[j] > maxVal) {
                        maxVal = xLocal[j];
                        maxIdx = static_cast<int64_t>(j);
                    }
                }
                // Store result
                AscendC::LocalTensor<int64_t> yLocal = outQueueY_.AllocTensor<int64_t>();
                yLocal[0] = maxIdx;
                // Transfer to global
                AscendC::DataCopy(yGm_[outIdx], yLocal, 1);
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            base += stride;
            if (base >= outputTotalLen_) break;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int64_t dimAttr_ { -1 };
    uint32_t outerDim_ {1};
    uint32_t reduceDim_ {1};
    uint32_t innerDim_ {1};
    uint32_t outputTotalLen_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.dimAttr, tiling_data.outerDim,
            tiling_data.reduceDim, tiling_data.innerDim,
            tiling_data.outputTotalLength);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor argmax_over_a_dimension_npu(const at::Tensor& self, int64_t dim) {
    // Output type is int64
    std::vector<int64_t> outputShape;
    int64_t ndim = self.dim();
    if (dim < 0) dim += ndim;
    for (int64_t i = 0; i < ndim; ++i) {
        if (i != dim) outputShape.push_back(self.size(i));
    }
    at::Tensor result = at::empty(outputShape, self.options().dtype(at::kLong));
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result);
    return result;
}
