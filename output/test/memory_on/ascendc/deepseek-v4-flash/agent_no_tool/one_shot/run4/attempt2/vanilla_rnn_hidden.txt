kernel_src="""

#include "kernel_operator.h"
#include "register/op_def.h"
using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_NUM = 8;  // placeholder for internal tiling

class KernelVanillaRnnHidden {
public:
    __aicore__ inline KernelVanillaRnnHidden() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR h, GM_ADDR w_i2h, GM_ADDR b_i2h,
        GM_ADDR w_h2o, GM_ADDR b_h2o,
        GM_ADDR y, GM_ADDR h_new,
        uint32_t batchSize, uint32_t inputSize,
        uint32_t hiddenSize, uint32_t outputSize)
    {
        batchSize_ = batchSize;
        inputSize_ = inputSize;
        hiddenSize_ = hiddenSize;
        outputSize_ = outputSize;
        K_ = inputSize_ + hiddenSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        hGm_.SetGlobalBuffer((__gm__ float*)h);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        h_newGm_.SetGlobalBuffer((__gm__ float*)h_new);

        // Allocate local tensors for matmul intermediates
        pipe_.InitBuffer(inQueueIn_, BUFFER_NUM, batchSize_ * K_ * sizeof(float));
        pipe_.InitBuffer(inQueueH_, BUFFER_NUM, batchSize_ * hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_i2h_, BUFFER_NUM, hiddenSize_ * K_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_i2h_, BUFFER_NUM, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_h2o_, BUFFER_NUM, outputSize_ * hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_h2o_, BUFFER_NUM, outputSize_ * sizeof(float));
        pipe_.InitBuffer(outQueuePre_, BUFFER_NUM, batchSize_ * hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, batchSize_ * outputSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueH_, BUFFER_NUM, batchSize_ * hiddenSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Step 1: Concatenate x and h -> inp [batchSize, K]
        // But we avoid explicit concat; we load x and h and treat as contiguous.
        // Instead, we copy x and h to inpLocal as a single block.
        LocalTensor<float> inpLocal = inQueueIn_.AllocTensor<float>();
        LocalTensor<float> hLocal = inQueueH_.AllocTensor<float>();
        // Copy x to first part, then h to second part
        DataCopy(inpLocal, xGm_, batchSize_ * inputSize_);
        DataCopy(inpLocal[batchSize_ * inputSize_], hGm_, batchSize_ * hiddenSize_);
        inQueueIn_.EnQue(inpLocal);
        inQueueH_.EnQue(hLocal);

        // Load weights and biases
        LocalTensor<float> w_i2hLocal = inQueueW_i2h_.AllocTensor<float>();
        DataCopy(w_i2hLocal, w_i2hGm_, hiddenSize_ * K_);
        inQueueW_i2h_.EnQue(w_i2hLocal);
        LocalTensor<float> b_i2hLocal = inQueueB_i2h_.AllocTensor<float>();
        DataCopy(b_i2hLocal, b_i2hGm_, hiddenSize_);
        inQueueB_i2h_.EnQue(b_i2hLocal);

        // Matmul pre = inp * w_i2h^T   shape: [B, H]
        // Use Matmul with transB = true
        LocalTensor<float> preLocal = outQueuePre_.AllocTensor<float>();
        // We need to set up Matmul parameters
        Matmul<MatmulTraits<DataType::DT_FLOAT, CubeFormat::ND, false, true>> matmul1;
        matmul1.SetTensorA(inpLocal);        // [B, K]
        matmul1.SetTensorB(w_i2hLocal);      // [H, K]
        matmul1.SetBias(b_i2hLocal);         // [H]
        matmul1.SetTensorC(preLocal);        // [B, H]
        matmul1.IterateAll();
        outQueuePre_.EnQue(preLocal);

        // Free intermediate inputs
        inQueueIn_.FreeTensor(inpLocal);
        inQueueW_i2h_.FreeTensor(w_i2hLocal);
        inQueueB_i2h_.FreeTensor(b_i2hLocal);

        // Tanh activation
        LocalTensor<float> hNewLocal = outQueueH_.AllocTensor<float>();
        Tanh(hNewLocal, preLocal, batchSize_ * hiddenSize_);
        outQueueH_.EnQue(hNewLocal);
        outQueuePre_.FreeTensor(preLocal);

        // Copy h_new to global
        DataCopy(h_newGm_, hNewLocal, batchSize_ * hiddenSize_);
        outQueueH_.FreeTensor(hNewLocal);

        // Step 2: Output projection: y = h_new * w_h2o^T + b_h2o
        LocalTensor<float> hNewLocal2 = outQueueH_.AllocTensor<float>();
        // Re-read h_new from global (or we could have saved local, but easier to reload)
        DataCopy(hNewLocal2, h_newGm_, batchSize_ * hiddenSize_);
        outQueueH_.EnQue(hNewLocal2);

        LocalTensor<float> w_h2oLocal = inQueueW_h2o_.AllocTensor<float>();
        DataCopy(w_h2oLocal, w_h2oGm_, outputSize_ * hiddenSize_);
        inQueueW_h2o_.EnQue(w_h2oLocal);
        LocalTensor<float> b_h2oLocal = inQueueB_h2o_.AllocTensor<float>();
        DataCopy(b_h2oLocal, b_h2oGm_, outputSize_);
        inQueueB_h2o_.EnQue(b_h2oLocal);

        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        Matmul<MatmulTraits<DataType::DT_FLOAT, CubeFormat::ND, false, true>> matmul2;
        matmul2.SetTensorA(hNewLocal2);      // [B, H]
        matmul2.SetTensorB(w_h2oLocal);       // [O, H]
        matmul2.SetBias(b_h2oLocal);          // [O]
        matmul2.SetTensorC(yLocal);           // [B, O]
        matmul2.IterateAll();
        outQueueY_.EnQue(yLocal);

        // Free intermediates
        outQueueH_.FreeTensor(hNewLocal2);
        inQueueW_h2o_.FreeTensor(w_h2oLocal);
        inQueueB_h2o_.FreeTensor(b_h2oLocal);

        // Copy y to global
        DataCopy(yGm_, yLocal, batchSize_ * outputSize_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueIn_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueH_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueW_i2h_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueB_i2h_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueW_h2o_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueB_h2o_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueuePre_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueH_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> hGm_;
    GlobalTensor<float> w_i2hGm_;
    GlobalTensor<float> b_i2hGm_;
    GlobalTensor<float> w_h2oGm_;
    GlobalTensor<float> b_h2oGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> h_newGm_;

    uint32_t batchSize_;
    uint32_t inputSize_;
    uint32_t hiddenSize_;
    uint32_t outputSize_;
    uint32_t K_;  // = inputSize + hiddenSize
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(
    GM_ADDR x, GM_ADDR h, GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o,
    GM_ADDR y, GM_ADDR h_new,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHidden op;
    op.Init(x, h, w_i2h, b_i2h, w_h2o, b_h2o, y, h_new,
            tiling_data.batchSize, tiling_data.inputSize,
            tiling_data.hiddenSize, tiling_data.outputSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'vanilla_rnn_hidden_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, x, h0, w_i2h, b_i2h, w_h2o, b_h2o):
        T, B, I = x.shape
        H = w_i2h.size(0)
        O = w_h2o.size(0)
        h = h0
        outs = []
        fn = getattr(custom_ops_lib, _FN)
        for t in range(T):
            xt = x[t]  # [B, I]
            y_t, h_new = fn(xt, h, w_i2h, b_i2h, w_h2o, b_h2o)
            outs.append(y_t)
            h = h_new
        return torch.stack(outs, dim=0)
