host_operator_src="""
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;
static constexpr uint32_t OVERHEAD_BYTES = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);

    if (shapeA->GetDimNum() != 2 || shapeB->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    uint32_t M = static_cast<uint32_t>(shapeA->GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA->GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB->GetDim(1));

    if (static_cast<uint32_t>(shapeB->GetDim(0)) != K) {
        return ge::GRAPH_FAILED;
    }

    uint32_t bytesB = K * N * sizeof(float);
    uint32_t remaining = UB_BUDGET_BYTES - bytesB - OVERHEAD_BYTES;

    uint32_t tileM = 1;
    if (remaining > 0 && (K + N) > 0) {
        uint32_t bytesPerRowPair = 2 * (K + N) * sizeof(float);
        tileM = remaining / bytesPerRowPair;
        if (tileM > M) tileM = M;
        tileM = AlignDown(tileM, ALIGN_ELEMS);
        if (tileM < 1) tileM = 1;
    }

    uint32_t tileK = K;

    uint32_t numTilesM = CeilDiv(M, tileM);
    uint32_t blockDim = numTilesM;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);
    uint32_t M = static_cast<uint32_t>(shapeA->GetDim(0));
    uint32_t N = static_cast<uint32_t>(shapeB->GetDim(1));

    gert::Shape *outShape = context->GetOutputShape(0);
    outShape->SetDimNum(2);
    outShape->SetDim(0, static_cast<int64_t>(M));
    outShape->SetDim(1, static_cast<int64_t>(N));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileK,
                                uint32_t blockDim) {
        A_gm_.SetGlobalBuffer((__gm__ float*)A, M * K);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, M * N);
        M_ = M; K_ = K; N_ = N;
        tileM_ = tileM; tileK_ = tileK;
        blockDim_ = blockDim;
        uint32_t sizeATile = tileM_ * K_;
        uint32_t sizeB = K_ * N_;
        uint32_t sizeCTile = tileM_ * N_;
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, sizeATile * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, sizeB * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, sizeCTile * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Load B into local (assume fits in UB)
        AscendC::LocalTensor<float> B_local = inQueueB_.AllocTensor<float>();
        AscendC::DataCopy(B_local, B_gm_, K_ * N_);
        inQueueB_.EnQue(B_local);

        for (uint32_t row = blockIdx * tileM_; row < M_; row += blockNum * tileM_) {
            uint32_t curTileM = (row + tileM_ <= M_) ? tileM_ : M_ - row;

            // Load A tile
            AscendC::LocalTensor<float> A_local = inQueueA_.AllocTensor<float>();
            uint32_t offsetA = row * K_;
            AscendC::DataCopy(A_local, A_gm_[offsetA], curTileM * K_);
            inQueueA_.EnQue(A_local);

            // Dequeue B
            AscendC::LocalTensor<float> B_local_deq = inQueueB_.DeQue<float>();

            // Allocate C tile
            AscendC::LocalTensor<float> C_local = outQueueC_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

            // Compute C = A * B per row
            for (uint32_t r = 0; r < curTileM; r++) {
                AscendC::LocalTensor<float> rowA = A_local[r * K_];
                for (uint32_t n = 0; n < N_; n++) {
                    AscendC::LocalTensor<float> colB = B_local_deq[n * K_];
                    AscendC::Mul(tmp, rowA, colB, K_);
                    AscendC::LocalTensor<float> scalar = AscendC::ReduceSum(tmp, K_);
                    C_local(r * N_ + n) = scalar(0);
                }
            }

            tmpQueue_.FreeTensor(tmp);
            inQueueB_.EnQue(B_local_deq);
            outQueueC_.EnQue(C_local);

            // Write C tile to global
            AscendC::LocalTensor<float> C_out = outQueueC_.DeQue<float>();
            AscendC::DataCopy(C_gm_[row * N_], C_out, curTileM * N_);
            outQueueC_.FreeTensor(C_out);

            // Free A tile
            AscendC::LocalTensor<float> A_done = inQueueA_.DeQue<float>();
            inQueueA_.FreeTensor(A_done);
        }

        // Free B at the end
        AscendC::LocalTensor<float> B_done = inQueueB_.DeQue<float>();
        inQueueB_.FreeTensor(B_done);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::GlobalTensor<float> A_gm_, B_gm_, C_gm_;
    uint32_t M_, K_, N_;
    uint32_t tileM_, tileK_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileK,
            tiling_data.blockDim);
    op.Process();
}
