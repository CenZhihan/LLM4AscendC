host_operator_src="""

#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0); // [M, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K]
    const gert::Shape *b_shape = context->GetInputShape(2); // [N]

    if (x_shape->GetOriginShape().GetDimNum() != 2 ||
        w_shape->GetOriginShape().GetDimNum() != 2 ||
        b_shape->GetOriginShape().GetDimNum() != 1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    // w is [N, K], bias is [N]
    (void)K; // already consistent by front-end

    // Decide blockDim from tiles of rows
    // Favor one block per few rows to exploit parallelism
    uint32_t blockDim = M;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // Post-epilogue ReLU will stream over C of length M*N
    // Choose tile length from UB budget: two buffers y-in/y-out
    uint32_t maxTileElems = UB_BUDGET_BYTES / (2u * sizeof(float));
    maxTileElems = AlignDown(maxTileElems, ALIGN_ELEMS);
    if (maxTileElems < ALIGN_ELEMS) maxTileElems = ALIGN_ELEMS;
    if (maxTileElems > HARD_CAP_ELEMS) maxTileElems = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t postTileLen = maxTileElems;
    const uint64_t totalElems = static_cast<uint64_t>(M) * static_cast<uint64_t>(N);
    if (totalElems < postTileLen) {
        uint32_t aligned = AlignDown(static_cast<uint32_t>(totalElems), ALIGN_ELEMS);
        postTileLen = (aligned > 0u) ? aligned : static_cast<uint32_t>(totalElems);
    }

    tiling.set_m(M);
    tiling.set_n(N);
    tiling.set_k(static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1)));
    tiling.set_blockDim(blockDim);
    tiling.set_postTileLen(postTileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // [M, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K]
    if (x_shape->GetOriginShape().GetDimNum() != 2 ||
        w_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = x_shape->GetOriginShape().GetDim(0);
    const int64_t N = w_shape->GetOriginShape().GetDim(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, M);
    y_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

using AscendC::GetBlockIdx;
using AscendC::GetBlockNum;

class PostRelu {
public:
    __aicore__ inline PostRelu() {}

    __aicore__ inline void Init(GM_ADDR c, uint32_t totalLen, uint32_t tileLen)
    {
        totalLen_ = totalLen;
        tileLen_ = tileLen;
        cGm_.SetGlobalBuffer((__gm__ float*)c, totalLen_);
        pipe_.InitBuffer(inQ_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQ_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        const uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t cur = totalLen_ - base;
        if (cur > tileLen_) cur = tileLen_;
        CopyIn(base, cur);

        while (true) {
            const uint32_t nextBase = base + stride;
            uint32_t next = 0;
            if (nextBase < totalLen_) {
                next = totalLen_ - nextBase;
                if (next > tileLen_) next = tileLen_;
                CopyIn(nextBase, next);
            }

            Compute(cur);
            CopyOut(base, cur);

            if (next == 0) break;
            base = nextBase;
            cur = next;
        }
    }

private:
    static constexpr int32_t BUFFER_NUM = 2;

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> t = inQ_.AllocTensor<float>();
        AscendC::DataCopy(t, cGm_[gmOffset], len);
        inQ_.EnQue<float>(t);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> src = inQ_.DeQue<float>();
        AscendC::LocalTensor<float> dst = outQ_.AllocTensor<float>();

        AscendC::Maxs(dst, src, 0.0f, len);

        outQ_.EnQue<float>(dst);
        inQ_.FreeTensor(src);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> dst = outQ_.DeQue<float>();
        AscendC::DataCopy(cGm_[gmOffset], dst, len);
        outQ_.FreeTensor(dst);
    }

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQ_;
    AscendC::GlobalTensor<float> cGm_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
#if defined(ASCENDC_CPU_DEBUG)
    if (g_coreType == AscendC::AIV) {
        return;
    }
#endif
    // Prepare tiling
    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    // Define matmul types: A [M,K], B is provided as [N,K] but transposed to [K,N]
    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, true,  LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    AscendC::TPipe pipe;

    // Create and register matmul object
    // Note: this relies on AscendC matmul builder available in the toolchain.
    MatmulL0CacheCustom::MatmulKernel<half, half, float, float> matmulKernel;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulKernel.matmulObj, &tiling);

    // Initialize and compute GEMM with bias: y = x * w^T + bias
    matmulKernel.Init(x, w, bias, y, tiling, false, true);
    matmulKernel.Process();

    // Epilogue: ReLU in-place on y
    const uint32_t M = tiling.m;
    const uint32_t N = tiling.n;
    const uint32_t total = M * N;
    const uint32_t postTileLen = tiling.postTileLen;

    PostRelu relu;
    relu.Init(y, total, postTileLen);
    relu.Process();
}

#ifndef ASCENDC_CPU_DEBUG
#endif
"""

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor gemm_add_relu_impl_npu(const at::Tensor& x, const at::Tensor& w, const at::Tensor& bias) {
    // x: [M, K], w: [N, K], bias: [N]
    TORCH_CHECK(x.dim() == 2, \"x must be 2D\");
    TORCH_CHECK(w.dim() == 2, \"w must be 2D\");
    TORCH_CHECK(bias.dim() == 1, \"bias must be 1D\");
    TORCH_CHECK(x.size(1) == w.size(1), \"in_features mismatch\");
    TORCH_CHECK(w.size(0) == bias.size(0), \"out_features mismatch\");

    const auto M = x.size(0);
    const auto N = w.size(0);

    at::Tensor y = at::empty({M, N}, x.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnGemmAddReluCustom, x, w, bias, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"gemm_add_relu_custom\", &gemm_add_relu_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"gemm_add_relu_custom\", &gemm_add_relu_impl_npu, \"GEMM + Bias + ReLU fused (GemmAddReluCustom on NPU)\");
}
