#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
static inline int64_t FloorDiv(int64_t a, int64_t b) {
    // assumes b>0
    if (a >= 0) return a / b;
    return -(( -a + b - 1 ) / b);
}
static inline int64_t ClampNonNegative(int64_t v) {
    return v > 0 ? v : 0;
}
} // anonymous namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    if (x_shape->GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }
    const int64_t N = x_shape->GetOriginShape().GetDim(0);
    const int64_t C = x_shape->GetOriginShape().GetDim(1);
    const int64_t L = x_shape->GetOriginShape().GetDim(2);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride_ptr  = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding_ptr = attrs->GetAttrPointer<int64_t>(2);
    int64_t K = *kernel_size;
    int64_t S = *stride_ptr;
    int64_t P = *padding_ptr;
    if (S == 0) S = K; // default stride == kernel_size

    // output length: floor((L + 2P - K)/S) + 1, clamped to >=0
    int64_t numer = L + 2 * P - K;
    int64_t outL = 0;
    if (numer >= 0) {
        outL = numer / S + 1;
    } else {
        outL = 0;
    }

    const uint32_t uN = static_cast<uint32_t>(N);
    const uint32_t uC = static_cast<uint32_t>(C);
    const uint32_t uL = static_cast<uint32_t>(L);
    const uint32_t uOutL = static_cast<uint32_t>(ClampNonNegative(outL));
    const uint32_t uK = static_cast<uint32_t>(K);
    const uint32_t uS = static_cast<uint32_t>(S);
    const uint32_t uP = static_cast<uint32_t>(P);

    const uint32_t totalLines = uN * uC;
    uint32_t blockDim = totalLines;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(uN);
    tiling.set_C(uC);
    tiling.set_L(uL);
    tiling.set_outL(uOutL);
    tiling.set_kernelSize(uK);
    tiling.set_stride(uS);
    tiling.set_padding(uP);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 1;
constexpr float NEG_INF_F = -3.402823466e+38f;

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t L, uint32_t outL,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding)
    {
        N_ = N; C_ = C; L_ = L; outL_ = outL;
        K_ = kernelSize; S_ = stride; P_ = padding;

        totalLines_ = N_ * C_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * C_ * L_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * C_ * outL_);

        pipe_.InitBuffer(inQueue_, IN_BUF_NUM, L_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUT_BUF_NUM, outL_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLines_ == 0 || L_ == 0 || outL_ == 0) {
            return;
        }
        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        // grid-stride over lines (N*C)
        for (uint32_t line = blockIdx; line < totalLines_; line += blockNum) {
            // Copy one full line of length L_ from GM to UB
            CopyInLine(line);
            // Compute pooling for the line
            ComputeLine();
            // Copy result line back to GM
            CopyOutLine(line);
        }
    }

private:
    __aicore__ inline void CopyInLine(uint32_t line)
    {
        AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
        const uint64_t base = (uint64_t)line * L_;
        AscendC::DataCopy(xLocal, xGm_[base], L_);
        inQueue_.EnQue<float>(xLocal);
    }

    __aicore__ inline void ComputeLine()
    {
        AscendC::LocalTensor<float> xLocal = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

        // Sliding window max over [0, L_)
        for (uint32_t o = 0; o < outL_; ++o) {
            // start index in input considering padding
            int64_t start = static_cast<int64_t>(o) * static_cast<int64_t>(S_) - static_cast<int64_t>(P_);
            float maxv = NEG_INF_F;
            // Iterate kernel window
            for (uint32_t k = 0; k < K_; ++k) {
                int64_t idx = start + static_cast<int64_t>(k);
                if (idx >= 0 && idx < static_cast<int64_t>(L_)) {
                    float v = xLocal.GetValue(static_cast<uint32_t>(idx));
                    maxv = (v > maxv) ? v : maxv;
                } else {
                    // padded region contributes -inf
                    // maxv unchanged
                }
            }
            yLocal.SetValue(o, maxv);
        }

        outQueue_.EnQue<float>(yLocal);
        inQueue_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutLine(uint32_t line)
    {
        AscendC::LocalTensor<float> yLocal = outQueue_.DeQue<float>();
        const uint64_t base = (uint64_t)line * outL_;
        AscendC::DataCopy(yGm_[base], yLocal, outL_);
        outQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, L_{0}, outL_{0};
    uint32_t K_{1}, S_{1}, P_{0};
    uint32_t totalLines_{0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.L, tiling_data.outL,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t compute_out_len(int64_t L, int64_t K, int64_t S, int64_t P) {
    if (S == 0) S = K;
    int64_t numer = L + 2 * P - K;
    if (numer < 0) return 0;
    return numer / S + 1;
}

at::Tensor max_pooling_1d_impl_npu(const at::Tensor& self,
                                   int64_t kernel_size,
                                   int64_t stride,
                                   int64_t padding,
                                   bool return_indices)
{
    TORCH_CHECK(self.dim() == 3, "max_pooling_1d_custom expects 3D input (N, C, L)");
    if (stride == 0) stride = kernel_size;
    const int64_t N = self.size(0);
    const int64_t C = self.size(1);
    const int64_t L = self.size(2);
    const int64_t outL = compute_out_len(L, kernel_size, stride, padding);
    at::Tensor output = at::empty({N, C, outL}, self.options());
    // Indices are not produced by this custom kernel; return_indices is accepted but ignored.

    EXEC_NPU_CMD(aclnnMaxPooling1dCustom, self, kernel_size, stride, padding, return_indices, output);
    return output;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_1d_custom", &max_pooling_1d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_1d_custom", &max_pooling_1d_impl_npu,
          "MaxPooling1d (MaxPooling1dCustom on NPU)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_1d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0, return_indices: bool = False):
        super(ModelNew, self).__init__()
        self.kernel_size = int(kernel_size)
        self.stride = int(stride) if stride is not None else 0  # 0 means default to kernel_size in kernel
        self.padding = int(padding)
        self.return_indices = bool(return_indices)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding, self.return_indices)
