host_operator_src="""

#include "conv_pointwise2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
}

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core UB share
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MIN_BLOCKS = 8;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape* xShape = context->GetInputShape(0);
    const gert::Shape* wShape = context->GetInputShape(1);

    if (xShape->GetOriginShape().GetDimNum() != 4 || wShape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t n64 = xShape->GetOriginShape().GetDim(0);
    const uint64_t cIn64 = xShape->GetOriginShape().GetDim(1);
    const uint64_t h64 = xShape->GetOriginShape().GetDim(2);
    const uint64_t w64 = xShape->GetOriginShape().GetDim(3);

    const uint64_t cOut64 = wShape->GetOriginShape().GetDim(0);
    const uint64_t cInW64 = wShape->GetOriginShape().GetDim(1);
    const uint64_t kH64 = wShape->GetOriginShape().GetDim(2);
    const uint64_t kW64 = wShape->GetOriginShape().GetDim(3);

    if (cIn64 != cInW64 || kH64 != 1 || kW64 != 1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t n = static_cast<uint32_t>(n64);
    const uint32_t cIn = static_cast<uint32_t>(cIn64);
    const uint32_t cOut = static_cast<uint32_t>(cOut64);
    const uint32_t H = static_cast<uint32_t>(h64);
    const uint32_t W = static_cast<uint32_t>(w64);
    const uint32_t hwSize = H * W;
    const uint32_t totalP = n * hwSize;

    // UB layout per tile:
    //   - weights buffer (cOut * cIn) floats
    //   - x buffer: tileP floats
    //   - y accumulators: ocTile * tileP floats
    const uint32_t weightsFloats = cOut * cIn;
    const uint32_t weightsBytes = weightsFloats * 4u;

    uint32_t ocTile = (cOut >= 8u) ? 8u : (cOut >= 4u ? 4u : cOut);
    if (ocTile == 0u) ocTile = 1u;

    // Compute max tileP given UB budget
    uint32_t availBytes = (UB_BUDGET_BYTES > weightsBytes) ? (UB_BUDGET_BYTES - weightsBytes) : 0u;
    // bytes needed per tileP = (1 + ocTile) * tileP * sizeof(float)
    uint32_t tileP = 0u;
    if (availBytes > 0u) {
        tileP = availBytes / ((1u + ocTile) * 4u);
    }
    if (tileP == 0u) tileP = ALIGN_ELEMS;
    tileP = AlignDown(tileP, ALIGN_ELEMS);
    if (tileP == 0u) tileP = ALIGN_ELEMS;
    if (tileP > hwSize) {
        tileP = AlignDown(hwSize, ALIGN_ELEMS);
        if (tileP == 0u) tileP = hwSize;
    }

    const uint32_t numTiles = CeilDiv(totalP, tileP);
    uint32_t blockDim = numTiles;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    ConvPointwise2dCustomTilingData tiling;
    tiling.set_n(n);
    tiling.set_cIn(cIn);
    tiling.set_cOut(cOut);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_hwSize(hwSize);
    tiling.set_totalP(totalP);
    tiling.set_tileP(tileP);
    tiling.set_ocTile(ocTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src="""

#include "kernel_operator.h"
#include "register/tilingdata_base.h"
#include "conv_pointwise2d_custom_tiling.h"
#include "utils/fill_utils.h"

using namespace AscendC;

// We keep buffers conservative to fit UB budget across diverse shapes.
constexpr int32_t IN_X_BUFFERS = 1;
constexpr int32_t YACC_BUFFERS = 1;
constexpr int32_t WEIGHT_BUFFERS = 1;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t h, uint32_t wDim,
                                uint32_t hwSize, uint32_t totalP,
                                uint32_t tileP, uint32_t ocTile)
    {
        n_ = n;
        cIn_ = cIn;
        cOut_ = cOut;
        h_ = h;
        w_ = wDim;
        hwSize_ = hwSize;
        totalP_ = totalP;
        tileP_ = tileP;
        ocTile_ = ocTile;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * cIn_ * hwSize_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)cOut_ * cIn_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * cOut_ * hwSize_);

        pipe_.InitBuffer(inXQueue_, IN_X_BUFFERS, tileP_ * sizeof(float));
        pipe_.InitBuffer(yAccQueue_, ocTile_, tileP_ * sizeof(float));
        // weights buffer will hold full matrix cOut_ * cIn_ floats
        pipe_.InitBuffer(wLocalQueue_, WEIGHT_BUFFERS, (uint32_t)((uint64_t)cOut_ * cIn_ * sizeof(float)));
        pipe_.InitBuffer(tmpQueue_, 1, tileP_ * sizeof(float)); // tmp for fma
    }

    __aicore__ inline void Process()
    {
        if (totalP_ == 0 || tileP_ == 0) return;

        // Preload full weights into UB once
        AscendC::LocalTensor<float> wUb = wLocalQueue_.AllocTensor<float>();
        AscendC::DataCopy(wUb, wGm_, (uint64_t)cOut_ * cIn_);
        // wUb layout: [oc][ci], contiguous along ci, length cIn_, stride between oc rows = cIn_.

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileP_;
        uint32_t base = blockIdx * tileP_;
        if (base >= totalP_) {
            wLocalQueue_.FreeTensor(wUb);
            return;
        }

        while (true) {
            // compute current batch index and position offset inside H*W
            const uint32_t nIdx = base / hwSize_;
            const uint32_t posHW = base % hwSize_;
            uint32_t curLen = hwSize_ - posHW;
            if (curLen > tileP_) curLen = tileP_;

            // Process output channels in tiles
            for (uint32_t ocStart = 0; ocStart < cOut_; ocStart += ocTile_) {
                const uint32_t ocCount = (ocStart + ocTile_ <= cOut_) ? ocTile_ : (cOut_ - ocStart);

                // allocate y accumulators for ocCount
                for (uint32_t t = 0; t < ocCount; ++t) {
                    AscendC::LocalTensor<float> yAcc = yAccQueue_.AllocTensor<float>();
                    AscendC::Duplicate<float>(yAcc, 0.0f, curLen);
                }

                // iterating over input channels
                for (uint32_t ci = 0; ci < cIn_; ++ci) {
                    // Load x vector for this ci and (nIdx, posHW:posHW+curLen)
                    AscendC::LocalTensor<float> xLocal = inXQueue_.AllocTensor<float>();
                    const uint64_t xOffset = ((uint64_t)nIdx * cIn_ + ci) * hwSize_ + posHW;
                    AscendC::DataCopy(xLocal, xGm_[xOffset], curLen);

                    // For each oc in this tile, accumulate yAcc += xLocal * wVal
                    for (uint32_t t = 0; t < ocCount; ++t) {
                        const uint32_t oc = ocStart + t;
                        AscendC::LocalTensor<float> yAccT = yAccQueue_.Get<float>(t);

                        // Load the scalar weight from wUb row major [oc][ci]
                        // UB index = oc * cIn_ + ci
                        const uint64_t wIndex = (uint64_t)oc * cIn_ + ci;
                        // Create a temporary scalar vector by scaling x with scalar weight.
                        // Muls accepts scalar; value taken from UB load to a temp scalar variable.
                        // Note: we rely on compiler support for per-iter scalar.
                        float wVal = wUb.GetValue(wIndex);

                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::Muls(tmp, xLocal, wVal, curLen);
                        AscendC::Add(yAccT, yAccT, tmp, curLen);
                        tmpQueue_.FreeTensor(tmp);
                    }

                    inXQueue_.FreeTensor(xLocal);
                }

                // Write back y accumulators to GM
                for (uint32_t t = 0; t < ocCount; ++t) {
                    const uint32_t oc = ocStart + t;
                    AscendC::LocalTensor<float> yAccT = yAccQueue_.Get<float>(t);
                    const uint64_t yOffset = ((uint64_t)nIdx * cOut_ + oc) * hwSize_ + posHW;
                    AscendC::DataCopy(yGm_[yOffset], yAccT, curLen);
                }

                // Free y accumulators
                for (uint32_t t = 0; t < ocCount; ++t) {
                    AscendC::LocalTensor<float> yAccT = yAccQueue_.Get<float>(t);
                    yAccQueue_.FreeTensor(yAccT);
                }
            }

            // advance base by grid-stride
            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalP_) break;
            base = nextBase;
        }

        wLocalQueue_.FreeTensor(wUb);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_X_BUFFERS> inXQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, WEIGHT_BUFFERS> wLocalQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 32> yAccQueue_; // set capacity upper bound, actual used = ocTile_

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, cOut_{0};
    uint32_t h_{0}, w_{0}, hwSize_{0};
    uint32_t totalP_{0}, tileP_{0}, ocTile_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cIn, tiling_data.cOut,
            tiling_data.h, tiling_data.w,
            tiling_data.hwSize, tiling_data.totalP,
            tiling_data.tileP, tiling_data.ocTile);
    op.Process();
}
