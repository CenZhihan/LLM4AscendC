host_operator_src="""

#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (a == 0) ? x : (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (b == 0) ? 0u : (a + b - 1u) / b; }
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative
static constexpr uint32_t ALIGN_ELEMS = 256;             // favor aligned DMA size
static constexpr uint32_t MIN_TILEW = 16;                // avoid too-small tiles

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    if (x_shape->GetStorageShape().GetDimNum() != 4 || w_shape->GetStorageShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(0));
    const uint32_t wCin = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(1));
    // Expect 1x1 kernel
    if (w_shape->GetStorageShape().GetDim(2) != 1 || w_shape->GetStorageShape().GetDim(3) != 1 || wCin != Cin) {
        return ge::GRAPH_FAILED;
    }

    // UB usage model:
    //   yLocal: tileW
    //   xLocal: tileW
    //   wRow  : Cin
    // Total floats = (2 * tileW + Cin)
    // bytes = 4 * (2*tileW + Cin) <= UB_BUDGET_BYTES
    const uint32_t budgetElems = UB_BUDGET_BYTES / 4u;
    uint32_t maxTileW = 0;
    if (budgetElems > Cin + 2u * MIN_TILEW) {
        maxTileW = (budgetElems - Cin) / 2u;
        maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
        if (maxTileW < MIN_TILEW) maxTileW = MIN_TILEW;
    } else {
        maxTileW = MIN_TILEW;
    }
    if (maxTileW > W) {
        // cap by width, still aligned if possible
        uint32_t aligned = AlignDown(W, ALIGN_ELEMS);
        maxTileW = (aligned > 0) ? aligned : W;
    }
    const uint32_t tileW = (maxTileW == 0) ? 1u : maxTileW;

    const uint32_t tilesPerRow = CeilDiv(W, tileW);
    const uint64_t totalTiles64 = static_cast<uint64_t>(N) * static_cast<uint64_t>(H) * static_cast<uint64_t>(tilesPerRow);
    uint32_t totalTiles = (totalTiles64 > 0xFFFFFFFFull) ? 0xFFFFFFFFu : static_cast<uint32_t>(totalTiles64);

    // Choose blockDim based on tiles
    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = (totalTiles == 0u ? 1u : totalTiles);
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_tileW(tileW);
    tiling.set_tilesPerRow(tilesPerRow);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetStorageShape().GetDimNum() != 4 || w_shape->GetStorageShape().GetDimNum() != 4) {
        return GRAPH_FAILED;
    }
    const int64_t N   = x_shape->GetStorageShape().GetDim(0);
    const int64_t Cin = x_shape->GetStorageShape().GetDim(1);
    const int64_t H   = x_shape->GetStorageShape().GetDim(2);
    const int64_t W   = x_shape->GetStorageShape().GetDim(3);

    const int64_t Cout = w_shape->GetStorageShape().GetDim(0);
    const int64_t wCin = w_shape->GetStorageShape().GetDim(1);
    if (w_shape->GetStorageShape().GetDim(2) != 1 || w_shape->GetStorageShape().GetDim(3) != 1 || wCin != Cin) {

    y_shape->SetStorageShape(out_ss);
    y_shape->SetOriginShape(out_ss);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_Q_DEPTH = 1;
constexpr int32_t OUT_Q_DEPTH = 1;
constexpr int32_t W_Q_DEPTH = 1;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W,
                                uint32_t tileW, uint32_t tilesPerRow)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout; H_ = H; W_ = W;
        tileW_ = tileW; tilesPerRow_ = tilesPerRow;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x));
        wGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w));
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y));

        pipe_.InitBuffer(inQueueX_, IN_Q_DEPTH, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_Q_DEPTH, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, W_Q_DEPTH, Cin_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || H_ == 0 || W_ == 0 || tileW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t tilesPerNH = static_cast<uint64_t>(H_) * static_cast<uint64_t>(tilesPerRow_);
        const uint64_t totalTiles = static_cast<uint64_t>(N_) * tilesPerNH;

        for (uint64_t t = blockIdx; t < totalTiles; t += blockNum) {
            // Decode tile index to (n, h, wBase)
            const uint64_t n = t / tilesPerNH;
            const uint64_t rem = t % tilesPerNH;
            const uint64_t h = rem / tilesPerRow_;
            const uint64_t wTileIdx = rem % tilesPerRow_;
            const uint32_t wBase = static_cast<uint32_t>(wTileIdx * tileW_);
            uint32_t curW = W_ - wBase;
            if (curW > tileW_) curW = tileW_;

            // Per output channel compute
            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Duplicate<float>(yLocal, 0.0f, curW);

                // Load weight row for oc: length Cin_
                AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                const uint64_t wRowOffset = static_cast<uint64_t>(oc) * static_cast<uint64_t>(Cin_);
                AscendC::DataCopy(wLocal, wGm_[wRowOffset], Cin_);

                // Accumulate over input channels
                __ubuf__ float* wRowPtr = reinterpret_cast<__ubuf__ float*>(wLocal.GetPhyAddr());
                for (uint32_t ic = 0; ic < Cin_; ++ic) {
                    const float wval = wRowPtr[ic];

                    // Load x segment [n, ic, h, wBase:wBase+curW)
                    const uint64_t xOffset =
                        (((n * static_cast<uint64_t>(Cin_) + ic) * static_cast<uint64_t>(H_) + h)
                         * static_cast<uint64_t>(W_)) + wBase;
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[xOffset], curW);

                    // y += x * wval
                    AscendC::Muls(xLocal, xLocal, wval, curW);
                    AscendC::Add(yLocal, yLocal, xLocal, curW);

                    inQueueX_.FreeTensor(xLocal);
                }

                // Store y segment
                const uint64_t yOffset =
                    (((n * static_cast<uint64_t>(Cout_) + oc) * static_cast<uint64_t>(H_) + h)
                     * static_cast<uint64_t>(W_)) + wBase;
                AscendC::DataCopy(yGm_[yOffset], yLocal, curW);

                wQueue_.FreeTensor(wLocal);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_Q_DEPTH> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_Q_DEPTH> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, W_Q_DEPTH> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0}, H_{0}, W_{0};
    uint32_t tileW_{0}, tilesPerRow_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W,
            tiling_data.tileW, tiling_data.tilesPerRow);
    op.Process();
}
