host_operator_src="""
#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (a == 0) ? x : (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// UB model (floats):
//   xLocal:     2 * tileIn
//   yLocal:     2 * tileOut
//   tmpFull:    1 * (tileIn - (K - 1))   // only used when stride > 1, budget for worst case as tileIn
// bytes ~= 4 * (3 * tileIn + 2 * tileOut), where tileIn = (tileOut - 1) * stride + kernel
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS     = 256;
static constexpr uint32_t BLOCKDIM_MIN    = 8;
static constexpr uint32_t BLOCKDIM_MAX    = 24;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    // Expect ND shape (N, C, L). Fallbacks if fewer dims.
    uint32_t dims = static_cast<uint32_t>(in_shape->GetStorageShape().GetDimNum());
    uint32_t N = 1, C = 1, L = 1;
    if (dims == 3) {
        N = static_cast<uint32_t>(in_shape->GetStorageShape().GetDim(0));
        C = static_cast<uint32_t>(in_shape->GetStorageShape().GetDim(1));
        L = static_cast<uint32_t>(in_shape->GetStorageShape().GetDim(2));
    } else if (dims == 2) {
        N = static_cast<uint32_t>(in_shape->GetStorageShape().GetDim(0));
        C = 1;
        L = static_cast<uint32_t>(in_shape->GetStorageShape().GetDim(1));
    } else if (dims == 1) {
        N = 1;
        C = 1;
        L = static_cast<uint32_t>(in_shape->GetStorageShape().GetDim(0));
    } else {
        // unknown dims -> use origin shape size as L
        L = static_cast<uint32_t>(in_shape->GetOriginShape().GetShapeSize());
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernelSizeAttr   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *strideAttr       = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *paddingAttr      = attrs->GetAttrPointer<int32_t>(2);
    const bool    *returnIdxAttr    = attrs->GetAttrPointer<bool>(3);

    uint32_t K = static_cast<uint32_t>(*kernelSizeAttr);
    uint32_t S = static_cast<uint32_t>(*strideAttr);
    uint32_t P = static_cast<uint32_t>(*paddingAttr);
    if (S == 0) S = K;  // default stride to kernel_size like PyTorch None

    // Compute output length
    int64_t temp = static_cast<int64_t>(L) + 2 * static_cast<int64_t>(P) - static_cast<int64_t>(K);
    uint32_t outL = (temp >= 0) ? static_cast<uint32_t>(temp / static_cast<int64_t>(S) + 1) : 0u;

    // Solve tileOut from UB budget: bytes = 4 * ( (3S+2)*tileOut + (3K - 3S) )
    uint32_t denom = (3u * S + 2u);
    uint32_t numer = (UB_BUDGET_BYTES / 4u);
    uint32_t base  = (3u * K >= 3u * S) ? (3u * K - 3u * S) : 0u;
    uint32_t tileOut = (numer > base && denom > 0u) ? (numer - base) / denom : ALIGN_ELEMS;

    // Align and clamp
    tileOut = AlignDown(tileOut, ALIGN_ELEMS);
    if (tileOut == 0u) tileOut = ALIGN_ELEMS;
    if (tileOut > outL && outL > 0) {
        uint32_t aligned = AlignDown(outL, ALIGN_ELEMS);
        tileOut = (aligned > 0) ? aligned : outL;
    }

    // Parallel across lines (N*C)
    const uint32_t lines = N * C;
    uint32_t blockDim = lines;
    if (blockDim < BLOCKDIM_MIN) blockDim = BLOCKDIM_MIN;
    if (blockDim > BLOCKDIM_MAX) blockDim = BLOCKDIM_MAX;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_l(L);
    tiling.set_outL(outL);
    tiling.set_tileOut(tileOut);
    tiling.set_kernelSize(K);
    tiling.set_stride(S);
    tiling.set_padding(P);
    tiling.set_returnIndices(static_cast<uint32_t>(*returnIdxAttr ? 1 : 0));
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);

    // Read attrs to infer output length on last dim
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernelSizeAttr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *strideAttr     = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *paddingAttr    = attrs->GetAttrPointer<int32_t>(2);

    uint32_t K = static_cast<uint32_t>(*kernelSizeAttr);
    uint32_t S = static_cast<uint32_t>(*strideAttr);
    uint32_t P = static_cast<uint32_t>(*paddingAttr);
    if (S == 0) S = K;

    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;

    // assume last dim is sequence length
    uint32_t dims = static_cast<uint32_t>(x_shape->GetStorageShape().GetDimNum());
    if (dims >= 1) {
        uint32_t L = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(dims - 1));
        int64_t temp = static_cast<int64_t>(L) + 2 * static_cast<int64_t>(P) - static_cast<int64_t>(K);
        uint32_t outL = (temp >= 0) ? static_cast<uint32_t>(temp / static_cast<int64_t>(S) + 1) : 0u;
        y_shape->GetStorageShape().SetDim(dims - 1, outL);
        y_shape->GetOriginShape().SetDim(dims - 1, outL);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t l, uint32_t outL,
                                uint32_t tileOut, uint32_t kernelSize, uint32_t stride, uint32_t padding,
                                uint32_t returnIndices)
    {
        n_ = n; c_ = c; l_ = l; outL_ = outL;
        tileOut_ = tileOut;
        k_ = kernelSize;
        s_ = stride == 0 ? kernelSize : stride;
        p_ = padding;
        returnIdx_ = returnIndices;

        totalIn_  = n_ * c_ * l_;
        totalOut_ = n_ * c_ * outL_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalIn_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut_);

        // tileIn = (tileOut - 1) * s + k
        uint32_t tileIn = (tileOut_ > 0) ? ((tileOut_ - 1u) * s_ + k_) : k_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileIn * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOut_ * sizeof(float));
        // tmp buffer to hold yFull when s_ > 1; allocate worst-case tileIn
        pipe_.InitBuffer(tmpQueue_, 1, tileIn * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outL_ == 0 || tileOut_ == 0 || n_ == 0 || c_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t lines = n_ * c_;

        // Grid-stride over (n*c) independent lines
        for (uint32_t line = blockIdx; line < lines; line += blockNum) {
            ProcessOneLine(line);
        }
    }

private:
    __aicore__ inline void ProcessOneLine(uint32_t line)
    {
        const uint32_t inLineBase  = line * l_;
        const uint32_t outLineBase = line * outL_;
        uint32_t outBase = 0;

        // Sliding over outputs in tiles of tileOut_
        while (outBase < outL_) {
            uint32_t remain = outL_ - outBase;
            uint32_t curTileOut = (remain < tileOut_) ? remain : tileOut_;
            // For this output tile, input span length and start index
            const uint32_t tileInLen = (curTileOut - 1u) * s_ + k_;
            int64_t startIn = static_cast<int64_t>(outBase) * static_cast<int64_t>(s_) - static_cast<int64_t>(p_);

            uint32_t leftPad = 0;
            uint32_t gmStart = 0;
            if (startIn < 0) {
                leftPad = static_cast<uint32_t>(-startIn);
                gmStart = 0;
            } else {
                gmStart = static_cast<uint32_t>(startIn);
            }

            uint32_t validIn = 0;
            if (gmStart < l_) {
                uint32_t maxAvail = l_ - gmStart;
                validIn = (tileInLen - leftPad < maxAvail) ? (tileInLen - leftPad) : maxAvail;
            } else {
                validIn = 0;
            }
            uint32_t rightPad = tileInLen - leftPad - validIn;

            CopyIn(inLineBase + gmStart, tileInLen, leftPad, validIn, rightPad);

            if (s_ == 1u) {
                ComputeStride1(curTileOut);
                CopyOut(outLineBase + outBase, curTileOut);
            } else {
                ComputeGeneral(curTileOut, tileInLen);
                CopyOut(outLineBase + outBase, curTileOut);
            }

            outBase += curTileOut;
        }
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t tileInLen,
                                  uint32_t leftPad, uint32_t validIn, uint32_t rightPad)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        // Fill with -INF to emulate max padding
        const float negInf = -3.4028235e38f;
        AscendC::Duplicate<float>(xLocal, negInf, tileInLen);
        if (validIn > 0) {
            AscendC::DataCopy(xLocal[leftPad], xGm_[gmOffset], validIn);
        }
        inQueueX_.EnQue<float>(xLocal);
        (void)rightPad; // rightPad is implied by initialization
    }

    // Fast path: stride == 1
    __aicore__ inline void ComputeStride1(uint32_t curTileOut)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // y = max over k windows across xLocal
        // Initialize yLocal with first window
        AscendC::DataCopy(yLocal, xLocal, curTileOut);
        for (uint32_t w = 1; w < k_; ++w) {
            AscendC::Max(yLocal, yLocal, xLocal[w], curTileOut);
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    // General path: stride >= 1 (when s_ > 1, perform downsample from full conv of stride 1)
    __aicore__ inline void ComputeGeneral(uint32_t curTileOut, uint32_t tileInLen)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> fullLocal = tmpQueue_.AllocTensor<float>();

        const uint32_t yFullLen = tileInLen - (k_ - 1u);
        // Compute full stride-1 pooled output into fullLocal
        AscendC::DataCopy(fullLocal, xLocal, yFullLen);
        for (uint32_t w = 1; w < k_; ++w) {
            AscendC::Max(fullLocal, fullLocal, xLocal[w], yFullLen);
        }
        AscendC::PipeBarrier<PIPE_V>();

        // Downsample: y[i] = full[i * s_]
        __ubuf__ float* fullPtr = reinterpret_cast<__ubuf__ float*>(fullLocal.GetPhyAddr());
        __ubuf__ float* yPtr    = reinterpret_cast<__ubuf__ float*>(yLocal.GetPhyAddr());
        for (uint32_t i = 0; i < curTileOut; ++i) {
            yPtr[i] = fullPtr[i * s_];
        }

        tmpQueue_.FreeTensor(fullLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>         tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, l_{0}, outL_{0};
    uint32_t tileOut_{0};
    uint32_t k_{1}, s_{1}, p_{0};
    uint32_t returnIdx_{0};
    uint32_t totalIn_{0}, totalOut_{0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.l, tiling_data.outL,
            tiling_data.tileOut, tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.returnIndices);
    op.Process();
}
