#include "max_pooling_1d_custom_tiling.h"

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const int64_t dims = x_shape->GetOriginShape().GetDimNum();

    uint32_t n = 1, c = 1, l = 1;
    if (dims == 3) {
        n = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
        c = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
        l = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    } else if (dims == 2) {
        // Accept (N, L) with implicit C=1
        n = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
        c = 1;
        l = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    } else {
        // Fallback: flatten everything to last dimension
        const uint32_t totalLength =
            static_cast<uint32_t>(x_shape->GetOriginShape().GetShapeSize());
        n = 1; c = 1; l = totalLength;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernelSizePtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stridePtr     = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *paddingPtr    = attrs->GetAttrPointer<int32_t>(2);
    const bool    *retIdxPtr     = attrs->GetAttrPointer<bool>(3);

    uint32_t k = static_cast<uint32_t>(*kernelSizePtr);
    uint32_t s = static_cast<uint32_t>((*stridePtr <= 0) ? *kernelSizePtr : *stridePtr);
    uint32_t p = static_cast<uint32_t>((*paddingPtr < 0) ? 0 : *paddingPtr);
    (void)retIdxPtr; // indices not produced in this custom kernel

    // Output length using PyTorch convention: floor((L + 2P - K)/S) + 1, clamp to >= 0
    int64_t numer = static_cast<int64_t>(l) + 2LL * static_cast<int64_t>(p) - static_cast<int64_t>(k);
    int64_t outLen64 = (numer >= 0) ? (numer / static_cast<int64_t>(s) + 1) : 0;
    uint32_t outLen = static_cast<uint32_t>(std::max<int64_t>(0, outLen64));

    // Choose blockDim based on number of (n*c) sequences to process.
    const uint32_t seqCount = n * c;
    uint32_t blockDim = seqCount;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 24u) blockDim = 24u; // conservative range for 910/910B series
    if (blockDim < 8u)  blockDim = std::min<uint32_t>(8u, std::max<uint32_t>(1u, seqCount));

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inLen(l);
    tiling.set_outLen(outLen);
    tiling.set_kernelSize(k);
    tiling.set_stride(s);
    tiling.set_padding(p);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernelSizePtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stridePtr     = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *paddingPtr    = attrs->GetAttrPointer<int32_t>(2);

    const int64_t dims = x_shape->GetOriginShape().GetDimNum();
    int64_t n = 1, c = 1, l = 1;
    if (dims == 3) {
        n = x_shape->GetOriginShape().GetDim(0);
        c = x_shape->GetOriginShape().GetDim(1);
        l = x_shape->GetOriginShape().GetDim(2);
    } else if (dims == 2) {
        n = x_shape->GetOriginShape().GetDim(0);
        c = 1;
        l = x_shape->GetOriginShape().GetDim(1);
    } else {
        n = 1; c = 1;
        l = x_shape->GetOriginShape().GetShapeSize();
    }

    const int64_t k = static_cast<int64_t>(*kernelSizePtr);
    const int64_t s = static_cast<int64_t>((*stridePtr <= 0) ? *kernelSizePtr : *stridePtr);
    const int64_t p = static_cast<int64_t>((*paddingPtr < 0) ? 0 : *paddingPtr);

    int64_t numer = l + 2 * p - k;
    int64_t outLen = (numer >= 0) ? (numer / s + 1) : 0;
    if (outLen < 0) outLen = 0;

    gert::Shape *y_shape = context->GetOutputShape(0);
    if (dims == 3) {
        y_shape->SetDimNum(3);
        y_shape->SetDim(0, n);
        y_shape->SetDim(1, c);
        y_shape->SetDim(2, outLen);
    } else {
        y_shape->SetDimNum(2);
        y_shape->SetDim(0, n * c);
        y_shape->SetDim(1, outLen);
    }

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include <cfloat>

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t inLen, uint32_t outLen,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding)
    {
        n_ = n; c_ = c;
        inLen_ = inLen; outLen_ = outLen;
        k_ = kernelSize; s_ = stride; p_ = padding;

        totalSeq_ = n_ * c_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalSeq_ * inLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (totalSeq_ == 0 || outLen_ == 0) ? 0 : (totalSeq_ * outLen_));

        // UB: one input sequence + one output sequence per task
        pipe_.InitBuffer(inQ_, 1, inLen_ * sizeof(float));
        pipe_.InitBuffer(outQ_, 1, outLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalSeq_ == 0 || inLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t seq = blockIdx; seq < totalSeq_; seq += blockNum) {
            const uint32_t xBase = seq * inLen_;
            const uint32_t yBase = seq * outLen_;
            CopyIn(xBase, inLen_);
            Compute();
            if (outLen_ > 0) {
                CopyOut(yBase, outLen_);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQ_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQ_.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> xLocal = inQ_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQ_.AllocTensor<float>();

        // For each output position compute max over window with zero padding outside range.
        for (uint32_t o = 0; o < outLen_; ++o) {
            // window [start, start + k_)
            int32_t start = static_cast<int32_t>(o) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);
            int32_t end = start + static_cast<int32_t>(k_);

            int32_t l = start < 0 ? 0 : start;
            int32_t r = end > static_cast<int32_t>(inLen_) ? static_cast<int32_t>(inLen_) : end;

            bool outside = (start < 0) || (end > static_cast<int32_t>(inLen_));

            float m;
            if (l < r) {
                // Initialize with first valid element
                m = xLocal.GetValue(static_cast<uint32_t>(l));
                for (int32_t idx = l + 1; idx < r; ++idx) {
                    float v = xLocal.GetValue(static_cast<uint32_t>(idx));
                    m = (v > m) ? v : m;
                }
                if (outside) {
                    // Compare with 0 introduced by padding
                    m = (0.0f > m) ? 0.0f : m;
                }
            } else {
                // Entire window is out of input range: all zeros
                m = 0.0f;
            }

            yLocal.SetValue(o, m);
        }

        outQ_.EnQue<float>(yLocal);
        inQ_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQ_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQ_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0};
    uint32_t inLen_{0}, outLen_{0};
    uint32_t k_{1}, s_{1}, p_{0};
    uint32_t totalSeq_{0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.inLen, tiling_data.outLen,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding);
    op.Process();
}
