host_operator_src="""

#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core
static constexpr uint32_t DMA_ALIGN_ELEMS = 4096;        // 16KB burst on fp32
static constexpr uint32_t MIN_BLOCKS = 1;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;

    // Shapes:
    // x: [M, K]
    // w: [N, K] (note: PyTorch Linear uses weight [N, K] and performs x @ w^T, we handle transB in kernel)
    // bias: [N]
    const gert::Shape* xShape = context->GetInputShape(0);
    const gert::Shape* wShape = context->GetInputShape(1);
    const gert::Shape* bShape = context->GetInputShape(2);

    if (xShape->GetOriginShape().GetDimNum() != 2 || wShape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t M64 = xShape->GetOriginShape().GetDim(0);
    const uint64_t Kx = xShape->GetOriginShape().GetDim(1);
    const uint64_t N64 = wShape->GetOriginShape().GetDim(0);
    const uint64_t Kw = wShape->GetOriginShape().GetDim(1);

    if (Kx != Kw) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(M64);
    const uint32_t N = static_cast<uint32_t>(N64);
    const uint32_t K = static_cast<uint32_t>(Kx);

    // Choose conservative blockDim=1 (single-core) for maximal correctness without cross-core offset logic.
    uint32_t blockDim = 1;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    // ReLU post-pass vector tiling: double-buffer (in/out)
    // bytes = 2 * tile * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (2u * sizeof(float));
    maxTile = AlignDown(maxTile, DMA_ALIGN_ELEMS);
    if (maxTile < DMA_ALIGN_ELEMS) maxTile = DMA_ALIGN_ELEMS;

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_blockDim(blockDim);
    tiling.set_reluTileLen(maxTile);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape* x = context->GetInputShape(0);
    const gert::Shape* w = context->GetInputShape(1);

    if (x->GetOriginShape().GetDimNum() != 2 || w->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const uint64_t M = x->GetOriginShape().GetDim(0);
    const uint64_t Kx = x->GetOriginShape().GetDim(1);
    const uint64_t N  = w->GetOriginShape().GetDim(0);
    const uint64_t Kw = w->GetOriginShape().GetDim(1);
    if (Kx != Kw) return GRAPH_FAILED;

    gert::Shape* y = context->GetOutputShape(0);
    std::vector<int64_t> yDims = {static_cast<int64_t>(M), static_cast<int64_t>(N)};
    y->GetStorageShape().SetDim(yDims.size(), yDims.data());
    y->GetOriginShape().SetDim(yDims.size(), yDims.data());
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

// We fuse: Y = ReLU( X @ W^T + bias ), with:
//   X: [M, K] in GM, row-major
//   W: [N, K] in GM, row-major (treated as transposed on B: isTransB = true)
//   bias: [N] in GM
//   Y: [M, N] in GM

struct GemmAddReluCustomTilingData {
    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t blockDim;
    uint32_t reluTileLen;
};

constexpr int32_t VEC_BUFFER_NUM = 2;

class KernelGemmAddReluCustom {
public:
    __aicore__ inline KernelGemmAddReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                const GemmAddReluCustomTilingData& td)
    {
        M_ = td.M;
        N_ = td.N;
        K_ = td.K;
        reluTileLen_ = td.reluTileLen;

        aGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), static_cast<uint64_t>(M_) * K_);
        bGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w), static_cast<uint64_t>(N_) * K_);
        biasGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bias), N_);
        cGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), static_cast<uint64_t>(M_) * N_);

        // Vector pipe for post ReLU pass
        pipe_.InitBuffer(inQ_, VEC_BUFFER_NUM, reluTileLen_ * sizeof(float));
        pipe_.InitBuffer(outQ_, VEC_BUFFER_NUM, reluTileLen_ * sizeof(float));
    }

    __aicore__ inline void ComputeGemmBias()
    {
        // Configure Matmul: A[M,K], B[N,K] with transB=true => B^T shape [K,N]
        // C[M,N] = A[M,K] * (B[N,K])^T + bias[N]
        // Bias is broadcast across rows.
        // Using ND formats in GM.
        // The internal library handles cube tiling; here we execute single-core (blockDim=1).
        // Note: The underlying Matmul config is assumed default (CFG_NORM).
        AscendC::Matmul<
            AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false>,
            AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true>,
            AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>,
            AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>,
            CFG_NORM> matmulObj;

        // Bind GM tensors directly
        matmulObj.SetTensorA(aGm_, M_, K_);
        matmulObj.SetTensorB(bGm_, N_, K_); // B stored as [N,K], transB=true
        matmulObj.SetTensorBias(biasGm_, N_);
        matmulObj.SetTensorC(cGm_, M_, N_);

        // Execute matmul + bias
        matmulObj.Process(&pipe_);
    }

    __aicore__ inline void PostRelu()
    {
        // Grid-stride over C[M,N] in GM and apply ReLU
        const uint32_t total = M_ * N_;
        if (total == 0) return;
        if (reluTileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t stride = blockNum * reluTileLen_;

        uint32_t base = blockIdx * reluTileLen_;
        if (base >= total) return;

        uint32_t curLen = total - base;
        if (curLen > reluTileLen_) curLen = reluTileLen_;
        CopyIn(base, curLen);

        while (true) {
            // Prefetch next
            const uint32_t nextBase = base + stride;
            uint32_t nextLen = 0;
            if (nextBase < total) {
                nextLen = total - nextBase;
                if (nextLen > reluTileLen_) nextLen = reluTileLen_;
                CopyIn(nextBase, nextLen);
            }

            ComputeRelu(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> inLocal = inQ_.AllocTensor<float>();
        AscendC::DataCopy(inLocal, cGm_[gmOffset], len);
        inQ_.EnQue(inLocal);
    }

    __aicore__ inline void ComputeRelu(uint32_t len)
    {
        AscendC::LocalTensor<float> inLocal = inQ_.DeQue<float>();
        AscendC::LocalTensor<float> outLocal = outQ_.AllocTensor<float>();

        AscendC::Maxs(outLocal, inLocal, 0.0f, len);

        inQ_.FreeTensor(inLocal);
        outQ_.EnQue(outLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> outLocal = outQ_.DeQue<float>();
        AscendC::DataCopy(cGm_[gmOffset], outLocal, len);
        outQ_.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, VEC_BUFFER_NUM> inQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, VEC_BUFFER_NUM> outQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t reluTileLen_ {0};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                                            GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    GemmAddReluCustomTilingData td;
    td.M = tiling_data.M;
    td.N = tiling_data.N;
    td.K = tiling_data.K;
    td.blockDim = tiling_data.blockDim;
    td.reluTileLen = tiling_data.reluTileLen;

    KernelGemmAddReluCustom op;
    op.Init(x, w, bias, y, td);
    op.ComputeGemmBias();
    op.PostRelu();
}
