namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    if (x_shape->GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t C = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t L = static_cast<uint32_t>(x_shape->GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kptr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *sptr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *pptr = attrs->GetAttrPointer<int32_t>(2);
    const bool *iptr = attrs->GetAttrPointer<bool>(3);

    const uint32_t K = static_cast<uint32_t>((kptr != nullptr && *kptr > 0) ? *kptr : 1);
    uint32_t stride = static_cast<uint32_t>((sptr != nullptr && *sptr > 0) ? *sptr : static_cast<int32_t>(K));
    const uint32_t padding = static_cast<uint32_t>((pptr != nullptr && *pptr >= 0) ? *pptr : 0);
    (void)iptr; // return_indices is not used in this implementation

    // Dilation fixed to 1 for this custom kernel
    const uint32_t dilation = 1u;
    if (L == 0 || K == 0) {
        return ge::GRAPH_FAILED;
    }

    // output length: floor((L + 2*padding - dilation*(K-1) - 1)/stride + 1)
    const int64_t numer = static_cast<int64_t>(L) + 2LL * static_cast<int64_t>(padding) - static_cast<int64_t>(dilation) * static_cast<int64_t>(K - 1) - 1LL;
    const uint32_t outL = (numer >= 0) ? static_cast<uint32_t>(numer / static_cast<int64_t>(stride) + 1LL) : 0u;

    // Tiling for output positions per iteration (tileOut)
    uint32_t tileOut = 0u;
    if (stride == 1u) {
        // Memory usage per iteration roughly: inSeg = tileOut + K - 1; y = tileOut; tmp = tileOut
        // bytes ~= 4 * ((K - 1) + 3 * tileOut). Leave extra headroom.
        const uint32_t budget = UB_BUDGET_BYTES - 4096; // keep some margin
        if (budget <= 4u * (K + 1u)) {
            tileOut = ALIGN_ELEMS; // minimal
        } else {
            uint32_t est = (budget / 4u);
            if (est > (K > 0 ? (K - 1u) : 0u)) {
                est = (est - (K - 1u)) / 3u; // from (K-1) + 3*tileOut <= est_bytes/4
            } else {
                est = ALIGN_ELEMS;
            }
            if (est > HARD_CAP_ELEMS) est = HARD_CAP_ELEMS;
            tileOut = AlignDown(est, ALIGN_ELEMS);
            if (tileOut < ALIGN_ELEMS) tileOut = ALIGN_ELEMS;
        }
    } else {
        // Generic fallback path processes 1 output at a time to avoid complex stride handling
        tileOut = 1u;
    }
    if (outL > 0 && tileOut > outL) {
        // small tensors
        tileOut = outL;
    }

    // Blocks: map over rows (N*C)
    const uint32_t rows = N * C;
    uint32_t blockDim = rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // Save tiling
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_L(L);
    tiling.set_K(K);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_outL(outL);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // no workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    if (x_shape->GetDimNum() != 3) {
        return GRAPH_FAILED;
    }
    const int64_t N = x_shape->GetDim(0);
    const int64_t C = x_shape->GetDim(1);
    const int64_t L = x_shape->GetDim(2);

    const int32_t* kptr = context->GetAttrs()->GetAttrPointer<int32_t>(0);
    const int32_t* sptr = context->GetAttrs()->GetAttrPointer<int32_t>(1);
    const int32_t* pptr = context->GetAttrs()->GetAttrPointer<int32_t>(2);

    const int64_t K = (kptr != nullptr && *kptr > 0) ? static_cast<int64_t>(*kptr) : 1;
    int64_t stride = (sptr != nullptr && *sptr > 0) ? static_cast<int64_t>(*sptr) : K;
    const int64_t padding = (pptr != nullptr && *pptr >= 0) ? static_cast<int64_t>(*pptr) : 0;

    const int64_t numer = L + 2 * padding - (K - 1) - 1;
    const int64_t outL = (numer >= 0) ? (numer / stride + 1) : 0;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, outL);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include <stdint.h>

constexpr int32_t OUT_BUFFER_NUM = 2;
constexpr float NEG_INF_F = -3.402823466e+38f; // approximate -FLT_MAX

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t L,
                                uint32_t K, uint32_t stride, uint32_t padding,
                                uint32_t outL, uint32_t tileOut)
    {
        N_ = N; C_ = C; L_ = L;
        K_ = K; stride_ = stride; padding_ = padding;
        outL_ = outL; tileOut_ = (tileOut == 0 ? 1u : tileOut);

        const uint32_t totalIn = N_ * C_ * L_;
        const uint32_t totalOut = N_ * C_ * outL_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalIn);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut);

        // Max input segment per tile: tileOut*stride + K - stride
        uint32_t inSegMax = (tileOut_ - 1u) * stride_ + K_;
        if (inSegMax < K_) inSegMax = K_;

        pipe_.InitBuffer(inQueueIn_, 1, inSegMax * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUFFER_NUM, tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || L_ == 0 || outL_ == 0 || K_ == 0) {
            return;
        }

        const uint32_t rows = N_ * C_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows; row += blockNum) {
            const uint32_t inRowBase = row * L_;
            const uint32_t outRowBase = row * outL_;

            for (uint32_t p0 = 0; p0 < outL_; p0 += tileOut_) {
                const uint32_t tileOutCur = AscendC::Min(tileOut_, outL_ - p0);

                // Compute segment to load from input with clipping (no explicit padding load)
                const int32_t baseStart = static_cast<int32_t>(p0) * static_cast<int32_t>(stride_) - static_cast<int32_t>(padding_);
                const int32_t lastOut = static_cast<int32_t>(p0 + tileOutCur - 1u);
                const int32_t lastEndExclusive = lastOut * static_cast<int32_t>(stride_) - static_cast<int32_t>(padding_) + static_cast<int32_t>(K_);

                const int32_t segStart = AscendC::Max(0, baseStart);
                const int32_t segEnd = AscendC::Min(static_cast<int32_t>(L_), lastEndExclusive);
                const uint32_t segLen = (segEnd > segStart) ? static_cast<uint32_t>(segEnd - segStart) : 0u;

                // Load input segment
                AscendC::LocalTensor<float> inLocal = inQueueIn_.AllocTensor<float>();
                if (segLen > 0) {
                    AscendC::DataCopy(inLocal, xGm_[inRowBase + static_cast<uint32_t>(segStart)], segLen);
                }
                inQueueIn_.EnQue(inLocal);

                // Compute max pooling for this tile
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

                // Initialize output with NEG_INF
                AscendC::Duplicate(yLocal, NEG_INF_F, tileOutCur);

                inLocal = inQueueIn_.DeQue<float>();

                if (stride_ == 1u) {
                    // Vectorized sliding window for stride=1 using segmented max with boundary checks
                    for (uint32_t i = 0; i < K_; ++i) {
                        // Valid j range: j in [jL, jR]
                        int32_t jL = segStart - (baseStart + static_cast<int32_t>(i));
                        if (jL < 0) jL = 0;
                        int32_t jR = segEnd - 1 - (baseStart + static_cast<int32_t>(i));
                        if (jR > static_cast<int32_t>(tileOutCur) - 1) jR = static_cast<int32_t>(tileOutCur) - 1;

                        if (jL <= jR) {
                            const uint32_t len = static_cast<uint32_t>(jR - jL + 1);
                            const uint32_t inOff = static_cast<uint32_t>((baseStart + jL + static_cast<int32_t>(i)) - segStart);
                            const uint32_t outOff = static_cast<uint32_t>(jL);

                            AscendC::LocalTensor<float> inSeg = inLocal[inOff];
                            AscendC::LocalTensor<float> ySeg = yLocal[outOff];
                            AscendC::Max(ySeg, ySeg, inSeg, len);
                        }
                    }
                } else {
                    // Generic fallback: process one output at a time
                    for (uint32_t j = 0; j < tileOutCur; ++j) {
                        int32_t s = baseStart + static_cast<int32_t>(j) * static_cast<int32_t>(stride_);
                        float curMax = NEG_INF_F;
                        for (uint32_t i = 0; i < K_; ++i) {
                            const int32_t idx = s + static_cast<int32_t>(i);
                            if (idx >= segStart && idx < segEnd) {
                                const uint32_t inIdx = static_cast<uint32_t>(idx - segStart);
                                float val = inLocal.GetValue(inIdx);
                                curMax = (val > curMax) ? val : curMax;
                            }
                        }
                        tmpLocal.SetValue(j, curMax);
                    }
                    AscendC::DataCopy(yLocal, tmpLocal, tileOutCur);
                }

                tmpQueue_.FreeTensor(tmpLocal);
                outQueueY_.EnQue<float>(yLocal);
                inQueueIn_.FreeTensor(inLocal);

                // Store results
                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[outRowBase + p0], yLocal, tileOutCur);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueIn_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, L_{0};
    uint32_t K_{0}, stride_{0}, padding_{0};
    uint32_t outL_{0}, tileOut_{0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.L,
            tiling_data.K, tiling_data.stride, tiling_data.padding,
            tiling_data.outL, tiling_data.tileOut);
    op.Process();
}
