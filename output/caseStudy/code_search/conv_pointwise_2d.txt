#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (a == 0) ? x : (x / a) * a; }
static inline uint32_t MinU32(uint32_t a, uint32_t b) { return (a < b) ? a : b; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

// Per-core UB budget (bytes)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// Favor larger DMA bursts and vector lanes
static constexpr uint32_t ALIGN_P = 256;
static constexpr uint32_t ALIGN_CO = 16;
static constexpr uint32_t MAX_BLOCKS = 24;

} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);   // [N, Cin, H, W]
    const gert::Shape *w_shape = context->GetInputShape(1);   // [Cout, Cin, 1, 1]
    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetDim(3));
    const uint32_t Cout= static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t CinW= static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t KH  = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t KW  = static_cast<uint32_t>(w_shape->GetDim(3));

    if (CinW != Cin || KH != 1 || KW != 1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Pn = H * W;

    // Choose tileCo first
    uint32_t tileCo = Cout;
    if (tileCo > 64u) tileCo = 64u;
    tileCo = AlignDown(tileCo, ALIGN_CO);
    if (tileCo == 0u) tileCo = MinU32(Cout, ALIGN_CO);

    // Given tileCo, choose tileP s.t.
    // 4 bytes * (Cin*tileCo + tileP*tileCo + 2*tileP) <= UB_BUDGET
    // tileP <= (UB/4 - Cin*tileCo) / (tileCo + 2)
    uint32_t ub_floats = UB_BUDGET_BYTES / 4u;
    uint32_t maxTileP = 0u;
    while (tileCo >= 1u) {
        if (ub_floats > Cin * tileCo) {
            uint32_t denom = tileCo + 2u;
            maxTileP = (ub_floats - Cin * tileCo) / ((denom == 0u) ? 1u : denom);
            maxTileP = AlignDown(maxTileP, ALIGN_P);
            if (maxTileP == 0u) {
                // try smaller tileCo
                tileCo = AlignDown(tileCo - 1u, ALIGN_CO);
                continue;
            }
            break;
        } else {
            tileCo = AlignDown(tileCo - 1u, ALIGN_CO);
        }
    }
    if (tileCo == 0u || maxTileP == 0u) {
        // fallback minimal tiling
        tileCo = MinU32(Cout, ALIGN_CO);
        if (tileCo == 0u) tileCo = Cout;
        uint32_t denom = tileCo + 2u;
        maxTileP = (ub_floats > Cin * tileCo) ? (ub_floats - Cin * tileCo) / ((denom == 0u) ? 1u : denom) : ALIGN_P;
        if (maxTileP == 0u) maxTileP = ALIGN_P;
        maxTileP = AlignDown(maxTileP, ALIGN_P);
        if (maxTileP == 0u) maxTileP = MinU32(Pn, ALIGN_P);
    }

    uint32_t tileP = maxTileP;
    if (Pn < tileP) {
        uint32_t aligned = AlignDown(Pn, ALIGN_P);
        tileP = (aligned > 0u) ? aligned : Pn;
    }

    // Blocks: parallelize across N (batch)
    uint32_t blockDim = (N == 0u) ? 1u : MinU32(N, MAX_BLOCKS);
    if (blockDim == 0u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    ConvPointwise2dCustomTilingData tiling;
    tiling.set_n(N);
    tiling.set_cIn(Cin);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_cOut(Cout);
    tiling.set_tileP(tileP);
    tiling.set_tileCo(tileCo);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

{
    const gert::Shape *x_shape = context->GetInputShape(0);  // [N,Cin,H,W]
    const gert::Shape *w_shape = context->GetInputShape(1);  // [Cout,Cin,1,1]

    const int64_t H = x_shape->GetDim(2);
    const int64_t W = x_shape->GetDim(3);

    y_shape->SetDim(0, N);

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t wdim, uint32_t cOut,
                                uint32_t tileP, uint32_t tileCo)
    {
        n_ = n;
        cin_ = cIn;
        cout_ = cOut;
        h_ = h;
        w_ = wdim;
        pPerImg_ = h_ * w_;
        tileP_ = tileP;
        tileCo_ = tileCo;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * cin_ * pPerImg_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cout_) * cin_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * cout_ * pPerImg_);

        // Buffers in UB
        // xVec: tileP_
        pipe_.InitBuffer(bufX_, 1, tileP_ * sizeof(float));
        // tmpVec: tileP_
        pipe_.InitBuffer(bufTmp_, 1, tileP_ * sizeof(float));
        // yBuf: tileP_ * tileCo_
        pipe_.InitBuffer(bufY_, 1, tileP_ * tileCo_ * sizeof(float));
        // wSub: cin_ * tileCo_
        pipe_.InitBuffer(bufWSub_, 1, cin_ * tileCo_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || pPerImg_ == 0 || tileP_ == 0 || tileCo_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        for (uint32_t ni = blockIdx; ni < n_; ni += blockNum) {
            // Iterate co tiles
            for (uint32_t coBase = 0; coBase < cout_; coBase += tileCo_) {
                const uint32_t curCo = (coBase + tileCo_ <= cout_) ? tileCo_ : (cout_ - coBase);

                // Preload weight submatrix [curCo, cin_] into UB, row-major per co
                LocalTensor<float> wSub = bufWSub_.AllocTensor<float>();
                for (uint32_t kk = 0; kk < curCo; ++kk) {
                    const uint32_t wOff = (coBase + kk) * cin_;
                    DataCopy(wSub[kk * cin_], wGm_[wOff], cin_);
                }

                // Iterate spatial tiles within image ni
                for (uint32_t pBase = 0; pBase < pPerImg_; pBase += tileP_) {
                    const uint32_t curP = (pBase + tileP_ <= pPerImg_) ? tileP_ : (pPerImg_ - pBase);

                    LocalTensor<float> yBuf = bufY_.AllocTensor<float>();
                    LocalTensor<float> tmp = bufTmp_.AllocTensor<float>();
                    // We will write yBuf rows (each length tileP_), so yBuf[k * tileP_] is y for channel (coBase+k)

                    bool firstCi = true;
                    for (uint32_t ci = 0; ci < cin_; ++ci) {
                        // Load x vector for this channel and spatial tile
                        LocalTensor<float> xVec = bufX_.AllocTensor<float>();
                        const uint64_t xOff = (static_cast<uint64_t>(ni) * cin_ + ci) * pPerImg_ + pBase;
                        DataCopy(xVec, xGm_[xOff], curP);

                        if (firstCi) {
                            // Initialize yBuf = w * x for first ci
                            for (uint32_t kk = 0; kk < curCo; ++kk) {
                                const float wval = wSub.GetValue(kk * cin_ + ci);
                                LocalTensor<float> yRow = yBuf[kk * tileP_];
                                Muls(yRow, xVec, wval, curP);
                            }
                            firstCi = false;
                        } else {
                            // Accumulate yBuf += w * x
                            for (uint32_t kk = 0; kk < curCo; ++kk) {
                                const float wval = wSub.GetValue(kk * cin_ + ci);
                                LocalTensor<float> yRow = yBuf[kk * tileP_];
                                Muls(tmp, xVec, wval, curP);
                                Add(yRow, yRow, tmp, curP);
                            }
                        }

                        bufX_.FreeTensor(xVec);
                    }

                    // Write y rows back to GM
                    for (uint32_t kk = 0; kk < curCo; ++kk) {
                        const uint64_t yOff = (static_cast<uint64_t>(ni) * cout_ + (coBase + kk)) * pPerImg_ + pBase;
                        LocalTensor<float> yRow = yBuf[kk * tileP_];
                        DataCopy(yGm_[yOff], yRow, curP);
                    }

                    bufTmp_.FreeTensor(tmp);
                    bufY_.FreeTensor(yBuf);
                }

                bufWSub_.FreeTensor(wSub);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufTmp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufWSub_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t h_{0}, w_{0}, pPerImg_{0};
    uint32_t tileP_{0}, tileCo_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cIn, tiling_data.h, tiling_data.w, tiling_data.cOut,
            tiling_data.tileP, tiling_data.tileCo);
    op.Process();
}
