host_operator_src="""
#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"
#include "tiling/platform/platform_ascendc.h"

namespace {
static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Shapes:
    // x: [M, K]
    // w: [N, K] (PyTorch Linear uses w^T in GEMM, so we set isTransB = true)
    // bias: [N]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    const uint64_t x_total = x_shape->GetOriginShape().GetShapeSize();
    const uint64_t w_total = w_shape->GetOriginShape().GetShapeSize();
    const uint64_t bias_total = b_shape->GetOriginShape().GetShapeSize();

    if (bias_total == 0 || w_total == 0) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(bias_total);
    const uint32_t K = static_cast<uint32_t>(w_total / bias_total);
    if (K == 0) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t M = static_cast<uint32_t>(x_total / K);

    // Use CANN matmul tiling API to generate TCubeTiling
    platform_ascendc::PlatformAscendC *platform = platform_ascendc::PlatformAscendCManager::GetInstance();
    if (platform == nullptr) {
        return ge::GRAPH_FAILED;
    }

    matmul_tiling::MultiCoreMatmulTiling tilingApi(*platform);

    // Half inputs, float accumulation/output, bias as float
    matmul_tiling::TPosition leftPosition = matmul_tiling::TPosition::GM;
    matmul_tiling::CubeFormat leftFormat = matmul_tiling::CubeFormat::ND;
    matmul_tiling::DataType leftDtype = matmul_tiling::DataType::DT_FLOAT16;
    bool isTransA = false;

    matmul_tiling::TPosition rightPosition = matmul_tiling::TPosition::GM;
    matmul_tiling::CubeFormat rightFormat = matmul_tiling::CubeFormat::ND;
    matmul_tiling::DataType rightDtype = matmul_tiling::DataType::DT_FLOAT16;
    bool isTransB = true; // w is [N, K]; need B^T

    matmul_tiling::TPosition resultPosition = matmul_tiling::TPosition::GM;
    matmul_tiling::CubeFormat resultFormat = matmul_tiling::CubeFormat::ND;
    matmul_tiling::DataType resultDtype = matmul_tiling::DataType::DT_FLOAT;

    matmul_tiling::TPosition biasPosition = matmul_tiling::TPosition::GM;
    matmul_tiling::CubeFormat biasFormat = matmul_tiling::CubeFormat::ND;
    matmul_tiling::DataType biasDtype = matmul_tiling::DataType::DT_FLOAT;

    // Choose a reasonable core number and base split
    const uint32_t baseM = 256;
    const uint32_t baseN = 128;
    uint32_t gridM = CeilDiv(M, baseM);
    uint32_t gridN = CeilDiv(N, baseN);
    uint32_t blockDim = gridM * gridN;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 24u) blockDim = 24u; // AICore cap

    tilingApi.SetDim(blockDim);
    tilingApi.SetAType(leftPosition, leftFormat, leftDtype, isTransA);
    tilingApi.SetBType(rightPosition, rightFormat, rightDtype, isTransB);
    tilingApi.SetCType(resultPosition, resultFormat, resultDtype);
    tilingApi.SetBiasType(biasPosition, biasFormat, biasDtype);

    tilingApi.SetOrgShape(M, N, K);
    tilingApi.SetShape(M, N, K);
    tilingApi.SetBias(true);
    tilingApi.SetTraverse(matmul_tiling::MatrixTraverse::FIRSTM);
    tilingApi.SetFixSplit(baseM, baseN, -1);
    tilingApi.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int64_t res = tilingApi.GetTiling(tilingData);
    tilingData.set_stepM(1);
    tilingData.set_stepN(1);
    if (res == -1) {
        return ge::GRAPH_FAILED;
    }

    context->SetBlockDim(blockDim);
    tilingData.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tilingData.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0; // no user workspace

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // x: [M, K], w: [N, K], bias: [N] -> y: [M, N]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(2);

    const uint64_t x_total = x_shape->GetOriginShape().GetShapeSize();
    const uint64_t bias_total = b_shape->GetOriginShape().GetShapeSize();

    const gert::Shape *w_shape = context->GetInputShape(1);
    const uint64_t w_total = w_shape->GetOriginShape().GetShapeSize();

    if (bias_total == 0) return GRAPH_FAILED;
    const uint32_t N = static_cast<uint32_t>(bias_total);
    const uint32_t K = static_cast<uint32_t>(w_total / bias_total);
    if (K == 0) return GRAPH_FAILED;
    const uint32_t M = static_cast<uint32_t>(x_total / K);

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> dims = {static_cast<int64_t>(M), static_cast<int64_t>(N)};
    y_shape->SetDimNum(dims.size());
    for (size_t i = 0; i < dims.size(); ++i) {
        y_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output is float
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"
#include "tiling/tiling_api.h"

using namespace matmul;
using namespace AscendC;

__aicore__ inline void CopyTiling(optiling::TCubeTiling* tiling, GM_ADDR tilingGM)
{
    uint32_t* ptr = reinterpret_cast<uint32_t*>(tiling);
    auto tiling32 = reinterpret_cast<__gm__ uint32_t*>(tilingGM);
    for (uint32_t i = 0; i < sizeof(optiling::TCubeTiling) / sizeof(uint32_t); i++, ptr++) {
        *ptr = *(tiling32 + i);
    }
}

template <typename AType, typename BType, typename CType, typename BiasType>
class MatmulReluKernel {
public:
    __aicore__ inline MatmulReluKernel() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR bias, GM_ADDR c, GM_ADDR workspace,
                                const optiling::TCubeTiling& tiling, TPipe* pipe)
    {
        this->tiling_ = tiling;
        aGm_.SetGlobalBuffer(reinterpret_cast<__gm__ AType*>(a), tiling_.M * tiling_.Ka);
        bGm_.SetGlobalBuffer(reinterpret_cast<__gm__ BType*>(b), tiling_.Kb * tiling_.N);
        cGm_.SetGlobalBuffer(reinterpret_cast<__gm__ CType*>(c), tiling_.M * tiling_.N);
        biasGm_.SetGlobalBuffer(reinterpret_cast<__gm__ BiasType*>(bias), tiling_.N);
        wsGm_.SetGlobalBuffer(reinterpret_cast<__gm__ CType*>(workspace), tiling_.M * tiling_.N);

        int32_t offsetA, offsetB, offsetC, offsetBias;
        CalcOffset(GetBlockIdx(), tiling_, offsetA, offsetB, offsetC, offsetBias);
        aGm_ = aGm_[offsetA];
        bGm_ = bGm_[offsetB];
        cGm_ = cGm_[offsetC];
        biasGm_ = biasGm_[offsetBias];
        wsGm_ = wsGm_[GetBlockIdx() * tiling_.singleCoreM * tiling_.singleCoreN];

        // Queues for matmul output and post-activation
        pipe->InitBuffer(reluInQ_, 1, tiling_.baseM * tiling_.baseN * sizeof(CType));
        pipe->InitBuffer(reluOutQ_, 1, tiling_.baseM * tiling_.baseN * sizeof(CType));
    }

    __aicore__ inline void Process()
    {
        mm_.SetWorkspace(wsGm_);
        mm_.SetTensorA(aGm_);
        mm_.SetTensorB(bGm_);
        mm_.SetBias(biasGm_);
        mm_.template Iterate<false>();

        const uint32_t tilesPerCore =
            tiling_.singleCoreM * tiling_.singleCoreN / (tiling_.baseM * tiling_.baseN);

        for (uint32_t t = 0; t < tilesPerCore; ++t) {
            // Compute GEMM+bias into local
            AscendC::LocalTensor<CType> cLocalIn = reluInQ_.AllocTensor<CType>();
            mm_.template GetTensorC<false>(cLocalIn, false, true);
            reluInQ_.EnQue(cLocalIn);

            // ReLU activation
            cLocalIn = reluInQ_.DeQue<CType>();
            AscendC::LocalTensor<CType> cLocalOut = reluOutQ_.AllocTensor<CType>();
            Maxs(cLocalOut, cLocalIn, static_cast<CType>(0.0f),
                 tiling_.baseM * tiling_.baseN);
            reluOutQ_.EnQue(cLocalOut);
            reluInQ_.FreeTensor(cLocalIn);

            // Copy out to GM
            cLocalOut = reluOutQ_.DeQue<CType>();
            const uint32_t roundM = tiling_.singleCoreM / tiling_.baseM;
            const uint32_t startOffset = (t % roundM) * tiling_.baseM * tiling_.N +
                                         (t / roundM) * tiling_.baseN;

            AscendC::DataCopyParams copyParam = {
                (uint16_t)tiling_.baseM,
                (uint16_t)(tiling_.baseN * sizeof(CType) / AscendC::DEFAULT_C0_SIZE),
                0,
                (uint16_t)((tiling_.N - tiling_.baseN) * sizeof(CType) / AscendC::DEFAULT_C0_SIZE)};
            DataCopy(cGm_[startOffset], cLocalOut, copyParam);
            reluOutQ_.FreeTensor(cLocalOut);
        }
        mm_.End();
    }

private:
    __aicore__ inline void CalcOffset(int32_t blockIdx, const optiling::TCubeTiling& tiling,
                                      int32_t& offsetA, int32_t& offsetB, int32_t& offsetC, int32_t& offsetBias)
    {
        auto Ceiling = [](__aicore__ uint32_t a, __aicore__ uint32_t b) { return (a + b - 1) / b; };
        const uint32_t mSingleBlocks = Ceiling(tiling.M, tiling.singleCoreM);
        const uint32_t mCoreIndx = static_cast<uint32_t>(blockIdx) % mSingleBlocks;
        const uint32_t nCoreIndx = static_cast<uint32_t>(blockIdx) / mSingleBlocks;

        offsetA = mCoreIndx * tiling.Ka * tiling.singleCoreM;
        offsetB = nCoreIndx * tiling.singleCoreN;
        offsetC = mCoreIndx * tiling.N * tiling.singleCoreM + nCoreIndx * tiling.singleCoreN;
        offsetBias = nCoreIndx * tiling.singleCoreN;
    }

private:
    optiling::TCubeTiling tiling_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> reluInQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> reluOutQ_;

public:
    Matmul<MatmulType<AscendC::TPosition::GM, CubeFormat::ND, AType>,
           MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BType>,
           MatmulType<AscendC::TPosition::VECIN, CubeFormat::ND, CType>,
           MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BiasType>>
        mm_;

    AscendC::GlobalTensor<AType> aGm_;
    AscendC::GlobalTensor<BType> bGm_;
    AscendC::GlobalTensor<CType> cGm_;
    AscendC::GlobalTensor<BiasType> biasGm_;
    AscendC::GlobalTensor<CType> wsGm_;
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                                           GM_ADDR __kfc_workspace__ workspace, GM_ADDR tilingGm)
{
    optiling::TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    AscendC::TPipe pipe;
    MatmulReluKernel<half, half, float, float> kernel;
    kernel.Init(x, w, bias, y, workspace, tiling, &pipe);
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), kernel.mm_, &tiling);
    kernel.Process();
}
