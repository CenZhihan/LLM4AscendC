host_operator_src="""
#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// UB budget planning:
//   - Weight buffer (O*C floats) loaded once per block
//   - xLocal: tileM floats
//   - tmpLocal: tileM floats
//   - yLocal: tileO * tileM floats
// Total bytes ~= 4 * (O*C + tileM * (tileO + 2))
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core available UB (conservative)
static constexpr uint32_t MIN_TILE_M = 512;              // minimal spatial tile for efficiency
static constexpr uint32_t ALIGN_TILE_M = 512;            // DMA-friendly alignment
static constexpr uint32_t DEFAULT_TILE_O = 16;           // try to process 16 output channels together
static constexpr uint32_t MAX_BLOCKDIM = 24;
static constexpr uint32_t MIN_BLOCKDIM = 8;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const uint64_t N = static_cast<uint64_t>(x_shape->GetDim(0));
    const uint64_t C = static_cast<uint64_t>(x_shape->GetDim(1));
    const uint64_t H = static_cast<uint64_t>(x_shape->GetDim(2));
    const uint64_t W = static_cast<uint64_t>(x_shape->GetDim(3));

    const uint64_t O = static_cast<uint64_t>(w_shape->GetDim(0));
    const uint64_t Ci = static_cast<uint64_t>(w_shape->GetDim(1));
    const uint64_t Kh = static_cast<uint64_t>(w_shape->GetDim(2));
    const uint64_t Kw = static_cast<uint64_t>(w_shape->GetDim(3));

    if (Ci != C || Kh != 1 || Kw != 1) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const uint64_t posPerBatch64 = H * W;
    const uint32_t posPerBatch = static_cast<uint32_t>(posPerBatch64);

    const uint32_t Ou32 = static_cast<uint32_t>(O);
    const uint32_t Cu32 = static_cast<uint32_t>(C);

    const uint32_t weightBytes = Ou32 * Cu32 * sizeof(float);
    uint32_t availBytes = (UB_BUDGET_BYTES > weightBytes) ? (UB_BUDGET_BYTES - weightBytes) : 0u;

    uint32_t tileO = DEFAULT_TILE_O;
    if (tileO > Ou32) tileO = Ou32;
    if (tileO == 0u) tileO = 1u;

    // Adjust tileO to secure reasonable tileM
    uint32_t tileM = 0u;
    for (; tileO >= 1u; --tileO) {
        uint32_t denom = (tileO + 2u) * sizeof(float);
        if (denom == 0) continue;
        if (availBytes / denom == 0) continue;
        tileM = availBytes / denom;
        tileM = AlignDown(tileM, ALIGN_TILE_M);
        if (tileM == 0u) continue;
        if (tileM > posPerBatch) tileM = AlignDown(posPerBatch, ALIGN_TILE_M);
        if (tileM >= MIN_TILE_M || tileO == 1u) break;
    }
    if (tileO == 0u) tileO = 1u;
    if (tileM == 0u) tileM = (posPerBatch >= MIN_TILE_M) ? MIN_TILE_M : posPerBatch;

    const uint32_t tilesPerBatch = CeilDiv(posPerBatch, tileM);
    uint32_t blockDim = static_cast<uint32_t>(N) * tilesPerBatch;
    if (blockDim < MIN_BLOCKDIM) blockDim = MIN_BLOCKDIM;
    if (blockDim > MAX_BLOCKDIM) blockDim = MAX_BLOCKDIM;
    context->SetBlockDim(blockDim);

    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_c(static_cast<uint32_t>(C));
    tiling.set_h(static_cast<uint32_t>(H));
    tiling.set_w(static_cast<uint32_t>(W));
    tiling.set_o(static_cast<uint32_t>(O));
    tiling.set_tileM(tileM);
    tiling.set_tileO(tileO);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t wdim, uint32_t o,
                                uint32_t tileM, uint32_t tileO)
    {
        n_ = n; c_ = c; h_ = h; w_ = wdim; o_ = o;
        posPerBatch_ = h_ * w_;
        tileM_ = tileM;
        tileO_ = tileO;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * c_ * posPerBatch_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)o_ * c_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * o_ * posPerBatch_);

        // Queues
        // xLocal: tileM
        // tmpLocal: tileM
        // yLocal: tileO * tileM
        // wLocal: o * c
        pipe_.InitBuffer(inQueueX_, 1, tileM_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * sizeof(float));
        pipe_.InitBuffer(outCalcQueue_, 1, tileO_ * tileM_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, o_ * c_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || o_ == 0 || posPerBatch_ == 0 || tileM_ == 0) return;

        // Preload weights (contiguous O*C)
        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, (uint64_t)o_ * c_);
        wQueue_.EnQue(wLocal); // keep allocated in queue during compute
        AscendC::LocalTensor<float> wBuf = wQueue_.DeQue<float>(); // obtain handle

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // We assign blocks across batches and spatial tiles within a batch:
        const uint32_t tilesPerBatch = (posPerBatch_ + tileM_ - 1u) / tileM_;
        const uint32_t totalTiles = n_ * tilesPerBatch;

        // Grid-stride over tiles
        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            const uint32_t nIdx = t / tilesPerBatch;
            const uint32_t tileIdx = t % tilesPerBatch;
            const uint32_t posBase = tileIdx * tileM_;
            uint32_t curM = posPerBatch_ - posBase;
            if (curM > tileM_) curM = tileM_;

            // Process output channels in chunks of tileO_
            for (uint32_t oBase = 0; oBase < o_; oBase += tileO_) {
                const uint32_t curO = (oBase + tileO_ <= o_) ? tileO_ : (o_ - oBase);

                AscendC::LocalTensor<float> yLocal = outCalcQueue_.AllocTensor<float>();
                AscendC::Duplicate(yLocal, 0.0f, curO * curM);

                // Accumulate across input channels
                for (uint32_t ci = 0; ci < c_; ++ci) {
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    const uint64_t xOff = ((uint64_t)nIdx * c_ + ci) * posPerBatch_ + posBase;
                    AscendC::DataCopy(xLocal, xGm_[xOff], curM);
                    inQueueX_.EnQue(xLocal);

                    AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();
                    AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

                    // For each output channel in chunk: y[o,:] += w[o,ci] * x[:]
                    for (uint32_t oi = 0; oi < curO; ++oi) {
                        const float wVal = wBuf[(uint64_t)(oBase + oi) * c_ + ci];
                        AscendC::Muls(tmpLocal, xTile, wVal, curM);
                        AscendC::LocalTensor<float> yRow = yLocal[curM * oi];
                        AscendC::Add(yRow, yRow, tmpLocal, curM);
                    }

                    tmpQueue_.FreeTensor(tmpLocal);
                    inQueueX_.FreeTensor(xTile);
                }

                // Write back curO rows
                for (uint32_t oi = 0; oi < curO; ++oi) {
                    AscendC::LocalTensor<float> yRow = yLocal[curM * oi];
                    const uint64_t yOff = ((uint64_t)nIdx * o_ + (oBase + oi)) * posPerBatch_ + posBase;
                    AscendC::DataCopy(yGm_[yOff], yRow, curM);
                }

                outCalcQueue_.FreeTensor(yLocal);
            }
        }

        // release weight buffer
        wQueue_.FreeTensor(wBuf);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> outCalcQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0}, o_{0};
    uint32_t posPerBatch_{0};
    uint32_t tileM_{0}, tileO_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                                               GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tilingData, tilingGm);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tilingData.n, tilingData.c, tilingData.h, tilingData.w, tilingData.o,
            tilingData.tileM, tilingData.tileO);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv_pointwise_2d_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    // Output shape: (N, O, H, W) with O = weight.size(0)
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, \"Expect 4D x and weight\");
    TORCH_CHECK(x.size(1) == weight.size(1), \"C_in mismatch between x and weight\");
    TORCH_CHECK(weight.size(2) == 1 && weight.size(3) == 1, \"Only 1x1 kernel supported\");

    auto N = x.size(0);
    auto O = weight.size(0);
    auto H = x.size(2);
    auto W = x.size(3);
    at::Tensor y = at::empty({N, O, H, W}, x.options());

    EXEC_NPU_CMD(aclnnConvPointwise2dCustom, x, weight, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv_pointwise_2d_custom\", &conv_pointwise_2d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"conv_pointwise_2d_custom\", &conv_pointwise_2d_impl_npu, \"Pointwise 1x1 conv (ConvPointwise2dCustom on NPU)\");
