#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

// UB budget for post-activation pass (ReLU):
// We'll use inQueueY + outQueueY = 2 * tile elements => bytes = 8 * tile for fp32
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core share (conservative)
static constexpr uint32_t ALIGN_ELEMS = 2048;            // DMA/alignment friendly
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // avoid very large per-iter tiles

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;

    // Shapes:
    // x: [M, K]
    // w: [N, K] (PyTorch linear uses weight [out_features, in_features])
    // bias: [N]
    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t totalLength = M * N;

    // Tile length for post-activation ReLU pass on y (contiguous y flatten)
    uint32_t maxTile = UB_BUDGET_BYTES / 8;  // fp32 -> 4B, 2 * tile * 4
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // Choose blockDim conservatively based on M rows so GEMM split per-core is reasonable.
    // Keep in [8, 24] range for 910B-like devices.
    uint32_t blockDim = (M < 8u) ? 8u : (M > 24u ? 24u : M);

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

// Fused kernel: GEMM (x @ w^T + bias) followed by ReLU activation.
// Shapes:
//   x: [M, K]
//   w: [N, K]   (PyTorch Linear uses weight [out_features, in_features])
//   bias: [N]
//   y: [M, N]
//
// Strategy:
// 1) Use a Matmul library object to compute y = x * w^T + bias in GM.
//    We set isTransA = false, isTransB = true to accept w as [N, K].
// 2) Post-process ReLU in tiles using vector cores (in-place on y).
//
// Tiling data (from host):
//   M, N, K, totalLength (M*N), tileLength (for ReLU), blockDim
template <typename AType, typename BType, typename CType, typename BiasType,
          bool isTransA, bool isTransB>
class KernelGemmAddReluCustomT {
public:
    __aicore__ inline KernelGemmAddReluCustomT() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t totalLength, uint32_t tileLength)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        aGm_.SetGlobalBuffer((__gm__ AType*)x, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ BType*)w, N_ * K_);   // [N, K]
        cGm_.SetGlobalBuffer((__gm__ CType*)y, M_ * N_);
        biasGm_.SetGlobalBuffer((__gm__ BiasType*)bias, N_);

        // UB queues for post-activation pass
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(CType));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(CType));
    }

    __aicore__ inline void Process()
    {
        // Phase 1: GEMM + Bias
        // Split rows (M) across blocks for GEMM phase
        const int32_t blockIdx = AscendC::GetBlockIdx();
        const int32_t blockNum = AscendC::GetBlockNum();

        // Rows per core split
        const uint32_t rowsPerCore = (M_ + blockNum - 1) / blockNum;
        const uint32_t mStart = rowsPerCore * blockIdx;
        const uint32_t mLen = (mStart >= M_) ? 0u : AscendC::Min(rowsPerCore, M_ - mStart);

        if (mLen > 0) {
            // Construct Matmul object for this core's slice
            AscendC::GlobalTensor<AType> aLocal = aGm_[mStart * K_];
            AscendC::GlobalTensor<CType> cLocal = cGm_[mStart * N_];

            // Configure cube matmul: A:[mLen, K], B:[K, N] via isTransB=true, C:[mLen, N], Bias:[N]
            AscendC::Matmul<
                AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, AType, isTransA, LayoutMode::NORMAL>,
                AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BType, isTransB, LayoutMode::NORMAL>,
                AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, CType, false, LayoutMode::NORMAL>,
                AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BiasType>>
                matmulObj;

            // Minimal tiling descriptor
            AscendC::TCubeTiling cubeTiling;
            cubeTiling.M = static_cast<int32_t>(mLen);
            cubeTiling.N = static_cast<int32_t>(N_);
            cubeTiling.Ka = static_cast<int32_t>(K_);
            cubeTiling.Kb = static_cast<int32_t>(K_);
            cubeTiling.BatchNum = 1;

            // Prepare tensors for matmul
            // Note: bGm_ and biasGm_ share across cores; aLocal/cLocal are per-core slices.
            matmulObj.SetTensor(aLocal, bGm_, cLocal, biasGm_, cubeTiling);
            matmulObj.Process(&pipe_);
        }

        // Phase 2: ReLU in-place on y (C)
        if (totalLen_ == 0 || tileLen_ == 0) return;

        // Grid-stride distribution across total output elements for ReLU
        const uint32_t uBlockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t uBlockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = uBlockNum * tileLen_;
        uint32_t base = uBlockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Prefetch first tile
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyInY(base, curLen);

        while (true) {
            // Prefetch next tile
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyInY(nextBase, nextLen);
            }

            // ReLU: y = max(y, 0)
            ComputeRelu(curLen);
            CopyOutY(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyInY(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<CType> yLocal = inQueueY_.AllocTensor<CType>();
        AscendC::DataCopy(yLocal, cGm_[gmOffset], len);
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void ComputeRelu(uint32_t len)
    {
        AscendC::LocalTensor<CType> yIn = inQueueY_.DeQue<CType>();
        AscendC::LocalTensor<CType> yOut = outQueueY_.AllocTensor<CType>();

        AscendC::Maxs(yOut, yIn, static_cast<CType>(0.0f), len);

        outQueueY_.EnQue<CType>(yOut);
        inQueueY_.FreeTensor(yIn);
    }

    __aicore__ inline void CopyOutY(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<CType> yLocal = outQueueY_.DeQue<CType>();
        AscendC::DataCopy(cGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<AType> aGm_;
    AscendC::GlobalTensor<BType> bGm_;
    AscendC::GlobalTensor<CType> cGm_;
    AscendC::GlobalTensor<BiasType> biasGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

using KernelImpl = KernelGemmAddReluCustomT<float, float, float, float, false, true>;

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                                           GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelImpl op;
    op.Init(x, w, bias, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
