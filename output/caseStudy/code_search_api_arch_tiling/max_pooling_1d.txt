#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <cstdint>

namespace optiling {

// UB model for this kernel (float32):
//   inQueueX: 2 * ((tileOut - 1) * stride + kernelSize)
//   outQueueY: 2 * tileOut
//   tmpQueue: 1 * tileOut
// Total elements = 2*((tileOut-1)*stride + kernelSize) + 3*tileOut
// Bytes = 4 * [2*(tileOut*stride + kernelSize) + 3*tileOut] = 4*(tileOut*(2*stride+3) + 2*kernelSize)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // favor larger DMA bursts
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape &orig = context->GetInputShape(0)->GetOriginShape();
    const uint32_t dimNum = static_cast<uint32_t>(orig.GetDimNum());
    uint32_t n = 1, c = 1, inLength = 1;
    if (dimNum >= 1) inLength = static_cast<uint32_t>(orig.GetDim(dimNum - 1));
    if (dimNum >= 2) c        = static_cast<uint32_t>(orig.GetDim(dimNum - 2));
    if (dimNum >= 3) n        = static_cast<uint32_t>(orig.GetDim(dimNum - 3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernelSizePtr    = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stridePtr        = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *paddingPtr       = attrs->GetAttrPointer<int64_t>(2);
    const bool    *returnIdxPtr     = attrs->GetAttrPointer<bool>(3);

    uint32_t kernelSize = static_cast<uint32_t>(*kernelSizePtr);
    uint32_t stride     = (stridePtr == nullptr || *stridePtr <= 0) ? kernelSize
                                                                    : static_cast<uint32_t>(*stridePtr);
    uint32_t padding    = (paddingPtr == nullptr || *paddingPtr < 0) ? 0u
                                                                     : static_cast<uint32_t>(*paddingPtr);
    (void)returnIdxPtr; // indices are not produced in this custom op

    const uint64_t inPadded = static_cast<uint64_t>(inLength) + 2ull * static_cast<uint64_t>(padding);
    uint32_t outLength = 0u;
    if (inPadded >= kernelSize) {
        outLength = static_cast<uint32_t>((inPadded - kernelSize) / stride + 1ull);
    } else {
        outLength = 0u;
    }

    // tileOut from UB budget: 4*(tileOut*(2*stride+3) + 2*kernelSize) <= UB_BUDGET_BYTES
    uint32_t maxTileOut = 1u;
    if (2u * stride + 3u > 0u) {
        const uint32_t rhs = (UB_BUDGET_BYTES / 4u);
        if (rhs > 2u * kernelSize) {
            maxTileOut = (rhs - 2u * kernelSize) / (2u * stride + 3u);
        } else {
            maxTileOut = 1u;
        }
    }
    maxTileOut = std::max<uint32_t>(AlignDown(maxTileOut, ALIGN_ELEMS), 1u);
    uint32_t tileOut = std::min<uint32_t>(outLength > 0 ? outLength : 1u, maxTileOut);
    if (tileOut == 0) tileOut = 1u;

    // blocks across rows (n * c)
    const uint32_t rows = n * c;
    uint32_t blockDim = rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;
    if (blockDim > rows) blockDim = rows == 0 ? 1u : rows;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inLength(inLength);
    tiling.set_outLength(outLength);
    tiling.set_tileOut(tileOut);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernelSizePtr    = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stridePtr        = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *paddingPtr       = attrs->GetAttrPointer<int64_t>(2);

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape &orig = x_shape->GetOriginShape();
    const uint32_t dimNum = static_cast<uint32_t>(orig.GetDimNum());

    int64_t n = 1, c = 1, inLength = 1;
    if (dimNum >= 1) inLength = orig.GetDim(dimNum - 1);
    if (dimNum >= 2) c        = orig.GetDim(dimNum - 2);
    if (dimNum >= 3) n        = orig.GetDim(dimNum - 3);

    const int64_t kernelSize = (kernelSizePtr == nullptr) ? 1 : *kernelSizePtr;
    const int64_t stride     = (stridePtr == nullptr || *stridePtr <= 0) ? kernelSize : *stridePtr;
    const int64_t padding    = (paddingPtr == nullptr || *paddingPtr < 0) ? 0 : *paddingPtr;

    const int64_t inPadded = inLength + 2 * padding;
    int64_t outLength = 0;
    if (inPadded >= kernelSize) {
        outLength = (inPadded - kernelSize) / stride + 1;
    } else {
        outLength = 0;
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    gert::Shape y(orig);
    if (dimNum >= 1) y.SetDim(dimNum - 1, outLength);
    if (dimNum < 2) y.SetDimNum(1);
    *y_shape = y;
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include <cstdint>

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPool1d {
public:
    __aicore__ inline KernelMaxPool1d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t inLen, uint32_t outLen,
                                uint32_t tileOut,
                                uint32_t ksize, uint32_t stride, uint32_t padding)
    {
        n_ = n;
        c_ = c;
        inLen_ = inLen;
        outLen_ = outLen;
        tileOut_ = tileOut;
        ksize_ = ksize;
        stride_ = stride;
        pad_ = padding;

        totalInElems_ = n_ * c_ * inLen_;
        totalOutElems_ = n_ * c_ * outLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutElems_);

        // buffers: input chunk = (tileOut -1)*stride + ksize, output tile = tileOut
        chunkInMax_ = (tileOut_ - 1u) * stride_ + ksize_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, chunkInMax_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || outLen_ == 0) return;

        const uint32_t rows = n_ * c_;
        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows; row += blockNum) {
            const uint32_t gmRowInBase  = row * inLen_;
            const uint32_t gmRowOutBase = row * outLen_;

            for (uint32_t bo = 0; bo < outLen_; bo += tileOut_) {
                const uint32_t curOut = (bo + tileOut_ <= outLen_) ? tileOut_ : (outLen_ - bo);
                const uint32_t chunkLen = (curOut - 1u) * stride_ + ksize_;

                CopyIn(gmRowInBase, bo, curOut, chunkLen);
                Compute(curOut, chunkLen);
                CopyOut(gmRowOutBase, bo, curOut);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmRowInBase, uint32_t bo, uint32_t curOut, uint32_t chunkLen)
    {
        LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        // zero-fill
        Dup(xLocal, 0.0f, chunkLen);

        // effective padded space start index
        int64_t ip0 = static_cast<int64_t>(bo) * static_cast<int64_t>(stride_) - static_cast<int64_t>(pad_);
        uint32_t gmCopyStart = 0;
        uint32_t gmCopyLen = 0;
        uint32_t localStart = 0;

        if (chunkLen > 0) {
            if (ip0 < 0) {
                localStart = static_cast<uint32_t>(-ip0);
                gmCopyStart = 0;
            } else {
                localStart = 0;
                gmCopyStart = static_cast<uint32_t>(ip0);
            }
            int64_t ipEnd = ip0 + static_cast<int64_t>(chunkLen);
            int64_t gmEnd = ipEnd > static_cast<int64_t>(inLen_) ? static_cast<int64_t>(inLen_) : ipEnd;
            int64_t gmBegin = ip0 < 0 ? 0 : ip0;
            gmCopyLen = (gmEnd > gmBegin) ? static_cast<uint32_t>(gmEnd - gmBegin) : 0u;
        }

        if (gmCopyLen > 0) {
            DataCopy(xLocal[localStart], xGm_[gmRowInBase + gmCopyStart], gmCopyLen);
        }

        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t curOut, uint32_t chunkLen)
    {
        LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        if (stride_ == 1) {
            // Vector-friendly path
            DataCopy(yLocal, xLocal, curOut);
            if (ksize_ > 1) {
                LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                for (uint32_t t = 1; t < ksize_; ++t) {
                    DataCopy(tLocal, xLocal[t], curOut);
                    Max(yLocal, yLocal, tLocal, curOut);
                }
                tmpQueue_.FreeTensor(tLocal);
            }
        } else {
            // Generic path (handle stride > 1)
            // Build maxima per output by scanning ksize values with stride
            for (uint32_t j = 0; j < curOut; ++j) {
                // initial
                float m = xLocal.GetValue(j * stride_);
                for (uint32_t t = 1; t < ksize_; ++t) {
                    float v = xLocal.GetValue(j * stride_ + t);
                    m = (v > m) ? v : m;
                }
                yLocal.SetValue(j, m);
            }
        }

        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmRowOutBase, uint32_t bo, uint32_t curOut)
    {
        LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        DataCopy(yGm_[gmRowOutBase + bo], yLocal, curOut);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0};
    uint32_t inLen_{0}, outLen_{0}, tileOut_{0};
    uint32_t ksize_{1}, stride_{1}, pad_{0};
    uint32_t totalInElems_{0}, totalOutElems_{0};
    uint32_t chunkInMax_{0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    REGISTER_TILING_DEFAULT(MaxPooling1dCustomTilingData);
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPool1d op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.inLength, tiling_data.outLength,
            tiling_data.tileOut,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t calc_out_len(int64_t in_len, int64_t kernel_size, int64_t stride, int64_t padding) {
    const int64_t s = (stride <= 0) ? kernel_size : stride;
    const int64_t in_pad = in_len + 2 * padding;
    if (in_pad < kernel_size) return 0;
    return (in_pad - kernel_size) / s + 1;
}

at::Tensor max_pooling_1d_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding, bool return_indices) {
    TORCH_CHECK(self.dim() == 3, "max_pooling_1d_custom expects input of shape (N, C, L)");
    const int64_t N = self.size(0);
    const int64_t C = self.size(1);
    const int64_t L = self.size(2);
    const int64_t S = (stride <= 0) ? kernel_size : stride;
    const int64_t Lout = calc_out_len(L, kernel_size, S, padding);
    at::Tensor result = at::empty({N, C, Lout}, self.options());
    EXEC_NPU_CMD(aclnnMaxPooling1dCustom, self, kernel_size, S, padding, return_indices, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_1d_custom", &max_pooling_1d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_1d_custom", &max_pooling_1d_impl_npu, "MaxPooling1d (MaxPooling1dCustom on NPU)");
