host_operator_src="""

#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (a == 0) ? x : (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return (a == 0) ? x : ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// UB model per core:
//   B_tile: tileK * tileN
//   A_row:  tileK
//   C_row:  tileN
//   Tmp:    tileN
// Total floats = tileK*tileN + tileK + 2*tileN
// Bytes = 4 * (tileK*tileN + tileK + 2*tileN)
static constexpr uint32_t UB_BUDGET_BYTES = 192u * 1024u; // conservative
static constexpr uint32_t DMA_ALIGN = 128u;               // favor large alignment
static constexpr uint32_t MAX_BLOCKDIM = 24u;             // typical 910B cubes available for aicore schedule
static constexpr uint32_t MAX_TILEK = 256u;               // clamp tileK for stability
static constexpr uint32_t MIN_TILEN = 64u;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    // Input 0: A matrix [M, K]
    // Input 1: B matrix [K, N]
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);

    if (aShape->GetDimNum() != 2 || bShape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(aShape->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(aShape->GetDim(1));
    const uint32_t Kb = static_cast<uint32_t>(bShape->GetDim(0));
    const uint32_t N = static_cast<uint32_t>(bShape->GetDim(1));
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    // Choose tileK
    uint32_t tileK = (K < MAX_TILEK) ? K : MAX_TILEK;
    // Solve tileN from UB budget: 4*(tileK*tileN + tileK + 2*tileN) <= UB_BUDGET_BYTES
    // => tileN * (tileK + 2) <= UB_BUDGET_BYTES/4 - tileK
    uint32_t rhs = (UB_BUDGET_BYTES / 4u);
    if (rhs <= tileK) rhs = tileK + 1u;
    uint32_t tileN = (rhs - tileK) / (tileK + 2u);
    if (tileN == 0) tileN = MIN_TILEN;
    if (tileN > N) tileN = N;
    // alignment encourage bigger DMA
    tileN = AlignDown(tileN, DMA_ALIGN);
    if (tileN == 0) {
        tileN = (N < DMA_ALIGN) ? N : DMA_ALIGN;
    }

    // Decide blockDim based on rows M
    uint32_t blockDim = CeilDiv(M, 1024u); // heuristic: around 1K rows per core
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCKDIM) blockDim = MAX_BLOCKDIM;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a = context->GetInputShape(0);
    const gert::Shape *b = context->GetInputShape(1);
    if (a->GetDimNum() != 2 || b->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    gert::Shape *y = context->GetOutputShape(0);
    y->SetDimNum(2);
    y->SetDim(0, a->GetDim(0)); // M
    y->SetDim(1, b->GetDim(1)); // N
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    ge::DataType a_dtype = context->GetInputDataType(0);
    ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != ge::DT_FLOAT || b_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BBUF_NUM = 1;
constexpr int32_t ARBUF_NUM = 1;
constexpr int32_t CBUF_NUM = 1;
constexpr int32_t TBUF_NUM = 1;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tileN_ = tileN; tileK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)a, (uint64_t)M_ * (uint64_t)K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)K_ * (uint64_t)N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * (uint64_t)N_);

        pipe_.InitBuffer(bTileQ_, BBUF_NUM, tileN_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(aRowQ_, ARBUF_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(cRowQ_, CBUF_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_,  TBUF_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            uint64_t aRowBase = (uint64_t)m * (uint64_t)K_;
            uint64_t yRowBase = (uint64_t)m * (uint64_t)N_;

            for (uint32_t nBase = 0; nBase < N_; nBase += tileN_) {
                uint32_t curN = N_ - nBase;
                if (curN > tileN_) curN = tileN_;

                AscendC::LocalTensor<float> cRow = cRowQ_.AllocTensor<float>();
                AscendC::Dup(cRow, 0.0f, curN);

                AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();

                for (uint32_t kBase = 0; kBase < K_; kBase += tileK_) {
                    uint32_t curK = K_ - kBase;
                    if (curK > tileK_) curK = tileK_;

                    // Load B_tile [curK x curN] row by row
                    AscendC::LocalTensor<float> bTile = bTileQ_.AllocTensor<float>();
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        uint64_t bOff = ((uint64_t)(kBase + kk) * (uint64_t)N_) + (uint64_t)nBase;
                        AscendC::DataCopyPad(bTile[kk * tileN_], bGm_[bOff], curN);
                    }

                    // Load A_row [curK]
                    AscendC::LocalTensor<float> aRow = aRowQ_.AllocTensor<float>();
                    AscendC::DataCopyPad(aRow, aGm_[aRowBase + (uint64_t)kBase], curK);

                    // Accumulate cRow += sum_{kk} aRow[kk] * bTile[kk,:]
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        float aVal = aRow.GetValue(kk);
                        AscendC::Muls(tmp, bTile[kk * tileN_], aVal, curN);
                        AscendC::Add(cRow, cRow, tmp, curN);
                    }

                    aRowQ_.FreeTensor(aRow);
                    bTileQ_.FreeTensor(bTile);
                }

                AscendC::DataCopyPad(yGm_[yRowBase + (uint64_t)nBase], cRow, curN);
                tmpQ_.FreeTensor(tmp);
                cRowQ_.FreeTensor(cRow);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BBUF_NUM> bTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, ARBUF_NUM> aRowQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, CBUF_NUM> cRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TBUF_NUM> tmpQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileN_{0}, tileK_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR a, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(a, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileN, tiling_data.tileK);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_contiguous_if_needed(const at::Tensor& t) {
    return t.is_contiguous() ? t : t.contiguous();
}

at::Tensor conv_pointwise_2d_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    TORCH_CHECK(x.dim() == 4, "x must be NCHW 4D");
    TORCH_CHECK(weight.dim() == 4, "weight must be OIHW 4D with 1x1 kernel");
    TORCH_CHECK(weight.size(2) == 1 && weight.size(3) == 1, "only 1x1 kernels supported");
    TORCH_CHECK(x.dtype() == at::kFloat && weight.dtype() == at::kFloat, "only float supported");

    auto N = x.size(0);
    auto C_in = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);
    auto C_out = weight.size(0);
    TORCH_CHECK(weight.size(1) == C_in, "weight C_in mismatch");

    // Transform to matmul form: A [M,K], B [K,N]
    at::Tensor x_nhwc = x.permute({0, 2, 3, 1}).contiguous();  // [N,H,W,C]
    int64_t M = N * H * W;
    int64_t K = C_in;
    at::Tensor a_mat = x_nhwc.view({M, K});

    at::Tensor w_t = weight.view({C_out, C_in}).transpose(0, 1).contiguous(); // [K, N]
    at::Tensor y_mat = at::empty({M, C_out}, x.options());

    EXEC_NPU_CMD(aclnnConvPointwise2dCustom, a_mat, w_t, y_mat);

    at::Tensor y_nhwc = y_mat.view({N, H, W, C_out});
    at::Tensor y_nchw = y_nhwc.permute({0, 3, 1, 2}).contiguous();
    return y_nchw;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_pointwise_2d_custom", &conv_pointwise_2d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_pointwise_2d_custom", &conv_pointwise_2d_impl_npu, "1x1 Conv2D (ConvPointwise2dCustom on NPU)");
}
