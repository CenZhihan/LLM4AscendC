host_operator_src="""

#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"
#include "tiling/platform/platform_ascendc.h"
#include "kernel_tiling/kernel_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Shapes:
    //   x: [M, K]
    //   w: [N, K]   (PyTorch weight layout), need B^T in GEMM => isTransB = true
    //   bias: [N]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Prefer origin shapes if available
    const gert::Shape *x_origin = &x_shape->GetOriginShape();
    const gert::Shape *w_origin = &w_shape->GetOriginShape();

    // Fallbacks in case of unknown rank handled by GetShapeSize, but GEMM expects rank-2.
    int64_t M = (x_origin->GetRank() >= 1) ? x_origin->GetDim(0) : static_cast<int64_t>(1);
    int64_t K = (x_origin->GetRank() >= 2) ? x_origin->GetDim(1) :
                ((w_origin->GetRank() >= 2) ? w_origin->GetDim(1) : static_cast<int64_t>(1));
    int64_t N = (w_origin->GetRank() >= 1) ? w_origin->GetDim(0) : static_cast<int64_t>(1);

    if (M <= 0 || N <= 0 || K <= 0) {
        return ge::GRAPH_FAILED;
    }

    // Build TCubeTiling with Matmul tiling API
    auto ascendcPlatform = platform_ascendc::PlatformAscendCManager::GetInstance();

    optiling::TCubeTiling tilingData;
    matmul_tiling::MultiCoreMatmulTiling tilingApi(*ascendcPlatform);

    // Auto-select core number
    tilingApi.SetDim(-1);

    // A: x, GM/ND/float, not transposed
    tilingApi.SetAType(matmul_tiling::TPosition::GM,
                       matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT,
                       false);

    // B: w, GM/ND/float, to be transposed inside Matmul (since w is [N,K])
    tilingApi.SetBType(matmul_tiling::TPosition::GM,
                       matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT,
                       true);

    // C: y, GM/ND/float
    tilingApi.SetCType(matmul_tiling::TPosition::GM,
                       matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);

    // Bias: GM/ND/float
    tilingApi.SetBiasType(matmul_tiling::TPosition::GM,
                          matmul_tiling::CubeFormat::ND,
                          matmul_tiling::DataType::DT_FLOAT);

    tilingApi.SetOrgShape(static_cast<int>(M), static_cast<int>(N), static_cast<int>(K));
    tilingApi.SetShape(static_cast<int>(M), static_cast<int>(N), static_cast<int>(K));
    tilingApi.SetBias(true);
    tilingApi.SetTraverse(matmul_tiling::MatrixTraverse::FIRSTM);
    tilingApi.SetBufferSpace(-1, -1, -1);

    int64_t res = tilingApi.GetTiling(tilingData);
    // Ensure step settings
    tilingData.set_stepM(1);
    tilingData.set_stepN(1);

    if (res == -1) {
        return ge::GRAPH_FAILED;
    }

    // Write binary tiling blob to raw buffer
    uint32_t tcubeTilingSize = tilingData.GetDataSize();
    tilingData.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tcubeTilingSize);

    // Conservative block dim; runtime may choose to override depending on scheduler
    context->SetBlockDim(1u);

    // No additional workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // y has shape [M, N] like linear output
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const gert::Shape *x_origin = &x_shape->GetOriginShape();
    const gert::Shape *w_origin = &w_shape->GetOriginShape();

    gert::Shape *y_shape = context->GetOutputShape(0);

    if (x_origin->GetRank() == 2 && w_origin->GetRank() == 2) {
        // x: [M,K], w: [N,K] -> y: [M,N]
        std::vector<int64_t> out_dims(2);
        out_dims[0] = x_origin->GetDim(0);
        out_dims[1] = w_origin->GetDim(0);
        y_shape->SetOriginShape(out_dims);
        y_shape->SetStorageShape(out_dims);
    } else {
        // Fallback to x shape if ranks unknown
        *y_shape = *x_shape;
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Expect float32 inputs; propagate dtype
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"
#include "tiling/tiling_api.h"
#include "kernel_tiling/kernel_tiling.h"

using namespace matmul;

__aicore__ inline uint32_t CeilingU32(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

__aicore__ inline void CopyTiling(optiling::TCubeTiling* tiling, GM_ADDR tilingGM)
{
    uint32_t* ptr = reinterpret_cast<uint32_t*>(tiling);
    auto tiling32 = reinterpret_cast<__gm__ uint32_t*>(tilingGM);
    for (uint32_t i = 0; i < sizeof(optiling::TCubeTiling) / sizeof(uint32_t); ++i, ++ptr) {
        *ptr = *(tiling32 + i);
    }
}

template <typename AType, typename BType, typename CType, typename BiasType>
class MatmulReluKernel {
public:
    __aicore__ inline MatmulReluKernel() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR bias, GM_ADDR c, GM_ADDR workspace,
                                const optiling::TCubeTiling& tiling, AscendC::TPipe* pipe);
    __aicore__ inline void Process();

private:
    __aicore__ inline void MatmulCompute();
    __aicore__ inline void ReluCompute(uint32_t partIdx);
    __aicore__ inline void CopyOut(uint32_t partIdx);
    __aicore__ inline void CalcOffset(int32_t blockIdx, const optiling::TCubeTiling& tiling,
                                      int32_t& offsetA, int32_t& offsetB, int32_t& offsetC, int32_t& offsetBias);

private:
    // Matmul object: A in GM, B in GM, C in VECIN (local), Bias in GM
    Matmul<MatmulType<AscendC::TPosition::GM, CubeFormat::ND, AType>,
           MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BType>,
           MatmulType<AscendC::TPosition::VECIN, CubeFormat::ND, CType>,
           MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BiasType>>
        matmulObj_;

    AscendC::GlobalTensor<AType> aGlobal_;
    AscendC::GlobalTensor<BType> bGlobal_;
    AscendC::GlobalTensor<CType> cGlobal_;
    AscendC::GlobalTensor<BiasType> biasGlobal_;
    AscendC::GlobalTensor<CType> workspaceGlobal_;

    AscendC::LocalTensor<CType> reluInLocal_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> reluInQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> reluOutQueue_;

    optiling::TCubeTiling tiling_;
    uint32_t splitRowNums_ {4};
    uint32_t splitRowSize_ {0};
};

template <typename AType, typename BType, typename CType, typename BiasType>
__aicore__ inline void
MatmulReluKernel<AType, BType, CType, BiasType>::Init(GM_ADDR a, GM_ADDR b, GM_ADDR bias, GM_ADDR c, GM_ADDR workspace,
                                                      const optiling::TCubeTiling& tiling, AscendC::TPipe* pipe)
{
    tiling_ = tiling;
    splitRowSize_ = tiling_.baseM / splitRowNums_;

    aGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ AType*>(a), tiling_.M * tiling_.Ka);
    bGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ BType*>(b), tiling_.Kb * tiling_.N);
    cGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ CType*>(c), tiling_.M * tiling_.N);
    biasGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ BiasType*>(bias), tiling_.N);
    workspaceGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ CType*>(workspace), tiling_.M * tiling_.N);

    int32_t offsetA, offsetB, offsetC, offsetBias;
    CalcOffset(AscendC::GetBlockIdx(), tiling_, offsetA, offsetB, offsetC, offsetBias);

    aGlobal_ = aGlobal_[offsetA];
    bGlobal_ = bGlobal_[offsetB];
    cGlobal_ = cGlobal_[offsetC];
    biasGlobal_ = biasGlobal_[offsetBias];
    workspaceGlobal_ = workspaceGlobal_[AscendC::GetBlockIdx() * tiling_.singleCoreM * tiling_.singleCoreN];

    pipe->InitBuffer(reluInQueue_, 1, tiling_.baseM * tiling_.baseN * sizeof(CType));
    pipe->InitBuffer(reluOutQueue_, 1, splitRowSize_ * tiling_.baseN * sizeof(CType));
}

template <typename AType, typename BType, typename CType, typename BiasType>
__aicore__ inline void MatmulReluKernel<AType, BType, CType, BiasType>::Process()
{
    matmulObj_.SetWorkspace(workspaceGlobal_);
    matmulObj_.SetTensorA(aGlobal_);
    matmulObj_.SetTensorB(bGlobal_);
    matmulObj_.SetBias(biasGlobal_);
    matmulObj_.template Iterate<false>();

    const uint32_t iterNum = tiling_.singleCoreM * tiling_.singleCoreN / (tiling_.baseM * tiling_.baseN);
    for (uint32_t i = 0; i < iterNum; ++i) {
        MatmulCompute();
        reluInLocal_ = reluInQueue_.DeQue<CType>(); // wait for matmul compute finished
        for (uint32_t j = 0; j < splitRowNums_; ++j) {
            ReluCompute(j);
            CopyOut(i * splitRowNums_ + j);
        }
        reluInQueue_.FreeTensor(reluInLocal_);
    }
    matmulObj_.End();
}

template <typename AType, typename BType, typename CType, typename BiasType>
__aicore__ inline void MatmulReluKernel<AType, BType, CType, BiasType>::MatmulCompute()
{
    reluInLocal_ = reluInQueue_.AllocTensor<CType>();
    matmulObj_.template GetTensorC<false>(reluInLocal_, false, true);
    reluInQueue_.EnQue(reluInLocal_);
}

template <typename AType, typename BType, typename CType, typename BiasType>
__aicore__ inline void MatmulReluKernel<AType, BType, CType, BiasType>::ReluCompute(uint32_t partIdx)
{
    auto reluOutLocal = reluOutQueue_.AllocTensor<CType>();
    // y = max(x, 0)
    AscendC::Maxs(reluOutLocal, reluInLocal_[partIdx * splitRowSize_ * tiling_.baseN], static_cast<CType>(0.0f),
                  splitRowSize_ * tiling_.baseN);
    reluOutQueue_.EnQue(reluOutLocal);
}

template <typename AType, typename BType, typename CType, typename BiasType>
__aicore__ inline void MatmulReluKernel<AType, BType, CType, BiasType>::CopyOut(uint32_t count)
{
    auto reluOutLocal = reluOutQueue_.DeQue<CType>();
    const uint32_t roundM = tiling_.singleCoreM / splitRowSize_;
    const uint32_t roundN = tiling_.singleCoreN / tiling_.baseN;
    uint32_t startOffset = (count % roundM) * splitRowSize_ * tiling_.N + (count / roundM) * tiling_.baseN;
    AscendC::DataCopyParams copyParam = {
        (uint16_t)splitRowSize_,
        (uint16_t)(tiling_.baseN * sizeof(CType) / AscendC::DEFAULT_C0_SIZE),
        0,
        (uint16_t)((tiling_.N - tiling_.baseN) * sizeof(CType) / AscendC::DEFAULT_C0_SIZE)
    };
    AscendC::DataCopy(cGlobal_[startOffset], reluOutLocal, copyParam);
    reluOutQueue_.FreeTensor(reluOutLocal);
}

template <typename AType, typename BType, typename CType, typename BiasType>
__aicore__ inline void
MatmulReluKernel<AType, BType, CType, BiasType>::CalcOffset(int32_t blockIdx, const optiling::TCubeTiling& tiling,
                                                            int32_t& offsetA, int32_t& offsetB,
                                                            int32_t& offsetC, int32_t& offsetBias)
{
    auto mSingleBlocks = CeilingU32(tiling.M, tiling.singleCoreM);
    auto mCoreIdx = static_cast<uint32_t>(blockIdx) % mSingleBlocks;
    auto nCoreIdx = static_cast<uint32_t>(blockIdx) / mSingleBlocks;

    offsetA = mCoreIdx * tiling.Ka * tiling.singleCoreM;
    offsetB = nCoreIdx * tiling.singleCoreN;
    offsetC = mCoreIdx * tiling.N * tiling.singleCoreM + nCoreIdx * tiling.singleCoreN;
    offsetBias = nCoreIdx * tiling.singleCoreN;
}

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                                           GM_ADDR __kfc_workspace__ workspace, GM_ADDR tilingGm)
{
    AscendC::TPipe pipe;
    optiling::TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    MatmulReluKernel<float, float, float, float> kernel;
    kernel.Init(x, w, bias, y, workspace, tiling, &pipe);
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), kernel.matmulObj_, &kernel.tiling_);
    kernel.Process();
}
