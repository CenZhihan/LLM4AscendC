host_operator_src="""
#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : ((a + b - 1u) / b);
}
} // namespace

namespace optiling {

// UB budgeting (float32):
// Buffers in UB per core:
//   X_tile:  tileK * tileM
//   W_tile:  tileCo * tileK
//   Y_tile:  tileCo * tileM
//   tmp:     tileM
// Total elems = tileK*tileM + tileCo*tileK + tileCo*tileM + tileM
// Total bytes = 4 * Total elems
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
static constexpr uint32_t MIN_ALIGN_ELEMS = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N    = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t CIn  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H    = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W    = static_cast<uint32_t>(x_shape->GetDim(3));
    const uint32_t COut = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Kdim = static_cast<uint32_t>(w_shape->GetDim(1));
    if (Kdim != CIn) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t HW = H * W;

    // Initial candidates
    uint32_t tileK  = (CIn >= 64u) ? 64u : AlignDown(CIn, MIN_ALIGN_ELEMS);
    if (tileK == 0u) tileK = CIn;
    uint32_t tileCo = (COut >= 64u) ? 64u : AlignDown(COut, MIN_ALIGN_ELEMS);
    if (tileCo == 0u) tileCo = COut;

    const uint32_t ubElems = UB_BUDGET_BYTES / 4u;

    auto solveTileM = [&](uint32_t tCo, uint32_t tK) -> uint32_t {
        if (tCo == 0u || tK == 0u) return 0u;
        if (ubElems <= tCo * tK) return 0u;
        uint32_t num = ubElems - tCo * tK;
        uint32_t den = tCo + tK + 1u; // +tmp tileM
        if (den == 0u) return 0u;
        uint32_t tm = num / den;
        tm = AlignDown(tm, MIN_ALIGN_ELEMS);
        if (tm == 0u) tm = (num >= den) ? 1u : 0u;
        return tm;
    };

    uint32_t tileM = solveTileM(tileCo, tileK);
    if (tileM == 0u) {
        // Try shrinking tileCo then tileK
        if (tileCo > 32u) {
            tileCo = AlignDown(tileCo, 32u);
        }
        tileM = solveTileM(tileCo, tileK);
    }
    if (tileM == 0u && tileK > 32u) {
        tileK = AlignDown(tileK, 32u);
        tileM = solveTileM(tileCo, tileK);
    }
    if (tileM == 0u) {
        tileCo = (tileCo > 16u) ? 16u : tileCo;
        tileK  = (tileK > 16u) ? 16u : tileK;
        tileM  = solveTileM(tileCo, tileK);
        if (tileM == 0u) tileM = 16u;
    }

    if (tileM > HW) tileM = AlignDown(HW, MIN_ALIGN_ELEMS);
    if (tileM == 0u) tileM = HW;

    const uint32_t mTiles = CeilDiv(HW, tileM);
    uint32_t blockDim = N * mTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_CIn(CIn);
    tiling.set_COut(COut);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_tileK(tileK);
    tiling.set_tileCo(tileCo);
    tiling.set_tileM(tileM);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

static inline __aicore__ uint32_t CeilDivU32(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : ((a + b - 1u) / b);
}

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t CIn, uint32_t COut,
                                uint32_t H, uint32_t W,
                                uint32_t tileK, uint32_t tileCo, uint32_t tileM)
    {
        N_ = N; CIn_ = CIn; COut_ = COut; H_ = H; W_ = W;
        HW_ = H_ * W_;
        tileK_ = tileK; tileCo_ = tileCo; tileM_ = tileM;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint32_t)(N_ * CIn_ * HW_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint32_t)(COut_ * CIn_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint32_t)(N_ * COut_ * HW_));

        // UB buffers
        pipe_.InitBuffer(xQ_, 1, tileK_ * tileM_ * sizeof(float));
        pipe_.InitBuffer(wQ_, 1, tileCo_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(yQ_, 1, tileCo_ * tileM_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tileM_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t numMTilesPerN = CeilDivU32(HW_, tileM_);
        const uint32_t totalTiles = N_ * numMTilesPerN;
        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            const uint32_t n = t / numMTilesPerN;
            const uint32_t mTileIdx = t % numMTilesPerN;
            const uint32_t mStart = mTileIdx * tileM_;
            uint32_t curM = HW_ - mStart;
            if (curM > tileM_) curM = tileM_;

            for (uint32_t coStart = 0; coStart < COut_; coStart += tileCo_) {
                uint32_t curCo = COut_ - coStart;
                if (curCo > tileCo_) curCo = tileCo_;

                // Allocate tiles
                AscendC::LocalTensor<float> yTile = yQ_.AllocTensor<float>();
                // Zero yTile
                AscendC::Muls(yTile, yTile, 0.0f, curCo * curM);

                for (uint32_t kBase = 0; kBase < CIn_; kBase += tileK_) {
                    uint32_t curK = CIn_ - kBase;
                    if (curK > tileK_) curK = tileK_;

                    // Load X tile: shape [curK, curM], contiguous per channel
                    AscendC::LocalTensor<float> xTile = xQ_.AllocTensor<float>();
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        uint32_t c = kBase + kk;
                        uint32_t xBase = ((n * CIn_ + c) * HW_) + mStart;
                        AscendC::DataCopy(xTile[kk * tileM_], xGm_[xBase], curM);
                    }
                    xQ_.EnQue(xTile);

                    // Load W tile: shape [curCo, curK], contiguous per row
                    AscendC::LocalTensor<float> wTile = wQ_.AllocTensor<float>();
                    for (uint32_t r = 0; r < curCo; ++r) {
                        uint32_t co = coStart + r;
                        uint32_t wBase = (co * CIn_) + kBase;
                        AscendC::DataCopy(wTile[r * tileK_], wGm_[wBase], curK);
                    }
                    wQ_.EnQue(wTile);

                    // Compute: Y[curCo, curM] += W[curCo, curK] * X[curK, curM]
                    AscendC::LocalTensor<float> xTileUse = xQ_.DeQue<float>();
                    AscendC::LocalTensor<float> wTileUse = wQ_.DeQue<float>();

                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        AscendC::LocalTensor<float> xCol = xTileUse[kk * tileM_]; // len=curM
                        for (uint32_t r = 0; r < curCo; ++r) {
                            AscendC::LocalTensor<float> yRow = yTile[r * tileM_]; // len=curM
                            // Read scalar weight
                            float wScalar = wTileUse.GetValue(r * tileK_ + kk);
                            AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                            AscendC::Muls(tmp, xCol, wScalar, curM);
                            AscendC::Add(yRow, yRow, tmp, curM);
                            tmpQ_.FreeTensor(tmp);
                        }
                    }

                    xQ_.FreeTensor(xTileUse);
                    wQ_.FreeTensor(wTileUse);
                } // kBase loop

                // Store Y tile rows
                for (uint32_t r = 0; r < curCo; ++r) {
                    uint32_t co = coStart + r;
                    uint32_t yBase = ((n * COut_ + co) * HW_) + mStart;
                    AscendC::LocalTensor<float> yRow = yTile[r * tileM_];
                    AscendC::DataCopy(yGm_[yBase], yRow, curM);
                }

                yQ_.FreeTensor(yTile);
            } // coStart
        } // tiles
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  xQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  wQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, CIn_{0}, COut_{0}, H_{0}, W_{0}, HW_{0};
    uint32_t tileK_{0}, tileCo_{0}, tileM_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                                               GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.CIn, tiling_data.COut,
            tiling_data.H, tiling_data.W,
            tiling_data.tileK, tiling_data.tileCo, tiling_data.tileM);
    op.Process();
}
