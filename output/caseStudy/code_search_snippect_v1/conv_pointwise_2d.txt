#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// Memory budget and alignment constraints for UB tiling.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // DMA-friendly tile alignment
static constexpr uint32_t MIN_TILE = 256;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const auto &x_os = x_shape->GetOriginShape();
    const auto &w_os = w_shape->GetOriginShape();

    if (x_os.GetDimNum() != 4 || w_os.GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(x_os.GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_os.GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_os.GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_os.GetDim(3));
    const uint32_t Cout = static_cast<uint32_t>(w_os.GetDim(0));
    // w_os[1] should equal Cin; w_os[2] = 1; w_os[3] = 1

    const uint32_t P = H * W;

    // UB usage model:
    //   - WChunk: coChunk * Cin floats
    //   - YTile: coChunk * tileT floats
    //   - XRow double-buffer: 2 * tileT floats
    //   - Tmp vector: 1 * tileT floats
    // Total floats = coChunk*(Cin + tileT) + 3*tileT
    // Total bytes = 4 * [coChunk*(Cin + tileT) + 3*tileT]
    // Solve for tileT <= floor((UB/4 - coChunk*Cin) / (coChunk + 3))

    uint32_t coChunk = Cout < 32u ? Cout : 32u;
    if (coChunk == 0u) coChunk = 1u;

    uint32_t tileT = 0;
    uint32_t ubFloatBudget = UB_BUDGET_BYTES / 4u;

    // Try decreasing coChunk down to 1 until a feasible tileT is found.
    for (; coChunk >= 1u; coChunk /= 2u) {
        uint32_t denom = coChunk + 3u;
        if (ubFloatBudget <= coChunk * Cin || denom == 0u) {
            tileT = 0;
        } else {
            uint32_t t = (ubFloatBudget - coChunk * Cin) / denom;
            t = AlignDown(t, ALIGN_ELEMS);
            if (t > P) t = AlignDown(P, ALIGN_ELEMS);
            tileT = t;
        }
        if (tileT >= MIN_TILE || P < MIN_TILE) break;
        if (coChunk == 1u) break;
    }

    if (tileT == 0u) {
        // Fallback minimal tile size
        tileT = (P < ALIGN_ELEMS) ? P : ALIGN_ELEMS;
        tileT = (tileT == 0u) ? 1u : tileT;
        coChunk = coChunk == 0u ? 1u : coChunk;
    }

    // Final bounds
    if (tileT > P) tileT = P;
    if (tileT == 0u) tileT = 1u;
    if (coChunk == 0u) coChunk = 1u;
    if (coChunk > Cout) coChunk = Cout;

    const uint32_t tilesPerN = (tileT == 0u) ? 1u : CeilDiv(P, tileT);
    uint32_t totalTiles = N * tilesPerN;
    if (totalTiles == 0u) totalTiles = 1u;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;      // lower bound for better occupancy
    if (blockDim > 24u) blockDim = 24u;    // conservative upper bound

    context->SetBlockDim(blockDim);

    optiling::ConvPointwise2dCustomTilingData tiling;
    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_cout(Cout);
    tiling.set_tileT(tileT);
    tiling.set_coChunk(coChunk);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) return GRAPH_FAILED;

    // Output shape: [N, Cout, H, W]
    const auto &x_os = x_shape->GetOriginShape();
    const auto &w_os = w_shape->GetOriginShape();
    if (x_os.GetDimNum() != 4 || w_os.GetDimNum() != 4) return GRAPH_FAILED;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, x_os.GetDim(0));
    y_shape->SetDim(1, w_os.GetDim(0));
    y_shape->SetDim(2, x_os.GetDim(2));
    y_shape->SetDim(3, x_os.GetDim(3));
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t h, uint32_t wi, uint32_t cout,
                                uint32_t tileT, uint32_t coChunk)
    {
        N_ = n;
        Cin_ = cin;
        H_ = h;
        W_ = wi;
        Cout_ = cout;
        P_ = H_ * W_;
        tileT_ = tileT;
        coChunk_ = coChunk;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * Cin_ * P_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Cout_) * Cin_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * Cout_ * P_);

        // Buffers
        // - WChunk: coChunk * Cin
        // - XRow: tileT
        // - Tmp: tileT
        // - YTile: coChunk * tileT
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, coChunk_ * Cin_ * sizeof(float));
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileT_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileT_ * sizeof(float));
        pipe_.InitBuffer(veccalcQueue_, BUFFER_NUM, coChunk_ * tileT_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileT_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || P_ == 0 || tileT_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t tilesPerN = (tileT_ == 0u) ? 1u : ((P_ + tileT_ - 1u) / tileT_);
        const uint32_t totalTiles = N_ * tilesPerN;

        for (uint32_t tileLinear = blockIdx; tileLinear < totalTiles; tileLinear += blockNum) {
            const uint32_t nIdx = tileLinear / tilesPerN;
            const uint32_t tileIdxInN = tileLinear % tilesPerN;
            const uint32_t posStart = tileIdxInN * tileT_;
            uint32_t curT = P_ - posStart;
            if (curT > tileT_) curT = tileT_;
            if (curT == 0u) continue;

            for (uint32_t coBase = 0; coBase < Cout_; coBase += coChunk_) {
                uint32_t coCurr = Cout_ - coBase;
                if (coCurr > coChunk_) coCurr = coChunk_;

                // Load W chunk [coCurr, Cin_]
                AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
                const uint64_t wOffset = static_cast<uint64_t>(coBase) * Cin_;
                AscendC::DataCopy(wLocal, wGm_[wOffset], static_cast<uint64_t>(coCurr) * Cin_);
                inQueueW_.EnQue(wLocal); // keep resident until done

                // Allocate Y tile [coCurr, curT]
                AscendC::LocalTensor<float> yTile = veccalcQueue_.AllocTensor<float>();

                // Accumulate across Cin
                bool firstCi = true;
                for (uint32_t ci = 0; ci < Cin_; ++ci) {
                    // Load X row: x[nIdx, ci, :, :] positions [posStart : posStart + curT)
                    AscendC::LocalTensor<float> xRow = inQueueX_.AllocTensor<float>();
                    const uint64_t xOffset = (static_cast<uint64_t>(nIdx) * Cin_ + ci) * P_ + posStart;
                    AscendC::DataCopy(xRow, xGm_[xOffset], curT);

                    // For each co in chunk, yRow += xRow * wScalar
                    for (uint32_t coOff = 0; coOff < coCurr; ++coOff) {
                        const float wScalar = wLocal[coOff * Cin_ + ci];
                        const uint64_t yRowOffset = static_cast<uint64_t>(coOff) * curT;

                        if (firstCi) {
                            // yRow = xRow * wScalar
                            AscendC::Muls(yTile[yRowOffset], xRow, wScalar, curT);
                        } else {
                            AscendC::LocalTensor<float> tVec = tmpQueue_.AllocTensor<float>();
                            AscendC::Muls(tVec, xRow, wScalar, curT);
                            AscendC::Add(yTile[yRowOffset], yTile[yRowOffset], tVec, curT);
                            tmpQueue_.FreeTensor(tVec);
                        }
                    }

                    inQueueX_.FreeTensor(xRow);
                    firstCi = false;
                }

                // Write Y tile rows to GM
                for (uint32_t coOff = 0; coOff < coCurr; ++coOff) {
                    const uint32_t co = coBase + coOff;
                    const uint64_t yOffset = ((static_cast<uint64_t>(nIdx) * Cout_ + co) * P_) + posStart;
                    AscendC::DataCopy(yGm_[yOffset], yTile[static_cast<uint64_t>(coOff) * curT], curT);
                }

                veccalcQueue_.FreeTensor(yTile);
                AscendC::LocalTensor<float> wLocalDeq = inQueueW_.DeQue<float>();
                inQueueW_.FreeTensor(wLocalDeq);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> veccalcQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0};
    uint32_t Cin_{0};
    uint32_t H_{0};
    uint32_t W_{0};
    uint32_t Cout_{0};
    uint32_t P_{0};
    uint32_t tileT_{0};
    uint32_t coChunk_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.h, tiling_data.w, tiling_data.cout,
            tiling_data.tileT, tiling_data.coChunk);
    op.Process();
}
