namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();

    uint32_t dim_num = static_cast<uint32_t>(origin.GetDimNum());
    if (dim_num < 1) {
        return ge::GRAPH_FAILED;
    }

    uint64_t N64 = 1, C64 = 1, L64 = 1;
    if (dim_num == 1) {
        L64 = static_cast<uint64_t>(origin.GetDim(0));
    } else if (dim_num == 2) {
        C64 = static_cast<uint64_t>(origin.GetDim(0));
        L64 = static_cast<uint64_t>(origin.GetDim(1));
    } else {
        N64 = static_cast<uint64_t>(origin.GetDim(0));
        C64 = static_cast<uint64_t>(origin.GetDim(1));
        L64 = static_cast<uint64_t>(origin.GetDim(2));
    }

    uint32_t N = static_cast<uint32_t>(N64);
    uint32_t C = static_cast<uint32_t>(C64);
    uint32_t L = static_cast<uint32_t>(L64);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size_ptr = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride_ptr      = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding_ptr     = attrs->GetAttrPointer<int64_t>(2);
    const bool    *return_indices  = attrs->GetAttrPointer<bool>(3);
    (void)return_indices; // not used in this kernel variant

    uint32_t K = static_cast<uint32_t>(*kernel_size_ptr);
    uint32_t S = static_cast<uint32_t>(*stride_ptr);
    uint32_t P = static_cast<uint32_t>(*padding_ptr);

    if (S == 0) {
        S = K; // default stride to kernel_size
    }

    uint32_t Lout = 0;
    if ((int64_t)L + 2LL * (int64_t)P >= (int64_t)K && S > 0) {
        Lout = static_cast<uint32_t>(((int64_t)L + 2LL * (int64_t)P - (int64_t)K) / (int64_t)S + 1LL);
    }

    uint32_t M = (N == 0 || C == 0) ? 0u : (N * C);

    uint32_t blockDim = (M == 0) ? 1u : M;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_L(L);
    tiling.set_Lout(Lout);
    tiling.set_K(K);
    tiling.set_S(S);
    tiling.set_P(P);
    tiling.set_M(M);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t WIN_BUFFER_NUM = 1;

// A simple MaxPool1D kernel over the last dimension of an (N, C, L) tensor.
// Attributes: kernel_size (K), stride (S), padding (P). Dilation is assumed 1.
// Only float32 input/output are supported. return_indices is ignored.
class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t L, uint32_t Lout,
                                uint32_t K, uint32_t S, uint32_t P)
    {
        N_ = N; C_ = C; L_ = L; Lout_ = Lout;
        K_ = K; S_ = S; P_ = P;
        M_ = (N_ == 0 || C_ == 0) ? 0u : (N_ * C_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * C_ * L_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * C_ * Lout_);

        pipe_.InitBuffer(winQueue_, WIN_BUFFER_NUM, K_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || Lout_ == 0 || K_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over channels (N*C)
        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            const uint64_t xBase = (uint64_t)m * L_;
            const uint64_t yBase = (uint64_t)m * Lout_;

            for (uint32_t j = 0; j < Lout_; ++j) {
                int64_t start = (int64_t)j * (int64_t)S_ - (int64_t)P_;
                int64_t end_inclusive = start + (int64_t)K_ - 1;

                int64_t valid_start = start < 0 ? 0 : start;
                int64_t valid_end = end_inclusive > ((int64_t)L_ - 1) ? ((int64_t)L_ - 1) : end_inclusive;
                int64_t valid_len64 = valid_end - valid_start + 1;
                uint32_t valid_len = (valid_len64 > 0) ? static_cast<uint32_t>(valid_len64) : 0u;

                uint32_t left_pad = (start < 0) ? static_cast<uint32_t>(-start) : 0u;

                AscendC::LocalTensor<float> win = winQueue_.AllocTensor<float>();
                // Initialize with very negative numbers to model -inf padding
                float neg_inf = -3.4e38f;
                AscendC::Duplicate(win, neg_inf, K_);

                if (valid_len > 0) {
                    // Copy valid segment into window buffer at position left_pad
                    AscendC::DataCopy(win[left_pad], xGm_[xBase + static_cast<uint64_t>(valid_start)], valid_len);
                }

                // Reduce max over K elements
                float maxv = neg_inf;
                for (uint32_t k = 0; k < K_; ++k) {
                    float v = win.GetValue(k);
                    if (v > maxv) maxv = v;
                }

                // Write result
                win.SetValue(0, maxv);
                AscendC::DataCopy(yGm_[yBase + j], win, 1);

                winQueue_.FreeTensor(win);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, WIN_BUFFER_NUM> winQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, L_{0}, Lout_{0};
    uint32_t K_{0}, S_{0}, P_{0};
    uint32_t M_{0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.L, tiling_data.Lout,
            tiling_data.K, tiling_data.S, tiling_data.P);
    op.Process();
}
"""

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t compute_out_len(int64_t L, int64_t K, int64_t S, int64_t P)
{
    if (S == 0) S = K;
    const int64_t numer = L + 2 * P - K;
    if (numer < 0) return 0;
    return numer / S + 1;
}

at::Tensor max_pooling_1d_impl_npu(const at::Tensor& self,
                                   int64_t kernel_size,
                                   c10::optional<int64_t> stride_opt,
                                   int64_t padding,
                                   bool return_indices)
{
    int64_t stride = stride_opt.has_value() ? stride_opt.value() : 0;
    int64_t N = 1, C = 1, L = 1;
    const auto sizes = self.sizes();
    if (sizes.size() == 1) {
        L = sizes[0];
    } else if (sizes.size() == 2) {
        C = sizes[0];
        L = sizes[1];
    } else {
        N = sizes[0];
        C = sizes[1];
        L = sizes[2];
    }
    int64_t Lout = compute_out_len(L, kernel_size, stride, padding);
    torch::IntArrayRef out_sizes;
    std::vector<int64_t> out_vec;
    if (sizes.size() == 1) {
        out_vec = {Lout};
    } else if (sizes.size() == 2) {
        out_vec = {C, Lout};
    } else {
        out_vec = {N, C, Lout};
    }
    at::Tensor result = at::empty(out_vec, self.options());
    EXEC_NPU_CMD(aclnnMaxPooling1dCustom, self, kernel_size, stride, padding, return_indices, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_1d_custom", &max_pooling_1d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_1d_custom", &max_pooling_1d_impl_npu, "MaxPooling1D (MaxPooling1dCustom on NPU)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_1d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0, return_indices: bool = False):
        super(ModelNew, self).__init__()
        self.kernel_size = int(kernel_size)
        self.stride = int(stride) if stride is not None else None
        self.padding = int(padding)
        self.return_indices = bool(return_indices)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding, self.return_indices)
