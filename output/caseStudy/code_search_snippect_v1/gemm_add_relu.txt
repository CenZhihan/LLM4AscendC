#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
static constexpr uint32_t ELEMSIZE = 4;                 // fp32
static constexpr uint32_t ALIGN_K = 16;
static constexpr uint32_t ALIGN_M = 8;
static constexpr uint32_t ALIGN_N = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;

    // Shapes:
    // x: [M, K], w: [N, K], bias: [N], y: [M, N]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    uint32_t M = 1;
    if (x_shape->GetOriginShape().GetDimNum() == 1) {
        M = 1;
    } else {
        for (int i = 0; i < x_shape->GetOriginShape().GetDimNum() - 1; ++i) {
            M *= static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(i));
        }
    }
    const uint32_t K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(x_shape->GetOriginShape().GetDimNum() - 1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));

    // Initial tile candidates
    uint32_t tileM = 8;
    uint32_t tileN = 64;
    uint32_t tileK = 256;

    tileM = (tileM > M) ? ((M >= 1) ? AlignDown(M, 1) : 1) : tileM;
    tileN = (tileN > N) ? ((N >= 1) ? AlignDown(N, 1) : 1) : tileN;
    tileK = (tileK > K) ? ((K >= 1) ? AlignDown(K, 1) : 1) : tileK;

    // Ensure alignment constraints
    if (tileM > 0) tileM = AlignDown(tileM, ALIGN_M);
    if (tileN > 0) tileN = AlignDown(tileN, ALIGN_N);
    if (tileK > 0) tileK = AlignDown(tileK, ALIGN_K);
    if (tileM == 0) tileM = (M >= 1) ? 1 : 1;
    if (tileN == 0) tileN = (N >= 1) ? 1 : 1;
    if (tileK == 0) tileK = (K >= 1) ? 1 : 1;

    // Fit into UB budget: bytes = 4 * (tileM*tileK + tileN*tileK + tileM*tileN)
    auto fits = [&](uint32_t tm, uint32_t tn, uint32_t tk) -> bool {
        uint64_t elems = static_cast<uint64_t>(tm) * tk
                       + static_cast<uint64_t>(tn) * tk
                       + static_cast<uint64_t>(tm) * tn;
        uint64_t bytes = ELEMSIZE * elems;
        return bytes <= UB_BUDGET_BYTES;
    };

    // Adjust tiles down to fit budget
    while (!fits(tileM, tileN, tileK)) {
        if (tileK > ALIGN_K) {
            tileK = AlignDown(tileK / 2, ALIGN_K);
            if (tileK == 0) tileK = 1;
        } else if (tileN > ALIGN_N) {
            tileN = AlignDown(tileN / 2, ALIGN_N);
            if (tileN == 0) tileN = 1;
        } else if (tileM > ALIGN_M) {
            tileM = AlignDown(tileM / 2, ALIGN_M);
            if (tileM == 0) tileM = 1;
        } else {
            // Minimal tiles
            break;
        }
    }

    // Compute blockDim across M-tile grid for load balance
    const uint32_t mTiles = CeilDiv(M, tileM);
    uint32_t blockDim = mTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // no workspace
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

class KernelGemmAddReluCustom {
public:
    __aicore__ inline KernelGemmAddReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tM_ = tileM; tN_ = tileN; tK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        pipe_.InitBuffer(xQueue_, 1, tM_ * tK_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, tN_ * tK_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, 1, tM_ * tN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t mTiles = CeilDiv(M_, tM_);
        // grid-stride over M tiles
        for (uint32_t mt = blockIdx; mt < mTiles; mt += blockNum) {
            const uint32_t mBase = mt * tM_;
            const uint32_t mActual = (mBase + tM_ <= M_) ? tM_ : (M_ - mBase);
            if (mActual == 0) continue;

            // Iterate N tiles sequentially on this core
            for (uint32_t nBase = 0; nBase < N_; nBase += tN_) {
                const uint32_t nActual = (nBase + tN_ <= N_) ? tN_ : (N_ - nBase);
                if (nActual == 0) continue;

                // Allocate y tile and zero-init
                AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                ZeroTile(yLocal, mActual, nActual);

                // K loop: accumulate partial products
                for (uint32_t kBase = 0; kBase < K_; kBase += tK_) {
                    const uint32_t kActual = (kBase + tK_ <= K_) ? tK_ : (K_ - kBase);

                    // Copy X_sub: shape [mActual, kActual]
                    AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                    CopyXSub(xLocal, mBase, kBase, mActual, kActual);

                    // Copy W_sub: shape [nActual, kActual] (from weight rows)
                    AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                    CopyWSub(wLocal, nBase, kBase, nActual, kActual);

                    // GEMM accumulate: yLocal[m, n] += dot(xLocal[m,:], wLocal[n,:])
                    GemmAccumulate(yLocal, xLocal, wLocal, mActual, nActual, kActual);

                    wQueue_.FreeTensor(wLocal);
                    xQueue_.FreeTensor(xLocal);
                }

                // Add bias and apply ReLU
                AddBiasRelu(yLocal, nBase, mActual, nActual);

                // Copy out Y_sub back to GM
                CopyYSub(yLocal, mBase, nBase, mActual, nActual);

                yQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
    {
        return (a + b - 1u) / b;
    }

    __aicore__ inline void ZeroTile(AscendC::LocalTensor<float>& yLocal, uint32_t mActual, uint32_t nActual)
    {
        const uint32_t elems = mActual * nActual;
        AscendC::Dup(yLocal, 0.0f, elems);
    }

    __aicore__ inline void CopyXSub(AscendC::LocalTensor<float>& xLocal,
                                    uint32_t mBase, uint32_t kBase,
                                    uint32_t mActual, uint32_t kActual)
    {
        // Row-by-row copy into a compact [mActual, kActual]
        for (uint32_t mi = 0; mi < mActual; ++mi) {
            const uint32_t gmOff = (mBase + mi) * K_ + kBase;
            AscendC::DataCopy(xLocal[mi * tK_], xGm_[gmOff], kActual);
        }
    }

    __aicore__ inline void CopyWSub(AscendC::LocalTensor<float>& wLocal,
                                    uint32_t nBase, uint32_t kBase,
                                    uint32_t nActual, uint32_t kActual)
    {
        // Row-by-row copy of weight slice [nActual, kActual] (w is [N, K])
        for (uint32_t ni = 0; ni < nActual; ++ni) {
            const uint32_t gmOff = (nBase + ni) * K_ + kBase;
            AscendC::DataCopy(wLocal[ni * tK_], wGm_[gmOff], kActual);
        }
    }

    __aicore__ inline void GemmAccumulate(AscendC::LocalTensor<float>& yLocal,
                                          const AscendC::LocalTensor<float>& xLocal,
                                          const AscendC::LocalTensor<float>& wLocal,
                                          uint32_t mActual, uint32_t nActual, uint32_t kActual)
    {
        // Naive dot-product accumulation over kActual
        for (uint32_t mi = 0; mi < mActual; ++mi) {
            const uint32_t yRowBase = mi * tN_;
            const uint32_t xRowBase = mi * tK_;
            for (uint32_t ni = 0; ni < nActual; ++ni) {
                float acc = 0.0f;
                const uint32_t wRowBase = ni * tK_;
                for (uint32_t kk = 0; kk < kActual; ++kk) {
                    acc += xLocal[xRowBase + kk] * wLocal[wRowBase + kk];
                }
                yLocal[yRowBase + ni] = yLocal[yRowBase + ni] + acc;
            }
        }
    }

    __aicore__ inline void AddBiasRelu(AscendC::LocalTensor<float>& yLocal,
                                       uint32_t nBase, uint32_t mActual, uint32_t nActual)
    {
        for (uint32_t mi = 0; mi < mActual; ++mi) {
            const uint32_t yRowBase = mi * tN_;
            for (uint32_t ni = 0; ni < nActual; ++ni) {
                float v = yLocal[yRowBase + ni] + bGm_[nBase + ni];
                // ReLU
                yLocal[yRowBase + ni] = v > 0.0f ? v : 0.0f;
            }
        }
    }

    __aicore__ inline void CopyYSub(const AscendC::LocalTensor<float>& yLocal,
                                    uint32_t mBase, uint32_t nBase,
                                    uint32_t mActual, uint32_t nActual)
    {
        // Row-by-row copy back to GM (y is [M, N])
        for (uint32_t mi = 0; mi < mActual; ++mi) {
            const uint32_t gmOff = (mBase + mi) * N_ + nBase;
            AscendC::DataCopy(yGm_[gmOff], yLocal[mi * tN_], nActual);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tM_{0}, tN_{0}, tK_{0};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddReluCustom op;
    op.Init(x, w, bias, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
