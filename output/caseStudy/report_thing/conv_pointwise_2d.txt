host_operator_src="""
#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core usable UB budget (conservative)
static constexpr uint32_t ALIGN_ELEMS = 1024;            // DMA friendly alignment
static constexpr uint32_t MAX_TILE_N = 8;                // UB-friendly number of output channels to accumulate together

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    // Input shapes
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 4 || w_shape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Cin  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));
    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t Kchk = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    if (Kchk != Cin) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t P = H * W;
    const uint32_t elemBudget = UB_BUDGET_BYTES / 4u;

    // Choose tileN conservatively
    uint32_t tileN = Cout < MAX_TILE_N ? Cout : MAX_TILE_N;
    if (tileN == 0u) tileN = 1u;

    // Compute tileM from budget: bytes ~= 4 * (tileM * tileN + tileM + tileM + tileN * Cin)
    // (yAcc: tileM*tileN, xVec: tileM, tmpVec: tileM, weight tile: tileN*Cin)
    // => tileM * (tileN + 2) + tileN * Cin <= elemBudget
    uint32_t maxTileM = 0;
    if (elemBudget > tileN * Cin) {
        maxTileM = (elemBudget - tileN * Cin) / (tileN + 2u);
    } else {
        maxTileM = ALIGN_ELEMS;
    }
    maxTileM = AlignDown(maxTileM, ALIGN_ELEMS);
    if (maxTileM < ALIGN_ELEMS) maxTileM = ALIGN_ELEMS;

    uint32_t tileM = maxTileM;
    if (tileM > P) {
        const uint32_t aligned = AlignDown(P, ALIGN_ELEMS);
        tileM = (aligned > 0u) ? aligned : P;
    }
    if (tileM == 0u) tileM = 1u;

    // Compute blockDim based on number of segments
    const uint32_t segPerN = (tileM == 0u) ? 1u : CeilDiv(P, tileM);
    const uint32_t totalSegs = N * segPerN;
    uint32_t blockDim = totalSegs;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cIn(Cin);
    tiling.set_cOut(Cout);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t MAX_TILE_N = 8;   // must be >= tiling.tileN

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t h, uint32_t wi,
                                uint32_t tileM, uint32_t tileN)
    {
        n_ = n;
        cin_ = cIn;
        cout_ = cOut;
        h_ = h;
        w_ = wi;
        p_ = h_ * w_;
        tileM_ = tileM;
        tileN_ = tileN > (uint32_t)MAX_TILE_N ? (uint32_t)MAX_TILE_N : tileN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * cin_ * p_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(cout_ * cin_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_ * cout_ * p_));

        // UB buffers
        // accQ_: up to MAX_TILE_N buffers of length tileM_ each (y accumulators)
        pipe_.InitBuffer(accQ_, MAX_TILE_N, tileM_ * sizeof(float));
        // wQ_: up to MAX_TILE_N buffers of length cin_ each (weight vectors for each out channel in tile)
        pipe_.InitBuffer(wQ_, MAX_TILE_N, cin_ * sizeof(float));
        // x/tmp vectors: two buffers of length tileM_: one xVec and one tmp
        pipe_.InitBuffer(tmpQ_, 2, tileM_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || p_ == 0 || tileM_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride over segments of positions within each batch n
        const uint32_t segPerN = (tileM_ == 0u) ? 1u : ((p_ + tileM_ - 1u) / tileM_);
        const uint32_t totalSegs = n_ * segPerN;

        for (uint32_t segId = blockIdx; segId < totalSegs; segId += blockNum) {
            const uint32_t nIdx = segId / segPerN;
            const uint32_t segInN = segId % segPerN;
            const uint32_t basePos = segInN * tileM_;
            uint32_t lenM = p_ - basePos;
            if (lenM > tileM_) lenM = tileM_;

            // Loop over output channels in tiles of tileN_
            for (uint32_t outBase = 0; outBase < cout_; outBase += tileN_) {
                const uint32_t tN = (outBase + tileN_ <= cout_) ? tileN_ : (cout_ - outBase);

                // Allocate accumulators and weight vectors for this tile
                AscendC::LocalTensor<float> yAcc[MAX_TILE_N];
                AscendC::LocalTensor<float> wVec[MAX_TILE_N];

                for (uint32_t j = 0; j < tN; ++j) {
                    yAcc[j] = accQ_.AllocTensor<float>();
                }
                for (uint32_t j = 0; j < tN; ++j) {
                    wVec[j] = wQ_.AllocTensor<float>();
                    // copy contiguous weight vector for out channel (outBase + j)
                    const uint32_t wOffset = (outBase + j) * cin_;
                    AscendC::DataCopy(wVec[j], wGm_[wOffset], cin_);
                }

                // Temporary vectors
                AscendC::LocalTensor<float> xVec = tmpQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();

                // k loop (input channels)
                for (uint32_t k = 0; k < cin_; ++k) {
                    // Load x vector for this segment, channel k
                    const uint32_t xOffset = (nIdx * cin_ + k) * p_ + basePos;
                    AscendC::DataCopy(xVec, xGm_[xOffset], lenM);

                    // Accumulate into each output accumulator
                    for (uint32_t j = 0; j < tN; ++j) {
                        // Get weight scalar w[outBase + j, k] from wVec[j]
                        float wk = wVec[j].GetValue(k);
                        if (k == 0u) {
                            // yAcc[j] = xVec * wk
                            AscendC::Muls(yAcc[j], xVec, wk, lenM);
                        } else {
                            // tmp = xVec * wk, yAcc[j] += tmp
                            AscendC::Muls(tmp, xVec, wk, lenM);
                            AscendC::Add(yAcc[j], yAcc[j], tmp, lenM);
                        }
                    }
                }

                // Write accumulators to GM
                for (uint32_t j = 0; j < tN; ++j) {
                    const uint32_t yOffset = (nIdx * cout_ + (outBase + j)) * p_ + basePos;
                    AscendC::DataCopy(yGm_[yOffset], yAcc[j], lenM);
                }

                // Free local tensors
                tmpQ_.FreeTensor(tmp);
                tmpQ_.FreeTensor(xVec);
                for (uint32_t j = 0; j < tN; ++j) {
                    wQ_.FreeTensor(wVec[j]);
                }
                for (uint32_t j = 0; j < tN; ++j) {
                    accQ_.FreeTensor(yAcc[j]);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECCALC, MAX_TILE_N> accQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, MAX_TILE_N>   wQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2>          tmpQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t h_{0}, w_{0}, p_{0};
    uint32_t tileM_{0}, tileN_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cIn, tiling_data.cOut,
            tiling_data.h, tiling_data.w,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
