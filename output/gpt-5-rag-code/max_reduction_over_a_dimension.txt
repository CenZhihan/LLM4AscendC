#include "max_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <vector>

namespace {
inline int64_t NormalizeAxis(int64_t axis, int64_t rank) {
    if (axis < 0) axis += rank;
    return axis;
}
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaxReductionOverADimensionCustomTilingData tiling;

    const gert::Shape* inShape = context->GetInputShape(0);
    const auto& origin = inShape->GetOriginShape();
    int64_t rank = static_cast<int64_t>(origin.GetDimNum());

    int64_t dimAttr = 0;
    if (context->GetAttrs() != nullptr) {
        // try get attribute 'dim'
        auto attrs = context->GetAttrs();
        if (attrs->HasAttr("dim")) {
            dimAttr = attrs->GetInt("dim");
        }
    }
    int64_t axis = NormalizeAxis(dimAttr, rank);
    axis = std::max<int64_t>(0, std::min<int64_t>(axis, rank - 1));

    uint64_t totalElems = static_cast<uint64_t>(origin.GetShapeSize());
    std::vector<uint64_t> dims(rank);
    for (int64_t i = 0; i < rank; ++i) {
        dims[i] = static_cast<uint64_t>(origin.GetDim(i));
    }

    uint64_t R = dims[axis];
    uint64_t S = 1; // stride between adjacent elements along the reduce axis
    for (int64_t i = axis + 1; i < rank; ++i) {
        S *= dims[i];
    }
    uint64_t totalA = (R == 0) ? 0 : (totalElems / R);

    // set block dim according to parallelism on A dimension
    uint32_t blockDim = static_cast<uint32_t>(std::min<uint64_t>(32, std::max<uint64_t>(1, totalA)));
    context->SetBlockDim(blockDim);

    tiling.set_totalA(static_cast<uint32_t>(totalA));
    tiling.set_R(static_cast<uint32_t>(R));
    tiling.set_S(static_cast<uint32_t>(S));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* inShape = context->GetInputShape(0);
    const auto& origin = inShape->GetOriginShape();
    int64_t rank = static_cast<int64_t>(origin.GetDimNum());

    int64_t dimAttr = 0;
    if (context->GetAttrs() != nullptr) {
        auto attrs = context->GetAttrs();
        if (attrs->HasAttr("dim")) {
            dimAttr = attrs->GetInt("dim");
        }
    }
    int64_t axis = dimAttr;
    if (axis < 0) axis += rank;
    if (axis < 0) axis = 0;
    if (axis >= rank) axis = rank - 1;

    std::vector<int64_t> outDims;
    outDims.reserve(std::max<int64_t>(0, rank - 1));
    for (int64_t i = 0; i < rank; ++i) {
        if (i == axis) continue;
        outDims.push_back(origin.GetDim(i));
    }

    gert::Shape* outShape = context->GetOutputShape(0);
    outShape->SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) {
        outShape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}

    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 1;

class KernelMaxReduction {
public:
    __aicore__ inline KernelMaxReduction() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalA, uint32_t R, uint32_t S)
    {
        this->totalA = totalA;
        this->R = R;
        this->S = S;

        uint32_t totalX = totalA * R;
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, totalX);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, totalA);

        pipe.InitBuffer(inQueueX, IN_BUF_NUM, sizeof(DTYPE_X));  // buffer for single element load
        pipe.InitBuffer(outQueueY, OUT_BUF_NUM, sizeof(DTYPE_Y)); // buffer for single element store
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t chunk = (totalA + blockNum - 1) / blockNum;
        uint32_t startA = blockIdx * chunk;
        uint32_t endA = AscendC::Min(startA + chunk, totalA);

        for (uint32_t a = startA; a < endA; ++a) {
            uint32_t base = (a / S) * (R * S) + (a % S);

            // Load first element to initialize max
            AscendC::LocalTensor<DTYPE_X> tmpFirst = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(tmpFirst, xGm[base], 1);
            DTYPE_X maxVal = tmpFirst[0];
            inQueueX.FreeTensor(tmpFirst);

            // Iterate remaining R-1 elements
            for (uint32_t r = 1; r < R; ++r) {
                AscendC::LocalTensor<DTYPE_X> tmp = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(tmp, xGm[base + r * S], 1);
                DTYPE_X cur = tmp[0];
                if (cur > maxVal) maxVal = cur;
                inQueueX.FreeTensor(tmp);
            }

            // write result
            AscendC::LocalTensor<DTYPE_Y> outLocal = outQueueY.AllocTensor<DTYPE_Y>();
            outLocal[0] = static_cast<DTYPE_Y>(maxVal);
            AscendC::DataCopy(yGm[a], outLocal, 1);
            outQueueY.FreeTensor(outLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalA;
    uint32_t R;
    uint32_t S;
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReduction op;
    op.Init(x, y, tiling_data.totalA, tiling_data.R, tiling_data.S);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor max_reduction_over_a_dimension_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    // Normalize dim similar to PyTorch
    int64_t rank = self.dim();
    int64_t axis = dim >= 0 ? dim : (dim + rank);
    if (axis < 0) axis = 0;
    if (axis >= rank) axis = rank - 1;

    // Build output sizes (reduce along axis)
    std::vector<int64_t> out_sizes;
    out_sizes.reserve(rank > 0 ? (rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == axis) continue;
        out_sizes.push_back(self.size(i));
    }
    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnMaxReductionOverADimensionCustom, self, (int)axis, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_reduction_over_a_dimension_custom", &max_reduction_over_a_dimension_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_reduction_over_a_dimension_custom", &max_reduction_over_a_dimension_custom_impl_npu, "Max reduce over a dimension");
