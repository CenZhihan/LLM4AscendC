kernel_src="""
#include "kernel_operator.h"
#include <cmath>

extern \"C\" __global__ __aicore__ void conv3d_multiply_instance_norm_clamp_multiply_max_custom(
    GM_ADDR x,
    GM_ADDR weight,
    GM_ADDR bias,
    GM_ADDR multiplier,
    GM_ADDR clamp_min,
    GM_ADDR clamp_max,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    // Dimensions
    const uint32_t N = tiling_data.n;
    const uint32_t Cin = tiling_data.cin;
    const uint32_t Cout = tiling_data.cout;
    const uint32_t InD = tiling_data.in_d;
    const uint32_t InH = tiling_data.in_h;
    const uint32_t InW = tiling_data.in_w;
    const uint32_t Kd = tiling_data.k_d;
    const uint32_t Kh = tiling_data.k_h;
    const uint32_t Kw = tiling_data.k_w;
    const uint32_t OutD = tiling_data.out_d;
    const uint32_t OutH = tiling_data.out_h;
    const uint32_t OutW = tiling_data.out_w;

    // Cast pointers
    __gm__ float* x_gm = (__gm__ float*)x;                               // [N, Cin, InD, InH, InW]
    __gm__ float* w_gm = (__gm__ float*)weight;                          // [Cout, Cin, Kd, Kh, Kw]
    __gm__ float* b_gm = (__gm__ float*)bias;                            // [Cout]
    __gm__ float* m_gm = (__gm__ float*)multiplier;                      // [Cout, 1, 1, 1] -> [Cout]
    __gm__ float* y_gm = (__gm__ float*)y;                               // [N, OutD, OutH, OutW]
    __gm__ float* cmin_gm = (__gm__ float*)clamp_min;                    // scalar
    __gm__ float* cmax_gm = (__gm__ float*)clamp_max;                    // scalar

    float cmin = cmin_gm[0];
    float cmax = cmax_gm[0];

    // Workspace layout
    // ws0: y2 = (conv(x)+bias) * multiplier  shape [N, Cout, OutD, OutH, OutW]
    // ws1: mean per (N,Cout) size N*Cout
    // ws2: var per (N,Cout) size N*Cout
    size_t out_spatial = (size_t)OutD * (size_t)OutH * (size_t)OutW;
    size_t y2_elems = (size_t)N * (size_t)Cout * out_spatial;
    __gm__ float* ws0_y2 = (__gm__ float*)workspace;
    __gm__ float* ws1_mean = ws0_y2 + y2_elems;
    __gm__ float* ws2_var  = ws1_mean + ((size_t)N * (size_t)Cout);

    const float eps = 1.0e-5f;

    // Pass 1: compute conv, apply first multiplier, store y2, and accumulate mean and var sums
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t co = 0; co < Cout; ++co) {
            double sum_val = 0.0;
            double sum_sq = 0.0;

            for (uint32_t od = 0; od < OutD; ++od) {
                for (uint32_t oh = 0; oh < OutH; ++oh) {
                    for (uint32_t ow = 0; ow < OutW; ++ow) {
                        // Convolution at (n, co, od, oh, ow)
                        float acc = 0.0f;
                        for (uint32_t ci = 0; ci < Cin; ++ci) {
                            for (uint32_t kd = 0; kd < Kd; ++kd) {
                                for (uint32_t kh = 0; kh < Kh; ++kh) {
                                    for (uint32_t kw = 0; kw < Kw; ++kw) {
                                        uint32_t id = od + kd;
                                        uint32_t ih = oh + kh;
                                        uint32_t iw = ow + kw;

                                        size_t x_idx = (((size_t)n * Cin + ci) * InD + id) * InH * InW
                                                     + ((size_t)ih * InW + iw);
                                        size_t w_idx = ((((size_t)co * Cin + ci) * Kd + kd) * Kh + kh) * Kw + kw;

                                        acc += x_gm[x_idx] * w_gm[w_idx];
                                    }
                                }
                            }
                        }
                        acc += b_gm[co];
                        float mult = m_gm[co];
                        float y2 = acc * mult;

                        size_t y2_idx = ((((size_t)n * Cout + co) * OutD + od) * OutH + oh) * OutW + ow;
                        ws0_y2[y2_idx] = y2;

                        sum_val += (double)y2;
                        sum_sq  += (double)y2 * (double)y2;
                    }
                }
            }

            double count = (double)out_spatial;
            float mean = (float)(sum_val / count);
            float var  = (float)((sum_sq / count) - (mean * mean));
            ws1_mean[(size_t)n * Cout + co] = mean;
            ws2_var[(size_t)n * Cout + co]  = var;
        }
    }

    // Pass 2: normalize, clamp, second multiply, channelwise max, write to y
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t od = 0; od < OutD; ++od) {
            for (uint32_t oh = 0; oh < OutH; ++oh) {
                for (uint32_t ow = 0; ow < OutW; ++ow) {
                    float max_val = -INFINITY;
                    for (uint32_t co = 0; co < Cout; ++co) {
                        size_t y2_idx = ((((size_t)n * Cout + co) * OutD + od) * OutH + oh) * OutW + ow;
                        float y2 = ws0_y2[y2_idx];

                        float mean = ws1_mean[(size_t)n * Cout + co];
                        float var  = ws2_var[(size_t)n * Cout + co];
                        float inv_std = 1.0f / sqrtf(var + eps);
                        float norm = (y2 - mean) * inv_std;

                        float clamped = norm;
                        if (clamped < cmin) clamped = cmin;
                        if (clamped > cmax) clamped = cmax;

                        float y3 = clamped * m_gm[co];

                        if (y3 > max_val) {
                            max_val = y3;
                        }
                    }
                    size_t y_idx = (((size_t)n * OutD + od) * OutH + oh) * OutW + ow;
                    y_gm[y_idx] = max_val;
                }
            }
        }
    }
}

import custom_ops_lib

class ModelNew(torch.nn.Module):

        super().__init__()
        # Hold parameters for conv and multiplier

        clamp_max_t = torch.tensor(self.clamp_max, dtype=x.dtype, device=x.device)
        return custom_ops_lib.conv3d_multiply_instance_norm_clamp_multiply_max_custom(
            x, self.conv.weight, self.conv.bias, self.multiplier, clamp_min_t, clamp_max_t
