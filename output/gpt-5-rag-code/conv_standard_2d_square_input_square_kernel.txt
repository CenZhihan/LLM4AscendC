kernel_src="""
#include "kernel_operator.h"
#include <cstdint>

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N      = tiling_data.n;
    const uint32_t C_in   = tiling_data.cIn;
    const uint32_t H_in   = tiling_data.hIn;
    const uint32_t W_in   = tiling_data.wIn;
    const uint32_t C_out  = tiling_data.cOut;
    const uint32_t K      = tiling_data.k;
    const uint32_t STRIDE = tiling_data.stride;
    const uint32_t PAD    = tiling_data.pad;
    const uint32_t DIL    = tiling_data.dilation;
    const uint32_t H_out  = tiling_data.hOut;
    const uint32_t W_out  = tiling_data.wOut;

    const float* __restrict__ x_ptr = reinterpret_cast<const float*>(x);
    const float* __restrict__ w_ptr = reinterpret_cast<const float*>(w);
    const float* __restrict__ b_ptr = reinterpret_cast<const float*>(b);
    float* __restrict__ y_ptr = reinterpret_cast<float*>(y);

    uint32_t block_num = AscendC::GetBlockNum();
    uint32_t block_idx = AscendC::GetBlockIdx();

    uint64_t totalOut = static_cast<uint64_t>(N) * C_out * H_out * W_out;
    uint64_t workPerBlock = (totalOut + block_num - 1) / block_num;
    uint64_t start = workPerBlock * block_idx;
    uint64_t end = start + workPerBlock;
    if (end > totalOut) end = totalOut;

    for (uint64_t linear = start; linear < end; ++linear) {
        uint64_t t = linear;
        uint32_t ow = static_cast<uint32_t>(t % W_out); t /= W_out;
        uint32_t oh = static_cast<uint32_t>(t % H_out); t /= H_out;
        uint32_t oc = static_cast<uint32_t>(t % C_out); t /= C_out;
        uint32_t n  = static_cast<uint32_t>(t);

        float acc = 0.0f;
        if (b_ptr != nullptr) {
            acc = b_ptr[oc];
        }

        int32_t base_h = static_cast<int32_t>(oh) * static_cast<int32_t>(STRIDE) - static_cast<int32_t>(PAD);
        int32_t base_w = static_cast<int32_t>(ow) * static_cast<int32_t>(STRIDE) - static_cast<int32_t>(PAD);

        for (uint32_t ci = 0; ci < C_in; ++ci) {
            for (uint32_t ky = 0; ky < K; ++ky) {
                int32_t ih = base_h + static_cast<int32_t>(ky) * static_cast<int32_t>(DIL);
                if (ih < 0 || ih >= static_cast<int32_t>(H_in)) continue;
                for (uint32_t kx = 0; kx < K; ++kx) {
                    int32_t iw = base_w + static_cast<int32_t>(kx) * static_cast<int32_t>(DIL);
                    if (iw < 0 || iw >= static_cast<int32_t>(W_in)) continue;

                    uint64_t x_off = (((static_cast<uint64_t>(n) * C_in + ci) * H_in + static_cast<uint32_t>(ih)) * W_in + static_cast<uint32_t>(iw));
                    uint64_t w_off = ((((static_cast<uint64_t>(oc) * C_in) + ci) * K + ky) * K + kx);
                    float xv = x_ptr[x_off];
                    float wv = w_ptr[w_off];
                    acc += xv * wv;
                }
            }
        }

        uint64_t y_off = (((static_cast<uint64_t>(n) * C_out + oc) * H_out + oh) * W_out + ow);
        y_ptr[y_off] = acc;
    }
}
