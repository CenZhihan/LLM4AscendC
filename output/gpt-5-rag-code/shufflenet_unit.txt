namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ShufflenetUnitCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    // Expecting 4D NCHW
    uint64_t n = in_shape->GetOriginShape().GetDim(0);
    uint64_t c = in_shape->GetOriginShape().GetDim(1);
    uint64_t h = in_shape->GetOriginShape().GetDim(2);
    uint64_t w = in_shape->GetOriginShape().GetDim(3);

    uint64_t totalPlanes = n * c;        // number of channel planes across batch
    uint64_t planeSize = h * w;          // elements per plane

    // Fetch groups attribute (default 1 if not set)
    uint32_t groups = 1;
    auto attr_groups = context->GetAttrs()->GetAttr("groups");
    if (attr_groups != nullptr) {
        groups = static_cast<uint32_t>(attr_groups->GetInt());
    }

    context->SetBlockDim(BLOCK_DIM);

    // distribute planes among blocks (ceil-div)
    uint32_t blockPlanes = static_cast<uint32_t>((totalPlanes + BLOCK_DIM - 1) / BLOCK_DIM);

    tiling.set_totalPlanes(static_cast<uint32_t>(totalPlanes));
    tiling.set_planeSize(static_cast<uint32_t>(planeSize));
    tiling.set_channels(static_cast<uint32_t>(c));
    tiling.set_groups(groups);
    tiling.set_blockPlanes(blockPlanes);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // pipeline buffers

class KernelShuffleUnit {
public:
    __aicore__ inline KernelShuffleUnit() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalPlanes, uint32_t planeSize,
                                uint32_t channels, uint32_t groups,
                                uint32_t blockPlanes)
    {
        this->totalPlanes = totalPlanes;
        this->planeSize = planeSize;
        this->channels = channels;
        this->groups = groups;
        this->blockPlanes = blockPlanes;

        this->blockStartPlane = AscendC::GetBlockIdx() * this->blockPlanes;
        uint32_t remain = (this->totalPlanes > this->blockStartPlane) ? (this->totalPlanes - this->blockStartPlane) : 0;
        this->loopCount = remain < this->blockPlanes ? remain : this->blockPlanes;

        // process one plane per buffer slot
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint64_t>(this->totalPlanes) * this->planeSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, static_cast<uint64_t>(this->totalPlanes) * this->planeSize);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->planeSize * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < this->loopCount; ++i) {
            CopyIn(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t i)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        uint64_t planeIdx = static_cast<uint64_t>(this->blockStartPlane + i);
        AscendC::DataCopy(xLocal, xGm[planeIdx * this->planeSize], this->planeSize);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t i)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        // compute output plane index mapping
        uint32_t planeIdx = this->blockStartPlane + i;
        uint32_t n = planeIdx / this->channels;
        uint32_t c = planeIdx % this->channels;
        uint32_t channelsPerGroup = this->channels / this->groups;
        uint32_t newc = (c % channelsPerGroup) * this->groups + (c / channelsPerGroup);
        uint64_t outPlaneIdx = static_cast<uint64_t>(n) * this->channels + newc;

        // write to output plane
        AscendC::DataCopy(yGm[outPlaneIdx * this->planeSize], xLocal, this->planeSize);

        inQueueX.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalPlanes;
    uint32_t planeSize;
    uint32_t channels;
    uint32_t groups;
    uint32_t blockPlanes;
    uint32_t blockStartPlane;
    uint32_t loopCount;
};

extern "C" __global__ __aicore__ void shufflenet_unit_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelShuffleUnit op;
    op.Init(x, y,
            tiling_data.totalPlanes,
            tiling_data.planeSize,
            tiling_data.channels,
            tiling_data.groups,
            tiling_data.blockPlanes);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ChannelShuffle(nn.Module):
    def __init__(self, groups):
        super(ChannelShuffle, self).__init__()
        self.groups = groups

    def forward(self, x):
        # This module is kept for reference but not used in ModelNew
        batch_size, channels, height, width = x.size()
        channels_per_group = channels // self.groups
        x = x.view(batch_size, self.groups, channels_per_group, height, width)
        x = x.transpose(1, 2).contiguous()
        x = x.view(batch_size, -1, height, width)
        return x

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, groups=3):
        super(ModelNew, self).__init__()
        assert out_channels % 4 == 0
        mid_channels = out_channels // 4

        self.groups = groups

        self.conv1 = nn.Conv2d(in_channels, mid_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn1 = nn.BatchNorm2d(mid_channels)

        self.conv2 = nn.Conv2d(mid_channels, mid_channels, kernel_size=3, stride=1, padding=1, groups=mid_channels, bias=False)
        self.bn2 = nn.BatchNorm2d(mid_channels)

        self.conv3 = nn.Conv2d(mid_channels, out_channels, kernel_size=1, stride=1, padding=0, groups=groups, bias=False)
        self.bn3 = nn.BatchNorm2d(out_channels)

        if in_channels == out_channels:
            self.shortcut = nn.Sequential()
        else:
            self.shortcut = nn.Sequential(
                nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=False),
                nn.BatchNorm2d(out_channels)
            )

    def forward(self, x):
        out = F.relu(self.bn1(self.conv1(x)))
        out = self.bn2(self.conv2(out))
        # Replace ChannelShuffle with custom AscendC kernel
        out = custom_ops_lib.shufflenet_unit_custom(out, self.groups)
        out = F.relu(self.bn3(self.conv3(out)))
        out = out + self.shortcut(x)
        return out
'''
