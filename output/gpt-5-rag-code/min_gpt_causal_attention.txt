kernel_src="""
#include "kernel_operator.h"

constexpr int32_t QKV_BUFFER_NUM = 1;

class KernelMinGptCausalAttention {
public:
    __aicore__ inline KernelMinGptCausalAttention() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y,
                                uint32_t b, uint32_t h, uint32_t t, uint32_t hs, float scale)
    {
        this->B = b;
        this->H = h;
        this->T = t;
        this->HS = hs;
        this->scale = scale;

        this->totalQueries = b * h * t;

        // Set GM buffers to full ranges
        qGm.SetGlobalBuffer((__gm__ DTYPE_X *)q, (uint64_t)this->totalQueries * this->HS);
        kGm.SetGlobalBuffer((__gm__ DTYPE_X *)k, (uint64_t)this->totalQueries * this->HS);
        vGm.SetGlobalBuffer((__gm__ DTYPE_X *)v, (uint64_t)this->totalQueries * this->HS);
        yGm.SetGlobalBuffer((__gm__ DTYPE_X *)y, (uint64_t)this->totalQueries * this->HS);

        // Local buffers for a single vector [HS]
        pipe.InitBuffer(qInQueue, QKV_BUFFER_NUM, this->HS * sizeof(DTYPE_X));
        pipe.InitBuffer(kInQueue, QKV_BUFFER_NUM, this->HS * sizeof(DTYPE_X));
        pipe.InitBuffer(vInQueue, QKV_BUFFER_NUM, this->HS * sizeof(DTYPE_X));
        pipe.InitBuffer(yOutQueue, QKV_BUFFER_NUM, this->HS * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint64_t queriesPerBlock = this->totalQueries / blockNum;
        uint64_t start = queriesPerBlock * blockIdx;
        uint64_t end = (blockIdx == blockNum - 1) ? this->totalQueries : (start + queriesPerBlock);

        for (uint64_t n = start; n < end; ++n) {
            // Decode indices
            uint64_t bhSpan = (uint64_t)H * (uint64_t)T;
            uint32_t b = static_cast<uint32_t>(n / bhSpan);
            uint64_t rem1 = n % bhSpan;
            uint32_t h = static_cast<uint32_t>(rem1 / T);
            uint32_t tq = static_cast<uint32_t>(rem1 % T);

            // Copy in q[tq]
            AscendC::LocalTensor<DTYPE_X> qLocal = qInQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(qLocal, qGm[n * (uint64_t)HS], HS);
            qInQueue.EnQue(qLocal);
            qLocal = qInQueue.DeQue<DTYPE_X>();

            // Pass 1: find max logit for numerical stability
            float maxLogit = -3.4e38f;
            for (uint32_t tk = 0; tk <= tq; ++tk) {
                uint64_t kvIndex = ((((uint64_t)b * H + (uint64_t)h) * T) + (uint64_t)tk) * (uint64_t)HS;
                AscendC::LocalTensor<DTYPE_X> kLocal = kInQueue.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(kLocal, kGm[kvIndex], HS);
                float dotVal = 0.0f;
                for (uint32_t j = 0; j < HS; ++j) {
                    dotVal += static_cast<float>(qLocal[j]) * static_cast<float>(kLocal[j]);
                }
                dotVal *= this->scale;
                if (dotVal > maxLogit) maxLogit = dotVal;
                kInQueue.FreeTensor(kLocal);
            }

            // Pass 2: compute sum of exp(logits - maxLogit)
            float sumExp = 0.0f;
            for (uint32_t tk = 0; tk <= tq; ++tk) {
                uint64_t kvIndex = ((((uint64_t)b * H + (uint64_t)h) * T) + (uint64_t)tk) * (uint64_t)HS;
                AscendC::LocalTensor<DTYPE_X> kLocal = kInQueue.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(kLocal, kGm[kvIndex], HS);
                float dotVal = 0.0f;
                for (uint32_t j = 0; j < HS; ++j) {
                    dotVal += static_cast<float>(qLocal[j]) * static_cast<float>(kLocal[j]);
                }
                dotVal *= this->scale;
                float expVal = expf(dotVal - maxLogit);
                sumExp += expVal;
                kInQueue.FreeTensor(kLocal);
            }

            // Pass 3: accumulate y = sum softmax * v
            AscendC::LocalTensor<DTYPE_X> yLocal = yOutQueue.AllocTensor<DTYPE_X>();
            for (uint32_t j = 0; j < HS; ++j) {
                yLocal[j] = static_cast<DTYPE_X>(0.0f);
            }

            for (uint32_t tk = 0; tk <= tq; ++tk) {
                uint64_t kvIndex = ((((uint64_t)b * H + (uint64_t)h) * T) + (uint64_t)tk) * (uint64_t)HS;

                AscendC::LocalTensor<DTYPE_X> kLocal = kInQueue.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(kLocal, kGm[kvIndex], HS);
                float dotVal = 0.0f;
                for (uint32_t j = 0; j < HS; ++j) {
                    dotVal += static_cast<float>(qLocal[j]) * static_cast<float>(kLocal[j]);
                }
                dotVal *= this->scale;
                float weight = expf(dotVal - maxLogit) / sumExp;
                kInQueue.FreeTensor(kLocal);

                AscendC::LocalTensor<DTYPE_X> vLocal = vInQueue.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(vLocal, vGm[kvIndex], HS);
                for (uint32_t j = 0; j < HS; ++j) {
                    float acc = static_cast<float>(yLocal[j]) + weight * static_cast<float>(vLocal[j]);
                    yLocal[j] = static_cast<DTYPE_X>(acc);
                }
                vInQueue.FreeTensor(vLocal);
            }

            yOutQueue.EnQue<DTYPE_X>(yLocal);
            AscendC::LocalTensor<DTYPE_X> yLocalOut = yOutQueue.DeQue<DTYPE_X>();
            AscendC::DataCopy(yGm[n * (uint64_t)HS], yLocalOut, HS);
            yOutQueue.FreeTensor(yLocalOut);

            qInQueue.FreeTensor(qLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, QKV_BUFFER_NUM> qInQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, QKV_BUFFER_NUM> kInQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, QKV_BUFFER_NUM> vInQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, QKV_BUFFER_NUM> yOutQueue;
    AscendC::GlobalTensor<DTYPE_X> qGm;
    AscendC::GlobalTensor<DTYPE_X> kGm;
    AscendC::GlobalTensor<DTYPE_X> vGm;
    AscendC::GlobalTensor<DTYPE_X> yGm;

    uint32_t B;
    uint32_t H;
    uint32_t T;
    uint32_t HS;
    uint64_t totalQueries;
    float scale;
};

extern "C" __global__ __aicore__ void min_gpt_causal_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMinGptCausalAttention op;
    op.Init(q, k, v, y,
            tiling_data.b, tiling_data.h, tiling_data.t, tiling_data.hs, tiling_data.scale);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import math
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Optimized masked self-attention using a custom AscendC kernel for the core attention computation.
    """
    def __init__(self, n_embd, n_head, attn_pdrop, resid_pdrop, max_seqlen):
        super().__init__()
        assert n_embd % n_head == 0
        self.c_attn = nn.Linear(n_embd, 3 * n_embd)
        self.c_proj = nn.Linear(n_embd, n_embd)
        self.attn_dropout = nn.Dropout(attn_pdrop)
        self.resid_dropout = nn.Dropout(resid_pdrop)
        self.register_buffer("bias", torch.tril(torch.ones(max_seqlen, max_seqlen))
                                     .view(1, 1, max_seqlen, max_seqlen))
        self.n_head = n_head
        self.n_embd = n_embd

    def forward(self, x):
        B, T, C = x.size()
        nh = self.n_head
        hs = C // nh

        # Compute Q, K, V
        qkv = self.c_attn(x)
        q, k, v = qkv.split(self.n_embd, dim=2)

        # Reshape to [B, nh, T, hs] and ensure contiguous layout
        k = k.view(B, T, nh, hs).transpose(1, 2).contiguous()
        q = q.view(B, T, nh, hs).transpose(1, 2).contiguous()
        v = v.view(B, T, nh, hs).transpose(1, 2).contiguous()

        # Custom kernel: causal masked scaled dot-product attention: y = Softmax((q k^T)/sqrt(hs) + mask) @ v
        # Kernel expects inputs [B, nh, T, hs] and returns [B, nh, T, hs]
        y = custom_ops_lib.min_gpt_causal_attention_custom(q, k, v)

        # Re-assemble heads and apply output projection
        y = y.transpose(1, 2).contiguous().view(B, T, C)
        y = self.resid_dropout(self.c_proj(y))
        return y
'''
