kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// This kernel computes a naive 2D convolution with stride=1, padding=0,
// followed by HardSwish and ReLU on the output.
class KernelConv2dHardSwishRelu {
public:
    __aicore__ inline KernelConv2dHardSwishRelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t Sh, uint32_t Sw, uint32_t Ph, uint32_t Pw,
                                uint32_t Hout, uint32_t Wout,
                                uint32_t totalLength, uint32_t tileNum)
    {
        // Set up parameters
        this->N = N; this->Cin = Cin; this->H = H; this->W = W;
        this->Cout = Cout; this->Kh = Kh; this->Kw = Kw;
        this->Sh = Sh; this->Sw = Sw; this->Ph = Ph; this->Pw = Pw;
        this->Hout = Hout; this->Wout = Wout;

        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = (this->blockLength / tileNum / BUFFER_NUM);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->xBlockBaseOffset(), this->xBlockSpan());
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint32_t)(Cout * Cin * Kh * Kw));
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b, (uint32_t)(Cout));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        // small scalar buffers for on-demand loads
        pipe.InitBuffer(scalarBufX, 1, sizeof(DTYPE_X));
        pipe.InitBuffer(scalarBufW, 1, sizeof(DTYPE_W));
        pipe.InitBuffer(scalarBufB, 1, sizeof(DTYPE_B));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline uint32_t xBlockBaseOffset() const
    {
        // The X tensor is reused across all outputs in a block; to simplify, we use full X span for GM view.
        // Return 0 so full X is visible (we set span large enough in Init).
        return 0;
    }
    __aicore__ inline uint32_t xBlockSpan() const
    {
        return (uint32_t)(N * Cin * H * W);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        uint32_t base = progress * this->tileLength;

        for (uint32_t idx = 0; idx < this->tileLength; ++idx) {
            uint32_t outLinear = base + idx;

            if (outLinear >= this->blockLength) {
                break;
            }

            // Map linear index (within this block) to (n, co, ho, wo)
            uint32_t globalLinear = this->blockLength * AscendC::GetBlockIdx() + outLinear;
            uint32_t wo = globalLinear % Wout;
            uint32_t tmp1 = globalLinear / Wout;
            uint32_t ho = tmp1 % Hout;
            uint32_t tmp2 = tmp1 / Hout;
            uint32_t co = tmp2 % Cout;
            uint32_t n  = tmp2 / Cout;

            // Accumulator
            float acc = 0.0f;

            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t kh = 0; kh < Kh; ++kh) {
                    int32_t hi = (int32_t)ho * (int32_t)Sh - (int32_t)Ph + (int32_t)kh;
                    if (hi < 0 || hi >= (int32_t)H) continue;
                    for (uint32_t kw = 0; kw < Kw; ++kw) {
                        int32_t wi = (int32_t)wo * (int32_t)Sw - (int32_t)Pw + (int32_t)kw;
                        if (wi < 0 || wi >= (int32_t)W) continue;

                        // Offsets
                        uint32_t xOff = (((n * Cin + ci) * H + (uint32_t)hi) * W + (uint32_t)wi);
                        uint32_t wOff = (((co * Cin + ci) * Kh + kh) * Kw + kw);

                        // Load scalar x and w
                        AscendC::LocalTensor<DTYPE_X> xScalar = scalarBufX.AllocTensor<DTYPE_X>();
                        AscendC::LocalTensor<DTYPE_W> wScalar = scalarBufW.AllocTensor<DTYPE_W>();
                        AscendC::DataCopy(xScalar, xGm[xOff], 1);
                        AscendC::DataCopy(wScalar, wGm[wOff], 1);
                        float xv = static_cast<float>(xScalar.GetValue(0));
                        float wv = static_cast<float>(wScalar.GetValue(0));
                        acc += xv * wv;
                        scalarBufX.FreeTensor(xScalar);
                        scalarBufW.FreeTensor(wScalar);
                    }
                }
            }

            // Add bias
            AscendC::LocalTensor<DTYPE_B> bScalar = scalarBufB.AllocTensor<DTYPE_B>();
            AscendC::DataCopy(bScalar, bGm[co], 1);
            float bv = static_cast<float>(bScalar.GetValue(0));
            scalarBufB.FreeTensor(bScalar);
            float val = acc + bv;

            // HardSwish: y = x * relu6(x + 3) / 6
            float t = val + 3.0f;
            float relu6_t = t < 0.0f ? 0.0f : (t > 6.0f ? 6.0f : t);
            float hsw = val * (relu6_t / 6.0f);

            // ReLU
            float outv = hsw > 0.0f ? hsw : 0.0f;

            yLocal.SetValue(idx, static_cast<DTYPE_Y>(outv));
        }
        outQueueY.EnQue<DTYPE_Y>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scalarBufX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scalarBufW;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scalarBufB;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, Cin, H, W;
    uint32_t Cout, Kh, Kw;
    uint32_t Sh, Sw, Ph, Pw;
    uint32_t Hout, Wout;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_hard_swish_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dHardSwishRelu op;
    op.Init(
        x, w, b, y,
        tiling_data.n(), tiling_data.c_in(), tiling_data.h(), tiling_data.w(),
        tiling_data.c_out(), tiling_data.k_h(), tiling_data.k_w(),
        tiling_data.stride_h(), tiling_data.stride_w(),
        tiling_data.pad_h(), tiling_data.pad_w(),
        tiling_data.h_out(), tiling_data.w_out(),
        tiling_data.totalLength(), tiling_data.tileNum()
    );
    op.Process();
}

    int64_t W = x.size(3);
    int64_t Cout = w.size(0);
    int64_t Kh = w.size(2);
    int64_t Kw = w.size(3);

    TORCH_CHECK(b.size(0) == Cout, "bias size mismatch with weight");

    EXEC_NPU_CMD(aclnnConv2dHardSwishReluCustom, x, w, b, y);
