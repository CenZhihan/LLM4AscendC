kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

__aicore__ inline float fast_tanh_approx(float x) {
    // tanh approximation: x*(27 + x^2)/(27 + 9*x^2)
    float x2 = x * x;
    float num = x * (27.0f + x2);
    float den = (27.0f + 9.0f * x2);
    return num / den;
}

__aicore__ inline float fast_gelu(float x) {
    // Approximate GELU via tanh formulation
    const float k0 = 0.7978845608f; // sqrt(2/pi)
    float x3 = x * x * x;
    float inner = k0 * (x + 0.044715f * x3);
    float t = fast_tanh_approx(inner);
    return 0.5f * x * (1.0f + t);
}

class KernelGemmSubtractGlobalAvgPoolLogSumExpGeluResidualAdd {
public:
    __aicore__ inline KernelGemmSubtractGlobalAvgPoolLogSumExpGeluResidualAdd() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w_mean, GM_ADDR c_scalar, GM_ADDR z,
                                uint32_t batch, uint32_t inFeatures, uint32_t tileK)
    {
        this->batch = batch;
        this->inFeatures = inFeatures;
        this->tileK = tileK;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)batch * inFeatures);
        wMeanGm.SetGlobalBuffer((__gm__ float*)w_mean, inFeatures);
        cGm.SetGlobalBuffer((__gm__ float*)c_scalar, 1);
        zGm.SetGlobalBuffer((__gm__ float*)z, (uint64_t)batch * inFeatures);

        pipe.InitBuffer(xTileQ, BUFFER_NUM, tileK * sizeof(float));
        pipe.InitBuffer(wTileQ, BUFFER_NUM, tileK * sizeof(float));
        pipe.InitBuffer(outTileQ, BUFFER_NUM, tileK * sizeof(float));
        pipe.InitBuffer(cBufQ, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t rowsPerBlock = (batch + blockNum - 1) / blockNum;
        uint32_t rowStart = rowsPerBlock * blockIdx;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > batch) rowEnd = batch;

        // load c scalar
        AscendC::LocalTensor<float> cLocal = cBufQ.AllocTensor<float>();
        AscendC::DataCopy(cLocal, cGm[0], 1);
        cBufQ.EnQue(cLocal);
        cLocal = cBufQ.DeQue<float>();
        float cVal = cLocal.GetValue(0); // assume element access supported
        cBufQ.FreeTensor(cLocal);

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            float m = 0.0f;

            // dot(x_row, w_mean)
            uint32_t processed = 0;
            while (processed < inFeatures) {
                uint32_t len = AscendC::min(this->tileK, inFeatures - processed);

                AscendC::LocalTensor<float> xLocal = xTileQ.AllocTensor<float>();
                AscendC::LocalTensor<float> wLocal = wTileQ.AllocTensor<float>();

                AscendC::DataCopy(xLocal, xGm[(uint64_t)row * inFeatures + processed], len);
                AscendC::DataCopy(wLocal, wMeanGm[processed], len);

                xTileQ.EnQue<float>(xLocal);
                wTileQ.EnQue<float>(wLocal);

                xLocal = xTileQ.DeQue<float>();
                wLocal = wTileQ.DeQue<float>();

                // accumulate partial dot
                for (uint32_t i = 0; i < len; ++i) {
                    m += xLocal.GetValue(i) * wLocal.GetValue(i);
                }

                xTileQ.FreeTensor(xLocal);
                wTileQ.FreeTensor(wLocal);

                processed += len;
            }

            // GlobalAvgPool + LogSumExp simplifies to identity here; add c and GELU
            float g = fast_gelu(m + cVal);

            // Write z_row = x_row + g (broadcast)
            uint32_t written = 0;
            while (written < inFeatures) {
                uint32_t len = AscendC::min(this->tileK, inFeatures - written);

                AscendC::LocalTensor<float> xLocal = xTileQ.AllocTensor<float>();
                AscendC::LocalTensor<float> zLocal = outTileQ.AllocTensor<float>();

                AscendC::DataCopy(xLocal, xGm[(uint64_t)row * inFeatures + written], len);

                // z = x + g
                for (uint32_t i = 0; i < len; ++i) {
                    zLocal.SetValue(i, xLocal.GetValue(i) + g);
                }

                outTileQ.EnQue<float>(zLocal);
                zLocal = outTileQ.DeQue<float>();
                AscendC::DataCopy(zGm[(uint64_t)row * inFeatures + written], zLocal, len);
                outTileQ.FreeTensor(zLocal);
                xTileQ.FreeTensor(xLocal);

                written += len;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xTileQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wTileQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> cBufQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outTileQ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wMeanGm;
    AscendC::GlobalTensor<float> cGm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t batch;
    uint32_t inFeatures;
    uint32_t tileK;
};

extern "C" __global__ __aicore__ void gemm_subtract_global_avg_pool_log_sum_exp_gelu_residual_add_custom(GM_ADDR x, GM_ADDR w_mean, GM_ADDR c_scalar, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    op.Init(x, w_mean, c_scalar, z, tiling_data.batch, tiling_data.inFeatures, tiling_data.tileK);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a fused custom AscendC kernel.
    The fusion leverages linearity to avoid materializing the full GEMM output before reduction.
    """
    def __init__(self, in_features, out_features, bias=True):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features, bias=bias)
        self.subtract = torch.nn.Parameter(torch.randn(out_features))

    def forward(self, x):
        # Precompute fused parameters:
        # mean over output features of weights per input feature
        w_mean = self.gemm.weight.mean(dim=0)  # [in_features]
        if self.gemm.bias is not None:
            c_scalar = (self.gemm.bias - self.subtract).mean()  # scalar
        else:
            c_scalar = (-self.subtract).mean()

        # Ensure shapes for inputs
        w_mean = w_mean.contiguous()
        c_scalar = c_scalar.view(1).contiguous()

        # Fused kernel computes:
        # m = x @ w_mean + c_scalar
        # y = gelu(m)
        # z = x + y (broadcast)
        return custom_ops_lib.gemm_subtract_global_avg_pool_log_sum_exp_gelu_residual_add_custom(x, w_mean, c_scalar)
