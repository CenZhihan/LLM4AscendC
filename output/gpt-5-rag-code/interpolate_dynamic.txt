#include "interpolate_dynamic_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    InterpolateDynamicCustomTilingData tiling;

    // Input x: NCHW
    const gert::Shape* in_shape = context->GetInputShape(0);
    uint64_t n = 1, c = 1, inH = 1, inW = 1;
    if (in_shape->GetDimNum() >= 4) {
        n   = in_shape->GetDim(0);
        c   = in_shape->GetDim(1);
        inH = in_shape->GetDim(2);
        inW = in_shape->GetDim(3);
    }

    // Output shape should be pre-allocated by frontend (PyTorch binding),
    // so we can read target outH/outW here.
    const gert::Shape* out_shape = context->GetOutputShape(0);
    uint64_t outH = inH;
    uint64_t outW = inW;
    if (out_shape->GetDimNum() >= 4) {
        outH = out_shape->GetDim(2);
        outW = out_shape->GetDim(3);
    }

    uint64_t totalOut = n * c * outH * outW;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(static_cast<uint32_t>(n));
    tiling.set_c(static_cast<uint32_t>(c));
    tiling.set_inH(static_cast<uint32_t>(inH));
    tiling.set_inW(static_cast<uint32_t>(inW));
    tiling.set_outH(static_cast<uint32_t>(outH));
    tiling.set_outW(static_cast<uint32_t>(outW));
    tiling.set_totalOut(totalOut);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Keep output same as input by default; frontend pre-allocates correct output shape.
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

class KernelInterpolateDynamic {
public:
    __aicore__ inline KernelInterpolateDynamic() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR size, GM_ADDR z,
                                uint32_t n, uint32_t c,
                                uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW,
                                uint64_t totalOut)
    {
        xPtr = (__gm__ DTYPE_X*)x;
        zPtr = (__gm__ DTYPE_Z*)z;
        n_ = n; c_ = c; inH_ = inH; inW_ = inW; outH_ = outH; outW_ = outW; totalOut_ = totalOut;

        // Block-based partition
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint64_t perBlock = (totalOut_ + blockNum - 1) / blockNum;
        start_ = perBlock * blockIdx;
        end_ = AscendC::Min(totalOut_, (uint64_t)perBlock * (blockIdx + 1));
    }

    __aicore__ inline void Process()
    {
        // Bilinear resize, align_corners=False
        float scale_h = (outH_ > 0) ? (static_cast<float>(inH_) / static_cast<float>(outH_)) : 0.0f;
        float scale_w = (outW_ > 0) ? (static_cast<float>(inW_) / static_cast<float>(outW_)) : 0.0f;

        for (uint64_t idx = start_; idx < end_; ++idx) {
            // unravel index into n,c,oh,ow in NCHW
            uint64_t tmp = idx;
            uint32_t ow = tmp % outW_; tmp /= outW_;
            uint32_t oh = tmp % outH_; tmp /= outH_;
            uint32_t c  = tmp % c_;    tmp /= c_;
            uint32_t n  = tmp;

            float src_h_f = (static_cast<float>(oh) + 0.5f) * scale_h - 0.5f;
            float src_w_f = (static_cast<float>(ow) + 0.5f) * scale_w - 0.5f;

            if (src_h_f < 0.0f) src_h_f = 0.0f;
            if (src_w_f < 0.0f) src_w_f = 0.0f;

            int32_t h0 = static_cast<int32_t>(AscendC::Floor(src_h_f));
            int32_t w0 = static_cast<int32_t>(AscendC::Floor(src_w_f));
            int32_t h1 = h0 + 1;
            int32_t w1 = w0 + 1;

            if (h0 < 0) h0 = 0;
            if (w0 < 0) w0 = 0;
            if (h1 >= (int32_t)inH_) h1 = (int32_t)inH_ - 1;
            if (w1 >= (int32_t)inW_) w1 = (int32_t)inW_ - 1;

            float dh = src_h_f - static_cast<float>(h0);
            float dw = src_w_f - static_cast<float>(w0);
            float h0w0_w = (1.0f - dh) * (1.0f - dw);
            float h0w1_w = (1.0f - dh) * dw;
            float h1w0_w = dh * (1.0f - dw);
            float h1w1_w = dh * dw;

            uint64_t base_nc = ((uint64_t)n * c_ + c) * inH_ * inW_;
            uint64_t idx00 = base_nc + (uint64_t)h0 * inW_ + (uint64_t)w0;
            uint64_t idx01 = base_nc + (uint64_t)h0 * inW_ + (uint64_t)w1;
            uint64_t idx10 = base_nc + (uint64_t)h1 * inW_ + (uint64_t)w0;
            uint64_t idx11 = base_nc + (uint64_t)h1 * inW_ + (uint64_t)w1;

            float v00 = static_cast<float>(xPtr[idx00]);
            float v01 = static_cast<float>(xPtr[idx01]);
            float v10 = static_cast<float>(xPtr[idx10]);
            float v11 = static_cast<float>(xPtr[idx11]);

            float outv = v00 * h0w0_w + v01 * h0w1_w + v10 * h1w0_w + v11 * h1w1_w;

            uint64_t outBase_nc = ((uint64_t)n * c_ + c) * outH_ * outW_;
            uint64_t outIdx = outBase_nc + (uint64_t)oh * outW_ + (uint64_t)ow;
            zPtr[outIdx] = (DTYPE_Z)outv;
        }
    }

private:
    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_Z* zPtr;
    uint32_t n_, c_, inH_, inW_, outH_, outW_;
    uint64_t totalOut_;
    uint64_t start_, end_;
};

extern "C" __global__ __aicore__ void interpolate_dynamic_custom(GM_ADDR x, GM_ADDR size, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelInterpolateDynamic op;
    op.Init(x, size, z,
            tiling_data.n, tiling_data.c,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.totalOut);
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor interpolate_dynamic_custom_impl_npu(const at::Tensor& self, at::IntArrayRef out_size) {
    TORCH_CHECK(out_size.size() == 2, \"size must be (outH, outW)\");
    auto N = self.size(0);
    auto C = self.size(1);
    auto outH = out_size[0];
    auto outW = out_size[1];
    at::Tensor result = at::empty({N, C, outH, outW}, self.options());
    EXEC_NPU_CMD(aclnnInterpolateDynamicCustom, self, out_size, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"interpolate_dynamic_custom\", &interpolate_dynamic_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"interpolate_dynamic_custom\", &interpolate_dynamic_custom_impl_npu, \"bilinear interpolate dynamic size\");
}
