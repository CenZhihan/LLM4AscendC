    uint32_t N = static_cast<uint32_t>(a_shape->GetStorageShape().GetDim(0));

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    return ge::GRAPH_SUCCESS;
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    // Basic checks and propagate to output
    if (a_shape->GetStorageShape().GetDimNum() != 2 || b_shape->GetStorageShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    if (a_shape->GetStorageShape().GetDim(1) != b_shape->GetStorageShape().GetDim(0) ||
        a_shape->GetStorageShape().GetDim(0) != a_shape->GetStorageShape().GetDim(1) ||
        b_shape->GetStorageShape().GetDim(0) != b_shape->GetStorageShape().GetDim(1)) {
        return ge::GRAPH_FAILED;
    }
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *a_shape;
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t N = tiling_data.N;

    // Partition rows among blocks
    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t rowsPerBlock = (N + blockNum - 1) / blockNum;
    const uint32_t rowStart = blockIdx * rowsPerBlock;
    const uint32_t rowEnd = (rowStart + rowsPerBlock) > N ? N : (rowStart + rowsPerBlock);

    __gm__ float* aPtr = reinterpret_cast<__gm__ float*>(A);
    __gm__ float* bPtr = reinterpret_cast<__gm__ float*>(B);
    __gm__ float* cPtr = reinterpret_cast<__gm__ float*>(C);

    for (uint32_t i = rowStart; i < rowEnd; ++i) {
        const uint64_t rowBaseA = static_cast<uint64_t>(i) * N;
        const uint64_t rowBaseC = static_cast<uint64_t>(i) * N;
        // Upper triangle set to zero
        for (uint32_t j = i + 1; j < N; ++j) {
            cPtr[rowBaseC + j] = 0.0f;
        }
        // Compute lower triangle entries for row i
        for (uint32_t j = 0; j <= i; ++j) {
            float acc = 0.0f;
            // Since A and B are lower-triangular, only k in [j, i] contribute
            for (uint32_t k = j; k <= i; ++k) {
                acc += aPtr[rowBaseA + k] * bPtr[static_cast<uint64_t>(k) * N + j];
            }
            cPtr[rowBaseC + j] = acc;
        }
    }
}
