kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // buffer num for queues

class KernelConv3dMaxLogSumExpRelu {
public:
    __aicore__ inline KernelConv3dMaxLogSumExpRelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z,
                                uint32_t n, uint32_t c, uint32_t d,
                                uint32_t h, uint32_t w, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->d = d;
        this->h = h;
        this->w = w;
        this->tileNum = tileNum;

        this->S = static_cast<uint32_t>(d) * static_cast<uint32_t>(h) * static_cast<uint32_t>(w); // spatial size
        this->positionsTotal = static_cast<uint32_t>(n) * this->S;

        uint32_t blockNum = AscendC::GetBlockNum();
        this->blockPositions = (positionsTotal + blockNum - 1) / blockNum; // simple block split
        this->tileLength = AscendC::Max((uint32_t)1, this->blockPositions / (this->tileNum * BUFFER_NUM));

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + (AscendC::GetBlockIdx() * this->blockPositions * this->c), (uint32_t)0);
        // output is shape (N,1,D,H,W), contiguous in positions
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + (AscendC::GetBlockIdx() * this->blockPositions), (uint32_t)0);

        pipe.InitBuffer(tmpInQueue, BUFFER_NUM, sizeof(DTYPE_X));                 // for scalar loads
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z)); // for a tile of outputs
    }

    __aicore__ inline void Process()
    {
        uint32_t loopCount = this->tileNum * BUFFER_NUM;
        for (uint32_t i = 0; i < loopCount; ++i) {
            uint32_t startPos = i * this->tileLength;
            if (startPos >= this->blockPositions) {
                break;
            }
            uint32_t currTile = AscendC::Min(this->tileLength, this->blockPositions - startPos);
            ComputeTile(startPos, currTile);
        }
    }

private:
    __aicore__ inline DTYPE_X LoadScalar(uint64_t idx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = tmpInQueue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[idx], 1);
        DTYPE_X val = xLocal.GetValue(0);
        tmpInQueue.FreeTensor(xLocal);
        return val;
    }

    __aicore__ inline void ComputeTile(uint32_t startPos, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        for (uint32_t t = 0; t < len; ++t) {
            uint32_t pos = startPos + t; // linear index over positions (n * S + s), within this block
            uint32_t globalPos = pos + AscendC::GetBlockIdx() * this->blockPositions; // absolute position across all blocks
            if (globalPos >= positionsTotal) {
                zLocal.SetValue(t, static_cast<DTYPE_Z>(0));
                continue;
            }

            uint32_t n_idx = globalPos / this->S;
            uint32_t s_idx = globalPos % this->S;

            // First pass: find max over channels for numeric stability
            DTYPE_X maxVal = static_cast<DTYPE_X>(-1.0e20f);
            uint64_t base_n = static_cast<uint64_t>(n_idx) * static_cast<uint64_t>(this->c) * static_cast<uint64_t>(this->S);
            for (uint32_t cc = 0; cc < this->c; ++cc) {
                uint64_t idx = base_n + static_cast<uint64_t>(cc) * static_cast<uint64_t>(this->S) + static_cast<uint64_t>(s_idx);
                DTYPE_X v = LoadScalar(idx);
                maxVal = v > maxVal ? v : maxVal;
            }

            // Second pass: sum exp(x - maxVal)
            DTYPE_X sumExp = static_cast<DTYPE_X>(0);
            for (uint32_t cc = 0; cc < this->c; ++cc) {
                uint64_t idx = base_n + static_cast<uint64_t>(cc) * static_cast<uint64_t>(this->S) + static_cast<uint64_t>(s_idx);
                DTYPE_X v = LoadScalar(idx);
                DTYPE_X ex = AscendC::Exp(v - maxVal);
                sumExp = sumExp + ex;
            }

            DTYPE_Z lse = maxVal + AscendC::Log(sumExp);
            // ReLU
            DTYPE_Z relu_v = lse > static_cast<DTYPE_Z>(0) ? lse : static_cast<DTYPE_Z>(0);
            zLocal.SetValue(t, relu_v);
        }

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        // write back contiguous results for this tile
        AscendC::DataCopy(zGm[startPos], zLocal, len);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpInQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t n;
    uint32_t c;
    uint32_t d;
    uint32_t h;
    uint32_t w;
    uint32_t S; // D*H*W
    uint32_t positionsTotal; // N*S
    uint32_t blockPositions;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv3d_max_log_sum_exp_relu_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMaxLogSumExpRelu op;
    op.Init(x, z,
            tiling_data.n(), tiling_data.c(), tiling_data.d(),
            tiling_data.h(), tiling_data.w(), tiling_data.tileNum());
    op.Process();
}

static inline std::vector<int64_t> to_3tuple(const c10::List<int64_t>& l, int64_t default_val=1) {
    std::vector<int64_t> out(3, default_val);
    for (size_t i = 0; i < l.size() && i < 3; ++i) out[i] = l.get(i);
    return out;
}

at::Tensor conv3d_max_log_sum_exp_relu_custom_impl_npu(const at::Tensor& x,
                                                       const at::Tensor& weight,
                                                       const c10::optional<at::Tensor>& bias_opt,
                                                       c10::List<int64_t> stride,
                                                       c10::List<int64_t> padding) {
    // 1) conv3d
    std::vector<int64_t> stride_v = to_3tuple(stride, 1);
    std::vector<int64_t> padding_v = to_3tuple(padding, 0);
    std::vector<int64_t> dilation_v = {1, 1, 1};
    int64_t groups = 1;
    at::Tensor conv_out;
    if (bias_opt.has_value()) {
        conv_out = at::conv3d(x, weight, bias_opt.value(), stride_v, padding_v, dilation_v, groups);
    } else {
        conv_out = at::conv3d(x, weight, at::Tensor(), stride_v, padding_v, dilation_v, groups);
    }

    // 2) max pool 3d with kernel_size=2, stride=2
    std::vector<int64_t> ksize = {2, 2, 2};
    std::vector<int64_t> kstride = {2, 2, 2};
    std::vector<int64_t> kpad = {0, 0, 0};
    std::vector<int64_t> kdil = {1, 1, 1};
    bool ceil_mode = false;
    at::Tensor pooled = at::max_pool3d(conv_out, ksize, kstride, kpad, kdil, ceil_mode);

    // 3) custom kernel: logsumexp over channel dim=1 with keepdim=True, then ReLU
    auto sizes = pooled.sizes().vec(); // N,C,D,H,W
    TORCH_CHECK(sizes.size() == 5, "Input to custom kernel must be 5D NCDHW");
    std::vector<int64_t> out_sizes = {sizes[0], 1, sizes[2], sizes[3], sizes[4]};
    at::Tensor result = at::empty(out_sizes, pooled.options());

    // Launch custom AscendC kernel
    EXEC_NPU_CMD(aclnnConv3dMaxLogSumExpReluCustom, pooled, result);
    return result;
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that fuses Conv3d, MaxPool3d(2,2), LogSumExp(dim=1, keepdim=True), and ReLU
    using a custom AscendC kernel for the reduction + activation.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride=1, padding=0):
        super().__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)

    def forward(self, x):
        # Use custom fused operator that performs:
        # conv3d -> max_pool3d(kernel_size=2, stride=2) on library kernels
        # then a custom AscendC kernel for logsumexp across channels and ReLU
        stride = self.conv.stride if isinstance(self.conv.stride, tuple) else (self.conv.stride,) * 3
        padding = self.conv.padding if isinstance(self.conv.padding, tuple) else (self.conv.padding,) * 3
        return custom_ops_lib.conv3d_max_log_sum_exp_relu_custom(
            x, self.conv.weight, self.conv.bias, list(stride), list(padding)
        )
