kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelCumsumExclusive {
public:
    __aicore__ inline KernelCumsumExclusive() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR axis, GM_ADDR y, uint32_t totalLength, uint32_t rowLen, uint32_t tileNum)
    {
        this->totalLength = totalLength;
        this->rowLen = rowLen;
        this->tileNum = tileNum;

        // Rows are contiguous when axis is the last dimension
        this->totalRows = (rowLen == 0) ? 0 : (totalLength / rowLen);

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t rowsPerBlock = (blockNum == 0) ? 0 : (totalRows / blockNum);
        uint32_t remain = (blockNum == 0) ? 0 : (totalRows % blockNum);

        this->startRow = rowsPerBlock * blockIdx + (blockIdx < remain ? blockIdx : remain);
        this->rowsForThisBlock = rowsPerBlock + (blockIdx < remain ? 1 : 0);

        // Base pointers for this block's first row
        uint32_t startOffset = this->startRow * this->rowLen;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + startOffset, this->rowsForThisBlock * this->rowLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + startOffset, this->rowsForThisBlock * this->rowLen);

        // Choose tile length to fit rows in segments
        uint32_t minSeg = this->rowLen / (this->tileNum * BUFFER_NUM);
        this->tileLength = minSeg > 0 ? minSeg : (this->rowLen > 0 ? this->rowLen : 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsForThisBlock; ++r) {
            uint32_t processed = 0;
            DTYPE_Y running_sum = static_cast<DTYPE_Y>(0);
            while (processed < this->rowLen) {
                uint32_t curLen = this->rowLen - processed > this->tileLength ? this->tileLength : (this->rowLen - processed);

                // CopyIn
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[r * this->rowLen + processed], curLen);
                inQueueX.EnQue(xLocal);

                // Compute
                AscendC::LocalTensor<DTYPE_X> xLocalDeq = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
                for (uint32_t i = 0; i < curLen; ++i) {
                    // exclusive cumsum: y[i] = running_sum; running_sum += x[i]
                    yLocal.SetValue(i, running_sum);
                    running_sum = running_sum + static_cast<DTYPE_Y>(xLocalDeq.GetValue(i));
                }
                outQueueY.EnQue(yLocal);
                inQueueX.FreeTensor(xLocalDeq);

                // CopyOut
                AscendC::LocalTensor<DTYPE_Y> yLocalDeq = outQueueY.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[r * this->rowLen + processed], yLocalDeq, curLen);
                outQueueY.FreeTensor(yLocalDeq);

                processed += curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalLength;
    uint32_t rowLen;
    uint32_t totalRows;
    uint32_t startRow;
    uint32_t rowsForThisBlock;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void cumsum_exclusive_custom(GM_ADDR x, GM_ADDR axis, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumExclusive op;
    op.Init(x, axis, y, tiling_data.totalLength, tiling_data.rowLen, tiling_data.tileNum);
    op.Process();
}
"""
