kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelDeconv3D {
public:
    __aicore__ inline KernelDeconv3D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t din, uint32_t hin, uint32_t win,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout,
                                uint64_t totalLength, uint32_t tileNum)
    {
        this->N = n; this->Cin = cin; this->Cout = cout;
        this->Din = din; this->Hin = hin; this->Win = win;
        this->kD = kd; this->kH = kh; this->kW = kw;
        this->Dout = dout; this->Hout = hout; this->Wout = wout;

        this->totalLength = totalLength;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        if (this->tileLength == 0) { this->tileLength = 1; }

        // Set global buffers. x and w point to full tensors; y points at per-block region.
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N * Cin * Din * Hin * Win);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y*)w, (uint64_t)Cin * Cout * kD * kH * kW);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y*)b, (uint64_t)Cout);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + (this->blockLength * AscendC::GetBlockIdx()), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        uint64_t baseOut = (uint64_t)this->blockLength * AscendC::GetBlockIdx();
        for (int32_t i = 0; i < loopCount; ++i) {
            uint64_t start = (uint64_t)i * this->tileLength;
            uint64_t end = start + this->tileLength;
            if (end > this->blockLength) { end = this->blockLength; }
            for (uint64_t off = start; off < end; ++off) {
                uint64_t outLinear = baseOut + off; // global index within full output
                // map to (n, co, do, ho, wo)
                uint64_t tmp = outLinear;
                uint32_t wo = tmp % Wout; tmp /= Wout;
                uint32_t ho = tmp % Hout; tmp /= Hout;
                uint32_t do_ = tmp % Dout; tmp /= Dout;
                uint32_t co = tmp % Cout; tmp /= Cout;
                uint32_t n = tmp % N;

                // Load bias[co] into accumulator
                AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
                AscendC::DataCopy(bLocal, bGm[co], 1);
                inQueueB.EnQue(bLocal);

                AscendC::LocalTensor<DTYPE_Z> accLocal = outQueueY.AllocTensor<DTYPE_Z>();
                // Initialize acc with bias
                AscendC::LocalTensor<DTYPE_Y> bLocalUse = inQueueB.DeQue<DTYPE_Y>();
                AscendC::DataCopy(accLocal, bLocalUse, 1);
                inQueueB.FreeTensor(bLocalUse);

                // Naive conv transpose (stride=1, pad=0, dilation=1):
                for (uint32_t ci = 0; ci < Cin; ++ci) {
                    for (uint32_t kd = 0; kd < kD; ++kd) {
                        int32_t di = (int32_t)do_ - (int32_t)kd;
                        if (di < 0 || di >= (int32_t)Din) continue;
                        for (uint32_t kh = 0; kh < kH; ++kh) {
                            int32_t hi = (int32_t)ho - (int32_t)kh;
                            if (hi < 0 || hi >= (int32_t)Hin) continue;
                            for (uint32_t kw = 0; kw < kW; ++kw) {
                                int32_t wi = (int32_t)wo - (int32_t)kw;
                                if (wi < 0 || wi >= (int32_t)Win) continue;

                                uint64_t xIdx = (((uint64_t)n * Cin + ci) * Din + (uint64_t)di) * Hin * Win
                                              + (uint64_t)hi * Win + (uint64_t)wi;

                                uint64_t wIdx = ((((uint64_t)ci * Cout + co) * kD + kd) * kH + kh) * kW + kw;

                                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                                AscendC::LocalTensor<DTYPE_Y> wLocal = inQueueW.AllocTensor<DTYPE_Y>();
                                AscendC::DataCopy(xLocal, xGm[xIdx], 1);
                                AscendC::DataCopy(wLocal, wGm[wIdx], 1);
                                inQueueX.EnQue(xLocal);
                                inQueueW.EnQue(wLocal);

                                AscendC::LocalTensor<DTYPE_Z> mulLocal = tmpQueue.AllocTensor<DTYPE_Z>();
                                AscendC::LocalTensor<DTYPE_X> xUse = inQueueX.DeQue<DTYPE_X>();
                                AscendC::LocalTensor<DTYPE_Y> wUse = inQueueW.DeQue<DTYPE_Y>();
                                AscendC::Mul(mulLocal, xUse, wUse, 1);
                                inQueueX.FreeTensor(xUse);
                                inQueueW.FreeTensor(wUse);

                                AscendC::Add(accLocal, accLocal, mulLocal, 1);
                                tmpQueue.FreeTensor(mulLocal);
                            }
                        }
                    }
                }

                // Store result to GM
                uint64_t yBlockOffset = off; // since yGm is already block-based
                AscendC::DataCopy(yGm[yBlockOffset], accLocal, 1);
                outQueueY.FreeTensor(accLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N, Cin, Cout;
    uint32_t Din, Hin, Win;
    uint32_t kD, kH, kW;
    uint32_t Dout, Hout, Wout;

    uint64_t totalLength;
    uint64_t blockLength;
    uint32_t tileNum;
    uint64_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDeconv3D op;
    op.Init(x, weight, bias, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.din, tiling_data.hin, tiling_data.win,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.dout, tiling_data.hout, tiling_data.wout,
            tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom AscendC kernel replacement for ConvTranspose3d with square input and asymmetric kernel.
    Supports default stride=(1,1,1), padding=(0,0,0), dilation=(1,1,1), output_padding=(0,0,0), groups=1.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: tuple = (1, 1, 1), padding: tuple = (0, 0, 0), output_padding: tuple = (0, 0, 0), groups: int = 1, bias: bool = False):
        super().__init__()
        self.conv_transpose3d = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding, groups=groups, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        bias = self.conv_transpose3d.bias
        if bias is None:
            bias = torch.zeros(self.conv_transpose3d.out_channels, dtype=x.dtype, device=x.device)
        return custom_ops_lib.conv_transposed_3d_square_input_asymmetric_kernel_custom(x, self.conv_transpose3d.weight, bias)
