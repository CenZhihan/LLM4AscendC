kernel_src="""
#include "kernel_operator.h"

extern \"C\" __global__ __aicore__ void squeeze_net_fire_module_custom(
    GM_ADDR x,
    GM_ADDR w_squeeze,
    GM_ADDR b_squeeze,
    GM_ADDR w_expand1x1,
    GM_ADDR b_expand1x1,
    GM_ADDR w_expand3x3,
    GM_ADDR b_expand3x3,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N  = tiling_data.N();
    const uint32_t Cin = tiling_data.C_in();
    const uint32_t H  = tiling_data.H();
    const uint32_t W  = tiling_data.W();
    const uint32_t Cs = tiling_data.C_squeeze();
    const uint32_t Ce1 = tiling_data.C_e1();
    const uint32_t Ce3 = tiling_data.C_e3();

    __gm__ float* x_ptr = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_sq_ptr = reinterpret_cast<__gm__ float*>(w_squeeze);
    __gm__ float* b_sq_ptr = reinterpret_cast<__gm__ float*>(b_squeeze);
    __gm__ float* w_e1_ptr = reinterpret_cast<__gm__ float*>(w_expand1x1);
    __gm__ float* b_e1_ptr = reinterpret_cast<__gm__ float*>(b_expand1x1);
    __gm__ float* w_e3_ptr = reinterpret_cast<__gm__ float*>(w_expand3x3);
    __gm__ float* b_e3_ptr = reinterpret_cast<__gm__ float*>(b_expand3x3);
    __gm__ float* y_ptr = reinterpret_cast<__gm__ float*>(y);
    __gm__ float* s_ptr = reinterpret_cast<__gm__ float*>(workspace);

    auto idx4 = [H, W](uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t C, uint32_t H_, uint32_t W_) -> uint64_t {
        return ((static_cast<uint64_t>(n) * C + c) * H_ + h) * W_ + w;
    };

    // 1) Squeeze 1x1 conv + ReLU, store to workspace s_ptr: (N, Cs, H, W)
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t h = 0; h < H; ++h) {
            for (uint32_t w = 0; w < W; ++w) {
                for (uint32_t cs = 0; cs < Cs; ++cs) {
                    float acc = b_sq_ptr[cs];
                    const uint64_t w_base = static_cast<uint64_t>(cs) * Cin; // [Cs, Cin, 1,1]
                    for (uint32_t ci = 0; ci < Cin; ++ci) {
                        float xv = x_ptr[idx4(n, ci, h, w, Cin, H, W)];
                        float wv = w_sq_ptr[w_base + ci];
                        acc += xv * wv;
                    }
                    // ReLU
                    if (acc < 0.0f) acc = 0.0f;
                    s_ptr[idx4(n, cs, h, w, Cs, H, W)] = acc;
                }
            }
        }
    }

    // 2) Expand 1x1 conv + ReLU, write first Ce1 channels of y
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t h = 0; h < H; ++h) {
            for (uint32_t w = 0; w < W; ++w) {
                for (uint32_t ce = 0; ce < Ce1; ++ce) {
                    float acc = b_e1_ptr[ce];
                    const uint64_t w_base = static_cast<uint64_t>(ce) * Cs; // [Ce1, Cs,1,1]
                    for (uint32_t cs = 0; cs < Cs; ++cs) {
                        float sv = s_ptr[idx4(n, cs, h, w, Cs, H, W)];
                        float wv = w_e1_ptr[w_base + cs];
                        acc += sv * wv;
                    }
                    if (acc < 0.0f) acc = 0.0f;
                    y_ptr[idx4(n, ce, h, w, Ce1 + Ce3, H, W)] = acc;
                }
            }
        }
    }

    // 3) Expand 3x3 conv (padding=1) + ReLU, write next Ce3 channels of y
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t h = 0; h < H; ++h) {
            for (uint32_t w = 0; w < W; ++w) {
                for (uint32_t ce = 0; ce < Ce3; ++ce) {
                    float acc = b_e3_ptr[ce];
                    for (uint32_t cs = 0; cs < Cs; ++cs) {
                        const uint64_t w_base = ((static_cast<uint64_t>(ce) * Cs) + cs) * 9U; // [Ce3, Cs,3,3]
                        for (uint32_t kh = 0; kh < 3; ++kh) {
                            int32_t ih = static_cast<int32_t>(h) + static_cast<int32_t>(kh) - 1;
                            if (ih < 0 || ih >= static_cast<int32_t>(H)) continue;
                            for (uint32_t kw = 0; kw < 3; ++kw) {
                                int32_t iw = static_cast<int32_t>(w) + static_cast<int32_t>(kw) - 1;
                                if (iw < 0 || iw >= static_cast<int32_t>(W)) continue;
                                float sv = s_ptr[idx4(n, cs, static_cast<uint32_t>(ih), static_cast<uint32_t>(iw), Cs, H, W)];
                                float wv = w_e3_ptr[w_base + kh * 3U + kw];
                                acc += sv * wv;
                            }
                        }
                    }
                    if (acc < 0.0f) acc = 0.0f;
                    y_ptr[idx4(n, Ce1 + ce, h, w, Ce1 + Ce3, H, W)] = acc;
                }
            }
        }
    }
}

model_src='''
import torch
import torch.nn as nn
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, in_channels, squeeze_channels, expand1x1_channels, expand3x3_channels):
        super(ModelNew, self).__init__()
        self.squeeze = nn.Conv2d(in_channels, squeeze_channels, kernel_size=1, bias=True)
        self.expand1x1 = nn.Conv2d(squeeze_channels, expand1x1_channels, kernel_size=1, bias=True)
        self.expand3x3 = nn.Conv2d(squeeze_channels, expand3x3_channels, kernel_size=3, padding=1, bias=True)

    def forward(self, x):
        return custom_ops_lib.squeeze_net_fire_module_custom(
            x,
            self.squeeze.weight, self.squeeze.bias,
            self.expand1x1.weight, self.expand1x1.bias,
            self.expand3x3.weight, self.expand3x3.bias
        )
