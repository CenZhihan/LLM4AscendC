host_operator_src="""
#include "lamb_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <cstdint>

namespace {
inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1) / b);
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 1;      // use single block to simplify reduction-free update
const uint32_t DEFAULT_TILE_LEN = 4096; // elements per tile to fit UB
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    LambCustomTilingData tiling;

    uint64_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    float lr = 0.0f;
    float eps = 1e-6f;
    float trust_ratio = 1.0f;

    // fetch attributes if provided
    (void)context->GetAttr("lr", lr);
    (void)context->GetAttr("eps", eps);
    (void)context->GetAttr("trust_ratio", trust_ratio);

    uint32_t tileLength = DEFAULT_TILE_LEN;
    if (totalLength < tileLength) {
        tileLength = static_cast<uint32_t>(totalLength);
    }
    uint32_t tileNum = CeilDiv(totalLength, tileLength);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_tileNum(tileNum);
    tiling.set_tileLength(tileLength);
    tiling.set_lr(lr);
    tiling.set_eps(eps);
    tiling.set_trust_ratio(trust_ratio);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);
    *out_shape = *in_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {

public:
    explicit LambCustom(const char* name) : OpDef(name)
    {
        this->Input("param")

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1; // single-buffered simplicity

class KernelLamb {
public:
    __aicore__ inline KernelLamb() {}

    __aicore__ inline void Init(GM_ADDR param, GM_ADDR m, GM_ADDR v, GM_ADDR out,
                                uint32_t totalLength, uint32_t tileNum, uint32_t tileLength,
                                float lr, float eps, float trust_ratio)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum(); // blockDim is 1
        this->tileNum = tileNum;
        this->tileLength = tileLength;
        this->eps = eps;
        this->alpha = lr * trust_ratio;

        paramGm.SetGlobalBuffer((__gm__ float *)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mGm.SetGlobalBuffer((__gm__ float *)m + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ float *)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ float *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inParamQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inMQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inVQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQ, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQ, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t processed = 0;
        for (uint32_t t = 0; t < this->tileNum; ++t) {
            uint32_t remain = this->blockLength - processed;
            uint32_t currLen = remain < this->tileLength ? remain : this->tileLength;

            CopyIn(processed, currLen);
            Compute(currLen);
            CopyOut(processed, currLen);

            processed += currLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t currLen)
    {
        LocalTensor<float> pLocal = inParamQ.AllocTensor<float>();
        LocalTensor<float> mLocal = inMQ.AllocTensor<float>();
        LocalTensor<float> vLocal = inVQ.AllocTensor<float>();

        DataCopy(pLocal, paramGm[offset], currLen);
        DataCopy(mLocal, mGm[offset], currLen);
        DataCopy(vLocal, vGm[offset], currLen);

        inParamQ.EnQue(pLocal);
        inMQ.EnQue(mLocal);
        inVQ.EnQue(vLocal);
    }

    __aicore__ inline void Compute(uint32_t currLen)
    {
        LocalTensor<float> pLocal = inParamQ.DeQue<float>();
        LocalTensor<float> mLocal = inMQ.DeQue<float>();
        LocalTensor<float> vLocal = inVQ.DeQue<float>();

        LocalTensor<float> tmpLocal = tmpQ.AllocTensor<float>(); // tmp for sqrt(v)+eps
        LocalTensor<float> outLocal = outQ.AllocTensor<float>();

        // tmpLocal = sqrt(vLocal)
        Sqrt(tmpLocal, vLocal, currLen);
        // tmpLocal = tmpLocal + eps
        Adds(tmpLocal, tmpLocal, this->eps, currLen);
        // outLocal = mLocal / tmpLocal   (reuse outLocal as r)
        Div(outLocal, mLocal, tmpLocal, currLen);
        // tmpLocal = outLocal * alpha
        Muls(tmpLocal, outLocal, this->alpha, currLen);
        // outLocal = pLocal - tmpLocal
        Sub(outLocal, pLocal, tmpLocal, currLen);

        outQ.EnQue(outLocal);

        tmpQ.FreeTensor(tmpLocal);
        inParamQ.FreeTensor(pLocal);
        inMQ.FreeTensor(mLocal);
        inVQ.FreeTensor(vLocal);
    }

    __aicore__ inline void CopyOut(uint32_t offset, uint32_t currLen)
    {
        LocalTensor<float> outLocal = outQ.DeQue<float>();
        DataCopy(outGm[offset], outLocal, currLen);
        outQ.FreeTensor(outLocal);
    }

private:
    TPipe pipe;
    TQue<TPosition::VECIN, BUFFER_NUM> inParamQ;
    TQue<TPosition::VECIN, BUFFER_NUM> inMQ;
    TQue<TPosition::VECIN, BUFFER_NUM> inVQ;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQ;
    TQue<TPosition::VECIN, BUFFER_NUM> tmpQ;

    GlobalTensor<float> paramGm;
    GlobalTensor<float> mGm;
    GlobalTensor<float> vGm;
    GlobalTensor<float> outGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    float eps;
    float alpha;
};

extern "C" __global__ __aicore__ void lamb_custom(GM_ADDR param, GM_ADDR m, GM_ADDR v, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLamb op;
    op.Init(param, m, v, out,
            tiling_data.totalLength, tiling_data.tileNum, tiling_data.tileLength,
            tiling_data.lr, tiling_data.eps, tiling_data.trust_ratio);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor lamb_custom_impl_npu(const at::Tensor& param, const at::Tensor& m, const at::Tensor& v, double lr, double eps) {
    // compute trust ratio on framework side
    // r = m / (sqrt(v) + eps)
    at::Tensor r = m / (v.sqrt() + eps);
    double trust_ratio = param.norm(2).item<double>() / (r.norm(2).item<double>() + eps);

    at::Tensor out = at::empty_like(param);
    EXEC_NPU_CMD(aclnnLambCustom, param, m, v, static_cast<float>(lr), static_cast<float>(eps), static_cast<float>(trust_ratio), out);
    return out;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("lamb_custom", &lamb_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("lamb_custom", &lamb_custom_impl_npu, "LAMB param update: param - lr*trust_ratio*m/(sqrt(v)+eps)",
          pybind11::arg("param"), pybind11::arg("m"), pybind11::arg("v"),
          pybind11::arg("lr") = 1e-3, pybind11::arg("eps") = 1e-6);
}
