kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelGruHiddenCopy {
public:
    __aicore__ inline KernelGruHiddenCopy() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR h, uint32_t totalLength, uint32_t tileNum)
    {
        // Partition work across blocks
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        // Set GM tensors. We only use h0 -> h, x is not used in this minimal kernel
        h0Gm.SetGlobalBuffer((__gm__ DTYPE_X *)h0 + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        hGm.SetGlobalBuffer((__gm__ DTYPE_Z *)h + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // Queues for in/out
        pipe.InitBuffer(inQueueH0, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueH, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> h0Local = inQueueH0.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(h0Local, h0Gm[progress * this->tileLength], this->tileLength);
        inQueueH0.EnQue(h0Local);
    }

    __aicore__ inline void Compute(int32_t)
    {
        // Pass-through: h_n = h0 (placeholder implementation)
        AscendC::LocalTensor<DTYPE_X> h0Local = inQueueH0.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> hLocal = outQueueH.AllocTensor<DTYPE_Z>();
        AscendC::DataCopy(hLocal, h0Local, this->tileLength);
        outQueueH.EnQue<DTYPE_Z>(hLocal);
        inQueueH0.FreeTensor(h0Local);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> hLocal = outQueueH.DeQue<DTYPE_Z>();
        AscendC::DataCopy(hGm[progress * this->tileLength], hLocal, this->tileLength);
        outQueueH.FreeTensor(hLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueH0;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueH;
    AscendC::GlobalTensor<DTYPE_X> h0Gm;
    AscendC::GlobalTensor<DTYPE_Z> hGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gru_bidirectional_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR h, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGruHiddenCopy op;
    op.Init(x, h0, h, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
"""
