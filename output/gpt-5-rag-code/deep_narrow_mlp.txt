#include "kernel_operator.h"

constexpr int32_t NUM_HIDDEN = 16;

class KernelDeepNarrowMlp {
public:
    __aicore__ inline KernelDeepNarrowMlp() {}

    __aicore__ inline void Init(
        GM_ADDR x,
        GM_ADDR y,
        GM_ADDR w0, GM_ADDR b0,
        GM_ADDR w1, GM_ADDR b1,
        GM_ADDR w2, GM_ADDR b2,
        GM_ADDR w3, GM_ADDR b3,
        GM_ADDR w4, GM_ADDR b4,
        GM_ADDR w5, GM_ADDR b5,
        GM_ADDR w6, GM_ADDR b6,
        GM_ADDR w7, GM_ADDR b7,
        GM_ADDR w8, GM_ADDR b8,
        GM_ADDR w9, GM_ADDR b9,
        GM_ADDR w10, GM_ADDR b10,
        GM_ADDR w11, GM_ADDR b11,
        GM_ADDR w12, GM_ADDR b12,
        GM_ADDR w13, GM_ADDR b13,
        GM_ADDR w14, GM_ADDR b14,
        GM_ADDR w15, GM_ADDR b15,
        GM_ADDR w16, GM_ADDR b16,
        GM_ADDR workspace,
        uint32_t batchSize, uint32_t inputSize, uint32_t hiddenSize, uint32_t outputSize)
    {
        batch_size_ = batchSize;
        input_size_ = inputSize;
        hidden_size_ = hiddenSize;
        output_size_ = outputSize;

        xGm_.SetGlobalBuffer((__gm__ DTYPE_X *)x, batchSize * inputSize);
        yGm_.SetGlobalBuffer((__gm__ DTYPE_Z *)y, batchSize * outputSize);

        // Set global buffers for each layer's weights/biases
        wGm_[0].SetGlobalBuffer((__gm__ DTYPE_W *)w0, hiddenSize * inputSize);
        bGm_[0].SetGlobalBuffer((__gm__ DTYPE_B *)b0, hiddenSize);
        wGm_[1].SetGlobalBuffer((__gm__ DTYPE_W *)w1, hiddenSize * hiddenSize);
        bGm_[1].SetGlobalBuffer((__gm__ DTYPE_B *)b1, hiddenSize);
        wGm_[2].SetGlobalBuffer((__gm__ DTYPE_W *)w2, hiddenSize * hiddenSize);
        bGm_[2].SetGlobalBuffer((__gm__ DTYPE_B *)b2, hiddenSize);
        wGm_[3].SetGlobalBuffer((__gm__ DTYPE_W *)w3, hiddenSize * hiddenSize);
        bGm_[3].SetGlobalBuffer((__gm__ DTYPE_B *)b3, hiddenSize);
        wGm_[4].SetGlobalBuffer((__gm__ DTYPE_W *)w4, hiddenSize * hiddenSize);
        bGm_[4].SetGlobalBuffer((__gm__ DTYPE_B *)b4, hiddenSize);
        wGm_[5].SetGlobalBuffer((__gm__ DTYPE_W *)w5, hiddenSize * hiddenSize);
        bGm_[5].SetGlobalBuffer((__gm__ DTYPE_B *)b5, hiddenSize);
        wGm_[6].SetGlobalBuffer((__gm__ DTYPE_W *)w6, hiddenSize * hiddenSize);
        bGm_[6].SetGlobalBuffer((__gm__ DTYPE_B *)b6, hiddenSize);
        wGm_[7].SetGlobalBuffer((__gm__ DTYPE_W *)w7, hiddenSize * hiddenSize);
        bGm_[7].SetGlobalBuffer((__gm__ DTYPE_B *)b7, hiddenSize);
        wGm_[8].SetGlobalBuffer((__gm__ DTYPE_W *)w8, hiddenSize * hiddenSize);
        bGm_[8].SetGlobalBuffer((__gm__ DTYPE_B *)b8, hiddenSize);
        wGm_[9].SetGlobalBuffer((__gm__ DTYPE_W *)w9, hiddenSize * hiddenSize);
        bGm_[9].SetGlobalBuffer((__gm__ DTYPE_B *)b9, hiddenSize);
        wGm_[10].SetGlobalBuffer((__gm__ DTYPE_W *)w10, hiddenSize * hiddenSize);
        bGm_[10].SetGlobalBuffer((__gm__ DTYPE_B *)b10, hiddenSize);
        wGm_[11].SetGlobalBuffer((__gm__ DTYPE_W *)w11, hiddenSize * hiddenSize);
        bGm_[11].SetGlobalBuffer((__gm__ DTYPE_B *)b11, hiddenSize);
        wGm_[12].SetGlobalBuffer((__gm__ DTYPE_W *)w12, hiddenSize * hiddenSize);
        bGm_[12].SetGlobalBuffer((__gm__ DTYPE_B *)b12, hiddenSize);
        wGm_[13].SetGlobalBuffer((__gm__ DTYPE_W *)w13, hiddenSize * hiddenSize);
        bGm_[13].SetGlobalBuffer((__gm__ DTYPE_B *)b13, hiddenSize);
        wGm_[14].SetGlobalBuffer((__gm__ DTYPE_W *)w14, hiddenSize * hiddenSize);
        bGm_[14].SetGlobalBuffer((__gm__ DTYPE_B *)b14, hiddenSize);
        wGm_[15].SetGlobalBuffer((__gm__ DTYPE_W *)w15, hiddenSize * hiddenSize);
        bGm_[15].SetGlobalBuffer((__gm__ DTYPE_B *)b15, hiddenSize);

        wOutGm_.SetGlobalBuffer((__gm__ DTYPE_W *)w16, outputSize * hiddenSize);
        bOutGm_.SetGlobalBuffer((__gm__ DTYPE_B *)b16, outputSize);

        // workspace for intermediate activations (batch_size_ x hidden_size_)
        interGm_.SetGlobalBuffer((__gm__ DTYPE_Z *)workspace, batchSize * hiddenSize);
    }

    __aicore__ inline void Process()
    {
        // First layer: x @ w0^T + b0, ReLU -> inter
        LinearRelu(xGm_, wGm_[0], bGm_[0], interGm_, batch_size_, input_size_, hidden_size_, /*transposeW=*/true);

        // Hidden layers 1..15: inter @ wi^T + bi, ReLU -> inter
        for (int i = 1; i < NUM_HIDDEN; ++i) {
            LinearRelu(interGm_, wGm_[i], bGm_[i], interGm_, batch_size_, hidden_size_, hidden_size_, /*transposeW=*/true);
        }

        // Output layer: inter @ w16^T + b16 -> y (no ReLU)
        Linear(interGm_, wOutGm_, bOutGm_, yGm_, batch_size_, hidden_size_, output_size_, /*transposeW=*/true);
    }

private:
    // Naive GEMM + Bias + ReLU on GM buffers for demonstration
    __aicore__ inline void LinearRelu(
        AscendC::GlobalTensor<DTYPE_Z>& A,           // [M,K]
        AscendC::GlobalTensor<DTYPE_W>& W,           // [N,K]
        AscendC::GlobalTensor<DTYPE_B>& B,           // [N]
        AscendC::GlobalTensor<DTYPE_Z>& Out,         // [M,N]
        uint32_t M, uint32_t K, uint32_t N, bool transposeW)
    {
        // Compute Out[m, n] = ReLU( sum_{k} A[m,k] * W[n,k] + B[n] )
        for (uint32_t m = 0; m < M; ++m) {
            for (uint32_t n = 0; n < N; ++n) {
                DTYPE_Z acc = static_cast<DTYPE_Z>(0.f);
                for (uint32_t k = 0; k < K; ++k) {
                    // W is [N,K] so W[n*K + k]
                    acc += static_cast<DTYPE_Z>(A[m * K + k]) * static_cast<DTYPE_Z>(W[n * K + k]);
                }
                acc += static_cast<DTYPE_Z>(B[n]);
                // ReLU
                Out[m * N + n] = acc > static_cast<DTYPE_Z>(0.f) ? acc : static_cast<DTYPE_Z>(0.f);
            }
        }
    }

    __aicore__ inline void Linear(
        AscendC::GlobalTensor<DTYPE_Z>& A,           // [M,K]
        AscendC::GlobalTensor<DTYPE_W>& W,           // [N,K]
        AscendC::GlobalTensor<DTYPE_B>& B,           // [N]
        AscendC::GlobalTensor<DTYPE_Z>& Out,         // [M,N]
        uint32_t M, uint32_t K, uint32_t N, bool transposeW)
    {
        // Compute Out[m, n] = sum_{k} A[m,k] * W[n,k] + B[n]
        for (uint32_t m = 0; m < M; ++m) {
            for (uint32_t n = 0; n < N; ++n) {
                DTYPE_Z acc = static_cast<DTYPE_Z>(0.f);
                for (uint32_t k = 0; k < K; ++k) {
                    acc += static_cast<DTYPE_Z>(A[m * K + k]) * static_cast<DTYPE_Z>(W[n * K + k]);
                }
                acc += static_cast<DTYPE_Z>(B[n]);
                Out[m * N + n] = acc;
            }
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm_;
    AscendC::GlobalTensor<DTYPE_Z> yGm_;
    AscendC::GlobalTensor<DTYPE_W> wGm_[NUM_HIDDEN];
    AscendC::GlobalTensor<DTYPE_B> bGm_[NUM_HIDDEN];
    AscendC::GlobalTensor<DTYPE_W> wOutGm_;
    AscendC::GlobalTensor<DTYPE_B> bOutGm_;
    AscendC::GlobalTensor<DTYPE_Z> interGm_;

    uint32_t batch_size_;
    uint32_t input_size_;
    uint32_t hidden_size_;
    uint32_t output_size_;
};
