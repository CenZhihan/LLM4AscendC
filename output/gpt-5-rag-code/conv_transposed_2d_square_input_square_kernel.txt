kernel_src="""
#include "kernel_operator.h"

class KernelConvTranspose2d {
public:
    __aicore__ inline KernelConvTranspose2d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t HOut, uint32_t WOut, uint32_t ocPerBlock)
    {
        this->N = N;
        this->Cin = Cin;
        this->Cout = Cout;
        this->H = H;
        this->W = W;
        this->K = K;
        this->HOut = HOut;
        this->WOut = WOut;

        this->ocStart = AscendC::GetBlockIdx() * ocPerBlock;
        this->ocEnd = this->ocStart + ocPerBlock;
        if (this->ocEnd > Cout) this->ocEnd = Cout;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N * Cin * H * W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y*)w, (uint64_t)Cin * Cout * K * K);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y, (uint64_t)N * Cout * HOut * WOut);
    }

    __aicore__ inline void ZeroOutputBlock()
    {
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oc = ocStart; oc < ocEnd; ++oc) {
                uint64_t base = ((uint64_t)n * Cout + oc) * HOut * WOut;
                uint64_t size = (uint64_t)HOut * WOut;
                // zeroing in chunks
                uint64_t offset = 0;
                while (offset < size) {
                    uint64_t step = (size - offset) > 4096 ? 4096 : (size - offset);
                    AscendC::LocalTensor<DTYPE_Z> tmp = pipe.AllocLocalTensor<DTYPE_Z>(step);
                    AscendC::Fill(tmp, static_cast<DTYPE_Z>(0), step);
                    AscendC::DataCopy(yGm[base + offset], tmp, step);
                    pipe.FreeLocalTensor(tmp);
                    offset += step;
                }
            }
        }
    }

    __aicore__ inline void Process()
    {
        // Naive direct transposed convolution for stride=1, padding=0, groups=1 (square kernel KxK)
        // y[n, oc, h + kh, w + kw] += x[n, ic, h, w] * w[ic, oc, kh, kw]
        // Initialize output slice to zero
        ZeroOutputBlock();

        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oc = ocStart; oc < ocEnd; ++oc) {
                for (uint32_t ic = 0; ic < Cin; ++ic) {
                    for (uint32_t kh = 0; kh < K; ++kh) {
                        for (uint32_t h = 0; h < H; ++h) {
                            // For each input row, accumulate to output row yRow = h + kh
                            uint32_t yRow = h + kh;
                            // Load kernel row weights w_row[0..K-1] for (ic, oc, kh, :)
                            // process across width
                            for (uint32_t w = 0; w < W; ++w) {
                                // x value
                                uint64_t x_idx = ((uint64_t)n * Cin + ic) * H * W + (uint64_t)h * W + w;
                                DTYPE_X x_val = xGm.GetValue(x_idx);

                                // Contribute to K positions along width
                                for (uint32_t kw = 0; kw < K; ++kw) {
                                    uint64_t y_col = w + kw;
                                    uint64_t y_idx = ((uint64_t)n * Cout + oc) * HOut * WOut + (uint64_t)yRow * WOut + y_col;
                                    uint64_t w_idx = ((uint64_t)ic * Cout + oc) * K * K + (uint64_t)kh * K + kw;

                                    DTYPE_Y k_val = wGm.GetValue(w_idx);
                                    DTYPE_Z y_old = yGm.GetValue(y_idx);
                                    yGm.SetValue(y_idx, y_old + static_cast<DTYPE_Z>(x_val * k_val));
                                }
                            }
                        }
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N, Cin, Cout, H, W, K, HOut, WOut;
    uint32_t ocStart, ocEnd;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    KernelConvTranspose2d op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.K,
            tiling_data.HOut, tiling_data.WOut, tiling_data.ocPerBlock);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ConvTranspose2dSquareCustom(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int):
        super().__init__()
        # PyTorch ConvTranspose2d weight layout: [in_channels, out_channels, kH, kW]
        self.weight = torch.nn.Parameter(torch.empty(in_channels, out_channels, kernel_size, kernel_size))
        torch.nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Only supports stride=1, padding=0, output_padding=0, groups=1, bias=False
        return custom_ops_lib.conv_transposed_2d_square_input_square_kernel_custom(x, self.weight)

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0, groups: int = 1, bias: bool = False):
        super().__init__()
        assert stride == 1 and padding == 0 and output_padding == 0 and groups == 1 and bias is False, "Custom kernel supports only stride=1, padding=0, output_padding=0, groups=1, bias=False"
        self.op = ConvTranspose2dSquareCustom(in_channels, out_channels, kernel_size)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
