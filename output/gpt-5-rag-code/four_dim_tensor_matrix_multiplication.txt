inline uint32_t clamp_tile(uint32_t val, uint32_t maxv, uint32_t minv = 1) {
    if (val > maxv) return maxv;
    if (val < minv) return minv;
    return val;
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    FourDimTensorMatrixMultiplicationCustomTilingData tiling;

    const auto* A_shape = context->GetInputShape(0);
    const auto* B_shape = context->GetInputShape(1);

    // Expect A: (b,i,j,l), B: (l,k)
    uint32_t b = static_cast<uint32_t>(A_shape->GetDim(0));
    uint32_t i = static_cast<uint32_t>(A_shape->GetDim(1));
    uint32_t j = static_cast<uint32_t>(A_shape->GetDim(2));
    uint32_t l = static_cast<uint32_t>(A_shape->GetDim(3));
    uint32_t k = static_cast<uint32_t>(B_shape->GetDim(1));

    // Simple heuristic tiles
    uint32_t tileK = clamp_tile(256, k);
    uint32_t tileL = clamp_tile(256, l);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_b(b);
    tiling.set_i(i);
    tiling.set_j(j);
    tiling.set_l(l);
    tiling.set_k(k);
    tiling.set_tileK(tileK);
    tiling.set_tileL(tileL);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

    const gert::Shape* B_shape = context->GetInputShape(1);
    // A: (b,i,j,l), B: (l,k) -> C: (b,i,j,k)
    gert::Shape* C_shape = context->GetOutputShape(0);
    std::vector<int64_t> outDims(4);
    outDims[0] = A_shape->GetDim(0);
    outDims[1] = A_shape->GetDim(1);
    outDims[2] = A_shape->GetDim(2);

kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 2;

class KernelFourDimTensorMatrixMultiplication {
public:
    __aicore__ inline KernelFourDimTensorMatrixMultiplication() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t b, uint32_t i, uint32_t j, uint32_t l, uint32_t k,
                                uint32_t tileK, uint32_t tileL)
    {
        this->b = b; this->i = i; this->j = j; this->l = l; this->k = k;
        this->tileK = tileK; this->tileL = tileL;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        totalOuter = b * i * j;
        // Simple split: equal partitions as in sample
        outerPerBlock = totalOuter / blockNum; // tail ignored for simplicity

        aGm.SetGlobalBuffer((__gm__ DTYPE*)A + (uint64_t)outerPerBlock * l * blockIdx,
                            (uint64_t)outerPerBlock * l);
        bGm.SetGlobalBuffer((__gm__ DTYPE*)B, (uint64_t)l * k);
        cGm.SetGlobalBuffer((__gm__ DTYPE*)C + (uint64_t)outerPerBlock * k * blockIdx,
                            (uint64_t)outerPerBlock * k);

        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileL * sizeof(DTYPE));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileK * sizeof(DTYPE));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, this->tileK * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t outer = 0; outer < outerPerBlock; ++outer) {
            // process K in tiles
            for (uint32_t kBase = 0; kBase < k; kBase += tileK) {
                uint32_t curK = AscendC::min(tileK, k - kBase);

                AscendC::LocalTensor<DTYPE> acc = outQueueC.AllocTensor<DTYPE>();
                AscendC::Dup(acc, static_cast<DTYPE>(0), curK);

                // iterate reduction dim L by tiles
                for (uint32_t lBase = 0; lBase < l; lBase += tileL) {
                    uint32_t curL = AscendC::min(tileL, l - lBase);
                    // load A chunk for this (b,i,j)
                    AscendC::LocalTensor<DTYPE> aChunk = inQueueA.AllocTensor<DTYPE>();
                    AscendC::DataCopy(aChunk, aGm[outer * l + lBase], curL);
                    inQueueA.EnQue(aChunk);
                    AscendC::LocalTensor<DTYPE> aLocal = inQueueA.DeQue<DTYPE>();

                    // accumulate outer products across curL rows
                    for (uint32_t t = 0; t < curL; ++t) {
                        // load B row segment of length curK
                        AscendC::LocalTensor<DTYPE> bRow = inQueueB.AllocTensor<DTYPE>();
                        AscendC::DataCopy(bRow, bGm[(lBase + t) * k + kBase], curK);
                        // scale bRow by scalar aLocal[t] and add to accumulator
                        DTYPE alpha = aLocal.GetValue(t);
                        AscendC::Muls(bRow, bRow, alpha, curK);
                        AscendC::Add(acc, acc, bRow, curK);
                        inQueueB.FreeTensor(bRow);
                    }
                    inQueueA.FreeTensor(aLocal);
                }

                // store result tile
                AscendC::DataCopy(cGm[outer * k + kBase], acc, curK);
                outQueueC.FreeTensor(acc);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;
    AscendC::GlobalTensor<DTYPE> aGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> cGm;

    uint32_t b, i, j, l, k;
    uint32_t tileK, tileL;
    uint32_t totalOuter;
    uint32_t outerPerBlock;
};

extern "C" __global__ __aicore__ void four_dim_tensor_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);
    KernelFourDimTensorMatrixMultiplication op;
    op.Init(A, B, C, t.b, t.i, t.j, t.l, t.k, t.tileK, t.tileL);
    op.Process();
}

#include <torch/library.h>

at::Tensor four_dim_tensor_matrix_multiplication_custom_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 4, "A must be 4D (b,i,j,l)");
    TORCH_CHECK(B.dim() == 2, "B must be 2D (l,k)");
    TORCH_CHECK(A.size(3) == B.size(0), "Mismatch on reduction dim: A[..., l] and B[l, ...]");

    auto opts = A.options();
    at::Tensor C = at::empty({A.size(0), A.size(1), A.size(2), B.size(1)}, opts);
    EXEC_NPU_CMD(aclnnFourDimTensorMatrixMultiplicationCustom, A, B, C);
    return C;
}

    m.impl("four_dim_tensor_matrix_multiplication_custom", &four_dim_tensor_matrix_multiplication_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("four_dim_tensor_matrix_multiplication_custom", &four_dim_tensor_matrix_multiplication_custom_impl_npu,
          "4D tensor-matrix multiplication: C[b,i,j,k] = sum_l A[b,i,j,l] * B[l,k]");
