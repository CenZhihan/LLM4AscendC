inline uint64_t GetShapeSize(const gert::Shape &shape) {
    uint64_t sz = 1;
    for (int i = 0; i < shape.GetOriginShape().GetDimNum(); ++i) {
        int64_t d = shape.GetOriginShape().GetDim(i);
        if (d > 0) sz *= static_cast<uint64_t>(d);
    }
    return sz;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MinReductionOverADimensionCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin = in_shape->GetOriginShape();
    int dimNum = origin.GetDimNum();
    uint64_t totalLength = GetShapeSize(*in_shape);
    uint32_t reduceLen = 1;
    if (dimNum > 0) {
        int64_t lastDim = origin.GetDim(dimNum - 1);
        reduceLen = static_cast<uint32_t>(lastDim > 0 ? lastDim : 1);
    }
    uint32_t outerSize = static_cast<uint32_t>(totalLength / reduceLen);

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_outerSize(outerSize);
    tiling.set_reduceLen(reduceLen);
    tiling.set_tileNum(TILE_NUM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    auto ox = x_shape->GetOriginShape();
    int dimNum = ox.GetDimNum();

    if (dimNum <= 0) {
        *y_shape = *x_shape;
        return GRAPH_SUCCESS;
    }

    // Output shape is input with the last dimension reduced (removed).
    int outDimNum = dimNum - 1;
    std::vector<int64_t> dims;
    dims.reserve(outDimNum);
    for (int i = 0; i < outDimNum; ++i) {
        dims.push_back(ox.GetDim(i));
    }

    gert::Shape tmp;
    tmp.SetDimNum(outDimNum);
    for (int i = 0; i < outDimNum; ++i) {
        tmp.SetDim(i, dims[i]);
    }
    *y_shape = tmp;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMinReduceLastDim {
public:
    __aicore__ inline KernelMinReduceLastDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerSize, uint32_t reduceLen, uint32_t tileNum)
    {
        this->reduceLen = reduceLen;
        this->tileNum = tileNum;
        this->blockOuter = outerSize / AscendC::GetBlockNum();
        if (this->tileNum == 0) { this->tileNum = 1; }
        this->tileLen = this->blockOuter / this->tileNum;
        if (this->tileLen == 0) { this->tileLen = this->blockOuter; this->tileNum = 1; }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)this->blockOuter * (uint64_t)this->reduceLen * AscendC::GetBlockIdx(),
                            (uint64_t)this->blockOuter * (uint64_t)this->reduceLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + (uint64_t)this->blockOuter * AscendC::GetBlockIdx(),
                            (uint64_t)this->blockOuter);

        pipe.InitBuffer(inQueue, 1, this->reduceLen * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, 1, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t totalTiles = this->tileNum;
        for (int32_t t = 0; t < totalTiles; ++t) {
            int32_t baseOuter = t * this->tileLen;
            int32_t rows = (t == totalTiles - 1) ? (this->blockOuter - baseOuter) : this->tileLen;
            for (int32_t r = 0; r < rows; ++r) {
                int32_t rowIdx = baseOuter + r;
                ReduceOneRow(rowIdx);
            }
        }
    }

private:
    __aicore__ inline void ReduceOneRow(int32_t rowIdx)
    {
        // Copy one row [reduceLen] from GM to UB
        AscendC::LocalTensor<DTYPE_X> rowLocal = inQueue.AllocTensor<DTYPE_X>();
        uint64_t gmOffset = (uint64_t)rowIdx * (uint64_t)this->reduceLen;
        AscendC::DataCopy(rowLocal, xGm[gmOffset], this->reduceLen);

        // Compute min across the row
        DTYPE_X minVal = rowLocal.GetValue(0);
        for (uint32_t c = 1; c < this->reduceLen; ++c) {
            DTYPE_X v = rowLocal.GetValue(c);
            minVal = (v < minVal) ? v : minVal;
        }

        inQueue.FreeTensor(rowLocal);

        // Write out the min to GM
        AscendC::LocalTensor<DTYPE_Y> outLocal = outQueue.AllocTensor<DTYPE_Y>();
        outLocal.SetValue(0, (DTYPE_Y)minVal);
        AscendC::DataCopy(yGm[rowIdx], outLocal, 1);
        outQueue.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t reduceLen;
    uint32_t blockOuter;
    uint32_t tileNum;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void min_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMinReduceLastDim op;
    op.Init(x, y, tiling_data.outerSize, tiling_data.reduceLen, tiling_data.tileNum);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline at::Tensor move_dim_to_last_if_needed(const at::Tensor& x, int64_t dim) {
    int64_t nd = x.dim();
    int64_t d = dim < 0 ? dim + nd : dim;
    if (d < 0 || d >= nd) {
        TORCH_CHECK(false, "dim out of range");
    }
    if (d == nd - 1) {
        return x;
    }
    return at::movedim(x, d, nd - 1);
}

at::Tensor min_reduction_over_a_dimension_custom_impl_npu(const at::Tensor& x, int64_t dim) {
    at::Tensor x_last = move_dim_to_last_if_needed(x, dim);
    // Output shape is input shape with the last dimension reduced
    std::vector<int64_t> out_sizes;
    out_sizes.reserve(x_last.dim() > 0 ? (x_last.dim() - 1) : 0);
    for (int i = 0; i < x_last.dim() - 1; ++i) {
        out_sizes.push_back(x_last.size(i));
    }
    at::Tensor result = at::empty(out_sizes, x_last.options());
    EXEC_NPU_CMD(aclnnMinReductionOverADimensionCustom, x_last, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("min_reduction_over_a_dimension_custom", &min_reduction_over_a_dimension_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("min_reduction_over_a_dimension_custom", &min_reduction_over_a_dimension_custom_impl_npu, "min over a dimension");
