static ge::graphStatus TilingFunc(gert::TilingContext* context)

    Conv2dReluHardSwishCustomTilingData tiling;

    const gert::Shape& wShape = context->GetInputShape(1)->GetOriginShape();

    uint32_t kW   = static_cast<uint32_t>(wShape.GetDim(3));

    // Assume stride=1, padding=0, dilation=1
    uint32_t strideH = 1, strideW = 1, padH = 0, padW = 0;
    uint32_t outH = (H + 2 * padH - kH) / strideH + 1;
    uint32_t outW = (W + 2 * padW - kW) / strideW + 1;

    uint32_t blockDim = Cout < maxBlocks ? Cout : maxBlocks;

    uint32_t coChunk = (Cout + blockDim - 1) / blockDim;

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padH(padH);

kernel_src="""
#include "kernel_operator.h"

struct Conv2dReluHardSwishCustomTilingData {
    uint32_t n;
    uint32_t cin;
    uint32_t cout;
    uint32_t h;
    uint32_t w;
    uint32_t kH;
    uint32_t kW;
    uint32_t outH;
    uint32_t outW;
    uint32_t strideH;
    uint32_t strideW;
    uint32_t padH;
    uint32_t padW;
    uint32_t coChunk;
};

template <typename T>
__aicore__ inline T local_min(T a, T b) { return a < b ? a : b; }

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    Conv2dReluHardSwishCustomTilingData td = *reinterpret_cast<Conv2dReluHardSwishCustomTilingData*>(tiling_data);
    __gm__ float* x_ptr = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_ptr = reinterpret_cast<__gm__ float*>(w);
    __gm__ float* b_ptr = reinterpret_cast<__gm__ float*>(b);
    __gm__ float* y_ptr = reinterpret_cast<__gm__ float*>(y);

    uint32_t N = td.n;
    uint32_t Cin = td.cin;
    uint32_t Cout = td.cout;
    uint32_t H = td.h;
    uint32_t W = td.w;
    uint32_t kH = td.kH;
    uint32_t kW = td.kW;
    uint32_t outH = td.outH;
    uint32_t outW = td.outW;
    uint32_t strideH = td.strideH;
    uint32_t strideW = td.strideW;
    uint32_t padH = td.padH;
    uint32_t padW = td.padW;

    uint32_t coChunk = td.coChunk;
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t coStart = blockIdx * coChunk;
    uint32_t coEnd = local_min<uint32_t>(coStart + coChunk, Cout);

    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t co = coStart; co < coEnd; ++co) {
            float bias_val = b_ptr[co];
            for (uint32_t oh = 0; oh < outH; ++oh) {
                for (uint32_t ow = 0; ow < outW; ++ow) {
                    float acc = bias_val;
                    int32_t base_h = static_cast<int32_t>(oh) * static_cast<int32_t>(strideH) - static_cast<int32_t>(padH);
                    int32_t base_w = static_cast<int32_t>(ow) * static_cast<int32_t>(strideW) - static_cast<int32_t>(padW);
                    for (uint32_t ci = 0; ci < Cin; ++ci) {
                        for (uint32_t kh = 0; kh < kH; ++kh) {
                            int32_t ih = base_h + static_cast<int32_t>(kh);
                            if (ih < 0 || ih >= static_cast<int32_t>(H)) {
                                continue;
                            }
                            for (uint32_t kw = 0; kw < kW; ++kw) {
                                int32_t iw = base_w + static_cast<int32_t>(kw);
                                if (iw < 0 || iw >= static_cast<int32_t>(W)) {
                                    continue;
                                }
                                uint64_t x_index = (((uint64_t)n * Cin + ci) * H + static_cast<uint32_t>(ih)) * W + static_cast<uint32_t>(iw);
                                uint64_t w_index = (((uint64_t)co * Cin + ci) * kH + kh) * kW + kw;
                                float xv = x_ptr[x_index];
                                float wv = w_ptr[w_index];
                                acc += xv * wv;
                            }
                        }
                    }
                    // ReLU
                    if (acc < 0.0f) acc = 0.0f;
                    // HardSwish: acc * clamp((acc + 3)/6, 0, 1)
                    float t = (acc + 3.0f) * (1.0f / 6.0f);
                    if (t < 0.0f) t = 0.0f;
                    if (t > 1.0f) t = 1.0f;
                    float outv = acc * t;

                    uint64_t y_index = (((uint64_t)n * Cout + co) * outH + oh) * outW + ow;
                    y_ptr[y_index] = outv;
                }
            }
        }
    }
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv2d_relu_hard_swish_custom_impl_npu(const at::Tensor& x, const at::Tensor& w, const at::Tensor& b) {
    TORCH_CHECK(x.dim() == 4, "x must be 4D NCHW");
    TORCH_CHECK(w.dim() == 4, "w must be 4D OIHW");
    TORCH_CHECK(b.dim() == 1, "b must be 1D of size Cout");

    auto N = x.size(0);
    auto Cin = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);
    auto Cout = w.size(0);
    auto kH = w.size(2);
    auto kW = w.size(3);
    TORCH_CHECK(w.size(1) == Cin, "w.size(1) must equal Cin");
    TORCH_CHECK(b.size(0) == Cout, "bias size must equal Cout");

    // stride=1, pad=0
    auto outH = H - kH + 1;
    auto outW = W - kW + 1;
    TORCH_CHECK(outH > 0 && outW > 0, "Invalid kernel size for given input");

    at::Tensor y = at::empty({N, Cout, outH, outW}, x.options());
    EXEC_NPU_CMD(aclnnConv2dReluHardSwishCustom, x, w, b, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_relu_hard_swish_custom", &conv2d_relu_hard_swish_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_relu_hard_swish_custom", &conv2d_relu_hard_swish_custom_impl_npu, "Conv2d + ReLU + HardSwish fused");
}
