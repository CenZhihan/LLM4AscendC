kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelMatmulSwishSumGroupNormCustom {
public:
    __aicore__ inline KernelMatmulSwishSumGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bLinear, GM_ADDR bExtra, GM_ADDR gamma, GM_ADDR beta,
                                GM_ADDR z,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t numGroups, float epsilon,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        this->M = M;
        this->K = K;
        this->N = N;
        this->numGroups = numGroups;
        this->epsilon = epsilon;
        this->groupSize = (numGroups > 0) ? (N / numGroups) : N;
        this->tileM = tileM;
        this->tileN = tileN;
        this->tileK = tileK;

        // block partition by rows
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();
        this->rowStart = (M / blocks) * bid;
        this->rowEnd = (bid == blocks - 1) ? M : (M / blocks) * (bid + 1);
        this->rowsThisBlock = this->rowEnd - this->rowStart;

        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, (uint64_t)M * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE*)w, (uint64_t)N * K);
        bLinearGm.SetGlobalBuffer((__gm__ DTYPE*)bLinear, N);
        bExtraGm.SetGlobalBuffer((__gm__ DTYPE*)bExtra, N);
        gammaGm.SetGlobalBuffer((__gm__ DTYPE*)gamma, N);
        betaGm.SetGlobalBuffer((__gm__ DTYPE*)beta, N);
        zGm.SetGlobalBuffer((__gm__ DTYPE*)z, (uint64_t)M * N);

        // Initialize buffers for tiles
        pipe.InitBuffer(accBufQue, 2, tileN * sizeof(DTYPE));
        pipe.InitBuffer(tmpBufQue, 2, tileN * sizeof(DTYPE));
        pipe.InitBuffer(wTileQue, 1, tileN * tileK * sizeof(DTYPE));
        pipe.InitBuffer(xTileQue, 1, tileK * sizeof(DTYPE));
        pipe.InitBuffer(bChunkQue, 1, tileN * sizeof(DTYPE));
        pipe.InitBuffer(extraChunkQue, 1, tileN * sizeof(DTYPE));
        pipe.InitBuffer(gammaChunkQue, 1, tileN * sizeof(DTYPE));
        pipe.InitBuffer(betaChunkQue, 1, tileN * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // Two-pass algorithm per row to avoid large workspace:
        // Pass 1: compute per-group sums and sumsq over features after Swish and extra_bias
        // Pass 2: recompute values and write normalized output
        for (uint32_t m = 0; m < rowsThisBlock; ++m) {
            uint32_t rowIdx = rowStart + m;

            // init accumulators
            uint32_t maxGroups = numGroups > 0 ? numGroups : 1;
            for (uint32_t g = 0; g < maxGroups; ++g) {
                groupSum[g] = 0.0f;
                groupSumSq[g] = 0.0f;
            }

            // Pass 1: stats
            RowPass(rowIdx, /*doNormalize=*/false, /*writeOut=*/false);

            // compute mean and inv std per group
            for (uint32_t g = 0; g < maxGroups; ++g) {
                float count = (float)groupSize;
                float mean = groupSum[g] / count;
                float var = groupSumSq[g] / count - mean * mean;
                if (var < 0.0f) var = 0.0f;
                groupMean[g] = mean;
                groupInvStd[g] = 1.0f / AscendC::Sqrt(var + epsilon);
            }

            // Pass 2: normalize and write
            RowPass(rowIdx, /*doNormalize=*/true, /*writeOut=*/true);
        }
    }

private:
    __aicore__ inline void RowPass(uint32_t rowIdx, bool doNormalize, bool writeOut)
    {
        // Iterate tiles along N
        for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
            uint32_t curN = AscendC::Min(tileN, N - n0);

            // acc <- linear_bias[n0:n0+curN]
            AscendC::LocalTensor<DTYPE> acc = accBufQue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> tmp = tmpBufQue.AllocTensor<DTYPE>();
            AscendC::LocalTensor<DTYPE> bChunk = bChunkQue.AllocTensor<DTYPE>();
            AscendC::DataCopy(bChunk, bLinearGm[n0], curN);
            AscendC::DataCopy(acc, bChunk, curN);

            // GEMV: acc += X[rowIdx, :] * W[n0:n0+curN, :]^T
            for (uint32_t k0 = 0; k0 < K; k0 += tileK) {
                uint32_t curK = AscendC::Min(tileK, K - k0);
                AscendC::LocalTensor<DTYPE> xTile = xTileQue.AllocTensor<DTYPE>();
                AscendC::DataCopy(xTile, xGm[rowIdx * K + k0], curK);

                // For each output channel in this tile, accumulate dot
                for (uint32_t nOff = 0; nOff < curN; ++nOff) {
                    // Load weight row segment W[n0+nOff, k0:k0+curK]
                    AscendC::LocalTensor<DTYPE> wTile = wTileQue.AllocTensor<DTYPE>();
                    AscendC::DataCopy(wTile, wGm[(n0 + nOff) * K + k0], curK);

                    // Compute dot product for this (rowIdx, n0+nOff) partial
                    float partial = 0.0f;
                    // Scalar loop over curK
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        partial += xTile.GetValue(kk) * wTile.GetValue(kk);
                    }
                    // acc[nOff] += partial
                    float prev = acc.GetValue(nOff);
                    acc.SetValue(nOff, prev + partial);

                    wTileQue.FreeTensor(wTile);
                }
                xTileQue.FreeTensor(xTile);
            }

            // Swish: s = sigmoid(acc) * acc
            AscendC::Sigmoid(tmp, acc, curN);
            AscendC::Mul(acc, tmp, acc, curN);

            // Add extra bias
            AscendC::LocalTensor<DTYPE> extraChunk = extraChunkQue.AllocTensor<DTYPE>();
            AscendC::DataCopy(extraChunk, bExtraGm[n0], curN);
            AscendC::Add(acc, acc, extraChunk, curN);

            if (!doNormalize) {
                // accumulate stats per group
                for (uint32_t i = 0; i < curN; ++i) {
                    uint32_t ch = n0 + i;
                    uint32_t g = (numGroups > 0 && groupSize > 0) ? (ch / groupSize) : 0;
                    float v = acc.GetValue(i);
                    groupSum[g] += v;
                    groupSumSq[g] += v * v;
                }
            } else {
                // normalize and affine
                AscendC::LocalTensor<DTYPE> gammaChunk = gammaChunkQue.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> betaChunk = betaChunkQue.AllocTensor<DTYPE>();
                AscendC::DataCopy(gammaChunk, gammaGm[n0], curN);
                AscendC::DataCopy(betaChunk, betaGm[n0], curN);

                for (uint32_t i = 0; i < curN; ++i) {
                    uint32_t ch = n0 + i;
                    uint32_t g = (numGroups > 0 && groupSize > 0) ? (ch / groupSize) : 0;
                    float v = acc.GetValue(i);
                    float vn = (v - groupMean[g]) * groupInvStd[g];
                    float outv = vn * gammaChunk.GetValue(i) + betaChunk.GetValue(i);
                    acc.SetValue(i, outv);
                }

                if (writeOut) {
                    AscendC::DataCopy(zGm[rowIdx * N + n0], acc, curN);
                }

                gammaChunkQue.FreeTensor(gammaChunk);
                betaChunkQue.FreeTensor(betaChunk);
            }

            accBufQue.FreeTensor(acc);
            tmpBufQue.FreeTensor(tmp);
            bChunkQue.FreeTensor(bChunk);
            extraChunkQue.FreeTensor(extraChunk);
        }
    }

private:
    AscendC::TPipe pipe;
    // Queues for local buffers
    AscendC::TQue<AscendC::TPosition::VECIN, 2> accBufQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> tmpBufQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wTileQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xTileQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bChunkQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> extraChunkQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> gammaChunkQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> betaChunkQue;

    // Global tensors
    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> wGm;
    AscendC::GlobalTensor<DTYPE> bLinearGm;
    AscendC::GlobalTensor<DTYPE> bExtraGm;
    AscendC::GlobalTensor<DTYPE> gammaGm;
    AscendC::GlobalTensor<DTYPE> betaGm;
    AscendC::GlobalTensor<DTYPE> zGm;

    // dims and tiling
    uint32_t M, K, N;
    uint32_t numGroups;
    uint32_t groupSize;
    float epsilon;
    uint32_t tileM, tileN, tileK;
    uint32_t rowStart, rowEnd, rowsThisBlock;

    // per-row stats
    float groupSum[1024];
    float groupSumSq[1024];
    float groupMean[1024];
    float groupInvStd[1024];
};

extern "C" __global__ __aicore__ void matmul_swish_sum_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR linear_bias, GM_ADDR extra_bias, GM_ADDR gamma, GM_ADDR beta, GM_ADDR z,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSwishSumGroupNormCustom op;
    op.Init(x, weight, linear_bias, extra_bias, gamma, beta, z,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.numGroups, tiling_data.epsilon,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a custom AscendC kernel that fuses:
    Linear (matmul + bias) -> Swish -> extra bias add -> GroupNorm
    """
    def __init__(self, in_features, out_features, num_groups, bias_shape):
        super(ModelNew, self).__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # weight layout expected by kernel: [out_features, in_features]
        return custom_ops_lib.matmul_swish_sum_group_norm_custom(
            x,
            self.matmul.weight,
            self.matmul.bias,
            self.bias,
            self.group_norm.weight,
            self.group_norm.bias,
            self.group_norm.num_groups,
