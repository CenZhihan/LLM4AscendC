kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM_DEFAULT = 2;

class KernelMatmulWithDiagonal {
public:
    __aicore__ inline KernelMatmulWithDiagonal() {}
    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t rows, uint32_t cols, uint32_t tileCols, uint32_t bufferNum)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileCols = tileCols;
        this->bufferNum = bufferNum > 0 ? bufferNum : BUFFER_NUM_DEFAULT;

        // distribute rows across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = (rows + blocks - 1) / blocks;
        this->rowStart = blockIdx * this->rowsPerBlock;
        uint64_t tmpEnd = static_cast<uint64_t>(this->rowStart) + static_cast<uint64_t>(this->rowsPerBlock);
        this->rowEnd = (tmpEnd > rows) ? rows : static_cast<uint32_t>(tmpEnd);
        if (this->rowStart >= this->rowEnd) {
            this->rowEnd = this->rowStart; // no work
        }
        uint32_t numRowsThisBlock = this->rowEnd - this->rowStart;

        // set global buffers to this block's slice
        aGm.SetGlobalBuffer((__gm__ DTYPE_A *)a + this->rowStart, numRowsThisBlock);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b + static_cast<uint64_t>(this->rowStart) * this->cols, static_cast<uint64_t>(numRowsThisBlock) * this->cols);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C *)c + static_cast<uint64_t>(this->rowStart) * this->cols, static_cast<uint64_t>(numRowsThisBlock) * this->cols);

        // init UB queues
        pipe.InitBuffer(inQueueB, this->bufferNum, this->tileCols * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueC, this->bufferNum, this->tileCols * sizeof(DTYPE_C));
        pipe.InitBuffer(inQueueAScalar, 1, sizeof(DTYPE_A)); // scalar a_i per row
        pipe.InitBuffer(tmpQueueA, this->bufferNum, this->tileCols * sizeof(DTYPE_A)); // expanded scalar to vector
    }

    __aicore__ inline void Process()
    {
        if (this->rowStart >= this->rowEnd) {
            return;
        }
        // For each row assigned to this block
        for (uint32_t r = 0; r < (this->rowEnd - this->rowStart); ++r) {
            // copy scalar a[r]
            AscendC::LocalTensor<DTYPE_A> aScalarLocal = inQueueAScalar.AllocTensor<DTYPE_A>();
            AscendC::DataCopy(aScalarLocal, aGm[r], 1);
            inQueueAScalar.EnQue<DTYPE_A>(aScalarLocal);
            aScalarLocal = inQueueAScalar.DeQue<DTYPE_A>();

            // iterate columns in tiles
            uint32_t col = 0;
            uint32_t tiles = (this->cols + this->tileCols - 1) / this->tileCols;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t curTileLen = (col + this->tileCols <= this->cols) ? this->tileCols : (this->cols - col);

                // CopyIn B tile
                AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
                AscendC::DataCopy(bLocal, bGm[static_cast<uint64_t>(r) * this->cols + col], curTileLen);
                inQueueB.EnQue<DTYPE_B>(bLocal);

                // Prepare expanded A vector for current tile (duplicate scalar)
                AscendC::LocalTensor<DTYPE_A> aVecLocal = tmpQueueA.AllocTensor<DTYPE_A>();
                AscendC::Duplicate(aVecLocal, aScalarLocal.GetValue(0), curTileLen);
                tmpQueueA.EnQue<DTYPE_A>(aVecLocal);

                // Compute: C_tile = A_scalar * B_tile
                bLocal = inQueueB.DeQue<DTYPE_B>();
                aVecLocal = tmpQueueA.DeQue<DTYPE_A>();
                AscendC::LocalTensor<DTYPE_C> cLocal = outQueueC.AllocTensor<DTYPE_C>();
                AscendC::Mul(cLocal, bLocal, aVecLocal, curTileLen);

                // enqueue for write back and free inputs
                outQueueC.EnQue<DTYPE_C>(cLocal);
                inQueueB.FreeTensor(bLocal);
                tmpQueueA.FreeTensor(aVecLocal);

                // CopyOut
                cLocal = outQueueC.DeQue<DTYPE_C>();
                AscendC::DataCopy(cGm[static_cast<uint64_t>(r) * this->cols + col], cLocal, curTileLen);
                outQueueC.FreeTensor(cLocal);

                col += curTileLen;
            }

            inQueueAScalar.FreeTensor(aScalarLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outQueueC;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueAScalar;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> tmpQueueA;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tileCols;
    uint32_t bufferNum;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void matmul_with_diagonal_matrices_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithDiagonal op;
    op.Init(a, b, c, tiling_data.rows, tiling_data.cols, tiling_data.tileCols, tiling_data.bufferNum);
    op.Process();
}
