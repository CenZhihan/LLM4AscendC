kernel_src="""
#include "kernel_operator.h"
#include <cmath>

class KernelMatmulGroupNormLeakyReluSum {
public:
    __aicore__ inline KernelMatmulGroupNormLeakyReluSum() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR gn_w, GM_ADDR gn_b, GM_ADDR y,
                                uint32_t m, uint32_t k, uint32_t n,
                                uint32_t numGroups, float eps, float negativeSlope)
    {
        this->m = m;
        this->k = k;
        this->n = n;
        this->numGroups = numGroups;
        this->groupSize = n / numGroups;
        this->eps = eps;
        this->negativeSlope = negativeSlope;

        // block split by rows
        uint32_t blocks = AscendC::GetBlockNum();
        this->rowsPerBlock = (m + blocks - 1) / blocks;
        this->rowStart = this->rowsPerBlock * AscendC::GetBlockIdx();
        if (this->rowStart + this->rowsPerBlock > m) {
            this->rowsPerBlock = (m > this->rowStart) ? (m - this->rowStart) : 0;
        }

        xGm.SetGlobalBuffer((__gm__ float*)x + this->rowStart * k, this->rowsPerBlock * k);
        wGm.SetGlobalBuffer((__gm__ float*)w, n * k);
        bGm.SetGlobalBuffer((__gm__ float*)b, n);
        gnWGm.SetGlobalBuffer((__gm__ float*)gn_w, n);
        gnBGm.SetGlobalBuffer((__gm__ float*)gn_b, n);
        yGm.SetGlobalBuffer((__gm__ float*)y + this->rowStart * n, this->rowsPerBlock * n);

        // L1 buffers
        pipe.InitBuffer(qX, 1, k * sizeof(float));                                // x row
        pipe.InitBuffer(qW, 1, groupSize * k * sizeof(float));                    // weight tile
        pipe.InitBuffer(qB, 1, groupSize * sizeof(float));                        // bias tile
        pipe.InitBuffer(qGnW, 1, groupSize * sizeof(float));                      // gn weight tile
        pipe.InitBuffer(qGnB, 1, groupSize * sizeof(float));                      // gn bias tile
        pipe.InitBuffer(qY, 1, groupSize * sizeof(float));                        // output tile
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsPerBlock; ++r) {
            // load x row
            AscendC::LocalTensor<float> xRow = qX.AllocTensor<float>();
            AscendC::DataCopy(xRow, xGm[r * k], k);
            qX.EnQue(xRow);

            for (uint32_t g = 0; g < numGroups; ++g) {
                uint32_t colStart = g * groupSize;

                AscendC::LocalTensor<float> wTile = qW.AllocTensor<float>();
                AscendC::DataCopy(wTile, wGm[colStart * k], groupSize * k);
                qW.EnQue(wTile);

                AscendC::LocalTensor<float> bTile = qB.AllocTensor<float>();
                AscendC::DataCopy(bTile, bGm[colStart], groupSize);
                qB.EnQue(bTile);

                AscendC::LocalTensor<float> gnWTile = qGnW.AllocTensor<float>();
                AscendC::DataCopy(gnWTile, gnWGm[colStart], groupSize);
                qGnW.EnQue(gnWTile);

                AscendC::LocalTensor<float> gnBTile = qGnB.AllocTensor<float>();
                AscendC::DataCopy(gnBTile, gnBGm[colStart], groupSize);
                qGnB.EnQue(gnBTile);

                // compute matmul partial for this group: yTile = xRow @ W_group^T + b_group
                AscendC::LocalTensor<float> xRowL = qX.DeQue<float>();
                AscendC::LocalTensor<float> wTileL = qW.DeQue<float>();
                AscendC::LocalTensor<float> bTileL = qB.DeQue<float>();
                AscendC::LocalTensor<float> yTile = qY.AllocTensor<float>();

                for (uint32_t c = 0; c < groupSize; ++c) {
                    float acc = bTileL.GetValue(c);
                    const float* wRowPtr = &wTileL.GetValue(c * k);
                    for (uint32_t kk = 0; kk < k; ++kk) {
                        acc += xRowL.GetValue(kk) * wRowPtr[kk];
                    }
                    yTile.SetValue(c, acc);
                }

                // group norm stats
                float mean = 0.f;
                for (uint32_t c = 0; c < groupSize; ++c) {
                    mean += yTile.GetValue(c);
                }
                mean /= static_cast<float>(groupSize);

                float var = 0.f;
                for (uint32_t c = 0; c < groupSize; ++c) {
                    float d = yTile.GetValue(c) - mean;
                    var += d * d;
                }
                var /= static_cast<float>(groupSize);
                float invStd = 1.f / AscendC::Sqrt(var + eps);

                // affine + leaky relu + elementwise sum (x = x + x)
                AscendC::LocalTensor<float> gnWTileL = qGnW.DeQue<float>();
                AscendC::LocalTensor<float> gnBTileL = qGnB.DeQue<float>();

                for (uint32_t c = 0; c < groupSize; ++c) {
                    float norm = (yTile.GetValue(c) - mean) * invStd;
                    float affine = norm * gnWTileL.GetValue(c) + gnBTileL.GetValue(c);
                    float act = affine >= 0.f ? affine : (affine * negativeSlope);
                    float summed = act + act;
                    yTile.SetValue(c, summed);
                }

                // write back
                qY.EnQue(yTile);

                AscendC::LocalTensor<float> outTile = qY.DeQue<float>();
                AscendC::DataCopy(yGm[r * n + colStart], outTile, groupSize);
                qY.FreeTensor(outTile);

                // free locals
                qW.FreeTensor(wTileL);
                qB.FreeTensor(bTileL);
                qGnW.FreeTensor(gnWTileL);
                qGnB.FreeTensor(gnBTileL);
                // keep xRow for next groups, do not free here
                qX.EnQue(xRowL); // re-enqueue for reuse
                (void)qX.DeQue<float>(); // balance queue to maintain state
                qX.FreeTensor(xRowL);
                // reload xRow for next loop
                AscendC::LocalTensor<float> xRow2 = qX.AllocTensor<float>();
                AscendC::DataCopy(xRow2, xGm[r * k], k);
                qX.EnQue(xRow2);
            }
            // cleanup xRow for row
            AscendC::LocalTensor<float> xr = qX.DeQue<float>();
            qX.FreeTensor(xr);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qX, qW, qB, qGnW, qGnB;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> qY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> gnWGm;
    AscendC::GlobalTensor<float> gnBGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t m, k, n;
    uint32_t numGroups, groupSize;
    uint32_t rowsPerBlock, rowStart;
    float eps;
    float negativeSlope;
};

extern "C" __global__ __aicore__ void matmul_group_norm_leaky_relu_sum_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR gn_weight, GM_ADDR gn_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulGroupNormLeakyReluSum op;
    op.Init(x, weight, bias, gn_weight, gn_bias, y,
            tiling_data.m, tiling_data.k, tiling_data.n,
            tiling_data.numGroups, tiling_data.eps, tiling_data.negativeSlope);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_size, num_groups, eps=1e-5, negative_slope=0.01):
        super().__init__()
        self.fc = torch.nn.Linear(input_size, hidden_size)
        self.gn = torch.nn.GroupNorm(num_groups=num_groups, num_channels=hidden_size, eps=eps)
        self.leaky_relu = torch.nn.LeakyReLU(negative_slope=negative_slope)

    def forward(self, x):
        return custom_ops_lib.matmul_group_norm_leaky_relu_sum_custom(
            x,
            self.fc.weight, self.fc.bias,
            self.gn.weight, self.gn.bias,
            float(self.gn.eps),
            float(self.leaky_relu.negative_slope),
            int(self.gn.num_groups)
