kernel_src="""
#include "kernel_operator.h"

// Fused kernel:
// y[b, 0] = sum_j ( sum_k x[b,k] * weight[j,k] + bias[j] )
//         = dot(x[b,:], sum_j weight[j,:]) + sum_j bias[j]

constexpr int32_t BUFFER_NUM = 2;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, uint32_t B, uint32_t I, uint32_t O, uint32_t tileK) {
        this->B = B;
        this->I = I;
        this->O = O;
        this->tileK = tileK;

        // Partition batch across blocks
        uint32_t blk = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        this->batchPerBlock = (B + blk - 1) / blk;
        this->batchStart = idx * this->batchPerBlock;
        this->batchEnd = AscendC::Min(this->batchStart + this->batchPerBlock, B);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)this->batchStart * I, (uint64_t)(this->batchEnd - this->batchStart) * I);
        wGm.SetGlobalBuffer((__gm__ DTYPE_WEIGHT*)w, (uint64_t)O * I);
        bGm.SetGlobalBuffer((__gm__ DTYPE_BIAS*)b, (uint64_t)O);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->batchStart, (uint64_t)(this->batchEnd - this->batchStart));

        // Queues for UB buffers
        pipe.InitBuffer(wsumQ, BUFFER_NUM, tileK * sizeof(DTYPE_WEIGHT));
        pipe.InitBuffer(xQ, BUFFER_NUM, tileK * sizeof(DTYPE_X));
        pipe.InitBuffer(prodQ, BUFFER_NUM, tileK * sizeof(DTYPE_Y));
        pipe.InitBuffer(scalarQ, BUFFER_NUM, 1 * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process() {
        // Precompute bias sum
        DTYPE_Y bsum = BiasSum();

        // Initialize per-batch accumulators in registers when writing out
        for (uint32_t b = this->batchStart; b < this->batchEnd; ++b) {
            DTYPE_Y acc = (DTYPE_Y)0;
            // Iterate over K tiles
            for (uint32_t k = 0; k < I; k += this->tileK) {
                uint32_t tileLen = AscendC::Min(this->tileK, I - k);
                // Compute wsum tile = sum over rows j of weight[j, k:k+tileLen]
                AscendC::LocalTensor<DTYPE_WEIGHT> wsumTile = wsumQ.AllocTensor<DTYPE_WEIGHT>();
                // Initialize wsumTile with first row
                AscendC::DataCopy(wsumTile, wGm[k], tileLen);
                for (uint32_t j = 1; j < O; ++j) {
                    AscendC::LocalTensor<DTYPE_WEIGHT> wtmp = wsumQ.AllocTensor<DTYPE_WEIGHT>();
                    AscendC::DataCopy(wtmp, wGm[j * I + k], tileLen);
                    AscendC::Add(wsumTile, wsumTile, wtmp, tileLen);
                    wsumQ.FreeTensor(wtmp);
                }
                // Load x tile
                AscendC::LocalTensor<DTYPE_X> xTile = xQ.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xTile, xGm[(b - this->batchStart) * I + k], tileLen);
                // prod = xTile * wsumTile
                AscendC::LocalTensor<DTYPE_Y> prodTile = prodQ.AllocTensor<DTYPE_Y>();
                AscendC::Mul(prodTile, xTile, wsumTile, tileLen);
                // reduce sum to scalar
                AscendC::LocalTensor<DTYPE_Y> scalar = scalarQ.AllocTensor<DTYPE_Y>();
                AscendC::ReduceSum(scalar, prodTile, tileLen);
                // accumulate
                acc = acc + scalar.GetValue(0);
                // free
                scalarQ.FreeTensor(scalar);
                prodQ.FreeTensor(prodTile);
                xQ.FreeTensor(xTile);
                wsumQ.FreeTensor(wsumTile);
            }
            // Add bias sum and write out
            acc = acc + bsum;
            AscendC::DataCopy(yGm[b - this->batchStart], &acc, 1);
        }
    }

private:
    __aicore__ inline DTYPE_Y BiasSum() {
        DTYPE_Y bsum = (DTYPE_Y)0;
        // Sum bias vector
        // Tile over bias if needed
        uint32_t step = 1024;
        for (uint32_t j = 0; j < O; j += step) {
            uint32_t len = AscendC::Min(step, O - j);
            AscendC::LocalTensor<DTYPE_BIAS> bTile = xQ.AllocTensor<DTYPE_BIAS>();
            AscendC::DataCopy(bTile, bGm[j], len);
            // Reduce
            AscendC::LocalTensor<DTYPE_Y> s = scalarQ.AllocTensor<DTYPE_Y>();
            AscendC::ReduceSum(s, bTile, len);
            bsum = bsum + s.GetValue(0);
            scalarQ.FreeTensor(s);
            xQ.FreeTensor(bTile);
        }
        return bsum;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wsumQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> prodQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> scalarQ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_WEIGHT> wGm;
    AscendC::GlobalTensor<DTYPE_BIAS> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t B;
    uint32_t I;
    uint32_t O;
    uint32_t tileK;

    uint32_t batchStart;
    uint32_t batchEnd;
    uint32_t batchPerBlock;
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);
    KernelFused op;
    op.Init(x, weight, bias, y, t.batch, t.inFeat, t.outFeat, t.tileK);
    op.Process();
}
