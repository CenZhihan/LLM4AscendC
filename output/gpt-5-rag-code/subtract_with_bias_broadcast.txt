constexpr uint32_t BLOCK_DIM = 8;
constexpr uint32_t BUFFER_NUM = 2;

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    SubtractWithBiasBroadcastCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* bias_shape = context->GetInputShape(1);

    uint32_t totalLength = x_shape->GetOriginShape().GetShapeSize();
    uint32_t lastDim = bias_shape->GetOriginShape().GetShapeSize();
    if (lastDim == 0) {
        lastDim = 1;
    }

    context->SetBlockDim(BLOCK_DIM);

    uint32_t blockLengthHost = totalLength / BLOCK_DIM;
    // ensure tileLength <= lastDim by ceil division
    uint64_t denom = static_cast<uint64_t>(lastDim) * BUFFER_NUM;
    uint32_t tileNum = static_cast<uint32_t>((blockLengthHost + denom - 1) / denom);
    if (tileNum == 0) {
        tileNum = 1;
    }

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(tileNum);
    tiling.set_lastDim(lastDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelSubtractWithBiasBroadcast {
public:
    __aicore__ inline KernelSubtractWithBiasBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, uint32_t lastDim)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->lastDim = lastDim;

        // Compute tileLength so that tileLength <= lastDim
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        if (this->tileLength == 0) {
            this->tileLength = lastDim; // fallback
        }
        if (this->tileLength > this->lastDim) {
            this->tileLength = this->lastDim;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias, this->lastDim);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint32_t loopCount = this->tileNum * BUFFER_NUM;
        uint32_t processed = loopCount * this->tileLength;
        uint32_t tail = (processed < this->blockLength) ? (this->blockLength - processed) : 0;

        for (uint32_t i = 0; i < loopCount; i++) {
            CopyIn(i, this->tileLength);
            Compute(this->tileLength);
            CopyOut(i, this->tileLength);
        }

        if (tail > 0) {
            CopyIn(loopCount, tail);
            Compute(tail);
            CopyOut(loopCount, tail);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t progress, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();

        uint32_t base = progress * this->tileLength;
        AscendC::DataCopy(xLocal, xGm[base], len);

        uint32_t biasStart = (base) % this->lastDim;
        uint32_t firstPart = AscendC::min(len, this->lastDim - biasStart);
        if (firstPart > 0) {
            AscendC::DataCopy(bLocal, bGm[biasStart], firstPart);
        }
        if (len > firstPart) {
            AscendC::DataCopy(bLocal[firstPart], bGm[0], len - firstPart);
        }

        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        AscendC::Sub(yLocal, xLocal, bLocal, len);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, len);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t lastDim;
};

extern "C" __global__ __aicore__ void subtract_with_bias_broadcast_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSubtractWithBiasBroadcast op;
    op.Init(x, bias, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.lastDim);
    op.Process();
}
