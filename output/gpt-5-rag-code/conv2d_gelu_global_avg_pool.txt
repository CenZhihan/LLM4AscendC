kernel_src="""
#include "kernel_operator.h"

struct Tiling {
    uint32_t N;
    uint32_t Cin;
    uint32_t H;
    uint32_t W;
    uint32_t Cout;
    uint32_t Kh;
    uint32_t Kw;
    uint32_t Hout;
    uint32_t Wout;
    uint32_t totalNC;
};

__aicore__ inline float gelu_approx(float x)
{
    const float k0 = 0.7978845608028654f; // sqrt(2/pi)
    const float k1 = 0.044715f;
    float x3 = x * x * x;
    float t = k0 * (x + k1 * x3);
    // tanh approximation path
    float y = 0.5f * x * (1.0f + tanhf(t));
    return y;
}

class KernelConv2dGeluGAP {
public:
    __aicore__ inline KernelConv2dGeluGAP() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, const Tiling &t) {
        x_ptr = reinterpret_cast<__gm__ float*>(x);
        w_ptr = reinterpret_cast<__gm__ float*>(w);
        b_ptr = reinterpret_cast<__gm__ float*>(b);
        y_ptr = reinterpret_cast<__gm__ float*>(y);
        tiling = t;
    }

    __aicore__ inline void Process() {
        const uint32_t blkIdx = AscendC::GetBlockIdx();
        const uint32_t blkNum = AscendC::GetBlockNum();

        uint32_t total = tiling.N * tiling.Cout;
        uint32_t perBlock = (total + blkNum - 1) / blkNum;
        uint32_t start = blkIdx * perBlock;
        uint32_t end = AscendC::Min(total, start + perBlock);

        // Precompute strides for NCHW
        uint32_t Cin = tiling.Cin;
        uint32_t H = tiling.H;
        uint32_t W = tiling.W;
        uint32_t Kh = tiling.Kh;
        uint32_t Kw = tiling.Kw;
        uint32_t Hout = tiling.Hout;
        uint32_t Wout = tiling.Wout;

        uint32_t in_stride_n = Cin * H * W;
        uint32_t in_stride_c = H * W;
        uint32_t in_stride_h = W;

        uint32_t w_stride_oc = Cin * Kh * Kw;
        uint32_t w_stride_ic = Kh * Kw;
        uint32_t w_stride_kh = Kw;

        float denom = (Hout > 0 && Wout > 0) ? static_cast<float>(Hout * Wout) : 1.0f;

        for (uint32_t idx = start; idx < end; ++idx) {
            uint32_t n = idx / tiling.Cout;
            uint32_t oc = idx % tiling.Cout;

            float pooled_sum = 0.0f;

            // For each output spatial location
            for (uint32_t oh = 0; oh < Hout; ++oh) {
                for (uint32_t ow = 0; ow < Wout; ++ow) {
                    float acc = b_ptr[oc];

                    // Convolution accumulation (stride=1, pad=0, dilation=1)
                    for (uint32_t ic = 0; ic < Cin; ++ic) {
                        for (uint32_t kh = 0; kh < Kh; ++kh) {
                            uint32_t ih = oh + kh;
                            for (uint32_t kw = 0; kw < Kw; ++kw) {
                                uint32_t iw = ow + kw;

                                uint32_t x_off = n * in_stride_n + ic * in_stride_c + ih * in_stride_h + iw;
                                uint32_t w_off = oc * w_stride_oc + ic * w_stride_ic + kh * w_stride_kh + kw;

                                float xv = x_ptr[x_off];
                                float wv = w_ptr[w_off];
                                acc += xv * wv;
                            }
                        }
                    }

                    float act = gelu_approx(acc);
                    pooled_sum += act;
                }
            }

            float outv = pooled_sum / denom;
            y_ptr[n * tiling.Cout + oc] = outv;
        }
    }

private:
    __gm__ float* x_ptr;
    __gm__ float* w_ptr;
    __gm__ float* b_ptr;
    __gm__ float* y_ptr;
    Tiling tiling;
};

extern "C" __global__ __aicore__ void conv2d_gelu_global_avg_pool_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling_gm) {
    GET_TILING_DATA(tiling_data, tiling_gm);
    Tiling t;
    t.N      = tiling_data.N;
    t.Cin    = tiling_data.Cin;
    t.H      = tiling_data.H;
    t.W      = tiling_data.W;
    t.Cout   = tiling_data.Cout;
    t.Kh     = tiling_data.Kh;
    t.Kw     = tiling_data.Kw;
    t.Hout   = tiling_data.Hout;
    t.Wout   = tiling_data.Wout;
    t.totalNC= tiling_data.totalNC;

    KernelConv2dGeluGAP op;
    op.Init(x, weight, bias, y, t);
    op.Process();
}
