    const gert::Shape* in_shape = context->GetInputShape(0);
    // Expect 2D [B, D]
    uint32_t dim_num = in_shape->GetOriginShape().GetDimNum();
    uint32_t batch = 1;
    uint32_t dim = 1;
    if (dim_num >= 2) {
        batch = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(0));
        dim   = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(1));

        dim   = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(0));

kernel_src="""
#include "kernel_operator.h"
#include <math.h>

class KernelCosineSimilarityLossCustom {
public:
    __aicore__ inline KernelCosineSimilarityLossCustom() {}

    __aicore__ inline void Init(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss, uint32_t batchSize, uint32_t dimSize, uint32_t tileNum, uint32_t tileLength)
    {
        this->batchSize = batchSize;
        this->dimSize = dimSize;
        this->tileNum = tileNum;
        this->tileLength = tileLength;

        predsGm.SetGlobalBuffer((__gm__ float*)predictions, batchSize * dimSize);
        targsGm.SetGlobalBuffer((__gm__ float*)targets, batchSize * dimSize);
        lossGm.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe.InitBuffer(inQueuePred, 1, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueTarg, 1, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue, 1, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueue, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        float lossSum = 0.f;
        const float eps = 1e-8f;

        for (uint32_t b = 0; b < batchSize; ++b) {
            float dotSum = 0.f;
            float normPred = 0.f;
            float normTarg = 0.f;

            for (uint32_t t = 0; t < tileNum; ++t) {
                uint32_t offset = b * dimSize + t * tileLength;
                uint32_t len = (t == tileNum - 1) ? (dimSize - t * tileLength) : tileLength;

                AscendC::LocalTensor<float> pLocal = inQueuePred.AllocTensor<float>();
                AscendC::LocalTensor<float> qLocal = inQueueTarg.AllocTensor<float>();
                AscendC::DataCopy(pLocal, predsGm[offset], len);
                AscendC::DataCopy(qLocal, targsGm[offset], len);
                inQueuePred.EnQue(pLocal);
                inQueueTarg.EnQue(qLocal);

                pLocal = inQueuePred.DeQue<float>();
                qLocal = inQueueTarg.DeQue<float>();

                // dot product partial: tmp = p * q; reduce sum
                AscendC::LocalTensor<float> tmpLocal = tmpQueue.AllocTensor<float>();
                AscendC::Mul(tmpLocal, pLocal, qLocal, len);
                float dotPart = 0.f;
                // Reduce sum over tmpLocal
                for (uint32_t i = 0; i < len; ++i) {
                    dotPart += tmpLocal.GetValue(i);
                }

                // norm of predictions: tmp = p * p; reduce sum
                float normPredPart = 0.f;
                for (uint32_t i = 0; i < len; ++i) {
                    float v = pLocal.GetValue(i);
                    normPredPart += v * v;
                }

                // norm of targets: tmp = q * q; reduce sum
                float normTargPart = 0.f;
                for (uint32_t i = 0; i < len; ++i) {
                    float v = qLocal.GetValue(i);
                    normTargPart += v * v;
                }

                dotSum += dotPart;
                normPred += normPredPart;
                normTarg += normTargPart;

                tmpQueue.FreeTensor(tmpLocal);
                inQueuePred.FreeTensor(pLocal);
                inQueueTarg.FreeTensor(qLocal);
            }

            float denom = sqrtf(normPred) * sqrtf(normTarg) + eps;
            float cosine_sim = dotSum / denom;
            float loss_i = 1.0f - cosine_sim;
            lossSum += loss_i;
        }

        float meanLoss = lossSum / static_cast<float>(batchSize);

        // Write out meanLoss to GM as a single value tensor [1]
        AscendC::LocalTensor<float> outLocal = outQueue.AllocTensor<float>();
        outLocal.SetValue(0, meanLoss);
        outQueue.EnQue(outLocal);

        outLocal = outQueue.DeQue<float>();
        AscendC::DataCopy(lossGm, outLocal, 1);
        outQueue.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueuePred;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueTarg;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue;

    AscendC::GlobalTensor<float> predsGm;
    AscendC::GlobalTensor<float> targsGm;
    AscendC::GlobalTensor<float> lossGm;

    uint32_t batchSize;
    uint32_t dimSize;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void cosine_similarity_loss_custom(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCosineSimilarityLossCustom op;
    op.Init(predictions, targets, loss, tiling_data.batchSize, tiling_data.dimSize, tiling_data.tileNum, tiling_data.tileLength);
    op.Process();
}
