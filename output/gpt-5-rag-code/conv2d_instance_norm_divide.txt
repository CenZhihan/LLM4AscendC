kernel_src="""
#include "kernel_operator.h"

using DTYPE_T = float;

extern "C" __global__ __aicore__ void conv2d_instance_norm_divide_custom(GM_ADDR x,
                                                                         GM_ADDR weight,
                                                                         GM_ADDR bias,
                                                                         GM_ADDR divide_by,
                                                                         GM_ADDR y,
                                                                         GM_ADDR workspace,
                                                                         GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N   = tiling_data.n();
    const uint32_t Cin = tiling_data.cin();
    const uint32_t H   = tiling_data.h();
    const uint32_t W   = tiling_data.w();
    const uint32_t Cout= tiling_data.cout();
    const uint32_t Kh  = tiling_data.kh();
    const uint32_t Kw  = tiling_data.kw();
    const uint32_t Oh  = tiling_data.oh();
    const uint32_t Ow  = tiling_data.ow();
    const float eps    = tiling_data.eps();

    __gm__ const DTYPE_T* xPtr  = reinterpret_cast<__gm__ const DTYPE_T*>(x);
    __gm__ const DTYPE_T* wPtr  = reinterpret_cast<__gm__ const DTYPE_T*>(weight);
    __gm__ const DTYPE_T* bPtr  = reinterpret_cast<__gm__ const DTYPE_T*>(bias);
    __gm__ const DTYPE_T* dPtr  = reinterpret_cast<__gm__ const DTYPE_T*>(divide_by);
    __gm__ DTYPE_T* yPtr        = reinterpret_cast<__gm__ DTYPE_T*>(y);

    DTYPE_T div_val = dPtr[0];

    const uint32_t groups = N * Cout;
    const uint32_t blkNum = AscendC::GetBlockNum();
    const uint32_t blkIdx = AscendC::GetBlockIdx();
    const uint32_t chunk  = (groups + blkNum - 1) / blkNum;
    const uint32_t start  = blkIdx * chunk;
    const uint32_t end    = (start + chunk) > groups ? groups : (start + chunk);

    auto x_index = [=] (uint32_t n, uint32_t c, uint32_t h, uint32_t w_) -> uint64_t {
        return (((uint64_t)n * Cin + c) * H + h) * W + w_;
    };
    auto w_index = [=] (uint32_t co, uint32_t ci, uint32_t kh, uint32_t kw_) -> uint64_t {
        return (((uint64_t)co * Cin + ci) * Kh + kh) * Kw + kw_;
    };
    auto y_index = [=] (uint32_t n, uint32_t co, uint32_t h, uint32_t w_) -> uint64_t {
        return (((uint64_t)n * Cout + co) * Oh + h) * Ow + w_;
    };

    for (uint32_t g = start; g < end; ++g) {
        uint32_t n  = g / Cout;
        uint32_t co = g % Cout;

        for (uint32_t oh = 0; oh < Oh; ++oh) {
            for (uint32_t ow = 0; ow < Ow; ++ow) {
                DTYPE_T acc = bPtr[co];
                for (uint32_t ci = 0; ci < Cin; ++ci) {
                    for (uint32_t kh = 0; kh < Kh; ++kh) {
                        uint32_t ih = oh + kh;
                        for (uint32_t kw = 0; kw < Kw; ++kw) {
                            uint32_t iw = ow + kw;
                            DTYPE_T xv = xPtr[x_index(n, ci, ih, iw)];
                            DTYPE_T wv = wPtr[w_index(co, ci, kh, kw)];
                            acc += xv * wv;
                        }
                    }
                }
                yPtr[y_index(n, co, oh, ow)] = acc;
            }
        }

        // Instance Norm per (n, co) over spatial dims (Oh x Ow)
        // Compute mean and variance
        double sum = 0.0;
        double sumsq = 0.0;
        const uint64_t spatial = (uint64_t)Oh * (uint64_t)Ow;
        for (uint32_t oh = 0; oh < Oh; ++oh) {
            for (uint32_t ow = 0; ow < Ow; ++ow) {
                DTYPE_T v = yPtr[y_index(n, co, oh, ow)];
                sum += (double)v;
                sumsq += (double)v * (double)v;
            }
        }
        double mean = sum / (double)spatial;
        double var = sumsq / (double)spatial - mean * mean;
        if (var < 0.0) var = 0.0;
        double inv_std = 1.0 / sqrt(var + (double)eps);

        // Normalize and divide by constant
        for (uint32_t oh = 0; oh < Oh; ++oh) {
            for (uint32_t ow = 0; ow < Ow; ++ow) {
                uint64_t idx = y_index(n, co, oh, ow);
                double v = (double)yPtr[idx];
                v = (v - mean) * inv_std;
                v = v / (double)div_val;
                yPtr[idx] = (DTYPE_T)v;
            }
        }
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv2d_instance_norm_divide_custom_impl_npu(const at::Tensor& x,
                                                       const at::Tensor& weight,
                                                       const at::Tensor& bias,
                                                       const at::Tensor& divide_by) {
    TORCH_CHECK(x.dim() == 4, "Input x must be NCHW 4D tensor");
    TORCH_CHECK(weight.dim() == 4, "Weight must be [Cout, Cin, Kh, Kw]");
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);
    auto Cout = weight.size(0);
    auto Kh = weight.size(2);
    auto Kw = weight.size(3);
    TORCH_CHECK(bias.numel() == Cout, "Bias size must match Cout");
    int64_t Oh = H - Kh + 1;
    int64_t Ow = W - Kw + 1;
    TORCH_CHECK(Oh > 0 && Ow > 0, "Invalid output spatial size after conv (no padding, stride=1)");

    at::Tensor result = at::empty({N, Cout, Oh, Ow}, x.options());
    EXEC_NPU_CMD(aclnnConv2dInstanceNormDivideCustom, x, weight, bias, divide_by, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_instance_norm_divide_custom", &conv2d_instance_norm_divide_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_instance_norm_divide_custom", &conv2d_instance_norm_divide_custom_impl_npu, "fused conv2d + instance norm + divide");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused Conv2d + InstanceNorm (no affine) + divide by constant using custom AscendC kernel.
    Assumes stride=1, padding=0, dilation=1.
    """
    def __init__(self, in_channels, out_channels, kernel_size, divide_by):
        super().__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)
        self.divide_by = float(divide_by)

    def forward(self, x):
        div_tensor = torch.tensor(self.divide_by, dtype=x.dtype, device=x.device).view(1)
        return custom_ops_lib.conv2d_instance_norm_divide_custom(x, self.conv.weight, self.conv.bias, div_tensor)
