host_operator_src="""
#include "conv_standard_1d_dilated_strided_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <cstdint>

namespace {
inline int64_t CalcLout(int64_t Lin, int64_t K, int64_t stride, int64_t dilation, int64_t padding = 0) {
    // PyTorch conv1d default: floor((Lin + 2*padding - dilation*(K-1) - 1)/stride + 1)
    return (Lin + 2 * padding - dilation * (K - 1) - 1) / stride + 1;
}
}

namespace optiling {
const uint32_t DEFAULT_BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto xShape = context->GetInputShape(0)->GetOriginShape();
    auto wShape = context->GetInputShape(1)->GetOriginShape();
    if (xShape.GetDimNum() != 3 || wShape.GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }
    int64_t N  = xShape.GetDim(0);
    int64_t Cin  = xShape.GetDim(1);
    int64_t Lin  = xShape.GetDim(2);
    int64_t Cout = wShape.GetDim(0);
    int64_t CinW = wShape.GetDim(1);
    int64_t K    = wShape.GetDim(2);
    if (CinW != Cin) {
        return ge::GRAPH_FAILED;
    }

    // attrs
    int64_t stride = 1;
    int64_t dilation = 1;
    // Try to get attributes from context
    try {
        stride = context->GetAttrInt("stride");
    } catch (...) {}
    try {
        dilation = context->GetAttrInt("dilation");
    } catch (...) {}

    int64_t Lout = CalcLout(Lin, K, stride, dilation, 0);
    if (Lout <= 0) {
        return ge::GRAPH_FAILED;
    }

    // simple 1D tiling on output length
    uint32_t tileLen = static_cast<uint32_t>(std::min<int64_t>(Lout, 1024));
    uint32_t tileNum = static_cast<uint32_t>((Lout + tileLen - 1) / tileLen);

    ConvStandard1dDilatedStridedCustomTilingData tiling;
    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_Cin(static_cast<uint32_t>(Cin));
    tiling.set_Cout(static_cast<uint32_t>(Cout));
    tiling.set_Lin(static_cast<uint32_t>(Lin));
    tiling.set_K(static_cast<uint32_t>(K));
    tiling.set_Lout(static_cast<uint32_t>(Lout));
    tiling.set_stride(static_cast<uint32_t>(stride));
    tiling.set_dilation(static_cast<uint32_t>(dilation));
    tiling.set_tileLen(tileLen);
    tiling.set_tileNum(tileNum);

    context->SetBlockDim(DEFAULT_BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    if (x_shape->GetDimNum() != 3 || w_shape->GetDimNum() != 3) {
        return GRAPH_FAILED;
    }
    int64_t N = x_shape->GetDim(0);
    int64_t Cin = x_shape->GetDim(1);
    int64_t Lin = x_shape->GetDim(2);
    int64_t Cout = w_shape->GetDim(0);
    int64_t CinW = w_shape->GetDim(1);
    int64_t K = w_shape->GetDim(2);
    if (CinW != Cin) {
        return GRAPH_FAILED;
    }
    int64_t stride = 1;
    int64_t dilation = 1;
    try { stride = context->GetAttrInt("stride"); } catch (...) {}
    try { dilation = context->GetAttrInt("dilation"); } catch (...) {}
    int64_t Lout = (Lin + 0 - dilation * (K - 1) - 1) / stride + 1;

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Lout);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#include <cstdint>

using DTYPE = float;

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(GM_ADDR x,
                                                                              GM_ADDR w,
                                                                              GM_ADDR b,
                                                                              GM_ADDR y,
                                                                              GM_ADDR workspace,
                                                                              GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t N       = tiling_data.N;
    const uint32_t Cin     = tiling_data.Cin;
    const uint32_t Cout    = tiling_data.Cout;
    const uint32_t Lin     = tiling_data.Lin;
    const uint32_t K       = tiling_data.K;
    const uint32_t Lout    = tiling_data.Lout;
    const uint32_t stride  = tiling_data.stride;
    const uint32_t dilation= tiling_data.dilation;

    const __gm__ DTYPE* xgm = reinterpret_cast<const __gm__ DTYPE*>(x);
    const __gm__ DTYPE* wgm = reinterpret_cast<const __gm__ DTYPE*>(w);
    const __gm__ DTYPE* bgm = reinterpret_cast<const __gm__ DTYPE*>(b);
    __gm__ DTYPE* ygm       = reinterpret_cast<__gm__ DTYPE*>(y);

    uint32_t totalOut = N * Cout * Lout;
    uint32_t blkNum = AscendC::GetBlockNum();
    uint32_t blkIdx = AscendC::GetBlockIdx();
    uint32_t chunk = (totalOut + blkNum - 1) / blkNum;
    uint32_t start = blkIdx * chunk;
    uint32_t end = start + chunk;
    if (end > totalOut) end = totalOut;

    for (uint32_t idx = start; idx < end; ++idx) {
        // decode n, co, lo from linear index
        uint32_t n  = idx / (Cout * Lout);
        uint32_t rem = idx % (Cout * Lout);
        uint32_t co = rem / Lout;
        uint32_t lo = rem % Lout;

        DTYPE acc = bgm ? bgm[co] : (DTYPE)0;

        uint64_t xBatchBase = static_cast<uint64_t>(n) * Cin * Lin;
        uint64_t wOutChBase = static_cast<uint64_t>(co) * Cin * K;

        uint32_t inStart = lo * stride;
        // compute convolution sum
        for (uint32_t ci = 0; ci < Cin; ++ci) {
            uint64_t xChBase = xBatchBase + static_cast<uint64_t>(ci) * Lin;
            uint64_t wChBase = wOutChBase + static_cast<uint64_t>(ci) * K;
            #pragma unroll
            for (uint32_t k = 0; k < K; ++k) {
                uint32_t inPos = inStart + k * dilation;
                // No padding; positions are valid by construction of Lout
                DTYPE xv = xgm[xChBase + inPos];
                DTYPE ww = wgm[wChBase + k];
                acc = AscendC::Mad(xv, ww, acc);
            }
        }
        uint64_t yBase = static_cast<uint64_t>(n) * Cout * Lout + static_cast<uint64_t>(co) * Lout + lo;
        ygm[yBase] = acc;
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline int64_t calc_lout_1d(int64_t Lin, int64_t K, int64_t stride, int64_t dilation, int64_t padding=0) {
    return (Lin + 2 * padding - dilation * (K - 1) - 1) / stride + 1;
}

at::Tensor conv_standard_1d_dilated_strided_custom_impl_npu(const at::Tensor& x,
                                                            const at::Tensor& w,
                                                            const at::Tensor& b,
                                                            int64_t stride,
                                                            int64_t dilation) {
    TORCH_CHECK(x.dim() == 3, "x must be 3D (N,C,L)");
    TORCH_CHECK(w.dim() == 3, "w must be 3D (Cout,Cin,K)");
    TORCH_CHECK(x.size(1) == w.size(1), "Cin mismatch");
    int64_t N = x.size(0);
    int64_t Lin = x.size(2);
    int64_t Cout = w.size(0);
    int64_t K = w.size(2);
    int64_t Lout = calc_lout_1d(Lin, K, stride, dilation, 0);

    auto out_sizes = std::vector<int64_t>{N, Cout, Lout};
    at::Tensor result = at::empty(out_sizes, x.options().memory_format(at::MemoryFormat::Contiguous));

    // Pass attrs stride and dilation to the op
    EXEC_NPU_CMD(aclnnConvStandard1dDilatedStridedCustom, x, w, b, stride, dilation, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_standard_1d_dilated_strided_custom", &conv_standard_1d_dilated_strided_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_standard_1d_dilated_strided_custom", &conv_standard_1d_dilated_strided_custom_impl_npu, "Conv1d (stride, dilation) custom AscendC");

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Custom Conv1d implemented via AscendC kernel. Supports stride and dilation, no padding.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, dilation: int = 1, bias: bool = False):
        super().__init__()
        # initialize weights similar to nn.Conv1d for fair replacement
        conv_ref = nn.Conv1d(in_channels, out_channels, kernel_size, stride=stride, dilation=dilation, bias=bias)
        self.weight = nn.Parameter(conv_ref.weight.detach())
        if bias:
            self.bias = nn.Parameter(conv_ref.bias.detach())
        else:
            self.bias = None
        self.stride = int(stride)
        self.dilation = int(dilation)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        b = self.bias
        if b is None:
            # create a zero bias on the fly
            b = torch.zeros(self.weight.size(0), dtype=x.dtype, device=x.device)
        return custom_ops_lib.conv_standard_1d_dilated_strided_custom(x, self.weight, b, self.stride, self.dilation)
'''
