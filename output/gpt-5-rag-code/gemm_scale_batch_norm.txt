kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmScaleBatchNorm {
public:
    __aicore__ inline KernelGemmScaleBatchNorm() {}
    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
        GM_ADDR scale, GM_ADDR gamma, GM_ADDR beta,
        GM_ADDR running_mean, GM_ADDR running_var,
        GM_ADDR y, uint32_t M, uint32_t N, uint32_t K,
        uint32_t tileNum, float eps)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->eps = eps;
        this->blockM = M / AscendC::GetBlockNum();
        this->tileNum = this->blockM;
        this->tileLength = this->N;

        xGm.SetGlobalBuffer((__gm__ float *)x + (uint64_t)AscendC::GetBlockIdx() * (uint64_t)blockM * (uint64_t)K, (uint64_t)blockM * (uint64_t)K);
        wGm.SetGlobalBuffer((__gm__ float *)weight, (uint64_t)N * (uint64_t)K);
        bGm.SetGlobalBuffer((__gm__ float *)bias, (uint64_t)N);
        yGm.SetGlobalBuffer((__gm__ float *)y + (uint64_t)AscendC::GetBlockIdx() * (uint64_t)blockM * (uint64_t)N, (uint64_t)blockM * (uint64_t)N);

        scaleGm.SetGlobalBuffer((__gm__ float *)scale, (uint64_t)N);
        gammaGm.SetGlobalBuffer((__gm__ float *)gamma, (uint64_t)N);
        betaGm.SetGlobalBuffer((__gm__ float *)beta, (uint64_t)N);
        meanGm.SetGlobalBuffer((__gm__ float *)running_mean, (uint64_t)N);
        varGm.SetGlobalBuffer((__gm__ float *)running_var, (uint64_t)N);

        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueScale, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueGamma, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueBeta, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueMean, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueVar, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void GemmCompute(GM_ADDR workspace)
    {
        // Use Matmul engine to compute Y = X * W^T + B into yGm
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSN> A_TYPE;
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true, LayoutMode::BSN> B_TYPE;
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSN> C_TYPE;
        typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> BIAS_TYPE;
        BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> matmulKernel;

        TCubeTiling cubeTiling;
        cubeTiling.shareMode = 0;
        cubeTiling.shareL1Size = FULL_L1_SIZE;
        cubeTiling.shareL0CSize = FULL_L0C_SIZE;
        cubeTiling.shareUbSize = 0;

        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulKernel.matmulObj, &cubeTiling);
        // Inputs: A = xGm (blockM x K), B = wGm (N x K, transposed inside), bias = bGm, output = yGm (blockM x N)
        matmulKernel.Init(reinterpret_cast<GM_ADDR>(xGm.GetHeadPtr()), reinterpret_cast<GM_ADDR>(wGm.GetHeadPtr()),
                          reinterpret_cast<GM_ADDR>(bGm.GetHeadPtr()), reinterpret_cast<GM_ADDR>(yGm.GetHeadPtr()),
                          workspace, cubeTiling);
        matmulKernel.Process<true>(&pipe, 1, 1);
    }

    __aicore__ inline void Process()
    {
        // After GEMM, apply per-channel scale and batch norm: y = ((y * scale - mean) / sqrt(var + eps)) * gamma + beta
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY.AllocTensor<float>();
        AscendC::LocalTensor<float> scaleLocal = inQueueScale.AllocTensor<float>();
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma.AllocTensor<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueBeta.AllocTensor<float>();
        AscendC::LocalTensor<float> meanLocal = inQueueMean.AllocTensor<float>();
        AscendC::LocalTensor<float> varLocal = inQueueVar.AllocTensor<float>();

        // row index within this block
        uint32_t row = progress % this->tileNum;
        AscendC::DataCopy(yLocal, yGm[row * this->tileLength], this->tileLength);
        AscendC::DataCopy(scaleLocal, scaleGm[0], this->tileLength);
        AscendC::DataCopy(gammaLocal, gammaGm[0], this->tileLength);
        AscendC::DataCopy(betaLocal, betaGm[0], this->tileLength);
        AscendC::DataCopy(meanLocal, meanGm[0], this->tileLength);
        AscendC::DataCopy(varLocal, varGm[0], this->tileLength);

        inQueueY.EnQue(yLocal);
        inQueueScale.EnQue(scaleLocal);
        inQueueGamma.EnQue(gammaLocal);
        inQueueBeta.EnQue(betaLocal);
        inQueueMean.EnQue(meanLocal);
        inQueueVar.EnQue(varLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY.DeQue<float>();
        AscendC::LocalTensor<float> scaleLocal = inQueueScale.DeQue<float>();
        AscendC::LocalTensor<float> gammaLocal = inQueueGamma.DeQue<float>();
        AscendC::LocalTensor<float> betaLocal = inQueueBeta.DeQue<float>();
        AscendC::LocalTensor<float> meanLocal = inQueueMean.DeQue<float>();
        AscendC::LocalTensor<float> varLocal = inQueueVar.DeQue<float>();

        AscendC::LocalTensor<float> zLocal = outQueueZ.AllocTensor<float>();

        // y = y * scale
        AscendC::Mul(yLocal, yLocal, scaleLocal, this->tileLength);
        // y = y - mean
        AscendC::Sub(yLocal, yLocal, meanLocal, this->tileLength);
        // denom = sqrt(var + eps)
        AscendC::LocalTensor<float> denomLocal = pipe.AllocTmpTensor<float>(this->tileLength);
        AscendC::Adds(denomLocal, varLocal, eps, this->tileLength);
        AscendC::Sqrt(denomLocal, denomLocal, this->tileLength);
        // y = y / denom
        AscendC::Div(yLocal, yLocal, denomLocal, this->tileLength);
        // y = y * gamma
        AscendC::Mul(yLocal, yLocal, gammaLocal, this->tileLength);
        // z = y + beta
        AscendC::Add(zLocal, yLocal, betaLocal, this->tileLength);

        outQueueZ.EnQue<float>(zLocal);

        inQueueY.FreeTensor(yLocal);
        inQueueScale.FreeTensor(scaleLocal);
        inQueueGamma.FreeTensor(gammaLocal);
        inQueueBeta.FreeTensor(betaLocal);
        inQueueMean.FreeTensor(meanLocal);
        inQueueVar.FreeTensor(varLocal);
        pipe.FreeTmpTensor(denomLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ.DeQue<float>();
        uint32_t row = progress % this->tileNum;
        AscendC::DataCopy(yGm[row * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY, inQueueScale, inQueueGamma, inQueueBeta, inQueueMean, inQueueVar;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> yGm;

    AscendC::GlobalTensor<float> scaleGm;
    AscendC::GlobalTensor<float> gammaGm;
    AscendC::GlobalTensor<float> betaGm;
    AscendC::GlobalTensor<float> meanGm;
    AscendC::GlobalTensor<float> varGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t blockM;
    uint32_t tileNum;
    uint32_t tileLength;
    float eps;
};

extern "C" __global__ __aicore__ void gemm_scale_batch_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scale, GM_ADDR bn_gamma,
    GM_ADDR bn_beta, GM_ADDR running_mean, GM_ADDR running_var,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    uint32_t M = tiling_data.M;
    uint32_t N = tiling_data.N;
    uint32_t K = tiling_data.K;
    uint32_t tileNum = tiling_data.tileNum;
    float eps = tiling_data.eps;

    KernelGemmScaleBatchNorm op;
    op.Init(x, weight, bias, scale, bn_gamma, bn_beta, running_mean, running_var, y, M, N, K, tileNum, eps);
    op.GemmCompute(workspace);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused GEMM + scale + BatchNorm using a custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, scale_shape, eps=1e-5, momentum=0.1):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.scale = torch.nn.Parameter(torch.randn(scale_shape))
        self.bn = torch.nn.BatchNorm1d(out_features, eps=eps, momentum=momentum)

    def forward(self, x):
        return custom_ops_lib.gemm_scale_batch_norm_custom(
            x,
            self.gemm.weight,
            self.gemm.bias,
            self.scale,
            self.bn.weight,
            self.bn.bias,
            self.bn.running_mean,
            self.bn.running_var,
            float(self.bn.eps)
        )
