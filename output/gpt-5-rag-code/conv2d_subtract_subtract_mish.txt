kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConv2dSubtractSubtractMish {
public:
    __aicore__ inline KernelConv2dSubtractSubtractMish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR s1, GM_ADDR s2, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        s1Gm.SetGlobalBuffer((__gm__ DTYPE_Y *)s1 + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        s2Gm.SetGlobalBuffer((__gm__ DTYPE_Y *)s2 + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueS1, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueS2, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> s1Local = inQueueS1.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> s2Local = inQueueS2.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(s1Local, s1Gm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(s2Local, s2Gm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueS1.EnQue(s1Local);
        inQueueS2.EnQue(s2Local);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> s1Local = inQueueS1.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> s2Local = inQueueS2.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // yLocal = x - s1
        AscendC::Sub(yLocal, xLocal, s1Local, this->tileLength);
        // yLocal = yLocal - s2
        AscendC::Sub(yLocal, yLocal, s2Local, this->tileLength);

        // Reuse xLocal as temporary buffer: xLocal = exp(yLocal)
        AscendC::Exp(xLocal, yLocal, this->tileLength);
        // xLocal = xLocal + 1.0
        AscendC::Adds(xLocal, xLocal, static_cast<DTYPE_X>(1.0f), this->tileLength);
        // xLocal = log(xLocal)
        AscendC::Log(xLocal, xLocal, this->tileLength);
        // xLocal = tanh(xLocal)
        AscendC::Tanh(xLocal, xLocal, this->tileLength);
        // yLocal = yLocal * xLocal  (Mish: y = x * tanh(softplus(x)))
        AscendC::Mul(yLocal, yLocal, xLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueS1.FreeTensor(s1Local);
        inQueueS2.FreeTensor(s2Local);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueS1, inQueueS2;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> s1Gm;
    AscendC::GlobalTensor<DTYPE_Y> s2Gm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_subtract_subtract_mish_custom(GM_ADDR x, GM_ADDR s1, GM_ADDR s2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractSubtractMish op;
    op.Init(x, s1, s2, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: performs Conv2d in PyTorch, then fused subtract-subtract-Mish via custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract_value_1, subtract_value_2):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract_value_1 = float(subtract_value_1)
        self.subtract_value_2 = float(subtract_value_2)

    def forward(self, x):
        x = self.conv(x)
        s1 = torch.full_like(x, self.subtract_value_1)
        s2 = torch.full_like(x, self.subtract_value_2)
        y = custom_ops_lib.conv2d_subtract_subtract_mish_custom(x, s1, s2)
        return y
