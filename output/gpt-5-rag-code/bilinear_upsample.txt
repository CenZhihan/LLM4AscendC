kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 1;

class KernelBilinearUpsampleCustom {
public:
    __aicore__ inline KernelBilinearUpsampleCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                 uint64_t totalOutLength,
                                 uint32_t tileNum,
                                 uint32_t n, uint32_t c,
                                 uint32_t inH, uint32_t inW,
                                 uint32_t outH, uint32_t outW)
    {
        this->n = n;
        this->c = c;
        this->inH = inH;
        this->inW = inW;
        this->outH = outH;
        this->outW = outW;
        this->totalOutLength = totalOutLength;
        this->tileNum = tileNum;

        blockNum = AscendC::GetBlockNum();
        blockIdx = AscendC::GetBlockIdx();

        // ceil-divide total per-block length
        uint64_t perBlock = (totalOutLength + blockNum - 1) / blockNum;
        blockStart = perBlock * blockIdx;
        blockEnd = blockStart + perBlock;
        if (blockEnd > totalOutLength) blockEnd = totalOutLength;
        blockLength = (blockEnd > blockStart) ? (blockEnd - blockStart) : 0;

        uint64_t denom = static_cast<uint64_t>(tileNum) * BUFFER_NUM;
        tileLength = denom ? static_cast<uint32_t>((blockLength + denom - 1) / denom) : static_cast<uint32_t>(blockLength);
        if (tileLength == 0 && blockLength > 0) tileLength = static_cast<uint32_t>(blockLength);

        uint64_t totalInLength = static_cast<uint64_t>(n) * c * inH * inW;
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, totalInLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + blockStart, blockLength);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (blockLength == 0) return;
        int32_t loopCount = static_cast<int32_t>(tileNum) * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            uint64_t startWithinBlock = static_cast<uint64_t>(i) * tileLength;
            if (startWithinBlock >= blockLength) break;
            uint32_t curLen = static_cast<uint32_t>(AscendC::Min<uint64_t>(tileLength, blockLength - startWithinBlock));
            ComputeAndStore(startWithinBlock, curLen);
        }
    }

private:
    __aicore__ inline void ComputeAndStore(uint64_t startWithinBlock, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        float scaleH = (outH > 0) ? static_cast<float>(inH) / static_cast<float>(outH) : 0.f;
        float scaleW = (outW > 0) ? static_cast<float>(inW) / static_cast<float>(outW) : 0.f;

        for (uint32_t i = 0; i < curLen; ++i) {
            uint64_t gidx = blockStart + startWithinBlock + i;

            uint64_t ow = gidx % outW;
            uint64_t tmp1 = gidx / outW;
            uint64_t oh = tmp1 % outH;
            uint64_t tmp2 = tmp1 / outH;
            uint64_t cc = tmp2 % c;
            uint64_t nn = tmp2 / c;

            float fy = (static_cast<float>(oh) + 0.5f) * scaleH - 0.5f;
            float fx = (static_cast<float>(ow) + 0.5f) * scaleW - 0.5f;

            float y0f = floorf(fy);
            float x0f = floorf(fx);
            int32_t y0 = static_cast<int32_t>(y0f);
            int32_t x0 = static_cast<int32_t>(x0f);
            int32_t y1 = y0 + 1;
            int32_t x1 = x0 + 1;

            float wy1 = fy - y0f;
            float wx1 = fx - x0f;
            float wy0 = 1.0f - wy1;
            float wx0 = 1.0f - wx1;

            int32_t y0c = AscendC::Max<int32_t>(0, AscendC::Min<int32_t>(static_cast<int32_t>(inH - 1), y0));
            int32_t y1c = AscendC::Max<int32_t>(0, AscendC::Min<int32_t>(static_cast<int32_t>(inH - 1), y1));
            int32_t x0c = AscendC::Max<int32_t>(0, AscendC::Min<int32_t>(static_cast<int32_t>(inW - 1), x0));
            int32_t x1c = AscendC::Max<int32_t>(0, AscendC::Min<int32_t>(static_cast<int32_t>(inW - 1), x1));

            uint64_t baseNC = ((nn * c + cc) * inH);
            uint64_t idx00 = (baseNC + static_cast<uint64_t>(y0c)) * inW + static_cast<uint64_t>(x0c);
            uint64_t idx01 = (baseNC + static_cast<uint64_t>(y0c)) * inW + static_cast<uint64_t>(x1c);
            uint64_t idx10 = (baseNC + static_cast<uint64_t>(y1c)) * inW + static_cast<uint64_t>(x0c);
            uint64_t idx11 = (baseNC + static_cast<uint64_t>(y1c)) * inW + static_cast<uint64_t>(x1c);

            float v00 = xGm.GetValue(idx00);
            float v01 = xGm.GetValue(idx01);
            float v10 = xGm.GetValue(idx10);
            float v11 = xGm.GetValue(idx11);

            float outv = wy0 * (wx0 * v00 + wx1 * v01) + wy1 * (wx0 * v10 + wx1 * v11);
            yLocal.SetValue(i, static_cast<DTYPE_Y>(outv));
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        AscendC::LocalTensor<DTYPE_Y> yLocalOut = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[startWithinBlock], yLocalOut, curLen);
        outQueueY.FreeTensor(yLocalOut);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, inH, inW, outH, outW;
    uint64_t totalOutLength;
    uint32_t tileNum;
    uint64_t blockStart, blockEnd, blockLength;
    uint32_t tileLength;
    uint32_t blockNum;
    uint32_t blockIdx;
};

extern "C" __global__ __aicore__ void bilinear_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBilinearUpsampleCustom op;
    op.Init(x, y,
            tiling_data.totalOutLength,
            tiling_data.tileNum,
            tiling_data.n, tiling_data.c,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW);
    op.Process();
}
