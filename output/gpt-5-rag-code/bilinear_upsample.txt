#include "bilinear_upsample_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto *in_shape = context->GetInputShape(0);
    auto inN = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(0));
    auto inC = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(1));
    auto inH = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(2));
    auto inW = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(3));

    uint32_t outH = inH * 2;
    uint32_t outW = inW * 2;

    uint64_t totalOutLength = static_cast<uint64_t>(inN) * inC * outH * outW;

    BilinearUpsampleCustomTilingData tiling;
    context->SetBlockDim(BLOCK_DIM);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_totalOutLength(totalOutLength);
    tiling.set_n(inN);
    tiling.set_c(inC);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    auto n = x_shape->GetOriginShape().GetDim(0);
    auto c = x_shape->GetOriginShape().GetDim(1);
    auto h = x_shape->GetOriginShape().GetDim(2);
    auto w = x_shape->GetOriginShape().GetDim(3);

    gert::Shape out_shape = *x_shape;
    out_shape.GetOriginShape().SetDim(0, n);
    out_shape.GetOriginShape().SetDim(1, c);
    out_shape.GetOriginShape().SetDim(2, h * 2);
    out_shape.GetOriginShape().SetDim(3, w * 2);

    *y_shape = out_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src=""
"#include \"kernel_operator.h\"\n"
"#include <cmath>\n"
"\n"
"constexpr int32_t BUFFER_NUM = 1;\n"
"\n"
"class KernelBilinearUpsampleCustom {\n"
"public:\n"
"    __aicore__ inline KernelBilinearUpsampleCustom() {}\n"
"    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,\n"
"                                 uint64_t totalOutLength,\n"
"                                 uint32_t tileNum,\n"
"                                 uint32_t n, uint32_t c,\n"
"                                 uint32_t inH, uint32_t inW,\n"
"                                 uint32_t outH, uint32_t outW)\n"
"    {\n"
"        this->n = n;\n"
"        this->c = c;\n"
"        this->inH = inH;\n"
"        this->inW = inW;\n"
"        this->outH = outH;\n"
"        this->outW = outW;\n"
"        this->totalOutLength = totalOutLength;\n"
"        this->tileNum = tileNum;\n"
"\n"
"        blockNum = AscendC::GetBlockNum();\n"
"        blockIdx = AscendC::GetBlockIdx();\n"
"\n"
"        // ceil-divide total per-block length\n"
"        uint64_t perBlock = (totalOutLength + blockNum - 1) / blockNum;\n"
"        blockStart = perBlock * blockIdx;\n"
"        blockEnd = blockStart + perBlock;\n"
"        if (blockEnd > totalOutLength) blockEnd = totalOutLength;\n"
"        blockLength = (blockEnd > blockStart) ? (blockEnd - blockStart) : 0;\n"
"\n"
"        uint64_t denom = static_cast<uint64_t>(tileNum) * BUFFER_NUM;\n"
"        tileLength = denom ? static_cast<uint32_t>((blockLength + denom - 1) / denom) : static_cast<uint32_t>(blockLength);\n"
"        if (tileLength == 0 && blockLength > 0) tileLength = static_cast<uint32_t>(blockLength);\n"
"\n"
"        uint64_t totalInLength = static_cast<uint64_t>(n) * c * inH * inW;\n"
"        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, totalInLength);\n"
"        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + blockStart, blockLength);\n"
"\n"
"        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileLength * sizeof(DTYPE_Y));\n"
"    }\n"
"\n"
"    __aicore__ inline void Process()\n"
"    {\n"
"        if (blockLength == 0) return;\n"
"        int32_t loopCount = static_cast<int32_t>(tileNum) * BUFFER_NUM;\n"
"        for (int32_t i = 0; i < loopCount; ++i) {\n"
"            uint64_t startWithinBlock = static_cast<uint64_t>(i) * tileLength;\n"
"            if (startWithinBlock >= blockLength) break;\n"
"            uint32_t curLen = static_cast<uint32_t>(AscendC::Min<uint64_t>(tileLength, blockLength - startWithinBlock));\n"
"            ComputeAndStore(startWithinBlock, curLen);\n"
"        }\n"
"    }\n"
"\n"
"private:\n"
"    __aicore__ inline void ComputeAndStore(uint64_t startWithinBlock, uint32_t curLen)\n"
"    {\n"
"        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();\n"
"\n"
"        float scaleH = (outH > 0) ? static_cast<float>(inH) / static_cast<float>(outH) : 0.f;\n"
"        float scaleW = (outW > 0) ? static_cast<float>(inW) / static_cast<float>(outW) : 0.f;\n"
"\n"
"        for (uint32_t i = 0; i < curLen; ++i) {\n"
"            uint64_t gidx = blockStart + startWithinBlock + i;\n"
"\n"
"            uint64_t ow = gidx % outW;\n"
"            uint64_t tmp1 = gidx / outW;\n"
"            uint64_t oh = tmp1 % outH;\n"
"            uint64_t tmp2 = tmp1 / outH;\n"
"            uint64_t cc = tmp2 % c;\n"
"            uint64_t nn = tmp2 / c;\n"
"\n"
"            float fy = (static_cast<float>(oh) + 0.5f) * scaleH - 0.5f;\n"
"            float fx = (static_cast<float>(ow) + 0.5f) * scaleW - 0.5f;\n"
"\n"
"            float y0f = floorf(fy);\n"
"            float x0f = floorf(fx);\n"
"            int32_t y0 = static_cast<int32_t>(y0f);\n"
"            int32_t x0 = static_cast<int32_t>(x0f);\n"
"            int32_t y1 = y0 + 1;\n"
"            int32_t x1 = x0 + 1;\n"
"\n"
"            float wy1 = fy - y0f;\n"
"            float wx1 = fx - x0f;\n"
"            float wy0 = 1.0f - wy1;\n"
"            float wx0 = 1.0f - wx1;\n"
"\n"
"            int32_t y0c = AscendC::Max<int32_t>(0, AscendC::Min<int32_t>(static_cast<int32_t>(inH - 1), y0));\n"
"            int32_t y1c = AscendC::Max<int32_t>(0, AscendC::Min<int32_t>(static_cast<int32_t>(inH - 1), y1));\n"
"            int32_t x0c = AscendC::Max<int32_t>(0, AscendC::Min<int32_t>(static_cast<int32_t>(inW - 1), x0));\n"
"            int32_t x1c = AscendC::Max<int32_t>(0, AscendC::Min<int32_t>(static_cast<int32_t>(inW - 1), x1));\n"
"\n"
"            uint64_t baseNC = ((nn * c + cc) * inH);\n"
"            uint64_t idx00 = (baseNC + static_cast<uint64_t>(y0c)) * inW + static_cast<uint64_t>(x0c);\n"
"            uint64_t idx01 = (baseNC + static_cast<uint64_t>(y0c)) * inW + static_cast<uint64_t>(x1c);\n"
"            uint64_t idx10 = (baseNC + static_cast<uint64_t>(y1c)) * inW + static_cast<uint64_t>(x0c);\n"
"            uint64_t idx11 = (baseNC + static_cast<uint64_t>(y1c)) * inW + static_cast<uint64_t>(x1c);\n"
"\n"
"            float v00 = xGm.GetValue(idx00);\n"
"            float v01 = xGm.GetValue(idx01);\n"
"            float v10 = xGm.GetValue(idx10);\n"
"            float v11 = xGm.GetValue(idx11);\n"
"\n"
"            float outv = wy0 * (wx0 * v00 + wx1 * v01) + wy1 * (wx0 * v10 + wx1 * v11);\n"
"            yLocal.SetValue(i, static_cast<DTYPE_Y>(outv));\n"
"        }\n"
"\n"
"        outQueueY.EnQue<DTYPE_Y>(yLocal);\n"
"        AscendC::LocalTensor<DTYPE_Y> yLocalOut = outQueueY.DeQue<DTYPE_Y>();\n"
"        AscendC::DataCopy(yGm[startWithinBlock], yLocalOut, curLen);\n"
"        outQueueY.FreeTensor(yLocalOut);\n"
"    }\n"
"\n"
"private:\n"
"    AscendC::TPipe pipe;\n"
"    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;\n"
"    AscendC::GlobalTensor<DTYPE_X> xGm;\n"
"    AscendC::GlobalTensor<DTYPE_Y> yGm;\n"
"\n"
"    uint32_t n, c, inH, inW, outH, outW;\n"
"    uint64_t totalOutLength;\n"
"    uint32_t tileNum;\n"
"    uint64_t blockStart, blockEnd, blockLength;\n"
"    uint32_t tileLength;\n"
"    uint32_t blockNum;\n"
"    uint32_t blockIdx;\n"
"};\n"
"\n"
"extern \"C\" __global__ __aicore__ void bilinear_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {\n"
"    GET_TILING_DATA(tiling_data, tiling);\n"
"    KernelBilinearUpsampleCustom op;\n"
"    op.Init(x, y,\n"
"            tiling_data.totalOutLength,\n"
"            tiling_data.tileNum,\n"
"            tiling_data.n, tiling_data.c,\n"
"            tiling_data.inH, tiling_data.inW,\n"
"            tiling_data.outH, tiling_data.outW);\n"
"    op.Process();\n"
"}\n"
""
