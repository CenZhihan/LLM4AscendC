#include "index_add_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM_COPY = 8;
const uint32_t DEFAULT_UPDATES_PER_TILE = 64;
const uint32_t DEFAULT_COL_CHUNK = 1024;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    IndexAddCustomTilingData tiling;

    // x: [N, C, ...], values: [numUpdates, rowSize]
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();

    // Get rowSize from values' 2nd dimension
    const auto& vShape = context->GetInputShape(2)->GetOriginShape();
    uint32_t rowSize = 1;
    if (vShape.GetDimNum() >= 2) {
        rowSize = static_cast<uint32_t>(vShape.GetDim(1));
    } else {
        // Fallback: if values is 1D (unlikely), set rowSize to 1
        rowSize = 1;
    }

    uint32_t numUpdates = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNumCopy(TILE_NUM_COPY);
    tiling.set_rowSize(rowSize);
    tiling.set_numUpdates(numUpdates);
    tiling.set_updatesPerTile(DEFAULT_UPDATES_PER_TILE);
    tiling.set_colChunk(DEFAULT_COL_CHUNK);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // for x->y copy pipeline

class KernelIndexAdd {
public:
    __aicore__ inline KernelIndexAdd() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, GM_ADDR values, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNumCopy,
                                uint32_t rowSize, uint32_t numUpdates,
                                uint32_t updatesPerTile, uint32_t colChunk)
    {
        // Save parameters
        this->totalLength = totalLength;
        this->tileNumCopy = tileNumCopy;
        this->rowSize = rowSize;
        this->numUpdates = numUpdates;
        this->updatesPerTile = updatesPerTile;
        this->colChunk = colChunk;

        // Block partition for contiguous copy
        this->blockLength = (totalLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t startOffset = blockLength * AscendC::GetBlockIdx();
        uint32_t endOffset = AscendC::Min(this->totalLength, startOffset + this->blockLength);
        this->blockLength = (endOffset > startOffset) ? (endOffset - startOffset) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + startOffset, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + startOffset, this->blockLength);

        // Global buffers for indices and values (use full buffer)
        indicesGm.SetGlobalBuffer((__gm__ DTYPE_Y *)indices, this->numUpdates);
        valuesGm.SetGlobalBuffer((__gm__ DTYPE_Z *)values, (uint64_t)this->numUpdates * this->rowSize);

        // Init copy pipeline buffers
        pipe.InitBuffer(inQueueX, BUFFER_NUM, GetCopyTileLength() * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, GetCopyTileLength() * sizeof(DTYPE_Z));

        // Init buffers for scatter-add stage
        pipe.InitBuffer(idxQueue, 1, this->updatesPerTile * sizeof(DTYPE_Y));
        pipe.InitBuffer(valQueue, 1, this->colChunk * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // Phase 1: copy x -> y (contiguous)
        CopyPhase();

        // Phase 2: scatter-add values into y at rows specified by indices
        ScatterAddPhase();
    }

private:
    __aicore__ inline uint32_t GetCopyTileLength() const
    {
        uint32_t tiles = tileNumCopy * BUFFER_NUM;
        if (tiles == 0) return 0;
        uint32_t tileLen = (this->blockLength + tiles - 1) / tiles;
        // align to vector friendly size if needed; keep simple
        return tileLen;
    }

    __aicore__ inline void CopyPhase()
    {
        if (this->blockLength == 0) {
            return;
        }
        uint32_t tileLen = GetCopyTileLength();
        uint32_t loopCount = (tileLen == 0) ? 0 : (AscendC::CeilDiv(this->blockLength, tileLen));
        for (uint32_t i = 0; i < loopCount; ++i) {
            uint32_t currLen = AscendC::Min(tileLen, this->blockLength - i * tileLen);

            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
            AscendC::DataCopy(xLocal, xGm[i * tileLen], currLen);
            // Straight copy
            AscendC::Cast(yLocal, xLocal, currLen);
            inQueueX.EnQue<DTYPE_X>(xLocal);
            outQueueY.EnQue<DTYPE_Z>(yLocal);

            AscendC::LocalTensor<DTYPE_X> xLocalUse = inQueueX.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Z> yLocalUse = outQueueY.DeQue<DTYPE_Z>();
            AscendC::DataCopy(yGm[i * tileLen], yLocalUse, currLen);
            inQueueX.FreeTensor(xLocalUse);
            outQueueY.FreeTensor(yLocalUse);
        }
    }

    __aicore__ inline void ScatterAddPhase()
    {
        if (this->numUpdates == 0 || this->rowSize == 0) {
            return;
        }

        // Partition updates across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t updatesPerBlock = AscendC::CeilDiv(this->numUpdates, blockNum);
        uint32_t startUpdate = updatesPerBlock * blockIdx;
        uint32_t endUpdate = AscendC::Min(this->numUpdates, startUpdate + updatesPerBlock);
        if (startUpdate >= endUpdate) {
            return;
        }

        for (uint32_t u = startUpdate; u < endUpdate; u += this->updatesPerTile) {
            uint32_t currTileUpdates = AscendC::Min(this->updatesPerTile, endUpdate - u);

            // Load a tile of indices to UB
            AscendC::LocalTensor<DTYPE_Y> idxUb = idxQueue.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(idxUb, indicesGm[u], currTileUpdates);
            idxQueue.EnQue<DTYPE_Y>(idxUb);
            AscendC::LocalTensor<DTYPE_Y> idxUse = idxQueue.DeQue<DTYPE_Y>();

            // Process rowSize in chunks along columns
            for (uint32_t col = 0; col < this->rowSize; col += this->colChunk) {
                uint32_t currColChunk = AscendC::Min(this->colChunk, this->rowSize - col);

                for (uint32_t j = 0; j < currTileUpdates; ++j) {
                    // Read destination row index
                    uint64_t dstRow = (uint64_t)idxUse.GetValue(j); // supports int32/int64
                    uint64_t dstOffset = dstRow * (uint64_t)this->rowSize + col;
                    uint64_t srcOffset = ((uint64_t)(u + j)) * (uint64_t)this->rowSize + col;

                    // Copy values chunk and atomic add to y
                    AscendC::LocalTensor<DTYPE_Z> valUb = valQueue.AllocTensor<DTYPE_Z>();
                    AscendC::DataCopy(valUb, valuesGm[srcOffset], currColChunk);

                    AscendC::SetAtomicAdd<DTYPE_Z>();
                    AscendC::DataCopy(yGm[dstOffset], valUb, currColChunk);
                    AscendC::SetAtomicNone();

                    valQueue.FreeTensor(valUb);
                }
            }

            idxQueue.FreeTensor(idxUse);
        }
    }

private:
    AscendC::TPipe pipe;

    // Queues/buffers
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::TQue<AscendC::TPosition::VECIN, 1> idxQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> valQueue;

    // GM tensors
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> indicesGm;  // int32/int64
    AscendC::GlobalTensor<DTYPE_Z> valuesGm;   // float
    AscendC::GlobalTensor<DTYPE_Z> yGm;        // float

    // Tiling parameters
    uint32_t totalLength;
    uint32_t tileNumCopy;
    uint32_t rowSize;
    uint32_t numUpdates;
    uint32_t updatesPerTile;
    uint32_t colChunk;

    // block partition for copy phase
    uint32_t blockLength;
};

extern "C" __global__ __aicore__ void index_add_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR values, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    KernelIndexAdd op;
    op.Init(x, indices, values, y,
            tiling_data.totalLength,
            tiling_data.tileNumCopy,
            tiling_data.rowSize,
            tiling_data.numUpdates,
            tiling_data.updatesPerTile,
            tiling_data.colChunk);
    op.Process();
}
