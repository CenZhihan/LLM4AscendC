#include "batch_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
static const uint32_t BUFFER_NUM = 2;
static const uint32_t BLOCK_DIM = 1; // keep block splitting aligned to channel segments

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    BatchNormCustomTilingData tiling;
    const gert::Shape* x_shape = context->GetInputShape(0);
    auto origin = x_shape->GetOriginShape();
    uint32_t dim_num = origin.GetDimNum();
    uint64_t total_len_u64 = origin.GetShapeSize();

    uint32_t c = 1;
    uint32_t hw = 1;
    if (dim_num >= 2) {
        c = static_cast<uint32_t>(origin.GetDim(1));
    }
    if (dim_num >= 4) {
        uint32_t h = static_cast<uint32_t>(origin.GetDim(dim_num - 2));
        uint32_t w = static_cast<uint32_t>(origin.GetDim(dim_num - 1));
        hw = h * w;
    }

    uint32_t total_len = static_cast<uint32_t>(total_len_u64);
    tiling.set_totalLength(total_len);
    tiling.set_channels(c);
    tiling.set_hw(hw);

    uint32_t channelSegments = (hw == 0) ? 0 : (total_len / hw);
    uint32_t tileNum = channelSegments / BUFFER_NUM;
    if (tileNum == 0) { tileNum = 1; }
    tiling.set_tileNum(tileNum);

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {

{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    return GRAPH_SUCCESS;

    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensors queued per pipe

class KernelBatchNorm {
public:
    __aicore__ inline KernelBatchNorm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR shift, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum, uint32_t channels, uint32_t hw)
    {
        this->channels = channels;
        this->hw = hw;

        // Use single block to keep channel segments aligned
        this->blockLength = totalLength / AscendC::GetBlockNum();
        // Process by full channel-segment tiles (size = hw)
        this->tileLength = hw;
        // Derive number of tiles per block
        uint32_t channelSegments = (hw == 0) ? 0 : (this->blockLength / hw);
        this->loopTiles = channelSegments; // number of channel segments in this block

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_S *)scale, this->channels);
        shiftGm.SetGlobalBuffer((__gm__ DTYPE_S *)shift, this->channels);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        // Each tile corresponds to one channel's HxW plane (channel segment)
        // Double-buffering across BUFFER_NUM
        int32_t loopCount = this->loopTiles * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        uint32_t segId = progress; // segment index within this block
        AscendC::DataCopy(xLocal, xGm[segId * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        uint32_t segId = progress; // segment index within this block
        // With BLOCK_DIM=1, the starting segment aligns so channel index is segId % channels
        uint32_t cIdx = (this->channels == 0) ? 0 : (segId % this->channels);
        DTYPE_S s = scaleGm.GetValue(cIdx);
        DTYPE_S b = shiftGm.GetValue(cIdx);

        // Elementwise: y = x * s + b over this HxW segment
        for (uint32_t i = 0; i < this->tileLength; ++i) {
            yLocal.SetValue(i, xLocal.GetValue(i) * s + b);
        }

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        uint32_t segId = progress;
        AscendC::DataCopy(yGm[segId * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_S> scaleGm;
    AscendC::GlobalTensor<DTYPE_S> shiftGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t blockLength;
    uint32_t tileLength;
    uint32_t loopTiles;
    uint32_t channels;
    uint32_t hw;
};

extern "C" __global__ __aicore__ void batch_norm_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR shift, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBatchNorm op;
    op.Init(x, scale, shift, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.channels, tiling_data.hw);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    BatchNorm2d forward replaced by custom AscendC kernel using per-channel affine params.
    """
    def __init__(self, num_features: int):
        super().__init__()
        # Keep a PyTorch BatchNorm2d to hold parameters and running stats
        self.bn = torch.nn.BatchNorm2d(num_features=num_features)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Inference-style BN using running stats:
        # scale[c] = weight[c] / sqrt(running_var[c] + eps)
        # shift[c] = bias[c] - running_mean[c] * scale[c]
        weight = self.bn.weight
        bias = self.bn.bias
        running_mean = self.bn.running_mean
        running_var = self.bn.running_var
        eps = self.bn.eps

        # Ensure parameters are on the same device/dtype as input
        dtype = x.dtype
        device = x.device
        weight = weight.to(device=device, dtype=dtype)
        bias = bias.to(device=device, dtype=dtype)
        running_mean = running_mean.to(device=device, dtype=dtype)
        running_var = running_var.to(device=device, dtype=dtype)

        scale = weight / torch.sqrt(running_var + eps)
        shift = bias - running_mean * scale

        return custom_ops_lib.batch_norm_custom(x, scale, shift)
'''
