host_operator_src="""
#include "index_select_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}
}

namespace optiling {
static const uint32_t MAX_BLOCKS = 32;
static const uint32_t DEFAULT_TILE_K = 256;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    IndexSelectCustomTilingData tiling;
    const auto* x_shape = context->GetInputShape(0);
    const auto* idx_shape = context->GetInputShape(1);

    // Expect x: [N, C], indices: [K]
    uint32_t rank = x_shape->GetOriginShape().GetDimNum();
    if (rank < 2) {
        return ge::GRAPH_FAILED;
    }

    uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t C = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t K = static_cast<uint32_t>(idx_shape->GetOriginShape().GetShapeSize());

    uint32_t blockDim = N < MAX_BLOCKS ? (N == 0 ? 1 : N) : MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    uint32_t rowsPerBlock = CeilDiv(N, blockDim);
    uint32_t tileK = (K >= DEFAULT_TILE_K) ? DEFAULT_TILE_K : (K == 0 ? 1 : K);

    tiling.set_rows(N);
    tiling.set_cols(C);
    tiling.set_k(K);
    tiling.set_tileK(tileK);
    tiling.set_rowsPerBlock(rowsPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* idx_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() < 2) {
        return GRAPH_FAILED;
    }
    int64_t N = x_shape->GetOriginShape().GetDim(0);
    int64_t K = idx_shape->GetOriginShape().GetShapeSize();
    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, K);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto xType = context->GetInputDataType(0);
    context->SetOutputDataType(0, xType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1; // simple double-buffer not necessary

class KernelIndexSelect {
public:
    __aicore__ inline KernelIndexSelect() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t k,
                                uint32_t tileK, uint32_t rowsPerBlock)
    {
        this->rows = rows;
        this->cols = cols;
        this->k = k;
        this->tileK = tileK;
        this->rowsPerBlock = rowsPerBlock;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Assign rows per block; final block may have fewer rows
        this->rowStart = blockIdx * rowsPerBlock;
        uint32_t maxRowsForThisBlock = 0;
        if (this->rowStart < rows) {
            maxRowsForThisBlock = rows - this->rowStart;
            if (maxRowsForThisBlock > rowsPerBlock) maxRowsForThisBlock = rowsPerBlock;
        }
        this->rowsInBlock = maxRowsForThisBlock;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, rows * cols);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_Y *)indices, k);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, rows * k);

        // init local buffers
        pipe.InitBuffer(idxQ, BUFFER_NUM, this->tileK * sizeof(DTYPE_Y));
        pipe.InitBuffer(yQ, BUFFER_NUM, this->tileK * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (rowsInBlock == 0 || k == 0) {
            return;
        }

        // Process each row assigned to this block
        for (uint32_t r = 0; r < rowsInBlock; ++r) {
            uint32_t rowGlobal = rowStart + r;
            // base offsets for x and y
            uint64_t xRowBase = (uint64_t)rowGlobal * (uint64_t)cols;
            uint64_t yRowBase = (uint64_t)rowGlobal * (uint64_t)k;

            // Iterate over K in tiles
            for (uint32_t off = 0; off < k; off += tileK) {
                uint32_t curTile = tileK;
                if (off + curTile > k) curTile = k - off;

                // Load indices tile
                LocalTensor<DTYPE_Y> indicesLocal = idxQ.AllocTensor<DTYPE_Y>();
                DataCopy(indicesLocal, idxGm[off], curTile);
                idxQ.EnQue(indicesLocal);

                // Gather this tile for the current row
                indicesLocal = idxQ.DeQue<DTYPE_Y>();
                LocalTensor<DTYPE_Z> yLocal = yQ.AllocTensor<DTYPE_Z>();

                // For each index in the tile, copy one element from x row into yLocal
                for (uint32_t i = 0; i < curTile; ++i) {
                    // Read index value from local indices buffer
                    int32_t colIdx = static_cast<int32_t>(indicesLocal.GetValue(i));
                    // Bounds-safe: if out of range, clamp to zero (optional safe-guard)
                    if (colIdx < 0) colIdx = 0;
                    if ((uint32_t)colIdx >= cols) colIdx = (int32_t)(cols - 1);

                    // Copy single element from xGm to yLocal[i]
                    DataCopy(yLocal[i], xGm[xRowBase + (uint64_t)colIdx], 1);
                }

                yQ.EnQue(yLocal);
                idxQ.FreeTensor(indicesLocal);

                // Store gathered tile for the current row
                yLocal = yQ.DeQue<DTYPE_Z>();
                DataCopy(yGm[yRowBase + off], yLocal, curTile);
                yQ.FreeTensor(yLocal);
            }
        }
    }

private:
    TPipe pipe;
    TQue<TPosition::VECIN, BUFFER_NUM> idxQ;
    TQue<TPosition::VECOUT, BUFFER_NUM> yQ;

    GlobalTensor<DTYPE_X> xGm;
    GlobalTensor<DTYPE_Y> idxGm;  // int32 indices
    GlobalTensor<DTYPE_Z> yGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t k;
    uint32_t tileK;
    uint32_t rowsPerBlock;

    uint32_t rowStart;
    uint32_t rowsInBlock;
};

extern "C" __global__ __aicore__ void index_select_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelIndexSelect op;
    op.Init(x, indices, y,
            tiling_data.rows, tiling_data.cols, tiling_data.k,
            tiling_data.tileK, tiling_data.rowsPerBlock);
    op.Process();
}
"""

    auto out_sizes = std::vector<int64_t>{self.size(0), indices32.size(0)};
