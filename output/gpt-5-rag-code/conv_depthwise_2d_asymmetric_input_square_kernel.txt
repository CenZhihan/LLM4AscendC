kernel_src="""
#include "kernel_operator.h"

// Naive depthwise convolution kernel implementation for stride=1, padding=0.
// Inputs:
//   x: [N, C, H, W]
//   w: [C, K, K]
// Output:
//   y: [N, C, H_out, W_out] with H_out = H - K + 1, W_out = W - K + 1

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const int32_t N  = static_cast<int32_t>(tiling_data.n());
    const int32_t C  = static_cast<int32_t>(tiling_data.c());
    const int32_t H  = static_cast<int32_t>(tiling_data.h());
    const int32_t W  = static_cast<int32_t>(tiling_data.w());
    const int32_t K  = static_cast<int32_t>(tiling_data.k());
    const int32_t OH = static_cast<int32_t>(tiling_data.out_h());
    const int32_t OW = static_cast<int32_t>(tiling_data.out_w());

    __gm__ DTYPE_X* x_ptr = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_Y* w_ptr = reinterpret_cast<__gm__ DTYPE_Y*>(w);
    __gm__ DTYPE_Z* y_ptr = reinterpret_cast<__gm__ DTYPE_Z*>(y);

    const int64_t in_nc_stride  = static_cast<int64_t>(H) * static_cast<int64_t>(W);
    const int64_t out_nc_stride = static_cast<int64_t>(OH) * static_cast<int64_t>(OW);
    const int64_t k_size        = static_cast<int64_t>(K) * static_cast<int64_t>(K);

    // Single-core execution, simple nested loops
    for (int32_t n = 0; n < N; ++n) {
        for (int32_t c = 0; c < C; ++c) {
            const int64_t in_base  = (static_cast<int64_t>(n) * C + c) * in_nc_stride;
            const int64_t out_base = (static_cast<int64_t>(n) * C + c) * out_nc_stride;
            const int64_t w_base   = static_cast<int64_t>(c) * k_size;

            for (int32_t oh = 0; oh < OH; ++oh) {
                for (int32_t ow = 0; ow < OW; ++ow) {
                    DTYPE_Z acc = static_cast<DTYPE_Z>(0);
                    // Convolution window
                    for (int32_t kh = 0; kh < K; ++kh) {
                        const int32_t ih = oh + kh; // stride=1, pad=0
                        const int64_t row_base = in_base + static_cast<int64_t>(ih) * W;
                        const int64_t w_row_base = w_base + static_cast<int64_t>(kh) * K;
                        for (int32_t kw = 0; kw < K; ++kw) {
                            const int32_t iw = ow + kw;
                            DTYPE_X xv = x_ptr[row_base + iw];
                            DTYPE_Y ww = w_ptr[w_row_base + kw];
                            acc += xv * ww;
                        }
                    }
                    y_ptr[out_base + static_cast<int64_t>(oh) * OW + ow] = acc;
                }
            }
        }
    }
}

at::Tensor conv_depthwise_2d_asymmetric_input_square_kernel_custom_impl_npu(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, C, H, W], w: [C, K, K]
    // Output y: [N, C, H-K+1, W-K+1] (stride=1, padding=0)

    auto C = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);
    auto K = w.size(1);

    auto OW = W - K + 1;

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        # Keep original module to host weights; enforce depthwise configuration
        self.conv2d = torch.nn.Conv2d(in_channels, out_channels, kernel_size=(kernel_size, kernel_size),
                                      stride=stride, padding=padding, groups=in_channels, bias=bias)
        assert stride == 1 and padding == 0, "This custom kernel supports stride=1, padding=0 only."
        assert out_channels == in_channels, "Depthwise conv requires out_channels == in_channels."
        assert bias is False or (self.conv2d.bias is None), "Bias is not supported by this custom kernel."

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Use custom NPU kernel: depthwise conv with stride=1 and padding=0
        return custom_ops_lib.conv_depthwise_2d_asymmetric_input_square_kernel_custom(x, self.conv2d.weight)
