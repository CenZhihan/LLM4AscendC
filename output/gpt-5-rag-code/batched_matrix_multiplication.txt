kernel_src="""
#include "kernel_operator.h"

class KernelBmm {
public:
    __aicore__ inline KernelBmm() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t batch, uint32_t M, uint32_t K, uint32_t N)
    {
        this->batch = batch;
        this->M = M;
        this->K = K;
        this->N = N;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // distribute batches across blocks
        this->batchPerBlock = (batch + blocks - 1) / blocks;
        uint32_t start = blockIdx * this->batchPerBlock;
        uint32_t end = start + this->batchPerBlock;
        if (end > batch) end = batch;
        this->localBatch = (end > start) ? (end - start) : 0;
        this->startBatch = start;

        // total elements for local buffers
        uint64_t aElems = (uint64_t)this->localBatch * M * K;
        uint64_t bElems = (uint64_t)this->localBatch * K * N;
        uint64_t cElems = (uint64_t)this->localBatch * M * N;

        aGm.SetGlobalBuffer((__gm__ DTYPE_X*)a + (uint64_t)start * M * K, aElems);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y*)b + (uint64_t)start * K * N, bElems);
        cGm.SetGlobalBuffer((__gm__ DTYPE_Z*)c + (uint64_t)start * M * N, cElems);
    }

    __aicore__ inline void Process()
    {
        if (localBatch == 0 || M == 0 || N == 0 || K == 0) {
            return;
        }

        // Perform batched matmul: for each local batch, C[b] = A[b] @ B[b]
        for (uint32_t lb = 0; lb < localBatch; ++lb) {
            // Offsets within local GM buffers
            uint64_t aOff = (uint64_t)lb * M * K;
            uint64_t bOff = (uint64_t)lb * K * N;
            uint64_t cOff = (uint64_t)lb * M * N;

            // Use AscendC matmul primitive
            // Set tensors for this batch slice and compute single GEMM
            matmulObj.SetTensorA(aGm[aOff], false);
            matmulObj.SetTensorB(bGm[bOff], false);
            matmulObj.IterateAll(cGm[cOff]);
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> aGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> cGm;

    uint32_t batch;
    uint32_t M;
    uint32_t K;
    uint32_t N;

    uint32_t batchPerBlock;
    uint32_t startBatch;
    uint32_t localBatch;

    // Matmul object provided by AscendC runtime
    AscendC::Matmul matmulObj;
};

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBmm op;
    op.Init(A, B, C, tiling_data.batch, tiling_data.M, tiling_data.K, tiling_data.N);
    op.Process();
}
"""
