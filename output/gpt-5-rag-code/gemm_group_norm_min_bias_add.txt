kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;      // tensor num for x/z queues
constexpr int32_t CONST_BUFFER_NUM = 1; // one buffer for constants

class KernelGemmGroupNormMinBiasAdd {
public:
    __aicore__ inline KernelGemmGroupNormMinBiasAdd() {}
    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR weight,
                                GM_ADDR bias_linear,
                                GM_ADDR gamma,
                                GM_ADDR beta,
                                GM_ADDR bias_add,
                                GM_ADDR z,
                                uint32_t batch,
                                uint32_t inFeat,
                                uint32_t outFeat,
                                uint32_t numGroups,
                                uint32_t tileNum)
    {
        this->batch = batch;
        this->inFeat = inFeat;
        this->outFeat = outFeat;
        this->numGroups = numGroups;
        this->tileNum = tileNum;
        this->epsilon = 1e-5f;

        // rows (batch) per block
        this->blockRows = batch / AscendC::GetBlockNum();
        // rows per tile
        this->tileRows = this->blockRows / tileNum / BUFFER_NUM;

        // Set global buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockRows * this->inFeat * AscendC::GetBlockIdx(),
                            this->blockRows * this->inFeat);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)weight, this->outFeat * this->inFeat);
        biasLinearGm.SetGlobalBuffer((__gm__ DTYPE_X *)bias_linear, this->outFeat);
        gammaGm.SetGlobalBuffer((__gm__ DTYPE_X *)gamma, this->outFeat);
        betaGm.SetGlobalBuffer((__gm__ DTYPE_X *)beta, this->outFeat);
        biasAddGm.SetGlobalBuffer((__gm__ DTYPE_X *)bias_add, this->outFeat);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockRows * this->outFeat * AscendC::GetBlockIdx(),
                            this->blockRows * this->outFeat);

        // init buffers
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileRows * this->inFeat * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileRows * this->outFeat * sizeof(DTYPE_Z));
        pipe.InitBuffer(inQueueConst, CONST_BUFFER_NUM, 4 * this->outFeat * sizeof(DTYPE_X)); // gamma, beta, bias_linear, bias_add
    }

    __aicore__ inline void LoadConstants()
    {
        AscendC::LocalTensor<DTYPE_X> constLocal = inQueueConst.AllocTensor<DTYPE_X>();
        // layout: [0:outFeat) gamma | [outFeat:2*outFeat) beta | [2*outFeat:3*outFeat) bias_linear | [3*outFeat:4*outFeat) bias_add
        AscendC::DataCopy(constLocal, gammaGm, this->outFeat);
        AscendC::DataCopy(constLocal[this->outFeat], betaGm, this->outFeat);
        AscendC::DataCopy(constLocal[2 * this->outFeat], biasLinearGm, this->outFeat);
        AscendC::DataCopy(constLocal[3 * this->outFeat], biasAddGm, this->outFeat);
        inQueueConst.EnQue(constLocal);
    }

    __aicore__ inline void Process()
    {
        LoadConstants();
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileRows * this->inFeat], this->tileRows * this->inFeat);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_X> constLocal = inQueueConst.DeQue<DTYPE_X>();

        DTYPE_X* gammaLocalPtr = constLocal.GetPtr();
        DTYPE_X* betaLocalPtr = constLocal.GetPtr() + this->outFeat;
        DTYPE_X* biasLinearLocalPtr = constLocal.GetPtr() + 2 * this->outFeat;
        DTYPE_X* biasAddLocalPtr = constLocal.GetPtr() + 3 * this->outFeat;

        // For each row in this tile
        for (uint32_t r = 0; r < this->tileRows; r++) {
            // Compute GEMM: y = x_row * W^T + bias_linear
            // Use zLocal as temporary buffer for y
            for (uint32_t oc = 0; oc < this->outFeat; oc++) {
                float acc = 0.0f;
                // load weight row oc into temp accum via direct access
                for (uint32_t ic = 0; ic < this->inFeat; ic++) {
                    float xv = static_cast<float>(xLocal[r * this->inFeat + ic]);
                    float wv = static_cast<float>(wGm[oc * this->inFeat + ic]);
                    acc += xv * wv;
                }
                acc += static_cast<float>(biasLinearLocalPtr[oc]);
                zLocal[r * this->outFeat + oc] = static_cast<DTYPE_Z>(acc);
            }

            // GroupNorm over channels (outFeat)
            uint32_t groupSize = this->outFeat / this->numGroups;
            for (uint32_t g = 0; g < this->numGroups; g++) {
                uint32_t start = g * groupSize;
                uint32_t end = start + groupSize;

                // compute mean
                float mean = 0.0f;
                for (uint32_t c = start; c < end; c++) {
                    mean += static_cast<float>(zLocal[r * this->outFeat + c]);
                }
                mean /= static_cast<float>(groupSize);

                // compute variance
                float var = 0.0f;
                for (uint32_t c = start; c < end; c++) {
                    float v = static_cast<float>(zLocal[r * this->outFeat + c]) - mean;
                    var += v * v;
                }
                var /= static_cast<float>(groupSize);
                float inv_std = 1.0f / AscendC::Sqrt(var + this->epsilon);

                // apply affine: (x - mean) * inv_std * gamma + beta
                for (uint32_t c = start; c < end; c++) {
                    float xhat = (static_cast<float>(zLocal[r * this->outFeat + c]) - mean) * inv_std;
                    float y = xhat * static_cast<float>(gammaLocalPtr[c]) + static_cast<float>(betaLocalPtr[c]);
                    zLocal[r * this->outFeat + c] = static_cast<DTYPE_Z>(y);
                }
            }

            // Reduce min across channels
            float minVal = AscendC::INFINITY_F;
            for (uint32_t c = 0; c < this->outFeat; c++) {
                float v = static_cast<float>(zLocal[r * this->outFeat + c]);
                minVal = AscendC::Min(minVal, v);
            }

            // Bias add with broadcasting: result[r, oc] = minVal + bias_add[oc]
            for (uint32_t c = 0; c < this->outFeat; c++) {
                float outv = minVal + static_cast<float>(biasAddLocalPtr[c]);
                zLocal[r * this->outFeat + c] = static_cast<DTYPE_Z>(outv);
            }
        }

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        // keep constLocal queued for reuse across tiles
        inQueueConst.EnQue(constLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        // Output layout: [1, outFeat, batch, 1] flattened as outFeat * batch
        AscendC::DataCopy(zGm[progress * this->tileRows * this->outFeat], zLocal, this->tileRows * this->outFeat);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, CONST_BUFFER_NUM> inQueueConst;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_X> biasLinearGm;
    AscendC::GlobalTensor<DTYPE_X> gammaGm;
    AscendC::GlobalTensor<DTYPE_X> betaGm;
    AscendC::GlobalTensor<DTYPE_X> biasAddGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t batch;
    uint32_t inFeat;
    uint32_t outFeat;
    uint32_t numGroups;
    uint32_t tileNum;
    uint32_t blockRows;
    uint32_t tileRows;
    float epsilon;
};

extern "C" __global__ __aicore__ void gemm_group_norm_min_bias_add_custom(GM_ADDR x,
                                                                          GM_ADDR weight,
                                                                          GM_ADDR bias_linear,
                                                                          GM_ADDR gamma,
                                                                          GM_ADDR beta,
                                                                          GM_ADDR bias_add,
                                                                          GM_ADDR z,
                                                                          GM_ADDR workspace,
                                                                          GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmGroupNormMinBiasAdd op;
    op.Init(x, weight, bias_linear, gamma, beta, bias_add, z,
            tiling_data.batch, tiling_data.inFeat, tiling_data.outFeat, tiling_data.numGroups, tiling_data.tileNum);
    op.Process();
}
