#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>

namespace {
inline int64_t floor_div(int64_t a, int64_t b) {
    return a / b;
}
}

namespace optiling {
const uint32_t DEFAULT_BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    using namespace optiling;
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    // Expect shape [N, C, L]
    int64_t n = 1, c = 1, length = 1;
    if (in_shape->GetOriginShape().GetDimNum() >= 3) {
        n       = in_shape->GetOriginShape().GetDim(0);
        c       = in_shape->GetOriginShape().GetDim(1);
        length  = in_shape->GetOriginShape().GetDim(2);
    } else {
        // Fallback: flatten all but last
        length = in_shape->GetOriginShape().GetShapeSize();
    }

    // Read attributes; if stride <= 0, use kernel_size as stride
    int64_t kernel_size = 1;
    int64_t stride      = 0;
    int64_t padding     = 0;

    // These attribute getters are framework-dependent; placeholders are provided here
    // Try to fetch attributes if provided, otherwise keep defaults
    if (context->HaveAttr("kernel_size")) {
        kernel_size = context->GetAttrInt("kernel_size");
    }
    if (context->HaveAttr("stride")) {
        stride = context->GetAttrInt("stride");
    }
    if (context->HaveAttr("padding")) {
        padding = context->GetAttrInt("padding");
    }
    if (stride <= 0) stride = kernel_size;

    int64_t out_length = 0;
    if (length > 0 && kernel_size > 0 && stride > 0) {
        out_length = floor_div((length + 2 * padding - kernel_size), stride) + 1;
        if (out_length < 0) out_length = 0;
    }

    uint32_t blockDim = static_cast<uint32_t>(std::min<int64_t>(DEFAULT_BLOCK_DIM, std::max<int64_t>(1, n * c)));
    context->SetBlockDim(blockDim);

    tiling.set_n(static_cast<uint32_t>(std::max<int64_t>(1, n)));
    tiling.set_c(static_cast<uint32_t>(std::max<int64_t>(1, c)));
    tiling.set_length(static_cast<uint32_t>(std::max<int64_t>(0, length)));
    tiling.set_kernelSize(static_cast<uint32_t>(std::max<int64_t>(1, kernel_size)));
    tiling.set_stride(static_cast<uint32_t>(std::max<int64_t>(1, stride)));
    tiling.set_padding(static_cast<uint32_t>(std::max<int64_t>(0, padding)));
    tiling.set_outLength(static_cast<uint32_t>(std::max<int64_t>(0, out_length)));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    int64_t n = 1, c = 1, length = 1;
    if (x_shape->GetOriginShape().GetDimNum() >= 3) {
        n       = x_shape->GetOriginShape().GetDim(0);
        c       = x_shape->GetOriginShape().GetDim(1);
        length  = x_shape->GetOriginShape().GetDim(2);
    } else {
        length  = x_shape->GetOriginShape().GetShapeSize();
    }

    int64_t kernel_size = 1;
    int64_t stride      = 0;
    int64_t padding     = 0;
    if (context->HaveAttr("kernel_size")) {
        kernel_size = context->GetAttrInt("kernel_size");
    }
    if (context->HaveAttr("stride")) {
        stride = context->GetAttrInt("stride");
    }
    if (context->HaveAttr("padding")) {
        padding = context->GetAttrInt("padding");
    }
    if (stride <= 0) stride = kernel_size;

    int64_t out_length = 0;
    if (length > 0 && kernel_size > 0 && stride > 0) {
        out_length = (length + 2 * padding - kernel_size) / stride + 1;
        if (out_length < 0) out_length = 0;
    }

    std::vector<int64_t> out_dims{n, c, out_length};
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

class KernelMaxPooling1d {
public:
    __aicore__ inline KernelMaxPooling1d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t n, uint32_t c, uint32_t length,
                                uint32_t k, uint32_t s, uint32_t p, uint32_t outL)
    {
        this->n = n;
        this->c = c;
        this->length = length;
        this->kernel = k;
        this->stride = s;
        this->pad = p;
        this->outLength = outL;

        totalLines = n * c;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(totalLines * length));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint32_t)(totalLines * outLength));

        pipe.InitBuffer(inQueueX, 1, length * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, 1, outLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t linesPerBlock = (totalLines + blockNum - 1) / blockNum;
        uint32_t startLine = blockIdx * linesPerBlock;
        uint32_t endLine = startLine + linesPerBlock;
        if (endLine > totalLines) endLine = totalLines;

        for (uint32_t line = startLine; line < endLine; ++line) {
            CopyIn(line);
            Compute(line);
            CopyOut(line);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t line)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[line * length], length);
        inQueueX.EnQue<DTYPE_X>(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*line*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        for (uint32_t t = 0; t < outLength; ++t) {
            int32_t start = (int32_t)t * (int32_t)stride - (int32_t)pad;
            // Use a large negative number to emulate -inf for padded positions
            DTYPE_Y maxVal = static_cast<DTYPE_Y>(-1.0e30f);
            for (uint32_t j = 0; j < kernel; ++j) {
                int32_t idx = start + (int32_t)j;
                if (idx >= 0 && (uint32_t)idx < length) {
                    DTYPE_X v = xLocal.GetValue((uint32_t)idx);
                    maxVal = v > maxVal ? v : maxVal;
                }
            }
            yLocal.SetValue(t, maxVal);
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t line)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[line * outLength], yLocal, outLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t length;
    uint32_t kernel;
    uint32_t stride;
    uint32_t pad;
    uint32_t outLength;
    uint32_t totalLines;
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1d op;
    op.Init(x, y,
            tiling_data.n,
            tiling_data.c,
            tiling_data.length,
            tiling_data.kernelSize,
            tiling_data.stride,
            tiling_data.padding,
            tiling_data.outLength);
    op.Process();
}
