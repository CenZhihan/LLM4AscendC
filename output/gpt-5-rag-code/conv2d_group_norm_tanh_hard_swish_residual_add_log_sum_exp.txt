kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using namespace AscendC;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t groups, float epsilon)
    {
        x_ptr = reinterpret_cast<__gm__ float*>(x);
        gamma_ptr = reinterpret_cast<__gm__ float*>(gamma);
        beta_ptr = reinterpret_cast<__gm__ float*>(beta);
        y_ptr = reinterpret_cast<__gm__ float*>(y);

        this->N = n;
        this->C = c;
        this->H = h;
        this->W = w;
        this->G = groups;
        this->eps = epsilon;

        this->HW = H * W;
        this->CHW = C * H * W;
        this->cpg = (groups == 0) ? C : (C / groups);
    }

    __aicore__ inline void Process()
    {
        uint32_t block_num = GetBlockNum();
        uint32_t block_idx = GetBlockIdx();
        if (block_num == 0) {
            return;
        }
        uint32_t n_per_block = (N + block_num - 1) / block_num;
        uint32_t n_start = block_idx * n_per_block;
        uint32_t n_end = n_start + n_per_block;
        if (n_start >= N) return;
        if (n_end > N) n_end = N;

        for (uint32_t n = n_start; n < n_end; ++n) {
            ComputeOneBatch(n);
        }
    }

private:
    __aicore__ inline float clampf(float v, float lo, float hi) {
        return v < lo ? lo : (v > hi ? hi : v);
    }

    __aicore__ inline void ComputeOneBatch(uint32_t n)
    {
        // Allocate local arrays for group stats (assuming G is reasonably small)
        // Initialize sums
        const uint32_t MAX_G = 1024;
        float sum[MAX_G];
        float sumsq[MAX_G];
        float mean[MAX_G];
        float invStd[MAX_G];

        uint32_t groups_used = G > 0 ? G : 1;
        for (uint32_t g = 0; g < groups_used && g < MAX_G; ++g) {
            sum[g] = 0.0f;
            sumsq[g] = 0.0f;
            mean[g] = 0.0f;
            invStd[g] = 0.0f;
        }

        // First pass: accumulate sums and sumsq per group over all channels in group and all H*W
        uint64_t base_n = static_cast<uint64_t>(n) * CHW;
        uint64_t num_per_group = static_cast<uint64_t>(cpg) * HW;
        for (uint32_t c = 0; c < C; ++c) {
            uint32_t g = (G == 0) ? 0 : (c / cpg);
            uint64_t base_nc = base_n + static_cast<uint64_t>(c) * HW;
            for (uint32_t idx = 0; idx < HW; ++idx) {
                float v = x_ptr[base_nc + idx];
                sum[g] += v;
                sumsq[g] += v * v;
            }
        }
        for (uint32_t g = 0; g < groups_used; ++g) {
            float m = sum[g] / static_cast<float>(num_per_group);
            float ex2 = sumsq[g] / static_cast<float>(num_per_group);
            float var = ex2 - m * m;
            if (var < 0.0f) var = 0.0f;
            mean[g] = m;
            invStd[g] = 1.0f / sqrtf(var + eps);
        }

        // Second stage: for each spatial location (h,w), compute logsumexp over channels of residual(x + hswish(tanh(gn(x))))
        for (uint32_t hw = 0; hw < HW; ++hw) {
            // First, find max over channels for numerical stability
            float maxv = -INFINITY;
            for (uint32_t c = 0; c < C; ++c) {
                uint32_t g = (G == 0) ? 0 : (c / cpg);
                uint64_t idx = base_n + static_cast<uint64_t>(c) * HW + hw;
                float xval = x_ptr[idx];

                // GroupNorm -> Tanh -> HardSwish -> Residual add
                float norm = (xval - mean[g]) * invStd[g];
                float affine = norm * gamma_ptr[c] + beta_ptr[c];
                float t = tanhf(affine);
                float hs = t * (clampf(t + 3.0f, 0.0f, 6.0f)) * (1.0f / 6.0f);
                float r = xval + hs;

                if (r > maxv) maxv = r;
            }
            if (!std::isfinite(maxv)) {
                maxv = 0.0f;
            }
            // Sum exp(r - maxv)
            float sumExp = 0.0f;
            for (uint32_t c = 0; c < C; ++c) {
                uint32_t g = (G == 0) ? 0 : (c / cpg);
                uint64_t idx = base_n + static_cast<uint64_t>(c) * HW + hw;
                float xval = x_ptr[idx];
                float norm = (xval - mean[g]) * invStd[g];
                float affine = norm * gamma_ptr[c] + beta_ptr[c];
                float t = tanhf(affine);
                float hs = t * (clampf(t + 3.0f, 0.0f, 6.0f)) * (1.0f / 6.0f);
                float r = xval + hs;
                sumExp += expf(r - maxv);
            }
            float out = logf(sumExp) + maxv;
            // write to y at channel 0
            uint64_t y_idx = static_cast<uint64_t>(n) * HW + hw; // since output C=1
            y_ptr[y_idx] = out;
        }
    }

private:
    __gm__ float* x_ptr;
    __gm__ float* gamma_ptr;
    __gm__ float* beta_ptr;
    __gm__ float* y_ptr;

    uint32_t N, C, H, W, G;
    uint32_t HW, CHW, cpg;
    float eps;
};

extern "C" __global__ __aicore__ void conv2d_group_norm_tanh_hard_swish_residual_add_log_sum_exp_custom(
    GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, gamma, beta, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.groups, tiling_data.epsilon);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a custom AscendC fused operator for:
    GroupNorm -> Tanh -> HardSwish -> Residual Add -> LogSumExp,
    applied on the Conv2d output.
    """
    def __init__(self, in_channels, out_channels, kernel_size, groups, eps=1e-5):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        # Keep GroupNorm module to manage parameters (weight/bias) and attrs.
        self.group_norm = torch.nn.GroupNorm(groups, out_channels, eps=eps)
        self.tanh = torch.nn.Tanh()
        self.hard_swish = torch.nn.Hardswish()

    def forward(self, x):
        x_conv = self.conv(x)
        # Fused custom op replaces: GroupNorm -> Tanh -> HardSwish -> Residual Add -> LogSumExp
        y = custom_ops_lib.conv2d_group_norm_tanh_hard_swish_residual_add_log_sum_exp_custom(
            x_conv, self.group_norm.weight, self.group_norm.bias, self.group_norm.num_groups, float(self.group_norm.eps)
        )
        return y
