kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // for double-buffering

// The kernel assumes input is laid out so that the normalization axis is the last dimension.
// That is, x is shaped as (G, C) flattened, where each group of length C is normalized.
// gamma is a 1-D tensor of length C applied elementwise after normalization.

class KernelRmsNorm {
public:
    __aicore__ inline KernelRmsNorm() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR y,
                                uint32_t totalLength, uint32_t groupSize,
                                uint32_t groupCount, uint32_t tileNum, float epsilon)
    {
        this->groupSize = groupSize;
        this->groupCountTotal = groupCount;
        this->tileNum = tileNum;
        this->epsilon = static_cast<DTYPE_X>(epsilon);

        // distribute groups among blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t groupsPerBlock = groupCount / blockNum;
        uint32_t remain = groupCount % blockNum;
        this->groupStart = groupsPerBlock * blockIdx + (blockIdx < remain ? blockIdx : remain);
        this->groupCountThisBlock = groupsPerBlock + (blockIdx < remain ? 1 : 0);

        // set global buffers starting at this block's offset
        uint32_t offsetElems = this->groupStart * this->groupSize;
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + offsetElems, this->groupCountThisBlock * this->groupSize);
        gammaGm.SetGlobalBuffer((__gm__ DTYPE_Y *)gamma, this->groupSize); // shared across groups
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + offsetElems, this->groupCountThisBlock * this->groupSize);

        // init local queues
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->groupSize * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueGamma, 1, this->groupSize * sizeof(DTYPE_Y)); // gamma loaded once
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->groupSize * sizeof(DTYPE_Z));

        // preload gamma once into a private local tensor
        gammaLocal = inQueueGamma.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(gammaLocal, gammaGm, this->groupSize);
        inQueueGamma.EnQue(gammaLocal);
    }

    __aicore__ inline void Process()
    {
        // process each group sequentially with double buffering
        for (uint32_t g = 0; g < this->groupCountThisBlock; ++g) {
            CopyIn(g);
            Compute(g);
            CopyOut(g);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t groupIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[groupIdx * this->groupSize], this->groupSize);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t groupIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> gammaLocalUse = inQueueGamma.DeQue<DTYPE_Y>(); // same tensor every time
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // temp buffer for squares
        AscendC::LocalTensor<DTYPE_X> sqLocal = tmpBuf.Alloc<DTYPE_X>(this->groupSize);

        // sqLocal = xLocal * xLocal
        AscendC::Mul(sqLocal, xLocal, xLocal, this->groupSize);

        // reduce sum of squares to a scalar
        // Assume existence of vector reduce API; reduce into first element of sqLocal
        AscendC::ReduceSum(sqLocal, sqLocal, this->groupSize);

        // read the first lane as scalar via lane broadcast
        AscendC::LocalTensor<DTYPE_X> denomVec = tmpBuf.Alloc<DTYPE_X>(this->groupSize);
        // broadcast mean = (sqLocal[0] / groupSize) + epsilon
        AscendC::Muls(denomVec, sqLocal, static_cast<DTYPE_X>(1.0f / static_cast<float>(this->groupSize)), 1);
        AscendC::Adds(denomVec, denomVec, this->epsilon, 1);
        // broadcast the scalar across vector
        AscendC::Broadcast(denomVec, denomVec, this->groupSize);
        // denomVec = sqrt(denomVec)
        AscendC::Sqrt(denomVec, denomVec, this->groupSize);

        // yLocal = xLocal / denom * gamma
        AscendC::Div(yLocal, xLocal, denomVec, this->groupSize);
        AscendC::Mul(yLocal, yLocal, gammaLocalUse, this->groupSize);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        // recycle
        inQueueX.FreeTensor(xLocal);
        inQueueGamma.EnQue(gammaLocalUse); // put back for reuse
        tmpBuf.Free(denomVec);
        tmpBuf.Free(sqLocal);
    }

    __aicore__ inline void CopyOut(uint32_t groupIdx)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[groupIdx * this->groupSize], yLocal, this->groupSize);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::VectorTmpBuff tmpBuf;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueGamma;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> gammaGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    AscendC::LocalTensor<DTYPE_Y> gammaLocal;

    uint32_t groupSize;
    uint32_t groupCountTotal;
    uint32_t groupStart;
    uint32_t groupCountThisBlock;
    uint32_t tileNum;
    DTYPE_X epsilon;
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNorm op;
    op.Init(x, gamma, y,
            tiling_data.totalLength, tiling_data.groupSize,
            tiling_data.groupCount, tiling_data.tileNum, tiling_data.epsilon);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized RMSNorm using custom AscendC kernel. It normalizes along the feature dimension.
    To match the kernel (which normalizes along the last dimension), we permute the tensor to NHWC.
    """
    def __init__(self, num_features: int, eps: float = 1e-5):
        super().__init__()
        self.num_features = num_features
        self.eps = eps
        # fixed gamma as ones to match x / rms (no learnable scale)
        self.register_buffer("gamma", torch.ones(num_features))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Move feature/channel dimension to the last to match kernel's expectation
        # Input is (N, C, H, W, ...). For 4D, we convert NCHW -> NHWC.
        # For generic ND: bring dim=1 to the last.
        perm = list(range(x.dim()))
        cdim = 1
        perm_no_c = [i for i in perm if i != cdim]
        permuted = x.permute(perm_no_c + [cdim]).contiguous()
        # Call custom kernel: normalize along last dimension with gamma=ones
        y_perm = custom_ops_lib.rms_norm_custom(permuted, self.gamma.to(permuted.dtype).to(permuted.device), self.eps)
        # Permute back to original layout
        inv_perm = list(range(x.dim()))
        for i, p in enumerate(perm_no_c + [cdim]):
            inv_perm[p] = i
        return y_perm.permute(inv_perm).contiguous()
'''
