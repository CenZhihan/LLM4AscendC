host_operator_src="""
#include "average_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline int64_t calc_out(int64_t in, int64_t kernel, int64_t stride, int64_t pad) {
    return (in + 2 * pad - kernel) / stride + 1;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    AveragePooling3dCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    auto origin = x_shape->GetOriginShape();
    if (origin.GetDimNum() != 5) {
        return ge::GRAPH_FAILED;
    }

    int64_t N = origin.GetDim(0);
    int64_t C = origin.GetDim(1);
    int64_t D = origin.GetDim(2);
    int64_t H = origin.GetDim(3);
    int64_t W = origin.GetDim(4);

    // get attributes
    int64_t k = 1;
    int64_t s = k;
    int64_t p = 0;

    auto k_attr = context->GetAttr("kernel_size");
    if (k_attr != nullptr) {
        k = k_attr->GetInt();
    }
    auto s_attr = context->GetAttr("stride");
    if (s_attr != nullptr) {
        s = s_attr->GetInt();
    }
    auto p_attr = context->GetAttr("padding");
    if (p_attr != nullptr) {
        p = p_attr->GetInt();
    }
    if (s <= 0) s = k;

    int64_t Do = calc_out(D, k, s, p);
    int64_t Ho = calc_out(H, k, s, p);
    int64_t Wo = calc_out(W, k, s, p);

    // Set output shape
    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, Do);
    y_shape->SetDim(3, Ho);
    y_shape->SetDim(4, Wo);

    // BlockDim and tiling pack
    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_c(static_cast<uint32_t>(C));
    tiling.set_d(static_cast<uint32_t>(D));
    tiling.set_h(static_cast<uint32_t>(H));
    tiling.set_w(static_cast<uint32_t>(W));

    tiling.set_do_(static_cast<uint32_t>(Do));
    tiling.set_ho(static_cast<uint32_t>(Ho));
    tiling.set_wo(static_cast<uint32_t>(Wo));

    tiling.set_k(static_cast<int32_t>(k));
    tiling.set_s(static_cast<int32_t>(s));
    tiling.set_p(static_cast<int32_t>(p));

    uint64_t totalOut = static_cast<uint64_t>(N) * static_cast<uint64_t>(C) *
                        static_cast<uint64_t>(Do) * static_cast<uint64_t>(Ho) * static_cast<uint64_t>(Wo);
    tiling.set_totalOut(totalOut);

    // save tiling buffer
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // no workspace
    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

    if (in.GetDimNum() != 5) {
        return GRAPH_FAILED;
    }

    int64_t N = in.GetDim(0);
    int64_t C = in.GetDim(1);
    int64_t D = in.GetDim(2);
    int64_t H = in.GetDim(3);
    int64_t W = in.GetDim(4);

    int64_t k = 1;

    int64_t Ho = calc_out(H, k, s, p);

kernel_src="""
#include "kernel_operator.h"

static inline __aicore__ uint64_t min_u64(uint64_t a, uint64_t b) { return a < b ? a : b; }

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    // Input/Output pointers
    __gm__ DTYPE_X* xPtr = (__gm__ DTYPE_X*)x;
    __gm__ DTYPE_Y* yPtr = (__gm__ DTYPE_Y*)y;

    // Shapes
    const uint32_t N  = tiling_data.n();
    const uint32_t C  = tiling_data.c();
    const uint32_t D  = tiling_data.d();
    const uint32_t H  = tiling_data.h();
    const uint32_t W  = tiling_data.w();

    const uint32_t Do = tiling_data.do_();
    const uint32_t Ho = tiling_data.ho();
    const uint32_t Wo = tiling_data.wo();

    const int32_t K = tiling_data.k();
    const int32_t S = tiling_data.s();
    const int32_t P = tiling_data.p();

    const uint64_t totalOut = tiling_data.totalOut();
    const uint32_t blocks = AscendC::GetBlockNum();
    const uint32_t blockId = AscendC::GetBlockIdx();
    const uint64_t perBlock = (totalOut + blocks - 1) / blocks;
    const uint64_t start = (uint64_t)blockId * perBlock;
    const uint64_t end = min_u64(totalOut, start + perBlock);

    // Helper lambdas for linear index mapping
    auto in_index = [=](__aicore__ int n, int c, int id, int ih, int iw) -> uint64_t {
        return (((uint64_t)n * C + (uint64_t)c) * D + (uint64_t)id) * H * W + (uint64_t)ih * W + (uint64_t)iw;
    };
    auto out_unpack = [=](__aicore__ uint64_t p, int &n, int &c, int &od, int &oh, int &ow) {
        ow = (int)(p % Wo); p /= Wo;
        oh = (int)(p % Ho); p /= Ho;
        od = (int)(p % Do); p /= Do;
        c  = (int)(p % C);  p /= C;
        n  = (int)p;
    };
    auto out_index = [=](__aicore__ int n, int c, int od, int oh, int ow) -> uint64_t {
        return (((uint64_t)n * C + (uint64_t)c) * Do + (uint64_t)od) * Ho * Wo + (uint64_t)oh * Wo + (uint64_t)ow;
    };

    for (uint64_t pos = start; pos < end; ++pos) {
        int n, c, od, oh, ow;
        out_unpack(pos, n, c, od, oh, ow);

        const int id0 = od * S - P;
        const int ih0 = oh * S - P;
        const int iw0 = ow * S - P;

        float sum = 0.0f;
        int count = 0;

        for (int kd = 0; kd < K; ++kd) {
            int id = id0 + kd;
            if (id < 0 || id >= (int)D) continue;
            for (int kh = 0; kh < K; ++kh) {
                int ih = ih0 + kh;
                if (ih < 0 || ih >= (int)H) continue;
                for (int kw = 0; kw < K; ++kw) {
                    int iw = iw0 + kw;
                    if (iw < 0 || iw >= (int)W) continue;
                    uint64_t idx_in = in_index(n, c, id, ih, iw);
                    // Direct GM read
                    sum += (float)xPtr[idx_in];
                    ++count;
                }
            }
        }
        float avg = (count > 0) ? (sum / (float)count) : 0.0f; // exclude pad region from count
        uint64_t idx_out = out_index(n, c, od, oh, ow);
        yPtr[idx_out] = (DTYPE_Y)avg;
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline std::vector<int64_t> calc_output_3d(const at::Tensor& x, int64_t k, int64_t s, int64_t p) {
    auto sizes = x.sizes();
    int64_t N = sizes[0], C = sizes[1], D = sizes[2], H = sizes[3], W = sizes[4];
    int64_t Do = (D + 2 * p - k) / s + 1;
    int64_t Ho = (H + 2 * p - k) / s + 1;
    int64_t Wo = (W + 2 * p - k) / s + 1;
    return {N, C, Do, Ho, Wo};
}

at::Tensor average_pooling_3d_custom_impl_npu(const at::Tensor& self, int64_t kernel_size, c10::optional<int64_t> stride_opt, int64_t padding) {
    TORCH_CHECK(self.dim() == 5, "Input must be 5D (N, C, D, H, W)");
    int64_t stride = stride_opt.has_value() ? stride_opt.value() : kernel_size;
    TORCH_CHECK(kernel_size > 0, "kernel_size must be > 0");
    TORCH_CHECK(stride > 0, "stride must be > 0");
    TORCH_CHECK(padding >= 0, "padding must be >= 0");

    auto out_sizes = calc_output_3d(self, kernel_size, stride, padding);
    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnAveragePooling3dCustom, self, kernel_size, stride, padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("average_pooling_3d_custom", &average_pooling_3d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("average_pooling_3d_custom", &average_pooling_3d_custom_impl_npu, "AvgPool3d custom (NCDHW)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Simple model that performs 3D Average Pooling using a custom AscendC kernel.
    """
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0):
        super().__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Use kernel_size if stride is None to match PyTorch AvgPool3d semantics
        stride = self.stride if self.stride is not None else self.kernel_size
        return custom_ops_lib.average_pooling_3d_custom(x, self.kernel_size, stride, self.padding)
