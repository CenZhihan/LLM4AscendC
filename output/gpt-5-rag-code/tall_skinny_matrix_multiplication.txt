#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t get_dim_safe(const gert::Shape* s, size_t i) {
    if (s->GetOriginShape().GetDimNum() > i) {
        return static_cast<uint32_t>(s->GetOriginShape().GetDim(i));
    }
    return 1;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
// Conservative tiles to fit UB for tall-skinny cases
const uint32_t DEFAULT_TILE_M = 512;
const uint32_t DEFAULT_TILE_K = 32;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    // Assume A: [M, K], B: [K, N]
    uint32_t M = get_dim_safe(a_shape, 0);
    uint32_t K = get_dim_safe(a_shape, 1);
    uint32_t Kb = get_dim_safe(b_shape, 0);
    uint32_t N = get_dim_safe(b_shape, 1);
    // Fallback if user passes transposed shapes: if a is [K, M] and b is [N, K], try to infer
    if (K != Kb && get_dim_safe(a_shape, 0) == Kb && get_dim_safe(b_shape, 1) == get_dim_safe(a_shape, 0)) {
        // A: [K, M], B: [N, K] -> treat as A^T [M,K] * B^T [K,N]
        M = get_dim_safe(a_shape, 1);
        K = get_dim_safe(a_shape, 0);
        Kb = get_dim_safe(b_shape, 1);
        N = get_dim_safe(b_shape, 0);
    }

    context->SetBlockDim(BLOCK_DIM);

    // Set tiling
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(DEFAULT_TILE_M);
    tiling.set_tileK(DEFAULT_TILE_K);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_PARAM_INVALID;
    }
    int64_t M = a_shape->GetOriginShape().GetDim(0);
    int64_t K = a_shape->GetOriginShape().GetDim(1);
    int64_t Kb = b_shape->GetOriginShape().GetDim(0);
    int64_t N = b_shape->GetOriginShape().GetDim(1);

    // try alternate orientation if not directly compatible
    if (K != Kb && a_shape->GetOriginShape().GetDim(0) == Kb) {
        // A: [K, M], B: [N, K] treat as A^T @ B^T
        M = a_shape->GetOriginShape().GetDim(1);
        K = a_shape->GetOriginShape().GetDim(0);
        Kb = b_shape->GetOriginShape().GetDim(1);
        N = b_shape->GetOriginShape().GetDim(0);
    }

    if (K != Kb) {
        return ge::GRAPH_PARAM_INVALID;
    }

    gert::Shape* out_shape = context->GetOutputShape(0);
    out_shape->SetDimNum(2);
    out_shape->SetDim(0, M);
    out_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t QUEUE_DEPTH = 1;

class KernelTallSkinnyMM {
public:
    __aicore__ inline KernelTallSkinnyMM() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileK)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileM = tileM;
        this->tileK = tileK;

        aGm.SetGlobalBuffer((__gm__ float*)a, M * K);
        bGm.SetGlobalBuffer((__gm__ float*)b, K * N);
        cGm.SetGlobalBuffer((__gm__ float*)c, M * N);

        // UB buffers
        pipe.InitBuffer(bQueue, QUEUE_DEPTH, tileK * N * sizeof(float)); // B_sub [tileK x N]
        pipe.InitBuffer(aRowQueue, QUEUE_DEPTH, tileK * sizeof(float));  // A row segment [tileK]
        pipe.InitBuffer(vecQueueP, QUEUE_DEPTH, N * sizeof(float));      // partial vector p [N]
        pipe.InitBuffer(vecQueueTmp, QUEUE_DEPTH, N * sizeof(float));    // temp vector [N]
        pipe.InitBuffer(vecQueueCold, QUEUE_DEPTH, N * sizeof(float));   // c_old [N]
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockId = AscendC::GetBlockIdx();

        uint32_t rowsPerBlock = (M + blockNum - 1) / blockNum;
        uint32_t rowStart = blockId * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > M) rowEnd = M;

        if (rowStart >= rowEnd) {
            return;
        }

        for (uint32_t k0 = 0; k0 < K; k0 += tileK) {
            uint32_t kThis = (k0 + tileK <= K) ? tileK : (K - k0);

            // Load B_sub [kThis x N] once per k0
            AscendC::LocalTensor<float> bLocal = bQueue.AllocTensor<float>();
            AscendC::DataCopy(bLocal, bGm[k0 * N], kThis * N);
            bQueue.EnQue(bLocal);
            bLocal = bQueue.DeQue<float>();

            for (uint32_t m0 = rowStart; m0 < rowEnd; m0 += tileM) {
                uint32_t rowsThisTile = ((m0 + tileM) <= rowEnd) ? tileM : (rowEnd - m0);

                for (uint32_t r = 0; r < rowsThisTile; ++r) {
                    uint32_t globalRow = m0 + r;

                    // Load A_row segment [kThis]
                    AscendC::LocalTensor<float> aRowLocal = aRowQueue.AllocTensor<float>();
                    AscendC::DataCopy(aRowLocal, aGm[globalRow * K + k0], kThis);
                    aRowQueue.EnQue(aRowLocal);
                    aRowLocal = aRowQueue.DeQue<float>();

                    // Prepare partial vector p = sum_{k in tile} a * b_rowk
                    AscendC::LocalTensor<float> pLocal = vecQueueP.AllocTensor<float>();
                    AscendC::LocalTensor<float> tmpLocal = vecQueueTmp.AllocTensor<float>();

                    // Compute pLocal without zero fill: first k initializes pLocal, subsequent ks accumulate
                    if (kThis > 0) {
                        // Copy first b_row (length N) from bLocal to tmpLocal and scale
                        AscendC::DataCopy(tmpLocal, bLocal[0], N);
                        float a0 = aRowLocal.GetValue(0);
                        AscendC::Muls(pLocal, tmpLocal, a0, N);

                        for (uint32_t kk = 1; kk < kThis; ++kk) {
                            AscendC::DataCopy(tmpLocal, bLocal[kk * N], N);
                            float aval = aRowLocal.GetValue(kk);
                            AscendC::Muls(tmpLocal, tmpLocal, aval, N);
                            AscendC::Add(pLocal, pLocal, tmpLocal, N);
                        }
                    }

                    // Accumulate into global C row
                    if (k0 == 0) {
                        AscendC::DataCopy(cGm[globalRow * N], pLocal, N);
                    } else {
                        AscendC::LocalTensor<float> cOldLocal = vecQueueCold.AllocTensor<float>();
                        AscendC::DataCopy(cOldLocal, cGm[globalRow * N], N);
                        AscendC::Add(pLocal, pLocal, cOldLocal, N);
                        AscendC::DataCopy(cGm[globalRow * N], pLocal, N);
                        vecQueueCold.FreeTensor(cOldLocal);
                    }

                    vecQueueP.FreeTensor(pLocal);
                    vecQueueTmp.FreeTensor(tmpLocal);
                    aRowQueue.FreeTensor(aRowLocal);
                }
            }
            bQueue.FreeTensor(bLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, QUEUE_DEPTH> bQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, QUEUE_DEPTH> aRowQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, QUEUE_DEPTH> vecQueueP;
    AscendC::TQue<AscendC::TPosition::VECOUT, QUEUE_DEPTH> vecQueueTmp;
    AscendC::TQue<AscendC::TPosition::VECOUT, QUEUE_DEPTH> vecQueueCold;

    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> cGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileM;
    uint32_t tileK;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm) {
    GET_TILING_DATA(tiling_data, tilingGm);
    KernelTallSkinnyMM op;
    op.Init(a, b, c,
            tiling_data.M,
            tiling_data.N,
            tiling_data.K,
            tiling_data.tileM,
            tiling_data.tileK);
    op.Process();
}
