host_operator_src="""
#include "cumsum_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <cstdint>
#include <vector>

namespace {
inline uint64_t Product(const std::vector<int64_t>& v, size_t begin, size_t end) {
    uint64_t p = 1;
    for (size_t i = begin; i < end; ++i) {
        p *= static_cast<uint64_t>(v[i]);
    }
    return p;
}
}

namespace optiling {
const uint32_t DEFAULT_BLOCK_DIM = 8;
const uint32_t DEFAULT_TILE_LEN = 2048;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CumsumCustomTilingData tiling;

    // Get input original shape
    const auto* in_shape = context->GetInputShape(0);
    const auto& origin = in_shape->GetOriginShape();
    uint64_t totalLength = origin.GetShapeSize();

    // Fetch dims
    std::vector<int64_t> dims;
    dims.reserve(origin.GetDimNum());
    for (uint32_t i = 0; i < origin.GetDimNum(); ++i) {
        dims.push_back(origin.GetDim(i));
    }

    // Read attribute dim
    int64_t dim = 0;
    if (context->GetAttrs() != nullptr) {
        (void)context->GetAttrs()->GetInt("dim", dim);
    }
    if (dim < 0) {
        dim += static_cast<int64_t>(dims.size());
    }
    // Axis length and rows (flatten to 2D: rows x axisLen when dim is the last dimension)
    uint32_t axisLen = static_cast<uint32_t>(dims[dim]);
    uint64_t inner = Product(dims, static_cast<size_t>(dim + 1), dims.size());
    uint64_t outer = Product(dims, 0, static_cast<size_t>(dim));
    // We only support contiguous axis (inner == 1) efficiently
    uint64_t rows = outer * inner; // when inner==1, rows == outer; otherwise it flattens lanes

    // Decide block dim and tile length
    uint32_t blockDim = DEFAULT_BLOCK_DIM;
    if (rows < blockDim) {
        blockDim = static_cast<uint32_t>(rows == 0 ? 1 : rows);
    }
    uint32_t tileLen = std::min<uint32_t>(DEFAULT_TILE_LEN, axisLen);

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_rows(static_cast<uint32_t>(rows));
    tiling.set_axisLen(axisLen);
    tiling.set_tileLen(tileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);
    *out_shape = *in_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // local tensor buffers per queue

class KernelCumsum {
public:
    __aicore__ inline KernelCumsum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rows, uint32_t axisLen, uint32_t tileLen)
    {
        this->totalLength = totalLength;
        this->rows = rows;
        this->axisLen = axisLen;
        this->tileLen = tileLen;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, totalLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, totalLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (rows + blockNum - 1) / blockNum;

        uint64_t rowStart = static_cast<uint64_t>(rowsPerBlock) * blockIdx;
        uint64_t rowEnd = AscendC::Min(static_cast<uint64_t>(rows),
                                       static_cast<uint64_t>(rowsPerBlock) * (blockIdx + 1));

        for (uint64_t r = rowStart; r < rowEnd; ++r) {
            uint64_t base = r * static_cast<uint64_t>(axisLen);
            uint32_t remaining = axisLen;
            uint64_t offset = base;
            DTYPE_X carry = static_cast<DTYPE_X>(0);

            while (remaining > 0) {
                uint32_t curLen = AscendC::Min(remaining, tileLen);

                // CopyIn
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[offset], curLen);
                inQueueX.EnQue(xLocal);

                // Compute
                AscendC::LocalTensor<DTYPE_X> xTile = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

                DTYPE_X acc = carry;
                for (uint32_t i = 0; i < curLen; ++i) {
                    acc = acc + static_cast<DTYPE_X>(xTile(i));
                    yLocal(i) = static_cast<DTYPE_Y>(acc);
                }
                carry = acc;

                outQueueY.EnQue(yLocal);
                inQueueX.FreeTensor(xTile);

                // CopyOut
                AscendC::LocalTensor<DTYPE_Y> yTile = outQueueY.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[offset], yTile, curLen);
                outQueueY.FreeTensor(yTile);

                offset += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalLength;
    uint32_t rows;
    uint32_t axisLen;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void cumsum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsum op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.rows,
            tiling_data.axisLen,
            tiling_data.tileLen);
    op.Process();
}
