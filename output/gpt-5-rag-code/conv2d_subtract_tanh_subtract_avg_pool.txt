kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConv2dSubtractTanhSubtractAvgPool {
public:
    __aicore__ inline KernelConv2dSubtractTanhSubtractAvgPool() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR s1, GM_ADDR s2, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = (totalLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t remain = totalLength - this->blockLength * AscendC::GetBlockNum();
        if (AscendC::GetBlockIdx() < remain) {
            this->startOffset = AscendC::GetBlockIdx() * (this->blockLength + 1);
            this->blockLength = this->blockLength + 1;
        } else {
            this->startOffset = remain * (this->blockLength + 1) + (AscendC::GetBlockIdx() - remain) * this->blockLength;
        }

        this->tileNum = tileNum;
        this->tileLength = (this->blockLength / (tileNum * BUFFER_NUM)) > 0 ? (this->blockLength / (tileNum * BUFFER_NUM)) : this->blockLength;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->startOffset, this->blockLength);
        s1Gm.SetGlobalBuffer((__gm__ DTYPE_Y *)s1 + this->startOffset, this->blockLength);
        s2Gm.SetGlobalBuffer((__gm__ DTYPE_Z *)s2 + this->startOffset, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->startOffset, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueS1, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueS2, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loops = (this->blockLength + this->tileLength - 1) / this->tileLength;
        for (int32_t i = 0; i < loops; ++i) {
            uint32_t curLen = (i != loops - 1) ? this->tileLength : (this->blockLength - i * this->tileLength);
            CopyIn(i, curLen);
            Compute(i, curLen);
            CopyOut(i, curLen);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> s1Local = inQueueS1.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> s2Local = inQueueS2.AllocTensor<DTYPE_Z>();

        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], curLen);
        AscendC::DataCopy(s1Local, s1Gm[progress * this->tileLength], curLen);
        AscendC::DataCopy(s2Local, s2Gm[progress * this->tileLength], curLen);

        inQueueX.EnQue(xLocal);
        inQueueS1.EnQue(s1Local);
        inQueueS2.EnQue(s2Local);
    }

    __aicore__ inline void Compute(int32_t progress, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> s1Local = inQueueS1.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> s2Local = inQueueS2.DeQue<DTYPE_Z>();

        AscendC::LocalTensor<DTYPE_Z> tmpLocal = tmpQueue.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // tmp = x - s1
        AscendC::Sub(tmpLocal, xLocal, s1Local, curLen);
        // tmp = tanh(tmp)
        AscendC::Tanh(tmpLocal, tmpLocal, curLen);
        // y = tmp - s2
        AscendC::Sub(yLocal, tmpLocal, s2Local, curLen);

        outQueueY.EnQue<DTYPE_Z>(yLocal);

        tmpQueue.FreeTensor(tmpLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueS1.FreeTensor(s1Local);
        inQueueS2.FreeTensor(s2Local);
    }

    __aicore__ inline void CopyOut(int32_t progress, uint32_t curLen)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, curLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueS1, inQueueS2;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> s1Gm;
    AscendC::GlobalTensor<DTYPE_Z> s2Gm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t blockLength;
    uint32_t startOffset;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(GM_ADDR x, GM_ADDR s1, GM_ADDR s2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractTanhSubtractAvgPool op;
    op.Init(x, s1, s2, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using custom AscendC kernel to fuse (subtract -> tanh -> subtract).
    AvgPool remains as a standard module for now.
    """
    def __init__(self, in_channels, out_channels, kernel_size, subtract1_value, subtract2_value, kernel_size_pool):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.subtract1_value = subtract1_value
        self.subtract2_value = subtract2_value
        self.avgpool = torch.nn.AvgPool2d(kernel_size_pool)

    def forward(self, x):
        x = self.conv(x)
        s1 = torch.full_like(x, self.subtract1_value)
        s2 = torch.full_like(x, self.subtract2_value)
        x = custom_ops_lib.conv2d_subtract_tanh_subtract_avg_pool_custom(x, s1, s2)
        x = self.avgpool(x)
        return x
