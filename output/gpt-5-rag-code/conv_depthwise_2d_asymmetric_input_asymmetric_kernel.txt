kernel_src="""
#include "kernel_operator.h"

class KernelDWConv {
public:
    __aicore__ inline KernelDWConv() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t Kh, uint32_t Kw,
                                uint32_t Sh, uint32_t Sw,
                                uint32_t Ph, uint32_t Pw,
                                uint32_t Dh, uint32_t Dw,
                                uint32_t OH, uint32_t OW)
    {
        this->N = N; this->C = C; this->H = H; this->W = W;
        this->Kh = Kh; this->Kw = Kw;
        this->Sh = Sh; this->Sw = Sw;
        this->Ph = Ph; this->Pw = Pw;
        this->Dh = Dh; this->Dw = Dw;
        this->OH = OH; this->OW = OW;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N*C*H*W);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)C*Kh*Kw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, (uint64_t)N*C*OH*OW);

        // Buffers for Kh rows of input, one output row, and one kernel
        uint32_t xRowsSize = Kh * W;
        uint32_t wSize = Kh * Kw;
        uint32_t yRowSize = OW;

        pipe.InitBuffer(inQueueXRows, 1, xRowsSize * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, 1, wSize * sizeof(DTYPE_W));
        pipe.InitBuffer(outQueueYRow, 1, yRowSize * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t chPerBlock = (C + blocks - 1) / blocks;
        uint32_t cStart = blockIdx * chPerBlock;
        uint32_t cEnd = (cStart + chPerBlock) > C ? C : (cStart + chPerBlock);

        for (uint32_t c = cStart; c < cEnd; ++c) {
            // Load weight kernel for this channel
            AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.AllocTensor<DTYPE_W>();
            uint64_t wBase = (uint64_t)c * (uint64_t)Kh * (uint64_t)Kw;
            AscendC::DataCopy(wLocal, wGm[wBase], (uint32_t)(Kh * Kw));

            for (uint32_t n = 0; n < N; ++n) {
                uint64_t inPlaneBase = ((uint64_t)n * C + (uint64_t)c) * (uint64_t)H * (uint64_t)W;
                uint64_t outPlaneBase = ((uint64_t)n * C + (uint64_t)c) * (uint64_t)OH * (uint64_t)OW;

                for (uint32_t oh = 0; oh < OH; ++oh) {
                    // Copy Kh consecutive input rows starting at ih = oh (since Sh=1, Ph=0, Dh=1)
                    uint32_t ih = oh * Sh;
                    AscendC::LocalTensor<DTYPE_X> xRowsLocal = inQueueXRows.AllocTensor<DTYPE_X>();
                    uint64_t xCopyBase = inPlaneBase + (uint64_t)ih * (uint64_t)W;
                    AscendC::DataCopy(xRowsLocal, xGm[xCopyBase], (uint32_t)(Kh * W));

                    AscendC::LocalTensor<DTYPE_Y> yRowLocal = outQueueYRow.AllocTensor<DTYPE_Y>();

                    // Compute one output row
                    for (uint32_t ow = 0; ow < OW; ++ow) {
                        uint32_t iw = ow * Sw;
                        DTYPE_Y acc = (DTYPE_Y)0;
                        for (uint32_t kh = 0; kh < Kh; ++kh) {
                            uint32_t rowOff = kh * W + iw;
                            uint32_t wOff = kh * Kw;
                            for (uint32_t kw = 0; kw < Kw; ++kw) {
                                acc += (DTYPE_Y)xRowsLocal[rowOff + kw] * (DTYPE_Y)wLocal[wOff + kw];
                            }
                        }
                        yRowLocal[ow] = acc;
                    }

                    // Write out this output row
                    uint64_t yStoreBase = outPlaneBase + (uint64_t)oh * (uint64_t)OW;
                    AscendC::DataCopy(yGm[yStoreBase], yRowLocal, OW);

                    outQueueYRow.FreeTensor(yRowLocal);
                    inQueueXRows.FreeTensor(xRowsLocal);
                }
            }
            inQueueW.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueXRows;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueYRow;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, C, H, W;
    uint32_t Kh, Kw;
    uint32_t Sh, Sw;
    uint32_t Ph, Pw;
    uint32_t Dh, Dw;
    uint32_t OH, OW;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelDWConv op;
    op.Init(x, w, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.Kh, tiling_data.Kw,
            tiling_data.Sh, tiling_data.Sw,
            tiling_data.Ph, tiling_data.Pw,
            tiling_data.Dh, tiling_data.Dw,
            tiling_data.OH, tiling_data.OW);
    op.Process();
}

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size_h: int, kernel_size_w: int, stride_h: int = 1, stride_w: int = 1, padding_h: int = 0, padding_w: int = 0, dilation_h: int = 1, dilation_w: int = 1, groups: int = 1, bias: bool = False):

        # Keep a standard Conv2d to store weights; we will invoke a custom op in forward
        assert stride_h == 1 and stride_w == 1, "custom kernel currently supports stride=1"
        assert padding_h == 0 and padding_w == 0, "custom kernel currently supports no padding"
        assert dilation_h == 1 and dilation_w == 1, "custom kernel currently supports dilation=1"
        assert groups == in_channels and out_channels == in_channels, "depthwise conv: groups==in_channels and out_channels==in_channels"
        self.conv2d = torch.nn.Conv2d(in_channels, in_channels, (kernel_size_h, kernel_size_w),

                                      padding=(padding_h, padding_w),
                                      dilation=(dilation_h, dilation_w),
                                      groups=in_channels,
                                      bias=False)  # bias unsupported in custom kernel

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # x: [N, C, H, W]
        # weight: [C, 1, Kh, Kw]
