namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);

    auto origin = in_shape->GetOriginShape();
    uint32_t dim_num = origin.GetDimNum();
    std::vector<int64_t> out_dims;

            out_dims.push_back(1);
        } else {
            out_dims.push_back(origin.GetDim(i));
        }

        out_dims = {1};
    }
    out_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        out_shape->SetDim(i, out_dims[i]);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSumReduceAxis1 {
public:
    __aicore__ inline KernelSumReduceAxis1() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t N, uint32_t D1, uint32_t D2, uint32_t tileK)
    {
        this->N = N;
        this->D1 = D1;
        this->D2 = D2;
        this->tileK = tileK;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->nPerBlock = (N + blockNum - 1) / blockNum;
        this->nStart = blockIdx * this->nPerBlock;
        if (this->nStart > N) this->nStart = N;
        uint32_t nEnd = this->nStart + this->nPerBlock;
        if (nEnd > N) nEnd = N;
        this->nCount = (nEnd > this->nStart) ? (nEnd - this->nStart) : 0;

        (__gm__ DTYPE_X*)xBase = (__gm__ DTYPE_X*)x + (uint64_t)this->nStart * (uint64_t)this->D1 * (uint64_t)this->D2;
        (__gm__ DTYPE_Z*)yBase = (__gm__ DTYPE_Z*)y + (uint64_t)this->nStart * (uint64_t)this->D2;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)xBase, (uint32_t)((uint64_t)this->nCount * (uint64_t)this->D1 * (uint64_t)this->D2));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)yBase, (uint32_t)((uint64_t)this->nCount * (uint64_t)this->D2));

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileK * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileK * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t nLocal = 0; nLocal < this->nCount; ++nLocal) {
            uint32_t k = 0;
            while (k < this->D2) {
                uint32_t currTile = this->tileK;
                if (k + currTile > this->D2) {
                    currTile = this->D2 - k;
                }

                AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
                bool first = true;

                for (uint32_t j = 0; j < this->D1; ++j) {
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();

                    uint64_t xOff = ((uint64_t)nLocal * (uint64_t)this->D1 + (uint64_t)j) * (uint64_t)this->D2 + (uint64_t)k;
                    AscendC::DataCopy(xLocal, xGm[xOff], currTile);

                    if (first) {
                        AscendC::DataCopy(yLocal, xLocal, currTile);
                        first = false;
                    } else {
                        AscendC::Add(yLocal, yLocal, xLocal, currTile);
                    }
                    inQueueX.FreeTensor(xLocal);
                }

                outQueueY.EnQue<DTYPE_Z>(yLocal);

                AscendC::LocalTensor<DTYPE_Z> yLocalOut = outQueueY.DeQue<DTYPE_Z>();
                uint64_t yOff = (uint64_t)nLocal * (uint64_t)this->D2 + (uint64_t)k;
                AscendC::DataCopy(yGm[yOff], yLocalOut, currTile);
                outQueueY.FreeTensor(yLocalOut);

                k += currTile;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N;
    uint32_t D1;
    uint32_t D2;
    uint32_t tileK;

    uint32_t nPerBlock;
    uint32_t nStart;
    uint32_t nCount;

    __gm__ DTYPE_X* xBase;
    __gm__ DTYPE_Z* yBase;
};

extern "C" __global__ __aicore__ void sum_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSumReduceAxis1 op;
    op.Init(x, y, tiling_data.N, tiling_data.D1, tiling_data.D2, tiling_data.tileK);
    op.Process();
}
