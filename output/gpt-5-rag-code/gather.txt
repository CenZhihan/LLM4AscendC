#include "gather_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GatherCustomTilingData tiling;

    const auto* x_shape = context->GetInputShape(0);
    const auto* idx_shape = context->GetInputShape(1);

    uint32_t rows = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t cols = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t outCols = static_cast<uint32_t>(idx_shape->GetDim(1));

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_outCols(outCols);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* idx_shape = context->GetInputShape(1);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *idx_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGather {
public:
    __aicore__ inline KernelGather() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t outCols, uint32_t tileNum)
    {
        this->rows = rows;
        this->cols = cols;
        this->outCols = outCols;
        this->tileNum = tileNum;

        // rows per block
        this->rowsPerBlock = (rows + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t startRow = this->rowsPerBlock * AscendC::GetBlockIdx();
        if (startRow + this->rowsPerBlock > rows) {
            this->rowsPerBlock = rows - startRow;
        }

        // segment length per tile for output columns
        uint32_t baseTile = outCols / (tileNum * BUFFER_NUM);
        this->tileLen = baseTile > 0 ? baseTile : 1;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + startRow * cols, this->rowsPerBlock * cols);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_INDEX *)idx + startRow * outCols, this->rowsPerBlock * outCols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + startRow * outCols, this->rowsPerBlock * outCols);

        pipe.InitBuffer(idxQueue, BUFFER_NUM, this->tileLen * sizeof(DTYPE_INDEX));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Y));
        pipe.InitBuffer(rowQueue, 1, this->cols * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            // Load one full row of x into local buffer
            AscendC::LocalTensor<DTYPE_X> rowLocal = rowQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(rowLocal, xGm[r * this->cols], this->cols);
            rowQueue.EnQue(rowLocal);
            AscendC::LocalTensor<DTYPE_X> xRow = rowQueue.DeQue<DTYPE_X>();

            uint32_t tiles = (this->outCols + this->tileLen - 1) / this->tileLen;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t start = t * this->tileLen;
                uint32_t validLen = this->outCols > start ? AscendC::Min(this->tileLen, this->outCols - start) : 0;
                if (validLen == 0) { break; }

                // Copy indices tile
                AscendC::LocalTensor<DTYPE_INDEX> idxLocal = idxQueue.AllocTensor<DTYPE_INDEX>();
                AscendC::DataCopy(idxLocal, idxGm[r * this->outCols + start], validLen);
                idxQueue.EnQue(idxLocal);

                // Compute gather on local row
                AscendC::LocalTensor<DTYPE_INDEX> idxVals = idxQueue.DeQue<DTYPE_INDEX>();
                AscendC::LocalTensor<DTYPE_Y> outLocal = outQueue.AllocTensor<DTYPE_Y>();

                for (uint32_t j = 0; j < validLen; ++j) {
                    int32_t colIdx = static_cast<int32_t>(idxVals[j]);
                    // Clamp to valid range for safety
                    if (colIdx < 0) colIdx = 0;
                    if (colIdx >= static_cast<int32_t>(this->cols)) colIdx = this->cols - 1;
                    outLocal[j] = static_cast<DTYPE_Y>(xRow[colIdx]);
                }

                outQueue.EnQue(outLocal);
                idxQueue.FreeTensor(idxVals);

                // Copy out gathered tile
                AscendC::LocalTensor<DTYPE_Y> outVals = outQueue.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[r * this->outCols + start], outVals, validLen);
                outQueue.FreeTensor(outVals);
            }

            rowQueue.FreeTensor(xRow);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> idxQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> rowQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_INDEX> idxGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t outCols;
    uint32_t tileNum;
    uint32_t rowsPerBlock;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void gather_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGather op;
    op.Init(x, idx, y, tiling_data.rows, tiling_data.cols, tiling_data.outCols, tiling_data.tileNum);
    op.Process();
}
"""
