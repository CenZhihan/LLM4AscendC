kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // number of tiles buffered

template<typename T>
__aicore__ inline void idx_to_coords(uint32_t idx, uint32_t Cout, uint32_t Hout, uint32_t Wout,
                                     uint32_t &n, uint32_t &oc, uint32_t &oh, uint32_t &ow) {
    uint32_t ow_ = idx % Wout;
    uint32_t tmp1 = idx / Wout;
    uint32_t oh_ = tmp1 % Hout;
    uint32_t tmp2 = tmp1 / Hout;
    uint32_t oc_ = tmp2 % Cout;
    uint32_t n_ = tmp2 / Cout;
    n = n_; oc = oc_; oh = oh_; ow = ow_;
}

class KernelConv2dActBN {
public:
    __aicore__ inline KernelConv2dActBN() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR gamma, GM_ADDR beta, GM_ADDR mean, GM_ADDR inv_std,
                                GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t K, uint32_t Hout, uint32_t Wout,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->N = N; this->Cin = Cin; this->H = H; this->W = W;
        this->Cout = Cout; this->K = K; this->Hout = Hout; this->Wout = Wout;
        this->totalLength = totalLength;
        this->blockLength = totalLength / AscendC::GetBlockNum() + ((AscendC::GetBlockIdx() < (totalLength % AscendC::GetBlockNum())) ? 1 : 0);
        // Use even split ignoring remainder complexity for demo
        this->blockStart = (totalLength / AscendC::GetBlockNum()) * AscendC::GetBlockIdx();
        this->tileNum = tileNum;
        this->tileLength = (this->blockLength == 0) ? 0 : (this->blockLength + (tileNum * BUFFER_NUM - 1)) / (tileNum * BUFFER_NUM);

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint32_t)(N * Cin * H * W));
        wGm.SetGlobalBuffer((__gm__ float*)w, (uint32_t)(Cout * Cin * K * K));
        bGm.SetGlobalBuffer((__gm__ float*)b, (uint32_t)(Cout));
        gammaGm.SetGlobalBuffer((__gm__ float*)gamma, (uint32_t)(Cout));
        betaGm.SetGlobalBuffer((__gm__ float*)beta, (uint32_t)(Cout));
        meanGm.SetGlobalBuffer((__gm__ float*)mean, (uint32_t)(Cout));
        invStdGm.SetGlobalBuffer((__gm__ float*)inv_std, (uint32_t)(Cout));
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint32_t)(totalLength));

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (this->blockLength == 0 || this->tileLength == 0) { return; }
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline uint32_t offset_x(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const {
        return (((n * Cin) + c) * H + h) * W + w;
    }
    __aicore__ inline uint32_t offset_w(uint32_t oc, uint32_t ic, uint32_t kh, uint32_t kw) const {
        return ((((oc * Cin) + ic) * K + kh) * K + kw);
    }
    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        uint32_t tileBase = progress * this->tileLength;
        for (uint32_t t = 0; t < this->tileLength; ++t) {
            uint32_t localIdx = tileBase + t;
            if (localIdx >= this->blockLength) {
                continue;
            }
            uint32_t globalIdx = this->blockStart + localIdx;
            uint32_t n, oc, oh, ow;
            idx_to_coords<float>(globalIdx, Cout, Hout, Wout, n, oc, oh, ow);

            float acc = 0.0f;
            for (uint32_t ic = 0; ic < Cin; ++ic) {
                for (uint32_t kh = 0; kh < K; ++kh) {
                    uint32_t ih = oh + kh;
                    for (uint32_t kw = 0; kw < K; ++kw) {
                        uint32_t iw = ow + kw;
                        float xv = xGm.GetValue(offset_x(n, ic, ih, iw));
                        float wv = wGm.GetValue(offset_w(oc, ic, kh, kw));
                        acc += xv * wv;
                    }
                }
            }
            acc += bGm.GetValue(oc);

            // activation: x * tanh(softplus(x)), with softplus(x) = log(1 + exp(x))
            float sp = logf(1.0f + expf(acc));
            float act = tanhf(sp) * acc;

            // batch norm (inference): y = gamma * (act - mean) * inv_std + beta
            float g = gammaGm.GetValue(oc);
            float be = betaGm.GetValue(oc);
            float m = meanGm.GetValue(oc);
            float istd = invStdGm.GetValue(oc);
            float outv = g * ((act - m) * istd) + be;

            yLocal.SetValue(t, outv);
        }
        outQueueY.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        uint32_t tileBase = progress * this->tileLength;
        // Write back element by element to handle tail
        for (uint32_t t = 0; t < this->tileLength; ++t) {
            uint32_t localIdx = tileBase + t;
            if (localIdx >= this->blockLength) { break; }
            yGm.SetValue(this->blockStart + localIdx, yLocal.GetValue(t));
        }
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> gammaGm;
    AscendC::GlobalTensor<float> betaGm;
    AscendC::GlobalTensor<float> meanGm;
    AscendC::GlobalTensor<float> invStdGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t N, Cin, H, W, Cout, K, Hout, Wout;
    uint32_t totalLength;
    uint32_t blockLength;
    uint32_t blockStart;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_activation_batch_norm_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                           GM_ADDR gamma, GM_ADDR beta, GM_ADDR mean, GM_ADDR inv_std,
                                                                           GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dActBN op;
    op.Init(x, weight, bias, gamma, beta, mean, inv_std, y,
            tiling_data.N, tiling_data.Cin, tiling_data.H, tiling_data.W,
            tiling_data.Cout, tiling_data.K, tiling_data.Hout, tiling_data.Wout,
            tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, eps=1e-5, momentum=0.1):
        super().__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size, stride=1, padding=0, bias=True)
        self.bn = torch.nn.BatchNorm2d(out_channels, eps=eps, momentum=momentum, affine=True, track_running_stats=True)

    def forward(self, x):
        # Prepare parameters
        weight = self.conv.weight
        bias = self.conv.bias if self.conv.bias is not None else torch.zeros(self.conv.out_channels, device=x.device, dtype=x.dtype)
        gamma = self.bn.weight if self.bn.affine else torch.ones(self.bn.num_features, device=x.device, dtype=x.dtype)
        beta = self.bn.bias if self.bn.affine else torch.zeros(self.bn.num_features, device=x.device, dtype=x.dtype)
        mean = self.bn.running_mean if self.bn.track_running_stats else torch.zeros(self.bn.num_features, device=x.device, dtype=x.dtype)
        var = self.bn.running_var if self.bn.track_running_stats else torch.ones(self.bn.num_features, device=x.device, dtype=x.dtype)
        inv_std = torch.rsqrt(var + self.bn.eps)

        return custom_ops_lib.conv2d_activation_batch_norm_custom(x, weight, bias, gamma, beta, mean, inv_std)
'''
