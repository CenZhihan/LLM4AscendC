kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void bmm_instance_norm_sum_residual_add_multiply_custom(
    GM_ADDR x,
    GM_ADDR y,
    GM_ADDR weight,
    GM_ADDR bias,
    GM_ADDR gamma,
    GM_ADDR beta,
    GM_ADDR z,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t B = tiling_data.batch();
    const uint32_t K = tiling_data.inFeat();
    const uint32_t N = tiling_data.outFeat();

    __gm__ float* x_ptr = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* y_ptr = reinterpret_cast<__gm__ float*>(y);
    __gm__ float* w_ptr = reinterpret_cast<__gm__ float*>(weight); // shape [N, K], row-major
    __gm__ float* b_ptr = reinterpret_cast<__gm__ float*>(bias);   // shape [N]
    __gm__ float* g_ptr = reinterpret_cast<__gm__ float*>(gamma);  // shape [N]
    __gm__ float* be_ptr = reinterpret_cast<__gm__ float*>(beta);  // shape [N]
    __gm__ float* z_ptr = reinterpret_cast<__gm__ float*>(z);      // shape [B, N]

    const uint32_t block_num = AscendC::GetBlockNum();
    const uint32_t block_idx = AscendC::GetBlockIdx();

    uint32_t batch_per_block = (B + block_num - 1) / block_num;
    uint32_t b_start = block_idx * batch_per_block;
    uint32_t b_end = b_start + batch_per_block;
    if (b_end > B) b_end = B;

    const float eps = 1e-5f;

    for (uint32_t b = b_start; b < b_end; ++b) {
        float mean_acc = 0.0f;
        float sq_acc = 0.0f;

        // First pass: compute z_temp = x @ W^T + bias, accumulate stats, store into z
        for (uint32_t n = 0; n < N; ++n) {
            float acc = 0.0f;
            __gm__ float* w_row = w_ptr + (uint64_t)n * K;
            __gm__ float* x_row = x_ptr + (uint64_t)b * K;
            for (uint32_t k = 0; k < K; ++k) {
                acc += x_row[k] * w_row[k];
            }
            acc += b_ptr[n];
            z_ptr[(uint64_t)b * N + n] = acc;
            mean_acc += acc;
            sq_acc += acc * acc;
        }

        float mean = (N > 0) ? (mean_acc / (float)N) : 0.0f;
        float var = (N > 0) ? (sq_acc / (float)N - mean * mean) : 0.0f;
        if (var < 0.0f) var = 0.0f;
        float inv_std = AscendC::Rsqrt(var + eps);

        // Second pass: instance-norm-like (per sample over feature dim), affine, residual add and multiply
        for (uint32_t n = 0; n < N; ++n) {
            float val = z_ptr[(uint64_t)b * N + n];
            float norm = (val - mean) * inv_std;
            float affine = norm * g_ptr[n] + be_ptr[n];
            float yy = y_ptr[(uint64_t)b * N + n];
            float outv = (affine + yy) * yy;
            z_ptr[(uint64_t)b * N + n] = outv;
        }
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, eps=1e-5, momentum=0.1):
        super().__init__()
        self.bmm = torch.nn.Linear(in_features, out_features, bias=True)
        # Keep original InstanceNorm2d definition for parameter shapes (gamma/beta of length out_features)
        self.instance_norm = torch.nn.InstanceNorm2d(out_features, eps=eps, momentum=momentum, affine=True)

    def forward(self, x, y):
        # Pull parameters needed by the fused custom kernel
        W = self.bmm.weight
        b = self.bmm.bias

        gamma = self.instance_norm.weight
        beta = self.instance_norm.bias

        # Ensure all operands are on the same device/dtype and contiguous
        x = x.contiguous()
        y = y.contiguous()
        W = W.contiguous()
        b = b.contiguous()
        gamma = gamma.contiguous()
        beta = beta.contiguous()

        return custom_ops_lib.bmm_instance_norm_sum_residual_add_multiply_custom(x, y, W, b, gamma, beta)
