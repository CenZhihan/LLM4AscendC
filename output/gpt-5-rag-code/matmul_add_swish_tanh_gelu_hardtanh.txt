kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFusedEltwise {
public:
    __aicore__ inline KernelFusedEltwise() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, float hardMin, float hardMax)
    {
        this->blockLength = (totalLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        this->blockLength = this->blockLength > 0 ? this->blockLength : 1;
        this->tileNum = tileNum > 0 ? tileNum : 1;
        this->tileLength = (this->blockLength + (tileNum * BUFFER_NUM) - 1) / (tileNum * BUFFER_NUM);
        this->hardMin = hardMin;
        this->hardMax = hardMax;

        uint32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + blockOffset, this->blockLength);
        addGm.SetGlobalBuffer((__gm__ DTYPE_ADD *)add + blockOffset, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + blockOffset, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueAdd, BUFFER_NUM, this->tileLength * sizeof(DTYPE_ADD));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        // temp buffers for constants (hardtanh bounds)
        pipe.InitBuffer(tmpConstQueueMin, 1, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpConstQueueMax, 1, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        // prepare min/max tensors
        AscendC::LocalTensor<DTYPE_Y> minConst = tmpConstQueueMin.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> maxConst = tmpConstQueueMax.AllocTensor<DTYPE_Y>();
        AscendC::Duplicate(minConst, (DTYPE_Y)hardMin, this->tileLength);
        AscendC::Duplicate(maxConst, (DTYPE_Y)hardMax, this->tileLength);
        tmpConstQueueMin.EnQue(minConst);
        tmpConstQueueMax.EnQue(maxConst);

        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute();
            CopyOut(i);
        }

        // free const buffers
        minConst = tmpConstQueueMin.DeQue<DTYPE_Y>();
        maxConst = tmpConstQueueMax.DeQue<DTYPE_Y>();
        tmpConstQueueMin.FreeTensor(minConst);
        tmpConstQueueMax.FreeTensor(maxConst);
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint32_t remain = blockLength - progress * this->tileLength;
        uint32_t curLen = remain > this->tileLength ? this->tileLength : remain;

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_ADD> addLocal = inQueueAdd.AllocTensor<DTYPE_ADD>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], curLen);
        AscendC::DataCopy(addLocal, addGm[progress * this->tileLength], curLen);
        inQueueX.EnQue<DTYPE_X>(xLocal);
        inQueueAdd.EnQue<DTYPE_ADD>(addLocal);
        curTileLen = curLen;
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_ADD> addLocal = inQueueAdd.DeQue<DTYPE_ADD>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // 1) add
        AscendC::Add(yLocal, xLocal, addLocal, curTileLen);
        // 2) swish: sigmoid(y) * y
        AscendC::LocalTensor<DTYPE_Y> tmp1 = pipe.AllocTmpBuffer<DTYPE_Y>(curTileLen);
        AscendC::Sigmoid(tmp1, yLocal, curTileLen);
        AscendC::Mul(yLocal, tmp1, yLocal, curTileLen);
        // 3) tanh
        AscendC::Tanh(yLocal, yLocal, curTileLen);
        // 4) gelu
        AscendC::Gelu(yLocal, yLocal, curTileLen);
        // 5) hardtanh clamp to [hardMin, hardMax]
        AscendC::LocalTensor<DTYPE_Y> minConst = tmpConstQueueMin.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Y> maxConst = tmpConstQueueMax.DeQue<DTYPE_Y>();
        AscendC::Max(yLocal, yLocal, minConst, curTileLen);
        AscendC::Min(yLocal, yLocal, maxConst, curTileLen);
        tmpConstQueueMin.EnQue(minConst);
        tmpConstQueueMax.EnQue(maxConst);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        pipe.FreeTmpBuffer(tmp1);
        inQueueX.FreeTensor(xLocal);
        inQueueAdd.FreeTensor(addLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, curTileLen);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAdd;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpConstQueueMin;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpConstQueueMax;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_ADD> addGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t curTileLen;
    float hardMin;
    float hardMax;
};

extern "C" __global__ __aicore__ void matmul_add_swish_tanh_gelu_hardtanh_custom(GM_ADDR x, GM_ADDR add, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedEltwise op;
    op.Init(x, add, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.hardMin, tiling_data.hardMax);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib
import torch.nn.functional as F

class ModelNew(torch.nn.Module):
    """
    Optimized model: compute linear (matmul + bias) with PyTorch, then fuse: +add_value, Swish, Tanh, GELU, Hardtanh in a custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, add_value_shape):
        super(ModelNew, self).__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.add_value = torch.nn.Parameter(torch.randn(add_value_shape))

    def forward(self, x):
        # PyTorch linear for matmul + bias (runs on NPU if tensors are on NPU)
        y = F.linear(x, self.matmul.weight, self.matmul.bias)
        # Prepare broadcasted add tensor to match y's shape
        add = self.add_value.view(1, -1).expand_as(y).contiguous()
        # Fused elementwise block on Ascend (add + Swish + Tanh + GELU + Hardtanh)
        y = custom_ops_lib.matmul_add_swish_tanh_gelu_hardtanh_custom(y, add)
        return y
