kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConv2D {
public:
    __aicore__ inline KernelConv2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t hin, uint32_t win,
                                uint32_t cout, uint32_t kh, uint32_t kw,
                                uint32_t hout, uint32_t wout,
                                uint32_t totalOutElems, uint32_t tileNum)
    {
        this->n = n; this->cin = cin; this->hin = hin; this->win = win;
        this->cout = cout; this->kh = kh; this->kw = kw; this->hout = hout; this->wout = wout;
        this->K = cin * kh * kw;
        this->totalOutElems = totalOutElems;
        this->tileNum = tileNum;

        // Partition work across blocks
        this->blockWork = (totalOutElems + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t blockStart = this->blockWork * AscendC::GetBlockIdx();
        uint32_t blockEnd = blockStart + this->blockWork;
        if (blockEnd > totalOutElems) blockEnd = totalOutElems;
        this->startIdx = blockStart;
        this->endIdx = blockEnd;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)((uint64_t)n * cin * hin * win));
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)w, (uint32_t)((uint64_t)cout * cin * kh * kw));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, (uint32_t)((uint64_t)n * cout * hout * wout));

        // UB buffers
        // weights for one out-channel (K elements)
        pipe.InitBuffer(weightBufQ, BUFFER_NUM, this->K * sizeof(DTYPE_Y));
        // im2col patch for one output position (K elements)
        pipe.InitBuffer(patchBufQ, BUFFER_NUM, this->K * sizeof(DTYPE_X));
        // partial output (1 element per compute)
        pipe.InitBuffer(outBufQ, BUFFER_NUM, sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t idx = this->startIdx; idx < this->endIdx; ++idx) {
            // Decode linear index to (n, co, ho, wo)
            uint32_t tmp = idx;
            uint32_t wo = tmp % wout; tmp /= wout;
            uint32_t ho = tmp % hout; tmp /= hout;
            uint32_t co = tmp % cout; tmp /= cout;
            uint32_t n_ = tmp;

            // Load weight vector for this co: contiguous K
            AscendC::LocalTensor<DTYPE_Y> wLocal = weightBufQ.AllocTensor<DTYPE_Y>();
            uint32_t w_off = co * this->K;
            AscendC::DataCopy(wLocal, wGm[w_off], this->K);
            weightBufQ.EnQue(wLocal);

            // Build im2col patch for this (n, ho, wo): copy (Cin x Kh) rows of length Kw contiguous
            AscendC::LocalTensor<DTYPE_X> xPatch = patchBufQ.AllocTensor<DTYPE_X>();
            uint32_t patchBase = 0;
            for (uint32_t c = 0; c < cin; ++c) {
                for (uint32_t r = 0; r < kh; ++r) {
                    uint32_t hi = ho + r; // dilation=1, padding=0
                    uint32_t wi = wo;
                    // source offset for this row segment
                    uint64_t x_off = (((uint64_t)n_ * cin + c) * hin + hi) * win + wi;
                    AscendC::DataCopy(xPatch[patchBase], xGm[x_off], kw);
                    patchBase += kw;
                }
            }
            patchBufQ.EnQue(xPatch);

            // DeQue for compute
            wLocal = weightBufQ.DeQue<DTYPE_Y>();
            xPatch = patchBufQ.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Z> outLocal = outBufQ.AllocTensor<DTYPE_Z>();

            // Compute dot product: sum(xPatch * wLocal)
            // Vector multiply into temp and reduce
            // temp buffer reused in xPatch to save UB: multiply in-place
            AscendC::Mul(xPatch, xPatch, wLocal, this->K);
            // Reduce sum across K into a scalar
            DTYPE_Z acc = static_cast<DTYPE_Z>(0);
            // Accumulate in chunks
            const uint32_t vec = 256; // arbitrary chunk
            uint32_t k = 0;
            for (; k + vec <= this->K; k += vec) {
                AscendC::LocalTensor<DTYPE_Z> tmpBuf = outBufQ.AllocTensor<DTYPE_Z>();
                AscendC::ReduceSum(tmpBuf, xPatch[k], vec);
                // Copy tmp scalar to acc (host-like accumulation)
                // Simulate by reading back from tmpBuf[0]
                // Note: In actual AscendC, you would keep reductions in UB
                acc = acc + tmpBuf.GetValue(0);
                outBufQ.FreeTensor(tmpBuf);
            }
            // Tail
            if (k < this->K) {
                AscendC::LocalTensor<DTYPE_Z> tmpBuf2 = outBufQ.AllocTensor<DTYPE_Z>();
                AscendC::ReduceSum(tmpBuf2, xPatch[k], this->K - k);
                acc = acc + tmpBuf2.GetValue(0);
                outBufQ.FreeTensor(tmpBuf2);
            }

            // Write acc to outLocal[0]
            outLocal.SetValue(0, acc);

            // Enqueue and free
            outBufQ.EnQue(outLocal);
            weightBufQ.FreeTensor(wLocal);
            patchBufQ.FreeTensor(xPatch);

            // Copy out
            outLocal = outBufQ.DeQue<DTYPE_Z>();
            uint64_t y_off = (((uint64_t)n_ * cout + co) * hout + ho) * wout + wo;
            AscendC::DataCopy(yGm[y_off], outLocal, 1);
            outBufQ.FreeTensor(outLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> weightBufQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> patchBufQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outBufQ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t n, cin, hin, win;
    uint32_t cout, kh, kw, hout, wout;
    uint32_t K;
    uint32_t totalOutElems;
    uint32_t tileNum;

    uint32_t blockWork;
    uint32_t startIdx;
    uint32_t endIdx;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2D op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.hin, tiling_data.win,
            tiling_data.cout, tiling_data.kh, tiling_data.kw,
            tiling_data.hout, tiling_data.wout,
            tiling_data.totalOutElems, tiling_data.tileNum);
    op.Process();
}

import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Optimized 2D convolution using custom AscendC kernel for asymmetric inputs and square kernel.
    Constraints:
    - stride=1, padding=0, dilation=1, groups=1
    - square kernel
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        # Keep an internal Conv2d for parameter management; we will call custom op in forward.
        assert stride == 1 and padding == 0 and dilation == 1 and groups == 1, "Custom kernel supports stride=1, padding=0, dilation=1, groups=1"
        self.conv2d = nn.Conv2d(in_channels, out_channels, (kernel_size, kernel_size), stride=stride, padding=padding, dilation=dilation, groups=groups, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        y = custom_ops_lib.conv_standard_2d_asymmetric_input_square_kernel_custom(x, self.conv2d.weight)
        if self.conv2d.bias is not None:
            y = y + self.conv2d.bias.view(1, -1, 1, 1)
        return y
