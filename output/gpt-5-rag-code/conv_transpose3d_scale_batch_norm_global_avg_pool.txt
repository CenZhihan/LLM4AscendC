#include "kernel_operator.h"
#include <cmath>

using DTYPE = float;

constexpr int32_t IN_BUFFER_NUM = 2;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR bn_scale, GM_ADDR bn_bias,
                                GM_ADDR running_mean, GM_ADDR running_var, GM_ADDR eps,
                                GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t totalPairs, uint32_t spatialSize, uint32_t blockDimUsed, uint32_t tileLength)
    {
        this->n_ = n;
        this->c_ = c;
        this->d_ = d;
        this->h_ = h;
        this->w_ = w;
        this->totalPairs_ = totalPairs;
        this->spatialSize_ = spatialSize;
        this->blockDimUsed_ = blockDimUsed;
        this->tileLength_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ DTYPE *)x, static_cast<uint32_t>(n) * c * d * h * w);
        scaleGm_.SetGlobalBuffer((__gm__ DTYPE *)scale, 1);
        bnScaleGm_.SetGlobalBuffer((__gm__ DTYPE *)bn_scale, c);
        bnBiasGm_.SetGlobalBuffer((__gm__ DTYPE *)bn_bias, c);
        runningMeanGm_.SetGlobalBuffer((__gm__ DTYPE *)running_mean, c);
        runningVarGm_.SetGlobalBuffer((__gm__ DTYPE *)running_var, c);
        epsGm_.SetGlobalBuffer((__gm__ DTYPE *)eps, 1);
        yGm_.SetGlobalBuffer((__gm__ DTYPE *)y, n * c);

        pipe_.InitBuffer(inQueue_, IN_BUFFER_NUM, this->tileLength_ * sizeof(DTYPE));
        pipe_.InitBuffer(tmpQueue_, IN_BUFFER_NUM, this->tileLength_ * sizeof(DTYPE));
        pipe_.InitBuffer(outQueue_, IN_BUFFER_NUM, sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        uint32_t base = totalPairs_ / blockDimUsed_;
        uint32_t rem  = totalPairs_ % blockDimUsed_;
        uint32_t blk  = AscendC::GetBlockIdx();

        uint32_t myCount = (blk < rem) ? (base + 1) : base;
        uint32_t myStart = (blk < rem) ? (blk * (base + 1)) : (rem * (base + 1) + (blk - rem) * base);

        // Load scalar scale and eps once
        DTYPE scaleVal = 1.0f;
        DTYPE epsVal = 0.0f;
        {
            AscendC::LocalTensor<DTYPE> sLocal = tmpQueue_.AllocTensor<DTYPE>();
            AscendC::DataCopy(sLocal, scaleGm_, 1);
            tmpQueue_.EnQue<DTYPE>(sLocal);
            AscendC::LocalTensor<DTYPE> sUsed = tmpQueue_.DeQue<DTYPE>();
            scaleVal = sUsed.GetValue(0);
            tmpQueue_.FreeTensor(sUsed);
        }
        {
            AscendC::LocalTensor<DTYPE> eLocal = tmpQueue_.AllocTensor<DTYPE>();
            AscendC::DataCopy(eLocal, epsGm_, 1);
            tmpQueue_.EnQue<DTYPE>(eLocal);
            AscendC::LocalTensor<DTYPE> eUsed = tmpQueue_.DeQue<DTYPE>();
            epsVal = eUsed.GetValue(0);
            tmpQueue_.FreeTensor(eUsed);
        }

        for (uint32_t t = 0; t < myCount; ++t) {
            uint32_t pairIdx = myStart + t;
            uint32_t nIdx = pairIdx / c_;
            uint32_t cIdx = pairIdx % c_;

            // load bn params for this channel
            DTYPE gammaVal, betaVal, meanVal, varVal;
            {
                AscendC::LocalTensor<DTYPE> tmp = tmpQueue_.AllocTensor<DTYPE>();
                AscendC::DataCopy(tmp, bnScaleGm_[cIdx], 1);
                tmpQueue_.EnQue<DTYPE>(tmp);
                AscendC::LocalTensor<DTYPE> t0 = tmpQueue_.DeQue<DTYPE>();
                gammaVal = t0.GetValue(0);
                tmpQueue_.FreeTensor(t0);
            }
            {
                AscendC::LocalTensor<DTYPE> tmp = tmpQueue_.AllocTensor<DTYPE>();
                AscendC::DataCopy(tmp, bnBiasGm_[cIdx], 1);
                tmpQueue_.EnQue<DTYPE>(tmp);
                AscendC::LocalTensor<DTYPE> t0 = tmpQueue_.DeQue<DTYPE>();
                betaVal = t0.GetValue(0);
                tmpQueue_.FreeTensor(t0);
            }
            {
                AscendC::LocalTensor<DTYPE> tmp = tmpQueue_.AllocTensor<DTYPE>();
                AscendC::DataCopy(tmp, runningMeanGm_[cIdx], 1);
                tmpQueue_.EnQue<DTYPE>(tmp);
                AscendC::LocalTensor<DTYPE> t0 = tmpQueue_.DeQue<DTYPE>();
                meanVal = t0.GetValue(0);
                tmpQueue_.FreeTensor(t0);
            }
            {
                AscendC::LocalTensor<DTYPE> tmp = tmpQueue_.AllocTensor<DTYPE>();
                AscendC::DataCopy(tmp, runningVarGm_[cIdx], 1);
                tmpQueue_.EnQue<DTYPE>(tmp);
                AscendC::LocalTensor<DTYPE> t0 = tmpQueue_.DeQue<DTYPE>();
                varVal = t0.GetValue(0);
                tmpQueue_.FreeTensor(t0);
            }

            // sum over spatial for this (n,c)
            uint32_t baseOffset = (nIdx * c_ + cIdx) * spatialSize_;
            DTYPE acc = 0.0f;
            uint32_t processed = 0;
            while (processed < spatialSize_) {
                uint32_t len = AscendC::Min(this->tileLength_, spatialSize_ - processed);
                AscendC::LocalTensor<DTYPE> inLocal = inQueue_.AllocTensor<DTYPE>();
                AscendC::DataCopy(inLocal, xGm_[baseOffset + processed], len);
                inQueue_.EnQue<DTYPE>(inLocal);
                AscendC::LocalTensor<DTYPE> computeLocal = inQueue_.DeQue<DTYPE>();
#pragma unroll
                for (uint32_t i = 0; i < len; ++i) {
                    acc += computeLocal.GetValue(i);
                }
                inQueue_.FreeTensor(computeLocal);
                processed += len;
            }

            DTYPE avg = acc * (scaleVal / static_cast<DTYPE>(spatialSize_));
            DTYPE invstd = static_cast<DTYPE>(1.0f) / static_cast<DTYPE>(sqrt(static_cast<float>(varVal + epsVal)));
            DTYPE outVal = (avg - meanVal) * invstd * gammaVal + betaVal;

            AscendC::LocalTensor<DTYPE> outLocal = outQueue_.AllocTensor<DTYPE>();
            outLocal.SetValue(0, outVal);
            outQueue_.EnQue<DTYPE>(outLocal);
            AscendC::LocalTensor<DTYPE> outUsed = outQueue_.DeQue<DTYPE>();
            AscendC::DataCopy(yGm_[nIdx * c_ + cIdx], outUsed, 1);
            outQueue_.FreeTensor(outUsed);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, IN_BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<DTYPE> xGm_;
    AscendC::GlobalTensor<DTYPE> scaleGm_;
    AscendC::GlobalTensor<DTYPE> bnScaleGm_;
    AscendC::GlobalTensor<DTYPE> bnBiasGm_;
    AscendC::GlobalTensor<DTYPE> runningMeanGm_;
    AscendC::GlobalTensor<DTYPE> runningVarGm_;
    AscendC::GlobalTensor<DTYPE> epsGm_;
    AscendC::GlobalTensor<DTYPE> yGm_;

    uint32_t n_, c_, d_, h_, w_;
    uint32_t totalPairs_, spatialSize_, blockDimUsed_, tileLength_;
};

extern "C" __global__ __aicore__ void conv_transpose3d_scale_batch_norm_global_avg_pool_custom(GM_ADDR x,
                                                                                                GM_ADDR scale,
                                                                                                GM_ADDR bn_scale,
                                                                                                GM_ADDR bn_bias,
                                                                                                GM_ADDR running_mean,
                                                                                                GM_ADDR running_var,
                                                                                                GM_ADDR eps,
                                                                                                GM_ADDR y,
                                                                                                GM_ADDR workspace,
                                                                                                GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, scale, bn_scale, bn_bias, running_mean, running_var, eps, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.totalPairs, tiling_data.spatialSize, tiling_data.blockDimUsed, tiling_data.tileLength);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized: keeps ConvTranspose3d in PyTorch, fuses scale + BatchNorm3d (inference) + global avg pool into one AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor, eps=1e-5, momentum=0.1):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size)
        self.scale_factor = float(scale_factor)
        self.batch_norm = torch.nn.BatchNorm3d(out_channels, eps=eps, momentum=momentum)

    def forward(self, x):
        # y: [N, C, D, H, W]
        y = self.conv_transpose(x)
        # Use custom fused kernel to perform: scale -> batchnorm (inference affine using running stats) -> global avg pool (to 1x1x1)
        scale_t = torch.tensor([self.scale_factor], dtype=y.dtype, device=y.device)
        eps_t = torch.tensor([self.batch_norm.eps], dtype=y.dtype, device=y.device)
        return custom_ops_lib.conv_transpose3d_scale_batch_norm_global_avg_pool_custom(
            y,
            scale_t,
            self.batch_norm.weight,     # gamma
            self.batch_norm.bias,       # beta
            self.batch_norm.running_mean,
            self.batch_norm.running_var,
            eps_t
        )
'''
