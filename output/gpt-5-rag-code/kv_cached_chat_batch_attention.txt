    tiling.set_q_len(q_len);

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelKvCachedChatBatchAttention {
public:
    __aicore__ inline KernelKvCachedChatBatchAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y,
                                uint32_t batch, uint32_t q_len, uint32_t kv_len, uint32_t d_model,
                                uint32_t tile_k, float scale)
    {
        this->batch = batch;
        this->q_len = q_len;
        this->kv_len = kv_len;
        this->d_model = d_model;
        this->tile_k = tile_k;
        this->scale = scale;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blocks = AscendC::GetBlockNum();
        // map block to (b, q_index)
        uint32_t bq_span = batch * q_len;
        uint32_t mapped = blockIdx % bq_span;
        this->b = mapped / q_len;
        this->q_index = mapped % q_len;

        // set base pointers for this block
        // Global layout:
        // Q: [B, QL, D]
        // K/V: [B, KV, D]
        // Y: [B, QL, D]
        // point Q and Y to the specific (b, q_index)
        qGm.SetGlobalBuffer((__gm__ DTYPE*)q + ((uint64_t)this->b * this->q_len + this->q_index) * this->d_model, this->d_model);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + ((uint64_t)this->b * this->q_len + this->q_index) * this->d_model, this->d_model);

        // K/V base at (b, 0, 0)
        uint64_t kv_base_offset = (uint64_t)this->b * this->kv_len * this->d_model;
        kGm.SetGlobalBuffer((__gm__ DTYPE*)k + kv_base_offset, this->kv_len * this->d_model);
        vGm.SetGlobalBuffer((__gm__ DTYPE*)v + kv_base_offset, this->kv_len * this->d_model);

        // init UB buffers
        pipe.InitBuffer(qBuf, 1, this->d_model * sizeof(DTYPE));
        pipe.InitBuffer(kBuf, 1, this->d_model * sizeof(DTYPE));
        pipe.InitBuffer(vBuf, 1, this->d_model * sizeof(DTYPE));
        pipe.InitBuffer(oBuf, 1, this->d_model * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // Load Q vector
        AscendC::LocalTensor<DTYPE> qLocal = qBuf.AllocTensor<DTYPE>();
        AscendC::DataCopy(qLocal, qGm[0], this->d_model);
        qBuf.EnQue(qLocal);

        // Prepare output accumulator and streaming softmax state
        AscendC::LocalTensor<DTYPE> outLocal = oBuf.AllocTensor<DTYPE>();
        // zero init outLocal
        for (uint32_t i = 0; i < this->d_model; ++i) {
            outLocal.SetValue(i, (DTYPE)0);
        }

        float m = -INFINITY;
        float sumExp = 0.0f;

        // DeQue Q for compute
        AscendC::LocalTensor<DTYPE> qVec = qBuf.DeQue<DTYPE>();

        // Stream over KV dimension
        for (uint32_t start = 0; start < this->kv_len; start += this->tile_k) {
            uint32_t curr = AscendC::min(this->tile_k, this->kv_len - start);
            for (uint32_t r = 0; r < curr; ++r) {
                uint32_t kv_index = start + r;

                AscendC::LocalTensor<DTYPE> kLocal = kBuf.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> vLocal = vBuf.AllocTensor<DTYPE>();

                // copy one row of K and V
                AscendC::DataCopy(kLocal, kGm[(uint64_t)kv_index * this->d_model], this->d_model);
                AscendC::DataCopy(vLocal, vGm[(uint64_t)kv_index * this->d_model], this->d_model);

                // compute score = (q dot k) * scale
                float dot = 0.0f;
                for (uint32_t i = 0; i < this->d_model; ++i) {
                    dot += static_cast<float>(qVec.GetValue(i)) * static_cast<float>(kLocal.GetValue(i));
                }
                float score = dot * this->scale;

                // streaming softmax update
                if (score > m) {
                    float factor = std::exp(m - score);
                    sumExp *= factor;
                    // scale outLocal by factor
                    for (uint32_t i = 0; i < this->d_model; ++i) {
                        float val = static_cast<float>(outLocal.GetValue(i));
                        outLocal.SetValue(i, static_cast<DTYPE>(val * factor));
                    }
                    m = score;
                }
                float w = std::exp(score - m);
                sumExp += w;
                // out += w * v
                for (uint32_t i = 0; i < this->d_model; ++i) {
                    float val = static_cast<float>(outLocal.GetValue(i));
                    float vv = static_cast<float>(vLocal.GetValue(i));
                    outLocal.SetValue(i, static_cast<DTYPE>(val + w * vv));
                }

                vBuf.FreeTensor(vLocal);
                kBuf.FreeTensor(kLocal);
            }
        }

        // normalize by sumExp
        float invSum = 1.0f / sumExp;
        for (uint32_t i = 0; i < this->d_model; ++i) {
            float val = static_cast<float>(outLocal.GetValue(i));
            outLocal.SetValue(i, static_cast<DTYPE>(val * invSum));
        }

        // write back
        AscendC::DataCopy(yGm[0], outLocal, this->d_model);

        oBuf.FreeTensor(outLocal);
        qBuf.FreeTensor(qVec);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qBuf, kBuf, vBuf;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> oBuf;

    AscendC::GlobalTensor<DTYPE> qGm;
    AscendC::GlobalTensor<DTYPE> kGm;
    AscendC::GlobalTensor<DTYPE> vGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t batch;
    uint32_t q_len;
    uint32_t kv_len;
    uint32_t d_model;
    uint32_t tile_k;
    float scale;
    uint32_t b;
    uint32_t q_index;
};

extern "C" __global__ __aicore__ void kv_cached_chat_batch_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelKvCachedChatBatchAttention op;
    op.Init(q, k, v, y,
            tiling_data.batch, tiling_data.q_len, tiling_data.kv_len, tiling_data.d_model,
            tiling_data.tile_k, tiling_data.scale);
    op.Process();
}
