kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmSwishDivideClampTanhClamp {
public:
    __aicore__ inline KernelGemmSwishDivideClampTanhClamp() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t blockM, uint32_t tileN, uint32_t kStep)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->blockM = blockM;
        this->tileN = tileN;
        this->kStep = kStep;

        uint32_t blockStartM = AscendC::GetBlockIdx() * blockM;
        if (blockStartM + blockM > M) {
            this->blockM = M - blockStartM;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + blockStartM * K, this->blockM * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)w, N * K);   // weight: [N, K]
        bGm.SetGlobalBuffer((__gm__ DTYPE_Z *)b, N);       // bias: [N]
        yGm.SetGlobalBuffer((__gm__ DTYPE_W *)y + blockStartM * N, this->blockM * N);

        // UB buffers
        pipe.InitBuffer(xRowQueue, 1, kStep * sizeof(DTYPE_X));               // x chunk
        pipe.InitBuffer(wTileQueue, 1, tileN * kStep * sizeof(DTYPE_Y));      // weight tile chunk
        pipe.InitBuffer(yTileQueue, BUFFER_NUM, tileN * sizeof(DTYPE_W));     // partial / result tile
        pipe.InitBuffer(bTileQueue, 1, tileN * sizeof(DTYPE_Z));              // bias tile
    }

    __aicore__ inline void Process()
    {
        for (uint32_t m = 0; m < this->blockM; ++m) {
            // We'll iterate N by tiles
            for (uint32_t nStart = 0; nStart < this->N; nStart += this->tileN) {
                uint32_t curTileN = AscendC::Min(this->tileN, this->N - nStart);
                // Allocate y tile and initialize with bias
                AscendC::LocalTensor<DTYPE_W> yTile = yTileQueue.AllocTensor<DTYPE_W>();
                ZeroTile(yTile, curTileN);

                // Load bias tile
                AscendC::LocalTensor<DTYPE_Z> bTile = bTileQueue.AllocTensor<DTYPE_Z>();
                AscendC::DataCopy(bTile, bGm[nStart], curTileN);
                bTileQueue.EnQue<DTYPE_Z>(bTile);

                // GEMM accumulation over K dimension by chunks
                uint32_t kProcessed = 0;
                while (kProcessed < this->K) {
                    uint32_t curK = AscendC::Min(this->kStep, this->K - kProcessed);

                    // Load x row chunk [curK]
                    AscendC::LocalTensor<DTYPE_X> xChunk = xRowQueue.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(xChunk, xGm[m * this->K + kProcessed], curK);
                    xRowQueue.EnQue<DTYPE_X>(xChunk);

                    // Load weight tile chunk of shape [curTileN, curK] laid as row-major blocks
                    AscendC::LocalTensor<DTYPE_Y> wChunk = wTileQueue.AllocTensor<DTYPE_Y>();
                    // Copy row by row
                    for (uint32_t tn = 0; tn < curTileN; ++tn) {
                        AscendC::DataCopy(wChunk[tn * curK], wGm[(nStart + tn) * this->K + kProcessed], curK);
                    }
                    wTileQueue.EnQue<DTYPE_Y>(wChunk);

                    // Compute partial yTile += xChunk dot wChunk rows
                    xChunk = xRowQueue.DeQue<DTYPE_X>();
                    wChunk = wTileQueue.DeQue<DTYPE_Y>();

                    for (uint32_t tn = 0; tn < curTileN; ++tn) {
                        DTYPE_W acc = static_cast<DTYPE_W>(0);
                        // unrolled/fused inner product for this (m, nStart+tn) over curK
                        for (uint32_t kk = 0; kk < curK; ++kk) {
                            acc += static_cast<DTYPE_W>(xChunk[kk]) * static_cast<DTYPE_W>(wChunk[tn * curK + kk]);
                        }
                        yTile[tn] += acc;
                    }

                    // Free consumed UB tensors
                    xRowQueue.FreeTensor(xChunk);
                    wTileQueue.FreeTensor(wChunk);

                    kProcessed += curK;
                }

                // Add bias
                bTile = bTileQueue.DeQue<DTYPE_Z>();
                for (uint32_t tn = 0; tn < curTileN; ++tn) {
                    yTile[tn] += static_cast<DTYPE_W>(bTile[tn]);
                }
                bTileQueue.FreeTensor(bTile);

                // Fused pointwise: swish, divide by 2, clamp[-1,1], tanh, clamp[-1,1]
                FusedPointwise(yTile, curTileN);

                // Store result
                yTileQueue.EnQue<DTYPE_W>(yTile);
                AscendC::LocalTensor<DTYPE_W> yOut = yTileQueue.DeQue<DTYPE_W>();
                AscendC::DataCopy(yGm[m * this->N + nStart], yOut, curTileN);
                yTileQueue.FreeTensor(yOut);
            }
        }
    }

private:
    __aicore__ inline void ZeroTile(AscendC::LocalTensor<DTYPE_W>& t, uint32_t len) {
        for (uint32_t i = 0; i < len; ++i) {
            t[i] = static_cast<DTYPE_W>(0);
        }
    }

    __aicore__ inline DTYPE_W ClampUnit(DTYPE_W v) {
        if (v > static_cast<DTYPE_W>(1.0f)) return static_cast<DTYPE_W>(1.0f);
        if (v < static_cast<DTYPE_W>(-1.0f)) return static_cast<DTYPE_W>(-1.0f);
        return v;
    }

    __aicore__ inline DTYPE_W FastTanh(DTYPE_W x) {
        // Use builtin tanh if available, else a simple polynomial approximation
        // tanh(x) ~ x*(27 + x^2)/(27 + 9*x^2)
        DTYPE_W x2 = x * x;
        return x * (static_cast<DTYPE_W>(27.0f) + x2) / (static_cast<DTYPE_W>(27.0f) + static_cast<DTYPE_W>(9.0f) * x2);
    }

    __aicore__ inline DTYPE_W FastExp(DTYPE_W x) {
        // Very rough exp approximation for demonstration, not numerically stable
        // exp(x) ~ 1 + x + x^2/2 + x^3/6 for small x
        DTYPE_W x2 = x * x;
        DTYPE_W x3 = x2 * x;
        return static_cast<DTYPE_W>(1.0f) + x + static_cast<DTYPE_W>(0.5f) * x2 + static_cast<DTYPE_W>(0.16666667f) * x3;
    }

    __aicore__ inline DTYPE_W FastSigmoid(DTYPE_W x) {
        // sigmoid(x) = 1 / (1 + exp(-x))
        DTYPE_W ex = FastExp(-x);
        return static_cast<DTYPE_W>(1.0f) / (static_cast<DTYPE_W>(1.0f) + ex);
    }

    __aicore__ inline void FusedPointwise(AscendC::LocalTensor<DTYPE_W>& t, uint32_t len) {
        for (uint32_t i = 0; i < len; ++i) {
            DTYPE_W v = t[i];
            DTYPE_W s = FastSigmoid(v);
            v = v * s;                           // swish
            v = v * static_cast<DTYPE_W>(0.5f);  // divide by 2
            v = ClampUnit(v);                    // clamp [-1, 1]
            v = FastTanh(v);                     // tanh
            v = ClampUnit(v);                    // clamp [-1, 1]
            t[i] = v;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xRowQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wTileQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bTileQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yTileQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm; // [blockM * K]
    AscendC::GlobalTensor<DTYPE_Y> wGm; // [N * K]
    AscendC::GlobalTensor<DTYPE_Z> bGm; // [N]
    AscendC::GlobalTensor<DTYPE_W> yGm; // [blockM * N]

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t blockM;
    uint32_t tileN;
    uint32_t kStep;
};

extern "C" __global__ __aicore__ void gemm_swish_divide_clamp_tanh_clamp_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSwishDivideClampTanhClamp op;
    op.Init(x, weight, bias, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.blockM, tiling_data.tileN, tiling_data.kStep);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused Linear + Swish + /2 + Clamp + Tanh + Clamp using custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, bias=True):
        super(ModelNew, self).__init__()
        self.weight = torch.nn.Parameter(torch.empty(out_features, in_features))
        torch.nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        if bias:
            self.bias = torch.nn.Parameter(torch.zeros(out_features))
        else:
            self.register_parameter("bias", None)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        if self.bias is None:
            # Use zero bias if not provided
            bias = torch.zeros(self.weight.size(0), dtype=x.dtype, device=x.device)
        else:
            bias = self.bias
        return custom_ops_lib.gemm_swish_divide_clamp_tanh_clamp_custom(x, self.weight, bias)
