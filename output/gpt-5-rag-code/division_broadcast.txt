kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // double buffer for pipeline

// This kernel divides x by channel-wise scale, broadcasting scale over N,H,W.
// Shapes:
//   x: [N, C, H, W] contiguous (NCHW)
//   scale: [C, 1, 1] => effectively [C]
class KernelDivisionBroadcast {
public:
    __aicore__ inline KernelDivisionBroadcast() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t tileLen)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->groups = n * c;
        this->groupSize = h * w;
        this->tileLenMax = tileLen;

        // distribute groups across blocks
        coreNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        blockId = static_cast<uint32_t>(AscendC::GetBlockIdx());
        perCoreGroup = groups / coreNum;
        tailGroup = groups % coreNum;
        startGroup = blockId * perCoreGroup;
        if (blockId < tailGroup) {
            perCoreGroup++;
            startGroup += blockId;
        } else {
            startGroup += tailGroup;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, n * c * h * w);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, n * c * h * w);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_Y *)scale, c);

        // queues for x and y tiles
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLenMax * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLenMax * sizeof(DTYPE_Z));
        // queue for loading one scalar scale value (1 element)
        pipe.InitBuffer(inQueueScale, BUFFER_NUM, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t g = 0; g < perCoreGroup; ++g) {
            uint32_t groupId = startGroup + g;                      // group over (n,c)
            uint32_t cIdx = groupId % c;                            // channel index
            uint32_t baseOffset = groupId * groupSize;              // base offset in elements

            // load the scalar scale[cIdx] once per group into local
            AscendC::LocalTensor<DTYPE_Y> scaleLocal = inQueueScale.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(scaleLocal, scaleGm[cIdx], 1);
            inQueueScale.EnQue(scaleLocal);

            // process this group's H*W elements in tiles
            uint32_t done = 0;
            while (done < groupSize) {
                uint32_t curLen = AscendC::min(this->tileLenMax, groupSize - done);

                CopyIn(baseOffset + done, curLen);
                Compute(curLen);
                CopyOut(baseOffset + done, curLen);

                done += curLen;
            }

            // free scaleLocal after done
            scaleLocal = inQueueScale.DeQue<DTYPE_Y>();
            inQueueScale.FreeTensor(scaleLocal);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[offset], len);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> scaleLocal = inQueueScale.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // scalar value for this group
        DTYPE_Y s = scaleLocal.GetValue(0);
        // elementwise division by scalar: y = x / s
        AscendC::Divs(yLocal, xLocal, s, len);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        // keep scaleLocal for this group: re-enqueue it for reuse until group is done
        inQueueScale.EnQue(scaleLocal);
    }

    __aicore__ inline void CopyOut(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[offset], yLocal, len);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueScale;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> scaleGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t n, c, h, w;
    uint32_t groups;
    uint32_t groupSize;
    uint32_t tileLenMax;

    uint32_t coreNum;
    uint32_t blockId;
    uint32_t perCoreGroup;
    uint32_t tailGroup;
    uint32_t startGroup;
};

extern "C" __global__ __aicore__ void division_broadcast_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDivisionBroadcast op;
    op.Init(x, scale, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.tileLen);
    op.Process();
}
"""
