kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConv3dGroupNormMinClampDropoutCustom {
public:
    __aicore__ inline KernelConv3dGroupNormMinClampDropoutCustom() : minVal(0.f), maxVal(1.f), dropP(0.f) {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR min_val, GM_ADDR max_val, GM_ADDR dropout_p, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // Scalar GM buffers (expect 1-element tensors)
        minGm.SetGlobalBuffer((__gm__ DTYPE_X *)min_val, 1);
        maxGm.SetGlobalBuffer((__gm__ DTYPE_X *)max_val, 1);
        probGm.SetGlobalBuffer((__gm__ DTYPE_X *)dropout_p, 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));

        // small queues for constants
        pipe.InitBuffer(inQueueMin, 1, sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueMax, 1, sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueProb, 1, sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        LoadConstants();

        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void LoadConstants()
    {
        // Load min
        {
            AscendC::LocalTensor<DTYPE_X> minLocal = inQueueMin.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(minLocal, minGm, 1);
            inQueueMin.EnQue(minLocal);
            AscendC::LocalTensor<DTYPE_X> minTmp = inQueueMin.DeQue<DTYPE_X>();
            minVal = minTmp.GetValue(0);
            inQueueMin.FreeTensor(minTmp);
        }
        // Load max
        {
            AscendC::LocalTensor<DTYPE_X> maxLocal = inQueueMax.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(maxLocal, maxGm, 1);
            inQueueMax.EnQue(maxLocal);
            AscendC::LocalTensor<DTYPE_X> maxTmp = inQueueMax.DeQue<DTYPE_X>();
            maxVal = maxTmp.GetValue(0);
            inQueueMax.FreeTensor(maxTmp);
        }
        // Load dropout probability
        {
            AscendC::LocalTensor<DTYPE_X> probLocal = inQueueProb.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(probLocal, probGm, 1);
            inQueueProb.EnQue(probLocal);
            AscendC::LocalTensor<DTYPE_X> probTmp = inQueueProb.DeQue<DTYPE_X>();
            dropP = probTmp.GetValue(0);
            inQueueProb.FreeTensor(probTmp);
        }
    }

    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // y1 = min(x, minVal)
        AscendC::Mins(yLocal, xLocal, minVal, this->tileLength);
        // y2 = clamp lower bound: max(y1, minVal)
        AscendC::Maxs(yLocal, yLocal, minVal, this->tileLength);
        // y3 = clamp upper bound: min(y2, maxVal)
        AscendC::Mins(yLocal, yLocal, maxVal, this->tileLength);

        // Dropout (inference-style pass-through; no randomness)
        // scale = 1.0f; if desired, could use (1 - dropP)
        // AscendC::Muls(yLocal, yLocal, 1.0f, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    // queues for scalar constants
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueMin, inQueueMax, inQueueProb;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    AscendC::GlobalTensor<DTYPE_X> minGm;
    AscendC::GlobalTensor<DTYPE_X> maxGm;
    AscendC::GlobalTensor<DTYPE_X> probGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    float minVal;
    float maxVal;
    float dropP;
};

extern "C" __global__ __aicore__ void conv3d_group_norm_min_clamp_dropout_custom(GM_ADDR x,
                                                                                  GM_ADDR min_val,
                                                                                  GM_ADDR max_val,
                                                                                  GM_ADDR dropout_p,
                                                                                  GM_ADDR y,
                                                                                  GM_ADDR workspace,
                                                                                  GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dGroupNormMinClampDropoutCustom op;
    op.Init(x, min_val, max_val, dropout_p, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: retains conv3d and groupnorm in PyTorch, fuses min + clamp + dropout into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, groups, min_value, max_value, dropout_p):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.norm = torch.nn.GroupNorm(groups, out_channels)
        self.min_value = float(min_value)
        self.max_value = float(max_value)
        self.dropout_p = float(dropout_p)

    def forward(self, x):
        x = self.conv(x)
        x = self.norm(x)
        x = custom_ops_lib.conv3d_group_norm_min_clamp_dropout_custom(
            x, self.min_value, self.max_value, self.dropout_p
        )
        return x
