host_operator_src="""
#include "clamp_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t BUFFER_NUM = 2;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ClampBroadcastCustomTilingData tiling;

    // total number of elements in x
    uint32_t totalLength = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    // inner dimension size taken from min_val (broadcast along last dim)
    uint32_t innerK = context->GetInputShape(1)->GetOriginShape().GetShapeSize();

    context->SetBlockDim(BLOCK_DIM);

    uint32_t blockLength = totalLength / BLOCK_DIM;
    // Ensure tileLength equals innerK so each tile corresponds to one row (or innermost dimension span)
    uint32_t loopsPerBlock = (innerK == 0) ? 0 : (blockLength / innerK);
    uint32_t tileNum = (loopsPerBlock >= BUFFER_NUM) ? (loopsPerBlock / BUFFER_NUM) : 0;

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(tileNum);
    tiling.set_innerK(innerK);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}


namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelClampBroadcast {
public:
    __aicore__ inline KernelClampBroadcast() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR minv, GM_ADDR maxv, GM_ADDR y, uint32_t totalLength, uint32_t tileNum, uint32_t innerK)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = innerK;
        this->innerK = innerK;

        uint32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + blockOffset, this->blockLength);
        minGm.SetGlobalBuffer((__gm__ DTYPE_Y *)minv, this->innerK);
        maxGm.SetGlobalBuffer((__gm__ DTYPE_Z *)maxv, this->innerK);
        yGm.SetGlobalBuffer((__gm__ DTYPE_W *)y + blockOffset, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueMin, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(inQueueMax, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_W));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> minLocal = inQueueMin.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> maxLocal = inQueueMax.AllocTensor<DTYPE_Z>();

        // x tile advances by tileLength for each progress
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        // min/max are broadcast along the last dimension: always copy from start
        AscendC::DataCopy(minLocal, minGm[0], this->tileLength);
        AscendC::DataCopy(maxLocal, maxGm[0], this->tileLength);

        inQueueX.EnQue(xLocal);
        inQueueMin.EnQue(minLocal);
        inQueueMax.EnQue(maxLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> minLocal = inQueueMin.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> maxLocal = inQueueMax.DeQue<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_W> yLocal = outQueueY.AllocTensor<DTYPE_W>();

        // y = min(max(x, min_val), max_val)
        AscendC::Maximum(yLocal, xLocal, minLocal, this->tileLength);
        AscendC::Minimum(yLocal, yLocal, maxLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_W>(yLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueMin.FreeTensor(minLocal);
        inQueueMax.FreeTensor(maxLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_W> yLocal = outQueueY.DeQue<DTYPE_W>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMin;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMax;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> minGm;
    AscendC::GlobalTensor<DTYPE_Z> maxGm;
    AscendC::GlobalTensor<DTYPE_W> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t innerK;
};

extern "C" __global__ __aicore__ void clamp_broadcast_custom(GM_ADDR x, GM_ADDR min_val, GM_ADDR max_val, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelClampBroadcast op;
    op.Init(x, min_val, max_val, y, tiling_data.totalLength, tiling_data.tileNum, tiling_data.innerK);
    op.Process();
}
