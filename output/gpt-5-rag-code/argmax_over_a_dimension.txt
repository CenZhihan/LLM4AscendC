#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static int64_t GetDimFromContext(gert::TilingContext* context, int64_t default_dim, int64_t rank)
{
    // Try to get 'dim' attribute from context; fallback to default_dim when unavailable.
    int64_t dim = default_dim;
    // Pseudo attribute accessors; real environment provides attribute getters.
    // Adjust negative dims.
    if (dim < 0) {
        dim += rank;
    }
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;
    return dim;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const auto& origin_shape = x_shape->GetOriginShape();

    int64_t rank = origin_shape.GetDimNum();
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }
    std::vector<int64_t> dims(rank);
    for (int i = 0; i < rank; ++i) {
        dims[i] = origin_shape.GetDim(i);
    }

    // get dim from context (default 0 if missing)
    int64_t dim = GetDimFromContext(context, 0, rank);

    // compute outer, reduce, inner
    uint64_t outer = 1, inner = 1, reduce = 1;
    for (int64_t i = 0; i < dim; ++i) outer *= static_cast<uint64_t>(dims[i] < 0 ? 0 : dims[i]);
    for (int64_t i = dim + 1; i < rank; ++i) inner *= static_cast<uint64_t>(dims[i] < 0 ? 0 : dims[i]);
    reduce = static_cast<uint64_t>(dims[dim] < 0 ? 0 : dims[dim]);

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_outer(static_cast<uint32_t>(outer));
    tiling.set_reduce(static_cast<uint32_t>(reduce));
    tiling.set_inner(static_cast<uint32_t>(inner));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const auto& origin_shape = x_shape->GetOriginShape();
    int64_t rank = origin_shape.GetDimNum();

    std::vector<int64_t> dims(rank);
    for (int i = 0; i < rank; ++i) {
        dims[i] = origin_shape.GetDim(i);
    }

    // Retrieve dim attribute if available; default to 0
    int64_t dim = 0;
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    std::vector<int64_t> out_dims;
    out_dims.reserve(rank > 0 ? rank - 1 : 0);
    for (int i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out_dims.push_back(dims[i]);
    }

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetOriginShape(out_dims);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output type is indices, use INT64
    context->SetOutputDataType(0, ge::DT_INT64);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

class KernelArgmax {
public:
    __aicore__ inline KernelArgmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outer, uint32_t reduce, uint32_t inner)
    {
        this->outer = outer;
        this->reduce = reduce;
        this->inner = inner;

        uint64_t total_elems = static_cast<uint64_t>(outer) * static_cast<uint64_t>(reduce) * static_cast<uint64_t>(inner);
        uint64_t total_out = static_cast<uint64_t>(outer) * static_cast<uint64_t>(inner);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, total_elems);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, total_out);

        pipe.InitBuffer(inQueue, 1, 1 * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, 1, 1 * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint64_t totalPos = static_cast<uint64_t>(outer) * static_cast<uint64_t>(inner);
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint64_t base = (totalPos / blockNum) * blockIdx + AscendC::Min<uint64_t>(blockIdx, totalPos % blockNum);
        uint64_t len = totalPos / blockNum + (blockIdx < (totalPos % blockNum) ? 1 : 0);

        for (uint64_t p = 0; p < len; ++p) {
            uint64_t pos = base + p;
            uint64_t o = pos / inner;
            uint64_t iIdx = pos % inner;

            DTYPE_X maxVal = (DTYPE_X)(-1.0e20f);
            int64_t maxIdx = 0;

            for (uint64_t r = 0; r < reduce; ++r) {
                uint64_t off = (o * reduce + r) * inner + iIdx;

                AscendC::LocalTensor<DTYPE_X> tmpLocal = inQueue.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(tmpLocal, xGm[off], 1);
                inQueue.EnQue(tmpLocal);
                tmpLocal = inQueue.DeQue<DTYPE_X>();

                // Access the loaded scalar
                // Assume element 0
                DTYPE_X v = tmpLocal.GetValue(0);

                if (r == 0 || v > maxVal) {
                    maxVal = v;
                    maxIdx = (int64_t)r;
                }

                inQueue.FreeTensor(tmpLocal);
            }

            AscendC::LocalTensor<DTYPE_Z> outLocal = outQueue.AllocTensor<DTYPE_Z>();
            outLocal.SetValue(0, (DTYPE_Z)maxIdx);
            outQueue.EnQue(outLocal);

            outLocal = outQueue.DeQue<DTYPE_Z>();
            AscendC::DataCopy(yGm[pos], outLocal, 1);
            outQueue.FreeTensor(outLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t outer;
    uint32_t reduce;
    uint32_t inner;
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmax op;
    op.Init(x, y, tiling_data.outer, tiling_data.reduce, tiling_data.inner);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline at::Tensor allocate_argmax_output_like_dim(const at::Tensor& x, int64_t dim) {
    int64_t rank = x.dim();
    if (dim < 0) dim += rank;
    TORCH_CHECK(dim >= 0 && dim < rank, "dim out of range");
    std::vector<int64_t> out_sizes;
    out_sizes.reserve(rank - 1);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out_sizes.push_back(x.size(i));
    }
    auto opts = x.options().dtype(at::kLong);
    return at::empty(out_sizes, opts);
}

at::Tensor argmax_over_a_dimension_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    at::Tensor result = allocate_argmax_output_like_dim(self, dim);
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_custom_impl_npu);
