kernel_src="""
#include "kernel_operator.h"

// Summation over channel dimension for 5D tensor [N, C, D, H, W] -> [N, 1, D, H, W]
// Input x is flattened as [C][N*D*H*W]; Output y is [N*D*H*W]
constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose3dMaxMaxSum {
public:
    __aicore__ inline KernelConvTranspose3dMaxMaxSum() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalInner, uint32_t channels, uint32_t tileNum)
    {
        this->totalInner = totalInner;           // N*D*H*W
        this->channels = channels;               // C
        this->tileNum = tileNum;

        this->blockInner = totalInner / AscendC::GetBlockNum();
        this->tileLength = this->blockInner / (tileNum * BUFFER_NUM);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(this->totalInner * this->channels));
        // y has no channel dim; assign per-block region as base
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockInner * AscendC::GetBlockIdx(), this->blockInner);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            ComputeTile(i);
        }
        // Drain remaining outputs if any
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void ComputeTile(int32_t progress)
    {
        // Allocate output local buffer and initialize by copying first channel
        uint32_t baseOffset = AscendC::GetBlockIdx() * this->blockInner + progress * this->tileLength;

        // First channel
        {
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[0 * this->totalInner + baseOffset], this->tileLength);
            inQueueX.EnQue(xLocal);

            AscendC::LocalTensor<DTYPE_X> xLocalDe = inQueueX.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
            // Initialize yLocal with first channel
            AscendC::DataCopy(yLocal, xLocalDe, this->tileLength);
            outQueueY.EnQue(yLocal);
            inQueueX.FreeTensor(xLocalDe);
        }

        // Accumulate remaining channels
        for (uint32_t c = 1; c < this->channels; ++c) {
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[c * this->totalInner + baseOffset], this->tileLength);
            inQueueX.EnQue(xLocal);

            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
            AscendC::LocalTensor<DTYPE_X> xLocalDe = inQueueX.DeQue<DTYPE_X>();
            AscendC::Add(yLocal, yLocal, xLocalDe, this->tileLength);
            outQueueY.EnQue(yLocal);

            inQueueX.FreeTensor(xLocalDe);
        }
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        uint32_t baseOffset = progress * this->tileLength;
        AscendC::DataCopy(yGm[baseOffset], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalInner; // N*D*H*W
    uint32_t channels;   // C
    uint32_t blockInner; // per block inner length
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_max_max_sum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dMaxMaxSum op;
    op.Init(x, y, tiling_data.totalInner, tiling_data.channels, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model keeping ConvTranspose3d and MaxPool3d in PyTorch, and replacing channel-reduction sum
    with a custom AscendC kernel conv_transpose3d_max_max_sum_custom.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.max_pool1 = torch.nn.MaxPool3d(kernel_size=2)
        self.max_pool2 = torch.nn.MaxPool3d(kernel_size=3)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.max_pool1(x)
        x = self.max_pool2(x)
        # Replace torch.sum(x, dim=1, keepdim=True) with custom op
        x = custom_ops_lib.conv_transpose3d_max_max_sum_custom(x)
        return x
