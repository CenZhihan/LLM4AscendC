    int64_t K    = w_shape->GetDim(2);

        return GRAPH_PARAM_INVALID;
    }

    int64_t stride = 1, padding = 0, dilation = 1;
    auto* attrs = context->GetAttrs();
    if (attrs != nullptr) {
        if (attrs->HasAttribute("stride"))   stride   = attrs->GetInt("stride");
        if (attrs->HasAttribute("padding"))  padding  = attrs->GetInt("padding");
        if (attrs->HasAttribute("dilation")) dilation = attrs->GetInt("dilation");

        return GRAPH_PARAM_INVALID;

    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Lout);

    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // simple scalar buffers

class KernelConvTransposed1d {
public:
    __aicore__ inline KernelConvTransposed1d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Lin, uint32_t Lout, uint32_t K,
                                uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t hasBias) {
        this->N = N;
        this->Cin = Cin;
        this->Cout = Cout;
        this->Lin = Lin;
        this->Lout = Lout;
        this->K = K;
        this->stride = stride;
        this->padding = padding;
        this->dilation = dilation;
        this->hasBias = hasBias;

        // One block handles one batch index if possible
        this->nIdx = AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint64_t>(N) * Cin * Lin);
        wGm.SetGlobalBuffer((__gm__ DTYPE_WEIGHT *)w, static_cast<uint64_t>(Cin) * Cout * K);
        if (hasBias) {
            bGm.SetGlobalBuffer((__gm__ DTYPE_BIAS *)b, Cout);
        }
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, static_cast<uint64_t>(N) * Cout * Lout);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, sizeof(DTYPE_WEIGHT));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, sizeof(DTYPE_BIAS));
        pipe.InitBuffer(accQueue, BUFFER_NUM, sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process() {
        if (nIdx >= N) {
            return;
        }

        for (uint32_t co = 0; co < Cout; ++co) {
            for (uint32_t t = 0; t < Lout; ++t) {
                // sumLocal = 0
                AscendC::LocalTensor<DTYPE_Y> sumLocal = accQueue.AllocTensor<DTYPE_Y>();
                AscendC::Duplicate(sumLocal, (DTYPE_Y)0, 1);

                for (uint32_t k = 0; k < K; ++k) {
                    // li_num = t + padding - k * dilation
                    int64_t li_num = static_cast<int64_t>(t) + static_cast<int64_t>(padding) - static_cast<int64_t>(k) * static_cast<int64_t>(dilation);
                    if (li_num < 0) { continue; }
                    if (static_cast<uint64_t>(li_num) > static_cast<uint64_t>(0x7fffffff)) { continue; } // guard
                    if ((li_num % static_cast<int64_t>(stride)) != 0) { continue; }
                    uint32_t li = static_cast<uint32_t>(li_num / static_cast<int64_t>(stride));
                    if (li >= Lin) { continue; }

                    for (uint32_t ci = 0; ci < Cin; ++ci) {
                        // Load x[n, ci, li] (1 element)
                        uint64_t x_off = (((uint64_t)nIdx * Cin + ci) * Lin + li);
                        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                        AscendC::DataCopy(xLocal, xGm[x_off], 1);

                        // Load w[ci, co, k] (1 element)
                        uint64_t w_off = (((uint64_t)ci * Cout + co) * K + k);
                        AscendC::LocalTensor<DTYPE_WEIGHT> wLocal = inQueueW.AllocTensor<DTYPE_WEIGHT>();
                        AscendC::DataCopy(wLocal, wGm[w_off], 1);

                        // tmp = x * w
                        AscendC::LocalTensor<DTYPE_Y> tmpLocal = accQueue.AllocTensor<DTYPE_Y>();
                        AscendC::Mul(tmpLocal, xLocal, wLocal, 1);

                        // sumLocal += tmp
                        AscendC::Add(sumLocal, sumLocal, tmpLocal, 1);

                        accQueue.FreeTensor(tmpLocal);
                        inQueueX.FreeTensor(xLocal);
                        inQueueW.FreeTensor(wLocal);
                    }
                }

                if (hasBias) {
                    // sumLocal += bias[co]
                    AscendC::LocalTensor<DTYPE_BIAS> bLocal = inQueueB.AllocTensor<DTYPE_BIAS>();
                    AscendC::DataCopy(bLocal, bGm[co], 1);
                    AscendC::Add(sumLocal, sumLocal, bLocal, 1);
                    inQueueB.FreeTensor(bLocal);
                }

                // Store y[n, co, t]
                uint64_t y_off = (((uint64_t)nIdx * Cout + co) * Lout + t);
                AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
                AscendC::DataCopy(yLocal, sumLocal, 1);
                AscendC::DataCopy(yGm[y_off], yLocal, 1);
                outQueueY.FreeTensor(yLocal);

                accQueue.FreeTensor(sumLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> accQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_WEIGHT> wGm;
    AscendC::GlobalTensor<DTYPE_BIAS> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, Cin, Cout, Lin, Lout, K;
    uint32_t stride, padding, dilation, hasBias;
    uint32_t nIdx;
};

extern "C" __global__ __aicore__ void conv_transposed_1d_asymmetric_input_square_kernel_padded_strided_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1d op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.Lin, tiling_data.Lout, tiling_data.K,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation,
            tiling_data.hasBias);
    op.Process();
}

static inline int64_t infer_lout(int64_t Lin, int64_t K, int64_t stride, int64_t padding, int64_t dilation) {
    return (Lin - 1) * stride - 2 * padding + dilation * (K - 1) + 1;

at::Tensor conv_transposed_1d_asymmetric_input_square_kernel_padded_strided_dilated_custom_impl_npu(
    const at::Tensor& x, const at::Tensor& weight, c10::optional<at::Tensor> bias_opt,
    int64_t stride, int64_t padding, int64_t dilation) {

    TORCH_CHECK(x.dim() == 3, "x must be 3D [N, Cin, Lin]");
    TORCH_CHECK(weight.dim() == 3, "weight must be 3D [Cin, Cout, K]");
    TORCH_CHECK(x.dtype() == at::kFloat, "Only float32 is supported");
    TORCH_CHECK(weight.dtype() == at::kFloat, "Only float32 weight is supported");

    auto Cin = x.size(1);
    auto Lin = x.size(2);
    auto CinW = weight.size(0);
    auto Cout = weight.size(1);
    auto K = weight.size(2);

    auto Lout = infer_lout(Lin, K, stride, padding, dilation);
    TORCH_CHECK(Lout > 0, "Computed output length must be positive");

    at::Tensor y = at::empty({N, Cout, Lout}, x.options());

    if (bias_opt.has_value()) {
        at::Tensor bias = bias_opt.value();
        TORCH_CHECK(bias.numel() == 0 || (bias.dim() == 1 && bias.size(0) == Cout),
