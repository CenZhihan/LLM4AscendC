kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelKvCachedAttention {
public:
    __aicore__ inline KernelKvCachedAttention() {}

    __aicore__ inline void Init(GM_ADDR query, GM_ADDR key_cache, GM_ADDR value_cache, GM_ADDR output,
                                uint32_t batch, uint32_t qLen, uint32_t kvLen, uint32_t dModel, uint32_t tileD)
    {
        this->batch = batch;
        this->qLen = qLen;
        this->kvLen = kvLen;
        this->dModel = dModel;
        this->tileD = tileD;

        uint32_t bq_index = AscendC::GetBlockIdx();
        uint32_t b = (qLen == 0) ? 0 : (bq_index / qLen);
        uint32_t q = (qLen == 0) ? 0 : (bq_index % qLen);
        if (b >= batch) {
            b = 0;
            q = 0;
        }
        uint64_t q_offset = ((uint64_t)b * qLen + q) * dModel;
        uint64_t kv_base  = (uint64_t)b * kvLen * dModel;
        uint64_t o_offset = q_offset;

        qGm.SetGlobalBuffer((__gm__ float*)query + q_offset, dModel);
        kGm.SetGlobalBuffer((__gm__ float*)key_cache + kv_base, kvLen * dModel);
        vGm.SetGlobalBuffer((__gm__ float*)value_cache + kv_base, kvLen * dModel);
        oGm.SetGlobalBuffer((__gm__ float*)output + o_offset, dModel);

        pipe.InitBuffer(vQueue, BUFFER_NUM, tileD * sizeof(float));
        pipe.InitBuffer(outQueue, BUFFER_NUM, tileD * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Simplified kernel: Aggregate value cache across time dimension for each (b, q) and write to output.
        // This example focuses on demonstrating AscendC tiling and memory movement patterns.
        uint32_t segs = (dModel + tileD - 1) / tileD;

        for (uint32_t s = 0; s < segs; ++s) {
            uint32_t start = s * tileD;
            uint32_t segLen = (start + tileD <= dModel) ? tileD : (dModel - start);

            // Initialize out segment with the first V row segment
            AscendC::LocalTensor<float> vLocal = vQueue.AllocTensor<float>();
            AscendC::DataCopy(vLocal, vGm[(uint64_t)start], segLen);
            vQueue.EnQue(vLocal);

            AscendC::LocalTensor<float> outLocal = outQueue.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpV = vQueue.DeQue<float>();
            // outLocal = tmpV
            AscendC::DataCopy(outLocal, tmpV, segLen);
            vQueue.FreeTensor(tmpV);
            outQueue.EnQue(outLocal);

            // Accumulate remaining V rows
            for (uint32_t t = 1; t < kvLen; ++t) {
                AscendC::LocalTensor<float> vSeg = vQueue.AllocTensor<float>();
                AscendC::DataCopy(vSeg, vGm[(uint64_t)t * dModel + start], segLen);
                vQueue.EnQue(vSeg);

                AscendC::LocalTensor<float> acc = outQueue.DeQue<float>();
                AscendC::LocalTensor<float> addend = vQueue.DeQue<float>();

                AscendC::Add(acc, acc, addend, segLen);

                outQueue.EnQue(acc);
                vQueue.FreeTensor(addend);
            }

            // Write accumulated segment to output
            AscendC::LocalTensor<float> finalOut = outQueue.DeQue<float>();
            AscendC::DataCopy(oGm[start], finalOut, segLen);
            outQueue.FreeTensor(finalOut);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> vQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<float> qGm;
    AscendC::GlobalTensor<float> kGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> oGm;

    uint32_t batch;
    uint32_t qLen;
    uint32_t kvLen;
    uint32_t dModel;
    uint32_t tileD;
};

extern "C" __global__ __aicore__ void kv_cached_attention_inference_custom(GM_ADDR query, GM_ADDR key_cache, GM_ADDR value_cache, GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelKvCachedAttention op;
    op.Init(query, key_cache, value_cache, output,
            tiling_data.batch, tiling_data.qLen, tiling_data.kvLen, tiling_data.dModel, tiling_data.tileD);
    op.Process();
}
