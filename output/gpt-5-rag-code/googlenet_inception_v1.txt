kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConcat4 {
public:
    __aicore__ inline KernelConcat4() {}
    __aicore__ inline void Init(GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR x4, GM_ADDR y,
                                uint32_t n, uint32_t c1, uint32_t c2, uint32_t c3, uint32_t c4,
                                uint32_t h, uint32_t w, uint32_t tileLength)
    {
        this->n = n;
        this->c1 = c1; this->c2 = c2; this->c3 = c3; this->c4 = c4;
        this->h = h; this->w = w;
        this->tileLength = tileLength;

        x1Gm.SetGlobalBuffer((__gm__ float*)x1, n * c1 * h * w);
        x2Gm.SetGlobalBuffer((__gm__ float*)x2, n * c2 * h * w);
        x3Gm.SetGlobalBuffer((__gm__ float*)x3, n * c3 * h * w);
        x4Gm.SetGlobalBuffer((__gm__ float*)x4, n * c4 * h * w);
        yGm.SetGlobalBuffer((__gm__ float*)y,  n * (c1 + c2 + c3 + c4) * h * w);

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t hw = h * w;
        uint32_t s1 = c1 * hw;
        uint32_t s2 = c2 * hw;
        uint32_t s3 = c3 * hw;
        uint32_t s4 = c4 * hw;
        uint32_t so = (c1 + c2 + c3 + c4) * hw;

        for (uint32_t b = 0; b < n; ++b) {
            uint32_t outBase = b * so;
            uint32_t in1Base = b * s1;
            uint32_t in2Base = b * s2;
            uint32_t in3Base = b * s3;
            uint32_t in4Base = b * s4;

            CopySegment(x1Gm, in1Base, yGm, outBase + 0,  s1);
            CopySegment(x2Gm, in2Base, yGm, outBase + s1, s2);
            CopySegment(x3Gm, in3Base, yGm, outBase + s1 + s2, s3);
            CopySegment(x4Gm, in4Base, yGm, outBase + s1 + s2 + s3, s4);
        }
    }

private:
    __aicore__ inline void CopySegment(AscendC::GlobalTensor<float>& src, uint32_t srcOffset,
                                       AscendC::GlobalTensor<float>& dst, uint32_t dstOffset,
                                       uint32_t length)
    {
        uint32_t copied = 0;
        while (copied < length) {
            uint32_t curr = AscendC::min(this->tileLength, length - copied);
            AscendC::LocalTensor<float> buf = inQueue.AllocTensor<float>();
            AscendC::DataCopy(buf, src[srcOffset + copied], curr);
            AscendC::DataCopy(dst[dstOffset + copied], buf, curr);
            inQueue.FreeTensor(buf);
            copied += curr;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::GlobalTensor<float> x1Gm, x2Gm, x3Gm, x4Gm, yGm;
    uint32_t n, c1, c2, c3, c4, h, w;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void googlenet_inception_v1_custom(GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR x4, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConcat4 op;
    op.Init(x1, x2, x3, x4, y,
            tiling_data.n, tiling_data.c1, tiling_data.c2, tiling_data.c3, tiling_data.c4,
            tiling_data.h, tiling_data.w, tiling_data.tileLength);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class InceptionModule(nn.Module):
    def __init__(self, in_channels, out_1x1, reduce_3x3, out_3x3, reduce_5x5, out_5x5, pool_proj):
        super(InceptionModule, self).__init__()
        self.branch1x1 = nn.Conv2d(in_channels, out_1x1, kernel_size=1)
        self.branch3x3 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_3x3, kernel_size=1),
            nn.Conv2d(reduce_3x3, out_3x3, kernel_size=3, padding=1)
        )
        self.branch5x5 = nn.Sequential(
            nn.Conv2d(in_channels, reduce_5x5, kernel_size=1),
            nn.Conv2d(reduce_5x5, out_5x5, kernel_size=5, padding=2)
        )
        self.branch_pool = nn.Sequential(
            nn.MaxPool2d(kernel_size=3, stride=1, padding=1),
            nn.Conv2d(in_channels, pool_proj, kernel_size=1)
        )

    def forward(self, x):
        b1 = self.branch1x1(x)
        b3 = self.branch3x3(x)
        b5 = self.branch5x5(x)
        bp = self.branch_pool(x)
        # Use custom AscendC concat along channel dimension
        return custom_ops_lib.googlenet_inception_v1_custom(b1, b3, b5, bp)

class ModelNew(nn.Module):
    def __init__(self, num_classes=1000):
        super(ModelNew, self).__init__()
        self.conv1 = nn.Conv2d(3, 64, kernel_size=7, stride=2, padding=3)
        self.maxpool1 = nn.MaxPool2d(3, stride=2, padding=1)
        self.conv2 = nn.Conv2d(64, 64, kernel_size=1)
        self.conv3 = nn.Conv2d(64, 192, kernel_size=3, padding=1)
        self.maxpool2 = nn.MaxPool2d(3, stride=2, padding=1)

        self.inception3a = InceptionModule(192, 64, 96, 128, 16, 32, 32)
        self.inception3b = InceptionModule(256, 128, 128, 192, 32, 96, 64)
        self.maxpool3 = nn.MaxPool2d(3, stride=2, padding=1)

        self.inception4a = InceptionModule(480, 192, 96, 208, 16, 48, 64)
        self.inception4b = InceptionModule(512, 160, 112, 224, 24, 64, 64)
        self.inception4c = InceptionModule(512, 128, 128, 256, 24, 64, 64)
        self.inception4d = InceptionModule(512, 112, 144, 288, 32, 64, 64)
        self.inception4e = InceptionModule(528, 256, 160, 320, 32, 128, 128)
        self.maxpool4 = nn.MaxPool2d(3, stride=2, padding=1)

        self.inception5a = InceptionModule(832, 256, 160, 320, 32, 128, 128)
        self.inception5b = InceptionModule(832, 384, 192, 384, 48, 128, 128)

        self.avgpool = nn.AdaptiveAvgPool2d((1, 1))
        self.dropout = nn.Dropout(0.0)
        self.fc = nn.Linear(1024, num_classes)

    def forward(self, x):
        x = self.maxpool1(F.relu(self.conv1(x)))
        x = F.relu(self.conv2(x))
        x = self.maxpool2(F.relu(self.conv3(x)))

        x = self.inception3a(x)
        x = self.inception3b(x)
        x = self.maxpool3(x)

        x = self.inception4a(x)
        x = self.inception4b(x)
        x = self.inception4c(x)
        x = self.inception4d(x)
        x = self.inception4e(x)
        x = self.maxpool4(x)

        x = self.inception5a(x)
        x = self.inception5b(x)

        x = self.avgpool(x)
        x = torch.flatten(x, 1)
        x = self.dropout(x)
        x = self.fc(x)
        return x
'''
