#include "kernel_operator.h"

using uint32 = unsigned int;

class KernelConvTranspose2dBiasAddClampScalingClampDivide {
public:
    __aicore__ inline KernelConvTranspose2dBiasAddClampScalingClampDivide() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR conv_b, GM_ADDR add_b, GM_ADDR y,
                                uint32 n, uint32 c_in, uint32 h_in, uint32 w_in,
                                uint32 c_out, uint32 k_h, uint32 k_w,
                                uint32 stride_h, uint32 stride_w,
                                uint32 pad_h, uint32 pad_w,
                                uint32 out_pad_h, uint32 out_pad_w,
                                uint32 h_out, uint32 w_out,
                                uint32 totalPairs, uint32 pairsPerBlock, float scale) {
        xGm = reinterpret_cast<__gm__ float*>(x);
        wGm = reinterpret_cast<__gm__ float*>(w);
        cbGm = reinterpret_cast<__gm__ float*>(conv_b);
        abGm = reinterpret_cast<__gm__ float*>(add_b);
        yGm = reinterpret_cast<__gm__ float*>(y);

        this->N = n; this->CIN = c_in; this->HIN = h_in; this->WIN = w_in;
        this->COUT = c_out; this->KH = k_h; this->KW = k_w;
        this->SH = stride_h; this->SW = stride_w;
        this->PH = pad_h; this->PW = pad_w;
        this->OPH = out_pad_h; this->OPW = out_pad_w;
        this->HOUT = h_out; this->WOUT = w_out;
        this->totalPairs = totalPairs; this->pairsPerBlock = pairsPerBlock;
        this->scale = scale;
    }

    __aicore__ inline void Process()
    {
        uint32 blockIdx = AscendC::GetBlockIdx();
        uint32 startPair = blockIdx * pairsPerBlock;
        uint32 endPair = startPair + pairsPerBlock;
        if (endPair > totalPairs) endPair = totalPairs;

        for (uint32 p = startPair; p < endPair; ++p) {
            uint32 n = p / COUT;
            uint32 co = p % COUT;

            // zero initialize output plane
            for (uint32 oh = 0; oh < HOUT; ++oh) {
                for (uint32 ow = 0; ow < WOUT; ++ow) {
                    uint64 out_idx = index_out(n, co, oh, ow);
                    yGm[out_idx] = 0.0f;
                }
            }

            // accumulate transposed convolution
            for (uint32 ci = 0; ci < CIN; ++ci) {
                for (uint32 ih = 0; ih < HIN; ++ih) {
                    for (uint32 iw = 0; iw < WIN; ++iw) {
                        float in_val = xGm[index_in(n, ci, ih, iw)];
                        if (in_val == 0.0f) {
                            // optional micro-optimization: skip zeros
                        }
                        for (uint32 kh = 0; kh < KH; ++kh) {
                            int oh = static_cast<int>(ih) * static_cast<int>(SH) - static_cast<int>(PH) + static_cast<int>(kh);
                            if (oh < 0 || oh >= static_cast<int>(HOUT)) continue;
                            for (uint32 kw = 0; kw < KW; ++kw) {
                                int ow = static_cast<int>(iw) * static_cast<int>(SW) - static_cast<int>(PW) + static_cast<int>(kw);
                                if (ow < 0 || ow >= static_cast<int>(WOUT)) continue;
                                float w_val = wGm[index_w(ci, co, kh, kw)];
                                uint64 out_idx = index_out(n, co, (uint32)oh, (uint32)ow);
                                yGm[out_idx] += in_val * w_val;
                            }
                        }
                    }
                }
            }

            float cbias = cbGm ? cbGm[co] : 0.0f;
            float abias = abGm ? abGm[co] : 0.0f; // expects add_bias shape [C_out] or [C_out,1,1] flattened
            float s = scale;

            // post-processing: +bias, clamp -> [0,1], *s, clamp -> [0,1], /s
            for (uint32 oh = 0; oh < HOUT; ++oh) {
                for (uint32 ow = 0; ow < WOUT; ++ow) {
                    uint64 out_idx = index_out(n, co, oh, ow);
                    float v = yGm[out_idx] + cbias + abias;
                    if (v < 0.0f) v = 0.0f;
                    if (v > 1.0f) v = 1.0f;
                    v = v * s;
                    if (v < 0.0f) v = 0.0f;
                    if (v > 1.0f) v = 1.0f;
                    v = v / s;
                    yGm[out_idx] = v;
                }
            }
        }
    }

private:
    __aicore__ inline uint64 index_in(uint32 n, uint32 c, uint32 h, uint32 w) const {
        return (((uint64)n * CIN + c) * HIN + h) * WIN + w;
    }
    __aicore__ inline uint64 index_w(uint32 ci, uint32 co, uint32 kh, uint32 kw) const {
        // weight shape [C_in, C_out, kH, kW]
        return (((uint64)ci * COUT + co) * KH + kh) * KW + kw;
    }
    __aicore__ inline uint64 index_out(uint32 n, uint32 c, uint32 h, uint32 w) const {
        return (((uint64)n * COUT + c) * HOUT + h) * WOUT + w;
    }

private:
    __gm__ float* xGm;
    __gm__ float* wGm;
    __gm__ float* cbGm;
    __gm__ float* abGm;
    __gm__ float* yGm;

    uint32 N, CIN, HIN, WIN;
    uint32 COUT, KH, KW;
    uint32 SH, SW;
    uint32 PH, PW;
    uint32 OPH, OPW;
    uint32 HOUT, WOUT;
    uint32 totalPairs, pairsPerBlock;
    float scale;
};

extern "C" __global__ __aicore__ void conv_transpose2d_bias_add_clamp_scaling_clamp_divide_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR add_bias,
    GM_ADDR stride_h, GM_ADDR stride_w, GM_ADDR pad_h, GM_ADDR pad_w,
    GM_ADDR out_pad_h, GM_ADDR out_pad_w, GM_ADDR scale, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    (void)stride_h; (void)stride_w; (void)pad_h; (void)pad_w; (void)out_pad_h; (void)out_pad_w; (void)workspace; // not used directly in kernel; parameters provided by tiling.

    GET_TILING_DATA(tiling_data, tiling);

    // Read scale scalar from GM (1 element)
    float scale_val = 1.0f;
    if (scale != nullptr) {
        __gm__ float* sPtr = reinterpret_cast<__gm__ float*>(scale);
        scale_val = sPtr[0];
    }

    KernelConvTranspose2dBiasAddClampScalingClampDivide op;
    op.Init(x, weight, conv_bias, add_bias, y,
            tiling_data.n(), tiling_data.c_in(), tiling_data.h_in(), tiling_data.w_in(),
            tiling_data.c_out(), tiling_data.k_h(), tiling_data.k_w(),
            tiling_data.stride_h(), tiling_data.stride_w(),
            tiling_data.pad_h(), tiling_data.pad_w(),
            tiling_data.out_pad_h(), tiling_data.out_pad_w(),
            tiling_data.h_out(), tiling_data.w_out(),
            tiling_data.totalPairs(), tiling_data.pairsPerBlock(), scale_val);
    op.Process();
}
