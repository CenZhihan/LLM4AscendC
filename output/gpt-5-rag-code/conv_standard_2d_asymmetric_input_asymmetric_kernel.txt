kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 1; // not using UB queues in this simple implementation

struct Conv2dParams {
    uint32_t n, cin, hin, win;
    uint32_t cout, kh, kw;
    uint32_t stride_h, stride_w;
    uint32_t pad_t, pad_l, pad_b, pad_r;
    uint32_t dilation_h, dilation_w;
    uint32_t groups;
    uint32_t hout, wout;
    uint32_t totalOutElems;
    uint32_t tileNum;
};

class KernelConv2D {
public:
    __aicore__ inline KernelConv2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, const Conv2dParams& p)
    {
        x_gm_ = reinterpret_cast<__gm__ DTYPE_X*>(x);
        w_gm_ = reinterpret_cast<__gm__ DTYPE_W*>(w);
        y_gm_ = reinterpret_cast<__gm__ DTYPE_Y*>(y);
        params_ = p;

        block_out_elems_ = params_.totalOutElems / AscendC::GetBlockNum();
        block_offset_ = block_out_elems_ * AscendC::GetBlockIdx();

        // Tail handle: last block takes all remaining
        if (AscendC::GetBlockIdx() == AscendC::GetBlockNum() - 1) {
            block_out_elems_ = params_.totalOutElems - block_offset_;
        }
    }

    __aicore__ inline void Process()
    {
        // Naive conv implementation (groups supported)
        for (uint32_t t = 0; t < block_out_elems_; ++t) {
            uint64_t out_linear = block_offset_ + t;

            // De-linearize to N, C_out, H_out, W_out
            uint32_t wout = params_.wout;
            uint32_t hout = params_.hout;
            uint32_t cout = params_.cout;

            uint32_t ow = out_linear % wout;
            uint32_t tmp1 = out_linear / wout;
            uint32_t oh = tmp1 % hout;
            uint32_t tmp2 = tmp1 / hout;
            uint32_t oc = tmp2 % cout;
            uint32_t n  = tmp2 / cout;

            // Identify oc group and channel range
            uint32_t groups = params_.groups;
            uint32_t cout_per_g = cout / groups;
            uint32_t cin_per_g  = params_.cin / groups;
            uint32_t g = oc / cout_per_g;
            uint32_t cin_start = g * cin_per_g;
            uint32_t cin_end   = cin_start + cin_per_g;

            float acc = 0.0f;
            // Iterate over input channels within group and kernel window
            for (uint32_t ic = cin_start; ic < cin_end; ++ic) {
                uint32_t ic_in_g = ic - cin_start; // index inside group for weight
                for (uint32_t kh = 0; kh < params_.kh; ++kh) {
                    int32_t ih = static_cast<int32_t>(oh) * static_cast<int32_t>(params_.stride_h)
                               - static_cast<int32_t>(params_.pad_t)
                               + static_cast<int32_t>(kh) * static_cast<int32_t>(params_.dilation_h);
                    if (ih < 0 || ih >= static_cast<int32_t>(params_.hin)) continue;

                    for (uint32_t kw = 0; kw < params_.kw; ++kw) {
                        int32_t iw = static_cast<int32_t>(ow) * static_cast<int32_t>(params_.stride_w)
                                   - static_cast<int32_t>(params_.pad_l)
                                   + static_cast<int32_t>(kw) * static_cast<int32_t>(params_.dilation_w);
                        if (iw < 0 || iw >= static_cast<int32_t>(params_.win)) continue;

                        // x index: NCHW
                        uint64_t x_idx = (((uint64_t)n * params_.cin + ic) * params_.hin + static_cast<uint32_t>(ih)) * params_.win + static_cast<uint32_t>(iw);

                        // w index: [C_out, C_in/groups, K_h, K_w]
                        uint64_t w_idx = ((((uint64_t)oc) * cin_per_g + ic_in_g) * params_.kh + kh) * params_.kw + kw;

                        acc += static_cast<float>(x_gm_[x_idx]) * static_cast<float>(w_gm_[w_idx]);
                    }
                }
            }

            // y index: NCHW
            uint64_t y_idx = (((uint64_t)n * params_.cout + oc) * params_.hout + oh) * params_.wout + ow;
            y_gm_[y_idx] = static_cast<DTYPE_Y>(acc);
        }
    }

private:
    __gm__ DTYPE_X* x_gm_;
    __gm__ DTYPE_W* w_gm_;
    __gm__ DTYPE_Y* y_gm_;
    Conv2dParams params_;
    uint32_t block_out_elems_;
    uint32_t block_offset_;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA_WITH_STRUCT(ConvStandard2dAsymmetricInputAsymmetricKernelCustomTilingData, t, tiling);

    Conv2dParams p;
    p.n = t.n();
    p.cin = t.cin();
    p.hin = t.hin();
    p.win = t.win();
    p.cout = t.cout();
    p.kh = t.kh();
    p.kw = t.kw();
    p.stride_h = t.stride_h();
    p.stride_w = t.stride_w();
    p.pad_t = t.pad_t();
    p.pad_l = t.pad_l();
    p.pad_b = t.pad_b();
    p.pad_r = t.pad_r();
    p.dilation_h = t.dilation_h();
    p.dilation_w = t.dilation_w();
    p.groups = t.groups();
    p.hout = t.hout();
    p.wout = t.wout();
    p.totalOutElems = t.totalOutElems();
    p.tileNum = t.tileNum();

    KernelConv2D op;
    op.Init(x, w, y, p);
    op.Process();
}

static inline int64_t calc_out_dim(int64_t in, int64_t pad0, int64_t pad1, int64_t k, int64_t d, int64_t s) {
    int64_t eff_k = d * (k - 1) + 1;
    return (in + pad0 + pad1 - eff_k) / s + 1;
}

at::Tensor conv_standard_2d_asymmetric_input_asymmetric_kernel_custom_impl_npu(
    const at::Tensor& x,
    const at::Tensor& w,
    int64_t stride_h, int64_t stride_w,

    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple,
                 stride: tuple = (1, 1), padding: tuple = (0, 0),
                 dilation: tuple = (1, 1), groups: int = 1, bias: bool = False):
        super().__init__()
        kh, kw = kernel_size
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = (kh, kw)
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.groups = groups
        self.bias_flag = bias

        assert in_channels % groups == 0 and out_channels % groups == 0, "Channels must be divisible by groups"

        weight_shape = (out_channels, in_channels // groups, kh, kw)

        torch.nn.init.kaiming_uniform_(self.weight, a=math.sqrt(5))

            fan_in = in_channels * kh * kw // groups
