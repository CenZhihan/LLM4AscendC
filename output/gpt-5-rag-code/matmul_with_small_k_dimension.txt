    return static_cast<uint32_t>((a + b - 1) / b);
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t DEFAULT_TILE_M = 128;
const uint32_t DEFAULT_TILE_N = 128;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MatmulWithSmallKDimensionCustomTilingData tiling;

    const auto* a_shape = context->GetInputShape(0);
    const auto* b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_PARAM_INVALID;
    }
    uint64_t M = a_shape->GetOriginShape().GetDim(0);
    uint64_t K = a_shape->GetOriginShape().GetDim(1);
    uint64_t Kb = b_shape->GetOriginShape().GetDim(0);
    uint64_t N = b_shape->GetOriginShape().GetDim(1);
    if (K != Kb) {
        return ge::GRAPH_PARAM_INVALID;
    }

    // Set block dim to parallelize rows
    context->SetBlockDim(BLOCK_DIM);

    // Choose tiles; small K so we tile only M and N
    uint32_t tileM = DEFAULT_TILE_M;
    uint32_t tileN = DEFAULT_TILE_N;

    tiling.set_M(static_cast<uint32_t>(M));
    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_K(static_cast<uint32_t>(K));
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);

    // save tiling
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // no workspace
    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE_A = half;
using DTYPE_B = half;
using DTYPE_C = float;

constexpr int32_t A_BUFFER_NUM = 1;
constexpr int32_t B_BUFFER_NUM = 1;
constexpr int32_t C_BUFFER_NUM = 1;

class KernelMatmulSmallK {
public:
    __aicore__ inline KernelMatmulSmallK() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileM = tileM;
        this->tileN = tileN;

        // parallelize by rows
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blk = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (M + blocks - 1) / blocks;
        this->rowStart = blk * rowsPerBlock;
        this->rowEnd = AscendC::min((uint32_t)(this->rowStart + rowsPerBlock), M);

        aGm.SetGlobalBuffer((__gm__ DTYPE_A *)A, (uint64_t)M * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)B, (uint64_t)K * N);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C *)C, (uint64_t)M * N);

        // initialize UB buffers
        pipe.InitBuffer(aQueue, A_BUFFER_NUM, (uint64_t)tileM * K * sizeof(DTYPE_A));
        pipe.InitBuffer(bQueue, B_BUFFER_NUM, (uint64_t)K * tileN * sizeof(DTYPE_B));
        pipe.InitBuffer(cQueue, C_BUFFER_NUM, (uint64_t)tileM * tileN * sizeof(DTYPE_C));
        pipe.InitBuffer(tmpVecQueue, C_BUFFER_NUM, (uint64_t)tileN * sizeof(DTYPE_C));
    }

    __aicore__ inline void Process()
    {
        if (rowStart >= rowEnd) {
            return;
        }
        for (uint32_t m0 = rowStart; m0 < rowEnd; m0 += tileM) {
            uint32_t curTM = AscendC::min(tileM, rowEnd - m0);

            // load A tile [curTM, K] (contiguous)
            AscendC::LocalTensor<DTYPE_A> aLocal = aQueue.AllocTensor<DTYPE_A>();
            AscendC::DataCopy(aLocal, aGm[(uint64_t)m0 * K], (uint64_t)curTM * K);
            aQueue.EnQue(aLocal);

            for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
                uint32_t curTN = AscendC::min(tileN, N - n0);

                AscendC::LocalTensor<DTYPE_A> aL = aQueue.DeQue<DTYPE_A>(); // reuse A tile across N tiles

                // load B tile [K, curTN] (row-wise segments)
                AscendC::LocalTensor<DTYPE_B> bLocal = bQueue.AllocTensor<DTYPE_B>();
                for (uint32_t kk = 0; kk < K; ++kk) {
                    AscendC::DataCopy(bLocal[(uint64_t)kk * tileN], bGm[(uint64_t)kk * N + n0], curTN);
                }
                bQueue.EnQue(bLocal);

                // allocate and zero C tile [curTM, curTN]
                AscendC::LocalTensor<DTYPE_C> cLocal = cQueue.AllocTensor<DTYPE_C>();
                AscendC::Zero(cLocal, (uint64_t)curTM * curTN);

                // compute C_tile += A[:, k] * B[k, :]
                AscendC::LocalTensor<DTYPE_B> bL = bQueue.DeQue<DTYPE_B>();
                AscendC::LocalTensor<DTYPE_C> tmpVec = tmpVecQueue.AllocTensor<DTYPE_C>();

                for (uint32_t kk = 0; kk < K; ++kk) {
                    // B row kk -> float vector length curTN
                    // First cast half -> float into tmpVec
                    AscendC::Cast(tmpVec, bL[(uint64_t)kk * tileN], curTN);

                    for (uint32_t i = 0; i < curTM; ++i) {
                        // read scalar a(i,kk) in float
                        DTYPE_C a_val = static_cast<DTYPE_C>(aL[(uint64_t)i * K + kk]);

                        // c_row pointer
                        AscendC::LocalTensor<DTYPE_C> cRow = cLocal[(uint64_t)i * tileN];

                        // tmpRow = tmpVec * a_val
                        AscendC::LocalTensor<DTYPE_C> tmpScaled = tmpVecQueue.AllocTensor<DTYPE_C>();
                        AscendC::Muls(tmpScaled, tmpVec, a_val, curTN);

                        // c_row += tmpScaled
                        AscendC::Add(cRow, cRow, tmpScaled, curTN);

                        tmpVecQueue.FreeTensor(tmpScaled);
                    }
                }

                tmpVecQueue.FreeTensor(tmpVec);
                bQueue.FreeTensor(bL);

                // store C tile by rows
                for (uint32_t i = 0; i < curTM; ++i) {
                    AscendC::DataCopy(cGm[(uint64_t)(m0 + i) * N + n0], cLocal[(uint64_t)i * tileN], curTN);
                }

                cQueue.FreeTensor(cLocal);
                bQueue.FreeTensor(bLocal);
                // A tile kept for reuse across N tiles
                aQueue.EnQue(aL);
            }
            // done with this A tile
            AscendC::LocalTensor<DTYPE_A> aL_free = aQueue.DeQue<DTYPE_A>();
            aQueue.FreeTensor(aL_free);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, A_BUFFER_NUM> aQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, B_BUFFER_NUM> bQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, C_BUFFER_NUM> cQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, C_BUFFER_NUM> tmpVecQueue;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileM;
    uint32_t tileN;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSmallK op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
