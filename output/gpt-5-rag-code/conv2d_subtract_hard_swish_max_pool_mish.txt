    const __gm__ float* X = reinterpret_cast<const __gm__ float*>(x);
    const __gm__ float* W = reinterpret_cast<const __gm__ float*>(weight);
    const __gm__ float* B = reinterpret_cast<const __gm__ float*>(bias);
    __gm__ float* Y = reinterpret_cast<__gm__ float*>(y);

    // Read tiling params
    const int N = static_cast<int>(tiling_data.n);
    const int Cin = static_cast<int>(tiling_data.cIn);
    const int H = static_cast<int>(tiling_data.h);
    const int Wd = static_cast<int>(tiling_data.w);
    const int Cout = static_cast<int>(tiling_data.cOut);
    const int K = static_cast<int>(tiling_data.k);
    const int stride = static_cast<int>(tiling_data.stride);
    const int pad = static_cast<int>(tiling_data.pad);
    const int poolK = static_cast<int>(tiling_data.poolK);
    const int poolS = static_cast<int>(tiling_data.poolS);

    const int Hconv = (H + 2 * pad - K) / stride + 1;
    const int Wconv = (Wd + 2 * pad - K) / stride + 1;
    const int Hpool = (Hconv - poolK) / poolS + 1;
    const int Wpool = (Wconv - poolK) / poolS + 1;

    auto clampf = [](__aicore__ float v, __aicore__ float lo, __aicore__ float hi) -> __aicore__ float {
        return v < lo ? lo : (v > hi ? hi : v);
    };
    auto hswish = [&](__aicore__ float v) -> __aicore__ float {
        float t = v + 3.0f;
        t = clampf(t, 0.0f, 6.0f);
        return v * (t * (1.0f / 6.0f));
    };
    auto mish = [&](__aicore__ float v) -> __aicore__ float {
        float sp = logf(1.0f + expf(v));
        return v * tanhf(sp);
    };

    // Simple block-based split across Cout dimension for parallel blocks
    int blockNum = AscendC::GetBlockNum();
    int blockId = AscendC::GetBlockIdx();
    int coPerBlock = (Cout + blockNum - 1) / blockNum;
    int coStart = blockId * coPerBlock;
    int coEnd = coStart + coPerBlock;
    if (coEnd > Cout) coEnd = Cout;

    // We store final pooled+mish output directly into Y (N, Cout, Hpool, Wpool)
    for (int nIdx = 0; nIdx < N; ++nIdx) {
        for (int co = coStart; co < coEnd; ++co) {
            const float b = B[co];
            for (int ph = 0; ph < Hpool; ++ph) {
                for (int pw = 0; pw < Wpool; ++pw) {
                    float poolMax = -INFINITY;
                    // Iterate over pool window
                    for (int pi = 0; pi < poolK; ++pi) {
                        for (int pj = 0; pj < poolK; ++pj) {
                            int oh = ph * poolS + pi;
                            int ow = pw * poolS + pj;
                            // Compute conv at (oh, ow, co)
                            float acc = 0.0f;
                            for (int ci = 0; ci < Cin; ++ci) {
                                for (int kh = 0; kh < K; ++kh) {
                                    for (int kw = 0; kw < K; ++kw) {
                                        int ih = oh * stride - pad + kh;
                                        int iw = ow * stride - pad + kw;
                                        if (ih >= 0 && ih < H && iw >= 0 && iw < Wd) {
                                            int xIndex = ((nIdx * Cin + ci) * H + ih) * Wd + iw;
                                            int wIndex = (((co * Cin) + ci) * K + kh) * K + kw;
                                            acc += X[xIndex] * W[wIndex];
                                        }
                                    }
                                }
                            }
                            acc += b;

                            // subtract value: subtract[0]
                            float subV = reinterpret_cast<const __gm__ float*>(bias)[0]; // placeholder avoid unused warning
                            (void)subV; // will be overridden below
