kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using float32 = float;

struct Dim3D {
    uint32_t B, Cin, Cout;
    uint32_t Din, Hin, Win;
    uint32_t Kd, Kh, Kw;
    uint32_t Groups;
};

class KernelConv3dGroupNormMean {
public:
    __aicore__ inline KernelConv3dGroupNormMean() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR g, GM_ADDR be, GM_ADDR y,
                                const Dim3D& dims)
    {
        xGm = reinterpret_cast<__gm__ const float32*>(x);
        wGm = reinterpret_cast<__gm__ const float32*>(w);
        bGm = reinterpret_cast<__gm__ const float32*>(b);
        gammaGm = reinterpret_cast<__gm__ const float32*>(g);
        betaGm  = reinterpret_cast<__gm__ const float32*>(be);
        yGm = reinterpret_cast<__gm__ float32*>(y);
        this->dims = dims;
    }

    __aicore__ inline void Process(float32 epsilon)
    {
        uint32_t bIdx = AscendC::GetBlockIdx();
        if (bIdx >= dims.B) {
            return;
        }

        const uint32_t Dout = dims.Din - dims.Kd + 1;
        const uint32_t Hout = dims.Hin - dims.Kh + 1;
        const uint32_t Wout = dims.Win - dims.Kw + 1;
        const uint32_t channelsPerGroup = dims.Cout / dims.Groups;
        const uint32_t groupElem = channelsPerGroup * Dout * Hout * Wout;

        // Accumulators for group statistics
        // Assuming reasonable number of groups (e.g., <= 64). For generality, compute on the fly.
        // We'll use dynamic indexing within fixed-size loop bounds if needed.

        // First pass: accumulate sum and sumsq per group
        // To avoid dynamic arrays, we allocate in registers for a safe upper bound of groups (e.g., 128).
        const uint32_t MAX_G = 256;
        float32 sum_g[MAX_G];
        float32 sumsq_g[MAX_G];

        for (uint32_t g = 0; g < dims.Groups && g < MAX_G; ++g) {
            sum_g[g] = 0.f;
            sumsq_g[g] = 0.f;
        }

        for (uint32_t co = 0; co < dims.Cout; ++co) {
            uint32_t g = co / channelsPerGroup;
            for (uint32_t od = 0; od < Dout; ++od) {
                for (uint32_t oh = 0; oh < Hout; ++oh) {
                    for (uint32_t ow = 0; ow < Wout; ++ow) {
                        float32 acc = 0.f;
                        for (uint32_t ci = 0; ci < dims.Cin; ++ci) {
                            for (uint32_t kd = 0; kd < dims.Kd; ++kd) {
                                for (uint32_t kh = 0; kh < dims.Kh; ++kh) {
                                    for (uint32_t kw = 0; kw < dims.Kw; ++kw) {
                                        uint32_t id = od + kd;
                                        uint32_t ih = oh + kh;
                                        uint32_t iw = ow + kw;
                                        uint64_t x_idx = (((((uint64_t)bIdx * dims.Cin + ci) * dims.Din + id) * dims.Hin + ih) * dims.Win + iw);
                                        uint64_t w_idx = (((((uint64_t)co * dims.Cin + ci) * dims.Kd + kd) * dims.Kh + kh) * dims.Kw + kw);
                                        acc += xGm[x_idx] * wGm[w_idx];
                                    }
                                }
                            }
                        }
                        acc += bGm[co];
                        sum_g[g] += acc;
                        sumsq_g[g] += acc * acc;
                    }
                }
            }
        }

        float32 mean_g[MAX_G];
        float32 var_g[MAX_G];
        for (uint32_t g = 0; g < dims.Groups && g < MAX_G; ++g) {
            float32 m = sum_g[g] / static_cast<float32>(groupElem);
            float32 v = sumsq_g[g] / static_cast<float32>(groupElem) - m * m;
            mean_g[g] = m;
            var_g[g] = v;
        }

        // Second pass: apply group norm (with affine) and compute final mean over all [Cout, Dout, Hout, Wout]
        float32 totalSum = 0.f;
        const float32 denom = static_cast<float32>(dims.Cout) * static_cast<float32>(Dout) * static_cast<float32>(Hout) * static_cast<float32>(Wout);

        for (uint32_t co = 0; co < dims.Cout; ++co) {
            uint32_t g = co / channelsPerGroup;
            float32 gamma = gammaGm[co];
            float32 beta  = betaGm[co];
            float32 inv_std = 1.0f / sqrtf(var_g[g] + epsilon);

            for (uint32_t od = 0; od < Dout; ++od) {
                for (uint32_t oh = 0; oh < Hout; ++oh) {
                    for (uint32_t ow = 0; ow < Wout; ++ow) {
                        float32 acc = 0.f;
                        for (uint32_t ci = 0; ci < dims.Cin; ++ci) {
                            for (uint32_t kd = 0; kd < dims.Kd; ++kd) {
                                for (uint32_t kh = 0; kh < dims.Kh; ++kh) {
                                    for (uint32_t kw = 0; kw < dims.Kw; ++kw) {
                                        uint32_t id = od + kd;
                                        uint32_t ih = oh + kh;
                                        uint32_t iw = ow + kw;
                                        uint64_t x_idx = (((((uint64_t)bIdx * dims.Cin + ci) * dims.Din + id) * dims.Hin + ih) * dims.Win + iw);
                                        uint64_t w_idx = (((((uint64_t)co * dims.Cin + ci) * dims.Kd + kd) * dims.Kh + kh) * dims.Kw + kw);
                                        acc += xGm[x_idx] * wGm[w_idx];
                                    }
                                }
                            }
                        }
                        acc += bGm[co];
                        float32 norm = (acc - mean_g[g]) * inv_std;
                        float32 val = norm * gamma + beta;
                        totalSum += val;
                    }
                }
            }
        }

        float32 finalMean = totalSum / denom;
        yGm[bIdx] = finalMean;
    }

private:
    __gm__ const float32* xGm;
    __gm__ const float32* wGm;
    __gm__ const float32* bGm;
    __gm__ const float32* gammaGm;
    __gm__ const float32* betaGm;
    __gm__ float32* yGm;
    Dim3D dims;
};

extern "C" __global__ __aicore__ void conv3d_group_norm_mean_custom(GM_ADDR x,
                                                                     GM_ADDR weight,
                                                                     GM_ADDR bias,
                                                                     GM_ADDR gamma,
                                                                     GM_ADDR beta,
                                                                     GM_ADDR y,
                                                                     GM_ADDR workspace,
                                                                     GM_ADDR tiling,
                                                                     GM_ADDR epsilon)
{
    GET_TILING_DATA(tiling_data, tiling);
    Dim3D dims;
    dims.B = tiling_data.batch;
    dims.Cin = tiling_data.cin;
    dims.Cout = tiling_data.cout;
    dims.Din = tiling_data.din;
    dims.Hin = tiling_data.hin;
    dims.Win = tiling_data.win;
    dims.Kd = tiling_data.kd;
    dims.Kh = tiling_data.kh;
    dims.Kw = tiling_data.kw;
    dims.Groups = tiling_data.groups;

    // epsilon is a scalar tensor at address epsilon
    __gm__ const float32* epsGm = reinterpret_cast<__gm__ const float32*>(epsilon);
    float32 eps_val = *epsGm;

    KernelConv3dGroupNormMean op;
    op.Init(x, weight, bias, gamma, beta, y, dims);
    op.Process(eps_val);
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, num_groups):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_channels)

    def forward(self, x):
        eps = torch.tensor(self.group_norm.eps, dtype=x.dtype, device=x.device)
        # Call fused custom op: conv3d -> groupnorm -> mean over [C, D, H, W]
        return custom_ops_lib.conv3d_group_norm_mean_custom(
            x, self.conv.weight, self.conv.bias, self.group_norm.weight, self.group_norm.bias, eps
        )
