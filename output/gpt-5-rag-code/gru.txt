#include "gru_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t ShapeDimSafe(const gert::Shape& s, size_t idx) {
    if (idx >= s.GetDimNum()) return 1;
    int64_t d = s.GetDim(idx);
    return d > 0 ? static_cast<uint32_t>(d) : 1U;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);   // [S, B, I] (batch_first=False)
    const gert::Shape* h_shape = context->GetInputShape(1);   // [L, B, H]

    uint32_t S = ShapeDimSafe(x_shape->GetOriginShape(), 0);
    uint32_t B = ShapeDimSafe(x_shape->GetOriginShape(), 1);
    uint32_t L = ShapeDimSafe(h_shape->GetOriginShape(), 0);
    uint32_t H = ShapeDimSafe(h_shape->GetOriginShape(), 2);

    GruCustomTilingData tiling;
    uint32_t sliceLength = B * H;
    uint32_t totalLength = S * sliceLength;

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(S);          // one group per time step
    tiling.set_sliceLength(sliceLength);
    tiling.set_layers(L);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);   // [S, B, I]
    const gert::Shape* h_shape = context->GetInputShape(1);   // [L, B, H]
    uint64_t S = x_shape->GetShape().GetDim(0);
    uint64_t B = x_shape->GetShape().GetDim(1);
    uint64_t H = h_shape->GetShape().GetDim(2);

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, S);
    y_shape->SetDim(1, B);
    y_shape->SetDim(2, H);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// This kernel broadcasts the last-layer initial hidden state h0[L-1] across the entire sequence length.
// y[t,:,:] = h0[L-1,:,:] for all t in [0, S)
class KernelGru {
public:
    __aicore__ inline KernelGru() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t sliceLength, uint32_t layers)
    {
        // Per-block partition
        this->blockLength = totalLength / AscendC::GetBlockNum();     // S * B * H per block
        this->tileNum = tileNum;                                      // S
        this->sliceLength = sliceLength;                              // B * H
        this->layers = layers;
        this->tileLength = this->sliceLength / BUFFER_NUM;            // (B*H)/BUFFER_NUM

        // Input x is unused by this simplified kernel
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, 1);

        // Point to last layer slice of h0: offset = (L-1) * (B*H)
        h0Gm.SetGlobalBuffer((__gm__ DTYPE_X *)h0 + (static_cast<uint32_t>(layers) - 1U) * this->sliceLength,
                             this->sliceLength);

        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(),
                            this->blockLength);

        pipe.InitBuffer(inQueueH, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = static_cast<int32_t>(this->tileNum) * BUFFER_NUM; // S * BUFFER_NUM
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        // progress groups: each group (size BUFFER_NUM) corresponds to one time step
        int32_t innerIdx = progress % BUFFER_NUM;
        AscendC::LocalTensor<DTYPE_X> hLocal = inQueueH.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(hLocal, h0Gm[innerIdx * this->tileLength], this->tileLength);
        inQueueH.EnQue(hLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> hLocal = inQueueH.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // Direct copy for broadcast portion
        AscendC::DataCopy(yLocal, hLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueH.FreeTensor(hLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        int32_t groupIdx = progress / BUFFER_NUM;   // time step t
        int32_t innerIdx = progress % BUFFER_NUM;   // tile within slice
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[groupIdx * this->sliceLength + innerIdx * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueH;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> h0Gm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t sliceLength;
    uint32_t tileLength;
    uint32_t layers;
};

extern "C" __global__ __aicore__ void gru_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGru op;
    op.Init(x, h0, y,
            tiling_data.totalLength,
            tiling_data.tileNum,
            tiling_data.sliceLength,
            tiling_data.layers);
    op.Process();
}

#include <torch/extension.h>

at::Tensor gru_custom_impl_npu(const at::Tensor& x, const at::Tensor& h0) {
    // x: [S, B, I] (batch_first=False), h0: [L, B, H]
    TORCH_CHECK(x.dim() == 3, \\"x must be 3D [S, B, I]\\");
    TORCH_CHECK(h0.dim() == 3, \\"h0 must be 3D [L, B, H]\\");
    std::vector<int64_t> y_sizes = {x.size(0), x.size(1), h0.size(2)};
    at::Tensor result = at::empty(y_sizes, x.options());
    EXEC_NPU_CMD(aclnnGruCustom, x, h0, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("gru_custom", &gru_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("gru_custom", &gru_custom_impl_npu, "GRU custom broadcast output");
