namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaxPooling2dCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    // Expect NCHW
    int64_t N = (in_shape->GetOriginShape().GetDimNum() > 0) ? in_shape->GetOriginShape().GetDim(0) : 1;
    int64_t C = (in_shape->GetOriginShape().GetDimNum() > 1) ? in_shape->GetOriginShape().GetDim(1) : 1;
    int64_t H = (in_shape->GetOriginShape().GetDimNum() > 2) ? in_shape->GetOriginShape().GetDim(2) : 1;
    int64_t W = (in_shape->GetOriginShape().GetDimNum() > 3) ? in_shape->GetOriginShape().GetDim(3) : 1;

    // Read attributes; allow int or list<int> for kernel_size/stride/padding
    std::vector<int64_t> ksz = getListIntAttr(context, "kernel_size", {});
    std::vector<int64_t> strd = getListIntAttr(context, "stride", {});
    std::vector<int64_t> pad = getListIntAttr(context, "padding", {});
    if (ksz.empty()) {
        int64_t k = getIntAttr(context, "kernel_size", 2);
        ksz = {k, k};
    } else if (ksz.size() == 1) {
        ksz = {ksz[0], ksz[0]};
    }
    if (strd.empty()) {
        int64_t s = getIntAttr(context, "stride", 1);
        strd = {s, s};
    } else if (strd.size() == 1) {
        strd = {strd[0], strd[0]};
    }
    if (pad.empty()) {
        int64_t p = getIntAttr(context, "padding", 0);
        pad = {p, p};
    } else if (pad.size() == 1) {
        pad = {pad[0], pad[0]};
    }

    int64_t KH = ksz[0], KW = ksz[1];
    int64_t SH = strd[0], SW = strd[1];
    int64_t PH = pad[0], PW = pad[1];

    int64_t OH = (H + 2 * PH - KH) / SH + 1;
    int64_t OW = (W + 2 * PW - KW) / SW + 1;
    OH = std::max<int64_t>(0, OH);
    OW = std::max<int64_t>(0, OW);

    // BlockDim strategy: split over N*C
    const uint32_t kMaxBlocks = 16;
    uint32_t nc = static_cast<uint32_t>(N * C);
    uint32_t blk = std::min(kMaxBlocks, nc == 0 ? 1 : nc);
    context->SetBlockDim(blk);

    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_c(static_cast<uint32_t>(C));
    tiling.set_h(static_cast<uint32_t>(H));
    tiling.set_w(static_cast<uint32_t>(W));
    tiling.set_kh(static_cast<uint32_t>(KH));
    tiling.set_kw(static_cast<uint32_t>(KW));
    tiling.set_sh(static_cast<uint32_t>(SH));
    tiling.set_sw(static_cast<uint32_t>(SW));
    tiling.set_ph(static_cast<uint32_t>(PH));
    tiling.set_pw(static_cast<uint32_t>(PW));
    tiling.set_oh(static_cast<uint32_t>(OH));
    tiling.set_ow(static_cast<uint32_t>(OW));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    const auto& os = in_shape->GetOriginShape();
    int64_t N = os.GetDim(0);
    int64_t C = os.GetDim(1);
    int64_t H = os.GetDim(2);
    int64_t W = os.GetDim(3);

    // Retrieve attributes in InferShape stage
    std::vector<int64_t> ksz, strd, pad;
    context->GetAttrs()->GetListInt("kernel_size", ksz);
    context->GetAttrs()->GetListInt("stride", strd);
    context->GetAttrs()->GetListInt("padding", pad);
    if (ksz.empty()) {
        int64_t k; bool ok;
        std::tie(k, ok) = context->GetAttrs()->GetInt("kernel_size");
        ksz = { ok ? k : 2, ok ? k : 2 };
    } else if (ksz.size() == 1) {
        ksz = {ksz[0], ksz[0]};
    }
    if (strd.empty()) {
        int64_t s; bool ok;
        std::tie(s, ok) = context->GetAttrs()->GetInt("stride");
        strd = { ok ? s : 1, ok ? s : 1 };
    } else if (strd.size() == 1) {
        strd = {strd[0], strd[0]};
    }
    if (pad.empty()) {
        int64_t p; bool ok;
        std::tie(p, ok) = context->GetAttrs()->GetInt("padding");
        pad = { ok ? p : 0, ok ? p : 0 };
    } else if (pad.size() == 1) {
        pad = {pad[0], pad[0]};
    }

    int64_t KH = ksz[0], KW = ksz[1];
    int64_t SH = strd[0], SW = strd[1];
    int64_t PH = pad[0], PW = pad[1];

    int64_t OH = (H + 2 * PH - KH) / SH + 1;
    int64_t OW = (W + 2 * PW - KW) / SW + 1;
    OH = std::max<int64_t>(0, OH);
    OW = std::max<int64_t>(0, OW);

    gert::Shape* out_shape = context->GetOutputShape(0);
    out_shape->SetOriginShape({N, C, OH, OW});
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

// A simple, correctness-first max pooling kernel for NCHW tensors.
// It uses scalar local buffers and elementwise Maximum to compute window maxima.
// For high performance, further tiling/vectorization should be added.

extern \"C\" __global__ __aicore__ void max_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    // Tiling data must match host side MaxPooling2dCustomTilingData
    GET_TILING_DATA(tiling_data, tiling);

    using DTYPE = float;

    const uint32_t N = tiling_data.n();
    const uint32_t C = tiling_data.c();
    const uint32_t H = tiling_data.h();
    const uint32_t W = tiling_data.w();
    const uint32_t KH = tiling_data.kh();
    const uint32_t KW = tiling_data.kw();
    const uint32_t SH = tiling_data.sh();
    const uint32_t SW = tiling_data.sw();
    const uint32_t PH = tiling_data.ph();
    const uint32_t PW = tiling_data.pw();
    const uint32_t OH = tiling_data.oh();
    const uint32_t OW = tiling_data.ow();

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;
    xGm.SetGlobalBuffer((__gm__ DTYPE *)x, (uint32_t)(N * C * H * W));
    yGm.SetGlobalBuffer((__gm__ DTYPE *)y, (uint32_t)(N * C * OH * OW));

    // Split work along N*C across blocks
    const uint32_t ncTotal = N * C;
    const uint32_t blocks = AscendC::GetBlockNum();
    const uint32_t blkIdx = AscendC::GetBlockIdx();
    uint32_t ncPerBlk = (ncTotal + blocks - 1) / blocks;
    uint32_t ncStart = blkIdx * ncPerBlk;
    uint32_t ncEnd = ncStart + ncPerBlk;
    if (ncEnd > ncTotal) ncEnd = ncTotal;

    AscendC::TPipe pipe;
    // Two 1-element local buffers: current value and running max
    constexpr int32_t BUFFER_NUM = 2;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpInQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpOutQ;
    pipe.InitBuffer(tmpInQ, BUFFER_NUM, sizeof(DTYPE));
    pipe.InitBuffer(tmpOutQ, BUFFER_NUM, sizeof(DTYPE));

    for (uint32_t nc = ncStart; nc < ncEnd; ++nc) {
        const uint32_t n = nc / C;
        const uint32_t c = nc % C;
        const uint32_t in_base = ((n * C + c) * H) * W;
        const uint32_t out_base = ((n * C + c) * OH) * OW;

        for (uint32_t oh = 0; oh < OH; ++oh) {
            int32_t hstart = (int32_t)oh * (int32_t)SH - (int32_t)PH;
            int32_t hend = hstart + (int32_t)KH;
            if (hstart < 0) hstart = 0;
            if (hend > (int32_t)H) hend = (int32_t)H;

            for (uint32_t ow = 0; ow < OW; ++ow) {
                int32_t wstart = (int32_t)ow * (int32_t)SW - (int32_t)PW;
                int32_t wend = wstart + (int32_t)KW;
                if (wstart < 0) wstart = 0;
                if (wend > (int32_t)W) wend = (int32_t)W;

                bool first = true;
                // allocate local tensors for running max and current scalar
                AscendC::LocalTensor<DTYPE> maxLocal = tmpOutQ.AllocTensor<DTYPE>();
                AscendC::LocalTensor<DTYPE> curLocal = tmpInQ.AllocTensor<DTYPE>();

                for (int32_t ih = hstart; ih < hend; ++ih) {
                    for (int32_t iw = wstart; iw < wend; ++iw) {
                        uint32_t in_off = in_base + (uint32_t)ih * W + (uint32_t)iw;
                        AscendC::DataCopy(curLocal, xGm[in_off], 1);  // copy one element
                        if (first) {
                            AscendC::DataCopy(maxLocal, curLocal, 1);
                            first = false;
                        } else {
                            AscendC::Maximum(maxLocal, maxLocal, curLocal, 1);
                        }
                    }
                }

                uint32_t out_off = out_base + oh * OW + ow;
                AscendC::DataCopy(yGm[out_off], maxLocal, 1);

                tmpInQ.FreeTensor(curLocal);
                tmpOutQ.FreeTensor(maxLocal);
            }
        }
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor max_pooling_2d_custom_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding) {
    std::vector<int64_t> ksz = {kernel_size, kernel_size};
    std::vector<int64_t> strd = {stride, stride};
    std::vector<int64_t> pad = {padding, padding};
    // Infer output size on host to allocate tensor
    TORCH_CHECK(self.dim() == 4, "Input must be NCHW");
    auto N = self.size(0);
    auto C = self.size(1);
    auto H = self.size(2);
    auto W = self.size(3);
    int64_t OH = (H + 2 * padding - kernel_size) / stride + 1;
    int64_t OW = (W + 2 * padding - kernel_size) / stride + 1;
    OH = std::max<int64_t>(0, OH);
    OW = std::max<int64_t>(0, OW);
    at::Tensor output = at::empty({N, C, OH, OW}, self.options());
    EXEC_NPU_CMD(aclnnMaxPooling2dCustom, self, ksz, strd, pad, output);
    return output;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_2d_custom", &max_pooling_2d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_2d_custom", &max_pooling_2d_custom_impl_npu, "Max Pool2D Custom");
}
