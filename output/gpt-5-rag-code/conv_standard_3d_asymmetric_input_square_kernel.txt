static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    auto *xShape = context->GetInputShape(0);
    auto *wShape = context->GetInputShape(1);

    // Expect x: [N, C_in, D, H, W], w: [C_out, C_in, 1, K, K]
    uint32_t N = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t Cin = static_cast<uint32_t>(xShape->GetDim(1));
    uint32_t D = static_cast<uint32_t>(xShape->GetDim(2));
    uint32_t H = static_cast<uint32_t>(xShape->GetDim(3));
    uint32_t W = static_cast<uint32_t>(xShape->GetDim(4));

    uint32_t Cout = static_cast<uint32_t>(wShape->GetDim(0));
    uint32_t Kd = static_cast<uint32_t>(wShape->GetDim(2)); // should be 1
    uint32_t Kh = static_cast<uint32_t>(wShape->GetDim(3));
    uint32_t Kw = static_cast<uint32_t>(wShape->GetDim(4));
    (void)Kd; // kd is 1 by design
    uint32_t K = Kh; // square kernel Kh == Kw

    // stride=1, padding=0, dilation=1
    uint32_t Dout = D;
    uint32_t Hout = (H >= K) ? (H - K + 1) : 0;
    uint32_t Wout = (W >= K) ? (W - K + 1) : 0;

    ConvStandard3dAsymmetricInputSquareKernelCustomTilingData tiling;
    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_D(D);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_K(K);
    tiling.set_Dout(Dout);
    tiling.set_Hout(Hout);
    tiling.set_Wout(Wout);
    uint64_t totalOutput = static_cast<uint64_t>(N) * Cout * Dout * Hout * Wout;
    tiling.set_totalOutput(totalOutput);

    context->SetBlockDim(BLOCK_DIM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""
#include "kernel_operator.h"

class KernelConv3D {
public:
    __aicore__ inline KernelConv3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W, uint32_t K,
                                uint32_t Dout, uint32_t Hout, uint32_t Wout,
                                uint64_t totalOutput)
    {
        xGm = reinterpret_cast<__gm__ DTYPE_X*>(x);
        wGm = reinterpret_cast<__gm__ DTYPE_W*>(w);
        yGm = reinterpret_cast<__gm__ DTYPE_Y*>(y);

        this->N = N; this->Cin = Cin; this->Cout = Cout;
        this->D = D; this->H = H; this->W = W; this->K = K;
        this->Dout = Dout; this->Hout = Hout; this->Wout = Wout;
        this->totalOutput = totalOutput;

        // compute per-block range
        uint64_t blocks = AscendC::GetBlockNum();
        uint64_t base = (totalOutput + blocks - 1) / blocks;
        this->start = base * AscendC::GetBlockIdx();
        this->end = this->start + base;
        if (this->end > totalOutput) this->end = totalOutput;
    }

    __aicore__ inline void Process()
    {
        // linear index over output: [n, oc, d, h, w]
        for (uint64_t idx = start; idx < end; ++idx) {
            uint64_t rem = idx;
            uint32_t w_out = static_cast<uint32_t>(rem % Wout); rem /= Wout;
            uint32_t h_out = static_cast<uint32_t>(rem % Hout); rem /= Hout;
            uint32_t d_out = static_cast<uint32_t>(rem % Dout); rem /= Dout;
            uint32_t oc = static_cast<uint32_t>(rem % Cout); rem /= Cout;
            uint32_t n = static_cast<uint32_t>(rem);

            DTYPE_Y acc = static_cast<DTYPE_Y>(0);

            // For stride=1, pad=0, dilation=1, kd=1
            for (uint32_t ic = 0; ic < Cin; ++ic) {
                for (uint32_t kh = 0; kh < K; ++kh) {
                    for (uint32_t kw = 0; kw < K; ++kw) {
                        uint32_t ih = h_out + kh;
                        uint32_t iw = w_out + kw;
                        uint32_t id = d_out;

                        uint64_t x_index = ((((uint64_t)n * Cin + ic) * D + id) * H + ih) * W + iw;
                        uint64_t w_index = (((((uint64_t)oc * Cin + ic) * 1 + 0) * K + kh) * K + kw);

                        acc += static_cast<DTYPE_Y>(xGm[x_index]) * static_cast<DTYPE_Y>(wGm[w_index]);
                    }
                }
            }

            uint64_t y_index = ((((uint64_t)n * Cout + oc) * Dout + d_out) * Hout + h_out) * Wout + w_out;
            yGm[y_index] = acc;
        }
    }

private:
    __gm__ DTYPE_X* xGm;
    __gm__ DTYPE_W* wGm;
    __gm__ DTYPE_Y* yGm;

    uint32_t N, Cin, Cout;
    uint32_t D, H, W, K;
    uint32_t Dout, Hout, Wout;
    uint64_t totalOutput;
    uint64_t start, end;
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workSpace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3D op;
    op.Init(x, w, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.D, tiling_data.H, tiling_data.W, tiling_data.K,
            tiling_data.Dout, tiling_data.Hout, tiling_data.Wout,
            tiling_data.totalOutput);
    op.Process();
}

    TORCH_CHECK(w.dim() == 5, "w must be 5D (Cout, Cin, 1, K, K)");
    int64_t N = x.size(0);

    int64_t W = x.size(4);
    int64_t Cout = w.size(0);
    int64_t K = w.size(3);

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom 3D convolution using AscendC kernel for (KxKx1) kernels, stride=1, padding=0, dilation=1, groups=1, no bias.
    Input expected in NCDHW layout.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super().__init__()
        assert stride == 1 and padding == 0 and dilation == 1 and groups == 1 and bias is False, "Custom op supports only stride=1, padding=0, dilation=1, groups=1, bias=False"
        # weight: [Cout, Cin, 1, K, K]
        self.weight = torch.nn.Parameter(torch.randn(out_channels, in_channels, 1, kernel_size, kernel_size))

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_standard_3d_asymmetric_input_square_kernel_custom(x, self.weight)
