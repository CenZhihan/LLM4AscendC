kernel_src="""
#include "kernel_operator.h"

class KernelConvTranspose2D {
public:
    __aicore__ inline KernelConvTranspose2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t w_in, uint32_t kh, uint32_t kw,
                                uint32_t ho, uint32_t wo, uint64_t totalWork)
    {
        xGm = reinterpret_cast<__gm__ DTYPE_X*>(x);
        wGm = reinterpret_cast<__gm__ DTYPE_Y*>(w);
        yGm = reinterpret_cast<__gm__ DTYPE_Z*>(y);

        this->N = n;
        this->Cin = cin;
        this->Cout = cout;
        this->H = h;
        this->W = w_in;
        this->Kh = kh;
        this->Kw = kw;
        this->Ho = ho;
        this->Wo = wo;
        this->Total = totalWork;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        uint64_t base = (totalWork / blocks) * idx;
        uint64_t tail = (idx == (blocks - 1)) ? (totalWork - (totalWork / blocks) * (blocks - 1))
                                              : (totalWork / blocks);
        this->start = base;
        this->work = tail;
    }

    __aicore__ inline void Process()
    {
        for (uint64_t t = 0; t < work; ++t) {
            uint64_t outIndex = start + t;

            // Unravel index into (n, co, ho, wo)
            uint64_t tmp = outIndex;
            uint32_t wo = static_cast<uint32_t>(tmp % Wo); tmp /= Wo;
            uint32_t ho = static_cast<uint32_t>(tmp % Ho); tmp /= Ho;
            uint32_t co = static_cast<uint32_t>(tmp % Cout); tmp /= Cout;
            uint32_t n = static_cast<uint32_t>(tmp);

            float acc = 0.0f;

            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t r = 0; r < Kh; ++r) {
                    int32_t ih = static_cast<int32_t>(ho) - static_cast<int32_t>(r);
                    if (ih < 0 || ih >= static_cast<int32_t>(H)) {
                        continue;
                    }
                    for (uint32_t s = 0; s < Kw; ++s) {
                        int32_t iw = static_cast<int32_t>(wo) - static_cast<int32_t>(s);
                        if (iw < 0 || iw >= static_cast<int32_t>(W)) {
                            continue;
                        }
                        uint64_t xIdx = (((static_cast<uint64_t>(n) * Cin + ci) * H + static_cast<uint32_t>(ih)) * W + static_cast<uint32_t>(iw));
                        uint64_t wIdx = (((static_cast<uint64_t>(ci) * Cout + co) * Kh + r) * Kw + s);
                        acc += static_cast<float>(xGm[xIdx]) * static_cast<float>(wGm[wIdx]);
                    }
                }
            }

            uint64_t yIdx = (((static_cast<uint64_t>(n) * Cout + co) * Ho + ho) * Wo + wo);
            yGm[yIdx] = static_cast<DTYPE_Z>(acc);
        }
    }

private:
    __gm__ DTYPE_X* xGm;
    __gm__ DTYPE_Y* wGm;
    __gm__ DTYPE_Z* yGm;

    uint32_t N, Cin, Cout, H, W, Kh, Kw, Ho, Wo;
    uint64_t Total;
    uint64_t start;
    uint64_t work;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose2D op;
    op.Init(x, weight, y,
            tiling_data.n(), tiling_data.cin(), tiling_data.cout(),
            tiling_data.h(), tiling_data.w(), tiling_data.kh(), tiling_data.kw(),
            tiling_data.ho(), tiling_data.wo(), tiling_data.totalWork());
    op.Process();
}

    int64_t Wo = W + Kw - 1;
