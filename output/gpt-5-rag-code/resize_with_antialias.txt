host_operator_src="""
#include "resize_with_antialias_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
constexpr uint32_t BLOCK_DIM = 8;
constexpr uint32_t OUT_H = 128;
constexpr uint32_t OUT_W = 128;
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const auto* inShape = context->GetInputShape(0);
    const auto& shape = inShape->GetOriginShape();
    uint32_t n = (shape.GetDimNum() >= 1) ? static_cast<uint32_t>(shape.GetDim(0)) : 1;
    uint32_t c = (shape.GetDimNum() >= 2) ? static_cast<uint32_t>(shape.GetDim(1)) : 1;
    uint32_t in_h = (shape.GetDimNum() >= 3) ? static_cast<uint32_t>(shape.GetDim(2)) : 1;
    uint32_t in_w = (shape.GetDimNum() >= 4) ? static_cast<uint32_t>(shape.GetDim(3)) : 1;

    ResizeWithAntialiasCustomTilingData tiling;
    uint32_t out_h = OUT_H;
    uint32_t out_w = OUT_W;

    uint32_t total_rows = n * c * out_h;
    uint32_t rows_per_block = (total_rows + BLOCK_DIM - 1) / BLOCK_DIM;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_in_h(in_h);
    tiling.set_in_w(in_w);
    tiling.set_out_h(out_h);
    tiling.set_out_w(out_w);
    tiling.set_total_rows(total_rows);
    tiling.set_rows_per_block(rows_per_block);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    gert::Shape out;
    out.SetDimNum(4);
    out.SetDim(0, x_shape->GetDim(0)); // N
    out.SetDim(1, x_shape->GetDim(1)); // C
    out.SetDim(2, 128); // H_out
    out.SetDim(3, 128); // W_out
    *y_shape = out;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class ResizeWithAntialiasCustom : public OpDef {
public:
    explicit ResizeWithAntialiasCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("z")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(ResizeWithAntialiasCustom);
}

kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE_X;
typedef float DTYPE_Z;

extern "C" __global__ __aicore__ void resize_with_antialias_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.n;
    const uint32_t C = tiling_data.c;
    const uint32_t inH = tiling_data.in_h;
    const uint32_t inW = tiling_data.in_w;
    const uint32_t outH = tiling_data.out_h;
    const uint32_t outW = tiling_data.out_w;
    const uint32_t totalRows = tiling_data.total_rows;
    const uint32_t rowsPerBlock = tiling_data.rows_per_block;

    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t startRow = blockIdx * rowsPerBlock;
    uint32_t endRow = startRow + rowsPerBlock;
    if (endRow > totalRows) endRow = totalRows;

    __gm__ DTYPE_X* x_ptr = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_Z* z_ptr = reinterpret_cast<__gm__ DTYPE_Z*>(z);

    for (uint32_t r = startRow; r < endRow; ++r) {
        uint32_t plane = r / outH;
        uint32_t oh = r - plane * outH;

        uint32_t n_idx = plane / C;
        uint32_t c_idx = plane - n_idx * C;

        uint64_t in_plane_base  = ((uint64_t)n_idx * C + c_idx) * (uint64_t)inH * (uint64_t)inW;
        uint64_t out_plane_base = ((uint64_t)n_idx * C + c_idx) * (uint64_t)outH * (uint64_t)outW;

        uint64_t in_row0_base = in_plane_base + (uint64_t)(2 * oh) * (uint64_t)inW;
        uint64_t in_row1_base = in_plane_base + (uint64_t)(2 * oh + 1) * (uint64_t)inW;
        uint64_t out_row_base = out_plane_base + (uint64_t)oh * (uint64_t)outW;

        for (uint32_t ow = 0; ow < outW; ++ow) {
            uint64_t i00 = in_row0_base + (uint64_t)(2 * ow);
            uint64_t i01 = i00 + 1;
            uint64_t i10 = in_row1_base + (uint64_t)(2 * ow);
            uint64_t i11 = i10 + 1;

            float v00 = x_ptr[i00];
            float v01 = x_ptr[i01];
            float v10 = x_ptr[i10];
            float v11 = x_ptr[i11];

            float sum = v00 + v01 + v10 + v11;
            z_ptr[out_row_base + ow] = sum * 0.25f;
        }
    }
}
"""

at::Tensor resize_with_antialias_custom_impl_npu(const at::Tensor& self) {
    TORCH_CHECK(self.dim() == 4, "Input must be 4D NCHW");
    auto N = self.size(0);
    auto C = self.size(1);

    at::Tensor result = at::empty({N, C, 128, 128}, self.options());
    EXEC_NPU_CMD(aclnnResizeWithAntialiasCustom, self, result);
    return result;
}
