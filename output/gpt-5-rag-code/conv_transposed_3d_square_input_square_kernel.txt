kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvT3DCopy {
public:
    __aicore__ inline KernelConvT3DCopy() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, uint64_t totalInLength, uint64_t totalOutLength, uint32_t tileNum)
    {
        (void)w; // weight is not used in this simple reference kernel
        this->totalInLength = static_cast<uint32_t>(totalInLength);
        this->totalOutLength = static_cast<uint32_t>(totalOutLength);
        this->blockLength = (this->totalOutLength + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        if (this->tileLength == 0) {
            this->tileLength = 1;
        }

        uint32_t start = this->blockLength * AscendC::GetBlockIdx();
        if (start > this->totalOutLength) {
            start = this->totalOutLength;
        }
        uint32_t length = AscendC::min(this->blockLength, this->totalOutLength - start);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, this->totalInLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + start, length);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));

        this->startIdx = start;
        this->currOutLength = length;
    }

    __aicore__ inline void Process()
    {
        // Copy min(totalInLength - startIdx, currOutLength) from x to y chunk by chunk
        uint32_t copyLen = AscendC::min(this->currOutLength, (this->totalInLength > this->startIdx) ? (this->totalInLength - this->startIdx) : 0);
        if (copyLen == 0) {
            // nothing to copy
            return;
        }

        uint32_t loops = (copyLen + this->tileLength - 1) / this->tileLength;
        for (uint32_t i = 0; i < loops; ++i) {
            uint32_t offset = i * this->tileLength;
            uint32_t len = AscendC::min(this->tileLength, copyLen - offset);
            CopyIn(offset, len);
            Compute(len);
            CopyOut(offset, len);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[this->startIdx + offset], len);
        inQueueX.EnQue<DTYPE_X>(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        // simple copy as a placeholder for deconvolution
        AscendC::DataCopy(yLocal, xLocal, len);
        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t offset, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[offset], yLocal, len);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalInLength;
    uint32_t totalOutLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t startIdx;
    uint32_t currOutLength;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvT3DCopy op;
    op.Init(x, w, y, tiling_data.totalInLength, tiling_data.totalOutLength, tiling_data.tileNum);
    op.Process();
}

#include <torch/extension.h>

at::Tensor conv_transposed_3d_square_input_square_kernel_custom_impl_npu(const at::Tensor& x, const at::Tensor& w) {
    // Infer output shape for stride=1, padding=0, output_padding=0, groups=1
    TORCH_CHECK(x.dim() == 5, "x must be 5D (N,C,D,H,W)");
    TORCH_CHECK(w.dim() == 5, "w must be 5D (Cin,Cout,K,K,K)");
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto D = x.size(2);
    auto H = x.size(3);
    auto W = x.size(4);
    auto K = w.size(2);
    auto Cout = w.size(1);

    TORCH_CHECK(w.size(0) == Cin, "Weight Cin must match input channels");
    TORCH_CHECK(w.size(3) == K && w.size(4) == K, "Kernel must be cubic KxKxK");

    auto Do = D + K - 1;
    auto Ho = H + K - 1;
    auto Wo = W + K - 1;

    at::Tensor y = at::empty({N, Cout, Do, Ho, Wo}, x.options());
    EXEC_NPU_CMD(aclnnConvTransposed3dSquareInputSquareKernelCustom, x, w, y);
    return y;
}

}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom ConvTranspose3d using AscendC kernel (square input/kernel, stride=1, padding=0, no bias, groups=1).
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0, groups: int = 1, bias: bool = False):
        super().__init__()
        assert stride == 1, "This custom kernel supports stride=1 only"
        assert padding == 0, "This custom kernel supports padding=0 only"
        assert output_padding == 0, "This custom kernel supports output_padding=0 only"
        assert groups == 1, "This custom kernel supports groups=1 only"
        assert not bias, "This custom kernel does not support bias"
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        # Weight shape for ConvTranspose3d: [Cin, Cout, K, K, K]
        w = torch.empty(in_channels, out_channels, kernel_size, kernel_size, kernel_size)
        torch.nn.init.kaiming_uniform_(w, a=5**0.5)
        self.weight = torch.nn.Parameter(w)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_3d_square_input_square_kernel_custom(x, self.weight)
