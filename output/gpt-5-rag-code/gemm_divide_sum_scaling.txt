kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE_X;
typedef float DTYPE_W;
typedef float DTYPE_A;
typedef float DTYPE_Y;

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmDivideSumScaling {
public:
    __aicore__ inline KernelGemmDivideSumScaling() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR alpha, GM_ADDR y,
                                uint32_t batchSize, uint32_t inputSize, uint32_t hiddenSize,
                                uint32_t tileNumCols, uint32_t tileNumRows, uint32_t rowsPerBlock)
    {
        this->batchSize = batchSize;
        this->inputSize = inputSize;
        this->hiddenSize = hiddenSize;

        this->rowsPerBlock = rowsPerBlock;
        this->tileNumCols = tileNumCols;
        this->tileNumRows = tileNumRows;

        // Each block processes a contiguous range of rows in x and y
        uint32_t startRow = AscendC::GetBlockIdx() * this->rowsPerBlock;
        uint32_t endRow = AscendC::Min(startRow + this->rowsPerBlock, this->batchSize);
        this->procRows = endRow - startRow;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + startRow * this->inputSize, this->procRows * this->inputSize);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, this->hiddenSize * this->inputSize);
        aGm.SetGlobalBuffer((__gm__ DTYPE_A *)alpha, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + startRow, this->procRows);

        // Determine tile lengths for column and row tiling
        this->tileColsLen = AscendC::Max<uint32_t>(1, this->inputSize / (this->tileNumCols * BUFFER_NUM));
        this->tileRowsLen = AscendC::Max<uint32_t>(1, this->hiddenSize / this->tileNumRows);

        // Initialize queues for column chunks and temporary buffers
        pipe.InitBuffer(xQueue, BUFFER_NUM, this->tileColsLen * sizeof(DTYPE_X));
        pipe.InitBuffer(wRowQueue, BUFFER_NUM, this->tileColsLen * sizeof(DTYPE_W));
        pipe.InitBuffer(wSumQueue, BUFFER_NUM, this->tileColsLen * sizeof(DTYPE_W));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileColsLen * sizeof(DTYPE_Y));

        // Output accumulation per row
        pipe.InitBuffer(yPartialQueue, 1, this->procRows * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Load alpha and compute combined scaling factor (alpha / 2)
        DTYPE_A alphaVal = 0;
        {
            AscendC::LocalTensor<DTYPE_A> aLocal = tmpQueue.AllocTensor<DTYPE_A>();
            AscendC::DataCopy(aLocal, aGm, 1);
            alphaVal = aLocal.GetValue(0);
            tmpQueue.FreeTensor(aLocal);
        }
        DTYPE_A scaleVal = alphaVal * static_cast<DTYPE_A>(0.5f);

        // Initialize output partial sums for all rows to zero
        AscendC::LocalTensor<DTYPE_Y> yPartial = yPartialQueue.AllocTensor<DTYPE_Y>();
        for (uint32_t r = 0; r < this->procRows; ++r) {
            yPartial.SetValue(r, static_cast<DTYPE_Y>(0));
        }

        // Iterate over column tiles
        uint32_t totalColTiles = (this->inputSize + this->tileColsLen - 1) / this->tileColsLen;
        for (uint32_t t = 0; t < totalColTiles; ++t) {
            uint32_t colStart = t * this->tileColsLen;
            uint32_t colLen = AscendC::Min<uint32_t>(this->tileColsLen, this->inputSize - colStart);

            // Accumulate weight column sums over hidden dimension for current column tile
            AscendC::LocalTensor<DTYPE_W> wSumLocal = wSumQueue.AllocTensor<DTYPE_W>();
            // Initialize with first row chunk
            uint32_t rowStart0 = 0;
            uint32_t firstRowLen = AscendC::Min<uint32_t>(this->tileRowsLen, this->hiddenSize - rowStart0);
            // Copy first row of the first group
            {
                AscendC::LocalTensor<DTYPE_W> wRowLocal = wRowQueue.AllocTensor<DTYPE_W>();
                AscendC::DataCopy(wRowLocal, wGm[rowStart0 * this->inputSize + colStart], colLen);
                AscendC::DataCopy(wSumLocal, wGm[rowStart0 * this->inputSize + colStart], colLen);
                wRowQueue.FreeTensor(wRowLocal);
            }

            // Continue accumulating for remaining rows
            for (uint32_t r = 1; r < this->hiddenSize; ++r) {
                AscendC::LocalTensor<DTYPE_W> wRowLocal = wRowQueue.AllocTensor<DTYPE_W>();
                AscendC::DataCopy(wRowLocal, wGm[r * this->inputSize + colStart], colLen);
                AscendC::Add(wSumLocal, wSumLocal, wRowLocal, colLen);
                wRowQueue.FreeTensor(wRowLocal);
            }

            // For each row in the block, compute partial dot product with x chunk and accumulate
            for (uint32_t br = 0; br < this->procRows; ++br) {
                AscendC::LocalTensor<DTYPE_X> xLocal = xQueue.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> tmpLocal = tmpQueue.AllocTensor<DTYPE_Y>();
                AscendC::DataCopy(xLocal, xGm[br * this->inputSize + colStart], colLen);

                // tmpLocal = xLocal * wSumLocal
                AscendC::Mul(tmpLocal, xLocal, wSumLocal, colLen);

                // Reduce sum tmpLocal into a scalar and add to yPartial[br]
                DTYPE_Y tileSum = AscendC::ReduceSum(tmpLocal, colLen);
                DTYPE_Y prevVal = yPartial.GetValue(br);
                yPartial.SetValue(br, prevVal + tileSum);

                tmpQueue.FreeTensor(tmpLocal);
                xQueue.FreeTensor(xLocal);
            }

            wSumQueue.FreeTensor(wSumLocal);
        }

        // Apply scaling factor and store results
        for (uint32_t br = 0; br < this->procRows; ++br) {
            DTYPE_Y v = yPartial.GetValue(br);
            yPartial.SetValue(br, static_cast<DTYPE_Y>(v * scaleVal));
        }
        AscendC::DataCopy(yGm, yPartial, this->procRows);
        yPartialQueue.FreeTensor(yPartial);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wRowQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wSumQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1>        yPartialQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t batchSize;
    uint32_t inputSize;
    uint32_t hiddenSize;

    uint32_t rowsPerBlock;
    uint32_t procRows;

    uint32_t tileNumCols;
    uint32_t tileNumRows;
    uint32_t tileColsLen;
    uint32_t tileRowsLen;
};

extern "C" __global__ __aicore__ void gemm_divide_sum_scaling_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR alpha, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmDivideSumScaling op;
    op.Init(x, weight, alpha, y,
            tiling_data.batchSize,
            tiling_data.inputSize,
            tiling_data.hiddenSize,
            tiling_data.tileNumCols,
            tiling_data.tileNumRows,
            tiling_data.rowsPerBlock);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a custom AscendC kernel:
    Performs x @ weight.T, then divides by 2, sums over dim=1 (keepdim=True), and scales by scaling_factor.
    """
    def __init__(self, input_size, hidden_size, scaling_factor):
        super(ModelNew, self).__init__()
        self.weight = torch.nn.Parameter(torch.randn(hidden_size, input_size))
        self.scaling_factor = scaling_factor

    def forward(self, x):
        alpha = torch.tensor(self.scaling_factor, dtype=x.dtype, device=x.device)
        return custom_ops_lib.gemm_divide_sum_scaling_custom(x, self.weight, alpha)
