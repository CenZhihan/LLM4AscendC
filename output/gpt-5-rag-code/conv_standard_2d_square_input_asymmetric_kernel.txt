kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    // Extract tiling parameters
    const uint32_t N   = tiling_data.N;
    const uint32_t C   = tiling_data.C;
    const uint32_t H   = tiling_data.H;
    const uint32_t W   = tiling_data.W;
    const uint32_t OC  = tiling_data.OC;
    const uint32_t KH  = tiling_data.KH;
    const uint32_t KW  = tiling_data.KW;
    const uint32_t outH = tiling_data.outH;
    const uint32_t outW = tiling_data.outW;
    const uint32_t tileOc = tiling_data.tileOc;

    // Assume float dtype for this custom kernel
    __gm__ DTYPE_X* xPtr = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_Y* yPtr = reinterpret_cast<__gm__ DTYPE_Y*>(y);
    __gm__ DTYPE_W* wPtr = reinterpret_cast<__gm__ DTYPE_W*>(w);

    // Simple direct convolution: stride=1, padding=0, dilation=1, groups=1
    // NCHW input, OIHW weights
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t oc = 0; oc < OC; ++oc) {
            for (uint32_t oh = 0; oh < outH; ++oh) {
                for (uint32_t ow = 0; ow < outW; ++ow) {
                    float acc = 0.f;
                    for (uint32_t c = 0; c < C; ++c) {
                        for (uint32_t kh = 0; kh < KH; ++kh) {
                            const uint32_t ih = oh + kh;
                            const uint32_t in_base = (((n * C + c) * H) + ih) * W + ow;
                            const uint32_t w_row_base = (((oc * C) + c) * KH + kh) * KW;
                            // Unroll along KW
                            for (uint32_t kw = 0; kw < KW; ++kw) {
                                const uint32_t iw = ow + kw;
                                const uint32_t x_idx = (((n * C + c) * H) + ih) * W + iw;
                                const uint32_t w_idx = w_row_base + kw;
                                acc += static_cast<float>(xPtr[x_idx]) * static_cast<float>(wPtr[w_idx]);
                            }
                        }
                    }
                    const uint32_t y_idx = (((n * OC + oc) * outH) + oh) * outW + ow;
                    yPtr[y_idx] = static_cast<DTYPE_Y>(acc);
                }
            }
        }
    }
}
