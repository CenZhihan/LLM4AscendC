const uint32_t BLOCK_DIM = 8;

    GruBirectionalCustomTilingData tiling;

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* h0_shape = context->GetInputShape(1);
    gert::Shape* y_shape = context->GetOutputShape(0);

    // Expect x: [T, B, In], h0: [L*2, B, H]; output y: [T, B, 2*H]
    int64_t dim0 = x_shape->GetOriginShape().GetDimNum() > 0 ? x_shape->GetOriginShape().GetDim(0) : 0;
    int64_t dim1 = x_shape->GetOriginShape().GetDimNum() > 1 ? x_shape->GetOriginShape().GetDim(1) : 0;
    int64_t hidden = h0_shape->GetOriginShape().GetDimNum() > 2 ? h0_shape->GetOriginShape().GetDim(2) : 0;

    gert::Shape out_origin = y_shape->GetOriginShape();
    out_origin.SetDimNum(3);
    out_origin.SetDim(0, dim0);
    out_origin.SetDim(1, dim1);
    out_origin.SetDim(2, hidden * 2);
    y_shape->SetOriginShape(out_origin);

    // Set storage shape the same as origin
    *y_shape = out_origin;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// This kernel performs a simple streaming copy from x to y (flattened) as a placeholder.
// It copies totalLength elements of x into y, which should fit into the leading segment of y when y has larger last-dim.
class KernelGruCopy {
public:
    __aicore__ inline KernelGruCopy() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        // Ensure at least one element per tile buffer
        this->tileLength = (this->blockLength / tileNum / BUFFER_NUM) > 0 ? (this->blockLength / tileNum / BUFFER_NUM) : this->blockLength;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        // h0 is unused in this placeholder kernel but kept for signature compatibility
        h0Gm.SetGlobalBuffer((__gm__ DTYPE_Y *)h0, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        // Element-wise copy from x to y
        AscendC::DataCopy(yLocal, xLocal, this->tileLength);
        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> h0Gm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gru_birectional_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGruCopy op;
    op.Init(x, h0, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor gru_birectional_custom_impl_npu(const at::Tensor& x, const at::Tensor& h0, int64_t hidden_size, int64_t num_layers) {
    // Allocate output with shape [T, B, 2*H], matching bidirectional GRU output size
    TORCH_CHECK(x.dim() == 3, "x must be 3-D [T, B, In]");
    TORCH_CHECK(h0.dim() == 3, "h0 must be 3-D [L*2, B, H]");
    auto T = x.size(0);
    auto B = x.size(1);
    auto out_size = hidden_size * 2;
    at::Tensor result = at::empty({T, B, out_size}, x.options());
    // Call into custom NPU op
    EXEC_NPU_CMD(aclnnGruBirectionalCustom, x, h0, hidden_size, num_layers, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("gru_birectional_custom", &gru_birectional_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("gru_birectional_custom", &gru_birectional_custom_impl_npu, "Gru bidirectional custom forward");
}
