kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelReduceMinTanh2 {
public:
    __aicore__ inline KernelReduceMinTanh2() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t tileNum)
    {
        this->N = n;
        this->C = c;
        this->H = h;
        this->W = w;
        this->P = n * h * w; // positions per channel
        this->tileNum = tileNum;

        uint32_t blockNum = AscendC::GetBlockNum();
        this->blockLength = this->P / blockNum; // per block positions
        this->blockStart = this->blockLength * AscendC::GetBlockIdx();
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, this->C * this->P);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, this->P);

        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress)
    {
        uint32_t localOffset = progress * this->tileLength;
        uint32_t globalOffset = this->blockStart + localOffset;

        AscendC::LocalTensor<DTYPE_Z> minLocal = outQueue.AllocTensor<DTYPE_Z>();
        // Initialize minLocal with channel 0 slice
        AscendC::DataCopy(minLocal, xGm[globalOffset], this->tileLength);

        // Reduce min across channels
        for (uint32_t c = 1; c < this->C; ++c) {
            AscendC::LocalTensor<DTYPE_X> tmp = tmpQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(tmp, xGm[c * this->P + globalOffset], this->tileLength);
            AscendC::Minimum(minLocal, minLocal, tmp, this->tileLength);
            tmpQueue.FreeTensor(tmp);
        }

        // Apply tanh twice
        AscendC::Tanh(minLocal, minLocal, this->tileLength);
        AscendC::Tanh(minLocal, minLocal, this->tileLength);

        outQueue.EnQue<DTYPE_Z>(minLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        uint32_t localOffset = progress * this->tileLength;
        uint32_t globalOffset = this->blockStart + localOffset;

        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueue.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[globalOffset], outLocal, this->tileLength);
        outQueue.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t N;
    uint32_t C;
    uint32_t H;
    uint32_t W;
    uint32_t P;

    uint32_t blockLength;
    uint32_t blockStart;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelReduceMinTanh2 op;
    op.Init(x, y, tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w, tiling_data.tileNum);
    op.Process();
}
