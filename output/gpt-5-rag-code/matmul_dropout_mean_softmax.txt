kernel_src="""
#include "kernel_operator.h"

using DTYPE_T = float;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelMatmulDropoutMeanSoftmax {
public:
    __aicore__ inline KernelMatmulDropoutMeanSoftmax() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR p, GM_ADDR y, uint32_t batch, uint32_t features, uint32_t tileNum)
    {
        this->batch = batch;
        this->features = features;
        this->tileNum = tileNum;

        // distribute rows across blocks
        this->rowsPerBlock = (batch + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t blockStart = this->rowsPerBlock * AscendC::GetBlockIdx();
        if (blockStart + this->rowsPerBlock > batch) {
            this->rowsPerBlock = (batch > blockStart) ? (batch - blockStart) : 0;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_T*)x + blockStart * features, this->rowsPerBlock * features);
        yGm.SetGlobalBuffer((__gm__ DTYPE_T*)y + blockStart * features, this->rowsPerBlock * features);

        pGm.SetGlobalBuffer((__gm__ DTYPE_T*)p, 1);

        // tile along feature dimension for streaming
        this->tileLen = (features / tileNum == 0) ? features : (features / tileNum);
        if (this->tileLen == 0) this->tileLen = features;

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(DTYPE_T));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLen * sizeof(DTYPE_T));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLen * sizeof(DTYPE_T));
    }

    __aicore__ inline void Process()
    {
        if (this->rowsPerBlock == 0) {
            return;
        }

        // load dropout prob scalar (assumed shape [1])
        DTYPE_T prob = 0.0f;
        {
            AscendC::LocalTensor<DTYPE_T> pLocal = tmpQueue.AllocTensor<DTYPE_T>();
            AscendC::DataCopy(pLocal, pGm[0], 1);
            tmpQueue.EnQue(pLocal);
            AscendC::LocalTensor<DTYPE_T> pLocal2 = tmpQueue.DeQue<DTYPE_T>();
            prob = pLocal2.GetValue(0);
            tmpQueue.FreeTensor(pLocal2);
        }

        DTYPE_T scale = (DTYPE_T)(1.0f - prob);

        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            // Pass 1: compute mean of scaled row
            DTYPE_T sumScaled = 0.0f;
            uint32_t tiles = (features + tileLen - 1) / tileLen;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = r * features + t * tileLen;
                uint32_t curr = (t == tiles - 1) ? (features - t * tileLen) : tileLen;

                AscendC::LocalTensor<DTYPE_T> xLocal = inQueueX.AllocTensor<DTYPE_T>();
                AscendC::DataCopy(xLocal, xGm[offset], curr);
                inQueueX.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_T> xLocal2 = inQueueX.DeQue<DTYPE_T>();
                // accumulate scaled sum
                for (uint32_t i = 0; i < curr; ++i) {
                    sumScaled += xLocal2.GetValue(i) * scale;
                }
                inQueueX.FreeTensor(xLocal2);
            }
            DTYPE_T meanVal = sumScaled / (DTYPE_T)features;

            // Pass 2: compute sum of exp(x*scale - mean)
            DTYPE_T sumExp = 0.0f;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = r * features + t * tileLen;
                uint32_t curr = (t == tiles - 1) ? (features - t * tileLen) : tileLen;

                AscendC::LocalTensor<DTYPE_T> xLocal = inQueueX.AllocTensor<DTYPE_T>();
                AscendC::DataCopy(xLocal, xGm[offset], curr);
                inQueueX.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_T> xLocal2 = inQueueX.DeQue<DTYPE_T>();
                for (uint32_t i = 0; i < curr; ++i) {
                    DTYPE_T v = xLocal2.GetValue(i) * scale - meanVal;
                    DTYPE_T ev = AscendC::Exp(v);
                    sumExp += ev;
                }
                inQueueX.FreeTensor(xLocal2);
            }

            // Pass 3: write normalized softmax
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t offset = r * features + t * tileLen;
                uint32_t curr = (t == tiles - 1) ? (features - t * tileLen) : tileLen;

                AscendC::LocalTensor<DTYPE_T> xLocal = inQueueX.AllocTensor<DTYPE_T>();
                AscendC::LocalTensor<DTYPE_T> yLocal = outQueueY.AllocTensor<DTYPE_T>();
                AscendC::DataCopy(xLocal, xGm[offset], curr);
                inQueueX.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_T> xLocal2 = inQueueX.DeQue<DTYPE_T>();
                for (uint32_t i = 0; i < curr; ++i) {
                    DTYPE_T v = xLocal2.GetValue(i) * scale - meanVal;
                    DTYPE_T ev = AscendC::Exp(v);
                    yLocal.SetValue(i, ev / sumExp);
                }
                outQueueY.EnQue(yLocal);
                inQueueX.FreeTensor(xLocal2);

                AscendC::LocalTensor<DTYPE_T> yLocal2 = outQueueY.DeQue<DTYPE_T>();
                AscendC::DataCopy(yGm[offset], yLocal2, curr);
                outQueueY.FreeTensor(yLocal2);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_T> xGm;
    AscendC::GlobalTensor<DTYPE_T> pGm;
    AscendC::GlobalTensor<DTYPE_T> yGm;

    uint32_t batch;
    uint32_t features;
    uint32_t tileNum;
    uint32_t rowsPerBlock;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void matmul_dropout_mean_softmax_custom(GM_ADDR x, GM_ADDR p, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulDropoutMeanSoftmax op;
    op.Init(x, p, y, tiling_data.batch, tiling_data.features, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: performs Linear, then fused Dropout-Mean-Softmax using a custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, dropout_p):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features)
        self.dropout_p = float(dropout_p)

    def forward(self, x):
        # MatMul + Bias via nn.Linear
        logits = self.linear(x)
        # Fused Dropout-Mean-Softmax custom op
        p = torch.tensor(self.dropout_p, dtype=logits.dtype, device=logits.device)
        return custom_ops_lib.matmul_dropout_mean_softmax_custom(logits, p)
