kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void conv_transpose3d_scaling_avg_pool_bias_add_scaling_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR scale1, GM_ADDR scale2, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    uint32_t N = tiling_data.n;
    uint32_t C = tiling_data.c;
    uint32_t D = tiling_data.d;
    uint32_t H = tiling_data.h;
    uint32_t W = tiling_data.w;
    uint32_t outD = tiling_data.outD;
    uint32_t outH = tiling_data.outH;
    uint32_t outW = tiling_data.outW;
    uint32_t totalOut = tiling_data.totalOutLength;

    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();

    uint32_t blockLen = (totalOut + blockNum - 1) / blockNum;
    uint32_t start = blockIdx * blockLen;
    uint32_t end = start + blockLen;
    if (end > totalOut) end = totalOut;

    __gm__ float* xGM = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* biasGM = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* s1GM = reinterpret_cast<__gm__ float*>(scale1);
    __gm__ float* s2GM = reinterpret_cast<__gm__ float*>(scale2);
    __gm__ float* yGM = reinterpret_cast<__gm__ float*>(y);

    float s1 = s1GM[0];
    float s2 = s2GM[0];

    uint64_t strideW = 1;
    uint64_t strideH = static_cast<uint64_t>(W);
    uint64_t strideD = static_cast<uint64_t>(H) * static_cast<uint64_t>(W);
    uint64_t strideC = static_cast<uint64_t>(D) * static_cast<uint64_t>(H) * static_cast<uint64_t>(W);
    uint64_t strideN = static_cast<uint64_t>(C) * strideC;

    for (uint32_t idx = start; idx < end; ++idx) {
        uint32_t t = idx;
        uint32_t wo = t % outW; t /= outW;
        uint32_t ho = t % outH; t /= outH;
        uint32_t do_ = t % outD; t /= outD;
        uint32_t c = t % C; t /= C;
        uint32_t n = t;

        uint64_t baseNC = static_cast<uint64_t>(n) * strideN + static_cast<uint64_t>(c) * strideC;

        uint64_t di0 = static_cast<uint64_t>(do_) * 2;
        uint64_t hi0 = static_cast<uint64_t>(ho) * 2;
        uint64_t wi0 = static_cast<uint64_t>(wo) * 2;

        uint64_t o0 = baseNC + di0 * strideD + hi0 * strideH + wi0 * strideW;

        float sum = 0.0f;
        sum += xGM[o0];
        sum += xGM[o0 + 1];
        sum += xGM[o0 + strideH];
        sum += xGM[o0 + strideH + 1];
        sum += xGM[o0 + strideD];
        sum += xGM[o0 + strideD + 1];
        sum += xGM[o0 + strideD + strideH];
        sum += xGM[o0 + strideD + strideH + 1];

        float avg = (sum * s1) * 0.125f; // scale1 then average over 2x2x2
        float b = biasGM[c];             // bias is broadcast (C,1,1,1)
        yGM[idx] = s2 * (avg + b);
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: ConvTranspose3d followed by custom fused kernel
    performing: y = scale2 * (AvgPool3d(x * scale1, k=2, s=2) + bias).
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, scale1, scale2, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.scale1 = torch.nn.Parameter(torch.tensor(scale1, dtype=torch.float32))
        self.bias = torch.nn.Parameter(torch.randn(bias_shape, dtype=torch.float32))
        self.scale2 = torch.nn.Parameter(torch.tensor(scale2, dtype=torch.float32))

    def forward(self, x):
        x = self.conv_transpose(x)
        # fused processing after deconvolution
        return custom_ops_lib.conv_transpose3d_scaling_avg_pool_bias_add_scaling_custom(x, self.bias, self.scale1, self.scale2)
