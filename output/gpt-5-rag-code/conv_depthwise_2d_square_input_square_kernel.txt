kernel_src="""
#include "kernel_operator.h"

#ifndef DTYPE_X
#define DTYPE_X float
#endif
#ifndef DTYPE_W
#define DTYPE_W float
#endif
#ifndef DTYPE_Y
#define DTYPE_Y float
#endif

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t N = tiling_data.n();
    const uint32_t C = tiling_data.c();
    const uint32_t H = tiling_data.h();
    const uint32_t W = tiling_data.w();
    const uint32_t K = tiling_data.k();
    const uint32_t OH = tiling_data.out_h();
    const uint32_t OW = tiling_data.out_w();
    const uint32_t STRIDE = tiling_data.stride();
    const uint32_t PAD = tiling_data.padding();

    __gm__ DTYPE_X* x_gm = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_W* w_gm = reinterpret_cast<__gm__ DTYPE_W*>(w);
    __gm__ DTYPE_Y* y_gm = reinterpret_cast<__gm__ DTYPE_Y*>(y);

    const uint32_t block_num = AscendC::GetBlockNum();
    const uint32_t block_idx = AscendC::GetBlockIdx();

    const uint64_t units = static_cast<uint64_t>(N) * static_cast<uint64_t>(C); // per (n,c)
    const uint64_t units_per_block = (units + block_num - 1) / block_num;
    const uint64_t unit_begin = units_per_block * block_idx;
    const uint64_t unit_end = (unit_begin + units_per_block) > units ? units : (unit_begin + units_per_block);

    // Simple per-(n,c) computation
    for (uint64_t u = unit_begin; u < unit_end; ++u) {
        uint32_t n = static_cast<uint32_t>(u / C);
        uint32_t c = static_cast<uint32_t>(u % C);

        // Load filter for channel c into local buffer
        // Assuming small K (square kernel)
        // Local buffer for filter
        DTYPE_W w_local[64]; // supports up to 8x8 kernel
        uint32_t k_area = K * K;
        // Copy filter from GM: offset = c*K*K
        for (uint32_t t = 0; t < k_area; ++t) {
            w_local[t] = w_gm[static_cast<uint64_t>(c) * k_area + t];
        }

        // Base offsets
        const uint64_t x_nc_base = (static_cast<uint64_t>(n) * C + c) * H * W;
        const uint64_t y_nc_base = (static_cast<uint64_t>(n) * C + c) * OH * OW;

        for (uint32_t oh = 0; oh < OH; ++oh) {
            for (uint32_t ow = 0; ow < OW; ++ow) {
                float acc = 0.0f;
                // Convolution window
                for (uint32_t kh = 0; kh < K; ++kh) {
                    int32_t ih = static_cast<int32_t>(oh) * static_cast<int32_t>(STRIDE) + static_cast<int32_t>(kh) - static_cast<int32_t>(PAD);
                    if (ih < 0 || ih >= static_cast<int32_t>(H)) {
                        continue;
                    }
                    for (uint32_t kw = 0; kw < K; ++kw) {
                        int32_t iw = static_cast<int32_t>(ow) * static_cast<int32_t>(STRIDE) + static_cast<int32_t>(kw) - static_cast<int32_t>(PAD);
                        if (iw < 0 || iw >= static_cast<int32_t>(W)) {
                            continue;
                        }
                        uint64_t x_off = x_nc_base + static_cast<uint64_t>(ih) * W + static_cast<uint64_t>(iw);
                        uint32_t w_off = kh * K + kw;
                        acc += static_cast<float>(x_gm[x_off]) * static_cast<float>(w_local[w_off]);
                    }
                }
                uint64_t y_off = y_nc_base + static_cast<uint64_t>(oh) * OW + static_cast<uint64_t>(ow);
                y_gm[y_off] = static_cast<DTYPE_Y>(acc);
            }
        }
    }
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline int64_t compute_out_dim(int64_t in, int64_t k, int64_t stride, int64_t padding) {
    return (in + 2 * padding - k) / stride + 1;
}

at::Tensor conv_depthwise_2d_square_input_square_kernel_custom_impl_npu(
    const at::Tensor& x, const at::Tensor& w, int64_t stride, int64_t padding) {

    TORCH_CHECK(x.dim() == 4, "Input x must be 4-D NCHW");
    TORCH_CHECK(w.dim() == 4, "Weight w must be 4-D (C,1,K,K)");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "Only float32 supported");
    TORCH_CHECK(w.scalar_type() == at::kFloat, "Only float32 supported");
    int64_t N = x.size(0);
    int64_t C = x.size(1);
    int64_t H = x.size(2);
    int64_t W = x.size(3);
    int64_t K = w.size(2);
    TORCH_CHECK(w.size(0) == C && w.size(1) == 1 && w.size(2) == w.size(3), "Invalid weight shape for depthwise");

    int64_t OH = compute_out_dim(H, K, stride, padding);
    int64_t OW = compute_out_dim(W, K, stride, padding);
    auto y = at::empty({N, C, OH, OW}, x.options());

    EXEC_NPU_CMD(aclnnConvDepthwise2dSquareInputSquareKernelCustom, x, w, stride, padding, y);
    return y;
}

import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Depthwise 2D convolution using custom AscendC kernel for square input and square kernel.
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, bias: bool = False):
        super().__init__()
        if bias:
            raise NotImplementedError("Bias is not supported by the custom kernel in this example.")
        # Keep a Conv2d only to manage parameters (weights) and initialization
        self.conv2d = nn.Conv2d(in_channels, in_channels, kernel_size, stride=stride, padding=padding,
                                groups=in_channels, bias=False)
        self.stride = stride
        self.padding = padding

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Use custom kernel instead of PyTorch conv operator
        return custom_ops_lib.conv_depthwise_2d_square_input_square_kernel_custom(
            x, self.conv2d.weight, self.stride, self.padding
        )
