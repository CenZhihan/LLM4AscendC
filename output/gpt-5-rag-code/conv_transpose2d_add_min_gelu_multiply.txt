kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelFusedPostOps {
public:
    __aicore__ inline KernelFusedPostOps() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add, GM_ADDR mul, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ float *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        addGm.SetGlobalBuffer((__gm__ float *)add + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mulGm.SetGlobalBuffer((__gm__ float *)mul + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ float *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueAdd, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueMul, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue1, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue2, BUFFER_NUM, this->tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
        AscendC::LocalTensor<float> addLocal = inQueueAdd.AllocTensor<float>();
        AscendC::LocalTensor<float> mulLocal = inQueueMul.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(addLocal, addGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(mulLocal, mulGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueAdd.EnQue(addLocal);
        inQueueMul.EnQue(mulLocal);
    }

    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX.DeQue<float>();
        AscendC::LocalTensor<float> addLocal = inQueueAdd.DeQue<float>();
        AscendC::LocalTensor<float> mulLocal = inQueueMul.DeQue<float>();

        AscendC::LocalTensor<float> tmp1 = tmpQueue1.AllocTensor<float>();
        AscendC::LocalTensor<float> zeroLocal = tmpQueue2.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

        // x + add
        AscendC::Add(tmp1, xLocal, addLocal, this->tileLength);
        // min(tmp1, 0.0)
        AscendC::Dup(zeroLocal, 0.0f, this->tileLength);
        AscendC::Min(yLocal, tmp1, zeroLocal, this->tileLength);
        // gelu(yLocal)
        AscendC::Gelu(yLocal, yLocal, this->tileLength);
        // yLocal *= mul
        AscendC::Mul(yLocal, yLocal, mulLocal, this->tileLength);

        outQueueY.EnQue<float>(yLocal);

        tmpQueue1.FreeTensor(tmp1);
        tmpQueue2.FreeTensor(zeroLocal);

        inQueueX.FreeTensor(xLocal);
        inQueueAdd.FreeTensor(addLocal);
        inQueueMul.FreeTensor(mulLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueAdd, inQueueMul;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue1, tmpQueue2;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> addGm;
    AscendC::GlobalTensor<float> mulGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose2d_add_min_gelu_multiply_custom(GM_ADDR x, GM_ADDR add, GM_ADDR mul, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedPostOps op;
    op.Init(x, add, mul, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Uses standard ConvTranspose2d, then fuses add -> min(.,0) -> GELU -> multiply by a custom AscendC kernel.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, add_value, multiply_value):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride)
        self.add_value = float(add_value)
        self.multiply_value = float(multiply_value)

    def forward(self, x):
        y = self.conv_transpose(x)
        add_t = torch.full_like(y, self.add_value)
        mul_t = torch.full_like(y, self.multiply_value)
        y = custom_ops_lib.conv_transpose2d_add_min_gelu_multiply_custom(y, add_t, mul_t)
        return y
