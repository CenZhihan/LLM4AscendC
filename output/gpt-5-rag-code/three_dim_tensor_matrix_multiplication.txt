inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ThreeDimTensorMatrixMultiplicationCustomTilingData tiling;

    const auto* a_shape = context->GetInputShape(0);
    const auto* b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    // Expect A: (N, M, K), B: (K, L)
    uint32_t n = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    uint32_t m = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    uint32_t k = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(2));
    uint32_t l = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(1));

    uint32_t rows = n * m;

    // Set a conservative block dim
    const uint32_t MAX_BLOCKS = 8;
    uint32_t blockDim = rows < MAX_BLOCKS ? rows : MAX_BLOCKS;
    blockDim = blockDim == 0 ? 1 : blockDim;
    context->SetBlockDim(blockDim);

    uint32_t rowsPerBlock = CeilDiv(rows, blockDim);

    tiling.set_n(n);
    tiling.set_m(m);
    tiling.set_k(k);
    tiling.set_l(l);
    tiling.set_rowsPerBlock(rowsPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

    gert::Shape* c_shape = context->GetOutputShape(0);

kernel_src="""
#include "kernel_operator.h"

using DTYPE_A = float;
using DTYPE_B = float;
using DTYPE_C = float;

constexpr int32_t BUFFER_NUM = 1;

class KernelThreeDimTensorMatMul {
public:
    __aicore__ inline KernelThreeDimTensorMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t n, uint32_t m, uint32_t k, uint32_t l,
                                uint32_t rowsPerBlock)
    {
        this->n = n;
        this->m = m;
        this->k = k;
        this->l = l;

        uint32_t rows = n * m;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        this->startRow = rowsPerBlock * blockIdx;
        this->endRow = this->startRow + rowsPerBlock;
        if (this->endRow > rows) {
            this->endRow = rows;
        }
        if (this->startRow >= this->endRow) {
            this->startRow = this->endRow;
        }
        this->rowsToProcess = (this->endRow > this->startRow) ? (this->endRow - this->startRow) : 0;

        aGm.SetGlobalBuffer((__gm__ DTYPE_A*)A + (uint64_t)this->startRow * k, (uint64_t)this->rowsToProcess * k);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B*)B, (uint64_t)k * l);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C*)C + (uint64_t)this->startRow * l, (uint64_t)this->rowsToProcess * l);

        pipe.InitBuffer(inQueueA, BUFFER_NUM, sizeof(DTYPE_A));           // scalar buffer for A element
        pipe.InitBuffer(inQueueB, BUFFER_NUM, sizeof(DTYPE_B));           // scalar buffer for B element
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, sizeof(DTYPE_C));           // scalar temp buffer
        pipe.InitBuffer(outQueueC, BUFFER_NUM, sizeof(DTYPE_C));          // scalar buffer for C element
    }

    __aicore__ inline void Process()
    {
        if (rowsToProcess == 0) {
            return;
        }
        for (uint32_t row = 0; row < rowsToProcess; ++row) {                  // flatten over N*M
            uint64_t aRowBase = (uint64_t)row * k;
            uint64_t cRowBase = (uint64_t)row * l;

            for (uint32_t col = 0; col < l; ++col) {                          // for each output column
                // cLocal holds the scalar accumulation for this (row, col)
                AscendC::LocalTensor<DTYPE_C> cLocal = outQueueC.AllocTensor<DTYPE_C>();

                bool initialized = false;

                // iterate over K dimension
                for (uint32_t kk = 0; kk < k; ++kk) {
                    // load A[row, kk]
                    AscendC::LocalTensor<DTYPE_A> aLocal = inQueueA.AllocTensor<DTYPE_A>();
                    AscendC::DataCopy(aLocal, aGm[aRowBase + kk], 1);
                    inQueueA.EnQue(aLocal);
                    aLocal = inQueueA.DeQue<DTYPE_A>();

                    // load B[kk, col]
                    AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
                    AscendC::DataCopy(bLocal, bGm[(uint64_t)kk * l + col], 1);
                    inQueueB.EnQue(bLocal);
                    bLocal = inQueueB.DeQue<DTYPE_B>();

                    if (!initialized) {
                        // c = a * b
                        AscendC::Mul(cLocal, aLocal, bLocal, 1);
                        initialized = true;
                    } else {
                        // tmp = a * b; c += tmp
                        AscendC::LocalTensor<DTYPE_C> tmpLocal = tmpQueue.AllocTensor<DTYPE_C>();
                        AscendC::Mul(tmpLocal, aLocal, bLocal, 1);
                        AscendC::Add(cLocal, cLocal, tmpLocal, 1);
                        tmpQueue.FreeTensor(tmpLocal);
                    }

                    inQueueA.FreeTensor(aLocal);
                    inQueueB.FreeTensor(bLocal);
                }

                // store C[row, col]
                AscendC::DataCopy(cGm[cRowBase + col], cLocal, 1);
                outQueueC.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t n, m, k, l;
    uint32_t startRow, endRow, rowsToProcess;
};

extern "C" __global__ __aicore__ void three_dim_tensor_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelThreeDimTensorMatMul op;
    op.Init(A, B, C,
            tiling_data.n, tiling_data.m, tiling_data.k, tiling_data.l,
            tiling_data.rowsPerBlock);
    op.Process();
}
