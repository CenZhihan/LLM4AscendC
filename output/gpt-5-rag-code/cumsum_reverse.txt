host_operator_src="""
#include "cumsum_reverse_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    CumsumReverseCustomTilingData tiling;

    // Get input shape and compute dims
    const gert::Shape* in_shape = context->GetInputShape(0);
    uint32_t rank = in_shape->GetOriginShape().GetDimNum();
    uint64_t total_len_u64 = in_shape->GetOriginShape().GetShapeSize();
    if (rank == 0) {
        return ge::GRAPH_FAILED;
    }
    uint64_t last_dim_u64 = in_shape->GetOriginShape().GetDim(rank - 1);
    if (last_dim_u64 == 0) {
        return ge::GRAPH_FAILED;
    }
    uint64_t rows_u64 = total_len_u64 / last_dim_u64;

    // Choose a reasonable tile size along the last dimension
    // Keep it small enough to fit into UB comfortably
    uint32_t tileSize = 1024;
    if (last_dim_u64 < tileSize) {
        tileSize = static_cast<uint32_t>(last_dim_u64);
    }

    // Parallelize over rows across blocks
    uint32_t rows = static_cast<uint32_t>(rows_u64);
    uint32_t dimSize = static_cast<uint32_t>(last_dim_u64);
    uint32_t blockLines = CeilDiv(rows, BLOCK_DIM);

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(static_cast<uint32_t>(total_len_u64));
    tiling.set_rows(rows);
    tiling.set_dimSize(dimSize);
    tiling.set_tileSize(tileSize);
    tiling.set_blockLines(blockLines);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class CumsumReverseCustom : public OpDef {
public:
    explicit CumsumReverseCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // number of buffers per queue

class KernelCumsumReverse {
public:
    __aicore__ inline KernelCumsumReverse() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t rows, uint32_t dimSize,
                                uint32_t tileSize, uint32_t blockLines)
    {
        this->rows = rows;
        this->dimSize = dimSize;
        this->tileSize = tileSize;
        this->blockLines = blockLines;

        uint32_t startLine = AscendC::GetBlockIdx() * blockLines;
        if (startLine >= this->rows) {
            this->linesThisBlock = 0;
        } else {
            uint32_t remain = this->rows - startLine;
            this->linesThisBlock = (remain < blockLines) ? remain : blockLines;
        }
        uint32_t baseOffset = startLine * this->dimSize;
        uint32_t totalElemsThisBlock = this->linesThisBlock * this->dimSize;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + baseOffset, totalElemsThisBlock);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + baseOffset, totalElemsThisBlock);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileSize * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileSize * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // For each line handled by this block, compute reverse cumsum along the last dimension
        for (uint32_t line = 0; line < this->linesThisBlock; ++line) {
            // Offset of the beginning of this line in GM
            uint32_t lineBase = line * this->dimSize;

            // Number of segments along the last dim
            uint32_t segNum = (this->dimSize + this->tileSize - 1) / this->tileSize;
            // Accumulator for cross-segment carry
            float carry = 0.0f;

            // Traverse from last segment to first
            for (int32_t seg = static_cast<int32_t>(segNum) - 1; seg >= 0; --seg) {
                uint32_t segStart = static_cast<uint32_t>(seg) * this->tileSize;
                uint32_t segEnd = segStart + this->tileSize;
                if (segEnd > this->dimSize) segEnd = this->dimSize;
                uint32_t currLen = segEnd - segStart;

                // Copy in current segment
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[lineBase + segStart], currLen);
                inQueueX.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_X> xSeg = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> ySeg = outQueueY.AllocTensor<DTYPE_Y>();

                // Compute reverse cumsum inside the segment with carry from the next segment
                for (int32_t j = static_cast<int32_t>(currLen) - 1; j >= 0; --j) {
                    float v = static_cast<float>(xSeg.GetValue(j));
                    carry += v;
                    ySeg.SetValue(j, static_cast<DTYPE_Y>(carry));
                }

                outQueueY.EnQue(ySeg);
                inQueueX.FreeTensor(xSeg);

                // Copy out
                AscendC::LocalTensor<DTYPE_Y> yOut = outQueueY.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[lineBase + segStart], yOut, currLen);
                outQueueY.FreeTensor(yOut);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows;
    uint32_t dimSize;
    uint32_t tileSize;
    uint32_t blockLines;
    uint32_t linesThisBlock;
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverse op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.rows,
            tiling_data.dimSize,
            tiling_data.tileSize,
            tiling_data.blockLines);
    if (AscendC::GetBlockIdx() * tiling_data.blockLines < tiling_data.rows) {
        op.Process();
    }
}
"""

#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor cumsum_reverse_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    // This custom kernel implements reverse cumsum along the last dimension.
    // The 'dim' argument is accepted to match the PyTorch API, but the kernel assumes last dimension.
    at::Tensor result = at::empty_like(self);
    EXEC_NPU_CMD(aclnnCumsumReverseCustom, self, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("cumsum_reverse_custom", &cumsum_reverse_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("cumsum_reverse_custom", &cumsum_reverse_custom_impl_npu, "reverse cumsum along last dim");
}
