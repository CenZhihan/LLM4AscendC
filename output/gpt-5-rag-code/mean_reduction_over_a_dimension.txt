kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMeanReductionOverLastDim {
public:
    __aicore__ inline KernelMeanReductionOverLastDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outRows, uint32_t reduceLen)
    {
        this->outRows = outRows;
        this->reduceLen = reduceLen;

        uint32_t blockId = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Split rows among blocks
        uint32_t start = static_cast<uint32_t>((static_cast<uint64_t>(outRows) * blockId) / blockNum);
        uint32_t end   = static_cast<uint32_t>((static_cast<uint64_t>(outRows) * (blockId + 1)) / blockNum);
        this->blockStartRow = start;
        this->blockRows = (end > start) ? (end - start) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + static_cast<uint64_t>(this->blockStartRow) * this->reduceLen,
                            static_cast<uint64_t>(this->blockRows) * this->reduceLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockStartRow, this->blockRows);

        // Each iteration handles one row worth of elements
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->reduceLen * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t row = 0; row < this->blockRows; ++row) {
            CopyIn(row);
            Compute();
            CopyOut(row);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[row * this->reduceLen], this->reduceLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // Sum over the last dimension and compute mean
        AscendC::ReduceSum(yLocal, xLocal, this->reduceLen);
        float scale = 1.0f / static_cast<float>(this->reduceLen);
        AscendC::Muls(yLocal, yLocal, scale, 1);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[row], yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    uint32_t outRows;
    uint32_t reduceLen;
    uint32_t blockStartRow;
    uint32_t blockRows;
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverLastDim op;
    op.Init(x, y, tiling_data.outRows, tiling_data.reduceLen);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using a custom AscendC kernel that reduces over the last dimension.
    For generic dim, we permute to move the target dim to the last, apply the custom op, and return.
    """
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        dim = self.dim
        ndim = x.dim()
        if dim < 0:
            dim += ndim
        assert 0 <= dim < ndim, "Invalid reduction dimension"
        if dim != ndim - 1:
            perm = [i for i in range(ndim) if i != dim] + [dim]
            x = x.permute(perm).contiguous()
        # Now reduce over last dimension using the custom kernel
        y = custom_ops_lib.mean_reduction_over_a_dimension_custom(x)
        # No need to permute back because remaining axes preserve original order when dim is removed
        return y
