#include "mlp_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MlpCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const auto& origin = x_shape->GetOriginShape();

    uint32_t rank = origin.GetDimNum();
    if (rank != 2) {
        return ge::GRAPH_FAILED;
    }
    uint32_t rows = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t cols = static_cast<uint32_t>(origin.GetDim(1));

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_rows(rows);
    tiling.set_cols(cols);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 2; // double buffer for pipeline

class KernelMlpCustom {
public:
    __aicore__ inline KernelMlpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y, uint32_t rows, uint32_t cols)
    {
        this->rows = rows;
        this->cols = cols;

        // Set full buffers; we'll use block-stride over rows
        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, rows * cols);
        bGm.SetGlobalBuffer((__gm__ DTYPE*)bias, cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, rows * cols);

        // One tile processes one full row [cols]
        uint32_t tileSizeBytes = cols * sizeof(DTYPE);
        pipe.InitBuffer(inQueueX, BUFFER_NUM, tileSizeBytes);
        pipe.InitBuffer(inQueueB, BUFFER_NUM, tileSizeBytes);
        pipe.InitBuffer(outQueueY, BUFFER_NUM, tileSizeBytes);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // block-strided over row dimension
        for (uint32_t r = blockIdx; r < rows; r += blockNum) {
            CopyIn(r);
            Compute();
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIndex)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = inQueueB.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[rowIndex * cols], cols);
        AscendC::DataCopy(bLocal, bGm[0], cols);
        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = inQueueB.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();

        AscendC::Add(yLocal, xLocal, bLocal, cols);
        AscendC::Relu(yLocal, yLocal, cols);

        outQueueY.EnQue(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIndex)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[rowIndex * cols], yLocal, cols);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t rows;
    uint32_t cols;
};

extern "C" __global__ __aicore__ void mlp_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMlpCustom op;
    op.Init(x, bias, y, tiling_data.rows, tiling_data.cols);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, input_size, layer_sizes, output_size):
        super(ModelNew, self).__init__()
        layers = []
        biases = []
        current_input_size = input_size

        # Hidden layers: Linear (no bias) + custom kernel (bias add + ReLU)
        for layer_size in layer_sizes:
            lin = nn.Linear(current_input_size, layer_size, bias=False)
            layers.append(lin)
            biases.append(nn.Parameter(torch.zeros(layer_size)))
            current_input_size = layer_size

        # Output layer: Linear (no bias) + bias add (no ReLU)
        self.output_linear = nn.Linear(current_input_size, output_size, bias=False)
        self.output_bias = nn.Parameter(torch.zeros(output_size))

        self.hidden_linears = nn.ModuleList(layers)
        self.hidden_biases = nn.ParameterList(biases)

    def forward(self, x):
        for lin, b in zip(self.hidden_linears, self.hidden_biases):
            x = lin(x)
            x = custom_ops_lib.mlp_custom(x, b)  # fused bias add + ReLU
        x = self.output_linear(x)
        x = x + self.output_bias  # last layer: bias add only
        return x
'''
