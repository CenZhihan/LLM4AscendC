kernel_src="""
#include "kernel_operator.h"

using dtype = float;

class KernelPostBnBiasDivSwish {
public:
    __aicore__ inline KernelPostBnBiasDivSwish() {}

    __aicore__ inline void Init(
        GM_ADDR y, GM_ADDR bn_mean, GM_ADDR bn_var, GM_ADDR bn_weight, GM_ADDR bn_bias,
        GM_ADDR extra_bias, GM_ADDR bn_eps, GM_ADDR divide_value,
        uint32_t M, uint32_t N, uint32_t extraBiasIsScalar) {

        this->M = M;
        this->N = N;
        this->rowsPerBlock = M / AscendC::GetBlockNum();
        this->startRow = this->rowsPerBlock * AscendC::GetBlockIdx();
        this->extraBiasIsScalar = extraBiasIsScalar;

        yGm.SetGlobalBuffer((__gm__ dtype *)y + (uint64_t)startRow * N, (uint64_t)rowsPerBlock * N);
        meanGm.SetGlobalBuffer((__gm__ dtype *)bn_mean, N);
        varGm.SetGlobalBuffer((__gm__ dtype *)bn_var, N);
        gammaGm.SetGlobalBuffer((__gm__ dtype *)bn_weight, N);
        betaGm.SetGlobalBuffer((__gm__ dtype *)bn_bias, N);
        extraBiasGm.SetGlobalBuffer((__gm__ dtype *)extra_bias, extraBiasIsScalar ? 1 : N);
        epsGm.SetGlobalBuffer((__gm__ dtype *)bn_eps, 1);
        divGm.SetGlobalBuffer((__gm__ dtype *)divide_value, 1);

        pipe.InitBuffer(inQueueY, 2, N * sizeof(dtype));
        pipe.InitBuffer(outQueueY, 2, N * sizeof(dtype));
        pipe.InitBuffer(tmpQueueA, 2, N * sizeof(dtype));
        pipe.InitBuffer(tmpQueueB, 2, N * sizeof(dtype));
    }

    __aicore__ inline void Process() {
        for (uint32_t r = 0; r < rowsPerBlock; ++r) {
            // Load row y
            AscendC::LocalTensor<dtype> yLocal = inQueueY.AllocTensor<dtype>();
            AscendC::DataCopy(yLocal, yGm[(uint64_t)r * N], N);
            inQueueY.EnQue(yLocal);

            // Load BN params
            AscendC::LocalTensor<dtype> meanLocal = tmpQueueA.AllocTensor<dtype>();
            AscendC::LocalTensor<dtype> varLocal = tmpQueueB.AllocTensor<dtype>();
            AscendC::LocalTensor<dtype> gammaLocal = outQueueY.AllocTensor<dtype>(); // reuse buffer
            AscendC::LocalTensor<dtype> betaLocal = inQueueY.AllocTensor<dtype>();   // reuse buffer
            AscendC::DataCopy(meanLocal, meanGm, N);
            AscendC::DataCopy(varLocal, varGm, N);
            AscendC::DataCopy(gammaLocal, gammaGm, N);
            AscendC::DataCopy(betaLocal, betaGm, N);

            // Prepare scalars and extra bias vector
            dtype epsScalar;
            dtype divScalar;
            AscendC::LocalTensor<dtype> epsLocal = tmpQueueA.AllocTensor<dtype>();
            AscendC::LocalTensor<dtype> divLocal = tmpQueueB.AllocTensor<dtype>();
            AscendC::DataCopy(epsLocal[0], epsGm, 1);
            AscendC::DataCopy(divLocal[0], divGm, 1);
            // Duplicate scalar to vector
            AscendC::Dup(epsLocal, epsLocal.GetValue(0), N);
            AscendC::Dup(divLocal, divLocal.GetValue(0), N);

            AscendC::LocalTensor<dtype> extraBiasLocal = outQueueY.AllocTensor<dtype>();
            if (extraBiasIsScalar) {
                // load scalar, then duplicate
                AscendC::DataCopy(extraBiasLocal[0], extraBiasGm, 1);
                AscendC::Dup(extraBiasLocal, extraBiasLocal.GetValue(0), N);
            } else {
                AscendC::DataCopy(extraBiasLocal, extraBiasGm, N);
            }

            // Compute
            AscendC::LocalTensor<dtype> yIn = inQueueY.DeQue<dtype>();
            AscendC::LocalTensor<dtype> yOut = outQueueY.AllocTensor<dtype>();

            // y = ((y - mean) / sqrt(var + eps)) * gamma + beta
            AscendC::LocalTensor<dtype> t0 = tmpQueueA.AllocTensor<dtype>(); // y - mean
            AscendC::Sub(t0, yIn, meanLocal, N);

            AscendC::LocalTensor<dtype> var_eps = tmpQueueB.AllocTensor<dtype>(); // var + eps
            AscendC::Add(var_eps, varLocal, epsLocal, N);

            AscendC::Sqrt(var_eps, var_eps, N); // sqrt(var + eps)
            AscendC::Div(t0, t0, var_eps, N);   // (y - mean)/sqrt

            AscendC::Mul(t0, t0, gammaLocal, N); // * gamma
            AscendC::Add(t0, t0, betaLocal, N);  // + beta

            // + extra_bias
            AscendC::Add(t0, t0, extraBiasLocal, N);

            // / divide_value
            AscendC::Div(t0, t0, divLocal, N);

            // swish: t0 * sigmoid(t0)
            AscendC::LocalTensor<dtype> sig = tmpQueueB.AllocTensor<dtype>();
            AscendC::Sigmoid(sig, t0, N);
            AscendC::Mul(yOut, t0, sig, N);

            // Store
            outQueueY.EnQue(yOut);
            AscendC::LocalTensor<dtype> yRes = outQueueY.DeQue<dtype>();
            AscendC::DataCopy(yGm[(uint64_t)r * N], yRes, N);
            outQueueY.FreeTensor(yRes);

            // Free temps
            inQueueY.FreeTensor(yIn);
            tmpQueueA.FreeTensor(meanLocal);
            tmpQueueB.FreeTensor(varLocal);
            outQueueY.FreeTensor(gammaLocal);
            inQueueY.FreeTensor(betaLocal);
            tmpQueueA.FreeTensor(epsLocal);
            tmpQueueB.FreeTensor(divLocal);
            outQueueY.FreeTensor(extraBiasLocal);
            tmpQueueA.FreeTensor(t0);
            tmpQueueB.FreeTensor(var_eps);
            tmpQueueB.FreeTensor(sig);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> tmpQueueA;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> tmpQueueB;

    AscendC::GlobalTensor<dtype> yGm;
    AscendC::GlobalTensor<dtype> meanGm, varGm, gammaGm, betaGm;
    AscendC::GlobalTensor<dtype> extraBiasGm, epsGm, divGm;

    uint32_t M, N;
    uint32_t rowsPerBlock;
    uint32_t startRow;
    uint32_t extraBiasIsScalar;
};

extern "C" __global__ __aicore__ void matmul_batch_norm_bias_add_divide_swish_custom(
    GM_ADDR x, GM_ADDR weight_t, GM_ADDR linear_bias,
    GM_ADDR bn_running_mean, GM_ADDR bn_running_var, GM_ADDR bn_weight, GM_ADDR bn_bias,
    GM_ADDR extra_bias, GM_ADDR y, GM_ADDR bn_eps, GM_ADDR divide_value,
    GM_ADDR workspace, GM_ADDR tiling) 
{
    // Prepare tiling
    GET_TILING_DATA(tilingData, tiling);

    // Define matmul kernel types
    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype, true, LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype>;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> batchMatmulKernel;
    AscendC::TPipe pipe;

    // Configure shared memory sizes for matmul
    tilingData.cubeTilingData.shareMode = 0;              // share mode
    tilingData.cubeTilingData.shareL1Size = FULL_L1_SIZE; // full L1
    tilingData.cubeTilingData.shareL0CSize = FULL_L0C_SIZE; // full L0C
    tilingData.cubeTilingData.shareUbSize = 0;            // no UB

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), batchMatmulKernel.matmulObj, &tilingData.cubeTilingData);

    // Init matmul kernel and compute y = x @ weight_t + linear_bias
    batchMatmulKernel.Init(x, weight_t, linear_bias, y, workspace, tilingData.cubeTilingData);
    batchMatmulKernel.Process(&pipe, 1, 1);

    // Post processing: BN + extra_bias + divide + swish in-place on y
    KernelPostBnBiasDivSwish post;
    post.Init(y, bn_running_mean, bn_running_var, bn_weight, bn_bias, extra_bias, bn_eps, divide_value,
              tilingData.M, tilingData.N, tilingData.extraBiasIsScalar);
    post.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, in_features, out_features, bn_eps=1e-5, bn_momentum=0.1, bias_shape=(1,), divide_value=1.0):
        super().__init__()
        self.matmul = nn.Linear(in_features, out_features)
        self.bn = nn.BatchNorm1d(out_features, eps=bn_eps, momentum=bn_momentum)
        self.bias = nn.Parameter(torch.randn(bias_shape))
        self.divide_value = divide_value

    def forward(self, x):
        # Prepare tensors
        wt = self.matmul.weight.t().contiguous()
        if self.matmul.bias is not None:
            lin_bias = self.matmul.bias.contiguous()
        else:
            lin_bias = torch.zeros(wt.size(1), dtype=x.dtype, device=x.device)

        # Ensure shapes and dtypes
        running_mean = self.bn.running_mean.contiguous().to(dtype=x.dtype, device=x.device)
        running_var = self.bn.running_var.contiguous().to(dtype=x.dtype, device=x.device)
        gamma = (self.bn.weight if self.bn.weight is not None else torch.ones_like(running_mean)).contiguous().to(dtype=x.dtype, device=x.device)
        beta = (self.bn.bias if self.bn.bias is not None else torch.zeros_like(running_mean)).contiguous().to(dtype=x.dtype, device=x.device)

        extra_bias = self.bias
        if extra_bias.numel() == 1:
            extra_bias = extra_bias.view(1).to(dtype=x.dtype, device=x.device).contiguous()
        else:
            extra_bias = extra_bias.reshape(-1).to(dtype=x.dtype, device=x.device).contiguous()

        bn_eps_t = torch.tensor([self.bn.eps], dtype=x.dtype, device=x.device)
        divide_value_t = torch.tensor([self.divide_value], dtype=x.dtype, device=x.device)

        return custom_ops_lib.matmul_batch_norm_bias_add_divide_swish_custom(
            x.contiguous(), wt, lin_bias,
            running_mean, running_var, gamma, beta,
            extra_bias, bn_eps_t, divide_value_t
        )
