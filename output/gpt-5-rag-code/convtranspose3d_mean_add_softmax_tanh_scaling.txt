#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvtranspose3dMeanAddSoftmaxTanhScaling {
public:
    __aicore__ inline KernelConvtranspose3dMeanAddSoftmaxTanhScaling() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR y,
                                uint32_t n, uint32_t d, uint32_t h, uint32_t w, uint32_t c,
                                uint32_t groups, uint32_t tileNum)
    {
        this->N = n;
        this->D = d;
        this->H = h;
        this->W = w;
        this->C = c;
        this->groups = groups;
        this->tileNum = tileNum;

        // Each block handles a slice of groups
        this->groupsPerBlock = groups / AscendC::GetBlockNum();
        uint32_t remainder = groups % AscendC::GetBlockNum();
        // distribute remainder to early blocks
        uint32_t myBlockIdx = AscendC::GetBlockIdx();
        this->groupStart = (this->groupsPerBlock * myBlockIdx) + AscendC::Min(myBlockIdx, remainder);
        this->myGroups = this->groupsPerBlock + (myBlockIdx < remainder ? 1 : 0);

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N * D * H * W * C);
        biasGm.SetGlobalBuffer((__gm__ float*)bias, (uint64_t)C);
        scaleGm.SetGlobalBuffer((__gm__ float*)scale, (uint64_t)1);
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N * D * H * W * C);

        // local buffers for per-group processing (channel vector)
        pipe.InitBuffer(channelQueue, BUFFER_NUM, C * sizeof(float));
        pipe.InitBuffer(workQueue, BUFFER_NUM, C * sizeof(float));
        pipe.InitBuffer(biasQueue, 1, C * sizeof(float));

        // Load bias once to local buffer
        AscendC::LocalTensor<float> biasLocal = biasQueue.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, biasGm[0], C);
        biasQueue.EnQue(biasLocal);

        // Load scale
        AscendC::LocalTensor<float> scaleLocal = workQueue.AllocTensor<float>();
        AscendC::DataCopy(scaleLocal, scaleGm[0], 1);
        this->scaleVal = scaleLocal.GetValue(0); // assuming API to read local scalar
        workQueue.FreeTensor(scaleLocal);
    }

    __aicore__ inline void Process()
    {
        // Iterate groups assigned to this block; each group corresponds to a contiguous vector of C channels (NHWDC channels-last)
        for (uint32_t g = 0; g < this->myGroups; ++g) {
            uint64_t baseGroup = (uint64_t)(this->groupStart + g);
            uint64_t baseOffset = baseGroup * C;

            // Load input channel vector
            AscendC::LocalTensor<float> xLocal = channelQueue.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm[baseOffset], C);

            // Load bias (already in queue)
            AscendC::LocalTensor<float> biasLocal = biasQueue.DeQue<float>();

            // Add bias
            AscendC::LocalTensor<float> yLocal = workQueue.AllocTensor<float>();
            AscendC::Add(yLocal, xLocal, biasLocal, C);

            // Compute softmax across channels
            // 1) find max
            float maxVal = yLocal.GetValue(0);
            for (uint32_t i = 1; i < C; ++i) {
                float vi = yLocal.GetValue(i);
                maxVal = (vi > maxVal) ? vi : maxVal;
            }
            // 2) subtract max and exponentiate, compute sum
            float sumExp = 0.f;
            for (uint32_t i = 0; i < C; ++i) {
                float vi = yLocal.GetValue(i) - maxVal;
                float evi = AscendC::ExpScalar(vi); // assuming scalar exp helper
                yLocal.SetValue(i, evi);
                sumExp += evi;
            }
            float invSum = 1.0f / sumExp;
            for (uint32_t i = 0; i < C; ++i) {
                float si = yLocal.GetValue(i) * invSum;
                // tanh
                float ti = AscendC::TanhScalar(si); // assuming scalar tanh helper
                // scaling
                ti = ti * this->scaleVal;
                yLocal.SetValue(i, ti);
            }

            // Store result
            AscendC::DataCopy(yGm[baseOffset], yLocal, C);

            // free locals
            workQueue.FreeTensor(yLocal);
            biasQueue.EnQue(biasLocal);
            channelQueue.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> channelQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> workQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQueue;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> biasGm;
    AscendC::GlobalTensor<float> scaleGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t N, D, H, W, C;
    uint32_t groups;
    uint32_t tileNum;
    uint32_t groupsPerBlock;
    uint32_t groupStart;
    uint32_t myGroups;
    float scaleVal;
};

extern "C" __global__ __aicore__ void convtranspose3d_mean_add_softmax_tanh_scaling_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{

    op.Init(x, bias, scale, y,
            tiling_data.n(), tiling_data.d(), tiling_data.h(), tiling_data.w(), tiling_data.c(),
            tiling_data.groups(), tiling_data.tileNum());

}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model:
    - Keep ConvTranspose3d in PyTorch
    - Mean pooling in PyTorch
    - Permute to channels-last for contiguous channel vectors
    - Fused bias add + softmax (across channels) + tanh + scaling via custom AscendC kernel
    - Permute back to channels-first
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.bias = torch.nn.Parameter(torch.randn(1, out_channels, 1, 1, 1))
        self.scaling_factor = scaling_factor

    def forward(self, x):
        x = self.conv_transpose(x)                            # (N, C, D, H, W)
        x = x.mean(dim=2, keepdim=True)                       # (N, C, 1, H, W)

        # Convert to channels-last NHWDC for contiguous channels
        x_cl = x.permute(0, 2, 3, 4, 1).contiguous()          # (N, 1, H, W, C)
        bias_cl = self.bias.permute(0, 2, 3, 4, 1).contiguous()  # (1, 1, 1, 1, C)
        scale_tensor = torch.tensor([self.scaling_factor], dtype=x_cl.dtype, device=x_cl.device)

        y_cl = custom_ops_lib.convtranspose3d_mean_add_softmax_tanh_scaling_custom(x_cl, bias_cl, scale_tensor)
        y = y_cl.permute(0, 4, 1, 2, 3).contiguous()          # (N, C, 1, H, W)
        return y
'''
