kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // double buffering for pipeline

class KernelConvPointwise2D {
public:
    __aicore__ inline KernelConvPointwise2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalNHW, uint32_t cin, uint32_t cout, uint32_t tileNum)
    {
        this->cin = cin;
        this->cout = cout;
        this->tileNum = tileNum;
        // Split NHW across blocks
        this->blockNHW = (totalNHW + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t startNHW = this->blockNHW * AscendC::GetBlockIdx();
        if (startNHW + this->blockNHW > totalNHW) {
            this->blockNHW = (totalNHW > startNHW) ? (totalNHW - startNHW) : 0;
        }
        // Flat NHWC representation, x size per block is blockNHW * cin
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)startNHW * cin, (uint64_t)this->blockNHW * cin);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)cin * cout);
        if (b != nullptr) {
            bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, cout);
            hasBias = true;
        } else {
            hasBias = false;
        }
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + (uint64_t)startNHW * cout, (uint64_t)this->blockNHW * cout);

        // tile along NHW
        uint32_t rawTile = this->blockNHW / (tileNum * BUFFER_NUM);
        this->tileNHW = (rawTile == 0) ? this->blockNHW : rawTile;

        // buffers: X tile (tileNHW * cin), W (cin * cout), Y tile (tileNHW * cout), B (cout)
        pipe.InitBuffer(inQueueX, BUFFER_NUM, (uint64_t)this->tileNHW * cin * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, 1, (uint64_t)cin * cout * sizeof(DTYPE_W));
        pipe.InitBuffer(inQueueB, 1, (uint64_t)cout * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, (uint64_t)this->tileNHW * cout * sizeof(DTYPE_Y));

        // Preload weight and bias once
        AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.AllocTensor<DTYPE_W>();
        AscendC::DataCopy(wLocal, wGm, (uint64_t)cin * cout);
        inQueueW.EnQue(wLocal);

        if (hasBias) {
            AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
            AscendC::DataCopy(bLocal, bGm, cout);
            inQueueB.EnQue(bLocal);
        }
    }

    __aicore__ inline void Process()
    {
        if (blockNHW == 0) {
            return;
        }
        uint32_t loops = DivCeil(this->blockNHW, this->tileNHW);
        for (uint32_t i = 0; i < loops; ++i) {
            uint32_t curNHW = ((i + 1) * this->tileNHW <= this->blockNHW) ? this->tileNHW : (this->blockNHW - i * this->tileNHW);
            CopyIn(i, curNHW);
            Compute(i, curNHW);
            CopyOut(i, curNHW);
        }
    }

private:
    __aicore__ inline uint32_t DivCeil(uint32_t a, uint32_t b) {
        return (a + b - 1) / b;
    }

    __aicore__ inline void CopyIn(uint32_t progress, uint32_t curNHW)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        // Copy x tile: contiguous NHW*Cin
        AscendC::DataCopy(xLocal, xGm[(uint64_t)progress * tileNHW * cin], (uint64_t)curNHW * cin);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t progress, uint32_t curNHW)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.DeQue<DTYPE_W>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_B> bLocal;
        if (hasBias) {
            bLocal = inQueueB.DeQue<DTYPE_B>();
        }

        // yLocal is size curNHW * cout
        // Initialize with bias if present, else zeros
        for (uint32_t nhw = 0; nhw < curNHW; ++nhw) {
            for (uint32_t oc = 0; oc < cout; ++oc) {
                DTYPE_Y val = (DTYPE_Y)0;
                if (hasBias) {
                    val = (DTYPE_Y)bLocal[oc];
                }
                yLocal[(uint64_t)nhw * cout + oc] = val;
            }
        }

        // Accumulate over input channels
        for (uint32_t ic = 0; ic < cin; ++ic) {
            for (uint32_t nhw = 0; nhw < curNHW; ++nhw) {
                DTYPE_X xval = xLocal[(uint64_t)nhw * cin + ic];
                // Fused multiply-add for all output channels
                for (uint32_t oc = 0; oc < cout; ++oc) {
                    DTYPE_W wval = wLocal[(uint64_t)ic * cout + oc];
                    yLocal[(uint64_t)nhw * cout + oc] = yLocal[(uint64_t)nhw * cout + oc] + (DTYPE_Y)(xval * wval);
                }
            }
        }

        outQueueY.EnQue(yLocal);
        inQueueX.FreeTensor(xLocal);
        if (hasBias) {
            // Keep bias resident for entire run; do not free/enqueue back to keep it accessible
            inQueueB.EnQue(bLocal);
        }
        inQueueW.EnQue(wLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress, uint32_t curNHW)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[(uint64_t)progress * tileNHW * cout], yLocal, (uint64_t)curNHW * cout);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t cin;
    uint32_t cout;
    uint32_t tileNum;
    uint32_t blockNHW;
    uint32_t tileNHW;
    bool hasBias;
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2D op;
    op.Init(x, weight, bias, y,
            tiling_data.totalNHW,
            tiling_data.cin,
            tiling_data.cout,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized pointwise 2D convolution (1x1) using custom AscendC kernel.
    Expects and computes using NHWC layout inside, with weight flattened to [Cin, Cout].
    """
    def __init__(self, in_channels: int, out_channels: int, bias: bool = False):
        super().__init__()
        self.conv1d = torch.nn.Conv2d(in_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Convert to NHWC for contiguous tiles along NHW
        x_nhwc = x.permute(0, 2, 3, 1).contiguous()
        # Prepare weight as [Cin, Cout]
        # Original weight: [Cout, Cin, 1, 1]
        w = self.conv1d.weight.squeeze(-1).squeeze(-1).permute(1, 0).contiguous()
        if self.conv1d.bias is not None:
            b = self.conv1d.bias
        else:
            b = None
        y_nhwc = custom_ops_lib.conv_pointwise_2d_custom(x_nhwc, w, b)
        # Convert back to NCHW
        y = y_nhwc.permute(0, 3, 1, 2).contiguous()
        return y
