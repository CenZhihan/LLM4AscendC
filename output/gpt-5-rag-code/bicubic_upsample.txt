#include "bicubic_upsample_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    BicubicUpsampleCustomTilingData tiling;

    // Input shape assumed to be NCHW
    auto inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t N = static_cast<uint32_t>(inShape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(inShape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(inShape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(inShape.GetDim(3));

    // Fixed target size per model: 256 x 256, align_corners=True
    uint32_t outH = 256;
    uint32_t outW = 256;

    uint32_t totalLength = N * C * outH * outW;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_inH(H);
    tiling.set_inW(W);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_alignCorners(1);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* in_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);

    std::vector<int64_t> origin_dims = in_shape->GetOriginShape().GetDims();
    // Force output shape: N, C, 256, 256
    if (origin_dims.size() != 4) {
        return ge::GRAPH_FAILED;
    }
    origin_dims[2] = 256;
    origin_dims[3] = 256;

    gert::Shape y_origin;
    y_origin.SetDimNum(origin_dims.size());
    for (size_t i = 0; i < origin_dims.size(); ++i) {
        y_origin.SetDim(i, origin_dims[i]);
    }

    out_shape->SetOriginShape(y_origin);
    out_shape->SetStorageShape(y_origin);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelBicubicUpsample {
public:
    __aicore__ inline KernelBicubicUpsample() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t N, uint32_t C,
                                uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW,
                                uint32_t alignCorners)
    {
        this->N = N; this->C = C;
        this->inH = inH; this->inW = inW;
        this->outH = outH; this->outW = outW;
        this->alignCorners = alignCorners != 0;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        if (this->tileLength == 0) {
            this->tileLength = 1;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));

        // precompute scales for align_corners=True
        if (this->alignCorners) {
            this->scaleH = (outH > 1 && inH > 1) ? static_cast<float>(inH - 1) / static_cast<float>(outH - 1) : 0.0f;
            this->scaleW = (outW > 1 && inW > 1) ? static_cast<float>(inW - 1) / static_cast<float>(outW - 1) : 0.0f;
        } else {
            // not used in this model, but set for completeness
            this->scaleH = (outH > 0) ? static_cast<float>(inH) / static_cast<float>(outH) : 0.0f;
            this->scaleW = (outW > 0) ? static_cast<float>(inW) / static_cast<float>(outW) : 0.0f;
        }
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            ComputeAndCopy(i);
        }
    }

private:
    __aicore__ inline float cubic_kernel(float x, float a) {
        float ax = AscendC::fabs(x);
        float ax2 = ax * ax;
        float ax3 = ax2 * ax;
        if (ax <= 1.0f) {
            return (a + 2.0f) * ax3 - (a + 3.0f) * ax2 + 1.0f;
        } else if (ax < 2.0f) {
            return a * ax3 - 5.0f * a * ax2 + 8.0f * a * ax - 4.0f * a;
        } else {
            return 0.0f;
        }
    }

    __aicore__ inline int64_t get_in_index(uint32_t n, uint32_t c, uint32_t h, uint32_t w) {
        return (((int64_t)n * C + c) * inH + h) * inW + w;
    }

    __aicore__ inline void ComputeAndCopy(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.AllocTensor<DTYPE_Y>();

        // Align with current tile segment
        uint32_t offsetBase = progress * this->tileLength;

        // Bicubic with Catmull-Rom a = -0.75
        const float a = -0.75f;

        for (uint32_t i = 0; i < this->tileLength; ++i) {
            uint32_t outIndex = offsetBase + i;
            if (outIndex >= this->blockLength) {
                break;
            }

            // unravel within block: order is (n, c, oh, ow)
            uint32_t ow = outIndex % outW;
            uint32_t tmp1 = outIndex / outW;
            uint32_t oh = tmp1 % outH;
            uint32_t tmp2 = tmp1 / outH;
            uint32_t c = tmp2 % C;
            uint32_t n = tmp2 / C;

            // compute source coordinates
            float srcY, srcX;
            if (alignCorners) {
                srcY = (outH > 1) ? static_cast<float>(oh) * scaleH : 0.0f;
                srcX = (outW > 1) ? static_cast<float>(ow) * scaleW : 0.0f;
            } else {
                srcY = (static_cast<float>(oh) + 0.5f) * scaleH - 0.5f;
                srcX = (static_cast<float>(ow) + 0.5f) * scaleW - 0.5f;
            }

            int32_t sy = static_cast<int32_t>(AscendC::floor(srcY));
            int32_t sx = static_cast<int32_t>(AscendC::floor(srcX));
            float ty = srcY - static_cast<float>(sy);
            float tx = srcX - static_cast<float>(sx);

            // compute weights for 4 neighbors in each dimension
            float wy[4];
            wy[0] = cubic_kernel(1.0f + ty, a);
            wy[1] = cubic_kernel(ty, a);
            wy[2] = cubic_kernel(1.0f - ty, a);
            wy[3] = cubic_kernel(2.0f - ty, a);

            float wx[4];
            wx[0] = cubic_kernel(1.0f + tx, a);
            wx[1] = cubic_kernel(tx, a);
            wx[2] = cubic_kernel(1.0f - tx, a);
            wx[3] = cubic_kernel(2.0f - tx, a);

            // accumulate 4x4 neighborhood
            float acc = 0.0f;
            for (int dy = 0; dy < 4; ++dy) {
                int32_t iy = sy - 1 + dy;
                if (iy < 0) iy = 0;
                if (iy >= static_cast<int32_t>(inH)) iy = static_cast<int32_t>(inH) - 1;
                float wyv = wy[dy];
                for (int dx = 0; dx < 4; ++dx) {
                    int32_t ix = sx - 1 + dx;
                    if (ix < 0) ix = 0;
                    if (ix >= static_cast<int32_t>(inW)) ix = static_cast<int32_t>(inW) - 1;
                    float wxv = wx[dx];

                    int64_t inIdx = get_in_index(n, c, static_cast<uint32_t>(iy), static_cast<uint32_t>(ix));
                    // Direct access; assume GlobalTensor supports element access for illustration
                    float v = static_cast<float>(xGm[inIdx]);
                    acc += wyv * wxv * v;
                }
            }

            yLocal[i] = static_cast<DTYPE_Y>(acc);
        }

        outQueue.EnQue<DTYPE_Y>(yLocal);
        AscendC::DataCopy(yGm[offsetBase], yLocal, this->tileLength);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, C, inH, inW, outH, outW;
    bool alignCorners;
    float scaleH;
    float scaleW;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void bicubic_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBicubicUpsample op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.n, tiling_data.c,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.alignCorners);
    op.Process();
}
