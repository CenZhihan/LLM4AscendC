    // Input shape: NCHW

    auto origin = in_shape->GetOriginShape();

    uint32_t n = (origin.GetDimNum() > 0) ? static_cast<uint32_t>(origin.GetDim(0)) : 1;
    uint32_t c = (origin.GetDimNum() > 1) ? static_cast<uint32_t>(origin.GetDim(1)) : 1;
    uint32_t h = (origin.GetDimNum() > 2) ? static_cast<uint32_t>(origin.GetDim(2)) : 1;
    uint32_t w = (origin.GetDimNum() > 3) ? static_cast<uint32_t>(origin.GetDim(3)) : 1;

    int64_t kernel = 1;
    int64_t stride = 0;
    int64_t padding = 0;

    // Read attributes (best-effort; default handling included)
    auto attrs = context->GetAttrs();
    if (attrs != nullptr) {
        (void)attrs->GetInt("kernel_size", kernel);
        (void)attrs->GetInt("stride", stride);
        (void)attrs->GetInt("padding", padding);

    if (kernel <= 0) kernel = 1;
    if (stride <= 0) stride = kernel; // PyTorch default when stride=None
    if (padding < 0) padding = 0;

    int64_t out_w = div_floor(static_cast<int64_t>(w) + 2 * padding - kernel, stride) + 1;
    if (out_h < 1) out_h = 1;
    if (out_w < 1) out_w = 1;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_out_h(static_cast<uint32_t>(out_h));
    tiling.set_out_w(static_cast<uint32_t>(out_w));
    tiling.set_kernel(static_cast<uint32_t>(kernel));
    tiling.set_stride(static_cast<uint32_t>(stride));
    tiling.set_padding(static_cast<uint32_t>(padding));

kernel_src="""
#include "kernel_operator.h"

extern \"C\" __global__ __aicore__ void average_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.n;
    const uint32_t C = tiling_data.c;
    const uint32_t H = tiling_data.h;
    const uint32_t W = tiling_data.w;
    const uint32_t OH = tiling_data.out_h;
    const uint32_t OW = tiling_data.out_w;
    const uint32_t K = tiling_data.kernel;
    const uint32_t S = tiling_data.stride;
    const uint32_t P = tiling_data.padding;

    const uint64_t in_spatial = static_cast<uint64_t>(H) * static_cast<uint64_t>(W);
    const uint64_t out_spatial = static_cast<uint64_t>(OH) * static_cast<uint64_t>(OW);

    __gm__ DTYPE_X* in  = (__gm__ DTYPE_X*)x;
    __gm__ DTYPE_Y* out = (__gm__ DTYPE_Y*)y;

    const uint32_t block_num = AscendC::GetBlockNum();
    const uint32_t block_idx = AscendC::GetBlockIdx();

    const uint64_t total_outputs = static_cast<uint64_t>(N) * static_cast<uint64_t>(C) * out_spatial;
    const uint64_t base = (total_outputs / block_num) * block_idx;
    const uint64_t tail = (block_idx == block_num - 1) ? (total_outputs - (total_outputs / block_num) * (block_num - 1))
                                                      : (total_outputs / block_num);

    for (uint64_t linear = 0; linear < tail; ++linear) {
        uint64_t idx = base + linear;

        uint64_t ow = idx % OW; idx /= OW;
        uint64_t oh = idx % OH; idx /= OH;
        uint64_t c  = idx % C;  idx /= C;
        uint64_t n  = idx;

        int64_t ih0 = static_cast<int64_t>(oh) * static_cast<int64_t>(S) - static_cast<int64_t>(P);
        int64_t iw0 = static_cast<int64_t>(ow) * static_cast<int64_t>(S) - static_cast<int64_t>(P);

        float sum = 0.0f;
        uint32_t count = 0;

        for (uint32_t kh = 0; kh < K; ++kh) {
            int64_t ih = ih0 + static_cast<int64_t>(kh);
            if (ih < 0 || ih >= static_cast<int64_t>(H)) {
                continue;
            }
            for (uint32_t kw = 0; kw < K; ++kw) {
                int64_t iw = iw0 + static_cast<int64_t>(kw);
                if (iw < 0 || iw >= static_cast<int64_t>(W)) {
                    continue;
                }
                uint64_t in_index = (((n * C + c) * H) + static_cast<uint64_t>(ih)) * W + static_cast<uint64_t>(iw);
                sum += static_cast<float>(in[in_index]);
                ++count;
            }
        }

        float result = (count > 0) ? (sum / static_cast<float>(count)) : 0.0f;
        uint64_t out_index = (((n * C + c) * OH) + oh) * OW + ow;
        out[out_index] = static_cast<DTYPE_Y>(result);
    }
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor average_pooling_2d_custom_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding) {
    if (stride <= 0) {
        stride = kernel_size;
    }
    at::Tensor result;
    // Infer output shape for allocation
    auto N = self.size(0);
    auto C = self.size(1);
    auto H = self.size(2);
    auto W = self.size(3);

    int64_t out_h = (H + 2 * padding - kernel_size) / stride + 1;
    int64_t out_w = (W + 2 * padding - kernel_size) / stride + 1;
    if (out_h < 1) out_h = 1;
    if (out_w < 1) out_w = 1;

    result = at::empty({N, C, out_h, out_w}, self.options());

    EXEC_NPU_CMD(aclnnAveragePooling2dCustom, self, kernel_size, stride, padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("average_pooling_2d_custom", &average_pooling_2d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("average_pooling_2d_custom", &average_pooling_2d_custom_impl_npu, "AveragePooling2dCustom");
}
