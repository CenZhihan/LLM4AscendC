kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmAddRelu {
public:
    __aicore__ inline KernelGemmAddRelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t tileN, uint32_t rowsPerBlock)
    {
        this->m = m;
        this->n = n;
        this->tileN = tileN;
        this->rowsPerBlock = rowsPerBlock;

        xGm.SetGlobalBuffer((__gm__ DTYPE *)x, (uint64_t)m * n);
        bGm.SetGlobalBuffer((__gm__ DTYPE *)bias, n);
        yGm.SetGlobalBuffer((__gm__ DTYPE *)y, (uint64_t)m * n);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileN * sizeof(DTYPE));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileN * sizeof(DTYPE));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileN * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        uint32_t startRow = AscendC::GetBlockIdx() * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > m) endRow = m;

        for (uint32_t r = startRow; r < endRow; ++r) {
            for (uint32_t c = 0; c < n; c += tileN) {
                uint32_t currTile = (c + tileN <= n) ? tileN : (n - c);
                CopyIn(r, c, currTile);
                Compute(currTile);
                CopyOut(r, c, currTile);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row, uint32_t colStart, uint32_t currTile)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = inQueueB.AllocTensor<DTYPE>();

        AscendC::DataCopy(xLocal, xGm[(uint64_t)row * n + colStart], currTile);
        AscendC::DataCopy(bLocal, bGm[colStart], currTile);

        inQueueX.EnQue(xLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t currTile)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> bLocal = inQueueB.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();

        AscendC::Add(yLocal, xLocal, bLocal, currTile);
        // ReLU: y = max(y, 0)
        AscendC::Max(yLocal, yLocal, static_cast<DTYPE>(0.0f), currTile);

        outQueueY.EnQue<DTYPE>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueB.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row, uint32_t colStart, uint32_t currTile)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[(uint64_t)row * n + colStart], yLocal, currTile);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> bGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t m;
    uint32_t n;
    uint32_t tileN;
    uint32_t rowsPerBlock;
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddRelu op;
    op.Init(x, bias, y, tiling_data.m, tiling_data.n, tiling_data.tileN, tiling_data.rowsPerBlock);
    op.Process();
}
