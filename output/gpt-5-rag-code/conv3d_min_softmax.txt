kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;

class KernelConv3dMinSoftmax {
public:
    __aicore__ inline KernelConv3dMinSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->d = d;
        this->h = h;
        this->w = w;
        this->tileNum = tileNum;

        uint64_t totalIn = (uint64_t)n * (uint64_t)c * (uint64_t)d * (uint64_t)h * (uint64_t)w;
        xGm.SetGlobalBuffer((__gm__ DTYPE*)x, totalIn);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y, totalIn);

        // Local buffers:
        // vecC: length = c, holds per-channel reduced minima
        // vecD: length = d, holds per-depth values for one channel
        // tmp: length = max(c, d) but we use 1 for scalar temporaries too
        uint32_t maxLen = c > d ? c : d;
        pipe.InitBuffer(vecCQueue, 1, c * sizeof(DTYPE));
        pipe.InitBuffer(vecDQueue, 1, d * sizeof(DTYPE));
        pipe.InitBuffer(tmpQueue, 2, maxLen * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        uint64_t P = (uint64_t)n * (uint64_t)h * (uint64_t)w; // number of (n,h,w) positions
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint64_t basePerBlock = (P + blockNum - 1) / blockNum;
        uint64_t startP = (uint64_t)blockIdx * basePerBlock;
        uint64_t endP = startP + basePerBlock;
        if (endP > P) endP = P;

        for (uint64_t p = startP; p < endP; ++p) {
            // decode p -> (nIdx, hIdx, wIdx)
            uint32_t nIdx = (uint32_t)(p / ((uint64_t)h * (uint64_t)w));
            uint64_t rem = p % ((uint64_t)h * (uint64_t)w);
            uint32_t hIdx = (uint32_t)(rem / (uint64_t)w);
            uint32_t wIdx = (uint32_t)(rem % (uint64_t)w);

            // Allocate local buffers
            AscendC::LocalTensor<DTYPE> vecC = vecCQueue.AllocTensor<DTYPE>();
            // Fill vecC[c] with +inf initially
            FillWithInf(vecC, c);

            for (uint32_t ci = 0; ci < c; ++ci) {
                // Load along D for this (nIdx, ci, :, hIdx, wIdx) into vecD
                AscendC::LocalTensor<DTYPE> vecD = vecDQueue.AllocTensor<DTYPE>();
                for (uint32_t di = 0; di < d; ++di) {
                    uint64_t off = Offset5D(nIdx, ci, di, hIdx, wIdx);
                    // Copy single element
                    AscendC::DataCopy(vecD[di], xGm[off], 1);
                }

                // ReduceMin over D -> store into vecC[ci]
                ReduceMinTo(vecC, ci, vecD, d);

                vecDQueue.FreeTensor(vecD);
            }

            // Softmax over C for this (nIdx, :, hIdx, wIdx): vecC holds logits
            SoftmaxInplace(vecC, c);

            // Store result to output at D index 0 (i.e., y[n, c, 0, h, w])
            for (uint32_t ci = 0; ci < c; ++ci) {
                uint64_t out_off = Offset5D(nIdx, ci, 0, hIdx, wIdx);
                AscendC::DataCopy(yGm[out_off], vecC[ci], 1);
            }

            vecCQueue.FreeTensor(vecC);
        }
    }

private:
    __aicore__ inline uint64_t Offset5D(uint32_t nIdx, uint32_t cIdx, uint32_t dIdx, uint32_t hIdx, uint32_t wIdx) const
    {
        // NCDHW contiguous: (((((n)*C + c)*D + d)*H + h)*W + w)
        return (((((uint64_t)nIdx) * (uint64_t)c + (uint64_t)cIdx) * (uint64_t)d + (uint64_t)dIdx) * (uint64_t)h
               + (uint64_t)hIdx) * (uint64_t)w + (uint64_t)wIdx;
    }

    __aicore__ inline void FillWithInf(AscendC::LocalTensor<DTYPE>& t, uint32_t len)
    {
        // Use a large positive number as +inf
        AscendC::LocalTensor<DTYPE> scalar = tmpQueue.AllocTensor<DTYPE>();
        // set scalar = 1e30
        AscendC::Duplicate(scalar, (DTYPE)1.0e30f, 1);
        for (uint32_t i = 0; i < len; ++i) {
            AscendC::DataCopy(t[i], scalar, 1);
        }
        tmpQueue.FreeTensor(scalar);
    }

    __aicore__ inline void ReduceMinTo(AscendC::LocalTensor<DTYPE>& dstVecC, uint32_t idxC,
                                       AscendC::LocalTensor<DTYPE>& srcVecD, uint32_t lenD)
    {
        // Compute min over srcVecD[0:lenD) into a scalar, then write to dstVecC[idxC]
        AscendC::LocalTensor<DTYPE> curMin = tmpQueue.AllocTensor<DTYPE>();
        // Initialize curMin = srcVecD[0]
        AscendC::DataCopy(curMin, srcVecD[0], 1);
        for (uint32_t i = 1; i < lenD; ++i) {
            AscendC::LocalTensor<DTYPE> v = tmpQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(v, srcVecD[i], 1);
            AscendC::Min(curMin, curMin, v, 1);
            tmpQueue.FreeTensor(v);
        }
        AscendC::DataCopy(dstVecC[idxC], curMin, 1);
        tmpQueue.FreeTensor(curMin);
    }

    __aicore__ inline void SoftmaxInplace(AscendC::LocalTensor<DTYPE>& vec, uint32_t len)
    {
        // max
        AscendC::LocalTensor<DTYPE> curMax = tmpQueue.AllocTensor<DTYPE>();
        AscendC::DataCopy(curMax, vec[0], 1);
        for (uint32_t i = 1; i < len; ++i) {
            AscendC::LocalTensor<DTYPE> v = tmpQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(v, vec[i], 1);
            AscendC::Max(curMax, curMax, v, 1);
            tmpQueue.FreeTensor(v);
        }

        // subtract max and exp; also accumulate sum
        AscendC::LocalTensor<DTYPE> sumExp = tmpQueue.AllocTensor<DTYPE>();
        AscendC::Duplicate(sumExp, (DTYPE)0.0f, 1);

        for (uint32_t i = 0; i < len; ++i) {
            AscendC::LocalTensor<DTYPE> v = tmpQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(v, vec[i], 1);
            AscendC::Sub(v, v, curMax, 1);
            AscendC::Exp(v, v, 1);
            AscendC::Add(sumExp, sumExp, v, 1);
            AscendC::DataCopy(vec[i], v, 1);
            tmpQueue.FreeTensor(v);
        }

        // divide by sumExp
        for (uint32_t i = 0; i < len; ++i) {
            AscendC::LocalTensor<DTYPE> v = tmpQueue.AllocTensor<DTYPE>();
            AscendC::DataCopy(v, vec[i], 1);
            AscendC::Div(v, v, sumExp, 1);
            AscendC::DataCopy(vec[i], v, 1);
            tmpQueue.FreeTensor(v);
        }

        tmpQueue.FreeTensor(sumExp);
        tmpQueue.FreeTensor(curMax);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  vecCQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  vecDQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> tmpQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t n, c, d, h, w;
    uint32_t tileNum;
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMinSoftmax op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model that keeps PyTorch Conv3d and uses a fused custom AscendC kernel
    to compute min over depth (dim=2) followed by softmax over channels (dim=1).
    The custom kernel returns shape [N, C, D, H, W] with results written at D=0,
    so we squeeze dim 2 to match the original interface [N, C, H, W].
    """
    def __init__(self, in_channels, out_channels, kernel_size, dim):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.dim = dim  # expected to be 2

    def forward(self, x):
        x = self.conv(x)                     # [N, C, D, H, W]
        x = custom_ops_lib.conv3d_min_softmax_custom(x)  # [N, C, D, H, W], results at D=0
        x = x.squeeze(2)                     # [N, C, H, W]
        return x
'''
