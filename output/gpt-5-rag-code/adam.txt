kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor buffers per queue

class KernelAdam {
public:
    __aicore__ inline KernelAdam() {}
    __aicore__ inline void Init(GM_ADDR param, GM_ADDR grad, GM_ADDR m, GM_ADDR v,
                                GM_ADDR beta1, GM_ADDR beta2, GM_ADDR lr, GM_ADDR eps,
                                GM_ADDR m_hat_coeff, GM_ADDR v_hat_coeff,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        paramGm.SetGlobalBuffer((__gm__ float *)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gradGm.SetGlobalBuffer((__gm__ float *)grad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        mGm.SetGlobalBuffer((__gm__ float *)m + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ float *)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outParamGm.SetGlobalBuffer((__gm__ float *)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength); // write back

        beta1Gm.SetGlobalBuffer((__gm__ float *)beta1, 1);
        beta2Gm.SetGlobalBuffer((__gm__ float *)beta2, 1);
        lrGm.SetGlobalBuffer((__gm__ float *)lr, 1);
        epsGm.SetGlobalBuffer((__gm__ float *)eps, 1);
        mHatCoeffGm.SetGlobalBuffer((__gm__ float *)m_hat_coeff, 1);
        vHatCoeffGm.SetGlobalBuffer((__gm__ float *)v_hat_coeff, 1);

        pipe.InitBuffer(inQueueParam, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueGrad, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueM, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(inQueueV, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(outQueueParam, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue1, BUFFER_NUM, this->tileLength * sizeof(float));
        pipe.InitBuffer(tmpQueue2, BUFFER_NUM, this->tileLength * sizeof(float));

        // Load scalar hyperparameters once
        AscendC::LocalTensor<float> s1 = tmpQueue1.AllocTensor<float>();
        AscendC::LocalTensor<float> s2 = tmpQueue1.AllocTensor<float>();
        AscendC::LocalTensor<float> s3 = tmpQueue1.AllocTensor<float>();
        AscendC::LocalTensor<float> s4 = tmpQueue1.AllocTensor<float>();
        AscendC::LocalTensor<float> s5 = tmpQueue1.AllocTensor<float>();
        AscendC::LocalTensor<float> s6 = tmpQueue1.AllocTensor<float>();

        AscendC::DataCopy(s1, beta1Gm[0], 1);
        AscendC::DataCopy(s2, beta2Gm[0], 1);
        AscendC::DataCopy(s3, lrGm[0], 1);
        AscendC::DataCopy(s4, epsGm[0], 1);
        AscendC::DataCopy(s5, mHatCoeffGm[0], 1);
        AscendC::DataCopy(s6, vHatCoeffGm[0], 1);

        beta1_ = s1.GetValue(0);
        beta2_ = s2.GetValue(0);
        lr_ = s3.GetValue(0);
        eps_ = s4.GetValue(0);
        m_hat_coeff_ = s5.GetValue(0);
        v_hat_coeff_ = s6.GetValue(0);

        tmpQueue1.FreeTensor(s1);
        tmpQueue1.FreeTensor(s2);
        tmpQueue1.FreeTensor(s3);
        tmpQueue1.FreeTensor(s4);
        tmpQueue1.FreeTensor(s5);
        tmpQueue1.FreeTensor(s6);
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<float> pLocal = inQueueParam.AllocTensor<float>();
        AscendC::LocalTensor<float> gLocal = inQueueGrad.AllocTensor<float>();
        AscendC::LocalTensor<float> mLocal = inQueueM.AllocTensor<float>();
        AscendC::LocalTensor<float> vLocal = inQueueV.AllocTensor<float>();

        AscendC::DataCopy(pLocal, paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gLocal, gradGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(mLocal, mGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(vLocal, vGm[progress * this->tileLength], this->tileLength);

        inQueueParam.EnQue(pLocal);
        inQueueGrad.EnQue(gLocal);
        inQueueM.EnQue(mLocal);
        inQueueV.EnQue(vLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<float> pLocal = inQueueParam.DeQue<float>();
        AscendC::LocalTensor<float> gLocal = inQueueGrad.DeQue<float>();
        AscendC::LocalTensor<float> mLocal = inQueueM.DeQue<float>();
        AscendC::LocalTensor<float> vLocal = inQueueV.DeQue<float>();

        AscendC::LocalTensor<float> outLocal = outQueueParam.AllocTensor<float>();
        AscendC::LocalTensor<float> t1 = tmpQueue1.AllocTensor<float>(); // temp for m update and intermediates
        AscendC::LocalTensor<float> t2 = tmpQueue2.AllocTensor<float>(); // temp for v update and intermediates

        // m_upd = beta1 * m + (1 - beta1) * grad
        AscendC::Muls(t1, mLocal, beta1_, this->tileLength);
        AscendC::Muls(outLocal, gLocal, (1.0f - beta1_), this->tileLength);
        AscendC::Add(t1, t1, outLocal, this->tileLength);

        // v_upd = beta2 * v + (1 - beta2) * (grad^2)
        AscendC::Mul(t2, gLocal, gLocal, this->tileLength);                 // grad^2
        AscendC::Muls(t2, t2, (1.0f - beta2_), this->tileLength);           // (1-beta2)*grad^2
        AscendC::Muls(outLocal, vLocal, beta2_, this->tileLength);          // beta2*v
        AscendC::Add(outLocal, outLocal, t2, this->tileLength);             // v_upd in outLocal temporarily

        // Bias correction: m_hat = m_upd * m_hat_coeff; v_hat = v_upd * v_hat_coeff
        AscendC::Muls(t1, t1, m_hat_coeff_, this->tileLength);              // m_hat in t1
        AscendC::Muls(outLocal, outLocal, v_hat_coeff_, this->tileLength);  // v_hat in outLocal

        // denom = sqrt(v_hat) + eps
        AscendC::Sqrt(t2, outLocal, this->tileLength);
        AscendC::Adds(t2, t2, eps_, this->tileLength);

        // update = lr * (m_hat / denom)
        AscendC::Div(t2, t1, t2, this->tileLength);
        AscendC::Muls(t2, t2, lr_, this->tileLength);

        // param_new = param - update
        AscendC::Sub(outLocal, pLocal, t2, this->tileLength);

        outQueueParam.EnQue(outLocal);

        // free inputs
        inQueueParam.FreeTensor(pLocal);
        inQueueGrad.FreeTensor(gLocal);
        inQueueM.FreeTensor(mLocal);
        inQueueV.FreeTensor(vLocal);
        // free temps
        tmpQueue1.FreeTensor(t1);
        tmpQueue2.FreeTensor(t2);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<float> outLocal = outQueueParam.DeQue<float>();
        AscendC::DataCopy(outParamGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueParam.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueParam, inQueueGrad, inQueueM, inQueueV;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueParam;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue1, tmpQueue2;

    AscendC::GlobalTensor<float> paramGm;
    AscendC::GlobalTensor<float> gradGm;
    AscendC::GlobalTensor<float> mGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> outParamGm;

    AscendC::GlobalTensor<float> beta1Gm, beta2Gm, lrGm, epsGm, mHatCoeffGm, vHatCoeffGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    float beta1_;
    float beta2_;
    float lr_;
    float eps_;
    float m_hat_coeff_;
    float v_hat_coeff_;
};

extern "C" __global__ __aicore__ void adam_custom(GM_ADDR param, GM_ADDR grad, GM_ADDR m, GM_ADDR v,
                                                  GM_ADDR beta1, GM_ADDR beta2, GM_ADDR lr, GM_ADDR eps,
                                                  GM_ADDR m_hat_coeff, GM_ADDR v_hat_coeff,
                                                  GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAdam op;
    op.Init(param, grad, m, v, beta1, beta2, lr, eps, m_hat_coeff, v_hat_coeff,
            tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, beta1=0.9, beta2=0.999, lr=1e-3, eps=1e-8, step=1):
        super().__init__()
        self.beta1 = beta1
        self.beta2 = beta2
        self.lr = lr
        self.eps = eps
        self.step = step

    def forward(self, param, grad, m, v):
        # Bias correction coefficients computed on host
        m_hat_coeff = 1.0 / (1.0 - (self.beta1 ** self.step))
        v_hat_coeff = 1.0 / (1.0 - (self.beta2 ** self.step))
        device = param.device
        dtype = param.dtype
        return custom_ops_lib.adam_custom(
            param, grad, m, v,
            torch.tensor(self.beta1, device=device, dtype=dtype),
            torch.tensor(self.beta2, device=device, dtype=dtype),
            torch.tensor(self.lr, device=device, dtype=dtype),
            torch.tensor(self.eps, device=device, dtype=dtype),
            torch.tensor(m_hat_coeff, device=device, dtype=dtype),
            torch.tensor(v_hat_coeff, device=device, dtype=dtype),
        )
'''
