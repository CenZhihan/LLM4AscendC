kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBnScaleSoftmax {
public:
    __aicore__ inline KernelGemmBnScaleSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR running_mean,
                                GM_ADDR running_var,
                                GM_ADDR bn_weight,
                                GM_ADDR bn_bias,
                                GM_ADDR scale,
                                GM_ADDR eps,
                                GM_ADDR y,
                                uint32_t batch,
                                uint32_t featureLen,
                                uint32_t tileNum,
                                uint32_t segLen,
                                uint32_t scaleLen)
    {
        this->batch = batch;
        this->featureLen = featureLen;
        this->tileNum = tileNum;
        this->segLen = segLen;
        this->scaleLen = scaleLen;

        // Calculate row split among blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blk = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (batch + blocks - 1) / blocks;
        rowStart = blk * rowsPerBlock;
        rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > batch) rowEnd = batch;

        xGm.SetGlobalBuffer((__gm__ float*)x, batch * featureLen);
        meanGm.SetGlobalBuffer((__gm__ float*)running_mean, featureLen);
        varGm.SetGlobalBuffer((__gm__ float*)running_var, featureLen);
        gammaGm.SetGlobalBuffer((__gm__ float*)bn_weight, featureLen);
        betaGm.SetGlobalBuffer((__gm__ float*)bn_bias, featureLen);
        scaleGm.SetGlobalBuffer((__gm__ float*)scale, scaleLen);
        epsGm.SetGlobalBuffer((__gm__ float*)eps, 1);
        yGm.SetGlobalBuffer((__gm__ float*)y, batch * featureLen);

        // init queues
        pipe.InitBuffer(qX, BUFFER_NUM, segLen * sizeof(float));
        pipe.InitBuffer(qMean, BUFFER_NUM, segLen * sizeof(float));
        pipe.InitBuffer(qVar, BUFFER_NUM, segLen * sizeof(float));
        pipe.InitBuffer(qGamma, BUFFER_NUM, segLen * sizeof(float));
        pipe.InitBuffer(qBeta, BUFFER_NUM, segLen * sizeof(float));
        pipe.InitBuffer(qScale, BUFFER_NUM, segLen * sizeof(float));
        pipe.InitBuffer(qY, BUFFER_NUM, segLen * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // load eps scalar
        float epsVal = 0.0f;
        {
            AscendC::LocalTensor<float> tmp = qX.AllocTensor<float>();
            AscendC::DataCopy(tmp, epsGm, 1);
            epsVal = tmp.GetValue(0);
            qX.FreeTensor(tmp);
        }

        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            // Pass 1: BN + scale and compute max
            float rowMax = -__HALF_MAX__;
            uint32_t numSeg = (featureLen + segLen - 1) / segLen;
            for (uint32_t s = 0; s < numSeg; ++s) {
                uint32_t offset = s * segLen;
                uint32_t len = featureLen - offset;
                if (len > segLen) len = segLen;

                AscendC::LocalTensor<float> xLoc = qX.AllocTensor<float>();
                AscendC::LocalTensor<float> meanLoc = qMean.AllocTensor<float>();
                AscendC::LocalTensor<float> varLoc = qVar.AllocTensor<float>();
                AscendC::LocalTensor<float> gammaLoc = qGamma.AllocTensor<float>();
                AscendC::LocalTensor<float> betaLoc = qBeta.AllocTensor<float>();
                AscendC::LocalTensor<float> scaleLoc = qScale.AllocTensor<float>();
                AscendC::LocalTensor<float> yLoc = qY.AllocTensor<float>();

                AscendC::DataCopy(xLoc, xGm[r * featureLen + offset], len);
                AscendC::DataCopy(meanLoc, meanGm[offset], len);
                AscendC::DataCopy(varLoc, varGm[offset], len);
                AscendC::DataCopy(gammaLoc, gammaGm[offset], len);
                AscendC::DataCopy(betaLoc, betaGm[offset], len);

                if (scaleLen == 1) {
                    // broadcast scalar scale to segment
                    // Fill scaleLoc with scalar
                    AscendC::Duplicate(scaleLoc, scaleGm, len); // assume Duplicate fills with scale[0]
                } else {
                    AscendC::DataCopy(scaleLoc, scaleGm[offset], len);
                }

                // tmp = (x - mean)
                AscendC::Sub(yLoc, xLoc, meanLoc, len);
                // var_eps = var + eps
                AscendC::Adds(varLoc, varLoc, epsVal, len);
                // rs = rsqrt(var_eps)
                AscendC::Rsqrt(varLoc, varLoc, len);
                // norm = tmp * rs
                AscendC::Mul(yLoc, yLoc, varLoc, len);
                // affine = norm * gamma + beta
                AscendC::Mul(yLoc, yLoc, gammaLoc, len);
                AscendC::Add(yLoc, yLoc, betaLoc, len);
                // scale
                AscendC::Mul(yLoc, yLoc, scaleLoc, len);

                // compute segment max
                float segMax = AscendC::ReduceMax(yLoc, len);
                rowMax = rowMax > segMax ? rowMax : segMax;

                // store intermediate
                AscendC::DataCopy(yGm[r * featureLen + offset], yLoc, len);

                qX.FreeTensor(xLoc);
                qMean.FreeTensor(meanLoc);
                qVar.FreeTensor(varLoc);
                qGamma.FreeTensor(gammaLoc);
                qBeta.FreeTensor(betaLoc);
                qScale.FreeTensor(scaleLoc);
                qY.FreeTensor(yLoc);
            }

            // Pass 2: exp and sum
            float sumExp = 0.0f;
            for (uint32_t s = 0; s < (featureLen + segLen - 1) / segLen; ++s) {
                uint32_t offset = s * segLen;
                uint32_t len = featureLen - offset;
                if (len > segLen) len = segLen;

                AscendC::LocalTensor<float> yLoc = qY.AllocTensor<float>();
                AscendC::DataCopy(yLoc, yGm[r * featureLen + offset], len);
                // yLoc = exp(yLoc - rowMax)
                AscendC::Adds(yLoc, yLoc, -rowMax, len);
                AscendC::Exp(yLoc, yLoc, len);
                float segSum = AscendC::ReduceSum(yLoc, len);
                sumExp += segSum;
                AscendC::DataCopy(yGm[r * featureLen + offset], yLoc, len);
                qY.FreeTensor(yLoc);
            }

            float invSum = 1.0f / sumExp;
            // Pass 3: normalize
            for (uint32_t s = 0; s < (featureLen + segLen - 1) / segLen; ++s) {
                uint32_t offset = s * segLen;
                uint32_t len = featureLen - offset;
                if (len > segLen) len = segLen;

                AscendC::LocalTensor<float> yLoc = qY.AllocTensor<float>();
                AscendC::DataCopy(yLoc, yGm[r * featureLen + offset], len);
                AscendC::Muls(yLoc, yLoc, invSum, len);
                AscendC::DataCopy(yGm[r * featureLen + offset], yLoc, len);
                qY.FreeTensor(yLoc);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qX, qMean, qVar, qGamma, qBeta, qScale;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> qY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> meanGm;
    AscendC::GlobalTensor<float> varGm;
    AscendC::GlobalTensor<float> gammaGm;
    AscendC::GlobalTensor<float> betaGm;
    AscendC::GlobalTensor<float> scaleGm;
    AscendC::GlobalTensor<float> epsGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t batch;
    uint32_t featureLen;
    uint32_t tileNum;
    uint32_t segLen;
    uint32_t scaleLen;

    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void gemm_batch_norm_scaling_softmax_custom(GM_ADDR x,
                                                                             GM_ADDR running_mean,
                                                                             GM_ADDR running_var,
                                                                             GM_ADDR bn_weight,
                                                                             GM_ADDR bn_bias,
                                                                             GM_ADDR scale,
                                                                             GM_ADDR eps,
                                                                             GM_ADDR y,
                                                                             GM_ADDR workspace,
                                                                             GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBnScaleSoftmax op;
    op.Init(x, running_mean, running_var, bn_weight, bn_bias, scale, eps, y,
            tiling_data.batch(), tiling_data.featureLen(), tiling_data.tileNum(),
            tiling_data.segLen(), tiling_data.scaleLen());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: calls fused custom AscendC kernel for BatchNorm + scaling + softmax
    following a GEMM (Linear) layer.
    """
    def __init__(self, in_features, out_features, bn_eps=1e-5, bn_momentum=0.1, scale_shape=(1,)):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.bn = torch.nn.BatchNorm1d(out_features, eps=bn_eps, momentum=bn_momentum)
        self.scale = torch.nn.Parameter(torch.ones(scale_shape))
        self.softmax_dim = 1

    def forward(self, x):
        x = self.gemm(x)
        eps_tensor = torch.tensor([self.bn.eps], dtype=x.dtype, device=x.device)
        y = custom_ops_lib.gemm_batch_norm_scaling_softmax_custom(
            x,
            self.bn.running_mean,
            self.bn.running_var,
            self.bn.weight,
            self.bn.bias,
            self.scale,
            eps_tensor
        )
        return y
