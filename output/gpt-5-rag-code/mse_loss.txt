kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelMseLoss {
public:
    __aicore__ inline KernelMseLoss() {}
    __aicore__ inline void Init(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        // Each block processes blockLength elements; split into tiles for double-buffering
        this->tileLength = this->blockLength / (tileNum * BUFFER_NUM);

        predGm.SetGlobalBuffer((__gm__ DTYPE_X *)predictions + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        tgtGm.SetGlobalBuffer((__gm__ DTYPE_Y *)targets + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        lossGm.SetGlobalBuffer((__gm__ DTYPE_Z *)loss, 1); // scalar output

        pipe.InitBuffer(inQueuePred, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueTgt,  BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(reduceQueue, BUFFER_NUM, sizeof(DTYPE_Z)); // for tile partial sum (1 element)
    }

    __aicore__ inline void Process()
    {
        // Accumulate sum of squared errors within this block
        float accSum = 0.0f;
        int32_t loopCount = this->tileNum * BUFFER_NUM;

        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            accSum += ComputeTile(i);
        }

        // Only one block is scheduled (BLOCK_DIM = 1), write final mean
        float mean = 0.0f;
        if (this->blockLength > 0) {
            mean = accSum / static_cast<float>(this->blockLength);
        }
        lossGm.SetValue(0, static_cast<DTYPE_Z>(mean));
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> predLocal = inQueuePred.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> tgtLocal  = inQueueTgt.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(predLocal, predGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(tgtLocal,  tgtGm[progress * this->tileLength],  this->tileLength);
        inQueuePred.EnQue(predLocal);
        inQueueTgt.EnQue(tgtLocal);
    }

    __aicore__ inline float ComputeTile(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> predLocal = inQueuePred.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> tgtLocal  = inQueueTgt.DeQue<DTYPE_Y>();
        // In-place: predLocal = (predLocal - tgtLocal)^2
        AscendC::Sub(predLocal, predLocal, tgtLocal, this->tileLength);
        AscendC::Mul(predLocal, predLocal, predLocal, this->tileLength);

        AscendC::LocalTensor<DTYPE_Z> tileSumLocal = reduceQueue.AllocTensor<DTYPE_Z>();
        // Reduce sum over tile
        AscendC::ReduceSum(tileSumLocal, predLocal, this->tileLength);
        float tileSum = static_cast<float>(tileSumLocal.GetValue(0));

        // Free locals
        reduceQueue.FreeTensor(tileSumLocal);
        inQueuePred.FreeTensor(predLocal);
        inQueueTgt.FreeTensor(tgtLocal);

        return tileSum;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePred, inQueueTgt;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> reduceQueue;

    AscendC::GlobalTensor<DTYPE_X> predGm;
    AscendC::GlobalTensor<DTYPE_Y> tgtGm;
    AscendC::GlobalTensor<DTYPE_Z> lossGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void mse_loss_custom(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMseLoss op;
    op.Init(predictions, targets, loss, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
