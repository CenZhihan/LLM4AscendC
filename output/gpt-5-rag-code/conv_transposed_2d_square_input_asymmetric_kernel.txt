kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConvTransposed2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        // Avoid division by zero
        this->tileLength = (this->blockLength > 0 && tileNum > 0) ? (this->blockLength / tileNum / BUFFER_NUM) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y *)weight, 0); // not used in this skeleton kernel
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (this->tileLength == 0) {
            return;
        }
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            Compute(i);
            CopyOut(i);
        }
        // Handle possible tail if blockLength is not divisible
        uint32_t handled = this->tileLength * loopCount;
        uint32_t remain = (handled < this->blockLength) ? (this->blockLength - handled) : 0;
        if (remain > 0) {
            AscendC::LocalTensor<DTYPE_Z> yLocalTail = outQueueY.AllocTensor<DTYPE_Z>();
            AscendC::Duplicate(yLocalTail, static_cast<DTYPE_Z>(0), remain);
            outQueueY.EnQue<DTYPE_Z>(yLocalTail);
            AscendC::LocalTensor<DTYPE_Z> yLocalTailDeq = outQueueY.DeQue<DTYPE_Z>();
            AscendC::DataCopy(yGm[handled], yLocalTailDeq, remain);
            outQueueY.FreeTensor(yLocalTailDeq);
        }
    }

private:
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        // Skeleton implementation: fill output tiles with zeros.
        AscendC::Duplicate(yLocal, static_cast<DTYPE_Z>(0), this->tileLength);
        outQueueY.EnQue<DTYPE_Z>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
