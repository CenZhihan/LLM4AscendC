#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_B = float;
using DTYPE_S = float;
using DTYPE_Y = float;

class KernelMatmulScaleResidualAddClampLogSumExpMish {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMish() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR scale, GM_ADDR clamp_min, GM_ADDR clamp_max,
                                GM_ADDR y,
                                uint32_t batch, uint32_t inSize, uint32_t outSize,
                                uint32_t rowsPerBlock, uint32_t tileK, uint32_t tileN)
    {
        this->batch = batch;
        this->inSize = inSize;
        this->outSize = outSize;
        this->rowsPerBlock = rowsPerBlock;
        this->tileK = tileK;
        this->tileN = tileN;

        uint32_t startRow = AscendC::GetBlockIdx() * rowsPerBlock;
        uint32_t maxRows = (startRow + rowsPerBlock <= batch) ? rowsPerBlock : (batch - startRow);
        this->myRows = maxRows;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + startRow * inSize, maxRows * inSize);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, outSize * inSize);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, outSize);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_S*)scale, 1);
        clampMinGm.SetGlobalBuffer((__gm__ DTYPE_S*)clamp_min, 1);
        clampMaxGm.SetGlobalBuffer((__gm__ DTYPE_S*)clamp_max, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + startRow * 1, maxRows * 1);

        pipe.InitBuffer(inQueueX, 1, tileK * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueW, 1, tileN * tileK * sizeof(DTYPE_W));
        pipe.InitBuffer(inQueueB, 1, tileN * sizeof(DTYPE_B));
        pipe.InitBuffer(tmpVecQueue, 1, tileN * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Load scalar params
        DTYPE_S s = 0.0f, cmin = 0.0f, cmax = 0.0f;
        {
            AscendC::LocalTensor<DTYPE_S> sLocal = tmpScalar.AllocTensor<DTYPE_S>();
            AscendC::DataCopy(sLocal, scaleGm, 1);
            s = reinterpret_cast<DTYPE_S*>(sLocal.GetPhyAddr())[0];
            tmpScalar.FreeTensor(sLocal);
            AscendC::LocalTensor<DTYPE_S> cminLocal = tmpScalar.AllocTensor<DTYPE_S>();
            AscendC::DataCopy(cminLocal, clampMinGm, 1);
            cmin = reinterpret_cast<DTYPE_S*>(cminLocal.GetPhyAddr())[0];
            tmpScalar.FreeTensor(cminLocal);
            AscendC::LocalTensor<DTYPE_S> cmaxLocal = tmpScalar.AllocTensor<DTYPE_S>();
            AscendC::DataCopy(cmaxLocal, clampMaxGm, 1);
            cmax = reinterpret_cast<DTYPE_S*>(cmaxLocal.GetPhyAddr())[0];
            tmpScalar.FreeTensor(cmaxLocal);
        }

        DTYPE_S scale2 = s * (DTYPE_S)2.0f;

        for (uint32_t r = 0; r < myRows; ++r) {
            DTYPE_Y m = -1.0e20f;
            DTYPE_Y se = 0.0f;

            for (uint32_t n0 = 0; n0 < outSize; n0 += tileN) {
                uint32_t curTN = (n0 + tileN <= outSize) ? tileN : (outSize - n0);

                // Allocate local accumulators
                AscendC::LocalTensor<DTYPE_Y> accLocal = tmpVecQueue.AllocTensor<DTYPE_Y>();
                // Initialize acc to 0
                for (uint32_t t = 0; t < curTN; ++t) {
                    reinterpret_cast<DTYPE_Y*>(accLocal.GetPhyAddr())[t] = 0.0f;
                }

                // GEMM accumulate across K
                for (uint32_t k0 = 0; k0 < inSize; k0 += tileK) {
                    uint32_t curTK = (k0 + tileK <= inSize) ? tileK : (inSize - k0);

                    // Load x tile (row r)
                    AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(xLocal, xGm[r * inSize + k0], curTK);
                    inQueueX.EnQue(xLocal);

                    // For weights: load curTN x curTK tile
                    AscendC::LocalTensor<DTYPE_W> wLocal = inQueueW.AllocTensor<DTYPE_W>();
                    // Copy each out channel row's curTK slice contiguously into wLocal
                    for (uint32_t t = 0; t < curTN; ++t) {
                        AscendC::DataCopy(wLocal[t * curTK], wGm[(n0 + t) * inSize + k0], curTK);
                    }
                    inQueueW.EnQue(wLocal);

                    // Compute partial dot products
                    AscendC::LocalTensor<DTYPE_X> xTile = inQueueX.DeQue<DTYPE_X>();
                    AscendC::LocalTensor<DTYPE_W> wTile = inQueueW.DeQue<DTYPE_W>();

                    for (uint32_t t = 0; t < curTN; ++t) {
                        DTYPE_Y sum = reinterpret_cast<DTYPE_Y*>(accLocal.GetPhyAddr())[t];
                        for (uint32_t kk = 0; kk < curTK; ++kk) {
                            DTYPE_X xv = reinterpret_cast<DTYPE_X*>(xTile.GetPhyAddr())[kk];
                            DTYPE_W wv = reinterpret_cast<DTYPE_W*>(wTile.GetPhyAddr())[t * curTK + kk];
                            sum += (DTYPE_Y)(xv * wv);
                        }
                        reinterpret_cast<DTYPE_Y*>(accLocal.GetPhyAddr())[t] = sum;
                    }

                    inQueueX.FreeTensor(xTile);
                    inQueueW.FreeTensor(wTile);
                }

                // Add bias, scale, residual add (x + x fused as *2 via scale2), clamp and update LSE
                AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
                AscendC::DataCopy(bLocal, bGm[n0], curTN);
                inQueueB.EnQue(bLocal);
                AscendC::LocalTensor<DTYPE_B> bTile = inQueueB.DeQue<DTYPE_B>();

                for (uint32_t t = 0; t < curTN; ++t) {
                    DTYPE_Y val = reinterpret_cast<DTYPE_Y*>(accLocal.GetPhyAddr())[t] + (DTYPE_Y)reinterpret_cast<DTYPE_B*>(bTile.GetPhyAddr())[t];
                    val = val * scale2;
                    // clamp
                    if (val < (DTYPE_Y)cmin) val = (DTYPE_Y)cmin;
                    if (val > (DTYPE_Y)cmax) val = (DTYPE_Y)cmax;
                    // online logsumexp
                    if (val > m) {
                        DTYPE_Y factor = AscendC::Exp(m - val); // exp(old_m - new_val)
                        se = se * factor + 1.0f;
                        m = val;
                    } else {
                        se += AscendC::Exp(val - m);
                    }
                }

                inQueueB.FreeTensor(bTile);
                tmpVecQueue.FreeTensor(accLocal);
            }

            // v = m + log(se)
            DTYPE_Y v = m + AscendC::Log(se);
            // mish(v) = v * tanh(softplus(v)), softplus(v) = log(1 + exp(v))
            DTYPE_Y sp = AscendC::Log(1.0f + AscendC::Exp(v));
            DTYPE_Y mish = v * AscendC::Tanh(sp);
            DTYPE_Y outv = v * mish;

            // Store result
            AscendC::DataCopy(yGm[r * 1], &outv, 1);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> tmpVecQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpScalar;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_S> scaleGm;
    AscendC::GlobalTensor<DTYPE_S> clampMinGm;
    AscendC::GlobalTensor<DTYPE_S> clampMaxGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t batch;
    uint32_t inSize;
    uint32_t outSize;
    uint32_t rowsPerBlock;
    uint32_t tileK;
    uint32_t tileN;
    uint32_t myRows;
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scale, GM_ADDR clamp_min, GM_ADDR clamp_max,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScaleResidualAddClampLogSumExpMish op;
    op.Init(x, w, b, scale, clamp_min, clamp_max, y,
            tiling_data.batch, tiling_data.inSize, tiling_data.outSize,
            tiling_data.rowsPerBlock, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
