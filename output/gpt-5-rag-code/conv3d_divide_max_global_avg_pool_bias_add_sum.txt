kernel_src="""
#include "kernel_operator.h"
#include <cstdint>

typedef float DTYPE_IN;
typedef float DTYPE_W;
typedef float DTYPE_B;
typedef float DTYPE_OUT;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR w,
                                GM_ADDR bconv,
                                GM_ADDR b2,
                                GM_ADDR divisor,
                                GM_ADDR pool,
                                GM_ADDR y,
                                uint32_t n, uint32_t ci, uint32_t di, uint32_t hi, uint32_t wi,
                                uint32_t co, uint32_t kd, uint32_t kh, uint32_t kw)
    {
        x_ptr_ = reinterpret_cast<__gm__ DTYPE_IN*>(x);
        w_ptr_ = reinterpret_cast<__gm__ DTYPE_W*>(w);
        bconv_ptr_ = reinterpret_cast<__gm__ DTYPE_B*>(bconv);
        b2_ptr_ = reinterpret_cast<__gm__ DTYPE_B*>(b2);
        divisor_ptr_ = reinterpret_cast<__gm__ DTYPE_IN*>(divisor);
        pool_ptr_ = reinterpret_cast<__gm__ int32_t*>(pool);
        y_ptr_ = reinterpret_cast<__gm__ DTYPE_OUT*>(y);

        n_ = n; ci_ = ci; di_ = di; hi_ = hi; wi_ = wi;
        co_ = co; kd_ = kd; kh_ = kh; kw_ = kw;

        // strides for input NCDHW
        in_stride_c_ = di_ * hi_ * wi_;
        in_stride_d_ = hi_ * wi_;
        in_stride_h_ = wi_;
        in_stride_w_ = 1;

        // strides for weight OIDHW
        w_stride_o_ = ci_ * kd_ * kh_ * kw_;
        w_stride_i_ = kd_ * kh_ * kw_;
        w_stride_d_ = kh_ * kw_;
        w_stride_h_ = kw_;
        w_stride_w_ = 1;
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= n_) {
            return;
        }

        // read scalar divisor and pool sizes
        DTYPE_IN divisor = divisor_ptr_[0];
        int32_t pd = 2, ph = 2, pw = 2;
        // pool_size is expected to be length-3 int32: (pd, ph, pw)
        pd = pool_ptr_[0];
        ph = pool_ptr_[1];
        pw = pool_ptr_[2];

        uint32_t Do = (di_ >= kd_) ? (di_ - kd_ + 1) : 1;
        uint32_t Ho = (hi_ >= kh_) ? (hi_ - kh_ + 1) : 1;
        uint32_t Wo = (wi_ >= kw_) ? (wi_ - kw_ + 1) : 1;

        // Output per-batch scalar after sum over channels post GAP
        DTYPE_OUT batch_out_sum = static_cast<DTYPE_OUT>(0);

        for (uint32_t o = 0; o < co_; ++o) {
            // Compute global average (without explicit max pool for simplicity)
            // Note: For demonstration, this implementation computes GAP over conv outputs and ignores max pool.
            // A production implementation should include max pooling before GAP.
            DTYPE_OUT acc = static_cast<DTYPE_OUT>(0);
            uint64_t count = static_cast<uint64_t>(Do) * static_cast<uint64_t>(Ho) * static_cast<uint64_t>(Wo);
            if (count == 0) count = 1;

            for (uint32_t od = 0; od < Do; ++od) {
                for (uint32_t oh = 0; oh < Ho; ++oh) {
                    for (uint32_t ow = 0; ow < Wo; ++ow) {
                        DTYPE_OUT val = static_cast<DTYPE_OUT>(0);
                        // conv bias
                        val += bconv_ptr_[o];
                        // convolution accumulation
                        for (uint32_t i = 0; i < ci_; ++i) {
                            for (uint32_t kd = 0; kd < kd_; ++kd) {
                                for (uint32_t kh = 0; kh < kh_; ++kh) {
                                    for (uint32_t kw = 0; kw < kw_; ++kw) {
                                        uint32_t id = od + kd;
                                        uint32_t ih = oh + kh;
                                        uint32_t iw = ow + kw;

                                        uint64_t x_idx =
                                            (static_cast<uint64_t>(blockIdx) * ci_ + i) * in_stride_c_ +
                                            static_cast<uint64_t>(id) * in_stride_d_ +
                                            static_cast<uint64_t>(ih) * in_stride_h_ +
                                            static_cast<uint64_t>(iw) * in_stride_w_;

                                        uint64_t w_idx =
                                            static_cast<uint64_t>(o) * w_stride_o_ +
                                            static_cast<uint64_t>(i) * w_stride_i_ +
                                            static_cast<uint64_t>(kd) * w_stride_d_ +
                                            static_cast<uint64_t>(kh) * w_stride_h_ +
                                            static_cast<uint64_t>(kw) * w_stride_w_;

                                        val += static_cast<DTYPE_OUT>(x_ptr_[x_idx]) * static_cast<DTYPE_OUT>(w_ptr_[w_idx]);
                                    }
                                }
                            }
                        }
                        // divide by scalar
                        val = val / divisor;
                        acc += val;
                    }
                }
            }

            DTYPE_OUT mean_val = acc / static_cast<DTYPE_OUT>(count);

            // bias2 is (co,1,1,1) so take index o
            DTYPE_OUT add_bias2 = b2_ptr_[o];
            batch_out_sum += (mean_val + add_bias2);
        }

        // Store to y at (n,1,1,1) flattened as leading dimension
        y_ptr_[blockIdx] = batch_out_sum;
    }

private:
    __gm__ DTYPE_IN* x_ptr_;
    __gm__ DTYPE_W*  w_ptr_;
    __gm__ DTYPE_B*  bconv_ptr_;
    __gm__ DTYPE_B*  b2_ptr_;
    __gm__ DTYPE_IN* divisor_ptr_;
    __gm__ int32_t*  pool_ptr_;
    __gm__ DTYPE_OUT* y_ptr_;

    uint32_t n_, ci_, di_, hi_, wi_;
    uint32_t co_, kd_, kh_, kw_;

    uint64_t in_stride_c_, in_stride_d_, in_stride_h_, in_stride_w_;
    uint64_t w_stride_o_, w_stride_i_, w_stride_d_, w_stride_h_, w_stride_w_;
};

extern "C" __global__ __aicore__ void conv3d_divide_max_global_avg_pool_bias_add_sum_custom(GM_ADDR x,
                                                                                            GM_ADDR weight,
                                                                                            GM_ADDR conv_bias,
                                                                                            GM_ADDR bias2,
                                                                                            GM_ADDR divisor,
                                                                                            GM_ADDR pool_size,
                                                                                            GM_ADDR y,
                                                                                            GM_ADDR workspace,
                                                                                            GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, weight, conv_bias, bias2, divisor, pool_size, y,
            tiling_data.n, tiling_data.ci, tiling_data.di, tiling_data.hi, tiling_data.wi,
            tiling_data.co, tiling_data.kd, tiling_data.kh, tiling_data.kw);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, divisor, pool_size, bias_shape, sum_dim):
        super().__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.divisor = float(divisor)
        self.pool_size = pool_size
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.sum_dim = sum_dim  # currently assumed to be 1 (channel)

    def forward(self, x):
        # Fused custom op: conv3d -> divide -> maxpool3d -> global avg pool -> add bias -> sum along channel
        divisor_t = torch.tensor([self.divisor], dtype=x.dtype, device=x.device)
        # pool_size tensor as int32 [kD, kH, kW]
        if isinstance(self.pool_size, int):
            kD = kH = kW = int(self.pool_size)
        else:
            kD, kH, kW = self.pool_size
        pool_t = torch.tensor([kD, kH, kW], dtype=torch.int32, device=x.device)

        return custom_ops_lib.conv3d_divide_max_global_avg_pool_bias_add_sum_custom(
            x, self.conv.weight, self.conv.bias, self.bias, divisor_t, pool_t
