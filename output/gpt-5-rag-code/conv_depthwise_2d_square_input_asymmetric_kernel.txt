kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_B = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelDepthwiseConv {
public:
    __aicore__ inline KernelDepthwiseConv() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR tiling)
    {
        GET_TILING_DATA(tiling_data, tiling);
        n = tiling_data.n;
        c = tiling_data.c;
        h = tiling_data.h;
        wIn = tiling_data.w;
        kH = tiling_data.kH;
        strideH = tiling_data.strideH;
        strideW = tiling_data.strideW;
        padH = tiling_data.padH;
        padW = tiling_data.padW;
        dilH = tiling_data.dilateH;
        dilW = tiling_data.dilateW;
        outH = tiling_data.outH;
        outW = tiling_data.outW;
        hasBias = tiling_data.hasBias != 0 ? 1 : 0;

        totalLen = tiling_data.totalLength;
        tileNum = tiling_data.tileNum;
        blockLen = totalLen / AscendC::GetBlockNum();
        startIdx = blockLen * AscendC::GetBlockIdx();
        tileLen = blockLen / (tileNum * BUFFER_NUM);
        if (tileLen == 0) tileLen = blockLen; // fallback for small sizes

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, n * c * h * wIn);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, c * kH);
        if (hasBias) {
            bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, c);
        }
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, n * c * outH * outW);

        pipe.InitBuffer(tmpQueueX, BUFFER_NUM, sizeof(DTYPE_X)); // scalar buffers
        pipe.InitBuffer(tmpQueueW, BUFFER_NUM, sizeof(DTYPE_W));
        pipe.InitBuffer(tmpQueueB, BUFFER_NUM, sizeof(DTYPE_B));
        pipe.InitBuffer(tmpQueueY, BUFFER_NUM, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loops = (blockLen + tileLen - 1) / tileLen;
        for (int32_t lp = 0; lp < loops; ++lp) {
            uint32_t curTileBase = startIdx + lp * tileLen;
            uint32_t curTileSize = AscendC::min(tileLen, blockLen - lp * tileLen);
            for (uint32_t i = 0; i < curTileSize; ++i) {
                uint32_t outIndex = curTileBase + i;
                ComputeOne(outIndex);
            }
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t outLinearIndex)
    {
        // Decode to (n, c, oh, ow)
        uint32_t outHW = outH * outW;
        uint32_t batchStride = c * outHW;

        uint32_t nIdx = outLinearIndex / batchStride;
        uint32_t rem1 = outLinearIndex - nIdx * batchStride;
        uint32_t cIdx = rem1 / outHW;
        uint32_t rem2 = rem1 - cIdx * outHW;
        uint32_t oh = rem2 / outW;
        uint32_t ow = rem2 - oh * outW;

        // Compute input base indices
        int32_t in_h0 = static_cast<int32_t>(oh) * static_cast<int32_t>(strideH) - static_cast<int32_t>(padH);
        int32_t in_w0 = static_cast<int32_t>(ow) * static_cast<int32_t>(strideW) - static_cast<int32_t>(padW);

        float acc = 0.0f;

        // width kernel is 1, only check single in_w
        bool w_valid = (in_w0 >= 0) && (in_w0 < static_cast<int32_t>(wIn));
        if (w_valid) {
            for (uint32_t kh = 0; kh < kH; ++kh) {
                int32_t in_h = in_h0 + static_cast<int32_t>(kh) * static_cast<int32_t>(dilH);
                if (in_h >= 0 && in_h < static_cast<int32_t>(h)) {
                    // x index: (((nIdx * C + cIdx) * H + in_h) * W + in_w0)
                    uint32_t xIndex = (((nIdx * c + cIdx) * h + static_cast<uint32_t>(in_h)) * wIn + static_cast<uint32_t>(in_w0));
                    // w index: cIdx * kH + kh
                    uint32_t wIndex = cIdx * kH + kh;

                    // Load one x
                    AscendC::LocalTensor<DTYPE_X> xScalar = tmpQueueX.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(xScalar, xGm[xIndex], 1);
                    tmpQueueX.EnQue<DTYPE_X>(xScalar);
                    xScalar = tmpQueueX.DeQue<DTYPE_X>();
                    float xv = static_cast<float>(xScalar.GetValue(0));
                    tmpQueueX.FreeTensor(xScalar);

                    // Load one w
                    AscendC::LocalTensor<DTYPE_W> wScalar = tmpQueueW.AllocTensor<DTYPE_W>();
                    AscendC::DataCopy(wScalar, wGm[wIndex], 1);
                    tmpQueueW.EnQue<DTYPE_W>(wScalar);
                    wScalar = tmpQueueW.DeQue<DTYPE_W>();
                    float wv = static_cast<float>(wScalar.GetValue(0));
                    tmpQueueW.FreeTensor(wScalar);

                    acc += xv * wv;
                }
            }
        }

        if (hasBias) {
            // Load bias[cIdx]
            AscendC::LocalTensor<DTYPE_B> bScalar = tmpQueueB.AllocTensor<DTYPE_B>();
            AscendC::DataCopy(bScalar, bGm[cIdx], 1);
            tmpQueueB.EnQue<DTYPE_B>(bScalar);
            bScalar = tmpQueueB.DeQue<DTYPE_B>();
            acc += static_cast<float>(bScalar.GetValue(0));
            tmpQueueB.FreeTensor(bScalar);
        }

        // Store result
        AscendC::LocalTensor<DTYPE_Y> yScalar = tmpQueueY.AllocTensor<DTYPE_Y>();
        yScalar.SetValue(0, static_cast<DTYPE_Y>(acc));
        tmpQueueY.EnQue<DTYPE_Y>(yScalar);
        yScalar = tmpQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[outLinearIndex], yScalar, 1);
        tmpQueueY.FreeTensor(yScalar);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> tmpQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> tmpQueueW;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> tmpQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, h, wIn;
    uint32_t kH;
    uint32_t strideH, strideW;
    uint32_t padH, padW;
    uint32_t dilH, dilW;
    uint32_t outH, outW;
    uint32_t hasBias;

    uint32_t totalLen;
    uint32_t tileNum;
    uint32_t blockLen;
    uint32_t startIdx;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    KernelDepthwiseConv op;
    op.Init(x, w, b, y, tiling);
    op.Process();
}

    return {v[0], v[1]};

at::Tensor conv_depthwise_2d_square_input_asymmetric_kernel_custom_impl_npu(
    const at::Tensor& x,
    const at::Tensor& w,
    const c10::optional<at::Tensor>& b_opt,
    c10::IntArrayRef stride,
    c10::IntArrayRef padding,
    c10::IntArrayRef dilation) {

    TORCH_CHECK(x.dim() == 4, "x must be NCHW");
    TORCH_CHECK(w.dim() == 4, "w must be [C,1,kH,1]");

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom depthwise 2D convolution using AscendC kernel specialized for square inputs with asymmetric kernel (kH,1).
    """
    def __init__(self, in_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        # keep standard module to own weights/bias, but we will call custom kernel in forward
        self.conv2d = torch.nn.Conv2d(
            in_channels, in_channels,
            kernel_size=(kernel_size, 1),
            stride=stride,
            padding=padding,
            dilation=dilation,
            groups=in_channels,
            bias=bias
        )

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_depthwise_2d_square_input_asymmetric_kernel_custom(
            x, self.conv2d.weight, self.conv2d.bias,
            self.conv2d.stride, self.conv2d.padding, self.conv2d.dilation
        )

def get_inputs():
    # This mirrors the original test harness
    batch_size = 64
    in_channels = 8
    width = 512
    height = 512
    x = torch.rand(batch_size, in_channels, height, width)
    return [x]

def get_init_inputs():
    in_channels = 8
    kernel_size = 3
    stride = 1
