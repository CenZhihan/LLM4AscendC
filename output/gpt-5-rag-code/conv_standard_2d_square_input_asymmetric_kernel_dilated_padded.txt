#include "kernel_operator.h"

extern "C" __global__ __aicore__
void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(GM_ADDR x,
                                                                           GM_ADDR w,
                                                                           GM_ADDR b,
                                                                           GM_ADDR y,
                                                                           GM_ADDR workspace,
                                                                           GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N   = tiling_data.N;
    const uint32_t Cin = tiling_data.Cin;
    const uint32_t H   = tiling_data.H;
    const uint32_t W   = tiling_data.W;
    const uint32_t Cout= tiling_data.Cout;
    const uint32_t Kh  = tiling_data.Kh;
    const uint32_t Kw  = tiling_data.Kw;
    const uint32_t Sh  = tiling_data.Sh;
    const uint32_t Sw  = tiling_data.Sw;
    const uint32_t Ph  = tiling_data.Ph;
    const uint32_t Pw  = tiling_data.Pw;
    const uint32_t Dh  = tiling_data.Dh;
    const uint32_t Dw  = tiling_data.Dw;
    const uint32_t Hout= tiling_data.Hout;
    const uint32_t Wout= tiling_data.Wout;

    const __gm__ float* xgm = reinterpret_cast<const __gm__ float*>(x);
    const __gm__ float* wgm = reinterpret_cast<const __gm__ float*>(w);
    const __gm__ float* bgm = reinterpret_cast<const __gm__ float*>(b);
    __gm__ float* ygm = reinterpret_cast<__gm__ float*>(y);

    // Strides for NCHW
    uint64_t strideN_x = (uint64_t)Cin * H * W;
    uint64_t strideC_x = (uint64_t)H * W;
    uint64_t strideH_x = W;
    uint64_t strideW_x = 1;

    // Weight OIHW
    uint64_t strideO_w = (uint64_t)Cin * Kh * Kw;
    uint64_t strideI_w = (uint64_t)Kh * Kw;
    uint64_t strideH_w = Kw;
    uint64_t strideW_w = 1;

    // Output NCHW
    uint64_t strideN_y = (uint64_t)Cout * Hout * Wout;
    uint64_t strideC_y = (uint64_t)Hout * Wout;
    uint64_t strideH_y = Wout;
    uint64_t strideW_y = 1;

    // Simple, single-block naive convolution
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t oc = 0; oc < Cout; ++oc) {
            float bias_val = bgm ? bgm[oc] : 0.0f;
            for (uint32_t oh = 0; oh < Hout; ++oh) {
                for (uint32_t ow = 0; ow < Wout; ++ow) {
                    float acc = 0.0f;
                    for (uint32_t ic = 0; ic < Cin; ++ic) {
                        for (uint32_t kh = 0; kh < Kh; ++kh) {
                            int64_t ih = (int64_t)oh * (int64_t)Sh - (int64_t)Ph + (int64_t)kh * (int64_t)Dh;
                            if (ih < 0 || ih >= (int64_t)H) continue;
                            for (uint32_t kw = 0; kw < Kw; ++kw) {
                                int64_t iw = (int64_t)ow * (int64_t)Sw - (int64_t)Pw + (int64_t)kw * (int64_t)Dw;
                                if (iw < 0 || iw >= (int64_t)W) continue;

                                uint64_t x_idx = (uint64_t)n * strideN_x + (uint64_t)ic * strideC_x + (uint64_t)ih * strideH_x + (uint64_t)iw * strideW_x;
                                uint64_t w_idx = (uint64_t)oc * strideO_w + (uint64_t)ic * strideI_w + (uint64_t)kh * strideH_w + (uint64_t)kw * strideW_w;
                                float xv = xgm[x_idx];
                                float wv = wgm[w_idx];
                                acc += xv * wv;
                            }
                        }
                    }
                    acc += bias_val;
                    uint64_t y_idx = (uint64_t)n * strideN_y + (uint64_t)oc * strideC_y + (uint64_t)oh * strideH_y + (uint64_t)ow * strideW_y;
                    ygm[y_idx] = acc;
                }
            }
        }
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom AscendC-optimized 2D convolution with dilation and asymmetric padding.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: tuple, stride: int = 1, padding: tuple = (0, 0), dilation: tuple = (1, 1), bias: bool = False):
        super().__init__()
        # Use nn.Conv2d only to hold weights/bias as parameters
        if isinstance(stride, int):
            stride = (stride, stride)
        self.stride = stride
        self.padding = padding
        self.dilation = dilation
        self.conv2d = torch.nn.Conv2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        b = self.conv2d.bias
        if b is None:
            # Provide zero bias to match required input of custom op
            b = torch.zeros(self.conv2d.weight.size(0), dtype=self.conv2d.weight.dtype, device=self.conv2d.weight.device)
        # Call custom op: expects x, w, bias, strides, pads, dilations
        return custom_ops_lib.conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
            x, self.conv2d.weight, b, list(self.stride), list(self.padding), list(self.dilation)
