    uint32_t c_out = (uint32_t)pw_origin.GetDim(0);

    uint32_t stride_h = 1, stride_w = 1;
    uint32_t dilation_h = 1, dilation_w = 1;
    // heuristically set padding to half kernel to keep output size same when stride=1
    uint32_t pad_h = k_h > 0 ? (k_h - 1) / 2 : 0;
    uint32_t pad_w = k_w > 0 ? (k_w - 1) / 2 : 0;

    out.SetDim(0, xN);
    out.SetDim(1, cOut);
    out.SetDim(2, outH);
    out.SetDim(3, outW);

kernel_src="""
#include "kernel_operator.h"

// A simple, reference-style implementation of depthwise separable 2D convolution:
// - Depthwise: [N, C_in, H, W] * [C_in, 1, K_h, K_w] -> [N, C_in, H_out, W_out]
// - Pointwise: [N, C_in, H_out, W_out] * [C_out, C_in, 1, 1] -> [N, C_out, H_out, W_out]
// Assumptions for simplicity: float32 data type, NCHW layout, stride/dilation/padding from tiling.

extern \"C\" __global__ __aicore__ void conv_depthwise_separable_2d_custom(GM_ADDR x,
                                                                           GM_ADDR depthwise_filter,
                                                                           GM_ADDR pointwise_filter,
                                                                           GM_ADDR y,
                                                                           GM_ADDR workspace,
                                                                           GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.n();
    const uint32_t C_in = tiling_data.c_in();
    const uint32_t H = tiling_data.h();
    const uint32_t W = tiling_data.w();
    const uint32_t C_out = tiling_data.c_out();
    const uint32_t K_h = tiling_data.k_h();
    const uint32_t K_w = tiling_data.k_w();
    const uint32_t stride_h = tiling_data.stride_h();
    const uint32_t stride_w = tiling_data.stride_w();
    const uint32_t pad_h = tiling_data.pad_h();
    const uint32_t pad_w = tiling_data.pad_w();
    const uint32_t dilation_h = tiling_data.dilation_h();
    const uint32_t dilation_w = tiling_data.dilation_w();
    const uint32_t H_out = tiling_data.out_h();
    const uint32_t W_out = tiling_data.out_w();

    float* in_ptr = reinterpret_cast<float*>(x);
    float* dw_ptr = reinterpret_cast<float*>(depthwise_filter);
    float* pw_ptr = reinterpret_cast<float*>(pointwise_filter);
    float* out_ptr = reinterpret_cast<float*>(y);

    auto in_index = [=](uint32_t n, uint32_t c, uint32_t h, uint32_t w) -> uint64_t {
        return ((uint64_t)n * C_in * H * W) + ((uint64_t)c * H * W) + ((uint64_t)h * W) + w;
    };
    auto out_index = [=](uint32_t n, uint32_t c, uint32_t h, uint32_t w) -> uint64_t {
        return ((uint64_t)n * C_out * H_out * W_out) + ((uint64_t)c * H_out * W_out) + ((uint64_t)h * W_out) + w;
    };
    auto dw_index = [=](uint32_t c, uint32_t kh, uint32_t kw) -> uint64_t {
        // [C_in, 1, K_h, K_w]
        return ((uint64_t)c * K_h * K_w) + ((uint64_t)kh * K_w) + kw;
    };
    auto pw_index = [=](uint32_t co, uint32_t ci) -> uint64_t {
        // [C_out, C_in, 1, 1]
        return ((uint64_t)co * C_in) + ci;
    };

    // Simple block scheduling: split batch among blocks
    uint32_t blocks = AscendC::GetBlockNum();
    uint32_t block_id = AscendC::GetBlockIdx();
    uint32_t n_per_block = (N + blocks - 1) / blocks;
    uint32_t n_begin = block_id * n_per_block;
    uint32_t n_end = n_begin + n_per_block;
    if (n_end > N) n_end = N;

    for (uint32_t n = n_begin; n < n_end; ++n) {
        for (uint32_t oh = 0; oh < H_out; ++oh) {
            for (uint32_t ow = 0; ow < W_out; ++ow) {
                // Accumulate output over C_out
                // Initialize out accumulators
                // To save memory, accumulate directly to output after init
                for (uint32_t co = 0; co < C_out; ++co) {
                    out_ptr[out_index(n, co, oh, ow)] = 0.0f;
                }

                // Fuse depthwise + pointwise
                for (uint32_t ci = 0; ci < C_in; ++ci) {
                    // Compute one depthwise output value at (n, ci, oh, ow)
                    float dw_val = 0.0f;
                    int32_t base_h = (int32_t)oh * (int32_t)stride_h - (int32_t)pad_h;
                    int32_t base_w = (int32_t)ow * (int32_t)stride_w - (int32_t)pad_w;
                    for (uint32_t kh = 0; kh < K_h; ++kh) {
                        int32_t ih = base_h + (int32_t)kh * (int32_t)dilation_h;
                        if (ih < 0 || ih >= (int32_t)H) continue;
                        for (uint32_t kw = 0; kw < K_w; ++kw) {
                            int32_t iw = base_w + (int32_t)kw * (int32_t)dilation_w;
                            if (iw < 0 || iw >= (int32_t)W) continue;
                            float in_v = in_ptr[in_index(n, ci, (uint32_t)ih, (uint32_t)iw)];
                            float w_v = dw_ptr[dw_index(ci, kh, kw)];
                            dw_val += in_v * w_v;
                        }
                    }
                    // Pointwise accumulate into all output channels
                    for (uint32_t co = 0; co < C_out; ++co) {
                        float pw_v = pw_ptr[pw_index(co, ci)];
                        out_ptr[out_index(n, co, oh, ow)] += dw_val * pw_v;
                    }
                }
            }
        }
    }
}

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        # Keep PyTorch modules for parameter management

        # Note: this custom kernel path currently ignores bias for simplicity.
