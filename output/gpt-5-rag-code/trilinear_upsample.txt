inline void GetNHWDC(const gert::Shape* x_shape, uint32_t &n, uint32_t &c, uint32_t &d, uint32_t &h, uint32_t &w) {
    // Expect 5D: N, C, D, H, W

    c = x_shape->GetOriginShape().GetDim(1);
    d = x_shape->GetOriginShape().GetDim(2);
    h = x_shape->GetOriginShape().GetDim(3);
    w = x_shape->GetOriginShape().GetDim(4);
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    TrilinearUpsampleCustomTilingData tiling;
    uint32_t n, c, di, hi, wi;
    GetNHWDC(context->GetInputShape(0), n, c, di, hi, wi);

    // align_corners=True with scale_factor=2 -> out = (in - 1) * 2 + 1
    uint32_t do_ = (di > 1) ? ((di - 1) * 2 + 1) : di;
    uint32_t ho  = (hi > 1) ? ((hi - 1) * 2 + 1) : hi;
    uint32_t wo  = (wi > 1) ? ((wi - 1) * 2 + 1) : wi;

    uint64_t totalInLength  = static_cast<uint64_t>(n) * c * di * hi * wi;
    uint64_t totalOutLength = static_cast<uint64_t>(n) * c * do_ * ho * wo;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_di(di);
    tiling.set_hi(hi);
    tiling.set_wi(wi);
    tiling.set_do_(do_);
    tiling.set_ho(ho);
    tiling.set_wo(wo);
    tiling.set_totalInLength(totalInLength);
    tiling.set_totalOutLength(totalOutLength);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // small scalar buffers

class KernelTrilinearUpsample {
public:
    __aicore__ inline KernelTrilinearUpsample() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t di, uint32_t hi, uint32_t wi,
                                uint32_t do_, uint32_t ho, uint32_t wo,
                                uint64_t totalInLength, uint64_t totalOutLength,
                                uint32_t tileNum)
    {
        this->n = n; this->c = c;
        this->di = di; this->hi = hi; this->wi = wi;
        this->do_ = do_; this->ho = ho; this->wo = wo;

        this->sW_in = 1;
        this->sH_in = this->wi;
        this->sD_in = this->hi * this->wi;
        this->sC_in = this->di * this->hi * this->wi;
        this->sN_in = this->c * this->sC_in;

        this->sW_out = 1;
        this->sH_out = this->wo;
        this->sD_out = this->ho * this->wo;
        this->sC_out = this->do_ * this->ho * this->wo;
        this->sN_out = this->c * this->sC_out;

        this->blockOutLength = static_cast<uint64_t>(totalOutLength) / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = static_cast<uint64_t>(blockOutLength) / tileNum;
        if (this->tileLength == 0) this->tileLength = blockOutLength; // fallback

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, totalInLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockOutLength * AscendC::GetBlockIdx(), this->blockOutLength);

        pipe.InitBuffer(inBuf, BUFFER_NUM, sizeof(DTYPE_X));
        pipe.InitBuffer(tmpBuf1, BUFFER_NUM, sizeof(DTYPE_X));
        pipe.InitBuffer(tmpBuf2, BUFFER_NUM, sizeof(DTYPE_X));
        pipe.InitBuffer(outBuf, BUFFER_NUM, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        uint64_t loops = (blockOutLength + tileLength - 1) / tileLength;
        for (uint64_t p = 0; p < loops; ++p) {
            uint64_t start = p * tileLength;
            uint64_t remain = blockOutLength - start;
            uint64_t len = remain < tileLength ? remain : tileLength;
            ComputeRange(start, len);
        }
    }

private:
    __aicore__ inline void ComputeRange(uint64_t start, uint64_t len)
    {
        for (uint64_t i = 0; i < len; ++i) {
            uint64_t outIndex = start + i;

            // decode indices in output tensor: N,C,Do,Ho,Wo
            uint64_t tmp = outIndex;
            uint32_t w = tmp % wo; tmp /= wo;
            uint32_t h = tmp % ho; tmp /= ho;
            uint32_t d = tmp % do_; tmp /= do_;
            uint32_t cidx = tmp % c;
            uint32_t nidx = tmp / c;

            // align_corners = true mapping
            float sd = (do_ > 1) ? (static_cast<float>(di - 1) / static_cast<float>(do_ - 1) * d) : 0.f;
            float sh = (ho  > 1) ? (static_cast<float>(hi - 1) / static_cast<float>(ho  - 1) * h) : 0.f;
            float sw = (wo  > 1) ? (static_cast<float>(wi - 1) / static_cast<float>(wo  - 1) * w) : 0.f;

            int32_t d0 = static_cast<int32_t>(AscendC::Floor(sd));
            int32_t h0 = static_cast<int32_t>(AscendC::Floor(sh));
            int32_t w0 = static_cast<int32_t>(AscendC::Floor(sw));
            int32_t d1 = d0 < static_cast<int32_t>(di - 1) ? d0 + 1 : d0;
            int32_t h1 = h0 < static_cast<int32_t>(hi - 1) ? h0 + 1 : h0;
            int32_t w1 = w0 < static_cast<int32_t>(wi - 1) ? w0 + 1 : w0;

            float wd = sd - static_cast<float>(d0);
            float wh = sh - static_cast<float>(h0);
            float ww = sw - static_cast<float>(w0);

            float wd0 = 1.0f - wd, wd1 = wd;
            float wh0 = 1.0f - wh, wh1 = wh;
            float ww0 = 1.0f - ww, ww1 = ww;

            // compute base input strides
            uint64_t baseNC = static_cast<uint64_t>(nidx) * sN_in + static_cast<uint64_t>(cidx) * sC_in;

            uint64_t off000 = baseNC + static_cast<uint64_t>(d0) * sD_in + static_cast<uint64_t>(h0) * sH_in + static_cast<uint64_t>(w0);
            uint64_t off001 = baseNC + static_cast<uint64_t>(d0) * sD_in + static_cast<uint64_t>(h0) * sH_in + static_cast<uint64_t>(w1);
            uint64_t off010 = baseNC + static_cast<uint64_t>(d0) * sD_in + static_cast<uint64_t>(h1) * sH_in + static_cast<uint64_t>(w0);
            uint64_t off011 = baseNC + static_cast<uint64_t>(d0) * sD_in + static_cast<uint64_t>(h1) * sH_in + static_cast<uint64_t>(w1);
            uint64_t off100 = baseNC + static_cast<uint64_t>(d1) * sD_in + static_cast<uint64_t>(h0) * sH_in + static_cast<uint64_t>(w0);
            uint64_t off101 = baseNC + static_cast<uint64_t>(d1) * sD_in + static_cast<uint64_t>(h0) * sH_in + static_cast<uint64_t>(w1);
            uint64_t off110 = baseNC + static_cast<uint64_t>(d1) * sD_in + static_cast<uint64_t>(h1) * sH_in + static_cast<uint64_t>(w0);
            uint64_t off111 = baseNC + static_cast<uint64_t>(d1) * sD_in + static_cast<uint64_t>(h1) * sH_in + static_cast<uint64_t>(w1);

            // Load 8 neighbors (scalar copies)
            AscendC::LocalTensor<DTYPE_X> v000 = inBuf.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> v001 = inBuf.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> v010 = inBuf.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> v011 = inBuf.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> v100 = inBuf.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> v101 = inBuf.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> v110 = inBuf.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> v111 = inBuf.AllocTensor<DTYPE_X>();

            AscendC::DataCopy(v000, xGm[off000], 1);
            AscendC::DataCopy(v001, xGm[off001], 1);
            AscendC::DataCopy(v010, xGm[off010], 1);
            AscendC::DataCopy(v011, xGm[off011], 1);
            AscendC::DataCopy(v100, xGm[off100], 1);
            AscendC::DataCopy(v101, xGm[off101], 1);
            AscendC::DataCopy(v110, xGm[off110], 1);
            AscendC::DataCopy(v111, xGm[off111], 1);

            // x interpolation
            AscendC::LocalTensor<DTYPE_X> t00 = tmpBuf1.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> t01 = tmpBuf1.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> t10 = tmpBuf1.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> t11 = tmpBuf1.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> tmp = tmpBuf2.AllocTensor<DTYPE_X>();

            AscendC::Muls(t00, v000, ww0, 1);
            AscendC::Muls(tmp, v001, ww1, 1);
            AscendC::Add(t00, t00, tmp, 1);

            AscendC::Muls(t01, v010, ww0, 1);
            AscendC::Muls(tmp, v011, ww1, 1);
            AscendC::Add(t01, t01, tmp, 1);

            AscendC::Muls(t10, v100, ww0, 1);
            AscendC::Muls(tmp, v101, ww1, 1);
            AscendC::Add(t10, t10, tmp, 1);

            AscendC::Muls(t11, v110, ww0, 1);
            AscendC::Muls(tmp, v111, ww1, 1);
            AscendC::Add(t11, t11, tmp, 1);

            // y interpolation
            AscendC::LocalTensor<DTYPE_X> ty0 = tmpBuf2.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> ty1 = tmpBuf2.AllocTensor<DTYPE_X>();

            AscendC::Muls(ty0, t00, wh0, 1);
            AscendC::Muls(tmp, t01, wh1, 1);
            AscendC::Add(ty0, ty0, tmp, 1);

            AscendC::Muls(ty1, t10, wh0, 1);
            AscendC::Muls(tmp, t11, wh1, 1);
            AscendC::Add(ty1, ty1, tmp, 1);

            // z interpolation
            AscendC::LocalTensor<DTYPE_Y> outV = outBuf.AllocTensor<DTYPE_Y>();
            AscendC::Muls(outV, ty0, wd0, 1);
            AscendC::Muls(tmp, ty1, wd1, 1);
            AscendC::Add(outV, outV, tmp, 1);

            // store to output
            AscendC::DataCopy(yGm[outIndex], outV, 1);

            // free locals
            inBuf.FreeTensor(v000); inBuf.FreeTensor(v001); inBuf.FreeTensor(v010); inBuf.FreeTensor(v011);
            inBuf.FreeTensor(v100); inBuf.FreeTensor(v101); inBuf.FreeTensor(v110); inBuf.FreeTensor(v111);
            tmpBuf1.FreeTensor(t00); tmpBuf1.FreeTensor(t01); tmpBuf1.FreeTensor(t10); tmpBuf1.FreeTensor(t11);
            tmpBuf2.FreeTensor(ty0); tmpBuf2.FreeTensor(ty1); tmpBuf2.FreeTensor(tmp);
            outBuf.FreeTensor(outV);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpBuf1;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpBuf2;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outBuf;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, di, hi, wi, do_, ho, wo;
    uint64_t blockOutLength;
    uint64_t tileLength;
    uint32_t tileNum;

    // strides
    uint64_t sW_in, sH_in, sD_in, sC_in, sN_in;
    uint64_t sW_out, sH_out, sD_out, sC_out, sN_out;
};

extern "C" __global__ __aicore__ void trilinear_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTrilinearUpsample op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.di, tiling_data.hi, tiling_data.wi,
            tiling_data.do_, tiling_data.ho, tiling_data.wo,
            tiling_data.totalInLength, tiling_data.totalOutLength,
            tiling_data.tileNum);
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline std::array<int64_t, 5> infer_out_shape_align_corners_s2(const at::Tensor& x) {
    // x: [N,C,D,H,W]; out: ((dim-1)*2+1) for align_corners=True
    auto sizes = x.sizes();
    int64_t n = sizes[0], c = sizes[1], d = sizes[2], h = sizes[3], w = sizes[4];
    int64_t do_ = d > 1 ? ((d - 1) * 2 + 1) : d;
    int64_t ho  = h > 1 ? ((h - 1) * 2 + 1) : h;
    int64_t wo  = w > 1 ? ((w - 1) * 2 + 1) : w;
    return {n, c, do_, ho, wo};
}

at::Tensor trilinear_upsample_custom_impl_npu(const at::Tensor& x) {
    auto out_shape = infer_out_shape_align_corners_s2(x);
    at::Tensor result = at::empty(out_shape, x.options());
    EXEC_NPU_CMD(aclnnTrilinearUpsampleCustom, x, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("trilinear_upsample_custom", &trilinear_upsample_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("trilinear_upsample_custom", &trilinear_upsample_custom_impl_npu, "Trilinear Upsample (align_corners=True, scale=2)");
