kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelConv3dReluLeakyReluGeluSigmoidBiasAdd {
public:
    __aicore__ inline KernelConv3dReluLeakyReluGeluSigmoidBiasAdd() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileNum,
                                float negativeSlope, uint32_t channels, uint32_t innerSize) {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->negativeSlope = negativeSlope;
        this->channels = channels;
        this->innerSize = innerSize;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias, this->channels); // bias length equals channels
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process() {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress) {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline DTYPE_Z relu_scalar(DTYPE_Z v) {
        return v > (DTYPE_Z)0 ? v : (DTYPE_Z)0;
    }

    __aicore__ inline DTYPE_Z leaky_relu_scalar(DTYPE_Z v) {
        return v >= (DTYPE_Z)0 ? v : (DTYPE_Z)(v * (DTYPE_Z)this->negativeSlope);
    }

    __aicore__ inline DTYPE_Z gelu_approx(DTYPE_Z x) {
        // Approximate GELU: 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715*x^3)))
        DTYPE_Z c0 = (DTYPE_Z)0.5f;
        DTYPE_Z c1 = (DTYPE_Z)0.7978845608f; // sqrt(2/pi)
        DTYPE_Z c2 = (DTYPE_Z)0.044715f;
        DTYPE_Z x3 = x * x * x;
        DTYPE_Z t = c1 * (x + c2 * x3);
        // use AscendC fast tanh if available; fallback to polynomial if not
        DTYPE_Z th = AscendC::Tanh(t);
        return c0 * x * ((DTYPE_Z)1 + th);
    }

    __aicore__ inline DTYPE_Z sigmoid_scalar(DTYPE_Z x) {
        DTYPE_Z one = (DTYPE_Z)1.0f;
        DTYPE_Z ex = AscendC::Exp(-x);
        return one / (one + ex);
    }

    __aicore__ inline void Compute(int32_t progress) {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Compute global base index for this tile within the whole tensor
        uint32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();
        uint32_t tileOffset = progress * this->tileLength;
        uint32_t globalBase = blockOffset + tileOffset;

        for (uint32_t i = 0; i < this->tileLength; ++i) {
            uint32_t gidx = globalBase + i;
            // channel index assuming layout [N, C, innerSize]
            uint32_t ch = (this->innerSize > 0 && this->channels > 0) ? ((gidx / this->innerSize) % this->channels) : 0U;
            DTYPE_Z v = (DTYPE_Z)xLocal.GetValue(i);

            // ReLU
            DTYPE_Z v_relu = relu_scalar(v);
            // LeakyReLU after ReLU (kept for semantics)
            DTYPE_Z v_leaky = leaky_relu_scalar(v_relu);
            // GELU
            DTYPE_Z v_gelu = gelu_approx(v_leaky);
            // Sigmoid
            DTYPE_Z v_sigmoid = sigmoid_scalar(v_gelu);
            // Bias add (broadcast per channel)
            DTYPE_Z b = (DTYPE_Z)biasGm.GetValue(ch);
            yLocal.SetValue(i, v_sigmoid + b);
        }

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress) {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> biasGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    float negativeSlope;
    uint32_t channels;
    uint32_t innerSize;
};

extern "C" __global__ __aicore__ void conv3d_relu_leaky_relu_gelu_sigmoid_bias_add_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dReluLeakyReluGeluSigmoidBiasAdd op;
    op.Init(x, bias, y,
            tiling_data.totalLength,
            tiling_data.tileNum,
            tiling_data.negativeSlope,
            tiling_data.channels,
            tiling_data.innerSize);
    op.Process();
}

import torch

class ModelNew(torch.nn.Module):
    """
    Optimized model: run Conv3d on PyTorch, then fuse ReLU -> LeakyReLU -> GELU -> Sigmoid -> BiasAdd
    into a single custom AscendC operator.
    """
    def __init__(self, in_channels, out_channels, kernel_size, bias_shape):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))

    def forward(self, x):
        x = self.conv(x)
        x = custom_ops_lib.conv3d_relu_leaky_relu_gelu_sigmoid_bias_add_custom(x, self.bias)
