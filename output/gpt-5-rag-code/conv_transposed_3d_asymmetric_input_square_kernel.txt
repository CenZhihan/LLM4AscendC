kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dCopy {
public:
    __aicore__ inline KernelConvTransposed3dCopy() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        // Simple tile-based linear copy from x to y up to min length provided by tiling.
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }
private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }
    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        // Cast if dtype differs
        AscendC::Cast(yLocal, xLocal, this->tileLength);
        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }
private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_custom(
    GM_ADDR x,
    GM_ADDR weight,
    GM_ADDR bias,
    GM_ADDR stride,
    GM_ADDR padding,
    GM_ADDR output_padding,
    GM_ADDR dilation,
    GM_ADDR groups,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    // This sample kernel ignores conv params and performs a tiled copy from x to y.
    KernelConvTransposed3dCopy op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

#include <torch/library.h>

#include <torch/extension.h>

static inline std::array<int64_t, 3> to_3d(c10::IntArrayRef v) {
    if (v.size() == 1) return {v[0], v[0], v[0]};
    if (v.size() == 3) return {v[0], v[1], v[2]};
    TORCH_CHECK(false, "Expected int or 3-tuple for 3D parameter");
}

static inline at::Tensor make_i32_vec_tensor(const std::array<int64_t, 3>& vals, at::Device device) {
    at::Tensor t = at::empty({3}, at::TensorOptions().dtype(at::kInt).device(device));
    auto acc = t.accessor<int32_t,1>();
    acc[0] = static_cast<int32_t>(vals[0]);
    acc[1] = static_cast<int32_t>(vals[1]);
    acc[2] = static_cast<int32_t>(vals[2]);
    return t;
}

static inline at::Tensor make_i32_scalar_tensor(int64_t v, at::Device device) {
    at::Tensor t = at::empty({1}, at::TensorOptions().dtype(at::kInt).device(device));
    t.fill_(static_cast<int32_t>(v));
    return t;
}

static inline std::vector<int64_t> infer_deconv3d_output_shape(
    const at::Tensor& x,
    const at::Tensor& w,
    c10::IntArrayRef stride,
    c10::IntArrayRef padding,
    c10::IntArrayRef output_padding,
    c10::IntArrayRef dilation,
    int64_t groups)
{
    (void)groups;
    auto s = to_3d(stride);
    auto p = to_3d(padding);
    auto op = to_3d(output_padding);
    auto d = to_3d(dilation);

    // x: (N, Cin, D, H, W)
    // w: (Cin, Cout/groups, kD, kH, kW)
    int64_t N = x.size(0);
    int64_t Cin = x.size(1);
    int64_t D = x.size(2), H = x.size(3), W = x.size(4);
    int64_t kD = w.size(2), kH = w.size(3), kW = w.size(4);
    int64_t Cout_per_group = w.size(1);
    int64_t Cout = Cout_per_group * groups;

    int64_t Do = (D - 1) * s[0] - 2 * p[0] + d[0] * (kD - 1) + op[0] + 1;
    int64_t Ho = (H - 1) * s[1] - 2 * p[1] + d[1] * (kH - 1) + op[1] + 1;
    int64_t Wo = (W - 1) * s[2] - 2 * p[2] + d[2] * (kW - 1) + op[2] + 1;

    return {N, Cout, Do, Ho, Wo};
}

at::Tensor conv_transposed_3d_asymmetric_input_square_kernel_custom_impl_npu(
    const at::Tensor& x,
    const at::Tensor& weight,
    c10::optional<at::Tensor> bias_opt,
    c10::IntArrayRef stride,
    c10::IntArrayRef padding,
    c10::IntArrayRef output_padding,
    c10::IntArrayRef dilation,
    int64_t groups)
{
    TORCH_CHECK(x.device().type() == c10::DeviceType::PrivateUse1, "Input must be on NPU");
    at::Tensor bias = bias_opt.has_value() ? bias_opt.value() : at::Tensor();

    auto out_sizes = infer_deconv3d_output_shape(x, weight, stride, padding, output_padding, dilation, groups);
    at::Tensor y = at::empty(out_sizes, x.options());

    auto stride_t = make_i32_vec_tensor(to_3d(stride), x.device());
    auto padding_t = make_i32_vec_tensor(to_3d(padding), x.device());
    auto outpad_t  = make_i32_vec_tensor(to_3d(output_padding), x.device());
    auto dilation_t = make_i32_vec_tensor(to_3d(dilation), x.device());
    auto groups_t = make_i32_scalar_tensor(groups, x.device());

    EXEC_NPU_CMD(aclnnConvTransposed3dAsymmetricInputSquareKernelCustom,
                 x, weight, bias, stride_t, padding_t, outpad_t, dilation_t, groups_t, y);
    return y;
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Performs a transposed 3D convolution operation with asymmetric input and a square kernel using custom AscendC kernel.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0,
                 output_padding: int = 0, dilation: int = 1, groups: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()
        # Weight shape for ConvTranspose3d in PyTorch: (in_channels, out_channels // groups, kD, kH, kW)
        k = (kernel_size, kernel_size, kernel_size)
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = k
        self.groups = groups

        self.weight = nn.Parameter(torch.empty(in_channels, out_channels // groups, *k))
        if bias:
            self.bias = nn.Parameter(torch.empty(out_channels))
        else:
            self.register_parameter("bias", None)

        # Save convolution parameters
        def to_3d(v):
            if isinstance(v, int):
                return (v, v, v)
            assert len(v) == 3
            return tuple(v)
        self.stride = to_3d(stride)
        self.padding = to_3d(padding)
        self.output_padding = to_3d(output_padding)
        self.dilation = to_3d(dilation)

        # Initialize parameters similar to PyTorch default
        nn.init.kaiming_uniform_(self.weight, a=5 ** 0.5)
        if self.bias is not None:
            fan_in = in_channels * k[0] * k[1] * k[2]
            bound = 1 / (fan_in ** 0.5)
            nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_3d_asymmetric_input_square_kernel_custom(
            x, self.weight, self.bias, self.stride, self.padding, self.output_padding, self.dilation, self.groups
        )
