host_operator_src="""
#include "standard_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    gert::Shape* c_shape = context->GetOutputShape(0);
    // A: (M, K), B: (K, N) => C: (M, N)
    int64_t m = a_shape->GetOriginShape().GetDim(0);
    int64_t k_a = a_shape->GetOriginShape().GetDim(1);
    int64_t k_b = b_shape->GetOriginShape().GetDim(0);
    int64_t n = b_shape->GetOriginShape().GetDim(1);
    if (k_a != k_b) {
        return GRAPH_FAILED;
    }
    gert::Shape out;
    out.SetDimNum(2);
    out.SetDim(0, m);
    out.SetDim(1, n);
    *c_shape = out;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType0 = context->GetInputDataType(0);
    const auto inputDataType1 = context->GetInputDataType(1);
    if (inputDataType0 != inputDataType1) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, inputDataType0);
    return GRAPH_SUCCESS;
}
}

namespace optiling {
extern ge::graphStatus TilingFunc(gert::TilingContext* context);
}

namespace ops {
class StandardMatrixMultiplicationCustom : public OpDef {
public:
    explicit StandardMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("a")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})

        this->Input("b")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})

            .UnknownShapeFormat({ge::FORMAT_ND});

            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void standard_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm)
{
    // Prepare tiling for matmul
    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    AscendC::TPipe pipe;

    // Share full on-chip resources for better matmul throughput
    tiling.shareMode = 0;
    tiling.shareL1Size = FULL_L1_SIZE;
    tiling.shareL0CSize = FULL_L0C_SIZE;
    tiling.shareUbSize = 0;

    // Register and run matmul
    AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> matmulObj;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulObj, &tiling);

    // Set A/B tensors and run single-batch GEMM: C = A * B
    matmulObj.SetTensorA(reinterpret_cast<__gm__ float*>(a), false);
    matmulObj.SetTensorB(reinterpret_cast<__gm__ float*>(b), false);
    matmulObj.IterateBatch(reinterpret_cast<__gm__ float*>(c), 1, 1, false);
}
"""
