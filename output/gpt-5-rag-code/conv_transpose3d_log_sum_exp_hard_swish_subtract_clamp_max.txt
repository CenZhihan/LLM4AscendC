kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFusedLSEHswishSubClamp {
public:
    __aicore__ inline KernelFusedLSEHswishSubClamp() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t tileNum, uint32_t spatialPerCore, uint32_t spatialTail)
    {
        this->n = n;
        this->c = c;
        this->d = d;
        this->h = h;
        this->w = w;
        this->tileNum = tileNum;
        this->S = d * h * w;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t coreId = AscendC::GetBlockIdx();

        uint32_t basePerCore = spatialPerCore;
        uint32_t tail = spatialTail;
        this->coreStart = coreId * basePerCore + (coreId < tail ? coreId : tail);
        this->coreSpatial = basePerCore + (coreId < tail ? 1 : 0);

        this->tileLen = (this->coreSpatial + tileNum - 1) / tileNum;
        if (this->tileLen == 0) this->tileLen = this->coreSpatial;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint64_t>(n) * c * S);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_Y *)bias, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, static_cast<uint64_t>(n) * 1 * S);

        pipe.InitBuffer(bufMaxQ, 1, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(bufSumQ, 1, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(bufTmpQ, 1, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(bufOutQ, 1, this->tileLen * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        if (this->coreSpatial == 0) {
            return;
        }
        // Load scalar bias
        AscendC::LocalTensor<DTYPE_X> biasLocal = bufTmpQ.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(biasLocal, biasGm, 1);
        float biasScalar = biasLocal.GetValue(0);
        bufTmpQ.FreeTensor(biasLocal);

        for (uint32_t nIdx = 0; nIdx < n; ++nIdx) {
            uint32_t processed = 0;
            while (processed < this->coreSpatial) {
                uint32_t curLen = this->tileLen;
                if (processed + curLen > this->coreSpatial) {
                    curLen = this->coreSpatial - processed;
                }
                uint32_t sStart = this->coreStart + processed;

                // Allocate local buffers
                AscendC::LocalTensor<DTYPE_X> maxLocal = bufMaxQ.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_X> sumLocal = bufSumQ.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_X> tmpLocal = bufTmpQ.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_X> outLocal = bufOutQ.AllocTensor<DTYPE_X>();

                // First pass: compute per-position max across channels
                // Load channel 0
                uint64_t baseOff0 = ((static_cast<uint64_t>(nIdx) * c + 0) * S) + sStart;
                AscendC::DataCopy(maxLocal, xGm[baseOff0], curLen);
                // For remaining channels
                for (uint32_t cIdx = 1; cIdx < c; ++cIdx) {
                    uint64_t off = ((static_cast<uint64_t>(nIdx) * c + cIdx) * S) + sStart;
                    AscendC::DataCopy(tmpLocal, xGm[off], curLen);
                    AscendC::Maximum(maxLocal, maxLocal, tmpLocal, curLen);
                }

                // Second pass: sum exp(x - max)
                // sumLocal = exp(c0 - maxLocal)
                AscendC::DataCopy(sumLocal, xGm[baseOff0], curLen);
                AscendC::Sub(sumLocal, sumLocal, maxLocal, curLen);
                AscendC::Exp(sumLocal, sumLocal, curLen);
                for (uint32_t cIdx = 1; cIdx < c; ++cIdx) {
                    uint64_t off = ((static_cast<uint64_t>(nIdx) * c + cIdx) * S) + sStart;
                    AscendC::DataCopy(tmpLocal, xGm[off], curLen);
                    AscendC::Sub(tmpLocal, tmpLocal, maxLocal, curLen);
                    AscendC::Exp(tmpLocal, tmpLocal, curLen);
                    AscendC::Add(sumLocal, sumLocal, tmpLocal, curLen);
                }

                // lse = log(sumLocal) + maxLocal -> reuse outLocal
                AscendC::Log(outLocal, sumLocal, curLen);
                AscendC::Add(outLocal, outLocal, maxLocal, curLen);

                // HardSwish-like (as specified): y = x * sigmoid(x + 3) / 6, x is lse here
                // t = outLocal + 3
                AscendC::Adds(tmpLocal, outLocal, 3.0f, curLen);
                // exp(-t)
                AscendC::Muls(tmpLocal, tmpLocal, -1.0f, curLen);
                AscendC::Exp(tmpLocal, tmpLocal, curLen);
                // denom = 1 + exp(-t)
                AscendC::Adds(tmpLocal, tmpLocal, 1.0f, curLen);
                // outLocal = outLocal / denom
                AscendC::Div(outLocal, outLocal, tmpLocal, curLen);
                // outLocal = outLocal / 6
                AscendC::Muls(outLocal, outLocal, 1.0f/6.0f, curLen);

                // Subtract bias scalar
                AscendC::Subs(outLocal, outLocal, biasScalar, curLen);

                // Clamp to [-1, 1]
                // outLocal = max(outLocal, -1)
                AscendC::Maximums(outLocal, outLocal, -1.0f, curLen);
                // outLocal = min(outLocal, 1)
                AscendC::Minimums(outLocal, outLocal, 1.0f, curLen);

                // Write back to y: shape [N, 1, D, H, W]
                uint64_t yOff = (static_cast<uint64_t>(nIdx) * 1 * S) + sStart;
                AscendC::DataCopy(yGm[yOff], outLocal, curLen);

                // Free locals
                bufMaxQ.FreeTensor(maxLocal);
                bufSumQ.FreeTensor(sumLocal);
                bufTmpQ.FreeTensor(tmpLocal);
                bufOutQ.FreeTensor(outLocal);

                processed += curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bufMaxQ;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bufSumQ;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bufTmpQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> bufOutQ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> biasGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t n, c, d, h, w;
    uint32_t S;
    uint32_t tileNum;
    uint32_t coreStart;
    uint32_t coreSpatial;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void conv_transpose3d_log_sum_exp_hard_swish_subtract_clamp_max_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedLSEHswishSubClamp op;
    op.Init(x, bias, y,
            tiling_data.n(), tiling_data.c(), tiling_data.d(), tiling_data.h(), tiling_data.w(),
            tiling_data.tileNum(), tiling_data.spatialPerCore(), tiling_data.spatialTail());
    op.Process();
}

    """
    ConvTranspose3d followed by fused LogSumExp (over channel dim with keepdim=True),

    """

        x = self.conv_transpose(x)

        x = custom_ops_lib.conv_transpose3d_log_sum_exp_hard_swish_subtract_clamp_max_custom(x, self.bias)
