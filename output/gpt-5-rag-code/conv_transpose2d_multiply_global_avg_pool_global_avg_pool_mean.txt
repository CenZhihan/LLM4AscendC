kernel_src="""
#include "kernel_operator.h"
#include "kernel_utils.h"

constexpr int32_t BUFFER_NUM = 2; // number of buffers for pipelining

class KernelConvTranspose2dMultiplyGlobalAvgPoolGlobalAvgPoolMean {
public:
    __aicore__ inline KernelConvTranspose2dMultiplyGlobalAvgPoolGlobalAvgPoolMean() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR m, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t nc, uint32_t hw, uint32_t groupsPerBlock, uint32_t tileLen)
    {
        this->nc = nc;
        this->hw = hw;
        this->groupsPerBlock = groupsPerBlock;
        this->tileLen = tileLen;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->startGroup = blockIdx * groupsPerBlock;
        uint32_t endGroup = (this->startGroup + groupsPerBlock > nc) ? nc : (this->startGroup + groupsPerBlock);
        this->groupCount = (endGroup > this->startGroup) ? (endGroup - this->startGroup) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + static_cast<uint64_t>(this->startGroup) * hw,
                            static_cast<uint64_t>(this->groupCount) * hw);
        mGm.SetGlobalBuffer((__gm__ DTYPE_Y *)m, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->startGroup, this->groupCount);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueM, 1, 1 * sizeof(DTYPE_Y)); // multiplier scalar
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, 1 * sizeof(DTYPE_Z)); // for 1-element intermediates
    }

    __aicore__ inline void Process()
    {
        if (groupCount == 0) {
            return;
        }
        for (uint32_t g = 0; g < groupCount; ++g) {
            // Accumulate sum over H*W for group g
            uint32_t remaining = hw;
            uint64_t offset = static_cast<uint64_t>(g) * hw;
            DTYPE_Z sumAcc = (DTYPE_Z)0;

            while (remaining > 0) {
                uint32_t len = remaining > tileLen ? tileLen : remaining;
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[offset], len);
                inQueueX.EnQue(xLocal);

                xLocal = inQueueX.DeQue<DTYPE_X>();
                // Compute partial sum over this tile
                DTYPE_Z partSum = ComputeSum(xLocal, xLocal, len);
                sumAcc = sumAcc + partSum;

                inQueueX.FreeTensor(xLocal);
                offset += len;
                remaining -= len;
            }

            // Compute mean
            DTYPE_Z meanVal = sumAcc / static_cast<DTYPE_Z>(hw);

            // Prepare one-element tensor with mean
            AscendC::LocalTensor<DTYPE_Z> meanLocal = tmpQueue.AllocTensor<DTYPE_Z>();
            Duplicate(meanLocal, meanVal, 1);

            // Load multiplier scalar to 1-element tensor
            AscendC::LocalTensor<DTYPE_Y> mLocal = inQueueM.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(mLocal, mGm[0], 1);

            // Multiply mean by multiplier
            AscendC::LocalTensor<DTYPE_Z> outLocal = tmpQueue.AllocTensor<DTYPE_Z>();
            Mul(outLocal, meanLocal, mLocal, 1);

            // Copy result to GM (output shape is (N,C,1,1) flattened to nc)
            AscendC::DataCopy(yGm[g], outLocal, 1);

            // Free locals
            tmpQueue.FreeTensor(meanLocal);
            inQueueM.FreeTensor(mLocal);
            tmpQueue.FreeTensor(outLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueM;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> mGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t nc;
    uint32_t hw;
    uint32_t groupsPerBlock;
    uint32_t tileLen;
    uint32_t startGroup;
    uint32_t groupCount;
};

extern "C" __global__ __aicore__ void conv_transpose2d_multiply_global_avg_pool_global_avg_pool_mean_custom(
    GM_ADDR x, GM_ADDR m, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose2dMultiplyGlobalAvgPoolGlobalAvgPoolMean op;
    op.Init(x, m, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.nc, tiling_data.hw, tiling_data.groupsPerBlock, tiling_data.tileLen);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: uses PyTorch ConvTranspose2d, then custom AscendC op to
    multiply by scalar and perform global average pooling twice.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, multiplier):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(
            in_channels, out_channels, kernel_size,
            stride=stride, padding=padding, output_padding=output_padding)
        self.multiplier = float(multiplier)

    def forward(self, x):
        x = self.conv_transpose(x)  # (N, C, H, W)
        m = torch.tensor([self.multiplier], dtype=x.dtype, device=x.device)
        # Custom kernel performs: (x * m) -> GAP(HW) -> GAP(HW) with keepdim=True
        y = custom_ops_lib.conv_transpose2d_multiply_global_avg_pool_global_avg_pool_mean_custom(x, m)
        return y
