    const gert::Shape* w_shape = context->GetInputShape(1); // [N,K]
    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;

    int64_t Kx = x_shape->GetOriginShape().GetDim(1);
    int64_t N = w_shape->GetOriginShape().GetDim(0);
    int64_t Kw = w_shape->GetOriginShape().GetDim(1);

        return GRAPH_FAILED;

    y_shape->SetDimNum(2);
    y_shape->SetDim(0, M);
    y_shape->SetDim(1, N);
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_W = float;
using DTYPE_B = float;
using DTYPE_Y = float;

constexpr uint32_t ONE = 1;

class KernelMatmulSwishScale {
public:
    __aicore__ inline KernelMatmulSwishScale() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scale, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileK, uint32_t tileN)
    {
        this->M = M; this->N = N; this->K = K;
        this->tileK = tileK; this->tileN = tileN;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)M * K);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)N * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, (uint64_t)N);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_Y*)scale, (uint64_t)ONE);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, (uint64_t)M * N);

        pipe.InitBuffer(aBuf, 1, tileK * sizeof(DTYPE_X));
        pipe.InitBuffer(wBuf, 1, tileK * sizeof(DTYPE_W));
        pipe.InitBuffer(biasBuf, 1, tileN * sizeof(DTYPE_B));
        pipe.InitBuffer(outBuf, 1, tileN * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpBuf, 1, tileN * sizeof(DTYPE_Y));

        // load scalar scale
        LocalTensor<DTYPE_Y> sLocal = tmpBuf.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(sLocal, scaleGm[0], ONE);
        tmpBuf.EnQue(sLocal);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t rowsPerStride = (M + blockNum - 1) / blockNum;

        uint32_t rowStart = blockIdx * rowsPerStride;
        uint32_t rowEnd = AscendC::Min(M, rowStart + rowsPerStride);

        LocalTensor<DTYPE_Y> sLocal = tmpBuf.DeQue<DTYPE_Y>();
        DTYPE_Y scaleVal = sLocal.GetValue(0);
        tmpBuf.FreeTensor(sLocal);

        for (uint32_t m = rowStart; m < rowEnd; ++m) {
            // traverse N by tiles
            for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
                uint32_t curTileN = AscendC::Min(tileN, N - n0);

                LocalTensor<DTYPE_Y> yLocal = outBuf.AllocTensor<DTYPE_Y>();
                LocalTensor<DTYPE_B> bLocal = biasBuf.AllocTensor<DTYPE_B>();

                // init yLocal with bias
                AscendC::DataCopy(bLocal, bGm[n0], curTileN);
                // copy bias into yLocal
                for (uint32_t j = 0; j < curTileN; ++j) {
                    yLocal.SetValue(j, bLocal.GetValue(j));
                }
                biasBuf.FreeTensor(bLocal);

                // accumulate K dimension
                for (uint32_t k0 = 0; k0 < K; k0 += tileK) {
                    uint32_t curTileK = AscendC::Min(tileK, K - k0);

                    LocalTensor<DTYPE_X> aLocal = aBuf.AllocTensor<DTYPE_X>();
                    AscendC::DataCopy(aLocal, xGm[m * K + k0], curTileK);
                    aBuf.EnQue(aLocal);
                    aLocal = aBuf.DeQue<DTYPE_X>();

                    // for each output column within tile, load a slice of W and accumulate
                    for (uint32_t j = 0; j < curTileN; ++j) {
                        LocalTensor<DTYPE_W> wLocal = wBuf.AllocTensor<DTYPE_W>();
                        AscendC::DataCopy(wLocal, wGm[(n0 + j) * K + k0], curTileK);
                        wBuf.EnQue(wLocal);
                        wLocal = wBuf.DeQue<DTYPE_W>();

                        DTYPE_Y acc = yLocal.GetValue(j);
                        for (uint32_t p = 0; p < curTileK; ++p) {
                            acc += static_cast<DTYPE_Y>(aLocal.GetValue(p)) * static_cast<DTYPE_Y>(wLocal.GetValue(p));
                        }
                        yLocal.SetValue(j, acc);

                        wBuf.FreeTensor(wLocal);
                    }
                    aBuf.FreeTensor(aLocal);
                }

                // swish: y = y * sigmoid(y)
                LocalTensor<DTYPE_Y> sigLocal = tmpBuf.AllocTensor<DTYPE_Y>();
                AscendC::Sigmoid(sigLocal, yLocal, curTileN);
                AscendC::Mul(yLocal, yLocal, sigLocal, curTileN);
                tmpBuf.FreeTensor(sigLocal);

                // scale
                AscendC::Muls(yLocal, yLocal, scaleVal, curTileN);

                // write back
                AscendC::DataCopy(yGm[m * N + n0], yLocal, curTileN);
                outBuf.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasBuf;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpBuf;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> scaleGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t M, N, K;
    uint32_t tileK, tileN;
};

extern "C" __global__ __aicore__ void matmul_swish_scaling_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scale,
                                                                  GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    uint32_t M = tiling_data.M();
    uint32_t N = tiling_data.N();
    uint32_t K = tiling_data.K();
    uint32_t tileK = tiling_data.tileK();
    uint32_t tileN = tiling_data.tileN();

    KernelMatmulSwishScale op;
    op.Init(x, w, b, scale, y, M, N, K, tileK, tileN);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, scaling_factor):
        super().__init__()
        self.weight = torch.nn.Parameter(torch.empty(out_features, in_features))
        self.bias = torch.nn.Parameter(torch.zeros(out_features))
        self.reset_parameters()
        self.scaling_factor = scaling_factor

    def reset_parameters(self):
        torch.nn.init.kaiming_uniform_(self.weight, a=5**0.5)
        fan_in, _ = torch.nn.init._calculate_fan_in_and_fan_out(self.weight)
        bound = 1 / fan_in**0.5 if fan_in > 0 else 0
        torch.nn.init.uniform_(self.bias, -bound, bound)

    def forward(self, x):
        scale = torch.tensor([self.scaling_factor], device=x.device, dtype=x.dtype)
        return custom_ops_lib.matmul_swish_scaling_custom(x, self.weight, self.bias, scale)
'''
