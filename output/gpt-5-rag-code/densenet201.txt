namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* y_shape = context->GetInputShape(1);

    // Expect shapes: [N, C, H, W]
    if (x_shape->GetDimNum() != 4 || y_shape->GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    uint32_t n  = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t c1 = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t h  = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t w  = static_cast<uint32_t>(x_shape->GetDim(3));
    uint32_t c2 = static_cast<uint32_t>(y_shape->GetDim(1));

    uint32_t hw = h * w;
    uint64_t totalLen64 = static_cast<uint64_t>(n) * static_cast<uint64_t>(c1 + c2) * static_cast<uint64_t>(hw);
    if (totalLen64 > std::numeric_limits<uint32_t>::max()) {
        return ge::GRAPH_FAILED;
    }
    uint32_t totalLength = static_cast<uint32_t>(totalLen64);

    Densenet201CustomTilingData tiling;
    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_tileNum(TILE_NUM);
    tiling.set_n(n);
    tiling.set_c1(c1);
    tiling.set_c2(c2);
    tiling.set_hw(hw);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* y_shape = context->GetInputShape(1);
    if (x_shape->GetDimNum() != 4 || y_shape->GetDimNum() != 4) {
        return GRAPH_FAILED;
    }
    if (x_shape->GetDim(0) != y_shape->GetDim(0) ||
        x_shape->GetDim(2) != y_shape->GetDim(2) ||
        x_shape->GetDim(3) != y_shape->GetDim(3)) {
        return GRAPH_FAILED;
    }
    gert::Shape* z_shape = context->GetOutputShape(0);
    z_shape->SetDimNum(4);
    z_shape->SetDim(0, x_shape->GetDim(0));
    z_shape->SetDim(1, x_shape->GetDim(1) + y_shape->GetDim(1));
    z_shape->SetDim(2, x_shape->GetDim(2));
    z_shape->SetDim(3, x_shape->GetDim(3));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    if (inputDataType != context->GetInputDataType(1)) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelDensenet201 {
public:
    __aicore__ inline KernelDensenet201() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t n, uint32_t c1, uint32_t c2, uint32_t hw)
    {
        this->totalLength = totalLength;
        this->tileNum = tileNum;
        this->n = n;
        this->c1 = c1;
        this->c2 = c2;
        this->hw = hw;

        this->segX = c1 * hw;
        this->segY = c2 * hw;
        this->segZ = (c1 + c2) * hw;

        uint32_t blocks = AscendC::GetBlockNum();
        this->blockLength = (totalLength + blocks - 1) / blocks; // ceil
        this->tileLength = this->blockLength / (tileNum * BUFFER_NUM);
        if (this->tileLength == 0) { this->tileLength = 1024; } // fallback minimal tile

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, n * segX);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, n * segY);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, n * segZ);

        pipe.InitBuffer(ubQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint32_t idx = AscendC::GetBlockIdx();
        uint32_t blocks = AscendC::GetBlockNum();

        // partition batches across blocks
        uint32_t base = n / blocks;
        uint32_t rem = n % blocks;
        uint32_t myN = base + (idx < rem ? 1 : 0);
        if (myN == 0) { return; }
        uint32_t start = idx * base + (idx < rem ? idx : rem);

        for (uint32_t b = 0; b < myN; ++b) {
            uint32_t bn = start + b;
            uint32_t offX = bn * segX;
            uint32_t offY = bn * segY;
            uint32_t offZ = bn * segZ;

            CopySegmentX(offZ, offX, segX);
            CopySegmentY(offZ + segX, offY, segY);
        }
    }

private:
    __aicore__ inline void CopySegmentX(uint32_t dstOff, uint32_t srcOff, uint32_t len)
    {
        uint32_t copied = 0;
        while (copied < len) {
            uint32_t eff = AscendC::Min(this->tileLength, len - copied);
            AscendC::LocalTensor<DTYPE_X> ub = ubQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(ub, xGm[srcOff + copied], eff);
            AscendC::DataCopy(zGm[dstOff + copied], ub, eff);
            ubQueue.FreeTensor(ub);
            copied += eff;
        }
    }

    __aicore__ inline void CopySegmentY(uint32_t dstOff, uint32_t srcOff, uint32_t len)
    {
        uint32_t copied = 0;
        while (copied < len) {
            uint32_t eff = AscendC::Min(this->tileLength, len - copied);
            AscendC::LocalTensor<DTYPE_Y> ub = ubQueue.AllocTensor<DTYPE_Y>();
            AscendC::DataCopy(ub, yGm[srcOff + copied], eff);
            AscendC::DataCopy(zGm[dstOff + copied], ub, eff);
            ubQueue.FreeTensor(ub);
            copied += eff;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> ubQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t totalLength;
    uint32_t tileNum;
    uint32_t n, c1, c2, hw;
    uint32_t segX, segY, segZ;
    uint32_t blockLength;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void densenet201_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelDensenet201 op;
    op.Init(x, y, z,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.n, tiling_data.c1, tiling_data.c2, tiling_data.hw);
    op.Process();
}

#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline at::Tensor make_output_like_cat_c1(const at::Tensor& x, const at::Tensor& y) {
    TORCH_CHECK(x.dim() == 4 && y.dim() == 4, "Densenet201Custom expects 4D tensors");
    TORCH_CHECK(x.size(0) == y.size(0) && x.size(2) == y.size(2) && x.size(3) == y.size(3),
                "Input shapes must match on N, H, W");
    auto sizes = x.sizes().vec();
    sizes[1] = x.size(1) + y.size(1);
    return at::empty(sizes, x.options());

    auto result = make_output_like_cat_c1(x, y);
    EXEC_NPU_CMD(aclnnDensenet201Custom, x, y, result);
    return result;

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("densenet201_custom", &densenet201_custom_impl_npu);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("densenet201_custom", &densenet201_custom_impl_npu, "Concat along C for DenseNet201");

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class DenseBlock(nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        super(DenseBlock, self).__init__()
        layers = []
        for i in range(num_layers):
            layers.append(self._make_layer(num_input_features + i * growth_rate, growth_rate))
        self.layers = nn.ModuleList(layers)

    def _make_layer(self, in_features: int, growth_rate: int):
        return nn.Sequential(
            nn.BatchNorm2d(in_features),
            nn.ReLU(inplace=True),
            nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
            nn.Dropout(0.0)
        )

    def forward(self, x):
        for layer in self.layers:
            new_feature = layer(x)
            # Replace torch.cat(features, 1) with custom AscendC concat (two-input incremental)
            x = custom_ops_lib.densenet201_custom(x, new_feature)
        return x

class TransitionLayer(nn.Module):
    def __init__(self, num_input_features: int, num_output_features: int):
        super(TransitionLayer, self).__init__()
        self.transition = nn.Sequential(
            nn.BatchNorm2d(num_input_features),
            nn.ReLU(inplace=True),
            nn.Conv2d(num_input_features, num_output_features, kernel_size=1, bias=False),
            nn.AvgPool2d(kernel_size=2, stride=2)
        )

    def forward(self, x):
        return self.transition(x)

class ModelNew(nn.Module):
    def __init__(self, growth_rate: int = 32, num_classes: int = 1000):
        super(ModelNew, self).__init__()
        self.features = nn.Sequential(
            nn.Conv2d(3, 64, kernel_size=7, stride=2, padding=3, bias=False),
            nn.BatchNorm2d(64),
            nn.ReLU(inplace=True),
            nn.MaxPool2d(kernel_size=3, stride=2, padding=1)
        )

        num_features = 64
        block_layers = [6, 12, 48, 32]  # DenseNet201

        self.dense_blocks = nn.ModuleList()
        self.transition_layers = nn.ModuleList()

        for i, num_layers in enumerate(block_layers):
            block = DenseBlock(num_layers=num_layers, num_input_features=num_features, growth_rate=growth_rate)
            self.dense_blocks.append(block)
            num_features = num_features + num_layers * growth_rate

            if i != len(block_layers) - 1:
                transition = TransitionLayer(num_input_features=num_features, num_output_features=num_features // 2)
                self.transition_layers.append(transition)
                num_features = num_features // 2

        self.final_bn = nn.BatchNorm2d(num_features)
        self.classifier = nn.Linear(num_features, num_classes)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        x = self.features(x)

        for i, block in enumerate(self.dense_blocks):
            x = block(x)
            if i != len(self.dense_blocks) - 1:
                x = self.transition_layers[i](x)

        x = self.final_bn(x)
        x = F.relu(x, inplace=True)
        x = F.adaptive_avg_pool2d(x, (1, 1)).view(x.size(0), -1)
        x = self.classifier(x)
        return x

def get_inputs():
    batch_size = 10
    height, width = 224, 224
    return [torch.rand(batch_size, 3, height, width)]

def get_init_inputs():
    growth_rate = 32
    num_classes = 10
    return [growth_rate, num_classes]
'''
