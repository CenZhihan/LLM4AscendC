kernel_src="""
#include "kernel_operator.h"
#include <limits>

using DTYPE_X = float;
using DTYPE_Y = int32_t;
using DTYPE_Z = float;

constexpr int32_t BUFFER_NUM = 1;

class KernelCrossEntropyLoss {
public:
    __aicore__ inline KernelCrossEntropyLoss() {}
    __aicore__ inline void Init(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss,
                                uint32_t rows, uint32_t cols)
    {
        this->rows = rows;
        this->cols = cols;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blk = AscendC::GetBlockIdx();

        uint32_t baseRows = rows / blockNum;
        uint32_t rem = rows % blockNum;

        this->rowStart = blk * baseRows + (blk < rem ? blk : rem);
        this->rowCount = baseRows + (blk < rem ? 1u : 0u);

        predsGm.SetGlobalBuffer((__gm__ DTYPE_X*)predictions + (uint64_t)rowStart * cols, (uint64_t)rowCount * cols);
        targsGm.SetGlobalBuffer((__gm__ DTYPE_Y*)targets + rowStart, rowCount);
        lossGm.SetGlobalBuffer((__gm__ DTYPE_Z*)loss + rowStart, rowCount);

        pipe.InitBuffer(inQueuePred, BUFFER_NUM, cols * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueLoss, BUFFER_NUM, 1 * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < rowCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t i)
    {
        AscendC::LocalTensor<DTYPE_X> predLocal = inQueuePred.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(predLocal, predsGm[i * cols], cols);
        inQueuePred.EnQue(predLocal);
    }

    __aicore__ inline void Compute(uint32_t i)
    {
        AscendC::LocalTensor<DTYPE_X> predLocal = inQueuePred.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> lossLocal = outQueueLoss.AllocTensor<DTYPE_Z>();

        // Read target index
        DTYPE_Y target_idx = targsGm.GetValue(i);

        // Compute log-sum-exp and pick target logit
        DTYPE_X maxVal = -std::numeric_limits<DTYPE_X>::infinity();
        for (uint32_t j = 0; j < cols; ++j) {
            DTYPE_X v = predLocal.GetValue(j);
            if (v > maxVal) maxVal = v;
        }

        DTYPE_X sumExp = 0.0f;
        for (uint32_t j = 0; j < cols; ++j) {
            DTYPE_X v = predLocal.GetValue(j);
            sumExp += AscendC::Exp(v - maxVal);
        }
        DTYPE_X logsumexp = maxVal + AscendC::Log(sumExp);

        DTYPE_X targetLogit = predLocal.GetValue((uint32_t)target_idx);
        DTYPE_Z li = (DTYPE_Z)(logsumexp - targetLogit);

        lossLocal.SetValue(0, li);

        outQueueLoss.EnQue<DTYPE_Z>(lossLocal);
        inQueuePred.FreeTensor(predLocal);
    }

    __aicore__ inline void CopyOut(uint32_t i)
    {
        AscendC::LocalTensor<DTYPE_Z> lossLocal = outQueueLoss.DeQue<DTYPE_Z>();
        AscendC::DataCopy(lossGm[i], lossLocal, 1);
        outQueueLoss.FreeTensor(lossLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePred;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueLoss;

    AscendC::GlobalTensor<DTYPE_X> predsGm;
    AscendC::GlobalTensor<DTYPE_Y> targsGm;
    AscendC::GlobalTensor<DTYPE_Z> lossGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t rowStart;
    uint32_t rowCount;
};

extern "C" __global__ __aicore__ void cross_entropy_loss_custom(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCrossEntropyLoss op;
    op.Init(predictions, targets, loss, tiling_data.rows, tiling_data.cols);
    op.Process();
}

#include <torch/library.h>

#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor cross_entropy_loss_custom_impl_npu(const at::Tensor& predictions, const at::Tensor& targets) {
    auto t = targets;
    if (t.dtype() != at::kInt) {
        t = targets.to(at::kInt);
    }
    // Output per-sample loss vector [N]
    at::Tensor result = at::empty({predictions.size(0)}, predictions.options());
    EXEC_NPU_CMD(aclnnCrossEntropyLossCustom, predictions, t, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("cross_entropy_loss_custom", &cross_entropy_loss_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("cross_entropy_loss_custom", &cross_entropy_loss_custom_impl_npu, "Cross Entropy Loss (per-sample)");
