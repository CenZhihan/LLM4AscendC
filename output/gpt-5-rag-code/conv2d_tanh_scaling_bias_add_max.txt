kernel_src="""
#include "kernel_operator.h"

using DType = float;

class KernelConv2dTanhScaleBiasAddMax {
public:
    __aicore__ inline KernelConv2dTanhScaleBiasAddMax() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR alpha, GM_ADDR pool, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W, uint32_t tileNum)
    {
        this->N = N;
        this->C = C;
        this->H = H;
        this->W = W;
        this->tileNum = tileNum;

        xGm.SetGlobalBuffer((__gm__ DType*)x, (uint32_t)(N*C*H*W));
        biasGm.SetGlobalBuffer((__gm__ DType*)bias, (uint32_t)(C)); // expect (C,1,1)
        alphaGm = (__gm__ DType*)alpha;
        poolGm = (__gm__ int32_t*)pool;
        yGm.SetGlobalBuffer((__gm__ DType*)y, (uint32_t)(N*C*H*W)); // allocate full; kernel writes pooled subset

        // local buffers: p rows of width W
        pipe.InitBuffer(rowBufQ, 2, (uint32_t)(maxRows * maxWidth * sizeof(DType)));
        pipe.InitBuffer(outBufQ, 2, (uint32_t)(maxWidth * sizeof(DType)));
    }

    __aicore__ inline void Process()
    {
        int32_t blk = AscendC::GetBlockIdx();
        int32_t blkNum = AscendC::GetBlockNum();

        // read alpha and pool_k (best effort)
        DType alphaVal = 1.0f;
        int32_t poolK = 4;
        // Attempt to read scalar alpha and pool size from GM into local
        {
            AscendC::LocalTensor<DType> tmp = rowBufQ.AllocTensor<DType>();
            AscendC::DataCopy(tmp, alphaGm, 1);
            alphaVal = tmp.GetValue(0);
            rowBufQ.FreeTensor(tmp);
        }
        {
            AscendC::LocalTensor<int32_t> tmpi = outBufQ.AllocTensor<int32_t>();
            AscendC::DataCopy(tmpi, poolGm, 1);
            poolK = tmpi.GetValue(0);
            outBufQ.FreeTensor(tmpi);
        }
        if (poolK <= 0) poolK = 1;
        // Effective output dims
        uint32_t Hout = H / (uint32_t)poolK;
        uint32_t Wout = W / (uint32_t)poolK;

        uint32_t totalNC = N * C;
        for (uint32_t idx = blk; idx < totalNC; idx += blkNum) {
            uint32_t n = idx / C;
            uint32_t c = idx % C;
            DType bval = LoadBias(c);

            for (uint32_t oh = 0; oh < Hout; ++oh) {
                // copy poolK rows from input for this (n,c)
                uint32_t inRowStart = oh * (uint32_t)poolK;
                uint64_t baseOff = ((((uint64_t)n * C + c) * H + inRowStart) * W);
                // allocate buffer for poolK rows
                AscendC::LocalTensor<DType> rowsLocal = rowBufQ.AllocTensor<DType>();
                uint32_t rowsCount = (uint32_t)poolK * W;
                // copy contiguous block of poolK*W
                AscendC::DataCopy(rowsLocal, xGm[baseOff], rowsCount);

                // apply tanh + scale + bias in-place
                for (uint32_t r = 0; r < (uint32_t)poolK; ++r) {
                    uint32_t rowOff = r * W;
                    for (uint32_t w = 0; w < W; ++w) {
                        DType v = rowsLocal.GetValue(rowOff + w);
                        // tanh
                        v = AscendC::tanh(v);
                        // scale
                        v = v * alphaVal;
                        // bias add
                        v = v + bval;
                        rowsLocal.SetValue(rowOff + w, v);
                    }
                }

                // max-pool across poolK x poolK windows along width dimension
                AscendC::LocalTensor<DType> outRowLocal = outBufQ.AllocTensor<DType>();
                for (uint32_t ow = 0; ow < Wout; ++ow) {
                    uint32_t wStart = ow * (uint32_t)poolK;
                    DType m = -INFINITY;
                    for (uint32_t r = 0; r < (uint32_t)poolK; ++r) {
                        uint32_t rowOff = r * W;
                        for (uint32_t kk = 0; kk < (uint32_t)poolK; ++kk) {
                            DType v = rowsLocal.GetValue(rowOff + wStart + kk);
                            m = v > m ? v : m;
                        }
                    }
                    outRowLocal.SetValue(ow, m);
                }

                // write pooled row to output
                uint64_t outBase = ((((uint64_t)n * C + c) * Hout + oh) * Wout);
                AscendC::DataCopy(yGm[outBase], outRowLocal, Wout);

                outBufQ.FreeTensor(outRowLocal);
                rowBufQ.FreeTensor(rowsLocal);
            }
        }
    }

private:
    __aicore__ inline DType LoadBias(uint32_t c)
    {
        // bias is (C,1,1) flattened
        return biasGm[c];
    }

private:
    AscendC::TPipe pipe;
    // Queues for local buffering
    static constexpr uint32_t maxRows = 8;
    static constexpr uint32_t maxWidth = 1024;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> rowBufQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outBufQ;

    AscendC::GlobalTensor<DType> xGm;
    AscendC::GlobalTensor<DType> yGm;
    AscendC::GlobalTensor<DType> biasGm;
    __gm__ DType* alphaGm;
    __gm__ int32_t* poolGm;

    uint32_t N;
    uint32_t C;
    uint32_t H;
    uint32_t W;
    uint32_t tileNum;
};

extern "C" __global__ __aicore__ void conv2d_tanh_scaling_bias_add_max_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR alpha, GM_ADDR pool_size, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dTanhScaleBiasAddMax op;
    op.Init(x, bias, alpha, pool_size, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W, tiling_data.tileNum);
    op.Process();
}

at::Tensor conv2d_tanh_scaling_bias_add_max_custom_impl_npu(const at::Tensor& x, const at::Tensor& bias, double alpha, int64_t pool_k) {
    TORCH_CHECK(x.dim() == 4, "x must be NCHW 4D tensor");
    TORCH_CHECK(bias.dim() == 3 || bias.dim() == 1, "bias must be (C,1,1) or (C)");
    int64_t N = x.size(0);
    int64_t C = x.size(1);
    int64_t H = x.size(2);
    int64_t W = x.size(3);
    TORCH_CHECK(pool_k > 0, "pool_k must be > 0");
    int64_t Hout = H / pool_k;
    int64_t Wout = W / pool_k;

    // Prepare scalar inputs as 1-element tensors
    at::Tensor alpha_t = at::empty({1}, x.options());
    alpha_t.fill_(alpha);
    // pool size tensor as int32
    at::Tensor pool_t = at::empty({1}, x.options().dtype(at::kInt));
    pool_t.fill_(static_cast<int>(pool_k));

    // bias expected as (C,1,1)
    at::Tensor bias_bcast;
    if (bias.dim() == 1) {
        bias_bcast = bias.view({C, 1, 1});
    } else {
        bias_bcast = bias;
    }

    // Allocate output
    at::Tensor y = at::empty({N, C, Hout, Wout}, x.options());
    EXEC_NPU_CMD(aclnnConv2dTanhScalingBiasAddMaxCustom, x, bias_bcast, alpha_t, pool_t, y);
    return y;
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, scaling_factor, bias_shape, pool_kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scaling_factor = scaling_factor
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.pool_kernel_size = pool_kernel_size

    def forward(self, x):
        x = self.conv(x)
        # Fused tanh, scaling, bias add and maxpool
        x = custom_ops_lib.conv2d_tanh_scaling_bias_add_max_custom(
            x, self.bias, float(self.scaling_factor), int(self.pool_kernel_size)
        )
        return x
