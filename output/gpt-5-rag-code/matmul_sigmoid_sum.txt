kernel_src="""
#include "kernel_operator.h"
#include <cmath>

__aicore__ inline float sigmoid_scalar(float x) {
    return 1.f / (1.f + expf(-x));
}

// Fused kernel: y[i] = sum_j sigmoid( dot(x[i], w[j]) + b[j] )
// Shapes:
// x: [batch, inFeatures]
// w: [outFeatures, inFeatures]
// b: [outFeatures]
// y: [batch, 1]
extern "C" __global__ __aicore__ void matmul_sigmoid_sum_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t batch = tiling_data.batch;
    const uint32_t inF = tiling_data.inFeatures;
    const uint32_t outF = tiling_data.outFeatures;

    __gm__ DTYPE_X* X = reinterpret_cast<__gm__ DTYPE_X*>(x);
    __gm__ DTYPE_W* W = reinterpret_cast<__gm__ DTYPE_W*>(w);
    __gm__ DTYPE_B* B = reinterpret_cast<__gm__ DTYPE_B*>(b);
    __gm__ DTYPE_Y* Y = reinterpret_cast<__gm__ DTYPE_Y*>(y);

    const uint32_t blockNum = AscendC::GetBlockNum();
    const uint32_t blockIdx = AscendC::GetBlockIdx();

    uint32_t baseRows = batch / blockNum;
    uint32_t remainder = batch % blockNum;
    uint32_t startRow = baseRows * blockIdx + ((blockIdx < remainder) ? blockIdx : remainder);
    uint32_t rowsThisBlock = baseRows + (blockIdx < remainder ? 1 : 0);

    for (uint32_t r = 0; r < rowsThisBlock; ++r) {
        uint32_t rowIndex = startRow + r;
        float outSum = 0.0f;

        // For each output neuron j
        for (uint32_t j = 0; j < outF; ++j) {
            float acc = 0.0f;
            // dot(x[row], w[j])
            const uint32_t wRowOff = j * inF;
            const uint32_t xRowOff = rowIndex * inF;
            for (uint32_t k = 0; k < inF; ++k) {
                float xv = static_cast<float>(X[xRowOff + k]);
                float wv = static_cast<float>(W[wRowOff + k]);
                acc += xv * wv;
            }
            acc += static_cast<float>(B[j]);
            outSum += sigmoid_scalar(acc);
        }
        Y[rowIndex] = static_cast<DTYPE_Y>(outSum);
    }
}
"""
