kernel_src="""
#include "kernel_operator.h"
#include <cmath>

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias0, GM_ADDR bias1, GM_ADDR gamma, GM_ADDR beta, GM_ADDR groups,
    GM_ADDR workspace, GM_ADDR tiling) {

    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t M = tiling_data.m;
    const uint32_t N = tiling_data.n;
    const uint32_t K = tiling_data.k;

    __gm__ float* x_ptr = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_ptr = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* b0_ptr = reinterpret_cast<__gm__ float*>(bias0);
    __gm__ float* b1_ptr = reinterpret_cast<__gm__ float*>(bias1);
    __gm__ float* g_ptr  = reinterpret_cast<__gm__ float*>(gamma);
    __gm__ float* be_ptr = reinterpret_cast<__gm__ float*>(beta);
    __gm__ int32_t* groups_ptr = reinterpret_cast<__gm__ int32_t*>(groups);
    __gm__ float* y_ptr = reinterpret_cast<__gm__ float*>(workspace); // not used, actual output mapped by runtime
    // In AscendC, the last GM_ADDR before tiling is typically output; adjust based on runtime binding.
    // For portability in this sample, we reuse 'workspace' as placeholder and rely on runtime binding for actual output.

    // Real output pointer is often passed as a separate GM_ADDR before workspace.
    // Adjust to match: x, weight, bias0, bias1, gamma, beta, groups, y, workspace, tiling
    // Hence, reinterpret the correct argument order:
    __gm__ float* y_real = reinterpret_cast<__gm__ float*>(beta); // temporary placeholder to avoid unused warning
#pragma clang diagnostic push
#pragma clang diagnostic ignored "-Wunused-variable"
#pragma clang diagnostic pop
    // Correct binding with known order:
    // Re-cast the 8th parameter as output
    __gm__ float* y_out = reinterpret_cast<__gm__ float*>(workspace);
    // If runtime maps differently, replace y_out mapping accordingly.

    const float eps = 1e-5f;
    const int32_t G = groups_ptr[0] > 0 ? groups_ptr[0] : 1;
    const uint32_t blocks = AscendC::GetBlockNum();
    const uint32_t block_id = AscendC::GetBlockIdx();

    uint32_t rows_per_block = (M + blocks - 1) / blocks;
    uint32_t row_begin = block_id * rows_per_block;
    uint32_t row_end = row_begin + rows_per_block;
    if (row_end > M) row_end = M;

    for (uint32_t i = row_begin; i < row_end; ++i) {
        // First pass: GEMM + BiasAdd + Hardtanh + Mish
        for (uint32_t n = 0; n < N; ++n) {
            float acc = 0.0f;
            // y[i, n] = sum_k x[i, k] * w[n, k]
            __gm__ float* x_row = x_ptr + i * K;
            __gm__ float* w_row = w_ptr + n * K;
            for (uint32_t k = 0; k < K; ++k) {
                acc += x_row[k] * w_row[k];
            }
            acc += b0_ptr[n];
            acc += b1_ptr[n];

            // Hardtanh (default min=-1, max=1)
            if (acc > 1.0f) acc = 1.0f;
            else if (acc < -1.0f) acc = -1.0f;

            // Mish: x * tanh(softplus(x)) where softplus(x)=log1p(exp(x))
            float sp = log1pf(expf(acc));
            float mish = acc * tanhf(sp);

            // Store intermediate to output buffer
            y_out[i * N + n] = mish;
        }

        // Second pass: GroupNorm per sample i over channel dimension N
        // groups G, group size Cg = N / G (assume divisible)
        uint32_t Cg = (G > 0) ? (N / static_cast<uint32_t>(G)) : N;

        for (int32_t g = 0; g < G; ++g) {
            uint32_t c_start = g * Cg;
            uint32_t c_end = c_start + Cg;
            if (c_end > N) c_end = N;

            // mean
            float mean = 0.0f;
            for (uint32_t c = c_start; c < c_end; ++c) {
                mean += y_out[i * N + c];
            }
            uint32_t denom = (c_end > c_start) ? (c_end - c_start) : 1;
            mean /= static_cast<float>(denom);

            // variance
            float var = 0.0f;
            for (uint32_t c = c_start; c < c_end; ++c) {
                float v = y_out[i * N + c] - mean;
                var += v * v;
            }
            var /= static_cast<float>(denom);
            float inv_std = 1.0f / sqrtf(var + eps);

            // normalize and affine
            for (uint32_t c = c_start; c < c_end; ++c) {
                float v = (y_out[i * N + c] - mean) * inv_std;
                float outv = v * g_ptr[c] + be_ptr[c];
                y_out[i * N + c] = outv;
            }
        }
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused GEMM + BiasAdd + Hardtanh + Mish + GroupNorm using custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, bias_shape, num_groups):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        # extra bias added after gemm
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        # groupnorm parameters (gamma and beta)
        self.groupnorm = torch.nn.GroupNorm(num_groups=num_groups, num_channels=out_features)
        self.num_groups = num_groups

    def forward(self, x: torch.Tensor):
        # Fused op consumes:
        # x [B, K], weight [N, K], bias0 [N] (from Linear), bias1 [N] (extra bias),
        # gamma [N], beta [N], groups [1] int32 tensor.
        groups_tensor = torch.tensor([self.num_groups], dtype=torch.int32, device=x.device)
        return custom_ops_lib.gemm_bias_add_hardtanh_mish_group_norm_custom(
            x, self.gemm.weight, self.gemm.bias, self.bias, self.groupnorm.weight, self.groupnorm.bias, groups_tensor
        )
