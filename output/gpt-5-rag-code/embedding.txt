host_operator_src="""
#include "embedding_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    EmbeddingCustomTilingData tiling;

    const gert::Shape* weightShape = context->GetInputShape(0);
    const gert::Shape* indicesShape = context->GetInputShape(1);

    uint64_t totalRows64 = 1;
    for (int i = 0; i < indicesShape->GetDimNum(); ++i) {
        totalRows64 *= static_cast<uint64_t>(indicesShape->GetDim(i));
    }
    uint32_t totalRows = static_cast<uint32_t>(totalRows64);
    uint32_t embDim = static_cast<uint32_t>(weightShape->GetDim(weightShape->GetDimNum() - 1));
    uint32_t vocabSize = static_cast<uint32_t>(weightShape->GetDim(0));

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalRows(totalRows);
    tiling.set_embDim(embDim);
    tiling.set_vocabSize(vocabSize);
    tiling.set_tileNum(TILE_NUM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* indicesShape = context->GetInputShape(1);
    const gert::Shape* weightShape = context->GetInputShape(0);
    gert::Shape* yShape = context->GetOutputShape(0);

    std::vector<int64_t> outDims;
    outDims.reserve(indicesShape->GetDimNum() + 1);
    for (int i = 0; i < indicesShape->GetDimNum(); ++i) {
        outDims.push_back(indicesShape->GetDim(i));
    }
    outDims.push_back(weightShape->GetDim(weightShape->GetDimNum() - 1));
    yShape->SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) {
        yShape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    const auto weightType = context->GetInputDataType(0);
    context->SetOutputDataType(0, weightType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class EmbeddingCustom : public OpDef {
public:
    explicit EmbeddingCustom(const char* name) : OpDef(name)
    {
        this->Input("weight")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

            .ParamType(REQUIRED)

        this->AICore()

    }

OP_ADD(EmbeddingCustom);
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // queue buffer num

class KernelEmbedding {
public:
    __aicore__ inline KernelEmbedding() {}
    __aicore__ inline void Init(GM_ADDR weight, GM_ADDR indices, GM_ADDR y,
                                uint32_t totalRows, uint32_t embDim, uint32_t vocabSize, uint32_t tileNum)
    {
        this->totalRows = totalRows;
        this->embDim = embDim;
        this->vocabSize = vocabSize;
        this->tileNum = tileNum;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint64_t start = (static_cast<uint64_t>(totalRows) * blockIdx) / blockNum;
        uint64_t end = (static_cast<uint64_t>(totalRows) * (blockIdx + 1)) / blockNum;
        this->blockStart = static_cast<uint32_t>(start);
        this->rowsThisBlock = static_cast<uint32_t>(end - start);

        weightGm.SetGlobalBuffer((__gm__ DTYPE_WEIGHT*)weight, static_cast<uint32_t>(vocabSize * embDim));
        indicesGm.SetGlobalBuffer((__gm__ DTYPE_INDICES*)indices + this->blockStart, this->rowsThisBlock);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->blockStart * embDim, this->rowsThisBlock * embDim);

        // determine tileRows (at least 1)
        this->tileRows = this->rowsThisBlock / (this->tileNum * BUFFER_NUM);
        if (this->tileRows == 0) this->tileRows = 1;

        pipe.InitBuffer(inQueueIdx, BUFFER_NUM, this->tileRows * sizeof(DTYPE_INDICES));
        // temporary buffer to hold one embedding vector in UB
        pipe.InitBuffer(tmpQueue, 1, this->embDim * sizeof(DTYPE_WEIGHT));
    }

    __aicore__ inline void Process()
    {
        if (rowsThisBlock == 0) {
            return;
        }
        uint32_t processed = 0;
        while (processed < rowsThisBlock) {
            uint32_t curRows = AscendC::Min(this->tileRows, this->rowsThisBlock - processed);
            CopyIn(processed, curRows);
            Compute(processed, curRows);
            processed += curRows;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t processed, uint32_t curRows)
    {
        AscendC::LocalTensor<DTYPE_INDICES> idxLocal = inQueueIdx.AllocTensor<DTYPE_INDICES>();
        AscendC::DataCopy(idxLocal, indicesGm[processed], curRows);
        inQueueIdx.EnQue(idxLocal);
    }

    __aicore__ inline void Compute(uint32_t processed, uint32_t curRows)
    {
        AscendC::LocalTensor<DTYPE_INDICES> idxLocal = inQueueIdx.DeQue<DTYPE_INDICES>();
        AscendC::LocalTensor<DTYPE_WEIGHT> vecLocal = tmpQueue.AllocTensor<DTYPE_WEIGHT>();

        for (uint32_t i = 0; i < curRows; ++i) {
            uint32_t id = static_cast<uint32_t>(idxLocal.GetValue(i));
            // bounds check is omitted for performance; assume valid indices
            uint32_t srcOffset = id * this->embDim;
            uint32_t dstOffset = (processed + i) * this->embDim;

            AscendC::DataCopy(vecLocal, weightGm[srcOffset], this->embDim);
            AscendC::DataCopy(yGm[dstOffset], vecLocal, this->embDim);
        }

        tmpQueue.FreeTensor(vecLocal);
        inQueueIdx.FreeTensor(idxLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue;

    AscendC::GlobalTensor<DTYPE_WEIGHT> weightGm;
    AscendC::GlobalTensor<DTYPE_INDICES> indicesGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalRows;
    uint32_t embDim;
    uint32_t vocabSize;

    uint32_t blockStart;
    uint32_t rowsThisBlock;
    uint32_t tileRows;
    uint32_t tileNum;
};

extern "C" __global__ __aicore__ void embedding_custom(GM_ADDR weight, GM_ADDR indices, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelEmbedding op;
    op.Init(weight, indices, y,
            tiling_data.totalRows, tiling_data.embDim, tiling_data.vocabSize, tiling_data.tileNum);
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor embedding_custom_impl_npu(const at::Tensor& weight, const at::Tensor& indices) {
    at::Tensor indices_i32 = indices;
    if (indices.scalar_type() != at::kInt) {
        indices_i32 = indices.to(at::kInt);
    }
    std::vector<int64_t> out_sizes = indices_i32.sizes().vec();
    out_sizes.push_back(weight.size(-1));
    at::Tensor result = at::empty(out_sizes, weight.options());
    EXEC_NPU_CMD(aclnnEmbeddingCustom, weight, indices_i32, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("embedding_custom", &embedding_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("embedding_custom", &embedding_custom_impl_npu, "Embedding lookup custom op");
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self):
        super().__init__()
        self.weight = torch.nn.Parameter(torch.empty(100000, 768))
        # Initialize similar to nn.Embedding default
        bound = 1.0 / self.weight.size(1) ** 0.5
        torch.nn.init.uniform_(self.weight, -bound, bound)

    def forward(self, indices):
        return custom_ops_lib.embedding_custom(self.weight, indices)
