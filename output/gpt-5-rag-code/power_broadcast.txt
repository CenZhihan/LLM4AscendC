#include "power_broadcast_custom_tiling.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    PowerBroadcastCustomTilingData tiling;

    // Input 0: x [N, C...] (we only use last dim as cols)
    const gert::Shape* x_shape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(x_shape->GetOriginShape().GetShapeSize());

    // Get last-dimension size as cols for broadcasting of exponent
    auto dims = x_shape->GetOriginShape().GetDims();
    uint32_t cols = 1;
    if (!dims.empty()) {
        cols = static_cast<uint32_t>(dims.back());
    }

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_totalLength(totalLength);
    tiling.set_cols(cols);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Output shape follows broadcasting (exponent broadcasts on last dim)
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);

}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

// Helper: ceil-div
__aicore__ inline uint32_t ceil_div_u32(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

class KernelPowerBroadcast {
public:
    __aicore__ inline KernelPowerBroadcast() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR exponent, GM_ADDR z,
                                uint32_t totalLength, uint32_t cols, uint32_t tileNum)
    {
        this->cols = cols;
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / (tileNum * BUFFER_NUM);
        if (this->tileLength == 0) {
            this->tileLength = this->blockLength > 0 ? this->blockLength : 1;
        }

        uint32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + blockOffset, this->blockLength);
        expGm.SetGlobalBuffer((__gm__ DTYPE_X *)exponent, this->cols); // shared across blocks
        zGm.SetGlobalBuffer((__gm__ DTYPE_X *)z + blockOffset, this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueueExp, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        uint32_t loopCount = ceil_div_u32(this->blockLength, this->tileLength);
        for (uint32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline uint32_t GetTileStart(uint32_t progress) {
        return progress * this->tileLength;
    }
    __aicore__ inline uint32_t GetCurTileLen(uint32_t progress) {
        uint32_t start = GetTileStart(progress);
        uint32_t remain = this->blockLength - start;
        return remain > this->tileLength ? this->tileLength : remain;
    }

    // Fill expLocal by repeating exponent across rows, based on global start index.
    __aicore__ inline void FillExponentLocal(AscendC::LocalTensor<DTYPE_X>& expLocal, uint32_t globalStart, uint32_t len)
    {
        // globalStart is relative to start of this block
        uint32_t idx_mod = (globalStart % this->cols);
        uint32_t filled = 0;
        while (filled < len) {
            uint32_t chunk = this->cols - idx_mod;
            if (chunk > (len - filled)) {
                chunk = len - filled;
            }
            AscendC::DataCopy(expLocal[filled], expGm[idx_mod], chunk);
            filled += chunk;
            idx_mod = 0;
        }
    }

    __aicore__ inline void CopyIn(uint32_t progress)
    {
        uint32_t curLen = GetCurTileLen(progress);
        uint32_t start = GetTileStart(progress);

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[start], curLen);
        inQueueX.EnQue(xLocal);

        AscendC::LocalTensor<DTYPE_X> eLocal = tmpQueueExp.AllocTensor<DTYPE_X>();
        FillExponentLocal(eLocal, start, curLen);
        tmpQueueExp.EnQue(eLocal);
    }

    __aicore__ inline void Compute(uint32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> eLocal = tmpQueueExp.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> zLocal = outQueueZ.AllocTensor<DTYPE_X>();

        uint32_t validCount = xLocal.GetSize();

        // z = exp(log(x) * exponent)
        AscendC::Log(zLocal, xLocal, validCount);
        AscendC::Mul(zLocal, zLocal, eLocal, validCount);
        AscendC::Exp(zLocal, zLocal, validCount);

        outQueueZ.EnQue<DTYPE_X>(zLocal);
        inQueueX.FreeTensor(xLocal);
        tmpQueueExp.FreeTensor(eLocal);
    }

    __aicore__ inline void CopyOut(uint32_t progress)
    {
        uint32_t curLen = GetCurTileLen(progress);
        uint32_t start = GetTileStart(progress);

        AscendC::LocalTensor<DTYPE_X> zLocal = outQueueZ.DeQue<DTYPE_X>();
        AscendC::DataCopy(zGm[start], zLocal, curLen);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> tmpQueueExp;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> expGm;
    AscendC::GlobalTensor<DTYPE_X> zGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t cols;
};

extern "C" __global__ __aicore__ void power_broadcast_custom(GM_ADDR x, GM_ADDR exponent, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelPowerBroadcast op;
    op.Init(x, exponent, z, tiling_data.totalLength, tiling_data.cols, tiling_data.tileNum);
    op.Process();
}
"""
