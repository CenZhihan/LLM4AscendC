kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConcatChannel {
public:
    __aicore__ inline KernelConcatChannel() {}
    __aicore__ inline void Init(GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR x4, GM_ADDR y,
                                uint64_t len1, uint64_t len2, uint64_t len3, uint64_t len4,
                                uint32_t tileSize)
    {
        this->len1 = len1;
        this->len2 = len2;
        this->len3 = len3;
        this->len4 = len4;
        this->tileSize = tileSize;

        x1Gm.SetGlobalBuffer((__gm__ float*)x1, len1);
        x2Gm.SetGlobalBuffer((__gm__ float*)x2, len2);
        x3Gm.SetGlobalBuffer((__gm__ float*)x3, len3);
        x4Gm.SetGlobalBuffer((__gm__ float*)x4, len4);
        yGm.SetGlobalBuffer((__gm__ float*)y, len1 + len2 + len3 + len4);

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileSize * sizeof(float));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint64_t base = 0;
        ConcatOne(x1Gm, len1, base);
        base += len1;
        ConcatOne(x2Gm, len2, base);
        base += len2;
        ConcatOne(x3Gm, len3, base);
        base += len3;
        ConcatOne(x4Gm, len4, base);
    }

private:
    template<typename GTensor>
    __aicore__ inline void ConcatOne(GTensor& srcGm, uint64_t len, uint64_t baseDst)
    {
        uint64_t copied = 0;
        while (copied < len) {
            uint32_t cur = static_cast<uint32_t>(AscendC::Min<uint64_t>(len - copied, tileSize));
            AscendC::LocalTensor<float> tmpIn = inQueue.AllocTensor<float>();
            AscendC::DataCopy(tmpIn, srcGm[copied], cur);
            inQueue.EnQue(tmpIn);

            AscendC::LocalTensor<float> tmp = inQueue.DeQue<float>();
            AscendC::LocalTensor<float> tmpOut = outQueue.AllocTensor<float>();
            // Move data within local tensor (no-op, reuse buffer)
            AscendC::DataCopy(tmpOut, tmp, cur);
            outQueue.EnQue(tmpOut);
            inQueue.FreeTensor(tmp);

            AscendC::LocalTensor<float> toWrite = outQueue.DeQue<float>();
            AscendC::DataCopy(yGm[baseDst + copied], toWrite, cur);
            outQueue.FreeTensor(toWrite);

            copied += cur;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<float> x1Gm;
    AscendC::GlobalTensor<float> x2Gm;
    AscendC::GlobalTensor<float> x3Gm;
    AscendC::GlobalTensor<float> x4Gm;
    AscendC::GlobalTensor<float> yGm;
    uint64_t len1;
    uint64_t len2;
    uint64_t len3;
    uint64_t len4;
    uint32_t tileSize;
};

extern "C" __global__ __aicore__ void googlenet_inception_module_custom(GM_ADDR x1, GM_ADDR x2, GM_ADDR x3, GM_ADDR x4, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConcatChannel op;
    op.Init(x1, x2, x3, x4, y,
            tiling_data.len1, tiling_data.len2, tiling_data.len3, tiling_data.len4,
            tiling_data.tileSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_1x1, reduce_3x3, out_3x3, reduce_5x5, out_5x5, pool_proj):
        super(ModelNew, self).__init__()
        self.branch1x1 = torch.nn.Conv2d(in_channels, out_1x1, kernel_size=1)

        self.branch3x3 = torch.nn.Sequential(
            torch.nn.Conv2d(in_channels, reduce_3x3, kernel_size=1),
            torch.nn.Conv2d(reduce_3x3, out_3x3, kernel_size=3, padding=1)
        )

        self.branch5x5 = torch.nn.Sequential(
            torch.nn.Conv2d(in_channels, reduce_5x5, kernel_size=1),
            torch.nn.Conv2d(reduce_5x5, out_5x5, kernel_size=5, padding=2)
        )

        self.branch_pool = torch.nn.Sequential(
            torch.nn.MaxPool2d(kernel_size=3, stride=1, padding=1),
            torch.nn.Conv2d(in_channels, pool_proj, kernel_size=1)
        )

    def forward(self, x):
        b1 = self.branch1x1(x)
        b2 = self.branch3x3(x)
        b3 = self.branch5x5(x)
        b4 = self.branch_pool(x)
        return custom_ops_lib.googlenet_inception_module_custom(b1, b2, b3, b4)
'''
