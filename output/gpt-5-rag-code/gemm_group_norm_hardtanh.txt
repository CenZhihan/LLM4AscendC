kernel_src="""
#include "kernel_operator.h"
using namespace AscendC;

extern "C" __global__ __aicore__ void gemm_group_norm_hardtanh_custom(
    GM_ADDR x,
    GM_ADDR weight,
    GM_ADDR bias,
    GM_ADDR gamma,
    GM_ADDR beta,
    GM_ADDR num_groups,
    GM_ADDR epsilon,
    GM_ADDR min_val,
    GM_ADDR max_val,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t M = tiling_data.M;
    const uint32_t K = tiling_data.K;
    const uint32_t N = tiling_data.N;

    // Cast GM pointers
    __gm__ float* x_gm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_gm = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* b_gm = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* gamma_gm = reinterpret_cast<__gm__ float*>(gamma);
    __gm__ float* beta_gm  = reinterpret_cast<__gm__ float*>(beta);
    __gm__ float* y_gm = reinterpret_cast<__gm__ float*>(y);

    // Scalar parameters from GM
    __gm__ int32_t* groups_gm = reinterpret_cast<__gm__ int32_t*>(num_groups);
    __gm__ float* eps_gm = reinterpret_cast<__gm__ float*>(epsilon);
    __gm__ float* min_gm = reinterpret_cast<__gm__ float*>(min_val);
    __gm__ float* max_gm = reinterpret_cast<__gm__ float*>(max_val);

    const int32_t G = groups_gm[0];
    const float eps = eps_gm[0];
    const float clip_min = min_gm[0];
    const float clip_max = max_gm[0];

    if (G <= 0 || N == 0 || K == 0 || M == 0) {
        return;
    }
    const uint32_t group_size = N / static_cast<uint32_t>(G);

    // Block-based row split
    uint32_t block_num = GetBlockNum();
    uint32_t block_idx = GetBlockIdx();
    uint32_t rows_per_block = (M + block_num - 1) / block_num;
    uint32_t m_start = block_idx * rows_per_block;
    uint32_t m_end = (m_start + rows_per_block < M) ? (m_start + rows_per_block) : M;

    for (uint32_t m = m_start; m < m_end; ++m) {
        // Process each group independently
        for (int32_t g = 0; g < G; ++g) {
            uint32_t c_start = static_cast<uint32_t>(g) * group_size;
            uint32_t c_end = c_start + group_size;
            if (c_end > N) c_end = N;

            // Pass 1: compute mean and variance for the group
            float sum = 0.0f;
            float sumsq = 0.0f;
            for (uint32_t n = c_start; n < c_end; ++n) {
                float acc = 0.0f;
                __gm__ float* w_row = w_gm + n * K;
                __gm__ float* x_row = x_gm + m * K;
                for (uint32_t k = 0; k < K; ++k) {
                    acc += x_row[k] * w_row[k];
                }
                acc += b_gm[n];
                sum += acc;
                sumsq += acc * acc;
            }
            float count = static_cast<float>(c_end - c_start);
            float mean = sum / count;
            float var = sumsq / count - mean * mean;
            if (var < 0.0f) var = 0.0f;
            float rstd = rsqrt(var + eps);

            // Pass 2: normalize, scale/shift and clamp, store to output
            for (uint32_t n = c_start; n < c_end; ++n) {
                float acc = 0.0f;
                __gm__ float* w_row = w_gm + n * K;
                __gm__ float* x_row = x_gm + m * K;
                for (uint32_t k = 0; k < K; ++k) {
                    acc += x_row[k] * w_row[k];
                }
                acc += b_gm[n];

                float normed = (acc - mean) * rstd;
                float scaled = normed * gamma_gm[n] + beta_gm[n];
                float clipped = scaled < clip_min ? clip_min : (scaled > clip_max ? clip_max : scaled);
                y_gm[m * N + n] = clipped;
            }
        }
    }
}
"""

at::Tensor gemm_group_norm_hardtanh_custom_impl_npu(
    const at::Tensor& x,
    const at::Tensor& weight,
    const at::Tensor& bias,
    const at::Tensor& gamma,
    const at::Tensor& beta,
    int64_t num_groups,
    double epsilon,
    double min_val,
    double max_val) {

    TORCH_CHECK(x.dim() == 2, "x must be 2D [M, K]");
    TORCH_CHECK(weight.dim() == 2, "weight must be 2D [N, K]");
    TORCH_CHECK(bias.dim() == 1, "bias must be 1D [N]");
    TORCH_CHECK(gamma.dim() == 1 && beta.dim() == 1, "gamma and beta must be 1D [N]");

    auto options = x.options();
    at::Tensor result = at::empty({x.size(0), weight.size(0)}, options);

    // Create scalar tensors on NPU
    at::Tensor num_groups_tensor = at::empty({1}, options.dtype(at::kInt));
    at::Tensor epsilon_tensor = at::empty({1}, options);
    at::Tensor min_tensor = at::empty({1}, options);
    at::Tensor max_tensor = at::empty({1}, options);

    num_groups_tensor.fill_(static_cast<int>(num_groups));
    epsilon_tensor.fill_(static_cast<float>(epsilon));
    min_tensor.fill_(static_cast<float>(min_val));
    max_tensor.fill_(static_cast<float>(max_val));

    EXEC_NPU_CMD(aclnnGemmGroupNormHardtanhCustom,
                 x, weight, bias, gamma, beta,
                 num_groups_tensor, epsilon_tensor, min_tensor, max_tensor,
                 result);
    return result;
}

    m.impl("gemm_group_norm_hardtanh_custom", &gemm_group_norm_hardtanh_custom_impl_npu);

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using fused custom AscendC kernel:
    Y = HardTanh(GroupNorm( X @ W^T + b, gamma, beta ))
    """
    def __init__(self, in_features, out_features, num_groups, hardtanh_min, hardtanh_max):
        super(ModelNew, self).__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features)
        self.hardtanh = torch.nn.Hardtanh(min_val=hardtanh_min, max_val=hardtanh_max)

    def forward(self, x):
        return custom_ops_lib.gemm_group_norm_hardtanh_custom(
            x,
            self.gemm.weight,    # [out_features, in_features]
            self.gemm.bias,      # [out_features]
            self.group_norm.weight,  # gamma [out_features]
            self.group_norm.bias,    # beta  [out_features]
            self.group_norm.num_groups,
            self.group_norm.eps,
            float(self.hardtanh.min_val),
            float(self.hardtanh.max_val),
        )
