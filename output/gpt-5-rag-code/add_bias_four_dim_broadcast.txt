static ge::graphStatus TilingFunc(gert::TilingContext* context)

    AddBiasFourDimBroadcastCustomTilingData tiling;

    const gert::Shape* xShape = context->GetInputShape(0);
    const gert::Shape* bShape = context->GetInputShape(1);
    if (xShape->GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }
    uint32_t n = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t c = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    uint32_t h = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
    uint32_t w = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));

    // bias should be length C
    if (bShape->GetOriginShape().GetShapeSize() != c) {
        // simple validation; runtime should ensure correctness
    }

    const uint32_t hw = h * w;

    // choose a reasonable channel tile for UB usage
    const uint32_t maxTileChannels = 32; // heuristic
    uint32_t tileChannels = (c < maxTileChannels) ? c : maxTileChannels;
    if (tileChannels == 0) tileChannels = 1;
    uint32_t tileLength = tileChannels * hw;

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_hw(hw);
    tiling.set_tileChannels(tileChannels);
    tiling.set_tileLength(tileLength);

    // set block dim based on batch
    uint32_t blockDim = (n >= 8) ? 8 : (n == 0 ? 1 : n);
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);

    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_B = float;
using DTYPE_Z = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelAddBiasFourDim {
public:
    __aicore__ inline KernelAddBiasFourDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t hw, uint32_t tileChannels, uint32_t tileLength)
    {
        this->N = n;
        this->C = c;
        this->H = h;
        this->W = w;
        this->HW = hw;
        this->tileChannels = tileChannels;
        this->tileLength = tileLength;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)((uint64_t)n * c * hw));
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)bias, c);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z, (uint32_t)((uint64_t)n * c * hw));

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueBChan, BUFFER_NUM, this->tileChannels * sizeof(DTYPE_B));
        pipe.InitBuffer(tmpQueueBiasExpanded, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // distribute N across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t basePerBlock = N / blockNum;
        uint32_t remainder = N % blockNum;
        uint32_t myN = basePerBlock + (blockIdx < remainder ? 1u : 0u);
        uint32_t startN = blockIdx * basePerBlock + (blockIdx < remainder ? blockIdx : remainder);

        for (uint32_t nIdx = 0; nIdx < myN; ++nIdx) {
            uint32_t nAbs = startN + nIdx;

            uint32_t cProcessed = 0;
            while (cProcessed < C) {
                uint32_t curTileC = tileChannels;
                if (curTileC > (C - cProcessed)) {
                    curTileC = (C - cProcessed);
                }
                uint32_t curTileLen = curTileC * HW;

                // Offsets in GM for this (n, c_tile) segment
                uint64_t offsetElems = ((uint64_t)nAbs * C + cProcessed) * HW;

                // CopyIn
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[offsetElems], curTileLen);
                inQueueX.EnQue(xLocal);

                AscendC::LocalTensor<DTYPE_B> bChanLocal = inQueueBChan.AllocTensor<DTYPE_B>();
                AscendC::DataCopy(bChanLocal, bGm[cProcessed], curTileC);
                inQueueBChan.EnQue(bChanLocal);

                // Compute
                AscendC::LocalTensor<DTYPE_X> xLocalC = inQueueX.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_B> bChanLocalC = inQueueBChan.DeQue<DTYPE_B>();
                AscendC::LocalTensor<DTYPE_Z> bExpanded = tmpQueueBiasExpanded.AllocTensor<DTYPE_Z>();

                // Expand bias channels across HW by duplicating each channel scalar into HW block
                for (uint32_t k = 0; k < curTileC; ++k) {
                    // get bias value for channel k in this tile
                    DTYPE_Z bval = bChanLocalC.GetValue(k);
                    AscendC::LocalTensor<DTYPE_Z> sub = bExpanded[k * HW];
                    AscendC::Duplicate(sub, bval, HW);
                }

                AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
                AscendC::Add(zLocal, xLocalC, bExpanded, curTileLen);

                outQueueZ.EnQue<DTYPE_Z>(zLocal);
                inQueueX.FreeTensor(xLocalC);
                inQueueBChan.FreeTensor(bChanLocalC);
                tmpQueueBiasExpanded.FreeTensor(bExpanded);

                // CopyOut
                AscendC::LocalTensor<DTYPE_Z> zLocalOut = outQueueZ.DeQue<DTYPE_Z>();
                AscendC::DataCopy(zGm[offsetElems], zLocalOut, curTileLen);
                outQueueZ.FreeTensor(zLocalOut);

                cProcessed += curTileC;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBChan;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueueBiasExpanded;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t N;
    uint32_t C;
    uint32_t H;
    uint32_t W;
    uint32_t HW;
    uint32_t tileChannels;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void add_bias_four_dim_broadcast_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAddBiasFourDim op;
    op.Init(x, bias, z,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.hw, tiling_data.tileChannels, tiling_data.tileLength);
    op.Process();
}
