kernel_src="""
#include "kernel_operator.h"

// This kernel implements 1D transposed convolution (stride=1, padding=0, groups=1, no bias)
// x: [N, Cin, Lin]
// w: [Cin, Cout, K]
// y: [N, Cout, Lout] where Lout = Lin + K - 1
// Note: This is a naive implementation focusing on correctness and tiling along the length dimension.

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed1d {
public:
    __aicore__ inline KernelConvTransposed1d() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t ksize, uint32_t lout, uint32_t tileLen)
    {
        this->n = n;
        this->cin = cin;
        this->cout = cout;
        this->lin = lin;
        this->ksize = ksize;
        this->lout = lout;
        this->tileLen = tileLen;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)n * cin * lin);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint64_t)cin * cout * ksize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint64_t)n * cout * lout);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Distribute output channels across blocks for parallelism
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockId = AscendC::GetBlockIdx();

        uint32_t coutPerBlock = (cout + blocks - 1) / blocks;
        uint32_t coutStart = coutPerBlock * blockId;
        uint32_t coutEnd = AscendC::Min(coutStart + coutPerBlock, cout);

        for (uint32_t b = 0; b < n; ++b) {
            for (uint32_t co = coutStart; co < coutEnd; ++co) {
                // Tile along output length
                for (uint32_t t0 = 0; t0 < lout; t0 += tileLen) {
                    uint32_t curLen = AscendC::Min(tileLen, lout - t0);

                    // Allocate and zero-initialize output tile
                    AscendC::LocalTensor<DTYPE_Y> yTile = outQueueY.AllocTensor<DTYPE_Y>();
                    ZeroLocal(yTile, curLen);

                    // Accumulate over input channels and kernel taps
                    for (uint32_t ci = 0; ci < cin; ++ci) {
                        // For each kernel position k
                        for (uint32_t k = 0; k < ksize; ++k) {
                            // Compute valid overlap range for input indices mapping to output [t0, t0+curLen)
                            // For output t, input index is xIdx = t - k
                            int64_t startT = (int64_t)t0;
                            int64_t endT = (int64_t)t0 + (int64_t)curLen - 1;
                            int64_t validStartT = AscendC::Max(startT, (int64_t)0);
                            int64_t validEndT = AscendC::Min(endT, (int64_t)lin - 1 + (int64_t)ksize - 1); // bounds before mapping
                            // Mapping constraints: xIdx in [0, lin-1] => t in [k, k + lin - 1]
                            validStartT = AscendC::Max(validStartT, (int64_t)k);
                            validEndT = AscendC::Min(validEndT, (int64_t)k + (int64_t)lin - 1);

                            if (validEndT < validStartT) {
                                continue;
                            }
                            uint32_t outRangeStart = (uint32_t)(validStartT - startT);
                            uint32_t outRangeLen = (uint32_t)(validEndT - validStartT + 1);
                            uint32_t xStartIdx = (uint32_t)(validStartT - (int64_t)k); // x index corresponding to validStartT

                            // Load contiguous x slice for this (b, ci) and kernel tap k
                            AscendC::LocalTensor<DTYPE_X> xSlice = inQueueX.AllocTensor<DTYPE_X>();
                            AscendC::DataCopy(xSlice, xBase(b, ci, xStartIdx), outRangeLen);

                            // Multiply by scalar weight w[ci, co, k]
                            DTYPE_W wScalar = wAt(ci, co, k);

                            AscendC::LocalTensor<DTYPE_Y> prod = tmpQueue.AllocTensor<DTYPE_Y>();
                            AscendC::Muls(prod, xSlice, wScalar, outRangeLen);

                            // Accumulate into yTile at proper offset
                            AscendC::Add(yTile[outRangeStart], yTile[outRangeStart], prod, outRangeLen);

                            tmpQueue.FreeTensor(prod);
                            inQueueX.FreeTensor(xSlice);
                        }
                    }

                    // Write yTile to GM
                    AscendC::DataCopy(yBase(b, co, t0), yTile, curLen);
                    outQueueY.FreeTensor(yTile);
                }
            }
        }
    }

private:
    __aicore__ inline void ZeroLocal(AscendC::LocalTensor<DTYPE_Y>& t, uint32_t len)
    {
        // Fill local tensor with zeros
        AscendC::Duplicate(t, (DTYPE_Y)0, len);
    }

    __aicore__ inline AscendC::GlobalTensor<DTYPE_X> xBase(uint32_t b, uint32_t ci, uint32_t xStart)
    {
        uint64_t offset = ((uint64_t)b * cin + ci) * lin + xStart;
        return xGm[offset];
    }

    __aicore__ inline AscendC::GlobalTensor<DTYPE_Y> yBase(uint32_t b, uint32_t co, uint32_t yStart)
    {
        uint64_t offset = ((uint64_t)b * cout + co) * lout + yStart;
        return yGm[offset];
    }

    __aicore__ inline DTYPE_W wAt(uint32_t ci, uint32_t co, uint32_t k)
    {
        uint64_t offset = ((uint64_t)ci * cout + co) * ksize + k;
        // Load scalar weight from GM
        // Using a temporary local of length 1 to fetch the scalar
        AscendC::LocalTensor<DTYPE_W> tmp = tmpQueue.AllocTensor<DTYPE_W>();
        AscendC::DataCopy(tmp, wGm[offset], 1);
        DTYPE_W val = tmp.GetValue(0);
        tmpQueue.FreeTensor(tmp);
        return val;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n;
    uint32_t cin;
    uint32_t cout;
    uint32_t lin;
    uint32_t ksize;
    uint32_t lout;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void conv_transposed_1d_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1d op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.ksize, tiling_data.lout, tiling_data.tileLen);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized 1D transposed convolution using custom AscendC kernel.
    Assumptions: stride=1, padding=0, output_padding=0, groups=1, bias=False.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, output_padding: int = 0, groups: int = 1, bias: bool = False):
        super().__init__()
        assert stride == 1, "Only stride=1 is supported in this custom kernel"
        assert padding == 0, "Only padding=0 is supported in this custom kernel"
        assert output_padding == 0, "Only output_padding=0 is supported in this custom kernel"
        assert groups == 1, "Only groups=1 is supported in this custom kernel"
        assert not bias, "bias=False is required in this custom kernel"
        self.in_channels = in_channels
        self.out_channels = out_channels
        self.kernel_size = kernel_size
        # PyTorch ConvTranspose1d uses weight shape [in_channels, out_channels, kernel_size]
        w = torch.empty(in_channels, out_channels, kernel_size)
        torch.nn.init.kaiming_uniform_(w, a=5**0.5)
        self.weight = torch.nn.Parameter(w)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_transposed_1d_custom(x, self.weight)
