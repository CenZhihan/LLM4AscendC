kernel_src="""
#include "kernel_operator.h"

__aicore__ inline float leaky_relu_val(float v, float negative_slope) {
    return v >= 0.f ? v : negative_slope * v;
}

extern "C" __global__ __aicore__ void conv_transpose3d_leaky_relu_multiply_leaky_relu_max_custom(GM_ADDR x, GM_ADDR multiplier, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    // Input: NCDHW
    const uint32_t N = tiling_data.n();
    const uint32_t C = tiling_data.c();
    const uint32_t D = tiling_data.d();
    const uint32_t H = tiling_data.h();
    const uint32_t W = tiling_data.w();
    const uint32_t OD = tiling_data.out_d();
    const uint32_t OH = tiling_data.out_h();
    const uint32_t OW = tiling_data.out_w();
    const float neg = tiling_data.negativeSlope();

    __gm__ float* x_ptr  = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* m_ptr  = reinterpret_cast<__gm__ float*>(multiplier);
    __gm__ float* y_ptr  = reinterpret_cast<__gm__ float*>(y);

    // Strides for NCDHW
    const uint64_t stride_w = 1ULL;
    const uint64_t stride_h = static_cast<uint64_t>(W);
    const uint64_t stride_d = static_cast<uint64_t>(H) * static_cast<uint64_t>(W);
    const uint64_t stride_c = static_cast<uint64_t>(D) * static_cast<uint64_t>(H) * static_cast<uint64_t>(W);
    const uint64_t stride_n = static_cast<uint64_t>(C) * static_cast<uint64_t>(D) * static_cast<uint64_t>(H) * static_cast<uint64_t>(W);

    const uint64_t out_stride_w = 1ULL;
    const uint64_t out_stride_h = static_cast<uint64_t>(OW);
    const uint64_t out_stride_d = static_cast<uint64_t>(OH) * static_cast<uint64_t>(OW);
    const uint64_t out_stride_c = static_cast<uint64_t>(OD) * static_cast<uint64_t>(OH) * static_cast<uint64_t>(OW);
    const uint64_t out_stride_n = static_cast<uint64_t>(C) * static_cast<uint64_t>(OD) * static_cast<uint64_t>(OH) * static_cast<uint64_t>(OW);

    // Note: This kernel assumes that input x is already the output of ConvTranspose3d.
    // It then applies: LeakyReLU -> multiply (broadcast over C) -> LeakyReLU -> 3D max pool with k=2,s=2.
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t c = 0; c < C; ++c) {
            float mul_val = m_ptr[c]; // multiplier is shaped (C,1,1,1) and broadcast
            for (uint32_t od = 0; od < OD; ++od) {
                uint32_t id = od * 2U;
                for (uint32_t oh = 0; oh < OH; ++oh) {
                    uint32_t ih = oh * 2U;
                    for (uint32_t ow = 0; ow < OW; ++ow) {
                        uint32_t iw = ow * 2U;

                        float max_val = -3.4e38f;
                        // Iterate over 2x2x2 window
                        for (uint32_t kd = 0; kd < 2U; ++kd) {
                            for (uint32_t kh = 0; kh < 2U; ++kh) {
                                for (uint32_t kw = 0; kw < 2U; ++kw) {
                                    uint32_t sd = id + kd;
                                    uint32_t sh = ih + kh;
                                    uint32_t sw = iw + kw;
                                    uint64_t idx = static_cast<uint64_t>(n) * stride_n
                                                 + static_cast<uint64_t>(c) * stride_c
                                                 + static_cast<uint64_t>(sd) * stride_d
                                                 + static_cast<uint64_t>(sh) * stride_h
                                                 + static_cast<uint64_t>(sw) * stride_w;

                                    float val = x_ptr[idx];
                                    // First LeakyReLU
                                    val = leaky_relu_val(val, neg);
                                    // Multiply by channel-wise scalar
                                    val = val * mul_val;
                                    // Second LeakyReLU
                                    val = leaky_relu_val(val, neg);

                                    if (val > max_val) {
                                        max_val = val;
                                    }
                                }
                            }
                        }

                        uint64_t out_idx = static_cast<uint64_t>(n) * out_stride_n
                                         + static_cast<uint64_t>(c) * out_stride_c
                                         + static_cast<uint64_t>(od) * out_stride_d
                                         + static_cast<uint64_t>(oh) * out_stride_h
                                         + static_cast<uint64_t>(ow) * out_stride_w;
                        y_ptr[out_idx] = max_val;
                    }
                }
            }
        }
    }
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keep ConvTranspose3d on framework, fuse LeakyReLU -> Multiply -> LeakyReLU -> MaxPool3d into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, multiplier_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(
            in_channels, out_channels, kernel_size,
            stride=stride, padding=padding, output_padding=output_padding
        )
        self.multiplier = torch.nn.Parameter(torch.randn(multiplier_shape))

    def forward(self, x):
        x = self.conv_transpose(x)
        # Fused: LeakyReLU(0.2) -> Multiply(channel-wise) -> LeakyReLU(0.2) -> MaxPool3d(k=2,s=2)
        x = custom_ops_lib.conv_transpose3d_leaky_relu_multiply_leaky_relu_max_custom(x, self.multiplier)
        return x
