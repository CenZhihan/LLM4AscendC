kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelBroadcastSubMean {
public:
    __aicore__ inline KernelBroadcastSubMean() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mean, GM_ADDR y,
                                uint32_t groups, uint32_t dhw,
                                uint32_t groupsPerBlock, uint32_t tileLen, uint32_t tileNum)
    {
        this->groups = groups;
        this->dhw = dhw;
        this->groupsPerBlock = groupsPerBlock;
        this->tileLen = tileLen;
        this->tileNum = tileNum;

        uint32_t startGroup = AscendC::GetBlockIdx() * groupsPerBlock;
        uint32_t endGroup = startGroup + groupsPerBlock;
        if (endGroup > groups) {
            endGroup = groups;
        }
        this->myStartGroup = startGroup;
        this->myGroupCount = (endGroup > startGroup) ? (endGroup - startGroup) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)startGroup * dhw, (uint64_t)this->myGroupCount * dhw);
        mGm.SetGlobalBuffer((__gm__ DTYPE_Y*)mean + startGroup, this->myGroupCount);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + (uint64_t)startGroup * dhw, (uint64_t)this->myGroupCount * dhw);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(tmpMeanQueue, 1, this->tileLen * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t g = 0; g < this->myGroupCount; ++g) {
            // Load mean value for this (n,c) group (1 element)
            DTYPE_Y meanScalar = LoadMeanScalar(g);

            // Prepare a local tensor filled with the mean scalar for broadcast subtraction
            for (uint32_t t = 0; t < this->tileNum; ++t) {
                uint32_t remain = this->dhw - t * this->tileLen;
                uint32_t curLen = (remain > this->tileLen) ? this->tileLen : remain;

                CopyIn(g, t, curLen);
                Compute(curLen, meanScalar);
                CopyOut(g, t, curLen);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t g, uint32_t tileIdx, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[g * this->dhw + tileIdx * this->tileLen], len);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len, DTYPE_Y meanScalar)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        AscendC::LocalTensor<DTYPE_Y> meanLocal = tmpMeanQueue.AllocTensor<DTYPE_Y>();

        // Duplicate mean scalar to vector
        AscendC::Dup(meanLocal, meanScalar, len);

        // y = x - mean
        AscendC::Sub(yLocal, xLocal, meanLocal, len);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        tmpMeanQueue.FreeTensor(meanLocal);
    }

    __aicore__ inline void CopyOut(uint32_t g, uint32_t tileIdx, uint32_t len)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[g * this->dhw + tileIdx * this->tileLen], yLocal, len);
        outQueueY.FreeTensor(yLocal);
    }

    __aicore__ inline DTYPE_Y LoadMeanScalar(uint32_t g)
    {
        // Load single mean element from GM
        AscendC::LocalTensor<DTYPE_Y> buf = tmpMeanQueue.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(buf, mGm[g], 1);
        // Read the first element. Using a workaround: duplicate then take first lane when used.
        // Here we return it as a scalar by using a vector-to-scalar intrinsic equivalent.
        // For simplicity, we assume returning the first lane via cast is supported.
        DTYPE_Y val = buf.GetValue(0);
        tmpMeanQueue.FreeTensor(buf);
        return val;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpMeanQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> mGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t groups;
    uint32_t dhw;
    uint32_t groupsPerBlock;
    uint32_t tileLen;
    uint32_t tileNum;

    uint32_t myStartGroup;
    uint32_t myGroupCount;
};

extern "C" __global__ __aicore__ void conv_transpose3d_batch_norm_subtract_custom(GM_ADDR x, GM_ADDR mean, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelBroadcastSubMean op;
    op.Init(x, mean, y,
            tiling_data.groups(), tiling_data.dhw(),
            tiling_data.groupsPerBlock(), tiling_data.tileLen(), tiling_data.tileNum());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, bias=True):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size,
                                                       stride=stride, padding=padding, bias=bias)
        self.batch_norm = torch.nn.BatchNorm3d(out_channels)

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.batch_norm(x)
        mean_nc = x.mean(dim=(2, 3, 4), keepdim=False)  # shape [N, C]
        y = custom_ops_lib.conv_transpose3d_batch_norm_subtract_custom(x, mean_nc)
        return y
