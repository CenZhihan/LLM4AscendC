kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE_X;
typedef float DTYPE_W;
typedef float DTYPE_B;
typedef float DTYPE_Y;

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulDivideGelu {
public:
    __aicore__ inline KernelMatmulDivideGelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileNum, float divisor)
    {
        this->m = m;
        this->n = n;
        this->k = k;
        this->tileNum = tileNum;
        this->totalLength = m * n;
        this->invDiv = divisor == 0.0f ? 0.0f : (1.0f / divisor);

        // Assign per-block ranges
        this->rowsPerBlock = (m + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        this->rowStart = AscendC::GetBlockIdx() * this->rowsPerBlock;
        this->rowEnd = AscendC::Min(this->rowStart + this->rowsPerBlock, (uint32_t)m);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(m * k));
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint32_t)(n * k));
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b, (uint32_t)(n));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint32_t)(m * n));

        // UB buffer init: rows of X and partial output tiles over N
        // We'll process output along N dimension in tiles to limit UB usage
        // Select a reasonable tile size for N
        this->tileN = 128; // can be tuned
        if (this->tileN > this->n) this->tileN = this->n;

        // Buffers
        // x row buffer length: k
        pipe.InitBuffer(xQueue, 1, this->k * sizeof(DTYPE_X));
        // weight tile buffer length: tileN * k
        pipe.InitBuffer(wQueue, 1, this->tileN * this->k * sizeof(DTYPE_W));
        // output tile buffer length: tileN
        pipe.InitBuffer(yQueue, BUFFER_NUM, this->tileN * sizeof(DTYPE_Y));
        // bias tile buffer length: tileN
        pipe.InitBuffer(bQueue, 1, this->tileN * sizeof(DTYPE_B));
    }

    __aicore__ inline void Process()
    {
        // For each row in this block range
        for (uint32_t row = this->rowStart; row < this->rowEnd; ++row) {
            // Load x row [k]
            AscendC::LocalTensor<DTYPE_X> xUb = xQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xUb, xGm[row * this->k], this->k);
            xQueue.EnQue(xUb);

            // Iterate N in tiles
            for (uint32_t colBase = 0; colBase < this->n; colBase += this->tileN) {
                uint32_t curTileN = AscendC::Min(this->tileN, this->n - colBase);

                // Load weight tile [curTileN, k]
                AscendC::LocalTensor<DTYPE_W> wUb = wQueue.AllocTensor<DTYPE_W>();
                // GM layout: weight is [n, k]
                AscendC::DataCopy(wUb, wGm[(colBase * this->k)], curTileN * this->k);
                wQueue.EnQue(wUb);

                // Load bias tile [curTileN]
                AscendC::LocalTensor<DTYPE_B> bUb = bQueue.AllocTensor<DTYPE_B>();
                AscendC::DataCopy(bUb, bGm[colBase], curTileN);
                bQueue.EnQue(bUb);

                // DeQue the loaded tensors
                AscendC::LocalTensor<DTYPE_X> xRow = xQueue.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_W> wTile = wQueue.DeQue<DTYPE_W>();
                AscendC::LocalTensor<DTYPE_B> bTile = bQueue.DeQue<DTYPE_B>();
                AscendC::LocalTensor<DTYPE_Y> yTile = yQueue.AllocTensor<DTYPE_Y>();

                // Initialize yTile with bias
                AscendC::DataCopy(yTile, bTile, curTileN);

                // GEMV accumulation: y[col] += dot(xRow, wTile[col, :])
                // Naive accumulation over K
                for (uint32_t kk = 0; kk < this->k; ++kk) {
                    // Load x scalar
                    // Broadcast xRow[kk] and accumulate yTile += xRow[kk] * wTile[:, kk]
                    // Use a temporary vector wCol = wTile[:, kk]
                    AscendC::LocalTensor<DTYPE_W> wCol; // view into wTile
                    wCol.SetGlobalBuffer(wTile.GetPhyAddr() + kk, curTileN, this->k);
                    AscendC::LocalTensor<DTYPE_Y> tmp = yQueue.AllocTensor<DTYPE_Y>();
                    // tmp = wCol * xRow[kk] + yTile
                    AscendC::Muls(tmp, wCol, xRow.GetValue(kk), curTileN);
                    AscendC::Add(yTile, yTile, tmp, curTileN);
                    yQueue.FreeTensor(tmp);
                }

                // Post-process: divide by divisor and apply GELU
                // z = yTile * invDiv
                AscendC::Muls(yTile, yTile, this->invDiv, curTileN);

                // GELU approximation: 0.5*z*(1 + tanh( sqrt(2/pi)*(z + 0.044715*z^3) ))
                const float kAlpha = 0.7978845608f; // sqrt(2/pi)
                const float kCoeff = 0.044715f;
                AscendC::LocalTensor<DTYPE_Y> z = yQueue.AllocTensor<DTYPE_Y>();
                AscendC::DataCopy(z, yTile, curTileN);

                AscendC::LocalTensor<DTYPE_Y> z2 = yQueue.AllocTensor<DTYPE_Y>();
                AscendC::Mul(z2, z, z, curTileN); // z^2

                AscendC::LocalTensor<DTYPE_Y> z3 = yQueue.AllocTensor<DTYPE_Y>();
                AscendC::Mul(z3, z2, z, curTileN); // z^3

                AscendC::LocalTensor<DTYPE_Y> inner = yQueue.AllocTensor<DTYPE_Y>();
                AscendC::Muls(inner, z3, kCoeff, curTileN);  // 0.044715*z^3
                AscendC::Add(inner, inner, z, curTileN);     // z + 0.044715*z^3
                AscendC::Muls(inner, inner, kAlpha, curTileN); // sqrt(2/pi)*(...)

                AscendC::LocalTensor<DTYPE_Y> t = yQueue.AllocTensor<DTYPE_Y>();
                AscendC::Tanh(t, inner, curTileN); // tanh(inner)

                // yTile = 0.5 * z * (1 + t)
                AscendC::LocalTensor<DTYPE_Y> onePlusT = yQueue.AllocTensor<DTYPE_Y>();
                AscendC::Adds(onePlusT, t, 1.0f, curTileN);
                AscendC::Mul(yTile, z, onePlusT, curTileN);
                AscendC::Muls(yTile, yTile, 0.5f, curTileN);

                // Free temps
                yQueue.FreeTensor(z);
                yQueue.FreeTensor(z2);
                yQueue.FreeTensor(z3);
                yQueue.FreeTensor(inner);
                yQueue.FreeTensor(t);
                yQueue.FreeTensor(onePlusT);

                // Enqueue yTile for copy-out
                yQueue.EnQue(yTile);

                // Copy out to GM y[row, colBase:colBase+curTileN]
                AscendC::LocalTensor<DTYPE_Y> yTileOut = yQueue.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yGm[row * this->n + colBase], yTileOut, curTileN);
                yQueue.FreeTensor(yTileOut);

                // Free used inputs
                wQueue.FreeTensor(wTile);
                bQueue.FreeTensor(bTile);
                // xRow kept for entire row
            }

            xQueue.FreeTensor(xRow);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t m;
    uint32_t n;
    uint32_t k;
    uint32_t tileNum;
    uint32_t totalLength;
    uint32_t tileN;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;

    float invDiv;
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulDivideGelu op;
    op.Init(x, weight, bias, y, tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileNum, tiling_data.divisor);
    op.Process();
}
