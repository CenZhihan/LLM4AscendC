kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2;

class KernelAvgPool2D_2x2S2 {
public:
    __aicore__ inline KernelAvgPool2D_2x2S2() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t totalOutLength, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->h_out = h / 2;
        this->w_out = w / 2;
        this->totalOut = totalOutLength;

        this->blockLength = totalOutLength / AscendC::GetBlockNum();
        this->tileNum = tileNum > 0 ? tileNum : 1;
        this->tileLength = (this->blockLength / this->tileNum / BUFFER_NUM);
        if (this->tileLength == 0) {
            this->tileLength = 1;
        }

        // Set GM buffers starting at block offset for output to avoid overlap
        uint32_t blockBase = this->blockLength * AscendC::GetBlockIdx();
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint32_t>(n * c * h * w));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + blockBase, this->blockLength);

        pipe.InitBuffer(inRowBuf, BUFFER_NUM, 2 * sizeof(DTYPE_X));   // 2 elements per row
        pipe.InitBuffer(outBuf, BUFFER_NUM, 1 * sizeof(DTYPE_Y));     // 1 element per output
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            ComputeTile(i);
        }
    }

private:
    __aicore__ inline void ComputeTile(int32_t progress)
    {
        uint32_t startOutIndex = progress * this->tileLength;
        if (startOutIndex >= this->blockLength) {
            return;
        }
        uint32_t valid = this->tileLength;
        if (startOutIndex + valid > this->blockLength) {
            valid = this->blockLength - startOutIndex;
        }

        for (uint32_t t = 0; t < valid; ++t) {
            uint32_t globalOutIndex = blockBaseOut() + startOutIndex + t;

            // Map flattened output index to (n, c, oh, ow)
            uint32_t tmp = globalOutIndex;
            uint32_t ow = tmp % w_out; tmp /= w_out;
            uint32_t oh = tmp % h_out; tmp /= h_out;
            uint32_t cc = tmp % c;     tmp /= c;
            uint32_t nn = tmp;

            // Compute input base indices
            uint32_t h0 = oh * 2;
            uint32_t w0 = ow * 2;

            uint32_t in_base_row0 = (((nn * c + cc) * h + h0) * w + w0);
            uint32_t in_base_row1 = in_base_row0 + w; // next row, same col

            // Load 2 elements from two rows
            AscendC::LocalTensor<DTYPE_X> row0 = inRowBuf.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> row1 = inRowBuf.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(row0, xGm[in_base_row0], 2);
            AscendC::DataCopy(row1, xGm[in_base_row1], 2);

            // Enqueue and dequeue to get local pointers
            inRowBuf.EnQue(row0);
            inRowBuf.EnQue(row1);
            AscendC::LocalTensor<DTYPE_X> r0 = inRowBuf.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> r1 = inRowBuf.DeQue<DTYPE_X>();

            // Compute average of 2x2 patch
            AscendC::LocalTensor<DTYPE_Y> outLocal = outBuf.AllocTensor<DTYPE_Y>();
            // Manually compute into first element
            // Note: Assume LocalTensor supports direct element access as a pointer-like array in device
            DTYPE_X v00 = r0.GetValue(0);
            DTYPE_X v01 = r0.GetValue(1);
            DTYPE_X v10 = r1.GetValue(0);
            DTYPE_X v11 = r1.GetValue(1);
            DTYPE_Y avg = static_cast<DTYPE_Y>((v00 + v01 + v10 + v11) * 0.25f);
            outLocal.SetValue(0, avg);

            // Release input buffers
            inRowBuf.FreeTensor(r0);
            inRowBuf.FreeTensor(r1);

            // Enqueue output
            outBuf.EnQue(outLocal);

            // Dequeue and write to GM
            AscendC::LocalTensor<DTYPE_Y> outDeq = outBuf.DeQue<DTYPE_Y>();
            AscendC::DataCopy(yGm[startOutIndex + t], outDeq, 1);
            outBuf.FreeTensor(outDeq);
        }
    }

    __aicore__ inline uint32_t blockBaseOut() const {
        return this->blockLength * AscendC::GetBlockIdx();
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inRowBuf;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outBuf;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t h;
    uint32_t w;
    uint32_t h_out;
    uint32_t w_out;

    uint32_t totalOut;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void densenet121_transition_layer_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAvgPool2D_2x2S2 op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.totalOutLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, num_input_features: int, num_output_features: int):
        super().__init__()
        # Keep modules to use their parameters; computation will be partially offloaded
        self.bn = nn.BatchNorm2d(num_input_features)
        self.relu = nn.ReLU(inplace=True)
        self.conv = nn.Conv2d(num_input_features, num_output_features, kernel_size=1, bias=False)

    def forward(self, x):
        # BN -> ReLU -> 1x1 Conv as in original transition layer
        x = self.bn(x)
        x = self.relu(x)
        x = self.conv(x)
        # Replace AvgPool2d(kernel=2, stride=2) with custom AscendC kernel
        y = custom_ops_lib.densenet121_transition_layer_custom(x)
        return y
