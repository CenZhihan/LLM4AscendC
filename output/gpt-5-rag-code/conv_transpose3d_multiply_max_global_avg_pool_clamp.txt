kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;  // double buffer for input tiles
constexpr uint32_t DEFAULT_TILE_LEN = 1024;

class KernelGAPClamp {
public:
    __aicore__ inline KernelGAPClamp() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t N, uint32_t C, uint32_t D, uint32_t H, uint32_t W, uint32_t tileLen)
    {
        this->N = N;
        this->C = C;
        this->D = D;
        this->H = H;
        this->W = W;
        this->groups = N * C;
        this->spatial = D * H * W;
        this->tileLen = tileLen > 0 ? tileLen : DEFAULT_TILE_LEN;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)N * C * D * H * W);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint64_t)N * C); // one value per (N,C)

        pipe.InitBuffer(inQueueX, IN_BUF_NUM, this->tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, 1, 1 * sizeof(DTYPE_Y)); // 1-element result per (n,c)
        pipe.InitBuffer(tmpQueue, 1, 1 * sizeof(DTYPE_Y));  // temporary 1-element tensors
    }

    __aicore__ inline void Process()
    {
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockId = AscendC::GetBlockIdx();

        // ceil-div to split (N*C) groups across blocks
        uint32_t groupsPerBlock = (groups + blocks - 1) / blocks;
        uint32_t startIdx = blockId * groupsPerBlock;
        uint32_t endIdx = AscendC::Min(startIdx + groupsPerBlock, groups);

        for (uint32_t g = startIdx; g < endIdx; ++g) {
            ComputeOneGroup(g);
        }
    }

private:
    __aicore__ inline void ReduceToSingle(AscendC::LocalTensor<DTYPE_X> &buf, uint32_t validLen)
    {
        uint32_t curr = validLen;
        while (curr > 1) {
            uint32_t half = curr >> 1;
            if (half > 0) {
                // buf[0:half] += buf[half:half]
                AscendC::Add(buf, buf, buf[half], half);
            }
            if (curr & 1) {
                // add leftover last element into first element
                AscendC::Add(buf, buf, buf[curr - 1], 1);
            }
            curr = half + (curr & 1);
        }
    }

    __aicore__ inline void ComputeOneGroup(uint32_t groupIdx)
    {
        // groupIdx corresponds to (n, c) pair; spatial elements contiguous in memory
        uint64_t baseOffset = (uint64_t)groupIdx * (uint64_t)spatial;

        // sumLocal will hold the running sum (1 element)
        AscendC::LocalTensor<DTYPE_Y> sumLocal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::Dup(sumLocal, (DTYPE_Y)0.0f, 1);

        uint32_t remaining = spatial;
        uint64_t offset = 0;
        int32_t tileIdx = 0;
        while (remaining > 0) {
            uint32_t currLen = remaining > tileLen ? tileLen : remaining;

            // Copy in a tile
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[baseOffset + offset], currLen);
            inQueueX.EnQue(xLocal);

            // Compute partial sum
            AscendC::LocalTensor<DTYPE_X> xTile = inQueueX.DeQue<DTYPE_X>();
            ReduceToSingle(xTile, currLen); // reduce to one value at xTile[0]
            // sumLocal = sumLocal + xTile[0]
            AscendC::Add(sumLocal, sumLocal, xTile, 1);

            inQueueX.FreeTensor(xTile);

            remaining -= currLen;
            offset += currLen;
            tileIdx++;
        }

        // Divide by spatial size: sumLocal *= (1.0 / spatial)
        AscendC::LocalTensor<DTYPE_Y> coefLocal = tmpQueue.AllocTensor<DTYPE_Y>();
        DTYPE_Y coef = (DTYPE_Y)(1.0f / (float)spatial);
        AscendC::Dup(coefLocal, coef, 1);
        AscendC::Mul(sumLocal, sumLocal, coefLocal, 1);
        tmpQueue.FreeTensor(coefLocal);

        // Clamp to [0, 1]
        AscendC::LocalTensor<DTYPE_Y> zeroLocal = tmpQueue.AllocTensor<DTYPE_Y>();
        AscendC::Dup(zeroLocal, (DTYPE_Y)0.0f, 1);
        AscendC::Max(sumLocal, sumLocal, zeroLocal, 1);
        tmpQueue.FreeTensor(zeroLocal);

        AscendC::LocalTensor<DTYPE_Y> oneLocal = tmpQueue.AllocTensor<DTYPE_Y>();
        AscendC::Dup(oneLocal, (DTYPE_Y)1.0f, 1);
        AscendC::Min(sumLocal, sumLocal, oneLocal, 1);
        tmpQueue.FreeTensor(oneLocal);

        // Enqueue result and write to GM at index groupIdx
        outQueueY.EnQue(sumLocal);

        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[groupIdx], yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, C, D, H, W;
    uint32_t groups;
    uint32_t spatial;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void conv_transpose3d_multiply_max_global_avg_pool_clamp_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGAPClamp op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.tileLen);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized: keep ConvTranspose3d and MaxPool3d in PyTorch, fuse global average pooling + clamp into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, scale, maxpool_kernel_size):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding)
        self.scale = scale
        self.maxpool = torch.nn.MaxPool3d(kernel_size=maxpool_kernel_size)
        self.clamp_min = 0.0
        self.clamp_max = 1.0

    def forward(self, x):
        x = self.conv_transpose(x)
        x = x * self.scale
        x = self.maxpool(x)
        # Fused global average pooling to (1,1,1) and clamp to [0,1]
        x = custom_ops_lib.conv_transpose3d_multiply_max_global_avg_pool_clamp_custom(x)
        return x
