kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelResBlockAddRelu {
public:
    __aicore__ inline KernelResBlockAddRelu() {}
    __aicore__ inline void Init(GM_ADDR main, GM_ADDR skip, GM_ADDR out, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        mainGm.SetGlobalBuffer((__gm__ DTYPE_X *)main + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        skipGm.SetGlobalBuffer((__gm__ DTYPE_Y *)skip + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z *)out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueMain, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueSkip, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueOut, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> mainLocal = inQueueMain.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> skipLocal = inQueueSkip.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(mainLocal, mainGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(skipLocal, skipGm[progress * this->tileLength], this->tileLength);
        inQueueMain.EnQue(mainLocal);
        inQueueSkip.EnQue(skipLocal);
    }

    __aicore__ inline void Compute(int32_t)
    {
        AscendC::LocalTensor<DTYPE_X> mainLocal = inQueueMain.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> skipLocal = inQueueSkip.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.AllocTensor<DTYPE_Z>();

        // outLocal = mainLocal + skipLocal
        AscendC::Add(outLocal, mainLocal, skipLocal, this->tileLength);
        // ReLU
        AscendC::Relu(outLocal, outLocal, this->tileLength);

        outQueueOut.EnQue<DTYPE_Z>(outLocal);
        inQueueMain.FreeTensor(mainLocal);
        inQueueSkip.FreeTensor(skipLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> outLocal = outQueueOut.DeQue<DTYPE_Z>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        outQueueOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMain, inQueueSkip;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueOut;
    AscendC::GlobalTensor<DTYPE_X> mainGm;
    AscendC::GlobalTensor<DTYPE_Y> skipGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void resnet_basic_block_custom(GM_ADDR main, GM_ADDR skip, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelResBlockAddRelu op;
    op.Init(main, skip, out, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    expansion = 1

    def __init__(self, in_channels, out_channels, stride=1):
        super(ModelNew, self).__init__()
        self.conv1 = nn.Conv2d(in_channels, out_channels, kernel_size=3, stride=stride, padding=1, bias=False)
        self.bn1 = nn.BatchNorm2d(out_channels)
        self.relu = nn.ReLU(inplace=True)
        self.conv2 = nn.Conv2d(out_channels, out_channels, kernel_size=3, stride=1, padding=1, bias=False)
        self.bn2 = nn.BatchNorm2d(out_channels)
        self.downsample = nn.Sequential(
            nn.Conv2d(in_channels, out_channels * self.expansion, kernel_size=1, stride=stride, bias=False),
            nn.BatchNorm2d(out_channels * self.expansion),
        )
        self.stride = stride

    def forward(self, x):
        identity = self.downsample(x)

        out = self.conv1(x)
        out = self.bn1(out)
        out = self.relu(out)

        out = self.conv2(out)
        out = self.bn2(out)

        out = custom_ops_lib.resnet_basic_block_custom(out, identity)
        return out
'''
