kernel_src="""
#include "kernel_operator.h"
#include <cmath>

class KernelFusedGemmBnGeluRelu {
public:
    __aicore__ inline KernelFusedGemmBnGeluRelu() {}
    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR w,
                                GM_ADDR b,
                                GM_ADDR gamma,
                                GM_ADDR beta,
                                GM_ADDR running_mean,
                                GM_ADDR running_var,
                                GM_ADDR epsilon,
                                GM_ADDR y,
                                uint32_t N, uint32_t K, uint32_t M)
    {
        this->N = N;
        this->K = K;
        this->M = M;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint32_t)N * K);
        wGm.SetGlobalBuffer((__gm__ float*)w, (uint32_t)M * K);
        bGm.SetGlobalBuffer((__gm__ float*)b, M);
        gammaGm.SetGlobalBuffer((__gm__ float*)gamma, M);
        betaGm.SetGlobalBuffer((__gm__ float*)beta, M);
        meanGm.SetGlobalBuffer((__gm__ float*)running_mean, M);
        varGm.SetGlobalBuffer((__gm__ float*)running_var, M);
        epsGm.SetGlobalBuffer((__gm__ float*)epsilon, 1);
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint32_t)N * M);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (N + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > N) rowEnd = N;

        float eps = epsGm.GetValue(0);

        for (uint32_t n = rowStart; n < rowEnd; ++n) {
            for (uint32_t m = 0; m < M; ++m) {
                float acc = 0.f;
                // GEMM: y[n, m] = x[n, :] dot w[m, :] + b[m]
                for (uint32_t k = 0; k < K; ++k) {
                    float xv = xGm.GetValue(n * K + k);
                    float wv = wGm.GetValue(m * K + k);
                    acc += xv * wv;
                }
                float val = acc + bGm.GetValue(m);

                // BatchNorm inference: gamma * (val - mean) / sqrt(var + eps) + beta
                float mean = meanGm.GetValue(m);
                float var = varGm.GetValue(m);
                float inv_std = 1.0f / sqrtf(var + eps);
                float bn = (val - mean) * inv_std;
                bn = bn * gammaGm.GetValue(m) + betaGm.GetValue(m);

                // GELU approximation using tanh
                float x3 = bn * bn * bn;
                const float kAlpha = 0.7978845608f; // sqrt(2/pi)
                const float kBeta = 0.044715f;
                float gelu = 0.5f * bn * (1.0f + tanhf(kAlpha * (bn + kBeta * x3)));

                // ReLU
                float relu = gelu > 0.f ? gelu : 0.f;

                yGm.SetValue(n * M + m, relu);
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> gammaGm;
    AscendC::GlobalTensor<float> betaGm;
    AscendC::GlobalTensor<float> meanGm;
    AscendC::GlobalTensor<float> varGm;
    AscendC::GlobalTensor<float> epsGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t N;
    uint32_t K;
    uint32_t M;
};

extern "C" __global__ __aicore__ void gemm_batch_norm_gelu_group_norm_mean_relu_custom(
    GM_ADDR x,
    GM_ADDR weight,
    GM_ADDR bias,
    GM_ADDR bn_scale,
    GM_ADDR bn_bias,
    GM_ADDR running_mean,
    GM_ADDR running_var,
    GM_ADDR epsilon,
    GM_ADDR y,
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedGemmBnGeluRelu op;
    op.Init(x, weight, bias, bn_scale, bn_bias, running_mean, running_var, epsilon,
            y, tiling_data.N, tiling_data.K, tiling_data.M);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.batch_norm = torch.nn.BatchNorm1d(out_features)

    def forward(self, x):
        # weight: [out, in], kernel expects [M, K] where M=out_features, K=in_features
        # BatchNorm parameters are taken from the module (inference-style)
        eps_tensor = torch.tensor([self.batch_norm.eps], dtype=x.dtype, device=x.device)
        return custom_ops_lib.gemm_batch_norm_gelu_group_norm_mean_relu_custom(
            x,
            self.gemm.weight,        # [out, in]
            self.gemm.bias,          # [out]
            self.batch_norm.weight,  # gamma [out]
            self.batch_norm.bias,    # beta  [out]
            self.batch_norm.running_mean,  # [out]
            self.batch_norm.running_var,   # [out]
            eps_tensor                # [1]
        )
'''
