inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1) / b);
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    GemmSigmoidSumLogSumExpCustomTilingData tiling;

    const auto* inShape = context->GetInputShape(0);
    auto shapeVec = inShape->GetOriginShape().GetDims();

    if (shapeVec.size() != 2) {
        return ge::GRAPH_FAILED;
    }

    uint32_t rows = static_cast<uint32_t>(shapeVec[0]);
    uint32_t cols = static_cast<uint32_t>(shapeVec[1]);

    context->SetBlockDim(BLOCK_DIM);

    uint32_t rowsPerBlock = CeilDiv(rows, BLOCK_DIM);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_rowsPerBlock(rowsPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* workspaces = context->GetWorkspaceSizes(1);

}

kernel_src="""
#include "kernel_operator.h"
#include <cfloat>
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSumExpRows {
public:
    __aicore__ inline KernelLogSumExpRows() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t rows, uint32_t cols, uint32_t rowsPerBlock)
    {
        this->rows = rows;
        this->cols = cols;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t startRow = (rows / blockNum) * blockIdx + AscendC::min((uint32_t)(rows % blockNum), (uint32_t)blockIdx);
        uint32_t endRow = (rows / blockNum) * (blockIdx + 1) + AscendC::min((uint32_t)(rows % blockNum), (uint32_t)(blockIdx + 1));
        this->localRows = endRow - startRow;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + startRow * cols, this->localRows * cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + startRow, this->localRows);

        tileLen = cols;

        pipe.InitBuffer(inQueueX, BUFFER_NUM, tileLen * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, 1 * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, tileLen * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < localRows; ++r) {
            CopyIn(r);
            Compute();
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * tileLen], tileLen);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_X> tmpLocal = tmpQueue.AllocTensor<DTYPE_X>();

        // 1) Find max over features for numerical stability
        float maxVal = -FLT_MAX;
        for (uint32_t i = 0; i < tileLen; ++i) {
            float v = xLocal.GetValue(i);
            maxVal = v > maxVal ? v : maxVal;
        }

        // 2) Subtract max, exp, and sum
        float sumExp = 0.f;
        for (uint32_t i = 0; i < tileLen; ++i) {
            float shifted = xLocal.GetValue(i) - maxVal;
            float ev = expf(shifted);
            tmpLocal.SetValue(i, ev);
            sumExp += ev;
        }

        // 3) y = max + log(sumExp)
        float outv = logf(sumExp) + maxVal;
        yLocal.SetValue(0, outv);

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[rowIdx], yLocal, 1);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t localRows;
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void gemm_sigmoid_sum_log_sum_exp_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(t, tiling);
    KernelLogSumExpRows op;
    op.Init(x, y, t.rows, t.cols, t.rowsPerBlock);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: retains Linear and Sigmoid in PyTorch,
    replaces final LogSumExp with a custom AscendC kernel.
    """
    def __init__(self, input_size, hidden_size, output_size):
        super(ModelNew, self).__init__()
        self.linear1 = torch.nn.Linear(input_size, hidden_size)
        self.linear2 = torch.nn.Linear(hidden_size, output_size)

    def forward(self, x):
        x = self.linear1(x)
        x = torch.sigmoid(x)
        x = self.linear2(x)
        # Custom kernel computes logsumexp over features (dim=1)
        x = custom_ops_lib.gemm_sigmoid_sum_log_sum_exp_custom(x)
        return x
