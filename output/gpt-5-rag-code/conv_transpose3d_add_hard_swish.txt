kernel_src="""
#include "kernel_operator.h"

// Fused elementwise kernel: z = t * hardswish(t) where t = x + add_input
// Here, hardswish(t) = t * relu6(t + 3) / 6, so final is t^2 * relu6(t + 3) / 6
// relu6(u) is implemented as relu(u) - relu(u - 6)

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose3dAddHardSwish {
public:
    __aicore__ inline KernelConvTranspose3dAddHardSwish() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add_in, GM_ADDR z, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        addGm.SetGlobalBuffer((__gm__ DTYPE_Y *)add_in + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueAdd, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> addLocal = inQueueAdd.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(addLocal, addGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueAdd.EnQue(addLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        // xLocal: will be reused to store t^2
        // addLocal: will be reused as scratch
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> addLocal = inQueueAdd.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // t = x + add_input -> put in zLocal
        AscendC::Add(zLocal, xLocal, addLocal, this->tileLength);
        // t^2 -> store in xLocal (overwrite xLocal)
        AscendC::Mul(xLocal, zLocal, zLocal, this->tileLength);

        // u = t + 3 -> store in addLocal
        AscendC::Adds(addLocal, zLocal, static_cast<DTYPE_Z>(3.0f), this->tileLength);
        // relu(u) -> store in zLocal
        AscendC::Relu(zLocal, addLocal, this->tileLength);
        // u - 6 -> keep in addLocal
        AscendC::Adds(addLocal, addLocal, static_cast<DTYPE_Z>(-6.0f), this->tileLength);
        // relu(u - 6) -> in addLocal
        AscendC::Relu(addLocal, addLocal, this->tileLength);
        // relu6(u) = relu(u) - relu(u - 6) -> in zLocal
        AscendC::Sub(zLocal, zLocal, addLocal, this->tileLength);

        // t^2 * relu6(u) -> in zLocal
        AscendC::Mul(zLocal, xLocal, zLocal, this->tileLength);
        // multiply by 1/6
        AscendC::Muls(zLocal, zLocal, static_cast<DTYPE_Z>(1.0f / 6.0f), this->tileLength);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueAdd.FreeTensor(addLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueAdd;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> addGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_add_hard_swish_custom(GM_ADDR x, GM_ADDR add_input, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dAddHardSwish op;
    op.Init(x, add_input, z, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Optimized model: uses custom AscendC kernel to fuse add + HardSwish
    after ConvTranspose3d. The ConvTranspose3d itself stays as a standard
    module for weight management and shape inference.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape):
        super(ModelNew, self).__init__()
        self.conv_transpose = nn.ConvTranspose3d(
            in_channels,
            out_channels,
            kernel_size,
            stride=stride,
            padding=padding,
            output_padding=output_padding
        )
        # Keep the original unused bias parameter to match state_dict if needed
        self.bias = nn.Parameter(torch.randn(bias_shape))

    def forward(self, x, add_input):
        # Standard deconvolution
        x = self.conv_transpose(x)
        # Fused: (x + add_input) * hardswish(x + add_input)
        x = custom_ops_lib.conv_transpose3d_add_hard_swish_custom(x, add_input)
        return x

# Example input constructors retained from the original specification
batch_size = 128
in_channels = 32
out_channels = 64
D, H, W = 16, 16, 16
kernel_size = 3
stride = 2
padding = 1
output_padding = 1
bias_shape = (out_channels, 1, 1, 1, 1)

def get_inputs():
    return [
        torch.rand(batch_size, in_channels, D, H, W),
        torch.rand(batch_size, out_channels, D*stride, H*stride, W*stride),
    ]

def get_init_inputs():
    return [in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape]
