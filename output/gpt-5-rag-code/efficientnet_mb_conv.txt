kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelEfficientnetMbConvCustom {
public:
    __aicore__ inline KernelEfficientnetMbConvCustom() : useResidual(false) {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR identity, GM_ADDR z, GM_ADDR flag, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        idGm.SetGlobalBuffer((__gm__ DTYPE_Y *)identity + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        flagGm.SetGlobalBuffer((__gm__ uint32_t *)flag, 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueId, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));

        // Load flag once
        AscendC::LocalTensor<uint32_t> flagLocal = pipe.AllocTmpBuffer<uint32_t>();
        AscendC::DataCopy(flagLocal, flagGm[0], 1);
        useResidual = (flagLocal.GetValue(0) != 0);
        pipe.FreeTmpBuffer(flagLocal);
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> idLocal = inQueueId.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(idLocal, idGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueId.EnQue(idLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> idLocal = inQueueId.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        if (useResidual) {
            AscendC::Add(zLocal, xLocal, idLocal, this->tileLength);
        } else {
            AscendC::DataCopy(zLocal, xLocal, this->tileLength);
        }

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueId.FreeTensor(idLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[progress * this->tileLength], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueId;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> idGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    AscendC::GlobalTensor<uint32_t> flagGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    bool useResidual;
};

extern "C" __global__ __aicore__ void efficientnet_mb_conv_custom(GM_ADDR x, GM_ADDR identity, GM_ADDR use_residual, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelEfficientnetMbConvCustom op;
    op.Init(x, identity, z, use_residual, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, stride, expand_ratio):
        super(ModelNew, self).__init__()
        self.use_residual = (stride == 1 and in_channels == out_channels)
        hidden_dim = in_channels * expand_ratio

        if expand_ratio != 1:
            self.expand_conv = nn.Sequential(
                nn.Conv2d(in_channels, hidden_dim, kernel_size=1, stride=1, padding=0, bias=False),
                nn.BatchNorm2d(hidden_dim),
                nn.ReLU6(inplace=True)
            )
        else:
            # To keep attribute existence consistent with original logic
            self.expand_conv = None

        self.depthwise_conv = nn.Sequential(
            nn.Conv2d(hidden_dim, hidden_dim, kernel_size=kernel_size, stride=stride, padding=(kernel_size-1)//2, groups=hidden_dim, bias=False),
            nn.BatchNorm2d(hidden_dim),
            nn.ReLU6(inplace=True)
        )

        self.project_conv = nn.Sequential(
            nn.Conv2d(hidden_dim, out_channels, kernel_size=1, stride=1, padding=0, bias=False),
            nn.BatchNorm2d(out_channels)
        )

    def forward(self, x):
        identity = x
        if self.expand_conv is not None:
            x = self.expand_conv(x)
        x = self.depthwise_conv(x)
        x = self.project_conv(x)

        if self.use_residual:
            # Create a scalar flag tensor (int32) on device to control residual
            use_residual_flag = torch.tensor([1], dtype=torch.int32, device=x.device)
            x = custom_ops_lib.efficientnet_mb_conv_custom(x, identity, use_residual_flag)
        return x
'''
