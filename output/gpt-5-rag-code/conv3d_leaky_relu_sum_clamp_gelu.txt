kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using DTYPE = float;

class KernelConv3dLeakyReluSumClampGelu {
public:
    __aicore__ inline KernelConv3dLeakyReluSumClampGelu() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR w,
                                GM_ADDR b,
                                GM_ADDR s,
                                GM_ADDR y,
                                uint32_t n, uint32_t cin,
                                uint32_t din, uint32_t hin, uint32_t win,
                                uint32_t cout,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout,
                                uint32_t totalOutputLength,
                                float negativeSlope,
                                float clampMin, float clampMax)
    {
        xGm = reinterpret_cast<__gm__ const DTYPE*>(x);
        wGm = reinterpret_cast<__gm__ const DTYPE*>(w);
        bGm = reinterpret_cast<__gm__ const DTYPE*>(b);
        sGm = reinterpret_cast<__gm__ const DTYPE*>(s);
        yGm = reinterpret_cast<__gm__ DTYPE*>(y);

        this->N = n; this->Cin = cin; this->Din = din; this->Hin = hin; this->Win = win;
        this->Cout = cout; this->Kd = kd; this->Kh = kh; this->Kw = kw;
        this->Dout = dout; this->Hout = hout; this->Wout = wout;
        this->TotalOut = totalOutputLength;

        this->negSlope = negativeSlope;
        this->cMin = clampMin;
        this->cMax = clampMax;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        uint32_t base = (totalOutputLength / blocks) * idx;
        uint32_t len = (idx == (blocks - 1)) ? (totalOutputLength - base) : (totalOutputLength / blocks);
        this->blockStart = base;
        this->blockLen = len;
    }

    __aicore__ inline void Process()
    {
        // Iterate over each output element assigned to this block and compute fused conv + activations
        for (uint32_t linearIdx = 0; linearIdx < blockLen; ++linearIdx) {
            uint32_t outIndex = blockStart + linearIdx;

            // unravel outIndex -> (n, co, od, oh, ow)
            uint32_t ow = outIndex % Wout;
            uint32_t tmp1 = outIndex / Wout;
            uint32_t oh = tmp1 % Hout;
            uint32_t tmp2 = tmp1 / Hout;
            uint32_t od = tmp2 % Dout;
            uint32_t tmp3 = tmp2 / Dout;
            uint32_t co = tmp3 % Cout;
            uint32_t n = tmp3 / Cout;

            DTYPE acc = static_cast<DTYPE>(0.0f);

            // convolution sum
            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t kd = 0; kd < Kd; ++kd) {
                    uint32_t in_d = od + kd;
                    for (uint32_t kh = 0; kh < Kh; ++kh) {
                        uint32_t in_h = oh + kh;
                        for (uint32_t kw = 0; kw < Kw; ++kw) {
                            uint32_t in_w = ow + kw;

                            uint64_t xIdx = ((((uint64_t)n * Cin + ci) * Din + in_d) * Hin + in_h) * Win + in_w;
                            uint64_t wIdx = ((((uint64_t)co * Cin + ci) * Kd + kd) * Kh + kh) * Kw + kw;

                            acc += xGm[xIdx] * wGm[wIdx];
                        }
                    }
                }
            }

            // add bias
            acc += bGm[co];

            // LeakyReLU
            DTYPE x_act = (acc >= static_cast<DTYPE>(0.0f)) ? acc : (negSlope * acc);

            // add sum_tensor (broadcast over N, D, H, W)
            x_act += sGm[co];

            // clamp
            if (x_act < cMin) x_act = cMin;
            if (x_act > cMax) x_act = cMax;

            // GELU (approximate with tanh)
            // gelu(x) ~ 0.5 * x * (1 + tanh(0.79788456 * (x + 0.044715 * x^3)))
            DTYPE x3 = x_act * x_act * x_act;
            DTYPE inner = static_cast<DTYPE>(0.79788456f) * (x_act + static_cast<DTYPE>(0.044715f) * x3);
            DTYPE gelu = static_cast<DTYPE>(0.5f) * x_act * (static_cast<DTYPE>(1.0f) + tanh(inner));

            uint64_t yIdx = ((((uint64_t)n * Cout + co) * Dout + od) * Hout + oh) * Wout + ow;
            yGm[yIdx] = gelu;
        }
    }

private:
    __gm__ const DTYPE* xGm;
    __gm__ const DTYPE* wGm;
    __gm__ const DTYPE* bGm;
    __gm__ const DTYPE* sGm;
    __gm__ DTYPE* yGm;

    uint32_t N, Cin, Din, Hin, Win;
    uint32_t Cout, Kd, Kh, Kw;
    uint32_t Dout, Hout, Wout;
    uint32_t TotalOut;

    float negSlope;
    float cMin, cMax;

    uint32_t blockStart;
    uint32_t blockLen;
};

extern "C" __global__ __aicore__ void conv3d_leaky_relu_sum_clamp_gelu_custom(GM_ADDR x,
                                                                              GM_ADDR weight,
                                                                              GM_ADDR bias,
                                                                              GM_ADDR sum_tensor,
                                                                              GM_ADDR y,
                                                                              GM_ADDR workspace,
                                                                              GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dLeakyReluSumClampGelu op;
    op.Init(x, weight, bias, sum_tensor, y,
            tiling_data.n, tiling_data.cin,
            tiling_data.din, tiling_data.hin, tiling_data.win,
            tiling_data.cout,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.dout, tiling_data.hout, tiling_data.wout,
            tiling_data.totalOutputLength,
            tiling_data.negativeSlope,
            tiling_data.clampMin, tiling_data.clampMax);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused Conv3D -> LeakyReLU -> +sum_tensor -> clamp -> GELU using AscendC custom kernel.
    Assumes stride=1, padding=0, dilation=1; sum_tensor is shaped (Cout,1,1,1).
    """
    def __init__(self, in_channels, out_channels, kernel_size, sum_tensor_shape):
        super(ModelNew, self).__init__()
        # Keep PyTorch Conv3d only as parameter container; forward will call fused custom op
        self.conv = torch.nn.Conv3d(in_channels, out_channels, kernel_size)
        # sum_tensor is learnable parameter broadcast across N,D,H,W
        self.sum_tensor = torch.nn.Parameter(torch.randn(sum_tensor_shape))

    def forward(self, x):
        # Use fused operator with conv weights and bias
        return custom_ops_lib.conv3d_leaky_relu_sum_clamp_gelu_custom(
            x, self.conv.weight, self.conv.bias, self.sum_tensor
        )
