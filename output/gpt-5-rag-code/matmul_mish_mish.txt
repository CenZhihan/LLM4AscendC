kernel_src="""
#include "kernel_operator.h"

class KernelMatmulMishMish {
public:
    __aicore__ inline KernelMatmulMishMish() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN, uint32_t tileK)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileN = tileN;
        this->tileK = tileK;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M * (uint64_t)K);
        wGm.SetGlobalBuffer((__gm__ float*)w, (uint64_t)N * (uint64_t)K);
        bGm.SetGlobalBuffer((__gm__ float*)b, (uint64_t)N);
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M * (uint64_t)N);

        // UB buffers
        // store one row of x (length K)
        pipe.InitBuffer(xRowBuf, 1, K * sizeof(float));
        // store a tile of weight of shape [tileN, tileK]
        pipe.InitBuffer(wTileBuf, 1, tileN * tileK * sizeof(float));
        // store a tile of bias (length tileN)
        pipe.InitBuffer(bTileBuf, 1, tileN * sizeof(float));
    }

    __aicore__ inline float mish_scalar(float v) {
        // mish(x) = x * tanh(softplus(x)), softplus(x) = ln(1 + exp(x))
        // a basic implementation; in practice, vectorized intrinsics are preferred
        float e = AscendC::Exp(v);
        float sp = AscendC::Log(1.0f + e);
        float th = AscendC::Tanh(sp);
        return v * th;
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // split rows by blocks
        uint32_t rowsPerBlock = M / blockNum;
        uint32_t remain = M % blockNum;
        uint32_t startRow = rowsPerBlock * blockIdx + (blockIdx < remain ? blockIdx : remain);
        uint32_t myRows = rowsPerBlock + (blockIdx < remain ? 1 : 0);
        if (myRows == 0) {
            return;
        }

        for (uint32_t i = 0; i < myRows; ++i) {
            uint32_t row = startRow + i;
            // load x row
            AscendC::LocalTensor<float> xRowLocal = xRowBuf.AllocTensor<float>();
            AscendC::DataCopy(xRowLocal, xGm[row * K], K);
            xRowBuf.EnQue(xRowLocal);
            AscendC::LocalTensor<float> xRow = xRowBuf.DeQue<float>();

            for (uint32_t nStart = 0; nStart < N; nStart += tileN) {
                uint32_t curTileN = AscendC::Min(tileN, N - nStart);

                // load bias tile
                AscendC::LocalTensor<float> bLocal = bTileBuf.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm[nStart], curTileN);
                bTileBuf.EnQue(bLocal);
                AscendC::LocalTensor<float> biasTile = bTileBuf.DeQue<float>();

                // accumulators in registers for the current tileN
                // initialize to zero
                // we store partial sums in local tensor w.r.t tileN for reuse; use vector as scratch
                float accMax = 0.0f; // dummy to avoid empty var warnings
                (void)accMax;

                // We'll accumulate into an out buffer for curTileN
                // create local array in UB to store accumulation
                AscendC::LocalTensor<float> accTile;
                accTile.SetGlobalBuffer(nullptr, 0); // silent initialization
                // allocate via weight tile buffer if capacity suffices
                AscendC::LocalTensor<float> accLocal = wTileBuf.AllocTensor<float>(); // reuse buffer head
                // initialize accumulators to 0
                for (uint32_t t = 0; t < curTileN; ++t) {
                    accLocal.SetValue(t, 0.0f);
                }

                for (uint32_t kStart = 0; kStart < K; kStart += tileK) {
                    uint32_t curTileK = AscendC::Min(tileK, K - kStart);
                    // load weight tile [curTileN, curTileK] row by row
                    AscendC::LocalTensor<float> wTileLocal = wTileBuf.AllocTensor<float>();
                    // copy each row segment
                    for (uint32_t t = 0; t < curTileN; ++t) {
                        AscendC::DataCopy(
                            wTileLocal[t * curTileK],
                            wGm[(nStart + t) * K + kStart],
                            curTileK
                        );
                    }

                    // accumulation
                    for (uint32_t t = 0; t < curTileN; ++t) {
                        float s = accLocal.GetValue(t);
                        for (uint32_t kk = 0; kk < curTileK; ++kk) {
                            float xv = xRow.GetValue(kStart + kk);
                            float wv = wTileLocal.GetValue(t * curTileK + kk);
                            s += xv * wv;
                        }
                        accLocal.SetValue(t, s);
                    }
                }

                // add bias and apply Mish twice, then write to y
                for (uint32_t t = 0; t < curTileN; ++t) {
                    float v = accLocal.GetValue(t) + biasTile.GetValue(t);
                    v = mish_scalar(mish_scalar(v));
                    // store
                    yGm[row * N + (nStart + t)] = v;
                }

                // free local buffers
                wTileBuf.FreeTensor(accLocal);
                bTileBuf.FreeTensor(biasTile);
            }

            xRowBuf.FreeTensor(xRow);
        }
    }

private:
    AscendC::TPipe pipe;
    // Buffers
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xRowBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wTileBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bTileBuf;

    // GM tensors
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> yGm;

    // shapes and tiles
    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileN;
    uint32_t tileK;
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMishMish op;
    op.Init(x, weight, bias, y,
            tiling_data.M,
            tiling_data.N,
            tiling_data.K,
            tiling_data.tileN,
            tiling_data.tileK);
    op.Process();
}

#include <torch/library.h>

#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor matmul_mish_mish_custom_impl_npu(const at::Tensor& x,
                                            const at::Tensor& weight,
                                            const at::Tensor& bias) {
    TORCH_CHECK(x.dim() == 2, "x must be 2D [M, K]");
    TORCH_CHECK(weight.dim() == 2, "weight must be 2D [N, K]");
    TORCH_CHECK(bias.dim() == 1, "bias must be 1D [N]");
    TORCH_CHECK(x.size(1) == weight.size(1), "K mismatch between x and weight");
    TORCH_CHECK(weight.size(0) == bias.size(0), "N mismatch between weight and bias");

    auto M = x.size(0);
    auto N = weight.size(0);
    at::Tensor y = at::empty({M, N}, x.options());
    EXEC_NPU_CMD(aclnnMatmulMishMishCustom, x, weight, bias, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_mish_mish_custom", &matmul_mish_mish_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_mish_mish_custom", &matmul_mish_mish_custom_impl_npu, "fused matmul+bias+mish+mish");
}

import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Fused MatMul + BiasAdd + Mish + Mish using a custom AscendC kernel.
    weight is expected in shape [out_features, in_features] (same as nn.Linear.weight).
    """
    def __init__(self, in_features, out_features):
        super().__init__()
        self.linear = nn.Linear(in_features, out_features)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Compute y = mish(mish(x @ W^T + b)) with W provided as [N, K]
        # Our custom op expects (x[M,K], weight[N,K], bias[N]) and produces (M,N)
        return custom_ops_lib.matmul_mish_mish_custom(x, self.linear.weight, self.linear.bias)
