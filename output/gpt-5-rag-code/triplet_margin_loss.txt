static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    TripletMarginLossCustomTilingData tiling;

    const gert::Shape* anchor_shape = context->GetInputShape(0);
    // Expecting [batch, dim]
    uint64_t rank = anchor_shape->GetOriginShape().GetDimNum();
    uint64_t rows = rank >= 1 ? static_cast<uint64_t>(anchor_shape->GetOriginShape().GetDim(0)) : 1;
    uint64_t cols = 1;
    if (rank >= 2) {
        cols = static_cast<uint64_t>(anchor_shape->GetOriginShape().GetDim(1));
        for (uint64_t i = 2; i < rank; ++i) {
            cols *= static_cast<uint64_t>(anchor_shape->GetOriginShape().GetDim(i));
        }
    }

    float margin = 1.0f;
    auto attr_margin = context->GetAttr("margin");
    if (attr_margin != nullptr) {
        margin = attr_margin->GetFloat();
    }

    // Simple fixed tiling; can be tuned for chip L1 size.
    uint32_t tileRows = 128;
    uint32_t tileCols = 256;
    if (rows < tileRows) tileRows = static_cast<uint32_t>(rows);
    if (cols < tileCols) tileCols = static_cast<uint32_t>(cols);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_rows(static_cast<uint32_t>(rows));
    tiling.set_cols(static_cast<uint32_t>(cols));
    tiling.set_tileRows(tileRows);
    tiling.set_tileCols(tileCols);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLoss {
public:
    __aicore__ inline KernelTripletMarginLoss() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
                                uint32_t rows, uint32_t cols, uint32_t tileRows, uint32_t tileCols, float margin)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileRows = tileRows;
        this->tileCols = tileCols;
        this->margin = margin;

        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();

        uint32_t baseRowsPerBlock = rows / blockNum;
        uint32_t remainder = rows % blockNum;
        this->rowStart = baseRowsPerBlock * blockIdx + (blockIdx < remainder ? blockIdx : remainder);
        this->rowCount = baseRowsPerBlock + (blockIdx < remainder ? 1 : 0);

        aGm.SetGlobalBuffer((__gm__ float*)anchor + (uint64_t)rowStart * cols, (uint64_t)rowCount * cols);
        pGm.SetGlobalBuffer((__gm__ float*)positive + (uint64_t)rowStart * cols, (uint64_t)rowCount * cols);
        nGm.SetGlobalBuffer((__gm__ float*)negative + (uint64_t)rowStart * cols, (uint64_t)rowCount * cols);
        lGm.SetGlobalBuffer((__gm__ float*)loss + rowStart, rowCount);

        // UB buffers
        uint32_t maxTileElems = tileRows * tileCols;
        pipe.InitBuffer(inQueueA, BUFFER_NUM, maxTileElems * sizeof(float));
        pipe.InitBuffer(inQueueP, BUFFER_NUM, maxTileElems * sizeof(float));
        pipe.InitBuffer(inQueueN, BUFFER_NUM, maxTileElems * sizeof(float));
        pipe.InitBuffer(tmpQueue1, BUFFER_NUM, maxTileElems * sizeof(float));
        pipe.InitBuffer(tmpQueue2, BUFFER_NUM, maxTileElems * sizeof(float));
        pipe.InitBuffer(outQueueL, BUFFER_NUM, tileRows * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t rProcessed = 0;
        while (rProcessed < rowCount) {
            uint32_t curRows = AscendC::Min(tileRows, rowCount - rProcessed);

            // local accumulators for this row tile
            AscendC::LocalTensor<float> posAccLocal = outQueueL.AllocTensor<float>();
            AscendC::LocalTensor<float> negAccLocal = tmpQueue1.AllocTensor<float>(); // reuse buffer size tileRows

            // initialize to zeros
            AscendC::Duplicate<float>(posAccLocal, 0.0f, curRows);
            AscendC::Duplicate<float>(negAccLocal, 0.0f, curRows);

            uint32_t cProcessed = 0;
            while (cProcessed < cols) {
                uint32_t curCols = AscendC::Min(tileCols, cols - cProcessed);
                uint32_t curElems = curRows * curCols;

                // CopyIn
                AscendC::LocalTensor<float> aLocal = inQueueA.AllocTensor<float>();
                AscendC::LocalTensor<float> pLocal = inQueueP.AllocTensor<float>();
                AscendC::LocalTensor<float> nLocal = inQueueN.AllocTensor<float>();

                uint64_t gmOffset = ((uint64_t)rProcessed * cols + cProcessed);
                AscendC::DataCopy(aLocal, aGm[gmOffset], curElems);
                AscendC::DataCopy(pLocal, pGm[gmOffset], curElems);
                AscendC::DataCopy(nLocal, nGm[gmOffset], curElems);

                inQueueA.EnQue(aLocal);
                inQueueP.EnQue(pLocal);
                inQueueN.EnQue(nLocal);

                // Compute partial sums
                aLocal = inQueueA.DeQue<float>();
                pLocal = inQueueP.DeQue<float>();
                nLocal = inQueueN.DeQue<float>();

                AscendC::LocalTensor<float> diffAP = tmpQueue1.AllocTensor<float>();
                AscendC::LocalTensor<float> diffAN = tmpQueue2.AllocTensor<float>();

                AscendC::Sub(diffAP, aLocal, pLocal, curElems);
                AscendC::Sub(diffAN, aLocal, nLocal, curElems);

                AscendC::Mul(diffAP, diffAP, diffAP, curElems); // square
                AscendC::Mul(diffAN, diffAN, diffAN, curElems); // square

                // Row-wise reduction: sum over curCols into curRows accumulators
                // Here we do a simple loop accumulating per row tile; illustration purpose.
                for (uint32_t r = 0; r < curRows; ++r) {
                    uint32_t base = r * curCols;
                    float sumPos = 0.0f;
                    float sumNeg = 0.0f;
                    // unrolled accumulation could be used; keep simple
                    for (uint32_t c = 0; c < curCols; ++c) {
                        sumPos += diffAP.GetValue(base + c);
                        sumNeg += diffAN.GetValue(base + c);
                    }
                    float oldPos = posAccLocal.GetValue(r);
                    float oldNeg = negAccLocal.GetValue(r);
                    posAccLocal.SetValue(r, oldPos + sumPos);
                    negAccLocal.SetValue(r, oldNeg + sumNeg);
                }

                tmpQueue2.FreeTensor(diffAN);
                tmpQueue1.FreeTensor(diffAP);

                inQueueA.FreeTensor(aLocal);
                inQueueP.FreeTensor(pLocal);
                inQueueN.FreeTensor(nLocal);

                cProcessed += curCols;
            }

            // After finishing cols for this row tile, compute sqrt and final loss, then write out
            AscendC::LocalTensor<float> lossLocal = outQueueL.AllocTensor<float>();
            for (uint32_t r = 0; r < curRows; ++r) {
                float dp = posAccLocal.GetValue(r);
                float dn = negAccLocal.GetValue(r);
                float distPos = AscendC::Sqrt(dp);
                float distNeg = AscendC::Sqrt(dn);
                float val = margin + distPos - distNeg;
                if (val < 0.0f) val = 0.0f;
                lossLocal.SetValue(r, val);
            }

            // CopyOut
            AscendC::DataCopy(lGm[rProcessed], lossLocal, curRows);

            outQueueL.FreeTensor(lossLocal);
            outQueueL.FreeTensor(posAccLocal);
            tmpQueue1.FreeTensor(negAccLocal);

            rProcessed += curRows;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueP, inQueueN;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue1, tmpQueue2;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueL;

    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> pGm;
    AscendC::GlobalTensor<float> nGm;
    AscendC::GlobalTensor<float> lGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tileRows;
    uint32_t tileCols;
    float margin;

    uint32_t blockNum;
    uint32_t blockIdx;
    uint32_t rowStart;
    uint32_t rowCount;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLoss op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.rows, tiling_data.cols, tiling_data.tileRows, tiling_data.tileCols, tiling_data.margin);
    op.Process();
}
