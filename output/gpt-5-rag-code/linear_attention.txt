#include "linear_attention_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t ceil_div(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // Shapes:
    // q: [B, Q, D]
    // k: [B, K, D]
    // v: [B, K, V]
    const gert::Shape* q_shape = context->GetInputShape(0);
    const gert::Shape* k_shape = context->GetInputShape(1);
    const gert::Shape* v_shape = context->GetInputShape(2);

    if (q_shape->GetDimNum() != 3 || k_shape->GetDimNum() != 3 || v_shape->GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }

    uint32_t B = static_cast<uint32_t>(q_shape->GetDim(0));
    uint32_t Q = static_cast<uint32_t>(q_shape->GetDim(1));
    uint32_t D = static_cast<uint32_t>(q_shape->GetDim(2));
    uint32_t K = static_cast<uint32_t>(k_shape->GetDim(1));
    uint32_t V = static_cast<uint32_t>(v_shape->GetDim(2));

    // simple default tiles
    const uint32_t dTile = 64;
    const uint32_t vTile = 64;
    const uint32_t kTile = 64;

    LinearAttentionCustomTilingData tiling;
    tiling.set_batch(B);
    tiling.set_qLen(Q);
    tiling.set_kLen(K);
    tiling.set_dModel(D);
    tiling.set_vDim(V);
    tiling.set_dTile(dTile);
    tiling.set_vTile(vTile);
    tiling.set_kTile(kTile);
    tiling.set_batchPerBlock(ceil_div(B, BLOCK_DIM));

    context->SetBlockDim(BLOCK_DIM);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* q_shape = context->GetInputShape(0);
    const gert::Shape* v_shape = context->GetInputShape(2);
    gert::Shape* out_shape = context->GetOutputShape(0);

    // Output: [B, Q, V]
    std::vector<int64_t> out_dims(3);
    out_dims[0] = q_shape->GetDim(0);
    out_dims[1] = q_shape->GetDim(1);
    out_dims[2] = v_shape->GetDim(2);
    out_shape->SetDimNum(3);
    for (size_t i = 0; i < 3; ++i) {
        out_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr float EPS = 1e-6f;

class KernelLinearAttention {
public:
    __aicore__ inline KernelLinearAttention() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR out,
                                uint32_t B, uint32_t Q, uint32_t K, uint32_t D, uint32_t V,
                                uint32_t dTile, uint32_t vTile, uint32_t kTile, uint32_t batchPerBlock)
    {
        this->B = B;
        this->Q = Q;
        this->K = K;
        this->D = D;
        this->V = V;
        this->dTile = dTile;
        this->vTile = vTile;
        this->kTile = kTile;
        this->batchPerBlock = batchPerBlock;

        qGm.SetGlobalBuffer((__gm__ DTYPE_X *)q, (uint32_t)(B * Q * D));
        kGm.SetGlobalBuffer((__gm__ DTYPE_Y *)k, (uint32_t)(B * K * D));
        vGm.SetGlobalBuffer((__gm__ DTYPE_W *)v, (uint32_t)(B * K * V));
        oGm.SetGlobalBuffer((__gm__ DTYPE_Z *)out, (uint32_t)(B * Q * V));

        // allocate local buffers
        pipe.InitBuffer(qBuf, 1, dTile * sizeof(DTYPE_X));
        pipe.InitBuffer(kBuf, 1, dTile * sizeof(DTYPE_Y));
        pipe.InitBuffer(vBuf, 1, vTile * sizeof(DTYPE_W));
        pipe.InitBuffer(oBuf, 1, vTile * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint32_t bStart = AscendC::GetBlockIdx() * batchPerBlock;
        uint32_t bEnd   = bStart + batchPerBlock;
        if (bEnd > B) bEnd = B;

        for (uint32_t b = bStart; b < bEnd; ++b) {
            uint64_t qBatchBase = ((uint64_t)b) * Q * D;
            uint64_t kBatchBase = ((uint64_t)b) * K * D;
            uint64_t vBatchBase = ((uint64_t)b) * K * V;
            uint64_t oBatchBase = ((uint64_t)b) * Q * V;

            for (uint32_t qIdx = 0; qIdx < Q; ++qIdx) {
                uint64_t qVecBase = qBatchBase + ((uint64_t)qIdx) * D;

                for (uint32_t v0 = 0; v0 < V; v0 += vTile) {
                    uint32_t curVTile = (v0 + vTile <= V) ? vTile : (V - v0);

                    AscendC::LocalTensor<DTYPE_Z> outLocal = oBuf.AllocTensor<DTYPE_Z>();
                    // zero init outLocal
                    for (uint32_t i = 0; i < curVTile; ++i) {
                        outLocal.SetValue(i, (DTYPE_Z)0);
                    }

                    // iterate over K (sequence) dimension
                    for (uint32_t kIdx = 0; kIdx < K; ++kIdx) {
                        // compute s = dot_d( relu(Q[q,:]) * relu(K[k,:]) ) with eps
                        float s = 0.0f;
                        uint64_t kVecBase = kBatchBase + ((uint64_t)kIdx) * D;

                        for (uint32_t d0 = 0; d0 < D; d0 += dTile) {
                            uint32_t curDTile = (d0 + dTile <= D) ? dTile : (D - d0);
                            AscendC::LocalTensor<DTYPE_X> qLocal = qBuf.AllocTensor<DTYPE_X>();
                            AscendC::LocalTensor<DTYPE_Y> kLocal = kBuf.AllocTensor<DTYPE_Y>();

                            AscendC::DataCopy(qLocal, qGm[qVecBase + d0], curDTile);
                            AscendC::DataCopy(kLocal, kGm[kVecBase + d0], curDTile);

                            for (uint32_t i = 0; i < curDTile; ++i) {
                                float qv = (float)qLocal.GetValue(i);
                                float kv = (float)kLocal.GetValue(i);
                                float qp = (qv > 0.0f ? qv : 0.0f) + EPS;
                                float kp = (kv > 0.0f ? kv : 0.0f) + EPS;
                                s += qp * kp;
                            }

                            qBuf.FreeTensor(qLocal);
                            kBuf.FreeTensor(kLocal);
                        }

                        // load V tile for this (b, kIdx, v0:curVTile)
                        AscendC::LocalTensor<DTYPE_W> vLocal = vBuf.AllocTensor<DTYPE_W>();
                        uint64_t vTileBase = vBatchBase + ((uint64_t)kIdx) * V + v0;
                        AscendC::DataCopy(vLocal, vGm[vTileBase], curVTile);

                        // outLocal += s * vLocal
                        for (uint32_t i = 0; i < curVTile; ++i) {
                            float prev = (float)outLocal.GetValue(i);
                            float vv = (float)vLocal.GetValue(i);
                            float upd = prev + s * vv;
                            outLocal.SetValue(i, (DTYPE_Z)upd);
                        }
                        vBuf.FreeTensor(vLocal);
                    }

                    // write out tile
                    uint64_t oTileBase = oBatchBase + ((uint64_t)qIdx) * V + v0;
                    AscendC::DataCopy(oGm[oTileBase], outLocal, curVTile);
                    oBuf.FreeTensor(outLocal);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> kBuf;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> vBuf;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> oBuf;

    AscendC::GlobalTensor<DTYPE_X> qGm;
    AscendC::GlobalTensor<DTYPE_Y> kGm;
    AscendC::GlobalTensor<DTYPE_W> vGm;
    AscendC::GlobalTensor<DTYPE_Z> oGm;

    uint32_t B;
    uint32_t Q;
    uint32_t K;
    uint32_t D;
    uint32_t V;
    uint32_t dTile;
    uint32_t vTile;
    uint32_t kTile;
    uint32_t batchPerBlock;
};

extern "C" __global__ __aicore__ void linear_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLinearAttention op;
    op.Init(q, k, v, out,
            tiling_data.batch, tiling_data.qLen, tiling_data.kLen,
            tiling_data.dModel, tiling_data.vDim,
            tiling_data.dTile, tiling_data.vTile, tiling_data.kTile,
            tiling_data.batchPerBlock);
    op.Process();
}
"""
