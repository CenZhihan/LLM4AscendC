static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ScaledDotProductAttentionLongContextCustomTilingData tiling;

    const gert::Shape* q_shape = context->GetInputShape(0);
    // Expect Q, K, V shapes: [B, S, D]
    uint32_t dims = q_shape->GetOriginShape().GetDimNum();
    uint32_t B = 1, S = 1, D = 1;
    if (dims == 3) {
        B = static_cast<uint32_t>(q_shape->GetOriginShape().GetDim(0));
        S = static_cast<uint32_t>(q_shape->GetOriginShape().GetDim(1));
        D = static_cast<uint32_t>(q_shape->GetOriginShape().GetDim(2));
    } else if (dims == 2) {
        // Fallback: [S, D], assume B=1
        B = 1;
        S = static_cast<uint32_t>(q_shape->GetOriginShape().GetDim(0));
        D = static_cast<uint32_t>(q_shape->GetOriginShape().GetDim(1));
    }

    // Simple fixed tiling suitable for long context; can be tuned
    const uint32_t TILE_Q = 8;
    const uint32_t TILE_K = 256;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_batchSize(B);
    tiling.set_seqLen(S);
    tiling.set_headDim(D);
    tiling.set_tileQLen(TILE_Q);
    tiling.set_tileKLen(TILE_K);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

class KernelSDPALongCtx {
public:
    __aicore__ inline KernelSDPALongCtx() {}

    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR o,
                                uint32_t batchSize, uint32_t seqLen, uint32_t headDim,
                                uint32_t tileQLen, uint32_t tileKLen)
    {
        this->B = batchSize;
        this->S = seqLen;
        this->D = headDim;
        this->TQ = tileQLen;
        this->TK = tileKLen;

        // partition queries across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        uint32_t qPerBlock = (S + blocks - 1) / blocks; // ceil
        qStart = idx * qPerBlock;
        qEnd = qStart + qPerBlock;
        if (qEnd > S) qEnd = S;
        qLenBlock = (qEnd > qStart) ? (qEnd - qStart) : 0;

        qGm.SetGlobalBuffer((__gm__ float*)q + ((size_t)0), (size_t)B * S * D);
        kGm.SetGlobalBuffer((__gm__ float*)k + ((size_t)0), (size_t)B * S * D);
        vGm.SetGlobalBuffer((__gm__ float*)v + ((size_t)0), (size_t)B * S * D);
        oGm.SetGlobalBuffer((__gm__ float*)o + ((size_t)0), (size_t)B * S * D);

        // Buffers
        pipe.InitBuffer(qInQue, 1, (size_t)TQ * D * sizeof(float));
        pipe.InitBuffer(kInQue, 1, (size_t)TK * D * sizeof(float));
        pipe.InitBuffer(vInQue, 1, (size_t)TK * D * sizeof(float));
        pipe.InitBuffer(oOutQue, 1, (size_t)TQ * D * sizeof(float));

        // Aux buffers for online softmax per Q row
        rowMaxBuf.SetSize(TQ * sizeof(float));
        rowSumBuf.SetSize(TQ * sizeof(float));
        // Accumulator buffer for output tile
        oAccBuf.SetSize((size_t)TQ * D * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (qLenBlock == 0) return;
        // For each batch independently
        for (uint32_t b = 0; b < B; ++b) {
            // Iterate over query tiles in this block
            for (uint32_t qBase = qStart; qBase < qEnd; qBase += TQ) {
                uint32_t curTQ = AscendC::min((uint32_t)TQ, qEnd - qBase);
                // Load Q tile
                CopyQTile(b, qBase, curTQ);

                // Initialize online softmax stats and accumulators
                InitOnlineState(curTQ);

                // Iterate over K/V tiles across entire sequence
                for (uint32_t kBase = 0; kBase < S; kBase += TK) {
                    uint32_t curTK = AscendC::min((uint32_t)TK, S - kBase);
                    CopyKTile(b, kBase, curTK);
                    CopyVTile(b, kBase, curTK);
                    // Compute partial scores and update online softmax accumulators
                    ComputeKVContribution(curTQ, curTK);
                }

                // Normalize accumulated output by row sums and write back
                FinalizeAndStore(b, qBase, curTQ);
            }
        }
    }

private:
    __aicore__ inline void CopyQTile(uint32_t b, uint32_t qBase, uint32_t curTQ)
    {
        AscendC::LocalTensor<float> qLocal = qInQue.AllocTensor<float>();
        // Q layout: [B, S, D], contiguous in last dim
        size_t offset = ((size_t)b * S + qBase) * D;
        AscendC::DataCopy(qLocal, qGm[offset], (size_t)curTQ * D);
        qInQue.EnQue(qLocal);
    }

    __aicore__ inline void CopyKTile(uint32_t b, uint32_t kBase, uint32_t curTK)
    {
        AscendC::LocalTensor<float> kLocal = kInQue.AllocTensor<float>();
        size_t offset = ((size_t)b * S + kBase) * D;
        AscendC::DataCopy(kLocal, kGm[offset], (size_t)curTK * D);
        kInQue.EnQue(kLocal);
    }

    __aicore__ inline void CopyVTile(uint32_t b, uint32_t kBase, uint32_t curTK)
    {
        AscendC::LocalTensor<float> vLocal = vInQue.AllocTensor<float>();
        size_t offset = ((size_t)b * S + kBase) * D;
        AscendC::DataCopy(vLocal, vGm[offset], (size_t)curTK * D);
        vInQue.EnQue(vLocal);
    }

    __aicore__ inline void InitOnlineState(uint32_t curTQ)
    {
        // Initialize row max to very small number and row sum to 0, accumulators to 0
        float* rowMax = (float*)rowMaxBuf.Get<AscendC::TPosition::VECCALC>();
        float* rowSum = (float*)rowSumBuf.Get<AscendC::TPosition::VECCALC>();
        float* oAcc = (float*)oAccBuf.Get<AscendC::TPosition::VECCALC>();
        for (uint32_t i = 0; i < curTQ; ++i) {
            rowMax[i] = -1e30f;
            rowSum[i] = 0.0f;
        }
        for (uint32_t i = 0; i < (size_t)curTQ * D; ++i) {
            oAcc[i] = 0.0f;
        }
    }

    __aicore__ inline void ComputeKVContribution(uint32_t curTQ, uint32_t curTK)
    {
        AscendC::LocalTensor<float> qLocal = qInQue.DeQue<float>();
        AscendC::LocalTensor<float> kLocal = kInQue.DeQue<float>();
        AscendC::LocalTensor<float> vLocal = vInQue.DeQue<float>();

        float* rowMax = (float*)rowMaxBuf.Get<AscendC::TPosition::VECCALC>();
        float* rowSum = (float*)rowSumBuf.Get<AscendC::TPosition::VECCALC>();
        float* oAcc   = (float*)oAccBuf.Get<AscendC::TPosition::VECCALC>();

        const float scale = 1.0f / sqrtf((float)D);

        // For each query row in the tile
        for (uint32_t qi = 0; qi < curTQ; ++qi) {
            float* qRow = qLocal.Get<float>() + (size_t)qi * D;

            // Compute logits for this q against current K tile, track max
            // We'll reuse a small temporary buffer in VECOUT for logits of size curTK
            // If not available, compute on the fly.
            float localMax = rowMax[qi];

            // We need to store exp(logits - newMax) to update accumulators; store in scratch if needed
            // Here we'll compute in two passes: first pass to find new max, second to accumulate
            // Pass 1: find max
            for (uint32_t kk = 0; kk < curTK; ++kk) {
                float* kRow = kLocal.Get<float>() + (size_t)kk * D;
                float dot = 0.0f;
                // dot product qRow·kRow
                for (uint32_t d = 0; d < D; ++d) {
                    dot += qRow[d] * kRow[d];
                }
                float logit = dot * scale;
                if (logit > localMax) localMax = logit;
            }

            // Compute scale factor for previous accumulators when max increases
            float prevMax = rowMax[qi];
            float prevSum = rowSum[qi];
            float maxScale = (prevMax > -1e20f) ? expf(prevMax - localMax) : 0.0f;

            // Scale existing accumulators by exp(prevMax - localMax)
            if (prevSum > 0.0f) {
                float* outRowAcc = oAcc + (size_t)qi * D;
                for (uint32_t d = 0; d < D; ++d) {
                    outRowAcc[d] *= maxScale;
                }
                prevSum *= maxScale;
            }

            // Pass 2: accumulate contributions from this KV tile
            float newSum = prevSum;
            for (uint32_t kk = 0; kk < curTK; ++kk) {
                float* kRow = kLocal.Get<float>() + (size_t)kk * D;
                float dot = 0.0f;
                for (uint32_t d = 0; d < D; ++d) {
                    dot += qRow[d] * kRow[d];
                }
                float logit = dot * scale;
                float w = expf(logit - localMax);
                newSum += w;

                // outAcc += w * V[kk]
                float* vRow = vLocal.Get<float>() + (size_t)kk * D;
                float* outRowAcc = oAcc + (size_t)qi * D;
                for (uint32_t d = 0; d < D; ++d) {
                    outRowAcc[d] += w * vRow[d];
                }
            }

            // Update running stats
            rowMax[qi] = localMax;
            rowSum[qi] = newSum;
        }

        vInQue.FreeTensor(vLocal);
        kInQue.FreeTensor(kLocal);
        qInQue.EnQue(qLocal); // keep Q tile for next KV tiles
    }

    __aicore__ inline void FinalizeAndStore(uint32_t b, uint32_t qBase, uint32_t curTQ)
    {
        AscendC::LocalTensor<float> qLocal = qInQue.DeQue<float>();
        float* rowSum = (float*)rowSumBuf.Get<AscendC::TPosition::VECCALC>();
        float* oAcc   = (float*)oAccBuf.Get<AscendC::TPosition::VECCALC>();

        AscendC::LocalTensor<float> oLocal = oOutQue.AllocTensor<float>();
        float* oPtr = oLocal.Get<float>();

        // Normalize and store
        for (uint32_t qi = 0; qi < curTQ; ++qi) {
            float invSum = 1.0f / (rowSum[qi] + 1e-9f);
            float* srcRow = oAcc + (size_t)qi * D;
            float* dstRow = oPtr + (size_t)qi * D;
            for (uint32_t d = 0; d < D; ++d) {
                dstRow[d] = srcRow[d] * invSum;
            }
        }

        oOutQue.EnQue(oLocal);
        qInQue.FreeTensor(qLocal);

        // Write back to GM
        AscendC::LocalTensor<float> oLocalW = oOutQue.DeQue<float>();
        size_t offset = ((size_t)b * S + qBase) * D;
        AscendC::DataCopy(oGm[offset], oLocalW, (size_t)curTQ * D);
        oOutQue.FreeTensor(oLocalW);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> qInQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> kInQue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> vInQue;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> oOutQue;

    AscendC::TBuf<AscendC::TPosition::VECCALC> rowMaxBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> rowSumBuf;
    AscendC::TBuf<AscendC::TPosition::VECCALC> oAccBuf;

    AscendC::GlobalTensor<float> qGm;
    AscendC::GlobalTensor<float> kGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> oGm;

    uint32_t B, S, D;
    uint32_t TQ, TK;
    uint32_t qStart, qEnd, qLenBlock;
};

extern "C" __global__ __aicore__ void scaled_dot_product_attention_long_context_custom(GM_ADDR Q, GM_ADDR K, GM_ADDR V, GM_ADDR O, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSDPALongCtx op;
    op.Init(Q, K, V, O,
            tiling_data.batchSize, tiling_data.seqLen, tiling_data.headDim,
            tiling_data.tileQLen, tiling_data.tileKLen);
    op.Process();
}
