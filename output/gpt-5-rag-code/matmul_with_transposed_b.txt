    const auto* a_shape = context->GetInputShape(0);
    const auto* bt_shape = context->GetInputShape(1);

    uint32_t k = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));

    uint32_t n = static_cast<uint32_t>(bt_shape->GetOriginShape().GetDim(1));

        return ge::GRAPH_FAILED;

    uint32_t tileN = DEFAULT_TILE_N;

        tileN = n;

    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_tileN(tileN);

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

// This kernel computes C = A @ B_t
// A shape: (M, K) in row-major
// B_t shape: (K, N) in row-major (we pass B^T from host/python)
// C shape: (M, N) in row-major
// Simple streaming implementation along N dimension with inner K accumulation.
// Note: This version uses a straightforward approach focusing on correctness;
// performance can be improved via Cube/MM APIs and advanced tiling.

constexpr int32_t BUFFER_NUM = 1;

using DTYPE_A = float;
using DTYPE_B = float;
using DTYPE_C = float;

struct TilingData {
    uint32_t m;
    uint32_t n;
    uint32_t k;
    uint32_t tileN;
};

class KernelMatmulTransposedB {
public:
    __aicore__ inline KernelMatmulTransposedB() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR bt, GM_ADDR c,
                                uint32_t m, uint32_t n, uint32_t k, uint32_t tileN)
    {
        this->M = m;
        this->N = n;
        this->K = k;
        this->tileN = tileN;

        // Partition rows of A/C across blocks
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = (M + blockNum - 1) / blockNum;
        uint32_t rowStart = rowsPerBlock * blockIdx;
        if (rowStart > M) rowStart = M;
        uint32_t rowEnd = rowsPerBlock * (blockIdx + 1);
        if (rowEnd > M) rowEnd = M;
        this->localRows = rowEnd - rowStart;

        aGm.SetGlobalBuffer((__gm__ DTYPE_A*)a + static_cast<uint64_t>(rowStart) * K, static_cast<uint64_t>(localRows) * K);
        btGm.SetGlobalBuffer((__gm__ DTYPE_B*)bt, static_cast<uint64_t>(K) * N);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C*)c + static_cast<uint64_t>(rowStart) * N, static_cast<uint64_t>(localRows) * N);

        // Allocate UB buffers
        // bVec: one tile along N
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileN * sizeof(DTYPE_B));
        // accVec: accumulator for C row over a tile-N segment
        pipe.InitBuffer(accQueue, BUFFER_NUM, this->tileN * sizeof(DTYPE_C));
        // tmpVec: temporary for multiply
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileN * sizeof(DTYPE_C));
        // aFillVec: holds broadcasted A[i,k] across tileN
        pipe.InitBuffer(aVecQueue, BUFFER_NUM, this->tileN * sizeof(DTYPE_A));
    }

    __aicore__ inline void Process()
    {
        // Iterate rows assigned to this block
        for (uint32_t i = 0; i < localRows; ++i) {
            // For each row, sweep N by tiles
            for (uint32_t j = 0; j < N; j += tileN) {
                uint32_t curTileN = AscendC::u32min(tileN, N - j);

                // Allocate/zero accumulator
                AscendC::LocalTensor<DTYPE_C> acc = accQueue.AllocTensor<DTYPE_C>();
                AscendC::Dup(acc, static_cast<DTYPE_C>(0.0f), curTileN);
                accQueue.EnQue(acc);
                acc = accQueue.DeQue<DTYPE_C>();

                // Accumulate over K
                for (uint32_t kIdx = 0; kIdx < K; ++kIdx) {
                    // Load B_t row segment: B_t[kIdx, j : j+curTileN]
                    AscendC::LocalTensor<DTYPE_B> bVec = inQueueB.AllocTensor<DTYPE_B>();
                    AscendC::DataCopy(bVec, btGm[static_cast<uint64_t>(kIdx) * N + j], curTileN);
                    inQueueB.EnQue(bVec);
                    bVec = inQueueB.DeQue<DTYPE_B>();

                    // Prepare a broadcast vector filled with A[i, kIdx]
                    AscendC::LocalTensor<DTYPE_A> aFill = aVecQueue.AllocTensor<DTYPE_A>();

                    // Load scalar A[i, kIdx] via a 1-element copy using aFill then duplicate across curTileN
                    // Load into first element
                    AscendC::DataCopy(aFill, aGm[static_cast<uint64_t>(i) * K + kIdx], 1);
                    // Broadcast first element across curTileN positions
                    AscendC::BroadCast(aFill, aFill, 0, curTileN); // broadcast from index 0 across curTileN
                    aVecQueue.EnQue(aFill);
                    aFill = aVecQueue.DeQue<DTYPE_A>();

                    // tmp = bVec * aFill
                    AscendC::LocalTensor<DTYPE_C> tmp = tmpQueue.AllocTensor<DTYPE_C>();
                    AscendC::Mul(tmp, bVec, aFill, curTileN);
                    tmpQueue.EnQue(tmp);
                    tmp = tmpQueue.DeQue<DTYPE_C>();

                    // acc += tmp
                    AscendC::Add(acc, acc, tmp, curTileN);

                    // Free temporaries
                    tmpQueue.FreeTensor(tmp);
                    aVecQueue.FreeTensor(aFill);
                    inQueueB.FreeTensor(bVec);
                }

                // Write back C[i, j : j+curTileN]
                AscendC::DataCopy(cGm[static_cast<uint64_t>(i) * N + j], acc, curTileN);
                accQueue.FreeTensor(acc);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> accQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> aVecQueue;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> btGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t M, N, K;
    uint32_t tileN;
    uint32_t rowsPerBlock;
    uint32_t localRows;
};

extern "C" __global__ __aicore__ void matmul_with_transposed_b_custom(GM_ADDR a, GM_ADDR b_t, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling, tilingGm);
    KernelMatmulTransposedB op;
    op.Init(a, b_t, c, tiling.m, tiling.n, tiling.k, tiling.tileN);
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor matmul_with_transposed_b_custom_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Original model computes A @ B.T where B shape is (N, K).
    // We pass B_t (K, N) to the custom kernel for contiguous access along N.
    at::Tensor B_t = B.transpose(-2, -1).contiguous();
    TORCH_CHECK(A.dim() == 2 && B_t.dim() == 2, "Inputs must be 2D");
    TORCH_CHECK(A.size(1) == B_t.size(0), "Incompatible shapes for matmul");

    auto M = A.size(0);
    auto N = B_t.size(1);
    at::Tensor result = at::empty({M, N}, A.options());

    EXEC_NPU_CMD(aclnnMatmulWithTransposedBCustom, A, B_t, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_with_transposed_b_custom", &matmul_with_transposed_b_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_with_transposed_b_custom", &matmul_with_transposed_b_custom_impl_npu, "C = A @ B.T via custom kernel");
}
