kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 1;

class KernelVanillaRnnHidden {
public:
    __aicore__ inline KernelVanillaRnnHidden() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h_prev, GM_ADDR w, GM_ADDR b, GM_ADDR h_new,
                                uint32_t totalLength, uint32_t tileNum,
                                uint32_t batchSize, uint32_t inputSize, uint32_t hiddenSize)
    {
        this->batchSize = batchSize;
        this->inputSize = inputSize;
        this->hiddenSize = hiddenSize;

        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum;

        xGm  = (__gm__ float*)x  + (AscendC::GetBlockIdx() * (this->blockLength / this->hiddenSize)) * this->inputSize;
        hGm  = (__gm__ float*)h_prev + (AscendC::GetBlockIdx() * (this->blockLength / this->hiddenSize)) * this->hiddenSize;
        wGm  = (__gm__ float*)w;
        bGm  = (__gm__ float*)b;
        yGm  = (__gm__ float*)h_new + AscendC::GetBlockIdx() * this->blockLength;
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum;
        for (int32_t i = 0; i < loopCount; ++i) {
            Compute(i);
        }
    }

private:
    __aicore__ inline void Compute(int32_t progress)
    {
        uint32_t start = progress * this->tileLength;
        uint32_t end   = (progress == this->tileNum - 1) ? this->blockLength : (start + this->tileLength);

        for (uint32_t idx = start; idx < end; ++idx) {
            uint32_t bIdx = idx / this->hiddenSize;
            uint32_t hIdx = idx % this->hiddenSize;

            float acc = bGm[hIdx];

            // x contribution: [B, I] * [I, H]
            const __gm__ float* xRow = xGm + bIdx * this->inputSize;
            const __gm__ float* wCol0 = wGm + hIdx; // column hIdx, step hiddenSize
            for (uint32_t j = 0; j < this->inputSize; ++j) {
                float xv = xRow[j];
                float wv = wCol0[j * this->hiddenSize];
                acc += xv * wv;
            }

            // h_prev contribution: [B, H] * [H, H]
            const __gm__ float* hRow = hGm + bIdx * this->hiddenSize;
            const __gm__ float* wCol1 = wGm + (this->inputSize * this->hiddenSize) + hIdx;
            for (uint32_t j = 0; j < this->hiddenSize; ++j) {
                float hv = hRow[j];
                float wv = wCol1[j * this->hiddenSize];
                acc += hv * wv;
            }

            // tanh activation
            float y = tanhf(acc);
            yGm[idx] = y;
        }
    }

private:
    __gm__ float* xGm;
    __gm__ float* hGm;
    __gm__ float* wGm;
    __gm__ float* bGm;
    __gm__ float* yGm;

    uint32_t batchSize;
    uint32_t inputSize;
    uint32_t hiddenSize;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h_prev, GM_ADDR w, GM_ADDR b, GM_ADDR h_new, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHidden op;
    op.Init(x, h_prev, w, b, h_new,
            tiling_data.totalLength(), tiling_data.tileNum(),
            tiling_data.batchSize(), tiling_data.inputSize(), tiling_data.hiddenSize());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size: int, hidden_size: int, output_size: int):
        super(ModelNew, self).__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.output_size = output_size

        # Keep same modules to hold parameters for output projection
        self.i2h = torch.nn.Linear(input_size + hidden_size, hidden_size)
        self.h2o = torch.nn.Linear(hidden_size, output_size)

    def forward(self, x: torch.Tensor, h0: torch.Tensor) -> torch.Tensor:
        seq_len, batch_size, _ = x.size()
        hidden = h0.to(x.device)
        outputs = []

        # Prepare transposed weight for custom kernel once
        # PyTorch Linear: weight shape [H, I+H] -> transpose to [I+H, H]
        w_t = self.i2h.weight.t().contiguous().to(x.device)
        b = self.i2h.bias.to(x.device)

        for t in range(seq_len):
            # custom op computes: tanh([x_t, h] @ W + b)
            hidden = custom_ops_lib.vanilla_rnn_hidden_custom(x[t], hidden, w_t, b)
            outputs.append(self.h2o(hidden))

        return torch.stack(outputs, dim=0)
'''
