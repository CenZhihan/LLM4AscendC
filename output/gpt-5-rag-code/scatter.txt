kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_UPD = float;
using DTYPE_Y = float;
using DTYPE_IDX = int64_t;

constexpr int32_t BUFFER_NUM = 2;

class KernelScatter {
public:
    __aicore__ inline KernelScatter() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR index, GM_ADDR updates, GM_ADDR y,
                                uint32_t rows, uint32_t xCols, uint32_t idxCols, uint32_t tileSize) {
        this->rows = rows;
        this->xCols = xCols;
        this->idxCols = idxCols;
        this->tileSize = tileSize;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        this->rowStart = (rows * blockIdx) / blockNum;
        this->rowEnd   = (rows * (blockIdx + 1)) / blockNum;
        this->localRows = (this->rowEnd > this->rowStart) ? (this->rowEnd - this->rowStart) : 0;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)this->rowStart * xCols, (uint64_t)this->localRows * xCols);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_IDX*)index + (uint64_t)this->rowStart * idxCols, (uint64_t)this->localRows * idxCols);
        updGm.SetGlobalBuffer((__gm__ DTYPE_UPD*)updates + (uint64_t)this->rowStart * idxCols, (uint64_t)this->localRows * idxCols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + (uint64_t)this->rowStart * xCols, (uint64_t)this->localRows * xCols);

        pipe.InitBuffer(idxQueue, BUFFER_NUM, this->tileSize * sizeof(DTYPE_IDX));
        pipe.InitBuffer(updQueue, BUFFER_NUM, this->tileSize * sizeof(DTYPE_UPD));
    }

    __aicore__ inline void Process() {
        if (localRows == 0) {
            return;
        }
        // Step 1: Copy x -> y for owned rows
        for (uint32_t r = 0; r < localRows; ++r) {
            AscendC::DataCopy(yGm[r * xCols], xGm[r * xCols], xCols);
        }

        // Step 2: Scatter updates along dim=1 using indices
        for (uint32_t r = 0; r < localRows; ++r) {
            uint32_t tiles = (idxCols + tileSize - 1) / tileSize;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t base = t * tileSize;
                uint32_t curLen = (base + tileSize <= idxCols) ? tileSize : (idxCols - base);
                CopyIn(r, base, curLen);
                Compute(r, curLen);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t r, uint32_t base, uint32_t len) {
        AscendC::LocalTensor<DTYPE_IDX> idxLocal = idxQueue.AllocTensor<DTYPE_IDX>();
        AscendC::LocalTensor<DTYPE_UPD> updLocal = updQueue.AllocTensor<DTYPE_UPD>();
        AscendC::DataCopy(idxLocal, idxGm[r * idxCols + base], len);
        AscendC::DataCopy(updLocal, updGm[r * idxCols + base], len);
        idxQueue.EnQue(idxLocal);
        updQueue.EnQue(updLocal);
    }

    __aicore__ inline void Compute(uint32_t r, uint32_t len) {
        AscendC::LocalTensor<DTYPE_IDX> idxLocal = idxQueue.DeQue<DTYPE_IDX>();
        AscendC::LocalTensor<DTYPE_UPD> updLocal = updQueue.DeQue<DTYPE_UPD>();

        // For each index in the tile, write the corresponding update into y
        // Note: If indices have duplicates, the last one processed wins.
        for (uint32_t i = 0; i < len; ++i) {
            // Read column index and value from local tensors
            DTYPE_IDX col = idxLocal.GetValue(i);
            // Bounds check (optional; indices are assumed valid)
            if (col >= 0 && (uint64_t)col < (uint64_t)xCols) {
                // Store single element from local updates to global y at (r, col)
                AscendC::DataCopy(yGm[(uint64_t)r * xCols + (uint64_t)col], updLocal[i], 1);
            }
        }

        idxQueue.FreeTensor(idxLocal);
        updQueue.FreeTensor(updLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> idxQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> updQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_IDX> idxGm;
    AscendC::GlobalTensor<DTYPE_UPD> updGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t rows;
    uint32_t xCols;
    uint32_t idxCols;
    uint32_t tileSize;

    uint32_t rowStart;
    uint32_t rowEnd;
    uint32_t localRows;
};

extern "C" __global__ __aicore__ void scatter_custom(GM_ADDR x, GM_ADDR index, GM_ADDR updates, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelScatter op;
    op.Init(x, index, updates, y,
            tiling_data.rows, tiling_data.xCols, tiling_data.idxCols, tiling_data.tileSize);
    op.Process();
}
