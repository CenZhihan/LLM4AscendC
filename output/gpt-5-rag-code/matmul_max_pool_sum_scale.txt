kernel_src="""
#include "kernel_operator.h"

class KernelMatmulMaxPoolSumScaleCustom {
public:
    __aicore__ inline KernelMatmulMaxPoolSumScaleCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scale, GM_ADDR ks, GM_ADDR y,
                                uint32_t batch, uint32_t inFeat, uint32_t outFeat, uint32_t tileK)
    {
        this->batch = batch;
        this->inFeat = inFeat;
        this->outFeat = outFeat;
        this->tileK = tileK;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint64_t)batch * inFeat);
        wGm.SetGlobalBuffer((__gm__ float*)w, (uint64_t)outFeat * inFeat);
        bGm.SetGlobalBuffer((__gm__ float*)b, outFeat);
        sGm.SetGlobalBuffer((__gm__ float*)scale, 1);
        ksGm.SetGlobalBuffer((__gm__ int32_t*)ks, 1);
        yGm.SetGlobalBuffer((__gm__ float*)y, batch);
    }
    __aicore__ inline void Process()
    {
        // Read scale and kernel size
        AscendC::LocalTensor<float> sLocal = tmpScalarF.AllocTensor<float>();
        AscendC::LocalTensor<int32_t> ksLocal = tmpScalarI.AllocTensor<int32_t>();
        AscendC::DataCopy(sLocal, sGm, 1);
        AscendC::DataCopy(ksLocal, ksGm, 1);
        float scale = sLocal.GetValue(0);
        int32_t ksize = ksLocal.GetValue(0);
        if (ksize <= 0) { ksize = 1; }

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = (batch + blockNum - 1) / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > batch) endRow = batch;

        // Allocate local buffers
        AscendC::LocalTensor<float> xLocal = vecBufX.AllocTensor<float>();
        AscendC::LocalTensor<float> wLocal = vecBufW.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = vecBufT.AllocTensor<float>();

        for (uint32_t i = startRow; i < endRow; ++i) {
            float sumVal = 0.0f;
            // iterate output features by pooling window
            for (uint32_t j = 0; j < outFeat; j += (uint32_t)ksize) {
                float maxVal = -3.4e38f;
                uint32_t window = (j + (uint32_t)ksize <= outFeat) ? (uint32_t)ksize : (outFeat - j);
                for (uint32_t t = 0; t < window; ++t) {
                    uint32_t outCol = j + t;
                    float acc = 0.0f;
                    // dot product over inFeat with tiling
                    for (uint32_t k0 = 0; k0 < inFeat; k0 += tileK) {
                        uint32_t tk = ((k0 + tileK) <= inFeat) ? tileK : (inFeat - k0);
                        // copy x segment
                        AscendC::DataCopy(xLocal, xGm[i * inFeat + k0], tk);
                        // copy w segment for column outCol
                        AscendC::DataCopy(wLocal, wGm[outCol * inFeat + k0], tk);
                        // elementwise multiply into tmpLocal
                        AscendC::Mul(tmpLocal, xLocal, wLocal, tk);
                        // reduce sum of tmpLocal[0:tk]
                        float partial = 0.0f;
                        for (uint32_t u = 0; u < tk; ++u) {
                            partial += tmpLocal.GetValue(u);
                        }
                        acc += partial;
                    }
                    // add bias
                    AscendC::LocalTensor<float> bLocal = tmpScalarF2.AllocTensor<float>();
                    AscendC::DataCopy(bLocal, bGm[outCol], 1);
                    acc += bLocal.GetValue(0);
                    tmpScalarF2.FreeTensor(bLocal);

                    if (acc > maxVal) maxVal = acc;
                }
                sumVal += maxVal;
            }
            // scale and write out
            sumVal *= scale;
            AscendC::LocalTensor<float> yLocal = tmpScalarF3.AllocTensor<float>();
            yLocal.SetValue(0, sumVal);
            AscendC::DataCopy(yGm[i], yLocal, 1);
            tmpScalarF3.FreeTensor(yLocal);
        }

        // free locals
        vecBufX.FreeTensor(xLocal);
        vecBufW.FreeTensor(wLocal);
        vecBufT.FreeTensor(tmpLocal);
        tmpScalarF.FreeTensor(sLocal);
        tmpScalarI.FreeTensor(ksLocal);
    }

private:
    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> wGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> sGm;
    AscendC::GlobalTensor<int32_t> ksGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t batch;
    uint32_t inFeat;
    uint32_t outFeat;
    uint32_t tileK;

    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::TPosition::VECCALC> vecBufX;
    AscendC::TBuf<AscendC::TPosition::VECCALC> vecBufW;
    AscendC::TBuf<AscendC::TPosition::VECCALC> vecBufT;

    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpScalarF;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpScalarF2;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpScalarF3;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpScalarI;
};

extern "C" __global__ __aicore__ void matmul_max_pool_sum_scale_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scale, GM_ADDR ks, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMaxPoolSumScaleCustom op;
    op.Init(x, weight, bias, scale, ks, y,
            tiling_data.batch, tiling_data.inFeat, tiling_data.outFeat, tiling_data.tileK);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using custom AscendC fused operator:
    y = scale * sum(MaxPool1d(Linear(x)), dim=1)
    """
    def __init__(self, in_features, out_features, kernel_size, scale_factor):
        super(ModelNew, self).__init__()
        self.matmul = torch.nn.Linear(in_features, out_features)
        self.max_pool = torch.nn.MaxPool1d(kernel_size)
        self.scale_factor = scale_factor

    def forward(self, x):
        # Use fused custom op: pass parameters and attributes
        ks = self.max_pool.kernel_size if isinstance(self.max_pool.kernel_size, int) else int(self.max_pool.kernel_size)
        return custom_ops_lib.matmul_max_pool_sum_scale_custom(x, self.matmul.weight, self.matmul.bias, float(self.scale_factor), int(ks))
