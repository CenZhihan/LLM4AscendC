kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR z,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t tileNum)
    {
        this->N = n;
        this->C = c;
        this->H = h;
        this->W = w;
        this->tileNum = tileNum;

        // Assign batch range per core
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blk = AscendC::GetBlockIdx();
        this->nPerCore = (N + blockNum - 1) / blockNum;
        this->nStart = blk * this->nPerCore;
        this->nEnd = this->nStart + this->nPerCore;
        if (this->nEnd > N) this->nEnd = N;

        // Set GM buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint32_t)N*C*H*W);
        bGm.SetGlobalBuffer((__gm__ DTYPE_X*)bias, (uint32_t)C); // bias size Cx1x1 flattened
        zGm.SetGlobalBuffer((__gm__ DTYPE_X*)z, (uint32_t)N);

        // Allocate local buffers:
        // - One for a single (H*W) tile per (n,c)
        // - One for channel vector per n
        uint32_t spatial = H * W;
        pipe.InitBuffer(inQueueX, BUFFER_NUM, spatial * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueBias, 1, C * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, sizeof(DTYPE_X));

        // Load bias into local once
        AscendC::LocalTensor<DTYPE_X> bLocal = inQueueBias.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(bLocal, bGm, C);
        inQueueBias.EnQue(bLocal);
    }

    __aicore__ inline void Process()
    {
        if (this->nStart >= this->nEnd) {
            return;
        }
        AscendC::LocalTensor<DTYPE_X> bLocal = inQueueBias.DeQue<DTYPE_X>();

        for (uint32_t nIdx = this->nStart; nIdx < this->nEnd; ++nIdx) {
            // Temporary storage for per-channel aggregated values (after global avg and bias add)
            // Allocate in UB using inQueueBias buffer when not used; here we simulate a local vector
            // We will reuse the input buffer to compute channel-wise results sequentially
            // After computing all channels, compute logsumexp across C.

            // Accumulate LSE components
            // First pass: compute exp values and sum them
            DTYPE_X sumExp = static_cast<DTYPE_X>(0);

            for (uint32_t cIdx = 0; cIdx < C; ++cIdx) {
                // Copy one (H*W) map into local
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                uint32_t spatial = H * W;
                uint64_t base = ((uint64_t)nIdx * C + cIdx) * (uint64_t)spatial;
                AscendC::DataCopy(xLocal, xGm[base], spatial);

                // Reduce sum over H*W
                // Accumulate in scalar
                DTYPE_X sumVal = static_cast<DTYPE_X>(0);
                // Vectorized partial sums
                // Use a simple loop; each element contributes to sumVal
                for (uint32_t i = 0; i < spatial; ++i) {
                    sumVal += xLocal.GetValue(i);
                }

                // Global average
                DTYPE_X meanVal = sumVal / static_cast<DTYPE_X>(spatial);

                // Add bias (C,1,1)
                DTYPE_X biased = meanVal + bLocal.GetValue(cIdx);

                // Accumulate exp for logsumexp along channel
                DTYPE_X e = AscendC::Exp(biased);
                sumExp += e;

                inQueueX.FreeTensor(xLocal);
            }

            // logsumexp over channel dimension (keepdim=True)
            DTYPE_X lse = AscendC::Log(sumExp);

            // Sum over (2,3) is trivial since it's 1x1 after GAP; multiply by 10
            DTYPE_X outVal = lse * static_cast<DTYPE_X>(10.0f);

            // Write output for this n
            AscendC::LocalTensor<DTYPE_X> zLocal = outQueueZ.AllocTensor<DTYPE_X>();
            zLocal.SetValue(0, outVal);
            outQueueZ.EnQue(zLocal);

            AscendC::LocalTensor<DTYPE_X> zLocalDeq = outQueueZ.DeQue<DTYPE_X>();
            AscendC::DataCopy(zGm[nIdx], zLocalDeq, 1);
            outQueueZ.FreeTensor(zLocalDeq);
        }

        inQueueBias.FreeTensor(bLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueBias;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> bGm;
    AscendC::GlobalTensor<DTYPE_X> zGm;

    uint32_t N, C, H, W;
    uint32_t tileNum;

    uint32_t nPerCore;
    uint32_t nStart;
    uint32_t nEnd;
};

extern "C" __global__ __aicore__ void convtranspose2d_globalavgpool_biasadd_logsumexp_sum_multiply_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, bias, z,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.tileNum);
    op.Process();
}
