kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // buffers per queue

class KernelCVitConcat {
public:
    __aicore__ inline KernelCVitConcat() {}
    __aicore__ inline void Init(GM_ADDR cls, GM_ADDR x, GM_ADDR y,
                                 uint32_t batch, uint32_t seq1, uint32_t seq2, uint32_t hidden)
    {
        this->batch = batch;
        this->seq1 = seq1;
        this->seq2 = seq2;
        this->hidden = hidden;

        uint32_t in0_len = batch * seq1 * hidden;
        uint32_t in1_len = batch * seq2 * hidden;
        uint32_t out_len = batch * (seq1 + seq2) * hidden;

        clsGm.SetGlobalBuffer((__gm__ DTYPE_X*)cls, in0_len);
        xGm.SetGlobalBuffer((__gm__ DTYPE_Y*)x, in1_len);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y, out_len);

        pipe.InitBuffer(inQueueCls, BUFFER_NUM, hidden * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueX, BUFFER_NUM, hidden * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, hidden * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t b = 0; b < batch; ++b) {
            // copy seq1 slices from cls -> y
            for (uint32_t s = 0; s < seq1; ++s) {
                CopyInCls(b, s);
                CopyOutToY(b, s, /*fromX=*/false);
            }
            // copy seq2 slices from x   -> y (after seq1)
            for (uint32_t s = 0; s < seq2; ++s) {
                CopyInX(b, s);
                CopyOutToY(b, s + seq1, /*fromX=*/true);
            }
        }
    }

private:
    __aicore__ inline void CopyInCls(uint32_t b, uint32_t s)
    {
        AscendC::LocalTensor<DTYPE_X> clsLocal = inQueueCls.AllocTensor<DTYPE_X>();
        uint32_t in_offset = (b * seq1 + s) * hidden;
        AscendC::DataCopy(clsLocal, clsGm[in_offset], hidden);
        inQueueCls.EnQue<DTYPE_X>(clsLocal);
    }

    __aicore__ inline void CopyInX(uint32_t b, uint32_t s)
    {
        AscendC::LocalTensor<DTYPE_Y> xLocal = inQueueX.AllocTensor<DTYPE_Y>();
        uint32_t in_offset = (b * seq2 + s) * hidden;
        AscendC::DataCopy(xLocal, xGm[in_offset], hidden);
        inQueueX.EnQue<DTYPE_Y>(xLocal);
    }

    __aicore__ inline void CopyOutToY(uint32_t b, uint32_t out_s, bool fromX)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        if (fromX) {
            AscendC::LocalTensor<DTYPE_Y> xLocal = inQueueX.DeQue<DTYPE_Y>();
            // reinterpret local tensor to output type width (same dtype here)
            AscendC::DataCopy(yLocal, xLocal, hidden);
            inQueueX.FreeTensor(xLocal);
        } else {
            AscendC::LocalTensor<DTYPE_X> clsLocal = inQueueCls.DeQue<DTYPE_X>();
            AscendC::DataCopy(yLocal, clsLocal, hidden);
            inQueueCls.FreeTensor(clsLocal);
        }
        outQueueY.EnQue<DTYPE_Z>(yLocal);

        AscendC::LocalTensor<DTYPE_Z> yOut = outQueueY.DeQue<DTYPE_Z>();
        uint32_t out_offset = (b * (seq1 + seq2) + out_s) * hidden;
        AscendC::DataCopy(yGm[out_offset], yOut, hidden);
        outQueueY.FreeTensor(yOut);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueCls, inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> clsGm;
    AscendC::GlobalTensor<DTYPE_Y> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t batch;
    uint32_t seq1;
    uint32_t seq2;
    uint32_t hidden;
};

extern "C" __global__ __aicore__ void convolutional_vision_transformer_custom(GM_ADDR cls, GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelCVitConcat op;
    op.Init(cls, x, y,
            tiling_data.batch(),
            tiling_data.seq1(),
            tiling_data.seq2(),
            tiling_data.hidden());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_classes, embed_dim=512, num_heads=8, num_layers=6,
                 mlp_ratio=4.0, patch_size=4, in_channels=3, image_size=32):
        super().__init__()
        self.patch_size = patch_size
        self.image_size = image_size
        self.embed_dim = embed_dim

        self.conv1 = torch.nn.Conv2d(in_channels, embed_dim, kernel_size=patch_size, stride=patch_size)
        num_patches = (image_size // patch_size) ** 2
        self.linear_proj = torch.nn.Linear(embed_dim * num_patches, embed_dim)

        self.transformer_layers = torch.nn.ModuleList([
            torch.nn.TransformerEncoderLayer(
                d_model=embed_dim,
                nhead=num_heads,
                dim_feedforward=int(embed_dim * mlp_ratio),
                dropout=0.0,
                batch_first=True
            ) for _ in range(num_layers)
        ])

        self.cls_token = torch.nn.Parameter(torch.zeros(1, 1, embed_dim))
        self.fc_out = torch.nn.Linear(embed_dim, num_classes)

    def forward(self, x):
        B = x.size(0)
        x = self.conv1(x)                  # (B, embed_dim, H/patch_size, W/patch_size)
        x = x.flatten(start_dim=1)         # (B, embed_dim * num_patches)
        x = self.linear_proj(x)            # (B, embed_dim)

        cls_tokens = self.cls_token.expand(B, -1, -1)  # (B, 1, embed_dim)
        # Use custom AscendC op to build (B, 2, embed_dim) by concatenating cls and x
        x = custom_ops_lib.convolutional_vision_transformer_custom(cls_tokens, x)  # (B, 2, embed_dim)

        for layer in self.transformer_layers:
            x = layer(x)

        return self.fc_out(x[:, 0])
'''
