kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 1; // simple buffering

// Assume data type float
using DTYPE = float;

class KernelMHA {
public:
    __aicore__ inline KernelMHA() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y, uint32_t b, uint32_t h, uint32_t s, uint32_t d, uint32_t tileNum)
    {
        this->B = b;
        this->H = h;
        this->S = s;
        this->D = d;
        this->N = b * h; // number of (batch, head) groups
        this->tileNum = tileNum > 0 ? tileNum : 1;

        // each block handles one (batch, head) group
        uint32_t blkIdx = AscendC::GetBlockIdx();
        qGm.SetGlobalBuffer((__gm__ DTYPE*)q + blkIdx * (S * D), S * D);
        kGm.SetGlobalBuffer((__gm__ DTYPE*)k + blkIdx * (S * D), S * D);
        vGm.SetGlobalBuffer((__gm__ DTYPE*)v + blkIdx * (S * D), S * D);
        yGm.SetGlobalBuffer((__gm__ DTYPE*)y + blkIdx * (S * D), S * D);

        // buffers: one vector of length D, and one score buffer of length S, and one acc vector of length D
        pipe.InitBuffer(qVecQ, BUFFER_NUM, D * sizeof(DTYPE));
        pipe.InitBuffer(accVecQ, BUFFER_NUM, D * sizeof(DTYPE));
        pipe.InitBuffer(scoreQ, BUFFER_NUM, S * sizeof(DTYPE));
    }

    __aicore__ inline void Process()
    {
        // For each query position t in [0, S)
        for (uint32_t t = 0; t < S; ++t) {
            // Load q[t, :]
            AscendC::LocalTensor<DTYPE> qVec = qVecQ.AllocTensor<DTYPE>();
            AscendC::DataCopy(qVec, qGm[t * D], D);

            // Compute scores = q[t,:] dot K[j,:] for j in [0,S)
            AscendC::LocalTensor<DTYPE> scores = scoreQ.AllocTensor<DTYPE>();
            ComputeScores(qVec, scores);

            // Softmax over scores
            SoftmaxInplace(scores);

            // Compute output acc = sum_j scores[j] * V[j,:]
            AscendC::LocalTensor<DTYPE> acc = accVecQ.AllocTensor<DTYPE>();
            ComputeWeightedSum(scores, acc);

            // Store acc to y[t, :]
            AscendC::DataCopy(yGm[t * D], acc, D);

            // release local tensors
            scoreQ.FreeTensor(scores);
            accVecQ.FreeTensor(acc);
            qVecQ.FreeTensor(qVec);
        }
    }

private:
    __aicore__ inline void ComputeScores(const AscendC::LocalTensor<DTYPE>& qVec, AscendC::LocalTensor<DTYPE>& scores)
    {
        // For each key position j
        for (uint32_t j = 0; j < S; ++j) {
            // dot product qVec . k[j,:]
            DTYPE sum = (DTYPE)0;
            // Load k[j,:] chunk-by-chunk
            // For simplicity, iterate element-wise
            for (uint32_t dd = 0; dd < D; ++dd) {
                DTYPE qv = qVec.GetValue(dd);
                DTYPE kv = kGm.GetValue(j * D + dd);
                sum += qv * kv;
            }
            // scale by sqrt(D)
            DTYPE scale = (DTYPE)1.0 / (DTYPE)AscendC::Sqrt((DTYPE)D);
            scores.SetValue(j, sum * scale);
        }
    }

    __aicore__ inline void SoftmaxInplace(AscendC::LocalTensor<DTYPE>& scores)
    {
        // find max for numerical stability
        DTYPE maxV = scores.GetValue(0);
        for (uint32_t j = 1; j < S; ++j) {
            DTYPE v = scores.GetValue(j);
            maxV = v > maxV ? v : maxV;
        }
        // compute exp and sum
        DTYPE sum = (DTYPE)0;
        for (uint32_t j = 0; j < S; ++j) {
            DTYPE v = AscendC::Exp(scores.GetValue(j) - maxV);
            scores.SetValue(j, v);
            sum += v;
        }
        // normalize
        DTYPE invSum = (DTYPE)1.0 / sum;
        for (uint32_t j = 0; j < S; ++j) {
            scores.SetValue(j, scores.GetValue(j) * invSum);
        }
    }

    __aicore__ inline void ComputeWeightedSum(const AscendC::LocalTensor<DTYPE>& scores, AscendC::LocalTensor<DTYPE>& acc)
    {
        // initialize acc to zeros
        for (uint32_t dd = 0; dd < D; ++dd) {
            acc.SetValue(dd, (DTYPE)0);
        }
        // accumulate scores[j] * v[j,:]
        for (uint32_t j = 0; j < S; ++j) {
            DTYPE w = scores.GetValue(j);
            for (uint32_t dd = 0; dd < D; ++dd) {
                DTYPE prev = acc.GetValue(dd);
                DTYPE vv = vGm.GetValue(j * D + dd);
                acc.SetValue(dd, prev + w * vv);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qVecQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> scoreQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> accVecQ;

    AscendC::GlobalTensor<DTYPE> qGm;
    AscendC::GlobalTensor<DTYPE> kGm;
    AscendC::GlobalTensor<DTYPE> vGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t B;
    uint32_t H;
    uint32_t S;
    uint32_t D;
    uint32_t N;
    uint32_t tileNum;
};

// Kernel expects q, k, v, y packed as [B, H, S, Dh] and launches B*H blocks
extern "C" __global__ __aicore__ void multi_head_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMHA op;
    op.Init(q, k, v, y,
            tiling_data.b, tiling_data.h, tiling_data.s, tiling_data.d,
            tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    """
    Multi-head attention with custom AscendC kernel for scaled dot-product attention.
    Projections (Q,K,V and output) are standard Linear layers; attention core is custom op.
    """
    def __init__(self, d_model=512, num_heads=8):
        super().__init__()
        assert d_model % num_heads == 0, "d_model must be divisible by num_heads"
        self.d_model = d_model
        self.num_heads = num_heads
        self.head_dim = d_model // num_heads
        self.q_proj = nn.Linear(d_model, d_model)
        self.k_proj = nn.Linear(d_model, d_model)
        self.v_proj = nn.Linear(d_model, d_model)
        self.out_proj = nn.Linear(d_model, d_model)

    def forward(self, x):
        # x: [B, S, D]
        B, S, D = x.shape
        # Linear projections
        q = self.q_proj(x)  # [B, S, D]
        k = self.k_proj(x)
        v = self.v_proj(x)
        # Reshape to [B, H, S, Dh]
        q = q.view(B, S, self.num_heads, self.head_dim).permute(0, 2, 1, 3).contiguous()
        k = k.view(B, S, self.num_heads, self.head_dim).permute(0, 2, 1, 3).contiguous()
        v = v.view(B, S, self.num_heads, self.head_dim).permute(0, 2, 1, 3).contiguous()
        # Custom attention op: returns [B, H, S, Dh]
        y = custom_ops_lib.multi_head_attention_custom(q, k, v)
        # Merge heads and project
        y = y.permute(0, 2, 1, 3).contiguous().view(B, S, D)
        out = self.out_proj(y)
        return out
'''
