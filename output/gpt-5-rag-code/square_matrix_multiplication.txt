host_operator_src="""
#include "square_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    SquareMatrixMultiplicationCustomTilingData tiling;
    const auto* a_shape = context->GetInputShape(0);
    if (a_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }
    // Expect square (N, N)
    uint32_t m = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    uint32_t n = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetDim(1));
    uint32_t k = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));

    context->SetBlockDim(BLOCK_DIM);
    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_blockDim(BLOCK_DIM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // no workspace
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return GRAPH_FAILED;
    }
    gert::Shape* c_shape = context->GetOutputShape(0);
    // C = (m x k) * (k x n) -> (m x n)
    std::vector<int64_t> out_dims = {a_shape->GetDim(0), b_shape->GetDim(1)};
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, out_dims[0]);
    c_shape->SetDim(1, out_dims[1]);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class SquareMatrixMultiplicationCustom : public OpDef {
public:
    explicit SquareMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore()
            .SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SquareMatrixMultiplicationCustom);
}

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm)
{
    // Use Matmul library to implement C = A * B
    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD> A_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD> B_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD> C_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> BIAS_TYPE;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> mmKernel;
    AscendC::TPipe pipe;

    // Share all on-chip resources to the Matmul library
    tiling.shareMode = 0;
#if defined(FULL_L1_SIZE)
    tiling.shareL1Size = FULL_L1_SIZE;
#else
    tiling.shareL1Size = 0;
#endif
#if defined(FULL_L0C_SIZE)
    tiling.shareL0CSize = FULL_L0C_SIZE;
#else
    tiling.shareL0CSize = 0;
#endif
    tiling.shareUbSize = 0;

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mmKernel.matmulObj, &tiling);
    mmKernel.Init(a, b, nullptr, c, workspace, tiling);
    // No bias, default repeat params
    mmKernel.Process<false>(&pipe, 1, 1);
}
