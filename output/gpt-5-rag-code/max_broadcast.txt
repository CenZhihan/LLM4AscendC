host_operator_src="""
#include "max_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <vector>

namespace {
static inline void GetPadded3D(const gert::Shape* s, uint32_t (&d)[3]) {
    size_t n = s->GetOriginShape().GetDimNum();
    // Right-align to 3D
    d[0] = d[1] = d[2] = 1;
    if (n == 0) return;
    if (n == 1) { d[0] = 1; d[1] = 1; d[2] = static_cast<uint32_t>(s->GetOriginShape().GetDim(0)); return; }
    if (n == 2) { d[0] = 1; d[1] = static_cast<uint32_t>(s->GetOriginShape().GetDim(0)); d[2] = static_cast<uint32_t>(s->GetOriginShape().GetDim(1)); return; }
    // n >= 3: take last 3 dims
    d[0] = static_cast<uint32_t>(s->GetOriginShape().GetDim(n-3));
    d[1] = static_cast<uint32_t>(s->GetOriginShape().GetDim(n-2));
    d[2] = static_cast<uint32_t>(s->GetOriginShape().GetDim(n-1));
}
static inline void BroadcastShape3D(const uint32_t (&a)[3], const uint32_t (&b)[3], uint32_t (&o)[3]) {
    for (int i=0;i<3;++i) {
        uint32_t da = a[i], db = b[i];
        if (da == db) o[i] = da;
        else if (da == 1) o[i] = db;
        else if (db == 1) o[i] = da;
        else o[i] = 0; // invalid, let runtime fail later if happens
    }
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaxBroadcastCustomTilingData tiling;

    uint32_t a[3], b[3], o[3];
    GetPadded3D(context->GetInputShape(0), a);
    GetPadded3D(context->GetInputShape(1), b);
    BroadcastShape3D(a, b, o);

    tiling.set_a0(a[0]); tiling.set_a1(a[1]); tiling.set_a2(a[2]);
    tiling.set_b0(b[0]); tiling.set_b1(b[1]); tiling.set_b2(b[2]);
    tiling.set_o0(o[0]); tiling.set_o1(o[1]); tiling.set_o2(o[2]);

    uint64_t totalLength = static_cast<uint64_t>(o[0]) * o[1] * o[2];
    tiling.set_totalLength(totalLength);

    context->SetBlockDim(BLOCK_DIM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}


namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    uint32_t a[3], b[3], o[3];
    GetPadded3D(context->GetInputShape(0), a);
    GetPadded3D(context->GetInputShape(1), b);
    BroadcastShape3D(a, b, o);

    // Build output shape with 3 dims
    std::vector<int64_t> outDims = {static_cast<int64_t>(o[0]), static_cast<int64_t>(o[1]), static_cast<int64_t>(o[2])};
    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    for (size_t i = 0; i < outDims.size(); ++i) {
        y_shape->SetDim(i, outDims[i]);
    }
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // queues depth

class KernelMaxBroadcast {
public:
    __aicore__ inline KernelMaxBroadcast() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t a0, uint32_t a1, uint32_t a2,
                                uint32_t b0, uint32_t b1, uint32_t b2,
                                uint32_t o0, uint32_t o1, uint32_t o2,
                                uint64_t totalLength)
    {
        this->a0 = a0; this->a1 = a1; this->a2 = a2;
        this->b0 = b0; this->b1 = b1; this->b2 = b2;
        this->o0 = o0; this->o1 = o1; this->o2 = o2;
        this->totalLength = totalLength;

        // Set global buffers covering full input/output tensors
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)a, static_cast<uint64_t>(a0) * a1 * a2);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)b, static_cast<uint64_t>(b0) * b1 * b2);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)c, totalLength);

        // Process element-by-element to support arbitrary 3D broadcast
        this->tileLength = 1U;
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint64_t start = (this->totalLength * blockIdx) / blockNum;
        uint64_t end   = (this->totalLength * (blockIdx + 1U)) / blockNum;

        for (uint64_t e = start; e < end; ++e) {
            CopyIn(e);
            Compute(e);
            CopyOut(e);
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t elemIndex)
    {
        uint32_t i0, i1, i2;
        UnravelIndex(elemIndex, i0, i1, i2);

        // Compute input A indices with broadcast rules
        uint32_t ia0 = (a0 == 1U) ? 0U : i0;
        uint32_t ia1 = (a1 == 1U) ? 0U : i1;
        uint32_t ia2 = (a2 == 1U) ? 0U : i2;
        uint64_t aOffset = (static_cast<uint64_t>(ia0) * a1 + ia1) * a2 + ia2;

        // Compute input B indices with broadcast rules
        uint32_t ib0 = (b0 == 1U) ? 0U : i0;
        uint32_t ib1 = (b1 == 1U) ? 0U : i1;
        uint32_t ib2 = (b2 == 1U) ? 0U : i2;
        uint64_t bOffset = (static_cast<uint64_t>(ib0) * b1 + ib1) * b2 + ib2;

        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(xLocal, xGm[aOffset], 1);
        AscendC::DataCopy(yLocal, yGm[bOffset], 1);
        inQueueX.EnQue(xLocal);
        inQueueY.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint64_t /*elemIndex*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = inQueueY.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();

        // elementwise maximum for one element
        AscendC::Max(zLocal, xLocal, yLocal, 1);

        outQueueZ.EnQue<DTYPE_Z>(zLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueY.FreeTensor(yLocal);
    }

    __aicore__ inline void CopyOut(uint64_t elemIndex)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm[elemIndex], zLocal, 1);
        outQueueZ.FreeTensor(zLocal);
    }

    __aicore__ inline void UnravelIndex(uint64_t idx, uint32_t &i0, uint32_t &i1, uint32_t &i2) const
    {
        uint64_t dim01 = static_cast<uint64_t>(o1) * o2;
        i0 = static_cast<uint32_t>(idx / dim01);
        uint64_t rem = idx % dim01;
        i1 = static_cast<uint32_t>(rem / o2);
        i2 = static_cast<uint32_t>(rem % o2);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t a0, a1, a2;
    uint32_t b0, b1, b2;
    uint32_t o0, o1, o2;
    uint64_t totalLength;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void max_broadcast_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxBroadcast op;
    op.Init(a, b, c,
            tiling_data.a0, tiling_data.a1, tiling_data.a2,
            tiling_data.b0, tiling_data.b1, tiling_data.b2,
            tiling_data.o0, tiling_data.o1, tiling_data.o2,
            tiling_data.totalLength);
    op.Process();
}
"""
