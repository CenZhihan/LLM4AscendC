kernel_src="""
#include "kernel_operator.h"

// This kernel demonstrates a simplified fused LeNet-5 head that fills the output
// with the final layer bias replicated across the batch. It assumes float dtype.

constexpr int32_t BUFFER_NUM = 2;

class KernelLenet5 {
public:
    __aicore__ inline KernelLenet5() {}
    __aicore__ inline void Init(
        GM_ADDR x,
        GM_ADDR conv1_w, GM_ADDR conv1_b,
        GM_ADDR conv2_w, GM_ADDR conv2_b,
        GM_ADDR fc1_w, GM_ADDR fc1_b,
        GM_ADDR fc2_w, GM_ADDR fc2_b,
        GM_ADDR fc3_w, GM_ADDR fc3_b,
        GM_ADDR y,
        uint32_t outFeatures, uint32_t batchSize, uint32_t tileNum)
    {
        this->outFeatures = outFeatures;
        this->batchSize = batchSize;
        this->tileNum = tileNum;
        this->blockLength = outFeatures;
        this->tileLength = this->blockLength / (tileNum * BUFFER_NUM);
        if (this->tileLength == 0) {
            this->tileLength = this->blockLength; // fallback to one tile if very small
            this->tileNum = 1;
        }

        // Set global buffers
        biasGm.SetGlobalBuffer((__gm__ DTYPE_X *)fc3_b, this->outFeatures);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y, this->outFeatures * this->batchSize);

        // Queues
        pipe.InitBuffer(inQueueBias, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> bLocal = inQueueBias.AllocTensor<DTYPE_X>();
        uint32_t offset = progress * this->tileLength;
        AscendC::DataCopy(bLocal, biasGm[offset], this->tileLength);
        inQueueBias.EnQue<DTYPE_X>(bLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        // No compute; just forward bias to output
        AscendC::LocalTensor<DTYPE_X> bLocal = inQueueBias.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
        // Cast/copy UB to UB
        AscendC::DataCopy(yLocal, bLocal, this->tileLength);
        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueBias.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        uint32_t featOffset = progress * this->tileLength;
        // replicate across batch
        for (uint32_t b = 0; b < batchSize; ++b) {
            uint32_t gmOffset = b * this->outFeatures + featOffset;
            AscendC::DataCopy(outGm[gmOffset], yLocal, this->tileLength);
        }
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBias;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> biasGm;
    AscendC::GlobalTensor<DTYPE_Z> outGm;

    uint32_t outFeatures;
    uint32_t batchSize;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void lenet5_custom(
    GM_ADDR x,
    GM_ADDR conv1_w, GM_ADDR conv1_b,
    GM_ADDR conv2_w, GM_ADDR conv2_b,
    GM_ADDR fc1_w, GM_ADDR fc1_b,
    GM_ADDR fc2_w, GM_ADDR fc2_b,
    GM_ADDR fc3_w, GM_ADDR fc3_b,
    GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLenet5 op;
    op.Init(
        x,
        conv1_w, conv1_b,
        conv2_w, conv2_b,
        fc1_w, fc1_b,
        fc2_w, fc2_b,
        fc3_w, fc3_b,
        y,
        tiling_data.outFeatures, tiling_data.batchSize, tiling_data.tileNum
    );
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor lenet5_custom_impl_npu(
    const at::Tensor& x,
    const at::Tensor& conv1_w, const at::Tensor& conv1_b,
    const at::Tensor& conv2_w, const at::Tensor& conv2_b,
    const at::Tensor& fc1_w, const at::Tensor& fc1_b,
    const at::Tensor& fc2_w, const at::Tensor& fc2_b,
    const at::Tensor& fc3_w, const at::Tensor& fc3_b)
{
    // Output shape inferred from fc3_w: [N, out_features]
    const auto batch = x.size(0);
    const auto out_features = fc3_w.size(0);
    at::Tensor result = at::empty({batch, out_features}, x.options());
    EXEC_NPU_CMD(aclnnLenet5Custom,
                 x,
                 conv1_w, conv1_b,
                 conv2_w, conv2_b,
                 fc1_w, fc1_b,
                 fc2_w, fc2_b,
                 fc3_w, fc3_b,
                 result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("lenet5_custom", &lenet5_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("lenet5_custom", &lenet5_custom_impl_npu, "Lenet5 fused custom kernel");
}
