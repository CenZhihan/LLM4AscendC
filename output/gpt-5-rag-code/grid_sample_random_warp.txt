kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2; // tile buffers for output

// Bilinear sampling with zeros padding and align_corners = false, channel_last = false (NCHW)
class KernelGridSampleRandomWarp {
public:
    __aicore__ inline KernelGridSampleRandomWarp() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR grid, GM_ADDR y, uint32_t n, uint32_t c, uint32_t hin, uint32_t win,
                                uint32_t hout, uint32_t wout, uint32_t totalOutputLength, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->hin = hin;
        this->win = win;
        this->hout = hout;
        this->wout = wout;
        this->totalOutputLength = totalOutputLength;
        this->tileNum = tileNum;

        this->blockLength = totalOutputLength / AscendC::GetBlockNum();
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->blockOutOffset = this->blockLength * AscendC::GetBlockIdx();

        // Full view for x and grid, block-partitioned view for y
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint32_t>(n * c * hin * win));
        gridGm.SetGlobalBuffer((__gm__ DTYPE_Y *)grid, static_cast<uint32_t>(n * hout * wout * 2));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockOutOffset, this->blockLength);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline bool in_bounds(int32_t y, int32_t x) const
    {
        return (y >= 0 && y < (int32_t)hin && x >= 0 && x < (int32_t)win);
    }

    __aicore__ inline DTYPE_Z sample_bilinear(uint32_t n_idx, uint32_t c_idx, float y_src, float x_src) const
    {
        // floor/ceil indices
        int32_t y0 = static_cast<int32_t>(floorf(y_src));
        int32_t x0 = static_cast<int32_t>(floorf(x_src));
        int32_t y1 = y0 + 1;
        int32_t x1 = x0 + 1;

        float wy1 = y_src - (float)y0;
        float wx1 = x_src - (float)x0;
        float wy0 = 1.0f - wy1;
        float wx0 = 1.0f - wx1;

        float acc = 0.0f;

        // p00
        if (in_bounds(y0, x0)) {
            uint32_t off00 = (((n_idx * c + c_idx) * hin + (uint32_t)y0) * win + (uint32_t)x0);
            float v00 = static_cast<float>(xGm[off00]);
            acc += v00 * (wy0 * wx0);
        }
        // p01
        if (in_bounds(y0, x1)) {
            uint32_t off01 = (((n_idx * c + c_idx) * hin + (uint32_t)y0) * win + (uint32_t)x1);
            float v01 = static_cast<float>(xGm[off01]);
            acc += v01 * (wy0 * wx1);
        }
        // p10
        if (in_bounds(y1, x0)) {
            uint32_t off10 = (((n_idx * c + c_idx) * hin + (uint32_t)y1) * win + (uint32_t)x0);
            float v10 = static_cast<float>(xGm[off10]);
            acc += v10 * (wy1 * wx0);
        }
        // p11
        if (in_bounds(y1, x1)) {
            uint32_t off11 = (((n_idx * c + c_idx) * hin + (uint32_t)y1) * win + (uint32_t)x1);
            float v11 = static_cast<float>(xGm[off11]);
            acc += v11 * (wy1 * wx1);
        }

        return static_cast<DTYPE_Z>(acc);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        for (uint32_t i = 0; i < this->tileLength; ++i) {
            uint32_t out_idx_in_block = progress * this->tileLength + i;
            uint32_t out_idx_total = this->blockOutOffset + out_idx_in_block;

            // unravel index: [N, C, H_out, W_out]
            uint32_t w = out_idx_total % wout;
            uint32_t tmp = out_idx_total / wout;
            uint32_t h = tmp % hout;
            tmp = tmp / hout;
            uint32_t ch = tmp % c;
            uint32_t n_idx = tmp / c;

            // grid idx: [N, H_out, W_out, 2]
            uint32_t grid_base = (((n_idx * hout + h) * wout + w) * 2);
            float gx = static_cast<float>(gridGm[grid_base + 0]);
            float gy = static_cast<float>(gridGm[grid_base + 1]);

            // align_corners = false mapping
            float x_src = (gx + 1.0f) * (float)win * 0.5f - 0.5f;
            float y_src = (gy + 1.0f) * (float)hin * 0.5f - 0.5f;

            // sample with bilinear and zeros padding
            yLocal[i] = sample_bilinear(n_idx, ch, y_src, x_src);
        }

        outQueueY.EnQue<DTYPE_Z>(yLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> gridGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t n, c, hin, win, hout, wout;
    uint32_t totalOutputLength;
    uint32_t tileNum;
    uint32_t blockLength;
    uint32_t tileLength;
    uint32_t blockOutOffset;
};

extern "C" __global__ __aicore__ void grid_sample_random_warp_custom(GM_ADDR x, GM_ADDR grid, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelGridSampleRandomWarp op;
    op.Init(x, grid, y,
            tiling_data.n(),
            tiling_data.c(),
            tiling_data.hin(),
            tiling_data.win(),
            tiling_data.hout(),
            tiling_data.wout(),
            tiling_data.totalOutputLength(),
            tiling_data.tileNum());
    op.Process();
}
