kernel_src="""
#include "kernel_operator.h"
#include <cmath>

static __aicore__ inline float sigmoid_scalar(float x) {
    return 1.f / (1.f + expf(-x));
}

extern "C" __global__ __aicore__ void gemm_group_norm_swish_multiply_swish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR gamma, GM_ADDR beta, GM_ADDR multiply_weight,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t M = tiling_data.M;
    const uint32_t N = tiling_data.N;
    const uint32_t K = tiling_data.K;
    const uint32_t groups = tiling_data.groups;
    const float eps = tiling_data.epsilon;

    __gm__ half* X = reinterpret_cast<__gm__ half*>(x);
    __gm__ half* W = reinterpret_cast<__gm__ half*>(weight);
    __gm__ float* B = reinterpret_cast<__gm__ float*>(bias);
    __gm__ float* G = reinterpret_cast<__gm__ float*>(gamma);
    __gm__ float* Be = reinterpret_cast<__gm__ float*>(beta);
    __gm__ float* MW = reinterpret_cast<__gm__ float*>(multiply_weight);
    __gm__ float* Y = reinterpret_cast<__gm__ float*>(y);

    uint32_t block_num = AscendC::GetBlockNum();
    uint32_t block_idx = AscendC::GetBlockIdx();

    uint32_t rows_per_block = (M + block_num - 1) / block_num;
    uint32_t row_start = block_idx * rows_per_block;
    uint32_t row_end = row_start + rows_per_block;
    if (row_end > M) row_end = M;

    uint32_t group_size = (groups == 0) ? N : (N / groups);

    // For each row assigned to this block
    for (uint32_t m = row_start; m < row_end; ++m) {
        // Iterate groups
        uint32_t gStart = 0;
        for (uint32_t g = 0; g < groups; ++g) {
            uint32_t cStart = g * group_size;
            uint32_t cEnd = cStart + group_size;

            // Pass 1: compute group mean and var from GEMM output
            float sum = 0.f;
            float sumsq = 0.f;
            for (uint32_t c = cStart; c < cEnd; ++c) {
                // dot product between X[m, :] and W[c, :]
                float acc = B[c];
                uint32_t xRowOff = m * K;
                uint32_t wRowOff = c * K;
                for (uint32_t k = 0; k < K; ++k) {
                    float xv = static_cast<float>(X[xRowOff + k]);
                    float wv = static_cast<float>(W[wRowOff + k]);
                    acc += xv * wv;
                }
                sum += acc;
                sumsq += acc * acc;
            }
            float mean = sum / static_cast<float>(group_size);
            float var = sumsq / static_cast<float>(group_size) - mean * mean;
            float inv_std = 1.0f / sqrtf(var + eps);

            // Pass 2: normalize + affine + swish + multiply + swish, write to Y
            for (uint32_t c = cStart; c < cEnd; ++c) {
                float acc = B[c];
                uint32_t xRowOff = m * K;
                uint32_t wRowOff = c * K;
                for (uint32_t k = 0; k < K; ++k) {
                    float xv = static_cast<float>(X[xRowOff + k]);
                    float wv = static_cast<float>(W[wRowOff + k]);
                    acc += xv * wv;
                }
                float normed = (acc - mean) * inv_std;
                float y1 = G[c] * normed + Be[c];
                float s1 = y1 * sigmoid_scalar(y1);
                float s2 = s1 * MW[c];
                float outv = s2 * sigmoid_scalar(s2);
                Y[m * N + c] = outv;
            }
        }
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, num_groups, multiply_weight_shape):
        super(ModelNew, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.num_groups = num_groups
        # retain original modules to reuse parameters
        self.gemm = torch.nn.Linear(in_features, out_features, bias=True)
        self.group_norm = torch.nn.GroupNorm(num_groups, out_features)
        self.multiply_weight = torch.nn.Parameter(torch.randn(multiply_weight_shape))

    def forward(self, x):
        # Call fused custom op: expects half inputs for GEMM, float params for norms/multiply
        x_half = x.to(torch.half)
        weight_half = self.gemm.weight.to(torch.half)
        bias = self.gemm.bias
        gamma = self.group_norm.weight
        beta = self.group_norm.bias
        mulw = self.multiply_weight
        eps = self.group_norm.eps
        return custom_ops_lib.gemm_group_norm_swish_multiply_swish_custom(
            x_half, weight_half, bias, gamma, beta, mulw, self.num_groups, eps
        )
