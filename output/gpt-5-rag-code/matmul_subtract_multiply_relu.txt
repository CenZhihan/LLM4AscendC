kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelMatmulSubtractMultiplyRelu {
public:
    __aicore__ inline KernelMatmulSubtractMultiplyRelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR subtract, GM_ADDR multiply, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        subGm.SetGlobalBuffer((__gm__ DTYPE_X *)subtract, 1);
        mulGm.SetGlobalBuffer((__gm__ DTYPE_X *)multiply, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
        pipe.InitBuffer(scalarSubQue, BUFFER_NUM, sizeof(DTYPE_X));
        pipe.InitBuffer(scalarMulQue, BUFFER_NUM, sizeof(DTYPE_X));
    }
    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);

        AscendC::LocalTensor<DTYPE_X> subLocal = scalarSubQue.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> mulLocal = scalarMulQue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(subLocal, subGm, 1);
        AscendC::DataCopy(mulLocal, mulGm, 1);
        scalarSubQue.EnQue(subLocal);
        scalarMulQue.EnQue(mulLocal);
    }
    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> subLocal = scalarSubQue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> mulLocal = scalarMulQue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // yLocal = relu( (xLocal - subtract_value) * multiply_value )
        // Assuming Adds and Muls have scalar versions.
        DTYPE_X subtract_scalar = subLocal.GetValue(0);
        DTYPE_X multiply_scalar = mulLocal.GetValue(0);

        AscendC::Adds(yLocal, xLocal, -subtract_scalar, this->tileLength);
        AscendC::Muls(yLocal, yLocal, multiply_scalar, this->tileLength);
        AscendC::Relu(yLocal, yLocal, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        scalarSubQue.FreeTensor(subLocal);
        scalarMulQue.FreeTensor(mulLocal);
    }
    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> scalarSubQue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> scalarMulQue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> subGm;
    AscendC::GlobalTensor<DTYPE_X> mulGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void matmul_subtract_multiply_relu_custom(GM_ADDR x, GM_ADDR subtract, GM_ADDR multiply, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSubtractMultiplyRelu op;
    op.Init(x, subtract, multiply, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, subtract_value, multiply_value):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features)
        self.subtract_value = subtract_value
        self.multiply_value = multiply_value

    def forward(self, x):
        # Preserve matmul (linear) in PyTorch, fuse subsequent ops into custom kernel
        x = self.linear(x)
        subtract_tensor = torch.tensor(self.subtract_value, dtype=x.dtype, device=x.device)
        multiply_tensor = torch.tensor(self.multiply_value, dtype=x.dtype, device=x.device)
        return custom_ops_lib.matmul_subtract_multiply_relu_custom(x, subtract_tensor, multiply_tensor)
