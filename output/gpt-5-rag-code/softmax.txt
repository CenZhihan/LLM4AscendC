host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>

namespace {
inline uint32_t CeilDivU32(uint32_t a, uint32_t b) { return (a + b - 1) / b; }
}

namespace optiling {
static const uint32_t DEFAULT_BLOCK_DIM = 32;
static const uint32_t DEFAULT_TILE_COLS = 4096;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    // Expect 2D: [batch, dim]
    uint32_t dims = in_shape->GetDimNum();
    if (dims < 2) {
        // Flatten all but last dim into rows if needed
        uint32_t total_elems = in_shape->GetOriginShape().GetShapeSize();
        tiling.set_rows(1);
        tiling.set_cols(total_elems);
    } else {
        uint32_t rows = static_cast<uint32_t>(in_shape->GetDim(0));
        uint32_t cols = static_cast<uint32_t>(in_shape->GetDim(1));
        tiling.set_rows(rows);
        tiling.set_cols(cols);
    }

    uint32_t rows = tiling.get_rows();
    uint32_t cols = tiling.get_cols();

    // Choose block dim based on rows
    uint32_t blockDim = std::min(DEFAULT_BLOCK_DIM, std::max<uint32_t>(1, rows));
    context->SetBlockDim(blockDim);

    // Tile along columns
    uint32_t tileCols = std::min(DEFAULT_TILE_COLS, cols);
    if (tileCols == 0) tileCols = cols;
    uint32_t tileNum = CeilDivU32(cols, tileCols);

    tiling.set_tileCols(tileCols);
    tiling.set_tileNum(tileNum);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0; // no workspace
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    auto in_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, in_dtype);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmax {
public:
    __aicore__ inline KernelSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t tileCols)
    {
        this->rows = rows;
        this->cols = cols;
        this->tileCols = tileCols;
        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();

        // assign rows to blocks
        this->rowsPerBlock = (rows + blockNum - 1) / blockNum;
        this->rowStart = this->rowsPerBlock * this->blockIdx;
        this->rowEnd = this->rowStart + this->rowsPerBlock;
        if (this->rowEnd > this->rows) this->rowEnd = this->rows;
        if (this->rowStart > this->rows) this->rowStart = this->rows;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)rowStart * cols, (uint64_t)(rowEnd - rowStart) * cols);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + (uint64_t)rowStart * cols, (uint64_t)(rowEnd - rowStart) * cols);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileCols * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Z));
        pipe.InitBuffer(inQueueY, BUFFER_NUM, this->tileCols * sizeof(DTYPE_Z)); // reuse for pass3
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            uint32_t localRowIndex = r - rowStart;
            // Pass 1: compute row max
            float maxScalar = -3.402823466e+38F; // -FLT_MAX
            uint32_t tiles = (cols + tileCols - 1) / tileCols;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t currLen = GetCurrTileLen(t);
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[localRowIndex * cols + t * tileCols], currLen);

                // reduce max
                AscendC::LocalTensor<float> tmpMax = calcBuf.AllocTensor<float>(1);
                AscendC::ReduceMax(tmpMax, xLocal, currLen);
                // read back scalar
                float tileMax = tmpMax.GetValue(0);
                if (tileMax > maxScalar) maxScalar = tileMax;
                calcBuf.FreeTensor(tmpMax);

                inQueueX.EnQue(xLocal);
                AscendC::LocalTensor<DTYPE_X> xLocalDeq = inQueueX.DeQue<DTYPE_X>();
                inQueueX.FreeTensor(xLocalDeq);
            }

            // Pass 2: compute sum of exp(x - max) and optionally write exp to y (temporary)
            float sumScalar = 0.0f;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t currLen = GetCurrTileLen(t);
                AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(xLocal, xGm[localRowIndex * cols + t * tileCols], currLen);

                AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

                AscendC::LocalTensor<float> bcastMax = calcBuf.AllocTensor<float>(currLen);
                AscendC::Duplicate(bcastMax, maxScalar, currLen);
                AscendC::Sub(yLocal, xLocal, bcastMax, currLen); // yLocal = x - max
                AscendC::Exp(yLocal, yLocal, currLen);           // yLocal = exp(x - max)

                // partial sum
                AscendC::LocalTensor<float> tmpSum = calcBuf.AllocTensor<float>(1);
                AscendC::ReduceSum(tmpSum, yLocal, currLen);
                sumScalar += tmpSum.GetValue(0);

                // Write temporary exp to GM y
                outQueueY.EnQue<DTYPE_Z>(yLocal);
                AscendC::DataCopy(yGm[localRowIndex * cols + t * tileCols], yLocal, currLen);
                outQueueY.FreeTensor(yLocal);

                // free temps
                calcBuf.FreeTensor(bcastMax);
                calcBuf.FreeTensor(tmpSum);

                inQueueX.EnQue(xLocal);
                AscendC::LocalTensor<DTYPE_X> xLocalDeq = inQueueX.DeQue<DTYPE_X>();
                inQueueX.FreeTensor(xLocalDeq);
            }

            // Pass 3: normalize exp by sum
            float invSum = 1.0f / sumScalar;
            for (uint32_t t = 0; t < tiles; ++t) {
                uint32_t currLen = GetCurrTileLen(t);

                AscendC::LocalTensor<DTYPE_Z> yLocalIn = inQueueY.AllocTensor<DTYPE_Z>();
                AscendC::DataCopy(yLocalIn, yGm[localRowIndex * cols + t * tileCols], currLen);

                AscendC::LocalTensor<float> bcastInv = calcBuf.AllocTensor<float>(currLen);
                AscendC::Duplicate(bcastInv, invSum, currLen);

                AscendC::LocalTensor<DTYPE_Z> yLocalOut = outQueueY.AllocTensor<DTYPE_Z>();
                AscendC::Mul(yLocalOut, yLocalIn, bcastInv, currLen);

                AscendC::DataCopy(yGm[localRowIndex * cols + t * tileCols], yLocalOut, currLen);

                // queue housekeeping
                inQueueY.EnQue(yLocalIn);
                AscendC::LocalTensor<DTYPE_Z> yLocalInDeq = inQueueY.DeQue<DTYPE_Z>();
                inQueueY.FreeTensor(yLocalInDeq);

                outQueueY.EnQue(yLocalOut);
                AscendC::LocalTensor<DTYPE_Z> yLocalOutDeq = outQueueY.DeQue<DTYPE_Z>();
                outQueueY.FreeTensor(yLocalOutDeq);

                calcBuf.FreeTensor(bcastInv);
            }
        }
    }

private:
    __aicore__ inline uint32_t GetCurrTileLen(uint32_t t) const
    {
        uint32_t start = t * tileCols;
        if (start + tileCols <= cols) return tileCols;
        return cols - start;
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueY;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBuf;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t rows;
    uint32_t cols;
    uint32_t tileCols;

    uint32_t blockNum;
    uint32_t blockIdx;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmax op;
    op.Init(x, y,
            tiling_data.rows,
            tiling_data.cols,
            tiling_data.tileCols);
    op.Process();
}
"""
