kernel_src="""
#include <cmath>
#include "kernel_operator.h"

class KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom {
public:
    __aicore__ inline KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR z, uint32_t M, uint32_t N, float negativeSlope) {
        this->M = M;
        this->N = N;
        this->negativeSlope = negativeSlope;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(M * N));
        zGm.SetGlobalBuffer((__gm__ DTYPE_Y *)z, (uint32_t)(M));
    }

    __aicore__ inline float LeakyRelu(float v) {
        return v >= 0 ? v : negativeSlope * v;
    }

    __aicore__ inline float GeluApprox(float x) {
        // Approximate GELU: 0.5 * x * (1 + tanh( sqrt(2/pi) * (x + 0.044715*x^3) ))
        const float k0 = 0.7978845608028654f; // sqrt(2/pi)
        const float k1 = 0.044715f;
        float x3 = x * x * x;
        float inner = k0 * (x + k1 * x3);
        float t = tanhf(inner);
        return 0.5f * x * (1.0f + t);
    }

    __aicore__ inline void Process() {
        int32_t blocks = AscendC::GetBlockNum();
        int32_t blockId = AscendC::GetBlockIdx();

        uint32_t startRow = (uint64_t)M * blockId / blocks;
        uint32_t endRow = (uint64_t)M * (blockId + 1) / blocks;

        for (uint32_t r = startRow; r < endRow; ++r) {
            // Compute logsumexp over dim=1 (columns)
            float maxVal = -INFINITY;
            uint64_t base = (uint64_t)r * N;

            for (uint32_t c = 0; c < N; ++c) {
                float v = xGm.GetValue(base + c);
                if (v > maxVal) maxVal = v;
            }

            float sumExp = 0.0f;
            for (uint32_t c = 0; c < N; ++c) {
                float v = xGm.GetValue(base + c);
                sumExp += expf(v - maxVal);
            }
            float lse = maxVal + logf(sumExp);

            // Two LeakyReLU
            float y = LeakyRelu(lse);
            y = LeakyRelu(y);

            // Two GELU
            y = GeluApprox(y);
            y = GeluApprox(y);

            zGm.SetValue(r, (DTYPE_Y)y);
        }
    }

private:
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> zGm;

    uint32_t M;
    uint32_t N;
    float negativeSlope;
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(GM_ADDR x, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling){
    GET_TILING_DATA(t, tiling);
    KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom op;
    op.Init(x, z, t.M, t.N, t.negativeSlope);
    op.Process();
}
