#include "product_reduction_over_a_dimension_custom_tiling.h"

namespace {
inline uint64_t SafeMul(uint64_t a, uint64_t b) {
    return a * b;
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // Read input shape
    const gert::Shape* x_shape = context->GetInputShape(0);
    auto origin = x_shape->GetOriginShape();

    int64_t dim_attr = 0;
    // Try to get 'dim' attribute; if not provided, default 0
    (void)context->GetAttr("dim", dim_attr);

    uint32_t rank = static_cast<uint32_t>(origin.GetDimNum());
    if (rank == 0) {
        // Scalar
        context->SetBlockDim(1);
        ProductReductionOverADimensionCustomTilingData tiling;
        tiling.set_outer(1);
        tiling.set_inner(1);
        tiling.set_reduceLen(1);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t* ws = context->GetWorkspaceSizes(1);
        ws[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    int64_t dim_norm = dim_attr;
    if (dim_norm < 0) dim_norm += rank;
    if (dim_norm < 0) dim_norm = 0;
    if (dim_norm >= rank) dim_norm = rank - 1;

    uint64_t outer = 1;
    for (int64_t i = 0; i < dim_norm; ++i) {
        outer = SafeMul(outer, static_cast<uint64_t>(origin.GetDim(i)));
    }
    uint64_t reduceLen = static_cast<uint64_t>(origin.GetDim(dim_norm));
    uint64_t inner = 1;
    for (int64_t i = dim_norm + 1; i < rank; ++i) {
        inner = SafeMul(inner, static_cast<uint64_t>(origin.GetDim(i)));
    }

    ProductReductionOverADimensionCustomTilingData tiling;
    tiling.set_outer(static_cast<uint32_t>(outer));
    tiling.set_inner(static_cast<uint32_t>(inner));
    tiling.set_reduceLen(static_cast<uint32_t>(reduceLen));

    // Choose a reasonable block dim
    uint32_t block_dim = 8;
    if (outer < block_dim) block_dim = static_cast<uint32_t>(outer == 0 ? 1 : outer);
    if (block_dim == 0) block_dim = 1;
    context->SetBlockDim(block_dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    auto origin = x_shape->GetOriginShape();

    int64_t dim_attr = 0;
    (void)context->GetAttr("dim", dim_attr);

    int64_t rank = static_cast<int64_t>(origin.GetDimNum());
    int64_t dim_norm = dim_attr;
    if (dim_norm < 0) dim_norm += rank;
    if (dim_norm < 0) dim_norm = 0;
    if (dim_norm >= rank) dim_norm = rank - 1;

    std::vector<int64_t> out_dims;
    out_dims.reserve(static_cast<size_t>(rank > 0 ? rank - 1 : 0));
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim_norm) continue;
        out_dims.push_back(origin.GetDim(i));
    }
    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelProdReduce {
public:
    __aicore__ inline KernelProdReduce() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outer, uint32_t inner, uint32_t reduceLen)
    {
        this->outer = outer;
        this->inner = inner;
        this->reduceLen = reduceLen;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint32_t>(outer) * static_cast<uint32_t>(reduceLen) * static_cast<uint32_t>(inner));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, static_cast<uint32_t>(outer) * static_cast<uint32_t>(inner));

        // We use local buffers sized by 'inner' to vectorize along contiguous dimension
        uint32_t bufSizeBytes = this->inner * sizeof(DTYPE_X);
        pipe.InitBuffer(inQueueX, 1, bufSizeBytes);
        pipe.InitBuffer(outQueueY, 1, bufSizeBytes);
    }

    __aicore__ inline void Process()
    {
        // Distribute 'outer' segments across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockId = AscendC::GetBlockIdx();
        uint32_t segPerBlock = (outer + blocks - 1) / blocks;
        uint32_t start = blockId * segPerBlock;
        uint32_t end = start + segPerBlock;
        if (end > outer) end = outer;

        for (uint32_t seg = start; seg < end; ++seg) {
            ReduceOneSegment(seg);
        }
    }

private:
    __aicore__ inline void ReduceOneSegment(uint32_t seg)
    {
        // Each segment covers a contiguous group of 'inner' outputs
        uint64_t baseIndex = static_cast<uint64_t>(seg) * static_cast<uint64_t>(reduceLen) * static_cast<uint64_t>(inner);

        // Allocate accumulator buffer (output type)
        AscendC::LocalTensor<DTYPE_Y> accLocal = outQueueY.AllocTensor<DTYPE_Y>();
        // Load first slice into accumulator
        AscendC::DataCopy(accLocal, xGm[baseIndex], inner);

        // Iteratively multiply with remaining slices
        for (uint32_t k = 1; k < reduceLen; ++k) {
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[baseIndex + static_cast<uint64_t>(k) * static_cast<uint64_t>(inner)], inner);
            AscendC::Mul(accLocal, accLocal, xLocal, inner);
            inQueueX.FreeTensor(xLocal);
        }

        // Write out
        AscendC::DataCopy(yGm[static_cast<uint64_t>(seg) * static_cast<uint64_t>(inner)], accLocal, inner);
        outQueueY.FreeTensor(accLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t outer;
    uint32_t inner;
    uint32_t reduceLen;
};

extern "C" __global__ __aicore__ void product_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelProdReduce op;
    op.Init(x, y, tiling_data.outer, tiling_data.inner, tiling_data.reduceLen);
    op.Process();
}
"""

static at::Tensor make_output_for_prod(const at::Tensor& self, int64_t dim) {
    int64_t ndim = self.dim();
    int64_t d = dim < 0 ? dim + ndim : dim;
    if (d < 0) d = 0;
    if (d >= ndim) d = ndim - 1;
    std::vector<int64_t> out_sizes = self.sizes().vec();
    out_sizes.erase(out_sizes.begin() + d);
    return at::empty(out_sizes, self.options());
