kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelWindowedCausalAttention {
public:
    __aicore__ inline KernelWindowedCausalAttention() {}
    __aicore__ inline void Init(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y,
                                uint32_t B, uint32_t N, uint32_t H, uint32_t Dh, uint32_t window)
    {
        this->B = B;
        this->N = N;
        this->H = H;
        this->Dh = Dh;
        this->window = window;

        uint64_t totalElems = static_cast<uint64_t>(B) * N * H * Dh;
        qGm.SetGlobalBuffer((__gm__ float*)q, totalElems);
        kGm.SetGlobalBuffer((__gm__ float*)k, totalElems);
        vGm.SetGlobalBuffer((__gm__ float*)v, totalElems);
        yGm.SetGlobalBuffer((__gm__ float*)y, totalElems);

        // UB buffers for one Dh-vector at a time
        pipe.InitBuffer(inVQueue, BUFFER_NUM, Dh * sizeof(float));
        pipe.InitBuffer(tmpAccQueue, BUFFER_NUM, Dh * sizeof(float));
        pipe.InitBuffer(outYQueue, BUFFER_NUM, Dh * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blk = AscendC::GetBlockIdx();
        uint32_t blocks = AscendC::GetBlockNum();
        // map block to (b,h)
        // Aim: one block per (b,h). If fewer blocks than B*H, stride.
        for (uint32_t bh = blk; bh < B * H; bh += blocks) {
            uint32_t b = bh / H;
            uint32_t h = bh % H;
            ComputeBH(b, h);
        }
    }

private:
    __aicore__ inline uint64_t Offset(uint32_t b, uint32_t i, uint32_t h, uint32_t d) const
    {
        // layout: [B, N, H, Dh]
        return (((static_cast<uint64_t>(b) * N + i) * H + h) * Dh + d);
    }

    __aicore__ inline void LoadV(LocalTensor<float>& vLocal, uint32_t b, uint32_t pos, uint32_t h) {
        AscendC::DataCopy(vLocal, vGm[Offset(b, pos, h, 0)], Dh);
    }
    __aicore__ inline void StoreY(LocalTensor<float>& yLocal, uint32_t b, uint32_t pos, uint32_t h) {
        AscendC::DataCopy(yGm[Offset(b, pos, h, 0)], yLocal, Dh);
    }

    __aicore__ inline void ComputeBH(uint32_t b, uint32_t h)
    {
        for (uint32_t i = 0; i < N; ++i) {
            uint32_t start = (i > window) ? (i - window) : 0;

            // acc = V[b, start, h, :]
            AscendC::LocalTensor<float> vLocal = inVQueue.AllocTensor<float>();
            LoadV(vLocal, b, start, h);

            AscendC::LocalTensor<float> accLocal = tmpAccQueue.AllocTensor<float>();
            AscendC::DataCopy(accLocal, vLocal, Dh);
            inVQueue.FreeTensor(vLocal);

            for (uint32_t j = start + 1; j <= i; ++j) {
                AscendC::LocalTensor<float> vLocal2 = inVQueue.AllocTensor<float>();
                LoadV(vLocal2, b, j, h);
                AscendC::Add(accLocal, accLocal, vLocal2, Dh);
                inVQueue.FreeTensor(vLocal2);
            }

            // Output y = accumulated V (sum over window). This is a simple local attention aggregation.
            AscendC::LocalTensor<float> yLocal = outYQueue.AllocTensor<float>();
            AscendC::DataCopy(yLocal, accLocal, Dh);
            outYQueue.EnQue(yLocal);
            tmpAccQueue.FreeTensor(accLocal);

            AscendC::LocalTensor<float> yDeq = outYQueue.DeQue<float>();
            StoreY(yDeq, b, i, h);
            outYQueue.FreeTensor(yDeq);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inVQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpAccQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outYQueue;

    AscendC::GlobalTensor<float> qGm;
    AscendC::GlobalTensor<float> kGm;
    AscendC::GlobalTensor<float> vGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t B;
    uint32_t N;
    uint32_t H;
    uint32_t Dh;
    uint32_t window;
};

extern "C" __global__ __aicore__ void windowed_causal_attention_custom(GM_ADDR q, GM_ADDR k, GM_ADDR v, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelWindowedCausalAttention op;
    op.Init(q, k, v, y,
            tiling_data.B, tiling_data.N, tiling_data.H, tiling_data.Dh, tiling_data.window);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, d_model=1024, num_heads=16, window_size=256):
        super().__init__()
        self.q_proj = nn.Linear(d_model, d_model)
        self.k_proj = nn.Linear(d_model, d_model)
        self.v_proj = nn.Linear(d_model, d_model)
        self.out_proj = nn.Linear(d_model, d_model)
        self.num_heads = num_heads
        self.window_size = window_size

    def forward(self, x):
        B, N, D = x.shape
        H = self.num_heads
        d_h = D // H

        Q = self.q_proj(x).view(B, N, H, d_h).contiguous()
        K = self.k_proj(x).view(B, N, H, d_h).contiguous()
        V = self.v_proj(x).view(B, N, H, d_h).contiguous()

        # Call custom NPU kernel
        attn_out = custom_ops_lib.windowed_causal_attention_custom(Q, K, V)

        return self.out_proj(attn_out.reshape(B, N, D))
