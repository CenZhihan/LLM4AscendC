#include "instance_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }
}

namespace optiling {
static const uint32_t MAX_BLOCKS = 32;
static const uint32_t BUFFER_NUM = 2;
static const uint32_t DEFAULT_TILE_NUM = 8;
static const float DEFAULT_EPS = 1e-5f;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    InstanceNormCustomTilingData tiling;

    const auto* ishape = context->GetInputShape(0);
    if (ishape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    uint32_t n = static_cast<uint32_t>(ishape->GetOriginShape().GetDim(0));
    uint32_t c = static_cast<uint32_t>(ishape->GetOriginShape().GetDim(1));
    uint32_t h = static_cast<uint32_t>(ishape->GetOriginShape().GetDim(2));
    uint32_t w = static_cast<uint32_t>(ishape->GetOriginShape().GetDim(3));
    uint32_t groups = n * c;
    uint32_t hwSize = h * w;

    uint32_t blockDim = groups < MAX_BLOCKS ? groups : MAX_BLOCKS;
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    uint32_t tileNum = DEFAULT_TILE_NUM;
    if (hwSize < tileNum * BUFFER_NUM) {
        tileNum = 1;
    }
    uint32_t tileSize = CeilDiv(hwSize, tileNum * BUFFER_NUM);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_groups(groups);
    tiling.set_hwSize(hwSize);
    tiling.set_tileNum(tileNum);
    tiling.set_tileSize(tileSize);
    tiling.set_epsilon(DEFAULT_EPS);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // Two arrays in workspace: mean[groups], rstd[groups]
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = static_cast<size_t>(groups) * 2U * sizeof(float);

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class InstanceNormCustom : public OpDef {
public:
    explicit InstanceNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore()
            .SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(InstanceNormCustom);
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor slots for double-buffering

class KernelInstanceNorm {
public:
    __aicore__ inline KernelInstanceNorm() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, uint32_t n, uint32_t c,
                                uint32_t h, uint32_t w, uint32_t groups, uint32_t hwSize,
                                uint32_t tileNum, uint32_t tileSize, float epsilon)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->groups = groups;
        this->hwSize = hwSize;
        this->tileNum = tileNum;
        this->tileLength = tileSize;
        this->epsilon = epsilon;

        // Assign GM buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, static_cast<uint64_t>(groups) * hwSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, static_cast<uint64_t>(groups) * hwSize);

        // Workspace layout: [mean (groups), rstd (groups)]
        meanGm.SetGlobalBuffer((__gm__ float *)workspace, groups);
        rstdGm.SetGlobalBuffer(((__gm__ float *)workspace) + groups, groups);

        // Double buffers for tiles
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));

        // Block split along groups dimension
        uint32_t blkNum = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();
        uint32_t base = groups / blkNum;
        uint32_t rem = groups % blkNum;
        startGroup = blkIdx * base + (blkIdx < rem ? blkIdx : rem);
        groupsThisBlock = base + (blkIdx < rem ? 1U : 0U);
    }

    __aicore__ inline void Process()
    {
        // Phase 1: compute per (N,C) mean and rstd, store to workspace
        for (uint32_t g = 0; g < groupsThisBlock; ++g) {
            uint32_t groupIndex = startGroup + g;
            ComputeStatsForGroup(groupIndex);
        }
        // Phase 2: normalize using computed stats
        for (uint32_t g = 0; g < groupsThisBlock; ++g) {
            uint32_t groupIndex = startGroup + g;
            NormalizeGroup(groupIndex);
        }
    }

private:
    __aicore__ inline void ComputeStatsForGroup(uint32_t groupIndex)
    {
        float sum = 0.f;
        float sumsq = 0.f;

        uint32_t fullTiles = hwSize / tileLength;
        uint32_t tail = hwSize % tileLength;
        uint32_t tileIters = fullTiles + (tail > 0 ? 1U : 0U);

        for (uint32_t t = 0; t < tileIters; ++t) {
            uint32_t curLen = (t < fullTiles) ? tileLength : tail;
            if (curLen == 0) continue;

            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[groupIndex * hwSize + t * tileLength], curLen);
            inQueueX.EnQue(xLocal);

            xLocal = inQueueX.DeQue<DTYPE_X>();
            // Accumulate sum and sum of squares
            for (uint32_t i = 0; i < curLen; ++i) {
                DTYPE_X v = xLocal.GetValue(i);
                float vf = static_cast<float>(v);
                sum += vf;
                sumsq += vf * vf;
            }
            inQueueX.FreeTensor(xLocal);
        }

        float count = static_cast<float>(hwSize);
        float mean = sum / count;
        float var = sumsq / count - mean * mean;
        float rstd = 1.0f / AscendC::Sqrt(var + epsilon);

        // Store to GM workspace
        AscendC::SetValue(meanGm[groupIndex], mean);
        AscendC::SetValue(rstdGm[groupIndex], rstd);
    }

    __aicore__ inline void NormalizeGroup(uint32_t groupIndex)
    {
        // Load stats
        float mean = AscendC::GetValue(meanGm[groupIndex]);
        float rstd = AscendC::GetValue(rstdGm[groupIndex]);

        uint32_t fullTiles = hwSize / tileLength;
        uint32_t tail = hwSize % tileLength;
        uint32_t tileIters = fullTiles + (tail > 0 ? 1U : 0U);

        for (uint32_t t = 0; t < tileIters; ++t) {
            uint32_t curLen = (t < fullTiles) ? tileLength : tail;
            if (curLen == 0) continue;

            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

            AscendC::DataCopy(xLocal, xGm[groupIndex * hwSize + t * tileLength], curLen);

            inQueueX.EnQue(xLocal);
            xLocal = inQueueX.DeQue<DTYPE_X>();

            // y = (x - mean) * rstd
            for (uint32_t i = 0; i < curLen; ++i) {
                DTYPE_X vx = xLocal.GetValue(i);
                float xf = static_cast<float>(vx);
                float yf = (xf - mean) * rstd;
                yLocal.SetValue(i, static_cast<DTYPE_Y>(yf));
            }

            outQueueY.EnQue<DTYPE_Y>(yLocal);

            // Commit to GM
            yLocal = outQueueY.DeQue<DTYPE_Y>();
            AscendC::DataCopy(yGm[groupIndex * hwSize + t * tileLength], yLocal, curLen);

            outQueueY.FreeTensor(yLocal);
            inQueueX.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<float> meanGm;
    AscendC::GlobalTensor<float> rstdGm;

    uint32_t n, c, h, w;
    uint32_t groups;
    uint32_t hwSize;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t startGroup;
    uint32_t groupsThisBlock;
    float epsilon;
};

extern "C" __global__ __aicore__ void instance_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelInstanceNorm op;
    op.Init(x, y, workspace,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.groups, tiling_data.hwSize,
            tiling_data.tileNum, tiling_data.tileSize, tiling_data.epsilon);
    op.Process();
}
"""
