kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using DTYPE = float;

static __aicore__ inline DTYPE tanh_approx(DTYPE x) {
    // Simple rational approximation for tanh
    // Clamp to avoid overflow
    if (x > 5.0f) return 1.0f;
    if (x < -5.0f) return -1.0f;
    DTYPE x2 = x * x;
    DTYPE num = x * (27.0f + x2);
    DTYPE den = 27.0f + 9.0f * x2;
    return num / den;
}

extern "C" __global__ __aicore__ void conv_transpose2d_max_pool_hardtanh_mean_tanh_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N    = tiling_data.n;
    const uint32_t Cin  = tiling_data.cin;
    const uint32_t Cout = tiling_data.cout;
    const uint32_t H    = tiling_data.h;
    const uint32_t W    = tiling_data.w;
    const uint32_t kH   = tiling_data.kH;
    const uint32_t kW   = tiling_data.kW;

    const uint32_t strideH = tiling_data.strideH;
    const uint32_t strideW = tiling_data.strideW;
    const uint32_t padH    = tiling_data.padH;
    const uint32_t padW    = tiling_data.padW;

    const uint32_t poolKH  = tiling_data.poolKH;
    const uint32_t poolKW  = tiling_data.poolKW;
    const uint32_t poolSH  = tiling_data.poolSH;
    const uint32_t poolSW  = tiling_data.poolSW;

    const uint32_t outH    = tiling_data.outH;
    const uint32_t outW    = tiling_data.outW;
    const uint32_t poolOutH = tiling_data.poolOutH;
    const uint32_t poolOutW = tiling_data.poolOutW;

    __gm__ DTYPE* x_gm  = reinterpret_cast<__gm__ DTYPE*>(x);
    __gm__ DTYPE* w_gm  = reinterpret_cast<__gm__ DTYPE*>(weight);
    __gm__ DTYPE* b_gm  = reinterpret_cast<__gm__ DTYPE*>(bias);
    __gm__ DTYPE* y_gm  = reinterpret_cast<__gm__ DTYPE*>(y);
    __gm__ DTYPE* convOut_gm = reinterpret_cast<__gm__ DTYPE*>(workspace);

    // Index helpers
    auto x_idx = [&](uint32_t n, uint32_t ci, uint32_t h, uint32_t w) -> uint64_t {
        return (((uint64_t)n * Cin + ci) * H + h) * W + w;
    };
    auto w_idx = [&](uint32_t ci, uint32_t co, uint32_t kh, uint32_t kw) -> uint64_t {
        return ((((uint64_t)ci * Cout) + co) * kH + kh) * kW + kw;
    };
    auto co_idx = [&](uint32_t n, uint32_t co, uint32_t h, uint32_t w) -> uint64_t {
        return ((((uint64_t)n * Cout) + co) * outH + h) * outW + w;
    };

    // 1) Initialize convOut with bias
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t co = 0; co < Cout; ++co) {
            DTYPE bias_v = b_gm[co];
            for (uint32_t oh = 0; oh < outH; ++oh) {
                for (uint32_t ow = 0; ow < outW; ++ow) {
                    convOut_gm[co_idx(n, co, oh, ow)] = bias_v;
                }
            }
        }
    }

    // 2) ConvTranspose2d accumulate
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t ci = 0; ci < Cin; ++ci) {
            for (uint32_t ih = 0; ih < H; ++ih) {
                for (uint32_t iw = 0; iw < W; ++iw) {
                    DTYPE x_val = x_gm[x_idx(n, ci, ih, iw)];
                    // Scatter to output
                    for (uint32_t co = 0; co < Cout; ++co) {
                        for (uint32_t kh = 0; kh < kH; ++kh) {
                            int64_t oh = (int64_t)ih * (int64_t)strideH - (int64_t)padH + (int64_t)kh;
                            if (oh < 0 || oh >= (int64_t)outH) continue;
                            for (uint32_t kw = 0; kw < kW; ++kw) {
                                int64_t ow = (int64_t)iw * (int64_t)strideW - (int64_t)padW + (int64_t)kw;
                                if (ow < 0 || ow >= (int64_t)outW) continue;
                                DTYPE w_val = w_gm[w_idx(ci, co, kh, kw)];
                                convOut_gm[co_idx(n, co, (uint32_t)oh, (uint32_t)ow)] += x_val * w_val;
                            }
                        }
                    }
                }
            }
        }
    }

    // Constants for HardTanh
    const DTYPE ht_min = -1.0f;
    const DTYPE ht_max = 1.0f;

    // 3) MaxPool2d + HardTanh + Mean over spatial + Tanh
    // Output y has shape [N, Cout, 1, 1]
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t co = 0; co < Cout; ++co) {
            DTYPE accum = 0.0f;
            uint64_t count = (uint64_t)poolOutH * (uint64_t)poolOutW;
            for (uint32_t ph = 0; ph < poolOutH; ++ph) {
                for (uint32_t pw = 0; pw < poolOutW; ++pw) {
                    // Pooling window origin
                    uint32_t h0 = ph * poolSH;
                    uint32_t w0 = pw * poolSW;
                    // Max over pool window
                    DTYPE m = -INFINITY;
                    for (uint32_t kh = 0; kh < poolKH; ++kh) {
                        for (uint32_t kw = 0; kw < poolKW; ++kw) {
                            uint32_t oh = h0 + kh;
                            uint32_t ow = w0 + kw;
                            if (oh < outH && ow < outW) {
                                DTYPE v = convOut_gm[co_idx(n, co, oh, ow)];
                                if (v > m) m = v;
                            }
                        }
                    }
                    // HardTanh
                    if (m < ht_min) m = ht_min;
                    if (m > ht_max) m = ht_max;
                    accum += m;
                }
            }
            DTYPE mean_v = (count > 0) ? (accum / (DTYPE)count) : 0.0f;
            DTYPE out_v = tanh_approx(mean_v);
            // y index for [N, Cout, 1, 1] contiguous
            uint64_t y_index = ((uint64_t)n * Cout + co);
            y_gm[y_index] = out_v;
        }
    }
}
