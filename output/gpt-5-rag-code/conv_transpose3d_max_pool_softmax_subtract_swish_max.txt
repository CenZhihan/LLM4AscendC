kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_B = float;
using DTYPE_Y = float;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelFused {
public:
    __aicore__ inline KernelFused() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR b, GM_ADDR y, uint32_t innerSize, uint32_t channels, uint32_t tileNum)
    {
        this->channels = channels;
        this->innerSize = innerSize;
        this->tileNum = tileNum;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        // split by innerSize across blocks
        this->blockInner = this->innerSize / blocks;
        this->tileLength = this->blockInner / tileNum / BUFFER_NUM;

        // set GM base pointers
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + blockInner * blockIdx, this->blockInner * this->channels);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B *)b + blockInner * blockIdx, this->blockInner * this->channels);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + blockInner * blockIdx, this->blockInner);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            ComputeTile(i);
        }
    }

private:
    __aicore__ inline void ComputeTile(int32_t progress)
    {
        // Allocate local tensors
        AscendC::LocalTensor<DTYPE_Y> bestLocal = outQueueY.AllocTensor<DTYPE_Y>();
        // Initialize best with large negative value
        AscendC::Duplicate(bestLocal, static_cast<DTYPE_Y>(-3.402823466e+38F), this->tileLength);

        for (uint32_t c = 0; c < this->channels; ++c) {
            // Load current channel tile
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
            uint32_t baseOffset = c * this->blockInner + progress * this->tileLength;
            AscendC::DataCopy(xLocal, xGm[baseOffset], this->tileLength);
            AscendC::DataCopy(bLocal, bGm[baseOffset], this->tileLength);
            inQueueX.EnQue(xLocal);
            inQueueB.EnQue(bLocal);

            // Dequeue for compute
            AscendC::LocalTensor<DTYPE_X> xVec = inQueueX.DeQue<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_B> bVec = inQueueB.DeQue<DTYPE_B>();
            AscendC::LocalTensor<DTYPE_Y> tmpLocal = tmpQueue.AllocTensor<DTYPE_Y>();

            // tmp = x - b
            AscendC::Sub(tmpLocal, xVec, bVec, this->tileLength);

            // swish(tmp) = tmp * sigmoid(tmp) with sigmoid = 1 / (1 + exp(-tmp))
            AscendC::LocalTensor<DTYPE_Y> negTmp = tmpQueue.AllocTensor<DTYPE_Y>();
            AscendC::LocalTensor<DTYPE_Y> expNeg = tmpQueue.AllocTensor<DTYPE_Y>();
            AscendC::LocalTensor<DTYPE_Y> denom = tmpQueue.AllocTensor<DTYPE_Y>();
            AscendC::LocalTensor<DTYPE_Y> sigm = tmpQueue.AllocTensor<DTYPE_Y>();
            AscendC::Mul(negTmp, tmpLocal, static_cast<DTYPE_Y>(-1.0f), this->tileLength);
            AscendC::Exp(expNeg, negTmp, this->tileLength);
            AscendC::Add(denom, expNeg, static_cast<DTYPE_Y>(1.0f), this->tileLength);
            AscendC::Reciprocal(sigm, denom, this->tileLength);
            AscendC::Mul(tmpLocal, tmpLocal, sigm, this->tileLength);

            // bestLocal = max(bestLocal, tmpLocal)
            AscendC::Max(bestLocal, bestLocal, tmpLocal, this->tileLength);

            // Free locals
            tmpQueue.FreeTensor(sigm);
            tmpQueue.FreeTensor(denom);
            tmpQueue.FreeTensor(expNeg);
            tmpQueue.FreeTensor(negTmp);
            tmpQueue.FreeTensor(tmpLocal);
            inQueueX.FreeTensor(xVec);
            inQueueB.FreeTensor(bVec);
        }

        // Enqueue and copy out
        outQueueY.EnQue<DTYPE_Y>(bestLocal);
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t innerSize;
    uint32_t channels;
    uint32_t tileNum;
    uint32_t blockInner;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv_transpose3d_max_pool_softmax_subtract_swish_max_custom(GM_ADDR x, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFused op;
    op.Init(x, b, y, tiling_data.innerSize, tiling_data.channels, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused custom kernel for the tail: after ConvTranspose3d and MaxPool3d and Softmax,
    perform channel-wise subtract with swish activation and reduce-max across channels.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, pool_kernel_size, pool_stride, pool_padding):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.max_pool = torch.nn.MaxPool3d(kernel_size=pool_kernel_size, stride=pool_stride, padding=pool_padding)
        self.subtract = torch.nn.Parameter(torch.randn(out_channels))

    def forward(self, x):
        x = self.conv_transpose(x)
        x = self.max_pool(x)
        x = torch.softmax(x, dim=1)
        bias = self.subtract.view(1, -1, 1, 1, 1).expand_as(x).contiguous()
        y = custom_ops_lib.conv_transpose3d_max_pool_softmax_subtract_swish_max_custom(x, bias)
        return y
