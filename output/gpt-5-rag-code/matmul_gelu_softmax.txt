kernel_src="""
#include "kernel_operator.h"
#include <cmath>

using fp32 = float;

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulGeluSoftmax {
public:
    __aicore__ inline KernelMatmulGeluSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t K, uint32_t M, uint32_t blockDim)
    {
        this->N = N;
        this->K = K;
        this->M = M;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // rows per block
        uint32_t rowsPerBlock = (N + blockNum - 1) / blockNum;
        rowStart = blockIdx * rowsPerBlock;
        rowEnd = AscendC::Min(rowStart + rowsPerBlock, N);

        xGm.SetGlobalBuffer((__gm__ fp32*)x, (uint32_t)((uint64_t)N * (uint64_t)K));
        wGm.SetGlobalBuffer((__gm__ fp32*)w, (uint32_t)((uint64_t)M * (uint64_t)K));
        bGm.SetGlobalBuffer((__gm__ fp32*)b, M);
        yGm.SetGlobalBuffer((__gm__ fp32*)y, (uint32_t)((uint64_t)N * (uint64_t)M));

        // Init UB buffers/queues: one row of x (length K), one row of y (length M), and one row of W (length K)
        pipe.InitBuffer(inQueueX, BUFFER_NUM, K * sizeof(fp32));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, M * sizeof(fp32));
        pipe.InitBuffer(inQueueW, BUFFER_NUM, K * sizeof(fp32));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            CopyInX(r);
            ComputeRow(r);
            CopyOutY(r);
        }
    }

private:
    __aicore__ inline void CopyInX(uint32_t row)
    {
        AscendC::LocalTensor<fp32> xLocal = inQueueX.AllocTensor<fp32>();
        AscendC::DataCopy(xLocal, xGm[(uint64_t)row * (uint64_t)K], K);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRow(uint32_t /*row*/)
    {
        AscendC::LocalTensor<fp32> xLocal = inQueueX.DeQue<fp32>();
        AscendC::LocalTensor<fp32> yLocal = outQueueY.AllocTensor<fp32>();

        // Initialize yLocal with bias
        AscendC::DataCopy(yLocal, bGm, M);

        // y = W @ x + b
        for (uint32_t mIdx = 0; mIdx < M; ++mIdx) {
            AscendC::LocalTensor<fp32> wLocal = inQueueW.AllocTensor<fp32>();
            // Load W[mIdx, :]
            AscendC::DataCopy(wLocal, wGm[(uint64_t)mIdx * (uint64_t)K], K);

            // Dot product
            fp32 acc = (fp32)0.0f;
            for (uint32_t kIdx = 0; kIdx < K; ++kIdx) {
                acc += xLocal.GetValue(kIdx) * wLocal.GetValue(kIdx);
            }

            // yLocal[mIdx] = acc + bias[mIdx]
            fp32 base = yLocal.GetValue(mIdx);
            yLocal.SetValue(mIdx, acc + base);

            inQueueW.FreeTensor(wLocal);
        }

        // Apply GELU (tanh approximation) elementwise
        for (uint32_t mIdx = 0; mIdx < M; ++mIdx) {
            fp32 v = yLocal.GetValue(mIdx);
            fp32 v3 = v * v * v;
            fp32 inner = (fp32)0.79788456f * (v + (fp32)0.044715f * v3);
            fp32 gelu = (fp32)0.5f * v * ((fp32)1.0f + tanhf(inner));
            yLocal.SetValue(mIdx, gelu);
        }

        // Softmax along the row
        // 1) Max reduction
        fp32 maxv = -1e30f;
        for (uint32_t mIdx = 0; mIdx < M; ++mIdx) {
            fp32 v = yLocal.GetValue(mIdx);
            if (v > maxv) maxv = v;
        }
        // 2) Exponentiate and sum
        fp32 sumv = 0.0f;
        for (uint32_t mIdx = 0; mIdx < M; ++mIdx) {
            fp32 e = expf(yLocal.GetValue(mIdx) - maxv);
            yLocal.SetValue(mIdx, e);
            sumv += e;
        }
        // 3) Normalize
        fp32 inv = 1.0f / sumv;
        for (uint32_t mIdx = 0; mIdx < M; ++mIdx) {
            yLocal.SetValue(mIdx, yLocal.GetValue(mIdx) * inv);
        }

        outQueueY.EnQue<fp32>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutY(uint32_t row)
    {
        AscendC::LocalTensor<fp32> yLocal = outQueueY.DeQue<fp32>();
        AscendC::DataCopy(yGm[(uint64_t)row * (uint64_t)M], yLocal, M);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueW;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<fp32> xGm;
    AscendC::GlobalTensor<fp32> wGm;
    AscendC::GlobalTensor<fp32> bGm;
    AscendC::GlobalTensor<fp32> yGm;

    uint32_t N;
    uint32_t K;
    uint32_t M;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void matmul_gelu_softmax_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                 GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    if ASCEND_IS_AIC {
        GET_TILING_DATA(tiling_data, tiling);
        KernelMatmulGeluSoftmax op;
        op.Init(x, weight, bias, y,
                tiling_data.n, tiling_data.k, tiling_data.m, tiling_data.blockDim);
        op.Process();
    }
}

    auto M = weight.size(0);

    return result;

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused Matmul + GELU + Softmax using custom AscendC kernel.
    Linear: y = x @ W^T + b, then GELU, then Softmax(dim=1).
    """
    def __init__(self, in_features, out_features):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features)

    def forward(self, x):
        # x: [N, K], weight: [M, K], bias: [M]
        # torch.nn.Linear stores weight as [M, K] already; matmul uses x @ W^T.
        return custom_ops_lib.matmul_gelu_softmax_custom(
            x, self.linear.weight, self.linear.bias
        )
