kernel_src="""
#include "kernel_operator.h"

// Streamed GEMV: C(M,1) = A(M,K) * B(K,1)
// Process rows in blocks, stream along K dimension

constexpr int32_t BUFFER_NUM = 2;

class KernelMatrixVectorMultiplication {
public:
    __aicore__ inline KernelMatrixVectorMultiplication() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t tileK)
    {
        this->M = M;
        this->K = K;
        this->tileK = tileK;

        // Block parallelization across M rows
        this->blockRows = M / AscendC::GetBlockNum();
        this->rowStart = this->blockRows * AscendC::GetBlockIdx();

        aGm.SetGlobalBuffer((__gm__ float*)A + rowStart * K, blockRows * K);
        bGm.SetGlobalBuffer((__gm__ float*)B, K);
        cGm.SetGlobalBuffer((__gm__ float*)C + rowStart, blockRows);

        pipe.InitBuffer(inQueueA, BUFFER_NUM, tileK * sizeof(float));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, tileK * sizeof(float));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, sizeof(float)); // one scalar per row
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, tileK * sizeof(float)); // temp mul buffer
    }

    __aicore__ inline void Process()
    {
        // For each row in this block
        for (uint32_t r = 0; r < blockRows; ++r) {
            // Accumulator in local
            float acc = 0.f;

            uint32_t fullTiles = K / tileK;
            uint32_t tail = K % tileK;

            // Stream full tiles
            for (uint32_t t = 0; t < fullTiles * BUFFER_NUM; ++t) {
                CopyIn(r, t, tileK);
                Compute(acc, tileK);
            }
            // Tail tile if exists
            if (tail) {
                CopyIn(r, fullTiles * BUFFER_NUM, tail);
                Compute(acc, tail);
            }

            // Write out result for row r
            AscendC::LocalTensor<float> cLocal = outQueueC.AllocTensor<float>();
            // Store scalar 'acc' to local tensor
            ((float*)cLocal.GetPtr())[0] = acc;
            outQueueC.EnQue(cLocal);

            AscendC::LocalTensor<float> cOutLocal = outQueueC.DeQue<float>();
            AscendC::DataCopy(cGm[r], cOutLocal, 1);
            outQueueC.FreeTensor(cOutLocal);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t r, uint32_t progress, uint32_t len)
    {
        // progress encodes tile index across K
        uint32_t tileIdx = progress / BUFFER_NUM;
        uint32_t kOffset = tileIdx * tileK;

        AscendC::LocalTensor<float> aLocal = inQueueA.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB.AllocTensor<float>();

        AscendC::DataCopy(aLocal, aGm[r * K + kOffset], len);
        AscendC::DataCopy(bLocal, bGm[kOffset], len);

        inQueueA.EnQue(aLocal);
        inQueueB.EnQue(bLocal);
    }

    __aicore__ inline void Compute(float &acc, uint32_t len)
    {
        AscendC::LocalTensor<float> aLocal = inQueueA.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB.DeQue<float>();
        AscendC::LocalTensor<float> mulLocal = tmpQueue.AllocTensor<float>();

        // Elementwise multiply
        AscendC::Mul(mulLocal, aLocal, bLocal, len);

        // Reduce sum of mulLocal into scalar
        float partial = AscendC::ReduceSum(mulLocal, len);
        acc += partial;

        tmpQueue.EnQue(mulLocal);
        tmpQueue.FreeTensor(mulLocal);
        inQueueA.FreeTensor(aLocal);
        inQueueB.FreeTensor(bLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> cGm;

    uint32_t M;
    uint32_t K;
    uint32_t tileK;
    uint32_t blockRows;
    uint32_t rowStart;
};

extern "C" __global__ __aicore__ void matrix_vector_multiplication_custom(
    GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatrixVectorMultiplication op;
    op.Init(A, B, C, tiling_data.M, tiling_data.K, tiling_data.tileK);
    op.Process();
}
