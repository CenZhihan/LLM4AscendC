kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelHuberLoss {
public:
    __aicore__ inline KernelHuberLoss() {}
    __aicore__ inline void Init(GM_ADDR pred, GM_ADDR target, GM_ADDR loss, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        predGm.SetGlobalBuffer((__gm__ DTYPE_X *)pred + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        targetGm.SetGlobalBuffer((__gm__ DTYPE_Y *)target + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        lossGm.SetGlobalBuffer((__gm__ DTYPE_Z *)loss + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(inQueuePred, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueTarget, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueLoss, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> predLocal = inQueuePred.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> targetLocal = inQueueTarget.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(predLocal, predGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(targetLocal, targetGm[progress * this->tileLength], this->tileLength);
        inQueuePred.EnQue(predLocal);
        inQueueTarget.EnQue(targetLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> predLocal = inQueuePred.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> targetLocal = inQueueTarget.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> lossLocal = outQueueLoss.AllocTensor<DTYPE_Z>();

        // diff = pred - target
        AscendC::Sub(lossLocal, predLocal, targetLocal, this->tileLength); // reuse lossLocal as temp for diff
        // adiff = abs(diff)
        AscendC::Abs(lossLocal, lossLocal, this->tileLength);

        // Prepare temporaries
        AscendC::LocalTensor<DTYPE_Z> diffLocal = outQueueLoss.AllocTensor<DTYPE_Z>();
        AscendC::Sub(diffLocal, predLocal, targetLocal, this->tileLength); // diff again for square

        AscendC::LocalTensor<DTYPE_Z> sqLocal = outQueueLoss.AllocTensor<DTYPE_Z>();
        AscendC::Mul(sqLocal, diffLocal, diffLocal, this->tileLength);     // diff^2
        AscendC::Muls(sqLocal, sqLocal, static_cast<DTYPE_Z>(0.5f), this->tileLength); // 0.5 * diff^2

        AscendC::LocalTensor<DTYPE_Z> largeLocal = outQueueLoss.AllocTensor<DTYPE_Z>();
        AscendC::Sub(largeLocal, lossLocal, static_cast<DTYPE_Z>(0.5f), this->tileLength); // |diff| - 0.5

        // mask: adiff < 1.0
        AscendC::LocalTensor<uint8_t> maskLocal = outQueueLoss.AllocTensor<uint8_t>();
        AscendC::Less(maskLocal, lossLocal, static_cast<DTYPE_Z>(1.0f), this->tileLength);

        // result = select(mask, small, large)
        AscendC::Select(lossLocal, maskLocal, sqLocal, largeLocal, this->tileLength);

        // Enqueue result
        outQueueLoss.EnQue<DTYPE_Z>(lossLocal);

        // Free temps
        outQueueLoss.FreeTensor(largeLocal);
        outQueueLoss.FreeTensor(sqLocal);
        outQueueLoss.FreeTensor(diffLocal);

        inQueuePred.FreeTensor(predLocal);
        inQueueTarget.FreeTensor(targetLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> lossLocal = outQueueLoss.DeQue<DTYPE_Z>();
        AscendC::DataCopy(lossGm[progress * this->tileLength], lossLocal, this->tileLength);
        outQueueLoss.FreeTensor(lossLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePred, inQueueTarget;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueLoss;

    AscendC::GlobalTensor<DTYPE_X> predGm;
    AscendC::GlobalTensor<DTYPE_Y> targetGm;
    AscendC::GlobalTensor<DTYPE_Z> lossGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void huber_loss_custom(GM_ADDR pred, GM_ADDR target, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelHuberLoss op;
    op.Init(pred, target, loss, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
