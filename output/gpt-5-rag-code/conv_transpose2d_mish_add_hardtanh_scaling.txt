kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelFusedPost {
public:
    __aicore__ inline KernelFusedPost() : addScalar(0), scaleScalar(1) {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR add, GM_ADDR scale, GM_ADDR y, uint32_t totalLength, uint32_t tileNum)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        // Scalars: expect 1-element tensors
        addGm.SetGlobalBuffer((__gm__ DTYPE_X *)add, 1);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_X *)scale, 1);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));

        // Read scalar parameters once
        AscendC::LocalTensor<DTYPE_X> tmpLocal = outQueueY.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(tmpLocal, addGm[0], 1);
        addScalar = tmpLocal.GetValue(0);
        AscendC::DataCopy(tmpLocal, scaleGm[0], 1);
        scaleScalar = tmpLocal.GetValue(0);
        outQueueY.FreeTensor(tmpLocal);
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[progress * this->tileLength], this->tileLength);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Mish: x * tanh(ln(1 + exp(x)))
        AscendC::Exp(yLocal, xLocal, this->tileLength);                      // yLocal = exp(x)
        AscendC::Add(yLocal, yLocal, (DTYPE_Z)1.0f, this->tileLength);       // yLocal = exp(x) + 1
        AscendC::Log(yLocal, yLocal, this->tileLength);                      // yLocal = ln(1 + exp(x))
        AscendC::Tanh(yLocal, yLocal, this->tileLength);                     // yLocal = tanh(softplus(x))
        AscendC::Mul(yLocal, yLocal, xLocal, this->tileLength);              // yLocal = x * tanh(softplus(x)) = Mish(x)

        // Add scalar
        AscendC::Add(yLocal, yLocal, (DTYPE_Z)addScalar, this->tileLength);

        // Hardtanh clamp to [-1, 1]
        AscendC::Maximum(yLocal, yLocal, (DTYPE_Z)(-1.0f), this->tileLength);
        AscendC::Minimum(yLocal, yLocal, (DTYPE_Z)(1.0f), this->tileLength);

        // Scale by scalar
        AscendC::Mul(yLocal, yLocal, (DTYPE_Z)scaleScalar, this->tileLength);

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    AscendC::GlobalTensor<DTYPE_X> addGm;
    AscendC::GlobalTensor<DTYPE_X> scaleGm;

    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;

    DTYPE_X addScalar;
    DTYPE_X scaleScalar;
};

extern "C" __global__ __aicore__ void conv_transpose2d_mish_add_hardtanh_scaling_custom(GM_ADDR x, GM_ADDR add, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelFusedPost op;
    op.Init(x, add, scale, y, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model using custom AscendC kernel to apply Mish, add, Hardtanh, and scaling
    after ConvTranspose2d.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, add_value, scale):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride, padding, output_padding)
        self.add_value = add_value
        self.scale = scale

    def forward(self, x):
        x = self.conv_transpose(x)
        add_tensor = torch.tensor([self.add_value], dtype=x.dtype, device=x.device)
        scale_tensor = torch.tensor([self.scale], dtype=x.dtype, device=x.device)
        x = custom_ops_lib.conv_transpose2d_mish_add_hardtanh_scaling_custom(x, add_tensor, scale_tensor)
        return x
