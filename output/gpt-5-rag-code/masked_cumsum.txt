inline uint32_t GetDimSize(const gert::Shape* shape, uint32_t idx, uint32_t defaultVal) {
    if (shape == nullptr) { return defaultVal; }
    if (shape->GetOriginShape().GetDimNum() <= idx) { return defaultVal; }
    return static_cast<uint32_t>(shape->GetOriginShape().GetDim(idx));
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;
const uint32_t TILE_NUM = 8;

// Simple tiling: assume 2D shape [B, L] and cumsum along dim=1
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MaskedCumsumCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    uint32_t dimNum = x_shape ? x_shape->GetOriginShape().GetDimNum() : 0;
    uint32_t batch = 1;
    uint32_t length = 1;
    if (dimNum >= 2) {
        batch = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
        length = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    } else if (dimNum == 1) {
        batch = 1;
        length = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    }
    // Set block dim and tiling params
    context->SetBlockDim(BLOCK_DIM);
    tiling.set_batch(batch);
    tiling.set_length(length);
    tiling.set_tileNum(TILE_NUM);
    // We treat cumsum along last dim by default (dim=1 for 2D)
    tiling.set_dim(dimNum >= 2 ? 1U : 0U);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsum {
public:
    __aicore__ inline KernelMaskedCumsum() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR m, GM_ADDR y,
                                uint32_t batch, uint32_t length, uint32_t tileNum, uint32_t dim)
    {
        this->batch = batch;
        this->length = length;
        this->tileNum = tileNum;
        this->dim = dim;

        // Assume cumsum along last dimension (dim==1 for 2D), process row-by-row
        // Distribute rows across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockId = AscendC::GetBlockIdx();

        // Simple partition: contiguous rows per block
        this->rowStart = (batch / blocks) * blockId;
        this->rowsPerBlock = (blockId == blocks - 1) ? (batch - (batch / blocks) * (blocks - 1)) : (batch / blocks);
        if (rowsPerBlock == 0) { this->rowsPerBlock = 0; return; }

        // Tile length per tile buffer
        this->tileLength = (this->length / (this->tileNum * BUFFER_NUM));
        if (this->tileLength == 0) { this->tileLength = this->length; this->tileNum = 1; }

        // Set GM buffers starting at rowStart
        uint32_t baseOffset = this->rowStart * this->length;
        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + baseOffset, this->rowsPerBlock * this->length);
        mGm.SetGlobalBuffer((__gm__ DTYPE_Y *)m + baseOffset, this->rowsPerBlock * this->length);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + baseOffset, this->rowsPerBlock * this->length);

        // Init queues
        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueM, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (this->rowsPerBlock == 0) { return; }
        // For each row assigned to this block
        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            // Running carry for cumsum across tiles within a row
            DTYPE_Z carry = static_cast<DTYPE_Z>(0);
            int32_t loopCount = this->tileNum * BUFFER_NUM;
            for (int32_t i = 0; i < loopCount; ++i) {
                CopyIn(r, i);
                carry = Compute(r, i, carry);
                CopyOut(r, i);
            }
            // Handle tail elements if length not divisible by tileLength*loopCount
            uint32_t processed = static_cast<uint32_t>(loopCount) * this->tileLength;
            if (processed < this->length) {
                uint32_t tail = this->length - processed;
                TailProcess(r, processed, tail, carry);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row, int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> mLocal = inQueueM.AllocTensor<DTYPE_Y>();
        uint32_t offset = row * this->length + progress * this->tileLength;
        AscendC::DataCopy(xLocal, xGm[offset], this->tileLength);
        AscendC::DataCopy(mLocal, mGm[offset], this->tileLength);
        inQueueX.EnQue(xLocal);
        inQueueM.EnQue(mLocal);
    }

    __aicore__ inline DTYPE_Z Compute(uint32_t row, int32_t progress, DTYPE_Z carryIn)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> mLocal = inQueueM.DeQue<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        // Multiply x by mask into yLocal
        AscendC::Mul(yLocal, xLocal, mLocal, this->tileLength);

        // Inclusive prefix sum on yLocal with carryIn
        DTYPE_Z carry = carryIn;
        for (uint32_t i = 0; i < this->tileLength; ++i) {
            // Read, accumulate, write back
            DTYPE_Z val = static_cast<DTYPE_Z>(yLocal.GetValue(i));
            carry = carry + val;
            yLocal.SetValue(i, carry);
        }

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueX.FreeTensor(xLocal);
        inQueueM.FreeTensor(mLocal);
        return carry;
    }

    __aicore__ inline void CopyOut(uint32_t row, int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        uint32_t offset = row * this->length + progress * this->tileLength;
        AscendC::DataCopy(yGm[offset], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

    __aicore__ inline void TailProcess(uint32_t row, uint32_t start, uint32_t tailLen, DTYPE_Z carryIn)
    {
        // Process remaining elements with a single-buffer path
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> mLocal = inQueueM.AllocTensor<DTYPE_Y>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        uint32_t offset = row * this->length + start;
        AscendC::DataCopy(xLocal, xGm[offset], tailLen);
        AscendC::DataCopy(mLocal, mGm[offset], tailLen);

        AscendC::Mul(yLocal, xLocal, mLocal, tailLen);

        DTYPE_Z carry = carryIn;
        for (uint32_t i = 0; i < tailLen; ++i) {
            DTYPE_Z val = static_cast<DTYPE_Z>(yLocal.GetValue(i));
            carry = carry + val;
            yLocal.SetValue(i, carry);
        }
        AscendC::DataCopy(yGm[offset], yLocal, tailLen);

        inQueueX.FreeTensor(xLocal);
        inQueueM.FreeTensor(mLocal);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueueM;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> mGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t batch;
    uint32_t length;
    uint32_t tileNum;
    uint32_t dim;

    uint32_t tileLength;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsum op;
    op.Init(x, mask, y, tiling_data.batch, tiling_data.length, tiling_data.tileNum, tiling_data.dim);
    op.Process();
}

        super(ModelNew, self).__init__()
