#include "conv_standard_1d_custom_tiling.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ConvStandard1dCustomTilingData tiling;

    const auto* xShape = context->GetInputShape(0);
    const auto* wShape = context->GetInputShape(1);

    uint32_t N = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t Cin = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    uint32_t Lin = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));

    uint32_t Cout = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(0));
    uint32_t K = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(2));

    uint32_t Lout = Lin >= K ? (Lin - K + 1) : 0;

    // Basic tiling strategy: split by Cout across blocks and by length with a fixed tile
    const uint32_t MAX_BLOCK = 8;
    uint32_t blockDim = Cout < MAX_BLOCK ? Cout : MAX_BLOCK;
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    uint32_t coutPerBlock = (Cout + blockDim - 1) / blockDim;

    // Choose a reasonable tile for length dimension
    uint32_t tileT = 2048;
    if (Lout < tileT) tileT = Lout == 0 ? 1 : Lout;

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_lin(Lin);
    tiling.set_lout(Lout);
    tiling.set_ksize(K);
    tiling.set_tileT(tileT);
    tiling.set_coutPerBlock(coutPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

    int64_t Cin = x_shape->GetOriginShape().GetDim(1);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv1D {
public:
    __aicore__ inline KernelConv1D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t lout, uint32_t ksize,
                                uint32_t coutPerBlock, uint32_t tileT)
    {
        this->N = n;
        this->Cin = cin;
        this->Cout = cout;
        this->Lin = lin;
        this->Lout = lout;
        this->K = ksize;
        this->tileT = tileT;

        this->blockStartCout = AscendC::GetBlockIdx() * coutPerBlock;
        this->blockEndCout = this->blockStartCout + coutPerBlock;
        if (this->blockEndCout > this->Cout) this->blockEndCout = this->Cout;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint64_t)N * Cin * Lin);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W *)w, (uint64_t)Cout * Cin * K);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint64_t)N * Cout * Lout);

        // UB buffers sized by tileT
        pipe.InitBuffer(xQueue, BUFFER_NUM, this->tileT * sizeof(DTYPE_X));
        pipe.InitBuffer(yQueue, BUFFER_NUM, this->tileT * sizeof(DTYPE_Y));
        // A tiny buffer for reading a single weight element if needed
        pipe.InitBuffer(wScalarQueue, BUFFER_NUM, sizeof(DTYPE_W));
    }

    __aicore__ inline void Process()
    {
        if (Lout == 0 || blockStartCout >= blockEndCout) {
            return;
        }
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t co = blockStartCout; co < blockEndCout; ++co) {
                for (uint32_t t = 0; t < Lout; t += tileT) {
                    uint32_t validLen = (t + tileT <= Lout) ? tileT : (Lout - t);

                    AscendC::LocalTensor<DTYPE_Y> yLocal = yQueue.AllocTensor<DTYPE_Y>();
                    AscendC::Dup(yLocal, static_cast<DTYPE_Y>(0.0f), validLen);

                    for (uint32_t ci = 0; ci < Cin; ++ci) {
                        for (uint32_t k = 0; k < K; ++k) {
                            // Load input slice x[n, ci, t+k : t+k+validLen]
                            AscendC::LocalTensor<DTYPE_X> xLocal = xQueue.AllocTensor<DTYPE_X>();
                            uint64_t xOffset = ((uint64_t)n * Cin + ci) * Lin + (t + k);
                            AscendC::DataCopy(xLocal, xGm[xOffset], validLen);

                            // Load weight scalar w[co, ci, k] -> wVal
                            AscendC::LocalTensor<DTYPE_W> wLocal = wScalarQueue.AllocTensor<DTYPE_W>();
                            uint64_t wOffset = ((uint64_t)co * Cin + ci) * K + k;
                            AscendC::DataCopy(wLocal, wGm[wOffset], 1);
                            // Multiply-accumulate: yLocal += xLocal * wVal
                            // Use in-place: scale xLocal then add
                            DTYPE_W wVal = static_cast<DTYPE_W>(0);
                            // Assuming API allows get scalar from UB; if not, replace with vector Mul using duplicated scalar
                            // Hypothetical access to the first element:
                            // wVal = wLocal.GetValue(0);
                            AscendC::Muls(xLocal, xLocal, wVal, validLen);
                            AscendC::Add(yLocal, yLocal, xLocal, validLen);

                            xQueue.FreeTensor(xLocal);
                            wScalarQueue.FreeTensor(wLocal);
                        }
                    }

                    uint64_t yOffset = ((uint64_t)n * Cout + co) * Lout + t;
                    AscendC::DataCopy(yGm[yOffset], yLocal, validLen);
                    yQueue.FreeTensor(yLocal);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wScalarQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, Cin, Cout, Lin, Lout, K;
    uint32_t tileT;
    uint32_t blockStartCout, blockEndCout;
};

extern "C" __global__ __aicore__ void conv_standard_1d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv1D op;
    op.Init(
        x, weight, y,
        tiling_data.n, tiling_data.cin, tiling_data.cout,
        tiling_data.lin, tiling_data.lout, tiling_data.ksize,
        tiling_data.coutPerBlock, tiling_data.tileT
    );
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor conv_standard_1d_custom_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    TORCH_CHECK(x.dim() == 3, "x must be 3D (N, C_in, L)");
    TORCH_CHECK(weight.dim() == 3, "weight must be 3D (C_out, C_in, K)");
    TORCH_CHECK(x.size(1) == weight.size(1), "Input channels mismatch between x and weight");
    int64_t N = x.size(0);
    int64_t Cin = x.size(1);
    int64_t Lin = x.size(2);
    int64_t Cout = weight.size(0);
    int64_t K = weight.size(2);
    TORCH_CHECK(Lin >= K, "Kernel size cannot exceed input length without padding in this custom op");
    int64_t Lout = Lin - K + 1;

    auto opts = x.options();
    at::Tensor result = at::empty({N, Cout, Lout}, opts);
    EXEC_NPU_CMD(aclnnConvStandard1dCustom, x, weight, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_standard_1d_custom", &conv_standard_1d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_standard_1d_custom", &conv_standard_1d_custom_impl_npu, "1D standard convolution (custom AscendC)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized 1D convolution using custom AscendC kernel.
    Constraints:
      - stride=1, padding=0, dilation=1, groups=1
      - bias is not supported in custom kernel
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int,
                 stride: int = 1, padding: int = 0, dilation: int = 1,
                 groups: int = 1, bias: bool = False):
        super().__init__()
        assert stride == 1 and padding == 0 and dilation == 1 and groups == 1, \
            "conv_standard_1d_custom currently supports stride=1, padding=0, dilation=1, groups=1"
        assert bias is False, "conv_standard_1d_custom currently does not support bias"
        # Use nn.Conv1d only to hold weights with correct initialization
        self.conv1d = torch.nn.Conv1d(in_channels, out_channels, kernel_size,
                                      stride=stride, padding=padding, dilation=dilation,
                                      groups=groups, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        return custom_ops_lib.conv_standard_1d_custom(x, self.conv1d.weight)
