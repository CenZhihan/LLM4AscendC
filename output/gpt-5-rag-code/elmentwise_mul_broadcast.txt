kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMulBroadcast {
public:
    __aicore__ inline KernelMulBroadcast() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR z, uint32_t n, uint32_t m, uint32_t k, uint32_t kTile)
    {
        this->n = n;
        this->m = m;
        this->k = k;
        this->kTile = kTile;

        // Each block handles one 'i' along N
        uint32_t iIdx = AscendC::GetBlockIdx();
        if (iIdx >= this->n) {
            // No work for this block
            this->active = false;
            return;
        }
        this->active = true;

        // a: (n,1,k), b: (1,m,k), z: (n,m,k)
        aGm.SetGlobalBuffer((__gm__ DTYPE_X *)a + iIdx * this->k, this->k);             // base at a[i,0,0]
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)b, this->m * this->k);                    // base at b[0,0,0]
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z *)z + iIdx * this->m * this->k, this->m * this->k);  // base at z[i,0,0]

        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->kTile * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->kTile * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->kTile * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (!this->active) { return; }
        for (uint32_t j = 0; j < this->m; ++j) {
            for (uint32_t ko = 0; ko < this->k; ko += this->kTile) {
                uint32_t len = (ko + this->kTile <= this->k) ? this->kTile : (this->k - ko);
                // CopyIn
                AscendC::LocalTensor<DTYPE_X> aLocal = inQueueA.AllocTensor<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> bLocal = inQueueB.AllocTensor<DTYPE_Y>();
                AscendC::DataCopy(aLocal, aGm[ko], len);
                AscendC::DataCopy(bLocal, bGm[j * this->k + ko], len);
                inQueueA.EnQue(aLocal);
                inQueueB.EnQue(bLocal);

                // Compute
                AscendC::LocalTensor<DTYPE_X> aL = inQueueA.DeQue<DTYPE_X>();
                AscendC::LocalTensor<DTYPE_Y> bL = inQueueB.DeQue<DTYPE_Y>();
                AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
                AscendC::Mul(zLocal, aL, bL, len);
                outQueueZ.EnQue<DTYPE_Z>(zLocal);
                inQueueA.FreeTensor(aL);
                inQueueB.FreeTensor(bL);

                // CopyOut
                AscendC::LocalTensor<DTYPE_Z> zL = outQueueZ.DeQue<DTYPE_Z>();
                AscendC::DataCopy(zGm[j * this->k + ko], zL, len);
                outQueueZ.FreeTensor(zL);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;
    AscendC::GlobalTensor<DTYPE_X> aGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;

    uint32_t n;
    uint32_t m;
    uint32_t k;
    uint32_t kTile;
    bool active{false};
};

extern "C" __global__ __aicore__ void elmentwise_mul_broadcast_custom(GM_ADDR a, GM_ADDR b, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMulBroadcast op;
    op.Init(a, b, z, tiling_data.n, tiling_data.m, tiling_data.k, tiling_data.kTile);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline at::IntArrayRef _broadcast_out_shape_3d(const at::Tensor& a, const at::Tensor& b, std::vector<int64_t>& tmp) {
    int64_t a0 = (a.dim() > 0) ? a.size(0) : 1;
    int64_t a1 = (a.dim() > 1) ? a.size(1) : 1;
    int64_t a2 = (a.dim() > 2) ? a.size(2) : 1;
    int64_t b0 = (b.dim() > 0) ? b.size(0) : 1;
    int64_t b1 = (b.dim() > 1) ? b.size(1) : 1;
    int64_t b2 = (b.dim() > 2) ? b.size(2) : 1;
    tmp.resize(3);
    tmp[0] = std::max<int64_t>(a0, b0);
    tmp[1] = std::max<int64_t>(a1, b1);
    tmp[2] = std::max<int64_t>(a2, b2);
    return at::IntArrayRef(tmp);
}

at::Tensor elmentwise_mul_broadcast_custom_impl_npu(const at::Tensor& a, const at::Tensor& b) {
    std::vector<int64_t> out_shape_vec;
    auto out_sizes = _broadcast_out_shape_3d(a, b, out_shape_vec);
    at::Tensor result = at::empty(out_sizes, a.options());
    EXEC_NPU_CMD(aclnnElmentwiseMulBroadcastCustom, a, b, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("elmentwise_mul_broadcast_custom", &elmentwise_mul_broadcast_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("elmentwise_mul_broadcast_custom", &elmentwise_mul_broadcast_custom_impl_npu, "broadcast multiply");
}
