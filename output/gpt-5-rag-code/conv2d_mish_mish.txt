kernel_src="""
#include "kernel_operator.h"

__aicore__ inline float mish_once(float x) {
    // mish(x) = x * tanh(softplus(x)) = x * tanh(log(1 + exp(x)))
    float e = AscendC::Exp(x);
    float sp = AscendC::Log(1.0f + e);
    float t = AscendC::Tanh(sp);
    return x * t;
}

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N  = tiling_data.n();
    const uint32_t IC = tiling_data.ic();
    const uint32_t OC = tiling_data.oc();
    const uint32_t IH = tiling_data.ih();
    const uint32_t IW = tiling_data.iw();
    const uint32_t KH = tiling_data.kh();
    const uint32_t KW = tiling_data.kw();
    const uint32_t OH = tiling_data.oh();
    const uint32_t OW = tiling_data.ow();
    const uint32_t stride_h = tiling_data.stride_h();
    const uint32_t stride_w = tiling_data.stride_w();
    const uint32_t pad_h = tiling_data.pad_h();
    const uint32_t pad_w = tiling_data.pad_w();
    const uint32_t dilation_h = tiling_data.dilation_h();
    const uint32_t dilation_w = tiling_data.dilation_w();

    const __gm__ float* x_ptr = (const __gm__ float*)x;
    const __gm__ float* w_ptr = (const __gm__ float*)w;
    const __gm__ float* b_ptr = (const __gm__ float*)b;
    __gm__ float* y_ptr = (__gm__ float*)y;

    // Split OC across blocks
    uint32_t blocks = AscendC::GetBlockNum();
    uint32_t block_id = AscendC::GetBlockIdx();
    uint32_t oc_per_block = OC / blocks + ((OC % blocks) > block_id ? 1 : 0);
    uint32_t oc_start = (OC / blocks) * block_id + AscendC::Min<uint32_t>(OC % blocks, block_id);
    uint32_t oc_end = oc_start + oc_per_block;
    if (oc_end > OC) oc_end = OC;

    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t oc = oc_start; oc < oc_end; ++oc) {
            for (uint32_t oh = 0; oh < OH; ++oh) {
                for (uint32_t ow = 0; ow < OW; ++ow) {
                    float acc = b_ptr[oc];
                    for (uint32_t ic = 0; ic < IC; ++ic) {
                        for (uint32_t kh = 0; kh < KH; ++kh) {
                            int32_t in_h = static_cast<int32_t>(oh * stride_h + kh * dilation_h - pad_h);
                            if (in_h < 0 || in_h >= static_cast<int32_t>(IH)) continue;
                            for (uint32_t kw = 0; kw < KW; ++kw) {
                                int32_t in_w = static_cast<int32_t>(ow * stride_w + kw * dilation_w - pad_w);
                                if (in_w < 0 || in_w >= static_cast<int32_t>(IW)) continue;

                                uint64_t x_idx = (((uint64_t)n * IC + ic) * IH + (uint64_t)in_h) * IW + (uint64_t)in_w;
                                uint64_t w_idx = (((uint64_t)oc * IC + ic) * KH + kh) * KW + kw;

                                acc += x_ptr[x_idx] * w_ptr[w_idx];
                            }
                        }
                    }
                    // Apply Mish twice
                    float y0 = mish_once(acc);
                    float y1 = mish_once(y0);

                    uint64_t y_idx = (((uint64_t)n * OC + oc) * OH + oh) * OW + ow;
                    y_ptr[y_idx] = y1;
                }
            }
        }
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline at::Tensor make_output_tensor_like_conv(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, C_in, H, W], w: [C_out, C_in, KH, KW]
    auto sizes = x.sizes();
    int64_t N = sizes[0];
    int64_t H = sizes[2];
    int64_t W = sizes[3];
    int64_t KH = w.size(2);
    int64_t KW = w.size(3);
    // Defaults: stride=1, padding=0, dilation=1
    int64_t OH = (H - KH) + 1;
    int64_t OW = (W - KW) + 1;
    int64_t OC = w.size(0);
    auto out_sizes = std::vector<int64_t>{N, OC, OH, OW};
    return at::empty(out_sizes, x.options());
}

at::Tensor conv2d_mish_mish_custom_impl_npu(const at::Tensor& x, const at::Tensor& w, const at::Tensor& b) {
    at::Tensor y = make_output_tensor_like_conv(x, w);
    EXEC_NPU_CMD(aclnnConv2dMishMishCustom, x, w, b, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_mish_mish_custom", &conv2d_mish_mish_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_mish_mish_custom", &conv2d_mish_mish_custom_impl_npu, "conv2d + mish + mish fused");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Model using fused custom AscendC kernel: conv2d followed by two Mish activations.
    """
    def __init__(self, in_channels, out_channels, kernel_size):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size, bias=True)

    def forward(self, x):
        return custom_ops_lib.conv2d_mish_mish_custom(x, self.conv.weight, self.conv.bias)
