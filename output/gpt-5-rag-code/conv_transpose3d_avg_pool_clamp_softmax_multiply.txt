kernel_src="""
#include "kernel_operator.h"

typedef float DTYPE;

struct CT3DAvgPoolSoftmaxMulTile {
    uint32_t n, cin, cout;
    uint32_t din, hin, win;
    uint32_t kd, kh, kw;
    uint32_t sd, sh, sw;
    uint32_t pd, ph, pw;
    uint32_t opd, oph, opw;
    uint32_t poolk;
    uint32_t dpool, hpool, wpool;
    uint32_t dout, hout, wout;
    uint32_t groupSize;
    uint32_t groupNum;
    uint32_t tileNum;
    uint32_t blockDim;
    float clampMin;
    float clampMax;
};

__aicore__ inline uint64_t offset_ncdhw(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                        uint32_t N, uint32_t C, uint32_t D, uint32_t H, uint32_t W)
{
    return (((uint64_t)n * C + c) * D + d) * H * W + (uint64_t)h * W + w;
}

extern "C" __global__ __aicore__ void conv_transpose3d_avg_pool_clamp_softmax_multiply_custom(GM_ADDR x,
                                                                                               GM_ADDR weight,
                                                                                               GM_ADDR bias,
                                                                                               GM_ADDR scale,
                                                                                               GM_ADDR y,
                                                                                               GM_ADDR workspace,
                                                                                               GM_ADDR tiling) {
    (void)workspace;
    GET_TILING_DATA(tiling_data_buf, tiling);
    CT3DAvgPoolSoftmaxMulTile t;
    // Load tiling data
    t.n        = tiling_data_buf.n;
    t.cin      = tiling_data_buf.cin;
    t.cout     = tiling_data_buf.cout;
    t.din      = tiling_data_buf.din;
    t.hin      = tiling_data_buf.hin;
    t.win      = tiling_data_buf.win;
    t.kd       = tiling_data_buf.kd;
    t.kh       = tiling_data_buf.kh;
    t.kw       = tiling_data_buf.kw;
    t.sd       = tiling_data_buf.sd;
    t.sh       = tiling_data_buf.sh;
    t.sw       = tiling_data_buf.sw;
    t.pd       = tiling_data_buf.pd;
    t.ph       = tiling_data_buf.ph;
    t.pw       = tiling_data_buf.pw;
    t.opd      = tiling_data_buf.opd;
    t.oph      = tiling_data_buf.oph;
    t.opw      = tiling_data_buf.opw;
    t.poolk    = tiling_data_buf.poolk;
    t.dpool    = tiling_data_buf.dpool;
    t.hpool    = tiling_data_buf.hpool;
    t.wpool    = tiling_data_buf.wpool;
    t.dout     = tiling_data_buf.dout;
    t.hout     = tiling_data_buf.hout;
    t.wout     = tiling_data_buf.wout;
    t.groupSize= tiling_data_buf.groupSize;
    t.groupNum = tiling_data_buf.groupNum;
    t.tileNum  = tiling_data_buf.tileNum;
    t.blockDim = tiling_data_buf.blockDim;
    t.clampMin = tiling_data_buf.clampMin;
    t.clampMax = tiling_data_buf.clampMax;

    // Global memory pointers
    __gm__ DTYPE* xGm  = (__gm__ DTYPE*)x;
    __gm__ DTYPE* wGm  = (__gm__ DTYPE*)weight;  // [Cin, Cout, Kd, Kh, Kw]
    __gm__ DTYPE* bGm  = (__gm__ DTYPE*)bias;    // [Cout]
    __gm__ DTYPE* sGm  = (__gm__ DTYPE*)scale;   // [1, Cout, 1, 1, 1] or [Cout]
    __gm__ DTYPE* yGm  = (__gm__ DTYPE*)y;

    // Determine block assignment
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t blocks   = AscendC::GetBlockNum();
    uint32_t groupsPerBlock = (t.groupNum + blocks - 1) / blocks;
    uint32_t startGroup = blockIdx * groupsPerBlock;
    uint32_t endGroup   = AscendC::Min((uint32_t)t.groupNum, startGroup + groupsPerBlock);

    // For each (n, oc) group assigned to this block:
    for (uint32_t g = startGroup; g < endGroup; ++g) {
        uint32_t nIdx = g / t.cout;
        uint32_t oc   = g % t.cout;

        // Local buffer for one output group (dout*hout*wout)
        // Allocate in vector out queue style
        AscendC::TPipe pipe;
        AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue;
        pipe.InitBuffer(outQueue, 1, t.groupSize * sizeof(DTYPE));
        AscendC::LocalTensor<DTYPE> yLocal = outQueue.AllocTensor<DTYPE>();
        // Initialize local buffer to bias
        for (uint32_t i = 0; i < t.groupSize; ++i) {
            yLocal.SetValue(i, bGm[oc]);
        }

        // ConvTranspose3d accumulate: iterate pooled input positions and scatter-add to outputs
        for (uint32_t ic = 0; ic < t.cin; ++ic) {
            for (uint32_t dp = 0; dp < t.dpool; ++dp) {
                for (uint32_t hp = 0; hp < t.hpool; ++hp) {
                    for (uint32_t wp = 0; wp < t.wpool; ++wp) {
                        // AvgPool3d value for pooled position (ic, dp, hp, wp)
                        // Pool kernel stride==poolk, kernel==poolk, no padding
                        DTYPE avgVal = 0;
                        uint32_t count = 0;
                        for (uint32_t kd0 = 0; kd0 < t.poolk; ++kd0) {
                            uint32_t dsrc = dp * t.poolk + kd0;
                            if (dsrc >= t.din) continue;
                            for (uint32_t kh0 = 0; kh0 < t.poolk; ++kh0) {
                                uint32_t hsrc = hp * t.poolk + kh0;
                                if (hsrc >= t.hin) continue;
                                for (uint32_t kw0 = 0; kw0 < t.poolk; ++kw0) {
                                    uint32_t wsrc = wp * t.poolk + kw0;
                                    if (wsrc >= t.win) continue;
                                    uint64_t xOff = offset_ncdhw(nIdx, ic, dsrc, hsrc, wsrc, t.n, t.cin, t.din, t.hin, t.win);
                                    avgVal += xGm[xOff];
                                    ++count;
                                }
                            }
                        }
                        if (count > 0) avgVal /= (DTYPE)count;

                        // Base output coordinate for this pooled position (before kernel offset)
                        int32_t baseOd = (int32_t)dp * (int32_t)t.sd - (int32_t)t.pd;
                        int32_t baseOh = (int32_t)hp * (int32_t)t.sh - (int32_t)t.ph;
                        int32_t baseOw = (int32_t)wp * (int32_t)t.sw - (int32_t)t.pw;

                        // Scatter to output positions through kernel window
                        for (uint32_t kd1 = 0; kd1 < t.kd; ++kd1) {
                            int32_t od = baseOd + (int32_t)kd1 + (int32_t)t.opd;
                            if (od < 0 || od >= (int32_t)t.dout) continue;
                            for (uint32_t kh1 = 0; kh1 < t.kh; ++kh1) {
                                int32_t oh = baseOh + (int32_t)kh1 + (int32_t)t.oph;
                                if (oh < 0 || oh >= (int32_t)t.hout) continue;
                                for (uint32_t kw1 = 0; kw1 < t.kw; ++kw1) {
                                    int32_t ow = baseOw + (int32_t)kw1 + (int32_t)t.opw;
                                    if (ow < 0 || ow >= (int32_t)t.wout) continue;
                                    // Weight index [ic, oc, kd1, kh1, kw1]
                                    uint64_t wOff = (((uint64_t)ic * t.cout + oc) * t.kd + kd1) * t.kh * t.kw
                                                    + (uint64_t)kh1 * t.kw + kw1;
                                    DTYPE wval = wGm[wOff];
                                    // Output local index
                                    uint32_t outIdx = ((uint32_t)od * t.hout + (uint32_t)oh) * t.wout + (uint32_t)ow;
                                    DTYPE acc = yLocal.GetValue(outIdx);
                                    acc += avgVal * wval;
                                    yLocal.SetValue(outIdx, acc);
                                }
                            }
                        }
                    }
                }
            }
        }

        // Clamp
        for (uint32_t i = 0; i < t.groupSize; ++i) {
            DTYPE v = yLocal.GetValue(i);
            if (v < t.clampMin) v = t.clampMin;
            if (v > t.clampMax) v = t.clampMax;
            yLocal.SetValue(i, v);
        }

        // Softmax over spatial for this (n, oc) group
        DTYPE maxVal = (DTYPE)-1.0e30f;
        for (uint32_t i = 0; i < t.groupSize; ++i) {
            DTYPE v = yLocal.GetValue(i);
            if (v > maxVal) maxVal = v;
        }
        DTYPE sumExp = (DTYPE)0;
        for (uint32_t i = 0; i < t.groupSize; ++i) {
            DTYPE v = yLocal.GetValue(i);
            DTYPE ev = AscendC::Exp(v - maxVal);
            sumExp += ev;
            yLocal.SetValue(i, ev);
        }
        DTYPE invSum = (sumExp > (DTYPE)0) ? ((DTYPE)1 / sumExp) : (DTYPE)0;
        // Scale per channel (broadcast)
        DTYPE scaleVal = sGm[oc];
        for (uint32_t i = 0; i < t.groupSize; ++i) {
            DTYPE v = yLocal.GetValue(i) * invSum;
            v *= scaleVal;
            yLocal.SetValue(i, v);
        }

        // Write back to global output y: contiguous group
        // Compute group base offset
        uint64_t yGroupBase = ((uint64_t)nIdx * t.cout + oc) * t.dout * t.hout * t.wout;
        // Copy local to GM
        for (uint32_t i = 0; i < t.groupSize; ++i) {
            yGm[yGroupBase + i] = yLocal.GetValue(i);
        }

        outQueue.EnQue<DTYPE>(yLocal);
        AscendC::LocalTensor<DTYPE> outLocal2 = outQueue.DeQue<DTYPE>();
        outQueue.FreeTensor(outLocal2);
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor allocate_output_like_transpose3d(
    const at::Tensor& x,
    const at::Tensor& weight,
    int64_t kernel_size,
    int64_t stride,
    int64_t padding,
    int64_t output_padding,
    int64_t pool_kernel_size)
{
    // x: [N, Cin, Din, Hin, Win], weight: [Cin, Cout, Kd, Kh, Kw]
    auto sizes = x.sizes();
    int64_t N = sizes[0];
    int64_t Din = sizes[2];
    int64_t Hin = sizes[3];
    int64_t Win = sizes[4];

    int64_t Cin = weight.size(0);
    int64_t Cout = weight.size(1);
    (void)Cin;

    int64_t Kd = kernel_size;
    int64_t Kh = kernel_size;
    int64_t Kw = kernel_size;

    int64_t sd = stride, sh = stride, sw = stride;
    int64_t pd = padding, ph = padding, pw = padding;
    int64_t opd = output_padding, oph = output_padding, opw = output_padding;

    int64_t poolk = pool_kernel_size;
    int64_t Dpool = Din / poolk;
    int64_t Hpool = Hin / poolk;
    int64_t Wpool = Win / poolk;

    int64_t Dout = (Dpool > 0 ? ((Dpool - 1) * sd - 2 * pd + Kd + opd) : 0);
    int64_t Hout = (Hpool > 0 ? ((Hpool - 1) * sh - 2 * ph + Kh + oph) : 0);
    int64_t Wout = (Wpool > 0 ? ((Wpool - 1) * sw - 2 * pw + Kw + opw) : 0);

    return at::empty({N, Cout, Dout, Hout, Wout}, x.options());
}
