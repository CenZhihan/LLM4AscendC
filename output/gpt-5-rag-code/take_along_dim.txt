kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_Z = float;
using DTYPE_INDICES = int32_t;

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelTakeAlongDim {
public:
    __aicore__ inline KernelTakeAlongDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, GM_ADDR y,
                                uint32_t totalRows, uint32_t lastDim, uint32_t indicesLen,
                                uint32_t tileNum)
    {
        this->rows = totalRows;
        this->lastDim = lastDim;
        this->indicesLen = indicesLen;
        this->tileNum = tileNum;

        // Partition by flattened output length (rows * indicesLen)
        this->blockOutLen = (rows * indicesLen) / AscendC::GetBlockNum();
        this->baseOutStart = this->blockOutLen * AscendC::GetBlockIdx();
        this->baseRowStart = this->baseOutStart / indicesLen;

        // Set GM buffers
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + baseRowStart * lastDim, (rows - baseRowStart) * lastDim);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_INDICES*)idx + baseOutStart, blockOutLen);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + baseOutStart, blockOutLen);

        // Tile length
        this->tileLength = this->blockOutLen / tileNum / BUFFER_NUM;

        pipe.InitBuffer(inQueueIdx,  BUFFER_NUM, this->tileLength * sizeof(DTYPE_INDICES));
        pipe.InitBuffer(outQueueY,   BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_INDICES> idxLocal = inQueueIdx.AllocTensor<DTYPE_INDICES>();
        AscendC::DataCopy(idxLocal, idxGm[progress * this->tileLength], this->tileLength);
        inQueueIdx.EnQue(idxLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_INDICES> idxLocal = inQueueIdx.DeQue<DTYPE_INDICES>();
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();

        uint32_t base = progress * this->tileLength;
        for (uint32_t j = 0; j < this->tileLength; ++j) {
            uint32_t globalOutIdx = this->baseOutStart + base + j;
            uint32_t row = globalOutIdx / indicesLen;
            // uint32_t colPos = globalOutIdx % indicesLen; // unused in computation directly
            DTYPE_INDICES srcCol = idxLocal.GetValue(j);

            uint32_t srcOffsetInX = (row - baseRowStart) * lastDim + static_cast<uint32_t>(srcCol);
            DTYPE_X xVal = xGm.GetValue(srcOffsetInX);
            yLocal.SetValue(j, static_cast<DTYPE_Z>(xVal));
        }

        outQueueY.EnQue<DTYPE_Z>(yLocal);
        inQueueIdx.FreeTensor(idxLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[progress * this->tileLength], yLocal, this->tileLength);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueIdx;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_INDICES> idxGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t rows;
    uint32_t lastDim;
    uint32_t indicesLen;

    uint32_t tileNum;
    uint32_t tileLength;

    uint32_t blockOutLen;
    uint32_t baseOutStart;
    uint32_t baseRowStart;
};

extern "C" __global__ __aicore__ void take_along_dim_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTakeAlongDim op;
    op.Init(x, idx, y,
            tiling_data.totalRows, tiling_data.lastDim, tiling_data.indicesLen,
            tiling_data.tileNum);
    op.Process();
}
