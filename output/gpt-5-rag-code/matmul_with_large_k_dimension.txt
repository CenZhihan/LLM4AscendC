namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MatmulWithLargeKDimensionCustomTilingData tiling;

    const auto *shapeA = context->GetInputShape(0);
    const auto *shapeB = context->GetInputShape(1);

    // Expect A: (M, K), B: (K, N)
    if (shapeA->GetDimNum() != 2 || shapeB->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    uint64_t M = shapeA->GetDim(0);
    uint64_t K = shapeA->GetDim(1);
    uint64_t Kb = shapeB->GetDim(0);
    uint64_t N = shapeB->GetDim(1);
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    // Choose tiles tuned for very large K to reduce GM access
    // tileK must be a multiple of 256 for better vectorization
    const uint32_t preferredTileK = 4096; // adjust on need
    const uint32_t preferredTileN = 32;   // vector-friendly width

    uint32_t tileK = (K >= preferredTileK) ? preferredTileK : static_cast<uint32_t>(K);
    // Make tileK a multiple of 256 where possible
    tileK = (tileK / 256) * 256;
    if (tileK == 0) tileK = static_cast<uint32_t>(K);

    uint32_t tileN = (N >= preferredTileN) ? preferredTileN : static_cast<uint32_t>(N);

    // BlockDim: map aicore blocks to rows with stride
    uint32_t maxBlocks = 32; // typical aicore count on 910B
    uint32_t blockDim = (M < maxBlocks) ? static_cast<uint32_t>(M) : maxBlocks;
    context->SetBlockDim(blockDim);

    tiling.set_M(static_cast<uint32_t>(M));
    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_K(static_cast<uint32_t>(K));
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0; // no extra workspace
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t AICORE_MAX_BLOCKS = 32;

class KernelMatmulLargeK {
public:
    __aicore__ inline KernelMatmulLargeK() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C, uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileK, uint32_t tileN)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileK = tileK;
        this->tileN = tileN;

        aGm.SetGlobalBuffer((__gm__ float *)A, (uint64_t)M * (uint64_t)K);
        bGm.SetGlobalBuffer((__gm__ float *)B, (uint64_t)K * (uint64_t)N);
        cGm.SetGlobalBuffer((__gm__ float *)C, (uint64_t)M * (uint64_t)N);

        // Buffers: A chunk [tileK], B chunk [tileK * tileN], C accum [tileN]
        pipe.InitBuffer(inA, 1, this->tileK * sizeof(float));
        pipe.InitBuffer(inB, 1, (uint32_t)this->tileK * this->tileN * sizeof(float));
        pipe.InitBuffer(outC, 1, this->tileN * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        if (blockNum == 0) blockNum = 1;
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Row stripe per block
        for (uint32_t row = blockIdx; row < M; row += blockNum) {
            // iterate N by tileN
            for (uint32_t nBase = 0; nBase < N; nBase += tileN) {
                uint32_t nLen = AscendC::Min(tileN, N - nBase);

                // allocate accum buffer and zero-init
                AscendC::LocalTensor<float> cAcc = outC.AllocTensor<float>();
                ZeroVec(cAcc, nLen);

                // iterate K by tileK
                for (uint32_t kBase = 0; kBase < K; kBase += tileK) {
                    uint32_t kLen = AscendC::Min(tileK, K - kBase);

                    // Copy A[row, kBase:kBase+kLen] contiguous
                    AscendC::LocalTensor<float> aLoc = inA.AllocTensor<float>();
                    AscendC::DataCopy(aLoc, aGm[(uint64_t)row * K + kBase], kLen);
                    inA.EnQue(aLoc);

                    // Copy B[kBase:kBase+kLen, nBase:nBase+nLen] contiguous as one block
                    AscendC::LocalTensor<float> bLoc = inB.AllocTensor<float>();
                    // B is row-major [K, N], contiguous slice starting at (kBase, nBase) size kLen*nLen
                    AscendC::DataCopy(bLoc, bGm[(uint64_t)kBase * N + nBase], (uint32_t)kLen * nLen);
                    inB.EnQue(bLoc);

                    // Deque for compute
                    aLoc = inA.DeQue<float>();
                    bLoc = inB.DeQue<float>();

                    // Compute: cAcc[n] += sum_{kk=0..kLen-1} aLoc[kk] * bLoc[kk, n]
                    FmaOuterProduct(cAcc, aLoc, bLoc, kLen, nLen);

                    // Free aLoc and bLoc (no need to keep)
                    inA.FreeTensor(aLoc);
                    inB.FreeTensor(bLoc);
                }

                // Write back cAcc -> C[row, nBase:nBase+nLen]
                outC.EnQue<float>(cAcc);
                cAcc = outC.DeQue<float>();
                AscendC::DataCopy(cGm[(uint64_t)row * N + nBase], cAcc, nLen);
                outC.FreeTensor(cAcc);
            }
        }
    }

private:
    __aicore__ inline void ZeroVec(AscendC::LocalTensor<float>& v, uint32_t len)
    {
        // Simple zero-initialization
        // Use vector set if available; fall back to loop
        for (uint32_t i = 0; i < len; ++i) {
            v.SetValue(i, 0.0f);
        }
    }

    __aicore__ inline void FmaOuterProduct(AscendC::LocalTensor<float>& cAcc,
                                           AscendC::LocalTensor<float>& aLoc,
                                           AscendC::LocalTensor<float>& bLoc,
                                           uint32_t kLen, uint32_t nLen)
    {
        // Accumulate outer-product contributions into cAcc
        for (uint32_t kk = 0; kk < kLen; ++kk) {
            float aVal = aLoc.GetValue(kk);
            uint32_t bRowOffset = kk * nLen;
            for (uint32_t nn = 0; nn < nLen; ++nn) {
                float prev = cAcc.GetValue(nn);
                float bVal = bLoc.GetValue(bRowOffset + nn);
                cAcc.SetValue(nn, prev + aVal * bVal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inA, inB;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outC;

    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> bGm;
    AscendC::GlobalTensor<float> cGm;

    uint32_t M, N, K;
    uint32_t tileK, tileN;
};

extern "C" __global__ __aicore__ void matmul_with_large_k_dimension_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(t, tiling);
    KernelMatmulLargeK op;
    op.Init(A, B, C, t.M, t.N, t.K, t.tileK, t.tileN);
    op.Process();
}
