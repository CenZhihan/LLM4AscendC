static ge::graphStatus TilingFunc(gert::TilingContext* context)

    GemmReluDivideCustomTilingData tiling;

    const auto* x_shape = context->GetInputShape(0);
    // weight_t: [K, N]
    const auto* wt_shape = context->GetInputShape(1);

    uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));

    uint32_t N = static_cast<uint32_t>(wt_shape->GetOriginShape().GetDim(1));
    if (K1 != K) {
        return ge::GRAPH_FAILED;

    const uint32_t BLOCK_DIM = 16;
    context->SetBlockDim(BLOCK_DIM);

    // Simple tile strategy
    uint32_t tileM = 64;
    uint32_t tileN = 64;
    uint32_t tileK = 64;

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

class KernelGemmReluDivide {
public:
    __aicore__ inline KernelGemmReluDivide() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight_t, GM_ADDR bias, GM_ADDR div, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        this->M_ = M;
        this->N_ = N;
        this->K_ = K;
        this->tileM_ = tileM;
        this->tileN_ = tileN;
        this->tileK_ = tileK;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blk = AscendC::GetBlockIdx();
        this->mPerBlock_ = (M + blocks - 1) / blocks;
        this->mStart_ = blk * this->mPerBlock_;
        if (this->mStart_ > M) this->mStart_ = M;
        this->mEnd_ = this->mStart_ + this->mPerBlock_;
        if (this->mEnd_ > M) this->mEnd_ = M;

        xGm_.SetGlobalBuffer((__gm__ float*)x + this->mStart_ * K, (this->mEnd_ - this->mStart_) * K);
        wGm_.SetGlobalBuffer((__gm__ float*)weight_t, K * N);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, N);
        yGm_.SetGlobalBuffer((__gm__ float*)y + this->mStart_ * N, (this->mEnd_ - this->mStart_) * N);
        divGm_.SetGlobalBuffer((__gm__ float*)div, 1);

        // Buffers
        pipe_.InitBuffer(inQueueA_, 1, tileK_ * sizeof(float));                         // a row chunk
        pipe_.InitBuffer(inQueueB_, 1, tileK_ * tileN_ * sizeof(float));               // b tile
        pipe_.InitBuffer(inQueueBias_, 1, tileN_ * sizeof(float));                     // bias slice
        pipe_.InitBuffer(inQueueDiv_, 1, sizeof(float));                               // divisor scalar
        pipe_.InitBuffer(outQueueC_, 1, tileN_ * sizeof(float));                       // c row tile
    }

    __aicore__ inline void Process()
    {
        // Load divisor once
        float divisor = 1.0f;
        {
            AscendC::LocalTensor<float> divLocal = inQueueDiv_.AllocTensor<float>();
            AscendC::DataCopy(divLocal, divGm_[0], 1);
            inQueueDiv_.EnQue(divLocal);
            divLocal = inQueueDiv_.DeQue<float>();
            divisor = divLocal.GetValue(0);
            inQueueDiv_.FreeTensor(divLocal);
        }
        float invDiv = 1.0f / divisor;

        for (uint32_t mi = this->mStart_; mi < this->mEnd_; ) {
            uint32_t curM = AscendC::min(this->tileM_, this->mEnd_ - mi);
            for (uint32_t i = 0; i < curM; ++i) {
                // Process N dimension in tiles
                for (uint32_t nj = 0; nj < N_; nj += tileN_) {
                    uint32_t curN = AscendC::min(tileN_, N_ - nj);

                    // Initialize C row tile to zero
                    AscendC::LocalTensor<float> cTile = outQueueC_.AllocTensor<float>();
                    for (uint32_t j = 0; j < curN; ++j) {
                        cTile.SetValue(j, 0.0f);
                    }

                    // K loop
                    for (uint32_t kk = 0; kk < K_; kk += tileK_) {
                        uint32_t curK = AscendC::min(tileK_, K_ - kk);

                        // Load a row chunk [curK]
                        AscendC::LocalTensor<float> aChunk = inQueueA_.AllocTensor<float>();
                        AscendC::DataCopy(aChunk, xGm_[(mi - mStart_) * K_ + i * K_ + kk], curK);
                        inQueueA_.EnQue(aChunk);

                        // Load B tile [curK x curN] laid out as contiguous rows in GM: B is [K x N]
                        AscendC::LocalTensor<float> bTile = inQueueB_.AllocTensor<float>();
                        for (uint32_t rk = 0; rk < curK; ++rk) {
                            AscendC::DataCopy(bTile[rk * curN], wGm_[(kk + rk) * N_ + nj], curN);
                        }
                        inQueueB_.EnQue(bTile);

                        // Compute partial c += aChunk * bTile
                        aChunk = inQueueA_.DeQue<float>();
                        bTile = inQueueB_.DeQue<float>();
                        for (uint32_t rk = 0; rk < curK; ++rk) {
                            float aval = aChunk.GetValue(rk);
                            for (uint32_t j = 0; j < curN; ++j) {
                                float prev = cTile.GetValue(j);
                                float bv = bTile.GetValue(rk * curN + j);
                                cTile.SetValue(j, prev + aval * bv);
                            }
                        }
                        inQueueA_.FreeTensor(aChunk);
                        inQueueB_.FreeTensor(bTile);
                    }

                    // Add bias slice
                    AscendC::LocalTensor<float> biasSlice = inQueueBias_.AllocTensor<float>();
                    AscendC::DataCopy(biasSlice, bGm_[nj], curN);
                    inQueueBias_.EnQue(biasSlice);
                    biasSlice = inQueueBias_.DeQue<float>();
                    for (uint32_t j = 0; j < curN; ++j) {
                        float v = cTile.GetValue(j) + biasSlice.GetValue(j);
                        // ReLU
                        if (v < 0.0f) v = 0.0f;
                        // divide
                        v = v * invDiv;
                        cTile.SetValue(j, v);
                    }
                    inQueueBias_.FreeTensor(biasSlice);

                    // Write back C row tile
                    AscendC::DataCopy(yGm_[(mi - mStart_) * N_ + i * N_ + nj], cTile, curN);
                    outQueueC_.FreeTensor(cTile);
                }
            }
            mi += curM;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueBias_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueDiv_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueC_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> divGm_;

    uint32_t M_, N_, K_;
    uint32_t tileM_, tileN_, tileK_;
    uint32_t mPerBlock_;
    uint32_t mStart_;
    uint32_t mEnd_;
};

extern "C" __global__ __aicore__ void gemm_relu_divide_custom(GM_ADDR x, GM_ADDR weight_t, GM_ADDR bias, GM_ADDR div, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmReluDivide op;
    op.Init(x, weight_t, bias, div, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused GEMM + ReLU + divide using custom AscendC kernel.
    """
    def __init__(self, in_features, out_features, divisor):
        super(ModelNew, self).__init__()
        self.linear = torch.nn.Linear(in_features, out_features)
        self.divisor = divisor

    def forward(self, x):
        # weight_t should be [K, N]
        wt = self.linear.weight.t()
        b = self.linear.bias
        div = torch.tensor([self.divisor], dtype=x.dtype, device=x.device)
        return custom_ops_lib.gemm_relu_divide_custom(x, wt, b, div)
