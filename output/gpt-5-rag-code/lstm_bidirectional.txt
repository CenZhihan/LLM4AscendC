kernel_src="""
#include "kernel_operator.h"

class KernelLstmBidir {
public:
    __aicore__ inline KernelLstmBidir() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR c0, GM_ADDR y, uint32_t batch, uint32_t hidden, uint32_t layers2)
    {
        (void)x;
        (void)c0;
        this->batch = batch;
        this->hidden = hidden;
        this->layers2 = layers2;

        // Set global buffers
        h0Gm.SetGlobalBuffer((__gm__ float *)h0, (uint32_t)layers2 * batch * hidden);
        yGm.SetGlobalBuffer((__gm__ float *)y, batch * hidden * 2);
    }

    __aicore__ inline void Process()
    {
        // Distribute batches across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t idx = AscendC::GetBlockIdx();
        uint32_t base = batch / blocks;
        uint32_t rem = batch % blocks;
        uint32_t start = base * idx + (idx < rem ? idx : rem);
        uint32_t count = base + (idx < rem ? 1 : 0);
        uint32_t end = start + count;

        if (hidden == 0 || layers2 < 2 || start >= end) {
            return;
        }

        // Forward last layer index and backward last layer index
        uint32_t f_idx = layers2 - 2; // forward of last layer
        uint32_t b_idx = layers2 - 1; // backward of last layer

        // Allocate local buffers for one hidden vector
        AscendC::LocalTensor<float> bufF = AscendC::AllocTensor<float>(hidden);
        AscendC::LocalTensor<float> bufB = AscendC::AllocTensor<float>(hidden);

        uint32_t strideBH = batch * hidden;

        for (uint32_t b = start; b < end; ++b) {
            uint32_t offF = f_idx * strideBH + b * hidden;
            uint32_t offB = b_idx * strideBH + b * hidden;
            uint32_t outOff = b * (hidden * 2);

            // Copy forward hidden
            AscendC::DataCopy(bufF, h0Gm[offF], hidden);
            // Copy backward hidden
            AscendC::DataCopy(bufB, h0Gm[offB], hidden);
            // Write to output: [bufF, bufB]
            AscendC::DataCopy(yGm[outOff], bufF, hidden);
            AscendC::DataCopy(yGm[outOff + hidden], bufB, hidden);
        }

        AscendC::FreeTensor(bufF);
        AscendC::FreeTensor(bufB);
    }

private:
    AscendC::GlobalTensor<float> h0Gm;
    AscendC::GlobalTensor<float> yGm;
    uint32_t batch;
    uint32_t hidden;
    uint32_t layers2;
};

extern "C" __global__ __aicore__ void lstm_bidirectional_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR c0, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelLstmBidir op;
    op.Init(x, h0, c0, y, tiling_data.batch, tiling_data.hidden, tiling_data.layers2);
    op.Process();
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor lstm_bidirectional_custom_impl_npu(const at::Tensor& x, const at::Tensor& h0, const at::Tensor& c0) {
    // h0: (layers2, batch, hidden)
    TORCH_CHECK(h0.dim() == 3, "h0 must be 3D (layers2, batch, hidden)");
    int64_t batch = h0.size(1);
    int64_t hidden = h0.size(2);
    at::Tensor result = at::empty({batch, hidden * 2}, h0.options());
    EXEC_NPU_CMD(aclnnLstmBidirectionalCustom, x, h0, c0, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("lstm_bidirectional_custom", &lstm_bidirectional_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("lstm_bidirectional_custom", &lstm_bidirectional_custom_impl_npu, "LSTM Bidirectional Custom");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, output_size, dropout=0.0):
        super().__init__()
        self.input_size = input_size
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        self.dropout = dropout
        self.fc = torch.nn.Linear(hidden_size * 2, output_size)

    def forward(self, x, h0, c0):
        # Custom NPU op produces (batch, 2 * hidden_size) by extracting the last layer's
        # forward and backward hidden states and concatenating them.
        y = custom_ops_lib.lstm_bidirectional_custom(x, h0, c0)
        out = self.fc(y)
        return out
'''
