host_operator_src="""
#include "downsample_bilinear_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    DownsampleBilinearCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    // Expect NCHW
    uint32_t n = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(0));
    uint32_t c = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(1));
    uint32_t h_in = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(2));
    uint32_t w_in = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(3));

    // Fixed target size (60, 80) as per model definition
    uint32_t h_out = 60;
    uint32_t w_out = 80;

    context->SetBlockDim(BLOCK_DIM);

    uint32_t nc = n * c;
    uint32_t ncPerBlock = (nc + BLOCK_DIM - 1) / BLOCK_DIM;

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h_in(h_in);
    tiling.set_w_in(w_in);
    tiling.set_h_out(h_out);
    tiling.set_w_out(w_out);
    tiling.set_ncPerBlock(ncPerBlock);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);

    // Copy N and C from input, set H=60, W=80
    gert::Shape out = *x_shape;
    out.SetDim(2, 60);
    out.SetDim(3, 80);
    *y_shape = out;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class DownsampleBilinearCustom : public OpDef {
public:
    explicit DownsampleBilinearCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(DownsampleBilinearCustom);
}

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void downsample_bilinear_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N = tiling_data.n();
    const uint32_t C = tiling_data.c();
    const uint32_t H_in = tiling_data.h_in();
    const uint32_t W_in = tiling_data.w_in();
    const uint32_t H_out = tiling_data.h_out();
    const uint32_t W_out = tiling_data.w_out();
    const uint32_t ncPerBlock = tiling_data.ncPerBlock();

    __gm__ const float* x_gm = reinterpret_cast<__gm__ const float*>(x);
    __gm__ float* y_gm = reinterpret_cast<__gm__ float*>(y);

    uint32_t block_idx = AscendC::GetBlockIdx();
    uint32_t start_nc = block_idx * ncPerBlock;
    uint32_t total_nc = N * C;
    uint32_t end_nc = start_nc + ncPerBlock;
    if (end_nc > total_nc) {
        end_nc = total_nc;
    }

    float scale_h = static_cast<float>(H_in) / static_cast<float>(H_out);
    float scale_w = static_cast<float>(W_in) / static_cast<float>(W_out);

    for (uint32_t nc_idx = start_nc; nc_idx < end_nc; ++nc_idx) {
        uint32_t n = nc_idx / C;
        uint32_t c = nc_idx % C;

        uint64_t base_in = ((static_cast<uint64_t>(n) * C + c) * H_in) * W_in;
        uint64_t base_out = ((static_cast<uint64_t>(n) * C + c) * H_out) * W_out;

        for (uint32_t oh = 0; oh < H_out; ++oh) {
            float in_y = (static_cast<float>(oh) + 0.5f) * scale_h - 0.5f; // align_corners=False
            int32_t y0 = static_cast<int32_t>(AscendC::Floor(in_y));
            int32_t y1 = y0 + 1;
            float wy = in_y - static_cast<float>(y0);
            if (y0 < 0) { y0 = 0; wy = 0.0f; }
            if (y1 >= static_cast<int32_t>(H_in)) { y1 = H_in - 1; }
            float wy0 = 1.0f - wy;
            float wy1 = wy;

            for (uint32_t ow = 0; ow < W_out; ++ow) {
                float in_x = (static_cast<float>(ow) + 0.5f) * scale_w - 0.5f;
                int32_t x0 = static_cast<int32_t>(AscendC::Floor(in_x));
                int32_t x1 = x0 + 1;
                float wx = in_x - static_cast<float>(x0);
                if (x0 < 0) { x0 = 0; wx = 0.0f; }
                if (x1 >= static_cast<int32_t>(W_in)) { x1 = W_in - 1; }
                float wx0 = 1.0f - wx;
                float wx1 = wx;

                uint64_t idx00 = base_in + static_cast<uint64_t>(y0) * W_in + static_cast<uint64_t>(x0);
                uint64_t idx01 = base_in + static_cast<uint64_t>(y0) * W_in + static_cast<uint64_t>(x1);
                uint64_t idx10 = base_in + static_cast<uint64_t>(y1) * W_in + static_cast<uint64_t>(x0);
                uint64_t idx11 = base_in + static_cast<uint64_t>(y1) * W_in + static_cast<uint64_t>(x1);

                float v00 = x_gm[idx00];
                float v01 = x_gm[idx01];
                float v10 = x_gm[idx10];
                float v11 = x_gm[idx11];

                float top = v00 * wx0 + v01 * wx1;
                float bottom = v10 * wx0 + v11 * wx1;
                float val = top * wy0 + bottom * wy1;

                uint64_t out_idx = base_out + static_cast<uint64_t>(oh) * W_out + static_cast<uint64_t>(ow);
                y_gm[out_idx] = val;
            }
        }
    }
}
"""

#include <torch/library.h>

#include <torch/extension.h>

at::Tensor downsample_bilinear_custom_impl_npu(const at::Tensor& x) {
    auto sizes = x.sizes();
    TORCH_CHECK(sizes.size() == 4, "Expected 4D NCHW tensor");
    int64_t N = sizes[0];
    int64_t C = sizes[1];
    int64_t H_out = 60;
    int64_t W_out = 80;
    at::Tensor result = at::empty({N, C, H_out, W_out}, x.options());
    EXEC_NPU_CMD(aclnnDownsampleBilinearCustom, x, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("downsample_bilinear_custom", &downsample_bilinear_custom_impl_npu);
}
