    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;

    int64_t K1 = a_shape->GetDim(0);
    int64_t M  = a_shape->GetDim(1);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

static __aicore__ inline uint32_t ceil_div_u32(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

class KernelMatmulTransposedA {
public:
    __aicore__ inline KernelMatmulTransposedA() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileM = tileM;
        this->tileN = tileN;
        this->tileK = tileK;

        rowsPerBlock = ceil_div_u32(M, AscendC::GetBlockNum());
        mStart = AscendC::GetBlockIdx() * rowsPerBlock;
        if (mStart > M) { mStart = M; }
        mEnd = mStart + rowsPerBlock;
        if (mEnd > M) { mEnd = M; }
        if (mEnd < mStart) { mEnd = mStart; }

        // Set global buffers: a is (K, M), b is (K, N), c is (M, N)
        aGm.SetGlobalBuffer((__gm__ DTYPE_A*)a, (uint64_t)K * (uint64_t)M);
        bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, (uint64_t)K * (uint64_t)N);
        cGm.SetGlobalBuffer((__gm__ DTYPE_C*)c, (uint64_t)M * (uint64_t)N);

        // UB buffers for tiles
        pipe.InitBuffer(inQueueA, BUFFER_NUM, tileK * tileM * sizeof(DTYPE_A));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, tileK * tileN * sizeof(DTYPE_B));
        pipe.InitBuffer(outQueueC, BUFFER_NUM, tileM * tileN * sizeof(DTYPE_C));
    }

    __aicore__ inline void Process()
    {
        if (mStart >= mEnd) {
            return;
        }
        for (uint32_t m0 = mStart; m0 < mEnd; m0 += tileM) {
            uint32_t curTM = (m0 + tileM <= mEnd) ? tileM : (mEnd - m0);
            for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
                uint32_t curTN = (n0 + tileN <= N) ? tileN : (N - n0);

                // Allocate C tile and zero it
                AscendC::LocalTensor<DTYPE_C> cLocal = outQueueC.AllocTensor<DTYPE_C>();
                uint32_t cSize = curTM * curTN;
                // Zero initialize
                for (uint32_t i = 0; i < cSize; ++i) {
                    cLocal.SetValue(i, (DTYPE_C)0);
                }

                // Accumulate over K
                for (uint32_t k0 = 0; k0 < K; k0 += tileK) {
                    uint32_t curTK = (k0 + tileK <= K) ? tileK : (K - k0);

                    // Load A panel of shape (curTK, curTM) from A[k0:k0+curTK, m0:m0+curTM]
                    AscendC::LocalTensor<DTYPE_A> aLocal = inQueueA.AllocTensor<DTYPE_A>();
                    for (uint32_t rk = 0; rk < curTK; ++rk) {
                        uint64_t aSrcOffset = (uint64_t)(k0 + rk) * (uint64_t)M + (uint64_t)m0;
                        AscendC::DataCopy(aLocal[rk * curTM], aGm[aSrcOffset], curTM);
                    }
                    inQueueA.EnQue(aLocal);

                    // Load B panel of shape (curTK, curTN) from B[k0:k0+curTK, n0:n0+curTN]
                    AscendC::LocalTensor<DTYPE_B> bLocal = inQueueB.AllocTensor<DTYPE_B>();
                    for (uint32_t rk = 0; rk < curTK; ++rk) {
                        uint64_t bSrcOffset = (uint64_t)(k0 + rk) * (uint64_t)N + (uint64_t)n0;
                        AscendC::DataCopy(bLocal[rk * curTN], bGm[bSrcOffset], curTN);
                    }
                    inQueueB.EnQue(bLocal);

                    // Compute C += A^T_panel @ B_panel
                    AscendC::LocalTensor<DTYPE_A> aTile = inQueueA.DeQue<DTYPE_A>();
                    AscendC::LocalTensor<DTYPE_B> bTile = inQueueB.DeQue<DTYPE_B>();
                    for (uint32_t i = 0; i < curTM; ++i) {
                        for (uint32_t rk = 0; rk < curTK; ++rk) {
                            DTYPE_A aval = aTile[rk * curTM + i];
                            const uint32_t bRowBase = rk * curTN;
                            const uint32_t cRowBase = i * curTN;
                            for (uint32_t j = 0; j < curTN; ++j) {
                                cLocal[cRowBase + j] = cLocal[cRowBase + j] + (DTYPE_C)(aval * bTile[bRowBase + j]);
                            }
                        }
                    }
                    inQueueA.FreeTensor(aTile);
                    inQueueB.FreeTensor(bTile);
                }

                // Store C tile back to GM at c[m0:m0+curTM, n0:n0+curTN]
                for (uint32_t i = 0; i < curTM; ++i) {
                    uint64_t cDstOffset = (uint64_t)(m0 + i) * (uint64_t)N + (uint64_t)n0;
                    AscendC::DataCopy(cGm[cDstOffset], cLocal[i * curTN], curTN);
                }
                outQueueC.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC;

    AscendC::GlobalTensor<DTYPE_A> aGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_C> cGm;

    uint32_t M, N, K;
    uint32_t tileM, tileN, tileK;
    uint32_t rowsPerBlock;
    uint32_t mStart, mEnd;
};

extern "C" __global__ __aicore__ void matmul_with_transposed_a_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulTransposedA op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
"""

at::Tensor matmul_with_transposed_a_custom_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // A: (K, M), B: (K, N), output: (M, N)
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "Inputs must be 2D");
    TORCH_CHECK(A.size(0) == B.size(0), "K dimensions must match");
    auto M = A.size(1);
    auto N = B.size(1);
    at::Tensor result = at::empty({M, N}, A.options());
    EXEC_NPU_CMD(aclnnMatmulWithTransposedACustom, A, B, result);
    return result;

    m.impl("matmul_with_transposed_a_custom", &matmul_with_transposed_a_custom_impl_npu);
