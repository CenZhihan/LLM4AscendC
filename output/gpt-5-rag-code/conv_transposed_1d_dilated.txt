inline uint32_t div_up(uint64_t a, uint64_t b) { return static_cast<uint32_t>((a + b - 1) / b); }

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ConvTransposed1dDilatedCustomTilingData tiling;

    const auto* xShape = context->GetInputShape(0);
    const auto* wShape = context->GetInputShape(1);
    if (xShape == nullptr || wShape == nullptr) {
        return ge::GRAPH_FAILED;
    }
    // x: [N, Cin, Lin], w: [Cin, Cout, K]
    uint64_t N = xShape->GetOriginShape().GetDim(0);
    uint64_t Cin = xShape->GetOriginShape().GetDim(1);
    uint64_t Lin = xShape->GetOriginShape().GetDim(2);

    uint64_t CinW = wShape->GetOriginShape().GetDim(0);
    uint64_t Cout = wShape->GetOriginShape().GetDim(1);
    uint64_t K = wShape->GetOriginShape().GetDim(2);

    if (CinW != Cin) {
        return ge::GRAPH_FAILED;
    }

    int64_t stride = 1;
    int64_t padding = 0;
    int64_t dilation = 1;
    (void)context->GetAttrInt("stride", stride);
    (void)context->GetAttrInt("padding", padding);
    (void)context->GetAttrInt("dilation", dilation);

    // Lout = (Lin-1)*stride - 2*padding + dilation*(K-1) + 1
    int64_t Lout = static_cast<int64_t>((Lin - 1) * stride - 2 * padding + dilation * (K - 1) + 1);
    if (Lout <= 0) {
        return ge::GRAPH_FAILED;
    }

    // Simple parallelization strategy: split by Cout across blocks
    const uint32_t maxBlocks = 32;
    uint32_t blockDim = static_cast<uint32_t>(Cout < maxBlocks ? Cout : maxBlocks);
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    uint32_t ocPerBlock = div_up(Cout, blockDim);

    // Tile along output length dimension to control L1 usage
    const uint32_t defaultTileLen = 256; // tune if needed

    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_Cin(static_cast<uint32_t>(Cin));
    tiling.set_Cout(static_cast<uint32_t>(Cout));
    tiling.set_Lin(static_cast<uint32_t>(Lin));
    tiling.set_Lout(static_cast<uint32_t>(Lout));
    tiling.set_K(static_cast<uint32_t>(K));
    tiling.set_stride(static_cast<uint32_t>(stride));
    tiling.set_padding(static_cast<uint32_t>(padding));
    tiling.set_dilation(static_cast<uint32_t>(dilation));
    tiling.set_ocPerBlock(ocPerBlock);
    tiling.set_tileLen(defaultTileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t QUEUE_DEPTH = 2;

class KernelConvTransposed1dDilated {
public:
    __aicore__ inline KernelConvTransposed1dDilated() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, uint32_t N, uint32_t Cin,
                                uint32_t Cout, uint32_t Lin, uint32_t Lout, uint32_t K,
                                uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t ocPerBlock, uint32_t tileLen)
    {
        this->N = N;
        this->Cin = Cin;
        this->Cout = Cout;
        this->Lin = Lin;
        this->Lout = Lout;
        this->K = K;
        this->stride = stride;
        this->padding = padding;
        this->dilation = dilation;
        this->ocPerBlock = ocPerBlock;
        this->tileLen = tileLen;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, (uint64_t)N * Cin * Lin);
        wGm.SetGlobalBuffer((__gm__ DTYPE_W*)w, (uint64_t)Cin * Cout * K);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y, (uint64_t)N * Cout * Lout);
        if (b != nullptr) {
            bGm.SetGlobalBuffer((__gm__ DTYPE_B*)b, Cout);
            hasBias = true;
        } else {
            hasBias = false;
        }

        pipe.InitBuffer(bufX, QUEUE_DEPTH, sizeof(DTYPE_X));     // scalar buffer
        pipe.InitBuffer(bufW, QUEUE_DEPTH, sizeof(DTYPE_W));     // scalar buffer
        pipe.InitBuffer(bufZ, QUEUE_DEPTH, sizeof(DTYPE_Y));     // scalar accumulation buffer (1 element)
    }

    __aicore__ inline void Process()
    {
        uint32_t blkIdx = AscendC::GetBlockIdx();
        uint32_t blkNum = AscendC::GetBlockNum();

        // assigned output channel range
        uint32_t ocStart = blkIdx * ocPerBlock;
        if (ocStart >= Cout) return;
        uint32_t ocEnd = ocStart + ocPerBlock;
        if (ocEnd > Cout) ocEnd = Cout;

        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t oc = ocStart; oc < ocEnd; ++oc) {
                // process along output length with tiles
                for (uint32_t tStart = 0; tStart < Lout; tStart += tileLen) {
                    uint32_t tEnd = tStart + tileLen;
                    if (tEnd > Lout) tEnd = Lout;

                    for (uint32_t r = tStart; r < tEnd; ++r) {
                        // accLocal = 0
                        AscendC::LocalTensor<DTYPE_Y> accLocal = bufZ.AllocTensor<DTYPE_Y>();
                        // zero-init by multiplying itself with 0
                        AscendC::Muls(accLocal, accLocal, static_cast<DTYPE_Y>(0), 1);

                        // sum over Cin and K
                        for (uint32_t ic = 0; ic < Cin; ++ic) {
                            // base offsets
                            uint64_t xBase = ((uint64_t)n * Cin + ic) * Lin;
                            uint64_t wBase = ((uint64_t)ic * Cout + oc) * K;

                            for (uint32_t k = 0; k < K; ++k) {
                                // For general stride and dilation:
                                // t (input index) must satisfy r + padding - dilation*k = t * stride
                                int64_t numer = static_cast<int64_t>(r) + static_cast<int64_t>(padding) - static_cast<int64_t>(dilation) * static_cast<int64_t>(k);
                                if (numer < 0) continue;
                                if (numer % static_cast<int64_t>(stride) != 0) continue;
                                int64_t tIn = numer / static_cast<int64_t>(stride);
                                if (tIn < 0 || tIn >= static_cast<int64_t>(Lin)) continue;

                                // load x scalar
                                AscendC::LocalTensor<DTYPE_X> xLocal = bufX.AllocTensor<DTYPE_X>();
                                AscendC::DataCopy(xLocal, xGm[xBase + static_cast<uint64_t>(tIn)], 1);

                                // load w scalar
                                AscendC::LocalTensor<DTYPE_W> wLocal = bufW.AllocTensor<DTYPE_W>();
                                AscendC::DataCopy(wLocal, wGm[wBase + k], 1);

                                // tmp = x * w
                                AscendC::LocalTensor<DTYPE_Y> tmpLocal = bufZ.AllocTensor<DTYPE_Y>();
                                AscendC::Mul(tmpLocal, xLocal, wLocal, 1);

                                // accLocal = accLocal + tmpLocal
                                AscendC::LocalTensor<DTYPE_Y> sumLocal = bufZ.AllocTensor<DTYPE_Y>();
                                AscendC::Add(sumLocal, accLocal, tmpLocal, 1);

                                // recycle
                                bufX.FreeTensor(xLocal);
                                bufW.FreeTensor(wLocal);
                                bufZ.FreeTensor(tmpLocal);
                                bufZ.FreeTensor(accLocal);

                                // keep new acc
                                accLocal = sumLocal;
                            }
                        }

                        // add bias
                        if (hasBias) {
                            AscendC::LocalTensor<DTYPE_B> bLocal = bufW.AllocTensor<DTYPE_B>();
                            AscendC::DataCopy(bLocal, bGm[oc], 1);
                            AscendC::LocalTensor<DTYPE_Y> sumLocal2 = bufZ.AllocTensor<DTYPE_Y>();
                            AscendC::Add(sumLocal2, accLocal, bLocal, 1);
                            bufW.FreeTensor(bLocal);
                            bufZ.FreeTensor(accLocal);
                            accLocal = sumLocal2;
                        }

                        // write to y
                        uint64_t yOff = ((uint64_t)n * Cout + oc) * Lout + r;
                        AscendC::DataCopy(yGm[yOff], accLocal, 1);
                        bufZ.FreeTensor(accLocal);
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN,  QUEUE_DEPTH> bufX;  // for scalar x or reuse
    AscendC::TQue<AscendC::TPosition::VECIN,  QUEUE_DEPTH> bufW;  // for scalar w/bias
    AscendC::TQue<AscendC::TPosition::VECOUT, QUEUE_DEPTH> bufZ;  // for accumulators

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_W> wGm;
    AscendC::GlobalTensor<DTYPE_B> bGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t N, Cin, Cout, Lin, Lout, K;
    uint32_t stride, padding, dilation;
    uint32_t ocPerBlock, tileLen;
    bool hasBias;
};

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1dDilated op;
    op.Init(x, w, b, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.Lin, tiling_data.Lout, tiling_data.K,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation,
            tiling_data.ocPerBlock, tiling_data.tileLen);
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline int64_t compute_lout(int64_t lin, int64_t k, int64_t stride, int64_t padding, int64_t dilation) {
    return (lin - 1) * stride - 2 * padding + dilation * (k - 1) + 1;
}

at::Tensor conv_transposed_1d_dilated_custom_impl_npu(const at::Tensor& x,
                                                      const at::Tensor& w,
                                                      c10::optional<at::Tensor> bias_opt,
                                                      int64_t stride,
                                                      int64_t padding,
                                                      int64_t dilation) {
    TORCH_CHECK(x.dim() == 3, "x must be [N, Cin, L]");
    TORCH_CHECK(w.dim() == 3, "w must be [Cin, Cout, K]");
    TORCH_CHECK(x.scalar_type() == at::kFloat && w.scalar_type() == at::kFloat, "float32 only");

    at::Tensor b;
    if (bias_opt.has_value() && bias_opt.value().defined() && bias_opt.value().numel() > 0) {
        b = bias_opt.value();
    } else {
        // create zero bias of size Cout
        int64_t Cout = w.size(1);
        b = at::zeros({Cout}, w.options());
    }

    int64_t N = x.size(0);
    int64_t Cin = x.size(1);
    int64_t Lin = x.size(2);
    TORCH_CHECK(Cin == w.size(0), "Cin mismatch with weight");
    int64_t Cout = w.size(1);
    int64_t K = w.size(2);

    int64_t Lout = compute_lout(Lin, K, stride, padding, dilation);
    TORCH_CHECK(Lout > 0, "computed output length must be positive");
    at::Tensor y = at::empty({N, Cout, Lout}, x.options());

    EXEC_NPU_CMD(aclnnConvTransposed1dDilatedCustom, x, w, b, stride, padding, dilation, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_transposed_1d_dilated_custom", &conv_transposed_1d_dilated_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_transposed_1d_dilated_custom", &conv_transposed_1d_dilated_custom_impl_npu, "ConvTranspose1d with dilation (custom NPU)");

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Custom NPU-optimized ConvTranspose1d with dilation using AscendC kernel.
    """
    def __init__(self, in_channels: int, out_channels: int, kernel_size: int, stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super().__init__()
        self.in_channels = in_channels

        self.padding = padding
        self.dilation = dilation

        self.conv1d_transpose = torch.nn.ConvTranspose1d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, dilation=dilation, bias=bias)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        w = self.conv1d_transpose.weight
        b = self.conv1d_transpose.bias if self.bias_flag else None
        return custom_ops_lib.conv_transposed_1d_dilated_custom(x, w, b, self.stride, self.padding, self.dilation)
