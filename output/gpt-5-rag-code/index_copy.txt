static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    IndexCopyCustomTilingData tiling;

    const auto* xShape = context->GetInputShape(0);
    const auto* idxShape = context->GetInputShape(1);
    // x: [N, D]
    uint64_t totalSize = xShape->GetOriginShape().GetShapeSize();
    uint64_t nRowsU64 = xShape->GetOriginShape().GetDimNum() > 0 ? xShape->GetOriginShape().GetDim(0) : 1;
    uint64_t nColsU64 = (nRowsU64 == 0) ? 0 : (totalSize / nRowsU64);

    uint64_t indexNumU64 = idxShape->GetOriginShape().GetShapeSize();

    uint32_t nRows = static_cast<uint32_t>(nRowsU64);
    uint32_t nCols = static_cast<uint32_t>(nColsU64);
    uint32_t indexNum = static_cast<uint32_t>(indexNumU64);

    // simple column tile
    const uint32_t TILE_COLS_DEFAULT = 256;
    uint32_t tileCols = (nCols < TILE_COLS_DEFAULT) ? nCols : TILE_COLS_DEFAULT;

    // Use single core to keep index order semantics (last write wins)
    context->SetBlockDim(1);

    tiling.set_nRows(nRows);
    tiling.set_nCols(nCols);
    tiling.set_indexNum(indexNum);
    tiling.set_tileCols(tileCols);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelIndexCopy {
public:
    __aicore__ inline KernelIndexCopy() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR indices, GM_ADDR src, GM_ADDR z,
                                uint32_t nRows, uint32_t nCols, uint32_t indexNum, uint32_t tileCols) {
        this->nRows = nRows;
        this->nCols = nCols;
        this->indexNum = indexNum;
        this->tileCols = tileCols == 0 ? 1 : tileCols;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x, nRows * nCols);
        srcGm.SetGlobalBuffer((__gm__ DTYPE_Y*)src, indexNum * nCols);
        zGm.SetGlobalBuffer((__gm__ DTYPE_Z*)z, nRows * nCols);
        indicesPtr = (__gm__ int64_t*)indices;

        // one generic vecin queue for staging tiles
        pipe.InitBuffer(tileQueueIn, BUFFER_NUM, this->tileCols * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process() {
        // Phase 1: z = x (copy all rows)
        for (uint32_t r = 0; r < nRows; ++r) {
            CopyRow(xGm, r, zGm, r);
        }
        // Phase 2: overwrite rows in z according to indices and src
        for (uint32_t j = 0; j < indexNum; ++j) {
            int64_t idx64 = indicesPtr[j];
            if (idx64 < 0) { continue; }
            uint32_t idx = static_cast<uint32_t>(idx64);
            if (idx >= nRows) { continue; }
            CopyRow(srcGm, j, zGm, idx);
        }
    }

private:
    template <typename Tin, typename Tout>
    __aicore__ inline void CopyRow(AscendC::GlobalTensor<Tin>& in, uint32_t inRow,
                                   AscendC::GlobalTensor<Tout>& out, uint32_t outRow) {
        uint32_t offsetIn = inRow * nCols;
        uint32_t offsetOut = outRow * nCols;
        for (uint32_t c = 0; c < nCols; c += tileCols) {
            uint32_t cur = (c + tileCols <= nCols) ? tileCols : (nCols - c);
            AscendC::LocalTensor<Tin> buf = tileQueueIn.AllocTensor<Tin>();
            AscendC::DataCopy(buf, in[offsetIn + c], cur);
            AscendC::DataCopy(out[offsetOut + c], buf, cur);
            tileQueueIn.FreeTensor(buf);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tileQueueIn;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> srcGm;
    AscendC::GlobalTensor<DTYPE_Z> zGm;
    __gm__ int64_t* indicesPtr;

    uint32_t nRows;
    uint32_t nCols;
    uint32_t indexNum;
    uint32_t tileCols;
};

extern "C" __global__ __aicore__ void index_copy_custom(GM_ADDR x, GM_ADDR indices, GM_ADDR src, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    if (AscendC::GetBlockIdx() > 0) { return; } // single core execution
    KernelIndexCopy op;
    op.Init(x, indices, src, z,
            tiling_data.nRows, tiling_data.nCols, tiling_data.indexNum, tiling_data.tileCols);
    op.Process();
}
