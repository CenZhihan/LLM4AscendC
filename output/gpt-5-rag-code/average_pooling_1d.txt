    tiling.set_stride(static_cast<uint32_t>(s));
    tiling.set_padding(static_cast<uint32_t>(p));

kernel_src="""
#include "kernel_operator.h"

class KernelAveragePool1D {
public:
    __aicore__ inline KernelAveragePool1D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t inLen, uint32_t outLen,
                                uint32_t k, uint32_t s, uint32_t p,
                                uint64_t totalOutput)
    {
        this->xPtr = reinterpret_cast<__gm__ DTYPE_X*>(x);
        this->yPtr = reinterpret_cast<__gm__ DTYPE_Y*>(y);
        this->n = n;
        this->c = c;
        this->inLen = inLen;
        this->outLen = outLen;
        this->k = k;
        this->s = s;
        this->p = p;
        this->totalOutput = totalOutput;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t bid = AscendC::GetBlockIdx();
        uint64_t blockLen = (totalOutput + blocks - 1) / blocks;
        this->startIdx = blockLen * bid;
        uint64_t endTmp = blockLen * (static_cast<uint64_t>(bid) + 1);
        this->endIdx = endTmp > totalOutput ? totalOutput : endTmp;
    }

    __aicore__ inline void Process()
    {
        if (outLen == 0 || totalOutput == 0) {
            return;
        }

        for (uint64_t idx = startIdx; idx < endIdx; ++idx) {
            uint64_t groupAbs = idx / outLen; // [0, n*c)
            uint32_t t = static_cast<uint32_t>(idx % outLen);

            int32_t inStart = static_cast<int32_t>(t) * static_cast<int32_t>(s) - static_cast<int32_t>(p);
            int32_t validStart = inStart < 0 ? 0 : inStart;
            int32_t validEnd = inStart + static_cast<int32_t>(k);
            if (validEnd > static_cast<int32_t>(inLen)) validEnd = static_cast<int32_t>(inLen);

            int32_t validCount = validEnd - validStart;
            float sum = 0.0f;

            if (validCount > 0) {
                uint64_t baseIn = groupAbs * static_cast<uint64_t>(inLen);
                for (int32_t i = 0; i < validCount; ++i) {
                    uint64_t pos = baseIn + static_cast<uint64_t>(validStart + i);
                    sum += static_cast<float>(xPtr[pos]);
                }
                float avg = sum / static_cast<float>(validCount); // count_include_pad = False
                yPtr[idx] = static_cast<DTYPE_Y>(avg);
            } else {
                yPtr[idx] = static_cast<DTYPE_Y>(0.0f);
            }
        }
    }

private:
    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_Y* yPtr;
    uint32_t n, c;
    uint32_t inLen, outLen;
    uint32_t k, s, p;
    uint64_t totalOutput;
    uint64_t startIdx, endIdx;
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePool1D op;
    op.Init(x, y,
            tiling_data.n(), tiling_data.c(),
            tiling_data.inLen(), tiling_data.outLen(),
            tiling_data.kernelSize(), tiling_data.stride(), tiling_data.padding(),
            tiling_data.totalOutput());
    op.Process();
}
