#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConv3D {
public:
    __aicore__ inline KernelConv3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR b,
                                uint64_t totalOut, uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t K, uint32_t Do, uint32_t Ho, uint32_t Wo,
                                uint32_t tileNum)
    {
        // store meta
        this->N = n; this->Cin = cin; this->Cout = cout;
        this->D = D; this->H = H; this->W = W;
        this->K = K; this->Do = Do; this->Ho = Ho; this->Wo = Wo;

        this->totalOut = totalOut;
        this->blockLength = totalOut / AscendC::GetBlockNum();
        this->tileNum = tileNum == 0 ? 1 : tileNum;
        this->tileLength = (this->blockLength / this->tileNum);
        if (this->tileLength == 0) { this->tileLength = this->blockLength; this->tileNum = 1; }

        xPtr = reinterpret_cast<__gm__ float *>(x);
        wPtr = reinterpret_cast<__gm__ float *>(w);
        yPtr = reinterpret_cast<__gm__ float *>(y);
        bPtr = reinterpret_cast<__gm__ float *>(b);

        // no on-chip buffering in this simple reference kernel
    }

    __aicore__ inline void Process()
    {
        uint64_t start = this->blockLength * AscendC::GetBlockIdx();
        uint64_t end   = start + this->blockLength;

        // process in simple tiles
        uint64_t cursor = start;
        while (cursor < end) {
            uint64_t curLen = AscendC::Min(this->tileLength, end - cursor);
            ComputeRange(cursor, curLen);
            cursor += curLen;
        }
    }

private:
    __aicore__ inline void ComputeRange(uint64_t startOutIdx, uint64_t len)
    {
        for (uint64_t idx = 0; idx < len; ++idx) {
            uint64_t flat = startOutIdx + idx;
            // decode flat index into (n, co, od, oh, ow)
            uint64_t t = flat;
            uint32_t ow = static_cast<uint32_t>(t % Wo); t /= Wo;
            uint32_t oh = static_cast<uint32_t>(t % Ho); t /= Ho;
            uint32_t od = static_cast<uint32_t>(t % Do); t /= Do;
            uint32_t co = static_cast<uint32_t>(t % Cout); t /= Cout;
            uint32_t n  = static_cast<uint32_t>(t % N);

            float acc = 0.0f;
            // optional bias
            if (bPtr != nullptr) {
                acc = bPtr[co];
            }

            // convolution loops: stride=1, padding=0, dilation=1, groups=1
            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t kd = 0; kd < K; ++kd) {
                    for (uint32_t kh = 0; kh < K; ++kh) {
                        for (uint32_t kw = 0; kw < K; ++kw) {
                            uint32_t id = od + kd;
                            uint32_t ih = oh + kh;
                            uint32_t iw = ow + kw;
                            // input index: (((n*Cin+ci)*D+id)*H+ih)*W+iw
                            uint64_t xIndex = (((static_cast<uint64_t>(n) * Cin + ci) * D + id) * H + ih) * W + iw;
                            // weight index: ((((co)*Cin+ci)*K+kd)*K+kh)*K+kw
                            uint64_t wIndex = ((((static_cast<uint64_t>(co) * Cin + ci) * K + kd) * K + kh) * K + kw);
                            float xv = xPtr[xIndex];
                            float wv = wPtr[wIndex];
                            acc += xv * wv;
                        }
                    }
                }
            }

            // output index: (((n*Cout+co)*Do+od)*Ho+oh)*Wo+ow
            uint64_t yIndex = (((static_cast<uint64_t>(n) * Cout + co) * Do + od) * Ho + oh) * Wo + ow;
            yPtr[yIndex] = acc;
        }
    }

private:
    __gm__ float *xPtr;
    __gm__ float *wPtr;
    __gm__ float *yPtr;
    __gm__ float *bPtr;

    uint64_t totalOut;
    uint32_t N, Cin, Cout;
    uint32_t D, H, W;
    uint32_t K;
    uint32_t Do, Ho, Wo;

    uint64_t blockLength;
    uint32_t tileNum;
    uint64_t tileLength;
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR b, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3D op;
    op.Init(x, w, y, b,
            tiling_data.totalOut,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.k, tiling_data.do_, tiling_data.ho, tiling_data.wo,
            tiling_data.tileNum);
    op.Process();
}
