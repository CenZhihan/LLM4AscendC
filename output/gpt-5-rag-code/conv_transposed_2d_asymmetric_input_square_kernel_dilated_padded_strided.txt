    const auto* groupsAttr    = context->GetAttr("groups");

    int64_t strideW = (stridesAttr && stridesAttr->GetListInt().size() > 1) ? stridesAttr->GetListInt()[1] : strideH;

    int64_t dilW    = (dilationsAttr && dilationsAttr->GetListInt().size() > 1) ? dilationsAttr->GetListInt()[1] : dilH;
    int64_t groups  = (groupsAttr) ? groupsAttr->GetInt() : 1;

    (void)Cin; (void)KcIn; // layout consistency check omitted

    int64_t Cout = KcOut * groups;
    int64_t Hout = (Hin - 1) * strideH - 2 * padH + dilH * (kH - 1) + 1;

kernel_src="""
#include "kernel_operator.h"

class KernelConvTranspose2D {
public:
    __aicore__ inline KernelConvTranspose2D() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t hIn, uint32_t wIn,
                                uint32_t kH, uint32_t kW,
                                uint32_t hOut, uint32_t wOut,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups,
                                uint32_t blockTileOutHW, uint32_t tileNum)
    {
        (void)groups; (void)tileNum;
        this->N = n; this->Cin = cIn; this->Cout = cOut;
        this->Hin = hIn; this->Win = wIn;
        this->KH = kH; this->KW = kW;
        this->Hout = hOut; this->Wout = wOut;
        this->strideH = strideH; this->strideW = strideW;
        this->padH = padH; this->padW = padW;
        this->dilH = dilationH; this->dilW = dilationW;
        this->blockTileOutHW = blockTileOutHW;

        xPtr = (__gm__ DTYPE_X*)x;
        wPtr = (__gm__ DTYPE_W*)w;
        yPtr = (__gm__ DTYPE_Y*)y;
        bPtr = (__gm__ DTYPE_B*)b;

        totalOut = N * Cout * Hout * Wout;
        // Block-based partition over output elements
        blockStartOut = blockTileOutHW * AscendC::GetBlockIdx();
        blockEndOut = (AscendC::GetBlockIdx() == AscendC::GetBlockNum() - 1) ? totalOut : blockStartOut + blockTileOutHW;

        // Initialize a simple pipe for optional local buffers (not used heavily)
        pipe.InitBuffer(outQueue, 1, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Zero-initialize output for the block range
        for (uint32_t idx = blockStartOut; idx < blockEndOut; ++idx) {
            yPtr[idx] = static_cast<DTYPE_Y>(0);
        }

        // Accumulate transposed convolution contributions
        for (uint32_t n = 0; n < N; ++n) {
            for (uint32_t ci = 0; ci < Cin; ++ci) {
                for (uint32_t hi = 0; hi < Hin; ++hi) {
                    for (uint32_t wi = 0; wi < Win; ++wi) {
                        // Load input value
                        uint64_t xIndex = (((uint64_t)n * Cin + ci) * Hin + hi) * Win + wi;
                        DTYPE_X xval = xPtr[xIndex];
                        // Scatter add to outputs
                        for (uint32_t kh = 0; kh < KH; ++kh) {
                            int64_t hOutPos = static_cast<int64_t>(hi) * strideH - static_cast<int64_t>(padH) + static_cast<int64_t>(kh) * static_cast<int64_t>(dilH);
                            if (hOutPos < 0 || hOutPos >= static_cast<int64_t>(Hout)) continue;
                            for (uint32_t kw = 0; kw < KW; ++kw) {
                                int64_t wOutPos = static_cast<int64_t>(wi) * strideW - static_cast<int64_t>(padW) + static_cast<int64_t>(kw) * static_cast<int64_t>(dilW);
                                if (wOutPos < 0 || wOutPos >= static_cast<int64_t>(Wout)) continue;
                                for (uint32_t co = 0; co < Cout; ++co) {
                                    // weight layout: [Cin, Cout, KH, KW]
                                    uint64_t wIndex = (((uint64_t)ci * Cout + co) * KH + kh) * KW + kw;
                                    DTYPE_W wval = wPtr[wIndex];
                                    uint64_t yIndex = (((uint64_t)n * Cout + co) * Hout + static_cast<uint32_t>(hOutPos)) * Wout + static_cast<uint32_t>(wOutPos);
                                    // Only update those yIndex within this block's partition to avoid race
                                    if (yIndex >= blockStartOut && yIndex < blockEndOut) {
                                        yPtr[yIndex] = yPtr[yIndex] + static_cast<DTYPE_Y>(xval * wval);
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }

        // Add bias if provided
        if (bPtr != nullptr) {
            for (uint32_t n = 0; n < N; ++n) {
                for (uint32_t co = 0; co < Cout; ++co) {
                    DTYPE_B biasVal = bPtr[co];
                    for (uint32_t h = 0; h < Hout; ++h) {
                        for (uint32_t w = 0; w < Wout; ++w) {
                            uint64_t yIndex = (((uint64_t)n * Cout + co) * Hout + h) * Wout + w;
                            if (yIndex >= blockStartOut && yIndex < blockEndOut) {
                                yPtr[yIndex] = yPtr[yIndex] + static_cast<DTYPE_Y>(biasVal);
                            }
                        }
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue;

    __gm__ DTYPE_X* xPtr;
    __gm__ DTYPE_W* wPtr;
    __gm__ DTYPE_B* bPtr;
    __gm__ DTYPE_Y* yPtr;

    uint32_t N, Cin, Cout;
    uint32_t Hin, Win, KH, KW;
    uint32_t Hout, Wout;
    uint32_t strideH, strideW, padH, padW, dilH, dilW;
    uint32_t blockTileOutHW;
    uint64_t totalOut;
    uint64_t blockStartOut;
    uint64_t blockEndOut;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_square_kernel_dilated_padded_strided_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose2D op;
    op.Init(x, w, b, y,
            tiling_data.n, tiling_data.cIn, tiling_data.cOut,
            tiling_data.hIn, tiling_data.wIn,
            tiling_data.kH, tiling_data.kW,
            tiling_data.hOut, tiling_data.wOut,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups,
            tiling_data.blockTileOutHW, tiling_data.tileNum);
    op.Process();
}
