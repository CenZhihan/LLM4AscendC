kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

template<typename T>
__aicore__ inline void sigmoid_inplace(AscendC::LocalTensor<T>& x, uint32_t len) {
    // y = 1 / (1 + exp(-x))
    AscendC::LocalTensor<T> tmp = AscendC::AllocLocalTensor<T>(len);
    AscendC::Muls(tmp, x, static_cast<T>(-1.0), len);
    AscendC::Exp(tmp, tmp, len);
    AscendC::Adds(tmp, tmp, static_cast<T>(1.0), len);
    AscendC::Reciprocal(x, tmp, len);
    AscendC::FreeLocalTensor(tmp);
}

class KernelLstm {
public:
    __aicore__ inline KernelLstm() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR w_ih, GM_ADDR w_hh,
                                GM_ADDR b_ih, GM_ADDR b_hh,
                                GM_ADDR h0, GM_ADDR c0,
                                GM_ADDR y, GM_ADDR hn, GM_ADDR cn,
                                uint32_t B, uint32_t T, uint32_t I, uint32_t H, uint32_t tileNum)
    {
        this->B = B;
        this->T = T;
        this->I = I;
        this->H = H;
        this->tileNum = tileNum;

        // Block-level partitioning on batch dimension
        this->batchPerBlock = (B + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        this->bStart = this->batchPerBlock * AscendC::GetBlockIdx();
        if (this->bStart + this->batchPerBlock > B) {
            this->batchPerBlock = B - this->bStart;
        }

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->bStart * (T * I), this->batchPerBlock * (T * I));
        wihGm.SetGlobalBuffer((__gm__ DTYPE_X*)w_ih, 4 * H * I);
        whhGm.SetGlobalBuffer((__gm__ DTYPE_X*)w_hh, 4 * H * H);
        bihGm.SetGlobalBuffer((__gm__ DTYPE_X*)b_ih, 4 * H);
        bhhGm.SetGlobalBuffer((__gm__ DTYPE_X*)b_hh, 4 * H);
        h0Gm.SetGlobalBuffer((__gm__ DTYPE_X*)h0 + this->bStart * H, this->batchPerBlock * H);
        c0Gm.SetGlobalBuffer((__gm__ DTYPE_X*)c0 + this->bStart * H, this->batchPerBlock * H);
        yGm.SetGlobalBuffer((__gm__ DTYPE_X*)y + this->bStart * (T * H), this->batchPerBlock * (T * H));
        hnGm.SetGlobalBuffer((__gm__ DTYPE_X*)hn + this->bStart * H, this->batchPerBlock * H);
        cnGm.SetGlobalBuffer((__gm__ DTYPE_X*)cn + this->bStart * H, this->batchPerBlock * H);

        // Local buffer sizes
        pipe.InitBuffer(xInQ, BUFFER_NUM, I * sizeof(DTYPE_X));
        pipe.InitBuffer(hInQ, BUFFER_NUM, H * sizeof(DTYPE_X));
        pipe.InitBuffer(gateQ, BUFFER_NUM, (4 * H) * sizeof(DTYPE_X));
        pipe.InitBuffer(hOutQ, BUFFER_NUM, H * sizeof(DTYPE_X));
        pipe.InitBuffer(cOutQ, BUFFER_NUM, H * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t b = 0; b < this->batchPerBlock; ++b) {
            // Load initial states h_t, c_t
            AscendC::LocalTensor<DTYPE_X> h_t = hInQ.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_X> c_t = cOutQ.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(h_t, h0Gm[b * H], H);
            AscendC::DataCopy(c_t, c0Gm[b * H], H);

            for (uint32_t t = 0; t < this->T; ++t) {
                // x_t
                AscendC::LocalTensor<DTYPE_X> x_t_local = xInQ.AllocTensor<DTYPE_X>();
                AscendC::DataCopy(x_t_local, xGm[b * (T * I) + t * I], I);
                xInQ.EnQue(x_t_local);

                // Compute gates: g = x_t @ W_ih^T + h_t @ W_hh^T + b_ih + b_hh, shape [4H]
                AscendC::LocalTensor<DTYPE_X> gates = gateQ.AllocTensor<DTYPE_X>();
                // MatMul placeholders: expect (1, I) x (I, 4H) + (1, H) x (H, 4H)
                AscendC::MatMul(gates, x_t_local, wihGm.Get<DTYPE_X>(), 1, I, 4 * H, /*transA*/false, /*transB*/true);
                AscendC::MatMul(gates, h_t, whhGm.Get<DTYPE_X>(), 1, H, 4 * H, /*transA*/false, /*transB*/true, /*beta*/1.0);
                // Add biases
                AscendC::Add(gates, gates, bihGm.Get<DTYPE_X>(), 4 * H);
                AscendC::Add(gates, gates, bhhGm.Get<DTYPE_X>(), 4 * H);

                // Slice gates into i, f, g, o, each length H
                AscendC::LocalTensor<DTYPE_X> i_gate = gates; // offset 0
                AscendC::LocalTensor<DTYPE_X> f_gate = gates[H]; // offset H
                AscendC::LocalTensor<DTYPE_X> g_gate = gates[2 * H]; // offset 2H
                AscendC::LocalTensor<DTYPE_X> o_gate = gates[3 * H]; // offset 3H

                // Activations
                sigmoid_inplace(i_gate, H);
                sigmoid_inplace(f_gate, H);
                AscendC::Tanh(g_gate, g_gate, H);
                sigmoid_inplace(o_gate, H);

                // c_t = f_gate * c_t + i_gate * g_gate
                AscendC::Mul(c_t, f_gate, c_t, H);
                AscendC::LocalTensor<DTYPE_X> tmp = hOutQ.AllocTensor<DTYPE_X>();
                AscendC::Mul(tmp, i_gate, g_gate, H);
                AscendC::Add(c_t, c_t, tmp, H);

                // h_t = o_gate * tanh(c_t)
                AscendC::Tanh(tmp, c_t, H);
                AscendC::Mul(h_t, o_gate, tmp, H);

                // Write out y at time t
                AscendC::DataCopy(yGm[b * (T * H) + t * H], h_t, H);

                // Free temps
                hOutQ.FreeTensor(tmp);
                gateQ.FreeTensor(gates);
                xInQ.FreeTensor(x_t_local);
            }

            // Store hn, cn
            AscendC::DataCopy(hnGm[b * H], h_t, H);
            AscendC::DataCopy(cnGm[b * H], c_t, H);

            // Free state tensors
            hInQ.FreeTensor(h_t);
            cOutQ.FreeTensor(c_t);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xInQ, hInQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> gateQ, hOutQ, cOutQ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_X> wihGm;
    AscendC::GlobalTensor<DTYPE_X> whhGm;
    AscendC::GlobalTensor<DTYPE_X> bihGm;
    AscendC::GlobalTensor<DTYPE_X> bhhGm;
    AscendC::GlobalTensor<DTYPE_X> h0Gm;
    AscendC::GlobalTensor<DTYPE_X> c0Gm;
    AscendC::GlobalTensor<DTYPE_X> yGm;
    AscendC::GlobalTensor<DTYPE_X> hnGm;
    AscendC::GlobalTensor<DTYPE_X> cnGm;

    uint32_t B, T, I, H;
    uint32_t tileNum;
    uint32_t batchPerBlock;
    uint32_t bStart;
};

extern "C" __global__ __aicore__ void lstm_custom(GM_ADDR x,
                                                  GM_ADDR w_ih, GM_ADDR w_hh,
                                                  GM_ADDR b_ih, GM_ADDR b_hh,
                                                  GM_ADDR h0, GM_ADDR c0,
                                                  GM_ADDR y, GM_ADDR hn, GM_ADDR cn,
                                                  GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLstm op;
    op.Init(x, w_ih, w_hh, b_ih, b_hh, h0, c0, y, hn, cn,
            tiling_data.batch, tiling_data.seqLen, tiling_data.inputSize, tiling_data.hiddenSize, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, hidden_size, num_layers, output_size, dropout=0.0):
        super().__init__()
        self.hidden_size = hidden_size
        self.num_layers = num_layers
        # Keep original PyTorch modules to hold parameters for convenience
        self.lstm = torch.nn.LSTM(input_size, hidden_size, num_layers, batch_first=True, dropout=dropout, bidirectional=False)
        self.fc = torch.nn.Linear(hidden_size, output_size)

    def forward(self, x, h0=None, c0=None):
        batch_size = x.size(0)
        device = x.device
        if h0 is None:
            h0 = torch.randn(self.num_layers, batch_size, self.hidden_size, device=device, dtype=x.dtype)
        if c0 is None:
            c0 = torch.randn(self.num_layers, batch_size, self.hidden_size, device=device, dtype=x.dtype)

        # Sequentially apply custom single-layer LSTM per layer
        y = x
        hn_list = []
        cn_list = []
        for layer in range(self.num_layers):
            w_ih = getattr(self.lstm, f"weight_ih_l{layer}")
            w_hh = getattr(self.lstm, f"weight_hh_l{layer}")
            b_ih = getattr(self.lstm, f"bias_ih_l{layer}")
            b_hh = getattr(self.lstm, f"bias_hh_l{layer}")
            h0_l = h0[layer]
            c0_l = c0[layer]
            y, hn_l, cn_l = custom_ops_lib.lstm_custom(y, w_ih, w_hh, b_ih, b_hh, h0_l, c0_l)
            hn_list.append(hn_l.unsqueeze(0))
            cn_list.append(cn_l.unsqueeze(0))

        # Final output: take last time step of top layer
        out = self.fc(y[:, -1, :])
        return out
'''
