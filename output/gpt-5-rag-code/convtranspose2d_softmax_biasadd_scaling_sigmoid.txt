kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1; // single buffer for per-group vector

// Fused kernel: softmax along channel (dim=1), then +bias(C,1,1), *scale, sigmoid.
// Input x is expected to be in NCHW layout. bias is length C (broadcasted over H,W,N).
class KernelConvT2dSoftmaxBiasScaleSigmoid {
public:
    __aicore__ inline KernelConvT2dSoftmaxBiasScaleSigmoid() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t groupNum, uint32_t /*tileNum*/)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->groupNum = groupNum;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, n * c * h * w);
        biasGm.SetGlobalBuffer((__gm__ DTYPE_BIAS *)bias, c);
        scaleGm.SetGlobalBuffer((__gm__ DTYPE_SCALE *)scale, 1);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, n * c * h * w);

        // local buffers: per-group channel vector
        uint32_t vecBytes = c * sizeof(DTYPE_X);
        pipe.InitBuffer(inQueueVec, BUFFER_NUM, vecBytes);
        pipe.InitBuffer(outQueueVec, BUFFER_NUM, vecBytes);

        // bias cache
        pipe.InitBuffer(biasQueue, 1, vecBytes);

        // scale scalar buffer
        pipe.InitBuffer(scaleQueue, 1, sizeof(DTYPE_SCALE));

        // preload bias and scale
        AscendC::LocalTensor<DTYPE_BIAS> biasLocal = biasQueue.AllocTensor<DTYPE_BIAS>();
        AscendC::DataCopy(biasLocal, biasGm, c);
        biasQueue.EnQue(biasLocal);

        AscendC::LocalTensor<DTYPE_SCALE> scaleLocal = scaleQueue.AllocTensor<DTYPE_SCALE>();
        AscendC::DataCopy(scaleLocal, scaleGm, 1);
        scaleQueue.EnQue(scaleLocal);
    }

    __aicore__ inline void Process()
    {
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t groupsPerBlock = groupNum / blocks;
        uint32_t start = groupsPerBlock * blockIdx;
        uint32_t end = (blockIdx == blocks - 1) ? groupNum : (start + groupsPerBlock);

        // Fetch cached bias and scale
        AscendC::LocalTensor<DTYPE_BIAS> biasLocal = biasQueue.DeQue<DTYPE_BIAS>();
        AscendC::LocalTensor<DTYPE_SCALE> scaleLocal = scaleQueue.DeQue<DTYPE_SCALE>();
        DTYPE_SCALE scaleVal = scaleLocal.GetValue(0);

        for (uint32_t g = start; g < end; ++g) {
            CopyInGroup(g);
            ComputeGroup(biasLocal, scaleVal);
            CopyOutGroup(g);
        }

        biasQueue.FreeTensor(biasLocal);
        scaleQueue.FreeTensor(scaleLocal);
    }

private:
    // Map group index to (n,h,w)
    __aicore__ inline void GroupIndexToNHW(uint32_t g, uint32_t &nIdx, uint32_t &hIdx, uint32_t &wIdx)
    {
        uint32_t hwSize = h * w;
        nIdx = g / hwSize;
        uint32_t rem = g % hwSize;
        hIdx = rem / w;
        wIdx = rem % w;
    }

    __aicore__ inline uint32_t Offset(uint32_t nIdx, uint32_t cIdx, uint32_t hIdx, uint32_t wIdx)
    {
        // NCHW contiguous indexing
        return ((nIdx * c + cIdx) * h + hIdx) * w + wIdx;
    }

    __aicore__ inline void CopyInGroup(uint32_t g)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueVec.AllocTensor<DTYPE_X>();
        uint32_t nIdx, hIdx, wIdx;
        GroupIndexToNHW(g, nIdx, hIdx, wIdx);

        // Gather along channels with stride H*W
        for (uint32_t ch = 0; ch < c; ++ch) {
            uint32_t off = Offset(nIdx, ch, hIdx, wIdx);
            // copy 1 element to xLocal[ch]
            AscendC::DataCopy(xLocal[ch], xGm[off], 1);
        }
        inQueueVec.EnQue(xLocal);
    }

    __aicore__ inline void ComputeGroup(AscendC::LocalTensor<DTYPE_BIAS> &biasLocal, DTYPE_SCALE scaleVal)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueVec.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueVec.AllocTensor<DTYPE_Y>();

        // Softmax along channel dimension
        // Step 1: find max
        DTYPE_X maxVal = xLocal.GetValue(0);
        for (uint32_t i = 1; i < c; ++i) {
            DTYPE_X v = xLocal.GetValue(i);
            maxVal = v > maxVal ? v : maxVal;
        }

        // Step 2: exp(x - max) and sum
        DTYPE_X sumVal = static_cast<DTYPE_X>(0);
        for (uint32_t i = 0; i < c; ++i) {
            DTYPE_X shifted = xLocal.GetValue(i) - maxVal;
            DTYPE_X ev = AscendC::Exp(shifted);
            yLocal.SetValue(i, ev);
            sumVal += ev;
        }

        // Step 3: normalize, add bias, scale, sigmoid
        for (uint32_t i = 0; i < c; ++i) {
            DTYPE_Y sm = static_cast<DTYPE_Y>(yLocal.GetValue(i) / sumVal);
            DTYPE_Y withBias = sm + static_cast<DTYPE_Y>(biasLocal.GetValue(i));
            DTYPE_Y scaled = withBias * static_cast<DTYPE_Y>(scaleVal);
            DTYPE_Y sig = AscendC::Sigmoid(scaled);
            yLocal.SetValue(i, sig);
        }

        outQueueVec.EnQue<DTYPE_Y>(yLocal);
        inQueueVec.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutGroup(uint32_t g)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueVec.DeQue<DTYPE_Y>();
        uint32_t nIdx, hIdx, wIdx;
        GroupIndexToNHW(g, nIdx, hIdx, wIdx);

        // Scatter back along channels
        for (uint32_t ch = 0; ch < c; ++ch) {
            uint32_t off = Offset(nIdx, ch, hIdx, wIdx);
            AscendC::DataCopy(yGm[off], yLocal[ch], 1);
        }
        outQueueVec.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueVec;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueVec;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scaleQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_BIAS> biasGm;
    AscendC::GlobalTensor<DTYPE_SCALE> scaleGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n;
    uint32_t c;
    uint32_t h;
    uint32_t w;
    uint32_t groupNum;
};

extern "C" __global__ __aicore__ void convtranspose2d_softmax_biasadd_scaling_sigmoid_custom(GM_ADDR x, GM_ADDR bias, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvT2dSoftmaxBiasScaleSigmoid op;
    op.Init(x, bias, scale, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.groupNum, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: keeps ConvTranspose2d on PyTorch, fuses softmax (dim=1), bias add, scaling and sigmoid into a custom AscendC op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, bias_shape, scaling_factor):
        super(ModelNew, self).__init__()
        self.conv_transpose = torch.nn.ConvTranspose2d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        self.bias = torch.nn.Parameter(torch.randn(bias_shape))
        self.scaling_factor = scaling_factor

    def forward(self, x):
        x = self.conv_transpose(x)
        scale_tensor = torch.tensor(self.scaling_factor, dtype=x.dtype, device=x.device)
        # Fused: softmax (dim=1) + bias add + scale + sigmoid
        x = custom_ops_lib.convtranspose2d_softmax_biasadd_scaling_sigmoid_custom(x, self.bias, scale_tensor)
        return x
