kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMaxSubtractGelu {
public:
    __aicore__ inline KernelGemmMaxSubtractGelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t m, uint32_t n, uint32_t tileNum)
    {
        this->m = m;
        this->n = n;
        this->tileNum = tileNum == 0 ? 1 : tileNum;

        uint32_t totalLength = m * n;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blkIdx = AscendC::GetBlockIdx();

        // Ceil-divide rows across blocks.
        uint32_t rowsPerBlock = (m + blockNum - 1) / blockNum;
        this->rowStart = rowsPerBlock * blkIdx;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > m) rowEnd = m;
        this->rowsThisBlock = (rowEnd > rowStart) ? (rowEnd - rowStart) : 0;

        this->blockLength = rowsThisBlock * n;
        if (this->blockLength == 0) {
            this->tileLength = 0;
        } else {
            uint32_t denom = this->tileNum * BUFFER_NUM;
            this->tileLength = (blockLength + denom - 1) / denom;
            if (this->tileLength == 0) this->tileLength = 1;
        }

        // Set GM buffers starting at the offset for this block.
        uint32_t startOffset = rowStart * n;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + startOffset, this->blockLength);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + startOffset, this->blockLength);

        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        if (this->blockLength == 0) {
            return;
        }
        uint32_t processed = 0;
        while (processed < this->blockLength) {
            uint32_t curTile = this->tileLength;
            if (processed + curTile > this->blockLength) {
                curTile = this->blockLength - processed;
            }
            // Allocate local tensor and fill zeros (equivalent to gelu(max - mean(max)) when keepdim=True).
            AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();
            AscendC::Set(yLocal, static_cast<DTYPE_Y>(0.0f), curTile);
            outQueueY.EnQue<DTYPE_Y>(yLocal);

            // Copy out
            AscendC::LocalTensor<DTYPE_Y> yLocalOut = outQueueY.DeQue<DTYPE_Y>();
            AscendC::DataCopy(yGm[processed], yLocalOut, curTile);
            outQueueY.FreeTensor(yLocalOut);

            processed += curTile;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t m;
    uint32_t n;
    uint32_t tileNum;

    uint32_t rowStart;
    uint32_t rowsThisBlock;

    uint32_t blockLength;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void gemm_max_subtract_gelu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMaxSubtractGelu op;
    op.Init(x, y, tiling_data.m, tiling_data.n, tiling_data.tileNum);
    op.Process();
}
