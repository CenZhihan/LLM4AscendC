kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConcatC {
public:
    __aicore__ inline KernelConcatC() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t c1, uint32_t c2, uint32_t hw, uint32_t tileNum)
    {
        this->n = n;
        this->c1 = c1;
        this->c2 = c2;
        this->hw = hw;
        this->tileNum = tileNum;

        this->perBatchA = c1 * hw;
        this->perBatchB = c2 * hw;
        this->perBatchY = (c1 + c2) * hw;

        // block-based batch split
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t base = n / blockNum;
        uint32_t rem = n % blockNum;
        this->startBatch = blockIdx * base + (blockIdx < rem ? blockIdx : rem);
        this->batchCount = base + (blockIdx < rem ? 1 : 0);
        if (this->batchCount == 0) {
            this->tileLength = 0;
            return;
        }

        uint32_t maxPerBatch = perBatchA > perBatchB ? perBatchA : perBatchB;
        uint32_t denom = this->tileNum * BUFFER_NUM;
        this->tileLength = maxPerBatch / denom;
        if (this->tileLength == 0) {
            this->tileLength = 1;
        }

        aGm.SetGlobalBuffer((__gm__ DTYPE_X *)a + (uint64_t)startBatch * (uint64_t)perBatchA,
                            (uint64_t)batchCount * (uint64_t)perBatchA);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y *)b + (uint64_t)startBatch * (uint64_t)perBatchB,
                            (uint64_t)batchCount * (uint64_t)perBatchB);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + (uint64_t)startBatch * (uint64_t)perBatchY,
                            (uint64_t)batchCount * (uint64_t)perBatchY);

        pipe.InitBuffer(inQueueA, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueB, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        if (this->batchCount == 0 || this->tileLength == 0) {
            return;
        }
        for (uint32_t bi = 0; bi < this->batchCount; ++bi) {
            // copy A part
            CopyConcateSegment(inQueueA, perBatchA, bi, 0, true);
            // copy B part (offset by perBatchA in output)
            CopyConcateSegment(inQueueB, perBatchB, bi, perBatchA, false);
        }
    }

private:
    template<typename Q>
    __aicore__ inline void CopyConcateSegment(Q& inQ, uint32_t segLen, uint32_t batchIdx, uint32_t yOffsetBase, bool useA)
    {
        uint32_t processed = 0;
        while (processed < segLen) {
            // CopyIn
            AscendC::LocalTensor<uint8_t> srcLocalU8 = inQ.AllocTensor<uint8_t>();
            uint32_t currLen = segLen - processed;
            if (currLen > this->tileLength) currLen = this->tileLength;

            if (useA) {
                AscendC::LocalTensor<DTYPE_X> srcLocal = srcLocalU8.ReinterpretCast<DTYPE_X>();
                AscendC::DataCopy(srcLocal, aGm[(uint64_t)batchIdx * (uint64_t)perBatchA + processed], currLen);
                inQ.EnQue<DTYPE_X>(srcLocal);
            } else {
                AscendC::LocalTensor<DTYPE_Y> srcLocal = srcLocalU8.ReinterpretCast<DTYPE_Y>();
                AscendC::DataCopy(srcLocal, bGm[(uint64_t)batchIdx * (uint64_t)perBatchB + processed], currLen);
                inQ.EnQue<DTYPE_Y>(srcLocal);
            }

            // Compute (pass-through)
            AscendC::LocalTensor<DTYPE_Z> yLocal = outQueueY.AllocTensor<DTYPE_Z>();
            if (useA) {
                AscendC::LocalTensor<DTYPE_X> srcLocal = inQ.DeQue<DTYPE_X>();
                AscendC::DataCopy(yLocal, srcLocal, currLen);
                inQ.FreeTensor(srcLocal);
            } else {
                AscendC::LocalTensor<DTYPE_Y> srcLocal = inQ.DeQue<DTYPE_Y>();
                AscendC::DataCopy(yLocal, srcLocal, currLen);
                inQ.FreeTensor(srcLocal);
            }
            outQueueY.EnQue<DTYPE_Z>(yLocal);

            // CopyOut
            AscendC::LocalTensor<DTYPE_Z> yLocal2 = outQueueY.DeQue<DTYPE_Z>();
            AscendC::DataCopy(yGm[(uint64_t)batchIdx * (uint64_t)perBatchY + yOffsetBase + processed], yLocal2, currLen);
            outQueueY.FreeTensor(yLocal2);

            processed += currLen;
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA, inQueueB;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> aGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t n;
    uint32_t c1;
    uint32_t c2;
    uint32_t hw;
    uint32_t tileNum;

    uint32_t perBatchA;
    uint32_t perBatchB;
    uint32_t perBatchY;

    uint32_t startBatch;
    uint32_t batchCount;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void densenet121_dense_block_custom(GM_ADDR a, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConcatC op;
    op.Init(a, b, y,
            tiling_data.n, tiling_data.c1, tiling_data.c2, tiling_data.hw, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_layers: int, num_input_features: int, growth_rate: int):
        super(ModelNew, self).__init__()
        layers = []
        for i in range(num_layers):
            layers.append(self._make_layer(num_input_features + i * growth_rate, growth_rate))
        self.layers = torch.nn.ModuleList(layers)

    def _make_layer(self, in_features: int, growth_rate: int):
        return torch.nn.Sequential(
            torch.nn.BatchNorm2d(in_features),
            torch.nn.ReLU(inplace=True),
            torch.nn.Conv2d(in_features, growth_rate, kernel_size=3, padding=1, bias=False),
            torch.nn.Dropout(0.0)
        )

    def forward(self, x):
        for layer in self.layers:
            new_feature = layer(x)
            x = custom_ops_lib.densenet121_dense_block_custom(x, new_feature)
        return x
'''
