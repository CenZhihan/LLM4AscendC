kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelHingeLoss {
public:
    __aicore__ inline KernelHingeLoss() : blockLength(0), tileNum(0), tileLength(0), sumAcc(0.f) {}
    __aicore__ inline void Init(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss, uint32_t totalLength, uint32_t tileNumIn)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNumIn;
        this->tileLength = (this->blockLength + (this->tileNum * BUFFER_NUM) - 1) / (this->tileNum * BUFFER_NUM); // ceil

        predsGm.SetGlobalBuffer((__gm__ DTYPE_X *)predictions + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        targsGm.SetGlobalBuffer((__gm__ DTYPE_Y *)targets + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        lossGm.SetGlobalBuffer((__gm__ DTYPE_Z *)loss, 1);

        pipe.InitBuffer(inQueuePreds, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(inQueueTargs, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Y));

        sumAcc = 0.f;
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            uint32_t offset = i * this->tileLength;
            if (offset >= this->blockLength) {
                break;
            }
            CopyIn(i);
            Compute(i);
        }
        float mean = 0.f;
        if (this->blockLength > 0) {
            mean = sumAcc / static_cast<float>(this->blockLength);
        }
        lossGm.SetValue(0, static_cast<DTYPE_Z>(mean));
    }

private:
    __aicore__ inline uint32_t CurrentLen(int32_t progress) const
    {
        uint32_t offset = progress * this->tileLength;
        uint32_t remain = (offset < this->blockLength) ? (this->blockLength - offset) : 0;
        return (remain > this->tileLength) ? this->tileLength : remain;
    }

    __aicore__ inline void CopyIn(int32_t progress)
    {
        uint32_t curLen = CurrentLen(progress);
        AscendC::LocalTensor<DTYPE_X> predsLocal = inQueuePreds.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> targsLocal = inQueueTargs.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(predsLocal, predsGm[progress * this->tileLength], curLen);
        AscendC::DataCopy(targsLocal, targsGm[progress * this->tileLength], curLen);
        inQueuePreds.EnQue(predsLocal);
        inQueueTargs.EnQue(targsLocal);
    }

    __aicore__ inline void Compute(int32_t progress)
    {
        uint32_t curLen = CurrentLen(progress);
        AscendC::LocalTensor<DTYPE_X> predsLocal = inQueuePreds.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> targsLocal = inQueueTargs.DeQue<DTYPE_Y>();

        for (uint32_t i = 0; i < curLen; ++i) {
            float p = static_cast<float>(predsLocal(i));
            float t = static_cast<float>(targsLocal(i));
            float v = 1.0f - p * t;
            if (v < 0.0f) v = 0.0f;
            sumAcc += v;
        }

        inQueuePreds.FreeTensor(predsLocal);
        inQueueTargs.FreeTensor(targsLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePreds, inQueueTargs;
    AscendC::GlobalTensor<DTYPE_X> predsGm;
    AscendC::GlobalTensor<DTYPE_Y> targsGm;
    AscendC::GlobalTensor<DTYPE_Z> lossGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    float sumAcc;
};

extern "C" __global__ __aicore__ void hinge_loss_custom(GM_ADDR predictions, GM_ADDR targets, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelHingeLoss op;
    op.Init(predictions, targets, loss, tiling_data.totalLength, tiling_data.tileNum);
    op.Process();
}
