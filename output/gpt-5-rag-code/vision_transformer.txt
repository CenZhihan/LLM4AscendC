kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelVisionTransformerCustom {
public:
    __aicore__ inline KernelVisionTransformerCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR cls, GM_ADDR pos, GM_ADDR y,
                                uint32_t B, uint32_t N, uint32_t D, uint32_t /*tileNum*/)
    {
        this->B = B;
        this->N = N;
        this->D = D;

        xGm.SetGlobalBuffer((__gm__ float*)x, (uint32_t)(B * N * D));
        clsGm.SetGlobalBuffer((__gm__ float*)cls, (uint32_t)D);
        posGm.SetGlobalBuffer((__gm__ float*)pos, (uint32_t)((N + 1) * D));
        yGm.SetGlobalBuffer((__gm__ float*)y, (uint32_t)(B * (N + 1) * D));

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->D * sizeof(float));
        pipe.InitBuffer(inQueuePos, BUFFER_NUM, this->D * sizeof(float));
        pipe.InitBuffer(inQueueCls, BUFFER_NUM, this->D * sizeof(float));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->D * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t b = 0; b < B; ++b) {
            // First token: y[b,0,:] = cls_token[0,0,:] + pos[0,0,:]
            {
                AscendC::LocalTensor<float> clsLocal = inQueueCls.AllocTensor<float>();
                AscendC::LocalTensor<float> posLocal = inQueuePos.AllocTensor<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

                AscendC::DataCopy(clsLocal, clsGm, D);
                AscendC::DataCopy(posLocal, posGm, D);
                AscendC::Add(yLocal, clsLocal, posLocal, D);

                outQueueY.EnQue<float>(yLocal);
                inQueueCls.EnQue<float>(clsLocal);
                inQueuePos.EnQue<float>(posLocal);

                AscendC::LocalTensor<float> yOut = outQueueY.DeQue<float>();
                AscendC::DataCopy(yGm[b * (N + 1) * D], yOut, D);
                outQueueY.FreeTensor(yOut);

                AscendC::LocalTensor<float> tmp1 = inQueueCls.DeQue<float>();
                AscendC::LocalTensor<float> tmp2 = inQueuePos.DeQue<float>();
                inQueueCls.FreeTensor(tmp1);
                inQueuePos.FreeTensor(tmp2);
            }

            // Remaining tokens: for t in [1..N]: y[b,t,:] = x[b,t-1,:] + pos[0,t,:]
            for (uint32_t t = 1; t <= N; ++t) {
                AscendC::LocalTensor<float> xLocal = inQueueX.AllocTensor<float>();
                AscendC::LocalTensor<float> posLocal = inQueuePos.AllocTensor<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY.AllocTensor<float>();

                uint32_t x_offset = b * N * D + (t - 1) * D;
                uint32_t p_offset = t * D;
                AscendC::DataCopy(xLocal, xGm[x_offset], D);
                AscendC::DataCopy(posLocal, posGm[p_offset], D);

                AscendC::Add(yLocal, xLocal, posLocal, D);

                outQueueY.EnQue<float>(yLocal);
                inQueueX.EnQue<float>(xLocal);
                inQueuePos.EnQue<float>(posLocal);

                AscendC::LocalTensor<float> yOut = outQueueY.DeQue<float>();
                AscendC::DataCopy(yGm[b * (N + 1) * D + t * D], yOut, D);
                outQueueY.FreeTensor(yOut);

                AscendC::LocalTensor<float> tmp1 = inQueueX.DeQue<float>();
                AscendC::LocalTensor<float> tmp2 = inQueuePos.DeQue<float>();
                inQueueX.FreeTensor(tmp1);
                inQueuePos.FreeTensor(tmp2);
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX, inQueuePos, inQueueCls;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> clsGm;
    AscendC::GlobalTensor<float> posGm;
    AscendC::GlobalTensor<float> yGm;

    uint32_t B;
    uint32_t N;
    uint32_t D;
};

extern "C" __global__ __aicore__ void vision_transformer_custom(GM_ADDR x, GM_ADDR cls_token, GM_ADDR pos_embedding, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelVisionTransformerCustom op;
    op.Init(x, cls_token, pos_embedding, y, tiling_data.B, tiling_data.N, tiling_data.D, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, image_size, patch_size, num_classes, dim, depth, heads, mlp_dim, channels=3, dropout=0.1, emb_dropout=0.1):
        super(ModelNew, self).__init__()
        assert image_size % patch_size == 0, "Image dimensions must be divisible by the patch size."
        num_patches = (image_size // patch_size) ** 2
        patch_dim = channels * patch_size ** 2

        self.patch_size = patch_size
        self.pos_embedding = torch.nn.Parameter(torch.randn(1, num_patches + 1, dim))
        self.patch_to_embedding = torch.nn.Linear(patch_dim, dim)
        self.cls_token = torch.nn.Parameter(torch.randn(1, 1, dim))
        self.dropout = torch.nn.Dropout(emb_dropout)

        self.transformer = torch.nn.TransformerEncoder(
            torch.nn.TransformerEncoderLayer(d_model=dim, nhead=heads, dim_feedforward=mlp_dim, dropout=dropout),
            num_layers=depth
        )

        self.to_cls_token = torch.nn.Identity()
        self.mlp_head = torch.nn.Sequential(
            torch.nn.Linear(dim, mlp_dim),
            torch.nn.GELU(),
            torch.nn.Dropout(dropout),
            torch.nn.Linear(mlp_dim, num_classes)
        )

    def forward(self, img):
        p = self.patch_size
        # Patchify
        x = img.unfold(2, p, p).unfold(3, p, p).reshape(img.shape[0], -1, p * p * img.shape[1])
        # Linear projection
        x = self.patch_to_embedding(x)
        # Custom kernel: concat cls token and add positional embedding
        x = custom_ops_lib.vision_transformer_custom(x, self.cls_token, self.pos_embedding)
        # Keep dropout behavior after embedding
        x = self.dropout(x)
        # Transformer encoder
        x = self.transformer(x)
        # CLS token head
        x = self.to_cls_token(x[:, 0])
        return self.mlp_head(x)
'''
