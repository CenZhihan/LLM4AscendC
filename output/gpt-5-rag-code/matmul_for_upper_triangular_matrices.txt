kernel_src="""
#include "kernel_operator.h"

class KernelMatmulTriU {
public:
    __aicore__ inline KernelMatmulTriU() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t n) {
        aGm = reinterpret_cast<__gm__ float*>(a);
        bGm = reinterpret_cast<__gm__ float*>(b);
        cGm = reinterpret_cast<__gm__ float*>(c);
        this->n = n;

        uint32_t block_num = AscendC::GetBlockNum();
        uint32_t block_idx = AscendC::GetBlockIdx();

        // Partition rows among blocks
        uint32_t rows_per_block = (n + block_num - 1) / block_num;
        rowStart = block_idx * rows_per_block;
        rowEnd = rowStart + rows_per_block;
        if (rowEnd > n) rowEnd = n;
    }

    __aicore__ inline void Process() {
        // For each row assigned to this block
        for (uint32_t i = rowStart; i < rowEnd; ++i) {
            // Zero lower triangle elements for this row
            for (uint32_t j = 0; j < i; ++j) {
                cGm[(uint64_t)i * n + j] = 0.0f;
            }
            // Compute upper triangle entries C[i, j] = sum_{k=i..j} A[i,k] * B[k,j]
            for (uint32_t j = i; j < n; ++j) {
                float acc = 0.0f;
                // Due to upper triangular structure:
                // A[i,k] != 0 only if k >= i; B[k,j] != 0 only if k <= j.
                for (uint32_t k = i; k <= j; ++k) {
                    acc += aGm[(uint64_t)i * n + k] * bGm[(uint64_t)k * n + j];
                }
                cGm[(uint64_t)i * n + j] = acc;
            }
        }
    }

private:
    __gm__ float* aGm;
    __gm__ float* bGm;
    __gm__ float* cGm;
    uint32_t n;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void matmul_for_upper_triangular_matrices_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulTriU op;
    op.Init(a, b, c, tiling_data.n());
    op.Process();
}
"""
