    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // tensor num for each queue

class KernelRmsprop {
public:
    __aicore__ inline KernelRmsprop() {}
    __aicore__ inline void Init(GM_ADDR param, GM_ADDR grad, GM_ADDR v,
                                GM_ADDR param_out, uint32_t totalLength, uint32_t tileNum,
                                float lr, float alpha, float eps)
    {
        this->blockLength = totalLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->tileLength = this->blockLength / tileNum / BUFFER_NUM;
        this->lr_ = lr;
        this->alpha_ = alpha;
        this->eps_ = eps;

        paramGm.SetGlobalBuffer((__gm__ DTYPE_X *)param + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        gradGm.SetGlobalBuffer((__gm__ DTYPE_X *)grad + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        vGm.SetGlobalBuffer((__gm__ DTYPE_X *)v + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);
        outGm.SetGlobalBuffer((__gm__ DTYPE_X *)param_out + this->blockLength * AscendC::GetBlockIdx(), this->blockLength);

        pipe.InitBuffer(qParam, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(qGrad, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(qV, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(qOut, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; i++) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> paramLocal = qParam.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> gradLocal = qGrad.AllocTensor<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> vLocal = qV.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(paramLocal, paramGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(gradLocal, gradGm[progress * this->tileLength], this->tileLength);
        AscendC::DataCopy(vLocal, vGm[progress * this->tileLength], this->tileLength);
        qParam.EnQue(paramLocal);
        qGrad.EnQue(gradLocal);
        qV.EnQue(vLocal);
    }

    __aicore__ inline void Compute(int32_t /*progress*/)
    {
        AscendC::LocalTensor<DTYPE_X> paramLocal = qParam.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> gradLocal = qGrad.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> vLocal = qV.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_X> outLocal = qOut.AllocTensor<DTYPE_X>();

        // z = grad^2 (reuse outLocal as temp)
        AscendC::Mul(outLocal, gradLocal, gradLocal, this->tileLength); // outLocal = g^2

        // v_new = alpha * v + (1 - alpha) * z
        AscendC::Muls(vLocal, vLocal, this->alpha_, this->tileLength);             // vLocal = alpha * v
        AscendC::Muls(outLocal, outLocal, (1.0f - this->alpha_), this->tileLength); // outLocal = (1 - alpha) * g^2
        AscendC::Add(vLocal, vLocal, outLocal, this->tileLength);                   // vLocal = v_new

        // outLocal = sqrt(v_new) + eps
        AscendC::Sqrt(outLocal, vLocal, this->tileLength);                          // outLocal = sqrt(v_new)
        AscendC::Adds(outLocal, outLocal, this->eps_, this->tileLength);            // outLocal = sqrt(v_new) + eps

        // outLocal = lr * grad / outLocal
        AscendC::Div(outLocal, gradLocal, outLocal, this->tileLength);              // outLocal = grad / (sqrt(v_new)+eps)
        AscendC::Muls(outLocal, outLocal, this->lr_, this->tileLength);             // outLocal = lr * grad / denom

        // outLocal = param - outLocal
        AscendC::Sub(outLocal, paramLocal, outLocal, this->tileLength);             // outLocal = param - update

        qOut.EnQue(outLocal);
        qParam.FreeTensor(paramLocal);
        qGrad.FreeTensor(gradLocal);
        qV.FreeTensor(vLocal);
    }

    __aicore__ inline void CopyOut(int32_t progress)
    {
        AscendC::LocalTensor<DTYPE_X> outLocal = qOut.DeQue<DTYPE_X>();
        AscendC::DataCopy(outGm[progress * this->tileLength], outLocal, this->tileLength);
        qOut.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qParam, qGrad, qV;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> qOut;
    AscendC::GlobalTensor<DTYPE_X> paramGm;
    AscendC::GlobalTensor<DTYPE_X> gradGm;
    AscendC::GlobalTensor<DTYPE_X> vGm;
    AscendC::GlobalTensor<DTYPE_X> outGm;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    float lr_;
    float alpha_;
    float eps_;
};

extern "C" __global__ __aicore__ void rmsprop_custom(GM_ADDR param, GM_ADDR grad, GM_ADDR v, GM_ADDR param_out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsprop op;
    op.Init(param, grad, v, param_out,
            tiling_data.totalLength, tiling_data.tileNum,
            tiling_data.lr, tiling_data.alpha, tiling_data.eps);
    op.Process();
}
