kernel_src="""
#include "kernel_operator.h"
#include <cmath>

extern "C" __global__ __aicore__ void convtranspose3d_relu_groupnorm_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR gamma, GM_ADDR beta, GM_ADDR groups, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N    = tiling_data.N();
    const uint32_t Cin  = tiling_data.Cin();
    const uint32_t Cout = tiling_data.Cout();
    const uint32_t D    = tiling_data.D();
    const uint32_t H    = tiling_data.H();
    const uint32_t W    = tiling_data.W();
    const uint32_t Kd   = tiling_data.Kd();
    const uint32_t Kh   = tiling_data.Kh();
    const uint32_t Kw   = tiling_data.Kw();
    const uint32_t Dout = tiling_data.Dout();
    const uint32_t Hout = tiling_data.Hout();
    const uint32_t Wout = tiling_data.Wout();

    __gm__ float* x_ptr      = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* w_ptr      = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* gamma_ptr  = reinterpret_cast<__gm__ float*>(gamma);
    __gm__ float* beta_ptr   = reinterpret_cast<__gm__ float*>(beta);
    __gm__ int32_t* g_ptr    = reinterpret_cast<__gm__ int32_t*>(groups);
    __gm__ float* y_ptr      = reinterpret_cast<__gm__ float*>(y);

    const float eps = 1.0e-5f;
    int32_t G = static_cast<int32_t>(g_ptr[0]);
    if (G <= 0) { G = 1; }
    // initialize output to zero
    uint64_t total = tiling_data.totalLength();
    for (uint64_t i = 0; i < total; ++i) {
        y_ptr[i] = 0.0f;
    }

    auto x_offset = [&](uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) -> uint64_t {
        return (((static_cast<uint64_t>(n) * Cin + c) * D + d) * H + h) * W + w;
    };
    auto y_offset = [&](uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) -> uint64_t {
        return (((static_cast<uint64_t>(n) * Cout + c) * Dout + d) * Hout + h) * Wout + w;
    };
    auto w_offset = [&](uint32_t ic, uint32_t oc, uint32_t kd, uint32_t kh, uint32_t kw) -> uint64_t {
        return ((((static_cast<uint64_t>(ic) * Cout + oc) * Kd + kd) * Kh + kh) * Kw + kw);
    };

    // Deconvolution (stride=1, padding=0, dilation=1): full convolution with flipped kernel
    for (uint32_t n = 0; n < N; ++n) {
        for (uint32_t oc = 0; oc < Cout; ++oc) {
            for (uint32_t od = 0; od < Dout; ++od) {
                for (uint32_t oh = 0; oh < Hout; ++oh) {
                    for (uint32_t ow = 0; ow < Wout; ++ow) {
                        float acc = 0.0f;
                        for (uint32_t ic = 0; ic < Cin; ++ic) {
                            for (uint32_t rkd = 0; rkd < Kd; ++rkd) {
                                int32_t id = static_cast<int32_t>(od) - static_cast<int32_t>(rkd);
                                if (id < 0 || id >= static_cast<int32_t>(D)) continue;
                                for (uint32_t rkh = 0; rkh < Kh; ++rkh) {
                                    int32_t ih = static_cast<int32_t>(oh) - static_cast<int32_t>(rkh);
                                    if (ih < 0 || ih >= static_cast<int32_t>(H)) continue;
                                    for (uint32_t rkw = 0; rkw < Kw; ++rkw) {
                                        int32_t iw = static_cast<int32_t>(ow) - static_cast<int32_t>(rkw);
                                        if (iw < 0 || iw >= static_cast<int32_t>(W)) continue;
                                        uint32_t fk = Kd - 1 - rkd;
                                        uint32_t fh = Kh - 1 - rkh;
                                        uint32_t fw = Kw - 1 - rkw;
                                        float xv = x_ptr[x_offset(n, ic, static_cast<uint32_t>(id), static_cast<uint32_t>(ih), static_cast<uint32_t>(iw))];
                                        float wv = w_ptr[w_offset(ic, oc, fk, fh, fw)];
                                        acc += xv * wv;
                                    }
                                }
                            }
                        }
                        // ReLU
                        if (acc < 0.0f) acc = 0.0f;
                        y_ptr[y_offset(n, oc, od, oh, ow)] = acc;
                    }
                }
            }
        }
    }

    // GroupNorm over y (after ReLU)
    uint32_t channels_per_group = Cout / static_cast<uint32_t>(G);
    if (channels_per_group == 0) { channels_per_group = 1; }
    uint64_t spatial = static_cast<uint64_t>(Dout) * Hout * Wout;
    for (uint32_t n = 0; n < N; ++n) {
        for (int32_t g = 0; g < G; ++g) {
            uint32_t c_start = static_cast<uint32_t>(g) * channels_per_group;
            uint32_t c_end = (g == G - 1) ? Cout : c_start + channels_per_group;
            // compute mean and variance
            double sum = 0.0;
            double sumsq = 0.0;
            uint64_t count = static_cast<uint64_t>(c_end - c_start) * spatial;
            for (uint32_t oc = c_start; oc < c_end; ++oc) {
                for (uint32_t od = 0; od < Dout; ++od) {
                    for (uint32_t oh = 0; oh < Hout; ++oh) {
                        for (uint32_t ow = 0; ow < Wout; ++ow) {
                            float v = y_ptr[y_offset(n, oc, od, oh, ow)];
                            sum += static_cast<double>(v);
                            sumsq += static_cast<double>(v) * static_cast<double>(v);
                        }
                    }
                }
            }
            float mean = static_cast<float>(sum / static_cast<double>(count));
            float var = static_cast<float>(sumsq / static_cast<double>(count) - static_cast<double>(mean) * static_cast<double>(mean));
            if (var < 0.0f) var = 0.0f;
            float rstd = 1.0f / sqrtf(var + eps);
            // normalize and affine
            for (uint32_t oc = c_start; oc < c_end; ++oc) {
                float gscale = gamma_ptr[oc];
                float gbias  = beta_ptr[oc];
                for (uint32_t od = 0; od < Dout; ++od) {
                    for (uint32_t oh = 0; oh < Hout; ++oh) {
                        for (uint32_t ow = 0; ow < Wout; ++ow) {
                            uint64_t idx = y_offset(n, oc, od, oh, ow);
                            float v = y_ptr[idx];
                            float vn = (v - mean) * rstd;
                            y_ptr[idx] = vn * gscale + gbias;
                        }
                    }
                }
            }
        }
    }
}

at::Tensor convtranspose3d_relu_groupnorm_custom_impl_npu(const at::Tensor& x,
                                                          const at::Tensor& weight,
                                                          const at::Tensor& gamma,
                                                          const at::Tensor& beta,
                                                          int64_t groups) {
    TORCH_CHECK(x.dim() == 5, "x must be 5D NCDHW");
    TORCH_CHECK(weight.dim() == 5, "weight must be 5D [Cin, Cout, Kd, Kh, Kw]");
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto D = x.size(2);
    auto H = x.size(3);
    auto W = x.size(4);

    auto Kd = weight.size(2);
    auto Kh = weight.size(3);
    auto Kw = weight.size(4);
    auto Cout = weight.size(1);

    int64_t Dout = D + Kd - 1;
    int64_t Hout = H + Kh - 1;
    int64_t Wout = W + Kw - 1;

    at::Tensor result = at::empty({N, Cout, Dout, Hout, Wout}, x.options());

    at::Tensor groups_tensor = make_groups_tensor(groups, x.device());

    EXEC_NPU_CMD(aclnnConvtranspose3dReluGroupnormCustom, x, weight, gamma, beta, groups_tensor, result);
    return result;
}
