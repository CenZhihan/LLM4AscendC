kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // double-buffering for vec in/out

// Compute final state:
// out[b, h, p, n] = sum_{c=0..C-1} states[b, c, h, p, n] * decay_chunk[b, h, Z-1, c]
class KernelMambaReturnFinalStateCustom {
public:
    __aicore__ inline KernelMambaReturnFinalStateCustom() {}
    __aicore__ inline void Init(GM_ADDR decay, GM_ADDR states, GM_ADDR out, uint32_t b, uint32_t h, uint32_t z,
                                uint32_t c, uint32_t p, uint32_t n, uint32_t q, uint32_t tileNum)
    {
        this->B = b;
        this->H = h;
        this->Z = z;
        this->C = c;
        this->P = p;
        this->N = n;
        this->Q = q;
        this->tileNum = tileNum;

        // Partition BH groups across blocks
        this->numBH = B * H;
        this->blockNum = AscendC::GetBlockNum();
        this->blockIdx = AscendC::GetBlockIdx();
        this->bhPerBlock = (numBH + blockNum - 1) / blockNum;
        this->bhStart = this->blockIdx * this->bhPerBlock;
        this->bhEnd = AscendC::Min(this->bhStart + this->bhPerBlock, this->numBH);

        // strides in elements
        stride_states_N = 1;
        stride_states_P = N;
        stride_states_H = P * N;
        stride_states_C = H * P * N;
        stride_states_B = C * H * P * N;

        stride_decay_C = 1;
        stride_decay_Z = C;
        stride_decay_H = Z * C;
        stride_decay_B = H * Z * C;

        stride_out_N = 1;
        stride_out_P = N;
        stride_out_H = P * N;
        stride_out_B = H * P * N;

        // set GM buffers
        decayGm.SetGlobalBuffer((__gm__ DTYPE_X*)decay, (uint64_t)B * H * Z * C);
        statesGm.SetGlobalBuffer((__gm__ DTYPE_Y*)states, (uint64_t)B * C * H * P * N);
        outGm.SetGlobalBuffer((__gm__ DTYPE_Z*)out, (uint64_t)B * H * P * N);

        // tile config along Q dimension
        this->tileLen = AscendC::Max((uint32_t)1, (uint32_t)(Q / tileNum / BUFFER_NUM));
        if (this->tileLen == 0) { this->tileLen = 1; }

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Y));
        pipe.InitBuffer(outQueue, BUFFER_NUM, this->tileLen * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t bh = this->bhStart; bh < this->bhEnd; ++bh) {
            uint32_t b = bh / H;
            uint32_t h = bh % H;

            // base offsets
            uint64_t states_base_bh = (uint64_t)b * stride_states_B + (uint64_t)h * stride_states_H;
            uint64_t decay_base_bh_lastz = (uint64_t)b * stride_decay_B + (uint64_t)h * stride_decay_H + (uint64_t)(Z - 1) * stride_decay_Z;
            uint64_t out_base_bh = (uint64_t)b * stride_out_B + (uint64_t)h * stride_out_H;

            // tile over Q = P*N
            uint32_t processed = 0;
            while (processed < Q) {
                uint32_t remain = Q - processed;
                uint32_t currTile = remain < this->tileLen ? remain : this->tileLen;

                // Accumulator buffer for this (b,h) tile
                AscendC::LocalTensor<DTYPE_Z> accLocal = outQueue.AllocTensor<DTYPE_Z>();

                // c loop: accumulate weighted states into accLocal
                for (uint32_t ci = 0; ci < C; ++ci) {
                    // load states tile for this c
                    AscendC::LocalTensor<DTYPE_Y> stLocal = inQueue.AllocTensor<DTYPE_Y>();
                    uint64_t st_offset = states_base_bh + (uint64_t)ci * stride_states_C + (uint64_t)processed;
                    AscendC::DataCopy(stLocal, statesGm[st_offset], currTile);

                    // read decay scalar weight
                    DTYPE_X weight = decayGm[decay_base_bh_lastz + (uint64_t)ci * stride_decay_C];

                    if (ci == 0) {
                        // accLocal = stLocal * weight
                        AscendC::Muls(accLocal, stLocal, (DTYPE_X)weight, currTile);
                    } else {
                        // stLocal = stLocal * weight; accLocal += stLocal
                        AscendC::Muls(stLocal, stLocal, (DTYPE_X)weight, currTile);
                        AscendC::Add(accLocal, accLocal, stLocal, currTile);
                    }
                    inQueue.FreeTensor(stLocal);
                }

                outQueue.EnQue(accLocal);

                // write back
                AscendC::LocalTensor<DTYPE_Z> outLocal = outQueue.DeQue<DTYPE_Z>();
                AscendC::DataCopy(outGm[out_base_bh + (uint64_t)processed], outLocal, currTile);
                outQueue.FreeTensor(outLocal);

                processed += currTile;
            }
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<DTYPE_X> decayGm;   // [B,H,Z,C]
    AscendC::GlobalTensor<DTYPE_Y> statesGm;  // [B,C,H,P,N]
    AscendC::GlobalTensor<DTYPE_Z> outGm;     // [B,H,P,N]

    // dims
    uint32_t B, H, Z, C, P, N, Q, tileNum;

    // BH partition
    uint32_t numBH, blockNum, blockIdx, bhPerBlock, bhStart, bhEnd;

    // strides
    uint64_t stride_states_N, stride_states_P, stride_states_H, stride_states_C, stride_states_B;
    uint64_t stride_decay_C, stride_decay_Z, stride_decay_H, stride_decay_B;
    uint64_t stride_out_N, stride_out_P, stride_out_H, stride_out_B;

    // tile
    uint32_t tileLen;
};

extern "C" __global__ __aicore__ void mamba_return_final_state_custom(GM_ADDR decay_chunk, GM_ADDR states, GM_ADDR out, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMambaReturnFinalStateCustom op;
    op.Init(decay_chunk, states, out,
            tiling_data.b, tiling_data.h, tiling_data.z, tiling_data.c,
            tiling_data.p, tiling_data.n, tiling_data.q, tiling_data.tileNum);
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
from einops import rearrange
import torch_npu
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, batch_size, seq_length, n_heads, d_head, d_state, block_len=64):
        super(ModelNew, self).__init__()
        assert seq_length % block_len == 0, "Sequence length must be divisible by block length"
        self.batch_size = batch_size
        self.seq_length = seq_length
        self.n_heads = n_heads
        self.d_head = d_head
        self.d_state = d_state
        self.block_len = block_len

        self.A = nn.Parameter(torch.randn(batch_size, seq_length, n_heads))
        self.B = nn.Parameter(torch.randn(batch_size, seq_length, n_heads, d_state))
        self.C = nn.Parameter(torch.randn(batch_size, seq_length, n_heads, d_state))

    def segsum(self, x):
        T = x.size(-1)
        x_cumsum = torch.cumsum(x, dim=-1)
        x_segsum = x_cumsum[..., :, None] - x_cumsum[..., None, :]
        mask = torch.tril(torch.ones(T, T, device=x.device, dtype=bool), diagonal=0)
        x_segsum = x_segsum.masked_fill(~mask, -torch.inf)
        return x_segsum

    def forward(self, X, initial_states=None):
        # Blockify
        X_blocks, A_blocks, B_blocks, C_blocks = [
            rearrange(x, "b (c l) ... -> b c l ...", l=self.block_len)
            for x in (X, self.A, self.B, self.C)
        ]
        A_blocks = rearrange(A_blocks, "b c l h -> b h c l")
        A_cumsum = torch.cumsum(A_blocks, dim=-1)

        # 1. Diagonal block outputs (kept as in original for correctness)
        L = torch.exp(self.segsum(A_blocks))  # [b,h,c,l,l]
        _ = torch.einsum("bclhn,bcshn,bhcls,bcshp->bclhp",
                         C_blocks, B_blocks, L, X_blocks)

        # 2. Intra-chunk states
        decay_states = torch.exp((A_cumsum[:, :, :, -1:] - A_cumsum))  # [b,h,c,l]
        states = torch.einsum("bclhn,bhcl,bclhp->bchpn",
                              B_blocks, decay_states, X_blocks)  # [b,c,h,p,n]

        # 3. Inter-chunk with custom kernel for final state
        if initial_states is None:
            initial_states = torch.zeros_like(states[:, :1])
        states = torch.cat([initial_states, states], dim=1)  # [b,c+1,h,p,n] (c includes the prepended state)

        decay_chunk = torch.exp(self.segsum(F.pad(A_cumsum[:, :, :, -1], (1, 0))))  # [b,h,c+1,c+1]

        # Custom kernel computes: final_state = (decay_chunk @ states) selecting last z row, reducing over c
        final_state = custom_ops_lib.mamba_return_final_state_custom(decay_chunk, states)  # [b,h,p,n]
        return final_state
'''
