    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"
#include <cmath>
#include <cfloat>

constexpr int32_t BUFFER_NUM = 1; // one row per buffer

class KernelLogSoftmax {
public:
    __aicore__ inline KernelLogSoftmax() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalRows, uint32_t rowSize)
    {
        this->totalRows = totalRows;
        this->rowSize = rowSize;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Split rows by blocks evenly (simple partition)
        this->rowsPerBlock = totalRows / blocks;
        this->rowStart = this->rowsPerBlock * blockIdx;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->rowStart * this->rowSize, this->rowsPerBlock * this->rowSize);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y*)y + this->rowStart * this->rowSize, this->rowsPerBlock * this->rowSize);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->rowSize * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueY, BUFFER_NUM, this->rowSize * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < this->rowsPerBlock; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->rowSize], this->rowSize);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.AllocTensor<DTYPE_Y>();

        // Compute max for numerical stability
        float maxVal = -FLT_MAX;
        for (uint32_t i = 0; i < this->rowSize; ++i) {
            float v = static_cast<float>(xLocal.GetValue(i));
            if (v > maxVal) maxVal = v;
        }

        // Compute sum(exp(x - max))
        float sumExp = 0.0f;
        for (uint32_t i = 0; i < this->rowSize; ++i) {
            float v = static_cast<float>(xLocal.GetValue(i));
            float e = expf(v - maxVal);
            sumExp += e;
        }
        float logSumExp = logf(sumExp);

        // y = x - max - logSumExp
        for (uint32_t i = 0; i < this->rowSize; ++i) {
            float v = static_cast<float>(xLocal.GetValue(i));
            float outv = v - maxVal - logSumExp;
            yLocal.SetValue(i, static_cast<DTYPE_Y>(outv));
        }

        outQueueY.EnQue<DTYPE_Y>(yLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueueY.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[rowIdx * this->rowSize], yLocal, this->rowSize);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t totalRows;
    uint32_t rowSize;
    uint32_t rowsPerBlock;
    uint32_t rowStart;
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmax op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.rowSize);
    op.Process();
}
"""

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

at::Tensor log_softmax_custom_impl_npu(const at::Tensor& self, int64_t dim) {
    // Only dim=1 is supported in this custom kernel
    TORCH_CHECK(dim == 1 || dim == -1, "log_softmax_custom supports dim=1 (last dim) only");
    at::Tensor result = at::empty_like(self);
    EXEC_NPU_CMD(aclnnLogSoftmaxCustom, self, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("log_softmax_custom", &log_softmax_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("log_softmax_custom", &log_softmax_custom_impl_npu, "log_softmax along dim=1");
}
