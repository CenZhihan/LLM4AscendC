host_operator_src="""
#include "argmin_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline int64_t ProductExcept(const std::vector<int64_t>& shape, size_t except_idx) {
    int64_t prod = 1;
    for (size_t i = 0; i < shape.size(); ++i) {
        if ((int)i == (int)except_idx) continue;
        prod *= shape[i];
    }
    return prod;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    ArgminOverADimensionCustomTilingData tiling;
    // Get input shape
    const gert::Shape* in_shape = context->GetInputShape(0);
    std::vector<int64_t> shape_vec = in_shape->GetOriginShape().GetDims();
    int32_t rank = static_cast<int32_t>(shape_vec.size());
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }

    // Get attribute dim
    int64_t dim_attr = 0;
    (void)context->GetAttr("dim", dim_attr);
    int32_t dim = static_cast<int32_t>(dim_attr);
    // normalize to [0, rank)
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    uint64_t totalLength = in_shape->GetOriginShape().GetShapeSize();
    uint32_t reduceK = static_cast<uint32_t>(shape_vec[dim]);

    // In kernel we will reduce along last dim, so we plan based on K along last dim after movedim in model.
    // We therefore set axis to -1 to indicate last dim is reduced.
    int32_t axis = -1;

    // outerSize is product of all dims except the reduced one (after moving reduced dim to last)
    uint64_t outerSize = static_cast<uint64_t>(ProductExcept(shape_vec, dim));

    // UB tile size along K
    const uint32_t TILE_NUM = 1;
    // choose a moderate tileK: multiple of 256 up to 4096, but not exceed reduceK
    uint32_t tileK = reduceK;
    if (tileK > 4096u) tileK = 4096u;
    // round down to multiple of 256 if possible
    if (tileK >= 256u) tileK = (tileK / 256u) * 256u;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_outerSize(static_cast<uint32_t>(outerSize));
    tiling.set_reduceK(reduceK);
    tiling.set_axis(axis);
    tiling.set_tileK(tileK == 0 ? reduceK : tileK);
    tiling.set_tileNum(TILE_NUM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    std::vector<int64_t> in_dims = x_shape->GetOriginShape().GetDims();
    int32_t rank = static_cast<int32_t>(in_dims.size());
    if (rank <= 0) {
        return GRAPH_FAILED;
    }
    int64_t dim_attr = 0;
    (void)context->GetAttr("dim", dim_attr);
    int32_t dim = static_cast<int32_t>(dim_attr);
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    std::vector<int64_t> out_dims;
    out_dims.reserve(rank - 1);
    for (int i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out_dims.push_back(in_dims[i]);
    }

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

using DTYPE_X = float;
using DTYPE_IDX = int64_t;

constexpr int32_t BUFFER_NUM = 1;

class KernelArgminLastDim {
public:
    __aicore__ inline KernelArgminLastDim() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR idx, uint32_t outerSize, uint32_t reduceK, uint32_t tileK)
    {
        this->outerSize = outerSize;
        this->reduceK = reduceK;
        this->tileK = (tileK == 0 || tileK > reduceK) ? reduceK : tileK;

        // Each block handles a contiguous range of outer groups
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockId = AscendC::GetBlockIdx();
        uint32_t baseOuterPerBlock = outerSize / blocks;
        uint32_t rem = outerSize % blocks;
        this->outerStart = blockId * baseOuterPerBlock + (blockId < rem ? blockId : rem);
        this->outerCount = baseOuterPerBlock + (blockId < rem ? 1u : 0u);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + this->outerStart * this->reduceK, this->outerCount * this->reduceK);
        idxGm.SetGlobalBuffer((__gm__ DTYPE_IDX*)idx + this->outerStart, this->outerCount);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileK * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t g = 0; g < this->outerCount; ++g) {
            ComputeOneGroup(g);
        }
    }

private:
    __aicore__ inline void ComputeOneGroup(uint32_t g)
    {
        // We reduce across contiguous K values for this group
        uint32_t K = this->reduceK;
        DTYPE_X currentMin = AscendC::numeric_limits<DTYPE_X>::infinity();
        uint32_t currentIdx = 0;

        uint32_t processed = 0;
        uint32_t baseOffset = g * K;

        while (processed < K) {
            uint32_t curLen = AscendC::min(this->tileK, K - processed);
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[baseOffset + processed], curLen);
            inQueueX.EnQue(xLocal);

            xLocal = inQueueX.DeQue<DTYPE_X>();
            // Iterate over local tile to update min and index
            for (uint32_t i = 0; i < curLen; ++i) {
                DTYPE_X val = xLocal.GetValue(i);
                if (val < currentMin) {
                    currentMin = val;
                    currentIdx = processed + i;
                }
            }
            inQueueX.FreeTensor(xLocal);
            processed += curLen;
        }

        // write index as int64
        DTYPE_IDX outVal = static_cast<DTYPE_IDX>(currentIdx);
        // store result
        AscendC::LocalTensor<DTYPE_IDX> outLocal;
        // use a small on-the-fly store via vector out queue
        // directly write to GM
        idxGm.SetValue(g, outVal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_IDX> idxGm;

    uint32_t outerSize;
    uint32_t reduceK;
    uint32_t tileK;

    uint32_t outerStart;
    uint32_t outerCount;
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR idx, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgminLastDim op;
    op.Init(x, idx, tiling_data.outerSize, tiling_data.reduceK, tiling_data.tileK);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Argmin using a custom AscendC kernel that reduces along the last dimension.
    For arbitrary dim, the tensor is moved so that dim becomes the last dimension,
    then reduced, matching torch.argmin(..., keepdim=False).
    """
    def __init__(self, dim: int):
        super().__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        rank = x.dim()
        dim = self.dim
        if dim < 0:
            dim += rank
        if dim < 0:
            dim = 0
        if dim >= rank:
            dim = rank - 1

        # Move target dim to last for kernel
        if dim != rank - 1:
            y = x.movedim(dim, -1).contiguous()
            # Kernel expects dim attribute but internally reduces last dim; pass -1
            out = custom_ops_lib.argmin_over_a_dimension_custom(y, -1)
            return out
        else:
            y = x.contiguous()
            out = custom_ops_lib.argmin_over_a_dimension_custom(y, -1)
            return out
'''
