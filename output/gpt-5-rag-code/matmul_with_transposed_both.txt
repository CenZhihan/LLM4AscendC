namespace {
inline uint64_t GetDimSafe(const gert::Shape* shape, size_t idx) {
    if (shape->GetOriginShape().GetDimNum() <= idx) return 1;
    return static_cast<uint64_t>(shape->GetOriginShape().GetDim(idx));
}
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    // Expecting inputs to already be transposed: a: [M, K], b: [K, N]
    uint64_t M = GetDimSafe(a_shape, 0);
    uint64_t K_a = GetDimSafe(a_shape, 1);
    uint64_t K_b = GetDimSafe(b_shape, 0);
    uint64_t N = GetDimSafe(b_shape, 1);

    if (K_a != K_b) {
        // Basic validation: K dims must match
        return ge::GRAPH_FAILED;
    }

    const uint32_t BLOCK_DIM = 1; // compute whole GEMM in one block to keep simple and robust

    MatmulWithTransposedBothCustomTilingData tiling;
    tiling.set_M(static_cast<uint32_t>(M));
    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_K(static_cast<uint32_t>(K_a));
    tiling.set_blockDim(BLOCK_DIM);

    context->SetBlockDim(BLOCK_DIM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    uint64_t M = a_shape->GetOriginShape().GetDim(0);
    uint64_t N = b_shape->GetOriginShape().GetDim(1);

    gert::Shape* c_shape = context->GetOutputShape(0);
    gert::Dims dims;
    dims.SetDimNum(2);
    dims.SetDim(0, M);
    dims.SetDim(1, N);
    c_shape->SetOriginShape(dims);
    c_shape->SetStorageShape(dims);
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

using dtype = float;

struct MatmulTiling {
    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t blockDim;
};

constexpr int32_t BUFFER_NUM = 2;

// A minimal GEMM kernel wrapper leveraging AscendC Matmul API
class KernelMatmul {
public:
    __aicore__ inline KernelMatmul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, const MatmulTiling& t) {
        M = t.M;
        N = t.N;
        K = t.K;

        aGm.SetGlobalBuffer((__gm__ dtype *)a, static_cast<uint64_t>(M) * K);
        bGm.SetGlobalBuffer((__gm__ dtype *)b, static_cast<uint64_t>(K) * N);
        cGm.SetGlobalBuffer((__gm__ dtype *)c, static_cast<uint64_t>(M) * N);
    }

    __aicore__ inline void Process() {
        // Use Matmul API
        using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype>;
        using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype>;
        using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype>;
        using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, dtype>;

        // Create pipe and register matmul object
        AscendC::TPipe pipe;
        // Set a default TCubeTiling for API registration (share full L0/L1 to let API handle blocking)
        TCubeTiling apiTiling;
        apiTiling.shareMode = 0;
        apiTiling.shareL1Size = FULL_L1_SIZE;
        apiTiling.shareL0CSize = FULL_L0C_SIZE;
        apiTiling.shareUbSize = 0;

        AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> matmulObj;
        REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulObj, &apiTiling);

        // Set input/output tensors
        matmulObj.SetTensorA(aGm, false); // inputs are already transposed outside; treat as normal A[M, K]
        matmulObj.SetTensorB(bGm, false); // B[K, N]
        matmulObj.IterateBatch(cGm, 1, 1, false);
    }

private:
    AscendC::GlobalTensor<dtype> aGm;
    AscendC::GlobalTensor<dtype> bGm;
    AscendC::GlobalTensor<dtype> cGm;
    uint32_t M;
    uint32_t N;
    uint32_t K;
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    MatmulTiling tConf;
    tConf.M = tiling_data.M;
    tConf.N = tiling_data.N;
    tConf.K = tiling_data.K;
    tConf.blockDim = tiling_data.blockDim;

    KernelMatmul op;
    op.Init(a, b, c, tConf);
    op.Process();
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>
#include "pytorch_npu_helper.hpp"
#include <torch/extension.h>

static inline at::Tensor ensure_contiguous_if_needed(const at::Tensor& t) {
    if (t.is_contiguous()) return t;
    return t.contiguous();
}

at::Tensor matmul_with_transposed_both_custom_impl_npu(const at::Tensor& a, const at::Tensor& b) {
    // Expecting a: [M, K], b: [K, N]
    at::Tensor a_c = ensure_contiguous_if_needed(a);
    at::Tensor b_c = ensure_contiguous_if_needed(b);
    TORCH_CHECK(a_c.dim() == 2 && b_c.dim() == 2, "Inputs must be rank-2 tensors");
    TORCH_CHECK(a_c.size(1) == b_c.size(0), "Inner dimensions must match for matmul");
    at::Tensor result = at::empty({a_c.size(0), b_c.size(1)}, a_c.options());
    EXEC_NPU_CMD(aclnnMatmulWithTransposedBothCustom, a_c, b_c, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_with_transposed_both_custom", &matmul_with_transposed_both_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_with_transposed_both_custom", &matmul_with_transposed_both_custom_impl_npu, "MatMul with both inputs transposed outside");
