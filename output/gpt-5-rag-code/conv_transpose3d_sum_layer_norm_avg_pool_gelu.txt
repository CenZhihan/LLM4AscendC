kernel_src="""
#include "kernel_operator.h"

constexpr uint32_t BUFFER_NUM = 1; // streaming voxel-by-voxel
constexpr float GELU_COEF_A = 0.7978845608028654f; // sqrt(2/pi)
constexpr float GELU_COEF_B = 0.044715f;

class KernelFused {
public:
    __aicore__ inline KernelFused() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR sumw, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t N, uint32_t D, uint32_t H, uint32_t W, uint32_t C,
                                uint32_t Do, uint32_t Ho, uint32_t Wo,
                                float eps, uint32_t poolK)
    {
        this->N = N; this->D = D; this->H = H; this->W = W; this->C = C;
        this->Do = Do; this->Ho = Ho; this->Wo = Wo;
        this->eps = eps; this->poolK = poolK;

        voxelsIn = N * D * H * W;
        voxelsOut = N * Do * Ho * Wo;

        // Partition on output voxels across blocks
        this->blockVoxelOut = (voxelsOut + AscendC::GetBlockNum() - 1) / AscendC::GetBlockNum();
        uint32_t start = AscendC::GetBlockIdx() * this->blockVoxelOut;
        uint32_t end = AscendC::Min(start + this->blockVoxelOut, voxelsOut);
        this->startVoxelOut = start;
        this->endVoxelOut = end;

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, (uint32_t)(N * D * H * W * C));
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y, (uint32_t)(N * Do * Ho * Wo * C));
        gammaGm.SetGlobalBuffer((__gm__ DTYPE_X *)gamma, C);
        betaGm.SetGlobalBuffer((__gm__ DTYPE_X *)beta, C);
        sumwGm.SetGlobalBuffer((__gm__ DTYPE_X *)sumw, 1);

        pipe.InitBuffer(inQue, BUFFER_NUM, C * sizeof(DTYPE_X));
        pipe.InitBuffer(outQue, BUFFER_NUM, C * sizeof(DTYPE_Y));

        // load scalar sum weight and gamma/beta once
        DTYPE_X sumv;
        AscendC::DataCopy(&sumv, sumwGm.GetPhyAddr(), (uint32_t)1);
        this->sumWeight = sumv;

        gammaLocal = pipe.AllocWorkspace<DTYPE_X>(C);
        betaLocal = pipe.AllocWorkspace<DTYPE_X>(C);
        AscendC::DataCopy(gammaLocal, gammaGm, C);
        AscendC::DataCopy(betaLocal, betaGm, C);
    }

    __aicore__ inline void Process()
    {
        for (uint32_t outIdx = startVoxelOut; outIdx < endVoxelOut; ++outIdx) {
            // decode out index -> n, do, ho, wo
            uint32_t n, do_, ho_, wo_;
            indexToCoordsOut(outIdx, n, do_, ho_, wo_);

            uint32_t di = do_ * poolK;
            uint32_t hi = ho_ * poolK;
            uint32_t wi = wo_ * poolK;

            // accumulator for 8 neighbors
            AscendC::LocalTensor<DTYPE_Y> acc = outQue.AllocTensor<DTYPE_Y>();
            zeroVec(acc, C);

            for (uint32_t kd = 0; kd < poolK; ++kd) {
                for (uint32_t kh = 0; kh < poolK; ++kh) {
                    for (uint32_t kw = 0; kw < poolK; ++kw) {
                        uint32_t d = di + kd;
                        uint32_t h = hi + kh;
                        uint32_t w = wi + kw;
                        // load one voxel vector of length C (NDHWC contiguous on C)
                        AscendC::LocalTensor<DTYPE_X> v = inQue.AllocTensor<DTYPE_X>();
                        loadVecNDHWC(v, n, d, h, w);

                        // add scalar sum_weight
                        addScalar(v, sumWeight, C);

                        // layer norm across channel dimension (last dim)
                        // compute mean and var
                        float mean = 0.f, var = 0.f;
                        computeMeanVar(v, mean, var, C);
                        float invStd = AscendC::rsqrt(var + eps);

                        // normalize + affine: (v-mean)*invStd*gamma + beta
                        normAffine(v, mean, invStd, C);

                        // accumulate
                        vecAddInplace(acc, v, C);

                        inQue.FreeTensor(v);
                    }
                }
            }

            // average pooling (divide by poolK^3)
            float scale = 1.0f / float(poolK * poolK * poolK);
            mulScalarInplace(acc, scale, C);

            // GELU activation (approximate)
            geluInplace(acc, C);

            // store to output at (n, do_, ho_, wo_)
            storeVecNDHWC(acc, n, do_, ho_, wo_);

            outQue.FreeTensor(acc);
        }
    }

private:
    __aicore__ inline void indexToCoordsOut(uint32_t idx, uint32_t &n, uint32_t &d, uint32_t &h, uint32_t &w)
    {
        uint32_t wh = Wo;
        uint32_t dh = Ho * wh;
        uint32_t nh = Do * dh;
        n = idx / nh;
        uint32_t rem = idx % nh;
        d = rem / dh;
        rem = rem % dh;
        h = rem / wh;
        w = rem % wh;
    }

    __aicore__ inline uint64_t offsetNDHWC(uint32_t n, uint32_t d, uint32_t h, uint32_t w, uint32_t c)
    {
        // linear index for NDHWC
        return (((uint64_t)n * D + d) * H + h) * W * (uint64_t)C + (uint64_t)w * C + c;
    }

    __aicore__ inline void loadVecNDHWC(AscendC::LocalTensor<DTYPE_X> &dst, uint32_t n, uint32_t d, uint32_t h, uint32_t w)
    {
        // contiguous C
        uint64_t base = offsetNDHWC(n, d, h, w, 0);
        AscendC::DataCopy(dst, xGm[base], C);
    }

    __aicore__ inline void storeVecNDHWC(AscendC::LocalTensor<DTYPE_Y> &src, uint32_t n, uint32_t d, uint32_t h, uint32_t w)
    {
        uint64_t base = (((uint64_t)n * Do + d) * Ho + h) * Wo * (uint64_t)C + (uint64_t)w * C;
        AscendC::DataCopy(yGm[base], src, C);
    }

    __aicore__ inline void zeroVec(AscendC::LocalTensor<DTYPE_Y> &v, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            v.SetValue(i, (DTYPE_Y)0);
        }
    }

    __aicore__ inline void addScalar(AscendC::LocalTensor<DTYPE_X> &v, float s, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            float t = (float)v.GetValue(i) + s;
            v.SetValue(i, (DTYPE_X)t);
        }
    }

    __aicore__ inline void mulScalarInplace(AscendC::LocalTensor<DTYPE_Y> &v, float s, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            float t = (float)v.GetValue(i) * s;
            v.SetValue(i, (DTYPE_Y)t);
        }
    }

    __aicore__ inline void vecAddInplace(AscendC::LocalTensor<DTYPE_Y> &acc, AscendC::LocalTensor<DTYPE_X> &v, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            float t = (float)acc.GetValue(i) + (float)v.GetValue(i);
            acc.SetValue(i, (DTYPE_Y)t);
        }
    }

    __aicore__ inline void computeMeanVar(AscendC::LocalTensor<DTYPE_X> &v, float &mean, float &var, uint32_t len)
    {
        float sum = 0.f, sum2 = 0.f;
        for (uint32_t i = 0; i < len; ++i) {
            float t = (float)v.GetValue(i);
            sum += t;
            sum2 += t * t;
        }
        mean = sum / float(len);
        var = sum2 / float(len) - mean * mean;
    }

    __aicore__ inline void normAffine(AscendC::LocalTensor<DTYPE_X> &v, float mean, float invStd, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            float x = (float)v.GetValue(i);
            float g = (float)gammaLocal.GetValue(i);
            float b = (float)betaLocal.GetValue(i);
            float y = (x - mean) * invStd;
            y = y * g + b;
            v.SetValue(i, (DTYPE_X)y);
        }
    }

    __aicore__ inline float tanh_approx(float x)
    {
        // simple approximation using std-like tanh via intrinsic if available
        return AscendC::tanh(x);
    }

    __aicore__ inline void geluInplace(AscendC::LocalTensor<DTYPE_Y> &v, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            float x = (float)v.GetValue(i);
            float u = GELU_COEF_A * (x + GELU_COEF_B * x * x * x);
            float t = 0.5f * x * (1.0f + tanh_approx(u));
            v.SetValue(i, (DTYPE_Y)t);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;
    AscendC::GlobalTensor<DTYPE_X> gammaGm;
    AscendC::GlobalTensor<DTYPE_X> betaGm;
    AscendC::GlobalTensor<DTYPE_X> sumwGm;

    AscendC::LocalTensor<DTYPE_X> gammaLocal;
    AscendC::LocalTensor<DTYPE_X> betaLocal;

    uint32_t N, D, H, W, C;
    uint32_t Do, Ho, Wo;
    uint32_t poolK;
    float eps;
    float sumWeight;

    uint32_t voxelsIn;
    uint32_t voxelsOut;
    uint32_t blockVoxelOut;
    uint32_t startVoxelOut;
    uint32_t endVoxelOut;
};

extern "C" __global__ __aicore__ void conv_transpose3d_sum_layer_norm_avg_pool_gelu_custom(GM_ADDR x, GM_ADDR sumw, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelFused op;
    op.Init(x, sumw, gamma, beta, y,
            tiling_data.N, tiling_data.D, tiling_data.H, tiling_data.W, tiling_data.C,
            tiling_data.Do, tiling_data.Ho, tiling_data.Wo,
            tiling_data.eps, tiling_data.poolK);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused: TransposedConv3D (PyTorch) -> channels-last transform -> custom fused Sum+LayerNorm+AvgPool+GELU kernel -> back to NCDHW
    LayerNorm is applied across channel dimension (channels-last) with weight/bias from self.norm.
    AvgPool3d is fixed kernel=stride=(2,2,2) inside custom kernel as in the original model config.
    """
    def __init__(self, in_channels, out_channels, kernel_size, stride, padding, output_padding, sum_weight, norm_shape, pool_kernel_size):
        super().__init__()
        self.conv_transpose = torch.nn.ConvTranspose3d(in_channels, out_channels, kernel_size, stride=stride, padding=padding, output_padding=output_padding)
        # sum weight as Parameter to keep parity with original
        self.sum_weight = torch.nn.Parameter(torch.tensor(sum_weight, dtype=torch.float32))
        # interpret LayerNorm across channels in channels-last layout
        self.norm = torch.nn.LayerNorm(norm_shape)
        self.pool_kernel_size = pool_kernel_size
        self.gelu = torch.nn.GELU()

    def forward(self, x):
        # x: [N, Cin, D, H, W]
        x = self.conv_transpose(x)  # [N, Cout, D, H, W]
        # switch to channels-last (NDHWC) for fused kernel which normalizes across channel dimension
        x_ndhwc = x.permute(0, 2, 3, 4, 1).contiguous()
        y_ndhwc = custom_ops_lib.conv_transpose3d_sum_layer_norm_avg_pool_gelu_custom(
            x_ndhwc, self.sum_weight, self.norm.weight, self.norm.bias
        )
        # back to NCDHW
        y = y_ndhwc.permute(0, 4, 1, 2, 3).contiguous()
        return y

def get_inputs():
    batch_size = 32
    in_channels = 32
    depth, height, width = 16, 32, 32
    return [torch.rand(batch_size, in_channels, depth, height, width)]

def get_init_inputs():
    batch_size = 32
    in_channels = 32
    out_channels = 64
    depth, height, width = 16, 32, 32
    kernel_size = (3, 3, 3)
    stride = (2, 2, 2)
    padding = (1, 1, 1)
    output_padding = (1, 1, 1)
    sum_weight = 1.0
    norm_shape = (out_channels,)
    pool_kernel_size = (2, 2, 2)
    return [in_channels, out_channels, kernel_size, stride, padding, output_padding, sum_weight, norm_shape, pool_kernel_size]
