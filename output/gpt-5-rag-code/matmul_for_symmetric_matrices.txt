host_operator_src="""
#include "matmul_for_symmetric_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MatmulForSymmetricMatricesCustomTilingData tiling;

    // Derive M, N, K from input shapes (assume square symmetric matrices)
    const auto* aShape = context->GetInputShape(0);
    const auto* bShape = context->GetInputShape(1);

    // Basic block dim setup
    context->SetBlockDim(BLOCK_DIM);

    // Save tiling to buffer
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace
    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    (void)aShape;
    (void)bShape;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    // Output C shape is (N, N) where N is last dim of B and first dim of A
    if (a_shape->GetRank() != 2 || b_shape->GetRank() != 2) {
        return ge::GRAPH_PARAM_INVALID;
    }
    gert::Shape* c_shape = context->GetOutputShape(0);
    std::vector<int64_t> outDims(2);
    outDims[0] = a_shape->GetDim(0);
    outDims[1] = b_shape->GetDim(1);
    c_shape->SetDimNum(2);
    c_shape->SetDims(outDims.data(), outDims.size());
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class MatmulForSymmetricMatricesCustom : public OpDef {
public:
    explicit MatmulForSymmetricMatricesCustom(const char* name) : OpDef(name)
    {
        this->Input("a")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("b")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("c")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulForSymmetricMatricesCustom);
}

kernel_src="""
#include "kernel_operator.h"

// Using AscendC Matmul pipeline similar to batch_matmul examples.
extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    // prepare tiling
    GET_TILING_DATA(tilingData, tiling);

    // Define matmul kernel types: ND float for A, B, C (no transpose)
    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> batchMatmulKernel;
    AscendC::TPipe pipe;

    // Basic share settings
    tilingData.cubeTilingData.shareMode = 0;
    tilingData.cubeTilingData.shareL1Size = FULL_L1_SIZE;
    tilingData.cubeTilingData.shareL0CSize = FULL_L0C_SIZE;
    tilingData.cubeTilingData.shareUbSize = 0;

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), batchMatmulKernel.matmulObj, &tilingData.cubeTilingData);

    // batch size 1 for simple 2D matmul
    int batchNum = 1;

    // init matmul kernel
    batchMatmulKernel.Init(a, b, nullptr, c, workspace, tilingData.cubeTilingData);
    // matmul kernel process
    batchMatmulKernel.Process<false>(&pipe, batchNum, batchNum);
}
