#include "logic_and_broadcast_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint64_t SafeProd(const std::vector<int64_t>& dims, size_t begin, size_t end) {
    uint64_t p = 1;
    for (size_t i = begin; i < end; ++i) p *= static_cast<uint64_t>(dims[i]);
    return p;
}
}

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    LogicAndBroadcastCustomTilingData tiling;

    const gert::Shape* x_shape_ptr = context->GetInputShape(0);
    const gert::Shape* y_shape_ptr = context->GetInputShape(1);
    std::vector<int64_t> x_dims = x_shape_ptr->GetOriginShape().GetDims();
    std::vector<int64_t> y_dims = y_shape_ptr->GetOriginShape().GetDims();

    if (x_dims.empty()) { x_dims = {1}; }
    if (y_dims.empty()) { y_dims = {1}; }

    // Align to at least 3D [N, H, W]
    // For higher dims, fold leading dims into N, keep last two as H and W.
    uint64_t W = static_cast<uint64_t>(x_dims.back());
    uint64_t H = static_cast<uint64_t>(x_dims.size() >= 2 ? x_dims[x_dims.size()-2] : 1);
    uint64_t N = SafeProd(x_dims, 0, (x_dims.size() >= 2 ? x_dims.size()-2 : 0));

    // Basic validation for broadcast pattern y of shape [1, H, 1] (after folding/align)
    // When original rank of y differs, check compat along last two dims.
    uint64_t yW = (y_dims.empty() ? 1 : static_cast<uint64_t>(y_dims.back()));
    uint64_t yH = (y_dims.size() >= 2 ? static_cast<uint64_t>(y_dims[y_dims.size()-2]) : 1);
    uint64_t yN = SafeProd(y_dims, 0, (y_dims.size() >= 2 ? y_dims.size()-2 : 0));
    (void)yN; // broadcast over leading dims
    // We expect yW == 1 and yH == H
    // Not strictly enforcing here to keep robustness; kernel assumes this layout.

    const uint32_t BLOCK_DIM = 8;
    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalRows(static_cast<uint32_t>(N * H));
    tiling.set_rowLength(static_cast<uint32_t>(W));
    tiling.set_height(static_cast<uint32_t>(H));
    tiling.set_blockDim(BLOCK_DIM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    // Output shape is broadcasted shape; since x is the expanded target, use x's shape.
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* out_shape = context->GetOutputShape(0);
    *out_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto dt = context->GetInputDataType(0);
    context->SetOutputDataType(0, dt);
    return ge::GRAPH_SUCCESS;
}
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2; // number of tensors per queue

class KernelLogicAndBroadcast {
public:
    __aicore__ inline KernelLogicAndBroadcast() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t totalRows, uint32_t rowLength, uint32_t height)
    {
        totalRows_ = totalRows;
        rowLen_ = rowLength;
        height_ = height;

        coreNum_ = static_cast<uint32_t>(AscendC::GetBlockNum());
        blockId_ = static_cast<uint32_t>(AscendC::GetBlockIdx());

        rowsPerCore_ = totalRows_ / coreNum_;
        tailRows_ = totalRows_ % coreNum_;

        startRow_ = blockId_ * rowsPerCore_;
        if (blockId_ < tailRows_) {
            rowsPerCore_ += 1;
            startRow_ += blockId_;
        } else {
            startRow_ += tailRows_;
        }

        // Bind global buffers, offset to the beginning row of this core
        xGm_.SetGlobalBuffer((__gm__ DTYPE_X *)x + static_cast<uint64_t>(startRow_) * rowLen_,
                             static_cast<uint64_t>(rowsPerCore_) * rowLen_);
        zGm_.SetGlobalBuffer((__gm__ DTYPE_Z *)z + static_cast<uint64_t>(startRow_) * rowLen_,
                             static_cast<uint64_t>(rowsPerCore_) * rowLen_);
        // y is indexed by H only, we bind full buffer
        yGm_.SetGlobalBuffer((__gm__ DTYPE_Y *)y, static_cast<uint64_t>(height_));

        pipe_.InitBuffer(inQX_, BUFFER_NUM, rowLen_ * sizeof(DTYPE_X));
        pipe_.InitBuffer(tmpQYVec_, BUFFER_NUM, rowLen_ * sizeof(DTYPE_Y));
        pipe_.InitBuffer(outQZ_, BUFFER_NUM, rowLen_ * sizeof(DTYPE_Z));
        // scalar buffer for y[h], minimal 1 element
        pipe_.InitBuffer(inQYScalar_, BUFFER_NUM, sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t r = 0; r < rowsPerCore_; ++r) {
            CopyIn(r);
            Compute(r);
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t r)
    {
        // Copy one row of x
        AscendC::LocalTensor<DTYPE_X> xLocal = inQX_.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm_[static_cast<uint64_t>(r) * rowLen_], rowLen_);
        inQX_.EnQue(xLocal);

        // Load scalar y at index h = (globalRow % H)
        uint32_t globalRow = startRow_ + r;
        uint32_t h = (height_ == 0U ? 0U : (globalRow % height_));
        AscendC::LocalTensor<DTYPE_Y> yScalarLocal = inQYScalar_.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(yScalarLocal, yGm_[h], 1);
        inQYScalar_.EnQue(yScalarLocal);
    }

    __aicore__ inline void Compute(uint32_t /*r*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQX_.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yScalarLocal = inQYScalar_.DeQue<DTYPE_Y>();

        // Broadcast scalar y to a vector
        AscendC::LocalTensor<DTYPE_Y> yVecLocal = tmpQYVec_.AllocTensor<DTYPE_Y>();
        DTYPE_Y yv = yScalarLocal.GetValue(0);
        AscendC::Dup(yVecLocal, yv, rowLen_);

        // z = x AND y_broadcast
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQZ_.AllocTensor<DTYPE_Z>();
        AscendC::LogicalAnd(zLocal, xLocal, yVecLocal, rowLen_);

        outQZ_.EnQue<DTYPE_Z>(zLocal);
        tmpQYVec_.FreeTensor(yVecLocal);
        inQX_.FreeTensor(xLocal);
        inQYScalar_.FreeTensor(yScalarLocal);
    }

    __aicore__ inline void CopyOut(uint32_t r)
    {
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQZ_.DeQue<DTYPE_Z>();
        AscendC::DataCopy(zGm_[static_cast<uint64_t>(r) * rowLen_], zLocal, rowLen_);
        outQZ_.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQYScalar_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQYVec_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQZ_;

    AscendC::GlobalTensor<DTYPE_X> xGm_;
    AscendC::GlobalTensor<DTYPE_Y> yGm_;
    AscendC::GlobalTensor<DTYPE_Z> zGm_;

    uint32_t totalRows_;
    uint32_t rowLen_;
    uint32_t height_;

    uint32_t coreNum_;
    uint32_t blockId_;
    uint32_t startRow_;
    uint32_t rowsPerCore_;
    uint32_t tailRows_;
};

extern "C" __global__ __aicore__ void logic_and_broadcast_custom(GM_ADDR x, GM_ADDR y, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    (void)workspace;
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogicAndBroadcast op;
    op.Init(x, y, z,
            tiling_data.totalRows,
            tiling_data.rowLength,
            tiling_data.height);
    op.Process();
}
"""
