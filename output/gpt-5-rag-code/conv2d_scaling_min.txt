kernel_src="""
#include "kernel_operator.h"

#ifndef DTYPE_X
#define DTYPE_X float
#endif

#ifndef DTYPE_S
#define DTYPE_S float
#endif

#ifndef DTYPE_Z
#define DTYPE_Z float
#endif

constexpr int32_t BUFFER_NUM = 2; // number of buffers per queue

class KernelConv2dScalingMin {
public:
    __aicore__ inline KernelConv2dScalingMin() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR scale, GM_ADDR y, uint32_t hw, uint32_t channels, uint32_t tileNum)
    {
        this->hw = hw;
        this->channels = channels;
        this->tileNum = tileNum;

        // one core handles one batch item
        uint32_t block_idx = AscendC::GetBlockIdx();
        uint64_t per_batch_size = static_cast<uint64_t>(channels) * static_cast<uint64_t>(hw);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + block_idx * per_batch_size, per_batch_size);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + block_idx * static_cast<uint64_t>(hw), hw);
        sGm.SetGlobalBuffer((__gm__ DTYPE_S*)scale, 1);

        this->tileLength = this->hw / tileNum / BUFFER_NUM;

        pipe.InitBuffer(inQueueX, BUFFER_NUM, this->tileLength * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueueZ, BUFFER_NUM, this->tileLength * sizeof(DTYPE_Z));
    }

    __aicore__ inline void Process()
    {
        // load scale once
        DTYPE_S s_val = LoadScaleOnce();

        int32_t loopCount = this->tileNum * BUFFER_NUM;
        for (int32_t i = 0; i < loopCount; ++i) {
            ComputeTile(i, s_val);
        }
    }

private:
    __aicore__ inline DTYPE_S LoadScaleOnce()
    {
        // allocate a small local tensor from input queue to fetch scale
        AscendC::LocalTensor<DTYPE_S> sLocal = inQueueX.AllocTensor<DTYPE_S>();
        AscendC::DataCopy(sLocal, sGm, 1);
        // assume LocalTensor supports GetValue(0). If not, this represents loading scalar into register.
        DTYPE_S s = sLocal.GetValue(0);
        inQueueX.FreeTensor(sLocal);
        return s;
    }

    __aicore__ inline void ComputeTile(int32_t progress, DTYPE_S s_val)
    {
        uint32_t offset = static_cast<uint32_t>(progress) * this->tileLength;

        // First channel initializes the running minimum
        {
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.AllocTensor<DTYPE_Z>();
            AscendC::DataCopy(xLocal, xGm[offset], this->tileLength);
            AscendC::Muls(zLocal, xLocal, s_val, this->tileLength);
            outQueueZ.EnQue(zLocal);
            inQueueX.FreeTensor(xLocal);
        }

        // Process remaining channels
        for (uint32_t c = 1; c < this->channels; ++c) {
            AscendC::LocalTensor<DTYPE_X> xLocal = inQueueX.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(xLocal, xGm[c * this->hw + offset], this->tileLength);
            // scale in place
            AscendC::Muls(xLocal, xLocal, s_val, this->tileLength);

            AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
            AscendC::Min(zLocal, zLocal, xLocal, this->tileLength);
            outQueueZ.EnQue(zLocal);

            inQueueX.FreeTensor(xLocal);
        }

        // write out the tile result
        AscendC::LocalTensor<DTYPE_Z> zLocal = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[offset], zLocal, this->tileLength);
        outQueueZ.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_S> sGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t hw;          // H*W per image
    uint32_t channels;    // C
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dScalingMin op;
    op.Init(x, scale, y, tiling_data.hw, tiling_data.channels, tiling_data.tileNum);
    op.Process();
}

#include <torch/library.h>

at::Tensor conv2d_scaling_min_custom_impl_npu(const at::Tensor& x, const at::Tensor& scale) {
    TORCH_CHECK(scale.numel() == 1, "scale must be a scalar tensor with 1 element");
    TORCH_CHECK(x.dim() == 4, "input must be NCHW 4D tensor");

    auto sizes = x.sizes();
    at::Tensor result = at::empty({sizes[0], 1, sizes[2], sizes[3]}, x.options());
    EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, scale, result);
    return result;
}

    m.impl("conv2d_scaling_min_custom", &conv2d_scaling_min_custom_impl_npu);

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Optimized model: Conv2d computed by PyTorch, followed by fused scaling + channel-wise min via AscendC custom op.
    """
    def __init__(self, in_channels, out_channels, kernel_size, scale_factor):
        super(ModelNew, self).__init__()
        self.conv = torch.nn.Conv2d(in_channels, out_channels, kernel_size)
        self.scale_factor = float(scale_factor)

    def forward(self, x):
        x = self.conv(x)
        scale_tensor = torch.tensor([self.scale_factor], dtype=x.dtype, device=x.device)
        return custom_ops_lib.conv2d_scaling_min_custom(x, scale_tensor)
