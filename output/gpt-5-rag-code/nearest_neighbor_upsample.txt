    NearestNeighborUpsampleCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    // Expect NCHW
    uint32_t n = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(0));
    uint32_t c = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(1));
    uint32_t h = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(2));
    uint32_t w = static_cast<uint32_t>(in_shape->GetOriginShape().GetDim(3));

    const uint32_t scale = 4; // fixed scale factor as in model
    uint32_t hout = h * scale;
    uint32_t wout = w * scale;

    uint32_t nc = n * c;
    uint64_t totalOutLen64 = static_cast<uint64_t>(nc) * static_cast<uint64_t>(hout) * static_cast<uint64_t>(wout);

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_totalOutLength(static_cast<uint32_t>(totalOutLen64));
    tiling.set_tileNum(TILE_NUM);
    tiling.set_nc(nc);
    tiling.set_hin(h);
    tiling.set_win(w);
    tiling.set_hout(hout);
    tiling.set_wout(wout);
    tiling.set_scale(scale);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;

    uint64_t c = in_shape->GetOriginShape().GetDim(1);

    uint64_t w = in_shape->GetOriginShape().GetDim(3);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t OUT_BUFFER_NUM = 2;
constexpr int32_t SCALAR_BUFFER_NUM = 1;

class KernelNearestNeighborUpsample {
public:
    __aicore__ inline KernelNearestNeighborUpsample() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutLength, uint32_t tileNum,
                                uint32_t nc, uint32_t hin, uint32_t win,
                                uint32_t hout, uint32_t wout, uint32_t scale)
    {
        this->totalOutLength = totalOutLength;
        this->blockLength = totalOutLength / AscendC::GetBlockNum();
        this->tileNum = tileNum;
        this->nc = nc;
        this->hin = hin;
        this->win = win;
        this->hout = hout;
        this->wout = wout;
        this->scale = scale;
        this->hwIn = hin * win;
        this->hwOut = hout * wout;

        uint32_t blockOffset = this->blockLength * AscendC::GetBlockIdx();

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x, totalOutLength); // input length not equal to output, but we only use index mapping
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z *)y + blockOffset, this->blockLength);

        // choose tile length as multiple of wout (process an integer number of rows)
        uint32_t rowsPerTile = AscendC::max((uint32_t)1, (this->blockLength / (this->tileNum * OUT_BUFFER_NUM)) / this->wout);
        this->tileRowCount = rowsPerTile;
        this->tileLength = rowsPerTile * this->wout;

        // init local buffers
        pipe.InitBuffer(outQueueZ, OUT_BUFFER_NUM, this->wout * sizeof(DTYPE_Z));
        pipe.InitBuffer(inScalarQueue, SCALAR_BUFFER_NUM, 1 * sizeof(DTYPE_X));
    }

    __aicore__ inline void Process()
    {
        int32_t loopCount = this->tileNum * OUT_BUFFER_NUM;
        uint32_t blockBaseGlobal = this->blockLength * AscendC::GetBlockIdx();
        for (int32_t i = 0; i < loopCount; ++i) {
            // compute and enqueue tile rows
            for (uint32_t r = 0; r < this->tileRowCount; ++r) {
                uint32_t rowGlobalIndex = blockBaseGlobal + i * this->tileLength + r * this->wout;
                if (rowGlobalIndex >= blockBaseGlobal + this->blockLength) {
                    break;
                }
                ComputeRow(rowGlobalIndex);
            }
            // copy out tile rows
            for (uint32_t r = 0; r < this->tileRowCount; ++r) {
                uint32_t rowGlobalIndex = blockBaseGlobal + i * this->tileLength + r * this->wout;
                if (rowGlobalIndex >= blockBaseGlobal + this->blockLength) {
                    break;
                }
                CopyOutRow(rowGlobalIndex - blockBaseGlobal);
            }
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t rowGlobalIndex)
    {
        // Determine channel and row indices
        uint32_t channelIdx = (rowGlobalIndex / this->hwOut);
        uint32_t rowOffsetInChannel = (rowGlobalIndex % this->hwOut) / this->wout; // oh
        uint32_t ih = rowOffsetInChannel / this->scale;
        uint32_t inBase = channelIdx * this->hwIn + ih * this->win;

        AscendC::LocalTensor<DTYPE_Z> zRow = outQueueZ.AllocTensor<DTYPE_Z>();

        // Fill one output row by repeating each input element along width
        for (uint32_t iw = 0; iw < this->win; ++iw) {
            AscendC::LocalTensor<DTYPE_X> tmpVal = inScalarQueue.AllocTensor<DTYPE_X>();
            AscendC::DataCopy(tmpVal, xGm[inBase + iw], 1);
            // repeat value 'scale' times into output row
            uint32_t start = iw * this->scale;
            for (uint32_t rep = 0; rep < this->scale; ++rep) {
                // copy single element to the corresponding position in local row
                AscendC::DataCopy(zRow[start + rep], tmpVal, 1);
            }
            inScalarQueue.FreeTensor(tmpVal);
        }
        outQueueZ.EnQue<DTYPE_Z>(zRow);
    }

    __aicore__ inline void CopyOutRow(uint32_t rowOffsetWithinBlock)
    {
        AscendC::LocalTensor<DTYPE_Z> zRow = outQueueZ.DeQue<DTYPE_Z>();
        AscendC::DataCopy(yGm[rowOffsetWithinBlock], zRow, this->wout);
        outQueueZ.FreeTensor(zRow);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueZ;
    AscendC::TQue<AscendC::TPosition::VECIN, SCALAR_BUFFER_NUM> inScalarQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t totalOutLength;
    uint32_t blockLength;
    uint32_t tileNum;
    uint32_t tileLength;
    uint32_t tileRowCount;

    uint32_t nc;
    uint32_t hin;
    uint32_t win;
    uint32_t hout;
    uint32_t wout;
    uint32_t scale;
    uint32_t hwIn;
    uint32_t hwOut;
};

extern "C" __global__ __aicore__ void nearest_neighbor_upsample_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelNearestNeighborUpsample op;
    op.Init(x, y,
            tiling_data.totalOutLength,
            tiling_data.tileNum,
            tiling_data.nc,
            tiling_data.hin,
            tiling_data.win,
            tiling_data.hout,
            tiling_data.wout,
            tiling_data.scale);
    op.Process();
}

at::Tensor nearest_neighbor_upsample_custom_impl_npu(const at::Tensor& self) {
    TORCH_CHECK(self.dim() == 4, "Input must be NCHW 4D tensor");
    int64_t n = self.size(0);

    int64_t h = self.size(2);
    int64_t w = self.size(3);

    std::vector<int64_t> out_size = {n, c, h * scale, w * scale};
    at::Tensor result = at::empty(out_size, self.options());

    return result;
}

    m.impl("nearest_neighbor_upsample_custom", &nearest_neighbor_upsample_custom_impl_npu);
