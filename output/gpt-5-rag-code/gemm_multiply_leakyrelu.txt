kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelGemmMulLeakyRelu {
public:
    __aicore__ inline KernelGemmMulLeakyRelu() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR multiplier, GM_ADDR negative_slope, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileN)
    {
        this->M = M;
        this->N = N;
        this->K = K;
        this->tileN = tileN;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        this->rowsPerBlock = (M + blockNum - 1) / blockNum;
        this->rowStart = blockIdx * this->rowsPerBlock;
        if (this->rowStart > M) this->rowStart = M;
        this->rowEnd = this->rowStart + this->rowsPerBlock;
        if (this->rowEnd > M) this->rowEnd = M;

        // Set GM buffers scoped to this block's rows for x and y
        xGm.SetGlobalBuffer((__gm__ DTYPE_X*)x + (uint64_t)rowStart * K, (rowEnd - rowStart) * K);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Z*)y + (uint64_t)rowStart * N, (rowEnd - rowStart) * N);
        // Full buffers for weight and bias and scalars
        wGm.SetGlobalBuffer((__gm__ DTYPE_Y*)weight, (uint64_t)N * K);
        bGm.SetGlobalBuffer((__gm__ DTYPE_Y*)bias, N);
        multGm.SetGlobalBuffer((__gm__ DTYPE_Y*)multiplier, 1);
        nsGm.SetGlobalBuffer((__gm__ DTYPE_Y*)negative_slope, 1);

        // Init local queues
        pipe.InitBuffer(xRowQ, BUFFER_NUM, K * sizeof(DTYPE_X));
        pipe.InitBuffer(wTileQ, BUFFER_NUM, tileN * K * sizeof(DTYPE_Y));
        pipe.InitBuffer(bTileQ, BUFFER_NUM, tileN * sizeof(DTYPE_Y));
        pipe.InitBuffer(yTileQ, BUFFER_NUM, tileN * sizeof(DTYPE_Z));
        pipe.InitBuffer(scalarQ, BUFFER_NUM, 2 * sizeof(DTYPE_Y));
    }

    __aicore__ inline void Process()
    {
        // Load scalars multiplier and negative slope
        AscendC::LocalTensor<DTYPE_Y> scalarLocal = scalarQ.AllocTensor<DTYPE_Y>();
        AscendC::DataCopy(scalarLocal, multGm, 1);
        AscendC::DataCopy(scalarLocal[1], nsGm, 1);
        DTYPE_Y multiplierVal = scalarLocal.GetValue(0);
        DTYPE_Y negativeSlopeVal = scalarLocal.GetValue(1);
        scalarQ.FreeTensor(scalarLocal);

        for (uint32_t m = rowStart; m < rowEnd; ++m) {
            // Copy one input row of length K
            AscendC::LocalTensor<DTYPE_X> xRowLocal = xRowQ.AllocTensor<DTYPE_X>();
            uint32_t localRowIdx = m - rowStart;
            AscendC::DataCopy(xRowLocal, xGm[(uint64_t)localRowIdx * K], K);
            xRowQ.EnQue(xRowLocal);
            xRowLocal = xRowQ.DeQue<DTYPE_X>();

            for (uint32_t n0 = 0; n0 < N; n0 += tileN) {
                uint32_t currN = (n0 + tileN <= N) ? tileN : (N - n0);

                AscendC::LocalTensor<DTYPE_Y> wLocal = wTileQ.AllocTensor<DTYPE_Y>();
                AscendC::LocalTensor<DTYPE_Y> bLocal = bTileQ.AllocTensor<DTYPE_Y>();
                AscendC::LocalTensor<DTYPE_Z> yLocal = yTileQ.AllocTensor<DTYPE_Z>();

                // Copy W tile: rows [n0, n0+currN), each length K
                AscendC::DataCopy(wLocal, wGm[(uint64_t)n0 * K], (uint64_t)currN * K);
                // Copy bias tile
                AscendC::DataCopy(bLocal, bGm[n0], currN);

                // Initialize output accumulators to 0
                for (uint32_t i = 0; i < currN; ++i) {
                    yLocal.SetValue(i, static_cast<DTYPE_Z>(0));
                }

                // Compute partial GEMM for the tile
                // y_i = dot(xRow, w_row_i)
                for (uint32_t kk = 0; kk < K; ++kk) {
                    DTYPE_X xv = xRowLocal.GetValue(kk);
                    for (uint32_t i = 0; i < currN; ++i) {
                        // wLocal is laid out as [currN, K] row-major
                        DTYPE_Y wv = wLocal.GetValue((uint64_t)i * K + kk);
                        DTYPE_Z acc = yLocal.GetValue(i);
                        acc = acc + static_cast<DTYPE_Z>(xv * wv);
                        yLocal.SetValue(i, acc);
                    }
                }

                // Add bias, apply multiplier and LeakyReLU
                for (uint32_t i = 0; i < currN; ++i) {
                    DTYPE_Z v = yLocal.GetValue(i);
                    v = v + static_cast<DTYPE_Z>(bLocal.GetValue(i));
                    v = v * static_cast<DTYPE_Z>(multiplierVal);
                    // leaky relu
                    if (v < static_cast<DTYPE_Z>(0)) {
                        v = v * static_cast<DTYPE_Z>(negativeSlopeVal);
                    }
                    yLocal.SetValue(i, v);
                }

                // Write back output tile for row m: contiguous along N
                AscendC::DataCopy(yGm[(uint64_t)localRowIdx * N + n0], yLocal, currN);

                yTileQ.FreeTensor(yLocal);
                bTileQ.FreeTensor(bLocal);
                wTileQ.FreeTensor(wLocal);
            }
            xRowQ.FreeTensor(xRowLocal);
        }
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xRowQ, wTileQ, bTileQ, scalarQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yTileQ;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> wGm;
    AscendC::GlobalTensor<DTYPE_Y> bGm;
    AscendC::GlobalTensor<DTYPE_Y> multGm;
    AscendC::GlobalTensor<DTYPE_Y> nsGm;
    AscendC::GlobalTensor<DTYPE_Z> yGm;

    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t tileN;

    uint32_t rowsPerBlock;
    uint32_t rowStart;
    uint32_t rowEnd;
};

extern "C" __global__ __aicore__ void gemm_multiply_leakyrelu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR multiplier, GM_ADDR negative_slope, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMulLeakyRelu op;
    op.Init(x, weight, bias, multiplier, negative_slope, y, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileN);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    """
    Fused Gemm + multiply + LeakyReLU using custom AscendC operator.
    """
    def __init__(self, in_features, out_features, multiplier, negative_slope):
        super().__init__()
        self.gemm = torch.nn.Linear(in_features, out_features)
        self.multiplier = float(multiplier)
        self.negative_slope = float(negative_slope)

    def forward(self, x):
        # Use weight and bias from the linear layer but compute via fused custom op
        return custom_ops_lib.gemm_multiply_leakyrelu_custom(
            x, self.gemm.weight, self.gemm.bias, self.multiplier, self.negative_slope
        )
