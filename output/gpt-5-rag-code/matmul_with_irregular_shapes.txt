host_operator_src="""
#include "matmul_with_irregular_shapes_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}
}

namespace optiling {
const uint32_t BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    MatmulWithIrregularShapesCustomTilingData tiling;

    const auto* A_shape = context->GetInputShape(0);
    const auto* B_shape = context->GetInputShape(1);

    // Expect 2D: A [M, K], B [K, N]
    uint32_t M = (uint32_t)A_shape->GetOriginShape().GetDim(0);
    uint32_t K = (uint32_t)A_shape->GetOriginShape().GetDim(1);
    uint32_t N = (uint32_t)B_shape->GetOriginShape().GetDim(1);

    // Basic tiling heuristic for irregular shapes
    uint32_t rowsPerBlock = CeilDiv(M, BLOCK_DIM);
    uint32_t tileN = 256; // UB-friendly N-tile
    if (tileN > N) tileN = N;
    uint32_t tileK = 64;  // K blocking
    if (tileK > K) tileK = K;

    context->SetBlockDim(BLOCK_DIM);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0; // no extra workspace
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* A_shape = context->GetInputShape(0);
    const gert::Shape* B_shape = context->GetInputShape(1);

    // A: [M, K], B: [K, N] => C: [M, N]
    gert::Shape* C_shape = context->GetOutputShape(0);
    gert::Shape out;
    out.SetDimNum(2);
    out.SetDim(0, A_shape->GetOriginShape().GetDim(0));
    out.SetDim(1, B_shape->GetOriginShape().GetDim(1));
    *C_shape = out;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return GRAPH_SUCCESS;
}
}

namespace ops {
class MatmulWithIrregularShapesCustom : public OpDef {
public:
    explicit MatmulWithIrregularShapesCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulWithIrregularShapesCustom);
}

kernel_src="""
#include "kernel_operator.h"

// A simple wrapper to use hardware matmul engine for irregular shapes.
// A: [M, K], B: [K, N], C: [M, N], dtype float32

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tilingGm)
{
    // Use cube matmul tiling prepared by framework. If unavailable, this still sets defaults.
    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    // Matmul kernel object provided by AscendC advanced API samples
    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> mmKernel;
    AscendC::TPipe pipe;

    // Share entire L1/L0C to matmul engine for better performance
    tiling.shareMode = 0;                 // share mode
    tiling.shareL1Size = FULL_L1_SIZE;    // use full L1
    tiling.shareL0CSize = FULL_L0C_SIZE;  // use full L0C
    tiling.shareUbSize = 0;               // no UB share

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mmKernel.matmulObj, &tiling);

    // Initialize and run
    mmKernel.Init(A, B, nullptr, C, workspace, tiling);
    // single-batch execution for plain 2D matmul
    mmKernel.Process<false>(&pipe, 1, 1);
}

#include <torch/library.h>
#include <torch/csrc/autograd/custom_function.h>

#include <torch/extension.h>

at::Tensor matmul_with_irregular_shapes_custom_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "Inputs must be 2D");
    TORCH_CHECK(A.size(1) == B.size(0), "Incompatible shapes for matmul");
    at::Tensor C = at::empty({A.size(0), B.size(1)}, A.options());
    EXEC_NPU_CMD(aclnnMatmulWithIrregularShapesCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_with_irregular_shapes_custom", &matmul_with_irregular_shapes_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_with_irregular_shapes_custom", &matmul_with_irregular_shapes_custom_impl_npu, "Matmul(A, B) -> C");
}
