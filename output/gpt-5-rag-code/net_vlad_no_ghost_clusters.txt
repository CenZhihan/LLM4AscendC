kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelNetVladNoGhostClusters {
public:
    __aicore__ inline KernelNetVladNoGhostClusters() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR assignment, GM_ADDR clusters2, GM_ADDR z,
                                uint32_t B, uint32_t N, uint32_t D, uint32_t K, uint32_t batchPerBlock)
    {
        this->B = B;
        this->N = N;
        this->D = D;
        this->K = K;
        this->batchPerBlock = batchPerBlock;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t startB = blockIdx * batchPerBlock;
        this->startB = startB;

        xGm.SetGlobalBuffer((__gm__ float *)x + startB * N * D, (B - startB) * N * D);
        aGm.SetGlobalBuffer((__gm__ float *)assignment + startB * N * K, (B - startB) * N * K);
        c2Gm.SetGlobalBuffer((__gm__ float *)clusters2, D * K);
        zGm.SetGlobalBuffer((__gm__ float *)z + startB * (D * K), (B - startB) * (D * K));

        // Buffers: per-batch x[N*D], s[N*K], v[D*K], c2[D*K], z[D*K], aSum[K]
        uint32_t xSize = N * D * sizeof(float);
        uint32_t sSize = N * K * sizeof(float);
        uint32_t vkSize = D * K * sizeof(float);
        uint32_t c2Size = D * K * sizeof(float);
        uint32_t zSize = D * K * sizeof(float);
        uint32_t aSumSize = K * sizeof(float);

        pipe.InitBuffer(bufX, BUFFER_NUM, xSize);
        pipe.InitBuffer(bufS, BUFFER_NUM, sSize);
        pipe.InitBuffer(bufV, BUFFER_NUM, vkSize);
        pipe.InitBuffer(bufC2, BUFFER_NUM, c2Size);
        pipe.InitBuffer(bufZ, BUFFER_NUM, zSize);
        pipe.InitBuffer(bufAS, BUFFER_NUM, aSumSize);
    }

    __aicore__ inline void Process()
    {
        // Load clusters2 once
        AscendC::LocalTensor<float> c2Local = bufC2.AllocTensor<float>();
        AscendC::DataCopy(c2Local, c2Gm, D * K);
        bufC2.EnQue(c2Local);

        uint32_t remaining = B - startB;
        uint32_t handleB = remaining < batchPerBlock ? remaining : batchPerBlock;

        for (uint32_t b = 0; b < handleB; ++b) {
            // CopyIn x and assignment for this batch
            AscendC::LocalTensor<float> xLocal = bufX.AllocTensor<float>();
            AscendC::LocalTensor<float> sLocal = bufS.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm[b * (N * D)], N * D);
            AscendC::DataCopy(sLocal, aGm[b * (N * K)], N * K);
            bufX.EnQue(xLocal);
            bufS.EnQue(sLocal);

            // Compute vlad
            AscendC::LocalTensor<float> vLocal = bufV.AllocTensor<float>();
            AscendC::LocalTensor<float> aSumLocal = bufAS.AllocTensor<float>();
            AscendC::LocalTensor<float> zLocal = bufZ.AllocTensor<float>();

            AscendC::LocalTensor<float> xL = bufX.DeQue<float>();
            AscendC::LocalTensor<float> sL = bufS.DeQue<float>();
            AscendC::LocalTensor<float> c2L = bufC2.DeQue<float>();

            // Initialize vLocal and aSumLocal to 0
            for (uint32_t i = 0; i < D * K; ++i) {
                vLocal.SetValue(i, 0.f);
            }
            for (uint32_t i = 0; i < K; ++i) {
                aSumLocal.SetValue(i, 0.f);
            }

            // a_sum over N: K-length vector
            for (uint32_t n = 0; n < N; ++n) {
                for (uint32_t k = 0; k < K; ++k) {
                    float val = sL.GetValue(n * K + k);
                    aSumLocal.SetValue(k, aSumLocal.GetValue(k) + val);
                }
            }

            // Accumulate vLocal[d*K + k] = sum_n s[n,k] * x[n,d]
            for (uint32_t n = 0; n < N; ++n) {
                for (uint32_t k = 0; k < K; ++k) {
                    float s_val = sL.GetValue(n * K + k);
                    if (s_val != 0.f) {
                        uint32_t baseX = n * D;
                        for (uint32_t d = 0; d < D; ++d) {
                            float xv = xL.GetValue(baseX + d);
                            uint32_t idx = d * K + k;
                            vLocal.SetValue(idx, vLocal.GetValue(idx) + s_val * xv);
                        }
                    }
                }
            }

            // Compute a = a_sum[k] * clusters2[d,k] and vlad = vLocal - a
            for (uint32_t k = 0; k < K; ++k) {
                float aSumVal = aSumLocal.GetValue(k);
                for (uint32_t d = 0; d < D; ++d) {
                    uint32_t idx = d * K + k;
                    float a_val = aSumVal * c2L.GetValue(idx);
                    float v = vLocal.GetValue(idx) - a_val;
                    vLocal.SetValue(idx, v);
                }
            }

            // L2 intra-norm over D for each k
            const float eps = 1e-12f;
            for (uint32_t k = 0; k < K; ++k) {
                float acc = 0.f;
                for (uint32_t d = 0; d < D; ++d) {
                    float v = vLocal.GetValue(d * K + k);
                    acc += v * v;
                }
                float denom = acc > eps ? AscendC::Sqrt(acc + eps) : 1.0f;
                for (uint32_t d = 0; d < D; ++d) {
                    uint32_t idx = d * K + k;
                    vLocal.SetValue(idx, vLocal.GetValue(idx) / denom);
                }
            }

            // Flatten to zLocal [D*K] and global L2 norm
            float accAll = 0.f;
            for (uint32_t i = 0; i < D * K; ++i) {
                float v = vLocal.GetValue(i);
                zLocal.SetValue(i, v);
                accAll += v * v;
            }
            float denomAll = accAll > eps ? AscendC::Sqrt(accAll + eps) : 1.0f;
            for (uint32_t i = 0; i < D * K; ++i) {
                zLocal.SetValue(i, zLocal.GetValue(i) / denomAll);
            }

            // Enqueue z and free buffers
            bufZ.EnQue(zLocal);

            bufV.FreeTensor(vLocal);
            bufAS.FreeTensor(aSumLocal);
            bufX.FreeTensor(xL);
            bufS.FreeTensor(sL);
            // keep c2L in queue for reuse
            bufC2.EnQue(c2L);

            // CopyOut
            AscendC::LocalTensor<float> zOut = bufZ.DeQue<float>();
            AscendC::DataCopy(zGm[b * (D * K)], zOut, D * K);
            bufZ.FreeTensor(zOut);
        }

        // Free c2Local
        AscendC::LocalTensor<float> c2LFinal = bufC2.DeQue<float>();
        bufC2.FreeTensor(c2LFinal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TBuf<AscendC::TPosition::VECIN>  bufX;
    AscendC::TBuf<AscendC::TPosition::VECIN>  bufS;
    AscendC::TBuf<AscendC::TPosition::VECIN>  bufC2;
    AscendC::TBuf<AscendC::TPosition::VECOUT> bufV;
    AscendC::TBuf<AscendC::TPosition::VECOUT> bufZ;
    AscendC::TBuf<AscendC::TPosition::VECOUT> bufAS;

    AscendC::GlobalTensor<float> xGm;
    AscendC::GlobalTensor<float> aGm;
    AscendC::GlobalTensor<float> c2Gm;
    AscendC::GlobalTensor<float> zGm;

    uint32_t B;
    uint32_t N;
    uint32_t D;
    uint32_t K;
    uint32_t batchPerBlock;
    uint32_t startB;
};

extern "C" __global__ __aicore__ void net_vlad_no_ghost_clusters_custom(GM_ADDR x, GM_ADDR assignment, GM_ADDR clusters2, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelNetVladNoGhostClusters op;
    op.Init(x, assignment, clusters2, z,
            tiling_data.B(), tiling_data.N(), tiling_data.D(), tiling_data.K(),
            tiling_data.batchPerBlock());
    op.Process();
}

model_src='''
import torch
import torch.nn as nn
import torch.nn.functional as F
import custom_ops_lib

class ModelNew(nn.Module):
    def __init__(self, cluster_size, feature_size, ghost_clusters):
        super(ModelNew, self).__init__()
        self.feature_size = feature_size
        self.cluster_size = cluster_size
        self.ghost_clusters = ghost_clusters

        import math
        init_sc = (1 / math.sqrt(feature_size))
        clusters = cluster_size + ghost_clusters

        self.clusters = nn.Parameter(init_sc * torch.randn(feature_size, clusters))
        self.batch_norm = nn.BatchNorm1d(clusters)
        self.clusters2 = nn.Parameter(init_sc * torch.randn(1, feature_size, cluster_size))
        self.out_dim = self.cluster_size * feature_size

    def forward(self, x, mask=None):
        max_sample = x.size(1)  # N
        x_flat = x.view(-1, self.feature_size)  # BN x D

        assignment = torch.matmul(x_flat, self.clusters)  # BN x (K+G)
        assignment = self.batch_norm(assignment)
        assignment = F.softmax(assignment, dim=1)  # BN x (K+G)
        assignment = assignment[:, :self.cluster_size]  # BN x K
        assignment = assignment.view(-1, max_sample, self.cluster_size)  # B x N x K

        # Fused custom kernel: computes vlad residuals, intra-norm, flatten and global L2 norm
        vlad = custom_ops_lib.net_vlad_no_ghost_clusters_custom(x, assignment, self.clusters2)
        return vlad
'''
