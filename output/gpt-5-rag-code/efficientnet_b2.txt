    uint32_t c = static_cast<uint32_t>(in_shape->GetDim(1));

kernel_src="""
#include "kernel_operator.h"

// Global Average Pool 2D to 1x1 per (N, C) group: y[n,c,0,0] = mean(x[n,c,:,:])
// Input: x (N, C, H, W) float
// Output: y (N, C, 1, 1) float

constexpr int32_t BUFFER_NUM = 2;

class KernelEfficientnetB2Custom {
public:
    __aicore__ inline KernelEfficientnetB2Custom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t n, uint32_t c, uint32_t h, uint32_t w, uint32_t tileNum)
    {
        this->n = n;
        this->c = c;
        this->h = h;
        this->w = w;
        this->tileNum = tileNum;

        this->groups = n * c;
        this->hw = h * w;

        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t baseGroupPerBlock = groups / blocks;
        uint32_t remainder = groups % blocks;
        // Distribute remainder groups to first 'remainder' blocks
        this->groupStart = baseGroupPerBlock * blockIdx + (blockIdx < remainder ? blockIdx : remainder);
        this->groupCount = baseGroupPerBlock + (blockIdx < remainder ? 1 : 0);

        xGm.SetGlobalBuffer((__gm__ DTYPE_X *)x + this->groupStart * this->hw, this->groupCount * this->hw);
        yGm.SetGlobalBuffer((__gm__ DTYPE_Y *)y + this->groupStart, this->groupCount);

        pipe.InitBuffer(inQueue, BUFFER_NUM, this->hw * sizeof(DTYPE_X));
        pipe.InitBuffer(outQueue, BUFFER_NUM, sizeof(DTYPE_Y));
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->hw * sizeof(DTYPE_X)); // for intermediate usage if needed
    }

    __aicore__ inline void Process()
    {
        for (uint32_t i = 0; i < this->groupCount; ++i) {
            CopyIn(i);
            Compute(i);
            CopyOut(i);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t idx)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.AllocTensor<DTYPE_X>();
        AscendC::DataCopy(xLocal, xGm[idx * this->hw], this->hw);
        inQueue.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t /*idx*/)
    {
        AscendC::LocalTensor<DTYPE_X> xLocal = inQueue.DeQue<DTYPE_X>();
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.AllocTensor<DTYPE_Y>();

        // Reduce sum over H*W
        // Allocate a temporary 1-element buffer for sum
        AscendC::LocalTensor<DTYPE_X> sumBuf = tmpQueue.AllocTensor<DTYPE_X>();
        // Initialize sum to 0
        AscendC::Fill(sumBuf, static_cast<DTYPE_X>(0), 1);
        // Use vector reduce if available
        AscendC::ReduceSum(sumBuf, xLocal, this->hw);
        // Compute mean = sum / (H*W)
        DTYPE_X inv = static_cast<DTYPE_X>(1.0f) / static_cast<DTYPE_X>(this->hw);
        AscendC::Muls(sumBuf, sumBuf, inv, 1);
        // Write result to output local tensor
        AscendC::DataCopy(yLocal, sumBuf, 1);

        outQueue.EnQue<DTYPE_Y>(yLocal);
        inQueue.FreeTensor(xLocal);
        tmpQueue.FreeTensor(sumBuf);
    }

    __aicore__ inline void CopyOut(uint32_t idx)
    {
        AscendC::LocalTensor<DTYPE_Y> yLocal = outQueue.DeQue<DTYPE_Y>();
        AscendC::DataCopy(yGm[idx], yLocal, 1);
        outQueue.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE_X> xGm;
    AscendC::GlobalTensor<DTYPE_Y> yGm;

    uint32_t n, c, h, w;
    uint32_t groups;
    uint32_t hw;
    uint32_t groupStart;
    uint32_t groupCount;
    uint32_t tileNum;
};

extern "C" __global__ __aicore__ void efficientnet_b2_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelEfficientnetB2Custom op;
    op.Init(x, y,
            tiling_data.n(),
            tiling_data.c(),
            tiling_data.h(),
            tiling_data.w(),
            tiling_data.tileNum());
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

class ModelNew(torch.nn.Module):
    def __init__(self, num_classes=1000):
        super(ModelNew, self).__init__()
        # Stem
        self.conv1 = torch.nn.Conv2d(3, 32, kernel_size=3, stride=2, padding=1, bias=False)
        self.bn1 = torch.nn.BatchNorm2d(32)
        self.relu = torch.nn.ReLU(inplace=True)

        # MBConv blocks (same as original)
        self.mbconv1 = self._make_mbconv_block(32, 96, 1, 3)
        self.mbconv2 = self._make_mbconv_block(96, 144, 2, 6)
        self.mbconv3 = self._make_mbconv_block(144, 192, 2, 6)
        self.mbconv4 = self._make_mbconv_block(192, 288, 2, 6)
        self.mbconv5 = self._make_mbconv_block(288, 384, 1, 6)

        # Final conv + bn
        self.conv_final = torch.nn.Conv2d(384, 1408, kernel_size=1, stride=1, padding=0, bias=False)
        self.bn_final = torch.nn.BatchNorm2d(1408)

        # Replace AdaptiveAvgPool2d with custom NPU kernel
        # self.avgpool = torch.nn.AdaptiveAvgPool2d((1, 1))  # replaced by efficientnet_b2_custom
        self.fc = torch.nn.Linear(1408, num_classes)

    def _make_mbconv_block(self, in_channels, out_channels, stride, expand_ratio):
        layers = []
        expanded_channels = in_channels * expand_ratio
        if expand_ratio != 1:
            layers.append(torch.nn.Conv2d(in_channels, expanded_channels, kernel_size=1, stride=1, padding=0, bias=False))
            layers.append(torch.nn.BatchNorm2d(expanded_channels))
            layers.append(torch.nn.ReLU(inplace=True))
        layers.append(torch.nn.Conv2d(expanded_channels, expanded_channels, kernel_size=3, stride=stride, padding=1, groups=expanded_channels, bias=False))
        layers.append(torch.nn.BatchNorm2d(expanded_channels))
        layers.append(torch.nn.ReLU(inplace=True))
        # Squeeze and Excitation
        layers.append(torch.nn.AdaptiveAvgPool2d((1, 1)))
        layers.append(torch.nn.Conv2d(expanded_channels, expanded_channels // 4, kernel_size=1, stride=1, padding=0, bias=False))
        layers.append(torch.nn.ReLU(inplace=True))
        layers.append(torch.nn.Conv2d(expanded_channels // 4, expanded_channels, kernel_size=1, stride=1, padding=0, bias=False))
        layers.append(torch.nn.Sigmoid())
        # Output phase
        layers.append(torch.nn.Conv2d(expanded_channels, out_channels, kernel_size=1, stride=1, padding=0, bias=False))
        layers.append(torch.nn.BatchNorm2d(out_channels))
        return torch.nn.Sequential(*layers)

    def forward(self, x):
        x = self.relu(self.bn1(self.conv1(x)))
        x = self.mbconv1(x)
        x = self.mbconv2(x)
        x = self.mbconv3(x)
        x = self.mbconv4(x)
        x = self.mbconv5(x)
        x = self.relu(self.bn_final(self.conv_final(x)))
        # Custom kernel replaces AdaptiveAvgPool2d((1,1))
        x = custom_ops_lib.efficientnet_b2_custom(x)
        x = torch.flatten(x, 1)
        x = self.fc(x)
        return x
'''
