static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(x_shape.GetShapeSize());

    uint32_t ndim = static_cast<uint32_t>(x_shape.GetDimNum());
    uint32_t n_cols = static_cast<uint32_t>(x_shape.GetDim(ndim - 1));
    uint32_t n_rows = totalLength / n_cols;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilon_ptr = attrs->GetAttrPointer<float>(0);
    float epsilon = (epsilon_ptr != nullptr) ? *epsilon_ptr : 1e-5f;

    uint32_t tileLen = n_cols;

    uint32_t blockDim = n_rows;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_n_rows(n_rows);
    tiling.set_n_cols(n_cols);
    tiling.set_blockDim(blockDim);
    tiling.set_epsilon(epsilon);
    tiling.set_invN(1.0f / static_cast<float>(n_cols)); // compute on host, avoid on-device cast

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t n_rows, uint32_t n_cols,
                                float epsilon, float invN)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        n_rows_ = n_rows;
        n_cols_ = n_cols;
        epsilon_ = epsilon;
        invN_ = invN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, n_cols_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, n_cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Allocate persistent local tensors for gamma and beta (not queued)
        gammaLocal_ = pipe_.AllocTensor<float>(n_cols_ * sizeof(float));
        betaLocal_ = pipe_.AllocTensor<float>(n_cols_ * sizeof(float));

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || n_cols_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Distribute rows across blocks
        const uint32_t rowsPerBlock = CeilDiv(n_rows_, blockNum);
        const uint32_t rowStart = blockIdx * rowsPerBlock;
        const uint32_t rowEnd = (rowStart + rowsPerBlock > n_rows_) ? n_rows_ : rowStart + rowsPerBlock;
        if (rowStart >= n_rows_) return;

        // Load gamma and beta once
        AscendC::DataCopy(gammaLocal_, gammaGm_, n_cols_);
        AscendC::DataCopy(betaLocal_, betaGm_, n_cols_);

        // Process each assigned row
        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            uint32_t rowOffset = row * n_cols_;
            CopyIn(rowOffset);
            Compute();
            CopyOut(rowOffset);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowOffset)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowOffset], n_cols_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Compute mean using scalar loop (invN avoids division by n_cols)
        float mean = 0.0f;
        for (uint32_t i = 0; i < n_cols_; ++i) {
            mean += xLocal.GetValue(i);
        }
        mean = mean * invN_;  // avoid division

        // Compute variance
        float var = 0.0f;
        for (uint32_t i = 0; i < n_cols_; ++i) {
            float diff = xLocal.GetValue(i) - mean;
            var += diff * diff;
        }
        var = var * invN_;  // avoid division

        float rstd = 1.0f / AscendC::Sqrt(var + epsilon_);

        // Normalize, scale, shift
        for (uint32_t i = 0; i < n_cols_; ++i) {
            float diff = xLocal.GetValue(i) - mean;
            float val = diff * rstd;
            val = val * gammaLocal_.GetValue(i) + betaLocal_.GetValue(i);
            yLocal.SetValue(i, val);
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t rowOffset)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[rowOffset], yLocal, n_cols_);
        outQueueY_.FreeTensor(yLocal);
    }

    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b)
    {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::LocalTensor<float> gammaLocal_;
    AscendC::LocalTensor<float> betaLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t n_rows_ {0};
    uint32_t n_cols_ {0};
    float epsilon_ {1e-5f};
    float invN_ {0.0f};   // 1.0f / n_cols_ (computed on host)
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n_rows, tiling_data.n_cols,
            tiling_data.epsilon, tiling_data.invN);
    op.Process();
}
"""

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        normalized_shape = init_inputs[0]
        self.register_buffer('gamma', torch.ones(normalized_shape))
        self.register_buffer('beta', torch.zeros(normalized_shape))

    def forward(self, *inputs):
        fn = getattr(custom_ops_lib, _FN)
        x = inputs[0]
        return fn(x, self.gamma, self.beta, 1e-5)
