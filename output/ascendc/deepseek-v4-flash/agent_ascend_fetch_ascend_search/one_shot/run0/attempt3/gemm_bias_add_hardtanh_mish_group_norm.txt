#include "kernel_operator.h"

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t totalB, uint32_t totalO,
                                uint32_t numGroups, float eps,
                                float hardtanhMin, float hardtanhMax,
                                uint32_t blockDim)
    {
        totalB_ = totalB;
        totalO_ = totalO;
        numGroups_ = numGroups;
        eps_ = eps;
        hardtanhMin_ = hardtanhMin;
        hardtanhMax_ = hardtanhMax;
        blockDim_ = blockDim;
        groupSize_ = totalO_ / numGroups_;
        invGroupSize_ = 1.0f / groupSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalB_ * totalO_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, totalO_);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, totalO_);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, totalO_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalB_ * totalO_);

        // Allocate local buffers: one row for x, one for y, one for tmp, plus bias, gamma, beta
        pipe_.InitBuffer(bufX_, totalO_ * sizeof(float));
        pipe_.InitBuffer(bufY_, totalO_ * sizeof(float));
        pipe_.InitBuffer(bufTmp_, totalO_ * sizeof(float));
        pipe_.InitBuffer(bufBias_, totalO_ * sizeof(float));
        pipe_.InitBuffer(bufGamma_, totalO_ * sizeof(float));
        pipe_.InitBuffer(bufBeta_, totalO_ * sizeof(float));

        xLocal_ = bufX_.Get<float>(0);
        yLocal_ = bufY_.Get<float>(0);
        tmpLocal_ = bufTmp_.Get<float>(0);
        biasLocal_ = bufBias_.Get<float>(0);
        gammaLocal_ = bufGamma_.Get<float>(0);
        betaLocal_ = bufBeta_.Get<float>(0);

        // Preload bias, gamma, beta (constant across rows)
        AscendC::DataCopy(biasLocal_, biasGm_, totalO_);
        AscendC::DataCopy(gammaLocal_, gnGammaGm_, totalO_);
        AscendC::DataCopy(betaLocal_, gnBetaGm_, totalO_);
    }

    __aicore__ inline void Process()
    {
        if (totalB_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t step = blockDim_;
        uint32_t startRow = blockIdx;
        uint32_t endRow = totalB_;

        for (uint32_t row = startRow; row < endRow; row += step) {
            // Load input row
            AscendC::DataCopy(xLocal_, xGm_[row * totalO_], totalO_);

            // Step 1: x = x + bias
            AscendC::Add(xLocal_, xLocal_, biasLocal_, totalO_);

            // Step 2: HardTanh
            AscendC::Maxs(xLocal_, xLocal_, hardtanhMin_, totalO_);
            AscendC::Mins(xLocal_, xLocal_, hardtanhMax_, totalO_);

            // Step 3: Mish
            AscendC::Exp(tmpLocal_, xLocal_, totalO_);              // tmp = exp(x)
            AscendC::Adds(tmpLocal_, tmpLocal_, 1.0f, totalO_);    // tmp = 1+exp(x)
            AscendC::Ln(tmpLocal_, tmpLocal_, totalO_);             // tmp = log(1+exp(x))
            AscendC::Tanh(tmpLocal_, tmpLocal_, totalO_);           // tmp = tanh(softplus)
            AscendC::Mul(xLocal_, xLocal_, tmpLocal_, totalO_);     // x = x * tanh(softplus)

            // Step 4: Group Normalization
            for (uint32_t g = 0; g < numGroups_; ++g) {
                uint32_t start = g * groupSize_;
                // Mean
                float sum = 0.0f;
                for (uint32_t i = start; i < start + groupSize_; ++i) {
                    sum += xLocal_(i);
                }
                float mean = sum * invGroupSize_;

                // Variance
                float var = 0.0f;
                for (uint32_t i = start; i < start + groupSize_; ++i) {
                    float diff = xLocal_(i) - mean;
                    var += diff * diff;
                }
                var = var * invGroupSize_ + eps_;
                var = sqrt(var);

                // Normalize, apply gamma and beta
                for (uint32_t i = start; i < start + groupSize_; ++i) {
                    float normalized = (xLocal_(i) - mean) / var;
                    xLocal_(i) = normalized * gammaLocal_(i) + betaLocal_(i);
                }
            }

            // Write output row
            AscendC::DataCopy(yGm_[row * totalO_], xLocal_, totalO_);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufX_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufY_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufTmp_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufBias_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufGamma_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bufBeta_;

    AscendC::LocalTensor<float> xLocal_;
    AscendC::LocalTensor<float> yLocal_;
    AscendC::LocalTensor<float> tmpLocal_;
    AscendC::LocalTensor<float> biasLocal_;
    AscendC::LocalTensor<float> gammaLocal_;
    AscendC::LocalTensor<float> betaLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalB_ {0};
    uint32_t totalO_ {0};
    uint32_t numGroups_ {1};
    uint32_t groupSize_ {0};
    float invGroupSize_ {0.0f};
    float eps_ {1e-5f};
    float hardtanhMin_ {-1.0f};
    float hardtanhMax_ {1.0f};
    uint32_t blockDim_ {1};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
    GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, bias, gn_gamma, gn_beta, y,
            tiling_data.totalB, tiling_data.totalO,
            tiling_data.numGroups, tiling_data.eps,
            tiling_data.hardtanhMin, tiling_data.hardtanhMax,
            tiling_data.blockDim);
    op.Process();
}
