host_operator_src="""

#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MAX_TILE_OUTPUT = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static int64_t ComputeOutSize(int64_t in, int64_t k, int64_t s, int64_t p, bool ceil_mode) {
    int64_t effective = in + 2 * p - k;
    if (ceil_mode) {
        return (effective + s - 1) / s + 1;
    } else {
        return effective / s + 1;
    }
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaxPooling3dCustomTilingData tiling;

    auto inShape = context->GetInputShape(0)->GetOriginShape();
    int32_t N = inShape.GetDim(0);
    int32_t C = inShape.GetDim(1);
    int32_t D = inShape.GetDim(2);
    int32_t H = inShape.GetDim(3);
    int32_t W = inShape.GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernel = *attrs->GetAttrPointer<int32_t>(0);
    int32_t stride = *attrs->GetAttrPointer<int32_t>(1);
    int32_t padding = *attrs->GetAttrPointer<int32_t>(2);
    bool ceilMode = *attrs->GetAttrPointer<bool>(3);

    if (stride == 0) stride = kernel;

    int32_t outD = ComputeOutSize(D, kernel, stride, padding, ceilMode);
    int32_t outH = ComputeOutSize(H, kernel, stride, padding, ceilMode);
    int32_t outW = ComputeOutSize(W, kernel, stride, padding, ceilMode);
    int32_t totalOutSize = N * C * outD * outH * outW;

    uint32_t tileLength = MAX_TILE_OUTPUT;
    if (tileLength > (uint32_t)totalOutSize) tileLength = totalOutSize;
    tileLength = AlignDown(tileLength, ALIGN_ELEMS);
    if (tileLength == 0) tileLength = totalOutSize;

    uint32_t numTiles = CeilDiv(totalOutSize, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_D(D);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_kernel(kernel);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_ceilMode(ceilMode);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_totalOutSize(totalOutSize);
    tiling.set_blockDim(blockDim);
    tiling.set_tileLength(tileLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_KERNEL = 7; // max kernel size 7

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
        GET_TILING_DATA(td, tiling);
        N_ = td.N;
        C_ = td.C;
        D_ = td.D;
        H_ = td.H;
        W_ = td.W;
        kernel_ = td.kernel;
        stride_ = td.stride;
        padding_ = td.padding;
        ceilMode_ = td.ceilMode;
        outD_ = td.outD;
        outH_ = td.outH;
        outW_ = td.outW;
        totalOutSize_ = td.totalOutSize;
        tileLen_ = td.tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * D_ * H_ * W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutSize_);

        // Allocate a pipe with a buffer for window data (max kernel³ floats)
        pipe_.InitBuffer(inWin_, BUFFER_NUM, MAX_KERNEL * MAX_KERNEL * MAX_KERNEL * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalOutSize_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Each block processes a contiguous chunk of output elements (flat order)
        uint32_t start = blockIdx * tileLen_;
        if (start >= totalOutSize_) return;
        uint32_t end = start + tileLen_;
        if (end > totalOutSize_) end = totalOutSize_;

        for (uint32_t idx = start; idx < end; ++idx) {
            // Convert flat index to output coords: (n, c, od, oh, ow)
            int32_t tmp = idx;
            int32_t ow = tmp % outW_; tmp /= outW_;
            int32_t oh = tmp % outH_; tmp /= outH_;
            int32_t od = tmp % outD_; tmp /= outD_;
            int32_t c = tmp % C_; tmp /= C_;
            int32_t n = tmp;

            // Compute input window start in each dimension
            int32_t in_d_start = od * stride_ - padding_;
            int32_t in_h_start = oh * stride_ - padding_;
            int32_t in_w_start = ow * stride_ - padding_;

            // Load window into local tensor
            AscendC::LocalTensor<float> win = inWin_.AllocTensor<float>();
            int32_t count = 0;
            float maxVal = -FLT_MAX;
            for (int32_t kd = 0; kd < kernel_; ++kd) {
                int32_t id = in_d_start + kd;
                if (id < 0 || id >= D_) continue;
                for (int32_t kh = 0; kh < kernel_; ++kh) {
                    int32_t ih = in_h_start + kh;
                    if (ih < 0 || ih >= H_) continue;
                    for (int32_t kw = 0; kw < kernel_; ++kw) {
                        int32_t iw = in_w_start + kw;
                        if (iw < 0 || iw >= W_) continue;
                        // Global index
                        uint32_t gidx = ((n * C_ + c) * D_ + id) * H_ * W_ + ih * W_ + iw;
                        win.SetValue(count, xGm_.GetValue(gidx));
                        count++;
                    }
                }
            }
            // Scalar max reduction
            float curMax = -FLT_MAX;
            for (int32_t i = 0; i < count; ++i) {
                float val = win.GetValue(i);
                if (val > curMax) curMax = val;
            }
            // Write output
            yGm_.SetValue(idx, curMax);
            inWin_.FreeTensor(win);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inWin_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    int32_t N_, C_, D_, H_, W_;
    int32_t kernel_, stride_, padding_;
    bool ceilMode_;
    int32_t outD_, outH_, outW_;
    int32_t totalOutSize_;
    int32_t tileLen_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    KernelMaxPooling3dCustom op;
    op.Init(x, y, workspace, tiling);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0, ceil_mode: bool = False):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride if stride is not None else kernel_size
        self.padding = padding
        self.ceil_mode = ceil_mode

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding, self.ceil_mode)
