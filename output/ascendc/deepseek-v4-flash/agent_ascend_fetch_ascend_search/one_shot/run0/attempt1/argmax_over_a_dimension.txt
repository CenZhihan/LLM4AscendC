host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MAX_OUTER_TILE = 8;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape *inShape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(inShape->GetOriginShape().GetShapeSize());
    uint32_t ndim = static_cast<uint32_t>(inShape->GetOriginShape().GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += ndim;  // normalize to positive
    if (dim < 0 || dim >= ndim) dim = ndim - 1;  // fallback to last dim

    // Compute outerSize = product of all dims except dim, reductionSize = size of dim
    uint32_t outerSize = 1;
    uint32_t reductionSize = 1;
    for (uint32_t i = 0; i < ndim; i++) {
        uint32_t s = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(i));
        if (i == static_cast<uint32_t>(dim))
            reductionSize = s;
        else
            outerSize *= s;
    }

    // Estimate per‑outer‑element work: load reductionSize floats, store one int64
    // UB budget: need buffer for floats (reductionSize), plus output int64s, plus temp for indices (if needed)
    uint32_t bytesPerOuter = reductionSize * sizeof(float) + sizeof(int64_t);
    uint32_t maxTileOuter = UB_BUDGET_BYTES / bytesPerOuter;
    if (maxTileOuter > MAX_OUTER_TILE) maxTileOuter = MAX_OUTER_TILE;
    if (maxTileOuter < 1) maxTileOuter = 1;
    uint32_t tileOuter = (outerSize < maxTileOuter) ? outerSize : maxTileOuter;

    uint32_t blockDim = CeilDiv(outerSize, tileOuter);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerSize(outerSize);
    tiling.set_reductionSize(reductionSize);
    tiling.set_blockDim(blockDim);
    tiling.set_tileOuter(tileOuter);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *inShape = context->GetInputShape(0);
    uint32_t ndim = static_cast<uint32_t>(inShape->GetOriginShape().GetDimNum());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= ndim) dim = ndim - 1;

    std::vector<int64_t> outDims;
    for (int64_t i = 0; i < ndim; i++) {
        if (i != dim) {
            outDims.push_back(inShape->GetOriginShape().GetDim(i));
        }
    }
    gert::Shape outShape(outDims);
    *context->GetOutputShape(0) = outShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class ArgmaxOverADimensionCustom : public OpDef {
public:
    explicit ArgmaxOverADimensionCustom(const char* name) : OpDef(name)
    {
        this->Input("x")

        this->Attr("dim").AttrType(OPTIONAL).Int(-1);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outerSize,
                                uint32_t reductionSize, uint32_t tileOuter)
    {
        totalLen_ = totalLength;
        outerSize_ = outerSize;
        reductionSize_ = reductionSize;
        tileOuter_ = tileOuter;
        blockDim_ = AscendC::GetBlockNum();

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, outerSize_);  // output size = outerSize

        // UB buffers: one tile of input floats and output indices
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileOuter_ * reductionSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOuter_ * sizeof(int64_t));
        pipe_.InitBuffer(tmpQueue_, 1, tileOuter_ * reductionSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outerSize_ == 0 || reductionSize_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        // grid stride over outer elements
        for (uint32_t baseOuter = blockIdx * tileOuter_; baseOuter < outerSize_; baseOuter += blockDim_ * tileOuter_) {
            uint32_t curTileOuter = tileOuter_;
            if (baseOuter + curTileOuter > outerSize_)
                curTileOuter = outerSize_ - baseOuter;

            // prefetch input for this tile
            uint32_t srcOffset = baseOuter * reductionSize_;
            CopyIn(srcOffset, curTileOuter);
            Compute(curTileOuter);
            CopyOut(baseOuter, curTileOuter);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t numOuter)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], numOuter * reductionSize_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t numOuter)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.AllocTensor<int64_t>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        for (uint32_t j = 0; j < numOuter; j++) {
            // pointer to the current outer's reduction chunk
            AscendC::LocalTensor<float> chunk = xLocal[j * reductionSize_];
            // use a built-in ArgMax (assumed to exist in AscendC)
            AscendC::ArgMax(tLocal, chunk, reductionSize_);  // returns index as float in tLocal[0]
            AscendC::Cast(yLocal[j], tLocal[0], AscendC::RoundMode::CAST_ROUND, 1);
        }

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<int64_t>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t numOuter)
    {
        AscendC::LocalTensor<int64_t> yLocal = outQueueY_.DeQue<int64_t>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, numOuter);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalLen_ {0};
    uint32_t outerSize_ {0};
    uint32_t reductionSize_ {0};
    uint32_t tileOuter_ {1};
    uint32_t blockDim_ {1};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.outerSize,
            tiling_data.reductionSize, tiling_data.tileOuter);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'argmax_over_a_dimension_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, dim: int):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Permute so that the reduction dimension becomes the last dimension.
        # The custom kernel always reduces the last dimension.
        dim = self.dim if self.dim >= 0 else x.dim() + self.dim
        perm = [i for i in range(x.dim()) if i != dim] + [dim]
        x_perm = x.permute(perm)
        fn = getattr(custom_ops_lib, _FN)
        out = fn(x_perm)
        # Output shape is input shape with the reduction dimension removed.
        return out
'''
