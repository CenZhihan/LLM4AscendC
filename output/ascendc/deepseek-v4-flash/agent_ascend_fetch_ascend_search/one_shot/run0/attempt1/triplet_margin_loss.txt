namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const gert::Shape *anchor_shape = context->GetInputShape(0);
    const auto &shape = anchor_shape->GetOriginShape();
    uint32_t totalSamples = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t featureDim   = static_cast<uint32_t>(shape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // Compute tileSamples: each sample uses two temporary vectors of size featureDim
    // (anchor-pos diff, anchor-neg diff) plus one for square. Total floats ~ 3*featureDim.
    uint32_t bytesPerSample = 3 * featureDim * sizeof(float);
    uint32_t maxSamplesPerTile = UB_BUDGET_BYTES / bytesPerSample;
    if (maxSamplesPerTile == 0) maxSamplesPerTile = 1;
    if (maxSamplesPerTile > totalSamples) maxSamplesPerTile = totalSamples;

    // Keep blockDim = 1 for simplicity (single-block reduction)
    context->SetBlockDim(1);

    tiling.set_totalSamples(totalSamples);
    tiling.set_featureDim(featureDim);
    tiling.set_tileSamples(maxSamplesPerTile);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR y, uint32_t totalSamples, uint32_t featureDim,
                                uint32_t tileSamples, float margin)
    {
        totalSamples_ = totalSamples;
        featureDim_   = featureDim;
        tileSamples_  = tileSamples;
        margin_       = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalSamples_ * featureDim_);
        posGm_.SetGlobalBuffer((__gm__ float*)positive, totalSamples_ * featureDim_);
        negGm_.SetGlobalBuffer((__gm__ float*)negative, totalSamples_ * featureDim_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        // Allocate per-tile buffers for one sample (we will process sample-by-sample within tile)
        pipe_.InitBuffer(anchorLocal_, BUFFER_NUM, featureDim_ * sizeof(float));
        pipe_.InitBuffer(posLocal_, BUFFER_NUM, featureDim_ * sizeof(float));
        pipe_.InitBuffer(negLocal_, BUFFER_NUM, featureDim_ * sizeof(float));
        pipe_.InitBuffer(diff1Local_, 1, featureDim_ * sizeof(float));
        pipe_.InitBuffer(diff2Local_, 1, featureDim_ * sizeof(float));
        pipe_.InitBuffer(sqLocal_, 1, featureDim_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalSamples_ == 0) return;

        float totalLoss = 0.0f;
        uint32_t processed = 0;

        while (processed < totalSamples_) {
            uint32_t cur = tileSamples_;
            if (cur > totalSamples_ - processed) cur = totalSamples_ - processed;

            for (uint32_t i = 0; i < cur; i++) {
                uint32_t idx = processed + i;

                // Load one sample (anchor, positive, negative rows)
                uint32_t offset = idx * featureDim_;
                AscendC::LocalTensor<float> anchorVec = anchorLocal_.AllocTensor<float>();
                AscendC::LocalTensor<float> posVec = posLocal_.AllocTensor<float>();
                AscendC::LocalTensor<float> negVec = negLocal_.AllocTensor<float>();
                AscendC::DataCopy(anchorVec, anchorGm_[offset], featureDim_);
                AscendC::DataCopy(posVec, posGm_[offset], featureDim_);
                AscendC::DataCopy(negVec, negGm_[offset], featureDim_);

                // diff1 = anchor - positive
                AscendC::LocalTensor<float> diff1Vec = diff1Local_.Get();
                AscendC::Sub(diff1Vec, anchorVec, posVec, featureDim_);
                // square, then reduce sum, then sqrt -> posDist
                AscendC::LocalTensor<float> sqVec = sqLocal_.Get();
                AscendC::Mul(sqVec, diff1Vec, diff1Vec, featureDim_);
                float posSqSum = AscendC::ReduceAdd(sqVec, featureDim_);  // returns sum as scalar
                float posDist = sqrtf(posSqSum);

                // diff2 = anchor - negative
                AscendC::LocalTensor<float> diff2Vec = diff2Local_.Get();
                AscendC::Sub(diff2Vec, anchorVec, negVec, featureDim_);
                AscendC::Mul(sqVec, diff2Vec, diff2Vec, featureDim_);
                float negSqSum = AscendC::ReduceAdd(sqVec, featureDim_);
                float negDist = sqrtf(negSqSum);

                float loss = fmaxf(0.0f, posDist - negDist + margin_);
                totalLoss += loss;

                // Free tensors
                anchorLocal_.FreeTensor(anchorVec);
                posLocal_.FreeTensor(posVec);
                negLocal_.FreeTensor(negVec);
            }
            processed += cur;
        }

        float meanLoss = totalLoss / static_cast<float>(totalSamples_);
        yGm_.SetValue(0, meanLoss);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> anchorLocal_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> posLocal_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> negLocal_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> diff1Local_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> diff2Local_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sqLocal_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> posGm_;
    AscendC::GlobalTensor<float> negGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalSamples_ {0};
    uint32_t featureDim_ {0};
    uint32_t tileSamples_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(
    GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y,
            tiling_data.totalSamples, tiling_data.featureDim,
            tiling_data.tileSamples, tiling_data.margin);
    op.Process();
}
