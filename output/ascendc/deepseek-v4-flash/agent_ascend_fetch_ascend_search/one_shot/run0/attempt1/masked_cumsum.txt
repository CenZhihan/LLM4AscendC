#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(x_shape->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;

    // Compute sequence length along dim
    const auto &shape = x_shape->GetOriginShape();
    int32_t ndim = shape.GetDimNum();
    if (dim < 0) dim += ndim;
    uint32_t seqLen = static_cast<uint32_t>(shape.GetDim(static_cast<uint32_t>(dim)));

    // Tile length: we need two buffers (x and mask) of tileLen floats each
    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));  // 2 * 4 = 8 bytes/elem
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = (seqLen < maxTile) ? seqLen : maxTile;

    // Block dimension: at most one block per sequence, but cap for efficiency
    uint32_t numSeqs = totalLength / seqLen;
    uint32_t blockDim = (numSeqs < 24) ? numSeqs : 24;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_seqLen(seqLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t seqLen,
                                uint32_t tileLen, uint32_t blockDim)
    {
        totalLen_ = totalLength;
        seqLen_ = seqLen;
        tileLen_ = tileLen;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(float));
        // output buffer not needed directly; we write out element by element
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || seqLen_ == 0 || tileLen_ == 0) return;

        uint32_t numSeqs = totalLen_ / seqLen_;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        // grid-stride over sequences
        for (uint32_t seq = blockIdx; seq < numSeqs; seq += blockDim_)
        {
            uint32_t base = seq * seqLen_;
            float curSum = 0.0f;

            // Process sequence in tiles
            uint32_t remaining = seqLen_;
            uint32_t offset = 0;
            while (remaining > 0)
            {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;

                // Load x and mask tiles (double buffered)
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
                AscendC::DataCopy(maskLocal, maskGm_[base + offset], curLen);
                inQueueX_.EnQue(xLocal);
                inQueueMask_.EnQue(maskLocal);

                // Dequeue and compute
                xLocal = inQueueX_.DeQue<float>();
                maskLocal = inQueueMask_.DeQue<float>();
                // element-wise product: xLocal = xLocal * maskLocal
                AscendC::Mul(xLocal, xLocal, maskLocal, curLen);

                // sequential cumulative sum over this tile
                for (uint32_t i = 0; i < curLen; i++)
                {
                    curSum += xLocal(i);
                    yGm_(base + offset + i) = curSum;
                }

                // free tensors
                inQueueX_.FreeTensor(xLocal);
                inQueueMask_.FreeTensor(maskLocal);

                offset += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t seqLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.totalLength, tiling_data.seqLen, tiling_data.tileLen, tiling_data.blockDim);
    op.Process();
}
