kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t totalB, uint32_t totalO,
                                uint32_t numGroups, float eps,
                                float hardtanhMin, float hardtanhMax,
                                uint32_t blockDim)
    {
        totalB_ = totalB;
        totalO_ = totalO;
        numGroups_ = numGroups;
        eps_ = eps;
        hardtanhMin_ = hardtanhMin;
        hardtanhMax_ = hardtanhMax;
        blockDim_ = blockDim;
        groupSize_ = totalO_ / numGroups_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalB_ * totalO_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, totalO_);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, totalO_);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, totalO_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalB_ * totalO_);

        // Buffers: one row for input/output, one for temporary
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, totalO_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, totalO_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, totalO_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalB_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t step = blockDim_;
        uint32_t startRow = blockIdx;
        uint32_t endRow = totalB_;

        for (uint32_t row = startRow; row < endRow; row += step) {
            // Load one row of x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[row * totalO_], totalO_);
            inQueueX_.EnQue(xLocal);

            // Allocate output and temporary
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

            // Step 1: x = x + bias
            AscendC::Add(xLocal, xLocal, biasGm_, totalO_);

            // Step 2: HardTanh
            AscendC::Maximums(xLocal, xLocal, hardtanhMin_, totalO_);
            AscendC::Minimums(xLocal, xLocal, hardtanhMax_, totalO_);

            // Step 3: Mish -> x * tanh(softplus(x))
            // softplus = log(1+exp(x))
            AscendC::Exp(tmp, xLocal, totalO_);               // tmp = exp(x)
            AscendC::Adds(tmp, tmp, 1.0f, totalO_);           // tmp = 1+exp(x)
            AscendC::Ln(tmp, tmp, totalO_);                   // tmp = log(1+exp(x))
            AscendC::Tanh(tmp, tmp, totalO_);                 // tmp = tanh(softplus)
            AscendC::Mul(xLocal, xLocal, tmp, totalO_);       // x = x * tanh(softplus)

            // Step 4: Group Normalization
            // For each group, normalize using mean and variance
            for (uint32_t g = 0; g < numGroups_; ++g) {
                uint32_t start = g * groupSize_;

                // Mean
                float sum = 0.0f;
                for (uint32_t i = start; i < start + groupSize_; ++i) {
                    sum += xLocal(i);
                }
                float mean = sum / static_cast<float>(groupSize_);

                // Variance
                float var = 0.0f;
                for (uint32_t i = start; i < start + groupSize_; ++i) {
                    float diff = xLocal(i) - mean;
                    var += diff * diff;
                }
                var = var / static_cast<float>(groupSize_) + eps_;
                var = sqrtf(var);

                // Normalize, then apply gamma and beta
                for (uint32_t i = start; i < start + groupSize_; ++i) {
                    float normalized = (xLocal(i) - mean) / var;
                    xLocal(i) = normalized * gnGammaGm_(i) + gnBetaGm_(i);
                }
            }

            // Write output row
            AscendC::DataCopy(yGm_[row * totalO_], xLocal, totalO_);

            // Free tensors
            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            tmpQueue_.FreeTensor(tmp);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalB_ {0};
    uint32_t totalO_ {0};
    uint32_t numGroups_ {1};
    uint32_t groupSize_ {0};
    float eps_ {1e-5f};
    float hardtanhMin_ {-1.0f};
    float hardtanhMax_ {1.0f};
    uint32_t blockDim_ {1};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
    GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, bias, gn_gamma, gn_beta, y,
            tiling_data.totalB, tiling_data.totalO,
            tiling_data.numGroups, tiling_data.eps,
            tiling_data.hardtanhMin, tiling_data.hardtanhMax,
            tiling_data.blockDim);
    op.Process();
}

import torch.nn.functional as F
import torch_npu
import custom_ops_lib

_FN = "gemm_bias_add_hardtanh_mish_group_norm_custom"

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, bias_shape, num_groups):
        super(ModelNew, self).__init__()
        # We keep weight and lin_bias as parameters? 
        # For simplicity, we assume linear is done externally (e.g., using F.linear)
        # The custom kernel only does bias_add + hardtanh + mish + group_norm
        # So we store num_groups, but weight/lin_bias are not used.
        self.num_groups = num_groups

    def forward(self, x, weight, lin_bias, bias, gn_gamma, gn_beta):
        # First compute linear using PyTorch
        x = F.linear(x, weight, lin_bias)
        # Then apply the custom kernel for the rest
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, bias, gn_gamma, gn_beta,
                  num_groups=self.num_groups,
                  eps=1e-5,
                  hardtanh_min=-1.0,
