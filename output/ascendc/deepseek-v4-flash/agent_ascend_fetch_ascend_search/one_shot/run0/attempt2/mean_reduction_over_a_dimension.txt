#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cstdint>
#include <algorithm>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;
static constexpr uint32_t MAX_NDIM = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;
    const gert::StorageShape *inShape = context->GetInputShape(0);
    const gert::Shape &originShape = inShape->GetOriginShape();
    uint32_t ndim = static_cast<uint32_t>(originShape.GetDimNum());
    if (ndim > MAX_NDIM) ndim = MAX_NDIM; // defensive

    uint32_t dims[MAX_NDIM];
    for (uint32_t i = 0; i < ndim; ++i) {
        dims[i] = static_cast<uint32_t>(originShape.GetDim(i));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;
    if (dim < 0) dim += ndim;

    uint32_t preSize = 1;
    for (uint32_t i = 0; i < static_cast<uint32_t>(dim); ++i) preSize *= dims[i];
    uint32_t reduceSize = dims[dim];
    uint32_t postSize = 1;
    for (uint32_t i = static_cast<uint32_t>(dim) + 1; i < ndim; ++i) postSize *= dims[i];

    uint32_t totalInputLen = preSize * reduceSize * postSize;
    uint32_t totalOutputLen = preSize * postSize;

    // tileOutput from UB: need 2 * tileOutput floats (one sum, one copy)
    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileOutput = maxTile;
    if (totalOutputLen < tileOutput) {
        uint32_t aligned = AlignDown(totalOutputLen, ALIGN_ELEMS);
        tileOutput = (aligned > 0) ? aligned : totalOutputLen;
    }
    if (tileOutput == 0) tileOutput = 1;

    uint32_t numTiles = CeilDiv(totalOutputLen, tileOutput);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalInputLen(totalInputLen);
    tiling.set_totalOutputLen(totalOutputLen);
    tiling.set_reduceSize(reduceSize);
    tiling.set_preSize(preSize);
    tiling.set_postSize(postSize);
    tiling.set_tileOutput(tileOutput);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t ndim = x_shape->GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;
    if (dim < 0) dim += ndim;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(ndim - 1);
    uint32_t idx = 0;

            y_shape->SetDim(idx++, x_shape->GetDim(i));
        }
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputLen, uint32_t totalOutputLen,
                                uint32_t reduceSize, uint32_t preSize, uint32_t postSize,
                                uint32_t tileOutput)
    {
        totalInputLen_ = totalInputLen;
        totalOutputLen_ = totalOutputLen;
        reduceSize_ = reduceSize;
        preSize_ = preSize;
        postSize_ = postSize;
        tileOutput_ = tileOutput;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileOutput_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOutput_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileOutput_ * sizeof(float));
        pipe_.InitBuffer(sumQueue_, BUFFER_NUM, tileOutput_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputLen_ == 0 || tileOutput_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideOutput = blockNum * tileOutput_;
        uint32_t outputBase = blockIdx * tileOutput_;
        if (outputBase >= totalOutputLen_) return;

        while (outputBase < totalOutputLen_) {
            uint32_t curLen = totalOutputLen_ - outputBase;
            if (curLen > tileOutput_) curLen = tileOutput_;

            ComputeAndStore(outputBase, curLen);

            outputBase += strideOutput;
        }
    }

private:
    __aicore__ inline void ComputeAndStore(uint32_t outputBase, uint32_t len)
    {
        // outputBase is flat index into output tensor
        // output shape: (preSize, postSize)
        // outputBase = p * postSize + q_start
        uint32_t p = outputBase / postSize_;
        uint32_t q_start = outputBase % postSize_;

        AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // zero sum
        AscendC::SetZero(sumLocal, len);

        for (uint32_t i = 0; i < reduceSize_; ++i) {
            // input offset for this slice:
            // p * (reduceSize * postSize) + i * postSize + q_start
            uint32_t inputOffset = p * (reduceSize_ * postSize_) + i * postSize_ + q_start;

            AscendC::DataCopy(xLocal, xGm_[inputOffset], len);
            AscendC::Add(sumLocal, sumLocal, xLocal, len);
        }

        // divide by reduceSize
        float invReduceSize = 1.0f / static_cast<float>(reduceSize_);
        AscendC::Muls(tLocal, sumLocal, invReduceSize, len);

        // store to output
        AscendC::DataCopy(yGm_[outputBase], tLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        inQueueX_.FreeTensor(xLocal);
        sumQueue_.FreeTensor(sumLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> sumQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputLen_ {0};
    uint32_t totalOutputLen_ {0};
    uint32_t reduceSize_ {0};
    uint32_t preSize_ {0};
    uint32_t postSize_ {0};
    uint32_t tileOutput_ {0};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalInputLen, tiling_data.totalOutputLen,
            tiling_data.reduceSize, tiling_data.preSize, tiling_data.postSize,
            tiling_data.tileOutput);
    op.Process();
}
