host_operator_src="""

#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// Budget per core for UB (conservative)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_UNIT = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    // Use StorageShape* (returned by GetInputShape in TilingContext)
    const gert::StorageShape* aShape = context->GetInputShape(0);
    const gert::StorageShape* bShape = context->GetInputShape(1);
    uint32_t M = static_cast<uint32_t>(aShape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(aShape->GetDim(1));
    uint32_t N = static_cast<uint32_t>(bShape->GetDim(1));

    uint32_t tileK = (K < 1024) ? K : AlignDown(1024, ALIGN_UNIT);
    uint32_t tileM;
    uint32_t tileN;

    if (M >= N) {
        tileN = N;
        uint32_t maxTileM = (UB_BUDGET_BYTES / 4) / (K + N);
        if (maxTileM < ALIGN_UNIT) maxTileM = ALIGN_UNIT;
        tileM = (maxTileM > M) ? M : AlignDown(maxTileM, ALIGN_UNIT);
    } else {
        tileM = M;
        uint32_t maxTileN = (UB_BUDGET_BYTES / 4) / (K + M);
        if (maxTileN < ALIGN_UNIT) maxTileN = ALIGN_UNIT;
        tileN = (maxTileN > N) ? N : AlignDown(maxTileN, ALIGN_UNIT);
    }

    if (tileM == 0) tileM = ALIGN_UNIT;
    if (tileN == 0) tileN = ALIGN_UNIT;
    if (tileK == 0) tileK = ALIGN_UNIT;

    uint32_t numTilesM = CeilDiv(M, tileM);
    uint32_t numTilesN = CeilDiv(N, tileN);
    uint32_t blockDim = numTilesM * numTilesN;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // GetInputShape in InferShapeContext returns const gert::Shape*
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    uint32_t M = a_shape->GetDim(0);
    uint32_t N = b_shape->GetDim(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        M_ = M; K_ = K; N_ = N;
        tileM_ = tileM; tileK_ = tileK; tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)a, M * K);
        bGm_.SetGlobalBuffer((__gm__ float*)b, K * N);
        cGm_.SetGlobalBuffer((__gm__ float*)c, M * N);

        uint32_t aTileSize = tileM_ * tileK_ * sizeof(float);
        uint32_t bTileSize = tileK_ * tileN_ * sizeof(float);
        uint32_t cTileSize = tileM_ * tileN_ * sizeof(float);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, aTileSize);
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, bTileSize);
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, cTileSize);
        pipe_.InitBuffer(tmpQueue_, 1, cTileSize);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t numTilesM = (M_ + tileM_ - 1) / tileM_;
        uint32_t numTilesN = (N_ + tileN_ - 1) / tileN_;
        uint32_t totalTiles = numTilesM * numTilesN;
        if (blockIdx >= totalTiles) return;
        uint32_t mTile = blockIdx / numTilesN;
        uint32_t nTile = blockIdx % numTilesN;

        uint32_t mStart = mTile * tileM_;
        uint32_t nStart = nTile * tileN_;
        uint32_t curTileM = (mStart + tileM_ > M_) ? M_ - mStart : tileM_;
        uint32_t curTileN = (nStart + tileN_ > N_) ? N_ - nStart : tileN_;

        AscendC::LocalTensor<float> acc = tmpQueue_.AllocTensor<float>();
        AscendC::Duplicate(acc, 0.0f, curTileM * curTileN);

        for (uint32_t kBase = 0; kBase < K_; kBase += tileK_) {
            uint32_t curTileK = (kBase + tileK_ > K_) ? K_ - kBase : tileK_;

            // Load A tile (curTileM x curTileK)
            AscendC::LocalTensor<float> aTile = inQueueA_.AllocTensor<float>();
            for (uint32_t i = 0; i < curTileM; i++) {
                uint32_t gmOffset = ((mStart + i) * K_) + kBase;
                uint32_t localOffset = i * curTileK;
                AscendC::DataCopy(aTile[localOffset], aGm_[gmOffset], curTileK);
            }
            inQueueA_.EnQue(aTile);

            // Load B tile (curTileK x curTileN)
            AscendC::LocalTensor<float> bTile = inQueueB_.AllocTensor<float>();
            for (uint32_t j = 0; j < curTileK; j++) {
                uint32_t gmOffset = ((kBase + j) * N_) + nStart;
                uint32_t localOffset = j * curTileN;
                AscendC::DataCopy(bTile[localOffset], bGm_[gmOffset], curTileN);
            }
            inQueueB_.EnQue(bTile);

            aTile = inQueueA_.DeQue<float>();
            bTile = inQueueB_.DeQue<float>();

            // acc += aTile * bTile (simple triple loop over K)
            for (uint32_t kk = 0; kk < curTileK; kk++) {
                for (uint32_t i = 0; i < curTileM; i++) {
                    float aVal = aTile(i * curTileK + kk);
                    for (uint32_t j = 0; j < curTileN; j++) {
                        float bVal = bTile(kk * curTileN + j);
                        acc(i * curTileN + j) += aVal * bVal;
                    }
                }
            }

            inQueueA_.FreeTensor(aTile);
            inQueueB_.FreeTensor(bTile);
        }

        // Write result to global memory
        for (uint32_t i = 0; i < curTileM; i++) {
            uint32_t localOffset = i * curTileN;
            uint32_t gmOffset = ((mStart + i) * N_) + nStart;
            AscendC::DataCopy(cGm_[gmOffset], acc[localOffset], curTileN);
        }
        tmpQueue_.FreeTensor(acc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {0};
    uint32_t tileK_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
