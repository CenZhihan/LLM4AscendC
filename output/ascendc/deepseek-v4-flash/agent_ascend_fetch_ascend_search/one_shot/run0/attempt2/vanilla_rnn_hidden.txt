#include "vanilla_rnn_hidden_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    VanillaRnnHiddenCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape *h_shape = context->GetInputShape(1)->GetOriginShape();
    const gert::Shape *w_i2h_shape = context->GetInputShape(2)->GetOriginShape();
    const gert::Shape *w_h2o_shape = context->GetInputShape(4)->GetOriginShape();

    uint32_t B = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t I = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t H = static_cast<uint32_t>(w_i2h_shape->GetDim(0));
    uint32_t O = static_cast<uint32_t>(w_h2o_shape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(w_i2h_shape->GetDim(1));

    // Distribute batch across cores
    uint32_t blockDim = context->GetCoreNum();
    if (blockDim > B) blockDim = B;
    context->SetBlockDim(blockDim);

    uint32_t batchPerBlock = B / blockDim;
    uint32_t remainingBatch = B % blockDim;

    tiling.set_B(B);
    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_batchPerBlock(batchPerBlock);
    tiling.set_remainingBatch(remainingBatch);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *h_shape = context->GetInputShape(1);
    const gert::Shape *w_h2o_shape = context->GetInputShape(4);

    gert::Shape *h_new_shape = context->GetOutputShape(0);
    *h_new_shape = *h_shape;

    gert::Shape *y_shape = context->GetOutputShape(1);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, h_shape->GetDim(0));                  // B
    y_shape->SetDim(1, static_cast<int64_t>(w_h2o_shape->GetDim(0))); // O

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    context->SetOutputDataType(1, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge
