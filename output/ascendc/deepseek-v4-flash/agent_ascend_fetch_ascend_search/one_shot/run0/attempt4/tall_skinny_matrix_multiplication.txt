host_operator_src="""

#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_UNIT = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape &aOriginShape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape &bOriginShape = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = static_cast<uint32_t>(aOriginShape.GetDim(0));
    uint32_t K = static_cast<uint32_t>(aOriginShape.GetDim(1));
    uint32_t N = static_cast<uint32_t>(bOriginShape.GetDim(1));

    uint32_t tileK = (K < 1024) ? K : AlignDown(1024, ALIGN_UNIT);
    uint32_t tileM;
    uint32_t tileN;

    if (M >= N) {
        tileN = N;
        uint32_t maxTileM = (UB_BUDGET_BYTES / 4) / (K + N);
        if (maxTileM < ALIGN_UNIT) maxTileM = ALIGN_UNIT;
        tileM = (maxTileM > M) ? M : AlignDown(maxTileM, ALIGN_UNIT);
    } else {
        tileM = M;
        uint32_t maxTileN = (UB_BUDGET_BYTES / 4) / (K + M);
        if (maxTileN < ALIGN_UNIT) maxTileN = ALIGN_UNIT;
        tileN = (maxTileN > N) ? N : AlignDown(maxTileN, ALIGN_UNIT);
    }

    if (tileM == 0) tileM = ALIGN_UNIT;
    if (tileN == 0) tileN = ALIGN_UNIT;
    if (tileK == 0) tileK = ALIGN_UNIT;

    uint32_t numTilesM = CeilDiv(M, tileM);
    uint32_t numTilesN = CeilDiv(N, tileN);
    uint32_t blockDim = numTilesM * numTilesN;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    uint32_t M = a_shape->GetDim(0);
    uint32_t N = b_shape->GetDim(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class TallSkinnyMatrixMultiplicationCustom : public OpDef {
public:
    explicit TallSkinnyMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("a")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("b")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("c")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(TallSkinnyMatrixMultiplicationCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include "matmul.h"

using namespace AscendC;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        M_ = M; K_ = K; N_ = N;
        tileM_ = tileM; tileK_ = tileK; tileN_ = tileN;
        aGm_.SetGlobalBuffer((__gm__ float*)a, M * K);
        bGm_.SetGlobalBuffer((__gm__ float*)b, K * N);
        cGm_.SetGlobalBuffer((__gm__ float*)c, M * N);
    }

    __aicore__ inline void Process()
    {
        MatMul<float, float, float> matMul;
        matMul.SetSrcA(aGm_, K_);
        matMul.SetSrcB(bGm_, N_);
        matMul.SetDst(cGm_, N_);
        matMul.SetTranspose(false, false);
        matMul.SetBatch(1);
        matMul.SetTile(tileM_, tileK_, tileN_);
        matMul.SetMode(MatMulMode::NO_BIAS);
        matMul.IterateAll();
    }

private:
    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;
    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {0};
    uint32_t tileK_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(a, b, c, tiling_data.M, tiling_data.K, tiling_data.N, tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
