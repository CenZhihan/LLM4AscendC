#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_FLOATS = 192 * 1024 / 4; // 49152
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(x_shape.GetShapeSize());

    uint32_t ndim = static_cast<uint32_t>(x_shape.GetDimNum());
    uint32_t n_cols = static_cast<uint32_t>(x_shape.GetDim(ndim - 1));
    uint32_t n_rows = totalLength / n_cols;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilon_ptr = attrs->GetAttrPointer<float>(0);
    float epsilon = (epsilon_ptr != nullptr) ? *epsilon_ptr : 1e-5f;

    uint32_t tileLen = n_cols;

    uint32_t blockDim = n_rows;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_n_rows(n_rows);
    tiling.set_n_cols(n_cols);
    tiling.set_blockDim(blockDim);
    tiling.set_epsilon(epsilon);
    tiling.set_invN(1.0f / static_cast<float>(n_cols));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
