host_operator_src="""

#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const gert::StorageShape *anchor_storage = context->GetInputShape(0);
    const auto &shape = anchor_storage->GetOriginShape();
    uint32_t totalSamples = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t featureDim   = static_cast<uint32_t>(shape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // Compute tileSamples: each sample uses temporary queues for diffPos, diffNeg,
    // sqPos, sqNeg, loss. With double buffering we need 2 copies of diffPos/diffNeg
    // (each size featureDim) and 2 copies of sqPos/sqNeg/loss (each size tileSamples).
    // Total floats ≈ 2*featureDim*2 + 3*tileSamples.
    // We approximate bytes per tile: (2*featureDim*2 + 3*tileSamples) * 4.
    // For simplicity, use a conservative estimate: (4*featureDim + 3*tileSamples) * 4.
    // Since tileSamples is variable, we solve a quadratic-like budget.
    // Heuristic: start with maxSamples = totalSamples, reduce until fits.
    uint32_t maxSamples = totalSamples;
    while (maxSamples > 0) {
        uint32_t bytesNeeded = (4u * featureDim + 3u * maxSamples) * static_cast<uint32_t>(sizeof(float));
        if (bytesNeeded <= UB_BUDGET_BYTES) break;
        maxSamples = maxSamples / 2;
    }
    if (maxSamples == 0) maxSamples = 1;

    // Use single block for simplicity (no cross-block reduction)
    context->SetBlockDim(1);

    tiling.set_totalSamples(totalSamples);
    tiling.set_featureDim(featureDim);
    tiling.set_tileSamples(maxSamples);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR y,
                                uint32_t totalSamples, uint32_t featureDim,
                                uint32_t tileSamples, float margin)
    {
        totalSamples_ = totalSamples;
        featureDim_   = featureDim;
        tileSamples_  = tileSamples;
        margin_       = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalSamples * featureDim);
        posGm_.SetGlobalBuffer((__gm__ float*)positive, totalSamples * featureDim);
        negGm_.SetGlobalBuffer((__gm__ float*)negative, totalSamples * featureDim);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        // Input queues (double buffered)
        pipe_.InitBuffer(inQueueAnchor_, BUFFER_NUM, tileSamples * featureDim * sizeof(float));
        pipe_.InitBuffer(inQueuePos_,   BUFFER_NUM, tileSamples * featureDim * sizeof(float));
        pipe_.InitBuffer(inQueueNeg_,   BUFFER_NUM, tileSamples * featureDim * sizeof(float));

        // Temporary queues for per-sample processing (single buffer each)
        pipe_.InitBuffer(tmpDiffPos_, 1, tileSamples * featureDim * sizeof(float));
        pipe_.InitBuffer(tmpDiffNeg_, 1, tileSamples * featureDim * sizeof(float));
        pipe_.InitBuffer(tmpSqPos_,   1, tileSamples * sizeof(float));
        pipe_.InitBuffer(tmpSqNeg_,   1, tileSamples * sizeof(float));
        pipe_.InitBuffer(tmpLoss_,    1, tileSamples * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalSamples_ == 0 || tileSamples_ == 0) return;

        float totalLoss = 0.0f;
        uint32_t offset = 0;

        while (offset < totalSamples_) {
            uint32_t curSamples = totalSamples_ - offset;
            if (curSamples > tileSamples_) curSamples = tileSamples_;

            CopyIn(offset, curSamples);
            ComputeAndAccumulate(totalLoss, curSamples);

            offset += curSamples;
        }

        // Write scalar result to global memory
        float avgLoss = totalLoss / static_cast<float>(totalSamples_);
        AscendC::LocalTensor<float> scalarTensor = scalarQueue_.AllocTensor<float>();
        scalarTensor.SetValue(0, avgLoss);
        AscendC::DataCopy(yGm_, scalarTensor, 1);
        scalarQueue_.FreeTensor(scalarTensor);
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t numSamples)
    {
        uint32_t len = numSamples * featureDim_;
        AscendC::LocalTensor<float> anchorLocal = inQueueAnchor_.AllocTensor<float>();
        AscendC::DataCopy(anchorLocal, anchorGm_[offset * featureDim_], len);
        inQueueAnchor_.EnQue(anchorLocal);

        AscendC::LocalTensor<float> posLocal = inQueuePos_.AllocTensor<float>();
        AscendC::DataCopy(posLocal, posGm_[offset * featureDim_], len);
        inQueuePos_.EnQue(posLocal);

        AscendC::LocalTensor<float> negLocal = inQueueNeg_.AllocTensor<float>();
        AscendC::DataCopy(negLocal, negGm_[offset * featureDim_], len);
        inQueueNeg_.EnQue(negLocal);
    }

    __aicore__ inline void ComputeAndAccumulate(float &totalLoss, uint32_t numSamples)
    {
        AscendC::LocalTensor<float> anchorLocal = inQueueAnchor_.DeQue<float>();
        AscendC::LocalTensor<float> posLocal    = inQueuePos_.DeQue<float>();
        AscendC::LocalTensor<float> negLocal    = inQueueNeg_.DeQue<float>();

        AscendC::LocalTensor<float> diffPos = tmpDiffPos_.AllocTensor<float>();
        AscendC::LocalTensor<float> diffNeg = tmpDiffNeg_.AllocTensor<float>();
        AscendC::LocalTensor<float> sqPos   = tmpSqPos_.AllocTensor<float>();
        AscendC::LocalTensor<float> sqNeg   = tmpSqNeg_.AllocTensor<float>();
        AscendC::LocalTensor<float> loss    = tmpLoss_.AllocTensor<float>();

        uint32_t len = numSamples * featureDim_;

        // diffPos = anchor - positive
        AscendC::Sub(diffPos, anchorLocal, posLocal, len);
        // diffNeg = anchor - negative
        AscendC::Sub(diffNeg, anchorLocal, negLocal, len);

        // Initialize sqPos and sqNeg to zero
        AscendC::Muls(sqPos, sqPos, 0.0f, numSamples);
        AscendC::Muls(sqNeg, sqNeg, 0.0f, numSamples);

        // Accumulate squared elements per sample (iterate over feature dimensions)
        for (uint32_t i = 0; i < featureDim_; i++) {
            // Extract i-th feature column from diffPos and diffNeg
            // The tensors are row-major, column i is at offset i.
            AscendC::LocalTensor<float> diffPosCol = diffPos[i];
            AscendC::LocalTensor<float> diffNegCol = diffNeg[i];
            // Square the column elements
            AscendC::Mul(diffPosCol, diffPosCol, diffPosCol, numSamples);
            AscendC::Mul(diffNegCol, diffNegCol, diffNegCol, numSamples);
            // Accumulate into sqPos and sqNeg
            AscendC::Add(sqPos, sqPos, diffPosCol, numSamples);
            AscendC::Add(sqNeg, sqNeg, diffNegCol, numSamples);
        }

        // loss = margin + sqPos - sqNeg, then clamp to 0 (max with scalar)
        AscendC::Sub(loss, sqPos, sqNeg, numSamples);
        AscendC::Adds(loss, loss, margin_, numSamples);
        AscendC::Maxs(loss, loss, 0.0f, numSamples);

        // Accumulate totalLoss: sum over samples
        float batchLoss = 0.0f;
        for (uint32_t i = 0; i < numSamples; i++) {
            batchLoss += loss.GetValue(i);
        }
        totalLoss += batchLoss;

        // Free temporary buffers
        tmpDiffPos_.FreeTensor(diffPos);
        tmpDiffNeg_.FreeTensor(diffNeg);
        tmpSqPos_.FreeTensor(sqPos);
        tmpSqNeg_.FreeTensor(sqNeg);
        tmpLoss_.FreeTensor(loss);

        // Free input queues
        inQueueAnchor_.FreeTensor(anchorLocal);
        inQueuePos_.FreeTensor(posLocal);
        inQueueNeg_.FreeTensor(negLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePos_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNeg_;

    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDiffPos_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDiffNeg_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSqPos_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpSqNeg_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpLoss_;

    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> posGm_;
    AscendC::GlobalTensor<float> negGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalSamples_ {0};
    uint32_t featureDim_   {0};
    uint32_t tileSamples_  {0};
    float margin_          {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                  GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y,
            tiling_data.totalSamples, tiling_data.featureDim,
            tiling_data.tileSamples, tiling_data.margin);
    op.Process();
}
