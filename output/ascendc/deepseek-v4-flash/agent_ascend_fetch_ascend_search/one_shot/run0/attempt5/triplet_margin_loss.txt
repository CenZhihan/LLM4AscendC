#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const gert::StorageShape *anchor_storage = context->GetInputShape(0);
    const auto &shape = anchor_storage->GetOriginShape();
    uint32_t totalSamples = static_cast<uint32_t>(shape.GetDim(0));
    uint32_t featureDim   = static_cast<uint32_t>(shape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // Simple tiling: process all samples at once if possible.
    // Each input tile: anchor, positive, negative each of size tileSamples*featureDim*4 bytes.
    // Using double buffering: 2 * (tileSamples*featureDim*4) per input queue => 3 * 2 * tileSamples*featureDim*4 = 24*tileSamples*featureDim bytes.
    // UB budget ~192KB.
    uint32_t maxSamples = totalSamples;
    while (maxSamples > 0) {
        uint32_t bytesNeeded = 24u * maxSamples * featureDim * static_cast<uint32_t>(sizeof(float));
        if (bytesNeeded <= UB_BUDGET_BYTES) break;
        maxSamples = maxSamples / 2;
    }
    if (maxSamples == 0) maxSamples = 1;

    // Use single block (no cross-block reduction)
    context->SetBlockDim(1);

    tiling.set_totalSamples(totalSamples);
    tiling.set_featureDim(featureDim);
    tiling.set_tileSamples(maxSamples);
    tiling.set_margin(*margin);
    tiling.set_invTotalSamples(1.0f / static_cast<float>(totalSamples));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    y_shape->SetDimNum(1);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR y,
                                uint32_t totalSamples, uint32_t featureDim,
                                uint32_t tileSamples, float margin,
                                float invTotalSamples)
    {
        totalSamples_   = totalSamples;
        featureDim_     = featureDim;
        tileSamples_    = tileSamples;
        margin_         = margin;
        invTotalSamples_ = invTotalSamples;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalSamples * featureDim);
        posGm_.SetGlobalBuffer((__gm__ float*)positive, totalSamples * featureDim);
        negGm_.SetGlobalBuffer((__gm__ float*)negative, totalSamples * featureDim);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        // Input queues (double buffered)
        pipe_.InitBuffer(inQueueAnchor_, BUFFER_NUM, tileSamples * featureDim * sizeof(float));
        pipe_.InitBuffer(inQueuePos_,   BUFFER_NUM, tileSamples * featureDim * sizeof(float));
        pipe_.InitBuffer(inQueueNeg_,   BUFFER_NUM, tileSamples * featureDim * sizeof(float));

        // Scalar queue for output
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalSamples_ == 0 || tileSamples_ == 0) return;

        float totalLoss = 0.0f;
        uint32_t offset = 0;

        while (offset < totalSamples_) {
            uint32_t curSamples = totalSamples_ - offset;
            if (curSamples > tileSamples_) curSamples = tileSamples_;

            CopyIn(offset, curSamples);
            totalLoss += ComputeTile(curSamples);

            offset += curSamples;
        }

        // Write average loss to global memory
        float avgLoss = totalLoss * invTotalSamples_;
        AscendC::LocalTensor<float> outScalar = scalarQueue_.AllocTensor<float>();
        outScalar.SetValue(0, avgLoss);
        AscendC::DataCopy(yGm_, outScalar, 1);
        scalarQueue_.FreeTensor(outScalar);
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t numSamples)
    {
        uint32_t len = numSamples * featureDim_;
        AscendC::LocalTensor<float> anchorLocal = inQueueAnchor_.AllocTensor<float>();
        AscendC::DataCopy(anchorLocal, anchorGm_[offset * featureDim_], len);
        inQueueAnchor_.EnQue(anchorLocal);

        AscendC::LocalTensor<float> posLocal = inQueuePos_.AllocTensor<float>();
        AscendC::DataCopy(posLocal, posGm_[offset * featureDim_], len);
        inQueuePos_.EnQue(posLocal);

        AscendC::LocalTensor<float> negLocal = inQueueNeg_.AllocTensor<float>();
        AscendC::DataCopy(negLocal, negGm_[offset * featureDim_], len);
        inQueueNeg_.EnQue(negLocal);
    }

    __aicore__ inline float ComputeTile(uint32_t numSamples)
    {
        AscendC::LocalTensor<float> anchorLocal = inQueueAnchor_.DeQue<float>();
        AscendC::LocalTensor<float> posLocal    = inQueuePos_.DeQue<float>();
        AscendC::LocalTensor<float> negLocal    = inQueueNeg_.DeQue<float>();

        float tileLoss = 0.0f;

        for (uint32_t s = 0; s < numSamples; s++) {
            float sqPos = 0.0f;
            float sqNeg = 0.0f;
            uint32_t base = s * featureDim_;
            for (uint32_t f = 0; f < featureDim_; f++) {
                float a = anchorLocal.GetValue(base + f);
                float p = posLocal.GetValue(base + f);
                float n = negLocal.GetValue(base + f);
                float diffPos = a - p;
                float diffNeg = a - n;
                sqPos += diffPos * diffPos;
                sqNeg += diffNeg * diffNeg;
            }
            float loss = margin_ + sqPos - sqNeg;
            if (loss < 0.0f) loss = 0.0f;
            tileLoss += loss;
        }

        inQueueAnchor_.FreeTensor(anchorLocal);
        inQueuePos_.FreeTensor(posLocal);
        inQueueNeg_.FreeTensor(negLocal);

        return tileLoss;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePos_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNeg_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> posGm_;
    AscendC::GlobalTensor<float> negGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalSamples_ {0};
    uint32_t featureDim_   {0};
    uint32_t tileSamples_  {0};
    float margin_          {1.0f};
    float invTotalSamples_ {0.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                  GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y,
            tiling_data.totalSamples, tiling_data.featureDim,
            tiling_data.tileSamples, tiling_data.margin,
            tiling_data.invTotalSamples);
    op.Process();
}
