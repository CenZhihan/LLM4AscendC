#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::Shape &inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(inShape.GetShapeSize());
    uint32_t ndim = static_cast<uint32_t>(inShape.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += static_cast<int32_t>(ndim);
    uint32_t reductionDim = static_cast<uint32_t>(dim);

    uint32_t outerSize = 1;
    for (uint32_t i = 0; i < reductionDim; ++i) {
        outerSize *= static_cast<uint32_t>(inShape.GetDim(i));
    }
    uint32_t reductionSize = static_cast<uint32_t>(inShape.GetDim(reductionDim));
    uint32_t innerSize = 1;
    for (uint32_t i = reductionDim + 1; i < ndim; ++i) {
        innerSize *= static_cast<uint32_t>(inShape.GetDim(i));
    }

    uint32_t outputLength = outerSize * innerSize;

    uint32_t tileOuter = 4096;
    if (tileOuter > outputLength) tileOuter = outputLength;

    uint32_t blockDim = CeilDiv(outputLength, tileOuter);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerSize(outerSize);
    tiling.set_reductionSize(reductionSize);
    tiling.set_innerSize(innerSize);
    tiling.set_blockDim(blockDim);
    tiling.set_tileOuter(tileOuter);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t ndim = x_shape->GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += static_cast<int32_t>(ndim);
    uint32_t reductionDim = static_cast<uint32_t>(dim);

    gert::Shape *y_shape = context->GetOutputShape(0);
    uint32_t outDims = ndim - 1;
    std::vector<int64_t> outDimsVec(outDims);
    for (uint32_t i = 0; i < reductionDim; ++i) {
        outDimsVec[i] = x_shape->GetDim(i);
    }
    for (uint32_t i = reductionDim + 1; i < ndim; ++i) {
        outDimsVec[i-1] = x_shape->GetDim(i);
    }
    y_shape->SetDimNum(outDims);
    for (uint32_t i = 0; i < outDims; ++i) {
        y_shape->SetDim(i, outDimsVec[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelArgmax {
public:
    __aicore__ inline KernelArgmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerSize,
                                uint32_t reductionSize,
                                uint32_t innerSize,
                                uint32_t tileOuter)
    {
        outerSize_ = outerSize;
        reductionSize_ = reductionSize;
        innerSize_ = innerSize;
        tileOuter_ = tileOuter;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y);

        pipe_.InitBuffer(localFloatBuf_, 1, sizeof(float));
        pipe_.InitBuffer(localInt64Buf_, 1, sizeof(int64_t));
    }

    __aicore__ inline void Process()
    {
        if (outerSize_ == 0 || reductionSize_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        uint32_t outputElems = outerSize_ * innerSize_;
        for (uint32_t base = blockIdx * tileOuter_; base < outputElems; base += blockNum * tileOuter_) {
            uint32_t curTile = tileOuter_;
            if (base + curTile > outputElems) curTile = outputElems - base;
            ComputeTile(base, curTile);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t baseIdx, uint32_t count)
    {
        LocalTensor<float> localVal = localFloatBuf_.Get<float>();
        LocalTensor<int64_t> localIdx = localInt64Buf_.Get<int64_t>();

        for (uint32_t i = 0; i < count; ++i) {
            uint32_t outputIdx = baseIdx + i;
            uint32_t innerIdx = outputIdx % innerSize_;
            uint32_t outerIdx = outputIdx / innerSize_;

            uint32_t inputBase = outerIdx * reductionSize_ * innerSize_ + innerIdx;
            uint32_t inputStride = innerSize_;

            // copy first element
            DataCopy(localVal, xGm_[inputBase], 1);
            float maxVal = localVal.GetValue(0);
            int64_t maxIdx = 0;

            for (uint32_t r = 1; r < reductionSize_; ++r) {
                DataCopy(localVal, xGm_[inputBase + r * inputStride], 1);
                float curVal = localVal.GetValue(0);
                if (curVal > maxVal) {
                    maxVal = curVal;
                    maxIdx = static_cast<int64_t>(r);
                }
            }

            // write output
            localIdx.SetValue(0, maxIdx);
            DataCopy(yGm_[outputIdx], localIdx, 1);
        }
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<int64_t> yGm_;

    uint32_t outerSize_;
    uint32_t reductionSize_;
    uint32_t innerSize_;
    uint32_t tileOuter_;

    TPipe pipe_;
    TBuf<TPosition::VECIN> localFloatBuf_;
    TBuf<TPosition::VECIN> localInt64Buf_;
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmax op;
    op.Init(x, y,
            tiling_data.outerSize,
            tiling_data.reductionSize,
            tiling_data.innerSize,
            tiling_data.tileOuter);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor argmax_impl_npu(const at::Tensor& self, int64_t dim) {
    auto ndim = self.dim();
    if (dim < 0) dim += ndim;
    std::vector<int64_t> outShape;
    for (int64_t i = 0; i < ndim; ++i) {
        if (i != dim) outShape.push_back(self.size(i));
    }
    at::Tensor result = at::empty(outShape, self.options().dtype(at::kLong));
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_impl_npu, "Argmax over a dimension (ArgmaxOverADimensionCustom on NPU)");
}
