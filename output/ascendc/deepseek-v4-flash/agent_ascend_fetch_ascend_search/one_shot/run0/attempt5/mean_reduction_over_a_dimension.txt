#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cstdint>
#include <algorithm>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;      // 32‑byte alignment
static constexpr uint32_t MAX_NDIM = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;
    const gert::StorageShape *inShape = context->GetInputShape(0);
    const gert::Shape &originShape = inShape->GetOriginShape();
    uint32_t ndim = static_cast<uint32_t>(originShape.GetDimNum());
    if (ndim > MAX_NDIM) ndim = MAX_NDIM;

    uint32_t dims[MAX_NDIM];
    for (uint32_t i = 0; i < ndim; ++i) {
        dims[i] = static_cast<uint32_t>(originShape.GetDim(i));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;
    if (dim < 0) dim += ndim;

    uint32_t preSize = 1;
    for (uint32_t i = 0; i < static_cast<uint32_t>(dim); ++i) preSize *= dims[i];
    uint32_t reduceSize = dims[dim];
    uint32_t postSize = 1;
    for (uint32_t i = static_cast<uint32_t>(dim) + 1; i < ndim; ++i) postSize *= dims[i];

    uint32_t totalInputLen = preSize * reduceSize * postSize;
    uint32_t totalOutputLen = preSize * postSize;

    // UB budget: we use 4 * tileOutput floats at peak (inQueueX, outQueueY, tmpQueue, sumQueue)
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileOutput = std::min(maxTile, postSize);
    tileOutput = AlignDown(tileOutput, ALIGN_ELEMS);
    if (tileOutput == 0) tileOutput = ALIGN_ELEMS;

    uint32_t numTiles = CeilDiv(totalOutputLen, tileOutput);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalInputLen(totalInputLen);
    tiling.set_totalOutputLen(totalOutputLen);
    tiling.set_reduceSize(reduceSize);
    tiling.set_preSize(preSize);
    tiling.set_postSize(postSize);
    tiling.set_tileOutput(tileOutput);
    tiling.set_blockDim(blockDim);

    float invReduceSize = (reduceSize == 0) ? 0.0f : 1.0f / static_cast<float>(reduceSize);
    tiling.set_invReduceSize(invReduceSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputLen, uint32_t totalOutputLen,
                                uint32_t reduceSize, uint32_t preSize, uint32_t postSize,
                                uint32_t tileOutput, float invReduceSize)
    {
        totalInputLen_ = totalInputLen;
        totalOutputLen_ = totalOutputLen;
        reduceSize_ = reduceSize;
        preSize_ = preSize;
        postSize_ = postSize;
        tileOutput_ = tileOutput;
        invReduceSize_ = invReduceSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileOutput_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOutput_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileOutput_ * sizeof(float));
        pipe_.InitBuffer(sumQueue_, BUFFER_NUM, tileOutput_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputLen_ == 0 || tileOutput_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid‑stride over flat output indices
        const uint32_t stride = blockNum * tileOutput_;
        uint32_t flatStart = blockIdx * tileOutput_;
        while (flatStart < totalOutputLen_) {
            uint32_t len = totalOutputLen_ - flatStart;
            if (len > tileOutput_) len = tileOutput_;
            // Ensure len is aligned to 8 elements (32 bytes)
            len = (len / 8) * 8;
            if (len == 0) break;

            // Compute p (pre index) and q (post index) from flat index
            uint32_t p = flatStart / postSize_;
            uint32_t q = flatStart % postSize_;

            // Accumulate reduction over reduceSize_ slices
            AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            // First slice: copy to sumLocal
            uint32_t firstOffset = p * (reduceSize_ * postSize_) + 0 * postSize_ + q;
            AscendC::DataCopy(sumLocal, xGm_[firstOffset], len);

            // Remaining slices
            for (uint32_t r = 1; r < reduceSize_; ++r) {
                uint32_t inputOffset = p * (reduceSize_ * postSize_) + r * postSize_ + q;
                AscendC::DataCopy(xLocal, xGm_[inputOffset], len);
                AscendC::Add(sumLocal, sumLocal, xLocal, len);
            }

            // Divide by reduceSize
            AscendC::Muls(tLocal, sumLocal, invReduceSize_, len);

            // Move to outQueueY and copy to global memory
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, tLocal, len);
            uint32_t outputOffset = p * postSize_ + q;
            AscendC::DataCopy(yGm_[outputOffset], yLocal, len);

            // Free tensors
            outQueueY_.FreeTensor(yLocal);
            tmpQueue_.FreeTensor(tLocal);
            inQueueX_.FreeTensor(xLocal);
            sumQueue_.FreeTensor(sumLocal);

            flatStart += stride;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> sumQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputLen_ {0};
    uint32_t totalOutputLen_ {0};
    uint32_t reduceSize_ {0};
    uint32_t preSize_ {0};
    uint32_t postSize_ {0};
    uint32_t tileOutput_ {0};
    float invReduceSize_ {0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalInputLen, tiling_data.totalOutputLen,
            tiling_data.reduceSize, tiling_data.preSize, tiling_data.postSize,
            tiling_data.tileOutput, tiling_data.invReduceSize);
    op.Process();
}

    if (dim < 0) dim += ndim;
