    uint32_t B = static_cast<uint32_t>(x_shape.GetDim(0));

    uint32_t blockDim = B < 24 ? B : 24;

    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_blockStartBatch(0);  // unused – kernel computes
    tiling.set_blockNbatches(batchPerCore + (rem > 0 ? 1 : 0)); // also unused

kernel_src="""

#include "kernel_operator.h"
#include <type_traits>

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHidden {
public:
    __aicore__ inline KernelVanillaRnnHidden() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h,
                                GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o,
                                GM_ADDR h_new, GM_ADDR y,
                                uint32_t B, uint32_t I, uint32_t H, uint32_t O, uint32_t K)
    {
        B_ = B; I_ = I; H_ = H; O_ = O; K_ = K;

        xGm_.SetGlobalBuffer((__gm__ float*)x, B_ * I_);
        hGm_.SetGlobalBuffer((__gm__ float*)h, B_ * H_);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H_ * K_);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O_ * H_);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);
        hNewGm_.SetGlobalBuffer((__gm__ float*)h_new, B_ * H_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, B_ * O_);

        // Tiling constants
        tileMatRows_ = 64;
        tileMatCols_ = 512;

        // Allocate local buffers (UB)
        pipe_.InitBuffer(xLocal_, 1, I_ * sizeof(float));
        pipe_.InitBuffer(hLocal_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(inpLocal_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(w_i2hTile_, 1, tileMatRows_ * K_ * sizeof(float)); // large but within UB if K <= 512? We'll adjust
        // Actually K=1280, tileMatCols=512, so we tile over columns as well.
        // For simplicity, we load entire w_i2h per row? But UB limited.
        // We'll use a simpler approach: for each batch, we iterate over H rows,
        // and for each row compute dot product via scalar loop.
        // This avoids large local buffers for matrix.
    }

    __aicore__ inline void Process()
    {
        if (B_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        // Distribute batches across cores
        uint32_t batchPerCore = B_ / blockNum;
        uint32_t rem = B_ % blockNum;
        uint32_t startB = blockIdx * batchPerCore + (blockIdx < rem ? blockIdx : rem);
        uint32_t endB = startB + batchPerCore + (blockIdx < rem ? 1 : 0);
        if (startB >= B_) return;

        // Temporary local variables for one batch
        LocalTensor<float> xLocal = xLocal_.Get<float>();
        LocalTensor<float> hLocal = hLocal_.Get<float>();
        LocalTensor<float> inpLocal = inpLocal_.Get<float>();
        LocalTensor<float> preLocal = pipe_.AllocTensor<float>(H_);
        LocalTensor<float> hNewLocal = pipe_.AllocTensor<float>(H_);
        LocalTensor<float> yLocal = pipe_.AllocTensor<float>(O_);
        LocalTensor<float> wRowLocal = pipe_.AllocTensor<float>(K_); // one row of w_i2h
        LocalTensor<float> wRowLocal_h2o = pipe_.AllocTensor<float>(H_); // one row of w_h2o

        for (uint32_t b = startB; b < endB; ++b) {
            // Load x[b] (I) and h[b] (H)
            DataCopy(xLocal, xGm_[b * I_], I_);
            DataCopy(hLocal, hGm_[b * H_], H_);

            // Build inp = cat(x, h)
            // Copy x part
            for (uint32_t i = 0; i < I_; ++i) inpLocal(i) = xLocal(i);
            // Copy h part
            for (uint32_t i = 0; i < H_; ++i) inpLocal(I_ + i) = hLocal(i);

            // Compute pre = w_i2h * inp + b_i2h, with tiling over rows (H)
            // Initialize pre with bias
            DataCopy(preLocal, b_i2hGm_, H_);

            // Process rows of w_i2h in tiles of tileMatRows_
            for (uint32_t rowStart = 0; rowStart < H_; rowStart += tileMatRows_) {
                uint32_t numRows = (rowStart + tileMatRows_ <= H_) ? tileMatRows_ : (H_ - rowStart);
                // For each row in this tile
                for (uint32_t r = 0; r < numRows; ++r) {
                    uint32_t rowIdx = rowStart + r;
                    // Load one row of w_i2h (K elements)
                    DataCopy(wRowLocal, w_i2hGm_[rowIdx * K_], K_);
                    // Dot product: sum over K (tiled over columns)
                    float dot = 0.0f;
                    for (uint32_t colStart = 0; colStart < K_; colStart += tileMatCols_) {
                        uint32_t numCols = (colStart + tileMatCols_ <= K_) ? tileMatCols_ : (K_ - colStart);
                        // Multiply elementwise
                        LocalTensor<float> mulTmp = pipe_.AllocTensor<float>(numCols);
                        Mul(mulTmp, inpLocal[colStart], wRowLocal[colStart], numCols);
                        // Reduce add
                        float partSum = 0.0f;
                        for (uint32_t i = 0; i < numCols; ++i) partSum += mulTmp(i);
                        dot += partSum;
                        pipe_.FreeTensor(mulTmp);
                    }
                    preLocal(rowIdx) += dot;
                }
            }

            // h_new = tanh(pre)
            Tanh(hNewLocal, preLocal, H_);

            // Compute y = w_h2o * h_new + b_h2o, tiled similarly
            DataCopy(yLocal, b_h2oGm_, O_);
            for (uint32_t rowStart = 0; rowStart < O_; rowStart += tileMatRows_) {
                uint32_t numRows = (rowStart + tileMatRows_ <= O_) ? tileMatRows_ : (O_ - rowStart);
                for (uint32_t r = 0; r < numRows; ++r) {
                    uint32_t rowIdx = rowStart + r;
                    // Load one row of w_h2o (H elements)
                    DataCopy(wRowLocal_h2o, w_h2oGm_[rowIdx * H_], H_);
                    float dot = 0.0f;
                    for (uint32_t colStart = 0; colStart < H_; colStart += tileMatCols_) {
                        uint32_t numCols = (colStart + tileMatCols_ <= H_) ? tileMatCols_ : (H_ - colStart);
                        LocalTensor<float> mulTmp = pipe_.AllocTensor<float>(numCols);
                        Mul(mulTmp, hNewLocal[colStart], wRowLocal_h2o[colStart], numCols);
                        float partSum = 0.0f;
                        for (uint32_t i = 0; i < numCols; ++i) partSum += mulTmp(i);
                        dot += partSum;
                        pipe_.FreeTensor(mulTmp);
                    }
                    yLocal(rowIdx) += dot;
                }
            }

            // Write outputs
            DataCopy(hNewGm_[b * H_], hNewLocal, H_);
            DataCopy(yGm_[b * O_], yLocal, O_);
        }

        pipe_.FreeTensor(preLocal);
        pipe_.FreeTensor(hNewLocal);
        pipe_.FreeTensor(yLocal);
        pipe_.FreeTensor(wRowLocal);
        pipe_.FreeTensor(wRowLocal_h2o);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> xLocal_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> hLocal_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> inpLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> hGm_;
    AscendC::GlobalTensor<float> w_i2hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> hNewGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t B_, I_, H_, O_, K_;
    uint32_t tileMatRows_, tileMatCols_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h,
    GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o,
    GM_ADDR h_new, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHidden op;
    op.Init(x, h, w_i2h, b_i2h, w_h2o, b_h2o, h_new, y,
            tiling_data.B, tiling_data.I, tiling_data.H, tiling_data.O, tiling_data.K);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'vanilla_rnn_hidden_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, x, h0, w_i2h, b_i2h, w_h2o, b_h2o):
        T, B, I = x.shape
        H = w_i2h.shape[0]
        O = w_h2o.shape[0]
        h = h0
        outs = []
        for t in range(T):
            x_t = x[t]  # (B, I)
            fn = getattr(custom_ops_lib, _FN)
            h_new, y_t = fn(x_t, h, w_i2h, b_i2h, w_h2o, b_h2o)
            outs.append(y_t)
            h = h_new
        return torch.stack(outs, dim=0)
'''
