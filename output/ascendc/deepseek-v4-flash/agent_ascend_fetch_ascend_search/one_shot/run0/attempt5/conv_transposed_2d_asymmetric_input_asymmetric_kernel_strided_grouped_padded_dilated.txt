static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustomTilingData tiling;

    const auto* x_shape = context->GetInputShape(0);
    const auto* w_shape = context->GetInputShape(1);

    int32_t N = x_shape->GetDim(0);
    int32_t C_in = x_shape->GetDim(1);
    int32_t H_in = x_shape->GetDim(2);
    int32_t W_in = x_shape->GetDim(3);
    int32_t kH = w_shape->GetDim(2);
    int32_t kW = w_shape->GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *stride_ptr = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *padding_ptr = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *output_padding_ptr = attrs->GetAttrPointer<int64_t>(2);
    int64_t groups = *attrs->GetAttrPointer<int64_t>(3);
    const int64_t *dilation_ptr = attrs->GetAttrPointer<int64_t>(4);

    int32_t strideH = static_cast<int32_t>(stride_ptr[0]);
    int32_t strideW = static_cast<int32_t>(stride_ptr[1]);
    int32_t padH = static_cast<int32_t>(padding_ptr[0]);
    int32_t padW = static_cast<int32_t>(padding_ptr[1]);
    int32_t outPadH = static_cast<int32_t>(output_padding_ptr[0]);
    int32_t outPadW = static_cast<int32_t>(output_padding_ptr[1]);
    int32_t dilationH = static_cast<int32_t>(dilation_ptr[0]);
    int32_t dilationW = static_cast<int32_t>(dilation_ptr[1]);

    int32_t C_out = static_cast<int32_t>(w_shape->GetDim(1)) * static_cast<int32_t>(groups);

    int32_t H_out = (H_in - 1) * strideH - 2 * padH + dilationH * (kH - 1) + outPadH + 1;
    int32_t W_out = (W_in - 1) * strideW - 2 * padW + dilationW * (kW - 1) + outPadW + 1;

    uint32_t totalLength = static_cast<uint32_t>(N) * static_cast<uint32_t>(C_out) *
                           static_cast<uint32_t>(H_out) * static_cast<uint32_t>(W_out);

    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C_in(C_in);
    tiling.set_H_in(H_in);
    tiling.set_W_in(W_in);
    tiling.set_C_out(C_out);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_dilationH(dilationH);
    tiling.set_dilationW(dilationW);
    tiling.set_groups(static_cast<int32_t>(groups));
    tiling.set_outPadH(outPadH);
    tiling.set_outPadW(outPadW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
