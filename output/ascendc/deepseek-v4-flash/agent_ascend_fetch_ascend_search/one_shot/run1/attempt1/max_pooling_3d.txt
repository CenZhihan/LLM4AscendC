host_operator_src="""

#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget for this kernel: 
//   input tile: D*H*W floats + output tile: dOut*hOut*wOut floats + temp for max
// We use 2 input tiles + 2 output tiles + intermediate for pipeline
// Simplified: each slice limited to UB size.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static int32_t ComputeOutSize(int32_t inSize, int32_t pad, int32_t dilation, 
                              int32_t kernel, int32_t stride, bool ceilMode) {
    int32_t numerator = inSize + 2 * pad - dilation * (kernel - 1) - 1;
    int32_t outSize;
    if (ceilMode) {
        outSize = CeilDiv(numerator, stride) + 1;
    } else {
        outSize = numerator / stride + 1;
    }
    return outSize;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaxPooling3dCustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = xShape->GetShapeSize();
    
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrInt(0);
    int32_t stride = attrs->GetAttrInt(1);
    int32_t padding = attrs->GetAttrInt(2);
    int32_t dilation = attrs->GetAttrInt(3);
    bool ceilMode = attrs->GetAttrBool(4);
    
    if (stride <= 0) stride = kernelSize;
    if (kernelSize <= 0) kernelSize = 1;
    
    uint32_t n = xShape->GetDim(0);
    uint32_t c = xShape->GetDim(1);
    uint32_t d = xShape->GetDim(2);
    uint32_t h = xShape->GetDim(3);
    uint32_t w = xShape->GetDim(4);
    
    int32_t dOut = ComputeOutSize(d, padding, dilation, kernelSize, stride, ceilMode);
    int32_t hOut = ComputeOutSize(h, padding, dilation, kernelSize, stride, ceilMode);
    int32_t wOut = ComputeOutSize(w, padding, dilation, kernelSize, stride, ceilMode);
    
    // Block dim: one block per element in NC plane (grid-stride over NC)
    uint32_t totalElements = n * c * dOut * hOut * wOut;
    uint32_t blockDim = (totalElements < 1) ? 1 : totalElements;
    if (blockDim > 32) blockDim = 32; // conservative
    
    // Tile length: number of output spatial elements per inner tile along D 
    // We tile along depth (D) to maximize UB usage.
    uint32_t spatialOut = dOut * hOut * wOut;
    uint32_t maxTileSpatial = UB_BUDGET_BYTES / (sizeof(float) * (1 + 1)); // input tile + output tile approx
    maxTileSpatial = (maxTileSpatial / ALIGN_ELEMS) * ALIGN_ELEMS;
    if (maxTileSpatial > 4096) maxTileSpatial = 4096;
    if (spatialOut < maxTileSpatial) maxTileSpatial = spatialOut;
    
    context->SetBlockDim(blockDim);
    
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(maxTileSpatial);
    tiling.set_blockDim(blockDim);
    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_d(d);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_dOut(dOut);
    tiling.set_hOut(hOut);
    tiling.set_wOut(wOut);
    tiling.set_kernelD(kernelSize);
    tiling.set_kernelH(kernelSize);
    tiling.set_kernelW(kernelSize);
    tiling.set_strideD(stride);
    tiling.set_strideH(stride);
    tiling.set_strideW(stride);
    tiling.set_padD(padding);
    tiling.set_padH(padding);
    tiling.set_padW(padding);
    tiling.set_dilationD(dilation);
    tiling.set_dilationH(dilation);
    tiling.set_dilationW(dilation);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    
    // We need attrs to compute output dims; we retrieve from context
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrInt(0);
    int32_t stride = attrs->GetAttrInt(1);
    int32_t padding = attrs->GetAttrInt(2);
    int32_t dilation = attrs->GetAttrInt(3);
    bool ceilMode = attrs->GetAttrBool(4);
    if (stride <= 0) stride = kernelSize;
    
    int32_t d = x_shape->GetDim(2);
    int32_t h = x_shape->GetDim(3);
    int32_t w = x_shape->GetDim(4);
    int32_t dOut = ComputeOutSize(d, padding, dilation, kernelSize, stride, ceilMode);
    int32_t hOut = ComputeOutSize(h, padding, dilation, kernelSize, stride, ceilMode);
    int32_t wOut = ComputeOutSize(w, padding, dilation, kernelSize, stride, ceilMode);
    
    y_shape->SetDimNum(5);
    y_shape->SetDim(0, x_shape->GetDim(0));
    y_shape->SetDim(1, x_shape->GetDim(1));
    y_shape->SetDim(2, dOut);
    y_shape->SetDim(3, hOut);
    y_shape->SetDim(4, wOut);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

static int32_t ComputeOutSize(int32_t inSize, int32_t pad, int32_t dilation,
                              int32_t kernel, int32_t stride, bool ceilMode) {
    int32_t numerator = inSize + 2 * pad - dilation * (kernel - 1) - 1;
    int32_t outSize;
    if (ceilMode) {
        outSize = (numerator + stride - 1) / stride + 1;
    } else {
        outSize = numerator / stride + 1;
    }
    return outSize;
}

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;   // double buffer for pipeline

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t dOut, uint32_t hOut, uint32_t wOut,
                                int32_t kernelD, int32_t kernelH, int32_t kernelW,
                                int32_t strideD, int32_t strideH, int32_t strideW,
                                int32_t padD, int32_t padH, int32_t padW,
                                int32_t dilationD, int32_t dilationH, int32_t dilationW) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n * c * dOut * hOut * wOut);
        
        n_ = n; c_ = c; d_ = d; h_ = h; w_ = w;
        dOut_ = dOut; hOut_ = hOut; wOut_ = wOut;
        kernelD_ = kernelD; kernelH_ = kernelH; kernelW_ = kernelW;
        strideD_ = strideD; strideH_ = strideH; strideW_ = strideW;
        padD_ = padD; padH_ = padH; padW_ = padW;
        dilationD_ = dilationD; dilationH_ = dilationH; dilationW_ = dilationW;
        tileLen_ = tileLength; // number of output spatial elements per tile
        totalLength_ = totalLength;
        
        // Determine the max input spatial size needed per tile (worst case depth x height x width)
        int32_t inputSpatial = (dOut_ * strideD_ + (kernelD_ - 1) * dilationD_ - 2 * padD_ + 1) *
                               (hOut_ * strideH_ + (kernelH_ - 1) * dilationH_ - 2 * padH_ + 1) *
                               (wOut_ * strideW_ + (kernelW_ - 1) * dilationW_ - 2 * padW_ + 1);
        if (inputSpatial < (int32_t)(d_ * h_ * w_)) inputSpatial = d_ * h_ * w_;
        
        uint32_t inputTileElems = tileLen_ * kernelD_ * kernelH_ * kernelW_; // approximate
        if (inputTileElems > (uint32_t)inputSpatial) inputTileElems = inputSpatial;
        
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputTileElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileLen_ * sizeof(float)); // temp for max result
    }

    __aicore__ inline void Process() {
        // Grid-stride over output spatial elements across blocks
        const uint32_t blockNum = GetBlockNum();
        const uint32_t blockIdx = GetBlockIdx();
        uint32_t totalOutputElems = n_ * c_ * dOut_ * hOut_ * wOut_;
        uint32_t strideElems = blockNum;
        uint32_t base = blockIdx;
        if (base >= totalOutputElems) return;
        
        // Prefetch first tile
        ProcessTile(base);
        base += strideElems;
        
        while (base < totalOutputElems) {
            ProcessTile(base);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t outLinearIdx) {
        // Compute output indices (n,c,od,oh,ow) from linear index
        uint32_t od = outLinearIdx % dOut_;
        uint32_t tmp = outLinearIdx / dOut_;
        uint32_t oh = tmp % hOut_;
        tmp = tmp / hOut_;
        uint32_t ow = tmp % wOut_;
        tmp = tmp / wOut_;
        uint32_t oc = tmp % c_;
        uint32_t n = tmp / c_;
        
        // Compute input start positions
        int32_t dStart = od * strideD_ - padD_;
        int32_t hStart = oh * strideH_ - padH_;
        int32_t wStart = ow * strideW_ - padW_;
        
        // Load input window
        float maxVal = -FLT_MAX;
        for (int kd = 0; kd < kernelD_; kd++) {
            int32_t id = dStart + kd * dilationD_;
            if (id < 0 || id >= (int32_t)d_) continue;
            for (int kh = 0; kh < kernelH_; kh++) {
                int32_t ih = hStart + kh * dilationH_;
                if (ih < 0 || ih >= (int32_t)h_) continue;
                for (int kw = 0; kw < kernelW_; kw++) {
                    int32_t iw = wStart + kw * dilationW_;
                    if (iw < 0 || iw >= (int32_t)w_) continue;
                    uint32_t inIdx = ((n * c_ + oc) * d_ + id) * h_ * w_ + ih * w_ + iw;
                    float val = xGm_.GetValue(inIdx);
                    if (val > maxVal) maxVal = val;
                }
            }
        }
        
        uint32_t outIdx = outLinearIdx;
        yGm_.SetValue(outIdx, maxVal);
    }
    
private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpBuf_;
    
    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;
    
    uint32_t n_, c_, d_, h_, w_;
    uint32_t dOut_, hOut_, wOut_;
    int32_t kernelD_, kernelH_, kernelW_;
    int32_t strideD_, strideH_, strideW_;
    int32_t padD_, padH_, padW_;
    int32_t dilationD_, dilationH_, dilationW_;
    uint32_t tileLen_;
    uint32_t totalLength_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.dOut, tiling_data.hOut, tiling_data.wOut,
            tiling_data.kernelD, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW,
            tiling_data.dilationD, tiling_data.dilationH, tiling_data.dilationW);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0,
                 dilation: int = 1, ceil_mode: bool = False):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride if stride is not None else kernel_size
        self.padding = padding
        self.dilation = dilation
        self.ceil_mode = ceil_mode

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding, self.dilation, self.ceil_mode)
