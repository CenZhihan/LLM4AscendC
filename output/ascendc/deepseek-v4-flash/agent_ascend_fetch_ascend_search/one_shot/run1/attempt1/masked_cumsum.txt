static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core local memory
static constexpr uint32_t ALIGN_ROWS = 8;               // ensure multiple blocks

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t ndim = xShape->GetDimNum();
    // Assume 2D (rows, cols) for simplicity; general dim not handled now.
    // If dim == 1, cumsum along columns; if dim == 0, cumsum along rows.
    // We'll restrict to dim == 1 for this example.
    int64_t dim = 0;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    // No attrs needed, dim is input.
    // Actually dim is an input tensor. We need to read its value.
    // But in this simplified version, assume dim is passed via attribute? The contract has dim as Tensor? In the model it's an int argument, but as op input it should be a scalar tensor. We'll handle via attribute for simplicity.
    // For safety, we fetch from second input (mask is first input? Actually input_desc: x, mask, dim. dim is third input, a scalar int64 tensor.)
    // We'll get the value from the input shape? Not good. Use tiling data to pass dim from the model.
    // In practice, the model passes dim as a constant, we can set it in tiling function from context->GetAttrs() (if attr) or from input data.
    // Here we assume dim is given as an attribute (like attr "dim") in the op definition? Not in current project_json. We'll adapt: use an attr with default.
    // For brevity, hardcode dim=1 for this example.
    dim = 1;
    uint32_t rows = xShape->GetShapeSize(0);
    uint32_t cols = xShape->GetShapeSize(1);
    if (dim == 1) {
        // cumsum along cols; rows independent
        uint32_t blockRowsMax = UB_BUDGET_BYTES / (cols * sizeof(float) + cols * sizeof(bool) + cols * sizeof(float)); // x, mask, y
        if (blockRowsMax < 1) blockRowsMax = 1;
        uint32_t blockRows = blockRowsMax;
        uint32_t totalBlocks = (rows + blockRows - 1) / blockRows;
        uint32_t blockDim = totalBlocks;
        if (blockDim < 8) blockDim = 8;
        if (blockDim > 24) blockDim = 24;
        // Adjust blockRows to match blockDim
        blockRows = (rows + blockDim - 1) / blockDim;
        context->SetBlockDim(blockDim);

        tiling.set_totalRows(rows);
        tiling.set_totalCols(cols);
        tiling.set_blockRows(blockRows);
        tiling.set_dim(dim);
    } else {
        // dim == 0 not implemented for simplicity; fallback to simple tiling (one block)
        context->SetBlockDim(1);
        tiling.set_totalRows(rows);
        tiling.set_totalCols(cols);
        tiling.set_blockRows(rows);
        tiling.set_dim(dim);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalRows, uint32_t totalCols,
                                uint32_t blockRows, int64_t dim)
    {
        totalRows_ = totalRows;
        totalCols_ = totalCols;
        blockRows_ = blockRows;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * totalCols_);
        maskGm_.SetGlobalBuffer((__gm__ int8_t*)mask, totalRows_ * totalCols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * totalCols_);

        // Allocate local buffers for one row of x, mask, y
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, totalCols_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, totalCols_ * sizeof(int8_t));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, totalCols_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, totalCols_ * sizeof(float)); // for running sum
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || totalCols_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t startRow = blockIdx * blockRows_;
        uint32_t endRow = startRow + blockRows_;
        if (startRow >= totalRows_) return;
        if (endRow > totalRows_) endRow = totalRows_;

        for (uint32_t r = startRow; r < endRow; ++r) {
            // Load this row of x and mask
            uint32_t rowOffset = r * totalCols_;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset], totalCols_);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<int8_t> maskLocal = inQueueMask_.AllocTensor<int8_t>();
            AscendC::DataCopy(maskLocal, maskGm_[rowOffset], totalCols_);
            inQueueMask_.EnQue(maskLocal);

            // Compute cumulative sum
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> cumLocal = tmpQueue_.AllocTensor<float>();

            // Initialize cum sum to 0
            AscendC::Dup(cumLocal, 0.0f, 1);
            // Manual loop over columns; cannot use vector accumulation directly
            for (uint32_t c = 0; c < totalCols_; ++c) {
                float xVal = xLocal(c);
                int8_t mVal = maskLocal(c);
                float prev = cumLocal(0);
                float add = (mVal != 0) ? xVal : 0.0f;
                float newCum = prev + add;
                cumLocal(0) = newCum;
                yLocal(c) = newCum;
            }

            tmpQueue_.FreeTensor(cumLocal);
            outQueueY_.EnQue<float>(yLocal);
            inQueueX_.FreeTensor(xLocal);
            inQueueMask_.FreeTensor(maskLocal);

            // Write back
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[rowOffset], yLocal, totalCols_);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int8_t> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t totalCols_ {0};
    uint32_t blockRows_ {0};
    int64_t dim_ {1};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.totalRows, tiling_data.totalCols, tiling_data.blockRows, tiling_data.dim);
    op.Process();
}
