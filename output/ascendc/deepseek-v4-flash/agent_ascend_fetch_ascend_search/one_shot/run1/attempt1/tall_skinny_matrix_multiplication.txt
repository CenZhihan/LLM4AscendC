#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include "matmul_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Get input shapes
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    int64_t M = a_shape->GetDim(0);
    int64_t K = a_shape->GetDim(1);
    int64_t N = b_shape->GetDim(1);

    // Create Matmul tiling object
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling tiling(ascendcPlatform);

    // Set data types and formats
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    // No bias

    // Set shapes
    tiling.SetOrgShape(matmul_tiling::TPosition::GM, matmul_tiling::MatrixDim::M, M);
    tiling.SetOrgShape(matmul_tiling::TPosition::GM, matmul_tiling::MatrixDim::K, K);
    tiling.SetOrgShape(matmul_tiling::TPosition::GM, matmul_tiling::MatrixDim::N, N);

    // Generate tiling data
    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) return ge::GRAPH_FAILED;

    // Save tiling data into opaque buffer
    TallSkinnyMatrixMultiplicationCustomTilingData outputData;
    memcpy(outputData.tilingBuffer, &tilingData, sizeof(tilingData));
    outputData.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(sizeof(tilingData));

    // Set block dimension (cube unit can use many cores)
    // Typically 1-8 cores for cube; we set based on size
    uint32_t blockDim = 1;
    if (M * N > 1024 * 1024) blockDim = 8;
    else if (M * N > 256 * 256) blockDim = 4;
    context->SetBlockDim(blockDim);

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    int64_t M = a_shape->GetDim(0);
    int64_t N = b_shape->GetDim(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

{
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

// Matrix types: A and B are half (fp16) in GM, C is float in GM, no bias
using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half>;
using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half>;
using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;
using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR tiling)
    {
        // Cast tiling pointer to the expected structure
        optiling::TCubeTiling *tilingData = reinterpret_cast<optiling::TCubeTiling*>(tiling);
        mm_.Init(tilingData);

        // Set global tensor buffers
        aGm_.SetGlobalBuffer((__gm__ half*)a);
        bGm_.SetGlobalBuffer((__gm__ half*)b);
        cGm_.SetGlobalBuffer((__gm__ float*)c);
    }

    __aicore__ inline void Process()
    {
        // Issue matmul computation
        mm_.SetTensorA(aGm_);
        mm_.SetTensorB(bGm_);
        mm_.IterateAll(cGm_);
    }

private:
    AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> mm_;
    AscendC::GlobalTensor<half> aGm_;
    AscendC::GlobalTensor<half> bGm_;
    AscendC::GlobalTensor<float> cGm_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(a, b, c, tilingData.tilingBuffer);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Convert inputs to half, output float
    at::Tensor A_half = A.to(at::kHalf);
    at::Tensor B_half = B.to(at::kHalf);
    at::Tensor C = at::empty({A.size(0), B.size(1)}, A.options().dtype(at::kFloat));
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, A_half, B_half, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("tall_skinny_matrix_multiplication_custom", &matmul_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("tall_skinny_matrix_multiplication_custom", &matmul_impl_npu,
          "Tall‑Skinny Matrix Multiplication (float16 input, float32 output)");
