#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -1e38f; // or -FLT_MAX but avoid FLT

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t lastDim,
                                uint32_t tileLength, uint32_t blockDim) {
        totalLen_ = totalLength;
        lastDim_ = lastDim;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        rows_ = totalLen_ / lastDim_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        // No sharedTmp buffer needed; use 4-arg reduce
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || lastDim_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t rowsPerBlock = (rows_ + blockDim_ - 1) / blockDim_;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > rows_) rowEnd = rows_;

        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            ProcessRow(r);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row) {
        uint32_t base = row * lastDim_;
        uint32_t numTiles = (lastDim_ + tileLen_ - 1) / tileLen_;
        float rowMax = NEG_INF;

        // Pass 1: compute max across the row
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = t * tileLen_;
            uint32_t len = (offset + tileLen_ <= lastDim_) ? tileLen_ : (lastDim_ - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], len);
            AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
            uint32_t shape[1] = {len};
            AscendC::ReduceMax(maxLocal, xLocal, shape, false);
            float tileMax = maxLocal(0);
            if (tileMax > rowMax) rowMax = tileMax;
            inQueueX_.FreeTensor(xLocal);
            tmpQueue_.FreeTensor(maxLocal);
        }

        // Pass 2: compute sum of exp(x - rowMax)
        float sumExp = 0.0f;
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = t * tileLen_;
            uint32_t len = (offset + tileLen_ <= lastDim_) ? tileLen_ : (lastDim_ - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], len);
            AscendC::LocalTensor<float> diffLocal = tmpQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> rowMaxTensor = tmpQueue_.AllocTensor<float>();
            AscendC::Dup(rowMaxTensor, rowMax, len);
            AscendC::Sub(diffLocal, xLocal, rowMaxTensor, len);
            AscendC::Exp(diffLocal, diffLocal, len);
            AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();
            uint32_t shape[1] = {len};
            AscendC::ReduceSum(sumLocal, diffLocal, shape, false);
            sumExp += sumLocal(0);
            inQueueX_.FreeTensor(xLocal);
            tmpQueue_.FreeTensor(diffLocal);
            tmpQueue_.FreeTensor(rowMaxTensor);
            tmpQueue_.FreeTensor(sumLocal);
        }

        float logSum = logf(sumExp);  // include <cmath>

        // Pass 3: compute output y = x - (rowMax + logSum)
        float shift = rowMax + logSum;
        for (uint32_t t = 0; t < numTiles; ++t) {
            uint32_t offset = t * tileLen_;
            uint32_t len = (offset + tileLen_ <= lastDim_) ? tileLen_ : (lastDim_ - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], len);
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> shiftTensor = tmpQueue_.AllocTensor<float>();
            AscendC::Dup(shiftTensor, shift, len);
            AscendC::Sub(yLocal, xLocal, shiftTensor, len);
            AscendC::DataCopy(yGm_[base + offset], yLocal, len);
            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            tmpQueue_.FreeTensor(shiftTensor);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t lastDim_{0};
    uint32_t tileLen_{0};
    uint32_t blockDim_{0};
    uint32_t rows_{0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.lastDim, tiling_data.tileLength, tiling_data.blockDim);
    op.Process();
}
