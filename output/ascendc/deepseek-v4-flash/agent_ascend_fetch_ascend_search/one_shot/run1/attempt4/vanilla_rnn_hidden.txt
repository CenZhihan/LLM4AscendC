kernel_src = '''
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHidden {
public:
    __aicore__ inline KernelVanillaRnnHidden() {}

    __aicore__ inline void Init(GM_ADDR x_t, GM_ADDR h,
                                GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o,
                                GM_ADDR y_t, GM_ADDR h_new,
                                GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        B_ = td.B; I_ = td.I; H_ = td.H; O_ = td.O;

        xGm_.SetGlobalBuffer((__gm__ float*)x_t, B_ * I_);
        hGm_.SetGlobalBuffer((__gm__ float*)h, B_ * H_);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H_ * (I_ + H_));
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O_ * H_);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y_t, B_ * O_);
        hNewGm_.SetGlobalBuffer((__gm__ float*)h_new, B_ * H_);

        pipe_.InitBuffer(inBufX_, BUFFER_NUM, 4 * 1024 * sizeof(float));
        pipe_.InitBuffer(inBufH_, BUFFER_NUM, 4 * 1024 * sizeof(float));
        pipe_.InitBuffer(outBufH_, BUFFER_NUM, 4 * 1024 * sizeof(float));
        pipe_.InitBuffer(outBufY_, BUFFER_NUM, 4 * 1024 * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, 4 * 1024 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Single block processes all rows for simplicity.
        // For each batch row, compute first matmul (i2h) + tanh, then second (h2o).
        for (uint32_t b = 0; b < B_; ++b) {
            // Compute h_new[b][*] = tanh( x_t[b] * w_i2h^T + h[b] * w_i2h^T + b_i2h )
            // w_i2h is (H, K), where K = I + H.
            // For each output neuron i (0..H-1), compute dot product.
            for (uint32_t i = 0; i < H_; ++i) {
                float sum = b_i2hGm_.GetValue(i); // bias
                // contribution from x_t[b][j] * w_i2h[i][j]
                for (uint32_t j = 0; j < I_; ++j) {
                    sum += xGm_.GetValue(b * I_ + j) * w_i2hGm_.GetValue(i * (I_ + H_) + j);
                }
                // contribution from h[b][k] * w_i2h[i][I_+k]
                for (uint32_t k = 0; k < H_; ++k) {
                    sum += hGm_.GetValue(b * H_ + k) * w_i2hGm_.GetValue(i * (I_ + H_) + I_ + k);
                }
                // tanh (approximate using exp)
                float tanh_val = (expf(2.0f * sum) - 1.0f) / (expf(2.0f * sum) + 1.0f);
                hNewGm_.SetValue(b * H_ + i, tanh_val);
            }
            // Compute y_t[b][*] = h_new[b] * w_h2o^T + b_h2o
            for (uint32_t i = 0; i < O_; ++i) {
                float sum = b_h2oGm_.GetValue(i);
                for (uint32_t k = 0; k < H_; ++k) {
                    sum += hNewGm_.GetValue(b * H_ + k) * w_h2oGm_.GetValue(i * H_ + k);
                }
                yGm_.SetValue(b * O_ + i, sum);
            }
        }
    }

private:
    uint32_t B_, I_, H_, O_;
    GlobalTensor<float> xGm_, hGm_, w_i2hGm_, b_i2hGm_, w_h2oGm_, b_h2oGm_, yGm_, hNewGm_;
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inBufX_, inBufH_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outBufH_, outBufY_;
    TQue<TPosition::VECCALC, 1> tmpBuf_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(
    GM_ADDR x_t, GM_ADDR h, GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o,
    GM_ADDR y_t, GM_ADDR h_new,
    GM_ADDR workspace, GM_ADDR tiling)
{
    KernelVanillaRnnHidden op;
    op.Init(x_t, h, w_i2h, b_i2h, w_h2o, b_h2o, y_t, h_new, workspace, tiling);
    op.Process();
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'vanilla_rnn_hidden_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super().__init__()

    def forward(self, x, h0, w_i2h, b_i2h, w_h2o, b_h2o):
        T, B, I = x.shape
        H = h0.shape[1]
        O = w_h2o.shape[0]
        h = h0
        outs = []
        for t in range(T):
            x_t = x[t]
            y_t, h_new = getattr(custom_ops_lib, _FN)(x_t, h, w_i2h, b_i2h, w_h2o, b_h2o)
            h = h_new
            outs.append(y_t)
        return torch.stack(outs, dim=0)
'''
