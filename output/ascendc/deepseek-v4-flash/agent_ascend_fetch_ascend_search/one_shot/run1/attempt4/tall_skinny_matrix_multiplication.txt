namespace optiling {

static constexpr uint32_t TILE_M = 256;  // rows per tile
static constexpr uint32_t ALIGN = 32;    // for vector alignment
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    uint32_t M = static_cast<uint32_t>(a_shape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(a_shape->GetDim(1));
    uint32_t N = static_cast<uint32_t>(b_shape->GetDim(1));

    // Adjust tileM to be aligned
    uint32_t tileM = TILE_M;
    if (M < tileM) tileM = M;
    tileM = ((tileM + ALIGN - 1) / ALIGN) * ALIGN;
    if (tileM > M) tileM = M;

    uint32_t blockDim = CeilDiv(M, tileM);
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

// Kernel for tall-skinny matrix multiplication: C = A * B
// A: float16 [M, K], B: float16 [K, N], C: float32 [M, N]
// Assumes N is small (e.g., up to 32) so we can vectorize across K.
// Each core processes a contiguous chunk of rows.

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM)
    {
        M_ = M;
        K_ = K;
        N_ = N;
        tileM_ = tileM;
        aGm_.SetGlobalBuffer((__gm__ half*)a);
        bGm_.SetGlobalBuffer((__gm__ half*)b);
        cGm_.SetGlobalBuffer((__gm__ float*)c);
        // Allocate local buffers: row chunk of A (tileM x K) is too large,
        // we process one row at a time to reduce UB usage.
        pipe_.InitBuffer(rowBuf_, 1, K_ * sizeof(half));          // one row of A
        pipe_.InitBuffer(colBuf_, 1, K_ * sizeof(half));          // one column of B
        pipe_.InitBuffer(accBuf_, 1, N_ * sizeof(float));         // accumulator for N columns
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockDim = AscendC::GetBlockNum();
        uint32_t startRow = blockIdx * tileM_;
        if (startRow >= M_) return;
        uint32_t endRow = startRow + tileM_;
        if (endRow > M_) endRow = M_;

        // For each row in this block's range
        for (uint32_t row = startRow; row < endRow; ++row) {
            // Load the entire row of A from GM into rowBuf
            AscendC::LocalTensor<half> rowLocal = rowBuf_.Get<half>();
            AscendC::DataCopy(rowLocal, aGm_[row * K_], K_);

            // Initialize accumulator to zero for this row
            AscendC::LocalTensor<float> accLocal = accBuf_.Get<float>();
            for (uint32_t j = 0; j < N_; ++j) {
                accLocal(j) = 0.0f;
            }

            // For each column of result C
            for (uint32_t col = 0; col < N_; ++col) {
                // Scalar dot product for column j
                float dot = 0.0f;
                for (uint32_t k = 0; k < K_; ++k) {
                    dot += (float)(rowLocal(k)) * (float)(bGm_[k * N_ + col]);
                }
                accLocal(col) = dot;
            }

            // Write row to output
            AscendC::DataCopy(cGm_[row * N_], accLocal, N_);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> rowBuf_; // for one row of A (half)
    AscendC::TBuf<AscendC::TPosition::VECCALC> colBuf_; // for one column of B (half)
    AscendC::TBuf<AscendC::TPosition::VECCALC> accBuf_; // for N accumulators (float)

    AscendC::GlobalTensor<half> aGm_;
    AscendC::GlobalTensor<half> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_;
    uint32_t K_;
    uint32_t N_;
    uint32_t tileM_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(a, b, c, tilingData.M, tilingData.K, tilingData.N, tilingData.tileM);
    op.Process();
}

#include <torch/library.h>

at::Tensor matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    at::Tensor A_half = A.to(at::kHalf);
    at::Tensor B_half = B.to(at::kHalf);
    at::Tensor C = at::empty({A.size(0), B.size(1)}, A.options().dtype(at::kFloat));
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, A_half, B_half, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("tall_skinny_matrix_multiplication_custom", &matmul_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("tall_skinny_matrix_multiplication_custom", &matmul_impl_npu,
