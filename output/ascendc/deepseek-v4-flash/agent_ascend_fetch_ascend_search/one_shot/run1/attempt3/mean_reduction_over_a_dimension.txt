#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/utils/type_utils.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    // Correct API: GetInputShape returns const gert::Shape*
    const gert::Shape *xShape = context->GetInputShape(0);
    const int64_t ndim = xShape->GetOriginShape().GetDimNum();
    const std::vector<int64_t> &dims = xShape->GetOriginShape().GetDims();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += ndim;

    // Compute outerSize, reduceSize, innerSize
    uint32_t outerSize = 1;
    for (int64_t i = 0; i < dim; i++) outerSize *= static_cast<uint32_t>(dims[i]);
    uint32_t reduceSize = static_cast<uint32_t>(dims[dim]);
    uint32_t innerSize = 1;
    for (int64_t i = dim + 1; i < ndim; i++) innerSize *= static_cast<uint32_t>(dims[i]);

    // UB budget: per-row compute needs innerSize floats for sum and innerSize floats for input slice
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    uint32_t perRowBytes = innerSize * sizeof(float) * 2;  // sum + input
    uint32_t maxRows = (perRowBytes == 0) ? 1 : UB_BUDGET_BYTES / perRowBytes;
    if (maxRows == 0) maxRows = 1;

    uint32_t tileOuter = (maxRows > outerSize) ? outerSize : maxRows;
    if (tileOuter < 1) tileOuter = 1;

    uint32_t blockDim = outerSize;
    if (blockDim > 8) blockDim = 8;   // clamp for typical 910B

    context->SetBlockDim(blockDim);

    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);
    tiling.set_reduceSize(reduceSize);
    tiling.set_tileOuter(tileOuter);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const int64_t ndim = xShape->GetOriginShape().GetDimNum();
    const std::vector<int64_t> &dims = xShape->GetOriginShape().GetDims();
    int64_t dim = context->GetAttrs()->GetAttrPointer<int64_t>(0)[0];
    if (dim < 0) dim += ndim;

    std::vector<int64_t> out_dims;
    for (int64_t i = 0; i < ndim; i++) {
        if (i != dim) out_dims.push_back(dims[i]);
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(static_cast<uint64_t>(out_dims.size()));
    for (size_t i = 0; i < out_dims.size(); i++) {
        y_shape->SetDim(static_cast<int64_t>(i), out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerSize, uint32_t innerSize,
                                uint32_t reduceSize, uint32_t tileOuter,
                                uint32_t blockDim, int64_t dim)
    {
        outerSize_ = outerSize;
        innerSize_ = innerSize;
        reduceSize_ = reduceSize;
        tileOuter_ = tileOuter;
        blockDim_ = blockDim;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)outerSize * reduceSize * innerSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)outerSize * innerSize);

        // Allocate local buffers: sum and input slice
        uint32_t bufSize = innerSize_ * sizeof(float);
        pipe_.InitBuffer(sumBuf_, 1, bufSize);
        pipe_.InitBuffer(inBuf_, 1, bufSize);
    }

    __aicore__ inline void Process()
    {
        if (outerSize_ == 0 || innerSize_ == 0 || reduceSize_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = blockDim_;  // Use the blockDim from tiling

        // Grid-stride over outer rows
        uint32_t outerStart = blockIdx * tileOuter_;
        while (outerStart < outerSize_) {
            uint32_t outerLen = outerSize_ - outerStart;
            if (outerLen > tileOuter_) outerLen = tileOuter_;

            // Process a tile of outer rows at once
            for (uint32_t i = 0; i < outerLen; i++) {
                ProcessRow(outerStart + i);
            }

            outerStart += blockNum * tileOuter_;
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t outerIdx)
    {
        // Initialize sum to zero
        AscendC::LocalTensor<float> sumLocal = sumBuf_.Get<float>();
        AscendC::LocalTensor<float> inLocal = inBuf_.Get<float>();
        AscendC::SetZero(sumLocal, innerSize_);

        // Accumulate over reduction dimension
        uint64_t baseRowOffset = (uint64_t)outerIdx * reduceSize_ * innerSize_;
        for (uint32_t k = 0; k < reduceSize_; k++) {
            uint64_t sliceOffset = baseRowOffset + (uint64_t)k * innerSize_;
            AscendC::DataCopy(inLocal, xGm_[sliceOffset], innerSize_);
            AscendC::Add(sumLocal, sumLocal, inLocal, innerSize_);
        }

        // Divide by reduceSize
        float invReduceSize = 1.0f / (float)reduceSize_;
        AscendC::Muls(sumLocal, sumLocal, invReduceSize, innerSize_);

        // Store output
        uint64_t outOffset = (uint64_t)outerIdx * innerSize_;
        AscendC::DataCopy(yGm_[outOffset], sumLocal, innerSize_);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outerSize_ {0};
    uint32_t innerSize_ {0};
    uint32_t reduceSize_ {0};
    uint32_t tileOuter_ {1};
    uint32_t blockDim_ {1};
    int64_t dim_ {0};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.outerSize, tiling_data.innerSize,
            tiling_data.reduceSize, tiling_data.tileOuter,
            tiling_data.blockDim, tiling_data.dim);
    op.Process();
}
