kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dCustom {
public:
    __aicore__ inline KernelConvTransposed2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t hIn, uint32_t wIn,
                                uint32_t cOutPerGroup, uint32_t kH, uint32_t kW,
                                uint32_t groups,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t outputPadH, uint32_t outputPadW,
                                uint32_t dilationH, uint32_t dilationW)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, n * cIn * hIn * wIn);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, cIn * cOutPerGroup * kH * kW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n * (groups * cOutPerGroup) * 
                             ((hIn-1)*strideH - 2*padH + dilationH*(kH-1) + outputPadH + 1) *
                             ((wIn-1)*strideW - 2*padW + dilationW*(kW-1) + outputPadW + 1));

        conv_.SetInputTensor(xGm_);
        conv_.SetWeightTensor(wGm_);
        conv_.SetOutputTensor(yGm_);
        conv_.SetStride(strideH, strideW);
        conv_.SetPadding(padH, padW, outputPadH, outputPadW);
        conv_.SetDilation(dilationH, dilationW);
        conv_.SetGroups(groups);
        conv_.SetTranspose(true);
    }

    __aicore__ inline void Process()
    {
        conv_.IterateAll();
    }

private:
    AscendC::ConvolutionTranspose2D<AscendC::ConvolutionTranspose2DTraits<4, 4, 4>> conv_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cIn, tiling_data.hIn, tiling_data.wIn,
            tiling_data.cOutPerGroup, tiling_data.kH, tiling_data.kW,
            tiling_data.groups,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.outputPadH, tiling_data.outputPadW,
            tiling_data.dilationH, tiling_data.dilationW);
    op.Process();
}
