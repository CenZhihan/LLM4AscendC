#include "triplet_margin_loss_custom_tiling.h"

namespace optiling {

// UB budget: 5 * tileLen * sizeof(float) + overhead
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;   // alignment for DMA
static constexpr uint32_t MAX_TILE_ELEMS = 4096; // per-tile limit

static uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    // Get input shapes: anchor, positive, negative all same shape (N, D)
    // Note: GetInputShape returns const gert::StorageShape* in CANN 9.0
    const gert::StorageShape *a_shape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(a_shape->GetDim(0));
    uint32_t D = static_cast<uint32_t>(a_shape->GetDim(1));
    uint32_t totalElems = N * D;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float margin = 1.0f;
    if (attrs->GetAttrNum() > 0) {
        margin = *attrs->GetAttrPointer<float>(0);
    }

    // Compute tile length for dimension chunks
    // Per iteration UB: 5 tiles (anchor, pos, neg, diff_ap, diff_an) -> 5 * tileLen floats
    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = ALIGN_ELEMS;
    if (maxTile > MAX_TILE_ELEMS) maxTile = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);

    // tileLen should be at most D
    uint32_t tileLen = (maxTile > D) ? AlignDown(D, ALIGN_ELEMS) : maxTile;
    if (tileLen == 0) tileLen = D;

    // For simplicity, use single block
    uint32_t blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_D(D);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t N, uint32_t D, uint32_t tileLen,
                                float margin)
    {
        N_ = N;
        D_ = D;
        tileLen_ = tileLen;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, N * D);
        posGm_.SetGlobalBuffer((__gm__ float*)positive, N * D);
        negGm_.SetGlobalBuffer((__gm__ float*)negative, N * D);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        // Allocate local tensors for tiled processing
        pipe_.InitBuffer(inQueueAnchor_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueuePos_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueNeg_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueA_, 1, tileLen_ * sizeof(float));  // difference anchor-pos
        pipe_.InitBuffer(tmpQueueB_, 1, tileLen_ * sizeof(float));  // difference anchor-neg
        // For reduction we use scalar registers, no extra queue needed.
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || D_ == 0) {
            lossGm_.SetValue(0, 0.0f);
            return;
        }

        float totalLoss = 0.0f;
        uint32_t validCount = 0;

        // Iterate over each triplet in the batch
        for (uint32_t i = 0; i < N_; ++i) {
            float distAP = 0.0f;
            float distAN = 0.0f;

            // Iterate over tiles of the dimension
            for (uint32_t tileOff = 0; tileOff < D_; tileOff += tileLen_) {
                uint32_t curTileLen = (tileOff + tileLen_ <= D_) ? tileLen_ : (D_ - tileOff);

                // Load anchor tile
                AscendC::LocalTensor<float> aTile = inQueueAnchor_.AllocTensor<float>();
                AscendC::DataCopy(aTile, anchorGm_[i * D_ + tileOff], curTileLen);
                inQueueAnchor_.EnQue(aTile);

                // Load positive tile
                AscendC::LocalTensor<float> pTile = inQueuePos_.AllocTensor<float>();
                AscendC::DataCopy(pTile, posGm_[i * D_ + tileOff], curTileLen);
                inQueuePos_.EnQue(pTile);

                // Load negative tile
                AscendC::LocalTensor<float> nTile = inQueueNeg_.AllocTensor<float>();
                AscendC::DataCopy(nTile, negGm_[i * D_ + tileOff], curTileLen);
                inQueueNeg_.EnQue(nTile);

                // Dequeue and compute
                aTile = inQueueAnchor_.DeQue<float>();
                pTile = inQueuePos_.DeQue<float>();
                nTile = inQueueNeg_.DeQue<float>();

                AscendC::LocalTensor<float> diffAP = tmpQueueA_.AllocTensor<float>();
                AscendC::LocalTensor<float> diffAN = tmpQueueB_.AllocTensor<float>();

                // diffAP = a - p, elementwise
                AscendC::Sub(diffAP, aTile, pTile, curTileLen);
                // diffAN = a - n
                AscendC::Sub(diffAN, aTile, nTile, curTileLen);

                // squares: diffAP = diffAP * diffAP, diffAN = diffAN * diffAN
                AscendC::Mul(diffAP, diffAP, diffAP, curTileLen);
                AscendC::Mul(diffAN, diffAN, diffAN, curTileLen);

                // reduce sum to get partial distances
                float apSum = 0.0f;
                float anSum = 0.0f;
                AscendC::ReduceSum(apSum, diffAP, curTileLen);
                AscendC::ReduceSum(anSum, diffAN, curTileLen);
                distAP += apSum;
                distAN += anSum;

                // Free local tensors
                tmpQueueA_.FreeTensor(diffAP);
                tmpQueueB_.FreeTensor(diffAN);
                inQueueAnchor_.FreeTensor(aTile);
                inQueuePos_.FreeTensor(pTile);
                inQueueNeg_.FreeTensor(nTile);
            }

            // Compute triplet loss for this sample
            float loss_i = distAP - distAN + margin_;
            if (loss_i < 0.0f) loss_i = 0.0f;
            totalLoss += loss_i;
            validCount++;
        }

        float finalLoss = totalLoss / (float)validCount;
        lossGm_.SetValue(0, finalLoss);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePos_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNeg_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueA_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueB_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> posGm_;
    AscendC::GlobalTensor<float> negGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t N_{0};
    uint32_t D_{0};
    uint32_t tileLen_{0};
    float margin_{1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive,
                                                                  GM_ADDR negative, GM_ADDR loss,
                                                                  GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.N, tiling_data.D, tiling_data.tileLen, tiling_data.margin);
    op.Process();
}

                                         const at::Tensor& positive,
                                         const at::Tensor& negative,
                                         double margin) {
    // Prepare inputs: ensure they are contiguous float on NPU
    at::Tensor anchor_contig = anchor.contiguous().npu();
    at::Tensor pos_contig = positive.contiguous().npu();
    at::Tensor neg_contig = negative.contiguous().npu();

    // Output scalar tensor

    EXEC_NPU_CMD(aclnnTripletMarginLossCustom, anchor_contig, pos_contig, neg_contig, margin, loss);
    return loss;
}

    m.impl("triplet_margin_loss_custom", &triplet_margin_loss_impl_npu);
