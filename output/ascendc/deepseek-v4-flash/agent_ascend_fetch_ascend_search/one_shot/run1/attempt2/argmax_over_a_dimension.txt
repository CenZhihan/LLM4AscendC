host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/utils/type_utils.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // ~188KB
static constexpr uint32_t ALIGN_ELEMS = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    // Get input shape (StorageShape)
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const uint32_t dims = xShape->GetDimNum();

    // Get the axis attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    int64_t axis = *dimAttr;
    if (axis < 0) axis += static_cast<int64_t>(dims);
    uint32_t uAxis = static_cast<uint32_t>(axis);

    // Compute outerTotal (product of all dims except axis) and axisSize
    uint32_t outerTotal = 1;
    uint32_t axisSize = 1;
    for (uint32_t i = 0; i < dims; ++i) {
        uint32_t s = static_cast<uint32_t>(xShape->GetDim(i));
        if (i == uAxis) {
            axisSize = s;
        } else {
            outerTotal *= s;
        }
    }

    // UB requirements: one float buffer of axisSize and one int32 buffer of 1 element
    uint32_t neededBytes = axisSize * sizeof(float) + sizeof(int32_t);
    uint32_t maxAxisElements = (UB_BUDGET_BYTES - sizeof(int32_t)) / sizeof(float);
    if (axisSize > maxAxisElements) {
        axisSize = AlignDown(maxAxisElements, ALIGN_ELEMS);
    }

    // Block dim heuristic: based on outerTotal
    uint32_t blockDim = (outerTotal + 255) / 256; // at least 1
    if (blockDim < 4) blockDim = 4;
    if (blockDim > 32) blockDim = 32;

    context->SetBlockDim(blockDim);

    tiling.set_outerTotal(outerTotal);
    tiling.set_axisSize(axisSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const uint32_t dims = xShape->GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    int64_t axis = *dimAttr;
    if (axis < 0) axis += static_cast<int64_t>(dims);
    uint32_t uAxis = static_cast<uint32_t>(axis);

    uint32_t newDims = dims - 1;
    gert::Shape outShape;
    outShape.SetDimNum(newDims);
    uint32_t idx = 0;
    for (uint32_t i = 0; i < dims; ++i) {
        if (i != uAxis) {
            outShape.SetDim(idx, static_cast<int64_t>(xShape->GetDim(i)));
            ++idx;
        }
    }
    *context->GetOutputShape(0) = outShape;
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, ge::DT_INT32);
    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerTotal, uint32_t axisSize)
    {
        outerTotal_ = outerTotal;
        axisSize_ = axisSize;
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ int32_t*)y);

        // Allocate local buffers: one float buffer of axisSize and one int32 buffer of 1 element
        pipe_.InitBuffer(inXBuf_, 1, axisSize_ * sizeof(float));
        pipe_.InitBuffer(outIdxBuf_, 1, sizeof(int32_t));
    }

    __aicore__ inline void Process()
    {
        if (outerTotal_ == 0 || axisSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a range of outer indices
        uint32_t start = blockIdx;
        uint32_t end = outerTotal_;
        for (uint32_t outerIdx = start; outerIdx < end; outerIdx += blockNum) {
            // Copy one row of axisSize floats from GM to local
            AscendC::LocalTensor<float> xLocal = inXBuf_.AllocTensor<float>();
            uint32_t gmOffset = outerIdx * axisSize_; // row offset in global memory
            AscendC::DataCopy(xLocal, xGm_[gmOffset], axisSize_);

            // Compute argmax on this row
            AscendC::LocalTensor<int32_t> idxLocal = outIdxBuf_.AllocTensor<int32_t>();
            AscendC::ReduceMax(xLocal, xLocal, idxLocal, axisSize_, 1); // per-segment (entire row)

            // Write the index to global output at position outerIdx
            AscendC::DataCopy(yGm_[outerIdx], idxLocal, 1);

            // Free local tensors
            outIdxBuf_.FreeTensor(idxLocal);
            inXBuf_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inXBuf_;     // single float buffer for row input
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outIdxBuf_; // single int32 buffer for index output

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int32_t> yGm_;

    uint32_t outerTotal_ {0};
    uint32_t axisSize_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.outerTotal, tiling_data.axisSize);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor argmax_impl_npu(const at::Tensor& self, int64_t dim) {
    // The custom kernel returns int32; convert to int64 to match PyTorch's argmax.
    at::Tensor result_int32 = at::empty(self.sizes(), self.options().dtype(at::kInt));
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result_int32);
    return result_int32.to(at::kLong);
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_impl_npu, "Argmax over a dimension (ArgmaxOverADimensionCustom on NPU)");
}
