host_operator_src="""
#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    const int32_t ndim = static_cast<int32_t>(x_shape.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim_ptr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dim_ptr;
    if (dim < 0) dim += ndim;

    uint32_t before = 1, rowLen = 1, after = 1;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i == dim) {
            rowLen = x_shape.GetDim(i);
        } else if (i < dim) {
            before *= x_shape.GetDim(i);
        } else {
            after *= x_shape.GetDim(i);
        }
    }

    uint32_t totalBatches = before * after;
    uint32_t colLen = after;

    // Determine tileLen for columns: need 3 float buffers + 1 int64 buffer per tile
    // items per column tile: 3 floats + 1 int64 => 3*4 + 8 = 20 bytes per element
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    // Adapt to colLen if smaller
    uint32_t tileLen = (colLen < maxTile) ? colLen : maxTile;

    // Each block processes a set of output elements (batch, col)
    // Distribute tileLen columns across blocks: each block gets a column tile
    uint32_t numColTiles = (colLen + tileLen - 1) / tileLen;
    uint32_t numBlocks = totalBatches * numColTiles;
    if (numBlocks < 8) numBlocks = 8;
    if (numBlocks > 24) numBlocks = 24;

    context->SetBlockDim(numBlocks);

    tiling.set_totalBatches(totalBatches);
    tiling.set_rowLen(rowLen);
    tiling.set_colLen(colLen);
    tiling.set_blockDim(numBlocks);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const int32_t ndim = static_cast<int32_t>(x_shape->GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim_ptr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dim_ptr;
    if (dim < 0) dim += ndim;

    std::vector<int64_t> out_dims;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i == dim) continue;
        out_dims.push_back(x_shape->GetDim(i));
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(static_cast<uint32_t>(out_dims.size()));
    for (uint32_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalBatches, uint32_t rowLen,
                                uint32_t colLen, uint32_t blockDim,
                                int32_t dim)
    {
        totalBatches_ = totalBatches;
        rowLen_ = rowLen;
        colLen_ = colLen;
        blockDim_ = blockDim;
        dim_ = dim;

        // For output, each output element corresponds to a (batch, col) pair.
        // We distribute work across blocks: each block handles a range of (batch, col)
        // but we process columns tile by tile within each batch.
        // Total output elements: totalBatches * colLen
        // We'll assign each block a contiguous set of output elements.
        // However, tiling for columns: we process tileLen columns per iteration.
        // For simplicity, we treat output index as (batch * colLen + col) linear.
        // Each block will iterate over its assigned output indices, but we need to
        // load the corresponding input rows. To make data access contiguous,
        // we process column tiles: for each batch, we load all rows for a tile of columns.
        // So we map each block to a (batch, colTile) pair.

        uint32_t numColTiles = (colLen_ + tileLen_ - 1) / tileLen_;
        uint32_t batchTileIdx = blockIdx_ / numColTiles;
        uint32_t colTileIdx = blockIdx_ % numColTiles;

        batchStart_ = batchTileIdx;
        colStart_ = colTileIdx * tileLen_;
        colEnd_ = colStart_ + tileLen_;
        if (colEnd_ > colLen_) colEnd_ = colLen_;
        tileLenActual_ = colEnd_ - colStart_;

        inputBase_ = (__gm__ float*)x;
        outputBase_ = (__gm__ int64_t*)y;

        // Initialize buffers
        pipe_.InitBuffer(inQueueMaxVals_, BUFFER_NUM, tileLenActual_ * sizeof(float));
        pipe_.InitBuffer(inQueueMaxIdxs_, BUFFER_NUM, tileLenActual_ * sizeof(int64_t));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileLenActual_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rowLen_ == 0 || colLen_ == 0) return;

        // Each block processes one batch (batchStart_) and one column tile.
        uint32_t batch = batchStart_;
        if (batch >= totalBatches_) return;

        // Load first row of the tile
        uint32_t colOffset = colStart_;
        uint32_t tileCols = tileLenActual_;

        // Initialize max values with first row's tile and indices to 0
        uint32_t baseIdx = batch * rowLen_ * colLen_ + 0 * colLen_ + colOffset; // row 0
        AscendC::LocalTensor<float> maxVals = inQueueMaxVals_.AllocTensor<float>();
        AscendC::LocalTensor<int64_t> maxIdxs = inQueueMaxIdxs_.AllocTensor<float>(); // reuse float queue for int64? Actually need separate. We'll use tmpQueue for int64 after conversion.
        // Better: use two separate queues: one for maxVals (float), one for maxIdxs (int64). We'll use inQueueMaxVals_ for maxVals, inQueueMaxIdxs_ for maxIdxs.
        // We already have inQueueMaxVals_ and inQueueMaxIdxs_ declared.
        AscendC::LocalTensor<int64_t> idxLocal = inQueueMaxIdxs_.AllocTensor<int64_t>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Copy first row tile into maxVals
        AscendC::DataCopy(maxVals, inputBase_[baseIdx], tileCols);
        // Set initial indices to 0
        AscendC::Mul(idxLocal, 0.0f, tileCols); // not correct; we need to set to 0, but Mul expects float. Use SetValue.
        AscendC::SetValue(idxLocal, 0, tileCols);

        // Loop over remaining rows
        for (uint32_t r = 1; r < rowLen_; ++r) {
            uint32_t baseRow = batch * rowLen_ * colLen_ + r * colLen_ + colOffset;
            AscendC::DataCopy(tLocal, inputBase_[baseRow], tileCols);

            // Compare: where tLocal > maxVals, update maxVals and index
            AscendC::Compare(idxTemp_, tLocal, maxVals, AscendC::CMPMODE::GT, tileCols); // idxTemp_ uint8
            // Use MaskedUpdate: maxVals = idxTemp_ ? tLocal : maxVals
            AscendC::Select(maxVals, idxTemp_, tLocal, maxVals, tileCols);
            // For indices: need to set to r where condition true
            AscendC::Select(idxLocal, idxTemp_, (int64_t)(r), idxLocal, tileCols);
        }

        // Write result: base output index for this batch and column tile
        uint32_t outBaseIdx = batch * colLen_ + colOffset;
        AscendC::DataCopy(outputBase_[outBaseIdx], idxLocal, tileCols);

        // Free tensors
        inQueueMaxVals_.FreeTensor(maxVals);
        inQueueMaxIdxs_.FreeTensor(idxLocal);
        tmpQueue_.FreeTensor(tLocal);
    }

private:
    // Queues: two input, one output, one temp
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMaxVals_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMaxIdxs_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> inputBase_;
    AscendC::GlobalTensor<int64_t> outputBase_;

    uint32_t totalBatches_;
    uint32_t rowLen_;
    uint32_t colLen_;
    uint32_t blockDim_;
    int32_t dim_;
    uint32_t batchStart_;
    uint32_t colStart_;
    uint32_t colEnd_;
    uint32_t tileLenActual_;
    uint32_t tileLen_ = 2048; // will be overridden by tiling data

    // We'll need a temporary for compare mask, but can reuse a small local
    AscendC::LocalTensor<uint8_t> idxTemp_;
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalBatches, tiling_data.rowLen,
            tiling_data.colLen, tiling_data.blockDim,
            tiling_data.dim);
    op.Process();
}
