#include "triplet_margin_loss_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const auto* anchorShape = context->GetInputShape(0);
    const uint32_t N = anchorShape->GetDim(0);
    const uint32_t D = anchorShape->GetDim(1);

    // UB budget: each tile holds 3 * tileBatch * D floats + intermediates
    uint32_t maxElem = (UB_BUDGET_BYTES / (3 * sizeof(float))) / 2;
    uint32_t tileBatch = AlignDown(maxElem / D, 8);
    if (tileBatch < 1) tileBatch = 1;
    if (tileBatch > N) tileBatch = N;
    tileBatch = AlignDown(tileBatch, 16);
    if (tileBatch == 0) tileBatch = 1;

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float margin = (attrs->GetAttrPointer<float>(0) != nullptr) ? *attrs->GetAttrPointer<float>(0) : 1.0f;

    tiling.set_totalBatch(N);
    tiling.set_featureDim(D);
    tiling.set_tileBatch(tileBatch);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t totalBatch, uint32_t featureDim,
                                uint32_t tileBatch, float margin)
    {
        totalBatch_ = totalBatch;
        featureDim_ = featureDim;
        tileBatch_ = tileBatch;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss);

        // Buffer for one tile (each sample vector of length D)
        // Double buffering for anchor, positive, negative
        pipe_.InitBuffer(anchorQue_, BUFFER_NUM, tileBatch_ * featureDim_ * sizeof(float));
        pipe_.InitBuffer(positiveQue_, BUFFER_NUM, tileBatch_ * featureDim_ * sizeof(float));
        pipe_.InitBuffer(negativeQue_, BUFFER_NUM, tileBatch_ * featureDim_ * sizeof(float));
        // Temporary for one sample's diff (D floats) and squared diff
        pipe_.InitBuffer(diffQue_, 1, featureDim_ * sizeof(float));
        pipe_.InitBuffer(sqQue_, 1, featureDim_ * sizeof(float));
        // Per-tile distance arrays (tileBatch floats each)
        pipe_.InitBuffer(posDistQue_, 1, tileBatch_ * sizeof(float));
        pipe_.InitBuffer(negDistQue_, 1, tileBatch_ * sizeof(float));
        // Output pad buffer (8 floats = 32B aligned)
        pipe_.InitBuffer(outPadQue_, 1, 8 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalBatch_ == 0 || featureDim_ == 0) return;

        float totalLoss = 0.0f;
        uint32_t processed = 0;

        while (processed < totalBatch_) {
            uint32_t curBatch = tileBatch_;
            if (processed + curBatch > totalBatch_)
                curBatch = totalBatch_ - processed;

            uint32_t offset = processed * featureDim_;

            // Prefetch tile (anchor, positive, negative)
            CopyIn(offset, curBatch);

            // Compute distances for all samples in this tile
            Compute(curBatch);

            // Compute loss for each sample and accumulate
            AscendC::LocalTensor<float> posDist = posDistQue_.DeQue<float>();
            AscendC::LocalTensor<float> negDist = negDistQue_.DeQue<float>();

            for (uint32_t i = 0; i < curBatch; ++i) {
                float dPos = posDist(i);
                float dNeg = negDist(i);
                float loss_i = dPos - dNeg + margin_;
                if (loss_i < 0.0f) loss_i = 0.0f;
                totalLoss += loss_i;
            }

            posDistQue_.FreeTensor(posDist);
            negDistQue_.FreeTensor(negDist);

            processed += curBatch;
        }

        // Write scalar result using DataCopyPad for 32B alignment
        AscendC::LocalTensor<float> outPad = outPadQue_.AllocTensor<float>();
        outPad.SetValue(0, totalLoss);
        // Pad with zeros (rest already zero from Init?)
        // We'll clear to be safe
        for (uint32_t i = 1; i < 8; ++i) {
            outPad.SetValue(i, 0.0f);
        }
        AscendC::DataCopyPad(lossGm_[0], outPad, 8);
        outPadQue_.FreeTensor(outPad);
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t batch)
    {
        uint32_t bytes = batch * featureDim_ * sizeof(float);
        auto anchorLocal = anchorQue_.AllocTensor<float>();
        auto positiveLocal = positiveQue_.AllocTensor<float>();
        auto negativeLocal = negativeQue_.AllocTensor<float>();
        AscendC::DataCopy(anchorLocal, anchorGm_[offset], bytes / sizeof(float));
        AscendC::DataCopy(positiveLocal, positiveGm_[offset], bytes / sizeof(float));
        AscendC::DataCopy(negativeLocal, negativeGm_[offset], bytes / sizeof(float));
        anchorQue_.EnQue(anchorLocal);
        positiveQue_.EnQue(positiveLocal);
        negativeQue_.EnQue(negativeLocal);
    }

    __aicore__ inline void Compute(uint32_t batch)
    {
        // Dequeue the tile data
        AscendC::LocalTensor<float> anchorTile = anchorQue_.DeQue<float>();
        AscendC::LocalTensor<float> positiveTile = positiveQue_.DeQue<float>();
        AscendC::LocalTensor<float> negativeTile = negativeQue_.DeQue<float>();

        // Allocate per-sample distance arrays
        AscendC::LocalTensor<float> posDist = posDistQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> negDist = negDistQue_.AllocTensor<float>();

        // For each sample, compute Euclidean distance
        for (uint32_t i = 0; i < batch; ++i) {
            uint32_t rowOffset = i * featureDim_;

            // Extract row pointers (actually use SubTensor)
            AscendC::LocalTensor<float> anchorRow = anchorTile[rowOffset];
            AscendC::LocalTensor<float> positiveRow = positiveTile[rowOffset];
            AscendC::LocalTensor<float> negativeRow = negativeTile[rowOffset];

            // Compute posDist = sqrt(sum((anchor-positive)^2))
            AscendC::LocalTensor<float> diff = diffQue_.AllocTensor<float>();
            AscendC::Sub(diff, anchorRow, positiveRow, featureDim_);
            AscendC::Mul(diff, diff, diff, featureDim_);
            float posSq = AscendC::ReduceSum(diff, featureDim_);
            float posSqrt = AscendC::Sqrt(posSq);
            posDist.SetValue(i, posSqrt);
            diffQue_.FreeTensor(diff);

            // Compute negDist = sqrt(sum((anchor-negative)^2))
            AscendC::LocalTensor<float> diff2 = sqQue_.AllocTensor<float>(); // reuse sq queue
            AscendC::Sub(diff2, anchorRow, negativeRow, featureDim_);
            AscendC::Mul(diff2, diff2, diff2, featureDim_);
            float negSq = AscendC::ReduceSum(diff2, featureDim_);
            float negSqrt = AscendC::Sqrt(negSq);
            negDist.SetValue(i, negSqrt);
            sqQue_.FreeTensor(diff2);
        }

        // Enqueue distance arrays for loss computation
        posDistQue_.EnQue(posDist);
        negDistQue_.EnQue(negDist);

        // Free tile buffers
        anchorQue_.FreeTensor(anchorTile);
        positiveQue_.FreeTensor(positiveTile);
        negativeQue_.FreeTensor(negativeTile);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> anchorQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> positiveQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> negativeQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> diffQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sqQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> posDistQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> negDistQue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outPadQue_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalBatch_;
    uint32_t featureDim_;
    uint32_t tileBatch_;
    float margin_;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(
    GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalBatch, tiling_data.featureDim,
            tiling_data.tileBatch, tiling_data.margin);
    op.Process();
}
