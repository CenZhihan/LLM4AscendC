#include "vanilla_rnn_hidden_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ELEM_SIZE = sizeof(float);

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    VanillaRnnHiddenCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_i2h_shape = context->GetInputShape(2);
    const gert::Shape *w_h2o_shape = context->GetInputShape(4);

    uint32_t T = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t B = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t I = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t H = static_cast<uint32_t>(w_i2h_shape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(w_i2h_shape->GetDim(1));
    uint32_t O = static_cast<uint32_t>(w_h2o_shape->GetDim(0));

    // Calculate memory for one time step
    uint32_t inpBytes = B * K * ELEM_SIZE;
    uint32_t preBytes = B * H * ELEM_SIZE;
    uint32_t hBytes = B * H * ELEM_SIZE;
    uint32_t yBytes = B * O * ELEM_SIZE;
    uint32_t totalStepBytes = inpBytes + preBytes + hBytes + yBytes;

    // Tiling for matmuls: reduce outer dim (H for first, O for second) if needed
    uint32_t tileH = H;
    uint32_t tileO = O;
    uint32_t tileK = K;

    // First matmul: inp (B,K) @ w_i2h.T (K,H) -> pre (B,H)
    // We'll tile along H (output dim)
    uint32_t workspaceBytes = totalStepBytes + 2 * B * tileH * ELEM_SIZE;
    while (workspaceBytes > UB_BUDGET_BYTES && tileH > 1) {
        tileH /= 2;
        workspaceBytes = totalStepBytes + 2 * B * tileH * ELEM_SIZE;
    }

    // Second matmul: h (B,H) @ w_h2o.T (H,O) -> y (B,O)
    workspaceBytes = totalStepBytes + 2 * B * tileO * ELEM_SIZE;
    while (workspaceBytes > UB_BUDGET_BYTES && tileO > 1) {
        tileO /= 2;
        workspaceBytes = totalStepBytes + 2 * B * tileO * ELEM_SIZE;
    }

    // Use single block (sequential time steps)
    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_T(T);
    tiling.set_B(B);
    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_tileH(tileH);
    tiling.set_tileO(tileO);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_h2o_shape = context->GetInputShape(4);
    gert::Shape y_shape = *x_shape;
    y_shape.SetDim(2, w_h2o_shape->GetDim(0));
    *context->GetOutputShape(0) = y_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType dt = context->GetInputDataType(0);
    context->SetOutputDataType(0, dt);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"
#include "matmul_api.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                 GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                 uint32_t T, uint32_t B, uint32_t I, uint32_t H,
                                 uint32_t O, uint32_t K,
                                 uint32_t tileK, uint32_t tileH, uint32_t tileO) {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O; K_ = K;
        tileK_ = tileK; tileH_ = tileH; tileO_ = tileO;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T * B * I);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B * H);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H * K);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O * H);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O);
        yGm_.SetGlobalBuffer((__gm__ float*)y, T * B * O);

        // Allocate local buffers for one time step
        pipe_.InitBuffer(inpQueue_, BUFFER_NUM, B_ * K_ * sizeof(float));
        pipe_.InitBuffer(hQueue_, BUFFER_NUM, B_ * H_ * sizeof(float));
        pipe_.InitBuffer(preQueue_, BUFFER_NUM, B_ * H_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, B_ * O_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, B_ * H_ * sizeof(float)); // for tanh input copy
    }

    __aicore__ inline void Process() {
        // Initialize h with h0
        LocalTensor<float> hLocal = hQueue_.AllocTensor<float>();
        DataCopy(hLocal, h0Gm_, B_ * H_);
        hQueue_.EnQue(hLocal);

        // Prefetch weights and biases into UB (constant reuse)
        pipe_.InitBuffer(w_i2hLocal_, 1, H_ * K_ * sizeof(float));
        pipe_.InitBuffer(b_i2hLocal_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(w_h2oLocal_, 1, O_ * H_ * sizeof(float));
        pipe_.InitBuffer(b_h2oLocal_, 1, O_ * sizeof(float));
        DataCopy(w_i2hLocal_.Get<float>(), w_i2hGm_, H_ * K_);
