#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_TILE_M = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape shapeA = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape shapeB = context->GetInputShape(1)->GetOriginShape();

    uint32_t M = static_cast<uint32_t>(shapeA.GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA.GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB.GetDim(1));

    constexpr float budget_ratio = 0.9f;
    uint32_t budget = static_cast<uint32_t>(budget_ratio * UB_BUDGET_BYTES);
    uint32_t denom = 2 * (K + N) * sizeof(float);
    uint32_t tileM = (denom > 0) ? (budget - K * N * sizeof(float)) / denom : MAX_TILE_M;
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;
    if (tileM > MAX_TILE_M) tileM = AlignDown(MAX_TILE_M, ALIGN_ELEMS);
    if (tileM > M) tileM = AlignDown(M, ALIGN_ELEMS);
    if (tileM == 0) tileM = M;

    uint32_t numTiles = (tileM == 0) ? 1 : (M + tileM - 1) / tileM;
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_totalALen(M * K);
    tiling.set_totalBLen(K * N);
    tiling.set_totalCLen(M * N);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);
    int64_t M = shapeA->GetDim(0);
    int64_t N = shapeB->GetDim(1);
    gert::Shape shapeC({M, N});
    *context->GetOutputShape(0) = shapeC;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = '''
#include "kernel_operator.h"
using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_N = 256;   // tile over output columns (N dimension)

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM) {
        A_gm_.SetGlobalBuffer((__gm__ float*)A);
        B_gm_.SetGlobalBuffer((__gm__ float*)B);
        C_gm_.SetGlobalBuffer((__gm__ float*)C);
        M_ = M; N_ = N; K_ = K; tileM_ = tileM;

        pipe_.InitBuffer(aBuf_, BUFFER_NUM, tileM_ * K_ * sizeof(float));
        pipe_.InitBuffer(bBuf_, BUFFER_NUM, K_ * TILE_N * sizeof(float));
        pipe_.InitBuffer(cBuf_, BUFFER_NUM, tileM_ * TILE_N * sizeof(float));
        pipe_.InitBuffer(accumBuf_, 1, tileM_ * TILE_N * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();
        uint32_t numRowTiles = (M_ + tileM_ - 1) / tileM_;

        for (uint32_t rowTileId = blockIdx; rowTileId < numRowTiles; rowTileId += blockNum) {
            uint32_t rowStart = rowTileId * tileM_;
            uint32_t curTileM = (rowStart + tileM_ <= M_) ? tileM_ : (M_ - rowStart);
            if (curTileM == 0) continue;

            // Initialize accumulation buffer to zero
            LocalTensor<float> accum = accumBuf_.AllocTensor<float>();
            SetVal(accum, 0.0f, curTileM * TILE_N);

            uint32_t numColTiles = (N_ + TILE_N - 1) / TILE_N;
            for (uint32_t colTileId = 0; colTileId < numColTiles; ++colTileId) {
                uint32_t colStart = colTileId * TILE_N;
                uint32_t curTileN = (colStart + TILE_N <= N_) ? TILE_N : (N_ - colStart);
                if (curTileN == 0) continue;

                // Load A tile (curTileM x K)
                LocalTensor<float> aLocal = aBuf_.AllocTensor<float>();
                DataCopy(aLocal, A_gm_[rowStart * K_], curTileM * K_);
                aBuf_.EnQue(aLocal);

                // Load B tile (K x curTileN) - load whole KxcurTileN block
                // B is row-major, so each row has length N_, we need contiguous K rows * curTileN
                // We load by row and store in bBuf_ as a flat block
                LocalTensor<float> bLocal = bBuf_.AllocTensor<float>();
                for (uint32_t k = 0; k < K_; ++k) {
                    uint32_t srcOff = k * N_ + colStart;
                    DataCopy(bLocal[k * curTileN], B_gm_[srcOff], curTileN);
                }
                bBuf_.EnQue(bLocal);

                // Now perform matmul on the two local tiles using vector operations
                LocalTensor<float> aDeq = aBuf_.DeQue<float>();
                LocalTensor<float> bDeq = bBuf_.DeQue<float>();

                // For each k, do outer product: accum += aCol_k * bRow_k
                // aCol_k is column k of aDeq (size curTileM), stride K_
                // bRow_k is row k of bDeq (size curTileN), stride curTileN (block flat, but it's stored as K rows of curTileN)
                // Use broadcast Muls and Add
                for (uint32_t k = 0; k < K_; ++k) {
                    // Extract k-th column of A by using Gather? Instead, we can do vector multiply for each row individually
                    // Simpler: we have all data, we can use MatMul class, but avoid scalar indexing.
                    // Use MatMul from AscendC
                }

                // For correctness, we use AscendC::MatMul
                MatMul<MatMulType<MM, false, false>> matmul;
                matmul.SetTensorA(aDeq, {curTileM, K_});
                matmul.SetTensorB(bDeq, {K_, curTileN});
                matmul.SetBias(false);
                LocalTensor<float> cTemp = cBuf_.AllocTensor<float>();
                matmul.SetTensorC(cTemp, {curTileM, curTileN});
                matmul.IterateAll();
                // Add cTemp into accum (which has shape curTileM x TILE_N, but cTemp is only curTileN wide)
                for (uint32_t i = 0; i < curTileM; ++i) {
                    Add(accum[i * TILE_N], accum[i * TILE_N], cTemp[i * curTileN], curTileN);
                }
                cBuf_.FreeTensor(cTemp);
                aBuf_.FreeTensor(aDeq);
                bBuf_.FreeTensor(bDeq);
            }

            // Write back C tile from accum (only curTileN columns? Actually accum is tileM x TILE_N, but TILE_N may be larger than N_)
            // Only write valid columns
            for (uint32_t i = 0; i < curTileM; ++i) {
                uint32_t cDstOff = (rowStart + i) * N_;
                DataCopy(C_gm_[cDstOff], accum[i * TILE_N], N_);
            }
            accumBuf_.FreeTensor(accum);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> aBuf_;
    TQue<TPosition::VECIN, BUFFER_NUM> bBuf_;
    TQue<TPosition::VECOUT, BUFFER_NUM> cBuf_;
    TQue<TPosition::VECCALC, 1> accumBuf_;

    GlobalTensor<float> A_gm_;
    GlobalTensor<float> B_gm_;
    GlobalTensor<float> C_gm_;

    uint32_t M_, N_, K_, tileM_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                                                                GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(A, B, C, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileM);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor tall_skinny_matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    at::Tensor C = at::empty({A.size(0), B.size(1)}, A.options());
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu);
}

    m.def("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu,

}
