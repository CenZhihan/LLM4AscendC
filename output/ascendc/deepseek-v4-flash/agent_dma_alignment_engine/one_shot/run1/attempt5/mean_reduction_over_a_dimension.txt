#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MeanReductionOverADimensionCustomTilingData tiling;

    // Fix: use GetOriginShape() to obtain a const Shape&
    const gert::Shape &shape = context->GetInputShape(0)->GetOriginShape();
    int32_t ndim = shape.GetDimNum();

    std::vector<int64_t> dims_vec(ndim);
    for (int32_t i = 0; i < ndim; i++) {
        dims_vec[i] = shape.GetDim(i);
    }
    const int64_t *dims = dims_vec.data();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;

    uint32_t outerSize = 1;
    for (int32_t i = 0; i < dim; i++) {
        outerSize *= dims[i];
    }
    uint32_t reduceSize = dims[dim];
    uint32_t innerSize = 1;
    for (int32_t i = dim + 1; i < ndim; i++) {
        innerSize *= dims[i];
    }

    uint32_t reduceStride = innerSize;
    uint32_t outerStride = reduceSize * innerSize;
    uint32_t totalInputElems = shape.GetShapeSize();
    uint32_t totalOutputElems = outerSize * innerSize;

    // UB budget: 1 accumulator + 1 input row = 2 * tile_inner floats (no double buffer for simplicity)
    uint32_t maxTileInner = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTileInner = AlignDown(maxTileInner, ALIGN_ELEMS);
    if (maxTileInner < ALIGN_ELEMS) maxTileInner = ALIGN_ELEMS;
    if (maxTileInner > HARD_CAP_ELEMS) maxTileInner = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = (maxTileInner < innerSize) ? maxTileInner : innerSize;

    uint32_t blockDim = outerSize;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalInputElems(totalInputElems);
    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);
    tiling.set_reduceSize(reduceSize);
    tiling.set_reduceStride(reduceStride);
    tiling.set_outerStride(outerStride);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputElems, uint32_t totalOutputElems,
                                uint32_t outerSize, uint32_t innerSize,
                                uint32_t reduceSize, uint32_t reduceStride,
                                uint32_t outerStride, uint32_t tileLen,
                                int32_t dim)
    {
        totalInputElems_ = totalInputElems;
        totalOutputElems_ = totalOutputElems;
        outerSize_ = outerSize;
        innerSize_ = innerSize;
        reduceSize_ = reduceSize;
        reduceStride_ = reduceStride;
        outerStride_ = outerStride;
        tileLen_ = tileLen;
        dim_ = dim;
        invReduceSize_ = 1.0f / reduceSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputElems);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElems);

        // Buffers:
        // input row (tileLen floats) with double buffer
        // accumulator (tileLen floats)
        // output row (tileLen floats) with double buffer
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // Single accumulator reused per tile
        pipe_.InitBuffer(accQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid stride over outer indices
        for (uint32_t outerIdx = blockIdx; outerIdx < outerSize_; outerIdx += blockNum) {
            uint32_t baseOut = outerIdx * innerSize_;
            uint32_t baseIn = outerIdx * outerStride_;

            // Tile over inner dimension
            for (uint32_t iStart = 0; iStart < innerSize_; iStart += tileLen_) {
                uint32_t curTile = innerSize_ - iStart;
                if (curTile > tileLen_) curTile = tileLen_;

                // Allocate accumulator once per tile
                AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();

                // Process all reduce rows
                for (uint32_t r = 0; r < reduceSize_; r++) {
                    // Fetch input row
                    uint32_t rowOffset = baseIn + iStart + r * innerSize_;
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[rowOffset], curTile);
                    // Accumulate
                    if (r == 0) {
                        AscendC::DataCopy(acc, xLocal, curTile);
                    } else {
                        AscendC::Add(acc, acc, xLocal, curTile);
                    }
                    inQueueX_.FreeTensor(xLocal);
                }

                // Divide by reduce size and write output
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Muls(yLocal, acc, invReduceSize_, curTile);
                AscendC::DataCopy(yGm_[baseOut + iStart], yLocal, curTile);
                outQueueY_.FreeTensor(yLocal);
                accQueue_.FreeTensor(acc);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputElems_ {0};
    uint32_t totalOutputElems_ {0};
    uint32_t outerSize_ {1};
    uint32_t innerSize_ {1};
    uint32_t reduceSize_ {8};
    uint32_t reduceStride_ {1};
    uint32_t outerStride_ {1};
    uint32_t tileLen_ {1};
    int32_t dim_ {0};
    float invReduceSize_ {1.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalInputElems, tiling_data.totalOutputElems,
            tiling_data.outerSize, tiling_data.innerSize,
            tiling_data.reduceSize, tiling_data.reduceStride,
            tiling_data.outerStride, tiling_data.tileLen,
            tiling_data.dim);
    op.Process();
}

#include <torch/library.h>

at::Tensor mean_reduction_impl_npu(const at::Tensor& self, int64_t dim) {
    auto ndim = self.dim();
    auto dim_bounded = dim < 0 ? dim + ndim : dim;
    std::vector<int64_t> out_shape;
    for (int64_t i = 0; i < ndim; i++) {
        if (i != dim_bounded) out_shape.push_back(self.size(i));
    }
    at::Tensor result = at::empty(out_shape, self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, dim, result);
    return result;
}
