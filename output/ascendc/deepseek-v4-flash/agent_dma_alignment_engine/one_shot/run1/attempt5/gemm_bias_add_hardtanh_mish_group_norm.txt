kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
                                GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t numGroups, float eps)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        eps_ = eps;

        blockIdx_ = AscendC::GetBlockIdx();
        if (blockIdx_ >= batchSize_) {
            initialized_ = false;
            return;
        }
        initialized_ = true;

        // Set global pointers for this sample
        xGm_.SetGlobalBuffer((__gm__ float*)x + blockIdx_ * inFeatures_, inFeatures_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures_ * inFeatures_);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures_);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures_);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures_);
        yGm_.SetGlobalBuffer((__gm__ float*)y + blockIdx_ * outFeatures_, outFeatures_);
    }

    __aicore__ inline void Process()
    {
        if (!initialized_) return;

        // Allocate local buffers
        AscendC::LocalTensor<float> xLocal;
        AscendC::LocalTensor<float> yLocal;
        AscendC::LocalTensor<float> wRow;
        AscendC::LocalTensor<float> tmp1;
        AscendC::LocalTensor<float> tmp2;
        AscendC::LocalTensor<float> linBiasLocal;
        AscendC::LocalTensor<float> biasLocal;
        AscendC::LocalTensor<float> gammaLocal;
        AscendC::LocalTensor<float> betaLocal;
        AscendC::LocalTensor<float> meanLocal;
        AscendC::LocalTensor<float> varLocal;

        pipe_.InitBuffer(xBuf_, 1, inFeatures_ * sizeof(float));
        pipe_.InitBuffer(yBuf_, 1, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(wRowBuf_, 1, inFeatures_ * sizeof(float));
        pipe_.InitBuffer(tBuf1_, 1, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(tBuf2_, 1, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(linBiasBuf_, 1, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(biasBuf_, 1, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(gammaBuf_, 1, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(betaBuf_, 1, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(meanBuf_, 1, numGroups_ * sizeof(float));
        pipe_.InitBuffer(varBuf_, 1, numGroups_ * sizeof(float));

        xLocal = xBuf_.Get<float>();
        yLocal = yBuf_.Get<float>();
        wRow = wRowBuf_.Get<float>();
        tmp1 = tBuf1_.Get<float>();
        tmp2 = tBuf2_.Get<float>();
        linBiasLocal = linBiasBuf_.Get<float>();
        biasLocal = biasBuf_.Get<float>();
        gammaLocal = gammaBuf_.Get<float>();
        betaLocal = betaBuf_.Get<float>();
        meanLocal = meanBuf_.Get<float>();
        varLocal = varBuf_.Get<float>();

        // Load constant buffers
        AscendC::DataCopy(linBiasLocal, linBiasGm_, outFeatures_);
        AscendC::DataCopy(biasLocal, biasGm_, outFeatures_);
        AscendC::DataCopy(gammaLocal, gnGammaGm_, outFeatures_);
        AscendC::DataCopy(betaLocal, gnBetaGm_, outFeatures_);

        // Load input x
        AscendC::DataCopy(xLocal, xGm_, inFeatures_);

        // Step 1: GEMM + bias + hardtanh + mish
        for (uint32_t i = 0; i < outFeatures_; i++) {
            // Load weight row i
            AscendC::DataCopy(wRow, weightGm_[i * inFeatures_], inFeatures_);

            // Dot product manually using scalar accumulation
            float dot = 0.0f;
            for (uint32_t k = 0; k < inFeatures_; k++) {
                dot += xLocal(k) * wRow(k);
            }
            // Add lin_bias and bias
            float val = dot + linBiasLocal(i) + biasLocal(i);
            // hardtanh
            if (val < -1.0f) val = -1.0f;
            if (val > 1.0f) val = 1.0f;
            // mish = val * tanh(log(1+exp(val)))
            float exp_v = AscendC::Exp(val);
            float softplus = AscendC::Log(1.0f + exp_v);
            float tanh_sp = AscendC::Tanh(softplus);
            val = val * tanh_sp;
            yLocal(i) = val;
        }

        // Step 2: Group norm
        uint32_t groupSize = outFeatures_ / numGroups_;
        // Compute mean per group
        for (uint32_t g = 0; g < numGroups_; g++) {
            float sum = 0.0f;
            for (uint32_t j = 0; j < groupSize; j++) {
                sum += yLocal(g * groupSize + j);
            }
            meanLocal(g) = sum / (float)groupSize;
        }
        // Compute variance per group
        for (uint32_t g = 0; g < numGroups_; g++) {
            float sumSq = 0.0f;
            float mean = meanLocal(g);
            for (uint32_t j = 0; j < groupSize; j++) {
                float diff = yLocal(g * groupSize + j) - mean;
                sumSq += diff * diff;
            }
            varLocal(g) = sumSq / (float)groupSize;
        }
        // Normalize
        for (uint32_t i = 0; i < outFeatures_; i++) {
            uint32_t g = i / groupSize;
            float norm = (yLocal(i) - meanLocal(g)) / AscendC::Sqrt(varLocal(g) + eps_);
            yLocal(i) = norm * gammaLocal(i) + betaLocal(i);
        }

        // Write output
        AscendC::DataCopy(yGm_, yLocal, outFeatures_);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECIN> xBuf_;
    AscendC::TBuf<AscendC::TPosition::VECOUT> yBuf_;
    AscendC::TBuf<AscendC::TPosition::VECIN> wRowBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tBuf1_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tBuf2_;
    AscendC::TBuf<AscendC::TPosition::VECIN> linBiasBuf_;
    AscendC::TBuf<AscendC::TPosition::VECIN> biasBuf_;
    AscendC::TBuf<AscendC::TPosition::VECIN> gammaBuf_;
    AscendC::TBuf<AscendC::TPosition::VECIN> betaBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> meanBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> varBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t inFeatures_ {0};
    uint32_t outFeatures_ {0};
    uint32_t numGroups_ {0};
    float eps_ {1e-5f};
    uint32_t blockIdx_ {0};
    bool initialized_ {false};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
    GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.batchSize, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.numGroups, tiling_data.eps);
    op.Process();
}
