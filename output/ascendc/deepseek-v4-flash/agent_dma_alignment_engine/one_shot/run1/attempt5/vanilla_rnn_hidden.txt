#include "vanilla_rnn_hidden_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ELEM_SIZE = sizeof(float);

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    VanillaRnnHiddenCustomTilingData tiling;

    const gert::StorageShape *x_shape = context->GetInputShape(0);
    const gert::StorageShape *w_i2h_shape = context->GetInputShape(2);
    const gert::StorageShape *w_h2o_shape = context->GetInputShape(4);

    uint32_t T = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t B = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t I = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t H = static_cast<uint32_t>(w_i2h_shape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(w_i2h_shape->GetDim(1));
    uint32_t O = static_cast<uint32_t>(w_h2o_shape->GetDim(0));

    // UB budget calculation for tile sizes
    uint32_t tileH = H;
    uint32_t tileO = O;
    uint32_t tileK = K;

    while (true) {
        uint32_t needed = B*K + B*H + 2*B*tileH + K*tileH;
        if (needed * ELEM_SIZE <= UB_BUDGET_BYTES) break;
        tileH /= 2;
        if (tileH == 0) {
            tileH = 1;
            break;
        }
    }

    while (true) {
        uint32_t needed = B*H + 2*B*tileO + H*tileO;
        if (needed * ELEM_SIZE <= UB_BUDGET_BYTES) break;
        tileO /= 2;
        if (tileO == 0) {
            tileO = 1;
            break;
        }
    }

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_T(T);
    tiling.set_B(B);
    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_tileH(tileH);
    tiling.set_tileO(tileO);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = """
#include "kernel_operator.h"
#include "matmul_api.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                 GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                 uint32_t T, uint32_t B, uint32_t I, uint32_t H,
                                 uint32_t O, uint32_t K,
                                 uint32_t tileK, uint32_t tileH, uint32_t tileO) {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O; K_ = K;
        tileH_ = tileH; tileO_ = tileO;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T * B * I);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B * H);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H * K);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O * H);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O);
        yGm_.SetGlobalBuffer((__gm__ float*)y, T * B * O);

        pipe_.InitBuffer(inpQueue_, BUFFER_NUM, B_ * K_ * sizeof(float));
        pipe_.InitBuffer(hQueue_, BUFFER_NUM, B_ * H_ * sizeof(float));
        pipe_.InitBuffer(preQueue_, BUFFER_NUM, B_ * H_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, B_ * O_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, B_ * max(H_, O_) * sizeof(float));

        pipe_.InitBuffer(w_i2hUb_, 1, H_ * K_ * sizeof(float));
        pipe_.InitBuffer(b_i2hUb_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(w_h2oUb_, 1, O_ * H_ * sizeof(float));
        pipe_.InitBuffer(b_h2oUb_, 1, O_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        DataCopy(w_i2hUb_.Get<float>(), w_i2hGm_, H_ * K_);
        DataCopy(b_i2hUb_.Get<float>(), b_i2hGm_, H_);
        DataCopy(w_h2oUb_.Get<float>(), w_h2oGm_, O_ * H_);
        DataCopy(b_h2oUb_.Get<float>(), b_h2oGm_, O_);

        LocalTensor<float> hLocal = hQueue_.AllocTensor<float>();
        DataCopy(hLocal, h0Gm_, B_ * H_);
        hQueue_.EnQue(hLocal);

        for (uint32_t t = 0; t < T_; t++) {
            // Step 1: inp = cat(x[t], h)
            LocalTensor<float> inpLocal = inpQueue_.AllocTensor<float>();
            DataCopy(inpLocal, xGm_[t * B_ * I_], B_ * I_);
            LocalTensor<float> hCur = hQueue_.DeQue<float>();
            DataCopy(inpLocal[B_ * I_], hCur, B_ * H_);
            hQueue_.EnQue(hCur);
            inpQueue_.EnQue(inpLocal);

            // Step 2: pre = inp @ w_i2h.T + b_i2h
            LocalTensor<float> preLocal = preQueue_.AllocTensor<float>();
            LocalTensor<float> inpDeq = inpQueue_.DeQue<float>();
            uint32_t hOffset = 0;
            while (hOffset < H_) {
                uint32_t curTileH = min(tileH_, H_ - hOffset);
                LocalTensor<float> wSub = w_i2hUb_[hOffset * K_];
                LocalTensor<float> preTile = preLocal[hOffset * B_];
                MatMul<MatMulType<MatMulCastPolicy<float, float, float, float>, MatMulConfig<true>>>
                    matmulTile;
                matmulTile.SetReuse(true);
                matmulTile.SetBroadCast(false);
                matmulTile.SetTransA(false);
                matmulTile.SetTransB(true);
                matmulTile.SetTensorA(inpDeq, B_, K_);
                matmulTile.SetTensorB(wSub, curTileH, K_);
                matmulTile.SetTensorC(preTile, B_, curTileH);
                matmulTile.IterateAll();
                LocalTensor<float> biasTile = b_i2hUb_[hOffset];
                BroadcastAdd(preTile, preTile, biasTile, B_ * curTileH);
                hOffset += curTileH;
            }
            inpQueue_.FreeTensor(inpDeq);
            preQueue_.EnQue(preLocal);

            // Step 3: h = tanh(pre)
            LocalTensor<float> hNew = hQueue_.AllocTensor<float>();
            Tanh(hNew, preLocal, B_ * H_);
            preQueue_.FreeTensor(preLocal);
            hQueue_.EnQue(hNew);

            // Step 4: y_t = h @ w_h2o.T + b_h2o
            LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
            LocalTensor<float> hCur2 = hQueue_.DeQue<float>();
            uint32_t oOffset = 0;
            while (oOffset < O_) {
                uint32_t curTileO = min(tileO_, O_ - oOffset);
                LocalTensor<float> wSub = w_h2oUb_[oOffset * H_];
                LocalTensor<float> yTile = yLocal[oOffset * B_];
                MatMul<MatMulType<MatMulCastPolicy<float, float, float, float>, MatMulConfig<true>>>
                    matmulTile;
                matmulTile.SetReuse(true);
                matmulTile.SetBroadCast(false);
                matmulTile.SetTransA(false);
                matmulTile.SetTransB(true);
                matmulTile.SetTensorA(hCur2, B_, H_);
                matmulTile.SetTensorB(wSub, curTileO, H_);
                matmulTile.SetTensorC(yTile, B_, curTileO);
                matmulTile.IterateAll();
                LocalTensor<float> biasTile = b_h2oUb_[oOffset];
                BroadcastAdd(yTile, yTile, biasTile, B_ * curTileO);
                oOffset += curTileO;
            }
            hQueue_.EnQue(hCur2);
            yQueue_.EnQue(yLocal);

            // Step 5: Write y_t to global memory
            LocalTensor<float> yOut = yQueue_.DeQue<float>();
            DataCopy(yGm_[t * B_ * O_], yOut, B_ * O_);
            yQueue_.FreeTensor(yOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> hQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> preQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::LocalTensor<float> w_i2hUb_;
    AscendC::LocalTensor<float> b_i2hUb_;
    AscendC::LocalTensor<float> w_h2oUb_;
    AscendC::LocalTensor<float> b_h2oUb_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> w_i2hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t T_ {0};
    uint32_t B_ {0};
    uint32_t I_ {0};
    uint32_t H_ {0};
    uint32_t O_ {0};
    uint32_t K_ {0};
    uint32_t tileH_ {0};
    uint32_t tileO_ {0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                 GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                                                 GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I, tiling_data.H,
            tiling_data.O, tiling_data.K,
            tiling_data.tileK, tiling_data.tileH, tiling_data.tileO);
    op.Process();
}
