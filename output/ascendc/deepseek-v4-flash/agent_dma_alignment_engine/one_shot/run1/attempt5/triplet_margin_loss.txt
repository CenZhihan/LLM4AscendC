static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const auto* anchorShape = context->GetInputShape(0);
    const uint32_t N = anchorShape->GetOriginShape().GetDim(0);
    const uint32_t D = anchorShape->GetOriginShape().GetDim(1);

    uint32_t maxElem = (UB_BUDGET_BYTES / (3 * sizeof(float))) / 2;
    uint32_t tileBatch = AlignDown(maxElem / D, 8);
    if (tileBatch < 1) tileBatch = 1;
    if (tileBatch > N) tileBatch = N;
    tileBatch = AlignDown(tileBatch, 16);
    if (tileBatch == 0) tileBatch = 1;

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float margin = (attrs->GetAttrPointer<float>(0) != nullptr) ? *attrs->GetAttrPointer<float>(0) : 1.0f;

    tiling.set_totalBatch(N);
    tiling.set_featureDim(D);
    tiling.set_tileBatch(tileBatch);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t totalBatch, uint32_t featureDim,
                                uint32_t tileBatch, float margin)
    {
        totalBatch_ = totalBatch;
        featureDim_ = featureDim;
        tileBatch_ = tileBatch;
        margin_ = margin;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss);

        // Buffers for one tile (batch * D floats each)
        pipe_.InitBuffer(anchorQue_, BUFFER_NUM, tileBatch_ * featureDim_ * sizeof(float));
        pipe_.InitBuffer(positiveQue_, BUFFER_NUM, tileBatch_ * featureDim_ * sizeof(float));
        pipe_.InitBuffer(negativeQue_, BUFFER_NUM, tileBatch_ * featureDim_ * sizeof(float));
        // Temporary per-sample vectors
        pipe_.InitBuffer(diffQue_, 1, featureDim_ * sizeof(float));
        pipe_.InitBuffer(posRowQue_, 1, featureDim_ * sizeof(float));
        pipe_.InitBuffer(negRowQue_, 1, featureDim_ * sizeof(float));
        pipe_.InitBuffer(sumQue_, 1, sizeof(float));      // scalar after reduce
        pipe_.InitBuffer(sqrtQue_, 1, sizeof(float));     // scalar for sqrt
        // Output buffer (at least 8 floats for DataCopyPad alignment)
        pipe_.InitBuffer(outPadQue_, 1, 8 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalBatch_ == 0 || featureDim_ == 0) return;

        float totalLoss = 0.0f;
        uint32_t processed = 0;

        // DataCopyParams for all DMA operations (default 32B-aligned)
        AscendC::DataCopyParams dcParams;

        while (processed < totalBatch_) {
            uint32_t curBatch = tileBatch_;
            if (processed + curBatch > totalBatch_)
                curBatch = totalBatch_ - processed;

            // Copy tile data
            CopyIn(processed * featureDim_, curBatch);

            // Allocate tile tensors
            AscendC::LocalTensor<float> anchorTile = anchorQue_.DeQue<float>();
            AscendC::LocalTensor<float> positiveTile = positiveQue_.DeQue<float>();
            AscendC::LocalTensor<float> negativeTile = negativeQue_.DeQue<float>();

            // Loop over samples in tile
            for (uint32_t i = 0; i < curBatch; ++i) {
                uint32_t rowOff = i * featureDim_;

                // Allocate local tensors for rows
                AscendC::LocalTensor<float> diff = diffQue_.AllocTensor<float>();
                AscendC::LocalTensor<float> posRow = posRowQue_.AllocTensor<float>();
                AscendC::LocalTensor<float> negRow = negRowQue_.AllocTensor<float>();
                AscendC::LocalTensor<float> sum = sumQue_.AllocTensor<float>();
                AscendC::LocalTensor<float> sqrtT = sqrtQue_.AllocTensor<float>();

                // Copy anchor row to diff
                AscendC::DataCopy(diff, anchorGm_[processed * featureDim_ + rowOff], featureDim_, dcParams);
                // Copy positive row to posRow
                AscendC::DataCopy(posRow, positiveGm_[processed * featureDim_ + rowOff], featureDim_, dcParams);
                // posDist = sqrt(sum((anchor-positive)^2))
                AscendC::Sub(diff, diff, posRow, featureDim_);
                AscendC::Mul(diff, diff, diff, featureDim_);
                AscendC::ReduceSum(sum, diff, featureDim_, 0);
                AscendC::Sqrt(sqrtT, sum, 1);
                float posDist = sqrtT(0);

                // Refresh diff with anchor again
                AscendC::DataCopy(diff, anchorGm_[processed * featureDim_ + rowOff], featureDim_, dcParams);
                // Copy negative row to negRow
                AscendC::DataCopy(negRow, negativeGm_[processed * featureDim_ + rowOff], featureDim_, dcParams);
                // negDist = sqrt(sum((anchor-negative)^2))
                AscendC::Sub(diff, diff, negRow, featureDim_);
                AscendC::Mul(diff, diff, diff, featureDim_);
                AscendC::ReduceSum(sum, diff, featureDim_, 0);
                AscendC::Sqrt(sqrtT, sum, 1);
                float negDist = sqrtT(0);

                // loss_i = max(0, posDist - negDist + margin)
                float loss_i = posDist - negDist + margin_;
                if (loss_i < 0.0f) loss_i = 0.0f;
                totalLoss += loss_i;

                // Free per-sample tensors
                diffQue_.FreeTensor(diff);
                posRowQue_.FreeTensor(posRow);
                negRowQue_.FreeTensor(negRow);
                sumQue_.FreeTensor(sum);
                sqrtQue_.FreeTensor(sqrtT);
            }

            // Free tile buffers
            anchorQue_.FreeTensor(anchorTile);
            positiveQue_.FreeTensor(positiveTile);
            negativeQue_.FreeTensor(negativeTile);

            processed += curBatch;
        }

        // Compute mean: avoid static_cast
        float fBatch = static_cast<float>(totalBatch_);
        totalLoss = totalLoss / fBatch;

        // Write scalar result with DataCopyPad (ensuring 32B-aligned GM address)
        AscendC::LocalTensor<float> outPad = outPadQue_.AllocTensor<float>();
        outPad.SetValue(0, totalLoss);
        for (uint32_t i = 1; i < 8; ++i) {
            outPad.SetValue(i, 0.0f);
        }
        AscendC::DataCopyParams outDcParams;
        AscendC::DataCopyPad(lossGm_, outPad, outDcParams);  // writes 8 floats, only first is meaningful
        outPadQue_.FreeTensor(outPad);
    }

private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t batch)
    {
        AscendC::DataCopyParams dcParams;
        uint32_t len = batch * featureDim_;
        auto anchorLocal = anchorQue_.AllocTensor<float>();
        auto positiveLocal = positiveQue_.AllocTensor<float>();
        auto negativeLocal = negativeQue_.AllocTensor<float>();
        AscendC::DataCopy(anchorLocal, anchorGm_[offset], len, dcParams);
        AscendC::DataCopy(positiveLocal, positiveGm_[offset], len, dcParams);
        AscendC::DataCopy(negativeLocal, negativeGm_[offset], len, dcParams);
        anchorQue_.EnQue(anchorLocal);
        positiveQue_.EnQue(positiveLocal);
        negativeQue_.EnQue(negativeLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> anchorQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> positiveQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> negativeQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> diffQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> posRowQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> negRowQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sqrtQue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outPadQue_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t totalBatch_;
    uint32_t featureDim_;
    uint32_t tileBatch_;
    float margin_;
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(
    GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalBatch, tiling_data.featureDim,
            tiling_data.tileBatch, tiling_data.margin);
    op.Process();
}
