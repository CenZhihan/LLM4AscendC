#include "log_softmax_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;
static constexpr uint32_t HARD_CAP_TILE = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LogSoftmaxCustomTilingData tiling;
    auto *input_shape = context->GetInputShape(0);
    auto &origin = input_shape->GetOriginShape();
    int64_t dims = origin.GetDimNum();
    int64_t totalCols = origin.GetDim(dims - 1);
    int64_t totalRows = 1;
    for (int64_t i = 0; i < dims - 1; i++) {
        totalRows *= origin.GetDim(i);
    }
    uint32_t totalRowsU = static_cast<uint32_t>(totalRows);
    uint32_t totalColsU = static_cast<uint32_t>(totalCols);

    uint32_t maxTile = AlignDown(UB_BUDGET_BYTES / 16, ALIGN_ELEMS);
    if (maxTile > HARD_CAP_TILE) maxTile = AlignDown(HARD_CAP_TILE, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileCols = (totalColsU < maxTile) ? totalColsU : maxTile;
    tileCols = AlignDown(tileCols, ALIGN_ELEMS);
    if (tileCols == 0) tileCols = totalColsU;

    uint32_t blockDim = (totalRowsU < 24) ? totalRowsU : 24;
    blockDim = blockDim < 1 ? 1 : blockDim;
    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRowsU);
    tiling.set_totalCols(totalColsU);
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -1.0e38f;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t totalCols,
                                uint32_t tileCols, uint32_t blockDim) {
        totalRows_ = totalRows;
        totalCols_ = totalCols;
        tileCols_ = tileCols;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalRows_ * totalCols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalRows_ * totalCols_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileCols_ * sizeof(float));
        pipe_.InitBuffer(scalarQue_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalRows_ == 0 || totalCols_ == 0 || tileCols_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t rowsPerBlock = (totalRows_ + blockDim_ - 1u) / blockDim_;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        if (rowStart >= totalRows_) return;
        uint32_t rowEnd = (rowStart + rowsPerBlock < totalRows_) ? (rowStart + rowsPerBlock) : totalRows_;

        for (uint32_t r = rowStart; r < rowEnd; r++) {
            ProcessRow(r);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx) {
        uint32_t rowOffset = rowIdx * totalCols_;

        // -------- Pass 1: compute global max over the row --------
        float globalMax = NEG_INF;
        uint32_t col = 0;
        while (col < totalCols_) {
            uint32_t tileLen = (col + tileCols_ <= totalCols_) ? tileCols_ : (totalCols_ - col);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + col], tileLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> maxLocal = scalarQue_.AllocTensor<float>();
            AscendC::ReduceMax(maxLocal, xLocal, NEG_INF, tileLen);
            float localMax = maxLocal.GetValue(0);
            if (localMax > globalMax) globalMax = localMax;
            scalarQue_.FreeTensor(maxLocal);
            inQueueX_.FreeTensor(xLocal);
            col += tileLen;
        }

        // -------- Pass 2: compute sum = sum(exp(x - globalMax)) --------
        float globalSum = 0.0f;
        col = 0;
        while (col < totalCols_) {
            uint32_t tileLen = (col + tileCols_ <= totalCols_) ? tileCols_ : (totalCols_ - col);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + col], tileLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            // tLocal = x - globalMax
            AscendC::Adds(tLocal, xLocal, -globalMax, tileLen);
            // tLocal = exp(tLocal)
            AscendC::Exp(tLocal, tLocal, tileLen);

            // Reduce sum into scalar
            AscendC::LocalTensor<float> sumLocal = scalarQue_.AllocTensor<float>();
            AscendC::ReduceSum(sumLocal, tLocal, 0.0f, tileLen);
            float localSum = sumLocal.GetValue(0);
            globalSum += localSum;

            scalarQue_.FreeTensor(sumLocal);
            tmpQueue_.FreeTensor(tLocal);
            inQueueX_.FreeTensor(xLocal);
            col += tileLen;
        }

        // Compute LSE = globalMax + log(globalSum) using scalar queue
        AscendC::LocalTensor<float> sLocal = scalarQue_.AllocTensor<float>();
        sLocal.SetValue(0, globalSum);
        AscendC::Log(sLocal, sLocal, 1);
        float LSE = globalMax + sLocal.GetValue(0);
        scalarQue_.FreeTensor(sLocal);

        // -------- Pass 3: output y = x - LSE --------
        col = 0;
        while (col < totalCols_) {
            uint32_t tileLen = (col + tileCols_ <= totalCols_) ? tileCols_ : (totalCols_ - col);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset + col], tileLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Adds(yLocal, xLocal, -LSE, tileLen);
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);

            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[rowOffset + col], yLocal, tileLen);
            outQueueY_.FreeTensor(yLocal);
            col += tileLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_{0};
    uint32_t totalCols_{0};
    uint32_t tileCols_{0};
    uint32_t blockDim_{0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.totalCols, tiling_data.tileCols, tiling_data.blockDim);
    op.Process();
}
