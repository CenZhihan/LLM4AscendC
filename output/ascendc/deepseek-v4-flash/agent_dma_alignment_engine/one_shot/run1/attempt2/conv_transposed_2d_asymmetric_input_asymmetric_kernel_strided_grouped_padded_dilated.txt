#include "kernel_operator.h"

class KernelConvTransposed2dAsymmetric {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetric() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t N, uint32_t C_in, uint32_t H_in, uint32_t W_in,
                                uint32_t C_out, uint32_t H_out, uint32_t W_out,
                                uint32_t kH, uint32_t kW,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t padding_h, uint32_t padding_w,
                                uint32_t dilation_h, uint32_t dilation_w,
                                uint32_t groups)
    {
        N_ = N; C_in_ = C_in; H_in_ = H_in; W_in_ = W_in;
        C_out_ = C_out; H_out_ = H_out; W_out_ = W_out;
        kH_ = kH; kW_ = kW;
        stride_h_ = stride_h; stride_w_ = stride_w;
        padding_h_ = padding_h; padding_w_ = padding_w;
        dilation_h_ = dilation_h; dilation_w_ = dilation_w;
        groups_ = groups;
        C_out_per_grp_ = C_out_ / groups_;
        C_per_grp_ = C_in_ / groups_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_in_ * H_in_ * W_in_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, C_in_ * C_out_per_grp_ * kH_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_out_ * H_out_ * W_out_);
    }

    __aicore__ inline void Process()
    {
        uint32_t totalPixels = N_ * C_out_ * H_out_ * W_out_;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t pixelsPerBlock = (totalPixels + blockNum - 1) / blockNum;
        uint32_t startPixel = blockIdx * pixelsPerBlock;
        uint32_t endPixel = startPixel + pixelsPerBlock;
        if (endPixel > totalPixels) endPixel = totalPixels;

        for (uint32_t p = startPixel; p < endPixel; ++p) {
            // decode pixel index (n, c_out, h, w)
            uint32_t n = p / (C_out_ * H_out_ * W_out_);
            uint32_t rem1 = p % (C_out_ * H_out_ * W_out_);
            uint32_t c_out = rem1 / (H_out_ * W_out_);
            uint32_t rem2 = rem1 % (H_out_ * W_out_);
            uint32_t h_out = rem2 / W_out_;
            uint32_t w_out = rem2 % W_out_;

            float acc = 0.0f;
            uint32_t group = c_out / C_out_per_grp_;
            uint32_t c_out_rel = c_out % C_out_per_grp_;
            uint32_t c_in_start = group * C_per_grp_;

            for (uint32_t kh = 0; kh < kH_; ++kh) {
                for (uint32_t kw = 0; kw < kW_; ++kw) {
                    int32_t h_in = (h_out + padding_h_ - kh * dilation_h_);
                    int32_t w_in = (w_out + padding_w_ - kw * dilation_w_);
                    if (h_in < 0 || w_in < 0) continue;
                    if (h_in % stride_h_ != 0 || w_in % stride_w_ != 0) continue;
                    h_in /= stride_h_;
                    w_in /= stride_w_;
                    if (h_in >= (int32_t)H_in_ || w_in >= (int32_t)W_in_) continue;

                    for (uint32_t ci = 0; ci < C_per_grp_; ++ci) {
                        uint32_t c_in = c_in_start + ci;
                        // input value
                        uint32_t inpIdx = n * C_in_ * H_in_ * W_in_ + c_in * H_in_ * W_in_ + h_in * W_in_ + w_in;
                        float inp = xGm_.GetValue(inpIdx);
                        // weight value
                        uint32_t wIdx = c_in * C_out_per_grp_ * kH_ * kW_ + c_out_rel * kH_ * kW_ + kh * kW_ + kw;
                        float w = wGm_.GetValue(wIdx);
                        acc += inp * w;
                    }
                }
            }

            uint32_t outIdx = n * C_out_ * H_out_ * W_out_ + c_out * H_out_ * W_out_ + h_out * W_out_ + w_out;
            yGm_.SetValue(outIdx, acc);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_in_, H_in_, W_in_;
    uint32_t C_out_, H_out_, W_out_;
    uint32_t kH_, kW_;
    uint32_t stride_h_, stride_w_;
    uint32_t padding_h_, padding_w_;
    uint32_t dilation_h_, dilation_w_;
    uint32_t groups_;
    uint32_t C_out_per_grp_, C_per_grp_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    KernelConvTransposed2dAsymmetric op;
    op.Init(x, weight, y,
            td.N, td.C_in, td.H_in, td.W_in,
            td.C_out, td.H_out, td.W_out,
            td.kH, td.kW,
            td.stride_h, td.stride_w,
            td.padding_h, td.padding_w,
            td.dilation_h, td.dilation_w,
            td.groups);
    op.Process();
}
