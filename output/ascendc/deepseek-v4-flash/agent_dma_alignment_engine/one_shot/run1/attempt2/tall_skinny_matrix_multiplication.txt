host_operator_src = '''
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_TILE_M = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;
    const gert::Shape *shapeA = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape *shapeB = context->GetInputShape(1)->GetOriginShape();

    uint32_t M = static_cast<uint32_t>(shapeA->GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA->GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB->GetDim(1));

    constexpr float budget_ratio = 0.9f;
    uint32_t budget = static_cast<uint32_t>(budget_ratio * UB_BUDGET_BYTES);
    uint32_t denom = 2 * (K + N) * sizeof(float); // bytes per tileM row
    uint32_t tileM = (denom > 0) ? (budget - K * N * sizeof(float)) / denom : MAX_TILE_M;
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;
    if (tileM > MAX_TILE_M) tileM = AlignDown(MAX_TILE_M, ALIGN_ELEMS);
    if (tileM > M) tileM = AlignDown(M, ALIGN_ELEMS);
    if (tileM == 0) tileM = M;

    uint32_t numTiles = (tileM == 0) ? 1 : (M + tileM - 1) / tileM;
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_totalALen(M * K);
    tiling.set_totalBLen(K * N);
    tiling.set_totalCLen(M * N);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);
    int64_t M = shapeA->GetDim(0);
    int64_t N = shapeB->GetDim(1);
    gert::Shape shapeC({M, N});
    *context->GetOutputShape(0) = shapeC;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class TallSkinnyMatrixMultiplicationCustom : public OpDef {
public:
    explicit TallSkinnyMatrixMultiplicationCustom(const char* name) : OpDef(name) {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})

            .ParamType(REQUIRED)

        this->Output("C")
            .ParamType(REQUIRED)

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

kernel_src = '''
#include "kernel_operator.h"
#include "matmul.h"

constexpr int32_t DOUBLE_BUFFER = 2;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM) {
        A_gm_.SetGlobalBuffer((__gm__ float*)A);
        B_gm_.SetGlobalBuffer((__gm__ float*)B);
        C_gm_.SetGlobalBuffer((__gm__ float*)C);
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;

        // UB buffers: double buffer for A tile, single for B (constant), double for C tile
        pipe_.InitBuffer(inBufA_, DOUBLE_BUFFER, tileM_ * K_ * sizeof(float));
        pipe_.InitBuffer(bufB_, 1, K_ * N_ * sizeof(float));
        pipe_.InitBuffer(outBufC_, DOUBLE_BUFFER, tileM_ * N_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        // Load entire B matrix into UB (K and N are small)
        AscendC::LocalTensor<float> bLocal = bufB_.AllocTensor<float>();
        AscendC::DataCopy(bLocal, B_gm_, K_ * N_);
        bufB_.EnQue(bLocal);

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t numTiles = (M_ + tileM_ - 1) / tileM_;

        for (uint32_t tileId = blockIdx; tileId < numTiles; tileId += blockNum) {
            uint32_t rowStart = tileId * tileM_;
            uint32_t curTileM = (rowStart + tileM_ <= M_) ? tileM_ : (M_ - rowStart);
            if (curTileM == 0) continue;

            // Copy A tile (row-major, contiguous)
            uint32_t aOffset = rowStart * K_;
            uint32_t aLen = curTileM * K_;
            AscendC::LocalTensor<float> aLocal = inBufA_.AllocTensor<float>();
            AscendC::DataCopy(aLocal, A_gm_[aOffset], aLen);
            inBufA_.EnQue(aLocal);

            // Dequeue A and B for matmul
            AscendC::LocalTensor<float> aDeq = inBufA_.DeQue<float>();
            AscendC::LocalTensor<float> bDeq = bufB_.DeQue<float>();
            AscendC::LocalTensor<float> cLocal = outBufC_.AllocTensor<float>();

            // Use AscendC MatMul API: C = A * B
            // A: curTileM x K, row-major
            // B: K x N, row-major
            // C: curTileM x N, row-major
            AscendC::MatMul<float> mm;
            mm.SetSrcA(aDeq, AscendC::MATRIX_OP_N, curTileM, K_);
            mm.SetSrcB(bDeq, AscendC::MATRIX_OP_N, K_, N_);
            mm.SetDst(cLocal, curTileM * N_);
            mm.IterateAll();

            outBufC_.EnQue(cLocal);
            inBufA_.FreeTensor(aDeq);
            bufB_.EnQue(bDeq); // reuse B for next tile

            // Write back C tile
            AscendC::LocalTensor<float> cDeq = outBufC_.DeQue<float>();
            uint32_t cOffset = rowStart * N_;
            AscendC::DataCopy(C_gm_[cOffset], cDeq, curTileM * N_);
            outBufC_.FreeTensor(cDeq);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, DOUBLE_BUFFER> inBufA_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bufB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, DOUBLE_BUFFER> outBufC_;

    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t M_, N_, K_;
    uint32_t tileM_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                                                                GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(A, B, C, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileM);
    op.Process();
}
'''

at::Tensor tall_skinny_matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    at::Tensor C = at::empty_like(at::matmul(A, B));
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, A, B, C);
    return C;
}
