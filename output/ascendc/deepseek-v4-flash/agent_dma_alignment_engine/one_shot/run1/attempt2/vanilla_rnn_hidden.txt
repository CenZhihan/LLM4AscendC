host_operator_src="""
#include "vanilla_rnn_hidden_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ELEM_SIZE = sizeof(float);

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    VanillaRnnHiddenCustomTilingData tiling;

    auto x_shape = context->GetInputShape(0)->GetOriginShape();
    auto w_i2h_shape = context->GetInputShape(2)->GetOriginShape();
    auto w_h2o_shape = context->GetInputShape(4)->GetOriginShape();

    uint32_t T = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t B = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t I = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t H = static_cast<uint32_t>(w_i2h_shape.GetDim(0));
    uint32_t K = static_cast<uint32_t>(w_i2h_shape.GetDim(1));
    uint32_t O = static_cast<uint32_t>(w_h2o_shape.GetDim(0));

    // Calculate memory for one time step
    uint32_t inpBytes = B * K * ELEM_SIZE;
    uint32_t preBytes = B * H * ELEM_SIZE;
    uint32_t hBytes = B * H * ELEM_SIZE;
    uint32_t yBytes = B * O * ELEM_SIZE;
    uint32_t totalStepBytes = inpBytes + preBytes + hBytes + yBytes;

    // Tiling for matmuls: reduce outer dim (H for first, O for second) if needed
    uint32_t tileH = H;
    uint32_t tileO = O;
    uint32_t tileK = K;

    // First matmul: inp (B,K) @ w_i2h.T (K,H) -> pre (B,H)
    // We'll tile along H (output dim)
    uint32_t workspaceBytes = totalStepBytes + 2 * B * tileH * ELEM_SIZE; // extra for partial sums? assume minimal
    while (workspaceBytes > UB_BUDGET_BYTES && tileH > 1) {
        tileH /= 2;
        workspaceBytes = totalStepBytes + 2 * B * tileH * ELEM_SIZE;
    }

    // Second matmul: h (B,H) @ w_h2o.T (H,O) -> y (B,O)
    workspaceBytes = totalStepBytes + 2 * B * tileO * ELEM_SIZE;
    while (workspaceBytes > UB_BUDGET_BYTES && tileO > 1) {
        tileO /= 2;
        workspaceBytes = totalStepBytes + 2 * B * tileO * ELEM_SIZE;
    }

    // Use single block (sequential time steps)
    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_T(T);
    tiling.set_B(B);
    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_tileH(tileH);
    tiling.set_tileO(tileO);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"
#include "matmul_api.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                 GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                 uint32_t T, uint32_t B, uint32_t I, uint32_t H,
                                 uint32_t O, uint32_t K,
                                 uint32_t tileK, uint32_t tileH, uint32_t tileO) {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O; K_ = K;
        tileK_ = tileK; tileH_ = tileH; tileO_ = tileO;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T * B * I);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B * H);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H * K);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O * H);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O);
        yGm_.SetGlobalBuffer((__gm__ float*)y, T * B * O);

        // Allocate local buffers for one time step
        pipe_.InitBuffer(inpQueue_, BUFFER_NUM, B_ * K_ * sizeof(float));
        pipe_.InitBuffer(hQueue_, BUFFER_NUM, B_ * H_ * sizeof(float));
        pipe_.InitBuffer(preQueue_, BUFFER_NUM, B_ * H_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, B_ * O_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, B_ * H_ * sizeof(float)); // for tanh input copy
    }

    __aicore__ inline void Process() {
        // Initialize h with h0
        LocalTensor<float> hLocal = hQueue_.AllocTensor<float>();
        DataCopy(hLocal, h0Gm_, B_ * H_);
        hQueue_.EnQue(hLocal);

        // Prefetch weights and biases into UB (constant reuse)
        pipe_.InitBuffer(w_i2hLocal_, 1, H_ * K_ * sizeof(float));
        pipe_.InitBuffer(b_i2hLocal_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(w_h2oLocal_, 1, O_ * H_ * sizeof(float));
        pipe_.InitBuffer(b_h2oLocal_, 1, O_ * sizeof(float));
        DataCopy(w_i2hLocal_.Get<float>(), w_i2hGm_, H_ * K_);
        DataCopy(b_i2hLocal_.Get<float>(), b_i2hGm_, H_);
        DataCopy(w_h2oLocal_.Get<float>(), w_h2oGm_, O_ * H_);
        DataCopy(b_h2oLocal_.Get<float>(), b_h2oGm_, O_);

        for (uint32_t t = 0; t < T_; t++) {
            // Load x[t] (shape B*I) and concatenate with h (B*H) -> inp (B*K)
            LocalTensor<float> inpLocal = inpQueue_.AllocTensor<float>();
            uint32_t offset_x = t * B_ * I_;
            DataCopy(inpLocal, xGm_[offset_x], B_ * I_);
            // Copy h (already in UB) to second part of inp
            LocalTensor<float> hFrom = hQueue_.DeQue<float>();
            DataCopy(inpLocal[B_ * I_], hFrom, B_ * H_);
            // Now inpLocal contains full inp

            // First matmul: inp (B,K) * w_i2h.T (K,H) -> pre (B,H) using tiling
            LocalTensor<float> preLocal = preQueue_.AllocTensor<float>();
            MatMulOverlap(mm1_);
            mm1_.SetTensorA(inpLocal);
            mm1_.SetTensorB(w_i2hLocal_.Get<float>()); // w_i2h stored row-major (H,K), need transpose? MatMul expects (K,H) if not transposed?
            // Actually w_i2h is (H,K) in row-major. We want matmul: inp (B,K) * w_i2h.T (K,H) -> pre (B,H). So we set transB to true.
            // But to simplify, we can treat w_i2h as (K,H) by loading transposed. Instead, load w_i2h as (K,H) in host. We'll keep as (H,K) and set TransposeB.
            mm1_.SetTransposeB(true);
            mm1_.SetBias(b_i2hLocal_.Get<float>());
            mm1_.SetResult(preLocal);
            mm1_.IterateAll(tileK_, tileH_, tileK_); // M=B, N=H, K=K

            // Tanh activation
            LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            Tanh(tmp, preLocal, B_ * H_);
            // Update h (reuse hLocal queue)
            LocalTensor<float> hNew = hQueue_.AllocTensor<float>();
            DataCopy(hNew, tmp, B_ * H_); // new h = tanh(pre)
            tmpQueue_.FreeTensor(tmp);
            preQueue_.FreeTensor(preLocal); // pre no longer needed
            hQueue_.EnQue(hNew); // enqueue for next step

            // Second matmul: hNew (B,H) * w_h2o.T (H,O) -> y (B,O)
            LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
            MatMulOverlap mm2_;
            mm2_.SetTensorA(hNew);
            mm2_.SetTensorB(w_h2oLocal_.Get<float>()); // w_h2o (O,H) row-major, we need (H,O) -> set TransposeB
            mm2_.SetTransposeB(true);
            mm2_.SetBias(b_h2oLocal_.Get<float>());
            mm2_.SetResult(yLocal);
            mm2_.IterateAll(tileH_, tileO_, tileH_); // M=B, N=O, K=H

            // Store y[t] to global
            uint32_t offset_y = t * B_ * O_;
            DataCopy(yGm_[offset_y], yLocal, B_ * O_);

            // Free tensors
            inpQueue_.FreeTensor(inpLocal);
            yQueue_.FreeTensor(yLocal);
            hNew is already enqueued, will be dequeued next iteration
        }
        // Free final h
        LocalTensor<float> hFinal = hQueue_.DeQue<float>();
        hQueue_.FreeTensor(hFinal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> hQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> preQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::LocalTensor<float> w_i2hLocal_; // stored as row-major H*K
    AscendC::LocalTensor<float> b_i2hLocal_; // H
    AscendC::LocalTensor<float> w_h2oLocal_; // O*H
    AscendC::LocalTensor<float> b_h2oLocal_; // O

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> w_i2hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t T_, B_, I_, H_, O_, K_;
    uint32_t tileK_, tileH_, tileO_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I, tiling_data.H,
            tiling_data.O, tiling_data.K,
            tiling_data.tileK, tiling_data.tileH, tiling_data.tileO);
    op.Process();
}
