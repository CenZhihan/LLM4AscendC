host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MeanReductionOverADimensionCustomTilingData tiling;

    // Get the original shape from storage shape
    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &shape = storageShape->GetOriginShape();

    int32_t ndim = shape.GetDimNum();
    // Retrieve dimensions into a vector for convenience
    std::vector<int64_t> dims_vec(ndim);
    for (int32_t i = 0; i < ndim; i++) {
        dims_vec[i] = shape.GetDim(i);
    }
    const int64_t *dims = dims_vec.data();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;

    // compute outerSize, reduceSize, innerSize
    uint32_t outerSize = 1;
    for (int32_t i = 0; i < dim; i++) {
        outerSize *= dims[i];
    }
    uint32_t reduceSize = dims[dim];
    uint32_t innerSize = 1;
    for (int32_t i = dim + 1; i < ndim; i++) {
        innerSize *= dims[i];
    }

    // strides in floats:
    uint32_t reduceStride = innerSize;            // step along reduce dimension
    uint32_t outerStride = reduceSize * innerSize; // step to next outer element
    uint32_t totalInputElems = shape.GetShapeSize();
    uint32_t totalOutputElems = outerSize * innerSize;

    // UB budget: we need buffer for one reduce-size vector (reduceSize floats)
    // plus one output element (1 float) plus double buffering overhead
    uint32_t tileBytes = reduceSize * sizeof(float) * 2; // double buffer for input
    uint32_t maxOutputTile = UB_BUDGET_BYTES / tileBytes; // number of output elements processed simultaneously
    if (maxOutputTile < 1) maxOutputTile = 1;
    uint32_t tileLen = maxOutputTile;
    // cap and align
    tileLen = std::min(tileLen, totalOutputElems);
    tileLen = AlignDown(tileLen, 1); // no alignment requirement for output offset (but output is contiguous)
    if (tileLen > HARD_CAP_ELEMS) tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // blockDim: reasonable number of blocks
    uint32_t numOutTiles = CeilDiv(totalOutputElems, tileLen);
    uint32_t blockDim = std::min(numOutTiles, (uint32_t)24);
    if (blockDim < 8) blockDim = 8;

    context->SetBlockDim(blockDim);

    tiling.set_totalInputElems(totalInputElems);
    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_outerSize(outerSize);
    tiling.set_innerSize(innerSize);
    tiling.set_reduceSize(reduceSize);
    tiling.set_reduceStride(reduceStride);
    tiling.set_outerStride(outerStride);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &x_shape = storageShape->GetOriginShape();

    int32_t ndim = x_shape.GetDimNum();
    int32_t dim = context->GetAttrs()->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;

    std::vector<int64_t> out_dims;
    for (int32_t i = 0; i < ndim; i++) {
        if (i != dim) out_dims.push_back(x_shape.GetDim(i));
    }
    int32_t out_ndim = out_dims.size();

    // Modify the output shape in place
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(out_ndim);
    for (int32_t i = 0; i < out_ndim; i++) {
        y_shape->SetDim(i, out_dims[i]);
    }

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputElems, uint32_t totalOutputElems,
                                uint32_t outerSize, uint32_t innerSize,
                                uint32_t reduceSize, uint32_t reduceStride,
                                uint32_t outerStride, uint32_t tileLen,
                                int32_t dim)
    {
        totalInputElems_ = totalInputElems;
        totalOutputElems_ = totalOutputElems;
        outerSize_ = outerSize;
        innerSize_ = innerSize;
        reduceSize_ = reduceSize;
        reduceStride_ = reduceStride;
        outerStride_ = outerStride;
        tileLen_ = tileLen;
        dim_ = dim;
        invReduceSize_ = 1.0f / reduceSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputElems);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElems);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, reduceSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(tmpAccQueue_, 1, reduceSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride over output elements
        uint32_t start = blockIdx * tileLen_;
        uint32_t end = totalOutputElems_;
        if (start >= end) return;

        // we process one output element per iteration; double buffer input across output elements
        uint32_t outputIdx = start;
        // Prefetch first input vector for first output element
        prefetchInput(outputIdx);

        while (outputIdx < end) {
            // compute for current output element
            float sum = computeSum();

            // Prefetch next input if more output elements remain
            uint32_t nextOutput = outputIdx + blockNum * tileLen_;
            if (nextOutput < end) {
                prefetchInput(nextOutput);
            }

            // write output
            yGm_.SetValue(outputIdx, sum * invReduceSize_);

            if (nextOutput < end) {
                outputIdx = nextOutput;
            } else {
                break;
            }
        }
    }

private:
    __aicore__ inline void prefetchInput(uint32_t outputIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        // compute GM address for reduce-dimension elements for this output element
        // outputIdx = outerIdx * innerSize + innerIdx
        uint32_t innerIdx = outputIdx % innerSize_;
        uint32_t outerIdx = outputIdx / innerSize_;
        uint32_t baseOffset = outerIdx * outerStride_ + innerIdx; // in floats

        // Use DataCopyStride to load reduceSize elements with stride = reduceStride
        AscendC::DataCopyStride(xLocal, xGm_[baseOffset], reduceSize_, reduceStride_, sizeof(float));
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline float computeSum()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> acc = tmpAccQueue_.AllocTensor<float>();

        // ReduceSum: sum elements in xLocal to first element of acc
        AscendC::ReduceSum(acc, xLocal, reduceSize_);
        float sum = acc.GetValue(0);

        tmpAccQueue_.FreeTensor(acc);
        inQueueX_.FreeTensor(xLocal);
        return sum;
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, float val)
    {
        // No need, we write directly in Process
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_; // not used, but we keep for pattern
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpAccQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalInputElems_ {0};
    uint32_t totalOutputElems_ {0};
    uint32_t outerSize_ {1};
    uint32_t innerSize_ {1};
    uint32_t reduceSize_ {8};
    uint32_t reduceStride_ {1};
    uint32_t outerStride_ {1};
    uint32_t tileLen_ {1};
    int32_t dim_ {0};
    float invReduceSize_ {1.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalInputElems, tiling_data.totalOutputElems,
            tiling_data.outerSize, tiling_data.innerSize,
            tiling_data.reduceSize, tiling_data.reduceStride,
            tiling_data.outerStride, tiling_data.tileLen,
            tiling_data.dim);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor mean_reduction_impl_npu(const at::Tensor& self, int64_t dim) {
    auto ndim = self.dim();
    auto dim_bounded = dim < 0 ? dim + ndim : dim;
    std::vector<int64_t> out_shape;
    for (int64_t i = 0; i < ndim; i++) {
        if (i != dim_bounded) out_shape.push_back(self.size(i));
    }
    at::Tensor result = at::empty(out_shape, self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
