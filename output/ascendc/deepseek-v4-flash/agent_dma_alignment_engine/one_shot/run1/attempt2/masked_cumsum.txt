#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget for one row:
//   x row: rowLength * sizeof(float)
//   mask row: rowLength * sizeof(bool)
//   y row: rowLength * sizeof(float)
// total = rowLength * (4 + 1 + 4) = rowLength * 9 bytes
// We limit rowLength so total <= UB_BUDGET (192KB)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t MAX_ROW_ELEMS = 32768;  // fits UB: 32768*9= ~288KB? Actually 32768*9=294912 > 192KB. Let's adjust: 32768*9 = 294912 > 192KB. So we need to tile within row if rowLength > UB cap. But for simplicity we set MAX_ROW_ELEMS = 18000? Let's compute: max bytes = 9 * rowLen <= 192K => rowLen <= 21845. So we set MAX_ROW_ELEMS = 20480 (20KB elements) to stay safe.
// However, the test case has rowLength=32768, so we must tile within row.
// We'll implement two-level tiling: row tiling inside kernel. For tiling, we set tileLength = ALIGN_DOWN(UB_BUDGET_BYTES/9, ALIGN_ELEMS) as max per inner tile.
// Let's compute: max inner tile elements = UB_BUDGET_BYTES / 9 = 21845, aligned down to 20480.
// We'll set tileLength accordingly.

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    const uint32_t totalLength = static_cast<uint32_t>(x_shape->GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *p_dim = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (p_dim != nullptr) ? *p_dim : -1;

    // Determine rowLength (size of the dimension along which cumsum is performed)
    // If dim is -1 or out of range, default to last dimension.
    uint32_t ndim = x_shape->GetDimNum();
    if (dim < 0) dim = ndim - 1;
    uint32_t rowLength = static_cast<uint32_t>(x_shape->GetDim(dim));

    // Compute inner tile length (max elements that fit UB for one inner tile)
    const uint32_t bytes_per_elem = 9; // x=4, mask=1, y=4
    uint32_t max_inner = UB_BUDGET_BYTES / bytes_per_elem;
    max_inner = AlignDown(max_inner, ALIGN_ELEMS);
    if (max_inner < ALIGN_ELEMS) max_inner = ALIGN_ELEMS;
    uint32_t tileLength = (rowLength < max_inner) ? rowLength : max_inner;
    // To simplify tiling, we also ensure that tileLength divides rowLength (or we handle remainder)
    // We'll handle remainder in kernel.

    // blockDim: number of rows to process in parallel? For now set to number of rows capped.
    uint32_t numRows = (rowLength == 0) ? 1 : (totalLength / rowLength);
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t tileLength, int32_t dim)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        dim_ = dim;
        numRows_ = (rowLen_ == 0) ? 1 : (totalLen_ / rowLen_);
        // innerTileCount is for row tiling
        innerTileCount_ = (rowLen_ + tileLen_ - 1) / tileLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ bool*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(bool));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid stride over rows: each block processes multiple rows
        uint32_t rowIdx = blockIdx;
        while (rowIdx < numRows_) {
            // Process one row: compute cumulative sum along rowLen elements
            ProcessRow(rowIdx);
            rowIdx += blockNum;
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        uint32_t rowOffset = rowIdx * rowLen_;
        float state = 0.0f;

        // Process row in tiles
        for (uint32_t tile = 0; tile < innerTileCount_; ++tile) {
            uint32_t tileOffset = rowOffset + tile * tileLen_;
            uint32_t curLen = tileLen_;
            if (tile == innerTileCount_ - 1) {
                // Last tile may be smaller
                uint32_t remaining = rowLen_ - tile * tileLen_;
                curLen = (remaining > 0) ? remaining : 0;
            }
            if (curLen == 0) break;

            // Load x and mask for this tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<bool> maskLocal = inQueueMask_.AllocTensor<bool>();
            AscendC::DataCopy(xLocal, xGm_[tileOffset], curLen);
            AscendC::DataCopy(maskLocal, maskGm_[tileOffset], curLen);

            // Compute cumulative sum over tile
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            ComputeTile(xLocal, maskLocal, yLocal, curLen, state);

            // Write out y for this tile
            AscendC::DataCopy(yGm_[tileOffset], yLocal, curLen);

            // Free tensors
            inQueueX_.FreeTensor(xLocal);
            inQueueMask_.FreeTensor(maskLocal);
            outQueueY_.FreeTensor(yLocal);
        }
    }

    __aicore__ inline void ComputeTile(
        AscendC::LocalTensor<float>& x,
        AscendC::LocalTensor<bool>& mask,
        AscendC::LocalTensor<float>& y,
        uint32_t len,
        float& state)
    {
        // Scalar loop over tile elements; state is carried across tiles.
        for (uint32_t i = 0; i < len; ++i) {
            // Use DataCopy for scalar load/store? In AscendC, we can access tensors with scalar index.
            // But direct indexing on LocalTensor may not be supported; we need to use a scalar variable.
            // Workaround: use AscendC::LocalTensor::GetValue/SetValue (not standard API).
            // Instead, we can use AscendC::Duplicate for broadcast then conditional add? 
            // Simpler: use a loop with pointer arithmetic using AscendC::LocalTensor::operator[]? 
            // The canonical way is to use a for loop with scalar variable and use AscendC::LoadData/StoreData vector operations.
            // However, for simplicity and correctness, we rely on the fact that the tile is small and we can use AscendC::Gather/Scatter? 
            // Given the complexity, we implement a straightforward loop using scalar operations from local tensors via pointer casts? 
            // AscendC does not allow direct indexing. We will use a simple approach: 
            //   - Use AscendC::DataCopy to a scalar variable? Not possible.
            //   - Use AscendC::MemSet and then add? 
            //   - Use a custom vector prefix sum kernel: use two stages: first multiply x by mask (vector Muls with 0/1), then do prefix sum via a loop that uses vec_add for increasing windows.
            // We adopt a simpler approach: for each element, we use AscendC::Broadcast and compare? Not efficient.
            // The safest for correctness is to implement cumsum using a serial loop with scalar load/store via AscendC::LocalTensor::operator[] (which may be available in some versions).
            // According to AscendC documentation, LocalTensor supports indexing via operator[] for scalar access (returns reference).
            // So we can do: float x_val = x[i]; bool m_val = mask[i]; state = (m_val ? state + x_val : state); y[i] = state;
            float xVal = x(i);  // scalar access
            bool mVal = mask(i);
            if (mVal) {
                state += xVal;
            }
            y(i) = state;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<bool> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ {-1};
    uint32_t numRows_ {0};
    uint32_t innerTileCount_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.totalLength, tiling_data.rowLength, tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
