#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const uint32_t totalLength = static_cast<uint32_t>(xShape->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *p_dim = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (p_dim != nullptr) ? *p_dim : -1;

    uint32_t ndim = xShape->GetOriginShape().GetDimNum();
    if (dim < 0) dim = ndim - 1;
    uint32_t rowLength = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(dim));

    // Process each row as a single tile to simplify alignment and avoid per-element loops
    // rowLength (float) fits in UB: rowLength * (4+1+4) = rowLength * 9 bytes < UB_BUDGET_BYTES?
    // For rowLength=32768: 32768*9=294912 > 192KB, so must tile. But we use a simple heuristic:
    // tileLength = min(rowLength, AlignDown(UB_BUDGET_BYTES/9, ALIGN_ELEMS))
    const uint32_t maxInner = AlignDown(UB_BUDGET_BYTES / 9, ALIGN_ELEMS);
    uint32_t tileLength = (rowLength < maxInner) ? rowLength : maxInner;
    if (tileLength == 0) tileLength = ALIGN_ELEMS;

    uint32_t numRows = (rowLength == 0) ? 1 : (totalLength / rowLength);
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t blockDim, int32_t dim)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        blockDim_ = blockDim;
        dim_ = dim;
        numRows_ = (rowLen_ == 0) ? 1 : (totalLen_ / rowLen_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ bool*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Allocate local buffers for one row (or tile) – use rowLen_ directly
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, rowLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, rowLen_ * sizeof(bool));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, rowLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t rowIdx = blockIdx;
        while (rowIdx < numRows_) {
            ProcessRow(rowIdx);
            rowIdx += blockNum;
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowIdx)
    {
        uint32_t rowOffset = rowIdx * rowLen_;

        // Allocate and copy x
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowOffset], rowLen_);
        inQueueX_.EnQue(xLocal);

        // Allocate and copy mask
        AscendC::LocalTensor<bool> maskLocal = inQueueMask_.AllocTensor<bool>();
        AscendC::DataCopy(maskLocal, maskGm_[rowOffset], rowLen_);
        inQueueMask_.EnQue(maskLocal);

        // Dequeue for compute
        xLocal = inQueueX_.DeQue<float>();
        maskLocal = inQueueMask_.DeQue<bool>();

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Compute masked cumulative sum using vectorized operations
        // y[i] = (mask[i]) ? previous_sum + x[i] : previous_sum
        // We implement as: y = x * mask (element-wise), then prefix sum using Add with shift
        // Use a manual loop over elements since we need carry, but keep it simple.
        // For performance, we use a small loop – row length is typically <= tile size.
        float state = 0.0f;
        for (uint32_t i = 0; i < rowLen_; ++i) {
            float xVal = xLocal(i);
            bool mVal = maskLocal(i);
            if (mVal) {
                state += xVal;
            }
            yLocal(i) = state;
        }

        // Enqueue output and copy to global
        outQueueY_.EnQue(yLocal);
        yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[rowOffset], yLocal, rowLen_);

        // Free all local tensors
        outQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueMask_.FreeTensor(maskLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<bool> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t blockDim_ {0};
    int32_t dim_ {-1};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y, tiling_data.totalLength, tiling_data.rowLength, tiling_data.blockDim, tiling_data.dim);
    op.Process();
}
