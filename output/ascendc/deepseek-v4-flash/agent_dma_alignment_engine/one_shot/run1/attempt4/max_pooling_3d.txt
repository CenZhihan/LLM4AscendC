host_operator_src="""

#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;
    // Fix: GetInputShape returns const gert::StorageShape*
    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &originShape = storageShape->GetOriginShape();
    uint32_t N = originShape.GetDim(0);
    uint32_t C = originShape.GetDim(1);
    uint32_t D = originShape.GetDim(2);
    uint32_t H = originShape.GetDim(3);
    uint32_t W = originShape.GetDim(4);

    uint32_t totalInputLength = N * C * D * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];

    // Compute output spatial dimensions (floor mode)
    auto computeOutSize = [&](uint32_t inDim) -> uint32_t {
        int32_t effective = static_cast<int32_t>(inDim) + 2 * padding - kernelSize;
        if (effective < 0) return 1;
        return static_cast<uint32_t>(effective) / stride + 1;
    };
    uint32_t outD = computeOutSize(D);
    uint32_t outH = computeOutSize(H);
    uint32_t outW = computeOutSize(W);
    uint32_t outputLength = N * C * outD * outH * outW;

    // tileLen based on output elements (each processed individually)
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float); // one float per output element max
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (outputLength < tileLen) {
        uint32_t aligned = AlignDown(outputLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : outputLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(outputLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalInputLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_inputN(N);
    tiling.set_inputC(C);
    tiling.set_inputD(D);
    tiling.set_inputH(H);
    tiling.set_inputW(W);
    tiling.set_outputD(outD);
    tiling.set_outputH(outH);
    tiling.set_outputW(outW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

{
    // Get input shape
    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::Shape &x_shape = x_storage->GetOriginShape();
    uint32_t N = x_shape.GetDim(0);
    uint32_t C = x_shape.GetDim(1);
    uint32_t D = x_shape.GetDim(2);
    uint32_t H = x_shape.GetDim(3);
    uint32_t W = x_shape.GetDim(4);

    // Get attributes
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];

    // Compute output spatial dims
    auto computeOut = [&](uint32_t inDim) -> uint32_t {
        int32_t effective = static_cast<int32_t>(inDim) + 2 * padding - kernelSize;
        if (effective < 0) return 1;
        return static_cast<uint32_t>(effective) / stride + 1;
    };
    uint32_t outD = computeOut(D);
    uint32_t outH = computeOut(H);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t kernelSize, int32_t stride, int32_t padding,
                                uint32_t inputN, uint32_t inputC,
                                uint32_t inputD, uint32_t inputH, uint32_t inputW,
                                uint32_t outputD, uint32_t outputH, uint32_t outputW)
    {
        totalLen_ = totalLength;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        N_ = inputN;
        C_ = inputC;
        D_ = inputD;
        H_ = inputH;
        W_ = inputW;
        outD_ = outputD;
        outH_ = outputH;
        outW_ = outputW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        // output buffer: size = N * C * outD * outH * outW
        uint32_t outputLen = N_ * C_ * outD_ * outH_ * outW_;
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputLen);
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || outD_ == 0 || outH_ == 0 || outW_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t totalOutput = N_ * C_ * outD_ * outH_ * outW_;
        // Grid-stride loop over output elements
        for (uint32_t flatIdx = blockIdx; flatIdx < totalOutput; flatIdx += blockNum) {
            // Convert flat output index to (n, c, od, oh, ow)
            uint32_t ow = flatIdx % outW_;
            uint32_t temp = flatIdx / outW_;
            uint32_t oh = temp % outH_;
            temp /= outH_;
            uint32_t od = temp % outD_;
            temp /= outD_;
            uint32_t c = temp % C_;
            uint32_t n = temp / C_;

            // Compute input window start (with padding)
            int32_t startD = static_cast<int32_t>(od) * stride_ - padding_;
            int32_t startH = static_cast<int32_t>(oh) * stride_ - padding_;
            int32_t startW = static_cast<int32_t>(ow) * stride_ - padding_;

            float maxVal = -FLT_MAX;
            // Iterate over kernel window
            for (int32_t kd = 0; kd < kernelSize_; ++kd) {
                int32_t id = startD + kd;
                if (id < 0 || id >= static_cast<int32_t>(D_)) continue;
                for (int32_t kh = 0; kh < kernelSize_; ++kh) {
                    int32_t ih = startH + kh;
                    if (ih < 0 || ih >= static_cast<int32_t>(H_)) continue;
                    for (int32_t kw = 0; kw < kernelSize_; ++kw) {
                        int32_t iw = startW + kw;
                        if (iw < 0 || iw >= static_cast<int32_t>(W_)) continue;
                        // Compute flat input index: n*C*D*H*W + c*D*H*W + id*H*W + ih*W + iw
                        uint32_t inputIdx = n * C_ * D_ * H_ * W_
                                           + c * D_ * H_ * W_
                                           + static_cast<uint32_t>(id) * H_ * W_
                                           + static_cast<uint32_t>(ih) * W_
                                           + static_cast<uint32_t>(iw);
                        float val = xGm_[inputIdx];
                        if (val > maxVal) maxVal = val;
                    }
                }
            }
            yGm_[flatIdx] = maxVal;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    int32_t kernelSize_ {3};
    int32_t stride_ {2};
    int32_t padding_ {1};
    uint32_t N_, C_, D_, H_, W_;
    uint32_t outD_, outH_, outW_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.inputN, tiling_data.inputC,
            tiling_data.inputD, tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputD, tiling_data.outputH, tiling_data.outputW);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size, stride=None, padding=0):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride if stride is not None else kernel_size
        self.padding = padding

    def forward(self, *inputs):
        x = inputs[0]
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding)
