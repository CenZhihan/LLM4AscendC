#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_TILE_M = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape shapeA = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape shapeB = context->GetInputShape(1)->GetOriginShape();

    uint32_t M = static_cast<uint32_t>(shapeA.GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA.GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB.GetDim(1));

    constexpr float budget_ratio = 0.9f;
    uint32_t budget = static_cast<uint32_t>(budget_ratio * UB_BUDGET_BYTES);
    uint32_t denom = 2 * (K + N) * sizeof(float);
    uint32_t tileM = (denom > 0) ? (budget - K * N * sizeof(float)) / denom : MAX_TILE_M;
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;
    if (tileM > MAX_TILE_M) tileM = AlignDown(MAX_TILE_M, ALIGN_ELEMS);
    if (tileM > M) tileM = AlignDown(M, ALIGN_ELEMS);
    if (tileM == 0) tileM = M;

    uint32_t numTiles = (tileM == 0) ? 1 : (M + tileM - 1) / tileM;
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_totalALen(M * K);
    tiling.set_totalBLen(K * N);
    tiling.set_totalCLen(M * N);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    int64_t N = shapeB->GetDim(1);
    gert::Shape shapeC({M, N});

    return GRAPH_SUCCESS;

    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_N = 256;   // tile over output columns (N dimension)

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM) {
        A_gm_.SetGlobalBuffer((__gm__ float*)A);
        B_gm_.SetGlobalBuffer((__gm__ float*)B);
        C_gm_.SetGlobalBuffer((__gm__ float*)C);
        M_ = M; N_ = N; K_ = K; tileM_ = tileM;

        // UB buffer sizing:
        //   A tile: tileM * K
        //   B tile: K * TILE_N  (transposed? no, but we load row slices)
        //   C accumulation: tileM * TILE_N
        //   temporary for one row of B (we load row by row)
        pipe_.InitBuffer(aBuf_, BUFFER_NUM, tileM_ * K_ * sizeof(float));
        pipe_.InitBuffer(bRowBuf_, 1, TILE_N * sizeof(float));   // single row of B
        pipe_.InitBuffer(cBuf_, BUFFER_NUM, tileM_ * TILE_N * sizeof(float));
        pipe_.InitBuffer(partialBuf_, 1, tileM_ * TILE_N * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t numRowTiles = (M_ + tileM_ - 1) / tileM_;

        for (uint32_t rowTileId = blockIdx; rowTileId < numRowTiles; rowTileId += blockNum) {
            uint32_t rowStart = rowTileId * tileM_;
            uint32_t curTileM = (rowStart + tileM_ <= M_) ? tileM_ : (M_ - rowStart);
            if (curTileM == 0) continue;

            // Initialize accumulation buffer to zero
            AscendC::LocalTensor<float> accum = partialBuf_.AllocTensor<float>();
            AscendC::SetZero(accum, curTileM * TILE_N);

            uint32_t numColTiles = (N_ + TILE_N - 1) / TILE_N;
            for (uint32_t colTileId = 0; colTileId < numColTiles; ++colTileId) {
                uint32_t colStart = colTileId * TILE_N;
                uint32_t curTileN = (colStart + TILE_N <= N_) ? TILE_N : (N_ - colStart);
                if (curTileN == 0) continue;

                // Load A tile (curTileM x K)
                AscendC::LocalTensor<float> aLocal = aBuf_.AllocTensor<float>();
                AscendC::DataCopy(aLocal, A_gm_[rowStart * K_], curTileM * K_);
                aBuf_.EnQue(aLocal);

                // Load B tile column by column? Actually we need K rows of B, each of length curTileN.
                // B is stored as (K, N) row-major. We load each row slice using a loop.
                // We'll store the tile in a temporary buffer bTile (not queued) – for simplicity we reuse bRowBuf.
                // Since K is small, we can load each row into a separate buffer or accumulate directly.
                // Here we load one row at a time and accumulate into accum.
                AscendC::LocalTensor<float> bRow = bRowBuf_.AllocTensor<float>();
                for (uint32_t k = 0; k < K_; ++k) {
                    uint32_t srcOff = k * N_ + colStart;
                    AscendC::DataCopy(bRow, B_gm_[srcOff], curTileN);
                    // Now we have the k-th row of B tile (size curTileN).
                    // For each i in 0..curTileM-1, we need to multiply A[i][k] by bRow and add to accum[i * TILE_N : i * TILE_N + curTileN]
                    // We'll do this with a vector scale and add.
                    // Use AscendC::Muls(bRow, bRow, aCol[k], curTileN) where aCol[k] is the k-th element of the i-th row of A.
                    // But A is row-major, so A[i][k] = aDeq[i * K_ + k]
                    // We'll dequeue A tile first.
                    AscendC::LocalTensor<float> aDeq = aBuf_.DeQue<float>();
                    // For each row i, scale bRow and add to accum
                    for (uint32_t i = 0; i < curTileM; ++i) {
                        float a_ik = aDeq[i * K_ + k];
                        // scale bRow by a_ik and add to accum row i
                        // We'll use a temporary scaled vector
                        AscendC::LocalTensor<float> scaled = bRowBuf_.AllocTensor<float>(); // reuse? careful
                        // For simplicity, we use a small local array
                        AscendC::Muls(scaled, bRow, a_ik, curTileN);
                        // add to accum at row i
                        uint32_t rowOff = i * TILE_N;
                        AscendC::Add(accum[rowOff], accum[rowOff], scaled, curTileN);
                    }
                    aBuf_.FreeTensor(aDeq); // free and re-allocate next iteration
                }
                bRowBuf_.FreeTensor(bRow);

                // After finishing K loops, accumulation for this column tile is done.
                // Write back C tile
                AscendC::LocalTensor<float> cLocal = cBuf_.AllocTensor<float>();
                // Copy accum to cLocal (maybe trim curTileN)
                AscendC::DataCopy(cLocal, accum, curTileM * TILE_N);
                // Then copy from UB to GM
                for (uint32_t i = 0; i < curTileM; ++i) {
                    uint32_t cDstOff = (rowStart + i) * N_ + colStart;
                    AscendC::DataCopy(C_gm_[cDstOff], cLocal[i * TILE_N], curTileN);
                }
                cBuf_.FreeTensor(cLocal);
            }
            partialBuf_.FreeTensor(accum);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bRowBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> cBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> partialBuf_;

    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t M_, N_, K_, tileM_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                                                                GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(A, B, C, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileM);
    op.Process();
}
'''
