host_operator_src="""
#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    const int32_t ndim = static_cast<int32_t>(x_shape.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim_ptr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dim_ptr;
    if (dim < 0) dim += ndim;

    uint32_t before = 1, rowLen = 1, after = 1;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i == dim) {
            rowLen = x_shape.GetDim(i);
        } else if (i < dim) {
            before *= x_shape.GetDim(i);
        } else {
            after *= x_shape.GetDim(i);
        }
    }

    uint32_t totalBatches = before * after;
    uint32_t colLen = after;

    // UB budget: 3 float buffers (maxVals, tmpRow, tmpVar) + 1 int64 buffer (maxIdxs) + 1 uint8 mask
    // bytes per column element: 3*4 + 8 + 1 = 21 bytes, round up to 24 for alignment
    uint32_t maxTile = UB_BUDGET_BYTES / 24;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > colLen) maxTile = colLen;

    uint32_t tileLen = maxTile;
    if (tileLen == 0) tileLen = colLen;

    // Number of column tiles per batch
    uint32_t numColTiles = CeilDiv(colLen, tileLen);
    uint32_t numBlocks = totalBatches * numColTiles;
    if (numBlocks < 8) numBlocks = 8;
    if (numBlocks > 24) numBlocks = 24;

    context->SetBlockDim(numBlocks);

    tiling.set_totalBatches(totalBatches);
    tiling.set_rowLen(rowLen);
    tiling.set_colLen(colLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(numBlocks);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const int32_t ndim = static_cast<int32_t>(x_shape->GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim_ptr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dim_ptr;
    if (dim < 0) dim += ndim;

    std::vector<int64_t> out_dims;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i == dim) continue;
        out_dims.push_back(x_shape->GetDim(i));
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(static_cast<uint32_t>(out_dims.size()));
    for (uint32_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalBatches, uint32_t rowLen,
                                uint32_t colLen, uint32_t tileLen,
                                uint32_t blockDim, int32_t dim)
    {
        totalBatches_ = totalBatches;
        rowLen_ = rowLen;
        colLen_ = colLen;
        tileLen_ = tileLen;
        blockDim_ = blockDim;
        dim_ = dim;

        // Map block id to (batch, colTile)
        uint32_t numColTiles = (colLen_ + tileLen_ - 1) / tileLen_;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        batchStart_ = blockIdx / numColTiles;
        colTileIdx_ = blockIdx % numColTiles;

        colStart_ = colTileIdx_ * tileLen_;
        colEnd_ = colStart_ + tileLen_;
        if (colEnd_ > colLen_) colEnd_ = colLen_;
        tileLenActual_ = colEnd_ - colStart_;

        // Initialize global tensors
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y);

        // Allocate local buffers: maxVals (float), maxIdxs (int64), tmpRow (float), mask (uint8)
        pipe_.InitBuffer(inQueueMaxVals_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMaxIdxs_, BUFFER_NUM, tileLen_ * sizeof(int64_t));
        pipe_.InitBuffer(inQueueTmpRow_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(uint8_t));
    }

    __aicore__ inline void Process()
    {
        if (rowLen_ == 0 || colLen_ == 0 || tileLenActual_ == 0) return;
        if (batchStart_ >= totalBatches_) return;

        auto maxVals = inQueueMaxVals_.AllocTensor<float>();
        auto maxIdxs = inQueueMaxIdxs_.AllocTensor<int64_t>();
        auto tmpRow = inQueueTmpRow_.AllocTensor<float>();
        auto mask = inQueueMask_.AllocTensor<uint8_t>();

        // Copy first row's tile into maxVals, set indices to 0
        uint32_t baseIdx = batchStart_ * rowLen_ * colLen_ + colStart_;
        AscendC::DataCopy(maxVals, xGm_[baseIdx], tileLenActual_);
        AscendC::SetValue(maxIdxs, static_cast<int64_t>(0), tileLenActual_);

        // Loop over remaining rows
        for (uint32_t r = 1; r < rowLen_; ++r) {
            uint32_t rowBase = batchStart_ * rowLen_ * colLen_ + r * colLen_ + colStart_;
            AscendC::DataCopy(tmpRow, xGm_[rowBase], tileLenActual_);

            // Compare: where tmpRow > maxVals, mask = 1
            AscendC::Compare(mask, tmpRow, maxVals, AscendC::CMPMODE::GT, tileLenActual_);

            // Update maxVals where mask == 1
            AscendC::Select(maxVals, tmpRow, maxVals, mask, tileLenActual_);

            // Create a local int64 tensor filled with current row index r
            AscendC::LocalTensor<int64_t> idxVals;
            idxVals = pipe_.AllocLocalTensor<int64_t>(tileLenActual_);
            AscendC::SetValue(idxVals, static_cast<int64_t>(r), tileLenActual_);

            // Update maxIdxs where mask == 1
            AscendC::Select(maxIdxs, idxVals, maxIdxs, mask, tileLenActual_);

            pipe_.FreeTensor(idxVals);
        }

        // Write result: output index = batchStart_ * colLen_ + colStart_
        uint32_t outBase = batchStart_ * colLen_ + colStart_;
        AscendC::DataCopy(yGm_[outBase], maxIdxs, tileLenActual_);

        inQueueMaxVals_.FreeTensor(maxVals);
        inQueueMaxIdxs_.FreeTensor(maxIdxs);
        inQueueTmpRow_.FreeTensor(tmpRow);
        inQueueMask_.FreeTensor(mask);
    }

private:
    AscendC::TPipe pipe_;
    // Buffers (used as persistent local tensors)
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMaxVals_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMaxIdxs_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueTmpRow_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalBatches_;
    uint32_t rowLen_;
    uint32_t colLen_;
    uint32_t tileLen_;
    uint32_t blockDim_;
    int32_t dim_;
    uint32_t batchStart_;
    uint32_t colTileIdx_;
    uint32_t colStart_;
    uint32_t colEnd_;
    uint32_t tileLenActual_;
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalBatches, tiling_data.rowLen,
            tiling_data.colLen, tiling_data.tileLen,
            tiling_data.blockDim, tiling_data.dim);
    op.Process();
}
