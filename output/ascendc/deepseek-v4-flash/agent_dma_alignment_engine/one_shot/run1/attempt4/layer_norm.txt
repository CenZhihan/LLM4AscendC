#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    LayerNormCustomTilingData tiling;
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    const gert::Shape &originShape = inputShape->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(originShape.GetShapeSize());

    int ndim = originShape.GetDimNum();
    uint32_t firstDim = static_cast<uint32_t>(originShape.GetDim(0));
    uint32_t innerSize = (firstDim == 0) ? totalLength : (totalLength / firstDim);

    float eps = 1e-5f;

    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (innerSize < tileLen) {
        tileLen = AlignDown(innerSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = innerSize;
    }

    uint32_t numRows = totalLength / innerSize;
    uint32_t blockDim = numRows;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_innerSize(innerSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t innerSize,
                                uint32_t tileLength, float eps)
    {
        totalLen_ = totalLength;
        innerSize_ = innerSize;
        invInnerSize_ = 1.0f / static_cast<float>(innerSize_);
        tileLen_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerSize_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t numRows = totalLen_ / innerSize_;
        for (uint32_t row = blockIdx; row < numRows; row += blockNum) {
            uint32_t rowOffset = row * innerSize_;
            // First pass: compute sum and sumSq over all tiles of this row
            float rowSum = 0.0f;
            float rowSumSq = 0.0f;
            uint32_t remaining = innerSize_;
            uint32_t tileOffset = rowOffset;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[tileOffset], curLen);
                for (uint32_t i = 0; i < curLen; i++) {
                    float val = xLocal(i);
                    rowSum += val;
                    rowSumSq += val * val;
                }
                inQueueX_.FreeTensor(xLocal);
                tileOffset += curLen;
                remaining -= curLen;
            }

            float mean = rowSum * invInnerSize_;
            float variance = rowSumSq * invInnerSize_ - mean * mean;
            if (variance < 0.0f) variance = 0.0f;
            float invStd = 1.0f / sqrt(variance + eps_);

            // Second pass: normalize and store
            remaining = innerSize_;
            tileOffset = rowOffset;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[tileOffset], curLen);
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                // y = x - mean  (using Add with negative scalar)
                AscendC::Add(yLocal, xLocal, -mean, curLen);
                AscendC::Muls(yLocal, yLocal, invStd, curLen);
                AscendC::DataCopy(yGm_[tileOffset], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
                tileOffset += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t innerSize_ {0};
    float invInnerSize_ {0.0f};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.innerSize, tiling_data.tileLength, tiling_data.eps);
    op.Process();
}
