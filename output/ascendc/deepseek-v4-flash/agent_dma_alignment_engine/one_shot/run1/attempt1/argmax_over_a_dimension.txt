#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: 192KB
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// Per-element storage in UB for one row chunk: max (float), idx (int64), row (float) => 16 bytes
// Plus overhead for other local arrays; use conservative max elements = UB_BUDGET_BYTES / 20
static constexpr uint32_t MAX_ELEMS_PER_BLOCK = 12288;  // safe estimate

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0)->GetOriginShape();
    const int32_t ndim = x_shape->GetDimNum();
    const int32_t *dims = x_shape->GetDims();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim_ptr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dim_ptr;
    if (dim < 0) dim += ndim;

    // Compute sizes along dim and total batches (product of other dims)
    uint32_t rowLen = 1, colLen = 1, totalBatches = 1;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i == dim) {
            rowLen = dims[i];
        } else {
            totalBatches *= dims[i];
        }
    }
    // colLen is product of dims after dim? Actually after reduction, we keep other dims.
    // For dim=1 example shape (B,R,C): output (B,C). colLen = C, totalBatches = B.
    // Handle generic: dim splits shape into prefix (batch dims) and suffix (col dims).
    // Assume simple case: 2D or 3D where dim=1 -> rowLen = dims[1], colLen = dims[2], totalBatches = dims[0].
    // For simplicity we only support 3D with dim=1 (as example). 
    // More general tiling would need proper handling.
    // We'll compute colLen as product of dims after dim, totalBatches as product of dims before dim and after col.
    // Given example dim=1:
    uint32_t before = 1, after = 1;
    for (int32_t i = 0; i < dim; ++i) before *= dims[i];
    for (int32_t i = dim + 1; i < ndim; ++i) after *= dims[i];
    totalBatches = before * after;
    rowLen = dims[dim];
    colLen = (ndim > dim + 1) ? after : 1;  // for 2D case colLen=1

    // Tiling: process one batch per tile (since each batch requires all rows for that col set)
    // We tile totalBatches into blocks, each block handles multiple batches via grid stride.
    uint32_t tileBatch = 1;  // one batch per tile
    uint32_t numTiles = (totalBatches + tileBatch - 1) / tileBatch;

    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalBatches(totalBatches);
    tiling.set_rowLen(rowLen);
    tiling.set_colLen(colLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverDimension {
public:
    __aicore__ inline KernelArgmaxOverDimension() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalBatches, uint32_t rowLen, uint32_t colLen,
                                int32_t dim)
    {
        totalBatches_ = totalBatches;
        rowLen_ = rowLen;
        colLen_ = colLen;
        dim_ = dim;
        // Stride in GM: each row has colLen floats, each batch has rowLen rows.
        // Input: x shape (B, R, C) or generic; we assume contiguous layout.
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalBatches_ * rowLen_ * colLen_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalBatches_ * colLen_);

        // UB allocation: for one batch, we need:
        //   maxLocal: colLen float  (4 * colLen)
        //   idxLocal: colLen int64  (8 * colLen)
        //   rowChunk: 1*colLen float (4 * colLen) but we process in chunks of 8 floats
        //   plus other small buffers.
        // Total: (4+8+4)*colLen = 16*colLen. With colLen=4095, ~65KB, within 192KB.
        // We also need temporary buffers for masks and duplication.
        // Use double buffering only for chunks? Not necessary.
        // We'll allocate maxLocal and idxLocal as single tensors.
        // rowChunk will be allocated per chunk.
        const uint32_t batchElem = colLen_;
        pipe_.InitBuffer(inQueueRow_, BUFFER_NUM, 32 * sizeof(float)); // 8 floats per chunk
        pipe_.InitBuffer(outQueueIdx_, 1, batchElem * sizeof(int64_t));
        pipe_.InitBuffer(tmpMaxBuf_, 1, batchElem * sizeof(float));
        pipe_.InitBuffer(tmpRowBuf_, 1, 8 * sizeof(float)); // for row chunk
        pipe_.InitBuffer(tmpMaskBuf_, 1, 8 * sizeof(uint8_t));
        pipe_.InitBuffer(tmpDupBuf_, 1, 8 * sizeof(int64_t));
    }

    __aicore__ inline void Process()
    {
        if (totalBatches_ == 0 || rowLen_ == 0 || colLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid stride over batches
        const uint32_t strideBatch = blockNum;
        uint32_t batchBase = blockIdx;
        while (batchBase < totalBatches_) {
            ProcessBatch(batchBase);
            batchBase += strideBatch;
        }
    }

private:
    __aicore__ inline void ProcessBatch(uint32_t batchIdx)
    {
        // Allocate local tensors for column-wide max and index
        AscendC::LocalTensor<float> maxLocal = tmpMaxBuf_.AllocTensor<float>();
        AscendC::LocalTensor<int64_t> idxLocal = outQueueIdx_.AllocTensor<int64_t>();

        // Initialize with first row
        ProcessFirstRow(batchIdx, maxLocal, idxLocal);

        // Process remaining rows
        for (uint32_t r = 1; r < rowLen_; ++r) {
            ProcessRow(batchIdx, r, maxLocal, idxLocal);
        }

        // Write out final indices for this batch
        WriteBatchIndices(batchIdx, idxLocal);

        // Free tensors
        tmpMaxBuf_.FreeTensor(maxLocal);
        outQueueIdx_.FreeTensor(idxLocal);
    }

    __aicore__ inline void ProcessFirstRow(uint32_t batchIdx,
                                           AscendC::LocalTensor<float>& maxLocal,
                                           AscendC::LocalTensor<int64_t>& idxLocal)
    {
        // Load row and set max = row, idx = 0
        // Process in chunks of 8 floats for alignment
        uint32_t offset = 0;
        uint32_t remaining = colLen_;
        while (remaining > 0) {
            uint32_t chunkSize = (remaining >= 8) ? 8 : remaining;
            uint32_t padSize = (chunkSize < 8) ? (8 - chunkSize) : 0;

            AscendC::LocalTensor<float> rowChunk = inQueueRow_.AllocTensor<float>();
            // Load from GM: row at (batchIdx * rowLen * colLen + r * colLen + offset)
            uint32_t gmOffset = batchIdx * rowLen_ * colLen_ + 0 * colLen_ + offset;
            if (chunkSize == 8) {
                AscendC::DataCopy(rowChunk, xGm_[gmOffset], chunkSize);
            } else {
                // Use DataCopyPad for tail
                AscendC::DataCopyPad(rowChunk, xGm_[gmOffset], chunkSize, padSize);
            }

            // Copy rowChunk to maxLocal at offset
            AscendC::DataCopy(maxLocal[offset], rowChunk, chunkSize);

            // Set indices to 0 for this chunk
            AscendC::LocalTensor<int64_t> idxChunk = tmpDupBuf_.AllocTensor<int64_t>();
            AscendC::Duplicate(idxChunk, static_cast<int64_t>(0), chunkSize);
            AscendC::DataCopy(idxLocal[offset], idxChunk, chunkSize);
            tmpDupBuf_.FreeTensor(idxChunk);

            // Free rowChunk
            inQueueRow_.FreeTensor(rowChunk);

            offset += chunkSize;
            remaining -= chunkSize;
        }
    }

    __aicore__ inline void ProcessRow(uint32_t batchIdx, uint32_t r,
                                      AscendC::LocalTensor<float>& maxLocal,
                                      AscendC::LocalTensor<int64_t>& idxLocal)
    {
        uint32_t offset = 0;
        uint32_t remaining = colLen_;
        while (remaining > 0) {
            uint32_t chunkSize = (remaining >= 8) ? 8 : remaining;
            uint32_t padSize = (chunkSize < 8) ? (8 - chunkSize) : 0;

            AscendC::LocalTensor<float> rowChunk = inQueueRow_.AllocTensor<float>();
            uint32_t gmOffset = batchIdx * rowLen_ * colLen_ + r * colLen_ + offset;
            if (chunkSize == 8) {
                AscendC::DataCopy(rowChunk, xGm_[gmOffset], chunkSize);
            } else {
                AscendC::DataCopyPad(rowChunk, xGm_[gmOffset], chunkSize, padSize);
            }

            // Compare rowChunk > maxLocal[offset] -> mask
            AscendC::LocalTensor<uint8_t> mask = tmpMaskBuf_.AllocTensor<uint8_t>();
            AscendC::Compare(mask, rowChunk, maxLocal[offset], AscendC::CMPMODE::GT, chunkSize);

            // Update maxLocal where mask true
            AscendC::Select(maxLocal[offset], rowChunk, maxLocal[offset], mask, AscendC::SEL_MODE::VSEL_CONDITIONAL, chunkSize);

            // Update idxLocal where mask true
            AscendC::LocalTensor<int64_t> idxDup = tmpDupBuf_.AllocTensor<int64_t>();
            AscendC::Duplicate(idxDup, static_cast<int64_t>(r), chunkSize);
            AscendC::Select(idxLocal[offset], idxDup, idxLocal[offset], mask, AscendC::SEL_MODE::VSEL_CONDITIONAL, chunkSize);
            tmpDupBuf_.FreeTensor(idxDup);

            // Free tensors
            tmpMaskBuf_.FreeTensor(mask);
            inQueueRow_.FreeTensor(rowChunk);

            offset += chunkSize;
            remaining -= chunkSize;
        }
    }

    __aicore__ inline void WriteBatchIndices(uint32_t batchIdx,
                                             AscendC::LocalTensor<int64_t>& idxLocal)
    {
        // Write indices to GM for this batch
        // Process in chunks of 4 int64 (32 bytes) for alignment
        uint32_t offset = 0;
        uint32_t remaining = colLen_;
        while (remaining > 0) {
            uint32_t chunkSize = (remaining >= 4) ? 4 : remaining;
            uint32_t padSize = (chunkSize < 4) ? (4 - chunkSize) : 0;

            uint32_t gmOffset = batchIdx * colLen_ + offset;
            if (chunkSize == 4) {
                AscendC::DataCopy(yGm_[gmOffset], idxLocal[offset], chunkSize);
            } else {
                // Use DataCopyPad for last chunk
                AscendC::LocalTensor<int64_t> tmp = inQueueRow_.AllocTensor<int64_t>(); // reuse row queue
                AscendC::DataCopy(tmp, idxLocal[offset], chunkSize);
                AscendC::DataCopyPad(yGm_[gmOffset], tmp, chunkSize, padSize);
                inQueueRow_.FreeTensor(tmp);
            }

            offset += chunkSize;
            remaining -= chunkSize;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueRow_;  // for row chunks (float)
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueIdx_;         // for final idx (int64)
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMaxBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpRowBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMaskBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDupBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalBatches_ {0};
    uint32_t rowLen_ {0};
    uint32_t colLen_ {0};
    int32_t dim_ {1};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverDimension op;
    op.Init(x, y,
            tiling_data.totalBatches, tiling_data.rowLen, tiling_data.colLen,
            tiling_data.dim);
    op.Process();
}
