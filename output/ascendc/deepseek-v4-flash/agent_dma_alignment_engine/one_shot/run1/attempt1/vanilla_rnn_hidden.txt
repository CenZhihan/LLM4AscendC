#include "vanilla_rnn_hidden_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: 192 KB shared. Estimate temporary buffers per time step:
//   inp (B*K), pre (B*H), h (B*H), y (B*O), matmul workspace.
// We set tile parameters conservatively.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ELEM_SIZE = sizeof(float);

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    VanillaRnnHiddenCustomTilingData tiling;

    // Retrieve shapes from inputs
    auto x_shape = context->GetInputShape(0)->GetOriginShape();
    auto w_i2h_shape = context->GetInputShape(2)->GetOriginShape();
    auto w_h2o_shape = context->GetInputShape(4)->GetOriginShape();

    uint32_t T = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t B = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t I = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t H = static_cast<uint32_t>(w_i2h_shape.GetDim(0));
    uint32_t K = static_cast<uint32_t>(w_i2h_shape.GetDim(1));
    uint32_t O = static_cast<uint32_t>(w_h2o_shape.GetDim(0));

    // Simple tiling for matmuls: tile based on UB budget for one step.
    // We allocate: inp = B*K floats, pre = B*H, h = B*H, y = B*O, temp for matmul.
    // Also allow room for matmul pipeline.
    uint32_t inpBytes = B * K * ELEM_SIZE;
    uint32_t preBytes = B * H * ELEM_SIZE;
    uint32_t hBytes = B * H * ELEM_SIZE;
    uint32_t yBytes = B * O * ELEM_SIZE;
    uint32_t totalStepBytes = inpBytes + preBytes + hBytes + yBytes;

    // Determine tile size for reduction dimension (K and H) by attempting to limit workspace
    uint32_t tileK = K;          // promote full K if possible
    uint32_t tileH = H;          // full H for matmul1
    uint32_t tileO = O;          // full O for matmul2

    // If too large, reduce tileH first (matmul1 outer)
    uint32_t workspaceBytes = totalStepBytes + 2 * B * tileH * ELEM_SIZE;  // rough estimate
    while (workspaceBytes > UB_BUDGET_BYTES && tileH > 1) {
        tileH /= 2;
        workspaceBytes = totalStepBytes + 2 * B * tileH * ELEM_SIZE;
    }
    // Similarly for tileO
    workspaceBytes = totalStepBytes + 2 * B * tileO * ELEM_SIZE;
    while (workspaceBytes > UB_BUDGET_BYTES && tileO > 1) {
        tileO /= 2;
        workspaceBytes = totalStepBytes + 2 * B * tileO * ELEM_SIZE;
    }

    // For simplicity, single block (sequential across time)
    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_T(T);
    tiling.set_B(B);
    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_tileH(tileH);
    tiling.set_tileO(tileO);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    auto x_shape = context->GetInputShape(0)->GetOriginShape();
    auto w_h2o_shape = context->GetInputShape(4)->GetOriginShape();
    // Output shape: [T, B, O]
    gert::Shape y_shape = x_shape;
    y_shape.SetDim(2, static_cast<int64_t>(w_h2o_shape.GetDim(0)));
    *context->GetOutputShape(0) = y_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType dt = context->GetInputDataType(0);
    context->SetOutputDataType(0, dt);
    return GRAPH_SUCCESS;
}
} // namespace ge
