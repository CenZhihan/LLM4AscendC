kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t innerSize,
                                uint32_t tileLength, float eps)
    {
        totalLen_ = totalLength;
        innerSize_ = innerSize;
        tileLen_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // Temporary buffer for sum/sumsq accumulation across tiles (scalars kept in registers)
        // Not needed as queue.
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerSize_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Number of rows
        uint32_t numRows = totalLen_ / innerSize_;
        // Grid-stride over rows
        for (uint32_t row = blockIdx; row < numRows; row += blockNum) {
            // Compute row baseline offsets
            uint32_t rowOffset = row * innerSize_;
            // First pass: compute sum and sumSq over all tiles of this row
            float rowSum = 0.0f;
            float rowSumSq = 0.0f;
            uint32_t remaining = innerSize_;
            uint32_t tileOffset = rowOffset;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                // Load tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[tileOffset], curLen);
                inQueueX_.EnQue(xLocal);
                // Compute partial sum and sumSq using vector ReduceAdd
                // We can use AscendC::ReduceAdd to get scalar sum? Not directly.
                // Instead we do elementwise operations and then reduce via scalar loop or built-in reduction.
                // For efficiency, we can compute sum and sumSq using vector operations and then reduce to scalar.
                // Simpler: iterate over elements in the tile (small tile size) to compute sum and sumSq.
                // But that defeats parallelism. We can use AscendC::Sums and SumsSq?
                // Actually AscendC has VecSum which returns a scalar. We'll use that.
                // However, VecSum is not exposed in kernel_operator.h? It is in vec_reduce.h.
                // For brevity, we do a simple loop for each tile (acceptable if tileLen moderate).
                // To demonstrate vectorization, we'll use a reduction function: AscendC::LocalTensor<float> tmp = ...;
                // We'll use AscendC::VecSum which returns a float. Let's assume it exists: AscendC::VecSum(xLocal, curLen);
                // In actual code, you'd include "reduce_operator.h" etc. For this example, we write a scalar loop.
                float sumTile = 0.0f;
                float sumSqTile = 0.0f;
                // Dequeue tensor for reading
                AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
                for (uint32_t i = 0; i < curLen; i++) {
                    float val = xDeq(i);
                    sumTile += val;
                    sumSqTile += val * val;
                }
                inQueueX_.FreeTensor(xDeq);
                rowSum += sumTile;
                rowSumSq += sumSqTile;
                tileOffset += curLen;
                remaining -= curLen;
            }

            // Compute mean and invStd
            float mean = rowSum / (float)innerSize_;
            float variance = rowSumSq / (float)innerSize_ - mean * mean;
            // Clamp variance to avoid negative due to numerical error
            if (variance < 0.0f) variance = 0.0f;
            float invStd = 1.0f / sqrtf(variance + eps_);

            // Second pass: normalize each tile and store to output
            remaining = innerSize_;
            tileOffset = rowOffset;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                // Load tile
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[tileOffset], curLen);
                inQueueX_.EnQue(xLocal);
                // Dequeue and compute normalized
                AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                // y = (x - mean) * invStd
                AscendC::Subs(yLocal, xDeq, mean, curLen);
                AscendC::Muls(yLocal, yLocal, invStd, curLen);
                // Store to output
                AscendC::DataCopy(yGm_[tileOffset], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xDeq);
                tileOffset += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t innerSize_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.innerSize, tiling_data.tileLength, tiling_data.eps);
    op.Process();
}
