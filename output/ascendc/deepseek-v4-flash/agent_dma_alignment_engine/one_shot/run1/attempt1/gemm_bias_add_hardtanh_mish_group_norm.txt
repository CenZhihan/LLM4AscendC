kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
                                GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t totalBatch, uint32_t outDim, uint32_t inDim,
                                uint32_t numGroups, float eps) {
        totalBatch_ = totalBatch;
        outDim_ = outDim;
        inDim_ = inDim;
        numGroups_ = numGroups;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalBatch_ * inDim_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outDim_ * inDim_);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outDim_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outDim_);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outDim_);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outDim_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalBatch_ * outDim_);

        // UB buffers
        // Maximum size: xLocal (inDim), weightCol (inDim), output row (outDim), misc.
        // All buffers allocated at Init time
        pipe_.InitBuffer(xBuf_, BUFFER_NUM, inDim_ * sizeof(float));
        pipe_.InitBuffer(weightColBuf_, 1, inDim_ * sizeof(float));
        pipe_.InitBuffer(outputBuf_, 1, outDim_ * sizeof(float));
        // Temp for dot product, softplus, tanh, group norm stats
        pipe_.InitBuffer(tmpBuf_, 1, (inDim_ > outDim_ ? inDim_ : outDim_) * sizeof(float));
        pipe_.InitBuffer(gnMeanBuf_, 1, numGroups_ * sizeof(float));
        pipe_.InitBuffer(gnVarBuf_, 1, numGroups_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalBatch_ == 0 || outDim_ == 0 || inDim_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Row-level grid stride: each core processes rows blockIdx, blockIdx+blockNum, ...
        for (uint32_t r = blockIdx; r < totalBatch_; r += blockNum) {
            ProcessRow(r);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row) {
        // Step 1: Load x row into xBuf
        AscendC::LocalTensor<float> xLocal = xBuf_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[row * inDim_], inDim_);

        // Step 2: Allocate output row buffer
        AscendC::LocalTensor<float> outLocal = outputBuf_.AllocTensor<float>();

        // Step 3: For each output column, compute dot product
        for (uint32_t j = 0; j < outDim_; ++j) {
            // Load weight column j
            AscendC::LocalTensor<float> wCol = weightColBuf_.AllocTensor<float>();
            AscendC::DataCopy(wCol, weightGm_[j * inDim_], inDim_);

            // Compute dot product: tmp = x * wCol, then reduce
            AscendC::LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>();
            AscendC::Mul(tmp, xLocal, wCol, inDim_);
            float dotVal = 0.0f;
            AscendC::ReduceAdd(dotVal, tmp, inDim_); // scalar

            // Add linear bias
            float linBiasVal = linBiasGm_.GetValue(j);
            outLocal.SetValue(j, dotVal + linBiasVal);

            // Free weight column and tmp
            weightColBuf_.FreeTensor(wCol);
            tmpBuf_.FreeTensor(tmp);
        }

        // Free xLocal (we are done with x)
        xBuf_.FreeTensor(xLocal);

        // Step 4: Add bias
        for (uint32_t j = 0; j < outDim_; ++j) {
            float b = biasGm_.GetValue(j);
            float val = outLocal.GetValue(j);
            outLocal.SetValue(j, val + b);
        }

        // Step 5: Hardtanh (clamp to [-1, 1])
        AscendC::LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>(); // reuse tmp
        // max with -1, min with 1
        AscendC::Maxs(tmp, outLocal, -1.0f, outDim_);
        AscendC::Mins(outLocal, tmp, 1.0f, outDim_);
        tmpBuf_.FreeTensor(tmp);

        // Step 6: Mish: out = x * tanh(softplus(x))
        // softplus(x) = ln(1 + exp(x))
        // We'll compute element-wise using AscendC math functions.
        AscendC::LocalTensor<float> xCopy = tmpBuf_.AllocTensor<float>(); // reuse tmp
        AscendC::Copy(xCopy, outLocal, outDim_); // copy current output
        // Compute softplus: t = exp(xCopy), then t = log1p(t)
        AscendC::Exp(xCopy, xCopy, outDim_);
        AscendC::Log1p(xCopy, xCopy, outDim_); // softplus now in xCopy
        // Compute tanh(softplus)
        AscendC::Tanh(xCopy, xCopy, outDim_);
        // Multiply by original value (still in outLocal? but we overwrote, so we need original)
        // Actually we copied to xCopy, then computed softplus etc. We need original stored separately.
        // Let's use a third buffer or recompute? For simplicity, we'll use another tempo buffer.
        // But we only have one tmpBuf. We'll allocate from outputBuf? We'll use another region.
        // For simplicity, we'll do in-place: keep original in outLocal, use xCopy for tanh, then multiply.
        // We need original value before mish for multiplication.
        // We'll store original in a separate temp. Since we have outputBuf, we can re-read? We'll use another buffer.
        // Actually we can compute mish as: y = x * tanh(log1p(exp(x))).
        // We'll compute exp(x) store in xCopy, then log1p, then tanh, then multiply with original (still in outLocal).
        // But outLocal holds the x after hardtanh. We need that x for final multiplication.
        // We'll use a copy: we stored xCopy = outLocal (hardtanh result). Then compute softplus on xCopy.
        // After tanh, we have tanh(softplus) in xCopy.
        // Then we multiply: outLocal = outLocal * xCopy (elementwise).
        AscendC::Mul(outLocal, outLocal, xCopy, outDim_); // mish result
        tmpBuf_.FreeTensor(xCopy);

        // Step 7: Group norm
        uint32_t dimPerGroup = outDim_ / numGroups_;
        AscendC::LocalTensor<float> meanPerGroup = gnMeanBuf_.AllocTensor<float>();
        AscendC::LocalTensor<float> varPerGroup = gnVarBuf_.AllocTensor<float>();
        // Compute mean for each group
        for (uint32_t g = 0; g < numGroups_; ++g) {
            uint32_t start = g * dimPerGroup;
            float sum = 0.0f;
            for (uint32_t k = 0; k < dimPerGroup; ++k) {
                sum += outLocal.GetValue(start + k);
            }
            float mean = sum / static_cast<float>(dimPerGroup);
            meanPerGroup.SetValue(g, mean);
        }
        // Compute variance for each group
        for (uint32_t g = 0; g < numGroups_; ++g) {
            uint32_t start = g * dimPerGroup;
            float mean = meanPerGroup.GetValue(g);
            float sqDiffSum = 0.0f;
            for (uint32_t k = 0; k < dimPerGroup; ++k) {
                float diff = outLocal.GetValue(start + k) - mean;
                sqDiffSum += diff * diff;
            }
            float var = sqDiffSum / static_cast<float>(dimPerGroup);
            varPerGroup.SetValue(g, var);
        }
        // Normalize: (x - mean) * gamma / sqrt(var+eps) + beta
        for (uint32_t j = 0; j < outDim_; ++j) {
            uint32_t g = j / dimPerGroup;
            float mean = meanPerGroup.GetValue(g);
            float var = varPerGroup.GetValue(g);
            float gamma = gnGammaGm_.GetValue(j);
            float beta = gnBetaGm_.GetValue(j);
            float x_val = outLocal.GetValue(j);
            float norm = (x_val - mean) / sqrtf(var + eps_);
            outLocal.SetValue(j, norm * gamma + beta);
        }

        // Step 8: Copy output row back to GM
        AscendC::DataCopy(yGm_[row * outDim_], outLocal, outDim_);

        // Free buffers
        outputBuf_.FreeTensor(outLocal);
        gnMeanBuf_.FreeTensor(meanPerGroup);
        gnVarBuf_.FreeTensor(varPerGroup);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightColBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outputBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> gnMeanBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> gnVarBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalBatch_ {0};
    uint32_t outDim_ {0};
    uint32_t inDim_ {0};
    uint32_t numGroups_ {1};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
    GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.totalBatch, tiling_data.outDim, tiling_data.inDim,
            tiling_data.numGroups, tiling_data.eps);
    op.Process();
}
