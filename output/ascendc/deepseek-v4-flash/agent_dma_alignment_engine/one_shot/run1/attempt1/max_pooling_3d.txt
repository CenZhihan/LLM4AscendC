#include "max_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;                // 256B / 4
static constexpr uint32_t WINDOW_FLOATS = 27;              // 3x3x3

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t inputN = static_cast<uint32_t>(inputShape->GetDim(0));
    uint32_t inputC = static_cast<uint32_t>(inputShape->GetDim(1));
    uint32_t inputD = static_cast<uint32_t>(inputShape->GetDim(2));
    uint32_t inputH = static_cast<uint32_t>(inputShape->GetDim(3));
    uint32_t inputW = static_cast<uint32_t>(inputShape->GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];

    // compute output dimensions
    auto outDim = [&](uint32_t in) -> uint32_t {
        return (in + 2 * padding - kernelSize) / stride + 1;
    };
    uint32_t outputD = outDim(inputD);
    uint32_t outputH = outDim(inputH);
    uint32_t outputW = outDim(inputW);
    uint32_t totalLength = inputN * inputC * outputD * outputH * outputW;

    // UB budget: For each tile we need:
    // - double buffer for the 27 floats per window: 2 * 27 * 4 = 216 bytes
    // - one output buffer (tileLength * 4 bytes)
    // - a small temp for max (use scalar, negligible)
    uint32_t tileElems = (UB_BUDGET_BYTES - 216) / sizeof(float);
    tileElems = AlignDown(tileElems, ALIGN_ELEMS);
    if (tileElems < ALIGN_ELEMS) tileElems = ALIGN_ELEMS;
    if (tileElems > totalLength) tileElems = AlignDown(totalLength, ALIGN_ELEMS);
    if (tileElems == 0) tileElems = totalLength;

    uint32_t numTiles = CeilDiv(totalLength, tileElems);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileElems);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_inputN(inputN);
    tiling.set_inputC(inputC);
    tiling.set_inputD(inputD);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_outputD(outputD);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    int32_t stride = context->GetAttrs()->GetAttrPointer<int32_t>(1)[0];

    uint32_t inputD = x_shape->GetDim(2);
    uint32_t inputH = x_shape->GetDim(3);

    auto outDim = [&](uint32_t in) -> uint32_t {

    };
    uint32_t outputD = outDim(inputD);
    uint32_t outputH = outDim(inputH);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR tiling)
    {
        GET_TILING_DATA(tiling_data, tiling);

        totalLen_ = tiling_data.totalLength;
        tileLen_ = tiling_data.tileLength;
        kernelSize_ = tiling_data.kernelSize;
        stride_ = tiling_data.stride;
        padding_ = tiling_data.padding;
        inputN_ = tiling_data.inputN;
        inputC_ = tiling_data.inputC;
        inputD_ = tiling_data.inputD;
        inputH_ = tiling_data.inputH;
        inputW_ = tiling_data.inputW;
        outputD_ = tiling_data.outputD;
        outputH_ = tiling_data.outputH;
        outputW_ = tiling_data.outputW;

        // Global tensor for x (full input) and y (full output)
        xGm_.SetGlobalBuffer((__gm__ float*)x, inputN_ * inputC_ * inputD_ * inputH_ * inputW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Local buffers: double buffer for 27 floats (window) and output buffer
        pipe_.InitBuffer(inWindowBuf0_, BUFFER_NUM, WINDOW_FLOATS * sizeof(float));
        pipe_.InitBuffer(inWindowBuf1_, BUFFER_NUM, WINDOW_FLOATS * sizeof(float));
        pipe_.InitBuffer(outBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output elements
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Stride in output elements
        uint32_t strideOut = blockNum * tileLen_;

        // Local output buffer
        AscendC::LocalTensor<float> outLocal = outBuf_.AllocTensor<float>();

        // Process tiles
        uint32_t remaining = totalLen_ - base;
        uint32_t curTile = (remaining > tileLen_) ? tileLen_ : remaining;

        while (true) {
            // Prefetch next tile
            uint32_t nextBase = base + strideOut;
            uint32_t nextTile = 0;
            if (nextBase < totalLen_) {
                uint32_t nextRem = totalLen_ - nextBase;
                nextTile = (nextRem > tileLen_) ? tileLen_ : nextRem;
            }

            // Launch DMA reads for current tile (each output element reads a window)
            PrefetchWindows(base, curTile);

            // Compute max for each output element in current tile
            ComputeWindows(base, curTile, outLocal);

            // Wait for previous write done (if any) – since we use double buffer for read, we need to synchronize writes
            // Use pipe barrier between tiles to ensure no overlap
            if (nextBase >= totalLen_ || nextTile == 0) {
                // Last tile: write out directly after compute
                WriteOut(base, curTile, outLocal);
                break;
            }

            // Write out previous tile while prefetching next
            // But to keep simple, we write out after compute and then prefetch next
            WriteOut(base, curTile, outLocal);

            // Proceed to next tile
            base = nextBase;
            curTile = nextTile;
        }

        outBuf_.FreeTensor(outLocal);
    }

private:
    __aicore__ inline void PrefetchWindows(uint32_t baseOut, uint32_t numOut)
    {
        // We'll load windows one by one inside ComputeWindows using double buffering.
        // This function is a placeholder for pipelining if needed.
        // For simplicity, we load inside ComputeWindows without prefetch.
        // But to demonstrate double buffering, we will load the first window here.
        // Instead we integrate loading inside ComputeWindows to keep code concise.
        // For a production kernel we would implement a proper producer-consumer.
    }

    __aicore__ inline void ComputeWindows(uint32_t baseOut, uint32_t numOut,
                                           const AscendC::LocalTensor<float>& outLocal)
    {
        // Loop over each output element in the tile
        for (uint32_t idx = 0; idx < numOut; ++idx) {
            uint32_t linearOut = baseOut + idx;
            // Compute spatial indices from linear index (N, C, D, H, W)
            uint32_t tmp = linearOut;
            uint32_t wOut = tmp % outputW_; tmp /= outputW_;
            uint32_t hOut = tmp % outputH_; tmp /= outputH_;
            uint32_t dOut = tmp % outputD_; tmp /= outputD_;
            uint32_t c = tmp % inputC_; tmp /= inputC_;
            uint32_t n = tmp;

            // Input window origin (with padding considered)
            int32_t dStart = static_cast<int32_t>(dOut) * stride_ - padding_;
            int32_t hStart = static_cast<int32_t>(hOut) * stride_ - padding_;
            int32_t wStart = static_cast<int32_t>(wOut) * stride_ - padding_;

            // Load the 3x3x3 window using double buffering for efficiency
            // Buffer A for loading, buffer B for computing, alternate
            AscendC::LocalTensor<float> winLocal = (idx % 2 == 0) ?
                inWindowBuf0_.AllocTensor<float>() :
                inWindowBuf1_.AllocTensor<float>();

            float maxVal = -__builtin_huge_valf();
            uint32_t loadIdx = 0;
            for (int32_t kd = 0; kd < kernelSize_; ++kd) {
                int32_t d = dStart + kd;
                for (int32_t kh = 0; kh < kernelSize_; ++kh) {
                    int32_t h = hStart + kh;
                    for (int32_t kw = 0; kw < kernelSize_; ++kw) {
                        int32_t w = wStart + kw;

                        bool inside = (n < inputN_ && c < inputC_ &&
                                       d >= 0 && d < static_cast<int32_t>(inputD_) &&
                                       h >= 0 && h < static_cast<int32_t>(inputH_) &&
                                       w >= 0 && w < static_cast<int32_t>(inputW_));
                        float val;
                        if (!inside) {
                            val = -__builtin_huge_valf();
                        } else {
                            // Compute linear index in xGm_
                            uint32_t inIdx = ((n * inputC_ + c) * inputD_ + d) * inputH_ * inputW_ + h * inputW_ + w;
                            // Load single element using DataCopyPad (non-aligned small size)
                            AscendC::DataCopyPad(winLocal[loadIdx], xGm_[inIdx], 1);
                            val = winLocal.GetValue(loadIdx);
                        }
                        maxVal = (val > maxVal) ? val : maxVal;
                        ++loadIdx;
                    }
                }
            }

            // Free window buffer (we are done loading)
            if (idx % 2 == 0)
                inWindowBuf0_.FreeTensor(winLocal);
            else
                inWindowBuf1_.FreeTensor(winLocal);

            // Store result to output local buffer
            outLocal.SetValue(idx, maxVal);
        }
    }

    __aicore__ inline void WriteOut(uint32_t baseOut, uint32_t numOut,
                                     const AscendC::LocalTensor<float>& outLocal)
    {
        // Write back to GM using DataCopyPad if not aligned
        // Here we assume numOut is multiple of ALIGN_ELEMS (64) – tiling design ensures that.
        AscendC::DataCopy(yGm_[baseOut], outLocal, numOut);
    }

private:
    // Pipe
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inWindowBuf0_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inWindowBuf1_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outBuf_;

    // Global tensors
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    // Tile config
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    int32_t kernelSize_{3};
    int32_t stride_{2};
    int32_t padding_{1};
    uint32_t inputN_{0}, inputC_{0}, inputD_{0}, inputH_{0}, inputW_{0};
    uint32_t outputD_{0}, outputH_{0}, outputW_{0};

    static constexpr uint32_t WINDOW_FLOATS = 27;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelMaxPooling3dCustom op;
    op.Init(x, y, tiling);
    op.Process();
}
