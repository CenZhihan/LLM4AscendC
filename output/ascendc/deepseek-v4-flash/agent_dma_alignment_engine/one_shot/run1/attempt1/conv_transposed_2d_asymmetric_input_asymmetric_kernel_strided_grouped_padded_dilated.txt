#include "kernel_operator.h"

class KernelConvTransposed2dAsymmetric {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetric() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t N, uint32_t C_in, uint32_t H_in, uint32_t W_in,
                                uint32_t C_out, uint32_t H_out, uint32_t W_out,
                                uint32_t kH, uint32_t kW,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t padding_h, uint32_t padding_w,
                                uint32_t dilation_h, uint32_t dilation_w,
                                uint32_t groups,
                                uint32_t tileH, uint32_t tileW)
    {
        // store parameters
        N_ = N; C_in_ = C_in; H_in_ = H_in; W_in_ = W_in;
        C_out_ = C_out; H_out_ = H_out; W_out_ = W_out;
        kH_ = kH; kW_ = kW;
        stride_h_ = stride_h; stride_w_ = stride_w;
        padding_h_ = padding_h; padding_w_ = padding_w;
        dilation_h_ = dilation_h; dilation_w_ = dilation_w;
        groups_ = groups;
        tileH_ = tileH; tileW_ = tileW;

        C_per_grp_ = C_in / groups;  // input channels per group (also weight groups)
        // Note: For transpose, weight shape (C_in, C_out_per_group, kH, kW)
        // We'll weight tensor: [C_in][C_out_per_group][kH][kW] in ND format (row-major)
        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C_in * H_in * W_in);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, C_in * (C_out_/groups) * kH * kW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N * C_out * H_out * W_out);

        // Buffer sizes: output tile: tileH * tileW * C_out floats
        // We need work for accumulation: double buffer for partial sums or atomic addition.
        // For simplicity, use single output accumulation in UB and then write back.
        // Also need input tile and weights for each group.
        // Using a conservative allocation:
        // - outBuf: tileH * tileW * C_out floats
        // - inBuf: (tileH_out_contrib * tileW_out_contrib) * C_in? complex, skip. Use small fixed.
        // We'll design a simpler blocked approach: each block processes one spatial tile of output.
        // For each output pixel, we need to gather input pixels and compute dot product.
        // Because of complexity, we'll do sequential per-pixel using scalar? Not efficient.
        // Instead, we'll rely on matmul or convolution using AscendC's built-in? Not allowed.
        // We'll implement a naive triple loop per output pixel using vector operations.
        // Use local memory for input window and weights.

        // Determine max input window per output tile: (tileH_ * stride_h + (kH-1)*dilation_h) x (tileW_ * stride_w + (kW-1)*dilation_w) pixels, each channel group.
        // That may exceed UB. We'll process each output pixel independently.
        // We'll allocate UB for single pixel accumulation: C_out floats, and weight for that pixel per group.
        // Use double buffering.
        pipe_.InitBuffer(inQueueX_, 1, C_per_grp_ * kH_ * kW_ * sizeof(float)); // enough for one kernel window for one group
        pipe_.InitBuffer(outQueueY_, 1, C_out_ * sizeof(float)); // for accumulated output per spatial location
        pipe_.InitBuffer(weightQueue_, 1, C_per_grp_ * (C_out_/groups_) * kH_ * kW_ * sizeof(float)); // weight for one group? too big, we'll stream smaller.
        // For simplicity, we will load weights per group inside the inner loop using global reads.
        // This kernel is a placeholder; real implementation would be more optimized.
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t totalTiles = N_ * CeilDiv(H_out_, tileH_) * CeilDiv(W_out_, tileW_);
        uint32_t tilePerBlock = (totalTiles + blockNum - 1) / blockNum;
        uint32_t startTile = blockIdx * tilePerBlock;
        uint32_t endTile = (startTile + tilePerBlock > totalTiles) ? totalTiles : startTile + tilePerBlock;

        for (uint32_t t = startTile; t < endTile; ++t) {
            // Decode tile to (n, t_h, t_w)
            uint32_t numTH = CeilDiv(H_out_, tileH_);
            uint32_t numTW = CeilDiv(W_out_, tileW_);
            uint32_t n = t / (numTH * numTW);
            uint32_t rem = t % (numTH * numTW);
            uint32_t t_h = rem / numTW;
            uint32_t t_w = rem % numTW;

            uint32_t hStart = t_h * tileH_;
            uint32_t hEnd = hStart + tileH_;
            if (hEnd > H_out_) hEnd = H_out_;
            uint32_t wStart = t_w * tileW_;
            uint32_t wEnd = wStart + tileW_;
            if (wEnd > W_out_) wEnd = W_out_;

            // Allocate output buffer for this tile
            AscendC::LocalTensor<float> outTile = outQueueY_.AllocTensor<float>();
            outTile.SetValue(0, 0.0f, (hEnd-hStart)*(wEnd-wStart)*C_out_);

            for (uint32_t h = hStart; h < hEnd; ++h) {
                for (uint32_t w = wStart; w < wEnd; ++w) {
                    // compute output pixel (n, c_out, h, w)
                    // for each group g, compute contribution
                    for (uint32_t g = 0; g < groups_; ++g) {
                        uint32_t c_out_start = g * (C_out_/groups_);
                        uint32_t c_in_start = g * C_per_grp_;
                        // For each input channel in group: c_in in [c_in_start, c_in_start+C_per_grp_)
                        // For each kernel element: (kh, kw) with dilation
                        // Input pixel position: (n, c_in, h_in, w_in)
                        // where h_in = (h + padding_h - kh*dilation_h) / stride_h? careful: transposed inverse.
                        // For transposed conv: h_in = (h - output_padding_h + padding_h - kh*dilation_h) / stride_h? Standard formula:
                        // output_pixel_pos = stride * input_pixel_pos + dilation * kernel_pos - padding
                        // So input_pixel_pos = (output_pixel_pos + padding - dilation*kernel_pos) / stride
                        // Only valid if divisible. We'll iterate over kernel positions that produce valid input indices.
                        for (uint32_t kh = 0; kh < kH_; ++kh) {
                            for (uint32_t kw = 0; kw < kW_; ++kw) {
                                int32_t h_in = (h + padding_h_ - (kh * dilation_h_));
                                int32_t w_in = (w + padding_w_ - (kw * dilation_w_));
                                if (h_in < 0 || w_in < 0) continue;
                                if (h_in % stride_h_ != 0 || w_in % stride_w_ != 0) continue;
                                h_in /= stride_h_;
                                w_in /= stride_w_;
                                if ((uint32_t)h_in >= H_in_ || (uint32_t)w_in >= W_in_) continue;

                                // load input value
                                uint32_t inpIdx = n * C_in_ * H_in_ * W_in_ + c_in_start * H_in_ * W_in_ + h_in * W_in_ + w_in;
                                float inp = xGm_.GetValue(inpIdx); // scalar load

                                // load weight for this kernel position and output channel
                                uint32_t wIdx = (c_in_start + 0) * (C_out_/groups_) * kH_ * kW_ + kh * kW_ + kw; // for first in group
                                // Actually weight layout: [c_in][c_out_per_group][kh][kw]
                                // We need to load weight for all output channels in this group (C_out_per_group)
                                AscendC::LocalTensor<float> wLocal = weightQueue_.AllocTensor<float>();
                                for (uint32_t co = 0; co < C_out_/groups_; ++co) {
                                    uint32_t wOffset = (c_in_start + 0) * (C_out_/groups_) * kH_ * kW_ + co * kH_ * kW_ + kh * kW_ + kw;
                                    wLocal.SetValue(co, wGm_.GetValue(wOffset));
                                }
                                // accumulate for each output channel
                                for (uint32_t co = 0; co < C_out_/groups_; ++co) {
                                    uint32_t outOff = n * C_out_ * H_out_ * W_out_ + (c_out_start + co) * H_out_ * W_out_ + h * W_out_ + w;
                                    float acc = outTile.GetValue(outOff);
                                    acc += inp * wLocal.GetValue(co);
                                    outTile.SetValue(outOff, acc);
                                }
                                weightQueue_.FreeTensor(wLocal);
                            }
                        }
                    }
                }
            }

            // Write output tile to global
            // This is simplified: we need to copy only the tile region.
            // Use DataCopy from LocalTensor to GlobalTensor with offsets.
            // We'll iterate over each output channel and spatial.
            for (uint32_t co = 0; co < C_out_; ++co) {
                for (uint32_t h = hStart; h < hEnd; ++h) {
                    uint32_t localOffset = (co * (hEnd-hStart) * (wEnd-wStart)) + (h-hStart)*(wEnd-wStart);
                    uint32_t gmOffset = n * C_out_ * H_out_ * W_out_ + co * H_out_ * W_out_ + h * W_out_ + wStart;
                    AscendC::DataCopy(yGm_[gmOffset], outTile[localOffset], (wEnd-wStart));
                }
            }
            outQueueY_.FreeTensor(outTile);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_in_, H_in_, W_in_;
    uint32_t C_out_, H_out_, W_out_;
    uint32_t kH_, kW_;
    uint32_t stride_h_, stride_w_;
    uint32_t padding_h_, padding_w_;
    uint32_t dilation_h_, dilation_w_;
    uint32_t groups_;
    uint32_t tileH_, tileW_;
    uint32_t C_per_grp_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    KernelConvTransposed2dAsymmetric op;
    op.Init(x, weight, y,
            td.N, td.C_in, td.H_in, td.W_in,
            td.C_out, td.H_out, td.W_out,
            td.kH, td.kW,
            td.stride_h, td.stride_w,
            td.padding_h, td.padding_w,
            td.dilation_h, td.dilation_w,
            td.groups,
            td.tileH, td.tileW);
    op.Process();
}
