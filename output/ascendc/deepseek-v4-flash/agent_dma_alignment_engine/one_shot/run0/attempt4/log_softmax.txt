#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;               // 32B alignment
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)   { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    gert::Shape x_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batchSize = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t dimSize   = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t totalLength = batchSize * dimSize;

    // UB needs: 2 input tiles, 2 output tiles, plus reduction temporaries (max, sum, logtmp)
    // Use 5x tileLen floats for safety.
    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (dimSize < tileLen) {
        tileLen = AlignDown(dimSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = dimSize;
    }

    uint32_t blockDim = (batchSize < MAX_BLOCKS) ? batchSize : MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_batchSize(batchSize);
    tiling.set_dimSize(dimSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t batchSize,
                                uint32_t dimSize, uint32_t tileLength,
                                uint32_t blockDim)
    {
        totalLen_ = totalLength;
        batchSize_ = batchSize;
        dimSize_ = dimSize;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        // scalars for reduction (1 element each)
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t rowsPerBlock = (batchSize_ + blockDim_ - 1) / blockDim_;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > batchSize_) endRow = batchSize_;
        if (startRow >= batchSize_) return;

        for (uint32_t row = startRow; row < endRow; ++row) {
            // ---- Pass 1: compute global maximum of the row ----
            AscendC::LocalTensor<float> globalMax = scalarQueue_.AllocTensor<float>();
            globalMax.SetValue(0, -1.0e38f);

            for (uint32_t tile = 0; tile < dimSize_; tile += tileLen_) {
                uint32_t len = (tile + tileLen_ <= dimSize_) ? tileLen_ : (dimSize_ - tile);
                uint32_t gmOffset = row * dimSize_ + tile;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
                inQueueX_.EnQue(xLocal);
                xLocal = inQueueX_.DeQue<float>();

                // compute max of this tile
                AscendC::LocalTensor<float> tileMax = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceMax(tileMax, xLocal, len);
                // update global max (elementwise max on 1-element tensors)
                AscendC::Max(globalMax, globalMax, tileMax, 1);

                inQueueX_.FreeTensor(xLocal);
                tmpQueue_.FreeTensor(tileMax);
            }
            float rowMax = globalMax.GetValue(0);
            scalarQueue_.FreeTensor(globalMax);

            // ---- Pass 2: compute sum of exp(x - max) ----
            AscendC::LocalTensor<float> globalSum = scalarQueue_.AllocTensor<float>();
            globalSum.SetValue(0, 0.0f);

            for (uint32_t tile = 0; tile < dimSize_; tile += tileLen_) {
                uint32_t len = (tile + tileLen_ <= dimSize_) ? tileLen_ : (dimSize_ - tile);
                uint32_t gmOffset = row * dimSize_ + tile;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
                inQueueX_.EnQue(xLocal);
                xLocal = inQueueX_.DeQue<float>();

                AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
                // compute exp(x - rowMax)
                AscendC::Adds(expLocal, xLocal, -rowMax, len);
                AscendC::Exp(expLocal, expLocal, len);
                // accumulate sum
                AscendC::LocalTensor<float> tileSum = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceAdd(tileSum, expLocal, len);
                AscendC::Add(globalSum, globalSum, tileSum, 1);

                inQueueX_.FreeTensor(xLocal);
                tmpQueue_.FreeTensor(expLocal);
                tmpQueue_.FreeTensor(tileSum);
            }
            float rowSum = globalSum.GetValue(0);
            scalarQueue_.FreeTensor(globalSum);

            // ---- compute logSum = rowMax + log(rowSum) ----
            AscendC::LocalTensor<float> logSumTensor = scalarQueue_.AllocTensor<float>();
            logSumTensor.SetValue(0, rowSum);
            AscendC::Log(logSumTensor, logSumTensor, 1);
            float logSum = logSumTensor.GetValue(0) + rowMax;
            scalarQueue_.FreeTensor(logSumTensor);

            // ---- Pass 3: compute output = x - logSum ----
            for (uint32_t tile = 0; tile < dimSize_; tile += tileLen_) {
                uint32_t len = (tile + tileLen_ <= dimSize_) ? tileLen_ : (dimSize_ - tile);
                uint32_t gmOffset = row * dimSize_ + tile;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
                inQueueX_.EnQue(xLocal);
                xLocal = inQueueX_.DeQue<float>();

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                // y = x - logSum
                AscendC::Adds(yLocal, xLocal, -logSum, len);

                inQueueX_.FreeTensor(xLocal);
                outQueueY_.EnQue(yLocal);

                yLocal = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    static constexpr int32_t BUFFER_NUM = 2;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t dimSize_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.batchSize,
            tiling_data.dimSize,
            tiling_data.tileLength,
            tiling_data.blockDim);
    op.Process();
}
