#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    // FIX: GetOriginShape() returns a value, not a pointer
    gert::Shape x_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batchSize = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t dimSize   = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t totalLength = batchSize * dimSize;

    // UB budget: 3 tiles + small overhead
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (dimSize < tileLen) {
        tileLen = AlignDown(dimSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = dimSize;
    }

    uint32_t blockDim = (batchSize < MAX_BLOCKS) ? batchSize : MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_batchSize(batchSize);
    tiling.set_dimSize(dimSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

        this->Output("y")

            .UnknownShapeFormat({ge::FORMAT_ND});

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");

};

OP_ADD(LogSoftmaxCustom);

kernel_src="""

#include "kernel_operator.h"

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t batchSize,
                                uint32_t dimSize, uint32_t tileLength,
                                uint32_t blockDim)
    {
        totalLen_ = totalLength;
        batchSize_ = batchSize;
        dimSize_ = dimSize;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t rowsPerBlock = (batchSize_ + blockDim_ - 1) / blockDim_;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (endRow > batchSize_) endRow = batchSize_;
        if (startRow >= batchSize_) return;

        // Allocate UB buffers reused per row
        AscendC::TPipe pipe;
        AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueue;
        AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue;
        AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcQueue;

        pipe.InitBuffer(inQueue, 1, tileLen_ * sizeof(float));
        pipe.InitBuffer(outQueue, 1, tileLen_ * sizeof(float));
        pipe.InitBuffer(calcQueue, 1, tileLen_ * sizeof(float));

        for (uint32_t row = startRow; row < endRow; ++row) {
            float rowMax = -3.402823466e+38f;
            float rowSum = 0.0f;

            // ---------- Pass 1: compute max ----------
            for (uint32_t tile = 0; tile < dimSize_; tile += tileLen_) {
                uint32_t len = (tile + tileLen_ <= dimSize_) ? tileLen_ : (dimSize_ - tile);
                uint32_t gmOffset = row * dimSize_ + tile;

                AscendC::LocalTensor<float> xLocal = inQueue.AllocTensor<float>();
                LoadTile(xLocal, xGm_, gmOffset, len);
                inQueue.EnQue(xLocal);

                xLocal = inQueue.DeQue<float>();
                AscendC::LocalTensor<float> maxLocal = calcQueue.AllocTensor<float>();
                AscendC::LocalTensor<float> oneLocal = calcQueue.AllocTensor<float>();
                AscendC::ReduceMax(maxLocal, xLocal, len);
                AscendC::BroadCast(oneLocal, maxLocal, 1);
                rowMax = (rowMax > oneLocal.GetValue(0)) ? rowMax : oneLocal.GetValue(0);
                calcQueue.FreeTensor(oneLocal);
                calcQueue.FreeTensor(maxLocal);
                inQueue.FreeTensor(xLocal);
            }

            // ---------- Pass 2: compute sum of exp(x - max) ----------
            for (uint32_t tile = 0; tile < dimSize_; tile += tileLen_) {
                uint32_t len = (tile + tileLen_ <= dimSize_) ? tileLen_ : (dimSize_ - tile);
                uint32_t gmOffset = row * dimSize_ + tile;

                AscendC::LocalTensor<float> xLocal = inQueue.AllocTensor<float>();
                LoadTile(xLocal, xGm_, gmOffset, len);
                inQueue.EnQue(xLocal);

                xLocal = inQueue.DeQue<float>();
                AscendC::LocalTensor<float> subLocal = calcQueue.AllocTensor<float>();
                AscendC::LocalTensor<float> expLocal = calcQueue.AllocTensor<float>();
                AscendC::LocalTensor<float> sumLocal = calcQueue.AllocTensor<float>();

                AscendC::Subs(subLocal, xLocal, rowMax, len);
                AscendC::Exp(expLocal, subLocal, len);
                AscendC::ReduceSum(sumLocal, expLocal, len);
                rowSum += sumLocal.GetValue(0);

                calcQueue.FreeTensor(sumLocal);
                calcQueue.FreeTensor(expLocal);
                calcQueue.FreeTensor(subLocal);
                inQueue.FreeTensor(xLocal);
            }

            float logSum = logf(rowSum);

            // ---------- Pass 3: compute output ----------
            for (uint32_t tile = 0; tile < dimSize_; tile += tileLen_) {
                uint32_t len = (tile + tileLen_ <= dimSize_) ? tileLen_ : (dimSize_ - tile);
                uint32_t gmOffset = row * dimSize_ + tile;

                AscendC::LocalTensor<float> xLocal = inQueue.AllocTensor<float>();
                LoadTile(xLocal, xGm_, gmOffset, len);
                inQueue.EnQue(xLocal);

                xLocal = inQueue.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueue.AllocTensor<float>();

                AscendC::Subs(yLocal, xLocal, rowMax, len);
                AscendC::Subs(yLocal, yLocal, logSum, len);

                inQueue.FreeTensor(xLocal);
                outQueue.EnQue(yLocal);

                yLocal = outQueue.DeQue<float>();
                StoreTile(yGm_, gmOffset, yLocal, len);
                outQueue.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline void LoadTile(const AscendC::LocalTensor<float>& dst,
                                    const AscendC::GlobalTensor<float>& src,
                                    uint32_t offset, uint32_t len)
    {
        if ((offset % 8 == 0) && (len % 8 == 0)) {
            AscendC::DataCopy(dst, src[offset], len);
        } else {
            AscendC::DataCopyPad(dst, src[offset], len);
        }
    }

    __aicore__ inline void StoreTile(const AscendC::GlobalTensor<float>& dst,
                                     uint32_t offset,
                                     const AscendC::LocalTensor<float>& src,
                                     uint32_t len)
    {
        if ((offset % 8 == 0) && (len % 8 == 0)) {
            AscendC::DataCopy(dst[offset], src, len);
        } else {
            AscendC::DataCopyPad(dst[offset], src, len);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t dimSize_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.batchSize,
            tiling_data.dimSize,
            tiling_data.tileLength,
            tiling_data.blockDim);
    op.Process();
}
