host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget estimate:
//   inQueueX: 1 tile
//   outQueueY: 1 tile
//   tmpQueue: 1 tile (used for intermediate exp values in second pass)
// Total floats = 3 * tile => bytes = 12 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core share
static constexpr uint32_t ALIGN_ELEMS = 64;              // 256B alignment for DMA (64 floats)
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid excessive latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t numDims = context->GetInputShape(0)->GetOriginShape().GetDimNum();
    // Assumes ND: last dim is the softmax axis (dim=1)
    const uint32_t rowLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(numDims - 1));

    // Compute tileLength from UB budget: 3 buffers of tileLen floats
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));   // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        tileLen = AlignDown(rowLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowLength;
    }
    // tileLen must at most rowLength
    if (tileLen > rowLength) tileLen = rowLength;

    // Adaptive blockDim based on number of rows * tiles
    const uint32_t numRows = totalLength / rowLength;
    // Each block will process some rows. Load balancing: use number of tiles per row as factor.
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_rowLength(rowLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t rowLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        rowLen_ = rowLength;
        numRows_ = totalLen_ / rowLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileLen_ * sizeof(float));  // for exp values
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || rowLen_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Each block processes a contiguous set of rows.
        const uint32_t rowsPerBlock = (numRows_ + blockNum - 1) / blockNum;
        const uint32_t startRow = blockIdx * rowsPerBlock;
        const uint32_t endRow = (startRow + rowsPerBlock > numRows_) ? numRows_ : startRow + rowsPerBlock;

        for (uint32_t rowIdx = startRow; rowIdx < endRow; ++rowIdx) {
            processRow(rowIdx);
        }
    }

private:
    // Phase 1: Find the maximum value along the row.
    __aicore__ inline void phase1FindMax(uint32_t rowOffset, uint32_t numTiles, float &globalMax)
    {
        const float NEG_INF = -1e38f;
        globalMax = NEG_INF;
        for (uint32_t t = 0; t < numTiles; ++t) {
            const uint32_t tileBase = rowOffset + t * tileLen_;
            const uint32_t curLen = (tileLen_ > (rowLen_ - t * tileLen_)) ? (rowLen_ - t * tileLen_) : tileLen_;
            if (curLen == 0) break;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[tileBase], curLen);

            AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceMax(maxLocal, xLocal, 0, curLen, AscendC::REDUCE_DIM);
            float localMax = maxLocal.GetValue(0);
            tmpQueue_.FreeTensor(maxLocal);
            inQueueX_.FreeTensor(xLocal);

            if (localMax > globalMax) globalMax = localMax;
        }
    }

    // Phase 2: Compute sum of exp(x - globalMax)
    __aicore__ inline void phase2ComputeSum(uint32_t rowOffset, uint32_t numTiles, float globalMax, float &globalSum)
    {
        globalSum = 0.0f;
        for (uint32_t t = 0; t < numTiles; ++t) {
            const uint32_t tileBase = rowOffset + t * tileLen_;
            const uint32_t curLen = (tileLen_ > (rowLen_ - t * tileLen_)) ? (rowLen_ - t * tileLen_) : tileLen_;
            if (curLen == 0) break;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[tileBase], curLen);

            // exp(x - globalMax)
            AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(expLocal, xLocal, globalMax, curLen);
            AscendC::Exp(expLocal, expLocal, curLen);

            // Compute local sum
            AscendC::LocalTensor<float> sumLocal = inQueueX_.AllocTensor<float>(); // reuse input buffer
            AscendC::ReduceSum(sumLocal, expLocal, 0, curLen, AscendC::REDUCE_DIM);
            float localSum = sumLocal.GetValue(0);
            globalSum += localSum;

            inQueueX_.FreeTensor(xLocal);
            inQueueX_.FreeTensor(sumLocal);
            tmpQueue_.FreeTensor(expLocal);
        }
    }

    // Phase 3: Normalize to produce final output.
    __aicore__ inline void phase3Normalize(uint32_t rowOffset, uint32_t numTiles, float globalMax, float globalSum)
    {
        for (uint32_t t = 0; t < numTiles; ++t) {
            const uint32_t tileBase = rowOffset + t * tileLen_;
            const uint32_t curLen = (tileLen_ > (rowLen_ - t * tileLen_)) ? (rowLen_ - t * tileLen_) : tileLen_;
            if (curLen == 0) break;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[tileBase], curLen);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // y = x - globalMax
            AscendC::Sub(yLocal, xLocal, globalMax, curLen);
            // exp(y)
            AscendC::Exp(yLocal, yLocal, curLen);
            // y = y / globalSum
            AscendC::Div(yLocal, yLocal, globalSum, curLen);

            AscendC::DataCopy(yGm_[tileBase], yLocal, curLen);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
        }
    }

    __aicore__ inline void processRow(uint32_t rowIdx)
    {
        const uint32_t rowOffset = rowIdx * rowLen_;
        const uint32_t numTiles = (rowLen_ + tileLen_ - 1) / tileLen_;

        float globalMax;
        phase1FindMax(rowOffset, numTiles, globalMax);

        float globalSum;
        phase2ComputeSum(rowOffset, numTiles, globalMax, globalSum);

        // Avoid division by zero (though sum should be >0)
        if (globalSum < 1e-30f) globalSum = 1e-30f;

        phase3Normalize(rowOffset, numTiles, globalMax, globalSum);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
    uint32_t rowLen_ {0};
    uint32_t numRows_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.blockDim, tiling_data.rowLength);
    op.Process();
}
