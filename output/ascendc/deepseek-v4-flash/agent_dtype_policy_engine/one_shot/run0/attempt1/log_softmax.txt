host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: 2 inQueueX + 2 outQueueY + 1 tmpQueue = 5 floats per tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 32;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape &xShape = context->GetInputShape(0)->GetOriginShape();
    // x is 2D: (batchSize, featureSize)
    uint32_t batchSize = static_cast<uint32_t>(xShape.GetDim(0));
    uint32_t featureSize = static_cast<uint32_t>(xShape.GetDim(1));

    // Tile length from UB budget: 5 * tileLen * sizeof(float) <= UB_BUDGET_BYTES
    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN);
    if (maxTile < ALIGN) maxTile = ALIGN;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN);

    uint32_t tileLen = maxTile;
    if (featureSize < tileLen) {
        uint32_t aligned = AlignDown(featureSize, ALIGN);
        tileLen = (aligned > 0) ? aligned : featureSize;
    }

    // Block dimension: number of rows per block, keep conservative
    uint32_t blockDim = batchSize;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_featureSize(featureSize);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t featureSize,
                                uint32_t tileLen)
    {
        batchSize_ = batchSize;
        featureSize_ = featureSize;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * featureSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * featureSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || featureSize_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Partition rows among blocks
        uint32_t rowsPerBlock = (batchSize_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > batchSize_) rowEnd = batchSize_;

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t rowOffset = row * featureSize_;

        // Pass 1: compute per-row max
        float rowMax = -FLT_MAX;
        {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset], tileLen_);
            inQueueX_.EnQue(xLocal);

            uint32_t remaining = featureSize_;
            uint32_t offset = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                // Prefetch next tile if any
                if (remaining > tileLen_) {
                    AscendC::LocalTensor<float> nextX = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(nextX, xGm_[rowOffset + offset + tileLen_], tileLen_);
                    inQueueX_.EnQue(nextX);
                }

                AscendC::LocalTensor<float> xCur = inQueueX_.DeQue<float>();
                // Compute local max
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                AscendC::ReduceMax(xCur, tmp, curLen); // tmp[0] = max
                float localMax;
                AscendC::ScalarGet(tmp, 0, localMax);
                if (localMax > rowMax) rowMax = localMax;
                tmpQueue_.FreeTensor(tmp);
                inQueueX_.FreeTensor(xCur);

                offset += curLen;
                remaining -= curLen;
            }
        }

        // Pass 2: compute sum of exp(x - rowMax) across row
        float rowSumExp = 0.0f;
        {
            uint32_t remaining = featureSize_;
            uint32_t offset = 0;
            // Prefetch first tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset], tileLen_);
            inQueueX_.EnQue(xLocal);

            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                if (remaining > tileLen_) {
                    AscendC::LocalTensor<float> nextX = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(nextX, xGm_[rowOffset + offset + tileLen_], tileLen_);
                    inQueueX_.EnQue(nextX);
                }

                AscendC::LocalTensor<float> xCur = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                // Compute exp(x - rowMax)
                AscendC::Sub(tmp, xCur, rowMax, curLen);
                AscendC::Exp(tmp, tmp, curLen);
                // Sum over tile
                AscendC::ReduceSum(tmp, tmp, curLen);
                float tileSum;
                AscendC::ScalarGet(tmp, 0, tileSum);
                rowSumExp += tileSum;
                tmpQueue_.FreeTensor(tmp);
                inQueueX_.FreeTensor(xCur);

                offset += curLen;
                remaining -= curLen;
            }
        }

        float logSum = AscendC::Ln(rowSumExp);
        float logShift = rowMax + logSum;  // rowMax + log(rowSumExp)

        // Pass 3: compute output y = x - logShift
        {
            uint32_t remaining = featureSize_;
            uint32_t offset = 0;
            // Prefetch first x tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset], tileLen_);
            inQueueX_.EnQue(xLocal);

            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                if (remaining > tileLen_) {
                    AscendC::LocalTensor<float> nextX = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(nextX, xGm_[rowOffset + offset + tileLen_], tileLen_);
                    inQueueX_.EnQue(nextX);
                }

                AscendC::LocalTensor<float> xCur = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Sub(yLocal, xCur, logShift, curLen);
                inQueueX_.FreeTensor(xCur);
                outQueueY_.EnQue<float>(yLocal);

                // Write output
                AscendC::LocalTensor<float> yWrite = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[rowOffset + offset], yWrite, curLen);
                outQueueY_.FreeTensor(yWrite);

                offset += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t featureSize_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.batchSize, tiling_data.featureSize, tiling_data.tileLen);
    op.Process();
}
"""
