host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const gert::StorageShape *inShape = context->GetInputShape(0);
    const gert::Shape &origShape = inShape->GetOriginShape();
    const uint32_t ndims = static_cast<uint32_t>(origShape.GetDimNum());
    // Assume 2D tensor: [batchSize, numFeatures]
    const uint32_t batchSize = (ndims > 0) ? static_cast<uint32_t>(origShape.GetDim(0)) : 0;
    const uint32_t numFeatures = (ndims > 1) ? static_cast<uint32_t>(origShape.GetDim(1)) : 0;

    // UB budget: 5 tiles (2 in, 2 out, 1 tmp) each of tileLength floats
    uint32_t maxTile = UB_BUDGET_BYTES / (5 * 4);  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (numFeatures < tileLen) {
        tileLen = AlignDown(numFeatures, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = numFeatures;
    }

    // blockDim: one block per row ideally, but cap for hardware occupancy
    uint32_t blockDim = batchSize;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_numFeatures(numFeatures);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float INF_NEG = -3.40282347e+38f;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t numFeatures,
                                uint32_t tileLength)
    {
        batchSize_ = batchSize;
        numFeatures_ = numFeatures;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * numFeatures_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * numFeatures_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(redQueue_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || numFeatures_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes rows in a grid-stride loop
        for (uint32_t rowIdx = blockIdx; rowIdx < batchSize_; rowIdx += blockNum) {
            uint32_t base = rowIdx * numFeatures_;
            ProcessRow(base);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t base)
    {
        // 1) Compute row maximum across all tiles
        float rowMax = INF_NEG;
        for (uint32_t offset = 0; offset < numFeatures_; offset += tileLen_) {
            uint32_t curLen = numFeatures_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tileMax = redQueue_.AllocTensor<float>();
            AscendC::ReduceMax(tileMax, xLocal, 1, curLen);
            float tileMaxVal = tileMax.GetValue(0);
            redQueue_.FreeTensor(tileMax);
            rowMax = (tileMaxVal > rowMax) ? tileMaxVal : rowMax;
            inQueueX_.FreeTensor(xLocal);
        }

        // 2) Compute sum of exp(x - rowMax) across all tiles
        float sumExp = 0.0f;
        for (uint32_t offset = 0; offset < numFeatures_; offset += tileLen_) {
            uint32_t curLen = numFeatures_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
            inQueueX_.EnQue(xLocal);

            xLocal = inQueueX_.DeQue<float>();
            AscendC::Subs(xLocal, xLocal, rowMax, curLen);
            AscendC::Exp(xLocal, xLocal, curLen);
            AscendC::LocalTensor<float> tileSum = redQueue_.AllocTensor<float>();
            AscendC::ReduceSum(tileSum, xLocal, 1, curLen);
            float tileSumVal = tileSum.GetValue(0);
            redQueue_.FreeTensor(tileSum);
            sumExp += tileSumVal;
            inQueueX_.FreeTensor(xLocal);
        }

        // 3) Write output: y = exp(x - rowMax) / sumExp
        float invSumExp = 1.0f / sumExp;
        for (uint32_t offset = 0; offset < numFeatures_; offset += tileLen_) {
            uint32_t curLen = numFeatures_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
            inQueueX_.EnQue(xLocal);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            xLocal = inQueueX_.DeQue<float>();
            AscendC::Subs(xLocal, xLocal, rowMax, curLen);
            AscendC::Exp(xLocal, xLocal, curLen);
            AscendC::Muls(yLocal, xLocal, invSumExp, curLen);
            inQueueX_.FreeTensor(xLocal);

            outQueueY_.EnQue<float>(yLocal);
            yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base + offset], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> redQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t numFeatures_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.batchSize, tiling_data.numFeatures, tiling_data.tileLength);
    op.Process();
}
