#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 32;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const gert::Shape &xShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batchSize = static_cast<uint32_t>(xShape.GetDim(0));
    uint32_t featureSize = static_cast<uint32_t>(xShape.GetDim(1));

    // UB budget: 2 inQueue (input) + 2 outQueue (output) + 1 tmp = 5 * tileLen * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN);
    if (maxTile < ALIGN) maxTile = ALIGN;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN);

    uint32_t tileLen = maxTile;
    if (featureSize < tileLen) {
        uint32_t aligned = AlignDown(featureSize, ALIGN);
        tileLen = (aligned > 0) ? aligned : featureSize;
    }

    // Block dim: partition rows among blocks (max 24 blocks)
    uint32_t blockDim = batchSize;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;
    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_featureSize(featureSize);
    tiling.set_tileLen(tileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF = -1e38f;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t featureSize,
                                uint32_t tileLen)
    {
        batchSize_ = batchSize;
        featureSize_ = featureSize;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * featureSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * featureSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float)); // for 1-element scalar results
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || featureSize_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Partition rows among blocks
        uint32_t rowsPerBlock = (batchSize_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = rowStart + rowsPerBlock;
        if (rowEnd > batchSize_) rowEnd = batchSize_;

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t rowOffset = row * featureSize_;

        // Pass 1: compute per-row max
        float rowMax = NEG_INF;
        uint32_t remaining = featureSize_;
        uint32_t offset = 0;
        // Prefetch first tile
        {
            AscendC::LocalTensor<float> xFirst = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xFirst, xGm_[rowOffset], tileLen_);
            inQueueX_.EnQue(xFirst);
        }

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // Prefetch next tile if any
            if (remaining > tileLen_) {
                AscendC::LocalTensor<float> nextX = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(nextX, xGm_[rowOffset + offset + tileLen_], tileLen_);
                inQueueX_.EnQue(nextX);
            }

            AscendC::LocalTensor<float> xCur = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tmpMax = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceMax<float, AscendC::REDUCE_TYPE::MAX>(tmpMax, xCur, curLen, 1);
            float localMax = tmpMax(0);
            if (localMax > rowMax) rowMax = localMax;
            scalarQueue_.FreeTensor(tmpMax);
            inQueueX_.FreeTensor(xCur);

            offset += curLen;
            remaining -= curLen;
        }

        // Pass 2: compute sum of exp(x - rowMax)
        float rowSumExp = 0.0f;
        remaining = featureSize_;
        offset = 0;
        // Prefetch first tile
        {
            AscendC::LocalTensor<float> xFirst = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xFirst, xGm_[rowOffset], tileLen_);
            inQueueX_.EnQue(xFirst);
        }

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            if (remaining > tileLen_) {
                AscendC::LocalTensor<float> nextX = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(nextX, xGm_[rowOffset + offset + tileLen_], tileLen_);
                inQueueX_.EnQue(nextX);
            }

            AscendC::LocalTensor<float> xCur = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            // tmp = x - rowMax  (use Add with negative scalar)
            AscendC::Adds(tmp, xCur, -rowMax, curLen);
            AscendC::Exp(tmp, tmp, curLen);
            AscendC::LocalTensor<float> tmpSum = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceSum<float, AscendC::REDUCE_TYPE::SUM>(tmpSum, tmp, curLen, 1);
            float tileSum = tmpSum(0);
            rowSumExp += tileSum;
            scalarQueue_.FreeTensor(tmpSum);
            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xCur);

            offset += curLen;
            remaining -= curLen;
        }

        // Compute log of sum: create a 1-element tensor with rowSumExp
        AscendC::LocalTensor<float> scalarSrc = scalarQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> logTensor = scalarQueue_.AllocTensor<float>();
        AscendC::SetValue<float>(scalarSrc, rowSumExp);
        AscendC::Ln(logTensor, scalarSrc, 1);
        float logSum = logTensor(0);
        scalarQueue_.FreeTensor(scalarSrc);
        scalarQueue_.FreeTensor(logTensor);

        float logShift = rowMax + logSum;

        // Pass 3: compute output y = x - logShift
        remaining = featureSize_;
        offset = 0;
        // Prefetch first tile of x
        {
            AscendC::LocalTensor<float> xFirst = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xFirst, xGm_[rowOffset], tileLen_);
            inQueueX_.EnQue(xFirst);
        }

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            if (remaining > tileLen_) {
                AscendC::LocalTensor<float> nextX = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(nextX, xGm_[rowOffset + offset + tileLen_], tileLen_);
                inQueueX_.EnQue(nextX);
            }

            AscendC::LocalTensor<float> xCur = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // y = x - logShift
            AscendC::Adds(yLocal, xCur, -logShift, curLen);
            inQueueX_.FreeTensor(xCur);
            outQueueY_.EnQue<float>(yLocal);

            // Write output
            AscendC::LocalTensor<float> yWrite = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[rowOffset + offset], yWrite, curLen);
            outQueueY_.FreeTensor(yWrite);

            offset += curLen;
            remaining -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t featureSize_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.batchSize, tiling_data.featureSize, tiling_data.tileLen);
    op.Process();
}
