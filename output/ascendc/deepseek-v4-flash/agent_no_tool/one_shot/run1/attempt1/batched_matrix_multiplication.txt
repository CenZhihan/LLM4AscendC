host_operator_src="""

#include "batched_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget for a single tile:
// A_tile: tileM * tileK floats
// B_tile: tileK * tileN floats
// C_tile: tileM * tileN floats
// Total floats = tileM*(tileK + tileN) + tileK*tileN
// Bytes = 4 * totalFloats
// We also need some workspace for MatMul intermediate; we allocate extra 20%
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    BatchedMatrixMultiplicationCustomTilingData tiling;

    const auto &shapeA = context->GetInputShape(0)->GetOriginShape();
    const auto &shapeB = context->GetInputShape(1)->GetOriginShape();

    // Assume shapeA: [batch, M, K]; shapeB: [batch, K, N]
    uint32_t batchSize = static_cast<uint32_t>(shapeA.GetDim(0));
    uint32_t M = static_cast<uint32_t>(shapeA.GetDim(1));
    uint32_t K = static_cast<uint32_t>(shapeA.GetDim(2));
    uint32_t N = static_cast<uint32_t>(shapeB.GetDim(2));

    // Compute tile sizes from UB budget
    // Heuristic: tileM = tileN = 64, adjust tileK to fit budget
    uint32_t tileM = 64, tileN = 64;
    uint32_t maxTileFloats = UB_BUDGET_BYTES / 4;
    // Full tile requires: tileM*tileK + tileK*tileN + tileM*tileN + overhead
    // We'll set overhead factor = 1.5
    uint32_t tileK = (maxTileFloats) / (tileM + tileN + tileM*tileN/tileK? iterative) 
    // Simplified: set tileK = 128 (common size)
    uint32_t tileK = 128;
    // Ensure alignment
    tileM = AlignDown(tileM, ALIGN);
    tileK = AlignDown(tileK, ALIGN);
    tileN = AlignDown(tileN, ALIGN);
    if (tileM == 0) tileM = 32;
    if (tileK == 0) tileK = 32;
    if (tileN == 0) tileN = 32;

    // Block dimension: we assign one block per batch (up to 24)
    uint32_t blockDim = batchSize;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);
    // Assuming shapeA = [batch, m, k], shapeB = [batch, k, n]
    std::vector<int64_t> dims = {shapeA->GetDim(0), shapeA->GetDim(1), shapeB->GetDim(2)};
    gert::Shape *shapeC = context->GetOutputShape(0);
    *shapeC = gert::Shape(dims);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "matmul.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelBatchedMatMul {
public:
    __aicore__ inline KernelBatchedMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t batchSize, uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        batchSize_ = batchSize;
        M_ = M;
        K_ = K;
        N_ = N;
        tileM_ = tileM;
        tileK_ = tileK;
        tileN_ = tileN;

        strideA_ = M * K;
        strideB_ = K * N;
        strideC_ = M * N;

        A_gm_.SetGlobalBuffer((__gm__ float*)A, batchSize_ * strideA_);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, batchSize_ * strideB_);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, batchSize_ * strideC_);

        // Allocate local tensors for tiles
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
        // Workspace for MatMul
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockDim = GetBlockNum();

        // Distribute batches across blocks
        for (uint32_t b = blockIdx; b < batchSize_; b += blockDim) {
            // For each batch, perform tiled matrix multiplication
            uint32_t baseA = b * strideA_;
            uint32_t baseB = b * strideB_;
            uint32_t baseC = b * strideC_;

            // Loop over M, K, N tiles
            for (uint32_t iM = 0; iM < M_; iM += tileM_) {
                uint32_t curM = min(tileM_, M_ - iM);
                for (uint32_t iN = 0; iN < N_; iN += tileN_) {
                    uint32_t curN = min(tileN_, N_ - iN);

                    // Initialize C tile to zero (will accumulate)
                    // Use temporary tensor for C tile
                    LocalTensor<float> C_tile = outQueueC_.AllocTensor<float>();
                    Duplicate(C_tile, 0.0f, curM * curN);

                    for (uint32_t iK = 0; iK < K_; iK += tileK_) {
                        uint32_t curK = min(tileK_, K_ - iK);

                        // Load A tile
                        LocalTensor<float> A_tile = inQueueA_.AllocTensor<float>();
                        DataCopy(A_tile, A_gm_[baseA + iM * K_ + iK], curM * curK);
                        inQueueA_.EnQue(A_tile);

                        // Load B tile
                        LocalTensor<float> B_tile = inQueueB_.AllocTensor<float>();
                        DataCopy(B_tile, B_gm_[baseB + iK * N_ + iN], curK * curN);
                        inQueueB_.EnQue(B_tile);

                        // Compute partial matmul: C_tile += A_tile * B_tile
                        // Use MatMul API
                        MatMul<MatMulMode::REPLICATE, float, float, float> mm;
                        // Set dimensions (transA=false, transB=false)
                        mm.SetOrgShape(curM, curK, curN);
                        uint32_t lda = K_; // stride between rows of A in global? But we use local tensors, lda = curK
                        uint32_t ldb = N_;
                        uint32_t ldc = N_;
                        // Actually for local tensors, leading dimensions are the tile dimensions
                        mm.SetStride(curK, N_, N_); // lda, ldb, ldc? Need to check API
                        // Alternative: manual accumulation using vector operations
                        // We'll implement a simple triple loop for demonstration
                        for (uint32_t m = 0; m < curM; m++) {
                            for (uint32_t n = 0; n < curN; n++) {
                                float sum = C_tile(m * curN + n);
                                for (uint32_t k = 0; k < curK; k++) {
                                    sum += A_tile(m * curK + k) * B_tile(k * curN + n);
                                }
                                C_tile(m * curN + n) = sum;
                            }
                        }

                        inQueueA_.FreeTensor(inQueueA_.DeQue<float>());
                        inQueueB_.FreeTensor(inQueueB_.DeQue<float>());
                    }

                    // Store C tile to global memory
                    DataCopy(C_gm_[baseC + iM * N_ + iN], C_tile, curM * curN);
                    outQueueC_.FreeTensor(C_tile);
                }
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueA_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueB_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> A_gm_;
    GlobalTensor<float> B_gm_;
    GlobalTensor<float> C_gm_;

    uint32_t batchSize_, M_, K_, N_;
    uint32_t tileM_, tileK_, tileN_;
    uint32_t strideA_, strideB_, strideC_;
};

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(
    GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBatchedMatMul op;
    op.Init(A, B, C,
            tiling_data.batchSize, tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
