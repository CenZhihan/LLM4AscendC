kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t IMG_H = 32;
constexpr int32_t IMG_W = 32;
constexpr int32_t POOL_K = 2;
constexpr int32_t POOL_S = 2;

class KernelLenet5Custom {
public:
    __aicore__ inline KernelLenet5Custom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR conv1_w, GM_ADDR conv1_b,
        GM_ADDR conv2_w, GM_ADDR conv2_b,
        GM_ADDR fc1_w, GM_ADDR fc1_b,
        GM_ADDR fc2_w, GM_ADDR fc2_b,
        GM_ADDR fc3_w, GM_ADDR fc3_b,
        GM_ADDR y, Lenet5CustomTilingData tiling)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        conv1wGm_.SetGlobalBuffer((__gm__ float*)conv1_w);
        conv1bGm_.SetGlobalBuffer((__gm__ float*)conv1_b);
        conv2wGm_.SetGlobalBuffer((__gm__ float*)conv2_w);
        conv2bGm_.SetGlobalBuffer((__gm__ float*)conv2_b);
        fc1wGm_.SetGlobalBuffer((__gm__ float*)fc1_w);
        fc1bGm_.SetGlobalBuffer((__gm__ float*)fc1_b);
        fc2wGm_.SetGlobalBuffer((__gm__ float*)fc2_w);
        fc2bGm_.SetGlobalBuffer((__gm__ float*)fc2_b);
        fc3wGm_.SetGlobalBuffer((__gm__ float*)fc3_w);
        fc3bGm_.SetGlobalBuffer((__gm__ float*)fc3_b);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        batch_ = tiling.batch;
        numClasses_ = tiling.numClasses;
        conv1OC_ = tiling.conv1OC;
        conv1IC_ = tiling.conv1IC;
        conv1KH_ = tiling.conv1KH;
        conv1KW_ = tiling.conv1KW;
        conv2OC_ = tiling.conv2OC;
        conv2IC_ = tiling.conv2IC;
        conv2KH_ = tiling.conv2KH;
        conv2KW_ = tiling.conv2KW;
        fc1In_ = tiling.fc1In;
        fc1Out_ = tiling.fc1Out;
        fc2In_ = tiling.fc2In;
        fc2Out_ = tiling.fc2Out;
        fc3In_ = tiling.fc3In;
        fc3Out_ = tiling.fc3Out;

        // Derived sizes
        conv1OutH_ = IMG_H - conv1KH_ + 1; // 28
        conv1OutW_ = IMG_W - conv1KW_ + 1; // 28
        pool1OutH_ = conv1OutH_ / POOL_S;   // 14
        pool1OutW_ = conv1OutW_ / POOL_S;
        conv2OutH_ = pool1OutH_ - conv2KH_ + 1; // 10
        conv2OutW_ = pool1OutW_ - conv2KW_ + 1; // 10
        pool2OutH_ = conv2OutH_ / POOL_S;   // 5
        pool2OutW_ = conv2OutW_ / POOL_S;

        // Allocate local buffers for one image
        // x_local: 1x32x32 = 1024 floats
        // conv1_out: conv1OC x 28x28
        // conv1_relu: same
        // pool1_out: conv1OC x 14x14
        // conv2_out: conv2OC x 10x10
        // conv2_relu: same
        // pool2_out: conv2OC x 5x5 = 400 floats
        // fc1_out: fc1Out (120)
        // fc2_out: fc2Out (84)
        // final: numClasses (20)

        // We'll allocate via pipe_.InitBuffer with total byte size
        // For simplicity, allocate a large buffer and suballocate in Process
        uint32_t maxLocalBytes = 0;
        maxLocalBytes += IMG_H * IMG_W * sizeof(float); // input
        maxLocalBytes += conv1OC_ * conv1OutH_ * conv1OutW_ * sizeof(float); // conv1_out
        maxLocalBytes += conv1OC_ * pool1OutH_ * pool1OutW_ * sizeof(float); // pool1_out
        maxLocalBytes += conv2OC_ * conv2OutH_ * conv2OutW_ * sizeof(float); // conv2_out
        maxLocalBytes += conv2OC_ * pool2OutH_ * pool2OutW_ * sizeof(float); // pool2_out
        maxLocalBytes += fc1Out_ * sizeof(float);
        maxLocalBytes += fc2Out_ * sizeof(float);
        maxLocalBytes += numClasses_ * sizeof(float);
        pipe_.InitBuffer(localBuf_, 1, maxLocalBytes);
        localBufBase_ = localBuf_.Get<float>();
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx >= batch_) return;

        // Pointer offsets for sub-buffers
        float *base = (float*)localBufBase_;

        // Input image
        uint32_t imgSize = IMG_H * IMG_W;
        float *xLocal = base; base += imgSize;

        // Conv1 output
        uint32_t conv1OutSize = conv1OC_ * conv1OutH_ * conv1OutW_;
        float *conv1Out = base; base += conv1OutSize;

        // Pool1 output
        uint32_t pool1OutSize = conv1OC_ * pool1OutH_ * pool1OutW_;
        float *pool1Out = base; base += pool1OutSize;

        // Conv2 output
        uint32_t conv2OutSize = conv2OC_ * conv2OutH_ * conv2OutW_;
        float *conv2Out = base; base += conv2OutSize;

        // Pool2 output
        uint32_t pool2OutSize = conv2OC_ * pool2OutH_ * pool2OutW_;
        float *pool2Out = base; base += pool2OutSize;

        // FC1 input (flattened pool2)
        uint32_t fc1InSize = fc1In_; // 400
        float *fc1In = pool2Out; // reuse pool2 space as flat
        // FC1 output
        float *fc1Out = base; base += fc1Out_;

        // FC2 output
        float *fc2Out = base; base += fc2Out_;

        // Final output
        float *yLocal = base; base += numClasses_;

        // Step 1: Copy input image
        AscendC::DataCopy(xLocal, xGm_[blockIdx * imgSize], imgSize);

        // Step 2: Conv1 + ReLU
        Conv2D(xLocal, IMG_H, IMG_W, 1,
               conv1wGm_, conv1bGm_,
               conv1OC_, conv1IC_, conv1KH_, conv1KW_,
               conv1Out, conv1OutH_, conv1OutW_);
        Relu(conv1Out, conv1OutSize);

        // Step 3: MaxPool 2x2
        MaxPool2x2(conv1Out, conv1OC_, conv1OutH_, conv1OutW_, pool1Out);

        // Step 4: Conv2 + ReLU
        Conv2D(pool1Out, pool1OutH_, pool1OutW_, conv1OC_,
               conv2wGm_, conv2bGm_,
               conv2OC_, conv2IC_, conv2KH_, conv2KW_,
               conv2Out, conv2OutH_, conv2OutW_);
        Relu(conv2Out, conv2OutSize);

        // Step 5: MaxPool 2x2
        MaxPool2x2(conv2Out, conv2OC_, conv2OutH_, conv2OutW_, pool2Out);

        // Step 6: FC1 + ReLU (using flattened pool2)
        Linear(pool2Out, fc1wGm_, fc1bGm_, fc1In_, fc1Out_, fc1Out);
        Relu(fc1Out, fc1Out_);

        // Step 7: FC2 + ReLU
        Linear(fc1Out, fc2wGm_, fc2bGm_, fc2In_, fc2Out_, fc2Out);
        Relu(fc2Out, fc2Out_);

        // Step 8: FC3 (no ReLU)
        Linear(fc2Out, fc3wGm_, fc3bGm_, fc3In_, fc3Out_, yLocal);

        // Step 9: Copy output
        AscendC::DataCopy(yGm_[blockIdx * numClasses_], yLocal, numClasses_);
    }

private:
    // Helper: convolution (naive, no vectorization for clarity)
    __aicore__ inline void Conv2D(
        const float *in, int inH, int inW, int inC,
        const AscendC::GlobalTensor<float> &w,
        const AscendC::GlobalTensor<float> &b,
        int outC, int inC_, int kH, int kW,
        float *out, int outH, int outW)
    {
        // Each thread processes one output pixel? For simplicity, we loop.
        // Using scalar accesses - very slow but demonstrates the pattern.
        // In practice, use AscendC vector operations.
        for (int oc = 0; oc < outC; oc++) {
            for (int oh = 0; oh < outH; oh++) {
                for (int ow = 0; ow < outW; ow++) {
                    float sum = b[oc];
                    for (int ic = 0; ic < inC_; ic++) {
                        for (int kh = 0; kh < kH; kh++) {
                            for (int kw = 0; kw < kW; kw++) {
                                int ih = oh + kh;
                                int iw = ow + kw;
                                sum += in[ic * inH * inW + ih * inW + iw] *
                                       w[oc * (inC_ * kH * kW) + ic * kH * kW + kh * kW + kw];
                            }
                        }
                    }
                    out[oc * outH * outW + oh * outW + ow] = sum;
                }
            }
        }
    }

    __aicore__ inline void Relu(float *data, int size)
    {
        for (int i = 0; i < size; i++) {
            if (data[i] < 0.0f) data[i] = 0.0f;
        }
    }

    __aicore__ inline void MaxPool2x2(
        const float *in, int channels, int inH, int inW, float *out)
    {
        int outH = inH / 2;
        int outW = inW / 2;
        for (int c = 0; c < channels; c++) {
            for (int oh = 0; oh < outH; oh++) {
                for (int ow = 0; ow < outW; ow++) {
                    float maxVal = -1e38f;
                    for (int dh = 0; dh < 2; dh++) {
                        for (int dw = 0; dw < 2; dw++) {
                            int ih = oh * 2 + dh;
                            int iw = ow * 2 + dw;
                            float val = in[c * inH * inW + ih * inW + iw];
                            if (val > maxVal) maxVal = val;
                        }
                    }
                    out[c * outH * outW + oh * outW + ow] = maxVal;
                }
            }
        }
    }

    __aicore__ inline void Linear(
        const float *in,
        const AscendC::GlobalTensor<float> &w,
        const AscendC::GlobalTensor<float> &b,
        int inDim, int outDim, float *out)
    {
        for (int oc = 0; oc < outDim; oc++) {
            float sum = b[oc];
            for (int ic = 0; ic < inDim; ic++) {
                sum += in[ic] * w[oc * inDim + ic];
            }
            out[oc] = sum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> localBuf_;
    uintptr_t localBufBase_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> conv1wGm_, conv1bGm_;
    AscendC::GlobalTensor<float> conv2wGm_, conv2bGm_;
    AscendC::GlobalTensor<float> fc1wGm_, fc1bGm_;
    AscendC::GlobalTensor<float> fc2wGm_, fc2bGm_;
    AscendC::GlobalTensor<float> fc3wGm_, fc3bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_, numClasses_;
    uint32_t conv1OC_, conv1IC_, conv1KH_, conv1KW_;
    uint32_t conv2OC_, conv2IC_, conv2KH_, conv2KW_;
    uint32_t fc1In_, fc1Out_, fc2In_, fc2Out_, fc3In_, fc3Out_;
    uint32_t conv1OutH_, conv1OutW_;
    uint32_t pool1OutH_, pool1OutW_;
    uint32_t conv2OutH_, conv2OutW_;
    uint32_t pool2OutH_, pool2OutW_;
};

extern "C" __global__ __aicore__ void lenet5_custom(
    GM_ADDR x, GM_ADDR conv1_w, GM_ADDR conv1_b,
    GM_ADDR conv2_w, GM_ADDR conv2_b,
    GM_ADDR fc1_w, GM_ADDR fc1_b,
    GM_ADDR fc2_w, GM_ADDR fc2_b,
    GM_ADDR fc3_w, GM_ADDR fc3_b,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLenet5Custom op;
    op.Init(x, conv1_w, conv1_b, conv2_w, conv2_b,
            fc1_w, fc1_b, fc2_w, fc2_b, fc3_w, fc3_b,
            y, tiling_data);
    op.Process();
}
