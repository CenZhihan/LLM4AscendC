host_operator_src="""

#include "gemm_relu_divide_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_FLOATS = 48 * 1024;  // 192KB / 4B = 48K floats
static constexpr uint32_t ALIGN = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    GemmReluDivideCustomTilingData tiling;

    auto x_shape = context->GetInputShape(0)->GetOriginShape();
    auto w_shape = context->GetInputShape(1)->GetOriginShape();

    uint32_t M = x_shape.GetDim(0);
    uint32_t K = x_shape.GetDim(1);
    uint32_t N = w_shape.GetDim(0);  // out_features

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *divisor = attrs->GetAttrPointer<float>(0);

    // Tile parameters: use square-ish tiles, limit by UB.
    // Each tile requires: A (tileM x tileK), B (tileN x tileK), C (tileM x tileN)
    // Budget: (tileM*tileK + tileN*tileK + tileM*tileN) floats <= UB_BUDGET_FLOATS / 2 (for double buffering)
    uint32_t maxTile = static_cast<uint32_t>(sqrt(UB_BUDGET_FLOATS / 4));
    maxTile = AlignDown(maxTile, ALIGN);
    if (maxTile < ALIGN) maxTile = ALIGN;

    uint32_t tileM = std::min(maxTile, M);
    uint32_t tileN = std::min(maxTile, N);
    uint32_t tileK = std::min(maxTile, K);

    // Adjust to fit UB: we require 2 * (tileM*tileK + tileN*tileK + tileM*tileN) < UB_BUDGET_FLOATS
    while (2 * (tileM*tileK + tileN*tileK + tileM*tileN) > UB_BUDGET_FLOATS && tileM > ALIGN) {
        if (tileM > tileN) tileM = AlignDown(tileM / 2, ALIGN);
        else tileN = AlignDown(tileN / 2, ALIGN);
        tileK = std::min(maxTile, K);
    }

    // Block dimension = number of M tiles * number of N tiles, bounded by available cores (~24)
    uint32_t numMTiles = CeilDiv(M, tileM);
    uint32_t numNTiles = CeilDiv(N, tileN);
    uint32_t blockDim = numMTiles * numNTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);
    tiling.set_divisor(*divisor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // y = (M, N)
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, x_shape->GetDim(0));
    y_shape->SetDim(1, w_shape->GetDim(0));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"
#include "matmul_api.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmReluDivideCustom {
public:
    __aicore__ inline KernelGemmReluDivideCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                float divisor)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN; tileK_ = tileK;
        divisor_ = divisor;
        invDivisor_ = 1.0f / divisor_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M * K);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N * K);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M * N);

        // Initialize MatMul object with default cube unit
        matMul_.SetOrgShape(FORMAT_ND, FORMAT_ND, FORMAT_ND);
        matMul_.SetOrgShape(FORMAT_ND, FORMAT_ND, FORMAT_ND);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        // Distribute M tiles and N tiles across cores
        uint32_t numMTiles = (M_ + tileM_ - 1) / tileM_;
        uint32_t numNTiles = (N_ + tileN_ - 1) / tileN_;
        uint32_t totalTiles = numMTiles * numNTiles;

        for (uint32_t tid = blockIdx; tid < totalTiles; tid += blockNum) {
            uint32_t mi = tid / numNTiles; // M tile index
            uint32_t ni = tid % numNTiles; // N tile index

            uint32_t mStart = mi * tileM_;
            uint32_t mEnd = min(mStart + tileM_, M_);
            uint32_t actualM = mEnd - mStart;

            uint32_t nStart = ni * tileN_;
            uint32_t nEnd = min(nStart + tileN_, N_);
            uint32_t actualN = nEnd - nStart;

            // Compute y = x * w^T + b for this tile
            MatMul<MatMulTYPE> mm;   // cube matmul
            mm.SetTensorA(xGm_[mStart * K_]); // A: (actualM, K)
            mm.SetTensorB(wGm_[nStart * K_]); // B: (actualN, K) stored row-major
            mm.SetBias(bGm_[nStart]);          // bias per output column
            mm.SetTransposeB(true);            // y = A * B^T + bias
            mm.SetResult(yGm_[mStart * N_ + nStart]); // C: (actualM, N)

            // Set strides: A: K, B: K (since stored as row-major, after transpose we need to read by columns? Actually SetTransposeB expects B stored as (K, N) but we have (N, K). We'll use default.
            // Better to use SetInputStride for A: K, B: K, result stride: N
            mm.SetInputStride(MatMulType::A, K_);
            mm.SetInputStride(MatMulType::B, K_);
            mm.SetResultStride(N_);

            // Set tile dimensions
            mm.SetShape(actualM, actualN, K_);
            mm.SetTile(tileM_, tileN_, tileK_);
            mm.IterateAll();

            // Apply ReLU and division on the result tile
            LocalTensor<float> yLocal = mm.GetResultTensor();
            Maxs(yLocal, yLocal, 0.0f, actualM * actualN);
            Muls(yLocal, yLocal, invDivisor_, actualM * actualN);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    MatMul<MatMulTYPE> matMul_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
    uint32_t tileK_ {0};
    float divisor_ {1.0f};
    float invDivisor_ {1.0f};
};

extern "C" __global__ __aicore__ void gemm_relu_divide_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                                               GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmReluDivideCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.divisor);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'gemm_relu_divide_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, divisor):
        super(ModelNew, self).__init__()
        self.in_features = in_features
        self.out_features = out_features
        self.divisor = divisor

    def forward(self, x, w, b, divisor_tensor):

        d = float(divisor_tensor.detach().cpu().item())
        fn = getattr(custom_ops_lib, _FN)
