#include "conv2d_min_tanh_tanh_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Conv2dMinTanhTanhCustomTilingData tiling;
    // Input shapes: x (N, C, H, W), weight (OC, IC, kH, kW), optional bias (OC)
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    // x: N, C, H, W
    uint32_t N = xShape->GetDim(0);
    uint32_t C = xShape->GetDim(1);
    uint32_t H = xShape->GetDim(2);
    uint32_t W = xShape->GetDim(3);
    // weight: OC, IC, kH, kW
    uint32_t OC = wShape->GetDim(0);
    uint32_t kH = wShape->GetDim(2);
    uint32_t kW = wShape->GetDim(3);
    // conv with stride=1, pad=0, dilation=1
    uint32_t Ho = H - kH + 1;
    uint32_t Wo = W - kW + 1;
    uint32_t totalPixels = N * Ho * Wo;  // output elements (each pixel is single channel after min)

    // Determine tile size: each output pixel requires processing all OC conv values.
    // UB budget: we need 1 buffer for input window (kH*kW*C floats), 1 for weight (OC*kH*kW*C ? no, we load per OC)
    // For simplicity, we allocate a small fixed window of 1024 floats per tile.
    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * 1024); // conservative
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tilePixels = maxTile;
    if (totalPixels < tilePixels)
        tilePixels = AlignDown(totalPixels, ALIGN_ELEMS);
    if (tilePixels == 0) tilePixels = 1;

    uint32_t numTiles = CeilDiv(totalPixels, tilePixels);
    uint32_t blockDim = numTiles;
    if (blockDim < 4) blockDim = 4;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_batch(N);
    tiling.set_inC(C);
    tiling.set_outC(OC);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_totalPixels(totalPixels);
    tiling.set_tilePixels(tilePixels);
    tiling.set_blockDim(blockDim);
    tiling.set_dummySlope(0.01f);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
