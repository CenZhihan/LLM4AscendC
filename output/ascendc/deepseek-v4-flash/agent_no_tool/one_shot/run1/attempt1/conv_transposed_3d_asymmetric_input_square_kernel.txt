#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                ConvTransposed3dAsymmetricInputSquareKernelCustomTilingData* tiling)
    {
        totalLen_ = tiling->totalLength;
        tileLen_ = tiling->tileLength;
        N_ = tiling->N;
        C_in_ = tiling->C_in;
        D_ = tiling->D;
        H_ = tiling->H;
        W_ = tiling->W;
        C_out_ = tiling->C_out;
        kD_ = tiling->kD;
        kH_ = tiling->kH;
        kW_ = tiling->kW;
        D_out_ = tiling->D_out;
        H_out_ = tiling->H_out;
        W_out_ = tiling->W_out;
        strideD_ = tiling->strideD;
        strideH_ = tiling->strideH;
        strideW_ = tiling->strideW;
        padD_ = tiling->padD;
        padH_ = tiling->padH;
        padW_ = tiling->padW;
        outputPadD_ = tiling->outputPadD;
        outputPadH_ = tiling->outputPadH;
        outputPadW_ = tiling->outputPadW;
        dilationD_ = tiling->dilationD;
        dilationH_ = tiling->dilationH;
        dilationW_ = tiling->dilationW;
        groups_ = tiling->groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t stride = blockNum * tileLen_;
        uint32_t start = blockIdx * tileLen_;

        for (uint32_t offset = start; offset < totalLen_; offset += stride) {
            uint32_t curLen = totalLen_ - offset;
            if (curLen > tileLen_) curLen = tileLen_;

            // Precompute output strides for index decomposition
            int32_t dims_out[5] = {N_, C_out_, D_out_, H_out_, W_out_};
            int32_t strides_out[5];
            strides_out[4] = 1;
            for (int32_t j = 3; j >= 0; --j) {
                strides_out[j] = strides_out[j+1] * dims_out[j+1];
            }
            // Precompute input strides
            int32_t strides_in[5];
            strides_in[4] = 1;
            int32_t dims_in[5] = {N_, C_in_, D_, H_, W_};
            for (int32_t j = 3; j >= 0; --j) {
                strides_in[j] = strides_in[j+1] * dims_in[j+1];
            }
            // Weight strides: [C_in, C_out, kD, kH, kW]
            int32_t weight_strides[5];
            int32_t wdims[5] = {C_in_, C_out_, kD_, kH_, kW_};
            weight_strides[4] = 1;
            for (int32_t j = 3; j >= 0; --j) {
                weight_strides[j] = weight_strides[j+1] * wdims[j+1];
            }

            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t linearIdx = offset + i;

                // Decompose linear index to output indices
                int32_t n, c_out, d_out, h_out, w_out;
                uint32_t rem = linearIdx;
                n = rem / strides_out[0];
                rem %= strides_out[0];
                c_out = rem / strides_out[1];
                rem %= strides_out[1];
                d_out = rem / strides_out[2];
                rem %= strides_out[2];
                h_out = rem / strides_out[3];
                rem %= strides_out[3];
                w_out = rem;

                float sum = 0.0f;
                // Loop over input channels (groups=1)
                for (int32_t c_in = 0; c_in < C_in_; c_in++) {
                    // Loop over kernel positions
                    for (int32_t kd = 0; kd < kD_; kd++) {
                        for (int32_t kh = 0; kh < kH_; kh++) {
                            for (int32_t kw = 0; kw < kW_; kw++) {
                                // Compute input spatial indices
                                int32_t d_in = d_out - (kd * dilationD_ - padD_);
                                int32_t h_in = h_out - (kh * dilationH_ - padH_);
                                int32_t w_in = w_out - (kw * dilationW_ - padW_);

                                // Check input bounds
                                if (d_in >= 0 && d_in < D_ &&
                                    h_in >= 0 && h_in < H_ &&
                                    w_in >= 0 && w_in < W_) {

                                    uint32_t x_offset = n * strides_in[0] +
                                                        c_in * strides_in[1] +
                                                        d_in * strides_in[2] +
                                                        h_in * strides_in[3] +
                                                        w_in;
                                    uint32_t w_offset = c_in * weight_strides[0] +
                                                        c_out * weight_strides[1] +
                                                        kd * weight_strides[2] +
                                                        kh * weight_strides[3] +
                                                        kw;
                                    sum += xGm_[x_offset] * wGm_[w_offset];
                                }
                            }
                        }
                    }
                }
                yGm_[linearIdx] = sum;
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    int32_t N_, C_in_, D_, H_, W_;
    int32_t C_out_, kD_, kH_, kW_;
    int32_t D_out_, H_out_, W_out_;
    int32_t strideD_, strideH_, strideW_;
    int32_t padD_, padH_, padW_;
    int32_t outputPadD_, outputPadH_, outputPadW_;
    int32_t dilationD_, dilationH_, dilationW_;
    int32_t groups_;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dAsymmetricInputSquareKernelCustom op;
    op.Init(x, weight, y, &tiling_data);
    op.Process();
}
