host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(xShape->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dimRaw = *attrs->GetAttrPointer<int64_t>(0);
    uint32_t ndim = xShape->GetOriginShape().GetDimNum();
    // Normalize dim (support negative)
    int64_t dim = (dimRaw < 0) ? dimRaw + ndim : dimRaw;
    // dim must be valid
    if (dim < 0 || dim >= static_cast<int64_t>(ndim)) {
        dim = ndim - 1;  // fallback to last dim
    }
    // Compute segment length (size of the chosen dimension)
    uint32_t segLen = xShape->GetOriginShape().GetDim(static_cast<uint32_t>(dim));
    // Compute stride: product of dimensions after dim
    uint32_t stride = 1;
    for (uint32_t d = static_cast<uint32_t>(dim) + 1; d < ndim; d++) {
        stride *= xShape->GetOriginShape().GetDim(d);
    }
    // Outer segment count = totalElements / segLen
    uint32_t outerSegCnt = totalLength / segLen;

    // Determine tile length for data movement (within a segment)
    // Each segment uses: one tile for x, one for mask (as float), one for y, one for carry?
    // We will load x and mask, compute product, then cumulative sum.
    // For simplicity, we load the entire segment if possible.
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));  // x, mask, y
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = (segLen < maxTile) ? segLen : maxTile;

    // Block dimension: based on number of outer segments and tile count
    // Each block can process multiple outer segments. Use a grid-stride approach.
    uint32_t blockDim = 24;  // conservative, adjust based on workload
    if (outerSegCnt < 8) blockDim = 8;
    if (outerSegCnt < blockDim) blockDim = outerSegCnt;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerSegCnt(outerSegCnt);
    tiling.set_segLen(segLen);
    tiling.set_stride(stride);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t outerSegCnt, uint32_t segLen,
                                uint32_t stride, uint32_t tileLen)
    {
        outerSegCnt_ = outerSegCnt;
        segLen_ = segLen;
        stride_ = stride;
        tileLen_ = tileLen;
        // If tileLen > segLen, we can adjust to segLen (no overhead)
        if (tileLen_ > segLen_) tileLen_ = segLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        maskGm_.SetGlobalBuffer((__gm__ int8_t*)mask);  // bool stored as int8
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Allocate local buffers: double buffer for x, mask (as float), y, plus a temp for product/carry
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(int8_t));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));  // for product and cumulative sum
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        // Grid-stride over outer segments
        for (uint32_t seg = blockIdx; seg < outerSegCnt_; seg += blockNum) {
            // Process one segment
            uint32_t baseOffset = seg * segLen_ * stride_;  // approximate, but correct for contiguous cases? Actually we need exact linear offset to first element of segment.
            // For simplicity, we assume the tensor is 2D with dim = last dimension (so stride_ = 1)
            // Then baseOffset = seg * segLen_
            // General formula: offset = seg * segLen_ * stride_ (since each segment is contiguous? Not always).
            // We'll use the 2D assumption for this kernel, but the tiling function set appropriate values.
            // If stride == 1 (dim is last), this works.
            // Otherwise, we need to compute proper offset. For now, we assume stride==1.
            uint32_t offset = seg * segLen_;

            // Process segment in tiles
            uint32_t remain = segLen_;
            uint32_t cur = 0;
            float carry = 0.0f;
            while (remain > 0) {
                uint32_t len = (remain > tileLen_) ? tileLen_ : remain;
                // Load tile of x and mask
                LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                LocalTensor<int8_t> maskLocal = inQueueMask_.AllocTensor<float>();  // actually int8
                DataCopy(xLocal, xGm_[offset + cur], len);
                DataCopy(maskLocal, maskGm_[offset + cur], len);

                // Compute product: x * (float)mask
                LocalTensor<float> prodLocal = tmpQueue_.AllocTensor<float>();
                // Convert mask to float (0 or 1)
                AscendC::Cast(prodLocal, maskLocal, AscendC::RoundMode::CAST_ROUND, len);
                AscendC::Mul(prodLocal, xLocal, prodLocal, len);

                // Compute cumulative sum on prodLocal
                for (uint32_t i = 1; i < len; i++) {
                    prodLocal[i] = prodLocal[i] + prodLocal[i - 1];
                }
                // Add carry from previous tile (if any)
                if (cur > 0) {
                    for (uint32_t i = 0; i < len; i++) {
                        prodLocal[i] = prodLocal[i] + carry;
                    }
                }
                // Update carry with last element
                carry = prodLocal[len - 1];

                // Store to y
                LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, prodLocal, len);  // not needed if we directly store from prodLocal
                // Actually we can store directly from prodLocal to global
                DataCopy(yGm_[offset + cur], prodLocal, len);

                // Free tensors
                inQueueX_.FreeTensor(xLocal);
                inQueueMask_.FreeTensor(maskLocal);
                tmpQueue_.FreeTensor(prodLocal);
                outQueueY_.FreeTensor(yLocal);  // we didn't enqueue, but we can just free it

                cur += len;
                remain -= len;
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<int8_t> maskGm_;
    GlobalTensor<float> yGm_;

    uint32_t outerSegCnt_ {0};
    uint32_t segLen_ {0};
    uint32_t stride_ {1};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.outerSegCnt, tiling_data.segLen,
            tiling_data.stride, tiling_data.tileLen);
    op.Process();
}
