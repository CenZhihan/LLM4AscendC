static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

#include "kernel_operator.h"
#include "matmul/matmul.h"

using namespace AscendC;

class KernelMatmulWithTransposedBoth {
public:
    __aicore__ inline KernelMatmulWithTransposedBoth() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t blockM, uint32_t blockN, uint32_t blockK,
                                uint32_t numBlocksM, uint32_t numBlocksN,
                                uint32_t blockDim)
    {
        M_ = M; N_ = N; K_ = K;
        blockM_ = blockM; blockN_ = blockN;
        numBlocksM_ = numBlocksM; numBlocksN_ = numBlocksN;

        A_gm_.SetGlobalBuffer((__gm__ float*)A, M_ * K_);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, K_ * N_);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, M_ * N_);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = GetBlockIdx();
        if (blockIdx >= numBlocksM_ * numBlocksN_) return;

        uint32_t i = blockIdx / numBlocksN_;
        uint32_t j = blockIdx % numBlocksN_;

        uint32_t baseM = i * blockM_;
        uint32_t baseN = j * blockN_;
        uint32_t curM = (baseM + blockM_ > M_) ? (M_ - baseM) : blockM_;
        uint32_t curN = (baseN + blockN_ > N_) ? (N_ - baseN) : blockN_;

        // MatMul object
        MatMul<float> mm;
        mm.SetTransposeA(true);
        mm.SetTransposeB(true);

        // Set global tensors with strides
        // A: physical shape (K, M) row-major, stride = M
        // B: physical shape (N, K) row-major, stride = K
        // Output: physical shape (M, N) row-major, stride = N
        mm.SetGlobalTensorA(A_gm_, M_, K_, M_);      // logical rows, logical cols, stride
        mm.SetGlobalTensorB(B_gm_, K_, N_, K_);
        mm.SetGlobalTensorC(C_gm_, M_, N_, N_);

        // Set output tile size
        mm.SetTileSize(curM, curN);

        // Set output offset (row-major)
        uint32_t offset = baseM * N_ + baseN;
        mm.SetOutputOffset(offset);

        // Execute
        mm.Process();
    }

private:
    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t M_, N_, K_;
    uint32_t blockM_, blockN_;
    uint32_t numBlocksM_, numBlocksN_;
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedBoth op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,

            tiling_data.numBlocksM, tiling_data.numBlocksN,
            tiling_data.blockDim);
    op.Process();
}
