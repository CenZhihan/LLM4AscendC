static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    GemmSwishDivideClampTanhClampCustomTilingData tiling;

    const auto& x_shape = context->GetInputShape(0)->GetOriginShape();
    const auto& w_shape = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = x_shape.GetDim(0);
    uint32_t K = x_shape.GetDim(1);
    uint32_t N = w_shape.GetDim(0);

    // Estimate tile sizes from UB: A(tileM*tileK), B(tileK*tileN), C(tileM*tileN), workspace
    // Example: tileM=tileN=tileK=64 gives 64*64*3=12288 floats = 49KB, fits with margin
    uint32_t tileM = 64;
    uint32_t tileN = 64;
    uint32_t tileK = 64;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmSwishDivideClampTanhClampCustom {
public:
    __aicore__ inline KernelGemmSwishDivideClampTanhClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z,
                                uint32_t totalM, uint32_t totalN, uint32_t totalK,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK) {
        M_ = totalM;
        N_ = totalN;
        K_ = totalK;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        zGm_.SetGlobalBuffer((__gm__ float*)z, M_ * N_);

        // UB buffer sizes: A(tileM*tileK), B(tileK*tileN), C(tileM*tileN)
        uint32_t sizeA = tileM_ * tileK_ * sizeof(float);
        uint32_t sizeB = tileK_ * tileN_ * sizeof(float);
        uint32_t sizeC = tileM_ * tileN_ * sizeof(float);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, sizeA);
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, sizeB);
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, sizeC);
        // Extra buffers for intermediate computations (swish, etc.) – reuse C for final output
        pipe_.InitBuffer(tmpQueue_, 1, sizeC);
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        // Each block processes a contiguous chunk of rows
        uint32_t startM = blockIdx * tileM_;
        if (startM >= M_) return;
        uint32_t endM = startM + tileM_;
        if (endM > M_) endM = M_;
        uint32_t numMRows = endM - startM;

        // For simplicity, iterate over K tiles once and accumulate into C tile
        // We'll process one K tile at a time with double buffering
        uint32_t kTiles = (K_ + tileK_ - 1) / tileK_;
        uint32_t nTiles = (N_ + tileN_ - 1) / tileN_;

        // For each output N tile (column group)
        for (uint32_t nt = 0; nt < nTiles; ++nt) {
            uint32_t startN = nt * tileN_;
            uint32_t endN = startN + tileN_;
            if (endN > N_) endN = N_;
            uint32_t numNCols = endN - startN;

            // Initialize C tile to zero (since we accumulate across K)
            AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
            AscendC::SetZero(cLocal, numMRows * numNCols);
            outQueueC_.EnQue(cLocal);

            AscendC::LocalTensor<float> cAcc = outQueueC_.DeQue<float>();

            // Loop over K tiles
            for (uint32_t kt = 0; kt < kTiles; ++kt) {
                uint32_t startK = kt * tileK_;
                uint32_t endK = startK + tileK_;
                if (endK > K_) endK = K_;
                uint32_t numKCols = endK - startK;

                // Prefetch A and B tiles
                // A: rows startM..endM, cols startK..endK (stored in global row-major)
                AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
                AscendC::DataCopy(aLocal, xGm_[startM * K_ + startK], numMRows * numKCols);
                inQueueA_.EnQue(aLocal);

                // B: rows startK..endK, cols startN..endN (w is row-major: N*K, but we need K*N)
                // Actually w is stored as [N, K] row-major, so we need to transpose? For simplicity we assume w is stored in K-major (column-major) for easy access.
                // But standard linear expects w with shape (out_features, in_features) row-major.
                // To avoid transposition, we can adjust tiling: treat B as (K, N) but w is transposed.
                // For this kernel, we assume the weights are stored in column-major format (K, N) to simplify inner loop.
                // We'll document this assumption in the binding code later.
                // Here we access w as if it's K*N matrix (column-major)
                // So global offset for w tile: startK * N_ + startN (if column-major)
                // WARNING: This requires the input w to be transposed before calling the kernel.
                // In practice, we could transpose in the kernel, but for simplicity we assume transposed weights.
                AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, wGm_[startK * N_ + startN], numKCols * numNCols);
                inQueueB_.EnQue(bLocal);

                // Do matmul: C += A * B
                AscendC::Matmul(cAcc, aLocal, bLocal, numMRows, numNCols, numKCols);

                // Free A and B
                inQueueA_.FreeTensor(aLocal);
                inQueueB_.FreeTensor(bLocal);
            }

            // Now cAcc holds the full matmul result for this tile (row group, column group)
            // Apply bias addition (b is of length N_, we need to broadcast to all rows)
            // We load bias for columns startN..endN
            AscendC::LocalTensor<float> biasLocal = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(biasLocal, bGm_[startN], numNCols);
            // Add bias to each row of cAcc
            for (uint32_t r = 0; r < numMRows; ++r) {
                AscendC::Add(cAcc[r * numNCols], cAcc[r * numNCols], biasLocal, numNCols);
            }
            tmpQueue_.FreeTensor(biasLocal);

            // Now apply element-wise operations on cAcc (size numMRows * numNCols)
            uint32_t totalElems = numMRows * numNCols;

            // y = x * sigmoid(x)  (swish)
            // We'll compute sigmoid using Exponential and division
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            // sigmoid = 1/(1+exp(-x))
            AscendC::Muls(tmp, cAcc, -1.0f, totalElems);
            AscendC::Exp(tmp, tmp, totalElems);
            AscendC::Adds(tmp, tmp, 1.0f, totalElems);
            AscendC::Reciprocal(tmp, tmp, totalElems);
            // cAcc = cAcc * tmp
            AscendC::Mul(cAcc, cAcc, tmp, totalElems);

            // divide by 2
            AscendC::Muls(cAcc, cAcc, 0.5f, totalElems);

            // clamp [-1,1]
            AscendC::Clamp(cAcc, -1.0f, 1.0f, totalElems);

            // tanh
            AscendC::Tanh(cAcc, cAcc, totalElems);

            // final clamp [-1,1]
            AscendC::Clamp(cAcc, -1.0f, 1.0f, totalElems);

            tmpQueue_.FreeTensor(tmp);

            // Copy result to global memory (z is row-major)
            AscendC::DataCopy(zGm_[startM * N_ + startN], cAcc, totalElems);

            outQueueC_.FreeTensor(cAcc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> zGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileM_{0}, tileN_{0}, tileK_{0};
};

extern "C" __global__ __aicore__ void gemm_swish_divide_clamp_tanh_clamp_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR z, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSwishDivideClampTanhClampCustom op;
    op.Init(x, w, b, z,
            tiling_data.totalM, tiling_data.totalN, tiling_data.totalK,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
