static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMultiplyLeakyreluCustomTilingData tiling;

    // Input shapes: x (M, K), w (N, K), b (N,), multiplier (1,), negative_slope (1,)
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    uint32_t M = xShape->GetDim(0);
    uint32_t K = xShape->GetDim(1);
    uint32_t N = wShape->GetDim(0);

    // Read scalar tensors as attributes (converted at host side for simplicity)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    // In practice, scalar tensors are inputs; we extract values here.
    // For this example we assume attrs contain multiplier and negative_slope.
    float multiplier = *(attrs->GetAttrPointer<float>(0));
    float negativeSlope = *(attrs->GetAttrPointer<float>(1));

    // Tiling: aim for tileM = 512, tileN = 512, tileK = 512 (adjust to UB)
    // Each tile computation needs: tileM*tileK + tileN*tileK + tileM*tileN elements
    uint32_t tileM = 512;
    uint32_t tileN = 512;
    uint32_t tileK = 512;

    // Rough UB check: (tileM*tileK + tileN*tileK + tileM*tileN) * 4 bytes
    uint32_t ubNeeded = (tileM*tileK + tileN*tileK + tileM*tileN) * sizeof(float);
    if (ubNeeded > UB_BUDGET_BYTES) {
        // Scale down proportionally
        float ratio = (float)UB_BUDGET_BYTES / ubNeeded;
        tileM = (uint32_t)(tileM * sqrtf(ratio));
        tileN = (uint32_t)(tileN * sqrtf(ratio));
        tileK = (uint32_t)(tileK * sqrtf(ratio));
        tileM = AlignDown(tileM, ALIGN_ELEMS);
        tileN = AlignDown(tileN, ALIGN_ELEMS);
        tileK = AlignDown(tileK, ALIGN_ELEMS);
        if (tileM < 16) tileM = 16;
        if (tileN < 16) tileN = 16;
        if (tileK < 16) tileK = 16;
    }

    // Block number based on tiles along M and N dimensions
    uint32_t tilesM = (M + tileM - 1) / tileM;
    uint32_t tilesN = (N + tileN - 1) / tileN;
    uint32_t blockNum = tilesM * tilesN;
    if (blockNum < 8) blockNum = 8;
    if (blockNum > 24) blockNum = 24;

    context->SetBlockDim(blockNum);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockNum(blockNum);
    tiling.set_multiplier(multiplier);
    tiling.set_negativeSlope(negativeSlope);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

        const uint32_t blockNum = AscendC::GetBlockNum();

        // Distribute (tileM,tileN) blocks among blocks using a 2D block mapping
        uint32_t tilesM = (M_ + tileM_ - 1) / tileM_;
        uint32_t tilesN = (N_ + tileN_ - 1) / tileN_;
        // Each block processes a subset of tiles along M and N dimensions.
        // For simplicity, we assign a single (tileM, tileN) output tile per block.
        // This is inefficient but demonstrates the pattern.
        uint32_t tileIdx = blockIdx;
        if (tileIdx >= tilesM * tilesN) return;
        uint32_t tM = tileIdx / tilesN;
        uint32_t tN = tileIdx % tilesN;
        uint32_t rowStart = tM * tileM_;
        uint32_t colStart = tN * tileN_;
        uint32_t curM = (rowStart + tileM_ > M_) ? (M_ - rowStart) : tileM_;
        uint32_t curN = (colStart + tileN_ > N_) ? (N_ - colStart) : tileN_;

        // Process K dimension in tiles
        for (uint32_t kOff = 0; kOff < K_; kOff += tileK_) {
            uint32_t curK = (kOff + tileK_ > K_) ? (K_ - kOff) : tileK_;
            // Load tile of x: shape (curM, curK) at position (rowStart, kOff)
            CopyInX(rowStart, kOff, curM, curK);
            // Load tile of w: shape (curN, curK) at position (colStart, kOff)
            CopyInW(colStart, kOff, curN, curK);
            // Compute partial product and accumulate into output tile
            CopyInBias(colStart, curN); // load bias once per output tile (could be optimized)
            Compute(curM, curN, curK);
        }
        // Finalize: apply multiplier and leaky relu
        ApplyActivation(curM, curN);
        CopyOut(rowStart, colStart, curM, curN);
