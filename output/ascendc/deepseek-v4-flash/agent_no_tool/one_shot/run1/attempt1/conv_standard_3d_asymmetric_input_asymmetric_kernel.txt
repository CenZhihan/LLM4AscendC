    uint32_t strideW = static_cast<uint32_t>(stride[2]);
    uint32_t padD    = static_cast<uint32_t>(padding[0]);
    uint32_t padH    = static_cast<uint32_t>(padding[1]);
    uint32_t padW    = static_cast<uint32_t>(padding[2]);

    uint32_t D_out = (D + 2 * padD - dilationD * (kD - 1) - 1) / strideD + 1;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv3dCustom {
public:
    __aicore__ inline KernelConv3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim,
                                uint32_t N, uint32_t outC, uint32_t inC,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t padD, uint32_t padH, uint32_t padW,
                                uint32_t dilationD, uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        N_ = N;
        outC_ = outC;
        inC_ = inC;
        D_ = D;
        H_ = H;
        W_ = W;
        kD_ = kD;
        kH_ = kH;
        kW_ = kW;
        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        padD_ = padD;
        padH_ = padH;
        padW_ = padW;
        dilationD_ = dilationD;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Pre-compute output spatial strides for indexing
        D_out_ = (D_ + 2 * padD_ - dilationD_ * (kD_ - 1) - 1) / strideD_ + 1;
        H_out_ = (H_ + 2 * padH_ - dilationH_ * (kH_ - 1) - 1) / strideH_ + 1;
        W_out_ = (W_ + 2 * padW_ - dilationW_ * (kW_ - 1) - 1) / strideW_ + 1;

        // Strides in input
        inStrideD_ = H_ * W_;
        inStrideH_ = W_;
        inStrideW_ = 1;
        // Channel stride for input
        inCStride_ = D_ * H_ * W_;
        // Batch stride
        batchStride_ = inC_ * D_ * H_ * W_;

        // Weight strides: outC, inC/groups, kD, kH, kW
        wOutCStride_ = (inC_ / groups_) * kD_ * kH_ * kW_;
        wInCStride_ = kD_ * kH_ * kW_;
        wDStride_ = kH_ * kW_;
        wHStride_ = kW_;
        wWStride_ = 1;

        // Output strides
        outStrideN_ = outC_ * D_out_ * H_out_ * W_out_;
        outStrideC_ = D_out_ * H_out_ * W_out_;
        outStrideD_ = H_out_ * W_out_;
        outStrideH_ = W_out_;
        outStrideW_ = 1;
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = blockDim_; // use actual blockDim from tiling
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Process each output element in this tile
            for (uint32_t i = 0; i < curLen; ++i) {
                uint32_t flatIdx = base + i;
                // Convert flattened index to (n, c, d, h, w)
                uint32_t n = flatIdx / outStrideN_;
                uint32_t rem = flatIdx % outStrideN_;
                uint32_t c = rem / outStrideC_;
                rem = rem % outStrideC_;
                uint32_t d = rem / outStrideD_;
                rem = rem % outStrideD_;
                uint32_t h = rem / outStrideH_;
                uint32_t w = rem % outStrideW_;

                float acc = 0.0f;

                // Determine input channel group
                uint32_t cg = c / (outC_ / groups_); // group index
                uint32_t inCStart = cg * (inC_ / groups_);
                uint32_t inCEnd = inCStart + (inC_ / groups_);

                // Kernel loops
                for (uint32_t kd = 0; kd < kD_; ++kd) {
                    int32_t inD = static_cast<int32_t>(d * strideD_ + kd * dilationD_ - padD_);
                    if (inD < 0 || inD >= static_cast<int32_t>(D_)) continue;
                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        int32_t inH = static_cast<int32_t>(h * strideH_ + kh * dilationH_ - padH_);
                        if (inH < 0 || inH >= static_cast<int32_t>(H_)) continue;
                        for (uint32_t kw = 0; kw < kW_; ++kw) {
                            int32_t inW = static_cast<int32_t>(w * strideW_ + kw * dilationW_ - padW_);
                            if (inW < 0 || inW >= static_cast<int32_t>(W_)) continue;

                            uint32_t inputOffset = n * batchStride_ +
                                                   inCStart * inCStride_ +
                                                   inD * inStrideD_ +
                                                   inH * inStrideH_ +
                                                   inW;
                            uint32_t weightOffset = c * wOutCStride_ +
                                                    0 * wInCStride_ + // groups handled by inCStart loop
                                                    kd * wDStride_ +
                                                    kh * wHStride_ +
                                                    kw;

                            // Accumulate over input channels within group
                            for (uint32_t ic = 0; ic < (inC_ / groups_); ++ic) {
                                float xVal = xGm_[inputOffset + ic * inCStride_];
                                float wVal = wGm_[weightOffset + ic * wInCStride_];
                                acc += xVal * wVal;
                            }
                        }
                    }
                }
                yGm_[flatIdx] = acc;
            }

            base += strideElems;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
    uint32_t N_ {0}, outC_ {0}, inC_ {0};
    uint32_t D_ {0}, H_ {0}, W_ {0};
    uint32_t kD_ {0}, kH_ {0}, kW_ {0};
    uint32_t strideD_ {1}, strideH_ {1}, strideW_ {1};
    uint32_t padD_ {0}, padH_ {0}, padW_ {0};
    uint32_t dilationD_ {1}, dilationH_ {1}, dilationW_ {1};
    uint32_t groups_ {1};

    uint32_t D_out_ {0}, H_out_ {0}, W_out_ {0};
    uint32_t inStrideD_ {0}, inStrideH_ {0}, inStrideW_ {0};
    uint32_t inCStride_ {0}, batchStride_ {0};
    uint32_t wOutCStride_ {0}, wInCStride_ {0}, wDStride_ {0}, wHStride_ {0}, wWStride_ {0};
    uint32_t outStrideN_ {0}, outStrideC_ {0}, outStrideD_ {0}, outStrideH_ {0}, outStrideW_ {0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dCustom op;
    op.Init(x, weight, y,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim,
            tiling_data.N, tiling_data.outC, tiling_data.inC,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW,
            tiling_data.dilationD, tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups);
    op.Process();
}
