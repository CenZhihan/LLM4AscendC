static constexpr uint32_t ALIGN_ELEMS = 16;               // byte alignment of float
static constexpr uint32_t MAX_TILE_ELEMS = 8192;          // safety limit

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Conv2dScalingMinCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *bias_shape = context->GetOptionalInputShape(2);

    auto x_dims = x_shape->GetOriginShape().GetDims();
    auto w_dims = w_shape->GetOriginShape().GetDims();

    uint32_t N = x_dims[0];
    uint32_t C_in = x_dims[1];
    uint32_t H = x_dims[2];
    uint32_t W = x_dims[3];
    uint32_t C_out = w_dims[0];
    uint32_t K_h = w_dims[2];
    uint32_t K_w = w_dims[3];
    uint32_t H_out = H - K_h + 1;
    uint32_t W_out = W - K_w + 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scalePtr = attrs->GetAttrPointer<float>(0);
    float scaleFactor = (scalePtr != nullptr) ? *scalePtr : 1.0f;

    // Estimate tile size: each output pixel requires loading tile of input and weights.
    // We'll tile over spatial output pixels (H_out * W_out).
    // For each output pixel, we need K_h*K_w*C_in input elements and C_out*K_h*K_w weight elements.
    // UB needed for input slice, weight slice, intermediate accumulation, bias, scale.
    // Simplify: tile over a contiguous block of output pixels.
    uint32_t totalOutElems = N * C_out * H_out * W_out;
    uint32_t bytesPerOutElem = sizeof(float);   // one float per output element after min across channels
    // Actually the final output is N * 1 * H_out * W_out, but intermediate conv output is N*C_out*H_out*W_out.
    // To compute min we need all C_out values per spatial location.
    // We can process one spatial location at a time (1x1 patch) to reduce UB usage.
    uint32_t tileSpatial = 1; // process one output pixel per iteration
    // But we can group multiple spatial pixels if UB allows.
    // For simplicity, we process one output pixel (H=1,W=1) at a time.
    // That requires loading all C_out weights and one input patch.
    uint32_t inputPatch = K_h * K_w * C_in;
    uint32_t weightSlice = C_out * K_h * K_w;
    uint32_t biasSize = (bias_shape != nullptr) ? bias_shape->GetOriginShape().GetShapeSize() : 0;
    uint32_t elementsNeeded = inputPatch + weightSlice + C_out + (biasSize > 0 ? C_out : 0) + 1;
    uint32_t bytesNeeded = elementsNeeded * sizeof(float);
    // If too high, reduce tile further (but already 1 spatial pixel)
    // Adjust block dim based on total out elements
    uint32_t totalSpatialLocs = N * H_out * W_out;
    uint32_t blockDim = CeilDiv(totalSpatialLocs, 256); // heuristic
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_batchSize(N);
    tiling.set_outChannels(C_out);
    tiling.set_outH(H_out);
    tiling.set_outW(W_out);
    tiling.set_kernelH(K_h);
    tiling.set_kernelW(K_w);
    tiling.set_inChannels(C_in);
    tiling.set_scaleFactor(scaleFactor);
    tiling.set_tileSize(1); // one spatial pixel per tile
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

        }
