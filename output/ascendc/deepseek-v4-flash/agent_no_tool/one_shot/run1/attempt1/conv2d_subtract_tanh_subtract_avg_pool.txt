#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dSubtractTanhSubtractAvgPoolCustom {
public:
    __aicore__ inline KernelConv2dSubtractTanhSubtractAvgPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y,
                                const Conv2dSubtractTanhSubtractAvgPoolCustomTilingData &tiling) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, tiling.N * tiling.C * tiling.H * tiling.W);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, tiling.outC * tiling.inC * tiling.kH * tiling.kW);
        if (conv_bias != nullptr)
            bGm_.SetGlobalBuffer((__gm__ float*)conv_bias, tiling.outC);
        yGm_.SetGlobalBuffer((__gm__ float*)y, tiling.totalYLen);
        til_ = tiling;
        // Precompute strides
        strideN_ = til_.C * til_.H * til_.W;
        strideC_ = til_.H * til_.W;
        strideH_ = til_.W;
        strideW_ = 1;
        strideOutC_ = til_.H_conv * til_.W_conv;
        strideOutH_ = til_.W_conv;
        strideOutW_ = 1;
        poolArea_ = til_.K * til_.K;
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();
        uint32_t total = til_.totalYLen;
        uint32_t tileLen = til_.tileLen;
        uint32_t start = blockIdx * tileLen;
        if (start >= total) return;
        uint32_t end = start + tileLen;
        if (end > total) end = total;
        for (uint32_t idx = start; idx < end; ++idx) {
            ProcessOne(idx);
        }
    }

private:
    __aicore__ inline void ProcessOne(uint32_t linearIdx) {
        // linearIdx = n * outC * H_out * W_out + oc * H_out * W_out + h_out * W_out + w_out
        uint32_t w_out = linearIdx % til_.W_out;
        uint32_t h_out = (linearIdx / til_.W_out) % til_.H_out;
        uint32_t oc = (linearIdx / (til_.W_out * til_.H_out)) % til_.outC;
        uint32_t n = linearIdx / (til_.W_out * til_.H_out * til_.outC);

        // Compute conv region for this pool element
        float sum = 0.0f;
        for (uint32_t dy = 0; dy < til_.K; ++dy) {
            for (uint32_t dx = 0; dx < til_.K; ++dx) {
                uint32_t conv_h = h_out * til_.K + dy;
                uint32_t conv_w = w_out * til_.K + dx;
                // Compute conv output at (n, oc, conv_h, conv_w)
                float convVal = 0.0f;
                // Convolution over inC, kH, kW
                uint32_t x_n_offset = n * strideN_;
                for (uint32_t ic = 0; ic < til_.inC; ++ic) {
                    for (uint32_t ky = 0; ky < til_.kH; ++ky) {
                        for (uint32_t kx = 0; kx < til_.kW; ++kx) {
                            uint32_t x_idx = x_n_offset + ic * strideC_ + (conv_h + ky) * strideH_ + (conv_w + kx);
                            uint32_t w_idx = oc * til_.inC * til_.kH * til_.kW + ic * til_.kH * til_.kW + ky * til_.kW + kx;
                            convVal += xGm_.GetValue(x_idx) * wGm_.GetValue(w_idx);
                        }
                    }
                }
                // add bias
                if (bGm_.GetDataSize() > 0)
                    convVal += bGm_.GetValue(oc);
                // apply subtract1 and tanh and subtract2
                float t = convVal - til_.sub1;
                float tanhVal = TanhApprox(t); // Use approximate tanh
                float finalVal = tanhVal - til_.sub2;
                sum += finalVal;
            }
        }
        float avg = sum / poolArea_;
        yGm_.SetValue(linearIdx, avg);
    }

    __aicore__ inline float TanhApprox(float x) {
        // Using built-in tanh from AscendC? For simplicity use local approximation
        // Actually AscendC provides Tanh local tensor function, but here scalar.
        // For scalar we can use fast approximation or use vector later.
        // Placeholder: use (exp(2x)-1)/(exp(2x)+1)
        float e2x = exp(2.0f * x);
        return (e2x - 1.0f) / (e2x + 1.0f);
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;
    Conv2dSubtractTanhSubtractAvgPoolCustomTilingData til_;
    uint32_t strideN_, strideC_, strideH_, strideW_;
    uint32_t strideOutC_, strideOutH_, strideOutW_;
    float poolArea_;
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractTanhSubtractAvgPoolCustom op;
    op.Init(x, weight, conv_bias, y, tiling_data);
    op.Process();
}
