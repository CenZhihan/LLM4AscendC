#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// Assumptions:
// - float, row-major, no transposition
// - M >> K or N >> K? We tailor for tall-skinny A (large M, small K) and wide B (large N, small K).
// The output C is (M x N).
//
// UB budget: each tile consumes:
//   inA:  tileM * tileK floats
//   inB:  tileK * tileN floats
//   out:  tileM * tileN floats
//   tmp:  maybe tileM * tileK + ... but we simplify: total floats = tileM*tileK + tileK*tileN + tileM*tileN.
// Budget bytes: 192KB = 49152 floats (approx). We'll pick a conservative maximum.

static constexpr uint32_t UB_BUDGET_FLOATS = 48 * 1024; // ~192KB / 4
static constexpr uint32_t ALIGN_DIM = 32;              // align tile sizes to 32

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a)   { return ( (x + a - 1) / a ) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)   { return (a + b - 1) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    // Extract shapes: assume A [M, K], B [K, N]
    auto shapeA = context->GetInputShape(0)->GetOriginShape();
    auto shapeB = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = static_cast<uint32_t>(shapeA.GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA.GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB.GetDim(1));

    if (M == 0 || N == 0 || K == 0) return ge::GRAPH_FAILED;

    // Estimate tile sizes from UB budget: choose tileM, tileK, tileN such that
    // tileM*tileK + tileK*tileN + tileM*tileN <= UB_BUDGET_FLOATS
    // We'll fix tileK = min(K, 128) and then adjust tileM and tileN.
    uint32_t tileK = AlignDown(std::min(K, 256u), ALIGN_DIM);
    if (tileK == 0) tileK = ALIGN_DIM;
    // Remaining budget for tileM*tileN: UB_BUDGET_FLOATS - tileM*tileK - tileK*tileN - tileM*tileN <= budget
    // We'll simplistically pick tileM = min(M, 128), tileN = min(N, 128) and then reduce if over.
    uint32_t tileM = AlignDown(std::min(M, 512u), ALIGN_DIM);
    uint32_t tileN = AlignDown(std::min(N, 512u), ALIGN_DIM);
    // Iteratively shrink
    while (tileM * tileK + tileK * tileN + tileM * tileN > UB_BUDGET_FLOATS) {
        if (tileM > tileN) tileM = AlignDown(tileM / 2, ALIGN_DIM);
        else               tileN = AlignDown(tileN / 2, ALIGN_DIM);
        if (tileM < ALIGN_DIM) tileM = ALIGN_DIM;
        if (tileN < ALIGN_DIM) tileN = ALIGN_DIM;
    }

    // Block dim: distribute across M dimension (each block handles a strip of rows)
    uint32_t numRowTiles = CeilDiv(M, tileM);
    uint32_t blockDim = std::min(numRowTiles, 24u);
    if (blockDim < 8) blockDim = 8;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *shapeA = context->GetInputShape(0);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileK_ = tileK; tileN_ = tileN;

        // Row-major: strideA = K, strideB = N, strideC = N
        A_gm_.SetGlobalBuffer((__gm__ float*)A, M_ * K_);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, K_ * N_);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, M_ * N_);

        // Allocate buffers for A, B tiles and accumulator.
        pipe_.InitBuffer(inQueA_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueB_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueC_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQue_, 1, tileM_ * tileN_ * sizeof(float)); // for accumulation
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockId = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        // Distribute row tiles among blocks (grid-stride)
        uint32_t numRowTiles = (M_ + tileM_ - 1) / tileM_;
        uint32_t startTileRow = blockId;
        for (uint32_t tRow = startTileRow; tRow < numRowTiles; tRow += blockNum) {
            uint32_t rowBase = tRow * tileM_;
            uint32_t curTileM = (rowBase + tileM_ > M_) ? (M_ - rowBase) : tileM_;

            // For each tile in N dimension (column tiles) we will accumulate over K tiles.
            // We'll loop over N tiles outside, and for each N tile we compute partial sums over K tiles.
            uint32_t numColTiles = (N_ + tileN_ - 1) / tileN_;
            uint32_t numKTiles   = (K_ + tileK_ - 1) / tileK_;

            // For each column tile, start with zero accumulator.
            for (uint32_t tCol = 0; tCol < numColTiles; ++tCol) {
                uint32_t colBase = tCol * tileN_;
                uint32_t curTileN = (colBase + tileN_ > N_) ? (N_ - colBase) : tileN_;

                // Pre-fetch first K tile
                uint32_t kIdx = 0;
                uint32_t kBase, curTileK;
                // We'll do a double-buffered outer product over K.
                // Initialize accumulator to zero.
                AscendC::LocalTensor<float> accLocal = tmpQue_.AllocTensor<float>();
                AscendC::SetZero(accLocal, curTileM * curTileN);

                // prefetch first A and B tiles
                kBase = kIdx * tileK_;
                curTileK = (kBase + tileK_ > K_) ? (K_ - kBase) : tileK_;
                PipelineIn(rowBase, colBase, kBase, curTileM, curTileK, curTileN);

                while (true) {
                    // Prefetch next tile (if exists) for overlap
                    uint32_t nextK = kIdx + 1;
                    bool hasNext = (nextK < numKTiles);
                    if (hasNext) {
                        uint32_t nkBase = nextK * tileK_;
                        uint32_t ncurTileK = (nkBase + tileK_ > K_) ? (K_ - nkBase) : tileK_;
                        PipelineIn(rowBase, colBase, nkBase, curTileM, ncurTileK, curTileN);
                    }

                    // Compute partial product and accumulate
                    ComputePartial(curTileM, curTileK, curTileN, accLocal);

                    // Free current tiles
                    // (inside ComputePartial we dequeue and free)

                    if (!hasNext) break;
                    kIdx = nextK;
                    kBase = nextK * tileK_;
                    curTileK = (kBase + tileK_ > K_) ? (K_ - kBase) : tileK_;
                }

                // Write result to global memory for this column tile
                PipelineOut(rowBase, colBase, curTileM, curTileN, accLocal);
                tmpQue_.FreeTensor(accLocal);
            }
        }
    }

private:
    __aicore__ inline void PipelineIn(uint32_t rowBase, uint32_t colBase,
                                       uint32_t kBase,
                                       uint32_t curTileM, uint32_t curTileK, uint32_t curTileN)
    {
        // Allocate and copy A tile [rowBase:rowBase+curTileM, kBase:kBase+curTileK]
        auto aLocal = inQueA_.AllocTensor<float>();
        uint32_t aSize = curTileM * curTileK;
        AscendC::DataCopy(aLocal, A_gm_[rowBase * K_ + kBase], aSize);
        inQueA_.EnQue(aLocal);

        // Allocate and copy B tile [kBase:kBase+curTileK, colBase:colBase+curTileN]
        auto bLocal = inQueB_.AllocTensor<float>();
        uint32_t bSize = curTileK * curTileN;
        AscendC::DataCopy(bLocal, B_gm_[kBase * N_ + colBase], bSize);
        inQueB_.EnQue(bLocal);
    }

    __aicore__ inline void ComputePartial(uint32_t curTileM, uint32_t curTileK,
                                           uint32_t curTileN,
                                           AscendC::LocalTensor<float>& accum)
    {
        auto aLocal = inQueA_.DeQue<float>();
        auto bLocal = inQueB_.DeQue<float>();

        // C += A * B  (accumulation)
        // Using AscendC's matmul or manual? For simplicity we rely on built-in MatMul
        // but that requires 2D tensors. We'll use batch matmul with unit batch.
        // Alternatively, we can use matrix multiply from vec instructions.
        // Given time, we assume a custom inline matmul using Mul and Add? Not feasible.
        // We'll simulate with AscendC::MatMul (if available) or use a loop.
        // For demonstration, we'll use a naive triple loop (inefficient but illustrative).
        // In practice, you'd use AscendC's MatMul or optimized vector gemm.
        for (uint32_t i = 0; i < curTileM; ++i) {
            for (uint32_t j = 0; j < curTileN; ++j) {
                float sum = 0.0f;
                for (uint32_t k = 0; k < curTileK; ++k) {
                    sum += aLocal[i * curTileK + k] * bLocal[k * curTileN + j];
                }
                accum[i * curTileN + j] += sum;
            }
        }

        inQueA_.FreeTensor(aLocal);
        inQueB_.FreeTensor(bLocal);
    }

    __aicore__ inline void PipelineOut(uint32_t rowBase, uint32_t colBase,
                                        uint32_t curTileM, uint32_t curTileN,
                                        AscendC::LocalTensor<float>& accum)
    {
        // Write accumulated result to global memory
        auto cLocal = outQueC_.AllocTensor<float>();
        AscendC::DataCopy(cLocal, accum, curTileM * curTileN);
        outQueC_.EnQue(cLocal);

        // Dequeue and memcpy to global
        auto cDeq = outQueC_.DeQue<float>();
        uint32_t offset = rowBase * N_ + colBase;
        for (uint32_t i = 0; i < curTileM; ++i) {
            AscendC::DataCopy(C_gm_[offset + i * N_], &cDeq[i * curTileN], curTileN);
        }
        outQueC_.FreeTensor(cDeq);
    }

    // Pipeline queues
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQue_;

    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t M_, N_, K_;
    uint32_t tileM_, tileK_, tileN_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
