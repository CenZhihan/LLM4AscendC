    uint32_t cin = x_shape->GetDim(1);
    uint32_t h_in = x_shape->GetDim(2);
    uint32_t w_in = x_shape->GetDim(3);
    uint32_t cout = w_shape->GetDim(0);
    uint32_t kernel_h = w_shape->GetDim(2);
    uint32_t kernel_w = w_shape->GetDim(3);

    uint32_t w_out = w_in - kernel_w + 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    float scaling_factor = atts->GetAttrPointer<float>(1)[0];

    // Estimate UB usage per tile: we need buffers for x tile, weight tile, conv_bias, post_bias,
    // intermediate result, output tile. For simplicity, assume each tile processes one output pixel
    // across channels? Actually we tile by output spatial dimensions and output channels.
    // A tile will process a block of output pixels: tile_n * tile_cout * (tile_h_out * tile_w_out)
    // We set tile_n = min(tile_n_max, n) and tile_cout = min(tile_cout_max, cout)

    // plus output tile. For 3x3 kernel, each x tile approx kernel_h * kernel_w * cin
    // We choose small tile to fit. For simplicity, tile by output pixel: 1 output pixel requires
    // kernel_h * kernel_w * cin input elements and cout weights per pixel? Actually per output pixel,
    // need whole weight kernel for that output channel. Better to tile by output channels.
    uint32_t tile_cout = 32; // align to 32
    uint32_t tile_n = 2;
    // Adjust to fit UB: rough estimate
    uint32_t input_per_tile = tile_n * tile_cout * kernel_h * kernel_w * cin;
    uint32_t weight_per_tile = tile_cout * cin * kernel_h * kernel_w;

    tiling.set_constant_value(constant_value);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dMinAddMultiplyCustom {
public:
    __aicore__ inline KernelConv2dMinAddMultiplyCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias,
                                GM_ADDR post_bias, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t h_in, uint32_t w_in,
                                uint32_t cout, uint32_t kernel_h, uint32_t kernel_w,
                                uint32_t h_out, uint32_t w_out,
                                uint32_t tile_n, uint32_t tile_cout,
                                float constant_value, float scaling_factor) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, n * cin * h_in * w_in);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, cout * cin * kernel_h * kernel_w);
        if (conv_bias != nullptr) {
            convBiasGm_.SetGlobalBuffer((__gm__ float*)conv_bias, cout);
            hasConvBias_ = true;
        }
        postBiasGm_.SetGlobalBuffer((__gm__ float*)post_bias, cout);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n * cout * h_out * w_out);

        n_ = n;
        cin_ = cin;
        hIn_ = h_in;
        wIn_ = w_in;
        cout_ = cout;
        kernelH_ = kernel_h;
        kernelW_ = kernel_w;
        hOut_ = h_out;
        wOut_ = w_out;
        tileN_ = tile_n;
        tileCout_ = tile_cout;
        constantValue_ = constant_value;
        scalingFactor_ = scaling_factor;

        // Compute tile sizes in elements
        uint32_t xTileElems = tileN_ * tileCout_ * kernelH_ * kernelW_ * cin_;
        uint32_t wTileElems = tileCout_ * cin_ * kernelH_ * kernelW_;
        uint32_t yTileElems = tileN_ * tileCout_ * hOut_ * wOut_; // but we actually tile only N and Cout, spatial all
        // For simplicity, we allocate fixed sizes. In production, we need dynamic.
        uint32_t bufSize = (xTileElems > wTileElems ? xTileElems : wTileElems);
        bufSize = (bufSize > yTileElems ? bufSize : yTileElems);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, bufSize * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, wTileElems * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, yTileElems * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, (tileN_ * tileCout_) * sizeof(float)); // for min/add/mul temporaries
    }

    __aicore__ inline void Process() {
        if (n_ == 0 || cout_ == 0) return;
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Distribute tiles: each block processes a set of (n_tile, cout_tile) pairs
        uint32_t numNTiles = (n_ + tileN_ - 1) / tileN_;
        uint32_t numCoutTiles = (cout_ + tileCout_ - 1) / tileCout_;
        uint32_t totalTiles = numNTiles * numCoutTiles;
        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            uint32_t nTileIdx = t / numCoutTiles;
            uint32_t cTileIdx = t % numCoutTiles;
            uint32_t nStart = nTileIdx * tileN_;
            uint32_t nEnd = (nStart + tileN_ > n_) ? n_ : nStart + tileN_;
            uint32_t cStart = cTileIdx * tileCout_;
            uint32_t cEnd = (cStart + tileCout_ > cout_) ? cout_ : cStart + tileCout_;
            uint32_t curTileN = nEnd - nStart;
            uint32_t curTileCout = cEnd - cStart;
            // Process this tile: convolve and fuse
            ProcessTile(nStart, curTileN, cStart, curTileCout);
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t nStart, uint32_t curN,
                                        uint32_t cStart, uint32_t curCout) {
        // For each output pixel (n, h_out, w_out), compute convolution over kernel window.
        // We'll iterate over output spatial dimensions.
        for (uint32_t n = nStart; n < nStart + curN; n++) {
            for (uint32_t h = 0; h < hOut_; h++) {
                for (uint32_t w = 0; w < wOut_; w++) {
                    // Load input window for all cout channels? That would be huge.
                    // Instead, for each output channel in this tile, compute partial sum.
                    // Optimize by loading input window once per output pixel.
                    // Allocate local buffer for input window (kernelH_ * kernelW_ * cin_)
                    AscendC::LocalTensor<float> xWin = inQueueX_.AllocTensor<float>();
                    AscendC::LocalTensor<float> wSlice = inQueueW_.AllocTensor<float>();
                    AscendC::LocalTensor<float> outPixel = outQueueY_.AllocTensor<float>();
                    AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

                    // Copy input window for this pixel (N, h, w)
                    // Input shape: [N, C, H, W], window is [C, kernelH, kernelW] at (h_in, w_in) = (h, w)
                    for (uint32_t c = 0; c < cin_; c++) {
                        for (uint32_t kh = 0; kh < kernelH_; kh++) {
                            for (uint32_t kw = 0; kw < kernelW_; kw++) {
                                uint32_t srcIdx = (n * cin_ + c) * (hIn_ * wIn_) + (h + kh) * wIn_ + (w + kw);
                                uint32_t dstIdx = (c * kernelH_ + kh) * kernelW_ + kw;
                                xWin.SetValue(dstIdx, xGm_.GetValue(srcIdx));
                            }
                        }
                    }

                    // For each output channel in this tile, compute convolution, then apply bias, min, add, mul
                    for (uint32_t co = 0; co < curCout; co++) {
                        uint32_t globalCo = cStart + co;
                        // Compute convolution for this channel co
                        float convVal = 0.0f;
                        for (uint32_t c = 0; c < cin_; c++) {
                            for (uint32_t kh = 0; kh < kernelH_; kh++) {
                                for (uint32_t kw = 0; kw < kernelW_; kw++) {
                                    uint32_t wIdx = (co * cin_ + c) * kernelH_ * kernelW_ + kh * kernelW_ + kw;
                                    uint32_t xIdx = (c * kernelH_ + kh) * kernelW_ + kw;
                                    convVal += wGm_.GetValue(wIdx) * xWin.GetValue(xIdx);
                                }
                            }
                        }
                        // Add conv bias if present
                        if (hasConvBias_) {
                            convVal += convBiasGm_.GetValue(globalCo);
                        }
                        // Apply min with constant value
                        if (convVal > constantValue_) {
                            convVal = constantValue_;
                        }
                        // Add post_bias
                        convVal += postBiasGm_.GetValue(globalCo);
                        // Multiply by scaling factor
                        convVal *= scalingFactor_;
                        // Store to output
                        uint32_t outIdx = (n * cout_ + globalCo) * hOut_ * wOut_ + h * wOut_ + w;
                        yGm_.SetValue(outIdx, convVal);
                    }

                    // Free locals
                    inQueueX_.FreeTensor(xWin);
                    inQueueW_.FreeTensor(wSlice);
                    outQueueY_.FreeTensor(outPixel);
                    tmpQueue_.FreeTensor(tmp);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> convBiasGm_;
    AscendC::GlobalTensor<float> postBiasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_, cin_, hIn_, wIn_, cout_, kernelH_, kernelW_, hOut_, wOut_;
    uint32_t tileN_, tileCout_;
    float constantValue_, scalingFactor_;
    bool hasConvBias_ = false;
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias,
    GM_ADDR post_bias, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinAddMultiplyCustom op;
    op.Init(x, weight, conv_bias, post_bias, y,
            tiling_data.n, tiling_data.cin, tiling_data.h_in, tiling_data.w_in,
            tiling_data.cout, tiling_data.kernel_h, tiling_data.kernel_w,
            tiling_data.h_out, tiling_data.w_out,
            tiling_data.tile_n, tiling_data.tile_cout,
            tiling_data.constant_value, tiling_data.scaling_factor);
    op.Process();
}

class ModelNew(torch.nn.Module):
