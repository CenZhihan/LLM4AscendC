kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t batch, uint32_t inChannels, uint32_t outChannels,
                                uint32_t inHeight, uint32_t inWidth,
                                uint32_t outHeight, uint32_t outWidth,
                                uint32_t kernelH, uint32_t kernelW,
                                int32_t strideH, int32_t strideW,
                                int32_t padH, int32_t padW,
                                int32_t dilationH, int32_t dilationW,
                                int32_t outputPadH, int32_t outputPadW,
                                uint32_t tileNum, uint32_t tileOutElems)
    {
        batch_ = batch;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        inHeight_ = inHeight;
        inWidth_ = inWidth;
        outHeight_ = outHeight;
        outWidth_ = outWidth;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        outputPadH_ = outputPadH;
        outputPadW_ = outputPadW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // For simplicity, we allocate a fixed-size local buffer per tile.
        // Tile size: we process one spatial tile at a time for all output channels.
        // We need local buffers for input window, weight slice, and output tile.
        // UB budget is limited; we use small fixed sizes.
        // In a real kernel, we would compute proper sizes.
        tileOutH_ = 8;
        tileOutW_ = 8;
        if (tileOutH_ > outHeight_) tileOutH_ = outHeight_;
        if (tileOutW_ > outWidth_) tileOutW_ = outWidth_;
        tileOutElems_ = tileOutH_ * tileOutW_ * outChannels_;  // per batch tile

        // Input window size needed: approximate
        uint32_t inputHWin = (tileOutH_ - 1) * strideH_ + dilationH_ * (kernelH_ - 1) + 1;
        uint32_t inputWWin = (tileOutW_ - 1) * strideW_ + dilationW_ * (kernelW_ - 1) + 1;
        inputWinSize_ = inputHWin * inputWWin * inChannels_;  // per batch tile

        // Weight size: all outChannels, inChannels, kernelH, kernelW (but we load only needed part)
        // We'll load entire weight? That's too large. Instead we load per-output-channel? For simplicity,
        // we allocate a buffer for the entire weight? That's not feasible.
        // In this simplified kernel, we will read weight in a nested loop directly from global memory.
        // To avoid huge local buffers, we load one kernel at a time.
        // So we only need local for input and output tiles.
        // But we still need weight values. We'll load weight on the fly.

        // Initialize pipe and queues for input and output.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputWinSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOutElems_ * sizeof(float));
        // No separate tmp buffer needed for this simple implementation.
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || outHeight_ == 0 || outWidth_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a subset of tiles. We use grid stride over tiles.
        uint32_t totalTiles = ( (outHeight_ + tileOutH_ - 1) / tileOutH_ ) *
                              ( (outWidth_ + tileOutW_ - 1) / tileOutW_ );
        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            uint32_t tileRow = tileIdx / ((outWidth_ + tileOutW_ - 1) / tileOutW_);
            uint32_t tileCol = tileIdx % ((outWidth_ + tileOutW_ - 1) / tileOutW_);
            uint32_t oStartH = tileRow * tileOutH_;
            uint32_t oStartW = tileCol * tileOutW_;
            uint32_t oEndH = oStartH + tileOutH_;
            uint32_t oEndW = oStartW + tileOutW_;
            if (oEndH > outHeight_) oEndH = outHeight_;
            if (oEndW > outWidth_) oEndW = outWidth_;

            uint32_t curOutH = oEndH - oStartH;
            uint32_t curOutW = oEndW - oStartW;

            // For each batch, process the tile
            for (uint32_t n = 0; n < batch_; n++) {
                // For each output channel
                // We'll allocate local tensors for output tile (zero-init)
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::SetZero(yLocal, curOutH * curOutW * outChannels_);

                // For each output pixel in tile
                for (uint32_t oh = 0; oh < curOutH; oh++) {
                    for (uint32_t ow = 0; ow < curOutW; ow++) {
                        uint32_t outPixelIdx = (oh * curOutW + ow) * outChannels_;
                        // For each output channel
                        for (uint32_t oc = 0; oc < outChannels_; oc++) {
                            float sum = 0.0f;
                            // For each input channel (groups=1)
                            for (uint32_t ic = 0; ic < inChannels_; ic++) {
                                // For each kernel element
                                for (uint32_t kh = 0; kh < kernelH_; kh++) {
                                    for (uint32_t kw = 0; kw < kernelW_; kw++) {
                                        // Compute input indices (transposed convolution)
                                        // out = (in - 1)*stride - 2*pad + dil*(k-1) + outpad + 1
                                        // For each output pixel (oh, ow), the contributing input position is
                                        // ih = (oh + padH - kh * dilationH) / strideH
                                        // And similarly for width.
                                        // But careful: This is the "backward" formula.
                                        // Standard conv_transpose2d: For each output pixel, input indices:
                                        // ih = (oh + padH - dilationH * kh) / strideH
                                        // iw = (ow + padW - dilationW * kw) / strideW
                                        // Must be exact division (no remainder) for contribution.
                                        uint32_t ih_num = oh + padH_ - dilationH_ * kh;
                                        uint32_t iw_num = ow + padW_ - dilationW_ * kw;
                                        // Check if divisible by stride and within input bounds
                                        if (ih_num % strideH_ == 0 && iw_num % strideW_ == 0) {
                                            uint32_t ih = ih_num / strideH_;
                                            uint32_t iw = iw_num / strideW_;
                                            if (ih < inHeight_ && iw < inWidth_) {
                                                // Get weight: index in weight tensor: (ic, oc, kh, kw)
                                                uint32_t wIdx = ((ic * outChannels_ + oc) * kernelH_ + kh) * kernelW_ + kw;
                                                // Get input: index in x: (n, ic, ih, iw)
                                                uint32_t xIdx = ((n * inChannels_ + ic) * inHeight_ + ih) * inWidth_ + iw;
                                                // Read from global memory (inefficient, but for demonstration)
                                                sum += wGm_.GetValue(wIdx) * xGm_.GetValue(xIdx);
                                            }
                                        }
                                    }
                                }
                            }
                            // Store in local output tile
                            yLocal.SetValue(outPixelIdx + oc, sum);
                        }
                    }
                }
                // Write output tile to global memory
                // Compute base offset for this batch and tile
                uint32_t yBase = ((n * outChannels_) * outHeight_ + oStartH) * outWidth_ + oStartW;
                // But output is contiguous in NCHW? We assume ND layout (NCHW). So we write row-major but careful: 
                // For each batch n, output shape: [outChannels, outHeight, outWidth]
                // The tile spans rows oStartH..oEndH columns oStartW..oEndW for all output channels.
                // So we need to interleave output channels. For simplicity, we copy the local tensor directly assuming it's laid out as [outChannels, curOutH, curOutW] contiguous.
                // But the global layout is [N, outChannels, H, W]. So we must compute strides.
                // Simplified: we copy per-channel per-row.
                for (uint32_t oc = 0; oc < outChannels_; oc++) {
                    for (uint32_t row = 0; row < curOutH; row++) {
                        uint32_t gRow = oStartH + row;
                        uint32_t gBase = ((n * outChannels_ + oc) * outHeight_ + gRow) * outWidth_ + oStartW;
                        for (uint32_t col = 0; col < curOutW; col++) {
                            uint32_t lIdx = (oc * curOutH + row) * curOutW + col;
                            yGm_.SetValue(gBase + col, yLocal.GetValue(lIdx));
                        }
                    }
                }
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t inChannels_;
    uint32_t outChannels_;
    uint32_t inHeight_;
    uint32_t inWidth_;
    uint32_t outHeight_;
    uint32_t outWidth_;
    uint32_t kernelH_;
    uint32_t kernelW_;
    int32_t strideH_;
    int32_t strideW_;
    int32_t padH_;
    int32_t padW_;
    int32_t dilationH_;
    int32_t dilationW_;
    int32_t outputPadH_;
    int32_t outputPadW_;
    uint32_t tileOutH_;
    uint32_t tileOutW_;
    uint32_t tileOutElems_;
    uint32_t inputWinSize_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(
