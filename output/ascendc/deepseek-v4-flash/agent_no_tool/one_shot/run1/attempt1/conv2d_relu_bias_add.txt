static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dReluBiasAddCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *pb_shape = context->GetInputShape(3);  // post_bias is index 3

    // x: N C H W
    uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t C_in = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t H_in = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t W_in = static_cast<uint32_t>(x_shape->GetDim(3));

    uint32_t C_out = static_cast<uint32_t>(w_shape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(w_shape->GetDim(2)); // assume square kernel

    uint32_t H_out = H_in - K + 1;
    uint32_t W_out = W_in - K + 1;
    if (H_out == 0 || W_out == 0) {
        return ge::GRAPH_FAILED;
    }
    uint32_t totalLength = N * C_out * H_out * W_out;

    // determine if conv_bias is provided (optional input index 2)
    bool hasConvBias = (context->GetInputShape(2) != nullptr);

    // tiling heuristic: each iteration processes a tile of output elements
    // We also need room for weight and input patches. For simplicity, use small tile.
    uint32_t tileEl = 512; // fixed small tile; can be tuned
    if (tileEl > totalLength) tileEl = totalLength;
    tileEl = AlignDown(tileEl, ALIGN_ELEMS);
    if (tileEl == 0) tileEl = ALIGN_ELEMS;

    uint32_t numTiles = CeilDiv(totalLength, tileEl);
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileEl);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C_in(C_in);
    tiling.set_C_out(C_out);
    tiling.set_H_in(H_in);
    tiling.set_W_in(W_in);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_kernelSize(K);
    tiling.set_hasConvBias(hasConvBias);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dReluBiasAddCustom {
public:
    __aicore__ inline KernelConv2dReluBiasAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR post_bias,
                                GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C_in, uint32_t C_out,
                                uint32_t H_in, uint32_t W_in, uint32_t H_out, uint32_t W_out,
                                uint32_t kernelSize, bool hasConvBias)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        if (hasConvBias) convBiasGm_.SetGlobalBuffer((__gm__ float*)conv_bias);
        postBiasGm_.SetGlobalBuffer((__gm__ float*)post_bias);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N;
        C_in_ = C_in;
        C_out_ = C_out;
        H_in_ = H_in;
        W_in_ = W_in;
        H_out_ = H_out;
        W_out_ = W_out;
        K_ = kernelSize;
        hasConvBias_ = hasConvBias;

        // Allocate local buffers
        pipe_.InitBuffer(inWeightBuf_, BUFFER_NUM, K_ * K_ * C_in_ * sizeof(float)); // one output channel's weight slice
        pipe_.InitBuffer(inPatchBuf_, BUFFER_NUM, K_ * K_ * C_in_ * sizeof(float));
        pipe_.InitBuffer(psumBuf_, 1, sizeof(float)); // for partial sum
        pipe_.InitBuffer(convBiasBuf_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(postBiasBuf_, BUFFER_NUM, sizeof(float));
        // For output, we use direct DataCopy to global
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        // Grid-stride over output elements
        uint32_t stride = blockNum * tileLen_;
        uint32_t start = blockIdx * tileLen_;
        uint32_t end = start + tileLen_;
        if (start >= totalLen_) return;
        if (end > totalLen_) end = totalLen_;

        uint32_t outputIndex = start;
        while (outputIndex < end) {
            // Map linear index to (n, c, h, w)
            uint32_t remaining = end - outputIndex;
            uint32_t curTile = (remaining < tileLen_) ? remaining : tileLen_;

            // Process tile of output elements (each element requires convolution)
            for (uint32_t i = 0; i < curTile; i++) {
                uint32_t idx = outputIndex + i;
                uint32_t n = idx / (C_out_ * H_out_ * W_out_);
                uint32_t tmp = idx % (C_out_ * H_out_ * W_out_);
                uint32_t c = tmp / (H_out_ * W_out_);
                uint32_t spatial = tmp % (H_out_ * W_out_);
                uint32_t h = spatial / W_out_;
                uint32_t w = spatial % W_out_;

                // Compute convolution for output (n, c, h, w)
                // Load weight for output channel c (all input channels)
                // Weight layout: (C_out, C_in, K, K)
                float sum = 0.0f;
                for (uint32_t ci = 0; ci < C_in_; ci++) {
                    for (uint32_t kh = 0; kh < K_; kh++) {
                        for (uint32_t kw = 0; kw < K_; kw++) {
                            // Input coordinates
                            uint32_t in_h = h + kh;
                            uint32_t in_w = w + kw;
                            // Load input value (using global tensor)
                            float in_val = xGm_[n * C_in_ * H_in_ * W_in_ + ci * H_in_ * W_in_ + in_h * W_in_ + in_w];
                            // Load weight value
                            float w_val = weightGm_[c * C_in_ * K_ * K_ + ci * K_ * K_ + kh * K_ + kw];
                            sum += in_val * w_val;
                        }
                    }
                }

                // Add conv_bias if present
                if (hasConvBias_) {
                    sum += convBiasGm_[c];
                }

                // ReLU
                if (sum < 0.0f) sum = 0.0f;

                // Add post_bias (broadcast across spatial and batch)
                sum += postBiasGm_[c];

                // Store output
                yGm_[idx] = sum;
            }

            outputIndex += curTile;
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inWeightBuf_;
    TQue<TPosition::VECIN, BUFFER_NUM> inPatchBuf_;
    TQue<TPosition::VECCALC, 1> psumBuf_;
    TQue<TPosition::VECIN, BUFFER_NUM> convBiasBuf_;
    TQue<TPosition::VECIN, BUFFER_NUM> postBiasBuf_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> weightGm_;
    GlobalTensor<float> convBiasGm_;
    GlobalTensor<float> postBiasGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t N_, C_in_, C_out_, H_in_, W_in_, H_out_, W_out_;
    uint32_t K_;
    bool hasConvBias_;
};

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR post_bias,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluBiasAddCustom op;
    op.Init(x, weight, conv_bias, post_bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C_in, tiling_data.C_out,
            tiling_data.H_in, tiling_data.W_in, tiling_data.H_out, tiling_data.W_out,
            tiling_data.kernelSize, tiling_data.hasConvBias);
    op.Process();
}
