static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t MAX_TILE_AREA = 8192; // safe upper bound

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ConvDepthwise2dSquareInputAsymmetricKernelCustomTilingData tiling;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernel_size_ptr = attrs->GetAttrPointer<int32_t>(0);
    int32_t kernel_h = *kernel_size_ptr;
    int32_t kernel_w = 1;  // asymmetric kernel width fixed to 1

    // Get input shapes
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t C = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t H = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    uint32_t W = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    uint32_t outputH = H - static_cast<uint32_t>(kernel_h) + 1;
    uint32_t outputW = W - static_cast<uint32_t>(kernel_w) + 1; // = W

    // Compute tileH and tileW based on UB budget
    uint32_t maxFloats = UB_BUDGET_BYTES / sizeof(float);
    // Rough UB usage per channel tile: input = (tileH + kernelH - 1)*tileW, weight = kernelH, output = tileH*tileW, temp = tileH*tileW
    // So total ~ 3*tileH*tileW + kernelH*tileW + kernelH <= maxFloats
    // Conservative: require 3*area <= maxFloats - 2*kernelH*tileW, but approximate with 3*area <= maxFloats
    uint32_t maxArea = maxFloats / 3;
    if (maxArea > MAX_TILE_AREA) maxArea = MAX_TILE_AREA;
    maxArea = AlignDown(maxArea, ALIGN_ELEMS);
    if (maxArea == 0) maxArea = ALIGN_ELEMS;

    uint32_t tileW = outputW;
    if (tileW > 512) tileW = 512; // reasonable max width per tile
    tileW = AlignDown(tileW, ALIGN_ELEMS);
    if (tileW == 0) tileW = ALIGN_ELEMS;
    uint32_t tileH = maxArea / tileW;
    if (tileH > outputH) tileH = outputH;
    if (tileH == 0) tileH = 1;
    tileH = AlignDown(tileH, ALIGN_ELEMS);
    if (tileH == 0) tileH = ALIGN_ELEMS;
    // Ensure tileH <= outputH
    while (tileH > outputH) tileH = AlignDown(outputH, ALIGN_ELEMS);
    if (tileH == 0) tileH = outputH;

    // blockDim: distribute across batch*channel
    uint32_t totalChannels = N * C;
    uint32_t blockDim = totalChannels;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_kernelH(static_cast<uint32_t>(kernel_h));
    tiling.set_kernelW(static_cast<uint32_t>(kernel_w));
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t kernelH, uint32_t kernelW,
                                uint32_t outputH, uint32_t outputW,
                                uint32_t tileH, uint32_t tileW) {
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        outputH_ = outputH;
        outputW_ = outputW;
        tileH_ = tileH;
        tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // buffers: weight, input tile, output tile, temp for accumulation
        uint32_t inputTileSize = (tileH_ + kernelH_ - 1) * tileW_;
        uint32_t outputTileSize = tileH_ * tileW_;
        pipe_.InitBuffer(wQueue_, 1, kernelH_ * sizeof(float));
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputTileSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outputTileSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, outputTileSize * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (N_ == 0 || C_ == 0 || outputH_ == 0 || outputW_ == 0) return;

        uint32_t blockDim = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each core processes a contiguous chunk of (batchIdx, channelIdx) pairs
        uint32_t totalChannels = N_ * C_;
        uint32_t channelsPerCore = (totalChannels + blockDim - 1) / blockDim;
        uint32_t startChan = blockIdx * channelsPerCore;
        uint32_t endChan = (blockIdx + 1) * channelsPerCore;
        if (endChan > totalChannels) endChan = totalChannels;

        for (uint32_t globalChan = startChan; globalChan < endChan; ++globalChan) {
            uint32_t batchIdx = globalChan / C_;
            uint32_t chanIdx = globalChan % C_;

            // Load weight for this channel
            uint32_t weightOffset = chanIdx * kernelH_ * kernelW_; // kernelW==1
            LoadWeight(weightOffset);

            // Process output tiles for this channel
            for (uint32_t rowStart = 0; rowStart < outputH_; rowStart += tileH_) {
                uint32_t curTileH = outputH_ - rowStart;
                if (curTileH > tileH_) curTileH = tileH_;
                for (uint32_t colStart = 0; colStart < outputW_; colStart += tileW_) {
                    uint32_t curTileW = outputW_ - colStart;
                    if (curTileW > tileW_) curTileW = tileW_;

                    // Load input tile (needs extra rows at top and bottom for kernel)
                    uint32_t inputRowStart = rowStart; // since padding=0
                    uint32_t inputTileH = curTileH + kernelH_ - 1;
                    LoadInputTile(batchIdx, chanIdx, inputRowStart, colStart, inputTileH, curTileW);

                    // Compute convolution
                    ComputeTile(curTileH, curTileW, colStart);

                    // Store output tile
                    StoreOutputTile(batchIdx, chanIdx, rowStart, colStart, curTileH, curTileW);
                }
            }
        }
    }

private:
    __aicore__ inline void LoadWeight(uint32_t weightOffset) {
        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_[weightOffset], kernelH_);
        wQueue_.EnQue<float>(wLocal);
    }

    __aicore__ inline void LoadInputTile(uint32_t batchIdx, uint32_t chanIdx,
                                         uint32_t rowStart, uint32_t colStart,
                                         uint32_t inputTileH, uint32_t inputTileW) {
        // Compute global offset: batchIdx * C * H * W + chanIdx * H * W + rowStart * W + colStart
        uint32_t baseOffset = (batchIdx * C_ + chanIdx) * H_ * W_;
        uint32_t lineOffset = baseOffset + rowStart * W_ + colStart;

        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        // Copy row by row if needed (contiguous if tileW == W, else not contiguous)
        if (colStart == 0 && inputTileW == W_) {
            // Entire rows are contiguous
            AscendC::DataCopy(xLocal, xGm_[lineOffset], inputTileH * W_);
        } else {
            // Non-contiguous: copy row by row
            for (uint32_t row = 0; row < inputTileH; ++row) {
                uint32_t srcOffset = lineOffset + row * W_;
                AscendC::DataCopy(xLocal[row * inputTileW], xGm_[srcOffset], inputTileW);
            }
        }
        inQueueX_.EnQue<float>(xLocal);
    }

    __aicore__ inline void ComputeTile(uint32_t tileH, uint32_t tileW, uint32_t colStart) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = wQueue_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> temp = tmpQueue_.AllocTensor<float>();

        // Perform 1D vertical convolution over each column
        for (uint32_t col = 0; col < tileW; ++col) {
            // For each output row in tile
            for (uint32_t row = 0; row < tileH; ++row) {
                float sum = 0.0f;
                for (uint32_t k = 0; k < kernelH_; ++k) {
                    float xVal = xLocal[(row + k) * tileW + col];
                    float wVal = wLocal[k];
                    sum += xVal * wVal;
                }
                yLocal[row * tileW + col] = sum;
            }
        }

        tmpQueue_.FreeTensor(temp);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        wQueue_.FreeTensor(wLocal);
    }

    __aicore__ inline void StoreOutputTile(uint32_t batchIdx, uint32_t chanIdx,
                                            uint32_t rowStart, uint32_t colStart,
                                            uint32_t tileH, uint32_t tileW) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint32_t baseOffset = (batchIdx * C_ + chanIdx) * outputH_ * outputW_;
        uint32_t lineOffset = baseOffset + rowStart * outputW_ + colStart;

        if (colStart == 0 && tileW == outputW_) {
            AscendC::DataCopy(yGm_[lineOffset], yLocal, tileH * outputW_);
        } else {
            for (uint32_t row = 0; row < tileH; ++row) {
                uint32_t dstOffset = lineOffset + row * outputW_;
                AscendC::DataCopy(yGm_[dstOffset], yLocal[row * tileW], tileW);
            }
        }
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wQueue_;  // weight per channel
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, H_, W_;
    uint32_t kernelH_, kernelW_;
    uint32_t outputH_, outputW_;
    uint32_t tileH_, tileW_;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.C,
            tiling_data.H, tiling_data.W,
            tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.outputH, tiling_data.outputW,
            tiling_data.tileH, tiling_data.tileW);
    op.Process();
}
