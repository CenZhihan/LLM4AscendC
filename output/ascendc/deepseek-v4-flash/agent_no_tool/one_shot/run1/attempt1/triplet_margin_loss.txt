#include "triplet_margin_loss_custom_tiling.h"

namespace optiling {

// UB budget estimate:
// - For each sample we need: diff (1 tile), square (1 tile), sum (1 tile), sqrt (1 tile), distances (2 tiles), loss (1 tile)
//   partial sums can reuse. Conservatively use 6 tiles of dim floats.
// - Bytes per tile: dim * sizeof(float). UB budget: 192KB.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128; // align dim tile

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;
    const gert::Shape *shapeAnchor = context->GetInputShape(0);
    const gert::Shape *shapePos = context->GetInputShape(1);
    const gert::Shape *shapeNeg = context->GetInputShape(2);

    uint32_t totalSamples = static_cast<uint32_t>(shapeAnchor->GetOriginShape().GetDim(0));
    uint32_t dim = static_cast<uint32_t>(shapeAnchor->GetOriginShape().GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *marginPtr = attrs->GetAttrPointer<float>(0);
    float margin = marginPtr ? *marginPtr : 1.0f;

    // Compute tileDim from UB budget
    uint32_t maxTileDim = UB_BUDGET_BYTES / (6 * sizeof(float)); // 6 tiles
    maxTileDim = AlignDown(maxTileDim, ALIGN_ELEMS);
    if (maxTileDim < ALIGN_ELEMS) maxTileDim = ALIGN_ELEMS;

    uint32_t tileDim = (dim < maxTileDim) ? dim : maxTileDim;
    // Ensure tileDim covers whole dim if possible
    if (dim < tileDim) {
        tileDim = dim;
    }

    // BlockDim: assign one block per sample chunk. target blocks ~8-24
    uint32_t blockDim = 8;
    if (totalSamples > 24) blockDim = 24;
    else if (totalSamples > 8) blockDim = totalSamples;

    context->SetBlockDim(blockDim);

    tiling.set_totalSamples(totalSamples);
    tiling.set_dim(dim);
    tiling.set_tileDim(tileDim);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR output,
                                uint32_t totalSamples, uint32_t dim, uint32_t tileDim,
                                float margin)
    {
        totalSamples_ = totalSamples;
        dim_ = dim;
        tileDim_ = tileDim;
        margin_ = margin;

        // Input tensors are 2D (n, d) row-major
        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalSamples_ * dim_);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalSamples_ * dim_);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalSamples_ * dim_);
        outputGm_.SetGlobalBuffer((__gm__ float*)output, totalSamples_);

        // Local buffer sizes: we need for each sample: diff, square, sum (scalar), distances, loss
        // We'll allocate buffers per tile of dim
        pipe_.InitBuffer(inQueAnchor_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(inQuePos_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(inQueNeg_, BUFFER_NUM, tileDim_ * sizeof(float));
        pipe_.InitBuffer(outQueLoss_, BUFFER_NUM, sizeof(float)); // scalar per sample
        pipe_.InitBuffer(tmpBuf1_, BUFFER_NUM, tileDim_ * sizeof(float)); // diff
        pipe_.InitBuffer(tmpBuf2_, BUFFER_NUM, tileDim_ * sizeof(float)); // square
        // local accumulators
        pipe_.InitBuffer(localSum_, 1, sizeof(float));
        pipe_.InitBuffer(distAP_, 1, sizeof(float));
        pipe_.InitBuffer(distAN_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalSamples_ == 0 || dim_ == 0 || tileDim_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride loop over samples
        uint32_t sampleIdx = blockIdx;
        while (sampleIdx < totalSamples_) {
            ComputeSample(sampleIdx);
            sampleIdx += blockNum;
        }
    }

private:
    __aicore__ inline void ComputeSample(uint32_t sampleIdx)
    {
        // Compute L2 distances: dAP = sqrt(sum_i (a_i - p_i)^2), dAN = sqrt(sum_i (a_i - n_i)^2)
        // Then loss = max(dAP - dAN + margin, 0.0f)
        // Use tiling over dim dimension

        uint32_t baseOffset = sampleIdx * dim_;
        float dAP = 0.0f;
        float dAN = 0.0f;

        // Tile loop over dim
        uint32_t dimRemaining = dim_;
        uint32_t offset = 0;
        while (offset < dim_) {
            uint32_t curTile = (dimRemaining > tileDim_) ? tileDim_ : dimRemaining;

            // Load anchor, positive, negative for this tile
            AscendC::LocalTensor<float> aLocal = inQueAnchor_.AllocTensor<float>();
            AscendC::DataCopy(aLocal, anchorGm_[baseOffset + offset], curTile);
            inQueAnchor_.EnQue(aLocal);

            AscendC::LocalTensor<float> pLocal = inQuePos_.AllocTensor<float>();
            AscendC::DataCopy(pLocal, positiveGm_[baseOffset + offset], curTile);
            inQuePos_.EnQue(pLocal);

            AscendC::LocalTensor<float> nLocal = inQueNeg_.AllocTensor<float>();
            AscendC::DataCopy(nLocal, negativeGm_[baseOffset + offset], curTile);
            inQueNeg_.EnQue(nLocal);

            // Deque and compute diff, square, accumulate
            AscendC::LocalTensor<float> a = inQueAnchor_.DeQue<float>();
            AscendC::LocalTensor<float> p = inQuePos_.DeQue<float>();
            AscendC::LocalTensor<float> n = inQueNeg_.DeQue<float>();

            AscendC::LocalTensor<float> diffAP = tmpBuf1_.AllocTensor<float>();
            AscendC::LocalTensor<float> diffAN = tmpBuf2_.AllocTensor<float>(); // reuse different? 
            // Actually we need two separate diff buffers. Use allocated tmpBuf1 and tmpBuf2.
            // diffAP
            AscendC::Sub(diffAP, a, p, curTile);
            AscendC::Mul(diffAP, diffAP, diffAP, curTile); // square
            // diffAN
            AscendC::Sub(diffAN, a, n, curTile);
            AscendC::Mul(diffAN, diffAN, diffAN, curTile);

            // Sum tiles into local scalars
            float sumAP = 0.0f;
            float sumAN = 0.0f;
            // Use reduction intrinsic? AscendC has ReduceAdd? For simplicity, manual loop.
            // But we can use reduction via conv? For simplicity, we'll do a serial loop over tile.
            // For efficiency, use vbadd and then scalar? Use AddReduce? Not available.
            // We'll write a simple loop.
            for (uint32_t i = 0; i < curTile; i++) {
                sumAP += diffAP(i);
                sumAN += diffAN(i);
            }
            dAP += sumAP;
            dAN += sumAN;

            // Free buffers
            tmpBuf1_.FreeTensor(diffAP);
            tmpBuf2_.FreeTensor(diffAN);
            inQueAnchor_.FreeTensor(a);
            inQuePos_.FreeTensor(p);
            inQueNeg_.FreeTensor(n);

            offset += curTile;
            dimRemaining -= curTile;
        }

        // Take square root
        dAP = sqrt(dAP);
        dAN = sqrt(dAN);

        // Compute loss
        float loss = dAP - dAN + margin_;
        if (loss < 0.0f) loss = 0.0f;

        // Write output
        outputGm_.SetValue(sampleIdx, loss);
    }

private:
    AscendC::TPipe pipe_;
    // Input queues for anchor, positive, negative tiles
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQuePos_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueNeg_;
    // Output queue for loss (scalar per sample)
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueLoss_;
    // Temporary buffers
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpBuf1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpBuf2_;
    // Scalar local buffers (not used as queue)
    AscendC::TLocal<AscendC::TPosition::VECIN> localSum_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t totalSamples_ {0};
    uint32_t dim_ {0};
    uint32_t tileDim_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, output,
            tiling_data.totalSamples, tiling_data.dim, tiling_data.tileDim,
            tiling_data.margin);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'triplet_margin_loss_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, *inputs):
        # inputs: [anchor, positive, negative, margin]
        anchor, positive, negative, margin = inputs
        margin_scalar = float(margin.reshape(-1)[0].item()) if margin.numel() else 0.0
        fn = getattr(custom_ops_lib, _FN)
        per_sample_loss = fn(anchor, positive, negative, margin_scalar)
        # The original function returns a scalar (mean). Here we compute mean manually.
