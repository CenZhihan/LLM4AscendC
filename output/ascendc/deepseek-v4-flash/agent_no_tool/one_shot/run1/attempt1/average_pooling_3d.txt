#include "average_pooling_3d_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling3dCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    uint32_t n = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(0));
    uint32_t c = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(1));
    uint32_t d = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(2));
    uint32_t h = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(3));
    uint32_t w = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t kernelSize = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(0)[0]); // int stored as int64_t
    uint32_t stride = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(1)[0]);
    uint32_t padding = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(2)[0]);

    // Compute output dimensions
    uint32_t od = (d + 2 * padding - kernelSize) / stride + 1;
    uint32_t oh = (h + 2 * padding - kernelSize) / stride + 1;
    uint32_t ow = (w + 2 * padding - kernelSize) / stride + 1;

    tiling.inputShape[0] = n; tiling.inputShape[1] = c; tiling.inputShape[2] = d; tiling.inputShape[3] = h; tiling.inputShape[4] = w;
    tiling.outputShape[0] = n; tiling.outputShape[1] = c; tiling.outputShape[2] = od; tiling.outputShape[3] = oh; tiling.outputShape[4] = ow;
    tiling.kernelSize = kernelSize;
    tiling.stride = stride;
    tiling.padding = padding;

    uint32_t totalLength = n * c * od * oh * ow;
    tiling.totalLength = totalLength;

    // Simple tiling: each block processes a contiguous chunk of output elements
    uint32_t blockDim = 24; // conservative, can be tuned
    context->SetBlockDim(blockDim);
    uint32_t tileLength = (totalLength + blockDim - 1) / blockDim;
    tiling.tileLength = tileLength;
    tiling.blockDim = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    uint32_t n = x_shape->GetOriginShape().GetDim(0);
    uint32_t c = x_shape->GetOriginShape().GetDim(1);
    uint32_t d = x_shape->GetOriginShape().GetDim(2);
    uint32_t h = x_shape->GetOriginShape().GetDim(3);
    uint32_t w = x_shape->GetOriginShape().GetDim(4);

    // Need attrs? We'll get from context but for InferShape, we can read from the op attributes.
    // Since InferShape is called before tiling, we assume attrs are available.
    const gert::AttrHandle attr_handle = context->GetAttrHandle();
    int64_t kernelSize = attr_handle.GetAttr<int64_t>("kernel_size");
    int64_t stride = attr_handle.GetAttr<int64_t>("stride");
    int64_t padding = attr_handle.GetAttr<int64_t>("padding");
    
    uint32_t od = (d + 2 * static_cast<uint32_t>(padding) - static_cast<uint32_t>(kernelSize)) / static_cast<uint32_t>(stride) + 1;
    uint32_t oh = (h + 2 * static_cast<uint32_t>(padding) - static_cast<uint32_t>(kernelSize)) / static_cast<uint32_t>(stride) + 1;
    uint32_t ow = (w + 2 * static_cast<uint32_t>(padding) - static_cast<uint32_t>(kernelSize)) / static_cast<uint32_t>(stride) + 1;

    y_shape->SetDimNum(5);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, c);
    y_shape->SetDim(2, od);
    y_shape->SetDim(3, oh);
    y_shape->SetDim(4, ow);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelAveragePooling3dCustom {
public:
    __aicore__ inline KernelAveragePooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t inputShape[5], uint32_t outputShape[5],
                                uint32_t kernelSize, uint32_t stride, uint32_t padding,
                                uint32_t totalLength, uint32_t tileLength)
    {
        inputN_ = inputShape[0]; inputC_ = inputShape[1]; inputD_ = inputShape[2]; inputH_ = inputShape[3]; inputW_ = inputShape[4];
        outputN_ = outputShape[0]; outputC_ = outputShape[1]; outputD_ = outputShape[2]; outputH_ = outputShape[3]; outputW_ = outputShape[4];
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        totalLength_ = totalLength;
        tileLength_ = tileLength;

        // Compute strides for 5D tensors (in elements)
        inputStrideN_ = inputC_ * inputD_ * inputH_ * inputW_;
        inputStrideC_ = inputD_ * inputH_ * inputW_;
        inputStrideD_ = inputH_ * inputW_;
        inputStrideH_ = inputW_;
        inputStrideW_ = 1;

        outputStrideN_ = outputC_ * outputD_ * outputH_ * outputW_;
        outputStrideC_ = outputD_ * outputH_ * outputW_;
        outputStrideD_ = outputH_ * outputW_;
        outputStrideH_ = outputW_;
        outputStrideW_ = 1;

        xGm_ = (__gm__ float*)x;
        yGm_ = (__gm__ float*)y;
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockDim = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t startIdx = blockIdx * tileLength_;
        uint32_t endIdx = startIdx + tileLength_;
        if (startIdx >= totalLength_) return;
        if (endIdx > totalLength_) endIdx = totalLength_;

        for (uint32_t idx = startIdx; idx < endIdx; ++idx) {
            // Convert flat index to (n, c, od, oh, ow)
            uint32_t tmp = idx;
            uint32_t ow = tmp % outputW_; tmp /= outputW_;
            uint32_t oh = tmp % outputH_; tmp /= outputH_;
            uint32_t od = tmp % outputD_; tmp /= outputD_;
            uint32_t c = tmp % outputC_; tmp /= outputC_;
            uint32_t n = tmp;

            // Compute input starting position (top-left of kernel)
            int32_t dStart = static_cast<int32_t>(od * stride_) - static_cast<int32_t>(padding_);
            int32_t hStart = static_cast<int32_t>(oh * stride_) - static_cast<int32_t>(padding_);
            int32_t wStart = static_cast<int32_t>(ow * stride_) - static_cast<int32_t>(padding_);

            float sum = 0.0f;
            uint32_t count = 0;

            for (uint32_t kd = 0; kd < kernelSize_; ++kd) {
                int32_t dIn = dStart + static_cast<int32_t>(kd);
                if (dIn < 0 || dIn >= static_cast<int32_t>(inputD_)) continue;
                uint32_t ud = static_cast<uint32_t>(dIn);
                for (uint32_t kh = 0; kh < kernelSize_; ++kh) {
                    int32_t hIn = hStart + static_cast<int32_t>(kh);
                    if (hIn < 0 || hIn >= static_cast<int32_t>(inputH_)) continue;
                    uint32_t uh = static_cast<uint32_t>(hIn);
                    for (uint32_t kw = 0; kw < kernelSize_; ++kw) {
                        int32_t wIn = wStart + static_cast<int32_t>(kw);
                        if (wIn < 0 || wIn >= static_cast<int32_t>(inputW_)) continue;
                        uint32_t uw = static_cast<uint32_t>(wIn);

                        // Compute flat input index
                        uint32_t inputIdx = n * inputStrideN_ + c * inputStrideC_ +
                                            ud * inputStrideD_ + uh * inputStrideH_ + uw;
                        sum += xGm_[inputIdx];
                        ++count;
                    }
                }
            }

            // Compute output index and store average
            uint32_t outputIdx = n * outputStrideN_ + c * outputStrideC_ +
                                 od * outputStrideD_ + oh * outputStrideH_ + ow;
            yGm_[outputIdx] = (count > 0) ? (sum / static_cast<float>(count)) : 0.0f;
        }
    }

private:
    uint32_t inputN_, inputC_, inputD_, inputH_, inputW_;
    uint32_t outputN_, outputC_, outputD_, outputH_, outputW_;
    uint32_t kernelSize_, stride_, padding_;
    uint32_t totalLength_, tileLength_;

    uint32_t inputStrideN_, inputStrideC_, inputStrideD_, inputStrideH_, inputStrideW_;
    uint32_t outputStrideN_, outputStrideC_, outputStrideD_, outputStrideH_, outputStrideW_;

    __gm__ float* xGm_;
    __gm__ float* yGm_;
};

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling3dCustom op;
    op.Init(x, y, tiling_data.inputShape, tiling_data.outputShape,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
