```cpp
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MATMUL_ALIGN = 16;

class KernelGemmScalingHardTanhGeluCustom {
public:
    __aicore__ inline KernelGemmScalingHardTanhGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t blockSizeM, uint32_t blockSizeN, uint32_t blockSizeK,
                                float scaling, float hardtanh_min, float hardtanh_max)
    {
        M_ = M; N_ = N; K_ = K;
        blockM_ = blockSizeM; blockN_ = blockSizeN; blockK_ = blockSizeK;
        scaling_ = scaling;
        htanMin_ = hardtanh_min;
        htanMax_ = hardtanh_max;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M * K);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N * K);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M * N);

        // Initialize cube matmul
        matmul_.SetTensorA(xGm_, M, K, false, false); // A not transposed
        matmul_.SetTensorB(wGm_, N, K, true, false); // B is transposed (W^T) -> W is [N,K], B transposed gives [K,N]? Actually for x * W^T, we set B as W with transpose: W shape [N,K], so when B is transposed, it's [K,N]. Then C = A * B => MxN.
        matmul_.SetBias(bGm_); // bias shape [N] broadcasts along M
        matmul_.SetTensorC(yGm_, M, N);

        // Set tile sizes
        matmul_.SetLoopConfig(blockM_, blockN_, blockK_);

        // Init pipe for vector operations (scaling, hardtanh, gelu) on output tiles
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, blockM_ * blockN_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, blockM_ * blockN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, blockM_ * blockN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        // Iterate over block rows (M dimension) and columns (N dimension)
        // We'll distribute blocks across cores.
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        uint32_t blocksM = (M_ + blockM_ - 1) / blockM_;
        uint32_t blocksN = (N_ + blockN_ - 1) / blockN_;
        uint32_t totalBlocks = blocksM * blocksN;
        uint32_t myBlockIdx = blockIdx;
        while (myBlockIdx < totalBlocks) {
            uint32_t mBlock = myBlockIdx / blocksN;
            uint32_t nBlock = myBlockIdx % blocksN;
            uint32_t mStart = mBlock * blockM_;
            uint32_t nStart = nBlock * blockN_;
            uint32_t mEnd = min(mStart + blockM_, M_);
            uint32_t nEnd = min(nStart + blockN_, N_);
            uint32_t curM = mEnd - mStart;
            uint32_t curN = nEnd - nStart;

            // Perform matmul for this block (partial leading dimension)
            // Use cube compute
            matmul_.SetTensorA(xGm_.Slice(mStart, 0, curM, K), curM, K, false, false);
            matmul_.SetTensorB(wGm_.Slice(nStart, 0, curN, K), curN, K, true, false);
            matmul_.SetBias(bGm_.Slice(nStart, curN));
            matmul_.SetTensorC(yGm_.Slice(mStart, nStart, curM, curN), curM, curN);
            matmul_.IterateAll(); // Cube compute

            // Now yGm_ block has result of matmul + bias. Now apply scaling, hardtanh, gelu
            // Load block into local
            AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, yGm_[mStart * N_ + nStart], curM * curN);

            // scaling
            AscendC::Muls(yLocal, yLocal, scaling_, curM * curN);
            // hardtanh
            AscendC::Clamp(yLocal, yLocal, htanMin_, htanMax_, curM * curN);
            // gelu
            AscendC::Gelu(yLocal, yLocal, curM * curN);

            // store back
            AscendC::DataCopy(yGm_[mStart * N_ + nStart], yLocal, curM * curN);
            inQueueY_.FreeTensor(yLocal);

            myBlockIdx += blockNum;
        }
    }

private:
    AscendC::Matmul<float, float, float> matmul_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    uint32_t M_, N_, K_;
    uint32_t blockM_, blockN_, blockK_;
    float scaling_, htanMin_, htanMax_;
};

extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
```
