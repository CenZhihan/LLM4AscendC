namespace optiling {

static constexpr uint32_t UB_BUDGET = 192 * 1024;
static constexpr uint32_t ALIGN = 32;

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  VanillaRnnHiddenCustomTilingData tiling;

  const gert::Shape *xShape = context->GetInputShape(0);
  const gert::Shape *hShape = context->GetInputShape(1);
  auto batch = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
  auto inputSize = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
  auto hiddenSize = static_cast<uint32_t>(hShape->GetOriginShape().GetDim(1));
  auto K = inputSize + hiddenSize;

  const gert::Shape *w_h2oShape = context->GetInputShape(4);
  auto outputSize = static_cast<uint32_t>(w_h2oShape->GetOriginShape().GetDim(0));

  // tileBatch = min(batch, UB_BUDGET / (K*sizeof(float)*2 + hiddenSize*sizeof(float)*2 + outputSize*sizeof(float)*2))
  uint32_t tileBatch = batch;
  uint32_t neededPerBatch = (K * 2 + hiddenSize * 2 + outputSize * 2) * sizeof(float);
  uint32_t maxTileBatch = UB_BUDGET / (neededPerBatch > 0 ? neededPerBatch : 1);
  if (maxTileBatch < 1) maxTileBatch = 1;
  if (maxTileBatch > batch) maxTileBatch = batch;
  // Align down to 1 (no alignment requirement for batch)
  tileBatch = maxTileBatch;

  context->SetBlockDim(tileBatch);  // each block processes tileBatch rows

  tiling.set_batch(batch);
  tiling.set_inputSize(inputSize);
  tiling.set_hiddenSize(hiddenSize);
  tiling.set_outputSize(outputSize);
  tiling.set_K(K);
  tiling.set_tileBatch(tileBatch);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *workspace = context->GetWorkspaceSizes(1);
  workspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}
} // namespace optiling

kernel_src = '''
#include "kernel_operator.h"
#include "cblas.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
  __aicore__ inline KernelVanillaRnnHiddenCustom() {}

  __aicore__ inline void Init(GM_ADDR x_t, GM_ADDR h, GM_ADDR w_i2h, GM_ADDR b_i2h,
                               GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR h_new, GM_ADDR y_t,
                               uint32_t batch, uint32_t inputSize, uint32_t hiddenSize,
                               uint32_t outputSize, uint32_t K, uint32_t tileBatch) {
    batch_ = batch;
    inputSize_ = inputSize;
    hiddenSize_ = hiddenSize;
    outputSize_ = outputSize;
    K_ = K;
    tileBatch_ = tileBatch;

    xGm_.SetGlobalBuffer((__gm__ float*)x_t, batch * inputSize);
    hGm_.SetGlobalBuffer((__gm__ float*)h, batch * hiddenSize);
    w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, hiddenSize * K);
    b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, hiddenSize);
    w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, outputSize * hiddenSize);
    b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, outputSize);
    hNewGm_.SetGlobalBuffer((__gm__ float*)h_new, batch * hiddenSize);
    yTGm_.SetGlobalBuffer((__gm__ float*)y_t, batch * outputSize);

    // allocate local buffers
    pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileBatch * inputSize_ * sizeof(float));
    pipe_.InitBuffer(inQueueH_, BUFFER_NUM, tileBatch * hiddenSize_ * sizeof(float));
    pipe_.InitBuffer(inQueueW_i2h_, 1, hiddenSize_ * K_ * sizeof(float));
    pipe_.InitBuffer(inQueueB_i2h_, 1, hiddenSize_ * sizeof(float));
    pipe_.InitBuffer(inQueueW_h2o_, 1, outputSize_ * hiddenSize_ * sizeof(float));
    pipe_.InitBuffer(inQueueB_h2o_, 1, outputSize_ * sizeof(float));
    pipe_.InitBuffer(outQueueH_, BUFFER_NUM, tileBatch * hiddenSize_ * sizeof(float));
    pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileBatch * outputSize_ * sizeof(float));
    pipe_.InitBuffer(tmpQueue_, 1, tileBatch * K_ * sizeof(float));
  }

  __aicore__ inline void Process() {
    uint32_t blockIdx = GetBlockIdx();
    uint32_t numBlocks = GetBlockNum();
    uint32_t startRow = blockIdx * tileBatch_;
    while (startRow < batch_) {
      uint32_t curBatch = (startRow + tileBatch_ > batch_) ? (batch_ - startRow) : tileBatch_;
      // Load weights once (shared by all tiles)
      AscendC::LocalTensor<float> w_i2hLocal = inQueueW_i2h_.AllocTensor<float>();
      AscendC::DataCopy(w_i2hLocal, w_i2hGm_, hiddenSize_ * K_);
      AscendC::LocalTensor<float> b_i2hLocal = inQueueB_i2h_.AllocTensor<float>();
      AscendC::DataCopy(b_i2hLocal, b_i2hGm_, hiddenSize_);
      AscendC::LocalTensor<float> w_h2oLocal = inQueueW_h2o_.AllocTensor<float>();
      AscendC::DataCopy(w_h2oLocal, w_h2oGm_, outputSize_ * hiddenSize_);
      AscendC::LocalTensor<float> b_h2oLocal = inQueueB_h2o_.AllocTensor<float>();
      AscendC::DataCopy(b_h2oLocal, b_h2oGm_, outputSize_);

      // Load x_t and h for this tile
      AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
      AscendC::DataCopy(xLocal, xGm_[startRow * inputSize_], curBatch * inputSize_);
      AscendC::LocalTensor<float> hLocal = inQueueH_.AllocTensor<float>();
      AscendC::DataCopy(hLocal, hGm_[startRow * hiddenSize_], curBatch * hiddenSize_);

      // Concat: create local tensor for cat result (batch, K)
      AscendC::LocalTensor<float> catLocal = tmpQueue_.AllocTensor<float>();
      AscendC::LocalTensor<float> catLocal2 = tmpQueue_.AllocTensor<float>(); // for final matmul
      // Copy x part
      AscendC::DataCopy(catLocal, xLocal, curBatch * inputSize_);
      // Copy h part at offset inputSize_ per row
      for (uint32_t i = 0; i < curBatch; ++i) {
        AscendC::DataCopy(catLocal[i * K_ + inputSize_], hLocal[i * hiddenSize_], hiddenSize_);
      }

      // MatMul: cat (B,K) * w_i2h^T (K,H) -> pre (B,H)
      // Use AscendC::MatMul; assume matrices row-major, weights are stored as HxK? We have w_i2h in original shape HxK, so transpose.
      // We'll do a batched matmul using CBLAS? Simpler: use cube unit.
      // Here we use a hypothetical API.
      AscendC::LocalTensor<float> preLocal = outQueueH_.AllocTensor<float>();
      // matmul: cat (B,K) x w_i2h (K,H) => pre (B,H)
      AscendC::MatMul(preLocal, catLocal, false, w_i2hLocal, true, curBatch, hiddenSize_, K_,
                      K_, K_, hiddenSize_, curBatch * K_, curBatch * hiddenSize_, 1.0f);
      // Add bias
      for (uint32_t i = 0; i < curBatch; ++i) {
        AscendC::Add(preLocal[i * hiddenSize_], preLocal[i * hiddenSize_], b_i2hLocal, hiddenSize_);
      }
      // Tanh
      AscendC::Tanh(preLocal, preLocal, curBatch * hiddenSize_);
      // Store h_new
      AscendC::DataCopy(hNewGm_[startRow * hiddenSize_], preLocal, curBatch * hiddenSize_);

      // Output projection: pre (B,H) x w_h2o^T (H,O) -> y (B,O)
      AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
      AscendC::MatMul(yLocal, preLocal, false, w_h2oLocal, true, curBatch, outputSize_, hiddenSize_,
                      hiddenSize_, hiddenSize_, outputSize_, curBatch * hiddenSize_, curBatch * outputSize_, 1.0f);
      // Add bias
      for (uint32_t i = 0; i < curBatch; ++i) {
        AscendC::Add(yLocal[i * outputSize_], yLocal[i * outputSize_], b_h2oLocal, outputSize_);
      }
      AscendC::DataCopy(yTGm_[startRow * outputSize_], yLocal, curBatch * outputSize_);

      // Free tensors
      inQueueX_.FreeTensor(xLocal);
      inQueueH_.FreeTensor(hLocal);
      tmpQueue_.FreeTensor(catLocal);
      tmpQueue_.FreeTensor(catLocal2);
      inQueueW_i2h_.FreeTensor(w_i2hLocal);
      inQueueB_i2h_.FreeTensor(b_i2hLocal);
      inQueueW_h2o_.FreeTensor(w_h2oLocal);
      inQueueB_h2o_.FreeTensor(b_h2oLocal);
      outQueueH_.FreeTensor(preLocal);
      outQueueY_.FreeTensor(yLocal);

      startRow += numBlocks * tileBatch_;
    }
  }

private:
  AscendC::TPipe pipe_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
  AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueH_;
  AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_i2h_;
  AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_i2h_;
  AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_h2o_;
  AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_h2o_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueH_;
  AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
  AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

  AscendC::GlobalTensor<float> xGm_, hGm_, w_i2hGm_, b_i2hGm_, w_h2oGm_, b_h2oGm_, hNewGm_, yTGm_;
  uint32_t batch_, inputSize_, hiddenSize_, outputSize_, K_, tileBatch_;
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(
    GM_ADDR x_t, GM_ADDR h, GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR h_new, GM_ADDR y_t,
    GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelVanillaRnnHiddenCustom op;
  op.Init(x_t, h, w_i2h, b_i2h, w_h2o, b_h2o, h_new, y_t,
          tiling_data.batch, tiling_data.inputSize, tiling_data.hiddenSize,
          tiling_data.outputSize, tiling_data.K, tiling_data.tileBatch);
  op.Process();
}
'''

import torch
import torch_npu
import custom_ops_lib

_FN = 'vanilla_rnn_hidden_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super().__init__()

    def forward(self, x, h0, w_i2h, b_i2h, w_h2o, b_h2o):
        T, B, I = x.shape
        H = h0.shape[1]
        O = w_h2o.shape[0]
        h = h0
        outs = []
        for t in range(T):
            h, y_t = custom_ops_lib.vanilla_rnn_hidden_custom(
                x[t], h, w_i2h, b_i2h, w_h2o, b_h2o)
            outs.append(y_t)
        return torch.stack(outs, dim=0)
