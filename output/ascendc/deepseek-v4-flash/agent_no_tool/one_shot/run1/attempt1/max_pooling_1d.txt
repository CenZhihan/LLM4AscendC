kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalSeqs, uint32_t seqLen,
                                uint32_t outLen, uint32_t tileSeqs,
                                int32_t kernelSize, int32_t stride, int32_t padding)
    {
        totalSeqs_ = totalSeqs;
        seqLen_ = seqLen;
        outLen_ = outLen;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;

        // Each sequence stored contiguously: input layout is [batch, features, seqLen]
        // But we treat totalSeqs independent sequences.
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        // Output buffer: output size per sequence = outLen_, total outputs = totalSeqs_ * outLen_
        // We need to compute linear offset for each sequence.
        // We'll compute offsets based on seqLen and outLen strides.
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Buffer for one sequence input (including padding) and output
        uint32_t paddedLen = seqLen + 2 * padding;
        inputBufLen_ = paddedLen;
        outputBufLen_ = outLen;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputBufLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outputBufLen_ * sizeof(float));
        // Temporary buffer for window max computation (size kernelSize)
        pipe_.InitBuffer(tmpQueue_, 1, kernelSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalSeqs_ == 0 || outLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over sequences
        for (uint32_t seqBase = blockIdx; seqBase < totalSeqs_; seqBase += blockNum) {
            // Compute input and output global offsets
            uint32_t inputOffset = seqBase * seqLen_;   // input stride = seqLen_ (original without padding)
            uint32_t outputOffset = seqBase * outLen_;  // output stride = outLen_

            // Load one sequence input (padded by zeros on both sides) into local
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();

            // Zero-initialize the padded buffer
            AscendC::Duplicate(xLocal, 0.0f, inputBufLen_);
            // Copy central part from global (non-padded) into xLocal at offset padding_
            AscendC::DataCopy(xLocal[padding_], xGm_[inputOffset], seqLen_);

            // Compute max pooling over this sequence
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            for (uint32_t o = 0; o < outLen_; o++) {
                uint32_t start = o * stride_;
                // Load window into tLocal
                for (uint32_t k = 0; k < kernelSize_; k++) {
                    tLocal(k) = xLocal(start + k);
                }
                // Compute max across window
                float maxVal = tLocal(0);
                for (uint32_t k = 1; k < kernelSize_; k++) {
                    if (tLocal(k) > maxVal) maxVal = tLocal(k);
                }
                yLocal(o) = maxVal;
            }

            // Write output to global
            AscendC::DataCopy(yGm_[outputOffset], yLocal, outLen_);

            // Free tensors
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalSeqs_ {0};
    uint32_t seqLen_ {0};
    uint32_t outLen_ {0};
    uint32_t inputBufLen_ {0};
    uint32_t outputBufLen_ {0};
    int32_t kernelSize_ {8};
    int32_t stride_ {1};
    int32_t padding_ {0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y, tiling_data.totalSeqs, tiling_data.seqLen,
            tiling_data.outLen, tiling_data.tileSeqs,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding);
    op.Process();
}
