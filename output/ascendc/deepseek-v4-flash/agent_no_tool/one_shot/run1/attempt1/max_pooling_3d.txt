#include "max_pooling_3d_custom_tiling.h"

namespace optiling {

static inline uint32_t ComputeSize(uint32_t in, uint32_t k, uint32_t s, uint32_t p, bool ceilMode) {
    int32_t temp = static_cast<int32_t>(in + 2 * p) - static_cast<int32_t>(k);
    int32_t out = temp / static_cast<int32_t>(s) + 1;
    if (ceilMode && (temp % static_cast<int32_t>(s) != 0)) {
        out += 1;
    }
    if (out < 0) out = 0;
    return static_cast<uint32_t>(out);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    tiling.n = shape.GetDim(0);
    tiling.c = shape.GetDim(1);
    tiling.dIn = shape.GetDim(2);
    tiling.hIn = shape.GetDim(3);
    tiling.wIn = shape.GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const auto *kernel_size = attrs->GetListInt(0);
    const auto *stride = attrs->GetListInt(1);
    const auto *padding = attrs->GetListInt(2);
    bool ceilMode = attrs->GetAttrBool(3);

    tiling.kD = static_cast<uint32_t>(kernel_size->GetData()[0]);
    tiling.kH = static_cast<uint32_t>(kernel_size->GetData()[1]);
    tiling.kW = static_cast<uint32_t>(kernel_size->GetData()[2]);
    tiling.sD = static_cast<uint32_t>(stride->GetData()[0]);
    tiling.sH = static_cast<uint32_t>(stride->GetData()[1]);
    tiling.sW = static_cast<uint32_t>(stride->GetData()[2]);
    tiling.pD = static_cast<uint32_t>(padding->GetData()[0]);
    tiling.pH = static_cast<uint32_t>(padding->GetData()[1]);
    tiling.pW = static_cast<uint32_t>(padding->GetData()[2]);
    tiling.ceilMode = ceilMode;

    tiling.dOut = ComputeSize(tiling.dIn, tiling.kD, tiling.sD, tiling.pD, ceilMode);
    tiling.hOut = ComputeSize(tiling.hIn, tiling.kH, tiling.sH, tiling.pH, ceilMode);
    tiling.wOut = ComputeSize(tiling.wIn, tiling.kW, tiling.sW, tiling.pW, ceilMode);
    tiling.totalOut = tiling.n * tiling.c * tiling.dOut * tiling.hOut * tiling.wOut;

    // Set blockDim based on parallelism (could be made adaptive)
    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);
    tiling.blockDim = blockDim;

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t dIn, uint32_t hIn, uint32_t wIn,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t sD, uint32_t sH, uint32_t sW,
                                uint32_t pD, uint32_t pH, uint32_t pW,
                                uint32_t dOut, uint32_t hOut, uint32_t wOut,
                                uint32_t totalOut, bool ceilMode)
    {
        n_ = n; c_ = c;
        dIn_ = dIn; hIn_ = hIn; wIn_ = wIn;
        kD_ = kD; kH_ = kH; kW_ = kW;
        sD_ = sD; sH_ = sH; sW_ = sW;
        pD_ = pD; pH_ = pH; pW_ = pW;
        dOut_ = dOut; hOut_ = hOut; wOut_ = wOut;
        totalOut_ = totalOut;
        ceilMode_ = ceilMode;

        xGm_.SetGlobalBuffer((__gm__ float*)x, 1); // dummy, we manage offsets manually
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);
        // Instead of using GlobalTensor with offset, we will use pointer arithmetic directly.
        xPtr_ = (__gm__ float*)x;
        yPtr_ = (__gm__ float*)y;

        // Tile size: process a block of output elements per iteration.
        // For simplicity, use a fixed tile size based on UB budget.
        // Each iteration processes up to tileOut_ output elements.
        // We'll allocate buffers for one output tile and for reading a window.
        tileOut_ = 1024; // tuned
        pipe_.InitBuffer(inQueueWindow_, BUFFER_NUM, tileOut_ * kD_ * kH_ * kW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpMax_, 1, tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpIdx_, 1, tileOut_ * sizeof(float)); // not used for indices
    }

    __aicore__ inline void Process()
    {
        uint32_t blockDim = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t totalOutPerBlock = (totalOut_ + blockDim - 1) / blockDim;
        uint32_t startIdx = blockIdx * totalOutPerBlock;
        uint32_t endIdx = startIdx + totalOutPerBlock;
        if (startIdx >= totalOut_) return;
        if (endIdx > totalOut_) endIdx = totalOut_;

        // Iterate over output elements in tiles
        for (uint32_t idx = startIdx; idx < endIdx; idx += tileOut_) {
            uint32_t curTile = (endIdx - idx) < tileOut_ ? (endIdx - idx) : tileOut_;

            // Compute global output coordinates for each element in tile (simplified: compute for first element and then increment)
            // We'll compute for each element individually in the kernel loop.
            // But we need to load the input windows. Instead, we process one output element at a time inside a loop.
            // For Ascend C vectorization, we can't have per-element branching easily. So we'll unroll over the tile and use scalar loops.
            // This kernel is for demonstration; a more efficient version would use 2D tiling.
            // We'll use a scalar loop for clarity.

            // For each output element in the tile
            for (uint32_t i = 0; i < curTile; i++) {
                uint32_t outIdx = idx + i;
                // Convert flat index to N, C, D, H, W
                uint32_t wOutIdx = outIdx % wOut_;
                uint32_t remainder = outIdx / wOut_;
                uint32_t hOutIdx = remainder % hOut_;
                remainder /= hOut_;
                uint32_t dOutIdx = remainder % dOut_;
                remainder /= dOut_;
                uint32_t cIdx = remainder % c_;
                uint32_t nIdx = remainder / c_;

                // Compute input window bounds
                int32_t dStart = static_cast<int32_t>(dOutIdx * sD_) - static_cast<int32_t>(pD_);
                int32_t hStart = static_cast<int32_t>(hOutIdx * sH_) - static_cast<int32_t>(pH_);
                int32_t wStart = static_cast<int32_t>(wOutIdx * sW_) - static_cast<int32_t>(pW_);

                float maxVal = -FLT_MAX;
                for (uint32_t kd = 0; kd < kD_; kd++) {
                    int32_t d = dStart + kd;
                    if (d < 0 || d >= static_cast<int32_t>(dIn_)) continue;
                    for (uint32_t kh = 0; kh < kH_; kh++) {
                        int32_t h = hStart + kh;
                        if (h < 0 || h >= static_cast<int32_t>(hIn_)) continue;
                        for (uint32_t kw = 0; kw < kW_; kw++) {
                            int32_t w = wStart + kw;
                            if (w < 0 || w >= static_cast<int32_t>(wIn_)) continue;
                            uint32_t inIdx = ((nIdx * c_ + cIdx) * dIn_ + d) * hIn_ + h) * wIn_ + w;
                            float val = xPtr_[inIdx];
                            if (val > maxVal) maxVal = val;
                        }
                    }
                }
                yPtr_[outIdx] = maxVal;
            }
        }
    }

private:
    uint32_t n_, c_, dIn_, hIn_, wIn_;
    uint32_t kD_, kH_, kW_;
    uint32_t sD_, sH_, sW_;
    uint32_t pD_, pH_, pW_;
    uint32_t dOut_, hOut_, wOut_;
    uint32_t totalOut_;
    bool ceilMode_;

    __gm__ float *xPtr_;
    __gm__ float *yPtr_;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWindow_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpIdx_;

    uint32_t tileOut_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.dIn, tiling_data.hIn, tiling_data.wIn,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.sD, tiling_data.sH, tiling_data.sW,
            tiling_data.pD, tiling_data.pH, tiling_data.pW,
            tiling_data.dOut, tiling_data.hOut, tiling_data.wOut,
            tiling_data.totalOut, tiling_data.ceilMode);
    op.Process();
}

import torch

_FN = 'max_pooling_3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0, return_indices: bool = False, ceil_mode: bool = False):

        self.kernel_size = kernel_size if isinstance(kernel_size, (list, tuple)) else [kernel_size]*3

        if not isinstance(self.stride, (list, tuple)):

        self.padding = padding if isinstance(padding, (list, tuple)) else [padding]*3
        self.ceil_mode = ceil_mode

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding, self.ceil_mode)
