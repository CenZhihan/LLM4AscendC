#include "conv_transposed_3d_square_input_asymmetric_kernel_custom_tiling.h"

namespace optiling {

// UB model for this kernel:
// Input tile: simd load input block, weight block, accumulator tile.
// We conservatively estimate 8 * tileLength floats for temp buffers.
// UB budget 192KB -> elements = 192*1024 / 4 = 49152.
// tileLength chosen as approx 1024 elements per core.
static constexpr uint32_t UB_BUDGET_ELEMS = 49152;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed3dSquareInputAsymmetricKernelCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetOutputShape(0)->GetOriginShape().GetShapeSize());

    // Get input shapes
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    // x: N, C_in, D, H, W  (ND)
    uint32_t N = x_shape->GetDim(0);
    uint32_t C_in = x_shape->GetDim(1);
    uint32_t D_in = x_shape->GetDim(2);
    uint32_t H_in = x_shape->GetDim(3);
    uint32_t W_in = x_shape->GetDim(4);
    // w: C_out, C_in, kD, kH, kW
    uint32_t C_out = w_shape->GetDim(0);
    uint32_t kD = w_shape->GetDim(2);
    uint32_t kH = w_shape->GetDim(3);
    uint32_t kW = w_shape->GetDim(4);

    // Assume stride=1, padding=0, dilation=1, output_padding=0, groups=1 (from attributes)
    uint32_t D_out = D_in + kD - 1;
    uint32_t H_out = H_in + kH - 1;
    uint32_t W_out = W_in + kW - 1;

    // tileLength from UB budget per core
    uint32_t maxTile = UB_BUDGET_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }

    // Determine blockDim based on number of tiles
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C_out(C_out);
    tiling.set_D_out(D_out);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_C_in(C_in);
    tiling.set_D_in(D_in);
    tiling.set_H_in(H_in);
    tiling.set_W_in(W_in);
    tiling.set_kD(kD);
    tiling.set_kH(kH);
    tiling.set_kW(kW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

{
    const gert::Shape *x_shape = context->GetInputShape(0);

    uint32_t W_in = x_shape->GetDim(4);

    uint32_t kW = w_shape->GetDim(4);
