host_operator_src="""

#include "frobenius_norm_custom_tiling.h"
#include "register/op_def_registry.h"
#include "kernel_operator.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    FrobeniusNormCustomTilingData tiling;
    const uint32_t totalLength = static_cast<uint32_t>(
        context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // UB budget: we need 2 tiles (input + output) + 1 temp for reduction
    // Elements per tile: 3 * tileLen floats => 12 * tileLen bytes
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // Determine blockDim: at least 8, at most 24, based on number of tiles
    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    // Workspace for partial reduction results: blockDim floats + 1 float for norm
    currentWorkspace[0] = (blockDim + 1) * sizeof(float);
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    *context->GetOutputShape(0) = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class FrobeniusNormCustom : public OpDef {
public:
    explicit FrobeniusNormCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(FrobeniusNormCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

// Kernel 1: compute partial sums of squares and store in workspace
class KernelFrobeniusNormReduce {
public:
    __aicore__ inline KernelFrobeniusNormReduce() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR workspace,
                                uint32_t totalLength, uint32_t tileLength) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        partialSumsGm_.SetGlobalBuffer((__gm__ float*)workspace, blockDim_ + 1);
        blockDim_ = AscendC::GetBlockNum();
        blockIdx_ = AscendC::GetBlockIdx();
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        uint32_t stride = blockDim_ * tileLen_;
        uint32_t start = blockIdx_ * tileLen_;
        float localSum = 0.0f;

        while (start < totalLen_) {
            uint32_t len = totalLen_ - start;
            if (len > tileLen_) len = tileLen_;

            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[start], len);

            // Square elements
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(tmp, xLocal, xLocal, len);

            // Reduce within tile (simple loop for demo; for efficiency use reduction API)
            for (uint32_t i = 0; i < len; ++i) {
                localSum += tmp(i);
            }

            inQueueX_.FreeTensor(xLocal);
            tmpQueue_.FreeTensor(tmp);

            start += stride;
        }

        // Write partial sum to global memory
        partialSumsGm_.SetValue(blockIdx_, localSum);
        __syncthreads();  // sync within block (but needed for correctness of reduction)
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> partialSumsGm_;
    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t blockDim_;
    uint32_t blockIdx_;
};

// Kernel 2: compute norm from partial sums and divide
class KernelFrobeniusNormDivide {
public:
    __aicore__ inline KernelFrobeniusNormDivide() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace,
                                uint32_t totalLength, uint32_t tileLength) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);
        partialSumsGm_.SetGlobalBuffer((__gm__ float*)workspace, blockDim_ + 1);
        blockDim_ = AscendC::GetBlockNum();
        blockIdx_ = AscendC::GetBlockIdx();
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        // Block 0 computes global norm from partial sums and writes to workspace[blockDim]
        if (blockIdx_ == 0) {
            float totalSum = 0.0f;
            for (uint32_t i = 0; i < blockDim_; ++i) {
                totalSum += partialSumsGm_.GetValue(i);
            }
            float norm = AscendC::Sqrt(totalSum);
            partialSumsGm_.SetValue(blockDim_, norm);
        }
        __syncthreads();  // ensure norm is written before reading

        // All blocks read the norm
        float norm = partialSumsGm_.GetValue(blockDim_);

        // Now divide tiles
        uint32_t stride = blockDim_ * tileLen_;
        uint32_t start = blockIdx_ * tileLen_;

        while (start < totalLen_) {
            uint32_t len = totalLen_ - start;
            if (len > tileLen_) len = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[start], len);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Div(yLocal, xLocal, norm, len);

            AscendC::DataCopy(yGm_[start], yLocal, len);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);

            start += stride;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> partialSumsGm_;
    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t blockDim_;
    uint32_t blockIdx_;
};

// Combined kernel entry: we perform two phases sequentially
extern "C" __global__ __aicore__ void frobenius_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    uint32_t totalLength = tiling_data.totalLength;
    uint32_t tileLength = tiling_data.tileLength;

    // Phase 1: reduction
    KernelFrobeniusNormReduce reduce;
    reduce.Init(x, workspace, totalLength, tileLength);
    reduce.Process();

    // Phase 2: division
    KernelFrobeniusNormDivide divide;
    divide.Init(x, y, workspace, totalLength, tileLength);
    divide.Process();
}
