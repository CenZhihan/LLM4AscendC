static constexpr uint32_t MAX_TILE_ELEMS = 32768; // avoid extreme latency

    uint32_t totalLength = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    uint32_t normalizedSize = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());
