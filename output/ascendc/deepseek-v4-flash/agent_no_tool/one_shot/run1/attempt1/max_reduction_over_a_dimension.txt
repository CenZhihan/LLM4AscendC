#include "max_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: input chunk: tileOuter * CHUNK_SIZE * 4 bytes
//            output buffer: tileOuter * 4 bytes
//            total bytes ~ 4 * tileOuter * (CHUNK_SIZE + 1)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t CHUNK_SIZE = 64;  // vector width

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxReductionOverADimensionCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t numDims = x_shape->GetDimNum();
    const int64_t *dims = x_shape->GetDim();

    // Get attribute dim (default -1 means last dim)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *attrDim = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *attrDim;
    if (dim < 0) {
        dim += numDims;
    }
    // Convert to positive 0-based dim
    uint32_t reductionDim = static_cast<uint32_t>(dim);

    // Compute outer size (product of all dims except reduction dim)
    uint32_t totalOutputLength = 1;
    for (uint32_t i = 0; i < numDims; ++i) {
        if (i != reductionDim) {
            totalOutputLength *= static_cast<uint32_t>(dims[i]);
        }
    }
    uint32_t reductionLength = static_cast<uint32_t>(dims[reductionDim]);

    // Compute tileOuter from UB budget:
    // Input chunk: tileOuter * CHUNK_SIZE * 4 bytes
    // Output buffer: tileOuter * 4 bytes
    uint32_t maxBytes = UB_BUDGET_BYTES;
    uint32_t elemPerOuter = CHUNK_SIZE + 1;
    uint32_t maxTileOuter = maxBytes / (elemPerOuter * sizeof(float));
    maxTileOuter = AlignDown(maxTileOuter, ALIGN_ELEMS);
    if (maxTileOuter < ALIGN_ELEMS) maxTileOuter = ALIGN_ELEMS;

    uint32_t tileOuter = maxTileOuter;
    if (totalOutputLength < tileOuter) {
        tileOuter = AlignDown(totalOutputLength, ALIGN_ELEMS);
        if (tileOuter == 0) tileOuter = totalOutputLength;
    }

    // Adaptive blockDim
    uint32_t numTiles = CeilDiv(totalOutputLength, tileOuter);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalOutputLength(totalOutputLength);
    tiling.set_reductionLength(reductionLength);
    tiling.set_tileOuter(tileOuter);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t CHUNK_SIZE = 64;      // vector width for reduction tile

class KernelMaxReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMaxReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputLength, uint32_t reductionLength,
                                uint32_t tileOuter)
    {
        totalOutputLen_ = totalOutputLength;
        reductionLen_ = reductionLength;
        tileOuter_ = tileOuter;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalOutputLen_ * reductionLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputLen_);

        // Pipeline buffers:
        // inQueue: holds input chunk of size tileOuter * CHUNK_SIZE floats
        // outQueue: holds output tile of size tileOuter floats
        // tmpQueue: holds current max values (size tileOuter floats)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileOuter_ * CHUNK_SIZE * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOuter_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileOuter_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputLen_ == 0 || reductionLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over outer dimension
        const uint32_t strideOuter = blockNum * tileOuter_;
        uint32_t outerStart = blockIdx * tileOuter_;
        if (outerStart >= totalOutputLen_) return;

        uint32_t curOuter = totalOutputLen_ - outerStart;
        if (curOuter > tileOuter_) curOuter = tileOuter_;

        // Initialize current max values to -inf
        AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();
        AscendC::Duplicate<float>(maxLocal, -std::numeric_limits<float>::infinity(), curOuter);

        // Process reduction dimension in chunks
        uint32_t chunkStart = 0;
        while (chunkStart < reductionLen_) {
            uint32_t chunkLen = reductionLen_ - chunkStart;
            if (chunkLen > CHUNK_SIZE) chunkLen = CHUNK_SIZE;

            // Load input chunk: for each outer index, load a vector of length chunkLen
            // Input layout: [outer][reduction] contiguous along reduction dim (last dim after permute)
            // So each outer row has contiguous reduction data.
            // We load tileOuter rows of chunkLen elements each.
            // Use a pipeline to overlap copy and compute
            AscendC::LocalTensor<float> inChunk = inQueueX_.AllocTensor<float>();
            uint32_t srcOffset = outerStart * reductionLen_ + chunkStart;
            // Copy a 2D block: total elements = curOuter * chunkLen
            AscendC::DataCopy(inChunk, xGm_[srcOffset], curOuter * chunkLen);
            inQueueX_.EnQue(inChunk);

            AscendC::LocalTensor<float> chunkLocal = inQueueX_.DeQue<float>();
            // Compute element-wise max between current max and chunk
            // We need to reduce along the chunk dimension: for each outer index, max over the chunk length.
            // Use ReduceMax on each row? No, we can use AscendC::Max followed by horizontal max? Simpler: use a loop over chunk length.
            // But for vectorization, we can use AscendC::Max to compare current max with the chunk elements one by one.
            // Since chunkLen <= 64, we can store a temporary local variable for each row? Not needed.
            // We'll use a scalar reduction: for each outer index, process each element in chunk.
            for (uint32_t j = 0; j < chunkLen; ++j) {
                AscendC::Max(maxLocal, maxLocal, chunkLocal[j * curOuter], curOuter);
            }
            inQueueX_.FreeTensor(chunkLocal);
            chunkStart += chunkLen;
        }

        // Write final max values to global output
        AscendC::LocalTensor<float> outTile = outQueueY_.AllocTensor<float>();
        AscendC::DataCopy(outTile, maxLocal, curOuter);
        outQueueY_.EnQue(outTile);

        AscendC::LocalTensor<float> outTileFinal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[outerStart], outTileFinal, curOuter);
        outQueueY_.FreeTensor(outTileFinal);

        tmpQueue_.FreeTensor(maxLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutputLen_ {0};
    uint32_t reductionLen_ {0};
    uint32_t tileOuter_ {0};
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReductionOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalOutputLength, tiling_data.reductionLength, tiling_data.tileOuter);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'max_reduction_over_a_dimension_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, dim):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x):
        # Permute x so that the reduction dimension is last (contiguous)
        # Then call custom kernel which reduces over last dimension
        perm = list(range(x.dim()))
        perm.pop(self.dim)
        perm.append(self.dim)
        x_perm = x.permute(*perm)
        y = getattr(custom_ops_lib, _FN)(x_perm)
        # The output y has shape x_perm.shape[:-1], which is equivalent to original output shape
        return y
