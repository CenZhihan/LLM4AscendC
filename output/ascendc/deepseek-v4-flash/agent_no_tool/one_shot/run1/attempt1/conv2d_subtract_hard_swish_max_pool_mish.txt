#include "conv2d_subtract_hard_swish_max_pool_mish_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Conv2dSubtractHardSwishMaxPoolMishCustomTilingData tiling;

    auto *xShape = context->GetInputShape(0);
    auto *wShape = context->GetInputShape(1);
    auto *yShape = context->GetOutputShape(0);

    uint32_t batch      = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t inC        = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    uint32_t inH        = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
    uint32_t inW        = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));
    uint32_t outC       = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(0));
    uint32_t kH         = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(2));
    uint32_t kW         = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(3));
    uint32_t outH       = inH - kH + 1;
    uint32_t outW       = inW - kW + 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *subValPtr = attrs->GetAttrPointer<float>(0);
    const int64_t *poolKPtr = attrs->GetAttrPointer<int64_t>(1);
    uint32_t poolK = static_cast<uint32_t>(*poolKPtr);
    uint32_t poolH = outH / poolK;
    uint32_t poolW = outW / poolK;

    tiling.set_batch(batch);
    tiling.set_inChannels(inC);
    tiling.set_inHeight(inH);
    tiling.set_inWidth(inW);
    tiling.set_outChannels(outC);
    tiling.set_kernelHeight(kH);
    tiling.set_kernelWidth(kW);
    tiling.set_outHeight(outH);
    tiling.set_outWidth(outW);
    tiling.set_poolHeight(poolH);
    tiling.set_poolWidth(poolW);
    tiling.set_subtractValue(*subValPtr);

    uint32_t totalLength = batch * outC * poolH * poolW;
    tiling.set_totalLength(totalLength);

    // Simple tiling on flattened output
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float) / 10; // conservative
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileLen = (totalLength < maxTile) ? totalLength : maxTile;
    tiling.set_tileLength(tileLen);

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles > 24 ? 24 : (numTiles < 8 ? 8 : numTiles);
    tiling.set_blockDim(blockDim);
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
