#include "matmul_with_irregular_shapes_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: we need space for A tile (tileM * tileK), B tile (tileK * tileN), C tile (tileM * tileN) and some temporaries.
// We'll assume 192KB per core. Each float is 4 bytes.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 32;  // align to 32 for better access

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithIrregularShapesCustomTilingData tiling;
    uint32_t M = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(0));
    uint32_t K = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));
    uint32_t N = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetDim(1));

    // Estimate tile sizes based on UB budget
    // We need: tileM*tileK + tileK*tileN + tileM*tileN floats.
    // Solve for roughly equal tile sizes: let t = tileM = tileN, tileK = 256 (fixed typical).
    // Then 2*t*256 + t*t <= UB_BUDGET/4
    // t^2 + 512t - (192*1024/4)=0 => t^2+512t-49152=0 => t~148 (approx). But we can compute dynamically.
    constexpr uint32_t MAX_K_TILE = 256;  // typical K tile to reuse
    uint32_t tileK = (K < MAX_K_TILE) ? K : MAX_K_TILE;
    uint32_t maxTileMN = (UB_BUDGET_BYTES / 4) / (tileK + 512); // rough estimate
    if (maxTileMN > 128) maxTileMN = 128;  // cap for stability
    uint32_t tileM = maxTileMN < M ? AlignDown(maxTileMN, ALIGN) : AlignDown(M, ALIGN);
    if (tileM == 0) tileM = (M < ALIGN) ? M : ALIGN;
    uint32_t tileN = maxTileMN < N ? AlignDown(maxTileMN, ALIGN) : AlignDown(N, ALIGN);
    if (tileN == 0) tileN = (N < ALIGN) ? N : ALIGN;

    // Number of tiles in M and N dimensions
    uint32_t tilesM = (M + tileM - 1) / tileM;
    uint32_t tilesN = (N + tileN - 1) / tileN;
    uint32_t totalTiles = tilesM * tilesN;
    uint32_t blockDim = totalTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulWithIrregularShapes {
public:
    __aicore__ inline KernelMatmulWithIrregularShapes() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        M_ = M; K_ = K; N_ = N;
        tileM_ = tileM; tileK_ = tileK; tileN_ = tileN;

        // Global pointers: row-major
        AGm_.SetGlobalBuffer((__gm__ float*)A, M * K);
        BGm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        CGm_.SetGlobalBuffer((__gm__ float*)C, M * N);

        // Local buffers: double buffering for A and B, single for C accumulator
        pipe_.InitBuffer(inQueA_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueB_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueC_, 1, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Distribute tiles: each block handles one or more (tileM x tileN) output tiles
        uint32_t tilesM = (M_ + tileM_ - 1) / tileM_;
        uint32_t tilesN = (N_ + tileN_ - 1) / tileN_;
        uint32_t totalTiles = tilesM * tilesN;

        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            uint32_t i = t / tilesN;  // which tile row
            uint32_t j = t % tilesN;  // which tile col
            uint32_t mOffset = i * tileM_;
            uint32_t nOffset = j * tileN_;
            uint32_t curM = (mOffset + tileM_ <= M_) ? tileM_ : M_ - mOffset;
            uint32_t curN = (nOffset + tileN_ <= N_) ? tileN_ : N_ - nOffset;

            // Compute output tile at (mOffset, nOffset)
            ComputeTile(mOffset, nOffset, curM, curN);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t mStart, uint32_t nStart, uint32_t mSize, uint32_t nSize)
    {
        // Accumulator tile
        AscendC::LocalTensor<float> accTile = outQueC_.AllocTensor<float>();
        AscendC::SetZero(accTile, mSize * nSize);

        // Loop over K dimension in tiles of tileK_
        uint32_t tilesK = (K_ + tileK_ - 1) / tileK_;
        for (uint32_t kIdx = 0; kIdx < tilesK; kIdx++) {
            uint32_t kStart = kIdx * tileK_;
            uint32_t kSize = (kStart + tileK_ <= K_) ? tileK_ : K_ - kStart;

            // Load A tile: mSize x kSize from global A at (mStart, kStart)
            AscendC::LocalTensor<float> aTile = inQueA_.AllocTensor<float>();
            AscendC::DataCopy(aTile, AGm_[mStart * K_ + kStart], mSize * kSize);

            // Load B tile: kSize x nSize from global B at (kStart, nStart)
            // But global B is row-major of shape K x N, so element (kStart, nStart) is at index kStart*N + nStart
            AscendC::LocalTensor<float> bTile = inQueB_.AllocTensor<float>();
            AscendC::DataCopy(bTile, BGm_[kStart * N_ + nStart], kSize * nSize);

            // Compute partial product: acc += a * b
            // We need a matrix multiply of (mSize x kSize) * (kSize x nSize). Use AscendC intrinsic mmad?
            // For simplicity, we use a naive triply nested loop in a single core (though not optimal).
            // In practice, we would invoke the cube unit. Here we approximate with general matrix multiplication via loop.
            // But to keep within AscendC, we'll implement a simple tiled gemm using vector instructions.
            // Since we cannot use cube here without complex setup, we assume a helper function.
            // For demonstration, we call a placeholder function.
            MatmulPlaceholder(accTile, aTile, bTile, mSize, kSize, nSize);

            inQueA_.FreeTensor(aTile);
            inQueB_.FreeTensor(bTile);
        }

        // Store result to global C
        // C is stored row-major with stride N_
        for (uint32_t r = 0; r < mSize; r++) {
            AscendC::DataCopy(CGm_[(mStart + r) * N_ + nStart], accTile[r * nSize], nSize);
        }

        outQueC_.FreeTensor(accTile);
    }

    __aicore__ inline void MatmulPlaceholder(AscendC::LocalTensor<float>& acc,
                                              AscendC::LocalTensor<float>& a,
                                              AscendC::LocalTensor<float>& b,
                                              uint32_t m, uint32_t k, uint32_t n)
    {
        // This is a simple naive implementation using vector operations.
        // For each output element (i,j): sum over k of a[i][k] * b[k][j]
        // We'll use loops and vectorized multiply-add.
        for (uint32_t i = 0; i < m; i++) {
            for (uint32_t j = 0; j < n; j++) {
                float sum = 0.0f;
                for (uint32_t p = 0; p < k; p++) {
                    sum += a[i * k + p] * b[p * n + j];
                }
                acc[i * n + j] += sum;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueC_;

    AscendC::GlobalTensor<float> AGm_;
    AscendC::GlobalTensor<float> BGm_;
    AscendC::GlobalTensor<float> CGm_;

    uint32_t M_, K_, N_;
    uint32_t tileM_, tileK_, tileN_;
};

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithIrregularShapes op;
    op.Init(A, B, C, tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
