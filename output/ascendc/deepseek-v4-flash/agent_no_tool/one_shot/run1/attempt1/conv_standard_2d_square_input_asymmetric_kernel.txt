#include "conv_standard_2d_square_input_asymmetric_kernel_custom_tiling.h"

namespace optiling {

// UB budget: conservative per-core
// Each tile processes a portion of output: we need input tile of size (tileH * tileW * inChannels) and weight tile of (outChannels_tile * inChannels * kernelH * kernelW)
// For simplicity, we tile along output spatial dimensions.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MAX_TILE_OUT = 4096; // max output elements per tile

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    uint32_t batch = x_shape->GetDim(0);
    uint32_t inChannels = x_shape->GetDim(1);
    uint32_t inH = x_shape->GetDim(2);
    uint32_t inW = x_shape->GetDim(3);

    uint32_t outChannels = w_shape->GetDim(0);
    uint32_t kernelH = w_shape->GetDim(2);
    uint32_t kernelW = w_shape->GetDim(3);

    // Fixed parameters for this specific conv
    uint32_t strideH = 1, strideW = 1;
    uint32_t padH = 0, padW = 0;
    uint32_t dilationH = 1, dilationW = 1;

    uint32_t outH = (inH + 2*padH - dilationH*(kernelH-1) - 1) / strideH + 1;
    uint32_t outW = (inW + 2*padW - dilationW*(kernelW-1) - 1) / strideW + 1;
    uint32_t totalLength = batch * outChannels * outH * outW;

    // Tiling: process output elements per tile. Each tile handles a consecutive block of output elements (spatial).
    uint32_t tileLength = totalLength;
    if (totalLength > MAX_TILE_OUT) {
        tileLength = AlignDown(MAX_TILE_OUT, ALIGN_ELEMS);
        if (tileLength == 0) tileLength = ALIGN_ELEMS;
    }

    // Block dim: number of tiles covering output
    uint32_t numTiles = (totalLength + tileLength - 1) / tileLength;
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_batch(batch);
    tiling.set_inChannels(inChannels);
    tiling.set_outChannels(outChannels);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_kernelH(kernelH);
    tiling.set_kernelW(kernelW);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_dilationH(dilationH);
    tiling.set_dilationW(dilationW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batch, uint32_t inChannels, uint32_t outChannels,
                                uint32_t inH, uint32_t inW, uint32_t kernelH, uint32_t kernelW,
                                uint32_t outH, uint32_t outW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batch_ = batch;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        inH_ = inH;
        inW_ = inW;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        outH_ = outH;
        outW_ = outW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_ * inChannels_ * inH_ * inW_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outChannels_ * inChannels_ * kernelH_ * kernelW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Each tile processes a chunk of output elements (flattened spatial per batch and channel)
        // We need to allocate local memory for input patch, weight, and output partial sums
        // Max input patch size per output element: inChannels * kernelH * kernelW
        // For a tile of tileLen_ output elements, we need storage for that many patches.
        // To keep UB usage manageable, we process one output element at a time within a tile.
        // So we allocate for one pattern.
        uint32_t patchSize = inChannels_ * kernelH_ * kernelW_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, patchSize * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, patchSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, patchSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output elements
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        // Prefetch first tile's output? Actually we compute output per element, so we need to prefetch weight and input patches.
        // We'll loop over each output element in the tile.
        for (uint32_t i = 0; i < curLen; i++) {
            uint32_t outIdx = base + i;
            // Compute output position (b, oc, oh, ow)
            uint32_t b = outIdx / (outChannels_ * outH_ * outW_);
            uint32_t rem = outIdx % (outChannels_ * outH_ * outW_);
            uint32_t oc = rem / (outH_ * outW_);
            uint32_t rem2 = rem % (outH_ * outW_);
            uint32_t oh = rem2 / outW_;
            uint32_t ow = rem2 % outW_;

            // For each input channel, compute convolution sum
            float sumVal = 0.0f;
            for (uint32_t ic = 0; ic < inChannels_; ic++) {
                for (uint32_t kh = 0; kh < kernelH_; kh++) {
                    for (uint32_t kw = 0; kw < kernelW_; kw++) {
                        int32_t ih = static_cast<int32_t>(oh * strideH_ + kh * dilationH_ - padH_);
                        int32_t iw = static_cast<int32_t>(ow * strideW_ + kw * dilationW_ - padW_);
                        if (ih >= 0 && ih < static_cast<int32_t>(inH_) && iw >= 0 && iw < static_cast<int32_t>(inW_)) {
                            uint32_t xIdx = b * inChannels_ * inH_ * inW_ + ic * inH_ * inW_ + ih * inW_ + iw;
                            uint32_t wIdx = oc * inChannels_ * kernelH_ * kernelW_ + ic * kernelH_ * kernelW_ + kh * kernelW_ + kw;
                            sumVal += xGm_.GetValue(xIdx) * wGm_.GetValue(wIdx);
                        }
                    }
                }
            }
            yGm_.SetValue(outIdx, sumVal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t batch_ {0};
    uint32_t inChannels_ {0};
    uint32_t outChannels_ {0};
    uint32_t inH_ {0};
    uint32_t inW_ {0};
    uint32_t kernelH_ {0};
    uint32_t kernelW_ {0};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
    uint32_t strideH_ {1};
    uint32_t strideW_ {1};
    uint32_t padH_ {0};
    uint32_t padW_ {0};
    uint32_t dilationH_ {1};
    uint32_t dilationW_ {1};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.batch, tiling_data.inChannels, tiling_data.outChannels,
            tiling_data.inH, tiling_data.inW, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW);
    op.Process();
}
