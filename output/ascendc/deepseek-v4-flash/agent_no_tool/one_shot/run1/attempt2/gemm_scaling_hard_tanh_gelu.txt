static constexpr uint32_t CUBE_BLOCK_M = 16;

static uint32_t AlignUp(uint32_t x, uint32_t a) {
    return (x + a - 1) / a * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{

    // Ensure block sizes are multiples of cube block sizes for efficient matmul
    uint32_t blockSizeM = AlignUp(std::min(CUBE_BLOCK_M, M), CUBE_BLOCK_M);

    uint32_t blocksN = (N + blockSizeN - 1) / blockSizeN;
    uint32_t totalBlocks = blocksM * blocksN;

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmScalingHardTanhGeluCustom {
public:
    __aicore__ inline KernelGemmScalingHardTanhGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t blockSizeM, uint32_t blockSizeN, uint32_t blockSizeK,
                                float scaling, float hardtanh_min, float hardtanh_max)
    {
        M_ = M; N_ = N; K_ = K;
        blockM_ = blockSizeM; blockN_ = blockSizeN; blockK_ = blockSizeK;
        scaling_ = scaling;
        htanMin_ = hardtanh_min;
        htanMax_ = hardtanh_max;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M * K);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N * K);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M * N);

        // Configure matmul: A = x (MxK), B = w^T (KxN), bias = b (N). Output = y (MxN)
        matmul_.SetTensorA(xGm_, M, K, false, false);
        matmul_.SetTensorB(wGm_, N, K, true, false); // transpose w (shape [N,K] -> [K,N])
        matmul_.SetBias(bGm_);
        matmul_.SetTensorC(yGm_, M, N);
        matmul_.SetLoopConfig(blockM_, blockN_, blockK_);

        // Vector processing pipe (for scaling, hardtanh, gelu on output tiles)
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, blockM_ * blockN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        uint32_t blocksM = (M_ + blockM_ - 1) / blockM_;
        uint32_t blocksN = (N_ + blockN_ - 1) / blockN_;
        uint32_t totalBlocks = blocksM * blocksN;

        // Grid-stride loop over output blocks
        uint32_t myBlockIdx = blockIdx;
        while (myBlockIdx < totalBlocks) {
            uint32_t mBlock = myBlockIdx / blocksN;
            uint32_t nBlock = myBlockIdx % blocksN;
            uint32_t mStart = mBlock * blockM_;
            uint32_t nStart = nBlock * blockN_;
            uint32_t mEnd = (mStart + blockM_ > M_) ? M_ : mStart + blockM_;
            uint32_t nEnd = (nStart + blockN_ > N_) ? N_ : nStart + blockN_;
            uint32_t curM = mEnd - mStart;
            uint32_t curN = nEnd - nStart;
            uint32_t blockSize = curM * curN;

            // Perform matmul for this block
            matmul_.SetTensorA(xGm_.Slice(mStart, 0, curM, K), curM, K, false, false);
            matmul_.SetTensorB(wGm_.Slice(nStart, 0, curN, K), curN, K, true, false);
            matmul_.SetBias(bGm_.Slice(nStart, curN));
            matmul_.SetTensorC(yGm_.Slice(mStart, nStart, curM, curN), curM, curN);
            matmul_.IterateAll();

            // Load computed block from global to local
            AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, yGm_[mStart * N_ + nStart], blockSize);

            // scaling
            AscendC::Muls(yLocal, yLocal, scaling_, blockSize);
            // hardtanh
            AscendC::Clamp(yLocal, yLocal, htanMin_, htanMax_, blockSize);
            // gelu approximation (exact using erf)
            AscendC::Gelu(yLocal, yLocal, blockSize);

            // Store back to global
            AscendC::DataCopy(yGm_[mStart * N_ + nStart], yLocal, blockSize);
            inQueueY_.FreeTensor(yLocal);

            myBlockIdx += blockNum;
        }
    }

private:
    AscendC::Matmul<float, float, float> matmul_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    uint32_t M_, N_, K_;
    uint32_t blockM_, blockN_, blockK_;
    float scaling_, htanMin_, htanMax_;
};

extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScalingHardTanhGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.blockSizeM, tiling_data.blockSizeN, tiling_data.blockSizeK,
            tiling_data.scaling, tiling_data.hardtanh_min, tiling_data.hardtanh_max);
    op.Process();
}
