static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Lenet5CustomTilingData tiling;

    // Input shapes (batch, C, H, W)
    const gert::Shape *x_shape = context->GetInputShape(0);
    auto x_dims = x_shape->GetOriginShape().GetDims();
    uint32_t batch = std::max(1u, static_cast<uint32_t>(x_dims[0]));

    // Conv1 weight shape
    const gert::Shape *c1w_shape = context->GetInputShape(1);
    auto c1w_dims = c1w_shape->GetOriginShape().GetDims();
    uint32_t conv1OC = c1w_dims[0];
    uint32_t conv1IC = c1w_dims[1];
    uint32_t conv1KH = c1w_dims[2];
    uint32_t conv1KW = c1w_dims[3];

    // Conv2 weight shape
    const gert::Shape *c2w_shape = context->GetInputShape(3);
    auto c2w_dims = c2w_shape->GetOriginShape().GetDims();
    uint32_t conv2OC = c2w_dims[0];
    uint32_t conv2IC = c2w_dims[1];
    uint32_t conv2KH = c2w_dims[2];
    uint32_t conv2KW = c2w_dims[3];

    // FC weight shapes
    const gert::Shape *fc1w_shape = context->GetInputShape(5);
    auto fc1w_dims = fc1w_shape->GetOriginShape().GetDims();
    uint32_t fc1Out = fc1w_dims[0];
    uint32_t fc1In  = fc1w_dims[1];

    const gert::Shape *fc2w_shape = context->GetInputShape(7);
    auto fc2w_dims = fc2w_shape->GetOriginShape().GetDims();
    uint32_t fc2Out = fc2w_dims[0];
    uint32_t fc2In  = fc2w_dims[1];

    const gert::Shape *fc3w_shape = context->GetInputShape(9);
    auto fc3w_dims = fc3w_shape->GetOriginShape().GetDims();
    uint32_t fc3Out = fc3w_dims[0];
    uint32_t fc3In  = fc3w_dims[1];

    // Attribute num_classes (optional, default to fc3Out)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t numClasses = attrs->GetAttr<int64_t>(0);

    // Set blockDim = min(batch, 256) but no more than physical cores
    uint32_t blockDim = std::min(batch, (uint32_t)256);
    if (blockDim < 8) blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_numClasses(static_cast<uint32_t>(numClasses));
    tiling.set_conv1OC(conv1OC);
    tiling.set_conv1IC(conv1IC);
    tiling.set_conv1KH(conv1KH);
    tiling.set_conv1KW(conv1KW);
    tiling.set_conv2OC(conv2OC);
    tiling.set_conv2IC(conv2IC);
    tiling.set_conv2KH(conv2KH);
    tiling.set_conv2KW(conv2KW);
    tiling.set_fc1In(fc1In);
    tiling.set_fc1Out(fc1Out);
    tiling.set_fc2In(fc2In);
    tiling.set_fc2Out(fc2Out);
    tiling.set_fc3In(fc3In);
    tiling.set_fc3Out(fc3Out);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    auto x_dims = x_shape->GetOriginShape().GetDims();
    // output shape [batch, numClasses]
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, x_dims[0]);
    // numClasses from attribute or from fc3 weight
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t numClasses = attrs->GetAttr<int64_t>(0);
    y_shape->SetDim(1, static_cast<int64_t>(numClasses));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
