kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t n, uint32_t cOut, uint32_t dOut, uint32_t hOut, uint32_t wOut,
                                uint32_t cIn, uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t dIn, uint32_t hIn, uint32_t wIn)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        n_ = n; cOut_ = cOut; dOut_ = dOut; hOut_ = hOut; wOut_ = wOut;
        cIn_ = cIn; kD_ = kD; kH_ = kH; kW_ = kW;
        dIn_ = dIn; hIn_ = hIn; wIn_ = wIn;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n * cIn * dIn * hIn * wIn);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, cIn * cOut * kD * kH * kW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            ComputeTile(base, curLen);

            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalLen_) break;
            base = nextBase;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t outOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        AscendC::Mul(yLocal, yLocal, 0.0f, len);

        for (uint32_t i = 0; i < len; i++) {
            uint32_t flatIdx = outOffset + i;
            uint32_t wIdx = flatIdx % wOut_;
            uint32_t hIdx = (flatIdx / wOut_) % hOut_;
            uint32_t dIdx = (flatIdx / (wOut_ * hOut_)) % dOut_;
            uint32_t coIdx = (flatIdx / (wOut_ * hOut_ * dOut_)) % cOut_;
            uint32_t nIdx = flatIdx / (wOut_ * hOut_ * dOut_ * cOut_);

            float sum = 0.0f;
            for (uint32_t kd = 0; kd < kD_; kd++) {
                int32_t dInIdx = static_cast<int32_t>(dIdx) - static_cast<int32_t>(kd);
                if (dInIdx < 0 || dInIdx >= static_cast<int32_t>(dIn_)) continue;
                for (uint32_t kh = 0; kh < kH_; kh++) {
                    int32_t hInIdx = static_cast<int32_t>(hIdx) - static_cast<int32_t>(kh);
                    if (hInIdx < 0 || hInIdx >= static_cast<int32_t>(hIn_)) continue;
                    for (uint32_t kw = 0; kw < kW_; kw++) {
                        int32_t wInIdx = static_cast<int32_t>(wIdx) - static_cast<int32_t>(kw);
                        if (wInIdx < 0 || wInIdx >= static_cast<int32_t>(wIn_)) continue;
                        for (uint32_t ci = 0; ci < cIn_; ci++) {
                            uint32_t xOff = nIdx * cIn_ * dIn_ * hIn_ * wIn_
                                          + ci * dIn_ * hIn_ * wIn_
                                          + dInIdx * hIn_ * wIn_
                                          + hInIdx * wIn_
                                          + wInIdx;
                            uint32_t wOff = ci * cOut_ * kD_ * kH_ * kW_
                                          + coIdx * kD_ * kH_ * kW_
                                          + kd * kH_ * kW_
                                          + kh * kW_
                                          + kw;
                            sum += xGm_.GetValue(xOff) * wGm_.GetValue(wOff);
                        }
                    }
                }
            }
            yLocal.SetValue(i, sum);
        }

        AscendC::DataCopy(yGm_[outOffset], yLocal, len);

        outQueueY_.FreeTensor(yLocal);
        tmpQueue_.FreeTensor(tLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t n_, cOut_, dOut_, hOut_, wOut_;
    uint32_t cIn_, kD_, kH_, kW_;
    uint32_t dIn_, hIn_, wIn_;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.cOut, tiling_data.dOut, tiling_data.hOut, tiling_data.wOut,
            tiling_data.cIn, tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.dIn, tiling_data.hIn, tiling_data.wIn);
    op.Process();
}
