host_operator_src="""

#include "cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumCustomTilingData tiling;
    const gert::Shape x_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalLength = static_cast<uint32_t>(x_shape.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t ndim = x_shape.GetDimNum();
    if (dim < 0) dim += ndim;
    if (dim < 0 || dim >= ndim) dim = 0;
    uint32_t rowLength = 1;
    uint32_t totalRows = 1;
    for (int32_t i = 0; i < ndim; i++) {
        if (i == dim) continue;
        totalRows *= static_cast<uint32_t>(x_shape.GetDim(i));
    }
    if (totalRows == 0) totalRows = 1;
    rowLength = totalLength / totalRows;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = (rowLength < maxTile) ? rowLength : maxTile;

    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_totalRows(totalRows);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCumsumCustom {
public:
    __aicore__ inline KernelCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLength,
                                uint32_t totalLength)
    {
        totalRows_ = totalRows;
        rowLength_ = rowLength;
        totalLength_ = totalLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, rowLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, rowLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowLength_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t rowIdx = blockIdx;
        while (rowIdx < totalRows_) {
            uint32_t rowOffset = rowIdx * rowLength_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowOffset], rowLength_);

            AscendC::LocalTensor<float> sumLocal = outQueueY_.AllocTensor<float>();
            sumLocal.SetValue(0, xLocal(0));

            for (uint32_t i = 1; i < rowLength_; i++) {
                float prev = sumLocal.GetValue(i - 1);
                float curr = xLocal.GetValue(i);
                sumLocal.SetValue(i, prev + curr);
            }

            AscendC::DataCopy(yGm_[rowOffset], sumLocal, rowLength_);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(sumLocal);

            rowIdx += blockNum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_ {0};
    uint32_t rowLength_ {0};
    uint32_t totalLength_ {0};
};

extern "C" __global__ __aicore__ void cumsum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.rowLength, tiling_data.totalLength);
    op.Process();
}
