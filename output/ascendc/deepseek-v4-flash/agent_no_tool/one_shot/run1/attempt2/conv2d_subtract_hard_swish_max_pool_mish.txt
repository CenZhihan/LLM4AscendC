static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

    uint32_t inC        = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dSubtractHardSwishMaxPoolMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractHardSwishMaxPoolMishCustom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
        GM_ADDR y,
        uint32_t batch, uint32_t inC, uint32_t inH, uint32_t inW,
        uint32_t outC, uint32_t kH, uint32_t kW,
        uint32_t outH, uint32_t outW,
        uint32_t poolH, uint32_t poolW,
        float subtractValue,
        uint32_t totalLength, uint32_t tileLength,
        uint8_t hasBias) {

        batch_ = batch; inC_ = inC; inH_ = inH; inW_ = inW;
        outC_ = outC; kH_ = kH; kW_ = kW;
        outH_ = outH; outW_ = outW;
        poolH_ = poolH; poolW_ = poolW;
        subtractValue_ = subtractValue;
        hasBias_ = hasBias;

        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_ * inC_ * inH_ * inW_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outC_ * inC_ * kH_ * kW_);
        if (hasBias_) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, outC_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t start = blockIdx * tileLen_;
        uint32_t end = start + tileLen_;
        if (start >= totalLen_) return;
        if (end > totalLen_) end = totalLen_;

        for (uint32_t idx = start; idx < end; ++idx) {
            uint32_t linear = idx;
            uint32_t pw = linear % poolW_; linear /= poolW_;
            uint32_t ph = linear % poolH_; linear /= poolH_;
            uint32_t oc = linear % outC_; linear /= outC_;
            uint32_t b = linear;

            // --- Convolution ---
            float convVal = 0.0f;
            // conv output pixel at (oc, ph, pw) corresponds to input region:
            // height: ph*poolK ... ph*poolK + kH - 1? Wait: conv output is outH x outW.
            // maxpool slides over conv output with kernel = poolK, stride = poolK.
            // So the pooled pixel at (ph, pw) covers conv output rows [ph*poolK, (ph+1)*poolK-1]
            // and columns similarly.  But the convolution is applied first to the input.
            // So we need to compute the convolution for the entire pool-sized block of output pixels
            // and then take the max.  Here we are computing the final output directly,
            // so we need to find the max over the conv output within that block.
            // We'll compute conv for each conv output pixel in the block and take max.
            float maxVal = -1e38f;
            for (uint32_t pkh = 0; pkh < poolK_; ++pkh) {
                int32_t convH = static_cast<int32_t>(ph * poolK_ + pkh);
                if (convH >= static_cast<int32_t>(outH_)) continue;
                for (uint32_t pkw = 0; pkw < poolK_; ++pkw) {
                    int32_t convW = static_cast<int32_t>(pw * poolK_ + pkw);
                    if (convW >= static_cast<int32_t>(outW_)) continue;
                    // conv output at (oc, convH, convW)
                    float sum = 0.0f;
                    for (uint32_t ic = 0; ic < inC_; ++ic) {
                        for (uint32_t kh = 0; kh < kH_; ++kh) {
                            int32_t ih = static_cast<int32_t>(convH + kh);
                            if (ih >= static_cast<int32_t>(inH_)) continue;
                            for (uint32_t kw = 0; kw < kW_; ++kw) {
                                int32_t iw = static_cast<int32_t>(convW + kw);
                                if (iw >= static_cast<int32_t>(inW_)) continue;
                                uint32_t xIdx = b * inC_ * inH_ * inW_ + ic * inH_ * inW_ + ih * inW_ + iw;
                                uint32_t wIdx = oc * inC_ * kH_ * kW_ + ic * kH_ * kW_ + kh * kW_ + kw;
                                sum += xGm_.GetValue(xIdx) * wGm_.GetValue(wIdx);
                            }
                        }
                    }
                    if (hasBias_) {
                        sum += biasGm_.GetValue(oc);
                    }
                    // subtract and hardswish
                    float sub = sum - subtractValue_;
                    float relu6 = sub + 3.0f;
                    if (relu6 < 0.0f) relu6 = 0.0f;
                    if (relu6 > 6.0f) relu6 = 6.0f;
                    float hs = sub * relu6 / 6.0f;
                    if (hs > maxVal) maxVal = hs;
                }
            }
            // mish
            float mishVal;
            if (maxVal > 0.0f) {
                mishVal = maxVal * tanhf_local(maxVal);
            } else {
                // softplus = log(1+exp(x))
                float sp = log1pf(expf(maxVal));
                mishVal = maxVal * tanhf_local(sp);
            }
            yGm_.SetValue(idx, mishVal);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t batch_ {0};
    uint32_t inC_ {0};
    uint32_t inH_ {0};
    uint32_t inW_ {0};
    uint32_t outC_ {0};
    uint32_t kH_ {0};
    uint32_t kW_ {0};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
    uint32_t poolH_ {0};
    uint32_t poolW_ {0};
    uint32_t poolK_ {2};   // from attribute, we'll set in Process using tiling? Set default.
    float subtractValue_ {0.0f};
    uint8_t hasBias_ {0};
};

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractHardSwishMaxPoolMishCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.batch, tiling_data.inChannels, tiling_data.inHeight, tiling_data.inWidth,
            tiling_data.outChannels, tiling_data.kernelHeight, tiling_data.kernelWidth,
            tiling_data.outHeight, tiling_data.outWidth,
            tiling_data.poolHeight, tiling_data.poolWidth,
            tiling_data.subtractValue,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.hasBias);
    // We need poolK from attribute; we can pass it via tiling or compute from outH/poolH.
    // For this simple kernel we assume poolK = outH / poolH, but we store it.
    // We'll set poolK_ based on tiling_data.outHeight / tiling_data.poolHeight.
    op.poolK_ = tiling_data.outHeight / tiling_data.poolHeight;  // assuming stride equals kernel
    op.Process();
}
"""
