#include "matmul_for_symmetric_matrices_custom_tiling.h"
#include <cmath>
#include "register/op_def_registry.h"

namespace optiling {

// UB model: for a tile of size tileSize x tileSize, we need:
//   inQueueA: 2 * tileSize * tileSize floats
//   inQueueB: 2 * tileSize * tileSize floats
//   outQueueC: 2 * tileSize * tileSize floats
//   tmpQueue: 1 * tileSize * tileSize floats (for accumulation)
// Total floats = 7 * tileSize * tileSize => bytes = 28 * tileSize * tileSize
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core share (conservative)
static constexpr uint32_t MIN_TILE = 64;
static constexpr uint32_t MAX_TILE = 512;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForSymmetricMatricesCustomTilingData tiling;
    // Input shapes: A and B are both (N, N)
    const uint32_t N = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(0));
    if (N == 0) {
        return ge::GRAPH_FAILED;
    }

    // Calculate maximum tile size from UB budget
    uint32_t maxTile = static_cast<uint32_t>(std::sqrt(UB_BUDGET_BYTES / 28.0f));
    if (maxTile < MIN_TILE) maxTile = MIN_TILE;
    if (maxTile > MAX_TILE) maxTile = MAX_TILE;
    if (maxTile > N) maxTile = N;
    // Align down to multiple of MIN_TILE for simplicity
    uint32_t tileSize = AlignDown(maxTile, MIN_TILE);
    if (tileSize == 0) tileSize = MIN_TILE;

    // Adaptive blockDim: number of tiles along one dimension
    uint32_t numTiles = CeilDiv(N, tileSize);
    uint32_t blockDim = numTiles * numTiles; // one block per output tile
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u; // typical limit for AICore

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_tileSize(tileSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *a_shape = context->GetInputShape(0);
    gert::Shape *c_shape = context->GetOutputShape(0);
    *c_shape = *a_shape; // output same shape as input

}

    return GRAPH_SUCCESS;
}

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulForSymmetricMatrices {
public:
    __aicore__ inline KernelMatmulForSymmetricMatrices() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t N, uint32_t tileSize)
    {
        N_ = N;
        tileSize_ = tileSize;
        aGm_.SetGlobalBuffer((__gm__ float*)a, N_ * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, N_ * N_);

        tileBytes_ = tileSize_ * tileSize_ * sizeof(float);
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileBytes_);
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileBytes_);
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileBytes_);
        pipe_.InitBuffer(tmpQueue_, 1, tileBytes_);
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || tileSize_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t tilesPerDim = (N_ + tileSize_ - 1) / tileSize_;
        // Each block handles one output tile.
        // Map blockIdx to tile row and column indices.
        // We use a cyclic assignment: block b handles tile (b / tilesPerDim, b % tilesPerDim) if b < tilesPerDim^2.
        uint32_t tileRow = blockIdx / tilesPerDim;
        uint32_t tileCol = blockIdx % tilesPerDim;
        if (tileRow >= tilesPerDim || tileCol >= tilesPerDim) return; // no work

        // Compute leading dimension (row-major)
        uint32_t rowStart = tileRow * tileSize_;
        uint32_t colStart = tileCol * tileSize_;
        uint32_t actualTileRows = (rowStart + tileSize_ <= N_) ? tileSize_ : N_ - rowStart;
        uint32_t actualTileCols = (colStart + tileSize_ <= N_) ? tileSize_ : N_ - colStart;

        // Accumulate over the reduction dimension K (tiled)
        // For each k tile, load A's tile (rowStart:rowStart+actualTileRows, k_start:k_start+tileK)
        // and B's tile (k_start:k_start+tileK, colStart:colStart+actualTileCols)
        // Multiply and accumulate.

        // initialize output tile to zero in temporary buffer
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> outTile = outQueueC_.AllocTensor<float>();
        // set outTile to zero (using vec_dup)
        AscendC::Duplicate(outTile, 0.0f, actualTileRows * actualTileCols);
        AscendC::LocalTensor<float> aTile;
        AscendC::LocalTensor<float> bTile;

        for (uint32_t k = 0; k < N_; k += tileSize_) {
            uint32_t kStart = k;
            uint32_t kLen = (kStart + tileSize_ <= N_) ? tileSize_ : N_ - kStart;

            // Load A tile: shape (actualTileRows, kLen)
            // Global offset in A: rowStart * N_ + kStart
            aTile = inQueueA_.AllocTensor<float>();
            AscendC::DataCopy(aTile, aGm_[rowStart * N_ + kStart], actualTileRows * kLen);
            inQueueA_.EnQue(aTile);

            // Load B tile: shape (kLen, actualTileCols)
            bTile = inQueueB_.AllocTensor<float>();
            AscendC::DataCopy(bTile, bGm_[kStart * N_ + colStart], kLen * actualTileCols);
            inQueueB_.EnQue(bTile);

            bTile = inQueueB_.DeQue<float>();
