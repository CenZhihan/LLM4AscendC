#include "conv_depthwise_2d_asymmetric_input_square_kernel_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;  // align to 128 bytes
static constexpr uint32_t TILE_HARD_CAP = 4096;

static uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ConvDepthwise2dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *dilation = attrs->GetAttrPointer<int64_t>(2);
    const int64_t groups = attrs->GetAttrPointer<int64_t>(3)[0];

    uint32_t N = x_shape->GetDim(0);
    uint32_t C = x_shape->GetDim(1);
    uint32_t H = x_shape->GetDim(2);
    uint32_t W = x_shape->GetDim(3);
    uint32_t KH = w_shape->GetDim(2);
    uint32_t KW = w_shape->GetDim(3);
    uint32_t strideH = stride[0];
    uint32_t strideW = stride[1];
    uint32_t padH = padding[0];
    uint32_t padW = padding[1];
    uint32_t dilationH = dilation[0];
    uint32_t dilationW = dilation[1];

    uint32_t outputH = (H + 2 * padH - dilationH * (KH - 1) - 1) / strideH + 1;
    uint32_t outputW = (W + 2 * padW - dilationW * (KW - 1) - 1) / strideW + 1;

    // Simple tiling: each tile = one output pixel (in one channel)
    // Total output elements per channel = outputH * outputW
    // UB budget: we need to load a patch of input: KH*KW floats + weight (KH*KW) + output float
    // plus some overhead. 3 * KH*KW * 4 bytes = 3*9*4=108 bytes, very small.
    uint32_t tileSize = 1;  // process one output pixel per iteration
    if (tileSize > TILE_HARD_CAP) tileSize = TILE_HARD_CAP;

    // blockDim: number of tiles per channel? Use many blocks to parallelize over output pixels.
    uint32_t totalPixels = N * C * outputH * outputW;
    uint32_t blockDim = 16; // fixed for now
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 40) blockDim = 40;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_KH(KH);
    tiling.set_KW(KW);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_dilationH(dilationH);
    tiling.set_dilationW(dilationW);
    tiling.set_groups(groups);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_tileSize(tileSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
