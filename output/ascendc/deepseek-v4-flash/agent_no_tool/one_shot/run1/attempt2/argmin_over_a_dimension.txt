static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ArgminOverADimensionCustomTilingData tiling;
    const auto* storageShape = context->GetInputShape(0);
    uint32_t ndim = storageShape->GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dimAttr = *attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimAttr < 0) ? dimAttr + ndim : dimAttr;
    if (dim < 0 || dim >= (int32_t)ndim) {
        dim = ndim - 1;
    }

    // Compute total input length
    uint32_t totalInputLength = 1;
    for (uint32_t i = 0; i < ndim; i++) totalInputLength *= storageShape->GetDim(i);

    // Compute output length = product of all dims except the reduction one
    uint32_t outputLength = 1;
    for (uint32_t i = 0; i < ndim; i++) if (i != (uint32_t)dim) outputLength *= storageShape->GetDim(i);

    // Compute reduction size, inner stride, outer size, last dim size
    uint32_t reduceSize = storageShape->GetDim(dim);
    uint32_t innerStride = 1;
    for (uint32_t i = dim + 1; i < ndim; i++) innerStride *= storageShape->GetDim(i);

    uint32_t outerSize = 1;
    for (uint32_t i = 0; i < (uint32_t)dim; i++) outerSize *= storageShape->GetDim(i);

    uint32_t lastDimSize = (ndim > 0) ? storageShape->GetDim(ndim - 1) : 1;

    // Block dimension based on number of output elements
    uint32_t blockDim = std::min(24u, CeilDiv(outputLength, 128u));
    if (blockDim < 1) blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalInputLength(totalInputLength);
    tiling.set_outputLength(outputLength);
    tiling.set_dimToReduce((uint32_t)dim);
    tiling.set_reduceSize(reduceSize);
    tiling.set_innerStride(innerStride);
    tiling.set_outerSize(outerSize);
    tiling.set_lastDimSize(lastDimSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dimAttr = attrs->GetAttrPointer<int32_t>(0) ? *attrs->GetAttrPointer<int32_t>(0) : -1;
