#include "conv_standard_3d_square_input_asymmetric_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget conservative for convolution: each tile needs input patch + weight + output
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard3dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *input_shape = context->GetInputShape(0);
    const gert::Shape *weight_shape = context->GetInputShape(1);

    uint32_t N  = input_shape->GetDim(0);
    uint32_t C  = input_shape->GetDim(1);
    uint32_t W  = input_shape->GetDim(2);
    uint32_t H  = input_shape->GetDim(3);
    uint32_t D  = input_shape->GetDim(4);

    uint32_t Cout = weight_shape->GetDim(0);
    uint32_t Cin  = weight_shape->GetDim(1);
    uint32_t Kw   = weight_shape->GetDim(2);
    uint32_t Kh   = weight_shape->GetDim(3);
    uint32_t Kd   = weight_shape->GetDim(4);

    // Output dimensions (padding=0, stride=1, dilation=1)
    uint32_t Wout = W - Kw + 1;
    uint32_t Hout = H - Kh + 1;
    uint32_t Dout = D - Kd + 1;

    uint32_t totalLength = N * Cout * Wout * Hout * Dout;

    // Estimate per-tile memory: for each output element we need the input patch (Cin*Kw*Kh*Kd) and weights (Cin*Kw*Kh*Kd) and output (1)
    // Roughly 2 * Cin * Kw * Kh * Kd per output element. Choose a small tile count.
    uint32_t patchSizeBytes = Cin * Kw * Kh * Kd * sizeof(float);
    uint32_t outputTileCost = patchSizeBytes * 2; // input patch + weights
    uint32_t maxTile = (UB_BUDGET_BYTES / outputTileCost);
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        tileLen = AlignDown(totalLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = totalLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_W(W);
    tiling.set_H(H);
    tiling.set_D(D);
    tiling.set_Cout(Cout);
    tiling.set_Cin(Cin);
    tiling.set_Kw(Kw);
    tiling.set_Kh(Kh);
    tiling.set_Kd(Kd);
    tiling.set_Wout(Wout);
    tiling.set_Hout(Hout);
    tiling.set_Dout(Dout);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv3d {
public:
    __aicore__ inline KernelConv3d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                const ConvStandard3dSquareInputAsymmetricKernelCustomTilingData& tiling)
    {
        // Store all tiling parameters
        totalLen_ = tiling.totalLength;
        tileLen_ = tiling.tileLength;
        N_ = tiling.N;
        C_ = tiling.C;
        W_ = tiling.W;
        H_ = tiling.H;
        D_ = tiling.D;
        Cout_ = tiling.Cout;
        Cin_ = tiling.Cin;
        Kw_ = tiling.Kw;
        Kh_ = tiling.Kh;
        Kd_ = tiling.Kd;
        Wout_ = tiling.Wout;
        Hout_ = tiling.Hout;
        Dout_ = tiling.Dout;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Allocate local buffers for one output tile: we need input patch and weight
        // Patch size: Cin * Kw * Kh * Kd
        patchElems_ = Cin_ * Kw_ * Kh_ * Kd_;
        // UB budget: allocate for patch, weight, and one output element accumulator
        const uint32_t ubElems = patchElems_ * 2 + 1;  // patch + weight + one output
        const uint32_t bufSize = ubElems * sizeof(float);
        pipe_.InitBuffer(inQueuePatch_, BUFFER_NUM, bufSize);
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, bufSize);
        pipe_.InitBuffer(tmpQueue_, 1, bufSize);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output elements (linearized)
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Prefetch data for current tile
            Prefetch(base, curLen);

            // Compute
            Compute(curLen);

            // Write back
            WriteBack(base, curLen);

            base += strideElems;
            if (base >= totalLen_) break;
        }
    }

private:
    __aicore__ inline void Prefetch(uint32_t base, uint32_t len)
    {
        // For each output element in this tile, we need to load the corresponding input patch and weight.
        // This is a simplified version: just load whole patch and weight for each output point one by one.
        // In practice, we would load larger chunks.
        // We'll iterate over the tile and load patch and weight into queues.
        for (uint32_t idx = 0; idx < len; ++idx) {
            uint32_t linear = base + idx;
            // Compute output coordinates
            uint32_t n   = linear / (Cout_ * Wout_ * Hout_ * Dout_);
            uint32_t r   = linear % (Cout_ * Wout_ * Hout_ * Dout_);
            uint32_t co  = r / (Wout_ * Hout_ * Dout_);
            uint32_t s   = r % (Wout_ * Hout_ * Dout_);
            uint32_t w   = s / (Hout_ * Dout_);
            uint32_t t   = s % (Hout_ * Dout_);
            uint32_t h   = t / Dout_;
            uint32_t d   = t % Dout_;

            // Load input patch
            AscendC::LocalTensor<float> patchLocal = inQueuePatch_.AllocTensor<float>();
            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kw = 0; kw < Kw_; ++kw) {
                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        for (uint32_t kd = 0; kd < Kd_; ++kd) {
                            uint32_t inIdx = n * (C_ * W_ * H_ * D_) +
                                             ci * (W_ * H_ * D_) +
                                             (w + kw) * (H_ * D_) +
                                             (h + kh) * D_ +
                                             (d + kd);
                            uint32_t localOff = ci * (Kw_ * Kh_ * Kd_) +
                                                kw * (Kh_ * Kd_) +
                                                kh * Kd_ +
                                                kd;
                            patchLocal(localOff) = xGm_[inIdx];
                        }
                    }
                }
            }
            // Load weight for this output channel
            AscendC::LocalTensor<float> weightLocal = inQueueW_.AllocTensor<float>();
            uint32_t wBase = co * (Cin_ * Kw_ * Kh_ * Kd_);
            for (uint32_t i = 0; i < patchElems_; ++i) {
                weightLocal(i) = wGm_[wBase + i];
            }

            inQueuePatch_.EnQue(patchLocal);
            inQueueW_.EnQue(weightLocal);
        }
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        for (uint32_t idx = 0; idx < len; ++idx) {
            AscendC::LocalTensor<float> patch = inQueuePatch_.DeQue<float>();
            AscendC::LocalTensor<float> weight = inQueueW_.DeQue<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

            // Compute dot product: accumulate over all patch elements
            AscendC::Mul(tmp, patch, weight, patchElems_);
            float sum = 0.0f;
            for (uint32_t i = 0; i < patchElems_; ++i) {
                sum += tmp(i);
            }
            // Store result in queue
            AscendC::LocalTensor<float> out = outQueueY_.AllocTensor<float>();
            out(0) = sum;

            outQueueY_.EnQue(out);
            inQueuePatch_.FreeTensor(patch);
            inQueueW_.FreeTensor(weight);
            tmpQueue_.FreeTensor(tmp);
        }
    }

    __aicore__ inline void WriteBack(uint32_t base, uint32_t len)
    {
        for (uint32_t idx = 0; idx < len; ++idx) {
            AscendC::LocalTensor<float> out = outQueueY_.DeQue<float>();
            yGm_[base + idx] = out(0);
            outQueueY_.FreeTensor(out);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePatch_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_, C_, W_, H_, D_;
    uint32_t Cout_, Cin_, Kw_, Kh_, Kd_;
    uint32_t Wout_, Hout_, Dout_;
    uint32_t patchElems_;
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3d op;
    op.Init(x, weight, y, tiling_data);
    op.Process();
}
