static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed2dAsymmetricInputAsymmetricKernelCustomTilingData tiling;

    // Fix: GetInputShape returns StorageShape; use GetOriginShape to get Shape.
    const gert::StorageShape *x_storage = context->GetInputShape(0);
    const gert::StorageShape *w_storage = context->GetInputShape(1);
    const gert::Shape &x_shape = x_storage->GetOriginShape();
    const gert::Shape &w_shape = w_storage->GetOriginShape();

    uint32_t batch = x_shape.GetDim(0);
    uint32_t inChannels = x_shape.GetDim(1);
    uint32_t inHeight = x_shape.GetDim(2);
    uint32_t inWidth = x_shape.GetDim(3);
    uint32_t outChannels = w_shape.GetDim(1);
    uint32_t kernelH = w_shape.GetDim(2);
    uint32_t kernelW = w_shape.GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t strideH = attrs->GetAttrPointer<int32_t>(0) ? *attrs->GetAttrPointer<int32_t>(0) : 1;
    int32_t strideW = attrs->GetAttrPointer<int32_t>(1) ? *attrs->GetAttrPointer<int32_t>(1) : 1;
    int32_t padH = attrs->GetAttrPointer<int32_t>(2) ? *attrs->GetAttrPointer<int32_t>(2) : 0;
    int32_t padW = attrs->GetAttrPointer<int32_t>(3) ? *attrs->GetAttrPointer<int32_t>(3) : 0;
    int32_t outputPadH = attrs->GetAttrPointer<int32_t>(4) ? *attrs->GetAttrPointer<int32_t>(4) : 0;
    int32_t outputPadW = attrs->GetAttrPointer<int32_t>(5) ? *attrs->GetAttrPointer<int32_t>(5) : 0;
    int32_t groups = attrs->GetAttrPointer<int32_t>(6) ? *attrs->GetAttrPointer<int32_t>(6) : 1;
    int32_t dilationH = attrs->GetAttrPointer<int32_t>(7) ? *attrs->GetAttrPointer<int32_t>(7) : 1;
    int32_t dilationW = attrs->GetAttrPointer<int32_t>(8) ? *attrs->GetAttrPointer<int32_t>(8) : 1;

    uint32_t outHeight = (inHeight - 1) * strideH - 2 * padH + dilationH * (kernelH - 1) + outputPadH + 1;
    uint32_t outWidth  = (inWidth - 1) * strideW - 2 * padW + dilationW * (kernelW - 1) + outputPadW + 1;

    uint32_t tileOutHeight = 8;
    uint32_t tileOutWidth = 8;
    if (tileOutHeight > outHeight) tileOutHeight = outHeight;
    if (tileOutWidth > outWidth) tileOutWidth = outWidth;

    uint32_t tilesH = CeilDiv(outHeight, tileOutHeight);
    uint32_t tilesW = CeilDiv(outWidth, tileOutWidth);
    uint32_t tileNum = tilesH * tilesW;
    uint32_t tileOutElems = tileOutHeight * tileOutWidth * outChannels;

    uint32_t blockDim = tileNum;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_inChannels(inChannels);
    tiling.set_outChannels(outChannels);
    tiling.set_inHeight(inHeight);
    tiling.set_inWidth(inWidth);
    tiling.set_outHeight(outHeight);
    tiling.set_outWidth(outWidth);
    tiling.set_kernelH(kernelH);
    tiling.set_kernelW(kernelW);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_dilationH(dilationH);
    tiling.set_dilationW(dilationW);
    tiling.set_outputPadH(outputPadH);
    tiling.set_outputPadW(outputPadW);
    tiling.set_tileNum(tileNum);
    tiling.set_tileOutElems(tileOutElems);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
