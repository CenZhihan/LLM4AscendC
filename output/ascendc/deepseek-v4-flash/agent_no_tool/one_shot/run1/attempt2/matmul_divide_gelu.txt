namespace optiling {

static constexpr uint32_t BLOCK_M = 128;
static constexpr uint32_t BLOCK_N = 128;
static constexpr uint32_t BLOCK_K = 128;
static constexpr uint32_t ALIGN = 32;

static inline uint32_t AlignUp(uint32_t x, uint32_t a) {
    return ((x + a - 1) / a) * a;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MatmulDivideGeluCustomTilingData tiling;
    const auto x_shape = context->GetInputShape(0)->GetOriginShape();
    const auto w_shape = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = x_shape.GetDim(0);
    uint32_t K = x_shape.GetDim(1);
    uint32_t N = w_shape.GetDim(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *divisor = attrs->GetAttrPointer<float>(0);

    uint32_t blockM = BLOCK_M;
    uint32_t blockN = BLOCK_N;
    uint32_t blockK = BLOCK_K;

    blockM = AlignUp(blockM, ALIGN);
    blockN = AlignUp(blockN, ALIGN);
    blockK = AlignUp(blockK, ALIGN);

    if (blockM > M) blockM = AlignDown(M, ALIGN);
    if (blockM == 0) blockM = M;
    if (blockN > N) blockN = AlignDown(N, ALIGN);
    if (blockN == 0) blockN = N;
    if (blockK > K) blockK = AlignDown(K, ALIGN);
    if (blockK == 0) blockK = K;

    tiling.set_totalM(M);
    tiling.set_totalN(N);
    tiling.set_totalK(K);
    tiling.set_blockM(blockM);
    tiling.set_blockN(blockN);
    tiling.set_blockK(blockK);
    tiling.set_divisor(*divisor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    static constexpr uint32_t cubeNum = 4;
    context->SetBlockDim(cubeNum);

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulDivideGeluCustom {
public:
    __aicore__ inline KernelMatmulDivideGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalM, uint32_t totalN, uint32_t totalK,
                                uint32_t blockM, uint32_t blockN, uint32_t blockK,
                                float divisor) {
        totalM_ = totalM;
        totalN_ = totalN;
        totalK_ = totalK;
        blockM_ = blockM;
        blockN_ = blockN;
        blockK_ = blockK;
        divisor_ = divisor;
        invDivisor_ = 1.0f / divisor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalM_ * totalK_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, totalN_ * totalK_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, totalN_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalM_ * totalN_);

        uint32_t tileXSize = blockM_ * blockK_ * sizeof(float);
        uint32_t tileWSize = blockN_ * blockK_ * sizeof(float);
        uint32_t tileYSize = blockM_ * blockN_ * sizeof(float);
        uint32_t tileBSize = blockN_ * sizeof(float);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileXSize);
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileWSize);
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileBSize);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileYSize);
        pipe_.InitBuffer(tmpQueue_, 1, tileYSize);
    }

    __aicore__ inline void Process() {
        if (totalM_ == 0 || totalN_ == 0 || totalK_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        // Each block processes a subset of M tiles
        for (uint32_t m0 = blockIdx * blockM_; m0 < totalM_; m0 += blockNum * blockM_) {
            uint32_t curM = blockM_;
            if (m0 + curM > totalM_) curM = totalM_ - m0;

            // Process all N tiles for this set of M rows
            for (uint32_t n0 = 0; n0 < totalN_; n0 += blockN_) {
                uint32_t curN = blockN_;
                if (n0 + curN > totalN_) curN = totalN_ - n0;

                // Load bias for current N block
                AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[n0], curN);

                // Initialize accumulator with bias
                AscendC::LocalTensor<float> accLocal = tmpQueue_.AllocTensor<float>();
                // Replicate bias for each row in curM
                for (uint32_t r = 0; r < curM; ++r) {
                    AscendC::LocalTensor<float> rowAcc = accLocal[r * curN];
                    AscendC::DataCopy(rowAcc, bLocal, curN);
                }

                // Iterate over K tiles
                for (uint32_t k0 = 0; k0 < totalK_; k0 += blockK_) {
                    uint32_t curK = blockK_;
                    if (k0 + curK > totalK_) curK = totalK_ - k0;

                    // Load X tile (curM x curK) – note X is stored row-major: [m0*m0_offset + k0]
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    uint32_t xOffset = m0 * totalK_ + k0;
                    AscendC::DataCopy(xLocal, xGm_[xOffset], curM * curK);

                    // Load W tile (curN x curK) – W is stored as (output_features, input_features)
                    AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
                    uint32_t wOffset = n0 * totalK_ + k0;
                    AscendC::DataCopy(wLocal, wGm_[wOffset], curN * curK);

                    // Accumulate: acc[m][n] += sum_k X[m][k] * W[n][k]
                    // We'll perform an outer product of each row of X with each row of W
                    // For each row m in curM:
                    //   for each row n in curN:
                    //      dot product over k (curK elements)
                    // We'll use vector multiplication and reduction.
                    for (uint32_t m = 0; m < curM; ++m) {
                        AscendC::LocalTensor<float> xRow = xLocal[m * curK];
                        for (uint32_t n = 0; n < curN; ++n) {
                            AscendC::LocalTensor<float> wRow = wLocal[n * curK];
                            AscendC::LocalTensor<float> prod = pipe_.AllocTensor<float>(curK);
                            AscendC::Mul(prod, xRow, wRow, curK);
                            float sum = 0.0f;
                            AscendC::ReduceAdd(pipe_, &sum, prod, curK);
                            AccLocal<float> accElem = accLocal[m * curN + n];
                            AccElem = accElem + sum;
                            pipe_.FreeTensor(prod);
                        }
                    }

                    inQueueX_.FreeTensor(xLocal);
                    inQueueW_.FreeTensor(wLocal);
                }

                // Now accLocal holds C = X * W^T + b for this N block
                // Apply division and GELU
                uint32_t totalElems = curM * curN;
                // Divide
                AscendC::Muls(accLocal, accLocal, invDivisor_, totalElems);
                // GELU: y = 0.5 * x * (1 + tanh( sqrt(2/pi) * (x + 0.044715 * x^3) ))
                // Compute x^3
                AscendC::LocalTensor<float> tmp = pipe_.AllocTensor<float>(totalElems);
                AscendC::Mul(tmp, accLocal, accLocal, totalElems);
                AscendC::Mul(tmp, tmp, accLocal, totalElems);
                AscendC::Muls(tmp, tmp, 0.044715f, totalElems);
                AscendC::Add(tmp, tmp, accLocal, totalElems);
                AscendC::Muls(tmp, tmp, 0.7978845608f, totalElems); // sqrt(2/pi)
                // tanh approximation? Use AscendC::Tanh
                AscendC::Tanh(tmp, tmp, totalElems);
                AscendC::Adds(tmp, tmp, 1.0f, totalElems);
                AscendC::Mul(accLocal, accLocal, tmp, totalElems);
                AscendC::Muls(accLocal, accLocal, 0.5f, totalElems);
                pipe_.FreeTensor(tmp);

                // Write result to global
                uint32_t yOffset = m0 * totalN_ + n0;
                AscendC::DataCopy(yGm_[yOffset], accLocal, totalElems);

                pipe_.FreeTensor(accLocal);
                inQueueB_.FreeTensor(bLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalM_ {0};
    uint32_t totalN_ {0};
    uint32_t totalK_ {0};
    uint32_t blockM_ {0};
    uint32_t blockN_ {0};
    uint32_t blockK_ {0};
    float divisor_ {1.0f};
    float invDivisor_ {1.0f};
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                                                   GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulDivideGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.totalM, tiling_data.totalN, tiling_data.totalK,
            tiling_data.blockM, tiling_data.blockN, tiling_data.blockK,
            tiling_data.divisor);
    op.Process();
}
