#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom {
public:
    __aicore__ inline KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalBatches, uint32_t inFeatures, uint32_t outFeatures)
    {
        batchTotal_ = totalBatches;
        inFeat_ = inFeatures;
        outFeat_ = outFeatures;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchTotal_ * inFeat_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, outFeat_ * inFeat_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, outFeat_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchTotal_);

        // Use TBuf for temporary accumulators; call InitBuffer with single size.
        pipe_.InitBuffer(wRowSumBuf_, inFeat_ * sizeof(float));
        // wTileBuf_ and bTileBuf_ will be allocated per-tile; we use the simplest tiling: tileOut_ = 1
        // to avoid huge local memory usage.
        tileOut_ = 1;
        pipe_.InitBuffer(wTileBuf_, tileOut_ * inFeat_ * sizeof(float));
        pipe_.InitBuffer(bTileBuf_, tileOut_ * sizeof(float));
        // x local buffer with double buffering
        pipe_.InitBuffer(xQue_, BUFFER_NUM, inFeat_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (inFeat_ == 0 || outFeat_ == 0 || batchTotal_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // ---- Step 1: Compute wRowSum and bSum (scalar) on this core ----
        AscendC::LocalTensor<float> wRowSum = wRowSumBuf_.Get<float>();
        AscendC::Duplicate(wRowSum, 0.0f, inFeat_);
        float bSum = 0.0f;

        uint32_t j = 0;
        while (j < outFeat_) {
            uint32_t curTileOut = tileOut_;  // tileOut_ is 1
            if (j + curTileOut > outFeat_) curTileOut = outFeat_ - j;

            // Load w[j:j+curTileOut, :] into local
            AscendC::LocalTensor<float> wTile = wTileBuf_.Get<float>();
            AscendC::DataCopy(wTile, wGm_[j * inFeat_], curTileOut * inFeat_);

            // Load b[j:j+curTileOut]
            AscendC::LocalTensor<float> bTile = bTileBuf_.Get<float>();
            AscendC::DataCopy(bTile, bGm_[j], curTileOut);

            // Update wRowSum
            for (uint32_t r = 0; r < curTileOut; ++r) {
                AscendC::LocalTensor<float> wRow = wTile[r * inFeat_];
                AscendC::Add(wRowSum, wRowSum, wRow, inFeat_);
            }

            // Update bSum using GetValue (since curTileOut is small, 1)
            for (uint32_t r = 0; r < curTileOut; ++r) {
                bSum += bTile.GetValue(r);
            }

            j += curTileOut;
        }

        // ---- Step 2: Process batches assigned to this core ----
        uint32_t batchStart = blockIdx;
        while (batchStart < batchTotal_) {
            // Load x[batchStart] into local via queue
            AscendC::LocalTensor<float> xLocal = xQue_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[batchStart * inFeat_], inFeat_);

            // Compute dot(xLocal, wRowSum) + bSum
            AscendC::LocalTensor<float> dotBuf = xQue_.AllocTensor<float>();  // use separate buffer
            AscendC::Mul(dotBuf, xLocal, wRowSum, inFeat_);
            float dot = 0.0f;
            for (uint32_t k = 0; k < inFeat_; ++k) {
                dot += dotBuf.GetValue(k);
            }
            float outVal = dot + bSum;

            // Write to global y
            yGm_.SetValue(batchStart, outVal);

            xQue_.FreeTensor(xLocal);
            xQue_.FreeTensor(dotBuf);

            batchStart += blockNum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> wRowSumBuf_;   // size inFeat_
    AscendC::TBuf<AscendC::TPosition::VECCALC> wTileBuf_;      // tileOut_ * inFeat_
    AscendC::TBuf<AscendC::TPosition::VECCALC> bTileBuf_;      // tileOut_
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchTotal_ {0};
    uint32_t inFeat_ {0};
    uint32_t outFeat_ {0};
    uint32_t tileOut_ {1};   // fixed to 1 to avoid excessive UB usage
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom op;
    op.Init(x, w, b, y, tiling_data.totalBatches, tiling_data.inFeatures, tiling_data.outFeatures);
    op.Process();
}
