host_operator_src="""

#include "average_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: we load a small window per output, plus intermediate sums and outputs
// For each output point we need kernel_size floats from input and intermediate.
// Working set per block: we process tileOut output points, each requiring up to kernel_size+1? 
// Actually we will load a contiguous block of input that covers the windows for tileOut outputs.
// Approximate: max window span = (tileOut-1)*stride + kernel_size. That can be large.
// Better: we process tileOut outputs, but load windows one by one in a loop.
// Easiest: allocate a local buffer of size kernel_size (max 16) and process in scalar mode.
// So UB budget is small. We'll set a conservative tileOut.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core share
static constexpr uint32_t ALIGN_ELEMS = 16;              // small alignment for loops
static constexpr uint32_t MAX_TILE_OUT = 4096;           // limit per block to avoid long latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling1dCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernelSize = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(2);

    const auto &inputShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t inputN = inputShape.GetDim(0);
    uint32_t inputC = inputShape.GetDim(1);
    uint32_t inputL = inputShape.GetDim(2);

    uint32_t kernel = static_cast<uint32_t>(*kernelSize);
    uint32_t s = static_cast<uint32_t>(*stride);
    uint32_t p = static_cast<uint32_t>(*padding);

    // Compute output length
    int32_t outL = static_cast<int32_t>((inputL + 2 * p - kernel) / s) + 1;
    if (outL < 0) outL = 0;
    uint32_t outputL = static_cast<uint32_t>(outL);

    uint32_t totalOutLength = inputN * inputC * outputL;

    // Compute tileOut: we process tileOut output positions per block.
    // Each output requires loading kernel elements, so UB usage is small.
    uint32_t tileOut = totalOutLength;
    if (totalOutLength > MAX_TILE_OUT)
        tileOut = MAX_TILE_OUT;
    if (tileOut < 1) tileOut = 1;

    // For high parallelism, set blockDim based on number of tiles.
    uint32_t numTiles = CeilDiv(totalOutLength, tileOut);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutLength(totalOutLength);
    tiling.set_tileOut(tileOut);
    tiling.set_inputN(inputN);
    tiling.set_inputC(inputC);
    tiling.set_inputL(inputL);
    tiling.set_kernelSize(kernel);
    tiling.set_stride(s);
    tiling.set_padding(p);
    tiling.set_outputL(outputL);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &dims = x_shape->GetDimNum();
    if (dims != 3) return GRAPH_FAILED;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernelSize = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding = attrs->GetAttrPointer<int64_t>(2);

    int64_t N = x_shape->GetDim(0);
    int64_t C = x_shape->GetDim(1);
    int64_t L = x_shape->GetDim(2);
    int64_t outL = (L + 2 * (*padding) - (*kernelSize)) / (*stride) + 1;
    if (outL < 0) outL = 0;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, outL);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelAveragePooling1dCustom {
public:
    __aicore__ inline KernelAveragePooling1dCustom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR y,
        uint32_t totalOutLength, uint32_t tileOut,
        uint32_t inputN, uint32_t inputC, uint32_t inputL,
        uint32_t kernelSize, uint32_t stride, uint32_t padding,
        uint32_t outputL)
    {
        xGm_.SetGlobalBuffer((__gm__ float *)x);
        yGm_.SetGlobalBuffer((__gm__ float *)y);
        totalOutLength_ = totalOutLength;
        tileOut_ = tileOut;
        inputN_ = inputN;
        inputC_ = inputC;
        inputL_ = inputL;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        outputL_ = outputL;
        invKernel_ = 1.0f / static_cast<float>(kernelSize_);

        // Allocate local buffers: one tile of output values (float) and a small window
        pipe_.InitBuffer(outQueueLocal_, BUFFER_NUM, tileOut_ * sizeof(float));
        // We'll use a local scalar for sum; no large temp needed.
    }

    __aicore__ inline void Process()
    {
        if (totalOutLength_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t strideBlocks = blockNum * tileOut_;

        // Each block processes a contiguous chunk of output positions (flattened)
        uint32_t base = blockIdx * tileOut_;
        if (base >= totalOutLength_) return;

        uint32_t remaining = totalOutLength_ - base;
        uint32_t curTile = (remaining < tileOut_) ? remaining : tileOut_;

        while (true) {
            // Compute and write one tile
            ComputeTile(base, curTile);
            base += strideBlocks;
            if (base >= totalOutLength_) break;
            remaining = totalOutLength_ - base;
            curTile = (remaining < tileOut_) ? remaining : tileOut_;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t baseOutFlat, uint32_t tileOut)
    {
        // Allocate output buffer
        AscendC::LocalTensor<float> yLocal = outQueueLocal_.AllocTensor<float>();

        // For each output index in this tile
        for (uint32_t i = 0; i < tileOut; i++) {
            uint32_t globalOutIdx = baseOutFlat + i;
            // Decode to (n, c, outPos)
            uint32_t outPos = globalOutIdx % outputL_;
            uint32_t rem = globalOutIdx / outputL_;
            uint32_t c = rem % inputC_;
            uint32_t n = rem / inputC_;

            // Compute input start index (virtual)
            int32_t start = static_cast<int32_t>(outPos * stride_) - static_cast<int32_t>(padding_);
            int32_t end = start + static_cast<int32_t>(kernelSize_); // exclusive

            float sum = 0.0f;
            int32_t inputIdx; // absolute input index (n, c, L)
            for (int32_t j = start; j < end; j++) {
                if (j >= 0 && j < static_cast<int32_t>(inputL_)) {
                    // input index = n * inputC_ * inputL_ + c * inputL_ + j
                    inputIdx = static_cast<int32_t>(n * inputC_ * inputL_ + c * inputL_ + j);
                    sum += xGm_[inputIdx];
                }
                // otherwise zero
            }
            yLocal.SetValue(i, sum * invKernel_);
        }

        // Write output
        uint32_t flatOffset = baseOutFlat; // output is flattened
        AscendC::DataCopy(yGm_[flatOffset], yLocal, tileOut);
        outQueueLocal_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> outQueueLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLength_;
    uint32_t tileOut_;
    uint32_t inputN_;
    uint32_t inputC_;
    uint32_t inputL_;
    uint32_t kernelSize_;
    uint32_t stride_;
    uint32_t padding_;
    uint32_t outputL_;
    float invKernel_;
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling1dCustom op;
    op.Init(
        x, y,
        tiling_data.totalOutLength, tiling_data.tileOut,
        tiling_data.inputN, tiling_data.inputC, tiling_data.inputL,
        tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
        tiling_data.outputL);
    op.Process();
}
