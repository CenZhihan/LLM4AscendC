host_operator_src="""

#include "square_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// Estimated UB usage per block:
//   halfA: blockM * blockK * 2 bytes
//   halfB: blockK * blockN * 2 bytes
//   halfC: blockM * blockN * 2 bytes
//   floatC: blockM * blockN * 4 bytes
//   tmp for cast: blockM*blockK*4 (float) + blockM*blockK*2 (half) etc.
// Conservative total ~ 2 * (blockM*blockK + blockK*blockN + blockM*blockN) * 2  bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 16;

static inline uint32_t AlignUp(uint32_t x, uint32_t a) {
    return (x + a - 1) / a * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SquareMatrixMultiplicationCustomTilingData tiling;

    const auto &shapeA = context->GetInputShape(0)->GetOriginShape();
    const auto &shapeB = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = static_cast<uint32_t>(shapeA.GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA.GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB.GetDim(1));

    // Determine cube-friendly block sizes
    // Aim for blocks that maximise utilisation while fitting UB
    uint32_t blockM = 64, blockN = 64, blockK = 128;

    // Adjust if matrix is smaller
    if (M < blockM) blockM = AlignUp(M, ALIGN);
    if (N < blockN) blockN = AlignUp(N, ALIGN);
    if (K < blockK) blockK = AlignUp(K, ALIGN);

    // Rough UB check: halfA+halfB+halfC+floatC
    uint32_t totalBytes = (blockM*blockK + blockK*blockN + blockM*blockN) * 2 * 2; // half size * 2 copies?
    // plus conversion temps – ensure fits.
    if (totalBytes > UB_BUDGET_BYTES) {
        // reduce until fits
        while (totalBytes > UB_BUDGET_BYTES && (blockM > 16 || blockN > 16 || blockK > 16)) {
            if (blockM > 16) blockM /= 2;
            if (blockN > 16) blockN /= 2;
            if (blockK > 16) blockK /= 2;
            blockM = AlignUp(blockM, ALIGN);
            blockN = AlignUp(blockN, ALIGN);
            blockK = AlignUp(blockK, ALIGN);
            totalBytes = (blockM*blockK + blockK*blockN + blockM*blockN) * 2 * 2;
        }
    }

    // Number of output tiles
    uint32_t tilesM = (M + blockM - 1) / blockM;
    uint32_t tilesN = (N + blockN - 1) / blockN;
    uint32_t numTiles = tilesM * tilesN;

    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_blockM(blockM);
    tiling.set_blockN(blockN);
    tiling.set_blockK(blockK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    const gert::Shape *shapeB = context->GetInputShape(1);

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelSquareMatMul {
public:
    __aicore__ inline KernelSquareMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t blockM, uint32_t blockN, uint32_t blockK)
    {
        M_ = M; N_ = N; K_ = K;
        bM_ = blockM; bN_ = blockN; bK_ = blockK;

        AGm_.SetGlobalBuffer((__gm__ float*)A, M_ * K_);
        BGm_.SetGlobalBuffer((__gm__ float*)B, K_ * N_);
        CGm_.SetGlobalBuffer((__gm__ float*)C, M_ * N_);

        // Tiles: we need local buffers for A (float -> half), B (float -> half), C (half accumulator) and float output
        uint32_t sizeA = bM_ * bK_ * sizeof(half);      // half copy
        uint32_t sizeB = bK_ * bN_ * sizeof(half);
        uint32_t sizeCacc = bM_ * bN_ * sizeof(half);   // accumulator
        uint32_t sizeCout = bM_ * bN_ * sizeof(float);

        // Additional temps for casting
        uint32_t sizeAfloat = bM_ * bK_ * sizeof(float);
        uint32_t sizeBfloat = bK_ * bN_ * sizeof(float);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, sizeAfloat);
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, sizeBfloat);
        pipe_.InitBuffer(halfQueueA_, 1, sizeA);
        pipe_.InitBuffer(halfQueueB_, 1, sizeB);
        pipe_.InitBuffer(accQueue_, 1, sizeCacc);
        pipe_.InitBuffer(tmpMatMulQueue_, 1, sizeCacc);
        pipe_.InitBuffer(outQueueC_, 1, sizeCout);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        // Grid-stride over output tiles (iM, iN)
        uint32_t tilesM = (M_ + bM_ - 1) / bM_;
        uint32_t tilesN = (N_ + bN_ - 1) / bN_;
        uint32_t totalTiles = tilesM * tilesN;

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            uint32_t iM = tileId / tilesN;
            uint32_t iN = tileId % tilesN;

            uint32_t rowStart = iM * bM_;
            uint32_t colStart = iN * bN_;
            uint32_t curM = (rowStart + bM_ <= M_) ? bM_ : (M_ - rowStart);
            uint32_t curN = (colStart + bN_ <= N_) ? bN_ : (N_ - colStart);

            // Allocate accumulator tile and set to zero
            LocalTensor<half> accTile = accQueue_.AllocTensor<half>();
            Duplicate(accTile, (half)0.0f, curM * curN);

            // Loop over K dimension in tiles
            uint32_t tilesK = (K_ + bK_ - 1) / bK_;
            for (uint32_t iK = 0; iK < tilesK; iK++) {
                uint32_t kStart = iK * bK_;
                uint32_t curK = (kStart + bK_ <= K_) ? bK_ : (K_ - kStart);

                // Load A tile (rowStart, kStart) size curM x curK
                LocalTensor<float> aFloat = inQueueA_.AllocTensor<float>();
                DataCopy(aFloat, AGm_[rowStart * K_ + kStart], curM * curK);

                // Load B tile (kStart, colStart) size curK x curN
                LocalTensor<float> bFloat = inQueueB_.AllocTensor<float>();
                DataCopy(bFloat, BGm_[kStart * N_ + colStart], curK * curN);

                // Convert to half
                LocalTensor<half> aHalf = halfQueueA_.AllocTensor<half>();
                Cast(aHalf, aFloat, RoundMode::CAST_ROUND, curM * curK);
                LocalTensor<half> bHalf = halfQueueB_.AllocTensor<half>();
                Cast(bHalf, bFloat, RoundMode::CAST_ROUND, curK * curN);

                // Free float buffers
                inQueueA_.FreeTensor(aFloat);
                inQueueB_.FreeTensor(bFloat);

                // Perform matrix multiplication using cube unit
                MatMul<half, half, half> matmul;
                matmul.SetTensorA(aHalf);
                matmul.SetTensorB(bHalf);
                LocalTensor<half> tmpC = tmpMatMulQueue_.AllocTensor<half>();
                matmul.SetTensorC(tmpC);
                matmul.SetBias(nullptr);
                matmul.SetM(curM);
                matmul.SetN(curN);
                matmul.SetK(curK);
                matmul.IterateAll();

                // Add to accumulator
                Add(accTile, accTile, tmpC, curM * curN);
                tmpMatMulQueue_.FreeTensor(tmpC);

                halfQueueA_.FreeTensor(aHalf);
                halfQueueB_.FreeTensor(bHalf);
            }

            // Convert accumulator to float
            LocalTensor<float> cFloat = outQueueC_.AllocTensor<float>();
            Cast(cFloat, accTile, RoundMode::CAST_ROUND, curM * curN);
            DataCopy(CGm_[rowStart * N_ + colStart], cFloat, curM * curN);

            outQueueC_.FreeTensor(cFloat);
            accQueue_.FreeTensor(accTile);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueA_;      // float A tile
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueB_;      // float B tile
    TQue<TPosition::VECIN, 1> halfQueueA_;             // half A tile
    TQue<TPosition::VECIN, 1> halfQueueB_;             // half B tile
    TQue<TPosition::VECCALC, 1> accQueue_;             // accumulator (half)
    TQue<TPosition::VECCALC, 1> tmpMatMulQueue_;       // temporary matmul result (half)
    TQue<TPosition::VECOUT, 1> outQueueC_;             // float C tile

    GlobalTensor<float> AGm_;
    GlobalTensor<float> BGm_;
    GlobalTensor<float> CGm_;

    uint32_t M_, N_, K_;
    uint32_t bM_, bN_, bK_;
};

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSquareMatMul op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.blockM, tiling_data.blockN, tiling_data.blockK);
    op.Process();
}
