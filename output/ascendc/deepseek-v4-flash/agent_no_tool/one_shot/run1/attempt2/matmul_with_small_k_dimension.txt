host_operator_src="""

#include "matmul_with_small_k_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget: 192 KB per core
// Buffers: 
//   B_tile : K * tileN * sizeof(float)
//   A_row : K * sizeof(float)
//   C_row_tile : tileN * sizeof(float)
//   temp for vector ops : tileN * sizeof(float)
// K is small (<=64), tileN chosen so total fits
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t MAX_K = 64;  // small K assumption
static constexpr uint32_t MIN_TILEN = 128;
static constexpr uint32_t MAX_TILEN = 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MatmulWithSmallKDimensionCustomTilingData tiling;

    const uint32_t M = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetDim(1));

    // Compute tileN such that B_tile (K * tileN * 4) fits in UB with margin
    uint32_t maxTileN = (UB_BUDGET_BYTES - 4 * (K + 2 * 128)) / (K * 4);  // reserve for A_row, C_row, temp
    if (maxTileN < MIN_TILEN) maxTileN = MIN_TILEN;
    if (maxTileN > MAX_TILEN) maxTileN = MAX_TILEN;
    uint32_t tileN = AlignDown(maxTileN, ALIGN_ELEMS);
    if (tileN < ALIGN_ELEMS) tileN = ALIGN_ELEMS;
    if (tileN > N) tileN = N;

    // BlockDim: number of cores, each processes roughly equal rows
    const uint32_t numCores = 24; // typical for 910B, could be adaptive
    uint32_t blockDim = numCores;
    uint32_t coreRows = CeilDiv(M, blockDim);

    // Adjust blockDim if M is small
    while (blockDim > 1 && coreRows * (uint64_t)K * tileN * 4 > UB_BUDGET_BYTES) {
        blockDim--;
        coreRows = CeilDiv(M, blockDim);
    }

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileN(tileN);
    tiling.set_coreRows(coreRows);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    std::vector<int64_t> dims = {a_shape->GetDim(0), b_shape->GetDim(1)};
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, dims[0]);
    c_shape->SetDim(1, dims[1]);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType a_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MatmulWithSmallKDimensionCustom : public OpDef {
public:
    explicit MatmulWithSmallKDimensionCustom(const char* name) : OpDef(name) {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(MatmulWithSmallKDimensionCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulSmallK {
public:
    __aicore__ inline KernelMatmulSmallK() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN, uint32_t coreRows, uint32_t blockDim)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileN_ = tileN;
        coreRows_ = coreRows;
        blockDim_ = blockDim;

        aGm_.SetGlobalBuffer((__gm__ float*)a, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, M_ * N_);

        // Initialize local buffers
        pipe_.InitBuffer(inBufB_, 1, K_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(inBufA_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(outBufC_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t startRow = blockIdx * coreRows_;
        uint32_t endRow = startRow + coreRows_;
        if (endRow > M_) endRow = M_;
        if (startRow >= M_) return;

        // Loop over column tiles
        for (uint32_t jStart = 0; jStart < N_; jStart += tileN_) {
            uint32_t curTileN = (jStart + tileN_ <= N_) ? tileN_ : (N_ - jStart);

            // Load B tile: K x curTileN from global into inBufB_
            AscendC::LocalTensor<float> bLocal = inBufB_.AllocTensor<float>();
            for (uint32_t k = 0; k < K_; k++) {
                // Copy row k of B tile: offset k*curTileN in local, source = global row k starting at column jStart
                AscendC::DataCopy(bLocal.Slice(k * curTileN, curTileN),
                                  bGm_[k * N_ + jStart],
                                  curTileN);
            }

            // Process rows in core's range
            for (uint32_t i = startRow; i < endRow; i++) {
                // Load A row
                AscendC::LocalTensor<float> aLocal = inBufA_.AllocTensor<float>();
                AscendC::DataCopy(aLocal, aGm_[i * K_], K_);

                // Initialize C tile to zero
                AscendC::LocalTensor<float> cLocal = outBufC_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpLocal = tmpBuf_.AllocTensor<float>();
                AscendC::Duplicate(cLocal, 0.0f, curTileN);

                // Outer-product over K
                for (uint32_t k = 0; k < K_; k++) {
                    float aScalar = aLocal(k);
                    // tmp = B_row_k * aScalar
                    AscendC::Muls(tmpLocal, bLocal.Slice(k * curTileN, curTileN), aScalar, curTileN);
                    // c = c + tmp
                    AscendC::Add(cLocal, cLocal, tmpLocal, curTileN);
                }

                // Write C row tile to global
                AscendC::DataCopy(cGm_[i * N_ + jStart], cLocal, curTileN);

                // Free local tensors
                tmpBuf_.FreeTensor(tmpLocal);
                outBufC_.FreeTensor(cLocal);
                inBufA_.FreeTensor(aLocal);
            }

            inBufB_.FreeTensor(bLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inBufB_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inBufA_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outBufC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_;
    uint32_t N_;
    uint32_t K_;
    uint32_t tileN_;
    uint32_t coreRows_;
    uint32_t blockDim_;
};

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSmallK op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileN, tiling_data.coreRows, tiling_data.blockDim);
    op.Process();
}
"""
