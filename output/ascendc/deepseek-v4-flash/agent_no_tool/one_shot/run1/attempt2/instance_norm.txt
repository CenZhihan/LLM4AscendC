host_operator_src="""

#include "instance_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (per block processes one channel of one sample):
//   xTile   :   tileLength * sizeof(float)
//   yTile   :   tileLength * sizeof(float)
//   tmpBuf  :   3 * tileLength * sizeof(float)   // for broadcast/scalar tensors
//   accBuf  :   1 * sizeof(float)  // scalar for reduction
// Total ~ 5 * tileLength * sizeof(float) = 20 * tileLength bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    InstanceNormCustomTilingData tiling;
    const gert::StorageShape *inShape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(0));
    uint32_t C = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(1));
    uint32_t H = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(2));
    uint32_t W = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(3));
    uint32_t totalLength = N * C * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);
    float epsVal = (eps != nullptr) ? *eps : 1e-5f;

    uint32_t channelSize = H * W;
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (channelSize < maxTile) ? channelSize : maxTile;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = (channelSize < ALIGN_ELEMS) ? channelSize : ALIGN_ELEMS;

    // Block dim = number of (sample, channel) pairs
    uint32_t blockDim = N * C;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 8u) blockDim = 8u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_eps(epsVal);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
