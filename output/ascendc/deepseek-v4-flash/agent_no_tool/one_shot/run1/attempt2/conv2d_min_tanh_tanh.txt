#include "conv2d_min_tanh_tanh_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Conv2dMinTanhTanhCustomTilingData tiling;
    // Input shapes: x (N, C, H, W), weight (OC, IC, kH, kW)
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    uint32_t N = xShape->GetDim(0);
    uint32_t C = xShape->GetDim(1);
    uint32_t H = xShape->GetDim(2);
    uint32_t W = xShape->GetDim(3);
    uint32_t OC = wShape->GetDim(0);
    uint32_t kH = wShape->GetDim(2);
    uint32_t kW = wShape->GetDim(3);
    uint32_t Ho = H - kH + 1;
    uint32_t Wo = W - kW + 1;
    uint32_t totalPixels = N * Ho * Wo;

    // Tile size: we need to fit one weight slice (OC * inC * kH * kW) + input window + temp
    // Weight size per OC = inC * kH * kW. We'll load one OC at a time.
    // We'll allocate a buffer for full weight (OC * windowSize) but that may be large.
    // Use a more conservative approach: tile based on number of output pixels.
    // For each pixel, we need windowSize floats input, and we iterate over OC, loading weight slice each time.
    // We can tile by batch*Ho*Wo pixels.
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * 4 * 512); // rough estimate
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tilePixels = maxTile;
    if (totalPixels < tilePixels)
        tilePixels = AlignDown(totalPixels, ALIGN_ELEMS);
    if (tilePixels == 0) tilePixels = 1;

    uint32_t numTiles = CeilDiv(totalPixels, tilePixels);
    uint32_t blockDim = numTiles;
    if (blockDim < 4) blockDim = 4;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_batch(N);
    tiling.set_inC(C);
    tiling.set_outC(OC);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_totalPixels(totalPixels);
    tiling.set_tilePixels(tilePixels);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dMinTanhTanhCustom {
public:
    __aicore__ inline KernelConv2dMinTanhTanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias,
                                GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        batch_ = td.batch;
        inC_ = td.inC;
        outC_ = td.outC;
        H_ = td.H;
        W_ = td.W;
        kH_ = td.kH;
        kW_ = td.kW;
        Ho_ = td.Ho;
        Wo_ = td.Wo;
        totalPixels_ = td.totalPixels;
        tilePixels_ = td.tilePixels;
        blockDim_ = td.blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_ * inC_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outC_ * inC_ * kH_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch_ * Ho_ * Wo_);
        if (conv_bias) {
            biasGm_.SetGlobalBuffer((__gm__ float*)conv_bias, outC_);
            hasBias_ = true;
        } else {
            hasBias_ = false;
        }

        windowSize_ = kH_ * kW_ * inC_;
        // Allocate local buffers
        pipe_.InitBuffer(inQ_, BUFFER_NUM, windowSize_ * sizeof(float));
        pipe_.InitBuffer(weightQ_, BUFFER_NUM, windowSize_ * sizeof(float));  // one OC slice at a time
        pipe_.InitBuffer(outQ_, BUFFER_NUM, 1 * sizeof(float));
        pipe_.InitBuffer(accQ_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalPixels_ == 0 || tilePixels_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = blockDim_;
        uint32_t pixelStart = blockIdx * tilePixels_;
        if (pixelStart >= totalPixels_) return;
        uint32_t pixelEnd = pixelStart + tilePixels_;
        if (pixelEnd > totalPixels_) pixelEnd = totalPixels_;

        // Prefetch first pixel's input window
        if (pixelStart < pixelEnd) {
            uint32_t globalIdx = pixelStart;
            LoadWindow(globalIdx);
        }

        for (uint32_t pixel = pixelStart; pixel < pixelEnd; ++pixel) {
            // Prefetch next window if not last
            if (pixel + 1 < pixelEnd) {
                LoadWindow(pixel + 1);
            }

            // Compute min over outC
            float minVal = 1.0e30f;
            for (uint32_t oc = 0; oc < outC_; ++oc) {
                // Load weight slice for this oc
                AscendC::LocalTensor<float> wLocal = weightQ_.AllocTensor<float>();
                uint32_t weightOffset = oc * windowSize_;
                AscendC::DataCopy(wLocal, wGm_[weightOffset], windowSize_);
                weightQ_.EnQue(wLocal);

                // Deque input window
                AscendC::LocalTensor<float> xLocal = inQ_.DeQue<float>();

                // Elementwise multiply into accumulator
                AscendC::LocalTensor<float> accLocal = accQ_.AllocTensor<float>();
                AscendC::Mul(accLocal, xLocal, wLocal, windowSize_);

                // Sum reduction via loop (scalar)
                float sum = 0.0f;
                for (uint32_t i = 0; i < windowSize_; ++i) {
                    sum += accLocal(i);
                }
                accQ_.FreeTensor(accLocal);
                weightQ_.FreeTensor(wLocal);
                inQ_.FreeTensor(xLocal);

                // Add bias if present
                if (hasBias_) {
                    sum += biasGm_.GetValue(oc);
                }

                // Update min
                if (sum < minVal) minVal = sum;
            }

            // Apply tanh twice: tanh(tanh(x))
            float t = tanhf(minVal);
            t = tanhf(t);

            // Write output
            AscendC::LocalTensor<float> outLocal = outQ_.AllocTensor<float>();
            outLocal(0) = t;
            outQ_.EnQue(outLocal);

            // Copy to global
            AscendC::LocalTensor<float> outCopy = outQ_.DeQue<float>();
            AscendC::DataCopy(yGm_[pixel], outCopy, 1);
            outQ_.FreeTensor(outCopy);
        }
    }

private:
    __aicore__ inline void LoadWindow(uint32_t pixelIdx)
    {
        // Convert pixelIdx to (n, oh, ow)
        uint32_t oh = pixelIdx % (Ho_ * Wo_) / Wo_;
        uint32_t ow = pixelIdx % Wo_;
        uint32_t n = pixelIdx / (Ho_ * Wo_);

        // Input window from x[n, :, oh:oh+kH, ow:ow+kW]
        uint32_t srcBase = n * inC_ * H_ * W_;
        AscendC::LocalTensor<float> xLocal = inQ_.AllocTensor<float>();
        // We need to copy a rectangular patch; easiest: loop over inC and rows
        // For simplicity, we copy row by row. To use DataCopy we need contiguous length.
        // We'll copy entire window by iterating over inC.
        uint32_t offset = 0;
        for (uint32_t c = 0; c < inC_; ++c) {
            for (uint32_t r = 0; r < kH_; ++r) {
                uint32_t rowStart = srcBase + c * H_ * W_ + (oh + r) * W_ + ow;
                AscendC::DataCopy(xLocal[offset], xGm_[rowStart], kW_);
                offset += kW_;
            }
        }
        inQ_.EnQue(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> weightQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t batch_, inC_, outC_, H_, W_, kH_, kW_, Ho_, Wo_;
    uint32_t totalPixels_, tilePixels_, blockDim_;
    uint32_t windowSize_;
    bool hasBias_ = false;
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias,
                                                                  GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanhCustom op;
    op.Init(x, weight, conv_bias, y, workspace, tiling);
    op.Process();
}
