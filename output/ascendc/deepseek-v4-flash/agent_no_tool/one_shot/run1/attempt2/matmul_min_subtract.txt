#include "matmul_min_subtract_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: 2 tiles of A, B, C + bias + workspace
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMinSubtractCustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::StorageShape *wShape = context->GetInputShape(1);
    const uint32_t M = xShape->GetDim(0);
    const uint32_t K = xShape->GetDim(1);
    const uint32_t N = wShape->GetDim(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float constant = attrs->GetAttrPointer<float>(0)[0];

    // Heuristic tile sizes
    uint32_t bestTileM = 1, bestTileN = 1, bestTileK = 1;
    for (uint32_t tM = 1; tM <= M && tM <= 128; tM *= 2) {
        for (uint32_t tN = 1; tN <= N && tN <= 128; tN *= 2) {
            for (uint32_t tK = 1; tK <= K && tK <= 256; tK *= 2) {
                uint32_t bytes = (tM * tK + tN * tK + tM * tN + tN) * sizeof(float);
                if (bytes <= UB_BUDGET_BYTES) {
                    bestTileM = tM;
                    bestTileN = tN;
                    bestTileK = tK;
                }
            }
        }
    }

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(bestTileM);
    tiling.set_tileN(bestTileN);
    tiling.set_tileK(bestTileK);
    tiling.set_constant(constant);

    uint32_t blockDim = 8;
    context->SetBlockDim(blockDim);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)

    y_shape->SetDimNum(2);
    y_shape->SetDim(0, x_shape->GetDim(0));
    y_shape->SetDim(1, w_shape->GetDim(0));

kernel_src="""

#include "kernel_operator.h"
#include "matmul.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulMinSubtractCustom {
public:
    __aicore__ inline KernelMatmulMinSubtractCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                float constant)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN; tileK_ = tileK;
        constant_ = constant;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M * K);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N * K);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M * N);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileN_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(biasBuf_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileM_ * tileN_ * sizeof(float));

        matmul_.SetTensorA(xGm_, {M, K});
        matmul_.SetTensorB(wGm_, {N, K});
        matmul_.SetTensorC(yGm_, {M, N});
        matmul_.SetTransposeB(true);
        matmul_.SetOrgShape(M, N, K);
        matmul_.SetTiling({tileM_, tileN_, tileK_});
    }

    __aicore__ inline void Process()
    {
        uint32_t curM, curN;
        for (uint32_t mIdx = 0; mIdx < M_; mIdx += tileM_) {
            curM = (mIdx + tileM_ <= M_) ? tileM_ : M_ - mIdx;
            for (uint32_t nIdx = 0; nIdx < N_; nIdx += tileN_) {
                curN = (nIdx + tileN_ <= N_) ? tileN_ : N_ - nIdx;
                matmul_.Iterate(mIdx, nIdx);

                AscendC::LocalTensor<float> biasLocal = biasBuf_.AllocTensor<float>();
                AscendC::DataCopy(biasLocal, bGm_[nIdx], curN);

                AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
                AscendC::DataCopy(cLocal, yGm_[mIdx * N_ + nIdx], curM * curN);

                for (uint32_t row = 0; row < curM; row++) {
                    AscendC::Add(cLocal[row * curN], cLocal[row * curN], biasLocal, curN);
                }

                AscendC::LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>();
                AscendC::Mins(tmp, cLocal, constant_, curM * curN);
                AscendC::Sub(cLocal, tmp, constant_, curM * curN);

                AscendC::DataCopy(yGm_[mIdx * N_ + nIdx], cLocal, curM * curN);

                outQueueC_.FreeTensor(cLocal);
                biasBuf_.FreeTensor(biasLocal);
                tmpBuf_.FreeTensor(tmp);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> biasBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_, wGm_, bGm_, yGm_;
    uint32_t M_, N_, K_;
    uint32_t tileM_, tileN_, tileK_;
    float constant_;
    AscendC::MatMul<AscendC::MatMulType<AscendC::ROUND, AscendC::ND, AscendC::ND>> matmul_;
};

extern "C" __global__ __aicore__ void matmul_min_subtract_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMinSubtractCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.constant);
    op.Process();
}
