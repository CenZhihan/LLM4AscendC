kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class MatmulScaleResidualAddClampLogSumExpMishCustomKernel {
public:
    __aicore__ inline MatmulScaleResidualAddClampLogSumExpMishCustomKernel() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR scaling, GM_ADDR clamp_min, GM_ADDR clamp_max,
                                GM_ADDR y,
                                uint32_t batch, uint32_t inputSize, uint32_t hiddenSize,
                                uint32_t tileBatch, float s, float lo, float hi)
    {
        batch_ = batch;
        inputSize_ = inputSize;
        hiddenSize_ = hiddenSize;
        tileBatch_ = tileBatch;
        scaling_ = s;
        clampMin_ = lo;
        clampMax_ = hi;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * inputSize);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, hiddenSize * inputSize);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, hiddenSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch);

        // Allocate local buffers
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileBatch_ * inputSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, 1, hiddenSize_ * inputSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, hiddenSize_ * sizeof(float));
        // Two rows per tile: one for matmul result, one for exp sum etc.
        pipe_.InitBuffer(resultQueue_, BUFFER_NUM, tileBatch_ * hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(workQueue_, 1, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || hiddenSize_ == 0 || inputSize_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Global row range for this block
        uint32_t rowsPerBlock = batch_ / blockNum;
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = (blockIdx == blockNum - 1) ? batch_ : startRow + rowsPerBlock;

        // Load weight and bias once (shared for all rows in block)
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, hiddenSize_ * inputSize_);
        inQueueW_.EnQue(wLocal);

        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        AscendC::DataCopy(bLocal, bGm_, hiddenSize_);
        inQueueB_.EnQue(bLocal);

        for (uint32_t row = startRow; row < endRow; row++) {
            // Load one row of x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[row * inputSize_], inputSize_);
            inQueueX_.EnQue(xLocal);

            // Matmul: y_row = x * W^T + bias
            AscendC::LocalTensor<float> resultLocal = resultQueue_.AllocTensor<float>();
            // Manual dot product for each output element
            for (uint32_t outIdx = 0; outIdx < hiddenSize_; outIdx++) {
                float sum = 0.0f;
                for (uint32_t inIdx = 0; inIdx < inputSize_; inIdx++) {
                    sum += xLocal(inIdx) * wLocal(outIdx * inputSize_ + inIdx);
                }
                sum += bLocal(outIdx);
                resultLocal(outIdx) = sum;
            }
            inQueueX_.FreeTensor(xLocal);

            // scaling + residual: result = result * scaling + result = result * (scaling + 1)
            AscendC::Muls(resultLocal, resultLocal, scaling_ + 1.0f, hiddenSize_);

            // clamp
            AscendC::Clamp(resultLocal, resultLocal, clampMin_, clampMax_, hiddenSize_);

            // logsumexp: compute max, exp, sum, log
            AscendC::LocalTensor<float> workLocal = workQueue_.AllocTensor<float>();
            // Find max
            float maxVal = AscendC::ReduceMax(resultLocal, hiddenSize_).GetValue(0);
            // Subtract max and exp
            AscendC::Subs(resultLocal, resultLocal, maxVal, hiddenSize_);
            AscendC::Exp(resultLocal, resultLocal, hiddenSize_);
            // Sum
            float sumExp = AscendC::ReduceSum(resultLocal, hiddenSize_).GetValue(0);
            // log(sumExp) + maxVal = logsumexp
            float lse = logf(sumExp) + maxVal;
            // mish(x) = x * tanh(log(1+exp(x)))   where x = lse
            float mish_val = lse * tanhf(logf(1.0f + expf(lse)));
            // Final output = lse * mish_val
            float output = lse * mish_val;

            // Store to global
            AscendC::LocalTensor<float> scalarLocal = scalarQueue_.AllocTensor<float>();
            scalarLocal(0) = output;
            AscendC::DataCopy(yGm_[row], scalarLocal, 1);
            scalarQueue_.FreeTensor(scalarLocal);

            workQueue_.FreeTensor(workLocal);
            resultQueue_.FreeTensor(resultLocal);
        }

        // Free weight and bias copies
        inQueueW_.FreeTensor(wLocal);
        inQueueB_.FreeTensor(bLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> resultQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t inputSize_;
    uint32_t hiddenSize_;
    uint32_t tileBatch_;
    float scaling_;
    float clampMin_;
    float clampMax_;
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
    GM_ADDR scaling, GM_ADDR clamp_min, GM_ADDR clamp_max,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    MatmulScaleResidualAddClampLogSumExpMishCustomKernel op;
    op.Init(x, weight, bias, scaling, clamp_min, clamp_max, y,
            tiling_data.batch, tiling_data.inputSize, tiling_data.hiddenSize,
            tiling_data.tileBatch, tiling_data.scaling, tiling_data.clampMin, tiling_data.clampMax);
    op.Process();
}
