namespace optiling {

static inline uint32_t ComputeSize(uint32_t in, uint32_t k, uint32_t s, uint32_t p, bool ceilMode) {
    int32_t temp = static_cast<int32_t>(in + 2 * p) - static_cast<int32_t>(k);
    int32_t out = temp / static_cast<int32_t>(s) + 1;
    if (ceilMode && (temp % static_cast<int32_t>(s) != 0)) {
        out += 1;
    }
    if (out < 0) out = 0;
    return static_cast<uint32_t>(out);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batchSize = shape.GetDim(0);
    uint32_t channels = shape.GetDim(1);
    uint32_t depthIn = shape.GetDim(2);
    uint32_t heightIn = shape.GetDim(3);
    uint32_t widthIn = shape.GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const auto *kernel_size = attrs->GetListInt(0);
    const auto *stride = attrs->GetListInt(1);
    const auto *padding = attrs->GetListInt(2);
    bool ceilMode = attrs->GetBool(3);

    uint32_t kernelDepth = static_cast<uint32_t>(kernel_size->GetData()[0]);
    uint32_t kernelHeight = static_cast<uint32_t>(kernel_size->GetData()[1]);
    uint32_t kernelWidth = static_cast<uint32_t>(kernel_size->GetData()[2]);
    uint32_t strideDepth = static_cast<uint32_t>(stride->GetData()[0]);
    uint32_t strideHeight = static_cast<uint32_t>(stride->GetData()[1]);
    uint32_t strideWidth = static_cast<uint32_t>(stride->GetData()[2]);
    uint32_t padDepth = static_cast<uint32_t>(padding->GetData()[0]);

    tiling.set_padWidth(padWidth);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t channels,
                                uint32_t depthIn, uint32_t heightIn, uint32_t widthIn,
                                uint32_t kernelDepth, uint32_t kernelHeight, uint32_t kernelWidth,
                                uint32_t strideDepth, uint32_t strideHeight, uint32_t strideWidth,
                                uint32_t padDepth, uint32_t padHeight, uint32_t padWidth,
                                uint32_t depthOut, uint32_t heightOut, uint32_t widthOut,
                                uint32_t totalOut, bool ceilMode)
    {
        batchSize_ = batchSize; channels_ = channels;
        depthIn_ = depthIn; heightIn_ = heightIn; widthIn_ = widthIn;
        kernelDepth_ = kernelDepth; kernelHeight_ = kernelHeight; kernelWidth_ = kernelWidth;
        strideDepth_ = strideDepth; strideHeight_ = strideHeight; strideWidth_ = strideWidth;
        padDepth_ = padDepth; padHeight_ = padHeight; padWidth_ = padWidth;
        depthOut_ = depthOut; heightOut_ = heightOut; widthOut_ = widthOut;
        totalOut_ = totalOut;
        ceilMode_ = ceilMode;

        xGm_.SetGlobalBuffer((__gm__ float*)x, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);
        xPtr_ = (__gm__ float*)x;
        yPtr_ = (__gm__ float*)y;

        tileOut_ = 1024;
        pipe_.InitBuffer(inQueueWindow_, BUFFER_NUM, tileOut_ * kernelDepth_ * kernelHeight_ * kernelWidth_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpMax_, 1, tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpIdx_, 1, tileOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockDim = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t totalOutPerBlock = (totalOut_ + blockDim - 1) / blockDim;
        uint32_t startIdx = blockIdx * totalOutPerBlock;
        uint32_t endIdx = startIdx + totalOutPerBlock;
        if (startIdx >= totalOut_) return;
        if (endIdx > totalOut_) endIdx = totalOut_;

        for (uint32_t idx = startIdx; idx < endIdx; idx += tileOut_) {
            uint32_t curTile = (endIdx - idx) < tileOut_ ? (endIdx - idx) : tileOut_;
            for (uint32_t i = 0; i < curTile; i++) {
                uint32_t outIdx = idx + i;
                uint32_t wOutIdx = outIdx % widthOut_;
                uint32_t remainder = outIdx / widthOut_;
                uint32_t hOutIdx = remainder % heightOut_;
                remainder /= heightOut_;
                uint32_t dOutIdx = remainder % depthOut_;
                remainder /= depthOut_;
                uint32_t cIdx = remainder % channels_;
                uint32_t nIdx = remainder / channels_;

                int32_t dStart = static_cast<int32_t>(dOutIdx * strideDepth_) - static_cast<int32_t>(padDepth_);
                int32_t hStart = static_cast<int32_t>(hOutIdx * strideHeight_) - static_cast<int32_t>(padHeight_);
                int32_t wStart = static_cast<int32_t>(wOutIdx * strideWidth_) - static_cast<int32_t>(padWidth_);

                float maxVal = -FLT_MAX;
                for (uint32_t kd = 0; kd < kernelDepth_; kd++) {
                    int32_t d = dStart + kd;
                    if (d < 0 || d >= static_cast<int32_t>(depthIn_)) continue;
                    for (uint32_t kh = 0; kh < kernelHeight_; kh++) {
                        int32_t h = hStart + kh;
                        if (h < 0 || h >= static_cast<int32_t>(heightIn_)) continue;
                        for (uint32_t kw = 0; kw < kernelWidth_; kw++) {
                            int32_t w = wStart + kw;
                            if (w < 0 || w >= static_cast<int32_t>(widthIn_)) continue;
                            uint32_t inIdx = ((nIdx * channels_ + cIdx) * depthIn_ + d) * heightIn_ + h) * widthIn_ + w;
                            float val = xPtr_[inIdx];
                            if (val > maxVal) maxVal = val;
                        }
                    }
                }
                yPtr_[outIdx] = maxVal;
            }
        }
    }

private:
    uint32_t batchSize_, channels_, depthIn_, heightIn_, widthIn_;
    uint32_t kernelDepth_, kernelHeight_, kernelWidth_;
    uint32_t strideDepth_, strideHeight_, strideWidth_;
    uint32_t padDepth_, padHeight_, padWidth_;
    uint32_t depthOut_, heightOut_, widthOut_;
    uint32_t totalOut_;
    bool ceilMode_;

    __gm__ float *xPtr_;
    __gm__ float *yPtr_;

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWindow_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpIdx_;

    uint32_t tileOut_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.batchSize, tiling_data.channels,
            tiling_data.depthIn, tiling_data.heightIn, tiling_data.widthIn,
            tiling_data.kernelDepth, tiling_data.kernelHeight, tiling_data.kernelWidth,
            tiling_data.strideDepth, tiling_data.strideHeight, tiling_data.strideWidth,
            tiling_data.padDepth, tiling_data.padHeight, tiling_data.padWidth,
            tiling_data.depthOut, tiling_data.heightOut, tiling_data.widthOut,
            tiling_data.totalOut, tiling_data.ceilMode);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0, return_indices: bool = False, ceil_mode: bool = False):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size if isinstance(kernel_size, (list, tuple)) else [kernel_size]*3
        self.stride = stride if stride is not None else self.kernel_size
        if not isinstance(self.stride, (list, tuple)):
            self.stride = [self.stride]*3
        self.padding = padding if isinstance(padding, (list, tuple)) else [padding]*3
        self.ceil_mode = ceil_mode

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding, self.ceil_mode)
'''
