kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom {
public:
    __aicore__ inline KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batchSize, uint32_t hiddenSize, uint32_t outputSize)
    {
        batchSize_ = batchSize;
        hiddenSize_ = hiddenSize;
        outputSize_ = outputSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * hiddenSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, outputSize * hiddenSize);
        bGm_.SetGlobalBuffer((__gm__ float*)b, outputSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize);

        // Allocate local buffers
        // inputRow and outputRow are the main buffers
        pipe_.InitBuffer(inQueue_, 1, hiddenSize_ * sizeof(float));   // input row
        pipe_.InitBuffer(outQueue_, 1, outputSize_ * sizeof(float));  // row output
        pipe_.InitBuffer(tmpQueue_, 3, hiddenSize_ * sizeof(float));  // weight row and temps
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || hiddenSize_ == 0 || outputSize_ == 0) return;

        for (uint32_t batchIdx = 0; batchIdx < batchSize_; ++batchIdx) {
            ProcessOneRow(batchIdx);
        }
    }

private:
    __aicore__ inline void ProcessOneRow(uint32_t batchIdx)
    {
        // Load input row
        AscendC::LocalTensor<float> inputRow = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(inputRow, xGm_[batchIdx * hiddenSize_], hiddenSize_);

        // Initialize output row with bias
        AscendC::LocalTensor<float> outputRow = outQueue_.AllocTensor<float>();
        AscendC::DataCopy(outputRow, bGm_, outputSize_);

        // For each output feature, compute dot product with weight row and add to outputRow
        for (uint32_t j = 0; j < outputSize_; ++j) {
            AscendC::LocalTensor<float> weightRow = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(weightRow, wGm_[j * hiddenSize_], hiddenSize_);

            // Compute dot product using vector multiply and add (MVP)
            // We'll use a temporary for partial sum
            AscendC::LocalTensor<float> prod = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(prod, inputRow, weightRow, hiddenSize_);
            float dot = 0.0f;
            for (uint32_t k = 0; k < hiddenSize_; ++k) {
                dot += prod(k);
            }
            // Add to outputRow[j]
            float oldVal = outputRow(j);
            outputRow(j) = oldVal + dot;

            tmpQueue_.FreeTensor(prod);
            tmpQueue_.FreeTensor(weightRow);
        }

        // Now compute logsumexp over outputRow (outputSize elements)
        // Find max
        float maxVal = -INFINITY;
        for (uint32_t j = 0; j < outputSize_; ++j) {
            float v = outputRow(j);
            if (v > maxVal) maxVal = v;
        }
        // Compute sum exp(x - max)
        float sumExp = 0.0f;
        for (uint32_t j = 0; j < outputSize_; ++j) {
            sumExp += expf(outputRow(j) - maxVal);
        }
        float logSumExp = logf(sumExp) + maxVal;

        // The output scalar for this batch is logSumExp
        float outVal = logSumExp;

        // Apply two leaky ReLU with negative_slope=0.01
        const float leakySlope = 0.01f;
        outVal = (outVal >= 0) ? outVal : leakySlope * outVal;
        outVal = (outVal >= 0) ? outVal : leakySlope * outVal;

        // Apply two GELU
        const float sqrt2 = 1.4142135623730951f;
        auto gelu = [](float x) -> float {
            return 0.5f * x * (1.0f + erff(x / sqrt2));
        };
        outVal = gelu(outVal);
        outVal = gelu(outVal);

        // Store result
        yGm_(batchIdx) = outVal;

        // Free buffers
        inQueue_.FreeTensor(inputRow);
        outQueue_.FreeTensor(outputRow);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_{0};
    uint32_t hiddenSize_{0};
    uint32_t outputSize_{0};
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom op;
    op.Init(x, w, b, y, tiling_data.batchSize, tiling_data.hiddenSize, tiling_data.outputSize);
    op.Process();
}
