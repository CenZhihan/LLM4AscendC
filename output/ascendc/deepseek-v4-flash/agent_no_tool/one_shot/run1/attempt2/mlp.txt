host_operator_src="""
#include "mlp_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 256;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t Min(uint32_t a, uint32_t b) {
    return a < b ? a : b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MlpCustomTilingData tiling;
    const auto &xShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t batch = static_cast<uint32_t>(xShape.GetDim(0));
    uint32_t inputSize = static_cast<uint32_t>(xShape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *p_hidden0 = attrs->GetAttrPointer<int32_t>(1); // hidden_0
    const int32_t *p_hidden1 = attrs->GetAttrPointer<int32_t>(2); // hidden_1
    const int32_t *p_outputSize = attrs->GetAttrPointer<int32_t>(3); // output_size
    uint32_t hidden0 = static_cast<uint32_t>(*p_hidden0);
    uint32_t hidden1 = static_cast<uint32_t>(*p_hidden1);
    uint32_t outputSize = static_cast<uint32_t>(*p_outputSize);

    // Heuristic tile sizes (fit in UB)
    uint32_t tileInner = 4096;
    if (inputSize < tileInner) tileInner = AlignDown(inputSize, ALIGN);
    if (tileInner == 0) tileInner = 1;

    uint32_t tileOuter = 2048;
    if (outputSize < tileOuter) tileOuter = AlignDown(outputSize, ALIGN);
    if (tileOuter == 0) tileOuter = 1;

    uint32_t tileBatch = 1;

    // Use single core for simplicity
    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalBatch(batch);
    tiling.set_inputSize(inputSize);
    tiling.set_hidden0(hidden0);
    tiling.set_hidden1(hidden1);
    tiling.set_outputSize(outputSize);
    tiling.set_tileBatch(tileBatch);
    tiling.set_tileOuter(tileOuter);
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // Workspace: store intermediate hidden0 and hidden1 vectors (one batch row)
    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = (hidden0 + hidden1) * sizeof(float);

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *xShape = context->GetInputShape(0);
    gert::Shape *yShape = context->GetOutputShape(0);
    *yShape = *xShape;
    const int32_t *p_outputSize = context->GetAttrs()->GetAttrPointer<int32_t>(3);
    yShape->SetDim(1, static_cast<int64_t>(*p_outputSize));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMlpCustom {
public:
    __aicore__ inline KernelMlpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed,
                                GM_ADDR y, GM_ADDR workspace,
                                uint32_t totalBatch, uint32_t inputSize,
                                uint32_t hidden0, uint32_t hidden1,
                                uint32_t outputSize,
                                uint32_t tileBatch, uint32_t tileOuter,
                                uint32_t tileInner)
    {
        totalBatch_ = totalBatch;
        inputSize_ = inputSize;
        hidden0_ = hidden0;
        hidden1_ = hidden1;
        outputSize_ = outputSize;
        tileOuter_ = tileOuter;
        tileInner_ = tileInner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalBatch * inputSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w_packed,
                             hidden0 * inputSize + hidden1 * hidden0 + outputSize * hidden1);
        bGm_.SetGlobalBuffer((__gm__ float*)b_packed, hidden0 + hidden1 + outputSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalBatch * outputSize);
        workspaceGm_.SetGlobalBuffer((__gm__ float*)workspace, hidden0 + hidden1);

        // Allocate UB buffers for tiled dot product
        pipe_.InitBuffer(inBufX_, 1, tileInner_ * sizeof(float));
        pipe_.InitBuffer(inBufW_, 1, tileInner_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t b = 0; b < totalBatch_; b++) {
            // Input row
            const float* xRow = xGm_.GetGlobalAddr() + b * inputSize_;

            // Layer 1: inputSize -> hidden0
            float* w1 = wGm_.GetGlobalAddr();
            float* b1 = bGm_.GetGlobalAddr();
            float* h0 = workspaceGm_.GetGlobalAddr();  // store at workspace offset 0
            MatVecMul(w1, xRow, b1, hidden0_, inputSize_, h0);
            ReluGlobal(h0, hidden0_);

            // Layer 2: hidden0 -> hidden1
            float* w2 = w1 + hidden0_ * inputSize_;
            float* b2 = b1 + hidden0_;
            float* h1 = workspaceGm_.GetGlobalAddr() + hidden0_;  // store after h0
            MatVecMul(w2, h0, b2, hidden1_, hidden0_, h1);
            ReluGlobal(h1, hidden1_);

            // Layer 3: hidden1 -> outputSize
            float* w3 = w2 + hidden1_ * hidden0_;
            float* b3 = b2 + hidden1_;
            float* yRow = yGm_.GetGlobalAddr() + b * outputSize_;
            MatVecMul(w3, h1, b3, outputSize_, hidden1_, yRow);
        }
    }

private:
    // Compute y = mat * vec + bias, where vec is a global array.
    // Write result to y_gm (global memory).
    __aicore__ inline void MatVecMul(const float* mat, const float* vec,
                                     const float* bias,
                                     uint32_t rows, uint32_t cols,
                                     float* y_gm)
    {
        for (uint32_t r = 0; r < rows; r++) {
            float sum = bias[r];
            for (uint32_t c = 0; c < cols; c += tileInner_) {
                uint32_t len = (cols - c) > tileInner_ ? tileInner_ : (cols - c);
                // Load vector segment
                DataCopy(inBufX_, vec + c, len);
                // Load weight row segment
                DataCopy(inBufW_, mat + r * cols + c, len);
                // Multiply and reduce sum
                Mul(tmpBuf_, inBufX_, inBufW_, len);
                sum += ReduceSum(tmpBuf_, len);
            }
            y_gm[r] = sum;
        }
    }

    __aicore__ inline void ReluGlobal(float* buf, uint32_t len)
    {
        // Apply ReLU in-place on global memory using local buffer tile
        LocalTensor<float> localTmp = tmpBuf_.GetTensor();
        for (uint32_t i = 0; i < len; i += tileInner_) {
            uint32_t blockLen = (len - i) > tileInner_ ? tileInner_ : (len - i);
            DataCopy(localTmp, buf + i, blockLen);
            Maxs(localTmp, localTmp, 0.0f, blockLen);
            DataCopy(buf + i, localTmp, blockLen);
        }
    }

private:
    TPipe pipe_;
    TBuf<TPosition::VECCALC> inBufX_;
    TBuf<TPosition::VECCALC> inBufW_;
    TBuf<TPosition::VECCALC> tmpBuf_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> workspaceGm_;

    uint32_t totalBatch_ = 0;
    uint32_t inputSize_ = 0;
    uint32_t hidden0_ = 0;
    uint32_t hidden1_ = 0;
    uint32_t outputSize_ = 0;
    uint32_t tileOuter_ = 0;
    uint32_t tileInner_ = 0;
};

extern "C" __global__ __aicore__ void mlp_custom(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed,
                                                  GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMlpCustom op;
    op.Init(x, w_packed, b_packed, y, workspace,
            tiling_data.totalBatch, tiling_data.inputSize,
            tiling_data.hidden0, tiling_data.hidden1,
            tiling_data.outputSize,
            tiling_data.tileBatch, tiling_data.tileOuter,
            tiling_data.tileInner);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'mlp_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, layer_sizes, output_size, *args, **kwargs):
        super(ModelNew, self).__init__()
        self.input_size = input_size
        self.hidden_0 = layer_sizes[0]
        self.hidden_1 = layer_sizes[1]
        self.output_size = output_size

    def forward(self, x, w_packed, b_packed):
        fn = getattr(custom_ops_lib, _FN)
        return fn(
            x, w_packed, b_packed,
            input_size=self.input_size,
            hidden_0=self.hidden_0,
            hidden_1=self.hidden_1,
            output_size=self.output_size
