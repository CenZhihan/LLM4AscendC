static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 16;
static constexpr uint32_t MAX_TILE_H = 8;
static constexpr uint32_t MAX_TILE_W = 8;

static uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Conv2dMishMishCustomTilingData tiling;

    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape &w_shape = context->GetInputShape(1)->GetOriginShape();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    tiling.set_n(x_shape.GetDim(0));
    tiling.set_cIn(x_shape.GetDim(1));
    tiling.set_hIn(x_shape.GetDim(2));
    tiling.set_wIn(x_shape.GetDim(3));
    tiling.set_k(w_shape.GetDim(0));
    tiling.set_r(w_shape.GetDim(2));
    tiling.set_s(w_shape.GetDim(3));
    uint32_t stride = attrs->GetAttrPointer<uint32_t>(0)[0];
    uint32_t padding = attrs->GetAttrPointer<uint32_t>(1)[0];
    uint32_t dilation = attrs->GetAttrPointer<uint32_t>(2)[0];
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_dilation(dilation);

    uint32_t hIn = tiling.get_hIn();
    uint32_t r = tiling.get_r();
    uint32_t hOut = (hIn + 2 * padding - dilation * (r - 1) - 1) / stride + 1;
    uint32_t wIn = tiling.get_wIn();
    uint32_t s = tiling.get_s();
    uint32_t wOut = (wIn + 2 * padding - dilation * (s - 1) - 1) / stride + 1;
    tiling.set_hOut(hOut);
    tiling.set_wOut(wOut);

    // Choose tile sizes for output spatial dims
    uint32_t bytesPerElem = 4;
    uint32_t tileH = std::min(MAX_TILE_H, hOut);
    uint32_t tileW = std::min(MAX_TILE_W, wOut);
    uint32_t cIn = tiling.get_cIn();
    // Adjust based on UB budget
    if (tileH * tileW * cIn * r * s * 4 > UB_BUDGET_BYTES * 0.4) {
        tileH = std::max(1u, tileH / 2);
        tileW = std::max(1u, tileW / 2);
    }

    tiling.set_tileH(AlignDown(tileH, ALIGN) > 0 ? AlignDown(tileH, ALIGN) : 1);
    tiling.set_tileW(AlignDown(tileW, ALIGN) > 0 ? AlignDown(tileW, ALIGN) : 1);

    uint32_t tileH_ = tiling.get_tileH();
    uint32_t tileW_ = tiling.get_tileW();
    uint32_t numTiles = ((hOut + tileH_ - 1) / tileH_) * ((wOut + tileW_ - 1) / tileW_);
    uint32_t blockDim = std::min(numTiles, 24u);
    if (blockDim < 8u) blockDim = 8u;
    tiling.set_blockDim(blockDim);

    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "conv2d_mish_mish_custom_tiling.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;
constexpr float MISH_THRESH = 20.0f;

class KernelConv2dMishMishCustom {
public:
    __aicore__ inline KernelConv2dMishMishCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                Conv2dMishMishCustomTilingData* tiling) {
        tiling_ = tiling;
        xGm_.SetGlobalBuffer((__gm__ float*)x, tiling_->n * tiling_->cIn * tiling_->hIn * tiling_->wIn);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, tiling_->k * tiling_->cIn * tiling_->r * tiling_->s);
        if (bias != nullptr) {
            hasBias_ = true;
            bGm_.SetGlobalBuffer((__gm__ float*)bias, tiling_->k);
        } else {
            hasBias_ = false;
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, tiling_->n * tiling_->k * tiling_->hOut * tiling_->wOut);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tiling_->tileH * tiling_->tileW * tiling_->cIn * tiling_->r * tiling_->s * sizeof(float));
        pipe_.InitBuffer(inQueueW_, 1, tiling_->k * tiling_->cIn * tiling_->r * tiling_->s * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tiling_->tileH * tiling_->tileW * tiling_->k * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tiling_->tileH * tiling_->tileW * tiling_->k * sizeof(float));
        pipe_.InitBuffer(biasQueue_, 1, tiling_->k * sizeof(float));
        if (hasBias_) {
            AscendC::LocalTensor<float> bLocal = biasQueue_.AllocTensor<float>();
            AscendC::DataCopy(bLocal, bGm_, tiling_->k);
            biasQueue_.EnQue(bLocal);
        }
    }

    __aicore__ inline void Process() {
        if (tiling_->n == 0 || tiling_->k == 0 || tiling_->hOut == 0 || tiling_->wOut == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t numTilesH = (tiling_->hOut + tiling_->tileH - 1) / tiling_->tileH;
        uint32_t numTilesW = (tiling_->wOut + tiling_->tileW - 1) / tiling_->tileW;
        uint32_t totalTiles = tiling_->n * numTilesH * numTilesW;
        uint32_t tilesPerBlock = (totalTiles + blockNum - 1) / blockNum;
        uint32_t startTile = blockIdx * tilesPerBlock;
        uint32_t endTile = (blockIdx + 1) * tilesPerBlock;
        if (endTile > totalTiles) endTile = totalTiles;

        for (uint32_t tileIdx = startTile; tileIdx < endTile; tileIdx++) {
            uint32_t nIdx = tileIdx / (numTilesH * numTilesW);
            uint32_t rem = tileIdx % (numTilesH * numTilesW);
            uint32_t hTileIdx = rem / numTilesW;
            uint32_t wTileIdx = rem % numTilesW;
            uint32_t hStart = hTileIdx * tiling_->tileH;
            uint32_t wStart = wTileIdx * tiling_->tileW;
            uint32_t hEnd = hStart + tiling_->tileH;
            if (hEnd > tiling_->hOut) hEnd = tiling_->hOut;
            uint32_t wEnd = wStart + tiling_->tileW;
            if (wEnd > tiling_->wOut) wEnd = tiling_->wOut;
            uint32_t tileH = hEnd - hStart;
            uint32_t tileW = wEnd - wStart;

            // im2col for this tile
            uint32_t colSize = tileH * tileW * tiling_->cIn * tiling_->r * tiling_->s;
            AscendC::LocalTensor<float> colLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_, tiling_->k * tiling_->cIn * tiling_->r * tiling_->s);

            // For each output position in tile, extract patch
            uint32_t idx = 0;
            for (uint32_t ho = hStart; ho < hEnd; ho++) {
                for (uint32_t wo = wStart; wo < wEnd; wo++) {
                    int32_t hi = static_cast<int32_t>(ho) * tiling_->stride - tiling_->padding;
                    int32_t wi = static_cast<int32_t>(wo) * tiling_->stride - tiling_->padding;
                    for (uint32_t ci = 0; ci < tiling_->cIn; ci++) {
                        for (uint32_t ri = 0; ri < tiling_->r; ri++) {
                            int32_t hIdx = hi + ri * tiling_->dilation;
                            for (uint32_t si = 0; si < tiling_->s; si++) {
                                int32_t wIdx = wi + si * tiling_->dilation;
                                float val = 0.0f;
                                if (hIdx >= 0 && hIdx < static_cast<int32_t>(tiling_->hIn) &&
                                    wIdx >= 0 && wIdx < static_cast<int32_t>(tiling_->wIn)) {
                                    uint32_t gmIdx = nIdx * tiling_->cIn * tiling_->hIn * tiling_->wIn +
                                                     ci * tiling_->hIn * tiling_->wIn +
                                                     hIdx * tiling_->wIn + wIdx;
                                    val = xGm_.GetValue(gmIdx);
                                }
                                colLocal.SetValue(idx++, val);
                            }
                        }
                    }
                }
            }
            inQueueX_.EnQue(colLocal);

            // Matmul: out = col * weight^T (col shape: [tileH*tileW, cIn*r*s], weight shape: [k, cIn*r*s])
            AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();
            AscendC::Matmul(outLocal, colLocal, wLocal, false, true, tileH*tileW, tiling_->k, tiling_->cIn * tiling_->r * tiling_->s);
            inQueueX_.FreeTensor(colLocal);
            inQueueW_.FreeTensor(wLocal);

            // Add bias if present
            if (hasBias_) {
                AscendC::LocalTensor<float> bLocal = biasQueue_.DeQue<float>();
                AscendC::Add(outLocal, outLocal, bLocal, tileH * tileW * tiling_->k);
                biasQueue_.EnQue(bLocal);
            }

            // Apply Mish x * tanh(softplus(x))
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            // softplus: log(1+exp(x)) for x<20, else x
            AscendC::Exp(tLocal, outLocal, tileH * tileW * tiling_->k);
            AscendC::Adds(tLocal, tLocal, 1.0f, tileH * tileW * tiling_->k);
            AscendC::Ln(tLocal, tLocal, tileH * tileW * tiling_->k);
            // for large values, clamp
            AscendC::Mins(tLocal, tLocal, MISH_THRESH, tileH * tileW * tiling_->k);
            AscendC::Maxs(tLocal, tLocal, -MISH_THRESH, tileH * tileW * tiling_->k);
            // tanh
            AscendC::Tanh(tLocal, tLocal, tileH * tileW * tiling_->k);
            // x * tanh(softplus(x))
            AscendC::Mul(outLocal, outLocal, tLocal, tileH * tileW * tiling_->k);
            // Second Mish (same gate)
            AscendC::Exp(tLocal, outLocal, tileH * tileW * tiling_->k);
            AscendC::Adds(tLocal, tLocal, 1.0f, tileH * tileW * tiling_->k);
            AscendC::Ln(tLocal, tLocal, tileH * tileW * tiling_->k);
            AscendC::Mins(tLocal, tLocal, MISH_THRESH, tileH * tileW * tiling_->k);
            AscendC::Maxs(tLocal, tLocal, -MISH_THRESH, tileH * tileW * tiling_->k);
            AscendC::Tanh(tLocal, tLocal, tileH * tileW * tiling_->k);
            AscendC::Mul(outLocal, outLocal, tLocal, tileH * tileW * tiling_->k);
            tmpQueue_.FreeTensor(tLocal);

            outQueueY_.EnQue(outLocal);

            // Copy out
            AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
            // Write to proper locations
            uint32_t offset = nIdx * tiling_->k * tiling_->hOut * tiling_->wOut +
                              hStart * tiling_->wOut * tiling_->k + wStart * tiling_->k;
            // For each row in tile
            for (uint32_t h = 0; h < tileH; h++) {
                uint32_t rowOffset = offset + h * tiling_->wOut * tiling_->k;
                AscendC::DataCopy(yGm_[rowOffset], yLocal[h * tileW * tiling_->k], tileW * tiling_->k);
            }
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> biasQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    Conv2dMishMishCustomTilingData* tiling_;
    bool hasBias_ = false;
};

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMishMishCustom op;
    op.Init(x, weight, bias, y, &tiling_data);
    op.Process();
}
