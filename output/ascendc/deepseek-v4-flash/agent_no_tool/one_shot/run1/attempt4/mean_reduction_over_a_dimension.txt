#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const auto* storageShape = context->GetInputShape(0);
    const auto& originShape = storageShape->GetOriginShape();
    uint32_t ndim = originShape.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;
    if (dim < 0) dim += static_cast<int64_t>(ndim);
    if (dim < 0 || dim >= static_cast<int64_t>(ndim)) {
        return ge::GRAPH_FAILED;
    }

    uint32_t reductionLength = static_cast<uint32_t>(originShape.GetDim(static_cast<uint32_t>(dim)));
    uint32_t outerSize = 1;
    for (int64_t i = 0; i < dim; ++i) {
        outerSize *= static_cast<uint32_t>(originShape.GetDim(static_cast<uint32_t>(i)));
    }
    uint32_t innerSize = 1;
    for (size_t i = static_cast<size_t>(dim) + 1; i < ndim; ++i) {
        innerSize *= static_cast<uint32_t>(originShape.GetDim(i));
    }

    uint32_t totalOutputElems = outerSize * innerSize;

    // UB budget: one slice of reductionLength floats
    uint32_t sliceBytes = reductionLength * sizeof(float);
    if (sliceBytes > UB_BUDGET_BYTES) {
        return ge::GRAPH_FAILED;
    }

    uint32_t blockDim = CeilDiv(totalOutputElems, 8u);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_reductionLength(reductionLength);
    tiling.set_innerSize(innerSize);
    tiling.set_outerSize(outerSize);
    tiling.set_blockDim(blockDim);
    tiling.set_invReductionLength(1.0f / static_cast<float>(reductionLength));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t ndim = x_shape->GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = *attrs->GetAttrPointer<int64_t>(0);
    if (dim < 0) dim += static_cast<int64_t>(ndim);
    if (dim < 0 || dim >= static_cast<int64_t>(ndim)) return GRAPH_FAILED;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(ndim - 1);
    for (uint32_t i = 0, j = 0; i < ndim; ++i) {
        if (static_cast<int64_t>(i) == dim) continue;
        y_shape->SetDim(j++, x_shape->GetDim(i));
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElems,
                                uint32_t reductionLength,
                                uint32_t innerSize,
                                uint32_t outerSize,
                                uint32_t blockDim,
                                float invReductionLength)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        totalOutputElems_ = totalOutputElems;
        reductionLength_ = reductionLength;
        innerSize_ = innerSize;
        outerSize_ = outerSize;
        blockDim_ = blockDim;
        invReductionLength_ = invReductionLength;

        // Single buffer for the slice (reductionLength floats)
        pipe_.InitBuffer(sliceQue_, 1, reductionLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());
        uint32_t blockNum = blockDim_;

        uint32_t elemsPerBlock = (totalOutputElems_ + blockNum - 1) / blockNum;
        uint32_t startIdx = blockIdx * elemsPerBlock;
        uint32_t endIdx = startIdx + elemsPerBlock;
        if (endIdx > totalOutputElems_) endIdx = totalOutputElems_;

        for (uint32_t idx = startIdx; idx < endIdx; ++idx) {
            uint32_t outerIdx = idx / innerSize_;
            uint32_t innerIdx = idx % innerSize_;

            uint32_t baseOffset = outerIdx * reductionLength_ * innerSize_ + innerIdx;
            uint32_t stride = innerSize_;

            // Allocate local tensor
            LocalTensor<float> sliceLocal = sliceQue_.AllocTensor<float>();

            // Strided copy: blockCount = reductionLength_, blockLen = 1,
            // srcStride = stride (gap between consecutive elements in global),
            // dstStride = 1 (contiguous in local)
            DataCopyParams params = {reductionLength_, 1, stride, 1};
            DataCopy(sliceLocal, xGm_[baseOffset], params);

            // Compute sum using a loop (vectorized by compiler)
            float sum = 0.0f;
            for (uint32_t k = 0; k < reductionLength_; ++k) {
                sum += sliceLocal(k);
            }

            float mean = sum * invReductionLength_;

            // Write output
            yGm_.SetValue(idx, mean);

            // Free local tensor
            sliceQue_.FreeTensor(sliceLocal);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, 1> sliceQue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalOutputElems_ {0};
    uint32_t reductionLength_ {0};
    uint32_t innerSize_ {0};
    uint32_t outerSize_ {0};
    uint32_t blockDim_ {0};
    float invReductionLength_ {0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalOutputElems,
            tiling_data.reductionLength,
            tiling_data.innerSize,
            tiling_data.outerSize,
            tiling_data.blockDim,
            tiling_data.invReductionLength);
    op.Process();
}
