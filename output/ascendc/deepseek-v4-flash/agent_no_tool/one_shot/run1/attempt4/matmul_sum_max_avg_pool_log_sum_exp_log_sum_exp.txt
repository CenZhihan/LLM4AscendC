#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom {
public:
    __aicore__ inline KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalBatches, uint32_t inFeatures, uint32_t outFeatures)
    {
        batchTotal_ = totalBatches;
        inFeat_ = inFeatures;
        outFeat_ = outFeatures;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchTotal_ * inFeat_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, outFeat_ * inFeat_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, outFeat_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchTotal_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inFeat_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, inFeat_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, inFeat_ * sizeof(float));
        pipe_.InitBuffer(wRowSumBuf_, 1, inFeat_ * sizeof(float));
        pipe_.InitBuffer(bSumBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (inFeat_ == 0 || outFeat_ == 0 || batchTotal_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Step 1: Compute row sum of W and sum of b
        AscendC::LocalTensor<float> wRowSum = wRowSumBuf_.Get<float>();
        AscendC::Duplicate(wRowSum, 0.0f, inFeat_);
        AscendC::LocalTensor<float> bSumLocal = bSumBuf_.Get<float>();
        AscendC::Duplicate(bSumLocal, 0.0f, 1);

        for (uint32_t j = 0; j < outFeat_; ++j) {
            // Load one row of W
            AscendC::LocalTensor<float> wTile = inQueueW_.AllocTensor<float>();
            AscendC::DataCopy(wTile, wGm_[j * inFeat_], inFeat_);
            inQueueW_.EnQue(wTile);
            wTile = inQueueW_.DeQue<float>();
            // Accumulate row sum
            AscendC::Add(wRowSum, wRowSum, wTile, inFeat_);
            inQueueW_.FreeTensor(wTile);

            // Load b[j]
            AscendC::LocalTensor<float> bTile = inQueueW_.AllocTensor<float>();
            AscendC::DataCopy(bTile, bGm_[j], 1);
            float bVal = bTile.GetValue(0);
            bSumLocal.SetValue(0, bSumLocal.GetValue(0) + bVal);
            inQueueW_.FreeTensor(bTile);
        }

        // Step 2: Process batches in grid-stride loop
        uint32_t batchStart = blockIdx;
        while (batchStart < batchTotal_) {
            // Load x[batchStart] into local
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[batchStart * inFeat_], inFeat_);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // Compute dot product: sum(xLocal * wRowSum)
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(tmp, xLocal, wRowSum, inFeat_);
            float dot = 0.0f;
            for (uint32_t k = 0; k < inFeat_; ++k) {
                dot += tmp.GetValue(k);
            }
            tmpQueue_.FreeTensor(tmp);
            float outVal = dot + bSumLocal.GetValue(0);

            // Write output
            yGm_.SetValue(batchStart, outVal);

            inQueueX_.FreeTensor(xLocal);
            batchStart += blockNum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> wRowSumBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bSumBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchTotal_ {0};
    uint32_t inFeat_ {0};
    uint32_t outFeat_ {0};
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{

    KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom op;
    op.Init(x, w, b, y, tiling_data.totalBatches, tiling_data.inFeatures, tiling_data.outFeatures);
    op.Process();
}
