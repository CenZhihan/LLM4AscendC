kernel_src="""

#include "kernel_operator.h"
#include "matmul_api.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmScalingHardTanhGelu {
public:
    __aicore__ inline KernelGemmScalingHardTanhGelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t blockSizeM, uint32_t blockSizeN, uint32_t blockSizeK,
                                float scaling, float hardtanh_min, float hardtanh_max)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        blockSizeM_ = blockSizeM;
        blockSizeN_ = blockSizeN;
        blockSizeK_ = blockSizeK;
        scaling_ = scaling;
        hardtanhMin_ = hardtanh_min;
        hardtanhMax_ = hardtanh_max;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M_ * N_);

        // Initialize matmul object
        matmul_.SetOrgShape(M_, N_, K_);
        matmul_.SetBlockShape(blockSizeM_, blockSizeN_, blockSizeK_);
        matmul_.SetA(xGm_);
        matmul_.SetB(wGm_);
        matmul_.SetBias(bGm_);

        // Allocate local tensor for intermediate result
        pipe_.InitBuffer(tmpMat_ , 1, blockSizeM_ * blockSizeN_ * sizeof(float));
        pipe_.InitBuffer(tmpAct_ , 1, blockSizeM_ * blockSizeN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        // Each block processes a slice of M and N dimensions
        uint32_t blocksM = (M_ + blockSizeM_ - 1) / blockSizeM_;
        uint32_t blocksN = (N_ + blockSizeN_ - 1) / blockSizeN_;
        uint32_t totalBlocks = blocksM * blocksN;

        for (uint32_t blockId = blockIdx; blockId < totalBlocks; blockId += blockNum) {
            uint32_t blockIdxM = blockId / blocksN;
            uint32_t blockIdxN = blockId % blocksN;

            uint32_t curM = std::min(blockSizeM_, M_ - blockIdxM * blockSizeM_);
            uint32_t curN = std::min(blockSizeN_, N_ - blockIdxN * blockSizeN_);

            // MatMul: C = A * B^T + bias (with transposition handled internally)
            matmul_.SetCallPos(blockIdxM, blockIdxN);
            matmul_.Process(tmpMat_, 1.0f, 0.0f);

            // Apply scaling, HardTanh, GELU on the block
            ApplyActivations(tmpMat_, curM, curN);

            // Copy result to global memory
            uint32_t gmOffset = blockIdxM * blockSizeM_ * N_ + blockIdxN * blockSizeN_;
            DataCopy(yGm_[gmOffset], tmpMat_, curM * curN);
        }
    }

private:
    __aicore__ inline void ApplyActivations(LocalTensor<float>& tensor, uint32_t rows, uint32_t cols)
    {
        uint32_t len = rows * cols;
        // scaling
        Muls(tensor, tensor, scaling_, len);
        // hardtanh: clamp to [hardtanhMin, hardtanhMax]
        // We'll use Max and Min: x = max(min(x, hardtanhMax), hardtanhMin)
        LocalTensor<float> tmp = tmpAct_.Get<float>();
        Maxs(tmp, tensor, hardtanhMin_, len);
        Mins(tensor, tmp, hardtanhMax_, len);
        // GELU approximate: x * 0.5 * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
        // We'll use the exact GELU: x * 0.5 * (1 + erf(x / sqrt(2)))
        // Ascend C has Erf intrinsic? There's Exp and but not Erf directly.
        // Use the tanh approximation for simplicity.
        // Let phi = 0.79788456f (sqrt(2/pi))
        // Compute x3 = x * x * x
        // tanh_input = phi * (x + 0.044715f * x3)
        // result = 0.5f * x * (1.0f + tanh(tanh_input))
        // We'll use vector tanh via Exp: tanh(z) = (exp(2z)-1)/(exp(2z)+1)
        // But lack of Exp? Ascend C provides Exp. Use that.
        // For simplicity and to avoid missing intrinsics, we implement the piecewise approximation:
        // GELU ≈ x * sigmoid(1.702f * x) (fast). But to be accurate, we use the erf-based.
        // We'll compute using Taylor? Better to use precomputed constants.
        // Since Ascend C has Exp, we can compute tanh.
        // However, to keep code concise, we use the piecewise linear? No, use the standard GELU formula with Erf.
        // Ascend C has no Erf, but we can implement with series or use the tanh approximation.
        // Let's implement the tanh approximation for GELU.
        const float sqrt2pi = 0.7978845608028654f; // sqrt(2/pi)
        const float coeff = 0.044715f;

        // Compute x^3
        LocalTensor<float> x3 = tmp; // reuse
        Mul(x3, tensor, tensor, len);
        Mul(x3, x3, tensor, len);

        // tanh_input = sqrt2pi * (x + coeff * x^3)
        LocalTensor<float> tanh_input = tmpAct_.Get<float>(); // reuse different buffer
        Madd(tanh_input, tensor, coeff, x3, len);
        Muls(tanh_input, tanh_input, sqrt2pi, len);

        // Compute tanh using: tanh(z) = (exp(2z)-1)/(exp(2z)+1)
        LocalTensor<float> exp2z = tmp;
        Muls(exp2z, tanh_input, 2.0f, len);
        Exp(exp2z, exp2z, len);  // e^(2z)
        LocalTensor<float> tanh_val = tmpAct_.Get<float>();
        // tanh_val = (exp2z - 1) / (exp2z + 1)
        Sub(tanh_val, exp2z, 1.0f, len);
        Add(exp2z, exp2z, 1.0f, len);
        Div(tanh_val, tanh_val, exp2z, len);

        // gelu = 0.5 * x * (1 + tanh_val)
        Add(tanh_val, tanh_val, 1.0f, len);
        Mul(tensor, tensor, tanh_val, len);
        Muls(tensor, tensor, 0.5f, len);
    }

    // Custom Madd: out = a + coeff * b
    __aicore__ inline void Madd(LocalTensor<float>& dst, LocalTensor<float>& a, float coeff, LocalTensor<float>& b, uint32_t len)
    {
        LocalTensor<float> tmp = tmpAct_.Get<float>();
        Muls(tmp, b, coeff, len);
        Add(dst, a, tmp, len);
    }

private:
    uint32_t M_, N_, K_;
    uint32_t blockSizeM_, blockSizeN_, blockSizeK_;
    float scaling_, hardtanhMin_, hardtanhMax_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    MatMul<MatMulType<ROUND_ROUND, ND_ND, ND_BD>> matmul_;

    TPipe pipe_;
    TBuf<TPosition::VECCALC> tmpMat_;   // for matmul output block
    TBuf<TPosition::VECCALC> tmpAct_;   // temporary for activations
};

extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScalingHardTanhGelu op;
    op.Init(x, w, b, y, tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.blockSizeM, tiling_data.blockSizeN, tiling_data.blockSizeK,
            tiling_data.scaling, tiling_data.hardtanh_min, tiling_data.hardtanh_max);
    op.Process();
}
