kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t batch, uint32_t inChannels, uint32_t inHeight, uint32_t inWidth,
                                uint32_t outChannels, uint32_t outHeight, uint32_t outWidth,
                                uint32_t kernelH, uint32_t kernelW,
                                uint32_t tileOutChannels, uint32_t tileOutHeight, uint32_t tileOutWidth)
    {
        batch_ = batch;
        inChannels_ = inChannels;
        inHeight_ = inHeight;
        inWidth_ = inWidth;
        outChannels_ = outChannels;
        outHeight_ = outHeight;
        outWidth_ = outWidth;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        tileOutChannels_ = tileOutChannels;
        tileOutHeight_ = tileOutHeight;
        tileOutWidth_ = tileOutWidth;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_ * inChannels_ * inHeight_ * inWidth_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outChannels_ * inChannels_ * kernelH_ * kernelW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch_ * outChannels_ * outHeight_ * outWidth_);

        uint32_t patchSize = inChannels_ * (tileOutHeight_ + kernelH_ - 1) * (tileOutWidth_ + kernelW_ - 1);
        uint32_t weightSize = tileOutChannels_ * inChannels_ * kernelH_ * kernelW_;
        uint32_t outSize = tileOutChannels_ * tileOutHeight_ * tileOutWidth_;

        pipe_.InitBuffer(inQueuePatch_, BUFFER_NUM, patchSize * sizeof(float));
        pipe_.InitBuffer(inQueueWeight_, BUFFER_NUM, weightSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, max(patchSize, max(weightSize, outSize)) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || outHeight_ == 0 || outWidth_ == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        uint32_t totalTilesPerBatch = CeilDiv(outChannels_, tileOutChannels_) *
                                      CeilDiv(outHeight_, tileOutHeight_) *
                                      CeilDiv(outWidth_, tileOutWidth_);
        uint32_t totalTiles = batch_ * totalTilesPerBatch;
        uint32_t tileStride = blockNum;
        uint32_t tileStart = blockIdx;

        for (uint32_t tIdx = tileStart; tIdx < totalTiles; tIdx += tileStride) {
            uint32_t tileFlat = tIdx % totalTilesPerBatch;
            uint32_t n = tIdx / totalTilesPerBatch;

            uint32_t tileChannels = CeilDiv(outChannels_, tileOutChannels_);
            uint32_t tileHeight = CeilDiv(outHeight_, tileOutHeight_);
            uint32_t tileWidth = CeilDiv(outWidth_, tileOutWidth_);

            uint32_t tC = tileFlat / (tileHeight * tileWidth);
            uint32_t tH = (tileFlat % (tileHeight * tileWidth)) / tileWidth;
            uint32_t tW = (tileFlat % (tileHeight * tileWidth)) % tileWidth;

            uint32_t ocStart = tC * tileOutChannels_;
            uint32_t ohStart = tH * tileOutHeight_;
            uint32_t owStart = tW * tileOutWidth_;

            uint32_t ocLen = min(tileOutChannels_, outChannels_ - ocStart);
            uint32_t ohLen = min(tileOutHeight_, outHeight_ - ohStart);
            uint32_t owLen = min(tileOutWidth_, outWidth_ - owStart);

            uint32_t ihStart = ohStart;
            uint32_t iwStart = owStart;
            uint32_t ihEnd = min(ihStart + ohLen + kernelH_ - 1, inHeight_);
            uint32_t iwEnd = min(iwStart + owLen + kernelW_ - 1, inWidth_);
            uint32_t ihLen = ihEnd - ihStart;
            uint32_t iwLen = iwEnd - iwStart;

            LocalTensor<float> patchLocal = inQueuePatch_.AllocTensor<float>();
            uint32_t patchBytes = inChannels_ * ihLen * iwLen * sizeof(float);
            uint32_t inOffset = n * inChannels_ * inHeight_ * inWidth_ +
                                inChannels_ * (ihStart * inWidth_ + iwStart);
            DataCopy(patchLocal, xGm_[inOffset], patchBytes / sizeof(float));

            LocalTensor<float> weightLocal = inQueueWeight_.AllocTensor<float>();
            uint32_t weightBytes = ocLen * inChannels_ * kernelH_ * kernelW_ * sizeof(float);
            uint32_t wOffset = ocStart * inChannels_ * kernelH_ * kernelW_;
            DataCopy(weightLocal, wGm_[wOffset], weightBytes / sizeof(float));

            Conv2D<float, float, float> conv;
            Conv2DParam convParam;
            convParam.strideH = 1;
            convParam.strideW = 1;
            convParam.padH = 0;
            convParam.padW = 0;
            convParam.dilationH = 1;
            convParam.dilationW = 1;
            conv.SetConvParam(convParam);

            LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();
            LocalTensor<float> workspaceLocal = tmpQueue_.AllocTensor<float>();
            conv.Process(patchLocal, weightLocal, outLocal,
                         {1, inChannels_, ihLen, iwLen},
                         {ocLen, inChannels_, kernelH_, kernelW_},
                         workspaceLocal, TPosition::VECCALC);
            tmpQueue_.FreeTensor(workspaceLocal);

            uint32_t outOffset = n * outChannels_ * outHeight_ * outWidth_ +
                                 ocStart * outHeight_ * outWidth_ +
                                 ohStart * outWidth_ + owStart;
            DataCopy(yGm_[outOffset], outLocal, ocLen * ohLen * owLen);

            outQueueY_.FreeTensor(outLocal);
            inQueueWeight_.FreeTensor(weightLocal);
            inQueuePatch_.FreeTensor(patchLocal);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueuePatch_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueWeight_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t inChannels_;
    uint32_t inHeight_;
    uint32_t inWidth_;
    uint32_t outChannels_;
    uint32_t outHeight_;
    uint32_t outWidth_;
    uint32_t kernelH_;
    uint32_t kernelW_;
    uint32_t tileOutChannels_;
    uint32_t tileOutHeight_;
    uint32_t tileOutWidth_;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.batch, tiling_data.inChannels, tiling_data.inHeight, tiling_data.inWidth,
            tiling_data.outChannels, tiling_data.outHeight, tiling_data.outWidth,
            tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.tileOutChannels, tiling_data.tileOutHeight, tiling_data.tileOutWidth);
    op.Process();
}
