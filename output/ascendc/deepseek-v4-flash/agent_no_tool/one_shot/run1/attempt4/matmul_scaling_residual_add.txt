    uint32_t tileN = (N > MAX_TILE_N) ? MAX_TILE_N : N;
    uint32_t tileK = (K > MAX_TILE_K) ? MAX_TILE_K : K;

    uint32_t tilesN = CeilDiv(N, tileN);

    tiling.set_N(N);

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulScalingResidualAdd {
public:
    __aicore__ inline KernelMatmulScalingResidualAdd() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                float scalingFactor)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN; tileK_ = tileK;
        scalingFactor_ = scalingFactor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M * K);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N * K);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M * N);

        matmul_.SetTensorA(xGm_);
        matmul_.SetTensorB(wGm_);
        matmul_.SetBias(bGm_);
        matmul_.SetResult(yGm_);
        matmul_.SetCubeDim(M, N, K);
        matmul_.SetTile(tileM_, tileN_, tileK_);

        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueueTmp_, 1, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        matmul_.IterateAll();

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        uint32_t totalElems = M_ * N_;
        uint32_t perBlock = (totalElems + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * perBlock;
        uint32_t end = (start + perBlock > totalElems) ? totalElems : start + perBlock;
        uint32_t len = end - start;

        if (len == 0) return;

        LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        LocalTensor<float> tmpLocal = outQueueTmp_.AllocTensor<float>();

        DataCopy(yLocal, yGm_[start], len);
        Muls(tmpLocal, yLocal, scalingFactor_, len);
        DataCopy(yGm_[start], tmpLocal, len);

        inQueueY_.FreeTensor(yLocal);
        outQueueTmp_.FreeTensor(tmpLocal);
    }

private:
    // Corrected MatMul declaration: use direct position enums, no MatMulType
    MatMul<TPosition::CUBE_MATRIX_A, TPosition::CUBE_MATRIX_B, TPosition::CUBE_BIAS, TPosition::CUBE_RESULT> matmul_;

    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueY_;
    TQue<TPosition::VECOUT, 1> outQueueTmp_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t M_, N_, K_;
    uint32_t tileM_, tileN_, tileK_;
    float scalingFactor_;
};

extern "C" __global__ __aicore__ void matmul_scaling_residual_add_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling,
                                                                         GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScalingResidualAdd op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.scalingFactor);
    op.Process();
}
