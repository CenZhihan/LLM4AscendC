static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard1dDilatedStridedCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t stride = attrs->GetAttrPointer<int32_t>(0)[0];
    int32_t dilation = attrs->GetAttrPointer<int32_t>(1)[0];
    int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];
    int32_t groups = attrs->GetAttrPointer<int32_t>(3)[0];

    // Use StorageShape pointer (correct for AscendC API)
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::StorageShape *wShape = context->GetInputShape(1);

    uint32_t batch = static_cast<uint32_t>(xShape->GetDim(0));
    uint32_t inChannels = static_cast<uint32_t>(xShape->GetDim(1));
    uint32_t inLength = static_cast<uint32_t>(xShape->GetDim(2));

    uint32_t outChannels = static_cast<uint32_t>(wShape->GetDim(0));
    uint32_t kernelSize = static_cast<uint32_t>(wShape->GetDim(2));

    uint32_t outLength = (inLength + 2*padding - dilation*(kernelSize-1) - 1) / stride + 1;
    if (padding == 0) {
        outLength = (inLength - dilation*(kernelSize-1) - 1) / stride + 1;
    }

    tiling.set_batch(batch);
    tiling.set_inChannels(inChannels);
    tiling.set_outChannels(outChannels);
    tiling.set_inLength(inLength);
    tiling.set_outLength(outLength);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(static_cast<uint32_t>(stride));
    tiling.set_dilation(static_cast<uint32_t>(dilation));
    tiling.set_padding(static_cast<uint32_t>(padding));

    uint32_t weightBytes = outChannels * inChannels * kernelSize * sizeof(float);
    uint32_t availBytes = UB_BUDGET_BYTES - weightBytes - 2*1024;
    uint32_t outPositions = availBytes / (outChannels * sizeof(float) + inChannels * kernelSize * sizeof(float));
    if (outPositions == 0) outPositions = 1;
    if (outPositions > 256) outPositions = 256;

    tiling.set_tileOut(outPositions);

    uint32_t blockDim = batch * (outLength / outPositions + 1);
    if (blockDim > 24) blockDim = 24;
    if (blockDim < 1) blockDim = 1;
    tiling.set_blockDim(blockDim);
    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv1dDilatedStrided {
public:
    __aicore__ inline KernelConv1dDilatedStrided() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t batch, uint32_t inCh, uint32_t outCh,
                                uint32_t inLen, uint32_t outLen,
                                uint32_t kern, uint32_t stride, uint32_t dilation,
                                uint32_t tileOut)
    {
        batch_ = batch;
        inCh_ = inCh;
        outCh_ = outCh;
        inLen_ = inLen;
        outLen_ = outLen;
        kern_ = kern;
        stride_ = stride;
        dilation_ = dilation;
        tileOut_ = tileOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, 0);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, 0);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 0);

        pipe_.InitBuffer(weightQueue_, 1, outCh_ * inCh_ * kern_ * sizeof(float));
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileOut_ * inCh_ * kern_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOut_ * outCh_ * sizeof(float));

        AscendC::LocalTensor<float> wLocal = weightQueue_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, outCh_ * inCh_ * kern_);
        weightQueue_.EnQue(wLocal);
    }

    __aicore__ inline void Process()
    {
        AscendC::LocalTensor<float> w = weightQueue_.DeQue<float>();

        for (uint32_t n = 0; n < batch_; ++n) {
            uint32_t xBatchOff = n * inCh_ * inLen_;
            uint32_t yBatchOff = n * outCh_ * outLen_;

            for (uint32_t outStart = 0; outStart < outLen_; outStart += tileOut_) {
                uint32_t curTileOut = outLen_ - outStart;
                if (curTileOut > tileOut_) curTileOut = tileOut_;

                AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();

                for (uint32_t i = 0; i < curTileOut; ++i) {
                    uint32_t outPos = outStart + i;
                    uint32_t inStart = outPos * stride_;
                    for (uint32_t oc = 0; oc < outCh_; ++oc) {
                        float sum = 0.0f;
                        for (uint32_t k = 0; k < kern_; ++k) {
                            uint32_t inIdx = inStart + k * dilation_;
                            if (inIdx < inLen_) {
                                for (uint32_t ic = 0; ic < inCh_; ++ic) {
                                    uint32_t wIdx = ((oc * inCh_ + ic) * kern_ + k);
                                    uint32_t xIdx = xBatchOff + ic * inLen_ + inIdx;
                                    sum += w.GetValue(wIdx) * xGm_.GetValue(xIdx);
                                }
                            }
                        }
                        uint32_t yIdx = i * outCh_ + oc;
                        yTile.SetValue(yIdx, sum);
                    }
                }

                AscendC::DataCopy(yGm_[yBatchOff + outStart * outCh_], yTile, curTileOut * outCh_);
                inQueueX_.FreeTensor(xTile);
                outQueueY_.FreeTensor(yTile);
            }
        }

        weightQueue_.FreeTensor(w);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t inCh_;
    uint32_t outCh_;
    uint32_t inLen_;
    uint32_t outLen_;
    uint32_t kern_;
    uint32_t stride_;
    uint32_t dilation_;
    uint32_t tileOut_;
};

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv1dDilatedStrided op;
    op.Init(x, weight, y,
            tiling_data.batch, tiling_data.inChannels, tiling_data.outChannels,
            tiling_data.inLength, tiling_data.outLength,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.dilation,
            tiling_data.tileOut);
    op.Process();
}
