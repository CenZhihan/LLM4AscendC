    uint32_t H = x_shape->GetDim(3);

    uint32_t Cout = w_shape->GetDim(0);

#include "kernel_operator.h"

class KernelConvStandard3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR tiling)
    {
        GET_TILING_DATA(td, tiling);
        N_ = td.N;
        C_ = td.C;
        D_ = td.D;
        H_ = td.H;
        W_ = td.W;
        Cout_ = td.Cout;
        K_ = td.K;
        Dout_ = td.Dout;
        Hout_ = td.Hout;
        Wout_ = td.Wout;
        totalLen_ = td.totalLength;
        tileLen_ = td.tileLength;

        inStrideN_ = C_ * D_ * H_ * W_;
        inStrideC_ = D_ * H_ * W_;
        inStrideD_ = H_ * W_;
        inStrideH_ = W_;
        inStrideW_ = 1;

        outStrideN_ = Cout_ * Dout_ * Hout_ * Wout_;
        outStrideCout_ = Dout_ * Hout_ * Wout_;
        outStrideD_ = Hout_ * Wout_;
        outStrideH_ = Wout_;
        outStrideW_ = 1;

        weightStrideCout_ = C_ * K_ * K_ * K_;
        weightStrideC_ = K_ * K_ * K_;
        weightStrideKd_ = K_ * K_;
        weightStrideKh_ = K_;
        weightStrideKw_ = 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(weightLocalBuf_, Cout_ * weightStrideCout_ * sizeof(float));
        weightLocal_ = weightLocalBuf_.Get<float>();

        AscendC::DataCopy(weightLocal_, wGm_, Cout_ * weightStrideCout_);

        pipe_.InitBuffer(inScalarBuf_, 1 * sizeof(float));
        pipe_.InitBuffer(outScalarBuf_, 1 * sizeof(float));
        inScalar_ = inScalarBuf_.Get<float>();
        outScalar_ = outScalarBuf_.Get<float>();
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t outBase = blockIdx * tileLen_; outBase < totalLen_; outBase += blockNum * tileLen_) {
            uint32_t curTileLen = (outBase + tileLen_ > totalLen_) ? (totalLen_ - outBase) : tileLen_;
            uint32_t end = outBase + curTileLen;

            for (uint32_t outIdx = outBase; outIdx < end; ++outIdx) {
                uint32_t n = outIdx / outStrideN_;
                uint32_t rem = outIdx % outStrideN_;
                uint32_t cout = rem / outStrideCout_;
                rem = rem % outStrideCout_;
                uint32_t d = rem / outStrideD_;
                rem = rem % outStrideD_;
                uint32_t h = rem / outStrideH_;
                uint32_t w = rem % outStrideH_;

                float sum = 0.0f;
                for (uint32_t cin = 0; cin < C_; ++cin) {
                    for (uint32_t kd = 0; kd < K_; ++kd) {
                        for (uint32_t kh = 0; kh < K_; ++kh) {
                            for (uint32_t kw = 0; kw < K_; ++kw) {
                                uint32_t inIdx = n * inStrideN_ +
                                                 cin * inStrideC_ +
                                                 (d + kd) * inStrideD_ +
                                                 (h + kh) * inStrideH_ +
                                                 (w + kw) * inStrideW_;

                                uint32_t wIdx = cout * weightStrideCout_ +
                                                cin * weightStrideC_ +
                                                kd * weightStrideKd_ +
                                                kh * weightStrideKh_ +
                                                kw * weightStrideKw_;

                                AscendC::DataCopy(inScalar_, xGm_[inIdx], 1);
                                float x_val = inScalar_.GetValue(0);
                                float w_val = weightLocal_.GetValue(wIdx);
                                sum += x_val * w_val;
                            }
                        }
                    }
                }
                outScalar_.SetValue(0, sum);
                AscendC::DataCopy(yGm_[outIdx], outScalar_, 1);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> weightLocalBuf_;
    AscendC::LocalTensor<float> weightLocal_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> inScalarBuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> outScalarBuf_;
    AscendC::LocalTensor<float> inScalar_;
    AscendC::LocalTensor<float> outScalar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, D_, H_, W_, Cout_, K_;
    uint32_t Dout_, Hout_, Wout_;
    uint32_t totalLen_, tileLen_;

    uint32_t inStrideN_, inStrideC_, inStrideD_, inStrideH_, inStrideW_;
    uint32_t outStrideN_, outStrideCout_, outStrideD_, outStrideH_, outStrideW_;
    uint32_t weightStrideCout_, weightStrideC_, weightStrideKd_, weightStrideKh_, weightStrideKw_;
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y, tiling);
    op.Process();
}
