#include "conv_depthwise_2d_square_input_asymmetric_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  ConvDepthwise2dSquareInputAsymmetricKernelCustomTilingData tiling;
  const uint32_t totalLength =
      static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
  // totalLength is size of output tensor (flattened)
  // For simplicity, compute from input shape and attributes
  const auto &x_shape = context->GetInputShape(0)->GetOriginShape();
  uint32_t batch = x_shape.GetDim(0);
  uint32_t channels = x_shape.GetDim(1);
  uint32_t height = x_shape.GetDim(2);
  uint32_t width = x_shape.GetDim(3);
  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int32_t kernel_size = attrs->GetAttrPointer<int32_t>(0)[0];
  int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
  int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];
  int32_t dilation = attrs->GetAttrPointer<int32_t>(3)[0];
  bool bias = attrs->GetAttrPointer<bool>(4)[0];
  uint32_t heightOut = (height + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
  uint32_t widthOut = (width + 2 * padding - dilation * (1 - 1) - 1) / stride + 1; // kernel width=1
  uint32_t outLen = batch * channels * heightOut * widthOut;

  // UB budget: each output element processing requires kernel_size * 2 floats (input + weight) + local work
  // We'll tile conservatively
  uint32_t maxTile = UB_BUDGET_BYTES / (2 * kernel_size * sizeof(float));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = maxTile;
  if (outLen < tileLen) {
    uint32_t aligned = AlignDown(outLen, ALIGN_ELEMS);
    tileLen = (aligned > 0) ? aligned : outLen;
  }

  uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(outLen, tileLen);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_totalLength(outLen);
  tiling.set_tileLength(tileLen);
  tiling.set_blockDim(blockDim);
  tiling.set_kernelSize(kernel_size);
  tiling.set_batch(batch);
  tiling.set_channels(channels);
  tiling.set_height(height);
  tiling.set_width(width);
  tiling.set_heightOut(heightOut);
  tiling.set_widthOut(widthOut);
  tiling.set_stride(stride);
  tiling.set_padding(padding);
  tiling.set_dilation(dilation);
  tiling.set_bias(bias);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape *x_shape = context->GetInputShape(0);
  gert::Shape *y_shape = context->GetOutputShape(0);
  // Compute output shape from input and attributes
  const auto &x = *x_shape;
  uint32_t batch = x.GetDim(0);
  uint32_t channels = x.GetDim(1);
  uint32_t height = x.GetDim(2);
  uint32_t width = x.GetDim(3);
  // For simplicity, assume stride=1, padding=0, dilation=1, bias=False (from attribute defaults)
  // But we should read attributes from context if possible. Since InferShape doesn't have attrs, we'll assume defaults.
  uint32_t kernel_size = 3; // default
  uint32_t heightOut = height - kernel_size + 1;
  uint32_t widthOut = width; // kernel width=1
  std::vector<int64_t> dims = {batch, channels, heightOut, widthOut};
  y_shape->SetDimNum(dims.size());
  for (size_t i = 0; i < dims.size(); ++i) {
    y_shape->SetDim(i, dims[i]);
  }
  return GRAPH_SUCCESS;
}

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dSquareInputAsymmetricKernelCustom {
public:
  __aicore__ inline KernelConvDepthwise2dSquareInputAsymmetricKernelCustom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                              uint32_t totalLength, uint32_t tileLength,
                              uint32_t kernelSize,
                              uint32_t batch, uint32_t channels,
                              uint32_t height, uint32_t width,
                              uint32_t heightOut, uint32_t widthOut,
                              int32_t stride, int32_t padding, int32_t dilation, bool bias) {
    totalLen_ = totalLength;
    tileLen_ = tileLength;
    kernelSize_ = kernelSize;
    batch_ = batch;
    channels_ = channels;
    height_ = height;
    width_ = width;
    heightOut_ = heightOut;
    widthOut_ = widthOut;
    stride_ = stride;
    padding_ = padding;
    dilation_ = dilation;
    bias_ = bias;

    xGm_.SetGlobalBuffer((__gm__ float*)x);
    weightGm_.SetGlobalBuffer((__gm__ float*)weight);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
  }

  __aicore__ inline void Process() {
    if (totalLen_ == 0 || tileLen_ == 0) return;
    uint32_t blockNum = AscendC::GetBlockNum();
    uint32_t blockIdx = AscendC::GetBlockIdx();
    uint32_t strideElems = blockNum * tileLen_;
    uint32_t base = blockIdx * tileLen_;
    if (base >= totalLen_) return;

    // Local buffer for weight (small kernel size, up to maybe 7)
    // We will load weight per output element, but we can load once per tile if same channel.
    // For simplicity, we load each time.
    while (true) {
      uint32_t curLen = totalLen_ - base;
      if (curLen > tileLen_) curLen = tileLen_;
      // Process each output element in this tile
      for (uint32_t i = 0; i < curLen; i++) {
        uint32_t outIdx = base + i;
        // Decode indices (N, C, H_out, W_out)
        uint32_t tmp = outIdx;
        uint32_t w_out = tmp % widthOut_;
        tmp /= widthOut_;
        uint32_t h_out = tmp % heightOut_;
        tmp /= heightOut_;
        uint32_t c = tmp % channels_;
        uint32_t n = tmp / channels_;
        // Compute input base offset for this batch and channel
        uint32_t inputBase = n * channels_ * height_ * width_ + c * height_ * width_ + h_out * width_ + w_out;
        // Weight base offset for this channel: weight shape [C,1,H,1] -> each channel has kernelSize_ floats
        uint32_t weightBase = c * kernelSize_;
        float sum = 0.0f;
        // Loop over kernel height (stride=1, dilation=1, no padding)
        for (uint32_t kh = 0; kh < kernelSize_; kh++) {
          float xVal = xGm_.GetValue(inputBase + kh * width_);
          float wVal = weightGm_.GetValue(weightBase + kh);
          sum += xVal * wVal;
        }
        // If bias is true, we would add bias (not implemented here, assume False)
        yGm_.SetValue(outIdx, sum);
      }
      base += strideElems;
      if (base >= totalLen_) break;
    }
  }

private:
  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> weightGm_;
  AscendC::GlobalTensor<float> yGm_;
  uint32_t totalLen_ {0};
  uint32_t tileLen_ {0};
  uint32_t kernelSize_ {3};
  uint32_t batch_ {1};
  uint32_t channels_ {1};
  uint32_t height_ {1};
  uint32_t width_ {1};
  uint32_t heightOut_ {1};
  uint32_t widthOut_ {1};
  int32_t stride_ {1};
  int32_t padding_ {0};
  int32_t dilation_ {1};
  bool bias_ {false};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelConvDepthwise2dSquareInputAsymmetricKernelCustom op;
  op.Init(x, weight, y,
          tiling_data.totalLength, tiling_data.tileLength,
          tiling_data.kernelSize,
          tiling_data.batch, tiling_data.channels,
          tiling_data.height, tiling_data.width,
          tiling_data.heightOut, tiling_data.widthOut,
          tiling_data.stride, tiling_data.padding, tiling_data.dilation, tiling_data.bias);
  op.Process();
}
