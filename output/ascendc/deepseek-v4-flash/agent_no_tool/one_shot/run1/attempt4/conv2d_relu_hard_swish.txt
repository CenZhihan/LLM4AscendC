host_operator_src="""

#include "conv2d_relu_hard_swish_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dReluHardSwishCustomTilingData tiling;

    // Get input shapes correctly
    const gert::Shape* input_shape = context->GetInputShape(0);
    const gert::Shape* weight_shape = context->GetInputShape(1);

    uint32_t N = input_shape->GetDim(0);
    uint32_t Cin = input_shape->GetDim(1);
    uint32_t H = input_shape->GetDim(2);
    uint32_t W = input_shape->GetDim(3);

    uint32_t Cout = weight_shape->GetDim(0);
    uint32_t K = weight_shape->GetDim(2);  // kernel size (square)

    // Convolution parameters (hardcoded from model)
    uint32_t stride = 1;
    uint32_t pad = 0;
    uint32_t dil = 1;

    uint32_t H_out = (H + 2 * pad - dil * (K - 1) - 1) / stride + 1;
    uint32_t W_out = (W + 2 * pad - dil * (K - 1) - 1) / stride + 1;

    uint32_t totalLength = N * Cout * H_out * W_out;

    // Determine tile size based on UB budget (conservative)
    constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    uint32_t tileLen = 4096;
    if (totalLength < tileLen) tileLen = totalLength;

    uint32_t numTiles = (tileLen == 0) ? 1 : (totalLength + tileLen - 1) / tileLen;
    uint32_t blockDim = (numTiles < 8) ? 8 : ((numTiles > 24) ? 24 : numTiles);

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_K(K);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_strideH(stride);
    tiling.set_strideW(stride);
    tiling.set_padH(pad);
    tiling.set_padW(pad);
    tiling.set_dilationH(dil);
    tiling.set_dilationW(dil);
    tiling.set_hasBias(context->GetInputShape(2) != nullptr);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
