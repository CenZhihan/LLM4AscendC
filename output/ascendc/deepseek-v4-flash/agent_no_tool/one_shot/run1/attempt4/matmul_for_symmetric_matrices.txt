host_operator_src="""

#include "matmul_for_symmetric_matrices_custom_tiling.h"
#include <cmath>
#include "register/op_def_registry.h"

namespace optiling {

// UB model: for a tile of size tileSize x tileSize, we need:
//   inQueueA: 2 * tileSize * tileSize floats
//   inQueueB: 2 * tileSize * tileSize floats
//   outQueueC: 2 * tileSize * tileSize floats
// Total floats = 6 * tileSize * tileSize => bytes = 24 * tileSize * tileSize
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core share (conservative)
static constexpr uint32_t MIN_TILE = 64;
static constexpr uint32_t MAX_TILE = 512;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForSymmetricMatricesCustomTilingData tiling;
    // Input shapes: A and B are both (N, N)
    const uint32_t N = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(0));
    if (N == 0) {
        return ge::GRAPH_FAILED;
    }

    // Calculate maximum tile size from UB budget
    uint32_t maxTile = static_cast<uint32_t>(std::sqrt(UB_BUDGET_BYTES / 24.0f));
    if (maxTile < MIN_TILE) maxTile = MIN_TILE;
    if (maxTile > MAX_TILE) maxTile = MAX_TILE;
    if (maxTile > N) maxTile = N;
    // Align down to multiple of MIN_TILE for simplicity
    uint32_t tileSize = AlignDown(maxTile, MIN_TILE);
    if (tileSize == 0) tileSize = MIN_TILE;

    // Set blockDim: with grid-stride kernel, blockDim can be fixed at a reasonable value
    // Use at most 24 blocks, but ensure coverage by kernel looping over tiles.
    uint32_t blockDim = 24u;
    // Alternative: could be derived from available cores, but 24 is safe.
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_tileSize(tileSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    gert::Shape *c_shape = context->GetOutputShape(0);
    *c_shape = *a_shape; // output same shape as input
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType dt = context->GetInputDataType(0);
    context->SetOutputDataType(0, dt);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulForSymmetricMatrices {
public:
    __aicore__ inline KernelMatmulForSymmetricMatrices() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t N, uint32_t tileSize)
    {
        N_ = N;
        tileSize_ = tileSize;
        aGm_.SetGlobalBuffer((__gm__ float*)a, N_ * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, N_ * N_);

        tileBytes_ = tileSize_ * tileSize_ * sizeof(float);
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileBytes_);
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileBytes_);
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileBytes_);
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || tileSize_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t tilesPerDim = (N_ + tileSize_ - 1) / tileSize_;
        uint32_t totalTiles = tilesPerDim * tilesPerDim;

        // Grid-stride over all output tiles
        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            uint32_t tileRow = tileIdx / tilesPerDim;
            uint32_t tileCol = tileIdx % tilesPerDim;

            uint32_t rowStart = tileRow * tileSize_;
            uint32_t colStart = tileCol * tileSize_;
            uint32_t actualTileRows = (rowStart + tileSize_ <= N_) ? tileSize_ : N_ - rowStart;
            uint32_t actualTileCols = (colStart + tileSize_ <= N_) ? tileSize_ : N_ - colStart;

            // Allocate output tile local tensor and initialize to zero
            AscendC::LocalTensor<float> outTile = outQueueC_.AllocTensor<float>();
            AscendC::Duplicate(outTile, 0.0f, actualTileRows * actualTileCols);

            for (uint32_t k = 0; k < N_; k += tileSize_) {
                uint32_t kStart = k;
                uint32_t kLen = (kStart + tileSize_ <= N_) ? tileSize_ : N_ - kStart;

                // Load A tile: (actualTileRows, kLen) row-major
                AscendC::LocalTensor<float> aTile = inQueueA_.AllocTensor<float>();
                AscendC::DataCopy(aTile, aGm_[rowStart * N_ + kStart], actualTileRows * kLen);
                inQueueA_.EnQue(aTile);

                // Load B tile: (kLen, actualTileCols) row-major
                AscendC::LocalTensor<float> bTile = inQueueB_.AllocTensor<float>();
                AscendC::DataCopy(bTile, bGm_[kStart * N_ + colStart], kLen * actualTileCols);
                inQueueB_.EnQue(bTile);

                aTile = inQueueA_.DeQue<float>();
                bTile = inQueueB_.DeQue<float>();

                // Outer product update: for each row i of outTile, for each k, add a_ik * B_k to outTile[i]
                for (uint32_t i = 0; i < actualTileRows; i++) {
                    float* aRow = (float*)aTile.GetPhyAddr() + i * kLen;
                    float* outRow = (float*)outTile.GetPhyAddr() + i * actualTileCols;
                    for (uint32_t kk = 0; kk < kLen; kk++) {
                        float a_val = aRow[kk];
                        float* bRow = (float*)bTile.GetPhyAddr() + kk * actualTileCols;
                        for (uint32_t j = 0; j < actualTileCols; j++) {
                            outRow[j] += a_val * bRow[j];
                        }
                    }
                }

                inQueueA_.FreeTensor(aTile);
                inQueueB_.FreeTensor(bTile);
            }

            // Copy outTile to global memory
            AscendC::DataCopy(cGm_[rowStart * N_ + colStart], outTile, actualTileRows * actualTileCols);
            outQueueC_.FreeTensor(outTile);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_ {0};
    uint32_t tileSize_ {0};
    uint32_t tileBytes_ {0};
};

extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForSymmetricMatrices op;
    op.Init(a, b, c, tiling_data.N, tiling_data.tileSize);
    op.Process();
}
