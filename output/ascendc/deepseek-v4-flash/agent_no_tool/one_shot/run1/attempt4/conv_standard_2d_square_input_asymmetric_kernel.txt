kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvCustom {
public:
    __aicore__ inline KernelConvCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t batch, uint32_t inChannels, uint32_t outChannels,
                                uint32_t inH, uint32_t inW, uint32_t kernelH, uint32_t kernelW,
                                uint32_t outH, uint32_t outW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW)
    {
        batch_ = batch;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        inH_ = inH;
        inW_ = inW;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        outH_ = outH;
        outW_ = outW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        uint32_t ocPerBlock = (outChannels_ + blockNum - 1) / blockNum;
        uint32_t ocStart = blockIdx * ocPerBlock;
        uint32_t ocEnd = (ocStart + ocPerBlock > outChannels_) ? outChannels_ : ocStart + ocPerBlock;

        if (ocStart >= outChannels_) return;

        AscendC::Conv2D<float, float, float> conv;
        conv.SetConvAttr(strideH_, strideW_, padH_, padW_, dilationH_, dilationW_, 1);
        conv.SetInputShape(batch_, inChannels_, inH_, inW_);
        conv.SetWeightShape(outChannels_, inChannels_, kernelH_, kernelW_);
        conv.SetOutputShape(batch_, outChannels_, outH_, outW_);

        conv.Process(xGm_, wGm_, yGm_);
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_, inChannels_, outChannels_, inH_, inW_, kernelH_, kernelW_, outH_, outW_;
    uint32_t strideH_, strideW_, padH_, padW_, dilationH_, dilationW_;
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(td, tiling);
    KernelConvCustom op;
    op.Init(x, weight, y,
            td.batch, td.inChannels, td.outChannels,
            td.inH, td.inW, td.kernelH, td.kernelW,
            td.outH, td.outW,
            td.strideH, td.strideW,
            td.padH, td.padW,
            td.dilationH, td.dilationW);
    op.Process();
}
