host_operator_src="""

#include "square_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget for double-buffered matmul;
// Each block: floatA (blockM*blockK*4), floatB (blockK*blockN*4), floatC (blockM*blockN*4)
// plus workspace for MatMul (ex. additional 2x blockM*blockN*4)
// Approx = 4*blockM*blockK + 4*blockK*blockN + 12*blockM*blockN  bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 16;  // 16-byte alignment (4 floats)

static inline uint32_t AlignUp(uint32_t x, uint32_t a) {
    return (x + a - 1) / a * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    SquareMatrixMultiplicationCustomTilingData tiling;

    const auto &shapeA = context->GetInputShape(0)->GetOriginShape();
    const auto &shapeB = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = static_cast<uint32_t>(shapeA.GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA.GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB.GetDim(1));

    // Start with common block sizes for square matmul
    uint32_t blockM = 64, blockN = 64, blockK = 128;
    if (M < blockM) blockM = AlignUp(M, ALIGN);
    if (N < blockN) blockN = AlignUp(N, ALIGN);
    if (K < blockK) blockK = AlignUp(K, ALIGN);

    // Estimate UB requirement: multiple buffers for double buffering + temp
    auto required_bytes = [](uint32_t bm, uint32_t bn, uint32_t bk) -> uint32_t {
        // A: 2 * (bm*bk*4) for double buffer
        // B: 2 * (bk*bn*4)
        // C: 2 * (bm*bn*4) for double buffer + temp accumulation
        // MatMul workspace: assume (bm*bn*4) extra
        return 2 * (bm * bk + bk * bn + 2 * bm * bn) * 4;  // conservative
    };

    uint32_t totalBytes = required_bytes(blockM, blockN, blockK);
    while (totalBytes > UB_BUDGET_BYTES &&
           (blockM > 16 || blockN > 16 || blockK > 16)) {
        if (blockM > 16) blockM /= 2;
        if (blockN > 16) blockN /= 2;
        if (blockK > 16) blockK /= 2;
        blockM = AlignUp(blockM, ALIGN);
        blockN = AlignUp(blockN, ALIGN);
        blockK = AlignUp(blockK, ALIGN);
        totalBytes = required_bytes(blockM, blockN, blockK);
    }

    // Number of output tiles in M and N dimensions
    uint32_t tilesM = CeilDiv(M, blockM);
    uint32_t tilesN = CeilDiv(N, blockN);
    uint32_t numTiles = tilesM * tilesN;

    // Block dimension: use as many as tiles, but clamp for stability
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_blockM(blockM);
    tiling.set_blockN(blockN);
    tiling.set_blockK(blockK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src="""

#include "kernel_operator.h"
#include "matmul.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelSquareMatMul {
public:
    __aicore__ inline KernelSquareMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t blockM, uint32_t blockN, uint32_t blockK)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        blockM_ = blockM;
        blockN_ = blockN;
        blockK_ = blockK;

        A_gm_.SetGlobalBuffer((__gm__ float*)A, M_ * K_);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, K_ * N_);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, M_ * N_);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockDim = GetBlockNum();

        // Compute which output tile this block handles
        uint32_t tilesM = (M_ + blockM_ - 1) / blockM_;
        uint32_t tilesN = (N_ + blockN_ - 1) / blockN_;
        uint32_t totalTiles = tilesM * tilesN;

        // Each block processes one tile (or none if out of range)
        if (blockIdx >= totalTiles) return;

        uint32_t tileIdx = blockIdx;
        uint32_t tileM = tileIdx / tilesN;
        uint32_t tileN = tileIdx % tilesN;

        uint32_t startM = tileM * blockM_;
        uint32_t startN = tileN * blockN_;
        uint32_t actualBlockM = min(blockM_, M_ - startM);
        uint32_t actualBlockN = min(blockN_, N_ - startN);

        // Local tensors for tile A, tile B, and output tile C
        // We'll use double buffering for A and B, but for simplicity we do single buffer.
        // For correctness and performance, we use MatMul class.
        LocalTensor<float> blockC;
        LocalTensor<float> blockA, blockB;

        // Allocate local storage for this tile
        pipe_.InitBuffer(tmpA_, 1, actualBlockM * blockK_ * sizeof(float));
        pipe_.InitBuffer(tmpB_, 1, blockK_ * actualBlockN * sizeof(float));
        pipe_.InitBuffer(tmpC_, 1, actualBlockM * actualBlockN * sizeof(float));
        pipe_.InitBuffer(work_, 1, actualBlockM * actualBlockN * sizeof(float) * 2); // workspace

        blockA = tmpA_.Get<float>();
        blockB = tmpB_.Get<float>();
        blockC = tmpC_.Get<float>();

        // Initialize output with zeros
        Duplicate(blockC, 0.0f, actualBlockM * actualBlockN);

        // Loop over K dimension in blocks
        uint32_t blocksK = (K_ + blockK_ - 1) / blockK_;
        for (uint32_t bk = 0; bk < blocksK; bk++) {
            uint32_t startK = bk * blockK_;
            uint32_t actualBlockK = min(blockK_, K_ - startK);

            // Load sub-block of A: shape (actualBlockM, actualBlockK) from global
            DataCopy(blockA, A_gm_[startM * K_ + startK], actualBlockM * actualBlockK);

            // Load sub-block of B: shape (actualBlockK, actualBlockN) from global
            DataCopy(blockB, B_gm_[startK * N_ + startN], actualBlockK * actualBlockN);

            // Perform block matrix multiplication C += A * B
            MatMul(blockC, blockA, blockB, actualBlockM, actualBlockN, actualBlockK,
                   false, false, work_.Get<float>());
        }

        // Write output tile to global memory
        DataCopy(C_gm_[startM * N_ + startN], blockC, actualBlockM * actualBlockN);
    }

private:
    TPipe pipe_;
    TBuf<TPosition::VECCALC> tmpA_, tmpB_, tmpC_, work_;

    GlobalTensor<float> A_gm_;
    GlobalTensor<float> B_gm_;
    GlobalTensor<float> C_gm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t blockM_ {0};
    uint32_t blockN_ {0};
    uint32_t blockK_ {0};
};

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSquareMatMul op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.blockM, tiling_data.blockN, tiling_data.blockK);
    op.Process();
}

#include <torch/library.h>

at::Tensor square_matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
