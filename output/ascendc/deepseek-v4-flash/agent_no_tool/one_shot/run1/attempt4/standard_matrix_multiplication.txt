kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelStandardMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelStandardMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t blockM, uint32_t blockN, uint32_t blockK)
    {
        M_ = M;
        K_ = K;
        N_ = N;
        blockM_ = blockM;
        blockN_ = blockN;
        blockK_ = blockK;

        A_gm_.SetGlobalBuffer((__gm__ float*)A, M * K);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, M * N);

        // Allocate buffers for tiles (double-buffered)
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, blockM_ * blockK_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, blockK_ * blockN_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, blockM_ * blockN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, blockM_ * blockN_ * sizeof(float)); // for accumulation
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Distribute blocks: each core handles a range of (M blocks, N blocks)
        uint32_t totalMB = (M_ + blockM_ - 1u) / blockM_;
        uint32_t totalNB = (N_ + blockN_ - 1u) / blockN_;
        uint32_t totalBlocks = totalMB * totalNB;
        uint32_t blocksPerCore = (totalBlocks + blockNum - 1u) / blockNum;
        uint32_t startBlock = blockIdx * blocksPerCore;
        uint32_t endBlock = (startBlock + blocksPerCore) < totalBlocks ? (startBlock + blocksPerCore) : totalBlocks;

        for (uint32_t blkIdx = startBlock; blkIdx < endBlock; blkIdx++) {
            uint32_t mBlock = blkIdx / totalNB;
            uint32_t nBlock = blkIdx % totalNB;
            uint32_t mStart = mBlock * blockM_;
            uint32_t mEnd = (mStart + blockM_) < M_ ? (mStart + blockM_) : M_;
            uint32_t mRow = mEnd - mStart;
            uint32_t nStart = nBlock * blockN_;
            uint32_t nEnd = (nStart + blockN_) < N_ ? (nStart + blockN_) : N_;
            uint32_t nCol = nEnd - nStart;
            uint32_t totalKB = (K_ + blockK_ - 1u) / blockK_;

            // Initialize accumulator tile to zero
            LocalTensor<float> accLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Duplicate(accLocal, 0.0f, mRow * nCol);

            for (uint32_t kBlock = 0; kBlock < totalKB; kBlock++) {
                uint32_t kStart = kBlock * blockK_;
                uint32_t kEnd = (kStart + blockK_) < K_ ? (kStart + blockK_) : K_;
                uint32_t kLen = kEnd - kStart;

                // Load A tile (mRow x kLen) from global memory
                LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
                // A is stored row-major: offset = mStart*K_ + kStart
                uint32_t aOffset = mStart * K_ + kStart;
                AscendC::DataCopy(aLocal, A_gm_[aOffset], mRow * kLen);

                // Load B tile (kLen x nCol) from global memory
                LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
                // B is stored row-major: offset = kStart*N_ + nStart
                uint32_t bOffset = kStart * N_ + nStart;
                AscendC::DataCopy(bLocal, B_gm_[bOffset], kLen * nCol);

                // Perform matrix multiplication using ASCEND C Matmul
                Matmul<MatmulType<AscendC::MatmulType::NORMAL>> mm;
                mm.Init(aLocal, bLocal, accLocal, false, false);
                mm.SetTail(kLen, mRow, nCol);
                uint32_t iterK = kLen; // single iteration
                mm.Process(iterK, iterK, 1);

                // Free loaded tensors
                inQueueA_.FreeTensor(aLocal);
                inQueueB_.FreeTensor(bLocal);
            }

            // Store result tile to global memory
            // accLocal now holds the partial result (accumulated across K tiles)
            LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
            // Move accumulated result to output queue (by copy)
            for (uint32_t i = 0; i < mRow * nCol; i++) {
                cLocal.SetValue(i, accLocal.GetValue(i));
            }
            outQueueC_.EnQue(cLocal);
            // Dequeue and write to global memory
            LocalTensor<float> cOut = outQueueC_.DeQue<float>();
            uint32_t cOffset = mStart * N_ + nStart;
            AscendC::DataCopy(C_gm_[cOffset], cOut, mRow * nCol);
            outQueueC_.FreeTensor(cOut);
            tmpQueue_.FreeTensor(accLocal);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueA_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueB_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> A_gm_;
    GlobalTensor<float> B_gm_;
    GlobalTensor<float> C_gm_;

    uint32_t M_, K_, N_;
    uint32_t blockM_, blockN_, blockK_;
};

extern "C" __global__ __aicore__ void standard_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelStandardMatrixMultiplicationCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.blockM, tiling_data.blockN, tiling_data.blockK);
    op.Process();
}
