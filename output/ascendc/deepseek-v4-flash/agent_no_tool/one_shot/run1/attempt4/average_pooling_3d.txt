#include "average_pooling_3d_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling3dCustomTilingData tiling;

    const gert::StorageShape *inputShape = context->GetInputShape(0);
    uint32_t n = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(0));
    uint32_t c = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(1));
    uint32_t d = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(2));
    uint32_t h = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(3));
    uint32_t w = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t kernelSize = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(0)[0]);
    uint32_t stride = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(1)[0]);
    uint32_t padding = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(2)[0]);

    uint32_t od = (d + 2 * padding - kernelSize) / stride + 1;
    uint32_t oh = (h + 2 * padding - kernelSize) / stride + 1;
    uint32_t ow = (w + 2 * padding - kernelSize) / stride + 1;

    tiling.set_inputN(n);
    tiling.set_inputC(c);
    tiling.set_inputD(d);
    tiling.set_inputH(h);
    tiling.set_inputW(w);
    tiling.set_outputN(n);
    tiling.set_outputC(c);
    tiling.set_outputD(od);
    tiling.set_outputH(oh);
    tiling.set_outputW(ow);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);

    uint32_t totalLength = n * c * od * oh * ow;
    tiling.set_totalLength(totalLength);

    uint32_t blockDim = 24;
    context->SetBlockDim(blockDim);
    uint32_t tileLength = (totalLength + blockDim - 1) / blockDim;
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::StorageShape *inputShape = context->GetInputShape(0);
    uint32_t n = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(0));
    uint32_t c = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(1));
    uint32_t d = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(2));
    uint32_t h = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(3));
    uint32_t w = static_cast<uint32_t>(inputShape->GetOriginShape().GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t kernelSize = attrs->GetAttrPointer<int64_t>(0)[0];
    int64_t stride = attrs->GetAttrPointer<int64_t>(1)[0];
    int64_t padding = attrs->GetAttrPointer<int64_t>(2)[0];

    uint32_t od = (d + 2 * padding - kernelSize) / stride + 1;
    uint32_t oh = (h + 2 * padding - kernelSize) / stride + 1;
    uint32_t ow = (w + 2 * padding - kernelSize) / stride + 1;
