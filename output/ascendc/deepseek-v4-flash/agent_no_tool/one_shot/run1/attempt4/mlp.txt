    const int32_t *p_outputSize = attrs->GetAttrPointer<int32_t>(3);
    uint32_t hidden0 = static_cast<uint32_t>(*p_hidden0);
    uint32_t hidden1 = static_cast<uint32_t>(*p_hidden1);
    uint32_t outputSize = static_cast<uint32_t>(*p_outputSize);

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_inputSize(inputSize);
    tiling.set_hidden0(hidden0);
    tiling.set_hidden1(hidden1);
    tiling.set_outputSize(outputSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMlpCustom {
public:
    __aicore__ inline KernelMlpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed,
                                GM_ADDR y,
                                uint32_t batch, uint32_t inputSize,
                                uint32_t hidden0, uint32_t hidden1,
                                uint32_t outputSize)
    {
        batch_ = batch;
        inputSize_ = inputSize;
        hidden0_ = hidden0;
        hidden1_ = hidden1;
        outputSize_ = outputSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * inputSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w_packed,
                             hidden0 * inputSize + hidden1 * hidden0 + outputSize * hidden1);
        bGm_.SetGlobalBuffer((__gm__ float*)b_packed,
                             hidden0 + hidden1 + outputSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * outputSize);

        // Local buffers
        pipe_.InitBuffer(xLocalQue_, BUFFER_NUM, inputSize_ * sizeof(float));
        pipe_.InitBuffer(wLocalQue_, BUFFER_NUM, inputSize_ * sizeof(float)); // weight row size max inputSize
        pipe_.InitBuffer(accumQue_, 1, inputSize_ * sizeof(float));          // accum for hidden states
        pipe_.InitBuffer(hidden0Que_, BUFFER_NUM, hidden0_ * sizeof(float));
        pipe_.InitBuffer(hidden1Que_, BUFFER_NUM, hidden1_ * sizeof(float));
        pipe_.InitBuffer(outQue_, BUFFER_NUM, outputSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t b = 0; b < batch_; b++) {
            // Load x row
            LocalTensor<float> xLocal = xLocalQue_.AllocTensor<float>();
            DataCopy(xLocal, xGm_[b * inputSize_], inputSize_);
            xLocalQue_.EnQue(xLocal);

            // ---- Layer 1: W1 * x + b1 -> h0, relu ----
            LocalTensor<float> h0 = hidden0Que_.AllocTensor<float>();
            for (uint32_t r = 0; r < hidden0_; r++) {
                // Load weight row
                LocalTensor<float> wRow = wLocalQue_.AllocTensor<float>();
                DataCopy(wRow, wGm_[r * inputSize_], inputSize_);
                wLocalQue_.EnQue(wRow);

                // Compute dot product
                LocalTensor<float> accum = accumQue_.AllocTensor<float>();
                LocalTensor<float> xDeq = xLocalQue_.DeQue<float>();
                LocalTensor<float> wDeq = wLocalQue_.DeQue<float>();
                Mul(accum, xDeq, wDeq, inputSize_);
                float sum = ReduceSum(accum, inputSize_);
                accumQue_.FreeTensor(accum);
                xLocalQue_.EnQue(xDeq);  // reuse x for next row
                wLocalQue_.FreeTensor(wDeq);

                // Bias and store
                float bias = bGm_[r];
                h0[r] = sum + bias;
            }
            // Apply relu
            Maxs(h0, h0, 0.0f, hidden0_);
            hidden0Que_.EnQue(h0);

            // ---- Layer 2: W2 * h0 + b2 -> h1, relu ----
            LocalTensor<float> h1 = hidden1Que_.AllocTensor<float>();
            uint32_t wOffset = hidden0_ * inputSize_;
            for (uint32_t r = 0; r < hidden1_; r++) {
                LocalTensor<float> wRow = wLocalQue_.AllocTensor<float>();
                DataCopy(wRow, wGm_[wOffset + r * hidden0_], hidden0_);
                wLocalQue_.EnQue(wRow);

                LocalTensor<float> accum = accumQue_.AllocTensor<float>();
                LocalTensor<float> h0Deq = hidden0Que_.DeQue<float>();
                LocalTensor<float> wDeq = wLocalQue_.DeQue<float>();
                Mul(accum, h0Deq, wDeq, hidden0_);
                float sum = ReduceSum(accum, hidden0_);
                accumQue_.FreeTensor(accum);
                hidden0Que_.EnQue(h0Deq);
                wLocalQue_.FreeTensor(wDeq);

                float bias = bGm_[hidden0_ + r];
                h1[r] = sum + bias;
            }
            Maxs(h1, h1, 0.0f, hidden1_);
            hidden1Que_.EnQue(h1);

            // ---- Layer 3: W3 * h1 + b3 -> output ----
            LocalTensor<float> out = outQue_.AllocTensor<float>();
            wOffset += hidden1_ * hidden0_;
            for (uint32_t r = 0; r < outputSize_; r++) {
                LocalTensor<float> wRow = wLocalQue_.AllocTensor<float>();
                DataCopy(wRow, wGm_[wOffset + r * hidden1_], hidden1_);
                wLocalQue_.EnQue(wRow);

                LocalTensor<float> accum = accumQue_.AllocTensor<float>();
                LocalTensor<float> h1Deq = hidden1Que_.DeQue<float>();
                LocalTensor<float> wDeq = wLocalQue_.DeQue<float>();
                Mul(accum, h1Deq, wDeq, hidden1_);
                float sum = ReduceSum(accum, hidden1_);
                accumQue_.FreeTensor(accum);
                hidden1Que_.EnQue(h1Deq);
                wLocalQue_.FreeTensor(wDeq);

                float bias = bGm_[hidden0_ + hidden1_ + r];
                out[r] = sum + bias;
            }
            outQue_.EnQue(out);

            // Write output row
            LocalTensor<float> outDeq = outQue_.DeQue<float>();
            DataCopy(yGm_[b * outputSize_], outDeq, outputSize_);
            outQue_.FreeTensor(outDeq);

            // Free remaining local tensors
            LocalTensor<float> xFinal = xLocalQue_.DeQue<float>();
            xLocalQue_.FreeTensor(xFinal);
            LocalTensor<float> h0Final = hidden0Que_.DeQue<float>();
            hidden0Que_.FreeTensor(h0Final);
            LocalTensor<float> h1Final = hidden1Que_.DeQue<float>();
            hidden1Que_.FreeTensor(h1Final);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> xLocalQue_;
    TQue<TPosition::VECIN, BUFFER_NUM> wLocalQue_;
    TQue<TPosition::VECCALC, 1> accumQue_;
    TQue<TPosition::VECOUT, BUFFER_NUM> hidden0Que_;
    TQue<TPosition::VECOUT, BUFFER_NUM> hidden1Que_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t batch_ = 0;
    uint32_t inputSize_ = 0;
    uint32_t hidden0_ = 0;
    uint32_t hidden1_ = 0;
    uint32_t outputSize_ = 0;
};

extern "C" __global__ __aicore__ void mlp_custom(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed,
                                                  GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMlpCustom op;
    op.Init(x, w_packed, b_packed, y,
            tiling_data.batch, tiling_data.inputSize,
            tiling_data.hidden0, tiling_data.hidden1,
            tiling_data.outputSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'mlp_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, layer_sizes, output_size, *args, **kwargs):
        super(ModelNew, self).__init__()
        self.input_size = input_size
        self.hidden_0 = layer_sizes[0]
        self.hidden_1 = layer_sizes[1]
        self.output_size = output_size

    def forward(self, x, w_packed, b_packed):
        fn = getattr(custom_ops_lib, _FN)
        return fn(
            x, w_packed, b_packed,
            input_size=self.input_size,
            hidden_0=self.hidden_0,
            hidden_1=self.hidden_1,
            output_size=self.output_size
