#include "conv2d_subtract_subtract_mish_custom_tiling.h"

namespace optiling {

// UB budget estimation: conv tile output + weight + input window + temp for mish
// Conservative 192KB per core
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dSubtractSubtractMishCustomTilingData tiling;

    // Use GetInputShape which returns const gert::Shape* directly
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    uint32_t inN  = static_cast<uint32_t>(x_shape->GetDim(0));
    uint32_t inC  = static_cast<uint32_t>(x_shape->GetDim(1));
    uint32_t inH  = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t inW  = static_cast<uint32_t>(x_shape->GetDim(3));
    uint32_t outC = static_cast<uint32_t>(w_shape->GetDim(0));
    uint32_t kH   = static_cast<uint32_t>(w_shape->GetDim(2));
    uint32_t kW   = static_cast<uint32_t>(w_shape->GetDim(3));

    // Convolution parameters (assume padding=0, stride=1, dilation=1)
    uint32_t pad = 0;
    uint32_t stride = 1;
    uint32_t dilation = 1;
    uint32_t outH = (inH + 2 * pad - dilation * (kH - 1) - 1) / stride + 1;
    uint32_t outW = (inW + 2 * pad - dilation * (kW - 1) - 1) / stride + 1;
    uint32_t totalLength = inN * outC * outH * outW;

    // Attrs
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *subtractValue1 = attrs->GetAttrPointer<float>(0);
    const float *subtractValue2 = attrs->GetAttrPointer<float>(1);

    // Check if bias is provided (input index 2 is optional)
    bool hasBias = (context->GetInputShape(2) != nullptr);

    // Tile length calculation: use UB budget for one output tile + weight + input window
    uint32_t weightElems = outC * inC * kH * kW;
    uint32_t inputWindowElems = kH * kW * inC;
    uint32_t miscElems = 512; // for mish temp
    uint32_t neededElems = weightElems + inputWindowElems + miscElems;
    uint32_t maxTile = (UB_BUDGET_BYTES / sizeof(float) >= neededElems) ? ((UB_BUDGET_BYTES / sizeof(float) - neededElems) / 2) : 256;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // Block dim
    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_inN(inN);
    tiling.set_inC(inC);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outC(outC);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_pad(pad);
    tiling.set_stride(stride);
    tiling.set_dilation(dilation);
    tiling.set_hasBias(hasBias ? 1 : 0);
    tiling.set_subtractValue1(*subtractValue1);
    tiling.set_subtractValue2(*subtractValue2);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dSubtractSubtractMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractSubtractMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inN, uint32_t inC, uint32_t inH, uint32_t inW,
                                uint32_t outC, uint32_t kH, uint32_t kW,
                                uint32_t pad, uint32_t stride, uint32_t dilation,
                                int32_t hasBias,
                                float subtractValue1, float subtractValue2)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inN_ = inN; inC_ = inC; inH_ = inH; inW_ = inW;
        outC_ = outC; kH_ = kH; kW_ = kW;
        pad_ = pad; stride_ = stride; dilation_ = dilation;
        hasBias_ = hasBias;
        subtractValue1_ = subtractValue1;
        subtractValue2_ = subtractValue2;

        // Output dimensions
        outH_ = (inH_ + 2 * pad_ - dilation_ * (kH_ - 1) - 1) / stride_ + 1;
        outW_ = (inW_ + 2 * pad_ - dilation_ * (kW_ - 1) - 1) / stride_ + 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        if (hasBias_) biasGm_.SetGlobalBuffer((__gm__ float*)bias, outC_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Temporary local buffers for tile processing
        auto pipe = AscendC::TPipe();
        pipe.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Per-element convolution using global memory reads
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t linearIdx = base + i;
                uint32_t n = linearIdx / (outC_ * outH_ * outW_);
                uint32_t rem1 = linearIdx % (outC_ * outH_ * outW_);
                uint32_t oc = rem1 / (outH_ * outW_);
                uint32_t rem2 = rem1 % (outH_ * outW_);
                uint32_t oh = rem2 / outW_;
                uint32_t ow = rem2 % outW_;

                int32_t h_start = static_cast<int32_t>(oh * stride_) - static_cast<int32_t>(pad_);
                int32_t w_start = static_cast<int32_t>(ow * stride_) - static_cast<int32_t>(pad_);

                float sum = 0.0f;
                for (uint32_t ic = 0; ic < inC_; ic++) {
                    for (uint32_t kh = 0; kh < kH_; kh++) {
                        for (uint32_t kw = 0; kw < kW_; kw++) {
                            int32_t h_idx = h_start + static_cast<int32_t>(kh * dilation_);
                            int32_t w_idx = w_start + static_cast<int32_t>(kw * dilation_);
                            if (h_idx >= 0 && h_idx < static_cast<int32_t>(inH_) &&
                                w_idx >= 0 && w_idx < static_cast<int32_t>(inW_)) {
                                float x_val = xGm_.GetValue(n * inC_ * inH_ * inW_ +
                                                             ic * inH_ * inW_ +
                                                             h_idx * inW_ + w_idx);
                                float w_val = weightGm_.GetValue(oc * inC_ * kH_ * kW_ +
                                                                  ic * kH_ * kW_ +
                                                                  kh * kW_ + kw);
                                sum += x_val * w_val;
                            }
                        }
                    }
                }
                if (hasBias_) {
                    sum += biasGm_.GetValue(oc);
                }
                sum -= subtractValue1_;
                sum -= subtractValue2_;

                // Mish activation using standard math functions
                float mish_val = sum * std::tanh(std::log(std::exp(sum) + 1.0f));
                yGm_.SetValue(linearIdx, mish_val);
            }

            base += strideElems;
        }
    }

private:
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t inN_ {1}, inC_ {1}, inH_ {1}, inW_ {1};
    uint32_t outC_ {1}, kH_ {1}, kW_ {1};
    uint32_t outH_ {1}, outW_ {1};
    uint32_t pad_ {0}, stride_ {1}, dilation_ {1};
    int32_t hasBias_ {0};
    float subtractValue1_ {0.5f};
    float subtractValue2_ {0.2f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_subtract_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractSubtractMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.inN, tiling_data.inC, tiling_data.inH, tiling_data.inW,
            tiling_data.outC, tiling_data.kH, tiling_data.kW,
            tiling_data.pad, tiling_data.stride, tiling_data.dilation,
            tiling_data.hasBias,
            tiling_data.subtractValue1, tiling_data.subtractValue2);
    op.Process();
}
