#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t MAX_TILE_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsilon = attrs->GetAttrPointer<float>(0);

    uint32_t totalLength = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    uint32_t normalizedSize = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());

    if (normalizedSize == 0) return ge::GRAPH_FAILED;
    uint32_t rows = totalLength / normalizedSize;
    if (rows * normalizedSize != totalLength) return ge::GRAPH_FAILED;

    uint32_t ubPerRow = 6 * normalizedSize * sizeof(float);
    uint32_t maxRows = UB_BUDGET_BYTES / ubPerRow;
    if (maxRows < 1) maxRows = 1;
    uint32_t rowsPerTile = maxRows;
    if (rowsPerTile * normalizedSize > MAX_TILE_ELEMS)
        rowsPerTile = AlignDown(MAX_TILE_ELEMS, normalizedSize) / normalizedSize;
    if (rowsPerTile < 1) rowsPerTile = 1;

    uint32_t tileLen = rowsPerTile * normalizedSize;

    uint32_t numTiles = CeilDiv(rows, rowsPerTile);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_normalizedSize(normalizedSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_epsilon(*epsilon);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t normalizedSize,
                                uint32_t tileLength, float epsilon)
    {
        totalLen_ = totalLength;
        normSize_ = normalizedSize;
        tileLen_ = tileLength;
        eps_ = epsilon;
        invNormSize_ = 1.0f / normSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, normSize_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, normSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, 1, normSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, normSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
        // Single element tensors for ReduceSum results
        pipe_.InitBuffer(sumQueue_, 1, sizeof(float));
        pipe_.InitBuffer(sqSumQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || normSize_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, normSize_);
        AscendC::DataCopy(bLocal, bGm_, normSize_);
        inQueueW_.EnQue(wLocal);
        inQueueB_.EnQue(bLocal);

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) break;

            CopyIn(base, curLen);
            Compute(curLen);
            CopyOut(base, curLen);

            base += strideElems;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp2 = tmpQueue2_.AllocTensor<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB_.DeQue<float>();

        AscendC::LocalTensor<float> sumScalar = sumQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> sqSumScalar = sqSumQueue_.AllocTensor<float>();

        uint32_t numRows = len / normSize_;
        for (uint32_t r = 0; r < numRows; ++r) {
            uint32_t offset = r * normSize_;
            AscendC::LocalTensor<float> rowX = xLocal[offset];
            AscendC::LocalTensor<float> rowY = yLocal[offset];

            // sum of x
            AscendC::ReduceSum(sumScalar, rowX, normSize_, normSize_);
            float mean = sumScalar.GetValue(0) * invNormSize_;

            // x - mean
            AscendC::Adds(tmp, rowX, -mean, normSize_);

            // (x - mean)^2
            AscendC::Mul(tmp2, tmp, tmp, normSize_);

            // sum of squares
            AscendC::ReduceSum(sqSumScalar, tmp2, normSize_, normSize_);
            float variance = sqSumScalar.GetValue(0) * invNormSize_;

            // standard deviation
            float invStd = 1.0f / sqrt(variance + eps_);

            // scale by invStd
            AscendC::Muls(tmp, tmp, invStd, normSize_);

            // multiply by weight
            AscendC::Mul(tmp, tmp, wLocal, normSize_);

            // add bias
            AscendC::Add(rowY, tmp, bLocal, normSize_);
        }

        sqSumQueue_.FreeTensor(sqSumScalar);
        sumQueue_.FreeTensor(sumScalar);
        inQueueW_.EnQue(wLocal);
        inQueueB_.EnQue(bLocal);
        tmpQueue2_.FreeTensor(tmp2);
        tmpQueue_.FreeTensor(tmp);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sqSumQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t normSize_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
    float invNormSize_ {0.0f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.normalizedSize,
            tiling_data.tileLength, tiling_data.epsilon);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        self.normalized_shape = normalized_shape
        self.weight = torch.nn.Parameter(torch.ones(normalized_shape, dtype=torch.float))
        self.bias = torch.nn.Parameter(torch.zeros(normalized_shape, dtype=torch.float))

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.weight, self.bias, 1e-5)
