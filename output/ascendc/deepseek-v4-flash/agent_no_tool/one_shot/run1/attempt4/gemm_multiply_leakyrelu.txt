namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMultiplyLeakyreluCustomTilingData tiling;

    const auto &xOrigin = context->GetInputShape(0)->GetOriginShape();
    const auto &wOrigin = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = xOrigin.GetDim(0);
    uint32_t K = xOrigin.GetDim(1);
    uint32_t N = wOrigin.GetDim(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float multiplier = *(attrs->GetAttrPointer<float>(0));
    float negativeSlope = *(attrs->GetAttrPointer<float>(1));

    // Tile sizes: start with 256 for each dimension and adapt to UB
    uint32_t tileM = 256, tileN = 256, tileK = 256;
    uint32_t ubNeeded = (tileM*tileK + tileN*tileK + tileM*tileN) * sizeof(float);
    if (ubNeeded > UB_BUDGET_BYTES) {
        float ratio = (float)UB_BUDGET_BYTES / ubNeeded;
        tileM = (uint32_t)(tileM * std::sqrt(ratio));
        tileN = (uint32_t)(tileN * std::sqrt(ratio));
        tileK = (uint32_t)(tileK * std::sqrt(ratio));
        tileM = AlignDown(tileM, ALIGN_ELEMS);
        tileN = AlignDown(tileN, ALIGN_ELEMS);
        tileK = AlignDown(tileK, ALIGN_ELEMS);
        if (tileM < 32) tileM = 32;
        if (tileN < 32) tileN = 32;
        if (tileK < 32) tileK = 32;
    }

    uint32_t tilesM = CeilDiv(M, tileM);
    uint32_t tilesN = CeilDiv(N, tileN);
    uint32_t blockNum = tilesM * tilesN;
    if (blockNum < 8) blockNum = 8;
    if (blockNum > 24) blockNum = 24;

    context->SetBlockDim(blockNum);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockNum(blockNum);
    tiling.set_multiplier(multiplier);
    tiling.set_negativeSlope(negativeSlope);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "matmul_api.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMultiplyLeakyreluCustom {
public:
    __aicore__ inline KernelGemmMultiplyLeakyreluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                float multiplier, float negativeSlope)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN; tileK_ = tileK;
        multiplier_ = multiplier;
        negativeSlope_ = negativeSlope;
        negCorr_ = negativeSlope_ - 1.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M * K);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N * K);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M * N);

        // Buffers for double-buffered tiles
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileN_ * tileK_ * sizeof(float));
        // Buffers for partial sum and intermediate products
        pipe_.InitBuffer(partialQueue_, 1, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(prodQueue_, 1, tileM_ * tileN_ * sizeof(float));
        // Bias buffer
        pipe_.InitBuffer(biasQueue_, 1, tileN_ * sizeof(float));
        // Output queue (single buffer used for writing)
        pipe_.InitBuffer(outQueueY_, 1, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t tilesM = (M_ + tileM_ - 1) / tileM_;
        uint32_t tilesN = (N_ + tileN_ - 1) / tileN_;
        if (blockIdx >= tilesM * tilesN) return;

        uint32_t tidxM = blockIdx / tilesN;
        uint32_t tidxN = blockIdx % tilesN;
        uint32_t rowStart = tidxM * tileM_;
        uint32_t colStart = tidxN * tileN_;
        uint32_t curM = min(tileM_, M_ - rowStart);
        uint32_t curN = min(tileN_, N_ - colStart);

        // Load bias once
        AscendC::LocalTensor<float> bias = biasQueue_.AllocTensor<float>();
        AscendC::DataCopy(bias, bGm_[colStart], curN);
        // Keep bias for later addition (we will not use queue for bias after this)

        // Initialize partial sum to zero
        AscendC::LocalTensor<float> partial = partialQueue_.AllocTensor<float>();
        AscendC::SetZero(partial, curM * curN);

        // K tiling loop
        for (uint32_t kOff = 0; kOff < K_; kOff += tileK_) {
            uint32_t curK = min(tileK_, K_ - kOff);

            // Load x tile (curM x curK) from global memory
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            for (uint32_t r = 0; r < curM; r++) {
                AscendC::DataCopy(xLocal[r * curK], xGm_[(rowStart + r) * K_ + kOff], curK);
            }
            inQueueX_.EnQue(xLocal);

            // Load w tile (curN x curK) from global memory. w is stored row-major (N x K).
            // We will use transposeB=true in MatMul, so load as is.
            AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
            for (uint32_t c = 0; c < curN; c++) {
                AscendC::DataCopy(wLocal[c * curK], wGm_[(colStart + c) * K_ + kOff], curK);
            }
            inQueueW_.EnQue(wLocal);

            // Compute product: x_tile * w_tile^T, result size curM * curN
            AscendC::LocalTensor<float> prod = prodQueue_.AllocTensor<float>();
            AscendC::MatMul(prod, inQueueX_.DeQue<float>(), inQueueW_.DeQue<float>(), curM, curN, curK, false, true);
            // Accumulate into partial
            AscendC::Add(partial, partial, prod, curM * curN);
            prodQueue_.FreeTensor(prod);

            // Free x and w tiles after use
            inQueueX_.FreeTensor(inQueueX_.DeQue<float>());
            inQueueW_.FreeTensor(inQueueW_.DeQue<float>());
        }

        // Apply bias
        AscendC::Add(partial, partial, bias, curM * curN);
        biasQueue_.FreeTensor(bias);

        // Multiply by multiplier
        AscendC::Muls(partial, partial, multiplier_, curM * curN);

        // LeakyReLU
        AscendC::LocalTensor<float> tmp = prodQueue_.AllocTensor<float>(); // reuse prod queue
        AscendC::Mins(tmp, partial, 0.0f, curM * curN);
        AscendC::Muls(tmp, tmp, negCorr_, curM * curN);
        AscendC::Add(partial, partial, tmp, curM * curN);
        prodQueue_.FreeTensor(tmp);

        // Copy to output buffer and write to global
        AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();
        AscendC::DataCopy(outLocal, partial, curM * curN);
        for (uint32_t r = 0; r < curM; r++) {
            AscendC::DataCopy(yGm_[(rowStart + r) * N_ + colStart], outLocal[r * curN], curN);
        }
        outQueueY_.FreeTensor(outLocal);

        // Free partial
        partialQueue_.FreeTensor(partial);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> partialQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> prodQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> biasQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_, N_, K_;
    uint32_t tileM_, tileN_, tileK_;
    float multiplier_, negativeSlope_, negCorr_;
};

extern "C" __global__ __aicore__ void gemm_multiply_leakyrelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMultiplyLeakyreluCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.multiplier, tiling_data.negativeSlope);
    op.Process();
}
