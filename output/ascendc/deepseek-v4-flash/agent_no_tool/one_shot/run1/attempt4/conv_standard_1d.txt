#include "conv_standard_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget and alignment (conservative)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ConvStandard1dCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    uint32_t N = xShape->GetDim(0);
    uint32_t C = xShape->GetDim(1);
    uint32_t L = xShape->GetDim(2);
    uint32_t OC = wShape->GetDim(0);
    uint32_t kW = wShape->GetDim(2);
    uint32_t L_out = (L - (kW - 1)) / 1 + 1;  // stride=1, padding=0, dilation=1

    uint32_t totalLength = N * OC * L_out;

    // Simple tiling: fixed tile size for demonstration
    uint32_t tileLength = 1024;
    tileLength = AlignDown(tileLength, ALIGN_ELEMS);
    if (tileLength == 0) tileLength = ALIGN_ELEMS;

    uint32_t blockDim = totalLength / tileLength;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_L(L);
    tiling.set_OC(OC);
    tiling.set_kW(kW);
    tiling.set_L_out(L_out);
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    uint32_t N = xShape->GetDim(0);
    uint32_t OC = wShape->GetDim(0);
    uint32_t L = xShape->GetDim(2);
    uint32_t kW = wShape->GetDim(2);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard1dCustom {
public:
    __aicore__ inline KernelConvStandard1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t L,
                                uint32_t OC, uint32_t kW, uint32_t L_out,
                                uint32_t tileLength)
    {
        N_ = N; C_ = C; L_ = L; OC_ = OC; kW_ = kW; L_out_ = L_out;
        tileLen_ = tileLength;
        totalLen_ = N_ * OC_ * L_out_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * L_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, OC_ * C_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t end = base + tileLen_;
        if (end > totalLen_) end = totalLen_;

        for (uint32_t idx = base; idx < end; ++idx) {
            uint32_t l = idx % L_out_;
            uint32_t oc = (idx / L_out_) % OC_;
            uint32_t n = idx / (OC_ * L_out_);

            float sum = 0.0f;
            for (uint32_t c = 0; c < C_; ++c) {
                for (uint32_t i = 0; i < kW_; ++i) {
                    sum += xGm_.GetValue(n * C_ * L_ + c * L_ + (l + i)) *
                           wGm_.GetValue(oc * C_ * kW_ + c * kW_ + i);
                }
            }
            yGm_.SetValue(idx, sum);
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, L_, OC_, kW_, L_out_;
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.C, tiling_data.L,
            tiling_data.OC, tiling_data.kW, tiling_data.L_out,
            tiling_data.tileLength);
    op.Process();
}
