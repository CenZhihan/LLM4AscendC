static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t totalLength =
        static_cast<uint32_t>(shape.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dim = attrs->GetAttrPointer<int32_t>(0);

    uint32_t batchSize = 1, featureSize = totalLength;
    if (shape.GetDimNum() >= 2) {
        uint32_t lastIdx = shape.GetDimNum() - 1;
        featureSize = shape.GetDim(lastIdx);
        for (uint32_t i = 0; i < lastIdx; ++i)
            batchSize *= shape.GetDim(i);
    }

    // tileLength = per-tile size, aligned and capped to fit UB budget
    uint32_t tileLen = AlignDown(featureSize, ALIGN_ELEMS);
    if (tileLen > HARD_CAP_ELEMS) tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = featureSize;

    // Ensure UB budget: need 4 * tileLen floats (x, y, tmp1, tmp2) + 3 scalars negligible
    while (tileLen > 0 && (4u * tileLen * 4u) > UB_BUDGET_BYTES) {
        tileLen = AlignDown(tileLen - ALIGN_ELEMS, ALIGN_ELEMS);
    }
    if (tileLen == 0) tileLen = ALIGN_ELEMS;

    uint32_t blockDim = batchSize;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_featureSize(featureSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(*dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

}
