class KernelMatmulForSymmetricMatrices {
public:
    __aicore__ inline KernelMatmulForSymmetricMatrices() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t N, uint32_t tileSize)
    {
        N_ = N;
        tileSize_ = tileSize;
        aGm_.SetGlobalBuffer((__gm__ float*)a, N_ * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, N_ * N_);

        tileBytes_ = tileSize_ * tileSize_ * sizeof(float);
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileBytes_);
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileBytes_);
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileBytes_);
        // No tmp queue needed for output; we use outQueueC directly
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || tileSize_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t tilesPerDim = (N_ + tileSize_ - 1) / tileSize_;
        uint32_t tileRow = blockIdx / tilesPerDim;
        uint32_t tileCol = blockIdx % tilesPerDim;
        if (tileRow >= tilesPerDim || tileCol >= tilesPerDim) return;

        uint32_t rowStart = tileRow * tileSize_;
        uint32_t colStart = tileCol * tileSize_;
        uint32_t actualTileRows = (rowStart + tileSize_ <= N_) ? tileSize_ : N_ - rowStart;
