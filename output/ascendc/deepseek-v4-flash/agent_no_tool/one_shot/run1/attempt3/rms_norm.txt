#include "rms_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    RmsNormCustomTilingData tiling;
    const gert::StorageShape *xStorageShape = context->GetInputShape(0);
    const gert::Shape &xOriginShape = xStorageShape->GetOriginShape();
    uint32_t shapeRank = xOriginShape.GetDimNum();
    if (shapeRank < 2) {
        return ge::GRAPH_FAILED;
    }
    uint32_t numRows = 1;
    uint32_t rowLength = xOriginShape.GetDim(shapeRank - 1);
    for (uint32_t i = 0; i < shapeRank - 1; ++i) {
        numRows *= xOriginShape.GetDim(i);
    }
    uint32_t totalLength = numRows * rowLength;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = (attrs->GetAttrNum() > 0) ? *attrs->GetAttrPointer<float>(0) : 1e-5f;

    uint32_t tileLen = rowLength;
    if (tileLen > HARD_CAP_ELEMS) {
        tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    }
    if (tileLen > (UB_BUDGET_BYTES / (3 * sizeof(float)))) {
        tileLen = AlignDown(UB_BUDGET_BYTES / (3 * sizeof(float)), ALIGN_ELEMS);
    }
    if (tileLen < 1) tileLen = 1;

    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_numRows(numRows);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelRmsNormCustom {
public:
    __aicore__ inline KernelRmsNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t numRows,
                                uint32_t rowLength, uint32_t tileLength,
                                float eps)
    {
        totalLen_ = totalLength;
        numRows_ = numRows;
        rowLength_ = rowLength;
        tileLen_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(squareBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLength_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < numRows_; row += blockNum) {
            uint32_t rowOffset = row * rowLength_;
            uint32_t remaining = rowLength_;

            // ---- First pass: compute sum of squares ----
            float sumSq = 0.0f;
            uint32_t tileStart = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curLen);
                AscendC::LocalTensor<float> squareLocal = squareBuf_.AllocTensor<float>();
                AscendC::Mul(squareLocal, xLocal, xLocal, curLen);
                // ReduceAdd is not available; manually accumulate using vector reduction
                // Use a simple loop over elements (inefficient but correct)
                for (uint32_t i = 0; i < curLen; ++i) {
                    sumSq += squareLocal.GetValue(i);
                }
                squareBuf_.FreeTensor(squareLocal);
                inQueueX_.FreeTensor(xLocal);
                tileStart += curLen;
                remaining -= curLen;
            }

            // ---- Compute inverse RMS ----
            float rowLenF = static_cast<float>(rowLength_);
            float meanSq = sumSq / rowLenF;
            float rms = sqrt(meanSq + eps_);
            float invRms = 1.0f / rms;

            // ---- Second pass: normalize ----
            remaining = rowLength_;
            tileStart = 0;
            while (remaining > 0) {
                uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowOffset + tileStart], curLen);
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Muls(yLocal, xLocal, invRms, curLen);
                AscendC::DataCopy(yGm_[rowOffset + tileStart], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
                tileStart += curLen;
                remaining -= curLen;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> squareBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t numRows_ {0};
    uint32_t rowLength_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNormCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.numRows,
            tiling_data.rowLength, tiling_data.tileLength, tiling_data.eps);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'rms_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, num_features: int, eps: float = 1e-5):
        super(ModelNew, self).__init__()
        self.num_features = num_features
        self.eps = eps

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Reshape input to 2D (batch*s, features) and apply custom RMSNorm
        orig_shape = x.shape
        x_flat = x.view(-1, self.num_features)
        y_flat = getattr(custom_ops_lib, _FN)(x_flat, self.eps)
        y = y_flat.view(orig_shape)
        return y
