host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t D = static_cast<uint32_t>(shape.GetDim(shape.GetDimNum() - 1));

    // UB budget: input double buffer (2) + output double buffer (2) + temp for exp (1) + scalar(1) = 6 tiles
    uint32_t maxTile = UB_BUDGET_BYTES / (6 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (D < tileLen) {
        uint32_t aligned = AlignDown(D, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : D;
    }

    // blockDim: up to 24 cores, but no more than number of rows
    uint32_t B = totalLength / D;
    uint32_t blockDim = (B < 24u) ? B : 24u;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_D(D);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t D)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        D_ = D;
        B_ = totalLen_ / D_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t rowsPerBlock = (B_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = (blockIdx + 1) * rowsPerBlock;
        if (rowEnd > B_) rowEnd = B_;

        for (uint32_t row = rowStart; row < rowEnd; row++) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t base = row * D_;

        // Phase 1: compute global row max
        float rowMax = -1e38f;
        for (uint32_t offset = 0; offset < D_; offset += tileLen_) {
            uint32_t curLen = (offset + tileLen_ <= D_) ? tileLen_ : (D_ - offset);
            // Allocate and load input tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            // ReduceMax to scalar
            AscendC::LocalTensor<float> maxScalar = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceMax(maxScalar, xLocal, curLen, 1);
            float localMax = maxScalar(0);
            if (localMax > rowMax) rowMax = localMax;

            scalarQueue_.FreeTensor(maxScalar);
            inQueueX_.FreeTensor(xLocal);
        }

        // Phase 2: compute sum of exp(x - rowMax)
        float rowSum = 0.0f;
        for (uint32_t offset = 0; offset < D_; offset += tileLen_) {
            uint32_t curLen = (offset + tileLen_ <= D_) ? tileLen_ : (D_ - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
            // tmp = x - rowMax
            AscendC::Sub(tmpLocal, xLocal, rowMax, curLen, 1);
            // exp(tmp)
            AscendC::Exp(tmpLocal, tmpLocal, curLen, 1);
            // sum of exp
            AscendC::LocalTensor<float> sumScalar = scalarQueue_.AllocTensor<float>();
            AscendC::ReduceSum(sumScalar, tmpLocal, curLen, 1);
            rowSum += sumScalar(0);

            scalarQueue_.FreeTensor(sumScalar);
            tmpQueue_.FreeTensor(tmpLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        // Phase 3: compute output y = exp(x - rowMax) / rowSum
        float invSum = 1.0f / rowSum;
        for (uint32_t offset = 0; offset < D_; offset += tileLen_) {
            uint32_t curLen = (offset + tileLen_ <= D_) ? tileLen_ : (D_ - offset);
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[base + offset], curLen);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // y = x - rowMax
            AscendC::Sub(yLocal, xLocal, rowMax, curLen, 1);
            // y = exp(y)
            AscendC::Exp(yLocal, yLocal, curLen, 1);
            // y = y * invSum
            AscendC::Mul(yLocal, yLocal, invSum, curLen, 1);
            // copy to global
            AscendC::DataCopy(yGm_[base + offset], yLocal, curLen);

            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t D_ {0};
    uint32_t B_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.D);
    op.Process();
}
