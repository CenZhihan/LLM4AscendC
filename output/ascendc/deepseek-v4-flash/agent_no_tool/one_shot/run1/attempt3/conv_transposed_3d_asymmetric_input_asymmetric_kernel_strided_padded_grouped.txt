#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3d {
public:
    __aicore__ inline KernelConvTransposed3d() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                const ConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustomTilingData &tiling)
    {
        totalLen_ = tiling.totalLength;
        tileLen_ = tiling.tileLength;
        blockDim_ = tiling.blockDim;
        batch_ = tiling.batch;
        inC_ = tiling.inC;
        outC_ = tiling.outC;
        inD_ = tiling.inD;
        inH_ = tiling.inH;
        inW_ = tiling.inW;
        outD_ = tiling.outD;
        outH_ = tiling.outH;
        outW_ = tiling.outW;
        kD_ = tiling.kD;
        kH_ = tiling.kH;
        kW_ = tiling.kW;
        strideD_ = tiling.strideD;
        strideH_ = tiling.strideH;
        strideW_ = tiling.strideW;
        padD_ = tiling.padD;
        padH_ = tiling.padH;
        padW_ = tiling.padW;
        outPadD_ = tiling.outPadD;
        outPadH_ = tiling.outPadH;
        outPadW_ = tiling.outPadW;
        dilationD_ = tiling.dilationD;
        dilationH_ = tiling.dilationH;
        dilationW_ = tiling.dilationW;
        groups_ = tiling.groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        uint32_t weightSize = static_cast<uint32_t>(inC_ * (outC_/groups_) * kD_ * kH_ * kW_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, weightSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t totalOutElems = static_cast<uint32_t>(batch_ * outC_ * outD_ * outH_ * outW_);
        uint32_t elemsPerBlock = CeilDiv(totalOutElems, blockNum);
        uint32_t start = blockIdx * elemsPerBlock;
        if (start >= totalOutElems) return;
        uint32_t end = start + elemsPerBlock;
        if (end > totalOutElems) end = totalOutElems;

        uint32_t pos = start;
        while (pos < end) {
            uint32_t curLen = end - pos;
            if (curLen > tileLen_) curLen = tileLen_;

            for (uint32_t i = 0; i < curLen; ++i, ++pos) {
                uint32_t tmp = pos;
                uint32_t w_out = tmp % static_cast<uint32_t>(outW_);
                tmp /= static_cast<uint32_t>(outW_);
                uint32_t h_out = tmp % static_cast<uint32_t>(outH_);
                tmp /= static_cast<uint32_t>(outH_);
                uint32_t d_out = tmp % static_cast<uint32_t>(outD_);
                tmp /= static_cast<uint32_t>(outD_);
                uint32_t c_out = tmp % static_cast<uint32_t>(outC_);
                tmp /= static_cast<uint32_t>(outC_);
                uint32_t n = tmp;

                int32_t group = c_out / (outC_ / groups_);
                int32_t c_in_start = group * (inC_ / groups_);
                int32_t c_in_end = (group + 1) * (inC_ / groups_);
                float sum = 0.0f;

                for (int32_t kd = 0; kd < kD_; ++kd) {
                    int32_t in_d = (d_out + padD_ - kd * dilationD_) / strideD_;
                    if ((d_out + padD_ - kd * dilationD_) % strideD_ != 0) continue;
                    if (in_d < 0 || in_d >= inD_) continue;
                    for (int32_t kh = 0; kh < kH_; ++kh) {
                        int32_t in_h = (h_out + padH_ - kh * dilationH_) / strideH_;
                        if ((h_out + padH_ - kh * dilationH_) % strideH_ != 0) continue;
                        if (in_h < 0 || in_h >= inH_) continue;
                        for (int32_t kw = 0; kw < kW_; ++kw) {
                            int32_t in_w = (w_out + padW_ - kw * dilationW_) / strideW_;
                            if ((w_out + padW_ - kw * dilationW_) % strideW_ != 0) continue;
                            if (in_w < 0 || in_w >= inW_) continue;

                            int32_t c_out_g = c_out % (outC_ / groups_);
                            for (int32_t ci = c_in_start; ci < c_in_end; ++ci) {
                                float x_val = xGm_[((n * inC_ + ci) * inD_ + in_d) * inH_ * inW_ + in_h * inW_ + in_w];
                                float w_val = wGm_[((ci * (outC_ / groups_) + c_out_g) * kD_ + kd) * kH_ * kW_ + kh * kW_ + kw];
                                sum += x_val * w_val;
                            }
                        }
                    }
                }
                yGm_[pos] = sum;
            }
        }
    }

private:
    static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
    int32_t batch_;
    int32_t inC_, outC_;
    int32_t inD_, inH_, inW_;
    int32_t outD_, outH_, outW_;
    int32_t kD_, kH_, kW_;
    int32_t strideD_, strideH_, strideW_;
    int32_t padD_, padH_, padW_;
    int32_t outPadD_, outPadH_, outPadW_;
    int32_t dilationD_, dilationH_, dilationW_;
    int32_t groups_;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_strided_padded_grouped_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3d op;
    op.Init(x, weight, y, tiling_data);
    op.Process();
}
