host_operator_src="""

#include "conv2d_min_tanh_tanh_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Conv2dMinTanhTanhCustomTilingData tiling;
    // Use StorageShape to access origin shape dimensions
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::StorageShape *wShape = context->GetInputShape(1);
    uint32_t N = xShape->GetOriginShape().GetDim(0);
    uint32_t C = xShape->GetOriginShape().GetDim(1);
    uint32_t H = xShape->GetOriginShape().GetDim(2);
    uint32_t W = xShape->GetOriginShape().GetDim(3);
    uint32_t OC = wShape->GetOriginShape().GetDim(0);
    uint32_t kH = wShape->GetOriginShape().GetDim(2);
    uint32_t kW = wShape->GetOriginShape().GetDim(3);
    uint32_t Ho = H - kH + 1;
    uint32_t Wo = W - kW + 1;
    uint32_t totalPixels = N * Ho * Wo;

    uint32_t maxTile = UB_BUDGET_BYTES / (4 * 4 * 512);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tilePixels = maxTile;
    if (totalPixels < tilePixels)
        tilePixels = AlignDown(totalPixels, ALIGN_ELEMS);
    if (tilePixels == 0) tilePixels = 1;

    uint32_t numTiles = CeilDiv(totalPixels, tilePixels);
    uint32_t blockDim = numTiles;
    if (blockDim < 4) blockDim = 4;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_batch(N);
    tiling.set_inC(C);
    tiling.set_outC(OC);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_totalPixels(totalPixels);
    tiling.set_tilePixels(tilePixels);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
