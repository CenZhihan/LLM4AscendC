kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dReluHardSwishCustom {
public:
    __aicore__ inline KernelConv2dReluHardSwishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t H_out, uint32_t W_out,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilH, uint32_t dilW,
                                bool hasBias)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        N_ = N;
        Cin_ = Cin;
        Cout_ = Cout;
        H_ = H;
        W_ = W;
        K_ = K;
        H_out_ = H_out;
        W_out_ = W_out;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilH_ = dilH;
        dilW_ = dilW;
        hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_*Cin_*H_*W_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, Cout_*Cin_*K_*K_);
        if (hasBias_) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        uint32_t patchSize = K_ * K_ * Cin_;
        pipe_.InitBuffer(vecIn_, BUFFER_NUM, patchSize * sizeof(float));
        pipe_.InitBuffer(vecWeight_, BUFFER_NUM, patchSize * sizeof(float));
        pipe_.InitBuffer(vecAcc_, 1, Cout_ * sizeof(float));
        pipe_.InitBuffer(vecAct_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t start = blockIdx * tileLen_;
        uint32_t end = (start + tileLen_ > totalLen_) ? totalLen_ : start + tileLen_;

        for (uint32_t idx = start; idx < end; ++idx) {
            uint32_t tmp = idx;
            uint32_t ow = tmp % W_out_; tmp /= W_out_;
            uint32_t oh = tmp % H_out_; tmp /= H_out_;
            uint32_t cout = tmp % Cout_; tmp /= Cout_;
            uint32_t n = tmp;

            uint32_t h_start = oh * strideH_ - padH_;
            uint32_t w_start = ow * strideW_ - padW_;

            uint32_t weightOffset = cout * Cin_ * K_ * K_;
            AscendC::LocalTensor<float> wLocal = vecWeight_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, weightGm_[weightOffset], Cin_ * K_ * K_);
            vecWeight_.EnQue(wLocal);

            AscendC::LocalTensor<float> xLocal = vecIn_.AllocTensor<float>();
            float *xLocalPtr = (float *)xLocal;
            for (uint32_t c = 0; c < Cin_; ++c) {
                for (uint32_t kh = 0; kh < K_; ++kh) {
                    for (uint32_t kw = 0; kw < K_; ++kw) {
                        uint32_t h = h_start + kh * dilH_;
                        uint32_t w = w_start + kw * dilW_;
                        float val = (h >= 0 && h < H_ && w >= 0 && w < W_) ?
                                   xGm_.GetValue(n * Cin_ * H_ * W_ + c * H_ * W_ + h * W_ + w) : 0.0f;
                        xLocalPtr[c * K_ * K_ + kh * K_ + kw] = val;
                    }
                }
            }
            vecIn_.EnQue(xLocal);

            AscendC::LocalTensor<float> acc = vecAcc_.AllocTensor<float>();
            AscendC::LocalTensor<float> xDeq = vecIn_.DeQue<float>();
            AscendC::LocalTensor<float> wDeq = vecWeight_.DeQue<float>();
            AscendC::Muls(acc, xDeq, wDeq, Cin_ * K_ * K_);
            AscendC::ReduceSum(acc, acc, 1);
            float sum = acc.GetValue(0);
            if (hasBias_) {
                sum += biasGm_.GetValue(cout);
            }
            // ReLU
            sum = (sum > 0.0f) ? sum : 0.0f;
            // HardSwish: y = x * min(max(x+3,0),6) / 6
            float xp3 = sum + 3.0f;
            float clamp = (xp3 > 6.0f) ? 6.0f : (xp3 < 0.0f) ? 0.0f : xp3;
            float y = sum * clamp / 6.0f;

            yGm_.SetValue(idx, y);

            vecAcc_.FreeTensor(acc);
            vecIn_.FreeTensor(xDeq);
            vecWeight_.FreeTensor(wDeq);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> vecIn_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> vecWeight_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> vecAcc_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> vecAct_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};

    uint32_t N_, Cin_, Cout_, H_, W_, K_;
    uint32_t H_out_, W_out_;
    uint32_t strideH_, strideW_, padH_, padW_, dilH_, dilW_;
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluHardSwishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.K,
            tiling_data.H_out, tiling_data.W_out,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.hasBias);
    op.Process();
}
