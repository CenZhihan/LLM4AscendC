#include "frobenius_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    FrobeniusNormCustomTilingData tiling;
    const uint32_t totalLength = static_cast<uint32_t>(
        context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Single block – no cross‑block sync needed
    context->SetBlockDim(1);

    // UB budget: we need 2 tiles (input + output) + 1 temp for square + Sqrt
    // Elements per tile: 3 * tileLen floats => 12 * tileLen bytes
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *x_shape = context->GetInputShape(0);
    *context->GetOutputShape(0) = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class FrobeniusNormCustom : public OpDef {
public:
    explicit FrobeniusNormCustom(const char* name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFrobeniusNormCustom {
public:
    __aicore__ inline KernelFrobeniusNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileLen_ * sizeof(float)); // extra for squaring & sqrt
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        // ----- First pass: compute sum of squares -----
        float totalSum = 0.0f;
        uint32_t offset = 0;
        while (offset < totalLen_) {
            uint32_t len = totalLen_ - offset;
            if (len > tileLen_) len = tileLen_;

            // Load input tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);

            // Square into temporary
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(tmp, xLocal, xLocal, len);

            // Accumulate manually (no cross‑block reduction needed)
            for (uint32_t i = 0; i < len; ++i) {
                totalSum += tmp(i);
            }

            inQueueX_.FreeTensor(xLocal);
            tmpQueue_.FreeTensor(tmp);
            offset += tileLen_;
        }

        // ----- Compute norm: sqrt(totalSum) -----
        // Use a local tensor of size 1 for sqrt
        AscendC::LocalTensor<float> normTensor = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> oneTensor = tmpQueue_.AllocTensor<float>();
        oneTensor.SetValue(0, totalSum);
        AscendC::Sqrt(normTensor, oneTensor, 1);
        float norm = normTensor.GetValue(0);
        tmpQueue_.FreeTensor(normTensor);
        tmpQueue_.FreeTensor(oneTensor);

        // Compute 1/norm for multiplication
        float invNorm = (norm > 1e-7f) ? 1.0f / norm : 0.0f;

        // ----- Second pass: divide by norm (multiply by invNorm) -----
        offset = 0;
        while (offset < totalLen_) {
            uint32_t len = totalLen_ - offset;
            if (len > tileLen_) len = tileLen_;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Muls(yLocal, xLocal, invNorm, len);

            AscendC::DataCopy(yGm_[offset], yLocal, len);

            inQueueX_.FreeTensor(xLocal);
            outQueueY_.FreeTensor(yLocal);
            offset += tileLen_;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void frobenius_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFrobeniusNormCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
