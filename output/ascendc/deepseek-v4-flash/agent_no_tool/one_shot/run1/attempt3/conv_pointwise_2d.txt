#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    uint32_t N  = x_shape->GetDim(0);
    uint32_t C  = x_shape->GetDim(1);
    uint32_t H  = x_shape->GetDim(2);
    uint32_t W  = x_shape->GetDim(3);
    uint32_t Cout = w_shape->GetDim(0);
    uint32_t Cin  = w_shape->GetDim(1);

    uint32_t M = N * H * W;
    uint32_t totalLength = M * Cout;

    // TileCout: number of output channels processed per iteration
    // We want to fit weight (Cout * Cin) + input (Cin) + output (tileCout) in UB
    // weight size: Cout * Cin * 4 bytes
    // input size: Cin * 4
    // output size: tileCout * 4
    // temporary for dot product: Cin * 4 (for vector mul) + 1 * 4 (for reduce sum)
    // total: (Cout*Cin + Cin + tileCout + Cin + 1) * 4
    // For simplicity, choose tileCout such that total <= UB_BUDGET_BYTES
    uint32_t weightBytes = Cout * Cin * sizeof(float);
    uint32_t inputBytes = Cin * sizeof(float);
    uint32_t outputBytes = Cout * sizeof(float); // max
    uint32_t tempBytes = (Cin + 1) * sizeof(float);
    uint32_t available = UB_BUDGET_BYTES > weightBytes ? UB_BUDGET_BYTES - weightBytes : 0;
    uint32_t tileCout = 1;
    if (available > 0) {
        uint32_t maxTileCout = (available - inputBytes - tempBytes) / sizeof(float);
        if (maxTileCout > Cout) maxTileCout = Cout;
        if (maxTileCout > 16) maxTileCout = 16; // conservative
        tileCout = maxTileCout > 0 ? maxTileCout : 1;
    }

    // Block dimension: assign contiguous spatial positions
    uint32_t blockDim = 8; // default
    if (M > 0) {
        uint32_t numTiles = CeilDiv(M, 1); // each block handles one or more positions
        blockDim = numTiles;
        if (blockDim < 8) blockDim = 8;
        if (blockDim > 24) blockDim = 24;
    }

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_M(M);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_tileCout(tileCout);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t M, uint32_t Cin,
                                uint32_t Cout, uint32_t tileCout,
                                uint32_t blockDim)
    {
        totalLen_ = totalLength;
        M_ = M;
        Cin_ = Cin;
        Cout_ = Cout;
        tileCout_ = tileCout;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M_ * Cin_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, Cout_ * Cin_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Initialize queues
        pipe_.InitBuffer(weightQueue_, BUFFER_NUM, Cout_ * Cin_ * sizeof(float));
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, Cin_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileCout_ * sizeof(float));
        pipe_.InitBuffer(mulTmp_, 1, Cin_ * sizeof(float));
        pipe_.InitBuffer(sumTmp_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0) return;

        // Load weight into UB (double buffered: load once into weightQueue_)
        AscendC::LocalTensor<float> weightLocal = weightQueue_.AllocTensor<float>();
        AscendC::DataCopy(weightLocal, weightGm_, Cout_ * Cin_);
        weightQueue_.EnQue(weightLocal);

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block handles a continuous range of spatial positions
        uint32_t spatialPerBlock = (M_ + blockNum - 1) / blockNum;
        uint32_t startS = blockIdx * spatialPerBlock;
        uint32_t endS = startS + spatialPerBlock;
        if (endS > M_) endS = M_;
        if (startS >= endS) return;

        for (uint32_t s = startS; s < endS; ++s) {
            // Load input for spatial position s
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[s * Cin_], Cin_);
            inQueueX_.EnQue(xLocal);

            // Process output channels in tiles of tileCout_
            uint32_t ocBase = 0;
            while (ocBase < Cout_) {
                uint32_t ocCount = (ocBase + tileCout_ > Cout_) ? (Cout_ - ocBase) : tileCout_;
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::SetZero(yLocal, ocCount);

                // For each output channel in this tile, compute dot product with input
                for (uint32_t oc = 0; oc < ocCount; ++oc) {
                    uint32_t actualOc = ocBase + oc;
                    // Load weight row for this output channel
                    AscendC::LocalTensor<float> wRow = weightLocal[actualOc * Cin_]; // this is not valid; need to use DataCopy
                    // Actually we need to copy from weightLocal to a local tensor
                    // Since weightLocal is already in UB, we can directly use pointer arithmetic but must ensure correct access.
                    // In AscendC, we can get a pointer to a sub-tensor: weightLocal[actualOc * Cin_] as a Tensor.
                    // However, the syntax is not straightforward.
                    // We'll use a simpler approach: copy the row into mulTmp_.
                    AscendC::LocalTensor<float> wRowCopy = mulTmp_.AllocTensor<float>();
                    AscendC::DataCopy(wRowCopy, weightLocal[actualOc * Cin_], Cin_);

                    // Elementwise multiply
                    AscendC::Mul(mulTmp_, xLocal, wRowCopy, Cin_);

                    // Reduce sum to get dot product
                    AscendC::LocalTensor<float> sumScalar = sumTmp_.AllocTensor<float>();
                    AscendC::ReduceSum(sumScalar, mulTmp_, Cin_);
                    yLocal[oc] = sumScalar[0];
                    sumTmp_.FreeTensor(sumScalar);
                    mulTmp_.FreeTensor(wRowCopy);
                }

                // Write output tile to global memory
                AscendC::DataCopy(yGm_[s * Cout_ + ocBase], yLocal, ocCount);

                outQueueY_.FreeTensor(yLocal);
                ocBase += tileCout_;
            }

            // Free input tensor
            inQueueX_.FreeTensor(inQueueX_.DeQue<float>());
        }

        // Free weight tensor
        weightQueue_.FreeTensor(weightQueue_.DeQue<float>());
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> weightQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> mulTmp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumTmp_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t M_ {0};
    uint32_t Cin_ {0};
    uint32_t Cout_ {0};
    uint32_t tileCout_ {16};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.totalLength, tiling_data.M, tiling_data.Cin, tiling_data.Cout,
            tiling_data.tileCout, tiling_data.blockDim);
    op.Process();
}
