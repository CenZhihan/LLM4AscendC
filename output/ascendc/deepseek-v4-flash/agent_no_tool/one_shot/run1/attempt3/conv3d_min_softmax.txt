kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv3dMinSoftmaxCustom {
public:
    __aicore__ inline KernelConv3dMinSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                const Conv3dMinSoftmaxCustomTilingData &tiling)
    {
        tiling_ = tiling;
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        bGm_.SetGlobalBuffer((__gm__ float*)bias);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Allocate one large local buffer for intermediate computations
        uint32_t maxSize = std::max({
            tiling_.K_D * tiling_.K_H * tiling_.K_W * tiling_.C_in * sizeof(float),
            tiling_.C_out * tiling_.C_in * tiling_.K_D * tiling_.K_H * tiling_.K_W * sizeof(float),
            tiling_.C_out * sizeof(float),
            tiling_.D_out * tiling_.C_out * sizeof(float)
        });
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, maxSize);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tiling_.C_out * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, maxSize);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t totalSpatial = tiling_.N * tiling_.H_out * tiling_.W_out;
        uint32_t stride = blockNum;
        uint32_t start = blockIdx;
        if (start >= totalSpatial) return;

        for (uint32_t idx = start; idx < totalSpatial; idx += stride)
        {
            uint32_t n = idx / (tiling_.H_out * tiling_.W_out);
            uint32_t rem = idx % (tiling_.H_out * tiling_.W_out);
            uint32_t h = rem / tiling_.W_out;
            uint32_t w = rem % tiling_.W_out;

            // Allocate weight and bias local tensors (reused across depths)
            AscendC::LocalTensor<float> wLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_, tiling_.C_out * tiling_.C_in * tiling_.K_D * tiling_.K_H * tiling_.K_W);
            AscendC::LocalTensor<float> bLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(bLocal, bGm_, tiling_.C_out);

            // Allocate conv output local tensor (D_out * C_out)
            AscendC::LocalTensor<float> convLocal = tmpQueue_.AllocTensor<float>();
            uint32_t convLen = tiling_.D_out * tiling_.C_out;
            AscendC::Fill(convLocal, 0.0f, convLen);

            // Loop over depths
            for (uint32_t d_out = 0; d_out < tiling_.D_out; d_out++)
            {
                uint32_t d_in_start = d_out;
                uint32_t h_in_start = h;
                uint32_t w_in_start = w;

                // Load input window (C_in * K_D * K_H * K_W) into local
                AscendC::LocalTensor<float> xLocal = tmpQueue_.AllocTensor<float>();
                uint32_t windowLen = tiling_.C_in * tiling_.K_D * tiling_.K_H * tiling_.K_W;
                for (uint32_t i = 0; i < windowLen; i++)
                {
                    uint32_t ci = i / (tiling_.K_D * tiling_.K_H * tiling_.K_W);
                    uint32_t rem1 = i % (tiling_.K_D * tiling_.K_H * tiling_.K_W);
                    uint32_t kd = rem1 / (tiling_.K_H * tiling_.K_W);
                    uint32_t rem2 = rem1 % (tiling_.K_H * tiling_.K_W);
                    uint32_t kh = rem2 / tiling_.K_W;
                    uint32_t kw = rem2 % tiling_.K_W;

                    uint32_t d_in = d_in_start + kd;
                    uint32_t h_in = h_in_start + kh;
                    uint32_t w_in = w_in_start + kw;
                    uint32_t globalOffset = ((n * tiling_.C_in + ci) * tiling_.D_in + d_in) * tiling_.H_in * tiling_.W_in
                                            + h_in * tiling_.W_in + w_in;
                    xLocal(i) = xGm_[globalOffset];
                }

                // For each output channel, compute dot product and accumulate into convLocal
                for (uint32_t co = 0; co < tiling_.C_out; co++)
                {
                    float sum = 0.0f;
                    for (uint32_t i = 0; i < windowLen; i++)
                    {
                        uint32_t weightIdx = co * windowLen + i;
                        sum += wLocal(weightIdx) * xLocal(i);
                    }
                    sum += bLocal(co);
                    convLocal(co * tiling_.D_out + d_out) = sum;
                }

                tmpQueue_.FreeTensor(xLocal);
            }

            // Compute min over depth for each channel
            AscendC::LocalTensor<float> minLocal = outQueueY_.AllocTensor<float>();
            for (uint32_t co = 0; co < tiling_.C_out; co++)
            {
                float minVal = convLocal(co * tiling_.D_out);
                for (uint32_t d = 1; d < tiling_.D_out; d++)
                {
                    if (convLocal(co * tiling_.D_out + d) < minVal)
                        minVal = convLocal(co * tiling_.D_out + d);
                }
                minLocal(co) = minVal;
            }

            // Compute softmax over channels
            float maxVal = minLocal(0);
            for (uint32_t co = 1; co < tiling_.C_out; co++)
            {
                if (minLocal(co) > maxVal) maxVal = minLocal(co);
            }
            float sum = 0.0f;
            AscendC::LocalTensor<float> softmaxLocal = tmpQueue_.AllocTensor<float>();
            for (uint32_t co = 0; co < tiling_.C_out; co++)
            {
                float val = AscendC::Exp(minLocal(co) - maxVal);
                softmaxLocal(co) = val;
                sum += val;
            }
            float invSum = 1.0f / sum;
            for (uint32_t co = 0; co < tiling_.C_out; co++)
            {
                yGm_[(n * tiling_.C_out + co) * tiling_.H_out * tiling_.W_out + h * tiling_.W_out + w] = softmaxLocal(co) * invSum;
            }

            // Free intermediate tensors
            tmpQueue_.FreeTensor(convLocal);
            tmpQueue_.FreeTensor(softmaxLocal);
            inQueueX_.FreeTensor(wLocal);
            inQueueX_.FreeTensor(bLocal);
            outQueueY_.FreeTensor(minLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    Conv3dMinSoftmaxCustomTilingData tiling_;
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMinSoftmaxCustom op;
    op.Init(x, weight, bias, y, tiling_data);
    op.Process();
}
