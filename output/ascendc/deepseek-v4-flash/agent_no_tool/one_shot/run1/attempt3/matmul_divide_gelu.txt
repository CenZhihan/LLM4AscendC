namespace optiling {

static constexpr uint32_t BLOCK_M = 128;
static constexpr uint32_t BLOCK_N = 128;
static constexpr uint32_t BLOCK_K = 128;
static constexpr uint32_t ALIGN = 32;

static inline uint32_t AlignUp(uint32_t x, uint32_t a) {
    return ((x + a - 1) / a) * a;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MatmulDivideGeluCustomTilingData tiling;
    const auto x_shape = context->GetInputShape(0)->GetOriginShape();
    const auto w_shape = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = x_shape.GetDim(0);
    uint32_t K = x_shape.GetDim(1);
    uint32_t N = w_shape.GetDim(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *divisor = attrs->GetAttrPointer<float>(0);

    uint32_t blockM = BLOCK_M;
    uint32_t blockN = BLOCK_N;
    uint32_t blockK = BLOCK_K;

    blockM = AlignUp(blockM, ALIGN);
    blockN = AlignUp(blockN, ALIGN);
    blockK = AlignUp(blockK, ALIGN);

    if (blockM > M) blockM = AlignDown(M, ALIGN);
    if (blockM == 0) blockM = M;
    if (blockN > N) blockN = AlignDown(N, ALIGN);
    if (blockN == 0) blockN = N;
    if (blockK > K) blockK = AlignDown(K, ALIGN);
    if (blockK == 0) blockK = K;

    tiling.set_totalM(M);
    tiling.set_totalN(N);
    tiling.set_totalK(K);
    tiling.set_blockM(blockM);
    tiling.set_blockN(blockN);
    tiling.set_blockK(blockK);
    tiling.set_divisor(*divisor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    static constexpr uint32_t cubeNum = 4;
    context->SetBlockDim(cubeNum);

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulDivideGeluCustom {
public:
    __aicore__ inline KernelMatmulDivideGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalM, uint32_t totalN, uint32_t totalK,
                                uint32_t blockM, uint32_t blockN, uint32_t blockK,
                                float divisor) {
        totalM_ = totalM;
        totalN_ = totalN;
        totalK_ = totalK;
        blockM_ = blockM;
        blockN_ = blockN;
        blockK_ = blockK;
        divisor_ = divisor;
        invDivisor_ = 1.0f / divisor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalM_ * totalK_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, totalN_ * totalK_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, totalN_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalM_ * totalN_);

        // For Cube matmul, we need to set up Matmul objects with proper dimensions.
        // We will process one block (M1xK1 * K1xN1) at a time.
        // Not using queues for Cube; Matmul handles its own buffers.
        // But we still need queues for bias and activation fusion.
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, blockN_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, blockM_ * blockN_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalM_ == 0 || totalN_ == 0 || totalK_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        // Each block processes a subset of M tiles
        for (uint32_t m0 = blockIdx * blockM_; m0 < totalM_; m0 += blockNum * blockM_) {
            uint32_t curM = blockM_;
            if (m0 + curM > totalM_) curM = totalM_ - m0;

            // Process all N tiles for this set of M rows
            for (uint32_t n0 = 0; n0 < totalN_; n0 += blockN_) {
                uint32_t curN = blockN_;
                if (n0 + curN > totalN_) curN = totalN_ - n0;

                // Load bias for current N block
                AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[n0], curN);

                // Prepare output buffer for this block
                AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();

                // Perform Matmul: outLocal = x[m0:m0+curM, :] * w[n0:n0+curN, :]^T?
                // Actually w shape is (N, K), so we need w[n0:n0+curN, :] (K is inner dim)
                // Using Cube: Matmul(x[m0, :, curK], w[n0, :, curK]) -> C(curM, curN)
                // We need to handle tiling over K as well. But for simplicity, if blockK = totalK, we do one matmul.
                // If not, we need to accumulate over K tiles.
                // Let's assume blockK = totalK for now (since it's set to at least 1, but we should handle K tiling).
                // Actually we must handle K tiling, but for brevity we can do a single matmul if blockK == totalK.
                // For maximum generality, we implement K-loop with accumulation.
                // Initialize outLocal with zeros (to accumulate)
                AscendC::LocalTensor<float> zeros = outLocal; // Use same memory, but we need to clear it.
                AscendC::SetZero(zeros, curM * curN);

                bool firstK = true;
                for (uint32_t k0 = 0; k0 < totalK_; k0 += blockK_) {
                    uint32_t curK = blockK_;
                    if (k0 + curK > totalK_) curK = totalK_ - k0;

                    // Prepare xBlock: curM x curK
                    // Prepare wBlock: curN x curK
                    // Need to copy to local buffers for Cube. We can use Matmul's built-in data copy.
                    // The Matmul object in AscendC can take GlobalTensor references.
                    // However, for simplicity, we'll manually copy tiles if needed.
                    // A more efficient approach: use Matmul with global tensors directly, but that requires contiguous sub-tensors.
                    // Since our data is row-major and w is also row-major (outputs, inputs), we can use SetGlobalBuffer for sub-sections.
                    // Actually Matmul expects inputs in specific formats; for simplicity, we'll use the Pipe and DataCopy for each tile.
                    // Allocate local tensors for x and w tiles (using manual buffers)
                    AscendC::LocalTensor<float> xTile;
                    AscendC::LocalTensor<float> wTile;
                    // We'll allocate from extra queues or use temporary buffers.
                    // For now, we use inQueueX_ (not used for anything else) for x tile.
                    // But we haven't inited inQueueX_ - let's do it on the fly? 
                    // To avoid complexity, we can use a single temporary buffer per iteration.
                    // Actually we can use AscendC::LocalTensor with AllocTensor from a queue.
                    // Let's define a small local buffer for x and w inside the loop.
                    // Since this is a fix, we keep it simple: reuse a single queue for both.
                    // BAD: This is messy. Instead, use Matmul with sub-global tensors by adjusting pointers.
                    // The Matmul API can accept GlobalTensor with offsets.
                    // Example: Matmul matmul; matmul.SetTensorA(xGm_[m0 * totalK_ + k0]); etc.
                    // But the Matmul object expects contiguous blocks; our sub-blocks are contiguous in memory.
                    // Therefore we can use:
                    //   matmul.SetTensorA(xGm_[m0 * totalK_ + k0]); // shape (curM, curK)
                    //   matmul.SetTensorB(wGm_[n0 * totalK_ + k0]); // shape (curN, curK) - but w is (N, K), so row-major with K as inner dim.
                    // However, matmul computes C = A * B^T in typical AscendC? Need to check documentation.
                    // Actually, the Cube in AscendC typically computes C = A * B, where A is MxK and B is KxN.
                    // Our w is stored as (N, K), which is transposed. So we need to treat w as B^T.
                    // That is, C = A * B^T, where A = x (MxK), B = w (NxK) -> C = x * w^T (MxN).
                    // This matches. So we can use Matmul(A, B, C) where B is w with shape (N, K) but treated as B^T.
                    // In AscendC MulMatmul? Actually the Cube has a specific API: `Matmul` class with methods.
                    // For simplicity, we can use the vectorized approach only for the first block of K and avoid loop?

                    // That will fix the errors without needing Cube.
                    // We'll use the following:
                    //   for each m, n:
                    //        dot = 0
                    //        for each k:
                    //            dot += x[m][k] * w[n][k]

                    // We'll do it correctly this time.

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalM_ {0};
    uint32_t totalN_ {0};
    uint32_t totalK_ {0};
    uint32_t blockM_ {0};
    uint32_t blockN_ {0};
    uint32_t blockK_ {0};
    float divisor_ {1.0f};
    float invDivisor_ {1.0f};

    KernelMatmulDivideGeluCustom op;
    op.Init(x, w, b, y,

}
