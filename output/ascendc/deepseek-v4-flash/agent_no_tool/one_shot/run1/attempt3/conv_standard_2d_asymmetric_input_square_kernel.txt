kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dAsymmetricInputSquareKernel {
public:
    __aicore__ inline KernelConv2dAsymmetricInputSquareKernel() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR tiling) {
        GET_TILING_DATA(data, tiling);
        totalOutputLength_ = data.totalOutputLength;
        tileOutputLength_ = data.tileOutputLength;
        N_ = data.N;
        C_ = data.C;
        H_ = data.H;
        W_ = data.W;
        O_ = data.O;
        Kh_ = data.Kh;
        Kw_ = data.Kw;
        Ho_ = data.Ho;
        Wo_ = data.Wo;
        strideH_ = data.strideH;
        strideW_ = data.strideW;
        padH_ = data.padH;
        padW_ = data.padW;
        dilationH_ = data.dilationH;
        dilationW_ = data.dilationW;
        groups_ = data.groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)w);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        uint32_t tileBytes = tileOutputLength_ * sizeof(float);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, C_ * Kh_ * Kw_ * sizeof(float)); // input patch
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, O_ * C_ * Kh_ * Kw_ * sizeof(float)); // weight (simplified: load all per tile)
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileBytes);
        // Allocate temporary buffer for accumulation
        pipe_.InitBuffer(tmpQueue_, 1, tileBytes);
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        // Tile over output elements
        uint32_t startIdx = blockIdx * tileOutputLength_;
        if (startIdx >= totalOutputLength_) return;
        uint32_t rem = totalOutputLength_ - startIdx;
        uint32_t curTile = (rem < tileOutputLength_) ? rem : tileOutputLength_;

        // For simplicity, we load entire weight for all output channels once.
        // In production, we would tile over output channels.
        // Load weight tensor into local: shape (O, C, Kh, Kw)
        // We'll allocate weight buffer and copy once.
        // Pipeline: enqueue weight into inQueueW_, then dequeue later for compute.
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, O_ * C_ * Kh_ * Kw_);
        inQueueW_.EnQue(wLocal);

        // Process each output element in this tile
        for (uint32_t i = 0; i < curTile; ++i) {
            uint32_t globalIdx = startIdx + i; // linear index in output
            // Decompose into n, o, h, w
            uint32_t n = globalIdx / (O_ * Ho_ * Wo_);
            uint32_t rem2 = globalIdx % (O_ * Ho_ * Wo_);
            uint32_t o = rem2 / (Ho_ * Wo_);
            uint32_t rem3 = rem2 % (Ho_ * Wo_);
            uint32_t h_out = rem3 / Wo_;
            uint32_t w_out = rem3 % Wo_;

            // Compute input patch coordinates
            int32_t h_in_start = static_cast<int32_t>(h_out) * strideH_ - static_cast<int32_t>(padH_);
            int32_t w_in_start = static_cast<int32_t>(w_out) * strideW_ - static_cast<int32_t>(padW_);

            // Load input patch for this output position
            AscendC::LocalTensor<float> xPatch = inQueueX_.AllocTensor<float>();
            uint32_t patchSize = C_ * Kh_ * Kw_;
            // Copy input patch from global memory (ignoring padding - simplified: assume valid)
            // In practice we need to handle boundaries, but for this example we assume no padding needed.
            for (uint32_t c = 0; c < C_; ++c) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        int32_t h_in = h_in_start + static_cast<int32_t>(kh * dilationH_);
                        int32_t w_in = w_in_start + static_cast<int32_t>(kw * dilationW_);
                        // Boundary check: if out of bounds, set zero
                        float val = 0.0f;
                        if (h_in >= 0 && h_in < static_cast<int32_t>(H_) && w_in >= 0 && w_in < static_cast<int32_t>(W_)) {
                            uint32_t idx = (n * C_ * H_ * W_) + (c * H_ * W_) + (h_in * W_) + w_in;
                            val = xGm_.GetValue(idx);
                        }
                        xPatch.SetValue((c * Kh_ * Kw_) + (kh * Kw_) + kw, val);
                    }
                }
            }
            inQueueX_.EnQue(xPatch);

            // Compute dot product for this output element
            // Dequeue weight and input patch
            AscendC::LocalTensor<float> wLocalDeq = inQueueW_.DeQue<float>();
            AscendC::LocalTensor<float> xPatchDeq = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // Initialize accumulator to 0
            yLocal.SetValue(0, 0.0f);

            // Perform vector multiply and reduce sum
            // We'll use a temporary local tensor for product
            AscendC::LocalTensor<float> prod = tmpQueue_.AllocTensor<float>();

            // For the specific weight slice for output channel o
            uint32_t weightOffset = o * C_ * Kh_ * Kw_;
            AscendC::Mul(prod, xPatchDeq, wLocalDeq[weightOffset], patchSize);
            AscendC::ReduceAdd(yLocal, prod, patchSize);

            tmpQueue_.FreeTensor(prod);
            inQueueW_.FreeTensor(wLocalDeq);
            inQueueX_.FreeTensor(xPatchDeq);

            // Write result to output global memory
            outQueueY_.EnQue(yLocal);
            AscendC::LocalTensor<float> yLocalOut = outQueueY_.DeQue<float>();
            yGm_.SetValue(globalIdx, yLocalOut.GetValue(0));
            outQueueY_.FreeTensor(yLocalOut);
        }

        // Free weight local buffer
        AscendC::LocalTensor<float> wFree = inQueueW_.DeQue<float>();
        inQueueW_.FreeTensor(wFree);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutputLength_ {0};
    uint32_t tileOutputLength_ {0};
    uint32_t N_, C_, H_, W_, O_, Kh_, Kw_, Ho_, Wo_;
    uint32_t strideH_, strideW_, padH_, padW_, dilationH_, dilationW_, groups_;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dAsymmetricInputSquareKernel op;
    op.Init(x, w, y, tiling);
    op.Process();
}
