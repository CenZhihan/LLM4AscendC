constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom {
public:
    __aicore__ inline KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias_opt,
                                GM_ADDR y, uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C_in, uint32_t D_in, uint32_t H_in, uint32_t W_in,
                                uint32_t C_out, uint32_t kD, uint32_t kH, uint32_t kW) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N; C_in_ = C_in; D_in_ = D_in; H_in_ = H_in; W_in_ = W_in;
        C_out_ = C_out; kD_ = kD; kH_ = kH; kW_ = kW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        biasGm_.SetGlobalBuffer((__gm__ float*)(conv_bias_opt ? conv_bias_opt : nullptr));
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // Workspace for reduction: ceil(tileLen/32) uint32_t elements
        pipe_.InitBuffer(wsQueue_, 1, ((tileLen_ + 31) / 32) * sizeof(uint32_t));
        // Scalar buffer for reduction result
        pipe_.InitBuffer(maxScalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        // Copy input tile
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[base], curLen);
        inQueueX_.EnQue(xLocal);

        // Dequeue and proceed
        AscendC::LocalTensor<float> tLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<uint32_t> ws = wsQueue_.AllocTensor<uint32_t>();
        AscendC::LocalTensor<float> maxScalar = maxScalarQueue_.AllocTensor<float>();

        // Compute max of the tile
        AscendC::ReduceMax<float, AscendC::ReductionPattern::AUTO>(maxScalar, tLocal, curLen, ws);
        float localMax = maxScalar.GetValue(0);

        // Multiply all elements by max
        AscendC::Muls(yLocal, tLocal, localMax, curLen);

        // Clamp to [0, 1] using Max and Min
        AscendC::Max(yLocal, yLocal, 0.0f, curLen);
        AscendC::Min(yLocal, yLocal, 1.0f, curLen);

        // Write output
        AscendC::DataCopy(yGm_[base], yLocal, curLen);

        wsQueue_.FreeTensor(ws);
        maxScalarQueue_.FreeTensor(maxScalar);
        outQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
