static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxReductionOverADimensionCustomTilingData tiling;
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    const int32_t nDims = static_cast<int32_t>(shape.GetDimNum());
    const uint32_t *dims = shape.GetDims();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrInt(0);
    if (dim < 0) dim += nDims;
    if (dim < 0 || dim >= nDims) return ge::GRAPH_FAILED;

    int32_t redDim = static_cast<int32_t>(dim);
    uint32_t outer = 1;
    for (int32_t i = 0; i < redDim; ++i) outer *= dims[i];
    uint32_t reductionLen = dims[redDim];
    uint32_t innerStride = 1;
    for (int32_t i = redDim + 1; i < nDims; ++i) innerStride *= dims[i];
    uint32_t totalOuter = outer * innerStride;  // number of independent reduction sequences

    // Tiling: each block processes a chunk of outer sequences
    uint32_t blockDim = 8;  // default
    if (totalOuter >= 24) blockDim = 24;
    else if (totalOuter >= 16) blockDim = 16;
    else if (totalOuter >= 8) blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_outerElements(totalOuter);
    tiling.set_reductionLength(reductionLen);
    tiling.set_innerStride(innerStride);
    tiling.set_blockDim(blockDim);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const auto &inShape = context->GetInputShape(0)->GetOriginShape();
    const int32_t nDims = inShape.GetDimNum();
    const uint32_t *dims = inShape.GetDims();

    // Determine dim from attr
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dim = attrs->GetAttrInt(0);
    if (dim < 0) dim += nDims;
    if (dim < 0 || dim >= nDims) return GRAPH_FAILED;

    // Build output shape: remove the reduced dimension
    std::vector<int64_t> outDims;
    for (int64_t i = 0; i < nDims; ++i) {
        if (i != dim) outDims.push_back(static_cast<int64_t>(dims[i]));
    }
    gert::Shape outShape;
    outShape.SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) {
        outShape.SetDim(static_cast<int32_t>(i), outDims[i]);
    }
    *context->GetOutputShape(0) = outShape;
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxReduction {
public:
    __aicore__ inline KernelMaxReduction() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerElements, uint32_t reductionLength,
                                uint32_t innerStride)
    {
        outerElems_ = outerElements;
        redLen_ = reductionLength;
        innerStride_ = innerStride;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Pre-allocate UB buffer for one reduction line + optional prefetch
        const uint32_t bufSize = redLen_ * sizeof(float);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, bufSize);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outerElems_ == 0 || redLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stride = (outerElems_ + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * stride;
        uint32_t end = (start + stride > outerElems_) ? outerElems_ : start + stride;

        for (uint32_t i = start; i < end; ++i) {
            // Compute offset for this reduction sequence
            uint32_t baseOffset = i;
            // Reduction loops over elements separated by innerStride_ * redLen_? Actually:
            // For a given outer index i, the positions are:
            // idx = base + j * innerStride_ where j from 0 to redLen_-1
            // base = i? Wait: need to map i properly.
            // We have outer = product of dims before reduction dim; innerStride = product after.
            // The linear index for the start of this sequence is i * redLen_ * innerStride_? No.
            // Better: treat i as a sequential index for the outer dimensions concatenated.
            // We'll compute using a mapping, but for simplicity and correctness, we can load the entire
            // reduction sequence address by scanning with a loop and using GM reads.
            // We'll do a simple loop over redLen_ elements, compute global offset each time.
            // This may be slow but works for arbitrary dimensions.
            // We'll prefetch the whole line into UB for better performance.
            // Let's compute the start offset for the first element of this reduction line.
            // Actually each reduction sequence is separated by innerStride_ elements.
            // The outer index i is from outerElements (product of outer dims).
            // But the reduction sequences are interleaved: for dimension d, the element at position
            // (i // innerStride_ , i % innerStride_) maybe? Simpler: we can treat the outer loop as a
            // flattened index over the first (outerElements) positions in the output? No, output size is outerElements.
            // We'll compute by reconstructing multidimensional indices via loops.
            // For kernel simplicity, use a scalar approach: iterate over redLen_ elements by linear index.
            // The linear index of the first element of this reduction line is:
            // (i / innerStride_) * (redLen_ * innerStride_) + (i % innerStride_) ? That's messy.
            // Instead, we can store the base address of each reduction line using tiling: we have outerElements
            // and innerStride; the step between consecutive reduction lines is innerStride_ * redLen_? No.
            // Actually for dimension d (0-indexed), the shape is [d0, d1, ..., d_{d-1}, redLen, d_{d+1}, ...].
            // The outer dims product = d0*d1*...*d_{d-1} * d_{d+1}*... = outerElements.
            // The innerStride = d_{d+1}*... (product after reduction).
            // So the tensor can be viewed as (outerElements / innerStride) blocks of size (redLen * innerStride).
            // Within each block, the reduction dimension is contiguous? Actually for dimension d, the loop order is:
            // for each outer element (using indices before and after), the reduction dim elements are strided by innerStride.
            // The linear index for a given outer element (i) and reduction step j is:
            // start = (i // innerStride) * (redLen * innerStride) + (i % innerStride) + j * innerStride.
            // Let's verify: i ranges 0..outerElements-1, innerStride is product of dims after reduction.
            // For a given i, the position in the last dims is i % innerStride, and the position in the first outer dims
            // is i // innerStride. Then the base of the block for that outer combination is (i // innerStride) * redLen * innerStride.
            // The reduction element at step j is base + (i % innerStride) + j * innerStride.
            // That gives the correct linear index.
            // So we can compute base and run loop.
            // For performance, we will load the entire reduction line into UB (size redLen) using a single DataCopy?
            // But DataCopy requires contiguous memory. The reduction dimension elements are not contiguous if
            // innerStride > 1 (they are spaced by innerStride). So we must load elements individually or use multi-DMA.
            // For simplicity, we will perform a scalar loop with repeated GM reads. Given that reduction length is typically
            // small (one dimension), this is acceptable. We'll also use a double buffer for the reduction line if needed.
            // To optimize, we can load the entire block of size redLen*innerStride and then pick strided elements? That wastes.
            // For now, use a simple loop with single float loads and comparisons.
            // We'll write a single reduction function that works for arbitrary stride.
            // Since the kernel is for demonstration, scalar approach is okay.

            // Compute the starting offsets
            uint32_t outerBlockIdx = i / innerStride_;
            uint32_t offsetInBlock = i % innerStride_;
            uint32_t baseIndex = outerBlockIdx * redLen_ * innerStride_ + offsetInBlock;

            // Initialize max with first element
            float maxVal = xGm_.GetValue(baseIndex);  // careful: GetValue returns float
            for (uint32_t j = 1; j < redLen_; ++j) {
                float val = xGm_.GetValue(baseIndex + j * innerStride_);
                if (val > maxVal) maxVal = val;
            }
            yGm_.SetValue(i, maxVal);
        }
    }

private:
    AscendC::TPipe pipe_;
    // Not using queues in this simplified version; just direct GM access
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;  // unused but required for compilation
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outerElems_ {0};
    uint32_t redLen_ {0};
    uint32_t innerStride_ {1};
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReduction op;
    op.Init(x, y, tiling_data.outerElements, tiling_data.reductionLength, tiling_data.innerStride);
    op.Process();
}
