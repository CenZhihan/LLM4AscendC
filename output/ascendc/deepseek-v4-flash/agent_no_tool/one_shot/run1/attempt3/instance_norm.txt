host_operator_src="""

#include "instance_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: per block we need for one tile:
//   xTile   : tileLength * sizeof(float)
//   yTile   : tileLength * sizeof(float)
//   tmpBuf  : 3 * tileLength * sizeof(float)  // for reduction temps
// Total <= 5 * tileLength * sizeof(float) = 20 * tileLength bytes.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    InstanceNormCustomTilingData tiling;
    const gert::StorageShape *inShape = context->GetInputShape(0);
    uint32_t N = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(0));
    uint32_t C = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(1));
    uint32_t H = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(2));
    uint32_t W = static_cast<uint32_t>(inShape->GetOriginShape().GetDim(3));
    uint32_t totalLength = N * C * H * W;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);
    float epsVal = (eps != nullptr) ? *eps : 1e-5f;

    uint32_t channelSize = H * W;
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (channelSize < maxTile) ? channelSize : maxTile;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = (channelSize < ALIGN_ELEMS) ? channelSize : ALIGN_ELEMS;

    // Block dim = number of (sample, channel) pairs
    uint32_t blockDim = N * C;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 8u) blockDim = 8u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_eps(epsVal);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class InstanceNormCustom : public OpDef {
public:
    explicit InstanceNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("eps").AttrType(OPTIONAL).Float(1e-5f);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(InstanceNormCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelInstanceNormCustom {
public:
    __aicore__ inline KernelInstanceNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                float eps)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        channelSize_ = H * W;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || channelSize_ == 0) return;

        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t numBlocks = AscendC::GetBlockNum();

        // Map blockIdx to (n, c)
        uint32_t totalChannels = N_ * C_;
        if (blockIdx >= totalChannels) return;
        uint32_t n = blockIdx / C_;
        uint32_t c = blockIdx % C_;

        uint32_t offset = (n * C_ + c) * channelSize_;
        uint32_t remaining = channelSize_;
        uint32_t curPos = 0;

        // First pass: compute sum and sum_sq
        float sum = 0.0f;
        float sumSq = 0.0f;

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            // Load tile data
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset + curPos], curLen);

            // Compute sum of elements
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceAdd(tmp, xLocal, curLen);
            sum += tmp.GetValue(0);

            // Compute sum of squares: square then reduce
            AscendC::Mul(tmp, xLocal, xLocal, curLen);
            AscendC::ReduceAdd(tmp, tmp, curLen);
            sumSq += tmp.GetValue(0);

            // Free tensors
            tmpQueue_.FreeTensor(tmp);
            inQueueX_.FreeTensor(xLocal);

            remaining -= curLen;
            curPos += curLen;
        }

        float mean = sum / static_cast<float>(channelSize_);
        float variance = (sumSq / static_cast<float>(channelSize_)) - (mean * mean);
        float invStd = 1.0f / sqrtf(variance + eps_);

        // Second pass: normalize
        remaining = channelSize_;
        curPos = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset + curPos], curLen);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

            // y = (x - mean) * invStd
            AscendC::Subs(tmp, xLocal, mean, curLen);
            AscendC::Muls(yLocal, tmp, invStd, curLen);

            // Copy out
            AscendC::DataCopy(yGm_[offset + curPos], yLocal, curLen);

            tmpQueue_.FreeTensor(tmp);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);

            remaining -= curLen;
            curPos += curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_ {1};
    uint32_t C_ {1};
    uint32_t H_ {1};
    uint32_t W_ {1};
    uint32_t channelSize_ {1};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void instance_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelInstanceNormCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.eps);
    op.Process();
}
