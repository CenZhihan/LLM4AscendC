kernel_src="""

#include "kernel_operator.h"
#include <kernel_operator/matmul.h>

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
                                GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t numGroups, float eps,
                                uint32_t tileLength)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        eps_ = eps;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inFeatures);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures * inFeatures);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures);
        betaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * outFeatures);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inFeatures_ * sizeof(float));
        pipe_.InitBuffer(inQueueWeight_, BUFFER_NUM, outFeatures_ * inFeatures_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, outFeatures_ * sizeof(float));
        pipe_.InitBuffer(meanVarBuf_, 1, (numGroups_ * 2) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t rowsPerBlock = CeilDiv(batchSize_, blockNum);
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = min(rowStart + rowsPerBlock, batchSize_);

        for (uint32_t r = rowStart; r < rowEnd; ++r) {
            OneRow(r);
        }
    }

private:
    __aicore__ inline void OneRow(uint32_t rowIdx)
    {
        AscendC::Matmul<AscendC::ROUND_MODE_NONE, AscendC::ND_CONVENTION> matmul;
        AscendC::LocalTensor<float> xRow = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xRow, xGm_[rowIdx * inFeatures_], inFeatures_);

        AscendC::LocalTensor<float> weightLocal = inQueueWeight_.AllocTensor<float>();
        AscendC::DataCopy(weightLocal, weightGm_, outFeatures_ * inFeatures_);

        AscendC::LocalTensor<float> result = outQueueY_.AllocTensor<float>();
        matmul.SetTensorA(xRow);
        matmul.SetTensorB(weightLocal);
        matmul.SetTensorC(result);
        matmul.SetTransB(true);
        matmul.IterateAll();

        inQueueX_.FreeTensor(xRow);
        inQueueWeight_.FreeTensor(weightLocal);

        AscendC::Add(result, result, linBiasGm_, outFeatures_);
        AscendC::Add(result, result, biasGm_, outFeatures_);

        AscendC::Mins(result, result, 1.0f, outFeatures_);
        AscendC::Maxs(result, result, -1.0f, outFeatures_);

        AscendC::LocalTensor<float> softplus = tmpQueue_.AllocTensor<float>();
        AscendC::Exp(softplus, result, outFeatures_);
        AscendC::Adds(softplus, softplus, 1.0f, outFeatures_);
        AscendC::Ln(softplus, softplus, outFeatures_);
        AscendC::Tanh(softplus, softplus, outFeatures_);
        AscendC::Mul(result, result, softplus, outFeatures_);
        tmpQueue_.FreeTensor(softplus);

        uint32_t groupSize = outFeatures_ / numGroups_;
        AscendC::LocalTensor<float> meanVar = meanVarBuf_.AllocTensor<float>();
        for (uint32_t g = 0; g < numGroups_; ++g) {
            float sum = 0.0f;
            for (uint32_t i = 0; i < groupSize; ++i) {
                sum += result[g * groupSize + i];
            }
            float mean = sum / (float)groupSize;
            meanVar[g] = mean;
            float varSum = 0.0f;
            for (uint32_t i = 0; i < groupSize; ++i) {
                float diff = result[g * groupSize + i] - mean;
                varSum += diff * diff;
            }
            float var = varSum / (float)groupSize;
            meanVar[numGroups_ + g] = var;
        }

        for (uint32_t g = 0; g < numGroups_; ++g) {
            float mean = meanVar[g];
            float var = meanVar[numGroups_ + g];
            float invStd = 1.0f / sqrtf(var + eps_);
            for (uint32_t i = 0; i < groupSize; ++i) {
                uint32_t idx = g * groupSize + i;
                result[idx] = (result[idx] - mean) * invStd * gammaGm_[idx] + betaGm_[idx];
            }
        }

        meanVarBuf_.FreeTensor(meanVar);

        AscendC::DataCopy(yGm_[rowIdx * outFeatures_], result, outFeatures_);
        outQueueY_.FreeTensor(result);
    }

    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1) / b; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueWeight_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> meanVarBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t inFeatures_ {0};
    uint32_t outFeatures_ {0};
    uint32_t numGroups_ {256};
    float eps_ {1e-5};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
    GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.batchSize, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.numGroups, tiling_data.eps, tiling_data.tileLength);
    op.Process();
}
