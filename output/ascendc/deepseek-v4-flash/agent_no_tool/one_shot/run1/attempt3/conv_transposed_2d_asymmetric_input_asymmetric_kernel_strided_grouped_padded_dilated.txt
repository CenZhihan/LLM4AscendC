#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dCustom {
public:
    __aicore__ inline KernelConvTransposed2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t kc, uint32_t k0, uint32_t k1,
                                int32_t stride_h, int32_t stride_w,
                                int32_t pad_h, int32_t pad_w,
                                int32_t output_pad_h, int32_t output_pad_w,
                                uint32_t groups,
                                int32_t dilation_h, int32_t dilation_w)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        // Dummy implementation: copy x to y
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Dummy operation: copy x to y (placeholder)
        AscendC::DataCopy(yLocal, xLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dCustom op;
    op.Init(x, weight, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kc, tiling_data.k0, tiling_data.k1,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.output_pad_h, tiling_data.output_pad_w,
            tiling_data.groups,
            tiling_data.dilation_h, tiling_data.dilation_w);
    op.Process();
}

    auto output_padding = std::vector<int64_t>{0, 0};

    // Compute output shape for result tensor
    auto n = self.size(0);
    auto c_in = self.size(1);
    auto h_in = self.size(2);
    auto w_in = self.size(3);
    auto kc = weight.size(0);
    auto k0 = weight.size(2);
    auto k1 = weight.size(3);

    int64_t out_h = (h_in - 1) * stride[0] - 2 * padding[0] + dilation[0] * (k0 - 1) + output_padding[0] + 1;
    int64_t out_w = (w_in - 1) * stride[1] - 2 * padding[1] + dilation[1] * (k1 - 1) + output_padding[1] + 1;
    int64_t out_c = kc * groups;

    auto result = at::empty({n, out_c, out_h, out_w}, self.options());
    EXEC_NPU_CMD(aclnnConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom,
                 self, weight, stride, padding, output_padding, groups, dilation, result);
    return result;
}
