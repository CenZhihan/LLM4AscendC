kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                const TPipeArgs &args)
    {
        N_ = args.N;
        C_ = args.C;
        H_ = args.H;
        W_ = args.W;
        KH_ = args.KH;
        KW_ = args.KW;
        strideH_ = args.strideH;
        strideW_ = args.strideW;
        padH_ = args.padH;
        padW_ = args.padW;
        dilationH_ = args.dilationH;
        dilationW_ = args.dilationW;
        groups_ = args.groups;
        outputH_ = args.outputH;
        outputW_ = args.outputW;
        tileSize_ = args.tileSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, groups_ * (C_ / groups_) * KH_ * KW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * outputH_ * outputW_);

        // Allocate local buffers for input patch and weight
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileSize_ * KH_ * KW_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileSize_ * KH_ * KW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (tileSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Total output elements = N * C * outputH * outputW
        uint32_t totalOut = N_ * C_ * outputH_ * outputW_;
        uint32_t strideElems = blockNum * tileSize_;
        uint32_t base = blockIdx * tileSize_;
        while (base < totalOut) {
            // Determine output indices for this tile
            uint32_t curLen = (base + tileSize_ <= totalOut) ? tileSize_ : (totalOut - base);
            // Process one output element at a time (tileSize==1)
            uint32_t n, c, oh, ow;
            LinearToIndex(base, n, c, oh, ow);

            // Load weight for this channel (depthwise: group = c)
            uint32_t group = c;  // groups == C
            uint32_t wOffset = group * KH_ * KW_;
            AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[wOffset], KH_ * KW_);
            inQueueW_.EnQue(wLocal);

            // Compute input coordinates
            int32_t hStart = oh * strideH_ - padH_;
            int32_t wStart = ow * strideW_ - padW_;
            float result = 0.0f;
            for (uint32_t kh = 0; kh < KH_; kh++) {
                for (uint32_t kw = 0; kw < KW_; kw++) {
                    int32_t ih = hStart + kh * dilationH_;
                    int32_t iw = wStart + kw * dilationW_;
                    // Check bounds (valid convolution with pad=0, but keep generic)
                    if (ih >= 0 && ih < H_ && iw >= 0 && iw < W_) {
                        uint32_t xIdx = n * C_ * H_ * W_ + c * H_ * W_ + ih * W_ + iw;
                        float xVal = xGm_.GetValue(xIdx);
                        float wVal = wGm_.GetValue(wOffset + kh * KW_ + kw);
                        result += xVal * wVal;
                    }
                }
            }

            // Store result
            yGm_.SetValue(base, result);

            // Free local tensors
            AscendC::LocalTensor<float> wLocalDeque = inQueueW_.DeQue<float>();
            inQueueW_.FreeTensor(wLocalDeque);

            base += strideElems;
        }
    }

private:
    __aicore__ inline void LinearToIndex(uint32_t idx, uint32_t &n, uint32_t &c,
                                          uint32_t &oh, uint32_t &ow)
    {
        uint32_t spatialSize = outputH_ * outputW_;
        n = idx / (C_ * spatialSize);
        uint32_t rem = idx % (C_ * spatialSize);
        c = rem / spatialSize;
        uint32_t spatialIdx = rem % spatialSize;
        oh = spatialIdx / outputW_;
        ow = spatialIdx % outputW_;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_, C_, H_, W_;
    uint32_t KH_, KW_;
    uint32_t strideH_, strideW_;
    uint32_t padH_, padW_;
    uint32_t dilationH_, dilationW_;
    uint32_t groups_;
    uint32_t outputH_, outputW_;
    uint32_t tileSize_;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymmetricInputSquareKernelCustom op;
    TPipeArgs args;
    args.N = tiling_data.N;
    args.C = tiling_data.C;
    args.H = tiling_data.H;
    args.W = tiling_data.W;
    args.KH = tiling_data.KH;
    args.KW = tiling_data.KW;
    args.strideH = tiling_data.strideH;
    args.strideW = tiling_data.strideW;
    args.padH = tiling_data.padH;
    args.padW = tiling_data.padW;
    args.dilationH = tiling_data.dilationH;
    args.dilationW = tiling_data.dilationW;
    args.groups = tiling_data.groups;
    args.outputH = tiling_data.outputH;
    args.outputW = tiling_data.outputW;
    args.tileSize = tiling_data.tileSize;
    op.Init(x, weight, y, args);
    op.Process();
}
