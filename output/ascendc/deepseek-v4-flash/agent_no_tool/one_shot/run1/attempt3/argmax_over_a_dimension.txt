#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    // Use GetOriginShape() to get the actual shape (not StorageShape)
    const auto& x_shape = context->GetInputShape(0)->GetOriginShape();
    int32_t ndim = x_shape.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;

    // Compute sliceSize = size of the dimension to be reduced
    uint32_t sliceSize = static_cast<uint32_t>(x_shape.GetDim(dim));

    // Compute numSlices = product of all other dimensions
    uint32_t numSlices = 1;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i != dim) {
            numSlices *= static_cast<uint32_t>(x_shape.GetDim(i));
        }
    }

    // Block dimension: cap to 24, but at least 1
    uint32_t blockDim = numSlices;
    if (blockDim > 24) blockDim = 24;
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(numSlices);
    tiling.set_sliceSize(sliceSize);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const auto& x_shape = context->GetInputShape(0)->GetOriginShape();
    int32_t ndim = x_shape.GetDimNum();

    int32_t dim = -1; // default, will be overwritten by attr
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs != nullptr) {
        dim = attrs->GetAttrPointer<int32_t>(0)[0];
        if (dim < 0) dim += ndim;
    }

    // Output shape: remove the specified dimension
    std::vector<int64_t> out_shape;
    for (int32_t i = 0; i < ndim; ++i) {
        if (i != dim) {
            out_shape.push_back(static_cast<int64_t>(x_shape.GetDim(i)));
        }
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(static_cast<int32_t>(out_shape.size()));
    for (size_t i = 0; i < out_shape.size(); ++i) {
        y_shape->SetDim(static_cast<int32_t>(i), out_shape[i]);
    }

    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // input is float, output is int64
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t sliceSize,
                                int32_t dim)
    {
        totalLen_ = totalLength;  // number of slices (output elements)
        sliceSize_ = sliceSize;
        dim_ = dim;  // not used, kept for interface compatibility

        // x is treated as 2D: [numSlices, sliceSize] in memory (contiguous rows)
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_ * sliceSize_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalLen_);

        // Double buffer for input: each buffer holds one slice
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, sliceSize_ * sizeof(float));
        // Output is scalar per slice, no buffering needed
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || sliceSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride loop over slices
        for (uint32_t sliceIdx = blockIdx; sliceIdx < totalLen_; sliceIdx += blockNum) {
            ProcessSlice(sliceIdx);
        }
    }

private:
    __aicore__ inline void ProcessSlice(uint32_t sliceIdx)
    {
        // Copy entire slice from global to local
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[sliceIdx * sliceSize_], sliceSize_);

        // Scalar argmax over the slice
        float maxVal = xLocal(0);
        int64_t maxIdx = 0;
        for (uint32_t i = 1; i < sliceSize_; ++i) {
            float val = xLocal(i);
            if (val > maxVal) {
                maxVal = val;
                maxIdx = static_cast<int64_t>(i);
            }
        }

        // Write result
        yGm_(sliceIdx) = maxIdx;

        // Free local tensor
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalLen_ {0};
    uint32_t sliceSize_ {0};
    int32_t dim_ {-1};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.sliceSize, tiling_data.dim);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    // Move the reduction dimension to the last position
    // The custom kernel always reduces the last dimension
    int64_t ndim = self.ndimension();
    if (dim < 0) dim += ndim;
    std::vector<int64_t> permute;
    for (int64_t i = 0; i < ndim; ++i) {
        if (i != dim) permute.push_back(i);
    }
    permute.push_back(dim);
    at::Tensor self_permuted = self.permute(permute).contiguous();

    at::Tensor result = at::empty_like(self_permuted, self_permuted.options().dtype(at::kLong));
    // Call the custom operator (ArgmaxOverADimensionCustom)
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self_permuted, -1, result);

    // The op reduces the last dimension, output shape is without that dim.
    // No need to squeeze further; result already has shape without the last dim.
    // The output dimensions are in the order after permute, which is all dims except dim in original order.
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu, "Argmax over a dimension (ArgmaxOverADimensionCustom on NPU)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'argmax_over_a_dimension_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, dim):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x):
        # Move the reduction dimension to the last position
        ndim = x.ndimension()
        dim = self.dim if self.dim >= 0 else ndim + self.dim
        permute = list(range(ndim))
        permute.pop(dim)
        permute.append(dim)
        x_perm = x.permute(permute).contiguous()
        # Call custom op (reduces last dimension)
        y = getattr(custom_ops_lib, _FN)(x_perm)
        return y
'''
