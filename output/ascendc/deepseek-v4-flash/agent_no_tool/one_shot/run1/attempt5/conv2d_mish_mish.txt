static constexpr uint32_t ALIGN = 16;
static constexpr uint32_t MAX_TILE_H = 8;
static constexpr uint32_t MAX_TILE_W = 8;

static uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Conv2dMishMishCustomTilingData tiling;

    const auto& x_shape = context->GetInputShape(0)->GetOriginShape();
    const auto& w_shape = context->GetInputShape(1)->GetOriginShape();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    tiling.set_n(x_shape.GetDim(0));
    tiling.set_cIn(x_shape.GetDim(1));
    tiling.set_hIn(x_shape.GetDim(2));
    tiling.set_wIn(x_shape.GetDim(3));
    tiling.set_k(w_shape.GetDim(0));
    tiling.set_r(w_shape.GetDim(2));
    tiling.set_s(w_shape.GetDim(3));
    uint32_t stride = attrs->GetAttrPointer<uint32_t>(0)[0];
    uint32_t padding = attrs->GetAttrPointer<uint32_t>(1)[0];
    uint32_t dilation = attrs->GetAttrPointer<uint32_t>(2)[0];
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_dilation(dilation);

    uint32_t hIn = tiling.get_hIn();
    uint32_t r = tiling.get_r();
    uint32_t hOut = (hIn + 2 * padding - dilation * (r - 1) - 1) / stride + 1;
    uint32_t wIn = tiling.get_wIn();
    uint32_t s = tiling.get_s();
    uint32_t wOut = (wIn + 2 * padding - dilation * (s - 1) - 1) / stride + 1;
    tiling.set_hOut(hOut);
    tiling.set_wOut(wOut);

    uint32_t bytesPerElem = 4;
    uint32_t tileH = std::min(MAX_TILE_H, hOut);
    uint32_t tileW = std::min(MAX_TILE_W, wOut);
    uint32_t cIn = tiling.get_cIn();
    uint32_t im2colSize = tileH * tileW * cIn * r * s * bytesPerElem;
    if (im2colSize > UB_BUDGET_BYTES * 0.4) {
        tileH = std::max(1u, tileH / 2);
        tileW = std::max(1u, tileW / 2);
    }

    tiling.set_tileH(AlignDown(tileH, ALIGN) > 0 ? AlignDown(tileH, ALIGN) : 1);
    tiling.set_tileW(AlignDown(tileW, ALIGN) > 0 ? AlignDown(tileW, ALIGN) : 1);

    uint32_t tileH_ = tiling.get_tileH();
    uint32_t tileW_ = tiling.get_tileW();
    uint32_t numTiles = ((hOut + tileH_ - 1) / tileH_) * ((wOut + tileW_ - 1) / tileW_);
    uint32_t blockDim = std::min(numTiles, 24u);
    if (blockDim < 8u) blockDim = 8u;
    tiling.set_blockDim(blockDim);

    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t MAX_OCH_GROUP = 32; // maximum output channels processed together in UB

class KernelConv2dMishMishCustom {
public:
    __aicore__ inline KernelConv2dMishMishCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t hIn, uint32_t wIn,
                                uint32_t k, uint32_t r, uint32_t s,
                                uint32_t hOut, uint32_t wOut,
                                uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t tileH, uint32_t tileW) {
        n_ = n; cIn_ = cIn; hIn_ = hIn; wIn_ = wIn;
        k_ = k; r_ = r; s_ = s;
        hOut_ = hOut; wOut_ = wOut;
        stride_ = stride; padding_ = padding; dilation_ = dilation;
        tileH_ = tileH; tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cIn_ * hIn_ * wIn_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, k_ * cIn_ * r_ * s_);
        hasBias_ = (bias != nullptr);
        if (hasBias_) bGm_.SetGlobalBuffer((__gm__ float*)bias, k_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * k_ * hOut_ * wOut_);

        uint32_t tileArea = tileH_ * tileW_;
        im2colSize_ = tileArea * cIn_ * r_ * s_;

        // allocate queues
        pipe_.InitBuffer(inQueueCol_, BUFFER_NUM, im2colSize_ * sizeof(float));
        // weight buffer for one ochGroup
        uint32_t wSliceSize = MAX_OCH_GROUP * cIn_ * r_ * s_;
        pipe_.InitBuffer(inQueueW_, 1, wSliceSize * sizeof(float));
        // output buffer for one ochGroup
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileArea * MAX_OCH_GROUP * sizeof(float));
        // temporary for mish
        pipe_.InitBuffer(tmpQueue_, 1, tileArea * MAX_OCH_GROUP * sizeof(float));
        // bias buffer (full length)
        if (hasBias_) {
            pipe_.InitBuffer(biasQueue_, 1, k_ * sizeof(float));
            AscendC::LocalTensor<float> bLocal = biasQueue_.AllocTensor<float>();
            AscendC::DataCopy(bLocal, bGm_, k_);
            biasQueue_.EnQue(bLocal);
        }
    }

    __aicore__ inline void Process() {
        if (n_ == 0 || k_ == 0 || hOut_ == 0 || wOut_ == 0) return;
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t numTilesH = (hOut_ + tileH_ - 1) / tileH_;
        uint32_t numTilesW = (wOut_ + tileW_ - 1) / tileW_;
        uint32_t totalTiles = n_ * numTilesH * numTilesW;
        uint32_t tilesPerBlock = (totalTiles + blockNum - 1) / blockNum;
        uint32_t startTile = blockIdx * tilesPerBlock;
        uint32_t endTile = startTile + tilesPerBlock;
        if (endTile > totalTiles) endTile = totalTiles;

        for (uint32_t tileIdx = startTile; tileIdx < endTile; tileIdx++) {
            uint32_t nIdx = tileIdx / (numTilesH * numTilesW);
            uint32_t rem = tileIdx % (numTilesH * numTilesW);
            uint32_t hTileIdx = rem / numTilesW;
            uint32_t wTileIdx = rem % numTilesW;
            uint32_t hStart = hTileIdx * tileH_;
            uint32_t wStart = wTileIdx * tileW_;
            uint32_t hEnd = hStart + tileH_;
            if (hEnd > hOut_) hEnd = hOut_;
            uint32_t wEnd = wStart + tileW_;
            if (wEnd > wOut_) wEnd = wOut_;
            uint32_t curH = hEnd - hStart;
            uint32_t curW = wEnd - wStart;
            uint32_t curArea = curH * curW;

            // im2col
            AscendC::LocalTensor<float> colLocal = inQueueCol_.AllocTensor<float>();
            uint32_t idx = 0;
            for (uint32_t ho = hStart; ho < hEnd; ho++) {
                for (uint32_t wo = wStart; wo < wEnd; wo++) {
                    int32_t hi = static_cast<int32_t>(ho) * stride_ - padding_;
                    int32_t wi = static_cast<int32_t>(wo) * stride_ - padding_;
                    for (uint32_t ci = 0; ci < cIn_; ci++) {
                        for (uint32_t ri = 0; ri < r_; ri++) {
                            int32_t hIdx = hi + ri * dilation_;
                            for (uint32_t si = 0; si < s_; si++) {
                                int32_t wIdx = wi + si * dilation_;
                                float val = 0.0f;
                                if (hIdx >= 0 && hIdx < static_cast<int32_t>(hIn_) &&
                                    wIdx >= 0 && wIdx < static_cast<int32_t>(wIn_)) {
                                    uint32_t gmIdx = nIdx * cIn_ * hIn_ * wIn_ +
                                                     ci * hIn_ * wIn_ +
                                                     hIdx * wIn_ + wIdx;
                                    val = xGm_.GetValue(gmIdx);
                                }
                                colLocal.SetValue(idx++, val);
                            }
                        }
                    }
                }
            }
            inQueueCol_.EnQue(colLocal);

            // Process output channels in groups
            for (uint32_t ochStart = 0; ochStart < k_; ochStart += MAX_OCH_GROUP) {
                uint32_t ochSize = (k_ - ochStart < MAX_OCH_GROUP) ? (k_ - ochStart) : MAX_OCH_GROUP;
                // load weight slice
                AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
                uint32_t wSliceLen = ochSize * cIn_ * r_ * s_;
                AscendC::DataCopy(wLocal, wGm_[ochStart * cIn_ * r_ * s_], wSliceLen);

                // matmul: col (curArea, cIn*r*s) * weight^T (ochSize, cIn*r*s) -> out (curArea, ochSize)
                AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();
                AscendC::Matmul(outLocal, colLocal, wLocal, false, true, curArea, ochSize, cIn_ * r_ * s_);
                inQueueW_.FreeTensor(wLocal);

                // add bias if present (broadcast along spatial dims)
                if (hasBias_) {
                    AscendC::LocalTensor<float> biasLocal = biasQueue_.DeQue<float>();
                    // bias is full length, we need slice for ochStart:ochStart+ochSize
                    for (uint32_t och = 0; och < ochSize; och++) {
                        float bVal = biasLocal.GetValue(ochStart + och);
                        for (uint32_t p = 0; p < curArea; p++) {
                            uint32_t outIdx = p * ochSize + och;
                            float cur = outLocal.GetValue(outIdx);
                            outLocal.SetValue(outIdx, cur + bVal);
                        }
                    }
                    biasQueue_.EnQue(biasLocal);
                }

                // Apply Mish twice: x * tanh(softplus(x))
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                // First Mish
                MishInPlace(outLocal, tLocal, curArea * ochSize);
                // Second Mish
                MishInPlace(outLocal, tLocal, curArea * ochSize);
                tmpQueue_.FreeTensor(tLocal);

                // write to global output
                for (uint32_t p = 0; p < curArea; p++) {
                    for (uint32_t och = 0; och < ochSize; och++) {
                        uint32_t hIdx = hStart + p / curW;
                        uint32_t wIdx = wStart + p % curW;
                        uint32_t gmIdx = nIdx * k_ * hOut_ * wOut_ +
                                         (ochStart + och) * hOut_ * wOut_ +
                                         hIdx * wOut_ + wIdx;
                        yGm_.SetValue(gmIdx, outLocal.GetValue(p * ochSize + och));
                    }
                }
                outQueueY_.FreeTensor(outLocal);
            }
            inQueueCol_.FreeTensor(colLocal);
        }
    }

private:
    __aicore__ inline void MishInPlace(AscendC::LocalTensor<float>& x, AscendC::LocalTensor<float>& tmp, uint32_t len) {
        // softplus: log(1+exp(x))
        for (uint32_t i = 0; i < len; i++) {
            float val = x.GetValue(i);
            float expVal = expf(val);
            float softplus = logf(1.0f + expVal);
            float tanhVal = tanhf(softplus);
            x.SetValue(i, val * tanhVal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueCol_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> biasQueue_;

    AscendC::GlobalTensor<float> xGm_, wGm_, bGm_, yGm_;
    uint32_t n_, cIn_, hIn_, wIn_, k_, r_, s_;
    uint32_t hOut_, wOut_, stride_, padding_, dilation_;
    uint32_t tileH_, tileW_, im2colSize_;
    bool hasBias_;
};

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                                               GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMishMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.n, tiling_data.cIn, tiling_data.hIn, tiling_data.wIn,
            tiling_data.k, tiling_data.r, tiling_data.s,
            tiling_data.hOut, tiling_data.wOut,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation,
            tiling_data.tileH, tiling_data.tileW);
    op.Process();
}
