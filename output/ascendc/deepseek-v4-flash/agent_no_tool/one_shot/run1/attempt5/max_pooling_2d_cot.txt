class KernelMaxPooling2dCustom {
public:
  __aicore__ inline KernelMaxPooling2dCustom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                              uint32_t batch, uint32_t channels,
                              uint32_t inH, uint32_t inW,
                              uint32_t kernel, uint32_t stride,
                              uint32_t padding,
                              uint32_t outH, uint32_t outW) {
    batch_ = batch;
    channels_ = channels;
    inH_ = inH;
    inW_ = inW;
    kernel_ = kernel;
    stride_ = stride;
    padding_ = padding;
    outH_ = outH;
    outW_ = outW;
    totalOutElems_ = batch * channels * outH * outW;

    yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutElems_);

  __aicore__ inline void Process() {

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    // Simple block-stride loop over all output elements
    for (uint32_t outIdx = blockIdx; outIdx < totalOutElems_; outIdx += blockNum) {

      uint32_t oh = tmp % outH_; tmp /= outH_;
