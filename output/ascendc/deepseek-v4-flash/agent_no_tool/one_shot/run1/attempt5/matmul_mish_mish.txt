#include "matmul_mish_mish_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t ALIGN_M = 16;
static constexpr uint32_t ALIGN_N = 16;
static constexpr uint32_t ALIGN_K = 16;
static constexpr uint32_t MAX_BLOCK = 24;
static constexpr uint32_t UB_BUDGET_M = 128;
static constexpr uint32_t UB_BUDGET_N = 128;
static constexpr uint32_t UB_BUDGET_K = 256;

static inline uint32_t AlignUp(uint32_t x, uint32_t a) {
    return (x + a - 1) / a * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MatmulMishMishCustomTilingData tiling;
    const gert::StorageShape *storage_x = context->GetInputShape(0);
    const gert::StorageShape *storage_w = context->GetInputShape(1);
    const gert::Shape &x_shape = storage_x->GetOriginShape();
    const gert::Shape &w_shape = storage_w->GetOriginShape();
    uint32_t M = x_shape.GetDim(0);
    uint32_t K = x_shape.GetDim(1);
    uint32_t N = w_shape.GetDim(0);

    uint32_t tileM = UB_BUDGET_M;
    uint32_t tileN = UB_BUDGET_N;
    uint32_t tileK = UB_BUDGET_K;
    if (tileM > M) tileM = M;
    if (tileN > N) tileN = N;
    if (tileK > K) tileK = K;

    tileM = AlignUp(tileM, ALIGN_M);
    tileN = AlignUp(tileN, ALIGN_N);
    tileK = AlignUp(tileK, ALIGN_K);

    uint32_t blockDim = MAX_BLOCK;
    if (blockDim > (M * N) / (tileM * tileN)) blockDim = (M * N) / (tileM * tileN);
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"
#include "matmul.h"

using namespace MatmulImpl;

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_MAX = 256;

class KernelMatmulMishMishCustom {
public:
    __aicore__ inline KernelMatmulMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M * K);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, N * K);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, N);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M * N);

        matmul_.SetDim(M_, N_, K_);
        matmul_.SetTensorA(xGm_);
        matmul_.SetTensorB(wGm_);
        matmul_.SetBias(bGm_);
        matmul_.SetTransposeB(true);
        matmul_.SetOutputTensor(yGm_);
        matmul_.SetTileSize(tileM_, tileN_, tileK_);

        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, TILE_MAX * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, TILE_MAX * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, TILE_MAX * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        matmul_.Process();
        ApplyMishTwice();
    }

private:
    __aicore__ inline void ApplyMishTwice()
    {
        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t totalElems = M_ * N_;
        uint32_t stride = blockNum * TILE_MAX;
        uint32_t start = blockIdx * TILE_MAX;

        while (start < totalElems) {
            uint32_t curTile = (totalElems - start);
            if (curTile > TILE_MAX) curTile = TILE_MAX;

            AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yLocal, yGm_[start], curTile);
            inQueueY_.EnQue(yLocal);

            AscendC::LocalTensor<float> mish1 = outQueueY_.AllocTensor<float>();
            Mish(yLocal, mish1, curTile);
            outQueueY_.EnQue(mish1);

            AscendC::LocalTensor<float> mish2 = inQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> mish1Deque = outQueueY_.DeQue<float>();
            Mish(mish1Deque, mish2, curTile);
            inQueueY_.EnQue(mish2);
            outQueueY_.FreeTensor(mish1Deque);

            AscendC::LocalTensor<float> result = inQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[start], result, curTile);
            inQueueY_.FreeTensor(result);

            start += stride;
        }
    }

    __aicore__ inline void Mish(AscendC::LocalTensor<float> &input,
                                AscendC::LocalTensor<float> &output,
                                uint32_t len)
    {
        AscendC::LocalTensor<float> t1 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2 = tmpQueue_.AllocTensor<float>();

        AscendC::Exp(t1, input, len);
        AscendC::Adds(t2, t1, 1.0f, len);
        AscendC::Log(t1, t2, len);
        AscendC::Tanh(t2, t1, len);
        AscendC::Mul(output, input, t2, len);

        tmpQueue_.FreeTensor(t1);
        tmpQueue_.FreeTensor(t2);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    Matmul<float, float, float> matmul_;

    uint32_t M_, N_, K_;
    uint32_t tileM_, tileN_, tileK_;
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                              GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMishMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
