    uint32_t hidden0 = static_cast<uint32_t>(*p_hidden0);
    uint32_t hidden1 = static_cast<uint32_t>(*p_hidden1);
    uint32_t outputSize = static_cast<uint32_t>(*p_outputSize);

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_inputSize(inputSize);
    tiling.set_hidden0(hidden0);
    tiling.set_hidden1(hidden1);
    tiling.set_outputSize(outputSize);
    tiling.set_blockDim(blockDim);

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMlpCustom {
public:
    __aicore__ inline KernelMlpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed,
                                GM_ADDR y,
                                uint32_t batch, uint32_t inputSize,
                                uint32_t hidden0, uint32_t hidden1,
                                uint32_t outputSize)
    {
        batch_ = batch;
        inputSize_ = inputSize;
        hidden0_ = hidden0;
        hidden1_ = hidden1;
        outputSize_ = outputSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * inputSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w_packed,
                             hidden0 * inputSize + hidden1 * hidden0 + outputSize * hidden1);
        bGm_.SetGlobalBuffer((__gm__ float*)b_packed,
                             hidden0 + hidden1 + outputSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * outputSize);
    }

    __aicore__ inline void Process()
    {
        // Allocate local tensors (size based on max dimensions)
        pipe_.InitBuffer(xLocalQue_, BUFFER_NUM, inputSize_ * sizeof(float));
        pipe_.InitBuffer(wRowQue_, BUFFER_NUM, inputSize_ * sizeof(float));
        pipe_.InitBuffer(accumQue_, 1, inputSize_ * sizeof(float));
        pipe_.InitBuffer(sumScalarQue_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(biasScalarQue_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(h0Que_, BUFFER_NUM, hidden0_ * sizeof(float));
        pipe_.InitBuffer(h1Que_, BUFFER_NUM, hidden1_ * sizeof(float));
        pipe_.InitBuffer(outQue_, BUFFER_NUM, outputSize_ * sizeof(float));

        for (uint32_t b = 0; b < batch_; b++) {
            // Load x row into xLocal
            LocalTensor<float> xLocal = xLocalQue_.AllocTensor<float>();
            DataCopy(xLocal, xGm_[b * inputSize_], inputSize_);

            // ---- Layer1: W1 * x + b1 -> h0, ReLU ----
            LocalTensor<float> h0 = h0Que_.AllocTensor<float>();
            for (uint32_t r = 0; r < hidden0_; r++) {
                // Load weight row
                LocalTensor<float> wRow = wRowQue_.AllocTensor<float>();
                DataCopy(wRow, wGm_[r * inputSize_], inputSize_);

                // accum = x * wRow (elementwise)
                LocalTensor<float> accum = accumQue_.AllocTensor<float>();
                Mul(accum, xLocal, wRow, inputSize_);

                // Reduce sum to scalar
                LocalTensor<float> sumScalar = sumScalarQue_.AllocTensor<float>();
                ReduceSum(sumScalar, accum, inputSize_, 0);
                accumQue_.FreeTensor(accum);

                // Load bias scalar
                LocalTensor<float> biasScalar = biasScalarQue_.AllocTensor<float>();
                DataCopy(biasScalar, bGm_[r], 1);

                // sum = sum + bias
                Add(sumScalar, sumScalar, biasScalar, 1);
                biasScalarQue_.FreeTensor(biasScalar);

                // Store into h0[r] via DataCopy (sub-tensor)
                DataCopy(h0[r], sumScalar, 1);
                sumScalarQue_.FreeTensor(sumScalar);
                wRowQue_.FreeTensor(wRow);
            }
            // ReLU in-place on h0
            Maxs(h0, h0, 0.0f, hidden0_);
            xLocalQue_.FreeTensor(xLocal);

            // ---- Layer2: W2 * h0 + b2 -> h1, ReLU ----
            LocalTensor<float> h1 = h1Que_.AllocTensor<float>();
            uint32_t wOffset = hidden0_ * inputSize_;
            for (uint32_t r = 0; r < hidden1_; r++) {
                LocalTensor<float> wRow = wRowQue_.AllocTensor<float>();
                DataCopy(wRow, wGm_[wOffset + r * hidden0_], hidden0_);

                LocalTensor<float> accum = accumQue_.AllocTensor<float>();
                Mul(accum, h0, wRow, hidden0_);

                LocalTensor<float> sumScalar = sumScalarQue_.AllocTensor<float>();
                ReduceSum(sumScalar, accum, hidden0_, 0);
                accumQue_.FreeTensor(accum);

                LocalTensor<float> biasScalar = biasScalarQue_.AllocTensor<float>();
                DataCopy(biasScalar, bGm_[hidden0_ + r], 1);

                Add(sumScalar, sumScalar, biasScalar, 1);
                biasScalarQue_.FreeTensor(biasScalar);

                DataCopy(h1[r], sumScalar, 1);
                sumScalarQue_.FreeTensor(sumScalar);
                wRowQue_.FreeTensor(wRow);
            }
            Maxs(h1, h1, 0.0f, hidden1_);
            h0Que_.FreeTensor(h0);

            // ---- Layer3: W3 * h1 + b3 -> output ----
            LocalTensor<float> out = outQue_.AllocTensor<float>();
            wOffset = hidden0_ * inputSize_ + hidden1_ * hidden0_;
            for (uint32_t r = 0; r < outputSize_; r++) {
                LocalTensor<float> wRow = wRowQue_.AllocTensor<float>();
                DataCopy(wRow, wGm_[wOffset + r * hidden1_], hidden1_);

                LocalTensor<float> accum = accumQue_.AllocTensor<float>();
                Mul(accum, h1, wRow, hidden1_);

                LocalTensor<float> sumScalar = sumScalarQue_.AllocTensor<float>();
                ReduceSum(sumScalar, accum, hidden1_, 0);
                accumQue_.FreeTensor(accum);

                LocalTensor<float> biasScalar = biasScalarQue_.AllocTensor<float>();
                DataCopy(biasScalar, bGm_[hidden0_ + hidden1_ + r], 1);

                Add(sumScalar, sumScalar, biasScalar, 1);
                biasScalarQue_.FreeTensor(biasScalar);

                DataCopy(out[r], sumScalar, 1);
                sumScalarQue_.FreeTensor(sumScalar);
                wRowQue_.FreeTensor(wRow);
            }
            h1Que_.FreeTensor(h1);

            // Write output row to global
            DataCopy(yGm_[b * outputSize_], out, outputSize_);
            outQue_.FreeTensor(out);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> xLocalQue_;
    TQue<TPosition::VECIN, BUFFER_NUM> wRowQue_;
    TQue<TPosition::VECCALC, 1> accumQue_;
    TQue<TPosition::VECCALC, 1> sumScalarQue_;
    TQue<TPosition::VECCALC, 1> biasScalarQue_;
    TQue<TPosition::VECOUT, BUFFER_NUM> h0Que_;
    TQue<TPosition::VECOUT, BUFFER_NUM> h1Que_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t batch_ = 0;
    uint32_t inputSize_ = 0;
    uint32_t hidden0_ = 0;
    uint32_t hidden1_ = 0;
    uint32_t outputSize_ = 0;
};

extern "C" __global__ __aicore__ void mlp_custom(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed,
                                                  GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMlpCustom op;
    op.Init(x, w_packed, b_packed, y,
            tiling_data.batch, tiling_data.inputSize,
            tiling_data.hidden0, tiling_data.hidden1,
            tiling_data.outputSize);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'mlp_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, layer_sizes, output_size, *args, **kwargs):
        super(ModelNew, self).__init__()
        self.input_size = input_size
        self.hidden_0 = layer_sizes[0]
        self.hidden_1 = layer_sizes[1]
        self.output_size = output_size

    def forward(self, x, w_packed, b_packed):
        fn = getattr(custom_ops_lib, _FN)
        return fn(
            x, w_packed, b_packed,
            input_size=self.input_size,
            hidden_0=self.hidden_0,
            hidden_1=self.hidden_1,
            output_size=self.output_size
