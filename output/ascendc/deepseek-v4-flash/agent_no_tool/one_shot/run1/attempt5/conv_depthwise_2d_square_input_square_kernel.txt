#include "conv_depthwise_2d_square_input_square_kernel_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ConvDepthwise2dSquareInputSquareKernelCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    auto origin_x = x_shape->GetOriginShape();
    auto origin_w = w_shape->GetOriginShape();

    const uint32_t N    = origin_x.GetDim(0);
    const uint32_t C_in = origin_x.GetDim(1);
    const uint32_t H    = origin_x.GetDim(2);
    const uint32_t W    = origin_x.GetDim(3);

    const uint32_t C_out = origin_w.GetDim(0);
    const uint32_t kh    = origin_w.GetDim(2);
    const uint32_t kw    = origin_w.GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *stride_h = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride_w = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *pad_h    = attrs->GetAttrPointer<int64_t>(2);
    const int64_t *pad_w    = attrs->GetAttrPointer<int64_t>(3);
    const int64_t *dil_h    = attrs->GetAttrPointer<int64_t>(4);
    const int64_t *dil_w    = attrs->GetAttrPointer<int64_t>(5);
    const int64_t *groups   = attrs->GetAttrPointer<int64_t>(6);

    const uint32_t H_out = (H + 2 * (*pad_h) - (*dil_h) * (kh - 1) - 1) / (*stride_h) + 1;
    const uint32_t W_out = (W + 2 * (*pad_w) - (*dil_w) * (kw - 1) - 1) / (*stride_w) + 1;

    // Compute tileH from UB budget
    const uint32_t maxTileH = (UB_BUDGET_BYTES / 2) / (4 * (W + W_out));
    uint32_t tileH = AlignDown(maxTileH, ALIGN_ELEMS);
    if (tileH < 1) tileH = 1;
    if (tileH > H_out) tileH = H_out;

    uint32_t totalPairs = N * C_out;
    uint32_t blockDim = CeilDiv(totalPairs, 2);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 40) blockDim = 40;
    context->SetBlockDim(blockDim);

    tiling.set_batchSize(N);
    tiling.set_channelsIn(C_in);
    tiling.set_heightIn(H);
    tiling.set_widthIn(W);
    tiling.set_channelsOut(C_out);
    tiling.set_kernelH(kh);
    tiling.set_kernelW(kw);
    tiling.set_strideH(*stride_h);
    tiling.set_strideW(*stride_w);
    tiling.set_padH(*pad_h);
    tiling.set_padW(*pad_w);
    tiling.set_dilationH(*dil_h);
    tiling.set_dilationW(*dil_w);
    tiling.set_groups(*groups);
    tiling.set_heightOut(H_out);
    tiling.set_widthOut(W_out);
    tiling.set_tileH(tileH);
    tiling.set_totalPairs(totalPairs);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    uint32_t W = origin_x.GetDim(3);
    uint32_t kh = origin_w.GetDim(2);
    uint32_t kw = origin_w.GetDim(3);

    uint32_t H_out = (H + 2 * (*pad_h) - (*dil_h) * (kh - 1) - 1) / (*stride_h) + 1;
    uint32_t W_out = (W + 2 * (*pad_w) - (*dil_w) * (kw - 1) - 1) / (*stride_w) + 1;

#include "kernel_operator.h"

using namespace AscendC;

class KernelConvDepthwise2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t batchSize, uint32_t channelsIn,
                                uint32_t heightIn, uint32_t widthIn,
                                uint32_t channelsOut, uint32_t kernelH, uint32_t kernelW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups, uint32_t heightOut, uint32_t widthOut,
                                uint32_t tileH, uint32_t totalPairs)
    {
        batchSize_ = batchSize; channelsIn_ = channelsIn;
        heightIn_ = heightIn; widthIn_ = widthIn;
        channelsOut_ = channelsOut; kernelH_ = kernelH; kernelW_ = kernelW;
        strideH_ = strideH; strideW_ = strideW;
        padH_ = padH; padW_ = padW;
        dilationH_ = dilationH; dilationW_ = dilationW;
        groups_ = groups;
        heightOut_ = heightOut; widthOut_ = widthOut;
        tileH_ = tileH; totalPairs_ = totalPairs;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * channelsIn_ * heightIn_ * widthIn_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, channelsOut_ * kernelH_ * kernelW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * channelsOut_ * heightOut_ * widthOut_);
    }

    __aicore__ inline void Process()
    {
        if (totalPairs_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        // Each block processes a set of (batch, channel) pairs in a grid-stride loop
        for (uint32_t pairIdx = blockIdx; pairIdx < totalPairs_; pairIdx += blockNum)
        {
            uint32_t b = pairIdx / channelsOut_;
            uint32_t c_out = pairIdx % channelsOut_;
            uint32_t c_in = c_out; // depthwise: groups == channelsOut == channelsIn

            // Base offsets
            uint32_t xBase = b * (channelsIn_ * heightIn_ * widthIn_) + c_in * (heightIn_ * widthIn_);
            uint32_t wBase = c_out * (kernelH_ * kernelW_);
            uint32_t yBase = b * (channelsOut_ * heightOut_ * widthOut_) + c_out * (heightOut_ * widthOut_);

            for (uint32_t hOut = 0; hOut < heightOut_; hOut++)
            {
                for (uint32_t wOut = 0; wOut < widthOut_; wOut++)
                {
                    float sum = 0.0f;
                    // Apply convolution
                    for (uint32_t i = 0; i < kernelH_; i++)
                    {
                        uint32_t hIn = hOut * strideH_ + i * dilationH_ - padH_;
                        if (hIn >= heightIn_) continue;
                        for (uint32_t j = 0; j < kernelW_; j++)
                        {
                            uint32_t wIn = wOut * strideW_ + j * dilationW_ - padW_;
                            if (wIn >= widthIn_) continue;
                            uint32_t xIdx = xBase + hIn * widthIn_ + wIn;
                            uint32_t wIdx = wBase + i * kernelW_ + j;
                            sum += xGm_.GetValue(xIdx) * wGm_.GetValue(wIdx);
                        }
                    }
                    uint32_t yIdx = yBase + hOut * widthOut_ + wOut;
                    yGm_.SetValue(yIdx, sum);
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_, channelsIn_, heightIn_, widthIn_;
    uint32_t channelsOut_, kernelH_, kernelW_;
    uint32_t strideH_, strideW_, padH_, padW_, dilationH_, dilationW_, groups_;
    uint32_t heightOut_, widthOut_;
    uint32_t tileH_, totalPairs_;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.batchSize, tiling_data.channelsIn,
            tiling_data.heightIn, tiling_data.widthIn,
            tiling_data.channelsOut, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups,
            tiling_data.heightOut, tiling_data.widthOut,
            tiling_data.tileH, tiling_data.totalPairs);
    op.Process();
}
