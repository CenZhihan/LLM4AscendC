#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dSubtractTanhSubtractAvgPoolCustom {
public:
    __aicore__ inline KernelConv2dSubtractTanhSubtractAvgPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y,
                                uint32_t totalYLen, uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t outC, uint32_t inC, uint32_t kH, uint32_t kW,
                                uint32_t H_conv, uint32_t W_conv, uint32_t K,
                                float sub1, float sub2, uint32_t tileLen) {
        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C * H * W);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outC * inC * kH * kW);
        hasBias_ = (conv_bias != nullptr);
        if (hasBias_)
            bGm_.SetGlobalBuffer((__gm__ float*)conv_bias, outC);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalYLen);

        N_ = N; C_ = C; H_ = H; W_ = W;
        outC_ = outC; inC_ = inC; kH_ = kH; kW_ = kW;
        H_conv_ = H_conv; W_conv_ = W_conv;
        K_ = K;
        sub1_ = sub1; sub2_ = sub2;
        tileLen_ = tileLen;
        totalYLen_ = totalYLen;

        // Precompute strides
        strideN_ = C_ * H_ * W_;
        strideC_ = H_ * W_;
        strideH_ = W_;
        strideOutC_ = H_conv_ * W_conv_;
        strideOutH_ = W_conv_;
        poolArea_ = static_cast<float>(K_ * K_);
        H_out_ = H_conv_ / K_;
        W_out_ = W_conv_ / K_;
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();
        uint32_t total = totalYLen_;
        uint32_t len = tileLen_;
        uint32_t start = blockIdx * len;
        if (start >= total) return;
        uint32_t end = start + len;
        if (end > total) end = total;
        for (uint32_t idx = start; idx < end; ++idx) {
            ProcessOne(idx);
        }
    }

private:
    __aicore__ inline void ProcessOne(uint32_t linearIdx) {
        uint32_t w_out = linearIdx % W_out_;
        uint32_t h_out = (linearIdx / W_out_) % H_out_;
        uint32_t oc = (linearIdx / (W_out_ * H_out_)) % outC_;
        uint32_t n = linearIdx / (W_out_ * H_out_ * outC_);

        float sum = 0.0f;
        for (uint32_t dy = 0; dy < K_; ++dy) {
            for (uint32_t dx = 0; dx < K_; ++dx) {
                uint32_t conv_h = h_out * K_ + dy;
                uint32_t conv_w = w_out * K_ + dx;

                float convVal = 0.0f;
                uint32_t x_n_offset = n * strideN_;
                for (uint32_t ic = 0; ic < inC_; ++ic) {
                    for (uint32_t ky = 0; ky < kH_; ++ky) {
                        for (uint32_t kx = 0; kx < kW_; ++kx) {
                            uint32_t x_idx = x_n_offset + ic * strideC_ + (conv_h + ky) * strideH_ + (conv_w + kx);
                            uint32_t w_idx = oc * inC_ * kH_ * kW_ + ic * kH_ * kW_ + ky * kW_ + kx;
                            convVal += xGm_.GetValue(x_idx) * wGm_.GetValue(w_idx);
                        }
                    }
                }
                if (hasBias_)
                    convVal += bGm_.GetValue(oc);
                float t = convVal - sub1_;
                float tanhVal = TanhApprox(t);
                float finalVal = tanhVal - sub2_;
                sum += finalVal;
            }
        }
        float avg = sum / poolArea_;
        yGm_.SetValue(linearIdx, avg);
    }

    __aicore__ inline float TanhApprox(float x) {
        float e2x = exp(2.0f * x);
        return (e2x - 1.0f) / (e2x + 1.0f);
    }

private:
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;
    bool hasBias_;
    uint32_t N_, C_, H_, W_;
    uint32_t outC_, inC_, kH_, kW_;
    uint32_t H_conv_, W_conv_;
    uint32_t K_;
    float sub1_, sub2_;
    uint32_t tileLen_;
    uint32_t totalYLen_;
    uint32_t strideN_, strideC_, strideH_;
    uint32_t strideOutC_, strideOutH_;
    float poolArea_;
    uint32_t H_out_, W_out_;
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractTanhSubtractAvgPoolCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.totalYLen, tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.outC, tiling_data.inC, tiling_data.kH, tiling_data.kW,
            tiling_data.H_conv, tiling_data.W_conv, tiling_data.K,
            tiling_data.sub1, tiling_data.sub2, tiling_data.tileLen);
    op.Process();
}
