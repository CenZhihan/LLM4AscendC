#include "conv2d_hard_swish_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dHardSwishReluCustomTilingData tiling;

    // Get origin shapes (storage shapes may be different)
    const auto& x_shape = context->GetInputShape(0)->GetOriginShape();
    const auto& w_shape = context->GetInputShape(1)->GetOriginShape();

    uint32_t batch = x_shape.GetDim(0);
    uint32_t inC = x_shape.GetDim(1);
    uint32_t inH = x_shape.GetDim(2);
    uint32_t inW = x_shape.GetDim(3);
    uint32_t outC = w_shape.GetDim(0);
    uint32_t kernelH = w_shape.GetDim(2);
    uint32_t kernelW = w_shape.GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int stride = attrs->GetAttrPointer<int>(0)[0];
    int padding = attrs->GetAttrPointer<int>(1)[0];
    int dilation = attrs->GetAttrPointer<int>(2)[0];
    int groups = attrs->GetAttrPointer<int>(3)[0];

    uint32_t outH = (inH + 2 * padding - dilation * (kernelH - 1) - 1) / stride + 1;
    uint32_t outW = (inW + 2 * padding - dilation * (kernelW - 1) - 1) / stride + 1;

    uint32_t totalLength = batch * outC * outH * outW;
    uint32_t tileLen = UB_BUDGET_BYTES / (4 * 5); // 5 floats per pixel (overestimate)
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;
    if (tileLen > HARD_CAP_ELEMS) tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    if (totalLength < tileLen) tileLen = AlignDown(totalLength, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = totalLength;

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batch(batch);
    tiling.set_inC(inC);
    tiling.set_inH(inH);
    tiling.set_inW(inW);
    tiling.set_outC(outC);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_kernelH(kernelH);
    tiling.set_kernelW(kernelW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float HS_CONST = 1.0f / 6.0f;

class KernelConv2dHardSwishReluCustom {
public:
    __aicore__ inline KernelConv2dHardSwishReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t batch,
                                uint32_t inC, uint32_t inH, uint32_t inW,
                                uint32_t outC, uint32_t outH, uint32_t outW,
                                uint32_t kernelH, uint32_t kernelW)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        batch_ = batch;
        inC_ = inC;
        inH_ = inH;
        inW_ = inW;
        outC_ = outC;
        outH_ = outH;
        outW_ = outW;
        kernelH_ = kernelH;
        kernelW_ = kernelW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * inC * inH * inW + 1);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outC * inC * kernelH * kernelW + 1);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, outC + 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmp1Queue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmp2Queue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, (inC_ * kernelH_ * kernelW_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t stride = blockDim_ * tileLen_;
        uint32_t base = blockIdx * tileLen_;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            ComputeTile(base, curLen);

            base += stride;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t outOffset, uint32_t len)
    {
        for (uint32_t i = 0; i < len; i++) {
            uint32_t globalIdx = outOffset + i;
            uint32_t tmp = globalIdx;
            uint32_t ow = tmp % outW_; tmp /= outW_;
            uint32_t oh = tmp % outH_; tmp /= outH_;
            uint32_t oc = tmp % outC_; tmp /= outC_;
            uint32_t b = tmp;

            float sum = 0.0f;
            for (uint32_t ic = 0; ic < inC_; ic++) {
                for (uint32_t kh = 0; kh < kernelH_; kh++) {
                    for (uint32_t kw = 0; kw < kernelW_; kw++) {
                        int in_h = static_cast<int>(oh * 1 + kh) - 0;
                        int in_w = static_cast<int>(ow * 1 + kw) - 0;
                        if (in_h >= 0 && in_h < static_cast<int>(inH_) &&
                            in_w >= 0 && in_w < static_cast<int>(inW_)) {
                            float x_val = xGm_.GetValue(b * inC_ * inH_ * inW_ +
                                                        ic * inH_ * inW_ +
                                                        in_h * inW_ + in_w);
                            float w_val = wGm_.GetValue(oc * inC_ * kernelH_ * kernelW_ +
                                                        ic * kernelH_ * kernelW_ +
                                                        kh * kernelW_ + kw);
                            sum += x_val * w_val;
                        }
                    }
                }
            }
            if (bGm_.GetValue(oc) != 0.0f) {
                sum += bGm_.GetValue(oc);
            }

            float hs_tmp = sum + 3.0f;
            if (hs_tmp < 0.0f) hs_tmp = 0.0f;
            if (hs_tmp > 6.0f) hs_tmp = 6.0f;
            float hs = sum * hs_tmp * HS_CONST;

            float y_val = hs > 0.0f ? hs : 0.0f;

            yGm_.SetValue(globalIdx, y_val);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp1Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp2Queue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
    uint32_t batch_ {1};
    uint32_t inC_ {1};
    uint32_t inH_ {1};
    uint32_t inW_ {1};
    uint32_t outC_ {1};
    uint32_t outH_ {1};
    uint32_t outW_ {1};
    uint32_t kernelH_ {1};
    uint32_t kernelW_ {1};
};

extern "C" __global__ __aicore__ void conv2d_hard_swish_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dHardSwishReluCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim,
            tiling_data.batch, tiling_data.inC, tiling_data.inH, tiling_data.inW,
            tiling_data.outC, tiling_data.outH, tiling_data.outW,
            tiling_data.kernelH, tiling_data.kernelW);
    op.Process();
}
