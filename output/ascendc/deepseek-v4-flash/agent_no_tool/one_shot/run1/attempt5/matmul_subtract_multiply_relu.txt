kernel_src="""

#include "kernel_operator.h"
#include "matmul.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t TILE_M = 128;
constexpr uint32_t TILE_N = 128;
constexpr uint32_t TILE_K = 128;

class KernelMatmulSubtractMultiplyReluCustom {
public:
    __aicore__ inline KernelMatmulSubtractMultiplyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                float subValue, float mulValue)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        subValue_ = subValue;
        mulValue_ = mulValue;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N_ * K_);  // w stored as (N, K)
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M_ * N_);

        // Init matmul object: transB = true because w is (N,K) but we need wT = (K,N)
        matmul_.SetParameter(M_, N_, K_, TILE_M, TILE_N, TILE_K,
                             false, true, true); // transA=false, transB=true, enableBias=true
        matmul_.Init(&pipe_);
        matmul_.SetOrgShape(M_, N_, K_);

        // Allocate buffers for matmul output (C) and temporary operations
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, TILE_M * TILE_N * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, TILE_M * TILE_N * sizeof(float));
        pipe_.InitBuffer(subMulReluQueue_, 1, TILE_M * TILE_N * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockIdx = GetBlockIdx();
        const uint32_t blockNum = GetBlockNum();

        // Each core processes a subset of rows (M dimension)
        uint32_t rowsPerBlock = (M_ + blockNum - 1) / blockNum;
        uint32_t rowStart = blockIdx * rowsPerBlock;
        uint32_t rowEnd = (rowStart + rowsPerBlock > M_) ? M_ : rowStart + rowsPerBlock;
        uint32_t numRows = rowEnd - rowStart;

        if (numRows == 0) return;

        // Tiling over rows and columns
        for (uint32_t mOffset = rowStart; mOffset < rowEnd; mOffset += TILE_M) {
            uint32_t curM = TILE_M;
            if (mOffset + curM > rowEnd) curM = rowEnd - mOffset;

            for (uint32_t nOffset = 0; nOffset < N_; nOffset += TILE_N) {
                uint32_t curN = TILE_N;
                if (nOffset + curN > N_) curN = N_ - nOffset;

                // Set sub-block for matmul
                matmul_.SetSubBlock(curM, curN, K_);

                // Allocate output tensor for matmul
                LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();

                // Perform matmul with bias: C = A * B + bias
                // where A = x, B = w transposed, bias = b
                matmul_.Iterate(outLocal, xGm_, wGm_, bGm_);

                // Subtract subValue
                LocalTensor<float> tmpLocal = subMulReluQueue_.AllocTensor<float>();
                Subs(tmpLocal, outLocal, subValue_, curM * curN);

                // Multiply mulValue
                LocalTensor<float> mulTmp = tmpQueue_.AllocTensor<float>();
                Muls(mulTmp, tmpLocal, mulValue_, curM * curN);

                // ReLU: max(0, mulTmp)
                Maxs(outLocal, mulTmp, 0.0f, curM * curN);

                // Free temporary tensors
                tmpQueue_.FreeTensor(mulTmp);
                subMulReluQueue_.FreeTensor(tmpLocal);

                // Copy out to global
                DataCopy(yGm_[mOffset * N_ + nOffset], outLocal, curM * curN);
                outQueueY_.FreeTensor(outLocal);
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;
    TQue<TPosition::VECCALC, 1> subMulReluQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    Matmul<float, float, float> matmul_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    float subValue_ {0.0f};
    float mulValue_ {1.0f};
};

extern "C" __global__ __aicore__ void matmul_subtract_multiply_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSubtractMultiplyReluCustom op;
    op.Init(x, w, b, y, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.subValue, tiling_data.mulValue);
    op.Process();
}
