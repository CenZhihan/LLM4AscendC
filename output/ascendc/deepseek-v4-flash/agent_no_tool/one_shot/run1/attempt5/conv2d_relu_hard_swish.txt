kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dReluHardSwishCustom {
public:
    __aicore__ inline KernelConv2dReluHardSwishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR y, GM_ADDR tiling)
    {
        // read tiling data
        GET_TILING_DATA(td, tiling);

        totalLen_ = td.totalLength;
        tileLen_ = td.tileLength;
        N_ = td.N;
        Cin_ = td.Cin;
        Cout_ = td.Cout;
        H_ = td.H;
        W_ = td.W;
        K_ = td.K;
        H_out_ = td.H_out;
        W_out_ = td.W_out;
        strideH_ = td.strideH;
        strideW_ = td.strideW;
        padH_ = td.padH;
        padW_ = td.padW;
        dilationH_ = td.dilationH;
        dilationW_ = td.dilationW;
        hasBias_ = td.hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, Cout_ * Cin_ * K_ * K_);
        if (hasBias_)
            bGm_.SetGlobalBuffer((__gm__ float*)bias, Cout_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Allocate local memory for weight and bias (once per block)
        pipe_.InitBuffer(wLbuf_, 1, Cout_ * Cin_ * K_ * K_ * sizeof(float));
        if (hasBias_)
            pipe_.InitBuffer(bLbuf_, 1, Cout_ * sizeof(float));
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));

        // Copy weight and bias to local
        AscendC::LocalTensor<float> wLocal = wLbuf_.Get<float>();
        AscendC::DataCopy(wLocal, wGm_, Cout_ * Cin_ * K_ * K_);
        if (hasBias_) {
            AscendC::LocalTensor<float> bLocal = bLbuf_.Get<float>();
            AscendC::DataCopy(bLocal, bGm_, Cout_);
        }
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Prefetch first tile
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen, base);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        // For convolution, we only need to load the output values? Actually we compute on the fly,
        // so we don't preload input tiles. This queue is used as scratch for temporary result storage.
        // We'll just allocate and enqueue dummy tensors to keep queue structure.
        // Alternatively, we can skip prefetch. Simplify: bypass queues for conv.
        // Override: we will not use CopyIn prefetch; compute directly.
        (void)gmOffset;
        (void)len;
    }

    __aicore__ inline void Compute(uint32_t len, uint32_t outOffset)
    {
        // Process 'len' output elements starting at global index outOffset.
        // Each output element is indexed by (n, cout, h, w).
        // We'll compute sequentially within the block.
        // Since len <= tileLen, we can process in a loop.
        AscendC::LocalTensor<float> oLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            uint32_t idx = outOffset + i;
            // Decode idx to (n, cout, h, w)
            uint32_t w = idx % W_out_;
            uint32_t temp = idx / W_out_;
            uint32_t h = temp % H_out_;
            temp /= H_out_;
            uint32_t cout = temp % Cout_;
            uint32_t n = temp / Cout_;

            // Compute convolution
            float sum = 0.0f;
            for (uint32_t c = 0; c < Cin_; ++c) {
                for (uint32_t ki = 0; ki < K_; ++ki) {
                    for (uint32_t kj = 0; kj < K_; ++kj) {
                        int32_t ih = static_cast<int32_t>(h) * strideH_ + ki * dilationH_ - padH_;
                        int32_t iw = static_cast<int32_t>(w) * strideW_ + kj * dilationW_ - padW_;
                        if (ih >= 0 && ih < static_cast<int32_t>(H_) && iw >= 0 && iw < static_cast<int32_t>(W_)) {
                            uint32_t xIdx = ((n * Cin_ + c) * H_ + ih) * W_ + iw;
                            uint32_t wIdx = ((cout * Cin_ + c) * K_ + ki) * K_ + kj;
                            sum += xGm_.GetValue(xIdx) * wGm_.GetValue(wIdx);
                        }
                    }
                }
            }
            if (hasBias_) {
                sum += bGm_.GetValue(cout);
            }
            // ReLU
            if (sum < 0.0f) sum = 0.0f;
            // HardSwish: x * clamp((x+3)/6, 0, 1)
            float t = (sum + 3.0f) / 6.0f;
            if (t < 0.0f) t = 0.0f;
            if (t > 1.0f) t = 1.0f;
            sum = sum * t;
            oLocal.SetValue(i, sum);
        }

        outQueueY_.EnQue(oLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::TBuf<AscendC::TPosition::VECCALC> wLbuf_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> bLbuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_, Cin_, Cout_, H_, W_, K_;
    uint32_t H_out_, W_out_;
    uint32_t strideH_, strideW_, padH_, padW_, dilationH_, dilationW_;
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelConv2dReluHardSwishCustom op;
    op.Init(x, weight, bias, y, tiling);
    op.Process();
}
