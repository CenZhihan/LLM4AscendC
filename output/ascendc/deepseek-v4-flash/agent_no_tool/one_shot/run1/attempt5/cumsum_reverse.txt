#include "cumsum_reverse_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cstdint>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumReverseCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim_ptr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = dim_ptr ? *dim_ptr : 1;

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    int32_t ndim = shape.GetDimNum();
    if (dim < 0) dim += ndim;
    uint32_t dimLength = static_cast<uint32_t>(shape.GetDim(dim));

    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (dimLength < tileLen) {
        tileLen = AlignDown(dimLength, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = dimLength;
    }

    uint32_t numRows = totalLength / dimLength;
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dimLength(dimLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t VEC_LEN = 8;  // vector length for float

class KernelCumsumReverseCustom {
public:
    __aicore__ inline KernelCumsumReverseCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t dimLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        dimLen_ = dimLength;
        rowNum_ = totalLength / dimLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || dimLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t r = blockIdx; r < rowNum_; r += blockNum) {
            float runningSum = 0.0f;
            uint32_t rowStart = r * dimLen_;
            uint32_t numTiles = (dimLen_ + tileLen_ - 1) / tileLen_;

            for (int32_t t = static_cast<int32_t>(numTiles) - 1; t >= 0; --t) {
                uint32_t tileStart = static_cast<uint32_t>(t) * tileLen_;
                uint32_t tileEnd = tileStart + tileLen_;
                if (tileEnd > dimLen_) tileEnd = dimLen_;
                uint32_t curLen = tileEnd - tileStart;

                // --- Load tile from global to local ---
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                // Ensure aligned copy, handle remainder with scalar
                uint32_t alignedLen = (curLen / VEC_LEN) * VEC_LEN;
                if (alignedLen > 0) {
                    AscendC::DataCopy(xLocal, xGm_[rowStart + tileStart], alignedLen);
                }
                for (uint32_t i = alignedLen; i < curLen; i++) {
                    xLocal.SetValue(i, xGm_.GetValue(rowStart + tileStart + i));
                }
                inQueueX_.EnQue(xLocal);

                // --- Dequeue input, compute output ---
                AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                for (int32_t i = static_cast<int32_t>(curLen) - 1; i >= 0; --i) {
                    float val = xDeque.GetValue(i);
                    runningSum += val;
                    yLocal.SetValue(i, runningSum);
                }

                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xDeque);

                // --- Dequeue output and write to global ---
                AscendC::LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
                alignedLen = (curLen / VEC_LEN) * VEC_LEN;
                if (alignedLen > 0) {
                    AscendC::DataCopy(yGm_[rowStart + tileStart], yDeque, alignedLen);
                }
                for (uint32_t i = alignedLen; i < curLen; i++) {
                    yGm_.SetValue(rowStart + tileStart + i, yDeque.GetValue(i));
                }
                outQueueY_.FreeTensor(yDeque);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t dimLen_ {0};
    uint32_t rowNum_ {0};
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverseCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.dimLength);
    op.Process();
}
