#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t batch, uint32_t inC, uint32_t outC,
                                uint32_t inH, uint32_t inW, uint32_t kH, uint32_t kW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t outputPadH, uint32_t outputPadW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups,
                                uint32_t outH, uint32_t outW,
                                uint32_t totalOutputLength, uint32_t tileLength,
                                uint32_t blockDim)
    {
        batch_ = batch;
        inC_ = inC;
        outC_ = outC;
        inH_ = inH;
        inW_ = inW;
        kH_ = kH;
        kW_ = kW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        outputPadH_ = outputPadH;
        outputPadW_ = outputPadW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;
        outH_ = outH;
        outW_ = outW;
        totalLen_ = totalOutputLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        // Input shapes (NCHW)
        inputStrideC_ = inH_ * inW_;
        inputStrideN_ = inC_ * inputStrideC_;
        // Weight shape: [inC, outC/groups, kH, kW] -> but groups=1 -> [inC, outC, kH, kW]
        weightStrideKH_ = kW_;
        weightStrideC_ = kH_ * weightStrideKH_;
        weightStrideInC_ = outC_ * weightStrideC_; // actually outC is per inC
        // Output shape
        outputStrideW_ = 1;
        outputStrideH_ = outW_;
        outputStrideC_ = outH_ * outputStrideH_;
        outputStrideN_ = outC_ * outputStrideC_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_); // not correct: x is not totalLen but we only access via offsets
        // Actually we need proper size: x size = batch * inC * inH * inW
        uint32_t xSize = batch_ * inC_ * inH_ * inW_;
        uint32_t wSize = inC_ * outC_ * kH_ * kW_;
        uint32_t ySize = totalLen_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // No local buffers needed for this simple implementation
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0) return;
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t blockNum = blockDim_;

        // Grid-stride over output elements
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            // Process one tile: compute output elements from base to base+curLen-1
            for (uint32_t i = 0; i < curLen; i++) {
                uint32_t idx = base + i;
                // Convert linear index to (n, oc, oh, ow)
                uint32_t n = idx / outputStrideN_;
                uint32_t rest = idx % outputStrideN_;
                uint32_t oc = rest / outputStrideC_;
                rest %= outputStrideC_;
                uint32_t oh = rest / outputStrideH_;
                uint32_t ow = rest % outputStrideH_;

                float sum = 0.0f;
                // Loop over input channels (all inC, since groups=1)
                for (uint32_t ic = 0; ic < inC_; ic++) {
                    // Loop over kernel
                    for (uint32_t kh = 0; kh < kH_; kh++) {
                        for (uint32_t kw = 0; kw < kW_; kw++) {
                            // Compute input spatial indices that contribute to this output
                            // Formula: oh = ih*strideH + dilationH*kh + padH - outputPadH
                            // Solve for ih:
                            // ih = (oh - dilationH*kh - padH + outputPadH) / strideH
                            int32_t numerator_h = static_cast<int32_t>(oh) - static_cast<int32_t>(dilationH_ * kh) - static_cast<int32_t>(padH_) + static_cast<int32_t>(outputPadH_);
                            if (numerator_h < 0) continue;
                            if (numerator_h % static_cast<int32_t>(strideH_) != 0) continue;
                            int32_t ih = numerator_h / static_cast<int32_t>(strideH_);
                            if (ih < 0 || ih >= static_cast<int32_t>(inH_)) continue;

                            int32_t numerator_w = static_cast<int32_t>(ow) - static_cast<int32_t>(dilationW_ * kw) - static_cast<int32_t>(padW_) + static_cast<int32_t>(outputPadW_);
                            if (numerator_w < 0) continue;
                            if (numerator_w % static_cast<int32_t>(strideW_) != 0) continue;
                            int32_t iw = numerator_w / static_cast<int32_t>(strideW_);
                            if (iw < 0 || iw >= static_cast<int32_t>(inW_)) continue;

                            // Accumulate
                            // Input offset: n*batchStride + ic*inStrideC + ih*inStrideH + iw
                            uint32_t xOff = n * inputStrideN_ + ic * inputStrideC_ + static_cast<uint32_t>(ih) * inW_ + static_cast<uint32_t>(iw);
                            // Weight offset: ic*weightStrideInC + oc*weightStrideC + kh*weightStrideKH + kw
                            uint32_t wOff = ic * weightStrideInC_ + oc * weightStrideC_ + kh * weightStrideKH_ + kw;
                            sum += xGm_.GetValue(xOff) * wGm_.GetValue(wOff);
                        }
                    }
                }
                yGm_.SetValue(idx, sum);
            }

            // Move to next tile (grid-stride)
            base += blockNum * tileLen_;
            if (base >= totalLen_) break;
            uint32_t nextLen = totalLen_ - base;
            if (nextLen > tileLen_) nextLen = tileLen_;
            if (nextLen == 0) break;
            // Continue loop
        }
    }

private:
    // Shape parameters
    uint32_t batch_, inC_, outC_, inH_, inW_, kH_, kW_;
    uint32_t strideH_, strideW_, padH_, padW_, outputPadH_, outputPadW_;
    uint32_t dilationH_, dilationW_, groups_;
    uint32_t outH_, outW_;
    uint32_t totalLen_, tileLen_, blockDim_;

    // Strides
    uint32_t inputStrideC_, inputStrideN_;
    uint32_t weightStrideKH_, weightStrideC_, weightStrideInC_;
    uint32_t outputStrideW_, outputStrideH_, outputStrideC_, outputStrideN_;

    // Global tensors
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.batch, tiling_data.inChannels, tiling_data.outChannels,
            tiling_data.inHeight, tiling_data.inWidth,
            tiling_data.kernelHeight, tiling_data.kernelWidth,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.outputPadH, tiling_data.outputPadW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups,
            tiling_data.outHeight, tiling_data.outWidth,
            tiling_data.totalOutputLength, tiling_data.tileLength,
            tiling_data.blockDim);
    op.Process();
}
