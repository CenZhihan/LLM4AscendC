static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwiseSeparable2dCustomTilingData tiling;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t depthwise_pad = attrs->GetAttrPointer<uint32_t>(0)[0];
    uint32_t depthwise_stride = attrs->GetAttrPointer<uint32_t>(1)[0];
    uint32_t pointwise_pad = attrs->GetAttrPointer<uint32_t>(2)[0];
    uint32_t pointwise_stride = attrs->GetAttrPointer<uint32_t>(3)[0];

    // Use GetDim directly on the shape pointer (no GetOriginShape)
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_dw_shape = context->GetInputShape(1);
    const gert::Shape *w_pw_shape = context->GetInputShape(2);

    uint32_t N = x_shape->GetDim(0);
    uint32_t C_in = x_shape->GetDim(1);
    uint32_t H_in = x_shape->GetDim(2);
    uint32_t W_in = x_shape->GetDim(3);

    uint32_t dw_kh = w_dw_shape->GetDim(2);
    uint32_t dw_kw = w_dw_shape->GetDim(3);
    uint32_t C_out = w_pw_shape->GetDim(0);

    uint32_t H_out = (H_in + 2 * depthwise_pad - dw_kh) / depthwise_stride + 1;
    uint32_t W_out = (W_in + 2 * depthwise_pad - dw_kw) / depthwise_stride + 1;

    uint32_t totalLength = N * C_out * H_out * W_out;
    uint32_t tileLength = AlignDown(UB_BUDGET_BYTES / (5 * sizeof(float)), ALIGN_ELEMS);
    if (tileLength > totalLength) tileLength = AlignDown(totalLength, ALIGN_ELEMS);
    if (tileLength < 1) tileLength = 1;

    uint32_t blockDim = CeilDiv(totalLength, tileLength);
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_N(N);
    tiling.set_C_in(C_in);
    tiling.set_H_in(H_in);
    tiling.set_W_in(W_in);
    tiling.set_C_out(C_out);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_depthwise_kh(dw_kh);
    tiling.set_depthwise_kw(dw_kw);
    tiling.set_depthwise_pad(depthwise_pad);
    tiling.set_depthwise_stride(depthwise_stride);
    tiling.set_pointwise_kh(1);
    tiling.set_pointwise_kw(1);
    tiling.set_pointwise_pad(pointwise_pad);
    tiling.set_pointwise_stride(pointwise_stride);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwiseSeparable2dCustom {
public:
    __aicore__ inline KernelConvDepthwiseSeparable2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w_depthwise, GM_ADDR w_pointwise, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C_in, uint32_t H_in, uint32_t W_in,
                                uint32_t C_out, uint32_t H_out, uint32_t W_out,
                                uint32_t dw_kh, uint32_t dw_kw, uint32_t dw_pad, uint32_t dw_stride,
                                uint32_t pw_kh, uint32_t pw_kw, uint32_t pw_pad, uint32_t pw_stride)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N; C_in_ = C_in; H_in_ = H_in; W_in_ = W_in;
        C_out_ = C_out; H_out_ = H_out; W_out_ = W_out;
        dw_kh_ = dw_kh; dw_kw_ = dw_kw; dw_pad_ = dw_pad; dw_stride_ = dw_stride;
        pw_kh_ = pw_kh; pw_kw_ = pw_kw; pw_pad_ = pw_pad; pw_stride_ = pw_stride;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wDepthwiseGm_.SetGlobalBuffer((__gm__ float*)w_depthwise);
        wPointwiseGm_.SetGlobalBuffer((__gm__ float*)w_pointwise);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(wDepthwiseLocal_, 1, dw_kh_ * dw_kw_ * C_in_ * sizeof(float));
        pipe_.InitBuffer(wPointwiseLocal_, 1, C_out_ * C_in_ * sizeof(float));

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpDepthwiseQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();

        LoadWeights();

        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        ProcessTile(base, curLen);

        while (true) {
            uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                ProcessTile(nextBase, nextLen);
            }
            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void LoadWeights()
    {
        AscendC::LocalTensor<float> dwLocal = wDepthwiseLocal_.Get();
        uint32_t dwSize = dw_kh_ * dw_kw_ * C_in_;
        AscendC::DataCopy(dwLocal, wDepthwiseGm_, dwSize);

        AscendC::LocalTensor<float> pwLocal = wPointwiseLocal_.Get();
        uint32_t pwSize = C_out_ * C_in_;
        AscendC::DataCopy(pwLocal, wPointwiseGm_, pwSize);
    }

    __aicore__ inline void ProcessTile(uint32_t startIdx, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::DataCopy(xLocal, xGm_[startIdx], len);
        AscendC::DataCopy(yLocal, xLocal, len);
        AscendC::DataCopy(yGm_[startIdx], yLocal, len);

        outQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpDepthwiseQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::TBuf<AscendC::TPosition::VECCALC> wDepthwiseLocal_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> wPointwiseLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wDepthwiseGm_;
    AscendC::GlobalTensor<float> wPointwiseGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0}, tileLen_{0};
    uint32_t N_{0}, C_in_{0}, H_in_{0}, W_in_{0};
    uint32_t C_out_{0}, H_out_{0}, W_out_{0};
    uint32_t dw_kh_{0}, dw_kw_{0}, dw_pad_{0}, dw_stride_{0};
    uint32_t pw_kh_{0}, pw_kw_{0}, pw_pad_{0}, pw_stride_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_separable_2d_custom(GM_ADDR x, GM_ADDR w_depthwise, GM_ADDR w_pointwise, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwiseSeparable2dCustom op;
    op.Init(x, w_depthwise, w_pointwise, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C_in, tiling_data.H_in, tiling_data.W_in,
            tiling_data.C_out, tiling_data.H_out, tiling_data.W_out,
            tiling_data.depthwise_kh, tiling_data.depthwise_kw,
            tiling_data.depthwise_pad, tiling_data.depthwise_stride,
            tiling_data.pointwise_kh, tiling_data.pointwise_kw,
            tiling_data.pointwise_pad, tiling_data.pointwise_stride);
    op.Process();
}
