host_operator_src="""
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget accounts for double buffering: 2 * (tileM*tileK + tileK*tileN) + tileM*tileN floats
static constexpr uint32_t UB_BUDGET_FLOATS = 48 * 1024;
static constexpr uint32_t ALIGN_DIM = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a)   { return ( (x + a - 1) / a ) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)   { return (a + b - 1) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    auto shapeA = context->GetInputShape(0)->GetOriginShape();
    auto shapeB = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = static_cast<uint32_t>(shapeA.GetDim(0));
    uint32_t K = static_cast<uint32_t>(shapeA.GetDim(1));
    uint32_t N = static_cast<uint32_t>(shapeB.GetDim(1));

    if (M == 0 || N == 0 || K == 0) return ge::GRAPH_FAILED;

    uint32_t tileK = AlignDown(std::min(K, 256u), ALIGN_DIM);
    if (tileK == 0) tileK = ALIGN_DIM;
    uint32_t tileM = AlignDown(std::min(M, 512u), ALIGN_DIM);
    uint32_t tileN = AlignDown(std::min(N, 512u), ALIGN_DIM);
    // Account for double buffering: 2*tileM*tileK + 2*tileK*tileN + tileM*tileN <= UB_BUDGET_FLOATS
    while (2*tileM*tileK + 2*tileK*tileN + tileM*tileN > UB_BUDGET_FLOATS) {
        if (tileM > tileN) tileM = AlignDown(tileM / 2, ALIGN_DIM);
        else               tileN = AlignDown(tileN / 2, ALIGN_DIM);
        if (tileM < ALIGN_DIM) tileM = ALIGN_DIM;
        if (tileN < ALIGN_DIM) tileN = ALIGN_DIM;
    }

    uint32_t numRowTiles = CeilDiv(M, tileM);
    uint32_t blockDim = std::min(numRowTiles, 24u);
    if (blockDim < 8) blockDim = 8;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::Shape *shapeB = context->GetInputShape(1);
    gert::Shape shapeC = {shapeA->GetDim(0), shapeB->GetDim(1)};
    *context->GetOutputShape(0) = shapeC;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileK_ = tileK; tileN_ = tileN;

        A_gm_.SetGlobalBuffer((__gm__ float*)A, M_ * K_);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, K_ * N_);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, M_ * N_);

        pipe_.InitBuffer(inQueA_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueB_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQue_, 1, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockId = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t numRowTiles = (M_ + tileM_ - 1) / tileM_;
        uint32_t startTileRow = blockId;
        for (uint32_t tRow = startTileRow; tRow < numRowTiles; tRow += blockNum) {
            uint32_t rowBase = tRow * tileM_;
            uint32_t curTileM = (rowBase + tileM_ > M_) ? (M_ - rowBase) : tileM_;

            uint32_t numColTiles = (N_ + tileN_ - 1) / tileN_;
            uint32_t numKTiles   = (K_ + tileK_ - 1) / tileK_;

            for (uint32_t tCol = 0; tCol < numColTiles; ++tCol) {
                uint32_t colBase = tCol * tileN_;
                uint32_t curTileN = (colBase + tileN_ > N_) ? (N_ - colBase) : tileN_;

                AscendC::LocalTensor<float> accLocal = tmpQue_.AllocTensor<float>();
                // Zero the entire accumulator buffer using Dup (safe vs Muls with uninitialized)
                AscendC::Dup(accLocal, 0.0f, tileM_ * tileN_);

                for (uint32_t kIdx = 0; kIdx < numKTiles; ++kIdx) {
                    uint32_t kBase = kIdx * tileK_;
                    uint32_t curTileK = (kBase + tileK_ > K_) ? (K_ - kBase) : tileK_;

                    PipelineIn(rowBase, colBase, kBase, curTileM, curTileK, curTileN);
                    ComputePartial(curTileM, curTileK, curTileN, accLocal);
                }

                PipelineOut(rowBase, colBase, curTileM, curTileN, accLocal);
                tmpQue_.FreeTensor(accLocal);
            }
        }
    }

private:
    __aicore__ inline void PipelineIn(uint32_t rowBase, uint32_t colBase,
                                       uint32_t kBase,
                                       uint32_t curTileM, uint32_t curTileK, uint32_t curTileN)
    {
        // Copy A tile: row-by-row because global rows have stride K_
        auto aLocal = inQueA_.AllocTensor<float>();
        for (uint32_t i = 0; i < curTileM; ++i) {
            AscendC::DataCopy(aLocal[i * curTileK],
                              A_gm_[(rowBase + i) * K_ + kBase],
                              curTileK);
        }
        inQueA_.EnQue(aLocal);

        // Copy B tile: row-by-row because global rows have stride N_
        auto bLocal = inQueB_.AllocTensor<float>();
        for (uint32_t k = 0; k < curTileK; ++k) {
            AscendC::DataCopy(bLocal[k * curTileN],
                              B_gm_[(kBase + k) * N_ + colBase],
                              curTileN);
        }
        inQueB_.EnQue(bLocal);
    }

    __aicore__ inline void ComputePartial(uint32_t curTileM, uint32_t curTileK,
                                           uint32_t curTileN,
                                           AscendC::LocalTensor<float>& accum)
    {
        auto aLocal = inQueA_.DeQue<float>();
        auto bLocal = inQueB_.DeQue<float>();

        // Compute C = A * B for the tile, accumulate into accum (row-major, stride tileN_)
        for (uint32_t i = 0; i < curTileM; ++i) {
            for (uint32_t j = 0; j < curTileN; ++j) {
                float sum = 0.0f;
                for (uint32_t k = 0; k < curTileK; ++k) {
                    sum += aLocal.GetValue(i * curTileK + k) *
                           bLocal.GetValue(k * curTileN + j);
                }
                uint32_t idx = i * tileN_ + j;
                float old = accum.GetValue(idx);
                accum.SetValue(idx, old + sum);
            }
        }

        inQueA_.FreeTensor(aLocal);
        inQueB_.FreeTensor(bLocal);
    }

    __aicore__ inline void PipelineOut(uint32_t rowBase, uint32_t colBase,
                                        uint32_t curTileM, uint32_t curTileN,
                                        AscendC::LocalTensor<float>& accum)
    {
        // Copy result rows from local accumulator to global C (row-major, stride N_)
        for (uint32_t i = 0; i < curTileM; ++i) {
            AscendC::DataCopy(C_gm_[(rowBase + i) * N_ + colBase],
                              accum[i * tileN_],
                              curTileN);
        }
    }

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQue_;

    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t M_, N_, K_;
    uint32_t tileM_, tileK_, tileN_;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
"""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor tall_skinny_matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, "Inputs must be 2D");
    TORCH_CHECK(A.size(1) == B.size(0), "Inner dimensions must match");
    at::Tensor C = at::empty({A.size(0), B.size(1)}, A.options());
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu,
          "Tall-skinny matrix multiplication (custom NPU)");
}
