static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;

    // Use GetOriginShape() to obtain Shape (not StorageShape*)
    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape &w_shape = context->GetInputShape(1)->GetOriginShape();

    uint32_t M = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t K = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t N = static_cast<uint32_t>(w_shape.GetDim(0));

    uint32_t tileM = 128;
    uint32_t tileN = 128;
    uint32_t tileK = 256;

    tileM = AlignUp(tileM, ALIGN_UNIT);
    tileN = AlignUp(tileN, ALIGN_UNIT);
    tileK = AlignUp(tileK, ALIGN_UNIT);

    if (tileM > M) tileM = M;
    if (tileN > N) tileN = N;
    if (tileK > K) tileK = K;

    uint32_t numTilesM = (M + tileM - 1) / tileM;
    uint32_t blockDim = numTilesM;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

kernel_src="""

#include "kernel_operator.h"
#include "matmul_api.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmAddReluCustom {
public:
    __aicore__ inline KernelGemmAddReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M * K);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N * K);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, N);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M * N);

        matmul_.SetTensorA(xGm_, M, K);
        matmul_.SetTensorB(wGm_, N, K);
        matmul_.SetTransB(true);
        matmul_.SetTensorC(yGm_, M, N);
        matmul_.SetBias(biasGm_, N);
        matmul_.SetActivationType(ActivationType::RELU);

        matmul_.SetTileM(tileM_);
        matmul_.SetTileN(tileN_);
        matmul_.SetTileK(tileK_);

        matmul_.Init();
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        uint32_t tilesM = (M_ + tileM_ - 1) / tileM_;
        uint32_t tilesPerBlock = (tilesM + blockNum - 1) / blockNum;
        uint32_t startTileM = blockIdx * tilesPerBlock;
        uint32_t endTileM = startTileM + tilesPerBlock;
        if (startTileM >= tilesM) return;
        if (endTileM > tilesM) endTileM = tilesM;

        for (uint32_t tm = startTileM; tm < endTileM; tm++) {
            uint32_t startRow = tm * tileM_;
            uint32_t numRows = tileM_;
            if (startRow + numRows > M_) numRows = M_ - startRow;

            matmul_.SetA(startRow, 0);
            matmul_.SetC(startRow, 0);
            matmul_.Compute();
        }
    }

private:
    MatMul<MatMulType<GM_ADDR, GM_ADDR, GM_ADDR>> matmul_;
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> biasGm_;
    GlobalTensor<float> yGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {128};
    uint32_t tileN_ {128};
    uint32_t tileK_ {256};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddReluCustom op;
    op.Init(x, w, bias, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
