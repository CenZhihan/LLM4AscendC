namespace optiling {

static constexpr uint32_t BLOCK_M = 128;
static constexpr uint32_t BLOCK_N = 128;
static constexpr uint32_t BLOCK_K = 128;
static constexpr uint32_t ALIGN = 32;

static inline uint32_t AlignUp(uint32_t x, uint32_t a) {
    return ((x + a - 1) / a) * a;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MatmulDivideGeluCustomTilingData tiling;
    const auto x_shape = context->GetInputShape(0)->GetOriginShape();
    const auto w_shape = context->GetInputShape(1)->GetOriginShape();
    uint32_t M = x_shape.GetDim(0);
    uint32_t K = x_shape.GetDim(1);
    uint32_t N = w_shape.GetDim(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *divisor = attrs->GetAttrPointer<float>(0);

    uint32_t blockM = BLOCK_M;
    uint32_t blockN = BLOCK_N;
    uint32_t blockK = BLOCK_K;

    blockM = AlignUp(blockM, ALIGN);
    blockN = AlignUp(blockN, ALIGN);
    blockK = AlignUp(blockK, ALIGN);

    if (blockM > M) blockM = AlignDown(M, ALIGN);
    if (blockM == 0) blockM = M;
    if (blockN > N) blockN = AlignDown(N, ALIGN);
    if (blockN == 0) blockN = N;
    if (blockK > K) blockK = AlignDown(K, ALIGN);
    if (blockK == 0) blockK = K;

    tiling.set_totalM(M);
    tiling.set_totalN(N);
    tiling.set_totalK(K);
    tiling.set_blockM(blockM);
    tiling.set_blockN(blockN);
    tiling.set_blockK(blockK);
    tiling.set_divisor(*divisor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    static constexpr uint32_t cubeNum = 4;
    context->SetBlockDim(cubeNum);

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t BLOCK_SIZE = 128;

class KernelMatmulDivideGeluCustom {
public:
    __aicore__ inline KernelMatmulDivideGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalM, uint32_t totalN, uint32_t totalK,
                                uint32_t blockM, uint32_t blockN, uint32_t blockK,
                                float divisor) {
        totalM_ = totalM;
        totalN_ = totalN;
        totalK_ = totalK;
        blockM_ = (blockM == 0) ? totalM : blockM;
        blockN_ = (blockN == 0) ? totalN : blockN;
        blockK_ = (blockK == 0) ? totalK : blockK;
        divisor_ = divisor;
        invDivisor_ = 1.0f / divisor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalM_ * totalK_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, totalN_ * totalK_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, totalN_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalM_ * totalN_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, blockM_ * blockN_ * sizeof(float));
        pipe_.InitBuffer(biasQueue_, 1, blockN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, blockM_ * blockN_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalM_ == 0 || totalN_ == 0 || totalK_ == 0) return;
        uint32_t blockIdx = GetBlockIdx();
        uint32_t blockNum = GetBlockNum();

        for (uint32_t m0 = blockIdx * blockM_; m0 < totalM_; m0 += blockNum * blockM_) {
            uint32_t curM = (m0 + blockM_ <= totalM_) ? blockM_ : (totalM_ - m0);
            if (curM == 0) break;

            for (uint32_t n0 = 0; n0 < totalN_; n0 += blockN_) {
                uint32_t curN = (n0 + blockN_ <= totalN_) ? blockN_ : (totalN_ - n0);
                if (curN == 0) continue;

                LocalTensor<float> biasLocal = biasQueue_.AllocTensor<float>();
                DataCopy(biasLocal, bGm_[n0], curN);

                LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();
                // Initialize outLocal to zero
                for (uint32_t i = 0; i < curM * curN; i++) {
                    outLocal.SetValue(i, 0.0f);
                }
                // Add bias (broadcast across rows)
                for (uint32_t r = 0; r < curM; r++) {
                    for (uint32_t c = 0; c < curN; c++) {
                        float biasVal = biasLocal.GetValue(c);
                        float old = outLocal.GetValue(r * curN + c);
                        outLocal.SetValue(r * curN + c, old + biasVal);
                    }
                }
                biasQueue_.FreeTensor(biasLocal);

                // Accumulate over K tiles
                for (uint32_t k0 = 0; k0 < totalK_; k0 += blockK_) {
                    uint32_t curK = (k0 + blockK_ <= totalK_) ? blockK_ : (totalK_ - k0);
                    if (curK == 0) continue;

                    LocalTensor<float> xTile = tmpQueue_.AllocTensor<float>();
                    DataCopy(xTile, xGm_[m0 * totalK_ + k0], curM * curK);

                    LocalTensor<float> wTile = tmpQueue_.AllocTensor<float>();
                    DataCopy(wTile, wGm_[n0 * totalK_ + k0], curN * curK);

                    // Manual dot product using GetValue/SetValue
                    for (uint32_t i = 0; i < curM; i++) {
                        for (uint32_t j = 0; j < curN; j++) {
                            float sum = 0.0f;
                            for (uint32_t k = 0; k < curK; k++) {
                                float xVal = xTile.GetValue(i * curK + k);
                                float wVal = wTile.GetValue(j * curK + k);
                                sum += xVal * wVal;
                            }
                            uint32_t idx = i * curN + j;
                            float old = outLocal.GetValue(idx);
                            outLocal.SetValue(idx, old + sum);
                        }
                    }

                    tmpQueue_.FreeTensor(xTile);
                    tmpQueue_.FreeTensor(wTile);
                }

                // Divide by divisor
                for (uint32_t i = 0; i < curM * curN; i++) {
                    float val = outLocal.GetValue(i);
                    outLocal.SetValue(i, val * invDivisor_);
                }

                // GELU approximation using sigmoid: x * sigmoid(1.702 * x)
                // First compute t = 1.702 * outLocal
                for (uint32_t i = 0; i < curM * curN; i++) {
                    float xVal = outLocal.GetValue(i);
                    float sigArg = 1.702f * xVal;
                    // Compute sigmoid: 1 / (1 + exp(-sigArg))
                    // We'll compute exp(-sigArg) using Exp on a local tensor
                    // But we need a temporary tensor for that.
                    // For simplicity, compute in-place using scalar math.
                    // However, Exp is not available as scalar. So we use a local tensor.
                }

                // Use a temporary tensor for exponential (reuse tmpQueue_)
                LocalTensor<float> expTmp = tmpQueue_.AllocTensor<float>();
                // Copy outLocal to expTmp, then multiply by 1.702, then negate, then exp, then reciprocal, then multiply
                for (uint32_t i = 0; i < curM * curN; i++) {
                    float xVal = outLocal.GetValue(i);
                    expTmp.SetValue(i, -1.702f * xVal);
                }
                // Exponential
                Exp(expTmp, expTmp, curM * curN);
                // Add 1
                Adds(expTmp, expTmp, 1.0f, curM * curN);
                // Divide outLocal by expTmp (i.e., out = out * 1/(1+exp(-1.702*out)) = out * sigmoid)
                Div(outLocal, outLocal, expTmp, curM * curN);
                tmpQueue_.FreeTensor(expTmp);

                // Copy out to global
                DataCopy(yGm_[m0 * totalN_ + n0], outLocal, curM * curN);
                outQueueY_.FreeTensor(outLocal);
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECIN, 1> biasQueue_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalM_ {0};
    uint32_t totalN_ {0};
    uint32_t totalK_ {0};
    uint32_t blockM_ {128};
    uint32_t blockN_ {128};
    uint32_t blockK_ {128};
    float divisor_ {1.0f};
    float invDivisor_ {1.0f};
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                                                   GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulDivideGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.totalM, tiling_data.totalN, tiling_data.totalK,
            tiling_data.blockM, tiling_data.blockN, tiling_data.blockK,
            tiling_data.divisor);
    op.Process();
}
