host_operator_src="""
#include "max_pooling2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model: one tile of input and output channels – conservative.
// Each pool operation needs to hold a whole input slice (one channel) plus output.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
  MaxPooling2dCustomTilingData tiling;
  const gert::Shape &input_shape = context->GetInputShape(0)->GetOriginShape();
  uint32_t batch = input_shape.GetDim(0);
  uint32_t channels = input_shape.GetDim(1);
  uint32_t inH = input_shape.GetDim(2);
  uint32_t inW = input_shape.GetDim(3);

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int32_t kernel = attrs->GetAttrPointer<int32_t>(0)[0];
  int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
  int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];
  int32_t dilation = attrs->GetAttrPointer<int32_t>(3)[0];

  int32_t outH = (inH + 2 * padding - dilation * (kernel - 1) - 1) / stride + 1;
  int32_t outW = (inW + 2 * padding - dilation * (kernel - 1) - 1) / stride + 1;

  uint32_t totalOutElems = batch * channels * outH * outW;

  uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * (kernel * kernel + 1));
  maxTile = AlignDown(maxTile, ALIGN_ELEMS);
  if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
  if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

  uint32_t tileLen = (totalOutElems < maxTile) ? totalOutElems : maxTile;

  uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutElems, tileLen);
  uint32_t blockDim = numTiles;
  if (blockDim < 8u) blockDim = 8u;
  if (blockDim > 24u) blockDim = 24u;

  context->SetBlockDim(blockDim);

  tiling.set_totalLength(batch * channels * inH * inW);
  tiling.set_tileLength(tileLen);
  tiling.set_blockDim(blockDim);
  tiling.set_batch(batch);
  tiling.set_channels(channels);
  tiling.set_inH(inH);
  tiling.set_inW(inW);
  tiling.set_kernel(kernel);
  tiling.set_stride(stride);
  tiling.set_padding(padding);
  tiling.set_outH(outH);
  tiling.set_outW(outW);

  tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                      context->GetRawTilingData()->GetCapacity());
  context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

  size_t *currentWorkspace = context->GetWorkspaceSizes(1);
  currentWorkspace[0] = 0;
  return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
  const gert::Shape *x_shape = context->GetInputShape(0);
  gert::Shape *y_shape = context->GetOutputShape(0);

  const gert::RuntimeAttrs *attrs = context->GetAttrs();
  int32_t kernel = attrs->GetAttrPointer<int32_t>(0)[0];
  int32_t stride = attrs->GetAttrPointer<int32_t>(1)[0];
  int32_t padding = attrs->GetAttrPointer<int32_t>(2)[0];
  int32_t dilation = attrs->GetAttrPointer<int32_t>(3)[0];

  uint32_t inH = x_shape->GetDim(2);
  uint32_t inW = x_shape->GetDim(3);
  int32_t outH = (inH + 2 * padding - dilation * (kernel - 1) - 1) / stride + 1;
  int32_t outW = (inW + 2 * padding - dilation * (kernel - 1) - 1) / stride + 1;

  y_shape->SetDim(0, x_shape->GetDim(0));
  y_shape->SetDim(1, x_shape->GetDim(1));
  y_shape->SetDim(2, outH);
  y_shape->SetDim(3, outW);
  return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
  context->SetOutputDataType(0, context->GetInputDataType(0));
  return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

class KernelMaxPooling2dCustom {
public:
  __aicore__ inline KernelMaxPooling2dCustom() {}

  __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                              uint32_t batch, uint32_t channels,
                              uint32_t inH, uint32_t inW,
                              uint32_t kernel, uint32_t stride,
                              uint32_t padding,
                              uint32_t outH, uint32_t outW) {
    batch_ = batch;
    channels_ = channels;
    inH_ = inH;
    inW_ = inW;
    kernel_ = kernel;
    stride_ = stride;
    padding_ = padding;
    outH_ = outH;
    outW_ = outW;
    totalOutElems_ = batch * channels * outH * outW;

    xGm_.SetGlobalBuffer((__gm__ float*)x, batch * channels * inH * inW);
    yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutElems_);
  }

  __aicore__ inline void Process() {
    if (totalOutElems_ == 0) return;

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    // Simple block-stride loop over all output elements
    for (uint32_t outIdx = blockIdx; outIdx < totalOutElems_; outIdx += blockNum) {
      uint32_t tmp = outIdx;
      uint32_t ow = tmp % outW_; tmp /= outW_;
      uint32_t oh = tmp % outH_; tmp /= outH_;
      uint32_t c = tmp % channels_; tmp /= channels_;
      uint32_t n = tmp;

      int32_t h_in_start = static_cast<int32_t>(oh * stride_) - static_cast<int32_t>(padding_);
      int32_t w_in_start = static_cast<int32_t>(ow * stride_) - static_cast<int32_t>(padding_);

      float maxVal = -3.4028235e+38f;

      for (uint32_t kh = 0; kh < kernel_; ++kh) {
        int32_t h_in = h_in_start + static_cast<int32_t>(kh);
        for (uint32_t kw = 0; kw < kernel_; ++kw) {
          int32_t w_in = w_in_start + static_cast<int32_t>(kw);
          float val;
          if (h_in >= 0 && h_in < static_cast<int32_t>(inH_) && w_in >= 0 && w_in < static_cast<int32_t>(inW_)) {
            uint32_t inputOffset = ((n * channels_ + c) * inH_ + h_in) * inW_ + w_in;
            val = xGm_.GetValue(inputOffset);
          } else {
            val = 0.0f;
          }
          if (val > maxVal) maxVal = val;
        }
      }
      yGm_.SetValue(outIdx, maxVal);
    }
  }

private:
  AscendC::GlobalTensor<float> xGm_;
  AscendC::GlobalTensor<float> yGm_;

  uint32_t batch_;
  uint32_t channels_;
  uint32_t inH_;
  uint32_t inW_;
  uint32_t kernel_;
  uint32_t stride_;
  uint32_t padding_;
  uint32_t outH_;
  uint32_t outW_;
  uint32_t totalOutElems_;
};

extern "C" __global__ __aicore__ void max_pooling2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
  GET_TILING_DATA(tiling_data, tiling);
  KernelMaxPooling2dCustom op;
  op.Init(x, y,
          tiling_data.batch, tiling_data.channels,
          tiling_data.inH, tiling_data.inW,
          tiling_data.kernel, tiling_data.stride,
          tiling_data.padding,
          tiling_data.outH, tiling_data.outW);
  op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor max_pooling2d_impl_npu(const at::Tensor& self,
                                   int64_t kernel_size,
                                   int64_t stride,
                                   int64_t padding,
                                   int64_t dilation = 1) {
    int64_t N = self.size(0);
    int64_t C = self.size(1);
    int64_t H = self.size(2);
    int64_t W = self.size(3);
    int64_t outH = (H + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    int64_t outW = (W + 2 * padding - dilation * (kernel_size - 1) - 1) / stride + 1;
    at::Tensor result = at::empty({N, C, outH, outW}, self.options());
    EXEC_NPU_CMD(aclnnMaxPooling2dCustom, self, kernel_size, stride, padding, dilation, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling2d_custom", &max_pooling2d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling2d_custom", &max_pooling2d_impl_npu, "Max Pooling 2D (MaxPooling2dCustom on NPU)");
}

import torch

import custom_ops_lib

_FN = 'max_pooling2d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size, stride, padding):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
