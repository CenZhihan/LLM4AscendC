    return (a + b - 1u) / b;

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Conv2dDivideLeakyReluCustomTilingData tiling;

    // Input shapes: x: NCHW, weight: CO CI KH KW
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    uint32_t N = x_shape->GetOriginShape().GetDim(0);
    uint32_t C = x_shape->GetOriginShape().GetDim(1);
    uint32_t H = x_shape->GetOriginShape().GetDim(2);
    uint32_t W = x_shape->GetOriginShape().GetDim(3);

    uint32_t CO = w_shape->GetOriginShape().GetDim(0);
    uint32_t CI = w_shape->GetOriginShape().GetDim(1); // should equal C
    uint32_t KH = w_shape->GetOriginShape().GetDim(2);
    uint32_t KW = w_shape->GetOriginShape().GetDim(3);

    (void)CI; // unused but required

    uint32_t HOut = H - KH + 1;
    uint32_t WOut = W - KW + 1;
    uint32_t totalOutputElems = N * CO * HOut * WOut;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *divisor_ptr = attrs->GetAttrPointer<float>(0);
    float divisor = *divisor_ptr;

    // Determine tile length: we process a contiguous chunk of output elements
    // Each output element requires: 1 float output, and we need input window floats for accumulation.
    // Simplified: treat each output element as a unit. UB size 192KB => 49152 floats.
    // We'll process tileLength output elements per iteration.
    uint32_t tileLen = 4096; // conservative
    if (totalOutputElems < tileLen) tileLen = totalOutputElems;

    // Compute blockDim: number of tiles enough to cover total output, clamped to reasonable range
    uint32_t numTiles = CeilDiv(totalOutputElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_CO(CO);
    tiling.set_KH(KH);
    tiling.set_KW(KW);
    tiling.set_HOut(HOut);
    tiling.set_WOut(WOut);
    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_divisor(divisor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

#include "kernel_operator.h"

class KernelConv2dDivideLeakyReluCustom {
public:
    __aicore__ inline KernelConv2dDivideLeakyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                GM_ADDR y, const Conv2dDivideLeakyReluCustomTilingData &tiling) {
        x_gm_.SetGlobalBuffer((__gm__ float*)x, 0);
        w_gm_.SetGlobalBuffer((__gm__ float*)weight, 0);
        bias_gm_ = (bias != nullptr) ? (__gm__ float*)bias : nullptr;
        y_gm_.SetGlobalBuffer((__gm__ float*)y, 0);

        N_ = tiling.N;
        C_ = tiling.C;
        H_ = tiling.H;
        W_ = tiling.W;
        CO_ = tiling.CO;
        KH_ = tiling.KH;
        KW_ = tiling.KW;
        HOut_ = tiling.HOut;
        WOut_ = tiling.WOut;
        totalOutputElems_ = tiling.totalOutputElems;
        tileLen_ = tiling.tileLength;
        divisor_ = tiling.divisor;
        negativeSlope_ = 0.01f;

        // Local buffer for output tile (we compute directly into y_gm to avoid double buffering)
        // Since convolution is compute-bound, we use a simple approach: each block processes its assigned output elements directly.
        // We allocate local tensor for loading input patches? For simplicity, we compute in-place using scalar loops.
        // This kernel is meant for correctness, not performance.
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint32_t blockNum = AscendC::GetBlockNum();

        uint32_t start = blockIdx * tileLen_;
        uint32_t end = (start + tileLen_ > totalOutputElems_) ? totalOutputElems_ : start + tileLen_;

        for (uint32_t idx = start; idx < end; ++idx) {
            // idx corresponds to a flat index in the output tensor: n, co, h, w
            uint32_t tmp = idx;
            uint32_t w_out = tmp % WOut_; tmp /= WOut_;
            uint32_t h_out = tmp % HOut_; tmp /= HOut_;
            uint32_t co = tmp % CO_; tmp /= CO_;
            uint32_t n = tmp;

            float sum = 0.0f;
            for (uint32_t kh = 0; kh < KH_; ++kh) {
                uint32_t h_in = h_out + kh;
                for (uint32_t kw = 0; kw < KW_; ++kw) {
                    uint32_t w_in = w_out + kw;
                    for (uint32_t ci = 0; ci < C_; ++ci) {
                        // x offset: n*C*H*W + ci*H*W + h_in*W + w_in
                        uint32_t x_offset = n * C_ * H_ * W_ + ci * H_ * W_ + h_in * W_ + w_in;
                        // weight offset: co*C*KH*KW + ci*KH*KW + kh*KW + kw
                        uint32_t w_offset = co * C_ * KH_ * KW_ + ci * KH_ * KW_ + kh * KW_ + kw;
                        sum += x_gm_[x_offset] * w_gm_[w_offset];
                    }
                }
            }

            // Add bias if present
            if (bias_gm_ != nullptr) {
                sum += bias_gm_[co];
            }

            // Divide
            sum = sum / divisor_;

            // Leaky ReLU
            sum = (sum > 0.0f) ? sum : negativeSlope_ * sum;

            // Write output
            uint32_t y_offset = n * CO_ * HOut_ * WOut_ + co * HOut_ * WOut_ + h_out * WOut_ + w_out;
            y_gm_[y_offset] = sum;
        }
    }

private:
    AscendC::GlobalTensor<float> x_gm_;
    AscendC::GlobalTensor<float> w_gm_;
    __gm__ float *bias_gm_;
    AscendC::GlobalTensor<float> y_gm_;

    uint32_t N_, C_, H_, W_, CO_, KH_, KW_, HOut_, WOut_;
    uint32_t totalOutputElems_, tileLen_;
    float divisor_, negativeSlope_;
};

extern "C" __global__ __aicore__ void conv2d_divide_leaky_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                       GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dDivideLeakyReluCustom op;
    op.Init(x, weight, bias, y, tiling_data);
    op.Process();
}
