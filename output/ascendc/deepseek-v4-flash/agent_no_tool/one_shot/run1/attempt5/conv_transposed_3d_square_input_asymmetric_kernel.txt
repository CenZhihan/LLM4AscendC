#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C_out, uint32_t D_out, uint32_t H_out, uint32_t W_out,
                                uint32_t C_in, uint32_t D_in, uint32_t H_in, uint32_t W_in,
                                uint32_t kD, uint32_t kH, uint32_t kW)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N; C_out_ = C_out; D_out_ = D_out; H_out_ = H_out; W_out_ = W_out;
        C_in_ = C_in; D_in_ = D_in; H_in_ = H_in; W_in_ = W_in;
        kD_ = kD; kH_ = kH; kW_ = kW;

        // Set global pointers (ND format, contiguous)
        xGm_.SetGlobalBuffer((__gm__ float*)x, N * C_in * D_in * H_in * W_in);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, C_out * C_in * kD * kH * kW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Local tensors: weight tile for one output channel
        // weight tile size = C_in * kD * kH * kW floats
        uint32_t weightTileSize = C_in_ * kD_ * kH_ * kW_;
        if (weightTileSize > 0) {
            pipe_.InitBuffer(weightLocal_, 1, weightTileSize * sizeof(float));
        }
        // No input double buffering – directly read from global
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride loop over output elements
        uint32_t start = blockIdx * tileLen_;
        if (start >= totalLen_) return;

        uint32_t end = start + tileLen_;
        if (end > totalLen_) end = totalLen_;

        // For each output element in this tile
        for (uint32_t outIdx = start; outIdx < end; ++outIdx) {
            // Map linear index to (b, c_out, d, h, w)
            uint32_t idx = outIdx;
            uint32_t w_out = idx % W_out_; idx /= W_out_;
            uint32_t h_out = idx % H_out_; idx /= H_out_;
            uint32_t d_out = idx % D_out_; idx /= D_out_;
            uint32_t c_out = idx % C_out_; idx /= C_out_;
            uint32_t b = idx;

            // Load weight for this c_out
            // weight layout: [C_out][C_in][kD][kH][kW]
            AscendC::LocalTensor<float> wLocal = weightLocal_.Get<float>();
            uint32_t weightOffset = c_out * (C_in_ * kD_ * kH_ * kW_);
            AscendC::DataCopy(wLocal, wGm_[weightOffset], C_in_ * kD_ * kH_ * kW_);

            float sum = 0.0f;
            // Loop over c_in and kernel positions
            for (uint32_t c_in = 0; c_in < C_in_; ++c_in) {
                for (uint32_t kd = 0; kd < kD_; ++kd) {
                    // Compute input index: input - kernel + 1 for transposed convolution
                    // For transposed with stride1, padding0, output pixel (d_out) receives contributions from input (d_out - kd)
                    if (d_out < kd || d_out >= D_in_ + kd) continue;  // out of bounds
                    uint32_t d_in = d_out - kd;
                    if (d_in >= D_in_) continue;
                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        if (h_out < kh || h_out >= H_in_ + kh) continue;
                        uint32_t h_in = h_out - kh;
                        if (h_in >= H_in_) continue;
                        for (uint32_t kw = 0; kw < kW_; ++kw) {
                            if (w_out < kw || w_out >= W_in_ + kw) continue;
                            uint32_t w_in = w_out - kw;
                            if (w_in >= W_in_) continue;

                            // Input index: [b][c_in][d_in][h_in][w_in]
                            uint32_t xIdx = ((b * C_in_ + c_in) * D_in_ + d_in) * H_in_ + h_in;
                            xIdx = xIdx * W_in_ + w_in;
                            float xVal = xGm_.GetValue(xIdx);

                            // Weight index: [c_out][c_in][kd][kh][kw] stored in local
                            uint32_t wIdx = (c_in * kD_ + kd) * kH_ + kh;
                            wIdx = wIdx * kW_ + kw;
                            float wVal = wLocal.GetValue(wIdx);

                            sum += xVal * wVal;
                        }
                    }
                }
            }
            yGm_.SetValue(outIdx, sum);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> weightLocal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_ {0}, C_out_ {0}, D_out_ {0}, H_out_ {0}, W_out_ {0};
    uint32_t C_in_ {0}, D_in_ {0}, H_in_ {0}, W_in_ {0};
    uint32_t kD_ {0}, kH_ {0}, kW_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.C_out, tiling_data.D_out, tiling_data.H_out, tiling_data.W_out,
            tiling_data.C_in, tiling_data.D_in, tiling_data.H_in, tiling_data.W_in,
            tiling_data.kD, tiling_data.kH, tiling_data.kW);
    op.Process();
}
