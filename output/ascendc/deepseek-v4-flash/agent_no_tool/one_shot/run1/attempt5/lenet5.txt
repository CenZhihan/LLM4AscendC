kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelLenet5Custom {
public:
    __aicore__ inline KernelLenet5Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR conv1_w, GM_ADDR conv1_b,
                                GM_ADDR conv2_w, GM_ADDR conv2_b,
                                GM_ADDR fc1_w, GM_ADDR fc1_b,
                                GM_ADDR fc2_w, GM_ADDR fc2_b,
                                GM_ADDR fc3_w, GM_ADDR fc3_b,
                                GM_ADDR y,
                                uint32_t batch, uint32_t numClasses,
                                uint32_t conv1OC, uint32_t conv1IC, uint32_t conv1KH, uint32_t conv1KW,
                                uint32_t conv2OC, uint32_t conv2IC, uint32_t conv2KH, uint32_t conv2KW,
                                uint32_t fc1In, uint32_t fc1Out,
                                uint32_t fc2In, uint32_t fc2Out,
                                uint32_t fc3In, uint32_t fc3Out)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * conv1IC * 32 * 32);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * numClasses);
        totalLen_ = batch * conv1IC * 32 * 32;
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();
        uint32_t lenPerBlock = totalLen_ / blockNum;
        uint32_t remain = totalLen_ % blockNum;
        uint32_t offset = blockIdx * lenPerBlock + (blockIdx < remain ? blockIdx : remain);
        uint32_t curLen = lenPerBlock + (blockIdx < remain ? 1 : 0);

        if (curLen == 0) return;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, curLen * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, curLen * sizeof(float));

        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offset], curLen);
        inQueueX_.EnQue(xLocal);

        AscendC::LocalTensor<float> xDeque = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Copy(yLocal, xDeque, curLen);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xDeque);

        AscendC::LocalTensor<float> yDeque = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], yDeque, curLen);
        outQueueY_.FreeTensor(yDeque);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLen_ {0};
};

extern "C" __global__ __aicore__ void lenet5_custom(GM_ADDR x, GM_ADDR conv1_w, GM_ADDR conv1_b,
                                                     GM_ADDR conv2_w, GM_ADDR conv2_b,
                                                     GM_ADDR fc1_w, GM_ADDR fc1_b,
                                                     GM_ADDR fc2_w, GM_ADDR fc2_b,
                                                     GM_ADDR fc3_w, GM_ADDR fc3_b,
                                                     GM_ADDR y,
                                                     GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLenet5Custom op;
    op.Init(x, conv1_w, conv1_b, conv2_w, conv2_b,
            fc1_w, fc1_b, fc2_w, fc2_b, fc3_w, fc3_b, y,
            tiling_data.batch, tiling_data.numClasses,
            tiling_data.conv1OC, tiling_data.conv1IC, tiling_data.conv1KH, tiling_data.conv1KW,
            tiling_data.conv2OC, tiling_data.conv2IC, tiling_data.conv2KH, tiling_data.conv2KW,
            tiling_data.fc1In, tiling_data.fc1Out,
            tiling_data.fc2In, tiling_data.fc2Out,
            tiling_data.fc3In, tiling_data.fc3Out);
    op.Process();
}

    const at::Tensor& conv1_w, const at::Tensor& conv1_b,
    const at::Tensor& conv2_w, const at::Tensor& conv2_b,
    const at::Tensor& fc1_w, const at::Tensor& fc1_b,

import torch
import torch_npu
import custom_ops_lib

_FN = 'lenet5_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        self.num_classes = init_inputs[0] if init_inputs else 20

    def forward(self, *inputs):
        # inputs: x, conv1_w, conv1_b, conv2_w, conv2_b, fc1_w, fc1_b, fc2_w, fc2_b, fc3_w, fc3_b
        x = inputs[0]
        conv1_w, conv1_b = inputs[1], inputs[2]
        conv2_w, conv2_b = inputs[3], inputs[4]
        fc1_w, fc1_b = inputs[5], inputs[6]
        fc2_w, fc2_b = inputs[7], inputs[8]
        fc3_w, fc3_b = inputs[9], inputs[10]
        fn = getattr(custom_ops_lib, _FN)
        result = fn(x, conv1_w, conv1_b, conv2_w, conv2_b,
                    fc1_w, fc1_b, fc2_w, fc2_b, fc3_w, fc3_b,
                    self.num_classes)
