host_operator_src="""

#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    auto x_shape = context->GetInputShape(0)->GetOriginShape();
    auto w_shape = context->GetInputShape(1)->GetOriginShape();
    uint32_t N  = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t C  = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t H  = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t W  = static_cast<uint32_t>(x_shape.GetDim(3));
    uint32_t Cout = static_cast<uint32_t>(w_shape.GetDim(0));
    uint32_t Cin  = static_cast<uint32_t>(w_shape.GetDim(1));

    uint32_t M = N * H * W;
    uint32_t totalLength = M * Cout;

    // TileCout: number of output channels processed per iteration
    // weight size: Cout * Cin * 4 bytes
    // input size: Cin * 4
    // output size: tileCout * 4
    uint32_t weightBytes = Cout * Cin * sizeof(float);
    uint32_t inputBytes = Cin * sizeof(float);
    uint32_t outputBytes = Cout * sizeof(float); // max
    uint32_t tempBytes = (Cin + 1) * sizeof(float);
    uint32_t available = UB_BUDGET_BYTES > weightBytes ? UB_BUDGET_BYTES - weightBytes : 0;
    uint32_t tileCout = 1;
    if (available > 0) {
        uint32_t maxTileCout = (available - inputBytes - tempBytes) / sizeof(float);
        if (maxTileCout > Cout) maxTileCout = Cout;
        if (maxTileCout > 16) maxTileCout = 16; // conservative
        tileCout = maxTileCout > 0 ? maxTileCout : 1;
    }

    // Block dimension: assign contiguous spatial positions
    uint32_t blockDim = 8; // default
    if (M > 0) {
        uint32_t numTiles = CeilDiv(M, 1); // each block handles one or more positions
        blockDim = numTiles;
        if (blockDim < 8) blockDim = 8;
        if (blockDim > 24) blockDim = 24;
    }

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_M(M);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_tileCout(tileCout);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    y_shape->SetDim(1, static_cast<int64_t>(w_shape->GetDim(0)));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class ConvPointwise2dCustom : public OpDef {
public:
    explicit ConvPointwise2dCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("weight")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("stride")   .AttrType(OPTIONAL).ListInt({1, 1});
        this->Attr("padding")  .AttrType(OPTIONAL).ListInt({0, 0});
        this->Attr("dilation") .AttrType(OPTIONAL).ListInt({1, 1});
        this->Attr("groups")   .AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(ConvPointwise2dCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t M, uint32_t Cin,
                                uint32_t Cout, uint32_t tileCout,
                                uint32_t blockDim)
    {
        M_ = M;
        Cin_ = Cin;
        Cout_ = Cout;
        tileCout_ = tileCout;
        totalLength_ = totalLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M_ * Cin_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, Cout_ * Cin_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M_ * Cout_);

        pipe_.InitBuffer(weightBuf_, 1, Cout_ * Cin_ * sizeof(float));
        pipe_.InitBuffer(inBuf_, 1, Cin_ * sizeof(float));
        pipe_.InitBuffer(outBuf_, 1, tileCout_ * sizeof(float));
        pipe_.InitBuffer(prodBuf_, 1, Cin_ * sizeof(float));
        pipe_.InitBuffer(sumBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0) return;

        // Load weight once
        AscendC::LocalTensor<float> weightLocal = weightBuf_.Get<float>();
        AscendC::DataCopy(weightLocal, weightGm_, Cout_ * Cin_);

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t spatialPerBlock = (M_ + blockNum - 1) / blockNum;
        uint32_t startS = blockIdx * spatialPerBlock;
        uint32_t endS = startS + spatialPerBlock;
        if (endS > M_) endS = M_;
        if (startS >= endS) return;

        for (uint32_t s = startS; s < endS; ++s) {
            // Load input for this spatial position
            AscendC::LocalTensor<float> xLocal = inBuf_.Get<float>();
            AscendC::DataCopy(xLocal, xGm_[s * Cin_], Cin_);

            // Process output channels in tiles
            AscendC::LocalTensor<float> prodLocal = prodBuf_.Get<float>();
            AscendC::LocalTensor<float> sumLocal = sumBuf_.Get<float>();

            for (uint32_t ocBase = 0; ocBase < Cout_; ocBase += tileCout_) {
                uint32_t ocCount = (ocBase + tileCout_ > Cout_) ? (Cout_ - ocBase) : tileCout_;

                AscendC::LocalTensor<float> yLocal = outBuf_.Get<float>();
                AscendC::SetZero(yLocal, ocCount);

                for (uint32_t oc = 0; oc < ocCount; ++oc) {
                    uint32_t actualOc = ocBase + oc;
                    // Get row of weight for this output channel
                    AscendC::LocalTensor<float> wRow = weightLocal[actualOc * Cin_];
                    // Elementwise multiply input and weight row
                    AscendC::Mul(prodLocal, xLocal, wRow, Cin_);
                    // Reduce sum
                    AscendC::ReduceSum(sumLocal, prodLocal, Cin_);
                    yLocal[oc] = sumLocal[0];
                }

                // Write output tile
                AscendC::DataCopy(yGm_[s * Cout_ + ocBase], yLocal, ocCount);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TBufPosition::VECCALC> weightBuf_;
    AscendC::TBuf<AscendC::TBufPosition::VECIN> inBuf_;
    AscendC::TBuf<AscendC::TBufPosition::VECOUT> outBuf_;
    AscendC::TBuf<AscendC::TBufPosition::VECCALC> prodBuf_;
    AscendC::TBuf<AscendC::TBufPosition::VECCALC> sumBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLength_ {0};
    uint32_t M_ {0};
    uint32_t Cin_ {0};
    uint32_t Cout_ {0};
    uint32_t tileCout_ {16};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.totalLength, tiling_data.M, tiling_data.Cin, tiling_data.Cout,
            tiling_data.tileCout, tiling_data.blockDim);
    op.Process();
}
