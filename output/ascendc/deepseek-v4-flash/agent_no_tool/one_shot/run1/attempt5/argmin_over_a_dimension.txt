#include "argmin_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cstdint>
#include <algorithm>

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ArgminOverADimensionCustomTilingData tiling;
    const auto& inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t ndim = inShape.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dimAttr = *attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimAttr < 0) ? dimAttr + (int32_t)ndim : dimAttr;
    if (dim < 0 || dim >= (int32_t)ndim) {
        dim = (int32_t)ndim - 1;
    }

    // Compute total input length
    uint32_t totalInputLength = 1;
    for (uint32_t i = 0; i < ndim; i++) totalInputLength *= inShape.GetDim(i);

    // Compute output length = product of all dims except the reduction one
    uint32_t outputLength = 1;
    for (uint32_t i = 0; i < ndim; i++) if (i != (uint32_t)dim) outputLength *= inShape.GetDim(i);

    // Reduction dimension size and strides
    uint32_t reduceSize = inShape.GetDim(dim);
    uint32_t innerStride = 1;
    for (uint32_t i = (uint32_t)dim + 1; i < ndim; i++) innerStride *= inShape.GetDim(i);

    uint32_t outerSize = 1;
    for (uint32_t i = 0; i < (uint32_t)dim; i++) outerSize *= inShape.GetDim(i);

    uint32_t lastDimSize = (ndim > 0) ? inShape.GetDim(ndim - 1) : 1;

    // Block dimension based on number of output elements
    uint32_t blockDim = std::min(24u, CeilDiv(outputLength, 128u));
    if (blockDim < 1) blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalInputLength(totalInputLength);
    tiling.set_outputLength(outputLength);
    tiling.set_dimToReduce((uint32_t)dim);
    tiling.set_reduceSize(reduceSize);
    tiling.set_innerStride(innerStride);
    tiling.set_outerSize(outerSize);
    tiling.set_lastDimSize(lastDimSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const auto& inShape = context->GetInputShape(0)->GetOriginShape();
    uint32_t ndim = inShape.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dimAttr = attrs->GetAttrPointer<int32_t>(0) ? *attrs->GetAttrPointer<int32_t>(0) : -1;
    int32_t dim = (dimAttr < 0) ? dimAttr + (int32_t)ndim : dimAttr;
    if (dim < 0 || dim >= (int32_t)ndim) dim = (int32_t)ndim - 1;

    // Construct output shape: remove the reduction dimension
    uint32_t outNdim = ndim - 1;
    gert::Shape outShape;
    outShape.SetDimNum(outNdim);
    uint32_t outIdx = 0;
    for (uint32_t i = 0; i < ndim; i++) {
        if (i != (uint32_t)dim) {
            outShape.SetDim(outIdx, inShape.GetDim(i));
            outIdx++;
        }
    }
    *context->GetOutputShape(0) = outShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"
#include <cfloat>

class KernelArgminOverADimensionCustom {
public:
    __aicore__ inline KernelArgminOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalInputLength, uint32_t outputLength,
                                uint32_t dimToReduce, uint32_t reduceSize,
                                uint32_t innerStride, uint32_t outerSize,
                                uint32_t lastDimSize) {
        xGm_ = (__gm__ float*)x;
        yGm_ = (__gm__ int64_t*)y;
        totalInputLength_ = totalInputLength;
        outputLength_ = outputLength;
        dimToReduce_ = dimToReduce;
        reduceSize_ = reduceSize;
        innerStride_ = innerStride;
        outerSize_ = outerSize;
        lastDimSize_ = lastDimSize;
    }

    __aicore__ inline void Process() {
        if (outputLength_ == 0 || reduceSize_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        // Grid-stride loop over output elements
        for (uint32_t outputIdx = blockIdx; outputIdx < outputLength_; outputIdx += blockNum) {
            // Recover outer and inner indices
            uint32_t innerOffset = outputIdx % innerStride_;
            uint32_t outerIdx = outputIdx / innerStride_;

            // Base address for the first element of the reduction slice
            uint32_t base = outerIdx * (reduceSize_ * innerStride_) + innerOffset;

            // Find min value and its index along the reduction dimension
            float minVal = FLT_MAX;
            int64_t minIdx = 0;
            for (uint32_t k = 0; k < reduceSize_; k++) {
                float val = xGm_[base + k * innerStride_];
                if (val < minVal) {
                    minVal = val;
                    minIdx = static_cast<int64_t>(k);
                }
            }
            yGm_[outputIdx] = minIdx;
        }
    }

private:
    __gm__ float* xGm_;
    __gm__ int64_t* yGm_;
    uint32_t totalInputLength_ {0};
    uint32_t outputLength_ {0};
    uint32_t dimToReduce_ {0};
    uint32_t reduceSize_ {0};
    uint32_t innerStride_ {1};
    uint32_t outerSize_ {1};
    uint32_t lastDimSize_ {1};
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgminOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalInputLength, tiling_data.outputLength,
            tiling_data.dimToReduce, tiling_data.reduceSize,
            tiling_data.innerStride, tiling_data.outerSize,
            tiling_data.lastDimSize);
    op.Process();
}
