#include "conv2d_min_tanh_tanh_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Conv2dMinTanhTanhCustomTilingData tiling;
    const gert::StorageShape *xShape = context->GetInputShape(0);
    const gert::StorageShape *wShape = context->GetInputShape(1);
    uint32_t N = xShape->GetOriginShape().GetDim(0);
    uint32_t C = xShape->GetOriginShape().GetDim(1);
    uint32_t H = xShape->GetOriginShape().GetDim(2);
    uint32_t W = xShape->GetOriginShape().GetDim(3);
    uint32_t OC = wShape->GetOriginShape().GetDim(0);
    uint32_t kH = wShape->GetOriginShape().GetDim(2);
    uint32_t kW = wShape->GetOriginShape().GetDim(3);
    uint32_t Ho = H - kH + 1;
    uint32_t Wo = W - kW + 1;
    uint32_t totalPixels = N * Ho * Wo;

    // Per-pixel scratch for convolution: patch (C*kH*kW) and weight chunk (outC*inC*kH*kW) and bias (outC)
    // We'll process output pixel by pixel, so we need space for one output pixel's all channel sums.
    uint32_t pixelCost = C * kH * kW + OC * C * kH * kW + OC; // approximate floats per pixel
    uint32_t maxPixel = UB_BUDGET_BYTES / (pixelCost * sizeof(float));
    maxPixel = AlignDown(maxPixel, ALIGN_ELEMS);
    if (maxPixel < 1) maxPixel = 1;
    if (maxPixel > HARD_CAP_ELEMS) maxPixel = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tilePixels = (maxPixel < totalPixels) ? maxPixel : totalPixels;
    // Ensure at least one pixel per tile
    if (tilePixels == 0) tilePixels = 1;

    uint32_t numTiles = CeilDiv(totalPixels, tilePixels);
    uint32_t blockDim = numTiles;
    if (blockDim < 4) blockDim = 4;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_batch(N);
    tiling.set_inC(C);
    tiling.set_outC(OC);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_totalPixels(totalPixels);
    tiling.set_tilePixels(tilePixels);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelConv2dMinTanhTanhCustom {
public:
    __aicore__ inline KernelConv2dMinTanhTanhCustom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y,
        uint32_t batch, uint32_t inC, uint32_t outC, uint32_t H, uint32_t W,
        uint32_t kH, uint32_t kW, uint32_t Ho, uint32_t Wo,
        uint32_t totalPixels, uint32_t tilePixels, uint32_t blockDim
    ) {
        batch_ = batch;
        inC_ = inC;
        outC_ = outC;
        H_ = H;
        W_ = W;
        kH_ = kH;
        kW_ = kW;
        Ho_ = Ho;
        Wo_ = Wo;
        totalPixels_ = totalPixels;
        tilePixels_ = tilePixels;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_ * inC_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outC_ * inC_ * kH_ * kW_);
        biasGm_.SetGlobalBuffer((__gm__ float*)conv_bias, outC_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch_ * 1 * Ho_ * Wo_);

        // Allocate local buffers:
        // patch: inC*kH*kW floats
        // weightChunk: outC * inC * kH * kW floats (we will load a few channels at a time)
        // bias: outC floats
        // sum: outC floats (per pixel)
        uint32_t patchSize = inC_ * kH_ * kW_;
        uint32_t weightSize = outC_ * inC_ * kH_ * kW_;
        uint32_t biasSize = outC_;

        pipe_.InitBuffer(tmpBuf1_, patchSize * sizeof(float));
        pipe_.InitBuffer(tmpBuf2_, weightSize * sizeof(float));
        pipe_.InitBuffer(tmpBuf3_, biasSize * sizeof(float));
        pipe_.InitBuffer(tmpBuf4_, outC_ * sizeof(float)); // sum per output pixel
    }

    __aicore__ inline void Process() {
        uint32_t blockIdx = GetBlockIdx();
        uint32_t pixelStart = blockIdx * tilePixels_;
        if (pixelStart >= totalPixels_) return;
        uint32_t pixelEnd = pixelStart + tilePixels_;
        if (pixelEnd > totalPixels_) pixelEnd = totalPixels_;

        LocalTensor<float> patchBuf = tmpBuf1_.Get<float>();
        LocalTensor<float> weightBuf = tmpBuf2_.Get<float>();
        LocalTensor<float> biasBuf = tmpBuf3_.Get<float>();
        LocalTensor<float> sumBuf = tmpBuf4_.Get<float>();

        // Preload bias once
        DataCopy(biasBuf, biasGm_, outC_);

        for (uint32_t p = pixelStart; p < pixelEnd; p++) {
            // Convert pixel index to batch, Ho, Wo
            uint32_t batchIdx = p / (Ho_ * Wo_);
            uint32_t ho = (p % (Ho_ * Wo_)) / Wo_;
            uint32_t wo = p % Wo_;

            // Load input patch for this pixel (all input channels, kernel window)
            // Input layout: NCHW
            // For each input channel c, load (c*H*W + h_start*W + w_start) with size kH*kW
            for (uint32_t c = 0; c < inC_; c++) {
                uint32_t inputOffset = (batchIdx * inC_ + c) * H_ * W_ + ho * W_ + wo;
                // The patch spans kH*kW: need to load kH lines of kW each
                // Use a loop for simplicity; can optimize with multi-line DataCopy but fine for now
                for (uint32_t i = 0; i < kH_; i++) {
                    uint32_t rowOffset = inputOffset + i * W_;
                    DataCopy(patchBuf[c * kH_ * kW_ + i * kW_], xGm_[rowOffset], kW_);
                }
            }

            // Compute convolution: for each output channel oc, sum over inC*kH*kW
            // We load weight on the fly (all outC*inC*kH*kW)
            DataCopy(weightBuf, wGm_, outC_ * inC_ * kH_ * kW_);

            // Compute per output channel
            for (uint32_t oc = 0; oc < outC_; oc++) {
                float sum = 0.0f;
                // Weight layout: (oc, ic, kh, kw)
                uint32_t weightOffset = oc * inC_ * kH_ * kW_;
                for (uint32_t ic = 0; ic < inC_; ic++) {
                    for (uint32_t kh = 0; kh < kH_; kh++) {
                        for (uint32_t kw = 0; kw < kW_; kw++) {
                            sum += patchBuf[ic * kH_ * kW_ + kh * kW_ + kw] *
                                   weightBuf[weightOffset + ic * kH_ * kW_ + kh * kW_ + kw];
                        }
                    }
                }
                sum += biasBuf[oc];
                sumBuf[oc] = sum;
            }

            // Compute min across channels (oc) and apply tanh twice
            float minVal = sumBuf[0];
            for (uint32_t oc = 1; oc < outC_; oc++) {
                if (sumBuf[oc] < minVal) minVal = sumBuf[oc];
            }
            // tanh
            float tanh1 = Tanh(minVal);
            float tanh2 = Tanh(tanh1);
            // Write to output
            yGm_[batchIdx * 1 * Ho_ * Wo_ + ho * Wo_ + wo] = tanh2;
        }
    }

private:
    // Inline tanh approximation (could use built-in, but for completeness)
    __aicore__ inline float Tanh(float x) {
        // Use built-in intrinsic if available; otherwise manual approximation
        // AscendC provides Tanh via math intrinsic
        return AscendC::Tanh(x);
    }

private:
    TPipe pipe_;
    TBuf<TPosition::VECCALC> tmpBuf1_;  // patch
    TBuf<TPosition::VECCALC> tmpBuf2_;  // weight
    TBuf<TPosition::VECCALC> tmpBuf3_;  // bias
    TBuf<TPosition::VECCALC> tmpBuf4_;  // sum

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> biasGm_;
    GlobalTensor<float> yGm_;

    uint32_t batch_, inC_, outC_, H_, W_, kH_, kW_, Ho_, Wo_;
    uint32_t totalPixels_, tilePixels_, blockDim_;
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanhCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.batch, tiling_data.inC, tiling_data.outC,
            tiling_data.H, tiling_data.W, tiling_data.kH, tiling_data.kW,
            tiling_data.Ho, tiling_data.Wo,
            tiling_data.totalPixels, tiling_data.tilePixels, tiling_data.blockDim);
    op.Process();
}
