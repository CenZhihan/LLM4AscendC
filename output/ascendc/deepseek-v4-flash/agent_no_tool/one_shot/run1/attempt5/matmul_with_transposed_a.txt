#include "matmul_with_transposed_a_custom_tiling.h"
#include "register/op_def_registry.h"
#include "graph/shape.h"

namespace optiling {

static constexpr uint32_t ALIGN_16 = 16;

static inline uint32_t AlignUp(uint32_t x, uint32_t a) {
    return (x + a - 1) / a * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MatmulWithTransposedACustomTilingData tiling;

    const ge::Shape &a_shape = context->GetInputShape(0)->GetOriginShape();
    const ge::Shape &b_shape = context->GetInputShape(1)->GetOriginShape();

    uint32_t K = static_cast<uint32_t>(a_shape.GetDim(0));
    uint32_t M = static_cast<uint32_t>(a_shape.GetDim(1));
    uint32_t N = static_cast<uint32_t>(b_shape.GetDim(1));

    uint32_t tileM = (M > 128) ? 128 : AlignUp(M, 16);
    uint32_t tileN = (N > 128) ? 128 : AlignUp(N, 16);
    uint32_t tileK = (K > 64)  ? 64  : AlignUp(K, 16);

    tileM = AlignUp(tileM, 16);
    tileN = AlignUp(tileN, 16);
    tileK = AlignUp(tileK, 16);

    uint32_t numTilesN = CeilDiv(N, tileN);
    uint32_t blockDim = (numTilesN < 1) ? 1 : numTilesN;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const ge::Shape &a_shape = context->GetInputShape(0)->GetOriginShape();
    const ge::Shape &b_shape = context->GetInputShape(1)->GetOriginShape();

    uint32_t M = static_cast<uint32_t>(a_shape.GetDim(1));
    uint32_t N = static_cast<uint32_t>(b_shape.GetDim(1));

    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}

kernel_src='''
#include "kernel_operator.h"
#include "cmath"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t ALIGN_16 = 16;

class KernelMatmulWithTransposedA {
public:
    __aicore__ inline KernelMatmulWithTransposedA() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                uint32_t blockIdx)
    {
        M_ = M;
        K_ = K;
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;
        blockIdx_ = blockIdx;

        aGm_.SetGlobalBuffer((__gm__ float*)a, K * M);
        bGm_.SetGlobalBuffer((__gm__ float*)b, K * N);
        cGm_.SetGlobalBuffer((__gm__ float*)c, M * N);

        pipe_.InitBuffer(inBufA_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(inBufB_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(outBufC_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        uint32_t nStart = blockIdx_ * tileN_;
        if (nStart >= N_) return;
        uint32_t nEnd = (nStart + tileN_ > N_) ? N_ : nStart + tileN_;
        uint32_t actualTileN = nEnd - nStart;

        uint32_t mStart = 0;
        while (mStart < M_) {
            uint32_t mEnd = (mStart + tileM_ > M_) ? M_ : mStart + tileM_;
            uint32_t actualTileM = mEnd - mStart;

            LocalTensor<float> cTile = outBufC_.AllocTensor<float>();
            cTile.SetZero(actualTileM * actualTileN);

            for (uint32_t kStart = 0; kStart < K_; kStart += tileK_) {
                uint32_t kEnd = (kStart + tileK_ > K_) ? K_ : kStart + tileK_;
                uint32_t actualTileK = kEnd - kStart;

                LocalTensor<float> aTile = inBufA_.AllocTensor<float>();
                uint32_t aOffset = kStart * M_ + mStart;
                DataCopy(aTile, aGm_[aOffset], actualTileK * actualTileM);

                LocalTensor<float> bTile = inBufB_.AllocTensor<float>();
                uint32_t bOffset = kStart * N_ + nStart;
                DataCopy(bTile, bGm_[bOffset], actualTileK * actualTileN);

                Matmul<MatmulType<ROUND_TYPE, INPUT_TYPE::FP32>> matmul;
                matmul.SetTransposeA(true);
                matmul.SetTransposeB(false);
                matmul.Init(aTile, bTile, cTile, actualTileM, actualTileK, actualTileN);
                matmul.Process();

                inBufA_.FreeTensor(aTile);
                inBufB_.FreeTensor(bTile);
            }

            uint32_t cOffset = mStart * N_ + nStart;
            DataCopy(cGm_[cOffset], cTile, actualTileM * actualTileN);
            outBufC_.FreeTensor(cTile);

            mStart = mEnd;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outBufC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
    uint32_t tileK_ {0};
    uint32_t blockIdx_ {0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_a_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
    KernelMatmulWithTransposedA op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            blockIdx);
    op.Process();
}
