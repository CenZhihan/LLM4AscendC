#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cmath>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;

    const gert::StorageShape *xShape = context->GetInputShape(0);
    uint32_t totalLength = static_cast<uint32_t>(xShape->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t dimRaw = *attrs->GetAttrPointer<int64_t>(0);
    uint32_t ndim = xShape->GetOriginShape().GetDimNum();
    int64_t dim = (dimRaw < 0) ? dimRaw + ndim : dimRaw;
    if (dim < 0 || dim >= static_cast<int64_t>(ndim)) {
        dim = ndim - 1;
    }

    uint32_t segLen = xShape->GetOriginShape().GetDim(static_cast<uint32_t>(dim));
    uint32_t stride = 1;
    for (uint32_t d = static_cast<uint32_t>(dim) + 1; d < ndim; d++) {
        stride *= xShape->GetOriginShape().GetDim(d);
    }
    uint32_t outerSegCnt = totalLength / segLen;

    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    uint32_t tileLen = (segLen < maxTile) ? segLen : maxTile;

    uint32_t blockDim = 24;
    if (outerSegCnt < 8) blockDim = 8;
    if (outerSegCnt < blockDim) blockDim = outerSegCnt;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerSegCnt(outerSegCnt);
    tiling.set_segLen(segLen);
    tiling.set_stride(stride);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t outerSegCnt, uint32_t segLen,
                                uint32_t stride, uint32_t tileLen)
    {
        outerSegCnt_ = outerSegCnt;
        segLen_ = segLen;
        stride_ = stride;
        tileLen_ = tileLen;
        if (tileLen_ > segLen_) tileLen_ = segLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        maskGm_.SetGlobalBuffer((__gm__ uint8_t*)mask);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(uint8_t));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        for (uint32_t seg = blockIdx; seg < outerSegCnt_; seg += blockNum) {
            uint32_t offset = seg * segLen_ * stride_;  // correct offset

            uint32_t remain = segLen_;
            uint32_t cur = 0;
            float carry = 0.0f;
            while (remain > 0) {
                uint32_t len = (remain > tileLen_) ? tileLen_ : remain;

                LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                LocalTensor<uint8_t> maskLocal = inQueueMask_.AllocTensor<uint8_t>();
                DataCopy(xLocal, xGm_[offset + cur * stride_], len);
                DataCopy(maskLocal, maskGm_[offset + cur * stride_], len);

                LocalTensor<float> prodLocal = tmpQueue_.AllocTensor<float>();
                // Use Select: prod = (mask != 0) ? x : 0
                Select(prodLocal, xLocal, 0.0f, maskLocal, SEL_MODE::SEL_MODE_MASK, len);

                // Manual cumulative sum using scalar access
                for (uint32_t i = 1; i < len; i++) {
                    float prev = prodLocal.GetValue(i - 1);
                    float curr = prodLocal.GetValue(i);
                    prodLocal.SetValue(i, curr + prev);
                }
                if (cur > 0) {
                    for (uint32_t i = 0; i < len; i++) {
                        float val = prodLocal.GetValue(i);
                        prodLocal.SetValue(i, val + carry);
                    }
                }
                carry = prodLocal.GetValue(len - 1);

                DataCopy(yGm_[offset + cur * stride_], prodLocal, len);

                inQueueX_.FreeTensor(xLocal);
                inQueueMask_.FreeTensor(maskLocal);
                tmpQueue_.FreeTensor(prodLocal);

                cur += len;
                remain -= len;
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueX_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<uint8_t> maskGm_;
    GlobalTensor<float> yGm_;

    uint32_t outerSegCnt_ {0};
    uint32_t segLen_ {0};
    uint32_t stride_ {1};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.outerSegCnt, tiling_data.segLen,
            tiling_data.stride, tiling_data.tileLen);
    op.Process();
}
