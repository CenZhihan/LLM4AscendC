constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t TILE_M = 128;
constexpr uint32_t TILE_N = 128;
constexpr uint32_t TILE_K = 128;

class KernelMatmulSubtractMultiplyReluCustom {
public:
    __aicore__ inline KernelMatmulSubtractMultiplyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                float subValue, float mulValue)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        subValue_ = subValue;
        mulValue_ = mulValue;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N_ * K_);  // w stored as (N, K)
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M_ * N_);

        matmul_.SetOrgShape(M_, N_, K_);

        pipe_.InitBuffer(tmpQueue_, 1, TILE_M * TILE_N * sizeof(float));
        pipe_.InitBuffer(subMulReluQueue_, 1, TILE_M * TILE_N * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockIdx = GetBlockIdx();
        const uint32_t blockNum = GetBlockNum();

        // Each core processes a subset of rows (M dimension)
        uint32_t rowsPerBlock = (M_ + blockNum - 1) / blockNum;

        uint32_t rowEnd = (rowStart + rowsPerBlock > M_) ? M_ : rowStart + rowsPerBlock;
        uint32_t numRows = rowEnd - rowStart;

        if (numRows == 0) return;

        // Tiling over rows and columns
        for (uint32_t mOffset = rowStart; mOffset < rowEnd; mOffset += TILE_M) {
            uint32_t curM = TILE_M;
            if (mOffset + curM > rowEnd) curM = rowEnd - mOffset;

            for (uint32_t nOffset = 0; nOffset < N_; nOffset += TILE_N) {

                // Set sub-block for matmul

                Maxs(outLocal, mulTmp, 0.0f, curM * curN);

                // Free temporary tensors

                subMulReluQueue_.FreeTensor(tmpLocal);

                // Copy out to global

                outQueueY_.FreeTensor(outLocal);
            }
