static constexpr uint32_t UB_BUDGET_BYTES = 240 * 1024;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    using T = MatmulWithTransposedBothCustomTilingData;
    T tiling;

    const auto &shapeA = context->GetInputShape(0)->GetOriginShape();
    const auto &shapeB = context->GetInputShape(1)->GetOriginShape();
    uint32_t K_input = static_cast<uint32_t>(shapeA.GetDim(0));
    uint32_t M       = static_cast<uint32_t>(shapeA.GetDim(1));
    uint32_t N       = static_cast<uint32_t>(shapeB.GetDim(0));

    // Just pass dimensions; kernel uses AscendC MatMul which handles its own tiling
    uint32_t blockM = 128, blockN = 128, blockK = 128;
    uint32_t numBlocksM = CeilDiv(M, blockM);
    uint32_t numBlocksN = CeilDiv(N, blockN);
    uint32_t blockDim = numBlocksM * numBlocksN;
    if (blockDim > 24U) blockDim = 24U;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K_input);
    tiling.set_blockM(blockM);
    tiling.set_blockN(blockN);
    tiling.set_blockK(blockK);
    tiling.set_numBlocksM(numBlocksM);
    tiling.set_numBlocksN(numBlocksN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"
#include "matmul.h"

using namespace AscendC;

class KernelMatmulWithTransposedBoth {
public:
    __aicore__ inline KernelMatmulWithTransposedBoth() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t blockM, uint32_t blockN, uint32_t blockK,
                                uint32_t numBlocksM, uint32_t numBlocksN,
                                uint32_t blockDim)
    {
        A_gm_.SetGlobalBuffer((__gm__ float*)A, M * K);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, M * N);
        M_ = M; N_ = N; K_ = K;
    }

    __aicore__ inline void Process()
    {
        // Use AscendC MatMul with both inputs transposed
        MatMul<float> matMul;
        matMul.Init(&pipe_);
        // A is (K, M) => after transposeA becomes (M, K)
        matMul.SetTensorA(A_gm_, true, M_, K_);
        // B is (N, K) => after transposeB becomes (K, N)
        matMul.SetTensorB(B_gm_, true, N_, K_);
        matMul.SetTensorC(C_gm_, M_, N_);
        matMul.SetBias(nullptr);
        matMul.Process();
    }

private:
    AscendC::TPipe pipe_;
    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;
    uint32_t M_, N_, K_;
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedBoth op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.blockM, tiling_data.blockN, tiling_data.blockK,
            tiling_data.numBlocksM, tiling_data.numBlocksN,
            tiling_data.blockDim);
    op.Process();
}
