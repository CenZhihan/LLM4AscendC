kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMaxSubtractGeluCustom {
public:
    __aicore__ inline KernelGemmMaxSubtractGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batch, uint32_t inDim, uint32_t outDim,
                                int32_t maxDim, int32_t meanDim,
                                uint32_t tileLen)
    {
        batch_ = batch;
        inDim_ = inDim;
        outDim_ = outDim;
        maxDim_ = maxDim;
        meanDim_ = meanDim;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * inDim);
        wGm_.SetGlobalBuffer((__gm__ float*)w, outDim * inDim);
        bGm_.SetGlobalBuffer((__gm__ float*)b, outDim);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * 1);

        // Use pipe for linear result storage (local memory)
        pipe_.InitBuffer(linearQueue_, BUFFER_NUM, outDim_ * sizeof(float));
        // Temporary buffers for max, mean, etc.
        pipe_.InitBuffer(tmpQueue_, 1, outDim_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || inDim_ == 0 || outDim_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t samplesPerBlock = tileLen_;
        if (samplesPerBlock == 0) samplesPerBlock = 1;

        uint32_t start = blockIdx * samplesPerBlock;
        if (start >= batch_) return;
        uint32_t end = start + samplesPerBlock;
        if (end > batch_) end = batch_;

        for (uint32_t s = start; s < end; s++) {
            // 1. Compute linear output for sample s: y_lin[i] = dot(x[s,:], w[i,:]) + b[i]
            AscendC::LocalTensor<float> linearLocal = linearQueue_.AllocTensor<float>();
            for (uint32_t i = 0; i < outDim_; i++) {
                float sum = 0.0f;
                for (uint32_t j = 0; j < inDim_; j++) {
                    sum += xGm_[s * inDim_ + j] * wGm_[i * inDim_ + j];
                }
                sum += bGm_[i];
                linearLocal(i) = sum;
            }

            // 2. Compute max along maxDim (always 1 in practice)
            float maxVal = linearLocal(0);
            for (uint32_t i = 1; i < outDim_; i++) {
                if (linearLocal(i) > maxVal) maxVal = linearLocal(i);
            }

            // 3. Compute mean along meanDim (always 1)
            float sum = 0.0f;
            for (uint32_t i = 0; i < outDim_; i++) {
                sum += linearLocal(i);
            }
            float meanVal = sum / static_cast<float>(outDim_);

            // 4. Compute x_adj = maxVal - meanVal
            float adj = maxVal - meanVal;

            // 5. Apply GELU activation: GELU(x) = 0.5 * x * (1 + tanh(sqrt(2/pi) * (x + 0.044715 * x^3)))
            float x = adj;
            float x3 = x * x * x;
            float inner = 0.7978845608028654f * (x + 0.044715f * x3);  // sqrt(2/pi) ≈ 0.79788456
            float tanh_inner = (AscendC::Exp(2.0f * inner) - 1.0f) / (AscendC::Exp(2.0f * inner) + 1.0f); // tanh approx
            float gelu_out = 0.5f * x * (1.0f + tanh_inner);

            // Store result
            yGm_[s] = gelu_out;

            linearQueue_.FreeTensor(linearLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> linearQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_ {0};
    uint32_t inDim_ {0};
    uint32_t outDim_ {0};
    int32_t maxDim_ {1};
    int32_t meanDim_ {1};
    uint32_t tileLen_ {1};
};

extern "C" __global__ __aicore__ void gemm_max_subtract_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMaxSubtractGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.batch, tiling_data.inDim, tiling_data.outDim,
            tiling_data.maxDim, tiling_data.meanDim,
            tiling_data.tileLength);
    op.Process();
}
