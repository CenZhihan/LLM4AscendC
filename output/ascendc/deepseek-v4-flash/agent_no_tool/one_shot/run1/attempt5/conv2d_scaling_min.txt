#include "conv2d_scaling_min_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    Conv2dScalingMinCustomTilingData tiling;

    // GetInputShape returns const gert::StorageShape*; use GetOriginShape() to obtain const gert::Shape&
    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape &w_shape = context->GetInputShape(1)->GetOriginShape();
    const gert::StorageShape *bias_storage = context->GetInputShape(2); // optional, may be nullptr

    uint32_t N = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t C_in = static_cast<uint32_t>(x_shape.GetDim(1));
    uint32_t H = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(x_shape.GetDim(3));
    uint32_t C_out = static_cast<uint32_t>(w_shape.GetDim(0));
    uint32_t K_h = static_cast<uint32_t>(w_shape.GetDim(2));
    uint32_t K_w = static_cast<uint32_t>(w_shape.GetDim(3));
    uint32_t H_out = H - K_h + 1;
    uint32_t W_out = W - K_w + 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scalePtr = attrs->GetAttrPointer<float>(0);
    float scaleFactor = (scalePtr != nullptr) ? *scalePtr : 1.0f;

    bool hasBias = (bias_storage != nullptr);

    uint32_t totalSpatial = N * H_out * W_out;
    uint32_t blockDim = (totalSpatial + 255) / 256;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;
    context->SetBlockDim(blockDim);

    tiling.set_batchSize(N);
    tiling.set_outChannels(C_out);
    tiling.set_outH(H_out);
    tiling.set_outW(W_out);
    tiling.set_kernelH(K_h);
    tiling.set_kernelW(K_w);
    tiling.set_inChannels(C_in);
    tiling.set_inH(H);
    tiling.set_inW(W);
    tiling.set_scaleFactor(scaleFactor);
    tiling.set_hasBias(hasBias ? 1u : 0u);
    tiling.set_tileSize(1);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape &w_shape = context->GetInputShape(1)->GetOriginShape();

    uint32_t N = static_cast<uint32_t>(x_shape.GetDim(0));
    uint32_t H = static_cast<uint32_t>(x_shape.GetDim(2));
    uint32_t W = static_cast<uint32_t>(x_shape.GetDim(3));
    uint32_t K_h = static_cast<uint32_t>(w_shape.GetDim(2));
    uint32_t K_w = static_cast<uint32_t>(w_shape.GetDim(3));
    uint32_t H_out = H - K_h + 1;
    uint32_t W_out = W - K_w + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, static_cast<int64_t>(N));
    y_shape->SetDim(1, 1);
    y_shape->SetDim(2, static_cast<int64_t>(H_out));
    y_shape->SetDim(3, static_cast<int64_t>(W_out));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
