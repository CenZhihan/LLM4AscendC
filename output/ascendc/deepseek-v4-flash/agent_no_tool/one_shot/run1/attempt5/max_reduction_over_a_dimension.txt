#include "max_reduction_over_a_dimension_custom_tiling.h"

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxReductionOverADimensionCustomTilingData tiling;
    auto shape = context->GetInputShape(0);  // returns const gert::StorageShape*
    const int32_t nDims = static_cast<int32_t>(shape->GetDimNum());
    std::vector<uint32_t> dims(nDims);
    for (int32_t i = 0; i < nDims; ++i) {
        dims[i] = static_cast<uint32_t>(shape->GetDim(i));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : -1;
    if (dim < 0) dim += nDims;
    if (dim < 0 || dim >= nDims) return ge::GRAPH_FAILED;

    int32_t redDim = static_cast<int32_t>(dim);
    uint32_t outer = 1;
    for (int32_t i = 0; i < redDim; ++i) outer *= dims[i];
    uint32_t reductionLen = dims[redDim];
    uint32_t innerStride = 1;
    for (int32_t i = redDim + 1; i < nDims; ++i) innerStride *= dims[i];
    uint32_t totalOuter = outer * innerStride;

    // Block dim adaptive based on totalOuter
    uint32_t blockDim = 8;
    if (totalOuter >= 24) blockDim = 24;
    else if (totalOuter >= 16) blockDim = 16;
    else if (totalOuter >= 8) blockDim = 8;
    context->SetBlockDim(blockDim);

    tiling.set_outerElements(totalOuter);
    tiling.set_reductionLength(reductionLen);
    tiling.set_innerStride(innerStride);
    tiling.set_blockDim(blockDim);
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    auto inShape = context->GetInputShape(0);  // returns const gert::StorageShape*
    const int32_t nDims = static_cast<int32_t>(inShape->GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : -1;
    if (dim < 0) dim += nDims;
    if (dim < 0 || dim >= nDims) return GRAPH_FAILED;

    // Build output shape: remove the reduced dimension
    gert::Shape outShape;
    outShape.SetDimNum(nDims - 1);
    int32_t outIdx = 0;
    for (int32_t i = 0; i < nDims; ++i) {
        if (i != dim) {
            outShape.SetDim(outIdx, inShape->GetDim(i));
            ++outIdx;
        }
    }
    *context->GetOutputShape(0) = outShape;
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, context->GetInputDataType(0));

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxReduction {
public:
    __aicore__ inline KernelMaxReduction() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerElements, uint32_t reductionLength,
                                uint32_t innerStride)
    {
        outerElems_ = outerElements;
        redLen_ = reductionLength;
        innerStride_ = innerStride;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        // Allocate UB for a single reduction line (contiguous copy if stride >1 we handle per element)
        // We'll load each element individually to handle strided access.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, sizeof(float));   // for each element load
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));  // for output element
        pipe_.InitBuffer(tmpBuf_, 1, sizeof(float));              // temp for max
    }

    __aicore__ inline void Process()
    {
        if (outerElems_ == 0 || redLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stride = (outerElems_ + blockNum - 1) / blockNum;
        uint32_t start = blockIdx * stride;
        uint32_t end = (start + stride > outerElems_) ? outerElems_ : start + stride;

        for (uint32_t i = start; i < end; ++i) {
            // Compute base offset for reduction sequence i
            uint32_t outerBlockIdx = i / innerStride_;
            uint32_t offsetInBlock = i % innerStride_;
            uint32_t baseIndex = outerBlockIdx * redLen_ * innerStride_ + offsetInBlock;

            // Load first element
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[baseIndex], 1);  // copy one float
            float maxVal = xLocal(0);
            inQueueX_.FreeTensor(xLocal);

            // Iterate over remaining reduction elements and update max
            for (uint32_t j = 1; j < redLen_; ++j) {
                AscendC::LocalTensor<float> xLocal2 = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal2, xGm_[baseIndex + j * innerStride_], 1);
                float val = xLocal2(0);
                inQueueX_.FreeTensor(xLocal2);
                if (val > maxVal) maxVal = val;
            }

            // Write result
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            yLocal(0) = maxVal;
            AscendC::DataCopy(yGm_[i], yLocal, 1);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outerElems_ {0};
    uint32_t redLen_ {0};
    uint32_t innerStride_ {1};
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReduction op;
    op.Init(x, y, tiling_data.outerElements, tiling_data.reductionLength, tiling_data.innerStride);
    op.Process();
}

#include <torch/library.h>

at::Tensor max_reduction_impl_npu(const at::Tensor& self, int64_t dim) {
    auto output_shape = self.sizes().vec();
    if (dim < 0) dim += self.dim();
    output_shape.erase(output_shape.begin() + dim);
    at::Tensor result = at::empty(output_shape, self.options());
    EXEC_NPU_CMD(aclnnMaxReductionOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_reduction_over_a_dimension_custom", &max_reduction_impl_npu);
