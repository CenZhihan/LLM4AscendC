host_operator_src = """
#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;
    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalDims = x_shape.GetDimNum();

    // Compute total output length (product of all dims except last)
    uint32_t totalOutputLength = 1;
    uint32_t reductionSize = x_shape.GetDim(totalDims - 1);
    for (uint32_t i = 0; i < totalDims - 1; ++i)
        totalOutputLength *= x_shape.GetDim(i);

    if (totalOutputLength == 0 || reductionSize == 0) {
        tiling.set_totalOutputLength(0);
        tiling.set_tileLength(0);
        tiling.set_reductionSize(0);
        tiling.set_blockDim(1);
        tiling.set_invRedSize(0.0f);
        context->SetBlockDim(1);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    // Tile over output elements
    uint32_t maxTileOutput = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTileOutput = AlignDown(maxTileOutput, ALIGN_ELEMS);
    if (maxTileOutput < ALIGN_ELEMS) maxTileOutput = ALIGN_ELEMS;
    if (maxTileOutput > HARD_CAP_ELEMS) maxTileOutput = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTileOutput;
    if (totalOutputLength < tileLen) {
        const uint32_t aligned = AlignDown(totalOutputLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutputLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutputLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    // Compute reciprocal on host to avoid cast in kernel
    float invRedSize = 1.0f / static_cast<float>(reductionSize);

    context->SetBlockDim(blockDim);

    tiling.set_totalOutputLength(totalOutputLength);
    tiling.set_tileLength(tileLen);
    tiling.set_reductionSize(reductionSize);
    tiling.set_blockDim(blockDim);
    tiling.set_invRedSize(invRedSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t totalDims = x_shape->GetDimNum();
    if (totalDims == 0) {
        return GRAPH_FAILED;
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    (void)y_shape->SetDimNum(totalDims - 1);
    for (uint32_t i = 0; i < totalDims - 1; ++i) {
        y_shape->SetDim(i, x_shape->GetDim(i));
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MeanReductionOverADimensionCustom : public OpDef {
public:
    explicit MeanReductionOverADimensionCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(-1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MeanReductionOverADimensionCustom);
} // namespace ops

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReduction {
public:
    __aicore__ inline KernelMeanReduction() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputLength, uint32_t tileLength,
                                uint32_t reductionSize, float invRedSize)
    {
        totalOutLen_ = totalOutputLength;
        tileLen_     = tileLength;
        redSize_     = reductionSize;
        invRedSize_  = invRedSize;   // directly from tiling data (float, no cast)

        uint32_t totalInputLen = totalOutLen_ * redSize_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);

        // Queues not used for scalar access, but keep for compatibility
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideOut = blockNum * tileLen_;
        uint32_t baseOut = blockIdx * tileLen_;

        while (baseOut < totalOutLen_) {
            uint32_t curOut = totalOutLen_ - baseOut;
            if (curOut > tileLen_) curOut = tileLen_;

            for (uint32_t i = 0; i < curOut; ++i) {
                uint32_t outIdx = baseOut + i;
                float sum = 0.0f;

                for (uint32_t j = 0; j < redSize_; ++j) {
                    float val = xGm_.GetValue(outIdx * redSize_ + j);
                    sum += val;
                }

                float mean = sum * invRedSize_;
                yGm_.SetValue(outIdx, mean);
            }

            baseOut += strideOut;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t redSize_ {0};
    float invRedSize_ {0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReduction op;
    op.Init(x, y, tiling_data.totalOutputLength, tiling_data.tileLength, tiling_data.reductionSize, tiling_data.invRedSize);
    op.Process();
}
"""

python_bind_src = """
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor mean_reduction_impl_npu(const at::Tensor& self, int64_t dim) {
    // Permute so that the reduction dimension becomes the last (innermost) dim
    at::Tensor x_perm = self.transpose(dim, -1);
    // The custom operator expects shape [..., redSize] and outputs [...,]
    at::Tensor result = at::empty(x_perm.sizes().slice(0, x_perm.dim()-1), self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, x_perm, -1, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("mean_reduction_over_a_dimension_custom", &mean_reduction_impl_npu, "Mean reduction over a dimension (MeanReductionOverADimensionCustom on NPU)");
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'mean_reduction_over_a_dimension_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, dim):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x):
        # Permute so the reduction dimension is last, then call custom kernel.
        fn = getattr(custom_ops_lib, _FN)
        x_perm = x.transpose(self.dim, -1)
        y = fn(x_perm)
        # y now has shape same as x_perm but with last dimension removed.
        return y
'''
