#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    // Get normalized_shape list (dimensions to normalize over)
    const int64_t *normShape = attrs->GetAttrPointer<int64_t>(0);
    uint32_t normDims = static_cast<uint32_t>(attrs->GetAttrSize(0));
    uint32_t innerSize = 1;
    for (uint32_t i = 0; i < normDims; ++i) {
        innerSize *= static_cast<uint32_t>(normShape[i]);
    }

    // Get eps
    const float *eps = attrs->GetAttrPointer<float>(1);

    uint32_t outerSize = (innerSize == 0) ? 0u : totalLength / innerSize;

    // tile length from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (innerSize < tileLen) {
        const uint32_t aligned = AlignDown(innerSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : innerSize;
    }

    uint32_t blockDim = outerSize;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_innerSize(innerSize);
    tiling.set_outerSize(outerSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t innerSize,
                                uint32_t outerSize, uint32_t tileLength,
                                float eps)
    {
        totalLen_ = totalLength;
        innerSize_ = innerSize;
        outerSize_ = outerSize;
        tileLen_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerSize_ == 0 || outerSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over outer (batch) dimension
        for (uint32_t outerIdx = blockIdx; outerIdx < outerSize_; outerIdx += blockNum) {
            const uint32_t baseGm = outerIdx * innerSize_;

            // ----- First pass: accumulate sum and sum of squares -----
            float sum = 0.0f;
            float sumSq = 0.0f;

            uint32_t offset = 0;
            while (offset < innerSize_) {
                uint32_t len = innerSize_ - offset;
                if (len > tileLen_) len = tileLen_;

                CopyIn(baseGm + offset, len);

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> sqLocal = tmpQueue_.AllocTensor<float>();

                // square and reduce sum for x and x^2
                AscendC::Mul(sqLocal, xLocal, xLocal, len);
                float tileSum = AscendC::ReduceSum(xLocal, len);
                float tileSumSq = AscendC::ReduceSum(sqLocal, len);

                sum += tileSum;
                sumSq += tileSumSq;

                tmpQueue_.FreeTensor(sqLocal);
                inQueueX_.FreeTensor(xLocal);

                offset += len;
            }

            // Compute mean and variance
            float mean = sum / static_cast<float>(innerSize_);
            float var = sumSq / static_cast<float>(innerSize_) - mean * mean;
            float invStd = 1.0f / AscendC::Sqrt(var + eps_);

            // ----- Second pass: normalize and write output -----
            offset = 0;
            while (offset < innerSize_) {
                uint32_t len = innerSize_ - offset;
                if (len > tileLen_) len = tileLen_;

                CopyIn(baseGm + offset, len);

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // y = (x - mean) * invStd
                AscendC::Adds(yLocal, xLocal, -mean, len);
                AscendC::Muls(yLocal, yLocal, invStd, len);

                outQueueY_.EnQue<float>(yLocal);
                inQueueX_.FreeTensor(xLocal);

                CopyOut(baseGm + offset, len);

                offset += len;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t innerSize_ {0};
    uint32_t outerSize_ {0};
    uint32_t tileLen_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.innerSize,
            tiling_data.outerSize, tiling_data.tileLength, tiling_data.eps);
    op.Process();
}
