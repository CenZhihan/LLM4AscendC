host_operator_src = """
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    int32_t dim = static_cast<int32_t>(*dimAttr);
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    int32_t ndims = shape.GetDimNum();
    if (dim < 0) dim += ndims;
    uint32_t rowLen = 1;
    if (ndims > 0) {
        rowLen = static_cast<uint32_t>(shape.GetDim(dim));
    } else {
        rowLen = totalLength;
    }

    // UB budget: for log_softmax we need at least 5 local tensors of tileLen floats:
    // x, max, sub, exp, sum, log -> we can reuse some.
    // Use 5 * tileLen * sizeof(float) <= UB_BUDGET_BYTES
    uint32_t maxTile = UB_BUDGET_BYTES / (5 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (rowLen < tileLen) {
        tileLen = AlignDown(rowLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = rowLen;
    }

    // number of rows
    uint32_t numRows = (rowLen == 0) ? 0 : (totalLength / rowLen);
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLen(rowLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLen,
                                uint32_t tileLen, float negativeSlope)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLen;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue3_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a contiguous set of rows (grid-stride over rows)
        uint32_t rowsPerBlock = CeilDiv(rowLen_, blockNum);  // approximate, but we want each block to handle several rows
        uint32_t startRow = blockIdx * rowsPerBlock;
        uint32_t endRow = startRow + rowsPerBlock;
        if (startRow >= rowLen_) return;
        if (endRow > rowLen_) endRow = rowLen_;

        for (uint32_t row = startRow; row < endRow; row++) {
            uint32_t base = row * rowLen_;
            ProcessRow(base);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t base)
    {
        uint32_t remaining = rowLen_;
        uint32_t offset = 0;

        // First pass: compute max
        float rowMax = -FLT_MAX;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            CopyIn(base + offset, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> maxLocal = tmpQueue1_.AllocTensor<float>();
            AscendC::LocalTensor<float> oneLocal = tmpQueue2_.AllocTensor<float>();

            // Use ReduceMax to get max of this tile
            AscendC::ReduceMax(maxLocal, xLocal, oneLocal, curLen);
            float tileMax = maxLocal[0];
            if (tileMax > rowMax) rowMax = tileMax;

            tmpQueue1_.FreeTensor(maxLocal);
            tmpQueue2_.FreeTensor(oneLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }

        // Second pass: compute sum of exp(x - max) and then log softmax
        remaining = rowLen_;
        offset = 0;
        float sumExp = 0.0f;

        // First sub-pass to compute sum of exp
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            CopyIn(base + offset, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> subLocal = tmpQueue1_.AllocTensor<float>();
            AscendC::LocalTensor<float> expLocal = tmpQueue2_.AllocTensor<float>();
            AscendC::LocalTensor<float> sumLocal = tmpQueue3_.AllocTensor<float>();

            // sub = x - rowMax
            AscendC::Subs(subLocal, xLocal, rowMax, curLen);
            // exp = exp(sub)
            AscendC::Exp(expLocal, subLocal, curLen);
            // reduce sum (we'll accumulate manually via scalar)
            AscendC::ReduceSum(sumLocal, expLocal, sumLocal, curLen);
            float tileSum = sumLocal[0];
            sumExp += tileSum;

            tmpQueue1_.FreeTensor(subLocal);
            tmpQueue2_.FreeTensor(expLocal);
            tmpQueue3_.FreeTensor(sumLocal);
            inQueueX_.FreeTensor(xLocal);
            offset += curLen;
            remaining -= curLen;
        }

        // Third pass: compute y = x - log(sumExp)   (since exp(x-max) sums to sumExp, log softmax = x - max - log(sumExp) but we already subtracted max? Actually formula: log_softmax = x - max - log(sum(exp(x-max))). We have sumExp = sum(exp(x-max)). So y = x - max - log(sumExp). Our sub = x-max, then sub - log(sumExp). We'll do in one pass.
        float logSumExp = logf(sumExp);
        remaining = rowLen_;
        offset = 0;
        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            CopyIn(base + offset, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> subLocal = tmpQueue1_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // sub = x - max
            AscendC::Subs(subLocal, xLocal, rowMax, curLen);
            // y = sub - logSumExp
            AscendC::Subs(yLocal, subLocal, logSumExp, curLen);

            tmpQueue1_.FreeTensor(subLocal);
            inQueueX_.FreeTensor(xLocal);
            outQueueY_.EnQue<float>(yLocal);

            CopyOut(base + offset, curLen);
            offset += curLen;
            remaining -= curLen;
        }
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

    static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue3_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLen, tiling_data.tileLen, 0.0f);
    op.Process();
}
"""
