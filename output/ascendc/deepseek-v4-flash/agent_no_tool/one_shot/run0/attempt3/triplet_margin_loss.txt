#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;

static uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t N = static_cast<uint32_t>(shape.GetDim(0));
    const uint32_t D = static_cast<uint32_t>(shape.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // Compute tile length for D based on UB budget (3 rows + temporary buffers)
    uint32_t tileLen = D;
    uint32_t bytesNeeded = tileLen * 3 * sizeof(float) + 4 * sizeof(float); // plus scalar buffers
    if (bytesNeeded > UB_BUDGET_BYTES) {
        tileLen = AlignDown(UB_BUDGET_BYTES / (3 * sizeof(float)), ALIGN_ELEMS);
        if (tileLen > D) tileLen = D;
        if (tileLen == 0) tileLen = ALIGN_ELEMS;
    }

    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    float invN = 1.0f / static_cast<float>(N);

    tiling.set_totalLength(N);
    tiling.set_fullD(D);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);
    tiling.set_invN(invN);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t SCALAR_SIZE = 1;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t totalLength, uint32_t fullD,
                                uint32_t tileLength, float margin, float invN)
    {
        N_ = totalLength;
        D_ = fullD;              // full feature dimension
        tileLen_ = tileLength;   // chunk size for D
        margin_ = margin;
        invN_ = invN;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, N_ * D_);
        posGm_.SetGlobalBuffer((__gm__ float*)positive, N_ * D_);
        negGm_.SetGlobalBuffer((__gm__ float*)negative, N_ * D_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        pipe_.InitBuffer(inAnchor_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inPos_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inNeg_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpDiff_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarBuf_, 1, SCALAR_SIZE * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || D_ == 0) {
            lossGm_.SetValue(0, 0.0f);
            return;
        }

        // Accumulator for sum of losses over all samples
        AscendC::LocalTensor<float> sumLoss = scalarBuf_.AllocTensor<float>();
        AscendC::SetZero(sumLoss, SCALAR_SIZE);

        uint32_t numTiles = (D_ + tileLen_ - 1) / tileLen_;

        for (uint32_t i = 0; i < N_; i++) {
            uint32_t baseOffset = i * D_;

            // Temporary per-sample partial sums for dap and dan (scalars)
            AscendC::LocalTensor<float> dapAcc = scalarBuf_.AllocTensor<float>();
            AscendC::SetZero(dapAcc, SCALAR_SIZE);
            AscendC::LocalTensor<float> danAcc = scalarBuf_.AllocTensor<float>();
            AscendC::SetZero(danAcc, SCALAR_SIZE);

            // Process each tile of the feature dimension
            for (uint32_t t = 0; t < numTiles; t++) {
                uint32_t offset = baseOffset + t * tileLen_;
                uint32_t curLen = (t == numTiles - 1) ? D_ - t * tileLen_ : tileLen_;
                if (curLen == 0) break;

                // Prefetch anchor, positive, negative for this tile
                AscendC::LocalTensor<float> aLocal = inAnchor_.AllocTensor<float>();
                AscendC::DataCopy(aLocal, anchorGm_[offset], curLen);
                AscendC::LocalTensor<float> pLocal = inPos_.AllocTensor<float>();
                AscendC::DataCopy(pLocal, posGm_[offset], curLen);
                AscendC::LocalTensor<float> nLocal = inNeg_.AllocTensor<float>();
                AscendC::DataCopy(nLocal, negGm_[offset], curLen);

                // dap += sum((a-p)^2)
                AscendC::LocalTensor<float> diff = tmpDiff_.AllocTensor<float>();
                AscendC::Sub(diff, aLocal, pLocal, curLen);
                AscendC::Mul(diff, diff, diff, curLen);
                AscendC::LocalTensor<float> tileDap = scalarBuf_.AllocTensor<float>();
                AscendC::ReduceSum(tileDap, diff, curLen, 1);
                AscendC::Add(dapAcc, dapAcc, tileDap, SCALAR_SIZE);
                scalarBuf_.FreeTensor(tileDap);

                // dan += sum((a-n)^2)
                AscendC::Sub(diff, aLocal, nLocal, curLen);
                AscendC::Mul(diff, diff, diff, curLen);
                AscendC::LocalTensor<float> tileDan = scalarBuf_.AllocTensor<float>();
                AscendC::ReduceSum(tileDan, diff, curLen, 1);
                AscendC::Add(danAcc, danAcc, tileDan, SCALAR_SIZE);
                scalarBuf_.FreeTensor(tileDan);

                // Free local tensors
                tmpDiff_.FreeTensor(diff);
                inAnchor_.FreeTensor(aLocal);
                inPos_.FreeTensor(pLocal);
                inNeg_.FreeTensor(nLocal);
            }

            // loss_i = max(0, dap - dan + margin)
            AscendC::Sub(dapAcc, dapAcc, danAcc, SCALAR_SIZE);
            scalarBuf_.FreeTensor(danAcc);
            AscendC::Adds(dapAcc, dapAcc, margin_, SCALAR_SIZE);
            AscendC::Relu(dapAcc, dapAcc, SCALAR_SIZE);  // max(0, x)

            // sumLoss += loss_i
            AscendC::Add(sumLoss, sumLoss, dapAcc, SCALAR_SIZE);
            scalarBuf_.FreeTensor(dapAcc);
        }

        // loss = sumLoss / N
        AscendC::Muls(sumLoss, sumLoss, invN_, SCALAR_SIZE);
        AscendC::DataCopy(lossGm_[0], sumLoss, SCALAR_SIZE);
        scalarBuf_.FreeTensor(sumLoss);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inPos_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inNeg_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> dummyOut_; // not used
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDiff_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBuf_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> posGm_;
    AscendC::GlobalTensor<float> negGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t N_ {0};
    uint32_t D_ {0};
    uint32_t tileLen_ {0};
    float margin_ {1.0f};
    float invN_ {0.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR loss, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalLength, tiling_data.fullD, tiling_data.tileLength,
            tiling_data.margin, tiling_data.invN);
    op.Process();
}
