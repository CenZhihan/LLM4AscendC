static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;
    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalDims = x_shape.GetDimNum();

    // Compute total output length (product of all dims except last)
    uint32_t totalOutputLength = 1;
    uint32_t reductionSize = x_shape.GetDim(totalDims - 1);
    for (uint32_t i = 0; i < totalDims - 1; ++i)
        totalOutputLength *= x_shape.GetDim(i);

    if (totalOutputLength == 0 || reductionSize == 0) {
        tiling.set_totalOutputLength(0);
        tiling.set_tileLength(0);
        tiling.set_reductionSize(0);
        tiling.set_blockDim(1);
        context->SetBlockDim(1);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    // Tile over output elements; each tile needs tileLen floats for accumulator and tileLen floats for one reduction step
    uint32_t maxTileOutput = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTileOutput = AlignDown(maxTileOutput, ALIGN_ELEMS);
    if (maxTileOutput < ALIGN_ELEMS) maxTileOutput = ALIGN_ELEMS;
    if (maxTileOutput > HARD_CAP_ELEMS) maxTileOutput = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTileOutput;
    if (totalOutputLength < tileLen) {
        const uint32_t aligned = AlignDown(totalOutputLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutputLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutputLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutputLength(totalOutputLength);
    tiling.set_tileLength(tileLen);
    tiling.set_reductionSize(reductionSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

                //   for each j in [0, redSize_), input index = (baseOut + i) * redSize_ + j, for i=0..curOut-1.
                // This is a contiguous block of size curOut * redSize_ when you take all j consecutively? No, the whole tile's input block is contiguous: [baseOut*redSize_, (baseOut+curOut)*redSize_).

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReduction {
public:
    __aicore__ inline KernelMeanReduction() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputLength, uint32_t tileLength,
                                uint32_t reductionSize)
    {
        totalOutLen_ = totalOutputLength;
        tileLen_     = tileLength;
        redSize_     = reductionSize;
        invRedSize_  = 1.0f / (float)reductionSize;

        uint32_t totalInputLen = totalOutLen_ * redSize_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, sizeof(float)); // single element per buffer for scalar reduction
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideOut = blockNum * tileLen_;
        uint32_t baseOut = blockIdx * tileLen_;

        while (baseOut < totalOutLen_) {
            uint32_t curOut = totalOutLen_ - baseOut;
            if (curOut > tileLen_) curOut = tileLen_;

            // Process each output element in this tile sequentially
            for (uint32_t i = 0; i < curOut; ++i) {
                uint32_t outIdx = baseOut + i;
                float sum = 0.0f;

                // Sum over reduction dimension using scalar loads
                for (uint32_t j = 0; j < redSize_; ++j) {
                    float val = xGm_.GetValue(outIdx * redSize_ + j);
                    sum += val;
                }

                // Compute mean
                float mean = sum * invRedSize_;

                // Store result
                yGm_.SetValue(outIdx, mean);
            }

            baseOut += strideOut;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t redSize_ {0};
    float invRedSize_ {0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReduction op;
    op.Init(x, y, tiling_data.totalOutputLength, tiling_data.tileLength, tiling_data.reductionSize);
    op.Process();
}
