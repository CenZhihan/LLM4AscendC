#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  
static constexpr uint32_t ALIGN_ELEMS = 2048;            

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttr = attrs->GetAttrPointer<int64_t>(0);
    int32_t dim = static_cast<int32_t>(*dimAttr);
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    int32_t ndims = shape.GetDimNum();
    if (dim < 0) dim += ndims;
    uint32_t dimSize = 1;
    if (ndims > 0) {
        dimSize = static_cast<uint32_t>(shape.GetDim(dim));
    } else {
        dimSize = totalLength;
    }

    // Determine tileLen from UB budget: we need approx 6 temporary tensors of size tileLen
    // plus some small ones. 6 * tileLen * 4 bytes <= UB_BUDGET_BYTES
    uint32_t maxTile = UB_BUDGET_BYTES / (6 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (dimSize < tileLen) {
        tileLen = AlignDown(dimSize, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = dimSize;
    }

    uint32_t numRows = (dimSize == 0) ? 0 : (totalLength / dimSize);
    uint32_t blockDim = numRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLen,
                                int32_t dim)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLen;
        dim_ = dim;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, rowLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, rowLen_ * sizeof(float));
        // tmpQueue: maxLocal(1), maxVec(rowLen), expLocal(rowLen), sumLocal(1), logSumLocal(1), logSumVec(rowLen)
        // Also need a few single-element tensors for reductions
        pipe_.InitBuffer(tmpQueue_, 6, rowLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numRows = totalLen_ / rowLen_;

        for (uint32_t row = blockIdx; row < numRows; row += blockNum) {
            uint32_t base = row * rowLen_;
            ProcessRow(base);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t base)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[base], rowLen_);
        inQueueX_.EnQue(xLocal);

        xLocal = inQueueX_.DeQue<float>();

        // Step 1: row max (reduce to scalar)
        AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();   // size 1
        AscendC::ReduceMax(maxLocal, xLocal, rowLen_, 1, AscendC::VECTOR);
        float maxVal = maxLocal.GetValue(0);
        tmpQueue_.FreeTensor(maxLocal);

        // Step 2: x - max (broadcast max to a vector)
        AscendC::LocalTensor<float> maxVec = tmpQueue_.AllocTensor<float>();     // size rowLen_
        AscendC::VecDup(maxVec, maxVal, rowLen_);
        AscendC::Sub(xLocal, xLocal, maxVec, rowLen_);
        tmpQueue_.FreeTensor(maxVec);

        // Step 3: exp(x - max)
        AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();   // size rowLen_
        AscendC::Exp(expLocal, xLocal, rowLen_);

        // Step 4: sum of exp (reduce to scalar)
        AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();   // size 1
        AscendC::ReduceSum(sumLocal, expLocal, rowLen_, 1, AscendC::VECTOR);
        float sumVal = sumLocal.GetValue(0);
        tmpQueue_.FreeTensor(sumLocal);

        // Step 5: log(sum)
        AscendC::LocalTensor<float> logSumLocal = tmpQueue_.AllocTensor<float>(); // size 1
        AscendC::LocalTensor<float> sumScalar = tmpQueue_.AllocTensor<float>();   // size 1 (temporary)
        sumScalar.SetValue(0, sumVal);
        AscendC::Log(logSumLocal, sumScalar, 1);
        float logSum = logSumLocal.GetValue(0);
        tmpQueue_.FreeTensor(logSumLocal);
        tmpQueue_.FreeTensor(sumScalar);

        // Step 6: y = (x - max) - log(sum)
        // Use expLocal as output buffer (reuse)
        AscendC::LocalTensor<float> logSumVec = tmpQueue_.AllocTensor<float>();  // size rowLen_
        AscendC::VecDup(logSumVec, logSum, rowLen_);
        AscendC::Sub(expLocal, xLocal, logSumVec, rowLen_);
        tmpQueue_.FreeTensor(logSumVec);

        // Free remaining temps
        tmpQueue_.FreeTensor(expLocal);

        // Enqueue result and copy out
        outQueueY_.EnQue<float>(xLocal);  // Note: xLocal now has final result? Actually we used expLocal as output, but xLocal still holds (x-max). We'll use expLocal.
        // But we freed expLocal. Let's fix: we need to keep expLocal as result.
        // Actually above we freed expLocal incorrectly. Let's reorder:
        // We'll not free expLocal after Sub; instead we enqueue it.
        // So we need to allocate expLocal from outQueueY_? Or reuse xLocal.
        // Let's restructure: Instead of allocating expLocal separately, we can reuse xLocal as output.
        // But we used xLocal for subtraction. After step 6, expLocal holds result.
        // So we should enqueue expLocal, not xLocal. So we shouldn't free expLocal.
        // Let's redo this method more cleanly.

        // Actually we will rewrite ProcessRow to be correct.
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 6> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t rowLen_{0};
    int32_t dim_{1};

    __aicore__ inline void ProcessRow(uint32_t base)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[base], rowLen_);
        inQueueX_.EnQue(xLocal);

        xLocal = inQueueX_.DeQue<float>();

        // Step 1: row max
        AscendC::LocalTensor<float> maxLocal = tmpQueue_.AllocTensor<float>();   // size 1
        AscendC::ReduceMax(maxLocal, xLocal, rowLen_, 1, AscendC::VECTOR);
        float maxVal = maxLocal.GetValue(0);
        tmpQueue_.FreeTensor(maxLocal);

        // Step 2: subtract max from x
        AscendC::LocalTensor<float> maxVec = tmpQueue_.AllocTensor<float>();     // size rowLen_
        AscendC::VecDup(maxVec, maxVal, rowLen_);
        AscendC::Sub(xLocal, xLocal, maxVec, rowLen_);
        tmpQueue_.FreeTensor(maxVec);

        // Step 3: exp(x - max)
        AscendC::LocalTensor<float> expLocal = tmpQueue_.AllocTensor<float>();   // size rowLen_
        AscendC::Exp(expLocal, xLocal, rowLen_);

        // Step 4: sum of exp
        AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();   // size 1
        AscendC::ReduceSum(sumLocal, expLocal, rowLen_, 1, AscendC::VECTOR);
        float sumVal = sumLocal.GetValue(0);
        tmpQueue_.FreeTensor(sumLocal);

        // Step 5: log(sum)
        AscendC::LocalTensor<float> logSumLocal = tmpQueue_.AllocTensor<float>(); // size 1
        AscendC::LocalTensor<float> sumScalar = tmpQueue_.AllocTensor<float>();   // size 1
        sumScalar.SetValue(0, sumVal);
        AscendC::Log(logSumLocal, sumScalar, 1);
        float logSum = logSumLocal.GetValue(0);
        tmpQueue_.FreeTensor(logSumLocal);
        tmpQueue_.FreeTensor(sumScalar);

        // Step 6: y = (x - max) - log(sum)   => expLocal reused as output
        AscendC::LocalTensor<float> logSumVec = tmpQueue_.AllocTensor<float>();  // size rowLen_
        AscendC::VecDup(logSumVec, logSum, rowLen_);
        AscendC::Sub(expLocal, xLocal, logSumVec, rowLen_);
        tmpQueue_.FreeTensor(logSumVec);

        // expLocal now holds final result
        outQueueY_.EnQue<float>(expLocal);
        inQueueX_.FreeTensor(xLocal);

        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[base], yLocal, rowLen_);
        outQueueY_.FreeTensor(yLocal);
    }
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.rowLen, tiling_data.dim);
    op.Process();
}
"""
