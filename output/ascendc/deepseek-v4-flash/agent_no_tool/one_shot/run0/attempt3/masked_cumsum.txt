#include "masked_cumsum_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t MAX_TILE_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaskedCumsumCustomTilingData tiling;

    const gert::StorageShape *storageShape = context->GetInputShape(0);
    const gert::Shape &originShape = storageShape->GetOriginShape();
    uint32_t numDims = originShape.GetDimNum();
    uint32_t totalLength = static_cast<uint32_t>(originShape.GetShapeSize());

    if (numDims == 0 || numDims > 2) return ge::GRAPH_FAILED;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dimPtr != nullptr) ? *dimPtr : 1;

    uint32_t rowLength = 0;
    if (numDims == 1) {
        rowLength = totalLength;
    } else {
        uint32_t dim0 = static_cast<uint32_t>(originShape.GetDim(0));
        uint32_t dim1 = static_cast<uint32_t>(originShape.GetDim(1));
        rowLength = (dim == 0) ? dim0 : dim1;
    }

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    if (maxTile > MAX_TILE_ELEMS) maxTile = MAX_TILE_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = ALIGN_ELEMS;

    uint32_t tileLength = (rowLength < maxTile) ? rowLength : maxTile;
    if (tileLength == 0) tileLength = rowLength;

    uint32_t seqCount = totalLength / rowLength;

    uint32_t blockDim = 24;
    if (seqCount < blockDim) blockDim = (seqCount > 0) ? seqCount : 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength,
                                uint32_t tileLength, uint32_t blockDim)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t seqCount = totalLen_ / rowLen_;

        for (uint32_t seq = blockIdx; seq < seqCount; seq += blockNum) {
            uint32_t baseOffset = seq * rowLen_;
            float accum = 0.0f;

            // Prefetch first tile
            uint32_t tileStart = 0;
            uint32_t curTileLen = (rowLen_ > tileLen_) ? tileLen_ : rowLen_;
            CopyIn(baseOffset, tileStart, curTileLen);

            while (true) {
                // Prefetch next tile
                uint32_t nextTileStart = tileStart + curTileLen;
                uint32_t nextTileLen = 0;
                if (nextTileStart < rowLen_) {
                    uint32_t remaining = rowLen_ - nextTileStart;
                    nextTileLen = (remaining > tileLen_) ? tileLen_ : remaining;
                    CopyIn(baseOffset, nextTileStart, nextTileLen);
                }

                // Compute current tile
                Compute(baseOffset, tileStart, curTileLen, accum);

                // Copy out
                CopyOut(baseOffset, tileStart, curTileLen);

                if (nextTileLen == 0) break;
                tileStart = nextTileStart;
                curTileLen = nextTileLen;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t baseOffset, uint32_t tileStart, uint32_t len)
    {
        // X buffer
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[baseOffset + tileStart], len);
        inQueueX_.EnQue(xLocal);

        // Mask buffer
        AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
        AscendC::DataCopy(maskLocal, maskGm_[baseOffset + tileStart], len);
        inQueueMask_.EnQue(maskLocal);
    }

    __aicore__ inline void Compute(uint32_t baseOffset, uint32_t tileStart,
                                   uint32_t len, float& accum)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> maskLocal = inQueueMask_.DeQue<float>();

        // x = x * mask
        AscendC::Mul(xLocal, xLocal, maskLocal, len);
        inQueueMask_.FreeTensor(maskLocal);

        // In-place prefix sum with carry from previous tile
        for (uint32_t i = 0; i < len; i++) {
            accum += xLocal(i);
            xLocal(i) = accum;
        }

        // Enqueue result for output
        outQueueY_.EnQue<float>(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t baseOffset, uint32_t tileStart, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[baseOffset + tileStart], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(
    GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.totalLength, tiling_data.rowLength,
            tiling_data.tileLength, tiling_data.blockDim);
    op.Process();
}
