#include "argmax_over_a_dimension_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_FLOATS = 192 * 1024 / 4; // 48k floats (192KB)
static constexpr uint32_t ALIGN = 32;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;
    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t ndim = xShape->GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;

    uint32_t outerSize = 1;
    for (uint32_t i = 0; i < static_cast<uint32_t>(dim); ++i) {
        outerSize *= static_cast<uint32_t>(xShape->GetDim(i));
    }
    uint32_t reducedDim = static_cast<uint32_t>(xShape->GetDim(dim));
    uint32_t innerSize = 1;
    for (uint32_t i = static_cast<uint32_t>(dim) + 1; i < ndim; ++i) {
        innerSize *= static_cast<uint32_t>(xShape->GetDim(i));
    }
    uint32_t outputElems = outerSize * innerSize;

    uint32_t tileLen = UB_BUDGET_FLOATS / reducedDim;
    if (tileLen == 0) tileLen = 1;
    if (tileLen > outputElems) tileLen = outputElems;

    uint32_t numTiles = (tileLen == 0) ? 1 : (outputElems + tileLen - 1) / tileLen;
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_outputElems(outputElems);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_outerSize(outerSize);
    tiling.set_reducedDim(reducedDim);
    tiling.set_innerSize(innerSize);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    uint32_t ndim = xShape->GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = attrs->GetAttrPointer<int32_t>(0)[0];
    if (dim < 0) dim += ndim;

    uint32_t outDimsNum = ndim - 1;
    gert::Shape outShape;
    outShape.SetDimNum(outDimsNum);
    uint32_t outIdx = 0;
    for (uint32_t i = 0; i < ndim; ++i) {
        if (i != static_cast<uint32_t>(dim)) {
            outShape.SetDim(outIdx, xShape->GetDim(i));
            ++outIdx;
        }
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = outShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outputElems, uint32_t tileLen,
                                uint32_t outerSize, uint32_t reducedDim,
                                uint32_t innerSize, int32_t dim)
    {
        outputElems_ = outputElems;
        tileLen_ = tileLen;
        outerSize_ = outerSize;
        reducedDim_ = reducedDim;
        innerSize_ = innerSize;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, reducedDim_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(int64_t));
    }

    __aicore__ inline void Process()
    {
        if (outputElems_ == 0 || reducedDim_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t idx = blockIdx * tileLen_; idx < outputElems_; idx += blockNum * tileLen_) {
            uint32_t curLimit = idx + tileLen_;
            if (curLimit > outputElems_) curLimit = outputElems_;

            for (uint32_t i = idx; i < curLimit; ++i) {
                uint32_t blockIdx_ = i / innerSize_;
                uint32_t innerIdx = i % innerSize_;
                uint32_t base = blockIdx_ * (reducedDim_ * innerSize_) + innerIdx;

                float maxVal = -__builtin_huge_valf();
                int64_t maxIdx = 0;
                for (uint32_t j = 0; j < reducedDim_; ++j) {
                    float val = xGm_[base + j * innerSize_];
                    if (val > maxVal) {
                        maxVal = val;
                        maxIdx = static_cast<int64_t>(j);
                    }
                }
                yGm_[i] = maxIdx;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t outputElems_ {0};
    uint32_t tileLen_ {0};
    uint32_t outerSize_ {0};
    uint32_t reducedDim_ {0};
    uint32_t innerSize_ {0};
    int32_t dim_ {-1};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.outputElems, tiling_data.tileLen,
            tiling_data.outerSize, tiling_data.reducedDim,
            tiling_data.innerSize, tiling_data.dim);
    op.Process();
}

#include <torch/library.h>

#include "pytorch_npu_helper.hpp"

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    int64_t ndim = self.dim();
    if (dim < 0) dim += ndim;
    std::vector<int64_t> outShape;
    for (int64_t i = 0; i < ndim; ++i) {
        if (i != dim) {
            outShape.push_back(self.size(i));
        }
    }
    at::Tensor result = at::empty(outShape, self.options().dtype(at::kLong));
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu);
