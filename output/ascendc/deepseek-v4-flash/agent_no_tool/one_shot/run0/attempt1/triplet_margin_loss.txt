host_operator_src="""

#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;    // per‑core UB
static constexpr uint32_t ALIGN_ELEMS = 16;               // minimal alignment

static uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    // anchor, positive, negative all have same shape: (N, D)
    const auto &shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t N = static_cast<uint32_t>(shape.GetDim(0));
    const uint32_t D = static_cast<uint32_t>(shape.GetDim(1));

    // Read margin attribute
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // Tile length = D (feature dimension) – one sample per iteration.
    // For very large D (> UB/ (3*sizeof(float)*4) ) we would need to tile D,
    // but for this example D=128 is safe.
    // Buffers per iteration: anchor_row, positive_row, negative_row = 3*D floats.
    uint32_t tileLen = D;
    uint32_t bytesNeeded = tileLen * 3 * sizeof(float);
    if (bytesNeeded > UB_BUDGET_BYTES) {
        // fallback: compute maximum safe D that fits (tileLen = min(D, UB/12))
        tileLen = AlignDown(UB_BUDGET_BYTES / (3 * sizeof(float)), ALIGN_ELEMS);
        if (tileLen > D) tileLen = D;
        if (tileLen == 0) tileLen = ALIGN_ELEMS;
    }

    // blockDim = 1 (all samples processed sequentially in one block)
    uint32_t blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(N);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // output is a scalar (1 element)
    gert::Shape *loss_shape = context->GetOutputShape(0);
    loss_shape->SetDimNum(1);
    loss_shape->SetDim(0, 1);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR loss,
                                uint32_t totalLength, uint32_t tileLength,
                                float margin)
    {
        N_ = totalLength;          // number of samples
        D_ = tileLength;           // feature dimension
        margin_ = margin;
        invN_ = 1.0f / static_cast<float>(N_);

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, N_ * D_);
        posGm_.SetGlobalBuffer((__gm__ float*)positive, N_ * D_);
        negGm_.SetGlobalBuffer((__gm__ float*)negative, N_ * D_);
        lossGm_.SetGlobalBuffer((__gm__ float*)loss, 1);

        // Local buffers: one row for each tensor
        pipe_.InitBuffer(inAnchor_, BUFFER_NUM, D_ * sizeof(float));
        pipe_.InitBuffer(inPos_, BUFFER_NUM, D_ * sizeof(float));
        pipe_.InitBuffer(inNeg_, BUFFER_NUM, D_ * sizeof(float));
        // Temporary per‑element diff and square (reused between samples)
        pipe_.InitBuffer(tmpDiff_, 1, D_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || D_ == 0) {
            lossGm_.SetValue(0, 0.0f);
            return;
        }

        float sumLoss = 0.0f;

        for (uint32_t i = 0; i < N_; i++) {
            // Load anchor[i], positive[i], negative[i] (each D floats)
            uint32_t offset = i * D_;

            AscendC::LocalTensor<float> aLocal = inAnchor_.AllocTensor<float>();
            AscendC::DataCopy(aLocal, anchorGm_[offset], D_);

            AscendC::LocalTensor<float> pLocal = inPos_.AllocTensor<float>();
            AscendC::DataCopy(pLocal, posGm_[offset], D_);

            AscendC::LocalTensor<float> nLocal = inNeg_.AllocTensor<float>();
            AscendC::DataCopy(nLocal, negGm_[offset], D_);

            // Compute squared distance between anchor and positive
            // d_ap = sum((a - p)^2)
            AscendC::LocalTensor<float> diff = tmpDiff_.AllocTensor<float>();
            AscendC::Sub(diff, aLocal, pLocal, D_);
            AscendC::Mul(diff, diff, diff, D_);
            float dap = AscendC::ReduceSum(diff, D_);

            // Compute squared distance between anchor and negative
            // d_an = sum((a - n)^2)
            AscendC::Sub(diff, aLocal, nLocal, D_);
            AscendC::Mul(diff, diff, diff, D_);
            float dan = AscendC::ReduceSum(diff, D_);

            // loss_i = max(0, d_ap - d_an + margin)
            float lossI = dap - dan + margin_;
            if (lossI < 0.0f) lossI = 0.0f;
            sumLoss += lossI;

            // Free tensors
            inAnchor_.FreeTensor(aLocal);
            inPos_.FreeTensor(pLocal);
            inNeg_.FreeTensor(nLocal);
            tmpDiff_.FreeTensor(diff);
        }

        // Output mean loss
        float meanLoss = sumLoss * invN_;
        lossGm_.SetValue(0, meanLoss);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inPos_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inNeg_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpDiff_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> posGm_;
    AscendC::GlobalTensor<float> negGm_;
    AscendC::GlobalTensor<float> lossGm_;

    uint32_t N_ {0};
    uint32_t D_ {0};
    float margin_ {1.0f};
    float invN_ {0.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive,
                                                                  GM_ADDR negative, GM_ADDR loss,
                                                                  GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, loss,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.margin);
    op.Process();
}
