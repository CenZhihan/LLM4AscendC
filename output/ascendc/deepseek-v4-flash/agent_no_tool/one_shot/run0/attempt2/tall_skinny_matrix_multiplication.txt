host_operator_src="""
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget for matmul:
//   A_sub : tileM * totalK floats
//   B_sub : totalK * tileN floats
//   C_sub : tileM * tileN floats
// Total floats = tileM*totalK + totalK*tileN + tileM*tileN
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;
    const gert::StorageShape *shapeA = context->GetInputShape(0);
    const gert::StorageShape *shapeB = context->GetInputShape(1);
    uint32_t M = shapeA->GetOriginShape().GetDim(0);
    uint32_t K = shapeA->GetOriginShape().GetDim(1);
    uint32_t N = shapeB->GetOriginShape().GetDim(1);

    uint32_t maxTileN = 128;
    if (N < maxTileN) maxTileN = N;
    uint32_t tileN = AlignDown(maxTileN, ALIGN);
    if (tileN == 0) tileN = maxTileN;

    // Solve for tileM: (tileM*K + K*tileN + tileM*tileN)*4 <= UB_BUDGET_BYTES
    uint32_t numerator = UB_BUDGET_BYTES / 4 - K * tileN;
    uint32_t denominator = K + tileN;
    uint32_t maxTileM = numerator / denominator;
    maxTileM = AlignDown(maxTileM, ALIGN);
    if (maxTileM < ALIGN) maxTileM = ALIGN;
    if (maxTileM > M) maxTileM = AlignDown(M, ALIGN);
    if (maxTileM == 0) maxTileM = M;

    uint32_t tileM = maxTileM;

    uint32_t numTilesM = CeilDiv(M, tileM);
    uint32_t blockDim = numTilesM;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalM(M);
    tiling.set_totalN(N);
    tiling.set_totalK(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *b_shape = context->GetInputShape(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    std::vector<int64_t> dims = {a_shape->GetDim(0), b_shape->GetDim(1)};
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, dims[0]);
    c_shape->SetDim(1, dims[1]);
    return GRAPH_SUCCESS;
}

    const ge::DataType a_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatMul {
public:
    __aicore__ inline KernelTallSkinnyMatMul() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t totalM, uint32_t totalN, uint32_t totalK,
                                uint32_t tileM, uint32_t tileN) {
        A_gm_.SetGlobalBuffer((__gm__ float*)A, totalM * totalK);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, totalK * totalN);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, totalM * totalN);

        totalM_ = totalM;
        totalN_ = totalN;
        totalK_ = totalK;
        tileM_ = tileM;
        tileN_ = tileN;

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM_ * totalK_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, totalK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalM_ == 0 || totalN_ == 0 || totalK_ == 0) return;

        uint32_t blockNum = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();

        uint32_t numTilesM = (totalM_ + tileM_ - 1) / tileM_;
        uint32_t tilesPerBlock = (numTilesM + blockNum - 1) / blockNum;
        uint32_t startTile = blockIdx * tilesPerBlock;
        uint32_t endTile = (blockIdx + 1) * tilesPerBlock;
        if (startTile >= numTilesM) return;
        if (endTile > numTilesM) endTile = numTilesM;

        for (uint32_t t = startTile; t < endTile; ++t) {
            uint32_t mOffset = t * tileM_;
            uint32_t curM = (mOffset + tileM_ <= totalM_) ? tileM_ : (totalM_ - mOffset);
            if (curM == 0) continue;

            uint32_t numTilesN = (totalN_ + tileN_ - 1) / tileN_;
            for (uint32_t s = 0; s < numTilesN; ++s) {
                uint32_t nOffset = s * tileN_;
                uint32_t curN = (nOffset + tileN_ <= totalN_) ? tileN_ : (totalN_ - nOffset);
                if (curN == 0) continue;

                // Load A submatrix (curM * totalK) – contiguous
                AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
                AscendC::DataCopy(aLocal, A_gm_ + mOffset * totalK_, curM * totalK_);
                inQueueA_.EnQue(aLocal);

                // Load B submatrix (totalK * curN) – row by row via Slice
                AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
                for (uint32_t k = 0; k < totalK_; ++k) {
                    auto bRow = bLocal.Slice(k * curN, curN);
                    AscendC::DataCopy(bRow, B_gm_ + k * totalN_ + nOffset, curN);
                }
                inQueueB_.EnQue(bLocal);

                // Compute C submatrix (curM * curN) – scalar dot product
                AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
                for (uint32_t m = 0; m < curM; ++m) {
                    for (uint32_t n = 0; n < curN; ++n) {
                        float sum = 0.0f;
                        for (uint32_t k = 0; k < totalK_; ++k) {
                            sum += aLocal(m * totalK_ + k) * bLocal(k * curN + n);
                        }
                        cLocal.SetValue(m * curN + n, sum);
                    }
                }
                outQueueC_.EnQue(cLocal);

                // Write C submatrix to global memory – row by row
                AscendC::LocalTensor<float> cOut = outQueueC_.DeQue<float>();
                for (uint32_t row = 0; row < curM; ++row) {
                    auto cRow = cOut.Slice(row * curN, curN);
                    uint32_t gmOffset = (mOffset + row) * totalN_ + nOffset;
                    AscendC::DataCopy(C_gm_ + gmOffset, cRow, curN);
                }
                outQueueC_.FreeTensor(cOut);

                AscendC::LocalTensor<float> aTmp = inQueueA_.DeQue<float>();
                inQueueA_.FreeTensor(aTmp);
                AscendC::LocalTensor<float> bTmp = inQueueB_.DeQue<float>();
                inQueueB_.FreeTensor(bTmp);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;

    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t totalM_ {0};
    uint32_t totalN_ {0};
    uint32_t totalK_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(
    GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatMul op;
    op.Init(A, B, C,
            tiling_data.totalM, tiling_data.totalN, tiling_data.totalK,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
