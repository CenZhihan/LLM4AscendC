kernel_src="""

#include "kernel_operator.h"
#include "vector_cmp.h"
#include "matmul_api.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
                                uint32_t T, uint32_t B, uint32_t I, uint32_t H, uint32_t O)
    {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O; K_ = I + H;

        xGm_.SetGlobalBuffer((__gm__ float*)x, T_ * B_ * I_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, B_ * H_);
        // w_i2h (H, K) split into w_x (H, I) and w_h (H, H)
        w_xGm_.SetGlobalBuffer((__gm__ float*)w_i2h, H_ * I_);
        w_hGm_.SetGlobalBuffer((__gm__ float*)w_i2h + H_ * I_, H_ * H_);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, O_ * H_);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, T_ * B_ * O_);

        // Allocate local buffers: biases, and per-time-step buffers
        pipe_.InitBuffer(inBufBiasI2H_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(inBufBiasH2O_, BUFFER_NUM, O_ * sizeof(float));
        pipe_.InitBuffer(inBufX_, BUFFER_NUM, I_ * sizeof(float));
        pipe_.InitBuffer(inBufH_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(inBufPreA_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(inBufPreB_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(inBufPre_, BUFFER_NUM, H_ * sizeof(float));
        pipe_.InitBuffer(inBufYRaw_, BUFFER_NUM, O_ * sizeof(float));
        pipe_.InitBuffer(inBufY_, BUFFER_NUM, O_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (T_ == 0 || B_ == 0) return;

        const uint32_t blockNum = GetBlockNum();
        const uint32_t blockIdx = GetBlockIdx();

        // Load biases into local once
        LocalTensor<float> biasI2H = inBufBiasI2H_.AllocTensor<float>();
        LocalTensor<float> biasH2O = inBufBiasH2O_.AllocTensor<float>();
        DataCopy(biasI2H, b_i2hGm_, H_);
        DataCopy(biasH2O, b_h2oGm_, O_);
        // Keep biases in local for entire kernel (no need to enqueue, just use these tensors)
        // Since we only need them once, we keep allocated.

        // Grid-stride over batches
        for (uint32_t b = blockIdx; b < B_; b += blockNum) {
            // Load initial hidden state for this batch
            LocalTensor<float> hLocal = inBufH_.AllocTensor<float>();
            DataCopy(hLocal, h0Gm_[b * H_], H_);

            for (uint32_t t = 0; t < T_; ++t) {
                // Load x[t, b, :]
                LocalTensor<float> xLocal = inBufX_.AllocTensor<float>();
                DataCopy(xLocal, xGm_[(t * B_ + b) * I_], I_);

                // Compute preA = x * w_x^T
                LocalTensor<float> preA = inBufPreA_.AllocTensor<float>();
                MatMul<false, true>(preA, xLocal, w_xGm_, 1, H_, I_, 1);
                // Compute preB = h * w_h^T
                LocalTensor<float> preB = inBufPreB_.AllocTensor<float>();
                MatMul<false, true>(preB, hLocal, w_hGm_, 1, H_, H_, 1);
                // pre = preA + preB + bias
                LocalTensor<float> pre = inBufPre_.AllocTensor<float>();
                Add(pre, preA, preB, H_);
                Add(pre, pre, biasI2H, H_);

                // tanh pre -> new hidden (in-place)
                Tanh(pre, pre, H_);
                // hLocal = pre (new hidden)
                Copy(hLocal, pre, H_);

                // Compute y = h * w_h2o^T + bias_h2o
                LocalTensor<float> yRaw = inBufYRaw_.AllocTensor<float>();
                MatMul<false, true>(yRaw, hLocal, w_h2oGm_, 1, O_, H_, 1);
                LocalTensor<float> yLocal = inBufY_.AllocTensor<float>();
                Add(yLocal, yRaw, biasH2O, O_);

                // Store y[t, b, :]
                DataCopy(yGm_[(t * B_ + b) * O_], yLocal, O_);

                // Free local tensors for this time step
                inBufX_.FreeTensor(xLocal);
                inBufPreA_.FreeTensor(preA);
                inBufPreB_.FreeTensor(preB);
                inBufPre_.FreeTensor(pre);
                inBufYRaw_.FreeTensor(yRaw);
                inBufY_.FreeTensor(yLocal);
            }
            // Free h local for this batch
            inBufH_.FreeTensor(hLocal);
        }
        // Free bias tensors
        inBufBiasI2H_.FreeTensor(biasI2H);
        inBufBiasH2O_.FreeTensor(biasH2O);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufH_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> inBufPreA_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> inBufPreB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> inBufPre_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> inBufYRaw_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> inBufY_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufBiasI2H_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inBufBiasH2O_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> w_xGm_;
    AscendC::GlobalTensor<float> w_hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t T_{0}, B_{0}, I_{0}, H_{0}, O_{0}, K_{0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(
    GM_ADDR x, GM_ADDR h0, GM_ADDR w_i2h, GM_ADDR b_i2h,
    GM_ADDR w_h2o, GM_ADDR b_h2o, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I, tiling_data.H, tiling_data.O);
    op.Process();
}
