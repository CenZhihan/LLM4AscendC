host_operator_src = """
#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB budget per block: input tile (tileLength * reductionSize), output tile (tileLength), temp for sum (reductionSize)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;
    const gert::Shape &x_shape = context->GetInputShape(0)->GetOriginShape();
    uint32_t totalDims = x_shape.GetDimNum();

    // Compute total output length (product of all dims except last)
    uint32_t totalOutputLength = 1;
    uint32_t reductionSize = x_shape.GetDim(totalDims - 1);
    for (uint32_t i = 0; i < totalDims - 1; ++i)
        totalOutputLength *= x_shape.GetDim(i);

    if (totalOutputLength == 0 || reductionSize == 0) {
        // zero-length tensor, safe fallback
        tiling.set_totalOutputLength(0);
        tiling.set_tileLength(0);
        tiling.set_reductionSize(0);
        tiling.set_blockDim(1);
        context->SetBlockDim(1);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    // Tile over output elements, each tile needs to load reductionSize floats per output element
    uint32_t floatsPerOutput = reductionSize;
    uint32_t maxTileOutput = UB_BUDGET_BYTES / (sizeof(float) * (floatsPerOutput + 1)); // +1 for output
    maxTileOutput = AlignDown(maxTileOutput, ALIGN_ELEMS);
    if (maxTileOutput < ALIGN_ELEMS) maxTileOutput = ALIGN_ELEMS;
    if (maxTileOutput > HARD_CAP_ELEMS) maxTileOutput = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTileOutput;
    if (totalOutputLength < tileLen) {
        const uint32_t aligned = AlignDown(totalOutputLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutputLength;
    }

    // Block dimension: based on number of tiles, keep reasonable
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutputLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutputLength(totalOutputLength);
    tiling.set_tileLength(tileLen);
    tiling.set_reductionSize(reductionSize);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    uint32_t totalDims = x_shape->GetDimNum();
    if (totalDims == 0) {
        return GRAPH_FAILED;
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    (void)y_shape->SetDimNum(totalDims - 1);
    for (uint32_t i = 0; i < totalDims - 1; ++i) {
        y_shape->SetDim(i, x_shape->GetDim(i));
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MeanReductionOverADimensionCustom : public OpDef {
public:
    explicit MeanReductionOverADimensionCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(-1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MeanReductionOverADimensionCustom);
} // namespace ops

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReduction {
public:
    __aicore__ inline KernelMeanReduction() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputLength, uint32_t tileLength,
                                uint32_t reductionSize)
    {
        totalOutLen_  = totalOutputLength;
        tileLen_      = tileLength;
        redSize_      = reductionSize;
        invRedSize_   = 1.0f / static_cast<float>(reductionSize);

        // Input is sized totalOutLen_ * redSize_
        uint32_t totalInputLen = totalOutLen_ * redSize_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * redSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, redSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over output elements
        const uint32_t strideOut = blockNum * tileLen_;
        uint32_t baseOut = blockIdx * tileLen_;
        if (baseOut >= totalOutLen_) return;

        // Data for a tile: each output element corresponds to redSize_ input floats
        // We load tileLen_ * redSize_ contiguous input for the tile.
        while (true) {
            uint32_t curOut = totalOutLen_ - baseOut;
            if (curOut > tileLen_) curOut = tileLen_;

            // Prefetch next tile
            uint32_t nextBase = baseOut + strideOut;
            uint32_t nextOut = 0;
            if (nextBase < totalOutLen_) {
                nextOut = totalOutLen_ - nextBase;
                if (nextOut > tileLen_) nextOut = tileLen_;
                CopyIn(nextBase * redSize_, nextOut * redSize_);
            }

            // Process current tile
            Compute(curOut);
            CopyOut(baseOut, curOut);

            if (nextOut == 0) break;
            baseOut = nextBase;
            // The next tile's input was already prefetched; on next iteration we compute it.
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t numOut)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // For each output element, sum over its reduction segment
        // xLocal is laid out as [numOut][redSize_]
        const uint32_t segSize = redSize_;

        // Use vector reduction patterns:
        // We can compute sum for first output element, then slide.
        // Since redSize_ can be up to 4096, we can use vector accumulation.
        // Approach: for each output element, load its segment (segment is contiguous) and sum.
        // We'll do loop over numOut, but with vectorized load on each segment.
        // Optimize: use vadd with accumulator across segments? Alternatively, use a loop that accumulates using vector for the whole segment.
        for (uint32_t i = 0; i < numOut; ++i) {
            AscendC::LocalTensor<float> src = xLocal[i * segSize]; // treat as offset
            AscendC::LocalTensor<float> sumLocal = yLocal[i]; // output scalar (as 1-element tensor)
            AscendC::VecSum(sumLocal, src, segSize); // sum of vector
            // Multiply by 1/redSize_ to get mean
            AscendC::Muls(sumLocal, sumLocal, invRedSize_, 1);
        }

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t redSize_ {0};
    float invRedSize_ {0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReduction op;
    op.Init(x, y, tiling_data.totalOutputLength, tiling_data.tileLength, tiling_data.reductionSize);
    op.Process();
}

python_bind_src = """
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor mean_reduction_impl_npu(const at::Tensor& self, int64_t dim) {
    // The custom operator always reduces over the last dimension.
    // Permute the input so that the desired dimension becomes last.
    at::Tensor x_perm = self.transpose(dim, -1);
    at::Tensor result = at::empty_like(x_perm, at::TensorOptions().dtype(self.dtype()).device(self.device()));
    // Remove the last dimension from result shape: we set empty_like with reduced dim? Use empty as target shape.
    // Actually the custom op will output a tensor with last dim removed.
    // For simplicity, let the op allocate output.
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, x_perm, -1, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("mean_reduction_over_a_dimension_custom", &mean_reduction_impl_npu, "Mean reduction over a dimension (MeanReductionOverADimensionCustom on NPU)");
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'mean_reduction_over_a_dimension_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, dim):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x):
        # Permute so the reduction dimension is last, then call custom kernel.
        fn = getattr(custom_ops_lib, _FN)
        # The custom kernel reduces over the last dimension.
        x_perm = x.transpose(self.dim, -1)
        y = fn(x_perm)
        # y now has shape same as x_perm but with last dimension removed.
        # No further reshape needed because the original dim is removed.
        return y
'''
