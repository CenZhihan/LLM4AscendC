#include <cmath>
#include "max_pooling3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t HARD_BLOCKS = 24;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;

    const gert::Shape &x_origin = context->GetInputShape(0)->GetOriginShape();
    uint32_t batch     = static_cast<uint32_t>(x_origin.GetDim(0));
    uint32_t channels  = static_cast<uint32_t>(x_origin.GetDim(1));
    uint32_t inputD    = static_cast<uint32_t>(x_origin.GetDim(2));
    uint32_t inputH    = static_cast<uint32_t>(x_origin.GetDim(3));
    uint32_t inputW    = static_cast<uint32_t>(x_origin.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t kernelSize = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(0)[0]);
    uint32_t stride     = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(1)[0]);
    uint32_t padding    = static_cast<uint32_t>(attrs->GetAttrPointer<int64_t>(2)[0]);
    bool ceilMode       = attrs->GetAttrPointer<bool>(3)[0];

    uint32_t outputD, outputH, outputW;
    auto calc_out = [&](uint32_t in, uint32_t pad, uint32_t k, uint32_t s) {
        if (ceilMode)
            return static_cast<uint32_t>(std::ceil((in + 2.0 * pad - (k - 1.0) - 1.0) / s + 1.0));
        else
            return static_cast<uint32_t>(std::floor((in + 2.0 * pad - (k - 1.0) - 1.0) / s + 1.0));
    };
    outputD = calc_out(inputD, padding, kernelSize, stride);
    outputH = calc_out(inputH, padding, kernelSize, stride);
    outputW = calc_out(inputW, padding, kernelSize, stride);

    uint32_t totalOutputElems = batch * channels * outputD * outputH * outputW;
    uint32_t totalInputElems = batch * channels * inputD * inputH * inputW;

    uint32_t blockDim = (totalOutputElems < HARD_BLOCKS) ? totalOutputElems : HARD_BLOCKS;
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_totalInputElems(totalInputElems);
    tiling.set_totalOutputElems(totalOutputElems);
    tiling.set_batch(batch);
    tiling.set_channels(channels);
    tiling.set_inputD(inputD);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_outputD(outputD);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_ceilMode(ceilMode);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                const MaxPooling3dCustomTilingData &tiling)
    {
        xGm_ = (__gm__ float*)x;
        yGm_ = (__gm__ float*)y;
        batch_ = tiling.batch;
        channels_ = tiling.channels;
        inputD_ = tiling.inputD;
        inputH_ = tiling.inputH;
        inputW_ = tiling.inputW;
        outputD_ = tiling.outputD;
        outputH_ = tiling.outputH;
        outputW_ = tiling.outputW;
        kernelSize_ = tiling.kernelSize;
        stride_ = tiling.stride;
        padding_ = tiling.padding;
        totalOutputElems_ = tiling.totalOutputElems;
        blockNum_ = tiling.blockDim;
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0) return;
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t idx = blockIdx; idx < totalOutputElems_; idx += blockNum_) {
            uint32_t tmp = idx;
            uint32_t ow = tmp % outputW_; tmp /= outputW_;
            uint32_t oh = tmp % outputH_; tmp /= outputH_;
            uint32_t od = tmp % outputD_; tmp /= outputD_;
            uint32_t c  = tmp % channels_; tmp /= channels_;
            uint32_t n  = tmp;

            int32_t id_start = static_cast<int32_t>(od) * static_cast<int32_t>(stride_)
                              - static_cast<int32_t>(padding_);
            int32_t ih_start = static_cast<int32_t>(oh) * static_cast<int32_t>(stride_)
                              - static_cast<int32_t>(padding_);
            int32_t iw_start = static_cast<int32_t>(ow) * static_cast<int32_t>(stride_)
                              - static_cast<int32_t>(padding_);

            float maxVal = -1e38f;
            for (uint32_t kd = 0; kd < kernelSize_; ++kd) {
                int32_t id = id_start + static_cast<int32_t>(kd);
                if (id < 0 || id >= static_cast<int32_t>(inputD_)) continue;
                for (uint32_t kh = 0; kh < kernelSize_; ++kh) {
                    int32_t ih = ih_start + static_cast<int32_t>(kh);
                    if (ih < 0 || ih >= static_cast<int32_t>(inputH_)) continue;
                    for (uint32_t kw = 0; kw < kernelSize_; ++kw) {
                        int32_t iw = iw_start + static_cast<int32_t>(kw);
                        if (iw < 0 || iw >= static_cast<int32_t>(inputW_)) continue;

                        uint32_t inIdx = ((n * channels_ + c) * inputD_ + static_cast<uint32_t>(id)) * inputH_ * inputW_
                                         + static_cast<uint32_t>(ih) * inputW_ + static_cast<uint32_t>(iw);
                        float val = xGm_[inIdx];
                        if (val > maxVal) maxVal = val;
                    }
                }
            }
            yGm_[idx] = maxVal;
        }
    }

private:
    __gm__ float *xGm_;
    __gm__ float *yGm_;
    uint32_t batch_;
    uint32_t channels_;
    uint32_t inputD_, inputH_, inputW_;
    uint32_t outputD_, outputH_, outputW_;
    uint32_t kernelSize_;
    uint32_t stride_;
    uint32_t padding_;
    uint32_t totalOutputElems_;
    uint32_t blockNum_;
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y, tiling_data);
    op.Process();
}

#include <torch/library.h>

at::Tensor max_pooling_3d_impl_npu(const at::Tensor& self,
                                   int64_t kernel_size,
                                   int64_t stride,
                                   int64_t padding,
                                   bool ceil_mode = false) {
    at::Tensor result = at::empty_like(self); // shape will be computed by ACL
    EXEC_NPU_CMD(aclnnMaxPooling3dCustom, self, kernel_size, stride, padding, ceil_mode, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_3d_custom", &max_pooling_3d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_3d_custom", &max_pooling_3d_impl_npu, "MaxPooling3D (MaxPooling3dCustom on NPU)");

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size=3, stride=2, padding=1, ceil_mode=False):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.ceil_mode = ceil_mode

    def forward(self, x):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding, self.ceil_mode)
