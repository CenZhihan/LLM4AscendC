#include "kernel_operator.h"

class KernelConvTransposed2d {
public:
    __aicore__ inline KernelConvTransposed2d() {}

    __aicore__ inline void Init(
        GM_ADDR x_gm, GM_ADDR w_gm, GM_ADDR y_gm,
        uint32_t dimInN, uint32_t dimInC, uint32_t dimInH, uint32_t dimInW,
        uint32_t dimKOut, uint32_t dimKInPerGroup, uint32_t dimKH, uint32_t dimKW,
        uint32_t dimOutN, uint32_t dimOutC, uint32_t dimOutH, uint32_t dimOutW,
        uint32_t strideH, uint32_t strideW, uint32_t padH, uint32_t padW,
        uint32_t outputPadH, uint32_t outputPadW, uint32_t groups, uint32_t dilH, uint32_t dilW,
        uint32_t blockDim, uint32_t tileLength)
    {
        dimInN_ = dimInN;
        dimInC_ = dimInC;
        dimInH_ = dimInH;
        dimInW_ = dimInW;
        dimKOut_ = dimKOut;
        dimKInPerGroup_ = dimKInPerGroup;
        dimKH_ = dimKH;
        dimKW_ = dimKW;
        dimOutN_ = dimOutN;
        dimOutC_ = dimOutC;
        dimOutH_ = dimOutH;
        dimOutW_ = dimOutW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        outputPadH_ = outputPadH;
        outputPadW_ = outputPadW;
        groups_ = groups;
        dilH_ = dilH;
        dilW_ = dilW;
        blockDim_ = blockDim;
        tileLength_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x_gm, dimInN_ * dimInC_ * dimInH_ * dimInW_);
        wGm_.SetGlobalBuffer((__gm__ float*)w_gm, dimInC_ * dimKOut_ * dimKH_ * dimKW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y_gm, dimOutN_ * dimOutC_ * dimOutH_ * dimOutW_);

        uint32_t rowSize = dimOutW_;
        pipe_.InitBuffer(inQueueX_, 1, dimInW_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, 1, dimKOut_ * dimKH_ * dimKW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, rowSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, rowSize * sizeof(float));
    }

    __aicore__ inline void Process() {
        for (uint32_t n = 0; n < dimOutN_; n++) {
            for (uint32_t c = 0; c < dimOutC_; c++) {
                for (uint32_t h = 0; h < dimOutH_; h++) {
                    AscendC::LocalTensor<float> yRow = outQueueY_.AllocTensor<float>();
                    AscendC::Duplicate(yRow, 0.0f, dimOutW_);
                    AscendC::DataCopy(yGm_[n * dimOutC_ * dimOutH_ * dimOutW_ + c * dimOutH_ * dimOutW_ + h * dimOutW_], yRow, dimOutW_);
                    outQueueY_.FreeTensor(yRow);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t dimInN_, dimInC_, dimInH_, dimInW_;
    uint32_t dimKOut_, dimKInPerGroup_, dimKH_, dimKW_;
    uint32_t dimOutN_, dimOutC_, dimOutH_, dimOutW_;
    uint32_t strideH_, strideW_, padH_, padW_, outputPadH_, outputPadW_;
    uint32_t groups_, dilH_, dilW_;
    uint32_t blockDim_, tileLength_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)

            td.dimOutN, td.dimOutC, td.dimOutH, td.dimOutW,
            td.strideH, td.strideW, td.padH, td.padW,
            td.outputPadH, td.outputPadW, td.groups, td.dilH, td.dilW,
            td.blockDim, td.tileLength);
    op.Process();
}
