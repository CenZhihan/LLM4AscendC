kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t IN_TILE = 2048;  // tile size for in_features loop

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
                                GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures,
                                uint32_t outFeatures, uint32_t numGroups,
                                float eps, uint32_t tileLen)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        numGroups_ = numGroups;
        eps_ = eps;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inFeatures);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures * inFeatures);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * outFeatures);

        // UB buffers for entire output tile (outFeatures)
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue3_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || inFeatures_ == 0 || outFeatures_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= batchSize_) return;
        uint32_t sampleIdx = blockIdx;

        // Output tensor in UB
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        const uint32_t n = outFeatures_;

        // Step 1: Linear(x, weight, lin_bias) and add bias
        for (uint32_t i = 0; i < n; i++) {
            float sum = linBiasGm_.GetValue(i) + biasGm_.GetValue(i);
            // dot product with weight row i
            uint32_t baseW = i * inFeatures_;
            for (uint32_t j = 0; j < inFeatures_; j += IN_TILE) {
                uint32_t len = (j + IN_TILE <= inFeatures_) ? IN_TILE : (inFeatures_ - j);
                AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> wTile = tmpQueue1_.AllocTensor<float>();
                // load x[sampleIdx * inFeatures + j] and w[baseW + j]
                AscendC::DataCopy(xTile, xGm_[sampleIdx * inFeatures_ + j], len);
                AscendC::DataCopy(wTile, wGm_[baseW + j], len);
                // accumulate: sum += dot(xTile, wTile)
                AscendC::LocalTensor<float> prod = tmpQueue2_.AllocTensor<float>();
                AscendC::Mul(prod, xTile, wTile, len);
                AscendC::LocalTensor<float> sumTensor = tmpQueue3_.AllocTensor<float>();
                AscendC::ReduceAdd(sumTensor, prod, len);
                sum += sumTensor.GetValue(0);
                inQueueX_.FreeTensor(xTile);
                tmpQueue1_.FreeTensor(wTile);
                tmpQueue2_.FreeTensor(prod);
                tmpQueue3_.FreeTensor(sumTensor);
            }
            yLocal.SetValue(i, sum);
        }

        // Step 2: Hardtanh (clamp to [-1, 1])
        AscendC::LocalTensor<float> t = tmpQueue1_.AllocTensor<float>();
        AscendC::Relu(t, yLocal, n);               // max(x, 0) -> later we need clamp
        // Actually we need min(max(x, -1), 1). Use Relu on x+1 then subtract 1? Simpler:
        AscendC::Adds(t, yLocal, 1.0f, n);
        AscendC::Relu(t, t, n);
        AscendC::Adds(t, t, -2.0f, n);
        AscendC::Relu(t, t, n);
        AscendC::Adds(yLocal, t, -1.0f, n);  // now y = clamp(orig, -1, 1)

        // Step 3: Mish = x * tanh(softplus(x))
        AscendC::LocalTensor<float> softplus = tmpQueue2_.AllocTensor<float>();
        AscendC::Exp(softplus, yLocal, n);           // exp(x)
        AscendC::Adds(softplus, softplus, 1.0f, n);  // 1+exp(x)
        AscendC::Log(softplus, softplus, n);         // log(1+exp(x)) = softplus
        AscendC::LocalTensor<float> tanhSoftplus = tmpQueue3_.AllocTensor<float>();
        AscendC::Tanh(tanhSoftplus, softplus, n);
        AscendC::Mul(yLocal, yLocal, tanhSoftplus, n);  // mish = x * tanh(softplus)
        tmpQueue1_.FreeTensor(t);
        tmpQueue2_.FreeTensor(softplus);
        tmpQueue3_.FreeTensor(tanhSoftplus);

        // Step 4: Group Normalization
        uint32_t groups = numGroups_;
        uint32_t channelsPerGroup = n / groups;
        // Compute per-group mean and variance from yLocal, then normalize
        for (uint32_t g = 0; g < groups; g++) {
            uint32_t start = g * channelsPerGroup;
            // mean
            float sum = 0.0f;
            for (uint32_t c = 0; c < channelsPerGroup; c++) {
                sum += yLocal.GetValue(start + c);
            }
            float mean = sum / static_cast<float>(channelsPerGroup);
            // variance
            float sqSum = 0.0f;
            for (uint32_t c = 0; c < channelsPerGroup; c++) {
                float diff = yLocal.GetValue(start + c) - mean;
                sqSum += diff * diff;
            }
            float var = sqSum / static_cast<float>(channelsPerGroup);
            float invStd = 1.0f / AscendC::Sqrt(var + eps_);
            // normalize
            for (uint32_t c = 0; c < channelsPerGroup; c++) {
                float val = (yLocal.GetValue(start + c) - mean) * invStd;
                val = val * gnGammaGm_.GetValue(start + c) + gnBetaGm_.GetValue(start + c);
                yLocal.SetValue(start + c, val);
            }
        }

        // Copy output to global memory
        AscendC::DataCopy(yGm_[sampleIdx * outFeatures_], yLocal, n);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue3_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t inFeatures_ {0};
    uint32_t outFeatures_ {0};
    uint32_t numGroups_ {1};
    float eps_ {1e-5f};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias,
                                                                  GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                                                  GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.batchSize, tiling_data.inFeatures,
            tiling_data.outFeatures, tiling_data.numGroups,
            tiling_data.eps, tiling_data.tileLen);
    op.Process();
}
