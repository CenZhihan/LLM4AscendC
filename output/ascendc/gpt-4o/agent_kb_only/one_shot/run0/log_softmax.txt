host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model:
// Maximum budget: tileSize, total floats = 3 * tileSize => bytes = 12 * tileSize
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
static constexpr uint32_t ALIGN_SIZE = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

ge::graphStatus LogSoftmaxTilingFunc(gert::TilingContext* context)
{
    LogSoftmaxCustomTilingData tilingData;
    const uint32_t inputSize = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t reduceSize = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDimSize(context->GetAttrInt(0)));

    uint32_t maxTileSize = UB_BUDGET_BYTES / 12; // bytes for floats
    maxTileSize = AlignDown(maxTileSize, ALIGN_SIZE);

    if (maxTileSize > reduceSize) {
        maxTileSize = AlignDown(reduceSize, ALIGN_SIZE);
    }

    uint32_t tileSize = maxTileSize;
    uint32_t numTiles = CeilDiv(inputSize, tileSize);
    uint32_t blockDim = numTiles;

    blockDim = (blockDim > 24) ? 24 : blockDim;

    context->SetBlockDim(blockDim);

    tilingData.set_totalSize(inputSize);
    tilingData.set_reduceSize(reduceSize);
    tilingData.set_tileSize(tileSize);
    tilingData.set_dim(context->GetAttrInt(0));

    tilingData.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tilingData.GetDataSize());

    context->GetWorkspaceSizes(1)[0] = 0;

    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    context->GetOutputShape(0)->CopyFrom(*x_shape);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::LogSoftmaxTilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR xGm, GM_ADDR yGm, uint32_t totalSize, uint32_t tileSize,
                                uint32_t reduceSize, uint32_t dim)
    {
        totalSize_ = totalSize;
        tileSize_ = tileSize;
        reduceSize_ = reduceSize;
        dim_ = dim;

        xGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(xGm), totalSize_);
        yGlobal_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(yGm), totalSize_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileSize_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalSize_ == 0 || tileSize_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t strideElems = blockNum * tileSize_;
        uint32_t base = blockIdx * tileSize_;
        if (base >= totalSize_) return;

        uint32_t currentSize = (totalSize_ - base < tileSize_) ? totalSize_ - base : tileSize_;

        CopyIn(base, currentSize);

        while (true) {
            uint32_t nextBase = base + strideElems;
            if (nextBase >= totalSize_) break;

            uint32_t nextSize = (totalSize_ - nextBase < tileSize_) ? totalSize_ - nextBase : tileSize_;
            CopyIn(nextBase, nextSize);

            Compute(currentSize, reduceSize_);
            CopyOut(base, currentSize);
            base = nextBase;
            currentSize = nextSize;
        }

        Compute(currentSize, reduceSize_);
        CopyOut(base, currentSize);
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> localX = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(localX, xGlobal_[gmOffset], len);
        inQueue_.EnQue(localX);
    }

    __aicore__ inline void Compute(uint32_t len, uint32_t reduceSize)
    {
        AscendC::LocalTensor<float> localX = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> localY = outQueue_.AllocTensor<float>();

        AscendC::LogSoftMax(localY, localX, reduceSize, len, dim_);

        outQueue_.EnQue(localY);
        inQueue_.FreeTensor(localX);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> localY = outQueue_.DeQue<float>();
        AscendC::DataCopy(yGlobal_[gmOffset], localY, len);
        outQueue_.FreeTensor(localY);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGlobal_;
    AscendC::GlobalTensor<float> yGlobal_;

    uint32_t totalSize_{0};
    uint32_t tileSize_{0};
    uint32_t reduceSize_{0};
    uint32_t dim_{1};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, tilingData.totalSize, tilingData.tileSize, tilingData.reduceSize, tilingData.dim);
    op.Process();
}
