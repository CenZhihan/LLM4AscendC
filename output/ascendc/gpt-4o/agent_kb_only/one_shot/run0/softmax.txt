#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_WIDTH = 32;
static constexpr uint32_t HARD_CAP_WIDTH = 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    SoftmaxCustomTilingData tiling;
    const uint32_t totalSize =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const uint32_t batchSize = context->GetInputShape(0)->GetShape(0);
    const uint32_t dimSize = context->GetInputShape(0)->GetShape(1);

    uint32_t width = dimSize;
    if (width > HARD_CAP_WIDTH) width = HARD_CAP_WIDTH;

    uint32_t alignedWidth = AlignDown(width, ALIGN_WIDTH);
    uint32_t blockNum = CeilDiv(batchSize * dimSize, alignedWidth);

    context->SetBlockDim(blockNum);
    tiling.set_height(batchSize);
    tiling.set_width(alignedWidth);
    tiling.set_blockNum(blockNum);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t height, uint32_t width)
    {
        height_ = height;
        width_ = width;
        xGm_.SetGlobalBuffer((__gm__ float*)x, height_ * width_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, height_ * width_);

        pipe_.InitBuffer(inQueueX_, 1, width_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, width_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, width_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t h = 0; h < height_; ++h) {
            CopyIn(h);
            Compute();
            CopyOut(h);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t h)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[h * width_], width_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

        // Compute softmax
        AscendC::Softmax<float>(yLocal, xLocal, tmpLocal, width_);

        tmpQueue_.FreeTensor(tmpLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t h)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[h * width_], yLocal, width_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t height_ {0};
    uint32_t width_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.height, tiling_data.width);
    op.Process();
}
"""

at::Tensor softmax_impl_npu(const at::Tensor& self, int64_t dim) {
    TORCH_CHECK(dim == 1, "Only dim=1 is supported for SoftmaxCustom");
    at::Tensor result = at::empty_like(self);
    EXEC_NPU_CMD(aclnnSoftmaxCustom, self, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("softmax_custom", &softmax_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("softmax_custom", &softmax_impl_npu, "Softmax activation (SoftmaxCustom on NPU)");
