#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvTransposed3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t xLen, uint32_t wLen, uint32_t yLen,
                                uint32_t totalOutLength, uint32_t tileLength,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Di, uint32_t Hi, uint32_t Wi,
                                uint32_t Do, uint32_t Ho, uint32_t Wo,
                                uint32_t kD, uint32_t kH, uint32_t kW)
    {
        xLen_ = xLen; wLen_ = wLen; yLen_ = yLen;
        totalOutLen_ = totalOutLength;
        tileLen_ = (tileLength == 0u) ? 1u : tileLength;

        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Di_ = Di; Hi_ = Hi; Wi_ = Wi;
        Do_ = Do; Ho_ = Ho; Wo_ = Wo;
        kD_ = kD; kH_ = kH; kW_ = kW;

        // Precompute strides
        i_sW_ = 1u;
        i_sH_ = Wi_;
        i_sD_ = Hi_ * Wi_;
        i_sC_ = Di_ * i_sD_;
        i_sN_ = Cin_ * i_sC_;

        o_sW_ = 1u;
        o_sH_ = Wo_;
        o_sD_ = Ho_ * o_sH_;
        o_sC_ = Do_ * o_sD_;
        o_sN_ = Cout_ * o_sC_;

        w_sW_ = 1u;
        w_sH_ = kW_;
        w_sD_ = kH_ * w_sH_;
        w_sCout_ = kD_ * w_sD_;
        w_sCin_  = Cout_ * w_sCout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yLen_);

        pipe_.InitBuffer(calcQueue_, 4, 4 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0u) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutLen_) return;

        uint32_t curLen = totalOutLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        while (true) {
            ComputeRange(base, curLen);

            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalOutLen_) break;
            base = nextBase;
            curLen = totalOutLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
        }
    }

private:
    __aicore__ inline void LinearToCoord(uint32_t lin,
                                         uint32_t &n, uint32_t &oc,
                                         uint32_t &od, uint32_t &oh, uint32_t &ow) const
    {
        uint32_t t = lin;
        ow = t % Wo_; t /= Wo_;
        oh = t % Ho_; t /= Ho_;
        od = t % Do_; t /= Do_;
        oc = t % Cout_; t /= Cout_;
        n  = t;
    }

    __aicore__ inline void ComputeRange(uint32_t base, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            const uint32_t outLin = base + i;

            uint32_t n, oc, od, oh, ow;
            LinearToCoord(outLin, n, oc, od, oh, ow);

            // Compute output GM offset
            const uint32_t yOff = n * o_sN_ + oc * o_sC_ + od * o_sD_ + oh * o_sH_ + ow;

            // acc = 0
            AscendC::LocalTensor<float> acc = calcQueue_.AllocTensor<float>();
            AscendC::Duplicate(acc, 0.0f, 1);

            // y[n, oc, od, oh, ow] = sum_ic sum_kd,kh,kw x[n, ic, od - kd, oh - kh, ow - kw] * w[ic, oc, kd, kh, kw]
            for (uint32_t ic = 0; ic < Cin_; ++ic) {
                for (uint32_t kd = 0; kd < kD_; ++kd) {
                    int32_t iz = static_cast<int32_t>(od) - static_cast<int32_t>(kd);
                    if (iz < 0 || iz >= static_cast<int32_t>(Di_)) continue;

                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        int32_t iy = static_cast<int32_t>(oh) - static_cast<int32_t>(kh);
                        if (iy < 0 || iy >= static_cast<int32_t>(Hi_)) continue;

                        for (uint32_t kw = 0; kw < kW_; ++kw) {
                            int32_t ix = static_cast<int32_t>(ow) - static_cast<int32_t>(kw);
                            if (ix < 0 || ix >= static_cast<int32_t>(Wi_)) continue;

                            const uint32_t xOff = n * i_sN_ + ic * i_sC_
                                                + static_cast<uint32_t>(iz) * i_sD_
                                                + static_cast<uint32_t>(iy) * i_sH_
                                                + static_cast<uint32_t>(ix);

                            const uint32_t wOff = ic * w_sCin_ + oc * w_sCout_
                                                + kd * w_sD_ + kh * w_sH_ + kw;

                            AscendC::LocalTensor<float> xLoc = calcQueue_.AllocTensor<float>();
                            AscendC::LocalTensor<float> wLoc = calcQueue_.AllocTensor<float>();
                            AscendC::LocalTensor<float> prod = calcQueue_.AllocTensor<float>();

                            AscendC::DataCopy(xLoc, xGm_[xOff], 1);
                            AscendC::DataCopy(wLoc, wGm_[wOff], 1);
                            AscendC::Mul(prod, xLoc, wLoc, 1);
                            AscendC::Add(acc, acc, prod, 1);

                            calcQueue_.FreeTensor(prod);
                            calcQueue_.FreeTensor(wLoc);
                            calcQueue_.FreeTensor(xLoc);
                        }
                    }
                }
            }

            // Write out
            AscendC::DataCopy(yGm_[yOff], acc, 1);
            calcQueue_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> calcQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t xLen_{0}, wLen_{0}, yLen_{0};
    uint32_t totalOutLen_{0}, tileLen_{1};

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Di_{0}, Hi_{0}, Wi_{0};
    uint32_t Do_{0}, Ho_{0}, Wo_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};

    // Strides
    uint32_t i_sW_{1}, i_sH_{0}, i_sD_{0}, i_sC_{0}, i_sN_{0};
    uint32_t o_sW_{1}, o_sH_{0}, o_sD_{0}, o_sC_{0}, o_sN_{0};
    uint32_t w_sW_{1}, w_sH_{0}, w_sD_{0}, w_sCout_{0}, w_sCin_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.xLength, tiling_data.wLength, tiling_data.yLength,
            tiling_data.totalOutLength, tiling_data.tileLength,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.Di, tiling_data.Hi, tiling_data.Wi,
            tiling_data.Do, tiling_data.Ho, tiling_data.Wo,
            tiling_data.kD, tiling_data.kH, tiling_data.kW);
    op.Process();
}
