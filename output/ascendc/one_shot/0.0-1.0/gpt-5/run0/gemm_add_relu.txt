#include "gemm_add_relu_custom_tiling.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// Target a conservative per-core unified buffer budget (bytes)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// Favor vector-friendly alignment (elements)
static constexpr uint32_t ALIGN_ELEMS = 16;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    const gert::Shape* b_shape = context->GetInputShape(2);

    // Shapes:
    // x: [M, K]
    // w: [N, K]   (PyTorch linear expects weight [N, K])
    // bias: [N]
    const uint64_t M64 = (x_shape->GetOriginShape().GetShapeSize() == 0) ? 0 :
                         static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(0));
    const uint64_t Kx64 = (x_shape->GetOriginShape().GetShapeSize() < 2) ? 0 :
                          static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(1));
    const uint64_t N64 = (w_shape->GetOriginShape().GetShapeSize() == 0) ? 0 :
                         static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(0));
    const uint64_t Kw64 = (w_shape->GetOriginShape().GetShapeSize() < 2) ? 0 :
                          static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(1));

    if (Kx64 != Kw64) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t M = static_cast<uint32_t>(M64);
    const uint32_t N = static_cast<uint32_t>(N64);
    const uint32_t K = static_cast<uint32_t>(Kx64);

    // Simple heuristic tiling under UB budget.
    // UB usage in floats approximately:
    //   X_tile (MT*KT) + W_tile (NT*KT) + Y_tile (MT*NT) + WT_tile (NT*KT) + bias (NT)
    // = MT*KT + 2*NT*KT + MT*NT + NT
    // bytes = 4 * that
    auto Fits = [](uint32_t MT, uint32_t NT, uint32_t KT) -> uint64_t {
        uint64_t f = static_cast<uint64_t>(MT) * KT
                   + 2ull * static_cast<uint64_t>(NT) * KT
                   + static_cast<uint64_t>(MT) * NT
                   + NT;
        return 4ull * f;
    };

    uint32_t KT = (K >= 256u) ? 256u : AlignDown(std::max(ALIGN_ELEMS, K), ALIGN_ELEMS);
    KT = std::min(KT, K == 0 ? 0u : AlignDown(K, ALIGN_ELEMS));

    if (KT == 0 && K > 0) KT = AlignDown(K, 1u);

    uint32_t NT = (N >= 128u) ? 128u : AlignDown(std::max(ALIGN_ELEMS, N), ALIGN_ELEMS);
    uint32_t MT = (M >= 32u) ? 32u : AlignDown(std::max(ALIGN_ELEMS, M), ALIGN_ELEMS);

    if (MT == 0 && M > 0) MT = 1u;
    if (NT == 0 && N > 0) NT = 1u;
    if (KT == 0 && K > 0) KT = 1u;

    // Tighten to fit UB
    while ((Fits(MT, NT, KT) > UB_BUDGET_BYTES) && (NT > ALIGN_ELEMS)) {
        NT = AlignDown(NT / 2u, ALIGN_ELEMS);
        if (NT == 0) NT = ALIGN_ELEMS;
    }
    while ((Fits(MT, NT, KT) > UB_BUDGET_BYTES) && (MT > ALIGN_ELEMS)) {
        MT = AlignDown(MT / 2u, ALIGN_ELEMS);
        if (MT == 0) MT = ALIGN_ELEMS;
    }
    while ((Fits(MT, NT, KT) > UB_BUDGET_BYTES) && (KT > ALIGN_ELEMS)) {
        KT = AlignDown(KT / 2u, ALIGN_ELEMS);
        if (KT == 0) KT = ALIGN_ELEMS;
    }
    // Clamp to problem dims
    if (MT > M) MT = (M >= ALIGN_ELEMS) ? AlignDown(M, ALIGN_ELEMS) : M;
    if (NT > N) NT = (N >= ALIGN_ELEMS) ? AlignDown(N, ALIGN_ELEMS) : N;
    if (KT > K) KT = (K >= ALIGN_ELEMS) ? AlignDown(K, ALIGN_ELEMS) : K;

    if (MT == 0) MT = std::min<uint32_t>(M, ALIGN_ELEMS);
    if (NT == 0) NT = std::min<uint32_t>(N, ALIGN_ELEMS);
    if (KT == 0) KT = std::min<uint32_t>(K, ALIGN_ELEMS);

    const uint32_t tilesM = (MT == 0) ? 1u : CeilDiv(M, MT);
    const uint32_t tilesN = (NT == 0) ? 1u : CeilDiv(N, NT);
    uint32_t totalTiles = tilesM * tilesN;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    GemmAddReluCustomTilingData tiling;
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_MT(MT);
    tiling.set_NT(NT);
    tiling.set_KT(KT);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace_sizes = context->GetWorkspaceSizes(1);
    workspace_sizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmAddReluCustom {
public:
    __aicore__ inline KernelGemmAddReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t MT, uint32_t NT, uint32_t KT)
    {
        M_ = M; N_ = N; K_ = K;
        MT_ = MT; NT_ = NT; KT_ = KT;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * N_);

        // Allocate pipeline buffers. Sizes are in elements.
        // X tile: MT * KT
        // W tile: NT * KT
        // WT tile (transposed): NT * KT
        // Y tile: MT * NT
        pipe_.InitBuffer(xTileQ_, 1, MT_ * KT_ * sizeof(float));
        pipe_.InitBuffer(wTileQ_, 1, NT_ * KT_ * sizeof(float));
        pipe_.InitBuffer(wTileTQ_, 1, NT_ * KT_ * sizeof(float));
        pipe_.InitBuffer(yTileQ_, 1, MT_ * NT_ * sizeof(float));
        pipe_.InitBuffer(biasQ_, 1, NT_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, NT_ * sizeof(float)); // for column adds
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t tilesM = (MT_ == 0) ? 1u : (uint32_t)((M_ + MT_ - 1u) / MT_);
        const uint32_t tilesN = (NT_ == 0) ? 1u : (uint32_t)((N_ + NT_ - 1u) / NT_);
        const uint32_t totalTiles = tilesM * tilesN;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t tm = tileId / tilesN;
            const uint32_t tn = tileId % tilesN;
            const uint32_t mBase = tm * MT_;
            const uint32_t nBase = tn * NT_;
            const uint32_t mLen = MinU(MT_, M_ - mBase);
            const uint32_t nLen = MinU(NT_, N_ - nBase);

            ComputeTile(mBase, nBase, mLen, nLen);
        }
    }

private:
    __aicore__ inline uint32_t MinU(uint32_t a, uint32_t b) { return (a < b) ? a : b; }

    __aicore__ inline void LoadBias(uint32_t nBase, uint32_t nLen, AscendC::LocalTensor<float>& bLocal)
    {
        AscendC::DataCopy(bLocal, bGm_[nBase], nLen);
    }

    __aicore__ inline void LoadX(uint32_t mBase, uint32_t kBase, uint32_t mLen, uint32_t kLen,
                                 AscendC::LocalTensor<float>& xLocal)
    {
        // Row-wise copy of submatrix X[mBase : mBase+mLen, kBase : kBase+kLen]
        for (uint32_t r = 0; r < mLen; ++r) {
            AscendC::DataCopy(xLocal[r * KT_], xGm_[(mBase + r) * K_ + kBase], kLen);
        }
    }

    __aicore__ inline void LoadW(uint32_t nBase, uint32_t kBase, uint32_t nLen, uint32_t kLen,
                                 AscendC::LocalTensor<float>& wLocal)
    {
        // Row-wise copy of submatrix W[nBase : nBase+nLen, kBase : kBase+kLen]
        for (uint32_t r = 0; r < nLen; ++r) {
            AscendC::DataCopy(wLocal[r * KT_], wGm_[(nBase + r) * K_ + kBase], kLen);
        }
    }

    __aicore__ inline void TransposeW(AscendC::LocalTensor<float>& wLocal,
                                      AscendC::LocalTensor<float>& wTLocal,
                                      uint32_t nLen, uint32_t kLen)
    {
        // Transpose wLocal[nLen x kLen] -> wTLocal[kLen x nLen]
        // Both are stored row-major with their respective leading dims KT_ and NT_.
        for (uint32_t r = 0; r < nLen; ++r) {
            // source row pointer: wLocal[r * KT_]
            // destination column r in wTLocal: write into wTLocal[k * NT_ + r]
            for (uint32_t c = 0; c < kLen; ++c) {
                // Fetch scalar
                float val = wLocal.GetValue(r * KT_ + c);
                wTLocal.SetValue(c * NT_ + r, val);
            }
        }
    }

    __aicore__ inline void GemmTile(AscendC::LocalTensor<float>& yLocal,
                                    AscendC::LocalTensor<float>& xLocal,
                                    AscendC::LocalTensor<float>& wTLocal,
                                    uint32_t mLen, uint32_t nLen, uint32_t kLen)
    {
        // y[mLen x nLen] += x[mLen x kLen] * wT[kLen x nLen]
        // Initialize yLocal to zeros
        AscendC::Zeros(yLocal, mLen * nLen);

        // Outer-product accumulation
        for (uint32_t kk = 0; kk < kLen; ++kk) {
            // wT row kk is contiguous of length nLen
            AscendC::LocalTensor<float> wRow = wTLocal[kk * NT_];
            for (uint32_t i = 0; i < mLen; ++i) {
                float a = xLocal.GetValue(i * KT_ + kk);
                AscendC::LocalTensor<float> yRow = yLocal[i * NT_];

                // tmp = a * wRow
                AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                AscendC::Muls(tmp, wRow, a, nLen);
                AscendC::Add(yRow, yRow, tmp, nLen);
                tmpQ_.FreeTensor(tmp);
            }
        }
    }

    __aicore__ inline void BiasRelu(AscendC::LocalTensor<float>& yLocal,
                                    AscendC::LocalTensor<float>& bLocal,
                                    uint32_t mLen, uint32_t nLen)
    {
        // Add bias per column and ReLU
        for (uint32_t i = 0; i < mLen; ++i) {
            AscendC::LocalTensor<float> yRow = yLocal[i * NT_];
            // yRow += bLocal
            AscendC::Add(yRow, yRow, bLocal, nLen);
            // ReLU
            AscendC::Maxs(yRow, yRow, 0.0f, nLen);
        }
    }

    __aicore__ inline void StoreY(uint32_t mBase, uint32_t nBase, uint32_t mLen, uint32_t nLen,
                                  AscendC::LocalTensor<float>& yLocal)
    {
        for (uint32_t r = 0; r < mLen; ++r) {
            AscendC::DataCopy(yGm_[(mBase + r) * N_ + nBase], yLocal[r * NT_], nLen);
        }
    }

    __aicore__ inline void ComputeTile(uint32_t mBase, uint32_t nBase, uint32_t mLen, uint32_t nLen)
    {
        // Allocate locals
        AscendC::LocalTensor<float> xLocal = xTileQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> wLocal = wTileQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> wTLocal = wTileTQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = yTileQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = biasQ_.AllocTensor<float>();

        // Load bias
        LoadBias(nBase, nLen, bLocal);

        // K loop
        bool first = true;
        // We accumulate into yLocal within each KT iteration via GemmTile which zeros first; to accumulate across K-chunks,
        // we keep partial sums in yLocal and add subsequent chunks.
        AscendC::Zeros(yLocal, mLen * nLen);

        for (uint32_t kBase = 0; kBase < K_; kBase += KT_) {
            const uint32_t kLen = MinU(KT_, K_ - kBase);

            LoadX(mBase, kBase, mLen, kLen, xLocal);
            LoadW(nBase, kBase, nLen, kLen, wLocal);
            TransposeW(wLocal, wTLocal, nLen, kLen);

            // yLocal += xLocal * wTLocal
            // Implement block GEMM accumulation for this k-chunk
            for (uint32_t kk = 0; kk < kLen; ++kk) {
                AscendC::LocalTensor<float> wRow = wTLocal[kk * NT_];
                for (uint32_t i = 0; i < mLen; ++i) {
                    float a = xLocal.GetValue(i * KT_ + kk);
                    AscendC::LocalTensor<float> yRow = yLocal[i * NT_];

                    AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                    AscendC::Muls(tmp, wRow, a, nLen);
                    AscendC::Add(yRow, yRow, tmp, nLen);
                    tmpQ_.FreeTensor(tmp);
                }
            }
        }

        // Bias + ReLU
        BiasRelu(yLocal, bLocal, mLen, nLen);

        // Store back
        StoreY(mBase, nBase, mLen, nLen);

        // Free
        biasQ_.FreeTensor(bLocal);
        yTileQ_.FreeTensor(yLocal);
        wTileTQ_.FreeTensor(wTLocal);
        wTileQ_.FreeTensor(wLocal);
        xTileQ_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, 1> xTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wTileTQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_ {0}, N_ {0}, K_ {0};
    uint32_t MT_ {0}, NT_ {0}, KT_ {0};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                                           GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddReluCustom op;
    op.Init(x, w, bias, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.MT, tiling_data.NT, tiling_data.KT);
    op.Process();
}
