kernel_src="""

#include "kernel_operator.h"
#include <cmath>

class KernelConv3dMinSoftmaxCustom {
public:
    __aicore__ inline KernelConv3dMinSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Dout, uint32_t Hout, uint32_t Wout,
                                uint32_t totalPositions, uint32_t tileLength)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N*Cin*D*H*W);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cout*Cin*Kd*Kh*Kw);
        bGm_.SetGlobalBuffer((__gm__ float*)b, Cout);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N*Cout*Hout*Wout);

        N_ = N; Cin_ = Cin; Cout_ = Cout;
        D_ = D; H_ = H; W_ = W;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        Dout_ = Dout; Hout_ = Hout; Wout_ = Wout;
        totalPos_ = totalPositions; tileLen_ = tileLength;
    }

    __aicore__ inline void Process()
    {
        if (totalPos_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalPos_) return;

        // Per-position scratch for channel values
        // Assuming Cout is reasonably small (e.g., 24 in given example)
        for (uint32_t pos = base; pos < totalPos_; pos += stride) {
            uint32_t maxCount = tileLen_;
            if (pos + maxCount > totalPos_) {
                maxCount = totalPos_ - pos;
            }
            for (uint32_t i = 0; i < maxCount; ++i) {
                const uint32_t p = pos + i;
                // Decode p -> (n, h_out, w_out)
                const uint32_t hwOut = Hout_ * Wout_;
                const uint32_t n = p / hwOut;
                const uint32_t rem = p % hwOut;
                const uint32_t ho = rem / Wout_;
                const uint32_t wo = rem % Wout_;

                // Compute min over depth for each output channel at this (n, ho, wo)
                // Store per-channel results in small local array
                float chValsMax = -3.402823466e+38f; // track max for softmax
                // We'll reuse a small UB-like local storage via registers
                // but keep also second pass variables.
                // Since Cout can be up to some hundred, stack array is fine
                // for this example scenario.
                // To be safe, compute in two passes storing in yGm_ temp area is not possible,
                // so use local fixed-size loop with dynamic Cout.
                // We'll compute into a temporary vector allocated on stack.
                // Note: VLAs are not allowed; use looped write/read into yTemp buffer in chunks is complex.
                // Here we use a fixed upper bound. For generality, we handle up to 1024 channels by chunks.
                const uint32_t MAX_C = 1024;
                uint32_t cProcessed = 0;
                while (cProcessed < Cout_) {
                    const uint32_t chunk = (Cout_ - cProcessed) < 1024u ? (Cout_ - cProcessed) : 1024u;
                    // Store in a small fixed array
                    float buf[1024];
                    float localMax = -3.402823466e+38f;

                    for (uint32_t c = 0; c < chunk; ++c) {
                        const uint32_t co = cProcessed + c;
                        float minVal = 3.402823466e+38f;
                        // Sweep depth dimension
                        for (uint32_t doIdx = 0; doIdx < Dout_; ++doIdx) {
                            float acc = bGm_[co];
                            // Convolution sum across Cin, Kd, Kh, Kw at (n, doIdx, ho, wo)
                            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                                for (uint32_t kd = 0; kd < Kd_; ++kd) {
                                    const uint32_t d = doIdx + kd;
                                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                                        const uint32_t h = ho + kh;
                                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                            const uint32_t w = wo + kw;
                                            uint64_t x_off = (((((uint64_t)n * Cin_ + ci) * D_ + d) * H_ + h) * W_) + w;
                                            uint64_t w_off = (((((uint64_t)co * Cin_ + ci) * Kd_ + kd) * Kh_ + kh) * Kw_) + kw;
                                            acc += xGm_[x_off] * wGm_[w_off];
                                        }
                                    }
                                }
                            }
                            if (acc < minVal) minVal = acc;
                        }
                        buf[c] = minVal;
                        if (minVal > localMax) localMax = minVal;
                    }

                    // Second pass: softmax partial write-out with normalization by chunk-local max/sum.
                    // For correct softmax across all channels, we need global max/sum.
                    // We compute global max first across chunks, then a second sweep to write final outputs.
                    // Accumulate global max
                    if (localMax > chValsMax) chValsMax = localMax;

                    // Temporarily store buf values into y buffer scratch space using y indices with an offset.
                    // We'll write final outputs after computing global sumexp.
                    for (uint32_t c = 0; c < chunk; ++c) {
                        const uint32_t co = cProcessed + c;
                        // Stash raw value into output buffer temporarily at its final location; normalize later.
                        uint64_t y_off = ((((uint64_t)n * Cout_ + co) * Hout_ + ho) * Wout_) + wo;
                        yGm_[y_off] = buf[c];
                    }

                    cProcessed += chunk;
                }

                // Compute global sumexp using chValsMax
                float sumExp = 0.f;
                for (uint32_t co = 0; co < Cout_; ++co) {
                    uint64_t y_off = ((((uint64_t)n * Cout_ + co) * Hout_ + ho) * Wout_) + wo;
                    const float v = yGm_[y_off];
                    sumExp += expf(v - chValsMax);
                }
                const float invSum = 1.f / sumExp;

                // Final write normalized probabilities
                for (uint32_t co = 0; co < Cout_; ++co) {
                    uint64_t y_off = ((((uint64_t)n * Cout_ + co) * Hout_ + ho) * Wout_) + wo;
                    const float v = yGm_[y_off];
                    const float e = expf(v - chValsMax) * invSum;
                    yGm_[y_off] = e;
                }
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t D_{0}, H_{0}, W_{0};
    uint32_t Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};

    uint32_t totalPos_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConv3dMinSoftmaxCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.Kd, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Dout, tiling_data.Hout, tiling_data.Wout,
            tiling_data.totalPositions, tiling_data.tileLength);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor allocate_output_for(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, Cin, D, H, W], w: [Cout, Cin, Kd, Kh, Kw]
    auto N = x.size(0);
    auto Cout = w.size(0);
    auto D  = x.size(2);
    auto H  = x.size(3);
    auto W  = x.size(4);
    auto Kd = w.size(2);
    auto Kh = w.size(3);
    auto Kw = w.size(4);
    auto Dout = D - Kd + 1; // reduced later by min over depth
    (void)Dout;
    auto Hout = H - Kh + 1;
    auto Wout = W - Kw + 1;
    return at::empty({N, Cout, Hout, Wout}, x.options());
}

at::Tensor conv3d_min_softmax_impl_npu(const at::Tensor& x,
                                       const at::Tensor& weight,
                                       const at::Tensor& bias) {
    at::Tensor result = allocate_output_for(x, weight);
    EXEC_NPU_CMD(aclnnConv3dMinSoftmaxCustom, x, weight, bias, result);
    return result;
}

    m.impl("conv3d_min_softmax_custom", &conv3d_min_softmax_impl_npu);
