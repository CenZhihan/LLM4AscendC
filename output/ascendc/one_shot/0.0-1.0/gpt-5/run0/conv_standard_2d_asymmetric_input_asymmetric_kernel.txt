static inline int64_t ConvOutDim(int64_t in, int64_t pad0, int64_t pad1, int64_t k, int64_t dilation, int64_t stride)

    const int64_t kernel = dilation * (k - 1) + 1;
    return (in + pad0 + pad1 - kernel) / stride + 1;
}

kernel_src="""
#include "kernel_operator.h"

// Naive direct-convolution implementation specialized for:
// - NCHW layout
// - stride_h = stride_w = 1
// - pad = 0
// - dilation = 1
// - groups = 1
// This implementation uses simple global-memory indexing for clarity.

class KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t K, uint32_t KH, uint32_t KW,
                                uint32_t OH, uint32_t OW)
    {
        xGm_ = reinterpret_cast<__gm__ float*>(x);
        wGm_ = reinterpret_cast<__gm__ float*>(w);
        yGm_ = reinterpret_cast<__gm__ float*>(y);

        N_ = N; C_ = C; H_ = H; W_ = W;
        K_ = K; KH_ = KH; KW_ = KW;
        OH_ = OH; OW_ = OW;
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || K_ == 0 || OH_ == 0 || OW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Parallelize across (n, oh) rows
        const uint32_t totalRows = N_ * OH_;
        for (uint32_t row = blockIdx; row < totalRows; row += blockNum) {
            const uint32_t n  = row / OH_;
            const uint32_t oh = row % OH_;
            // Iterate OC and OW locally
            for (uint32_t oc = 0; oc < K_; ++oc) {
                for (uint32_t ow = 0; ow < OW_; ++ow) {
                    float acc = 0.0f;

                    // Compute dot product across C, KH, KW
                    for (uint32_t ic = 0; ic < C_; ++ic) {
                        for (uint32_t kh = 0; kh < KH_; ++kh) {
                            const uint32_t ih = oh + kh; // stride=1, pad=0, dilation=1
                            const uint32_t baseX_h = (((n * C_ + ic) * H_) + ih) * W_;
                            const uint32_t baseW_h = (((oc * C_ + ic) * KH_) + kh) * KW_;

                            for (uint32_t kw = 0; kw < KW_; ++kw) {
                                const uint32_t iw = ow + kw;

                                const uint32_t xIdx = baseX_h + iw;
                                const uint32_t wIdx = baseW_h + kw;

                                acc += xGm_[xIdx] * wGm_[wIdx];
                            }
                        }
                    }

                    const uint32_t yIdx = ((((n * K_) + oc) * OH_) + oh) * OW_ + ow;
                    yGm_[yIdx] = acc;
                }
            }
        }
    }

private:
    __gm__ float* xGm_ {nullptr};
    __gm__ float* wGm_ {nullptr};
    __gm__ float* yGm_ {nullptr};

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t K_{0}, KH_{0}, KW_{0};
    uint32_t OH_{0}, OW_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.K, tiling_data.KH, tiling_data.KW,
            tiling_data.OH, tiling_data.OW);
    op.Process();
}

    TORCH_CHECK(weight.dim() == 4, \"weight must be OIHW\");

    const auto C  = x.size(1);

    const auto KH = weight.size(2);
    const auto KW = weight.size(3);

    const auto OH = H - KH + 1;
    const auto OW = W - KW + 1;
