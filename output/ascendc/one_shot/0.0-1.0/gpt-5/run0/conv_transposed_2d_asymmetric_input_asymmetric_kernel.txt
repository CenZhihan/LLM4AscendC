kernel_src="""

#include "kernel_operator.h"

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t hin, uint32_t win,
                                uint32_t cout, uint32_t kh, uint32_t kw,
                                uint32_t hout, uint32_t wout,
                                uint64_t totalOut, uint32_t tileLen)
    {
        n_ = n; cin_ = cin; hin_ = hin; win_ = win;
        cout_ = cout; kh_ = kh; kw_ = kw;
        hout_ = hout; wout_ = wout;
        totalOut_ = totalOut;
        tileLen_ = tileLen;

        xSize_ = static_cast<uint64_t>(n_) * cin_ * hin_ * win_;
        wSize_ = static_cast<uint64_t>(cin_) * cout_ * kh_ * kw_;
        ySize_ = static_cast<uint64_t>(n_) * cout_ * hout_ * wout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize_);

        // One-element accumulation tiles
        pipe_.InitBuffer(outQueueY_, 2, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpXA_, 1, sizeof(float));
        pipe_.InitBuffer(tmpWB_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint64_t base = static_cast<uint64_t>(blockIdx) * tileLen_;
        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * tileLen_;

        for (; base < totalOut_; base += strideElems) {
            uint32_t curLen = static_cast<uint32_t>((base + tileLen_ <= totalOut_) ? tileLen_ : (totalOut_ - base));
            ComputeTile(base, curLen);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint64_t base, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Dup(yLocal, 0.0f, len);

        // This kernel uses len == 1 as configured by tiling for simplicity
        for (uint32_t i = 0; i < len; ++i) {
            const uint64_t gidx = base + i;

            // Decode indices: [n, cout, hout, wout]
            uint64_t t = gidx;
            const uint32_t ow = static_cast<uint32_t>(t % wout_); t /= wout_;
            const uint32_t oh = static_cast<uint32_t>(t % hout_); t /= hout_;
            const uint32_t co = static_cast<uint32_t>(t % cout_); t /= cout_;
            const uint32_t n  = static_cast<uint32_t>(t);

            // Accumulate contributions
            const int32_t kh_min = (oh >= (kh_ - 1)) ? 0 : static_cast<int32_t>(kh_ - 1 - oh);
            const int32_t kh_max = (oh < kh_) ? static_cast<int32_t>(oh) : static_cast<int32_t>(kh_ - 1);
            const int32_t kw_min = (ow >= (kw_ - 1)) ? 0 : static_cast<int32_t>(kw_ - 1 - ow);
            const int32_t kw_max = (ow < kw_) ? static_cast<int32_t>(ow) : static_cast<int32_t>(kw_ - 1);

            for (uint32_t ci = 0; ci < cin_; ++ci) {
                for (int32_t kh = kh_min; kh <= kh_max; ++kh) {
                    const uint32_t ih = static_cast<uint32_t>(oh - kh);
                    for (int32_t kw = kw_min; kw <= kw_max; ++kw) {
                        const uint32_t iw = static_cast<uint32_t>(ow - kw);

                        // Offsets in flattened buffers
                        const uint64_t xoff = (((static_cast<uint64_t>(n) * cin_ + ci) * hin_ + ih) * win_ + iw);
                        const uint64_t woff = (((static_cast<uint64_t>(ci) * cout_ + co) * kh_ + static_cast<uint32_t>(kh)) * kw_ + static_cast<uint32_t>(kw));

                        AscendC::LocalTensor<float> x1 = tmpXA_.AllocTensor<float>();
                        AscendC::LocalTensor<float> w1 = tmpWB_.AllocTensor<float>();

                        AscendC::DataCopy(x1, xGm_[xoff], 1);
                        AscendC::DataCopy(w1, wGm_[woff], 1);

                        // x1 = x1 * w1
                        AscendC::Mul(x1, x1, w1, 1);

                        // yLocal[i] += x1
                        // Since len==1, i maps to offset 0 or we can rely on vector add with len=1 starting at yLocal base.
                        AscendC::Add(yLocal, yLocal, x1, 1);

                        tmpWB_.FreeTensor(w1);
                        tmpXA_.FreeTensor(x1);
                    }
                }
            }
        }

        outQueueY_.EnQue<float>(yLocal);

        // Write back
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        // Contiguous writeback for each element
        for (uint32_t i = 0; i < len; ++i) {
            const uint64_t gidx = base + i;
            uint64_t t = gidx;
            const uint32_t ow = static_cast<uint32_t>(t % wout_); t /= wout_;
            const uint32_t oh = static_cast<uint32_t>(t % hout_); t /= hout_;
            const uint32_t co = static_cast<uint32_t>(t % cout_); t /= cout_;
            const uint32_t n  = static_cast<uint32_t>(t);
            const uint64_t yoff = (((static_cast<uint64_t>(n) * cout_ + co) * hout_ + oh) * wout_ + ow);
            // Copy single element from local position 0 (since len == 1, only one value is present)
            AscendC::DataCopy(yGm_[yoff], yOut, 1);
        }
        outQueueY_.FreeTensor(yOut);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpXA_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpWB_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, hin_{0}, win_{0};
    uint32_t cout_{0}, kh_{0}, kw_{0};
    uint32_t hout_{0}, wout_{0};
    uint64_t totalOut_{0};
    uint32_t tileLen_{1};
    uint64_t xSize_{0}, wSize_{0}, ySize_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.hin, tiling_data.win,
            tiling_data.cout, tiling_data.kh, tiling_data.kw,
            tiling_data.hout, tiling_data.wout,
            tiling_data.totalOut, tiling_data.tileLength);
    op.Process();
}
