host_operator_src="""

#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model (floats):
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue: 1 * tile
//   gQueue: 1 * tile
//   bQueue: 1 * tile
// Total floats ~ 7 * tile => bytes ~ 28 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core safe budget
static constexpr uint32_t ALIGN_ELEMS    = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;

    // Shapes
    const uint64_t xSize     = context->GetInputShape(0)->GetOriginShape().GetShapeSize();
    const uint64_t gammaSize = context->GetInputShape(1)->GetOriginShape().GetShapeSize();
    const uint64_t betaSize  = context->GetInputShape(2)->GetOriginShape().GetShapeSize();

    if (gammaSize == 0 || betaSize == 0 || gammaSize != betaSize || xSize == 0) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t rowLen = static_cast<uint32_t>(gammaSize);
    if (xSize % rowLen != 0) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t rows = static_cast<uint32_t>(xSize / rowLen);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    const float eps = (epsPtr == nullptr) ? 1e-5f : *epsPtr;

    // tileLen from UB budget: bytes ~ 28 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 28;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (rowLen < tileLen) {
        const uint32_t aligned = AlignDown(rowLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : rowLen;
    }

    // Blocks schedule rows
    uint32_t blockDim = rows;
    if (blockDim < 8u)  blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_rowLength(rowLen);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t rows, uint32_t rowLength, uint32_t tileLength,
                                float eps)
    {
        rows_      = rows;
        rowLen_    = rowLength;
        tileLen_   = tileLength;
        eps_       = eps;

        const uint32_t totalLen = rows_ * rowLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen);
        gGm_.SetGlobalBuffer((__gm__ float*)gamma, rowLen_);
        bGm_.SetGlobalBuffer((__gm__ float*)beta, rowLen_);

        pipe_.InitBuffer(inQueueX_,  BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_,  1,           tileLen_ * sizeof(float));
        pipe_.InitBuffer(gQueue_,    1,           tileLen_ * sizeof(float));
        pipe_.InitBuffer(bQueue_,    1,           tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a grid-stride set of rows
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            const uint32_t rowBase = row * rowLen_;
            // Phase 1: compute mean and variance
            float sum = 0.0f;
            float sqSum = 0.0f;

            uint32_t offset = 0;
            while (offset < rowLen_) {
                const uint32_t curLen = AscendC::min(tileLen_, rowLen_ - offset);
                CopyInX(rowBase + offset, curLen);

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                // sum
                float partSum = AscendC::ReduceSum(xLocal, curLen);
                // square into tmp buffer
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                AscendC::Mul(tLocal, xLocal, xLocal, curLen);
                float partSqSum = AscendC::ReduceSum(tLocal, curLen);

                tmpQueue_.FreeTensor(tLocal);
                inQueueX_.FreeTensor(xLocal);

                sum   += partSum;
                sqSum += partSqSum;

                offset += curLen;
            }

            const float n = static_cast<float>(rowLen_);
            const float mean = sum / n;
            float var = sqSum / n - mean * mean;
            if (var < 0.0f) var = 0.0f; // numerical guard
            const float invStd = AscendC::rsqrt(var + eps_);

            // Phase 2: normalize, scale and shift
            offset = 0;
            while (offset < rowLen_) {
                const uint32_t curLen = AscendC::min(tileLen_, rowLen_ - offset);

                // Prefetch x, gamma, beta
                CopyInX(rowBase + offset, curLen);
                CopyInG(offset, curLen);
                CopyInB(offset, curLen);

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> gLocal = gQueue_.DeQue<float>();
                AscendC::LocalTensor<float> bLocal = bQueue_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // y = ((x - mean) * invStd) * gamma + beta
                AscendC::Adds(yLocal, xLocal, -mean, curLen);
                AscendC::Muls(yLocal, yLocal, invStd, curLen);
                AscendC::Mul(yLocal, yLocal, gLocal, curLen);
                AscendC::Add(yLocal, yLocal, bLocal, curLen);

                outQueueY_.EnQue(yLocal);
                gQueue_.FreeTensor(gLocal);
                bQueue_.FreeTensor(bLocal);
                inQueueX_.FreeTensor(xLocal);

                CopyOut(rowBase + offset, curLen);
                offset += curLen;
            }
        }
    }

private:
    __aicore__ inline void CopyInX(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInG(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> gLocal = gQueue_.AllocTensor<float>();
        AscendC::DataCopy(gLocal, gGm_[gmOffset], len);
        gQueue_.EnQue(gLocal);
    }

    __aicore__ inline void CopyInB(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();
        AscendC::DataCopy(bLocal, bGm_[gmOffset], len);
        bQueue_.EnQue(bLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>         tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN,  1>          gQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN,  1>          bQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> gGm_;
    AscendC::GlobalTensor<float> bGm_;

    uint32_t rows_{0};
    uint32_t rowLen_{0};
    uint32_t tileLen_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.rows, tiling_data.rowLength, tiling_data.tileLength,
            tiling_data.eps);
    op.Process();
}
"""

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape, *init_inputs):
        super(ModelNew, self).__init__()
        # Parameters to mimic nn.LayerNorm with affine=True
        self.weight = torch.nn.Parameter(torch.ones(normalized_shape, dtype=torch.float32))
        self.bias = torch.nn.Parameter(torch.zeros(normalized_shape, dtype=torch.float32))
        self.eps = 1e-5

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        # Ensure parameter device/dtype alignment
        gamma = self.weight.to(device=x.device, dtype=x.dtype)
        beta = self.bias.to(device=x.device, dtype=x.dtype)
        return fn(x, gamma, beta, self.eps)
'''
