#include "conv_standard_2d_square_input_asymmetric_kernel_custom_tiling.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model per core:
//   yTile: 1 * tileW
//   xTile: 1 * tileW
//   wRow:  kW
//   tmpAcc: 1 * tileW
// Total bytes ~= (3 * tileW + kW) * sizeof(float) with light safety margin
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expecting NCHW layouts
    const uint32_t n    = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t cIn  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t h    = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t w    = static_cast<uint32_t>(x_shape->GetDim(3));
    const uint32_t cOut = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t kH   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t kW   = static_cast<uint32_t>(w_shape->GetDim(3));

    // stride=1, padding=0, dilation=1
    const uint32_t hOut = (h >= kH) ? (h - kH + 1u) : 0u;
    const uint32_t wOut = (w >= kW) ? (w - kW + 1u) : 0u;

    // Determine tile width from UB budget
    const uint32_t bytesPerElem = 4u;
    uint32_t maxTileWElems = 0u;
    if (bytesPerElem > 0u) {
        // UB usage ~= (3*tileW + kW) * 4
        if (UB_BUDGET_BYTES > kW * bytesPerElem) {
            maxTileWElems = (UB_BUDGET_BYTES - kW * bytesPerElem) / (3u * bytesPerElem);
        }
    }
    if (maxTileWElems == 0u) maxTileWElems = 1024u;
    maxTileWElems = AlignDown(maxTileWElems, ALIGN_ELEMS);
    if (maxTileWElems < ALIGN_ELEMS) maxTileWElems = ALIGN_ELEMS;

    uint32_t tileW = maxTileWElems;
    if (wOut > 0u && wOut < tileW) {
        uint32_t aligned = AlignDown(wOut, ALIGN_ELEMS);
        tileW = (aligned > 0u) ? aligned : wOut;
    }
    if (wOut == 0u) tileW = 0u;

    // Decide blockDim by number of work tiles
    uint64_t tilesPerRow  = (tileW == 0u) ? 1u : CeilDiv(wOut, tileW);
    uint64_t totalTiles   = static_cast<uint64_t>(n) * cOut * hOut * tilesPerRow;
    uint32_t blockDim     = (totalTiles == 0u) ? 1u : static_cast<uint32_t>(totalTiles);
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim < 8u) blockDim = (totalTiles >= 8u) ? 8u : static_cast<uint32_t>(totalTiles);

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_cIn(cIn);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_cOut(cOut);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_hOut(hOut);
    tiling.set_wOut(wOut);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t wi,
                                uint32_t cOut, uint32_t kH, uint32_t kW,
                                uint32_t hOut, uint32_t wOut, uint32_t tileW)
    {
        n_ = n; cIn_ = cIn; h_ = h; w_ = wi;
        cOut_ = cOut; kH_ = kH; kW_ = kW;
        hOut_ = hOut; wOut_ = wOut; tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cIn_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, cOut_ * cIn_ * kH_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * cOut_ * hOut_ * wOut_);

        // UB buffers
        // xTile: tileW, yTile: tileW, tmpAcc: tileW, wRow: kW
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpAccQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpWRowQueue_, 1, (kW_ > 0 ? kW_ : 1) * sizeof(float));
        pipe_.InitBuffer(tmpScalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cOut_ == 0 || hOut_ == 0 || wOut_ == 0 || tileW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tilesPerRow  = (wOut_ + tileW_ - 1u) / tileW_;
        const uint64_t tilesPerImg  = static_cast<uint64_t>(cOut_) * hOut_ * tilesPerRow;
        const uint64_t totalTiles   = static_cast<uint64_t>(n_) * tilesPerImg;

        for (uint64_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            // Decode tileId -> nIdx, oc, oh, wTile
            uint64_t tmp = tileId;
            const uint32_t wTile = static_cast<uint32_t>(tmp % tilesPerRow); tmp /= tilesPerRow;
            const uint32_t oh    = static_cast<uint32_t>(tmp % hOut_);       tmp /= hOut_;
            const uint32_t oc    = static_cast<uint32_t>(tmp % cOut_);       tmp /= cOut_;
            const uint32_t nIdx  = static_cast<uint32_t>(tmp);

            const uint32_t wBase = wTile * tileW_;
            uint32_t curW = wOut_ - wBase;
            if (curW > tileW_) curW = tileW_;

            // Prepare output accumulator yLocal = 0
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Duplicate(yLocal, 0.0f, curW);

            // For each input channel and kernel element, accumulate
            for (uint32_t ci = 0; ci < cIn_; ++ci) {
                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    // Load weight row [kW] for current (oc,ci,kh,:)
                    const uint64_t wOffsetRow = ((((uint64_t)oc * cIn_ + ci) * kH_ + kh) * kW_);
                    AscendC::LocalTensor<float> wRow = tmpWRowQueue_.AllocTensor<float>();
                    AscendC::DataCopy(wRow, wGm_[wOffsetRow], kW_);

                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        // Load x vector for this window position
                        const uint32_t ih = oh + kh;
                        const uint32_t iwBase = wBase + kw;
                        const uint64_t xOffset = ((((uint64_t)nIdx * cIn_ + ci) * h_ + ih) * w_ + iwBase);
                        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                        AscendC::DataCopy(xLocal, xGm_[xOffset], curW);

                        // Load scalar weight from wRow[kw]
                        AscendC::LocalTensor<float> wScalarBuf = tmpScalarQueue_.AllocTensor<float>();
                        AscendC::DataCopy(wScalarBuf, wRow[kw], 1);
                        // Convert to scalar
                        float wScalar = wScalarBuf.GetValue(0);

                        // acc = xLocal * wScalar
                        AscendC::LocalTensor<float> accLocal = tmpAccQueue_.AllocTensor<float>();
                        AscendC::Muls(accLocal, xLocal, wScalar, curW);
                        // yLocal += accLocal
                        AscendC::Add(yLocal, yLocal, accLocal, curW);

                        tmpAccQueue_.FreeTensor(accLocal);
                        tmpScalarQueue_.FreeTensor(wScalarBuf);
                        inQueueX_.FreeTensor(xLocal);
                    }
                    tmpWRowQueue_.FreeTensor(wRow);
                }
            }

            // Store yLocal to GM
            const uint64_t yOffset = ((((uint64_t)nIdx * cOut_ + oc) * hOut_ + oh) * wOut_ + wBase);
            AscendC::DataCopy(yGm_[yOffset], yLocal, curW);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>         tmpAccQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>         tmpWRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>         tmpScalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, h_{0}, w_{0};
    uint32_t cOut_{0}, kH_{0}, kW_{0};
    uint32_t hOut_{0}, wOut_{0}, tileW_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cIn, tiling_data.h, tiling_data.w,
            tiling_data.cOut, tiling_data.kH, tiling_data.kW,
            tiling_data.hOut, tiling_data.wOut, tiling_data.tileW);
    op.Process();
}
