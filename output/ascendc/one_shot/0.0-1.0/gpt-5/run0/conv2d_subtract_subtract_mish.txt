#include "kernel_operator.h"
#include <cmath>

class KernelConv2dSubtractSubtractMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractSubtractMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t Co,
                                uint32_t H, uint32_t W,
                                uint32_t KH, uint32_t KW,
                                uint32_t Ho, uint32_t Wo,
                                int32_t strideH, int32_t strideW,
                                int32_t padH, int32_t padW,
                                float sub1, float sub2,
                                uint64_t totalOut)
    {
        x_ = reinterpret_cast<__gm__ float*>(x);
        w_ = reinterpret_cast<__gm__ float*>(w);
        b_ = reinterpret_cast<__gm__ float*>(b);
        y_ = reinterpret_cast<__gm__ float*>(y);

        N_ = N; Ci_ = Ci; Co_ = Co;
        H_ = H; W_ = W; KH_ = KH; KW_ = KW;
        Ho_ = Ho; Wo_ = Wo;
        sH_ = strideH; sW_ = strideW;
        pH_ = padH; pW_ = padW;
        sub1_ = sub1; sub2_ = sub2;
        totalOut_ = totalOut;
        hasBias_ = (b != nullptr);
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t idx = blockIdx; idx < totalOut_; idx += blockNum) {
            uint32_t ow = static_cast<uint32_t>(idx % Wo_);
            uint64_t tmp = idx / Wo_;
            uint32_t oh = static_cast<uint32_t>(tmp % Ho_);
            tmp /= Ho_;
            uint32_t oc = static_cast<uint32_t>(tmp % Co_);
            uint32_t n  = static_cast<uint32_t>(tmp / Co_);

            float acc = hasBias_ ? b_[oc] : 0.0f;

            for (uint32_t ic = 0; ic < Ci_; ++ic) {
                for (uint32_t ky = 0; ky < KH_; ++ky) {
                    const int in_y = static_cast<int>(oh) * sH_ + static_cast<int>(ky) - pH_;
                    if (in_y < 0 || in_y >= static_cast<int>(H_)) continue;
                    for (uint32_t kx = 0; kx < KW_; ++kx) {
                        const int in_x = static_cast<int>(ow) * sW_ + static_cast<int>(kx) - pW_;
                        if (in_x < 0 || in_x >= static_cast<int>(W_)) continue;

                        const uint64_t x_off =
                            (((uint64_t)n * Ci_ + ic) * H_ + static_cast<uint32_t>(in_y)) * W_
                            + static_cast<uint32_t>(in_x);
                        const uint64_t w_off =
                            (((uint64_t)oc * Ci_ + ic) * KH_ + ky) * KW_ + kx;

                        const float xv = x_[x_off];
                        const float wv = w_[w_off];
                        acc += xv * wv;
                    }
                }
            }

            float v = acc - sub1_ - sub2_;
            float sp = log1pf(expf(v));
            float act = v * tanhf(sp);

            const uint64_t y_off = idx;
            y_[y_off] = act;
        }
    }

private:
    __gm__ float* x_ {nullptr};
    __gm__ float* w_ {nullptr};
    __gm__ float* b_ {nullptr};
    __gm__ float* y_ {nullptr};

    uint32_t N_{0}, Ci_{0}, Co_{0};
    uint32_t H_{0}, W_{0}, KH_{0}, KW_{0}, Ho_{0}, Wo_{0};
    int32_t sH_{1}, sW_{1}, pH_{0}, pW_{0};
    float sub1_{0.f}, sub2_{0.f};
    uint64_t totalOut_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_subtract_subtract_mish_custom(GM_ADDR x,
                                                                            GM_ADDR weight,
                                                                            GM_ADDR conv_bias,
                                                                            GM_ADDR y,
                                                                            GM_ADDR workspace,
                                                                            GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractSubtractMishCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.N, tiling_data.Ci, tiling_data.Co,
            tiling_data.H, tiling_data.W,
            tiling_data.KH, tiling_data.KW,
            tiling_data.Ho, tiling_data.Wo,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.sub1, tiling_data.sub2,
            tiling_data.totalOut);
    op.Process();
}
