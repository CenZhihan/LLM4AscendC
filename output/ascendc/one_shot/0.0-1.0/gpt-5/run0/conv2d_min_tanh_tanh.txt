host_operator_src="""

#include "conv2d_min_tanh_tanh_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB is not the bottleneck here; choose a modest tile to balance scheduling.
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MIN_TILE = 256;
static constexpr uint32_t MAX_BLOCKS = 32;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMinTanhTanhCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const auto x_dims = x_shape->GetOriginShape().GetDims();
    const auto w_dims = w_shape->GetOriginShape().GetDims();

    if (x_dims.size() != 4 || w_dims.size() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N    = static_cast<uint32_t>(x_dims[0]);
    const uint32_t C_in = static_cast<uint32_t>(x_dims[1]);
    const uint32_t H_in = static_cast<uint32_t>(x_dims[2]);
    const uint32_t W_in = static_cast<uint32_t>(x_dims[3]);

    const uint32_t C_out = static_cast<uint32_t>(w_dims[0]);
    const uint32_t K     = static_cast<uint32_t>(w_dims[2]); // assume square kernel

    // stride=1, padding=0, dilation=1
    if (H_in < K || W_in < K) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t H_out = H_in - K + 1;
    const uint32_t W_out = W_in - K + 1;

    const uint32_t totalPos = N * H_out * W_out;

    // Optional bias check
    uint32_t hasBias = 0u;
    if (context->GetInputNum() >= 3) {
        const gert::Shape *b_shape = context->GetInputShape(2);
        if (b_shape != nullptr && b_shape->GetShapeSize() == C_out) {
            hasBias = 1u;
        }
    }

    // Choose tile over output positions
    uint32_t tilePos = totalPos;
    if (tilePos > MIN_TILE) {
        tilePos = AlignDown(tilePos, ALIGN_ELEMS);
        if (tilePos == 0) tilePos = MIN_TILE;
        if (tilePos > 8192u) tilePos = 8192u; // keep compute loop bounded per-iter
    }

    const uint32_t numTiles = (tilePos == 0) ? 1u : CeilDiv(totalPos, tilePos);
    uint32_t blockDim = numTiles;
    if (blockDim == 0) blockDim = 1u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim < 4u) blockDim = 4u;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cIn(C_in);
    tiling.set_cOut(C_out);
    tiling.set_inH(H_in);
    tiling.set_inW(W_in);
    tiling.set_k(K);
    tiling.set_outH(H_out);
    tiling.set_outW(W_out);
    tiling.set_totalPos(totalPos);
    tiling.set_tilePos(tilePos);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const auto &x_dims = x_shape->GetOriginShape().GetDims();
    const auto &w_dims = w_shape->GetOriginShape().GetDims();

    if (x_dims.size() != 4 || w_dims.size() != 4) {
        return GRAPH_FAILED;
    }

    const int64_t N  = x_dims[0];
    const int64_t H  = x_dims[2];
    const int64_t W  = x_dims[3];
    const int64_t K  = w_dims[2]; // assume square kernel KxK

    const int64_t H_out = H - K + 1;
    const int64_t W_out = W - K + 1;

    if (H_out <= 0 || W_out <= 0) return GRAPH_FAILED;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, 1);       // reduced over C_out -> keepdim=True
    y_shape->SetDim(2, H_out);
    y_shape->SetDim(3, W_out);

    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dMinTanhTanhCustom {
public:
    __aicore__ inline KernelConv2dMinTanhTanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t inH, uint32_t inW, uint32_t k,
                                uint32_t outH, uint32_t outW,
                                uint32_t totalPos, uint32_t tilePos,
                                uint32_t hasBias)
    {
        xGm_ = (__gm__ float*)x;
        wGm_ = (__gm__ float*)w;
        bGm_ = (__gm__ float*)b;
        yGm_ = (__gm__ float*)y;

        N_ = n; C_in_ = cIn; C_out_ = cOut;
        H_in_ = inH; W_in_ = inW; K_ = k;
        H_out_ = outH; W_out_ = outW;

        totalPos_ = totalPos;
        tilePos_ = (tilePos == 0 ? 1u : tilePos);
        hasBias_ = hasBias != 0;

        // Precompute strides
        xStrideC_ = H_in_ * W_in_;
        xStrideN_ = C_in_ * xStrideC_;
        wStrideCout_ = C_in_ * K_ * K_;
        wStrideCin_ = K_ * K_;
        yStrideN_ = H_out_ * W_out_;
    }

    __aicore__ inline float tanh_approx(float x) const
    {
        // A simple rational approximation with decent accuracy over wide range.
        float x2 = x * x;
        float num = x * (27.0f + x2);
        float den = 27.0f + 9.0f * x2;
        return num / den;
    }

    __aicore__ inline void Process()
    {
        if (totalPos_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stridePos = blockNum * tilePos_;
        uint32_t base = blockIdx * tilePos_;
        if (base >= totalPos_) return;

        while (base < totalPos_) {
            uint32_t len = totalPos_ - base;
            if (len > tilePos_) len = tilePos_;

            // Compute len output positions [base, base+len)
            for (uint32_t i = 0; i < len; ++i) {
                const uint32_t p = base + i;
                // map p -> (n, oh, ow)
                uint32_t n = p / yStrideN_;
                uint32_t rem = p - n * yStrideN_;
                uint32_t oh = rem / W_out_;
                uint32_t ow = rem - oh * W_out_;

                // sliding window top-left in input
                const uint32_t ih0 = oh;
                const uint32_t iw0 = ow;

                // compute min over C_out of conv result
                float minVal = 3.4e38f; // large positive

                for (uint32_t co = 0; co < C_out_; ++co) {
                    float acc = 0.0f;
                    // bias
                    if (hasBias_) {
                        acc += bGm_[co];
                    }
                    // convolution: sum over ci, kh, kw
                    const uint32_t wBaseCo = co * wStrideCout_;
                    const uint32_t xBaseN  = n * xStrideN_;
                    for (uint32_t ci = 0; ci < C_in_; ++ci) {
                        const uint32_t xBaseNC = xBaseN + ci * xStrideC_;
                        const uint32_t wBaseCoci = wBaseCo + ci * wStrideCin_;
                        for (uint32_t kh = 0; kh < K_; ++kh) {
                            const uint32_t ih = ih0 + kh;
                            const uint32_t xRowBase = xBaseNC + ih * W_in_;
                            const uint32_t wRowBase = wBaseCoci + kh * K_;
                            for (uint32_t kw = 0; kw < K_; ++kw) {
                                const uint32_t iw = iw0 + kw;
                                float xv = xGm_[xRowBase + iw];
                                float wv = wGm_[wRowBase + kw];
                                acc += xv * wv;
                            }
                        }
                    }
                    if (acc < minVal) minVal = acc;
                }

                // y = tanh(tanh(minVal))
                float t1 = tanh_approx(minVal);
                float t2 = tanh_approx(t1);
                yGm_[p] = t2; // y is [N,1,H_out,W_out] flattened -> index p
            }

            base += stridePos;
        }
    }

private:
    __gm__ float* xGm_ {nullptr};
    __gm__ float* wGm_ {nullptr};
    __gm__ float* bGm_ {nullptr};
    __gm__ float* yGm_ {nullptr};

    uint32_t N_{0}, C_in_{0}, C_out_{0};
    uint32_t H_in_{0}, W_in_{0}, K_{0};
    uint32_t H_out_{0}, W_out_{0};

    uint32_t totalPos_{0};
    uint32_t tilePos_{1};

    // strides
    uint32_t xStrideC_{0};
    uint32_t xStrideN_{0};
    uint32_t wStrideCout_{0};
    uint32_t wStrideCin_{0};
    uint32_t yStrideN_{0};

    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanhCustom op;
    op.Init(
        x, weight, bias, y,
        tiling_data.n, tiling_data.cIn, tiling_data.cOut,
        tiling_data.inH, tiling_data.inW, tiling_data.k,
        tiling_data.outH, tiling_data.outW,
        tiling_data.totalPos, tiling_data.tilePos,
        tiling_data.hasBias
    );
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor infer_output_like_conv_min(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, C_in, H, W], w: [C_out, C_in, K, K], stride=1, pad=0, dil=1
    TORCH_CHECK(x.dim() == 4 && w.dim() == 4, \"Input dims must be 4D\");
    auto N = x.size(0);
    auto Hin = x.size(2);
    auto Win = x.size(3);
    auto K = w.size(2);
    auto Hout = Hin - K + 1;
    auto Wout = Win - K + 1;
    TORCH_CHECK(Hout > 0 && Wout > 0, \"Invalid shapes for valid conv: H/W less than kernel\");
    auto options = x.options();
    return at::empty({N, 1, Hout, Wout}, options);
}

at::Tensor conv2d_min_tanh_tanh_impl_npu(const at::Tensor& x,
                                         const at::Tensor& weight,
                                         c10::optional<at::Tensor> bias_opt) {
    at::Tensor y = infer_output_like_conv_min(x, weight);
    if (bias_opt.has_value()) {
        EXEC_NPU_CMD(aclnnConv2dMinTanhTanhCustom, x, weight, bias_opt.value(), y);
    } else {
        EXEC_NPU_CMD(aclnnConv2dMinTanhTanhCustom, x, weight, at::Tensor(), y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv2d_min_tanh_tanh_custom\", &conv2d_min_tanh_tanh_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"conv2d_min_tanh_tanh_custom\", &conv2d_min_tanh_tanh_impl_npu,
          \"Conv2d + min(channel, keepdim) + tanh + tanh (Conv2dMinTanhTanhCustom on NPU)\");
}
