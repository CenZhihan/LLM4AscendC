host_operator_src="""
#include "matmul_for_upper_triangular_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// UB model (per core):
//   yBuf (accumulator): 1 * colTile
//   bBuf (B slice):     1 * colTile
//   tBuf (temp):        1 * colTile
//   aScalar:            1
// Total floats ≈ 3 * colTile + 1 => bytes ≈ 12 * colTile + 4
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ALIGN_ELEMS = 256;             // DMA alignment preference
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // guard

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForUpperTriangularMatricesCustomTilingData tiling;

    const gert::Shape *shapeA = context->GetInputShape(0);
    // Expect square (N,N)
    const uint32_t N = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(0));

    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    if (maxTile > N) {
        uint32_t aligned = AlignDown(N, ALIGN_ELEMS);
        maxTile = (aligned > 0u) ? aligned : N;
    }

    uint32_t colTile = maxTile;
    if (colTile == 0) colTile = 1;

    // Use rows distribution across blocks
    uint32_t blockDim = N;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_colTile(colTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    gert::Shape *c_shape = context->GetOutputShape(0);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulForUpperTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForUpperTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t colTile)
    {
        N_ = N;
        colTile_ = colTile;

        aGm_.SetGlobalBuffer((__gm__ float*)A, N_ * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, N_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, N_ * N_);

        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, colTile_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, colTile_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, colTile_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || colTile_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Distribute rows across blocks in a grid-stride loop.
        for (uint32_t i = blockIdx; i < N_; i += blockNum) {
            // 1) Zero lower-triangular part for row i: columns [0, i)
            uint32_t remainingLower = i;
            uint32_t jBase = 0;
            while (remainingLower > 0) {
                uint32_t len = remainingLower < colTile_ ? remainingLower : colTile_;
                AscendC::LocalTensor<float> yLocal = outQueueC_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, len);
                outQueueC_.EnQue(yLocal);

                AscendC::LocalTensor<float> yOut = outQueueC_.DeQue<float>();
                AscendC::DataCopy(cGm_[i * N_ + jBase], yOut, len);
                outQueueC_.FreeTensor(yOut);

                jBase += len;
                remainingLower -= len;
            }

            // 2) Compute upper-triangular part for row i: columns [i, N_)
            for (uint32_t colStart = i; colStart < N_; colStart += colTile_) {
                uint32_t len = N_ - colStart;
                if (len > colTile_) len = colTile_;

                // Accumulator buffer initialized to zeros
                AscendC::LocalTensor<float> accLocal = outQueueC_.AllocTensor<float>();
                AscendC::Dup(accLocal, 0.0f, len);

                // For this tile, we only need k in [i, colStart + len - 1]
                const uint32_t kEnd = colStart + len; // exclusive
                for (uint32_t k = i; k < kEnd; ++k) {
                    // Load scalar A[i, k]
                    AscendC::LocalTensor<float> aScalar = scalarQueue_.AllocTensor<float>();
                    AscendC::DataCopy(aScalar, aGm_[i * N_ + k], 1);
                    // Read scalar from local
                    float aVal = aScalar.GetValue(0);
                    scalarQueue_.FreeTensor(aScalar);

                    // Load B[k, colStart:colStart+len]
                    AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
                    AscendC::DataCopy(bLocal, bGm_[k * N_ + colStart], len);
                    inQueueB_.EnQue(bLocal);

                    // Compute accLocal += aVal * bLocal
                    AscendC::LocalTensor<float> bChunk = inQueueB_.DeQue<float>();
                    AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::Muls(tLocal, bChunk, aVal, len);
                    AscendC::Add(accLocal, accLocal, tLocal, len);
                    tmpQueue_.FreeTensor(tLocal);
                    inQueueB_.FreeTensor(bChunk);
                }

                // Store result to C[i, colStart:colStart+len]
                outQueueC_.EnQue(accLocal);
                AscendC::LocalTensor<float> yOut = outQueueC_.DeQue<float>();
                AscendC::DataCopy(cGm_[i * N_ + colStart], yOut, len);
                outQueueC_.FreeTensor(yOut);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t colTile_{0};
};

extern "C" __global__ __aicore__ void matmul_for_upper_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForUpperTriangularMatricesCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.colTile);
    op.Process();
}
