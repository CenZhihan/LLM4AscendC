#include "matmul_scaling_residual_add_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1ull) / b);
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulScalingResidualAddCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    if (x_shape->GetOriginShape().GetDimNum() != 2 ||
        w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t M = x_shape->GetOriginShape().GetDim(0);
    const uint64_t K = x_shape->GetOriginShape().GetDim(1);
    const uint64_t N = w_shape->GetOriginShape().GetDim(0);
    const uint64_t KW = w_shape->GetOriginShape().GetDim(1);

    if (K != KW) return ge::GRAPH_FAILED;
    if (b_shape->GetOriginShape().GetShapeSize() != N) return ge::GRAPH_FAILED;

    // Choose tiles (tm, tn, tk) based on UB memory:
    // UB bytes needed ~= 4 * (tm*tk + tk*tn + tm*tn) for A_sub, B_subT, C_sub
    // Start with tm and tk, then solve tn from budget.
    uint32_t tm = 16;
    uint32_t tk = 64;
    if (K < tk) tk = AlignDown(static_cast<uint32_t>(K), ALIGN_ELEMS);
    if (tk == 0) tk = static_cast<uint32_t>(K);

    const uint32_t bytes_per_float = 4;
    uint32_t budget_elems = UB_BUDGET_BYTES / bytes_per_float;
    // tn <= (budget - tm*tk) / (tk + tm)
    uint32_t tn = 0;
    if ((tm * tk) < budget_elems) {
        tn = (budget_elems - tm * tk) / (tk + tm);
        tn = AlignDown(tn, ALIGN_ELEMS);
    }
    if (tn < ALIGN_ELEMS) tn = ALIGN_ELEMS;
    if (tn > N) tn = AlignDown(static_cast<uint32_t>(N), ALIGN_ELEMS);
    if (tn == 0) tn = static_cast<uint32_t>(N);

    const uint32_t mTiles = CeilDiv(M, tm);
    const uint32_t nTiles = CeilDiv(N, tn);

    // Blocks over M tiles. If many N tiles too, we still stride M tiles per block.
    uint32_t blockDim = mTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_m(static_cast<uint32_t>(M));
    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_k(static_cast<uint32_t>(K));
    tiling.set_tm(tm);
    tiling.set_tn(tn);
    tiling.set_tk(tk);
    tiling.set_mTiles(mTiles);
    tiling.set_nTiles(nTiles);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

// Simple GEMM tile with fused scaling residual: y = (x @ w^T + b) * s + (x @ w^T)
// Inputs:
//   x: [M, K], row-major
//   w: [N, K], row-major (note: matmul uses w^T logically)
//   b: [N]
//   scaling: scalar tensor [1]
// Output:
//   y: [M, N]
constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulScalingResidualAddCustom {
public:
    __aicore__ inline KernelMatmulScalingResidualAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tm, uint32_t tn, uint32_t tk)
    {
        M_ = M; N_ = N; K_ = K;
        tm_ = tm; tn_ = tn; tk_ = tk;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        sGm_.SetGlobalBuffer((__gm__ float*)scaling, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * N_);

        // Allocate UB buffers: C_sub (tm*tn), bVec (tn)
        pipe_.InitBuffer(cTileQ_, BUFFER_NUM, tm_ * tn_ * sizeof(float));
        pipe_.InitBuffer(bVecQ_, BUFFER_NUM, tn_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        // Read scaling scalar once
        float s = 0.0f;
        {
            AscendC::LocalTensor<float> sLocal = bVecQ_.AllocTensor<float>();
            AscendC::DataCopy(sLocal, sGm_[0], 1);
            bVecQ_.EnQue(sLocal);
            AscendC::LocalTensor<float> sLocal2 = bVecQ_.DeQue<float>();
            s = sLocal2.GetValue(0);
            bVecQ_.FreeTensor(sLocal2);
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over M tiles per block
        const uint32_t mTiles = (M_ + tm_ - 1u) / tm_;
        const uint32_t nTiles = (N_ + tn_ - 1u) / tn_;

        for (uint32_t mt = blockIdx; mt < mTiles; mt += blockNum) {
            const uint32_t mStart = mt * tm_;
            const uint32_t mLen = (mStart + tm_ <= M_) ? tm_ : (M_ - mStart);

            for (uint32_t nt = 0; nt < nTiles; ++nt) {
                const uint32_t nStart = nt * tn_;
                const uint32_t nLen = (nStart + tn_ <= N_) ? tn_ : (N_ - nStart);

                // Allocate and zero C tile [mLen, nLen]
                AscendC::LocalTensor<float> cTile = cTileQ_.AllocTensor<float>();
                // Zero initialize
                AscendC::Zeros(cTile, mLen * nLen);

                // Accumulate C += A_sub @ B_subT over K
                for (uint32_t k0 = 0; k0 < K_; k0 += tk_) {
                    const uint32_t kLen = ((k0 + tk_) <= K_) ? tk_ : (K_ - k0);

                    // For each k within this chunk: outer product accumulate
                    for (uint32_t kp = 0; kp < kLen; ++kp) {
                        // Load bVec = w[:, k0 + kp] slice for columns nStart..nStart+nLen-1
                        AscendC::LocalTensor<float> bVec = bVecQ_.AllocTensor<float>();
                        // Manually gather strided elements along first dim
                        for (uint32_t j = 0; j < nLen; ++j) {
                            const uint32_t wRow = nStart + j;
                            const uint32_t wCol = k0 + kp;
                            const uint64_t wIdx = (uint64_t)wRow * K_ + wCol; // [N,K] row-major
                            float val;
                            AscendC::DataCopy(bVec[j], wGm_[wIdx], 1);
                        }

                        // For each row i in A_sub, accumulate C[i,:] += A[i, k0+kp] * bVec[:]
                        for (uint32_t i = 0; i < mLen; ++i) {
                            const uint32_t aRow = mStart + i;
                            const uint32_t aCol = k0 + kp;
                            const uint64_t aIdx = (uint64_t)aRow * K_ + aCol;
                            // Load scalar aVal
                            AscendC::LocalTensor<float> aScal = bVecQ_.AllocTensor<float>();
                            AscendC::DataCopy(aScal, xGm_[aIdx], 1);
                            float aVal = 0.0f;
                            aVal = aScal.GetValue(0);
                            bVecQ_.FreeTensor(aScal);

                            // cRow pointer
                            for (uint32_t j = 0; j < nLen; ++j) {
                                const uint32_t colOff = j;
                                const uint64_t cOff = (uint64_t)i * tn_ + colOff;
                                float prev = cTile.GetValue(cOff);
                                float bj = bVec.GetValue(j);
                                prev += aVal * bj;
                                cTile.SetValue(cOff, prev);
                            }
                        }
                        bVecQ_.FreeTensor(bVec);
                    }
                }

                // Add bias and apply fused scaling residual: out = c * s + c = c * (1 + s)
                const float alpha = 1.0f + s;

                // Prepare bias slice b[nStart:nStart+nLen]
                AscendC::LocalTensor<float> bVec = bVecQ_.AllocTensor<float>();
                AscendC::DataCopy(bVec, bGm_[nStart], nLen);

                for (uint32_t i = 0; i < mLen; ++i) {
                    for (uint32_t j = 0; j < nLen; ++j) {
                        const uint64_t cOff = (uint64_t)i * tn_ + j;
                        float v = cTile.GetValue(cOff);
                        v = v + bVec.GetValue(j);
                        float outv = v * alpha;
                        // Write to GM y at [mStart+i, nStart+j]
                        const uint64_t yIdx = (uint64_t)(mStart + i) * N_ + (nStart + j);
                        AscendC::LocalTensor<float> tmp1 = bVecQ_.AllocTensor<float>();
                        tmp1.SetValue(0, outv);
                        AscendC::DataCopy(yGm_[yIdx], tmp1, 1);
                        bVecQ_.FreeTensor(tmp1);
                    }
                }

                bVecQ_.FreeTensor(bVec);
                cTileQ_.FreeTensor(cTile);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> cTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bVecQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> sGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tm_{0}, tn_{0}, tk_{0};
};

extern "C" __global__ __aicore__ void matmul_scaling_residual_add_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScalingResidualAddCustom op;
    op.Init(x, w, b, scaling, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tm, tiling_data.tn, tiling_data.tk);
    op.Process();
}
