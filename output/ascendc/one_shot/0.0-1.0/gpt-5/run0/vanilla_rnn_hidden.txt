static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    VanillaRnnHiddenCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);     // [T,B,I]
    const gert::Shape *w_i2h_shape = context->GetInputShape(2); // [H,K]
    const gert::Shape *w_h2o_shape = context->GetInputShape(4); // [O,H]

    const int64_t T = x_shape->GetOriginShape().GetDim(0);
    const int64_t B = x_shape->GetOriginShape().GetDim(1);
    const int64_t I = x_shape->GetOriginShape().GetDim(2);

    const int64_t H = w_i2h_shape->GetOriginShape().GetDim(0);
    const int64_t K = w_i2h_shape->GetOriginShape().GetDim(1);

    const int64_t O = w_h2o_shape->GetOriginShape().GetDim(0);
    const int64_t H2 = w_h2o_shape->GetOriginShape().GetDim(1);

    if (H != H2) {
        return ge::GRAPH_FAILED;
    }
    if (K != (I + H)) {
        // Combined input size must be I + H
        // We proceed conservatively; runtime may still fail if mismatched.
    }

    uint32_t blockDim = static_cast<uint32_t>(B);
    if (blockDim == 0u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_T(static_cast<uint32_t>(T));
    tiling.set_B(static_cast<uint32_t>(B));
    tiling.set_I(static_cast<uint32_t>(I));
    tiling.set_H(static_cast<uint32_t>(H));
    tiling.set_O(static_cast<uint32_t>(O));
    tiling.set_K(static_cast<uint32_t>(K));
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // Workspace holds two buffers of size [B,H] for ping-pong hidden state
    size_t *workspaces = context->GetWorkspaceSizes(1);
    const uint64_t ws_elems = static_cast<uint64_t>(B) * static_cast<uint64_t>(H) * 2ull;
    workspaces[0] = static_cast<size_t>(ws_elems * sizeof(float));

    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

static __aicore__ inline float fast_tanh(float x)
{
    // A simple rational approximation to avoid exp on device:
    // tanh(x) ~ x * (27 + x^2) / (27 + 9 x^2)
    float x2 = x * x;
    float num = x * (27.0f + x2);
    float den = (27.0f + 9.0f * x2);
    return num / den;
}

extern \"C\" __global__ __aicore__ void vanilla_rnn_hidden_custom(
    GM_ADDR x,       // [T,B,I]
    GM_ADDR h0,      // [B,H]
    GM_ADDR w_i2h,   // [H,K] (K = I+H)
    GM_ADDR b_i2h,   // [H]
    GM_ADDR w_h2o,   // [O,H]
    GM_ADDR b_h2o,   // [O]
    GM_ADDR y,       // [T,B,O]
    GM_ADDR workspace,
    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t T = tiling_data.T;
    const uint32_t B = tiling_data.B;
    const uint32_t I = tiling_data.I;
    const uint32_t H = tiling_data.H;
    const uint32_t O = tiling_data.O;
    const uint32_t K = tiling_data.K;

    __gm__ float* x_gm  = (__gm__ float*)x;
    __gm__ float* h0_gm = (__gm__ float*)h0;
    __gm__ float* wi2h_gm = (__gm__ float*)w_i2h;
    __gm__ float* bi2h_gm = (__gm__ float*)b_i2h;
    __gm__ float* who_gm  = (__gm__ float*)w_h2o;
    __gm__ float* bho_gm  = (__gm__ float*)b_h2o;
    __gm__ float* y_gm  = (__gm__ float*)y;
    __gm__ float* ws_gm = (__gm__ float*)workspace;

    const uint64_t x_stride_tb = (uint64_t)B * (uint64_t)I;
    const uint64_t y_stride_tb = (uint64_t)B * (uint64_t)O;

    const uint64_t hbuf_stride_b = (uint64_t)H;
    const uint64_t hbuf_bank_size = (uint64_t)B * (uint64_t)H;

    const uint32_t block_num = (uint32_t)AscendC::GetBlockNum();
    const uint32_t block_idx = (uint32_t)AscendC::GetBlockIdx();

    if (B == 0 || T == 0 || (I + H) != K) {
        return;
    }

    for (uint32_t b = block_idx; b < B; b += block_num) {
        // Initialize hidden state buffer for this batch index
        uint64_t h_base0 = (uint64_t)b * hbuf_stride_b;           // bank 0 base for this b
        uint64_t h_base1 = hbuf_bank_size + h_base0;              // bank 1 base for this b

        // Copy h0[b, :] -> ws_gm[h_base0 : h_base0+H]
        for (uint32_t j = 0; j < H; ++j) {
            ws_gm[h_base0 + j] = h0_gm[(uint64_t)b * (uint64_t)H + j];
        }

        for (uint32_t t = 0; t < T; ++t) {
            const uint64_t x_base = (uint64_t)t * x_stride_tb + (uint64_t)b * (uint64_t)I;
            const uint64_t y_base = (uint64_t)t * y_stride_tb + (uint64_t)b * (uint64_t)O;

            const uint64_t old_h_base = ((t & 1u) == 0u) ? h_base0 : h_base1;
            const uint64_t new_h_base = ((t & 1u) == 0u) ? h_base1 : h_base0;

            // Compute pre-activation for each hidden unit j, then apply tanh and write to new_h_base
            for (uint32_t j = 0; j < H; ++j) {
                float acc = bi2h_gm[j]; // bias for hidden unit j

                const uint64_t w_row_base = (uint64_t)j * (uint64_t)K;

                // Input contribution: sum_{i=0..I-1} x[t,b,i] * w_i2h[j, i]
                for (uint32_t i = 0; i < I; ++i) {
                    const float xi = x_gm[x_base + i];
                    const float wij = wi2h_gm[w_row_base + i];
                    acc += xi * wij;
                }

                // Hidden contribution: sum_{h=0..H-1} h_prev[h] * w_i2h[j, I + h]
                for (uint32_t hh = 0; hh < H; ++hh) {
                    const float hprev = ws_gm[old_h_base + hh];
                    const float wjh = wi2h_gm[w_row_base + (uint64_t)I + (uint64_t)hh];
                    acc += hprev * wjh;
                }

                // Activation
                ws_gm[new_h_base + j] = fast_tanh(acc);
            }

            // Output projection y_t[b, :] = h_t[b,:] @ w_h2o^T + b_h2o
            for (uint32_t o = 0; o < O; ++o) {
                float acc_o = bho_gm[o];
                const uint64_t w_ho_row_base = (uint64_t)o * (uint64_t)H;
                for (uint32_t hh = 0; hh < H; ++hh) {
                    const float hval = ws_gm[new_h_base + hh];
                    const float w_oh = who_gm[w_ho_row_base + hh];
                    acc_o += hval * w_oh;
                }
                y_gm[y_base + o] = acc_o;
            }
        }
    }
}
"""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor vanilla_rnn_hidden_impl_npu(const at::Tensor& x,
                                       const at::Tensor& h0,
                                       const at::Tensor& w_i2h,
                                       const at::Tensor& b_i2h,
                                       const at::Tensor& w_h2o,
                                       const at::Tensor& b_h2o) {
    // Output shape [T, B, O]
    TORCH_CHECK(x.dim() == 3, \"x must be [T,B,I]\");
    TORCH_CHECK(w_h2o.dim() == 2, \"w_h2o must be [O,H]\");
    auto T = x.size(0);
    auto B = x.size(1);
    auto O = w_h2o.size(0);
    at::Tensor result = at::empty({T, B, O}, x.options());
    EXEC_NPU_CMD(aclnnVanillaRnnHiddenCustom, x, h0, w_i2h, b_i2h, w_h2o, b_h2o, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("vanilla_rnn_hidden_custom", &vanilla_rnn_hidden_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("vanilla_rnn_hidden_custom", &vanilla_rnn_hidden_impl_npu,
          "Vanilla RNN hidden with output projection (VanillaRnnHiddenCustom on NPU)");
}
