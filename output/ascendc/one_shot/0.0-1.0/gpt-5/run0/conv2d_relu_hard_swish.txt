    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));

    const uint32_t W   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Kh   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw   = static_cast<uint32_t>(w_shape->GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stride_h = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride_w = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *pad_h    = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *pad_w    = attrs->GetAttrPointer<int32_t>(3);
    const int32_t *dil_h    = attrs->GetAttrPointer<int32_t>(4);
    const int32_t *dil_w    = attrs->GetAttrPointer<int32_t>(5);

    const uint32_t sH = static_cast<uint32_t>(*stride_h);
    const uint32_t sW = static_cast<uint32_t>(*stride_w);
    const uint32_t pH = static_cast<uint32_t>(*pad_h);
    const uint32_t pW = static_cast<uint32_t>(*pad_w);
    const uint32_t dH = static_cast<uint32_t>(*dil_h);
    const uint32_t dW = static_cast<uint32_t>(*dil_w);

    const uint32_t Hout = static_cast<uint32_t>((static_cast<int64_t>(H) + 2ll * pH - static_cast<int64_t>(dH) * (Kh - 1) - 1) / sH + 1);
    const uint32_t Wout = static_cast<uint32_t>((static_cast<int64_t>(W) + 2ll * pW - static_cast<int64_t>(dW) * (Kw - 1) - 1) / sW + 1);

    const uint32_t totalOut = N * Cout * Hout * Wout;

    // Determine if bias provided
    bool hasBias = false;
    if (context->GetInputsNum() >= 3) {
        const gert::Shape *b_shape = context->GetInputShape(2);
        if (b_shape != nullptr && b_shape->GetShapeSize() == Cout) {
            hasBias = true;
        }
    }

    // Simple tiling strategy on flattened output elements
    static constexpr uint32_t ALIGN_ELEMS = 256;
    static constexpr uint32_t HARD_CAP_ELEMS = 65536;

    uint32_t tileLen = HARD_CAP_ELEMS;
    if (totalOut < tileLen) {
        uint32_t aligned = AlignDown(totalOut, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOut;
    } else {
        tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOut, tileLen);
    uint32_t blockDim = numTiles;

    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);

    tiling.set_Wout(Wout);

    tiling.set_strideH(sH);
    tiling.set_strideW(sW);
    tiling.set_padH(pH);
    tiling.set_padW(pW);
    tiling.set_dilH(dH);
    tiling.set_dilW(dW);

    tiling.set_hasBias(hasBias ? 1u : 0u);

    tiling.set_totalOut(totalOut);
    tiling.set_tileLength(tileLen);

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const int64_t N = x_shape->GetDim(0);

    const int64_t Cout = w_shape->GetDim(0);
    const int64_t Kh = w_shape->GetDim(2);
    const int64_t Kw = w_shape->GetDim(3);

    int64_t sH = 1, sW = 1, pH = 0, pW = 0, dH = 1, dW = 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs != nullptr) {
        const int32_t *stride_h = attrs->GetAttrPointer<int32_t>(0);
        const int32_t *stride_w = attrs->GetAttrPointer<int32_t>(1);
        const int32_t *pad_h    = attrs->GetAttrPointer<int32_t>(2);
        const int32_t *pad_w    = attrs->GetAttrPointer<int32_t>(3);
        const int32_t *dil_h    = attrs->GetAttrPointer<int32_t>(4);
        const int32_t *dil_w    = attrs->GetAttrPointer<int32_t>(5);
        if (stride_h) sH = *stride_h;
        if (stride_w) sW = *stride_w;
        if (pad_h)    pH = *pad_h;
        if (pad_w)    pW = *pad_w;
        if (dil_h)    dH = *dil_h;
        if (dil_w)    dW = *dil_w;
    }

    const int64_t Hout = (H + 2 * pH - dH * (Kh - 1) - 1) / sH + 1;
    const int64_t Wout = (W + 2 * pW - dW * (Kw - 1) - 1) / sW + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Hout);
    y_shape->SetDim(3, Wout);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dReluHardSwishCustom {
public:
    __aicore__ inline KernelConv2dReluHardSwishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, 
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t Kh, uint32_t Kw,
                                uint32_t Hout, uint32_t Wout,
                                uint32_t sH, uint32_t sW, uint32_t pH, uint32_t pW,
                                uint32_t dH, uint32_t dW,
                                uint32_t hasBias,
                                uint32_t totalOut, uint32_t tileLength)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; Kh_ = Kh; Kw_ = Kw;
        Hout_ = Hout; Wout_ = Wout;
        sH_ = sH; sW_ = sW; pH_ = pH; pW_ = pW; dH_ = dH; dW_ = dW;
        hasBias_ = hasBias;
        totalOut_ = totalOut;
        tileLen_ = tileLength;

        const uint32_t xSize = N_ * Cin_ * H_ * W_;
        const uint32_t wSize = Cout_ * Cin_ * Kh_ * Kw_;
        const uint32_t bSize = Cout_;
        const uint32_t ySize = totalOut_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        if (hasBias_) { bGm_.SetGlobalBuffer((__gm__ float*)b, bSize); }
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOut_) return;

        uint32_t curLen = totalOut_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        while (true) {
            ComputeStore(base, curLen);

            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalOut_) break;
            base = nextBase;
            curLen = totalOut_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
        }
    }

private:
    __aicore__ inline float relu(float v) const { return v > 0.f ? v : 0.f; }

    __aicore__ inline float hswish_from_relu(float r) const {
        // r is already ReLU output (>=0). h = min(max((r+3)/6, 0), 1)
        float h = (r + 3.0f) * (1.0f / 6.0f);
        if (h < 0.f) h = 0.f;
        if (h > 1.f) h = 1.f;
        return r * h;
    }

    __aicore__ inline void decode_index(uint32_t p, uint32_t &n, uint32_t &co, uint32_t &oh, uint32_t &ow) const
    {
        const uint32_t HWout = Hout_ * Wout_;
        const uint32_t CHWout = Cout_ * HWout;

        n  = p / CHWout;
        uint32_t r1 = p - n * CHWout;
        co = r1 / HWout;
        uint32_t r2 = r1 - co * HWout;
        oh = r2 / Wout_;
        ow = r2 - oh * Wout_;
    }

    __aicore__ inline uint32_t x_index(uint32_t n, uint32_t ci, uint32_t ih, uint32_t iw) const
    {
        return ((n * Cin_ + ci) * H_ + ih) * W_ + iw;
    }

    __aicore__ inline uint32_t w_index(uint32_t co, uint32_t ci, uint32_t kh, uint32_t kw) const
    {
        return ((co * Cin_ + ci) * Kh_ + kh) * Kw_ + kw;
    }

    __aicore__ inline void ComputeStore(uint32_t base, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            const uint32_t p = base + i;
            uint32_t n, co, oh, ow;
            decode_index(p, n, co, oh, ow);

            float acc = hasBias_ ? bGm_[co] : 0.f;

            // Convolution inner product for this output element
            const int32_t in_h_base = static_cast<int32_t>(oh) * static_cast<int32_t>(sH_) - static_cast<int32_t>(pH_);
            const int32_t in_w_base = static_cast<int32_t>(ow) * static_cast<int32_t>(sW_) - static_cast<int32_t>(pW_);

            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const int32_t ih = in_h_base + static_cast<int32_t>(kh) * static_cast<int32_t>(dH_);
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const int32_t iw = in_w_base + static_cast<int32_t>(kw) * static_cast<int32_t>(dW_);
                        // no implicit padding reads: skip if out of bounds
                        if (ih >= 0 && ih < static_cast<int32_t>(H_) && iw >= 0 && iw < static_cast<int32_t>(W_)) {
                            float xv = xGm_[x_index(n, ci, static_cast<uint32_t>(ih), static_cast<uint32_t>(iw))];
                            float wv = wGm_[w_index(co, ci, kh, kw)];
                            acc += xv * wv;
                        }
                    }
                }
            }

            float r = relu(acc);
            float y = hswish_from_relu(r);
            yGm_[p] = y;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, Kh_{0}, Kw_{0};
    uint32_t Hout_{0}, Wout_{0};
    uint32_t sH_{1}, sW_{1}, pH_{0}, pW_{0}, dH_{1}, dW_{1};
    uint32_t hasBias_{0};
    uint32_t totalOut_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluHardSwishCustom op;
    op.Init(
        x, w, b, y,
        tiling_data.N, tiling_data.Cin, tiling_data.Cout,
        tiling_data.H, tiling_data.W, tiling_data.Kh, tiling_data.Kw,
        tiling_data.Hout, tiling_data.Wout,
        tiling_data.strideH, tiling_data.strideW,
        tiling_data.padH, tiling_data.padW,
        tiling_data.dilH, tiling_data.dilW,
        tiling_data.hasBias,
        tiling_data.totalOut, tiling_data.tileLength
    );
    op.Process();
}
