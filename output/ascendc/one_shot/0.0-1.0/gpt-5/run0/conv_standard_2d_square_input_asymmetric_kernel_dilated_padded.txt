static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x = context->GetInputShape(0);
    const gert::Shape *w = context->GetInputShape(1);

    const int64_t N  = x->GetDim(0);
    const int64_t Ci = x->GetDim(1);
    const int64_t H  = x->GetDim(2);
    const int64_t W  = x->GetDim(3);

    const int64_t Co = w->GetDim(0);
    const int64_t Kh = w->GetDim(2);
    const int64_t Kw = w->GetDim(3);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t sh = static_cast<int64_t>(*attrs->GetAttrPointer<int32_t>(0));
    const int64_t sw = static_cast<int64_t>(*attrs->GetAttrPointer<int32_t>(1));
    const int64_t pt = static_cast<int64_t>(*attrs->GetAttrPointer<int32_t>(2));
    const int64_t pb = static_cast<int64_t>(*attrs->GetAttrPointer<int32_t>(3));
    const int64_t pl = static_cast<int64_t>(*attrs->GetAttrPointer<int32_t>(4));
    const int64_t pr = static_cast<int64_t>(*attrs->GetAttrPointer<int32_t>(5));
    const int64_t dh = static_cast<int64_t>(*attrs->GetAttrPointer<int32_t>(6));
    const int64_t dw = static_cast<int64_t>(*attrs->GetAttrPointer<int32_t>(7));
    const int64_t groups = static_cast<int64_t>(*attrs->GetAttrPointer<int32_t>(8));

    if (groups != 1 || Ci != w->GetDim(1)) {
        return GRAPH_FAILED;
    }

    const int64_t effKh = (Kh - 1) * dh + 1;
    const int64_t effKw = (Kw - 1) * dw + 1;

    const int64_t Ho = (H + pt + pb - effKh) / sh + 1;
    const int64_t Wo = (W + pl + pr - effKw) / sw + 1;

    gert::Shape *y = context->GetOutputShape(0);
    y->SetDimNum(4);
    y->SetDim(0, N);
    y->SetDim(1, Co);
    y->SetDim(2, Ho);
    y->SetDim(3, Wo);
    return GRAPH_SUCCESS;
}

#include "kernel_operator.h"

class KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t wi,
                                uint32_t cOut, uint32_t kH, uint32_t kW,
                                uint32_t outH, uint32_t outW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padT, uint32_t padB, uint32_t padL, uint32_t padR,
                                uint32_t dilH, uint32_t dilW, uint32_t groups,
                                uint32_t tileOW, uint32_t tileCO, uint32_t tileCI)
    {
        n_ = n; cIn_ = cIn; h_ = h; w_ = wi;
        cOut_ = cOut; kH_ = kH; kW_ = kW;
        outH_ = outH; outW_ = outW;

        sh_ = strideH; sw_ = strideW;
        pt_ = padT; pb_ = padB; pl_ = padL; pr_ = padR;
        dh_ = dilH; dw_ = dilW;
        groups_ = groups;

        tileOW_ = tileOW; tileCO_ = tileCO; tileCI_ = tileCI;

        const uint32_t xSize = n_ * cIn_ * h_ * w_;
        const uint32_t wSize = cOut_ * cIn_ * kH_ * kW_;
        const uint32_t ySize = n_ * cOut_ * outH_ * outW_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // UB buffer for one [tileCO x tileOW] output tile
        const uint32_t outTileElems = tileCO_ * tileOW_;
        pipe_.InitBuffer(yTileQueue_, 1, outTileElems * sizeof(float));
        // small temp to stage weights for a co-tile across one (ci, kh, kw)
        pipe_.InitBuffer(tmpWeightsQueue_, 1, tileCO_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cIn_ == 0 || cOut_ == 0 || outH_ == 0 || outW_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalBlocks = n_ * outH_;
        if (blockIdx >= totalBlocks) return;

        const uint32_t nIdx = blockIdx / outH_;
        const uint32_t ohIdx = blockIdx % outH_;

        for (uint32_t owBase = 0; owBase < outW_; owBase += tileOW_) {
            const uint32_t curOW = (owBase + tileOW_ <= outW_) ? tileOW_ : (outW_ - owBase);

            for (uint32_t coBase = 0; coBase < cOut_; coBase += tileCO_) {
                const uint32_t curCO = (coBase + tileCO_ <= cOut_) ? tileCO_ : (cOut_ - coBase);

                AscendC::LocalTensor<float> yTile = yTileQueue_.AllocTensor<float>();
                // Zero initialize tile
                AscendC::Dup(yTile, 0.0f, curCO * curOW);

                // Accumulate over input channels and kernel window
                for (uint32_t ci = 0; ci < cIn_; ci += 1) {
                    const uint32_t curCI = 1u;
                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        const int32_t in_h = static_cast<int32_t>(ohIdx) * static_cast<int32_t>(sh_) - static_cast<int32_t>(pt_) + static_cast<int32_t>(kh) * static_cast<int32_t>(dh_);
                        const bool h_valid = (in_h >= 0) && (in_h < static_cast<int32_t>(h_));
                        for (uint32_t kw = 0; kw < kW_; ++kw) {
                            // Stage weights for this (ci,kh,kw) across curCO
                            AscendC::LocalTensor<float> wTile = tmpWeightsQueue_.AllocTensor<float>();
                            for (uint32_t coOff = 0; coOff < curCO; ++coOff) {
                                const uint32_t co = coBase + coOff;
                                const uint32_t wIdx = IndexW(co, ci, kh, kw);
                                // load scalar weight to UB slot
                                // Using DataCopy for one element for clarity
                                AscendC::DataCopy(wTile[coOff], wGm_[wIdx], 1);
                            }

                            for (uint32_t owOff = 0; owOff < curOW; ++owOff) {
                                const uint32_t ow = owBase + owOff;
                                const int32_t in_w = static_cast<int32_t>(ow) * static_cast<int32_t>(sw_) - static_cast<int32_t>(pl_) + static_cast<int32_t>(kw) * static_cast<int32_t>(dw_);
                                const bool w_valid = (in_w >= 0) && (in_w < static_cast<int32_t>(w_));
                                float xval = 0.0f;
                                if (h_valid && w_valid) {
                                    const uint32_t xIdx = IndexX(nIdx, ci, static_cast<uint32_t>(in_h), static_cast<uint32_t>(in_w));
                                    // copy one element from GM to a temp scalar local
                                    AscendC::DataCopy(xScalar_, xGm_[xIdx], 1);
                                    xval = xScalar_.GetValue(0);
                                }
                                if (xval != 0.0f) {
                                    for (uint32_t coOff = 0; coOff < curCO; ++coOff) {
                                        const float wv = wTile.GetValue(coOff);
                                        const uint32_t yIndex = coOff * curOW + owOff;
                                        const float accum = yTile.GetValue(yIndex) + xval * wv;
                                        yTile.SetValue(yIndex, accum);
                                    }
                                }
                            }
                            tmpWeightsQueue_.FreeTensor(wTile);
                        }
                    }
                }

                // Write back yTile to GM for each output channel in the tile
                for (uint32_t coOff = 0; coOff < curCO; ++coOff) {
                    const uint32_t co = coBase + coOff;
                    const uint32_t yBaseIdx = IndexY(nIdx, co, ohIdx, owBase);
                    // Create a local view for this channel slice
                    AscendC::LocalTensor<float> ySlice = yTile[curOW * coOff];
                    AscendC::DataCopy(yGm_[yBaseIdx], ySlice, curOW);
                }

                yTileQueue_.FreeTensor(yTile);
            }
        }
    }

private:
    __aicore__ inline uint32_t IndexX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((n * cIn_ + c) * h_ + h) * w_ + w);
    }
    __aicore__ inline uint32_t IndexW(uint32_t co, uint32_t ci, uint32_t kh, uint32_t kw) const
    {
        return (((co * cIn_ + ci) * kH_ + kh) * kW_ + kw);
    }
    __aicore__ inline uint32_t IndexY(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((n * cOut_ + c) * outH_ + h) * outW_ + w);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yTileQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpWeightsQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    // scalar staging buffer for input element
    AscendC::LocalTensor<float> xScalar_;

    uint32_t n_{0}, cIn_{0}, h_{0}, w_{0};
    uint32_t cOut_{0}, kH_{0}, kW_{0};
    uint32_t outH_{0}, outW_{0};

    uint32_t sh_{1}, sw_{1};
    uint32_t pt_{0}, pb_{0}, pl_{0}, pr_{0};
    uint32_t dh_{1}, dw_{1};
    uint32_t groups_{1};

    uint32_t tileOW_{1}, tileCO_{1}, tileCI_{1};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom op;
    op.Init(
        x, weight, y,
        tiling_data.n, tiling_data.cIn, tiling_data.h, tiling_data.w,
        tiling_data.cOut, tiling_data.kH, tiling_data.kW,
        tiling_data.outH, tiling_data.outW,
        tiling_data.strideH, tiling_data.strideW,
        tiling_data.padT, tiling_data.padB, tiling_data.padL, tiling_data.padR,
        tiling_data.dilH, tiling_data.dilW, tiling_data.groups,
        tiling_data.tileOW, tiling_data.tileCO, tiling_data.tileCI);
    op.Process();
}
