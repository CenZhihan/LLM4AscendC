kernel_src="""

#include "kernel_operator.h"

// Fused kernel for:
// y = logsumexp(logsumexp(mean(max(sum(x @ w^T + b, dim=1, keepdim=True), dim=1, keepdim=True), dim=1, keepdim=True), dim=1, keepdim=True), dim=1, keepdim=True)
// which reduces to y[b] = sum_j ( x[b] · w[j]^T + b[j] ) for each batch b.
// Output shape: [B, 1] (stored as B contiguous elements).

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom {
public:
    __aicore__ inline KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batch, uint32_t inFeatures,
                                uint32_t outFeatures, uint32_t tileK)
    {
        batch_ = batch;
        in_ = inFeatures;
        out_ = outFeatures;
        tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(batch_) * in_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(out_) * in_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, out_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch_);

        pipe_.InitBuffer(qX_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(qW_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(qWS_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(qTMP_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(qScalar_, 2, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || in_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over batch
        for (uint32_t b = blockIdx; b < batch_; b += blockNum) {
            ComputeRow(b);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t bRow)
    {
        // Compute bias sum: sum_j b[j]
        bool biasInit = false;
        AscendC::LocalTensor<float> bsumLocal;
        for (uint32_t o = 0; o < out_; ) {
            const uint32_t len = (out_ - o) > tileK_ ? tileK_ : (out_ - o);
            // Load bias segment
            AscendC::LocalTensor<float> bSeg = qTMP_.AllocTensor<float>();
            AscendC::DataCopy(bSeg, bGm_[o], len);

            // Reduce sum of this segment to scalar
            AscendC::LocalTensor<float> bPart = qScalar_.AllocTensor<float>();
            AscendC::ReduceSum(bPart, bSeg, len);

            if (!biasInit) {
                // First segment initializes bsum
                bsumLocal = bPart;
                biasInit = true;
            } else {
                AscendC::Add(bsumLocal, bsumLocal, bPart, 1);
                qScalar_.FreeTensor(bPart);
            }
            qTMP_.FreeTensor(bSeg);
            o += len;
        }

        // Compute s = sum_i x[b,i] * (sum_j w[j,i])
        bool sumInit = false;
        AscendC::LocalTensor<float> sumLocal;
        for (uint32_t k = 0; k < in_; ) {
            const uint32_t klen = (in_ - k) > tileK_ ? tileK_ : (in_ - k);

            // wsSeg = sum over out rows of w row segments
            AscendC::LocalTensor<float> wsSeg = qWS_.AllocTensor<float>();
            bool wsInit = false;

            for (uint32_t o = 0; o < out_; ++o) {
                AscendC::LocalTensor<float> wSeg = qW_.AllocTensor<float>();
                // w is stored row-major by out, then in
                const uint64_t wOffset = static_cast<uint64_t>(o) * in_ + k;
                AscendC::DataCopy(wSeg, wGm_[wOffset], klen);
                if (!wsInit) {
                    AscendC::DataCopy(wsSeg, wSeg, klen);
                    wsInit = true;
                } else {
                    AscendC::Add(wsSeg, wsSeg, wSeg, klen);
                }
                qW_.FreeTensor(wSeg);
            }

            // Load x segment for row bRow
            AscendC::LocalTensor<float> xSeg = qX_.AllocTensor<float>();
            const uint64_t xOffset = static_cast<uint64_t>(bRow) * in_ + k;
            AscendC::DataCopy(xSeg, xGm_[xOffset], klen);

            // tmp = xSeg * wsSeg
            AscendC::LocalTensor<float> tmpSeg = qTMP_.AllocTensor<float>();
            AscendC::Muls(tmpSeg, xSeg, 1.0f, klen); // tmp = xSeg
            AscendC::Mul(tmpSeg, tmpSeg, wsSeg, klen); // tmp *= wsSeg

            // Reduce tmpSeg to scalar partial
            AscendC::LocalTensor<float> partLocal = qScalar_.AllocTensor<float>();
            AscendC::ReduceSum(partLocal, tmpSeg, klen);

            if (!sumInit) {
                sumLocal = partLocal;
                sumInit = true;
            } else {
                AscendC::Add(sumLocal, sumLocal, partLocal, 1);
                qScalar_.FreeTensor(partLocal);
            }

            qTMP_.FreeTensor(tmpSeg);
            qX_.FreeTensor(xSeg);
            qWS_.FreeTensor(wsSeg);

            k += klen;
        }

        // sumLocal += bsumLocal
        AscendC::Add(sumLocal, sumLocal, bsumLocal, 1);

        // Store result y[bRow]
        AscendC::DataCopy(yGm_[bRow], sumLocal, 1);

        // Free scalars that were not freed
        if (sumLocal.GetPtr() != nullptr) {
            qScalar_.FreeTensor(sumLocal);
        }
        if (bsumLocal.GetPtr() != nullptr) {
            qScalar_.FreeTensor(bsumLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>        qWS_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>        qTMP_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2>         qScalar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_{0};
    uint32_t in_{0};
    uint32_t out_{0};
    uint32_t tileK_{0};
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom op;
    op.Init(x, w, b, y,
            tiling_data.batch,
            tiling_data.inFeatures,
            tiling_data.outFeatures,
            tiling_data.tileK);
    op.Process();
}
