host_operator_src="""

#include "cumsum_reverse_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB budget (conservative)
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumReverseCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const gert::OriginShape &orig = in_shape->GetOriginShape();
    const uint32_t rank = static_cast<uint32_t>(orig.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimPtr != nullptr) ? *dimPtr : 0;

    // Normalize dim to [0, rank)
    if (rank == 0) {
        return ge::GRAPH_FAILED;
    }
    if (dim < 0) dim += static_cast<int32_t>(rank);
    if (dim < 0) dim = 0;
    if (dim >= static_cast<int32_t>(rank)) dim = static_cast<int32_t>(rank - 1);

    uint64_t totalLength64 = 1;
    for (uint32_t i = 0; i < rank; ++i) {
        totalLength64 *= static_cast<uint64_t>(orig.GetDim(i));
    }
    const uint32_t totalLength = static_cast<uint32_t>(totalLength64);

    const uint32_t axisLength = static_cast<uint32_t>(orig.GetDim(dim));

    // Derive tile length from UB budget
    // We need 3 buffers of size tile: input, output, temp
    uint32_t maxTile = UB_BUDGET_BYTES / (3u * sizeof(float));  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (axisLength < tileLen) {
        const uint32_t aligned = AlignDown(axisLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : axisLength;
    }

    // Number of independent segments along the chosen axis (assuming last-dim fast path)
    uint64_t segmentCount64 = (axisLength == 0) ? 0 : (totalLength64 / axisLength);
    uint32_t segmentCount = static_cast<uint32_t>(segmentCount64);

    // Choose block dim based on number of segments
    uint32_t blockDim = (segmentCount == 0) ? 1u : segmentCount;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_axisLength(axisLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumsumReverseCustom : public OpDef {
public:
    explicit CumsumReverseCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumsumReverseCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 1;
constexpr int32_t TMP_BUF_NUM = 1;

class KernelCumsumReverseCustom {
public:
    __aicore__ inline KernelCumsumReverseCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t axisLength,
                                uint32_t tileLength, int32_t dim)
    {
        (void)dim; // current kernel assumes last-dimension fast path
        totalLen_ = totalLength;
        axisLen_  = axisLength;
        tileLen_  = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMP_BUF_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || axisLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t segmentCount = totalLen_ / axisLen_;
        if (segmentCount == 0) return;

        for (uint32_t seg = blockIdx; seg < segmentCount; seg += blockNum) {
            const uint32_t segBase = seg * axisLen_;
            uint32_t remaining = axisLen_;
            float carry = 0.0f;

            while (remaining > 0) {
                const uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
                const uint32_t gmOffset = segBase + (remaining - curLen);

                CopyIn(gmOffset, curLen);
                carry = Compute(curLen, carry);
                CopyOut(gmOffset, curLen);

                remaining -= curLen;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    // Performs reverse inclusive scan within tile, adds incoming carry to all elements,
    // returns updated carry for the next (left) tile.
    __aicore__ inline float Compute(uint32_t len, float carryIn)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // yLocal = xLocal
        AscendC::DataCopy(yLocal, xLocal, len);

        // Doubling steps for reverse scan: y[i] += y[i + step]
        for (uint32_t step = 1; step < len; step <<= 1) {
            const uint32_t work = len - step;
            AscendC::DataCopy(tLocal, yLocal[step], work);
            AscendC::Add(yLocal, yLocal, tLocal, work);
        }

        // Add carry from right tile if any
        if (carryIn != 0.0f) {
            AscendC::Adds(yLocal, yLocal, carryIn, len);
        }

        // The first element now holds the sum of this tile plus carryIn
        float tileTotal = yLocal.GetValue(0);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);

        return tileTotal;
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t axisLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverseCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.axisLength,
            tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
