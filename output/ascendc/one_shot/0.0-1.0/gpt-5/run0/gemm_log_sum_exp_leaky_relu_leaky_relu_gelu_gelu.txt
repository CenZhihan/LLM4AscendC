static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

// UB model (float):
//   xTile: tileK
//   wTile: tileM * tileK
//   acc:   tileM
//   tmp:   tileM
// Total floats ~ tileK + tileM*tileK + 2*tileM
// Budget bytes per core (conservative)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_K = 64;     // favor vector-friendly lengths
static constexpr uint32_t ALIGN_M = 64;
static constexpr uint32_t MIN_TILE_K = 64;
static constexpr uint32_t MIN_TILE_M = 64;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmLogSumExpLeakyReluLeakyReluGeluGeluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0); // [N, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [M, K]
    const gert::Shape *b_shape = context->GetInputShape(2); // [M]

    const uint32_t batchSize = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t inFeatures = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t outFeatures = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));

    // Derive tile sizes based on UB
    const uint32_t budgetFloats = UB_BUDGET_BYTES / 4;
    // Try to pick tileK first
    uint32_t tileK = inFeatures;
    if (tileK > 4096u) tileK = 4096u;
    tileK = AlignDown(tileK, ALIGN_K);
    if (tileK < MIN_TILE_K) tileK = MIN_TILE_K;
    if (tileK > inFeatures) tileK = AlignDown(inFeatures, ALIGN_K) > 0 ? AlignDown(inFeatures, ALIGN_K) : inFeatures;

    // Given tileK, pick tileM such that tileM*tileK + tileK + 2*tileM <= budgetFloats
    uint32_t maxTileM = 0;
    if (tileK > 0) {
        uint32_t remain = budgetFloats > tileK ? budgetFloats - tileK : budgetFloats;
        // tileM * (tileK + 2) <= remain
        maxTileM = remain / (tileK + 2u);
        maxTileM = AlignDown(maxTileM, ALIGN_M);
        if (maxTileM < MIN_TILE_M) maxTileM = MIN_TILE_M;
    }
    if (maxTileM > outFeatures) {
        maxTileM = AlignDown(outFeatures, ALIGN_M) > 0 ? AlignDown(outFeatures, ALIGN_M) : outFeatures;
    }
    if (maxTileM == 0) maxTileM = outFeatures > 0 ? outFeatures : 1;

    // Grid along batch rows
    uint32_t numTilesN = batchSize;
    uint32_t blockDim = numTilesN;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_inFeatures(inFeatures);
    tiling.set_outFeatures(outFeatures);
    tiling.set_tileK(tileK);
    tiling.set_tileM(maxTileM);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEGATIVE_SLOPE = 0.01f;

class KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom {
public:
    __aicore__ inline KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t K, uint32_t M,
                                uint32_t tileK, uint32_t tileM)
    {
        N_ = N; K_ = K; M_ = M;
        tileK_ = tileK; tileM_ = tileM;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * (uint64_t)K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)M_ * (uint64_t)K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)M_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * 1u);

        // Buffers
        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1, tileM_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || K_ == 0 || M_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        for (uint32_t row = blockIdx; row < N_; row += blockNum) {
            float lse = ComputeRow(row);
            // activation chain: two leaky_relu, two gelu
            float v = lse;
            v = LeakyRelu(v);
            v = LeakyRelu(v);
            v = Gelu(v);
            v = Gelu(v);
            // write output
            yGm_[row * 1u + 0] = v;
        }
    }

private:
    __aicore__ inline float LeakyRelu(float x) const
    {
        return x >= 0.0f ? x : (NEGATIVE_SLOPE * x);
    }

    __aicore__ inline float Gelu(float x) const
    {
        // tanh approximation: 0.5*x*(1 + tanh(sqrt(2/pi)*(x + 0.044715*x^3)))
        const float kAlpha = 0.7978845608f;  // sqrt(2/pi)
        float x3 = x * x * x;
        float inner = kAlpha * (x + 0.044715f * x3);
        // use AscendC math if available; fallback to device lib
        float t = AscendC::Tanh(inner);
        return 0.5f * x * (1.0f + t);
    }

    __aicore__ inline float ComputeRow(uint32_t row)
    {
        // We perform two passes across M: first to get max(z), then to get sum(exp(z - max))
        float globalMax = -INFINITY;

        // Pass 1: compute z = xW^T + b and track max
        for (uint32_t mBase = 0; mBase < M_; mBase += tileM_) {
            const uint32_t curM = (mBase + tileM_ <= M_) ? tileM_ : (M_ - mBase);

            AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();
            // zero accumulators
            AscendC::Duplicate(acc, 0.0f, curM);

            for (uint32_t kBase = 0; kBase < K_; kBase += tileK_) {
                const uint32_t curK = (kBase + tileK_ <= K_) ? tileK_ : (K_ - kBase);

                // load x tile [curK]
                AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[row * K_ + kBase], curK);
                xQueue_.EnQue(xLocal);

                // load w tile [curM, curK]
                AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                // contiguous block since W is [M,K], tile is rows mBase.. and cols kBase..
                AscendC::DataCopy(wLocal, wGm_[mBase * K_ + kBase], curM * curK);
                wQueue_.EnQue(wLocal);

                // compute acc[m] += dot(xLocal, wRow)
                xLocal = xQueue_.DeQue<float>();
                wLocal = wQueue_.DeQue<float>();
                // iterate over each m in tile (uses device-side loops)
                for (uint32_t m = 0; m < curM; ++m) {
                    float sumv = 0.0f;
                    // vector dot for curK
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        float xv = xLocal.GetValue(kk);
                        float wv = wLocal.GetValue(m * curK + kk);
                        sumv = AscendC::__vadd(sumv, AscendC::__vmul(xv, wv));
                    }
                    float prev = acc.GetValue(m);
                    acc.SetValue(m, prev + sumv);
                }
                wQueue_.FreeTensor(wLocal);
                xQueue_.FreeTensor(xLocal);
            }

            // add bias and update global max
            for (uint32_t m = 0; m < curM; ++m) {
                float zv = acc.GetValue(m) + bGm_[mBase + m];
                acc.SetValue(m, zv);
                if (zv > globalMax) globalMax = zv;
            }
            accQueue_.FreeTensor(acc);
        }

        if (!AscendC::IsFinite(globalMax)) {
            globalMax = 0.0f; // fallback safety
        }

        // Pass 2: recompute z and accumulate sum(exp(z - max))
        float sumExp = 0.0f;
        for (uint32_t mBase = 0; mBase < M_; mBase += tileM_) {
            const uint32_t curM = (mBase + tileM_ <= M_) ? tileM_ : (M_ - mBase);

            AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();
            AscendC::Duplicate(acc, 0.0f, curM);

            for (uint32_t kBase = 0; kBase < K_; kBase += tileK_) {
                const uint32_t curK = (kBase + tileK_ <= K_) ? tileK_ : (K_ - kBase);

                AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[row * K_ + kBase], curK);
                xQueue_.EnQue(xLocal);

                AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                AscendC::DataCopy(wLocal, wGm_[mBase * K_ + kBase], curM * curK);
                wQueue_.EnQue(wLocal);

                xLocal = xQueue_.DeQue<float>();
                wLocal = wQueue_.DeQue<float>();
                for (uint32_t m = 0; m < curM; ++m) {
                    float sumv = 0.0f;
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        float xv = xLocal.GetValue(kk);
                        float wv = wLocal.GetValue(m * curK + kk);
                        sumv = AscendC::__vadd(sumv, AscendC::__vmul(xv, wv));
                    }
                    float prev = acc.GetValue(m);
                    acc.SetValue(m, prev + sumv);
                }
                wQueue_.FreeTensor(wLocal);
                xQueue_.FreeTensor(xLocal);
            }

            for (uint32_t m = 0; m < curM; ++m) {
                float zv = acc.GetValue(m) + bGm_[mBase + m];
                float ev = AscendC::Exp(zv - globalMax);
                sumExp += ev;
            }
            accQueue_.FreeTensor(acc);
        }

        float lse = AscendC::Log(sumExp) + globalMax;
        return lse;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>        accQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>        tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, K_{0}, M_{0};
    uint32_t tileK_{0}, tileM_{0};
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.batchSize,
            tiling_data.inFeatures,
            tiling_data.outFeatures,
            tiling_data.tileK,
            tiling_data.tileM);
    op.Process();
}
