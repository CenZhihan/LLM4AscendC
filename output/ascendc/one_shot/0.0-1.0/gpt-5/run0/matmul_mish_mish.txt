host_operator_src="""
#include "matmul_mish_mish_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (b == 0) ? 0 : (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// Memory budget and alignment heuristics (per-core UB share).
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ALIGN_ELEMS    = 256;          // favor DMA alignment
static constexpr uint32_t MAX_O_TILE     = 128;          // must match kernel guard
static constexpr uint32_t MIN_K_CHUNK    = 256;          // lower bound on K chunk

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMishMishCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    const gert::Shape *bShape = context->GetInputShape(2);

    const auto &xOrigin = xShape->GetOriginShape();
    const auto &wOrigin = wShape->GetOriginShape();
    const auto &bOrigin = bShape->GetOriginShape();

    const uint32_t dimX = static_cast<uint32_t>(xOrigin.GetDimNum());
    const uint32_t dimW = static_cast<uint32_t>(wOrigin.GetDimNum());

    const uint32_t B = (dimX >= 1) ? static_cast<uint32_t>(xOrigin.GetDim(0)) : 1u;
    const uint32_t K = (dimX >= 2) ? static_cast<uint32_t>(xOrigin.GetDim(dimX - 1)) : 1u;

    const uint32_t O = (dimW >= 1) ? static_cast<uint32_t>(wOrigin.GetDim(0)) : 1u;
    const uint32_t K2 = (dimW >= 2) ? static_cast<uint32_t>(wOrigin.GetDim(1)) : 1u;
    (void)bOrigin; // bias inferred as O-sized

    // Basic guards (assume valid shapes; keep going even if mismatch)
    const uint32_t Kuse = (K == K2) ? K : K2;

    // Choose oTile (outputs per inner tile).
    uint32_t oTile = O;
    if (oTile > MAX_O_TILE) oTile = MAX_O_TILE;
    if (oTile == 0) oTile = 1;

    // UB budget for: x_chunk (Kc), w_chunk (Kc), y/e scratch (2 * oTile)
    // bytes_per_elem = 4; totalElems = 2*Kc + 2*oTile + safety
    uint32_t budgetElems = UB_BUDGET_BYTES / 4;
    uint32_t safety = 256;
    uint32_t kChunk = 0;
    if (budgetElems > (2 * oTile + safety)) {
        kChunk = (budgetElems - (2 * oTile + safety)) / 2;
        kChunk = AlignDown(kChunk, ALIGN_ELEMS);
    }
    if (kChunk < MIN_K_CHUNK) kChunk = MIN_K_CHUNK;
    if (kChunk > Kuse) kChunk = AlignDown(Kuse, ALIGN_ELEMS) > 0 ? AlignDown(Kuse, ALIGN_ELEMS) : Kuse;

    // Blocks: spread across batch dimension
    uint32_t blockDim = (B > 0) ? B : 1u;
    if (blockDim < 8u)  blockDim = blockDim; // keep small if B small
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim == 0u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_B(B);
    tiling.set_K(Kuse);
    tiling.set_O(O);
    tiling.set_kChunk(kChunk);
    tiling.set_oTile(oTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    const auto &xOrigin = xShape->GetOriginShape();
    const auto &wOrigin = wShape->GetOriginShape();

    gert::Shape *yShape = context->GetOutputShape(0);

    std::vector<int64_t> yDims;
    const int64_t B = (xOrigin.GetDimNum() >= 1) ? xOrigin.GetDim(0) : 1;
    const int64_t O = (wOrigin.GetDimNum() >= 1) ? wOrigin.GetDim(0) : 1;

    if (xOrigin.GetDimNum() == 1) {
        // Treat 1D x as [1, K]
        yDims = {1, O};
    } else {
        // Keep leading dims of x except the last, then O
        if (xOrigin.GetDimNum() >= 2) {
            // Assume 2D: [B, K] -> [B, O]
            yDims = {B, O};
        } else {
            yDims = {B, O};
        }
    }
    yShape->SetDimNum(static_cast<uint32_t>(yDims.size()));
    for (size_t i = 0; i < yDims.size(); ++i) {
        yShape->SetDim(i, yDims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType xType = context->GetInputDataType(0);
    context->SetOutputDataType(0, xType);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t MAX_O_TILE = 128;

class KernelMatmulMishMishCustom {
public:
    __aicore__ inline KernelMatmulMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t B, uint32_t K, uint32_t O,
                                uint32_t kChunk, uint32_t oTile)
    {
        B_ = B; K_ = K; O_ = O;
        kChunk_ = kChunk;
        oTile_ = (oTile > MAX_O_TILE) ? MAX_O_TILE : oTile;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(B_ * K_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(O_ * K_));
        bGm_.SetGlobalBuffer((__gm__ float*)b, static_cast<uint32_t>(O_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(B_ * O_));

        // Buffers: x_chunk (Kc), w_chunk (Kc), yVec (oTile), tmpVec (oTile)
        pipe_.InitBuffer(xTmpQueue_, 1, kChunk_ * sizeof(float));
        pipe_.InitBuffer(wTmpQueue_, 1, kChunk_ * sizeof(float));
        pipe_.InitBuffer(vecQueue_, 2, oTile_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (B_ == 0 || K_ == 0 || O_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t b = blockIdx; b < B_; b += blockNum) {
            const uint32_t xRowBase = b * K_;
            const uint32_t yRowBase = b * O_;

            for (uint32_t oStart = 0; oStart < O_; oStart += oTile_) {
                const uint32_t oLen = (oStart + oTile_ <= O_) ? oTile_ : (O_ - oStart);

                float acc[MAX_O_TILE];
                for (uint32_t i = 0; i < oLen; ++i) acc[i] = 0.0f;

                for (uint32_t kStart = 0; kStart < K_; kStart += kChunk_) {
                    const uint32_t kLen = ((kStart + kChunk_) <= K_) ? kChunk_ : (K_ - kStart);

                    AscendC::LocalTensor<float> xLocal = xTmpQueue_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[xRowBase + kStart], kLen);

                    for (uint32_t o = 0; o < oLen; ++o) {
                        const uint32_t wRowBase = (oStart + o) * K_;
                        AscendC::LocalTensor<float> wLocal = wTmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(wLocal, wGm_[wRowBase + kStart], kLen);

                        // Scalar FMA loop over kLen on UB
                        for (uint32_t t = 0; t < kLen; ++t) {
                            acc[o] += xLocal[t] * wLocal[t];
                        }

                        wTmpQueue_.FreeTensor(wLocal);
                    }

                    xTmpQueue_.FreeTensor(xLocal);
                }

                // Prepare yLocal := acc + bias
                AscendC::LocalTensor<float> yLocal = vecQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> bLocal = vecQueue_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[oStart], oLen);
                for (uint32_t o = 0; o < oLen; ++o) {
                    yLocal[o] = acc[o] + bLocal[o];
                }

                // First Mish: y = y * tanh(ln(1 + exp(y)))
                AscendC::LocalTensor<float> tmp = vecQueue_.AllocTensor<float>();
                AscendC::Exp(tmp, yLocal, oLen);               // tmp = exp(y)
                AscendC::Adds(tmp, tmp, 1.0f, oLen);           // tmp = exp(y) + 1
                AscendC::Ln(tmp, tmp, oLen);                   // tmp = ln(1 + exp(y))
                AscendC::Tanh(tmp, tmp, oLen);                 // tmp = tanh(softplus(y))
                AscendC::Mul(yLocal, yLocal, tmp, oLen);       // y = y * tmp

                // Second Mish
                AscendC::Exp(tmp, yLocal, oLen);               // tmp = exp(y)
                AscendC::Adds(tmp, tmp, 1.0f, oLen);           // tmp = exp(y) + 1
                AscendC::Ln(tmp, tmp, oLen);                   // tmp = ln(1 + exp(y))
                AscendC::Tanh(tmp, tmp, oLen);                 // tmp = tanh(softplus(y))
                AscendC::Mul(yLocal, yLocal, tmp, oLen);       // y = y * tmp

                // Store
                AscendC::DataCopy(yGm_[yRowBase + oStart], yLocal, oLen);

                vecQueue_.FreeTensor(tmp);
                vecQueue_.FreeTensor(bLocal);
                vecQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xTmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wTmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> vecQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t B_ {0}, K_ {0}, O_ {0};
    uint32_t kChunk_ {256};
    uint32_t oTile_ {64};
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMishMishCustom op;
    op.Init(x, w, b, y,
            tiling_data.B, tiling_data.K, tiling_data.O,
            tiling_data.kChunk, tiling_data.oTile);
    op.Process();
}
