kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t numGroups, uint32_t groupSize,
                                float eps, float htMin, float htMax)
    {
        N_ = N;
        C_ = C;
        numGroups_ = numGroups;
        groupSize_ = groupSize;
        eps_ = eps;
        htMin_ = htMin;
        htMax_ = htMax;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * C_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * C_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, C_);
        betaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, C_);

        // Buffers sized to one group tile
        const uint32_t bytes = groupSize_ * sizeof(float);
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, bytes);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, bytes);
        pipe_.InitBuffer(tmpQueue_, 2, bytes);      // scratch
        pipe_.InitBuffer(paramQueue_, 2, bytes);    // gamma/beta tiles
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || groupSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t totalTiles = static_cast<uint64_t>(N_) * static_cast<uint64_t>(numGroups_);

        for (uint64_t tile = blockIdx; tile < totalTiles; tile += blockNum) {
            const uint32_t n = static_cast<uint32_t>(tile / numGroups_);
            const uint32_t g = static_cast<uint32_t>(tile % numGroups_);
            const uint32_t c0 = g * groupSize_;
            const uint64_t rowBase = static_cast<uint64_t>(n) * C_;

            // Load data tile
            CopyIn(rowBase + c0, groupSize_);
            // Load params
            CopyParam(c0, groupSize_);

            // Compute fused: hardtanh -> mish -> groupnorm -> affine
            Compute(groupSize_);

            // Store
            CopyOut(rowBase + c0, groupSize_);
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyParam(uint32_t c0, uint32_t len)
    {
        AscendC::LocalTensor<float> gLocal = paramQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = paramQueue_.AllocTensor<float>();
        AscendC::DataCopy(gLocal, gammaGm_[c0], len);
        AscendC::DataCopy(bLocal, betaGm_[c0], len);
        paramQueue_.EnQue(gLocal);
        paramQueue_.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> t0 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> t1 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> gLocal = paramQueue_.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = paramQueue_.DeQue<float>();

        // Hardtanh clamp: x = min(max(x, htMin), htMax)
        AscendC::Maxs(t0, xLocal, htMin_, len);
        AscendC::Mins(t0, t0, htMax_, len);

        // Mish: x * tanh(softplus(x)); softplus(x) = ln(1 + exp(x))
        AscendC::Exp(t1, t0, len);                 // t1 = exp(t0)
        AscendC::Adds(t1, t1, 1.0f, len);          // t1 = t1 + 1
        AscendC::Ln(t1, t1, len);                  // t1 = ln(t1)
        AscendC::Tanh(t1, t1, len);                // t1 = tanh(softplus)
        AscendC::Mul(t0, t0, t1, len);             // t0 = mish(x)

        // GroupNorm over this tile (one group):
        // mean = sum(x)/len
        float mean = 0.0f;
        float var = 0.0f;
        {
            // sum
            float sumVal = 0.0f;
            float sumSqVal = 0.0f;
            // Accumulate in chunks for numerical stability
            const uint32_t VEC = 256;
            uint32_t done = 0;
            while (done < len) {
                uint32_t step = (len - done > VEC) ? VEC : (len - done);
                AscendC::LocalTensor<float> v = t0[done]; // alias view
                // Temporary buffer for reduction
                AscendC::LocalTensor<float> red = tmpQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> red2 = tmpQueue_.AllocTensor<float>();
                AscendC::Mul(red2, v, v, step);            // v^2
                // ReduceSum into first element of red/red2
                AscendC::ReduceSum(red, v, step);
                AscendC::ReduceSum(red2, red2, step);
                // Read back scalars
                sumVal += red.GetValue(0);
                sumSqVal += red2.GetValue(0);
                tmpQueue_.FreeTensor(red2);
                tmpQueue_.FreeTensor(red);
                done += step;
            }
            mean = sumVal / static_cast<float>(len);
            var = sumSqVal / static_cast<float>(len) - mean * mean;
        }
        float invStd = AscendC::Rsqrt(var + eps_);

        // y = ((x - mean) * invStd) * gamma + beta
        AscendC::Adds(t1, t0, -mean, len);        // t1 = x - mean
        AscendC::Muls(t1, t1, invStd, len);       // t1 = (x-mean)/std
        AscendC::Mul(t1, t1, gLocal, len);        // * gamma
        AscendC::Add(yLocal, t1, bLocal, len);    // + beta

        // queue maintenance
        paramQueue_.FreeTensor(bLocal);
        paramQueue_.FreeTensor(gLocal);
        tmpQueue_.FreeTensor(t1);
        tmpQueue_.FreeTensor(t0);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> paramQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;

    uint32_t N_{0};
    uint32_t C_{0};
    uint32_t numGroups_{1};
    uint32_t groupSize_{0};
    float eps_{1e-5f};
    float htMin_{-1.0f};
    float htMax_{1.0f};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(GM_ADDR x,
                                                                                     GM_ADDR gn_gamma,
                                                                                     GM_ADDR gn_beta,
                                                                                     GM_ADDR y,
                                                                                     GM_ADDR workspace,
                                                                                     GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, gn_gamma, gn_beta, y,
            tiling_data.N, tiling_data.C, tiling_data.numGroups, tiling_data.groupSize,
            tiling_data.eps, tiling_data.htMin, tiling_data.htMax);
    op.Process();
}
