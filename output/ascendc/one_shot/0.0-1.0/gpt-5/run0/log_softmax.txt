host_operator_src="""
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) { return static_cast<uint32_t>((a + b - 1ull) / b); }
} // namespace

namespace optiling {

// Memory model per-core (rough estimate):
//   inQueueX: 2 * tileInner
//   outQueueY: 2 * tileInner
//   tmpQueue: 2 * tileInner
// Total floats ~= 6 * tileInner => bytes ~= 24 * tileInner
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS     = 2048;         // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS  = 8192;         // contain per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t axis = (dimAttr != nullptr) ? *dimAttr : 1;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const gert::Shape &origin = in_shape->GetOriginShape();

    const int64_t rank = static_cast<int64_t>(origin.GetDimNum());
    int64_t normAxis = axis;
    if (normAxis < 0) normAxis += rank;
    if (normAxis < 0) normAxis = 0;
    if (normAxis >= rank) normAxis = rank - 1;

    uint64_t totalLength = 1;
    for (int64_t i = 0; i < rank; ++i) totalLength *= static_cast<uint64_t>(origin.GetDim(i));

    uint32_t innerLen = static_cast<uint32_t>(origin.GetDim(normAxis));
    uint32_t outerLen = static_cast<uint32_t>(totalLength / static_cast<uint64_t>(innerLen));

    // tileInner from UB budget: bytes = 24 * tileInner
    uint32_t maxTile = UB_BUDGET_BYTES / 24;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileInner = maxTile;
    if (innerLen < tileInner) {
        const uint32_t aligned = AlignDown(innerLen, ALIGN_ELEMS);
        tileInner = (aligned > 0) ? aligned : innerLen;
    }

    // Block dim based on number of rows (outerLen)
    uint32_t blockDim = outerLen;
    if (blockDim < 8u) blockDim = (outerLen == 0u) ? 1u : outerLen;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_innerLength(innerLen);
    tiling.set_outerLength(outerLen);
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(normAxis));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

            .Format({ge::FORMAT_ND})

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t innerLength,
                                uint32_t outerLength,
                                uint32_t tileInner)
    {
        totalLen_  = totalLength;
        innerLen_  = innerLength;
        outerLen_  = outerLength;
        tileInner_ = tileInner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerLen_ == 0 || outerLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < outerLen_; row += blockNum) {
            ComputeRow(row);
        }
    }

private:
    __aicore__ inline void LoadSeg(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void StoreSeg(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

    __aicore__ inline void ComputeRow(uint32_t rowIdx)
    {
        const uint32_t rowBase = rowIdx * innerLen_;

        // Pass 1: rowMax
        bool firstMax = true;
        AscendC::LocalTensor<float> rowMaxLocal = tmpQueue_.AllocTensor<float>(); // use first element
        uint32_t processed = 0;
        while (processed < innerLen_) {
            const uint32_t curLen = AscendC::min(tileInner_, innerLen_ - processed);
            LoadSeg(rowBase + processed, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> segMax = tmpQueue_.AllocTensor<float>(); // length >= 1
            AscendC::ReduceMax(segMax, xLocal, curLen); // segMax[0] = max(xLocal)

            if (firstMax) {
                AscendC::DataCopy(rowMaxLocal, segMax, 1);
                firstMax = false;
            } else {
                AscendC::Max(rowMaxLocal, rowMaxLocal, segMax, 1);
            }

            tmpQueue_.FreeTensor(segMax);
            inQueueX_.FreeTensor(xLocal);
            processed += curLen;
        }

        // Pass 2: sumExp = sum(exp(x - rowMax))
        bool firstSum = true;
        AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>(); // use first element
        processed = 0;
        while (processed < innerLen_) {
            const uint32_t curLen = AscendC::min(tileInner_, innerLen_ - processed);
            LoadSeg(rowBase + processed, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> maxVec = tmpQueue_.AllocTensor<float>();
            AscendC::Broadcast(maxVec, rowMaxLocal, curLen);

            AscendC::LocalTensor<float> tmpVec = tmpQueue_.AllocTensor<float>();
            AscendC::Sub(tmpVec, xLocal, maxVec, curLen);
            AscendC::Exp(tmpVec, tmpVec, curLen);

            AscendC::LocalTensor<float> segSum = tmpQueue_.AllocTensor<float>();
            AscendC::ReduceSum(segSum, tmpVec, curLen); // segSum[0] = sum(exp(...))

            if (firstSum) {
                AscendC::DataCopy(sumLocal, segSum, 1);
                firstSum = false;
            } else {
                AscendC::Add(sumLocal, sumLocal, segSum, 1);
            }

            tmpQueue_.FreeTensor(segSum);
            tmpQueue_.FreeTensor(tmpVec);
            tmpQueue_.FreeTensor(maxVec);
            inQueueX_.FreeTensor(xLocal);
            processed += curLen;
        }

        // logSum = log(sumExp), bias = rowMax + logSum
        AscendC::LocalTensor<float> logSumLocal = tmpQueue_.AllocTensor<float>();
        AscendC::Log(logSumLocal, sumLocal, 1);
        AscendC::LocalTensor<float> biasLocal = tmpQueue_.AllocTensor<float>();
        AscendC::Add(biasLocal, rowMaxLocal, logSumLocal, 1);

        // Pass 3: y = x - bias
        processed = 0;
        while (processed < innerLen_) {
            const uint32_t curLen = AscendC::min(tileInner_, innerLen_ - processed);
            LoadSeg(rowBase + processed, curLen);
            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> biasVec = tmpQueue_.AllocTensor<float>();
            AscendC::Broadcast(biasVec, biasLocal, curLen);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Sub(yLocal, xLocal, biasVec, curLen);

            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xLocal);
            tmpQueue_.FreeTensor(biasVec);

            StoreSeg(rowBase + processed, curLen);
            processed += curLen;
        }

        tmpQueue_.FreeTensor(biasLocal);
        tmpQueue_.FreeTensor(logSumLocal);
        tmpQueue_.FreeTensor(sumLocal);
        tmpQueue_.FreeTensor(rowMaxLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2>         tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t innerLen_{0};
    uint32_t outerLen_{0};
    uint32_t tileInner_{0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.innerLength,
            tiling_data.outerLength,
            tiling_data.tileInner);
    op.Process();
}
"""
