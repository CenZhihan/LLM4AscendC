#include "conv_depthwise_2d_square_input_square_kernel_custom_tiling.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0u) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// Memory model for one output-row segment (length L):
//  - in rows: 3 * (L + 2)
//  - out row: 1 * L
//  - tmp vec: 1 * L
//  - weights: 16 floats (padding)
// total floats ~ 5L + 6 + 16 -> ~5L + 22; conservatively treat as 8L
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t BYTES_PER_FLOAT = 4;
static constexpr uint32_t ALIGN_ELEMS = 256;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dSquareInputSquareKernelCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    const uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t C = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    // Assume weight [C, 1, K, K]
    const uint32_t K = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    const uint32_t Hout = (H >= K) ? (H - K + 1) : 0u;
    const uint32_t Wout = (W >= K) ? (W - K + 1) : 0u;

    // tileW from UB budget:
    // bytes ~ 8L * 4
    uint32_t maxTileW = (UB_BUDGET_BYTES / (8u * BYTES_PER_FLOAT));
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (Wout > 0 && maxTileW > Wout) {
        const uint32_t aligned = AlignDown(Wout, ALIGN_ELEMS);
        maxTileW = (aligned > 0) ? aligned : Wout;
    }
    uint32_t tileW = (maxTileW == 0u) ? Wout : maxTileW;
    if (tileW == 0u && Wout > 0u) tileW = Wout;  // small shapes

    // blockDim: parallelize across (N*C*Hout) rows
    const uint64_t totalRows = static_cast<uint64_t>(N) * static_cast<uint64_t>(C) * static_cast<uint64_t>(Hout);
    uint32_t blockDim = static_cast<uint32_t>(totalRows);
    if (blockDim == 0u) blockDim = 1u;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_K(K);
    tiling.set_Hout(Hout);
    tiling.set_Wout(Wout);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t K, uint32_t Hout, uint32_t Wout, uint32_t tileW)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        K_ = K; Hout_ = Hout; Wout_ = Wout;
        tileW_ = (tileW == 0u) ? Wout_ : tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, C_ * K_ * K_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * Hout_ * Wout_);

        // buffers: 3*(tileW + K - 1) for input rows, 1*tileW for output, 1*tileW for tmp, weights 16 floats
        const uint32_t inRowLen = tileW_ + (K_ - 1);
        pipe_.InitBuffer(inRow0Q_, 1, inRowLen * sizeof(float));
        pipe_.InitBuffer(inRow1Q_, 1, inRowLen * sizeof(float));
        pipe_.InitBuffer(inRow2Q_, 1, inRowLen * sizeof(float));
        pipe_.InitBuffer(outQ_,     1, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_,     1, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQ_,       1, 16 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || Hout_ == 0 || Wout_ == 0) return;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t rowsPerBatch = static_cast<uint64_t>(C_) * static_cast<uint64_t>(Hout_);
        const uint64_t totalRows = static_cast<uint64_t>(N_) * rowsPerBatch;

        for (uint64_t rowId = blockIdx; rowId < totalRows; rowId += blockNum) {
            const uint32_t n = static_cast<uint32_t>(rowId / rowsPerBatch);
            const uint32_t rc = static_cast<uint32_t>(rowId % rowsPerBatch);
            const uint32_t c = rc / Hout_;
            const uint32_t oh = rc % Hout_;

            // load weights for channel c (assumes depthwise [C,1,K,K])
            float wv[9];
            LoadWeights(c, wv);

            const uint64_t inBase  = (((uint64_t)n * C_ + c) * H_) * W_;       // start of this (n,c) in x
            const uint64_t outBase = (((uint64_t)n * C_ + c) * Hout_ + oh) * Wout_; // start of this out row

            for (uint32_t ow0 = 0; ow0 < Wout_; ow0 += tileW_) {
                const uint32_t segW = AscendC::min(tileW_, Wout_ - ow0);
                ComputeRowSegment(inBase, outBase, oh, ow0, segW, wv);
            }
        }
    }

private:
    __aicore__ inline void LoadWeights(uint32_t c, float wv[9])
    {
        const uint32_t k2 = K_ * K_; // K*K
        AscendC::LocalTensor<float> wLocal = wQ_.AllocTensor<float>();
        const uint64_t wOff = static_cast<uint64_t>(c) * k2;
        // copy at least 9 for K=3; if larger K, we only use top-left 3x3 (per kernel naming assumption)
        const uint32_t copyNum = (k2 >= 9u) ? 9u : k2;
        AscendC::DataCopy(wLocal, wGm_[wOff], copyNum);

        // fetch values into scalars (API assumed)
        #pragma unroll
        for (uint32_t i = 0; i < 9; ++i) {
            float v = 0.f;
            if (i < copyNum) {
                v = wLocal.GetValue(i);
            }
            wv[i] = v;
        }
        wQ_.FreeTensor(wLocal);
    }

    __aicore__ inline void ComputeRowSegment(uint64_t inBase, uint64_t outBase,
                                             uint32_t oh, uint32_t ow0, uint32_t segW,
                                             const float wv[9])
    {
        const uint32_t needW = segW + (K_ - 1); // seg + halo (2)
        AscendC::LocalTensor<float> in0 = inRow0Q_.AllocTensor<float>();
        AscendC::LocalTensor<float> in1 = inRow1Q_.AllocTensor<float>();
        AscendC::LocalTensor<float> in2 = inRow2Q_.AllocTensor<float>();
        AscendC::LocalTensor<float> y   = outQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();

        // DMA input rows (contiguous segments including halo)
        const uint64_t row0 = inBase + static_cast<uint64_t>(oh + 0) * W_ + ow0;
        const uint64_t row1 = inBase + static_cast<uint64_t>(oh + 1) * W_ + ow0;
        const uint64_t row2 = inBase + static_cast<uint64_t>(oh + 2) * W_ + ow0;

        AscendC::DataCopy(in0, xGm_[row0], needW);
        AscendC::DataCopy(in1, xGm_[row1], needW);
        AscendC::DataCopy(in2, xGm_[row2], needW);

        // y = 0
        AscendC::Duplicate(y, 0.0f, segW);

        // Accumulate 3x3
        // top row
        AccumFMA(y, in0, 0, wv[0], segW, tmp);
        AccumFMA(y, in0, 1, wv[1], segW, tmp);
        AccumFMA(y, in0, 2, wv[2], segW, tmp);
        // mid row
        AccumFMA(y, in1, 0, wv[3], segW, tmp);
        AccumFMA(y, in1, 1, wv[4], segW, tmp);
        AccumFMA(y, in1, 2, wv[5], segW, tmp);
        // bot row
        AccumFMA(y, in2, 0, wv[6], segW, tmp);
        AccumFMA(y, in2, 1, wv[7], segW, tmp);
        AccumFMA(y, in2, 2, wv[8], segW, tmp);

        // store
        const uint64_t yOff = outBase + ow0;
        AscendC::DataCopy(yGm_[yOff], y, segW);

        tmpQ_.FreeTensor(tmp);
        outQ_.FreeTensor(y);
        inRow2Q_.FreeTensor(in2);
        inRow1Q_.FreeTensor(in1);
        inRow0Q_.FreeTensor(in0);
    }

    __aicore__ inline void AccumFMA(AscendC::LocalTensor<float>& y,
                                    AscendC::LocalTensor<float>& xin,
                                    uint32_t shift,
                                    float w, uint32_t len, AscendC::LocalTensor<float>& tmp)
    {
        // tmp = xin[shift:shift+len] * w
        AscendC::Muls(tmp, xin[shift], w, len);
        AscendC::Add(y, y, tmp, len);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inRow0Q_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inRow1Q_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inRow2Q_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t K_{3}, Hout_{0}, Wout_{0}, tileW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.K, tiling_data.Hout, tiling_data.Wout, tiling_data.tileW);
    op.Process();
}
