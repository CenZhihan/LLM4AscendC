host_operator_src="""

#include "matmul_for_symmetric_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model:
//   inQueueA: 1 * kTile
//   inQueueB: 1 * kTile
//   calcMul: 1 * kTile
//   redBuf: 1
//   sumBuf: 1
// Total bytes = (3 * kTile + 2) * 4
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // favor decent DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 8192;         // clamp to avoid long inner loops

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForSymmetricMatricesCustomTilingData tiling;

    const gert::Shape *aShape = context->GetInputShape(0);
    const auto &oShape = aShape->GetOriginShape();
    if (oShape.GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(oShape.GetDim(0));
    const uint32_t M = static_cast<uint32_t>(oShape.GetDim(1));
    if (N == 0 || M == 0 || N != M) {
        return ge::GRAPH_FAILED;
    }

    // kTile from UB budget: bytes = (3*kTile + 2) * 4
    uint32_t maxTile = (UB_BUDGET_BYTES / 4u > 2u) ? (UB_BUDGET_BYTES / 4u - 2u) / 3u : 1u;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    if (maxTile > N) {
        uint32_t aligned = AlignDown(N, ALIGN_ELEMS);
        maxTile = (aligned > 0) ? aligned : N;
    }
    const uint32_t kTile = maxTile;

    // BlockDim parallelizes across rows i
    uint32_t blockDim = N;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_kTile(kTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    const auto &ao = a_shape->GetOriginShape();
    const auto &bo = b_shape->GetOriginShape();
    if (ao.GetDimNum() != 2 || bo.GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    if (ao.GetDim(0) != ao.GetDim(1) || bo.GetDim(0) != bo.GetDim(1) || ao.GetDim(1) != bo.GetDim(0)) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    *c_shape = *a_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) return GRAPH_FAILED;
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;

class KernelMatmulForSymmetricMatricesCustom {
public:
    __aicore__ inline KernelMatmulForSymmetricMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t kTile)
    {
        N_ = N;
        kTile_ = kTile;

        const uint32_t totalElems = N_ * N_;
        aGm_.SetGlobalBuffer((__gm__ float*)A, totalElems);
        bGm_.SetGlobalBuffer((__gm__ float*)B, totalElems);
        cGm_.SetGlobalBuffer((__gm__ float*)C, totalElems);

        // Buffers: A chunk, B chunk, Mul tmp, Reduce tmp (1), Sum buf (1)
        pipe_.InitBuffer(inQueueA_, IN_BUF_NUM, kTile_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, IN_BUF_NUM, kTile_ * sizeof(float));
        pipe_.InitBuffer(calcMulQueue_, 1, kTile_ * sizeof(float));
        pipe_.InitBuffer(redQueue_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(sumQueue_, 1, 1 * sizeof(float));
        pipe_.InitBuffer(outQueueC_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || kTile_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t i = blockIdx; i < N_; i += blockNum) {
            const uint32_t aRowBase = i * N_;
            for (uint32_t j = 0; j < N_; ++j) {
                const uint32_t bRowBase = j * N_;
                bool firstChunk = true;

                // Allocate persistent sum buffer (1 element) for this (i,j)
                AscendC::LocalTensor<float> sumLocal = sumQueue_.AllocTensor<float>();

                for (uint32_t kBase = 0; kBase < N_; kBase += kTile_) {
                    uint32_t len = N_ - kBase;
                    if (len > kTile_) len = kTile_;

                    // Copy A row chunk
                    {
                        AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
                        AscendC::DataCopy(aLocal, aGm_[aRowBase + kBase], len);
                        inQueueA_.EnQue(aLocal);
                    }
                    // Copy B row chunk (leveraging symmetry: use row j instead of column j)
                    {
                        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
                        AscendC::DataCopy(bLocal, bGm_[bRowBase + kBase], len);
                        inQueueB_.EnQue(bLocal);
                    }

                    // Compute partial dot and accumulate
                    {
                        AscendC::LocalTensor<float> aLocal = inQueueA_.DeQue<float>();
                        AscendC::LocalTensor<float> bLocal = inQueueB_.DeQue<float>();
                        AscendC::LocalTensor<float> mulLocal = calcMulQueue_.AllocTensor<float>();
                        AscendC::Mul(mulLocal, aLocal, bLocal, len);

                        AscendC::LocalTensor<float> redLocal = redQueue_.AllocTensor<float>();
                        AscendC::ReduceSum(redLocal, mulLocal, len);

                        if (firstChunk) {
                            AscendC::DataCopy(sumLocal, redLocal, 1);
                            firstChunk = false;
                        } else {
                            AscendC::Add(sumLocal, sumLocal, redLocal, 1);
                        }

                        redQueue_.FreeTensor(redLocal);
                        calcMulQueue_.FreeTensor(mulLocal);
                        inQueueB_.FreeTensor(bLocal);
                        inQueueA_.FreeTensor(aLocal);
                    }
                }

                // Write result scalar to C[i, j]
                {
                    AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
                    AscendC::DataCopy(cLocal, sumLocal, 1);
                    outQueueC_.EnQue(cLocal);

                    AscendC::LocalTensor<float> cToWrite = outQueueC_.DeQue<float>();
                    AscendC::DataCopy(cGm_[aRowBase + j], cToWrite, 1);
                    outQueueC_.FreeTensor(cToWrite);
                }

                sumQueue_.FreeTensor(sumLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcMulQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> redQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_ {0};
    uint32_t kTile_ {0};
};

extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForSymmetricMatricesCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.kTile);
    op.Process();
}
