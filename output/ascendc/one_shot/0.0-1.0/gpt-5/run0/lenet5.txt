kernel_src="""
#include "kernel_operator.h"

constexpr int32_t QUE_SZ_X = 1;
constexpr int32_t QUE_SZ_TMPA = 1;
constexpr int32_t QUE_SZ_TMPB = 1;
constexpr int32_t QUE_SZ_TMPC = 1;
constexpr int32_t QUE_SZ_ROWW = 1;
constexpr int32_t QUE_SZ_BIAS = 1;

class KernelLenet5Custom {
public:
    __aicore__ inline KernelLenet5Custom() {}

    __aicore__ inline void Init(
        GM_ADDR x,
        GM_ADDR conv1_w, GM_ADDR conv1_b,
        GM_ADDR conv2_w, GM_ADDR conv2_b,
        GM_ADDR fc1_w,  GM_ADDR fc1_b,
        GM_ADDR fc2_w,  GM_ADDR fc2_b,
        GM_ADDR fc3_w,  GM_ADDR fc3_b,
        GM_ADDR y,
        uint32_t batch, uint32_t numClasses)
    {
        batch_ = batch;
        numClasses_ = numClasses;

        // Set global buffers sizes
        const uint32_t xLen = batch_ * 1u * 32u * 32u;
        const uint32_t yLen = batch_ * numClasses_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xLen);
        conv1wGm_.SetGlobalBuffer((__gm__ float*)conv1_w, 6u*1u*5u*5u);
        conv1bGm_.SetGlobalBuffer((__gm__ float*)conv1_b, 6u);
        conv2wGm_.SetGlobalBuffer((__gm__ float*)conv2_w, 16u*6u*5u*5u);
        conv2bGm_.SetGlobalBuffer((__gm__ float*)conv2_b, 16u);
        fc1wGm_.SetGlobalBuffer((__gm__ float*)fc1_w, 120u*400u);
        fc1bGm_.SetGlobalBuffer((__gm__ float*)fc1_b, 120u);
        fc2wGm_.SetGlobalBuffer((__gm__ float*)fc2_w, 84u*120u);
        fc2bGm_.SetGlobalBuffer((__gm__ float*)fc2_b, 84u);
        fc3wGm_.SetGlobalBuffer((__gm__ float*)fc3_w, numClasses_*84u);
        fc3bGm_.SetGlobalBuffer((__gm__ float*)fc3_b, numClasses_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yLen);

        // Init UB queues (sizes in elements)
        // x: 1*32*32=1024
        pipe_.InitBuffer(qX_,    QUE_SZ_X,    1024u * sizeof(float));
        // tmpA: max(conv1_out=4704, conv2_w=2400, fc row up to 400, also used for fc rows)
        pipe_.InitBuffer(qTmpA_, QUE_SZ_TMPA, 4704u * sizeof(float));
        // tmpB: max(pool1=1176, biases up to 120, also reuse for intermediate vecs)
        pipe_.InitBuffer(qTmpB_, QUE_SZ_TMPB, 2400u * sizeof(float)); // also can hold conv2_w if needed
        // tmpC: conv2_out=1600
        pipe_.InitBuffer(qTmpC_, QUE_SZ_TMPC, 1600u * sizeof(float));
        // row buffer for FC rows up to 400
        pipe_.InitBuffer(qRowW_, QUE_SZ_ROWW, 400u * sizeof(float));
        // bias/out small buffer up to 120 (fc1 out)
        pipe_.InitBuffer(qBias_, QUE_SZ_BIAS, 120u * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t n = blockIdx; n < batch_; n += blockNum) {
            ComputeOneSample(n);
        }
    }

private:
    // Helpers for indexing
    __aicore__ inline uint32_t idx2d(uint32_t y, uint32_t x, uint32_t W) const { return y * W + x; }
    __aicore__ inline uint32_t idx3d(uint32_t c, uint32_t y, uint32_t x, uint32_t H, uint32_t W) const { return c * (H * W) + y * W + x; }
    __aicore__ inline uint32_t w1_idx(uint32_t oc, uint32_t ky, uint32_t kx) const { return oc * (1u * 25u) + ky * 5u + kx; }
    __aicore__ inline uint32_t w2_idx(uint32_t oc, uint32_t ic, uint32_t ky, uint32_t kx) const { return oc * (6u * 25u) + ic * 25u + ky * 5u + kx; }

    __aicore__ inline void ComputeOneSample(uint32_t n)
    {
        // Constants
        const uint32_t IN_H = 32, IN_W = 32;
        const uint32_t C1 = 6, K1 = 5, O1_H = 28, O1_W = 28;
        const uint32_t P1_H = 14, P1_W = 14;
        const uint32_t C2 = 16, K2 = 5, O2_H = 10, O2_W = 10;
        const uint32_t P2_H = 5, P2_W = 5;
        const uint32_t FLAT = 16 * 5 * 5;

        // 1) Load input sample x[n]
        AscendC::LocalTensor<float> xLocal = qX_.AllocTensor<float>();
        {
            const uint32_t xoff = n * (IN_H * IN_W);
            AscendC::DataCopy(xLocal, xGm_[xoff], IN_H * IN_W);
        }

        // 2) Conv1 + ReLU -> out1 (C1,O1_H,O1_W)
        AscendC::LocalTensor<float> out1 = qTmpA_.AllocTensor<float>(); // 4704
        {
            AscendC::LocalTensor<float> w1 = qTmpB_.AllocTensor<float>(); // 150 used portion
            AscendC::LocalTensor<float> b1 = qBias_.AllocTensor<float>(); // 6
            AscendC::DataCopy(w1, conv1wGm_[0], C1 * 1u * K1 * K1);
            AscendC::DataCopy(b1, conv1bGm_[0], C1);

            for (uint32_t oc = 0; oc < C1; ++oc) {
                for (uint32_t oy = 0; oy < O1_H; ++oy) {
                    for (uint32_t ox = 0; ox < O1_W; ++ox) {
                        float sum = b1[oc];
                        for (uint32_t ky = 0; ky < K1; ++ky) {
                            for (uint32_t kx = 0; kx < K1; ++kx) {
                                const float xv = xLocal[idx2d(oy + ky, ox + kx, IN_W)];
                                const float wv = w1[w1_idx(oc, ky, kx)];
                                sum += xv * wv;
                            }
                        }
                        // ReLU
                        if (sum < 0.f) sum = 0.f;
                        out1[idx3d(oc, oy, ox, O1_H, O1_W)] = sum;
                    }
                }
            }
            qTmpB_.FreeTensor(w1);
            qBias_.FreeTensor(b1);
        }

        // 3) MaxPool2d 2x2 stride2 -> p1 (C1,P1_H,P1_W)
        AscendC::LocalTensor<float> p1 = qTmpB_.AllocTensor<float>(); // 1176 fits
        {
            for (uint32_t oc = 0; oc < C1; ++oc) {
                for (uint32_t py = 0; py < P1_H; ++py) {
                    for (uint32_t px = 0; px < P1_W; ++px) {
                        const uint32_t iy = py * 2u;
                        const uint32_t ix = px * 2u;
                        float m = out1[idx3d(oc, iy, ix, O1_H, O1_W)];
                        float t = out1[idx3d(oc, iy, ix + 1u, O1_H, O1_W)]; if (t > m) m = t;
                        t = out1[idx3d(oc, iy + 1u, ix, O1_H, O1_W)]; if (t > m) m = t;
                        t = out1[idx3d(oc, iy + 1u, ix + 1u, O1_H, O1_W)]; if (t > m) m = t;
                        p1[idx3d(oc, py, px, P1_H, P1_W)] = m;
                    }
                }
            }
        }
        qTmpA_.FreeTensor(out1); // free out1

        // 4) Conv2 + ReLU -> out2 (C2,O2_H,O2_W)
        AscendC::LocalTensor<float> out2 = qTmpC_.AllocTensor<float>(); // 1600
        {
            AscendC::LocalTensor<float> w2 = qTmpA_.AllocTensor<float>(); // 2400
            AscendC::LocalTensor<float> b2 = qBias_.AllocTensor<float>(); // 16 (use first 16)
            AscendC::DataCopy(w2, conv2wGm_[0], C2 * C1 * K2 * K2);
            AscendC::DataCopy(b2, conv2bGm_[0], C2);

            for (uint32_t oc = 0; oc < C2; ++oc) {
                for (uint32_t oy = 0; oy < O2_H; ++oy) {
                    for (uint32_t ox = 0; ox < O2_W; ++ox) {
                        float sum = b2[oc];
                        for (uint32_t ic = 0; ic < C1; ++ic) {
                            for (uint32_t ky = 0; ky < K2; ++ky) {
                                for (uint32_t kx = 0; kx < K2; ++kx) {
                                    const uint32_t iy = oy + ky;
                                    const uint32_t ix = ox + kx;
                                    const float xv = p1[idx3d(ic, iy, ix, P1_H, P1_W)];
                                    const float wv = w2[w2_idx(oc, ic, ky, kx)];
                                    sum += xv * wv;
                                }
                            }
                        }
                        if (sum < 0.f) sum = 0.f;
                        out2[idx3d(oc, oy, ox, O2_H, O2_W)] = sum;
                    }
                }
            }
            qTmpA_.FreeTensor(w2);
            qBias_.FreeTensor(b2);
        }
        qTmpB_.FreeTensor(p1); // free p1

        // 5) MaxPool2d 2x2 stride2 -> p2 (vector FLAT=400)
        AscendC::LocalTensor<float> p2 = qRowW_.AllocTensor<float>(); // size >= 400
        {
            for (uint32_t oc = 0; oc < C2; ++oc) {
                for (uint32_t py = 0; py < P2_H; ++py) {
                    for (uint32_t px = 0; px < P2_W; ++px) {
                        const uint32_t iy = py * 2u;
                        const uint32_t ix = px * 2u;
                        float m = out2[idx3d(oc, iy, ix, O2_H, O2_W)];
                        float t = out2[idx3d(oc, iy, ix + 1u, O2_H, O2_W)]; if (t > m) m = t;
                        t = out2[idx3d(oc, iy + 1u, ix, O2_H, O2_W)]; if (t > m) m = t;
                        t = out2[idx3d(oc, iy + 1u, ix + 1u, O2_H, O2_W)]; if (t > m) m = t;
                        const uint32_t flat_idx = oc * (P2_H * P2_W) + py * P2_W + px;
                        p2[flat_idx] = m;
                    }
                }
            }
        }
        qTmpC_.FreeTensor(out2);

        // 6) FC1 (120) + ReLU
        AscendC::LocalTensor<float> fc1out = qBias_.AllocTensor<float>(); // reuse bias buffer size 120
        {
            // Process each output row: 120 x 400
            for (uint32_t j = 0; j < 120u; ++j) {
                AscendC::LocalTensor<float> row = qTmpA_.AllocTensor<float>(); // >=400
                const uint32_t woff = j * FLAT;
                AscendC::DataCopy(row, fc1wGm_[woff], FLAT);
                // bias
                float acc = fc1bGm_[j];
                // dot
                for (uint32_t i = 0; i < FLAT; ++i) {
                    acc += row[i] * p2[i];
                }
                if (acc < 0.f) acc = 0.f;
                fc1out[j] = acc;
                qTmpA_.FreeTensor(row);
            }
        }
        qRowW_.FreeTensor(p2);

        // 7) FC2 (84) + ReLU
        AscendC::LocalTensor<float> fc2out = qRowW_.AllocTensor<float>(); // reuse row buffer size >=120, but we need 84 output; ok
        {
            for (uint32_t j = 0; j < 84u; ++j) {
                AscendC::LocalTensor<float> row = qTmpA_.AllocTensor<float>(); // >=120
                const uint32_t woff = j * 120u;
                AscendC::DataCopy(row, fc2wGm_[woff], 120u);
                float acc = fc2bGm_[j];
                for (uint32_t i = 0; i < 120u; ++i) {
                    acc += row[i] * fc1out[i];
                }
                if (acc < 0.f) acc = 0.f;
                fc2out[j] = acc;
                qTmpA_.FreeTensor(row);
            }
        }
        qBias_.FreeTensor(fc1out);

        // 8) FC3 (numClasses) -> output
        {
            const uint32_t yBase = n * numClasses_;
            for (uint32_t j = 0; j < numClasses_; ++j) {
                AscendC::LocalTensor<float> row = qTmpA_.AllocTensor<float>(); // >=84
                const uint32_t woff = j * 84u;
                AscendC::DataCopy(row, fc3wGm_[woff], 84u);
                float acc = fc3bGm_[j];
                for (uint32_t i = 0; i < 84u; ++i) {
                    acc += row[i] * fc2out[i];
                }
                yGm_[yBase + j] = acc;
                qTmpA_.FreeTensor(row);
            }
        }
        qRowW_.FreeTensor(fc2out);

        // Free input x
        qX_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN,  QUE_SZ_X>    qX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QUE_SZ_TMPA> qTmpA_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QUE_SZ_TMPB> qTmpB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QUE_SZ_TMPC> qTmpC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QUE_SZ_ROWW> qRowW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QUE_SZ_BIAS> qBias_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> conv1wGm_;
    AscendC::GlobalTensor<float> conv1bGm_;
    AscendC::GlobalTensor<float> conv2wGm_;
    AscendC::GlobalTensor<float> conv2bGm_;
    AscendC::GlobalTensor<float> fc1wGm_;
    AscendC::GlobalTensor<float> fc1bGm_;
    AscendC::GlobalTensor<float> fc2wGm_;
    AscendC::GlobalTensor<float> fc2bGm_;
    AscendC::GlobalTensor<float> fc3wGm_;
    AscendC::GlobalTensor<float> fc3bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_{0};
    uint32_t numClasses_{0};
};

extern "C" __global__ __aicore__ void lenet5_custom(
    GM_ADDR x,
    GM_ADDR conv1_w, GM_ADDR conv1_b,
    GM_ADDR conv2_w, GM_ADDR conv2_b,
