kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

// Fused kernel: y[n,1,ho,wo] = min_c_out( scale * (conv2d(x, weight) + bias) )
class KernelConv2dScalingMinCustom {
public:
    __aicore__ inline KernelConv2dScalingMinCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t hIn, uint32_t wIn,
                                uint32_t cOut, uint32_t kH, uint32_t kW,
                                uint32_t hOut, uint32_t wOut,
                                uint32_t totalTiles, uint32_t tileLen,
                                float scale, uint32_t hasBias,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padT, uint32_t padL,
                                uint32_t dilationH, uint32_t dilationW)
    {
        n_ = n; cIn_ = cIn; hIn_ = hIn; wIn_ = wIn;
        cOut_ = cOut; kH_ = kH; kW_ = kW; hOut_ = hOut; wOut_ = wOut;
        totalTiles_ = totalTiles; tileLen_ = tileLen;
        scale_ = scale; hasBias_ = hasBias;
        strideH_ = strideH; strideW_ = strideW;
        padT_ = padT; padL_ = padL;
        dilationH_ = dilationH; dilationW_ = dilationW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * cIn_ * hIn_ * wIn_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)cOut_ * cIn_ * kH_ * kW_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, cOut_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * 1 * hOut_ * wOut_);

        // UB buffers sized by cOut for accumulation and temporaries
        uint32_t vecBytes = cOut_ * sizeof(float);
        if (vecBytes == 0) vecBytes = 4;
        pipe_.InitBuffer(accQueue_, 1, vecBytes);
        pipe_.InitBuffer(tmpQueue_, 1, vecBytes);
        pipe_.InitBuffer(minQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalTiles_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideTiles = blockNum * tileLen_;
        uint32_t baseTile = blockIdx * tileLen_;
        if (baseTile >= totalTiles_) return;

        // Grid stride loop over output pixels (N * Ho * Wo)
        while (baseTile < totalTiles_) {
            uint32_t curTiles = totalTiles_ - baseTile;
            if (curTiles > tileLen_) curTiles = tileLen_;
            ComputeBlock(baseTile, curTiles);
            baseTile += strideTiles;
        }
    }

private:
    __aicore__ inline void ZeroAcc(AscendC::LocalTensor<float>& acc)
    {
        AscendC::Dup(acc, 0.0f, cOut_);
    }

    __aicore__ inline void AddBias(AscendC::LocalTensor<float>& acc)
    {
        if (!hasBias_) return;
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        // Copy bias to UB and add
        AscendC::DataCopy(tmp, bGm_, cOut_);
        AscendC::Add(acc, acc, tmp, cOut_);
        tmpQueue_.FreeTensor(tmp);
    }

    __aicore__ inline void ScaleAcc(AscendC::LocalTensor<float>& acc)
    {
        AscendC::Muls(acc, acc, scale_, cOut_);
    }

    // Compute one output pixel (min over cOut) at flattened index tIdx
    __aicore__ inline float ComputeOne(uint32_t tIdx)
    {
        // Decode n, ho, wo
        const uint32_t ho = tIdx % hOut_;
        const uint32_t nIdx = tIdx / (hOut_ * wOut_);
        const uint32_t wo = (tIdx / hOut_) % wOut_;

        AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();
        ZeroAcc(acc);

        // Iterate all input channels and kernel positions
        for (uint32_t ci = 0; ci < cIn_; ++ci) {
            for (uint32_t kh = 0; kh < kH_; ++kh) {
                const int32_t in_h = static_cast<int32_t>(ho) * static_cast<int32_t>(strideH_) - static_cast<int32_t>(padT_) + static_cast<int32_t>(kh) * static_cast<int32_t>(dilationH_);
                if (in_h < 0 || in_h >= static_cast<int32_t>(hIn_)) continue;
                for (uint32_t kw = 0; kw < kW_; ++kw) {
                    const int32_t in_w = static_cast<int32_t>(wo) * static_cast<int32_t>(strideW_) - static_cast<int32_t>(padL_) + static_cast<int32_t>(kw) * static_cast<int32_t>(dilationW_);
                    if (in_w < 0 || in_w >= static_cast<int32_t>(wIn_)) continue;

                    const uint64_t xOff = (((uint64_t)nIdx * cIn_ + ci) * hIn_ + (uint32_t)in_h) * wIn_ + (uint32_t)in_w;

                    // Load scalar input
                    AscendC::LocalTensor<float> xTmp = minQueue_.AllocTensor<float>(); // reuse small buffer
                    AscendC::DataCopy(xTmp, xGm_[xOff], 1);
                    // Broadcast input scalar to tmp vector
                    AscendC::LocalTensor<float> tmpVec = tmpQueue_.AllocTensor<float>();
                    AscendC::Dup(tmpVec, xTmp.GetValue(0), cOut_);

                    // Load weight vector across cOut for fixed (ci,kh,kw)
                    // Note: conceptual; actual layout may require strided loads.
                    const uint64_t wBase = ((uint64_t)ci * kH_ + kh) * kW_ + kw;
                    // Each oc located at oc*(cIn_*kH_*kW_) + wBase
                    // We fetch per-oc in a loop and accumulate: emulate via vector ops by assembling weights in UB
                    AscendC::LocalTensor<float> wVec = accQueue_.AllocTensor<float>(); // temporarily reuse accQueue_ (size cOut_)
                    // Build wVec via GM scatter-like loads (conceptual)
                    for (uint32_t oc = 0; oc < cOut_; ++oc) {
                        const uint64_t wOff = (uint64_t)oc * (uint64_t)(cIn_ * kH_ * kW_) + wBase;
                        AscendC::LocalTensor<float> wScalar = minQueue_.AllocTensor<float>();
                        AscendC::DataCopy(wScalar, wGm_[wOff], 1);
                        wVec.SetValue(oc, wScalar.GetValue(0));
                        minQueue_.FreeTensor(wScalar);
                    }

                    // acc += wVec * x_scalar
                    AscendC::Mul(tmpVec, wVec, tmpVec, cOut_);
                    AscendC::Add(acc, acc, tmpVec, cOut_);

                    accQueue_.FreeTensor(wVec);
                    tmpQueue_.FreeTensor(tmpVec);
                    minQueue_.FreeTensor(xTmp);
                }
            }
        }

        AddBias(acc);
        ScaleAcc(acc);

        // Reduce min across cOut
        AscendC::LocalTensor<float> red = minQueue_.AllocTensor<float>(); // size 1
        AscendC::ReduceMin(red, acc, cOut_);
        float outVal = red.GetValue(0);

        minQueue_.FreeTensor(red);
        accQueue_.FreeTensor(acc);
        return outVal;
    }

    __aicore__ inline void ComputeBlock(uint32_t baseTile, uint32_t numTiles)
    {
        // For each tile, compute and write Y scalar
        for (uint32_t t = 0; t < numTiles; ++t) {
            const uint32_t gIdx = baseTile + t;
            if (gIdx >= totalTiles_) break;

            float v = ComputeOne(gIdx);

            // Decode y offset: y[n,1,ho,wo]
            const uint32_t ho = gIdx % hOut_;
            const uint32_t nIdx = gIdx / (hOut_ * wOut_);
            const uint32_t wo = (gIdx / hOut_) % wOut_;
            const uint64_t yOff = (((uint64_t)nIdx) * 1u + 0u) * (uint64_t)(hOut_ * wOut_) + (uint64_t)ho * wOut_ + (uint64_t)wo;

            AscendC::LocalTensor<float> yScalar = minQueue_.AllocTensor<float>();
            yScalar.SetValue(0, v);
            AscendC::DataCopy(yGm_[yOff], yScalar, 1);
            minQueue_.FreeTensor(yScalar);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> minQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, hIn_{0}, wIn_{0};
    uint32_t cOut_{0}, kH_{0}, kW_{0}, hOut_{0}, wOut_{0};
    uint32_t totalTiles_{0}, tileLen_{0};

    float scale_{1.0f};
    uint32_t hasBias_{0};
    uint32_t strideH_{1}, strideW_{1}, padT_{0}, padL_{0}, dilationH_{1}, dilationW_{1};
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dScalingMinCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.n, tiling_data.cIn, tiling_data.hIn, tiling_data.wIn,
            tiling_data.cOut, tiling_data.kH, tiling_data.kW,
            tiling_data.hOut, tiling_data.wOut,
            tiling_data.totalTiles, tiling_data.tileLen,
            tiling_data.scale, tiling_data.hasBias,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padT, tiling_data.padL,
            tiling_data.dilationH, tiling_data.dilationW);
    op.Process();
}
