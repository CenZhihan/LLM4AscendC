host_operator_src="""

#include "cumsum_exclusive_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
// Total floats = 4 * tile => bytes = 16 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // avoid excessive per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumExclusiveCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    if (dimPtr == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const gert::Shape *inShape = context->GetInputShape(0);
    const gert::Shape &orig = inShape->GetOriginShape();
    const uint32_t rank = static_cast<uint32_t>(orig.GetDimNum());
    if (rank == 0u) {
        return ge::GRAPH_FAILED;
    }

    int32_t axis = *dimPtr;
    if (axis < 0) axis += static_cast<int32_t>(rank);
    if (axis < 0 || axis >= static_cast<int32_t>(rank)) {
        return ge::GRAPH_FAILED;
    }

    uint64_t axisLen = static_cast<uint64_t>(orig.GetDim(axis));
    uint64_t inner = 1;
    for (int32_t i = axis + 1; i < static_cast<int32_t>(rank); ++i) {
        inner *= static_cast<uint64_t>(orig.GetDim(i));
    }
    uint64_t outer = 1;
    for (int32_t i = 0; i < axis; ++i) {
        outer *= static_cast<uint64_t>(orig.GetDim(i));
    }

    uint64_t rows = outer * inner;
    if (axisLen == 0 || rows == 0) {
        // Degenerate, but still set minimal parameters
        context->SetBlockDim(1);
        tiling.set_axisLen(0);
        tiling.set_rows(0);
        tiling.set_tileLength(0);
        tiling.set_inner(static_cast<uint32_t>(inner));
        tiling.set_blockDim(1);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    // tileLen from UB budget: bytes = 16 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 16;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = static_cast<uint32_t>(axisLen);
    if (tileLen > maxTile) {
        tileLen = maxTile;
    } else {
        // align down for efficiency but ensure non-zero
        uint32_t aligned = AlignDown(tileLen, ALIGN_ELEMS);
        tileLen = (aligned > 0u) ? aligned : tileLen;
    }

    // blockDim based on number of independent rows
    uint32_t blockDim = static_cast<uint32_t>(rows);
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_axisLen(static_cast<uint32_t>(axisLen));
    tiling.set_rows(static_cast<uint32_t>(rows));
    tiling.set_tileLength(tileLen);
    tiling.set_inner(static_cast<uint32_t>(inner));
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    gert::Shape *y_shape = context->GetOutputShape(0);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCumsumExclusiveCustom {
public:
    __aicore__ inline KernelCumsumExclusiveCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t axisLen, uint32_t rows,
                                uint32_t tileLength, uint32_t inner)
    {
        axisLen_ = axisLen;
        rows_ = rows;
        tileLen_ = tileLength;
        inner_ = inner;

        totalElems_ = axisLen_ * rows_; // equals total number of elements

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElems_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (axisLen_ == 0 || rows_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Distribute rows across blocks
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            float carry = 0.0f;
            uint32_t pos = 0;

            // Prefetch first tile
            uint32_t curLen = (axisLen_ - pos);
            if (curLen > tileLen_) curLen = tileLen_;
            CopyInRow(row, pos, curLen);

            while (pos < axisLen_) {
                // Prefetch next tile early to overlap
                const uint32_t nextPos = pos + curLen;
                uint32_t nextLen = 0;
                if (nextPos < axisLen_) {
                    nextLen = axisLen_ - nextPos;
                    if (nextLen > tileLen_) nextLen = tileLen_;
                    CopyInRow(row, nextPos, nextLen);
                }

                // Compute current tile
                carry = ComputeExclusive(curLen, carry);
                // Write back
                CopyOutRow(row, pos, curLen);

                if (nextLen == 0) break;
                pos = nextPos;
                curLen = nextLen;
            }
        }
    }

private:
    __aicore__ inline void CopyInRow(uint32_t row, uint32_t pos, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        if (inner_ == 1) {
            // contiguous along the last dimension
            const uint32_t gmOffset = row * axisLen_ + pos;
            AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        } else {
            // Fallback not optimized: require last-dim for best performance
            // For unsupported stride, simply zero-fill to avoid undefined behavior.
            AscendC::Duplicate(xLocal, 0.0f, len);
        }
        inQueueX_.EnQue(xLocal);
    }

    // Returns updated carry after processing this tile
    __aicore__ inline float ComputeExclusive(uint32_t len, float carryIn)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        float prefix = carryIn;

        // Simple scalar loop within local memory to compute exclusive prefix sum
        // y[i] = prefix; prefix += x[i]
        for (uint32_t i = 0; i < len; ++i) {
            float xi = xLocal.GetValue(i);
            yLocal.SetValue(i, prefix);
            prefix += xi;
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        return prefix; // new carry for next tile
    }

    __aicore__ inline void CopyOutRow(uint32_t row, uint32_t pos, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        if (inner_ == 1) {
            const uint32_t gmOffset = row * axisLen_ + pos;
            AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        } else {
            // Skip write for unsupported stride to avoid invalid memory writes
        }
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t axisLen_ {0};
    uint32_t rows_ {0};
    uint32_t tileLen_ {0};
    uint32_t inner_ {1};
    uint32_t totalElems_ {0};
};

extern "C" __global__ __aicore__ void cumsum_exclusive_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumExclusiveCustom op;
    op.Init(x, y,
            tiling_data.axisLen, tiling_data.rows,
            tiling_data.tileLength, tiling_data.inner);
    op.Process();
}
