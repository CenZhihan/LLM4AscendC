host_operator_src="""
#include "conv_standard_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// UB model per core (oc chunk = 1):
//   xChanBuf: (tileLenOut + K - 1) floats
//   yBuf: tileLenOut floats
//   tmpBuf: tileLenOut floats
// Total ~ (3 * tileLenOut + K - 1) floats => bytes ~ (12 * tileLenOut + 4*(K - 1))
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_TILE_LEN = 32768;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);

    const uint32_t n    = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t cIn  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t lIn  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));

    const uint32_t cOut = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t k    = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));

    if (k == 0 || lIn < k) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t lOut = lIn - k + 1;

    // Derive tileLenOut from UB budget
    uint32_t maxTile = 0;
    if (UB_BUDGET_BYTES > 4u * (k - 1u)) {
        maxTile = (UB_BUDGET_BYTES - 4u * (k - 1u)) / 12u;
    }
    if (maxTile == 0) maxTile = ALIGN_ELEMS;
    if (maxTile > MAX_TILE_LEN) maxTile = MAX_TILE_LEN;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile == 0) maxTile = ALIGN_ELEMS;

    uint32_t tileLenOut = maxTile;
    if (lOut < tileLenOut) {
        const uint32_t aligned = AlignDown(lOut, ALIGN_ELEMS);
        tileLenOut = (aligned > 0) ? aligned : lOut;
    }
    if (tileLenOut == 0) tileLenOut = 1;

    const uint32_t posTiles = CeilDiv(lOut, tileLenOut);
    // Work units = n * cOut * posTiles (oc chunk = 1)
    const uint32_t totalWork = n * cOut * posTiles;

    // Conservative blockDim bounds
    uint32_t blockDim = totalWork;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    ConvStandard1dCustomTilingData tiling;
    tiling.set_n(n);
    tiling.set_cIn(cIn);
    tiling.set_cOut(cOut);
    tiling.set_lIn(lIn);
    tiling.set_k(k);
    tiling.set_lOut(lOut);
    tiling.set_tileLenOut(tileLenOut);
    tiling.set_posTiles(posTiles);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUFFERS  = 2;
constexpr int32_t OUT_BUFFERS = 2;
constexpr int32_t CALC_BUFFERS = 2;

class KernelConvStandard1dCustom {
public:
    __aicore__ inline KernelConvStandard1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t lIn, uint32_t k, uint32_t lOut,
                                uint32_t tileLenOut)
    {
        n_ = n; cIn_ = cIn; cOut_ = cOut;
        lIn_ = lIn; k_ = k; lOut_ = lOut; tileLenOut_ = tileLenOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_) * cIn_ * lIn_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(cOut_) * cIn_ * k_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_) * cOut_ * lOut_);

        // UB buffers:
        //   xChanBuf: tileLenOut + k - 1
        //   yBuf: tileLenOut
        //   tmpBuf: tileLenOut
        pipe_.InitBuffer(xInQ_, IN_BUFFERS, (tileLenOut_ + k_ - 1) * sizeof(float));
        pipe_.InitBuffer(yOutQ_, OUT_BUFFERS, tileLenOut_ * sizeof(float));
        pipe_.InitBuffer(calcQ_, CALC_BUFFERS, tileLenOut_ * sizeof(float));
        pipe_.InitBuffer(wScalarQ_, 1, sizeof(float)); // scalar staging
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cIn_ == 0 || cOut_ == 0 || lOut_ == 0 || tileLenOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t posTiles = (lOut_ + tileLenOut_ - 1u) / tileLenOut_;
        const uint32_t totalWork = n_ * cOut_ * posTiles;

        for (uint32_t workId = blockIdx; workId < totalWork; workId += blockNum) {
            // Decode linear workId -> (nIdx, ocIdx, posTileIdx)
            uint32_t tmp = workId;
            const uint32_t posTileIdx = tmp % posTiles;
            tmp /= posTiles;
            const uint32_t ocIdx = tmp % cOut_;
            const uint32_t nIdx = tmp / cOut_;

            const uint32_t posStart = posTileIdx * tileLenOut_;
            uint32_t curLen = lOut_ - posStart;
            if (curLen > tileLenOut_) curLen = tileLenOut_;

            // Allocate y buffer and zero it
            AscendC::LocalTensor<float> yLocal = yOutQ_.AllocTensor<float>();
            AscendC::Dup(yLocal, 0.0f, curLen);

            // Loop over input channels
            for (uint32_t c = 0; c < cIn_; ++c) {
                // Load input segment for this channel: length = curLen + k - 1
                const uint32_t xBase = ((nIdx * cIn_ + c) * lIn_) + posStart;
                AscendC::LocalTensor<float> xLocal = xInQ_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[xBase], curLen + k_ - 1);
                xInQ_.EnQue(xLocal);

                // DeQue to compute
                AscendC::LocalTensor<float> xBuf = xInQ_.DeQue<float>();

                // For each kernel tap
                for (uint32_t kk = 0; kk < k_; ++kk) {
                    // Load weight scalar w[ocIdx, c, kk]
                    const uint32_t wOffset = ((ocIdx * cIn_ + c) * k_) + kk;
                    AscendC::LocalTensor<float> wSc = wScalarQ_.AllocTensor<float>();
                    AscendC::DataCopy(wSc, wGm_[wOffset], 1);
                    // Assume ability to read scalar from local tensor
                    float wv = AscendC::GetValue(wSc, 0); // helper to fetch scalar
                    wScalarQ_.FreeTensor(wSc);

                    // tmp = xBuf[kk : kk+curLen] * wv
                    AscendC::LocalTensor<float> tmpVec = calcQ_.AllocTensor<float>();
                    AscendC::DataCopy(tmpVec, xBuf[kk], curLen);
                    AscendC::Muls(tmpVec, tmpVec, wv, curLen);
                    AscendC::Add(yLocal, yLocal, tmpVec, curLen);
                    calcQ_.FreeTensor(tmpVec);
                }

                xInQ_.FreeTensor(xBuf);
            }

            // Store y
            const uint32_t yBase = ((nIdx * cOut_ + ocIdx) * lOut_) + posStart;
            AscendC::DataCopy(yGm_[yBase], yLocal, curLen);
            yOutQ_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFERS> xInQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFERS> yOutQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, CALC_BUFFERS> calcQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wScalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, cOut_{0};
    uint32_t lIn_{0}, k_{0}, lOut_{0};
    uint32_t tileLenOut_{0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cIn, tiling_data.cOut,
            tiling_data.lIn, tiling_data.k, tiling_data.lOut,
            tiling_data.tileLenOut);
    op.Process();
}

at::Tensor conv1d_standard_impl_npu(const at::Tensor& x, const at::Tensor& weight) {

    auto n = x.size(0);
    auto c_out = weight.size(0);
    auto l_out = x.size(2) - weight.size(2) + 1;
