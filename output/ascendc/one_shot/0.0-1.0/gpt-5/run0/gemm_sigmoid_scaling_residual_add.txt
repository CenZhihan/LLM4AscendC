#include "gemm_sigmoid_scaling_residual_add_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB share
static constexpr uint32_t ELEM_SIZE = 4;                 // float
static constexpr uint32_t ALIGN_ELEMS = 256;             // moderate alignment for DMA
static constexpr uint32_t MIN_TILE_N = 64;
static constexpr uint32_t MAX_TILE_N = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmSigmoidScalingResidualAddCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    // Expect: x [M,K], w [N,K], b [N]
    const uint32_t dims_x = static_cast<uint32_t>(x_shape->GetStorageShape().GetDimNum());
    const uint32_t dims_w = static_cast<uint32_t>(w_shape->GetStorageShape().GetDimNum());
    if (dims_x < 2 || dims_w < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(dims_x - 2));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(dims_x - 1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(dims_w - 2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scalingAttr = attrs->GetAttrPointer<float>(0);
    const float scaling = (scalingAttr != nullptr) ? *scalingAttr : 1.0f;

    // UB model per-core (rough):
    //   x_row: K
    //   w_row: K
    //   y_tile: tileN
    //   b_tile: tileN
    // Total floats = 2*K + 2*tileN  => bytes = 4*(2*K + 2*tileN)
    // Solve for tileN given UB budget.
    uint32_t maxTileN = 0;
    if (K < (UB_BUDGET_BYTES / (2 * ELEM_SIZE))) {
        const uint32_t remainBytes = UB_BUDGET_BYTES - (2u * K * ELEM_SIZE);
        maxTileN = remainBytes / (2u * ELEM_SIZE);
    }
    if (maxTileN < MIN_TILE_N) maxTileN = MIN_TILE_N;
    if (maxTileN > MAX_TILE_N) maxTileN = MAX_TILE_N;
    maxTileN = AlignDown(maxTileN, ALIGN_ELEMS);
    if (maxTileN == 0) maxTileN = MIN_TILE_N;

    uint32_t tileN = maxTileN;
    if (N < tileN) {
        const uint32_t aligned = AlignDown(N, ALIGN_ELEMS);
        tileN = (aligned > 0) ? aligned : N;
    }

    // Block mapping on M dimension (rows)
    uint32_t blockDim = M;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_m(M);
    tiling.set_n(N);
    tiling.set_k(K);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);
    tiling.set_scaling(scaling);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

        return GRAPH_FAILED;

    gert::Shape *y_shape = context->GetOutputShape(0);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmSigmoidScalingResidualAddCustom {
public:
    __aicore__ inline KernelGemmSigmoidScalingResidualAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileN, float scaling)
    {
        m_ = m;
        n_ = n;
        k_ = k;
        tileN_ = tileN;
        scale_ = scaling;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)m_ * k_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)n_ * k_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)m_ * n_);

        // UB allocations (double buffers for overlap where appropriate)
        pipe_.InitBuffer(xRowQ_, 1, k_ * sizeof(float));
        pipe_.InitBuffer(wRowQ_, 1, k_ * sizeof(float));
        pipe_.InitBuffer(yTileQ_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(bTileQ_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpTileQ_, 1, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || n_ == 0 || k_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Row-parallel across M
        for (uint32_t row = blockIdx; row < m_; row += blockNum) {
            // Load one X row to UB
            AscendC::LocalTensor<float> xRow = xRowQ_.AllocTensor<float>();
            AscendC::DataCopy(xRow, xGm_[row * k_], k_);
            xRowQ_.EnQue(xRow);

            // Iterate N dimension by tiles
            for (uint32_t baseN = 0; baseN < n_; baseN += tileN_) {
                const uint32_t curN = ((baseN + tileN_) <= n_) ? tileN_ : (n_ - baseN);

                // yTile initialized from bias b[baseN: baseN+curN]
                AscendC::LocalTensor<float> yTile = yTileQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> bTile = bTileQ_.AllocTensor<float>();
                AscendC::DataCopy(bTile, bGm_[baseN], curN);
                // Initialize yTile with bias
                AscendC::DataCopy(yTile, bTile, curN);

                // Compute GEMM for this row and column tile: y[row, baseN:baseN+curN] = xRow * W_T[:, baseN:baseN+curN] + b
                for (uint32_t j = 0; j < curN; ++j) {
                    // Load W row (for output column baseN + j): contiguous length K
                    AscendC::LocalTensor<float> wRow = wRowQ_.AllocTensor<float>();
                    AscendC::DataCopy(wRow, wGm_[(baseN + j) * k_], k_);

                    // Dot product: acc = sum_{t=0..k_-1} xRow[t] * wRow[t]
                    // Note: For simplicity, accumulate using scalar loop over UB vectors.
                    float acc = 0.0f;
                    for (uint32_t t = 0; t < k_; ++t) {
                        acc += wRow.GetValue(t) * xRowQ_.GetFront<float>().GetValue(t);
                    }
                    // Add to yTile[j]
                    const float prev = yTile.GetValue(j);
                    yTile.SetValue(j, prev + acc);

                    wRowQ_.FreeTensor(wRow);
                }

                // Fused: y = y + scaling * sigmoid(y)
                AscendC::LocalTensor<float> tmp = tmpTileQ_.AllocTensor<float>();
                // tmp = y
                AscendC::DataCopy(tmp, yTile, curN);
                // tmp = sigmoid(y)
                AscendC::Sigmoid(tmp, tmp, curN);
                // tmp = scaling * sigmoid(y)
                AscendC::Muls(tmp, tmp, scale_, curN);
                // y = y + tmp
                AscendC::Add(yTile, yTile, tmp, curN);

                // Store result
                AscendC::DataCopy(yGm_[static_cast<uint64_t>(row) * n_ + baseN], yTile, curN);

                // Free locals
                tmpTileQ_.FreeTensor(tmp);
                bTileQ_.FreeTensor(bTile);
                yTileQ_.FreeTensor(yTile);
            }

            AscendC::LocalTensor<float> xRowFront = xRowQ_.DeQue<float>();
            xRowQ_.FreeTensor(xRowFront);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xRowQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wRowQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpTileQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t m_ {0};
    uint32_t n_ {0};
    uint32_t k_ {0};
    uint32_t tileN_ {0};
    float scale_ {1.0f};
};

extern "C" __global__ __aicore__ void gemm_sigmoid_scaling_residual_add_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSigmoidScalingResidualAddCustom op;
    op.Init(x, w, b, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileN, tiling_data.scaling);
    op.Process();
}
