host_operator_src="""
#include "batched_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // favor larger DMA bursts on vectors
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    BatchedMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *A_shape = context->GetInputShape(0);
    const gert::Shape *B_shape = context->GetInputShape(1);

    // Expect A: [B, M, K], B: [B, K, N]
    const uint64_t Bdim = static_cast<uint64_t>(A_shape->GetOriginShape().GetDim(0));
    const uint64_t Mdim = static_cast<uint64_t>(A_shape->GetOriginShape().GetDim(1));
    const uint64_t Kdim = static_cast<uint64_t>(A_shape->GetOriginShape().GetDim(2));
    const uint64_t Ndim = static_cast<uint64_t>(B_shape->GetOriginShape().GetDim(2));

    uint32_t Bv = static_cast<uint32_t>(Bdim);
    uint32_t Mv = static_cast<uint32_t>(Mdim);
    uint32_t Kv = static_cast<uint32_t>(Kdim);
    uint32_t Nv = static_cast<uint32_t>(Ndim);

    // UB model:
    // - inQueueB: 2 * tileN
    // - outQueueC(accum): 1 * tileN
    // - tmpQueueA (chunk of A row): 1 * tileK
    // Total floats = (3 * tileN + tileK) => bytes = 4 * (3 * tileN + tileK)
    // Choose tileK modest to balance looping and UB use.
    uint32_t tileK = 128;
    if (Kv < tileK) tileK = Kv;
    // compute maximum tileN allowed by UB budget
    uint32_t maxElemsByUB = (UB_BUDGET_BYTES / 4u);
    uint32_t maxTileN = 0;
    if (maxElemsByUB > tileK) {
        maxTileN = (maxElemsByUB - tileK) / 3u;
    }
    if (maxTileN < ALIGN_ELEMS) maxTileN = ALIGN_ELEMS;
    maxTileN = AlignDown(maxTileN, ALIGN_ELEMS);
    if (Nv < maxTileN) {
        uint32_t aligned = AlignDown(Nv, ALIGN_ELEMS);
        maxTileN = (aligned > 0) ? aligned : Nv;
    }
    uint32_t tileN = maxTileN;
    if (tileN == 0) tileN = Nv;  // degenerate

    // BlockDim selection: one core processes grid-stride rows (B*M)
    const uint64_t totalRows = static_cast<uint64_t>(Bv) * static_cast<uint64_t>(Mv);
    uint32_t blockDim = static_cast<uint32_t>(totalRows);
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_batch(Bv);
    tiling.set_m(Mv);
    tiling.set_k(Kv);
    tiling.set_n(Nv);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *A_shape = context->GetInputShape(0);
    const gert::Shape *B_shape = context->GetInputShape(1);
    // A: [B, M, K], B: [B, K, N] => C: [B, M, N]
    gert::Shape *C_shape = context->GetOutputShape(0);

    gert::Shape out;
    out.SetDimNum(3);
    out.SetDim(0, A_shape->GetOriginShape().GetDim(0));
    out.SetDim(1, A_shape->GetOriginShape().GetDim(1));
    out.SetDim(2, B_shape->GetOriginShape().GetDim(2));
    *C_shape = out;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    // Simple rule: expect both float and same dtype, propagate to output
    if (a_dtype != b_dtype) {
        // Fallback: prefer a_dtype
        context->SetOutputDataType(0, a_dtype);
    } else {
        context->SetOutputDataType(0, a_dtype);
    }
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelBatchedMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelBatchedMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t batch, uint32_t m, uint32_t k, uint32_t n,
                                uint32_t tileN, uint32_t tileK)
    {
        batch_ = batch;
        M_ = m;
        K_ = k;
        N_ = n;
        tileN_ = tileN;
        tileK_ = tileK;

        uint64_t sizeA = static_cast<uint64_t>(batch_) * M_ * K_;
        uint64_t sizeB = static_cast<uint64_t>(batch_) * K_ * N_;
        uint64_t sizeC = static_cast<uint64_t>(batch_) * M_ * N_;

        A_gm_.SetGlobalBuffer((__gm__ float*)A, sizeA);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, sizeB);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, sizeC);

        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueA_, 1, tileK_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalRows = static_cast<uint64_t>(batch_) * static_cast<uint64_t>(M_);
        if (totalRows == 0 || N_ == 0 || K_ == 0) return;

        for (uint64_t row = blockIdx; row < totalRows; row += blockNum) {
            const uint32_t b = static_cast<uint32_t>(row / M_);
            const uint32_t i = static_cast<uint32_t>(row % M_);

            for (uint32_t nBase = 0; nBase < N_; nBase += tileN_) {
                uint32_t curN = N_ - nBase;
                if (curN > tileN_) curN = tileN_;

                // Accumulator for C[b, i, nBase:nBase+curN]
                AscendC::LocalTensor<float> yLocal = outQueueC_.AllocTensor<float>();
                AscendC::Duplicate(yLocal, 0.0f, curN);

                for (uint32_t kBase = 0; kBase < K_; kBase += tileK_) {
                    uint32_t curK = K_ - kBase;
                    if (curK > tileK_) curK = tileK_;

                    // Load A row chunk: A[b, i, kBase : kBase+curK]
                    AscendC::LocalTensor<float> aLocal = tmpQueueA_.AllocTensor<float>();
                    uint64_t aOffset = ((uint64_t)b * M_ + i) * K_ + kBase;
                    AscendC::DataCopy(aLocal, A_gm_[aOffset], curK);

                    // Accumulate over k within the chunk
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        float aVal = aLocal.GetValue(kk);
                        if (aVal == 0.0f) {
                            continue;
                        }
                        // Load B row segment: B[b, kBase+kk, nBase : nBase+curN]
                        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
                        uint64_t bOffset = ((uint64_t)b * K_ + (kBase + kk)) * N_ + nBase;
                        AscendC::DataCopy(bLocal, B_gm_[bOffset], curN);

                        // yLocal += aVal * bLocal
                        AscendC::Muls(bLocal, bLocal, aVal, curN);
                        AscendC::Add(yLocal, yLocal, bLocal, curN);

                        inQueueB_.FreeTensor(bLocal);
                    }

                    tmpQueueA_.FreeTensor(aLocal);
                }

                // Write back C segment
                outQueueC_.EnQue<float>(yLocal);
                AscendC::LocalTensor<float> yForCopy = outQueueC_.DeQue<float>();
                uint64_t cOffset = ((uint64_t)b * M_ + i) * N_ + nBase;
                AscendC::DataCopy(C_gm_[cOffset], yForCopy, curN);
                outQueueC_.FreeTensor(yForCopy);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueA_;

    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t batch_ {0};
    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileN_ {0};
    uint32_t tileK_ {0};
};

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBatchedMatrixMultiplicationCustom op;
    op.Init(A, B, C,
            tiling_data.batch, tiling_data.m, tiling_data.k, tiling_data.n,
            tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
