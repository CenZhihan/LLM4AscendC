kernel_src="""
#include "kernel_operator.h"

class KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t din, uint32_t hin, uint32_t win,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout,
                                uint32_t tileW,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t padD, uint32_t padH, uint32_t padW,
                                uint32_t dilD, uint32_t dilH, uint32_t dilW,
                                uint32_t groups)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        din_ = din; hin_ = hin; win_ = win;
        kd_ = kd; kh_ = kh; kw_ = kw;
        dout_ = dout; hout_ = hout; wout_ = wout;
        tileW_ = tileW;
        sD_ = strideD; sH_ = strideH; sW_ = strideW;
        pD_ = padD; pH_ = padH; pW_ = padW;
        dD_ = dilD; dH_ = dilH; dW_ = dilW;
        groups_ = groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_)*cin_*din_*hin_*win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cin_)*cout_*kd_*kh_*kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_)*cout_*dout_*hout_*wout_);

        pipe_.InitBuffer(yBuf_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(xBuf_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_==0 || cin_==0 || cout_==0) return;
        const uint64_t tasks = static_cast<uint64_t>(n_) * cout_ * dout_ * hout_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t t = blockIdx; t < tasks; t += blockNum) {
            // decode t -> (n, co, d, h)
            uint64_t tmp = t;
            uint32_t h = static_cast<uint32_t>(tmp % hout_); tmp /= hout_;
            uint32_t d = static_cast<uint32_t>(tmp % dout_); tmp /= dout_;
            uint32_t co = static_cast<uint32_t>(tmp % cout_); tmp /= cout_;
            uint32_t n = static_cast<uint32_t>(tmp);

            // iterate across width in tiles
            for (uint32_t baseW = 0; baseW < wout_; baseW += tileW_) {
                uint32_t curW = wout_ - baseW;
                if (curW > tileW_) curW = tileW_;

                AscendC::LocalTensor<float> yLocal = yBuf_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, curW);

                // Accumulate contributions
                for (uint32_t ci = 0; ci < cin_; ++ci) {
                    for (uint32_t kd = 0; kd < kd_; ++kd) {
                        // di such that d == di*strideD - padD + kd*dilD
                        // For stride=1,pad=0,dil=1: di = d - kd
                        // Use general inverse mapping: di must satisfy d + pD - kd*dD divisible by sD
                        int64_t numerD = static_cast<int64_t>(d) + pD_ - static_cast<int64_t>(kd) * dD_;
                        if (numerD < 0 || numerD % sD_ != 0) continue;
                        uint32_t di = static_cast<uint32_t>(numerD / sD_);
                        if (di >= din_) continue;

                        for (uint32_t kh = 0; kh < kh_; ++kh) {
                            int64_t numerH = static_cast<int64_t>(h) + pH_ - static_cast<int64_t>(kh) * dH_;
                            if (numerH < 0 || numerH % sH_ != 0) continue;
                            uint32_t hi = static_cast<uint32_t>(numerH / sH_);
                            if (hi >= hin_) continue;

                            for (uint32_t kw = 0; kw < kw_; ++kw) {
                                int64_t numerW = static_cast<int64_t>(baseW) + pW_ - static_cast<int64_t>(kw) * dW_;
                                // handle segment intersection within current tile
                                // Valid y positions in this tile: w in [baseW, baseW+curW-1]
                                // corresponding wi must satisfy w = wi*sW - pW + kw*dW
                                // => wi = (w + pW - kw*dW) / sW and integer
                                // We'll compute the overlapping range explicitly by brute force step 1 when sW==1; else approximate by clipping
                                // For general sW, we compute startW as first w in tile that satisfies congruence modulo sW
                                // But keep simple for sW==1 (common case)
                                if (sW_ == 1) {
                                    // w range where wi in [0, win-1]: w in [kw - pW, kw - pW + win - 1]
                                    int64_t minW = static_cast<int64_t>(kw) * dW_ - pW_;
                                    int64_t maxW = minW + static_cast<int64_t>(win_) - 1;
                                    int64_t startW = baseW > static_cast<uint32_t>(minW) ? baseW : static_cast<uint32_t>(minW);
                                    int64_t endW = (baseW + curW - 1) < static_cast<uint32_t>(maxW) ? (baseW + curW - 1) : static_cast<uint32_t>(maxW);
                                    if (startW <= endW) {
                                        uint32_t len2 = static_cast<uint32_t>(endW - startW + 1);
                                        uint32_t wi0 = static_cast<uint32_t>(startW - (static_cast<int64_t>(kw) * dW_ - pW_));
                                        // Load x segment
                                        uint64_t xoff = OffsetX(n, ci, di, hi, wi0);
                                        AscendC::LocalTensor<float> xLocal = xBuf_.AllocTensor<float>();
                                        AscendC::DataCopy(xLocal, xGm_[xoff], len2);

                                        // multiply by weight scalar
                                        float wScalar = *(wGm_.GetPhyAddr() + OffsetW(ci, co, kd, kh, kw));
                                        AscendC::Muls(xLocal, xLocal, wScalar, len2);

                                        // yLocal segment add: yLocal[(startW-baseW):(startW-baseW)+len2-1] += xLocal
                                        uint32_t yoff = static_cast<uint32_t>(startW - baseW);
                                        // Add into the subrange
                                        AscendC::Add(yLocal[yoff], yLocal[yoff], xLocal, len2);

                                        xBuf_.FreeTensor(xLocal);
                                    }
                                } else {
                                    // General sW: iterate valid positions in this tile that align to sW.
                                    int64_t tileStart = baseW;
                                    int64_t tileEnd = baseW + curW - 1;
                                    // Find the smallest w>=tileStart s.t. (w + pW - kw*dW) % sW == 0 and wi in [0, win-1].
                                    for (int64_t wpos = tileStart; wpos <= tileEnd; ++wpos) {
                                        int64_t wi_num = wpos + pW_ - static_cast<int64_t>(kw) * dW_;
                                        if (wi_num < 0 || wi_num % sW_ != 0) continue;
                                        uint32_t wi = static_cast<uint32_t>(wi_num / sW_);
                                        if (wi >= win_) continue;

                                        // Load single element x and accumulate
                                        uint64_t xoff = OffsetX(n, ci, di, hi, wi);
                                        AscendC::LocalTensor<float> xLocal = xBuf_.AllocTensor<float>();
                                        AscendC::DataCopy(xLocal, xGm_[xoff], 1);

                                        float wScalar = *(wGm_.GetPhyAddr() + OffsetW(ci, co, kd, kh, kw));
                                        AscendC::Muls(xLocal, xLocal, wScalar, 1);

                                        uint32_t yoff = static_cast<uint32_t>(wpos - baseW);
                                        AscendC::Add(yLocal[yoff], yLocal[yoff], xLocal, 1);

                                        xBuf_.FreeTensor(xLocal);
                                    }
                                } // sW
                            } // kw
                        } // kh
                    } // kd
                } // ci

                // write out yLocal to GM
                uint64_t yoff = OffsetY(n, co, d, h, baseW);
                AscendC::DataCopy(yGm_[yoff], yLocal, curW);
                yBuf_.FreeTensor(yLocal);
            } // baseW
        } // tasks
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        return ((((uint64_t)n * cin_ + c) * din_ + d) * hin_ + h) * win_ + w;
    }
    __aicore__ inline uint64_t OffsetW(uint32_t ci, uint32_t co, uint32_t kd, uint32_t kh, uint32_t kw) const
    {
        return ((((uint64_t)ci * cout_ + co) * kd_ + kd) * kh_ + kh) * kw_ + kw;
    }
    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        return ((((uint64_t)n * cout_ + c) * dout_ + d) * hout_ + h) * wout_ + w;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf yBuf_;
    AscendC::TBuf xBuf_;
    AscendC::TBuf tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t din_{0}, hin_{0}, win_{0};
    uint32_t kd_{0}, kh_{0}, kw_{0};
    uint32_t dout_{0}, hout_{0}, wout_{0};
    uint32_t tileW_{0};

    uint32_t sD_{1}, sH_{1}, sW_{1};
    uint32_t pD_{0}, pH_{0}, pW_{0};
    uint32_t dD_{1}, dH_{1}, dW_{1};
    uint32_t groups_{1};
};
