host_operator_src="""

#include "matmul_with_small_k_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

} // anonymous

namespace optiling {

// UB model bytes needed: 4 * (tileM*K + K*tileN + tileM*tileN + tileN)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t MIN_TILE = 16;
static constexpr uint32_t ALIGN_TILE = 16;
static constexpr uint32_t INIT_TILE = 128;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(a_shape->GetDim(0));
    const uint32_t K1 = static_cast<uint32_t>(a_shape->GetDim(1));
    const uint32_t K2 = static_cast<uint32_t>(b_shape->GetDim(0));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetDim(1));
    if (K1 != K2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t K = K1;

    // Choose tile sizes to fit UB budget with small K emphasized
    uint32_t tileM = INIT_TILE;
    uint32_t tileN = INIT_TILE;

    auto usage_bytes = [&](uint32_t tm, uint32_t tn) -> uint64_t {
        uint64_t tm_u = tm, tn_u = tn, k_u = K;
        return 4ull * (tm_u * k_u + k_u * tn_u + tm_u * tn_u + tn_u);
    };

    // Ensure tiles are aligned and >= MIN_TILE
    tileM = AlignDown(tileM, ALIGN_TILE);
    tileN = AlignDown(tileN, ALIGN_TILE);
    if (tileM < MIN_TILE) tileM = MIN_TILE;
    if (tileN < MIN_TILE) tileN = MIN_TILE;

    // Reduce tileN until usage fits, then reduce tileM if still too big
    while (usage_bytes(tileM, tileN) > UB_BUDGET_BYTES && tileN > MIN_TILE) {
        tileN = AlignDown(tileN / 2u, ALIGN_TILE);
        if (tileN < MIN_TILE) tileN = MIN_TILE;
    }
    while (usage_bytes(tileM, tileN) > UB_BUDGET_BYTES && tileM > MIN_TILE) {
        tileM = AlignDown(tileM / 2u, ALIGN_TILE);
        if (tileM < MIN_TILE) tileM = MIN_TILE;
    }

    // Final guard
    if (usage_bytes(tileM, tileN) > UB_BUDGET_BYTES) {
        // As a last resort, clamp to MIN_TILE
        tileM = MIN_TILE;
        tileN = MIN_TILE;
    }

    const uint32_t tilesM = CeilDiv(M, tileM);
    const uint32_t tilesN = CeilDiv(N, tileN);
    uint32_t totalTiles = tilesM * tilesN;
    if (totalTiles == 0u) totalTiles = 1u;

    // Conservative block dim
    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    optiling::MatmulWithSmallKDimensionCustomTilingData tiling;
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    if (a_shape->GetDim(1) != b_shape->GetDim(0)) {
        return GRAPH_FAILED;
    }
    gert::Shape* c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, a_shape->GetDim(0));
    c_shape->SetDim(1, b_shape->GetDim(1));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

class KernelMatmulWithSmallKDimensionCustom {
public:
    __aicore__ inline KernelMatmulWithSmallKDimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, (uint64_t)M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, (uint64_t)K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, (uint64_t)M_ * N_);

        // Allocate UB buffers
        pipe_.InitBuffer(aBuf_, 1, (uint64_t)tileM_ * K_ * sizeof(float));
        pipe_.InitBuffer(bBuf_, 1, (uint64_t)K_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(cBuf_, 1, (uint64_t)tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, (uint64_t)tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t tilesM = CeilDiv(M_, tileM_);
        const uint32_t tilesN = CeilDiv(N_, tileN_);
        const uint32_t totalTiles = tilesM * tilesN;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            const uint32_t mTileIdx = tileIdx / tilesN;
            const uint32_t nTileIdx = tileIdx % tilesN;

            const uint32_t mStart = mTileIdx * tileM_;
            const uint32_t nStart = nTileIdx * tileN_;

            const uint32_t mLen = Min(tileM_, M_ - mStart);
            const uint32_t nLen = Min(tileN_, N_ - nStart);
            const uint32_t kLen = K_;

            // Allocate local tensors
            AscendC::LocalTensor<float> aLocal = aBuf_.AllocTensor<float>();
            AscendC::LocalTensor<float> bLocal = bBuf_.AllocTensor<float>();
            AscendC::LocalTensor<float> cLocal = cBuf_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpLocal = tmpBuf_.AllocTensor<float>();

            // Load A tile: contiguous block mLen * kLen
            const uint64_t aGmOffset = (uint64_t)mStart * K_;
            AscendC::DataCopy(aLocal, aGm_[aGmOffset], (uint64_t)mLen * kLen);

            // Load B tile: row by row due to column slicing
            for (uint32_t kk = 0; kk < kLen; ++kk) {
                AscendC::DataCopy(bLocal[kk * tileN_], bGm_[(uint64_t)kk * N_ + nStart], nLen);
            }

            // Initialize C tile to zeros
            AscendC::Dup(cLocal, 0.0f, (uint64_t)mLen * nLen);

            // Compute: C(mLen x nLen) += A(mLen x kLen) * B(kLen x nLen)
            for (uint32_t kk = 0; kk < kLen; ++kk) {
                AscendC::LocalTensor<float> bRow = bLocal[kk * tileN_];
                for (uint32_t mi = 0; mi < mLen; ++mi) {
                    const float aVal = aLocal[(uint64_t)mi * K_ + kk];
                    AscendC::LocalTensor<float> cRow = cLocal[(uint64_t)mi * tileN_];
                    // tmp = bRow * aVal
                    AscendC::Muls(tmpLocal, bRow, aVal, nLen);
                    // cRow += tmp
                    AscendC::Add(cRow, cRow, tmpLocal, nLen);
                }
            }

            // Store C tile: row by row
            for (uint32_t mi = 0; mi < mLen; ++mi) {
                AscendC::DataCopy(cGm_[(uint64_t)(mStart + mi) * N_ + nStart],
                                  cLocal[(uint64_t)mi * tileN_], nLen);
            }

            // Free locals
            tmpBuf_.FreeTensor(tmpLocal);
            cBuf_.FreeTensor(cLocal);
            bBuf_.FreeTensor(bLocal);
            aBuf_.FreeTensor(aLocal);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
        return (a + b - 1u) / b;
    }

    __aicore__ inline uint32_t Min(uint32_t a, uint32_t b) {
        return (a < b) ? a : b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> aBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> cBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileM_{0};
    uint32_t tileN_{0};
};

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithSmallKDimensionCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
