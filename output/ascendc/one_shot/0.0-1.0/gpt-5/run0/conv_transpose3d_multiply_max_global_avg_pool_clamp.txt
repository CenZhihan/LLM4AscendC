#include "kernel_operator.h"
#include <cfloat>

class KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom {
public:
    __aicore__ inline KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Din, uint32_t Hin, uint32_t Win,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Dout, uint32_t Hout, uint32_t Wout,
                                uint32_t hasBias)
    {
        xGm_ = (__gm__ float*)x;
        wGm_ = (__gm__ float*)w;
        bGm_ = (__gm__ float*)b;
        yGm_ = (__gm__ float*)y;

        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Din_ = Din; Hin_ = Hin; Win_ = Win;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        Dout_ = Dout; Hout_ = Hout; Wout_ = Wout;
        hasBias_ = hasBias;

        volIn_  = Din_ * Hin_ * Win_;
        volOut_ = Dout_ * Hout_ * Wout_;

        // Precompute strides
        x_n_stride_  = Cin_ * volIn_;
        x_c_stride_  = volIn_;
        x_d_stride_  = Hin_ * Win_;
        x_h_stride_  = Win_;
        x_w_stride_  = 1u;

        w_ci_stride_ = Cout_ * (Kd_ * Kh_ * Kw_);
        w_co_stride_ = (Kd_ * Kh_ * Kw_);
        w_kd_stride_ = (Kh_ * Kw_);
        w_kh_stride_ = Kw_;
        w_kw_stride_ = 1u;

        y_nc_stride_ = 1u; // y buffer is [N, C, 1, 1, 1], we will index as linear (N*C)
    }

    __aicore__ inline void Process()
    {
        // Single-block kernel by design for correct global max reduction
        const uint32_t bid = (uint32_t)AscendC::GetBlockIdx();
        if (bid != 0u) {
            return;
        }

        float globalMax = -FLT_MAX;

        // First pass: compute deconv outputs element-wise, accumulate per (n,c) sums, update global max.
        for (uint32_t n = 0; n < N_; ++n) {
            for (uint32_t co = 0; co < Cout_; ++co) {
                float sumNC = 0.0f;

                for (uint32_t od = 0; od < Dout_; ++od) {
                    for (uint32_t oh = 0; oh < Hout_; ++oh) {
                        for (uint32_t ow = 0; ow < Wout_; ++ow) {
                            float acc = hasBias_ ? bGm_[co] : 0.0f;

                            // deconvolution accumulation
                            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                                for (uint32_t kd = 0; kd < Kd_; ++kd) {
                                    // input index along depth
                                    int32_t id = (int32_t)od - (int32_t)kd;
                                    if (id < 0 || id >= (int32_t)Din_) continue;

                                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                                        int32_t ih = (int32_t)oh - (int32_t)kh;
                                        if (ih < 0 || ih >= (int32_t)Hin_) continue;

                                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                            int32_t iw = (int32_t)ow - (int32_t)kw;
                                            if (iw < 0 || iw >= (int32_t)Win_) continue;

                                            uint32_t x_index =
                                                n * x_n_stride_ +
                                                ci * x_c_stride_ +
                                                (uint32_t)id * x_d_stride_ +
                                                (uint32_t)ih * x_h_stride_ +
                                                (uint32_t)iw * x_w_stride_;

                                            uint32_t w_index =
                                                ci * w_ci_stride_ +
                                                co * w_co_stride_ +
                                                kd * w_kd_stride_ +
                                                kh * w_kh_stride_ +
                                                kw * w_kw_stride_;

                                            acc += xGm_[x_index] * wGm_[w_index];
                                        }
                                    }
                                }
                            }

                            if (acc > globalMax) {
                                globalMax = acc;
                            }
                            sumNC += acc;
                        }
                    }
                }

                // Store the sum for (n,co) in y buffer temporarily (linearized over N*Cout)
                uint32_t lin = n * Cout_ + co;
                yGm_[lin] = sumNC;
            }
        }

        const float scale = globalMax;
        const float invCount = 1.0f / static_cast<float>(volOut_);

        // Second pass: finalize pooled value per (n,c) = clamp(scale * avg, 0, 1), write to [N,C,1,1,1]
        for (uint32_t n = 0; n < N_; ++n) {
            for (uint32_t co = 0; co < Cout_; ++co) {
                uint32_t lin = n * Cout_ + co;
                float avg = yGm_[lin] * invCount;
                float val = scale * avg;
                if (val < 0.0f) val = 0.0f;
                if (val > 1.0f) val = 1.0f;

                // y layout: [N, C, 1, 1, 1] is contiguous; position is lin
                yGm_[lin] = val;
            }
        }
    }

private:
    __gm__ float* xGm_ {nullptr};
    __gm__ float* wGm_ {nullptr};
    __gm__ float* bGm_ {nullptr};
    __gm__ float* yGm_ {nullptr};

    uint32_t N_ {0}, Cin_ {0}, Cout_ {0};
    uint32_t Din_ {0}, Hin_ {0}, Win_ {0};
    uint32_t Kd_ {0}, Kh_ {0}, Kw_ {0};
    uint32_t Dout_ {0}, Hout_ {0}, Wout_ {0};
    uint32_t hasBias_ {0};

    uint32_t volIn_ {0}, volOut_ {0};

    // Strides for x
    uint32_t x_n_stride_ {0};
    uint32_t x_c_stride_ {0};
    uint32_t x_d_stride_ {0};
    uint32_t x_h_stride_ {0};
    uint32_t x_w_stride_ {0};

    // Strides for w
    uint32_t w_ci_stride_ {0};
    uint32_t w_co_stride_ {0};
    uint32_t w_kd_stride_ {0};
    uint32_t w_kh_stride_ {0};
    uint32_t w_kw_stride_ {0};

    // y stride (linearized N*C)
    uint32_t y_nc_stride_ {0};
};

extern "C" __global__ __aicore__ void conv_transpose3d_multiply_max_global_avg_pool_clamp_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{

            tiling_data.Dout, tiling_data.Hout, tiling_data.Wout,
            tiling_data.hasBias);
    op.Process();
}
