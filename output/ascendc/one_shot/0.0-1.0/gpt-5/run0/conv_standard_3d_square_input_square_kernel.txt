static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    // Expect shapes:
    // x: [N, C, D, H, W]
    // w: [M, C, KD=3, KH=3, KW=3]
    const uint32_t n = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t c = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t d = static_cast<uint32_t>(x_shape->GetDim(2));

    const uint32_t kc = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t kd = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t kh = static_cast<uint32_t>(w_shape->GetDim(3));
    const uint32_t kw = static_cast<uint32_t>(w_shape->GetDim(4));

    tiling.set_kh(kh);

kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t in_w,
                                uint32_t m, uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout,
                                uint32_t lineTileW)
    {
        n_ = n; c_ = c; d_ = d; h_ = h; w_ = in_w;
        m_ = m; kd_ = kd; kh_ = kh; kw_ = kw;
        dout_ = dout; hout_ = hout; wout_ = wout;
        lineTileW_ = lineTileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_) * c_ * d_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, m_ * c_ * kd_ * kh_ * kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_) * m_ * dout_ * hout_ * wout_);

        // Precompute strides for faster indexing
        sNin_ = c_ * d_ * h_ * w_;
        sCin_ = d_ * h_ * w_;
        sDin_ = h_ * w_;
        sHin_ = w_;

        sMwt_ = c_ * kd_ * kh_ * kw_;
        sCwt_ = kd_ * kh_ * kw_;
        sKZwt_ = kh_ * kw_;
        sKYwt_ = kw_;

        sNout_ = m_ * dout_ * hout_ * wout_;
        sMout_ = dout_ * hout_ * wout_;
        sDout_ = hout_ * wout_;
        sHout_ = wout_;
    }

    __aicore__ inline void Process()
    {
        if (dout_ == 0 || hout_ == 0 || wout_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalLines = static_cast<uint64_t>(n_) * static_cast<uint64_t>(m_) *
                                    static_cast<uint64_t>(dout_) * static_cast<uint64_t>(hout_);

        // Grid-stride over output lines (fixed n, m, d, h), compute all w positions per line
        for (uint64_t line = blockIdx; line < totalLines; line += blockNum) {
            uint32_t tmp = static_cast<uint32_t>(line);

            const uint32_t h_out = tmp % hout_;
            tmp /= hout_;
            const uint32_t d_out = tmp % dout_;
            tmp /= dout_;
            const uint32_t m_out = tmp % m_;
            tmp /= m_;
            const uint32_t n_out = tmp;

            // Compute each output along W for this (n_out, m_out, d_out, h_out)
            for (uint32_t w_out = 0; w_out < wout_; ++w_out) {
                float acc = 0.0f;

                // Convolution accumulation: sum over c, kz, ky, kx
                for (uint32_t c_in = 0; c_in < c_; ++c_in) {
                    for (uint32_t kz = 0; kz < 3; ++kz) {
                        const uint32_t id = d_out + kz; // valid since dout = d-2
                        for (uint32_t ky = 0; ky < 3; ++ky) {
                            const uint32_t ih = h_out + ky;
                            // base input offset for current (n,c,id,ih,w_out)
                            const uint32_t baseIn =
                                n_out * sNin_ + c_in * sCin_ + id * sDin_ + ih * sHin_ + w_out;

                            // base weight offset for current (m,c,kz,ky,0)
                            const uint32_t baseWt =
                                m_out * sMwt_ + c_in * sCwt_ + kz * sKZwt_ + ky * sKYwt_;

                            // kx loop (KW=3), contiguous in memory
                            float x0 = *(xGm_.GetPtr() + baseIn + 0);
                            float w0 = *(wGm_.GetPtr() + baseWt + 0);
                            acc += x0 * w0;

                            float x1 = *(xGm_.GetPtr() + baseIn + 1);
                            float w1 = *(wGm_.GetPtr() + baseWt + 1);
                            acc += x1 * w1;

                            float x2 = *(xGm_.GetPtr() + baseIn + 2);
                            float w2 = *(wGm_.GetPtr() + baseWt + 2);
                            acc += x2 * w2;
                        }
                    }
                }

                const uint32_t outOffset =
                    n_out * sNout_ + m_out * sMout_ + d_out * sDout_ + h_out * sHout_ + w_out;
                *(yGm_.GetPtr() + outOffset) = acc;
            }
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, d_{0}, h_{0}, w_{0};
    uint32_t m_{0}, kd_{3}, kh_{3}, kw_{3};
    uint32_t dout_{0}, hout_{0}, wout_{0};
    uint32_t lineTileW_{0};

    // Strides
    uint32_t sNin_{0}, sCin_{0}, sDin_{0}, sHin_{0};
    uint32_t sMwt_{0}, sCwt_{0}, sKZwt_{0}, sKYwt_{0};
    uint32_t sNout_{0}, sMout_{0}, sDout_{0}, sHout_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.m, tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.dout, tiling_data.hout, tiling_data.wout,
            tiling_data.lineTileW);
    op.Process();
}

    TORCH_CHECK(w.dim() == 5, "Weight must be 5D [M, C, KD, KH, KW]");

    auto Hout = H - 2;
    auto Wout = W - 2;
