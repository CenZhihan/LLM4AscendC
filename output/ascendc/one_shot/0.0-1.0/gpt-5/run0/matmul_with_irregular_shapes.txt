host_operator_src="""

#include "matmul_with_irregular_shapes_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// UB memory model (floats):
//   A_sub: tileM * tileK
//   B_sub: tileK * tileN
//   C_sub: tileM * tileN
//   tmpVec: tileN
// Total floats = tileM*tileK + tileK*tileN + tileM*tileN + tileN
// Bytes = 4 * Total floats
namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core conservative
static constexpr uint32_t ALIGN_ELEMS = 16;             // align to vector lanes
static constexpr uint32_t MAX_TILE_M = 16;
static constexpr uint32_t MAX_TILE_N = 256;
static constexpr uint32_t MAX_TILE_K = 256;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithIrregularShapesCustomTilingData tiling;

    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    const uint32_t K2 = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(0));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(1));
    if (K != K2) {
        return ge::GRAPH_FAILED;
    }

    // Initial tile guesses
    uint32_t tileM = (M >= 8) ? 8u : ((M >= 4) ? 4u : 1u);
    tileM = (tileM > MAX_TILE_M) ? MAX_TILE_M : tileM;
    uint32_t tileN = (N >= 128) ? 128u : AlignUp(N, ALIGN_ELEMS);
    if (tileN > MAX_TILE_N) tileN = MAX_TILE_N;
    if (tileN > N) tileN = AlignDown(N, ALIGN_ELEMS) > 0 ? AlignDown(N, ALIGN_ELEMS) : N;

    uint32_t tileK = (K >= 128) ? 128u : AlignUp(K, ALIGN_ELEMS);
    if (tileK > MAX_TILE_K) tileK = MAX_TILE_K;
    if (tileK > K) tileK = AlignDown(K, ALIGN_ELEMS) > 0 ? AlignDown(K, ALIGN_ELEMS) : K;

    auto fits = [](uint32_t tm, uint32_t tn, uint32_t tk) -> uint64_t {
        uint64_t floats = (uint64_t)tm * tk + (uint64_t)tk * tn + (uint64_t)tm * tn + tn;
        return 4ull * floats;
    };

    // Reduce tiles until fit UB budget
    while (fits(tileM, tileN, tileK) > UB_BUDGET_BYTES) {
        if (tileN > ALIGN_ELEMS) {
            uint32_t newTN = AlignDown(tileN / 2u, ALIGN_ELEMS);
            if (newTN >= ALIGN_ELEMS) tileN = newTN;
            else tileN = ALIGN_ELEMS;
        } else if (tileK > ALIGN_ELEMS) {
            uint32_t newTK = AlignDown(tileK / 2u, ALIGN_ELEMS);
            if (newTK >= ALIGN_ELEMS) tileK = newTK;
            else tileK = ALIGN_ELEMS;
        } else if (tileM > 1u) {
            tileM = tileM / 2u;
            if (tileM == 0) tileM = 1u;
        } else {
            break;
        }
    }

    if (tileM == 0) tileM = 1;
    if (tileN == 0) tileN = 1;
    if (tileK == 0) tileK = 1;

    const uint32_t tilesM = CeilDiv(M, tileM);
    const uint32_t tilesN = CeilDiv(N, tileN);
    const uint32_t totalTiles = tilesM * tilesN;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulWithIrregularShapesCustom {
public:
    __aicore__ inline KernelMatmulWithIrregularShapesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN; tileK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, (uint64_t)M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, (uint64_t)K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, (uint64_t)M_ * N_);

        pipe_.InitBuffer(aIn_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(bIn_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(cTile_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpVec_, BUFFER_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t tilesM = CeilDiv(M_, tileM_);
        const uint32_t tilesN = CeilDiv(N_, tileN_);
        const uint32_t totalTiles = tilesM * tilesN;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t tm = tileId / tilesN;
            const uint32_t tn = tileId % tilesN;

            const uint32_t mBase = tm * tileM_;
            const uint32_t nBase = tn * tileN_;
            const uint32_t mLen = MinU32(tileM_, M_ - mBase);
            const uint32_t nLen = MinU32(tileN_, N_ - nBase);

            AscendC::LocalTensor<float> aLocal = aIn_.AllocTensor<float>();
            AscendC::LocalTensor<float> bLocal = bIn_.AllocTensor<float>();
            AscendC::LocalTensor<float> cLocal = cTile_.AllocTensor<float>();
            AscendC::LocalTensor<float> tVec = tmpVec_.AllocTensor<float>();

            // Zero C tile
            AscendC::Duplicate(cLocal, 0.0f, tileM_ * tileN_);

            // Iterate K in chunks
            for (uint32_t kBase = 0; kBase < K_; kBase += tileK_) {
                const uint32_t kLen = MinU32(tileK_, K_ - kBase);

                // Load A_sub [mLen x kLen], row by row
                for (uint32_t m = 0; m < mLen; ++m) {
                    const uint32_t gmOff = (mBase + m) * K_ + kBase;
                    AscendC::LocalTensor<float> aRow = aLocal[m * tileK_];
                    AscendC::DataCopy(aRow, aGm_[gmOff], kLen);
                }

                // Load B_sub [kLen x nLen], row by row
                for (uint32_t kk = 0; kk < kLen; ++kk) {
                    const uint32_t gmOff = (kBase + kk) * N_ + nBase;
                    AscendC::LocalTensor<float> bRow = bLocal[kk * tileN_];
                    AscendC::DataCopy(bRow, bGm_[gmOff], nLen);
                }

                // Compute C_sub += A_sub * B_sub
                for (uint32_t m = 0; m < mLen; ++m) {
                    AscendC::LocalTensor<float> yRow = cLocal[m * tileN_];
                    for (uint32_t kk = 0; kk < kLen; ++kk) {
                        // Read a scalar from A_sub
                        float a_val = aLocal.GetValue(m * tileK_ + kk);
                        // Broadcast scalar into vector
                        AscendC::Duplicate(tVec, a_val, nLen);
                        // Multiply with B row kk
                        AscendC::LocalTensor<float> bRow = bLocal[kk * tileN_];
                        AscendC::Mul(tVec, tVec, bRow, nLen);
                        // Accumulate into yRow
                        AscendC::Add(yRow, yRow, tVec, nLen);
                    }
                }
            }

            // Store C_sub to GM, row by row
            for (uint32_t m = 0; m < mLen; ++m) {
                const uint32_t gmOff = (mBase + m) * N_ + nBase;
                AscendC::LocalTensor<float> yRow = cLocal[m * tileN_];
                AscendC::DataCopy(cGm_[gmOff], yRow, nLen);
            }

            tmpVec_.FreeTensor(tVec);
            cTile_.FreeTensor(cLocal);
            bIn_.FreeTensor(bLocal);
            aIn_.FreeTensor(aLocal);
        }
    }

private:
    __aicore__ static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
    __aicore__ static inline uint32_t MinU32(uint32_t a, uint32_t b) { return a < b ? a : b; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aIn_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bIn_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> cTile_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpVec_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileM_{0}, tileN_{0}, tileK_{0};
};

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithIrregularShapesCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
