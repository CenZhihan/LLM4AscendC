kernel_src="""
#include "kernel_operator.h"

class KernelSqueezeNetFireModuleCustom {
public:
    __aicore__ inline KernelSqueezeNetFireModuleCustom() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR w_sq, GM_ADDR b_sq,
                                GM_ADDR w_e1, GM_ADDR b_e1,
                                GM_ADDR w_e3, GM_ADDR b_e3,
                                GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t w,
                                uint32_t cSq, uint32_t cE1, uint32_t cE3,
                                uint32_t pad3)
    {
        n_ = n; cIn_ = cIn; h_ = h; w_ = w;
        cSq_ = cSq; cE1_ = cE1; cE3_ = cE3; pad3_ = pad3;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * cIn_ * h_ * w_);
        wSqGm_.SetGlobalBuffer((__gm__ float*)w_sq, (uint64_t)cSq_ * cIn_);
        bSqGm_.SetGlobalBuffer((__gm__ float*)b_sq, cSq_);
        wE1Gm_.SetGlobalBuffer((__gm__ float*)w_e1, (uint64_t)cE1_ * cSq_);
        bE1Gm_.SetGlobalBuffer((__gm__ float*)b_e1, cE1_);
        wE3Gm_.SetGlobalBuffer((__gm__ float*)w_e3, (uint64_t)cE3_ * cSq_ * 3 * 3);
        bE3Gm_.SetGlobalBuffer((__gm__ float*)b_e3, cE3_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * (cE1_ + cE3_) * h_ * w_);
    }

    __aicore__ inline void Process()
    {
        const uint64_t totalPos = (uint64_t)n_ * h_ * w_;
        if (totalPos == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint64_t pos = blockIdx; pos < totalPos; pos += blockNum) {
            uint32_t n, h, w;
            DecodePos(pos, n, h, w);

            // Expand1x1 branch (uses s at center)
            for (uint32_t c1 = 0; c1 < cE1_; ++c1) {
                float acc1 = bE1Gm_[c1];
                for (uint32_t csq = 0; csq < cSq_; ++csq) {
                    float s_val = ComputeS(n, h, w, csq);
                    const uint64_t wIdx = (uint64_t)c1 * cSq_ + csq;
                    acc1 += s_val * wE1Gm_[wIdx];
                }
                if (acc1 < 0.f) acc1 = 0.f;
                const uint64_t yOff = OffsetY(n, c1, h, w);
                yGm_[yOff] = acc1;
            }

            // Expand3x3 branch (uses 3x3 neighborhood of s)
            // Accumulate into registers per c3 channel
            // Initialize with bias
            // Given cE3_ can be large; accumulate scalar per channel
            for (uint32_t c3 = 0; c3 < cE3_; ++c3) {
                e3acc(c3) = bE3Gm_[c3];
            }

            for (uint32_t kh = 0; kh < 3; ++kh) {
                int32_t hh = (int32_t)h + (int32_t)kh - (int32_t)pad3_;
                if (hh < 0 || hh >= (int32_t)h_) continue;
                for (uint32_t kw = 0; kw < 3; ++kw) {
                    int32_t ww = (int32_t)w + (int32_t)kw - (int32_t)pad3_;
                    if (ww < 0 || ww >= (int32_t)w_) continue;

                    for (uint32_t csq = 0; csq < cSq_; ++csq) {
                        float s_val = ComputeS(n, (uint32_t)hh, (uint32_t)ww, csq);
                        const uint64_t baseIdx = ((uint64_t)kh * 3 + kw);
                        for (uint32_t c3 = 0; c3 < cE3_; ++c3) {
                            const uint64_t wIdx = (((uint64_t)c3 * cSq_ + csq) * 3 + kh) * 3 + kw;
                            e3acc(c3) += s_val * wE3Gm_[wIdx];
                        }
                    }
                }
            }

            for (uint32_t c3 = 0; c3 < cE3_; ++c3) {
                float v = e3acc(c3);
                if (v < 0.f) v = 0.f;
                const uint64_t yOff = OffsetY(n, cE1_ + c3, h, w);
                yGm_[yOff] = v;
            }
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return ((((uint64_t)n * cIn_ + c) * h_ + h) * w_ + w);
    }
    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return ((((uint64_t)n * (cE1_ + cE3_) + c) * h_ + h) * w_ + w);
    }

    __aicore__ inline float ComputeS(uint32_t n, uint32_t h, uint32_t w, uint32_t csq) const
    {
        float s = bSqGm_[csq];
        const uint64_t wBase = (uint64_t)csq * cIn_;
        for (uint32_t cin = 0; cin < cIn_; ++cin) {
            const float xv = xGm_[OffsetX(n, cin, h, w)];
            s += xv * wSqGm_[wBase + cin];
        }
        if (s < 0.f) s = 0.f;
        return s;
    }

    __aicore__ inline float & e3acc(uint32_t idx)
    {
        // Use a small statically sized array through shared local memory emulation.
        // cE3_ can vary, but we will map via GM scratch using a ring of registers.
        // For simplicity, allocate in private memory per thread with upper bound using dynamic indexing.
        return accBuf_[idx];
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wSqGm_;
    AscendC::GlobalTensor<float> bSqGm_;
    AscendC::GlobalTensor<float> wE1Gm_;
    AscendC::GlobalTensor<float> bE1Gm_;
    AscendC::GlobalTensor<float> wE3Gm_;
    AscendC::GlobalTensor<float> bE3Gm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, h_{0}, w_{0};
    uint32_t cSq_{0}, cE1_{0}, cE3_{0}, pad3_{1};

    // Note: For demonstration, allocate a reasonably sized accumulator buffer.
    // In production, tune or tile cE3_ to fit UB/register pressure.
    float accBuf_[512]; // supports up to 512 expand3 channels; adjust via tiling if needed.
};

extern "C" __global__ __aicore__ void squeeze_net_fire_module_custom(GM_ADDR x,
                                                                      GM_ADDR w_squeeze,
                                                                      GM_ADDR b_squeeze,
                                                                      GM_ADDR w_expand1,
                                                                      GM_ADDR b_expand1,
                                                                      GM_ADDR w_expand3,
                                                                      GM_ADDR b_expand3,
                                                                      GM_ADDR y,
                                                                      GM_ADDR workspace,
                                                                      GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSqueezeNetFireModuleCustom op;
    op.Init(x,
            w_squeeze, b_squeeze,
            w_expand1, b_expand1,
            w_expand3, b_expand3,
            y,
            tiling_data.n, tiling_data.cIn, tiling_data.h, tiling_data.w,
            tiling_data.cSq, tiling_data.cE1, tiling_data.cE3,
            tiling_data.pad3);
    op.Process();
}
