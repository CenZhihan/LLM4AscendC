host_operator_src="""
#include "average_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t FloorDiv(int64_t a, int64_t b) { return static_cast<uint32_t>(a / b); }
} // namespace

namespace optiling {

// UB not heavily used in this scalar-gather implementation, keep a modest tile to balance scheduling cost.
static constexpr uint32_t UB_BUDGET_BYTES = 64 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling3dCustomTilingData tiling;

    // Expect input as NCDHW
    const gert::Shape *in_shape = context->GetInputShape(0);
    if (in_shape->GetStorageShape().GetDimNum() != 5) {
        // Fallback to origin shape if storage not 5D
    }
    const auto &origin = in_shape->GetOriginShape();
    const uint32_t N  = static_cast<uint32_t>(origin.GetDim(0));
    const uint32_t C  = static_cast<uint32_t>(origin.GetDim(1));
    const uint32_t Di = static_cast<uint32_t>(origin.GetDim(2));
    const uint32_t Hi = static_cast<uint32_t>(origin.GetDim(3));
    const uint32_t Wi = static_cast<uint32_t>(origin.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *k_attr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *s_attr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *p_attr = attrs->GetAttrPointer<int32_t>(2);

    const uint32_t k = static_cast<uint32_t>(*k_attr);
    const uint32_t s_eff = static_cast<uint32_t>((*s_attr == 0) ? *k_attr : *s_attr);
    const uint32_t p = static_cast<uint32_t>(*p_attr);

    // PyTorch AvgPool3d default ceil_mode=False, count_include_pad=True
    const uint32_t Do = (Di + 2u * p >= k) ? FloorDiv(static_cast<int64_t>(Di + 2u * p - k), static_cast<int64_t>(s_eff)) + 1u : 0u;
    const uint32_t Ho = (Hi + 2u * p >= k) ? FloorDiv(static_cast<int64_t>(Hi + 2u * p - k), static_cast<int64_t>(s_eff)) + 1u : 0u;
    const uint32_t Wo = (Wi + 2u * p >= k) ? FloorDiv(static_cast<int64_t>(Wi + 2u * p - k), static_cast<int64_t>(s_eff)) + 1u : 0u;

    const uint64_t totalOut64 = static_cast<uint64_t>(N) * C * Do * Ho * Wo;
    const uint32_t totalOut = static_cast<uint32_t>(totalOut64);

    // Tile length heuristic
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOut < tileLen) {
        const uint32_t aligned = AlignDown(totalOut, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOut;
    }
    if (tileLen == 0) tileLen = 1;

    const uint32_t numTiles = CeilDiv(totalOut == 0 ? 1u : totalOut, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalOut(totalOut);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_Di(Di);
    tiling.set_Hi(Hi);
    tiling.set_Wi(Wi);

    tiling.set_Do(Do);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);

    tiling.set_k(k);
    tiling.set_s(s_eff);
    tiling.set_p(p);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin = x_shape->GetOriginShape();

    const int64_t N  = origin.GetDim(0);
    const int64_t C  = origin.GetDim(1);
    const int64_t Di = origin.GetDim(2);
    const int64_t Hi = origin.GetDim(3);
    const int64_t Wi = origin.GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *k_attr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *s_attr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *p_attr = attrs->GetAttrPointer<int32_t>(2);

    const int64_t k = static_cast<int64_t>(*k_attr);
    const int64_t s_eff = static_cast<int64_t>((*s_attr == 0) ? *k_attr : *s_attr);
    const int64_t p = static_cast<int64_t>(*p_attr);

    const int64_t Do = (Di + 2 * p - k) / s_eff + 1;
    const int64_t Ho = (Hi + 2 * p - k) / s_eff + 1;

    y_shape->SetStorageShape(y_origin);
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

class KernelAveragePooling3dCustom {
public:
    __aicore__ inline KernelAveragePooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOut, uint32_t tileLen,
                                uint32_t N, uint32_t C,
                                uint32_t Di, uint32_t Hi, uint32_t Wi,
                                uint32_t Do, uint32_t Ho, uint32_t Wo,
                                uint32_t k, uint32_t s, uint32_t p)
    {
        totalOut_ = totalOut;
        tileLen_ = (tileLen == 0) ? 1u : tileLen;

        N_ = N; C_ = C;
        Di_ = Di; Hi_ = Hi; Wi_ = Wi;
        Do_ = Do; Ho_ = Ho; Wo_ = Wo;

        k_ = k; s_ = s; p_ = p;
        denom_ = static_cast<float>(k_ * k_ * k_);

        xPtr_ = (__gm__ float*)x;
        yPtr_ = (__gm__ float*)y;
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOut_) return;

        while (base < totalOut_) {
            const uint32_t end = AscendC::Min(base + tileLen_, totalOut_);
            for (uint32_t outIdx = base; outIdx < end; ++outIdx) {
                ComputeOne(outIdx);
            }
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t outIdx)
    {
        // Map flat output index to (n,c,do,ho,wo)
        uint32_t tmp = outIdx;
        const uint32_t wo = tmp % Wo_; tmp /= Wo_;
        const uint32_t ho = tmp % Ho_; tmp /= Ho_;
        const uint32_t od = tmp % Do_; tmp /= Do_;
        const uint32_t c  = tmp % C_;  tmp /= C_;
        const uint32_t n  = tmp;

        const int32_t sd = static_cast<int32_t>(od) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);
        const int32_t sh = static_cast<int32_t>(ho) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);
        const int32_t sw = static_cast<int32_t>(wo) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);

        float sum = 0.0f;

        for (uint32_t kd = 0; kd < k_; ++kd) {
            const int32_t id = sd + static_cast<int32_t>(kd);
            if (id < 0 || id >= static_cast<int32_t>(Di_)) continue;
            for (uint32_t kh = 0; kh < k_; ++kh) {
                const int32_t ih = sh + static_cast<int32_t>(kh);
                if (ih < 0 || ih >= static_cast<int32_t>(Hi_)) continue;
                for (uint32_t kw = 0; kw < k_; ++kw) {
                    const int32_t iw = sw + static_cast<int32_t>(kw);
                    if (iw < 0 || iw >= static_cast<int32_t>(Wi_)) continue;

                    const uint32_t inIdx =
                        (((n * C_ + c) * Di_ + static_cast<uint32_t>(id)) * Hi_ + static_cast<uint32_t>(ih)) * Wi_ +
                        static_cast<uint32_t>(iw);
                    sum += xPtr_[inIdx];
                }
            }
        }

        yPtr_[outIdx] = sum / denom_;
    }

private:
    __gm__ float* xPtr_{nullptr};
    __gm__ float* yPtr_{nullptr};

    uint32_t totalOut_{0};
    uint32_t tileLen_{1};

    uint32_t N_{0}, C_{0};
    uint32_t Di_{0}, Hi_{0}, Wi_{0};
    uint32_t Do_{0}, Ho_{0}, Wo_{0};

    uint32_t k_{1}, s_{1}, p_{0};
    float denom_{1.0f};
};

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling3dCustom op;
    op.Init(x, y,
            tiling_data.totalOut, tiling_data.tileLength,
            tiling_data.N, tiling_data.C,
            tiling_data.Di, tiling_data.Hi, tiling_data.Wi,
            tiling_data.Do, tiling_data.Ho, tiling_data.Wo,
            tiling_data.k, tiling_data.s, tiling_data.p);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor make_output_like_avgpool3d(const at::Tensor& self, int64_t k, int64_t s, int64_t p) {
    const auto N = self.size(0);
    const auto C = self.size(1);
    const auto Di = self.size(2);
    const auto Hi = self.size(3);
    const auto Wi = self.size(4);

    const int64_t s_eff = (s == 0) ? k : s;
    const int64_t Do = (Di + 2 * p - k) / s_eff + 1;
    const int64_t Ho = (Hi + 2 * p - k) / s_eff + 1;
    const int64_t Wo = (Wi + 2 * p - k) / s_eff + 1;

    return at::empty({N, C, Do, Ho, Wo}, self.options());
}

at::Tensor average_pooling_3d_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding) {
    at::Tensor result = make_output_like_avgpool3d(self, kernel_size, stride, padding);
    EXEC_NPU_CMD(aclnnAveragePooling3dCustom, self, (int)kernel_size, (int)stride, (int)padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("average_pooling_3d_custom", &average_pooling_3d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("average_pooling_3d_custom", &average_pooling_3d_impl_npu, "3D Average Pooling (AveragePooling3dCustom on NPU)");
