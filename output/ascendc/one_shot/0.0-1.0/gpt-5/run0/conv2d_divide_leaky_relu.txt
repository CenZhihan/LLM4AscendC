    const uint64_t totalPixels = static_cast<uint64_t>(N) * Ho * Wo;
    uint32_t blockDim = static_cast<uint32_t>(totalPixels);

    if (blockDim > 24u) blockDim = 24u;

    tiling.set_cin(Cin);

    tiling.set_h(H);

    tiling.set_kh(Kh);
    tiling.set_kw(Kw);
    tiling.set_ho(Ho);
    tiling.set_wo(Wo);

    tiling.set_divisor(div);

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dDivideLeakyReluCustom {
public:
    __aicore__ inline KernelConv2dDivideLeakyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo,
                                float divisor, float negativeSlope)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;

        divisor_ = divisor;
        invDiv_ = (divisor_ != 0.0f) ? (1.0f / divisor_) : 0.5f; // expect 2.0
        negativeSlope_ = negativeSlope;
        negCorr_ = negativeSlope_ - 1.0f;

        const uint32_t xSize = N_ * Cin_ * H_ * W_;
        const uint32_t wSize = Cout_ * Cin_ * Kh_ * Kw_;
        const uint32_t bSize = Cout_;
        const uint32_t ySize = N_ * Cout_ * Ho_ * Wo_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        bGm_.SetGlobalBuffer((__gm__ float*)b, bSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // Buffers for scalar computations (length = 1)
        pipe_.InitBuffer(accBuf_, 1, sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, sizeof(float));
        pipe_.InitBuffer(tmp2Buf_, 1, sizeof(float));
        pipe_.InitBuffer(xBuf_, 1, sizeof(float));
        pipe_.InitBuffer(wBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || Ho_ == 0 || Wo_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint64_t totalPixels = (uint64_t)N_ * Ho_ * Wo_;
        uint64_t pix = blockIdx;
        while (pix < totalPixels) {
            // decode pixel -> (n, oh, ow)
            const uint32_t n = (uint32_t)(pix / (Ho_ * (uint64_t)Wo_));
            const uint32_t rem = (uint32_t)(pix % (Ho_ * (uint64_t)Wo_));
            const uint32_t oh = rem / Wo_;
            const uint32_t ow = rem % Wo_;

            // Loop over output channels; accumulate scalar
            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                AscendC::LocalTensor<float> acc = accBuf_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmp2 = tmp2Buf_.AllocTensor<float>();

                // acc = bias[oc]
                AscendC::DataCopy(acc, bGm_[oc], 1);

                for (uint32_t ic = 0; ic < Cin_; ++ic) {
                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        const uint32_t ih = oh + kh;
                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                            const uint32_t iw = ow + kw;

                            // x index
                            const uint64_t xIndex =
                                (((uint64_t)n * Cin_ + ic) * H_ + ih) * W_ + iw;
                            // w index
                            const uint64_t wIndex =
                                (((uint64_t)oc * Cin_ + ic) * Kh_ + kh) * Kw_ + kw;

                            AscendC::LocalTensor<float> xLoc = xBuf_.AllocTensor<float>();
                            AscendC::LocalTensor<float> wLoc = wBuf_.AllocTensor<float>();

                            AscendC::DataCopy(xLoc, xGm_[xIndex], 1);
                            AscendC::DataCopy(wLoc, wGm_[wIndex], 1);

                            // tmp = xLoc * wLoc
                            AscendC::Mul(tmp, xLoc, wLoc, 1);
                            // acc = acc + tmp
                            AscendC::Add(acc, acc, tmp, 1);

                            wBuf_.FreeTensor(wLoc);
                            xBuf_.FreeTensor(xLoc);
                        }
                    }
                }

                // Divide by divisor: acc *= invDiv_
                AscendC::Muls(acc, acc, invDiv_, 1);

                // Leaky ReLU: y = x + (negSlope - 1) * min(x, 0)
                AscendC::Mins(tmp2, acc, 0.0f, 1);           // tmp2 = min(acc, 0)
                AscendC::Muls(tmp2, tmp2, negCorr_, 1);      // tmp2 *= (negSlope - 1)
                AscendC::Add(acc, acc, tmp2, 1);             // acc += tmp2

                // Store to output
                const uint64_t yIndex =
                    (((uint64_t)n * Cout_ + oc) * Ho_ + oh) * Wo_ + ow;
                AscendC::DataCopy(yGm_[yIndex], acc, 1);

                tmp2Buf_.FreeTensor(tmp2);
                tmpBuf_.FreeTensor(tmp);
                accBuf_.FreeTensor(acc);
            }

            pix += blockNum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmp2Buf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    float divisor_{2.0f}, invDiv_{0.5f};
    float negativeSlope_{0.01f};
    float negCorr_{-0.99f};
};

extern "C" __global__ __aicore__ void conv2d_divide_leaky_relu_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConv2dDivideLeakyReluCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w, tiling_data.kh, tiling_data.kw,
            tiling_data.ho, tiling_data.wo,
            tiling_data.divisor, tiling_data.negativeSlope);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor ensure_bias(const c10::optional<at::Tensor>& bias_opt,
                                     int64_t cout, const at::Tensor& like) {
    if (bias_opt.has_value() && bias_opt.value().defined()) {
        return bias_opt.value().contiguous();
    }
    return at::zeros({cout}, like.options());
}

at::Tensor conv2d_divide_leaky_relu_impl_npu(const at::Tensor& x,
                                             const at::Tensor& weight,
                                             c10::optional<at::Tensor> bias_opt,
                                             double divisor) {
    TORCH_CHECK(x.dim() == 4, "x must be 4D NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be 4D OIHW");
    TORCH_CHECK(divisor == 2.0, "divisor must be 2.0 for Conv2dDivideLeakyReluCustom");

    const auto N = x.size(0);
    const auto Cin = x.size(1);
    const auto H = x.size(2);
    const auto W = x.size(3);
    const auto Cout = weight.size(0);
    const auto Cin_w = weight.size(1);
    const auto Kh = weight.size(2);
    const auto Kw = weight.size(3);

    TORCH_CHECK(Cin == Cin_w, "in_channels mismatch between x and weight");
    TORCH_CHECK(H >= Kh && W >= Kw, "Invalid kernel larger than input without padding");
    const auto Ho = H - Kh + 1;
    const auto Wo = W - Kw + 1;

    at::Tensor result = at::empty({N, Cout, Ho, Wo}, x.options());
    at::Tensor bias = ensure_bias(bias_opt, Cout, x);

    EXEC_NPU_CMD(aclnnConv2dDivideLeakyReluCustom, x, weight, bias, divisor, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_divide_leaky_relu_custom", &conv2d_divide_leaky_relu_impl_npu);
