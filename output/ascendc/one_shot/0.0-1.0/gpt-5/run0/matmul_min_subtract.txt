#include "matmul_min_subtract_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

// UB budget and alignment heuristics
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // per-core conservative share
static constexpr uint32_t ALIGN_ELEMS = 128;              // align tiles to favor DMA efficiency
static constexpr uint32_t MAX_BLOCKS = 24;                // cap blocks for stability

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMinSubtractCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    // Expect shapes:
    //   x: [M, K]
    //   w: [K, N]   (note: host binding will provide transposed weight)
    //   b: [N]
    const uint32_t x_dims = static_cast<uint32_t>(x_shape->GetOriginShape().GetDimNum());
    const uint32_t w_dims = static_cast<uint32_t>(w_shape->GetOriginShape().GetDimNum());
    const uint32_t b_dims = static_cast<uint32_t>(b_shape->GetOriginShape().GetDimNum());

    uint32_t M = 1, K = 1, N = 1;
    if (x_dims >= 2) {
        M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
        K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    } else {
        // Fallback: treat as vector
        M = 1;
        K = static_cast<uint32_t>(x_shape->GetOriginShape().GetShapeSize());
    }
    if (w_dims >= 2) {
        const uint32_t K_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
        const uint32_t N_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
        // Prefer host-transposed weight (K, N)
        K = K_w;
        N = N_w;
    } else {
        N = static_cast<uint32_t>(w_shape->GetOriginShape().GetShapeSize());
    }
    if (b_dims >= 1) {
        const uint32_t N_b = static_cast<uint32_t>(b_shape->GetOriginShape().GetShapeSize());
        // Ensure bias length matches N if possible
        if (N_b != 0) {
            N = N_b;
        }
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *c_ptr = attrs->GetAttrPointer<float>(0);
    const float cVal = (c_ptr != nullptr) ? *c_ptr : 0.0f;

    // Tile sizing: xSlice (tileK) + acc(tileN) + wVec(tileN) + bVec(tileN) ~ tileK + 3*tileN floats
    const uint32_t budgetFloats = UB_BUDGET_BYTES / 4;
    uint32_t tileN = AlignDown(1024u, ALIGN_ELEMS);
    uint32_t tileK = AlignDown(4096u, ALIGN_ELEMS);

    // Bound by dimensions
    if (N < tileN) tileN = (N >= ALIGN_ELEMS) ? AlignDown(N, ALIGN_ELEMS) : N;
    if (tileN == 0) tileN = (N == 0) ? 1u : N;

    if (K < tileK) tileK = (K >= ALIGN_ELEMS) ? AlignDown(K, ALIGN_ELEMS) : K;
    if (tileK == 0) tileK = (K == 0) ? 1u : K;

    // Ensure within UB budget
    // floatsNeeded ~ tileK + 3*tileN
    while (tileK + 3 * tileN > budgetFloats && (tileK > ALIGN_ELEMS || tileN > ALIGN_ELEMS)) {
        if (tileK >= tileN && tileK > ALIGN_ELEMS) {
            tileK = AlignDown(tileK / 2, ALIGN_ELEMS);
            if (tileK == 0) tileK = ALIGN_ELEMS;
        } else if (tileN > ALIGN_ELEMS) {
            tileN = AlignDown(tileN / 2, ALIGN_ELEMS);
            if (tileN == 0) tileN = ALIGN_ELEMS;
        } else {
            break;
        }
    }

    // Blocks over M (rows)
    uint32_t blockDim = M;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);
    tiling.set_cVal(cVal);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulMinSubtractCustom {
public:
    __aicore__ inline KernelMatmulMinSubtractCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN, uint32_t tileK, float cVal)
    {
        M_ = M; N_ = N; K_ = K;
        tileN_ = tileN; tileK_ = tileK;
        cVal_ = cVal;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M_ * K_);
        // w is provided as (K, N) contiguous in memory
        wGm_.SetGlobalBuffer((__gm__ float*)w, K_ * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M_ * N_);

        // Buffers:
        //  xSlice: tileK
        //  accVec: tileN
        //  wVec:   tileN
        //  bVec:   tileN
        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(vecQueue_, 3, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            ComputeRow(m);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t m)
    {
        const uint32_t rowXOffset = m * K_;
        const uint32_t rowYOffset = m * N_;

        for (uint32_t n0 = 0; n0 < N_; n0 += tileN_) {
            uint32_t lenN = N_ - n0;
            if (lenN > tileN_) lenN = tileN_;

            AscendC::LocalTensor<float> acc = vecQueue_.AllocTensor<float>();
            // acc = 0
            AscendC::Muls(acc, acc, 0.0f, lenN);

            // Iterate K in tiles
            for (uint32_t k0 = 0; k0 < K_; k0 += tileK_) {
                uint32_t lenK = K_ - k0;
                if (lenK > tileK_) lenK = tileK_;

                // x slice for this row
                AscendC::LocalTensor<float> xSlice = xQueue_.AllocTensor<float>();
                AscendC::DataCopy(xSlice, xGm_[rowXOffset + k0], lenK);

                // For each k in slice: acc += x[k] * w[k, n0:n0+lenN]
                for (uint32_t i = 0; i < lenK; ++i) {
                    // Load w vector for k = (k0 + i)
                    AscendC::LocalTensor<float> wVec = vecQueue_.AllocTensor<float>();
                    const uint32_t wOffset = (k0 + i) * N_ + n0;
                    AscendC::DataCopy(wVec, wGm_[wOffset], lenN);

                    // xi = xSlice[i]
                    float xi = AscendC::GetValue(xSlice, i);

                    // temp = wVec * xi
                    AscendC::Muls(wVec, wVec, xi, lenN);
                    // acc += temp
                    AscendC::Add(acc, acc, wVec, lenN);

                    vecQueue_.FreeTensor(wVec);
                }

                xQueue_.FreeTensor(xSlice);
            }

            // Add bias
            AscendC::LocalTensor<float> bVec = vecQueue_.AllocTensor<float>();
            AscendC::DataCopy(bVec, bGm_[n0], lenN);
            AscendC::Add(acc, acc, bVec, lenN);

            // Post process: y = min(acc, cVal_) - cVal_
            AscendC::Mins(acc, acc, cVal_, lenN);
            AscendC::Adds(acc, acc, -cVal_, lenN);

            // Write back
            AscendC::DataCopy(yGm_[rowYOffset + n0], acc, lenN);

            vecQueue_.FreeTensor(bVec);
            vecQueue_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3> vecQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileN_{0}, tileK_{0};
    float cVal_{0.0f};
};

extern "C" __global__ __aicore__ void matmul_min_subtract_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMinSubtractCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileN, tiling_data.tileK, tiling_data.cVal);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_transposed_contiguous(const at::Tensor& w) {
    // Expect original w: [out_features, in_features]
    // Kernel expects w_t: [in_features, out_features]
    return w.transpose(0, 1).contiguous();
}

at::Tensor matmul_min_subtract_impl_npu(const at::Tensor& x,
                                        const at::Tensor& w,
                                        const at::Tensor& b,
                                        const at::Tensor& c_tensor) {
    TORCH_CHECK(x.dim() == 2, "x must be 2D [M, K]");
    TORCH_CHECK(w.dim() == 2, "w must be 2D [N, K]");
    TORCH_CHECK(b.dim() == 1, "b must be 1D [N]");
    TORCH_CHECK(c_tensor.numel() == 1, "c must be scalar tensor");

    // Prepare arguments
    auto w_t = to_transposed_contiguous(w); // [K, N]
    double c_val = static_cast<double>(c_tensor.item<float>());

    // Output: [M, N]
    auto M = x.size(0);
    auto N = w.size(0);
    at::Tensor result = at::empty({M, N}, x.options());

    EXEC_NPU_CMD(aclnnMatmulMinSubtractCustom, x, w_t, b, c_val, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_min_subtract_custom", &matmul_min_subtract_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_min_subtract_custom", &matmul_min_subtract_impl_npu,
          "MatMul + bias + min(c) - c (MatmulMinSubtractCustom on NPU)");
}
