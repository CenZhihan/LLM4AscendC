host_operator_src="""
#include "conv_transposed_1d_dilated_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model:
//   yLocal: 1 * tile
//   xLocal: 1 * tile
//   tmpLocal: 1 * tile
//   small temp for scalars/weights (negligible vs tile)
// Total floats ~= 3 * tile => bytes ~= 12 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed1dDilatedCustomTilingData tiling;

    const gert::Shape* xShape = context->GetInputShape(0);
    const gert::Shape* wShape = context->GetInputShape(1);

    // Shapes: x: [N, Cin, Lin], weight: [Cin, Cout, K]
    const uint32_t n   = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    const uint32_t cin = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    const uint32_t lin = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));

    const uint32_t wcin = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(0));
    const uint32_t cout = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(1));
    const uint32_t k    = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stride   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *padding  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilation = attrs->GetAttrPointer<int32_t>(2);

    // Basic validation
    if (cin != wcin) {
        return ge::GRAPH_FAILED;
    }

    // lout = (lin - 1) * stride - 2*padding + dilation*(k-1) + 1
    const uint32_t s = static_cast<uint32_t>(*stride);
    const uint32_t p = static_cast<uint32_t>(*padding);
    const uint32_t d = static_cast<uint32_t>(*dilation);
    const uint32_t lout = (lin > 0) ? ((lin - 1u) * s) + 1u + d * (k - 1u) - 2u * p : 0u;

    // Derive tile length from UB budget: bytes ~= 12 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 12u;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (lout < tileLen) {
        const uint32_t aligned = AlignDown(lout, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : lout;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(lout, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_cin(cin);
    tiling.set_cout(cout);
    tiling.set_lin(lin);
    tiling.set_lout(lout);
    tiling.set_k(k);
    tiling.set_stride(s);
    tiling.set_padding(p);
    tiling.set_dilation(d);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const int64_t n   = x_shape->GetDim(0);
    const int64_t cin = x_shape->GetDim(1);
    const int64_t lin = x_shape->GetDim(2);

    const int64_t wcin = w_shape->GetDim(0);
    const int64_t cout = w_shape->GetDim(1);
    const int64_t k    = w_shape->GetDim(2);

    if (cin != wcin) {
        return GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t stride   = *(attrs->GetAttrPointer<int32_t>(0));
    const int32_t padding  = *(attrs->GetAttrPointer<int32_t>(1));
    const int32_t dilation = *(attrs->GetAttrPointer<int32_t>(2));

    // lout = (lin - 1) * stride - 2*padding + dilation*(k-1) + 1
    const int64_t lout = (lin - 1) * static_cast<int64_t>(stride) - 2 * static_cast<int64_t>(padding)
                         + static_cast<int64_t>(dilation) * (k - 1) + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, n);
    y_shape->SetDim(1, cout);
    y_shape->SetDim(2, lout);

    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed1dDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed1dDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t lout, uint32_t k,
                                uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t tileLength)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        lin_ = lin; lout_ = lout; k_ = k;
        stride_ = stride; padding_ = padding; dilation_ = dilation;
        tileLen_ = tileLength;

        const uint32_t xSize = n_ * cin_ * lin_;
        const uint32_t ySize = n_ * cout_ * lout_;
        const uint32_t wSize = cin_ * cout_ * k_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float)); // for scalar weight fetch
    }

    __aicore__ inline void Process()
    {
        if (lout_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;

        while (base < lout_) {
            uint32_t curLen = lout_ - base;
            if (curLen > tileLen_) curLen = tileLen_;
            ComputeTile(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline uint32_t XIndex(uint32_t n, uint32_t ic, uint32_t pos) const
    {
        return ((n * cin_ + ic) * lin_) + pos;
    }
    __aicore__ inline uint32_t YIndex(uint32_t n, uint32_t oc, uint32_t pos) const
    {
        return ((n * cout_ + oc) * lout_) + pos;
    }
    __aicore__ inline uint32_t WIndex(uint32_t ic, uint32_t oc, uint32_t t) const
    {
        return ((ic * cout_ + oc) * k_) + t;
    }

    __aicore__ inline float LoadWeightScalar(uint32_t ic, uint32_t oc, uint32_t t)
    {
        AscendC::LocalTensor<float> wLocal = scalarQueue_.AllocTensor<float>();
        const uint32_t wOff = WIndex(ic, oc, t);
        AscendC::DataCopy(wLocal, wGm_[wOff], 1);
        // Extract scalar from local tensor. API provides GetValue for scalar read.
        float w = wLocal.GetValue(0);
        scalarQueue_.FreeTensor(wLocal);
        return w;
    }

    __aicore__ inline void ComputeTile(uint32_t base, uint32_t len)
    {
        // For each batch and out channel, compute yLocal[base:base+len)
        for (uint32_t n = 0; n < n_; ++n) {
            for (uint32_t oc = 0; oc < cout_; ++oc) {
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Duplicate(yLocal, 0.0f, len);

                for (uint32_t ic = 0; ic < cin_; ++ic) {
                    // Iterate over kernel taps
                    for (uint32_t t = 0; t < k_; ++t) {
                        // For stride == 1: in_pos = out_pos + padding - dilation * t
                        // Only valid when 0 <= in_pos < lin_
                        // Compute contiguous overlap [pBegin, pEnd) in tile space
                        int32_t s = static_cast<int32_t>(base) + static_cast<int32_t>(padding_) - static_cast<int32_t>(dilation_) * static_cast<int32_t>(t);
                        int32_t pBegin = 0;
                        int32_t pEnd = static_cast<int32_t>(len);

                        if (s < 0) {
                            // need to start later in tile to bring in_pos to 0
                            pBegin = -s;
                        }
                        // in_pos end at s + (pEnd-1), must be < lin_
                        if (s + (pEnd - 1) >= static_cast<int32_t>(lin_)) {
                            pEnd = static_cast<int32_t>(lin_) - s;
                        }

                        if (pBegin < pEnd) {
                            const uint32_t segLen = static_cast<uint32_t>(pEnd - pBegin);
                            const uint32_t inStart = static_cast<uint32_t>(s + pBegin);

                            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[XIndex(n, ic, inStart)], segLen);

                            const float w = LoadWeightScalar(ic, oc, t);

                            AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                            AscendC::Muls(tmpLocal, xLocal, w, segLen);

                            AscendC::LocalTensor<float> ySeg = yLocal[static_cast<uint32_t>(pBegin)];
                            AscendC::Add(ySeg, ySeg, tmpLocal, segLen);

                            tmpQueue_.FreeTensor(tmpLocal);
                            inQueueX_.FreeTensor(xLocal);
                        }
                    } // t
                } // ic

                // Write out the completed tile for y[n, oc, base : base+len)
                AscendC::DataCopy(yGm_[YIndex(n, oc, base)], yLocal, len);
                outQueueY_.FreeTensor(yLocal);
            } // oc
        } // n
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t lin_{0}, lout_{0}, k_{0};
    uint32_t stride_{1}, padding_{0}, dilation_{1};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1dDilatedCustom op;
    op.Init(x,
            weight,
            y,
            tiling_data.n,
            tiling_data.cin,
            tiling_data.cout,
            tiling_data.lin,
            tiling_data.lout,
            tiling_data.k,
            tiling_data.stride,
            tiling_data.padding,
            tiling_data.dilation,
            tiling_data.tileLength);
    op.Process();
}

#include "pytorch_npu_helper.hpp"

static inline at::Tensor make_output_like(const at::Tensor& x, const at::Tensor& weight,
                                          int64_t stride, int64_t padding, int64_t dilation)
{
    // x: [N, Cin, Lin], weight: [Cin, Cout, K]
    const auto N = x.size(0);
    const auto Cout = weight.size(1);
    const auto Lin = x.size(2);
    const auto K = weight.size(2);
    const int64_t Lout = (Lin - 1) * stride - 2 * padding + dilation * (K - 1) + 1;
    auto opts = x.options();
    return at::empty({N, Cout, Lout}, opts);
}

at::Tensor conv_transposed_1d_dilated_impl_npu(const at::Tensor& x,
                                               const at::Tensor& weight,
                                               int64_t stride,
                                               int64_t padding,
                                               int64_t dilation)

    at::Tensor y = make_output_like(x, weight, stride, padding, dilation);
    EXEC_NPU_CMD(aclnnConvTransposed1dDilatedCustom, x, weight, stride, padding, dilation, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_transposed_1d_dilated_custom", &conv_transposed_1d_dilated_impl_npu);
