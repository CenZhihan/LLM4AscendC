host_operator_src="""

#include "matmul_divide_gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative UB share
static constexpr uint32_t ALIGN_N = 1024;
static constexpr uint32_t ALIGN_K = 256;
static constexpr uint32_t MIN_TILE_N = 512;
static constexpr uint32_t MIN_TILE_K = 256;
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t MinU32(uint32_t a, uint32_t b) { return a < b ? a : b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulDivideGeluCustomTilingData tiling;

    const auto *x_shape  = context->GetInputShape(0); // [M, K]
    const auto *wt_shape = context->GetInputShape(1); // [K, N] (contiguous across N)
    if (x_shape->GetOriginShape().GetDimNum() < 2 || wt_shape->GetOriginShape().GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t M64 = x_shape->GetOriginShape().GetDim(0);
    const uint64_t K64 = x_shape->GetOriginShape().GetDim(1);
    const uint64_t N64 = wt_shape->GetOriginShape().GetDim(1);

    const uint32_t M = static_cast<uint32_t>(M64);
    const uint32_t K = static_cast<uint32_t>(K64);
    const uint32_t N = static_cast<uint32_t>(N64);

    // UB usage in floats: yTile(Tn) + temp(Tn) + wtRow(Tn) + xSeg(Tk) = 3*Tn + Tk
    const uint32_t ub_floats = UB_BUDGET_BYTES / 4;

    uint32_t tileN = MinU32(N, 8192u);
    tileN = AlignDown(tileN, ALIGN_N);
    if (tileN == 0) tileN = MinU32(N, ALIGN_N);
    if (tileN < MIN_TILE_N) tileN = MinU32(N, MIN_TILE_N);

    // Ensure space for x segment
    uint32_t remaining = (3u * tileN <= ub_floats) ? (ub_floats - 3u * tileN) : 0u;
    while (remaining < MIN_TILE_K && tileN > MIN_TILE_N) {
        tileN = AlignDown(tileN - ALIGN_N, ALIGN_N);
        remaining = (3u * tileN <= ub_floats) ? (ub_floats - 3u * tileN) : 0u;
    }
    if (remaining < MIN_TILE_K) {
        // fallback minimal safe partitions
        tileN = AlignDown(MinU32(N, MIN_TILE_N), ALIGN_N);
        remaining = (3u * tileN <= ub_floats) ? (ub_floats - 3u * tileN) : 0u;
    }

    uint32_t tileK = remaining;
    tileK = MinU32(tileK, 4096u);
    tileK = AlignDown(tileK, ALIGN_K);
    if (tileK == 0) tileK = MinU32(K, MIN_TILE_K);
    if (tileK > K) tileK = AlignDown(K, ALIGN_K > 0 ? ALIGN_K : 1);

    uint32_t blockDim = M;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulDivideGeluCustom {
public:
    __aicore__ inline KernelMatmulDivideGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wt, GM_ADDR b, GM_ADDR divisor, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tileN_ = tileN; tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * K_);
        wtGm_.SetGlobalBuffer((__gm__ float*)wt, (uint64_t)K_ * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        divGm_.SetGlobalBuffer((__gm__ float*)divisor, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * N_);

        pipe_.InitBuffer(yTileQ_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tempQ_,  BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(wRowQ_,  BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(xSegQ_,  BUFFER_NUM, tileK_ * sizeof(float));

        // load divisor scalar to register
        AscendC::LocalTensor<float> dLocal = xSegQ_.AllocTensor<float>();
        AscendC::DataCopy(dLocal, divGm_[0], 1);
        // assume immediate availability; no need to enque/deque
        invDiv_ = 1.0f / dLocal.GetValue(0);
        xSegQ_.FreeTensor(dLocal);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            const uint64_t xRowBase = (uint64_t)m * K_;
            const uint64_t yRowBase = (uint64_t)m * N_;

            for (uint32_t n0 = 0; n0 < N_; n0 += tileN_) {
                uint32_t nLen = N_ - n0;
                if (nLen > tileN_) nLen = tileN_;

                AscendC::LocalTensor<float> yTile = yTileQ_.AllocTensor<float>();
                AscendC::DataCopy(yTile, bGm_[n0], nLen);

                for (uint32_t k0 = 0; k0 < K_; k0 += tileK_) {
                    uint32_t kLen = K_ - k0;
                    if (kLen > tileK_) kLen = tileK_;

                    AscendC::LocalTensor<float> xSeg = xSegQ_.AllocTensor<float>();
                    AscendC::DataCopy(xSeg, xGm_[xRowBase + k0], kLen);

                    for (uint32_t t = 0; t < kLen; ++t) {
                        const float scale = xSeg.GetValue(t);
                        AscendC::LocalTensor<float> wRow = wRowQ_.AllocTensor<float>();
                        AscendC::DataCopy(wRow, wtGm_[((uint64_t)(k0 + t) * N_) + n0], nLen);

                        AscendC::LocalTensor<float> tmp = tempQ_.AllocTensor<float>();
                        AscendC::Muls(tmp, wRow, scale, nLen);
                        AscendC::Add(yTile, yTile, tmp, nLen);

                        tempQ_.FreeTensor(tmp);
                        wRowQ_.FreeTensor(wRow);
                    }
                    xSegQ_.FreeTensor(xSeg);
                }

                // yTile = yTile / divisor
                AscendC::Muls(yTile, yTile, invDiv_, nLen);

                // GELU approximation: 0.5 * x * (1 + tanh(sqrt(2/pi)*(x + 0.044715*x^3)))
                AscendC::LocalTensor<float> tmp1 = tempQ_.AllocTensor<float>(); // reuse tempQ_
                AscendC::Mul(tmp1, yTile, yTile, nLen);             // y^2
                AscendC::Mul(tmp1, tmp1, yTile, nLen);              // y^3
                AscendC::Muls(tmp1, tmp1, 0.044715f, nLen);         // 0.044715*y^3
                AscendC::Add(tmp1, tmp1, yTile, nLen);              // y + 0.044715*y^3
                AscendC::Muls(tmp1, tmp1, 0.7978845608028654f, nLen); // sqrt(2/pi)*...
                AscendC::Tanh(tmp1, tmp1, nLen);                    // tanh(v)
                AscendC::Adds(tmp1, tmp1, 1.0f, nLen);              // 1 + tanh(...)
                AscendC::Mul(tmp1, tmp1, yTile, nLen);              // y * (1 + tanh(...))
                AscendC::Muls(yTile, tmp1, 0.5f, nLen);             // 0.5 * ...

                tempQ_.FreeTensor(tmp1);

                AscendC::DataCopy(yGm_[yRowBase + n0], yTile, nLen);
                yTileQ_.FreeTensor(yTile);
            }
        }
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tempQ_;
    AscendC::TQue<AscendC::TPosition::VECIN,   BUFFER_NUM> wRowQ_;
    AscendC::TQue<AscendC::TPosition::VECIN,   BUFFER_NUM> xSegQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wtGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> divGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileN_{0}, tileK_{0};
    float invDiv_{1.0f};
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w_t, GM_ADDR b, GM_ADDR divisor, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulDivideGeluCustom op;
    op.Init(x, w_t, b, divisor, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileN, tiling_data.tileK);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'matmul_divide_gelu_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, input_size, output_size, divisor):
        super(ModelNew, self).__init__()
        self.input_size = input_size
        self.output_size = output_size

    def forward(self, x, w, b, divisor):
        # Pass weight as transposed [K, N] to match kernel's contiguous access on N
        wt = w.t().contiguous()
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, wt, b, divisor)
