#include "gemm_scaling_hard_tanh_gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model (per core):
// - xChunk: 1 * tileK
// - wChunk: 1 * tileK
// - tmpMul: 1 * tileK
// - tiny scalars and scratch: negligible
// Total floats ~ 3 * tileK => bytes ~ 12 * tileK
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_TILEK = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmScalingHardTanhGeluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetStorageShape().GetDimNum() < 2 || w_shape->GetStorageShape().GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(0));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scaling = attrs->GetAttrPointer<float>(0);
    const float *htmin   = attrs->GetAttrPointer<float>(1);
    const float *htmax   = attrs->GetAttrPointer<float>(2);

    // tileK from UB budget: bytes ~ 12 * tileK
    uint32_t maxTileK = UB_BUDGET_BYTES / 12;
    if (maxTileK > HARD_CAP_TILEK) maxTileK = HARD_CAP_TILEK;
    maxTileK = AlignDown(maxTileK, ALIGN_ELEMS);
    if (maxTileK == 0) maxTileK = ALIGN_ELEMS;

    uint32_t tileK = (K >= maxTileK) ? maxTileK : AlignDown(K, ALIGN_ELEMS);
    if (tileK == 0) tileK = K; // small K fallback

    // Conservative blockDim based on M
    uint32_t blockDim = M;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim == 0u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);
    tiling.set_scaling(*scaling);
    tiling.set_hardtanhMin(*htmin);
    tiling.set_hardtanhMax(*htmax);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmScalingHardTanhGeluCustom {
public:
    __aicore__ inline KernelGemmScalingHardTanhGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileK,
                                float scaling, float htmin, float htmax)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileK_ = tileK;
        scaling_ = scaling;
        htmin_ = htmin;
        htmax_ = htmax;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * (uint64_t)K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)N_ * (uint64_t)K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * (uint64_t)N_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(tmpMulQueue_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(sumQueue_, 2, sizeof(float));   // 1-element buffers
        pipe_.InitBuffer(outQueueY_, 2, sizeof(float));  // 1-element buffers
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each core iterates rows with stride of blockNum
        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            ProcessRow(m);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t m)
    {
        const uint64_t xRowBase = (uint64_t)m * (uint64_t)K_;
        const uint64_t yRowBase = (uint64_t)m * (uint64_t)N_;

        for (uint32_t n = 0; n < N_; ++n) {
            // acc = 0
            AscendC::LocalTensor<float> acc = sumQueue_.AllocTensor<float>();
            AscendC::Dup(acc, 0.0f, 1);

            // Dot product over K in chunks
            for (uint32_t k0 = 0; k0 < K_; k0 += tileK_) {
                uint32_t kc = K_ - k0;
                if (kc > tileK_) kc = tileK_;

                // Load x chunk
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[xRowBase + k0], kc);
                inQueueX_.EnQue(xLocal);

                // Load w chunk for this column
                AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
                const uint64_t wBase = (uint64_t)n * (uint64_t)K_ + (uint64_t)k0;
                AscendC::DataCopy(wLocal, wGm_[wBase], kc);
                inQueueW_.EnQue(wLocal);

                // Compute partial dot
                AscendC::LocalTensor<float> xL = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> wL = inQueueW_.DeQue<float>();
                AscendC::LocalTensor<float> tmpMul = tmpMulQueue_.AllocTensor<float>();

                AscendC::Mul(tmpMul, xL, wL, kc);

                AscendC::LocalTensor<float> part = sumQueue_.AllocTensor<float>();
                AscendC::ReduceSum(part, tmpMul, kc);

                // acc += part
                AscendC::Add(acc, acc, part, 1);

                // Free temps
                sumQueue_.FreeTensor(part);
                tmpMulQueue_.FreeTensor(tmpMul);
                inQueueW_.FreeTensor(wL);
                inQueueX_.FreeTensor(xL);
            }

            // Add bias b[n]
            AscendC::LocalTensor<float> bLocal = sumQueue_.AllocTensor<float>();
            AscendC::DataCopy(bLocal, bGm_[n], 1);
            AscendC::Add(acc, acc, bLocal, 1);
            sumQueue_.FreeTensor(bLocal);

            // Scale
            AscendC::Muls(acc, acc, scaling_, 1);

            // HardTanh: clamp to [htmin_, htmax_]
            AscendC::Maxs(acc, acc, htmin_, 1);
            AscendC::Mins(acc, acc, htmax_, 1);

            // GELU approximation: 0.5 * x * (1 + tanh(0.79788456 * (x + 0.044715 * x^3)))
            // preserve x in acc
            AscendC::LocalTensor<float> x2 = sumQueue_.AllocTensor<float>();
            AscendC::Mul(x2, acc, acc, 1); // x^2

            AscendC::LocalTensor<float> x3 = sumQueue_.AllocTensor<float>();
            AscendC::Mul(x3, x2, acc, 1); // x^3

            AscendC::LocalTensor<float> t = sumQueue_.AllocTensor<float>();
            AscendC::Muls(t, x3, 0.044715f, 1); // 0.044715*x^3

            AscendC::LocalTensor<float> u = sumQueue_.AllocTensor<float>();
            AscendC::Add(u, t, acc, 1); // x + 0.044715*x^3

            AscendC::LocalTensor<float> v = sumQueue_.AllocTensor<float>();
            AscendC::Muls(v, u, 0.79788456f, 1); // sqrt(2/pi)*(...)

            AscendC::LocalTensor<float> w = sumQueue_.AllocTensor<float>();
            AscendC::Tanh(w, v, 1);

            AscendC::LocalTensor<float> onePlus = sumQueue_.AllocTensor<float>();
            AscendC::Adds(onePlus, w, 1.0f, 1); // 1 + tanh(...)

            AscendC::LocalTensor<float> halfx = sumQueue_.AllocTensor<float>();
            AscendC::Muls(halfx, acc, 0.5f, 1);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Mul(yLocal, halfx, onePlus, 1);

            // Store y[m, n]
            AscendC::DataCopy(yGm_[yRowBase + n], yLocal, 1);

            // Free locals
            outQueueY_.FreeTensor(yLocal);
            sumQueue_.FreeTensor(halfx);
            sumQueue_.FreeTensor(onePlus);
            sumQueue_.FreeTensor(w);
            sumQueue_.FreeTensor(v);
            sumQueue_.FreeTensor(u);
            sumQueue_.FreeTensor(t);
            sumQueue_.FreeTensor(x3);
            sumQueue_.FreeTensor(x2);
            sumQueue_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMulQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> sumQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileK_{0};

    float scaling_{1.0f};
    float htmin_{-1.0f};
    float htmax_{1.0f};
};

extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScalingHardTanhGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileK,
            tiling_data.scaling, tiling_data.hardtanhMin, tiling_data.hardtanhMax);
    op.Process();
}
