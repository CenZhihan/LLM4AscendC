host_operator_src="""
#include "cumprod_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // per-iter cap

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumprodCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &ori = in_shape->GetOriginShape();
    const uint32_t rank = static_cast<uint32_t>(ori.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);

    int32_t axis = (dimAttr != nullptr) ? *dimAttr : -1;
    if (axis < 0) axis += static_cast<int32_t>(rank);
    if (axis < 0) axis = static_cast<int32_t>(rank) - 1;
    if (axis >= static_cast<int32_t>(rank)) axis = static_cast<int32_t>(rank) - 1;

    uint64_t totalSize = static_cast<uint64_t>(ori.GetShapeSize());
    uint32_t rowLen = static_cast<uint32_t>(ori.GetDim(axis));
    uint32_t outer = (rowLen == 0) ? 0u : static_cast<uint32_t>(totalSize / rowLen);

    // tileLen from UB budget: need in + out = 2 * tileLen * 4 bytes
    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = (rowLen < maxTile) ? rowLen : maxTile;
    // ensure at least 1
    if (tileLen == 0 && rowLen > 0) tileLen = (rowLen < ALIGN_ELEMS) ? rowLen : ALIGN_ELEMS;

    // blockDim based on number of rows to process
    uint32_t blockDim = outer;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_outerCount(outer);
    tiling.set_rowLength(rowLen);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(axis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumprodCustom : public OpDef {
public:
    explicit CumprodCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(-1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumprodCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;

class KernelCumprodCustom {
public:
    __aicore__ inline KernelCumprodCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerCount, uint32_t rowLength,
                                uint32_t tileLength)
    {
        outerCount_ = outerCount;
        rowLen_ = rowLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(outerCount_) * rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(outerCount_) * rowLen_);

        pipe_.InitBuffer(inQ_, IN_BUF_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQ_, OUT_BUF_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outerCount_ == 0 || rowLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes multiple rows in a grid-stride loop
        for (uint32_t row = blockIdx; row < outerCount_; row += blockNum) {
            float acc = 1.0f;

            uint32_t processed = 0;
            while (processed < rowLen_) {
                const uint32_t curLen = AscendC::min(tileLen_, rowLen_ - processed);
                const uint32_t base = row * rowLen_ + processed;

                CopyIn(base, curLen);
                ComputeSegment(curLen, acc);
                CopyOut(base, curLen);

                processed += curLen;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQ_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQ_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeSegment(uint32_t len, float &acc)
    {
        AscendC::LocalTensor<float> xLocal = inQ_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQ_.AllocTensor<float>();

        // Sequential prefix product within the tile with carry 'acc'
        for (uint32_t i = 0; i < len; ++i) {
            float v = xLocal.GetValue(i);
            acc = acc * v;
            yLocal.SetValue(i, acc);
        }

        outQ_.EnQue(yLocal);
        inQ_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQ_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQ_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outerCount_{0};
    uint32_t rowLen_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void cumprod_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumprodCustom op;
    op.Init(x, y, tiling_data.outerCount, tiling_data.rowLength, tiling_data.tileLength);
    op.Process();
}
"""
