kernel_src="""

#include "kernel_operator.h"

constexpr int32_t SCALAR_BUF_DEPTH = 8;

class KernelConvTransposed3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t outD, uint32_t outH, uint32_t outW)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        inD_ = inD; inH_ = inH; inW_ = inW;
        kD_ = kD; kH_ = kH; kW_ = kW;
        outD_ = outD; outH_ = outH; outW_ = outW;

        inHW_ = inH_ * inW_;
        inDHW_ = inD_ * inHW_;
        outHW_ = outH_ * outW_;
        outDHW_ = outD_ * outHW_;
        kHW_ = kH_ * kW_;
        kDHW_ = kD_ * kHW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * cin_ * inDHW_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)cin_ * cout_ * kDHW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * cout_ * outDHW_);

        pipe_.InitBuffer(calcQueue_, SCALAR_BUF_DEPTH, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        const uint64_t totalPairs = static_cast<uint64_t>(n_) * static_cast<uint64_t>(cout_);
        if (blockIdx >= totalPairs) return;

        // map block -> (nIdx, oc)
        const uint32_t nIdx = blockIdx / cout_;
        const uint32_t oc   = blockIdx % cout_;

        // iterate all spatial output positions for this (nIdx, oc)
        const uint64_t S = static_cast<uint64_t>(outDHW_);
        for (uint64_t p = 0; p < S; ++p) {
            // decode to (od, oh, ow)
            uint32_t od = static_cast<uint32_t>(p / outHW_);
            uint32_t rem = static_cast<uint32_t>(p % outHW_);
            uint32_t oh = rem / outW_;
            uint32_t ow = rem % outW_;

            // yLocal = 0
            AscendC::LocalTensor<float> yLocal = calcQueue_.AllocTensor<float>();
            AscendC::Duplicate(yLocal, 0.0f, 1);

            // accumulate contributions: id = od + kd - (kD - 1), similarly for ih, iw
            for (uint32_t ci = 0; ci < cin_; ++ci) {
                for (uint32_t kd = 0; kd < kD_; ++kd) {
                    int32_t id = static_cast<int32_t>(od) + static_cast<int32_t>(kd) - static_cast<int32_t>(kD_ - 1);
                    if (id < 0 || id >= static_cast<int32_t>(inD_)) continue;
                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        int32_t ih = static_cast<int32_t>(oh) + static_cast<int32_t>(kh) - static_cast<int32_t>(kH_ - 1);
                        if (ih < 0 || ih >= static_cast<int32_t>(inH_)) continue;
                        for (uint32_t kw = 0; kw < kW_; ++kw) {
                            int32_t iw = static_cast<int32_t>(ow) + static_cast<int32_t>(kw) - static_cast<int32_t>(kW_ - 1);
                            if (iw < 0 || iw >= static_cast<int32_t>(inW_)) continue;

                            // GM indices
                            uint64_t xIndex =
                                static_cast<uint64_t>(nIdx) * (static_cast<uint64_t>(cin_) * inDHW_) +
                                static_cast<uint64_t>(ci)   * inDHW_ +
                                static_cast<uint64_t>(static_cast<uint32_t>(id)) * inHW_ +
                                static_cast<uint64_t>(static_cast<uint32_t>(ih)) * inW_ +
                                static_cast<uint64_t>(static_cast<uint32_t>(iw));

                            uint64_t wIndex =
                                static_cast<uint64_t>(ci)   * (static_cast<uint64_t>(cout_) * kDHW_) +
                                static_cast<uint64_t>(oc)   * kDHW_ +
                                static_cast<uint64_t>(kd)   * kHW_ +
                                static_cast<uint64_t>(kh)   * kW_ +
                                static_cast<uint64_t>(kw);

                            // load scalars from GM
                            AscendC::LocalTensor<float> xScalar = calcQueue_.AllocTensor<float>();
                            AscendC::LocalTensor<float> wScalar = calcQueue_.AllocTensor<float>();
                            AscendC::DataCopy(xScalar, xGm_[xIndex], 1);
                            AscendC::DataCopy(wScalar, wGm_[wIndex], 1);

                            // yLocal += xScalar * wScalar
                            AscendC::LocalTensor<float> prod = calcQueue_.AllocTensor<float>();
                            AscendC::Mul(prod, xScalar, wScalar, 1);
                            AscendC::Add(yLocal, yLocal, prod, 1);

                            calcQueue_.FreeTensor(prod);
                            calcQueue_.FreeTensor(wScalar);
                            calcQueue_.FreeTensor(xScalar);
                        }
                    }
                }
            }

            // store yLocal to GM
            uint64_t yIndex =
                static_cast<uint64_t>(nIdx) * (static_cast<uint64_t>(cout_) * outDHW_) +
                static_cast<uint64_t>(oc)   * outDHW_ +
                static_cast<uint64_t>(od)   * outHW_ +
                static_cast<uint64_t>(oh)   * outW_ +
                static_cast<uint64_t>(ow);

            AscendC::DataCopy(yGm_[yIndex], yLocal, 1);
            calcQueue_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, SCALAR_BUF_DEPTH> calcQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t inD_{0}, inH_{0}, inW_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};
    uint32_t outD_{0}, outH_{0}, outW_{0};

    uint32_t inHW_{0}, inDHW_{0};
    uint32_t outHW_{0}, outDHW_{0};
    uint32_t kHW_{0}, kDHW_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.inD, tiling_data.inH, tiling_data.inW,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW);
    op.Process();
}
