#include "kernel_operator.h"

class KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t Co,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Do, uint32_t Ho, uint32_t Wo,
                                int32_t strideD, int32_t strideH, int32_t strideW,
                                int32_t padD, int32_t padH, int32_t padW,
                                int32_t dilD, int32_t dilH, int32_t dilW)
    {
        x_ = reinterpret_cast<__gm__ float*>(x);
        w_ = reinterpret_cast<__gm__ float*>(w);
        y_ = reinterpret_cast<__gm__ float*>(y);

        N_ = N; Ci_ = Ci; Co_ = Co;
        D_ = D; H_ = H; W_ = W;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        Do_ = Do; Ho_ = Ho; Wo_ = Wo;

        sD_ = strideD; sH_ = strideH; sW_ = strideW;
        pD_ = padD;    pH_ = padH;    pW_ = padW;
        dD_ = dilD;    dH_ = dilH;    dW_ = dilW;
    }

    __aicore__ inline void Process()
    {
        const uint64_t totalOut = static_cast<uint64_t>(N_) * Co_ * Do_ * Ho_ * Wo_;
        if (totalOut == 0ULL) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t idx = blockIdx; idx < totalOut; idx += blockNum) {
            ComputeOne(idx);
        }
    }

private:
    __aicore__ inline void LinearToCoords(uint64_t idx,
                                          uint32_t &n, uint32_t &co,
                                          uint32_t &od, uint32_t &oh, uint32_t &ow) const
    {
        uint64_t t = idx;
        ow = static_cast<uint32_t>(t % Wo_); t /= Wo_;
        oh = static_cast<uint32_t>(t % Ho_); t /= Ho_;
        od = static_cast<uint32_t>(t % Do_); t /= Do_;
        co = static_cast<uint32_t>(t % Co_); t /= Co_;
        n  = static_cast<uint32_t>(t);
    }

    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t ci, uint32_t z, uint32_t y, uint32_t x) const
    {
        return (((static_cast<uint64_t>(n) * Ci_ + ci) * D_ + z) * H_ + y) * W_ + x;
    }

    __aicore__ inline uint64_t OffsetW(uint32_t co, uint32_t ci, uint32_t kz, uint32_t ky, uint32_t kx) const
    {
        return ((((static_cast<uint64_t>(co) * Ci_ + ci) * Kd_ + kz) * Kh_ + ky) * Kw_ + kx);
    }

    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t co, uint32_t z, uint32_t y, uint32_t x) const
    {
        return (((static_cast<uint64_t>(n) * Co_ + co) * Do_ + z) * Ho_ + y) * Wo_ + x;
    }

    __aicore__ inline void ComputeOne(uint64_t outLinear)
    {
        uint32_t n, co, od, oh, ow;
        LinearToCoords(outLinear, n, co, od, oh, ow);

        // Base input coordinates before kernel extent
        const int32_t baseZ = static_cast<int32_t>(od) * sD_ - pD_;
        const int32_t baseY = static_cast<int32_t>(oh) * sH_ - pH_;
        const int32_t baseX = static_cast<int32_t>(ow) * sW_ - pW_;

        float acc = 0.0f;

        for (uint32_t ci = 0; ci < Ci_; ++ci) {
            for (uint32_t kz = 0; kz < Kd_; ++kz) {
                const int32_t iz = baseZ + static_cast<int32_t>(kz) * dD_;
                if (iz < 0 || iz >= static_cast<int32_t>(D_)) continue;

                for (uint32_t ky = 0; ky < Kh_; ++ky) {
                    const int32_t iy = baseY + static_cast<int32_t>(ky) * dH_;
                    if (iy < 0 || iy >= static_cast<int32_t>(H_)) continue;

                    for (uint32_t kx = 0; kx < Kw_; ++kx) {
                        const int32_t ix = baseX + static_cast<int32_t>(kx) * dW_;
                        if (ix < 0 || ix >= static_cast<int32_t>(W_)) continue;

                        const uint64_t xOff = OffsetX(n, ci, static_cast<uint32_t>(iz),
                                                      static_cast<uint32_t>(iy),
                                                      static_cast<uint32_t>(ix));
                        const uint64_t wOff = OffsetW(co, ci, kz, ky, kx);

                        // Direct GM reads/writes (naive for clarity)
                        const float xv = x_[xOff];
                        const float wv = w_[wOff];
                        acc += xv * wv;
                    }
                }
            }
        }

        const uint64_t yOff = OffsetY(n, co, od, oh, ow);
        y_[yOff] = acc;
    }

private:
    __gm__ float* x_ {nullptr};
    __gm__ float* w_ {nullptr};
    __gm__ float* y_ {nullptr};

    uint32_t N_{0}, Ci_{0}, Co_{0};
    uint32_t D_{0}, H_{0}, W_{0};
    uint32_t Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Do_{0}, Ho_{0}, Wo_{0};

    int32_t sD_{1}, sH_{1}, sW_{1};
    int32_t pD_{0}, pH_{0}, pW_{0};
    int32_t dD_{1}, dH_{1}, dW_{1};
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Ci, tiling_data.Co,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.Kd, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Do, tiling_data.Ho, tiling_data.Wo,
            tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
            tiling_data.padD, tiling_data.padH, tiling_data.padW,
            tiling_data.dilD, tiling_data.dilH, tiling_data.dilW);
    op.Process();
}
