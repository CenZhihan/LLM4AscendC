host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();
    const uint32_t rank = static_cast<uint32_t>(origin.GetDimNum());
    uint64_t totalLength = origin.GetShapeSize();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);

    if (rank == 0) {
        return ge::GRAPH_PARAM_INVALID;
    }

    int32_t d = *dimAttr;
    if (d < 0) d += static_cast<int32_t>(rank);
    if (d < 0 || d >= static_cast<int32_t>(rank)) {
        return ge::GRAPH_PARAM_INVALID;
    }

    uint64_t P = 1, Q = 1;
    uint64_t R = static_cast<uint64_t>(origin.GetDim(d));
    for (int32_t i = 0; i < d; ++i) {
        P *= static_cast<uint64_t>(origin.GetDim(i));
    }
    for (uint32_t i = static_cast<uint32_t>(d + 1); i < rank; ++i) {
        Q *= static_cast<uint64_t>(origin.GetDim(i));
    }

    // UB model:
    //   acc: 1 * tileQ
    //   xbuf: 1 * tileQ
    // Total floats = 2 * tileQ => bytes = 8 * tileQ
    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t qLen = static_cast<uint32_t>(Q);
    uint32_t tileQ = maxTile;
    if (qLen < tileQ) {
        const uint32_t aligned = AlignDown(qLen, ALIGN_ELEMS);
        tileQ = (aligned > 0) ? aligned : qLen;
    }
    if (tileQ == 0) tileQ = 1; // safety

    const uint32_t numQChunks = CeilDiv(Q, tileQ);
    const uint64_t totalWorkItems64 = P * static_cast<uint64_t>(numQChunks);
    uint32_t totalWorkItems = static_cast<uint32_t>(totalWorkItems64 > 0xFFFFFFFFu ? 0xFFFFFFFFu : totalWorkItems64);

    uint32_t blockDim = totalWorkItems;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_pLen(static_cast<uint32_t>(P));
    tiling.set_rLen(static_cast<uint32_t>(R));
    tiling.set_qLen(static_cast<uint32_t>(Q));
    tiling.set_tileQ(tileQ);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(d);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &xs = x_shape->GetOriginShape();
    const uint32_t rank = static_cast<uint32_t>(xs.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);

    int32_t d = *dimAttr;
    if (d < 0) d += static_cast<int32_t>(rank);
    if (d < 0 || d >= static_cast<int32_t>(rank)) {
        return GRAPH_PARAM_INVALID;
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(rank - 1);
    uint32_t out_i = 0;
    for (uint32_t i = 0; i < rank; ++i) {
        if (static_cast<int32_t>(i) == d) continue;
        y_shape->SetDim(out_i++, xs.GetDim(i));
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t ACC_BUF_NUM = 1;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t pLen, uint32_t rLen, uint32_t qLen,
                                uint32_t tileQ)
    {
        totalLen_ = totalLength;
        P_ = pLen;
        R_ = rLen;
        Q_ = qLen;
        tileQ_ = tileQ;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, P_ * Q_);

        pipe_.InitBuffer(inQueue_, IN_BUF_NUM, tileQ_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, ACC_BUF_NUM, tileQ_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (P_ == 0 || Q_ == 0 || R_ == 0 || tileQ_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numQChunks = (Q_ + tileQ_ - 1u) / tileQ_;
        const uint64_t totalWorkItems = static_cast<uint64_t>(P_) * static_cast<uint64_t>(numQChunks);

        for (uint64_t work = blockIdx; work < totalWorkItems; work += blockNum) {
            const uint32_t p = static_cast<uint32_t>(work / numQChunks);
            const uint32_t qChunk = static_cast<uint32_t>(work % numQChunks);
            const uint32_t qBase = qChunk * tileQ_;
            uint32_t curLen = Q_ - qBase;
            if (curLen > tileQ_) curLen = tileQ_;

            // acc = 0
            AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();
            AscendC::Dup(acc, 0.0f, curLen);

            // Accumulate across R
            for (uint32_t r = 0; r < R_; ++r) {
                AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();

                uint64_t xOffset = (static_cast<uint64_t>(p) * static_cast<uint64_t>(R_) + static_cast<uint64_t>(r)) * static_cast<uint64_t>(Q_) + static_cast<uint64_t>(qBase);
                AscendC::DataCopy(xLocal, xGm_[static_cast<uint32_t>(xOffset)], curLen);

                AscendC::Add(acc, acc, xLocal, curLen);

                inQueue_.FreeTensor(xLocal);
            }

            // Mean: acc = acc / R
            const float invR = 1.0f / static_cast<float>(R_);
            AscendC::Muls(acc, acc, invR, curLen);

            // Store to y: shape [P, Q]
            uint64_t yOffset = static_cast<uint64_t>(p) * static_cast<uint64_t>(Q_) + static_cast<uint64_t>(qBase);
            AscendC::DataCopy(yGm_[static_cast<uint32_t>(yOffset)], acc, curLen);

            accQueue_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, ACC_BUF_NUM> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t P_ {0};
    uint32_t R_ {0};
    uint32_t Q_ {0};
    uint32_t tileQ_ {0};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.pLen, tiling_data.rLen, tiling_data.qLen,
            tiling_data.tileQ);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor mean_reduction_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    at::Tensor result;
    // Output shape is input shape with 'dim' removed
    auto in_sizes = self.sizes();
    int64_t nd = in_sizes.size();
    int64_t d = dim < 0 ? dim + nd : dim;
    TORCH_CHECK(d >= 0 && d < nd, \"dim out of range\");
    std::vector<int64_t> out_sizes;
    out_sizes.reserve(nd - 1);
    for (int64_t i = 0; i < nd; ++i) {
        if (i == d) continue;
        out_sizes.push_back(in_sizes[i]);
    }
    result = at::empty(out_sizes, self.options());

    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, static_cast<int>(dim), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("mean_reduction_over_a_dimension_custom", &mean_reduction_over_a_dimension_impl_npu, "Mean reduction over a given dimension (MeanReductionOverADimensionCustom on NPU)");
}
