host_operator_src="""

#include "matmul_with_transposed_b_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

// UB model per iteration (processing a K-segment for one output element):
//   aVec: 1 * tileK
//   bVec: 1 * tileK
//   mulVec: 1 * tileK
//   partial/sum scalars: negligible
// Total floats ~= 3 * tileK => bytes = 12 * tileK
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedBCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(a_shape->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(a_shape->GetDim(1));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetDim(0));
    const uint32_t Kb = static_cast<uint32_t>(b_shape->GetDim(1));
    if (Kb != K) {
        return ge::GRAPH_FAILED;
    }

    // tileK from UB budget: bytes = 12 * tileK
    uint32_t maxTileK = UB_BUDGET_BYTES / 12;
    maxTileK = AlignDown(maxTileK, ALIGN_ELEMS);
    if (maxTileK < ALIGN_ELEMS) maxTileK = ALIGN_ELEMS;
    if (maxTileK > HARD_CAP_ELEMS) maxTileK = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileK = maxTileK;
    if (K < tileK) {
        const uint32_t aligned = AlignDown(K, ALIGN_ELEMS);
        tileK = (aligned > 0) ? aligned : K;
    }

    // blockDim based on number of output elements
    const uint64_t totalOut = static_cast<uint64_t>(M) * static_cast<uint64_t>(N);
    uint32_t blockDim = static_cast<uint32_t>(totalOut);
    if (blockDim > 32u) blockDim = 32u;
    if (blockDim == 0u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    const int64_t M = a_shape->GetDim(0);
    const int64_t K = a_shape->GetDim(1);

    c_shape->SetDim(0, M);

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulWithTransposedBCustom {
public:
    __aicore__ inline KernelMatmulWithTransposedBCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileK)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileK_ = tileK;

        const uint32_t aSize = M_ * K_;
        const uint32_t bSize = N_ * K_;
        const uint32_t cSize = M_ * N_;
        aGm_.SetGlobalBuffer((__gm__ float*)A, aSize);
        bGm_.SetGlobalBuffer((__gm__ float*)B, bSize);
        cGm_.SetGlobalBuffer((__gm__ float*)C, cSize);

        pipe_.InitBuffer(inA_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(inB_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(mulBuf_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(scalarBuf_, BUFFER_NUM, sizeof(float)); // for partial and sum
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalOut = static_cast<uint64_t>(M_) * static_cast<uint64_t>(N_);
        for (uint64_t outIdx = blockIdx; outIdx < totalOut; outIdx += blockNum) {
            const uint32_t i = static_cast<uint32_t>(outIdx / N_);
            const uint32_t j = static_cast<uint32_t>(outIdx % N_);

            // sum accumulator as 1-element local tensor
            AscendC::LocalTensor<float> sumLocal = scalarBuf_.AllocTensor<float>();
            AscendC::Duplicate(sumLocal, 0.0f, 1);

            for (uint32_t kBase = 0; kBase < K_; kBase += tileK_) {
                uint32_t len = K_ - kBase;
                if (len > tileK_) len = tileK_;

                // Load A row segment
                AscendC::LocalTensor<float> aLocal = inA_.AllocTensor<float>();
                const uint32_t aOff = i * K_ + kBase;
                AscendC::DataCopy(aLocal, aGm_[aOff], len);
                inA_.EnQue(aLocal);

                // Load B row segment (B is [N, K], taking row j)
                AscendC::LocalTensor<float> bLocal = inB_.AllocTensor<float>();
                const uint32_t bOff = j * K_ + kBase;
                AscendC::DataCopy(bLocal, bGm_[bOff], len);
                inB_.EnQue(bLocal);

                // Dequeue and compute elementwise multiply
                AscendC::LocalTensor<float> aSeg = inA_.DeQue<float>();
                AscendC::LocalTensor<float> bSeg = inB_.DeQue<float>();
                AscendC::LocalTensor<float> mLocal = mulBuf_.AllocTensor<float>();
                AscendC::Mul(mLocal, aSeg, bSeg, len);

                // Reduce sum of mLocal into partial scalar
                AscendC::LocalTensor<float> partial = scalarBuf_.AllocTensor<float>();
                AscendC::ReduceSum(partial, mLocal, len);

                // Accumulate into sumLocal
                AscendC::Add(sumLocal, sumLocal, partial, 1);

                // Free locals
                scalarBuf_.FreeTensor(partial);
                mulBuf_.FreeTensor(mLocal);
                inB_.FreeTensor(bSeg);
                inA_.FreeTensor(aSeg);
            }

            // Write result to C[i, j]
            const uint32_t cOff = i * N_ + j;
            AscendC::DataCopy(cGm_[cOff], sumLocal, 1);
            scalarBuf_.FreeTensor(sumLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> mulBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scalarBuf_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileK_{0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_b_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedBCustom op;
    op.Init(A, B, C, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileK);
    op.Process();
}
