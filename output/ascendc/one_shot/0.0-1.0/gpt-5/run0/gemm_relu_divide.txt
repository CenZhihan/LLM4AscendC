#include "gemm_relu_divide_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (a == 0) ? x : (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return (a == 0) ? x : ((x + a - 1) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmReluDivideCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    const auto &x_origin = x_shape->GetOriginShape();
    const auto &w_origin = w_shape->GetOriginShape();
    const auto &b_origin = b_shape->GetOriginShape();

    if (x_origin.GetDimNum() != 2 || w_origin.GetDimNum() != 2 || b_origin.GetDimNum() != 1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_origin.GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_origin.GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_origin.GetDim(0)); // w [N, K]
    // sanity: K must match
    if (static_cast<uint32_t>(w_origin.GetDim(1)) != K) return ge::GRAPH_FAILED;
    if (static_cast<uint32_t>(b_origin.GetDim(0)) != N) return ge::GRAPH_FAILED;

    // Very conservative UB budget model (float32):
    //   WT (tileK * tileN) + Y (tileM * tileN) + rowK (tileK) + bias (tileN) + tmpN (tileN)
    // floats ~= tileN * (tileK + tileM + 1 + 1) + tileK
    // bytes = floats * 4
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t ALIGN_K = 16;
    static constexpr uint32_t ALIGN_N = 16;
    static constexpr uint32_t MAX_TILE_N = 256;
    static constexpr uint32_t MAX_TILE_K = 256;
    static constexpr uint32_t MAX_TILE_M = 16;

    uint32_t tileN = 128;
    uint32_t tileK = 128;
    uint32_t tileM = 8;

    tileN = (tileN > N) ? ((N >= 16) ? AlignDown(N, ALIGN_N) : N) : tileN;
    if (tileN == 0) tileN = (N == 0) ? 1u : ((N >= 16) ? AlignDown(N, ALIGN_N) : N);
    if (tileN > MAX_TILE_N) tileN = MAX_TILE_N;

    tileK = (tileK > K) ? ((K >= 16) ? AlignDown(K, ALIGN_K) : K) : tileK;
    if (tileK == 0) tileK = (K == 0) ? 1u : ((K >= 16) ? AlignDown(K, ALIGN_K) : K);
    if (tileK > MAX_TILE_K) tileK = MAX_TILE_K;

    if (tileM > M) tileM = (M == 0) ? 1u : ((M >= 8) ? 8u : M);
    if (tileM > MAX_TILE_M) tileM = MAX_TILE_M;

    auto FitsUB = [&](uint32_t tm, uint32_t tn, uint32_t tk) -> bool {
        uint64_t floats = static_cast<uint64_t>(tn) * (static_cast<uint64_t>(tk) + static_cast<uint64_t>(tm) + 2u)
                          + static_cast<uint64_t>(tk);
        uint64_t bytes = floats * 4u;
        return bytes <= UB_BUDGET_BYTES;
    };

    // Reduce tile sizes to fit UB if needed
    while (!FitsUB(tileM, tileN, tileK)) {
        if (tileN > 64) {
            tileN = (tileN >= 32) ? AlignDown(tileN / 2, ALIGN_N) : (tileN / 2);
            if (tileN == 0) tileN = 16;
        } else if (tileK > 64) {
            tileK = (tileK >= 32) ? AlignDown(tileK / 2, ALIGN_K) : (tileK / 2);
            if (tileK == 0) tileK = 16;
        } else if (tileM > 4) {
            tileM = tileM / 2;
            if (tileM == 0) tileM = 1;
        } else {
            break;
        }
    }

    uint32_t numRowTiles = (tileM == 0) ? 1u : CeilDiv(M, tileM);
    uint32_t blockDim = numRowTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    GemmReluDivideCustomTilingData td;
    td.set_M(M);
    td.set_N(N);
    td.set_K(K);
    td.set_tileM(tileM);
    td.set_tileN(tileN);
    td.set_tileK(tileK);
    td.set_blockDim(blockDim);

    td.SaveToBuffer(context->GetRawTilingData()->GetData(),
                    context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(td.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelGemmReluDivideCustom {
public:
    __aicore__ inline KernelGemmReluDivideCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tM_ = tileM; tN_ = tileN; tK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);
        dGm_.SetGlobalBuffer((__gm__ float*)divisor, 1);

        // Buffers in UB
        // WT buffer: tK_ * tN_
        pipe_.InitBuffer(wtQueue_, BUFFER_NUM, static_cast<uint64_t>(tK_) * tN_ * sizeof(float));
        // Y buffer: tM_ * tN_
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, static_cast<uint64_t>(tM_) * tN_ * sizeof(float));
        // row K buffer: tK_
        pipe_.InitBuffer(rowKQueue_, BUFFER_NUM, static_cast<uint64_t>(tK_) * sizeof(float));
        // bias vector: tN_
        pipe_.InitBuffer(biasQueue_, BUFFER_NUM, static_cast<uint64_t>(tN_) * sizeof(float));
        // temp N vector: tN_
        pipe_.InitBuffer(tmpNQueue_, BUFFER_NUM, static_cast<uint64_t>(tN_) * sizeof(float));
        // scalar buffer (for divisor): 1
        pipe_.InitBuffer(scalarQueue_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        // Load divisor scalar and compute reciprocal
        AscendC::LocalTensor<float> dLocal = scalarQueue_.AllocTensor<float>();
        AscendC::DataCopy(dLocal, dGm_[0], 1);
        scalarQueue_.EnQue(dLocal);
        dLocal = scalarQueue_.DeQue<float>();
        float invDiv = 1.0f / dLocal.GetValue(0);
        scalarQueue_.FreeTensor(dLocal);

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideRows = blockNum * tM_;
        uint32_t baseRow = blockIdx * tM_;
        if (baseRow >= M_) return;

        while (true) {
            uint32_t rows = M_ - baseRow;
            if (rows > tM_) rows = tM_;
            // Iterate over N dimension in tiles
            for (uint32_t n0 = 0; n0 < N_; n0 += tN_) {
                uint32_t nLen = N_ - n0;
                if (nLen > tN_) nLen = tN_;

                AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> biasLocal = biasQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpN = tmpNQueue_.AllocTensor<float>();

                // Load bias slice
                AscendC::DataCopy(biasLocal, bGm_[n0], nLen);

                // Initialize Y with bias for each active row
                for (uint32_t i = 0; i < rows; ++i) {
                    AscendC::DataCopy(yLocal[i * tN_], biasLocal, nLen);
                }

                // K-loop in tiles
                for (uint32_t k0 = 0; k0 < K_; k0 += tK_) {
                    uint32_t kLen = K_ - k0;
                    if (kLen > tK_) kLen = tK_;

                    // Load and transpose W tile into WT buffer [kLen, nLen] row-major
                    AscendC::LocalTensor<float> wtLocal = wtQueue_.AllocTensor<float>();
                    // For each column j in N-tile, copy row of W and scatter into wtLocal
                    AscendC::LocalTensor<float> rowK = rowKQueue_.AllocTensor<float>();
                    for (uint32_t j = 0; j < nLen; ++j) {
                        // Load W[n0 + j, k0 : k0 + kLen] into rowK
                        uint64_t wOff = static_cast<uint64_t>(n0 + j) * K_ + k0;
                        AscendC::DataCopy(rowK, wGm_[wOff], kLen);
                        // Scatter into wtLocal so that wtLocal[kk * tN_ + j] = rowK[kk]
                        for (uint32_t kk = 0; kk < kLen; ++kk) {
                            wtLocal.SetValue(kk * tN_ + j, rowK.GetValue(kk));
                        }
                    }
                    rowKQueue_.FreeTensor(rowK);

                    // For each row in X tile
                    for (uint32_t i = 0; i < rows; ++i) {
                        // Load X row segment
                        AscendC::LocalTensor<float> xRow = rowKQueue_.AllocTensor<float>();
                        uint64_t xOff = static_cast<uint64_t>(baseRow + i) * K_ + k0;
                        AscendC::DataCopy(xRow, xGm_[xOff], kLen);

                        // Accumulate outer-product via scalar a * vector b
                        for (uint32_t kk = 0; kk < kLen; ++kk) {
                            float a = xRow.GetValue(kk);
                            // bVec = wtLocal[kk, :]
                            AscendC::Muls(tmpN, wtLocal[kk * tN_], a, nLen);
                            AscendC::Add(yLocal[i * tN_], yLocal[i * tN_], tmpN, nLen);
                        }
                        rowKQueue_.FreeTensor(xRow);
                    }

                    wtQueue_.FreeTensor(wtLocal);
                } // end K tiles

                // Apply ReLU and divide by divisor
                for (uint32_t i = 0; i < rows; ++i) {
                    AscendC::Maxs(yLocal[i * tN_], yLocal[i * tN_], 0.0f, nLen);
                    AscendC::Muls(yLocal[i * tN_], yLocal[i * tN_], invDiv, nLen);
                }

                // Store Y rows back to GM
                for (uint32_t i = 0; i < rows; ++i) {
                    uint64_t yOff = static_cast<uint64_t>(baseRow + i) * N_ + n0;
                    AscendC::DataCopy(yGm_[yOff], yLocal[i * tN_], nLen);
                }

                tmpNQueue_.FreeTensor(tmpN);
                biasQueue_.FreeTensor(biasLocal);
                yQueue_.FreeTensor(yLocal);
            } // end N tiles

            // advance to next set of rows for this block (grid-stride)
            uint32_t nextBase = baseRow + strideRows;
            if (nextBase >= M_) break;
            baseRow = nextBase;
        }
    }

private:
    AscendC::TPipe pipe_;
    // Queues
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> wtQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> rowKQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> biasQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpNQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scalarQueue_;

    // GM tensors
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> dGm_;

    // Shapes
    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tM_{0}, tN_{0}, tK_{0};
};

extern "C" __global__ __aicore__ void gemm_relu_divide_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmReluDivideCustom op;
    op.Init(x, w, b, divisor, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
