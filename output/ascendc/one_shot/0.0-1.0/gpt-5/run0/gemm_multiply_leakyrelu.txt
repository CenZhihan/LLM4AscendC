#include "gemm_multiply_leakyrelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (per core, conservative):
//   xChunk: 1 * tileK
//   wChunk: 1 * tileK
//   tmp:    1 * tileK
//   acc/buf: a few scalars
// Total floats ~ 3 * tileK => bytes ~ 12 * tileK
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per core shared
static constexpr uint32_t ALIGN_ELEMS = 256;             // prefer DMA-friendly lengths
static constexpr uint32_t TILEK_HARD_CAP = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMultiplyLeakyreluCustomTilingData tiling;

    const auto *x_shape = context->GetInputShape(0)->GetOriginShapePtr();
    const auto *w_shape = context->GetInputShape(1)->GetOriginShapePtr();
    const auto *b_shape = context->GetInputShape(2)->GetOriginShapePtr();

    // Expect x: [M, K], w: [N, K], b: [N]
    if (x_shape->GetDimNum() != 2 || w_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t Nb = static_cast<uint32_t>(b_shape->GetDim(0));

    if (Kw != K || Nb != N) {
        return ge::GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *multiplier = attrs->GetAttrPointer<float>(0);
    const float *negativeSlope = attrs->GetAttrPointer<float>(1);

    // tileK from UB budget: bytes ~ 12 * tileK
    uint32_t maxTileK = UB_BUDGET_BYTES / 12;
    maxTileK = AlignDown(maxTileK, ALIGN_ELEMS);
    if (maxTileK < ALIGN_ELEMS) maxTileK = ALIGN_ELEMS;
    if (maxTileK > TILEK_HARD_CAP) maxTileK = AlignDown(TILEK_HARD_CAP, ALIGN_ELEMS);

    uint32_t tileK = maxTileK;
    if (K < tileK) {
        const uint32_t aligned = AlignDown(K, ALIGN_ELEMS);
        tileK = (aligned > 0) ? aligned : K;
    }

    // blockDim on rows (M). Keep within [8, 24]
    uint32_t blockDim = (M == 0) ? 8u : (M < 8u ? 8u : (M > 24u ? 24u : M));
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);
    tiling.set_multiplier(*multiplier);
    tiling.set_negativeSlope(*negativeSlope);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMultiplyLeakyreluCustom {
public:
    __aicore__ inline KernelGemmMultiplyLeakyreluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileK,
                                float multiplier, float negativeSlope)
    {
        M_ = M; N_ = N; K_ = K; tileK_ = tileK;
        mult_ = multiplier;
        negSlope_ = negativeSlope;
        negCorr_ = negSlope_ - 1.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        pipe_.InitBuffer(xVecQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(wVecQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQ_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(oneQ_, 1, sizeof(float));  // for scalar 1-length vectors
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0 || tileK_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            // Iterate columns one-by-one (robust and memory-safe)
            for (uint32_t n = 0; n < N_; ++n) {
                bool accInitialized = false;
                AscendC::LocalTensor<float> acc1 = oneQ_.AllocTensor<float>(); // 1 element
                // We will fill acc1 with first partial sum, then accumulate others
                for (uint32_t kb = 0; kb < K_; kb += tileK_) {
                    uint32_t curK = K_ - kb;
                    if (curK > tileK_) curK = tileK_;

                    // Copy chunks from GM
                    CopyInX(m, kb, curK);
                    CopyInW(n, kb, curK);

                    // Multiply elementwise: tmp = xChunk * wChunk
                    AscendC::LocalTensor<float> xLocal = xVecQ_.DeQue<float>();
                    AscendC::LocalTensor<float> wLocal = wVecQ_.DeQue<float>();
                    AscendC::LocalTensor<float> tmpLocal = tmpVecQ_.AllocTensor<float>();
                    AscendC::Mul(tmpLocal, xLocal, wLocal, curK);

                    // Reduce tmpLocal[curK] -> sumLocal[1]
                    AscendC::LocalTensor<float> sumLocal = oneQ_.AllocTensor<float>();
                    // Initialize sumLocal to zero by reusing it as destination of reduction
                    // Use a naive halving reduction into sumLocal by built-in ReduceSum (if available)
                    AscendC::ReduceSum(sumLocal, tmpLocal, curK);

                    tmpVecQ_.FreeTensor(tmpLocal);
                    xVecQ_.FreeTensor(xLocal);
                    wVecQ_.FreeTensor(wLocal);

                    if (!accInitialized) {
                        // acc1 = sumLocal
                        AscendC::DataCopy(acc1, sumLocal, 1);
                        accInitialized = true;
                    } else {
                        // acc1 += sumLocal
                        AscendC::Add(acc1, acc1, sumLocal, 1);
                    }
                    oneQ_.FreeTensor(sumLocal);
                }

                // y = acc1 + b[n]
                AscendC::LocalTensor<float> bLocal = oneQ_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[n], 1);
                AscendC::Add(acc1, acc1, bLocal, 1);
                oneQ_.FreeTensor(bLocal);

                // y *= multiplier
                AscendC::Muls(acc1, acc1, mult_, 1);

                // LeakyReLU: y = y + (negSlope-1) * min(y, 0)
                AscendC::LocalTensor<float> tLocal = oneQ_.AllocTensor<float>();
                AscendC::Mins(tLocal, acc1, 0.0f, 1);
                AscendC::Muls(tLocal, tLocal, negCorr_, 1);
                AscendC::Add(acc1, acc1, tLocal, 1);
                oneQ_.FreeTensor(tLocal);

                // Store to GM: y[m, n] => offset m*N_ + n
                const uint64_t yOff = static_cast<uint64_t>(m) * N_ + n;
                AscendC::DataCopy(yGm_[yOff], acc1, 1);

                oneQ_.FreeTensor(acc1);
            }
        }
    }

private:
    __aicore__ inline void CopyInX(uint32_t m, uint32_t kb, uint32_t curK)
    {
        const uint64_t xOff = static_cast<uint64_t>(m) * K_ + kb;
        AscendC::LocalTensor<float> xLocal = xVecQ_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[xOff], curK);
        xVecQ_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInW(uint32_t n, uint32_t kb, uint32_t curK)
    {
        const uint64_t wOff = static_cast<uint64_t>(n) * K_ + kb; // row-major contiguous
        AscendC::LocalTensor<float> wLocal = wVecQ_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_[wOff], curK);
        wVecQ_.EnQue(wLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xVecQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> oneQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileK_{0};
    float mult_{1.0f};
    float negSlope_{0.01f};
    float negCorr_{-0.99f};
};

extern "C" __global__ __aicore__ void gemm_multiply_leakyrelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMultiplyLeakyreluCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileK,
            tiling_data.multiplier, tiling_data.negativeSlope);
    op.Process();
}
