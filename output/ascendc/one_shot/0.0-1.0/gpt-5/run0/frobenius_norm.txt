host_operator_src="""

#include "frobenius_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue: 2 * tile
// Total floats = 6 * tile => bytes = 24 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    FrobeniusNormCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // tileLen from UB budget: bytes = 24 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 24;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // Adaptive blockDim based on number of tiles; keep conservative for stability.
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // Workspace layout (float):
    //   sums[0..blockDim-1]
    //   flags[blockDim..2*blockDim-1]    // per-block ready flags
    //   norm[2*blockDim]                 // single float of the global norm
    //   ready_flag[2*blockDim + 1]       // single float flag
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = static_cast<size_t>((2 * blockDim + 2) * sizeof(float));
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelFrobeniusNormCustom {
public:
    __aicore__ inline KernelFrobeniusNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR workspace,
                                uint32_t totalLength, uint32_t tileLength, uint32_t blockDim)
    {
        totalLen_ = totalLength;
        tileLen_  = tileLength;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        wsGm_.SetGlobalBuffer((__gm__ float*)workspace, (2 * blockDim_ + 2));

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) {
            // Mark done for phase 1 even if no work.
            WriteWSFloat(flagOffset(blockIdx), 1.0f);
            return;
        }

        // Phase 1: accumulate local sum of squares for this block.
        float localSum = 0.0f;
        for (uint32_t pos = base; pos < totalLen_; pos += strideElems) {
            uint32_t len = totalLen_ - pos;
            if (len > tileLen_) len = tileLen_;
            CopyIn(pos, len);
            localSum += SumSquares(len);
        }

        // Write partial sum and flag.
        WriteWSFloat(sumOffset(blockIdx), localSum);
        WriteWSFloat(flagOffset(blockIdx), 1.0f);

        // Block 0: wait for all flags then compute global norm and broadcast
        if (blockIdx == 0) {
            // Wait all blocks
            for (uint32_t i = 0; i < blockDim_; ++i) {
                while (ReadWSFloat(flagOffset(i)) < 0.5f) {
                    // spin-wait
                }
            }
            // Reduce partial sums
            float totalSum = 0.0f;
            for (uint32_t i = 0; i < blockDim_; ++i) {
                totalSum += ReadWSFloat(sumOffset(i));
            }
            float norm = AscendC::Sqrt(totalSum);
            if (!(norm > 0.0f)) {
                // handle zero norm
                norm = 0.0f;
            }
            WriteWSFloat(normIndex(), norm);
            // ready flag
            WriteWSFloat(readyIndex(), 1.0f);
        }

        // All blocks wait for ready flag to be set, then read norm and compute inv-norm.
        while (ReadWSFloat(readyIndex()) < 0.5f) {
            // spin-wait
        }
        const float normVal = ReadWSFloat(normIndex());
        const float invNorm = (normVal > 0.0f) ? (1.0f / normVal) : 0.0f;

        // Phase 2: Normalize and write out
        for (uint32_t pos = base; pos < totalLen_; pos += strideElems) {
            uint32_t len = totalLen_ - pos;
            if (len > tileLen_) len = tileLen_;
            NormalizeAndStore(pos, len, invNorm);
        }
    }

private:
    __aicore__ inline uint32_t sumOffset(uint32_t idx) const { return idx; }
    __aicore__ inline uint32_t flagOffset(uint32_t idx) const { return blockDim_ + idx; }
    __aicore__ inline uint32_t normIndex() const { return 2 * blockDim_; }
    __aicore__ inline uint32_t readyIndex() const { return 2 * blockDim_ + 1; }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline float SumSquares(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> sqLocal = tmpQueue_.AllocTensor<float>();
        AscendC::Mul(sqLocal, xLocal, xLocal, len);

        AscendC::LocalTensor<float> accLocal = tmpQueue_.AllocTensor<float>();
        AscendC::ReduceSum(accLocal, sqLocal, len);

        // Extract scalar from local tensor
        float val = AscendC::GetValue(accLocal, 0);

        tmpQueue_.FreeTensor(accLocal);
        tmpQueue_.FreeTensor(sqLocal);
        inQueueX_.FreeTensor(xLocal);
        return val;
    }

    __aicore__ inline void NormalizeAndStore(uint32_t gmOffset, uint32_t len, float invNorm)
    {
        // Load
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);

        AscendC::LocalTensor<float> xin = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::Muls(yLocal, xin, invNorm, len);

        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xin);

        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yOut, len);
        outQueueY_.FreeTensor(yOut);
    }

    __aicore__ inline void WriteWSFloat(uint32_t idx, float value)
    {
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        AscendC::Duplicate(tmp, value, 1);
        AscendC::DataCopy(wsGm_[idx], tmp, 1);
        tmpQueue_.FreeTensor(tmp);
    }

    __aicore__ inline float ReadWSFloat(uint32_t idx)
    {
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(tmp, wsGm_[idx], 1);
        float val = AscendC::GetValue(tmp, 0);
        tmpQueue_.FreeTensor(tmp);
        return val;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wsGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
};

extern "C" __global__ __aicore__ void frobenius_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelFrobeniusNormCustom op;
    op.Init(x, y, workspace, tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim);
    op.Process();
}
