host_operator_src="""

#include "instance_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    InstanceNormCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();

    uint32_t dimNum = static_cast<uint32_t>(origin.GetDimNum());
    uint32_t n = (dimNum >= 1) ? static_cast<uint32_t>(origin.GetDim(0)) : 1u;
    uint32_t c = (dimNum >= 2) ? static_cast<uint32_t>(origin.GetDim(1)) : 1u;

    uint32_t innerLen = 1u;
    for (uint32_t i = 2; i < dimNum; ++i) {
        innerLen *= static_cast<uint32_t>(origin.GetDim(i));
    }
    if (innerLen == 0u) innerLen = 1u;

    const uint32_t groupCount = n * c;
    const uint32_t totalLength =
        static_cast<uint32_t>(in_shape->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);

    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // 5 buffers * float(4B)
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen;
    if (innerLen >= ALIGN_ELEMS) {
        tileLen = AlignDown((innerLen < maxTile) ? innerLen : maxTile, ALIGN_ELEMS);
        if (tileLen == 0u) tileLen = ALIGN_ELEMS;
    } else {
        tileLen = innerLen;  // tiny spatial, no need to align
    }

    uint32_t blockDim = groupCount;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_innerLength(innerLen);
    tiling.set_tileLength(tileLen);
    tiling.set_groupCount(groupCount);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class InstanceNormCustom : public OpDef {
public:
    explicit InstanceNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("eps").AttrType(OPTIONAL).Float(1e-5);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(InstanceNormCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelInstanceNormCustom {
public:
    __aicore__ inline KernelInstanceNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t innerLength,
                                uint32_t groupCount,
                                uint32_t tileLength,
                                float eps)
    {
        totalLen_ = totalLength;
        innerLen_ = innerLength;
        groupCnt_ = groupCount;
        tileLen_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerLen_ == 0 || groupCnt_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t g = blockIdx; g < groupCnt_; g += blockNum) {
            float mean = 0.0f;
            float invStd = 1.0f;

            ReduceGroup(g, mean, invStd);
            NormalizeGroup(g, mean, invStd);
        }
    }

private:
    __aicore__ inline void ReduceGroup(uint32_t g, float &mean, float &invStd)
    {
        float sum = 0.0f;
        float sumSq = 0.0f;

        uint32_t processed = 0;
        while (processed < innerLen_) {
            const uint32_t len = (innerLen_ - processed > tileLen_) ? tileLen_ : (innerLen_ - processed);
            const uint32_t offset = g * innerLen_ + processed;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);

            // sum(x)
            float partSum = AscendC::ReduceSum(xLocal, len);

            // sum(x^2)
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(tLocal, xLocal, xLocal, len);
            float partSqSum = AscendC::ReduceSum(tLocal, len);

            tmpQueue_.FreeTensor(tLocal);
            inQueueX_.FreeTensor(xLocal);

            sum += partSum;
            sumSq += partSqSum;
            processed += len;
        }

        mean = sum / static_cast<float>(innerLen_);
        float var = sumSq / static_cast<float>(innerLen_) - mean * mean;
        if (var < 0.0f) var = 0.0f;
        invStd = 1.0f / sqrtf(var + eps_);
    }

    __aicore__ inline void NormalizeGroup(uint32_t g, float mean, float invStd)
    {
        uint32_t processed = 0;
        while (processed < innerLen_) {
            const uint32_t len = (innerLen_ - processed > tileLen_) ? tileLen_ : (innerLen_ - processed);
            const uint32_t offset = g * innerLen_ + processed;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            AscendC::DataCopy(xLocal, xGm_[offset], len);

            // t = x - mean
            AscendC::Adds(tLocal, xLocal, -mean, len);
            // y = (x - mean) * invStd
            AscendC::Muls(yLocal, tLocal, invStd, len);

            tmpQueue_.FreeTensor(tLocal);
            inQueueX_.FreeTensor(xLocal);
            outQueueY_.EnQue(yLocal);

            AscendC::LocalTensor<float> yDeq = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[offset], yDeq, len);
            outQueueY_.FreeTensor(yDeq);

            processed += len;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t innerLen_{0};
    uint32_t groupCnt_{0};
    uint32_t tileLen_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void instance_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelInstanceNormCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.innerLength,
            tiling_data.groupCount,
            tiling_data.tileLength,
            tiling_data.eps);
    op.Process();
}
"""
