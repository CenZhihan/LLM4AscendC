static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedACustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    // Expect A: (K, M), B: (K, N)
    uint32_t K = static_cast<uint32_t>(a_shape->GetOriginShape().GetDimNum() >= 1 ? a_shape->GetOriginShape().GetDim(0) : 1);
    uint32_t M = static_cast<uint32_t>(a_shape->GetOriginShape().GetDimNum() >= 2 ? a_shape->GetOriginShape().GetDim(1) : 1);
    uint32_t N = static_cast<uint32_t>(b_shape->GetOriginShape().GetDimNum() >= 2 ? b_shape->GetOriginShape().GetDim(1) : 1);

    // Tile defaults
    uint32_t tileM = 64;
    uint32_t tileN = 128;
    uint32_t tileK = 64;

    if (M < tileM) tileM = M;
    if (N < tileN) tileN = N;
    if (K < tileK) tileK = K;

    uint32_t tilesM = (tileM == 0) ? 1u : CeilDiv(M, tileM);
    uint32_t tilesN = (tileN == 0) ? 1u : CeilDiv(N, tileN);

    if (blockDim == 0u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_tilesM(tilesM);
    tiling.set_tilesN(tilesN);
    tiling.set_totalTiles(totalTiles);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulWithTransposedACustom {
public:
    __aicore__ inline KernelMatmulWithTransposedACustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN,
                                uint32_t tilesM, uint32_t tilesN)
    {
        M_ = M; K_ = K; N_ = N;
        tM_ = tileM; tK_ = tileK; tN_ = tileN;
        tilesM_ = tilesM; tilesN_ = tilesN;

        aGm_.SetGlobalBuffer((__gm__ float*)a, (uint64_t)K_ * (uint64_t)M_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)K_ * (uint64_t)N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * (uint64_t)N_);

        pipe_.InitBuffer(aRowQ_, BUFFER_NUM, tM_ * sizeof(float));
        pipe_.InitBuffer(bTileQ_, BUFFER_NUM, tK_ * tN_ * sizeof(float));
        pipe_.InitBuffer(cTileQ_, BUFFER_NUM, tM_ * tN_ * sizeof(float));
        pipe_.InitBuffer(tmpRowQ_, BUFFER_NUM, tN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalTiles = tilesM_ * tilesN_;
        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            const uint32_t mi = t / tilesN_;
            const uint32_t nj = t % tilesN_;
            const uint32_t m0 = mi * tM_;
            const uint32_t n0 = nj * tN_;

            uint32_t mLen = M_ - m0; if (mLen > tM_) mLen = tM_;
            uint32_t nLen = N_ - n0; if (nLen > tN_) nLen = tN_;

            if (mLen == 0 || nLen == 0) continue;

            // Allocate and zero C tile
            AscendC::LocalTensor<float> cTile = cTileQ_.AllocTensor<float>();
            AscendC::Duplicate(cTile, 0.0f, mLen * nLen);

            // Iterate K in chunks
            for (uint32_t k0 = 0; k0 < K_; k0 += tK_) {
                uint32_t kLen = K_ - k0; if (kLen > tK_) kLen = tK_;
                if (kLen == 0) break;

                // Load B tile: [kLen x nLen]
                AscendC::LocalTensor<float> bTile = bTileQ_.AllocTensor<float>();
                for (uint32_t kk = 0; kk < kLen; ++kk) {
                    const uint64_t bOffset = (uint64_t)(k0 + kk) * (uint64_t)N_ + (uint64_t)n0;
                    AscendC::DataCopy(bTile[kk * nLen], bGm_[bOffset], nLen);
                }
                bTileQ_.EnQue(bTile);
                AscendC::LocalTensor<float> bTileL = bTileQ_.DeQue<float>();

                AscendC::LocalTensor<float> tmpRow = tmpRowQ_.AllocTensor<float>();

                // Accumulate: C[m, :] += sum_{kk} A[k0+kk, m0:m0+mLen] * B_row(kk)
                for (uint32_t kk = 0; kk < kLen; ++kk) {
                    AscendC::LocalTensor<float> bRow = bTileL[kk * nLen];

                    // Load A row segment (length mLen) for this kk
                    AscendC::LocalTensor<float> aRow = aRowQ_.AllocTensor<float>();
                    const uint64_t aOffset = (uint64_t)(k0 + kk) * (uint64_t)M_ + (uint64_t)m0;
                    AscendC::DataCopy(aRow, aGm_[aOffset], mLen);

                    // For each m in tile, broadcast a scalar and FMA into C row
                    for (uint32_t m = 0; m < mLen; ++m) {
                        float aScalar = aRow.GetValue(m);
                        AscendC::Duplicate(tmpRow, aScalar, nLen);
                        AscendC::Mul(tmpRow, tmpRow, bRow, nLen);

                        AscendC::LocalTensor<float> cRow = cTile[m * nLen];
                        AscendC::Add(cRow, cRow, tmpRow, nLen);
                    }

                    aRowQ_.FreeTensor(aRow);
                }

                tmpRowQ_.FreeTensor(tmpRow);
                bTileQ_.FreeTensor(bTileL);
            }

            // Store C tile back: rows m0..m0+mLen-1, cols n0..n0+nLen-1
            for (uint32_t m = 0; m < mLen; ++m) {
                const uint64_t yOffset = (uint64_t)(m0 + m) * (uint64_t)N_ + (uint64_t)n0;
                AscendC::LocalTensor<float> cRow = cTile[m * nLen];
                AscendC::DataCopy(yGm_[yOffset], cRow, nLen);
            }
            cTileQ_.FreeTensor(cTile);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aRowQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bTileQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> cTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpRowQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, K_{0}, N_{0};
    uint32_t tM_{0}, tK_{0}, tN_{0};
    uint32_t tilesM_{0}, tilesN_{0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_a_custom(GM_ADDR A, GM_ADDR B, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedACustom op;
    op.Init(A, B, y,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN,
            tiling_data.tilesM, tiling_data.tilesN);
    op.Process();
}
