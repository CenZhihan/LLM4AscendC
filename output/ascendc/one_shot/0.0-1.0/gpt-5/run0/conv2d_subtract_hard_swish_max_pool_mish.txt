kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF_F = -1.0e30f;

class KernelConv2dSubtractHardSwishMaxPoolMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractHardSwishMaxPoolMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, const optiling::Conv2dSubtractHardSwishMaxPoolMishCustomTilingData &t)
    {
        n_ = t.n();
        ci_ = t.cIn();
        hi_ = t.hIn();
        wi_ = t.wIn();
        co_ = t.cOut();
        kH_ = t.kH();
        kW_ = t.kW();
        hConv_ = t.hConv();
        wConv_ = t.wConv();
        ho_ = t.hOut();
        wo_ = t.wOut();
        ocChunk_ = t.ocChunk();
        poolK_ = t.poolK();
        subtractValue_ = t.subtractValue();

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_*ci_*hi_*wi_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)co_*ci_*kH_*kW_);
        if (b != nullptr) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)co_);
            hasBias_ = true;
        } else {
            hasBias_ = false;
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_*co_*ho_*wo_);

        // UB buffers
        // We will reuse small buffers frequently: scalar cell, weight plane, x patch, vectors
        // Allocate enough to cover worst-case based on ocChunk_ and ci_*kH_*kW_
        const uint32_t planeSize = ci_ * kW_ * kH_;

        pipe_.InitBuffer(tmpScalarQ_, 1, sizeof(float));                 // scalar element
        pipe_.InitBuffer(wPlaneQ_, 1, planeSize * sizeof(float));        // one weight plane for a single oc
        pipe_.InitBuffer(xPatchQ_, 1, planeSize * sizeof(float));        // x patch covering all channels
        pipe_.InitBuffer(vecQ_, 4, ocChunk_ * sizeof(float));            // yVec, yMax, biasVec, tmp
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || ci_ == 0 || co_ == 0 || ho_ == 0 || wo_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t totalPooled = (uint64_t)n_ * ho_ * wo_;

        for (uint64_t pooledIdx = blockIdx; pooledIdx < totalPooled; pooledIdx += blockNum) {
            uint32_t nIdx, hoIdx, woIdx;
            DecomposePooledIndex(pooledIdx, nIdx, hoIdx, woIdx);

            // For each oc tile
            for (uint32_t ocBase = 0; ocBase < co_; ocBase += ocChunk_) {
                const uint32_t chunk = AscendC::min(ocChunk_, co_ - ocBase);

                AscendC::LocalTensor<float> yMax = vecQ_.AllocTensor<float>();
                FillConst(yMax, chunk, NEG_INF_F);

                // Iterate pool window
                for (uint32_t ph = 0; ph < poolK_; ++ph) {
                    for (uint32_t pw = 0; pw < poolK_; ++pw) {
                        const uint32_t hc = hoIdx * poolK_ + ph;
                        const uint32_t wc = woIdx * poolK_ + pw;

                        // Prepare x patch for this conv location across all input channels
                        AscendC::LocalTensor<float> xPatch = xPatchQ_.AllocTensor<float>();
                        LoadXPatch(nIdx, hc, wc, xPatch);

                        // yVec initialize with bias or zero
                        AscendC::LocalTensor<float> yVec = vecQ_.AllocTensor<float>();
                        if (hasBias_) {
                            LoadBiasVec(ocBase, chunk, yVec);
                        } else {
                            FillConst(yVec, chunk, 0.0f);
                        }

                        // Accumulate convolution: for each oc in chunk, dot(wPlane(oc), xPatch)
                        for (uint32_t ocOff = 0; ocOff < chunk; ++ocOff) {
                            const uint32_t oc = ocBase + ocOff;
                            AscendC::LocalTensor<float> wPlane = wPlaneQ_.AllocTensor<float>();
                            LoadWeightPlane(oc, wPlane);

                            float sum = 0.0f;
                            const uint32_t planeSize = ci_ * kH_ * kW_;
                            // Accumulate dot product
                            for (uint32_t idx = 0; idx < planeSize; ++idx) {
                                float a, b;
                                // load wPlane[idx]
                                LoadScalarFromLocal(wPlane, idx, a);
                                // load xPatch[idx]
                                LoadScalarFromLocal(xPatch, idx, b);
                                sum += a * b;
                            }
                            // write sum into yVec[ocOff] += sum
                            float prev;
                            LoadScalarFromLocal(yVec, ocOff, prev);
                            prev += sum;
                            StoreScalarToLocal(yVec, ocOff, prev);

                            wPlaneQ_.FreeTensor(wPlane);
                        }

                        // yVec = yVec - subtractValue
                        AddsInplace(yVec, chunk, -subtractValue_);

                        // HardSwish: y = y * clamp(y + 3, 0, 6) / 6
                        HardSwishInplace(yVec, chunk);

                        // yMax = max(yMax, yVec)
                        MaxInplace(yMax, yVec, chunk);

                        vecQ_.FreeTensor(yVec);
                        xPatchQ_.FreeTensor(xPatch);
                    }
                }

                // Mish: y = y * tanh(ln(1 + exp(y)))
                MishInplace(yMax, chunk);

                // Store results to GM at [nIdx, ocBase:ocBase+chunk, hoIdx, woIdx]
                StoreOutputVector(nIdx, hoIdx, woIdx, ocBase, yMax, chunk);

                vecQ_.FreeTensor(yMax);
            }
        }
    }

private:
    __aicore__ inline void DecomposePooledIndex(uint64_t pooledIdx, uint32_t &nIdx, uint32_t &hoIdx, uint32_t &woIdx)
    {
        const uint64_t hw = (uint64_t)ho_ * wo_;
        nIdx = static_cast<uint32_t>(pooledIdx / hw);
        const uint64_t rem = pooledIdx % hw;
        hoIdx = static_cast<uint32_t>(rem / wo_);
        woIdx = static_cast<uint32_t>(rem % wo_);
    }

    // Global layout assumptions: x[N, C, H, W] contiguous; w[O, I, kH, kW] contiguous; y[N, O, Ho, Wo]
    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    { return (((uint64_t)n * ci_ + c) * hi_ + h) * wi_ + w; }

    __aicore__ inline uint64_t WPlaneOffset(uint32_t o) const
    { return (uint64_t)o * ci_ * kH_ * kW_; }

    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t o, uint32_t h, uint32_t w) const
    { return (((uint64_t)n * co_ + o) * ho_ + h) * wo_ + w; }

    __aicore__ inline void LoadScalarFromGM(uint64_t gmOffset, float &value)
    {
        AscendC::LocalTensor<float> s = tmpScalarQ_.AllocTensor<float>();
        AscendC::DataCopy(s, xGm_[gmOffset], 1);
        value = *reinterpret_cast<float*>(s.GetPtr(0));
        tmpScalarQ_.FreeTensor(s);
    }

    __aicore__ inline void LoadScalarFromLocal(AscendC::LocalTensor<float> &buf, uint32_t idx, float &value)
    {
        // Access local tensor scalar via a 1-length copy
        AscendC::LocalTensor<float> s = tmpScalarQ_.AllocTensor<float>();
        AscendC::DataCopy(s, buf[idx], 1);
        value = *reinterpret_cast<float*>(s.GetPtr(0));
        tmpScalarQ_.FreeTensor(s);
    }

    __aicore__ inline void StoreScalarToLocal(AscendC::LocalTensor<float> &buf, uint32_t idx, float value)
    {
        AscendC::LocalTensor<float> s = tmpScalarQ_.AllocTensor<float>();
        *reinterpret_cast<float*>(s.GetPtr(0)) = value;
        AscendC::DataCopy(buf[idx], s, 1);
        tmpScalarQ_.FreeTensor(s);
    }

    __aicore__ inline void FillConst(AscendC::LocalTensor<float> &buf, uint32_t len, float v)
    {
        // set by scalar store
        for (uint32_t i = 0; i < len; ++i) {
            StoreScalarToLocal(buf, i, v);
        }
    }

    __aicore__ inline void LoadBiasVec(uint32_t ocBase, uint32_t len, AscendC::LocalTensor<float> &dst)
    {
        for (uint32_t i = 0; i < len; ++i) {
            AscendC::LocalTensor<float> s = tmpScalarQ_.AllocTensor<float>();
            AscendC::DataCopy(s, bGm_[ocBase + i], 1);
            AscendC::DataCopy(dst[i], s, 1);
            tmpScalarQ_.FreeTensor(s);
        }
    }

    __aicore__ inline void LoadWeightPlane(uint32_t oc, AscendC::LocalTensor<float> &dst)
    {
        const uint64_t off = WPlaneOffset(oc);
        const uint32_t planeSize = ci_ * kH_ * kW_;
        AscendC::DataCopy(dst, wGm_[off], planeSize);
    }

    __aicore__ inline void LoadXPatch(uint32_t nIdx, uint32_t hc, uint32_t wc, AscendC::LocalTensor<float> &dst)
    {
        // dst shape flattened: [ci, kH, kW]
        uint32_t pos = 0;
        for (uint32_t c = 0; c < ci_; ++c) {
            for (uint32_t kh = 0; kh < kH_; ++kh) {
                const uint32_t hIn = hc + kh;
                const uint32_t wStart = wc;
                // copy kW_ contiguous elements from x[n,c,hIn,wStart]
                const uint64_t base = XOffset(nIdx, c, hIn, wStart);
                AscendC::DataCopy(dst[pos], xGm_[base], kW_);
                pos += kW_;
            }
        }
    }

    __aicore__ inline void AddsInplace(AscendC::LocalTensor<float> &vec, uint32_t len, float v)
    {
        for (uint32_t i = 0; i < len; ++i) {
            float t;
            LoadScalarFromLocal(vec, i, t);
            t += v;
            StoreScalarToLocal(vec, i, t);
        }
    }

    __aicore__ inline float Clamp(float x, float lo, float hi) { return x < lo ? lo : (x > hi ? hi : x); }

    __aicore__ inline void HardSwishInplace(AscendC::LocalTensor<float> &vec, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            float x;
            LoadScalarFromLocal(vec, i, x);
            float t = Clamp(x + 3.0f, 0.0f, 6.0f);
            float y = x * (t * (1.0f / 6.0f));
            StoreScalarToLocal(vec, i, y);
        }
    }

    __aicore__ inline void MaxInplace(AscendC::LocalTensor<float> &dst, AscendC::LocalTensor<float> &src, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            float a, b;
            LoadScalarFromLocal(dst, i, a);
            LoadScalarFromLocal(src, i, b);
            StoreScalarToLocal(dst, i, (a > b) ? a : b);
        }
    }

    __aicore__ inline float TanhApprox(float x)
    {
        // Fallback to std::tanhf if supported; otherwise a simple polynomial approx
        return tanhf(x);
    }

    __aicore__ inline float Softplus(float x)
    {
        // numerically stable softplus
        if (x > 20.0f) return x;                  // exp(x) is huge => log1p(exp(x)) ~ x
        if (x < -20.0f) return expf(x);           // exp(x) tiny => log1p(exp(x)) ~ exp(x)
        return log1pf(expf(x));
    }

    __aicore__ inline void MishInplace(AscendC::LocalTensor<float> &vec, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            float x;
            LoadScalarFromLocal(vec, i, x);
            float sp = Softplus(x);
            float t = TanhApprox(sp);
            float y = x * t;
            StoreScalarToLocal(vec, i, y);
        }
    }

    __aicore__ inline void StoreOutputVector(uint32_t nIdx, uint32_t hoIdx, uint32_t woIdx, uint32_t ocBase,
                                             AscendC::LocalTensor<float> &vec, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            float v;
            LoadScalarFromLocal(vec, i, v);
            AscendC::LocalTensor<float> s = tmpScalarQ_.AllocTensor<float>();
            *reinterpret_cast<float*>(s.GetPtr(0)) = v;
            const uint64_t off = YOffset(nIdx, ocBase + i, hoIdx, woIdx);
            AscendC::DataCopy(yGm_[off], s, 1);
            tmpScalarQ_.FreeTensor(s);
        }
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpScalarQ_; // 1-float scratch
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wPlaneQ_;    // weight plane buffer
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xPatchQ_;    // x patch buffer
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> vecQ_;       // general vectors

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, ci_{0}, hi_{0}, wi_{0};
    uint32_t co_{0}, kH_{0}, kW_{0};
    uint32_t hConv_{0}, wConv_{0}, ho_{0}, wo_{0};
    uint32_t ocChunk_{1}, poolK_{1};
    float subtractValue_{0.0f};
    bool hasBias_{false};
};
