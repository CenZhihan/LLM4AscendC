    if (v > hi) return hi;
    return v;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedBothCustomTilingData tiling;

    const gert::Shape &a_shape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape &b_shape = context->GetInputShape(1)->GetOriginShape();

    if (a_shape.GetDimNum() != 2 || b_shape.GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t K = static_cast<uint32_t>(a_shape.GetDim(0));
    const uint32_t M = static_cast<uint32_t>(a_shape.GetDim(1));
    const uint32_t N = static_cast<uint32_t>(b_shape.GetDim(0));
    // Expect b_shape.GetDim(1) == K

    const uint64_t totalOut = static_cast<uint64_t>(M) * static_cast<uint64_t>(N);
    uint32_t blockDim = 1u;
    if (totalOut >= 24ull) {
        // Heuristic: more outputs -> more blocks, capped conservatively
        blockDim = Clamp(static_cast<uint32_t>(totalOut / 4096ull), 1u, 24u);
    }

    context->SetBlockDim(blockDim);

    tiling.set_m(M);
    tiling.set_n(N);
    tiling.set_k(K);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelMatmulWithTransposedBothCustom {
public:
    __aicore__ inline KernelMatmulWithTransposedBothCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K)
    {
        M_ = M;
        N_ = N;
        K_ = K;

        aGm_.SetGlobalBuffer((__gm__ float*)a, static_cast<uint64_t>(K_) * M_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, static_cast<uint64_t>(N_) * K_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        // Scalar buffers (1 element each) used for accumulation and multiply
        pipe_.InitBuffer(scalarQ_, 4, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalOut = static_cast<uint64_t>(M_) * static_cast<uint64_t>(N_);

        AscendC::LocalTensor<float> aVal = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> bVal = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> mulVal = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> accVal = scalarQ_.AllocTensor<float>();

        for (uint64_t idx = blockIdx; idx < totalOut; idx += blockNum) {
            const uint32_t i = static_cast<uint32_t>(idx / N_); // row in Y (and column in A)
            const uint32_t j = static_cast<uint32_t>(idx % N_); // col in Y (and row in B)

            AscendC::Duplicate(accVal, 0.0f, 1);

            // y[i, j] = sum_k A[k, i] * B[j, k]
            for (uint32_t k = 0; k < K_; ++k) {
                const uint64_t aOff = static_cast<uint64_t>(k) * M_ + i; // A is [K, M]
                const uint64_t bOff = static_cast<uint64_t>(j) * K_ + k; // B is [N, K]
                AscendC::DataCopy(aVal, aGm_[aOff], 1);
                AscendC::DataCopy(bVal, bGm_[bOff], 1);
                AscendC::Mul(mulVal, aVal, bVal, 1);
                AscendC::Add(accVal, accVal, mulVal, 1);
            }

            const uint64_t yOff = static_cast<uint64_t>(i) * N_ + j; // Y is [M, N]
            AscendC::DataCopy(yGm_[yOff], accVal, 1);
        }

        scalarQ_.FreeTensor(accVal);
        scalarQ_.FreeTensor(mulVal);
        scalarQ_.FreeTensor(bVal);
        scalarQ_.FreeTensor(aVal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> scalarQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR a, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedBothCustom op;
    op.Init(a, b, y, tiling_data.m, tiling_data.n, tiling_data.k);
    op.Process();
}
