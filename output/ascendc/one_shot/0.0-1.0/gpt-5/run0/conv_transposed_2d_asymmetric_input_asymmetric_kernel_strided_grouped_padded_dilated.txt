#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
                                uint32_t G, uint32_t CoutPg,
                                uint32_t kH, uint32_t kW,
                                uint32_t sH, uint32_t sW,
                                uint32_t pH, uint32_t pW,
                                uint32_t dH, uint32_t dW,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t tileW)
    {
        N_ = N; Cin_ = Cin; H_ = H; W_ = W;
        G_ = G; CoutPg_ = CoutPg;
        kH_ = kH; kW_ = kW;
        sH_ = sH; sW_ = sW;
        pH_ = pH; pW_ = pW;
        dH_ = dH; dW_ = dW;
        Ho_ = Ho; Wo_ = Wo;
        tileW_ = tileW;
        Co_ = G_ * CoutPg_;
        CinPg_ = Cin_ / G_;
        kArea_ = kH_ * kW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cin_ * CoutPg_ * kArea_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Co_ * Ho_ * Wo_);

        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(xQueue_, 1, tileW_ * sizeof(float));      // enough for segment
        pipe_.InitBuffer(wQueue_, 1, kArea_ * sizeof(float));       // per-cin kernel window
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Ho_ == 0 || Wo_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numLines = N_ * G_ * CoutPg_ * Ho_;
        uint32_t line = blockIdx;

        while (line < numLines) {
            // Decode line -> (n, g, co, ho)
            uint32_t tmp = line;
            const uint32_t ho = tmp % Ho_; tmp /= Ho_;
            const uint32_t co = tmp % CoutPg_; tmp /= CoutPg_;
            const uint32_t g  = tmp % G_; tmp /= G_;
            const uint32_t n  = tmp % N_;

            const uint32_t coGlobal = g * CoutPg_ + co;

            // Iterate width tiles
            for (uint32_t w0 = 0; w0 < Wo_; w0 += tileW_) {
                uint32_t len = Wo_ - w0;
                if (len > tileW_) len = tileW_;

                // y accumulation buffer
                AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                AscendC::SetValue(yLocal, 0.0f, len);

                // Loop over input channels within group
                for (uint32_t cinPg = 0; cinPg < CinPg_; ++cinPg) {
                    const uint32_t cinGlobal = g * CinPg_ + cinPg;

                    // Load weights for this (cinGlobal, co) across kH*kW into wLocal
                    // contiguous block starting at kh=0,kw=0
                    const uint64_t wBase = (((uint64_t)cinGlobal * CoutPg_ + co) * kH_) * kW_;
                    AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                    AscendC::DataCopy(wLocal, wGm_[wBase], kArea_);

                    // Traverse kernel window and accumulate
                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        // deconv mapping for height
                        int32_t hin_num = static_cast<int32_t>(ho) + static_cast<int32_t>(pH_) - static_cast<int32_t>(kh * dH_);
                        if (hin_num < 0) continue;
                        if (hin_num % static_cast<int32_t>(sH_) != 0) continue;
                        const uint32_t hin = static_cast<uint32_t>(hin_num / static_cast<int32_t>(sH_));
                        if (hin >= H_) continue;

                        for (uint32_t kw = 0; kw < kW_; ++kw) {
                            // deconv mapping for width: w_out = sW * win + kw - pW
                            // valid w in [w0, w0+len-1] with given residue
                            const int32_t residue = static_cast<int32_t>(kw) - static_cast<int32_t>(pW_);
                            // first w >= w0 with w % sW == residue mod sW
                            int32_t r = residue % static_cast<int32_t>(sW_);
                            if (r < 0) r += static_cast<int32_t>(sW_);
                            int32_t w0_mod = static_cast<int32_t>(w0 % sW_);
                            int32_t delta = r - w0_mod;
                            if (delta < 0) delta += static_cast<int32_t>(sW_);
                            uint32_t w_first = w0 + static_cast<uint32_t>(delta);
                            if (w_first >= w0 + len) continue;

                            // Enforce win >= 0 => w_first >= (kw - pW)
                            uint32_t lower_bound = 0;
                            if (residue > 0) {
                                lower_bound = static_cast<uint32_t>(residue);
                            }
                            if (w_first < lower_bound) {
                                uint32_t adjust = (lower_bound - w_first + (sW_ - 1)) / sW_;
                                w_first += adjust * sW_;
                                if (w_first >= w0 + len) continue;
                            }

                            // Upper bound: win <= W_-1 => w <= (W_-1)*sW_ + residue
                            uint64_t upper_valid_w = (uint64_t)(W_ - 1) * sW_ + (uint64_t)residue;
                            uint32_t w_end = w0 + len - 1;
                            if ((uint64_t)w_end > upper_valid_w) {
                                if (upper_valid_w < (uint64_t)w0) continue;
                                w_end = static_cast<uint32_t>(upper_valid_w);
                            }
                            if (w_end < w_first) continue;

                            // Now contiguous input segment from win_start to win_end with step 1, mapped to w_out step sW_
                            const uint32_t win_start = (w_first + pW_ - kw) / sW_;
                            const uint32_t steps = (w_end - w_first) / sW_ + 1;

                            // Copy input segment [win_start, win_start+steps)
                            const uint64_t xBase = (((uint64_t)n * Cin_ + cinGlobal) * H_ + hin) * W_ + win_start;
                            AscendC::LocalTensor<float> xSeg = xQueue_.AllocTensor<float>();
                            AscendC::DataCopy(xSeg, xGm_[xBase], steps);

                            // Fetch weight scalar for (kh,kw)
                            const float wScalar = wLocal[kh * kW_ + kw];

                            // Accumulate: yLocal[(w_first - w0) + m*sW_] += xSeg[m] * wScalar
                            uint32_t yIdx = (w_first - w0);
                            for (uint32_t m = 0; m < steps; ++m) {
                                yLocal[yIdx] = AscendC::Fma(xSeg[m], wScalar, yLocal[yIdx]);
                                yIdx += sW_;
                            }

                            xQueue_.FreeTensor(xSeg);
                        } // kw
                    } // kh

                    wQueue_.FreeTensor(wLocal);
                } // cinPg

                // Write back yLocal to GM: y[n, coGlobal, ho, w0 : w0+len)
                const uint64_t yBase = (((uint64_t)n * Co_ + coGlobal) * Ho_ + ho) * Wo_ + w0;
                AscendC::DataCopy(yGm_[yBase], yLocal, len);

                yQueue_.FreeTensor(yLocal);
            } // width tiles

            line += blockNum;
        } // lines
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  xQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0};
    uint32_t G_{0}, CoutPg_{0}, Co_{0}, CinPg_{0};
    uint32_t kH_{0}, kW_{0}, kArea_{0};
    uint32_t sH_{0}, sW_{0}, pH_{0}, pW_{0}, dH_{0}, dW_{0};
    uint32_t Ho_{0}, Wo_{0}, tileW_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
