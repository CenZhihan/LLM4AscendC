kernel_src="""
#include "kernel_operator.h"
#include <cmath>

class KernelConv2dSubtractTanhSubtractAvgPoolCustom {
public:
    __aicore__ inline KernelConv2dSubtractTanhSubtractAvgPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, const optiling::Conv2dSubtractTanhSubtractAvgPoolCustomTilingData &t)
    {
        x_ = reinterpret_cast<__gm__ float*>(x);
        w_ = reinterpret_cast<__gm__ float*>(w);
        b_ = reinterpret_cast<__gm__ float*>(b);
        y_ = reinterpret_cast<__gm__ float*>(y);

        N_ = t.N;
        Cin_ = t.Cin;
        H_ = t.H;
        W_ = t.W;
        Cout_ = t.Cout;
        Kh_ = t.Kh;
        Kw_ = t.Kw;
        Hout_ = t.Hout;
        Wout_ = t.Wout;
        PH_ = t.PH;
        PW_ = t.PW;
        poolK_ = t.poolK;
        hasBias_ = (t.hasBias != 0);
        totalPooled_ = static_cast<uint64_t>(t.N) * t.Cout * t.PH * t.PW;
        tileLen_ = t.tileLen;

        subtract1_ = t.subtract1;
        subtract2_ = t.subtract2;
    }

    __aicore__ inline void Process()
    {
        if (totalPooled_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * tileLen_;
        uint64_t base = static_cast<uint64_t>(blockIdx) * tileLen_;
        if (base >= totalPooled_) return;

        while (base < totalPooled_) {
            uint32_t curLen = static_cast<uint32_t>(totalPooled_ - base);
            if (curLen > tileLen_) curLen = tileLen_;
            ComputeRange(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeRange(uint64_t start, uint32_t len)
    {
        // Linear index over [N, Cout, PH, PW]
        for (uint64_t off = 0; off < len; ++off) {
            uint64_t idx = start + off;

            uint64_t tmp = idx;
            const uint32_t PW_ = PW__;
            const uint32_t PH_ = PH__;
            const uint32_t Cout_ = Cout__;

            const uint32_t pw = static_cast<uint32_t>(tmp % PW__);
            tmp /= PW__;
            const uint32_t ph = static_cast<uint32_t>(tmp % PH__);
            tmp /= PH__;
            const uint32_t oc = static_cast<uint32_t>(tmp % Cout__);
            const uint32_t n  = static_cast<uint32_t>(tmp / Cout__);

            // pool window starting points in conv output
            const uint32_t oh0 = ph * poolK_;
            const uint32_t ow0 = pw * poolK_;

            float pooledSum = 0.f;
            for (uint32_t dh = 0; dh < poolK_; ++dh) {
                const uint32_t oh = oh0 + dh;
                for (uint32_t dw = 0; dw < poolK_; ++dw) {
                    const uint32_t ow = ow0 + dw;
                    float acc = hasBias_ ? b_[oc] : 0.f;

                    // Direct conv for this (n, oc, oh, ow)
                    for (uint32_t ic = 0; ic < Cin_; ++ic) {
                        for (uint32_t kh = 0; kh < Kh_; ++kh) {
                            const uint32_t ih = oh + kh;
                            const uint32_t baseX_hw = ((n * Cin_ + ic) * H_ + ih) * W_;
                            const uint32_t baseW_hw = ((oc * Cin_ + ic) * Kh_ + kh) * Kw_;
                            for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                const uint32_t iw = ow + kw;
                                const float xv = x_[baseX_hw + iw];
                                const float wv = w_[baseW_hw + kw];
                                acc += xv * wv;
                            }
                        }
                    }
                    // fused pointwise: subtract1 -> tanh -> subtract2
                    acc = acc - subtract1_;
                    // tanh approximation for device side
                    float e2x = expf(2.0f * acc);
                    float tval = (e2x - 1.0f) / (e2x + 1.0f);
                    acc = tval - subtract2_;
                    pooledSum += acc;
                }
            }
            const float outv = pooledSum / static_cast<float>(poolK_ * poolK_);

            // y index [n, oc, ph, pw] in NCHW with PH, PW as H,W
            const uint64_t yOffset = (((static_cast<uint64_t>(n) * Cout_) + oc) * PH_) * PW_ + static_cast<uint64_t>(ph) * PW_ + pw;
            y_[yOffset] = outv;
        }
    }

private:
    __gm__ float* x_{nullptr};
    __gm__ float* w_{nullptr};
    __gm__ float* b_{nullptr};
    __gm__ float* y_{nullptr};

    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0}, Cout_{0}, Kh_{0}, Kw_{0};
    uint32_t Hout_{0}, Wout_{0}, PH_{0}, PW_{0}, poolK_{1};
    bool hasBias_{false};
    uint64_t totalPooled_{0};
    uint32_t tileLen_{1};

    float subtract1_{0.f};
    float subtract2_{0.f};

    // Shorthand getters to avoid capture issues inside ComputeRange
    __aicore__ inline uint32_t PH__() const { return PH_; }
    __aicore__ inline uint32_t PW__() const { return PW_; }
    __aicore__ inline uint32_t Cout__() const { return Cout_; }
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractTanhSubtractAvgPoolCustom op;
    op.Init(x, w, b, y, tiling_data);
    op.Process();
}
