host_operator_src="""
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core UB budget for our kernel
static constexpr uint32_t ALIGN_ELEMS_M = 128;
static constexpr uint32_t ALIGN_ELEMS_N = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    // Expect 2D inputs
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    const uint32_t Kb = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(0));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(1));

    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    // UB memory planning:
    // We keep:
    //   - C_tile (tileM * tileN)
    //   - B_panel (tileK * tileN)
    //   - aSeg (tileK)
    //   - tmpVec (tileN)
    // floats_budget = UB_BUDGET_BYTES / 4
    uint32_t floats_budget = UB_BUDGET_BYTES / 4;
    // choose tileN small to favor tall-skinny cases
    uint32_t tileN = N;
    if (tileN > 32u) tileN = 32u;
    if (tileN < 8u) tileN = (N == 0 ? 0u : N);
    tileN = AlignDown(tileN, ALIGN_ELEMS_N);
    if (tileN == 0u) tileN = (N > 0 ? (N < ALIGN_ELEMS_N ? N : ALIGN_ELEMS_N) : 0u);

    // choose tileK modest (amortize GM loads while keeping UB low)
    uint32_t tileK = 8u;
    if (K < tileK) tileK = K;

    // compute tileM from budget
    // tileM*tileN + tileK*tileN + tileK + tileN <= floats_budget * 0.95
    uint32_t guard = floats_budget / 20; // 5% guard
    uint32_t usable = (floats_budget > guard) ? (floats_budget - guard) : floats_budget;
    uint32_t denom = (tileN == 0u) ? 1u : tileN;
    uint32_t maxTileM = 0u;
    if (denom > 0) {
        uint32_t rhs = usable - (tileK * tileN + tileK + tileN);
        maxTileM = (rhs > denom) ? (rhs / denom) : 1u;
    }
    if (maxTileM == 0u) maxTileM = 1u;
    uint32_t tileM = maxTileM;
    tileM = AlignDown(tileM, ALIGN_ELEMS_M);
    if (tileM == 0u) {
        tileM = (M >= ALIGN_ELEMS_M ? ALIGN_ELEMS_M : AlignDown(M, 8u));
        if (tileM == 0u) tileM = (M == 0 ? 0u : 1u);
    }
    if (tileM > M) tileM = AlignDown(M, ALIGN_ELEMS_M);
    if (tileM == 0u && M > 0u) tileM = M;

    // Block dim sizing based on number of M-tiles (since N is skinny)
    const uint32_t mTiles = (tileM == 0u) ? 1u : CeilDiv(M, tileM);
    uint32_t blockDim = mTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tM_ = tileM; tN_ = tileN; tK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)a, static_cast<uint64_t>(M_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, static_cast<uint64_t>(K_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, static_cast<uint64_t>(M_) * N_);

        // UB buffers
        // cTile: tM_ * tN_
        // bPanel: tK_ * tN_
        // aSeg: tK_
        // tmpVec: tN_
        pipe_.InitBuffer(cQueue_, 1, tM_ * tN_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, BUFFER_NUM, tK_ * tN_ * sizeof(float));
        pipe_.InitBuffer(aQueue_, BUFFER_NUM, tK_ * sizeof(float));
        pipe_.InitBuffer(vQueue_, 1, tN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t baseM = blockIdx * tM_; baseM < M_; baseM += blockNum * tM_) {
            uint32_t curM = M_ - baseM;
            if (curM > tM_) curM = tM_;

            for (uint32_t n0 = 0; n0 < N_; n0 += tN_) {
                uint32_t curN = N_ - n0;
                if (curN > tN_) curN = tN_;

                AscendC::LocalTensor<float> cTile = cQueue_.AllocTensor<float>();
                // zero initialize C tile (curM x curN region)
                AscendC::Dup(cTile, 0.0f, curM * tN_); // clear rows, only curN will be used on store

                for (uint32_t k0 = 0; k0 < K_; k0 += tK_) {
                    uint32_t kc = K_ - k0;
                    if (kc > tK_) kc = tK_;

                    // Load B panel (kc x curN)
                    AscendC::LocalTensor<float> bPanel = bQueue_.AllocTensor<float>();
                    for (uint32_t kk = 0; kk < kc; ++kk) {
                        uint32_t bRowOffset = (k0 + kk) * N_ + n0;
                        AscendC::DataCopy(bPanel[kk * tN_], bGm_[bRowOffset], curN);
                    }

                    // Temporary vector for accumulating scaled B row
                    AscendC::LocalTensor<float> tmpVec = vQueue_.AllocTensor<float>();

                    // Iterate rows in the M tile
                    for (uint32_t i = 0; i < curM; ++i) {
                        // Load A row segment (kc)
                        AscendC::LocalTensor<float> aSeg = aQueue_.AllocTensor<float>();
                        uint32_t aSegOffset = (baseM + i) * K_ + k0;
                        AscendC::DataCopy(aSeg, aGm_[aSegOffset], kc);

                        // cRow points to row i in cTile
                        AscendC::LocalTensor<float> cRow = cTile[i * tN_];

                        // Accumulate over kc
                        for (uint32_t kk = 0; kk < kc; ++kk) {
                            // tmpVec = bPanel[kk, :] * aSeg[kk]
                            float alpha = aSeg.GetValue(kk);
                            AscendC::DataCopy(tmpVec, bPanel[kk * tN_], curN);
                            AscendC::Muls(tmpVec, tmpVec, alpha, curN);
                            // cRow += tmpVec
                            AscendC::Add(cRow, cRow, tmpVec, curN);
                        }

                        aQueue_.FreeTensor(aSeg);
                    }

                    vQueue_.FreeTensor(tmpVec);
                    bQueue_.FreeTensor(bPanel);
                }

                // Store C tile back (row by row)
                for (uint32_t i = 0; i < curM; ++i) {
                    uint32_t cOff = (baseM + i) * N_ + n0;
                    AscendC::DataCopy(cGm_[cOff], cTile[i * tN_], curN);
                }

                cQueue_.FreeTensor(cTile);
            }
        }
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECCALC, 1> cQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> vQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tM_{0}, tN_{0}, tK_{0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
