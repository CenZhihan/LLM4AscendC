#include "max_pooling_3d_custom_tiling.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
static inline uint32_t FloorDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const gert::Shape origin = in_shape->GetOriginShape();
    const int32_t dims = static_cast<int32_t>(origin.GetDimNum());
    if (dims != 5) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(origin.GetDim(0));
    const uint32_t C = static_cast<uint32_t>(origin.GetDim(1));
    const uint32_t D = static_cast<uint32_t>(origin.GetDim(2));
    const uint32_t H = static_cast<uint32_t>(origin.GetDim(3));
    const uint32_t W = static_cast<uint32_t>(origin.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *k_ptr = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *s_ptr = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *p_ptr = attrs->GetAttrPointer<int64_t>(2);
    const bool *ceil_ptr = attrs->GetAttrPointer<bool>(3);

    const uint32_t K = static_cast<uint32_t>(*k_ptr);
    uint32_t S = static_cast<uint32_t>(*s_ptr);
    if (S == 0u) S = K;
    const uint32_t P = static_cast<uint32_t>(*p_ptr);
    const bool ceilMode = (ceil_ptr != nullptr) ? *ceil_ptr : false;

    auto compute_out = [&](uint32_t in) -> uint32_t {
        if (in == 0u) return 0u;
        const int64_t tmp = static_cast<int64_t>(in) + 2LL * static_cast<int64_t>(P) - static_cast<int64_t>(K);
        if (tmp < 0) return 0u;
        if (ceilMode) {
            return static_cast<uint32_t>(CeilDiv(static_cast<uint32_t>(tmp), S) + 1u);
        } else {
            return static_cast<uint32_t>(FloorDiv(static_cast<uint32_t>(tmp), S) + 1u);
        }
    };

    const uint32_t outD = compute_out(D);
    const uint32_t outH = compute_out(H);
    const uint32_t outW = compute_out(W);

    const uint64_t totalIn = static_cast<uint64_t>(N) * C * D * H * W;
    const uint64_t totalOut = static_cast<uint64_t>(N) * C * outD * outH * outW;

    // Choose a moderate tile length for output elements; minimal UB usage per element.
    static constexpr uint32_t ALIGN_ELEMS = 256;
    static constexpr uint32_t HARD_CAP_ELEMS = 16384;
    uint32_t tileLen = HARD_CAP_ELEMS;
    if (totalOut < tileLen) {
        uint32_t aligned = AlignDown(static_cast<uint32_t>(totalOut), ALIGN_ELEMS);
        tileLen = (aligned > 0u) ? aligned : static_cast<uint32_t>(totalOut);
    } else {
        tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    }
    if (tileLen == 0u) tileLen = 1u;

    uint32_t numTiles = static_cast<uint32_t>(CeilDiv(static_cast<uint32_t>((totalOut > 0) ? ((totalOut > UINT32_MAX) ? UINT32_MAX : static_cast<uint32_t>(totalOut)) : 0u), tileLen));
    if (totalOut > 0 && numTiles == 0u) numTiles = 1u;

    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_D(D);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_K(K);
    tiling.set_S(S);
    tiling.set_P(P);
    tiling.set_ceilMode(ceilMode ? 1u : 0u);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_totalIn(totalIn);
    tiling.set_totalOut(totalOut);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr float NEG_INF = -3.402823466e+38F;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t D, uint32_t H, uint32_t W,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint32_t K, uint32_t S, uint32_t P,
                                uint32_t tileLen, uint64_t totalIn, uint64_t totalOut)
    {
        N_ = N; C_ = C; D_ = D; H_ = H; W_ = W;
        outD_ = outD; outH_ = outH; outW_ = outW;
        K_ = K; S_ = S; P_ = P;
        tileLen_ = tileLen;
        totalIn_ = totalIn;
        totalOut_ = totalOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalIn_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut_);

        // Buffers for one-element temporary values and accumulators
        pipe_.InitBuffer(tmpQueue_, 4, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * static_cast<uint64_t>(tileLen_);
        uint64_t base = static_cast<uint64_t>(blockIdx) * static_cast<uint64_t>(tileLen_);
        while (base < totalOut_) {
            uint32_t curLen = static_cast<uint32_t>(totalOut_ - base);
            if (curLen > tileLen_) curLen = tileLen_;
            ComputeTile(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint64_t baseOut, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            const uint64_t outIndex = baseOut + i;

            // Decode linear index to (n, c, od, oh, ow)
            uint64_t t = outIndex;
            const uint32_t ow = static_cast<uint32_t>(t % outW_); t /= outW_;
            const uint32_t oh = static_cast<uint32_t>(t % outH_); t /= outH_;
            const uint32_t od = static_cast<uint32_t>(t % outD_); t /= outD_;
            const uint32_t c  = static_cast<uint32_t>(t % C_);    t /= C_;
            const uint32_t n  = static_cast<uint32_t>(t);

            const int32_t id0 = static_cast<int32_t>(od) * static_cast<int32_t>(S_) - static_cast<int32_t>(P_);
            const int32_t ih0 = static_cast<int32_t>(oh) * static_cast<int32_t>(S_) - static_cast<int32_t>(P_);
            const int32_t iw0 = static_cast<int32_t>(ow) * static_cast<int32_t>(S_) - static_cast<int32_t>(P_);

            AscendC::LocalTensor<float> acc = tmpQueue_.AllocTensor<float>();
            AscendC::Duplicate(acc, NEG_INF, 1);

            for (uint32_t kd = 0; kd < K_; ++kd) {
                const int32_t id = id0 + static_cast<int32_t>(kd);
                if (id < 0 || id >= static_cast<int32_t>(D_)) continue;
                for (uint32_t kh = 0; kh < K_; ++kh) {
                    const int32_t ih = ih0 + static_cast<int32_t>(kh);
                    if (ih < 0 || ih >= static_cast<int32_t>(H_)) continue;
                    for (uint32_t kw = 0; kw < K_; ++kw) {
                        const int32_t iw = iw0 + static_cast<int32_t>(kw);
                        if (iw < 0 || iw >= static_cast<int32_t>(W_)) continue;

                        const uint64_t inIndex =
                            (((static_cast<uint64_t>(n) * C_ + c) * D_ + static_cast<uint32_t>(id)) * H_ + static_cast<uint32_t>(ih)) * W_ + static_cast<uint32_t>(iw);

                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, xGm_[inIndex], 1);
                        AscendC::Max(acc, acc, tmp, 1);
                        tmpQueue_.FreeTensor(tmp);
                    }
                }
            }

            AscendC::DataCopy(yGm_[outIndex], acc, 1);
            tmpQueue_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, D_{0}, H_{0}, W_{0};
    uint32_t outD_{0}, outH_{0}, outW_{0};
    uint32_t K_{1}, S_{1}, P_{0};
    uint32_t tileLen_{1};
    uint64_t totalIn_{0}, totalOut_{0};
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.K, tiling_data.S, tiling_data.P,
            tiling_data.tileLength, tiling_data.totalIn, tiling_data.totalOut);
    op.Process();
}
