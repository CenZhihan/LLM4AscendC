    }

    const uint32_t C    = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H    = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W    = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cw   = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t K    = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw   = static_cast<uint32_t>(w_shape->GetDim(3));

    if (C != Cw || K != Kw) {
        return ge::GRAPH_FAILED;
    }

    // stride=1, padding=0, dilation=1
    if (H < K || W < K) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t Hout = H - K + 1;
    const uint32_t Wout = W - K + 1;
    const uint32_t CKK  = C * K * K;

    // Determine if optional conv_bias is provided
    bool hasBias = false;
    if (context->GetOptionalInputTensorNum() >= 1) {
        const gert::Tensor* biasTensor = context->GetOptionalInputTensor(2);
        hasBias = (biasTensor != nullptr);
    }

    // Choose blockDim along (N * Hout) dimension
    const uint32_t rowTasks = N * Hout;
    uint32_t blockDim = rowTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Cout(Cout);
    tiling.set_K(K);
    tiling.set_Hout(Hout);
    tiling.set_Wout(Wout);
    tiling.set_CKK(CKK);
    tiling.set_blockDim(blockDim);
    tiling.set_hasConvBias(hasBias ? 1u : 0u);

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dReluBiasAddCustom {
public:
    __aicore__ inline KernelConv2dReluBiasAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR cb, GM_ADDR pb, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t K, uint32_t Hout, uint32_t Wout,
                                uint32_t CKK, uint32_t hasConvBias)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        Cout_ = Cout; K_ = K; Hout_ = Hout; Wout_ = Wout; CKK_ = CKK;
        hasConvBias_ = (hasConvBias != 0);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(N_ * C_ * H_ * W_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(Cout_ * CKK_));
        if (hasConvBias_) {
            cbGm_.SetGlobalBuffer((__gm__ float*)cb, Cout_);
        }
        pbGm_.SetGlobalBuffer((__gm__ float*)pb, Cout_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(N_ * Cout_ * Hout_ * Wout_));

        // UB buffers
        pipe_.InitBuffer(lineQueue_, 2, Wout_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1,  Wout_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1,  Wout_ * sizeof(float));
        pipe_.InitBuffer(wvecQueue_, 1, CKK_  * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || Hout_ == 0 || Wout_ == 0 || Cout_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalRows = N_ * Hout_;
        for (uint32_t row = blockIdx; row < totalRows; row += blockNum) {
            uint32_t n  = row / Hout_;
            uint32_t oh = row % Hout_;

            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                // acc = conv_bias[oc] or 0
                AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();
                float initVal = 0.0f;
                if (hasConvBias_) {
                    AscendC::LocalTensor<float> cbLocal = scalarQueue_.AllocTensor<float>();
                    AscendC::DataCopy(cbLocal, cbGm_[oc], 1);
                    initVal = cbLocal.GetValue(0);
                    scalarQueue_.FreeTensor(cbLocal);
                }
                AscendC::Dup(acc, initVal, Wout_);

                // Load weights for this oc: contiguous CKK_
                AscendC::LocalTensor<float> wvec = wvecQueue_.AllocTensor<float>();
                const uint32_t wBase = oc * CKK_;
                AscendC::DataCopy(wvec, wGm_[wBase], CKK_);

                uint32_t wIdx = 0;
                for (uint32_t ic = 0; ic < C_; ++ic) {
                    for (uint32_t kh = 0; kh < K_; ++kh) {
                        const uint32_t ih = oh + kh;
                        for (uint32_t kw = 0; kw < K_; ++kw) {
                            // Load input line x[n, ic, ih, kw:kw+Wout_-1]
                            const uint32_t iw = kw;
                            const uint32_t xOffset =
                                (((n * C_ + ic) * H_ + ih) * W_) + iw;
                            AscendC::LocalTensor<float> line = lineQueue_.AllocTensor<float>();
                            AscendC::DataCopy(line, xGm_[xOffset], Wout_);

                            // Multiply-accumulate acc += line * w
                            const float wval = wvec.GetValue(wIdx);
                            ++wIdx;

                            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                            AscendC::Muls(tmp, line, wval, Wout_);
                            AscendC::Add(acc, acc, tmp, Wout_);
                            tmpQueue_.FreeTensor(tmp);

                            lineQueue_.FreeTensor(line);
                        }
                    }
                }

                // ReLU
                AscendC::Mins(acc, acc, 0.0f, Wout_);
                // y = acc - min(acc,0) => Max(acc,0)
                AscendC::Sub(acc, acc, acc, Wout_); // acc = acc - min(acc,0)

                // Add post bias scalar
                AscendC::LocalTensor<float> pbLocal = scalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(pbLocal, pbGm_[oc], 1);
                const float pbVal = pbLocal.GetValue(0);
                scalarQueue_.FreeTensor(pbLocal);
                AscendC::Adds(acc, acc, pbVal, Wout_);

                // Store y[n, oc, oh, 0:Wout_]
                const uint32_t yOffset =
                    (((n * Cout_ + oc) * Hout_ + oh) * Wout_);
                AscendC::DataCopy(yGm_[yOffset], acc, Wout_);
                accQueue_.FreeTensor(acc);

                wvecQueue_.FreeTensor(wvec);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> lineQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wvecQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> cbGm_;
    AscendC::GlobalTensor<float> pbGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, K_{0}, Hout_{0}, Wout_{0}, CKK_{0};
    bool hasConvBias_{false};
};

// Kernel entry
extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR convBias,
                                                                  GM_ADDR postBias, GM_ADDR y,
                                                                  GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluBiasAddCustom op;
    op.Init(x, weight, convBias, postBias, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.Cout, tiling_data.K, tiling_data.Hout, tiling_data.Wout,
            tiling_data.CKK, tiling_data.hasConvBias);
    op.Process();
}

#include <torch/library.h>

static inline at::Tensor empty_out_for_conv2d(const at::Tensor& x, const at::Tensor& w)
{
    TORCH_CHECK(x.dim() == 4, "x must be NCHW");
    TORCH_CHECK(w.dim() == 4, "weight must be OIHW");

    const int64_t N  = x.size(0);
    const int64_t C  = x.size(1);
    const int64_t H  = x.size(2);
    const int64_t W  = x.size(3);
    const int64_t Oc = w.size(0);
    const int64_t Ci = w.size(1);
    const int64_t K  = w.size(2);
    const int64_t Kw = w.size(3);
    TORCH_CHECK(C == Ci, "in_channels mismatch");
    TORCH_CHECK(K == Kw, "kernel must be square");
    TORCH_CHECK(H >= K && W >= K, "kernel larger than input with zero padding");

    const int64_t Hout = H - K + 1;
    const int64_t Wout = W - K + 1;
    return at::empty({N, Oc, Hout, Wout}, x.options());
}

at::Tensor conv2d_relu_bias_add_impl_npu(const at::Tensor& x,
                                         const at::Tensor& weight,
                                         const at::Tensor& conv_bias_opt,
                                         const at::Tensor& post_bias)

    // conv_bias_opt may be undefined (passed from Python as None)
    EXEC_NPU_CMD(aclnnConv2dReluBiasAddCustom, x, weight, conv_bias_opt, post_bias, y);
    return y;
}
