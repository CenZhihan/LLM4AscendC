static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MIN_TILEW = 128;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect shapes:
    // x: [N, C, H, W], w: [C, 1, K, 1]
    const uint32_t dim_x = static_cast<uint32_t>(x_shape->GetOriginShape().GetDimNum());
    const uint32_t dim_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDimNum());
    if (dim_x != 4 || dim_w != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t C = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const uint32_t Cw = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    if (Cw != C) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Hout = (H >= Kw) ? (H - Kw + 1u) : 0u;

    // UB plan per inner tile:
    //   xVec: tileW floats
    //   yVec: tileW floats
    //   wLocal: Kw floats (loaded once per (n,c,ho) row)
    // Total floats ~= 2 * tileW + Kw
    uint32_t maxTileW = 0;
    if (Kw < (UB_BUDGET_BYTES / 4)) {
        const uint32_t budgetFloats = (UB_BUDGET_BYTES / 4);
        const uint32_t availFloats = (budgetFloats > Kw) ? (budgetFloats - Kw) : MIN_TILEW;
        maxTileW = AlignDown(availFloats / 2u, ALIGN_ELEMS);
    }
    if (maxTileW < MIN_TILEW) maxTileW = MIN_TILEW;
    if (maxTileW > W) {
        // prefer using whole width when smaller than max tile
        uint32_t alignedW = AlignDown(W, ALIGN_ELEMS);
        maxTileW = (alignedW == 0) ? W : alignedW;
    }

    const uint32_t totRows = (Hout == 0u) ? 1u : (N * C * Hout);
    uint32_t blockDim = totRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_kH(Kw);
    tiling.set_hOut(Hout);
    tiling.set_tileW(maxTileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t wdim,
                                uint32_t kH, uint32_t hOut, uint32_t tileW)
    {
        N_ = n; C_ = c; H_ = h; W_ = wdim;
        K_ = kH; Hout_ = hOut; tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(N_ * C_ * H_ * W_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(C_ * K_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(N_ * C_ * Hout_ * W_));

        // Buffers:
        // - inQueueX_: width tile sized
        // - outQueueY_: width tile sized
        // - tmpQueue_: used for weights (K_) and temporary vec reuse
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        // allocate enough to hold max(K_, tileW_)
        uint32_t tmpCap = (K_ > tileW_) ? K_ : tileW_;
        pipe_.InitBuffer(tmpQueue_, 1, tmpCap * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (Hout_ == 0 || N_ == 0 || C_ == 0 || W_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totRows = N_ * C_ * Hout_;
        // grid-stride over (n, c, ho)
        for (uint32_t rowIdx = blockIdx; rowIdx < totRows; rowIdx += blockNum) {
            // decode rowIdx -> (n, c, ho)
            const uint32_t NC = C_ * Hout_;
            const uint32_t n = rowIdx / NC;
            const uint32_t rem = rowIdx % NC;
            const uint32_t c = rem / Hout_;
            const uint32_t ho = rem % Hout_;

            // Load weights for this channel c: w[c, 0, :, 0] -> length K_
            AscendC::LocalTensor<float> wLocal = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[c * K_], K_);

            // Iterate across width in tiles
            for (uint32_t col = 0; col < W_; col += tileW_) {
                uint32_t curW = W_ - col;
                if (curW > tileW_) curW = tileW_;

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, curW);

                // Accumulate over kernel height
                for (uint32_t kh = 0; kh < K_; ++kh) {
                    // Load input row slice x[n,c,ho+kh, col : col+curW]
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    const uint32_t inRow = ho + kh;
                    const uint32_t inBase = (((n * C_ + c) * H_ + inRow) * W_) + col;
                    AscendC::DataCopy(xLocal, xGm_[inBase], curW);

                    // Read weight scalar w[c, kh]
                    float wval = wLocal.GetValue(kh);

                    // yLocal += xLocal * wval
                    AscendC::Muls(xLocal, xLocal, wval, curW);
                    AscendC::Add(yLocal, yLocal, xLocal, curW);

                    inQueueX_.FreeTensor(xLocal);
                }

                // Store output row slice y[n,c,ho, col : col+curW]
                const uint32_t outBase = (((n * C_ + c) * Hout_ + ho) * W_) + col;
                AscendC::DataCopy(yGm_[outBase], yLocal, curW);
                outQueueY_.FreeTensor(yLocal);
            }

            tmpQueue_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t K_{0}, Hout_{0}, tileW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, w, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kH, tiling_data.hOut, tiling_data.tileW);
    op.Process();
}
