namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// We only zero-fill output in device kernel; UB holds a single output tile.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 1u << 20;     // safety cap

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed3dAsymmetricInputSquareKernelCustomTilingData tiling;

    // Shapes
    const gert::Shape *x_shape = context->GetInputShape(0)->GetOriginShapePtr();
    const gert::Shape *w_shape = context->GetInputShape(1)->GetOriginShapePtr();

    // Expect shapes:
    // x: [N, Cin, D, H, W]
    // w: [Cin, Cout, kD, kH, kW]  (PyTorch conv_transpose3d convention)
    const int64_t N   = (x_shape->GetDimNum() >= 1) ? x_shape->GetDim(0) : 0;
    const int64_t Cin = (x_shape->GetDimNum() >= 2) ? x_shape->GetDim(1) : 0;
    const int64_t Di  = (x_shape->GetDimNum() >= 3) ? x_shape->GetDim(2) : 0;
    const int64_t Hi  = (x_shape->GetDimNum() >= 4) ? x_shape->GetDim(3) : 0;
    const int64_t Wi  = (x_shape->GetDimNum() >= 5) ? x_shape->GetDim(4) : 0;

    const int64_t kCin = (w_shape->GetDimNum() >= 1) ? w_shape->GetDim(0) : 0;
    const int64_t Cout = (w_shape->GetDimNum() >= 2) ? w_shape->GetDim(1) : 0;
    const int64_t kD   = (w_shape->GetDimNum() >= 3) ? w_shape->GetDim(2) : 0;
    const int64_t kH   = (w_shape->GetDimNum() >= 4) ? w_shape->GetDim(3) : 0;
    const int64_t kW   = (w_shape->GetDimNum() >= 5) ? w_shape->GetDim(4) : 0;

    // Fixed parameters per provided model
    const int64_t groups = 1;
    const int64_t strideD = 1, strideH = 1, strideW = 1;
    const int64_t padD = 0, padH = 0, padW = 0;
    const int64_t dilD = 1, dilH = 1, dilW = 1;
    const int64_t outPadD = 0, outPadH = 0, outPadW = 0;

    // Basic validation (best-effort)
    if (Cin != kCin || N <= 0 || Cin <= 0 || Di <= 0 || Hi <= 0 || Wi <= 0 ||
        kD <= 0 || kH <= 0 || kW <= 0) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const int64_t Co = Cout * groups;
    const int64_t Do = (Di - 1) * strideD - 2 * padD + dilD * (kD - 1) + outPadD + 1;
    const int64_t Ho = (Hi - 1) * strideH - 2 * padH + dilH * (kH - 1) + outPadH + 1;
    const int64_t Wo = (Wi - 1) * strideW - 2 * padW + dilW * (kW - 1) + outPadW + 1;

    const uint64_t totalOutElems64 =
        static_cast<uint64_t>(N) * static_cast<uint64_t>(Co) *
        static_cast<uint64_t>(Do) * static_cast<uint64_t>(Ho) * static_cast<uint64_t>(Wo);
    if (totalOutElems64 > 0xFFFFFFFFull) {
        return ge::GRAPH_PARAM_INVALID;
    }
    const uint32_t totalOutElems = static_cast<uint32_t>(totalOutElems64);

    // Tile length from UB budget: bytes = 4 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 4;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOutElems < tileLen) {
        const uint32_t aligned = AlignDown(totalOutElems, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutElems;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalOutLength(totalOutElems);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalOutLength, uint32_t tileLength)
    {
        totalOutLen_ = totalOutLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, 0);       // not used
        wGm_.SetGlobalBuffer((__gm__ float*)weight, 0);  // not used
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);

        pipe_.InitBuffer(zeroQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t base = blockIdx * tileLen_;
        const uint32_t strideElems = blockNum * tileLen_;
        if (base >= totalOutLen_) return;

        // Prepare first zero tile
        uint32_t curLen = totalOutLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        EnqueueZeros(curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalOutLen_) {
                nextLen = totalOutLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                EnqueueZeros(nextLen);
            }

            // Write zeros to y
            WriteTile(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void EnqueueZeros(uint32_t len)
    {
        AscendC::LocalTensor<float> zLocal = zeroQueue_.AllocTensor<float>();
        AscendC::Duplicate(zLocal, 0.0f, len);
        zeroQueue_.EnQue(zLocal);
    }

    __aicore__ inline void WriteTile(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> zLocal = zeroQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], zLocal, len);
        zeroQueue_.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> zeroQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dAsymmetricInputSquareKernelCustom op;
    op.Init(x, weight, y, tiling_data.totalOutLength, tiling_data.tileLength);
    op.Process();
}
