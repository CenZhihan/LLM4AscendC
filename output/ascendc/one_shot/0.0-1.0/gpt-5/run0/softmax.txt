host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
// inQueue + outQueue per tile => 2 * tileLen floats -> 8 * tileLen bytes (+ small overhead)
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const gert::Shape &origin = shape->GetOriginShape();

    const int64_t rank = origin.GetDimNum();
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *axisPtr = attrs->GetAttrPointer<int32_t>(0);

    int32_t axis = (axisPtr != nullptr) ? *axisPtr : -1;
    int32_t ax = axis;
    if (ax < 0) ax += static_cast<int32_t>(rank);
    if (ax < 0) ax = 0;
    if (ax >= rank) ax = static_cast<int32_t>(rank) - 1;

    // Compute outer, axisDim, inner
    uint64_t outer = 1;
    for (int64_t i = 0; i < ax; ++i) {
        outer *= static_cast<uint64_t>(origin.GetDim(i));
    }
    uint64_t axisDim = static_cast<uint64_t>(origin.GetDim(ax));
    uint64_t inner = 1;
    for (int64_t i = ax + 1; i < rank; ++i) {
        inner *= static_cast<uint64_t>(origin.GetDim(i));
    }

    // For this kernel we expect softmax along the last dimension for contiguous rows.
    // If not, we collapse inner into outer so that rows remain contiguous when axis is last.
    // Users should prefer axis == last_dim for best performance.
    if (inner != 1) {
        // Collapse to mimic last-dim behavior when tensor is flattened with inner.
        outer = outer * inner;
        inner = 1;
    }

    const uint32_t rows = static_cast<uint32_t>(outer);
    const uint32_t rowLength = static_cast<uint32_t>(axisDim);

    // tileLen from UB budget: bytes ~ 8 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = (rowLength >= maxTile) ? maxTile : AlignDown(rowLength, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = rowLength;  // small rows

    const uint32_t numTilesPerRow = (tileLen == 0) ? 1u : CeilDiv(rowLength, tileLen);

    // blockDim: parallelize over rows
    uint32_t blockDim = rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(rows);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_axis(ax);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

            .ParamType(REQUIRED)

            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("axis").AttrType(OPTIONAL).Int(-1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }

OP_ADD(SoftmaxCustom);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t rowLength,
                                uint32_t tileLength, int32_t axis)
    {
        rows_ = totalRows;
        cols_ = rowLength;
        tileLen_ = tileLength;
        axis_ = axis;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(rows_) * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(rows_) * cols_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block works on multiple rows in a grid-stride loop.
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            const uint64_t rowBase = static_cast<uint64_t>(row) * cols_;

            // Pass 1: reduce max over the row
            float rowMax = -1.0e30f;
            for (uint32_t base = 0; base < cols_; base += tileLen_) {
                const uint32_t cur = MinU32(cols_ - base, tileLen_);
                AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + base], cur);
                inQueue_.EnQue(xLocal);

                AscendC::LocalTensor<float> xTile = inQueue_.DeQue<float>();
                AscendC::LocalTensor<float> red = scalarQueue_.AllocTensor<float>();
                AscendC::ReduceMax(red, xTile, cur);
                const float tileMax = red.GetValue(0);

                if (tileMax > rowMax) rowMax = tileMax;

                scalarQueue_.FreeTensor(red);
                inQueue_.FreeTensor(xTile);
            }

            // Pass 2: compute exp(x - rowMax), accumulate sum, and store to y
            float rowSum = 0.0f;
            for (uint32_t base = 0; base < cols_; base += tileLen_) {
                const uint32_t cur = MinU32(cols_ - base, tileLen_);
                AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[rowBase + base], cur);
                inQueue_.EnQue(xLocal);

                AscendC::LocalTensor<float> xTile = inQueue_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

                // y = exp(x - rowMax)
                AscendC::Adds(yLocal, xTile, -rowMax, cur);
                AscendC::Exp(yLocal, yLocal, cur);

                AscendC::LocalTensor<float> red = scalarQueue_.AllocTensor<float>();
                AscendC::ReduceSum(red, yLocal, cur);
                rowSum += red.GetValue(0);
                scalarQueue_.FreeTensor(red);

                outQueue_.EnQue(yLocal);
                inQueue_.FreeTensor(xTile);

                AscendC::LocalTensor<float> yTile = outQueue_.DeQue<float>();
                AscendC::DataCopy(yGm_[rowBase + base], yTile, cur);
                outQueue_.FreeTensor(yTile);
            }

            // Pass 3: normalize stored exp values by rowSum
            const float invSum = 1.0f / rowSum;
            for (uint32_t base = 0; base < cols_; base += tileLen_) {
                const uint32_t cur = MinU32(cols_ - base, tileLen_);
                AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, yGm_[rowBase + base], cur);
                outQueue_.EnQue(yLocal);

                AscendC::LocalTensor<float> yTile = outQueue_.DeQue<float>();
                AscendC::Muls(yTile, yTile, invSum, cur);
                AscendC::DataCopy(yGm_[rowBase + base], yTile, cur);
                outQueue_.FreeTensor(yTile);
            }
        }
    }

private:
    __aicore__ inline uint32_t MinU32(uint32_t a, uint32_t b) { return (a < b) ? a : b; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t tileLen_ {0};
    int32_t axis_ { -1 };
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.totalRows, tiling_data.rowLength,
            tiling_data.tileLength, tiling_data.axis);
    op.Process();
}
"""
