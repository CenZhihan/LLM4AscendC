host_operator_src="""

#include "matmul_for_lower_triangular_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB budget (conservative)
static constexpr uint32_t ALIGN = 16;                    // favor vector alignment
static constexpr uint32_t MIN_TILE = 16;
static constexpr uint32_t MAX_TILE = 128;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForLowerTriangularMatricesCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    // Expect A is [N, N]
    const uint32_t N = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));

    // Choose base tile within UB budget for three tiles (A: MxK, B: KxN, C: MxN)
    // Memory = 4 * (M*K + K*N + M*N)
    uint32_t base = 64;
    auto mem_cost = [&](uint32_t m, uint32_t n, uint32_t k) -> uint32_t {
        uint64_t elems = static_cast<uint64_t>(m) * k + static_cast<uint64_t>(k) * n + static_cast<uint64_t>(m) * n;
        return static_cast<uint32_t>(elems * 4u);
    };

    base = AlignDown(base, ALIGN);
    if (base < MIN_TILE) base = MIN_TILE;
    if (base > MAX_TILE) base = MAX_TILE;

    // Try to find a suitable base that fits into UB
    uint32_t tileM = base, tileN = base, tileK = base;
    while (mem_cost(tileM, tileN, tileK) > UB_BUDGET_BYTES) {
        if (tileK > MIN_TILE) tileK = AlignDown(tileK / 2u, ALIGN);
        else if (tileN > MIN_TILE) tileN = AlignDown(tileN / 2u, ALIGN);
        else if (tileM > MIN_TILE) tileM = AlignDown(tileM / 2u, ALIGN);
        else break;
        if (tileM < MIN_TILE) tileM = MIN_TILE;
        if (tileN < MIN_TILE) tileN = MIN_TILE;
        if (tileK < MIN_TILE) tileK = MIN_TILE;
    }

    // Clamp tiles to problem size
    if (N < tileM) tileM = AlignDown(N, ALIGN) ? AlignDown(N, ALIGN) : N;
    if (N < tileN) tileN = AlignDown(N, ALIGN) ? AlignDown(N, ALIGN) : N;
    if (N < tileK) tileK = AlignDown(N, ALIGN) ? AlignDown(N, ALIGN) : N;

    // Compute blockDim based on lower-triangular tiles
    const uint32_t numTileRows = CeilDiv(N, tileM);
    const uint32_t numTileCols = CeilDiv(N, tileN);
    const uint64_t approxLowerTiles = static_cast<uint64_t>(numTileRows) * numTileCols; // upper bound
    uint32_t blockDim = static_cast<uint32_t>(approxLowerTiles);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulForLowerTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForLowerTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t N, uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        const uint32_t totalElems = N_ * N_;

        aGm_.SetGlobalBuffer((__gm__ float*)a, totalElems);
        bGm_.SetGlobalBuffer((__gm__ float*)b, totalElems);
        cGm_.SetGlobalBuffer((__gm__ float*)c, totalElems);

        pipe_.InitBuffer(aQueue_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(cQueue_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numTileRows = (N_ + tileM_ - 1u) / tileM_;
        const uint32_t numTileCols = (N_ + tileN_ - 1u) / tileN_;
        const uint32_t totalTiles = numTileRows * numTileCols;

        for (uint32_t linear = blockIdx; linear < totalTiles; linear += blockNum) {
            const uint32_t ti = linear / numTileCols;
            const uint32_t tj = linear % numTileCols;

            // Skip upper-triangular tiles (only compute ti >= tj)
            if (tj > ti) continue;

            const uint32_t i0 = ti * tileM_;
            const uint32_t j0 = tj * tileN_;
            if (i0 >= N_ || j0 >= N_) continue;

            const uint32_t mLen = MinU32(tileM_, N_ - i0);
            const uint32_t nLen = MinU32(tileN_, N_ - j0);

            // Accumulator tile
            AscendC::LocalTensor<float> cLocal = cQueue_.AllocTensor<float>();
            AscendC::Dup(cLocal, 0.0f, mLen * nLen);

            // Valid K range for lower-triangular product: k in [j0, i0 + mLen - 1]
            uint32_t kBegin = j0;
            uint32_t kEnd = i0 + mLen;
            if (kBegin > N_) kBegin = N_;
            if (kEnd > N_) kEnd = N_;
            if (kBegin >= kEnd) {
                // Nothing contributes; still need to zero upper half within tile
                MaskUpper(cLocal, i0, j0, mLen, nLen);
                StoreTile(cLocal, i0, j0, mLen, nLen);
                cQueue_.FreeTensor(cLocal);
                continue;
            }

            for (uint32_t k0 = kBegin; k0 < kEnd; k0 += tileK_) {
                const uint32_t kLen = MinU32(tileK_, kEnd - k0);

                // Copy A_sub: rows [i0, i0+mLen), cols [k0, k0+kLen)
                AscendC::LocalTensor<float> aLocal = aQueue_.AllocTensor<float>();
                for (uint32_t r = 0; r < mLen; ++r) {
                    const uint32_t a_gm_off = (i0 + r) * N_ + k0;
                    const uint32_t a_loc_off = r * tileK_;
                    AscendC::DataCopy(aLocal[a_loc_off], aGm_[a_gm_off], kLen);
                }
                aQueue_.EnQue(aLocal);

                // Copy B_sub: rows [k0, k0+kLen), cols [j0, j0+nLen)
                AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();
                for (uint32_t r = 0; r < kLen; ++r) {
                    const uint32_t b_gm_off = (k0 + r) * N_ + j0;
                    const uint32_t b_loc_off = r * tileN_;
                    AscendC::DataCopy(bLocal[b_loc_off], bGm_[b_gm_off], nLen);
                }
                bQueue_.EnQue(bLocal);

                // Compute C += A_sub * B_sub using outer-product accumulation
                aLocal = aQueue_.DeQue<float>();
                bLocal = bQueue_.DeQue<float>();

                for (uint32_t r = 0; r < mLen; ++r) {
                    // Optional early-stop upper bound for k based on row r: k <= i0 + r
                    uint32_t kMaxForRow = (i0 + r >= k0) ? MinU32(kLen, (i0 + r - k0 + 1)) : 0u;
                    if (kMaxForRow == 0u) continue;

                    for (uint32_t t = 0; t < kMaxForRow; ++t) {
                        const float alpha = aLocal[r * tileK_ + t];

                        // tmp = alpha * bLocal[t, :]
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::Muls(tmp, bLocal[t * tileN_], alpha, nLen);

                        // cLocal[r, :] += tmp
                        AscendC::Add(cLocal[r * tileN_], cLocal[r * tileN_], tmp, nLen);

                        tmpQueue_.FreeTensor(tmp);
                    }
                }

                bQueue_.FreeTensor(bLocal);
                aQueue_.FreeTensor(aLocal);
            }

            // Zero the strictly upper-triangular part within the output tile
            MaskUpper(cLocal, i0, j0, mLen, nLen);

            // Store tile to GM
            StoreTile(cLocal, i0, j0, mLen, nLen);

            cQueue_.FreeTensor(cLocal);
        }
    }

private:
    __aicore__ inline uint32_t MinU32(uint32_t a, uint32_t b) { return a < b ? a : b; }

    __aicore__ inline void MaskUpper(AscendC::LocalTensor<float>& cLocal,
                                     uint32_t i0, uint32_t j0,
                                     uint32_t mLen, uint32_t nLen)
    {
        for (uint32_t r = 0; r < mLen; ++r) {
            const uint32_t globalRow = i0 + r;
            // Max inclusive column allowed: globalCol <= globalRow
            // Allowed count within this tile
            int64_t allowed = static_cast<int64_t>(globalRow) - static_cast<int64_t>(j0) + 1;
            uint32_t keep = 0;
            if (allowed > 0) {
                keep = static_cast<uint32_t>(allowed);
                if (keep > nLen) keep = nLen;
            }
            if (keep < nLen) {
                AscendC::Dup(cLocal[r * tileN_ + keep], 0.0f, nLen - keep);
            }
        }
    }

    __aicore__ inline void StoreTile(AscendC::LocalTensor<float>& cLocal,
                                     uint32_t i0, uint32_t j0,
                                     uint32_t mLen, uint32_t nLen)
    {
        for (uint32_t r = 0; r < mLen; ++r) {
            const uint32_t c_gm_off = (i0 + r) * N_ + j0;
            const uint32_t c_loc_off = r * tileN_;
            AscendC::DataCopy(cGm_[c_gm_off], cLocal[c_loc_off], nLen);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> aQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> cQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t tileM_{0};
    uint32_t tileN_{0};
    uint32_t tileK_{0};
};

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForLowerTriangularMatricesCustom op;
    op.Init(A, B, C,
            tiling_data.N,
            tiling_data.tileM,
            tiling_data.tileN,
            tiling_data.tileK);
    op.Process();
}
