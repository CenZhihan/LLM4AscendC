namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dSquareInputSquareKernelCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    // Expect NCHW for x and OIHW for weight
    const uint32_t n  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t ci = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t h  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t w  = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t co = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t kH = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t kW = static_cast<uint32_t>(w_shape->GetDim(3));
    const uint32_t k  = kH; // square kernel assumed

    // Fixed stride=1, padding=0, dilation=1; groups=1
    const uint32_t hout = (h >= k) ? (h - k + 1) : 0u;
    const uint32_t wout = (w >= k) ? (w - k + 1) : 0u;

    // Tile across width-out per row; full row preferred
    uint32_t tileWout = wout;

    // Choose blockDim to cover (n * co) pairs with grid-stride in kernel
    uint32_t targetBlocks = n * co;
    uint32_t blockDim = MinU32(targetBlocks, 24u);
    if (blockDim == 0u) blockDim = 1u;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_ci(ci);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_co(co);
    tiling.set_k(k);
    tiling.set_hout(hout);
    tiling.set_wout(wout);
    tiling.set_tileWout(tileWout);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t ci, uint32_t h, uint32_t w_in,
                                uint32_t co, uint32_t k, uint32_t hout, uint32_t wout,
                                uint32_t tileWout)
    {
        n_ = n; ci_ = ci; h_ = h; w_ = w_in; co_ = co; k_ = k; hout_ = hout; wout_ = wout; tileWout_ = tileWout;

        const uint64_t xTotal = static_cast<uint64_t>(n_)*ci_*h_*w_;
        const uint64_t yTotal = static_cast<uint64_t>(n_)*co_*hout_*wout_;
        const uint64_t wTotal = static_cast<uint64_t>(co_)*ci_*k_*k_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xTotal);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wTotal);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yTotal);

        pipe_.InitBuffer(inQueueX_, 1, tileWout_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileWout_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQueue_, 1, tileWout_ * sizeof(float));
        pipe_.InitBuffer(wBufQueue_, 1, tileWout_ * sizeof(float)); // reuse; only first k*k used
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || co_ == 0 || ci_ == 0 || h_ == 0 || w_ == 0 || k_ == 0) return;
        if (hout_ == 0 || wout_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t pairsTotal = static_cast<uint64_t>(n_) * co_;
        for (uint64_t pair = blockIdx; pair < pairsTotal; pair += blockNum) {
            const uint32_t nIdx  = static_cast<uint32_t>(pair / co_);
            const uint32_t coIdx = static_cast<uint32_t>(pair % co_);

            // Iterate output rows
            for (uint32_t ho = 0; ho < hout_; ++ho) {
                AscendC::LocalTensor<float> yRow = outQueueY_.AllocTensor<float>();
                AscendC::Dup(yRow, 0.0f, wout_);

                // Accumulate over input channels
                for (uint32_t ciIdx = 0; ciIdx < ci_; ++ciIdx) {
                    // Load weights for this (coIdx, ciIdx) into local buffer (k*k floats)
                    AscendC::LocalTensor<float> wLoc = wBufQueue_.AllocTensor<float>();
                    const uint64_t wBase = (((uint64_t)coIdx * ci_ + ciIdx) * k_) * k_;
                    AscendC::DataCopy(wLoc, wGm_[wBase], k_ * k_);

                    // For each kernel row
                    for (uint32_t ky = 0; ky < k_; ++ky) {
                        const uint32_t in_h = ho + ky;
                        // For each kernel col
                        for (uint32_t kx = 0; kx < k_; ++kx) {
                            // Read weight scalar from local buffer
                            const float wVal = wLoc.GetValue(ky * k_ + kx);

                            // Load a contiguous segment of input row corresponding to this shift: length = wout_
                            AscendC::LocalTensor<float> xSeg = inQueueX_.AllocTensor<float>();
                            const uint64_t xBase = ((((uint64_t)nIdx * ci_) + ciIdx) * h_ + in_h) * w_ + kx;
                            AscendC::DataCopy(xSeg, xGm_[xBase], wout_);

                            // Multiply by scalar weight and accumulate
                            AscendC::LocalTensor<float> tmp = tmpVecQueue_.AllocTensor<float>();
                            AscendC::Muls(tmp, xSeg, wVal, wout_);
                            AscendC::Add(yRow, yRow, tmp, wout_);
                            tmpVecQueue_.FreeTensor(tmp);
                            inQueueX_.FreeTensor(xSeg);
                        }
                    }
                    wBufQueue_.FreeTensor(wLoc);
                }

                // Write output row to GM
                const uint64_t yBase = ((((uint64_t)nIdx * co_) + coIdx) * hout_ + ho) * wout_;
                AscendC::DataCopy(yGm_[yBase], yRow, wout_);
                outQueueY_.FreeTensor(yRow);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>   inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1>  outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVecQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wBufQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, ci_{0}, h_{0}, w_{0};
    uint32_t co_{0}, k_{0}, hout_{0}, wout_{0};
    uint32_t tileWout_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.ci, tiling_data.h, tiling_data.w,
            tiling_data.co, tiling_data.k, tiling_data.hout, tiling_data.wout,
            tiling_data.tileWout);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv2d_standard_square_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    TORCH_CHECK(x.dim() == 4, "x must be 4D NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be 4D OIHW");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "x must be float32");
    TORCH_CHECK(weight.scalar_type() == at::kFloat, "weight must be float32");

    const int64_t N  = x.size(0);
    const int64_t Ci = x.size(1);
    const int64_t H  = x.size(2);
    const int64_t W  = x.size(3);

    const int64_t Co = weight.size(0);
    const int64_t K  = weight.size(2); // assume square

    TORCH_CHECK(weight.size(1) == Ci, "weight Ci must match input channels");
    TORCH_CHECK(weight.size(2) == weight.size(3), "kernel must be square");
    TORCH_CHECK(H >= K && W >= K, "kernel larger than input with zero padding");

    const int64_t Hout = H - K + 1;
    const int64_t Wout = W - K + 1;

    at::Tensor result = at::empty({N, Co, Hout, Wout}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard2dSquareInputSquareKernelCustom, x, weight, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_standard_2d_square_input_square_kernel_custom", &conv2d_standard_square_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_standard_2d_square_input_square_kernel_custom", &conv2d_standard_square_impl_npu,
          "Conv2D (stride=1, pad=0) square input and kernel (ConvStandard2dSquareInputSquareKernelCustom on NPU)");
}
