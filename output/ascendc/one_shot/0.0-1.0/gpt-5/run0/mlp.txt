#include "mlp_custom_tiling.h"

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MlpCustomTilingData tiling;

    // Input 0 is x: [N, input_size]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const int64_t rank = x_shape->GetOriginShape().GetDimNum();
    uint32_t batch = 1;
    if (rank >= 1) {
        batch = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *inputSizePtr   = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *h1Ptr          = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *h2Ptr          = attrs->GetAttrPointer<int64_t>(2);
    const int64_t *outputSizePtr  = attrs->GetAttrPointer<int64_t>(3);

    const uint32_t inSz  = static_cast<uint32_t>(*inputSizePtr);
    const uint32_t h1    = static_cast<uint32_t>(*h1Ptr);
    const uint32_t h2    = static_cast<uint32_t>(*h2Ptr);
    const uint32_t outSz = static_cast<uint32_t>(*outputSizePtr);

    // BlockDim: parallelize across batch dimension conservatively.
    uint32_t blockDim = batch;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batch);
    tiling.set_inputSize(inSz);
    tiling.set_h1(h1);
    tiling.set_h2(h2);
    tiling.set_outputSize(outSz);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // Workspace: per-block hidden buffers for h1 and h2
    size_t *workspaces = context->GetWorkspaceSizes(1);
    const size_t perBlockFloats = static_cast<size_t>(h1) + static_cast<size_t>(h2);
    workspaces[0] = static_cast<size_t>(blockDim) * perBlockFloats * sizeof(float);
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    // Get attrs
    const int64_t output_size = context->GetAttrs()->GetAttr<int64_t>("output_size");
    gert::Shape out_shape = x_shape->GetOriginShape();

    if (out_shape.GetDimNum() == 0) {
        out_shape.SetDimNum(2);
        out_shape.SetDim(0, 1);
        out_shape.SetDim(1, output_size);
    } else if (out_shape.GetDimNum() == 1) {
        out_shape.SetDimNum(2);
        out_shape.SetDim(1, output_size);
    } else {
        out_shape.SetDim(1, output_size);
    }

    y_shape->SetOriginShape(out_shape);
    y_shape->SetStorageShape(out_shape);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

class KernelMlpCustom {
public:
    __aicore__ inline KernelMlpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wPacked, GM_ADDR bPacked, GM_ADDR y,
                                GM_ADDR workspace,
                                uint32_t batchSize, uint32_t inputSize,
                                uint32_t h1, uint32_t h2, uint32_t outputSize)
    {
        batch_ = batchSize;
        in_ = inputSize;
        h1_ = h1;
        h2_ = h2;
        out_ = outputSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)batch_ * in_);
        wGm_.SetGlobalBuffer((__gm__ float*)wPacked, (uint64_t)h1_ * in_ + (uint64_t)h2_ * h1_ + (uint64_t)out_ * h2_);
        bGm_.SetGlobalBuffer((__gm__ float*)bPacked, (uint64_t)h1_ + (uint64_t)h2_ + (uint64_t)out_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)batch_ * out_);

        // Workspace is per-block: [h1 + h2] floats
        wsBase_ = (__gm__ float*)workspace;

        // Allocate queues/buffers sized for worst-case row lengths
        const uint32_t maxVec = Max3(in_, h1_, h2_);
        pipe_.InitBuffer(xBufQ_, 1, in_ * sizeof(float));
        pipe_.InitBuffer(wBufQ_, 1, maxVec * sizeof(float));
        pipe_.InitBuffer(tmpBufQ_, 1, maxVec * sizeof(float));
        pipe_.InitBuffer(outBufQ_, 1, out_ * sizeof(float));
        pipe_.InitBuffer(hidBufQ_, 1, h1_ * sizeof(float)); // reuse for h2 when needed
        pipe_.InitBuffer(biasBufQ_, 1, Max3(h1_, h2_, out_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        // per-block workspace slice
        __gm__ float* hid1GM = wsBase_ + (size_t)blockIdx * (size_t)(h1_ + h2_);
        __gm__ float* hid2GM = hid1GM + (size_t)h1_;

        for (uint32_t n = blockIdx; n < batch_; n += blockNum) {
            // x_row -> xLocal
            AscendC::LocalTensor<float> xLocal = xBufQ_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[ (size_t)n * in_ ], in_);

            // Layer 1: y1 = ReLU(w1 * x + b1)
            ComputeDenseReLU(hid1GM, 0, xLocal, 0, in_, 0, h1_, /*wOffset*/0);

            // Layer 2: y2 = ReLU(w2 * y1 + b2)
            // Load hidden1 into local in chunks to reuse existing function
            ComputeDenseReLU(hid2GM, 0, hid1GM, 0, h1_, h1_, h2_, /*wOffset*/(uint64_t)h1_ * in_);

            // Layer 3: y = w3 * y2 + b3  (no ReLU)
            ComputeDenseLinearToOut(yGm_, (size_t)n * out_, hid2GM, 0, h2_, h2_, out_, /*wOffset*/(uint64_t)h1_ * in_ + (uint64_t)h2_ * h1_);

            xBufQ_.FreeTensor(xLocal);
        }
    }

private:
    // Helper to get max of three
    __aicore__ static inline uint32_t Max3(uint32_t a, uint32_t b, uint32_t c) {
        uint32_t m = (a > b) ? a : b;
        return (m > c) ? m : c;
    }

    // Compute z = ReLU(W * v + b)
    // vBuf can be in GM or Local; we accept as GM pointer when vIsGM=true else use LocalTensor
    __aicore__ inline void ComputeDenseReLU(__gm__ float* outGM, uint32_t outOff,
                                            AscendC::LocalTensor<float>& vLocal, uint32_t vOff, uint32_t vLen,
                                            uint32_t outLen, uint32_t rows,
                                            uint64_t wOffset)
    {
        // Load bias for this layer
        AscendC::LocalTensor<float> bLocal = biasBufQ_.AllocTensor<float>();
        uint64_t bOffset = BiasOffsetForRows(rows);
        AscendC::DataCopy(bLocal, bGm_[bOffset], rows);

        // For each output neuron i in [0, rows)
        for (uint32_t i = 0; i < rows; ++i) {
            // Load weight row i: length vLen
            AscendC::LocalTensor<float> wLocal = wBufQ_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[wOffset + (uint64_t)i * vLen], vLen);

            // Elementwise multiply into tmp and reduce sum to scalar
            AscendC::LocalTensor<float> tmpLocal = tmpBufQ_.AllocTensor<float>();
            AscendC::Mul(tmpLocal, wLocal, vLocal, vLen);

            // Simple reduction loop (fallback)
            float acc = 0.0f;
            for (uint32_t k = 0; k < vLen; ++k) {
                // Assuming direct indexing is supported for LocalTensor in device code
                acc += tmpLocal.GetValue(k);
            }
            acc += bLocal.GetValue(i);
            // ReLU
            if (acc < 0.0f) acc = 0.0f;

            // Store to GM
            outGM[outOff + i] = acc;

            tmpBufQ_.FreeTensor(tmpLocal);
            wBufQ_.FreeTensor(wLocal);
        }

        biasBufQ_.FreeTensor(bLocal);
    }

    // Overload: When input vector is in GM (hidden from previous layer)
    __aicore__ inline void ComputeDenseReLU(__gm__ float* outGM, uint32_t outOff,
                                            __gm__ float* vGM, uint32_t vOff, uint32_t vLen,
                                            uint32_t outLen, uint32_t rows,
                                            uint64_t wOffset)
    {
        AscendC::LocalTensor<float> vLocal = xBufQ_.AllocTensor<float>();
        // Load vGM into local once
        AscendC::DataCopy(vLocal, vGM[vOff], vLen);
        ComputeDenseReLU(outGM, outOff, vLocal, 0, vLen, outLen, rows, wOffset);
        xBufQ_.FreeTensor(vLocal);
    }

    // Compute y = W * v + b, store directly to output GM (no ReLU)
    __aicore__ inline void ComputeDenseLinearToOut(AscendC::GlobalTensor<float>& yGM, uint64_t yRowOff,
                                                   __gm__ float* vGM, uint32_t vOff, uint32_t vLen,
                                                   uint32_t outLen, uint32_t rows,
                                                   uint64_t wOffset)
    {
        // Load bias for this layer
        AscendC::LocalTensor<float> bLocal = biasBufQ_.AllocTensor<float>();
        uint64_t bOffset = BiasOffsetForRows(rows);
        AscendC::DataCopy(bLocal, bGm_[bOffset], rows);

        AscendC::LocalTensor<float> vLocal = xBufQ_.AllocTensor<float>();
        AscendC::DataCopy(vLocal, vGM[vOff], vLen);

        for (uint32_t i = 0; i < rows; ++i) {
            AscendC::LocalTensor<float> wLocal = wBufQ_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[wOffset + (uint64_t)i * vLen], vLen);

            AscendC::LocalTensor<float> tmpLocal = tmpBufQ_.AllocTensor<float>();
            AscendC::Mul(tmpLocal, wLocal, vLocal, vLen);

            float acc = 0.0f;
            for (uint32_t k = 0; k < vLen; ++k) {
                acc += tmpLocal.GetValue(k);
            }
            acc += bLocal.GetValue(i);

            yGM[yRowOff + i] = acc;

            tmpBufQ_.FreeTensor(tmpLocal);
            wBufQ_.FreeTensor(wLocal);
        }

        xBufQ_.FreeTensor(vLocal);
        biasBufQ_.FreeTensor(bLocal);
    }

    // Compute bias offset inside packed b: [b1 | b2 | b3]
    __aicore__ inline uint64_t BiasOffsetForRows(uint32_t rows) const
    {
        if (rows == h1_) return 0ull;
        if (rows == h2_) return (uint64_t)h1_;
        // else rows == out_
        return (uint64_t)h1_ + (uint64_t)h2_;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xBufQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBufQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> hidBufQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasBufQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    __gm__ float* wsBase_ {nullptr};

    uint32_t batch_ {0};
    uint32_t in_ {0};
    uint32_t h1_ {0};
    uint32_t h2_ {0};
    uint32_t out_ {0};
};

extern "C" __global__ __aicore__ void mlp_custom(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMlpCustom op;
    op.Init(x, w_packed, b_packed, y, workspace,
            tiling_data.batchSize, tiling_data.inputSize,
            tiling_data.h1, tiling_data.h2, tiling_data.outputSize);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'mlp_custom'

class ModelNew(torch.nn.Module):
    """
    Fused 3-layer MLP using custom AscendC kernel.
    PyBind: mlp_custom(x, w_packed, b_packed, input_size, layer_sizes, output_size)
    """
    def __init__(self, input_size, layer_sizes, output_size):
        super(ModelNew, self).__init__()
        self.input_size = int(input_size)
        self.layer_sizes = [int(layer_sizes[0]), int(layer_sizes[1])]
        self.output_size = int(output_size)

    def forward(self, x, w_packed, b_packed):
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, w_packed, b_packed, self.input_size, self.layer_sizes, self.output_size)
