host_operator_src="""

#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inRow: 1 * L
//   outRow: 1 * Lout
//   tmpWin: K
// Total floats ~= L + Lout + K => bytes ~= 4*(L + Lout + K)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t MIN_BLOCKS = 8;
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    const size_t dimNum = inShape->GetDimNum();
    if (dimNum != 3) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(inShape->GetDim(0));
    const uint32_t C = static_cast<uint32_t>(inShape->GetDim(1));
    const uint32_t L = static_cast<uint32_t>(inShape->GetDim(2));

    const uint32_t nRows = N * C;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const uint32_t kernelSize = *attrs->GetAttrPointer<uint32_t>(0);
    uint32_t stride = *attrs->GetAttrPointer<uint32_t>(1);
    const uint32_t padding = *attrs->GetAttrPointer<uint32_t>(2);
    const uint32_t returnIndices = *attrs->GetAttrPointer<uint32_t>(3);

    if (stride == 0u) stride = kernelSize;

    const int64_t Lpad = static_cast<int64_t>(L) + 2LL * static_cast<int64_t>(padding);
    const uint32_t outL = static_cast<uint32_t>((Lpad >= static_cast<int64_t>(kernelSize))
        ? ( (Lpad - static_cast<int64_t>(kernelSize)) / static_cast<int64_t>(stride) + 1LL )
        : 0LL);

    // Decide if a full row fits into UB
    uint32_t tileMaxRow = 0;
    const uint64_t bytesNeed = 4ULL * (static_cast<uint64_t>(L) + static_cast<uint64_t>(outL) + static_cast<uint64_t>(kernelSize));
    if (bytesNeed <= UB_BUDGET_BYTES) {
        tileMaxRow = L;
    } else {
        tileMaxRow = 0; // Fallback path will stream from GM without full-row buffering
    }

    // Blocks based on number of rows
    uint32_t blockDim = nRows;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_nRows(nRows);
    tiling.set_rowLength(L);
    tiling.set_outRowLength(outL);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_tileMaxRow(tileMaxRow);
    tiling.set_returnIndices(returnIndices);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const size_t dimNum = x_shape->GetDimNum();
    if (dimNum != 3) {
        return GRAPH_FAILED;
    }
    const int64_t N = x_shape->GetDim(0);
    const int64_t C = x_shape->GetDim(1);
    const int64_t L = x_shape->GetDim(2);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t kernelSize = static_cast<int64_t>(*attrs->GetAttrPointer<uint32_t>(0));
    int64_t stride = static_cast<int64_t>(*attrs->GetAttrPointer<uint32_t>(1));
    const int64_t padding = static_cast<int64_t>(*attrs->GetAttrPointer<uint32_t>(2));
    if (stride == 0) stride = kernelSize;

    const int64_t Lpad = L + 2 * padding;
    const int64_t outL = (Lpad >= kernelSize) ? ((Lpad - kernelSize) / stride + 1) : 0;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, outL);
    return GRAPH_SUCCESS;
}

    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr float NEG_INF_F = -3.402823e+38f;

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t nRows, uint32_t rowLen, uint32_t outRowLen,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding,
                                uint32_t tileMaxRow)
    {
        nRows_ = nRows;
        rowLen_ = rowLen;
        outRowLen_ = outRowLen;
        k_ = kernelSize;
        s_ = (stride == 0u) ? kernelSize : stride;
        p_ = padding;
        tileMaxRow_ = tileMaxRow;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(nRows_ * rowLen_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(nRows_ * outRowLen_));

        const uint32_t inBytes  = rowLen_    * sizeof(float);
        const uint32_t outBytes = outRowLen_ * sizeof(float);
        const uint32_t winBytes = k_         * sizeof(float);

        pipe_.InitBuffer(inQueueRow_, 1, inBytes);
        pipe_.InitBuffer(outQueueRow_, 1, outBytes);
        pipe_.InitBuffer(winQueue_, 1, winBytes);
    }

    __aicore__ inline void Process()
    {
        if (nRows_ == 0 || rowLen_ == 0 || outRowLen_ == 0 || k_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t rid = blockIdx; rid < nRows_; rid += blockNum) {
            const uint32_t xBase = rid * rowLen_;
            const uint32_t yBase = rid * outRowLen_;
            if (tileMaxRow_ >= rowLen_) {
                ComputeRowLocal(xBase, yBase);
            } else {
                ComputeRowGM(xBase, yBase);
            }
        }
    }

private:
    __aicore__ inline void ComputeRowLocal(uint32_t xBase, uint32_t yBase)
    {
        AscendC::LocalTensor<float> xLocal = inQueueRow_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueRow_.AllocTensor<float>();

        AscendC::DataCopy(xLocal, xGm_[xBase], rowLen_);

        // For each output index j
        for (uint32_t j = 0; j < outRowLen_; ++j) {
            // window start in padded coordinates
            int32_t start = static_cast<int32_t>(j) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);
            int32_t end   = start + static_cast<int32_t>(k_);

            // Clamp to [0, rowLen_)
            uint32_t e0 = static_cast<uint32_t>(AscendC::Max(0, start));
            uint32_t e1 = static_cast<uint32_t>(AscendC::Min(static_cast<int32_t>(rowLen_), end));

            float m = NEG_INF_F;
            for (uint32_t t = e0; t < e1; ++t) {
                // Local read
                float v = xLocal.GetValue(t);
                m = (v > m) ? v : m;
            }

            // If window extends outside valid range, include zero from padding
            if (start < 0 || end > static_cast<int32_t>(rowLen_)) {
                m = (0.0f > m) ? 0.0f : m;
            }

            yLocal.SetValue(j, m);
        }

        AscendC::DataCopy(yGm_[yBase], yLocal, outRowLen_);

        outQueueRow_.FreeTensor(yLocal);
        inQueueRow_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeRowGM(uint32_t xBase, uint32_t yBase)
    {
        AscendC::LocalTensor<float> yLocal = outQueueRow_.AllocTensor<float>();
        AscendC::LocalTensor<float> winLocal = winQueue_.AllocTensor<float>();

        // Compute output elements one by one; stream windows from GM
        for (uint32_t j = 0; j < outRowLen_; ++j) {
            int32_t start = static_cast<int32_t>(j) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);
            int32_t end   = start + static_cast<int32_t>(k_);

            uint32_t e0 = static_cast<uint32_t>(AscendC::Max(0, start));
            uint32_t e1 = static_cast<uint32_t>(AscendC::Min(static_cast<int32_t>(rowLen_), end));
            uint32_t span = (e1 > e0) ? (e1 - e0) : 0;

            float m = NEG_INF_F;
            if (span > 0) {
                AscendC::DataCopy(winLocal, xGm_[xBase + e0], span);
                for (uint32_t t = 0; t < span; ++t) {
                    float v = winLocal.GetValue(t);
                    m = (v > m) ? v : m;
                }
            }

            if (start < 0 || end > static_cast<int32_t>(rowLen_)) {
                m = (0.0f > m) ? 0.0f : m;
            }

            yLocal.SetValue(j, m);
        }

        AscendC::DataCopy(yGm_[yBase], yLocal, outRowLen_);
        winQueue_.FreeTensor(winLocal);
        outQueueRow_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  inQueueRow_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueRow_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> winQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t nRows_ {0};
    uint32_t rowLen_ {0};
    uint32_t outRowLen_ {0};
    uint32_t k_ {0};
    uint32_t s_ {0};
    uint32_t p_ {0};
    uint32_t tileMaxRow_ {0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y,
            tiling_data.nRows, tiling_data.rowLength, tiling_data.outRowLength,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.tileMaxRow);
    op.Process();
}
"""

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t compute_out_len(int64_t L, int64_t kernel, int64_t stride, int64_t padding) {
    if (stride <= 0) stride = kernel;
    const int64_t Lpad = L + 2 * padding;
    if (Lpad < kernel) return 0;
    return (Lpad - kernel) / stride + 1;
}

at::Tensor max_pooling_1d_impl_npu(const at::Tensor& self,
                                   int64_t kernel_size,
                                   c10::optional<int64_t> stride_opt,
                                   int64_t padding,
                                   bool return_indices) {
    TORCH_CHECK(self.dim() == 3, "Expected input of shape (N, C, L)");
    const int64_t N = self.size(0);
    const int64_t C = self.size(1);
    const int64_t L = self.size(2);

    int64_t stride = stride_opt.has_value() ? stride_opt.value() : kernel_size;
    const int64_t outL = compute_out_len(L, kernel_size, stride, padding);

    at::Tensor result = at::empty({N, C, outL}, self.options());
    // return_indices is ignored in this custom kernel (not produced)
    EXEC_NPU_CMD(aclnnMaxPooling1dCustom, self, (int64_t)kernel_size, (int64_t)stride, (int64_t)padding, (int64_t)(return_indices ? 1 : 0), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_1d_custom", &max_pooling_1d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_1d_custom", &max_pooling_1d_impl_npu, "MaxPooling1D (MaxPooling1dCustom on NPU)");
}

import torch

_FN = 'max_pooling_1d_custom'

class ModelNew(torch.nn.Module):

        super(ModelNew, self).__init__()
        self.kernel_size = int(kernel_size)
        self.stride = int(stride) if (stride is not None) else None
        self.padding = int(padding)
        self.return_indices = bool(return_indices)

    def forward(self, x: torch.Tensor) -> torch.Tensor:

        return fn(x, self.kernel_size, self.stride, self.padding, self.return_indices)
