kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Win, uint32_t Hin, uint32_t Din,
                                uint32_t Cout, uint32_t Kw, uint32_t Kh, uint32_t Kd,
                                uint32_t Wout, uint32_t Hout, uint32_t Dout,
                                uint32_t totalOut, uint32_t tileOut)
    {
        N_ = N; Cin_ = Cin; Win_ = Win; Hin_ = Hin; Din_ = Din;
        Cout_ = Cout; Kw_ = Kw; Kh_ = Kh; Kd_ = Kd;
        Wout_ = Wout; Hout_ = Hout; Dout_ = Dout;
        totalOut_ = totalOut; tileOut_ = tileOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_*Cin_*Win_*Hin_*Din_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cout_*Cin_*Kw_*Kh_*Kd_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_*Cout_*Wout_*Hout_*Dout_);

        // Minimal UB requirement: process 1 element at a time
        pipe_.InitBuffer(calcQ_, 1, tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tileOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        // Grid-stride over output elements (scalar tiles)
        for (uint32_t base = blockIdx; base < totalOut_; base += blockNum) {
            ComputeOne(base);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t p)
    {
        // Unravel p into [N, Cout, Wout, Hout, Dout] where D is the innermost (fastest) dim
        uint32_t d = p % Dout_; uint32_t t = p / Dout_;
        uint32_t h = t % Hout_; t /= Hout_;
        uint32_t w = t % Wout_; t /= Wout_;
        uint32_t co = t % Cout_; uint32_t n = t / Cout_;

        // Allocate 1-element local buffers
        AscendC::LocalTensor<float> acc = calcQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> xLoc = calcQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> wLoc = calcQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
        AscendC::Dup(acc, 0.0f, 1);

        // Accumulate over Cin and kernel window
        for (uint32_t ci = 0; ci < Cin_; ++ci) {
            for (uint32_t kw = 0; kw < Kw_; ++kw) {
                const uint32_t xw = w + kw; // within [0, Win-1]
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const uint32_t xh = h + kh; // within [0, Hin-1]
                    for (uint32_t kd = 0; kd < Kd_; ++kd) {
                        const uint32_t xd = d + kd; // within [0, Din-1]

                        // x offset: ((((n*Cin + ci)*Win + xw)*Hin + xh)*Din + xd)
                        uint64_t xOff = xd
                            + (uint64_t)Din_ * (xh
                            + (uint64_t)Hin_ * (xw
                            + (uint64_t)Win_ * (ci
                            + (uint64_t)Cin_ * n)));

                        // w offset: ((((co*Cin + ci)*Kw + kw)*Kh + kh)*Kd + kd)
                        uint64_t wOff = kd
                            + (uint64_t)Kd_ * (kh
                            + (uint64_t)Kh_ * (kw
                            + (uint64_t)Kw_ * (ci
                            + (uint64_t)Cin_ * co)));

                        AscendC::DataCopy(xLoc, xGm_[xOff], 1);
                        AscendC::DataCopy(wLoc, wGm_[wOff], 1);

                        AscendC::Mul(tmp, xLoc, wLoc, 1);
                        AscendC::Add(acc, acc, tmp, 1);
                    }
                }
            }
        }

        // y offset: ((((n*Cout + co)*Wout + w)*Hout + h)*Dout + d)
        uint64_t yOff = d
            + (uint64_t)Dout_ * (h
            + (uint64_t)Hout_ * (w
            + (uint64_t)Wout_ * (co
            + (uint64_t)Cout_ * n)));

        AscendC::DataCopy(yGm_[yOff], acc, 1);

        tmpQ_.FreeTensor(tmp);
        calcQ_.FreeTensor(wLoc);
        calcQ_.FreeTensor(xLoc);
        calcQ_.FreeTensor(acc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Win_{0}, Hin_{0}, Din_{0};
    uint32_t Cout_{0}, Kw_{0}, Kh_{0}, Kd_{0};
    uint32_t Wout_{0}, Hout_{0}, Dout_{0};
    uint32_t totalOut_{0}, tileOut_{1};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Win, tiling_data.Hin, tiling_data.Din,
            tiling_data.Cout, tiling_data.Kw, tiling_data.Kh, tiling_data.Kd,
            tiling_data.Wout, tiling_data.Hout, tiling_data.Dout,
            tiling_data.totalOut, tiling_data.tileOut);
    op.Process();
}
