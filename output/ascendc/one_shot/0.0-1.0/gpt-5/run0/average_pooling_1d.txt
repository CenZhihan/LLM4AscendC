#include "average_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// UB model for this kernel (per core):
// - xLocal: (tileOut + kernelSize - 1) floats
// - yLocal: tileOut floats
// Total floats = 2*tileOut + (kernelSize - 1) => bytes = 4*(2*tileOut + kernelSize - 1)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS     = 1024;
static constexpr uint32_t MIN_TILE_ELEMS  = 256;
static constexpr uint32_t MAX_BLOCKS      = 24;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling1dCustomTilingData tiling;

    const gert::Shape *shape0 = context->GetInputShape(0);
    if (shape0->GetOriginShape().GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }
    const int64_t n64 = shape0->GetOriginShape().GetDim(0);
    const int64_t c64 = shape0->GetOriginShape().GetDim(1);
    const int64_t l64 = shape0->GetOriginShape().GetDim(2);
    if (n64 <= 0 || c64 <= 0 || l64 <= 0) return ge::GRAPH_FAILED;

    const uint32_t n = static_cast<uint32_t>(n64);
    const uint32_t c = static_cast<uint32_t>(c64);
    const uint32_t inLen = static_cast<uint32_t>(l64);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *k_ptr = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *s_ptr = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *p_ptr = attrs->GetAttrPointer<int64_t>(2);

    const uint32_t kernelSize = static_cast<uint32_t>(*k_ptr);
    const uint32_t stride     = static_cast<uint32_t>(*s_ptr);
    const uint32_t padding    = static_cast<uint32_t>(*p_ptr);

    if (kernelSize == 0 || stride == 0) return ge::GRAPH_FAILED;

    // Output length: floor((inLen + 2*padding - kernelSize)/stride) + 1, saturate to zero
    int64_t outLen64 = (static_cast<int64_t>(inLen) + 2 * static_cast<int64_t>(padding) - static_cast<int64_t>(kernelSize));
    outLen64 = (outLen64 >= 0) ? (outLen64 / static_cast<int64_t>(stride) + 1) : 0;
    const uint32_t outLen = static_cast<uint32_t>(outLen64);

    // Tile size from UB budget
    uint32_t bytesPerFloat = 4u;
    uint32_t maxTileOut = 0u;
    if (UB_BUDGET_BYTES > bytesPerFloat * (kernelSize > 0 ? (kernelSize - 1) : 0)) {
        maxTileOut = (UB_BUDGET_BYTES / bytesPerFloat - (kernelSize - 1u)) / 2u;
    }
    if (maxTileOut < MIN_TILE_ELEMS) maxTileOut = MIN_TILE_ELEMS;
    maxTileOut = AlignDown(maxTileOut, ALIGN_ELEMS);
    if (maxTileOut == 0) maxTileOut = ALIGN_ELEMS;

    uint32_t tileOut = maxTileOut;
    if (outLen > 0 && outLen < tileOut) {
        const uint32_t aligned = AlignDown(outLen, ALIGN_ELEMS);
        tileOut = (aligned > 0u) ? aligned : outLen;
    }

    // Blocks over lines (N*C)
    const uint32_t totalLines = n * c;
    uint32_t blockDim = totalLines;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_inLen(inLen);
    tiling.set_outLen(outLen);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    if (x_shape->GetDimNum() != 3) return GRAPH_FAILED;

    const int64_t n = x_shape->GetDim(0);
    const int64_t c = x_shape->GetDim(1);
    const int64_t inLen = x_shape->GetDim(2);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t kernelSize = *attrs->GetAttrPointer<int64_t>(0);
    const int64_t stride     = *attrs->GetAttrPointer<int64_t>(1);

    y_shape->SetDim(0, n);
    y_shape->SetDim(1, c);
    y_shape->SetDim(2, outLen);
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

class KernelAveragePooling1dCustom {
public:
    __aicore__ inline KernelAveragePooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t inLen, uint32_t outLen,
                                uint32_t tileOut,
                                uint32_t kernelSize,
                                uint32_t stride,
                                uint32_t padding)
    {
        n_ = n; c_ = c;
        inLen_ = inLen; outLen_ = outLen;
        tileOut_ = tileOut;
        k_ = kernelSize;
        s_ = stride;
        p_ = padding;

        totalLines_ = n_ * c_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * c_ * inLen_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_ * c_ * outLen_));

        // Buffers:
        // xLocal: tileOut + k - 1
        // yLocal: tileOut
        uint32_t inLocal = tileOut_ + ((k_ > 0) ? (k_ - 1u) : 0u);
        uint32_t outLocal = tileOut_;

        pipe_.InitBuffer(inQueueX_, 1, inLocal * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, outLocal * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLines_ == 0 || outLen_ == 0) return;
        // Note: implementation optimized for stride == 1
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t line = blockIdx; line < totalLines_; line += blockNum) {
            const uint32_t xBase = line * inLen_;
            const uint32_t yBase = line * outLen_;

            for (uint32_t outBase = 0; outBase < outLen_; outBase += tileOut_) {
                uint32_t curOut = outLen_ - outBase;
                if (curOut > tileOut_) curOut = tileOut_;

                // Span of inputs needed for this output tile
                // spanLen = (curOut - 1)*stride + k
                uint32_t spanLen = (curOut - 1u) * s_ + k_;

                // Compute starting input index (can be negative due to padding)
                int64_t firstStart = static_cast<int64_t>(outBase) * static_cast<int64_t>(s_) - static_cast<int64_t>(p_);

                // Determine the contiguous valid GM copy range [gmStart, gmEnd)
                int64_t gmStart = firstStart < 0 ? 0 : firstStart;
                int64_t gmEnd = firstStart + static_cast<int64_t>(spanLen);
                if (gmEnd > static_cast<int64_t>(inLen_)) gmEnd = static_cast<int64_t>(inLen_);
                uint32_t gmCopyLen = (gmEnd > gmStart) ? static_cast<uint32_t>(gmEnd - gmStart) : 0u;

                // Local offset where valid GM data will be placed
                uint32_t localOff = (firstStart < 0) ? static_cast<uint32_t>(-firstStart) : 0u;

                // Prepare local buffers
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                // Zero-fill local input span
                AscendC::Duplicate(xLocal, 0.0f, spanLen);
                // Copy the valid segment from GM into xLocal[localOff]
                if (gmCopyLen > 0) {
                    AscendC::DataCopy(xLocal[localOff], xGm_[xBase + static_cast<uint32_t>(gmStart)], gmCopyLen);
                }

                // Compute: sliding window average (optimized for stride == 1)
                // yLocal[0:curOut] = sum_{i=0..k-1} xLocal[i : i+curOut]
                AscendC::Duplicate(yLocal, 0.0f, curOut);

                if (s_ == 1u) {
                    for (uint32_t i = 0; i < k_; ++i) {
                        AscendC::Add(yLocal, yLocal, xLocal[i], curOut);
                    }
                } else {
                    // Generic implementation for stride > 1 (fallback, less optimized):
                    // Accumulate each output position with scalar-like vector ops.
                    // We process in vector chunks when possible.
                    for (uint32_t j = 0; j < curOut; ++j) {
                        // position in xLocal
                        uint32_t start = j * s_;
                        // Accumulate k_ elements starting at start
                        float tmpScalar = 0.0f; // will be broadcast per element update
                        // To avoid scalar loops, do segment-wise add when possible
                        // But here we do a minimal approach: accumulate into a temp vector element-wise
                        // Implemented as: y[j] = sum(xLocal[start + t]), t in [0, k_)
                        // Use a small local temp to hold one element via Duplicate and AddN
                        for (uint32_t t = 0; t < k_; ++t) {
                            // load one value and accumulate to yLocal[j]
                            AscendC::LocalTensor<float> one = inQueueX_.AllocTensor<float>();
                            // reuse small section of xLocal as a 1-length tensor via pointer offset
                            // Duplicate approach: fetch xLocal[start + t] into scalar via ReduceAdd of length 1
                            AscendC::DataCopy(one, xLocal[start + t], 1);
                            // add into yLocal[j]
                            AscendC::Add(yLocal[j], yLocal[j], one, 1);
                            inQueueX_.FreeTensor(one);
                        }
                    }
                }

                // Scale by 1/k to get average, include padding in count
                const float invK = (k_ > 0) ? (1.0f / static_cast<float>(k_)) : 0.0f;
                AscendC::Muls(yLocal, yLocal, invK, curOut);

                // Write back
                AscendC::DataCopy(yGm_[yBase + outBase], yLocal, curOut);

                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0};
    uint32_t inLen_{0}, outLen_{0}, tileOut_{0};
    uint32_t k_{1}, s_{1}, p_{0};
    uint32_t totalLines_{0};
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling1dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.inLen, tiling_data.outLen,
            tiling_data.tileOut,
            tiling_data.kernelSize,
            tiling_data.stride,
            tiling_data.padding);
    op.Process();
}
