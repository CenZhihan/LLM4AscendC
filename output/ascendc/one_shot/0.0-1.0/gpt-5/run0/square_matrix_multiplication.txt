host_operator_src="""
#include "square_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0u) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

// UB memory budget per core (bytes). We will tile A, B, C + a small temp row.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ELEM_SIZE = 4;                  // float
static constexpr uint32_t ALIGN_M = 16;
static constexpr uint32_t ALIGN_N = 16;
static constexpr uint32_t ALIGN_K = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SquareMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const int64_t dim0 = a_shape->GetOriginShape().GetDim(0);
    const int64_t dim1 = a_shape->GetOriginShape().GetDim(1);
    if (dim0 <= 0 || dim1 <= 0 || dim0 != dim1) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(dim0);

    // Start with reasonable tile sizes that fit in UB:
    // A: tileM*tileK, B: tileK*tileN, C: tileM*tileN, tmp: tileN
    // Total floats = tileM*tileK + tileK*tileN + tileM*tileN + tileN
    // Choose defaults: M=64, N=64, K=128
    uint32_t tileM = 64u;
    uint32_t tileN = 64u;
    uint32_t tileK = 128u;

    // Clip tiles by problem size
    if (N < tileM) tileM = N;
    if (N < tileN) tileN = N;
    if (N < tileK) tileK = N;

    // Align tiles for DMA efficiency
    tileM = AlignDown(tileM, ALIGN_M);
    tileN = AlignDown(tileN, ALIGN_N);
    tileK = AlignDown(tileK, ALIGN_K);
    if (tileM == 0u) tileM = (N < ALIGN_M) ? N : ALIGN_M;
    if (tileN == 0u) tileN = (N < ALIGN_N) ? N : ALIGN_N;
    if (tileK == 0u) tileK = (N < ALIGN_K) ? N : ALIGN_K;

    // Ensure UB budget (very conservative check)
    // bytes = (tileM*tileK + tileK*tileN + tileM*tileN + tileN) * 4
    auto fits = [&](uint32_t m, uint32_t n, uint32_t k) -> bool {
        uint64_t elems = static_cast<uint64_t>(m) * k
                       + static_cast<uint64_t>(k) * n
                       + static_cast<uint64_t>(m) * n
                       + static_cast<uint64_t>(n);
        return elems * ELEM_SIZE <= UB_BUDGET_BYTES;
    };

    while (!fits(tileM, tileN, tileK)) {
        if (tileK > ALIGN_K) tileK = AlignDown(tileK / 2u, ALIGN_K);
        else if (tileN > ALIGN_N) tileN = AlignDown(tileN / 2u, ALIGN_N);
        else if (tileM > ALIGN_M) tileM = AlignDown(tileM / 2u, ALIGN_M);
        else break;
        if (tileM == 0u) tileM = 1u;
        if (tileN == 0u) tileN = 1u;
        if (tileK == 0u) tileK = 1u;
    }

    // Select blockDim based on number of M-tiles, clamp to [8, 24] for stability
    uint32_t mTiles = CeilDiv(N, tileM);
    uint32_t blockDim = mTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 ||
        b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    int64_t n0 = a_shape->GetOriginShape().GetDim(0);
    int64_t n1 = a_shape->GetOriginShape().GetDim(1);
    int64_t m0 = b_shape->GetOriginShape().GetDim(0);
    int64_t m1 = b_shape->GetOriginShape().GetDim(1);
    if (n0 != n1 || m0 != m1 || n0 != m0) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    *c_shape = *a_shape;
    return GRAPH_SUCCESS;
}

    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;

    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSquareMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelSquareMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(N_) * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(N_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(N_) * N_);

        pipe_.InitBuffer(aTileQ_, 1, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(bTileQ_, 1, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(cTileQ_, 1, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpRowQ_, 1, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || tileM_ == 0 || tileN_ == 0 || tileK_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t mStride = blockNum * tileM_;
        uint32_t mBase = blockIdx * tileM_;
        for (; mBase < N_; mBase += mStride) {
            const uint32_t curM = (mBase + tileM_ <= N_) ? tileM_ : (N_ - mBase);

            for (uint32_t nBase = 0; nBase < N_; nBase += tileN_) {
                const uint32_t curN = (nBase + tileN_ <= N_) ? tileN_ : (N_ - nBase);

                AscendC::LocalTensor<float> aTile = aTileQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> bTile = bTileQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> cTile = cTileQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpRow = tmpRowQ_.AllocTensor<float>();

                bool firstAcc = true;

                // Iterate over K dimension in chunks
                for (uint32_t kBase = 0; kBase < N_; kBase += tileK_) {
                    const uint32_t curK = (kBase + tileK_ <= N_) ? tileK_ : (N_ - kBase);

                    // Load A sub-tile: shape (curM, curK), row by row
                    for (uint32_t i = 0; i < curM; ++i) {
                        const uint64_t aOffset = (static_cast<uint64_t>(mBase + i) * N_) + kBase;
                        AscendC::DataCopy(aTile[i * tileK_], aGm_[aOffset], curK);
                    }

                    // Load B sub-tile: shape (curK, curN), row by row
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        const uint64_t bOffset = (static_cast<uint64_t>(kBase + kk) * N_) + nBase;
                        AscendC::DataCopy(bTile[kk * tileN_], bGm_[bOffset], curN);
                    }

                    // Compute C += A_sub @ B_sub
                    for (uint32_t i = 0; i < curM; ++i) {
                        AscendC::LocalTensor<float> cRow = cTile[i * tileN_];

                        for (uint32_t kk = 0; kk < curK; ++kk) {
                            const uint32_t aIdx = i * tileK_ + kk;
                            // Read scalar a(i, k)
                            float aVal = aTile.GetValue(aIdx);

                            AscendC::LocalTensor<float> bRow = bTile[kk * tileN_];
                            if (firstAcc) {
                                // cRow = aVal * bRow
                                AscendC::Muls(cRow, bRow, aVal, curN);
                            } else {
                                // tmpRow = aVal * bRow; cRow += tmpRow
                                AscendC::Muls(tmpRow, bRow, aVal, curN);
                                AscendC::Add(cRow, cRow, tmpRow, curN);
                            }
                        }
                        firstAcc = false;
                    }
                }

                // Store C sub-tile: shape (curM, curN), row by row
                for (uint32_t i = 0; i < curM; ++i) {
                    const uint64_t cOffset = (static_cast<uint64_t>(mBase + i) * N_) + nBase;
                    AscendC::DataCopy(cGm_[cOffset], cTile[i * tileN_], curN);
                }

                tmpRowQ_.FreeTensor(tmpRow);
                cTileQ_.FreeTensor(cTile);
                bTileQ_.FreeTensor(bTile);
                aTileQ_.FreeTensor(aTile);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bTileQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> cTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpRowQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t tileM_{0};
    uint32_t tileN_{0};
    uint32_t tileK_{0};
};

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSquareMatrixMultiplicationCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
