host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <cstdint>

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

} // anonymous namespace

namespace optiling {

// UB model per element in a tile across post-dimension:
//   row(f32): 1
//   max(f32): 1
//   idx(i32): 1
//   idx64(i64): 2
//   kvec(i32): 1
// ---------------------
// Total bytes ~= 24 * tileJ
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
static constexpr uint32_t BYTES_PER_ELEM = 24;
static constexpr uint32_t ALIGN_ELEMS = 1024;           // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 16384;       // cap to reduce per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const gert::OriginShape &origin = in_shape->GetOriginShape();
    const int64_t rank = origin.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *axisPtr = attrs->GetAttrPointer<int32_t>(0);
    int64_t axis = static_cast<int64_t>(*axisPtr);
    int64_t axisNorm = axis;
    if (axisNorm < 0) axisNorm += rank;
    if (axisNorm < 0 || axisNorm >= rank) {
        return ge::GRAPH_FAILED;
    }

    uint64_t preSize64 = 1, axisSize64 = 1, postSize64 = 1;
    for (int64_t i = 0; i < axisNorm; ++i) {
        preSize64 *= static_cast<uint64_t>(origin.GetDim(i));
    }
    axisSize64 = static_cast<uint64_t>(origin.GetDim(axisNorm));
    for (int64_t i = axisNorm + 1; i < rank; ++i) {
        postSize64 *= static_cast<uint64_t>(origin.GetDim(i));
    }

    const uint64_t totalInputSize = preSize64 * axisSize64 * postSize64;

    // Determine tileJ using UB budget
    uint32_t maxTileJ = UB_BUDGET_BYTES / BYTES_PER_ELEM;
    maxTileJ = AlignDown(maxTileJ, ALIGN_ELEMS);
    if (maxTileJ < ALIGN_ELEMS) maxTileJ = ALIGN_ELEMS;
    if (maxTileJ > HARD_CAP_ELEMS) maxTileJ = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    const uint32_t postSz = static_cast<uint32_t>(postSize64);
    uint32_t tileJ = maxTileJ;
    if (postSz < tileJ) {
        const uint32_t aligned = AlignDown(postSz, ALIGN_ELEMS);
        tileJ = (aligned > 0) ? aligned : postSz;
    }

    const uint32_t preSz = static_cast<uint32_t>(preSize64);
    const uint32_t numJTiles = (tileJ == 0) ? 1u : CeilDiv(postSz, tileJ);
    const uint64_t totalTasks64 = static_cast<uint64_t>(preSz) * static_cast<uint64_t>(numJTiles);
    uint32_t totalTasks = totalTasks64 > UINT32_MAX ? UINT32_MAX : static_cast<uint32_t>(totalTasks64);

    uint32_t blockDim = totalTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_preSize(static_cast<uint32_t>(preSize64));
    tiling.set_axisSize(static_cast<uint32_t>(axisSize64));
    tiling.set_postSize(static_cast<uint32_t>(postSize64));
    tiling.set_tileJ(tileJ);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    (void)totalInputSize;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::OriginShape &origin = x_shape->GetOriginShape();
    const int64_t rank = origin.GetDimNum();

    const int32_t axisAttr = *context->GetAttrs()->GetAttrPointer<int32_t>(0);
    int64_t axis = axisAttr;
    if (axis < 0) axis += rank;
    if (axis < 0 || axis >= rank) {
        return GRAPH_FAILED;
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    gert::Shape tmp;
    tmp.SetFormat(origin.GetFormat());
    // Build output dims with axis removed
    std::vector<int64_t> outDims;
    outDims.reserve(static_cast<size_t>(rank > 0 ? rank - 1 : 0));
    for (int64_t i = 0; i < rank; ++i) {
        if (i == axis) continue;
        outDims.push_back(origin.GetDim(i));
    }
    if (outDims.empty()) {
        // Scalar result if all dims reduced away; represent as [1]
        outDims.push_back(1);
    }
    tmp.SetDims(outDims);
    *y_shape = tmp;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM_IN = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t preSize, uint32_t axisSize,
                                uint32_t postSize, uint32_t tileJ)
    {
        preSize_ = preSize;
        axisSize_ = axisSize;
        postSize_ = postSize;
        tileJ_ = tileJ;

        const uint64_t totalIn = static_cast<uint64_t>(preSize_) * axisSize_ * postSize_;
        const uint64_t totalOut = static_cast<uint64_t>(preSize_) * postSize_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalIn);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalOut);

        pipe_.InitBuffer(inF32Queue_, BUFFER_NUM_IN, tileJ_ * sizeof(float));     // row + max
        pipe_.InitBuffer(i32Queue_, 1, tileJ_ * sizeof(int32_t));                 // idx32 / kvec
        pipe_.InitBuffer(outI64Queue_, 1, tileJ_ * sizeof(int64_t));              // output idx64
    }

    __aicore__ inline void Process()
    {
        if (preSize_ == 0 || postSize_ == 0) return;

        const uint32_t numJTiles = (tileJ_ == 0) ? 1u : (postSize_ + tileJ_ - 1u) / tileJ_;
        const uint64_t totalTasks = static_cast<uint64_t>(preSize_) * numJTiles;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t task = blockIdx; task < totalTasks; task += blockNum) {
            const uint32_t i = static_cast<uint32_t>(task / numJTiles);
            const uint32_t jTile = static_cast<uint32_t>(task % numJTiles);
            const uint32_t jStart = jTile * tileJ_;
            uint32_t curJ = postSize_ - jStart;
            if (curJ > tileJ_) curJ = tileJ_;
            ReduceOneTile(i, jStart, curJ);
        }
    }

private:
    __aicore__ inline void ReduceOneTile(uint32_t i, uint32_t jStart, uint32_t curJ)
    {
        // Allocate locals
        AscendC::LocalTensor<float> maxLocal = inF32Queue_.AllocTensor<float>();
        AscendC::Dup(maxLocal, -1.0e30f, curJ);

        AscendC::LocalTensor<int32_t> idxLocal = i32Queue_.AllocTensor<int32_t>();
        AscendC::Dup(idxLocal, 0, curJ);

        for (uint32_t k = 0; k < axisSize_; ++k) {
            // Load a contiguous row segment: length curJ
            const uint64_t base = (static_cast<uint64_t>(i) * axisSize_ + k) * postSize_ + jStart;
            AscendC::LocalTensor<float> rowLocal = inF32Queue_.AllocTensor<float>();
            AscendC::DataCopy(rowLocal, xGm_[base], curJ);

            // Update maxLocal and idxLocal where rowLocal > maxLocal
            AscendC::LocalTensor<uint8_t> maskLocal; // assume compare produces mask in temp
            // Compute new max in-place
            AscendC::Max(maxLocal, maxLocal, rowLocal, curJ);
            // Build vector of current k
            AscendC::LocalTensor<int32_t> kvecLocal = i32Queue_.AllocTensor<int32_t>();
            AscendC::Dup(kvecLocal, static_cast<int32_t>(k), curJ);
            // Update indices where rowLocal became new max: idx = select(row > prev, k, idx)
            AscendC::Select(idxLocal, rowLocal, maxLocal, kvecLocal, idxLocal, curJ);

            i32Queue_.FreeTensor(kvecLocal);
            inF32Queue_.FreeTensor(rowLocal);
        }

        // Cast to int64 and write out
        AscendC::LocalTensor<int64_t> outLocal = outI64Queue_.AllocTensor<int64_t>();
        AscendC::Cast(outLocal, idxLocal, curJ);
        const uint64_t yBase = static_cast<uint64_t>(i) * postSize_ + jStart;
        AscendC::DataCopy(yGm_[yBase], outLocal, curJ);

        outI64Queue_.FreeTensor(outLocal);
        i32Queue_.FreeTensor(idxLocal);
        inF32Queue_.FreeTensor(maxLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM_IN> inF32Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> i32Queue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outI64Queue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t preSize_ {0};
    uint32_t axisSize_ {0};
    uint32_t postSize_ {0};
    uint32_t tileJ_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.preSize,
            tiling_data.axisSize,
            tiling_data.postSize,
            tiling_data.tileJ);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline std::vector<int64_t> compute_out_sizes(const at::Tensor& self, int64_t axis) {
    const int64_t ndim = self.dim();
    int64_t ax = axis;
    if (ax < 0) ax += ndim;
    TORCH_CHECK(ax >= 0 && ax < ndim, "axis out of range");
    std::vector<int64_t> sizes;
    sizes.reserve(static_cast<size_t>(ndim > 0 ? ndim - 1 : 0));
    for (int64_t i = 0; i < ndim; ++i) {
        if (i == ax) continue;
        sizes.push_back(self.size(i));
    }
    if (sizes.empty()) sizes.push_back(1);
    return sizes;
}

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t axis) {
    auto out_sizes = compute_out_sizes(self, axis);
    at::Tensor result = at::empty(out_sizes, self.options().dtype(at::kLong));
    int32_t ax32 = static_cast<int32_t>(axis);
    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, ax32, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu, "Argmax over a dimension (ArgmaxOverADimensionCustom on NPU)");
}
