host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) { return static_cast<uint32_t>((a + b - 1u) / b); }
}  // namespace

namespace optiling {

// UB model for this kernel (float x + float mask + float y):
//   inQueueX: 2 * tile
//   inQueueM: 2 * tile
//   outQueueY: 2 * tile
// Total floats = 6 * tile => bytes = 24 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;             // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;         // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &ori = in_shape->GetOriginShape();
    const std::vector<int64_t> dims = ori.GetDims();
    const uint32_t rank = static_cast<uint32_t>(dims.size());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dim_attr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = (dim_attr != nullptr) ? *dim_attr : 0;
    if (dim < 0) dim += static_cast<int64_t>(rank);
    if (dim < 0) dim = 0;
    if (dim >= static_cast<int64_t>(rank)) dim = static_cast<int64_t>(rank) - 1;

    uint64_t outerLen = 1;
    uint64_t innerLen = (rank > 0) ? static_cast<uint64_t>(dims[dim]) : 1;
    uint64_t tail = 1;

    for (int64_t i = 0; i < dim; ++i) outerLen *= static_cast<uint64_t>(dims[i]);
    for (uint32_t i = static_cast<uint32_t>(dim + 1); i < rank; ++i) tail *= static_cast<uint64_t>(dims[i]);

    uint64_t totalLength = 1;
    for (auto d : dims) totalLength *= static_cast<uint64_t>(d);

    // tile length along the scan axis from UB budget: bytes = 24 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 24;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = static_cast<uint32_t>(innerLen);
    if (tileLen > maxTile) {
        tileLen = maxTile;
    } else {
        // try to keep alignment while not exceeding innerLen
        uint32_t aligned = AlignDown(tileLen, ALIGN_ELEMS);
        if (aligned > 0) tileLen = aligned;
    }
    if (innerLen < tileLen) {
        uint32_t aligned = AlignDown(static_cast<uint32_t>(innerLen), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(innerLen);
    }
    if (tileLen == 0) tileLen = 1;

    // lines to process = outerLen * tail; each line is a scan of length innerLen with stride=tail
    uint64_t lines = outerLen * tail;
    uint32_t blockDim = static_cast<uint32_t>(lines);
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_outerLength(static_cast<uint32_t>(outerLen));
    tiling.set_innerLength(static_cast<uint32_t>(innerLen));
    tiling.set_tail(static_cast<uint32_t>(tail));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(dim));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t outerLength,
                                uint32_t innerLength,
                                uint32_t tail,
                                uint32_t tileLength,
                                int32_t dim)
    {
        totalLen_ = totalLength;
        outerLen_ = outerLength;
        innerLen_ = innerLength;
        tail_ = tail;
        tileLen_ = tileLength;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        mGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueM_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // We scan along lines = outerLen_ * tail_
        const uint32_t lines = (outerLen_ == 0 || tail_ == 0) ? 0 : outerLen_ * tail_;
        if (lines == 0) return;

        for (uint32_t lineId = blockIdx; lineId < lines; lineId += blockNum) {
            // Decode line into (outerIndex, p) where p in [0, tail_)
            const uint32_t outerIndex = (tail_ == 0) ? 0 : (lineId / tail_);
            const uint32_t p = (tail_ == 0) ? 0 : (lineId % tail_);

            // Base element index for this line's first position
            // Layout: [outer][inner][tail]
            const uint64_t base = static_cast<uint64_t>(outerIndex) * static_cast<uint64_t>(innerLen_) * static_cast<uint64_t>(tail_) + static_cast<uint64_t>(p);

            float carry = 0.0f;

            // Prefetch first tile
            uint32_t pos = 0;
            uint32_t curLen = innerLen_ - pos;
            if (curLen > tileLen_) curLen = tileLen_;
            CopyIn(base, pos, curLen);

            while (pos < innerLen_) {
                // Prefetch next tile
                const uint32_t nextPos = pos + curLen;
                uint32_t nextLen = 0;
                if (nextPos < innerLen_) {
                    nextLen = innerLen_ - nextPos;
                    if (nextLen > tileLen_) nextLen = tileLen_;
                    CopyIn(base, nextPos, nextLen);
                }

                // Compute cumsum on current tile
                carry = Compute(curLen, carry);

                // Write back
                CopyOut(base, pos, curLen);

                if (nextLen == 0) break;

                pos = nextPos;
                curLen = nextLen;
            }
        }
    }

private:
    // Copy a contiguous or strided segment of a line into UB
    __aicore__ inline void CopyIn(uint64_t base, uint32_t pos, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> mLocal = inQueueM_.AllocTensor<float>();

        if (tail_ == 1) {
            const uint64_t gmOffset = base + static_cast<uint64_t>(pos);
            AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
            AscendC::DataCopy(mLocal, mGm_[gmOffset], len);
        } else {
            // Fallback: gather strided elements one by one
            float* xPtr = reinterpret_cast<float*>(xLocal.GetPtr(0));
            float* mPtr = reinterpret_cast<float*>(mLocal.GetPtr(0));
            for (uint32_t i = 0; i < len; ++i) {
                const uint64_t gmOffset = base + static_cast<uint64_t>(pos + i) * static_cast<uint64_t>(tail_);
                AscendC::LocalTensor<float> tmpX = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpM = inQueueM_.AllocTensor<float>();
                AscendC::DataCopy(tmpX, xGm_[gmOffset], 1);
                AscendC::DataCopy(tmpM, mGm_[gmOffset], 1);
                float* tx = reinterpret_cast<float*>(tmpX.GetPtr(0));
                float* tm = reinterpret_cast<float*>(tmpM.GetPtr(0));
                xPtr[i] = tx[0];
                mPtr[i] = tm[0];
                inQueueX_.FreeTensor(tmpX);
                inQueueM_.FreeTensor(tmpM);
            }
        }

        inQueueX_.EnQue(xLocal);
        inQueueM_.EnQue(mLocal);
    }

    // Compute masked cumsum for a tile; returns updated carry (running sum)
    __aicore__ inline float Compute(uint32_t len, float carry)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> mLocal = inQueueM_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        float* xPtr = reinterpret_cast<float*>(xLocal.GetPtr(0));
        float* mPtr = reinterpret_cast<float*>(mLocal.GetPtr(0));
        float* yPtr = reinterpret_cast<float*>(yLocal.GetPtr(0));

        for (uint32_t i = 0; i < len; ++i) {
            float xm = xPtr[i] * mPtr[i];
            carry += xm;
            yPtr[i] = carry;
        }

        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueM_.FreeTensor(mLocal);
        return carry;
    }

    // Copy a UB segment back to GM with contiguous or strided layout
    __aicore__ inline void CopyOut(uint64_t base, uint32_t pos, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();

        if (tail_ == 1) {
            const uint64_t gmOffset = base + static_cast<uint64_t>(pos);
            AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        } else {
            // Scatter strided elements one by one
            float* yPtr = reinterpret_cast<float*>(yLocal.GetPtr(0));
            for (uint32_t i = 0; i < len; ++i) {
                const uint64_t gmOffset = base + static_cast<uint64_t>(pos + i) * static_cast<uint64_t>(tail_);
                AscendC::LocalTensor<float> tmp = outQueueY_.AllocTensor<float>();
                float* tp = reinterpret_cast<float*>(tmp.GetPtr(0));
                tp[0] = yPtr[i];
                AscendC::DataCopy(yGm_[gmOffset], tmp, 1);
                outQueueY_.FreeTensor(tmp);
            }
        }

        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueM_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> mGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t outerLen_{0};
    uint32_t innerLen_{0};
    uint32_t tail_{1};
    uint32_t tileLen_{1};
    int32_t dim_{0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.totalLength,
            tiling_data.outerLength,
            tiling_data.innerLength,
            tiling_data.tail,
            tiling_data.tileLength,
            tiling_data.dim);
    op.Process();
}
