#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model (per tile along feature dimension):
// Inputs: anchor, positive, negative -> 3 * tile
// Temps: diff_ap, diff_an               2 * tile
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;             // DMA-friendly alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;         // cap per-iter length

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    if (a_shape->GetOriginShape().GetDimNum() < 1) {
        return ge::GRAPH_FAILED;
    }

    // Expect 2D [N, D]; support flatten if higher dims by collapsing trailing dims
    uint32_t dimNum = static_cast<uint32_t>(a_shape->GetOriginShape().GetDimNum());
    uint64_t rows64 = a_shape->GetOriginShape().GetDim(0);
    uint64_t cols64 = 1;
    for (uint32_t i = 1; i < dimNum; ++i) {
        cols64 *= a_shape->GetOriginShape().GetDim(i);
    }
    const uint32_t rows = static_cast<uint32_t>(rows64);
    const uint32_t cols = static_cast<uint32_t>(cols64);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *marginPtr = attrs->GetAttrPointer<float>(0);
    const float margin = (marginPtr != nullptr) ? *marginPtr : 1.0f;

    // tileCols from UB budget: bytes = 20 * tileCols
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements along cols
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileCols = maxTile;
    if (cols < tileCols) {
        const uint32_t aligned = AlignDown(cols, ALIGN_ELEMS);
        tileCols = (aligned > 0) ? aligned : cols;
    }

    // Use single core to avoid cross-core reduction for scalar output.
    const uint32_t blockDim = 1u;
    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t tileCols, float margin)
    {
        rows_ = rows;
        cols_ = cols;
        tileCols_ = tileCols > 0 ? tileCols : cols;
        margin_ = margin;

        aGm_.SetGlobalBuffer((__gm__ float*)anchor, rows_ * cols_);
        pGm_.SetGlobalBuffer((__gm__ float*)positive, rows_ * cols_);
        nGm_.SetGlobalBuffer((__gm__ float*)negative, rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(inQueueP_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(inQueueN_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(calcQ1_, 1, tileCols_ * sizeof(float));
        pipe_.InitBuffer(calcQ2_, 1, tileCols_ * sizeof(float));
        pipe_.InitBuffer(outQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) {
            // write zero
            AscendC::LocalTensor<float> outLocal = outQ_.AllocTensor<float>();
            AscendC::Dup(outLocal, 0.0f, 1);
            AscendC::DataCopy(yGm_[0], outLocal, 1);
            outQ_.FreeTensor(outLocal);
            return;
        }

        // Single-core reduction by design (blockDim=1 in tiling)
        float totalLoss = 0.0f;

        for (uint32_t r = 0; r < rows_; ++r) {
            float sumAp = 0.0f;
            float sumAn = 0.0f;

            uint32_t baseRowOffset = r * cols_;
            uint32_t processed = 0;

            // Prefetch first segment
            uint32_t curLen = cols_ - processed;
            if (curLen > tileCols_) curLen = tileCols_;
            CopyIn(baseRowOffset + processed, curLen);

            while (true) {
                // Prefetch next segment
                processed += curLen;
                uint32_t nextLen = 0;
                if (processed < cols_) {
                    nextLen = cols_ - processed;
                    if (nextLen > tileCols_) nextLen = tileCols_;
                    CopyIn(baseRowOffset + processed, nextLen);
                }

                // Compute partial sums for current segment
                ReduceSeg(curLen, sumAp, sumAn);

                if (nextLen == 0) break;
                curLen = nextLen;
            }

            // L2 norms with small epsilon for numerical stability
            float d_ap = SqrtF(sumAp + 1e-12f);
            float d_an = SqrtF(sumAn + 1e-12f);
            float z = d_ap - d_an + margin_;

            // ReLU via subtracting negative part
            float zmin = z < 0.0f ? z : 0.0f;
            float zpos = z - zmin;
            totalLoss += zpos;
        }

        float meanLoss = totalLoss / static_cast<float>(rows_);
        AscendC::LocalTensor<float> outLocal = outQ_.AllocTensor<float>();
        AscendC::Dup(outLocal, meanLoss, 1);
        AscendC::DataCopy(yGm_[0], outLocal, 1);
        outQ_.FreeTensor(outLocal);
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
        AscendC::LocalTensor<float> pLocal = inQueueP_.AllocTensor<float>();
        AscendC::LocalTensor<float> nLocal = inQueueN_.AllocTensor<float>();

        AscendC::DataCopy(aLocal, aGm_[gmOffset], len);
        AscendC::DataCopy(pLocal, pGm_[gmOffset], len);
        AscendC::DataCopy(nLocal, nGm_[gmOffset], len);

        inQueueA_.EnQue(aLocal);
        inQueueP_.EnQue(pLocal);
        inQueueN_.EnQue(nLocal);
    }

    __aicore__ inline void ReduceSeg(uint32_t len, float &sumAp, float &sumAn)
    {
        AscendC::LocalTensor<float> aLocal = inQueueA_.DeQue<float>();
        AscendC::LocalTensor<float> pLocal = inQueueP_.DeQue<float>();
        AscendC::LocalTensor<float> nLocal = inQueueN_.DeQue<float>();

        AscendC::LocalTensor<float> t1 = calcQ1_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2 = calcQ2_.AllocTensor<float>();

        // diff_ap = anchor - positive
        AscendC::Sub(t1, aLocal, pLocal, len);
        // diff_an = anchor - negative
        AscendC::Sub(t2, aLocal, nLocal, len);

        // square
        AscendC::Mul(t1, t1, t1, len);
        AscendC::Mul(t2, t2, t2, len);

        // partial reduce sum into scalars
        float partAp = ReduceSumLocal(t1, len);
        float partAn = ReduceSumLocal(t2, len);

        sumAp += partAp;
        sumAn += partAn;

        calcQ1_.FreeTensor(t1);
        calcQ2_.FreeTensor(t2);

        inQueueA_.FreeTensor(aLocal);
        inQueueP_.FreeTensor(pLocal);
        inQueueN_.FreeTensor(nLocal);
    }

    __aicore__ inline float ReduceSumLocal(AscendC::LocalTensor<float> vec, uint32_t len)
    {
        // Reduce sum of 'len' elements in vec into a single float
        AscendC::LocalTensor<float> sLocal = outQ_.AllocTensor<float>();
        AscendC::ReduceSum(sLocal, vec, len);
        // Read back the single value; modeled as index 0
        float sumVal = sLocal.GetValue(0);
        outQ_.FreeTensor(sLocal);
        return sumVal;
    }

    __aicore__ inline float SqrtF(float x)
    {
        // Scalar sqrt helper
        // Optionally use vector sqrt on a single element if needed
        return x <= 0.0f ? 0.0f : AscendC::Sqrt(x);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueP_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueN_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>        calcQ1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>        calcQ2_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1>         outQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> pGm_;
    AscendC::GlobalTensor<float> nGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_{0};
    uint32_t cols_{0};
    uint32_t tileCols_{0};
    float margin_{1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y,
            tiling_data.rows, tiling_data.cols, tiling_data.tileCols, tiling_data.margin);
    op.Process();
}
