host_operator_src="""
#include "rms_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}
} // anonymous namespace

namespace optiling {

// UB model for this kernel:
//   inQueueX: 1 * tile
//   outQueueY: 1 * tile
//   tmpQueue (sum/scale): 1 * tile
// Total floats = 3 * tile => bytes = 12 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536; // keep per-iter small enough

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    RmsNormCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);

    const int64_t dim_cnt = static_cast<int64_t>(shape->GetOriginShape().GetDimNum());
    if (dim_cnt < 2) {
        return ge::GRAPH_FAILED;
    }

    // Extract dims
    uint64_t totalLen_u64 = 1;
    for (int64_t i = 0; i < dim_cnt; ++i) {
        totalLen_u64 *= static_cast<uint64_t>(shape->GetOriginShape().GetDim(i));
    }
    const uint32_t totalLength = static_cast<uint32_t>(totalLen_u64);

    const uint32_t batchSize = static_cast<uint32_t>(shape->GetOriginShape().GetDim(0));
    const uint32_t channels  = static_cast<uint32_t>(shape->GetOriginShape().GetDim(1));
    uint32_t inner = 1u;
    for (int64_t i = 2; i < dim_cnt; ++i) {
        inner *= static_cast<uint32_t>(shape->GetOriginShape().GetDim(i));
    }

    // tileLen from UB budget: bytes = 12 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    // tileLen cannot exceed inner; keep aligned when possible
    uint32_t tileLen = maxTile;
    if (inner < tileLen) {
        const uint32_t aligned = AlignDown(inner, ALIGN_ELEMS);
        tileLen = (aligned > 0u) ? aligned : inner;
    }

    const uint32_t numInnerTiles = (tileLen == 0u) ? 1u : CeilDiv(inner, tileLen);
    const uint32_t totalWork = (batchSize == 0u) ? 0u : (batchSize * numInnerTiles);
    uint32_t blockDim = totalWork;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_batchSize(batchSize);
    tiling.set_channels(channels);
    tiling.set_inner(inner);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

public:
    explicit RmsNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("eps").AttrType(OPTIONAL).Float(1e-5f);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};
OP_ADD(RmsNormCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUFFERS = 2;
constexpr int32_t OUT_BUFFERS = 2;
constexpr int32_t TMP_BUFFERS = 1;

class KernelRmsNormCustom {
public:
    __aicore__ inline KernelRmsNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t batchSize,
                                uint32_t channels,
                                uint32_t inner,
                                uint32_t tileLength,
                                float eps)
    {
        totalLen_ = totalLength;
        batch_ = batchSize;
        channels_ = channels;
        inner_ = inner;
        tileLen_ = tileLength;
        eps_ = eps;
        invC_ = (channels_ > 0u) ? (1.0f / static_cast<float>(channels_)) : 0.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, IN_BUFFERS, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUFFERS, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMP_BUFFERS, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0u || tileLen_ == 0u || channels_ == 0u) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numInnerTiles = (inner_ + tileLen_ - 1u) / tileLen_;
        const uint32_t totalWork = batch_ * numInnerTiles;

        for (uint32_t work = blockIdx; work < totalWork; work += blockNum) {
            const uint32_t b = work / numInnerTiles;
            const uint32_t t = work % numInnerTiles;

            const uint32_t startInner = t * tileLen_;
            uint32_t curLen = inner_ - startInner;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0u) continue;

            const uint32_t batchBase = b * channels_ * inner_;

            // sumLocal will hold sum of squares, then transformed into scale (1 / rms)
            AscendC::LocalTensor<float> sumLocal = tmpQueue_.AllocTensor<float>();

            // Initialize accumulation with first channel
            {
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                const uint32_t off0 = batchBase + 0u * inner_ + startInner;
                AscendC::DataCopy(xLocal, xGm_[off0], curLen);
                // sumLocal = xLocal * xLocal
                AscendC::Mul(sumLocal, xLocal, xLocal, curLen);
                inQueueX_.FreeTensor(xLocal);
            }

            // Accumulate remaining channels
            for (uint32_t c = 1u; c < channels_; ++c) {
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                const uint32_t off = batchBase + c * inner_ + startInner;
                AscendC::DataCopy(xLocal, xGm_[off], curLen);
                AscendC::Mul(xLocal, xLocal, xLocal, curLen);            // x^2
                AscendC::Add(sumLocal, sumLocal, xLocal, curLen);         // sum += x^2
                inQueueX_.FreeTensor(xLocal);
            }

            // mean -> add eps -> inverse sqrt to get scale
            AscendC::Muls(sumLocal, sumLocal, invC_, curLen);
            AscendC::Adds(sumLocal, sumLocal, eps_, curLen);
            AscendC::Rsqrt(sumLocal, sumLocal, curLen); // scale = 1 / sqrt(mean + eps)

            // Second pass: y = x * scale for all channels
            for (uint32_t c = 0u; c < channels_; ++c) {
                const uint32_t off = batchBase + c * inner_ + startInner;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                AscendC::DataCopy(xLocal, xGm_[off], curLen);
                AscendC::Mul(yLocal, xLocal, sumLocal, curLen);
                AscendC::DataCopy(yGm_[off], yLocal, curLen);

                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            tmpQueue_.FreeTensor(sumLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFERS> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFERS> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUFFERS> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t batch_{0};
    uint32_t channels_{0};
    uint32_t inner_{0};
    uint32_t tileLen_{0};
    float eps_{1e-5f};
    float invC_{0.0f};
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNormCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.batchSize,
            tiling_data.channels,
            tiling_data.inner,
            tiling_data.tileLength,
            tiling_data.eps);
    op.Process();
}
"""
