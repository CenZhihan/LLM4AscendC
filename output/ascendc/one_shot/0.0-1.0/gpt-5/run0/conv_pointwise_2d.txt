#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // favor large vector ops
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t n   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t h   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t w   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t win  = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t kh   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t kw   = static_cast<uint32_t>(w_shape->GetDim(3));

    if (win != cin || kh != 1 || kw != 1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t s = h * w;

    // Choose KO (output-channel chunk) and spatial tile size to fit UB
    uint32_t ko = 8;
    if (cout < ko) ko = cout > 0 ? cout : 1;

    // UB model:
    //   yTile: ko * tileS
    //   xVec : 1 * tileS
    //   tmp  : 1 * tileS
    // Bytes per tile ~= (ko + 2) * tileS * 4
    uint32_t maxTileS = UB_BUDGET_BYTES / ((ko + 2u) * 4u);
    maxTileS = AlignDown(maxTileS, ALIGN_ELEMS);
    if (maxTileS < ALIGN_ELEMS) maxTileS = ALIGN_ELEMS;
    if (maxTileS > s) {
        const uint32_t aligned = AlignDown(s, ALIGN_ELEMS);
        maxTileS = (aligned > 0) ? aligned : s;
    }
    uint32_t tileS = (maxTileS == 0) ? 1u : maxTileS;

    const uint32_t numTilesS = CeilDiv(s, tileS);
    const uint32_t numChunksOc = CeilDiv(cout, ko);
    const uint32_t totalTasks = n * numTilesS * numChunksOc;

    uint32_t blockDim = totalTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_cin(cin);
    tiling.set_cout(cout);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_tileS(tileS);
    tiling.set_ko(ko);
    tiling.set_numTilesS(numTilesS);
    tiling.set_numChunksOc(numChunksOc);
    tiling.set_totalTasks(totalTasks);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return GRAPH_FAILED;
    }

    const int64_t n   = x_shape->GetDim(0);
    const int64_t h   = x_shape->GetDim(2);
    const int64_t w   = x_shape->GetDim(3);

    gert::Shape *y_shape = context->GetOutputShape(0);

    y_shape->SetDim(0, n);
    y_shape->SetDim(1, cout);
    y_shape->SetDim(2, h);
    y_shape->SetDim(3, w);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t ww,
                                uint32_t tileS, uint32_t ko,
                                uint32_t numTilesS, uint32_t numChunksOc,
                                uint32_t totalTasks)
    {
        n_ = n; cin_ = cin; cout_ = cout; h_ = h; w_ = ww;
        hw_ = h_ * w_;
        tileS_ = tileS;
        ko_ = ko;
        numTilesS_ = numTilesS;
        numChunksOc_ = numChunksOc;
        totalTasks_ = totalTasks;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * cin_ * hw_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)cout_ * cin_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * cout_ * hw_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileS_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, ko_ * tileS_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQueue_, BUFFER_NUM, tileS_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || hw_ == 0 || tileS_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t taskId = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t stride = blockNum;

        while (taskId < totalTasks_) {
            uint32_t nIdx, tileSIdx, ocChunkIdx;
            DecodeTask(taskId, nIdx, tileSIdx, ocChunkIdx);

            const uint32_t p0 = tileSIdx * tileS_;
            uint32_t curLen = hw_ - p0;
            if (curLen > tileS_) curLen = tileS_;

            const uint32_t oc_start = ocChunkIdx * ko_;
            uint32_t oc_eff = cout_ - oc_start;
            if (oc_eff > ko_) oc_eff = ko_;

            // Allocate and zero y tile buffer
            AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();
            AscendC::Dup(yTile, 0.0f, oc_eff * curLen);

            // Iterate over input channels
            for (uint32_t ic = 0; ic < cin_; ++ic) {
                // Load X vector for this (n, ic) tile
                AscendC::LocalTensor<float> xVec = inQueueX_.AllocTensor<float>();
                const uint64_t xOff = OffsetX(nIdx, ic, p0);
                AscendC::DataCopy(xVec, xGm_[xOff], curLen);
                inQueueX_.EnQue(xVec);
                xVec = inQueueX_.DeQue<float>();

                // For each output channel in this chunk
                for (uint32_t ko = 0; ko < oc_eff; ++ko) {
                    const uint32_t oc = oc_start + ko;
                    const uint64_t wOff = OffsetW(oc, ic);

                    // Load scalar weight w[oc, ic]
                    AscendC::LocalTensor<float> wLoc = scalarQueue_.AllocTensor<float>();
                    AscendC::DataCopy(wLoc, wGm_[wOff], 1);
                    scalarQueue_.EnQue(wLoc);
                    wLoc = scalarQueue_.DeQue<float>();

                    float wScalar = wLoc.GetValue(0);

                    // temp = xVec * wScalar
                    AscendC::LocalTensor<float> tmp = tmpVecQueue_.AllocTensor<float>();
                    AscendC::Muls(tmp, xVec, wScalar, curLen);

                    // y_col += temp
                    AscendC::LocalTensor<float> yCol = yTile[curLen * ko];
                    AscendC::Add(yCol, yCol, tmp, curLen);

                    tmpVecQueue_.FreeTensor(tmp);
                    scalarQueue_.FreeTensor(wLoc);
                }

                inQueueX_.FreeTensor(xVec);
            }

            // Store y tile to GM
            for (uint32_t ko = 0; ko < oc_eff; ++ko) {
                const uint32_t oc = oc_start + ko;
                const uint64_t yOff = OffsetY(nIdx, oc, p0);
                AscendC::LocalTensor<float> yCol = yTile[curLen * ko];
                AscendC::DataCopy(yGm_[yOff], yCol, curLen);
            }

            outQueueY_.FreeTensor(yTile);

            taskId += stride;
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t p) const
    {
        return ((uint64_t)n * cin_ + c) * hw_ + p;
    }
    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t c, uint32_t p) const
    {
        return ((uint64_t)n * cout_ + c) * hw_ + p;
    }
    __aicore__ inline uint64_t OffsetW(uint32_t oc, uint32_t ic) const
    {
        return (uint64_t)oc * cin_ + ic;
    }

    __aicore__ inline void DecodeTask(uint32_t taskId, uint32_t &nIdx, uint32_t &tileSIdx, uint32_t &ocChunkIdx) const
    {
        const uint32_t a = numChunksOc_;
        const uint32_t b = numTilesS_;
        ocChunkIdx = taskId % a;
        const uint32_t t = taskId / a;
        tileSIdx = t % b;
        nIdx = t / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpVecQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0}, h_{0}, w_{0}, hw_{0};
    uint32_t tileS_{0}, ko_{0};
    uint32_t numTilesS_{0}, numChunksOc_{0}, totalTasks_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w,
            tiling_data.tileS, tiling_data.ko,
            tiling_data.numTilesS, tiling_data.numChunksOc,
            tiling_data.totalTasks);
    op.Process();
}
