#include "conv_depthwise_2d_asymmetric_input_square_kernel_custom_tiling.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

// UB model:
//   yLocal: 1 * tileW
//   tLocal: 1 * tileW
//   wLocal: ~16 floats
// => bytes ≈ 8 * tileW + ~64
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape& x_shape = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape& w_shape = context->GetInputShape(1)->GetOriginShape();

    if (x_shape.GetDimNum() != 4 || w_shape.GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N  = static_cast<uint32_t>(x_shape.GetDim(0));
    const uint32_t C  = static_cast<uint32_t>(x_shape.GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape.GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(x_shape.GetDim(3));

    const uint32_t Co = static_cast<uint32_t>(w_shape.GetDim(0)); // should equal C
    const uint32_t K1 = static_cast<uint32_t>(w_shape.GetDim(1)); // should be 1
    const uint32_t KH = static_cast<uint32_t>(w_shape.GetDim(2));
    const uint32_t KW = static_cast<uint32_t>(w_shape.GetDim(3));

    if (Co != C || K1 != 1) {
        return ge::GRAPH_FAILED;
    }

    // stride=(1,1), padding=(0,0), dilation=(1,1)
    if (H < KH || W < KW) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t Ho = H - KH + 1;
    const uint32_t Wo = W - KW + 1;

    uint32_t maxTileW = UB_BUDGET_BYTES / 8; // elements
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;
    if (maxTileW > HARD_CAP_ELEMS) maxTileW = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileW = maxTileW;
    if (Wo < tileW) {
        const uint32_t aligned = AlignDown(Wo, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : Wo;
    }

    const uint64_t totalRows = static_cast<uint64_t>(N) * C * Ho;
    uint32_t blockDim = (totalRows < 8u) ? static_cast<uint32_t>(totalRows) : 8u;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_KH(KH);
    tiling.set_KW(KW);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t KH, uint32_t KW, uint32_t Ho, uint32_t Wo,
                                uint32_t tileW)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        KH_ = KH; KW_ = KW; Ho_ = Ho; Wo_ = Wo;
        tileW_ = tileW;

        const uint32_t xSize = N_ * C_ * H_ * W_;
        const uint32_t wSize = C_ * KH_ * KW_;
        const uint32_t ySize = N_ * C_ * Ho_ * Wo_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(weightQueue_, 1, 16 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (Ho_ == 0 || Wo_ == 0 || tileW_ == 0) return;

        const uint32_t totalRows = N_ * C_ * Ho_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < totalRows; row += blockNum) {
            ComputeRow(row);
        }
    }

private:
    __aicore__ inline void LoadWeights(uint32_t c, float (&wv)[9])
    {
        AscendC::LocalTensor<float> wLocal = weightQueue_.AllocTensor<float>();
        const uint32_t wOff = c * KH_ * KW_;
        AscendC::DataCopy(wLocal, wGm_[wOff], KH_ * KW_);
        // Read back to scalars
        // Assuming LocalTensor provides GetValue
        wv[0] = wLocal.GetValue(0);
        wv[1] = wLocal.GetValue(1);
        wv[2] = wLocal.GetValue(2);
        wv[3] = wLocal.GetValue(3);
        wv[4] = wLocal.GetValue(4);
        wv[5] = wLocal.GetValue(5);
        wv[6] = wLocal.GetValue(6);
        wv[7] = wLocal.GetValue(7);
        wv[8] = wLocal.GetValue(8);
        weightQueue_.FreeTensor(wLocal);
    }

    __aicore__ inline void ComputeRow(uint32_t row)
    {
        const uint32_t ho = row % Ho_;
        const uint32_t tmp = row / Ho_;
        const uint32_t c  = tmp % C_;
        const uint32_t n  = tmp / C_;

        float wv[9];
        LoadWeights(c, wv);

        const uint32_t yRowBase = (((n * C_ + c) * Ho_) + ho) * Wo_;

        for (uint32_t startW = 0; startW < Wo_; startW += tileW_) {
            uint32_t len = Wo_ - startW;
            if (len > tileW_) len = tileW_;

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

            // First kernel position (kh=0, kw=0) initializes yLocal
            {
                const uint32_t xBase = (((n * C_ + c) * H_) + (ho + 0u)) * W_ + (startW + 0u);
                AscendC::DataCopy(tLocal, xGm_[xBase], len);
                AscendC::Muls(yLocal, tLocal, wv[0], len);
            }

            // Remaining positions accumulate
            // kw=1,2 at kh=0
            {
                const uint32_t xBase1 = (((n * C_ + c) * H_) + (ho + 0u)) * W_ + (startW + 1u);
                AscendC::DataCopy(tLocal, xGm_[xBase1], len);
                AscendC::Muls(tLocal, tLocal, wv[1], len);
                AscendC::Add(yLocal, yLocal, tLocal, len);
            }
            {
                const uint32_t xBase2 = (((n * C_ + c) * H_) + (ho + 0u)) * W_ + (startW + 2u);
                AscendC::DataCopy(tLocal, xGm_[xBase2], len);
                AscendC::Muls(tLocal, tLocal, wv[2], len);
                AscendC::Add(yLocal, yLocal, tLocal, len);
            }
            // kh=1, kw=0..2
            {
                const uint32_t xBase3 = (((n * C_ + c) * H_) + (ho + 1u)) * W_ + (startW + 0u);
                AscendC::DataCopy(tLocal, xGm_[xBase3], len);
                AscendC::Muls(tLocal, tLocal, wv[3], len);
                AscendC::Add(yLocal, yLocal, tLocal, len);
            }
            {
                const uint32_t xBase4 = (((n * C_ + c) * H_) + (ho + 1u)) * W_ + (startW + 1u);
                AscendC::DataCopy(tLocal, xGm_[xBase4], len);
                AscendC::Muls(tLocal, tLocal, wv[4], len);
                AscendC::Add(yLocal, yLocal, tLocal, len);
            }
            {
                const uint32_t xBase5 = (((n * C_ + c) * H_) + (ho + 1u)) * W_ + (startW + 2u);
                AscendC::DataCopy(tLocal, xGm_[xBase5], len);
                AscendC::Muls(tLocal, tLocal, wv[5], len);
                AscendC::Add(yLocal, yLocal, tLocal, len);
            }
            // kh=2, kw=0..2
            {
                const uint32_t xBase6 = (((n * C_ + c) * H_) + (ho + 2u)) * W_ + (startW + 0u);
                AscendC::DataCopy(tLocal, xGm_[xBase6], len);
                AscendC::Muls(tLocal, tLocal, wv[6], len);
                AscendC::Add(yLocal, yLocal, tLocal, len);
            }
            {
                const uint32_t xBase7 = (((n * C_ + c) * H_) + (ho + 2u)) * W_ + (startW + 1u);
                AscendC::DataCopy(tLocal, xGm_[xBase7], len);
                AscendC::Muls(tLocal, tLocal, wv[7], len);
                AscendC::Add(yLocal, yLocal, tLocal, len);
            }
            {
                const uint32_t xBase8 = (((n * C_ + c) * H_) + (ho + 2u)) * W_ + (startW + 2u);
                AscendC::DataCopy(tLocal, xGm_[xBase8], len);
                AscendC::Muls(tLocal, tLocal, wv[8], len);
                AscendC::Add(yLocal, yLocal, tLocal, len);
            }

            tmpQueue_.FreeTensor(tLocal);
            outQueueY_.EnQue<float>(yLocal);

            // Store
            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            const uint32_t yBase = yRowBase + startW;
            AscendC::DataCopy(yGm_[yBase], yOut, len);
            outQueueY_.FreeTensor(yOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t KH_{0}, KW_{0}, Ho_{0}, Wo_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymmetricInputSquareKernelCustom op;
    op.Init(x, w, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.KH, tiling_data.KW, tiling_data.Ho, tiling_data.Wo,
            tiling_data.tileW);
    op.Process();
}

at::Tensor conv_dw2d_asym_sq_kernel_impl_npu(const at::Tensor& x, const at::Tensor& w) {

    TORCH_CHECK(w.dim() == 4, \"w must be 4D [C,1,KH,KW]\");

    auto C = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);
    auto KH = w.size(2);
    auto KW = w.size(3);

    auto Ho = H - KH + 1;
    auto Wo = W - KW + 1;

    at::Tensor result = at::empty({N, C, Ho, Wo}, x.options());
    EXEC_NPU_CMD(aclnnConvDepthwise2dAsymmetricInputSquareKernelCustom, x, w, result);
