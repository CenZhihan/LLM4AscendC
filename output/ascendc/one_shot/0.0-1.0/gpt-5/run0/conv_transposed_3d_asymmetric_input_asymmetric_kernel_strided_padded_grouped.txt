#include "kernel_operator.h"

class KernelConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputAsymmetricKernelStridedPaddedGroupedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t KD, uint32_t KH, uint32_t KW,
                                uint32_t OD, uint32_t OH, uint32_t OW,
                                uint32_t sD, uint32_t sH, uint32_t sW,
                                uint32_t pD, uint32_t pH, uint32_t pW,
                                uint32_t outPadD, uint32_t outPadH, uint32_t outPadW,
                                uint32_t groups)
    {
        x_ = (__gm__ float*)x;
        w_ = (__gm__ float*)w;
        y_ = (__gm__ float*)y;

        N_ = N; Cin_ = Cin; Cout_ = Cout;
        D_ = D; H_ = H; W_ = W;
        KD_ = KD; KH_ = KH; KW_ = KW;
        OD_ = OD; OH_ = OH; OW_ = OW;
        sD_ = sD; sH_ = sH; sW_ = sW;
        pD_ = pD; pH_ = pH; pW_ = pW;
        outPadD_ = outPadD; outPadH_ = outPadH; outPadW_ = outPadW;
        groups_ = groups;

        // Precompute strides (64-bit to avoid overflow on large tensors)
        sx_w_ = 1ull;
        sx_h_ = (uint64_t)W_;
        sx_d_ = (uint64_t)H_ * (uint64_t)W_;
        sx_c_ = (uint64_t)D_ * (uint64_t)H_ * (uint64_t)W_;
        sx_n_ = (uint64_t)Cin_ * (uint64_t)D_ * (uint64_t)H_ * (uint64_t)W_;

        CoG_ = Cout_ / groups_;
        // weight: [Cin, CoG, KD, KH, KW]
        sw_kw_ = 1ull;
        sw_kh_ = (uint64_t)KW_;
        sw_kd_ = (uint64_t)KH_ * (uint64_t)KW_;
        sw_cog_= (uint64_t)KD_ * (uint64_t)KH_ * (uint64_t)KW_;
        sw_cin_= (uint64_t)CoG_ * (uint64_t)KD_ * (uint64_t)KH_ * (uint64_t)KW_;

        sy_w_ = 1ull;
        sy_h_ = (uint64_t)OW_;
        sy_d_ = (uint64_t)OH_ * (uint64_t)OW_;
        sy_c_ = (uint64_t)OD_ * (uint64_t)OH_ * (uint64_t)OW_;
        sy_n_ = (uint64_t)Cout_ * (uint64_t)OD_ * (uint64_t)OH_ * (uint64_t)OW_;
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0) return;

        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();
        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();

        // Map each block to a unique (n, g) pair
        const uint32_t perNg = N_ * groups_;
        if (blockIdx >= perNg) return;

        const uint32_t n = blockIdx / groups_;
        const uint32_t g = blockIdx % groups_;

        // Zero initialize this (n,g) slice of output y across its group channels
        for (uint32_t co = 0; co < CoG_; ++co) {
            const uint32_t oc = g * CoG_ + co;
            for (uint32_t od = 0; od < OD_; ++od) {
                for (uint32_t oh = 0; oh < OH_; ++oh) {
                    for (uint32_t ow = 0; ow < OW_; ++ow) {
                        uint64_t idx_y = ((uint64_t)n * sy_n_) + ((uint64_t)oc * sy_c_) + ((uint64_t)od * sy_d_) + ((uint64_t)oh * sy_h_) + (uint64_t)ow;
                        y_[idx_y] = 0.0f;
                    }
                }
            }
        }

        // Accumulate contributions for (n,g) slice
        const uint32_t CiG = Cin_ / groups_;
        for (uint32_t ci = 0; ci < CiG; ++ci) {
            const uint32_t cabs = g * CiG + ci;
            for (uint32_t d = 0; d < D_; ++d) {
                for (uint32_t h = 0; h < H_; ++h) {
                    for (uint32_t w = 0; w < W_; ++w) {
                        // Load input scalar x[n, cabs, d, h, w]
                        uint64_t idx_x = ((uint64_t)n * sx_n_) + ((uint64_t)cabs * sx_c_) + ((uint64_t)d * sx_d_) + ((uint64_t)h * sx_h_) + (uint64_t)w;
                        float xval = x_[idx_x];

                        // For each kernel position
                        for (uint32_t kd = 0; kd < KD_; ++kd) {
                            int64_t od = (int64_t)d * (int64_t)sD_ - (int64_t)pD_ + (int64_t)kd;
                            if (od < 0 || od >= (int64_t)OD_) continue;
                            for (uint32_t kh = 0; kh < KH_; ++kh) {
                                int64_t oh = (int64_t)h * (int64_t)sH_ - (int64_t)pH_ + (int64_t)kh;
                                if (oh < 0 || oh >= (int64_t)OH_) continue;
                                for (uint32_t kw = 0; kw < KW_; ++kw) {
                                    int64_t ow = (int64_t)w * (int64_t)sW_ - (int64_t)pW_ + (int64_t)kw;
                                    if (ow < 0 || ow >= (int64_t)OW_) continue;

                                    // For each out channel within group
                                    for (uint32_t co = 0; co < CoG_; ++co) {
                                        // weight[cabs, co, kd, kh, kw]
                                        uint64_t idx_w = ((uint64_t)cabs * sw_cin_) + ((uint64_t)co * sw_cog_) + ((uint64_t)kd * sw_kd_) + ((uint64_t)kh * sw_kh_) + (uint64_t)kw;
                                        float wval = w_[idx_w];

                                        const uint32_t oc = g * CoG_ + co;
                                        uint64_t idx_y = ((uint64_t)n * sy_n_) + ((uint64_t)oc * sy_c_) + ((uint64_t)od * sy_d_) + ((uint64_t)oh * sy_h_) + (uint64_t)ow;
                                        y_[idx_y] += xval * wval;
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }
    }

private:
    __gm__ float* x_ {nullptr};
    __gm__ float* w_ {nullptr};
    __gm__ float* y_ {nullptr};

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t D_{0}, H_{0}, W_{0};
    uint32_t KD_{0}, KH_{0}, KW_{0};
    uint32_t OD_{0}, OH_{0}, OW_{0};
    uint32_t sD_{0}, sH_{0}, sW_{0};
    uint32_t pD_{0}, pH_{0}, pW_{0};
    uint32_t outPadD_{0}, outPadH_{0}, outPadW_{0};
    uint32_t groups_{1};
    uint32_t CoG_{0};

    // strides (64-bit)
    uint64_t sx_w_{0}, sx_h_{0}, sx_d_{0}, sx_c_{0}, sx_n_{0};
    uint64_t sw_kw_{0}, sw_kh_{0}, sw_kd_{0}, sw_cog_{0}, sw_cin_{0};
    uint64_t sy_w_{0}, sy_h_{0}, sy_d_{0}, sy_c_{0}, sy_n_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_strided_padded_grouped_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)

        tiling_data.N, tiling_data.Cin, tiling_data.Cout,

        tiling_data.padD, tiling_data.padH, tiling_data.padW,
        tiling_data.outPadD, tiling_data.outPadH, tiling_data.outPadW,
        tiling_data.groups
    );
    op.Process();
}
