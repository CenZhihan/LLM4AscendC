kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulScaleResidualAddClampLogSumExpMishCustom {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR scaling, GM_ADDR cmin, GM_ADDR cmax,
                                GM_ADDR y,
                                uint32_t N, uint32_t K, uint32_t C,
                                uint32_t tileK, uint32_t tileC)
    {
        N_ = N; K_ = K; C_ = C;
        tileK_ = tileK; tileC_ = tileC;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * (uint64_t)K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)C_ * (uint64_t)K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, C_);
        sGm_.SetGlobalBuffer((__gm__ float*)scaling, 1);
        cminGm_.SetGlobalBuffer((__gm__ float*)cmin, 1);
        cmaxGm_.SetGlobalBuffer((__gm__ float*)cmax, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_);

        // Queues for small working buffers
        pipe_.InitBuffer(xChunkQ_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(wChunkQ_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(accQ_, 1, tileC_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, 1, 4 * sizeof(float)); // scaling, cmin, cmax, one extra
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || K_ == 0 || C_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        // Load scalar params (scaling, clamp min/max)
        float s_val = 0.f, cmin_val = -INFINITY, cmax_val = INFINITY;
        {
            AscendC::LocalTensor<float> scalars = scalarQ_.AllocTensor<float>();
            // copy scaling, cmin, cmax into local buffer
            AscendC::DataCopy(scalars, sGm_[0], 1);
            AscendC::DataCopy(scalars[1], cminGm_[0], 1);
            AscendC::DataCopy(scalars[2], cmaxGm_[0], 1);
            s_val = scalars.GetValue(0);
            cmin_val = scalars.GetValue(1);
            cmax_val = scalars.GetValue(2);
            scalarQ_.FreeTensor(scalars);
        }
        const float scale_eff = 1.0f + s_val;

        // Grid-stride over rows
        for (uint32_t row = blockIdx; row < N_; row += blockNum) {
            // Online log-sum-exp across output channels without storing full row
            float M = -INFINITY;
            float S = 0.0f;

            for (uint32_t c0 = 0; c0 < C_; c0 += tileC_) {
                const uint32_t cLen = AscendC::min(tileC_, C_ - c0);
                AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
                AscendC::Dup(acc, 0.0f, cLen);

                // Compute acc[j] = dot(x[row,:], w[c0+j,:]) for j in [0, cLen)
                for (uint32_t k0 = 0; k0 < K_; k0 += tileK_) {
                    const uint32_t kLen = AscendC::min(tileK_, K_ - k0);

                    AscendC::LocalTensor<float> xChunk = xChunkQ_.AllocTensor<float>();
                    AscendC::DataCopy(xChunk, xGm_[row * K_ + k0], kLen);

                    for (uint32_t j = 0; j < cLen; ++j) {
                        AscendC::LocalTensor<float> wChunk = wChunkQ_.AllocTensor<float>();
                        AscendC::DataCopy(wChunk, wGm_[(c0 + j) * K_ + k0], kLen);

                        // Dot product
                        float sumv = 0.0f;
                        for (uint32_t t = 0; t < kLen; ++t) {
                            const float xv = xChunk.GetValue(t);
                            const float ww = wChunk.GetValue(t);
                            sumv = sumv + xv * ww;
                        }
                        // acc[j] += sumv
                        const float prev = acc.GetValue(j);
                        acc.SetValue(j, prev + sumv);

                        wChunkQ_.FreeTensor(wChunk);
                    }
                    xChunkQ_.FreeTensor(xChunk);
                }

                // Add bias and fused scale+residual: y = clamp(scale_eff * acc, cmin, cmax)
                // Note: original x*s + x = (1+s)*x applied to linear output
                for (uint32_t j = 0; j < cLen; ++j) {
                    float v = acc.GetValue(j);
                    // add bias
                    float bj = bGm_[c0 + j];
                    v = v + bj;
                    // fused scale-residual
                    v = v * scale_eff;
                    // clamp
                    if (v < cmin_val) v = cmin_val;
                    if (v > cmax_val) v = cmax_val;
                    acc.SetValue(j, v);
                }

                // Tile log-sum-exp partials
                float m_t = -INFINITY;
                for (uint32_t j = 0; j < cLen; ++j) {
                    const float v = acc.GetValue(j);
                    if (v > m_t) m_t = v;
                }
                float S_t = 0.0f;
                for (uint32_t j = 0; j < cLen; ++j) {
                    const float v = acc.GetValue(j);
                    S_t += expf(v - m_t);
                }
                // Combine with running M,S
                const float M_new = (M > m_t) ? M : m_t;
                const float S_new = S * expf(M - M_new) + S_t * expf(m_t - M_new);
                M = M_new;
                S = S_new;

                accQ_.FreeTensor(acc);
            }

            // Final LSE and Mish
            float lse = M + logf(S);
            // softplus stable: log1p(exp(-|x|)) + max(x, 0)
            const float ax = fabsf(lse);
            const float sp = log1pf(expf(-ax)) + ((lse > 0.0f) ? lse : 0.0f);
            const float mish = lse * tanhf(sp);
            const float outv = lse * mish;

            yGm_[row] = outv;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xChunkQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wChunkQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> sGm_;
    AscendC::GlobalTensor<float> cminGm_;
    AscendC::GlobalTensor<float> cmaxGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, K_{0}, C_{0};
    uint32_t tileK_{0}, tileC_{0};
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scaling, GM_ADDR clamp_min, GM_ADDR clamp_max,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScaleResidualAddClampLogSumExpMishCustom op;
    op.Init(x, weight, bias, scaling, clamp_min, clamp_max, y,
            tiling_data.nRows, tiling_data.nColsIn, tiling_data.nColsOut,
            tiling_data.tileK, tiling_data.tileC);
    op.Process();
}
