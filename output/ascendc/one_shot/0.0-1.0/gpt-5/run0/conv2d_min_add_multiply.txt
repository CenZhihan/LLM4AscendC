static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMinAddMultiplyCustomTilingData tiling;

    // Inputs: x [N,Cin,H,W], weight [Cout,Cin,Kh,Kw], conv_bias [Cout] optional, post_bias [Cout,1,1]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetOriginShape().GetDimNum() != 4 || w_shape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t Kh   = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    const uint32_t Kw   = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(3));

    if (Kh == 0 || Kw == 0 || H < Kh || W < Kw) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const uint32_t Hout = H - Kh + 1;
    const uint32_t Wout = W - Kw + 1;
    const uint64_t totalOut = static_cast<uint64_t>(N) * Cout * Hout * Wout;

    // Attributes
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *constantValue = attrs->GetAttrPointer<float>(0);
    const float *scalingFactor = attrs->GetAttrPointer<float>(1);

    // Block dim heuristic
    uint32_t blockDim = 16;
    if (totalOut > 0) {
        uint32_t target = CeilDiv(totalOut, 4096);
        if (target < 8u) target = 8u;
        if (target > 24u) target = 24u;
        blockDim = target;
    }
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Cout(Cout);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);
    tiling.set_Hout(Hout);
    tiling.set_Wout(Wout);
    tiling.set_totalOut(totalOut);
    tiling.set_blockDim(blockDim);
    tiling.set_constantValue(*constantValue);
    tiling.set_scalingFactor(*scalingFactor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dMinAddMultiplyCustom {
public:
    __aicore__ inline KernelConv2dMinAddMultiplyCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR conv_b, GM_ADDR post_b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t Hout, uint32_t Wout,
                                float constantValue, float scalingFactor)
    {
        N_ = N; Cin_ = Cin; H_ = H; W_ = W;
        Cout_ = Cout; Kh_ = Kh; Kw_ = Kw;
        Hout_ = Hout; Wout_ = Wout;
        constV_ = constantValue;
        scaleV_ = scalingFactor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_)*Cin_*H_*W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Cout_)*Cin_*Kh_*Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_)*Cout_*Hout_*Wout_);
        pbGm_.SetGlobalBuffer((__gm__ float*)post_b, Cout_);
        convBiasPtr_ = (__gm__ float*)conv_b;

        pipe_.InitBuffer(accQ_, 1, sizeof(float));      // accumulator
        pipe_.InitBuffer(tmpQ_, 1, sizeof(float));      // temp product
        pipe_.InitBuffer(xQ_, 1, sizeof(float));        // x scalar
        pipe_.InitBuffer(wQ_, 1, sizeof(float));        // w scalar
        pipe_.InitBuffer(auxQ_, 1, sizeof(float));      // bias/post_bias scalar
    }

    __aicore__ inline void Process()
    {
        const uint64_t totalOut = static_cast<uint64_t>(N_) * Cout_ * Hout_ * Wout_;
        if (totalOut == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t idx = blockIdx; idx < totalOut; idx += blockNum) {
            // Unravel index: [N, Cout, Hout, Wout]
            uint64_t t = idx;
            const uint32_t co = static_cast<uint32_t>(t % Cout_); t /= Cout_;
            const uint32_t wo = static_cast<uint32_t>(t % Wout_); t /= Wout_;
            const uint32_t ho = static_cast<uint32_t>(t % Hout_); t /= Hout_;
            const uint32_t n  = static_cast<uint32_t>(t);

            // acc = 0
            AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
            AscendC::Duplicate(acc, 0.0f, 1);

            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const uint32_t ih = ho + kh;
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const uint32_t iw = wo + kw;

                        const uint64_t xOff = (((static_cast<uint64_t>(n) * Cin_ + ci) * H_ + ih) * W_ + iw);
                        const uint64_t wOff = (((static_cast<uint64_t>(co) * Cin_ + ci) * Kh_ + kh) * Kw_ + kw);

                        AscendC::LocalTensor<float> xLoc = xQ_.AllocTensor<float>();
                        AscendC::LocalTensor<float> wLoc = wQ_.AllocTensor<float>();
                        AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();

                        AscendC::DataCopy(xLoc, xGm_[xOff], 1);
                        AscendC::DataCopy(wLoc, wGm_[wOff], 1);
                        AscendC::Mul(tmp, xLoc, wLoc, 1);
                        AscendC::Add(acc, acc, tmp, 1);

                        tmpQ_.FreeTensor(tmp);
                        wQ_.FreeTensor(wLoc);
                        xQ_.FreeTensor(xLoc);
                    }
                }
            }

            // Add conv bias if provided
            if (convBiasPtr_ != nullptr) {
                AscendC::LocalTensor<float> bLoc = auxQ_.AllocTensor<float>();
                AscendC::DataCopy(bLoc, AscendC::GlobalTensor<float>((__gm__ float*)convBiasPtr_, Cout_)[co], 1);
                AscendC::Add(acc, acc, bLoc, 1);
                auxQ_.FreeTensor(bLoc);
            }

            // cap = min(acc, constV_)
            AscendC::Mins(acc, acc, constV_, 1);

            // add post_bias[co]
            AscendC::LocalTensor<float> pbLoc = auxQ_.AllocTensor<float>();
            AscendC::DataCopy(pbLoc, pbGm_[co], 1);
            AscendC::Add(acc, acc, pbLoc, 1);
            auxQ_.FreeTensor(pbLoc);

            // multiply by scaling factor
            AscendC::Muls(acc, acc, scaleV_, 1);

            // write out
            const uint64_t yOff = (((static_cast<uint64_t>(n) * Cout_ + co) * Hout_ + ho) * Wout_ + wo);
            AscendC::DataCopy(yGm_[yOff], acc, 1);

            accQ_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> auxQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> pbGm_;
    __gm__ float* convBiasPtr_ {nullptr};

    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, Kh_{0}, Kw_{0};
    uint32_t Hout_{0}, Wout_{0};
    float constV_{0.0f};
    float scaleV_{1.0f};
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR post_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinAddMultiplyCustom op;
    op.Init(x, weight, conv_bias, post_bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.H, tiling_data.W,
            tiling_data.Cout, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Hout, tiling_data.Wout,
            tiling_data.constantValue, tiling_data.scalingFactor);
    op.Process();
}

#include <torch/library.h>

static inline at::IntArrayRef conv2d_out_shape(const at::Tensor& x, const at::Tensor& w) {
    // x: [N,Cin,H,W], w: [Cout,Cin,Kh,Kw], stride=1, padding=0, dilation=1
    const int64_t N = x.size(0);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);
    const int64_t Cout = w.size(0);
    const int64_t Kh = w.size(2);
    const int64_t Kw = w.size(3);
    const int64_t Hout = H - Kh + 1;
    const int64_t Wout = W - Kw + 1;
    static thread_local std::vector<int64_t> shape;
    shape = {N, Cout, Hout, Wout};
    return at::IntArrayRef(shape);
}

at::Tensor conv2d_min_add_multiply_impl_npu(const at::Tensor& x,
                                            const at::Tensor& weight,
                                            c10::optional<at::Tensor> conv_bias_opt,
                                            const at::Tensor& post_bias,
                                            double constant_value,
                                            double scaling_factor) {
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, "x and weight must be 4D");
    at::Tensor result = at::empty(conv2d_out_shape(x, weight), x.options());
    if (conv_bias_opt.has_value()) {
        EXEC_NPU_CMD(aclnnConv2dMinAddMultiplyCustom, x, weight, conv_bias_opt.value(), post_bias, constant_value, scaling_factor, result);
    } else {
        at::Tensor undef_bias;
        EXEC_NPU_CMD(aclnnConv2dMinAddMultiplyCustom, x, weight, undef_bias, post_bias, constant_value, scaling_factor, result);
    }
    return result;
}
