static inline uint32_t CeilDiv(uint64_t a, uint32_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgminOverADimensionCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> dims = x_shape->GetOriginShape().GetDims();
    const int32_t rank = static_cast<int32_t>(dims.size());
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }

    // dim attribute (we will reduce along the last dimension effectively; the front-end may permute)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimPtr != nullptr) ? *dimPtr : -1;
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) return ge::GRAPH_FAILED;

    const uint64_t reduceLen = static_cast<uint64_t>(dims[dim]);
    uint64_t totalElems = 1ull;
    for (int i = 0; i < rank; ++i) {
        totalElems *= static_cast<uint64_t>(dims[i]);
    }
    if (reduceLen == 0) return ge::GRAPH_FAILED;
    const uint64_t totalRows64 = totalElems / reduceLen;

    // Choose blockDim conservatively based on rows
    uint32_t blockDim = static_cast<uint32_t>(totalRows64);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(static_cast<uint32_t>(totalRows64));
    tiling.set_rowLength(static_cast<uint32_t>(reduceLen));
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const ge::TensorDesc *x_desc = context->GetInputDesc(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    std::vector<int64_t> dims = x_shape->GetDims();
    const int32_t rank = static_cast<int32_t>(dims.size());

    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimPtr != nullptr) ? *dimPtr : -1;
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) return GRAPH_FAILED;

    std::vector<int64_t> out_dims;
    out_dims.reserve(rank > 0 ? rank - 1 : 0);
    for (int i = 0; i < rank; ++i) {
        if (i != dim) out_dims.push_back(dims[i]);
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDims(out_dims);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUFFERS = 2;
constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core conservative
constexpr uint32_t ALIGN_ELEMS = 1024;
constexpr uint32_t HARD_CAP_ELEMS = 32768;

class KernelArgminOverADimensionCustom {
public:
    __aicore__ inline KernelArgminOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows,
                                uint32_t rowLength)
    {
        totalRows_ = totalRows;
        rowLen_ = rowLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)totalRows_ * (uint64_t)rowLen_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, totalRows_);

        // Determine a tile length within UB budget (float32)
        uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
        if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
        maxTile = (maxTile / ALIGN_ELEMS) * ALIGN_ELEMS;
        if (maxTile == 0) maxTile = ALIGN_ELEMS;

        tileLen_ = rowLen_ < maxTile ? rowLen_ : maxTile;

        pipe_.InitBuffer(inQueue_, IN_BUFFERS, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outIdxQueue_, 1, sizeof(int64_t));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || rowLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            ComputeRow(row);
        }
    }

private:
    __aicore__ inline void LoadSlice(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueue_.EnQue(xLocal);
    }

    __aicore__ inline void StoreIndex(uint32_t row, int64_t idx)
    {
        AscendC::LocalTensor<int64_t> yLocal = outIdxQueue_.AllocTensor<int64_t>();
        // Set the single element
        yLocal.SetValue(0, idx);
        outIdxQueue_.EnQue(yLocal);

        AscendC::LocalTensor<int64_t> yDeq = outIdxQueue_.DeQue<int64_t>();
        AscendC::DataCopy(yGm_[row], yDeq, 1);
        outIdxQueue_.FreeTensor(yDeq);
    }

    __aicore__ inline void ComputeRow(uint32_t row)
    {
        const uint64_t base = static_cast<uint64_t>(row) * static_cast<uint64_t>(rowLen_);
        float bestVal = AscendC::numeric_limits<float>::infinity();
        int64_t bestIdx = 0;

        uint32_t pos = 0;
        // Prefetch first slice
        uint32_t cur = rowLen_ - pos;
        if (cur > tileLen_) cur = tileLen_;
        LoadSlice(base + pos, cur);

        while (pos < rowLen_) {
            AscendC::LocalTensor<float> xLocal = inQueue_.DeQue<float>();

            // Iterate over current slice
            for (uint32_t i = 0; i < cur; ++i) {
                float v = xLocal.GetValue(i);
                if (v < bestVal) {
                    bestVal = v;
                    bestIdx = static_cast<int64_t>(pos + i);
                }
            }

            inQueue_.FreeTensor(xLocal);
            pos += cur;
            if (pos >= rowLen_) break;

            // Prefetch next slice
            cur = rowLen_ - pos;
            if (cur > tileLen_) cur = tileLen_;
            LoadSlice(base + pos, cur);
        }

        StoreIndex(row, bestIdx);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFERS> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outIdxQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t totalRows_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgminOverADimensionCustom op;
    op.Init(x, y, tiling_data.totalRows, tiling_data.rowLength);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor argmin_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    // Output shape = input shape with the specified dim removed
    std::vector<int64_t> out_sizes;
    out_sizes.reserve(self.dim() > 0 ? self.dim() - 1 : 0);
    int64_t rank = self.dim();
    int64_t d = dim;
    if (d < 0) d += rank;
    TORCH_CHECK(d >= 0 && d < rank, "dim out of range");
    for (int64_t i = 0; i < rank; ++i) {
        if (i != d) out_sizes.push_back(self.size(i));
    }
    at::Tensor result = at::empty(out_sizes, self.options().dtype(at::kLong));
    EXEC_NPU_CMD(aclnnArgminOverADimensionCustom, self, (int)dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmin_over_a_dimension_custom", &argmin_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmin_over_a_dimension_custom", &argmin_over_a_dimension_impl_npu,
          "Argmin over a given dimension (ArgminOverADimensionCustom on NPU)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'argmin_over_a_dimension_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, dim: int):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Move the reduction dimension to the last for better kernel memory access.
        rank = x.dim()
        dim = self.dim
        if dim < 0:
            dim += rank
        assert 0 <= dim < rank, "dim out of range"
        if dim != rank - 1:
            perm = [i for i in range(rank) if i != dim] + [dim]
            x = x.permute(perm)
            dim_for_op = x.dim() - 1
        else:
            dim_for_op = dim
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, dim_for_op)
'''
