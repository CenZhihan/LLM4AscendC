host_operator_src="""
#include "max_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   outQueue: 2 * tileInner
//   tmpRow:   1 * tileInner
// Total floats = 3 * tileInner => bytes = 12 * tileInner
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative
static constexpr uint32_t ALIGN_ELEMS = 1024;            // favor aligned DMA
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // avoid very large tiles

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) { return static_cast<uint32_t>((a + b - 1ull) / b); }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxReductionOverADimensionCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();

    const int32_t rank = static_cast<int32_t>(origin.GetDimNum());
    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int32_t dim = static_cast<int32_t>(*dimPtr);
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) {
        return ge::GRAPH_FAILED;
    }

    // Collect dimensions
    uint64_t outerNum = 1;
    for (int32_t i = 0; i < dim; ++i) {
        outerNum *= static_cast<uint64_t>(origin.GetDim(i));
    }
    uint64_t axisLen = static_cast<uint64_t>(origin.GetDim(dim));
    uint64_t innerSize = 1;
    for (int32_t i = dim + 1; i < rank; ++i) {
        innerSize *= static_cast<uint64_t>(origin.GetDim(i));
    }

    const uint64_t xTotalLen = outerNum * axisLen * innerSize;
    const uint64_t yTotalLen = outerNum * innerSize;

    // Compute tileInner from UB budget: bytes = 12 * tileInner
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileInner = maxTile;
    if (innerSize < tileInner) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(innerSize), ALIGN_ELEMS);
        tileInner = (aligned > 0) ? aligned : static_cast<uint32_t>(innerSize);
    }
    if (tileInner == 0) tileInner = 1;

    // BlockDim based on outer parallelism
    uint32_t blockDim = static_cast<uint32_t>(outerNum);
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_xTotalLength(static_cast<uint32_t>(xTotalLen));
    tiling.set_yTotalLength(static_cast<uint32_t>(yTotalLen));
    tiling.set_outerNum(static_cast<uint32_t>(outerNum));
    tiling.set_axisLen(static_cast<uint32_t>(axisLen));
    tiling.set_innerSize(static_cast<uint32_t>(innerSize));
    tiling.set_outerStride(static_cast<uint32_t>(axisLen * innerSize));
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin = x_shape->GetOriginShape();
    gert::Shape *y_shape = context->GetOutputShape(0);

    const int32_t rank = static_cast<int32_t>(origin.GetDimNum());
    const int64_t dimAttr = *context->GetAttrs()->GetAttrPointer<int64_t>(0);
    int32_t dim = static_cast<int32_t>(dimAttr);
    int32_t redDim = dim;
    if (redDim < 0) redDim += rank;

    // Build reduced shape
    std::vector<int64_t> out_dims;
    out_dims.reserve(std::max(1, rank - 1));
    for (int32_t i = 0; i < rank; ++i) {
        if (i == redDim) continue;
        out_dims.push_back(origin.GetDim(i));
    }
    if (out_dims.empty()) {
        // For scalar output, represent as shape [1]
        out_dims.push_back(1);
    }

    y_shape->SetOriginShape(ge::Shape(out_dims));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t OUT_BUFFER_NUM = 2;

class KernelMaxReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMaxReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t xTotalLength,
                                uint32_t yTotalLength,
                                uint32_t outerNum,
                                uint32_t axisLen,
                                uint32_t innerSize,
                                uint32_t outerStride,
                                uint32_t tileInner)
    {
        xTotalLen_ = xTotalLength;
        yTotalLen_ = yTotalLength;
        outerNum_ = outerNum;
        axisLen_ = axisLen;
        innerSize_ = innerSize;
        outerStride_ = outerStride;
        tileInner_ = tileInner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xTotalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yTotalLen_);

        pipe_.InitBuffer(outQueueY_, OUT_BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (xTotalLen_ == 0 || yTotalLen_ == 0 || axisLen_ == 0 || innerSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t outer = blockIdx; outer < outerNum_; outer += blockNum) {
            const uint32_t base = outer * outerStride_;
            // iterate inner columns in tiles
            for (uint32_t colBase = 0; colBase < innerSize_; colBase += tileInner_) {
                uint32_t curLen = innerSize_ - colBase;
                if (curLen > tileInner_) curLen = tileInner_;

                // yLocal holds current maxima; initialize with first row
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, xGm_[base + colBase], curLen);

                for (uint32_t r = 1; r < axisLen_; ++r) {
                    AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::DataCopy(tLocal, xGm_[base + r * innerSize_ + colBase], curLen);
                    AscendC::Max(yLocal, yLocal, tLocal, curLen);
                    tmpQueue_.FreeTensor(tLocal);
                }

                outQueueY_.EnQue(yLocal);

                // write out
                AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[outer * innerSize_ + colBase], yOut, curLen);
                outQueueY_.FreeTensor(yOut);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t xTotalLen_ {0};
    uint32_t yTotalLen_ {0};
    uint32_t outerNum_ {0};
    uint32_t axisLen_ {0};
    uint32_t innerSize_ {0};
    uint32_t outerStride_ {0};
    uint32_t tileInner_ {0};
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.xTotalLength,
            tiling_data.yTotalLength,
            tiling_data.outerNum,
            tiling_data.axisLen,
            tiling_data.innerSize,
            tiling_data.outerStride,
            tiling_data.tileInner);
    op.Process();
}
