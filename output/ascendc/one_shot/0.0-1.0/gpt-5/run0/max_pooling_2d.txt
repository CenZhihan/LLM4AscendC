host_operator_src="""

#include "max_pooling_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint64_t a, uint32_t b) {
    return static_cast<uint32_t>((a + b - 1ULL) / static_cast<uint64_t>(b));
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling2dCustomTilingData tiling;

    const gert::Shape& in_shape = context->GetInputShape(0)->GetOriginShape();
    if (in_shape.GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(in_shape.GetDim(0));
    const uint32_t C = static_cast<uint32_t>(in_shape.GetDim(1));
    const uint32_t H = static_cast<uint32_t>(in_shape.GetDim(2));
    const uint32_t W = static_cast<uint32_t>(in_shape.GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernel_ptr  = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride_ptr  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding_ptr = attrs->GetAttrPointer<int32_t>(2);

    const uint32_t K = static_cast<uint32_t>(*kernel_ptr);
    const uint32_t S = static_cast<uint32_t>(*stride_ptr);
    const uint32_t P = static_cast<uint32_t>(*padding_ptr);

    // Output dims following PyTorch MaxPool2d convention
    const int64_t outH_ll = (static_cast<int64_t>(H) + 2LL * static_cast<int64_t>(P) - static_cast<int64_t>(K)) / static_cast<int64_t>(S) + 1LL;
    const int64_t outW_ll = (static_cast<int64_t>(W) + 2LL * static_cast<int64_t>(P) - static_cast<int64_t>(K)) / static_cast<int64_t>(S) + 1LL;
    const uint32_t outH = outH_ll > 0 ? static_cast<uint32_t>(outH_ll) : 0u;
    const uint32_t outW = outW_ll > 0 ? static_cast<uint32_t>(outW_ll) : 0u;

    const uint64_t totalOut64 = static_cast<uint64_t>(N) * C * outH * outW;
    const uint32_t totalOut = static_cast<uint32_t>(totalOut64);

    // Choose a conservative tile size in number of output elements
    // This kernel processes outputs scalar-wise, so UB usage is minimal.
    uint32_t tileOut = 2048u;
    if (tileOut > totalOut && totalOut > 0) tileOut = totalOut;

    uint32_t numTiles = (tileOut == 0) ? 1u : CeilDiv(totalOut64, tileOut);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_kernel(K);
    tiling.set_stride(S);
    tiling.set_padding(P);
    tiling.set_totalOut(totalOut);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape& in_shape = context->GetInputShape(0)->GetOriginShape();
    if (in_shape.GetDimNum() != 4) {
        return GRAPH_FAILED;
    }

    const ge::RuntimeAttrs* attrs = context->GetAttrs();
    const int32_t *kernel_ptr  = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride_ptr  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding_ptr = attrs->GetAttrPointer<int32_t>(2);

    const int64_t N = in_shape.GetDim(0);
    const int64_t C = in_shape.GetDim(1);
    const int64_t H = in_shape.GetDim(2);
    const int64_t W = in_shape.GetDim(3);

    const int64_t K = *kernel_ptr;
    const int64_t S = *stride_ptr;
    const int64_t P = *padding_ptr;

    const int64_t outH = (H + 2 * P - K) / S + 1;
    const int64_t outW = (W + 2 * P - K) / S + 1;

    gert::Shape* y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, outH > 0 ? outH : 0);
    y_shape->SetDim(3, outW > 0 ? outW : 0);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""

#include "kernel_operator.h"
#include <cfloat>

class KernelMaxPooling2dCustom {
public:
    __aicore__ inline KernelMaxPooling2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t outH, uint32_t outW,
                                uint32_t kernel, uint32_t stride, uint32_t padding,
                                uint32_t totalOut, uint32_t tileOut)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        outH_ = outH; outW_ = outW;
        K_ = kernel; S_ = stride; P_ = padding;
        totalOut_ = totalOut; tileOut_ = (tileOut == 0) ? 1u : tileOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_)*C_*H_*W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_)*C_*outH_*outW_);

        // We process scalar values; minimal UB is required.
        pipe_.InitBuffer(tmpQueue_, 2, sizeof(float));     // for acc and temp
        pipe_.InitBuffer(outQueue_, 1, sizeof(float));     // for output staging
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint64_t strideElems = static_cast<uint64_t>(blockNum) * static_cast<uint64_t>(tileOut_);
        uint64_t base = static_cast<uint64_t>(blockIdx) * static_cast<uint64_t>(tileOut_);
        while (base < totalOut_) {
            uint32_t cur = static_cast<uint32_t>(totalOut_ - base);
            if (cur > tileOut_) cur = tileOut_;
            ComputeBlock(static_cast<uint32_t>(base), cur);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeBlock(uint32_t baseOutIdx, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t outIdx = baseOutIdx + i;

            // Decode flat index -> (n, c, oh, ow)
            uint32_t ow = outIdx % outW_;
            uint32_t t  = outIdx / outW_;
            uint32_t oh = t % outH_;
            t /= outH_;
            uint32_t c = t % C_;
            uint32_t n = t / C_;

            const int32_t in_y0 = static_cast<int32_t>(oh) * static_cast<int32_t>(S_) - static_cast<int32_t>(P_);
            const int32_t in_x0 = static_cast<int32_t>(ow) * static_cast<int32_t>(S_) - static_cast<int32_t>(P_);

            bool hasValid = false;

            AscendC::LocalTensor<float> acc = tmpQueue_.AllocTensor<float>();
            AscendC::Duplicate(acc, -FLT_MAX, 1);

            for (uint32_t ky = 0; ky < K_; ++ky) {
                const int32_t iy = in_y0 + static_cast<int32_t>(ky);
                if (iy < 0 || iy >= static_cast<int32_t>(H_)) continue;

                for (uint32_t kx = 0; kx < K_; ++kx) {
                    const int32_t ix = in_x0 + static_cast<int32_t>(kx);
                    if (ix < 0 || ix >= static_cast<int32_t>(W_)) continue;

                    hasValid = true;

                    const uint64_t inOffset =
                        (((static_cast<uint64_t>(n) * C_ + c) * H_ + static_cast<uint32_t>(iy)) * W_ + static_cast<uint32_t>(ix));

                    AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                    AscendC::DataCopy(tmp, xGm_[inOffset], 1);
                    AscendC::Max(acc, acc, tmp, 1);
                    tmpQueue_.FreeTensor(tmp);
                }
            }

            const uint64_t outOffset =
                (((static_cast<uint64_t>(n) * C_ + c) * outH_ + oh) * outW_ + ow);

            AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
            if (hasValid) {
                AscendC::DataCopy(yLocal, acc, 1);
            } else {
                // Entire window was outside input due to extreme padding; PyTorch behavior uses zero padding.
                AscendC::Duplicate(yLocal, 0.0f, 1);
            }
            AscendC::DataCopy(yGm_[outOffset], yLocal, 1);
            outQueue_.FreeTensor(yLocal);
            tmpQueue_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t K_{0}, S_{0}, P_{0};
    uint32_t totalOut_{0}, tileOut_{1};
};

extern "C" __global__ __aicore__ void max_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling2dCustom op;
    op.Init(
        x, y,
        tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
        tiling_data.outH, tiling_data.outW,
        tiling_data.kernel, tiling_data.stride, tiling_data.padding,
        tiling_data.totalOut, tiling_data.tileOut
    );
    op.Process();
}
"""
