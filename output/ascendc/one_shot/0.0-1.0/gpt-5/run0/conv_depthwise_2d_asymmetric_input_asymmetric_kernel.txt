#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t Kh, uint32_t Kw,
                                uint32_t Sh, uint32_t Sw,
                                uint32_t Ph, uint32_t Pw,
                                uint32_t Dh, uint32_t Dw,
                                uint32_t tileWoMax, uint32_t maxSegLen)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        Ho_ = Ho; Wo_ = Wo;
        Kh_ = Kh; Kw_ = Kw;
        Sh_ = Sh; Sw_ = Sw;
        Ph_ = Ph; Pw_ = Pw;
        Dh_ = Dh; Dw_ = Dw;
        tileWoMax_ = tileWoMax;
        maxSegLen_ = maxSegLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_)*C_*H_*W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(C_)*Kh_*Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_)*C_*Ho_*Wo_);

        // UB buffers: input stripes (Kh * maxSegLen), weights (Kh*Kw), output stripe (tileWoMax)
        pipe_.InitBuffer(inQueue_, 1, Kh_ * maxSegLen_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, Kh_ * Kw_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileWoMax_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || Ho_ == 0 || Wo_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Tasks over (N,C,Ho)
        const uint64_t tasks = static_cast<uint64_t>(N_) * C_ * Ho_;
        for (uint64_t task = blockIdx; task < tasks; task += blockNum) {
            uint32_t tmp = static_cast<uint32_t>(task);
            const uint32_t n = tmp / (C_ * Ho_);
            tmp = tmp % (C_ * Ho_);
            const uint32_t c = tmp / Ho_;
            const uint32_t ho = tmp % Ho_;

            DoOneRowTask(n, c, ho);
        }
    }

private:
    __aicore__ inline uint64_t XIndex(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * C_ + c) * H_ + h) * W_ + w;
    }

    __aicore__ inline uint64_t WIndex(uint32_t c, uint32_t r, uint32_t k) const
    {
        return ((uint64_t)c * Kh_ + r) * Kw_ + k;
    }

    __aicore__ inline uint64_t YIndex(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * C_ + c) * Ho_ + h) * Wo_ + w;
    }

    __aicore__ inline void LoadWeightsToUB(uint32_t c, AscendC::LocalTensor<float>& wLocal)
    {
        const uint64_t wOffset = WIndex(c, 0, 0);
        AscendC::DataCopy(wLocal, wGm_[wOffset], Kh_ * Kw_);
    }

    __aicore__ inline void DoOneRowTask(uint32_t n, uint32_t c, uint32_t ho)
    {
        // Weight local buffer
        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        LoadWeightsToUB(c, wLocal);

        const int32_t base_hi0 = static_cast<int32_t>(ho) * static_cast<int32_t>(Sh_) - static_cast<int32_t>(Ph_);

        // Iterate over output width in tiles
        for (uint32_t woBase = 0; woBase < Wo_; woBase += tileWoMax_) {
            const uint32_t tile = (woBase + tileWoMax_ <= Wo_) ? tileWoMax_ : (Wo_ - woBase);
            const uint32_t curSegLen = (tile - 1u) * Sw_ + 1u + (Kw_ - 1u) * Dw_;

            AscendC::LocalTensor<float> inLocal = inQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

            // Initialize output stripe to 0
            AscendC::Dup(yLocal, 0.0f, tile);

            // For each kernel row r, prepare input stripe with zero-padding as needed
            for (uint32_t r = 0; r < Kh_; ++r) {
                const int32_t hi = base_hi0 + static_cast<int32_t>(r) * static_cast<int32_t>(Dh_);
                float* rowBuf = inLocal.GetValuePtr(0) + r * maxSegLen_;

                // zero fill current row stripe
                for (uint32_t i = 0; i < curSegLen; ++i) {
                    rowBuf[i] = 0.0f;
                }

                if (hi >= 0 && hi < static_cast<int32_t>(H_)) {
                    const int32_t left = static_cast<int32_t>(woBase) * static_cast<int32_t>(Sw_) - static_cast<int32_t>(Pw_);
                    const int32_t copyStart = left < 0 ? 0 : left;
                    const int32_t copyEnd = left + static_cast<int32_t>(curSegLen) > static_cast<int32_t>(W_) ?
                                            static_cast<int32_t>(W_) : (left + static_cast<int32_t>(curSegLen));
                    const int32_t copyLen = copyEnd - copyStart;
                    if (copyLen > 0) {
                        const uint32_t dstOff = static_cast<uint32_t>(copyStart - left);
                        const uint64_t srcOffset = XIndex(n, c, static_cast<uint32_t>(hi), static_cast<uint32_t>(copyStart));
                        // Copy contiguous segment into row buffer position
                        AscendC::DataCopy(inLocal[ r * maxSegLen_ + dstOff ],
                                          xGm_[srcOffset],
                                          static_cast<uint32_t>(copyLen));
                    }
                }

                // Convolution accumulate for this row r
                for (uint32_t k = 0; k < Kw_; ++k) {
                    const float wval = wLocal[r * Kw_ + k];
                    const uint32_t baseOff = k * Dw_;
                    for (uint32_t t = 0; t < tile; ++t) {
                        const uint32_t inOff = baseOff + t * Sw_;
                        const float inval = rowBuf[inOff];
                        yLocal[t] = AscendC::Fma(inval, wval, yLocal[t]);
                    }
                }
            }

            // Write back output stripe
            const uint64_t yOffset = YIndex(n, c, ho, woBase);
            AscendC::DataCopy(yGm_[yOffset], yLocal, tile);

            outQueue_.FreeTensor(yLocal);
            inQueue_.FreeTensor(inLocal);
        }

        wQueue_.FreeTensor(wLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t Kh_{0}, Kw_{0};
    uint32_t Sh_{1}, Sw_{1}, Ph_{0}, Pw_{0}, Dh_{1}, Dw_{1};
    uint32_t tileWoMax_{0}, maxSegLen_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.Ho, tiling_data.Wo,
            tiling_data.Kh, tiling_data.Kw,
            tiling_data.Sh, tiling_data.Sw,
            tiling_data.Ph, tiling_data.Pw,
            tiling_data.Dh, tiling_data.Dw,
            tiling_data.tileWoMax, tiling_data.maxSegLen);
    op.Process();
}
