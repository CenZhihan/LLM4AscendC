kernel_src="""

#include "kernel_operator.h"

class KernelConv2dHardSwishReluCustom {
public:
    __aicore__ inline KernelConv2dHardSwishReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t hIn, uint32_t wIn,
                                uint32_t cOut, uint32_t kH, uint32_t kW,
                                uint32_t oH, uint32_t oW, uint32_t tileW)
    {
        n_ = n; cIn_ = cIn; hIn_ = hIn; wIn_ = wIn;
        cOut_ = cOut; kH_ = kH; kW_ = kW; oH_ = oH; oW_ = oW; tileW_ = tileW;

        xPtr_ = (__gm__ float*)x;
        wPtr_ = (__gm__ float*)w;
        bPtr_ = (__gm__ float*)b;
        yPtr_ = (__gm__ float*)y;
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cOut_ == 0 || oH_ == 0 || oW_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint64_t items = (uint64_t)n_ * (uint64_t)cOut_;
        for (uint64_t idx = blockIdx; idx < items; idx += blockNum) {
            const uint32_t n = (uint32_t)(idx / cOut_);
            const uint32_t oc = (uint32_t)(idx % cOut_);
            ComputeOneNO(n, oc);
        }
    }

private:
    __aicore__ inline uint64_t XIndex(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * cIn_ + c) * hIn_ + h) * wIn_ + w;
    }

    __aicore__ inline uint64_t WIndex(uint32_t oc, uint32_t c, uint32_t kh, uint32_t kw) const
    {
        return (((uint64_t)oc * cIn_ + c) * kH_ + kh) * kW_ + kw;
    }

    __aicore__ inline uint64_t YIndex(uint32_t n, uint32_t oc, uint32_t oh, uint32_t ow) const
    {
        return (((uint64_t)n * cOut_ + oc) * oH_ + oh) * oW_ + ow;
    }

    __aicore__ inline float FusedHardSwishRelu(float v) const
    {
        float t = v + 3.0f;
        if (t < 0.0f) t = 0.0f;
        if (t > 6.0f) t = 6.0f;
        float hs = v * (t * (1.0f / 6.0f));
        return hs > 0.0f ? hs : 0.0f;
    }

    __aicore__ inline void ComputeOneNO(uint32_t n, uint32_t oc)
    {
        const bool hasBias = (bPtr_ != nullptr);
        const float biasVal = hasBias ? bPtr_[oc] : 0.0f;

        for (uint32_t oh = 0; oh < oH_; ++oh) {
            uint32_t ow = 0;
            while (ow < oW_) {
                const uint32_t curTile = (ow + tileW_ <= oW_) ? tileW_ : (oW_ - ow);
                for (uint32_t t = 0; t < curTile; ++t) {
                    const uint32_t oww = ow + t;
                    float acc = 0.0f;
                    // Direct convolution (stride=1, pad=0)
                    for (uint32_t ci = 0; ci < cIn_; ++ci) {
                        for (uint32_t kh = 0; kh < kH_; ++kh) {
                            const uint32_t ih = oh + kh;
                            for (uint32_t kw = 0; kw < kW_; ++kw) {
                                const uint32_t iw = oww + kw;
                                const float xval = xPtr_[XIndex(n, ci, ih, iw)];
                                const float wval = wPtr_[WIndex(oc, ci, kh, kw)];
                                acc += xval * wval;
                            }
                        }
                    }
                    acc += biasVal;
                    const float outv = FusedHardSwishRelu(acc);
                    yPtr_[YIndex(n, oc, oh, oww)] = outv;
                }
                ow += curTile;
            }
        }
    }

private:
    __gm__ float *xPtr_ {nullptr};
    __gm__ float *wPtr_ {nullptr};
    __gm__ float *bPtr_ {nullptr};
    __gm__ float *yPtr_ {nullptr};

    uint32_t n_ {0}, cIn_ {0}, hIn_ {0}, wIn_ {0};
    uint32_t cOut_ {0}, kH_ {0}, kW_ {0};
    uint32_t oH_ {0}, oW_ {0}, tileW_ {1};
};

extern "C" __global__ __aicore__ void conv2d_hard_swish_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dHardSwishReluCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.n, tiling_data.cIn, tiling_data.hIn, tiling_data.wIn,
            tiling_data.cOut, tiling_data.kH, tiling_data.kW,
            tiling_data.oH, tiling_data.oW, tiling_data.tileW);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor ensure_bias_or_zero(const at::Tensor& weight, c10::optional<at::Tensor> bias_opt) {
    if (bias_opt.has_value() && bias_opt.value().defined()) {
        return bias_opt.value();
    }
    at::TensorOptions opts = weight.options();
    return at::zeros({weight.size(0)}, opts);
}

at::Tensor conv2d_hard_swish_relu_custom_impl_npu(const at::Tensor& x,
                                                  const at::Tensor& weight,
                                                  c10::optional<at::Tensor> bias_opt) {
    TORCH_CHECK(x.dim() == 4, "x must be NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be OIHW");
    TORCH_CHECK(x.size(1) == weight.size(1), "in_channels mismatch between x and weight");
    TORCH_CHECK(x.scalar_type() == at::kFloat && weight.scalar_type() == at::kFloat, "Only float supported");

    const int64_t N  = x.size(0);
    const int64_t Ci = x.size(1);
    const int64_t Hi = x.size(2);
    const int64_t Wi = x.size(3);

    const int64_t Co = weight.size(0);
    const int64_t Kh = weight.size(2);
    const int64_t Kw = weight.size(3);

    TORCH_CHECK(Hi >= Kh && Wi >= Kw, "Kernel must fit within input (no padding)");

    const int64_t Oh = Hi - Kh + 1;
    const int64_t Ow = Wi - Kw + 1;

    at::Tensor result = at::empty({N, Co, Oh, Ow}, x.options());
    at::Tensor bias = ensure_bias_or_zero(weight, bias_opt);

    EXEC_NPU_CMD(aclnnConv2dHardSwishReluCustom, x, weight, bias, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_hard_swish_relu_custom", &conv2d_hard_swish_relu_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_hard_swish_relu_custom", &conv2d_hard_swish_relu_custom_impl_npu,
          "Conv2d + HardSwish then ReLU fused (Conv2dHardSwishReluCustom on NPU)");
}
