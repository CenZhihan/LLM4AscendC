{
    return (a + b - 1u) / b;

static ge::graphStatus TilingFunc(gert::TilingContext *context)

    MatmulWithTransposedBCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    uint32_t M = static_cast<uint32_t>(a_shape->GetShapeSize());
    uint32_t K = static_cast<uint32_t>(a_shape->GetShape(1));
    uint32_t N = static_cast<uint32_t>(b_shape->GetShape(0));

    tiling.set_k(K);
    tiling.set_n(N);

    uint32_t nTileSize = TILE_N;

    if (M < mTileSize) mTileSize = M > 0 ? M : 1;
    if (N < nTileSize) nTileSize = N > 0 ? N : 1;

    uint32_t totalBlocks = CeilDiv(M, mTileSize) * CeilDiv(N, nTileSize);
    uint32_t blockDim = totalBlocks > 24 ? 24 : (totalBlocks > 0 ? totalBlocks : 1);

    tiling.set_blockDim(blockDim);
    tiling.set_mTileSize(mTileSize);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_K = 256;
constexpr int32_t TILE_M = 64;
constexpr int32_t TILE_N = 64;

class KernelMatmulWithTransposedBCustom {
public:
    __aicore__ inline KernelMatmulWithTransposedBCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t m, uint32_t k, uint32_t n,
                                uint32_t mTileSize, uint32_t nTileSize)
    {
        m_ = m;
        k_ = k;
        n_ = n;
        mTileSize_ = mTileSize;
        nTileSize_ = nTileSize;

        aGm_.SetGlobalBuffer((__gm__ float*)a, m * k);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n * k);
        cGm_.SetGlobalBuffer((__gm__ float*)c, m * n);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, mTileSize * TILE_K * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, nTileSize * TILE_K * sizeof(float));
        pipe_.InitBuffer(tmpQueueC_, BUFFER_NUM, mTileSize * nTileSize * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, mTileSize * nTileSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || n_ == 0 || k_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalBlocks = AscendC::GetBlockNum();

        const uint32_t nTilesM = (m_ + mTileSize_ - 1) / mTileSize_;
        const uint32_t nTilesN = (n_ + nTileSize_ - 1) / nTileSize_;
        const uint32_t totalTiles = nTilesM * nTilesN;

        if (blockIdx >= totalTiles) return;

        const uint32_t tileMIdx = blockIdx / nTilesN;
        const uint32_t tileNIdx = blockIdx % nTilesN;

        const uint32_t mStart = tileMIdx * mTileSize_;
        const uint32_t mLen = (mStart + mTileSize_ > m_) ? (m_ - mStart) : mTileSize_;

        const uint32_t nStart = tileNIdx * nTileSize_;
        const uint32_t nLen = (nStart + nTileSize_ > n_) ? (n_ - nStart) : nTileSize_;

        ComputeTile(mStart, mLen, nStart, nLen);
    }

private:
    __aicore__ inline void ComputeTile(uint32_t mStart, uint32_t mLen,
                                       uint32_t nStart, uint32_t nLen)
    {
        AscendC::LocalTensor<float> cLocal = tmpQueueC_.AllocTensor<float>();
        AscendC::Set(cLocal, 0.0f, mLen * nLen);

        const uint32_t kTiles = (k_ + TILE_K - 1) / TILE_K;

        for (uint32_t kt = 0; kt < kTiles; ++kt) {
            const uint32_t kOffset = kt * TILE_K;
            const uint32_t curK = (kOffset + TILE_K > k_) ? (k_ - kOffset) : TILE_K;

            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();

            AscendC::DataCopy(aLocal, aGm_[mStart * k_ + kOffset], mLen * curK);
            AscendC::DataCopy(bLocal, bGm_[nStart * k_ + kOffset], nLen * curK);

            inQueueA_.EnQue(aLocal);
            inQueueB_.EnQue(bLocal);

            AscendC::LocalTensor<float> aDeq = inQueueA_.DeQue<float>();
            AscendC::LocalTensor<float> bDeq = inQueueB_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueC_.AllocTensor<float>();

            for (uint32_t mi = 0; mi < mLen; ++mi) {
                for (uint32_t ni = 0; ni < nLen; ++ni) {
                    float sum = 0.0f;
                    uint32_t aIdx = mi * curK;
                    uint32_t bIdx = ni * curK;
                    for (uint32_t ki = 0; ki < curK; ki += 8) {
                        uint32_t remain = curK - ki;
                        uint32_t step = remain >= 8 ? 8 : remain;

                        float aVal[8] = {0};
                        float bVal[8] = {0};
                        for (uint32_t s = 0; s < step; ++s) {
                            aVal[s] = aDeq.GetValue(aIdx + ki + s);
                            bVal[s] = bDeq.GetValue(bIdx + ki + s);
                        }
                        for (uint32_t s = 0; s < step; ++s) {
                            sum += aVal[s] * bVal[s];
                        }
                    }
                    cLocal.SetValue(mi * nLen + ni, cLocal.GetValue(mi * nLen + ni) + sum);
                }
            }

            outQueueC_.FreeTensor(yLocal);
            inQueueA_.FreeTensor(aDeq);
            inQueueB_.FreeTensor(bDeq);
        }

        AscendC::LocalTensor<float> outC = outQueueC_.AllocTensor<float>();
        AscendC::DataCopy(outC, cLocal, mLen * nLen);
        outQueueC_.EnQue(outC);

        AscendC::LocalTensor<float> finalC = outQueueC_.DeQue<float>();
        AscendC::DataCopy(cGm_[mStart * n_ + nStart], finalC, mLen * nLen);
        outQueueC_.FreeTensor(finalC);
        tmpQueueC_.FreeTensor(cLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueueC_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t m_ {0};
    uint32_t k_ {0};
    uint32_t n_ {0};
    uint32_t mTileSize_ {TILE_M};
    uint32_t nTileSize_ {TILE_N};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_b_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                                                      GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedBCustom op;
    op.Init(a, b, c, tiling_data.m, tiling_data.k, tiling_data.n,
            tiling_data.mTileSize, tiling_data.nTileSize);
    op.Process();
}
