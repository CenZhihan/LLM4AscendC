#include "gemm_max_subtract_gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMaxSubtractGeluCustomTilingData tiling;
    
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    
    uint32_t batchSize = static_cast<uint32_t>(x_shape->GetShapeVector()[0]);
    uint32_t inFeatures = static_cast<uint32_t>(x_shape->GetShapeVector()[1]);
    uint32_t outFeatures = static_cast<uint32_t>(w_shape->GetShapeVector()[0]);
    
    // totalLength for GEMM output: batchSize * outFeatures
    uint32_t totalLength = batchSize * outFeatures;
    
    // For max operation along dim 1, we need the full row to compute max
    // Then need space for mean computation
    // UB buffer model: matmul output + max buffer + mean buffer + gelu output
    // Approximate: 4 * tile for simplicity
    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    
    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }
    
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;
    
    context->SetBlockDim(blockDim);
    
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t maxDim = attrs->HasAttr(2) ? static_cast<int32_t>(attrs->GetAttrPointer<int>(2)[0]) : 1;
    
    tiling.set_batchSize(batchSize);
    tiling.set_inFeatures(inFeatures);
    tiling.set_outFeatures(outFeatures);
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_maxDim(maxDim);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include <math.h>

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMaxSubtractGeluCustom {
public:
    __aicore__ inline KernelGemmMaxSubtractGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t totalLength, uint32_t tileLength, int32_t maxDim)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        maxDim_ = maxDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inFeatures);
        wGm_.SetGlobalBuffer((__gm__ float*)w, outFeatures * inFeatures);
        bGm_.SetGlobalBuffer((__gm__ float*)b, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, outFeatures * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueMax_, 1, batchSize * sizeof(float));
        pipe_.InitBuffer(tmpQueueMean_, 1, batchSize * sizeof(float));
        pipe_.InitBuffer(tmpQueueSub_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        if (inFeatures_ == 0 || outFeatures_ == 0) return;

        // Copy bias to local
        CopyBiasIn();

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block computes one batch's GEMM -> max - mean -> gelu
        // blockIdx maps to batch index
        uint32_t batchIdx = blockIdx;
        if (batchIdx >= batchSize_) return;

        // Compute GEMM for this batch: mat[b, :] = x[b, :] @ w[:, :]
        // Each block computes one output row
        ComputeGemmBatch(batchIdx);
    }

private:
    __aicore__ inline void CopyBiasIn()
    {
        bLocal_ = tmpQueueB_.AllocTensor<float>();
        AscendC::DataCopy(bLocal_, bGm_, outFeatures_);
        tmpQueueB_.EnQue(bLocal_);
    }

    __aicore__ inline void ComputeGemmBatch(uint32_t batchIdx)
    {
        // Step 1: GEMM - compute full row output for this batch
        // matOut[b, j] = sum_i(x[b, i] * w[j, i]) + b[j]
        
        AscendC::LocalTensor<float> matLocal = outQueueY_.AllocTensor<float>();
        
        // Initialize with bias
        AscendC::Broadcast(matLocal, bLocal_, outFeatures_, outFeatures_);
        
        // Process in tiles along K dimension (inFeatures)
        uint32_t kTiles = CeilDiv(inFeatures_, tileLen_);
        for (uint32_t kTileIdx = 0; kTileIdx < kTiles; ++kTileIdx) {
            uint32_t kOffset = kTileIdx * tileLen_;
            uint32_t kLen = inFeatures_ - kOffset;
            if (kLen > tileLen_) kLen = tileLen_;

            // Copy x tile for this batch
            AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
            uint32_t xOffset = batchIdx * inFeatures_ + kOffset;
            AscendC::DataCopy(xTile, xGm_[xOffset], kLen);

            // Copy w tile (column block j: j * tileLen_ to (j+1) * tileLen_)
            AscendC::LocalTensor<float> wTile = inQueueW_.AllocTensor<float>();
            uint32_t wOffset = kOffset; // w is stored as [outFeatures, inFeatures]
            // Actually need to access w[:, kOffset:kOffset+kLen]
            // wGm_[j * inFeatures_ + kOffset] for j from 0 to outFeatures-1 step tileLen_
            for (uint32_t outOffset = 0; outOffset < outFeatures_; outOffset += tileLen_) {
                uint32_t wCopyOffset = outOffset * inFeatures_ + kOffset;
                uint32_t wCopyLen = kLen;
                AscendC::DataCopy(wTile[outOffset], wGm_[wCopyOffset], wCopyLen);
            }

            // For each output column block
            uint32_t outTileCount = CeilDiv(outFeatures_, tileLen_);
            for (uint32_t outTileIdx = 0; outTileIdx < outTileCount; ++outTileIdx) {
                uint32_t outOffset = outTileIdx * tileLen_;
                uint32_t outLen = outFeatures_ - outOffset;
                if (outLen > tileLen_) outLen = tileLen_;

                // matOut[outOffset:outOffset+outLen] += xTile * wTile[outOffset:outOffset+outLen]
                AscendC::Matmul(matLocal[outOffset], xTile, wTile[outOffset], 
                               1, outLen, inFeatures_, kLen);
            }

            inQueueX_.FreeTensor(xTile);
            inQueueW_.FreeTensor(wTile);
        }

        outQueueY_.EnQue(matLocal);

        // Step 2: Max along dim 1 (maxDim=1)
        AscendC::LocalTensor<float> matDeq = outQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> maxLocal = tmpQueueMax_.AllocTensor<float>();
        AscendC::ReduceMax(maxLocal, matDeq, 0, outFeatures_, true);

        // Step 3: Subtract mean along dim 1
        AscendC::LocalTensor<float> meanLocal = tmpQueueMean_.AllocTensor<float>();
        AscendC::ReduceMean(meanLocal, matDeq, 0, outFeatures_, true);
        
        AscendC::LocalTensor<float> subLocal = tmpQueueSub_.AllocTensor<float>();
        AscendC::Sub(subLocal, maxLocal, meanLocal, batchSize_);

        // Step 4: GELU activation
        AscendC::LocalTensor<float> geluLocal = outQueueY_.AllocTensor<float>();
        AscendC::Gelu(geluLocal, subLocal, batchSize_);

        // Copy result to output
        AscendC::DataCopy(yGm_[batchIdx], geluLocal, 1);

        outQueueY_.FreeTensor(matDeq);
        tmpQueueMax_.FreeTensor(maxLocal);
        tmpQueueMean_.FreeTensor(meanLocal);
        tmpQueueSub_.FreeTensor(subLocal);
        tmpQueueB_.FreeTensor(bLocal_);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMean_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueSub_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::LocalTensor<float> bLocal_;

    uint32_t batchSize_ {0};
    uint32_t inFeatures_ {0};
    uint32_t outFeatures_ {0};
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t maxDim_ {1};
};

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

extern "C" __global__ __aicore__ void gemm_max_subtract_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMaxSubtractGeluCustom op;
    op.Init(x, w, b, y, 
            tiling_data.batchSize, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.maxDim);
    op.Process();
}
