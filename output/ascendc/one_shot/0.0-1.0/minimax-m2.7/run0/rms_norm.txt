#include "rms_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t FEATURE_TILE = 64;
static constexpr uint32_t ELEMS_PER_THREAD = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    RmsNormCustomTilingData tiling;
    
    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::Shape &shape = inputShape->GetOriginShape();
    
    uint32_t batchSize = static_cast<uint32_t>(shape.GetShapeDim(0));
    uint32_t numFeatures = static_cast<uint32_t>(shape.GetShapeDim(1));
    uint32_t innerSize = 1;
    for (int i = 2; i < shape.GetDims(); ++i) {
        innerSize *= static_cast<uint32_t>(shape.GetShapeDim(i));
    }
    
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);

    // tileLength represents number of elements per feature reduction
    // Each thread handles FEATURE_TILE features, with ELEMS_PER_THREAD inner elements
    uint32_t tileLength = FEATURE_TILE;
    if (tileLength > numFeatures) {
        tileLength = AlignDown(numFeatures, ALIGN_ELEMS);
        if (tileLength < ELEMS_PER_THREAD) tileLength = numFeatures;
    }

    // blockDim: grid over batchSize * innerSize
    const uint32_t totalOuter = batchSize * innerSize;
    uint32_t blockDim = CeilDiv(totalOuter, 1);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 256u) blockDim = 256u;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_numFeatures(numFeatures);
    tiling.set_innerSize(innerSize);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t REDUCE_BUFFER_NUM = 2;

class KernelRmsNormCustom {
public:
    __aicore__ inline KernelRmsNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t numFeatures,
                                uint32_t innerSize, uint32_t tileLength,
                                float eps)
    {
        batchSize_ = batchSize;
        numFeatures_ = numFeatures;
        innerSize_ = innerSize;
        tileLength_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * numFeatures_ * innerSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * numFeatures_ * innerSize_);

        const uint32_t maxFeatures = 64;  // max features per tile
        pipe_.InitBuffer(inQueueX_, REDUCE_BUFFER_NUM, maxFeatures * innerSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, maxFeatures * innerSize_ * sizeof(float));
        pipe_.InitBuffer(reduceQueue_, 1, maxFeatures * innerSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (numFeatures_ == 0 || innerSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalOuter = batchSize_ * innerSize_;
        const uint32_t strideBlocks = blockNum;

        uint32_t baseOuter = blockIdx;
        if (baseOuter >= totalOuter) return;

        while (baseOuter < totalOuter) {
            uint32_t batchIdx = baseOuter / innerSize_;
            uint32_t innerIdx = baseOuter % innerSize_;
            
            ReduceAndNormalize(batchIdx, innerIdx);
            
            baseOuter += strideBlocks;
        }
    }

private:
    __aicore__ inline void ReduceAndNormalize(uint32_t batchIdx, uint32_t innerIdx)
    {
        // Step 1: Compute sum of squares across all features for this (batch, inner) position
        float sumSquares = 0.0f;
        
        uint32_t featOffset = 0;
        while (featOffset < numFeatures_) {
            uint32_t curFeatLen = tileLength_;
            if (featOffset + curFeatLen > numFeatures_) {
                curFeatLen = numFeatures_ - featOffset;
            }
            
            // Load feature tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            for (uint32_t i = 0; i < curFeatLen; ++i) {
                for (uint32_t j = 0; j < innerSize_; ++j) {
                    if (innerIdx == j) {
                        uint32_t srcIdx = (batchIdx * numFeatures_ + featOffset + i) * innerSize_ + j;
                        xLocal.SetValue(i * innerSize_ + j, xGm_.GetValue(srcIdx));
                    }
                }
            }
            inQueueX_.EnQue(xLocal);
            
            // Compute sum of squares for this tile
            AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> sqLocal = reduceQueue_.AllocTensor<float>();
            
            AscendC::Muls(sqLocal, xDeq, xDeq, curFeatLen * innerSize_);
            
            // Reduce sum across features
            for (uint32_t j = 0; j < innerSize_; ++j) {
                if (innerIdx == j) {
                    float tileSum = 0.0f;
                    for (uint32_t i = 0; i < curFeatLen; ++i) {
                        tileSum += sqLocal.GetValue(i * innerSize_ + j);
                    }
                    sumSquares += tileSum;
                }
            }
            
            reduceQueue_.FreeTensor(sqLocal);
            inQueueX_.FreeTensor(xDeq);
            
            featOffset += curFeatLen;
        }

        // Step 2: Compute rms = sqrt(sumSquares / numFeatures + eps)
        float rms = sumSquares / static_cast<float>(numFeatures_) + eps_;
        rms = AscendC::Sqrt(AscendC::CastTo<float>(AscendC::Set1<float>(rms), RoundMode::CAST_NONE));
        
        // Compute 1/rms for multiplication
        float invRms = 1.0f / rms;

        // Step 3: Normalize - multiply x by 1/rms
        featOffset = 0;
        while (featOffset < numFeatures_) {
            uint32_t curFeatLen = tileLength_;
            if (featOffset + curFeatLen > numFeatures_) {
                curFeatLen = numFeatures_ - featOffset;
            }
            
            // Load feature tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            for (uint32_t i = 0; i < curFeatLen; ++i) {
                for (uint32_t j = 0; j < innerSize_; ++j) {
                    if (innerIdx == j) {
                        uint32_t srcIdx = (batchIdx * numFeatures_ + featOffset + i) * innerSize_ + j;
                        xLocal.SetValue(i * innerSize_ + j, xGm_.GetValue(srcIdx));
                    }
                }
            }
            inQueueX_.EnQue(xLocal);
            
            // Multiply by invRms
            AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            
            AscendC::Muls(yLocal, xDeq, invRms, curFeatLen * innerSize_);
            
            outQueueY_.EnQue(yLocal);
            inQueueX_.FreeTensor(xDeq);
            
            // Copy out
            AscendC::LocalTensor<float> yDeq = outQueueY_.DeQue<float>();
            for (uint32_t i = 0; i < curFeatLen; ++i) {
                for (uint32_t j = 0; j < innerSize_; ++j) {
                    if (innerIdx == j) {
                        uint32_t dstIdx = (batchIdx * numFeatures_ + featOffset + i) * innerSize_ + j;
                        yGm_.SetValue(dstIdx, yDeq.GetValue(i * innerSize_ + j));
                    }
                }
            }
            outQueueY_.FreeTensor(yDeq);
            
            featOffset += curFeatLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, REDUCE_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> reduceQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t numFeatures_ {0};
    uint32_t innerSize_ {0};
    uint32_t tileLength_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNormCustom op;
    op.Init(x, y, tiling_data.batchSize, tiling_data.numFeatures, 
            tiling_data.innerSize, tiling_data.tileLength, tiling_data.eps);
    op.Process();
}
