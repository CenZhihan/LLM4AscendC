#include "vanilla_rnn_hidden_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 128 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MIN_TILE_ELEMS = 64;
static constexpr uint32_t MAX_TILE_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    VanillaRnnHiddenCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *h_shape = context->GetInputShape(1);
    const gert::Shape *w_i2h_shape = context->GetInputShape(2);
    const gert::Shape *w_h2o_shape = context->GetInputShape(4);

    uint32_t kSize = static_cast<uint32_t>(w_i2h_shape->GetShapeSize());
    uint32_t hiddenSize = static_cast<uint32_t>(h_shape->GetShapeSize());
    uint32_t outputSize = static_cast<uint32_t>(w_h2o_shape->GetShapeSize());

    tiling.set_hiddenSize(hiddenSize);
    tiling.set_inputSize(kSize);
    tiling.set_outputSize(outputSize);
    tiling.set_batchSize(1);
    tiling.set_kSize(kSize);

    uint32_t tileLen = UB_BUDGET_BYTES / (4 * 6);
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    tileLen = std::max(tileLen, MIN_TILE_ELEMS);
    tileLen = std::min(tileLen, MAX_TILE_ELEMS);

    uint32_t blockDim = CeilDiv(hiddenSize, tileLen);
    blockDim = std::max(blockDim, 1u);
    blockDim = std::min(blockDim, 8u);

    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h, GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o,
                                GM_ADDR y, GM_ADDR h_new,
                                uint32_t hiddenSize, uint32_t inputSize,
                                uint32_t outputSize, uint32_t kSize,
                                uint32_t tileLength)
    {
        hiddenSize_ = hiddenSize;
        inputSize_ = inputSize;
        outputSize_ = outputSize;
        kSize_ = kSize;
        tileLength_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, inputSize_);
        hGm_.SetGlobalBuffer((__gm__ float*)h, hiddenSize_);
        w_i2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, hiddenSize_ * kSize_);
        b_i2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, hiddenSize_);
        w_h2oGm_.SetGlobalBuffer((__gm__ float*)w_h2o, outputSize_ * hiddenSize_);
        b_h2oGm_.SetGlobalBuffer((__gm__ float*)b_h2o, outputSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputSize_);
        hNewGm_.SetGlobalBuffer((__gm__ float*)h_new, hiddenSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, kSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueH_, BUFFER_NUM, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLength_ * kSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(outQueuePre_, BUFFER_NUM, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueH_, BUFFER_NUM, hiddenSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, std::max(hiddenSize_, tileLength_ * kSize_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (hiddenSize_ == 0 || kSize_ == 0 || outputSize_ == 0) return;

        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> hLocal = inQueueH_.AllocTensor<float>();

        AscendC::DataCopy(xLocal, xGm_[0], inputSize_);
        AscendC::DataCopy(hLocal, hGm_[0], hiddenSize_);

        inQueueX_.EnQue(xLocal);
        inQueueH_.EnQue(hLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> hLocal = inQueueH_.DeQue<float>();
        AscendC::LocalTensor<float> preLocal = outQueuePre_.AllocTensor<float>();

        AscendC::MatmulDirection dim = AscendC::MatmulDirection::OUT_COLLAPSE;
        AscendC::MatmulAttr matmulAttr;
        matmulAttr.matePos = 0;

        AscendC::LocalTensor<float> w_i2hLocal = w_i2hGm_.GetTensor();

        AscendC::MatMul(preLocal, w_i2hLocal, xLocal, hiddenSize_, kSize_, 1, dim, matmulAttr);

        AscendC::Adds(preLocal, preLocal, b_i2hGm_, hiddenSize_, hiddenSize_);

        AscendC::Tanh(preLocal, preLocal, hiddenSize_);

        outQueuePre_.EnQue(preLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<float> preLocal = outQueuePre_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueH_.AllocTensor<float>();
        AscendC::LocalTensor<float> hNewLocal = outQueueH_.AllocTensor<float>();

        AscendC::MatmulDirection dim = AscendC::MatmulDirection::OUT_COLLAPSE;
        AscendC::MatmulAttr matmulAttr;
        matmulAttr.matePos = 0;

        AscendC::LocalTensor<float> w_h2oLocal = w_h2oGm_.GetTensor();

        AscendC::MatMul(yLocal, w_h2oLocal, preLocal, outputSize_, hiddenSize_, 1, dim, matmulAttr);

        AscendC::Adds(yLocal, yLocal, b_h2oGm_, outputSize_, outputSize_);

        AscendC::DataCopy(hNewLocal, preLocal, hiddenSize_);

        AscendC::DataCopy(yGm_[0], yLocal, outputSize_);
        AscendC::DataCopy(hNewGm_[0], hNewLocal, hiddenSize_);

        outQueueH_.FreeTensor(yLocal);
        outQueueH_.FreeTensor(hNewLocal);

        inQueueX_.FreeTensor(xLocal);
        inQueueH_.FreeTensor(hLocal);
        outQueuePre_.FreeTensor(preLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueH_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueuePre_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueH_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> hGm_;
    AscendC::GlobalTensor<float> w_i2hGm_;
    AscendC::GlobalTensor<float> b_i2hGm_;
    AscendC::GlobalTensor<float> w_h2oGm_;
    AscendC::GlobalTensor<float> b_h2oGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> hNewGm_;

    uint32_t hiddenSize_ {0};
    uint32_t inputSize_ {0};
    uint32_t outputSize_ {0};
    uint32_t kSize_ {0};
    uint32_t tileLength_ {0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h, GM_ADDR w_i2h,
                                                                 GM_ADDR b_i2h, GM_ADDR w_h2o, GM_ADDR b_h2o,
                                                                 GM_ADDR y, GM_ADDR h_new,
                                                                 GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h, w_i2h, b_i2h, w_h2o, b_h2o, y, h_new,
            tiling_data.hiddenSize, tiling_data.inputSize,
            tiling_data.outputSize, tiling_data.kSize,
            tiling_data.tileLength);
    op.Process();
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'vanilla_rnn_hidden_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, x, h0, w_i2h, b_i2h, w_h2o, b_h2o):
        T, B, I = x.shape
        H, K = w_i2h.shape
        O = w_h2o.shape[0]
        h = h0
        outs = []
        for t in range(T):
            x_t = x[t].reshape(-1)
            fn = getattr(custom_ops_lib, _FN)
            y_t, h = fn(x_t, h, w_i2h, b_i2h, w_h2o, b_h2o)
            outs.append(y_t.squeeze(0).squeeze(0))
        return torch.stack(outs, dim=0)
'''
