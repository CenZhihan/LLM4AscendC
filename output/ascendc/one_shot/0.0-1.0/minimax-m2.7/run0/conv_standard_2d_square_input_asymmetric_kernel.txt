kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t Ci_TILE = 8;
constexpr int32_t Ho_TILE = 8;
constexpr int32_t Wo_TILE = 32;
constexpr int32_t Kh_MAX = 16;
constexpr int32_t Kw_MAX = 16;

class KernelConvStandard2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t inputN, uint32_t inputC, uint32_t inputH, uint32_t inputW,
                                uint32_t weightCO, uint32_t weightKH, uint32_t weightKW,
                                uint32_t outputH, uint32_t outputW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW)
    {
        inputN_ = inputN;
        inputC_ = inputC;
        inputH_ = inputH;
        inputW_ = inputW;
        weightCO_ = weightCO;
        weightKH_ = weightKH;
        weightKW_ = weightKW;
        outputH_ = outputH;
        outputW_ = outputW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)w);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, Ci_TILE * Ho_TILE * Wo_TILE * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, weightCO_ * Ci_TILE * weightKH_ * weightKW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, Ho_TILE * Wo_TILE * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, Ci_TILE * Ho_TILE * Wo_TILE * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        
        const uint32_t outHpBlk = (outputH_ + Ho_TILE - 1) / Ho_TILE;
        const uint32_t outWpBlk = (outputW_ + Wo_TILE - 1) / Wo_TILE;
        const uint32_t totalBlk = inputN_ * outHpBlk * outWpBlk;
        
        if (totalBlk == 0) return;
        
        const uint32_t blkPerGroup = (totalBlk + blockNum - 1) / blockNum;
        const uint32_t startBlk = blockIdx * blkPerGroup;
        const uint32_t endBlk = (blockIdx + 1) * blkPerGroup;
        
        for (uint32_t blkId = startBlk; blkId < endBlk && blkId < totalBlk; ++blkId) {
            uint32_t n = blkId / (outHpBlk * outWpBlk);
            uint32_t remain = blkId % (outHpBlk * outWpBlk);
            uint32_t hoIdx = remain / outWpBlk;
            uint32_t woIdx = remain % outWpBlk;
            
            ConvOneBlock(n, hoIdx, woIdx);
        }
    }

private:
    __aicore__ inline void ConvOneBlock(uint32_t n, uint32_t hoIdx, uint32_t woIdx)
    {
        const uint32_t hoStart = hoIdx * Ho_TILE;
        const uint32_t woStart = woIdx * Wo_TILE;
        const uint32_t hoEnd = (hoStart + Ho_TILE > outputH_) ? outputH_ : hoStart + Ho_TILE;
        const uint32_t woEnd = (woStart + Wo_TILE > outputW_) ? outputW_ : woStart + Wo_TILE;
        const uint32_t curHo = hoEnd - hoStart;
        const uint32_t curWo = woEnd - woStart;
        
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        
        for (uint32_t co = 0; co < weightCO_; ++co) {
            AscendC::SetVector<float>(yLocal, 0.0f, curHo * curWo);
            
            for (uint32_t ciBase = 0; ciBase < inputC_; ciBase += Ci_TILE) {
                const uint32_t ciEnd = (ciBase + Ci_TILE > inputC_) ? inputC_ : ciBase + Ci_TILE;
                const uint32_t curCi = ciEnd - ciBase;
                
                LoadWeightBlock(wGm_, co, ciBase, curCi);
                
                for (uint32_t kh = 0; kh < weightKH_; ++kh) {
                    const uint32_t hi = hoStart * strideH_ + kh * dilationH_ - padH_;
                    if (hi >= inputH_) continue;
                    const uint32_t hLimit = (hi < inputH_) ? 1 : 0;
                    
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    uint32_t wi = woStart * strideW_ - padW_;
                    uint32_t wiAligned = (wi / Wo_TILE) * Wo_TILE;
                    uint32_t offset = wiAligned;
                    
                    for (uint32_t woLocal = 0; woLocal < curWo + padW_; woLocal += Wo_TILE) {
                        const uint32_t hiLimit = (hi + 1 <= inputH_) ? hi + 1 : inputH_;
                        uint32_t wLimit = woStart + curWo;
                        
                        uint32_t win = (wi + Wo_TILE <= inputW_) ? Wo_TILE : (inputW_ - wi);
                        uint32_t loadW = (woLocal + Wo_TILE <= curWo + padW_) ? Wo_TILE : (curWo + padW_ - woLocal);
                        
                        uint32_t inHi = hi;
                        uint32_t inWi = wi;
                        uint32_t inLen = loadW;
                        
                        if (inWi >= inputW_) {
                            AscendC::SetVector<float>(xLocal, 0.0f, Ho_TILE * Wo_TILE);
                        } else {
                            if (inWi + inLen > inputW_) {
                                inLen = inputW_ - inWi;
                            }
                            uint32_t gmOffset = ((n * inputC_ + ciBase) * inputH_ + inHi) * inputW_ + inWi;
                            AscendC::DataCopy(xLocal, xGm_[gmOffset], inLen);
                            if (inLen < Ho_TILE * Wo_TILE) {
                                AscendC::SetVector<float>(xLocal[inLen], 0.0f, Ho_TILE * Wo_TILE - inLen);
                            }
                        }
                        inQueueX_.EnQue<float>(xLocal);
                        
                        AscendC::LocalTensor<float> xDeq = inQueueX_.DeQue<float>();
                        AscendC::LocalTensor<float> wDeq = inQueueW_.DeQue<float>();
                        
                        uint32_t khIdx = kh * weightKW_;
                        for (uint32_t kw = 0; kw < weightKW_; ++kw) {
                            AscendC::Matmul(yLocal, xDeq, wDeq[khIdx + kw], curHo * curWo, 1, curCi * weightKW_);
                        }
                        
                        inQueueX_.FreeTensor(xDeq);
                        inQueueW_.FreeTensor(wDeq);
                        
                        woLocal += Wo_TILE;
                        if (woLocal >= curWo + padW_) break;
                    }
                }
            }
            
            uint32_t yOffset = ((n * weightCO_ + co) * outputH_ + hoStart) * outputW_ + woStart;
            AscendC::DataCopy(yGm_[yOffset], yLocal, curHo * curWo);
        }
        
        outQueueY_.FreeTensor(yLocal);
        tmpQueue_.FreeTensor(tmpLocal);
    }
    
    __aicore__ inline void LoadWeightBlock(AscendC::GlobalTensor<float> wGm, uint32_t co, uint32_t ciBase, uint32_t curCi)
    {
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        uint32_t weightOffset = ((co * inputC_ + ciBase) * weightKH_ * weightKW_);
        
        const uint32_t cp0 = (curCi + 7) / 8;
        const uint32_t cp1 = 8;
        const uint32_t cp2 = weightKH_ * weightKW_;
        const uint32_t dstStep = cp0 * cp1 * cp2;
        
        for (uint32_t ci = 0; ci < curCi; ++ci) {
            for (uint32_t kh = 0; kh < weightKH_; ++kh) {
                for (uint32_t kw = 0; kw < weightKW_; ++kw) {
                    uint32_t srcIdx = ci * weightKH_ * weightKW_ + kh * weightKW_ + kw;
                    uint32_t dstIdx = (ci / cp1) * cp1 * cp2 + (ci % cp1) * cp2 + kh * weightKW_ + kw;
                    wLocal.SetValue(dstIdx, wGm[weightOffset + srcIdx]);
                }
            }
        }
        
        inQueueW_.EnQue<float>(wLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t inputN_ {0};
    uint32_t inputC_ {0};
    uint32_t inputH_ {0};
    uint32_t inputW_ {0};
    uint32_t weightCO_ {0};
    uint32_t weightKH_ {0};
    uint32_t weightKW_ {0};
    uint32_t outputH_ {0};
    uint32_t outputW_ {0};
    uint32_t strideH_ {1};
    uint32_t strideW_ {1};
    uint32_t padH_ {0};
    uint32_t padW_ {0};
    uint32_t dilationH_ {1};
    uint32_t dilationW_ {1};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, w, y,
            tiling_data.inputN, tiling_data.inputC, tiling_data.inputH, tiling_data.inputW,
            tiling_data.weightCO, tiling_data.weightKH, tiling_data.weightKW,
            tiling_data.outputH, tiling_data.outputW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW);
    op.Process();
}
