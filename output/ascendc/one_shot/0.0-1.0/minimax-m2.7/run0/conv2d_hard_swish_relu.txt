kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dHardSwishReluCustom {
public:
    __aicore__ inline KernelConv2dHardSwishReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength, uint32_t blockDim,
                                int32_t inC, int32_t outC,
                                int32_t inH, int32_t inW,
                                int32_t outH, int32_t outW,
                                int32_t kernelSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        inC_ = inC;
        outC_ = outC;
        inH_ = inH;
        inW_ = inW;
        outH_ = outH;
        outW_ = outW;
        kernelSize_ = kernelSize;

        const uint32_t patchSize = kernelSize_ * kernelSize_ * inC_;
        const uint32_t xQueueSize = tileLen_ * patchSize * sizeof(float);
        const uint32_t wQueueSize = outC_ * inC_ * kernelSize_ * kernelSize_ * sizeof(float);
        const uint32_t yQueueSize = tileLen_ * sizeof(float);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength * patchSize);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outC_ * inC_ * kernelSize_ * kernelSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, xQueueSize);
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, wQueueSize);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, yQueueSize);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        CopyIn(base, curLen);
        CopyInWeight();

        while (true) {
            const uint32_t nextBase = base + blockDim_ * tileLen_;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Conv(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t yOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();

        for (uint32_t e = 0; e < len; ++e) {
            uint32_t yIdx = yOffset + e;
            uint32_t n = yIdx / (outC_ * outH_ * outW_);
            uint32_t rem = yIdx % (outC_ * outH_ * outW_);
            uint32_t oc = rem / (outH_ * outW_);
            rem = rem % (outH_ * outW_);
            uint32_t oh = rem / outW_;
            uint32_t ow = rem % outW_;

            uint32_t inHStart = oh;
            uint32_t inWStart = ow;

            for (uint32_t ic = 0; ic < inC_; ++ic) {
                for (uint32_t kh = 0; kh < static_cast<uint32_t>(kernelSize_); ++kh) {
                    for (uint32_t kw = 0; kw < static_cast<uint32_t>(kernelSize_); ++kw) {
                        uint32_t inHIdx = inHStart + kh;
                        uint32_t inWIdx = inWStart + kw;
                        uint32_t xIdx = ((n * inC_ + ic) * inH_ + inHIdx) * inW_ + inWIdx;
                        uint32_t colIdx = (e * inC_ + ic) * kernelSize_ * kernelSize_ + kh * kernelSize_ + kw;
                        xLocal.SetValue(colIdx, xGm_.GetValue(xIdx));
                    }
                }
            }
        }

        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInWeight()
    {
        AscendC::LocalTensor<float> weightLocal = inQueueW_.AllocTensor<float>();
        for (uint32_t i = 0; i < static_cast<uint32_t>(outC_ * inC_ * kernelSize_ * kernelSize_); ++i) {
            weightLocal.SetValue(i, weightGm_.GetValue(i));
        }
        inQueueW_.EnQue(weightLocal);
    }

    __aicore__ inline void Conv(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> weightLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t e = 0; e < len; ++e) {
            float result = 0.0f;
            for (uint32_t oc = 0; oc < static_cast<uint32_t>(outC_); ++oc) {
                for (uint32_t ic = 0; ic < static_cast<uint32_t>(inC_); ++ic) {
                    for (uint32_t kh = 0; kh < static_cast<uint32_t>(kernelSize_); ++kh) {
                        for (uint32_t kw = 0; kw < static_cast<uint32_t>(kernelSize_); ++kw) {
                            uint32_t weightIdx = (oc * inC_ + ic) * kernelSize_ * kernelSize_ + kh * kernelSize_ + kw;
                            uint32_t colIdx = (e * inC_ + ic) * kernelSize_ * kernelSize_ + kh * kernelSize_ + kw;
                            result += weightLocal.GetValue(weightIdx) * xLocal.GetValue(colIdx);
                        }
                    }
                }
            }

            float hs = result * AscendC::Abs(result + 3.0f) / (result + 6.0f + AscendC::Abs(result - 3.0f));
            float relu = hs > 0.0f ? hs : 0.0f;
            yLocal.SetValue(e, relu);
        }

        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueW_.FreeTensor(weightLocal);
    }

    __aicore__ inline void CopyOut(uint32_t yOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        for (uint32_t e = 0; e < len; ++e) {
            yGm_.SetValue(yOffset + e, yLocal.GetValue(e));
        }
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
    int32_t inC_ {0};
    int32_t outC_ {0};
    int32_t inH_ {0};
    int32_t inW_ {0};
    int32_t outH_ {0};
    int32_t outW_ {0};
    int32_t kernelSize_ {0};
};

extern "C" __global__ __aicore__ void conv2d_hard_swish_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dHardSwishReluCustom op;
    op.Init(x, weight, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim,
           tiling_data.inC, tiling_data.outC, tiling_data.inH, tiling_data.inW,
           tiling_data.outH, tiling_data.outW, tiling_data.kernelSize);
    op.Process();
}
