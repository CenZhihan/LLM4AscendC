namespace optiling {

// UB budget model for this fused kernel:
//   MatA queue: 2 * tile_m * k (transposed from x, batch tile)
//   MatB queue: 2 * k * n (weights)
//   Bias queue: 1 * n
//   Result queue: 2 * tile_m * n
//   Temp queue: 1 * tile_m * n
//   LeakyRelu temp: 1 * tile_m * n
// Conservative: ~20 * tile_m * n + 4 * k * n bytes
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t HARD_CAP_TILE_M = 4096;
static constexpr uint32_t HARD_CAP_N = 4096;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMultiplyLeakyreluCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    const gert::Shape* b_shape = context->GetInputShape(2);

    const uint32_t batchSize = static_cast<uint32_t>(x_shape->GetShape().at(0));
    const uint32_t m = batchSize;
    const uint32_t k = static_cast<uint32_t>(x_shape->GetShape().at(1));
    const uint32_t n = static_cast<uint32_t>(w_shape->GetShape().at(0));

    float multiplierVal = 1.0f;

    tiling.set_inFeatures(k);
    tiling.set_outFeatures(n);

    tiling.set_k(k);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMultiplyLeakyreluCustom {
public:
    __aicore__ inline KernelGemmMultiplyLeakyreluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures,
                                uint32_t outFeatures, float negativeSlope,
                                float multiplierVal)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        m_ = batchSize;
        k_ = inFeatures;
        n_ = outFeatures;
        negativeSlope_ = negativeSlope;
        multiplierVal_ = multiplierVal;
        negCorr_ = negativeSlope_ - 1.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, m_ * k_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, n_ * k_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, m_ * n_);

        // Tile size: process rows in chunks for memory efficiency
        tileM_ = (m_ > 256) ? 256 : m_;
        tileM_ = ((tileM_ + 31) / 32) * 32;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileM_ * k_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, k_ * n_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, n_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileM_ * n_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * n_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || k_ == 0 || n_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalBlocks = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Each block handles tileM_ rows
        const uint32_t strideM = totalBlocks * tileM_;

        for (uint32_t rowStart = blockIdx * tileM_; rowStart < m_; rowStart += strideM) {
            uint32_t curRow = rowStart;
            uint32_t curTileM = tileM_;
            if (curRow + curTileM > m_) {
                curTileM = m_ - curRow;
            }

            // Process current tile
            ProcessTile(curRow, curTileM);
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t rowStart, uint32_t tileM)
    {
        // Load tile of x: tileM x k
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowStart * k_], tileM * k_);
        inQueueX_.EnQue(xLocal);

        // Load weight: k x n
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_[0], k_ * n_);
        inQueueW_.EnQue(wLocal);

        // Load bias: n
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        AscendC::DataCopy(bLocal, bGm_[0], n_);
        inQueueB_.EnQue(bLocal);

        // GEMM: xLocal @ wLocal^T + bias => yTmp (tileM x n)
        // Using multiply-add pattern for GEMM
        AscendC::LocalTensor<float> yTmp = outQueueY_.AllocTensor<float>();
        
        // yTmp = bias (broadcast row)
        AscendC::Duplicate(yTmp, bLocal, tileM * n_);

        // Matrix multiply accumulate: yTmp += xLocal @ wLocal^T
        // xLocal is tileM x k, wLocal is k x n, result is tileM x n
        AscendC::Matmul(yTmp, xLocal, wLocal, yTmp, false, false, 1.0f, 1.0f);

        // Free xLocal and wLocal
        inQueueX_.FreeTensor(xLocal);
        inQueueW_.FreeTensor(wLocal);

        // Multiply by scalar: yTmp *= multiplierVal
        AscendC::Muls(yTmp, yTmp, multiplierVal_, tileM * n_);

        // Apply LeakyReLU: y = yTmp + (negativeSlope - 1) * min(yTmp, 0)
        // Use temporary buffer for min computation
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> minTmp = tmpQueue_.AllocTensor<float>();

        // minTmp = min(yTmp, 0)
        AscendC::Mins(minTmp, yTmp, 0.0f, tileM * n_);
        // minTmp = (negativeSlope - 1) * minTmp
        AscendC::Muls(minTmp, minTmp, negCorr_, tileM * n_);
        // y = yTmp + minTmp
        AscendC::Add(yLocal, yTmp, minTmp, tileM * n_);

        // Free temp and intermediate
        tmpQueue_.FreeTensor(minTmp);
        inQueueB_.FreeTensor(bLocal);

        // Copy result to output
        AscendC::DataCopy(yGm_[rowStart * n_], yLocal, tileM * n_);

        // Cleanup
        outQueueY_.FreeTensor(yLocal);
        outQueueY_.FreeTensor(yTmp);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t inFeatures_ {0};
    uint32_t outFeatures_ {0};
    uint32_t m_ {0};
    uint32_t k_ {0};
    uint32_t n_ {0};
    uint32_t tileM_ {0};
    float negativeSlope_ {0.01f};
    float multiplierVal_ {1.0f};
    float negCorr_ {-0.99f};
};

extern "C" __global__ __aicore__ void gemm_multiply_leakyrelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                                                     GM_ADDR multiplier, GM_ADDR negative_slope,
                                                                     GM_ADDR y, GM_ADDR workspace,
                                                                     GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMultiplyLeakyreluCustom op;
    op.Init(x, w, b, y,
            tiling_data.batchSize, tiling_data.inFeatures,
            tiling_data.outFeatures, tiling_data.negativeSlope,
            tiling_data.multiplierVal);
    op.Process();
}
