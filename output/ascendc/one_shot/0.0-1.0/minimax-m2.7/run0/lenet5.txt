//   conv1_in: 6 * 28 * 28, conv1_out: 6 * 28 * 28
//   pool1_out: 6 * 14 * 14
//   conv2_in: 16 * 14 * 14, conv2_out: 16 * 10 * 10

static constexpr uint32_t UB_BUDGET_BYTES = 160 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

    Lenet5CustomTilingData tiling;

    uint32_t outputSize = batchSize * numClasses;

    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

// LeNet-5 dimensions
constexpr int32_t CONV1_OUT_CH = 6;
constexpr int32_t CONV2_OUT_CH = 16;
constexpr int32_t FC1_OUT = 120;
constexpr int32_t FC2_OUT = 84;
constexpr int32_t POOL_OUT_DIM = 5;
constexpr int32_t IMG_SIZE = 32;
constexpr int32_t KERNEL_SIZE = 5;

class KernelLenet5Custom {
public:
    __aicore__ inline KernelLenet5Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                GM_ADDR conv1_w, GM_ADDR conv1_b,
                                GM_ADDR conv2_w, GM_ADDR conv2_b,
                                GM_ADDR fc1_w, GM_ADDR fc1_b,
                                GM_ADDR fc2_w, GM_ADDR fc2_b,
                                GM_ADDR fc3_w, GM_ADDR fc3_b,
                                uint32_t batchSize, uint32_t numClasses,
                                uint32_t totalLength)
    {
        batchSize_ = batchSize;
        numClasses_ = numClasses;
        totalLen_ = totalLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * numClasses);

        conv1WGm_.SetGlobalBuffer((__gm__ float*)conv1_w, CONV1_OUT_CH * 1 * KERNEL_SIZE * KERNEL_SIZE);
        conv1BGm_.SetGlobalBuffer((__gm__ float*)conv1_b, CONV1_OUT_CH);
        conv2WGm_.SetGlobalBuffer((__gm__ float*)conv2_w, CONV2_OUT_CH * CONV1_OUT_CH * KERNEL_SIZE * KERNEL_SIZE);
        conv2BGm_.SetGlobalBuffer((__gm__ float*)conv2_b, CONV2_OUT_CH);
        fc1WGm_.SetGlobalBuffer((__gm__ float*)fc1_w, FC1_OUT * CONV2_OUT_CH * POOL_OUT_DIM * POOL_OUT_DIM);
        fc1BGm_.SetGlobalBuffer((__gm__ float*)fc1_b, FC1_OUT);
        fc2WGm_.SetGlobalBuffer((__gm__ float*)fc2_w, FC2_OUT * FC1_OUT);
        fc2BGm_.SetGlobalBuffer((__gm__ float*)fc2_b, FC2_OUT);
        fc3WGm_.SetGlobalBuffer((__gm__ float*)fc3_w, numClasses * FC2_OUT);
        fc3BGm_.SetGlobalBuffer((__gm__ float*)fc3_b, numClasses);

        // Allocate UB for each batch sample processing
        // conv1 output: 6*28*28 = 4704
        // pool1 output: 6*14*14 = 1176
        // conv2 output: 16*10*10 = 1600
        // pool2 output: 16*5*5 = 400
        // fc1 output: 120
        // fc2 output: 84
        constexpr uint32_t MAX_CONV1 = CONV1_OUT_CH * 28 * 28;
        constexpr uint32_t MAX_POOL1 = CONV1_OUT_CH * 14 * 14;
        constexpr uint32_t MAX_CONV2 = CONV2_OUT_CH * 10 * 10;
        constexpr uint32_t MAX_POOL2 = CONV2_OUT_CH * POOL_OUT_DIM * POOL_OUT_DIM;
        constexpr uint32_t MAX_FC1 = FC1_OUT;
        constexpr uint32_t MAX_FC2 = FC2_OUT;
        constexpr uint32_t MAX_FC3 = 256;  // max num_classes

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, IMG_SIZE * IMG_SIZE * sizeof(float));
        pipe_.InitBuffer(conv1Queue_, BUFFER_NUM, MAX_CONV1 * sizeof(float));
        pipe_.InitBuffer(pool1Queue_, BUFFER_NUM, MAX_POOL1 * sizeof(float));
        pipe_.InitBuffer(conv2Queue_, BUFFER_NUM, MAX_CONV2 * sizeof(float));
        pipe_.InitBuffer(pool2Queue_, BUFFER_NUM, MAX_POOL2 * sizeof(float));
        pipe_.InitBuffer(fc1Queue_, BUFFER_NUM, MAX_FC1 * sizeof(float));
        pipe_.InitBuffer(fc2Queue_, BUFFER_NUM, MAX_FC2 * sizeof(float));
        pipe_.InitBuffer(fc3Queue_, BUFFER_NUM, MAX_FC3 * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, MAX_CONV1 * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, MAX_CONV2 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideBatches = blockNum;
        uint32_t batchStart = blockIdx;

        while (batchStart < batchSize_) {
            ProcessSingleBatch(batchStart);
            batchStart += strideBatches;
        }
    }

private:
    __aicore__ inline void ProcessSingleBatch(uint32_t batchIdx)
    {
        // Process one batch sample through the entire LeNet-5 network

        // Stage 1: Conv1 + ReLU
        LocalTensor<float> conv1Out = conv1Queue_.AllocTensor<float>();
        Conv2dSingleChannel(batchIdx, conv1WGm_, conv1BGm_, conv1Out, 1, CONV1_OUT_CH, IMG_SIZE, 28);
        ReluInPlace(conv1Out, CONV1_OUT_CH * 28 * 28);
        conv1Queue_.EnQue(conv1Out);

        // Stage 2: MaxPool2d (2x2)
        LocalTensor<float> pool1Out = pool1Queue_.AllocTensor<float>();
        MaxPool2x2(conv1Queue_.DeQue<float>(), pool1Out, CONV1_OUT_CH, 28);
        pool1Queue_.EnQue(pool1Out);
        conv1Queue_.FreeTensor(conv1Queue_.DeQue<float>());

        // Stage 3: Conv2 + ReLU
        LocalTensor<float> conv2Out = conv2Queue_.AllocTensor<float>();
        Conv2dMultiChannel(pool1Queue_.DeQue<float>(), conv2WGm_, conv2BGm_, conv2Out,
                           CONV1_OUT_CH, CONV2_OUT_CH, 14, 10);
        ReluInPlace(conv2Out, CONV2_OUT_CH * 10 * 10);
        conv2Queue_.EnQue(conv2Out);
        pool1Queue_.FreeTensor(pool1Queue_.DeQue<float>());

        // Stage 4: MaxPool2d (2x2)
        LocalTensor<float> pool2Out = pool2Queue_.AllocTensor<float>();
        MaxPool2x2(conv2Queue_.DeQue<float>(), pool2Out, CONV2_OUT_CH, 10);
        pool2Queue_.EnQue(pool2Out);
        conv2Queue_.FreeTensor(conv2Queue_.DeQue<float>());

        // Stage 5: Flatten + FC1 + ReLU
        LocalTensor<float> fc1Out = fc1Queue_.AllocTensor<float>();
        FlattenAndLinear(pool2Queue_.DeQue<float>(), fc1WGm_, fc1BGm_, fc1Out,
                         CONV2_OUT_CH * POOL_OUT_DIM * POOL_OUT_DIM, FC1_OUT);
        ReluInPlace(fc1Out, FC1_OUT);
        fc1Queue_.EnQue(fc1Out);
        pool2Queue_.FreeTensor(pool2Queue_.DeQue<float>());

        // Stage 6: FC2 + ReLU
        LocalTensor<float> fc2Out = fc2Queue_.AllocTensor<float>();
        Linear(fc1Queue_.DeQue<float>(), fc2WGm_, fc2BGm_, fc2Out, FC1_OUT, FC2_OUT);
        ReluInPlace(fc2Out, FC2_OUT);
        fc2Queue_.EnQue(fc2Out);
        fc1Queue_.FreeTensor(fc1Queue_.DeQue<float>());

        // Stage 7: FC3 (output)
        LocalTensor<float> fc3Out = fc3Queue_.AllocTensor<float>();
        Linear(fc2Queue_.DeQue<float>(), fc3WGm_, fc3BGm_, fc3Out, FC2_OUT, numClasses_);
        fc2Queue_.FreeTensor(fc2Queue_.DeQue<float>());

        // Copy output to global memory
        uint32_t outOffset = batchIdx * numClasses_;
        AscendC::DataCopy(yGm_[outOffset], fc3Out, numClasses_);
        fc3Queue_.FreeTensor(fc3Out);
    }

    __aicore__ inline void Conv2dSingleChannel(uint32_t batchIdx,
                                               GlobalTensor<float>& weight,
                                               GlobalTensor<float>& bias,
                                               LocalTensor<float>& output,
                                               int32_t inChannels, int32_t outChannels,
                                               int32_t inSize, int32_t outSize)
    {
        LocalTensor<float> inputLocal = inQueueX_.AllocTensor<float>();
        uint32_t inOffset = batchIdx * inSize * inSize;
        AscendC::DataCopy(inputLocal, xGm_[inOffset], inSize * inSize);
        inQueueX_.EnQue(inputLocal);

        // Simple convolution: for each output channel, compute correlation
        LocalTensor<float> tmpLocal = tmpQueue1_.AllocTensor<float>();

        for (int32_t oc = 0; oc < outChannels; ++oc) {
            int32_t outBase = oc * outSize * outSize;

            // Initialize with bias
            for (int32_t i = 0; i < outSize * outSize; ++i) {
                output[outBase + i] = bias[oc];
            }

            // Compute convolution
            for (int32_t ic = 0; ic < inChannels; ++ic) {
                for (int32_t oh = 0; oh < outSize; ++oh) {
                    for (int32_t ow = 0; ow < outSize; ++ow) {
                        float sum = 0.0f;
                        for (int32_t kh = 0; kh < KERNEL_SIZE; ++kh) {
                            for (int32_t kw = 0; kw < KERNEL_SIZE; ++kw) {
                                int32_t ih = oh + kh;
                                int32_t iw = ow + kw;
                                if (ih < inSize && iw < inSize) {
                                    int32_t wIdx = oc * inChannels * KERNEL_SIZE * KERNEL_SIZE +
                                                   ic * KERNEL_SIZE * KERNEL_SIZE +
                                                   kh * KERNEL_SIZE + kw;
                                    int32_t inIdx = ic * inSize * inSize + ih * inSize + iw;
                                    sum += inputLocal[inIdx] * weight[wIdx];
                                }
                            }
                        }
                        output[outBase + oh * outSize + ow] += sum;
                    }
                }
            }
        }

        tmpQueue1_.FreeTensor(tmpLocal);
        inQueueX_.FreeTensor(inQueueX_.DeQue<float>());
    }

    __aicore__ inline void Conv2dMultiChannel(LocalTensor<float>& input,
                                              GlobalTensor<float>& weight,
                                              GlobalTensor<float>& bias,
                                              LocalTensor<float>& output,
                                              int32_t inChannels, int32_t outChannels,
                                              int32_t inSize, int32_t outSize)
    {
        LocalTensor<float> tmpLocal = tmpQueue2_.AllocTensor<float>();

        for (int32_t oc = 0; oc < outChannels; ++oc) {
            int32_t outBase = oc * outSize * outSize;

            for (int32_t i = 0; i < outSize * outSize; ++i) {
                output[outBase + i] = bias[oc];
            }

            for (int32_t ic = 0; ic < inChannels; ++ic) {
                for (int32_t oh = 0; oh < outSize; ++oh) {
                    for (int32_t ow = 0; ow < outSize; ++ow) {
                        float sum = 0.0f;
                        for (int32_t kh = 0; kh < KERNEL_SIZE; ++kh) {
                            for (int32_t kw = 0; kw < KERNEL_SIZE; ++kw) {
                                int32_t ih = oh + kh;
                                int32_t iw = ow + kw;
                                if (ih < inSize && iw < inSize) {
                                    int32_t wIdx = oc * inChannels * KERNEL_SIZE * KERNEL_SIZE +
                                                   ic * KERNEL_SIZE * KERNEL_SIZE +
                                                   kh * KERNEL_SIZE + kw;
                                    int32_t inIdx = ic * inSize * inSize + ih * inSize + iw;
                                    sum += input[inIdx] * weight[wIdx];
                                }
                            }
                        }
                        output[outBase + oh * outSize + ow] += sum;
                    }
                }
            }
        }

        tmpQueue2_.FreeTensor(tmpLocal);
    }

    __aicore__ inline void MaxPool2x2(LocalTensor<float>& input, LocalTensor<float>& output,
                                      int32_t channels, int32_t inSize)
    {
        int32_t outSize = inSize / 2;

        for (int32_t c = 0; c < channels; ++c) {
            for (int32_t oh = 0; oh < outSize; ++oh) {
                for (int32_t ow = 0; ow < outSize; ++ow) {
                    float maxVal = -3.4e38f;
                    int32_t baseIdx = c * inSize * inSize;
                    for (int32_t kh = 0; kh < 2; ++kh) {
                        for (int32_t kw = 0; kw < 2; ++kw) {
                            int32_t ih = oh * 2 + kh;
                            int32_t iw = ow * 2 + kw;
                            float val = input[baseIdx + ih * inSize + iw];
                            if (val > maxVal) maxVal = val;
                        }
                    }
                    int32_t outIdx = c * outSize * outSize + oh * outSize + ow;
                    output[outIdx] = maxVal;
                }
            }
        }
    }

    __aicore__ inline void FlattenAndLinear(LocalTensor<float>& input,
                                            GlobalTensor<float>& weight,
                                            GlobalTensor<float>& bias,
                                            LocalTensor<float>& output,
                                            int32_t inFeatures, int32_t outFeatures)
    {
        for (int32_t oc = 0; oc < outFeatures; ++oc) {
            float sum = bias[oc];
            for (int32_t ic = 0; ic < inFeatures; ++ic) {
                int32_t wIdx = oc * inFeatures + ic;
                sum += input[ic] * weight[wIdx];
            }
            output[oc] = sum;
        }
    }

    __aicore__ inline void Linear(LocalTensor<float>& input,
                                  GlobalTensor<float>& weight,
                                  GlobalTensor<float>& bias,
                                  LocalTensor<float>& output,
                                  int32_t inFeatures, int32_t outFeatures)
    {
        for (int32_t oc = 0; oc < outFeatures; ++oc) {
            float sum = bias[oc];
            for (int32_t ic = 0; ic < inFeatures; ++ic) {
                int32_t wIdx = oc * inFeatures + ic;
                sum += input[ic] * weight[wIdx];
            }
            output[oc] = sum;
        }
    }

    __aicore__ inline void ReluInPlace(LocalTensor<float>& tensor, int32_t length)
    {
        LocalTensor<float> tmpLocal = tmpQueue1_.AllocTensor<float>();
        AscendC::Mins(tmpLocal, tensor, 0.0f, length);
        AscendC::Sub(tensor, tensor, tmpLocal, length);
        tmpQueue1_.FreeTensor(tmpLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> conv1Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> pool1Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> conv2Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> pool2Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc1Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc2Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc3Queue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> conv1WGm_;
    AscendC::GlobalTensor<float> conv1BGm_;
    AscendC::GlobalTensor<float> conv2WGm_;
    AscendC::GlobalTensor<float> conv2BGm_;
    AscendC::GlobalTensor<float> fc1WGm_;
    AscendC::GlobalTensor<float> fc1BGm_;
    AscendC::GlobalTensor<float> fc2WGm_;
    AscendC::GlobalTensor<float> fc2BGm_;
    AscendC::GlobalTensor<float> fc3WGm_;
    AscendC::GlobalTensor<float> fc3BGm_;

    uint32_t batchSize_ {0};
    uint32_t numClasses_ {0};
    uint32_t totalLen_ {0};
};

extern "C" __global__ __aicore__ void lenet5_custom(GM_ADDR x, GM_ADDR y,
                                                     GM_ADDR conv1_w, GM_ADDR conv1_b,
                                                     GM_ADDR conv2_w, GM_ADDR conv2_b,
                                                     GM_ADDR fc1_w, GM_ADDR fc1_b,
                                                     GM_ADDR fc2_w, GM_ADDR fc2_b,
                                                     GM_ADDR fc3_w, GM_ADDR fc3_b,
