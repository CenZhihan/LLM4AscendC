#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR weight, GM_ADDR y,
        uint32_t inputBatch, uint32_t inputChannel, uint32_t inputDepth, uint32_t inputHeight, uint32_t inputWidth,
        uint32_t outputChannel, uint32_t kernelD, uint32_t kernelH, uint32_t kernelW,
        uint32_t strideD, uint32_t strideH, uint32_t strideW,
        uint32_t paddingD, uint32_t paddingH, uint32_t paddingW,
        uint32_t outputPaddingD, uint32_t outputPaddingH, uint32_t outputPaddingW,
        uint32_t dilationD, uint32_t dilationH, uint32_t dilationW,
        uint32_t groups, uint32_t outputDepth, uint32_t outputHeight, uint32_t outputWidth,
        uint32_t totalLength, uint32_t tileLength)
    {
        inputBatch_ = inputBatch;
        inputChannel_ = inputChannel;
        inputDepth_ = inputDepth;
        inputHeight_ = inputHeight;
        inputWidth_ = inputWidth;
        outputChannel_ = outputChannel;
        kernelD_ = kernelD;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        paddingD_ = paddingD;
        paddingH_ = paddingH;
        paddingW_ = paddingW;
        outputPaddingD_ = outputPaddingD;
        outputPaddingH_ = outputPaddingH;
        outputPaddingW_ = outputPaddingW;
        dilationD_ = dilationD;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;
        outputDepth_ = outputDepth;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        totalLength_ = totalLength;
        tileLength_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        inputSize_ = inputBatch * inputChannel * inputDepth * inputHeight * inputWidth;
        weightSize_ = outputChannel * inputChannel * kernelD * kernelH * kernelW;
        outputSize_ = outputChannel * outputDepth * outputHeight * outputWidth;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLength_ == 0 || tileLength_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLength_;
        uint32_t base = blockIdx * tileLength_;
        if (base >= totalLength_) return;

        uint32_t curLen = totalLength_ - base;
        if (curLen > tileLength_) curLen = tileLength_;

        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLength_) {
                nextLen = totalLength_ - nextBase;
                if (nextLen > tileLength_) nextLen = tileLength_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);

        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, weightGm_[0], weightSize_);
        inQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        AscendC::LocalTensor<float> yLocalDeq = outQueueY_.AllocTensor<float>();

        for (uint32_t n = 0; n < inputBatch_; ++n) {
            uint32_t outputBatchOffset = n * outputChannel_ * outputDepth_ * outputHeight_ * outputWidth_;
            uint32_t inputBatchOffset = n * inputChannel_ * inputDepth_ * inputHeight_ * inputWidth_;

            for (uint32_t oc = 0; oc < outputChannel_; ++oc) {
                uint32_t outputSpatialOffset = outputBatchOffset + oc * outputDepth_ * outputHeight_ * outputWidth_;
                
                for (uint32_t od = 0; od < outputDepth_; ++od) {
                    for (uint32_t oh = 0; oh < outputHeight_; ++oh) {
                        for (uint32_t ow = 0; ow < outputWidth_; ++ow) {
                            float sum = 0.0f;

                            int32_t inStartD = (int32_t)od - (int32_t)outputPaddingD_;
                            int32_t inStartH = (int32_t)oh - (int32_t)outputPaddingH_;
                            int32_t inStartW = (int32_t)ow - (int32_t)outputPaddingW_;

                            int32_t actualStartD = (inStartD + (int32_t)paddingD_) / (int32_t)strideD_;
                            int32_t actualStartH = (inStartH + (int32_t)paddingH_) / (int32_t)strideH_;
                            int32_t actualStartW = (inStartW + (int32_t)paddingW_) / (int32_t)strideW_;

                            int32_t actualEndD = (inStartD + (int32_t)paddingD_ + (int32_t)dilationD_ * ((int32_t)kernelD_ - 1)) / (int32_t)strideD_;
                            int32_t actualEndH = (inStartH + (int32_t)paddingH_ + (int32_t)dilationH_ * ((int32_t)kernelH_ - 1)) / (int32_t)strideH_;
                            int32_t actualEndW = (inStartW + (int32_t)paddingW_ + (int32_t)dilationW_ * ((int32_t)kernelW_ - 1)) / (int32_t)strideW_;

                            for (uint32_t ic = 0; ic < inputChannel_; ++ic) {
                                for (int32_t kd = 0; kd < (int32_t)kernelD_; ++kd) {
                                    int32_t id = actualStartD + kd * dilationD_;
                                    if (id < 0 || id >= (int32_t)inputDepth_) continue;

                                    for (int32_t kh = 0; kh < (int32_t)kernelH_; ++kh) {
                                        int32_t ih = actualStartH + kh * dilationH_;
                                        if (ih < 0 || ih >= (int32_t)inputHeight_) continue;

                                        for (int32_t kw = 0; kw < (int32_t)kernelW_; ++kw) {
                                            int32_t iw = actualStartW + kw * dilationW_;
                                            if (iw < 0 || iw >= (int32_t)inputWidth_) continue;

                                            int32_t effectiveId = (id * strideD_) - paddingD_ + kd * dilationD_;
                                            int32_t effectiveIh = (ih * strideH_) - paddingH_ + kh * dilationH_;
                                            int32_t effectiveIw = (iw * strideW_) - paddingW_ + kw * dilationW_;

                                            if (effectiveId < 0 || effectiveId >= (int32_t)inputDepth_) continue;
                                            if (effectiveIh < 0 || effectiveIh >= (int32_t)inputHeight_) continue;
                                            if (effectiveIw < 0 || effectiveIw >= (int32_t)inputWidth_) continue;

                                            uint32_t inputIdx = inputBatchOffset + ic * inputDepth_ * inputHeight_ * inputWidth_
                                                              + effectiveId * inputHeight_ * inputWidth_
                                                              + effectiveIh * inputWidth_
                                                              + effectiveIw;

                                            uint32_t weightIdx = oc * inputChannel_ * kernelD_ * kernelH_ * kernelW_
                                                               + ic * kernelD_ * kernelH_ * kernelW_
                                                               + kd * kernelH_ * kernelW_
                                                               + kh * kernelW_
                                                               + kw;

                                            sum += xLocal[inputIdx] * wLocal[weightIdx];
                                        }
                                    }
                                }
                            }

                            uint32_t outIdx = outputSpatialOffset + od * outputHeight_ * outputWidth_ + oh * outputWidth_ + ow;
                            yLocalDeq[outIdx] = sum;
                        }
                    }
                }
            }
        }

        outQueueY_.EnQue<float>(yLocalDeq);
        tmpQueue_.FreeTensor(tLocal);
        inQueueW_.FreeTensor(wLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t inputBatch_ {0};
    uint32_t inputChannel_ {0};
    uint32_t inputDepth_ {0};
    uint32_t inputHeight_ {0};
    uint32_t inputWidth_ {0};
    uint32_t outputChannel_ {0};
    uint32_t kernelD_ {0};
    uint32_t kernelH_ {0};
    uint32_t kernelW_ {0};
    uint32_t strideD_ {0};
    uint32_t strideH_ {0};
    uint32_t strideW_ {0};
    uint32_t paddingD_ {0};
    uint32_t paddingH_ {0};
    uint32_t paddingW_ {0};
    uint32_t outputPaddingD_ {0};
    uint32_t outputPaddingH_ {0};
    uint32_t outputPaddingW_ {0};
    uint32_t dilationD_ {0};
    uint32_t dilationH_ {0};
    uint32_t dilationW_ {0};
    uint32_t groups_ {0};
    uint32_t outputDepth_ {0};
    uint32_t outputHeight_ {0};
    uint32_t outputWidth_ {0};
    uint32_t totalLength_ {0};
    uint32_t tileLength_ {0};
    uint32_t inputSize_ {0};
    uint32_t weightSize_ {0};
    uint32_t outputSize_ {0};
};
