ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ConvStandard1dCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::Shape *weightShape = context->GetInputShape(1);
    const uint32_t N = inputShape->GetShapeVector()[0];
    const uint32_t Cin = inputShape->GetShapeVector()[1];
    const uint32_t Lin = inputShape->GetShapeVector()[2];
    const uint32_t Cout = weightShape->GetShapeVector()[0];
    const uint32_t Cw = weightShape->GetShapeVector()[1];
    const uint32_t K = weightShape->GetShapeVector()[2];

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int *stride = attrs->GetAttrPointer<int>(0);
    const int *padding = attrs->GetAttrPointer<int>(1);
    const int *dilation = attrs->GetAttrPointer<int>(2);
    const int *groups = attrs->GetAttrPointer<int>(3);

    uint32_t Lout = (Lin + 2 * (*padding) - (*dilation) * (K - 1) - 1) / (*stride) + 1;
    uint32_t totalLength = N * Cout * Lout;

    // compute tileLen based on UB budget
    uint32_t weightElems = Cout * Cin * K;
    uint32_t weightBytes = weightElems * sizeof(float);
    uint32_t ubBudgetBytes = 192 * 1024;
    uint32_t tileBudgetBytes = ubBudgetBytes - weightBytes;
    // each output element needs to load Cin*K input elements + 1 output.
    uint32_t bytesPerElem = (Cin * K + 1) * sizeof(float);
    uint32_t maxTile = tileBudgetBytes / bytesPerElem;
    // align down
    maxTile = (maxTile / ALIGN_ELEMS) * ALIGN_ELEMS;
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = (totalLength / ALIGN_ELEMS) * ALIGN_ELEMS;
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1 : (totalLength + tileLen - 1) / tileLen;
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(K);
    tiling.set_inChannels(Cin);
    tiling.set_outChannels(Cout);
    tiling.set_inputLength(Lin);
    tiling.set_stride(*stride);
    tiling.set_padding(*padding);
    tiling.set_dilation(*dilation);
    tiling.set_groups(*groups);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard1dCustom {
public:
    __aicore__ inline KernelConvStandard1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t kernelSize, uint32_t inChannels,
                                uint32_t outChannels, uint32_t inputLength,
                                uint32_t stride, uint32_t padding,
                                uint32_t dilation, uint32_t groups)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        kernelSize_ = kernelSize;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        inputLength_ = inputLength;
        stride_ = stride;
        padding_ = padding;
        dilation_ = dilation;
        groups_ = groups;

        // compute output length per channel
        Lout_ = (inputLength_ + 2 * padding_ - dilation_ * (kernelSize_ - 1) - 1) / stride_ + 1;

        // total input elements
        uint32_t N = totalLen_ / (outChannels_ * Lout_);
        totalInputLen_ = N * inChannels_ * inputLength_;
        weightLen_ = outChannels_ * inChannels_ * kernelSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, weightLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // init buffers: input queue, weight queue, output queue
        // weight queue size is weightLen_, input queue size is (tileLen_ + kernelSize_ - 1) * inChannels_
        // but we can allocate just enough for weight and a temporary.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, (tileLen_ + kernelSize_ - 1) * inChannels_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, weightLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // grid-stride loop
        uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // load weight into local memory once
        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, weightLen_);
        wQueue_.EnQue(wLocal);

        // process tiles
        while (true) {
            uint32_t curLen = (base + tileLen_ > totalLen_) ? (totalLen_ - base) : tileLen_;
            // compute tile of output
            ComputeTile(base, curLen, wLocal);
            // copy out
            AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);

            // move to next tile
            uint32_t nextBase = base + strideElems;
            if (nextBase >= totalLen_) break;
            base = nextBase;
        }

        // free weight
        wQueue_.FreeTensor(wLocal);
    }

private:
    __aicore__ inline void ComputeTile(uint32_t base, uint32_t curLen,
                                       AscendC::LocalTensor<float> &wLocal)
    {
        // allocate input segment: length = curLen + kernelSize_ - 1
        uint32_t inSegLen = curLen + kernelSize_ - 1;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        // load input segment: from global memory xGm_, start position depends on output position.
        // Determine starting output position: (n, cout, l) for base index.
        // Compute n, cout, l for first output in tile.
        uint32_t N = totalLen_ / (outChannels_ * Lout_);
        uint32_t Cout = outChannels_;
        uint32_t Lout = Lout_;
        uint32_t Cin = inChannels_;
        uint32_t K = kernelSize_;
        uint32_t stride = stride_;
        uint32_t padding = padding_;
        uint32_t dilation = dilation_;

        // Compute global input offset for start of this tile:
        // base corresponds to output index = base.
        // Determine the n, cout, l of this output index.
        uint32_t outIdx0 = base;
        uint32_t n0 = outIdx0 / (Cout * Lout);
        uint32_t rest0 = outIdx0 % (Cout * Lout);
        uint32_t cout0 = rest0 / Lout;
        uint32_t l0 = rest0 % Lout;

        // Compute the input start position for this tile: for channel 0, l0 * stride - padding.
        int32_t inStart = static_cast<int32_t>(l0 * stride) - static_cast<int32_t>(padding);
        // For each channel we will load inSegLen elements.
        // We'll load all channels consecutively: total input segment length = inSegLen * Cin.
        // Compute source pointer for xLocal: xGm_ offset for n0, channel 0, position inStart.
        uint32_t srcOffset = (n0 * Cin + 0) * inputLength_ + (inStart >= 0 ? inStart : 0);
        AscendC::DataCopy(xLocal, xGm_[srcOffset], inSegLen * Cin);
        inQueueX_.EnQue(xLocal);

        // For each output element i in tile (0..curLen-1):
        // compute output index = base + i
        // compute n, cout, l
        // compute sum
        // store result to yLocal

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < curLen; ++i) {
            uint32_t outIdx = base + i;
            uint32_t n = outIdx / (Cout * Lout);
            uint32_t rest = outIdx % (Cout * Lout);
            uint32_t cout = rest / Lout;
            uint32_t l = rest % Lout;

            float sum = 0.0f;
            // loop over channels and kernel
            for (uint32_t cin = 0; cin < Cin; ++cin) {

                    int32_t xIdx = static_cast<int32_t>(l * stride) + static_cast<int32_t>(kw * dilation) - static_cast<int32_t>(padding);
                    // clamp to valid range (should be within bounds due to padding)
                    uint32_t xIdxClamped = (xIdx >= 0) ? static_cast<uint32_t>(xIdx) : 0;
                    // compute offset in input segment: (cin * inSegLen) + (xIdxClamped - inStart)

                    sum += wVal * xVal;
                }

        outQueueY_.EnQue<float>(yLocal);
    }

    AscendC::GlobalTensor<float> yGm_;

    uint32_t tileLen_ {0};
    uint32_t kernelSize_ {3};

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard1dCustom {
public:
    __aicore__ inline KernelConvStandard1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t kernelSize, uint32_t inChannels,
                                uint32_t outChannels, uint32_t inputLength,
                                uint32_t stride, uint32_t padding,
                                uint32_t dilation, uint32_t groups)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        kernelSize_ = kernelSize;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        inputLength_ = inputLength;
        stride_ = stride;
        padding_ = padding;
        dilation_ = dilation;
        groups_ = groups;

        // compute output length per channel
        Lout_ = (inputLength_ + 2 * padding_ - dilation_ * (kernelSize_ - 1) - 1) / stride_ + 1;

        // total input and weight sizes
        uint32_t N = totalLen_ / (outChannels_ * Lout_);
        totalInputLen_ = N * inChannels_ * inputLength_;
        weightLen_ = outChannels_ * inChannels_ * kernelSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInputLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, weightLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // allocate double buffers for input segment and output
        // input segment length: tileLen + kernelSize - 1
        uint32_t inSegLen = tileLen_ + kernelSize_ - 1;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inSegLen * inChannels_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (true) {
            uint32_t curLen = (base + tileLen_ > totalLen_) ? (totalLen_ - base) : tileLen_;
            ComputeTile(base, curLen);
            // copy out
            AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);

            uint32_t nextBase = base + strideElems;
            if (nextBase >= totalLen_) break;
            base = nextBase;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t base, uint32_t curLen)
    {
        const uint32_t K = kernelSize_;
        const uint32_t Cin = inChannels_;
        const uint32_t Cout = outChannels_;
        const uint32_t Lout = Lout_;
        const uint32_t stride = stride_;
        const uint32_t padding = padding_;
        const uint32_t dilation = dilation_;

        // input segment length for this tile
        const uint32_t inSegLen = curLen + K - 1;

        // allocate input segment buffer
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();

        // compute starting output position (n, cout, l) for this tile
        uint32_t outIdx0 = base;
        uint32_t N = totalLen_ / (Cout * Lout);
        uint32_t n0 = outIdx0 / (Cout * Lout);
        uint32_t rest0 = outIdx0 % (Cout * Lout);
        uint32_t cout0 = rest0 / Lout;
        uint32_t l0 = rest0 % Lout;

        // compute start position in input for channel 0
        int32_t inStart = static_cast<int32_t>(l0 * stride) - static_cast<int32_t>(padding);
        uint32_t srcOffset = (n0 * Cin + 0) * inputLength_ + (inStart > 0 ? inStart : 0);
        // copy input segment for all channels
        AscendC::DataCopy(xLocal, xGm_[srcOffset], inSegLen * Cin);
        inQueueX_.EnQue(xLocal);

        // allocate output buffer
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // compute each output in tile
        for (uint32_t i = 0; i < curLen; ++i) {
            uint32_t outIdx = base + i;
            uint32_t n = outIdx / (Cout * Lout);
            uint32_t rest = outIdx % (Cout * Lout);
            uint32_t cout = rest / Lout;
            uint32_t l = rest % Lout;

            float sum = 0.0f;
            for (uint32_t cin = 0; cin < Cin; ++cin) {
                for (uint32_t kw = 0; kw < K; ++kw) {
                    int32_t xIdx = static_cast<int32_t>(l * stride) + static_cast<int32_t>(kw * dilation) - static_cast<int32_t>(padding);
                    uint32_t xIdxClamped = (xIdx >= 0) ? static_cast<uint32_t>(xIdx) : 0;
                    uint32_t offsetInSeg = cin * inSegLen + (xIdxClamped - (inStart > 0 ? inStart : 0));
                    float xVal = xLocal[offsetInSeg];
                    float wVal = wGm_[((cout * Cin) + cin) * K + kw];
                    sum += wVal * xVal;
                }
            }
            yLocal[i] = sum;
        }

        // free input segment, enque output
        inQueueX_.FreeTensor(xLocal);
        outQueueY_.EnQue<float>(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t kernelSize_ {3};
    uint32_t inChannels_ {64};
    uint32_t outChannels_ {128};
    uint32_t inputLength_ {131072};
    uint32_t stride_ {1};
    uint32_t padding_ {0};
    uint32_t dilation_ {1};
    uint32_t groups_ {1};
    uint32_t Lout_ {0};
    uint32_t totalInputLen_ {0};
    uint32_t weightLen_ {0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dCustom op;
    op.Init(x, w, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.kernelSize,
            tiling_data.inChannels,
            tiling_data.outChannels,
            tiling_data.inputLength,
            tiling_data.stride,
            tiling_data.padding,
            tiling_data.dilation,
            tiling_data.groups);
    op.Process();
}
