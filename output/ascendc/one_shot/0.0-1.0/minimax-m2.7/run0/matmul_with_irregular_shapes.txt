#include "matmul_with_irregular_shapes_custom_tiling.h"

namespace optiling {

static constexpr uint32_t BLOCK_DIM = 32;
static constexpr uint32_t M_TILE = 128;
static constexpr uint32_t K_TILE = 128;
static constexpr uint32_t N_TILE = 128;
static constexpr uint32_t K_SUB_TILE = 32;
static constexpr uint32_t UB_SIZE_BYTES = 60 * 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithIrregularShapesCustomTilingData tiling;
    
    const uint32_t mSize = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t kSize = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());
    const uint32_t nSize = static_cast<uint32_t>(context->GetOutputShape(0)->GetOriginShape().GetShapeSize());
    
    const uint32_t M = mSize;
    const uint32_t K = kSize / M;
    const uint32_t N = nSize / M;
    
    uint32_t mTile = M_TILE;
    uint32_t kTile = K_TILE;
    uint32_t nTile = N_TILE;
    
    if (M < mTile) mTile = AlignDown(M, 16);
    if (K < kTile) kTile = AlignDown(K, 16);
    if (N < nTile) nTile = AlignDown(N, 16);
    
    mTile = (mTile > 0) ? mTile : 16;
    kTile = (kTile > 0) ? kTile : 16;
    nTile = (nTile > 0) ? nTile : 16;
    
    uint32_t numBlocks = CeilDiv(M, mTile);
    if (numBlocks < BLOCK_DIM) numBlocks = BLOCK_DIM;
    if (numBlocks > 64) numBlocks = 64;
    
    context->SetBlockDim(numBlocks);
    
    tiling.set_mSize(M);
    tiling.set_kSize(K);
    tiling.set_nSize(N);
    tiling.set_mTile(mTile);
    tiling.set_kTile(kTile);
    tiling.set_nTile(nTile);
    tiling.set_blockDim(numBlocks);
    tiling.set_mOffset(0);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulIrregular {
public:
    __aicore__ inline KernelMatmulIrregular() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t mSize, uint32_t kSize, uint32_t nSize,
                                uint32_t mTile, uint32_t kTile, uint32_t nTile,
                                uint32_t blockIdx)
    {
        mSize_ = mSize;
        kSize_ = kSize;
        nSize_ = nSize;
        mTile_ = mTile;
        kTile_ = kTile;
        nTile_ = nTile;
        
        aGm_.SetGlobalBuffer((__gm__ float*)a, mSize * kSize);
        bGm_.SetGlobalBuffer((__gm__ float*)b, kSize * nSize);
        cGm_.SetGlobalBuffer((__gm__ float*)c, mSize * nSize);
        
        blockIdx_ = blockIdx;
        
        const uint32_t kSubTile = 32;
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, mTile_ * kSubTile * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, kSubTile * nTile_ * sizeof(float));
        pipe_.InitBuffer(ubufC_, mTile_ * nTile_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        
        uint32_t mStart = blockIdx_ * mTile_;
        if (mStart >= mSize_) return;
        
        const uint32_t mEnd = (mStart + mTile_ < mSize_) ? mStart + mTile_ : mSize_;
        const uint32_t mLen = mEnd - mStart;
        
        for (uint32_t nStart = 0; nStart < nSize_; nStart += nTile_) {
            const uint32_t nEnd = (nStart + nTile_ < nSize_) ? nStart + nTile_ : nSize_;
            const uint32_t nLen = nEnd - nStart;
            
            ComputeBlock(mStart, mLen, nStart, nLen);
        }
    }

private:
    __aicore__ inline void ComputeBlock(uint32_t mStart, uint32_t mLen, uint32_t nStart, uint32_t nLen)
    {
        AscendC::LocalTensor<float> cLocal = ubufC_.AllocTensor<float>();
        AscendC::Sets(cLocal, 0.0f, mLen * nLen);
        
        const uint32_t kSubTile = 32;
        
        for (uint32_t kPos = 0; kPos < kSize_; kPos += kSubTile) {
            const uint32_t kRemain = kSize_ - kPos;
            const uint32_t kCur = (kRemain >= kSubTile) ? kSubTile : kRemain;
            
            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            for (uint32_t mi = 0; mi < mLen; ++mi) {
                uint32_t srcOffset = (mStart + mi) * kSize_ + kPos;
                AscendC::DataCopyExtParam aParam;
                aParam.srcStride = kSize_ - kCur;
                aParam.dstStride = 0;
                aParam.srcRepStride = 0;
                aParam.dstRepStride = 0;
                AscendC::DataCopy(aLocal[mi * kCur], aGm_[srcOffset], kCur, aParam);
            }
            inQueueA_.EnQue(aLocal);
            
            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
            uint32_t bOffset = kPos * nSize_ + nStart;
            AscendC::DataCopy(bLocal, bGm_[bOffset], kCur * nLen);
            inQueueB_.EnQue(bLocal);
            
            AscendC::LocalTensor<float> aDeq = inQueueA_.DeQue<float>();
            AscendC::LocalTensor<float> bDeq = inQueueB_.DeQue<float>();
            
            for (uint32_t ki = 0; ki < kCur; ++ki) {
                for (uint32_t ni = 0; ni < nLen; ++ni) {
                    float bVal = bDeq[ki * nLen + ni];
                    for (uint32_t mi = 0; mi < mLen; ++mi) {
                        float aVal = aDeq[mi * kCur + ki];
                        cLocal[mi * nLen + ni] += aVal * bVal;
                    }
                }
            }
            
            inQueueA_.FreeTensor(aDeq);
            inQueueB_.FreeTensor(bDeq);
        }
        
        uint32_t cOffset = mStart * nSize_ + nStart;
        AscendC::DataCopyExtParam cParam;
        cParam.srcStride = nLen - mLen;
        cParam.dstStride = nSize_ - mLen;
        cParam.srcRepStride = nLen * (mLen - 1);
        cParam.dstRepStride = nSize_ * (mLen - 1);
        AscendC::DataCopy(cGm_[cOffset], cLocal, mLen * nLen, cParam);
        
        ubufC_.FreeTensor(cLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> ubufC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t mSize_ {0};
    uint32_t kSize_ {0};
    uint32_t nSize_ {0};
    uint32_t mTile_ {0};
    uint32_t kTile_ {0};
    uint32_t nTile_ {0};
    uint32_t blockIdx_ {0};
};

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulIrregular op;
    op.Init(a, b, c, 
            tiling_data.mSize, tiling_data.kSize, tiling_data.nSize,
            tiling_data.mTile, tiling_data.kTile, tiling_data.nTile,
            tiling_data.blockDim);
    op.Process();
}
