class KernelSquareMatrixMultiplication {
public:
    __aicore__ inline KernelSquareMatrixMultiplication() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, N * N);
        bGm_.SetGlobalBuffer((__gm__ float*)B, N * N);
        cGm_.SetGlobalBuffer((__gm__ float*)C, N * N);

        const uint32_t UB_SIZE = 32 * 1024;
        pipe_.InitBuffer(inQueueA_, 2, tileM * tileK * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 2, tileK * tileN * sizeof(float));
        pipe_.InitBuffer(outQueueC_, 2, tileM * tileN * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, BLOCK_SIZE * BLOCK_SIZE * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = GetBlockIdx();
        const uint32_t numBlocks = GetBlockNum();

        const uint32_t blocksM = (N_ + tileM_ - 1) / tileM_;
        const uint32_t blocksN = (N_ + tileN_ - 1) / tileN_;
        const uint32_t totalBlocks = blocksM * blocksN;

        if (blockIdx >= totalBlocks) return;

        const uint32_t blockM = blockIdx / blocksN;
        const uint32_t blockN = blockIdx % blocksN;

        const uint32_t mStart = blockM * tileM_;
        const uint32_t nStart = blockN * tileN_;
        const uint32_t mEnd = std::min(mStart + tileM_, N_);
        const uint32_t nEnd = std::min(nStart + tileN_, N_);

        const uint32_t mLen = mEnd - mStart;
        const uint32_t nLen = nEnd - nStart;

        // Clear output tile
        for (uint32_t i = 0; i < mLen; ++i) {
            for (uint32_t j = 0; j < nLen; ++j) {
                cLocal[i * nLen + j] = 0.0f;
            }
        }

        // GEMM: C[m,n] += A[m,k] * B[k,n]
        for (uint32_t k = 0; k < N_; k += tileK_) {
            const uint32_t kEnd = std::min(k + tileK_, N_);
            const uint32_t kLen = kEnd - k;

            // Load tiles
            LoadTileA(mStart, k, mLen, kLen);
            LoadTileB(k, nStart, kLen, nLen);

            // Multiply and accumulate
            Gemm(mLen, kLen, nLen);
        }

        // Store result
        StoreTileC(mStart, nStart, mLen, nLen);
    }

private:
    __aicore__ inline void LoadTileA(uint32_t m, uint32_t k, uint32_t mLen, uint32_t kLen)
    {
        AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
        for (uint32_t i = 0; i < mLen; ++i) {
            for (uint32_t j = 0; j < kLen; ++j) {
                aLocal[i * kLen + j] = aGm_[(m + i) * N_ + (k + j)];
            }
        }
        inQueueA_.EnQue(aLocal);
    }

    __aicore__ inline void LoadTileB(uint32_t k, uint32_t n, uint32_t kLen, uint32_t nLen)
    {
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        for (uint32_t i = 0; i < kLen; ++i) {
            for (uint32_t j = 0; j < nLen; ++j) {
                bLocal[i * nLen + j] = bGm_[(k + i) * N_ + (n + j)];
            }
        }
        inQueueB_.EnQue(bLocal);
    }

    __aicore__ inline void Gemm(uint32_t mLen, uint32_t kLen, uint32_t nLen)
    {
        AscendC::LocalTensor<float> aLocal = inQueueA_.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB_.DeQue<float>();
        AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();

        // Matrix multiplication using vector instructions
        for (uint32_t i = 0; i < mLen; ++i) {
            for (uint32_t j = 0; j < nLen; ++j) {
                float sum = 0.0f;
                for (uint32_t k = 0; k < kLen; ++k) {
                    sum += aLocal[i * kLen + k] * bLocal[k * nLen + j];
                }
                cLocal[i * nLen + j] += sum;
            }
        }

        outQueueC_.EnQue(cLocal);
        inQueueA_.FreeTensor(aLocal);
        inQueueB_.FreeTensor(bLocal);
    }

    __aicore__ inline void StoreTileC(uint32_t m, uint32_t n, uint32_t mLen, uint32_t nLen)
    {
        AscendC::LocalTensor<float> cLocal = outQueueC_.DeQue<float>();

            for (uint32_t j = 0; j < nLen; ++j) {
                cGm_[(m + i) * N_ + (n + j)] = cLocal[i * nLen + j];
            }
        }
        outQueueC_.EnQue(cLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_;
    uint32_t tileM_;
    uint32_t tileN_;
    uint32_t tileK_;

};

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,

        tileLen_ = tileLen;

        aGm_.SetGlobalBuffer((__gm__ float*)A, (size_t)N * N);
        bGm_.SetGlobalBuffer((__gm__ float*)B, (size_t)N * N);
        cGm_.SetGlobalBuffer((__gm__ float*)C, (size_t)N * N);

        const uint32_t aTileSize = tileLen_ * tileLen_;
        const uint32_t bTileSize = tileLen_ * tileLen_;
        const uint32_t cTileSize = tileLen_ * tileLen_;
        const uint32_t totalSize = aTileSize + bTileSize + cTileSize;

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, aTileSize * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, bTileSize * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, cTileSize * sizeof(float));
