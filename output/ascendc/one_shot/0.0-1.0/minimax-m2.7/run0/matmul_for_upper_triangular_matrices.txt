static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForUpperTriangularMatricesCustomTilingData tiling;
    
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    
    uint32_t m = static_cast<uint32_t>(a_shape->GetShapeSize());
    uint32_t n = static_cast<uint32_t>(b_shape->GetShapeSize());
    uint32_t k = static_cast<uint32_t>(a_shape->GetShape(1));
    
    uint32_t rowTileLen = UB_BUDGET_BYTES / (3 * sizeof(float));
    rowTileLen = AlignDown(rowTileLen, ALIGN_ELEMS);
    if (rowTileLen < ALIGN_ELEMS) rowTileLen = ALIGN_ELEMS;
    if (rowTileLen > HARD_CAP_ELEMS) rowTileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    
    uint32_t colTileLen = rowTileLen;
    uint32_t kTileLen = rowTileLen;
    
    uint32_t totalRowTiles = CeilDiv(m, rowTileLen);
    uint32_t totalColTiles = CeilDiv(n, colTileLen);
    uint32_t totalTiles = totalRowTiles * totalColTiles;
    
    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;
    
    context->SetBlockDim(blockDim);
    
    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_rowTileLength(rowTileLen);
    tiling.set_colTileLength(colTileLen);
    tiling.set_kTileLength(kTileLen);
    tiling.set_blockDim(blockDim);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulForUpperTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForUpperTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t rowTileLen, uint32_t colTileLen, uint32_t kTileLen)
    {
        m_ = m;
        n_ = n;
        k_ = k;
        rowTileLen_ = rowTileLen;
        colTileLen_ = colTileLen;
        kTileLen_ = kTileLen;
        
        aGm_.SetGlobalBuffer((__gm__ float*)a, m_ * k_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, k_ * n_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, m_ * n_);
        
        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, rowTileLen_ * kTileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, kTileLen_ * colTileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, rowTileLen_ * colTileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, rowTileLen_ * colTileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || n_ == 0 || k_ == 0) return;
        
        const uint32_t totalRowTiles = CeilDiv(m_, rowTileLen_);
        const uint32_t totalColTiles = CeilDiv(n_, colTileLen_);
        
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalBlocks = static_cast<uint32_t>(AscendC::GetBlockNum());
        
        uint32_t blockRow = blockIdx / totalColTiles;
        uint32_t blockCol = blockIdx % totalColTiles;
        
        uint32_t rowStart = blockRow * rowTileLen_;
        uint32_t colStart = blockCol * colTileLen_;
        
        uint32_t rowLen = (rowStart + rowTileLen_ > m_) ? (m_ - rowStart) : rowTileLen_;
        uint32_t colLen = (colStart + colTileLen_ > n_) ? (n_ - colStart) : colTileLen_;
        
        if (rowLen == 0 || colLen == 0) return;
        
        uint32_t iStart = rowStart;
        uint32_t jStart = colStart;
        
        if (iStart > jStart) return;
        
        AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpBuf_.AllocTensor<float>();
        
        AscendC::Sets(cLocal, 0.0f, rowLen * colLen);
        
        for (uint32_t kOffset = 0; kOffset < k_; kOffset += kTileLen_) {
            uint32_t curK = (kOffset + kTileLen_ > k_) ? (k_ - kOffset) : kTileLen_;
            
            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
            
            uint32_t aRowStart = iStart;
            uint32_t bColStart = jStart;
            uint32_t aRowLen = rowLen;
            uint32_t bColLen = colLen;
            
            uint32_t actualARowLen = 0;
            for (uint32_t r = 0; r < aRowLen; ++r) {
                uint32_t globalRow = aRowStart + r;
                if (globalRow < k_) {
                    actualARowLen = r + 1;
                }
            }
            if (actualARowLen > curK) actualARowLen = curK;
            
            uint32_t actualBColLen = 0;
            for (uint32_t c = 0; c < bColLen; ++c) {
                uint32_t globalCol = bColStart + c;
                if (globalCol < k_) {
                    actualBColLen = c + 1;
                }
            }
            if (actualBColLen > curK) actualBColLen = curK;
            
            if (actualARowLen > 0 && actualBColLen > 0) {
                AscendC::DataCopy(aLocal, aGm_.GetOffset(aRowStart * k_ + kOffset), 
                                  actualARowLen * curK * sizeof(float));
                AscendC::DataCopy(bLocal, bGm_.GetOffset(kOffset * n_ + bColStart), 
                                  curK * actualBColLen * sizeof(float));
                
                AscendC::MatMul(tmpLocal, aLocal, bLocal, 
                                actualARowLen, curK, actualBColLen,
                                0, 0, 0, 0, 1.0f);
                
                AscendC::Add(cLocal, cLocal, tmpLocal, rowLen * colLen);
            }
            
            inQueueA_.FreeTensor(aLocal);
            inQueueB_.FreeTensor(bLocal);
        }
        
        AscendC::DataCopy(cGm_.GetOffset(iStart * n_ + jStart), cLocal, 
                          rowLen * colLen * sizeof(float));
        
        tmpBuf_.FreeTensor(tmpLocal);
        outQueueC_.FreeTensor(cLocal);
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b)
    {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t m_ {0};
    uint32_t n_ {0};
    uint32_t k_ {0};
    uint32_t rowTileLen_ {0};
    uint32_t colTileLen_ {0};
    uint32_t kTileLen_ {0};
};

extern "C" __global__ __aicore__ void matmul_for_upper_triangular_matrices_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForUpperTriangularMatricesCustom op;
    op.Init(a, b, c, tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.rowTileLength, tiling_data.colTileLength, tiling_data.kTileLength);
    op.Process();
}
