constexpr uint32_t BLOCK_DIM = 8;
constexpr uint32_t TILE_LENGTH = 32768;
constexpr uint32_t UNROLL_FACTOR = 8;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}
    
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t totalLength, 
                                uint32_t tileLength, uint32_t innerSize, float eps) {
        totalLength_ = totalLength;
        tileLength_ = tileLength;
        innerSize_ = innerSize;
        eps_ = eps;
        
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength_);
        
        pipe_.InitBuffer(inQueueX_, BLOCK_DIM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BLOCK_DIM, tileLength_ * sizeof(float));
    }
    
    __aicore__ inline void Process() {
        for (uint32_t tileId = 0; tileId < totalLength_; tileId += tileLength_) {
            uint32_t actualTileLength = std::min(tileLength_, totalLength_ - tileId);
            CopyIn(tileId, actualTileLength);
            Compute(actualTileLength);
            CopyOut(tileId, actualTileLength);
        }
    }
    
private:
    __aicore__ inline void CopyIn(uint32_t offset, uint32_t length) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[offset], length);
        inQueueX_.EnQue(xLocal);
    }
    
    __aicore__ inline void Compute(uint32_t length) {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        
        // Compute mean
        float mean = 0.0f;
        for (uint32_t i = 0; i < length; ++i) {
            mean += xLocal[i];
        }
        mean /= length;
        
        // Compute variance
        float variance = 0.0f;
        for (uint32_t i = 0; i < length; ++i) {
            float diff = xLocal[i] - mean;
            variance += diff * diff;
        }
        variance /= length;
        
        // Normalize
        float sqrtVar = std::sqrt(variance + eps_);
        for (uint32_t i = 0; i < length; ++i) {
            yLocal[i] = (xLocal[i] - mean) / sqrtVar;
        }
        
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }
    
    __aicore__ inline void CopyOut(uint32_t offset, uint32_t length) {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[offset], yLocal, length);
        outQueueY_.FreeTensor(yLocal);
    }
    
private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BLOCK_DIM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BLOCK_DIM> outQueueY_;
    
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    
    uint32_t totalLength_;
    uint32_t tileLength_;
    uint32_t innerSize_;
    float eps_;
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling) {

    KernelLayerNormCustom op;
