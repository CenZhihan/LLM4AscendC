kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight,
                                uint32_t ni, uint32_t ci, uint32_t wi, uint32_t hi, uint32_t di,
                                uint32_t co, uint32_t kw, uint32_t kh, uint32_t kd,
                                uint32_t wo, uint32_t ho, uint32_t do_)
    {
        ni_ = ni;
        ci_ = ci;
        wi_ = wi;
        hi_ = hi;
        di_ = di;
        co_ = co;
        kw_ = kw;
        kh_ = kh;
        kd_ = kd;
        wo_ = wo;
        ho_ = ho;
        do__ = do_;

        xGm_.SetGlobalBuffer((__gm__ float *)x);
        yGm_.SetGlobalBuffer((__gm__ float *)y);
        wGm_.SetGlobalBuffer((__gm__ float *)weight);

        const uint32_t MAX_UB_ELEMS = 60 * 1024 / sizeof(float);
        uint32_t inTileSize = ci_ * kw_ * kh_ * kd_;
        uint32_t outTileSize = wo_ * ho_ * do__;

        inTileLength_ = (inTileSize < MAX_UB_ELEMS) ? inTileSize : MAX_UB_ELEMS;
        outTileLength_ = (outTileSize < MAX_UB_ELEMS) ? outTileSize : MAX_UB_ELEMS;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inTileLength_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, 1, inTileSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outTileLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (co_ == 0 || ni_ == 0 || ci_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        uint32_t coPerBlock = (co_ + blockNum - 1) / blockNum;
        uint32_t coStart = blockIdx * coPerBlock;
        uint32_t coEnd = (coStart + coPerBlock > co_) ? co_ : coStart + coPerBlock;

        if (coStart >= co_) return;

        uint32_t filterSize = ci_ * kw_ * kh_ * kd_;
        LoadWeight(coStart, filterSize);

        for (uint32_t coIdx = coStart; coIdx < coEnd; ++coIdx) {
            for (uint32_t nIdx = 0; nIdx < ni_; ++nIdx) {
                Compute(nIdx, coIdx);
            }
        }
    }

private:
    __aicore__ inline void LoadWeight(uint32_t coIdx, uint32_t filterSize)
    {
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        for (uint32_t i = 0; i < filterSize; ++i) {
            uint32_t wOffset = ((coIdx * ci_ + i / (kw_ * kh_ * kd_)) * kw_ * kh_ * kd_) + (i % (kw_ * kh_ * kd_));
            uint32_t ciIdx = i / (kw_ * kh_ * kd_);
            uint32_t kkIdx = (i % (kw_ * kh_ * kd_)) / (kh_ * kd_);
            uint32_t khIdx = (i % (kh_ * kd_)) / kd_;
            uint32_t kdIdx = i % kd_;
            uint32_t gmOffset = ((coIdx * ci_ + ciIdx) * kw_ + kkIdx) * kh_ + khIdx) * kd_ + kdIdx;
            wLocal.SetValue(i, wGm_.GetValue(gmOffset));
        }
        inQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void Compute(uint32_t nIdx, uint32_t coIdx)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();

        for (uint32_t dIdx = 0; dIdx < do__; ++dIdx) {
            for (uint32_t hIdx = 0; hIdx < ho_; ++hIdx) {
                for (uint32_t wIdx = 0; wIdx < wo_; ++wIdx) {
                    uint32_t outIdx = ((dIdx * ho_ + hIdx) * wo_ + wIdx);
                    float sum = 0.0f;

                    for (uint32_t kdIdx = 0; kdIdx < kd_; ++kdIdx) {
                        for (uint32_t khIdx = 0; khIdx < kh_; ++khIdx) {
                            for (uint32_t kwIdx = 0; kwIdx < kw_; ++kwIdx) {
                                for (uint32_t ciIdx = 0; ciIdx < ci_; ++ciIdx) {
                                    uint32_t xW = wIdx + kwIdx;
                                    uint32_t xH = hIdx + khIdx;
                                    uint32_t xD = dIdx + kdIdx;

                                    if (xW < wi_ && xH < hi_ && xD < di_) {
                                        uint32_t xOffset = (((nIdx * ci_ + ciIdx) * di_ + xD) * hi_ + xH) * wi_ + xW;
                                        uint32_t wOffset = (((ciIdx * kw_ + kwIdx) * kh_ + khIdx) * kd_ + kdIdx);
                                        sum += xGm_.GetValue(xOffset) * wLocal.GetValue(wOffset);
                                    }
                                }
                            }
                        }
                    }
                    yLocal.SetValue(outIdx, sum);
                }
            }
        }

        uint32_t outBase = ((nIdx * co_ + coIdx) * do__ * ho_ * wo_);
        for (uint32_t i = 0; i < do__ * ho_ * wo_; ++i) {
            yGm_.SetValue(outBase + i, yLocal.GetValue(i));
        }

        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wGm_;

    uint32_t ni_{0};
    uint32_t ci_{0};
    uint32_t wi_{0};
    uint32_t hi_{0};
    uint32_t di_{0};
    uint32_t co_{0};
    uint32_t kw_{0};
    uint32_t kh_{0};
    uint32_t kd_{0};
    uint32_t wo_{0};
    uint32_t ho_{0};
    uint32_t do__ {0};
    uint32_t inTileLength_{0};
    uint32_t outTileLength_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dSquareInputAsymmetricKernelCustom op;
    op.Init(x, y, weight,
            tiling_data.ni, tiling_data.ci, tiling_data.wi, tiling_data.hi, tiling_data.di,
            tiling_data.co, tiling_data.kw, tiling_data.kh, tiling_data.kd,
            tiling_data.wo, tiling_data.ho, tiling_data.do_);
    op.Process();
}
