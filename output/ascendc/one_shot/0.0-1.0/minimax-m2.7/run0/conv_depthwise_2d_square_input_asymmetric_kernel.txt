//   inQueueX: 2 * tileHeight * tileWidth
//   inQueueW: 1 * kernelSize
//   outQueueY: 2 * tileHeight * tileWidth
//   tmpQueue: 1 * tileHeight * tileWidth
// Total floats = 6 * tileHeight * tileWidth + kernelSize
// Conservative UB budget: 64 * 1024 bytes => max ~16K elements
static constexpr uint32_t UB_BUDGET_BYTES = 64 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dSquareInputAsymmetricKernelCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::Shape *weightShape = context->GetInputShape(1);

    const uint32_t c = static_cast<uint32_t>(inputShape->GetShape().at(1));
    const uint32_t h = static_cast<uint32_t>(inputShape->GetShape().at(2));
    const uint32_t w = static_cast<uint32_t>(inputShape->GetShape().at(3));
    const uint32_t kernelSize = static_cast<uint32_t>(weightShape->GetShape().at(2));

    const int32_t *dilationAttr = attrs->GetAttrPointer<int32_t>(2);

    const uint32_t stride = (strideAttr != nullptr) ? static_cast<uint32_t>(*strideAttr) : 1;
    const uint32_t padding = (paddingAttr != nullptr) ? static_cast<uint32_t>(*paddingAttr) : 0;
    const uint32_t dilation = (dilationAttr != nullptr) ? static_cast<uint32_t>(*dilationAttr) : 1;

    const uint32_t outH = (h + 2 * padding - dilation * (kernelSize - 1) - 1) / stride + 1;
    const uint32_t outW = w;

    const uint32_t totalLength = c * outH * outW;

    // Tile dimensions based on UB budget
    // UB usage: (6 * tileH * tileW + kernelSize) * sizeof(float)
    uint32_t maxTileElems = UB_BUDGET_BYTES / (6 * sizeof(float));
    maxTileElems = AlignDown(maxTileElems, ALIGN_ELEMS);
    if (maxTileElems < ALIGN_ELEMS) maxTileElems = ALIGN_ELEMS;
    if (maxTileElems > HARD_CAP_ELEMS) maxTileElems = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // For square input, use tileHeight = tileWidth for simplicity
    uint32_t tileSide = static_cast<uint32_t>(sqrt(static_cast<double>(maxTileElems)));
    tileSide = AlignDown(tileSide, 16u);
    if (tileSide < 16) tileSide = 16;

    uint32_t tileHeight = tileSide;
    uint32_t tileWidth = tileSide;

    const uint32_t numChannelTiles = c;

    uint32_t blockDim = numHTiles * numWTiles * numChannelTiles;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_KERNEL_SIZE = 32;

class KernelConvDepthwise2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight,
                                uint32_t h, uint32_t w, uint32_t c,
                                uint32_t kernelSize, uint32_t stride,
                                uint32_t padding, uint32_t dilation,
                                uint32_t outH, uint32_t outW,
                                uint32_t tileHeight, uint32_t tileWidth)
    {
        h_ = h;
        w_ = w;
        c_ = c;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        dilation_ = dilation;
        outH_ = outH;
        outW_ = outW;
        tileH_ = tileHeight;
        tileW_ = tileWidth;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);

        const uint32_t tileSize = tileH_ * tileW_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileSize * sizeof(float));
        pipe_.InitBuffer(weightQueue_, 1, kernelSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outH_ == 0 || outW_ == 0 || c_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid: (outH/tileH) * (outW/tileW) * c
        const uint32_t numHTiles = (outH_ + tileH_ - 1) / tileH_;
        const uint32_t numWTiles = (outW_ + tileW_ - 1) / tileW_;
        const uint32_t numChannelTiles = c_;

        const uint32_t totalBlocks = numHTiles * numWTiles * numChannelTiles;
        if (totalBlocks == 0) return;

        // Stride-style load balancing
        uint32_t taskId = blockIdx;
        uint32_t currentBlockIdx = 0;

        for (uint32_t cIdx = 0; cIdx < numChannelTiles; ++cIdx) {
            for (uint32_t hIdx = 0; hIdx < numHTiles; ++hIdx) {
                for (uint32_t wIdx = 0; wIdx < numWTiles; ++wIdx) {
                    if (currentBlockIdx == taskId % totalBlocks) {
                        ProcessTile(cIdx, hIdx, wIdx);
                        return;
                    }
                    currentBlockIdx++;
                }
            }
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t cIdx, uint32_t hIdx, uint32_t wIdx)
    {
        const uint32_t outHStart = hIdx * tileH_;
        const uint32_t outWStart = wIdx * tileW_;

        const uint32_t curTileH = (outHStart + tileH_ > outH_) ? (outH_ - outHStart) : tileH_;
        const uint32_t curTileW = (outWStart + tileW_ > outW_) ? (outW_ - outWStart) : tileW_;

        const uint32_t tileSize = tileH_ * tileW_;
        uint32_t curTileSize = curTileH * curTileW;

        // Load weight for this channel
        LoadWeight(cIdx);

        // Copy input tile
        CopyIn(cIdx, outHStart, outWStart, curTileH, curTileW);

        // Compute
        Compute(curTileH, curTileW, cIdx);

        // Copy output
        CopyOut(cIdx, outHStart, outWStart, curTileH, curTileW);
    }

    __aicore__ inline void LoadWeight(uint32_t cIdx)
    {
        AscendC::LocalTensor<float> weightLocal = weightQueue_.AllocTensor<float>();
        for (uint32_t k = 0; k < kernelSize_; ++k) {
            weightLocal.SetValue(k, weightGm_.GetValue(cIdx * kernelSize_ + k));
        }
        weightQueue_.EnQue(weightLocal);
    }

    __aicore__ inline void CopyIn(uint32_t cIdx, uint32_t outHStart, uint32_t outWStart,
                                   uint32_t tileH, uint32_t tileW)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();

        // Calculate input region based on output position (with stride, padding, dilation)
        const uint32_t inHStartBase = outHStart * stride_;
        const uint32_t inHStart = (inHStartBase >= padding_) ? (inHStartBase - padding_) : 0;

        // For each position in output tile, calculate corresponding input positions
        uint32_t idx = 0;
        for (uint32_t ih = 0; ih < tileH; ++ih) {
            for (uint32_t iw = 0; iw < tileW; ++iw) {
                const uint32_t outHPos = outHStart + ih;
                const uint32_t outWPos = outWStart + iw;

                if (outHPos < outH_ && outWPos < outW_) {
                    const uint32_t actualInHStart = outHPos * stride_ + padding_;
                    const uint32_t actualInH = actualInHStart;

                    for (uint32_t k = 0; k < kernelSize_; ++k) {
                        const uint32_t inH = actualInH + k * dilation_;
                        const uint32_t inW = outWPos;

                        if (inH < h_ && inW < w_) {
                            const uint32_t inIdx = cIdx * h_ * w_ + inH * w_ + inW;
                            xLocal.SetValue(idx + k * tileH * tileW, xGm_.GetValue(inIdx));
                        }
                    }
                }
                idx++;
            }
        }

        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t tileH, uint32_t tileW, uint32_t cIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> weightLocal = weightQueue_.DeQue<float>();

        const uint32_t tileSize = tileH_ * tileW_;
        const uint32_t curSize = tileH * tileW;

        // Initialize output to zero
        AscendC::Sets(yLocal, 0.0f, curSize);

        // Depthwise convolution: for each kernel element, multiply and accumulate
        for (uint32_t k = 0; k < kernelSize_; ++k) {
            // Extract k-th slice from input
            AscendC::LocalTensor<float> xSlice = xLocal[k * tileSize];

            // Multiply with weight
            AscendC::Muls(tLocal, xSlice, weightLocal.GetValue(k), curSize);

            // Accumulate to output
            AscendC::Add(yLocal, yLocal, tLocal, curSize);
        }

        tmpQueue_.FreeTensor(tLocal);
        weightQueue_.FreeTensor(weightLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t cIdx, uint32_t outHStart, uint32_t outWStart,
                                    uint32_t tileH, uint32_t tileW)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();

        for (uint32_t ih = 0; ih < tileH; ++ih) {
            for (uint32_t iw = 0; iw < tileW; ++iw) {
                const uint32_t outHPos = outHStart + ih;
                const uint32_t outWPos = outWStart + iw;

                if (outHPos < outH_ && outWPos < outW_) {
                    const uint32_t outIdx = cIdx * outH_ * outW_ + outHPos * outW_ + outWPos;
                    const uint32_t localIdx = ih * tileW_ + iw;
                    yGm_.SetValue(outIdx, yLocal.GetValue(localIdx));
                }
            }
        }

        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;

    uint32_t h_ {0};
    uint32_t w_ {0};
    uint32_t c_ {0};
    uint32_t kernelSize_ {3};
    uint32_t stride_ {1};
    uint32_t padding_ {0};
    uint32_t dilation_ {1};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
    uint32_t tileH_ {64};
    uint32_t tileW_ {64};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, y, weight,
            tiling_data.h, tiling_data.w, tiling_data.c,
            tiling_data.kernelSize, tiling_data.stride,
            tiling_data.padding, tiling_data.dilation,
            tiling_data.outH, tiling_data.outW,
            tiling_data.tileHeight, tiling_data.tileWidth);
    op.Process();
}
