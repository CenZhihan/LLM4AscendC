#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dGrouped {
public:
    __aicore__ inline KernelConvTransposed3dGrouped() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR weight, GM_ADDR y,
        uint32_t batchSize, uint32_t cIn, uint32_t cOut,
        uint32_t cOutGroup, uint32_t groups,
        uint32_t inputD, uint32_t inputH, uint32_t inputW,
        uint32_t kernelD, uint32_t kernelH, uint32_t kernelW,
        uint32_t outputD, uint32_t outputH, uint32_t outputW,
        int32_t strideD, int32_t strideH, int32_t strideW,
        int32_t padD, int32_t padH, int32_t padW,
        int32_t outPadD, int32_t outPadH, int32_t outPadW,
        int32_t dilationD, int32_t dilationH, int32_t dilationW,
        uint32_t tileLength)
    {
        batchSize_ = batchSize;
        cIn_ = cIn;
        cOut_ = cOut;
        cOutGroup_ = cOutGroup;
        groups_ = groups;
        inputD_ = inputD;
        inputH_ = inputH;
        inputW_ = inputW;
        kernelD_ = kernelD;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        outputD_ = outputD;
        outputH_ = outputH;
        outputW_ = outputW;
        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        padD_ = padD;
        padH_ = padH;
        padW_ = padW;
        outPadD_ = outPadD;
        outPadH_ = outPadH;
        outPadW_ = outPadW;
        dilationD_ = dilationD;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        tileLength_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        cInGroup_ = cIn / groups;
        cOutPerBlock_ = cOutGroup_;
        inputSize_ = inputD_ * inputH_ * inputW_;
        outputSize_ = outputD_ * outputH_ * outputW_;
        kernelSize_ = kernelD_ * kernelH_ * kernelW_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, kernelSize_ * cOutPerBlock_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t totalBlocks = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        if (totalBlocks == 0) return;

        for (uint32_t blk = blockIdx; blk < batchSize_ * groups_; blk += totalBlocks) {
            uint32_t batchIdx = blk / groups_;
            uint32_t groupIdx = blk % groups_;

            if (batchIdx >= batchSize_) continue;

            ProcessGroup(batchIdx, groupIdx);
        }
    }

private:
    __aicore__ inline void ProcessGroup(uint32_t batchIdx, uint32_t groupIdx)
    {
        uint32_t outputOffset = (batchIdx * cOut_ + groupIdx * cOutGroup_) * outputSize_;
        uint32_t inputOffset = (batchIdx * cIn_ + groupIdx * cInGroup_) * inputSize_;
        uint32_t weightOffset = groupIdx * cOutGroup_ * kernelSize_;

        // Process output in tiles along channel dimension (cOutGroup)
        uint32_t cRemaining = cOutGroup_;
        uint32_t cOffset = 0;

        while (cRemaining > 0) {
            uint32_t cCur = (cRemaining > cOutPerBlock_) ? cOutPerBlock_ : cRemaining;

            ProcessOutputTile(outputOffset + cOffset * outputSize_,
                              inputOffset,
                              weightOffset + cOffset * kernelSize_,
                              cCur);

            cOffset += cCur;
            cRemaining -= cCur;
        }
    }

    __aicore__ inline void ProcessOutputTile(
        uint32_t yOffset, uint32_t xOffset, uint32_t wOffset, uint32_t cCur)
    {
        // Each output point: accumulate over kernel positions and input channels
        uint32_t outputPlaneSize = outputD_ * outputH_ * outputW_;
        uint32_t totalElements = cCur * outputPlaneSize;

        uint32_t offset = 0;
        while (offset < totalElements) {
            uint32_t remain = totalElements - offset;
            uint32_t curLen = (remain > tileLength_) ? tileLength_ : remain;

            // Copy input feature map tile
            uint32_t inputElems = inputSize_;
            uint32_t inputOffsetCur = 0;
            while (inputOffsetCur < inputElems) {
                uint32_t inRemain = inputElems - inputOffsetCur;
                uint32_t inCur = (inRemain > tileLength_) ? tileLength_ : inRemain;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[xOffset + inputOffsetCur], inCur);
                inQueueX_.EnQue(xLocal);
                inputOffsetCur += inCur;
            }

            // Copy weight tile for this output channel chunk
            uint32_t weightElems = cCur * kernelSize_;
            AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[wOffset], weightElems);
            inQueueW_.EnQue(wLocal);

            // Initialize output to zero
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Sets(yLocal, 0.0f, curLen);
            tmpQueue_.FreeTensor(tLocal);

            // Perform transposed convolution for this tile
            ComputeTransposedConv(yLocal, xLocal, wLocal, curLen, cCur, outputPlaneSize);

            // Copy output
            AscendC::DataCopy(yGm_[yOffset + offset], yLocal, curLen);

            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
            inQueueW_.FreeTensor(wLocal);

            offset += curLen;
        }
    }

    __aicore__ inline void ComputeTransposedConv(
        AscendC::LocalTensor<float> &yLocal,
        AscendC::LocalTensor<float> &xLocal,
        AscendC::LocalTensor<float> &wLocal,
        uint32_t totalOutElems, uint32_t cCur, uint32_t outputPlaneSize)
    {
        // For transposed conv: each output element corresponds to one input position
        // dilated output positions are accumulated from input * weight
        // This is simplified: iterate output positions and accumulate contributions

        uint32_t outIdx = 0;
        for (uint32_t od = 0; od < outputD_; ++od) {
            for (uint32_t oh = 0; oh < outputH_; ++oh) {
                for (uint32_t ow = 0; ow < outputW_; ++ow) {
                    int32_t inDStart = od + padD_ - outPadD_;
                    int32_t inHStart = oh + padH_ - outPadH_;
                    int32_t inWStart = ow + padW_ - outPadW_;

                    for (uint32_t kd = 0; kd < kernelD_; ++kd) {
                        for (uint32_t kh = 0; kh < kernelH_; ++kh) {
                            for (uint32_t kw = 0; kw < kernelW_; ++kw) {
                                int32_t inD = (inDStart - kd * dilationD_) / strideD_;
                                int32_t inH = (inHStart - kh * dilationH_) / strideH_;
                                int32_t inW = (inWStart - kw * dilationW_) / strideW_;

                                if (inD >= 0 && inD < static_cast<int32_t>(inputD_) &&
                                    inH >= 0 && inH < static_cast<int32_t>(inputH_) &&
                                    inW >= 0 && inW < static_cast<int32_t>(inputW_) &&
                                    (inDStart - kd * dilationD_) % strideD_ == 0 &&
                                    (inHStart - kh * dilationH_) % strideH_ == 0 &&
                                    (inWStart - kw * dilationW_) % strideW_ == 0)
                                {
                                    uint32_t inIdx = inD * inputH_ * inputW_ + inH * inputW_ + inW;
                                    uint32_t wIdx = kd * kernelH_ * kernelW_ + kh * kernelW_ + kw;

                                    for (uint32_t oc = 0; oc < cCur; ++oc) {
                                        uint32_t outFlatIdx = oc * outputPlaneSize + outIdx;
                                        uint32_t wFlatIdx = oc * kernelSize_ + wIdx;

                                        if (outFlatIdx < totalOutElems) {
                                            yLocal.SetValue(outFlatIdx,
                                                yLocal.GetValue(outFlatIdx) +
                                                xLocal.GetValue(inIdx) * wLocal.GetValue(wFlatIdx));
                                        }
                                    }
                                }
                            }
                        }
                    }
                    outIdx++;
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_{0};
    uint32_t cIn_{0};
    uint32_t cOut_{0};
    uint32_t cOutGroup_{0};
    uint32_t groups_{0};
    uint32_t cInGroup_{0};
    uint32_t cOutPerBlock_{0};
    uint32_t inputD_{0}, inputH_{0}, inputW_{0};
    uint32_t kernelD_{0}, kernelH_{0}, kernelW_{0};
    uint32_t outputD_{0}, outputH_{0}, outputW_{0};
    int32_t strideD_{1}, strideH_{1}, strideW_{1};
    int32_t padD_{0}, padH_{0}, padW_{0};
    int32_t outPadD_{0}, outPadH_{0}, outPadW_{0};
    int32_t dilationD_{1}, dilationH_{1}, dilationW_{1};
    uint32_t inputSize_{0};
    uint32_t outputSize_{0};
    uint32_t kernelSize_{0};
    uint32_t tileLength_{0};
};
