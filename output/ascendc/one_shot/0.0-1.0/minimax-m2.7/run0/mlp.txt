host_operator_src="""

#include "mlp_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t MAX_TILE_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MlpCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const uint32_t batchSize = static_cast<uint32_t>(x_shape->GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int *inputSize = attrs->GetAttrPointer<int>(0);
    const int *hiddenSize1 = attrs->GetAttrPointer<int>(1);
    const int *hiddenSize2 = attrs->GetAttrPointer<int>(2);
    const int *outputSize = attrs->GetAttrPointer<int>(3);

    const uint32_t m = batchSize;
    const uint32_t k1 = *inputSize;
    const uint32_t h1 = *hiddenSize1;
    const uint32_t k2 = *hiddenSize1;
    const uint32_t h2 = *hiddenSize2;
    const uint32_t out = *outputSize;

    // UB budget: 3 mats (A,B,C) + 2 tmp
    // Per tile: A(mxk) + B(kxh) + C(mxh) + 2*tmp(mxk or mxh)
    // Conservative: use k dimension as limiting factor
    uint32_t maxTileM = UB_BUDGET_BYTES / (4 * sizeof(float) * k1);
    maxTileM = AlignDown(maxTileM, ALIGN_ELEMS);
    if (maxTileM < ALIGN_ELEMS) maxTileM = ALIGN_ELEMS;
    if (maxTileM > MAX_TILE_ELEMS) maxTileM = AlignDown(MAX_TILE_ELEMS, ALIGN_ELEMS);

    uint32_t tileM = maxTileM;
    if (m < tileM) {
        const uint32_t aligned = AlignDown(m, ALIGN_ELEMS);
        tileM = (aligned > 0) ? aligned : m;
    }

    uint32_t blockDim = CeilDiv(m, tileM);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_inputSize(k1);
    tiling.set_hiddenSize1(h1);
    tiling.set_hiddenSize2(h2);
    tiling.set_outputSize(out);
    tiling.set_mSize(m);
    tiling.set_k1Size(k1);
    tiling.set_k2Size(k2);
    tiling.set_tileM(tileM);
    tiling.set_tileK(AlignDown(maxTileM, ALIGN_ELEMS));
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int *outputSize = attrs->GetAttrPointer<int>(3);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMlpCustom {
public:
    __aicore__ inline KernelMlpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed, GM_ADDR y,
                                uint32_t batchSize, uint32_t inputSize, uint32_t hiddenSize1,
                                uint32_t hiddenSize2, uint32_t outputSize,
                                uint32_t mSize, uint32_t k1Size, uint32_t k2Size,
                                uint32_t tileM)
    {
        batchSize_ = batchSize;
        inputSize_ = inputSize;
        hiddenSize1_ = hiddenSize1;
        hiddenSize2_ = hiddenSize2;
        outputSize_ = outputSize;
        mSize_ = mSize;
        k1Size_ = k1Size;
        k2Size_ = k2Size;
        tileM_ = tileM;

        xGm_.SetGlobalBuffer((__gm__ float*)x, mSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * outputSize_);

        // w_packed layout: [w1(h1*k1), w2(h2*k2), w3(out*h2)]
        w1Offset_ = 0;
        w2Offset_ = hiddenSize1_ * inputSize_;
        w3Offset_ = w2Offset_ + hiddenSize2_ * hiddenSize1_;

        w1Gm_.SetGlobalBuffer((__gm__ float*)(w_packed + w1Offset_), hiddenSize1_ * inputSize_);
        w2Gm_.SetGlobalBuffer((__gm__ float*)(w_packed + w2Offset_), hiddenSize2_ * hiddenSize1_);
        w3Gm_.SetGlobalBuffer((__gm__ float*)(w_packed + w3Offset_), outputSize_ * hiddenSize2_);

        // b_packed layout: [b1(h1), b2(h2), b3(out)]
        b1Gm_.SetGlobalBuffer((__gm__ float*)b_packed, hiddenSize1_);
        b2Gm_.SetGlobalBuffer((__gm__ float*)(b_packed + hiddenSize1_), hiddenSize2_);
        b3Gm_.SetGlobalBuffer((__gm__ float*)(b_packed + hiddenSize1_ + hiddenSize2_), outputSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileM_ * sizeof(float));
        pipe_.InitBuffer(inQueueW1_, BUFFER_NUM, hiddenSize1_ * sizeof(float));
        pipe_.InitBuffer(inQueueW2_, BUFFER_NUM, hiddenSize2_ * sizeof(float));
        pipe_.InitBuffer(inQueueW3_, BUFFER_NUM, outputSize_ * sizeof(float));
        pipe_.InitBuffer(inQueueB1_, 1, hiddenSize1_ * sizeof(float));
        pipe_.InitBuffer(inQueueB2_, 1, hiddenSize2_ * sizeof(float));
        pipe_.InitBuffer(inQueueB3_, 1, outputSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileM_ * outputSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueA_, 1, tileM_ * hiddenSize1_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueB_, 1, tileM_ * hiddenSize2_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (mSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideM = blockNum * tileM_;
        uint32_t mStart = blockIdx * tileM_;
        if (mStart >= mSize_) return;

        // Prefetch biases
        CopyBiasesIn();

        uint32_t mLen = tileM_;
        bool firstTile = true;
        bool lastTile = false;

        while (true) {
            if (mStart + tileM_ >= mSize_) {
                mLen = mSize_ - mStart;
                lastTile = true;
            }

            // Layer 1: GEMM (m x k1) @ (h1 x k1)T -> (m x h1), then + bias1, then ReLU
            Layer1(mStart, mLen, firstTile);

            // Layer 2: GEMM (m x h1) @ (h2 x h1)T -> (m x h2), then + bias2, then ReLU
            Layer2(mLen, firstTile);

            // Layer 3: GEMM (m x h2) @ (out x h2)T -> (m x out), then + bias3
            Layer3(mStart, mLen, firstTile, lastTile);

            if (lastTile) break;

            mStart += strideM;
            if (mStart >= mSize_) break;
            mLen = (mStart + tileM_ > mSize_) ? (mSize_ - mStart) : tileM_;
            firstTile = false;
        }
    }

private:
    __aicore__ inline void CopyBiasesIn()
    {
        AscendC::LocalTensor<float> b1Local = inQueueB1_.AllocTensor<float>();
        AscendC::DataCopy(b1Local, b1Gm_, hiddenSize1_);
        inQueueB1_.EnQue(b1Local);

        AscendC::LocalTensor<float> b2Local = inQueueB2_.AllocTensor<float>();
        AscendC::DataCopy(b2Local, b2Gm_, hiddenSize2_);
        inQueueB2_.EnQue(b2Local);

        AscendC::LocalTensor<float> b3Local = inQueueB3_.AllocTensor<float>();
        AscendC::DataCopy(b3Local, b3Gm_, outputSize_);
        inQueueB3_.EnQue(b3Local);
    }

    __aicore__ inline void Layer1(uint32_t mStart, uint32_t mLen, bool firstTile)
    {
        // Copy input tile
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[mStart], mLen);
        inQueueX_.EnQue(xLocal);

        // Copy weights for layer 1
        AscendC::LocalTensor<float> w1Local = inQueueW1_.AllocTensor<float>();
        AscendC::DataCopy(w1Local, w1Gm_, hiddenSize1_ * inputSize_);
        inQueueW1_.EnQue(w1Local);

        // Allocate output for layer 1
        AscendC::LocalTensor<float> y1Local = outQueueY_.AllocTensor<float>();

        // GEMM: y1 = x @ w1.T
        // xLocal: (mLen x inputSize_), w1Local: (hiddenSize1_ x inputSize_) -> result: (mLen x hiddenSize1_)
        // Using MatMul: A(m,k) @ B(n,k)T -> C(m,n)
        AscendC::MatMul<float>(y1Local, xLocal, w1Local, mLen, hiddenSize1_, inputSize_,
                               1.0f, 0.0f, false, true);

        // Add bias1: broadcast b1 to (mLen x hiddenSize1_)
        AscendC::LocalTensor<float> b1Local = inQueueB1_.DeQue<float>();
        AscendC::BroadcastTo<float>(y1Local, y1Local, mLen, hiddenSize1_, b1Local, 0, hiddenSize1_);
        AscendC::Adds(y1Local, y1Local, b1Local, mLen * hiddenSize1_);

        // ReLU activation
        AscendC::Relu(y1Local, y1Local, mLen * hiddenSize1_);

        // Store intermediate result in tmpQueueA_
        AscendC::LocalTensor<float> tmpLocal = tmpQueueA_.AllocTensor<float>();
        AscendC::DataCopy(tmpLocal, y1Local, mLen * hiddenSize1_);
        tmpQueueA_.FreeTensor(tmpLocal);
        tmpQueueA_.EnQue(tmpLocal);

        // Cleanup
        inQueueX_.FreeTensor(xLocal);
        inQueueW1_.FreeTensor(w1Local);
        inQueueB1_.FreeTensor(b1Local);
        outQueueY_.FreeTensor(y1Local);
    }

    __aicore__ inline void Layer2(uint32_t mLen, bool firstTile)
    {
        // Get layer 1 output from tmpQueueA_
        AscendC::LocalTensor<float> h1Local = tmpQueueA_.DeQue<float>();

        // Copy weights for layer 2
        AscendC::LocalTensor<float> w2Local = inQueueW2_.AllocTensor<float>();
        AscendC::DataCopy(w2Local, w2Gm_, hiddenSize2_ * hiddenSize1_);
        inQueueW2_.EnQue(w2Local);

        // Allocate output for layer 2
        AscendC::LocalTensor<float> y2Local = outQueueY_.AllocTensor<float>();

        // GEMM: y2 = h1 @ w2.T
        AscendC::MatMul<float>(y2Local, h1Local, w2Local, mLen, hiddenSize2_, hiddenSize1_,
                               1.0f, 0.0f, false, true);

        // Add bias2
        AscendC::LocalTensor<float> b2Local = inQueueB2_.DeQue<float>();
        AscendC::BroadcastTo<float>(y2Local, y2Local, mLen, hiddenSize2_, b2Local, 0, hiddenSize2_);
        AscendC::Adds(y2Local, y2Local, b2Local, mLen * hiddenSize2_);

        // ReLU activation
        AscendC::Relu(y2Local, y2Local, mLen * hiddenSize2_);

        // Store intermediate result in tmpQueueB_
        AscendC::LocalTensor<float> tmpLocal2 = tmpQueueB_.AllocTensor<float>();
        AscendC::DataCopy(tmpLocal2, y2Local, mLen * hiddenSize2_);
        tmpQueueB_.FreeTensor(tmpLocal2);
        tmpQueueB_.EnQue(tmpLocal2);

        // Cleanup
        tmpQueueA_.FreeTensor(h1Local);
        inQueueW2_.FreeTensor(w2Local);
        inQueueB2_.FreeTensor(b2Local);
        outQueueY_.FreeTensor(y2Local);
    }

    __aicore__ inline void Layer3(uint32_t mStart, uint32_t mLen, bool firstTile, bool lastTile)
    {
        // Get layer 2 output from tmpQueueB_
        AscendC::LocalTensor<float> h2Local = tmpQueueB_.DeQue<float>();

        // Copy weights for layer 3
        AscendC::LocalTensor<float> w3Local = inQueueW3_.AllocTensor<float>();
        AscendC::DataCopy(w3Local, w3Gm_, outputSize_ * hiddenSize2_);
        inQueueW3_.EnQue(w3Local);

        // Allocate output buffer
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // GEMM: y = h2 @ w3.T
        AscendC::MatMul<float>(yLocal, h2Local, w3Local, mLen, outputSize_, hiddenSize2_,
                               1.0f, 0.0f, false, true);

        // Add bias3
        AscendC::LocalTensor<float> b3Local = inQueueB3_.DeQue<float>();
        AscendC::BroadcastTo<float>(yLocal, yLocal, mLen, outputSize_, b3Local, 0, outputSize_);
        AscendC::Adds(yLocal, yLocal, b3Local, mLen * outputSize_);

        // Copy result to global memory
        AscendC::DataCopy(yGm_[mStart * outputSize_], yLocal, mLen * outputSize_);

        // Cleanup
        tmpQueueB_.FreeTensor(h2Local);
        inQueueW3_.FreeTensor(w3Local);
        inQueueB3_.FreeTensor(b3Local);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW1_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW2_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW3_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB1_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB2_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB3_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueA_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueB_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> w1Gm_;
    AscendC::GlobalTensor<float> w2Gm_;
    AscendC::GlobalTensor<float> w3Gm_;
    AscendC::GlobalTensor<float> b1Gm_;
    AscendC::GlobalTensor<float> b2Gm_;
    AscendC::GlobalTensor<float> b3Gm_;

    uint32_t batchSize_ {0};
    uint32_t inputSize_ {0};
    uint32_t hiddenSize1_ {0};
    uint32_t hiddenSize2_ {0};
    uint32_t outputSize_ {0};
    uint32_t mSize_ {0};
    uint32_t k1Size_ {0};
    uint32_t k2Size_ {0};
    uint32_t tileM_ {0};
    uint32_t w1Offset_ {0};
    uint32_t w2Offset_ {0};
    uint32_t w3Offset_ {0};
};

extern "C" __global__ __aicore__ void mlp_custom(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMlpCustom op;
    op.Init(x, w_packed, b_packed, y,
            tiling_data.batchSize, tiling_data.inputSize, tiling_data.hiddenSize1,
            tiling_data.hiddenSize2, tiling_data.outputSize,
            tiling_data.mSize, tiling_data.k1Size, tiling_data.k2Size,
            tiling_data.tileM);
    op.Process();
}
