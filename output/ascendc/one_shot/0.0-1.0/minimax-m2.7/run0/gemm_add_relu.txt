#include "gemm_add_relu_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t minTileRows = 32;
static constexpr uint32_t minTileCols = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;
    
    auto inputShapes = context->GetInputShape(0);
    uint32_t batchSize = static_cast<uint32_t>(inputShapes->GetShape().GetDim(0));
    uint32_t inFeatures = static_cast<uint32_t>(inputShapes->GetShape().GetDim(1));
    
    auto weightShape = context->GetInputShape(1);
    uint32_t outFeatures = static_cast<uint32_t>(weightShape->GetShape().GetDim(0));
    
    // Compute row tile based on UB budget
    // UB usage: rowTile * inFeatures for x, rowTile for partial result, 
    //           colTile * inFeatures for w tile, rowTile * colTile for c tile
    uint32_t rowTile = UB_BUDGET_BYTES / (inFeatures * sizeof(float) * 2);
    rowTile = AlignDown(rowTile, minTileRows);
    if (rowTile < minTileRows) rowTile = minTileRows;
    if (rowTile > batchSize) rowTile = AlignDown(batchSize, minTileRows);
    if (rowTile == 0) rowTile = minTileRows;
    
    uint32_t colTile = UB_BUDGET_BYTES / (rowTile * sizeof(float) * 2);
    colTile = AlignDown(colTile, minTileCols);
    if (colTile < minTileCols) colTile = minTileCols;
    if (colTile > outFeatures) colTile = AlignDown(outFeatures, minTileCols);
    if (colTile == 0) colTile = minTileCols;
    
    // BlockDim for batching rows
    uint32_t numRowTiles = CeilDiv(batchSize, rowTile);
    uint32_t blockDim = numRowTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;
    
    tiling.set_batchSize(batchSize);
    tiling.set_inFeatures(inFeatures);
    tiling.set_outFeatures(outFeatures);
    tiling.set_rowTile(rowTile);
    tiling.set_colTile(colTile);
    tiling.set_blockDim(blockDim);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MM_BLOCK_M = 32;
constexpr int32_t MM_BLOCK_K = 32;
constexpr int32_t MM_BLOCK_N = 32;

class KernelGemmAddReluCustom {
public:
    __aicore__ inline KernelGemmAddReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures, 
                                uint32_t outFeatures, uint32_t rowTile, uint32_t colTile)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        rowTile_ = rowTile;
        colTile_ = colTile;
        
        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inFeatures);
        wGm_.SetGlobalBuffer((__gm__ float*)w, outFeatures * inFeatures);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * outFeatures);
        
        // UB buffers: need space for matrix tiles
        pipe_.InitBuffer(xQueueX_, BUFFER_NUM, rowTile_ * inFeatures_ * sizeof(float));
        pipe_.InitBuffer(wQueueW_, BUFFER_NUM, colTile_ * inFeatures_ * sizeof(float));
        pipe_.InitBuffer(cQueue_, BUFFER_NUM, rowTile_ * colTile_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, rowTile_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || inFeatures_ == 0 || outFeatures_ == 0) return;
        
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        
        // Each block handles a subset of rows (batch dimension)
        uint32_t numRowTiles = (batchSize_ + rowTile_ - 1) / rowTile_;
        
        uint32_t startRowTile = blockIdx;
        if (startRowTile >= numRowTiles) return;
        
        // Process row tiles assigned to this block
        for (uint32_t r = startRowTile; r < numRowTiles; r += 64) {  // max 64 blocks
            uint32_t rowStart = r * rowTile_;
            uint32_t rowEnd = (rowStart + rowTile_ > batchSize_) ? batchSize_ : rowStart + rowTile_;
            uint32_t rowsThisTile = rowEnd - rowStart;
            
            ProcessRowTile(rowStart, rowsThisTile);
        }
    }

private:
    __aicore__ inline void ProcessRowTile(uint32_t rowStart, uint32_t rows)
    {
        // Load x tile: [rows x inFeatures_]
        AscendC::LocalTensor<float> xLocal = xQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[rowStart * inFeatures_], rows * inFeatures_);
        xQueueX_.EnQue(xLocal);
        
        // Process columns in tiles
        uint32_t numColTiles = (outFeatures_ + colTile_ - 1) / colTile_;
        uint32_t yOffset = rowStart * outFeatures_;
        
        for (uint32_t c = 0; c < numColTiles; ++c) {
            uint32_t colStart = c * colTile_;
            uint32_t colEnd = (colStart + colTile_ > outFeatures_) ? outFeatures_ : colStart + colTile_;
            uint32_t colsThisTile = colEnd - colStart;
            
            // Compute partial GEMM for this column tile
            ComputeGemmTile(xLocal, rows, colStart, colsThisTile, yOffset);
            
            yOffset += colsThisTile;
        }
        
        xQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void ComputeGemmTile(AscendC::LocalTensor<float>& xLocal,
                                          uint32_t rows, uint32_t colStart,
                                          uint32_t cols, uint32_t yOffset)
    {
        // Load weight tile: [cols x inFeatures_]
        AscendC::LocalTensor<float> wLocal = wQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_[colStart * inFeatures_], cols * inFeatures_);
        wQueueW_.EnQue(wLocal);
        
        // Allocate output tile
        AscendC::LocalTensor<float> cLocal = cQueue_.AllocTensor<float>();
        
        // Initialize output to zeros
        AscendC::Sets(cLocal, 0.0f, rows * cols);
        
        // Compute GEMM using blocked algorithm
        uint32_t kTiles = (inFeatures_ + MM_BLOCK_K - 1) / MM_BLOCK_K;
        uint32_t mTiles = (rows + MM_BLOCK_M - 1) / MM_BLOCK_M;
        uint32_t nTiles = (cols + MM_BLOCK_N - 1) / MM_BLOCK_N;
        
        AscendC::LocalTensor<float> aBlock;
        AscendC::LocalTensor<float> bBlock;
        AscendC::LocalTensor<float> cBlock;
        
        for (uint32_t kt = 0; kt < kTiles; ++kt) {
            uint32_t kStart = kt * MM_BLOCK_K;
            uint32_t kEnd = (kStart + MM_BLOCK_K > inFeatures_) ? inFeatures_ : kStart + MM_BLOCK_K;
            uint32_t kLen = kEnd - kStart;
            
            for (uint32_t mt = 0; mt < mTiles; ++mt) {
                uint32_t mStart = mt * MM_BLOCK_M;
                uint32_t mEnd = (mStart + MM_BLOCK_M > rows) ? rows : mStart + MM_BLOCK_M;
                uint32_t mLen = mEnd - mStart;
                
                for (uint32_t nt = 0; nt < nTiles; ++nt) {
                    uint32_t nStart = nt * MM_BLOCK_N;
                    uint32_t nEnd = (nStart + MM_BLOCK_N > cols) ? cols : nStart + MM_BLOCK_N;
                    uint32_t nLen = nEnd - nStart;
                    
                    uint32_t aOffset = mStart * inFeatures_ + kStart;
                    uint32_t bOffset = nStart * inFeatures_ + kStart;
                    uint32_t cOffset = mStart * cols + nStart;
                    
                    // Use MatMul for blocked computation
                    AscendC::MatMul(cLocal[cOffset], 
                                   xLocal[aOffset], 
                                   wLocal[bOffset],
                                   mLen, kLen, nLen, cols);
                }
            }
        }
        
        wQueueW_.FreeTensor(wLocal);
        cQueue_.EnQue(cLocal);
        
        // Add bias and apply ReLU
        ApplyBiasRelu(rows, cols, yOffset);
        
        cQueue_.FreeTensor(cLocal);
    }

    __aicore__ inline void ApplyBiasRelu(uint32_t rows, uint32_t cols, uint32_t yOffset)
    {
        AscendC::LocalTensor<float> cLocal = cQueue_.DeQue<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        
        // Broadcast bias to all rows
        for (uint32_t i = 0; i < rows; ++i) {
            uint32_t rowOffset = i * cols;
            // Load bias for this column range
            for (uint32_t j = 0; j < cols; ++j) {
                cLocal[rowOffset + j] += biasGm_[yOffset + j];
            }
        }
        
        // Apply ReLU: y = max(0, y)
        AscendC::ReLU(cLocal, cLocal, rows * cols);
        
        // Copy result to output
        AscendC::DataCopy(yGm_[yOffset], cLocal, rows * cols);
        
        tmpQueue_.FreeTensor(tmpLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> cQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;
    
    uint32_t batchSize_ {0};
    uint32_t inFeatures_ {0};
    uint32_t outFeatures_ {0};
    uint32_t rowTile_ {0};
    uint32_t colTile_ {0};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddReluCustom op;
    op.Init(x, w, bias, y, tiling_data.batchSize, tiling_data.inFeatures, 
            tiling_data.outFeatures, tiling_data.rowTile, tiling_data.colTile);
    op.Process();
}
