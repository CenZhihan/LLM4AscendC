class KernelBatchedMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelBatchedMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t totalLengthX, uint32_t totalLengthY, uint32_t totalLengthZ,
                                uint32_t tileLength,
                                uint32_t batchSize, uint32_t M, uint32_t K, uint32_t N)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLengthX);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLengthY);
        zGm_.SetGlobalBuffer((__gm__ float*)z, totalLengthZ);

        batchSize_ = batchSize;
        M_ = M;
        K_ = K;
        N_ = N;
        tileLen_ = tileLength;

        // Allocate UB buffers: input X (tileM?), input Y (tileN?), output Z (tileLen).
        // For simplicity, allocate one tileLen for each.
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueZ_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (tileLen_ == 0) return;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= batchSize_ * M_ * N_) return;

        // Process tiles
        while (base < batchSize_ * M_ * N_) {
            uint32_t tileEnd = base + tileLen_;
            if (tileEnd > batchSize_ * M_ * N_) tileEnd = batchSize_ * M_ * N_;
            uint32_t curTileLen = tileEnd - base;

            ComputeTile(base, curTileLen);
            CopyOut(base, curTileLen);

            base += stride;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t start, uint32_t len)
    {
        // For each output element in this tile:
        for (uint32_t offset = 0; offset < len; ++offset) {
            uint32_t outIdx = start + offset;
            // compute b,i,j
            uint32_t b = outIdx / (M_ * N_);
            uint32_t rem = outIdx % (M_ * N_);
            uint32_t i = rem / N_;
            uint32_t j = rem % N_;

            // Compute dot product
            float sum = 0.0f;
            for (uint32_t k = 0; k < K_; ++k) {
                // A[b,i,k]
                uint32_t aIdx = ((b * M_ + i) * K_) + k;
                // B[b,k,j]
                uint32_t bIdx = ((b * K_ + k) * N_) + j;
                sum += xGm_[aIdx] * yGm_[bIdx];
            }
            // Store result
            zGm_[outIdx] = sum;
        }
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        // For output we directly wrote to zGm_, so no need to copy.
        // But we can leave empty.
    }

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;

static constexpr uint32_t ALIGN_ELEMS = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    BatchedMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *shapeX = context->GetInputShape(0);
    const gert::Shape *shapeY = context->GetInputShape(1);

    const auto &shapeXVec = shapeX->GetShapeVector();
    const auto &shapeYVec = shapeY->GetShapeVector();

    uint32_t batchSize = static_cast<uint32_t>(shapeXVec[0]);
    uint32_t M = static_cast<uint32_t>(shapeXVec[1]);
    uint32_t K = static_cast<uint32_t>(shapeXVec[2]);
    uint32_t N = static_cast<uint32_t>(shapeYVec[2]);

    uint32_t totalLengthY = batchSize * K * N;

    // Each tile holds up to tileLen floats for each of X, Y, Z.
    // Total bytes = 3 * tileLen * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));

#include "kernel_operator.h"
#include "batched_matrix_multiplication_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelBatchedMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelBatchedMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t totalLengthX, uint32_t totalLengthY, uint32_t totalLengthZ,
                                uint32_t tileLength,
                                uint32_t batchSize, uint32_t M, uint32_t K, uint32_t N)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLengthX);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLengthY);
        zGm_.SetGlobalBuffer((__gm__ float*)z, totalLengthZ);

        batchSize_ = batchSize;
        M_ = M;
        K_ = K;
        N_ = N;
        tileLen_ = tileLength;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueZ_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalOutputElems = batchSize_ * M_ * N_;
        const uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutputElems) return;

        while (base < totalOutputElems) {
            uint32_t curTileLen = tileLen_;
            if (base + curTileLen > totalOutputElems) {
                curTileLen = totalOutputElems - base;
            }
            ComputeTile(base, curTileLen);
            base += stride;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t start, uint32_t len)
    {
        for (uint32_t offset = 0; offset < len; ++offset) {
            uint32_t outIdx = start + offset;
            uint32_t b = outIdx / (M_ * N_);
            uint32_t rem = outIdx % (M_ * N_);
            uint32_t i = rem / N_;
            uint32_t j = rem % N_;

            float sum = 0.0f;
            for (uint32_t k = 0; k < K_; ++k) {
                uint32_t aIdx = ((b * M_ + i) * K_) + k;
                uint32_t bIdx = ((b * K_ + k) * N_) + j;
                sum += xGm_[aIdx] * yGm_[bIdx];
            }
            zGm_[outIdx] = sum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ_;

    AscendC::GlobalTensor<float> xGm_;

    AscendC::GlobalTensor<float> zGm_;

    uint32_t batchSize_;
    uint32_t M_;
    uint32_t K_;
    uint32_t N_;
    uint32_t tileLen_;
};
