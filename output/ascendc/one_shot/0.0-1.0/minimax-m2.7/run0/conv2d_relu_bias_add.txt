kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dReluBiasAddCustom {
public:
    __aicore__ inline KernelConv2dReluBiasAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias,
                                GM_ADDR bias, GM_ADDR y,
                                uint32_t batch, uint32_t cin, uint32_t height, uint32_t width,
                                uint32_t cout, uint32_t kH, uint32_t kW,
                                uint32_t outH, uint32_t outW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t hasConvBias)
    {
        batch_ = batch;
        cin_ = cin;
        height_ = height;
        width_ = width;
        cout_ = cout;
        kH_ = kH;
        kW_ = kW;
        outH_ = outH;
        outW_ = outW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        hasConvBias_ = hasConvBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * cin * height * width);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, cout * cin * kH * kW);

        if (hasConvBias_) {
            convBiasGm_.SetGlobalBuffer((__gm__ float*)conv_bias, cout);
        }

        biasGm_.SetGlobalBuffer((__gm__ float*)bias, cout);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * cout * outH * outW);

        inQueueX_.Init(pipe_, BUFFER_NUM);
        inQueueW_.Init(pipe_, BUFFER_NUM);
        outQueueY_.Init(pipe_, BUFFER_NUM);
        tmpQueue_.Init(pipe_, 1);
    }

    __aicore__ inline void Process()
    {
        uint32_t totalOutput = batch_ * cout_ * outH_ * outW_;
        if (totalOutput == 0) return;

        uint32_t blockNum = GetBlockNum();
        uint32_t blockIdx = GetBlockIdx();

        uint32_t tilesPerBlock = (totalOutput + blockNum - 1) / blockNum;
        uint32_t startTile = blockIdx * tilesPerBlock;
        uint32_t endTile = startTile + tilesPerBlock;
        if (endTile > totalOutput) endTile = totalOutput;

        for (uint32_t tile = startTile; tile < endTile; ++tile) {
            ComputeOneOutput(tile);
        }
    }

private:
    __aicore__ inline void ComputeOneOutput(uint32_t outputIdx)
    {
        uint32_t n = outputIdx / (cout_ * outH_ * outW_);
        uint32_t remainder = outputIdx % (cout_ * outH_ * outW_);
        uint32_t co = remainder / (outH_ * outW_);
        remainder = remainder % (outH_ * outW_);
        uint32_t oh = remainder / outW_;
        uint32_t ow = remainder % outW_;

        float result = 0.0f;

        for (uint32_t cc = 0; cc < cin_; ++cc) {
            for (uint32_t kh = 0; kh < kH_; ++kh) {
                for (uint32_t kw = 0; kw < kW_; ++kw) {
                    int32_t ih = oh * strideH_ + kh * dilationH_ - padH_;
                    int32_t iw = ow * strideW_ + kw * dilationW_ - padW_;

                    if (ih < 0 || ih >= height_ || iw < 0 || iw >= width_) {
                        continue;
                    }

                    uint32_t xIdx = n * cin_ * height_ * width_ + cc * height_ * width_ + ih * width_ + iw;
                    uint32_t wIdx = co * cin_ * kH_ * kW_ + cc * kH_ * kW_ + kh * kW_ + kw;

                    result += xGm_.GetValue(xIdx) * wGm_.GetValue(wIdx);
                }
            }
        }

        if (hasConvBias_) {
            result += convBiasGm_.GetValue(co);
        }

        if (result < 0.0f) {
            result = 0.0f;
        }

        result += biasGm_.GetValue(co);

        uint32_t yIdx = n * cout_ * outH_ * outW_ + co * outH_ * outW_ + oh * outW_ + ow;
        yGm_.SetValue(yIdx, result);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> convBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_;
    uint32_t cin_;
    uint32_t height_;
    uint32_t width_;
    uint32_t cout_;
    uint32_t kH_;
    uint32_t kW_;
    uint32_t outH_;
    uint32_t outW_;
    uint32_t strideH_;
    uint32_t strideW_;
    uint32_t padH_;
    uint32_t padW_;
    uint32_t dilationH_;
    uint32_t dilationW_;
    uint32_t hasConvBias_;
};

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x, GM_ADDR weight,
    GM_ADDR conv_bias, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluBiasAddCustom op;
    op.Init(x, weight, conv_bias, bias, y,
            tiling_data.batch, tiling_data.cin, tiling_data.height, tiling_data.width,
            tiling_data.cout, tiling_data.kH, tiling_data.kW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.hasConvBias);
    op.Process();
}
