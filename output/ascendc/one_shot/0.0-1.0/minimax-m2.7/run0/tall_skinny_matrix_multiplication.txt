host_operator_src = '''
#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 8;
static constexpr uint32_t TILE_N = 16;
static constexpr uint32_t TILE_K = 64;

ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);

    const uint32_t M = static_cast<uint32_t>(aShape->GetOriginShape().GetShapeVector()[0]);
    const uint32_t K = static_cast<uint32_t>(aShape->GetOriginShape().GetShapeVector()[1]);
    const uint32_t N = static_cast<uint32_t>(bShape->GetOriginShape().GetShapeVector()[1]);

    const uint32_t UB_FLOATS = UB_BUDGET_BYTES / sizeof(float);
    const uint32_t tileN = TILE_N;
    const uint32_t tileK = TILE_K;

    // Compute tileM based on UB budget
    uint32_t tileM = (UB_FLOATS - tileK * tileN) / (tileK + tileN);
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;
    if (M < tileM) tileM = M;
    else tileM = AlignDown(tileM, ALIGN_ELEMS);

    const uint32_t blockDimX = CeilDiv(M, tileM);
    const uint32_t blockDimY = CeilDiv(N, tileN);
    const uint32_t blockDim = blockDimX * blockDimY;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    const int64_t M = a_shape->GetOriginShape().GetShapeVector()[0];
    const int64_t K = a_shape->GetOriginShape().GetShapeVector()[1];
    const int64_t K2 = b_shape->GetOriginShape().GetShapeVector()[0];
    const int64_t N = b_shape->GetOriginShape().GetShapeVector()[1];

    // Optionally assert K == K2
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetShapeVector({M, N});
    return ge::GRAPH_SUCCESS;
}

{
    const ge::DataType dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, dtype);
    return ge::GRAPH_SUCCESS;
}

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTallSkinnyMatrixMultiplication {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplication() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                uint32_t blockDim)
    {
        M_ = M;
        K_ = K;
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        A_gm_.SetGlobalBuffer((__gm__ float *)A, M * K);
        B_gm_.SetGlobalBuffer((__gm__ float *)B, K * N);
        C_gm_.SetGlobalBuffer((__gm__ float *)C, M * N);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileM * tileK * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileK * tileN * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileM * tileN * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileM * tileN * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();

        const uint32_t blockDimX = (M_ + tileM_ - 1) / tileM_;
        const uint32_t blockDimY = (N_ + tileN_ - 1) / tileN_;

        const uint32_t idxX = blockIdx / blockDimY;
        const uint32_t idxY = blockIdx % blockDimY;

        const uint32_t startM = idxX * tileM_;
        const uint32_t startN = idxY * tileN_;

        if (startM >= M_ || startN >= N_) return;

        const uint32_t lenM = (startM + tileM_ <= M_) ? tileM_ : M_ - startM;
        const uint32_t lenN = (startN + tileN_ <= N_) ? tileN_ : N_ - startN;

        // Allocate output tile and zero‑initialize
        auto outTile = outQueueC_.AllocTensor<float>();
        AscendC::Sets(outTile, 0.0f, lenM * lenN);

        // Loop over K dimension
        for (uint32_t k = 0; k < K_; k += tileK_) {
            uint32_t kLen = (k + tileK_ <= K_) ? tileK_ : K_ - k;

            // Copy A tile
            auto aTile = inQueueA_.AllocTensor<float>();
            for (uint32_t i = 0; i < lenM; ++i) {
                uint32_t srcOffset = (startM + i) * K_ + k;
                AscendC::DataCopy(aTile[i * kLen], A_gm_[srcOffset], kLen);
            }
            inQueueA_.EnQue(aTile);

            // Copy B tile
            auto bTile = inQueueB_.AllocTensor<float>();
            for (uint32_t i = 0; i < kLen; ++i) {
                uint32_t srcOffset = (k + i) * N_ + startN;
                AscendC::DataCopy(bTile[i * lenN], B_gm_[srcOffset], lenN);
            }
            inQueueB_.EnQue(bTile);

            // Compute partial result
            auto tmpTile = tmpQueue_.AllocTensor<float>();
            AscendC::Sets(tmpTile, 0.0f, lenM * lenN);

            // Multiply‑accumulate
            for (uint32_t i = 0; i < lenM; ++i) {
                for (uint32_t j = 0; j < lenN; ++j) {
                    float sum = 0.0f;
                    for (uint32_t kk = 0; kk < kLen; ++kk) {
                        sum += aTile[i * kLen + kk] * bTile[kk * lenN + j];
                    }
                    tmpTile[i * lenN + j] = sum;
                }
            }

            // Accumulate into output tile
            AscendC::Add(outTile, outTile, tmpTile, lenM * lenN);

            // Free temporary buffers
            inQueueA_.FreeTensor(aTile);
            inQueueB_.FreeTensor(bTile);
            tmpQueue_.FreeTensor(tmpTile);
        }

        // Copy output tile to global memory
        outQueueC_.EnQue(outTile);
        auto outTile2 = outQueueC_.DeQue<float>();
        for (uint32_t i = 0; i < lenM; ++i) {
            uint32_t dstOffset = (startM + i) * N_ + startN;
            AscendC::DataCopy(C_gm_[dstOffset], outTile2[i * lenN], lenN);
        }
        outQueueC_.FreeTensor(outTile2);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t M_ = 0;
    uint32_t K_ = 0;
    uint32_t N_ = 0;
    uint32_t tileM_ = 0;
    uint32_t tileN_ = 0;
    uint32_t tileK_ = 0;
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplication op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.blockDim);
    op.Process();
}
'''
