#include "standard_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t BLOCK_DIM = 8;
static constexpr uint32_t M_TILE_SIZE = 64;
static constexpr uint32_t K_TILE_SIZE = 256;
static constexpr uint32_t N_TILE_SIZE = 64;
static constexpr uint32_t ELEMENTS_PER_BLOCK = 16;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    StandardMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);

    uint32_t M = static_cast<uint32_t>(aShape->GetShapeSize());
    uint32_t K = static_cast<uint32_t>(aShape->GetShape().GetDim(1));
    uint32_t N = static_cast<uint32_t>(bShape->GetShape().GetDim(1));

    uint32_t mTiles = CeilDiv(M, M_TILE_SIZE);
    uint32_t nTiles = CeilDiv(N, N_TILE_SIZE);
    uint32_t kTiles = CeilDiv(K, K_TILE_SIZE);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);

    std::vector<uint32_t> mTileVec;
    std::vector<uint32_t> nTileVec;
    mTileVec.reserve(mTiles);
    nTileVec.reserve(nTiles);

    for (uint32_t i = 0; i < mTiles; ++i) {
        mTileVec.push_back(M_TILE_SIZE);
    }
    if (M % M_TILE_SIZE != 0 && !mTileVec.empty()) {
        mTileVec.back() = M % M_TILE_SIZE;
    }

    for (uint32_t i = 0; i < nTiles; ++i) {
        nTileVec.push_back(N_TILE_SIZE);
    }
    if (N % N_TILE_SIZE != 0 && !nTileVec.empty()) {
        nTileVec.back() = N % N_TILE_SIZE;
    }

    tiling.set_mTiles(mTileVec);
    tiling.set_nTiles(nTileVec);

    uint32_t totalBlocks = mTiles * nTiles;
    uint32_t blockDim = totalBlocks < BLOCK_DIM ? totalBlocks : BLOCK_DIM;
    tiling.set_blockDim(blockDim);

    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);

    gert::Shape *cShape = context->GetOutputShape(0);

    std::vector<int64_t> cDims = {
        aShape->GetShape().GetDim(0),
        bShape->GetShape().GetDim(1)
    };

    cShape->SetDim(0, cDims[0]);
    cShape->SetDim(1, cDims[1]);
    cShape->SetShapeSize(cDims[0] * cDims[1]);

    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);

    if (a_dtype != b_dtype) {
        return ge::GRAPH_FAILED;
    }

    context->SetOutputDataType(0, a_dtype);
    return ge::GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t M_TILE = 64;
constexpr uint32_t K_TILE = 256;
constexpr uint32_t N_TILE = 64;
constexpr uint32_t BLOCK_ROWS = 16;

class KernelStandardMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelStandardMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t K, uint32_t N)
    {
        M_ = M;
        K_ = K;
        N_ = N;

        AGm_.SetGlobalBuffer((__gm__ float*)A, M * K);
        BGm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        CGm_.SetGlobalBuffer((__gm__ float*)C, M * N);

        pipe_.InitBuffer(AQueX_, BUFFER_NUM, M_TILE * K_TILE * sizeof(float));
        pipe_.InitBuffer(AQueY_, BUFFER_NUM, M_TILE * K_TILE * sizeof(float));
        pipe_.InitBuffer(BQueX_, BUFFER_NUM, K_TILE * N_TILE * sizeof(float));
        pipe_.InitBuffer(BQueY_, BUFFER_NUM, K_TILE * N_TILE * sizeof(float));
        pipe_.InitBuffer(CQue_, BUFFER_NUM, M_TILE * N_TILE * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, M_TILE * N_TILE * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t mTileCount = (M_ + M_TILE - 1) / M_TILE;
        uint32_t nTileCount = (N_ + N_TILE - 1) / N_TILE;
        uint32_t totalBlocks = mTileCount * nTileCount;

        if (totalBlocks == 0) return;

        uint32_t blocksPerGroup = (totalBlocks + blockNum - 1) / blockNum;
        uint32_t startBlock = blockIdx * blocksPerGroup;
        uint32_t endBlock = (startBlock + blocksPerGroup) > totalBlocks ? totalBlocks : (startBlock + blocksPerGroup);

        for (uint32_t blk = startBlock; blk < endBlock; ++blk) {
            uint32_t mTileIdx = blk / nTileCount;
            uint32_t nTileIdx = blk % nTileCount;

            uint32_t mOffset = mTileIdx * M_TILE;
            uint32_t nOffset = nTileIdx * N_TILE;

            uint32_t mLen = (mOffset + M_TILE <= M_) ? M_TILE : (M_ - mOffset);
            uint32_t nLen = (nOffset + N_TILE <= N_) ? N_TILE : (N_ - nOffset);

            ComputeTile(mOffset, nOffset, mLen, nLen);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t mOffset, uint32_t nOffset, uint32_t mLen, uint32_t nLen)
    {
        AscendC::LocalTensor<float> cLocal = CQue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpBuf_.GetTensor<float>(0, mLen * nLen);

        AscendC::Sets(cLocal, 0.0f, mLen * nLen);

        uint32_t kTileCount = (K_ + K_TILE - 1) / K_TILE;

        for (uint32_t kt = 0; kt < kTileCount; ++kt) {
            uint32_t kOffset = kt * K_TILE;
            uint32_t kLen = (kOffset + K_TILE <= K_) ? K_TILE : (K_ - kOffset);

            uint32_t aSize = mLen * kLen;
            uint32_t bSize = kLen * nLen;

            AscendC::LocalTensor<float> aLocal = AQueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> bLocal = BQueX_.AllocTensor<float>();

            AscendC::DataCopy(aLocal, AGm_[mOffset * K_ + kOffset], aSize);
            AscendC::DataCopy(bLocal, BGm_[kOffset * N_ + nOffset], bSize);

            AQueX_.EnQue(aLocal);
            BQueX_.EnQue(bLocal);

            AscendC::LocalTensor<float> aDeq = AQueY_.DeQue<float>();
            AscendC::LocalTensor<float> bDeq = BQueY_.DeQue<float>();

            AscendC::Matmul(cLocal, aDeq, bDeq, cLocal, tmpLocal, mLen, kLen, nLen);

            AQueY_.FreeTensor(aDeq);
            BQueY_.FreeTensor(bDeq);
        }

        CQue_.EnQue(cLocal);

        AscendC::LocalTensor<float> cOut = CQue_.DeQue<float>();
        AscendC::DataCopy(CGm_[mOffset * N_ + nOffset], cOut, mLen * nLen);
        CQue_.FreeTensor(cOut);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> AQueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> AQueY_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> BQueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> BQueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> CQue_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;

    AscendC::GlobalTensor<float> AGm_;
    AscendC::GlobalTensor<float> BGm_;
    AscendC::GlobalTensor<float> CGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
};

extern "C" __global__ __aicore__ void standard_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelStandardMatrixMultiplicationCustom op;
    op.Init(A, B, C, tiling_data.M, tiling_data.K, tiling_data.N);
    op.Process();
}
