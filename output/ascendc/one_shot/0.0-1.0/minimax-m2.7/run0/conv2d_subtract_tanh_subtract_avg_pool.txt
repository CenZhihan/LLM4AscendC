kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dSubtractTanhSubtractAvgPoolCustom {
public:
    __aicore__ inline KernelConv2dSubtractTanhSubtractAvgPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t N, uint32_t C_in, uint32_t C_out,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t poolSize, uint32_t outputSize,
                                float subtract1Value, float subtract2Value)
    {
        N_ = N;
        C_in_ = C_in;
        C_out_ = C_out;
        H_ = H;
        W_ = W;
        K_ = K;
        poolSize_ = poolSize;
        outputSize_ = outputSize;
        subtract1Value_ = subtract1Value;
        subtract2Value_ = subtract2Value;

        H_out_ = H - K + 1;
        W_out_ = W - K + 1;
        H_pool_ = H_out_ / poolSize;
        W_pool_ = W_out_ / poolSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_in_ * H_ * W_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, C_out_ * C_in_ * K_ * K_);

        if (bias != nullptr) {
            hasBias_ = true;
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, C_out_);
        } else {
            hasBias_ = false;
        }

        yGm_.SetGlobalBuffer((__gm__ float*)y, outputSize_);

        const uint32_t convOutSize = C_out_ * H_out_ * W_out_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, C_in_ * K_ * K_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, C_out_ * C_in_ * K_ * K_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, C_out_ * sizeof(float));
        pipe_.InitBuffer(convQueue_, BUFFER_NUM, convOutSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outputSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        if (blockIdx >= outputSize_) return;

        const uint32_t n = blockIdx / (C_out_ * H_pool_ * W_pool_);
        const uint32_t cOutBase = (blockIdx / (H_pool_ * W_pool_)) % C_out_;
        const uint32_t hPoolBase = (blockIdx / W_pool_) % H_pool_;
        const uint32_t wPoolBase = blockIdx % W_pool_;

        uint32_t hOutBase = hPoolBase * poolSize_;
        uint32_t wOutBase = wPoolBase * poolSize_;

        ConvolutionSinglePixel(n, cOutBase, hOutBase, wOutBase);
    }

private:
    __aicore__ inline void ConvolutionSinglePixel(uint32_t n, uint32_t cOutBase,
                                                  uint32_t hOutStart, uint32_t wOutStart)
    {
        AscendC::LocalTensor<float> convOut = convQueue_.AllocTensor<float>();

        for (uint32_t cOut = 0; cOut < C_out_; ++cOut) {
            float sum = hasBias_ ? biasGm_.GetValue(cOut) : 0.0f;

            for (uint32_t cIn = 0; cIn < C_in_; ++cIn) {
                for (uint32_t kh = 0; kh < K_; ++kh) {
                    for (uint32_t kw = 0; kw < K_; ++kw) {
                        uint32_t hIn = hOutStart + kh;
                        uint32_t wIn = wOutStart + kw;

                        uint32_t xIdx = ((n * C_in_ + cIn) * H_ + hIn) * W_ + wIn;
                        uint32_t wIdx = ((cOut * C_in_ + cIn) * K_ + kh) * K_ + kw;

                        sum += xGm_.GetValue(xIdx) * weightGm_.GetValue(wIdx);
                    }
                }
            }

            uint32_t outIdx = (cOut * H_out_ + hOutStart) * W_out_ + wOutStart;
            convOut.SetValue(outIdx, sum);
        }

        convQueue_.EnQue(convOut);
        ComputeAndPool();
    }

    __aicore__ inline void ComputeAndPool()
    {
        AscendC::LocalTensor<float> convLocal = convQueue_.DeQue<float>();
        AscendC::LocalTensor<float> tmpLocal1 = convQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal2 = convQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> poolLocal = outQueueY_.AllocTensor<float>();

        AscendC::Adds(tmpLocal1, convLocal, -subtract1Value_, C_out_ * H_out_ * W_out_);
        AscendC::Tanh(tmpLocal2, tmpLocal1, C_out_ * H_out_ * W_out_);
        AscendC::Adds(tmpLocal1, tmpLocal2, -subtract2Value_, C_out_ * H_out_ * W_out_);

        const uint32_t pooledSize = C_out_ * H_pool_ * W_pool_;
        const uint32_t windowSize = poolSize_ * poolSize_;

        for (uint32_t c = 0; c < C_out_; ++c) {
            for (uint32_t hp = 0; hp < H_pool_; ++hp) {
                for (uint32_t wp = 0; wp < W_pool_; ++wp) {
                    float poolSum = 0.0f;
                    uint32_t baseIdx = (c * H_out_ + hp * poolSize_) * W_out_ + wp * poolSize_;

                    for (uint32_t kh = 0; kh < poolSize_; ++kh) {
                        for (uint32_t kw = 0; kw < poolSize_; ++kw) {
                            poolSum += tmpLocal1.GetValue(baseIdx + kh * W_out_ + kw);
                        }
                    }

                    uint32_t outIdx = (c * H_pool_ + hp) * W_pool_ + wp;
                    poolLocal.SetValue(outIdx, poolSum / static_cast<float>(windowSize));
                }
            }
        }

        outQueueY_.EnQue<float>(poolLocal);
        convQueue_.FreeTensor(convLocal);
        convQueue_.FreeTensor(tmpLocal1);
        convQueue_.FreeTensor(tmpLocal2);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> convQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t C_in_ {0};
    uint32_t C_out_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t K_ {0};
    uint32_t poolSize_ {1};
    uint32_t outputSize_ {0};
    uint32_t H_out_ {0};
    uint32_t W_out_ {0};
    uint32_t H_pool_ {0};
    uint32_t W_pool_ {0};
    float subtract1Value_ {0.0f};
    float subtract2Value_ {0.0f};
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractTanhSubtractAvgPoolCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.C_in, tiling_data.C_out,
            tiling_data.H, tiling_data.W, tiling_data.K,
            tiling_data.poolSize, tiling_data.outputSize,
            tiling_data.subtract1Value, tiling_data.subtract2Value);
    op.Process();
}
