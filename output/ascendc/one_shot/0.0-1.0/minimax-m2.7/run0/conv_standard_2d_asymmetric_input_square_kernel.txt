#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t BLOCK_SIZE = 32;
constexpr uint32_t OUT_TILE_H = 16;
constexpr uint32_t OUT_TILE_W = 16;

class KernelConvStandard2dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileH, uint32_t tileW,
                                uint32_t kH, uint32_t kW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t cIn, uint32_t cOut,
                                uint32_t hIn, uint32_t wIn,
                                uint32_t hOut, uint32_t wOut,
                                uint32_t batchSize)
    {
        totalLen_ = totalLength;
        tileH_ = tileH > 0 ? tileH : OUT_TILE_H;
        tileW_ = tileW > 0 ? tileW : OUT_TILE_W;
        kH_ = kH;
        kW_ = kW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        cIn_ = cIn;
        cOut_ = cOut;
        hIn_ = hIn;
        wIn_ = wIn;
        hOut_ = hOut;
        wOut_ = wOut;
        batchSize_ = batchSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)w);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        const uint32_t maxTileSize = tileH_ * tileW_;
        const uint32_t inTileSize = maxTileSize * cIn_ * kH_ * kW_;
        const uint32_t outTileSize = maxTileSize * cOut_;
        
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inTileSize * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, cOut_ * cIn_ * kH_ * kW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outTileSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tilesPerBatch = ((hOut_ + tileH_ - 1) / tileH_) * ((wOut_ + tileW_ - 1) / tileW_);
        const uint32_t totalTiles = batchSize_ * tilesPerBatch;

        const uint32_t strideBlocks = blockNum * tilesPerBatch;
        uint32_t batchIdx = blockIdx / tilesPerBatch;
        uint32_t tileIdx = blockIdx % tilesPerBatch;

        while (batchIdx < batchSize_) {
            const uint32_t tilesH = (hOut_ + tileH_ - 1) / tileH_;
            const uint32_t tilesW = (wOut_ + tileW_ - 1) / tileW_;
            
            const uint32_t tileYIdx = tileIdx / tilesW;
            const uint32_t tileXIdx = tileIdx % tilesW;

            const uint32_t outOffsetH = tileYIdx * tileH_;
            const uint32_t outOffsetW = tileXIdx * tileW_;

            const uint32_t curTileH = (outOffsetH + tileH_ <= hOut_) ? tileH_ : (hOut_ - outOffsetH);
            const uint32_t curTileW = (outOffsetW + tileW_ <= wOut_) ? tileW_ : (wOut_ - outOffsetW);

            const uint32_t inOffsetH = outOffsetH * strideH_ - padH_;
            const uint32_t inOffsetW = outOffsetW * strideW_ - padW_;

            const uint32_t actualInH = (inOffsetH >= 0) ? inOffsetH : 0;
            const uint32_t actualInW = (inOffsetW >= 0) ? inOffsetW : 0;
            const uint32_t effectivePadTop = (inOffsetH >= 0) ? 0 : (-inOffsetH);
            const uint32_t effectivePadLeft = (inOffsetW >= 0) ? 0 : (-inOffsetW);

            CopyInWeight();
            CopyInInput(actualInH, actualInW, curTileH, curTileW, effectivePadTop, effectivePadLeft);
            Compute(curTileH, curTileW);
            CopyOut(outOffsetH, outOffsetW, curTileH, curTileW, batchIdx);

            batchIdx += strideBlocks;
            if (batchIdx >= batchSize_) break;
        }
    }

private:
    __aicore__ inline void CopyInWeight()
    {
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        const uint32_t weightSize = cOut_ * cIn_ * kH_ * kW_;
        AscendC::DataCopy(wLocal, wGm_, weightSize);
        inQueueW_.EnQue<float>(wLocal);
    }

    __aicore__ inline void CopyInInput(uint32_t inH, uint32_t inW, uint32_t tileH, uint32_t tileW,
                                       uint32_t padTop, uint32_t padLeft)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        
        const uint32_t actualH = (inH + tileH <= hIn_) ? tileH : (hIn_ - inH);
        const uint32_t actualW = (inW + tileW <= wIn_) ? tileW : (wIn_ - inW);
        
        AscendC::DataCopy(xLocal, xGm_, actualH * actualW * cIn_);
        inQueueX_.EnQue<float>(xLocal);
    }

    __aicore__ inline void Compute(uint32_t tileH, uint32_t tileW)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::AscendC::SetVectorFlag enumerator;
        
        const uint32_t outSize = tileH * tileW * cOut_;
        AscendC::Sets(yLocal, 0.0f, outSize);

        for (uint32_t c = 0; c < cIn_; ++c) {
            for (uint32_t kh = 0; kh < kH_; ++kh) {
                for (uint32_t kw = 0; kw < kW_; ++kw) {
                    uint32_t hStart = kh * dilationH_;
                    uint32_t wStart = kw * dilationW_;
                    
                    if (hStart >= tileH || wStart >= tileW) continue;
                    
                    uint32_t hEnd = tileH;
                    uint32_t wEnd = tileW;
                    
                    AscendC::LocalTensor<float> xSlice = xLocal[c * tileH * tileW];
                    AscendC::LocalTensor<float> wSlice = wLocal[c * kH_ * kW_ + kh * kW_ + kw];
                    
                    AscendC::Muls(wSlice, wSlice, xSlice, tileH * tileW);
                    AscendC::Add(yLocal, yLocal, wSlice, tileH * tileW);
                }
            }
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueW_.FreeTensor(wLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t outH, uint32_t outW, uint32_t tileH, uint32_t tileW, uint32_t batchIdx)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        
        const uint32_t offset = ((batchIdx * cOut_ + 0) * hOut_ + outH) * wOut_ + outW;
        
        AscendC::DataCopy(yGm_[offset], yLocal, tileH * tileW * cOut_);
        
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileH_ {OUT_TILE_H};
    uint32_t tileW_ {OUT_TILE_W};
    uint32_t kH_ {3};
    uint32_t kW_ {3};
    uint32_t strideH_ {1};
    uint32_t strideW_ {1};
    uint32_t padH_ {0};
    uint32_t padW_ {0};
    uint32_t dilationH_ {1};
    uint32_t dilationW_ {1};
    uint32_t cIn_ {64};
    uint32_t cOut_ {128};
    uint32_t hIn_ {512};
    uint32_t wIn_ {1024};
    uint32_t hOut_ {510};
    uint32_t wOut_ {1022};
    uint32_t batchSize_ {8};
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dAsymmetricInputSquareKernelCustom op;
    op.Init(x, w, y,
            tiling_data.totalLength, tiling_data.tileH, tiling_data.tileW,
            tiling_data.kH, tiling_data.kW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.cIn, tiling_data.cIn,
            tiling_data.hIn, tiling_data.wIn,
            tiling_data.hOut, tiling_data.wOut,
            tiling_data.batchSize);
    op.Process();
}
