host_operator_src = """
#include "argmin_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // UB budget per core
static constexpr uint32_t ALIGN_ELEMS = 2048;             // alignment for DMA
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // hard cap for tile length

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    using namespace std;
    ArgminOverADimensionCustomTilingData tiling;

    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::Shape::ShapeInfo &shapeInfo = inputShape->GetShape();
    uint32_t batch = static_cast<uint32_t>(shapeInfo.GetDim(0));
    uint32_t dim1  = static_cast<uint32_t>(shapeInfo.GetDim(1));
    uint32_t dim2  = static_cast<uint32_t>(shapeInfo.GetDim(2));

    // The reduction axis is dim (attribute). We assume dim == 1 (reduce over dim1).
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t axis = attrs->GetAttrPointer<int64_t>(0) ? *(attrs->GetAttrPointer<int64_t>(0)) : 1;

    // Output size = batch * dim2 (when axis == 1)
    uint32_t totalLengthOut = batch * dim2;
    uint32_t axisSize = (axis == 1) ? dim1 : (axis == 2 ? dim2 : batch);

    // Tile length: each output point needs space for a float min and an int index (8 bytes)
    uint32_t bytesPerPoint = sizeof(float) + sizeof(int32_t);
    uint32_t maxTile = UB_BUDGET_BYTES / bytesPerPoint;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLength = maxTile;
    if (totalLengthOut < tileLength) {
        uint32_t aligned = AlignDown(totalLengthOut, ALIGN_ELEMS);
        tileLength = (aligned > 0) ? aligned : totalLengthOut;
    }

    uint32_t numTiles = (tileLength == 0) ? 1u : CeilDiv(totalLengthOut, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLengthOut(totalLengthOut);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_axisSize(axisSize);
    tiling.set_dim2(dim2);
    tiling.set_batchSize(batch);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t axis = attrs->GetAttrPointer<int64_t>(0) ? *(attrs->GetAttrPointer<int64_t>(0)) : 1;

    gert::Shape *outputShape = context->GetOutputShape(0);
    // Remove the reduction axis from input shape
    if (axis == 1) {
        // input shape: (batch, dim1, dim2) -> output: (batch, dim2)
        outputShape->SetDim(0, inputShape->GetDim(0));
        outputShape->SetDim(1, inputShape->GetDim(2));
    } else if (axis == 2) {
        // (batch, dim1, dim2) -> (batch, dim1)
        outputShape->SetDim(0, inputShape->GetDim(0));
        outputShape->SetDim(1, inputShape->GetDim(1));
    } else {
        // fallback: keep shape unchanged
        *outputShape = *inputShape;
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    context->SetOutputDataType(0, ge::DT_INT32);
    return GRAPH_SUCCESS;
}

}  // namespace ge

namespace ops {

class ArgminOverADimensionCustom : public OpDef {
public:
    explicit ArgminOverADimensionCustom(const char *name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT32})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(ArgminOverADimensionCustom);

}  // namespace ops

kernel_src = """
#include "kernel_operator.h"
#include <limits>

constexpr int32_t BUFFER_NUM = 2;

class KernelArgminOverADimensionCustom {
public:
    __aicore__ inline KernelArgminOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLengthOut, uint32_t tileLength,
                                uint32_t axisSize, uint32_t dim2, uint32_t batchSize)
    {
        totalLengthOut_ = totalLengthOut;
        tileLength_ = tileLength;
        axisSize_ = axisSize;
        dim2_ = dim2;
        batchSize_ = batchSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ int32_t*)y);

        // Allocate UB for temporary min values and indices for the whole tile.
        // Each tile processes up to tileLength_ output points.
        pipe_.InitBuffer(minValBuf_, tileLength_ * sizeof(float));
        pipe_.InitBuffer(minIdxBuf_, tileLength_ * sizeof(int32_t));
    }

    __aicore__ inline void Process()
    {
        if (totalLengthOut_ == 0 || tileLength_ == 0 || axisSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride loop over output points.
        uint32_t strideTiles = blockNum * tileLength_;
        uint32_t base = blockIdx * tileLength_;
        if (base >= totalLengthOut_) return;

        while (base < totalLengthOut_) {
            uint32_t curTileLen = totalLengthOut_ - base;
            if (curTileLen > tileLength_) curTileLen = tileLength_;

            // Process 'curTileLen' output points in this tile.
            ProcessTile(base, curTileLen);

            base += strideTiles;
        }
    }

private:
    __aicore__ inline void ProcessTile(uint32_t tileStart, uint32_t tileLen)
    {
        // Load minValBuf and minIdxBuf into local tensors.
        AscendC::LocalTensor<float> minValTensor = pipe_.AllocTensor<float>();
        AscendC::LocalTensor<int32_t> minIdxTensor = pipe_.AllocTensor<int32_t>();

        // Initialize min values to large positive numbers and indices to 0.
        AscendC::Mins(minValTensor, minValTensor, minValTensor, tileLen); // will be zero; need init to INF.
        // Actually we need to fill with INF. Use a vector of INF constants.
        AscendC::Fill(minValTensor, std::numeric_limits<float>::max(), tileLen);
        AscendC::Fill(minIdxTensor, static_cast<int32_t>(0), tileLen);

        // Iterate over the reduction axis (axisSize_).
        for (uint32_t axisIdx = 0; axisIdx < axisSize_; ++axisIdx) {
            // Load the values for all output points at this axis index.
            // Each output point corresponds to a specific batch and dim2 index.
            // We need to gather x[batch][axisIdx][dim2Idx] for each output point.
            // Since we process tileLen output points, we compute the offsets.
            // Compute batch and dim2 indices for each output point in the tile.
            // For each output point i in tile: outIdx = tileStart + i
            // batch = outIdx / dim2_; dim2Idx = outIdx % dim2_;
            // offset = ((batch * axisSize_ + axisIdx) * dim2_ + dim2Idx)
            // We'll compute in a vectorized fashion if possible, but for simplicity use a loop.

            // Use a temporary local tensor to hold the loaded values.
            AscendC::LocalTensor<float> loadBuf = pipe_.AllocTensor<float>();
            // Copy from GM to loadBuf for these 'tileLen' values.
            // We need to copy one value per output point.
            // Use DataCopy for each point in a loop (could be optimized).
            for (uint32_t i = 0; i < tileLen; ++i) {
                uint32_t outIdx = tileStart + i;
                uint32_t batchIdx = outIdx / dim2_;
                uint32_t dim2Idx = outIdx % dim2_;
                uint32_t offset = ((batchIdx * axisSize_ + axisIdx) * dim2_ + dim2Idx);
                AscendC::LocalTensor<float> singleVal = pipe_.AllocTensor<float>();
                AscendC::DataCopy(singleVal, xGm_[offset], 1);
                // Move to loadBuf position i
                // Actually we can directly compute compare with minValTensor using scalar.
                // Let's just do a scalar compare.
                float val = singleVal.GetValue();
                if (val < minValTensor.GetValue(i)) {
                    minValTensor.SetValue(i, val);
                    minIdxTensor.SetValue(i, static_cast<int32_t>(axisIdx));
                }
                pipe_.FreeTensor(singleVal);
            }
            pipe_.FreeTensor(loadBuf);
        }

        // Write results back to GM.
        for (uint32_t i = 0; i < tileLen; ++i) {
            uint32_t outIdx = tileStart + i;
            yGm_[outIdx] = minIdxTensor.GetValue(i);
        }

        pipe_.FreeTensor(minValTensor);
        pipe_.FreeTensor(minIdxTensor);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int32_t> yGm_;

    AscendC::LocalTensor<float> minValBuf_;
    AscendC::LocalTensor<int32_t> minIdxBuf_;

    uint32_t totalLengthOut_ {0};
    uint32_t tileLength_ {0};
    uint32_t axisSize_ {0};
    uint32_t dim2_ {0};
    uint32_t batchSize_ {0};
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR y,
                                                                    GM_ADDR workspace,
                                                                    GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgminOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalLengthOut,
            tiling_data.tileLength,
            tiling_data.axisSize,
            tiling_data.dim2,
            tiling_data.batchSize);
    op.Process();
}
"""
