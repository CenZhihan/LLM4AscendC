#include "conv_depthwise_2d_asymmetric_input_square_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for depthwise conv2d kernel:
//   inQueueX: 1 * tile (input tile)
//   inQueueW: 1 * KH * KW (weight buffer)
//   outQueueY: 1 * tile (output tile)
//   tmpQueue: 2 * tile (for multiply-accumulate and intermediate results)
// Total floats ~= 5 * tile + 9 (for 3x3 kernel) => bytes ~= 20 * tile + 36
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    uint32_t numBatch = static_cast<uint32_t>(x_shape->GetShapeSize() / (w_shape->GetDim(1) * x_shape->GetDim(2) * x_shape->GetDim(3)));
    uint32_t numChannels = static_cast<uint32_t>(w_shape->GetDim(0));
    uint32_t inputHeight = static_cast<uint32_t>(x_shape->GetDim(2));
    uint32_t inputWidth = static_cast<uint32_t>(x_shape->GetDim(3));
    uint32_t kernelHeight = static_cast<uint32_t>(w_shape->GetDim(2));
    uint32_t kernelWidth = static_cast<uint32_t>(w_shape->GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    uint32_t strideH = static_cast<uint32_t>(attrs->GetAttrPointer<int>(0)[0]);
    uint32_t strideW = static_cast<uint32_t>(attrs->GetAttrPointer<int>(1)[0]);
    uint32_t padH = static_cast<uint32_t>(attrs->GetAttrPointer<int>(2)[0]);
    uint32_t padW = static_cast<uint32_t>(attrs->GetAttrPointer<int>(3)[0]);
    uint32_t dilationH = static_cast<uint32_t>(attrs->GetAttrPointer<int>(4)[0]);
    uint32_t dilationW = static_cast<uint32_t>(attrs->GetAttrPointer<int>(5)[0]);
    uint32_t groups = static_cast<uint32_t>(attrs->GetAttrPointer<int>(6)[0]);

    uint32_t outputHeight = (inputHeight + 2 * padH - dilationH * (kernelHeight - 1) - 1) / strideH + 1;
    uint32_t outputWidth = (inputWidth + 2 * padW - dilationW * (kernelWidth - 1) - 1) / strideW + 1;

    uint32_t totalLength = numBatch * numChannels * outputHeight * outputWidth;

    // tileLen from UB budget: bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_dilationH(dilationH);
    tiling.set_dilationW(dilationW);
    tiling.set_groups(groups);
    tiling.set_kernelHeight(kernelHeight);
    tiling.set_kernelWidth(kernelWidth);
    tiling.set_inputHeight(inputHeight);
    tiling.set_inputWidth(inputWidth);
    tiling.set_outputHeight(outputHeight);
    tiling.set_outputWidth(outputWidth);
    tiling.set_numBatch(numBatch);
    tiling.set_numChannels(numChannels);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

    uint32_t inputHeight = x_shape->GetDim(2);
    uint32_t inputWidth = x_shape->GetDim(3);
    uint32_t kernelHeight = w_shape->GetDim(2);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t KERNEL_HEIGHT = 3;
constexpr int32_t KERNEL_WIDTH = 3;
constexpr int32_t KERNEL_SIZE = KERNEL_HEIGHT * KERNEL_WIDTH;

class KernelConvDepthwise2dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR w,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups, uint32_t kernelHeight, uint32_t kernelWidth,
                                uint32_t inputHeight, uint32_t inputWidth,
                                uint32_t outputHeight, uint32_t outputWidth,
                                uint32_t numBatch, uint32_t numChannels)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;
        kernelHeight_ = kernelHeight;
        kernelWidth_ = kernelWidth;
        inputHeight_ = inputHeight;
        inputWidth_ = inputWidth;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        numBatch_ = numBatch;
        numChannels_ = numChannels;
        outputSizePerChannel_ = outputHeight_ * outputWidth_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, numBatch_ * numChannels_ * inputHeight_ * inputWidth_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, numChannels_ * kernelHeight_ * kernelWidth_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, 1, KERNEL_SIZE * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueY_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueAcc_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        CopyIn(base, curLen);
        CopyInWeight();

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInWeight()
    {
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_[0], KERNEL_SIZE);
        inQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> accLocal = tmpQueueAcc_.AllocTensor<float>();

        // Initialize accumulator with zeros
        AscendC::Sets(accLocal, 0.0f, len);

        // For each output position, compute convolution
        // Since this is depthwise, each channel processes independently
        // We need to handle the spatial convolution manually
        for (int32_t kh = 0; kh < kernelHeight_; ++kh) {
            for (int32_t kw = 0; kw < kernelWidth_; ++kw) {
                float wVal = wLocal.GetValue(kh * kernelWidth_ + kw);
                // Create broadcast tensor for weight
                AscendC::LocalTensor<float> wBroadcast = tmpQueueY_.AllocTensor<float>();
                AscendC::Sets(wBroadcast, wVal, len);
                
                // Multiply and accumulate
                AscendC::Muls(tmpQueueY_, xLocal, wBroadcast, len);
                AscendC::Add(accLocal, accLocal, tmpQueueY_, len);
                
                tmpQueueY_.FreeTensor(wBroadcast);
            }
        }

        // Copy result to output
        AscendC::Copy(yLocal, accLocal, len);

        tmpQueueAcc_.FreeTensor(accLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueW_.FreeTensor(wLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueAcc_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t strideH_ {1};
    uint32_t strideW_ {1};
    uint32_t padH_ {0};
    uint32_t padW_ {0};
    uint32_t dilationH_ {1};
    uint32_t dilationW_ {1};
    uint32_t groups_ {1};
    uint32_t kernelHeight_ {3};
    uint32_t kernelWidth_ {3};
    uint32_t inputHeight_ {0};
    uint32_t inputWidth_ {0};
    uint32_t outputHeight_ {0};
    uint32_t outputWidth_ {0};
    uint32_t numBatch_ {0};
    uint32_t numChannels_ {0};
    uint32_t outputSizePerChannel_ {0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR w, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymmetricInputSquareKernelCustom op;
    op.Init(x, y, w, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups, tiling_data.kernelHeight, tiling_data.kernelWidth,
            tiling_data.inputHeight, tiling_data.inputWidth,
            tiling_data.outputHeight, tiling_data.outputWidth,
            tiling_data.numBatch, tiling_data.numChannels);
    op.Process();
}
