constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t ELEMENTS_PER_VEC = 16;

class KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR weight, GM_ADDR y,
        uint32_t totalLength, uint32_t tileLength,
        uint32_t batchSize, uint32_t inChannels, uint32_t outChannels,
        uint32_t inDepth, uint32_t inHeight, uint32_t inWidth,
        uint32_t kernelD, uint32_t kernelH, uint32_t kernelW,
        uint32_t outDepth, uint32_t outHeight, uint32_t outWidth,
        int32_t strideD, int32_t strideH, int32_t strideW,
        int32_t paddingD, int32_t paddingH, int32_t paddingW,
        int32_t outputPaddingD, int32_t outputPaddingH, int32_t outputPaddingW,
        int32_t dilationD, int32_t dilationH, int32_t dilationW,
        int32_t groups)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        batchSize_ = batchSize;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        inDepth_ = inDepth;
        inHeight_ = inHeight;
        inWidth_ = inWidth;
        kernelD_ = kernelD;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        outDepth_ = outDepth;
        outHeight_ = outHeight;
        outWidth_ = outWidth;

        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        paddingD_ = paddingD;
        paddingH_ = paddingH;
        paddingW_ = paddingW;
        outputPaddingD_ = outputPaddingD;
        outputPaddingH_ = outputPaddingH;
        outputPaddingW_ = outputPaddingW;
        dilationD_ = dilationD;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;

        inChannelGroup_ = inChannels / groups;
        outChannelGroup_ = outChannels / groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, kernelD_ * kernelH_ * kernelW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        AscendC::SetVectorConstant(yLocal, 0.0f, len);

        for (uint32_t g = 0; g < static_cast<uint32_t>(groups_); ++g) {
            for (uint32_t oc = 0; oc < outChannelGroup_; ++oc) {
                uint32_t outChannelGlobal = g * outChannelGroup_ + oc;

                for (uint32_t d = 0; d < kernelD_; ++d) {
                    for (uint32_t h = 0; h < kernelH_; ++h) {
                        for (uint32_t w = 0; w < kernelW_; ++w) {
                            uint32_t weightIdx = outChannelGlobal * kernelD_ * kernelH_ * kernelW_ +
                                                 d * kernelH_ * kernelW_ +
                                                 h * kernelW_ +
                                                 w;

                            for (uint32_t ic = 0; ic < inChannelGroup_; ++ic) {
                                uint32_t inChannelGlobal = g * inChannelGroup_ + ic;

                                float weightVal = weightGm_.GetValue(weightIdx);

                                for (uint32_t i = 0; i < len; ++i) {
                                    uint32_t n = (gmOffsetBase_ + i) / (outChannels_ * outDepth_ * outHeight_ * outWidth_);
                                    uint32_t rem = (gmOffsetBase_ + i) % (outChannels_ * outDepth_ * outHeight_ * outWidth_);
                                    uint32_t od = rem / (outHeight_ * outWidth_);
                                    rem = rem % (outHeight_ * outWidth_);
                                    uint32_t oh = rem / outWidth_;
                                    uint32_t ow = rem % outWidth_;

                                    int32_t inD = od * strideD_ - paddingD_ + d * dilationD_;
                                    int32_t inH = oh * strideH_ - paddingH_ + h * dilationH_;
                                    int32_t inW = ow * strideW_ - paddingW_ + w * dilationW_;

                                    if (inD >= 0 && inD < static_cast<int32_t>(inDepth_) &&
                                        inH >= 0 && inH < static_cast<int32_t>(inHeight_) &&
                                        inW >= 0 && inW < static_cast<int32_t>(inWidth_)) {
                                        uint32_t inIdx = n * inChannels_ * inDepth_ * inHeight_ * inWidth_ +
                                                         inChannelGlobal * inDepth_ * inHeight_ * inWidth_ +
                                                         inD * inHeight_ * inWidth_ +
                                                         inH * inWidth_ +
                                                         inW;
                                        yLocal.SetValue(i, yLocal.GetValue(i) + xLocal.GetValue(inIdx) * weightVal);
                                    }
                                }
                            }
                        }
                    }
                }
            }
        }

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

    uint32_t gmOffsetBase_;

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};

    uint32_t batchSize_ {0};
    uint32_t inChannels_ {0};
    uint32_t outChannels_ {0};
    uint32_t inDepth_ {0};
    uint32_t inHeight_ {0};
    uint32_t inWidth_ {0};
    uint32_t kernelD_ {0};
    uint32_t kernelH_ {0};
    uint32_t kernelW_ {0};
    uint32_t outDepth_ {0};
    uint32_t outHeight_ {0};
    uint32_t outWidth_ {0};

    int32_t strideD_ {1};
    int32_t strideH_ {1};
    int32_t strideW_ {1};
    int32_t paddingD_ {0};
    int32_t paddingH_ {0};
    int32_t paddingW_ {0};
    int32_t outputPaddingD_ {0};
    int32_t outputPaddingH_ {0};
    int32_t outputPaddingW_ {0};
    int32_t dilationD_ {1};
    int32_t dilationH_ {1};
    int32_t dilationW_ {1};
    int32_t groups_ {1};

    uint32_t inChannelGroup_ {0};
