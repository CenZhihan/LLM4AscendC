static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedBothTilingData tiling;
    
    const uint32_t mSize = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t kSize = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());
    const uint32_t nSize = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());

    // Simplified tiling: divide work across m dimension
    // UB budget: M * kBlock * 4 + kBlock * N * 4 + M * N * 4 for temp
    // Conservative: use kTile = 1 to minimize UB usage, increase mBlocks for parallelism
    uint32_t mTile = MAX_M_TILE;
    uint32_t kTile = 1;
    uint32_t nTile = MAX_N_TILE;

    uint32_t mBlocks = CeilDiv(mSize, mTile);
    uint32_t kBlocks = CeilDiv(kSize, kTile);
    
    // Ensure reasonable block count
    uint32_t blockDim = mBlocks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;

    context->SetBlockDim(blockDim);

    tiling.set_mSize(mSize);
    tiling.set_kSize(kSize);
    tiling.set_nSize(nSize);
    tiling.set_mTile(mTile);
    tiling.set_kTile(kTile);
    tiling.set_nTile(nTile);
    tiling.set_blockDim(blockDim);
    tiling.set_mBlocks(mBlocks);
    tiling.set_kBlocks(kBlocks);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulWithTransposedBoth {
public:
    __aicore__ inline KernelMatmulWithTransposedBoth() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR output,
                                uint32_t mSize, uint32_t kSize, uint32_t nSize,
                                uint32_t mTile, uint32_t kTile, uint32_t nTile,
                                uint32_t mBlocks, uint32_t kBlocks)
    {
        mSize_ = mSize;
        kSize_ = kSize;
        nSize_ = nSize;
        mTile_ = mTile;
        kTile_ = kTile;
        nTile_ = nTile;
        mBlocks_ = mBlocks;
        kBlocks_ = kBlocks;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        outputGm_.SetGlobalBuffer((__gm__ float*)output);

        // UB allocation: A (M * mTile), B (nTile * N), C (M * nTile)
        uint32_t aSize = mSize_ * mTile_ * sizeof(float);
        uint32_t bSize = nTile_ * nSize_ * sizeof(float);
        uint32_t cSize = mSize_ * nTile_ * sizeof(float);
        
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, aSize);
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, bSize);
        pipe_.InitBuffer(outQueueZ_, BUFFER_NUM, cSize);
        pipe_.InitBuffer(tmpQueue_, 1, cSize);
    }

    __aicore__ inline void Process()
    {
        if (mSize_ == 0 || kSize_ == 0 || nSize_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        
        // Each block handles mTile rows of the output
        uint32_t mStart = blockIdx * mTile_;
        if (mStart >= mSize_) return;
        
        uint32_t mLen = mSize_ - mStart;
        if (mLen > mTile_) mLen = mTile_;

        // Copy entire M-tile of A into UB
        CopyInA(mStart, mLen);

        // Process all K dimension
        for (uint32_t kBlock = 0; kBlock < kBlocks_; ++kBlock) {
            uint32_t kStart = kBlock * kTile_;
            uint32_t kLen = kSize_ - kStart;
            if (kLen > kTile_) kLen = kTile_;

            // Copy K-tile of B
            CopyInB(kStart, kLen);

            // Compute partial: C[mStart:mStart+mLen, :] += A[mStart:mStart+mLen, kStart:kStart+kLen] * B[kStart:kStart+kLen, :]
            ComputePartial(mStart, mLen, kStart, kLen);

            if (kBlock < kBlocks_ - 1) {
                // Prefetch next B tile
                uint32_t nextKStart = (kBlock + 1) * kTile_;
                uint32_t nextKLen = kSize_ - nextKStart;
                if (nextKLen > kTile_) nextKLen = kTile_;
                CopyInB(nextKStart, nextKLen);
            }
        }

        // Copy result out
        CopyOut(mStart, mLen);
    }

private:
    __aicore__ inline void CopyInA(uint32_t mOffset, uint32_t mLen)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[mOffset], mLen * kSize_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInB(uint32_t kOffset, uint32_t kLen)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        // B is stored as (N, K), we need column kOffset with kLen elements
        for (uint32_t n = 0; n < nSize_; ++n) {
            AscendC::DataCopy(yLocal[n * kLen + 0], yGm_[n * kSize_ + kOffset], kLen);
        }
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void ComputePartial(uint32_t mOffset, uint32_t mLen, uint32_t kOffset, uint32_t kLen)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = inQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> zLocal = outQueueZ_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Access A: xLocal is [mLen, kSize_], need columns kOffset:kOffset+kLen
        // Access B: yLocal is [kLen, nSize_] transposed view
        
        // For each output position (m, n):
        // z[m, n] += sum over k of A[m, k] * B[k, n]
        // A is stored row-major, B is stored row-major (original B transposed)
        
        // Initialize zLocal to zero for first k-block
        if (kOffset == 0) {
            for (uint32_t i = 0; i < mLen * nSize_; ++i) {
                tLocal.SetValue<float>(i, 0.0f);
            }
        } else {
            AscendC::DataCopy(tLocal, zLocal, mLen * nSize_);
        }
        
        // Compute matmul for this k-block
        // A_block = xLocal[:, kOffset:kOffset+kLen] shape (mLen, kLen)
        // B_block needs to be accessed from yGm directly
        
        // Use vectorized approach: for each m and n, compute dot product
        for (uint32_t m = 0; m < mLen; ++m) {
            for (uint32_t n = 0; n < nSize_; ++n) {
                float sum = 0.0f;
                // sum over k: A[m, k] * B[k, n]
                for (uint32_t k = 0; k < kLen; ++k) {
                    // A[m, k] = xLocal[m * kSize_ + kOffset + k]
                    // B[k, n] = B_original[n, k] but we have yGm[n * kSize_ + k]
                    float a_val = xLocal.GetValue<float>(m * kSize_ + kOffset + k);
                    float b_val = yGm_[n * kSize_ + kOffset + k];
                    sum += a_val * b_val;
                }
                uint32_t z_idx = m * nSize_ + n;
                float prev = tLocal.GetValue<float>(z_idx);
                tLocal.SetValue<float>(z_idx, prev + sum);
            }
        }
        
        AscendC::DataCopy(zLocal, tLocal, mLen * nSize_);

        tmpQueue_.FreeTensor(tLocal);
        outQueueZ_.EnQue<float>(zLocal);
        inQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t mOffset, uint32_t mLen)
    {
        AscendC::LocalTensor<float> zLocal = outQueueZ_.DeQue<float>();
        AscendC::DataCopy(outputGm_[mOffset * nSize_], zLocal, mLen * nSize_);
        outQueueZ_.FreeTensor(zLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> outputGm_;

    uint32_t mSize_ {0};
    uint32_t kSize_ {0};
    uint32_t nSize_ {0};
    uint32_t mTile_ {0};
    uint32_t kTile_ {0};
    uint32_t nTile_ {0};
    uint32_t mBlocks_ {0};
    uint32_t kBlocks_ {0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR x, GM_ADDR y, GM_ADDR output, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedBoth op;
    op.Init(x, y, output,
            tiling_data.mSize, tiling_data.kSize, tiling_data.nSize,
            tiling_data.mTile, tiling_data.kTile, tiling_data.nTile,
            tiling_data.mBlocks, tiling_data.kBlocks);
    op.Process();
}
