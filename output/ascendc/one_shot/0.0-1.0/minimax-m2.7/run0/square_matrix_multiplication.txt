kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSquareMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelSquareMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileRow, uint32_t tileCol, uint32_t tileK)
    {
        N_ = N;
        tileRow_ = tileRow;
        tileCol_ = tileCol;
        tileK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, (size_t)N * N);
        bGm_.SetGlobalBuffer((__gm__ float*)B, (size_t)N * N);
        cGm_.SetGlobalBuffer((__gm__ float*)C, (size_t)N * N);

        const uint32_t aSize = tileRow_ * tileK_;
        const uint32_t bSize = tileK_ * tileCol_;
        const uint32_t cSize = tileRow_ * tileCol_;

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, aSize * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, bSize * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, cSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, cSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, cSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        const uint32_t tilesN = (N_ + tileCol_ - 1u) / tileCol_;
        const uint32_t mBlockIdx = blockIdx / tilesN;
        const uint32_t nBlockIdx = blockIdx % tilesN;

        const uint32_t mStart = mBlockIdx * tileRow_;
        const uint32_t nStart = nBlockIdx * tileCol_;
        const uint32_t mLen = (mStart + tileRow_ <= N_) ? tileRow_ : (N_ - mStart);
        const uint32_t nLen = (nStart + tileCol_ <= N_) ? tileCol_ : (N_ - nStart);

        const uint32_t cSize = mLen * nLen;

        AscendC::LocalTensor<float> cInit = outQueueC_.AllocTensor<float>();
        AscendC::VectorFill(cInit, 0.0f, cSize);
        outQueueC_.EnQue(cInit);

        uint32_t kBufIdx = 0;

        for (uint32_t k = 0; k < N_; k += tileK_) {
            const uint32_t kLen = (k + tileK_ <= N_) ? tileK_ : (N_ - k);
            const uint32_t nextK = k + tileK_;
            const bool hasNext = (nextK < N_);

            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            for (uint32_t i = 0; i < mLen; ++i) {
                for (uint32_t j = 0; j < kLen; ++j) {
                    aLocal.SetValue(i * kLen + j, aGm_.GetValue((mStart + i) * N_ + k + j));
                }
            }
            inQueueA_.EnQue(aLocal);

            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
            for (uint32_t i = 0; i < kLen; ++i) {
                for (uint32_t j = 0; j < nLen; ++j) {
                    bLocal.SetValue(i * nLen + j, bGm_.GetValue((k + i) * N_ + nStart + j));
                }
            }
            inQueueB_.EnQue(bLocal);

            AscendC::LocalTensor<float> cLocal = outQueueC_.DeQue<float>();
            MatMul(aLocal, bLocal, cLocal, mLen, kLen, nLen);
            outQueueC_.EnQue(cLocal);

            inQueueA_.FreeTensor(aLocal);
            inQueueB_.FreeTensor(bLocal);

            if (hasNext) {
                const uint32_t nextKLen = (nextK + tileK_ <= N_) ? tileK_ : (N_ - nextK);
                AscendC::LocalTensor<float> aNext = inQueueA_.AllocTensor<float>();
                for (uint32_t i = 0; i < mLen; ++i) {
                    for (uint32_t j = 0; j < nextKLen; ++j) {
                        aNext.SetValue(i * nextKLen + j, aGm_.GetValue((mStart + i) * N_ + nextK + j));
                    }
                }
                inQueueA_.EnQue(aNext);

                AscendC::LocalTensor<float> bNext = inQueueB_.AllocTensor<float>();
                for (uint32_t i = 0; i < nextKLen; ++i) {
                    for (uint32_t j = 0; j < nLen; ++j) {
                        bNext.SetValue(i * nLen + j, bGm_.GetValue((nextK + i) * N_ + nStart + j));
                    }
                }
                inQueueB_.EnQue(bNext);
            }
        }

        AscendC::LocalTensor<float> result = outQueueC_.DeQue<float>();
        for (uint32_t i = 0; i < mLen; ++i) {
            for (uint32_t j = 0; j < nLen; ++j) {
                cGm_.SetValue((mStart + i) * N_ + nStart + j, result.GetValue(i * nLen + j));
            }
        }
        outQueueC_.FreeTensor(result);
    }

private:
    __aicore__ inline void MatMul(const AscendC::LocalTensor<float> &aLocal,
                                  const AscendC::LocalTensor<float> &bLocal,
                                  const AscendC::LocalTensor<float> &cLocal,
                                  uint32_t m, uint32_t k, uint32_t n)
    {
        AscendC::LocalTensor<float> tmp1 = tmpQueue1_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp2 = tmpQueue2_.AllocTensor<float>();

        for (uint32_t i = 0; i < m; ++i) {
            for (uint32_t j = 0; j < n; ++j) {
                float sum = 0.0f;
                for (uint32_t p = 0; p < k; ++p) {
                    sum += aLocal.GetValue(i * k + p) * bLocal.GetValue(p * n + j);
                }
                tmp1.SetValue(i * n + j, sum);
            }
        }

        AscendC::Adds(tmp2, tmp1, cLocal, m * n);

        tmpQueue1_.FreeTensor(tmp1);
        tmpQueue2_.FreeTensor(tmp2);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_ {0};
    uint32_t tileRow_ {0};
    uint32_t tileCol_ {0};
    uint32_t tileK_ {0};
};

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSquareMatrixMultiplicationCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.tileRow, tiling_data.tileCol, tiling_data.tileK);
    op.Process();
}
