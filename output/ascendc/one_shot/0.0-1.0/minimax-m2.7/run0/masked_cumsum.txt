host_operator_src = """
#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per‑core share
static constexpr uint32_t FLOATS_PER_ELEM = 7;           // x(2)+mask(2)+out(2)+tmp(1)
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    MaskedCumsumCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const gert::OriginalShape &originShape = shape->GetOriginShape();
    const uint32_t totalLength = static_cast<uint32_t>(originShape.GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t dim = (dimPtr != nullptr) ? *dimPtr : 1;

    const int32_t ndim = static_cast<int32_t>(originShape.GetDims());
    // Clamp dim to a valid range (allow negative indexing)
    int32_t dimIdx = (dim >= 0) ? dim : ndim + dim;
    if (dimIdx < 0) dimIdx = 0;
    if (dimIdx >= ndim) dimIdx = ndim - 1;

    const uint32_t lineLength = static_cast<uint32_t>(originShape.GetDim(dimIdx));
    const uint32_t preDimProd = (lineLength != 0) ? (totalLength / lineLength) : 1u;

    // Tile length from UB budget: bytes = FLOATS_PER_ELEM * tileLen * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (FLOATS_PER_ELEM * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLength = maxTile;
    if (lineLength < tileLength) {
        const uint32_t aligned = AlignDown(lineLength, ALIGN_ELEMS);
        tileLength = (aligned > 0) ? aligned : lineLength;
    }

    uint32_t blockDim = preDimProd;               // one block per line
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<uint32_t>(dimIdx));
    tiling.set_lineLength(lineLength);
    tiling.set_preDimProd(preDimProd);
    tiling.set_phase(0);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *xShape = context->GetInputShape(0);
    gert::Shape *yShape = context->GetOutputShape(0);
    *yShape = *xShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
}  // namespace ge

namespace ops {
class MaskedCumsumCustom : public OpDef {
public:
    explicit MaskedCumsumCustom(const char *name) : OpDef(name) {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})

        this->Input("mask")
            .ParamType(REQUIRED)

        this->Output("y")

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

    }

OP_ADD(MaskedCumsumCustom);
}  // namespace ops

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t dim,
                                uint32_t lineLength, uint32_t preDimProd,
                                uint32_t phase) {
        totalLength_ = totalLength;
        tileLength_ = tileLength;
        blockDim_ = blockDim;
        dim_ = dim;
        lineLength_ = lineLength;
        preDimProd_ = preDimProd;
        phase_ = phase;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLength_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLength_ * sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLength_ == 0 || lineLength_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= preDimProd_) return;

        const uint32_t lineIdx = blockIdx;
        const uint32_t lineOffsetBase = lineIdx * lineLength_;
        const uint32_t tileCount = (lineLength_ + tileLength_ - 1) / tileLength_;

        float cum = 0.0f;
        for (uint32_t tileIdx = 0; tileIdx < tileCount; ++tileIdx) {
            uint32_t offset = lineOffsetBase + tileIdx * tileLength_;
            uint32_t len = (tileIdx == tileCount - 1) ? (lineLength_ - tileIdx * tileLength_) : tileLength_;

            // Load x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            inQueueX_.EnQue(xLocal);

            // Load mask
            AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
            AscendC::DataCopy(maskLocal, maskGm_[offset], len);
            inQueueMask_.EnQue(maskLocal);

            // Dequeue for computation
            xLocal = inQueueX_.DeQue<float>();
            maskLocal = inQueueMask_.DeQue<float>();

            // Compute masked = x * mask
            AscendC::LocalTensor<float> maskedLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(maskedLocal, xLocal, maskLocal, len);

            // Free input tensors
            inQueueX_.FreeTensor(xLocal);
            inQueueMask_.FreeTensor(maskLocal);

            // Inclusive prefix sum within the tile
            for (uint32_t i = 1; i < len; ++i) {
                maskedLocal[i] = maskedLocal[i] + maskedLocal[i - 1];
            }

            // Capture tile sum (last element before adding offset)
            float tileSum = maskedLocal[len - 1];

            // Add cumulative offset to the whole tile
            AscendC::Adds(maskedLocal, maskedLocal, cum, len);

            // Copy result to global memory
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yGm_[offset], maskedLocal, len);
            outQueueY_.EnQue(yLocal);
            tmpQueue_.FreeTensor(maskedLocal);

            // Update cumulative sum for next tile
            cum += tileSum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLength_ {0};
    uint32_t tileLength_ {0};
    uint32_t blockDim_ {0};
    uint32_t dim_ {0};
    uint32_t lineLength_ {0};
    uint32_t preDimProd_ {0};
    uint32_t phase_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                                          GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.blockDim,
            tiling_data.dim,
            tiling_data.lineLength,
            tiling_data.preDimProd,
            tiling_data.phase);
    op.Process();
}
