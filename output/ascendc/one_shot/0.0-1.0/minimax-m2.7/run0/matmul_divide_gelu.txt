kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t BLOCK_SIZE = 32;
constexpr float GELU_COEFF = 0.044714998f;
constexpr float SQRT_2_OVER_PI = 0.797884583f;

class KernelMatmulDivideGeluCustom {
public:
    __aicore__ inline KernelMatmulDivideGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor,
                                GM_ADDR y, uint32_t batchSize, uint32_t m, uint32_t k, uint32_t n,
                                uint32_t totalLength, uint32_t tileLength, float divisorVal,
                                uint32_t mSize, uint32_t kSize, uint32_t nSize)
    {
        batchSize_ = batchSize;
        m_ = m;
        k_ = k;
        n_ = n;
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        divisor_ = divisorVal;
        invDivisor_ = 1.0f / divisor_;
        mSize_ = mSize;
        kSize_ = kSize;
        nSize_ = nSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * m);
        wGm_.SetGlobalBuffer((__gm__ float*)w, n * k);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        divisorGm_.SetGlobalBuffer((__gm__ float*)divisor, 1);

        pipe_.InitBuffer(xQueueX_, BUFFER_NUM, mSize_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, kSize_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, 1, nSize_ * sizeof(float));
        pipe_.InitBuffer(matmulOutQueue_, BUFFER_NUM, nSize_ * sizeof(float));
        pipe_.InitBuffer(divQueue_, BUFFER_NUM, nSize_ * sizeof(float));
        pipe_.InitBuffer(yQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, nSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, nSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue3_, 1, nSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        const uint32_t batchIdx = gmOffset / n_;
        const uint32_t outIdx = gmOffset % n_;
        
        const uint32_t mStart = 0;
        const uint32_t mLen = m_;
        
        AscendC::LocalTensor<float> xLocal = xQueueX_.AllocTensor<float>();
        for (uint32_t i = 0; i < mLen; ++i) {
            AscendC::DataCopy(xLocal[i], xGm_[batchIdx * m_ + i], 1);
        }
        xQueueX_.EnQue(xLocal);
        
        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        for (uint32_t j = 0; j < n_; ++j) {
            for (uint32_t p = 0; p < k_; ++p) {
                AscendC::DataCopy(wLocal[j * k_ + p], wGm_[j * k_ + p], 1);
            }
        }
        wQueue_.EnQue(wLocal);
        
        AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();
        for (uint32_t j = 0; j < n_; ++j) {
            AscendC::DataCopy(bLocal[j], bGm_[outIdx + j], 1);
        }
        bQueue_.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = xQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = wQueue_.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = bQueue_.DeQue<float>();
        AscendC::LocalTensor<float> matmulOut = matmulOutQueue_.AllocTensor<float>();
        
        AscendC::Matmul(matmulOut, xLocal, wLocal, bLocal, m_, k_, n_, 
                        1.0f, 0.0f, AscendC::MatmulMode::MMC);
        matmulOutQueue_.EnQue(matmulOut);
        
        xQueueX_.FreeTensor(xLocal);
        wQueue_.FreeTensor(wLocal);
        bQueue_.FreeTensor(bLocal);
        
        AscendC::LocalTensor<float> matmulDeq = matmulOutQueue_.DeQue<float>();
        AscendC::LocalTensor<float> divOut = divQueue_.AllocTensor<float>();
        
        AscendC::Muls(divOut, matmulDeq, invDivisor_, n_);
        divQueue_.EnQue(divOut);
        matmulOutQueue_.FreeTensor(matmulDeq);
        
        AscendC::LocalTensor<float> divDeq = divQueue_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = yQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp1 = tmpQueue1_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp2 = tmpQueue2_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp3 = tmpQueue3_.AllocTensor<float>();
        
        AscendC::Muls(tmp1, divDeq, GELU_COEFF, n_);
        AscendC::Muls(tmp2, divDeq, divDeq, n_);
        AscendC::Muls(tmp3, tmp1, tmp2, n_);
        AscendC::Adds(tmp2, divDeq, tmp3, n_);
        AscendC::Muls(tmp1, tmp2, SQRT_2_OVER_PI, n_);
        
        AscendC::Tanh(tmp3, tmp1, n_);
        AscendC::Adds(tmp1, tmp3, 1.0f, n_);
        AscendC::Muls(yLocal, divDeq, tmp1, n_);
        AscendC::Muls(yLocal, yLocal, 0.5f, n_);
        
        tmpQueue3_.FreeTensor(tmp3);
        tmpQueue2_.FreeTensor(tmp2);
        tmpQueue1_.FreeTensor(tmp1);
        yQueueY_.EnQue(yLocal);
        divQueue_.FreeTensor(divDeq);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = yQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        yQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> matmulOutQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> divQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue3_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> divisorGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t m_ {0};
    uint32_t k_ {0};
    uint32_t n_ {0};
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t mSize_ {0};
    uint32_t kSize_ {0};
    uint32_t nSize_ {0};
    float divisor_ {1.0f};
    float invDivisor_ {1.0f};
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulDivideGeluCustom op;
    op.Init(x, w, b, divisor, y,
            tiling_data.batchSize, tiling_data.m, tiling_data.k, tiling_data.n,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.divisor,
            tiling_data.mSize, tiling_data.kSize, tiling_data.nSize);
    op.Process();
}
