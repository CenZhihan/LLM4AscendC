host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t ELEMS_PER_ROW = 393216;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;
    const gert::Shape *shape = context->GetInputShape(0);
    const int64_t dim = context->GetAttrs()->GetAttrPointer<int64_t>(0) ? 
                        *context->GetAttrs()->GetAttrPointer<int64_t>(0) : 1;
    
    const uint32_t totalLength = static_cast<uint32_t>(shape->GetShapeSize());
    uint32_t dimSize = 1;
    if (dim == 1 || dim == -1) {
        dimSize = static_cast<uint32_t>(shape->GetShape().GetDim(1));
    } else if (dim == 0) {
        dimSize = static_cast<uint32_t>(shape->GetShape().GetDim(0));
    }
    if (dimSize == 0) dimSize = ELEMS_PER_ROW;

    uint32_t tileLen = AlignDown(UB_BUDGET_BYTES / (16 * sizeof(float)), ALIGN_ELEMS);
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;
    if (tileLen > dimSize) tileLen = AlignDown(dimSize, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = ALIGN_ELEMS;

    uint32_t rowsPerBlock = CeilDiv(totalLength, dimSize) / 64;
    if (rowsPerBlock < 1) rowsPerBlock = 1;
    uint32_t blockDim = CeilDiv(CeilDiv(totalLength, dimSize), rowsPerBlock);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(dim));
    tiling.set_dimSize(dimSize);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t dim, uint32_t dimSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        dim_ = dim;
        dimSize_ = dimSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t rowsTotal = totalLen_ / dimSize_;
        const uint32_t rowsPerBlock = (rowsTotal + blockNum - 1) / blockNum;
        const uint32_t startRow = blockIdx * rowsPerBlock;
        if (startRow >= rowsTotal) return;
        const uint32_t endRow = (startRow + rowsPerBlock > rowsTotal) ? rowsTotal : startRow + rowsPerBlock;

        for (uint32_t row = startRow; row < endRow; ++row) {
            const uint32_t rowOffset = row * dimSize_;
            ProcessRow(rowOffset, dimSize_);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t rowOffset, uint32_t len)
    {
        uint32_t offset = 0;
        while (offset < len) {
            uint32_t chunkLen = (offset + tileLen_ <= len) ? tileLen_ : (len - offset);
            CopyIn(rowOffset + offset, chunkLen);
            
            const uint32_t nextOffset = offset + tileLen_;
            if (nextOffset < len) {
                CopyInNext(rowOffset + nextOffset, 
                          (nextOffset + tileLen_ <= len) ? tileLen_ : (len - nextOffset));
            }
            
            Compute(chunkLen);
            CopyOut(rowOffset + offset, chunkLen);
            
            offset += tileLen_;
        }
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInNext(uint32_t gmOffset, uint32_t len)
    {
        if (len == 0) return;
        AscendC::LocalTensor<float> xNextLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xNextLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xNextLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Find max: y = max(x, 0)
        AscendC::Mins(yLocal, xLocal, 0.0f, len);
        
        // t = x - max
        AscendC::Sub(tLocal, xLocal, yLocal, len);
        
        // y = exp(x - max)
        AscendC::Exp(yLocal, tLocal, len);
        
        // t = sum(exp(x - max))
        AscendC::Adds(tLocal, yLocal, 0.0f, len);
        
        // y = exp(x - max) / sum
        AscendC::Div(yLocal, yLocal, tLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ {1};
    uint32_t dimSize_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.dim, tiling_data.dimSize);
    op.Process();
}
