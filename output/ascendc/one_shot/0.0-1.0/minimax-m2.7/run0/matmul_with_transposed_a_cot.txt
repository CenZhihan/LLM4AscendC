kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulWithTransposedACustom {
public:
    __aicore__ inline KernelMatmulWithTransposedACustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                uint32_t blockDim)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;
        blockDim_ = blockDim;

        aGm_.SetGlobalBuffer((__gm__ float*)A, K * M);
        bGm_.SetGlobalBuffer((__gm__ float*)B, K * N);
        cGm_.SetGlobalBuffer((__gm__ float*)C, M * N);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileK_ * tileM_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tileRowCount = (M_ + tileM_ - 1) / tileM_;
        const uint32_t tileColCount = (N_ + tileN_ - 1) / tileN_;
        const uint32_t totalTiles = tileRowCount * tileColCount;

        const uint32_t stride = blockDim_; // number of blocks used
        uint32_t tileIdx = blockIdx;

        while (tileIdx < totalTiles) {
            uint32_t row = tileIdx / tileColCount;
            uint32_t col = tileIdx % tileColCount;
            uint32_t i0 = row * tileM_;
            uint32_t j0 = col * tileN_;

            uint32_t tileM_cur = (i0 + tileM_ <= M_) ? tileM_ : (M_ - i0);
            uint32_t tileN_cur = (j0 + tileN_ <= N_) ? tileN_ : (N_ - j0);

            // Allocate accumulator for output tile
            AscendC::LocalTensor<float> cLocal = pipe_.AllocTensor<float>(tileM_cur * tileN_cur);
            AscendC::Fill(cLocal, 0.0f, tileM_cur * tileN_cur);

            // Loop over K dimension
            for (uint32_t k = 0; k < K_; k += tileK_) {
                uint32_t tileK_cur = (k + tileK_ <= K_) ? tileK_ : (K_ - k);

                // Allocate buffers for A and B slices
                AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>(tileK_cur * tileM_cur);
                AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>(tileK_cur * tileN_cur);

                // Compute GM offsets
                uint32_t aOffset = k * M_ + i0; // row stride M_
                uint32_t bOffset = k * N_ + j0; // row stride N_

                // Copy A slice: shape tileK_cur x tileM_cur
                AscendC::DataCopy(aLocal, aGm_[aOffset], tileK_cur * tileM_cur);
                // Copy B slice: shape tileK_cur x tileN_cur
                AscendC::DataCopy(bLocal, bGm_[bOffset], tileK_cur * tileN_cur);

                inQueueA_.EnQue(aLocal);
                inQueueB_.EnQue(bLocal);

                // Compute partial result: (A^T) * B
                AscendC::LocalTensor<float> partialLocal = tmpQueue_.AllocTensor<float>(tileM_cur * tileN_cur);
                AscendC::Mmad<float> mmad;
                mmad.Init(tileM_cur, tileN_cur, tileK_cur, true, false);
                mmad.SetTensorA(aLocal);
                mmad.SetTensorB(bLocal);
                mmad.SetTensorC(partialLocal);
                mmad.Enque();
                mmad.Wait();

                // Accumulate partial result
                AscendC::Add(cLocal, cLocal, partialLocal, tileM_cur * tileN_cur);

                // Free buffers
                tmpQueue_.FreeTensor(partialLocal);
                inQueueA_.FreeTensor(aLocal);
                inQueueB_.FreeTensor(bLocal);
            }

            // Copy result to GM
            uint32_t cOffset = i0 * N_ + j0;
            AscendC::DataCopy(cGm_[cOffset], cLocal, tileM_cur * tileN_cur);

            pipe_.FreeTensor(cLocal);

            tileIdx += stride;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileM_{0};
    uint32_t tileN_{0};
    uint32_t tileK_{0};
    uint32_t blockDim_{0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_a_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{

            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.blockDim);
    op.Process();
}
