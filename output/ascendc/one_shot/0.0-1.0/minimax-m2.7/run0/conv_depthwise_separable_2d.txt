#include "conv_depthwise_separable_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 200 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwiseSeparable2dCustomTilingData tiling;

    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::Shape *dwShape = context->GetInputShape(1);

    uint32_t N = inputShape->GetShapeN();
    uint32_t inputC = inputShape->GetShapeC();
    uint32_t inputH = inputShape->GetShapeH();
    uint32_t inputW = inputShape->GetShapeW();

    uint32_t kernelH = dwShape->GetShapeH();
    uint32_t kernelW = dwShape->GetShapeW();

    uint32_t strideH = 1;
    uint32_t strideW = 1;
    uint32_t padH = 1;
    uint32_t padW = 1;
    uint32_t groups = inputC;

    uint32_t outputH = (inputH + 2 * padH - kernelH) / strideH + 1;
    uint32_t outputW = (inputW + 2 * padW - kernelW) / strideW + 1;

    const gert::Shape *pwShape = context->GetInputShape(2);
    uint32_t outputC = pwShape->GetShapeC();

    uint32_t totalLength = N * outputH * outputW;

    uint32_t tileLength = UB_BUDGET_BYTES / (4 * sizeof(float) * 6);
    tileLength = AlignDown(tileLength, ALIGN_ELEMS);
    if (tileLength < ALIGN_ELEMS) tileLength = ALIGN_ELEMS;
    if (tileLength > HARD_CAP_ELEMS) tileLength = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    if (totalLength < tileLength) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLength = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLength == 0) ? 1u : CeilDiv(totalLength, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_inputC(inputC);
    tiling.set_inputH(inputH);
    tiling.set_inputW(inputW);
    tiling.set_outputC(outputC);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_kernelH(kernelH);
    tiling.set_kernelW(kernelW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_groups(groups);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwiseSeparable2dCustom {
public:
    __aicore__ inline KernelConvDepthwiseSeparable2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wDepthwise, GM_ADDR wPointwise, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inputC, uint32_t inputH, uint32_t inputW,
                                uint32_t outputC, uint32_t outputH, uint32_t outputW,
                                uint32_t kernelH, uint32_t kernelW,
                                uint32_t padH, uint32_t padW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t groups)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inputC_ = inputC;
        inputH_ = inputH;
        inputW_ = inputW;
        outputC_ = outputC;
        outputH_ = outputH;
        outputW_ = outputW;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        padH_ = padH;
        padW_ = padW;
        strideH_ = strideH;
        strideW_ = strideW;
        groups_ = groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x, inputC * inputH * inputW);
        wDepthwiseGm_.SetGlobalBuffer((__gm__ float*)wDepthwise, inputC * kernelH * kernelW);
        wPointwiseGm_.SetGlobalBuffer((__gm__ float*)wPointwise, outputC * inputC);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputC * outputH * outputW);

        pipeDepthwise_.InitBuffer(inQueueXDepth_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipeDepthwise_.InitBuffer(outQueueYDepth_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipeDepthwise_.InitBuffer(tmpQueueDepth_, 1, tileLen_ * sizeof(float));

        pipePointwise_.InitBuffer(inQueueXPointwise_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipePointwise_.InitBuffer(outQueueYPointwise_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipePointwise_.InitBuffer(tmpQueuePointwise_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t pos = 0;
        while (pos < totalLen_) {
            uint32_t curPos = pos + base;
            if (curPos >= totalLen_) break;

            uint32_t curLen = totalLen_ - curPos;
            if (curLen > tileLen_) curLen = tileLen_;

            uint32_t n = curPos / (outputH_ * outputW_);
            uint32_t rest = curPos % (outputH_ * outputW_);
            uint32_t ho = rest / outputW_;
            uint32_t wo = rest % outputW_;

            ComputeDepthwisePointwise(n, ho, wo, curLen);

            pos += blockNum * tileLen_;
        }
    }

private:
    __aicore__ inline void ComputeDepthwisePointwise(uint32_t n, uint32_t ho, uint32_t wo, uint32_t len)
    {
        for (uint32_t e = 0; e < len; ++e) {
            uint32_t pos = e;
            uint32_t nCur = n;
            uint32_t hoCur = ho;
            uint32_t woCur = wo;

            uint32_t tilePos = pos;
            uint32_t tileNB = outputH_ * outputW_;
            while (tilePos >= tileNB) {
                tilePos -= tileNB;
                nCur++;
            }
            uint32_t tmpHo = tilePos / outputW_;
            uint32_t tmpWo = tilePos % outputW_;
            hoCur = tmpHo;
            woCur = tmpWo;

            float dwResult[64];
            for (uint32_t c = 0; c < inputC_; ++c) {
                float sum = 0.0f;
                for (uint32_t kh = 0; kh < kernelH_; ++kh) {
                    int32_t ih = static_cast<int32_t>(hoCur * strideH_ + kh) - padH_;
                    if (ih < 0 || ih >= static_cast<int32_t>(inputH_)) continue;
                    for (uint32_t kw = 0; kw < kernelW_; ++kw) {
                        int32_t iw = static_cast<int32_t>(woCur * strideW_ + kw) - padW_;
                        if (iw < 0 || iw >= static_cast<int32_t>(inputW_)) continue;

                        uint32_t inIdx = nCur * inputC_ * inputH_ * inputW_ + c * inputH_ * inputW_ + ih * inputW_ + iw;
                        uint32_t wIdx = c * kernelH_ * kernelW_ + kh * kernelW_ + kw;
                        sum += xGm_.GetValue(inIdx) * wDepthwiseGm_.GetValue(wIdx);
                    }
                }
                dwResult[c] = sum;
            }

            for (uint32_t oc = 0; oc < outputC_; ++oc) {
                float sum = 0.0f;
                for (uint32_t ic = 0; ic < inputC_; ++ic) {
                    uint32_t wIdx = oc * inputC_ + ic;
                    sum += dwResult[ic] * wPointwiseGm_.GetValue(wIdx);
                }
                uint32_t outIdx = nCur * outputC_ * outputH_ * outputW_ + oc * outputH_ * outputW_ + hoCur * outputW_ + woCur;
                yGm_.SetValue(outIdx, sum);
            }
        }
    }

private:
    AscendC::TPipe pipeDepthwise_;
    AscendC::TPipe pipePointwise_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueXDepth_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueDepth_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueYDepth_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueXPointwise_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueuePointwise_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueYPointwise_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wDepthwiseGm_;
    AscendC::GlobalTensor<float> wPointwiseGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t inputC_;
    uint32_t inputH_;
    uint32_t inputW_;
    uint32_t outputC_;
    uint32_t outputH_;
    uint32_t outputW_;
    uint32_t kernelH_;
    uint32_t kernelW_;
    uint32_t padH_;
    uint32_t padW_;
    uint32_t strideH_;
    uint32_t strideW_;
    uint32_t groups_;
};

extern "C" __global__ __aicore__ void conv_depthwise_separable_2d_custom(GM_ADDR x, GM_ADDR w_depthwise, GM_ADDR w_pointwise, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwiseSeparable2dCustom op;
    op.Init(x, w_depthwise, w_pointwise, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.inputC, tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputC, tiling_data.outputH, tiling_data.outputW,
            tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.groups);
    op.Process();
}
