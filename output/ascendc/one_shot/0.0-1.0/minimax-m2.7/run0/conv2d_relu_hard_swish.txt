kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dReluHardSwishCustom {
public:
    __aicore__ inline KernelConv2dReluHardSwishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t batchSize, uint32_t inputC, uint32_t inputH, uint32_t inputW,
                                uint32_t outputC, uint32_t kernelH, uint32_t kernelW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t outputH, uint32_t outputW,
                                bool hasBias)
    {
        batchSize_ = batchSize;
        inputC_ = inputC;
        inputH_ = inputH;
        inputW_ = inputW;
        outputC_ = outputC;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        outputH_ = outputH;
        outputW_ = outputW;
        hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ uint8_t*)x);
        yGm_.SetGlobalBuffer((__gm__ uint8_t*)y);
        wGm_.SetGlobalBuffer((__gm__ uint8_t*)weight);
        if (hasBias_) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, outputC_);
        }

        const uint32_t outTileSize = outputH_ * outputW_;
        const uint32_t inTileSize = (kernelH_ * kernelW_) * inputC_;
        const uint32_t ubSize = 48 * 1024;
        const uint32_t maxOutTile = AscendC::GetBlockNum() > 0 ? ubSize / sizeof(float) / 2 : 1024;
        outTileLen_ = outTileSize < maxOutTile ? outTileSize : maxOutTile;
        outTileLen_ = (outTileLen_ / 16) * 16;
        if (outTileLen_ < 16) outTileLen_ = 16;

        const uint32_t inBlockSize = kernelH_ * kernelW_ * inputC_ * sizeof(float);
        inTileLen_ = (ubSize * 3 / 8) / inBlockSize;
        if (inTileLen_ < 1) inTileLen_ = 1;
        inTileLen_ = (inTileLen_ / 4) * 4;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, outTileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, inTileLen_ * inBlockSize);
        pipe_.InitBuffer(inQueueBias_, 1, outputC_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outTileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, outTileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t totalOutputElements = batchSize_ * outputC_ * outputH_ * outputW_;
        if (totalOutputElements == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t outputPlaneSize = outputH_ * outputW_;
        const uint32_t totalPlanes = batchSize_ * outputC_;
        const uint32_t planesPerBlock = CeilDiv(totalPlanes, blockNum);
        const uint32_t startPlane = blockIdx * planesPerBlock;

        if (startPlane >= totalPlanes) return;

        const uint32_t endPlane = (startPlane + planesPerBlock > totalPlanes) ? totalPlanes : startPlane + planesPerBlock;

        for (uint32_t plane = startPlane; plane < endPlane; ++plane) {
            const uint32_t n = plane / outputC_;
            const uint32_t c = plane % outputC_;

            if (hasBias_) {
                LoadBias(c);
            }

            for (uint32_t oh = 0; oh < outputH_; ++oh) {
                for (uint32_t ow = 0; ow < outputW_; ++ow) {
                    float result = ComputeSinglePixel(n, c, oh, ow);

                    result = AscendC::Relu(result);

                    float hs = result + 3.0f;
                    hs = hs / 6.0f;
                    float clamp = AscendC::Max(hs, 0.0f);
                    clamp = AscendC::Min(clamp, 1.0f);
                    result = result * clamp;

                    const uint32_t outOffset = ((n * outputC_ + c) * outputH_ + oh) * outputW_ + ow;
                    yGm_[outOffset] = result;
                }
            }
        }
    }

private:
    __aicore__ inline void LoadBias(uint32_t outC)
    {
        AscendC::LocalTensor<float> biasLocal = inQueueBias_.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, biasGm_[outC], 1);
        inQueueBias_.EnQue(biasLocal);
    }

    __aicore__ inline float ComputeSinglePixel(uint32_t n, uint32_t c, uint32_t outH, uint32_t outW)
    {
        float sum = 0.0f;

        if (hasBias_) {
            AscendC::LocalTensor<float> biasLocal = inQueueBias_.DeQue<float>();
            sum = biasLocal.GetValue(0);
            inQueueBias_.FreeTensor(biasLocal);
        }

        for (uint32_t kh = 0; kh < kernelH_; ++kh) {
            int32_t inH = static_cast<int32_t>(outH * strideH_ + kh * dilationH_) - static_cast<int32_t>(padH_);
            if (inH < 0 || inH >= static_cast<int32_t>(inputH_)) continue;

            for (uint32_t kw = 0; kw < kernelW_; ++kw) {
                int32_t inW = static_cast<int32_t>(outW * strideW_ + kw * dilationW_) - static_cast<int32_t>(padW_);
                if (inW < 0 || inW >= static_cast<int32_t>(inputW_)) continue;

                const uint32_t inOffset = ((n * inputC_ + c) * inputH_ + inH) * inputW_ + inW;
                const uint32_t wOffset = (c * inputC_ + 0) * kernelH_ * kernelW_ + kh * kernelW_ + kw;

                sum += xGm_[inOffset] * wGm_[wOffset];
            }
        }

        return sum;
    }

    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b)
    {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueBias_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<uint8_t> xGm_;
    AscendC::GlobalTensor<uint8_t> yGm_;
    AscendC::GlobalTensor<uint8_t> wGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t batchSize_ {0};
    uint32_t inputC_ {0};
    uint32_t inputH_ {0};
    uint32_t inputW_ {0};
    uint32_t outputC_ {0};
    uint32_t kernelH_ {1};
    uint32_t kernelW_ {1};
    uint32_t strideH_ {1};
    uint32_t strideW_ {1};
    uint32_t padH_ {0};
    uint32_t padW_ {0};
    uint32_t dilationH_ {1};
    uint32_t dilationW_ {1};
    uint32_t outputH_ {0};
    uint32_t outputW_ {0};
    uint32_t outTileLen_ {0};
    uint32_t inTileLen_ {0};
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluHardSwishCustom op;
    op.Init(x, y, weight, bias,
            tiling_data.batchSize, tiling_data.inputC, tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputC, tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.outputH, tiling_data.outputW,
            tiling_data.hasBias);
    op.Process();
}
