namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulScalingResidualAddCustomTilingData tiling;
    const uint32_t B = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t N = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());
    const uint32_t K = static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());
    const uint32_t totalLength = B * N;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scaleFactor = attrs->GetAttrPointer<float>(0);

    uint32_t blockDim = context->GetBlockDim();
    uint32_t perBlockRows = CeilDiv(B, blockDim);

    uint32_t maxKBuf = UB_BUDGET_BYTES / (2 * ALIGN_ELEMS * sizeof(float));
    uint32_t tileK = (K < maxKBuf) ? CeilDiv(K, ALIGN_ELEMS) * ALIGN_ELEMS : CeilDiv(maxKBuf, ALIGN_ELEMS) * ALIGN_ELEMS;
    if (tileK < ALIGN_ELEMS) tileK = ALIGN_ELEMS;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_B(B);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_perBlockRows(perBlockRows);
    tiling.set_tileK(tileK);
    tiling.set_scaleFactor(*scaleFactor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulScalingResidualAddCustom {
public:
    __aicore__ inline KernelMatmulScalingResidualAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t B, uint32_t N, uint32_t K,
                                uint32_t perBlockRows, uint32_t tileK,
                                float scaleFactor)
    {
        B_ = B;
        N_ = N;
        K_ = K;
        perBlockRows_ = perBlockRows;
        tileK_ = tileK;
        scaleFactor_ = scaleFactor;
        offset_ = 0;

        xGm_.SetGlobalBuffer((__gm__ float*)x, B * K);
        wGm_.SetGlobalBuffer((__gm__ float*)w, K * N);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N);
        yGm_.SetGlobalBuffer((__gm__ float*)y, B * N);

        const uint32_t UB = UB_BUDGET_BYTES / sizeof(float);
        const uint32_t tileBAlign = (N_ + ALIGN_ELEMS - 1) / ALIGN_ELEMS * ALIGN_ELEMS;
        uint32_t tileB = UB / (2 * tileBAlign + ALIGN_ELEMS);
        tileB = (tileB / ALIGN_ELEMS) * ALIGN_ELEMS;
        if (tileB < ALIGN_ELEMS) tileB = ALIGN_ELEMS;
        tileB_ = tileB;

        uint32_t maxKBuf = UB / (2 * tileB_ + ALIGN_ELEMS);
        uint32_t tK = (K_ < maxKBuf) ? K_ : (maxKBuf / ALIGN_ELEMS) * ALIGN_ELEMS;
        if (tK < ALIGN_ELEMS) tK = ALIGN_ELEMS;
        tileK_ = tK;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileB_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, tileK_ * N_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileB_ * N_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, N_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (B_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockRowOffset = blockIdx * perBlockRows_;
        offset_ = blockRowOffset * N_;
        if (blockRowOffset >= B_) return;

        uint32_t blockRows = B_ - blockRowOffset;
        if (blockRows > tileB_) {
            blockRows = tileB_;
        }

        for (uint32_t k = 0; k < K_; k += tileK_) {
            uint32_t curK = tileK_;
            if (k + tileK_ > K_) {
                curK = K_ - k;
            }

            AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xTile, xGm_[blockRowOffset * K_ + k], blockRows * curK);
            inQueueX_.EnQue(xTile);

            AscendC::LocalTensor<float> wTile = wQueue_.AllocTensor<float>();
            AscendC::DataCopy(wTile, wGm_[k * N_], curK * N_);
            wQueue_.EnQue(wTile);

            AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> wLocal = wQueue_.DeQue<float>();
            AscendC::LocalTensor<float> yTile = outQueueY_.AllocTensor<float>();

            mmOp_.SetMatmulParam(AscendC::MatmulMode::ROW_VEC, blockRows, curK, N_, curK, N_, N_);
            mmOp_(xLocal, wLocal, yTile);

            inQueueX_.FreeTensor(xLocal);
            wQueue_.FreeTensor(wLocal);
            outQueueY_.EnQue(yTile);
        }

        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> biasLocal = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, bGm_, N_);

        for (uint32_t i = 0; i < blockRows; ++i) {
            AscendC::Add(yLocal[i * N_], yLocal[i * N_], biasLocal, N_);
        }

        AscendC::Muls(yLocal, yLocal, scaleFactor_, blockRows * N_);
        AscendC::DataCopy(yGm_[offset_], yLocal, blockRows * N_);

        outQueueY_.FreeTensor(yLocal);
        tmpQueue_.FreeTensor(biasLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    AscendC::MatmulAscend<AscendC::TPosition::VECCALC> mmOp_;

    uint32_t B_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t perBlockRows_{0};
    uint32_t tileB_{0};
    uint32_t tileK_{0};
    uint32_t offset_{0};
    float scaleFactor_{1.0f};
};

extern "C" __global__ __aicore__ void matmul_scaling_residual_add_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScalingResidualAddCustom op;
    op.Init(x, w, b, y, tiling_data.B, tiling_data.N, tiling_data.K,
            tiling_data.perBlockRows, tiling_data.tileK, tiling_data.scaleFactor);
    op.Process();
}
