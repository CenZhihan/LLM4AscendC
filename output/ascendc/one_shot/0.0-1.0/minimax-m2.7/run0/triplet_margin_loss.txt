#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per‑core share
static constexpr uint32_t ALIGN_ELEMS = 128;           // minimum embedding size (align to dim)
static constexpr uint32_t HARD_CAP_ELEMS = 32768;     // avoid too‑large per‑iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    TripletMarginLossCustomTilingData tiling;
    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Embedding dimension (second dimension of the input)
    uint32_t embedDim = static_cast<uint32_t>(context->GetInputShape(0)->GetShape().GetDimValue(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *margin = attrs->GetAttrPointer<float>(0);

    // tileLen from UB budget: 3 inputs (anchor, positive, negative) => 3 * tileLen floats
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float)); // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < embedDim) maxTile = embedDim;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, embedDim);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(*margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context) {
    const gert::Shape *anchor_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    // Output shape is 1‑D: batch size (first dimension of anchor)
    y_shape->SetDimCount(1);
    y_shape->SetDim(0, anchor_shape->GetDim(0));
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context) {
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);

}
} // namespace ge

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                GM_ADDR y, uint32_t totalLength,
                                uint32_t tileLength, float margin) {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        margin_ = margin;

        // Embedding dimension equals tileLength; batch size = totalLen_ / embedDim_
        embedDim_ = tileLength;
        batchSize_ = totalLen_ / embedDim_;

        anchorGm_.SetGlobalBuffer((__gm__ float*)anchor, totalLen_);
        positiveGm_.SetGlobalBuffer((__gm__ float*)positive, totalLen_);
        negativeGm_.SetGlobalBuffer((__gm__ float*)negative, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_);

        pipe_.InitBuffer(inQueueAnchor_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueuePositive_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueNegative_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueLoss_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process() {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes one sample (tileLen_ elements)
        uint32_t offset = blockIdx * tileLen_;
        if (offset >= totalLen_) return;

        uint32_t len = tileLen_;
        if (offset + len > totalLen_) {
            len = totalLen_ - offset;
        }

        // Copy input data for this sample
        CopyInAnchor(offset, len);
        CopyInPositive(offset, len);
        CopyInNegative(offset, len);

        // Compute per‑sample loss
        Compute(len);

        // Write loss_i to output vector
        CopyOutLoss(blockIdx, 1);
    }

private:
    __aicore__ inline void CopyInAnchor(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> local = inQueueAnchor_.AllocTensor<float>();
        AscendC::DataCopy(local, anchorGm_[gmOffset], len);
        inQueueAnchor_.EnQue(local);
    }

    __aicore__ inline void CopyInPositive(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> local = inQueuePositive_.AllocTensor<float>();
        AscendC::DataCopy(local, positiveGm_[gmOffset], len);
        inQueuePositive_.EnQue(local);
    }

    __aicore__ inline void CopyInNegative(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> local = inQueueNegative_.AllocTensor<float>();
        AscendC::DataCopy(local, negativeGm_[gmOffset], len);
        inQueueNegative_.EnQue(local);
    }

    __aicore__ inline void Compute(uint32_t len) {
        // Deque input tensors
        AscendC::LocalTensor<float> aLocal = inQueueAnchor_.DeQue<float>();
        AscendC::LocalTensor<float> pLocal = inQueuePositive_.DeQue<float>();
        AscendC::LocalTensor<float> nLocal = inQueueNegative_.DeQue<float>();

        // Temporary buffer for differences
        AscendC::LocalTensor<float> diff = tmpQueue_.AllocTensor<float>();

        // diff = anchor - positive
        AscendC::Sub(diff, aLocal, pLocal, len);
        // diff = diff * diff (square)
        AscendC::Muls(diff, diff, diff, len);
        // Sum of squared positive distance
        float sumPos = 0.0f;
        for (uint32_t i = 0; i < len; ++i) {
            sumPos += diff[i];
        }

        // Reuse diff for anchor - negative
        AscendC::Sub(diff, aLocal, nLocal, len);
        AscendC::Muls(diff, diff, diff, len);
        float sumNeg = 0.0f;
        for (uint32_t i = 0; i < len; ++i) {
            sumNeg += diff[i];
        }

        tmpQueue_.FreeTensor(diff);

        // Free input tensors
        inQueueAnchor_.FreeTensor(aLocal);
        inQueuePositive_.FreeTensor(pLocal);
        inQueueNegative_.FreeTensor(nLocal);

        // Triplet margin loss per sample
        float loss_i = sumPos - sumNeg + margin_;
        if (loss_i < 0.0f) loss_i = 0.0f;

        // Write loss to output queue
        AscendC::LocalTensor<float> outLocal = outQueueLoss_.AllocTensor<float>();
        outLocal[0] = loss_i;
        outQueueLoss_.EnQue(outLocal);
    }

    __aicore__ inline void CopyOutLoss(uint32_t gmOffset, uint32_t len) {
        AscendC::LocalTensor<float> outLocal = outQueueLoss_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], outLocal, len);
        outQueueLoss_.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueAnchor_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueuePositive_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueNegative_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueLoss_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> anchorGm_;
    AscendC::GlobalTensor<float> positiveGm_;
    AscendC::GlobalTensor<float> negativeGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t embedDim_ {0};
    uint32_t batchSize_ {0};
    float margin_ {1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive,
                                                                 GM_ADDR negative, GM_ADDR y,
                                                                 GM_ADDR workspace, GM_ADDR tiling) {
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y,
            tiling_data.totalLength, tiling_data.tileLength, tiling_data.margin);
    op.Process();
}

model_src = '''
import torch
import torch_npu
import custom_ops_lib

_FN = 'triplet_margin_loss_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, *inputs):
        anchor, positive, negative, margin = inputs
        m = float(margin.reshape(-1)[0].item()) if margin.numel() else 0.0
        loss_vec = getattr(custom_ops_lib, _FN)(anchor, positive, negative, m)
        return loss_vec.mean()
