    return (x / a) * a;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;

    uint32_t inputLength = static_cast<uint32_t>(x_shape->GetShape().at(2));

    uint32_t outChannel = static_cast<uint32_t>(w_shape->GetShape().at(0));

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard1dDilatedStridedCustom {
public:
    __aicore__ inline KernelConvStandard1dDilatedStridedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR w,
                               uint32_t batchSize, uint32_t inChannel, uint32_t outChannel,
                               uint32_t inputLength, uint32_t outputLength, uint32_t kernelSize,
                               uint32_t stride, uint32_t dilation, uint32_t groups,
                               uint32_t totalOutputElems, uint32_t tileLength)
    {
        batchSize_ = batchSize;
        inChannel_ = inChannel;
        outChannel_ = outChannel;
        inputLength_ = inputLength;
        outputLength_ = outputLength;
        kernelSize_ = kernelSize;
        stride_ = stride;
        dilation_ = dilation;
        groups_ = groups;
        totalOutputElems_ = totalOutputElems;
        tileLength_ = tileLength;

        inChannelPerGroup_ = inChannel_ / groups_;
        outChannelPerGroup_ = outChannel_ / groups_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * inChannel_ * inputLength_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, outChannel_ * inChannelPerGroup_ * kernelSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutputElems_);

        pipe_.InitBuffer(xQueueX_, BUFFER_NUM, tileLength_ * inChannelPerGroup_ * kernelSize_ * sizeof(float));
        pipe_.InitBuffer(wQueueW_, BUFFER_NUM, outChannelPerGroup_ * inChannelPerGroup_ * kernelSize_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1, tileLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElems_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t strideTiles = blockNum;
        uint32_t baseTile = blockIdx;
        uint32_t tileIdx = baseTile;

        while (tileIdx * tileLength_ < totalOutputElems_) {
            uint32_t startElem = tileIdx * tileLength_;
            uint32_t curTileLen = totalOutputElems_ - startElem;
            if (curTileLen > tileLength_) curTileLen = tileLength_;

            ComputeTile(startElem, curTileLen);

            tileIdx += strideTiles;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t startElem, uint32_t tileLen)
    {
        uint32_t startBatch = startElem / (outChannel_ * outputLength_);
        uint32_t startOutChannel = (startElem / outputLength_) % outChannel_;
        uint32_t startOutIdx = startElem % outputLength_;

        uint32_t endElem = startElem + tileLen;
        uint32_t endBatch = endElem / (outChannel_ * outputLength_);
        uint32_t endOutChannel = (endElem / outputLength_) % outChannel_;
        uint32_t endOutIdx = endElem % outputLength_;

        for (uint32_t b = startBatch; b <= endBatch; ++b) {
            uint32_t outChannelStart = (b == startBatch) ? startOutChannel : 0;
            uint32_t outChannelEnd = (b == endBatch) ? endOutChannel : outChannel_ - 1;

            for (uint32_t oc = outChannelStart; oc <= outChannelEnd; ++oc) {
                uint32_t groupIdx = oc / outChannelPerGroup_;
                uint32_t ocInGroup = oc % outChannelPerGroup_;

                uint32_t outIdxStart = ((b == startBatch) && (oc == outChannelStart)) ? startOutIdx : 0;
                uint32_t outIdxEnd = ((b == endBatch) && (oc == endOutChannel)) ? endOutIdx : outputLength_ - 1;

                for (uint32_t outIdx = outIdxStart; outIdx <= outIdxEnd; ++outIdx) {
                    uint32_t inIdxBase = outIdx * stride_;

                    float sum = 0.0f;

                    for (uint32_t ic = 0; ic < inChannelPerGroup_; ++ic) {
                        for (uint32_t k = 0; k < kernelSize_; ++k) {
                            uint32_t inIdx = inIdxBase + k * dilation_;
                            if (inIdx >= inputLength_) continue;

                            uint32_t xOffset = b * inChannel_ * inputLength_ + groupIdx * inChannelPerGroup_ * inputLength_ + ic * inputLength_ + inIdx;
                            uint32_t wOffset = oc * inChannelPerGroup_ * kernelSize_ + ic * kernelSize_ + k;

                            sum += xGm_.GetValue(xOffset) * wGm_.GetValue(wOffset);
                        }
                    }

                    uint32_t yOffset = b * outChannel_ * outputLength_ + oc * outputLength_ + outIdx;
                    yGm_.SetValue(yOffset, sum);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t inChannel_ {0};
    uint32_t outChannel_ {0};
    uint32_t inputLength_ {0};
    uint32_t outputLength_ {0};
    uint32_t kernelSize_ {0};
    uint32_t stride_ {1};
    uint32_t dilation_ {1};
    uint32_t groups_ {1};
    uint32_t inChannelPerGroup_ {0};
    uint32_t outChannelPerGroup_ {0};
    uint32_t totalOutputElems_ {0};
    uint32_t tileLength_ {0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR w, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dDilatedStridedCustom op;
    op.Init(x, y, w,
            tiling_data.batchSize, tiling_data.inChannel, tiling_data.outChannel,
            tiling_data.inputLength, tiling_data.outputLength, tiling_data.kernelSize,
            tiling_data.stride, tiling_data.dilation, tiling_data.groups,
            tiling_data.totalOutputElems, tiling_data.tileLength);
    op.Process();
}
