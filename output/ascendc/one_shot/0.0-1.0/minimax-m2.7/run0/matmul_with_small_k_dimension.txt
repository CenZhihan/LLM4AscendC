#include "matmul_with_small_k_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t TILE_M_ELEMS = 512;
static constexpr uint32_t TILE_N_ELEMS = 512;
static constexpr uint32_t ALIGN_M = 16;
static constexpr uint32_t ALIGN_N = 16;
static constexpr uint32_t BLOCK_DIM_CAP = 64;

static inline uint32_t AlignUp(uint32_t x, uint32_t a)
{
    return ((x + a - 1u) / a) * a;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithSmallKDimensionCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    const uint32_t M = static_cast<uint32_t>(a_shape->GetOriginShape().GetShapeSize());
    const uint32_t K = static_cast<uint32_t>(a_shape->GetOriginShape().GetShapeSize()) > 0 
                       ? static_cast<uint32_t>(b_shape->GetOriginShape().GetShapeSize()) / M
                       : 1;
    const uint32_t N = static_cast<uint32_t>(b_shape->GetOriginShape().GetShapeSize()) / K;

    tiling.set_mSize(M);
    tiling.set_kSize(K);
    tiling.set_nSize(N);

    uint32_t mTile = TILE_M_ELEMS;
    uint32_t nTile = TILE_N_ELEMS;

    if (M <= TILE_M_ELEMS) {
        mTile = AlignUp(M, ALIGN_M);
    }
    if (N <= TILE_N_ELEMS) {
        nTile = AlignUp(N, ALIGN_N);
    }

    tiling.set_mTile(mTile);
    tiling.set_nTile(nTile);

    const uint32_t mBlocks = CeilDiv(M, mTile);
    const uint32_t nBlocks = CeilDiv(N, nTile);
    const uint32_t totalBlocks = mBlocks * nBlocks;

    uint32_t blockDim = totalBlocks;
    if (blockDim > BLOCK_DIM_CAP) blockDim = BLOCK_DIM_CAP;

    tiling.set_blockDim(blockDim);
    tiling.set_mBlocks(mBlocks);
    tiling.set_nBlocks(nBlocks);

    context->SetBlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulWithSmallKDimensionCustom {
public:
    __aicore__ inline KernelMatmulWithSmallKDimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t mSize, uint32_t kSize, uint32_t nSize,
                                uint32_t mTile, uint32_t nTile,
                                uint32_t mBlocks, uint32_t nBlocks)
    {
        mSize_ = mSize;
        kSize_ = kSize;
        nSize_ = nSize;
        mTile_ = mTile;
        nTile_ = nTile;
        mBlocks_ = mBlocks;
        nBlocks_ = nBlocks;

        aGm_.SetGlobalBuffer((__gm__ float*)a, mSize * kSize);
        bGm_.SetGlobalBuffer((__gm__ float*)b, kSize * nSize);
        cGm_.SetGlobalBuffer((__gm__ float*)c, mSize * nSize);

        const uint32_t aTileSize = mTile_ * kSize_ * sizeof(float);
        const uint32_t bTileSize = kSize_ * nTile_ * sizeof(float);
        const uint32_t cTileSize = mTile_ * nTile_ * sizeof(float);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, aTileSize);
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, bTileSize);
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, cTileSize);
    }

    __aicore__ inline void Process()
    {
        if (mSize_ == 0 || kSize_ == 0 || nSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalBlocks = mBlocks_ * nBlocks_;
        if (totalBlocks == 0) return;

        for (uint32_t i = blockIdx; i < totalBlocks; i += blockNum) {
            const uint32_t mBlockIdx = i / nBlocks_;
            const uint32_t nBlockIdx = i % nBlocks_;

            uint32_t mOffset = mBlockIdx * mTile_;
            uint32_t nOffset = nBlockIdx * nTile_;

            uint32_t mLen = mSize_ - mOffset;
            if (mLen > mTile_) mLen = mTile_;
            mLen = ((mLen + 15) / 16) * 16;

            uint32_t nLen = nSize_ - nOffset;
            if (nLen > nTile_) nLen = nTile_;
            nLen = ((nLen + 15) / 16) * 16;

            ComputeTile(mOffset, nOffset, mLen, nLen);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t mOffset, uint32_t nOffset, uint32_t mLen, uint32_t nLen)
    {
        AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();

        AscendC::DataCopy(aLocal, aGm_[mOffset * kSize_], mLen * kSize_);
        AscendC::DataCopy(bLocal, bGm_[nOffset], kSize_ * nLen);

        AscendC::MatmulParams<float, float, float> matmulParams;
        matmulParams.m = mLen;
        matmulParams.k = kSize_;
        matmulParams.n = nLen;
        matmulParams.lda = kSize_;
        matmulParams.ldb = nLen_;
        matmulParams.ldc = nLen_;
        matmulParams.transA = 1;
        matmulParams.transB = 0;

        AscendC::Matmul(cLocal, aLocal, bLocal, matmulParams);

        AscendC::DataCopy(cGm_[mOffset * nSize_ + nOffset], cLocal, mLen * nLen);

        outQueueC_.FreeTensor(cLocal);
        inQueueB_.FreeTensor(bLocal);
        inQueueA_.FreeTensor(aLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t mSize_ {0};
    uint32_t kSize_ {0};
    uint32_t nSize_ {0};
    uint32_t mTile_ {0};
    uint32_t nTile_ {0};
    uint32_t mBlocks_ {0};
    uint32_t nBlocks_ {0};
};

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithSmallKDimensionCustom op;
    op.Init(a, b, c, tiling_data.mSize, tiling_data.kSize, tiling_data.nSize,
            tiling_data.mTile, tiling_data.nTile, tiling_data.mBlocks, tiling_data.nBlocks);
    op.Process();
}
