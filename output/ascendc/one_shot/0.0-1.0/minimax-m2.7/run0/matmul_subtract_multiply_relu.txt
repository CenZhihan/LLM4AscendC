static constexpr uint32_t TILE_SIZE = 4096;

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulSubtractMultiplyReluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    uint32_t M = 1;
    uint32_t K = 1;
    uint32_t N = 1;

    if (x_shape->GetShapeSize() > 0) {
        const auto &dims = x_shape->GetDims();
        if (dims.size() >= 2) {
            M = static_cast<uint32_t>(dims[0]);
            K = static_cast<uint32_t>(dims[1]);
        } else if (dims.size() == 1) {
            M = 1;
            K = static_cast<uint32_t>(dims[0]);
        }
    }

    if (w_shape->GetShapeSize() > 0) {
        const auto &w_dims = w_shape->GetDims();
        if (w_dims.size() >= 2) {
            N = static_cast<uint32_t>(w_dims[0]);
        } else if (w_dims.size() == 1) {
            N = static_cast<uint32_t>(w_dims[0]);
        }
    }

    uint32_t totalLength = M * N;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float subtractValue = attrs->GetAttrPointer<float>(0) ? *attrs->GetAttrPointer<float>(0) : 0.0f;
    float multiplyValue = attrs->GetAttrPointer<float>(1) ? *attrs->GetAttrPointer<float>(1) : 1.0f;

    uint32_t blockDim = CeilDiv(M * N, TILE_SIZE);
    if (blockDim < 1) blockDim = 1;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;

    uint32_t tileLength = CeilDiv(totalLength, blockDim);
    tileLength = (tileLength + ALIGN_ELEMS - 1) / ALIGN_ELEMS * ALIGN_ELEMS;
    if (tileLength > TILE_SIZE) tileLength = TILE_SIZE;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_m(M);
    tiling.set_k(K);
    tiling.set_n(N);
    tiling.set_subtractValue(subtractValue);
    tiling.set_multiplyValue(multiplyValue);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulSubtractMultiplyReluCustom {
public:
    __aicore__ inline KernelMatmulSubtractMultiplyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t m, uint32_t k, uint32_t n,
                                float subtractValue, float multiplyValue)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        m_ = m;
        k_ = k;
        n_ = n;
        subtractValue_ = subtractValue;
        multiplyValue_ = multiplyValue;

        xGm_.SetGlobalBuffer((__gm__ float*)x, m * k);
        wGm_.SetGlobalBuffer((__gm__ float*)w, n * k);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n);
        yGm_.SetGlobalBuffer((__gm__ float*)y, m * n);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        if (m_ == 0 || k_ == 0 || n_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t rowPerBlock = CeilDiv(m_, blockNum);
        const uint32_t rowStart = blockIdx * rowPerBlock;
        if (rowStart >= m_) return;
        const uint32_t rowEnd = (rowStart + rowPerBlock > m_) ? m_ : rowStart + rowPerBlock;
        const uint32_t rowsThisBlock = rowEnd - rowStart;

        for (uint32_t row = rowStart; row < rowEnd; ++row) {
            uint32_t base = row * k_;
            uint32_t curLen = k_;

            if (curLen <= tileLen_) {
                CopyInRow(xGm_, inQueueX_, base, curLen);
            }

            uint32_t outBase = row * n_;
            for (uint32_t col = 0; col < n_; ++col) {
                float result = bGm_.GetValue(col);

                uint32_t simdLen = curLen;
                uint32_t offset = 0;

                while (simdLen >= 128) {
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
                    AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                    AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

                    AscendC::DataCopy(xLocal, xGm_[base + offset], 128);
                    AscendC::DataCopy(wLocal, wGm_[col + offset], 128);

                    AscendC::Muls(tmpLocal, xLocal, wLocal, 128);
                    AscendC::Adds(yLocal, tmpLocal, 0.0f, 128);

                    uint32_t reduceOffset = 128;
                    while (reduceOffset < simdLen) {
                        uint32_t chunkSize = (reduceOffset + 128 <= simdLen) ? 128 : (simdLen - reduceOffset);
                        AscendC::DataCopy(xLocal, xGm_[base + reduceOffset], chunkSize);
                        AscendC::DataCopy(wLocal, wGm_[col + reduceOffset], chunkSize);
                        AscendC::Muls(tmpLocal, xLocal, wLocal, chunkSize);
                        AscendC::Add(yLocal, yLocal, tmpLocal, chunkSize);
                        reduceOffset += chunkSize;
                    }

                    AscendC::ReduceSum(yLocal, yLocal, 128);
                    result = yLocal.GetValue(0);

                    tmpQueue_.FreeTensor(tmpLocal);
                    outQueueY_.FreeTensor(yLocal);
                    inQueueW_.FreeTensor(wLocal);
                    inQueueX_.FreeTensor(xLocal);

                    offset += 128;
                    simdLen = (simdLen > 128) ? (simdLen - 128) : 0;
                }

                for (uint32_t i = 0; i < simdLen; ++i) {
                    result += xGm_.GetValue(base + offset + i) * wGm_.GetValue(col + offset + i);
                }

                result = result - subtractValue_;
                result = result * multiplyValue_;
                if (result < 0.0f) result = 0.0f;

                yGm_.SetValue(outBase + col, result);
            }

            if (curLen <= tileLen_) {
                inQueueX_.FreeTensor(inQueueX_.DeQue<float>());
            }
        }
    }

private:
    __aicore__ inline void CopyInRow(AscendC::GlobalTensor<float> &srcGm,
                                      AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> &queue,
                                      uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> local = queue.AllocTensor<float>();
        AscendC::DataCopy(local, srcGm[gmOffset], len);
        queue.EnQue(local);
    }

    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b)
    {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t m_ {1};
    uint32_t k_ {1};
    uint32_t n_ {1};
    float subtractValue_ {0.0f};
    float multiplyValue_ {1.0f};
};

extern "C" __global__ __aicore__ void matmul_subtract_multiply_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR sub, GM_ADDR mul, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSubtractMultiplyReluCustom op;
    op.Init(x, w, b, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.m, tiling_data.k, tiling_data.n,
            tiling_data.subtractValue, tiling_data.multiplyValue);
    op.Process();
}
