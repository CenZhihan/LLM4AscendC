constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_KERNEL_SIZE = 27;

class KernelConvTransposed3dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inputDepth, uint32_t inputHeight, uint32_t inputWidth,
                                uint32_t outputDepth, uint32_t outputHeight, uint32_t outputWidth,
                                uint32_t kernelDepth, uint32_t kernelHeight, uint32_t kernelWidth,
                                uint32_t inChannels, uint32_t outChannels,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t padD, uint32_t padH, uint32_t padW,
                                uint32_t dilationD, uint32_t dilationH, uint32_t dilationW,
                                uint32_t outputPadD, uint32_t outputPadH, uint32_t outputPadW,
                                uint32_t groups)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inputDepth_ = inputDepth;
        inputHeight_ = inputHeight;
        inputWidth_ = inputWidth;
        outputDepth_ = outputDepth;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        kernelDepth_ = kernelDepth;
        kernelHeight_ = kernelHeight;
        kernelWidth_ = kernelWidth;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        padD_ = padD;
        padH_ = padH;
        padW_ = padW;
        dilationD_ = dilationD;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        outputPadD_ = outputPadD;
        outputPadH_ = outputPadH;
        outputPadW_ = outputPadW;
        groups_ = groups;

        inputPlane_ = inputDepth_ * inputHeight_ * inputWidth_;
        outputPlane_ = outputDepth_ * outputHeight_ * outputWidth_;
        weightSize_ = kernelDepth_ * kernelHeight_ * kernelWidth_;
        kernelSize_ = kernelDepth_ * kernelHeight_ * kernelWidth_ * inChannels_;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, kernelSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueY_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalTiles = CeilDiv(totalLen_, tileLen_);
        uint32_t startTileIdx = blockIdx;
        uint32_t tilesPerBlock = CeilDiv(totalTiles, blockNum);

        CopyInWeight();

        for (uint32_t t = 0; t < tilesPerBlock; ++t) {
            uint32_t tileIdx = startTileIdx + t * blockNum;
            if (tileIdx >= totalTiles) break;

            uint32_t startPos = tileIdx * tileLen_;
            uint32_t curTileLen = totalLen_ - startPos;
            if (curTileLen > tileLen_) curTileLen = tileLen_;

            CopyIn(startPos, curTileLen);
            Compute(curTileLen);
            CopyOut(startPos, curTileLen);
        }
    }

private:
    __aicore__ inline void CopyInWeight()
    {
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, weightGm_, kernelSize_);
        inQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void CopyIn(uint32_t startPos, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[startPos], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueueY_.AllocTensor<float>();

        AscendC::Sets(tmpLocal, 0.0f, len);

        for (uint32_t pos = 0; pos < len; ++pos) {
            float sum = 0.0f;
            uint32_t outIdx = pos;

            uint32_t od = outIdx / (outputHeight_ * outputWidth_ * outChannels_);
            outIdx %= outputHeight_ * outputWidth_ * outChannels_;
            uint32_t oc = outIdx / (outputHeight_ * outputWidth_);
            outIdx %= outputHeight_ * outputWidth_;
            uint32_t oh = outIdx / outputWidth_;
            uint32_t ow = outIdx % outputWidth_;

            for (uint32_t ic = 0; ic < inChannels_; ++ic) {
                for (uint32_t kd = 0; kd < kernelDepth_; ++kd) {
                    int32_t id = od * strideD_ - padD_ + kd * dilationD_;
                    for (uint32_t kh = 0; kh < kernelHeight_; ++kh) {
                        int32_t ih = oh * strideH_ - padH_ + kh * dilationH_;
                        for (uint32_t kw = 0; kw < kernelWidth_; ++kw) {
                            int32_t iw = ow * strideW_ - padW_ + kw * dilationW_;

                            if (id >= 0 && id < static_cast<int32_t>(inputDepth_) &&
                                ih >= 0 && ih < static_cast<int32_t>(inputHeight_) &&
                                iw >= 0 && iw < static_cast<int32_t>(inputWidth_)) {
                                
                                uint32_t weightIdx = ((oc * inChannels_ + ic) * kernelDepth_ + kd) * kernelHeight_ * kernelWidth_ + kh * kernelWidth_ + kw;
                                uint32_t inputIdx = ((od * inputHeight_ + ih) * inputWidth_ + iw) * inChannels_ + ic;
                                sum += wLocal[weightIdx] * xLocal[inputIdx];
                            }
                        }
                    }
                }
            }
            tmpLocal[pos] = sum;
        }

        AscendC::Copy(yLocal, tmpLocal, len);

        tmpQueueY_.FreeTensor(tmpLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueW_.EnQue<float>(wLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t startPos, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[startPos], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b)
    {
        return (a + b - 1) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t inputDepth_;
    uint32_t inputHeight_;
    uint32_t inputWidth_;
    uint32_t outputDepth_;
    uint32_t outputHeight_;
    uint32_t outputWidth_;
    uint32_t kernelDepth_;
    uint32_t kernelHeight_;
    uint32_t kernelWidth_;
    uint32_t inChannels_;
    uint32_t outChannels_;
    uint32_t strideD_;
    uint32_t strideH_;
    uint32_t strideW_;
    uint32_t padD_;
    uint32_t padH_;
    uint32_t padW_;
    uint32_t dilationD_;
    uint32_t dilationH_;
    uint32_t dilationW_;
    uint32_t outputPadD_;
    uint32_t outputPadH_;
    uint32_t outputPadW_;
    uint32_t groups_;
    uint32_t inputPlane_;
    uint32_t outputPlane_;
    uint32_t weightSize_;
    uint32_t kernelSize_;
};
