namespace optiling {

// UB model for this kernel (per core):
//   inQueueX: 2 * tile (activation)
//   inQueueW: 1 * kSize (weight filter)
//   inQueueB: 1 * co (conv bias, optional)
//   inQueuePB: 1 * co (post bias)
//   outQueueY: 2 * tile
//   tmpQueue1: 1 * tile (min result)
//   tmpQueue2: 1 * tile (add result)
// Total ~ 7 * tile floats => bytes = 28 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMinAddMultiplyCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    const gert::Shape* post_bias_shape = context->GetInputShape(3);

    int32_t n = static_cast<int32_t>(x_shape->GetShape()[0]);
    int32_t c = static_cast<int32_t>(x_shape->GetShape()[1]);
    int32_t h = static_cast<int32_t>(x_shape->GetShape()[2]);
    int32_t w = static_cast<int32_t>(x_shape->GetShape()[3]);
    int32_t kH = static_cast<int32_t>(w_shape->GetShape()[2]);
    int32_t kW = static_cast<int32_t>(w_shape->GetShape()[3]);
    int32_t co = static_cast<int32_t>(w_shape->GetShape()[0]);

    int32_t out_w = w - kW + 1;
    uint32_t totalLength = static_cast<uint32_t>(n * co * out_h * out_w);

    const gert::RuntimeAttrs* attrs = context->GetAttrs();

    uint32_t maxTile = UB_BUDGET_BYTES / (28 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);

    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dMinAddMultiplyCustom {
public:
    __aicore__ inline KernelConv2dMinAddMultiplyCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR post_bias,
                                GM_ADDR y, uint32_t totalLength, uint32_t tileLength,
                                int32_t n, int32_t c, int32_t h, int32_t w,
                                int32_t kH, int32_t kW, int32_t co,
                                float constantValue, float scalingFactor,
                                uint32_t hasConvBias, uint32_t postBiasSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLen_;
        n_ = n;
        c_ = c;
        h_ = h;
        w_ = w;
        kH_ = kH;
        kW_ = kW;
        co_ = co;
        constantValue_ = constantValue;
        scalingFactor_ = scalingFactor;
        hasConvBias_ = hasConvBias;
        postBiasSize_ = postBiasSize;

        outH_ = h_ - kH_ + 1;
        outW_ = w_ - kW_ + 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * c_ * h_ * w_));
        wGm_.SetGlobalBuffer((__gm__ float*)weight, static_cast<uint32_t>(co_ * c_ * kH_ * kW_));

        if (hasConvBias_) {
            convBiasGm_.SetGlobalBuffer((__gm__ float*)conv_bias, static_cast<uint32_t>(co_));
        }
        postBiasGm_.SetGlobalBuffer((__gm__ float*)post_bias, postBiasSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        uint32_t ubSize = tileLen_ * sizeof(float);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, ubSize);
        pipe_.InitBuffer(inQueueW_, 1, static_cast<uint32_t>(co_ * c_ * kH_ * kW_) * sizeof(float));
        pipe_.InitBuffer(inQueueCB_, 1, static_cast<uint32_t>(co_) * sizeof(float));
        pipe_.InitBuffer(inQueuePB_, 1, postBiasSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, ubSize);
        pipe_.InitBuffer(tmpMinQueue_, 1, ubSize);
        pipe_.InitBuffer(tmpAddQueue_, 1, ubSize);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        LoadWeights();

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            ComputeConv(curLen);
            ComputeMin(curLen);
            ComputeAdd(curLen);
            ComputeScale(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void LoadWeights()
    {
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, static_cast<uint32_t>(co_ * c_ * kH_ * kW_));
        inQueueW_.EnQue(wLocal);

        if (hasConvBias_) {
            AscendC::LocalTensor<float> cbLocal = inQueueCB_.AllocTensor<float>();
            AscendC::DataCopy(cbLocal, convBiasGm_, static_cast<uint32_t>(co_));
            inQueueCB_.EnQue(cbLocal);
        }

        AscendC::LocalTensor<float> pbLocal = inQueuePB_.AllocTensor<float>();
        AscendC::DataCopy(pbLocal, postBiasGm_, postBiasSize_);
        inQueuePB_.EnQue(pbLocal);
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeConv(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        int32_t inH = h_;
        int32_t inW = w_;
        int32_t inC = c_;

        for (uint32_t i = 0; i < len; ++i) {
            int32_t outIdx = static_cast<int32_t>(gmOffsetToOutIdx(gmOffset_ + i));
            int32_t oc = outIdx / (outH_ * outW_);
            int32_t ohh = (outIdx % (outH_ * outW_)) / outW_;
            int32_t oww = outIdx % outW_;

            float sum = 0.0f;
            for (int32_t ic = 0; ic < inC; ++ic) {
                for (int32_t kh = 0; kh < kH_; ++kh) {
                    int32_t ih = ohh + kh;
                    for (int32_t kw = 0; kw < kW_; ++kw) {
                        int32_t iw = oww + kw;
                        if (ih >= 0 && ih < inH && iw >= 0 && iw < inW) {
                            int32_t inIdx = oc * inC * inH * inW + ic * inH * inW + ih * inW + iw;
                            int32_t wIdx = oc * inC * kH_ * kW_ + ic * kH_ * kW_ + kh * kW_ + kw;
                            sum += xLocal.GetValue(static_cast<uint32_t>(inIdx)) * wLocal.GetValue(static_cast<uint32_t>(wIdx));
                        }
                    }
                }
            }

            if (hasConvBias_) {
                AscendC::LocalTensor<float> cbLocal = inQueueCB_.PeekCurTensor();
                sum += cbLocal.GetValue(static_cast<uint32_t>(oc));
            }

            yLocal.SetValue(i, sum);
        }

        outQueueY_.EnQue(yLocal);
        inQueueW_.FreeTensor(wLocal);
        inQueueX_.FreeTensor(xLocal);
        if (hasConvBias_) {
            inQueueCB_.FreeTensor(inQueueCB_.DeQue<float>());
        }
    }

    __aicore__ inline uint32_t gmOffsetToOutIdx(uint32_t offset)
    {
        return offset;
    }

    __aicore__ inline void ComputeMin(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> minLocal = tmpMinQueue_.AllocTensor<float>();

        AscendC::Mins(minLocal, yLocal, constantValue_, len);

        tmpMinQueue_.EnQue(minLocal);
        outQueueY_.FreeTensor(yLocal);
    }

    __aicore__ inline void ComputeAdd(uint32_t len)
    {
        AscendC::LocalTensor<float> minLocal = tmpMinQueue_.DeQue<float>();
        AscendC::LocalTensor<float> addLocal = tmpAddQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> pbLocal = inQueuePB_.DeQue<float>();

        AscendC::Adds(addLocal, minLocal, pbLocal, len);

        tmpAddQueue_.EnQue(addLocal);
        inQueuePB_.FreeTensor(pbLocal);
        tmpMinQueue_.FreeTensor(minLocal);
    }

    __aicore__ inline void ComputeScale(uint32_t len)
    {
        AscendC::LocalTensor<float> addLocal = tmpAddQueue_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::Muls(yLocal, addLocal, scalingFactor_, len);

        outQueueY_.EnQue(yLocal);
        tmpAddQueue_.FreeTensor(addLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueCB_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueuePB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMinQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpAddQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> convBiasGm_;
    AscendC::GlobalTensor<float> postBiasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t gmOffset_ {0};
    int32_t n_ {0};
    int32_t c_ {0};
    int32_t h_ {0};
    int32_t w_ {0};
    int32_t kH_ {0};
    int32_t kW_ {0};
    int32_t co_ {0};
    int32_t outH_ {0};
    int32_t outW_ {0};
    float constantValue_ {0.5f};
    float scalingFactor_ {1.0f};
    uint32_t hasConvBias_ {0};
    uint32_t postBiasSize_ {0};
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias,
                                                                      GM_ADDR post_bias, GM_ADDR y,
                                                                      GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinAddMultiplyCustom op;
    op.Init(x, weight, conv_bias, post_bias, y,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kH, tiling_data.kW, tiling_data.co,
            tiling_data.constantValue, tiling_data.scalingFactor,
            tiling_data.hasConvBias, tiling_data.postBiasSize);
    op.Process();
}
