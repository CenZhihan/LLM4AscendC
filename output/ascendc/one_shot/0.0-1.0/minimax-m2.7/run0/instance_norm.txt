    InstanceNormCustomTilingData tiling;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t CHANNELS_PER_BLOCK = 16;

class KernelInstanceNormCustom {
public:
    __aicore__ inline KernelInstanceNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR gamma, GM_ADDR beta,
                                uint32_t batchSize, uint32_t numFeatures,
                                uint32_t height, uint32_t width,
                                uint32_t spatialSize, uint32_t totalLength,
                                float eps, bool hasGammaBeta)
    {
        batchSize_ = batchSize;
        numFeatures_ = numFeatures;
        height_ = height;
        width_ = width;
        spatialSize_ = spatialSize;
        totalLength_ = totalLength;
        eps_ = eps;
        hasGammaBeta_ = hasGammaBeta;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength_);

        if (hasGammaBeta_) {
            gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, numFeatures_);
            betaGm_.SetGlobalBuffer((__gm__ float*)beta, numFeatures_);
        }

        const uint32_t blockLength = CHANNELS_PER_BLOCK * spatialSize_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, blockLength * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, blockLength * sizeof(float));
        pipe_.InitBuffer(tmpQueueMean_, 1, CHANNELS_PER_BLOCK * sizeof(float));
        pipe_.InitBuffer(tmpQueueVar_, 1, CHANNELS_PER_BLOCK * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLength_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t channelsPerBlock = CHANNELS_PER_BLOCK;

        for (uint32_t baseIdx = blockIdx; baseIdx < batchSize_ * numFeatures_;
             baseIdx += channelsPerBlock) {

            uint32_t blockEnd = baseIdx + channelsPer_BLOCK;
            if (blockEnd > batchSize_ * numFeatures_) {
                blockEnd = batchSize_ * numFeatures_;
            }
            uint32_t actualChannels = blockEnd - baseIdx;
            uint32_t curLength = actualChannels * spatialSize_;

            CopyIn(baseIdx, actualChannels);
            Compute(actualChannels);
            CopyOut(baseIdx, actualChannels);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t startChannel, uint32_t numChannels)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        uint32_t offset = startChannel * spatialSize_;
        uint32_t length = numChannels * spatialSize_;
        AscendC::DataCopy(xLocal, xGm_[offset], length);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t numChannels)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> meanLocal = tmpQueueMean_.AllocTensor<float>();
        AscendC::LocalTensor<float> varLocal = tmpQueueVar_.AllocTensor<float>();

        // Compute mean per channel
        // Reshape to (numChannels, spatialSize_) conceptually
        // Reduce axis 1 (spatial dimension)
        AscendC::ReduceSum(meanLocal, xLocal, 0, numChannels, spatialSize_, spatialSize_);

        // Divide by spatial size to get mean
        float invSpatial = 1.0f / static_cast<float>(spatialSize_);
        AscendC::Muls(meanLocal, meanLocal, invSpatial, numChannels);

        // Compute variance: E[(x - mean)^2] = E[x^2] - E[x]^2
        AscendC::LocalTensor<float> xSqLocal = tmpQueueMean_.AllocTensor<float>();

        AscendC::Muls(xSqLocal, xLocal, xLocal, numChannels * spatialSize_);

        AscendC::LocalTensor<float> meanSqLocal = tmpQueueVar_.AllocTensor<float>();
        AscendC::ReduceSum(meanSqLocal, xSqLocal, 0, numChannels, spatialSize_, spatialSize_);
        AscendC::Muls(meanSqLocal, meanSqLocal, invSpatial, numChannels);

        AscendC::Sub(varLocal, meanSqLocal, meanLocal, numChannels);
        AscendC::Muls(meanLocal, meanLocal, meanLocal, numChannels);
        AscendC::Sub(varLocal, varLocal, meanLocal, numChannels);

        // Add epsilon and compute sqrt(1/(var + eps)) for normalization
        AscendC::Adds(varLocal, varLocal, eps_, numChannels);
        AscendC::Rsqrt(varLocal, varLocal, numChannels);

        // Broadcast multiply for normalization
        AscendC::LocalTensor<float> varBcast = varLocal;
        AscendC::Expand(varBcast, varBcast, {numChannels, spatialSize_});

        // y = (x - mean) * rsqrt(var + eps)
        AscendC::Sub(yLocal, xLocal, varBcast, numChannels * spatialSize_, 1,
                     numChannels, spatialSize_, 0, 0);
        AscendC::Muls(yLocal, yLocal, varBcast, numChannels * spatialSize_);

        // Apply gamma and beta if available
        if (hasGammaBeta_) {
            AscendC::LocalTensor<float> gammaLocal = tmpQueueMean_.AllocTensor<float>();
            AscendC::LocalTensor<float> betaLocal = tmpQueueVar_.AllocTensor<float>();

            for (uint32_t c = 0; c < numChannels; ++c) {
                gammaLocal.SetValue(c, gammaGm_.GetValue(startChannel_ + c));
                betaLocal.SetValue(c, betaGm_.GetValue(startChannel_ + c));
            }

            AscendC::LocalTensor<float> gammaBcast = gammaLocal;
            AscendC::Expand(gammaBcast, gammaBcast, {numChannels, spatialSize_});
            AscendC::LocalTensor<float> betaBcast = betaLocal;
            AscendC::Expand(betaBcast, betaBcast, {numChannels, spatialSize_});

            AscendC::Muls(yLocal, yLocal, gammaBcast, numChannels * spatialSize_);
            AscendC::Adds(yLocal, yLocal, betaBcast, numChannels * spatialSize_);
        }

        tmpQueueMean_.FreeTensor(meanLocal);
        tmpQueueVar_.FreeTensor(varLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t startChannel, uint32_t numChannels)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint32_t offset = startChannel * spatialSize_;
        uint32_t length = numChannels * spatialSize_;
        AscendC::DataCopy(yGm_[offset], yLocal, length);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMean_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueVar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;

    uint32_t batchSize_ {0};
    uint32_t numFeatures_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    uint32_t spatialSize_ {0};
    uint32_t totalLength_ {0};
    uint32_t startChannel_ {0};
    float eps_ {1e-5f};
    bool hasGammaBeta_ {false};
};

extern "C" __global__ __aicore__ void instance_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR gamma, GM_ADDR beta, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelInstanceNormCustom op;
    op.Init(x, y, gamma, beta,
            tiling_data.batchSize, tiling_data.numFeatures,
            tiling_data.height, tiling_data.width,
            tiling_data.spatialSize, tiling_data.totalLength,
            tiling_data.eps, tiling_data.hasGammaBeta);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor instance_norm_impl_npu(const at::Tensor& x,
                                   const at::Tensor& gamma,
                                   const at::Tensor& beta,
                                   float eps) {
    at::Tensor result = at::empty_like(x);
    if (gamma.defined() && beta.defined()) {
        EXEC_NPU_CMD(aclnnInstanceNormCustom, x, gamma, beta, eps, result);
    } else {
        at::Tensor emptyGamma = at::empty({x.size(1)}, x.options());
        at::Tensor emptyBeta = at::empty({x.size(1)}, x.options());
        EXEC_NPU_CMD(aclnnInstanceNormCustom, x, emptyGamma, emptyBeta, eps, result);
    }
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("instance_norm_custom", &instance_norm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("instance_norm_custom", &instance_norm_impl_npu, "Instance Normalization (InstanceNormCustom on NPU)");

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'instance_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        num_features = init_inputs[0] if init_inputs else 64
        self.inorm = torch.nn.InstanceNorm2d(num_features=num_features, affine=True)

    def forward(self, *inputs):
        x = inputs[0]
        gamma = self.inorm.weight
        beta = self.inorm.bias
        eps = self.inorm.eps
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, gamma, beta, eps)
