#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed1dDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed1dDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t stride, int32_t padding, int32_t dilation,
                                int32_t inChannels, int32_t outChannels,
                                int32_t kernelSize, int32_t inputLength,
                                int32_t outputLength, int32_t batchSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        stride_ = stride;
        padding_ = padding;
        dilation_ = dilation;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        kernelSize_ = kernelSize;
        inputLength_ = inputLength;
        outputLength_ = outputLength;
        batchSize_ = batchSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);

        const uint32_t weightSize = inChannels * outChannels * kernelSize;
        const uint32_t ySize = batchSize * outChannels * outputLength;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, weightSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        const uint32_t weightSize = inChannels_ * outChannels_ * kernelSize_;
        CopyInWeight(0, weightSize);

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInWeight(uint32_t offset, uint32_t len)
    {
        weightLocal_ = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(weightLocal_, weightGm_[offset], len);
        inQueueW_.EnQue(weightLocal_);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> t1Local = tmpQueue1_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2Local = tmpQueue2_.AllocTensor<float>();

        AscendC::SetVector<float>(yLocal, len, 0.0f);

        for (int32_t b = 0; b < batchSize_; ++b) {
            for (int32_t oc = 0; oc < outChannels_; ++oc) {
                for (int32_t ic = 0; ic < inChannels_; ++ic) {
                    for (int32_t ks = 0; ks < kernelSize_; ++ks) {
                        const int32_t dilatedKs = ks * dilation_;
                        const int32_t outPosBase = b * outChannels_ * outputLength_ + oc * outputLength_;
                        const int32_t inPosBase = b * inChannels_ * inputLength_ + ic * inputLength_;

                        for (int32_t outIdx = 0; outIdx < outputLength_; ++outIdx) {
                            const int32_t inIdx = outIdx - dilatedKs + padding_;
                            if (inIdx >= 0 && inIdx < inputLength_) {
                                const int32_t outGlobalIdx = outPosBase + outIdx;
                                const int32_t inGlobalIdx = inPosBase + inIdx;
                                const float wVal = wLocal.template GetValue<float>(ic * outChannels_ * kernelSize_ + oc * kernelSize_ + ks);
                                const float xVal = xLocal.template GetValue<float>(inGlobalIdx);
                                yLocal.template SetValue<float>(outGlobalIdx, yLocal.template GetValue<float>(outGlobalIdx) + wVal * xVal);
                            }
                        }
                    }
                }
            }
        }

        tmpQueue1_.FreeTensor(t1Local);
        tmpQueue2_.FreeTensor(t2Local);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueW_.EnQue<float>(wLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::LocalTensor<float> weightLocal_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t stride_ {1};
    int32_t padding_ {0};
    int32_t dilation_ {1};
    int32_t inChannels_ {0};
    int32_t outChannels_ {0};
    int32_t kernelSize_ {0};
    int32_t inputLength_ {0};
    int32_t outputLength_ {0};
    int32_t batchSize_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1dDilatedCustom op;
    op.Init(x, y, weight, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation,
            tiling_data.inChannels, tiling_data.outChannels, tiling_data.kernelSize,
            tiling_data.inputLength, tiling_data.outputLength, tiling_data.batchSize);
    op.Process();
}

_FN = 'conv_transposed_1d_dilated_custom'

class ModelNew(torch.nn.Module):

                 stride: int = 1, padding: int = 0, dilation: int = 1, bias: bool = False):
        super(ModelNew, self).__init__()

        self.out_channels = out_channels
        self.kernel_size = kernel_size
        self.stride = stride
        self.padding = padding
        self.dilation = dilation

        self.register_buffer('padding_tensor', torch.tensor([padding], dtype=torch.long))
        self.register_buffer('dilation_tensor', torch.tensor([dilation], dtype=torch.long))

        """
        Performs the transposed 1D convolution with dilation.

        Args:
            x (torch.Tensor): Input tensor of shape (batch_size, in_channels, length).
            weight: Transposed convolution weights

        Returns:
            torch.Tensor: Output tensor of shape (batch_size, out_channels, length_out).
