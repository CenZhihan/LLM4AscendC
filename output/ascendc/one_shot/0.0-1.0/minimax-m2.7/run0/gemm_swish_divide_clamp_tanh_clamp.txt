static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmSwishDivideClampTanhClampCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    int32_t m = static_cast<int32_t>(x_shape->GetShapeVector()[0]);
    int32_t k = static_cast<int32_t>(x_shape->GetShapeVector()[1]);
    int32_t n = static_cast<int32_t>(w_shape->GetShapeVector()[0]);

    uint32_t totalLength = static_cast<uint32_t>(m * n);

    uint32_t maxTile = UB_BUDGET_BYTES / (4 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_m(m);
    tiling.set_k(k);
    tiling.set_n(n);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmSwishDivideClampTanhClampCustom {
public:
    __aicore__ inline KernelGemmSwishDivideClampTanhClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t m, int32_t k, int32_t n)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        m_ = m;
        k_ = k;
        n_ = n;

        xGm_.SetGlobalBuffer((__gm__ float*)x, m * k);
        wGm_.SetGlobalBuffer((__gm__ float*)w, n * k);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n);
        yGm_.SetGlobalBuffer((__gm__ float*)y, m * n);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        uint32_t row = gmOffset / n_;
        uint32_t col = gmOffset % n_;

        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            uint32_t outIdx = gmOffset + i;
            uint32_t r = outIdx / n_;
            uint32_t c = outIdx % n_;
            uint32_t rowOffset = r * k;
            float biasVal = bGm_.GetValue(c);

            float sum = biasVal;
            for (int32_t j = 0; j < k_; ++j) {
                float xv = xGm_.GetValue(rowOffset + j);
                float wv = wGm_.GetValue(c * k_ + j);
                sum += xv * wv;
            }
            xLocal.SetValue(i, sum);
        }

        inQueueX_.EnQue(xLocal);
        inQueueW_.EnQue(wLocal);
        inQueueB_.EnQue(bLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> t1Local = tmpQueue1_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2Local = tmpQueue2_.AllocTensor<float>();

        // y = x * sigmoid(x) (swish)
        AscendC::Sigmoid(t1Local, xLocal, len);
        AscendC::Muls(yLocal, t1Local, xLocal, len);

        // y = y / 2.0
        AscendC::Muls(yLocal, yLocal, 0.5f, len);

        // y = clamp(y, -1.0, 1.0)
        AscendC::Clamp(yLocal, yLocal, -1.0f, 1.0f, len);

        // y = tanh(y)
        AscendC::Tanh(t2Local, yLocal, len);

        // y = clamp(y, -1.0, 1.0)
        AscendC::Clamp(yLocal, t2Local, -1.0f, 1.0f, len);

        tmpQueue1_.FreeTensor(t1Local);
        tmpQueue2_.FreeTensor(t2Local);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t m_ {0};
    int32_t k_ {0};
    int32_t n_ {0};
};

extern "C" __global__ __aicore__ void gemm_swish_divide_clamp_tanh_clamp_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSwishDivideClampTanhClampCustom op;
    op.Init(x, w, b, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.m, tiling_data.k, tiling_data.n);
    op.Process();
}
