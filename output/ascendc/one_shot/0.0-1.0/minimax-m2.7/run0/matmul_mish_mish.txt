namespace optiling {

static constexpr uint32_t BLOCK_DIM = 8;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMishMishCustomTilingData tiling;
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    uint32_t batchSize = static_cast<uint32_t>(x_shape->GetShapeSize()) / 
                         static_cast<uint32_t>(w_shape->GetShape().GetDim(1));
    uint32_t inFeatures = static_cast<uint32_t>(w_shape->GetShape().GetDim(1));
    uint32_t outFeatures = static_cast<uint32_t>(w_shape->GetShape().GetDim(0));

    uint32_t m = batchSize;
    uint32_t k = inFeatures;
    uint32_t n = outFeatures;

    // Tile sizes for matmul with activation
    uint32_t tileM = 128;
    uint32_t tileK = 256;
    uint32_t tileN = 128;

    if (m < tileM) tileM = AlignDown(m > 0 ? m : 1, ALIGN_ELEMS);
    if (k < tileK) tileK = AlignDown(k > 0 ? k : 1, ALIGN_ELEMS);
    if (n < tileN) tileN = AlignDown(n > 0 ? n : 1, ALIGN_ELEMS);

    tiling.set_batchSize(batchSize);
    tiling.set_inFeatures(inFeatures);
    tiling.set_outFeatures(outFeatures);
    tiling.set_m(m);
    tiling.set_k(k);
    tiling.set_n(n);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(BLOCK_DIM);

    context->SetBlockDim(BLOCK_DIM);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_SIZE = 128;

class KernelMatmulMishMishCustom {
public:
    __aicore__ inline KernelMatmulMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t m, uint32_t k, uint32_t n,
                                uint32_t tileM, uint32_t tileK, uint32_t tileN)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        m_ = m;
        k_ = k;
        n_ = n;
        tileM_ = tileM;
        tileK_ = tileK;
        tileN_ = tileN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, m * k);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, n * k);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, n);
        yGm_.SetGlobalBuffer((__gm__ float*)y, m * n);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileM * tileK * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileN * tileK * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, tileN * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileM * tileN * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileM * tileN * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileM * tileN * sizeof(float));
        pipe_.InitBuffer(tmpQueue3_, 1, tileM * tileN * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        
        for (uint32_t mStart = blockIdx * tileM_; mStart < m_; mStart += tileM_ * 8) {
            uint32_t mEnd = mStart + tileM_;
            if (mEnd > m_) mEnd = m_;
            uint32_t mLen = mEnd - mStart;

            for (uint32_t nStart = 0; nStart < n_; nStart += tileN_) {
                uint32_t nEnd = nStart + tileN_;
                if (nEnd > n_) nEnd = n_;
                uint32_t nLen = nEnd - nStart;

                MatmulTile(mStart, mLen, nStart, nLen);
            }
        }
    }

private:
    __aicore__ inline void MatmulTile(uint32_t mStart, uint32_t mLen, uint32_t nStart, uint32_t nLen)
    {
        AscendC::LocalTensor<float> accTensor = tmpQueue1_.AllocTensor<float>();
        AscendC::Sets(accTensor, 0.0f, mLen * nLen);

        for (uint32_t kStart = 0; kStart < k_; kStart += tileK_) {
            uint32_t kEnd = kStart + tileK_;
            if (kEnd > k_) kEnd = k_;
            uint32_t kLen = kEnd - kStart;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();

            for (uint32_t mi = 0; mi < mLen; ++mi) {
                uint32_t gmOffsetX = (mStart + mi) * k_ + kStart;
                AscendC::DataCopy(xLocal[mi * kLen], xGm_[gmOffsetX], kLen);
            }

            for (uint32_t ni = 0; ni < nLen; ++ni) {
                for (uint32_t ki = 0; ki < kLen; ++ki) {
                    uint32_t gmOffsetW = (nStart + ni) * k_ + kStart + ki;
                    wLocal[ni * kLen + ki] = wGm_[gmOffsetW];
                }
            }

            inQueueX_.EnQue(xLocal);
            inQueueW_.EnQue(wLocal);

            AscendC::LocalTensor<float> xQue = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> wQue = inQueueW_.DeQue<float>();
            AscendC::LocalTensor<float> prodLocal = tmpQueue2_.AllocTensor<float>();

            AscendC::Muls(prodLocal, wQue, xQue, mLen, wQue.GetShape(), xQue.GetShape(), mLen * nLen);

            AscendC::Add(accTensor, accTensor, prodLocal, mLen * nLen);

            tmpQueue2_.FreeTensor(prodLocal);
            inQueueX_.FreeTensor(xQue);
            inQueueW_.FreeTensor(wQue);
        }

        AscendC::LocalTensor<float> biasLocal = inQueueB_.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, bGm_[nStart], nLen);
        inQueueB_.EnQue(biasLocal);

        AscendC::LocalTensor<float> bQue = inQueueB_.DeQue<float>();
        AscendC::Adds(accTensor, accTensor, bQue, mLen * nLen);
        inQueueB_.FreeTensor(bQue);

        // First mish: mish(x) = x * tanh(softplus(x))
        AscendC::LocalTensor<float> softplusLocal = tmpQueue2_.AllocTensor<float>();
        AscendC::Softplus(softplusLocal, accTensor, mLen * nLen);

        AscendC::LocalTensor<float> tanhLocal = tmpQueue3_.AllocTensor<float>();
        AscendC::Tanh(tanhLocal, softplusLocal, mLen * nLen);

        AscendC::LocalTensor<float> mish1Local = tmpQueue1_.AllocTensor<float>();
        AscendC::Muls(mish1Local, accTensor, tanhLocal, mLen * nLen);

        tmpQueue2_.FreeTensor(softplusLocal);
        tmpQueue3_.FreeTensor(tanhLocal);

        // Second mish
        AscendC::LocalTensor<float> softplusLocal2 = tmpQueue2_.AllocTensor<float>();
        AscendC::Softplus(softplusLocal2, mish1Local, mLen * nLen);

        AscendC::LocalTensor<float> tanhLocal2 = tmpQueue3_.AllocTensor<float>();
        AscendC::Tanh(tanhLocal2, softplusLocal2, mLen * nLen);

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Muls(yLocal, mish1Local, tanhLocal2, mLen * nLen);

        tmpQueue1_.FreeTensor(accTensor);
        tmpQueue1_.FreeTensor(mish1Local);
        tmpQueue2_.FreeTensor(softplusLocal2);
        tmpQueue3_.FreeTensor(tanhLocal2);
        outQueueY_.EnQue(yLocal);

        for (uint32_t mi = 0; mi < mLen; ++mi) {
            uint32_t gmOffsetY = (mStart + mi) * n_ + nStart;
            AscendC::LocalTensor<float> yQue = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[gmOffsetY], yQue[mi * nLen], nLen);
            outQueueY_.FreeTensor(yQue);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue3_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t inFeatures_ {0};
    uint32_t outFeatures_ {0};
    uint32_t m_ {0};
    uint32_t k_ {0};
    uint32_t n_ {0};
    uint32_t tileM_ {TILE_SIZE};
    uint32_t tileK_ {256};
    uint32_t tileN_ {TILE_SIZE};
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMishMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.batchSize, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.m, tiling_data.k, tiling_data.n,
            tiling_data.tileM, tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
