                                uint32_t blockDim, uint32_t lineSize, int32_t dim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        lineSize_ = lineSize;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t numLines = (lineSize_ == 0) ? 0 : totalLen_ / lineSize_;
        if (numLines == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= numLines) return;

        // Compute start offset for this line.
        uint32_t lineStart = blockIdx * lineSize_;
        uint32_t offset = lineStart;
        uint32_t remaining = lineSize_;

        float cumProd = 1.0f;

        while (remaining > 0) {
            uint32_t curChunk = (remaining > tileLen_) ? tileLen_ : remaining;
            CopyIn(offset, curChunk);
            cumProd = Compute(curChunk, cumProd);
            CopyOut(offset, curChunk);
            offset += curChunk;
            remaining -= curChunk;
        }
    }
