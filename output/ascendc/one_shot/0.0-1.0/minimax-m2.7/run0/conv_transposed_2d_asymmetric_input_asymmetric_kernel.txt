kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 3;

class KernelConvTransposed2dAsymmetric {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetric() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR w,
                                int32_t ni, int32_t ci, int32_t hi, int32_t wi,
                                int32_t co, int32_t koH, int32_t koW,
                                int32_t ho, int32_t wo,
                                int32_t strideH, int32_t strideW,
                                int32_t paddingH, int32_t paddingW,
                                int32_t outputPaddingH, int32_t outputPaddingW,
                                int32_t groups, int32_t dilationH, int32_t dilationW,
                                int32_t coPerGroup, uint32_t tileLength)
    {
        ni_ = ni; ci_ = ci; hi_ = hi; wi_ = wi;
        co_ = co; koH_ = koH; koW_ = koW;
        ho_ = ho; wo_ = wo;
        strideH_ = strideH; strideW_ = strideW;
        paddingH_ = paddingH; paddingW_ = paddingW;
        outputPaddingH_ = outputPaddingH; outputPaddingW_ = outputPaddingW;
        groups_ = groups; dilationH_ = dilationH; dilationW_ = dilationW;
        coPerGroup_ = coPerGroup;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, ni_ * ci_ * hi_ * wi_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ni_ * co_ * ho_ * wo_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(co_ * coPerGroup_ * koH_ * koW_));

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, static_cast<uint32_t>(hi_ * wi_) * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, static_cast<uint32_t>(coPerGroup_ * koH_ * koW_) * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueY_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutput_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const int32_t cHoW = co_ * ho_ * wo_;
        const uint32_t totalTiles = (cHoW + tileLen_ - 1) / tileLen_;

        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            const int32_t tileStart = tileIdx * tileLen_;
            const int32_t curLen = (tileStart + tileLen_ > cHoW) ? (cHoW - tileStart) : tileLen_;

            ComputeTile(tileStart, curLen);
        }
    }

private:
    __aicore__ inline void ComputeTile(int32_t offset, int32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpY = tmpQueueY_.AllocTensor<float>();
        AscendC::Sets(tmpY, 0.0f, len);

        for (int32_t batchIdx = 0; batchIdx < ni_; ++batchIdx) {
            const int32_t batchOffsetBase = batchIdx * ci_ * hi_ * wi_;
            const int32_t yBatchOffset = batchIdx * co_ * ho_ * wo_;

            for (int32_t outC = 0; outC < co_; ++outC) {
                const int32_t groupIdx = outC / coPerGroup_;
                const int32_t outCInGroup = outC % coPerGroup_;
                const int32_t weightOffset = outCInGroup * coPerGroup_ * koH_ * koW_;

                for (int32_t inC = 0; inC < ci_; ++inC) {
                    const int32_t groupIdxIn = inC / (ci_ / groups_);
                    if (groupIdx != groupIdxIn) continue;

                    const int32_t inCInGroup = inC % (ci_ / groups_);
                    const int32_t wOffset = weightOffset + inCInGroup * koH_ * koW_;
                    const int32_t xOffset = batchOffsetBase + inC * hi_ * wi_;

                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[xOffset], static_cast<uint32_t>(hi_ * wi_));

                    AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
                    AscendC::DataCopy(wLocal, wGm_[wOffset], static_cast<uint32_t>(coPerGroup_ * koH_ * koW_));

                    for (int32_t oy = 0; oy < ho_; ++oy) {
                        for (int32_t ox = 0; ox < wo_; ++ox) {
                            const int32_t outPos = outC * ho_ * wo_ + oy * wo_ + ox;
                            const int32_t localIdx = outPos - offset;
                            if (localIdx < 0 || localIdx >= len) continue;

                            float sum = 0.0f;
                            for (int32_t ky = 0; ky < koH_; ++ky) {
                                for (int32_t kx = 0; kx < koW_; ++kx) {
                                    const int32_t iy = oy * strideH_ - paddingH_ + ky * dilationH_;
                                    const int32_t ix = ox * strideW_ - paddingW_ + kx * dilationW_;

                                    if (iy >= 0 && iy < hi_ && ix >= 0 && ix < wi_) {
                                        const int32_t xIdx = iy * wi_ + ix;
                                        const int32_t wIdx = ky * koW_ + kx;
                                        sum += xLocal[xIdx] * wLocal[wIdx];
                                    }
                                }
                            }

                            tmpY[localIdx] += sum;
                        }
                    }

                    inQueueX_.FreeTensor(xLocal);
                    inQueueW_.FreeTensor(wLocal);
                }
            }
        }

        AscendC::DataCopy(yLocal, tmpY, static_cast<uint32_t>(len));
        tmpQueueY_.FreeTensor(tmpY);
        outQueueY_.EnQue(yLocal);

        AscendC::LocalTensor<float> yGmLocal = outQueueY_.DeQue<float>();
        const int32_t yOffset = offset;
        AscendC::DataCopy(yGm_[yOffset], yGmLocal, static_cast<uint32_t>(len));
        outQueueY_.FreeTensor(yGmLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wGm_;

    int32_t ni_ {0}, ci_ {0}, hi_ {0}, wi_ {0};
    int32_t co_ {0}, koH_ {0}, koW_ {0};
    int32_t ho_ {0}, wo_ {0};
    int32_t strideH_ {1}, strideW_ {1};
    int32_t paddingH_ {0}, paddingW_ {0};
    int32_t outputPaddingH_ {0}, outputPaddingW_ {0};
    int32_t groups_ {1}, dilationH_ {1}, dilationW_ {1};
    int32_t coPerGroup_ {1};
    uint32_t tileLen_ {0};
    int32_t totalOutput_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR w, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvTransposed2dAsymmetric op;
    op.Init(x, y, w,
            tiling_data.ni, tiling_data.ci, tiling_data.hi, tiling_data.wi,
            tiling_data.co, tiling_data.koH, tiling_data.koW,
            tiling_data.ho, tiling_data.wo,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.paddingH, tiling_data.paddingW,
            tiling_data.outputPaddingH, tiling_data.outputPaddingW,
            tiling_data.groups, tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.coPerGroup, tiling_data.tileLength);
    op.Process();
}
