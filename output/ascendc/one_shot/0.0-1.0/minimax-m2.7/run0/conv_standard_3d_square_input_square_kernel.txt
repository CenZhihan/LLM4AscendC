#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t batch, uint32_t inChannel, uint32_t outChannel,
                                uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t padD, uint32_t padH, uint32_t padW,
                                uint32_t tileD, uint32_t tileH, uint32_t tileW,
                                uint32_t blockDim)
    {
        batch_ = batch; inChannel_ = inChannel; outChannel_ = outChannel;
        inD_ = inD; inH_ = inH; inW_ = inW;
        kD_ = kD; kH_ = kH; kW_ = kW;
        outD_ = outD; outH_ = outH; outW_ = outW;
        strideD_ = strideD; strideH_ = strideH; strideW_ = strideW;
        padD_ = padD; padH_ = padH; padW_ = padW;
        tileD_ = tileD; tileH_ = tileH; tileW_ = tileW;
        blockDim_ = blockDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * inChannel * inD * inH * inW);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outChannel * inChannel * kD * kH * kW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * outChannel * outD * outH * outW);

        inStrideD_ = inChannel * inH * inW;
        inStrideH_ = inW * inChannel;
        inStrideW_ = inChannel;
        inStrideC_ = 1;

        wStrideKD_ = inChannel * kH * kW;
        wStrideKH_ = kW * inChannel;
        wStrideKW_ = inChannel;
        wStrideC_ = 1;

        outStrideC_ = outD_ * outH_ * outW_;
        outStrideD_ = outH_ * outW_;
        outStrideH_ = outW_;
        outStrideW_ = 1;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileD * tileH * tileW * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, kD * kH * kW * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, 256 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outD_ == 0 || outH_ == 0 || outW_ == 0) return;

        const uint32_t totalBlocks = blockDim_;
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t totalOutputPoints = outD_ * outH_ * outW_;
        uint32_t pointsPerBlock = (totalOutputPoints + totalBlocks - 1) / totalBlocks;
        uint32_t startPoint = blockIdx * pointsPerBlock;
        uint32_t endPoint = startPoint + pointsPerBlock;
        if (endPoint > totalOutputPoints) endPoint = totalOutputPoints;

        for (uint32_t outIdx = startPoint; outIdx < endPoint; ++outIdx) {
            uint32_t od = outIdx / (outH_ * outW_);
            uint32_t rest = outIdx % (outH_ * outW_);
            uint32_t oh = rest / outW_;
            uint32_t ow = rest % outW_;

            for (uint32_t b = 0; b < batch_; ++b) {
                for (uint32_t oc = 0; oc < outChannel_; ++oc) {
                    float sum = 0.0f;
                    ComputeConv(b, oc, od, oh, ow, sum);
                    uint32_t outIdx_final = ((b * outChannel_ + oc) * outD_ + od) * outH_ * outW_ + (oh * outW_ + ow);
                    yGm_[outIdx_final] = sum;
                }
            }
        }
    }

private:
    __aicore__ inline void ComputeConv(uint32_t b, uint32_t oc, uint32_t od, uint32_t oh, uint32_t ow, float &sum)
    {
        for (uint32_t ic = 0; ic < inChannel_; ++ic) {
            for (uint32_t kd = 0; kd < kD_; ++kd) {
                int32_t id = od * strideD_ + kd - padD_;
                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    int32_t ih = oh * strideH_ + kh - padH_;
                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        int32_t iw = ow * strideW_ + kw - padW_;
                        if (id >= 0 && id < static_cast<int32_t>(inD_) &&
                            ih >= 0 && ih < static_cast<int32_t>(inH_) &&
                            iw >= 0 && iw < static_cast<int32_t>(inW_)) {
                            uint32_t xIdx = ((b * inChannel_ + ic) * inD_ + id) * inH_ * inW_ + (ih * inW_ + iw);
                            uint32_t wIdx = ((oc * inChannel_ + ic) * kD_ + kd) * kH_ * kW_ + (kh * kW_ + kw);
                            sum += xGm_[xIdx] * weightGm_[wIdx];
                        }
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_ {0};
    uint32_t inChannel_ {0};
    uint32_t outChannel_ {0};
    uint32_t inD_ {0};
    uint32_t inH_ {0};
    uint32_t inW_ {0};
    uint32_t kD_ {0};
    uint32_t kH_ {0};
    uint32_t kW_ {0};
    uint32_t outD_ {0};
    uint32_t outH_ {0};
    uint32_t outW_ {0};
    uint32_t strideD_ {1};
    uint32_t strideH_ {1};
    uint32_t strideW_ {1};
    uint32_t padD_ {0};
    uint32_t padH_ {0};
    uint32_t padW_ {0};
    uint32_t tileD_ {0};
    uint32_t tileH_ {0};
    uint32_t tileW_ {0};
    uint32_t blockDim_ {1};

    uint32_t inStrideD_ {0};
    uint32_t inStrideH_ {0};
    uint32_t inStrideW_ {0};
    uint32_t inStrideC_ {0};

    uint32_t wStrideKD_ {0};
    uint32_t wStrideKH_ {0};
    uint32_t wStrideKW_ {0};
    uint32_t wStrideC_ {0};

    uint32_t outStrideC_ {0};
    uint32_t outStrideD_ {0};
    uint32_t outStrideH_ {0};
    uint32_t outStrideW_ {0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
           tiling_data.batch, tiling_data.inChannel, tiling_data.outChannel,
           tiling_data.inD, tiling_data.inH, tiling_data.inW,
           tiling_data.kD, tiling_data.kH, tiling_data.kW,
           tiling_data.outD, tiling_data.outH, tiling_data.outW,
           tiling_data.strideD, tiling_data.strideH, tiling_data.strideW,
           tiling_data.padD, tiling_data.padH, tiling_data.padW,
           tiling_data.tileD, tiling_data.tileH, tiling_data.tileW,
           tiling_data.blockDim);
    op.Process();
}
