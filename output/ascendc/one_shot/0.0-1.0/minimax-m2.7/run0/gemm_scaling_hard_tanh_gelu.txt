// UB model for this kernel:
//   xQueue: tileLen * sizeof(float) * BUFFER_NUM
//   wQueue: tileLen * sizeof(float) * BUFFER_NUM
//   yQueue: tileLen * sizeof(float) * BUFFER_NUM
//   gemm uses xTile * kSize * sizeof(float) + wTile * kSize * sizeof(float) for matmul
//   plus additional workspace for intermediate results
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmScalingHardTanhGeluCustomTilingData tiling;
    
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    
    uint32_t mSize = static_cast<uint32_t>(x_shape->GetShapeSize());
    uint32_t kSize = static_cast<uint32_t>(w_shape->GetShape()[1]);
    uint32_t nSize = static_cast<uint32_t>(w_shape->GetShape()[0]);
    
    uint32_t totalLength = mSize;
    
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scaling = attrs->GetAttrPointer<float>(0);
    const float *hardtanhMin = attrs->GetAttrPointer<float>(1);
    const float *hardtanhMax = attrs->GetAttrPointer<float>(2);

    uint32_t tileLen = UB_BUDGET_BYTES / (3 * sizeof(float));
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen < ALIGN_ELEMS) tileLen = ALIGN_ELEMS;
    if (tileLen > HARD_CAP_ELEMS) tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;

    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.set_kSize(kSize);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_SIZE = 256;
constexpr float SQRT_2_PI = 0.7978845608f;

class KernelGemmScalingHardTanhGeluCustom {
public:
    __aicore__ inline KernelGemmScalingHardTanhGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t mSize, uint32_t kSize, uint32_t nSize,
                                float scaling, float hardtanhMin, float hardtanhMax)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        mSize_ = mSize;
        kSize_ = kSize;
        nSize_ = nSize;
        scaling_ = scaling;
        hardtanhMin_ = hardtanhMin;
        hardtanhMax_ = hardtanhMax;

        xGm_.SetGlobalBuffer((__gm__ float*)x, mSize * kSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, nSize * kSize);
        bGm_.SetGlobalBuffer((__gm__ float*)b, nSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, mSize * nSize);

        uint32_t maxTile = tileLen_;
        if (maxTile > mSize_) maxTile = mSize_;
        
        pipe_.InitBuffer(xQueueX_, BUFFER_NUM, maxTile * kSize_ * sizeof(float));
        pipe_.InitBuffer(xQueueW_, BUFFER_NUM, maxTile * kSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, maxTile * nSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, maxTile * nSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, maxTile * nSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue3_, 1, maxTile * nSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t mStride = blockNum * tileLen_;
        uint32_t mBase = blockIdx * tileLen_;
        if (mBase >= mSize_) return;

        uint32_t mRemain = mSize_ - mBase;
        uint32_t curMLen = (mRemain > tileLen_) ? tileLen_ : mRemain;

        uint32_t offsetX = mBase * kSize_;
        uint32_t offsetY = mBase * nSize_;

        CopyIn(xGm_, wGm_, offsetX, curMLen);
        Compute(curMLen);
        CopyOut(offsetY, curMLen);

        while (mRemain > curMLen) {
            mBase += mStride;
            if (mBase >= mSize_) break;
            
            offsetX = mBase * kSize_;
            offsetY = mBase * nSize_;
            mRemain = mSize_ - mBase;
            curMLen = (mRemain > tileLen_) ? tileLen_ : mRemain;

            CopyIn(xGm_, wGm_, offsetX, curMLen);
            Compute(curMLen);
            CopyOut(offsetY, curMLen);
        }
    }

private:
    __aicore__ inline void CopyIn(AscendC::GlobalTensor<float> &xGm, 
                                  AscendC::GlobalTensor<float> &wGm,
                                  uint32_t xOffset, uint32_t mLen)
    {
        AscendC::LocalTensor<float> xLocal = xQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> wLocal = xQueueW_.AllocTensor<float>();
        
        AscendC::DataCopy(xLocal, xGm[xOffset], mLen * kSize_);
        AscendC::DataCopy(wLocal, wGm[0], nSize_ * kSize_);
        
        xQueueX_.EnQue(xLocal);
        xQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void Compute(uint32_t mLen)
    {
        AscendC::LocalTensor<float> xLocal = xQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = xQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> t1Local = tmpQueue1_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2Local = tmpQueue2_.AllocTensor<float>();
        AscendC::LocalTensor<float> t3Local = tmpQueue3_.AllocTensor<float>();
        
        uint32_t ySize = mLen * nSize_;
        
        // GEMM: y = x @ w^T (y[m,n] = sum_k x[m,k] * w[n,k])
        AscendC::MatMul(yLocal, xLocal, wLocal, mLen, nSize_, kSize_, 0, 0, 0, 0, false, false);
        
        // Add bias: y = y + b
        AscendC::Adds(t1Local, yLocal, bGm_, ySize);
        
        // Scale: y = y * scaling
        AscendC::Muls(t2Local, t1Local, scaling_, ySize);
        
        // HardTanh: y = clamp(y, min, max)
        AscendC::Clamp(t3Local, t2Local, hardtanhMin_, hardtanhMax_, ySize);
        
        // GELU approximation: y * 0.5 * (1 + tanh(sqrt(2/pi) * (y + 0.044715 * y^3)))
        // Compute y^3
        AscendC::Muls(t1Local, t3Local, t3Local, ySize);
        AscendC::Muls(t2Local, t1Local, t3Local, ySize);
        
        // Compute 0.044715 * y^3
        AscendC::Muls(t1Local, t2Local, 0.044715f, ySize);
        
        // Compute y + 0.044715 * y^3
        AscendC::Adds(t2Local, t3Local, t1Local, ySize);
        
        // Compute sqrt(2/pi) * (y + 0.044715 * y^3)
        AscendC::Muls(t1Local, t2Local, SQRT_2_PI, ySize);
        
        // Compute tanh(sqrt(2/pi) * (y + 0.044715 * y^3))
        AscendC::Tanh(t2Local, t1Local, ySize);
        
        // Compute 1 + tanh(...)
        AscendC::Adds(t1Local, t2Local, 1.0f, ySize);
        
        // Compute 0.5 * (1 + tanh(...))
        AscendC::Muls(yLocal, t1Local, 0.5f, ySize);
        
        // Final: y * 0.5 * (1 + tanh(...))
        AscendC::Muls(yLocal, t3Local, yLocal, ySize);
        
        tmpQueue1_.FreeTensor(t1Local);
        tmpQueue2_.FreeTensor(t2Local);
        tmpQueue3_.FreeTensor(t3Local);
        outQueueY_.EnQue<float>(yLocal);
        xQueueX_.FreeTensor(xLocal);
        xQueueW_.FreeTensor(wLocal);
    }

    __aicore__ inline void CopyOut(uint32_t yOffset, uint32_t mLen)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[yOffset], yLocal, mLen * nSize_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue3_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t mSize_ {0};
    uint32_t kSize_ {0};
    uint32_t nSize_ {0};
    float scaling_ {1.0f};
    float hardtanhMin_ {-1.0f};
    float hardtanhMax_ {1.0f};
};

extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScalingHardTanhGeluCustom op;
    op.Init(x, w, b, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.mSize, tiling_data.kSize, tiling_data.nSize,
            tiling_data.scaling, tiling_data.hardtanhMin, tiling_data.hardtanhMax);
    op.Process();
}
