class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t dim,
                                uint32_t lineLength, uint32_t preDimProd,
                                uint32_t phase)
    {
        totalLength_ = totalLength;
        tileLength_ = tileLength;
        blockDim_ = blockDim;
        dim_ = dim;
        lineLength_ = lineLength;
        preDimProd_ = preDimProd;
        phase_ = phase;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength_);
        maskGm_.SetGlobalBuffer((__gm__ float*)mask, totalLength_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLength_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(inQueueMask_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLength_ == 0 || lineLength_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= preDimProd_) return;

        const uint32_t lineIdx = blockIdx;
        const uint32_t lineOffsetBase = lineIdx * lineLength_;
        const uint32_t tileCount = (lineLength_ + tileLength_ - 1) / tileLength_;

        float cum = 0.0f;
        for (uint32_t tileIdx = 0; tileIdx < tileCount; ++tileIdx) {
            uint32_t offset = lineOffsetBase + tileIdx * tileLength_;
            uint32_t len = (tileIdx == tileCount - 1) ? (lineLength_ - tileIdx * tileLength_) : tileLength_;
            // Load x
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[offset], len);
            inQueueX_.EnQue(xLocal);
            // Load mask
            AscendC::LocalTensor<float> maskLocal = inQueueMask_.AllocTensor<float>();
            AscendC::DataCopy(maskLocal, maskGm_[offset], len);
            inQueueMask_.EnQue(maskLocal);
            // Deque
            xLocal = inQueueX_.DeQue<float>();
            maskLocal = inQueueMask_.DeQue<float>();

            // Compute masked = x * mask
            AscendC::LocalTensor<float> maskedLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(maskedLocal, xLocal, maskLocal, len);

            // Free xLocal, maskLocal
            inQueueX_.FreeTensor(xLocal);
            inQueueMask_.FreeTensor(maskLocal);

            // Prefix sum (inclusive) in maskedLocal
            // maskedLocal[0] unchanged
            for (uint32_t i = 1; i < len; ++i) {
                maskedLocal[i] = maskedLocal[i] + maskedLocal[i-1];
            }

            // Capture tile sum (last element before adding offset)
            float tileSum = maskedLocal[len - 1];

            // Add cumulative offset
            AscendC::Adds(maskedLocal, maskedLocal, cum, len);

            // Copy result to output
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::DataCopy(yGm_[offset], maskedLocal, len);
            outQueueY_.EnQue(yLocal);
            // Free maskedLocal
            tmpQueue_.FreeTensor(maskedLocal);

            // Update cumulative
            cum += tileSum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueMask_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLength_ {0};
    uint32_t tileLength_ {0};
    uint32_t blockDim_ {0};
    uint32_t dim_ {0};
    uint32_t lineLength_ {0};
    uint32_t preDimProd_ {0};
    uint32_t phase_ {0};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.blockDim,
            tiling_data.dim,
            tiling_data.lineLength,
            tiling_data.preDimProd,
            tiling_data.phase);
    op.Process();
}
