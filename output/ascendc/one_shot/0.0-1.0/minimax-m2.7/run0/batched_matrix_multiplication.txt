host_operator_src = '''
#include "batched_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cstdint>
#include <vector>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    BatchedMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *shapeX = context->GetInputShape(0);
    const gert::Shape *shapeY = context->GetInputShape(1);

    const auto &shapeXVec = shapeX->GetShapeVector();
    const auto &shapeYVec = shapeY->GetShapeVector();

    uint32_t batchSize = static_cast<uint32_t>(shapeXVec[0]);
    uint32_t M = static_cast<uint32_t>(shapeXVec[1]);
    uint32_t K = static_cast<uint32_t>(shapeXVec[2]);
    uint32_t N = static_cast<uint32_t>(shapeYVec[2]);

    uint32_t totalLengthX = batchSize * M * K;
    uint32_t totalLengthY = batchSize * K * N;
    uint32_t totalLengthZ = batchSize * M * N;

    // tileLen based on UB budget: we need space for X, Y, Z tiles.
    // Total bytes = 3 * tileLen * sizeof(float)
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLengthZ < tileLen) {
        uint32_t aligned = AlignDown(totalLengthZ, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLengthZ;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLengthZ, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLengthX(totalLengthX);
    tiling.set_totalLengthY(totalLengthY);
    tiling.set_totalLengthZ(totalLengthZ);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_batchSize(batchSize);
    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *yShape = context->GetInputShape(1);
    gert::Shape *zShape = context->GetOutputShape(0);

    const auto &xVec = xShape->GetShapeVector();
    const auto &yVec = yShape->GetShapeVector();

    std::vector<int64_t> outVec = {xVec[0], xVec[1], yVec[2]};
    zShape->SetShapeVector(outVec);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

}  // namespace ge

namespace ops {

class BatchedMatrixMultiplicationCustom : public OpDef {
public:
    explicit BatchedMatrixMultiplicationCustom(const char *name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("z")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(BatchedMatrixMultiplicationCustom);

}  // namespace ops

kernel_src = '''
#include "kernel_operator.h"
#include "batched_matrix_multiplication_custom_tiling.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelBatchedMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelBatchedMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR z,
                                uint32_t totalLengthX, uint32_t totalLengthY, uint32_t totalLengthZ,
                                uint32_t tileLength,
                                uint32_t batchSize, uint32_t M, uint32_t K, uint32_t N)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLengthX);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLengthY);
        zGm_.SetGlobalBuffer((__gm__ float*)z, totalLengthZ);

        batchSize_ = batchSize;
        M_ = M;
        K_ = K;
        N_ = N;
        tileLen_ = tileLength;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueZ_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalOutputElems = batchSize_ * M_ * N_;
        const uint32_t stride = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutputElems) return;

        while (base < totalOutputElems) {
            uint32_t curTileLen = tileLen_;
            if (base + curTileLen > totalOutputElems) {
                curTileLen = totalOutputElems - base;
            }
            ComputeTile(base, curTileLen);
            base += stride;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t start, uint32_t len)
    {
        for (uint32_t offset = 0; offset < len; ++offset) {
            uint32_t outIdx = start + offset;
            uint32_t b = outIdx / (M_ * N_);
            uint32_t rem = outIdx % (M_ * N_);
            uint32_t i = rem / N_;
            uint32_t j = rem % N_;

            float sum = 0.0f;
            for (uint32_t k = 0; k < K_; ++k) {
                uint32_t aIdx = ((b * M_ + i) * K_) + k;
                uint32_t bIdx = ((b * K_ + k) * N_) + j;
                sum += xGm_[aIdx] * yGm_[bIdx];
            }
            zGm_[outIdx] = sum;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueZ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> zGm_;

    uint32_t batchSize_;
    uint32_t M_;
    uint32_t K_;
    uint32_t N_;
    uint32_t tileLen_;
};

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBatchedMatrixMultiplicationCustom op;
    op.Init(x, y, workspace,
            tiling_data.totalLengthX, tiling_data.totalLengthY, tiling_data.totalLengthZ,
            tiling_data.tileLength,
            tiling_data.batchSize, tiling_data.M, tiling_data.K, tiling_data.N);
    op.Process();
}
'''
