#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t TILE_H = 32;
constexpr int32_t TILE_W = 32;
constexpr int32_t ELEMENTS_PER_TILE = TILE_H * TILE_W;

class KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR weight, GM_ADDR y,
        uint32_t batchSize, uint32_t inputC, uint32_t outputC,
        uint32_t inputH, uint32_t inputW,
        uint32_t outputH, uint32_t outputW,
        uint32_t kernelH, uint32_t kernelW,
        uint32_t strideH, uint32_t strideW,
        uint32_t paddingH, uint32_t paddingW,
        uint32_t dilationH, uint32_t dilationW,
        uint32_t groups, uint32_t totalOutputElems,
        uint32_t outputTileH, uint32_t outputTileW)
    {
        batchSize_ = batchSize;
        inputC_ = inputC;
        outputC_ = outputC;
        inputH_ = inputH;
        inputW_ = inputW;
        outputH_ = outputH;
        outputW_ = outputW;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideH_ = strideH;
        strideW_ = strideW;
        paddingH_ = paddingH;
        paddingW_ = paddingW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;
        totalOutputElems_ = totalOutputElems;
        outputTileH_ = outputTileH;
        outputTileW_ = outputTileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        const uint32_t inputPlaneSize = inputH_ * inputW_;
        const uint32_t kernelPlaneSize = kernelH_ * kernelW_;
        const uint32_t filterSize = inputC_ * kernelH_ * kernelW_;
        const uint32_t inputBatchSize = inputC_ * inputH_ * inputW_;
        const uint32_t outputBatchSize = outputC_ * outputH_ * outputW_;

        xGm_.SetShape({batchSize_, inputC_, inputH_, inputW_});
        xGm_.SetStrides({inputBatchSize, inputPlaneSize, inputW_, 1});
        weightGm_.SetShape({outputC_, inputC_, kernelH_, kernelW_});
        weightGm_.SetStrides({filterSize, kernelPlaneSize, kernelW_, 1});
        yGm_.SetShape({batchSize_, outputC_, outputH_, outputW_});
        yGm_.SetStrides({outputBatchSize, outputH_ * outputW_, outputW_, 1});

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputC_ * kernelH_ * kernelW_ * TILE_H * TILE_W * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, outputC_ * inputC_ * kernelH_ * kernelW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, ELEMENTS_PER_TILE * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, ELEMENTS_PER_TILE * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t totalTiles = CeilDiv(totalOutputElems_, ELEMENTS_PER_TILE);
        uint32_t tilesPerBlock = CeilDiv(totalTiles, blockNum);

        uint32_t startTile = blockIdx * tilesPerBlock;
        if (startTile >= totalTiles) return;
        uint32_t endTile = startTile + tilesPerBlock;
        if (endTile > totalTiles) endTile = totalTiles;

        for (uint32_t tileId = startTile; tileId < endTile; ++tileId) {
            uint32_t outputIdx = tileId * ELEMENTS_PER_TILE;
            ComputeTile(outputIdx);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t outputStartIdx)
    {
        uint32_t outN = outputStartIdx / (outputC_ * outputH_ * outputW_);
        uint32_t remainder = outputStartIdx % (outputC_ * outputH_ * outputW_);
        uint32_t outC = remainder / (outputH_ * outputW_);
        uint32_t outIdx = remainder % (outputH_ * outputW_);
        uint32_t outH = outIdx / outputW_;
        uint32_t outW = outIdx % outputW_;

        uint32_t ihStart = outH * strideH_ - paddingH_;
        uint32_t iwStart = outW * strideW_ - paddingW_;

        LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Sets(yLocal, 0.0f, ELEMENTS_PER_TILE);

        for (uint32_t ic = 0; ic < inputC_; ++ic) {
            LocalTensor<float> wLocal = tmpQueue_.AllocTensor<float>();
            LoadWeight(wLocal, outC, ic);
            
            for (uint32_t h = 0; h < TILE_H && (outH + h) < outputH_; ++h) {
                for (uint32_t w = 0; w < TILE_W && (outW + w) < outputW_; ++w) {
                    uint32_t curH = outH + h;
                    uint32_t curW = outW + w;
                    uint32_t ihBase = curH * strideH_ - paddingH_;
                    uint32_t iwBase = curW * strideW_ - paddingW_;

                    float sum = 0.0f;
                    for (uint32_t kh = 0; kh < kernelH_; ++kh) {
                        int32_t ih = static_cast<int32_t>(ihBase) + kh * dilationH_;
                        if (ih < 0 || ih >= static_cast<int32_t>(inputH_)) continue;
                        for (uint32_t kw = 0; kw < kernelW_; ++kw) {
                            int32_t iw = static_cast<int32_t>(iwBase) + kw * dilationW_;
                            if (iw < 0 || iw >= static_cast<int32_t>(inputW_)) continue;
                            uint32_t inOffset = ((outN * inputC_ + ic) * inputH_ + ih) * inputW_ + iw;
                            float xVal = xGm_.GetValue(inOffset);
                            uint32_t wOffset = ((outC * inputC_ + ic) * kernelH_ + kh) * kernelW_ + kw;
                            float wVal = wLocal.GetValue(wOffset);
                            sum += xVal * wVal;
                        }
                    }
                    uint32_t yOffset = h * TILE_W + w;
                    yLocal.SetValue(yOffset, yLocal.GetValue(yOffset) + sum);
                }
            }
            
            tmpQueue_.FreeTensor(wLocal);
        }

        for (uint32_t h = 0; h < TILE_H && (outH + h) < outputH_; ++h) {
            for (uint32_t w = 0; w < TILE_W && (outW + w) < outputW_; ++w) {
                uint32_t yOffset = h * TILE_W + w;
                uint32_t outOffset = ((outN * outputC_ + outC) * outputH_ + (outH + h)) * outputW_ + (outW + w);
                yGm_.SetValue(outOffset, yLocal.GetValue(yOffset));
            }
        }

        outQueueY_.FreeTensor(yLocal);
    }

    __aicore__ inline void LoadWeight(LocalTensor<float>& wLocal, uint32_t outC, uint32_t inC)
    {
        uint32_t baseOffset = ((outC * inputC_) + inC) * kernelH_ * kernelW_;
        AscendC::DataCopy(wLocal, weightGm_[baseOffset], kernelH_ * kernelW_);
    }

    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b)
    {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_;
    uint32_t inputC_;
    uint32_t outputC_;
    uint32_t inputH_;
    uint32_t inputW_;
    uint32_t outputH_;
    uint32_t outputW_;
    uint32_t kernelH_;
    uint32_t kernelW_;
    uint32_t strideH_;
    uint32_t strideW_;
    uint32_t paddingH_;
    uint32_t paddingW_;
    uint32_t dilationH_;
    uint32_t dilationW_;
    uint32_t groups_;
    uint32_t totalOutputElems_;
    uint32_t outputTileH_;
    uint32_t outputTileW_;
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.batchSize, tiling_data.inputC, tiling_data.outputC,
            tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputH, tiling_data.outputW,
            tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.paddingH, tiling_data.paddingW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups, tiling_data.totalOutputElems,
            tiling_data.outputTileH, tiling_data.outputTileW);
    op.Process();
}
