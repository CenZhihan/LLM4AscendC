        // Loop over output indices assigned to this block
        for (uint32_t outIdx = outStart; outIdx < outEnd; ++outIdx) {
            // Compute indices for non-reduced dimensions
            uint32_t idx0, idx1, idx2;
            if (reductionDim_ == 0) {
                // output dims are dims_[1], dims_[2]; output index maps to idx1, idx2
                idx1 = outIdx / dims_[2];
                idx2 = outIdx % dims_[2];
                idx0 = 0; // not used
            } else if (reductionDim_ == 1) {
                idx0 = outIdx / dims_[2];
                idx2 = outIdx % dims_[2];
                idx1 = 0;
            } else if (reductionDim_ == 2) {
                idx0 = outIdx / dims_[1];
                idx1 = outIdx % dims_[1];
                idx2 = 0;
            } else {
                // default case
                idx0 = outIdx / dims_[2];
                idx2 = outIdx % dims_[2];
                idx1 = 0;
            }

            // Compute base offset for the slice (ignoring reduction dimension)
            uint32_t baseOffset = idx0 * stride0 + idx1 * stride1 + idx2 * stride2;

            // Compute max across reduction dimension
            float maxVal = -3.4e38f; // -INF
            // iterate over reduction dimension
            // We can loop without tiling
            for (uint32_t r = 0; r < redLength_; ++r) {
                uint32_t offset = baseOffset + r * (reductionDim_ == 0 ? stride0 : (reductionDim_ == 1 ? stride1 : stride2));
                float val = xGm_[offset];
                if (val > maxVal) {
                    maxVal = val;
                }
            }

            // Store result
            yGm_[outIdx] = maxVal;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLength_;
    uint32_t outputLength_;
    uint32_t redLength_;
    uint32_t blockDim_;
    uint32_t tileOut_;
    uint32_t dims_[3];
    int32_t reductionDim_;
};

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMaxReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                 uint32_t totalLength,
                                 uint32_t outputLength,
                                 uint32_t redLength,
                                 uint32_t blockDim,
                                 uint32_t tileOut,
                                 uint32_t dims[3],
                                 int32_t reductionDim)
    {
        totalLength_ = totalLength;
        outputLength_ = outputLength;
        redLength_ = redLength;
        blockDim_ = blockDim;
        tileOut_ = tileOut;
        dims_[0] = dims[0];
        dims_[1] = dims[1];
        dims_[2] = dims[2];
        reductionDim_ = reductionDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputLength_);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        if (blockIdx >= blockDim_) return;

        // Compute strides for each dimension
        uint32_t stride0 = dims_[1] * dims_[2];
        uint32_t stride1 = dims_[2];
        uint32_t stride2 = 1;

        // Determine output indices range for this block
        uint32_t outStart = blockIdx * tileOut_;
        uint32_t outEnd = outStart + tileOut_;
        if (outEnd > outputLength_) outEnd = outputLength_;

        // Loop over output indices assigned to this block
        for (uint32_t outIdx = outStart; outIdx < outEnd; ++outIdx) {
            // Compute indices for non-reduced dimensions
            uint32_t idx0 = 0, idx1 = 0, idx2 = 0;
            if (reductionDim_ == 0) {
                // output dims are dims_[1], dims_[2]; output index maps to idx1, idx2
                idx1 = outIdx / dims_[2];
                idx2 = outIdx % dims_[2];
                idx0 = 0;
            } else if (reductionDim_ == 1) {
                idx0 = outIdx / dims_[2];
                idx2 = outIdx % dims_[2];
                idx1 = 0;
            } else if (reductionDim_ == 2) {
                idx0 = outIdx / dims_[1];
                idx1 = outIdx % dims_[1];
                idx2 = 0;
            } else {
                // fallback: treat as dim=1
                idx0 = outIdx / dims_[2];
                idx2 = outIdx % dims_[2];
                idx1 = 0;
            }

            // Compute base offset for the slice (ignoring reduction dimension)
            uint32_t baseOffset = idx0 * stride0 + idx1 * stride1 + idx2 * stride2;

            // Compute max across reduction dimension
            float maxVal = -3.4e38f; // -INF
            // Determine stride for reduction dimension
            uint32_t redStride = (reductionDim_ == 0) ? stride0 : (reductionDim_ == 1) ? stride1 : stride2;
            for (uint32_t r = 0; r < redLength_; ++r) {
                uint32_t offset = baseOffset + r * redStride;
                float val = xGm_[offset];
                if (val > maxVal) {
                    maxVal = val;
                }

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMaxReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                 uint32_t totalLength,
                                 uint32_t outputLength,
                                 uint32_t redLength,
                                 uint32_t blockDim,
                                 uint32_t tileOut,
                                 uint32_t dims[3],
                                 int32_t reductionDim)
    {
        totalLength_ = totalLength;
        outputLength_ = outputLength;
        redLength_ = redLength;
        blockDim_ = blockDim;
        tileOut_ = tileOut;
        dims_[0] = dims[0];
        dims_[1] = dims[1];
        dims_[2] = dims[2];
        reductionDim_ = reductionDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLength_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outputLength_);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        if (blockIdx >= blockDim_) return;

        // Compute strides for each dimension
        uint32_t stride0 = dims_[1] * dims_[2];
        uint32_t stride1 = dims_[2];
        uint32_t stride2 = 1;

        // Determine output indices range for this block
        uint32_t outStart = blockIdx * tileOut_;
        uint32_t outEnd = outStart + tileOut_;
        if (outEnd > outputLength_) outEnd = outputLength_;

        // Loop over output indices assigned to this block
        for (uint32_t outIdx = outStart; outIdx < outEnd; ++outIdx) {
            // Compute indices for non-reduced dimensions
            uint32_t idx0 = 0, idx1 = 0, idx2 = 0;
            if (reductionDim_ == 0) {
                // output dims are dims_[1], dims_[2]; output index maps to idx1, idx2
                idx1 = outIdx / dims_[2];
                idx2 = outIdx % dims_[2];
                idx0 = 0;
            } else if (reductionDim_ == 1) {
                idx0 = outIdx / dims_[2];
                idx2 = outIdx % dims_[2];
                idx1 = 0;
            } else if (reductionDim_ == 2) {
                idx0 = outIdx / dims_[1];
                idx1 = outIdx % dims_[1];
                idx2 = 0;
            } else {
                // fallback: treat as dim=1
                idx0 = outIdx / dims_[2];
                idx2 = outIdx % dims_[2];
                idx1 = 0;
            }

            // Compute base offset for the slice (ignoring reduction dimension)
            uint32_t baseOffset = idx0 * stride0 + idx1 * stride1 + idx2 * stride2;

            // Compute max across reduction dimension
            float maxVal = -3.4e38f; // -INF
            // Determine stride for reduction dimension
            uint32_t redStride = (reductionDim_ == 0) ? stride0 : (reductionDim_ == 1) ? stride1 : stride2;
            for (uint32_t r = 0; r < redLength_; ++r) {
                uint32_t offset = baseOffset + r * redStride;
                float val = xGm_[offset];
                if (val > maxVal) {
                    maxVal = val;
                }
            }

            // Store result
            yGm_[outIdx] = maxVal;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t totalLength_;
    uint32_t outputLength_;
    uint32_t redLength_;
    uint32_t blockDim_;

    uint32_t dims_[3];

    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        if (blockIdx >= blockDim_) return;

        // Compute strides for each dimension
        uint32_t stride0 = dims_[1] * dims_[2];
        uint32_t stride1 = dims_[2];
        uint32_t stride2 = 1;

        uint32_t outStart = blockIdx * tileOut_;
        uint32_t outEnd = outStart + tileOut_;
        if (outEnd > outputLength_) outEnd = outputLength_;

        // Loop over output indices assigned to this block
        for (uint32_t outIdx = outStart; outIdx < outEnd; ++outIdx) {
            // Compute indices for non-reduced dimensions
            uint32_t idx0 = 0, idx1 = 0, idx2 = 0;
            if (reductionDim_ == 0) {

                idx2 = outIdx % dims_[2];

                idx2 = 0;

                idx2 = outIdx % dims_[2];
