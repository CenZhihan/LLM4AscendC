kernel_src = '''
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom {
public:
    __aicore__ inline KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batch, uint32_t inFeatures, uint32_t outFeatures,
                                float negativeSlope)
    {
        batch_ = batch;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        negativeSlope_ = negativeSlope;
        totalLen_ = batch * outFeatures;
        tileLen_ = outFeatures;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * inFeatures);
        wGm_.SetGlobalBuffer((__gm__ float*)w, outFeatures * inFeatures);
        bGm_.SetGlobalBuffer((__gm__ float*)b, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inFeatures * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        uint32_t batchIdx = blockIdx;
        if (batchIdx >= batch_) return;

        // Copy x row to local
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[batchIdx * inFeatures_], inFeatures_);
        inQueueX_.EnQue(xLocal);

        // Compute linear and reduction
        float sumExp = 0.0f;
        for (uint32_t j = 0; j < outFeatures_; ++j) {
            float sum = bGm_[j];
            for (uint32_t k = 0; k < inFeatures_; ++k) {
                sum += xLocal.GetValue(k) * wGm_[j * inFeatures_ + k];
            }
            sumExp += expf(sum);
        }

        // Release xLocal
        inQueueX_.FreeTensor(xLocal);

        float val = logf(sumExp);

        // LeakyReLU twice
        val = (val > 0.0f) ? val : negativeSlope_ * val;
        val = (val > 0.0f) ? val : negativeSlope_ * val;

        // GELU twice
        float sqrt2OverPi = 0.7978845608028654f;
        float kCoeff = 0.044715f;
        float tanhArg = sqrt2OverPi * (val + kCoeff * val * val * val);
        val = 0.5f * val * (1.0f + tanhf(tanhArg));
        tanhArg = sqrt2OverPi * (val + kCoeff * val * val * val);
        val = 0.5f * val * (1.0f + tanhf(tanhArg));

        // Write result
        yGm_[batchIdx] = val;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;
    uint32_t batch_;
    uint32_t inFeatures_;
    uint32_t outFeatures_;
    uint32_t totalLen_;
    uint32_t tileLen_;
    float negativeSlope_;
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.batch,
            tiling_data.inFeatures,
            tiling_data.outFeatures,
            tiling_data.negativeSlope);
    op.Process();
}
