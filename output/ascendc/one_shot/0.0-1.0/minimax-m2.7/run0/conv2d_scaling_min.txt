kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_KERNEL_SIZE = 7;

class KernelConv2dScalingMinCustom {
public:
    __aicore__ inline KernelConv2dScalingMinCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                               uint32_t batchSize, uint32_t inChannel, uint32_t outChannel,
                               uint32_t height, uint32_t width, uint32_t kernelSize,
                               uint32_t outHeight, uint32_t outWidth, uint32_t totalLength,
                               uint32_t tileLength, float scale, bool hasBias)
    {
        batchSize_ = batchSize;
        inChannel_ = inChannel;
        outChannel_ = outChannel;
        height_ = height;
        width_ = width;
        kernelSize_ = kernelSize;
        outHeight_ = outHeight;
        outWidth_ = outWidth;
        totalLength_ = totalLength;
        tileLength_ = tileLength;
        scale_ = scale;
        hasBias_ = hasBias;
        
        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * inChannel_ * height_ * width_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outChannel_ * inChannel_ * kernelSize_ * kernelSize_);
        
        if (hasBias_) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, outChannel_);
        }
        
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_ * outHeight_ * outWidth_);
        
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, outChannel_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue0_, 1, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLength_ == 0 || tileLength_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t outElementsPerBatch = outChannel_ * outHeight_ * outWidth_;
        const uint32_t strideElems = blockNum * tileLength_;
        uint32_t base = blockIdx * tileLength_;
        if (base >= totalLength_) return;

        uint32_t curLen = totalLength_ - base;
        if (curLen > tileLength_) curLen = tileLength_;
        
        uint32_t batchIdx = base / outElementsPerBatch;
        uint32_t elemIdxInBatch = base % outElementsPerBatch;
        uint32_t outChIdx = elemIdxInBatch / (outHeight_ * outWidth_);
        uint32_t spatialIdx = elemIdxInBatch % (outHeight_ * outWidth_);
        uint32_t outH = spatialIdx / outWidth_;
        uint32_t outW = spatialIdx % outWidth_;

        ComputeElement(batchIdx, outChIdx, outH, outW);
        CopyOut(base, curLen);
    }

private:
    __aicore__ inline void ComputeElement(uint32_t batchIdx, uint32_t outChIdx, uint32_t outH, uint32_t outW)
    {
        int32_t kHalf = static_cast<int32_t>(kernelSize_) / 2;
        
        float convSum = 0.0f;
        for (uint32_t inC = 0; inC < inChannel_; ++inC) {
            for (int32_t kh = -kHalf, kidx = 0; kh <= kHalf; ++kh, ++kidx) {
                for (int32_t kw = -kHalf; kw <= kHalf; ++kw) {
                    int32_t inH = static_cast<int32_t>(outH) + kh;
                    int32_t inW = static_cast<int32_t>(outW) + kw;
                    
                    if (inH >= 0 && inH < static_cast<int32_t>(height_) &&
                        inW >= 0 && inW < static_cast<int32_t>(width_)) {
                        uint32_t inOffset = batchIdx * inChannel_ * height_ * width_ +
                                           inC * height_ * width_ +
                                           inH * width_ + inW;
                        uint32_t wOffset = outChIdx * inChannel_ * kernelSize_ * kernelSize_ +
                                          inC * kernelSize_ * kernelSize_ +
                                          kidx * kernelSize_ + (kw + kHalf);
                        
                        convSum += xGm_.GetValue(inOffset) * weightGm_.GetValue(wOffset);
                    }
                }
            }
        }
        
        if (hasBias_) {
            convSum += biasGm_.GetValue(outChIdx);
        }
        
        float scaled = convSum * scale_;
        
        float minVal = scaled;
        for (uint32_t otherCh = 0; otherCh < outChannel_; ++otherCh) {
            float otherSum = ComputeConvForChannel(batchIdx, otherCh, outH, outW);
            if (hasBias_) {
                otherSum += biasGm_.GetValue(otherCh);
            }
            otherSum *= scale_;
            
            if (otherSum < minVal) {
                minVal = otherSum;
            }
        }
        
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        yLocal.SetValue({minVal});
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline float ComputeConvForChannel(uint32_t batchIdx, uint32_t outChIdx, uint32_t outH, uint32_t outW)
    {
        int32_t kHalf = static_cast<int32_t>(kernelSize_) / 2;
        float convSum = 0.0f;
        
        for (uint32_t inC = 0; inC < inChannel_; ++inC) {
            for (int32_t kh = -kHalf, kidx = 0; kh <= kHalf; ++kh, ++kidx) {
                for (int32_t kw = -kHalf; kw <= kHalf; ++kw) {
                    int32_t inH = static_cast<int32_t>(outH) + kh;
                    int32_t inW = static_cast<int32_t>(outW) + kw;
                    
                    if (inH >= 0 && inH < static_cast<int32_t>(height_) &&
                        inW >= 0 && inW < static_cast<int32_t>(width_)) {
                        uint32_t inOffset = batchIdx * inChannel_ * height_ * width_ +
                                           inC * height_ * width_ +
                                           inH * width_ + inW;
                        uint32_t wOffset = outChIdx * inChannel_ * kernelSize_ * kernelSize_ +
                                          inC * kernelSize_ * kernelSize_ +
                                          kidx * kernelSize_ + (kw + kHalf);
                        
                        convSum += xGm_.GetValue(inOffset) * weightGm_.GetValue(wOffset);
                    }
                }
            }
        }
        return convSum;
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue0_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t inChannel_ {0};
    uint32_t outChannel_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    uint32_t kernelSize_ {0};
    uint32_t outHeight_ {0};
    uint32_t outWidth_ {0};
    uint32_t totalLength_ {0};
    uint32_t tileLength_ {0};
    float scale_ {1.0f};
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dScalingMinCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.batchSize, tiling_data.inChannel, tiling_data.outChannel,
            tiling_data.height, tiling_data.width, tiling_data.kernelSize,
            tiling_data.outHeight, tiling_data.outWidth, tiling_data.totalLength,
            tiling_data.tileLength, tiling_data.scale, tiling_data.hasBias);
    op.Process();
}
