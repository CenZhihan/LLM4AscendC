#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t C0_SIZE = 16;
constexpr uint32_t VECTORIZED_WIDTH = 128;

class KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR weight, GM_ADDR y,
        uint32_t batchNum, uint32_t inputC, uint32_t inputH, uint32_t inputW,
        uint32_t outputC, uint32_t outputH, uint32_t outputW,
        uint32_t kernelH, uint32_t kernelW,
        uint32_t strideH, uint32_t strideW,
        uint32_t padH, uint32_t padW,
        uint32_t dilationH, uint32_t dilationW,
        uint32_t groups,
        uint32_t tileElems, uint32_t owPerBlock)
    {
        batchNum_ = batchNum;
        inputC_ = inputC;
        inputH_ = inputH;
        inputW_ = inputW;
        outputC_ = outputC;
        outputH_ = outputH;
        outputW_ = outputW;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;
        tileElems_ = tileElems;
        owPerBlock_ = owPerBlock;

        inputC1_ = inputC;
        outputC1_ = outputC;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        uint32_t inputPlaneSize = inputH_ * inputW_;
        uint32_t weightPlaneSize = kernelH_ * kernelW_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileElems_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, weightPlaneSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileElems_ * sizeof(float));
        pipe_.InitBuffer(ubufAcc_, 1, tileElems_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t totalBlocks = batchNum_ * outputH_ * ((outputW_ + owPerBlock_ - 1) / owPerBlock_);
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        if (blockIdx >= totalBlocks) return;

        uint32_t batchIdx = blockIdx / (outputH_ * ((outputW_ + owPerBlock_ - 1) / owPerBlock_));
        uint32_t rest = blockIdx % (outputH_ * ((outputW_ + owPerBlock_ - 1) / owPerBlock_));
        uint32_t ohStart = rest / ((outputW_ + owPerBlock_ - 1) / owPerBlock_);
        uint32_t owBlockIdx = rest % ((outputW_ + owPerBlock_ - 1) / owPerBlock_);

        for (uint32_t ocBase = 0; ocBase < outputC_; ++ocBase) {
            for (uint32_t icBase = 0; icBase < inputC_; ++icBase) {
                for (uint32_t oh = ohStart; oh < outputH_; oh += 1) {
                    ComputeRow(ocBase, icBase, oh, owBlockIdx, batchIdx);
                }
            }
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t ocBase, uint32_t icBase, uint32_t oh, uint32_t owBlockIdx, uint32_t batchIdx)
    {
        uint32_t owStart = owBlockIdx * owPerBlock_;
        uint32_t owEnd = owStart + owPerBlock_;
        if (owEnd > outputW_) owEnd = outputW_;
        if (owStart >= owEnd) return;

        uint32_t ihStart = oh * strideH_ - padH_;
        uint32_t ihEnd = ihStart + kernelH_ * dilationH_;

        AscendC::LocalTensor<float> accLocal = ubufAcc_.AllocTensor<float>();
        AscendC::Set<true>(accLocal, 0.0f, owEnd - owStart);

        for (uint32_t kh = 0; kh < kernelH_; ++kh) {
            int32_t ih = static_cast<int32_t>(ihStart) + static_cast<int32_t>(kh) * static_cast<int32_t>(dilationH_);
            if (ih < 0 || ih >= static_cast<int32_t>(inputH_)) continue;

            for (uint32_t kw = 0; kw < kernelW_; ++kw) {
                int32_t iw = static_cast<int32_t>(owStart) * static_cast<int32_t>(strideW_) - static_cast<int32_t>(padW_) +
                             static_cast<int32_t>(kw) * static_cast<int32_t>(dilationW_);
                if (iw < 0 || iw >= static_cast<int32_t>(inputW_)) continue;

                uint32_t xOffset = ((batchIdx * inputC_ + icBase) * inputH_ + ih) * inputW_ + iw;
                uint32_t wOffset = (ocBase * inputC_ + icBase) * kernelH_ * kernelW_ + kh * kernelW_ + kw;

                uint32_t curOwLen = owEnd - owStart;
                CopyIn(xOffset, wOffset, curOwLen, owStart);
                Compute(curOwLen, ocBase, icBase, kh, accLocal);
            }
        }

        CopyOut(ocBase, oh, owStart, owEnd - owStart, batchIdx, accLocal);
        ubufAcc_.FreeTensor(accLocal);
    }

    __aicore__ inline void CopyIn(uint32_t xOffset, uint32_t wOffset, uint32_t owLen, uint32_t owStart)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        uint32_t iwBase = owStart * strideW_ - padW_;
        for (uint32_t i = 0; i < owLen; ++i) {
            uint32_t iw = iwBase + i * strideW_;
            uint32_t srcIdx = xOffset + i * strideW_;
            if (iw < inputW_) {
                xLocal.SetValue(i, xGm_.GetValue(srcIdx));
            } else {
                xLocal.SetValue(i, 0.0f);
            }
        }
        inQueueX_.EnQue(xLocal);

        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, weightGm_[wOffset], 1);
        inQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void Compute(uint32_t len, uint32_t ocBase, uint32_t icBase, uint32_t kh, AscendC::LocalTensor<float>& accLocal)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        float wVal = wLocal.GetValue(0);
        AscendC::Muls(yLocal, xLocal, wVal, len);
        AscendC::Add(accLocal, accLocal, yLocal, len);

        outQueueY_.FreeTensor(yLocal);
        inQueueW_.FreeTensor(wLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t ocBase, uint32_t oh, uint32_t owStart, uint32_t owLen, uint32_t batchIdx, AscendC::LocalTensor<float>& accLocal)
    {
        uint32_t yOffset = ((batchIdx * outputC_ + ocBase) * outputH_ + oh) * outputW_ + owStart;
        AscendC::DataCopy(yGm_[yOffset], accLocal, owLen);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> ubufAcc_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchNum_;
    uint32_t inputC_, inputH_, inputW_;
    uint32_t outputC_, outputH_, outputW_;
    uint32_t kernelH_, kernelW_;
    uint32_t strideH_, strideW_;
    uint32_t padH_, padW_;
    uint32_t dilationH_, dilationW_;
    uint32_t groups_;
    uint32_t inputC1_, outputC1_;
    uint32_t tileElems_;
    uint32_t owPerBlock_;
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)

            tiling_data.batchNum, tiling_data.inputC, tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputC, tiling_data.outputH, tiling_data.outputW,

            tiling_data.strideH, tiling_data.strideW,
            tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.groups,
            tiling_data.tileElems, tiling_data.owPerBlock);
    op.Process();
}
