kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dMishMishCustom {
public:
    __aicore__ inline KernelConv2dMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t stride, int32_t padding, int32_t dilation,
                                int32_t kernelSize, int32_t inChannel, int32_t outChannel,
                                int32_t height, int32_t width, int32_t outHeight, int32_t outWidth,
                                bool hasBias)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        stride_ = stride;
        padding_ = padding;
        dilation_ = dilation;
        kernelSize_ = kernelSize;
        inChannel_ = inChannel;
        outChannel_ = outChannel;
        height_ = height;
        width_ = width;
        outHeight_ = outHeight;
        outWidth_ = outWidth;
        hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)bias);
        }

        const uint32_t convOutSize = kernelSize_ * kernelSize_ * inChannel_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, convOutSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
        if (hasBias_) {
            pipe_.InitBuffer(biasQueue_, 1, outChannel_ * sizeof(float));
        }
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2Local = tmpQueue2_.AllocTensor<float>();

        // Conv2d: im2col + gemm simplified for 3x3
        Conv2dDirect(yLocal, xLocal, len);

        // Add bias if present
        if (hasBias_) {
            AscendC::LocalTensor<float> biasLocal = biasQueue_.AllocTensor<float>();
            AscendC::DataCopy(biasLocal, bGm_[0], outChannel_);
            biasQueue_.EnQue(biasLocal);
            
            AscendC::LocalTensor<float> biasDeq = biasQueue_.DeQue<float>();
            AscendC::Adds(yLocal, yLocal, biasDeq, len);
            biasQueue_.FreeTensor(biasDeq);
        }

        // First Mish: mish(x) = x * tanh(softplus(x))
        // softplus(x) = log(1 + exp(x))
        AscendC::Exp(tLocal, xLocal, len);
        AscendC::Adds(tLocal, tLocal, 1.0f, len);
        AscendC::Ln(tLocal, tLocal, len);
        AscendC::Tanh(t2Local, tLocal, len);
        AscendC::Muls(yLocal, xLocal, t2Local, len);

        // Second Mish
        AscendC::Exp(tLocal, yLocal, len);
        AscendC::Adds(tLocal, tLocal, 1.0f, len);
        AscendC::Ln(tLocal, tLocal, len);
        AscendC::Tanh(t2Local, tLocal, len);
        AscendC::Muls(yLocal, yLocal, t2Local, len);

        tmpQueue_.FreeTensor(tLocal);
        tmpQueue2_.FreeTensor(t2Local);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void Conv2dDirect(AscendC::LocalTensor<float>& outTensor,
                                        AscendC::LocalTensor<float>& inTensor,
                                        uint32_t len)
    {
        // Simplified direct convolution for kernel 3x3, stride 1, padding 0, dilation 1
        // For each output element, compute sum of kernel * input region
        const int32_t outC = outChannel_;
        const int32_t outH = outHeight_;
        const int32_t outW = outWidth_;
        
        // Copy weight to local
        const uint32_t weightSize = kernelSize_ * kernelSize_ * inChannel_;
        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_[0], weightSize);
        wQueue_.EnQue(wLocal);
        AscendC::LocalTensor<float> wDeq = wQueue_.DeQue<float>();

        // Initialize output to zero
        AscendC::Sets(outTensor, 0.0f, len);

        // For each output channel
        for (int32_t oc = 0; oc < outC; ++oc) {
            for (int32_t oh = 0; oh < outH; ++oh) {
                for (int32_t ow = 0; ow < outW; ++ow) {
                    float sum = 0.0f;
                    int32_t outIdx = oc * outH * outW + oh * outW + ow;
                    
                    for (int32_t ic = 0; ic < inChannel_; ++ic) {
                        for (int32_t kh = 0; kh < kernelSize_; ++kh) {
                            for (int32_t kw = 0; kw < kernelSize_; ++kw) {
                                int32_t ih = oh * stride_ + kh * dilation_ - padding_;
                                int32_t iw = ow * stride_ + kw * dilation_ - padding_;
                                
                                if (ih >= 0 && ih < height_ && iw >= 0 && iw < width_) {
                                    int32_t inIdx = ic * height_ * width_ + ih * width_ + iw;
                                    int32_t wIdx = oc * inChannel_ * kernelSize_ * kernelSize_ +
                                                   ic * kernelSize_ * kernelSize_ +
                                                   kh * kernelSize_ + kw;
                                    sum += inTensor[inIdx] * wDeq[wIdx];
                                }
                            }
                        }
                    }
                    outTensor[outIdx] = sum;
                }
            }
        }

        wQueue_.FreeTensor(wDeq);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> biasQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t stride_ {1};
    int32_t padding_ {0};
    int32_t dilation_ {1};
    int32_t kernelSize_ {3};
    int32_t inChannel_ {1};
    int32_t outChannel_ {1};
    int32_t height_ {1};
    int32_t width_ {1};
    int32_t outHeight_ {1};
    int32_t outWidth_ {1};
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace,
                                                               GM_ADDR weight, GM_ADDR conv_bias,
                                                               GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMishMishCustom op;
    op.Init(x, y, weight, conv_bias,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation,
            tiling_data.kernelSize, tiling_data.inChannel, tiling_data.outChannel,
            tiling_data.height, tiling_data.width, tiling_data.outHeight, tiling_data.outWidth,
            tiling_data.hasBias);
    op.Process();
}
