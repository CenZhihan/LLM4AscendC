static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SqueezeNetFireModuleCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_squeeze_shape = context->GetInputShape(1);
    const gert::Shape *w_expand1_shape = context->GetInputShape(3);
    const gert::Shape *w_expand3_shape = context->GetInputShape(5);

    uint32_t inputBatch = x_shape->GetShape().at(0);
    uint32_t inputChannel = x_shape->GetShape().at(1);
    uint32_t inputHeight = x_shape->GetShape().at(2);
    uint32_t inputWidth = x_shape->GetShape().at(3);

    uint32_t squeezeOutChannel = w_squeeze_shape->GetShape().at(0);
    uint32_t expand1OutChannel = w_expand1_shape->GetShape().at(0);
    uint32_t expand3OutChannel = w_expand3_shape->GetShape().at(0);

    uint32_t outputHeight = inputHeight;
    uint32_t outputWidth = inputWidth;
    uint32_t totalOutputChannel = expand1OutChannel + expand3OutChannel;
    uint32_t totalLength = inputBatch * totalOutputChannel * outputHeight * outputWidth;

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_inputBatch(inputBatch);
    tiling.set_inputHeight(inputHeight);
    tiling.set_inputWidth(inputWidth);
    tiling.set_inputChannel(inputChannel);
    tiling.set_squeezeOutChannel(squeezeOutChannel);
    tiling.set_expand1OutChannel(expand1OutChannel);
    tiling.set_expand3OutChannel(expand3OutChannel);
    tiling.set_outputHeight(outputHeight);
    tiling.set_outputWidth(outputWidth);
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSqueezeNetFireModuleCustom {
public:
    __aicore__ inline KernelSqueezeNetFireModuleCustom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR w_squeeze, GM_ADDR b_squeeze,
        GM_ADDR w_expand1, GM_ADDR b_expand1,
        GM_ADDR w_expand3, GM_ADDR b_expand3, GM_ADDR y,
        uint32_t inputBatch, uint32_t inputHeight, uint32_t inputWidth,
        uint32_t inputChannel, uint32_t squeezeOutChannel,
        uint32_t expand1OutChannel, uint32_t expand3OutChannel,
        uint32_t outputHeight, uint32_t outputWidth, uint32_t totalLength)
    {
        inputBatch_ = inputBatch;
        inputHeight_ = inputHeight;
        inputWidth_ = inputWidth;
        inputChannel_ = inputChannel;
        squeezeOutChannel_ = squeezeOutChannel;
        expand1OutChannel_ = expand1OutChannel;
        expand3OutChannel_ = expand3OutChannel;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        totalLength_ = totalLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wSqueezeGm_.SetGlobalBuffer((__gm__ float*)w_squeeze);
        bSqueezeGm_.SetGlobalBuffer((__gm__ float*)b_squeeze);
        wExpand1Gm_.SetGlobalBuffer((__gm__ float*)w_expand1);
        bExpand1Gm_.SetGlobalBuffer((__gm__ float*)b_expand1);
        wExpand3Gm_.SetGlobalBuffer((__gm__ float*)w_expand3);
        bExpand3Gm_.SetGlobalBuffer((__gm__ float*)b_expand3);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        squeezeBufferSize_ = inputBatch_ * outputHeight_ * outputWidth_ * squeezeOutChannel_;
        expand1BufferSize_ = inputBatch_ * outputHeight_ * outputWidth_ * expand1OutChannel_;
        expand3BufferSize_ = inputBatch_ * outputHeight_ * outputWidth_ * expand3OutChannel_;
        maxBufferSize_ = squeezeBufferSize_;
        if (expand1BufferSize_ > maxBufferSize_) maxBufferSize_ = expand1BufferSize_;
        if (expand3BufferSize_ > maxBufferSize_) maxBufferSize_ = expand3BufferSize_;

        pipe_.InitBuffer(xQueueX_, BUFFER_NUM, maxBufferSize_ * sizeof(float));
        pipe_.InitBuffer(xQueueS_, BUFFER_NUM, maxBufferSize_ * sizeof(float));
        pipe_.InitBuffer(xQueueE1_, BUFFER_NUM, maxBufferSize_ * sizeof(float));
        pipe_.InitBuffer(xQueueE3_, BUFFER_NUM, maxBufferSize_ * sizeof(float));
        pipe_.InitBuffer(xQueueY_, BUFFER_NUM, maxBufferSize_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, maxBufferSize_ * sizeof(float));
        pipe_.InitBuffer(matmulTmp_, 1, maxBufferSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (uint32_t b = 0; b < inputBatch_; ++b) {
            ProcessSqueeze(b);
            ProcessExpand1x1(b);
            ProcessExpand3x3(b);
            ConcatenateOutputs(b);
        }
    }

private:
    __aicore__ inline void ProcessSqueeze(uint32_t batchIdx)
    {
        uint32_t inOffset = batchIdx * inputChannel_ * inputHeight_ * inputWidth_;
        uint32_t outOffset = batchIdx * squeezeOutChannel_ * outputHeight_ * outputWidth_;

        for (uint32_t oc = 0; oc < squeezeOutChannel_; ++oc) {
            for (uint32_t oh = 0; oh < outputHeight_; ++oh) {
                for (uint32_t ow = 0; ow < outputWidth_; ++ow) {
                    float sum = bSqueezeGm_[oc];

                    for (uint32_t ic = 0; ic < inputChannel_; ++ic) {
                        uint32_t inIdx = inOffset + ic * inputHeight_ * inputWidth_ + oh * inputWidth_ + ow;
                        uint32_t weightIdx = oc * inputChannel_ + ic;
                        sum += xGm_[inIdx] * wSqueezeGm_[weightIdx];
                    }

                    uint32_t outIdx = outOffset + oc * outputHeight_ * outputWidth_ + oh * outputWidth_ + ow;
                    float reluOut = sum > 0.0f ? sum : 0.0f;
                    tmpQueue_.AllocTensor<float>()[0] = reluOut;
                    AscendC::LocalTensor<float> outLocal = xQueueS_.AllocTensor<float>();
                    outLocal[0] = reluOut;
                    xQueueS_.EnQue(outLocal);
                }
            }
        }
    }

    __aicore__ inline void ProcessExpand1x1(uint32_t batchIdx)
    {
        uint32_t squeezeOffset = batchIdx * squeezeOutChannel_ * outputHeight_ * outputWidth_;
        uint32_t outOffset = batchIdx * expand1OutChannel_ * outputHeight_ * outputWidth_;

        for (uint32_t oc = 0; oc < expand1OutChannel_; ++oc) {
            for (uint32_t oh = 0; oh < outputHeight_; ++oh) {
                for (uint32_t ow = 0; ow < outputWidth_; ++ow) {
                    float sum = bExpand1Gm_[oc];

                    for (uint32_t ic = 0; ic < squeezeOutChannel_; ++ic) {
                        uint32_t inIdx = squeezeOffset + ic * outputHeight_ * outputWidth_ + oh * outputWidth_ + ow;
                        uint32_t weightIdx = oc * squeezeOutChannel_ + ic;
                        sum += xQueueS_.GetTensor()[inIdx] * wExpand1Gm_[weightIdx];
                    }

                    uint32_t outIdx = outOffset + oc * outputHeight_ * outputWidth_ + oh * outputWidth_ + ow;
                    float reluOut = sum > 0.0f ? sum : 0.0f;
                    AscendC::LocalTensor<float> outLocal = xQueueE1_.AllocTensor<float>();
                    outLocal[0] = reluOut;
                    xQueueE1_.EnQue(outLocal);
                }
            }
        }

        xQueueS_.FreeTensor(xQueueS_.GetTensor());
    }

    __aicore__ inline void ProcessExpand3x3(uint32_t batchIdx)
    {
        uint32_t inOffset = batchIdx * inputChannel_ * inputHeight_ * inputWidth_;
        uint32_t squeezeOffset = batchIdx * squeezeOutChannel_ * outputHeight_ * outputWidth_;
        uint32_t outOffset = batchIdx * expand3OutChannel_ * outputHeight_ * outputWidth_;

        for (uint32_t oc = 0; oc < expand3OutChannel_; ++oc) {
            for (uint32_t oh = 0; oh < outputHeight_; ++oh) {
                for (uint32_t ow = 0; ow < outputWidth_; ++ow) {
                    float sum = bExpand3Gm_[oc];

                    for (uint32_t ic = 0; ic < squeezeOutChannel_; ++ic) {
                        for (int32_t kh = -1; kh <= 1; ++kh) {
                            for (int32_t kw = -1; kw <= 1; ++kw) {
                                int32_t ih = static_cast<int32_t>(oh) + kh;
                                int32_t iw = static_cast<int32_t>(ow) + kw;

                                if (ih >= 0 && ih < static_cast<int32_t>(outputHeight_) &&
                                    iw >= 0 && iw < static_cast<int32_t>(outputWidth_)) {
                                    uint32_t inIdx = squeezeOffset + ic * outputHeight_ * outputWidth_ +
                                                     static_cast<uint32_t>(ih) * outputWidth_ + static_cast<uint32_t>(iw);
                                    uint32_t weightIdx = oc * squeezeOutChannel_ * 9 +
                                                         ic * 9 + (kh + 1) * 3 + (kw + 1);
                                    sum += xQueueS_.GetTensor()[inIdx] * wExpand3Gm_[weightIdx];
                                }
                            }
                        }
                    }

                    uint32_t outIdx = outOffset + oc * outputHeight_ * outputWidth_ + oh * outputWidth_ + ow;
                    float reluOut = sum > 0.0f ? sum : 0.0f;
                    AscendC::LocalTensor<float> outLocal = xQueueE3_.AllocTensor<float>();
                    outLocal[0] = reluOut;
                    xQueueE3_.EnQue(outLocal);
                }
            }
        }
    }

    __aicore__ inline void ConcatenateOutputs(uint32_t batchIdx)
    {
        uint32_t e1Offset = batchIdx * expand1OutChannel_ * outputHeight_ * outputWidth_;
        uint32_t e3Offset = batchIdx * expand3OutChannel_ * outputHeight_ * outputWidth_;
        uint32_t yOffset = batchIdx * (expand1OutChannel_ + expand3OutChannel_) * outputHeight_ * outputWidth_;

        uint32_t totalOutChannel = expand1OutChannel_ + expand3OutChannel_;

        for (uint32_t c = 0; c < expand1OutChannel_; ++c) {
            for (uint32_t h = 0; h < outputHeight_; ++h) {
                for (uint32_t w = 0; w < outputWidth_; ++w) {
                    uint32_t srcIdx = e1Offset + c * outputHeight_ * outputWidth_ + h * outputWidth_ + w;
                    uint32_t dstIdx = yOffset + c * outputHeight_ * outputWidth_ + h * outputWidth_ + w;
                    yGm_[dstIdx] = xQueueE1_.GetTensor()[srcIdx];
                }
            }
        }

        for (uint32_t c = 0; c < expand3OutChannel_; ++c) {
            for (uint32_t h = 0; h < outputHeight_; ++h) {
                for (uint32_t w = 0; w < outputWidth_; ++w) {
                    uint32_t srcIdx = e3Offset + c * outputHeight_ * outputWidth_ + h * outputWidth_ + w;
                    uint32_t dstIdx = yOffset + (expand1OutChannel_ + c) * outputHeight_ * outputWidth_ + h * outputWidth_ + w;
                    yGm_[dstIdx] = xQueueE3_.GetTensor()[srcIdx];
                }
            }
        }

        xQueueE1_.FreeTensor(xQueueE1_.GetTensor());
        xQueueE3_.FreeTensor(xQueueE3_.GetTensor());
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> xQueueS_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> xQueueE1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> xQueueE3_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> xQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> matmulTmp_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wSqueezeGm_;
    AscendC::GlobalTensor<float> bSqueezeGm_;
    AscendC::GlobalTensor<float> wExpand1Gm_;
    AscendC::GlobalTensor<float> bExpand1Gm_;
    AscendC::GlobalTensor<float> wExpand3Gm_;
    AscendC::GlobalTensor<float> bExpand3Gm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t inputBatch_;
    uint32_t inputHeight_;
    uint32_t inputWidth_;
    uint32_t inputChannel_;
    uint32_t squeezeOutChannel_;
    uint32_t expand1OutChannel_;
    uint32_t expand3OutChannel_;
    uint32_t outputHeight_;
    uint32_t outputWidth_;
    uint32_t totalLength_;
    uint32_t squeezeBufferSize_;
    uint32_t expand1BufferSize_;
    uint32_t expand3BufferSize_;
    uint32_t maxBufferSize_;
};

extern "C" __global__ __aicore__ void squeeze_net_fire_module_custom(
    GM_ADDR x, GM_ADDR w_squeeze, GM_ADDR b_squeeze,
    GM_ADDR w_expand1, GM_ADDR b_expand1,
    GM_ADDR w_expand3, GM_ADDR b_expand3,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSqueezeNetFireModuleCustom op;
    op.Init(x, w_squeeze, b_squeeze, w_expand1, b_expand1, w_expand3, b_expand3, y,
            tiling_data.inputBatch, tiling_data.inputHeight, tiling_data.inputWidth,
            tiling_data.inputChannel, tiling_data.squeezeOutChannel,
            tiling_data.expand1OutChannel, tiling_data.expand3OutChannel,
            tiling_data.outputHeight, tiling_data.outputWidth, tiling_data.totalLength);
    op.Process();
}
