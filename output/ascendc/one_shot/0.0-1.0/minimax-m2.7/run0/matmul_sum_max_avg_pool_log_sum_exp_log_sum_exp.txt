kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom {
public:
    __aicore__ inline KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom() {}

    __aicore__ inline void Init(uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t tileLength, float negativeSlope)
    {
        batchSize_ = batchSize;
        K_ = inFeatures;
        N_ = outFeatures;
        tileLen_ = tileLength;
        negativeSlope_ = negativeSlope;
        negCorr_ = negativeSlope_ - 1.0f;
        numTiles_ = (N_ + tileLen_ - 1) / tileLen_;
    }

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y)
    {
        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * K_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, N_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueMax_, 1, N_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueExp_, 1, N_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batchSize_ == 0 || K_ == 0 || N_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= batchSize_) return;

        uint32_t batchIdx = blockIdx;
        float out = 0.0f;

        // Load first tile of x
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[batchIdx * K_], K_);
        inQueueX_.EnQue(xLocal);

        // Load first tile of w
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, tileLen_ * K_);
        inQueueW_.EnQue(wLocal);

        for (uint32_t i = 0; i < numTiles_; ++i) {
            if (i + 1 < numTiles_) {
                AscendC::LocalTensor<float> xNext = inQueueX_.AllocTensor<float>();
                AscendC::LocalTensor<float> wNext = inQueueW_.AllocTensor<float>();
                AscendC::DataCopy(xNext, xGm_[batchIdx * K_], K_);
                AscendC::DataCopy(wNext, wGm_[(i + 1) * tileLen_ * K_], tileLen_ * K_);
                inQueueX_.EnQue(xNext);
                inQueueW_.EnQue(wNext);
            }

            AscendC::LocalTensor<float> xCur = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> wCur = inQueueW_.DeQue<float>();

            uint32_t actualTileLen = (i == numTiles_ - 1) ? (N_ - i * tileLen_) : tileLen_;

            for (uint32_t j = 0; j < actualTileLen; ++j) {
                float acc = 0.0f;
                for (uint32_t k = 0; k < K_; ++k) {
                    acc += xCur[k] * wCur[j * K_ + k];
                }
                out += acc + bGm_[batchIdx * N_ + i * tileLen_ + j];
            }

            inQueueX_.FreeTensor(xCur);
            inQueueW_.FreeTensor(wCur);
        }

        // Reduction phase
        AscendC::LocalTensor<float> maxLocal = tmpQueueMax_.AllocTensor<float>();
        AscendC::LocalTensor<float> expLocal = tmpQueueExp_.AllocTensor<float>();
        AscendC::Sets(maxLocal, -3.402823466e+38f, N_);
        maxLocal[0] = out;
        AscendC::ReduceMax(maxLocal, maxLocal, 0, 1, N_, true);
        float maxVal = maxLocal[0];
        tmpQueueMax_.FreeTensor(maxLocal);

        float expSum = expf(out - maxVal);
        expLocal[0] = expSum;
        AscendC::ReduceSum(expLocal, expLocal, 0, 1, N_, true);
        expSum = expLocal[0];
        tmpQueueExp_.FreeTensor(expLocal);

        float logsumexp1 = logf(expSum) + maxVal;
        float yFinal = logsumexp1;
        yGm_[batchIdx] = yFinal;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueExp_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileLen_ {512};
    uint32_t numTiles_ {1};
    float negativeSlope_ {0.01f};
    float negCorr_ {-0.99f};
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom op;
    op.Init(tiling_data.batchSize, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.tileLength, tiling_data.negativeSlope);
    op.Init(x, w, b, y);
    op.Process();
}
