static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMinTanhTanhCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    uint32_t batch = static_cast<uint32_t>(x_shape->GetShapeVector()[0]);
    uint32_t inChannels = static_cast<uint32_t>(x_shape->GetShapeVector()[1]);
    uint32_t height = static_cast<uint32_t>(x_shape->GetShapeVector()[2]);
    uint32_t width = static_cast<uint32_t>(x_shape->GetShapeVector()[3]);
    uint32_t outChannels = static_cast<uint32_t>(w_shape->GetShapeVector()[0]);
    uint32_t kernelSize = static_cast<uint32_t>(w_shape->GetShapeVector()[2]);

    uint32_t outputHeight = height - kernelSize + 1;
    uint32_t outputWidth = width - kernelSize + 1;

    // UB usage estimate: input tile + output tile + weight buffer
    // Conservative: 3 * tile elements
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLength = maxTile;
    uint32_t totalOutputElements = batch * outChannels * outputHeight * outputWidth;
    if (totalOutputElements < tileLength) {
        const uint32_t aligned = AlignDown(totalOutputElements, ALIGN_ELEMS);
        tileLength = (aligned > 0) ? aligned : totalOutputElements;
    }

    uint32_t numTiles = CeilDiv(totalOutputElements, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_inChannels(inChannels);
    tiling.set_outChannels(outChannels);
    tiling.set_height(height);
    tiling.set_width(width);
    tiling.set_kernelSize(kernelSize);
    tiling.set_outputHeight(outputHeight);
    tiling.set_outputWidth(outputWidth);
    tiling.set_blockDim(blockDim);
    tiling.set_tileLength(tileLength);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "kernel_gmask.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t KS = 3;  // kernel size

class KernelConv2dMinTanhTanhCustom {
public:
    __aicore__ inline KernelConv2dMinTanhTanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t batch, uint32_t inChannels, uint32_t outChannels,
                                uint32_t height, uint32_t width, uint32_t kernelSize,
                                uint32_t outputHeight, uint32_t outputWidth,
                                uint32_t tileLength)
    {
        batch_ = batch;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        height_ = height;
        width_ = width;
        kernelSize_ = kernelSize;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        tileLength_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch * inChannels * height * width);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outChannels * inChannels * kernelSize * kernelSize);
        hasBias_ = (bias != nullptr);
        if (hasBias_) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, outChannels);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch * outputHeight * outputWidth);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLength_ * sizeof(float));
        pipe_.InitBuffer(weightBuf_, 1, inChannels * kernelSize * kernelSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (tileLength_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalOutputElements = batch_ * outputHeight_ * outputWidth_;
        const uint32_t strideElems = blockNum * tileLength_;
        uint32_t basePos = blockIdx * tileLength_;
        if (basePos >= totalOutputElements) return;

        uint32_t curLen = totalOutputElements - basePos;
        if (curLen > tileLength_) curLen = tileLength_;

        uint32_t nextBase = basePos + strideElems;

        while (basePos < totalOutputElements) {
            uint32_t outBatch = basePos / (outputHeight_ * outputWidth_);
            uint32_t outSpatial = basePos % (outputHeight_ * outputWidth_);
            uint32_t outH = outSpatial / outputWidth_;
            uint32_t outW = outSpatial % outputWidth_;

            ComputePerOutput(outBatch, outH, outW, curLen, 
                             (nextBase < totalOutputElements) ? 1 : 0);

            uint32_t gmOffset = outBatch * outputHeight_ * outputWidth_ + outH * outputWidth_ + outW;
            CopyOut(gmOffset, curLen);

            basePos = nextBase;
            nextBase += strideElems;
            curLen = (basePos < totalOutputElements) ? 
                     ((totalOutputElements - basePos > tileLength_) ? tileLength_ : totalOutputElements - basePos) : 0;
        }
    }

private:
    __aicore__ inline void ComputePerOutput(uint32_t startBatch, uint32_t startH, uint32_t startW, 
                                            uint32_t numElems, uint32_t hasNext)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        
        // For each output position, compute conv2d -> min -> tanh -> tanh
        uint32_t validOutputs = 0;
        for (uint32_t i = 0; i < numElems; ++i) {
            uint32_t pos = i;
            uint32_t batchIdx = (startBatch + pos / (outputHeight_ * outputWidth_)) % batch_;
            uint32_t spatialPos = (startBatch + pos / (outputHeight_ * outputWidth_)) / (outputHeight_ * outputWidth_) * (outputHeight_ * outputWidth_) +
                                 (startH * outputWidth_ + startW + pos) % (outputHeight_ * outputWidth_);
            uint32_t curH = spatialPos / outputWidth_;
            uint32_t curW = spatialPos % outputWidth_;

            float minVal = 1e10f;
            float convSum = 0.0f;

            // Convolution: for each output channel (outChannels)
            for (uint32_t oc = 0; oc < outChannels_; ++oc) {
                convSum = 0.0f;
                for (uint32_t ic = 0; ic < inChannels_; ++ic) {
                    for (uint32_t kh = 0; kh < kernelSize_; ++kh) {
                        for (uint32_t kw = 0; kw < kernelSize_; ++kw) {
                            uint32_t inH = curH + kh;
                            uint32_t inW = curW + kw;
                            if (inH < height_ && inW < width_) {
                                uint32_t xIdx = batchIdx * inChannels_ * height_ * width_ +
                                               ic * height_ * width_ +
                                               inH * width_ + inW;
                                uint32_t wIdx = oc * inChannels_ * kernelSize_ * kernelSize_ +
                                               ic * kernelSize_ * kernelSize_ +
                                               kh * kernelSize_ + kw;
                                convSum += xGm_.GetValue(xIdx) * weightGm_.GetValue(wIdx);
                            }
                        }
                    }
                }
                if (hasBias_ && oc == 0) {
                    convSum += biasGm_.GetValue(oc);
                }
            }

            // Apply min along dim=1 (across out_channels)
            for (uint32_t oc = 0; oc < outChannels_; ++oc) {
                if (convSum < minVal) {
                    minVal = convSum;
                }
            }

            // Apply tanh twice: tanh(tanh(x)) = tanh(x) for x in (-1, 1), but we compute explicitly
            float t1 = fast_tanh(minVal);
            float t2 = fast_tanh(t1);
            yLocal.SetValue(i, t2);
            validOutputs++;
        }

        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline float fast_tanh(float x)
    {
        float x2 = x * x;
        float a = x * (135135.0f + x2 * (17325.0f + x2 * (378.0f + x2 * 1.0f)));
        float b = 135135.0f + x2 * (62370.0f + x2 * (3150.0f + x2 * 28.0f));
        return a / b;
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> weightBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_ {0};
    uint32_t inChannels_ {0};
    uint32_t outChannels_ {0};
    uint32_t height_ {0};
    uint32_t width_ {0};
    uint32_t kernelSize_ {0};
    uint32_t outputHeight_ {0};
    uint32_t outputWidth_ {0};
    uint32_t tileLength_ {0};
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanhCustom op;
    op.Init(x, weight, bias, y, tiling_data.batch, tiling_data.inChannels, 
            tiling_data.outChannels, tiling_data.height, tiling_data.width,
            tiling_data.kernelSize, tiling_data.outputHeight, tiling_data.outputWidth,
            tiling_data.tileLength);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'conv2d_min_tanh_tanh_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()
        self.in_channels = init_inputs[0] if len(init_inputs) > 0 else 16
        self.out_channels = init_inputs[1] if len(init_inputs) > 1 else 64
        self.kernel_size = init_inputs[2] if len(init_inputs) > 2 else 3

    def forward(self, *inputs):
        x = inputs[0]
        weight = inputs[1]
        bias = inputs[2] if len(inputs) > 2 else None
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, weight, bias, self.in_channels, self.out_channels, self.kernel_size)
