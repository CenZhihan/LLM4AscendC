namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dSubtractSubtractMishCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *weightShape = context->GetInputShape(1);
    const gert::Shape *biasShape = (context->GetInputShapeCount() > 2) ? context->GetInputShape(2) : nullptr;

    uint32_t N = xShape->GetShapeVector()[0];
    uint32_t C = xShape->GetShapeVector()[1];
    uint32_t H = xShape->GetShapeVector()[2];
    uint32_t W = xShape->GetShapeVector()[3];

    uint32_t OC = weightShape->GetShapeVector()[0];
    uint32_t IC = weightShape->GetShapeVector()[1];
    uint32_t KH = weightShape->GetShapeVector()[2];
    uint32_t KW = weightShape->GetShapeVector()[3];

    uint32_t outputH = H - KH + 1;
    uint32_t outputW = W - KW + 1;

    uint32_t totalLength = N * OC * outputH * outputW;

    uint8_t hasBias = (biasShape != nullptr && biasShape->GetShapeVector().size() > 0) ? 1 : 0;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *subtractValue1 = attrs->GetAttrPointer<float>(0);
    const float *subtractValue2 = attrs->GetAttrPointer<float>(1);

    uint32_t weightLen = OC * IC * KH * KW;
    uint32_t biasLen = hasBias ? OC : 0;

    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t BYTES_PER_ELEM = 4;
    static constexpr uint32_t TEMP_BUFFERS_PER_TILE = 4;

    uint32_t maxTileByBudget = (UB_BUDGET_BYTES - (weightLen + biasLen) * BYTES_PER_ELEM) / (TEMP_BUFFERS_PER_TILE * BYTES_PER_ELEM);
    static constexpr uint32_t ALIGN_ELEMS = 256;
    uint32_t tileLength = AlignDown(maxTileByBudget, ALIGN_ELEMS);
    if (tileLength < ALIGN_ELEMS) tileLength = ALIGN_ELEMS;
    static constexpr uint32_t HARD_CAP_ELEMS = 32768;
    if (tileLength > HARD_CAP_ELEMS) tileLength = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    const uint32_t numTiles = CeilDiv(totalLength, tileLength);
    uint32_t blockDim = numTiles;
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 64) blockDim = 64;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLength);
    tiling.set_blockDim(blockDim);
    tiling.set_inputN(N);
    tiling.set_inputC(C);
    tiling.set_inputH(H);
    tiling.set_inputW(W);
    tiling.set_weightOC(OC);
    tiling.set_weightIC(IC);
    tiling.set_weightKH(KH);
    tiling.set_weightKW(KW);
    tiling.set_outputH(outputH);
    tiling.set_outputW(outputW);
    tiling.set_subtractValue1(*subtractValue1);
    tiling.set_subtractValue2(*subtractValue2);
    tiling.set_hasBias(hasBias);

kernel_src = """
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dSubtractSubtractMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractSubtractMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inputN, uint32_t inputC, uint32_t inputH, uint32_t inputW,
                                uint32_t weightOC, uint32_t weightIC, uint32_t weightKH, uint32_t weightKW,
                                uint32_t outputH, uint32_t outputW,
                                float subtractValue1, float subtractValue2, bool hasBias)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        inputN_ = inputN;
        inputC_ = inputC;
        inputH_ = inputH;
        inputW_ = inputW;
        weightOC_ = weightOC;
        weightIC_ = weightIC;
        weightKH_ = weightKH;
        weightKW_ = weightKW;
        outputH_ = outputH;
        outputW_ = outputW;
        subtractValue1_ = subtractValue1;
        subtractValue2_ = subtractValue2;
        hasBias_ = hasBias;

        weightLen_ = weightOC_ * weightIC_ * weightKH_ * weightKW_;
        biasLen_ = hasBias_ ? weightOC_ : 0;

        xGm_.SetGlobalBuffer((__gm__ float*)x, inputN_ * inputC_ * inputH_ * inputW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, weightLen_);
        if (hasBias_) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, biasLen_);
        }

        pipe_.InitBuffer(weightQueue_, 1, weightLen_ * sizeof(float));
        if (hasBias_) {
            pipe_.InitBuffer(biasQueue_, 1, biasLen_ * sizeof(float));
        }
        pipe_.InitBuffer(tmpQueue_, 3, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        // Copy weight and bias to local memory once
        AscendC::LocalTensor<float> weightBuf = weightQueue_.AllocTensor<float>();
        AscendC::DataCopy(weightBuf, weightGm_, weightLen_);

        AscendC::LocalTensor<float> biasBuf;
        if (hasBias_) {
            biasBuf = biasQueue_.AllocTensor<float>();
            AscendC::DataCopy(biasBuf, biasGm_, biasLen_);
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t base = blockIdx * tileLen_; base < totalLen_; base += blockNum * tileLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            // Allocate sum buffer
            AscendC::LocalTensor<float> sumBuf = tmpQueue_.AllocTensor<float>();

            // Compute convolution sums for each output element in the tile
            for (uint32_t i = 0; i < curLen; ++i) {
                uint32_t idx = base + i;
                uint32_t n = idx / (weightOC_ * outputH_ * outputW_);
                uint32_t idxc = idx % (weightOC_ * outputH_ * outputW_);
                uint32_t oc = idxc / (outputH_ * outputW_);
                uint32_t idxh = idxc % (outputH_ * outputW_);
                uint32_t oh = idxh / outputW_;
                uint32_t ow = idxh % outputW_;

                float sum = 0.0f;
                for (uint32_t ic = 0; ic < inputC_; ++ic) {
                    for (uint32_t kh = 0; kh < weightKH_; ++kh) {
                        for (uint32_t kw = 0; kw < weightKW_; ++kw) {
                            uint32_t input_h = oh + kh;
                            uint32_t input_w = ow + kw;
                            uint32_t input_offset = ((n * inputC_ + ic) * inputH_ + input_h) * inputW_ + input_w;
                            uint32_t weight_offset = ((oc * inputC_ + ic) * weightKH_ + kh) * weightKW_ + kw;
                            sum += xGm_[input_offset] * weightBuf[weight_offset];
                        }
                    }
                }
                if (hasBias_) {
                    sum += biasBuf[oc];
                }
                sum -= subtractValue1_ + subtractValue2_;
                sumBuf.SetValue(i, sum);
            }

            // Compute Mish activation: y = x * tanh(softplus(x))
            // exp = exp(x)
            AscendC::LocalTensor<float> expBuf = tmpQueue_.AllocTensor<float>();
            AscendC::Exp(expBuf, sumBuf, curLen);
            // softplus = log(1 + exp)
            AscendC::Adds(expBuf, expBuf, 1.0f, curLen);
            AscendC::Ln(expBuf, expBuf, curLen);
            // tanh = tanh(softplus)
            AscendC::LocalTensor<float> tanhBuf = tmpQueue_.AllocTensor<float>();
            AscendC::Tanh(tanhBuf, expBuf, curLen);
            tmpQueue_.FreeTensor(expBuf);

            // y = x * tanh
            AscendC::LocalTensor<float> yBuf = outQueueY_.AllocTensor<float>();
            AscendC::Mul(yBuf, sumBuf, tanhBuf, curLen);
            tmpQueue_.FreeTensor(sumBuf);
            tmpQueue_.FreeTensor(tanhBuf);

            outQueueY_.EnQue(yBuf);

            // Copy output to global memory
            AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yLocal, curLen);
            outQueueY_.FreeTensor(yLocal);
        }

        // Free weight and bias buffers
        weightQueue_.FreeTensor(weightBuf);
        if (hasBias_) {
            biasQueue_.FreeTensor(biasBuf);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> weightQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t inputN_{0};
    uint32_t inputC_{0};
    uint32_t inputH_{0};
    uint32_t inputW_{0};
    uint32_t weightOC_{0};
    uint32_t weightIC_{0};
    uint32_t weightKH_{0};
    uint32_t weightKW_{0};
    uint32_t outputH_{0};
    uint32_t outputW_{0};
    uint32_t weightLen_{0};
    uint32_t biasLen_{0};
    float subtractValue1_{0.5f};
    float subtractValue2_{0.2f};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_subtract_subtract_mish_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(Conv2dSubtractSubtractMishCustomTilingData, tiling);
    KernelConv2dSubtractSubtractMishCustom op;
    op.Init(x, y, weight, bias,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.inputN, tiling_data.inputC, tiling_data.inputH, tiling_data.inputW,
            tiling_data.weightOC, tiling_data.weightIC, tiling_data.weightKH, tiling_data.weightKW,
            tiling_data.outputH, tiling_data.outputW,
            tiling_data.subtractValue1, tiling_data.subtractValue2,
            tiling_data.hasBias);
    op.Process();
}
