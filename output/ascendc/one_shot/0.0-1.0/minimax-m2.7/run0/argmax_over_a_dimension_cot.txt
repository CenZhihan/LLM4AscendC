class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}
    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outputLength,
                                uint32_t tileLength, uint32_t blockDim,
                                int32_t axis, int32_t dim0, int32_t dim1, int32_t dim2,
                                uint32_t reductionSize)
    {
        totalLen_ = totalLength;
        outputLen_ = outputLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        axis_ = axis;
        dim0_ = dim0;
        dim1_ = dim1;
        dim2_ = dim2;
        reductionSize_ = reductionSize;

        // compute strides
        stride0_ = dim1 * dim2;
        stride1_ = dim2;
        stride2_ = 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ int32_t*)y, outputLen_);
    }

    __aicore__ inline void Process()
    {
        if (outputLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stride = blockDim_ * tileLen_;
        uint32_t base = blockIdx * tileLen_;

        while (base < outputLen_) {
            uint32_t curTileLen = (outputLen_ - base) >= tileLen_ ? tileLen_ : (outputLen_ - base);
            for (uint32_t i = 0; i < curTileLen; ++i) {
                uint32_t outIdx = base + i;
                int32_t bestIdx = -1;
                float bestVal = -std::numeric_limits<float>::max();

                // compute coordinates based on axis
                uint32_t i0, i1, i2;
                if (axis_ == 0) {
                    i1 = outIdx / dim2_;
                    i2 = outIdx % dim2_;
                } else if (axis_ == 1) {
                    i0 = outIdx / dim2_;
                    i2 = outIdx % dim2_;
                } else { // axis_ == 2
                    i0 = outIdx / dim1_;
                    i1 = outIdx % dim1_;
                }

                // compute base offset
                uint32_t baseOffset = i0 * stride0_ + i1 * stride1_ + i2 * stride2_;

                // reduction loop
                uint32_t strideAxis = (axis_ == 0) ? stride0_ : (axis_ == 1) ? stride1_ : stride2_;
                for (uint32_t k = 0; k < reductionSize_; ++k) {
                    uint32_t offset = baseOffset + k * strideAxis;
                    float val = xGm_[offset];
                    if (val > bestVal) {
                        bestVal = val;
                        bestIdx = static_cast<int32_t>(k);
                    }
                }
                yGm_[outIdx] = bestIdx;
            }
            base += stride;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int32_t> yGm_;

    uint32_t tileLen_{0};
    uint32_t blockDim_{0};
    int32_t axis_{0};
    int32_t dim0_{0};

    int32_t dim2_{0};
    uint32_t reductionSize_{0};
    uint32_t stride0_{0};
    uint32_t stride1_{0};
    uint32_t stride2_{0};
};

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outputLength,
                                uint32_t tileLength, uint32_t blockDim,
                                int32_t axis, int32_t dim0, int32_t dim1, int32_t dim2,
                                uint32_t reductionSize)
    {
        totalLen_ = totalLength;
        outputLen_ = outputLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;

        dim0_ = dim0;
        dim1_ = dim1;
        dim2_ = dim2;

        stride1_ = dim2;

        yGm_.SetGlobalBuffer((__gm__ int32_t*)y, outputLen_);
    }

    __aicore__ inline void Process()
    {
        if (outputLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stride = blockDim_ * tileLen_;
        uint32_t base = blockIdx * tileLen_;

        while (base < outputLen_) {
            uint32_t curTileLen = (outputLen_ - base) >= tileLen_ ? tileLen_ : (outputLen_ - base);
            for (uint32_t i = 0; i < curTileLen; ++i) {
                uint32_t outIdx = base + i;
                int32_t bestIdx = -1;

                uint32_t i0 = 0, i1 = 0, i2 = 0;

                    i2 = outIdx % dim2_;

                    i0 = outIdx / dim2_;
                    i2 = outIdx % dim2_;
                } else { // axis_ == 2
                    i0 = outIdx / dim1_;
                    i1 = outIdx % dim1_;

    return (a + b - 1u) / b;

static ge::graphStatus TilingFunc(gert::TilingContext *context) {
    ArgmaxOverADimensionCustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t axis = attrs->GetAttrPointer<int32_t>(0)[0];

    std::vector<int64_t> dims = context->GetInputShape(0)->GetOriginShape().GetDims();
    int64_t dim0 = dims.size() > 0 ? dims[0] : 1;
    int64_t dim1 = dims.size() > 1 ? dims[1] : 1;

#include "kernel_operator.h"
#include <limits>
#include <stdint.h>

constexpr int32_t BUFFER_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t outputLength,
                                uint32_t tileLength, uint32_t blockDim,
                                int32_t axis, int32_t dim0, int32_t dim1, int32_t dim2,
                                uint32_t reductionSize)
    {
        totalLen_ = totalLength;
        outputLen_ = outputLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        axis_ = axis;
        dim0_ = dim0;
        dim1_ = dim1;
        dim2_ = dim2;
        reductionSize_ = reductionSize;

        // compute strides for original shape
        stride0_ = dim1 * dim2;
        stride1_ = dim2;
        stride2_ = 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ int32_t*)y, outputLen_);
    }

    __aicore__ inline void Process()
    {
        if (outputLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t stride = blockDim_ * tileLen_;
        uint32_t base = blockIdx * tileLen_;

        while (base < outputLen_) {
            uint32_t curTileLen = (outputLen_ - base) >= tileLen_ ? tileLen_ : (outputLen_ - base);
            for (uint32_t i = 0; i < curTileLen; ++i) {
                uint32_t outIdx = base + i;
                int32_t bestIdx = -1;
                float bestVal = -std::numeric_limits<float>::max();

                // compute coordinates based on axis
                uint32_t i0 = 0, i1 = 0, i2 = 0;
                if (axis_ == 0) {
                    i1 = outIdx / dim2_;
                    i2 = outIdx % dim2_;
                } else if (axis_ == 1) {
                    i0 = outIdx / dim2_;
                    i2 = outIdx % dim2_;
                } else { // axis_ == 2
                    i0 = outIdx / dim1_;
                    i1 = outIdx % dim1_;
                }

                uint32_t baseOffset = i0 * stride0_ + i1 * stride1_ + i2 * stride2_;
                uint32_t strideAxis = (axis_ == 0) ? stride0_ : (axis_ == 1) ? stride1_ : stride2_;

                for (uint32_t k = 0; k < reductionSize_; ++k) {
                    uint32_t offset = baseOffset + k * strideAxis;
                    float val = xGm_[offset];
                    if (val > bestVal) {
                        bestVal = val;
                        bestIdx = static_cast<int32_t>(k);
                    }
                }
                yGm_[outIdx] = bestIdx;
            }
            base += stride;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int32_t> yGm_;
    uint32_t totalLen_{0};
    uint32_t outputLen_{0};
    uint32_t tileLen_{0};
    uint32_t blockDim_{0};
    int32_t axis_{0};
    int32_t dim0_{0};
    int32_t dim1_{0};
    int32_t dim2_{0};
    uint32_t reductionSize_{0};
    uint32_t stride0_{0};
    uint32_t stride1_{0};
    uint32_t stride2_{0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
