static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static inline int32_t CeilDivInt(int32_t a, int32_t b)
{
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::Shape *outputShape = context->GetOutputShape(0);

    const int32_t channels = static_cast<int32_t>(inputShape->GetShapeVector()[1]);
    const int32_t inputDim1 = static_cast<int32_t>(inputShape->GetShapeVector()[2]);
    const int32_t inputDim2 = static_cast<int32_t>(inputShape->GetShapeVector()[3]);
    const int32_t inputDim3 = static_cast<int32_t>(inputShape->GetShapeVector()[4]);
    const int32_t outputDim1 = static_cast<int32_t>(outputShape->GetShapeVector()[2]);
    const int32_t outputDim2 = static_cast<int32_t>(outputShape->GetShapeVector()[3]);
    const int32_t outputDim3 = static_cast<int32_t>(outputShape->GetShapeVector()[4]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t kernelSize = static_cast<int32_t>(attrs->GetAttrPointer<int64_t>(0)[0]);
    const int64_t *stridePtr = attrs->GetAttrPointer<int64_t>(1);
    const int32_t stride = (stridePtr != nullptr && stridePtr[0] != 0) ? 
                           static_cast<int32_t>(stridePtr[0]) : kernelSize;
    const int64_t *paddingPtr = attrs->GetAttrPointer<int64_t>(2);
    const int32_t padding = (paddingPtr != nullptr) ? static_cast<int32_t>(paddingPtr[0]) : 0;
    const int64_t *ceilModePtr = attrs->GetAttrPointer<int64_t>(3);
    const bool ceilMode = (ceilModePtr != nullptr) ? (ceilModePtr[0] != 0) : false;

    const uint32_t totalLength = channels * outputDim1 * outputDim2 * outputDim3;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_ceilMode(ceilMode);
    tiling.set_channels(channels);
    tiling.set_inputDim1(inputDim1);
    tiling.set_inputDim2(inputDim2);
    tiling.set_inputDim3(inputDim3);
    tiling.set_outputDim1(outputDim1);
    tiling.set_outputDim2(outputDim2);
    tiling.set_outputDim3(outputDim3);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    
    const int64_t kernelSize = attrs->GetAttrPointer<int64_t>(0)[0];
    const int64_t *stridePtr = attrs->GetAttrPointer<int64_t>(1);
    const int32_t stride = (stridePtr != nullptr && stridePtr[0] != 0) ? 
                           static_cast<int32_t>(stridePtr[0]) : static_cast<int32_t>(kernelSize);
    const int64_t *paddingPtr = attrs->GetAttrPointer<int64_t>(2);
    const int32_t padding = (paddingPtr != nullptr) ? static_cast<int32_t>(paddingPtr[0]) : 0;
    const int64_t *ceilModePtr = attrs->GetAttrPointer<int64_t>(3);
    const bool ceilMode = (ceilModePtr != nullptr) ? (ceilModePtr[0] != 0) : false;

    const int32_t channels = x_shape->GetShapeVector()[1];
    const int32_t inputD = x_shape->GetShapeVector()[2];
    const int32_t inputH = x_shape->GetShapeVector()[3];
    const int32_t inputW = x_shape->GetShapeVector()[4];

    int32_t outputD, outputH, outputW;

    if (ceilMode) {
        outputD = CeilDivInt(inputD + 2 * padding - static_cast<int32_t>(kernelSize) + 1, stride);
        outputH = CeilDivInt(inputH + 2 * padding - static_cast<int32_t>(kernelSize) + 1, stride);
        outputW = CeilDivInt(inputW + 2 * padding - static_cast<int32_t>(kernelSize) + 1, stride);
    } else {
        outputD = (inputD + 2 * padding - static_cast<int32_t>(kernelSize)) / stride + 1;
        outputH = (inputH + 2 * padding - static_cast<int32_t>(kernelSize)) / stride + 1;
        outputW = (inputW + 2 * padding - static_cast<int32_t>(kernelSize)) / stride + 1;
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetShapeVector(0, channels);
    y_shape->SetShapeVector(1, outputD);
    y_shape->SetShapeVector(2, outputH);
    y_shape->SetShapeVector(3, outputW);

    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_KERNEL_SIZE = 8;

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t kernelSize, int32_t stride, int32_t padding,
                                bool ceilMode, int32_t channels,
                                int32_t inputDim1, int32_t inputDim2, int32_t inputDim3,
                                int32_t outputDim1, int32_t outputDim2, int32_t outputDim3)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        ceilMode_ = ceilMode;
        channels_ = channels;
        inputDim1_ = inputDim1;
        inputDim2_ = inputDim2;
        inputDim3_ = inputDim3;
        outputDim1_ = outputDim1;
        outputDim2_ = outputDim2;
        outputDim3_ = outputDim3;

        xGm_.SetGlobalBuffer((__gm__ float*)x, channels_ * inputDim1_ * inputDim2_ * inputDim3_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, channels_ * outputDim1_ * outputDim2_ * outputDim3_);

        const uint32_t ubSize = kernelSize_ * kernelSize_ * kernelSize_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(maxQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = maxQueue_.AllocTensor<float>();

        const int32_t ks = kernelSize_;
        const int32_t st = stride_;
        const int32_t pd = padding_;
        const int32_t ch = channels_;
        const int32_t iD = inputDim1_;
        const int32_t iH = inputDim2_;
        const int32_t iW = inputDim3_;
        const int32_t oD = outputDim1_;
        const int32_t oH = outputDim2_;
        const int32_t oW = outputDim3_;

        for (uint32_t idx = 0; idx < len; ++idx) {
            const int32_t outIdx = idx;
            const int32_t c = outIdx / (oD * oH * oW);
            const int32_t rem1 = outIdx % (oD * oH * oW);
            const int32_t od = rem1 / (oH * oW);
            const int32_t rem2 = rem1 % (oH * oW);
            const int32_t oh = rem2 / oW;
            const int32_t ow = rem2 % oW;

            const int32_t startD = od * st - pd;
            const int32_t startH = oh * st - pd;
            const int32_t startW = ow * st - pd;

            float maxVal = -3.4e38f;

            for (int32_t kd = 0; kd < ks; ++kd) {
                const int32_t inD = startD + kd;
                if (inD < 0 || inD >= iD) continue;
                for (int32_t kh = 0; kh < ks; ++kh) {
                    const int32_t inH = startH + kh;
                    if (inH < 0 || inH >= iH) continue;
                    for (int32_t kw = 0; kw < ks; ++kw) {
                        const int32_t inW = startW + kw;
                        if (inW < 0 || inW >= iW) continue;

                        const int32_t inIdx = c * iD * iH * iW + inD * iH * iW + inH * iW + inW;
                        float val = xLocal.GetValue(inIdx);
                        if (val > maxVal) {
                            maxVal = val;
                        }
                    }
                }
            }

            yLocal.SetValue(idx, maxVal);
        }

        maxQueue_.FreeTensor(maxLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maxQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t kernelSize_ {3};
    int32_t stride_ {2};
    int32_t padding_ {0};
    bool ceilMode_ {false};
    int32_t channels_ {32};
    int32_t inputDim1_ {128};
    int32_t inputDim2_ {128};
    int32_t inputDim3_ {128};
    int32_t outputDim1_ {64};
    int32_t outputDim2_ {64};
    int32_t outputDim3_ {64};
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.ceilMode, tiling_data.channels,
            tiling_data.inputDim1, tiling_data.inputDim2, tiling_data.inputDim3,
            tiling_data.outputDim1, tiling_data.outputDim2, tiling_data.outputDim3);
    op.Process();
}
