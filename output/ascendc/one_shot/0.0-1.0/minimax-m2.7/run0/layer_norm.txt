host_operator_src="""

#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;
static constexpr uint32_t INNER_SIZE_ALIGN = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;

    const gert::Shape *inputShape = context->GetInputShape(0);
    std::vector<int64_t> shape = inputShape->GetShape();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = 1e-05f;
    if (attrs && attrs->GetAttrCount() > 1) {
        eps = attrs->GetAttrPointer<float>(1);
    }

    uint32_t totalLength = static_cast<uint32_t>(inputShape->GetShapeSize());
    uint32_t innerSize = 1;
    if (shape.size() >= 3) {
        innerSize = static_cast<uint32_t>(shape[shape.size() - 1]);
        for (size_t i = 1; i < shape.size(); ++i) {
            innerSize *= static_cast<uint32_t>(shape[i]);
        }
    }

    uint32_t alignedInner = AlignDown(innerSize, INNER_SIZE_ALIGN);
    if (alignedInner < INNER_SIZE_ALIGN) alignedInner = innerSize;

    uint32_t tileLen = UB_BUDGET_BYTES / (8 * sizeof(float));
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen < alignedInner) {
        tileLen = alignedInner;
    }
    if (tileLen > HARD_CAP_ELEMS) {
        tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    }

    if (totalLength < tileLen) {
        uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_innerSize(innerSize);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LayerNormCustom : public OpDef {
public:
    explicit LayerNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("normalized_shape").AttrType(OPTIONAL).ListInt({64, 256, 256});
        this->Attr("eps").AttrType(OPTIONAL).Float(1e-05);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LayerNormCustom);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t innerSize,
                                float eps)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        innerSize_ = innerSize;
        eps_ = eps;
        invInnerSize_ = 1.0f / static_cast<float>(innerSize_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileLen_ * sizeof(float));
        pipe_.InitBuffer(workspaceQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> meanLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> varLocal = tmpQueue_.AllocTensor<float>();

        // mean = sum(x) / N
        AscendC::ReduceSumMean(meanLocal, xLocal, 0, len);
        AscendC::BroadcastTo(meanLocal, meanLocal, len, 1);

        // var = sum((x - mean)^2) / N
        AscendC::Sub(varLocal, xLocal, meanLocal, len);
        AscendC::Muls(varLocal, varLocal, varLocal, len);
        AscendC::ReduceSumMean(varLocal, varLocal, 0, len);
        AscendC::BroadcastTo(varLocal, varLocal, len, 1);

        // rsqrt_var = 1 / sqrt(var + eps)
        AscendC::Adds(varLocal, varLocal, eps_, len);
        AscendC::Rsqrt(varLocal, varLocal, len);

        // y = (x - mean) * rsqrt_var
        AscendC::Sub(yLocal, xLocal, meanLocal, len);
        AscendC::Mul(yLocal, yLocal, varLocal, len);

        tmpQueue_.FreeTensor(meanLocal);
        tmpQueue_.FreeTensor(varLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workspaceQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {1};
    uint32_t innerSize_ {1};
    float eps_ {1e-05f};
    float invInnerSize_ {1.0f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, 
            tiling_data.blockDim, tiling_data.innerSize, tiling_data.eps);
    op.Process();
}
