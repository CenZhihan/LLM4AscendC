#include "gemm_relu_divide_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 190 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t kBlockSize = 32;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmReluDivideCustomTilingData tiling;
    
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    
    const int64_t batch = x_shape->GetShape().GetDims()[0];
    const int64_t m = batch;
    const int64_t k = x_shape->GetShape().GetDims()[1];
    const int64_t n = w_shape->GetShape().GetDims()[0];
    
    tiling.set_m(static_cast<uint32_t>(m));
    tiling.set_n(static_cast<uint32_t>(n));
    tiling.set_k(static_cast<uint32_t>(k));
    tiling.set_totalLength(static_cast<uint32_t>(m * n));
    
    // Get divisor from attrs
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *divisor = attrs->GetAttrPointer<float>(0);
    tiling.set_divisor(*divisor);
    
    // Calculate tile sizes based on UB budget
    // UB usage: x tile (tileM * k) + w tile (tileN * k) + y tile (tileM * tileN) + bias (n)
    uint32_t tileM = std::sqrt(UB_BUDGET_BYTES / (4.0f * k + 4.0f));
    tileM = std::min(tileM, static_cast<uint32_t>(m));
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;
    
    uint32_t tileN = UB_BUDGET_BYTES / (4 * tileM + 4 * k);
    tileN = std::min(tileN, static_cast<uint32_t>(n));
    tileN = AlignDown(tileN, ALIGN_ELEMS);
    if (tileN < ALIGN_ELEMS) tileN = ALIGN_ELEMS;
    
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    
    // Calculate blockDim based on total tiles
    const uint32_t numM = CeilDiv(static_cast<uint32_t>(m), tileM);
    const uint32_t numN = CeilDiv(static_cast<uint32_t>(n), tileN);
    const uint32_t numTiles = numM * numN;
    
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;
    
    context->SetBlockDim(blockDim);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t kBlockSize = 32;

class KernelGemmReluDivideCustom {
public:
    __aicore__ inline KernelGemmReluDivideCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalLength, uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileM, uint32_t tileN, float divisor)
    {
        m_ = m;
        n_ = n;
        k_ = k;
        tileM_ = tileM;
        tileN_ = tileN;
        divisor_ = divisor;
        invDivisor_ = 1.0f / divisor;

        xGm_.SetGlobalBuffer((__gm__ float*)x, m * k);
        wGm_.SetGlobalBuffer((__gm__ float*)w, n * k);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n);
        yGm_.SetGlobalBuffer((__gm__ float*)y, m * n);

        const uint32_t maxTileSize = tileM_ * tileN_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, maxTileSize * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, maxTileSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, maxTileSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t numM = (m_ + tileM_ - 1) / tileM_;
        const uint32_t numN = (n_ + tileN_ - 1) / tileN_;
        const uint32_t tileIdx = blockIdx;

        if (tileIdx >= numM * numN) return;

        const uint32_t rowIdx = tileIdx / numN;
        const uint32_t colIdx = tileIdx % numN;

        const uint32_t outRowIdx = rowIdx * tileM_;
        const uint32_t outColIdx = colIdx * tileN_;

        if (outRowIdx >= m_ || outColIdx >= n_) return;

        const uint32_t curM = (outRowIdx + tileM_ <= m_) ? tileM_ : (m_ - outRowIdx);
        const uint32_t curN = (outColIdx + tileN_ <= n_) ? tileN_ : (n_ - outColIdx);

        CopyIn(xGm_, wGm_, outRowIdx, outColIdx, curM, curN);
        Compute(curM, curN);
        CopyOut(yGm_, outRowIdx, outColIdx, curM, curN);
    }

private:
    __aicore__ inline void CopyIn(AscendC::GlobalTensor<float> &xGm, 
                                  AscendC::GlobalTensor<float> &wGm,
                                  uint32_t outRowIdx, uint32_t outColIdx, 
                                  uint32_t curM, uint32_t curN)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();

        AscendC::Dat_copy_pad(xLocal, xGm[outRowIdx * k_], curM * k_, 0, 0);
        AscendC::Dat_copy_pad(wLocal, wGm[outColIdx * k_], curN * k_, 0, 0);

        inQueueX_.EnQue(xLocal);
        inQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void Compute(uint32_t curM, uint32_t curN)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::InitFloat(yLocal, 0.0f, curM * curN);

        const uint32_t kBlocks = (k_ + kBlockSize - 1) / kBlockSize;
        for (uint32_t kb = 0; kb < kBlocks; ++kb) {
            const uint32_t kOffset = kb * kBlockSize;
            const uint32_t curK = (kOffset + kBlockSize <= k_) ? kBlockSize : (k_ - kOffset);

            AscendC::LocalTensor<float> xSub = xLocal[0];
            AscendC::LocalTensor<float> wSub = wLocal[0];

            AscendC::Matmul(yLocal, xSub[kOffset], wSub[kOffset], yLocal, curM, curN, curK);
        }

        AscendC::Add(yLocal, yLocal, bGm_, yLocal, curM * curN);

        AscendC::Relu(yLocal, yLocal, curM * curN);

        AscendC::Muls(yLocal, yLocal, invDivisor_, yLocal, curM * curN);

        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueW_.FreeTensor(wLocal);
    }

    __aicore__ inline void CopyOut(AscendC::GlobalTensor<float> &yGm, 
                                   uint32_t outRowIdx, uint32_t outColIdx,
                                   uint32_t curM, uint32_t curN)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::Dat_copy_pad(yGm[outRowIdx * n_ + outColIdx], yLocal, curM * curN);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t m_ {0};
    uint32_t n_ {0};
    uint32_t k_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
    float divisor_ {1.0f};
    float invDivisor_ {1.0f};
};

extern "C" __global__ __aicore__ void gemm_relu_divide_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmReluDivideCustom op;
    op.Init(x, w, b, y, tiling_data.totalLength, tiling_data.m, tiling_data.n, tiling_data.k,
           tiling_data.tileM, tiling_data.tileN, tiling_data.divisor);
    op.Process();
}
