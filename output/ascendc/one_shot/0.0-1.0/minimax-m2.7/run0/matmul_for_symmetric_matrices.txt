#include "matmul_for_symmetric_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for matrix multiplication:
//   A block: mTile * kTile floats
//   B block: kTile * nTile floats
//   C block: mTile * nTile floats
// Total = (2 * kTile * (mTile + nTile) + mTile * nTile) * sizeof(float)
// Conservative: ~128KB budget for UB

static constexpr uint32_t UB_BUDGET_BYTES = 128 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 32;
static constexpr uint32_t MIN_TILE = 64;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForSymmetricMatricesCustomTilingData tiling;
    const uint32_t n = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());

    // Determine tile sizes based on UB budget
    // For simplicity, use square-ish tiles
    uint32_t mTile = 128;
    uint32_t nTile = 128;
    uint32_t kTile = 64;

    // Verify UB fits: (mTile * kTile + kTile * nTile + mTile * nTile) * 4 bytes
    uint32_t needed = (mTile * kTile + kTile * nTile + mTile * nTile) * sizeof(float);
    if (needed > UB_BUDGET_BYTES) {
        // Reduce tile sizes
        mTile = 64;
        nTile = 64;
        kTile = 32;
    }

    mTile = AlignDown(mTile, ALIGN_ELEMS);
    nTile = AlignDown(nTile, ALIGN_ELEMS);
    kTile = AlignDown(kTile, ALIGN_ELEMS);

    if (mTile < MIN_TILE) mTile = MIN_TILE;
    if (nTile < MIN_TILE) nTile = MIN_TILE;
    if (kTile < MIN_TILE) kTile = MIN_TILE;

    // Calculate blockDim: each block handles one mTile x nTile output block
    const uint32_t totalBlocks = CeilDiv(n, mTile) * CeilDiv(n, nTile);
    uint32_t blockDim = totalBlocks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(n * n);
    tiling.set_n(n);
    tiling.set_mTile(mTile);
    tiling.set_nTile(nTile);
    tiling.set_kTile(kTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    gert::Shape *c_shape = context->GetOutputShape(0);
    
    const int64_t n = a_shape->GetShapeSize();
    c_shape->Resize(2);
    c_shape->SetDim(0, n);
    c_shape->SetDim(1, n);
    return GRAPH_SUCCESS;
}

    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_TILE = 128;
constexpr int32_t K_TILE = 64;

class KernelMatmulForSymmetricMatricesCustom {
public:
    __aicore__ inline KernelMatmulForSymmetricMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t n, uint32_t mTile, uint32_t nTile, uint32_t kTile)
    {
        n_ = n;
        mTile_ = mTile;
        nTile_ = nTile;
        kTile_ = kTile;

        aGm_.SetGlobalBuffer((__gm__ float*)A, n_ * n_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, n_ * n_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, n_ * n_);

        // Allocate UB buffers for A, B, C blocks
        pipe_.InitBuffer(aQueueA_, BUFFER_NUM, mTile_ * kTile_ * sizeof(float));
        pipe_.InitBuffer(bQueueB_, BUFFER_NUM, kTile_ * nTile_ * sizeof(float));
        pipe_.InitBuffer(cQueueC_, 1, mTile_ * nTile_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, mTile_ * nTile_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t nTileBlocks = CeilDiv(n_, nTile_);
        const uint32_t totalBlocks = CeilDiv(n_, mTile_) * nTileBlocks;

        // Assign blocks cyclically
        uint32_t assignedBlocks = 0;
        uint32_t mStart = 0;
        uint32_t nStart = 0;

        for (uint32_t i = 0; i < totalBlocks && assignedBlocks <= blockIdx; ++i) {
            uint32_t mCur = i / nTileBlocks;
            uint32_t nCur = i % nTileBlocks;
            mStart = mCur * mTile_;
            nStart = nCur * nTile_;
            assignedBlocks++;
        }

        if (blockIdx >= totalBlocks) return;

        // Compute C[mStart:mStart+mTile_, nStart:nStart+nTile_] = A[mStart:mStart+mTile_, :] * B[:, nStart:nStart+nTile_]
        ComputeBlock(mStart, nStart);
    }

private:
    __aicore__ inline void ComputeBlock(uint32_t mStart, uint32_t nStart)
    {
        const uint32_t actualMRows = (mStart + mTile_ > n_) ? (n_ - mStart) : mTile_;
        const uint32_t actualNCols = (nStart + nTile_ > n_) ? (n_ - nStart) : nTile_;

        // Initialize C block to zero
        AscendC::LocalTensor<float> cLocal = cQueueC_.AllocTensor<float>();
        AscendC::VectorRepu(cLocal, 0.0f, actualMRows * actualNCols);

        // Loop over K dimension
        for (uint32_t k = 0; k < n_; k += kTile_) {
            const uint32_t actualK = (k + kTile_ > n_) ? (n_ - k) : kTile_;

            // Load A block: rows mStart:mStart+actualMRows, cols k:k+actualK
            AscendC::LocalTensor<float> aLocal = aQueueA_.AllocTensor<float>();
            LoadMatrixBlock(aGm_, aLocal, mStart, k, actualMRows, actualK, n_);
            aQueueA_.EnQue(aLocal);

            // Load B block: rows k:k+actualK, cols nStart:nStart+actualNCols
            AscendC::LocalTensor<float> bLocal = bQueueB_.AllocTensor<float>();
            LoadMatrixBlock(bGm_, bLocal, k, nStart, actualK, actualNCols, n_);
            bQueueB_.EnQue(bLocal);

            // Compute partial result: C_block += A_block @ B_block
            AscendC::LocalTensor<float> aTile = aQueueA_.DeQue<float>();
            AscendC::LocalTensor<float> bTile = bQueueB_.DeQue<float>();
            
            // Use MatMul for inner product
            AscendC::MatMul<float>(cLocal, aTile, bTile, cLocal, actualMRows, actualNCols, actualK, 
                                   actualMRows, actualK, actualNCols, actualNCols);

            aQueueA_.FreeTensor(aTile);
            bQueueB_.FreeTensor(bTile);
        }

        // Write C block back to GM
        StoreMatrixBlock(cLocal, cGm_, mStart, nStart, actualMRows, actualNCols, n_);
        cQueueC_.FreeTensor(cLocal);
    }

    __aicore__ inline void LoadMatrixBlock(AscendC::GlobalTensor<float>& srcGm,
                                           AscendC::LocalTensor<float>& dstLocal,
                                           uint32_t rowStart, uint32_t colStart,
                                           uint32_t rows, uint32_t cols, uint32_t stride)
    {
        AscendC::DataCopyExtParams params;
        params.srcStride = stride;
        params.dstStride = cols;
        params.srcRepStride = 0;
        params.dstRepStride = 0;
        AscendC::DataCopy(dstLocal, srcGm[rowStart * stride + colStart], rows, cols, params);
    }

    __aicore__ inline void StoreMatrixBlock(AscendC::LocalTensor<float>& srcLocal,
                                            AscendC::GlobalTensor<float>& dstGm,
                                            uint32_t rowStart, uint32_t colStart,
                                            uint32_t rows, uint32_t cols, uint32_t stride)
    {
        AscendC::DataCopyExtParams params;
        params.srcStride = cols;
        params.dstStride = stride;
        params.srcRepStride = 0;
        params.dstRepStride = 0;
        AscendC::DataCopy(dstGm[rowStart * stride + colStart], srcLocal, rows, cols, params);
    }

    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b)
    {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bQueueB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> cQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t n_ {0};
    uint32_t mTile_ {128};
    uint32_t nTile_ {128};
    uint32_t kTile_ {64};
};

extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForSymmetricMatricesCustom op;
    op.Init(A, B, C, tiling_data.n, tiling_data.mTile, tiling_data.nTile, tiling_data.kTile);
    op.Process();
}
