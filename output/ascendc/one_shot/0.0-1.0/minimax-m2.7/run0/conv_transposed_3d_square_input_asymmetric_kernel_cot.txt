        if (tileLen == 0) tileLen = totalLength;
    }

    uint32_t numTiles = CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_C_in(C_in);
    tiling.set_C_out(C_out);
    tiling.set_kD(kD);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_D_in(D_in);
    tiling.set_H_in(H_in);
    tiling.set_W_in(W_in);
    tiling.set_D_out(D_out);
    tiling.set_H_out(H_out);
    tiling.set_W_out(W_out);
    tiling.set_strideD(strideD);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_paddingD(paddingD);
    tiling.set_paddingH(paddingH);
    tiling.set_paddingW(paddingW);
    tiling.set_outputPaddingD(outputPaddingD);
    tiling.set_outputPaddingH(outputPaddingH);
    tiling.set_outputPaddingW(outputPaddingW);
    tiling.set_dilationD(dilationD);
    tiling.set_dilationH(dilationH);
    tiling.set_dilationW(dilationW);
    tiling.set_groups(groups);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *weightShape = context->GetInputShape(1);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    const int *stridePtr = attrs->GetAttrPointer<int>(0);
    const int *paddingPtr = attrs->GetAttrPointer<int>(1);
    const int *outputPaddingPtr = attrs->GetAttrPointer<int>(2);
    const int *groupsPtr = attrs->GetAttrPointer<int>(3);
    const int *dilationPtr = attrs->GetAttrPointer<int>(4);

    uint32_t strideD = stridePtr[0];
    uint32_t strideH = stridePtr[1];
    uint32_t strideW = stridePtr[2];
    uint32_t paddingD = paddingPtr[0];
    uint32_t paddingH = paddingPtr[1];
    uint32_t paddingW = paddingPtr[2];
    uint32_t outputPaddingD = outputPaddingPtr[0];
    uint32_t outputPaddingH = outputPaddingPtr[1];
    uint32_t outputPaddingW = outputPaddingPtr[2];
    uint32_t groups = groupsPtr[0];
    uint32_t dilationD = dilationPtr[0];
    uint32_t dilationH = dilationPtr[1];
    uint32_t dilationW = dilationPtr[2];

    uint32_t N = xShape->GetShapeVector()[0];
    uint32_t C_in = xShape->GetShapeVector()[1];
    uint32_t D_in = xShape->GetShapeVector()[2];
    uint32_t H_in = xShape->GetShapeVector()[3];
    uint32_t W_in = xShape->GetShapeVector()[4];

    uint32_t C_out = weightShape->GetShapeVector()[1];
    uint32_t kD = weightShape->GetShapeVector()[2];
    uint32_t kH = weightShape->GetShapeVector()[3];
    uint32_t kW = weightShape->GetShapeVector()[4];

    uint32_t D_out = (D_in - 1) * strideD + dilationD * (kD - 1) + 1 + outputPaddingD - 2 * paddingD;
    uint32_t H_out = (H_in - 1) * strideH + dilationH * (kH - 1) + 1 + outputPaddingH - 2 * paddingH;
    uint32_t W_out = (W_in - 1) * strideW + dilationW * (kW - 1) + 1 + outputPaddingW - 2 * paddingW;

    gert::Shape *yShape = context->GetOutputShape(0);
    yShape->GetShapeVector().push_back(N);
    yShape->GetShapeVector().push_back(C_out);
    yShape->GetShapeVector().push_back(D_out);
    yShape->GetShapeVector().push_back(H_out);
    yShape->GetShapeVector().push_back(W_out);
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return ge::GRAPH_SUCCESS;
}

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t C_in, uint32_t C_out,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t D_in, uint32_t H_in, uint32_t W_in,
                                uint32_t D_out, uint32_t H_out, uint32_t W_out,
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t paddingD, uint32_t paddingH, uint32_t paddingW,
                                uint32_t outputPaddingD, uint32_t outputPaddingH, uint32_t outputPaddingW,
                                uint32_t dilationD, uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        C_in_ = C_in;
        C_out_ = C_out;
        kD_ = kD;
        kH_ = kH;
        kW_ = kW;
        D_in_ = D_in;
        H_in_ = H_in;
        W_in_ = W_in;
        D_out_ = D_out;
        H_out_ = H_out;
        W_out_ = W_out;
        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        paddingD_ = paddingD;
        paddingH_ = paddingH;
        paddingW_ = paddingW;
        outputPaddingD_ = outputPaddingD;
        outputPaddingH_ = outputPaddingH;
        outputPaddingW_ = outputPaddingW;
        dilationD_ = dilationD;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;

        xGm_.SetGlobalBuffer((__gm__ float*)x, C_in_ * D_in_ * H_in_ * W_in_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, C_out_ * D_out_ * H_out_ * W_out_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, C_in_ * C_out_ * kD_ * kH_ * kW_);

        // Compute weight elements
        weightElems_ = C_in_ * C_out_ * kD_ * kH_ * kW_;

        // Allocate UB buffers: weight (weightElems), input region (weightElems), output (tileLen)
        pipe_.InitBuffer(weightQueue_, BUFFER_NUM, weightElems_ * sizeof(float));
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, weightElems_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        // Load weight once per block (or per tile)
        LoadWeight();

        // Process tiles in grid-stride loop
        while (true) {
            // Compute for current tile
            ComputeAndCopyOut(base, curLen);

            // Prepare next tile
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                // Optionally preload input for next tile if needed
            }

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void LoadWeight()
    {
        // Load entire weight into UB for reuse across the block
        AscendC::LocalTensor<float> weightLocal = weightQueue_.AllocTensor<float>();
        AscendC::DataCopy(weightLocal, weightGm_, weightElems_);
        weightQueue_.EnQue(weightLocal);
    }

    __aicore__ inline void ComputeAndCopyOut(uint32_t globalStart, uint32_t len)
    {
        // For each output element in this tile, compute convolution sum
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            uint32_t outIdx = globalStart + i;
            float sum = ComputeConvolution(outIdx);
            yLocal[i] = sum;
        }

        outQueueY_.EnQue(yLocal);

        // Copy output to global
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[globalStart], yOut, len);
        outQueueY_.FreeTensor(yOut);
    }

    __aicore__ inline float ComputeConvolution(uint32_t outIdx)
    {
        // Decode output coordinates from linear index
        uint32_t n = outIdx / (C_out_ * D_out_ * H_out_ * W_out_);
        outIdx %= (C_out_ * D_out_ * H_out_ * W_out_);
        uint32_t c_out = outIdx / (D_out_ * H_out_ * W_out_);
        outIdx %= (D_out_ * H_out_ * W_out_);
        uint32_t d_out = outIdx / (H_out_ * W_out_);
        outIdx %= (H_out_ * W_out_);
        uint32_t h_out = outIdx / W_out_;
        uint32_t w_out = outIdx % W_out_;

        float sum = 0.0f;

        // Loop over input channels and kernel dimensions
        for (uint32_t c_in = 0; c_in < C_in_; ++c_in) {
            for (uint32_t kd = 0; kd < kD_; ++kd) {
                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        // Compute input coordinate (applying stride, dilation, padding)
                        int32_t d_in = d_out * strideD_ - paddingD_ + kd * dilationD_;
                        int32_t h_in = h_out * strideH_ - paddingH_ + kh * dilationH_;
                        int32_t w_in = w_out * strideW_ - paddingW_ + kw * dilationW_;

                        // Check bounds
                        if (d_in >= 0 && d_in < D_in_ && h_in >= 0 && h_in < H_in_ && w_in >= 0 && w_in < W_in_) {
                            // Compute linear indices
                            uint32_t inIdx = ((n * C_in_ + c_in) * D_in_ + d_in) * H_in_ + h_in) * W_in_ + w_in;
                            uint32_t weightIdx = (((c_in * C_out_ + c_out) * kD_ + kd) * kH_ + kh) * kW_ + kw;

                            // Get weight and input values
                            // We have weight in UB (weightQueue_)
                            AscendC::LocalTensor<float> weightLocal = weightQueue_.DeQue<float>();
                            float w_val = weightLocal[weightIdx];
                            weightQueue_.EnQue(weightLocal);

                            // Load input value from global (or from a buffer)
                            float x_val = xGm_[inIdx];

                            sum += w_val * x_val;
                        }
                    }
                }
            }
        }

        return sum;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> weightQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t C_in_ {0};
    uint32_t C_out_ {0};
    uint32_t kD_ {0};
    uint32_t kH_ {0};
    uint32_t kW_ {0};
    uint32_t D_in_ {0};
    uint32_t H_in_ {0};
    uint32_t W_in_ {0};
    uint32_t D_out_ {0};
    uint32_t H_out_ {0};
    uint32_t W_out_ {0};
    uint32_t strideD_ {1};
    uint32_t strideH_ {1};
    uint32_t strideW_ {1};
    uint32_t paddingD_ {0};
    uint32_t paddingH_ {0};
    uint32_t paddingW_ {0};
    uint32_t outputPaddingD_ {0};
    uint32_t outputPaddingH_ {0};
    uint32_t outputPaddingW_ {0};
    uint32_t dilationD_ {1};
    uint32_t dilationH_ {1};
    uint32_t dilationW_ {1};
    uint32_t groups_ {1};
    uint32_t weightElems_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR workspace, GM_ADDR tiling)
