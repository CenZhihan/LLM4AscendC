class KernelConvDepthwise2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR w,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t H_out, uint32_t W_out,
                                uint32_t kernelSize, uint32_t pad,
                                uint32_t stride, uint32_t dilation,
                                uint32_t groups)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        H_out_ = H_out;
        W_out_ = W_out;
        kernelSize_ = kernelSize;
        pad_ = pad;
        stride_ = stride;
        dilation_ = dilation;
        groups_ = groups;
        weightSize_ = C_ * kernelSize_ * kernelSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * H_out_ * W_out_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, weightSize_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, weightSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Load weight once per block
        AscendC::LocalTensor<float> weightLocal = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(weightLocal, wGm_, weightSize_);

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            Compute(base, curLen, weightLocal);
            CopyOut(base, curLen);

            base += strideElems;
            if (base >= totalLen_) break;
        }

        tmpQueue_.FreeTensor(weightLocal);
    }

private:
    __aicore__ inline void Compute(uint32_t base, uint32_t len, AscendC::LocalTensor<float> weightLocal)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t idx = base + i;
            uint32_t n = idx / (C_ * H_out_ * W_out_);
            uint32_t idxNC = idx % (C_ * H_out_ * W_out_);
            uint32_t c = idxNC / (H_out_ * W_out_);
            uint32_t idxHW = idxNC % (H_out_ * W_out_);
            uint32_t h_out = idxHW / W_out_;
            uint32_t w_out = idxHW % W_out_;

            float sum = 0.0f;
            for (int32_t ki = 0; ki < kernelSize_; ++ki) {
                for (int32_t kj = 0; kj < kernelSize_; ++kj) {
                    uint32_t w_idx = c * (kernelSize_ * kernelSize_) + ki * kernelSize_ + kj;
                    float w_val = weightLocal[w_idx];
                    uint32_t in_idx = ((n * C_ + c) * H_ + (h_out + ki)) * W_ + (w_out + kj);
                    float x_val = xGm_[in_idx];
                    sum += w_val * x_val;
                }
            }
            yLocal[i] = sum;
        }
        outQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wGm_;

    uint32_t tileLen_ {0};
    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t H_out_ {0};
    uint32_t W_out_ {0};
    uint32_t kernelSize_ {3};
    uint32_t pad_ {0};
    uint32_t stride_ {1};
    uint32_t dilation_ {1};
    uint32_t groups_ {1};
    uint32_t weightSize_ {0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t H_out, uint32_t W_out,
                                uint32_t kernelSize, uint32_t pad,
                                uint32_t stride, uint32_t dilation,
                                uint32_t groups)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N;
        C_ = C;
        H_ = H;
        W_ = W;
        H_out_ = H_out;
        W_out_ = W_out;
        kernelSize_ = kernelSize;
        pad_ = pad;
        stride_ = stride;
        dilation_ = dilation;
        groups_ = groups;
        weightSize_ = C_ * kernelSize_ * kernelSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * C_ * H_out_ * W_out_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, weightSize_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, weightSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // Load weight once per block
        AscendC::LocalTensor<float> weightLocal = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(weightLocal, wGm_, weightSize_);

        while (true) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            Compute(base, curLen, weightLocal);
            CopyOut(base, curLen);

            base += strideElems;
            if (base >= totalLen_) break;
        }

        tmpQueue_.FreeTensor(weightLocal);
    }

private:
    __aicore__ inline void Compute(uint32_t base, uint32_t len, AscendC::LocalTensor<float> weightLocal)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t idx = base + i;
            uint32_t n = idx / (C_ * H_out_ * W_out_);
            uint32_t idxNC = idx % (C_ * H_out_ * W_out_);
            uint32_t c = idxNC / (H_out_ * W_out_);
            uint32_t idxHW = idxNC % (H_out_ * W_out_);
            uint32_t h_out = idxHW / W_out_;
            uint32_t w_out = idxHW % W_out_;

            float sum = 0.0f;
            for (int32_t ki = 0; ki < kernelSize_; ++ki) {
                for (int32_t kj = 0; kj < kernelSize_; ++kj) {
                    uint32_t w_idx = c * (kernelSize_ * kernelSize_) + ki * kernelSize_ + kj;
                    float w_val = weightLocal[w_idx];
                    uint32_t in_idx = ((n * C_ + c) * H_ + (h_out + ki)) * W_ + (w_out + kj);
                    float x_val = xGm_[in_idx];
                    sum += w_val * x_val;
                }
            }
            yLocal[i] = sum;
        }
        outQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t N_ {0};
    uint32_t C_ {0};
    uint32_t H_ {0};
    uint32_t W_ {0};
    uint32_t H_out_ {0};
    uint32_t W_out_ {0};
    uint32_t kernelSize_ {3};
    uint32_t pad_ {0};
    uint32_t stride_ {1};
    uint32_t dilation_ {1};
    uint32_t groups_ {1};
    uint32_t weightSize_ {0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
