#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile (input buffer)
//   outQueueY: 2 * tile (output buffer)
//   tmpQueueMax: 1 * tile (for max values)
//   tmpQueueIdx: 1 * tile (indices if needed)
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    
    // Input shape: (batch, features, length)
    const uint32_t batchSize = static_cast<uint32_t>(inputShape->GetShape().at(0));
    const uint32_t features = static_cast<uint32_t>(inputShape->GetShape().at(1));
    const uint32_t inputLength = static_cast<uint32_t>(inputShape->GetShape().at(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernelSizeAttr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *strideAttr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *paddingAttr = attrs->GetAttrPointer<int32_t>(2);

    uint32_t kernelSize = (*kernelSizeAttr > 0) ? static_cast<uint32_t>(*kernelSizeAttr) : 1;
    uint32_t stride = (*strideAttr > 0) ? static_cast<uint32_t>(*strideAttr) : kernelSize;
    uint32_t padding = (*paddingAttr > 0) ? static_cast<uint32_t>(*paddingAttr) : 0;

    // Calculate output length
    uint32_t outputLength = (inputLength + 2 * padding - kernelSize) / stride + 1;
    
    // Total elements to process (flattened as batch * features * outputLength for output)
    uint32_t totalLength = batchSize * features * outputLength;
    uint32_t inputTotalLength = batchSize * features * inputLength;

    // tileLen based on UB budget: bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    // Adaptive blockDim
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_inputLength(inputLength);
    tiling.set_outputLength(outputLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_batchSize(batchSize);
    tiling.set_features(features);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t inputLength,
                                uint32_t outputLength, uint32_t tileLength,
                                uint32_t kernelSize, uint32_t stride,
                                uint32_t padding, uint32_t batchSize,
                                uint32_t features)
    {
        totalLen_ = totalLength;
        inputLen_ = inputLength;
        outputLen_ = outputLength;
        tileLen_ = tileLength;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        batchSize_ = batchSize;
        features_ = features;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * features_ * inputLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * kernelSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueMax_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueIdx_, 1, tileLen_ * sizeof(uint32_t));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(base, curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t outOffset, uint32_t outLen)
    {
        // Each output element needs kernelSize input elements
        uint32_t inputElementsNeeded = outLen * kernelSize_;
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        
        // Calculate actual input indices for this tile
        uint32_t firstOutIdx = outOffset;
        uint32_t batchIdx = firstOutIdx / (features_ * outputLen_);
        uint32_t remainder = firstOutIdx % (features_ * outputLen_);
        uint32_t featIdx = remainder / outputLen_;
        uint32_t outIdx = remainder % outputLen_;
        
        // Calculate input start position
        int32_t inStart = static_cast<int32_t>(outIdx * stride_) - static_cast<int32_t>(padding_);
        
        for (uint32_t i = 0; i < outLen; ++i) {
            uint32_t currentOutIdx = firstOutIdx + i;
            uint32_t currBatch = currentOutIdx / (features_ * outputLen_);
            uint32_t currRem = currentOutIdx % (features_ * outputLen_);
            uint32_t currFeat = currRem / outputLen_;
            uint32_t currOutPos = currRem % outputLen_;
            
            int32_t currInStart = static_cast<int32_t>(currOutPos * stride_) - static_cast<int32_t>(padding_);
            
            for (uint32_t k = 0; k < kernelSize_; ++k) {
                int32_t inPos = currInStart + static_cast<int32_t>(k);
                if (inPos >= 0 && inPos < static_cast<int32_t>(inputLen_)) {
                    uint32_t flatInIdx = currBatch * features_ * inputLen_ + currFeat * inputLen_ + inPos;
                    xLocal[i * kernelSize_ + k] = xGm_.GetValue(flatInIdx);
                } else {
                    xLocal[i * kernelSize_ + k] = -65504.0f;  // Very small value for padding
                }
            }
        }
        
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t outOffset, uint32_t outLen)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxLocal = tmpQueueMax_.AllocTensor<float>();
        AscendC::LocalTensor<uint32_t> idxLocal = tmpQueueIdx_.AllocTensor<uint32_t>();

        // Find max within each kernel window
        // xLocal layout: [out0_k0, out0_k1, ..., out0_kK-1, out1_k0, ...]
        for (uint32_t i = 0; i < outLen; ++i) {
            float maxVal = xLocal[i * kernelSize_];
            uint32_t maxIdx = 0;
            for (uint32_t k = 1; k < kernelSize_; ++k) {
                float val = xLocal[i * kernelSize_ + k];
                if (val > maxVal) {
                    maxVal = val;
                    maxIdx = k;
                }
            }
            maxLocal.SetValue(i, maxVal);
            idxLocal.SetValue(i, maxIdx);
        }

        // Copy max values to output
        AscendC::DuplicatingTensor(yLocal, maxLocal, outLen);

        tmpQueueMax_.FreeTensor(maxLocal);
        tmpQueueIdx_.FreeTensor(idxLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        for (uint32_t i = 0; i < len; ++i) {
            yGm_.SetValue(gmOffset + i, yLocal.GetValue(i));
        }
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMax_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueIdx_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t inputLen_ {0};
    uint32_t outputLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t kernelSize_ {1};
    uint32_t stride_ {1};
    uint32_t padding_ {0};
    uint32_t batchSize_ {1};
    uint32_t features_ {1};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.inputLength, tiling_data.outputLength,
           tiling_data.tileLength, tiling_data.kernelSize, tiling_data.stride,
           tiling_data.padding, tiling_data.batchSize, tiling_data.features);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'max_pooling_1d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0, return_indices: bool = False):
        super(ModelNew, self).__init__()
        self.kernel_size = kernel_size
        self.stride = stride if stride is not None else kernel_size
        self.padding = padding
        self.return_indices = return_indices

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.kernel_size, self.stride, self.padding, self.return_indices)
