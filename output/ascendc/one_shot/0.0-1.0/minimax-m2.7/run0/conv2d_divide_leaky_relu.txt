kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dDivideLeakyReluCustom {
public:
    __aicore__ inline KernelConv2dDivideLeakyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t batch, uint32_t inputC, uint32_t outputC,
                                uint32_t inputH, uint32_t inputW,
                                uint32_t outputH, uint32_t outputW,
                                uint32_t kernelH, uint32_t kernelW,
                                int32_t padHT, int32_t padHB, int32_t padWL, int32_t padWR,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t tileH, uint32_t tileW,
                                float divisor)
    {
        xGm_.SetGlobalBuffer((__gm__ uint8_t*)x);
        yGm_.SetGlobalBuffer((__gm__ uint8_t*)y);
        wGm_.SetGlobalBuffer((__gm__ uint8_t*)weight);
        hasBias_ = (bias != nullptr);
        if (hasBias_) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias);
        }

        batch_ = batch;
        inputC_ = inputC;
        outputC_ = outputC;
        inputH_ = inputH;
        inputW_ = inputW;
        outputH_ = outputH;
        outputW_ = outputW;
        kernelH_ = kernelH;
        kernelW_ = kernelW;
        padHT_ = padHT;
        padHB_ = padHB;
        padWL_ = padWL;
        padWR_ = padWR;
        strideH_ = strideH;
        strideW_ = strideW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        tileH_ = tileH;
        tileW_ = tileW;
        divisor_ = divisor;
        negSlope_ = 0.01f;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileH_ * tileW_ * inputC_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, kernelH_ * kernelW_ * inputC_ * outputC_ * sizeof(float) / 4);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileH_ * tileW_ * outputC_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        uint32_t tileHIdx = 0;
        uint32_t tileWIdx = 0;

        for (uint32_t b = 0; b < batch_; ++b) {
            for (uint32_t oc = 0; oc < outputC_; ++oc) {
                tileHIdx = 0;
                tileWIdx = 0;
                while (tileHIdx < outputH_) {
                    while (tileWIdx < outputW_) {
                        uint32_t hOffset = tileHIdx * strideH_;
                        uint32_t wOffset = tileWIdx * strideW_;
                        uint32_t outTileH = (tileHIdx + tileH_ > outputH_) ? (outputH_ - tileHIdx) : tileH_;
                        uint32_t outTileW = (tileWIdx + tileW_ > outputW_) ? (outputW_ - tileWIdx) : tileW_;

                        uint32_t inTileH = (outTileH - 1) * dilationH_ + kernelH_;
                        uint32_t inTileW = (outTileW - 1) * dilationW_ + kernelW_;

                        int32_t inHStart = static_cast<int32_t>(hOffset) - padHT_;
                        int32_t inWStart = static_cast<int32_t>(wOffset) - padWL_;

                        uint32_t inTileHStart = (inHStart < 0) ? 0 : inHStart;
                        uint32_t inTileWStart = (inWStart < 0) ? 0 : inWStart;
                        uint32_t inTileHEnd = (inHStart + static_cast<int32_t>(inTileH) > static_cast<int32_t>(inputH_)) ? inputH_ : inHStart + inTileH;
                        uint32_t inTileWEnd = (inWStart + static_cast<int32_t>(inTileW) > static_cast<int32_t>(inputW_)) ? inputW_ : inWStart + inTileW;

                        uint32_t actualInH = inTileHEnd - inTileHStart;
                        uint32_t actualInW = inTileWEnd - inTileWStart;

                        CopyIn(xGm_, b, inputC_, inputH_, inputW_, inTileHStart, inTileWStart, actualInH, actualInW);
                        CopyInWeight(oc, inputC_, kernelH_, kernelW_);

                        Compute(actualInH, actualInW, outTileH, outTileW, hOffset, wOffset, oc, b, inTileHStart, inTileWStart, inHStart, inWStart);

                        CopyOut(yGm_, b, outputC_, outputH_, outputW_, tileHIdx, tileWIdx, outTileH, outTileW);

                        tileWIdx += tileW_;
                    }
                    tileWIdx = 0;
                    tileHIdx += tileH_;
                }
            }
        }
    }

private:
    __aicore__ inline void CopyIn(AscendC::GlobalTensor<uint8_t>& srcGm, uint32_t b, uint32_t c, uint32_t h, uint32_t w,
                                   uint32_t hStart, uint32_t wStart, uint32_t actualH, uint32_t actualW)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, srcGm, b, c, h, w, hStart, wStart, actualH, actualW);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInWeight(uint32_t ocStart, uint32_t inputC, uint32_t kernelH, uint32_t kernelW)
    {
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_, ocStart, 0, 0, 0, inputC, kernelH, kernelW);
        inQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void Compute(uint32_t actualInH, uint32_t actualInW,
                                    uint32_t outTileH, uint32_t outTileW,
                                    uint32_t hOffset, uint32_t wOffset,
                                    uint32_t ocStart, uint32_t batchIdx,
                                    uint32_t inTileHStart, uint32_t inTileWStart,
                                    int32_t inHStart, int32_t inWStart)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::Sets(yLocal, 0.0f, outTileH * outTileW);

        if (hasBias_) {
            for (uint32_t ow = 0; ow < outTileW; ++ow) {
                for (uint32_t oh = 0; oh < outTileH; ++oh) {
                    uint32_t yIdx = oh * outTileW + ow;
                    yLocal[yIdx] = biasGm_[ocStart];
                }
            }
        }

        for (uint32_t ic = 0; ic < inputC_; ++ic) {
            for (uint32_t kh = 0; kh < kernelH_; ++kh) {
                for (uint32_t kw = 0; kw < kernelW_; ++kw) {
                    int32_t actualIH = inHStart + static_cast<int32_t>(kh * dilationH_);
                    int32_t actualIW = inWStart + static_cast<int32_t>(kw * dilationW_);

                    if (actualIH >= 0 && actualIH < static_cast<int32_t>(inputH_) &&
                        actualIW >= 0 && actualIW < static_cast<int32_t>(inputW_)) {
                        uint32_t actualISrcH = actualIH - inTileHStart;
                        uint32_t actualISrcW = actualIW - inTileWStart;

                        for (uint32_t oh = 0; oh < outTileH; ++oh) {
                            for (uint32_t ow = 0; ow < outTileW; ++ow) {
                                uint32_t inputHIdx = actualISrcH + oh * dilationH_;
                                uint32_t inputWIdx = actualISrcW + ow * dilationW_;

                                if (inputHIdx < actualInH && inputWIdx < actualInW) {
                                    uint32_t xIdx = inputHIdx * actualInW + inputWIdx;
                                    uint32_t wIdx = ic * kernelH_ * kernelW_ + kh * kernelW_ + kw;
                                    uint32_t yIdx = oh * outTileW + ow;
                                    yLocal[yIdx] += xLocal[xIdx] * wLocal[wIdx];
                                }
                            }
                        }
                    }
                }
            }
        }

        for (uint32_t i = 0; i < outTileH * outTileW; ++i) {
            yLocal[i] = yLocal[i] / divisor_;
            if (yLocal[i] < 0.0f) {
                yLocal[i] = yLocal[i] * negSlope_;
            }
        }

        outQueueY_.EnQue(yLocal);
        inQueueW_.FreeTensor(wLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(AscendC::GlobalTensor<uint8_t>& dstGm, uint32_t b, uint32_t c, uint32_t h, uint32_t w,
                                    uint32_t hStart, uint32_t wStart, uint32_t tileH, uint32_t tileW)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(dstGm, yLocal, b, c, h, w, hStart, wStart, tileH, tileW);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<uint8_t> xGm_;
    AscendC::GlobalTensor<uint8_t> yGm_;
    AscendC::GlobalTensor<uint8_t> wGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t batch_ {0};
    uint32_t inputC_ {0};
    uint32_t outputC_ {0};
    uint32_t inputH_ {0};
    uint32_t inputW_ {0};
    uint32_t outputH_ {0};
    uint32_t outputW_ {0};
    uint32_t kernelH_ {0};
    uint32_t kernelW_ {0};
    int32_t padHT_ {0};
    int32_t padHB_ {0};
    int32_t padWL_ {0};
    int32_t padWR_ {0};
    uint32_t strideH_ {0};
    uint32_t strideW_ {0};
    uint32_t dilationH_ {0};
    uint32_t dilationW_ {0};
    uint32_t tileH_ {0};
    uint32_t tileW_ {0};
    float divisor_ {2.0f};
    float negSlope_ {0.01f};
    bool hasBias_ {false};
};

extern "C" __global__ __aicore__ void conv2d_divide_leaky_relu_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR weight, GM_ADDR bias, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dDivideLeakyReluCustom op;
    op.Init(x, y, weight, bias,
            tiling_data.batch, tiling_data.inputC, tiling_data.outputC,
            tiling_data.inputH, tiling_data.inputW,
            tiling_data.outputH, tiling_data.outputW,
            tiling_data.kernelH, tiling_data.kernelW,
            tiling_data.padHT, tiling_data.padHB, tiling_data.padWL, tiling_data.padWR,
            tiling_data.strideH, tiling_data.strideW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.tileH, tiling_data.tileW,
            tiling_data.divisor);
    op.Process();
}
