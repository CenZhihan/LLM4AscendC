#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv3dMinSoftmaxCustom {
public:
    __aicore__ inline KernelConv3dMinSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t inputBatch, uint32_t inputChannel, uint32_t inputDepth,
                                uint32_t inputHeight, uint32_t inputWidth,
                                uint32_t weightOutChannel, uint32_t weightInChannel,
                                uint32_t weightDepth, uint32_t weightHeight, uint32_t weightWidth,
                                uint32_t outputDepth, uint32_t outputHeight, uint32_t outputWidth,
                                uint32_t totalLength, uint32_t tileLength)
    {
        inputBatch_ = inputBatch;
        inputChannel_ = inputChannel;
        inputDepth_ = inputDepth;
        inputHeight_ = inputHeight;
        inputWidth_ = inputWidth;
        weightOutChannel_ = weightOutChannel;
        weightInChannel_ = weightInChannel;
        weightDepth_ = weightDepth;
        weightHeight_ = weightHeight;
        weightWidth_ = weightWidth;
        outputDepth_ = outputDepth;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        
        inputStrideD_ = inputChannel_ * inputHeight_ * inputWidth_;
        inputStrideH_ = inputWidth_;
        inputStrideW_ = 1;
        
        outputStrideC_ = outputHeight_ * outputWidth_;
        outputStrideH_ = outputWidth_;
        outputStrideW_ = 1;
        
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)weight);
        bGm_.SetGlobalBuffer((__gm__ float*)bias);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, weightOutChannel_ * weightInChannel_ * weightDepth_ * weightHeight_ * weightWidth_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, 1, weightOutChannel_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;
        
        CopyInConstant();
        
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;
        
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        
        CopyIn(base, curLen);
        
        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }
            
            Compute(curLen);
            CopyOut(base, curLen);
            
            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyInConstant()
    {
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        
        uint32_t weightSize = weightOutChannel_ * weightInChannel_ * weightDepth_ * weightHeight_ * weightWidth_;
        AscendC::DataCopy(wLocal, wGm_, weightSize);
        AscendC::DataCopy(bLocal, bGm_, weightOutChannel_);
        
        inQueueW_.EnQue<float>(wLocal);
        inQueueB_.EnQue<float>(bLocal);
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, yGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> bLocal = inQueueB_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
        
        uint32_t outputIdx = 0;
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t globalIdx = i + (AscendC::GetBlockIdx() * tileLen_);
            
            uint32_t n = globalIdx / outputStrideC_;
            uint32_t c = (globalIdx % outputStrideC_) / outputStrideH_;
            uint32_t h = (globalIdx % outputStrideH_) / outputStrideW_;
            uint32_t w = globalIdx % outputStrideW_;
            
            float minVal = 1e10f;
            
            for (uint32_t cd = 0; cd < weightDepth_; ++cd) {
                uint32_t inputD = cd;
                for (uint32_t ch = 0; ch < weightHeight_; ++ch) {
                    uint32_t inputH = h + ch;
                    for (uint32_t cw = 0; cw < weightWidth_; ++cw) {
                        uint32_t inputW = w + cw;
                        
                        float convVal = 0.0f;
                        for (uint32_t ic = 0; ic < weightInChannel_; ++ic) {
                            uint32_t inputIdx = n * inputStrideD_ * inputChannel_ + 
                                                ic * inputStrideD_ +
                                                inputD * inputHeight_ * inputWidth_ +
                                                inputH * inputWidth_ +
                                                inputW;
                            uint32_t weightIdx = c * weightInChannel_ * weightDepth_ * weightHeight_ * weightWidth_ +
                                                  ic * weightDepth_ * weightHeight_ * weightWidth_ +
                                                  cd * weightHeight_ * weightWidth_ +
                                                  ch * weightWidth_ +
                                                  cw;
                            convVal += xLocal[inputIdx] * wLocal[weightIdx];
                        }
                    }
                }
            }
            
            convVal += bLocal[c];
            
            if (convVal < minVal) {
                minVal = convVal;
            }
            
            yLocal[i] = minVal;
        }
        
        float maxVal = -1e10f;
        for (uint32_t i = 0; i < len; ++i) {
            if (yLocal[i] > maxVal) {
                maxVal = yLocal[i];
            }
        }
        
        float sumExp = 0.0f;
        for (uint32_t i = 0; i < len; ++i) {
            float expVal = AscendC::Exp(yLocal[i] - maxVal);
            tLocal[i] = expVal;
            sumExp += expVal;
        }
        
        for (uint32_t i = 0; i < len; ++i) {
            yLocal[i] = tLocal[i] / sumExp;
        }
        
        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueW_.FreeTensor(wLocal);
        inQueueB_.FreeTensor(bLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t inputBatch_ {0};
    uint32_t inputChannel_ {0};
    uint32_t inputDepth_ {0};
    uint32_t inputHeight_ {0};
    uint32_t inputWidth_ {0};
    uint32_t weightOutChannel_ {0};
    uint32_t weightInChannel_ {0};
    uint32_t weightDepth_ {0};
    uint32_t weightHeight_ {0};
    uint32_t weightWidth_ {0};
    uint32_t outputDepth_ {0};
    uint32_t outputHeight_ {0};
    uint32_t outputWidth_ {0};
    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    
    uint32_t inputStrideD_ {0};
    uint32_t inputStrideH_ {0};
    uint32_t inputStrideW_ {0};
    uint32_t outputStrideC_ {0};
    uint32_t outputStrideH_ {0};
    uint32_t outputStrideW_ {0};
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMinSoftmaxCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.inputBatch, tiling_data.inputChannel, tiling_data.inputDepth,
            tiling_data.inputHeight, tiling_data.inputWidth,
            tiling_data.weightOutChannel, tiling_data.weightInChannel,
            tiling_data.weightDepth, tiling_data.weightHeight, tiling_data.weightWidth,
            tiling_data.outputDepth, tiling_data.outputHeight, tiling_data.outputWidth,
            tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
