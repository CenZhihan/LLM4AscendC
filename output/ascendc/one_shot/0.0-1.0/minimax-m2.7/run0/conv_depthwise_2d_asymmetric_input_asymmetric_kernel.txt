// UB model for depthwise conv:
//   inQueueX: tileHeight * tileWidth * BUFFER_NUM
//   weightQueue: kernelSizeH * kernelSizeW * BUFFER_NUM
//   outQueueY: tileHeight * tileWidth * BUFFER_NUM
//   tmpQueue: tileHeight * tileWidth
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ELEMS_PER_BUFFER = 4096;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;
static constexpr uint32_t ALIGN_ELEMS = 256;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dAsymmetricTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::Shape *weightShape = context->GetInputShape(1);

    int32_t strideH = 1, strideW = 1;
    int32_t paddingH = 0, paddingW = 0;
    int32_t dilationH = 1, dilationW = 1;
    int32_t inChannels = 1, outChannels = 1;
    int32_t kernelSizeH = 1, kernelSizeW = 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs != nullptr) {
        auto getAttrInt = [&](int idx) -> int32_t {
            const void* ptr = attrs->GetAttrPointer<void>(idx);
            return ptr ? *(static_cast<const int32_t*>(ptr)) : 1;
        };
        strideH = getAttrInt(0);
        strideW = getAttrInt(1);
        paddingH = getAttrInt(2);
        paddingW = getAttrInt(3);
        dilationH = getAttrInt(4);
        dilationW = getAttrInt(5);
        inChannels = getAttrInt(6);
        outChannels = getAttrInt(7);
        kernelSizeH = getAttrInt(8);
        kernelSizeW = getAttrInt(9);
    }

    int32_t inputHeight = static_cast<int32_t>(inputShape->GetShape()[2]);
    int32_t inputWidth = static_cast<int32_t>(inputShape->GetShape()[3]);
    int32_t batchSize = static_cast<int32_t>(inputShape->GetShape()[0]);

    int32_t outputHeight = (inputHeight + 2 * paddingH - dilationH * (kernelSizeH - 1) - 1) / strideH + 1;
    int32_t outputWidth = (inputWidth + 2 * paddingW - dilationW * (kernelSizeW - 1) - 1) / strideW + 1;

    int32_t totalLength = batchSize * outChannels * outputHeight * outputWidth;
    int32_t channelSize = outputHeight * outputWidth;

    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (static_cast<uint32_t>(totalLength) < tileLen) {
        tileLen = AlignDown(static_cast<uint32_t>(totalLength), ALIGN_ELEMS);
        if (tileLen == 0) tileLen = static_cast<uint32_t>(totalLength);
    }

    uint32_t numTiles = CeilDiv(static_cast<uint32_t>(totalLength), tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_paddingH(paddingH);
    tiling.set_paddingW(paddingW);
    tiling.set_dilationH(dilationH);
    tiling.set_dilationW(dilationW);
    tiling.set_inChannels(inChannels);
    tiling.set_outChannels(outChannels);
    tiling.set_kernelSizeH(kernelSizeH);
    tiling.set_kernelSizeW(kernelSizeW);
    tiling.set_inputHeight(inputHeight);
    tiling.set_inputWidth(inputWidth);
    tiling.set_outputHeight(outputHeight);
    tiling.set_outputWidth(outputWidth);
    tiling.set_channelSize(channelSize);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_KERNEL_SIZE = 64;

class KernelDepthwiseConv2dAsymmetric {
public:
    __aicore__ inline KernelDepthwiseConv2dAsymmetric() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR weight, GM_ADDR y,
        uint32_t totalLength, uint32_t tileLength,
        int32_t strideH, int32_t strideW,
        int32_t paddingH, int32_t paddingW,
        int32_t dilationH, int32_t dilationW,
        int32_t inChannels, int32_t outChannels,
        int32_t kernelSizeH, int32_t kernelSizeW,
        int32_t inputHeight, int32_t inputWidth,
        int32_t outputHeight, int32_t outputWidth,
        int32_t channelSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        strideH_ = strideH;
        strideW_ = strideW;
        paddingH_ = paddingH;
        paddingW_ = paddingW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        inChannels_ = inChannels;
        outChannels_ = outChannels;
        kernelSizeH_ = kernelSizeH;
        kernelSizeW_ = kernelSizeW;
        inputHeight_ = inputHeight;
        inputWidth_ = inputWidth;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        channelSize_ = channelSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        kernelSizeTotal_ = kernelSizeH_ * kernelSizeW_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(weightQueue_, BUFFER_NUM, kernelSizeTotal_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> accLocal = accQueue_.AllocTensor<float>();

        uint32_t numChannels = len / channelSize_;
        uint32_t hTile = outputHeight_;
        uint32_t wTile = outputWidth_;

        for (uint32_t c = 0; c < numChannels; ++c) {
            uint32_t channelOffset = c * channelSize_;

            uint32_t weightOffset = c * kernelSizeTotal_;

            AscendC::LocalTensor<float> weightLocal = weightQueue_.AllocTensor<float>();
            AscendC::DataCopy(weightLocal, weightGm_[weightOffset], kernelSizeTotal_);
            weightQueue_.EnQue(weightLocal);

            for (uint32_t outH = 0; outH < hTile; ++outH) {
                for (uint32_t outW = 0; outW < wTile; ++outW) {
                    float sum = 0.0f;
                    int32_t inBaseH = outH * strideH_ - paddingH_;
                    int32_t inBaseW = outW * strideW_ - paddingW_;

                    for (int32_t kH = 0; kH < kernelSizeH_; ++kH) {
                        int32_t inH = inBaseH + kH * dilationH_;
                        for (int32_t kW = 0; kW < kernelSizeW_; ++kW) {
                            int32_t inW = inBaseW + kW * dilationW_;

                            if (inH >= 0 && inH < inputHeight_ && inW >= 0 && inW < inputWidth_) {
                                uint32_t inputIdx = channelOffset + outH * wTile + outW;
                                uint32_t kernelIdx = kH * kernelSizeW_ + kW;
                                uint32_t weightIdx = kernelIdx;
                                sum += xLocal[inputIdx] * weightLocal[weightIdx];
                            }
                        }
                    }
                    accLocal[channelOffset + outH * wTile + outW] = sum;
                }
            }

            AscendC::LocalTensor<float> wDeq = weightQueue_.DeQue<float>();
            weightQueue_.FreeTensor(wDeq);
        }

        AscendC::DataCopy(yLocal, accLocal, len);

        accQueue_.FreeTensor(accLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> weightQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t strideH_ {1};
    int32_t strideW_ {1};
    int32_t paddingH_ {0};
    int32_t paddingW_ {0};
    int32_t dilationH_ {1};
    int32_t dilationW_ {1};
    int32_t inChannels_ {1};
    int32_t outChannels_ {1};
    int32_t kernelSizeH_ {1};
    int32_t kernelSizeW_ {1};
    int32_t inputHeight_ {1};
    int32_t inputWidth_ {1};
    int32_t outputHeight_ {1};
    int32_t outputWidth_ {1};
    int32_t channelSize_ {1};
    int32_t kernelSizeTotal_ {1};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelDepthwiseConv2dAsymmetric op;
    op.Init(
        x, weight, y,
        tiling_data.totalLength, tiling_data.tileLength,
        tiling_data.strideH, tiling_data.strideW,
        tiling_data.paddingH, tiling_data.paddingW,
        tiling_data.dilationH, tiling_data.dilationW,
        tiling_data.inChannels, tiling_data.outChannels,
        tiling_data.kernelSizeH, tiling_data.kernelSizeW,
        tiling_data.inputHeight, tiling_data.inputWidth,
        tiling_data.outputHeight, tiling_data.outputWidth,
        tiling_data.channelSize);
    op.Process();
}
