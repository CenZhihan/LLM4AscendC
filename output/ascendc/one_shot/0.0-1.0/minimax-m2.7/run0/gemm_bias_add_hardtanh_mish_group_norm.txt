#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight,
                                GM_ADDR lin_bias, GM_ADDR bias,
                                GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t numGroups,
                                uint32_t inFeatures, uint32_t outFeatures)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        numGroups_ = numGroups;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        groupSize_ = outFeatures / numGroups;
        kTile_ = 64;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outFeatures * inFeatures);
        linBiasGm_.SetGlobalBuffer((__gm__ float*)lin_bias, outFeatures);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, outFeatures);
        gnGammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, outFeatures);
        gnBetaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, outFeatures);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * kTile_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLen_ * kTile_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueMish_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueMean_, 1, numGroups_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueVar_, 1, numGroups_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueTmp_, 1, groupSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t tileCount = (tileLen_ == 0) ? 1u : (totalLen_ + tileLen_ - 1u) / tileLen_;
        const uint32_t tilesPerBlock = (tileCount + blockDim_ - 1u) / blockDim_;
        const uint32_t blockIdx = GetBlockIdx();

        uint32_t pos = blockIdx * tilesPerBlock;
        uint32_t base = pos * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        pos_ = pos;
        base_ = base;
        curLen_ = curLen;

        for (uint32_t t = 0; t < tilesPerBlock && base < totalLen_; ++t) {
            Compute();
            CopyOut();

            ++pos_;
            base += tileLen_;
            curLen_ = totalLen_ - base;
            if (curLen_ > tileLen_) curLen_ = tileLen_;
        }
    }

private:
    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Muls(yLocal, yLocal, 0.0f, curLen_);

        for (uint32_t k = 0; k < inFeatures_; k += kTile_) {
            uint32_t curK = inFeatures_ - k;
            if (curK > kTile_) curK = kTile_;

            AscendC::LocalTensor<float> xL = inQueueX_.AllocTensor<float>();
            for (uint32_t o = 0; o < curLen_; ++o) {
                uint32_t gmOffset = (pos_ + o) * inFeatures_ + k;
                AscendC::DataCopy(xL[o * kTile_], xGm_[gmOffset], curK);
            }
            inQueueX_.EnQue(xL);

            AscendC::LocalTensor<float> wL = inQueueW_.AllocTensor<float>();
            for (uint32_t o = 0; o < curLen_; ++o) {
                uint32_t gmOffset = (pos_ + o) * inFeatures_ + k;
                AscendC::DataCopy(wL[o * kTile_], weightGm_[gmOffset], curK);
            }
            inQueueW_.EnQue(wL);

            AscendC::LocalTensor<float> xLT = inQueueX_.DeQue<float>();
            AscendC::LocalTensor<float> wLT = inQueueW_.DeQue<float>();

            AscendC::MultiplyAdd(yLocal, xLT, wLT, curLen_ * kTile_);

            inQueueX_.FreeTensor(xLT);
            inQueueW_.FreeTensor(wLT);
        }

        AscendC::Adds(yLocal, yLocal, linBiasGm_[pos_ * curLen_], curLen_);
        AscendC::Adds(yLocal, yLocal, biasGm_[pos_], curLen_);

        AscendC::Hardtanh(yLocal, yLocal, -1.0f, 1.0f, curLen_);

        AscendC::LocalTensor<float> mishLocal = tmpQueueMish_.AllocTensor<float>();
        AscendC::Adds(mishLocal, yLocal, 0.0f, curLen_);
        AscendC::Exp(mishLocal, mishLocal, curLen_);
        AscendC::Adds(mishLocal, mishLocal, 1.0f, curLen_);
        AscendC::Ln(mishLocal, mishLocal, curLen_);
        AscendC::Tanh(mishLocal, mishLocal, curLen_);
        AscendC::Mul(mishLocal, yLocal, mishLocal, curLen_);

        AscendC::LocalTensor<float> meanLocal = tmpQueueMean_.AllocTensor<float>();
        AscendC::LocalTensor<float> varLocal = tmpQueueVar_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueueTmp_.AllocTensor<float>();

        for (uint32_t g = 0; g < numGroups_; ++g) {
            uint32_t gOffset = g * groupSize_;
            AscendC::ReduceMean(meanLocal[g], mishLocal[gOffset], groupSize_);
        }

        for (uint32_t g = 0; g < numGroups_; ++g) {
            uint32_t gOffset = g * groupSize_;
            AscendC::Sub(tmpLocal, mishLocal[gOffset], meanLocal[g], groupSize_);
            AscendC::Muls(tmpLocal, tmpLocal, tmpLocal, groupSize_);
            AscendC::ReduceSum(varLocal[g], tmpLocal, groupSize_);
            AscendC::Muls(varLocal[g], varLocal[g], 1.0f / groupSize_, 1);
            AscendC::Adds(varLocal[g], varLocal[g], 1e-5f, 1);
            AscendC::Sqrt(varLocal[g], varLocal[g], 1);
            AscendC::Sub(tmpLocal, mishLocal[gOffset], meanLocal[g], groupSize_);
            AscendC::Div(tmpLocal, tmpLocal, varLocal[g], groupSize_);
            AscendC::Muls(tmpLocal, tmpLocal, gnGammaGm_[g], groupSize_);
            AscendC::Adds(tmpLocal, tmpLocal, gnBetaGm_[g], groupSize_);
            AscendC::DataCopy(yLocal[gOffset], tmpLocal, groupSize_);
        }

        outQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[base_], yLocal, curLen_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMish_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMean_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueVar_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueTmp_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> linBiasGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> gnGammaGm_;
    AscendC::GlobalTensor<float> gnBetaGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
    uint32_t numGroups_ {256};
    uint32_t inFeatures_ {8192};
    uint32_t outFeatures_ {8192};
    uint32_t groupSize_ {32};
    uint32_t kTile_ {64};
    uint32_t pos_ {0};
    uint32_t base_ {0};
    uint32_t curLen_ {0};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias,
    GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmBiasAddHardtanhMishGroupNormCustom op;
    op.Init(x, y, weight, lin_bias, bias, gn_gamma, gn_beta,
            tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.blockDim, tiling_data.numGroups,
            tiling_data.inFeatures, tiling_data.outFeatures);
    op.Process();
}
