#include "cumsum_reverse_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumReverseCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    const std::vector<int64_t> &shape = inputShape->GetOriginShape().GetDims();
    
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t dim = attrs->GetAttrPointer<int32_t>(0);
    
    int32_t normalizedDim = dim < 0 ? dim + static_cast<int32_t>(shape.size()) : dim;
    if (normalizedDim < 0 || normalizedDim >= static_cast<int32_t>(shape.size())) {
        normalizedDim = static_cast<int32_t>(shape.size()) - 1;
    }
    
    uint32_t dimSize = static_cast<uint32_t>(shape[normalizedDim]);
    
    uint32_t outerSize = 1;
    for (size_t i = 0; i < static_cast<size_t>(normalizedDim); ++i) {
        outerSize *= static_cast<uint32_t>(shape[i]);
    }
    
    uint32_t totalLength = 1;
    for (int64_t d : shape) {
        totalLength *= static_cast<uint32_t>(d);
    }
    
    // Tile along dim dimension
    uint32_t maxTile = UB_BUDGET_BYTES / (3 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > dimSize) maxTile = AlignDown(dimSize, ALIGN_ELEMS);
    
    uint32_t tileLen = maxTile;
    if (dimSize < tileLen) {
        tileLen = dimSize;
    }
    
    uint32_t numTiles = CeilDiv(dimSize, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;
    
    context->SetBlockDim(blockDim);
    
    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(normalizedDim);
    tiling.set_dimSize(dimSize);
    tiling.set_outerSize(outerSize);
    
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
    
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCumsumReverseCustom {
public:
    __aicore__ inline KernelCumsumReverseCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                               uint32_t totalLength, uint32_t tileLength,
                               int32_t dim, uint32_t dimSize, uint32_t outerSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLen;
        dim_ = dim;
        dimSize_ = dimSize;
        outerSize_ = outerSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Need 3 buffers: flip buffer, cumsum result, final result
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(flipQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || dimSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block handles one outer slice
        const uint32_t blocksPerOuter = 1;
        uint32_t outerIdx = blockIdx;
        if (outerIdx >= outerSize_) return;

        // Process along dim dimension with tiling
        uint32_t strideTiles = blockNum;
        uint32_t tileOffset = 0;

        while (tileOffset < dimSize_) {
            uint32_t curTileLen = tileLen_;
            if (tileOffset + curTileLen > dimSize_) {
                curTileLen = dimSize_ - tileOffset;
            }

            // CopyIn: load from original order
            CopyIn(outerIdx, tileOffset, curTileLen);
            Compute(outerIdx, tileOffset, curTileLen);
            CopyOut(outerIdx, tileOffset, curTileLen);

            tileOffset += tileLen_ * strideTiles;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t outerIdx, uint32_t offsetInDim, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        
        // Calculate GM offset for this slice
        uint32_t gmOffset = outerIdx * dimSize_ + offsetInDim;
        
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t outerIdx, uint32_t offsetInDim, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> flipLocal = flipQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // Reverse the segment: flipLocal[i] = xLocal[len-1-i]
        uint32_t actualLen = len;
        uint32_t gmOffset = outerIdx * dimSize_ + offsetInDim;
        
        // Load the full dim segment for proper cumsum
        uint32_t fullDimOffset = outerIdx * dimSize_;
        
        // For tiled processing, we need to handle the full reverse cumsum
        // Load entire dim into flip buffer (reversed)
        for (uint32_t i = 0; i < actualLen; ++i) {
            uint32_t srcIdx = offsetInDim + actualLen - 1 - i;
            uint32_t srcOffset = fullDimOffset + srcIdx;
            // Manual copy with offset
            tLocal.SetValue(i, xGm_.GetValue(srcOffset));
        }
        
        // Now tLocal has reversed data, do cumsum on it
        AscendC::CumSum(flipLocal, tLocal, actualLen, 0);
        
        // Reverse back to get reverse cumsum result
        for (uint32_t i = 0; i < actualLen; ++i) {
            yLocal.SetValue(i, flipLocal.GetValue(actualLen - 1 - i));
        }

        tmpQueue_.FreeTensor(tLocal);
        flipQueue_.FreeTensor(flipLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t outerIdx, uint32_t offsetInDim, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        uint32_t gmOffset = outerIdx * dimSize_ + offsetInDim;
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> flipQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ {0};
    uint32_t dimSize_ {0};
    uint32_t outerSize_ {0};
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverseCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, 
            tiling_data.dim, tiling_data.dimSize, tiling_data.outerSize);
    op.Process();
}
