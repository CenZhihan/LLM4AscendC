#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dSubtractHardSwishMaxPoolMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractHardSwishMaxPoolMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t batchSize, uint32_t outputHeight, uint32_t outputWidth,
                                uint32_t outputChannel, uint32_t kernelSize,
                                uint32_t poolKernelSize, float subtractValue, int32_t hasBias)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        batchSize_ = batchSize;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        outputChannel_ = outputChannel;
        kernelSize_ = kernelSize;
        poolKernelSize_ = poolKernelSize;
        subtractValue_ = subtractValue;
        hasBias_ = hasBias;
        
        // Calculate dimensions
        inHeight_ = outputHeight_ * poolKernelSize_ + kernelSize_ - 1;
        inWidth_ = outputWidth_ * poolKernelSize_ + kernelSize_ - 1;
        
        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize_ * (outputChannel_ / 8) * inHeight_ * inWidth_ * 8);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, outputChannel_ * (outputChannel_ / 8) * kernelSize_ * kernelSize_ * 8);
        
        if (hasBias_ && bias != nullptr) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, outputChannel_);
            hasBias_ = 1;
        } else {
            hasBias_ = 0;
        }

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(weightQueue_, 1, kernelSize_ * kernelSize_ * sizeof(float));
        pipe_.InitBuffer(biasQueue_, 1, sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue3_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        ComputeAndProcess(base, curLen);
    }

private:
    __aicore__ inline void ComputeAndProcess(uint32_t globalOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        
        // Calculate position in output tensor
        uint32_t n = globalOffset / (outputChannel_ * outputHeight_ * outputWidth_);
        uint32_t remain = globalOffset % (outputChannel_ * outputHeight_ * outputWidth_);
        uint32_t c = remain / (outputHeight_ * outputWidth_);
        uint32_t ph = (remain % (outputHeight_ * outputWidth_)) / outputWidth_;
        uint32_t pw = remain % outputWidth_;
        
        // Calculate corresponding pool window in conv output
        uint32_t convH = ph * poolKernelSize_;
        uint32_t convW = pw * poolKernelSize_;
        
        // Compute conv for this output position
        float result = 0.0f;
        for (uint32_t ci = 0; ci < outputChannel_; ++ci) {
            for (uint32_t kh = 0; kh < kernelSize_; ++kh) {
                for (uint32_t kw = 0; kw < kernelSize_; ++kw) {
                    uint32_t inH = convH + kh;
                    uint32_t inW = convW + kw;
                    uint32_t inIdx = n * outputChannel_ * inHeight_ * inWidth_ + 
                                    (outputChannel_ / 8) * inHeight_ * inWidth_ + 
                                    ci * inHeight_ * inWidth_ + 
                                    inH * inWidth_ + inW;
                    
                    uint32_t wIdx = ci * (outputChannel_ / 8) * kernelSize_ * kernelSize_ + 
                                   (outputChannel_ / 8) * kernelSize_ * kernelSize_ +
                                   kh * kernelSize_ + kw;
                    
                    if (inIdx < batchSize_ * (outputChannel_ / 8) * inHeight_ * inWidth_ * 8 &&
                        wIdx < outputChannel_ * (outputChannel_ / 8) * kernelSize_ * kernelSize_ * 8) {
                        float xVal = xGm_.GetValue(inIdx);
                        float wVal = weightGm_.GetValue(wIdx);
                        result += xVal * wVal;
                    }
                }
            }
            
            if (hasBias_ == 1) {
                float bVal = biasGm_.GetValue(ci);
                result += bVal;
            }
            
            // Apply subtract and hard_swish
            float afterSub = result - subtractValue_;
            float hardSwish = afterSub * AscendC::AscendClamp((afterSub + 3.0f) / 6.0f, 0.0f, 1.0f);
            
            // Apply mish after maxpool
            float expX = AscendC::Exp(hardSwish);
            float softplus = AscendC::Log(1.0f + expX);
            float tanhSoftplus = (exp(softplus) - exp(-softplus)) / (exp(softplus) + exp(-softplus));
            float mish = hardSwish * tanhSoftplus;
            
            yLocal.SetValue(ci, mish);
        }

        outQueueY_.EnQue(yLocal);
        
        // Copy out
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[globalOffset], yOut, len);
        outQueueY_.FreeTensor(yOut);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> biasQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue3_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t batchSize_ {0};
    uint32_t outputHeight_ {0};
    uint32_t outputWidth_ {0};
    uint32_t outputChannel_ {0};
    uint32_t kernelSize_ {0};
    uint32_t poolKernelSize_ {0};
    uint32_t inHeight_ {0};
    uint32_t inWidth_ {0};
    float subtractValue_ {0.5f};
    int32_t hasBias_ {0};
};

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(GM_ADDR x, GM_ADDR y, GM_ADDR weight, GM_ADDR bias, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractHardSwishMaxPoolMishCustom op;
    op.Init(x, y, weight, bias, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.batchSize, tiling_data.outputHeight, tiling_data.outputWidth,
            tiling_data.outputChannel, tiling_data.kernelSize, tiling_data.poolKernelSize,
            tiling_data.subtractValue, tiling_data.hasBias);
    op.Process();
}
