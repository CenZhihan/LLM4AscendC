__aicore__ inline void Process()
{
    if (M_ == 0 || K_ == 0 || N_ == 0) return;
    const uint32_t blockIdx = AscendC::GetBlockIdx();
    const uint32_t blockNum = AscendC::GetBlockNum();

    // Compute block grid dimensions
    const uint32_t blockDimX = (M_ + tileM_ - 1) / tileM_;
    const uint32_t blockDimY = (N_ + tileN_ - 1) / tileN_;
    // Total blocks should equal blockNum passed from host; but we can compute anyway.
    const uint32_t idxX = blockIdx / blockDimY;
    const uint32_t idxY = blockIdx % blockDimY;

    const uint32_t startM = idxX * tileM_;
    const uint32_t startN = idxY * tileN_;
    if (startM >= M_ || startN >= N_) return;

    const uint32_t lenM = (startM + tileM_ <= M_) ? tileM_ : M_ - startM;
    const uint32_t lenN = (startN + tileN_ <= N_) ? tileN_ : N_ - startN;

    // Allocate output tile and zero-initialize
    auto outTile = outQueueC_.AllocTensor<float>();
    AscendC::Sets(outTile, 0.0f, lenM * lenN);

    // Loop over K dimension
    for (uint32_t k = 0; k < K_; k += tileK_) {
        uint32_t kLen = (k + tileK_ <= K_) ? tileK_ : K_ - k;

        // Copy A tile
        auto aTile = inQueueA_.AllocTensor<float>();
        for (uint32_t i = 0; i < lenM; ++i) {
            uint32_t srcOffset = (startM + i) * K_ + k;
            AscendC::DataCopy(aTile[i * kLen], A_gm_[srcOffset], kLen);
        }
        inQueueA_.EnQue(aTile);

        // Copy B tile
        auto bTile = inQueueB_.AllocTensor<float>();
        for (uint32_t i = 0; i < kLen; ++i) {
            uint32_t srcOffset = (k + i) * N_ + startN;
            AscendC::DataCopy(bTile[i * lenN], B_gm_[srcOffset], lenN);
        }
        inQueueB_.EnQue(bTile);

        // Compute partial result
        auto tmpTile = tmpQueue_.AllocTensor<float>();
        AscendC::Sets(tmpTile, 0.0f, lenM * lenN);

        // Multiply and accumulate
        for (uint32_t i = 0; i < lenM; ++i) {
            for (uint32_t j = 0; j < lenN; ++j) {
                float sum = 0.0f;
                for (uint32_t kk = 0; kk < kLen; ++kk) {
                    sum += aTile[i * kLen + kk] * bTile[kk * lenN + j];
                }
                tmpTile[i * lenN + j] = sum;
            }
        }

        // Accumulate into output tile
        AscendC::Add(outTile, outTile, tmpTile, lenM * lenN);

        // Free buffers for next iteration
        inQueueA_.FreeTensor(aTile);
        inQueueB_.FreeTensor(bTile);
        tmpQueue_.FreeTensor(tmpTile);
    }

    // Enqueue output tile and copy to global memory
    outQueueC_.EnQue(outTile);
    auto outTile2 = outQueueC_.DeQue<float>();
    for (uint32_t i = 0; i < lenM; ++i) {
        uint32_t dstOffset = (startM + i) * N_ + startN;
        AscendC::DataCopy(C_gm_[dstOffset], outTile2[i * lenN], lenN);
    }
    outQueueC_.FreeTensor(outTile2);
}

ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;
    const gert::Shape* aShape = context->GetInputShape(0);
    const gert::Shape* bShape = context->GetInputShape(1);
    const uint32_t M = static_cast<uint32_t>(aShape->GetOriginShape().GetShapeVector()[0]);
    const uint32_t K = static_cast<uint32_t>(aShape->GetOriginShape().GetShapeVector()[1]);
    const uint32_t N = static_cast<uint32_t>(bShape->GetOriginShape().GetShapeVector()[1]);

    const uint32_t UB_FLOATS = UB_BUDGET_BYTES / sizeof(float);
    const uint32_t tileN = TILE_N;
    const uint32_t tileK = TILE_K;

    // Compute tileM based on UB budget
    uint32_t tileM = (UB_FLOATS - tileK * tileN) / (tileK + tileN);
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;
    if (M < tileM) tileM = M;
    else tileM = AlignDown(tileM, ALIGN_ELEMS); // ensure alignment

    const uint32_t blockDimX = CeilDiv(M, tileM);
    const uint32_t blockDimY = CeilDiv(N, tileN);
    const uint32_t blockDim = blockDimX * blockDimY;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        const uint32_t blockIdx = AscendC::GetBlockIdx();
        const uint32_t blockNum = AscendC::GetBlockNum();

        // Compute block grid dimensions
        const uint32_t blockDimX = (M_ + tileM_ - 1) / tileM_;
        const uint32_t blockDimY = (N_ + tileN_ - 1) / tileN_;

        const uint32_t idxX = blockIdx / blockDimY;
        const uint32_t idxY = blockIdx % blockDimY;

        const uint32_t startM = idxX * tileM_;
        const uint32_t startN = idxY * tileN_;

        if (startM >= M_ || startN >= N_) return;

        const uint32_t lenM = (startM + tileM_ <= M_) ? tileM_ : M_ - startM;
        const uint32_t lenN = (startN + tileN_ <= N_) ? tileN_ : N_ - startN;

        // Allocate output tile and zero-initialize
        auto outTile = outQueueC_.AllocTensor<float>();
        AscendC::Sets(outTile, 0.0f, lenM * lenN);

        // Loop over K dimension
        for (uint32_t k = 0; k < K_; k += tileK_) {
            uint32_t kLen = (k + tileK_ <= K_) ? tileK_ : K_ - k;

            // Copy A tile
            auto aTile = inQueueA_.AllocTensor<float>();
            for (uint32_t i = 0; i < lenM; ++i) {
                uint32_t srcOffset = (startM + i) * K_ + k;
                AscendC::DataCopy(aTile[i * kLen], A_gm_[srcOffset], kLen);
            }
            inQueueA_.EnQue(aTile);

            // Copy B tile
            auto bTile = inQueueB_.AllocTensor<float>();
            for (uint32_t i = 0; i < kLen; ++i) {
                uint32_t srcOffset = (k + i) * N_ + startN;
                AscendC::DataCopy(bTile[i * lenN], B_gm_[srcOffset], lenN);
            }
            inQueueB_.EnQue(bTile);

            // Compute partial result
            auto tmpTile = tmpQueue_.AllocTensor<float>();
            AscendC::Sets(tmpTile, 0.0f, lenM * lenN);

            // Multiply-accumulate
            for (uint32_t i = 0; i < lenM; ++i) {
                for (uint32_t j = 0; j < lenN; ++j) {
                    float sum = 0.0f;
                    for (uint32_t kk = 0; kk < kLen; ++kk) {
                        sum += aTile[i * kLen + kk] * bTile[kk * lenN + j];
                    }
                    tmpTile[i * lenN + j] = sum;
                }
            }

            // Accumulate into output tile
            AscendC::Add(outTile, outTile, tmpTile, lenM * lenN);

            // Free temporary buffers
            inQueueA_.FreeTensor(aTile);
            inQueueB_.FreeTensor(bTile);
            tmpQueue_.FreeTensor(tmpTile);
        }

        // Copy output tile to global memory
        outQueueC_.EnQue(outTile);
        auto outTile2 = outQueueC_.DeQue<float>();
        for (uint32_t i = 0; i < lenM; ++i) {
            uint32_t dstOffset = (startM + i) * N_ + startN;
            AscendC::DataCopy(C_gm_[dstOffset], outTile2[i * lenN], lenN);
        }
        outQueueC_.FreeTensor(outTile2);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t M_ = 0;
    uint32_t K_ = 0;
    uint32_t N_ = 0;
    uint32_t tileM_ = 0;
    uint32_t tileN_ = 0;
    uint32_t tileK_ = 0;
};
