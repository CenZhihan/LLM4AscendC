host_operator_src="""

#include "max_pooling_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling2dCustomTilingData tiling;
    const gert::Shape *inputShape = context->GetInputShape(0);
    
    int32_t batchSize = static_cast<int32_t>(inputShape->GetShape().at(0));
    int32_t channels = static_cast<int32_t>(inputShape->GetShape().at(1));
    int32_t height = static_cast<int32_t>(inputShape->GetShape().at(2));
    int32_t width = static_cast<int32_t>(inputShape->GetShape().at(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = static_cast<int32_t>(attrs->GetAttrPointer<int64_t>(0)[0]);
    int32_t stride = static_cast<int32_t>(attrs->GetAttrPointer<int64_t>(1)[0]);
    int32_t padding = static_cast<int32_t>(attrs->GetAttrPointer<int64_t>(2)[0]);

    int32_t outputHeight = (height + 2 * padding - kernelSize) / stride + 1;
    int32_t outputWidth = (width + 2 * padding - kernelSize) / stride + 1;
    int32_t outputSize = batchSize * channels * outputHeight * outputWidth;

    uint32_t totalLength = static_cast<uint32_t>(outputSize);
    
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_batchSize(batchSize);
    tiling.set_channels(channels);
    tiling.set_height(height);
    tiling.set_width(width);
    tiling.set_outputHeight(outputHeight);
    tiling.set_outputWidth(outputWidth);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_KERNEL_SIZE = 16;

class KernelMaxPooling2dCustom {
public:
    __aicore__ inline KernelMaxPooling2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                               uint32_t totalLength, uint32_t tileLength,
                               int32_t kernelSize, int32_t stride, int32_t padding,
                               int32_t batchSize, int32_t channels, int32_t height, int32_t width,
                               int32_t outputHeight, int32_t outputWidth)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        batchSize_ = batchSize;
        channels_ = channels;
        height_ = height;
        width_ = width;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        
        inputSize_ = batchSize_ * channels_ * height_ * width_;
        
        xGm_.SetGlobalBuffer((__gm__ float*)x, inputSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        
        uint32_t inputTileSize = tileLen_ * kernelSize_ * kernelSize_;
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputTileSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(maxQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t outputOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        
        int32_t outIdx = static_cast<int32_t>(outputOffset);
        int32_t n = outIdx / (channels_ * outputHeight_ * outputWidth_);
        int32_t remain = outIdx % (channels_ * outputHeight_ * outputWidth_);
        int32_t c = remain / (outputHeight_ * outputWidth_);
        int32_t outH = (remain % (outputHeight_ * outputWidth_)) / outputWidth_;
        int32_t outW = remain % outputWidth_;
        
        int32_t inHStart = outH * stride_ - padding_;
        int32_t inWStart = outW * stride_ - padding_;
        
        int32_t copyLen = 0;
        for (int32_t kh = 0; kh < kernelSize_; ++kh) {
            for (int32_t kw = 0; kw < kernelSize_; ++kw) {
                int32_t inH = inHStart + kh;
                int32_t inW = inWStart + kw;
                
                if (inH >= 0 && inH < height_ && inW >= 0 && inW < width_) {
                    for (int32_t i = 0; i < len; ++i) {
                        int32_t idx = outIdx + i;
                        int32_t n_i = idx / (channels_ * outputHeight_ * outputWidth_);
                        int32_t rem_i = idx % (channels_ * outputHeight_ * outputWidth_);
                        int32_t c_i = rem_i / (outputHeight_ * outputWidth_);
                        int32_t outH_i = (rem_i % (outputHeight_ * outputWidth_)) / outputWidth_;
                        int32_t outW_i = rem_i % outputWidth_;
                        
                        int32_t inH_i = outH_i * stride_ - padding_ + kh;
                        int32_t inW_i = outW_i * stride_ - padding_ + kw;
                        
                        int32_t inputIdx = n_i * channels_ * height_ * width_ + c_i * height_ * width_ + inH_i * width_ + inW_i;
                        
                        if (kh == 0 && kw == 0) {
                            xLocal.SetValue(i, xGm_.GetValue(inputIdx));
                        } else {
                            float currentVal = xLocal.GetValue(i);
                            float newVal = xGm_.GetValue(inputIdx);
                            xLocal.SetValue(i, currentVal > newVal ? currentVal : newVal);
                        }
                    }
                }
            }
        }
        
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        
        AscendC::Copy(yLocal, xLocal, len);
        
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> maxQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t kernelSize_ {2};
    int32_t stride_ {2};
    int32_t padding_ {0};
    int32_t batchSize_ {1};
    int32_t channels_ {1};
    int32_t height_ {1};
    int32_t width_ {1};
    int32_t outputHeight_ {1};
    int32_t outputWidth_ {1};
    uint32_t inputSize_ {0};
};

extern "C" __global__ __aicore__ void max_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling2dCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, 
           tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
           tiling_data.batchSize, tiling_data.channels, tiling_data.height, tiling_data.width,
           tiling_data.outputHeight, tiling_data.outputWidth);
    op.Process();
}
