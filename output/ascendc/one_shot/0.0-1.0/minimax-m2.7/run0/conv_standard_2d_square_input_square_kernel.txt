#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t batchSize, uint32_t outputChannel,
                                uint32_t outputHeight, uint32_t outputWidth,
                                uint32_t inputChannel, uint32_t inputHeight,
                                uint32_t inputWidth, uint32_t kernelHeight,
                                uint32_t kernelWidth, uint32_t strideH,
                                uint32_t strideW, uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t group, uint32_t tileLength)
    {
        batchSize_ = batchSize;
        outputChannel_ = outputChannel;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        inputChannel_ = inputChannel;
        inputHeight_ = inputHeight;
        inputWidth_ = inputWidth;
        kernelHeight_ = kernelHeight;
        kernelWidth_ = kernelWidth;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        group_ = group;
        tileLength_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inputChannel * inputHeight * inputWidth);
        wGm_.SetGlobalBuffer((__gm__ float*)w, outputChannel * inputChannel * kernelHeight * kernelWidth);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * outputChannel * outputHeight * outputWidth);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLength_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLength_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t totalOutputElems = batchSize_ * outputChannel_ * outputHeight_ * outputWidth_;
        if (totalOutputElems == 0 || tileLength_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLength_;
        uint32_t baseIdx = blockIdx * tileLength_;
        if (baseIdx >= totalOutputElems) return;

        uint32_t curLen = totalOutputElems - baseIdx;
        if (curLen > tileLength_) curLen = tileLength_;

        CopyIn(baseIdx, curLen);

        while (true) {
            const uint32_t nextBase = baseIdx + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalOutputElems) {
                nextLen = totalOutputElems - nextBase;
                if (nextLen > tileLength_) nextLen = tileLength_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);

            const uint32_t outBaseGm = baseIdx;
            CopyOut(outBaseGm, curLen);

            if (nextLen == 0) break;
            baseIdx = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t outputIdx, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            const uint32_t outIdx = outputIdx + i;
            xLocal[i] = 0.0f;
            wLocal[i] = 0.0f;
        }

        inQueueX_.EnQue(xLocal);
        inQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal1 = tmpQueue1_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal2 = tmpQueue2_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            const uint32_t outIdx = (AscendC::GetBlockIdx() * tileLength_ + i);
            yLocal[i] = 0.0f;
        }

        for (uint32_t b = 0; b < batchSize_; ++b) {
            for (uint32_t oc = 0; oc < outputChannel_; ++oc) {
                for (uint32_t oh = 0; oh < outputHeight_; ++oh) {
                    for (uint32_t ow = 0; ow < outputWidth_; ++ow) {
                        uint32_t outLinearIdx = b * outputChannel_ * outputHeight_ * outputWidth_ +
                                                 oc * outputHeight_ * outputWidth_ +
                                                 oh * outputWidth_ + ow;

                        if (outLinearIdx >= AscendC::GetBlockIdx() * tileLength_ &&
                            outLinearIdx < AscendC::GetBlockIdx() * tileLength_ + len) {
                            float sum = 0.0f;

                            for (uint32_t ic = 0; ic < inputChannel_; ++ic) {
                                for (uint32_t kh = 0; kh < kernelHeight_; ++kh) {
                                    for (uint32_t kw = 0; kw < kernelWidth_; ++kw) {
                                        int32_t ih = oh * strideH_ - padH_ + kh * dilationH_;
                                        int32_t iw = ow * strideW_ - padW_ + kw * dilationW_;

                                        if (ih >= 0 && ih < static_cast<int32_t>(inputHeight_) &&
                                            iw >= 0 && iw < static_cast<int32_t>(inputWidth_)) {
                                            uint32_t xIdx = b * inputChannel_ * inputHeight_ * inputWidth_ +
                                                            ic * inputHeight_ * inputWidth_ +
                                                            ih * inputWidth_ + iw;
                                            uint32_t wIdx = oc * inputChannel_ * kernelHeight_ * kernelWidth_ +
                                                           ic * kernelHeight_ * kernelWidth_ +
                                                           kh * kernelWidth_ + kw;

                                            sum += xGm_.GetValue(xIdx) * wGm_.GetValue(wIdx);
                                        }
                                    }
                                }
                            }

                            uint32_t localIdx = outLinearIdx - AscendC::GetBlockIdx() * tileLength_;
                            yLocal[localIdx] = sum;
                        }
                    }
                }
            }
        }

        tmpQueue1_.FreeTensor(tLocal1);
        tmpQueue2_.FreeTensor(tLocal2);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueW_.FreeTensor(wLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        for (uint32_t i = 0; i < len; ++i) {
            yGm_.SetValue(gmOffset + i, yLocal[i]);
        }
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {0};
    uint32_t outputChannel_ {0};
    uint32_t outputHeight_ {0};
    uint32_t outputWidth_ {0};
    uint32_t inputChannel_ {0};
    uint32_t inputHeight_ {0};
    uint32_t inputWidth_ {0};
    uint32_t kernelHeight_ {0};
    uint32_t kernelWidth_ {0};
    uint32_t strideH_ {1};
    uint32_t strideW_ {1};
    uint32_t padH_ {0};
    uint32_t padW_ {0};
    uint32_t dilationH_ {1};
    uint32_t dilationW_ {1};
    uint32_t group_ {1};
    uint32_t tileLength_ {0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputSquareKernelCustom op;
    op.Init(x, w, y,
            tiling_data.batchSize, tiling_data.outputChannel,
            tiling_data.outputHeight, tiling_data.outputWidth,
            tiling_data.inputChannel, tiling_data.inputHeight,
            tiling_data.inputWidth, tiling_data.kernelHeight,
            tiling_data.kernelWidth, tiling_data.strideH,
            tiling_data.strideW, tiling_data.padH, tiling_data.padW,
            tiling_data.dilationH, tiling_data.dilationW,
            tiling_data.group, tiling_data.tileLength);
    op.Process();
}
