constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t inputN, uint32_t inputC, uint32_t inputH, uint32_t inputW,
                                uint32_t outputC, uint32_t outputH, uint32_t outputW,
                                uint32_t kH, uint32_t kW,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups, uint32_t outputPadH, uint32_t outputPadW,
                                uint32_t hasBias)
    {
        totalLength_ = totalLength;
        inputN_ = inputN;
        inputC_ = inputC;
        inputH_ = inputH;
        inputW_ = inputW;
        outputC_ = outputC;
        outputH_ = outputH;
        outputW_ = outputW;
        kH_ = kH;
        kW_ = kW;
        strideH_ = strideH;
        strideW_ = strideW;
        padH_ = padH;
        padW_ = padW;
        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;
        hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ uint8_t*)x);
        wGm_.SetGlobalBuffer((__gm__ uint8_t*)w);
        biasGm_.SetGlobalBuffer((__gm__ uint8_t*)bias);
        yGm_.SetGlobalBuffer((__gm__ uint8_t*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLength * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, kH_ * kW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLength_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideBlocks = blockNum;
        uint32_t pos = blockIdx;

        while (pos < totalLength_) {
            const uint32_t n = pos / (outputC_ * outputH_ * outputW_);
            uint32_t remain = pos % (outputC_ * outputH_ * outputW_);
            const uint32_t oc = remain / (outputH_ * outputW_);
            remain = remain % (outputH_ * outputW_);
            const uint32_t oh = remain / outputW_;
            const uint32_t ow = remain % outputW_;

            float sum = 0.0f;

            for (uint32_t ki = 0; ki < kH_; ++ki) {
                for (uint32_t kj = 0; kj < kW_; ++kj) {
                    for (uint32_t ci = 0; ci < inputC_ / groups_; ++ci) {
                        uint32_t groupIdx = oc / (outputC_ / groups_);
                        uint32_t inC = groupIdx * (inputC_ / groups_) + ci;

                        uint32_t inH = oh * strideH_ - padH_ + ki * dilationH_;
                        uint32_t inW = ow * strideW_ - padW_ + kj * dilationW_;

                        if (inH >= inputH_ || inW >= inputW_) continue;

                        float val = GetX(n, inC, inH, inW);
                        float w = GetW(oc, ci, ki, kj);
                        sum += val * w;
                    }
                }
            }

            if (hasBias_ == 1) {
                sum += GetBias(oc);
            }

            GetY(n, oc, oh, ow) = sum;

            pos += strideBlocks;
        }
    }

private:
    __aicore__ inline float GetX(uint32_t n, uint32_t c, uint32_t h, uint32_t w)
    {
        uint32_t offset = ((n * inputC_ + c) * inputH_ + h) * inputW_ + w;
        return xGm_.GetValue(offset);
    }

    __aicore__ inline float GetW(uint32_t oc, uint32_t c, uint32_t kh, uint32_t kw)
    {
        uint32_t offset = ((oc * (inputC_ / groups_) + c) * kH_ + kh) * kW_ + kw;
        return wGm_.GetValue(offset);
    }

    __aicore__ inline float GetBias(uint32_t oc)
    {
        return biasGm_.GetValue(oc);
    }

    __aicore__ inline float& GetY(uint32_t n, uint32_t c, uint32_t h, uint32_t w)
    {
        uint32_t offset = ((n * outputC_ + c) * outputH_ + h) * outputW_ + w;
        return yGm_.GetValue(offset);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<uint8_t> xGm_;
    AscendC::GlobalTensor<uint8_t> wGm_;
    AscendC::GlobalTensor<uint8_t> biasGm_;
    AscendC::GlobalTensor<uint8_t> yGm_;

    uint32_t totalLength_ {0};
    uint32_t inputN_ {0};
    uint32_t inputC_ {0};
    uint32_t inputH_ {0};
    uint32_t inputW_ {0};
    uint32_t outputC_ {0};
    uint32_t outputH_ {0};
    uint32_t outputW_ {0};
    uint32_t kH_ {0};
    uint32_t kW_ {0};
    uint32_t strideH_ {1};
    uint32_t strideW_ {1};
    uint32_t padH_ {0};
    uint32_t padW_ {0};
    uint32_t dilationH_ {1};
    uint32_t dilationW_ {1};
    uint32_t groups_ {1};
    uint32_t hasBias_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
