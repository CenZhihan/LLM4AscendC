#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulScaleResidualAddClampLogSumExpMishCustom {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t rowLength, uint32_t tileLength,
                                float scaling, float clampMin, float clampMax)
    {
        totalLen_ = totalLength;
        rowLen_ = rowLength;
        tileLen_ = tileLength;
        scaling_ = scaling;
        clampMin_ = clampMin;
        clampMax_ = clampMax;
        scaleFactor_ = scaling_ + 1.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, rowLen_ * rowLen_);
        biasGm_.SetGlobalBuffer((__gm__ float*)bias, rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, rowLen_ * rowLength * sizeof(float));
        pipe_.InitBuffer(inQueueBias_, 1, rowLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue3_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        uint32_t rowStart = gmOffset / rowLen_;
        uint32_t rowEnd = (gmOffset + len + rowLen_ - 1) / rowLen_;
        uint32_t copyRows = rowEnd - rowStart;
        if (copyRows > rowLen_) copyRows = rowLen_;
        uint32_t weightSize = rowLen_ * rowLen_;

        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);

        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, weightGm_[0], weightSize);
        inQueueW_.EnQue(wLocal);

        AscendC::LocalTensor<float> biasLocal = inQueueBias_.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, biasGm_[0], rowLen_);
        inQueueBias_.EnQue(biasLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> biasLocal = inQueueBias_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        AscendC::MatMul(yLocal, xLocal, wLocal, biasLocal, rowLen_, rowLen_, rowLen_, len, len / rowLen_, 0, false, false);
        inQueueW_.FreeTensor(wLocal);

        AscendC::LocalTensor<float> tLocal1 = tmpQueue1_.AllocTensor<float>();
        AscendC::Adds(tLocal1, yLocal, 0.0f, len);
        outQueueY_.FreeTensor(yLocal);

        AscendC::Muls(yLocal, tLocal1, scaleFactor_, len);
        tmpQueue1_.FreeTensor(tLocal1);

        AscendC::LocalTensor<float> tLocal2 = tmpQueue2_.AllocTensor<float>();
        AscendC::Clamp(yLocal, yLocal, clampMin_, clampMax_, tLocal2, len);
        outQueueY_.FreeTensor(yLocal);

        AscendC::Adds(yLocal, tLocal2, 0.0f, len);
        tmpQueue2_.FreeTensor(tLocal2);

        AscendC::LocalTensor<float> tLocal3 = tmpQueue3_.AllocTensor<float>();
        AscendC::Muls(tLocal3, yLocal, 0.0f, len);
        
        uint32_t rows = len / rowLen_;
        uint32_t actualRowLen = rowLen_;
        
        for (uint32_t r = 0; r < rows; ++r) {
            uint32_t offset = r * rowLen_;
            float maxVal = yLocal.GetValue(offset);
            for (uint32_t c = 1; c < actualRowLen; ++c) {
                float val = yLocal.GetValue(offset + c);
                if (val > maxVal) maxVal = val;
            }
            tLocal3.SetValue(offset, maxVal);
            for (uint32_t c = 0; c < actualRowLen; ++c) {
                float expVal = expf(yLocal.GetValue(offset + c) - maxVal);
                tLocal3.SetValue(offset + c, expVal);
            }
        }
        
        AscendC::Adds(yLocal, tLocal3, 0.0f, len);
        tmpQueue3_.FreeTensor(tLocal3);

        AscendC::LocalTensor<float> tLocal4 = tmpQueue1_.AllocTensor<float>();
        for (uint32_t r = 0; r < rows; ++r) {
            uint32_t offset = r * rowLen_;
            float sumExp = 0.0f;
            for (uint32_t c = 0; c < actualRowLen; ++c) {
                sumExp += tLocal3.GetValue(offset + c);
            }
            float logSumExp = logf(sumExp + 1e-10f);
            float maxVal = tLocal3.GetValue(offset);
            yLocal.SetValue(offset, logSumExp + maxVal);
            for (uint32_t c = 1; c < actualRowLen; ++c) {
                yLocal.SetValue(offset + c, 0.0f);
            }
        }
        tmpQueue1_.FreeTensor(tLocal4);

        AscendC::Mish(yLocal, yLocal, len);

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueBias_.FreeTensor(biasLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueBias_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue3_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
    float scaling_ {1.0f};
    float clampMin_ {-3.40282e+38f};
    float clampMax_ {3.40282e+38f};
    float scaleFactor_ {2.0f};
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, 
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScaleResidualAddClampLogSumExpMishCustom op;
    op.Init(x, weight, bias, y, 
            tiling_data.totalLength, tiling_data.rowLength, tiling_data.tileLength,
            tiling_data.scaling, tiling_data.clampMin, tiling_data.clampMax);
    op.Process();
}
