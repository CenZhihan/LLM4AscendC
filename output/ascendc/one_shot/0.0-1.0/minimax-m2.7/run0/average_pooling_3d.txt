#include "average_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t HARD_CAP_ELEMS = 8192;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static int32_t CalcOutputDim(int32_t inputDim, int32_t kernelSize, int32_t stride, int32_t padding)
{
    return (inputDim + 2 * padding - kernelSize) / stride + 1;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling3dCustomTilingData tiling;

    const gert::Shape *inputShape = context->GetInputShape(0);
    std::vector<int64_t> shapeVec = inputShape->GetShapeVector();

    int32_t batch = static_cast<int32_t>(shapeVec[0]);
    int32_t channels = static_cast<int32_t>(shapeVec[1]);
    int32_t inputDepth = static_cast<int32_t>(shapeVec[2]);
    int32_t inputHeight = static_cast<int32_t>(shapeVec[3]);
    int32_t inputWidth = static_cast<int32_t>(shapeVec[4]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t kernelSize = static_cast<int32_t>(attrs->GetAttrPointer<float>(0))[0];
    int32_t stride = attrs->GetAttrPointer<float>(1) ? 
                     static_cast<int32_t>(attrs->GetAttrPointer<float>(1)[0]) : kernelSize;
    int32_t padding = attrs->GetAttrPointer<float>(2) ? 
                      static_cast<int32_t>(attrs->GetAttrPointer<float>(2)[0]) : 0;

    int32_t outputDepth = CalcOutputDim(inputDepth, kernelSize, stride, padding);
    int32_t outputHeight = CalcOutputDim(inputHeight, kernelSize, stride, padding);
    int32_t outputWidth = CalcOutputDim(inputWidth, kernelSize, stride, padding);

    uint32_t totalInputLength = static_cast<uint32_t>(inputShape->GetOriginShape().GetShapeSize());
    int32_t totalOutputCount = batch * channels * outputDepth * outputHeight * outputWidth;

    uint32_t maxTile = UB_BUDGET_BYTES / (sizeof(float) * 4);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if ((uint32_t)totalOutputCount < tileLen) {
        uint32_t aligned = AlignDown(totalOutputCount, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : (uint32_t)totalOutputCount;
    }

    uint32_t numTiles = CeilDiv((uint32_t)totalOutputCount, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalInputLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernelSize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_channels(channels);
    tiling.set_inputDepth(inputDepth);
    tiling.set_inputHeight(inputHeight);
    tiling.set_inputWidth(inputWidth);
    tiling.set_outputDepth(outputDepth);
    tiling.set_outputHeight(outputHeight);
    tiling.set_outputWidth(outputWidth);
    tiling.set_totalOutputCount(totalOutputCount);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    std::vector<int64_t> shapeVec = x_shape->GetShapeVector();
    int64_t batch = shapeVec[0];
    int64_t channels = shapeVec[1];
    int64_t depth = shapeVec[2];
    int64_t height = shapeVec[3];
    int64_t width = shapeVec[4];

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int64_t kernelSize = static_cast<int64_t>(attrs->GetAttrPointer<float>(0)[0]);
    int64_t stride = attrs->GetAttrPointer<float>(1) ? 
                     static_cast<int64_t>(attrs->GetAttrPointer<float>(1)[0]) : kernelSize;
    int64_t padding = attrs->GetAttrPointer<float>(2) ? 
                      static_cast<int64_t>(attrs->GetAttrPointer<float>(2)[0]) : 0;

    int64_t outputDepth = (depth + 2 * padding - kernelSize) / stride + 1;
    int64_t outputHeight = (height + 2 * padding - kernelSize) / stride + 1;
    int64_t outputWidth = (width + 2 * padding - kernelSize) / stride + 1;

    y_shape->SetShapeVector({batch, channels, outputDepth, outputHeight, outputWidth});
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelAveragePooling3dCustom {
public:
    __aicore__ inline KernelAveragePooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                int32_t kernelSize, int32_t stride, int32_t padding,
                                int32_t channels, int32_t inputDepth, int32_t inputHeight, int32_t inputWidth,
                                int32_t outputDepth, int32_t outputHeight, int32_t outputWidth,
                                int32_t totalOutputCount)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        channels_ = channels;
        inputDepth_ = inputDepth;
        inputHeight_ = inputHeight;
        inputWidth_ = inputWidth;
        outputDepth_ = outputDepth;
        outputHeight_ = outputHeight;
        outputWidth_ = outputWidth;
        totalOutputCount_ = totalOutputCount;
        
        kernelSizeSq_ = kernelSize_ * kernelSize_;
        kernelSizeCu_ = kernelSizeSq_ * kernelSize_;
        invKernelCount_ = 1.0f / static_cast<float>(kernelSizeCu_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(redQueue_, 1, kernelSizeCu_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputCount_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        int32_t batchSize = channels_;
        int32_t outPlaneSize = outputDepth_ * outputHeight_ * outputWidth_;
        int32_t inPlaneSize = inputDepth_ * inputHeight_ * inputWidth_;

        int32_t base = static_cast<int32_t>(blockIdx) * static_cast<int32_t>(tileLen_);
        if (base >= totalOutputCount_) return;

        int32_t curLen = totalOutputCount_ - base;
        if (curLen > static_cast<int32_t>(tileLen_)) curLen = static_cast<int32_t>(tileLen_);

        for (int32_t i = 0; i < curLen; ++i) {
            int32_t outIdx = base + i;
            int32_t b = outIdx / (channels_ * outPlaneSize);
            int32_t remain = outIdx % (channels_ * outPlaneSize);
            int32_t c = remain / outPlaneSize;
            int32_t outPlaneIdx = remain % outPlaneSize;
            
            int32_t od = outPlaneIdx / (outputHeight_ * outputWidth_);
            int32_t oh = (outPlaneIdx % (outputHeight_ * outputWidth_)) / outputWidth_;
            int32_t ow = outPlaneIdx % outputWidth_;

            int32_t id_start = od * stride_ - padding_;
            int32_t ih_start = oh * stride_ - padding_;
            int32_t iw_start = ow * stride_ - padding_;

            float sum = 0.0f;
            int32_t validCount = 0;

            for (int32_t kd = 0; kd < kernelSize_; ++kd) {
                int32_t id = id_start + kd;
                if (id < 0 || id >= inputDepth_) continue;
                
                for (int32_t kh = 0; kh < kernelSize_; ++kh) {
                    int32_t ih = ih_start + kh;
                    if (ih < 0 || ih >= inputHeight_) continue;
                    
                    for (int32_t kw = 0; kw < kernelSize_; ++kw) {
                        int32_t iw = iw_start + kw;
                        if (iw < 0 || iw >= inputWidth_) continue;

                        int32_t inIdx = ((b * channels_ + c) * inputDepth_ + id) * inputHeight_ * inputWidth_ + ih * inputWidth_ + iw;
                        float val = xGm_[inIdx];
                        sum += val;
                        ++validCount;
                    }
                }
            }

            float avg = (validCount > 0) ? (sum / static_cast<float>(validCount)) : 0.0f;
            int32_t yIdx = ((b * channels_ + c) * outputDepth_ + od) * outputHeight_ * outputWidth_ + oh * outputWidth_ + ow;
            yGm_[yIdx] = avg;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> redQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    int32_t kernelSize_ {3};
    int32_t stride_ {2};
    int32_t padding_ {0};
    int32_t channels_ {1};
    int32_t inputDepth_ {1};
    int32_t inputHeight_ {1};
    int32_t inputWidth_ {1};
    int32_t outputDepth_ {1};
    int32_t outputHeight_ {1};
    int32_t outputWidth_ {1};
    int32_t totalOutputCount_ {0};
    int32_t kernelSizeSq_ {9};
    int32_t kernelSizeCu_ {27};
    float invKernelCount_ {0.111111f};
};

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling3dCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.channels, tiling_data.inputDepth, tiling_data.inputHeight, tiling_data.inputWidth,
            tiling_data.outputDepth, tiling_data.outputHeight, tiling_data.outputWidth,
            tiling_data.totalOutputCount);
    op.Process();
}
