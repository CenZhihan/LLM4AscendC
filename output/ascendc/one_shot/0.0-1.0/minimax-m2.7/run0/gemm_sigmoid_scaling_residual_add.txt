static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmSigmoidScalingResidualAddCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    uint32_t m = static_cast<uint32_t>(x_shape->GetShapeSize());
    uint32_t k = static_cast<uint32_t>(x_shape->GetShapeDimension(1));
    uint32_t n = static_cast<uint32_t>(w_shape->GetShapeDimension(0));

    uint32_t mt = TILE_M;
    uint32_t nt = TILE_N;
    uint32_t kt = TILE_K;

    uint32_t tileMNum = CeilDiv(m, mt);
    uint32_t tileNNum = CeilDiv(n, nt);
    uint32_t tileKNum = CeilDiv(k, kt);

    uint32_t blockDim = tileMNum * tileNNum;
    if (blockDim > BLOCK_DIM_MAX) {
        blockDim = BLOCK_DIM_MAX;

    context->SetBlockDim(blockDim);

    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_mt(mt);
    tiling.set_nt(nt);
    tiling.set_kt(kt);
    tiling.set_kTileNum(tileKNum);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmSigmoidScalingResidualAddCustom {
public:
    __aicore__ inline KernelGemmSigmoidScalingResidualAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling,
                                GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t mt, uint32_t nt, uint32_t kt,
                                uint32_t kTileNum)
    {
        m_ = m;
        n_ = n;
        k_ = k;
        mt_ = mt;
        nt_ = nt;
        kt_ = kt;
        kTileNum_ = kTileNum;

        xGm_.SetGlobalBuffer((__gm__ float*)x, m_ * k_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, n_ * k_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, m_ * n_);

        scaling_ = *reinterpret_cast<float*>(scaling);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, mt_ * kt_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, nt_ * kt_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, nt_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, mt_ * nt_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t tileMNum = (m_ + mt_ - 1u) / mt_;
        uint32_t tileNNum = (n_ + nt_ - 1u) / nt_;
        uint32_t totalTiles = tileMNum * tileNNum;

        if (blockIdx >= totalTiles) return;

        uint32_t tileIdxM = blockIdx / tileNNum;
        uint32_t tileIdxN = blockIdx % tileNNum;

        uint32_t mStart = tileIdxM * mt_;
        uint32_t nStart = tileIdxN * nt_;

        uint32_t mLen = mt_;
        if (mStart + mLen > m_) mLen = m_ - mStart;
        uint32_t nLen = nt_;
        if (nStart + nLen > n_) nLen = n_ - nStart;

        uint32_t baseIdxA = mStart * k_;
        uint32_t baseIdxB = nStart * k_;

        uint32_t curIdxA = baseIdxA;
        uint32_t curIdxB = baseIdxB;

        int32_t bufferIdx = 0;
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Muls(yLocal, yLocal, 0.0f, mLen * nLen);

        if (kTileNum_ == 1) {
            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            AscendC::DataCopy(aLocal, xGm_[baseIdxA], mLen * k_);
            inQueueA_.EnQue(aLocal);

            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
            AscendC::DataCopy(bLocal, wGm_[baseIdxB], nLen * k_);
            inQueueB_.EnQue(bLocal);

            AscendC::LocalTensor<float> aOp = inQueueA_.DeQue<float>();
            AscendC::LocalTensor<float> bOp = inQueueB_.DeQue<float>();
            AscendC::Matmul(yLocal, aOp, bOp, yLocal, mLen, nLen, k_, k_, k_, nLen, 0, 0, 0, 0);
            inQueueB_.FreeTensor(bOp);
            inQueueA_.FreeTensor(aOp);
        } else {
            AscendC::LocalTensor<float> aFirst = inQueueA_.AllocTensor<float>();
            AscendC::DataCopy(aFirst, xGm_[baseIdxA], mLen * kt_);
            inQueueA_.EnQue(aFirst);

            AscendC::LocalTensor<float> bFirst = inQueueB_.AllocTensor<float>();
            AscendC::DataCopy(bFirst, wGm_[baseIdxB], nLen * kt_);
            inQueueB_.EnQue(bFirst);

            for (uint32_t kIdx = 0; kIdx < kTileNum_; ++kIdx) {
                uint32_t kOffset = kIdx * kt_;
                uint32_t kCur = kt_;
                if (kOffset + kCur > k_) kCur = k_ - kOffset;

                AscendC::LocalTensor<float> aOp = inQueueA_.DeQue<float>();
                AscendC::LocalTensor<float> bOp = inQueueB_.DeQue<float>();

                AscendC::LocalTensor<float> yTemp = tmpQueue_.AllocTensor<float>();
                AscendC::Muls(yTemp, yTemp, 0.0f, mLen * nLen);
                AscendC::Matmul(yTemp, aOp, bOp, yTemp, mLen, nLen, kCur, kCur, kCur, nLen, 0, 0, 0, 0);
                AscendC::Adds(yLocal, yLocal, yTemp, mLen * nLen);
                tmpQueue_.FreeTensor(yTemp);

                inQueueB_.FreeTensor(bOp);
                inQueueA_.FreeTensor(aOp);

                if (kIdx + 1 < kTileNum_) {
                    uint32_t kNextOffset = (kIdx + 1) * kt_;
                    uint32_t kNextCur = kt_;
                    if (kNextOffset + kNextCur > k_) kNextCur = k_ - kNextOffset;

                    AscendC::LocalTensor<float> aNext = inQueueA_.AllocTensor<float>();
                    AscendC::DataCopy(aNext, xGm_[baseIdxA + kNextOffset], mLen * kNextCur);
                    inQueueA_.EnQue(aNext);

                    AscendC::LocalTensor<float> bNext = inQueueB_.AllocTensor<float>();
                    AscendC::DataCopy(bNext, wGm_[baseIdxB + kNextOffset], nLen * kNextCur);
                    inQueueB_.EnQue(bNext);
                }
            }
        }

        AscendC::LocalTensor<float> bVec = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(bVec, bGm_[nStart], nLen);
        AscendC::Adds(yLocal, yLocal, bVec, mLen * nLen);
        tmpQueue_.FreeTensor(bVec);

        AscendC::Sigmoid(yLocal, yLocal, mLen * nLen);

        float s = scaling_;
        float s_minus_1 = s - 1.0f;

        AscendC::LocalTensor<float> yOrig = outQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yOrig, yGm_[mStart * n_ + nStart], mLen * nLen);

        AscendC::LocalTensor<float> tmpScale = tmpQueue_.AllocTensor<float>();
        AscendC::Muls(tmpScale, yLocal, s_minus_1, mLen * nLen);
        AscendC::Adds(yLocal, yLocal, tmpScale, mLen * nLen);
        tmpQueue_.FreeTensor(tmpScale);

        outQueueY_.EnQue(yLocal);
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t m_ {0};
    uint32_t n_ {0};
    uint32_t k_ {0};
    uint32_t mt_ {0};
    uint32_t nt_ {0};
    uint32_t kt_ {0};
    uint32_t kTileNum_ {0};
    float scaling_ {1.0f};

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
};

extern "C" __global__ __aicore__ void gemm_sigmoid_scaling_residual_add_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSigmoidScalingResidualAddCustom op;
    op.Init(x, w, b, scaling, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.mt, tiling_data.nt, tiling_data.kt,
            tiling_data.kTileNum);
    op.Process();
}
