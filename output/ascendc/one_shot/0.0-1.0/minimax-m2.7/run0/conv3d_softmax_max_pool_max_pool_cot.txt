        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        // Process each element in the tile
        for (uint32_t i = 0; i < curLen; ++i) {
            uint32_t idx = base + i;
            // Compute coordinates for final output
            uint32_t n = idx / (kOutChannels_ * kFinalDepth_ * kFinalHeight_ * kFinalWidth_);
            uint32_t rest = idx % (kOutChannels_ * kFinalDepth_ * kFinalHeight_ * kFinalWidth_);
            uint32_t c = rest / (kFinalDepth_ * kFinalHeight_ * kFinalWidth_);
            rest = rest % (kOutChannels_ * kFinalDepth_ * kFinalHeight_ * kFinalWidth_);

            uint32_t w = rest % kFinalWidth_;

            uint32_t base_h = h * kPoolFactor_;
            uint32_t base_w = w * kPoolFactor_;

            // Loop over region positions

                for (int dh = 0; dh < 4; ++dh) {

                        uint32_t conv_d = base_d + dd;
                        uint32_t conv_h = base_h + dh;
                        uint32_t conv_w = base_w + dw;

                        float conv_vals[16];
                        for (int c_out = 0; c_out < 16; ++c_out) {
                            float val = biasGm_[c_out];
                            // Convolution computation
                            for (int c_in = 0; c_in < 3; ++c_in) {
                                for (int kd = 0; kd < 3; ++kd) {
                                    for (int kh = 0; kh < 3; ++kh) {
                                        for (int kw = 0; kw < 3; ++kw) {

                                            uint32_t h_in = conv_h + kh;

                                            val += weightGm_[w_idx] * xGm_[x_idx];

                                    }
                                }

                            sum_exp += exp_vals[c_out];

                        }

                            max_val = exp_vals[c];
                        }
                    }
                }
            }

        }
