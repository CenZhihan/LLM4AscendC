kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulForLowerTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForLowerTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t m, uint32_t k, uint32_t n,
                                uint32_t blockDim, uint32_t tileRow, uint32_t tileCol)
    {
        m_ = m;
        k_ = k;
        n_ = n;
        blockDim_ = blockDim;
        tileRow_ = tileRow;
        tileCol_ = tileCol;

        AGm_.SetGlobalBuffer((__gm__ float*)A, m_ * k_);
        BGm_.SetGlobalBuffer((__gm__ float*)B, k_ * n_);
        CGm_.SetGlobalBuffer((__gm__ float*)C, m_ * n_);

        pipe_.InitBuffer(inQueueA_, BUFFER_NUM, tileRow_ * k_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, k_ * tileCol_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, BUFFER_NUM, tileRow_ * tileCol_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, tileRow_ * tileCol_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t rowBlocks = CeilDiv(m_, tileRow_);
        uint32_t colBlocks = CeilDiv(n_, tileCol_);
        uint32_t totalBlocks = rowBlocks * colBlocks;

        uint32_t blocksPerCore = CeilDiv(totalBlocks, blockDim_);
        uint32_t startBlock = blockIdx * blocksPerCore;
        uint32_t endBlock = startBlock + blocksPerCore;
        if (endBlock > totalBlocks) endBlock = totalBlocks;

        for (uint32_t blkIdx = startBlock; blkIdx < endBlock; ++blkIdx) {
            uint32_t rowBlk = blkIdx / colBlocks;
            uint32_t colBlk = blkIdx % colBlocks;

            uint32_t rowStart = rowBlk * tileRow_;
            uint32_t colStart = colBlk * tileCol_;
            uint32_t curRow = (rowStart + tileRow_ > m_) ? (m_ - rowStart) : tileRow_;
            uint32_t curCol = (colStart + tileCol_ > n_) ? (n_ - colStart) : tileCol_;

            if (rowStart >= m_ || colStart >= n_) continue;

            ComputeTile(rowStart, colStart, curRow, curCol);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t rowStart, uint32_t colStart,
                                       uint32_t curRow, uint32_t curCol)
    {
        AscendC::LocalTensor<float> cLocal = outQueueC_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpBuf_.GetTensor<float>(0, curRow * curCol);

        AscendC::Sets(cLocal, 0.0f, curRow * curCol);

        for (uint32_t kIdx = 0; kIdx < k_; ++kIdx) {
            bool kInRangeA = (kIdx <= rowStart + curRow - 1);
            bool kInRangeB = (kIdx < n_);

            if (!kInRangeA || !kInRangeB) continue;

            uint32_t aRowEnd = (kIdx < rowStart) ? curRow : (k_ < rowStart + curRow ? k_ - rowStart : curRow);
            if (kIdx < rowStart) continue;
            uint32_t aRows = rowStart + curRow - kIdx;
            if (aRows == 0) continue;
            aRows = (aRows > curRow) ? curRow : aRows;

            uint32_t bCols = curCol;
            uint32_t aLen = aRows;
            uint32_t bLen = aRows * bCols;

            AscendC::LocalTensor<float> aLocal = inQueueA_.AllocTensor<float>();
            AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();

            AscendC::DataCopy(aLocal, AGm_[kIdx * k_ + rowStart], aLen);

            uint32_t bRowEnd = kIdx + aRows;
            uint32_t actualBRows = (bRowEnd > k_) ? (k_ - kIdx) : aRows;
            if (kIdx >= n_) continue;
            uint32_t copyCols = (colStart + curCol > n_) ? (n_ - colStart) : curCol;

            for (uint32_t c = 0; c < copyCols && kIdx < n_; ++c) {
                uint32_t bIdx = kIdx * n_ + colStart + c;
                if (bIdx < k_ * n_) {
                    bLocal.SetValue(c, BGm_[bIdx]);
                }
            }
            for (uint32_t r = 1; r < actualBRows; ++r) {
                for (uint32_t c = 0; c < copyCols && (kIdx + r) < n_; ++c) {
                    uint32_t bIdx = (kIdx + r) * n_ + colStart + c;
                    if (bIdx < k_ * n_) {
                        bLocal.SetValue(r * curCol + c, BGm_[bIdx]);
                    }
                }
            }
            bLocal.SetShape({actualBRows, copyCols});

            AscendC::Matmul(cLocal, aLocal, bLocal, cLocal, curRow, curCol, k_, 1.0f);

            inQueueA_.FreeTensor(aLocal);
            inQueueB_.FreeTensor(bLocal);
        }

        AscendC::LocalTensor<float> cTmp = outQueueC_.AllocTensor<float>();
        AscendC::DataCopy(cTmp, cLocal, curRow * curCol);
        outQueueC_.FreeTensor(cLocal);

        uint32_t cOffset = rowStart * n_ + colStart;
        AscendC::DataCopy(CGm_[cOffset], cTmp, curRow * curCol);
        outQueueC_.FreeTensor(cTmp);
    }

    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b)
    {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueC_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> tmpBuf_;

    AscendC::GlobalTensor<float> AGm_;
    AscendC::GlobalTensor<float> BGm_;
    AscendC::GlobalTensor<float> CGm_;

    uint32_t m_ {0};
    uint32_t k_ {0};
    uint32_t n_ {0};
    uint32_t blockDim_ {8};
    uint32_t tileRow_ {64};
    uint32_t tileCol_ {64};
};

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForLowerTriangularMatricesCustom op;
    op.Init(A, B, C, tiling_data.m, tiling_data.k, tiling_data.n,
            tiling_data.blockDim, tiling_data.tileRow, tiling_data.tileCol);
    op.Process();
}

import torch
import torch_npu
import custom_ops_lib

_FN = 'matmul_for_lower_triangular_matrices_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, *init_inputs):
        super(ModelNew, self).__init__()

    def forward(self, A, B):
        """
        Performs matrix multiplication of lower triangular matrices A and B.
        Since A and B are lower triangular, result C is also lower triangular.

        Args:
            A (torch.Tensor): Lower triangular matrix of shape (N, N).
            B (torch.Tensor): Lower triangular matrix of shape (N, N).

        Returns:
            torch.Tensor: The result of matrix multiplication C of shape (N, N).
        """
        fn = getattr(custom_ops_lib, _FN)
        C = fn(A, B)
