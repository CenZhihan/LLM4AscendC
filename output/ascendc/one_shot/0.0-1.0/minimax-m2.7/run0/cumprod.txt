host_operator_src = '''
#include "cumprod_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumprodCustomTilingData tiling;
    const gert::Shape *shape = context->GetInputShape(0);
    const uint32_t totalLength = static_cast<uint32_t>(shape->GetOriginShape().GetShapeSize());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = dimAttr ? *dimAttr : 0;

    // Ensure dim is non‑negative; if negative, treat as last dimension
    if (dim < 0) {
        int32_t rank = static_cast<int32_t>(shape->GetShape().size());
        dim = rank + dim;
        if (dim < 0) dim = 0;
    }

    const auto &shapeVec = shape->GetShape();
    uint32_t lineSize = 1;
    if (dim >= 0 && dim < static_cast<int32_t>(shapeVec.size())) {
        lineSize = static_cast<uint32_t>(shapeVec[dim]);
    } else {
        lineSize = static_cast<uint32_t>(shapeVec.back());
    }

    // Determine tile length based on UB budget (two buffers of float)
    uint32_t maxTile = UB_BUDGET_BYTES / (2 * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = lineSize;
    if (tileLen > maxTile) tileLen = maxTile;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = ALIGN_ELEMS;

    uint32_t numLines = (lineSize == 0) ? 0 : totalLength / lineSize;
    uint32_t blockDim = numLines;
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_lineSize(lineSize);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

class CumprodCustom : public OpDef {
public:
    explicit CumprodCustom(const char *name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim")
            .AttrType(OPTIONAL)
            .Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumprodCustom);

} // namespace ops
'''

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCumprodCustom {
public:
    __aicore__ inline KernelCumprodCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t blockDim, uint32_t lineSize, int32_t dim)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        blockDim_ = blockDim;
        lineSize_ = lineSize;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t numLines = (lineSize_ == 0) ? 0 : totalLen_ / lineSize_;
        if (numLines == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= numLines) return;

        uint32_t lineStart = blockIdx * lineSize_;
        uint32_t offset = lineStart;
        uint32_t remaining = lineSize_;

        float cumProd = 1.0f;

        while (remaining > 0) {
            uint32_t curChunk = (remaining > tileLen_) ? tileLen_ : remaining;
            CopyIn(offset, curChunk);
            cumProd = Compute(curChunk, cumProd);
            CopyOut(offset, curChunk);
            offset += curChunk;
            remaining -= curChunk;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline float Compute(uint32_t len, float cumProd)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        float acc = cumProd;
        for (uint32_t i = 0; i < len; ++i) {
            acc *= xLocal.GetValue(i);
            yLocal.SetValue(i, acc);
        }

        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        return acc;
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t blockDim_ {0};
    uint32_t lineSize_ {0};
    int32_t dim_ {0};
};

extern "C" __global__ __aicore__ void cumprod_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumprodCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.blockDim, tiling_data.lineSize, tiling_data.dim);
    op.Process();
}
'''

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor cumprod_impl_npu(const at::Tensor& self, int64_t dim)
{
    at::Tensor result = at::empty_like(self);
    EXEC_NPU_CMD(aclnnCumprodCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("cumprod_custom", &cumprod_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("cumprod_custom", &cumprod_impl_npu, "Cumulative product (CumprodCustom on NPU)");
}
