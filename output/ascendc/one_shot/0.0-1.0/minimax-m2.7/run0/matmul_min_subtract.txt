kernel_src=""""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t K_TILE = 512;
constexpr int32_t M_TILE = 128;
constexpr int32_t N_TILE = 128;
constexpr int32_t K_UNROLL = 8;

class KernelMatmulMinSubtractCustom {
public:
    __aicore__ inline KernelMatmulMinSubtractCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures,
                                uint32_t outFeatures, float constant)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        constant_ = constant;
        constantMinus_ = constant - constant;  // placeholder
        negConst_ = -constant;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        wGm_.SetGlobalBuffer((__gm__ float*)w);
        bGm_.SetGlobalBuffer((__gm__ float*)b);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        uint32_t mSize = M_TILE * N_TILE * sizeof(float);
        uint32_t kSize = K_TILE * N_TILE * sizeof(float);
        uint32_t biasSize = N_TILE * sizeof(float);
        uint32_t tmpSize = M_TILE * N_TILE * sizeof(float);

        pipe_.InitBuffer(xQueueX_, BUFFER_NUM, kSize);
        pipe_.InitBuffer(wQueueW_, BUFFER_NUM, kSize);
        pipe_.InitBuffer(biasQueue_, 1, biasSize);
        pipe_.InitBuffer(mcQueueY_, BUFFER_NUM, mSize);
        pipe_.InitBuffer(tmpQueueMin_, 1, tmpSize);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tmpSize);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalBlocks = static_cast<uint32_t>(AscendC::GetBlockNum());

        for (uint32_t b = 0; b < batchSize_; ++b) {
            uint32_t mBlocks = (outFeatures_ + M_TILE - 1) / M_TILE;
            uint32_t totalMBlocks = mBlocks;
            uint32_t nBlocks = (outFeatures_ + N_TILE - 1) / N_TILE;

            uint32_t totalBlocksPerBatch = totalMBlocks * nBlocks;
            if (totalBlocksPerBatch == 0) continue;

            uint32_t startBlock = blockIdx;
            if (startBlock >= totalBlocksPerBatch) continue;

            uint32_t mBlockIdx = startBlock / nBlocks;
            uint32_t nBlockIdx = startBlock % nBlocks;

            uint32_t mStart = mBlockIdx * M_TILE;
            uint32_t nStart = nBlockIdx * N_TILE;

            uint32_t mLen = (mStart + M_TILE <= outFeatures_) ? M_TILE : (outFeatures_ - mStart);
            uint32_t nLen = (nStart + N_TILE <= outFeatures_) ? N_TILE : (outFeatures_ - nStart);

            MatmulBlock(b, mStart, nStart, mLen, nLen);
        }
    }

private:
    __aicore__ inline void MatmulBlock(uint32_t batchIdx, uint32_t mStart, uint32_t nStart,
                                       uint32_t mLen, uint32_t nLen)
    {
        uint32_t kBlocks = (inFeatures_ + K_TILE - 1) / K_TILE;

        AscendC::LocalTensor<float> yLocal = mcQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> wTile = wQueueW_.AllocTensor<float>();
        AscendC::LocalTensor<float> xTile = xQueueX_.AllocTensor<float>();

        AscendC::Sets(yLocal, 0.0f, mLen * nLen);

        uint32_t kTileActual = (K_TILE < inFeatures_) ? K_TILE : inFeatures_;
        uint32_t kRem = inFeatures_;

        for (uint32_t kb = 0; kb < kBlocks; ++kb) {
            uint32_t kOffset = kb * K_TILE;
            uint32_t curK = (kRem < K_TILE) ? kRem : K_TILE;
            kRem -= curK;

            uint32_t wOffset = nStart * inFeatures_ + kOffset;
            uint32_t xOffset = batchIdx * inFeatures_ + kOffset;

            AscendC::DataCopy(wTile, wGm_[wOffset], curK * nLen);
            AscendC::DataCopy(xTile, xGm_[xOffset], curK * mLen);

            AscendC::MatsMul(yLocal, xTile, wTile, yLocal, mLen, nLen, curK, curK, nLen, 1, false, false);

            if (kb < kBlocks - 1) {
                if (kRem > 0 && kRem < K_TILE) {
                    kTileActual = kRem;
                }
            }
        }

        mcQueueY_.FreeTensor(xTile);
        mcQueueY_.FreeTensor(wTile);

        AscendC::LocalTensor<float> biasLocal = biasQueue_.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, bGm_[nStart], nLen);

        AscendC::Adds(yLocal, yLocal, biasLocal, mLen * nLen);
        biasQueue_.FreeTensor(biasLocal);

        AscendC::LocalTensor<float> minLocal = tmpQueueMin_.AllocTensor<float>();
        AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();

        AscendC::Sets(minLocal, constant_, mLen * nLen);
        AscendC::Mins(outLocal, yLocal, minLocal, mLen * nLen);
        AscendC::AddScalar(outLocal, outLocal, negConst_, mLen * nLen);

        tmpQueueMin_.FreeTensor(minLocal);

        uint32_t yOffset = batchIdx * outFeatures_ + mStart * outFeatures_ + nStart;
        AscendC::DataCopy(yGm_[yOffset], outLocal, mLen * nLen);
        outQueueY_.FreeTensor(outLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueueW_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> mcQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMin_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ {1};
    uint32_t inFeatures_ {0};
    uint32_t outFeatures_ {0};
    float constant_ {0.0f};
    float constantMinus_ {0.0f};
    float negConst_ {0.0f};
};

extern "C" __global__ __aicore__ void matmul_min_subtract_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                                                  GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMinSubtractCustom op;
    op.Init(x, w, b, y,
            tiling_data.batchSize,
            tiling_data.inFeatures,
            tiling_data.outFeatures,
            tiling_data.constant);
    op.Process();
}
