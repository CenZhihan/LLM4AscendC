#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom {
public:
    __aicore__ inline KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t N, uint32_t C_in, uint32_t C_out,
                                uint32_t D_in, uint32_t H_in, uint32_t W_in,
                                uint32_t D_out, uint32_t H_out, uint32_t W_out,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t hasBias,
                                uint32_t outputSize, uint32_t sumSize, uint32_t spatialSize)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        N_ = N;
        C_in_ = C_in;
        C_out_ = C_out;
        D_in_ = D_in;
        H_in_ = H_in;
        W_in_ = W_in;
        D_out_ = D_out;
        H_out_ = H_out;
        W_out_ = W_out;
        kD_ = kD;
        kH_ = kH;
        kW_ = kW;
        hasBias_ = hasBias;
        outputSize_ = outputSize;
        sumSize_ = sumSize;
        spatialSize_ = spatialSize;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_in_ * D_in_ * H_in_ * W_in_);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight, C_in_ * C_out_ * kD_ * kH_ * kW_);
        if (hasBias_) {
            biasGm_.SetGlobalBuffer((__gm__ float*)bias, C_out_);
        }

        // Workspace layout: conv_out (outputSize) + sumBuf (sumSize) + maxVal (1)
        convOut_ = reinterpret_cast<__gm__ float*>(workspace);
        sumBuf_ = convOut_ + outputSize_;
        maxVal_ = sumBuf_ + sumSize_;
        y_ = reinterpret_cast<__gm__ float*>(y);

        // Allocate local buffer for weight in UB
        pipe_.InitBuffer(weightQueue_, 1, C_in_ * C_out_ * kD_ * kH_ * kW_ * sizeof(float));
        if (hasBias_) {
            pipe_.InitBuffer(biasQueue_, 1, C_out_ * sizeof(float));
        }
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        // Load weight and bias into UB
        AscendC::LocalTensor<float> weightLocal = weightQueue_.AllocTensor<float>();
        AscendC::DataCopy(weightLocal, weightGm_, C_in_ * C_out_ * kD_ * kH_ * kW_);
        weightQueue_.EnQue(weightLocal);

        AscendC::LocalTensor<float> biasLocal;
        if (hasBias_) {
            biasLocal = biasQueue_.AllocTensor<float>();
            AscendC::DataCopy(biasLocal, biasGm_, C_out_);
            biasQueue_.EnQue(biasLocal);
        }

        // Compute conv and max
        ComputeConvAndMax(weightLocal, biasLocal);

        // Second pass: multiply, sum, clamp, average
        ComputeFinalResult();

        // Free UB buffers
        weightQueue_.DeQue<float>();
        if (hasBias_) biasQueue_.DeQue<float>();
    }

private:
    __aicore__ inline void ComputeConvAndMax(AscendC::LocalTensor<float> &weightLocal,
                                            AscendC::LocalTensor<float> &biasLocal)
    {
        float localMax = -3.402823466e+38f; // -INF

        // Process entire output sequentially (single thread)
        for (uint32_t idx = 0; idx < totalLen_; ++idx) {
            // Decode output indices
            uint32_t n = idx / (C_out_ * D_out_ * H_out_ * W_out_);
            uint32_t rem = idx % (C_out_ * D_out_ * H_out_ * W_out_);
            uint32_t c_out = rem / (D_out_ * H_out_ * W_out_);
            rem = rem % (D_out_ * H_out_ * W_out_);
            uint32_t d = rem / (H_out_ * W_out_);
            rem = rem % (H_out_ * W_out_);
            uint32_t h = rem / W_out_;
            uint32_t w = rem % W_out_;

            float sum = 0.0f;
            // Convolution: sum over C_in and kernel
            for (uint32_t c_in = 0; c_in < C_in_; ++c_in) {
                for (uint32_t kd = 0; kd < kD_; ++kd) {
                    int32_t input_d = static_cast<int32_t>(d) - static_cast<int32_t>(kd);
                    if (input_d < 0 || input_d >= static_cast<int32_t>(D_in_)) continue;
                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        int32_t input_h = static_cast<int32_t>(h) - static_cast<int32_t>(kh);
                        if (input_h < 0 || input_h >= static_cast<int32_t>(H_in_)) continue;
                        for (uint32_t kw = 0; kw < kW_; ++kw) {
                            int32_t input_w = static_cast<int32_t>(w) - static_cast<int32_t>(kw);
                            if (input_w < 0 || input_w >= static_cast<int32_t>(W_in_)) continue;

                            // weight index: [c_in][c_out][kd][kh][kw]
                            uint32_t weight_idx = ((c_in * C_out_ + c_out) * kD_ + kd) * kH_ * kW_ + kh * kW_ + kw;
                            // input index: [n][c_in][d][h][w]
                            uint32_t input_idx = ((n * C_in_ + c_in) * D_in_ + static_cast<uint32_t>(input_d)) * H_in_ + static_cast<uint32_t>(input_h)) * W_in_ + static_cast<uint32_t>(input_w);

                            sum += weightLocal.GetValue(weight_idx) * xGm_.GetValue(input_idx);
                        }
                    }
                }
            }
            // Add bias if present
            if (hasBias_) {
                sum += biasLocal.GetValue(c_out);
            }

            // Store conv output
            convOut_[idx] = sum;
            // Update max
            if (sum > localMax) localMax = sum;
        }

        // Store max to global memory
        maxVal_[0] = localMax;
    }

    __aicore__ inline void ComputeFinalResult()
    {
        float globalMax = maxVal_[0];
        float countInv = 1.0f / static_cast<float>(spatialSize_);

        // Initialize sum buffer to zero
        for (uint32_t i = 0; i < sumSize_; ++i) {
            sumBuf_[i] = 0.0f;
        }

        // Second pass: multiply conv output by max, clamp, accumulate
        for (uint32_t idx = 0; idx < totalLen_; ++idx) {
            float val = convOut_[idx] * globalMax;
            // Clamp
            if (val < 0.0f) val = 0.0f;
            else if (val > 1.0f) val = 1.0f;

            // Decode indices for sum buffer
            uint32_t n = idx / (C_out_ * D_out_ * H_out_ * W_out_);
            uint32_t rem = idx % (C_out_ * D_out_ * H_out_ * W_out_);
            uint32_t c_out = rem / (D_out_ * H_out_ * W_out_);
            uint32_t sumIdx = n * C_out_ + c_out;
            sumBuf_[sumIdx] += val;
        }

        // Final pass: compute average and clamp
        for (uint32_t i = 0; i < sumSize_; ++i) {
            float avg = sumBuf_[i] * countInv;
            if (avg < 0.0f) avg = 0.0f;
            else if (avg > 1.0f) avg = 1.0f;
            y_[i] = avg;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> weightQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> biasGm_;

    __gm__ float* convOut_;
    __gm__ float* sumBuf_;
    __gm__ float* maxVal_;
    __gm__ float* y_;

    uint32_t totalLen_;
    uint32_t tileLen_;
    uint32_t N_;
    uint32_t C_in_;
    uint32_t C_out_;
    uint32_t D_in_;
    uint32_t H_in_;
    uint32_t W_in_;
    uint32_t D_out_;
    uint32_t H_out_;
    uint32_t W_out_;
    uint32_t kD_;
    uint32_t kH_;
    uint32_t kW_;
    uint32_t hasBias_;
    uint32_t outputSize_;
    uint32_t sumSize_;
    uint32_t spatialSize_;
};

extern "C" __global__ __aicore__ void conv_transpose3d_multiply_max_global_avg_pool_clamp_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.totalLength, tiling_data.tileLength,

            tiling_data.D_in, tiling_data.H_in, tiling_data.W_in,
            tiling_data.D_out, tiling_data.H_out, tiling_data.W_out,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.hasBias,
            tiling_data.outputSize, tiling_data.sumSize, tiling_data.spatialSize);
    op.Process();
}
