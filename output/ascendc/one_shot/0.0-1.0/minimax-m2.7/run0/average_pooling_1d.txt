kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelAveragePooling1dCustom {
public:
    __aicore__ inline KernelAveragePooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalOutputElements, uint32_t inputLength,
                                uint32_t outputLength, uint32_t batchSize,
                                uint32_t inChannels, uint32_t kernelSize,
                                uint32_t stride, uint32_t padding)
    {
        totalOutputElements_ = totalOutputElements;
        inputLength_ = inputLength;
        outputLength_ = outputLength;
        batchSize_ = batchSize;
        inChannels_ = inChannels;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, kernelSize_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutputElements_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t elementsPerBlock = (totalOutputElements_ + blockNum - 1) / blockNum;
        uint32_t startIdx = blockIdx * elementsPerBlock;
        uint32_t endIdx = startIdx + elementsPerBlock;
        if (endIdx > totalOutputElements_) endIdx = totalOutputElements_;

        uint32_t loopCount = endIdx - startIdx;
        for (uint32_t i = 0; i < loopCount; ++i) {
            uint32_t outIdx = startIdx + i;
            
            uint32_t pos = outIdx % outputLength_;
            uint32_t ch = (outIdx / outputLength_) % inChannels_;
            uint32_t batch = outIdx / (inChannels_ * outputLength_);
            
            int32_t inputStart = static_cast<int32_t>(pos * stride_) - static_cast<int32_t>(padding_);
            
            float sum = 0.0f;
            
            for (uint32_t k = 0; k < kernelSize_; ++k) {
                int32_t inputIdx = inputStart + static_cast<int32_t>(k);
                
                float val;
                if (inputIdx < 0 || inputIdx >= static_cast<int32_t>(inputLength_)) {
                    val = 0.0f;
                } else {
                    uint32_t linearInputIdx = batch * inChannels_ * inputLength_ + 
                                              ch * inputLength_ + 
                                              static_cast<uint32_t>(inputIdx);
                    val = xGm_.GetValue(linearInputIdx);
                }
                sum += val;
            }
            
            float avg = sum / static_cast<float>(kernelSize_);
            
            uint32_t linearOutIdx = batch * inChannels_ * outputLength_ + 
                                    ch * outputLength_ + 
                                    pos;
            yGm_.SetValue(linearOutIdx, avg);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutputElements_ {0};
    uint32_t inputLength_ {0};
    uint32_t outputLength_ {0};
    uint32_t batchSize_ {0};
    uint32_t inChannels_ {0};
    uint32_t kernelSize_ {0};
    uint32_t stride_ {0};
    uint32_t padding_ {0};
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling1dCustom op;
    op.Init(x, y, tiling_data.totalOutputElements, tiling_data.inputLength,
            tiling_data.outputLength, tiling_data.batchSize, tiling_data.inChannels,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding);
    op.Process();
}
