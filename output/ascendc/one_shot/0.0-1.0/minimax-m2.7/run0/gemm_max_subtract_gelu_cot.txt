constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMaxSubtractGeluCustom {
public:
    __aicore__ inline KernelGemmMaxSubtractGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures, uint32_t outFeatures,
                                int32_t maxDim)
    {
        batchSize_ = batchSize;
        inFeatures_ = inFeatures;
        outFeatures_ = outFeatures;
        maxDim_ = maxDim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batchSize * inFeatures);
        wGm_.SetGlobalBuffer((__gm__ float*)w, outFeatures * inFeatures);
        bGm_.SetGlobalBuffer((__gm__ float*)b, outFeatures);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batchSize * outFeatures);

        // Allocate buffers for intermediate results
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, batchSize * inFeatures * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, outFeatures * inFeatures * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, batchSize * outFeatures * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, batchSize * outFeatures * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Process in tiles
        for (uint32_t tileIdx = 0; tileIdx < numTiles_; ++tileIdx) {
            // Load input tile
            CopyIn(tileIdx);
            // Compute GEMM for this tile
            Compute(tileIdx);
            // Store output
            CopyOut(tileIdx);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t tileIdx)
    {
        uint32_t tileOffset = tileIdx * tileSize_;
        uint32_t copySize = min(tileSize_, batchSize_ * inFeatures_ - tileOffset);

        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[tileOffset], copySize);
        inQueueX_.EnQue(xLocal);

        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        AscendC::DataCopy(wLocal, wGm_[tileOffset], copySize);
        inQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void Compute(uint32_t tileIdx)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

        // GEMM computation: y = x * w^T + b
        AscendC::Gemm(yLocal, xLocal, wLocal, batchSize_, outFeatures_, inFeatures_);
        tmpQueue_.EnQue(tmpLocal);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueW_.FreeTensor(wLocal);
    }

    __aicore__ inline void CopyOut(uint32_t tileIdx)
    {
        uint32_t tileOffset = tileIdx * tileSize_;
        uint32_t copySize = min(tileSize_, batchSize_ * outFeatures_ - tileOffset);

        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[tileOffset], yLocal, copySize);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_ = 0;
    uint32_t inFeatures_ = 0;
    uint32_t outFeatures_ = 0;
    uint32_t numTiles_ = 0;
    uint32_t tileSize_ = 0;
};
