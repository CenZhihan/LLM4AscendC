host_operator_src = '''
#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape *inputShape = context->GetInputShape(0);
    const gert::Shape &shape = inputShape->GetOriginShape();
    uint32_t dimCount = shape.GetDimCount();

    uint32_t dim0 = (dimCount > 0) ? shape.GetDim(0) : 1;
    uint32_t dim1 = (dimCount > 1) ? shape.GetDim(1) : 1;
    uint32_t dim2 = (dimCount > 2) ? shape.GetDim(2) : 1;
    uint32_t dim3 = (dimCount > 3) ? shape.GetDim(3) : 1;

    const int32_t *attrDim = context->GetAttrs()->GetAttrPointer<int32_t>(0);
    uint32_t reduceDim = (attrDim != nullptr) ? static_cast<uint32_t>(*attrDim) : 0;

    uint32_t reduceDimSize = (reduceDim < dimCount) ? shape.GetDim(reduceDim) : 1;

    uint32_t totalInputLength = static_cast<uint32_t>(shape.GetShapeSize());
    uint32_t totalOutputLength = totalInputLength / reduceDimSize;

    uint32_t stride = 1;
    for (uint32_t i = reduceDim + 1; i < dimCount; ++i) {
        stride *= shape.GetDim(i);
    }

    uint32_t maxTile = UB_BUDGET_BYTES / (reduceDimSize * sizeof(float));
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOutputLength < tileLen) {
        uint32_t aligned = AlignDown(totalOutputLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutputLength;
    }

    uint32_t numTiles = (tileLen == 0) ? 1 : CeilDiv(totalOutputLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8) blockDim = 8;
    if (blockDim > 24) blockDim = 24;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalOutputLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_reduceDimSize(reduceDimSize);
    tiling.set_dim(reduceDim);
    tiling.set_shapeDim0(dim0);
    tiling.set_shapeDim1(dim1);
    tiling.set_shapeDim2(dim2);
    tiling.set_shapeDim3(dim3);
    tiling.set_stride(stride);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape &shape = xShape->GetOriginShape();

    const int32_t *attrDim = context->GetAttrs()->GetAttrPointer<int32_t>(0);
    uint32_t reduceDim = (attrDim != nullptr) ? static_cast<uint32_t>(*attrDim) : 0;
    if (reduceDim >= shape.GetDimCount()) reduceDim = 0;

    gert::Shape outputShape;
    for (uint32_t i = 0; i < shape.GetDimCount(); ++i) {
        if (i != reduceDim) {
            outputShape.AppendDim(shape.GetDim(i));
        }
    }
    *context->GetOutputShape(0) = outputShape;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return ge::GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

class MeanReductionOverADimensionCustom : public OpDef {
public:
    explicit MeanReductionOverADimensionCustom(const char *name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(ge::REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(ge::REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim")
            .AttrType(ge::OPTIONAL)
            .Int(0);

        this->SetInferShape(ge::InferShape);
        this->SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MeanReductionOverADimensionCustom);

} // namespace ops

kernel_src = '''
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t reduceDimSize, uint32_t dim,
                                uint32_t shapeDim0, uint32_t shapeDim1, uint32_t shapeDim2,
                                uint32_t stride)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        reduceDimSize_ = reduceDimSize;
        dim_ = dim;
        shapeDim0_ = shapeDim0;
        shapeDim1_ = shapeDim1;
        shapeDim2_ = shapeDim2;
        stride_ = stride;

        uint32_t inputLen = totalLen_ * reduceDimSize_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, inputLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        Compute(curLen, base);
        CopyOut(base, curLen);
    }

private:
    __aicore__ inline void Compute(uint32_t len, uint32_t outOffset)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            uint32_t outIdx = outOffset + i;
            uint32_t start = ComputeStartOffset(outIdx);
            float sum = 0.0f;
            for (uint32_t r = 0; r < reduceDimSize_; ++r) {
                uint32_t srcOffset = start + r * stride_;
                float val = xGm_[srcOffset];
                sum += val;
            }
            yLocal.SetValue(i, sum / reduceDimSize_);
        }

        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline uint32_t ComputeStartOffset(uint32_t outIdx)
    {
        if (dim_ == 0) {
            uint32_t d1 = shapeDim1_;
            uint32_t d2 = shapeDim2_;
            uint32_t i1 = outIdx / d2;
            uint32_t i2 = outIdx % d2;
            return i1 * d2 + i2;
        } else if (dim_ == 1) {
            uint32_t d0 = shapeDim0_;
            uint32_t d1 = shapeDim1_;
            uint32_t d2 = shapeDim2_;
            uint32_t i0 = outIdx / d2;
            uint32_t i2 = outIdx % d2;
            return i0 * d1 * d2 + i2;
        } else if (dim_ == 2) {
            uint32_t d0 = shapeDim0_;
            uint32_t d1 = shapeDim1_;
            uint32_t d2 = shapeDim2_;
            uint32_t i0 = outIdx / d1;
            uint32_t i1 = outIdx % d1;
            return (i0 * d1 + i1) * d2;
        } else {
            return outIdx * reduceDimSize_;
        }
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t reduceDimSize_{1};
    uint32_t dim_{0};
    uint32_t shapeDim0_{1};
    uint32_t shapeDim1_{1};
    uint32_t shapeDim2_{1};
    uint32_t stride_{1};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.reduceDimSize,
            tiling_data.dim,
            tiling_data.shapeDim0,
            tiling_data.shapeDim1,
            tiling_data.shapeDim2,
            tiling_data.stride);
    op.Process();
}
'''
