class KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom {
public:

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t inputN, uint32_t inputC,
                                uint32_t inputD, uint32_t inputH, uint32_t inputW,

                                uint32_t dilationD, uint32_t dilationH, uint32_t dilationW,
                                uint32_t groups, uint32_t outChannelTile, uint32_t batchTile)
    {
        inputN_ = inputN;
        inputC_ = inputC;
        inputD_ = inputD;
        inputH_ = inputH;
        inputW_ = inputW;
        kernelKD_ = kernelKD;
        kernelKH_ = kernelKH;
        kernelKW_ = kernelKW;
        outChannel_ = outChannel;
        outputD_ = outputD;
        outputH_ = outputH;
        outputW_ = outputW;
        strideD_ = strideD;
        strideH_ = strideH;
        strideW_ = strideW;
        padD_ = padD;
        padH_ = padH;
        padW_ = padW;

        dilationH_ = dilationH;
        dilationW_ = dilationW;
        groups_ = groups;
        outChannelTile_ = outChannelTile;
        batchTile_ = batchTile;

        inputHWin_ = inputD_ * inputH_ * inputW_;
        outputHWin_ = outputD_ * outputH_ * outputW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x);
        weightGm_.SetGlobalBuffer((__gm__ float*)weight);
        yGm_.SetGlobalBuffer((__gm__ float*)y);

        inputSize_ = inputN_ * inputC_ * inputD_ * inputH_ * inputW_;
        weightSize_ = outChannel_ * inputC_ * kernelKD_ * kernelKH_ * kernelKW_;
        outputSize_ = inputN_ * outChannel_ * outputD_ * outputH_ * outputW_;

        maxDim_ = inputD_ > inputH_ ? inputD_ : inputH_;
        maxDim_ = maxDim_ > inputW_ ? maxDim_ : inputW_;
        maxDim_ = maxDim_ > kernelKD_ ? maxDim_ : kernelKD_;
        maxDim_ = maxDim_ > kernelKH_ ? maxDim_ : kernelKH_;
        maxDim_ = maxDim_ > kernelKW_ ? maxDim_ : kernelKW_;
        maxDim_ = maxDim_ > outputD_ ? maxDim_ : outputD_;
        maxDim_ = maxDim_ > outputH_ ? maxDim_ : outputH_;
        maxDim_ = maxDim_ > outputW_ ? maxDim_ : outputW_;

        uint32_t inputBufSize = batchTile_ * inputC1_ * maxDim_ * maxDim_ * maxDim_;
        uint32_t weightBufSize = outChannelTile_ * inputC1_ * kernelKD_ * kernelKH_ * kernelKW_;
        uint32_t outputBufSize = batchTile_ * outChannelTile_ * outputD_ * outputH_ * outputW_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inputBufSize * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, weightBufSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outputBufSize * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t totalBlocks = inputN_ * ((outChannel_ + outChannelTile_ - 1) / outChannelTile_);
        if (blockIdx >= totalBlocks) return;

        uint32_t nIdx = blockIdx / ((outChannel_ + outChannelTile_ - 1) / outChannelTile_);
        uint32_t ocIdx = blockIdx % ((outChannel_ + outChannelTile_ - 1) / outChannelTile_);

        uint32_t startOC = ocIdx * outChannelTile_;
        uint32_t nOC = outChannel_;
        if (startOC >= nOC) return;

        uint32_t curOC = nOC - startOC;
        if (curOC > outChannelTile_) curOC = outChannelTile_;

        ProcessTile(nIdx, startOC, curOC);
    }

private:
    __aicore__ inline void ProcessTile(uint32_t nIdx, uint32_t startOC, uint32_t nOC)
    {
        uint32_t inStartD = 0;
        uint32_t inStartH = 0;
        uint32_t inStartW = 0;
        uint32_t kDStart = 0;
        uint32_t kHStart = 0;
        uint32_t kWStart = 0;

        uint32_t inputSizeL1 = nOC * kernelKD_ * kernelKH_ * kernelKW_;
        uint32_t outputSizeL1 = nOC * outputD_ * outputH_ * outputW_;

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();

        for (uint32_t ic = 0; ic < inputC_; ++ic) {
            uint32_t xOffset = nIdx * inputC_ * inputD_ * inputH_ * inputW_ +
                               ic * inputD_ * inputH_ * inputW_;
            uint32_t wOffset = startOC * inputC_ * kernelKD_ * kernelKH_ * kernelKW_ +
                               ic * kernelKD_ * kernelKH_ * kernelKW_;

            uint32_t copyInSize = inputD_ * inputH_ * inputW_;
            AscendC::DataCopy(xLocal, xGm_[xOffset], copyInSize);

            uint32_t copyWSize = nOC * kernelKD_ * kernelKH_ * kernelKW_;
            AscendC::DataCopy(wLocal, weightGm_[wOffset], copyWSize);

            Conv3d(xLocal, wLocal, yLocal, ic, startOC, nOC,
                   inputD_, inputH_, inputW_,
                   kernelKD_, kernelKH_, kernelKW_,
                   outputD_, outputH_, outputW_,
                   strideD_, strideH_, strideW_,
                   padD_, padH_, padW_,
                   dilationD_, dilationH_, dilationW_);
        }

        uint32_t yOffset = nIdx * outChannel_ * outputD_ * outputH_ * outputW_ +
                          startOC * outputD_ * outputH_ * outputW_;
        AscendC::DataCopy(yGm_[yOffset], yLocal, nOC * outputD_ * outputH_ * outputW_);

        outQueueY_.FreeTensor(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueW_.FreeTensor(wLocal);
    }

    __aicore__ inline void Conv3d(AscendC::LocalTensor<float>& xLocal,
                                  AscendC::LocalTensor<float>& wLocal,
                                  AscendC::LocalTensor<float>& yLocal,
                                  uint32_t ic, uint32_t startOC, uint32_t nOC,
                                  uint32_t inD, uint32_t inH, uint32_t inW,
                                  uint32_t kD, uint32_t kH, uint32_t kW,
                                  uint32_t outD, uint32_t outH, uint32_t outW,
                                  uint32_t sD, uint32_t sH, uint32_t sW,
                                  uint32_t pD, uint32_t pH, uint32_t pW,
                                  uint32_t dD, uint32_t dH, uint32_t dW)
    {
        for (uint32_t od = 0; od < outD; ++od) {
            for (uint32_t oh = 0; oh < outH; ++oh) {
                for (uint32_t ow = 0; ow < outW; ++ow) {
                    uint32_t baseID = od * sD - pD;
                    uint32_t baseIH = oh * sH - pH;
                    uint32_t baseIW = ow * sW - pW;

                    for (uint32_t oc = 0; oc < nOC; ++oc) {
                        float sum = 0.0f;

                        for (uint32_t kd = 0; kd < kD; ++kd) {
                            int32_t id = static_cast<int32_t>(baseID + kd * dD);
                            if (id < 0 || id >= static_cast<int32_t>(inD)) continue;

                            for (uint32_t kh = 0; kh < kH; ++kh) {
                                int32_t ih = static_cast<int32_t>(baseIH + kh * dH);
                                if (ih < 0 || ih >= static_cast<int32_t>(inH)) continue;

                                for (uint32_t kw = 0; kw < kW; ++kw) {
                                    int32_t iw = static_cast<int32_t>(baseIW + kw * dW);
                                    if (iw < 0 || iw >= static_cast<int32_t>(inW)) continue;

                                    uint32_t xIdx = id * inH * inW + ih * inW + iw;
                                    uint32_t wIdx = oc * kD * kH * kW + kd * kH * kW + kh * kW + kw;

                                    sum += xLocal.GetValue(xIdx) * wLocal.GetValue(wIdx);
                                }
                            }
                        }

                        uint32_t yIdx = oc * outD * outH * outW + od * outH * outW + oh * outW + ow;
                        yLocal.SetValue(yIdx, sum);
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> weightGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t inputN_ {0};
    uint32_t inputC_ {0};
    uint32_t inputD_ {0};
    uint32_t inputH_ {0};
    uint32_t inputW_ {0};
    uint32_t kernelKD_ {0};

    uint32_t outputD_ {0};
    uint32_t outputH_ {0};
    uint32_t outputW_ {0};
    uint32_t strideD_ {1};
    uint32_t strideH_ {1};
    uint32_t strideW_ {1};

    uint32_t outputHWin_ {0};
    uint32_t inputSize_ {0};
    uint32_t weightSize_ {0};
