kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr uint32_t TILE_H = 128;
constexpr uint32_t TILE_W = 1024;
constexpr uint32_t ELEMENTS_PER_TILE = TILE_H * TILE_W;
constexpr uint32_t OC_TILE = 8;
constexpr uint32_t IC_TILE = 8;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR w,
                                uint32_t batch, uint32_t inC, uint32_t outC,
                                uint32_t h, uint32_t w,
                                uint32_t tileH, uint32_t tileW,
                                uint32_t wTileC)
    {
        batch_ = batch;
        inC_ = inC;
        outC_ = outC;
        h_ = h;
        w_ = w;
        tileH_ = tileH;
        tileW_ = tileW;
        wTileC_ = wTileC;
        
        xGm_.SetGlobalBuffer((__gm__ float*)x);
        yGm_.SetGlobalBuffer((__gm__ float*)y);
        wGm_.SetGlobalBuffer((__gm__ float*)w);
        
        uint32_t inTileSize = ELEMENTS_PER_TILE;
        uint32_t outTileSize = ELEMENTS_PER_TILE;
        uint32_t wTileSize = inC_ * OC_TILE;
        
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inTileSize * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, outTileSize * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, wTileSize * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, ELEMENTS_PER_TILE * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (blockIdx >= batch_) return;
        
        for (uint32_t hTileIdx = 0; hTileIdx < h_; ++hTileIdx) {
            CopyInBlock(blockIdx, hTileIdx);
            
            uint32_t hOffset = hTileIdx * tileH_;
            uint32_t curTileH = (hOffset + tileH_ <= h_) ? tileH_ : (h_ - hOffset);
            uint32_t curTileSize = curTileH * w_;
            
            for (uint32_t wOffset = 0; wOffset < w_; wOffset += tileW_) {
                uint32_t curTileW = (wOffset + tileW_ <= w_) ? tileW_ : (w_ - wOffset);
                uint32_t curElems = curTileH * curTileW;
                
                uint32_t gmOffset = blockIdx * inC_ * h_ * w_ + hOffset * w_ + wOffset;
                CopyInTile(gmOffset, curElems);
                Compute(curElems);
                CopyOutTile(gmOffset, curElems);
            }
        }
    }

private:
    __aicore__ inline void CopyInBlock(uint32_t b, uint32_t hIdx)
    {
        uint32_t hOffset = hIdx * tileH_;
        uint32_t hLen = (hOffset + tileH_ <= h_) ? tileH_ : (h_ - hOffset);
        uint32_t blockOffset = b * inC_ * h_ * w_ + hOffset * w_;
        uint32_t len = inC_ * hLen * w_;
        
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[blockOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInTile(uint32_t gmOffset, uint32_t num)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], num);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t num)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        
        AscendC::AscendC::SetVectorizationLength(yLocal, num);
        AscendC::Sets(yLocal, 0.0f, num);
        
        for (uint32_t ocIdx = 0; ocIdx < outC_; ocIdx += OC_TILE) {
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            uint32_t wOffset = ocIdx * inC_;
            AscendC::DataCopy(wLocal, wGm_[wOffset], inC_ * OC_TILE);
            
            uint32_t hTileIdx = 0;
            uint32_t hOffset = 0;
            uint32_t wOffset = 0;
            
            for (uint32_t icIdx = 0; icIdx < inC_; icIdx += IC_TILE) {
                uint32_t icLen = (icIdx + IC_TILE <= inC_) ? IC_TILE : (inC_ - icIdx);
                
                uint32_t xOffset = icIdx * tileH_ * tileW_;
                uint32_t wSubOffset = icIdx * OC_TILE;
                
                AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                AscendC::Muls(tmpLocal, xLocal[xOffset], wLocal[wSubOffset], num, icLen, icLen, OC_TILE, OC_TILE);
                AscendC::Add(yLocal, yLocal, tmpLocal, num);
                tmpQueue_.FreeTensor(tmpLocal);
            }
            
            wQueue_.FreeTensor(wLocal);
        }
        
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutTile(uint32_t gmOffset, uint32_t num)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, num);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> wGm_;

    uint32_t batch_{0};
    uint32_t inC_{0};
    uint32_t outC_{0};
    uint32_t h_{0};
    uint32_t w_{0};
    uint32_t tileH_{0};
    uint32_t tileW_{0};
    uint32_t wTileC_{0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR w, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, y, w,
            tiling_data.batch, tiling_data.inC, tiling_data.outC,
            tiling_data.h, tiling_data.w,
            tiling_data.tileH, tiling_data.tileW,
            tiling_data.wTileC);
    op.Process();
}
