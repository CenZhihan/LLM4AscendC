#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include "tiling/tiling_api.h"

namespace optiling {
static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    LogSoftmaxCustomTilingData tiling;

    // Get input original shape
    const gert::Shape* in_shape = context->GetInputShape(0);
    auto origin_shape = in_shape->GetOriginShape();

    // Compose ge::Shape for tiling APIs and calculate m (product of non-last dims) and k (last dim)
    std::vector<int64_t> dims;
    dims.reserve(origin_shape.GetDimNum());
    for (int i = 0; i < origin_shape.GetDimNum(); ++i) {
        dims.emplace_back(origin_shape.GetDim(i));
    }
    if (dims.size() == 0) {
        return ge::GRAPH_FAILED;
    }
    ge::Shape srcShape(dims);

    uint32_t m = 1;
    for (size_t i = 0; i + 1 < dims.size(); ++i) {
        m *= static_cast<uint32_t>(dims[i]);
    }
    uint32_t k = static_cast<uint32_t>(dims.back());

    // Get tmp buffer size and tiling parameters for LogSoftMax
    const uint32_t typeSize = 4; // float
    uint32_t tmpMax = AscendC::GetLogSoftMaxMaxTmpSize(srcShape, typeSize, false);
    uint32_t tmpMin = AscendC::GetLogSoftMaxMinTmpSize(srcShape, typeSize, false);
    (void)tmpMin; // not used further, keep for reference
    uint32_t tmpSize = tmpMax;

    AscendC::LogSoftMaxTilingFunc(srcShape, typeSize, tmpSize, tiling.logSoftmaxTilingData);

    // Fill tiling data
    tiling.set_m(m);
    tiling.set_k(k);
    tiling.set_tmpBufferSize(tmpSize);

    // Set block dim (can be tuned)
    context->SetBlockDim(8);

    // Save tiling to buffer
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No GM workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
}

kernel_src="""
#include "kernel_operator.h"

using DTYPE = float;
constexpr int32_t BUFFER_NUM = 1; // simple single buffer per queue
constexpr uint32_t BLOCK_BYTES = 32;
constexpr uint32_t BLOCK_ELEMS_FLOAT = BLOCK_BYTES / sizeof(DTYPE);

class KernelLogSoftmax {
public:
    __aicore__ inline KernelLogSoftmax() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t m, uint32_t k,
                                uint32_t tmpBufferSize,
                                const LogSoftMaxTiling &logsoftmaxTilingIn)
    {
        this->m = m;
        this->k = k;
        this->tmpSize = tmpBufferSize;
        this->logsoftmaxTiling = logsoftmaxTilingIn;

        // Split work by M across blocks
        uint32_t blocks = AscendC::GetBlockNum();
        uint32_t blockIdx = AscendC::GetBlockIdx();
        uint64_t rowsPerBlock = m / blocks;
        uint64_t tailRows = m % blocks;
        this->rowStart = rowsPerBlock * blockIdx + (blockIdx < tailRows ? blockIdx : tailRows);
        this->rowCount = rowsPerBlock + (blockIdx < tailRows ? 1 : 0);

        // GM tensor binding starting at rowStart
        xGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(x) + this->rowStart * this->k, this->rowCount * this->k);
        yGm.SetGlobalBuffer(reinterpret_cast<__gm__ DTYPE *>(y) + this->rowStart * this->k, this->rowCount * this->k);

        // Queues: process one row at a time for simplicity
        uint32_t perTileElems = this->k; // one row
        uint32_t perTileBytes = perTileElems * sizeof(DTYPE);

        pipe.InitBuffer(inQueueX, BUFFER_NUM, perTileBytes);
        pipe.InitBuffer(outQueueY, BUFFER_NUM, perTileBytes);

        // max/sum locals: last axis is one data block (32 bytes) per row
        pipe.InitBuffer(auxQueueMax, BUFFER_NUM, BLOCK_ELEMS_FLOAT * sizeof(DTYPE));
        pipe.InitBuffer(auxQueueSum, BUFFER_NUM, BLOCK_ELEMS_FLOAT * sizeof(DTYPE));

        // tmp buffer for LogSoftMax API
        pipe.InitBuffer(tmpQueue, BUFFER_NUM, this->tmpSize);
    }

    __aicore__ inline void Process()
    {
        for (uint64_t r = 0; r < this->rowCount; ++r) {
            CopyIn(r);
            Compute();
            CopyOut(r);
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.AllocTensor<DTYPE>();
        AscendC::DataCopy(xLocal, xGm[rowIdx * this->k], this->k);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<DTYPE> xLocal = inQueueX.DeQue<DTYPE>();
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> maxLocal = auxQueueMax.AllocTensor<DTYPE>();
        AscendC::LocalTensor<DTYPE> sumLocal = auxQueueSum.AllocTensor<DTYPE>();
        AscendC::LocalTensor<uint8_t> tmpLocal = tmpQueue.AllocTensor<uint8_t>();

        // Shape info: process 1 x k at a time, last axis is reduction axis
        AscendC::SoftMaxShapeInfo softmaxInfo = {1, this->k, 1, this->k};

        // LogSoftMax: dst, sum, max, src, tmp, tiling, shape-info
        AscendC::LogSoftMax<DTYPE, false>(yLocal, sumLocal, maxLocal, xLocal, tmpLocal,
                                           this->logsoftmaxTiling, softmaxInfo);

        outQueueY.EnQue<DTYPE>(yLocal);

        inQueueX.FreeTensor(xLocal);
        auxQueueMax.FreeTensor(maxLocal);
        auxQueueSum.FreeTensor(sumLocal);
        tmpQueue.FreeTensor(tmpLocal);
    }

    __aicore__ inline void CopyOut(uint64_t rowIdx)
    {
        AscendC::LocalTensor<DTYPE> yLocal = outQueueY.DeQue<DTYPE>();
        AscendC::DataCopy(yGm[rowIdx * this->k], yLocal, this->k);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> auxQueueMax;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> auxQueueSum;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue;

    AscendC::GlobalTensor<DTYPE> xGm;
    AscendC::GlobalTensor<DTYPE> yGm;

    uint32_t m = 0;
    uint32_t k = 0;
    uint32_t tmpSize = 0;
    uint64_t rowStart = 0;
    uint64_t rowCount = 0;
    LogSoftMaxTiling logsoftmaxTiling;
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelLogSoftmax op;
    op.Init(x, y,
            tilingData.m, tilingData.k,
            tilingData.tmpBufferSize,
            tilingData.logSoftmaxTilingData);
    op.Process();
}
