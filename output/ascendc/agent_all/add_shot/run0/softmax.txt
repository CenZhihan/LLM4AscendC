host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {
const uint32_t BLOCK_DIM = 1;

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    SoftmaxCustomTilingData tiling;

    // Set block dim to 1 for simplicity
    context->SetBlockDim(BLOCK_DIM);

    // Obtain input shape
    const gert::Shape* inShape = context->GetInputShape(0);
    ge::Shape srcShape = inShape->GetOriginShape();

    // Decide a proper local workspace size for SoftMax tiling. Use the minimum to ensure correctness.
    const uint32_t localWorkSpaceSize = AscendC::GetSoftMaxMinTmpSize(srcShape, sizeof(float), false);

    // Get SoftMax tiling parameters
    AscendC::SoftMaxTilingFunc(srcShape, sizeof(float), localWorkSpaceSize, tiling.softmaxTilingData);

    // Save tiling
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No global workspace required
    size_t* currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    context->SetTilingKey(1);
    return ge::GRAPH_SUCCESS;
}
}

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* x_shape = context->GetInputShape(0);
    gert::Shape* y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}
static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    const auto inputDataType = context->GetInputDataType(0);
    context->SetOutputDataType(0, inputDataType);
    return ge::GRAPH_SUCCESS;
}
}

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});
        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
}

kernel_src="""
#include "kernel_operator.h"

template <typename T>
class KernelSoftmax {
public:
    __aicore__ inline KernelSoftmax() {}
    __aicore__ inline void Init(GM_ADDR x_gm, GM_ADDR y_gm, const AscendC::SoftMaxTiling &tilingData)
    {
        elementNumPerBlk = 32 / sizeof(T);
        // Set shape from tiling
        height = tilingData.srcM;
        width = tilingData.srcK;

        xGlobal.SetGlobalBuffer((__gm__ T*)x_gm);
        yGlobal.SetGlobalBuffer((__gm__ T*)y_gm);

        // Initialize local buffers
        pipe.InitBuffer(inQueueX, 1, height * width * sizeof(T));
        pipe.InitBuffer(maxQueue, 1, height * elementNumPerBlk * sizeof(T));
        pipe.InitBuffer(sumQueue, 1, height * elementNumPerBlk * sizeof(T));
        pipe.InitBuffer(outQueueY, 1, height * width * sizeof(T));

        tiling = tilingData;
    }

    __aicore__ inline void Process()
    {
        CopyIn();
        Compute();
        CopyOut();
    }

private:
    __aicore__ inline void CopyIn()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX.AllocTensor<T>();
        AscendC::DataCopy(xLocal, xGlobal, height * width);
        inQueueX.EnQue(xLocal);
    }

    __aicore__ inline void Compute()
    {
        AscendC::LocalTensor<T> xLocal = inQueueX.DeQue<T>();
        AscendC::LocalTensor<T> sumLocal = sumQueue.AllocTensor<T>();
        AscendC::LocalTensor<T> maxLocal = maxQueue.AllocTensor<T>();
        AscendC::LocalTensor<T> yLocal = outQueueY.AllocTensor<T>();

        AscendC::SoftMaxShapeInfo shapeInfo = {height, width, height, width};
        AscendC::SoftMax<T>(yLocal, sumLocal, maxLocal, xLocal, tiling, shapeInfo);

        outQueueY.EnQue<T>(yLocal);
        maxQueue.FreeTensor(maxLocal);
        sumQueue.FreeTensor(sumLocal);
        inQueueX.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut()
    {
        AscendC::LocalTensor<T> yLocal = outQueueY.DeQue<T>();
        AscendC::DataCopy(yGlobal, yLocal, height * width);
        outQueueY.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueueX;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> maxQueue;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> sumQueue;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueueY;
    AscendC::GlobalTensor<T> xGlobal, yGlobal;
    uint32_t elementNumPerBlk = 0;
    uint32_t width = 0;
    uint32_t height = 0;
    AscendC::SoftMaxTiling tiling;
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelSoftmax<float> op;
    op.Init(x, y, tilingData.softmaxTilingData);
    op.Process();
}
"""
