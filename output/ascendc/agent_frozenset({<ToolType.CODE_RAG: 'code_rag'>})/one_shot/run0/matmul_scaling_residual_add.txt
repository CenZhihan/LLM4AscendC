#include "register/tilingdata_base.h"
#include "register/op_def_registry.h"

namespace optiling {

BEGIN_TILING_DATA_DEF(MatmulScalingResidualAddCustomTilingData)
  TILING_DATA_FIELD_DEF(uint32_t, totalLen);
  TILING_DATA_FIELD_DEF(uint32_t, tileLen);
  TILING_DATA_FIELD_DEF(uint32_t, blockDim);
  // Matmul tiling blob
  TILING_DATA_FIELD_DEF(TCubeTiling, mmTilingData);
END_TILING_DATA_DEF;

REGISTER_TILING_DATA_CLASS(MatmulScalingResidualAddCustom, MatmulScalingResidualAddCustomTilingData)

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// UB budget for post-scale pass only (Y read/modify/write):
// inQueue: 1 * tile, outQueue: 1 * tile
// bytes = 2 * tile * 4
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulScalingResidualAddCustomTilingData td;

    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);
    const auto *b_shape = context->GetInputShape(2);

    // Shapes:
    // x: [M, K]
    // w: [N, K] (PyTorch linear expects weight [out_features, in_features])
    // b: [N]
    const uint64_t M = x_shape->GetOriginShape().GetDim(0);
    const uint64_t K = x_shape->GetOriginShape().GetDim(1);
    const uint64_t N = w_shape->GetOriginShape().GetDim(0);

    // Set output shape
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, M);
    y_shape->SetDim(1, N);

    // Prepare post-scale tiling (y = (x @ w^T + b) * (1 + s))
    const uint32_t totalLen = static_cast<uint32_t>(M * N);

    uint32_t maxTile = UB_BUDGET_BYTES / 8;  // 2 * tile * 4B
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLen < tileLen) {
        const uint32_t aligned = AlignDown(totalLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLen;
    }
    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    // Fill our post-scale fields
    td.set_totalLen(totalLen);
    td.set_tileLen(tileLen);
    td.set_blockDim(blockDim);

    // Generate matmul tiling for A(MxK) * B^T(NxK) => C(MxN), with bias
    // We request transB = true since w is [N, K] and we need w^T during matmul.
    // The platform tiler will fill mmTilingData accordingly.
    TCubeTiling &mm = td.mutable_mmTilingData();
    // Initialize as default, then set basic shapes for framework tiler to refine.
    mm.M = static_cast<uint32_t>(M);
    mm.N = static_cast<uint32_t>(N);
    mm.K = static_cast<uint32_t>(K);
    mm.typeA = ge::DT_FLOAT;
    mm.typeB = ge::DT_FLOAT;
    mm.typeC = ge::DT_FLOAT;
    mm.transposeA = 0;
    mm.transposeB = 1;
    // Leave other fields for backend default optimization.

    td.SaveToBuffer(context->GetRawTilingData()->GetData(),
                    context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(td.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t INOUT_BUFFERS = 2;

class KernelMatmulScalingResidualAddCustomPost {
public:
    __aicore__ inline KernelMatmulScalingResidualAddCustomPost() {}

    __aicore__ inline void Init(GM_ADDR y, GM_ADDR scaling,
                                uint32_t totalLen, uint32_t tileLen)
    {
        totalLen_ = totalLen;
        tileLen_ = tileLen;

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        sGm_.SetGlobalBuffer((__gm__ float*)scaling, 1);

        pipe_.InitBuffer(inQueueY_, INOUT_BUFFERS, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, INOUT_BUFFERS, tileLen_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        // Load scaling scalar, compute factor = 1 + s
        AscendC::LocalTensor<float> sLocal = scalarQueue_.AllocTensor<float>();
        AscendC::DataCopy(sLocal, sGm_[0], 1);
        scalarQueue_.EnQue<float>(sLocal);
        sLocal = scalarQueue_.DeQue<float>();
        float one = 1.0f;
        float sVal = 0.0f;
        AscendC::LocalTensor<float> tmp = scalarQueue_.AllocTensor<float>();
        // tmp[0] = sLocal[0] (copy through UB)
        AscendC::DataCopy(tmp, sLocal, 1);
        // Convert back via scalar read
        sVal = tmp.GetValue(0);
        scalarQueue_.FreeTensor(tmp);
        scalarQueue_.FreeTensor(sLocal);
        const float factor = one + sVal;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t stride = blockNum * tileLen_;

        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + stride;
            uint32_t nextLen = 0U;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen, factor);
            CopyOut(base, curLen);

            if (nextLen == 0U) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint32_t len, float factor)
    {
        AscendC::LocalTensor<float> yLocalIn = inQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> yLocalOut = outQueueY_.AllocTensor<float>();
        AscendC::Muls(yLocalOut, yLocalIn, factor, len);
        inQueueY_.FreeTensor(yLocalIn);
        outQueueY_.EnQue(yLocalOut);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INOUT_BUFFERS> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, INOUT_BUFFERS> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> sGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void matmul_scaling_residual_add_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    // 1) Run GEMM with bias: y = x @ w^T + b
    // Prepare Matmul tiling
    GET_TILING_DATA_MEMBER(MatmulScalingResidualAddCustomTilingData, td, tdObj, tilingGm);
    GET_TILING_DATA_MEMBER(MatmulScalingResidualAddCustomTilingData, mmTilingData, mmTilingDataObj, tilingGm);

    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    AscendC::TPipe mmPipe;
    AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> mm;
#ifndef __CCE_KT_TEST__
    REGIST_MATMUL_OBJ(&mmPipe, GetSysWorkSpacePtr(), mm, &mmTilingDataObj);
#endif
    // Init and process GEMM
    mm.Init(x, w, b, y, workspace, mmTilingDataObj);
    mm.Process(&mmPipe);

    // 2) Post-process: y = y * (1 + scaling[0]) using tile-based VEC pipeline
    KernelMatmulScalingResidualAddCustomPost post;
    post.Init(y, scaling, tdObj.totalLen, tdObj.tileLen);
    post.Process();
}
