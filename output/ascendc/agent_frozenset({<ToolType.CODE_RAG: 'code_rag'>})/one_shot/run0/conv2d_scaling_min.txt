kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dScalingMinCustom {
public:
    __aicore__ inline KernelConv2dScalingMinCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR s, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t hin, uint32_t win,
                                uint32_t cout, uint32_t kh, uint32_t kw,
                                uint32_t oh, uint32_t ow,
                                uint32_t tilePixels, uint32_t hasBias)
    {
        n_ = n; cin_ = cin; hin_ = hin; win_ = win;
        cout_ = cout; kh_ = kh; kw_ = kw; oh_ = oh; ow_ = ow;
        tilePixels_ = tilePixels;
        hasBias_ = (hasBias != 0);

        xSize_ = n_ * cin_ * hin_ * win_;
        wSize_ = cout_ * cin_ * kh_ * kw_;
        bSize_ = hasBias_ ? cout_ : 0u;
        sSize_ = 1u;
        ySize_ = n_ * oh_ * ow_; // channel is 1

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, bSize_);
        }
        sGm_.SetGlobalBuffer((__gm__ float*)s, sSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize_);

        // Buffers:
        //  - xRow: KW floats
        //  - wRow: KW floats
        //  - tmpVec: KW floats
        //  - scalars: length 1 for sum/min/bias/scale
        const uint32_t rowBytes = kw_ * sizeof(float);
        pipe_.InitBuffer(xRowQ_, 1, rowBytes);
        pipe_.InitBuffer(wRowQ_, 1, rowBytes);
        pipe_.InitBuffer(tmpVecQ_, 1, rowBytes);
        pipe_.InitBuffer(scalarQ_, 8, sizeof(float));
        pipe_.InitBuffer(outQ_, 2, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (oh_ == 0 || ow_ == 0 || n_ == 0) return;

        // Load scale scalar once per block
        AscendC::LocalTensor<float> scaleLocal = scalarQ_.AllocTensor<float>();
        AscendC::DataCopy(scaleLocal, sGm_[0], 1);

        const uint32_t totalPixels = n_ * oh_ * ow_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t idx = blockIdx; idx < totalPixels; idx += blockNum) {
            uint32_t nIdx, ohIdx, owIdx;
            UnravelPixelIndex(idx, nIdx, ohIdx, owIdx);

            // Initialize min accumulator with big value
            AscendC::LocalTensor<float> minLocal = scalarQ_.AllocTensor<float>();
            AscendC::Dup<float>(minLocal, 3.4028235e38f, 1);

            for (uint32_t co = 0; co < cout_; ++co) {
                // sum over (ci, kh, kw)
                AscendC::LocalTensor<float> sumLocal = scalarQ_.AllocTensor<float>();
                AscendC::Dup<float>(sumLocal, 0.0f, 1);

                for (uint32_t ci = 0; ci < cin_; ++ci) {
                    for (uint32_t r = 0; r < kh_; ++r) {
                        // Load x row [kw_] and w row [kw_]
                        AscendC::LocalTensor<float> xRow = xRowQ_.AllocTensor<float>();
                        AscendC::LocalTensor<float> wRow = wRowQ_.AllocTensor<float>();
                        const uint32_t xOff = OffsetX(nIdx, ci, ohIdx + r, owIdx);
                        const uint32_t wOff = OffsetW(co, ci, r, 0);
                        AscendC::DataCopy(xRow, xGm_[xOff], kw_);
                        AscendC::DataCopy(wRow, wGm_[wOff], kw_);

                        AscendC::LocalTensor<float> tmpVec = tmpVecQ_.AllocTensor<float>();
                        AscendC::Mul(tmpVec, xRow, wRow, kw_);

                        // Reduce sum of tmpVec into scalar partial
                        AscendC::LocalTensor<float> partial = scalarQ_.AllocTensor<float>();
                        AscendC::ReduceSum(partial, tmpVec, kw_);

                        // sumLocal += partial
                        AscendC::Add(sumLocal, sumLocal, partial, 1);

                        scalarQ_.FreeTensor(partial);
                        tmpVecQ_.FreeTensor(tmpVec);
                        wRowQ_.FreeTensor(wRow);
                        xRowQ_.FreeTensor(xRow);
                    }
                }

                // Add bias if present
                if (hasBias_) {
                    AscendC::LocalTensor<float> bLocal = scalarQ_.AllocTensor<float>();
                    AscendC::DataCopy(bLocal, bGm_[co], 1);
                    AscendC::Add(sumLocal, sumLocal, bLocal, 1);
                    scalarQ_.FreeTensor(bLocal);
                }

                // Multiply by scale
                AscendC::Mul(sumLocal, sumLocal, scaleLocal, 1);

                // minLocal = min(minLocal, sumLocal)
                AscendC::Min(minLocal, minLocal, sumLocal, 1);

                scalarQ_.FreeTensor(sumLocal);
            }

            // Write result to GM at y offset
            const uint32_t yOff = OffsetY(nIdx, ohIdx, owIdx);
            AscendC::LocalTensor<float> yScalar = outQ_.AllocTensor<float>();
            AscendC::DataCopy(yScalar, minLocal, 1);
            AscendC::DataCopy(yGm_[yOff], yScalar, 1);
            outQ_.FreeTensor(yScalar);
            scalarQ_.FreeTensor(minLocal);
        }

        scalarQ_.FreeTensor(scaleLocal);
    }

private:
    __aicore__ inline uint32_t OffsetX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((n * cin_ + c) * hin_ + h) * win_ + w);
    }

    __aicore__ inline uint32_t OffsetW(uint32_t co, uint32_t ci, uint32_t r, uint32_t c) const
    {
        return (((co * cin_ + ci) * kh_ + r) * kw_ + c);
    }

    __aicore__ inline uint32_t OffsetY(uint32_t n, uint32_t h, uint32_t w) const
    {
        return ((n * oh_ + h) * ow_ + w);
    }

    __aicore__ inline void UnravelPixelIndex(uint32_t idx, uint32_t &nIdx, uint32_t &ohIdx, uint32_t &owIdx) const
    {
        const uint32_t pixPerN = oh_ * ow_;
        nIdx = idx / pixPerN;
        const uint32_t rem = idx % pixPerN;
        ohIdx = rem / ow_;
        owIdx = rem % ow_;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xRowQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 8> scalarQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> sGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, hin_{0}, win_{0};
    uint32_t cout_{0}, kh_{0}, kw_{0}, oh_{0}, ow_{0};
    uint32_t tilePixels_{1};
    bool hasBias_{false};

    uint32_t xSize_{0}, wSize_{0}, bSize_{0}, sSize_{0}, ySize_{0};
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dScalingMinCustom op;
    op.Init(x, weight, bias, scale, y,
            tiling_data.n, tiling_data.cin, tiling_data.hin, tiling_data.win,
            tiling_data.cout, tiling_data.kh, tiling_data.kw,
            tiling_data.oh, tiling_data.ow,
            tiling_data.tilePixels, tiling_data.hasBias);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor conv2d_scaling_min_impl_npu(const at::Tensor& x,
                                              const at::Tensor& weight,
                                              c10::optional<at::Tensor> bias_opt,
                                              const at::Tensor& scale) {
    // Output shape: [N, 1, H-K+1, W-K+1]
    TORCH_CHECK(scale.numel() == 1, "scale must be a scalar tensor with numel == 1");
    auto N = x.size(0);
    auto H = x.size(2);
    auto W = x.size(3);
    auto Kh = weight.size(2);
    auto Kw = weight.size(3);
    TORCH_CHECK(H >= Kh && W >= Kw, "Invalid spatial size for valid conv (no padding).");

    auto Oh = H - Kh + 1;
    auto Ow = W - Kw + 1;

    at::Tensor y = at::empty({N, 1, Oh, Ow}, x.options());
    at::Tensor bias = bias_opt.has_value() ? bias_opt.value() : at::Tensor();

    EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, bias, scale, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_scaling_min_custom", &conv2d_scaling_min_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_scaling_min_custom", &conv2d_scaling_min_impl_npu,
          "Conv2d + scaling + channel-wise min (Conv2dScalingMinCustom on NPU)");
}
