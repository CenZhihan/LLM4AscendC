host_operator_src="""

#include "cumsum_exclusive_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// UB model for this kernel:
//   inQueueX:   2 * tile
//   outQueueY:  2 * tile
//   tmpQueue:   1 * tile
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumExclusiveCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const int64_t rank = static_cast<int64_t>(shape->GetOriginShape().GetDimNum());

    int32_t axis = -1;
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    if (attrs != nullptr && attrs->GetAttrNum() > 0) {
        const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
        if (dimAttr) axis = *dimAttr;
    }

    int64_t normAxis = axis;
    if (normAxis < 0) normAxis += rank;
    if (normAxis < 0 || normAxis >= rank) {
        // Fallback: clamp to last axis
        normAxis = rank - 1;
    }

    uint64_t totalSize = 1;
    for (int i = 0; i < rank; ++i) totalSize *= static_cast<uint64_t>(shape->GetOriginShape().GetDim(i));

    const uint32_t axisSize = static_cast<uint32_t>(shape->GetOriginShape().GetDim(normAxis));

    // inner = product of dims after axis; kernel assumes inner == 1 (axis is last dimension)
    uint64_t inner = 1;
    for (int i = static_cast<int>(normAxis) + 1; i < rank; ++i) inner *= static_cast<uint64_t>(shape->GetOriginShape().GetDim(i));

    // Compute outer number of rows to scan
    uint32_t outerNum = 1;
    if (axisSize > 0) {
        outerNum = static_cast<uint32_t>(totalSize / axisSize);
    }

    // tileLen from UB budget: bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (axisSize < tileLen) {
        const uint32_t aligned = AlignDown(axisSize, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : axisSize;
    }
    if (tileLen == 0) tileLen = axisSize;

    // Block dim: balance by number of rows (outer)
    uint32_t blockDim = outerNum;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    // If axis is not last (inner != 1), we still set tiling; kernel will only support inner==1 fast path.
    context->SetBlockDim(blockDim);

    tiling.set_outerNum(outerNum);
    tiling.set_axisSize(axisSize);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_axis(static_cast<int32_t>(normAxis));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCumsumExclusiveCustom {
public:
    __aicore__ inline KernelCumsumExclusiveCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerNum, uint32_t axisSize,
                                uint32_t tileLength, int32_t axis)
    {
        outer_ = outerNum;
        axisSize_ = axisSize;
        tileLen_ = tileLength;
        axis_ = axis; // normalized; kernel supports last-axis fast path

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(outer_) * axisSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(outer_) * axisSize_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outer_ == 0 || axisSize_ == 0 || tileLen_ == 0) return;

        // Kernel assumes cumsum axis is the last dimension (inner stride == 1).
        // For other cases, behavior is undefined.
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < outer_; row += blockNum) {
            float carry = 0.0f;
            uint32_t processed = 0;
            while (processed < axisSize_) {
                const uint32_t curLen = (axisSize_ - processed > tileLen_) ? tileLen_ : (axisSize_ - processed);
                const uint64_t gmOffset = static_cast<uint64_t>(row) * axisSize_ + processed;

                CopyIn(gmOffset, curLen);
                Compute(carry, curLen, carry);
                CopyOut(gmOffset, curLen);

                processed += curLen;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    // Compute exclusive cumsum for a tile with an incoming carry; returns updated carryOut.
    __aicore__ inline void Compute(float carryIn, uint32_t len, float &carryOut)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // yLocal = xLocal
        AscendC::DataCopy(yLocal, xLocal, len);

        // Inclusive scan in-place using Hillis-Steele: y += y shifted by 2^k
        for (uint32_t step = 1; step < len; step <<= 1) {
            // tLocal = shiftRight(yLocal, step), zero-filled
            // Zero tLocal
            AscendC::Muls(tLocal, yLocal, 0.0f, len); // fill zeros using mul with 0
            // Copy yLocal[0 : len-step] -> tLocal[step : len]
            AscendC::DataCopy(tLocal[step], yLocal, len - step);
            // yLocal = yLocal + tLocal
            AscendC::Add(yLocal, yLocal, tLocal, len);
        }

        // last element of inclusive scan is the tile sum
        float tileSum = yLocal.GetValue(len - 1);
        carryOut = carryIn + tileSum;

        // Compute exclusive: shiftRight(inclusive, 1) and add carryIn
        // tLocal = shiftRight(yLocal, 1)
        AscendC::Muls(tLocal, yLocal, 0.0f, len);
        if (len > 1) {
            AscendC::DataCopy(tLocal[1], yLocal, len - 1);
        }
        if (carryIn != 0.0f) {
            AscendC::Adds(tLocal, tLocal, carryIn, len);
        }

        // yLocal = tLocal (final output tile)
        AscendC::DataCopy(yLocal, tLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_ {0};
    uint32_t axisSize_ {0};
    uint32_t tileLen_ {0};
    int32_t axis_ { -1 };
};

extern "C" __global__ __aicore__ void cumsum_exclusive_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumExclusiveCustom op;
    op.Init(x, y, tiling_data.outerNum, tiling_data.axisSize, tiling_data.tileLength, tiling_data.axis);
    op.Process();
}
"""
