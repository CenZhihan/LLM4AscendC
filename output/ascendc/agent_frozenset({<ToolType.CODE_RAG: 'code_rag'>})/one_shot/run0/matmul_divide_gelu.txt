host_operator_src="""
#include "matmul_divide_gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0u) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulDivideGeluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);  // [M, K]
    const gert::Shape *w_shape = context->GetInputShape(1);  // [N, K]
    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    // w_shape dim1 should equal K (input_size)
    // b: [N], divisor: [1] (scalar tensor)

    // UB budget and kChunk choice:
    // Buffers per inner-loop:
    //   x_chunk: kChunk
    //   w_chunk: kChunk
    //   mul_tmp: kChunk
    //   scalar temps: ~8 floats
    // Bytes ~ (3*kChunk + 8) * 4
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
    static constexpr uint32_t ALIGN_ELEMS = 256;

    uint32_t maxKChunk = UB_BUDGET_BYTES / 12; // ~ (3*k) * 4 => 12 bytes/elem
    if (maxKChunk == 0) maxKChunk = ALIGN_ELEMS;
    maxKChunk = AlignDown(maxKChunk, ALIGN_ELEMS);
    if (maxKChunk < ALIGN_ELEMS) maxKChunk = ALIGN_ELEMS;

    uint32_t kChunk = (K >= ALIGN_ELEMS) ? AlignDown(maxKChunk, ALIGN_ELEMS) : K;
    if (kChunk == 0) kChunk = K;

    uint32_t blockDim = (M == 0) ? 1u : CeilDiv(M, 1u);
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_kChunk(kChunk);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr uint32_t XBUF_NUM = 1;
constexpr uint32_t WBUF_NUM = 1;
constexpr uint32_t TMPBUF_NUM = 1;
constexpr uint32_t SCALARBUF_NUM = 4;

class KernelMatmulDivideGeluCustom {
public:
    __aicore__ inline KernelMatmulDivideGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t kChunk)
    {
        M_ = M; N_ = N; K_ = K; kChunk_ = (kChunk == 0 ? K : kChunk);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        dGm_.SetGlobalBuffer((__gm__ float*)divisor, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        pipe_.InitBuffer(xQueue_, XBUF_NUM, kChunk_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, WBUF_NUM, kChunk_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMPBUF_NUM, kChunk_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, SCALARBUF_NUM, sizeof(float)); // 1-element scalars
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Load divisor and compute reciprocal once per core
        AscendC::LocalTensor<float> divLocal = scalarQueue_.AllocTensor<float>();
        AscendC::DataCopy(divLocal, dGm_[0], 1);
        AscendC::LocalTensor<float> invDivLocal = scalarQueue_.AllocTensor<float>();
        AscendC::Reciprocal(invDivLocal, divLocal, 1);

        // Constants for GELU approx
        AscendC::LocalTensor<float> halfLocal = scalarQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> oneLocal  = scalarQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> c0Local   = scalarQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> c1Local   = scalarQueue_.AllocTensor<float>();
        AscendC::Duplicate(halfLocal, 0.5f, 1);
        AscendC::Duplicate(oneLocal, 1.0f, 1);
        AscendC::Duplicate(c0Local, 0.7978845608028654f, 1); // sqrt(2/pi)
        AscendC::Duplicate(c1Local, 0.044715f, 1);

        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            // For each output feature n, compute dot(x_row, w_row_n) + b[n]
            const uint64_t xRowBase = static_cast<uint64_t>(m) * K_;
            const uint64_t yRowBase = static_cast<uint64_t>(m) * N_;
            for (uint32_t n = 0; n < N_; ++n) {
                // acc = b[n]
                AscendC::LocalTensor<float> acc = scalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(acc, bGm_[n], 1);

                for (uint32_t k0 = 0; k0 < K_; k0 += kChunk_) {
                    uint32_t curK = K_ - k0;
                    if (curK > kChunk_) curK = kChunk_;

                    AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                    AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                    AscendC::LocalTensor<float> mulLocal = tmpQueue_.AllocTensor<float>();

                    AscendC::DataCopy(xLocal, xGm_[xRowBase + k0], curK);
                    AscendC::DataCopy(wLocal, wGm_[static_cast<uint64_t>(n) * K_ + k0], curK);
                    AscendC::Mul(mulLocal, xLocal, wLocal, curK);

                    AscendC::LocalTensor<float> red = scalarQueue_.AllocTensor<float>();
                    AscendC::ReduceSum(red, mulLocal, curK);
                    AscendC::Add(acc, acc, red, 1);

                    scalarQueue_.FreeTensor(red);
                    tmpQueue_.FreeTensor(mulLocal);
                    wQueue_.FreeTensor(wLocal);
                    xQueue_.FreeTensor(xLocal);
                }

                // Divide by divisor
                AscendC::Mul(acc, acc, invDivLocal, 1);

                // GELU approximation using tanh
                // t1 = acc
                AscendC::LocalTensor<float> t1 = scalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(t1, acc, 1);
                // t2 = acc * acc
                AscendC::LocalTensor<float> t2 = scalarQueue_.AllocTensor<float>();
                AscendC::Mul(t2, acc, acc, 1);
                // t3 = t2 * acc = acc^3
                AscendC::LocalTensor<float> t3 = scalarQueue_.AllocTensor<float>();
                AscendC::Mul(t3, t2, acc, 1);
                // inner = c1 * t3 + t1
                AscendC::LocalTensor<float> inner = scalarQueue_.AllocTensor<float>();
                AscendC::Mul(inner, c1Local, t3, 1);
                AscendC::Add(inner, inner, t1, 1);
                // inner = c0 * inner
                AscendC::Mul(inner, c0Local, inner, 1);
                // t = tanh(inner)
                AscendC::LocalTensor<float> t = scalarQueue_.AllocTensor<float>();
                AscendC::Tanh(t, inner, 1);
                // out = 0.5 * t1 * (1 + t)
                AscendC::LocalTensor<float> onePlusT = scalarQueue_.AllocTensor<float>();
                AscendC::Add(onePlusT, oneLocal, t, 1);
                AscendC::LocalTensor<float> prod = scalarQueue_.AllocTensor<float>();
                AscendC::Mul(prod, t1, onePlusT, 1);
                AscendC::Mul(prod, prod, halfLocal, 1);

                // Store to GM
                AscendC::DataCopy(yGm_[yRowBase + n], prod, 1);

                // Free temporaries
                scalarQueue_.FreeTensor(prod);
                scalarQueue_.FreeTensor(onePlusT);
                scalarQueue_.FreeTensor(t);
                scalarQueue_.FreeTensor(inner);
                scalarQueue_.FreeTensor(t3);
                scalarQueue_.FreeTensor(t2);
                scalarQueue_.FreeTensor(t1);
                scalarQueue_.FreeTensor(acc);
            }
        }

        // Free constants
        scalarQueue_.FreeTensor(c1Local);
        scalarQueue_.FreeTensor(c0Local);
        scalarQueue_.FreeTensor(oneLocal);
        scalarQueue_.FreeTensor(halfLocal);
        scalarQueue_.FreeTensor(invDivLocal);
        scalarQueue_.FreeTensor(divLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, XBUF_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, WBUF_NUM> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMPBUF_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, SCALARBUF_NUM> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> dGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t kChunk_{0};
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulDivideGeluCustom op;
    op.Init(x, w, b, divisor, y, tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.kChunk);
    op.Process();
}
