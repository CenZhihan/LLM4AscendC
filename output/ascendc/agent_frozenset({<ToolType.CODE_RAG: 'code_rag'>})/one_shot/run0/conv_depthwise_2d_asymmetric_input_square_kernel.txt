static ge::graphStatus TilingFunc(gert::TilingContext *context)

    ConvDepthwise2dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect NCHW for x: [N, C, H, W]
    const uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t C = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetDim(3));

    // weight: [C, 1, kH, kW], depthwise
    const uint32_t Kh = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetDim(3));

    // Specialization: stride=1, pad=0, dilation=1
    const uint32_t outH = H - Kh + 1;
    const uint32_t outW = W - Kw + 1;

    // Tile scheme: one tile = one (n, c, outH_row), computing full width row (tileWidth = outW)
    const uint64_t totalTiles64 = static_cast<uint64_t>(N) * static_cast<uint64_t>(C) * static_cast<uint64_t>(outH);
    const uint32_t totalTiles = static_cast<uint32_t>(totalTiles64);

    uint32_t blockDim = totalTiles;
    blockDim = Clamp(blockDim, 1u, 24u); // conservative

    context->SetBlockDim(blockDim);

    tiling.set_totalTiles(totalTiles);
    tiling.set_tileWidth(outW);
    tiling.set_blockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_inH(H);
    tiling.set_inW(W);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_kH(Kh);
    tiling.set_kW(Kw);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;

class KernelConvDepthwise2dAsymmetricInputSquare {
public:
    __aicore__ inline KernelConvDepthwise2dAsymmetricInputSquare() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalTiles, uint32_t tileWidth,
                                uint32_t N, uint32_t C,
                                uint32_t inH, uint32_t inW,
                                uint32_t outH, uint32_t outW,
                                uint32_t kH, uint32_t kW)
    {
        totalTiles_ = totalTiles;
        tileW_ = tileWidth;

        n_ = N; c_ = C; inH_ = inH; inW_ = inW; outH_ = outH; outW_ = outW; kH_ = kH; kW_ = kW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N) * C * inH * inW);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(C) * kH * kW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N) * C * outH * outW);

        // Buffers:
        //   3 input rows of length (tileW + kW - 1) elements
        //   1 output row of length tileW
        //   1 tmp vector of length tileW
        //   1 weight vector of length kH*kW (<= 9)
        uint32_t rowCopyLen = tileW_ + kW_ - 1; // contiguous per row
        rowLen_ = rowCopyLen;

        pipe_.InitBuffer(inRowQ0_, BUFFER_NUM, rowCopyLen * sizeof(float));
        pipe_.InitBuffer(inRowQ1_, BUFFER_NUM, rowCopyLen * sizeof(float));
        pipe_.InitBuffer(inRowQ2_, BUFFER_NUM, rowCopyLen * sizeof(float));

        pipe_.InitBuffer(outRowQ_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, BUFFER_NUM, tileW_ * sizeof(float));

        pipe_.InitBuffer(wQ_, BUFFER_NUM, (kH_ * kW_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalTiles_ == 0 || tileW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles_; tileIdx += blockNum) {
            ComputeTile(tileIdx);
        }
    }

private:
    __aicore__ inline void LoadRow(TQue<TPosition::VECIN, BUFFER_NUM> &q, uint64_t gmBase, uint32_t len)
    {
        LocalTensor<float> rowLocal = q.AllocTensor<float>();
        DataCopy(rowLocal, xGm_[gmBase], len);
        q.EnQue(rowLocal);
    }

    __aicore__ inline void ComputeTile(uint32_t tileIdx)
    {
        // Decode tile -> (n, c, h), each tile computes full width row y[n,c,h,0:outW)
        uint32_t h = tileIdx % outH_;
        uint32_t tmp = tileIdx / outH_;
        uint32_t c = tmp % c_;
        uint32_t n = tmp / c_;

        // Base offsets
        // x offset for start of row h
        uint64_t xBase = (((static_cast<uint64_t>(n) * c_ + c) * inH_ + h) * inW_);
        // y offset for row h
        uint64_t yBase = (((static_cast<uint64_t>(n) * c_ + c) * outH_ + h) * outW_);

        // Copy 3 input rows
        const uint32_t rowLen = rowLen_; // tileW + kW - 1
        LoadRow(inRowQ0_, xBase + 0ULL * inW_, rowLen);
        LoadRow(inRowQ1_, xBase + 1ULL * inW_, rowLen);
        LoadRow(inRowQ2_, xBase + 2ULL * inW_, rowLen);

        // Copy weights for this channel c (kH*kW elements)
        LocalTensor<float> wLocal = wQ_.AllocTensor<float>();
        const uint64_t wBase = static_cast<uint64_t>(c) * (kH_ * kW_);
        DataCopy(wLocal, wGm_[wBase], kH_ * kW_);
        wQ_.EnQue(wLocal);

        // Prepare locals
        LocalTensor<float> row0 = inRowQ0_.DeQue<float>();
        LocalTensor<float> row1 = inRowQ1_.DeQue<float>();
        LocalTensor<float> row2 = inRowQ2_.DeQue<float>();
        LocalTensor<float> wVec = wQ_.DeQue<float>();

        LocalTensor<float> yLocal = outRowQ_.AllocTensor<float>();
        Dup(yLocal, 0.0f, tileW_);

        // Load weight scalars
        // Assuming LocalTensor supports GetValue to extract scalar
        float w00 = wVec.GetValue(0);
        float w01 = wVec.GetValue(1);
        float w02 = wVec.GetValue(2);
        float w10 = wVec.GetValue(3);
        float w11 = wVec.GetValue(4);
        float w12 = wVec.GetValue(5);
        float w20 = wVec.GetValue(6);
        float w21 = wVec.GetValue(7);
        float w22 = wVec.GetValue(8);

        // Accumulate contributions
        AccumConv(yLocal, row0, w00, 0);
        AccumConv(yLocal, row0, w01, 1);
        AccumConv(yLocal, row0, w02, 2);
        AccumConv(yLocal, row1, w10, 0);
        AccumConv(yLocal, row1, w11, 1);
        AccumConv(yLocal, row1, w12, 2);
        AccumConv(yLocal, row2, w20, 0);
        AccumConv(yLocal, row2, w21, 1);
        AccumConv(yLocal, row2, w22, 2);

        // Store output row
        outRowQ_.EnQue(yLocal);
        LocalTensor<float> yOutLocal = outRowQ_.DeQue<float>();
        DataCopy(yGm_[yBase], yOutLocal, tileW_);
        outRowQ_.FreeTensor(yOutLocal);

        // Free inputs and weights
        inRowQ0_.FreeTensor(row0);
        inRowQ1_.FreeTensor(row1);
        inRowQ2_.FreeTensor(row2);
        wQ_.FreeTensor(wVec);
    }

    __aicore__ inline void AccumConv(LocalTensor<float> &yLocal, LocalTensor<float> &row, float w, uint32_t shift)
    {
        // tmp = row[shift : shift + tileW] * w
        LocalTensor<float> tmpVec = tmpQ_.AllocTensor<float>();
        DataCopy(tmpVec, row[shift], tileW_);
        Muls(tmpVec, tmpVec, w, tileW_);
        Add(yLocal, yLocal, tmpVec, tileW_);
        tmpQ_.FreeTensor(tmpVec);
    }

private:
    TPipe pipe_;

    TQue<TPosition::VECIN, BUFFER_NUM> inRowQ0_;
    TQue<TPosition::VECIN, BUFFER_NUM> inRowQ1_;
    TQue<TPosition::VECIN, BUFFER_NUM> inRowQ2_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outRowQ_;
    TQue<TPosition::VECCALC, BUFFER_NUM> tmpQ_;
    TQue<TPosition::VECCALC, BUFFER_NUM> wQ_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalTiles_{0};
    uint32_t tileW_{0};
    uint32_t rowLen_{0};

    uint32_t n_{0}, c_{0}, inH_{0}, inW_{0}, outH_{0}, outW_{0}, kH_{0}, kW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymmetricInputSquare op;
    op.Init(x, weight, y,
            tiling_data.totalTiles, tiling_data.tileWidth,
            tiling_data.n, tiling_data.c,
            tiling_data.inH, tiling_data.inW,
            tiling_data.outH, tiling_data.outW,
            tiling_data.kH, tiling_data.kW);
    op.Process();
}
