static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard1dDilatedStridedCustomTilingData tiling;

    // Inputs
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const uint32_t N  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Ci = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t Lin = static_cast<uint32_t>(x_shape->GetDim(2));

    const uint32_t Co = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t K  = static_cast<uint32_t>(w_shape->GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stride   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *padding  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilation = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *groups   = attrs->GetAttrPointer<int32_t>(3);

    // Groups support is constrained to 1 for this kernel
    const int32_t G = (groups != nullptr) ? *groups : 1;

    // Output length
    const int64_t Lin_pad = static_cast<int64_t>(Lin) + 2LL * (padding ? *padding : 0);
    const int64_t dil_term = static_cast<int64_t>((dilation ? *dilation : 1) * (K - 1));
    const int64_t numer = Lin_pad - dil_term - 1;
    const uint32_t Lout = static_cast<uint32_t>(numer / (stride ? *stride : 1) + 1);

    // Very conservative tiling to avoid complex strided gather in UB: tileT=1
    const uint32_t tileT = 1;

    // Choose ciTile modestly to keep UB small and predictable
    uint32_t ciTile = Ci;
    if (ciTile > 64u) ciTile = 64u;

    // BlockDim: map blocks to (N * Co) pairs
    uint32_t totalPairs = N * Co;
    uint32_t blockDim = totalPairs;
    if (blockDim < 8u)  blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Ci(Ci);
    tiling.set_Co(Co);
    tiling.set_Lin(Lin);
    tiling.set_Lout(Lout);
    tiling.set_K(K);
    tiling.set_stride(stride ? *stride : 1);
    tiling.set_dilation(dilation ? *dilation : 1);
    tiling.set_padding(padding ? *padding : 0);
    tiling.set_groups(G);
    tiling.set_tileT(tileT);
    tiling.set_ciTile(ciTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    workspace[0] = 0;

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // [N, Ci, Lin]
    const gert::Shape *w_shape = context->GetInputShape(1); // [Co, Ci, K] for groups=1

    const int64_t N = x_shape->GetDim(0);
    const int64_t Ci = x_shape->GetDim(1);
    const int64_t Lin = x_shape->GetDim(2);

    const int64_t Co = w_shape->GetDim(0);
    const int64_t K  = w_shape->GetDim(2);

    // Default attrs
    int64_t stride = 3, padding = 0, dilation = 4, groups = 1;
    auto attrs = context->GetAttrs();
    if (attrs) {
        if (auto p = attrs->GetAttrPointer<int32_t>(0)) stride = *p;
        if (auto p = attrs->GetAttrPointer<int32_t>(1)) padding = *p;
        if (auto p = attrs->GetAttrPointer<int32_t>(2)) dilation = *p;
        if (auto p = attrs->GetAttrPointer<int32_t>(3)) groups = *p;
    }

    // groups == 1 assumed
    (void)groups;

    const int64_t Lin_pad = Lin + 2 * padding;
    const int64_t dil_term = dilation * (K - 1);
    const int64_t Lout = (Lin_pad - dil_term - 1) / stride + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Co);
    y_shape->SetDim(2, Lout);

    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard1dDilatedStridedCustom {
public:
    __aicore__ inline KernelConvStandard1dDilatedStridedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t Co,
                                uint32_t Lin, uint32_t Lout, uint32_t K,
                                int32_t stride, int32_t dilation,
                                uint32_t tileT, uint32_t ciTile)
    {
        N_ = N; Ci_ = Ci; Co_ = Co;
        Lin_ = Lin; Lout_ = Lout; K_ = K;
        stride_ = stride; dilation_ = dilation;
        tileT_ = tileT; ciTile_ = ciTile;

        totalX_ = static_cast<uint64_t>(N_) * Ci_ * Lin_;
        totalW_ = static_cast<uint64_t>(Co_) * Ci_ * K_;
        totalY_ = static_cast<uint64_t>(N_) * Co_ * Lout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalX_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, totalW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalY_);

        // lwin for given tileT
        lwin_ = static_cast<uint32_t>((tileT_ - 1u) * static_cast<uint32_t>(stride_) +
                                      static_cast<uint32_t>(dilation_) * (K_ - 1u) + 1u);

        // UB buffers
        // X-chunk buffer: ciTile * lwin floats
        pipe_.InitBuffer(xChunkQ_, 1, ciTile_ * lwin_ * sizeof(float));
        // W-chunk buffer: ciTile * K floats
        pipe_.InitBuffer(wChunkQ_, 1, ciTile_ * K_ * sizeof(float));
        // Y buffer: tileT floats (we set tileT=1 in tiling, but keep generic here)
        pipe_.InitBuffer(yQ_, 1, tileT_ * sizeof(float));
        // Tmp small buffers for 1-element intermediates
        pipe_.InitBuffer(tmpQ_, 4, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Ci_ == 0 || Co_ == 0 || Lout_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalPairs = static_cast<uint64_t>(N_) * Co_;
        for (uint64_t pair = blockIdx; pair < totalPairs; pair += blockNum) {
            const uint32_t nIdx  = static_cast<uint32_t>(pair / Co_);
            const uint32_t ocIdx = static_cast<uint32_t>(pair % Co_);

            for (uint32_t tBase = 0; tBase < Lout_; tBase += tileT_) {
                const uint32_t curT = AscendC::min(tileT_, Lout_ - tBase);

                // Prepare y tile
                AscendC::LocalTensor<float> yLocal = yQ_.AllocTensor<float>();
                AscendC::Muls(yLocal, yLocal, 0.0f, curT);

                for (uint32_t icBase = 0; icBase < Ci_; icBase += ciTile_) {
                    const uint32_t curCi = AscendC::min(ciTile_, Ci_ - icBase);

                    // Load X chunk for [nIdx, icBase:icBase+curCi) with window starting at tBase*stride
                    AscendC::LocalTensor<float> xLocal = xChunkQ_.AllocTensor<float>();
                    for (uint32_t i = 0; i < curCi; ++i) {
                        const uint32_t ic = icBase + i;
                        const uint64_t xOff = (static_cast<uint64_t>(nIdx) * Ci_ + ic) * Lin_
                                            + static_cast<uint64_t>(tBase) * static_cast<uint32_t>(stride_);
                        AscendC::DataCopy(xLocal[i * lwin_], xGm_[xOff], lwin_);
                    }
                    xChunkQ_.EnQue(xLocal);

                    // Load W chunk for [ocIdx, icBase:icBase+curCi), K taps
                    AscendC::LocalTensor<float> wLocal = wChunkQ_.AllocTensor<float>();
                    for (uint32_t i = 0; i < curCi; ++i) {
                        const uint32_t ic = icBase + i;
                        const uint64_t wOff = (static_cast<uint64_t>(ocIdx) * Ci_ + ic) * K_;
                        AscendC::DataCopy(wLocal[i * K_], wGm_[wOff], K_);
                    }
                    wChunkQ_.EnQue(wLocal);

                    // Compute accumulate for this chunk
                    AscendC::LocalTensor<float> xChunk = xChunkQ_.DeQue<float>();
                    AscendC::LocalTensor<float> wChunk = wChunkQ_.DeQue<float>();

                    for (uint32_t i = 0; i < curCi; ++i) {
                        const uint32_t xBase = i * lwin_;
                        const uint32_t wBase = i * K_;
                        // For each kernel tap
                        for (uint32_t k = 0; k < K_; ++k) {
                            // Gather one element from X (tileT is 1, so curT is 1)
                            AscendC::LocalTensor<float> x1 = tmpQ_.AllocTensor<float>();
                            AscendC::DataCopy(x1, xChunk[xBase + k * static_cast<uint32_t>(dilation_)], curT);

                            // Gather one weight scalar
                            AscendC::LocalTensor<float> w1 = tmpQ_.AllocTensor<float>();
                            AscendC::DataCopy(w1, wChunk[wBase + k], 1);

                            // Multiply and accumulate: prod = x1 * w1; yLocal += prod
                            AscendC::LocalTensor<float> prod = tmpQ_.AllocTensor<float>();
                            AscendC::Mul(prod, x1, w1, curT);
                            AscendC::Add(yLocal, yLocal, prod, curT);

                            tmpQ_.FreeTensor(prod);
                            tmpQ_.FreeTensor(w1);
                            tmpQ_.FreeTensor(x1);
                        }
                    }

                    xChunkQ_.FreeTensor(xChunk);
                    wChunkQ_.FreeTensor(wChunk);
                } // icBase loop

                // Store y tile
                const uint64_t yOff = (static_cast<uint64_t>(nIdx) * Co_ + ocIdx) * Lout_ + tBase;
                AscendC::DataCopy(yGm_[yOff], yLocal, curT);
                yQ_.FreeTensor(yLocal);
            } // tBase
        } // pair
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  xChunkQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  wChunkQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> tmpQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Ci_{0}, Co_{0};
    uint32_t Lin_{0}, Lout_{0}, K_{0};
    int32_t  stride_{1}, dilation_{1};
    uint32_t tileT_{1}, ciTile_{1};
    uint32_t lwin_{1};

    uint64_t totalX_{0}, totalW_{0}, totalY_{0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dDilatedStridedCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Ci, tiling_data.Co,
            tiling_data.Lin, tiling_data.Lout, tiling_data.K,
            tiling_data.stride, tiling_data.dilation,
            tiling_data.tileT, tiling_data.ciTile);
    op.Process();
}
