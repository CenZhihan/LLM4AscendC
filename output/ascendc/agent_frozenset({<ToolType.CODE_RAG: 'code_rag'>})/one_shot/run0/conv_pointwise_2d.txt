host_operator_src="""

#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// UB model per-core memory usage (in floats):
//   xLocal: 2 * tilePos   (double buffer optional)
//   accY:   tileOc * tilePos
//   tmp:    1 * tilePos
//   wRows:  tileOc * Cin
// Total floats ~= (3 + tileOc) * tilePos + tileOc * Cin
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core budget
static constexpr uint32_t ALIGN_ELEMS = 2048;            // for efficient DMA
static constexpr uint32_t HARD_CAP_POS = 16384;          // cap to avoid long iters
static constexpr uint32_t MAX_TILE_OC = 4;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    // Parse shapes
    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 4 || w_shape->GetOriginShape().GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N    = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Cin  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H    = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t W    = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));
    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t HW   = H * W;

    // Choose tileOc adaptively
    uint32_t tileOc = (Cout < MAX_TILE_OC) ? Cout : MAX_TILE_OC;
    if (tileOc == 0u) tileOc = 1u;

    const uint32_t ubFloats = UB_BUDGET_BYTES / 4u;

    // Compute tilePos subject to UB budget; adapt tileOc down if needed
    uint32_t tilePos = 0;
    while (tileOc >= 1u) {
        const uint32_t denom = 3u + tileOc; // 2*x + acc + tmp = 3*tilePos
        if (ubFloats > tileOc * Cin) {
            uint32_t maxPos = (ubFloats - tileOc * Cin) / denom;
            if (maxPos > HARD_CAP_POS) maxPos = HARD_CAP_POS;
            if (maxPos > HW) maxPos = HW;
            tilePos = AlignDown(maxPos, ALIGN_ELEMS);
        } else {
            tilePos = 0;
        }

        if (tilePos >= ALIGN_ELEMS || HW <= ALIGN_ELEMS) break;
        if (tileOc == 1u) break;
        tileOc >>= 1; // shrink oc tile and retry
    }

    if (tilePos == 0) {
        tilePos = (HW >= ALIGN_ELEMS) ? ALIGN_ELEMS : HW;
    }
    if (tilePos > HW) tilePos = HW;

    // Blocks across oc tiles
    const uint32_t numOcTiles = (tileOc == 0) ? 1u : CeilDiv(Cout, tileOc);
    uint32_t blockDim = numOcTiles;
    if (blockDim < 8u) blockDim = (numOcTiles == 0 ? 1u : (numOcTiles < 8u ? 8u : numOcTiles));
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_HW(HW);
    tiling.set_tilePos(tilePos);
    tiling.set_tileOc(tileOc);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr int32_t MAX_TILE_OC = 4;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout, uint32_t HW,
                                uint32_t tilePos, uint32_t tileOc)
    {
        N_ = N;
        Cin_ = Cin;
        Cout_ = Cout;
        HW_ = HW;
        tilePos_ = tilePos;
        tileOc_ = tileOc;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(N_ * Cin_ * HW_));
        wGm_.SetGlobalBuffer((__gm__ float*)weight, static_cast<uint32_t>(Cout_ * Cin_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(N_ * Cout_ * HW_));

        // Initialize UB buffers
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tilePos_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, MAX_TILE_OC, tilePos_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tilePos_ * sizeof(float));
        pipe_.InitBuffer(wRowQueue_, MAX_TILE_OC, Cin_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || HW_ == 0 || tilePos_ == 0) return;

        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t ocTileStart = blockIdx * tileOc_;
        if (ocTileStart >= Cout_) return;

        const uint32_t ocCount = (ocTileStart + tileOc_ <= Cout_) ? tileOc_ : (Cout_ - ocTileStart);

        // Load weight rows for this oc-tile into UB (each row length = Cin, contiguous)
        AscendC::LocalTensor<float> wRows[MAX_TILE_OC];
        for (uint32_t k = 0; k < ocCount; ++k) {
            AscendC::LocalTensor<float> wRow = wRowQueue_.AllocTensor<float>();
            const uint32_t wBase = (ocTileStart + k) * Cin_;
            AscendC::DataCopy(wRow, wGm_[wBase], Cin_);
            wRows[k] = wRow; // keep allocated until end
        }

        for (uint32_t n = 0; n < N_; ++n) {
            const uint32_t xBatchBase = n * Cin_ * HW_;
            const uint32_t yBatchBase = n * Cout_ * HW_;

            for (uint32_t p = 0; p < HW_; p += tilePos_) {
                uint32_t curLen = HW_ - p;
                if (curLen > tilePos_) curLen = tilePos_;

                // Allocate accumulators for each oc in tile and zero-initialize
                AscendC::LocalTensor<float> accs[MAX_TILE_OC];
                for (uint32_t k = 0; k < ocCount; ++k) {
                    AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();
                    AscendC::VectorDup(acc, 0.0f, curLen);
                    accs[k] = acc;
                }

                // Temporary buffer for scaled x
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

                // Accumulate across input channels
                for (uint32_t ic = 0; ic < Cin_; ++ic) {
                    // Load x tile for (n, ic)
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    const uint32_t xOff = xBatchBase + ic * HW_ + p;
                    AscendC::DataCopy(xLocal, xGm_[xOff], curLen);
                    inQueueX_.EnQue(xLocal);

                    AscendC::LocalTensor<float> xUse = inQueueX_.DeQue<float>();

                    // For each oc in tile, y += x * w[oc, ic]
                    for (uint32_t k = 0; k < ocCount; ++k) {
                        const float wScalar = wRows[k].GetValue(ic);
                        AscendC::Muls(tLocal, xUse, wScalar, curLen);
                        AscendC::Add(accs[k], accs[k], tLocal, curLen);
                    }

                    inQueueX_.FreeTensor(xUse);
                }

                tmpQueue_.FreeTensor(tLocal);

                // Store accumulators to output
                for (uint32_t k = 0; k < ocCount; ++k) {
                    const uint32_t oc = ocTileStart + k;
                    const uint32_t yOff = yBatchBase + oc * HW_ + p;
                    AscendC::DataCopy(yGm_[yOff], accs[k], curLen);
                    accQueue_.FreeTensor(accs[k]);
                }
            }
        }

        // Free weight rows
        for (uint32_t k = 0; k < ocCount; ++k) {
            wRowQueue_.FreeTensor(wRows[k]);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, MAX_TILE_OC> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, MAX_TILE_OC> wRowQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0};
    uint32_t Cin_{0};
    uint32_t Cout_{0};
    uint32_t HW_{0};
    uint32_t tilePos_{0};
    uint32_t tileOc_{1};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout, tiling_data.HW,
            tiling_data.tilePos, tiling_data.tileOc);
    op.Process();
}
