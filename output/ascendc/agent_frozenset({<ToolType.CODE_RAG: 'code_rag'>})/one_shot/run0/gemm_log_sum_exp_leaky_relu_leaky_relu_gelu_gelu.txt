#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF_F = -3.402823466e+38f;  // -FLT_MAX
constexpr float INV_SQRT2 = 0.7071067811865476f; // 1/sqrt(2)

class KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom {
public:
    __aicore__ inline KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batch, uint32_t inFeat, uint32_t outFeat,
                                uint32_t rowsPerBlock, uint32_t tileK,
                                float neg1, float neg2)
    {
        batch_ = batch;
        inFeat_ = inFeat;
        outFeat_ = outFeat;
        rowsPerBlock_ = rowsPerBlock;
        tileK_ = (tileK == 0) ? inFeat_ : tileK;
        neg1_ = neg1;
        neg2_ = neg2;
        negCorr1_ = neg1_ - 1.0f;
        negCorr2_ = neg2_ - 1.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(batch_) * inFeat_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(outFeat_) * inFeat_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, outFeat_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(batch_) * 1u);

        pipe_.InitBuffer(xTileQ_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(wTileQ_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, 4, 1 * sizeof(float)); // for 1-element reductions and scalars
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || inFeat_ == 0 || outFeat_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes multiple rows in a grid-stride loop
        for (uint32_t row = blockIdx; row < batch_; row += blockNum) {
            ComputeRow(row);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t row)
    {
        // First pass: compute max over out dimension of z = xW^T + b
        float maxVal = NEG_INF_F;

        for (uint32_t o = 0; o < outFeat_; ++o) {
            // accumulator scalar in UB as 1-element tensor
            AscendC::LocalTensor<float> acc = scalarQ_.AllocTensor<float>();
            AscendC::Dup(acc, 0.0f, 1);

            for (uint32_t kOff = 0; kOff < inFeat_; kOff += tileK_) {
                const uint32_t curK = AscendC::Min(tileK_, inFeat_ - kOff);

                AscendC::LocalTensor<float> xLocal = xTileQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> wLocal = wTileQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpMul = tmpQ_.AllocTensor<float>();
                // load x segment
                AscendC::DataCopy(xLocal, xGm_[row * inFeat_ + kOff], curK);
                // load w segment for current out channel
                AscendC::DataCopy(wLocal, wGm_[o * inFeat_ + kOff], curK);
                // tmp = x * w
                AscendC::Mul(tmpMul, xLocal, wLocal, curK);
                // reduce sum tmp into 1 element
                AscendC::LocalTensor<float> red = scalarQ_.AllocTensor<float>();
                AscendC::ReduceSum(red, tmpMul, curK);
                // acc += red
                AscendC::Add(acc, acc, red, 1);

                scalarQ_.FreeTensor(red);
                tmpQ_.FreeTensor(tmpMul);
                wTileQ_.FreeTensor(wLocal);
                xTileQ_.FreeTensor(xLocal);
            }

            // add bias
            AscendC::LocalTensor<float> biasT = scalarQ_.AllocTensor<float>();
            AscendC::DataCopy(biasT, bGm_[o], 1);
            AscendC::Add(acc, acc, biasT, 1);
            scalarQ_.FreeTensor(biasT);

            // update max
            float z_val = 0.0f;
            // move acc[0] to a temp local to read into scalar via DataCopy to GM not needed; use Cast through Adds with 0?
            // Here we assume we can read a 1-element LocalTensor into scalar via GetValue
            z_val = acc.GetValue(0);
            if (z_val > maxVal) maxVal = z_val;

            scalarQ_.FreeTensor(acc);
        }

        // Second pass: compute sum(exp(z - maxVal))
        AscendC::LocalTensor<float> sumExp = scalarQ_.AllocTensor<float>();
        AscendC::Dup(sumExp, 0.0f, 1);

        for (uint32_t o = 0; o < outFeat_; ++o) {
            // accumulate z for channel o
            AscendC::LocalTensor<float> acc = scalarQ_.AllocTensor<float>();
            AscendC::Dup(acc, 0.0f, 1);

            for (uint32_t kOff = 0; kOff < inFeat_; kOff += tileK_) {
                const uint32_t curK = AscendC::Min(tileK_, inFeat_ - kOff);

                AscendC::LocalTensor<float> xLocal = xTileQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> wLocal = wTileQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpMul = tmpQ_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[row * inFeat_ + kOff], curK);
                AscendC::DataCopy(wLocal, wGm_[o * inFeat_ + kOff], curK);
                AscendC::Mul(tmpMul, xLocal, wLocal, curK);

                AscendC::LocalTensor<float> red = scalarQ_.AllocTensor<float>();
                AscendC::ReduceSum(red, tmpMul, curK);
                AscendC::Add(acc, acc, red, 1);

                scalarQ_.FreeTensor(red);
                tmpQ_.FreeTensor(tmpMul);
                wTileQ_.FreeTensor(wLocal);
                xTileQ_.FreeTensor(xLocal);
            }

            // add bias
            AscendC::LocalTensor<float> biasT = scalarQ_.AllocTensor<float>();
            AscendC::DataCopy(biasT, bGm_[o], 1);
            AscendC::Add(acc, acc, biasT, 1);
            scalarQ_.FreeTensor(biasT);

            // delta = z - maxVal
            AscendC::LocalTensor<float> delta = scalarQ_.AllocTensor<float>();
            AscendC::Dup(delta, maxVal, 1);
            AscendC::Sub(delta, acc, delta, 1);

            // e = exp(delta)
            AscendC::LocalTensor<float> e = scalarQ_.AllocTensor<float>();
            AscendC::Exp(e, delta, 1);

            // sumExp += e
            AscendC::Add(sumExp, sumExp, e, 1);

            scalarQ_.FreeTensor(e);
            scalarQ_.FreeTensor(delta);
            scalarQ_.FreeTensor(acc);
        }

        // lse = maxVal + log(sumExp)
        AscendC::LocalTensor<float> lse = scalarQ_.AllocTensor<float>();
        AscendC::Dup(lse, maxVal, 1);

        AscendC::LocalTensor<float> logSum = scalarQ_.AllocTensor<float>();
        AscendC::Log(logSum, sumExp, 1);

        AscendC::Add(lse, lse, logSum, 1);
        scalarQ_.FreeTensor(logSum);
        scalarQ_.FreeTensor(sumExp);

        // two LeakyReLUs with negative slopes neg1_ and neg2_
        // y = x + (neg - 1) * min(x, 0)
        AscendC::LocalTensor<float> tmp = scalarQ_.AllocTensor<float>();
        AscendC::Mins(tmp, lse, 0.0f, 1);
        AscendC::Muls(tmp, tmp, negCorr1_, 1);
        AscendC::Add(lse, lse, tmp, 1);

        AscendC::Mins(tmp, lse, 0.0f, 1);
        AscendC::Muls(tmp, tmp, negCorr2_, 1);
        AscendC::Add(lse, lse, tmp, 1);

        // two GELUs (erf-based): 0.5*x*(1 + erf(x / sqrt(2)))
        GeluInplace(lse);
        GeluInplace(lse);

        // store result to y[row, 0]
        AscendC::DataCopy(yGm_[row], lse, 1);

        scalarQ_.FreeTensor(tmp);
        scalarQ_.FreeTensor(lse);
    }

    __aicore__ inline void GeluInplace(AscendC::LocalTensor<float>& x)
    {
        AscendC::LocalTensor<float> t = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> e = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> one = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> half = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpMul = scalarQ_.AllocTensor<float>();

        AscendC::Dup(one, 1.0f, 1);
        AscendC::Dup(half, 0.5f, 1);

        // t = x / sqrt(2)
        AscendC::Muls(t, x, INV_SQRT2, 1);
        // e = erf(t)
        AscendC::Erf(e, t, 1);
        // e = 1 + e
        AscendC::Add(e, e, one, 1);
        // tmpMul = 0.5 * x
        AscendC::Mul(tmpMul, half, x, 1);
        // x = tmpMul * e
        AscendC::Mul(x, tmpMul, e, 1);

        scalarQ_.FreeTensor(tmpMul);
        scalarQ_.FreeTensor(half);
        scalarQ_.FreeTensor(one);
        scalarQ_.FreeTensor(e);
        scalarQ_.FreeTensor(t);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_{0};
    uint32_t inFeat_{0};
    uint32_t outFeat_{0};
    uint32_t rowsPerBlock_{1};
    uint32_t tileK_{0};

    float neg1_{0.01f};
    float neg2_{0.01f};
    float negCorr1_{-0.99f};
    float negCorr2_{-0.99f};
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{

            tiling_data.rowsPerBlock, tiling_data.tileK,
            tiling_data.negativeSlope1, tiling_data.negativeSlope2);
    op.Process();
}
