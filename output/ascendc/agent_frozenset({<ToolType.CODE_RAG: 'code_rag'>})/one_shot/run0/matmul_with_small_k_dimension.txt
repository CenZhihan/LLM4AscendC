namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t AlignUp(uint32_t x, uint32_t a) {
    return (a == 0) ? x : ((x + a - 1) / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithSmallKDimensionCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    const uint32_t K2 = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(0));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(1));
    if (K != K2) {
        return ge::GRAPH_FAILED;
    }

    // UB model for this kernel (per core):
    // - inQueueB: 2 * tileN
    // - accBuffer: 1 * tileN
    // - aRow: 1 * K
    // Total bytes = (2 + 1) * tileN * 4 + K * 4 = 12 * tileN + 4 * K
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative share per core
    static constexpr uint32_t ALIGN_ELEMS = 2048;              // align for better DMA
    static constexpr uint32_t MIN_TILE_N = 512;

    uint32_t maxTileN = 0;
    if (UB_BUDGET_BYTES > 4u * K) {
        maxTileN = (UB_BUDGET_BYTES - 4u * K) / 12u; // elements
    }
    if (maxTileN < MIN_TILE_N) {
        maxTileN = MIN_TILE_N;
    }
    maxTileN = AlignDown(maxTileN, ALIGN_ELEMS);
    if (maxTileN == 0) {
        maxTileN = MIN_TILE_N;
    }
    uint32_t tileN = maxTileN;
    if (N < tileN) {
        const uint32_t aligned = AlignDown(N, ALIGN_ELEMS);
        tileN = (aligned > 0) ? aligned : N;
    }

    // Choose a conservative blockDim (number of AI Cores to launch)
    uint32_t blockDim = M;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulWithSmallKDimensionCustom {
public:
    __aicore__ inline KernelMatmulWithSmallKDimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(M_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(K_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(M_) * N_);

        // Queues: B double-buffer, accumulator, and aRow buffer
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(aRowQueue_, 1, K_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0 || tileN_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block computes multiple rows of C (grid-stride over M)
        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            ComputeOneRow(m);
        }
    }

private:
    __aicore__ inline void CopyInB(uint32_t k, uint32_t n0, uint32_t len)
    {
        AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
        const uint64_t bOff = static_cast<uint64_t>(k) * N_ + n0;
        AscendC::DataCopy(bLocal, bGm_[bOff], len);
        inQueueB_.EnQue(bLocal);
    }

    __aicore__ inline void ComputeOneRow(uint32_t m)
    {
        // Load aRow[m, :]
        AscendC::LocalTensor<float> aLocal = aRowQueue_.AllocTensor<float>();
        const uint64_t aOff = static_cast<uint64_t>(m) * K_;
        AscendC::DataCopy(aLocal, aGm_[aOff], K_);
        aRowQueue_.EnQue(aLocal);
        aLocal = aRowQueue_.DeQue<float>();

        for (uint32_t n0 = 0; n0 < N_; n0 += tileN_) {
            uint32_t curN = N_ - n0;
            if (curN > tileN_) curN = tileN_;

            // Prefetch and compute for k = 0
            CopyInB(0, n0, curN);
            AscendC::LocalTensor<float> bLocal0 = inQueueB_.DeQue<float>();
            AscendC::LocalTensor<float> yLocal = accQueue_.AllocTensor<float>();

            // get scalar a[m,0]
            float a0 = aLocal.GetValue(0);
            AscendC::Muls(yLocal, bLocal0, a0, curN);
            inQueueB_.FreeTensor(bLocal0);

            // Accumulate for remaining k
            for (uint32_t k = 1; k < K_; ++k) {
                CopyInB(k, n0, curN);
                AscendC::LocalTensor<float> bLocal = inQueueB_.DeQue<float>();
                float ak = aLocal.GetValue(k);
                AscendC::Muls(bLocal, bLocal, ak, curN);
                AscendC::Add(yLocal, yLocal, bLocal, curN);
                inQueueB_.FreeTensor(bLocal);
            }

            // Write back C[m, n0:n0+curN]
            const uint64_t cOff = static_cast<uint64_t>(m) * N_ + n0;
            AscendC::DataCopy(cGm_[cOff], yLocal, curN);
            accQueue_.FreeTensor(yLocal);
        }

        aRowQueue_.FreeTensor(aLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> aRowQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileN_{0};
};

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithSmallKDimensionCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileN);
    op.Process();
}
