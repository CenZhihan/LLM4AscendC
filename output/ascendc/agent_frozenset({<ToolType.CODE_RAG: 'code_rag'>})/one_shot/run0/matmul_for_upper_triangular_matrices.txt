#include "matmul_for_upper_triangular_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

// UB usage model per core (worst row i=0):
//   - aRow: up to N floats
//   - bSeg: up to N floats
//   - ySeg: up to N floats
// Total ~ 3 * N * 4 bytes. Keep below UB budget.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForUpperTriangularMatricesCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const gert::Shape origin = in_shape->GetOriginShape();
    const int64_t dim0 = origin.GetDim(0);
    const int64_t dim1 = origin.GetDim(1);
    if (dim0 <= 0 || dim1 <= 0 || dim0 != dim1) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(dim0);

    // tileCols determined by UB budget for 3*N floats
    uint32_t maxNFromUb = (UB_BUDGET_BYTES / 4u) / 3u;
    if (maxNFromUb < ALIGN_ELEMS) maxNFromUb = ALIGN_ELEMS;
    uint32_t tileCols = N;
    if (tileCols > maxNFromUb) {
        tileCols = AlignDown(maxNFromUb, ALIGN_ELEMS);
        if (tileCols == 0) tileCols = ALIGN_ELEMS;
    }
    tileCols = AlignDown(tileCols, ALIGN_ELEMS);
    if (tileCols == 0) tileCols = (N < ALIGN_ELEMS) ? N : ALIGN_ELEMS;

    // blockDim: distribute rows across cores
    uint32_t blockDim = N;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{

    const gert::Shape *B_shape = context->GetInputShape(1);
    if ((A_shape->GetOriginShape().GetDim(0) != B_shape->GetOriginShape().GetDim(0)) ||
        (A_shape->GetOriginShape().GetDim(1) != B_shape->GetOriginShape().GetDim(1))) {
        return GRAPH_FAILED;
    }
    gert::Shape *C_shape = context->GetOutputShape(0);
    *C_shape = *A_shape;
    return GRAPH_SUCCESS;

    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;

    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUF_DEPTH = 2;

class KernelMatmulForUpperTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForUpperTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileCols)
    {
        N_ = N;
        tileCols_ = tileCols;

        const uint32_t totalElems = N_ * N_;
        aGm_.SetGlobalBuffer((__gm__ float*)A, totalElems);
        bGm_.SetGlobalBuffer((__gm__ float*)B, totalElems);
        cGm_.SetGlobalBuffer((__gm__ float*)C, totalElems);

        pipe_.InitBuffer(aBufQ_, 1, tileCols_ * sizeof(float));         // aRow (max N elems)
        pipe_.InitBuffer(bBufQ_, BUF_DEPTH, tileCols_ * sizeof(float)); // bSeg (double-buffered)
        pipe_.InitBuffer(yBufQ_, 1, tileCols_ * sizeof(float));         // ySeg accumulate
        pipe_.InitBuffer(scalarQ_, 1, sizeof(float));                   // alpha scalar
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each core computes a subset of rows i using grid-stride across rows.
        for (uint32_t i = blockIdx; i < N_; i += blockNum) {
            ComputeRow(i);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t i)
    {
        const uint32_t rowBase = i * N_;
        const uint32_t rowLen = N_ - i;
        if (rowLen == 0) return;

        // Load A[i, i..N-1] into local buffer (up to tileCols_)
        // We may need several chunks if N_ > tileCols_, but tiling guarantees tileCols_ >= rowLen in typical cases.
        uint32_t remaining = rowLen;
        uint32_t aOffset = 0;

        // For robustness, handle possibly multiple chunks of the aRow (rare if tileCols_ >= N)
        while (remaining > 0) {
            const uint32_t chunk = remaining > tileCols_ ? tileCols_ : remaining;
            // Copy aRow chunk
            AscendC::LocalTensor<float> aRowChunk = aBufQ_.AllocTensor<float>();
            AscendC::DataCopy(aRowChunk, aGm_[rowBase + i + aOffset], chunk);
            aBufQ_.EnQue(aRowChunk);

            // For each k in this chunk, process contributions to C[i, k..N-1]
            for (uint32_t kOff = 0; kOff < chunk; ++kOff) {
                const uint32_t k = i + aOffset + kOff;
                const uint32_t segLen = N_ - k;
                if (segLen == 0) continue;

                // Load alpha = A[i, k]
                AscendC::LocalTensor<float> aRowChunkLocal = aBufQ_.DeQue<float>();
                float alpha = aRowChunkLocal.GetValue(kOff);
                aBufQ_.EnQue(aRowChunkLocal);

                // Process B[k, k..N-1] potentially in pieces if segLen > tileCols_
                uint32_t remSeg = segLen;
                uint32_t segColOff = 0;

                while (remSeg > 0) {
                    const uint32_t step = remSeg > tileCols_ ? tileCols_ : remSeg;

                    // Load B row segment
                    AscendC::LocalTensor<float> bSeg = bBufQ_.AllocTensor<float>();
                    AscendC::DataCopy(bSeg, bGm_[k * N_ + k + segColOff], step);

                    // Scale B segment by alpha in-place
                    AscendC::Muls(bSeg, bSeg, alpha, step);

                    // C segment start address: C[i, k + segColOff .. k + segColOff + step - 1]
                    const uint32_t cOff = rowBase + k + segColOff;

                    if (k == i && segColOff == 0) {
                        // First write for this row, write full initial segment (no need to read C)
                        AscendC::DataCopy(cGm_[cOff], bSeg, step);
                    } else {
                        // Accumulate into existing C segment: y = c + bSeg
                        AscendC::LocalTensor<float> ySeg = yBufQ_.AllocTensor<float>();
                        AscendC::DataCopy(ySeg, cGm_[cOff], step);
                        AscendC::Add(ySeg, ySeg, bSeg, step);
                        AscendC::DataCopy(cGm_[cOff], ySeg, step);
                        yBufQ_.FreeTensor(ySeg);
                    }

                    bBufQ_.FreeTensor(bSeg);
                    remSeg -= step;
                    segColOff += step;
                }
            }

            // Pop the aRowChunk after all kOff in this chunk done
            AscendC::LocalTensor<float> toFree = aBufQ_.DeQue<float>();
            aBufQ_.FreeTensor(toFree);

            remaining -= chunk;
            aOffset += chunk;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aBufQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_DEPTH> bBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t tileCols_{0};
};

extern "C" __global__ __aicore__ void matmul_for_upper_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForUpperTriangularMatricesCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.tileCols);
    op.Process();
}
