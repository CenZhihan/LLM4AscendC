{
    return (a + b - 1u) / b;

    // yAcc: tileD
    // xTile: tileD + Kd - 1
    // xSeg: tileD
    // Total floats ~ (3 * tileD + (Kd - 1)), bytes = 4 * (3 * tileD + Kd - 1)
    static constexpr uint32_t UB_BUDGET_BYTES = 160 * 1024; // per-core budget
    uint32_t maxTileD = 0;
    if (Kd > 0) {
        const uint32_t usableFloats = UB_BUDGET_BYTES / 4;
        if (usableFloats > (Kd - 1)) {
            maxTileD = (usableFloats - (Kd - 1)) / 3;
        }
    }
    if (maxTileD == 0) {
        maxTileD = Dout > 0 ? (Dout < 16 ? Dout : 16u) : 1u;
    }
    // Align tileD down to 64 for better DMA behavior, but keep >= 8
    maxTileD = AlignDown(maxTileD, 64u);
    if (maxTileD < 8u) maxTileD = (Dout < 8 ? Dout : 8u);
    if (maxTileD > Dout) maxTileD = Dout;

    uint32_t blockDim = 1u;

        if (blockDim < 8u) blockDim = 8u;
        if (blockDim > 24u) blockDim = 24u;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;
constexpr int32_t TMP_BUF_NUM = 2;

// Helper macro to decode linear task index into (n, co, wout, hout)
__aicore__ inline void DecodeTask(uint64_t t, uint32_t Wout, uint32_t Hout, uint32_t Co,
                                  uint32_t &n, uint32_t &co, uint32_t &wout, uint32_t &hout)
{
    uint64_t tmp = t;
    hout = static_cast<uint32_t>(tmp % Hout);
    tmp /= Hout;
    wout = static_cast<uint32_t>(tmp % Wout);
    tmp /= Wout;
    co = static_cast<uint32_t>(tmp % Co);
    tmp /= Co;
    n = static_cast<uint32_t>(tmp);
}

class KernelConvStandard3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t W, uint32_t H, uint32_t D,
                                uint32_t Kw, uint32_t Kh, uint32_t Kd,
                                uint32_t Wout, uint32_t Hout, uint32_t Dout,
                                uint32_t tileD)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        W_ = W; H_ = H; D_ = D;
        Kw_ = Kw; Kh_ = Kh; Kd_ = Kd;
        Wout_ = Wout; Hout_ = Hout; Dout_ = Dout;
        tileD_ = tileD;

        // Strides for x: [N, C, W, H, D]
        sx_d_ = 1;
        sx_h_ = D_;
        sx_w_ = H_ * D_;
        sx_c_ = W_ * H_ * D_;
        sx_n_ = cin_ * W_ * H_ * D_;

        // Strides for y: [N, C_out, Wout, Hout, Dout]
        sy_d_ = 1;
        sy_h_ = Dout_;
        sy_w_ = Hout_ * Dout_;
        sy_c_ = Wout_ * Hout_ * Dout_;
        sy_n_ = cout_ * Wout_ * Hout_ * Dout_;

        // Strides for weight: [Cout, Cin, Kw, Kh, Kd]
        sw_kd_ = 1;
        sw_kh_ = Kd_;
        sw_kw_ = Kh_ * Kd_;
        sw_ci_ = Kw_ * Kh_ * Kd_;
        sw_co_ = cin_ * Kw_ * Kh_ * Kd_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * cin_ * W_ * H_ * D_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cout_) * cin_ * Kw_ * Kh_ * Kd_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * cout_ * Wout_ * Hout_ * Dout_);

        // UB buffers
        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, (tileD_ + Kd_) * sizeof(float));   // xTile: tileD + Kd - 1
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileD_ * sizeof(float));         // yAcc / output
        pipe_.InitBuffer(tmpQueue_, TMP_BUF_NUM, tileD_ * sizeof(float));          // xSeg or tmp accum
        pipe_.InitBuffer(scalarBuf_, 1, sizeof(float));                            // scalar buffer (1 element)
    }

    __aicore__ inline void Process()
    {
        if (Dout_ == 0 || tileD_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalTasks = static_cast<uint64_t>(n_) * cout_ * Wout_ * Hout_;
        for (uint64_t task = blockIdx; task < totalTasks; task += blockNum) {
            uint32_t n, co, wout, hout;
            DecodeTask(task, Wout_, Hout_, cout_, n, co, wout, hout);

            // loop over d_out in tiles
            for (uint32_t dBase = 0; dBase < Dout_; dBase += tileD_) {
                const uint32_t curTile = AscendC::min(tileD_, Dout_ - dBase);

                // Enqueue an empty yAcc buffer (will be written)
                AscendC::LocalTensor<float> yAcc = outQueueY_.AllocTensor<float>();

                // We will lazily initialize yAcc on first accumulation
                bool firstAcc = true;

                for (uint32_t ci = 0; ci < cin_; ++ci) {
                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                            // Load xTile once for this (ci, kh, kw): contiguous along D
                            const uint64_t xBaseOff =
                                static_cast<uint64_t>(n) * sx_n_ +
                                static_cast<uint64_t>(ci) * sx_c_ +
                                static_cast<uint64_t>(wout + kw) * sx_w_ +
                                static_cast<uint64_t>(hout + kh) * sx_h_ +
                                static_cast<uint64_t>(dBase) * sx_d_;
                            const uint32_t xTileLen = curTile + (Kd_ - 1);

                            AscendC::LocalTensor<float> xTile = inQueueX_.AllocTensor<float>();
                            AscendC::DataCopy(xTile, xGm_[xBaseOff], xTileLen);
                            inQueueX_.EnQue(xTile);

                            for (uint32_t kd = 0; kd < Kd_; ++kd) {
                                // Load weight scalar w_val = weight[co, ci, kw, kh, kd]
                                const uint64_t wOff =
                                    static_cast<uint64_t>(co) * sw_co_ +
                                    static_cast<uint64_t>(ci) * sw_ci_ +
                                    static_cast<uint64_t>(kw) * sw_kw_ +
                                    static_cast<uint64_t>(kh) * sw_kh_ +
                                    static_cast<uint64_t>(kd) * sw_kd_;

                                AscendC::LocalTensor<float> wScalar = scalarBuf_.AllocTensor<float>();
                                AscendC::DataCopy(wScalar, wGm_[wOff], 1);
                                // Read scalar value into a C++ variable (device register)
                                float w_val = wScalar.GetValue(0);
                                scalarBuf_.FreeTensor(wScalar);

                                // Extract xSeg = xTile[kd : kd + curTile]
                                AscendC::LocalTensor<float> xTileLocal = inQueueX_.DeQue<float>();
                                AscendC::LocalTensor<float> xSeg = tmpQueue_.AllocTensor<float>();
                                AscendC::DataCopy(xSeg, xTileLocal[kd], curTile);

                                if (firstAcc) {
                                    // yAcc = xSeg * w_val
                                    AscendC::Muls(yAcc, xSeg, w_val, curTile);
                                    firstAcc = false;
                                } else {
                                    AscendC::LocalTensor<float> tmpMul = tmpQueue_.AllocTensor<float>();
                                    AscendC::Muls(tmpMul, xSeg, w_val, curTile);
                                    AscendC::Add(yAcc, yAcc, tmpMul, curTile);
                                    tmpQueue_.FreeTensor(tmpMul);
                                }
                                tmpQueue_.FreeTensor(xSeg);
                                inQueueX_.EnQue(xTileLocal);
                            }

                            // release xTile
                            AscendC::LocalTensor<float> xTileLocalFinal = inQueueX_.DeQue<float>();
                            inQueueX_.FreeTensor(xTileLocalFinal);
                        }
                    }
                }

                // Write yAcc to GM at [n, co, wout, hout, dBase : dBase + curTile]
                const uint64_t yBaseOff =
                    static_cast<uint64_t>(n) * sy_n_ +
                    static_cast<uint64_t>(co) * sy_c_ +
                    static_cast<uint64_t>(wout) * sy_w_ +
                    static_cast<uint64_t>(hout) * sy_h_ +
                    static_cast<uint64_t>(dBase) * sy_d_;

                outQueueY_.EnQue(yAcc);
                AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[yBaseOff], yOut, curTile);
                outQueueY_.FreeTensor(yOut);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t W_{0}, H_{0}, D_{0};
    uint32_t Kw_{0}, Kh_{0}, Kd_{0};
    uint32_t Wout_{0}, Hout_{0}, Dout_{0};
    uint32_t tileD_{0};

    // strides for x
    uint32_t sx_d_{0}, sx_h_{0}, sx_w_{0}, sx_c_{0}, sx_n_{0};
    // strides for y
    uint32_t sy_d_{0}, sy_h_{0}, sy_w_{0}, sy_c_{0}, sy_n_{0};
    // strides for w
    uint32_t sw_kd_{0}, sw_kh_{0}, sw_kw_{0}, sw_ci_{0}, sw_co_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tilingData, tilingGm);
    KernelConvStandard3dSquareInputAsymmetricKernelCustom op;
    op.Init(
        x, weight, y,
        tilingData.n, tilingData.cin, tilingData.cout,
        tilingData.w, tilingData.h, tilingData.d,
        tilingData.kw, tilingData.kh, tilingData.kd,
        tilingData.wout, tilingData.hout, tilingData.dout,
        tilingData.tileD
    );
    op.Process();
}
