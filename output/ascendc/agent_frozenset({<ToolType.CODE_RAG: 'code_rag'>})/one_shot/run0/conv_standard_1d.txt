host_operator_src="""
#include "conv_standard_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}

} // anonymous namespace

namespace optiling {

// UB model (floats):
//   yTile:         1 * tileT
//   xChannelWin:   1 * (tileT + K - 1)
//   tmpVec:        1 * tileT
//   wChannel:      1 * K
// Total floats ~= 3 * tileT + (tileT + K - 1) + K = 4 * tileT + (2K - 1)
// Bytes = 4 * (4 * tileT + 2K - 1)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative share per core
static constexpr uint32_t ALIGN_ELEMS = 64;              // favor vector alignment
static constexpr uint32_t HARD_CAP_TILE = 16384;         // avoid too-long inner loops

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard1dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 3 || w_shape->GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t Lin = static_cast<uint32_t>(x_shape->GetDim(2));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t CinW = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t K    = static_cast<uint32_t>(w_shape->GetDim(2));

    if (CinW != Cin) {
        return ge::GRAPH_FAILED;
    }
    if (Lin < K || K == 0) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Lout = Lin - K + 1;

    // Compute tileT based on UB budget: 4*(4*tileT + 2K-1) <= UB_BUDGET_BYTES
    const uint32_t Bfloats = UB_BUDGET_BYTES / 4;
    uint32_t tileT = 1;
    if (Bfloats > (2u * K - 1u)) {
        const uint32_t maxTileByUb = (Bfloats - (2u * K - 1u)) / 4u;
        tileT = maxTileByUb;
    }
    if (tileT > HARD_CAP_TILE) tileT = HARD_CAP_TILE;
    if (tileT > Lout) tileT = Lout;
    tileT = AlignDown(tileT, ALIGN_ELEMS);
    if (tileT == 0u) tileT = (Lout >= 1u ? 1u : 0u);

    uint32_t blockDim = Cout;
    if (blockDim == 0u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_Lin(Lin);
    tiling.set_K(K);
    tiling.set_Lout(Lout);
    tiling.set_tileT(tileT);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetDimNum() != 3 || w_shape->GetDimNum() != 3) {
        return GRAPH_FAILED;
    }
    const int64_t N   = x_shape->GetDim(0);
    const int64_t Cin = x_shape->GetDim(1);
    const int64_t Lin = x_shape->GetDim(2);

    const int64_t Cout = w_shape->GetDim(0);
    const int64_t CinW = w_shape->GetDim(1);
    const int64_t K    = w_shape->GetDim(2);

    if (Cin != CinW || Lin < K || K <= 0) {
        return GRAPH_FAILED;
    }
    const int64_t Lout = Lin - K + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Lout);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using AscendC::GetBlockIdx;
using AscendC::GetBlockNum;

class KernelConvStandard1dCustom {
public:
    __aicore__ inline KernelConvStandard1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Lin, uint32_t K, uint32_t Lout,
                                uint32_t tileT)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout; Lin_ = Lin; K_ = K; Lout_ = Lout; tileT_ = tileT;
        if (K_ == 0) return;
        tileWin_ = tileT_ + K_ - 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Cin_ * Lin_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cout_ * Cin_ * K_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Cout_ * Lout_);

        // Buffers:
        // yTile: tileT
        // xWin (per channel): tileT + K - 1
        // tmpVec: tileT
        // wChannel: K
        const uint32_t yBytes   = tileT_ * sizeof(float);
        const uint32_t xBytes   = tileWin_ * sizeof(float);
        const uint32_t tmpBytes = tileT_ * sizeof(float);
        const uint32_t wBytes   = K_ * sizeof(float);

        pipe_.InitBuffer(yQueue_,   1, yBytes);
        pipe_.InitBuffer(xWinQueue_,1, xBytes);
        pipe_.InitBuffer(tmpQueue_, 1, tmpBytes);
        pipe_.InitBuffer(wQueue_,   1, wBytes);
    }

    __aicore__ inline void Process()
    {
        if (Lout_ == 0 || tileT_ == 0 || K_ == 0) return;

        const uint32_t blockNum = (uint32_t)GetBlockNum();
        const uint32_t blockIdx = (uint32_t)GetBlockIdx();

        for (uint32_t oc = blockIdx; oc < Cout_; oc += blockNum) {
            for (uint32_t n = 0; n < N_; ++n) {
                for (uint32_t tBase = 0; tBase < Lout_; tBase += tileT_) {
                    uint32_t curT = Lout_ - tBase;
                    if (curT > tileT_) curT = tileT_;
                    ComputeTile(n, oc, tBase, curT);
                }
            }
        }
    }

private:
    __aicore__ inline void LoadXChannelWin(uint32_t n, uint32_t ci, uint32_t tBase, uint32_t curT,
                                           AscendC::LocalTensor<float>& xWin)
    {
        const uint64_t gmOffset = ((uint64_t)n * Cin_ + ci) * Lin_ + tBase;
        const uint32_t copyLen = curT + K_ - 1;
        AscendC::DataCopy(xWin, xGm_[gmOffset], copyLen);
    }

    __aicore__ inline void LoadWChannel(uint32_t oc, uint32_t ci, AscendC::LocalTensor<float>& wLocal)
    {
        const uint64_t gmOffset = ((uint64_t)oc * Cin_ + ci) * K_;
        AscendC::DataCopy(wLocal, wGm_[gmOffset], K_);
    }

    __aicore__ inline void StoreY(uint32_t n, uint32_t oc, uint32_t tBase, uint32_t curT,
                                  AscendC::LocalTensor<float>& yLocal)
    {
        const uint64_t gmOffset = ((uint64_t)n * Cout_ + oc) * Lout_ + tBase;
        AscendC::DataCopy(yGm_[gmOffset], yLocal, curT);
    }

    __aicore__ inline void ComputeTile(uint32_t n, uint32_t oc, uint32_t tBase, uint32_t curT)
    {
        // Allocate y tile
        AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();

        bool firstTerm = true;

        for (uint32_t ci = 0; ci < Cin_; ++ci) {
            AscendC::LocalTensor<float> xWin = xWinQueue_.AllocTensor<float>();
            LoadXChannelWin(n, ci, tBase, curT, xWin);

            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            LoadWChannel(oc, ci, wLocal);

            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

            for (uint32_t k = 0; k < K_; ++k) {
                // tmp = xWin shifted by k of length curT
                AscendC::DataCopy(tmp, xWin[k], curT);

                // Read weight scalar from local tensor
                float wk = wLocal.GetValue(k);

                if (firstTerm) {
                    AscendC::Muls(yLocal, tmp, wk, curT);
                    firstTerm = false;
                } else {
                    AscendC::Muls(tmp, tmp, wk, curT);
                    AscendC::Add(yLocal, yLocal, tmp, curT);
                }
            }

            tmpQueue_.FreeTensor(tmp);
            wQueue_.FreeTensor(wLocal);
            xWinQueue_.FreeTensor(xWin);
        }

        StoreY(n, oc, tBase, curT, yLocal);
        yQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1>   yQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>    xWinQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>  tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>    wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0}, Lin_{0}, K_{0}, Lout_{0}, tileT_{0}, tileWin_{0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.Lin, tiling_data.K, tiling_data.Lout,
            tiling_data.tileT);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv1d_standard_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    TORCH_CHECK(x.dim() == 3 && weight.dim() == 3, \"Expect x[N,Cin,L] and weight[Cout,Cin,K]\");
    TORCH_CHECK(x.scalar_type() == at::kFloat && weight.scalar_type() == at::kFloat, \"dtype must be float32\");
    TORCH_CHECK(x.size(1) == weight.size(1), \"Cin mismatch\");
    int64_t N = x.size(0);
    int64_t Cin = x.size(1);
    int64_t Lin = x.size(2);
    int64_t Cout = weight.size(0);
    int64_t K = weight.size(2);
    TORCH_CHECK(K > 0 && Lin >= K, \"Invalid kernel length for input\");
    int64_t Lout = Lin - K + 1;
    auto result = at::empty({N, Cout, Lout}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard1dCustom, x, weight, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv_standard_1d_custom\", &conv1d_standard_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"conv_standard_1d_custom\", &conv1d_standard_impl_npu, \"Standard 1D Conv (ConvStandard1dCustom on NPU)\");
}
