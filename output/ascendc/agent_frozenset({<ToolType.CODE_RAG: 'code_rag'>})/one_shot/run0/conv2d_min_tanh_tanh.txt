#include "conv2d_min_tanh_tanh_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

// UB model per-core for fused conv-reduce-tanh:
//   - vec buffers: minVec (P), outVec (P), tmpVec (P)
//   - row buffer: inRow (P + K - 1)
// Total floats ~ 4*P + (K - 1)  => bytes ~ 16*P + 4*(K-1)
// Add headroom and queues: budget ~ 192KB conservatively.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_TILEW = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMinTanhTanhCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    const uint32_t n     = static_cast<uint32_t>(xShape->GetDim(0));
    const uint32_t cIn   = static_cast<uint32_t>(xShape->GetDim(1));
    const uint32_t h     = static_cast<uint32_t>(xShape->GetDim(2));
    const uint32_t w     = static_cast<uint32_t>(xShape->GetDim(3));

    const uint32_t cOut  = static_cast<uint32_t>(wShape->GetDim(0));
    const uint32_t k     = static_cast<uint32_t>(wShape->GetDim(2)); // assume square kernel

    const uint32_t hOut  = h - k + 1;
    const uint32_t wOut  = w - k + 1;

    // tileW from UB budget: bytes ~ 16*tileW + 4*(k-1)
    uint32_t maxTileW = 0;
    if (UB_BUDGET_BYTES > 4 * (k - 1)) {
        maxTileW = (UB_BUDGET_BYTES - 4 * (k - 1)) / 16;
    }
    if (maxTileW == 0) maxTileW = ALIGN_ELEMS;
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW > MAX_TILEW) maxTileW = AlignDown(MAX_TILEW, ALIGN_ELEMS);
    uint32_t tileW = maxTileW;
    if (wOut < tileW) {
        const uint32_t aligned = AlignDown(wOut, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : wOut;
    }
    if (tileW == 0) tileW = 1;

    // block dim across N * H_out rows
    const uint32_t totalRows = (hOut == 0) ? 1u : (n * hOut);
    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // detect bias existence
    uint32_t hasBias = (context->GetInputShapeSize() >= 3 && context->GetInputShape(2)->GetShapeSize() > 0) ? 1u : 0u;

    tiling.set_n(n);
    tiling.set_cIn(cIn);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_cOut(cOut);
    tiling.set_k(k);
    tiling.set_hOut(hOut);
    tiling.set_wOut(wOut);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

// Fused Conv2D(stride=1, pad=0, dilate=1) + reduce_min over output channels + tanh + tanh.
// Data layout: NCHW for x; OIHW for weight. Output: [N,1,H-K+1,W-K+1].

constexpr int32_t VEC_BUFFERS = 4; // minVec, outVec, tmpVec, biasVec(when needed)
constexpr int32_t ROW_BUFFERS = 1;

class KernelConv2dMinTanhTanhCustom {
public:
    __aicore__ inline KernelConv2dMinTanhTanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t h, uint32_t wIn,
                                uint32_t cOut, uint32_t k, uint32_t hOut, uint32_t wOut,
                                uint32_t tileW, uint32_t hasBias)
    {
        n_ = n; cIn_ = cIn; h_ = h; wIn_ = wIn; cOut_ = cOut;
        k_ = k; hOut_ = hOut; wOut_ = wOut; tileW_ = tileW; hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * cIn_ * h_ * wIn_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cOut_) * cIn_ * k_ * k_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)bias, cOut_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * 1u * hOut_ * wOut_);

        pipe_.InitBuffer(vecQ_, VEC_BUFFERS, tileW_ * sizeof(float));
        pipe_.InitBuffer(rowQ_, ROW_BUFFERS, (tileW_ + k_ - 1) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cIn_ == 0 || cOut_ == 0 || hOut_ == 0 || wOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t rowsTotal = n_ * hOut_;
        for (uint32_t rowIdx = blockIdx; rowIdx < rowsTotal; rowIdx += blockNum) {
            const uint32_t n = rowIdx / hOut_;
            const uint32_t ho = rowIdx % hOut_;

            for (uint32_t wo = 0; wo < wOut_; wo += tileW_) {
                const uint32_t len = AscendC::min((uint32_t)tileW_, (uint32_t)(wOut_ - wo));

                AscendC::LocalTensor<float> minVec = vecQ_.AllocTensor<float>();
                bool firstCo = true;

                for (uint32_t co = 0; co < cOut_; ++co) {
                    AscendC::LocalTensor<float> outVec = vecQ_.AllocTensor<float>();
                    AscendC::Dup(outVec, 0.0f, len);

                    // accumulate dot products over C_in and KxK
                    for (uint32_t ci = 0; ci < cIn_; ++ci) {
                        for (uint32_t kh = 0; kh < k_; ++kh) {
                            // load contiguous input row segment [len + k - 1]
                            const uint64_t xoff = XOffset(n, ci, ho + kh, wo);
                            AscendC::LocalTensor<float> inRow = rowQ_.AllocTensor<float>();
                            AscendC::DataCopy(inRow, xGm_[xoff], len + k_ - 1);

                            // load weight block for this (co, ci, kh, :)
                            // read k_ scalars into a small local buffer
                            AscendC::LocalTensor<float> wLine = vecQ_.AllocTensor<float>();
                            const uint64_t woff = WOffset(co, ci, kh, 0);
                            AscendC::DataCopy(wLine, wGm_[woff], k_);

                            // for each kw, outVec += inRow[kw:kw+len] * wval
                            for (uint32_t kw = 0; kw < k_; ++kw) {
                                AscendC::LocalTensor<float> tmp = vecQ_.AllocTensor<float>();
                                // slice view: start at offset 'kw', length 'len'
                                AscendC::LocalTensor<float> inSlice = inRow[kw];
                                // broadcasting multiply by scalar wLine[kw]
                                float wval = GetScalarFromUb(wLine, kw);
                                AscendC::Muls(tmp, inSlice, wval, len);
                                AscendC::Add(outVec, outVec, tmp, len);
                                vecQ_.FreeTensor(tmp);
                            }

                            vecQ_.FreeTensor(wLine);
                            rowQ_.FreeTensor(inRow);
                        }
                    }

                    if (hasBias_) {
                        // add bias (scalar) to outVec
                        float biasVal = GetScalarFromGm(bGm_, co);
                        AscendC::Adds(outVec, outVec, biasVal, len);
                    }

                    if (firstCo) {
                        AscendC::DataCopy(minVec, outVec, len);
                        firstCo = false;
                    } else {
                        AscendC::Min(minVec, minVec, outVec, len);
                    }
                    vecQ_.FreeTensor(outVec);
                }

                // tanh then tanh
                AscendC::LocalTensor<float> tmpAct = vecQ_.AllocTensor<float>();
                AscendC::Tanh(tmpAct, minVec, len);
                AscendC::Tanh(minVec, tmpAct, len);
                vecQ_.FreeTensor(tmpAct);

                // write out to y[n,0,ho,wo:wo+len)
                const uint64_t yoff = YOffset(n, ho, wo);
                AscendC::DataCopy(yGm_[yoff], minVec, len);

                vecQ_.FreeTensor(minVec);
            }
        }
    }

private:
    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * cIn_ + c) * h_ + h) * wIn_ + w;
    }
    __aicore__ inline uint64_t WOffset(uint32_t co, uint32_t ci, uint32_t kh, uint32_t kw) const
    {
        return ((((uint64_t)co * cIn_ + ci) * k_ + kh) * k_) + kw;
    }
    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t ho, uint32_t wo) const
    {
        // y shape: [n,1,hOut,wOut]
        return (((uint64_t)n * 1u + 0u) * hOut_ + ho) * wOut_ + wo;
    }

    // Helper to fetch scalar from UB local tensor at index idx.
    __aicore__ inline float GetScalarFromUb(const AscendC::LocalTensor<float>& t, uint32_t idx) const
    {
        // Assume AscendC allows reading UB element to a scalar through a temporary 1-len vector op path.
        AscendC::LocalTensor<float> tmp = t[idx];
        // Using a vector-to-scalar move: Muls to itself with 1.0 then read via built-in conversion
        // Here we rely on intrinsic that returns the first lane as scalar for control usage.
        return AscendC::GetValue(tmp); // conceptual helper provided by AscendC
    }

    // Helper to fetch scalar from GM bias[co]
    __aicore__ inline float GetScalarFromGm(AscendC::GlobalTensor<float>& gm, uint32_t idx) const
    {
        AscendC::LocalTensor<float> bt = vecQ_.AllocTensor<float>();
        AscendC::DataCopy(bt, gm[idx], 1);
        float val = AscendC::GetValue(bt); // read first lane
        vecQ_.FreeTensor(bt);
        return val;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, VEC_BUFFERS> vecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, ROW_BUFFERS> rowQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, h_{0}, wIn_{0}, cOut_{0};
    uint32_t k_{0}, hOut_{0}, wOut_{0}, tileW_{0};
    uint32_t hasBias_{0};
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanhCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.n, tiling_data.cIn, tiling_data.h, tiling_data.w,
            tiling_data.cOut, tiling_data.k, tiling_data.hOut, tiling_data.wOut,
            tiling_data.tileW, tiling_data.hasBias);
    op.Process();
}

static inline at::Tensor allocate_output(const at::Tensor& x, const at::Tensor& w) {
    // shapes: x: [N, Ci, H, W], w: [Co, Ci, K, K]
    const auto N = x.size(0);
    const auto H = x.size(2);
    const auto W = x.size(3);
    const auto K = w.size(2);
    const auto Hout = H - K + 1;
    const auto Wout = W - K + 1;
    return at::empty({N, 1, Hout, Wout}, x.options());
}

                                         const at::Tensor& weight,
                                         c10::optional<at::Tensor> bias_opt) {

    if (bias_opt.has_value()) {
        EXEC_NPU_CMD(aclnnConv2dMinTanhTanhCustom, x, weight, bias_opt.value(), result);
    } else {
        EXEC_NPU_CMD(aclnnConv2dMinTanhTanhCustom, x, weight, c10::optional<at::Tensor>(), result);
    }
    return result;
}
