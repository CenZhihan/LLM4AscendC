kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom {
public:
    __aicore__ inline KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t d, uint32_t h, uint32_t w,
                                uint32_t tileSpatial, float scale)
    {
        n_ = n; c_ = c; d_ = d; h_ = h; w_ = w;
        spatial_ = d_ * h_ * w_;
        tileSpatial_ = tileSpatial;
        scale_ = scale;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * c_ * spatial_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * c_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileSpatial_ * sizeof(float));
        // two VECCALC buffers of size 1: accumulator and temp reduce
        pipe_.InitBuffer(calcQueue_, 2, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || spatial_ == 0) {
            return;
        }

        const uint32_t nc = n_ * c_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t idx = blockIdx; idx < nc; idx += blockNum) {
            // Determine (n,c) from linear index
            const uint32_t n_idx = idx / c_;
            const uint32_t c_idx = idx % c_;
            const uint64_t baseOffset = (static_cast<uint64_t>(n_idx) * c_ + c_idx) * spatial_;

            // accumulator initialized to 0
            AscendC::LocalTensor<float> accum = calcQueue_.AllocTensor<float>();
            AscendC::Duplicate(accum, 0.0f, 1);

            uint32_t processed = 0;
            while (processed < spatial_) {
                uint32_t cur = spatial_ - processed;
                if (cur > tileSpatial_) cur = tileSpatial_;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[baseOffset + processed], cur);
                inQueueX_.EnQue(xLocal);

                // compute
                AscendC::LocalTensor<float> inTile = inQueueX_.DeQue<float>();
                // x *= scale
                AscendC::Muls(inTile, inTile, scale_, cur);
                // reduce sum into a 1-element tmp
                AscendC::LocalTensor<float> tmp = calcQueue_.AllocTensor<float>();
                AscendC::ReduceSum(tmp, inTile, cur);
                // accum += tmp
                AscendC::Add(accum, accum, tmp, 1);

                calcQueue_.FreeTensor(tmp);
                inQueueX_.FreeTensor(inTile);

                processed += cur;
            }

            // average: accum /= spatial_
            const float invSpatial = 1.0f / static_cast<float>(spatial_);
            AscendC::Muls(accum, accum, invSpatial, 1);
            // clamp to [0, 1]
            AscendC::Maxs(accum, accum, 0.0f, 1);
            AscendC::Mins(accum, accum, 1.0f, 1);

            // write out single value at index idx
            AscendC::DataCopy(yGm_[idx], accum, 1);

            calcQueue_.FreeTensor(accum);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> calcQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, d_{0}, h_{0}, w_{0};
    uint32_t spatial_{0};
    uint32_t tileSpatial_{0};
    float scale_{1.0f};
};

extern "C" __global__ __aicore__ void conv_transpose3d_multiply_max_global_avg_pool_clamp_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{

    op.Init(x, y,

            tiling.tileSpatial, tiling.scale);
    op.Process();
}
