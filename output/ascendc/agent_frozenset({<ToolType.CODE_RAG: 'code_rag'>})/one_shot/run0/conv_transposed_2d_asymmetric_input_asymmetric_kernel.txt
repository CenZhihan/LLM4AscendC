#include "conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) { return static_cast<uint32_t>((a + b - 1) / b); }
}

// UB model:
//   yAcc: 1 * tileWo
//   xSlice: 1 * tileWo
//   small misc: ~ a few scalars
// bytes ~= (2 * tileWo) * 4
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed2dAsymmetricInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape* xShape = context->GetInputShape(0);
    const gert::Shape* wShape = context->GetInputShape(1);

    // Expect:
    // x: [N, Ci, Hi, Wi]
    // w: [Ci, Co, Kh, Kw]  (PyTorch conv_transpose2d)
    uint32_t N  = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    uint32_t Ci = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    uint32_t Hi = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
    uint32_t Wi = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));

    uint32_t Ci_w = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(0));
    uint32_t Co   = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(1));
    uint32_t Kh   = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(2));
    uint32_t Kw   = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(3));

    if (Ci_w != Ci) {
        return ge::GRAPH_FAILED;
    }

    // stride=(1,1), padding=(0,0), dilation=(1,1), output_padding=(0,0)
    const uint32_t Ho = Hi + Kh - 1;
    const uint32_t Wo = Wi + Kw - 1;

    // Tile width chosen from UB budget
    uint32_t maxTileWo = UB_BUDGET_BYTES / 8; // 2 buffers * 4B
    maxTileWo = AlignDown(maxTileWo, ALIGN_ELEMS);
    if (maxTileWo < ALIGN_ELEMS) maxTileWo = ALIGN_ELEMS;
    if (maxTileWo > Wo) {
        uint32_t aligned = AlignDown(Wo, ALIGN_ELEMS);
        maxTileWo = (aligned > 0) ? aligned : Wo;
    }
    uint32_t tileWo = maxTileWo;

    const uint64_t totalTilesW = CeilDiv(Wo, tileWo);
    const uint64_t totalItems = static_cast<uint64_t>(N) * Co * Ho * totalTilesW;

    uint32_t blockDim = 24;
    if (totalItems < 8) blockDim = static_cast<uint32_t>(totalItems);
    if (blockDim < 1) blockDim = 1;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Ci(Ci);
    tiling.set_Hi(Hi);
    tiling.set_Wi(Wi);
    tiling.set_Co(Co);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_tileWo(tileWo);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t Hi, uint32_t Wi,
                                uint32_t Co, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t tileWo)
    {
        N_ = N; Ci_ = Ci; Hi_ = Hi; Wi_ = Wi;
        Co_ = Co; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;
        tileWo_ = tileWo;

        xSize_ = static_cast<uint64_t>(N_) * Ci_ * Hi_ * Wi_;
        wSize_ = static_cast<uint64_t>(Ci_) * Co_ * Kh_ * Kw_;
        ySize_ = static_cast<uint64_t>(N_) * Co_ * Ho_ * Wo_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize_);

        pipe_.InitBuffer(yAccQueue_, 1, tileWo_ * sizeof(float));
        pipe_.InitBuffer(xSliceQueue_, 1, tileWo_ * sizeof(float));
        pipe_.InitBuffer(wScalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Ci_ == 0 || Co_ == 0 || Ho_ == 0 || Wo_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint32_t tilesW = (Wo_ + tileWo_ - 1) / tileWo_;
        const uint64_t totalItems = (uint64_t)N_ * Co_ * Ho_ * tilesW;

        for (uint64_t item = blockIdx; item < totalItems; item += blockNum) {
            // Unravel item -> (n, co, ho, tw)
            uint64_t t = item;
            uint32_t tw = (uint32_t)(t % tilesW);
            t /= tilesW;
            uint32_t ho = (uint32_t)(t % Ho_);
            t /= Ho_;
            uint32_t co = (uint32_t)(t % Co_);
            t /= Co_;
            uint32_t n = (uint32_t)t;

            uint32_t woBase = tw * tileWo_;
            uint32_t curWoLen = Wo_ - woBase;
            if (curWoLen > tileWo_) curWoLen = tileWo_;
            ComputeOneTile(n, co, ho, woBase, curWoLen);
        }
    }

private:
    __aicore__ inline float LoadWeightScalar(uint64_t offset)
    {
        AscendC::LocalTensor<float> wLoc = wScalarQueue_.AllocTensor<float>();
        AscendC::DataCopy(wLoc, wGm_[offset], 1);
        // Read back scalar from UB
        float val = wLoc.GetValue(0);
        wScalarQueue_.FreeTensor(wLoc);
        return val;
    }

    __aicore__ inline void ComputeOneTile(uint32_t n, uint32_t co, uint32_t ho,
                                          uint32_t woBase, uint32_t len)
    {
        AscendC::LocalTensor<float> yAcc = yAccQueue_.AllocTensor<float>();
        AscendC::Duplicate(yAcc, 0.0f, len);

        // For stride=1, padding=0, dilation=1
        const int32_t kH_1 = (int32_t)Kh_ - 1;
        const int32_t kW_1 = (int32_t)Kw_ - 1;

        for (uint32_t ci = 0; ci < Ci_; ++ci) {
            for (uint32_t kh = 0; kh < Kh_; ++kh) {
                const int32_t ih = (int32_t)ho - kH_1 + (int32_t)kh;
                if (ih < 0 || ih >= (int32_t)Hi_) continue;

                for (uint32_t kw = 0; kw < Kw_; ++kw) {
                    const int32_t iw_start = (int32_t)woBase - kW_1 + (int32_t)kw;
                    int32_t seg_start = iw_start;
                    int32_t seg_end   = iw_start + (int32_t)len; // exclusive

                    int32_t vb = seg_start < 0 ? 0 : seg_start;
                    int32_t ve = seg_end > (int32_t)Wi_ ? (int32_t)Wi_ : seg_end;
                    int32_t valid = ve - vb;
                    if (valid <= 0) continue;

                    uint32_t dst_off = (uint32_t)(vb - seg_start);

                    // Load input segment x[n,ci,ih, vb : vb+valid]
                    AscendC::LocalTensor<float> xSlice = xSliceQueue_.AllocTensor<float>();
                    const uint64_t xOffset = ((((uint64_t)n * Ci_ + ci) * Hi_ + (uint32_t)ih) * Wi_ + (uint32_t)vb);
                    AscendC::DataCopy(xSlice, xGm_[xOffset], (uint32_t)valid);

                    // Load weight scalar w[ci,co,kh,kw]
                    const uint64_t wOffset = ((((uint64_t)ci * Co_ + co) * Kh_ + kh) * Kw_ + kw);
                    float wVal = LoadWeightScalar(wOffset);

                    // yAcc[dst_off:dst_off+valid] += xSlice[0:valid] * wVal
                    AscendC::Muls(xSlice, xSlice, wVal, (uint32_t)valid);
                    AscendC::LocalTensor<float> ySub = yAcc[dst_off];
                    AscendC::Add(ySub, ySub, xSlice, (uint32_t)valid);

                    xSliceQueue_.FreeTensor(xSlice);
                }
            }
        }

        // Write back
        const uint64_t yOffset = ((((uint64_t)n * Co_ + co) * Ho_ + ho) * Wo_ + woBase);
        AscendC::DataCopy(yGm_[yOffset], yAcc, len);
        yAccQueue_.FreeTensor(yAcc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yAccQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xSliceQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wScalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Ci_{0}, Hi_{0}, Wi_{0};
    uint32_t Co_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t tileWo_{0};

    uint64_t xSize_{0}, wSize_{0}, ySize_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, w, y,
            tiling_data.N, tiling_data.Ci, tiling_data.Hi, tiling_data.Wi,
            tiling_data.Co, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo, tiling_data.tileWo);
    op.Process();
}
