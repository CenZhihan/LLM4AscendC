#include "gemm_multiply_leakyrelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB memory model (floats):
//   xChunk: 1 * tileN
//   wChunk: 1 * tileN
//   tmpVec: 1 * tileN
//   accVec: 1 * tileN
// Scalars and small temporaries: negligible
// Total bytes ~ 16 * tileN
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative
static constexpr uint32_t ALIGN_ELEMS = 256;             // align for better DMA
static constexpr uint32_t MIN_TILE = 256;
static constexpr uint32_t MAX_BLOCKS = 24;
static constexpr uint32_t MIN_BLOCKS = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMultiplyLeakyreluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0); // [M, N]
    const gert::Shape *w_shape = context->GetInputShape(1); // [K, N]

    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t K = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));

    uint32_t tileN = UB_BUDGET_BYTES / 16;  // bytes / (4 buffers * 4B)
    tileN = AlignDown(tileN, ALIGN_ELEMS);
    if (tileN < MIN_TILE) tileN = MIN_TILE;
    if (tileN > N) {
        uint32_t aligned = AlignDown(N, ALIGN_ELEMS);
        tileN = (aligned > 0) ? aligned : N;
    }
    if (tileN == 0) tileN = 1;

    uint32_t blockDim = (M == 0) ? 1u : M;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t QUEUE_DEPTH = 1;

class KernelGemmMultiplyLeakyreluCustom {
public:
    __aicore__ inline KernelGemmMultiplyLeakyreluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR multiplier, GM_ADDR negative_slope,
                                GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t tileN)
    {
        M_ = M; N_ = N; K_ = K; tileN_ = tileN > 0 ? tileN : 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * N_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)K_ * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)K_);
        mulGm_.SetGlobalBuffer((__gm__ float*)multiplier, 1);
        nsGm_.SetGlobalBuffer((__gm__ float*)negative_slope, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * K_);

        pipe_.InitBuffer(xQueue_, QUEUE_DEPTH, tileN_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, QUEUE_DEPTH, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, QUEUE_DEPTH, tileN_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, QUEUE_DEPTH, tileN_ * sizeof(float));

        pipe_.InitBuffer(scalarQueue_, 8, sizeof(float));   // small scalars
        pipe_.InitBuffer(outQueue_, 2, sizeof(float));      // output scalars
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            // For each output column k
            for (uint32_t k = 0; k < K_; ++k) {
                // Accumulate dot(x[m,:], w[k,:]) into accVec then reduce to scalar
                bool accInitialized = false;
                uint32_t accLen = 0;

                AscendC::LocalTensor<float> accLocal;

                for (uint32_t n0 = 0; n0 < N_; n0 += tileN_) {
                    uint32_t cur = N_ - n0;
                    if (cur > tileN_) cur = tileN_;

                    AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[(uint64_t)m * N_ + n0], cur);

                    AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                    AscendC::DataCopy(wLocal, wGm_[(uint64_t)k * N_ + n0], cur);

                    AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::Mul(tmpLocal, xLocal, wLocal, cur);

                    if (!accInitialized) {
                        accLocal = accQueue_.AllocTensor<float>();
                        AscendC::DataCopy(accLocal, tmpLocal, cur);
                        accLen = cur;
                        accInitialized = true;
                    } else {
                        // accumulate into first accLen elements
                        AscendC::Add(accLocal, accLocal, tmpLocal, cur);
                        // accLen remains size of current segment (cur) for reduction; we only keep meaningful region
                        accLen = cur;
                    }

                    tmpQueue_.FreeTensor(tmpLocal);
                    wQueue_.FreeTensor(wLocal);
                    xQueue_.FreeTensor(xLocal);
                }

                // Reduce accLocal[0:accLen] to scalar in-place
                if (!accInitialized) {
                    // No data means dot=0
                    accLocal = accQueue_.AllocTensor<float>();
                    // leave as undefined but we set to 0
                    AscendC::Zeros(accLocal, 1);
                    accLen = 1;
                }

                uint32_t len = accLen;
                while (len > 1) {
                    uint32_t half = len >> 1;
                    if (half > 0) {
                        AscendC::Add(accLocal, accLocal, accLocal[half], half);
                    }
                    if ((len & 1u) != 0u) {
                        AscendC::Add(accLocal, accLocal, accLocal[len - 1], 1);
                        len = half + 1;
                    } else {
                        len = half;
                    }
                }

                // yScalar = acc + b[k]
                AscendC::LocalTensor<float> yScalar = outQueue_.AllocTensor<float>();
                AscendC::DataCopy(yScalar, accLocal, 1);

                AscendC::LocalTensor<float> bLocal = scalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[k], 1);
                AscendC::Add(yScalar, yScalar, bLocal, 1);

                // multiply by multiplier scalar tensor
                AscendC::LocalTensor<float> mLocal = scalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(mLocal, mulGm_[0], 1);
                AscendC::Mul(yScalar, yScalar, mLocal, 1);

                // LeakyReLU: y = max(y,0) + ns * min(y,0)
                AscendC::LocalTensor<float> minV = scalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(minV, yScalar, 1);
                AscendC::Mins(minV, minV, 0.0f, 1);

                AscendC::LocalTensor<float> maxV = scalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(maxV, yScalar, 1);
                AscendC::Maxs(maxV, maxV, 0.0f, 1);

                AscendC::LocalTensor<float> nsLocal = scalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(nsLocal, nsGm_[0], 1);

                AscendC::LocalTensor<float> tmpS = scalarQueue_.AllocTensor<float>();
                AscendC::Mul(tmpS, minV, nsLocal, 1);

                AscendC::Add(yScalar, maxV, tmpS, 1);

                // Store to GM
                AscendC::DataCopy(yGm_[(uint64_t)m * K_ + k], yScalar, 1);

                // Free local buffers
                scalarQueue_.FreeTensor(tmpS);
                scalarQueue_.FreeTensor(nsLocal);
                scalarQueue_.FreeTensor(maxV);
                scalarQueue_.FreeTensor(minV);
                scalarQueue_.FreeTensor(mLocal);
                scalarQueue_.FreeTensor(bLocal);
                outQueue_.FreeTensor(yScalar);
                accQueue_.FreeTensor(accLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 8> scalarQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> mulGm_;
    AscendC::GlobalTensor<float> nsGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0}, tileN_{0};
};

extern "C" __global__ __aicore__ void gemm_multiply_leakyrelu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR multiplier, GM_ADDR negative_slope, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMultiplyLeakyreluCustom op;
    op.Init(x, w, b, multiplier, negative_slope, y,
            tiling_data.M, tiling_data.N, tiling_data.K, tiling_data.tileN);
    op.Process();
}
