host_operator_src="""
#include "rms_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : ((a + b - 1u) / b);
}

// UB model for this kernel (float32):
//   acc:    1 * tileP
//   tmp:    1 * tileP
//   in:     1 * tileP
//   out:    1 * tileP
// Total floats ~ 4 * tileP => bytes = 16 * tileP
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative budget
static constexpr uint32_t ALIGN_ELEMS = 1024;            // favor large DMA bursts
static constexpr uint32_t MIN_TILE = 1024;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    RmsNormCustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    const int64_t dimNum = inShape->GetOriginShape().GetDimNum();
    if (dimNum < 2) {
        return ge::GRAPH_FAILED;
    }

    // Axis layout: (batch, channel, spatial...)
    const int64_t batch = inShape->GetOriginShape().GetDim(0);
    const int64_t channels = inShape->GetOriginShape().GetDim(1);

    uint64_t s64 = 1;
    for (int64_t i = 2; i < dimNum; ++i) {
        s64 *= static_cast<uint64_t>(inShape->GetOriginShape().GetDim(i));
    }

    const uint32_t nOuter = static_cast<uint32_t>(batch);
    const uint32_t c = static_cast<uint32_t>(channels);
    const uint32_t s = static_cast<uint32_t>(s64);

    const uint32_t totalLength = (uint32_t)(static_cast<uint64_t>(nOuter) * c * s);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *eps = attrs->GetAttrPointer<float>(0);

    // tileP from UB budget: bytes = 16 * tileP
    uint32_t maxTileP = UB_BUDGET_BYTES / 16;
    if (maxTileP < MIN_TILE) maxTileP = MIN_TILE;
    maxTileP = AlignDown(maxTileP, ALIGN_ELEMS);

    uint32_t tileP = maxTileP;
    if (s > 0 && s < tileP) {
        uint32_t aligned = AlignDown(s, ALIGN_ELEMS);
        tileP = aligned > 0 ? aligned : s;
    }

    if (tileP == 0) tileP = MIN_TILE;

    const uint32_t tilesPerN = CeilDiv(s, tileP);
    uint32_t totalTiles = (nOuter == 0 || tilesPerN == 0) ? 0u : (nOuter * tilesPerN);
    if (totalTiles == 0) totalTiles = 1;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_nOuter(nOuter);
    tiling.set_c(c);
    tiling.set_s(s);
    tiling.set_tileP(tileP);
    tiling.set_blockDim(blockDim);
    tiling.set_totalLength(totalLength);
    tiling.set_eps(*eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class RmsNormCustom : public OpDef {
public:
    explicit RmsNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("eps").AttrType(OPTIONAL).Float(1e-5);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(RmsNormCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;
constexpr int32_t TMP_BUF_NUM = 2;

class KernelRmsNormCustom {
public:
    __aicore__ inline KernelRmsNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t nOuter, uint32_t c, uint32_t s,
                                uint32_t tileP, float eps)
    {
        nOuter_ = nOuter;
        c_ = c;
        s_ = s;
        tileP_ = tileP;
        eps_ = eps;

        uint32_t totalLen = nOuter_ * c_ * s_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen);

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileP_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileP_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMP_BUF_NUM, tileP_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (nOuter_ == 0 || c_ == 0 || s_ == 0 || tileP_ == 0) {
            return;
        }

        const uint32_t tilesPerN = (s_ + tileP_ - 1u) / tileP_;
        const uint32_t totalTiles = nOuter_ * tilesPerN;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t n = tileId / tilesPerN;
            const uint32_t tileInN = tileId - n * tilesPerN;
            uint32_t sStart = tileInN * tileP_;
            uint32_t curP = s_ - sStart;
            if (curP > tileP_) curP = tileP_;
            if (curP == 0) continue;

            // Allocate persistent accumulator buffer (will hold denom after reduction)
            AscendC::LocalTensor<float> accLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Duplicate(accLocal, 0.0f, curP);

            // First pass: accumulate sum of squares across channels
            for (uint32_t c = 0; c < c_; ++c) {
                const uint32_t base = ((n * c_) + c) * s_ + sStart;

                // Load x chunk
                AscendC::LocalTensor<float> xLocalAlloc = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocalAlloc, xGm_[base], curP);
                inQueueX_.EnQue(xLocalAlloc);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();

                // tmp = x * x
                AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                AscendC::Mul(tmpLocal, xLocal, xLocal, curP);

                // acc += tmp
                AscendC::Add(accLocal, accLocal, tmpLocal, curP);

                tmpQueue_.FreeTensor(tmpLocal);
                inQueueX_.FreeTensor(xLocal);
            }

            // Compute denom = sqrt(mean + eps)
            if (c_ > 1u) {
                float invC = 1.0f / static_cast<float>(c_);
                AscendC::Muls(accLocal, accLocal, invC, curP);
            }
            AscendC::Adds(accLocal, accLocal, eps_, curP);
            AscendC::Sqrt(accLocal, accLocal, curP);

            // Second pass: normalize and write out
            for (uint32_t c = 0; c < c_; ++c) {
                const uint32_t base = ((n * c_) + c) * s_ + sStart;

                // Load x
                AscendC::LocalTensor<float> xLocalAlloc = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocalAlloc, xGm_[base], curP);
                inQueueX_.EnQue(xLocalAlloc);
                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();

                // y = x / denom
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Div(yLocal, xLocal, accLocal, curP);
                outQueueY_.EnQue(yLocal);
                inQueueX_.FreeTensor(xLocal);

                AscendC::LocalTensor<float> yLocalOut = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[base], yLocalOut, curP);
                outQueueY_.FreeTensor(yLocalOut);
            }

            // Free persistent buffer
            tmpQueue_.FreeTensor(accLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t nOuter_ {0};
    uint32_t c_ {0};
    uint32_t s_ {0};
    uint32_t tileP_ {0};
    float eps_ {1e-5f};
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNormCustom op;
    op.Init(x, y,
            tiling_data.nOuter, tiling_data.c, tiling_data.s,
            tiling_data.tileP, tiling_data.eps);
    op.Process();
}
"""
