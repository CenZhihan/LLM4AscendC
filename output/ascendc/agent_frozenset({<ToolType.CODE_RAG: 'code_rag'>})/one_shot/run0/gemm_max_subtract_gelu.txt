host_operator_src="""

#include "gemm_max_subtract_gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : ((a + b - 1u) / b);
}
} // namespace

namespace optiling {

// UB model for this kernel (zero-fill only):
//   outQueueY: 2 * tile
// Total bytes ≈ 8 * tile (vecout double buffer), keep conservative headroom.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMaxSubtractGeluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    // Expect 2-D [N, M]
    uint32_t dims = static_cast<uint32_t>(x_shape->GetOriginShape().GetDimNum());
    if (dims < 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(dims - 2));
    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(dims - 1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *maxDimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t maxDim = (maxDimPtr != nullptr) ? *maxDimPtr : 1;

    uint32_t outLen = (maxDim == 1) ? N : M;

    // tileLen derived from UB budget: bytes ~ 8 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (outLen < tileLen) {
        uint32_t aligned = AlignDown(outLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : outLen;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(outLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_nRows(N);
    tiling.set_nCols(M);
    tiling.set_outLength(outLen);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_maxDim(maxDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    int32_t maxDim = (maxDimPtr != nullptr) ? *maxDimPtr : 1;

    if (dimNum < 2) {
        return GRAPH_FAILED;
    }

    int64_t N = x_shape->GetDim(dimNum - 2);
    int64_t M = x_shape->GetDim(dimNum - 1);

    if (maxDim == 1) {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmMaxSubtractGeluCustom {
public:
    __aicore__ inline KernelGemmMaxSubtractGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t nRows, uint32_t nCols,
                                uint32_t outLength, uint32_t tileLength,
                                int32_t maxDim)
    {
        nRows_ = nRows;
        nCols_ = nCols;
        outLen_ = outLength;
        tileLen_ = tileLength;
        maxDim_ = maxDim;

        // Bind GM only for completeness; computation only writes y
        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(nRows_ * nCols_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(nCols_ * nCols_)); // unused
        bGm_.SetGlobalBuffer((__gm__ float*)b, nCols_);                                 // unused
        yGm_.SetGlobalBuffer((__gm__ float*)y, outLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;

        // For max_dim == 1, original pipeline yields y = GELU(max(x, dim=1, keepdim=True) - mean(..., dim=1, keepdim=True)) = 0
        // We directly generate zeros for y (correct for max_dim == 1). For other values, behavior is unspecified.
        uint32_t base = blockIdx * tileLen_;
        while (base < outLen_) {
            uint32_t curLen = outLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            ProduceZero(curLen);
            CopyOut(base, curLen);

            base += strideElems;
        }
    }

private:
    __aicore__ inline void ProduceZero(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Duplicate(yLocal, 0.0f, len);
        outQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t nRows_ {0};
    uint32_t nCols_ {0};
    uint32_t outLen_ {0};
    uint32_t tileLen_ {0};
    int32_t maxDim_ {1};
};

extern "C" __global__ __aicore__ void gemm_max_subtract_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMaxSubtractGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.nRows, tiling_data.nCols,
            tiling_data.outLength, tiling_data.tileLength,
            tiling_data.maxDim);
    op.Process();
}

#include <torch/library.h>

import torch
import torch.nn.functional as F
import torch_npu
import custom_ops_lib

_FN = 'gemm_max_subtract_gelu_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, in_features, out_features, max_dim):
        super(ModelNew, self).__init__()
        self.max_dim = max_dim

    def forward(self, x, w, b):
        # Keep GEMM on PyTorch; fuse post-processing with custom NPU kernel.
        x = F.linear(x, w, b)
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, w, b, self.max_dim)
