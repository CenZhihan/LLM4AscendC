    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

    if (a_shape == nullptr || b_shape == nullptr) {
        return GRAPH_PARAM_INVALID;

    if (a_shape->GetStorageShape().GetDimNum() != 2 ||
        b_shape->GetStorageShape().GetDimNum() != 2) {
        return GRAPH_PARAM_INVALID;

kernel_src="""
#include "kernel_operator.h"

// This kernel computes C = tril(A @ B) for lower triangular A and B.
// Inputs A, B are float16; output C is float32 for accumulation precision.

class KernelMatmulForLowerTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForLowerTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c, uint32_t n)
    {
        n_ = n;
        aGm_.SetGlobalBuffer((__gm__ half*)a, static_cast<uint64_t>(n_) * n_);
        bGm_.SetGlobalBuffer((__gm__ half*)b, static_cast<uint64_t>(n_) * n_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, static_cast<uint64_t>(n_) * n_);

        // Simple per-row buffer for accumulation and a B-row cache.
        // UB budget is sufficient for typical N; choose a modest tile for columns.
        colTile_ = 1024;
        if (colTile_ > n_) colTile_ = n_;

        pipe_.InitBuffer(bRowQ_, 1, colTile_ * sizeof(half));
        pipe_.InitBuffer(outRowQ_, 1, colTile_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        for (uint32_t i = blockIdx; i < n_; i += blockNum) {
            // Iterate lower-triangular output columns in tiles
            for (uint32_t jStart = 0; jStart <= i; jStart += colTile_) {
                uint32_t validCols = colTile_;
                if (jStart + validCols > i + 1) {
                    validCols = i + 1 - jStart;
                }

                AscendC::LocalTensor<float> outRow = outRowQ_.AllocTensor<float>();
                AscendC::Dup(outRow, 0.0f, validCols);

                // Accumulate over k in [jStart, i] because A[i,k]==0 when k>i and B[k,j]==0 when k<j
                for (uint32_t k = jStart; k <= i; ++k) {
                    // Load B row k, columns jStart..jStart+validCols-1 (contiguous)
                    AscendC::LocalTensor<half> bRow = bRowQ_.AllocTensor<half>();
                    // GM offset for B[k, jStart]
                    uint64_t bOffset = static_cast<uint64_t>(k) * n_ + jStart;
                    AscendC::DataCopy(bRow, bGm_[bOffset], validCols);

                    // Load A[i, k] scalar from GM
                    // Compute GM offset for A[i,k]
                    uint64_t aIdx = static_cast<uint64_t>(i) * n_ + k;
                    // Fetch single element into a tiny local tensor
                    half aValHalfArr[1]; // will be placed into registers/stack
                    // Use a tiny local to DMA the scalar
                    AscendC::LocalTensor<half> aTmpLocal = bRowQ_.AllocTensor<half>();
                    AscendC::DataCopy(aTmpLocal, aGm_[aIdx], 1);
                    // Move scalar from local tensor to register by vector->scalar read
                    half aValHalf = aTmpLocal.GetValue(0); // requires AscendC LocalTensor scalar access
                    bRowQ_.FreeTensor(aTmpLocal);

                    // Convert half->float scalar for accumulation
                    float aVal = static_cast<float>(aValHalf);

                    // outRow += aVal * float(bRow)
                    // Convert bRow to float tensor
                    AscendC::LocalTensor<float> bRowF = outRowQ_.AllocTensor<float>();
                    AscendC::Cast(bRowF, bRow, validCols);
                    AscendC::Mla(outRow, bRowF, aVal, outRow, validCols); // outRow = bRowF * aVal + outRow

                    outRowQ_.FreeTensor(bRowF);
                    bRowQ_.FreeTensor(bRow);
                }

                // Store outRow to C[i, jStart : jStart+validCols)
                uint64_t cOffset = static_cast<uint64_t>(i) * n_ + jStart;
                AscendC::DataCopy(cGm_[cOffset], outRow, validCols);
                outRowQ_.FreeTensor(outRow);
            }

            // Zero the upper triangle for row i (j > i)
            if (i + 1 < n_) {
                uint32_t rem = n_ - (i + 1);
                uint64_t cOffset = static_cast<uint64_t>(i) * n_ + (i + 1);
                // Use a temporary local buffer to write zeros in chunks
                uint32_t chunk = colTile_;
                if (chunk > rem) chunk = rem;
                AscendC::LocalTensor<float> tmp = outRowQ_.AllocTensor<float>();
                uint32_t left = rem;
                AscendC::Dup(tmp, 0.0f, chunk);
                while (left > 0) {
                    uint32_t cur = (left > chunk) ? chunk : left;
                    AscendC::DataCopy(cGm_[cOffset], tmp, cur);
                    cOffset += cur;
                    left -= cur;
                }
                outRowQ_.FreeTensor(tmp);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bRowQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outRowQ_;

    AscendC::GlobalTensor<half> aGm_;
    AscendC::GlobalTensor<half> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t n_{0};
    uint32_t colTile_{1024};
};

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForLowerTriangularMatricesCustom op;
    uint32_t n = tiling_data.n;
    op.Init(A, B, C, n);
    op.Process();
}
