    const uint64_t Hout = Hin - kH + 1;
    const uint64_t Wout = Win - kW + 1;

    const uint64_t totalWorkVox = N * Dout * Hout * Wout;

    uint32_t blockDim = 24; // conservative default for 910B
    if (totalWorkVox < blockDim) {
        blockDim = static_cast<uint32_t>(totalWorkVox > 0 ? totalWorkVox : 1ULL);
        if (blockDim == 0) blockDim = 1;

        blockDim = 24u;

    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_cin(static_cast<uint32_t>(Cin));
    tiling.set_cout(static_cast<uint32_t>(Cout));
    tiling.set_inD(static_cast<uint32_t>(Din));
    tiling.set_inH(static_cast<uint32_t>(Hin));
    tiling.set_inW(static_cast<uint32_t>(Win));
    tiling.set_kD(static_cast<uint32_t>(kD));
    tiling.set_kH(static_cast<uint32_t>(kH));
    tiling.set_kW(static_cast<uint32_t>(kW));
    tiling.set_outD(static_cast<uint32_t>(Dout));
    tiling.set_outH(static_cast<uint32_t>(Hout));
    tiling.set_outW(static_cast<uint32_t>(Wout));

kernel_src="""
#include "kernel_operator.h"

extern \"C\" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling, tilingGm);

    const uint32_t N    = tiling.n;
    const uint32_t Cin  = tiling.cin;
    const uint32_t Cout = tiling.cout;
    const uint32_t Din  = tiling.inD;
    const uint32_t Hin  = tiling.inH;
    const uint32_t Win  = tiling.inW;

    const uint32_t kD = tiling.kD;
    const uint32_t kH = tiling.kH;
    const uint32_t kW = tiling.kW;

    const uint32_t Dout = tiling.outD;
    const uint32_t Hout = tiling.outH;
    const uint32_t Wout = tiling.outW;

    const uint64_t voxels = static_cast<uint64_t>(N) * Dout * Hout * Wout;

    if (voxels == 0 || Cout == 0 || Cin == 0) {
        return;
    }

    __gm__ float* xGm = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* wGm = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* yGm = reinterpret_cast<__gm__ float*>(y);

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

    for (uint64_t idx = blockIdx; idx < voxels; idx += blockNum) {
        // Decode (n, od, oh, ow) from linear index
        uint64_t t = idx;
        const uint32_t onePlane = Hout * Wout;
        const uint32_t oneDepth = onePlane;
        const uint32_t outDHW = Dout * Hout * Wout;

        const uint32_t n  = static_cast<uint32_t>(t / outDHW);
        t = t % outDHW;
        const uint32_t od = static_cast<uint32_t>(t / oneDepth);
        t = t % oneDepth;
        const uint32_t oh = static_cast<uint32_t>(t / Wout);
        const uint32_t ow = static_cast<uint32_t>(t % Wout);

        // For each output channel
        for (uint32_t oc = 0; oc < Cout; ++oc) {
            float acc = 0.0f;

            // Convolution sum over Cin and kernel
            for (uint32_t ic = 0; ic < Cin; ++ic) {
                for (uint32_t kz = 0; kz < kD; ++kz) {
                    const uint32_t iz = od + kz;
                    for (uint32_t ky = 0; ky < kH; ++ky) {
                        const uint32_t iy = oh + ky;
                        // kx loop: contiguous in W
                        for (uint32_t kx = 0; kx < kW; ++kx) {
                            const uint32_t ix = ow + kx;

                            // x offset: [n, ic, iz, iy, ix]
                            uint64_t xOff =
                                (((static_cast<uint64_t>(n)  * Cin + ic) * Din + iz) * Hin + iy) * Win + ix;

                            // weight offset: [oc, ic, kz, ky, kx]
                            uint64_t wOff =
                                (((static_cast<uint64_t>(oc) * Cin + ic) * kD + kz) * kH + ky) * kW + kx;

                            const float xv = xGm[xOff];
                            const float wv = wGm[wOff];
                            acc = AscendC::FAdd(acc, AscendC::FMul(xv, wv));
                        }
                    }
                }
            }

            // y offset: [n, oc, od, oh, ow]
            uint64_t yOff =
                (((static_cast<uint64_t>(n) * Cout + oc) * Dout + od) * Hout + oh) * Wout + ow;
            yGm[yOff] = acc;
        }
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv3d_std_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    // Output shape: N, Cout, D-kD+1, H-kH+1, W-kW+1 (stride=1, pad=0, dil=1)
    TORCH_CHECK(x.dim() == 5 && weight.dim() == 5, "Expected 5D tensors for x and weight");
    int64_t N = x.size(0);
    int64_t Cin = x.size(1);
    int64_t Din = x.size(2);
    int64_t Hin = x.size(3);
    int64_t Win = x.size(4);

    int64_t Cout = weight.size(0);
    int64_t CinW = weight.size(1);
    int64_t kD = weight.size(2);
    int64_t kH = weight.size(3);
    int64_t kW = weight.size(4);
    TORCH_CHECK(Cin == CinW, "Input channels must match weight in_channels");
    TORCH_CHECK(Din >= kD && Hin >= kH && Win >= kW, "Kernel larger than input with zero padding");

    at::Tensor y = at::empty({N, Cout, Din - kD + 1, Hin - kH + 1, Win - kW + 1}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard3dSquareInputSquareKernelCustom, x, weight, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_standard_3d_square_input_square_kernel_custom", &conv3d_std_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_standard_3d_square_input_square_kernel_custom", &conv3d_std_impl_npu,
          "Conv3D standard (3x3x3 no pad/stride=1) via ConvStandard3dSquareInputSquareKernelCustom on NPU");
