static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulSubtractMultiplyReluCustomTilingData tiling;

    // Shapes:
    // x: [M, K]
    // w: [N, K]  (PyTorch linear expects y = x @ w^T + b)
    // b: [N]
    // sub: scalar tensor
    // mul: scalar tensor
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    if (x_shape->GetDimNum() != 2 || w_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Kx = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetDim(1));

    if (Kx != Kw) {
        return ge::GRAPH_FAILED;
    }
    if (static_cast<uint32_t>(b_shape->GetDim(0)) != N) {
        return ge::GRAPH_FAILED;
    }

    // Choose conservative blockDim based on M
    uint32_t blockDim = M;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(Kx);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace_sizes = context->GetWorkspaceSizes(1);
    workspace_sizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

// Naive GEMM + fused (subtract, multiply, relu).
// x: [M, K], w: [N, K], b: [N], y: [M, N]
// y = relu( (x @ w^T + b) - sub ) * mul
// Note: w is [N, K] to match PyTorch Linear semantics.

struct TilingData {
    uint32_t M;
    uint32_t N;
    uint32_t K;
    uint32_t blockDim;
};

extern "C" __global__ __aicore__ void matmul_subtract_multiply_relu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR sub, GM_ADDR mul, GM_ADDR workspace, GM_ADDR tilingGm)
{
#if defined(ASCENDC_CPU_DEBUG)
    if (g_coreType == AscendC::AIV) {
        return;
    }
#endif

    // Load tiling data
    GET_TILING_DATA(tilingData, tilingGm);
    const uint32_t M = tilingData.M;
    const uint32_t N = tilingData.N;
    const uint32_t K = tilingData.K;

    // GM pointers
    __gm__ float* xG = (__gm__ float*)x;
    __gm__ float* wG = (__gm__ float*)w;
    __gm__ float* bG = (__gm__ float*)b;
    __gm__ float* yG = (__gm__ float*)y;
    __gm__ float* subG = (__gm__ float*)sub;
    __gm__ float* mulG = (__gm__ float*)mul;

    // Read scalar sub and mul from GM
    float subVal = 0.0f;
    float mulVal = 1.0f;
    if (subG != nullptr) {
        subVal = *subG;
    }
    if (mulG != nullptr) {
        mulVal = *mulG;
    }

    const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();
    const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();

    // Grid-stride loop over rows
    for (uint32_t m = blockIdx; m < M; m += blockNum) {
        const uint64_t xRowBase = (uint64_t)m * (uint64_t)K;
        const uint64_t yRowBase = (uint64_t)m * (uint64_t)N;

        for (uint32_t n = 0; n < N; ++n) {
            float acc = bG[n];
            const uint64_t wRowBase = (uint64_t)n * (uint64_t)K;

            // Dot product over K
            for (uint32_t k = 0; k < K; ++k) {
                const float xv = xG[xRowBase + k];
                const float wv = wG[wRowBase + k];
                acc = AscendC::Fma(xv, wv, acc); // acc += xv * wv
            }

            // Fused post-processing: (acc - subVal) * mulVal then ReLU
            float val = (acc - subVal) * mulVal;
            if (val < 0.0f) val = 0.0f;

            yG[yRowBase + n] = val;
        }
    }
}
"""
