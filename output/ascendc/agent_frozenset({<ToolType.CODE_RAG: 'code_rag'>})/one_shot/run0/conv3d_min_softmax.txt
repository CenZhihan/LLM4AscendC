static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1) / b);
}
static inline int64_t GetDim(const gert::Shape &s, int idx) {
    return s.GetDim(static_cast<uint32_t>(idx));
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv3dMinSoftmaxCustomTilingData tiling;

    const gert::Shape &x_ori = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape &w_ori = context->GetInputShape(1)->GetOriginShape();
    // x: [N, Cin, Din, Hin, Win]
    // w: [Cout, Cin, Kd, Kh, Kw]
    const uint32_t N   = static_cast<uint32_t>(GetDim(x_ori, 0));
    const uint32_t Cin = static_cast<uint32_t>(GetDim(x_ori, 1));
    const uint32_t Din = static_cast<uint32_t>(GetDim(x_ori, 2));
    const uint32_t Hin = static_cast<uint32_t>(GetDim(x_ori, 3));
    const uint32_t Win = static_cast<uint32_t>(GetDim(x_ori, 4));

    const uint32_t Cout = static_cast<uint32_t>(GetDim(w_ori, 0));
    const uint32_t CinW = static_cast<uint32_t>(GetDim(w_ori, 1));
    const uint32_t Kd   = static_cast<uint32_t>(GetDim(w_ori, 2));
    const uint32_t Kh   = static_cast<uint32_t>(GetDim(w_ori, 3));
    const uint32_t Kw   = static_cast<uint32_t>(GetDim(w_ori, 4));

    if (Cin != CinW) {
        return ge::GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();

    const int32_t *strideW = attrs->GetAttrPointer<int32_t>(2);

    const int32_t *dilH    = attrs->GetAttrPointer<int32_t>(7);
    const int32_t *dilW    = attrs->GetAttrPointer<int32_t>(8);
    const int32_t *groups  = attrs->GetAttrPointer<int32_t>(9);

    // For this kernel, we assume groups == 1
    if (*groups != 1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t sD = static_cast<uint32_t>(*strideD);
    const uint32_t sH = static_cast<uint32_t>(*strideH);
    const uint32_t sW = static_cast<uint32_t>(*strideW);
    const uint32_t pD = static_cast<uint32_t>(*padD);
    const uint32_t pH = static_cast<uint32_t>(*padH);
    const uint32_t pW = static_cast<uint32_t>(*padW);
    const uint32_t dD = static_cast<uint32_t>(*dilD);
    const uint32_t dH = static_cast<uint32_t>(*dilH);
    const uint32_t dW = static_cast<uint32_t>(*dilW);

    // Effective kernel sizes with dilation
    const uint32_t kEffD = (Kd - 1) * dD + 1;
    const uint32_t kEffH = (Kh - 1) * dH + 1;
    const uint32_t kEffW = (Kw - 1) * dW + 1;

    const uint32_t Hout = (Hin + 2 * pH - kEffH) / sH + 1;
    const uint32_t Wout = (Win + 2 * pW - kEffW) / sW + 1;

    const uint64_t totalSites = static_cast<uint64_t>(N) * Hout * Wout;

    // Conservative tiling as compute is heavy; use small tiles to increase parallelism
    uint32_t tileSites = 16;

    tiling.set_Hin(Hin);
    tiling.set_Win(Win);

    tiling.set_Kd(Kd);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);

    tiling.set_Hout(Hout);
    tiling.set_Wout(Wout);

    tiling.set_strideD(sD);
    tiling.set_strideH(sH);
    tiling.set_strideW(sW);

    tiling.set_padD(pD);
    tiling.set_padH(pH);
    tiling.set_padW(pW);

    tiling.set_dilationD(dD);
    tiling.set_dilationH(dH);
    tiling.set_dilationW(dW);

    tiling.set_groups(static_cast<uint32_t>(*groups));

{
    const gert::Shape &x_ori = context->GetInputShape(0)->GetOriginShape();
    const gert::Shape &w_ori = context->GetInputShape(1)->GetOriginShape();

    const int64_t Din = x_ori.GetDim(2);
    const int64_t Hin = x_ori.GetDim(3);
    const int64_t Win = x_ori.GetDim(4);

    const int64_t Cout = w_ori.GetDim(0);
    const int64_t CinW = w_ori.GetDim(1);
    const int64_t Kd   = w_ori.GetDim(2);

    const int64_t Kw   = w_ori.GetDim(4);
    if (Cin != CinW) return GRAPH_FAILED;

    gert::InferShapeContext *ctx = context;

    const gert::RuntimeAttrs *attrs = ctx->GetAttrs();
    const int32_t *strideD = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *strideH = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *strideW = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *padD    = attrs->GetAttrPointer<int32_t>(3);
    const int32_t *padH    = attrs->GetAttrPointer<int32_t>(4);
    const int32_t *padW    = attrs->GetAttrPointer<int32_t>(5);
    const int32_t *dilD    = attrs->GetAttrPointer<int32_t>(6);
    const int32_t *dilH    = attrs->GetAttrPointer<int32_t>(7);
    const int32_t *dilW    = attrs->GetAttrPointer<int32_t>(8);

    const int64_t kEffD = (Kd - 1) * (*dilD) + 1;
    const int64_t kEffH = (Kh - 1) * (*dilH) + 1;
    const int64_t kEffW = (Kw - 1) * (*dilW) + 1;

    const int64_t Dout = (Din + 2 * (*padD) - kEffD) / (*strideD) + 1;
    const int64_t Hout = (Hin + 2 * (*padH) - kEffH) / (*strideH) + 1;
    const int64_t Wout = (Win + 2 * (*padW) - kEffW) / (*strideW) + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);

    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Hout);
    y_shape->SetDim(3, Wout);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelConv3dMinSoftmaxCustom {
public:
    __aicore__ inline KernelConv3dMinSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                const optiling::Conv3dMinSoftmaxCustomTilingData &t)
    {
        xGm_ = reinterpret_cast<__gm__ float*>(x);
        wGm_ = reinterpret_cast<__gm__ float*>(w);
        bGm_ = reinterpret_cast<__gm__ float*>(b);
        yGm_ = reinterpret_cast<__gm__ float*>(y);

        N_ = t.N();
        Cin_ = t.Cin();
        Din_ = t.Din();
        Hin_ = t.Hin();
        Win_ = t.Win();

        Cout_ = t.Cout();
        Kd_ = t.Kd();
        Kh_ = t.Kh();
        Kw_ = t.Kw();

        Dout_ = t.Dout();
        Hout_ = t.Hout();
        Wout_ = t.Wout();

        sD_ = t.strideD();
        sH_ = t.strideH();
        sW_ = t.strideW();

        pD_ = t.padD();
        pH_ = t.padH();
        pW_ = t.padW();

        dD_ = t.dilationD();
        dH_ = t.dilationH();
        dW_ = t.dilationW();

        totalSites_ = static_cast<uint64_t>(N_) * Hout_ * Wout_;
        tileSites_ = t.tileSites();
    }

    __aicore__ inline void Process()
    {
        if (totalSites_ == 0 || Cout_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t strideSites = static_cast<uint64_t>(blockNum) * tileSites_;
        uint64_t baseSite = static_cast<uint64_t>(blockIdx) * tileSites_;

        while (baseSite < totalSites_) {
            uint64_t endSite = baseSite + tileSites_;
            if (endSite > totalSites_) endSite = totalSites_;
            for (uint64_t s = baseSite; s < endSite; ++s) {
                uint32_t n, ho, wo;
                DecodeSite(s, n, ho, wo);

                // First pass: find max across channels (for softmax stability)
                float maxVal = -AscendC::INFINITY_FLOAT;
                for (uint32_t co = 0; co < Cout_; ++co) {
                    float mv = ConvMinOverDepth(n, ho, wo, co);
                    if (mv > maxVal) maxVal = mv;
                }

                // Second pass: compute denominator (sum of exp)
                float sumExp = 0.f;
                for (uint32_t co = 0; co < Cout_; ++co) {
                    float mv = ConvMinOverDepth(n, ho, wo, co);
                    float e = AscendC::Exp(mv - maxVal);
                    sumExp += e;
                }

                // Third pass: write normalized softmax values
                for (uint32_t co = 0; co < Cout_; ++co) {
                    float mv = ConvMinOverDepth(n, ho, wo, co);
                    float outv = AscendC::Exp(mv - maxVal) / sumExp;
                    uint64_t yoff = YOffset(n, co, ho, wo);
                    yGm_[yoff] = outv;
                }
            }
            baseSite += strideSites;
        }
    }

private:
    __aicore__ inline void DecodeSite(uint64_t s, uint32_t &n, uint32_t &ho, uint32_t &wo) const
    {
        wo = static_cast<uint32_t>(s % Wout_);
        s /= Wout_;
        ho = static_cast<uint32_t>(s % Hout_);
        s /= Hout_;
        n = static_cast<uint32_t>(s);
    }

    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        return (((static_cast<uint64_t>(n) * Cin_ + c) * Din_ + d) * Hin_ + h) * Win_ + w;
    }

    __aicore__ inline uint64_t WOffset(uint32_t co, uint32_t ci, uint32_t kd, uint32_t kh, uint32_t kw) const
    {
        return ((((static_cast<uint64_t>(co) * Cin_ + ci) * Kd_ + kd) * Kh_ + kh) * Kw_ + kw);
    }

    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t co, uint32_t h, uint32_t w) const
    {
        return (((static_cast<uint64_t>(n) * Cout_ + co) * Hout_ + h) * Wout_ + w);
    }

    __aicore__ inline float ConvAt(uint32_t n, uint32_t d0, uint32_t h0, uint32_t w0, uint32_t co) const
    {
        float acc = 0.f;
        for (uint32_t ci = 0; ci < Cin_; ++ci) {
            for (uint32_t kd = 0; kd < Kd_; ++kd) {
                const int32_t di = static_cast<int32_t>(d0) + static_cast<int32_t>(kd) * static_cast<int32_t>(dD_) - static_cast<int32_t>(pD_);
                if (di < 0 || di >= static_cast<int32_t>(Din_)) continue;
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const int32_t hi = static_cast<int32_t>(h0) + static_cast<int32_t>(kh) * static_cast<int32_t>(dH_) - static_cast<int32_t>(pH_);
                    if (hi < 0 || hi >= static_cast<int32_t>(Hin_)) continue;
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const int32_t wi = static_cast<int32_t>(w0) + static_cast<int32_t>(kw) * static_cast<int32_t>(dW_) - static_cast<int32_t>(pW_);
                        if (wi < 0 || wi >= static_cast<int32_t>(Win_)) continue;
                        const uint64_t xoff = XOffset(n, ci, static_cast<uint32_t>(di), static_cast<uint32_t>(hi), static_cast<uint32_t>(wi));
                        const uint64_t woff = WOffset(co, ci, kd, kh, kw);
                        acc += xGm_[xoff] * wGm_[woff];
                    }
                }
            }
        }
        acc += bGm_[co];
        return acc;
    }

    __aicore__ inline float ConvMinOverDepth(uint32_t n, uint32_t ho, uint32_t wo, uint32_t co) const
    {
        float minv = AscendC::INFINITY_FLOAT;
        for (uint32_t dof = 0; dof < Dout_; ++dof) {
            const uint32_t d0 = dof * sD_;
            const uint32_t h0 = ho * sH_;
            const uint32_t w0 = wo * sW_;
            const float v = ConvAt(n, d0, h0, w0, co);
            if (v < minv) minv = v;
        }
        return minv;
    }

private:
    __gm__ float *xGm_{nullptr};
    __gm__ float *wGm_{nullptr};
    __gm__ float *bGm_{nullptr};
    __gm__ float *yGm_{nullptr};

    uint32_t N_{0}, Cin_{0}, Din_{0}, Hin_{0}, Win_{0};
    uint32_t Cout_{0}, Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};
    uint32_t sD_{1}, sH_{1}, sW_{1};
    uint32_t pD_{0}, pH_{0}, pW_{0};
    uint32_t dD_{1}, dH_{1}, dW_{1};

    uint64_t totalSites_{0};
    uint32_t tileSites_{1};
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMinSoftmaxCustom op;
    op.Init(x, w, b, y, tiling_data);
    op.Process();
}
