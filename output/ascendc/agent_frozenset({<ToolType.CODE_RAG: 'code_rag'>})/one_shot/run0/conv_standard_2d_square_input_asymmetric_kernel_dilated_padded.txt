{
    return (a + b - 1u) / b;

{
    return (a + b - 1ull) / b;

    const int pad_right  = attrs->GetAttr<int>("pad_right");
    const int dil_h      = attrs->GetAttr<int>("dilation_h");
    const int dil_w      = attrs->GetAttr<int>("dilation_w");

    const int64_t effKw = dil_w * (Kw - 1) + 1;

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelConv2DDirectScalar {
public:
    __aicore__ inline KernelConv2DDirectScalar() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                const optiling::ConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustomTilingData &t)
    {
        n_ = t.n();
        c_ = t.c();
        h_ = t.h();
        w_ = t.w();
        k_ = t.k();
        kh_ = t.kh();
        kw_ = t.kw();
        oh_ = t.oh();
        ow_ = t.ow();

        strideH_ = t.strideH();
        strideW_ = t.strideW();
        padTop_ = t.padTop();
        padBottom_ = t.padBottom();
        padLeft_ = t.padLeft();
        padRight_ = t.padRight();
        dilH_ = t.dilH();
        dilW_ = t.dilW();
        groups_ = t.groups();

        totalPos_ = t.totalPos();
        spanPerBlock_ = t.spanPerBlock();

        // GM tensors
        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * c_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)k_ * c_ * kh_ * kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * k_ * oh_ * ow_);

        // Init pipe and local buffers
        // For simplicity we use length-1 locals for scalar loads and accumulation
        pipe_.InitBuffer(xElmQ_, 1, sizeof(float));        // 1 float
        pipe_.InitBuffer(wElmQ_, 1, sizeof(float));        // 1 float
        pipe_.InitBuffer(accElmQ_, 1, sizeof(float));      // 1 float
        pipe_.InitBuffer(tmpElmQ_, 1, sizeof(float));      // 1 float
    }

    __aicore__ inline void Process()
    {
        if (groups_ != 1 || totalPos_ == 0) { return; }

        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());

        uint64_t startPos = (uint64_t)blockIdx * spanPerBlock_;
        if (startPos >= totalPos_) return;
        uint64_t endPos = startPos + spanPerBlock_;
        if (endPos > totalPos_) endPos = totalPos_;

        // Iterate over output channels one-by-one (ocTile = 1)
        for (uint32_t oc = 0; oc < k_; ++oc) {
            // For each assigned position
            for (uint64_t pos = startPos; pos < endPos; ++pos) {
                uint64_t tmp = pos;
                uint32_t ow = static_cast<uint32_t>(tmp % this->ow_);
                tmp /= this->ow_;
                uint32_t oh = static_cast<uint32_t>(tmp % this->oh_);
                uint32_t n  = static_cast<uint32_t>(tmp / this->oh_);

                // acc = 0
                LocalTensor<float> acc = accElmQ_.AllocTensor<float>();
                // Fill zero
                SetValue(acc, 0.0f, 1);

                // Convolution sum
                for (uint32_t ic = 0; ic < c_; ++ic) {
                    for (uint32_t kh = 0; kh < kh_; ++kh) {
                        int32_t ih_base = static_cast<int32_t>(oh) * static_cast<int32_t>(strideH_) - static_cast<int32_t>(padTop_) +
                                          static_cast<int32_t>(kh) * static_cast<int32_t>(dilH_);
                        if (ih_base < 0 || ih_base >= static_cast<int32_t>(h_)) {
                            continue;
                        }
                        for (uint32_t kw = 0; kw < this->kw_; ++kw) {
                            int32_t iw_val = static_cast<int32_t>(ow) * static_cast<int32_t>(strideW_) - static_cast<int32_t>(padLeft_) +
                                             static_cast<int32_t>(kw) * static_cast<int32_t>(dilW_);
                            if (iw_val < 0 || iw_val >= static_cast<int32_t>(w_)) {
                                continue;
                            }

                            uint64_t xIndex = (((uint64_t)n * c_ + ic) * h_ + static_cast<uint32_t>(ih_base)) * w_ + static_cast<uint32_t>(iw_val);
                            // Load x scalar
                            LocalTensor<float> xElm = xElmQ_.AllocTensor<float>();
                            DataCopy(xElm, xGm_[xIndex], 1);

                            // Load weight scalar for this (oc, ic, kh, kw)
                            uint64_t wIndex = ((((uint64_t)oc * c_ + ic) * kh_) + kh) * this->kw_ + kw;
                            LocalTensor<float> wElm = wElmQ_.AllocTensor<float>();
                            DataCopy(wElm, wGm_[wIndex], 1);

                            // temp = w * x (length 1), acc += temp
                            LocalTensor<float> tmpElm = tmpElmQ_.AllocTensor<float>();
                            Mul(tmpElm, wElm, xElm, 1);
                            Add(acc, acc, tmpElm, 1);

                            tmpElmQ_.FreeTensor(tmpElm);
                            wElmQ_.FreeTensor(wElm);
                            xElmQ_.FreeTensor(xElm);
                        }
                    }
                }

                // Store to y
                uint64_t yIndex = (((uint64_t)n * k_ + oc) * oh_ + oh) * this->ow_ + ow;
                DataCopy(yGm_[yIndex], acc, 1);
                accElmQ_.FreeTensor(acc);
            }
        }
    }

private:
    // Pipe and queues
    TPipe pipe_;
    TQue<TPosition::VECCALC, 1> xElmQ_;
    TQue<TPosition::VECCALC, 1> wElmQ_;
    TQue<TPosition::VECCALC, 1> accElmQ_;
    TQue<TPosition::VECCALC, 1> tmpElmQ_;

    // GM tensors
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    // Shapes and params
    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t k_{0}, kh_{0}, kw_{0};
    uint32_t oh_{0}, ow_{0};

    uint32_t strideH_{1}, strideW_{1};
    uint32_t padTop_{0}, padBottom_{0}, padLeft_{0}, padRight_{0};
    uint32_t dilH_{1}, dilW_{1};
    uint32_t groups_{1};

    uint64_t totalPos_{0};
    uint64_t spanPerBlock_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2DDirectScalar op;
    op.Init(x, weight, y, tiling_data);
    op.Process();
}
