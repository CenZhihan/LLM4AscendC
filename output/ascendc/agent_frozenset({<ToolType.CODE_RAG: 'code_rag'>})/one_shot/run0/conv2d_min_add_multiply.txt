kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv2dMinAddMultiplyCustom {
public:
    __aicore__ inline KernelConv2dMinAddMultiplyCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias,
                                GM_ADDR post_bias, GM_ADDR constant_value,
                                GM_ADDR scaling_factor, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t h, uint32_t w, uint32_t k,
                                uint32_t ho, uint32_t wo, uint32_t tileW,
                                uint32_t hasBiasFlag)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        h_ = h; w_ = w; k_ = k;
        ho_ = ho; wo_ = wo; tileW_ = tileW;
        hasBias_ = (hasBiasFlag != 0u);

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cin_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, cout_ * cin_ * k_ * k_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * cout_ * ho_ * wo_);

        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)conv_bias, cout_);
        }
        pbGm_.SetGlobalBuffer((__gm__ float*)post_bias, cout_ * 1 * 1);
        cvGm_.SetGlobalBuffer((__gm__ float*)constant_value, 1);
        sfGm_.SetGlobalBuffer((__gm__ float*)scaling_factor, 1);

        pipe_.InitBuffer(bufIn_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(bufTmp_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(bufOut_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(bufWvec_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(bufScalar_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (ho_ == 0 || wo_ == 0 || tileW_ == 0) return;

        float cv = LoadScalar(cvGm_, 0);
        float sf = LoadScalar(sfGm_, 0);

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalTasks = n_ * cout_ * ho_;
        for (uint32_t task = blockIdx; task < totalTasks; task += blockNum) {
            const uint32_t nIdx = task / (cout_ * ho_);
            const uint32_t rem1 = task % (cout_ * ho_);
            const uint32_t co = rem1 / ho_;
            const uint32_t hoIdx = rem1 % ho_;

            const float bias = hasBias_ ? LoadScalar(bGm_, co) : 0.0f;
            const float postb = LoadPostBias(co); // pb[co,1,1]

            for (uint32_t wStart = 0; wStart < wo_; wStart += tileW_) {
                uint32_t len = wo_ - wStart;
                if (len > tileW_) len = tileW_;

                AscendC::LocalTensor<float> yBuf = bufOut_.AllocTensor<float>();
                bool firstAccum = true;

                // Accumulate convolution
                for (uint32_t ci = 0; ci < cin_; ++ci) {
                    for (uint32_t ky = 0; ky < k_; ++ky) {
                        for (uint32_t kx = 0; kx < k_; ++kx) {
                            // Load x slice [len] contiguous along width
                            AscendC::LocalTensor<float> xBuf = bufIn_.AllocTensor<float>();
                            const uint32_t xBase = IndexX(nIdx, ci, hoIdx + ky, wStart + kx);
                            AscendC::DataCopy(xBuf, xGm_[xBase], len);

                            // Load weight scalar and duplicate to vector
                            const uint32_t wBase = IndexW(co, ci, ky, kx);
                            float wVal = LoadScalar(wGm_, wBase);
                            AscendC::LocalTensor<float> wVec = bufWvec_.AllocTensor<float>();
                            AscendC::Dup(wVec, wVal, len);

                            // temp = xBuf * wVec
                            AscendC::LocalTensor<float> tmp = bufTmp_.AllocTensor<float>();
                            AscendC::Mul(tmp, xBuf, wVec, len);

                            if (firstAccum) {
                                AscendC::DataCopy(yBuf, tmp, len);
                                firstAccum = false;
                            } else {
                                AscendC::Add(yBuf, yBuf, tmp, len);
                            }

                            bufTmp_.FreeTensor(tmp);
                            bufWvec_.FreeTensor(wVec);
                            bufIn_.FreeTensor(xBuf);
                        }
                    }
                }

                // Add conv bias if any
                if (hasBias_ && bias != 0.0f) {
                    AscendC::Adds(yBuf, yBuf, bias, len);
                }

                // Clip to constant_value
                AscendC::Mins(yBuf, yBuf, cv, len);

                // Add post_bias per output channel
                if (postb != 0.0f) {
                    AscendC::Adds(yBuf, yBuf, postb, len);
                }

                // Scale
                AscendC::Muls(yBuf, yBuf, sf, len);

                // Store back
                const uint32_t yBase = IndexY(nIdx, co, hoIdx, wStart);
                AscendC::DataCopy(yGm_[yBase], yBuf, len);
                bufOut_.FreeTensor(yBuf);
            }
        }
    }

private:
    __aicore__ inline uint32_t IndexX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((n * cin_ + c) * h_ + h) * w_ + w);
    }

    __aicore__ inline uint32_t IndexW(uint32_t co, uint32_t ci, uint32_t ky, uint32_t kx) const
    {
        return (((co * cin_ + ci) * k_ + ky) * k_ + kx);
    }

    __aicore__ inline uint32_t IndexY(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((n * cout_ + c) * ho_ + h) * wo_ + w);
    }

    __aicore__ inline float LoadScalar(const AscendC::GlobalTensor<float>& gm, uint32_t idx)
    {
        AscendC::LocalTensor<float> t = bufScalar_.AllocTensor<float>();
        AscendC::DataCopy(t, gm[idx], 1);
        float v = t.GetValue(0);
        bufScalar_.FreeTensor(t);
        return v;
    }

    __aicore__ inline float LoadPostBias(uint32_t co)
    {
        // post_bias is [Cout,1,1] laid out as Cout elements
        return LoadScalar(pbGm_, co);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bufIn_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bufTmp_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> bufOut_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufWvec_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufScalar_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> pbGm_;
    AscendC::GlobalTensor<float> cvGm_;
    AscendC::GlobalTensor<float> sfGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t h_{0}, w_{0}, k_{0};
    uint32_t ho_{0}, wo_{0}, tileW_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR post_bias,
    GM_ADDR constant_value, GM_ADDR scaling_factor, GM_ADDR y,
    GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinAddMultiplyCustom op;
    op.Init(x, weight, conv_bias, post_bias, constant_value, scaling_factor, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.h, tiling_data.w, tiling_data.k,
            tiling_data.ho, tiling_data.wo, tiling_data.tileW,
            tiling_data.hasBiasFlag);
    op.Process();
}
