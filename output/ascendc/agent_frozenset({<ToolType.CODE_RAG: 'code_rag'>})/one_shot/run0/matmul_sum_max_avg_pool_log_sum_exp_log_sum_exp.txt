static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // approximate per-core usable UB
static constexpr uint32_t ALIGN_ELEMS    = 256;           // favor aligned DMA
static constexpr uint32_t MIN_KTILE      = 256;
static constexpr uint32_t MAX_BLOCKS     = 24;
static constexpr uint32_t MIN_BLOCKS     = 1;

// Simple helpers
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulSumMaxAvgPoolLogSumExpLogSumExpCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0); // [batch, in]
    const gert::Shape *w_shape = context->GetInputShape(1); // [out, in]
    const gert::Shape *b_shape = context->GetInputShape(2); // [out]

    if (x_shape->GetOriginShape().GetDimNum() != 2 ||
        w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t batch   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t inFeat  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t outFeat = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));

    // UB model for this kernel per-core:
    // - wRowBuf:    kTile floats
    // - wSumBuf:    kTile floats
    // - xBuf:       kTile floats
    // - tmpBuf:     kTile floats
    // - scalarBuf:  1 float (round up into kTile alignment negligible)
    // Additionally we iterate w along jTile rows; we reuse wRowBuf per row.
    //
    // Memory ~ (4 * kTile + 1) * 4 bytes <= UB_BUDGET_BYTES
    uint32_t kTile = UB_BUDGET_BYTES / (4u * sizeof(float));
    kTile = AlignDown(kTile, ALIGN_ELEMS);
    if (kTile > inFeat) {
        kTile = AlignDown(inFeat, ALIGN_ELEMS);
        if (kTile == 0) kTile = inFeat;
    }
    if (kTile < MIN_KTILE) {
        kTile = (inFeat >= MIN_KTILE) ? MIN_KTILE : inFeat;
    }

    // jTile controls the number of W rows accumulated per inner loop step.
    // We use a small constant to save UB and lessen DMA granularity concerns.
    const uint32_t jTile = 16;

    // Blocks: distribute rows across cores by simple grid-stride.
    uint32_t blockDim = CeilDiv(batch, 1u); // one row at a time per iteration
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_inFeat(inFeat);
    tiling.set_outFeat(outFeat);
    tiling.set_kTile(kTile);
    tiling.set_jTile(jTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUF_NUM = 2;

class KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom {
public:
    __aicore__ inline KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batch, uint32_t inFeat, uint32_t outFeat,
                                uint32_t kTile, uint32_t jTile)
    {
        batch_ = batch;
        inFeat_ = inFeat;
        outFeat_ = outFeat;
        kTile_ = kTile;
        jTile_ = jTile;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(batch_) * inFeat_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(outFeat_) * inFeat_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, outFeat_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch_);

        // Buffers:
        // - xTile:    kTile
        // - wRow:     kTile
        // - wSum:     kTile
        // - tmp:      kTile
        // - scalar:   1
        pipe_.InitBuffer(xTileQ_, BUF_NUM, kTile_ * sizeof(float));
        pipe_.InitBuffer(wRowQ_, 1, kTile_ * sizeof(float));
        pipe_.InitBuffer(wSumQ_, 1, kTile_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, kTile_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, 2, sizeof(float)); // for reduce sums and y accumulate
        pipe_.InitBuffer(yInQ_, 1, sizeof(float));
        pipe_.InitBuffer(yOutQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || inFeat_ == 0 || outFeat_ == 0 || kTile_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Precompute bsum as a 1-element local tensor: bsum = sum_j b[j]
        bool bsumInit = false;
        AscendC::LocalTensor<float> bsumLocal;
        for (uint32_t boff = 0; boff < outFeat_; ) {
            uint32_t len = outFeat_ - boff;
            if (len > kTile_) len = kTile_;
            AscendC::LocalTensor<float> bChunk = scalarQ_.AllocTensor<float>(); // reuse scalarQ_ is too small
            // Instead, use xTileQ_ for loading b chunk
            bChunk = xTileQ_.AllocTensor<float>();
            AscendC::DataCopy(bChunk, bGm_[boff], len);
            if (!bsumInit) {
                bsumLocal = scalarQ_.AllocTensor<float>();
                AscendC::ReduceSum(bsumLocal, bChunk, len);
                bsumInit = true;
            } else {
                AscendC::LocalTensor<float> tmpSum = scalarQ_.AllocTensor<float>();
                AscendC::ReduceSum(tmpSum, bChunk, len);
                AscendC::Add(bsumLocal, bsumLocal, tmpSum, 1);
                scalarQ_.FreeTensor(tmpSum);
            }
            xTileQ_.FreeTensor(bChunk);
            boff += len;
        }
        if (!bsumInit) {
            bsumInit = true;
            bsumLocal = scalarQ_.AllocTensor<float>();
            // set zero if outFeat_ == 0 (shouldn't happen as checked)
            AscendC::Muls(bsumLocal, bsumLocal, 0.0f, 1);
        }

        // Iterate k tiles and accumulate contributions into y (in GM) per row
        for (uint32_t k0 = 0; k0 < inFeat_; k0 += kTile_) {
            uint32_t kLen = inFeat_ - k0;
            if (kLen > kTile_) kLen = kTile_;

            // Compute wsum over outFeat for this k-tile: wsum[k] = sum_j w[j, k0 + k]
            bool wsumInit = false;
            AscendC::LocalTensor<float> wSum = wSumQ_.AllocTensor<float>();
            for (uint32_t j0 = 0; j0 < outFeat_; j0 += jTile_) {
                uint32_t jt = outFeat_ - j0;
                if (jt > jTile_) jt = jTile_;
                // Accumulate over jt rows
                for (uint32_t t = 0; t < jt; ++t) {
                    AscendC::LocalTensor<float> wRow = wRowQ_.AllocTensor<float>();
                    const uint64_t rowOff = static_cast<uint64_t>(j0 + t) * inFeat_ + k0;
                    AscendC::DataCopy(wRow, wGm_[rowOff], kLen);
                    if (!wsumInit) {
                        // First row: wSum = wRow
                        AscendC::DataCopy(wSum, wRow, kLen);
                        wsumInit = true;
                    } else {
                        AscendC::Add(wSum, wSum, wRow, kLen);
                    }
                    wRowQ_.FreeTensor(wRow);
                }
            }
            if (!wsumInit) {
                // outFeat_ == 0, shouldn't happen
                AscendC::Muls(wSum, wSum, 0.0f, kLen);
            }

            // Grid-stride over rows
            for (uint32_t row = blockIdx; row < batch_; row += blockNum) {
                // x segment
                AscendC::LocalTensor<float> xTile = xTileQ_.AllocTensor<float>();
                const uint64_t xoff = static_cast<uint64_t>(row) * inFeat_ + k0;
                AscendC::DataCopy(xTile, xGm_[xoff], kLen);

                // tmp = xTile * wSum
                AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                AscendC::Muls(tmp, xTile, 1.0f, kLen);           // tmp = xTile
                AscendC::Mul(tmp, tmp, wSum, kLen);              // tmp *= wSum

                // sLocal = reduce_sum(tmp)
                AscendC::LocalTensor<float> sLocal = scalarQ_.AllocTensor<float>();
                AscendC::ReduceSum(sLocal, tmp, kLen);

                // Accumulate into y[row]
                if (k0 == 0) {
                    // first k-tile: y = sLocal
                    AscendC::LocalTensor<float> yLocal = yOutQ_.AllocTensor<float>();
                    AscendC::DataCopy(yLocal, sLocal, 1);
                    AscendC::DataCopy(yGm_[row], yLocal, 1);
                    yOutQ_.FreeTensor(yLocal);
                } else {
                    // y += sLocal
                    AscendC::LocalTensor<float> yPrev = yInQ_.AllocTensor<float>();
                    AscendC::DataCopy(yPrev, yGm_[row], 1);
                    AscendC::Add(yPrev, yPrev, sLocal, 1);
                    AscendC::DataCopy(yGm_[row], yPrev, 1);
                    yInQ_.FreeTensor(yPrev);
                }

                // free locals
                scalarQ_.FreeTensor(sLocal);
                tmpQ_.FreeTensor(tmp);
                xTileQ_.FreeTensor(xTile);
            }

            wSumQ_.FreeTensor(wSum);
        }

        // Add bsum to every y[row]
        for (uint32_t row = blockIdx; row < batch_; row += blockNum) {
            AscendC::LocalTensor<float> yVal = yInQ_.AllocTensor<float>();
            AscendC::DataCopy(yVal, yGm_[row], 1);
            AscendC::Add(yVal, yVal, bsumLocal, 1);
            AscendC::DataCopy(yGm_[row], yVal, 1);
            yInQ_.FreeTensor(yVal);
        }

        // free bsumLocal
        scalarQ_.FreeTensor(bsumLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_NUM> xTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>       wRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>     wSumQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>     tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2>     scalarQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>       yInQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1>      yOutQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batch_ {0};
    uint32_t inFeat_ {0};
    uint32_t outFeat_ {0};
    uint32_t kTile_ {0};
    uint32_t jTile_ {0};
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom op;
    op.Init(x, w, b, y,
            tiling_data.batch,
            tiling_data.inFeat,
            tiling_data.outFeat,
            tiling_data.kTile,
            tiling_data.jTile);
    op.Process();
}
