#include "kernel_operator.h"

class KernelConvTransposed3dAsymInAsymKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymInAsymKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t OD, uint32_t OH, uint32_t OW,
                                uint64_t totalOut, uint32_t tileLen)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        D_ = D; H_ = H; W_ = W;
        kD_ = kD; kH_ = kH; kW_ = kW;
        OD_ = OD; OH_ = OH; OW_ = OW;
        totalOut_ = totalOut;
        tileLen_ = tileLen > 0 ? tileLen : 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(N_ * Cin_ * D_ * H_ * W_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(Cin_ * Cout_ * kD_ * kH_ * kW_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(N_ * Cout_ * OD_ * OH_ * OW_));

        pipe_.InitBuffer(xInQ_, 2, sizeof(float));   // scalar buffers
        pipe_.InitBuffer(wInQ_, 2, sizeof(float));
        pipe_.InitBuffer(tmpQ_, 2, sizeof(float));
        pipe_.InitBuffer(accQ_, 1, sizeof(float));
        pipe_.InitBuffer(outQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint64_t strideElems = static_cast<uint64_t>(blockNum) * static_cast<uint64_t>(tileLen_);
        uint64_t base = static_cast<uint64_t>(blockIdx) * static_cast<uint64_t>(tileLen_);
        if (base >= totalOut_) return;

        while (base < totalOut_) {
            uint32_t curLen = tileLen_;
            if (base + curLen > totalOut_) {
                curLen = static_cast<uint32_t>(totalOut_ - base);
            }
            ComputeRange(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t ci, uint32_t iz, uint32_t iy, uint32_t ix) const
    {
        // [N, Cin, D, H, W]
        return (((static_cast<uint64_t>(n) * Cin_ + ci) * D_ + iz) * H_ + iy) * W_ + ix;
    }

    __aicore__ inline uint64_t OffsetW(uint32_t ci, uint32_t co, uint32_t kz, uint32_t ky, uint32_t kx) const
    {
        // [Cin, Cout, kD, kH, kW]
        return ((((static_cast<uint64_t>(ci) * Cout_ + co) * kD_ + kz) * kH_ + ky) * kW_ + kx);
    }

    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t co, uint32_t oz, uint32_t oy, uint32_t ox) const
    {
        // [N, Cout, OD, OH, OW]
        return (((static_cast<uint64_t>(n) * Cout_ + co) * OD_ + oz) * OH_ + oy) * OW_ + ox;
    }

    __aicore__ inline void DecodeOutIndex(uint64_t idx,
                                          uint32_t &n, uint32_t &co,
                                          uint32_t &oz, uint32_t &oy, uint32_t &ox) const
    {
        uint64_t t = idx;
        ox = static_cast<uint32_t>(t % OW_); t /= OW_;
        oy = static_cast<uint32_t>(t % OH_); t /= OH_;
        oz = static_cast<uint32_t>(t % OD_); t /= OD_;
        co = static_cast<uint32_t>(t % Cout_); t /= Cout_;
        n  = static_cast<uint32_t>(t);
    }

    __aicore__ inline void ComputeRange(uint64_t base, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            uint64_t outIdx = base + i;
            uint32_t n, co, oz, oy, ox;
            DecodeOutIndex(outIdx, n, co, oz, oy, ox);

            // acc = 0.0
            AscendC::LocalTensor<float> accLocal = accQ_.AllocTensor<float>();
            AscendC::Dup(accLocal, 0.0f, 1);

            // y[n, co, oz, oy, ox] = sum_{ci,kz,ky,kx} x[n,ci, iz,iy,ix] * w[ci,co,kz,ky,kx]
            // where iz = oz - kz, iy = oy - ky, ix = ox - kx; only if indices in-bound
            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kz = 0; kz < kD_; ++kz) {
                    int32_t iz = static_cast<int32_t>(oz) - static_cast<int32_t>(kz);
                    if (iz < 0 || iz >= static_cast<int32_t>(D_)) continue;
                    for (uint32_t ky = 0; ky < kH_; ++ky) {
                        int32_t iy = static_cast<int32_t>(oy) - static_cast<int32_t>(ky);
                        if (iy < 0 || iy >= static_cast<int32_t>(H_)) continue;
                        for (uint32_t kx = 0; kx < kW_; ++kx) {
                            int32_t ix = static_cast<int32_t>(ox) - static_cast<int32_t>(kx);
                            if (ix < 0 || ix >= static_cast<int32_t>(W_)) continue;

                            const uint64_t xOff = OffsetX(n, ci,
                                                          static_cast<uint32_t>(iz),
                                                          static_cast<uint32_t>(iy),
                                                          static_cast<uint32_t>(ix));
                            const uint64_t wOff = OffsetW(ci, co, kz, ky, kx);

                            AscendC::LocalTensor<float> xLocal = xInQ_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[static_cast<uint32_t>(xOff)], 1);
                            AscendC::LocalTensor<float> wLocal = wInQ_.AllocTensor<float>();
                            AscendC::DataCopy(wLocal, wGm_[static_cast<uint32_t>(wOff)], 1);

                            AscendC::LocalTensor<float> prodLocal = tmpQ_.AllocTensor<float>();
                            AscendC::Mul(prodLocal, xLocal, wLocal, 1);
                            AscendC::Add(accLocal, accLocal, prodLocal, 1);

                            tmpQ_.FreeTensor(prodLocal);
                            wInQ_.FreeTensor(wLocal);
                            xInQ_.FreeTensor(xLocal);
                        }
                    }
                }
            }

            // store to y
            const uint64_t yOff = OffsetY(n, co, oz, oy, ox);
            AscendC::DataCopy(yGm_[static_cast<uint32_t>(yOff)], accLocal, 1);
            accQ_.FreeTensor(accLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> xInQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> wInQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t D_{0}, H_{0}, W_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};
    uint32_t OD_{0}, OH_{0}, OW_{0};
    uint64_t totalOut_{0};
    uint32_t tileLen_{1};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x,
                                                                                                   GM_ADDR weight,
                                                                                                   GM_ADDR y,
                                                                                                   GM_ADDR workspace,
                                                                                                   GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dAsymInAsymKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.OD, tiling_data.OH, tiling_data.OW,
            tiling_data.totalOut, tiling_data.tileLen);
    op.Process();
}
