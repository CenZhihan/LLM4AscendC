kernel_src="""
#include "kernel_operator.h"

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0,
                                GM_ADDR w_i2h, GM_ADDR b_i2h,
                                GM_ADDR w_h2o, GM_ADDR b_h2o,
                                GM_ADDR y,
                                uint32_t T, uint32_t B, uint32_t I,
                                uint32_t H, uint32_t O)
    {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(T_)*B_*I_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, static_cast<uint64_t>(B_)*H_);

        wi2hGm_.SetGlobalBuffer((__gm__ float*)w_i2h, static_cast<uint64_t>(H_)*(I_+H_));
        bi2hGm_.SetGlobalBuffer((__gm__ float*)b_i2h, H_);

        whoGm_.SetGlobalBuffer((__gm__ float*)w_h2o, static_cast<uint64_t>(O_)*H_);
        bhoGm_.SetGlobalBuffer((__gm__ float*)b_h2o, O_);

        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(T_)*B_*O_);

        // UB buffers: vectors and rows
        pipe_.InitBuffer(xBufQ_, 1, I_ * sizeof(float));
        pipe_.InitBuffer(hBufQ_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(wRowBufQ_, 1, (I_ + H_) * sizeof(float));
        pipe_.InitBuffer(outHBufQ_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(outYBufQ_, 1, O_ * sizeof(float));
        pipe_.InitBuffer(bHBufQ_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(bYBufQ_, 1, O_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (B_ == 0 || T_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t b = blockIdx; b < B_; b += blockNum) {
            // Allocate UB tensors
            AscendC::LocalTensor<float> xUb = xBufQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> hUb = hBufQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> wRowUb = wRowBufQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> outHUb = outHBufQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> outYUb = outYBufQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> bHUb = bHBufQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> bYUb = bYBufQ_.AllocTensor<float>();

            // Load initial h and biases
            AscendC::DataCopy(hUb, h0Gm_[static_cast<uint64_t>(b) * H_], H_);
            AscendC::DataCopy(bHUb, bi2hGm_[0], H_);
            AscendC::DataCopy(bYUb, bhoGm_[0], O_);

            for (uint32_t t = 0; t < T_; ++t) {
                const uint64_t xOffset = (static_cast<uint64_t>(t) * B_ + b) * I_;
                AscendC::DataCopy(xUb, xGm_[xOffset], I_);

                // pre-activation hidden: outHUb[:] = bH + x*W_x^T + h*W_h^T
                // Initialize outHUb with bias
                // Vector copy bias to outHUb
                AscendC::DataCopy(outHUb, bHUb, H_);

                // Accumulate x * W_x^T
                // For each hidden unit row r
                for (uint32_t r = 0; r < H_; ++r) {
                    // Load full weight row: length K = I + H
                    const uint64_t wRowOff = static_cast<uint64_t>(r) * (I_ + H_);
                    AscendC::DataCopy(wRowUb, wi2hGm_[wRowOff], (I_ + H_));

                    float acc = GetScalar(outHUb, r); // start from bias

                    // Dot with x (first I elements)
                    acc += Dot(xUb, wRowUb, I_);
                    // Dot with h (last H elements)
                    acc += DotOffset(hUb, wRowUb, I_, H_);

                    SetScalar(outHUb, r, acc);
                }

                // h = tanh(outHUb)
                AscendC::Tanh(hUb, outHUb, H_);

                // y = h * W_ho^T + b_ho
                // Initialize y with bias
                AscendC::DataCopy(outYUb, bYUb, O_);
                for (uint32_t o = 0; o < O_; ++o) {
                    const uint64_t wOutRowOff = static_cast<uint64_t>(o) * H_;
                    // Reuse wRowUb buffer for output projection row
                    AscendC::DataCopy(wRowUb, whoGm_[wOutRowOff], H_);
                    float acc = GetScalar(outYUb, o);
                    acc += Dot(hUb, wRowUb, H_);
                    SetScalar(outYUb, o, acc);
                }

                // store y[t,b,:]
                const uint64_t yOffset = (static_cast<uint64_t>(t) * B_ + b) * O_;
                AscendC::DataCopy(yGm_[yOffset], outYUb, O_);
            }

            // Free UB tensors
            bYBufQ_.FreeTensor(bYUb);
            bHBufQ_.FreeTensor(bHUb);
            outYBufQ_.FreeTensor(outYUb);
            outHBufQ_.FreeTensor(outHUb);
            wRowBufQ_.FreeTensor(wRowUb);
            hBufQ_.FreeTensor(hUb);
            xBufQ_.FreeTensor(xUb);
        }
    }

private:
    // Helper: read scalar from local tensor
    __aicore__ inline float GetScalar(AscendC::LocalTensor<float> &t, uint32_t idx) {
        // Access via pointer arithmetic on local tensor
        return *(reinterpret_cast<__ub__ float*>(t.GetPhyAddr()) + idx);
    }

    __aicore__ inline void SetScalar(AscendC::LocalTensor<float> &t, uint32_t idx, float v) {
        *(reinterpret_cast<__ub__ float*>(t.GetPhyAddr()) + idx) = v;
    }

    // Dot product of two UB vectors of length len
    __aicore__ inline float Dot(AscendC::LocalTensor<float> &a,
                                AscendC::LocalTensor<float> &b,
                                uint32_t len)
    {
        __ub__ float* ap = reinterpret_cast<__ub__ float*>(a.GetPhyAddr());
        __ub__ float* bp = reinterpret_cast<__ub__ float*>(b.GetPhyAddr());
        float sum = 0.0f;
        for (uint32_t i = 0; i < len; ++i) {
            sum += ap[i] * bp[i];
        }
        return sum;
    }

    // Dot product where b is offset by off elements
    __aicore__ inline float DotOffset(AscendC::LocalTensor<float> &a,
                                      AscendC::LocalTensor<float> &b,
                                      uint32_t off,
                                      uint32_t len)
    {
        __ub__ float* ap = reinterpret_cast<__ub__ float*>(a.GetPhyAddr());
        __ub__ float* bp = reinterpret_cast<__ub__ float*>(b.GetPhyAddr()) + off;
        float sum = 0.0f;
        for (uint32_t i = 0; i < len; ++i) {
            sum += ap[i] * bp[i];
        }
        return sum;
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> hBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wRowBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> outHBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> outYBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bHBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bYBufQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> wi2hGm_;
    AscendC::GlobalTensor<float> bi2hGm_;
    AscendC::GlobalTensor<float> whoGm_;
    AscendC::GlobalTensor<float> bhoGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t T_{0}, B_{0}, I_{0}, H_{0}, O_{0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0,
                                                                GM_ADDR w_i2h, GM_ADDR b_i2h,
                                                                GM_ADDR w_h2o, GM_ADDR b_h2o,
                                                                GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I, tiling_data.H, tiling_data.O);
    op.Process();
}
