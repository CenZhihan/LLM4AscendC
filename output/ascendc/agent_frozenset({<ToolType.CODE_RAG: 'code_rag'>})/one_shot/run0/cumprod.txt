host_operator_src="""

#include "cumprod_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}

// UB usage model (contiguous inner dim):
//   inQueue: 2 * tile
//   outQueue: 2 * tile
// Total bytes ~= 8 * tile (float32)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumprodCustomTilingData tiling;

    const auto *inShape = context->GetInputShape(0);
    const auto origin = inShape->GetOriginShape();
    const uint32_t totalLength = static_cast<uint32_t>(origin.GetShapeSize());

    int64_t nDim = static_cast<int64_t>(origin.GetDimNum());
    if (nDim <= 0) {
        return ge::GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    int64_t dim = *dimPtr;
    if (dim < 0) dim += nDim;
    if (dim < 0 || dim >= nDim) {
        return ge::GRAPH_FAILED;
    }

    uint64_t innerLenU64 = static_cast<uint64_t>(origin.GetDim(dim));
    uint64_t innerStrideU64 = 1;
    for (int64_t i = dim + 1; i < nDim; ++i) {
        innerStrideU64 *= static_cast<uint64_t>(origin.GetDim(i));
    }
    const uint32_t innerLen = static_cast<uint32_t>(innerLenU64);
    const uint32_t innerStride = static_cast<uint32_t>(innerStrideU64 == 0 ? 1 : innerStrideU64);
    const uint32_t segments = (innerLen == 0) ? 0u : static_cast<uint32_t>(totalLength / innerLen);

    // tile length (elements) for contiguous path
    uint32_t maxTile = UB_BUDGET_BYTES / 8;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (innerLen < tileLen) {
        const uint32_t aligned = AlignDown(innerLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : innerLen;
    }
    if (tileLen == 0) tileLen = 1;  // robust guard

    // If strided (dim not last), choose conservative small tile to limit per-iter overhead.
    if (innerStride != 1u) {
        tileLen = 64;
        if (tileLen > innerLen) tileLen = innerLen;
    }

    uint32_t blockDim = segments;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_innerLen(innerLen);
    tiling.set_segments(segments);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_innerStride(innerStride);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumprodCustom : public OpDef {
public:
    explicit CumprodCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumprodCustom);
} // namespace ops
"""

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelCumprodCustom {
public:
    __aicore__ inline KernelCumprodCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t innerLen,
                                uint32_t segments,
                                uint32_t tileLength,
                                uint32_t innerStride)
    {
        totalLen_ = totalLength;
        innerLen_ = innerLen;
        segments_ = segments;
        tileLen_ = (tileLength == 0) ? 1u : tileLength;
        innerStride_ = (innerStride == 0) ? 1u : innerStride;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || innerLen_ == 0 || segments_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t seg = blockIdx; seg < segments_; seg += blockNum) {
            // Compute base offset of this segment and element stride inside it
            uint32_t base = 0;
            uint32_t step = innerStride_;
            if (innerStride_ == 1u) {
                base = seg * innerLen_;  // contiguous row
            } else {
                // Map flat segment id to (q, r): q over outer-before, r over tail
                // base = q * (innerLen * innerStride) + r
                const uint32_t q = seg / innerStride_;
                const uint32_t r = seg % innerStride_;
                base = q * (innerLen_ * innerStride_) + r;
            }

            float carry = 1.0f;  // running product for this segment

            if (innerStride_ == 1u) {
                // Fast contiguous path: stream tiles along the last dimension
                uint32_t pos = 0;
                while (pos < innerLen_) {
                    const uint32_t remain = innerLen_ - pos;
                    const uint32_t curLen = (remain < tileLen_) ? remain : tileLen_;

                    // Load a contiguous tile
                    CopyInContig(base + pos, curLen);

                    // Compute cumulative product within tile using scalar walk in UB
                    AscendC::LocalTensor<float> xLocal = inQueue_.DeQue<float>();
                    AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

                    for (uint32_t i = 0; i < curLen; ++i) {
                        const float xi = xLocal.GetValue(i);
                        carry = carry * xi;
                        yLocal.SetValue(i, carry);
                    }

                    outQueue_.EnQue<float>(yLocal);
                    inQueue_.FreeTensor(xLocal);

                    // Store tile
                    CopyOutContig(base + pos, curLen);

                    pos += curLen;
                }
            } else {
                // Generic strided path: element-by-element processing
                for (uint32_t i = 0; i < innerLen_; ++i) {
                    const uint32_t off = base + i * step;

                    AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[off], 1);
                    inQueue_.EnQue(xLocal);

                    AscendC::LocalTensor<float> xL = inQueue_.DeQue<float>();
                    AscendC::LocalTensor<float> yL = outQueue_.AllocTensor<float>();

                    const float xi = xL.GetValue(0);
                    carry = carry * xi;
                    yL.SetValue(0, carry);

                    outQueue_.EnQue<float>(yL);
                    inQueue_.FreeTensor(xL);

                    AscendC::LocalTensor<float> yLoc = outQueue_.DeQue<float>();
                    AscendC::DataCopy(yGm_[off], yLoc, 1);
                    outQueue_.FreeTensor(yLoc);
                }
            }
        }
    }

private:
    __aicore__ inline void CopyInContig(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueue_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOutContig(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t innerLen_ {0};
    uint32_t segments_ {0};
    uint32_t tileLen_ {1};
    uint32_t innerStride_ {1};
};

extern "C" __global__ __aicore__ void cumprod_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumprodCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.innerLen,
            tiling_data.segments,
            tiling_data.tileLength,
            tiling_data.innerStride);
    op.Process();
}
"""
