host_operator_src="""

#include "average_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * inTileLen
//   outQueueY: 2 * outTileLen
// Total floats = 2*(stride*outTileLen + kernelSize) + 2*outTileLen
// bytes = 4 * [2*(stride+1)*outTileLen + 2*kernelSize]
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // favor DMA bursts
static constexpr uint32_t HARD_CAP_OUT_ELEMS = 32768;    // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling1dCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const uint32_t *kernelSize = attrs->GetAttrPointer<uint32_t>(0);
    const uint32_t *stride     = attrs->GetAttrPointer<uint32_t>(1);
    const uint32_t *padding    = attrs->GetAttrPointer<uint32_t>(2);

    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &x_origin = x_shape->GetOriginShape();

    const uint32_t n = static_cast<uint32_t>(x_origin.GetDim(0));
    const uint32_t c = static_cast<uint32_t>(x_origin.GetDim(1));
    const uint32_t lenIn = static_cast<uint32_t>(x_origin.GetDim(2));

    const uint32_t lenOut = static_cast<uint32_t>(
        (static_cast<int64_t>(lenIn) + 2 * static_cast<int64_t>(*padding)
         - static_cast<int64_t>(*kernelSize)) / static_cast<int64_t>(*stride) + 1);

    // Derive max out tile from UB budget:
    // bytes <= UB_BUDGET_BYTES
    // 4 * [2*(stride+1)*outTile + 2*kernelSize] <= UB_BUDGET_BYTES
    uint32_t maxOutTile = 0;
    const uint32_t denom = 2u * (*stride + 1u);
    if (UB_BUDGET_BYTES > 8u * (*kernelSize)) {
        maxOutTile = (UB_BUDGET_BYTES / 4u - 2u * (*kernelSize)) / denom;
    }
    maxOutTile = AlignDown(maxOutTile, ALIGN_ELEMS);
    if (maxOutTile < ALIGN_ELEMS) maxOutTile = ALIGN_ELEMS;
    if (maxOutTile > HARD_CAP_OUT_ELEMS) maxOutTile = AlignDown(HARD_CAP_OUT_ELEMS, ALIGN_ELEMS);
    uint32_t tileOut = maxOutTile;
    if (lenOut < tileOut) {
        const uint32_t aligned = AlignDown(lenOut, ALIGN_ELEMS);
        tileOut = (aligned > 0) ? aligned : lenOut;
    }

    const uint32_t seqCount = n * c;
    uint32_t blockDim = seqCount;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_lenIn(lenIn);
    tiling.set_lenOut(lenOut);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(*kernelSize);
    tiling.set_stride(*stride);
    tiling.set_padding(*padding);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelAveragePooling1dCustom {
public:
    __aicore__ inline KernelAveragePooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t lenIn, uint32_t lenOut,
                                uint32_t tileOut,
                                uint32_t kernelSize,
                                uint32_t stride,
                                uint32_t padding)
    {
        n_ = n;
        c_ = c;
        seqCount_ = n_ * c_;
        lenIn_ = lenIn;
        lenOut_ = lenOut;
        tileOut_ = tileOut;
        k_ = kernelSize;
        s_ = stride;
        p_ = padding;
        invK_ = 1.0f / static_cast<float>(k_);

        const uint32_t totalX = seqCount_ * lenIn_;
        const uint32_t totalY = seqCount_ * lenOut_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalX);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalY);

        // Maximum input slice per output tile: s_ * tileOut_ + k_
        inTileCap_ = s_ * tileOut_ + k_;

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, inTileCap_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (seqCount_ == 0 || lenOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t seq = blockIdx; seq < seqCount_; seq += blockNum) {
            const uint32_t xBase = seq * lenIn_;
            const uint32_t yBase = seq * lenOut_;

            uint32_t outBase = 0;
            // Preload first slice
            int32_t startGlobal = static_cast<int32_t>(outBase) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);
            int32_t endGlobalEx = startGlobal + static_cast<int32_t>((tileOut_ > lenOut_ ? lenOut_ : tileOut_ - 1u)) * static_cast<int32_t>(s_) + static_cast<int32_t>(k_);

            uint32_t loadStart = static_cast<uint32_t>(AscendC::Max(0, startGlobal));
            uint32_t loadEndEx = static_cast<uint32_t>(AscendC::Min(static_cast<int32_t>(lenIn_), endGlobalEx));
            uint32_t loadLen = (loadEndEx > loadStart) ? (loadEndEx - loadStart) : 0u;

            if (loadLen > 0) {
                CopyIn(xBase + loadStart, loadLen);
            }

            while (outBase < lenOut_) {
                const uint32_t curOutLen = AscendC::Min(tileOut_, lenOut_ - outBase);

                // prefetch next slice for overlap
                const uint32_t nextOutBase = outBase + curOutLen;
                uint32_t nextLoadLen = 0u;
                uint32_t nextLoadStart = 0u;
                if (nextOutBase < lenOut_) {
                    int32_t nStartGlobal = static_cast<int32_t>(nextOutBase) * static_cast<int32_t>(s_) - static_cast<int32_t>(p_);
                    int32_t nEndGlobalEx = nStartGlobal + static_cast<int32_t>((AscendC::Min(tileOut_, lenOut_ - nextOutBase) - 1u)) * static_cast<int32_t>(s_) + static_cast<int32_t>(k_);
                    nextLoadStart = static_cast<uint32_t>(AscendC::Max(0, nStartGlobal));
                    uint32_t nextLoadEndEx = static_cast<uint32_t>(AscendC::Min(static_cast<int32_t>(lenIn_), nEndGlobalEx));
                    nextLoadLen = (nextLoadEndEx > nextLoadStart) ? (nextLoadEndEx - nextLoadStart) : 0u;
                    if (nextLoadLen > 0) {
                        CopyIn(xBase + nextLoadStart, nextLoadLen);
                    }
                }

                Compute(outBase, curOutLen, loadStart, loadLen);
                CopyOut(yBase + outBase, curOutLen);

                // rotate
                outBase = nextOutBase;
                loadStart = nextLoadStart;
                loadLen = nextLoadLen;
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t outBase, uint32_t outLen,
                                   uint32_t loadStart, uint32_t loadLen)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> xLocal;
        bool hasInput = (loadLen > 0);
        if (hasInput) {
            xLocal = inQueueX_.DeQue<float>();
        }

        // For each output in this tile, compute average over window
        for (uint32_t i = 0; i < outLen; ++i) {
            int32_t winStart = static_cast<int32_t>((outBase + i) * s_) - static_cast<int32_t>(p_);
            int32_t winEndEx = winStart + static_cast<int32_t>(k_);

            uint32_t clampedStart = static_cast<uint32_t>(AscendC::Max(0, winStart));
            uint32_t clampedEndEx = static_cast<uint32_t>(AscendC::Min(static_cast<int32_t>(lenIn_), winEndEx));

            float sum = 0.0f;
            if (hasInput && clampedEndEx > clampedStart) {
                uint32_t localStart = clampedStart - loadStart;
                uint32_t localLen = clampedEndEx - clampedStart;
                for (uint32_t j = 0; j < localLen; ++j) {
                    sum += xLocal[localStart + j];
                }
            }
            yLocal[i] = sum * invK_;
        }

        if (hasInput) {
            inQueueX_.FreeTensor(xLocal);
        }
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_ {0};
    uint32_t c_ {0};
    uint32_t seqCount_ {0};
    uint32_t lenIn_ {0};
    uint32_t lenOut_ {0};
    uint32_t tileOut_ {0};
    uint32_t inTileCap_ {0};
    uint32_t k_ {0};
    uint32_t s_ {1};
    uint32_t p_ {0};
    float invK_ {1.0f};
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling1dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.lenIn, tiling_data.lenOut,
            tiling_data.tileOut,
            tiling_data.kernelSize,
            tiling_data.stride,
            tiling_data.padding);
    op.Process();
}

static inline int64_t calc_out_len_1d(int64_t len_in, int64_t kernel_size, int64_t stride, int64_t padding) {
    return (len_in + 2 * padding - kernel_size) / stride + 1;
}

at::Tensor average_pooling_1d_custom_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding) {
    TORCH_CHECK(self.dim() == 3, \"Input must be 3D (N, C, L)\");
    const auto N = self.size(0);
    const auto C = self.size(1);
    const auto L = self.size(2);
    const auto outL = calc_out_len_1d(L, kernel_size, stride, padding);
    at::Tensor result = at::empty({N, C, outL}, self.options());
    EXEC_NPU_CMD(aclnnAveragePooling1dCustom, self, kernel_size, stride, padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("average_pooling_1d_custom", &average_pooling_1d_custom_impl_npu);
