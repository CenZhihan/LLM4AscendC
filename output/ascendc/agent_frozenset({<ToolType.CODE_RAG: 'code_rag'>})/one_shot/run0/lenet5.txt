static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Lenet5CustomTilingData tiling;

    // Expect x: [N, 1, 32, 32]
    const gert::Shape *xShape = context->GetInputShape(0);
    const auto dims = xShape->GetOriginShape().GetDims();
    if (dims.size() < 4) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N  = static_cast<uint32_t>(dims[0]);
    const uint32_t C  = static_cast<uint32_t>(dims[1]);
    const uint32_t H  = static_cast<uint32_t>(dims[2]);
    const uint32_t W  = static_cast<uint32_t>(dims[3]);

    // fc3_w: [num_classes, 84]
    const gert::Shape *fc3wShape = context->GetInputShape(9);
    const auto fc3wDims = fc3wShape->GetOriginShape().GetDims();
    if (fc3wDims.size() < 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t numClasses = static_cast<uint32_t>(fc3wDims[0]);

    // We tile only over batch for simplicity
    uint32_t tileBatch = 1;

    // Conservative blockDim
    uint32_t numTiles = (tileBatch == 0) ? 1u : CeilDiv(N, tileBatch);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = (numTiles < 8u) ? numTiles : 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_batch(N);
    tiling.set_inC(C);
    tiling.set_inH(H);
    tiling.set_inW(W);
    tiling.set_numClasses(numClasses);
    tiling.set_tileBatch(tileBatch);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelLenet5Custom {
public:
    __aicore__ inline KernelLenet5Custom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR conv1_w, GM_ADDR conv1_b,
                                GM_ADDR conv2_w, GM_ADDR conv2_b,
                                GM_ADDR fc1_w, GM_ADDR fc1_b,
                                GM_ADDR fc2_w, GM_ADDR fc2_b,
                                GM_ADDR fc3_w, GM_ADDR fc3_b,
                                GM_ADDR y,
                                uint32_t batch, uint32_t inC, uint32_t inH, uint32_t inW,
                                uint32_t numClasses, uint32_t tileBatch)
    {
        batch_ = batch;
        inC_ = inC;
        inH_ = inH;
        inW_ = inW;
        numClasses_ = numClasses;
        tileBatch_ = tileBatch;

        xGm_.SetGlobalBuffer((__gm__ float*)x, batch_ * inC_ * inH_ * inW_);
        conv1WGm_.SetGlobalBuffer((__gm__ float*)conv1_w, 6 * 1 * 5 * 5);
        conv1BGm_.SetGlobalBuffer((__gm__ float*)conv1_b, 6);
        conv2WGm_.SetGlobalBuffer((__gm__ float*)conv2_w, 16 * 6 * 5 * 5);
        conv2BGm_.SetGlobalBuffer((__gm__ float*)conv2_b, 16);
        fc1WGm_.SetGlobalBuffer((__gm__ float*)fc1_w, 120 * 400);
        fc1BGm_.SetGlobalBuffer((__gm__ float*)fc1_b, 120);
        fc2WGm_.SetGlobalBuffer((__gm__ float*)fc2_w, 84 * 120);
        fc2BGm_.SetGlobalBuffer((__gm__ float*)fc2_b, 84);
        fc3WGm_.SetGlobalBuffer((__gm__ float*)fc3_w, numClasses_ * 84);
        fc3BGm_.SetGlobalBuffer((__gm__ float*)fc3_b, numClasses_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, batch_ * numClasses_);

        // UB buffers
        // Input and intermediate
        pipe_.InitBuffer(xQueue_, BUFFER_NUM, 1 * 32 * 32 * sizeof(float));            // x: 1024
        pipe_.InitBuffer(conv1WQueue_, BUFFER_NUM, 6 * 1 * 5 * 5 * sizeof(float));     // 150
        pipe_.InitBuffer(conv1BQueue_, BUFFER_NUM, 6 * sizeof(float));                 // 6
        pipe_.InitBuffer(conv1OutQueue_, BUFFER_NUM, 6 * 28 * 28 * sizeof(float));     // 4704
        pipe_.InitBuffer(pool1Queue_, BUFFER_NUM, 6 * 14 * 14 * sizeof(float));        // 1176

        pipe_.InitBuffer(conv2WQueue_, BUFFER_NUM, 16 * 6 * 5 * 5 * sizeof(float));    // 2400
        pipe_.InitBuffer(conv2BQueue_, BUFFER_NUM, 16 * sizeof(float));                // 16
        pipe_.InitBuffer(conv2OutQueue_, BUFFER_NUM, 16 * 10 * 10 * sizeof(float));    // 1600
        pipe_.InitBuffer(pool2Queue_, BUFFER_NUM, 16 * 5 * 5 * sizeof(float));         // 400

        // FC buffers
        pipe_.InitBuffer(fc1RowQueue_, BUFFER_NUM, 400 * sizeof(float));
        pipe_.InitBuffer(fc1OutQueue_, BUFFER_NUM, 120 * sizeof(float));
        pipe_.InitBuffer(fc1BQueue_, BUFFER_NUM, 120 * sizeof(float));

        pipe_.InitBuffer(fc2RowQueue_, BUFFER_NUM, 120 * sizeof(float));
        pipe_.InitBuffer(fc2OutQueue_, BUFFER_NUM, 84 * sizeof(float));
        pipe_.InitBuffer(fc2BQueue_, BUFFER_NUM, 84 * sizeof(float));

        pipe_.InitBuffer(fc3RowQueue_, BUFFER_NUM, 84 * sizeof(float));
        pipe_.InitBuffer(fc3OutQueue_, BUFFER_NUM, /*numClasses_*/ 512 * sizeof(float)); // allocate upper bound; will use first numClasses_
        pipe_.InitBuffer(fc3BQueue_, BUFFER_NUM, 512 * sizeof(float)); // likewise
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0) { return; }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideB = blockNum * tileBatch_;
        uint32_t baseB = blockIdx * tileBatch_;
        if (baseB >= batch_) { return; }

        // Load constant weights/biases that are reused across samples
        AscendC::LocalTensor<float> w1 = conv1WQueue_.AllocTensor<float>();
        AscendC::DataCopy(w1, conv1WGm_[0], 6 * 1 * 5 * 5);
        AscendC::LocalTensor<float> b1 = conv1BQueue_.AllocTensor<float>();
        AscendC::DataCopy(b1, conv1BGm_[0], 6);

        AscendC::LocalTensor<float> w2 = conv2WQueue_.AllocTensor<float>();
        AscendC::DataCopy(w2, conv2WGm_[0], 16 * 6 * 5 * 5);
        AscendC::LocalTensor<float> b2 = conv2BQueue_.AllocTensor<float>();
        AscendC::DataCopy(b2, conv2BGm_[0], 16);

        AscendC::LocalTensor<float> bfc1 = fc1BQueue_.AllocTensor<float>();
        AscendC::DataCopy(bfc1, fc1BGm_[0], 120);
        AscendC::LocalTensor<float> bfc2 = fc2BQueue_.AllocTensor<float>();
        AscendC::DataCopy(bfc2, fc2BGm_[0], 84);
        AscendC::LocalTensor<float> bfc3 = fc3BQueue_.AllocTensor<float>();
        AscendC::DataCopy(bfc3, fc3BGm_[0], numClasses_);

        while (baseB < batch_) {
            uint32_t todo = tileBatch_;
            if (baseB + todo > batch_) { todo = batch_ - baseB; }
            for (uint32_t n = 0; n < todo; ++n) {
                ComputeOne(baseB + n, w1, b1, w2, b2, bfc1, bfc2, bfc3);
            }
            baseB += strideB;
        }

        // Free static buffers
        fc3BQueue_.FreeTensor(bfc3);
        fc2BQueue_.FreeTensor(bfc2);
        fc1BQueue_.FreeTensor(bfc1);
        conv2BQueue_.FreeTensor(b2);
        conv2WQueue_.FreeTensor(w2);
        conv1BQueue_.FreeTensor(b1);
        conv1WQueue_.FreeTensor(w1);
    }

private:
    __aicore__ inline float fmaxf(float a, float b) { return a > b ? a : b; }

    __aicore__ inline void ComputeOne(uint32_t n,
                                      AscendC::LocalTensor<float> &w1,
                                      AscendC::LocalTensor<float> &b1,
                                      AscendC::LocalTensor<float> &w2,
                                      AscendC::LocalTensor<float> &b2,
                                      AscendC::LocalTensor<float> &bfc1,
                                      AscendC::LocalTensor<float> &bfc2,
                                      AscendC::LocalTensor<float> &bfc3)
    {
        // Allocate/prepare per-sample local tensors
        AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> y1Local = conv1OutQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> p1Local = pool1Queue_.AllocTensor<float>();
        AscendC::LocalTensor<float> y2Local = conv2OutQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> p2Local = pool2Queue_.AllocTensor<float>();

        // Copy input image (N,1,32,32)
        const uint32_t imgSize = 1u * 32u * 32u;
        AscendC::DataCopy(xLocal, xGm_[n * imgSize], imgSize);

        // conv1: out [6,28,28] = conv(xLocal, w1) + b1, then ReLU
        for (int co = 0; co < 6; ++co) {
            for (int oh = 0; oh < 28; ++oh) {
                for (int ow = 0; ow < 28; ++ow) {
                    float acc = b1[co];
                    for (int kh = 0; kh < 5; ++kh) {
                        for (int kw = 0; kw < 5; ++kw) {
                            int ih = oh + kh;
                            int iw = ow + kw;
                            float xv = xLocal[ih * 32 + iw];
                            float wv = w1[co * (1 * 5 * 5) + (0 * 5 * 5) + kh * 5 + kw];
                            acc += xv * wv;
                        }
                    }
                    float relu = fmaxf(acc, 0.0f);
                    y1Local[co * (28 * 28) + oh * 28 + ow] = relu;
                }
            }
        }

        // max pool 2x2 stride2: [6,28,28] -> [6,14,14]
        for (int c = 0; c < 6; ++c) {
            for (int ph = 0; ph < 14; ++ph) {
                for (int pw = 0; pw < 14; ++pw) {
                    float m00 = y1Local[c * 28 * 28 + (2 * ph) * 28 + (2 * pw)];
                    float m01 = y1Local[c * 28 * 28 + (2 * ph) * 28 + (2 * pw + 1)];
                    float m10 = y1Local[c * 28 * 28 + (2 * ph + 1) * 28 + (2 * pw)];
                    float m11 = y1Local[c * 28 * 28 + (2 * ph + 1) * 28 + (2 * pw + 1)];
                    float m0 = fmaxf(m00, m01);
                    float m1 = fmaxf(m10, m11);
                    p1Local[c * 14 * 14 + ph * 14 + pw] = fmaxf(m0, m1);
                }
            }
        }

        // conv2: in [6,14,14], w2 [16,6,5,5] -> out [16,10,10], then ReLU
        for (int co = 0; co < 16; ++co) {
            for (int oh = 0; oh < 10; ++oh) {
                for (int ow = 0; ow < 10; ++ow) {
                    float acc = b2[co];
                    for (int ci = 0; ci < 6; ++ci) {
                        for (int kh = 0; kh < 5; ++kh) {
                            for (int kw = 0; kw < 5; ++kw) {
                                int ih = oh + kh;
                                int iw = ow + kw;
                                float xv = p1Local[ci * (14 * 14) + ih * 14 + iw];
                                float wv = w2[co * (6 * 5 * 5) + ci * (5 * 5) + kh * 5 + kw];
                                acc += xv * wv;
                            }
                        }
                    }
                    y2Local[co * (10 * 10) + oh * 10 + ow] = fmaxf(acc, 0.0f);
                }
            }
        }

        // max pool 2x2 stride2: [16,10,10] -> [16,5,5]
        for (int c = 0; c < 16; ++c) {
            for (int ph = 0; ph < 5; ++ph) {
                for (int pw = 0; pw < 5; ++pw) {
                    float m00 = y2Local[c * 10 * 10 + (2 * ph) * 10 + (2 * pw)];
                    float m01 = y2Local[c * 10 * 10 + (2 * ph) * 10 + (2 * pw + 1)];
                    float m10 = y2Local[c * 10 * 10 + (2 * ph + 1) * 10 + (2 * pw)];
                    float m11 = y2Local[c * 10 * 10 + (2 * ph + 1) * 10 + (2 * pw + 1)];
                    float m0 = fmaxf(m00, m01);
                    float m1 = fmaxf(m10, m11);
                    p2Local[c * 5 * 5 + ph * 5 + pw] = fmaxf(m0, m1);
                }
            }
        }

        // Flatten to 400
        // p2Local already laid out as [16,5,5] contiguous

        // FC1: [120,400] * 400 + b -> ReLU
        AscendC::LocalTensor<float> fc1Row = fc1RowQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> fc1Out = fc1OutQueue_.AllocTensor<float>();
        for (int i = 0; i < 120; ++i) {
            // Copy one row of fc1_w (length 400)
            AscendC::DataCopy(fc1Row, fc1WGm_[i * 400], 400);
            float sum = bfc1[i];
            float acc = sum;
            for (int k = 0; k < 400; ++k) {
                acc += fc1Row[k] * p2Local[k];
            }
            fc1Out[i] = fmaxf(acc, 0.0f);
        }

        // FC2: [84,120] * 120 + b -> ReLU
        AscendC::LocalTensor<float> fc2Row = fc2RowQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> fc2Out = fc2OutQueue_.AllocTensor<float>();
        for (int i = 0; i < 84; ++i) {
            AscendC::DataCopy(fc2Row, fc2WGm_[i * 120], 120);
            float acc = bfc2[i];
            for (int k = 0; k < 120; ++k) {
                acc += fc2Row[k] * fc1Out[k];
            }
            fc2Out[i] = fmaxf(acc, 0.0f);
        }

        // FC3: [numClasses,84] * 84 + b -> logits
        AscendC::LocalTensor<float> fc3Row = fc3RowQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> fc3Out = fc3OutQueue_.AllocTensor<float>();
        for (uint32_t i = 0; i < numClasses_; ++i) {
            AscendC::DataCopy(fc3Row, fc3WGm_[i * 84], 84);
            float acc = bfc3[i];
            for (int k = 0; k < 84; ++k) {
                acc += fc3Row[k] * fc2Out[k];
            }
            fc3Out[i] = acc;
        }

        // Write result to GM
        AscendC::DataCopy(yGm_[n * numClasses_], fc3Out, numClasses_);

        // Free locals
        fc3OutQueue_.FreeTensor(fc3Out);
        fc3RowQueue_.FreeTensor(fc3Row);
        fc2OutQueue_.FreeTensor(fc2Out);
        fc2RowQueue_.FreeTensor(fc2Row);
        fc1OutQueue_.FreeTensor(fc1Out);
        fc1RowQueue_.FreeTensor(fc1Row);

        pool2Queue_.FreeTensor(p2Local);
        conv2OutQueue_.FreeTensor(y2Local);
        pool1Queue_.FreeTensor(p1Local);
        conv1OutQueue_.FreeTensor(y1Local);
        xQueue_.FreeTensor(xLocal);
    }

private:
    AscendC::TPipe pipe_;
    // Queues
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> conv1WQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> conv1BQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> conv1OutQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> pool1Queue_;

    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> conv2WQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> conv2BQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> conv2OutQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> pool2Queue_;

    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc1RowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc1OutQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc1BQueue_;

    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc2RowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc2OutQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc2BQueue_;

    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc3RowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc3OutQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> fc3BQueue_;

    // GM tensors
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> conv1WGm_;
    AscendC::GlobalTensor<float> conv1BGm_;
    AscendC::GlobalTensor<float> conv2WGm_;
    AscendC::GlobalTensor<float> conv2BGm_;
    AscendC::GlobalTensor<float> fc1WGm_;
    AscendC::GlobalTensor<float> fc1BGm_;
    AscendC::GlobalTensor<float> fc2WGm_;
    AscendC::GlobalTensor<float> fc2BGm_;
    AscendC::GlobalTensor<float> fc3WGm_;
    AscendC::GlobalTensor<float> fc3BGm_;
    AscendC::GlobalTensor<float> yGm_;

    // Params
    uint32_t batch_{0};
    uint32_t inC_{1};
    uint32_t inH_{32};
    uint32_t inW_{32};
    uint32_t numClasses_{0};
    uint32_t tileBatch_{1};
};

extern "C" __global__ __aicore__ void lenet5_custom(GM_ADDR x, GM_ADDR conv1_w, GM_ADDR conv1_b,
                                                    GM_ADDR conv2_w, GM_ADDR conv2_b,
                                                    GM_ADDR fc1_w, GM_ADDR fc1_b,
