static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x = context->GetInputShape(0);
    const gert::Shape *w = context->GetInputShape(1);

    const int64_t N  = x->GetDim(0);
    const int64_t Ci = x->GetDim(1);
    const int64_t Di = x->GetDim(2);
    const int64_t Hi = x->GetDim(3);
    const int64_t Wi = x->GetDim(4);

    const int64_t W_Ci = w->GetDim(0);
    const int64_t Co   = w->GetDim(1);
    const int64_t Kd   = w->GetDim(2);
    const int64_t Kh   = w->GetDim(3);
    const int64_t Kw   = w->GetDim(4);

    if (W_Ci != Ci) {
        return GRAPH_FAILED;
    }

    // Attributes (defaulted)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t stride_d = attrs->GetAttrPointer<int32_t>(0) ? *(attrs->GetAttrPointer<int32_t>(0)) : 1;
    const int64_t stride_h = attrs->GetAttrPointer<int32_t>(1) ? *(attrs->GetAttrPointer<int32_t>(1)) : 1;
    const int64_t stride_w = attrs->GetAttrPointer<int32_t>(2) ? *(attrs->GetAttrPointer<int32_t>(2)) : 1;
    const int64_t pad_d    = attrs->GetAttrPointer<int32_t>(3) ? *(attrs->GetAttrPointer<int32_t>(3)) : 0;
    const int64_t pad_h    = attrs->GetAttrPointer<int32_t>(4) ? *(attrs->GetAttrPointer<int32_t>(4)) : 0;
    const int64_t pad_w    = attrs->GetAttrPointer<int32_t>(5) ? *(attrs->GetAttrPointer<int32_t>(5)) : 0;
    const int64_t dil_d    = attrs->GetAttrPointer<int32_t>(6) ? *(attrs->GetAttrPointer<int32_t>(6)) : 1;
    const int64_t dil_h    = attrs->GetAttrPointer<int32_t>(7) ? *(attrs->GetAttrPointer<int32_t>(7)) : 1;
    const int64_t dil_w    = attrs->GetAttrPointer<int32_t>(8) ? *(attrs->GetAttrPointer<int32_t>(8)) : 1;
    const int64_t out_pad_d= attrs->GetAttrPointer<int32_t>(9) ? *(attrs->GetAttrPointer<int32_t>(9)) : 0;
    const int64_t out_pad_h= attrs->GetAttrPointer<int32_t>(10)? *(attrs->GetAttrPointer<int32_t>(10)) : 0;
    const int64_t out_pad_w= attrs->GetAttrPointer<int32_t>(11)? *(attrs->GetAttrPointer<int32_t>(11)) : 0;

    const int64_t Do = (Di - 1) * stride_d - 2 * pad_d + dil_d * (Kd - 1) + out_pad_d + 1;
    const int64_t Ho = (Hi - 1) * stride_h - 2 * pad_h + dil_h * (Kh - 1) + out_pad_h + 1;
    const int64_t Wo = (Wi - 1) * stride_w - 2 * pad_w + dil_w * (Kw - 1) + out_pad_w + 1;

    gert::Shape *y = context->GetOutputShape(0);
    y->SetDimNum(5);
    y->SetDim(0, N);
    y->SetDim(1, Co);
    y->SetDim(2, Do);
    y->SetDim(3, Ho);
    y->SetDim(4, Wo);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t QBUF = 1;

class KernelConvTransposed3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t ci, uint32_t co,
                                uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t outD, uint32_t outH, uint32_t outW)
    {
        n_ = n; ci_ = ci; co_ = co;
        inD_ = inD; inH_ = inH; inW_ = inW;
        kD_ = kD; kH_ = kH; kW_ = kW;
        outD_ = outD; outH_ = outH; outW_ = outW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_)*ci_*inD_*inH_*inW_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(ci_)*co_*kD_*kH_*kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_)*co_*outD_*outH_*outW_);

        pipe_.InitBuffer(qX_, QBUF, sizeof(float));
        pipe_.InitBuffer(qW_, QBUF, sizeof(float));
        pipe_.InitBuffer(qY_, QBUF, sizeof(float));
        pipe_.InitBuffer(qTmp_, QBUF, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || ci_ == 0 || co_ == 0 || outD_ == 0 || outH_ == 0 || outW_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());
        const uint64_t pairTotal = static_cast<uint64_t>(n_) * co_;

        for (uint64_t pair = blockIdx; pair < pairTotal; pair += blockNum) {
            uint32_t nIdx = static_cast<uint32_t>(pair / co_);
            uint32_t coIdx = static_cast<uint32_t>(pair % co_);
            // Zero-out y[nIdx, coIdx, :, :, :]
            ZeroOutOutputChannel(nIdx, coIdx);
            // Accumulate contributions
            ComputePair(nIdx, coIdx);
        }
    }

private:
    __aicore__ inline uint64_t XIndex(uint32_t nIdx, uint32_t cIdx, uint32_t z, uint32_t y, uint32_t x) const
    {
        uint64_t strideW = 1ULL;
        uint64_t strideH = static_cast<uint64_t>(inW_);
        uint64_t strideD = static_cast<uint64_t>(inH_) * inW_;
        uint64_t strideC = static_cast<uint64_t>(inD_) * inH_ * inW_;
        uint64_t strideN = static_cast<uint64_t>(ci_) * strideC;
        return nIdx * strideN + cIdx * strideC + z * strideD + y * strideH + x;
    }

    __aicore__ inline uint64_t YIndex(uint32_t nIdx, uint32_t cIdx, uint32_t z, uint32_t y, uint32_t x) const
    {
        uint64_t strideW = 1ULL;
        uint64_t strideH = static_cast<uint64_t>(outW_);
        uint64_t strideD = static_cast<uint64_t>(outH_) * outW_;
        uint64_t strideC = static_cast<uint64_t>(outD_) * outH_ * outW_;
        uint64_t strideN = static_cast<uint64_t>(co_) * strideC;
        return nIdx * strideN + cIdx * strideC + z * strideD + y * strideH + x;
    }

    __aicore__ inline uint64_t WIndex(uint32_t ciIdx, uint32_t coIdx, uint32_t kd, uint32_t kh, uint32_t kw) const
    {
        uint64_t strideKw = 1ULL;
        uint64_t strideKh = static_cast<uint64_t>(kW_);
        uint64_t strideKd = static_cast<uint64_t>(kH_) * kW_;
        uint64_t strideCo = static_cast<uint64_t>(kD_) * kH_ * kW_;
        uint64_t strideCi = static_cast<uint64_t>(co_) * strideCo;
        return ciIdx * strideCi + coIdx * strideCo + kd * strideKd + kh * strideKh + kw;
    }

    __aicore__ inline void ZeroOutOutputChannel(uint32_t nIdx, uint32_t coIdx)
    {
        for (uint32_t z = 0; z < outD_; ++z) {
            for (uint32_t y = 0; y < outH_; ++y) {
                for (uint32_t x = 0; x < outW_; ++x) {
                    uint64_t yOff = YIndex(nIdx, coIdx, z, y, x);
                    LocalTensor<float> yLocal = qY_.AllocTensor<float>();
                    DataCopy(yLocal, yGm_[yOff], 1);
                    // y = y * 0.0f  -> zero
                    Muls(yLocal, yLocal, 0.0f, 1);
                    DataCopy(yGm_[yOff], yLocal, 1);
                    qY_.FreeTensor(yLocal);
                }
            }
        }
    }

    __aicore__ inline void ComputePair(uint32_t nIdx, uint32_t coIdx)
    {
        // For each input voxel and channel, scatter-add to output y[nIdx, coIdx, :, :, :]
        for (uint32_t ciIdx = 0; ciIdx < ci_; ++ciIdx) {
            for (uint32_t iz = 0; iz < inD_; ++iz) {
                for (uint32_t iy = 0; iy < inH_; ++iy) {
                    for (uint32_t ix = 0; ix < inW_; ++ix) {
                        // Load x scalar
                        uint64_t xOff = XIndex(nIdx, ciIdx, iz, iy, ix);
                        LocalTensor<float> xLocal = qX_.AllocTensor<float>();
                        DataCopy(xLocal, xGm_[xOff], 1);
                        // For each kernel element
                        for (uint32_t kd = 0; kd < kD_; ++kd) {
                            uint32_t oz = iz + kd;
                            if (oz >= outD_) continue;
                            for (uint32_t kh = 0; kh < kH_; ++kh) {
                                uint32_t oy = iy + kh;
                                if (oy >= outH_) continue;
                                for (uint32_t kw = 0; kw < kW_; ++kw) {
                                    uint32_t ox = ix + kw;
                                    if (ox >= outW_) continue;

                                    uint64_t wOff = WIndex(ciIdx, coIdx, kd, kh, kw);
                                    LocalTensor<float> wLocal = qW_.AllocTensor<float>();
                                    DataCopy(wLocal, wGm_[wOff], 1);

                                    uint64_t yOff = YIndex(nIdx, coIdx, oz, oy, ox);
                                    LocalTensor<float> yLocal = qY_.AllocTensor<float>();
                                    DataCopy(yLocal, yGm_[yOff], 1);

                                    LocalTensor<float> tmpLocal = qTmp_.AllocTensor<float>();
                                    // tmp = x * w
                                    Mul(tmpLocal, xLocal, wLocal, 1);
                                    // y = y + tmp
                                    Add(yLocal, yLocal, tmpLocal, 1);

                                    DataCopy(yGm_[yOff], yLocal, 1);

                                    qTmp_.FreeTensor(tmpLocal);
                                    qY_.FreeTensor(yLocal);
                                    qW_.FreeTensor(wLocal);
                                }
                            }
                        }
                        qX_.FreeTensor(xLocal);
                    }
                }
            }
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN,  QBUF> qX_;
    TQue<TPosition::VECIN,  QBUF> qW_;
    TQue<TPosition::VECOUT, QBUF> qY_;
    TQue<TPosition::VECCALC, QBUF> qTmp_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t n_{0}, ci_{0}, co_{0};
    uint32_t inD_{0}, inH_{0}, inW_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};
    uint32_t outD_{0}, outH_{0}, outW_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.ci, tiling_data.co,
            tiling_data.inD, tiling_data.inH, tiling_data.inW,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW);
    op.Process();
}
