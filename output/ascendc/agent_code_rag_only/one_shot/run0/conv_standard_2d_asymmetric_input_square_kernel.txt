namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 128;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    if (x_shape->GetActualShape().GetDimNum() != 4 || w_shape->GetActualShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Ci = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t Hi = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t Wi = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const uint32_t Co = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t Ci_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    const uint32_t Kh = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(3));

    if (Ci != Ci_w) {
        return ge::GRAPH_FAILED;
    }
    if (Kh == 0 || Kw == 0 || Hi < Kh || Wi < Kw) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Ho = Hi - Kh + 1;  // stride=1, pad=0, dilation=1
    const uint32_t Wo = Wi - Kw + 1;

    // UB requirement: 3 local vectors of tileW float32 => bytes = 12 * tileW
    uint32_t maxTile = UB_BUDGET_BYTES / 16; // keep headroom
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileW = maxTile;
    if (Wo < tileW) {
        const uint32_t aligned = AlignDown(Wo, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : Wo;
    }

    const uint64_t totalTasks64 = static_cast<uint64_t>(N) * static_cast<uint64_t>(Co) * static_cast<uint64_t>(Ho);
    uint32_t totalTasks = (totalTasks64 > 0xFFFFFFFFull) ? 0xFFFFFFFFu : static_cast<uint32_t>(totalTasks64);
    uint32_t blockDim = totalTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_ci(Ci);
    tiling.set_hi(Hi);
    tiling.set_wi(Wi);
    tiling.set_co(Co);
    tiling.set_kh(Kh);
    tiling.set_kw(Kw);
    tiling.set_ho(Ho);
    tiling.set_wo(Wo);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 1;
constexpr int32_t TMP_BUF_NUM = 1;

class KernelConvStandard2dAsymSq {
public:
    __aicore__ inline KernelConvStandard2dAsymSq() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t ci, uint32_t hi, uint32_t wi,
                                uint32_t co, uint32_t kh, uint32_t kw,
                                uint32_t ho, uint32_t wo, uint32_t tileW)
    {
        n_ = n; ci_ = ci; hi_ = hi; wi_ = wi;
        co_ = co; kh_ = kh; kw_ = kw; ho_ = ho; wo_ = wo;
        tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_)*ci_*hi_*wi_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(co_)*ci_*kh_*kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_)*co_*ho_*wo_);

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM,  tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMP_BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, 9 * sizeof(float)); // small buffer for 3x3 weights per (oc, ic)
    }

    __aicore__ inline void Process()
    {
        if (n_==0 || ci_==0 || co_==0 || ho_==0 || wo_==0 || tileW_==0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t tasksU64 = static_cast<uint64_t>(n_) * static_cast<uint64_t>(co_) * static_cast<uint64_t>(ho_);
        const uint32_t tasks = (tasksU64 > 0xFFFFFFFFull) ? 0xFFFFFFFFu : static_cast<uint32_t>(tasksU64);

        for (uint32_t t = blockIdx; t < tasks; t += blockNum) {
            // decompose task index into (n, oc, oh)
            uint32_t rem = t;
            const uint32_t nIdx = rem / (co_ * ho_);
            rem = rem % (co_ * ho_);
            const uint32_t ocIdx = rem / ho_;
            const uint32_t ohIdx = rem % ho_;

            // process width in tiles
            for (uint32_t owBase = 0; owBase < wo_; owBase += tileW_) {
                uint32_t curW = wo_ - owBase;
                if (curW > tileW_) curW = tileW_;

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                bool firstTerm = true;

                for (uint32_t icIdx = 0; icIdx < ci_; ++icIdx) {
                    // load 3x3 weights for this (oc, ic)
                    AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                    const uint64_t wBase = (((static_cast<uint64_t>(ocIdx) * ci_ + icIdx) * kh_) * kw_);
                    AscendC::DataCopy(wLocal, wGm_[wBase], kh_ * kw_);
                    // iterate over kernel window
                    for (uint32_t khIdx = 0; khIdx < kh_; ++khIdx) {
                        const uint32_t inH = ohIdx + khIdx; // stride=1
                        for (uint32_t kwIdx = 0; kwIdx < kw_; ++kwIdx) {
                            const uint32_t inWStart = owBase + kwIdx; // stride=1
                            // copy input slice of length curW starting at inWStart
                            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                            const uint64_t xBase = ((((static_cast<uint64_t>(nIdx) * ci_ + icIdx) * hi_ + inH) * wi_) + inWStart);
                            AscendC::DataCopy(xLocal, xGm_[xBase], curW);

                            // compute tmp = xLocal * weight_scalar, then accumulate into yLocal
                            const float wVal = wLocal.GetValue(khIdx * kw_ + kwIdx);
                            AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                            AscendC::Muls(tmpLocal, xLocal, wVal, curW);
                            if (firstTerm) {
                                // yLocal = tmpLocal
                                AscendC::DataCopy(yLocal, tmpLocal, curW);
                                firstTerm = false;
                            } else {
                                AscendC::Add(yLocal, yLocal, tmpLocal, curW);
                            }

                            tmpQueue_.FreeTensor(tmpLocal);
                            inQueueX_.FreeTensor(xLocal);
                        }
                    }
                    wQueue_.FreeTensor(wLocal);
                }

                // store output tile
                const uint64_t yBase = ((((static_cast<uint64_t>(nIdx) * co_ + ocIdx) * ho_ + ohIdx) * wo_) + owBase);
                AscendC::DataCopy(yGm_[yBase], yLocal, curW);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, ci_{0}, hi_{0}, wi_{0};
    uint32_t co_{0}, kh_{0}, kw_{0};
    uint32_t ho_{0}, wo_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dAsymSq op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.ci, tiling_data.hi, tiling_data.wi,
            tiling_data.co, tiling_data.kh, tiling_data.kw,
            tiling_data.ho, tiling_data.wo, tiling_data.tileW);
    op.Process();
}
