kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 1;

class KernelConv2dReluBiasAddCustom {
public:
    __aicore__ inline KernelConv2dReluBiasAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR conv_b, GM_ADDR post_b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo, uint32_t tileW,
                                uint32_t tilesPerRow, uint32_t totalTiles, uint32_t hasConvBias,
                                uint32_t sH, uint32_t sW, uint32_t pH, uint32_t pW,
                                uint32_t dH, uint32_t dW)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo; tileW_ = tileW;
        tilesPerRow_ = tilesPerRow; totalTiles_ = totalTiles;
        hasConvBias_ = hasConvBias;
        sH_ = sH; sW_ = sW; pH_ = pH; pW_ = pW; dH_ = dH; dW_ = dW;

        const uint32_t xSize = N_ * Cin_ * H_ * W_;
        const uint32_t wSize = Cout_ * Cin_ * Kh_ * Kw_;
        const uint32_t ySize = N_ * Cout_ * Ho_ * Wo_;
        const uint32_t cbSize = Cout_;
        const uint32_t pbSize = Cout_; // post_bias: [Cout,1,1] flattened

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);
        if (conv_b != nullptr) {
            cbGm_.SetGlobalBuffer((__gm__ float*)conv_b, cbSize);
        }
        pbGm_.SetGlobalBuffer((__gm__ float*)post_b, pbSize);

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileW_ * sizeof(float));
        // small temp buffer for scalar fetch if needed
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalTiles_ == 0 || tileW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tileId = blockIdx; tileId < totalTiles_; tileId += blockNum) {
            // Decode tileId -> (n, co, ho, tileX)
            uint32_t rem = tileId;
            const uint32_t tx = rem % tilesPerRow_;
            rem /= tilesPerRow_;
            const uint32_t ho = rem % Ho_;
            rem /= Ho_;
            const uint32_t co = rem % Cout_;
            const uint32_t n  = rem / Cout_;

            const uint32_t wBase = tx * tileW_;
            uint32_t curTile = Wo_ - wBase;
            if (curTile > tileW_) curTile = tileW_;
            if (curTile == 0) continue;

            // Accumulator yLocal
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Duplicate(yLocal, 0.0f, curTile);

            // Reduction over Cin, Kh, Kw
            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    // input row index (with padding/dilation/stride)
                    int32_t hOut = static_cast<int32_t>(ho);
                    int32_t hIn = hOut * static_cast<int32_t>(sH_) - static_cast<int32_t>(pH_) + static_cast<int32_t>(kh) * static_cast<int32_t>(dH_);
                    if (hIn < 0 || hIn >= static_cast<int32_t>(H_)) {
                        continue; // outside padding
                    }
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        int32_t wOut = static_cast<int32_t>(wBase);
                        int32_t wInBase = wOut * static_cast<int32_t>(sW_) - static_cast<int32_t>(pW_) + static_cast<int32_t>(kw) * static_cast<int32_t>(dW_);
                        if (wInBase < 0 || (wInBase + static_cast<int32_t>(curTile) - 1) >= static_cast<int32_t>(W_)) {
                            // This branch can occur only when padding/stride cause partial overlap
                            // Clamp the valid segment
                            uint32_t valid = 0;
                            uint32_t startShift = 0;
                            if (wInBase < 0) {
                                startShift = static_cast<uint32_t>(-wInBase);
                                wInBase = 0;
                            }
                            if (wInBase < static_cast<int32_t>(W_)) {
                                int32_t maxValid = static_cast<int32_t>(W_) - wInBase;
                                valid = (maxValid > 0) ? static_cast<uint32_t>(maxValid) : 0u;
                                if (valid > (curTile - startShift)) valid = curTile - startShift;
                            }
                            if (valid == 0) continue;

                            // Load weight scalar
                            const uint32_t wIndex = (((co * Cin_ + ci) * Kh_ + kh) * Kw_) + kw;
                            float wScalar = LoadScalarFromGM(wGm_, wIndex);

                            // Prepare x segment [valid] and accumulate into yLocal[startShift : startShift+valid]
                            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                            const uint32_t xIndex = (((n * Cin_ + ci) * H_ + static_cast<uint32_t>(hIn)) * W_) + static_cast<uint32_t>(wInBase);
                            AscendC::DataCopy(xLocal, xGm_[xIndex], valid);
                            AscendC::Muls(xLocal, xLocal, wScalar, valid);

                            // Accumulate into the subrange by operating on a shifted view
                            AscendC::LocalTensor<float> ySub = yLocal[startShift];
                            AscendC::Add(ySub, ySub, xLocal, valid);

                            inQueueX_.FreeTensor(xLocal);
                            continue;
                        }

                        // Fast path full tile inside image
                        const uint32_t wIndex = (((co * Cin_ + ci) * Kh_ + kh) * Kw_) + kw;
                        float wScalar = LoadScalarFromGM(wGm_, wIndex);

                        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                        const uint32_t xIndex = (((n * Cin_ + ci) * H_ + static_cast<uint32_t>(hIn)) * W_) + static_cast<uint32_t>(wInBase);
                        AscendC::DataCopy(xLocal, xGm_[xIndex], curTile);
                        AscendC::Muls(xLocal, xLocal, wScalar, curTile);
                        AscendC::Add(yLocal, yLocal, xLocal, curTile);
                        inQueueX_.FreeTensor(xLocal);
                    }
                }
            }

            // Add conv bias if available: y += conv_bias[co]
            if (hasConvBias_) {
                float cb = LoadScalarFromGM(cbGm_, co);
                AscendC::Adds(yLocal, yLocal, cb, curTile);
            }

            // ReLU: y = max(y, 0)
            AscendC::Maxs(yLocal, yLocal, 0.0f, curTile);

            // Add post bias (broadcast): y += post_bias[co]
            float pb = LoadScalarFromGM(pbGm_, co);
            AscendC::Adds(yLocal, yLocal, pb, curTile);

            // Store
            const uint32_t yIndex = (((n * Cout_ + co) * Ho_ + ho) * Wo_) + wBase;
            AscendC::DataCopy(yGm_[yIndex], yLocal, curTile);
            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    __aicore__ inline float LoadScalarFromGM(const AscendC::GlobalTensor<float>& gm, uint32_t idx)
    {
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(tmp, gm[idx], 1);
        // Assume AscendC provides GetValue to read scalar from local tensor
        float val = AscendC::GetValue(tmp, 0);
        tmpQueue_.FreeTensor(tmp);
        return val;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> cbGm_;
    AscendC::GlobalTensor<float> pbGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0}, tileW_{0}, tilesPerRow_{0}, totalTiles_{0};
    uint32_t hasConvBias_{0};
    uint32_t sH_{1}, sW_{1}, pH_{0}, pW_{0}, dH_{1}, dW_{1};
};

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR post_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluBiasAddCustom op;
    op.Init(
        x, weight, conv_bias, post_bias, y,
        tiling_data.N, tiling_data.Cin, tiling_data.Cout,
        tiling_data.H, tiling_data.W, tiling_data.Kh, tiling_data.Kw,
        tiling_data.Ho, tiling_data.Wo, tiling_data.tileW,
        tiling_data.tilesPerRow, tiling_data.totalTiles, tiling_data.hasConvBias,
        tiling_data.strideH, tiling_data.strideW, tiling_data.padH, tiling_data.padW,
        tiling_data.dilH, tiling_data.dilW
    );
    op.Process();
}
