static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const gert::Shape *in0 = context->GetInputShape(0);
    const auto &ori0 = in0->GetOriginShape();

    const uint32_t dimNum = static_cast<uint32_t>(ori0.GetDimNum());
    uint32_t cols = 1;
    if (dimNum == 0) {
        cols = static_cast<uint32_t>(ori0.GetShapeSize());
        if (cols == 0) cols = 1;
    } else {
        cols = static_cast<uint32_t>(ori0.GetDim(dimNum - 1));
        if (cols == 0) cols = 1;
    }
    const uint32_t total = static_cast<uint32_t>(ori0.GetShapeSize());
    const uint32_t rows = (cols == 0) ? 0u : (total / cols);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *marginPtr = attrs->GetAttrPointer<float>(0);
    const float margin = (marginPtr != nullptr) ? *marginPtr : 1.0f;

    // Use single-core reduction for deterministic scalar output.
    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y,
                                uint32_t rows, uint32_t cols, float margin)
    {
        rows_ = rows;
        cols_ = cols;
        margin_ = margin;

        aGm_.SetGlobalBuffer((__gm__ float*)anchor, rows_ * cols_);
        pGm_.SetGlobalBuffer((__gm__ float*)positive, rows_ * cols_);
        nGm_.SetGlobalBuffer((__gm__ float*)negative, rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        // VECCALC queue used for local temporaries and row buffers
        pipe_.InitBuffer(tmpQueue_, 6, cols_ * sizeof(float));
        // Scalar queue for 1-element temporaries
        pipe_.InitBuffer(scalarQueue_, 4, 64 * sizeof(float)); // reserve at least 64B
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) {
            WriteZero();
            return;
        }
        // Single-core reduction as blockDim set to 1 in tiling
        if (GetBlockIdx() != 0) {
            return;
        }

        const float eps = 1e-6f;

        LocalTensor<float> sumLocal = scalarQueue_.AllocTensor<float>();
        Dup(sumLocal, 0.0f, 1);

        LocalTensor<float> marginLocal = scalarQueue_.AllocTensor<float>();
        Dup(marginLocal, margin_, 1);

        LocalTensor<float> zeroLocal = scalarQueue_.AllocTensor<float>();
        Dup(zeroLocal, 0.0f, 1);

        for (uint32_t r = 0; r < rows_; ++r) {
            // Load a row for each of anchor, positive, negative
            LocalTensor<float> aLocal = tmpQueue_.AllocTensor<float>();
            LocalTensor<float> pLocal = tmpQueue_.AllocTensor<float>();
            LocalTensor<float> nLocal = tmpQueue_.AllocTensor<float>();

            DataCopy(aLocal, aGm_[r * cols_], cols_);
            DataCopy(pLocal, pGm_[r * cols_], cols_);
            DataCopy(nLocal, nGm_[r * cols_], cols_);

            // diffAP = a - p; diffAN = a - n
            LocalTensor<float> diffAP = tmpQueue_.AllocTensor<float>();
            LocalTensor<float> diffAN = tmpQueue_.AllocTensor<float>();
            Sub(diffAP, aLocal, pLocal, cols_);
            Sub(diffAN, aLocal, nLocal, cols_);

            // Square diffs
            Mul(diffAP, diffAP, diffAP, cols_);
            Mul(diffAN, diffAN, diffAN, cols_);

            // Reduce sum over cols to get squared L2 norms
            LocalTensor<float> workLocal = tmpQueue_.AllocTensor<float>();
            LocalTensor<float> sumAP = scalarQueue_.AllocTensor<float>();
            LocalTensor<float> sumAN = scalarQueue_.AllocTensor<float>();

            ReduceSumSmall(diffAP, workLocal, sumAP, cols_);
            ReduceSumSmall(diffAN, workLocal, sumAN, cols_);

            // d = sqrt(sum + eps)
            LocalTensor<float> epsLocal = scalarQueue_.AllocTensor<float>();
            Dup(epsLocal, eps, 1);

            Add(sumAP, sumAP, epsLocal, 1);
            Add(sumAN, sumAN, epsLocal, 1);
            Sqrt(sumAP, sumAP, 1);
            Sqrt(sumAN, sumAN, 1);

            // delta = max(d_ap - d_an + margin, 0)
            LocalTensor<float> delta = scalarQueue_.AllocTensor<float>();
            Sub(delta, sumAP, sumAN, 1);
            Add(delta, delta, marginLocal, 1);
            Max(delta, delta, zeroLocal, 1);

            // accumulate
            Add(sumLocal, sumLocal, delta, 1);

            // free locals
            scalarQueue_.FreeTensor(delta);
            scalarQueue_.FreeTensor(epsLocal);
            scalarQueue_.FreeTensor(sumAN);
            scalarQueue_.FreeTensor(sumAP);
            tmpQueue_.FreeTensor(workLocal);
            tmpQueue_.FreeTensor(diffAN);
            tmpQueue_.FreeTensor(diffAP);
            tmpQueue_.FreeTensor(nLocal);
            tmpQueue_.FreeTensor(pLocal);
            tmpQueue_.FreeTensor(aLocal);
        }

        // mean
        float invRows = 1.0f / static_cast<float>(rows_);
        LocalTensor<float> invLocal = scalarQueue_.AllocTensor<float>();
        Dup(invLocal, invRows, 1);
        Mul(sumLocal, sumLocal, invLocal, 1);

        // write back
        DataCopy(yGm_[0], sumLocal, 1);

        scalarQueue_.FreeTensor(invLocal);
        scalarQueue_.FreeTensor(zeroLocal);
        scalarQueue_.FreeTensor(marginLocal);
        scalarQueue_.FreeTensor(sumLocal);
    }

private:
    __aicore__ inline void ReduceSumSmall(const LocalTensor<float>& srcLocal,
                                          const LocalTensor<float>& workLocal,
                                          const LocalTensor<float>& dstLocal,
                                          const uint32_t len)
    {
        uint32_t repeat = len / 64;
        uint32_t tailNum = len % 64;
        if (repeat > 0) {
            WholeReduceSum(workLocal, srcLocal, 64, repeat, 1, 1, 8);
            pipe_barrier(PIPE_V);
        }
        if (tailNum != 0) {
            WholeReduceSum(workLocal[repeat], srcLocal[len - tailNum], tailNum, 1, 1, 1, 8);
            pipe_barrier(PIPE_V);
            repeat += 1;
        }
        WholeReduceSum(dstLocal, workLocal, repeat, 1, 1, 1, 8);
        pipe_barrier(PIPE_V);
    }

    __aicore__ inline void WriteZero()
    {
        if (GetBlockIdx() != 0) return;
        LocalTensor<float> zeroLocal = scalarQueue_.AllocTensor<float>();
        Dup(zeroLocal, 0.0f, 1);
        DataCopy(yGm_[0], zeroLocal, 1);
        scalarQueue_.FreeTensor(zeroLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, 6> tmpQueue_;
    TQue<TPosition::VECCALC, 4> scalarQueue_;

    GlobalTensor<float> aGm_;
    GlobalTensor<float> pGm_;
    GlobalTensor<float> nGm_;
    GlobalTensor<float> yGm_;

    uint32_t rows_{0};
    uint32_t cols_{0};
    float margin_{1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                 GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y,
            tiling_data.rows, tiling_data.cols, tiling_data.margin);
    op.Process();
}
