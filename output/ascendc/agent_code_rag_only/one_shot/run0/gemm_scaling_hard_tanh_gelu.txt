{

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmScalingHardTanhGeluCustom {
public:
    __aicore__ inline KernelGemmScalingHardTanhGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN,
                                float scaling, float htMin, float htMax)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;

        scale_ = scaling;
        htMin_ = htMin;
        htMax_ = htMax;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        // Local buffers
        pipe_.InitBuffer(accQue_, 1, tileM_ * tileN_ * sizeof(float));  // accum as [tileN, tileM]
        pipe_.InitBuffer(xcolQue_, 1, tileM_ * sizeof(float));          // x column for mLen
        pipe_.InitBuffer(wvecQue_, 1, tileN_ * sizeof(float));          // weights for current k across nLen
        pipe_.InitBuffer(tmpVecQue_, 1, tileM_ * sizeof(float));        // temp vec size mLen
        pipe_.InitBuffer(outRowQue_, 1, tileN_ * sizeof(float));        // row output buffer size nLen
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t mBase = blockIdx * tileM_;
        const uint32_t mStride = blockNum * tileM_;

        while (mBase < M_) {
            const uint32_t mLen = AscendC::min(tileM_, M_ - mBase);

            for (uint32_t n0 = 0; n0 < N_; n0 += tileN_) {
                const uint32_t nLen = AscendC::min(tileN_, N_ - n0);

                AscendC::LocalTensor<float> acc = accQue_.AllocTensor<float>();
                // Initialize acc[j, 0..mLen-1] with bias b[n0+j]
                for (uint32_t j = 0; j < nLen; ++j) {
                    // Load bias scalar
                    float bval = 0.0f;
                    {
                        AscendC::LocalTensor<float> tmp1 = wvecQue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp1, bGm_[n0 + j], 1);
                        // Read local value
                        bval = tmp1.GetValue(0);
                        wvecQue_.FreeTensor(tmp1);
                    }
                    // Fill acc segment for this j
                    uint32_t baseOff = j * tileM_;
                    for (uint32_t r = 0; r < mLen; ++r) {
                        acc.SetValue(baseOff + r, bval);
                    }
                }
                // Zero-init remaining (if any, but we'll keep unused elements untouched)

                // K loop, using kStep = 1 for simplicity
                for (uint32_t k = 0; k < K_; ++k) {
                    // Load weight vector wvec[0..nLen-1] = w[n0..n0+nLen-1, k]
                    AscendC::LocalTensor<float> wvec = wvecQue_.AllocTensor<float>();
                    for (uint32_t j = 0; j < nLen; ++j) {
                        AscendC::DataCopy(wvec[j], wGm_[(static_cast<uint64_t>(n0 + j) * K_) + k], 1);
                    }

                    // Load x column vector xcol[0..mLen-1] = x[mBase..mBase+mLen-1, k]
                    AscendC::LocalTensor<float> xcol = xcolQue_.AllocTensor<float>();
                    for (uint32_t r = 0; r < mLen; ++r) {
                        AscendC::DataCopy(xcol[r], xGm_[(static_cast<uint64_t>(mBase + r) * K_) + k], 1);
                    }

                    // Accumulate acc[j, :] += xcol[:] * wvec[j]
                    AscendC::LocalTensor<float> tmpVec = tmpVecQue_.AllocTensor<float>();
                    for (uint32_t j = 0; j < nLen; ++j) {
                        const float wj = wvec.GetValue(j);
                        AscendC::Muls(tmpVec, xcol, wj, mLen);
                        AscendC::Add(acc[j * tileM_], acc[j * tileM_], tmpVec, mLen);
                    }
                    tmpVecQue_.FreeTensor(tmpVec);
                    xcolQue_.FreeTensor(xcol);
                    wvecQue_.FreeTensor(wvec);
                }

                // Post-ops: scaling, hardtanh clamp, GELU (fast tanh approximation)
                for (uint32_t j = 0; j < nLen; ++j) {
                    AscendC::LocalTensor<float> vec = acc[j * tileM_];
                    // Scale
                    AscendC::Muls(vec, vec, scale_, mLen);
                    // Clamp to [htMin_, htMax_]
                    AscendC::Maximums(vec, vec, htMin_, mLen);
                    AscendC::Minimums(vec, vec, htMax_, mLen);

                    // GELU: 0.5*x*(1 + tanh(sqrt(2/pi)*(x + 0.044715*x^3)))
                    AscendC::LocalTensor<float> tmp = tmpVecQue_.AllocTensor<float>();
                    // tmp = x^2
                    AscendC::Mul(tmp, vec, vec, mLen);
                    // tmp = x^3
                    AscendC::Mul(tmp, tmp, vec, mLen);
                    // tmp = 0.044715 * x^3
                    AscendC::Muls(tmp, tmp, 0.044715f, mLen);
                    // tmp = x + 0.044715*x^3
                    AscendC::Add(tmp, tmp, vec, mLen);
                    // tmp = sqrt(2/pi) * tmp
                    AscendC::Muls(tmp, tmp, 0.7978845608028654f, mLen);
                    // tmp = tanh(tmp)
                    AscendC::Tanh(tmp, tmp, mLen);
                    // tmp = (1 + tanh(...)) * 0.5
                    AscendC::Adds(tmp, tmp, 1.0f, mLen);
                    AscendC::Muls(tmp, tmp, 0.5f, mLen);
                    // vec = x * tmp
                    AscendC::Mul(vec, vec, tmp, mLen);
                    tmpVecQue_.FreeTensor(tmp);
                }

                // Write out to GM: per row contiguous length nLen
                AscendC::LocalTensor<float> outRow = outRowQue_.AllocTensor<float>();
                for (uint32_t r = 0; r < mLen; ++r) {
                    for (uint32_t j = 0; j < nLen; ++j) {
                        float v = acc.GetValue(j * tileM_ + r);
                        outRow.SetValue(j, v);
                    }
                    AscendC::DataCopy(yGm_[(static_cast<uint64_t>(mBase + r) * N_) + n0], outRow, nLen);
                }
                outRowQue_.FreeTensor(outRow);

                accQue_.FreeTensor(acc);
            }

            mBase += mStride;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xcolQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wvecQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVecQue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outRowQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileM_{64};
    uint32_t tileN_{8};

    float scale_{1.0f};
    float htMin_{-1.0f};
    float htMax_{1.0f};
};

extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmScalingHardTanhGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN,
            tiling_data.scaling, tiling_data.htMin, tiling_data.htMax);
    op.Process();
}
