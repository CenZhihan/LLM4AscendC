    if (L < k) return 0u;
    return (L - k) / s + 1u;

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t MAX_COUT_LOCAL = 256; // softmax vector cap per position

class KernelConv3dSoftmaxMaxPoolMaxPoolCustom {
public:
    __aicore__ inline KernelConv3dSoftmaxMaxPoolMaxPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Di, uint32_t Hi, uint32_t Wi,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Do, uint32_t Ho, uint32_t Wo,
                                uint32_t D2, uint32_t H2, uint32_t W2,
                                uint32_t tilePositions, uint32_t chPerIter)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Di_ = Di; Hi_ = Hi; Wi_ = Wi;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        Do_ = Do; Ho_ = Ho; Wo_ = Wo;
        D2_ = D2; H2_ = H2; W2_ = W2;
        tilePos_ = tilePositions;
        chPerIter_ = chPerIter;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * Cin_ * Di_ * Hi_ * Wi_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, Cout_ * Cin_ * Kd_ * Kh_ * Kw_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, Cout_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * Cout_ * D2_ * H2_ * W2_);

        // UB: input tile for one pooled output region: Cin * (4+Kd-1) * (4+Kh-1) * (4+Kw-1)
        inD_ = (4u + Kd_ - 1u);
        inH_ = (4u + Kh_ - 1u);
        inW_ = (4u + Kw_ - 1u);

        uint32_t inTileElems = Cin_ * inD_ * inH_ * inW_;
        uint32_t wTileElems = Cin_ * Kd_ * Kh_ * Kw_;
        uint32_t biasElems = Cout_;

        uint32_t maxBytes = (inTileElems + wTileElems + biasElems + MAX_COUT_LOCAL * 2) * sizeof(float);
        // allocate a single VECCALC buffer big enough
        pipe_.InitBuffer(tmpBuf_, 1, maxBytes);
    }

    __aicore__ inline void Process()
    {
        if (D2_ == 0 || H2_ == 0 || W2_ == 0 || Cout_ == 0) return;

        const uint32_t totalPositions = N_ * D2_ * H2_ * W2_;
        const uint32_t blkNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blkIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t stride = blkNum * tilePos_;
        uint32_t base = blkIdx * tilePos_;
        if (base >= totalPositions) return;

        while (base < totalPositions) {
            uint32_t work = tilePos_;
            if (base + work > totalPositions) work = totalPositions - base;

            for (uint32_t i = 0; i < work; ++i) {
                ComputeOne(base + i);
            }
            base += stride;
        }
    }

private:
    __aicore__ inline void ComputeOne(uint32_t posIdx)
    {
        // Decode (n, d2, h2, w2)
        uint32_t wh2 = W2_;
        uint32_t hh2 = H2_;
        uint32_t dh2 = D2_;

        uint32_t n = posIdx / (dh2 * hh2 * wh2);
        uint32_t rem = posIdx % (dh2 * hh2 * wh2);
        uint32_t d2 = rem / (hh2 * wh2);
        rem = rem % (hh2 * wh2);
        uint32_t h2 = rem / wh2;
        uint32_t w2 = rem % wh2;

        // Base position in conv output grid covered by the two 2x2x2 pools -> 4x4x4 block
        uint32_t d0 = d2 * 4u;
        uint32_t h0 = h2 * 4u;
        uint32_t w0 = w2 * 4u;

        uint32_t dMax = d0 + 4u; if (dMax > Do_) dMax = Do_;
        uint32_t hMax = h0 + 4u; if (hMax > Ho_) hMax = Ho_;
        uint32_t wMax = w0 + 4u; if (wMax > Wo_) wMax = Wo_;

        // UB buffer layout: [input_tile | weights_per_c | bias | convBuf | maxSoft]
        AscendC::LocalTensor<float> ub = tmpBuf_.AllocTensor<float>();
        float* ubPtr = reinterpret_cast<float*>(ub.GetPhyAddr());

        float* inTile = ubPtr; // size Cin_ * inD_ * inH_ * inW_
        const uint32_t inTileElems = Cin_ * inD_ * inH_ * inW_;
        float* wTile = inTile + inTileElems;  // size Cin_ * Kd_ * Kh_ * Kw_
        const uint32_t wTileElems = Cin_ * Kd_ * Kh_ * Kw_;
        float* biasLocal = wTile + wTileElems; // size Cout_
        float* convBuf = biasLocal + Cout_;    // size MAX_COUT_LOCAL
        float* maxSoft = convBuf + MAX_COUT_LOCAL; // size MAX_COUT_LOCAL

        // Copy bias to UB
        AscendC::DataCopy(AscendC::LocalTensor<float>(biasLocal, Cout_), bGm_[0], Cout_);

        // Prefetch input tile once per pooled output position
        // Copy region: D in [d0, d0 + inD_ - 1], H in [h0, ...], W [w0, ...], across all Cin
        for (uint32_t ci = 0; ci < Cin_; ++ci) {
            for (uint32_t dd = 0; dd < inD_; ++dd) {
                uint32_t d = d0 + dd;
                if (d >= Di_) continue; // guard
                for (uint32_t hh = 0; hh < inH_; ++hh) {
                    uint32_t h = h0 + hh;
                    if (h >= Hi_) continue;
                    uint32_t gmOffset = IndexX(n, ci, d, h, w0);
                    uint32_t ubOffset = ((ci * inD_ + dd) * inH_ + hh) * inW_;
                    uint32_t validW = inW_;
                    if (w0 + validW > Wi_) validW = Wi_ - w0;
                    AscendC::DataCopy(
                        AscendC::LocalTensor<float>(inTile + ubOffset, validW),
                        xGm_[gmOffset], validW);
                    // zero-pad tail if needed
                    for (uint32_t t = validW; t < inW_; ++t) {
                        inTile[ubOffset + t] = 0.0f;
                    }
                }
            }
        }

        // Initialize maxSoft to -inf
        for (uint32_t c = 0; c < Cout_; ++c) {
            maxSoft[c] = -INFINITY;
        }

        // Iterate 4x4x4 region of conv-output positions; clamp for borders
        for (uint32_t od = d0; od < dMax; ++od) {
            for (uint32_t oh = h0; oh < hMax; ++oh) {
                for (uint32_t ow = w0; ow < wMax; ++ow) {
                    // At this spatial position, compute conv for all Cout channels -> convBuf
                    for (uint32_t co = 0; co < Cout_; ++co) {
                        // Load weights for this co into UB
                        CopyWeightToUb(co, wTile);
                        // Convolution sum at (od,oh,ow)
                        float acc = biasLocal[co];
                        for (uint32_t ci = 0; ci < Cin_; ++ci) {
                            for (uint32_t kd = 0; kd < Kd_; ++kd) {
                                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                                    const float* wBase = wTile + (((ci * Kd_ + kd) * Kh_ + kh) * Kw_);
                                    // input start in UB for this (ci, od, oh)
                                    uint32_t dd = (od - d0) + kd;
                                    uint32_t hh = (oh - h0) + kh;
                                    const float* xBase = inTile + (((ci * inD_ + dd) * inH_ + hh) * inW_) + (ow - w0);
                                    // unroll kw
                                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                        acc += xBase[kw] * wBase[kw];
                                    }
                                }
                            }
                        }
                        convBuf[co] = acc;
                    }
                    // Softmax across channels at this (n,od,oh,ow)
                    float maxv = -INFINITY;
                    for (uint32_t co = 0; co < Cout_; ++co) {
                        float v = convBuf[co];
                        if (v > maxv) maxv = v;
                    }
                    float sumExp = 0.0f;
                    for (uint32_t co = 0; co < Cout_; ++co) {
                        float e = expf(convBuf[co] - maxv);
                        convBuf[co] = e;
                        sumExp += e;
                    }
                    const float invSum = 1.0f / (sumExp + 1e-12f);
                    for (uint32_t co = 0; co < Cout_; ++co) {
                        float p = convBuf[co] * invSum;
                        // update max over 4x4x4 region for this channel
                        if (p > maxSoft[co]) {
                            maxSoft[co] = p;
                        }
                    }
                }
            }
        }

        // Write result vector maxSoft to GM at (n, :, d2, h2, w2)
        uint32_t outBase = IndexY(n, 0u, d2, h2, w2);
        AscendC::DataCopy(yGm_[outBase], AscendC::LocalTensor<float>(maxSoft, Cout_), Cout_);

        tmpBuf_.FreeTensor(ub);
    }

    __aicore__ inline uint32_t IndexX(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        // N, C, D, H, W
        return ((((n * Cin_ + c) * Di_ + d) * Hi_ + h) * Wi_ + w);
    }

    __aicore__ inline uint32_t IndexW(uint32_t co, uint32_t ci, uint32_t kd, uint32_t kh, uint32_t kw) const
    {
        // Cout, Cin, Kd, Kh, Kw
        return ((((co * Cin_ + ci) * Kd_ + kd) * Kh_ + kh) * Kw_ + kw);
    }

    __aicore__ inline uint32_t IndexY(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        // N, Cout, D2, H2, W2
        return ((((n * Cout_ + c) * D2_ + d) * H2_ + h) * W2_ + w);
    }

    __aicore__ inline void CopyWeightToUb(uint32_t co, float* wTile)
    {
        // Copy w[co, :, :, :, :] -> UB
        const uint32_t wPerC = Cin_ * Kd_ * Kh_ * Kw_;
        uint32_t gmOffset = co * wPerC;
        AscendC::DataCopy(AscendC::LocalTensor<float>(wTile, wPerC), wGm_[gmOffset], wPerC);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Di_{0}, Hi_{0}, Wi_{0};
    uint32_t Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Do_{0}, Ho_{0}, Wo_{0};
    uint32_t D2_{0}, H2_{0}, W2_{0};
    uint32_t tilePos_{1}, chPerIter_{0};
    uint32_t inD_{0}, inH_{0}, inW_{0};
};

extern "C" __global__ __aicore__ void conv3d_softmax_max_pool_max_pool_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dSoftmaxMaxPoolMaxPoolCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.Di, tiling_data.Hi, tiling_data.Wi,
            tiling_data.Kd, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Do, tiling_data.Ho, tiling_data.Wo,
            tiling_data.D2, tiling_data.H2, tiling_data.W2,
            tiling_data.tilePositions, tiling_data.chPerIter);
    op.Process();
}
