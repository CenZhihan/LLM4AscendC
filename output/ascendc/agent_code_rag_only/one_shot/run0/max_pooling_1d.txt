#include "max_pooling_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB share (conservative)
static constexpr uint32_t ALIGN_ELEMS = 1024;            // align tile sizes for dma bursts
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // cap to avoid long iters

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint64_t MaxU64(uint64_t a, uint64_t b) { return a > b ? a : b; }
static inline uint64_t MinU64(uint64_t a, uint64_t b) { return a < b ? a : b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();
    int32_t dimNum = static_cast<int32_t>(origin.GetDimNum());
    if (dimNum < 1) return ge::GRAPH_FAILED;

    uint64_t L = static_cast<uint64_t>(origin.GetDim(dimNum - 1));
    uint64_t channels = 1;
    for (int32_t i = 0; i < dimNum - 1; ++i) {
        channels *= static_cast<uint64_t>(origin.GetDim(i));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *ksize_ptr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride_ptr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding_ptr = attrs->GetAttrPointer<int32_t>(2);

    uint32_t ksize = static_cast<uint32_t>(*ksize_ptr);
    uint32_t stride = static_cast<uint32_t>(*stride_ptr);
    uint32_t padding = static_cast<uint32_t>(*padding_ptr);

    if (stride == 0) stride = ksize; // typical default if unspecified

    // Output length: floor((L + 2*padding - ksize)/stride) + 1; clamp to zero
    int64_t numer = static_cast<int64_t>(L) + 2 * static_cast<int64_t>(padding) - static_cast<int64_t>(ksize);
    uint64_t outL = 0;
    if (numer >= 0) {
        outL = static_cast<uint64_t>(numer / static_cast<int64_t>(stride)) + 1u;
    }

    // Memory model per tile (floats):
    //   inQueueX: BUFFER_NUM * needed = 2 * (tileOut * stride + ksize - 1)
    //   outQueueY: BUFFER_NUM * tileOut = 2 * tileOut
    //   tmpQueue: 1 * tileOut (for potential scratch)
    // Total floats = (2*(stride+1)+1) * tileOut + 2*(ksize - 1)
    // bytes = 4 * floats
    uint64_t budgetFloats = UB_BUDGET_BYTES / 4u;
    uint64_t constantTerm = 2ull * (ksize > 0 ? (ksize - 1u) : 0u);
    uint64_t coeff = 2ull * static_cast<uint64_t>(stride) + 3ull; // 2*(stride+1) + 1
    uint64_t maxTileOut = 0;
    if (budgetFloats > constantTerm && coeff > 0) {
        maxTileOut = (budgetFloats - constantTerm) / coeff;
    }
    if (maxTileOut == 0) maxTileOut = ALIGN_ELEMS;
    uint32_t tileOut = static_cast<uint32_t>(MaxU64(1, maxTileOut));
    tileOut = AlignDown(tileOut, ALIGN_ELEMS);
    if (tileOut == 0) tileOut = ALIGN_ELEMS;
    if (tileOut > HARD_CAP_ELEMS) tileOut = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    if (outL > 0 && outL < tileOut) {
        uint32_t aligned = AlignDown(static_cast<uint32_t>(outL), ALIGN_ELEMS);
        tileOut = (aligned > 0) ? aligned : static_cast<uint32_t>(outL);
    }

    // Total tiles across all channels
    uint64_t tilesPerCh = (tileOut == 0) ? 1ull : CeilDiv(static_cast<uint32_t>(outL), tileOut);
    uint64_t totalTiles = tilesPerCh * channels;

    uint32_t blockDim = static_cast<uint32_t>(totalTiles);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_channels(static_cast<uint32_t>(channels));
    tiling.set_L(static_cast<uint32_t>(L));
    tiling.set_outL(static_cast<uint32_t>(outL));
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);
    tiling.set_ksize(ksize);
    tiling.set_stride(stride);
    tiling.set_padding(padding);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    const auto &origin = x_shape->GetOriginShape();
    int32_t dimNum = static_cast<int32_t>(origin.GetDimNum());
    if (dimNum < 1) {
        return GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *ksize_ptr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride_ptr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding_ptr = attrs->GetAttrPointer<int32_t>(2);

    int64_t L = origin.GetDim(dimNum - 1);
    int32_t ksize = *ksize_ptr;
    int32_t stride = *stride_ptr == 0 ? *ksize_ptr : *stride_ptr;
    int32_t padding = *padding_ptr;

    int64_t numer = L + 2LL * padding - static_cast<int64_t>(ksize);
    int64_t outL = (numer >= 0) ? (numer / stride + 1) : 0;

    gert::Shape outShape = *x_shape;
    outShape.SetDim(dimNum - 1, outL);
    *y_shape = outShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t channels,
                                uint32_t L, uint32_t outL,
                                uint32_t tileOut,
                                uint32_t ksize,
                                uint32_t stride,
                                uint32_t padding)
    {
        channels_ = channels;
        L_ = L;
        outL_ = outL;
        tileOut_ = (tileOut == 0 ? 1u : tileOut);
        ksize_ = ksize;
        stride_ = (stride == 0 ? ksize_ : stride);
        padding_ = padding;

        // input segment length for the worst tile
        // neededLen = tileOut * stride + ksize - 1
        tileIn_ = tileOut_ * stride_ + (ksize_ > 0 ? (ksize_ - 1u) : 0u);

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)channels_ * L_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)channels_ * outL_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileIn_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileOut_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileOut_ * sizeof(float));

        negInf_ = -3.4028235e38f; // approx -FLT_MAX
    }

    __aicore__ inline void Process()
    {
        if (channels_ == 0 || L_ == 0 || outL_ == 0 || ksize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over channels
        for (uint32_t ch = blockIdx; ch < channels_; ch += blockNum) {
            uint64_t xChBase = (uint64_t)ch * L_;
            uint64_t yChBase = (uint64_t)ch * outL_;

            for (uint32_t outBase = 0; outBase < outL_; outBase += tileOut_) {
                uint32_t curOut = outL_ - outBase;
                if (curOut > tileOut_) curOut = tileOut_;

                int64_t start = (int64_t)outBase * (int64_t)stride_ - (int64_t)padding_;
                uint32_t needed = curOut * stride_ + (ksize_ > 0 ? (ksize_ - 1u) : 0u);
                int64_t segStart = start < 0 ? 0 : start;
                int64_t segEnd = start + (int64_t)needed;
                if (segEnd > (int64_t)L_) segEnd = (int64_t)L_;
                uint32_t segLen = (uint32_t)(segEnd - segStart);
                uint32_t leftPad = (uint32_t)(segStart - start); // if start < 0, leftPad > 0

                // Load input segment into UB with -INF padding on both sides
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::Dup(xLocal, negInf_, needed);
                if (segLen > 0) {
                    AscendC::DataCopy(xLocal[leftPad], xGm_[xChBase + (uint64_t)segStart], segLen);
                }
                inQueueX_.EnQue(xLocal);

                // Compute max pooling; optimized for stride == 1
                AscendC::LocalTensor<float> xin = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                if (stride_ == 1) {
                    // Initialize with first window slice
                    AscendC::DataCopy(yLocal, xin[leftPad], curOut);
                    // Merge remaining k-1 slices by elementwise max
                    for (uint32_t j = 1; j < ksize_; ++j) {
                        AscendC::Max(yLocal, yLocal, xin[leftPad + j], curOut);
                    }
                } else {
                    // Generic stride path: gather into tmp buffer then reduce by max
                    // NOTE: This path may be slower and uses simple looping.
                    AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                    // Initialize yLocal with -INF
                    AscendC::Dup(yLocal, negInf_, curOut);

                    for (uint32_t j = 0; j < ksize_; ++j) {
                        // Gather spaced elements with step=stride_ into tLocal
                        // tLocal[i] = xin[leftPad + j + i*stride_], i in [0, curOut)
                        for (uint32_t i = 0; i < curOut; ++i) {
                            // scalar read/write via small 1-len copies
                            AscendC::DataCopy(tLocal[i], xin[leftPad + j + i * stride_], 1);
                        }
                        AscendC::Max(yLocal, yLocal, tLocal, curOut);
                    }
                    tmpQueue_.FreeTensor(tLocal);
                }

                outQueueY_.EnQue(yLocal);
                inQueueX_.FreeTensor(xin);

                // Store results
                AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[yChBase + (uint64_t)outBase], yOut, curOut);
                outQueueY_.FreeTensor(yOut);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t channels_{0};
    uint32_t L_{0};
    uint32_t outL_{0};
    uint32_t tileOut_{0};
    uint32_t tileIn_{0};
    uint32_t ksize_{0};
    uint32_t stride_{1};
    uint32_t padding_{0};
    float negInf_{-3.4028235e38f};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y,
            tiling_data.channels, tiling_data.L, tiling_data.outL,
            tiling_data.tileOut, tiling_data.ksize, tiling_data.stride, tiling_data.padding);
    op.Process();
}
