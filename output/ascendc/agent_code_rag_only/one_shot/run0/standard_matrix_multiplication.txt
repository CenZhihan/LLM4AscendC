host_operator_src="""

#include "standard_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
static inline uint32_t AlignUp(uint32_t x, uint32_t a) {
    return ((x + a - 1u) / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model (float32):
//   A_tile: tileM * kChunk
//   B_tile: kChunk * tileN
//   C_tile: tileM * tileN
//   tmpVec: tileN
// Total floats = tileM*kChunk + kChunk*tileN + tileM*tileN + tileN
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 16;              // align tile edges to 16 for better DMA/vec efficiency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    StandardMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    // Expect A: [M, K], B: [K, N]
    const uint32_t M = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(1));

    // Budget in floats
    uint32_t budgetFloats = UB_BUDGET_BYTES / 4u;

    // Start with square-ish tiles, aligned, then compute kChunk from budget.
    uint32_t tileM = 64u;
    uint32_t tileN = 64u;
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    tileN = AlignDown(tileN, ALIGN_ELEMS);
    if (tileM == 0) tileM = ALIGN_ELEMS;
    if (tileN == 0) tileN = ALIGN_ELEMS;

    if (M < tileM) tileM = (M >= ALIGN_ELEMS) ? AlignDown(M, ALIGN_ELEMS) : M;
    if (N < tileN) tileN = (N >= ALIGN_ELEMS) ? AlignDown(N, ALIGN_ELEMS) : N;

    // Ensure at least 1
    if (tileM == 0) tileM = 1;
    if (tileN == 0) tileN = 1;

    uint32_t minKChunk = ALIGN_ELEMS;
    if (K < minKChunk) minKChunk = (K >= ALIGN_ELEMS) ? AlignDown(K, ALIGN_ELEMS) : K;
    if (minKChunk == 0) minKChunk = 1;

    // From budget: tileM*kChunk + kChunk*tileN + tileM*tileN + tileN <= budget
    uint32_t maxKChunk = 0;
    if (budgetFloats > (tileM * tileN + tileN)) {
        uint32_t remain = budgetFloats - (tileM * tileN + tileN);
        uint32_t denom = (tileM + tileN);
        if (denom > 0) {
            maxKChunk = remain / denom;
            maxKChunk = AlignDown(maxKChunk, ALIGN_ELEMS);
        }
    }
    if (maxKChunk == 0) maxKChunk = minKChunk;
    if (maxKChunk > K) maxKChunk = AlignDown(K, ALIGN_ELEMS);
    if (maxKChunk == 0) maxKChunk = (K >= ALIGN_ELEMS) ? ALIGN_ELEMS : K;

    // Compute blocks: tile over MxN grid
    const uint32_t tilesM = CeilDiv(M, tileM);
    const uint32_t tilesN = CeilDiv(N, tileN);
    const uint32_t totalTiles = tilesM * tilesN;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_kChunk(maxKChunk);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

// A simple tiled GEMM kernel: C[M,N] = A[M,K] x B[K,N]
// Data type: float32
// Tile strategy: each block computes multiple (tileM x tileN) tiles of C via grid-stride over tile grid.

class KernelStandardMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelStandardMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileN, uint32_t kChunk)
    {
        M_ = M; K_ = K; N_ = N;
        tM_ = tileM; tN_ = tileN; kC_ = kChunk;

        aGm_.SetGlobalBuffer((__gm__ float*)a, static_cast<uint64_t>(M_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, static_cast<uint64_t>(K_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, static_cast<uint64_t>(M_) * N_);

        // UB buffers
        // A_tile: tM_ * kC_
        // B_tile: kC_ * tN_
        // C_tile: tM_ * tN_
        // tmpVec: tN_
        pipe_.InitBuffer(aQueue_, 1, tM_ * kC_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, 1, kC_ * tN_ * sizeof(float));
        pipe_.InitBuffer(cQueue_, 1, tM_ * tN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t tilesM = (M_ + tM_ - 1u) / tM_;
        const uint32_t tilesN = (N_ + tN_ - 1u) / tN_;
        const uint32_t totalTiles = tilesM * tilesN;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tileIndex = blockIdx; tileIndex < totalTiles; tileIndex += blockNum) {
            uint32_t tileRow = tileIndex % tilesM;
            uint32_t tileCol = tileIndex / tilesM;

            uint32_t mBase = tileRow * tM_;
            uint32_t nBase = tileCol * tN_;

            uint32_t curM = M_ - mBase;
            if (curM > tM_) curM = tM_;
            uint32_t curN = N_ - nBase;
            if (curN > tN_) curN = tN_;

            ComputeTile(mBase, nBase, curM, curN);
        }
    }

private:
    __aicore__ inline void ZeroOut(AscendC::LocalTensor<float>& buf, uint32_t len)
    {
        AscendC::Duplicate(buf, 0.0f, len);
    }

    __aicore__ inline void LoadATile(uint32_t mBase, uint32_t kBase,
                                     uint32_t curM, uint32_t curK,
                                     AscendC::LocalTensor<float>& aTile)
    {
        // Copy A submatrix rows: each row length curK
        for (uint32_t r = 0; r < curM; ++r) {
            uint64_t gmOffset = (static_cast<uint64_t>(mBase + r) * K_) + kBase;
            AscendC::DataCopy(aTile[r * curK], aGm_[gmOffset], curK);
        }
    }

    __aicore__ inline void LoadBTile(uint32_t kBase, uint32_t nBase,
                                     uint32_t curK, uint32_t curN,
                                     AscendC::LocalTensor<float>& bTile)
    {
        // Copy B submatrix rows: each row length curN
        for (uint32_t kk = 0; kk < curK; ++kk) {
            uint64_t gmOffset = (static_cast<uint64_t>(kBase + kk) * N_) + nBase;
            AscendC::DataCopy(bTile[kk * curN], bGm_[gmOffset], curN);
        }
    }

    __aicore__ inline float ReadLocalElem(const AscendC::LocalTensor<float>& t, uint32_t idx) {
        // In practice, AscendC provides ways to read a scalar from UB.
        // Here we assume a helper API GetValue exists.
        return t.GetValue(idx);
    }

    __aicore__ inline void ComputeTile(uint32_t mBase, uint32_t nBase, uint32_t curM, uint32_t curN)
    {
        AscendC::LocalTensor<float> cTile = cQueue_.AllocTensor<float>();
        ZeroOut(cTile, curM * curN);

        for (uint32_t kBase = 0; kBase < K_; kBase += kC_) {
            uint32_t curK = K_ - kBase;
            if (curK > kC_) curK = kC_;

            AscendC::LocalTensor<float> aTile = aQueue_.AllocTensor<float>();
            AscendC::LocalTensor<float> bTile = bQueue_.AllocTensor<float>();
            LoadATile(mBase, kBase, curM, curK, aTile);
            LoadBTile(kBase, nBase, curK, curN, bTile);

            AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

            for (uint32_t r = 0; r < curM; ++r) {
                AscendC::LocalTensor<float> cRow = cTile[r * curN];
                for (uint32_t kk = 0; kk < curK; ++kk) {
                    // aVal = A[r, kk]
                    const uint32_t aIdx = r * curK + kk;
                    float aVal = ReadLocalElem(aTile, aIdx);

                    // tmp = B[kk, :] * aVal
                    AscendC::LocalTensor<float> bRow = bTile[kk * curN];
                    AscendC::Muls(tmp, bRow, aVal, curN);

                    // cRow += tmp
                    AscendC::Add(cRow, cRow, tmp, curN);
                }
            }

            tmpQueue_.FreeTensor(tmp);
            bQueue_.FreeTensor(bTile);
            aQueue_.FreeTensor(aTile);
        }

        // Store C tile back to GM
        for (uint32_t r = 0; r < curM; ++r) {
            uint64_t gmOffset = (static_cast<uint64_t>(mBase + r) * N_) + nBase;
            AscendC::DataCopy(cGm_[gmOffset], cTile[r * curN], curN);
        }

        cQueue_.FreeTensor(cTile);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> cQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0};
    uint32_t K_{0};
    uint32_t N_{0};
    uint32_t tM_{0};
    uint32_t tN_{0};
    uint32_t kC_{0};
};

extern "C" __global__ __aicore__ void standard_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelStandardMatrixMultiplicationCustom op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileN, tiling_data.kChunk);
    op.Process();
}
