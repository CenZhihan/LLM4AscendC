namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative UB per-core
static constexpr uint32_t MIN_ALIGN = 256;              // favor larger vector DMA bursts
static constexpr uint32_t MAX_BLOCKS = 24;              // typical safe upper bound
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dSquareInputSquareKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect NCHW input and weight [C,1,3,3]
    const uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t C = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetDim(3));
    // kernel must be 3x3, depth_multiplier=1
    (void)w_shape;

    const uint32_t stride_h = 1;
    const uint32_t stride_w = 1;
    const uint32_t pad_h = 0;
    const uint32_t pad_w = 0;
    const uint32_t dilation_h = 1;
    const uint32_t dilation_w = 1;

    // Output dims (no padding/dilation, stride=1), kernel=3x3
    const uint32_t outH = (H + 2 * pad_h - dilation_h * (3 - 1) - 1) / stride_h + 1;
    const uint32_t outW = (W + 2 * pad_w - dilation_w * (3 - 1) - 1) / stride_w + 1;

    // UB budget for two float vectors (y and tmp), each of length tileW
    // bytes = 2 * tileW * sizeof(float)
    uint32_t maxTileW = UB_BUDGET_BYTES / (2u * sizeof(float));
    maxTileW = AlignDown(maxTileW, MIN_ALIGN);
    if (maxTileW < MIN_ALIGN) maxTileW = MIN_ALIGN;
    if (maxTileW > outW) {
        uint32_t aligned = AlignDown(outW, MIN_ALIGN);
        maxTileW = (aligned > 0) ? aligned : outW;
    }
    const uint32_t tileW = maxTileW;

    // Blocks distribute across N*C channels
    uint32_t blockDim = N * C;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t outH, uint32_t outW, uint32_t tileW)
    {
        N_ = N; C_ = C; H_ = H; W_ = W; outH_ = outH; outW_ = outW; tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * C_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(C_) * 3 * 3);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * C_ * outH_ * outW_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileW_ * sizeof(float));

        // Small buffer for 3x3 filter (9 floats)
        pipe_.InitBuffer(filterQueue_, 1, 9 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || H_ < 3 || W_ < 3) { return; }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalPairs = static_cast<uint64_t>(N_) * C_;  // each (n,c) is independent
        for (uint64_t pair = blockIdx; pair < totalPairs; pair += blockNum) {
            const uint32_t n = static_cast<uint32_t>(pair / C_);
            const uint32_t c = static_cast<uint32_t>(pair % C_);

            // Load filter of channel c: weight shape [C,1,3,3] -> contiguous 9 floats per c
            float wf[9];
            {
                AscendC::LocalTensor<float> fLocal = filterQueue_.AllocTensor<float>();
                const uint64_t wOffset = static_cast<uint64_t>(c) * 9u;
                AscendC::DataCopy(fLocal, wGm_[wOffset], 9u);
                filterQueue_.FreeTensor(fLocal); // values copied into wf by element-wise reads below

                // Read back 9 filter scalars from GM via small copies to registers
                // Note: local to register read is compiler handled
                // Since we freed fLocal to reuse buffer, we reload scalars individually
                // To simplify, we directly copy each scalar to tmp buffer and read.
            }
            // Fallback: directly fetch each scalar into a 1-element local then to register
            for (int i = 0; i < 9; ++i) {
                AscendC::LocalTensor<float> tmp1 = filterQueue_.AllocTensor<float>();
                const uint64_t wOff = static_cast<uint64_t>(c) * 9u + static_cast<uint64_t>(i);
                AscendC::DataCopy(tmp1, wGm_[wOff], 1u);
                // Access first element; compiler places into register
                wf[i] = tmp1.GetValue(0);
                filterQueue_.FreeTensor(tmp1);
            }

            for (uint32_t oh = 0; oh < outH_; ++oh) {
                for (uint32_t ow0 = 0; ow0 < outW_; ow0 += tileW_) {
                    const uint32_t len = (ow0 + tileW_ <= outW_) ? tileW_ : (outW_ - ow0);

                    // Allocate output local buffer
                    AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                    // Compute first term to initialize yLocal:
                    // r=0,k=0 -> input row = oh + 0, col = ow0 + 0
                    {
                        AscendC::LocalTensor<float> seg = tmpQueue_.AllocTensor<float>();
                        const uint64_t xOff = OffsetX(n, c, oh + 0, ow0 + 0);
                        AscendC::DataCopy(seg, xGm_[xOff], len);
                        AscendC::Muls(yLocal, seg, wf[0], len);
                        tmpQueue_.FreeTensor(seg);
                    }

                    // Accumulate remaining 8 terms
                    // (r,k): (0,1)->wf[1], (0,2)->wf[2], (1,0)->wf[3], (1,1)->wf[4], (1,2)->wf[5], (2,0)->wf[6], (2,1)->wf[7], (2,2)->wf[8]
                    const uint32_t rIdx[8] = {0,0,1,1,1,2,2,2};
                    const uint32_t kIdx[8] = {1,2,0,1,2,0,1,2};
                    for (int t = 0; t < 8; ++t) {
                        AscendC::LocalTensor<float> seg = tmpQueue_.AllocTensor<float>();
                        const uint32_t r = rIdx[t];
                        const uint32_t k = kIdx[t];
                        const uint64_t xOff = OffsetX(n, c, oh + r, ow0 + k);
                        AscendC::DataCopy(seg, xGm_[xOff], len);
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::Muls(tmp, seg, wf[(r * 3u + k)], len);
                        AscendC::Add(yLocal, yLocal, tmp, len);
                        tmpQueue_.FreeTensor(tmp);
                        tmpQueue_.FreeTensor(seg);
                    }

                    // Store yLocal back to GM
                    const uint64_t yOff = OffsetY(n, c, oh, ow0);
                    AscendC::DataCopy(yGm_[yOff], yLocal, len);
                    outQueueY_.FreeTensor(yLocal);
                }
            }
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        // NCHW layout
        return static_cast<uint64_t>(n) * C_ * H_ * W_
             + static_cast<uint64_t>(c) * H_ * W_
             + static_cast<uint64_t>(h) * W_
             + static_cast<uint64_t>(w);
    }

    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        // NCHW layout
        return static_cast<uint64_t>(n) * C_ * outH_ * outW_
             + static_cast<uint64_t>(c) * outH_ * outW_
             + static_cast<uint64_t>(h) * outW_
             + static_cast<uint64_t>(w);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> filterQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.outH, tiling_data.outW, tiling_data.tileW);
    op.Process();
}
