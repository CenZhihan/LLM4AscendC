host_operator_src="""

#include "matmul_mish_mish_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// UB estimation:
// Buffers per core iteration (single-row, tileN columns):
//   xTile: 1 * tileK
//   wTile: tileN * tileK
//   mulBuf: 1 * tileK
//   accVec: tileN
//   biasVec: tileN
//   tmpVec: tileN
// Total floats ~= (tileN + 2) * tileK + 3 * tileN
// Bytes ~= 4 * ((tileN + 2) * tileK + 3 * tileN)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 512;
static constexpr uint32_t HARD_CAP_TILEK = 16384;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMishMishCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    // Expect:
    //   x: [B, K]
    //   weight: [N, K]
    //   bias: [N]
    const uint32_t B = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));

    // Choose tileN conservatively to keep wTile fit good
    uint32_t tileN = 4u;
    if (N < tileN) tileN = (N == 0 ? 1u : N);

    // Compute tileK from UB budget
    uint32_t usable = UB_BUDGET_BYTES - 8 * 1024; // small safety margin
    uint32_t denom = 4u * (tileN + 2u); // bytes per element times (tileN + 2) for xTile + wTile + mulBuf
    uint32_t tileK = usable / denom;
    tileK = AlignDown(tileK, ALIGN_ELEMS);
    if (tileK == 0) tileK = ALIGN_ELEMS;
    if (tileK > HARD_CAP_TILEK) tileK = AlignDown(HARD_CAP_TILEK, ALIGN_ELEMS);
    if (tileK > K) {
        uint32_t aligned = AlignDown(K, ALIGN_ELEMS);
        tileK = (aligned > 0) ? aligned : K;
    }

    // Block mapping over batch rows
    uint32_t blockDim = B;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_batch(B);
    tiling.set_inFeatures(K);
    tiling.set_outFeatures(N);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *x_shape = context->GetInputShape(0);   // [B, K]
    const gert::Shape *w_shape = context->GetInputShape(1);   // [N, K]

    const int64_t B = x_shape->GetOriginShape().GetDim(0);

    gert::Shape out;
    out.SetDimNum(2);
    out.SetDim(0, B);
    out.SetDim(1, N);

    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulMishMishCustom {
public:
    __aicore__ inline KernelMatmulMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t B, uint32_t K, uint32_t N,
                                uint32_t tileK, uint32_t tileN)
    {
        B_ = B;
        K_ = K;
        N_ = N;
        tileK_ = tileK;
        tileN_ = tileN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(B_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(B_) * N_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, tileN_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 2, tileN_ * sizeof(float));
        pipe_.InitBuffer(mulQueue_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(biasQueue_, BUFFER_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (B_ == 0 || K_ == 0 || N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t r = blockIdx; r < B_; r += (blockNum == 0 ? 1u : blockNum)) {
            // Process output columns in tiles of tileN_
            for (uint32_t o0 = 0; o0 < N_; o0 += tileN_) {
                const uint32_t lenN = (o0 + tileN_ <= N_) ? tileN_ : (N_ - o0);

                // init accumulators
                AscendC::LocalTensor<float> acc = accQueue_.AllocTensor<float>();
                AscendC::Dup(acc, 0.0f, lenN);

                // Iterate K in chunks
                for (uint32_t k0 = 0; k0 < K_; k0 += tileK_) {
                    const uint32_t lenK = (k0 + tileK_ <= K_) ? tileK_ : (K_ - k0);

                    // Load x row chunk
                    CopyInX(r, k0, lenK);

                    // Load weight tile (lenN rows, lenK cols)
                    CopyInW(o0, k0, lenN, lenK);

                    AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                    AscendC::LocalTensor<float> wLocal = inQueueW_.DeQue<float>();

                    // For each output column in the tile, compute partial dot
                    for (uint32_t c = 0; c < lenN; ++c) {
                        AscendC::LocalTensor<float> mulBuf = mulQueue_.AllocTensor<float>();
                        AscendC::LocalTensor<float> wRow = wLocal[c * tileK_];
                        AscendC::Mul(mulBuf, xLocal, wRow, lenK);

                        // Reduce sum mulBuf to mulBuf[0]
                        ReduceSumInplace(mulBuf, lenK);

                        // acc[c] += mulBuf[0]
                        AscendC::Add(acc[c], acc[c], mulBuf, 1);

                        mulQueue_.FreeTensor(mulBuf);
                    }

                    inQueueW_.FreeTensor(wLocal);
                    inQueueX_.FreeTensor(xLocal);
                }

                // Add bias
                AscendC::LocalTensor<float> biasLocal = biasQueue_.AllocTensor<float>();
                AscendC::DataCopy(biasLocal, bGm_[o0], lenN);
                AscendC::Add(acc, acc, biasLocal, lenN);
                biasQueue_.FreeTensor(biasLocal);

                // Apply Mish twice: y = mish(mish(acc))
                ApplyMishInplace(acc, lenN);
                ApplyMishInplace(acc, lenN);

                // Write out contiguous results for this row and tile of columns
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, acc, lenN);
                outQueueY_.EnQue(yLocal);

                // GM store
                AscendC::LocalTensor<float> yLocalOut = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[static_cast<uint64_t>(r) * N_ + o0], yLocalOut, lenN);
                outQueueY_.FreeTensor(yLocalOut);

                accQueue_.FreeTensor(acc);
            }
        }
    }

private:
    __aicore__ inline void CopyInX(uint32_t r, uint32_t k0, uint32_t lenK)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[static_cast<uint64_t>(r) * K_ + k0], lenK);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyInW(uint32_t o0, uint32_t k0, uint32_t lenN, uint32_t lenK)
    {
        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
        // Pack lenN rows, each of length lenK, into wLocal contiguous by rows
        for (uint32_t c = 0; c < lenN; ++c) {
            const uint64_t wOffset = static_cast<uint64_t>(o0 + c) * K_ + k0;
            AscendC::DataCopy(wLocal[c * tileK_], wGm_[wOffset], lenK);
        }
        inQueueW_.EnQue(wLocal);
    }

    __aicore__ inline void ReduceSumInplace(AscendC::LocalTensor<float>& buf, uint32_t len)
    {
        uint32_t cur = len;
        while (cur > 1u) {
            uint32_t half = cur >> 1;
            // sum first half with second half
            AscendC::Add(buf, buf, buf[half], half);
            if (cur & 1u) {
                // add the last odd element into position 0
                AscendC::Add(buf, buf, buf[cur - 1], 1);
                cur = half + 1u;
            } else {
                cur = half;
            }
        }
    }

    __aicore__ inline void ApplyMishInplace(AscendC::LocalTensor<float>& vec, uint32_t len)
    {
        // vec = vec * tanh(ln(1 + exp(vec)))
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
        AscendC::Exp(tmp, vec, len);
        AscendC::Adds(tmp, tmp, 1.0f, len);
        AscendC::Ln(tmp, tmp, len);
        AscendC::Tanh(tmp, tmp, len);
        AscendC::Mul(vec, vec, tmp, len);
        tmpQueue_.FreeTensor(tmp);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outQueueY_;

    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> mulQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> biasQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t B_{0};
    uint32_t K_{0};
    uint32_t N_{0};
    uint32_t tileK_{0};
    uint32_t tileN_{0};
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMishMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.batch, tiling_data.inFeatures, tiling_data.outFeatures,
            tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
