    const int64_t Kh   = w_shape->GetDim(2);

    if (Cin != C) {

    // Attributes
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *poolKPtr    = attrs->GetAttrPointer<int64_t>(1);
    const int64_t poolK = *poolKPtr;

    const int64_t OH = H - Kh + 1;
    const int64_t OW = W - Kw + 1;

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

class KernelConv2dSubtractHardSwishMaxPoolMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractHardSwishMaxPoolMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t Cin, uint32_t Kh, uint32_t Kw,
                                uint32_t OH, uint32_t OW, uint32_t Hp, uint32_t Wp,
                                uint32_t poolK, float subtractValue)
    {
        x_ = reinterpret_cast<__gm__ float*>(x);
        w_ = reinterpret_cast<__gm__ float*>(w);
        b_ = reinterpret_cast<__gm__ float*>(b);
        y_ = reinterpret_cast<__gm__ float*>(y);

        N_ = N; C_ = C; H_ = H; W_ = W;
        Cout_ = Cout; Cin_ = Cin; Kh_ = Kh; Kw_ = Kw;
        OH_ = OH; OW_ = OW; Hp_ = Hp; Wp_ = Wp;
        poolK_ = poolK; subtractValue_ = subtractValue;
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || Cout_ == 0 || Kh_ == 0 || Kw_ == 0 || Hp_ == 0 || Wp_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t oc = blockIdx; oc < Cout_; oc += (blockNum == 0 ? 1u : blockNum)) {
            for (uint32_t n = 0; n < N_; ++n) {
                for (uint32_t yp = 0; yp < Hp_; ++yp) {
                    for (uint32_t xp = 0; xp < Wp_; ++xp) {
                        float max_val = -INFINITY;

                        // Pool window over post-conv activations
                        for (uint32_t py = 0; py < poolK_; ++py) {
                            const uint32_t oy = yp * poolK_ + py;
                            for (uint32_t px = 0; px < poolK_; ++px) {
                                const uint32_t ox = xp * poolK_ + px;

                                // Direct conv2d at output (oy, ox) for (n, oc)
                                float acc = 0.0f;
                                if (b_ != nullptr) {
                                    acc = b_[oc];
                                }

                                for (uint32_t ic = 0; ic < Cin_; ++ic) {
                                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                                        const uint32_t in_y = oy + kh;
                                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                            const uint32_t in_x = ox + kw;

                                            const uint64_t x_idx =
                                                (((uint64_t)n * C_ + ic) * H_ + in_y) * W_ + in_x;
                                            const uint64_t w_idx =
                                                ((((uint64_t)oc * Cin_ + ic) * Kh_ + kh) * Kw_ + kw);

                                            const float xv = x_[x_idx];
                                            const float wv = w_[w_idx];
                                            acc += xv * wv;
                                        }
                                    }
                                }

                                // Subtract, then HardSwish
                                const float sub_v = acc - subtractValue_;
                                const float hs = hard_swish(sub_v);

                                // Max pool update
                                if (hs > max_val) {
                                    max_val = hs;
                                }
                            }
                        }

                        // Mish on pooled value
                        const float outv = mish(max_val);

                        const uint64_t y_idx =
                            (((uint64_t)n * Cout_ + oc) * Hp_ + yp) * Wp_ + xp;
                        y_[y_idx] = outv;
                    }
                }
            }
        }
    }

private:
    __aicore__ inline float hard_swish(float v) const
    {
        float t = v + 3.0f;
        if (t < 0.0f) t = 0.0f;
        if (t > 6.0f) t = 6.0f;
        return v * (t * (1.0f / 6.0f));
    }

    __aicore__ inline float mish(float x) const
    {
        // mish(x) = x * tanh(softplus(x)); softplus(x)=ln(1+exp(x))
        // Basic scalar implementation
        float expx = std::exp(x);
        float sp = std::log1p(expx);
        float th = std::tanh(sp);
        return x * th;
    }

private:
    __gm__ float *x_ {nullptr};
    __gm__ float *w_ {nullptr};
    __gm__ float *b_ {nullptr};
    __gm__ float *y_ {nullptr};

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, Cin_{0}, Kh_{0}, Kw_{0};
    uint32_t OH_{0}, OW_{0}, Hp_{0}, Wp_{0};
    uint32_t poolK_{0};
    float subtractValue_{0.f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(GM_ADDR x,
                                                                                      GM_ADDR weight,
                                                                                      GM_ADDR conv_bias,
                                                                                      GM_ADDR y,
                                                                                      GM_ADDR workspace,
                                                                                      GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractHardSwishMaxPoolMishCustom ker;
    ker.Init(x, weight, conv_bias, y,
             tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
             tiling_data.Cout, tiling_data.Cin, tiling_data.Kh, tiling_data.Kw,
             tiling_data.OH, tiling_data.OW, tiling_data.Hp, tiling_data.Wp,
             tiling_data.poolK, tiling_data.subtractValue);
    ker.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline std::vector<int64_t> infer_out_shape(const at::Tensor& x, const at::Tensor& w, int64_t pool_k) {
    // x: [N, C, H, W], w: [Cout, Cin, Kh, Kw]
    auto sizes_x = x.sizes();
    auto sizes_w = w.sizes();
    int64_t N = sizes_x[0];
    int64_t C = sizes_x[1];
    int64_t H = sizes_x[2];
    int64_t W = sizes_x[3];

    int64_t Cout = sizes_w[0];
    int64_t Cin  = sizes_w[1];
    int64_t Kh   = sizes_w[2];
    int64_t Kw   = sizes_w[3];
    TORCH_CHECK(C == Cin, "Input channels mismatch: x C=", C, ", weight Cin=", Cin);

    int64_t OH = H - Kh + 1;
    int64_t OW = W - Kw + 1;
    TORCH_CHECK(OH > 0 && OW > 0, "Invalid conv output spatial size: (", OH, ",", OW, ")");

    TORCH_CHECK(pool_k > 0, "pool_kernel_size must be > 0");
    int64_t Hp = OH / pool_k;
    int64_t Wp = OW / pool_k;
    TORCH_CHECK(Hp >= 0 && Wp >= 0, "Invalid pooled size");

    return {N, Cout, Hp, Wp};
}

at::Tensor conv2d_subtract_hard_swish_max_pool_mish_impl_npu(const at::Tensor& x,
                                                             const at::Tensor& weight,
                                                             c10::optional<at::Tensor> conv_bias_opt,
                                                             double subtract_value,
                                                             int64_t pool_kernel_size) {
    std::vector<int64_t> out_shape = infer_out_shape(x, weight, pool_kernel_size);
    at::Tensor result = at::empty(out_shape, x.options());
    at::Tensor bias_tensor;
    if (conv_bias_opt.has_value()) {
        bias_tensor = conv_bias_opt.value();

        bias_tensor = at::Tensor(); // undefined tensor
    }
    EXEC_NPU_CMD(aclnnConv2dSubtractHardSwishMaxPoolMishCustom, x, weight, bias_tensor, subtract_value, pool_kernel_size, result);
    return result;
}
