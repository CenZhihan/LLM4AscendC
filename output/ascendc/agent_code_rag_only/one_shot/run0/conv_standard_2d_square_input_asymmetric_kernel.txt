#include "conv_standard_2d_square_input_asymmetric_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}
}

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative UB share
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetOriginShape().GetDimNum() != 4 || w_shape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Ci = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t Hi = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t Wi = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const uint32_t Co = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t Ci_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    const uint32_t Kh = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(3));

    if (Ci_w != Ci) {
        return ge::GRAPH_FAILED;
    }
    if (Hi < Kh || Wi < Kw) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Ho = Hi - Kh + 1;
    const uint32_t Wo = Wi - Kw + 1;
    const uint32_t KLen = Ci * Kh * Kw;

    // UB model:
    //   - xVec: 1 * tileW
    //   - tmpVec: 1 * tileW
    //   - yAcc: 1 * tileW
    //   - yOut: 1 * tileW
    //   - weight buffer: KLen
    // Total bytes = 4 * tileW * 4 + KLen * 4
    const uint32_t overheadWeightsBytes = KLen * 4;
    uint32_t usable = (UB_BUDGET_BYTES > overheadWeightsBytes) ? (UB_BUDGET_BYTES - overheadWeightsBytes) : 0u;

    uint32_t tileW = (usable / (4u * 4u)); // floats
    tileW = AlignDown(tileW, ALIGN_ELEMS);
    if (tileW == 0) {
        tileW = (Wo >= ALIGN_ELEMS) ? ALIGN_ELEMS : Wo;
    }
    if (tileW > Wo) tileW = AlignDown(Wo, ALIGN_ELEMS);
    if (tileW == 0) tileW = Wo;

    const uint32_t rowsCount = N * Ho;
    uint32_t blockDim = rowsCount;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_ci(Ci);
    tiling.set_hi(Hi);
    tiling.set_wi(Wi);
    tiling.set_co(Co);
    tiling.set_kh(Kh);
    tiling.set_kw(Kw);
    tiling.set_ho(Ho);
    tiling.set_wo(Wo);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);
    tiling.set_kLen(KLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t ci, uint32_t hi, uint32_t wi,
                                uint32_t co, uint32_t kh, uint32_t kw,
                                uint32_t ho, uint32_t wo,
                                uint32_t tileW, uint32_t kLen)
    {
        n_ = n; ci_ = ci; hi_ = hi; wi_ = wi;
        co_ = co; kh_ = kh; kw_ = kw;
        ho_ = ho; wo_ = wo;
        tileW_ = tileW;
        kLen_ = kLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * ci_ * hi_ * wi_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(co_) * kLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * co_ * ho_ * wo_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        // Two calc buffers: yAcc and tmp
        pipe_.InitBuffer(calcQueue_, 2, tileW_ * sizeof(float));
        // One weight buffer sized for full KLen
        pipe_.InitBuffer(weightQueue_, 1, kLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || co_ == 0 || ho_ == 0 || wo_ == 0) { return; }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t rowsCount = n_ * ho_;
        for (uint32_t row = blockIdx; row < rowsCount; row += blockNum) {
            const uint32_t nIdx = row / ho_;
            const uint32_t hOut = row % ho_;

            for (uint32_t oc = 0; oc < co_; ++oc) {
                // Preload weights for this output channel to UB (KLen elements)
                AscendC::LocalTensor<float> wLocal = weightQueue_.AllocTensor<float>();
                AscendC::DataCopy(wLocal, wGm_[static_cast<uint64_t>(oc) * kLen_], kLen_);
                weightQueue_.EnQue(wLocal);
                wLocal = weightQueue_.DeQue<float>();

                for (uint32_t wBase = 0; wBase < wo_; wBase += tileW_) {
                    uint32_t curW = wo_ - wBase;
                    if (curW > tileW_) curW = tileW_;

                    // Allocate accumulators
                    AscendC::LocalTensor<float> yAcc = calcQueue_.AllocTensor<float>();
                    AscendC::LocalTensor<float> tmp = calcQueue_.AllocTensor<float>();
                    AscendC::Dup(yAcc, 0.0f, curW);

                    // Convolution accumulation over Ci, Kh, Kw
                    for (uint32_t ic = 0; ic < ci_; ++ic) {
                        for (uint32_t kh = 0; kh < kh_; ++kh) {
                            const uint32_t inRow = (hOut + kh);
                            for (uint32_t kw = 0; kw < kw_; ++kw) {
                                // Load input stripe [curW] contiguous along width
                                AscendC::LocalTensor<float> xLoc = inQueueX_.AllocTensor<float>();

                                const uint64_t xOffset =
                                    (((static_cast<uint64_t>(nIdx) * ci_ + ic) * hi_ + inRow) * wi_) + (wBase + kw);

                                AscendC::DataCopy(xLoc, xGm_[xOffset], curW);
                                inQueueX_.EnQue(xLoc);
                                xLoc = inQueueX_.DeQue<float>();

                                // fetch scalar weight value from UB
                                const uint32_t kIndex = (ic * kh_ + kh) * kw_ + kw;
                                float wv = wLocal.GetValue(kIndex);

                                // tmp = xLoc * wv; yAcc += tmp
                                AscendC::Muls(tmp, xLoc, wv, curW);
                                AscendC::Add(yAcc, yAcc, tmp, curW);

                                inQueueX_.FreeTensor(xLoc);
                            }
                        }
                    }

                    // Write yAcc to GM
                    AscendC::LocalTensor<float> yOut = outQueueY_.AllocTensor<float>();
                    AscendC::DataCopy(yOut, yAcc, curW);
                    outQueueY_.EnQue(yOut);

                    AscendC::LocalTensor<float> yOutDeq = outQueueY_.DeQue<float>();
                    const uint64_t yOffset =
                        (((static_cast<uint64_t>(nIdx) * co_ + oc) * ho_ + hOut) * wo_) + wBase;
                    AscendC::DataCopy(yGm_[yOffset], yOutDeq, curW);
                    outQueueY_.FreeTensor(yOutDeq);

                    calcQueue_.FreeTensor(tmp);
                    calcQueue_.FreeTensor(yAcc);
                }

                weightQueue_.FreeTensor(wLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> calcQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, ci_{0}, hi_{0}, wi_{0};
    uint32_t co_{0}, kh_{0}, kw_{0};
    uint32_t ho_{0}, wo_{0};
    uint32_t tileW_{0}, kLen_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.ci, tiling_data.hi, tiling_data.wi,
            tiling_data.co, tiling_data.kh, tiling_data.kw,
            tiling_data.ho, tiling_data.wo,
            tiling_data.tileW, tiling_data.kLen);
    op.Process();
}
