//   xTile: 1 * tileK

//   scalars and misc: negligible
// Total bytes ~= 3 * tileK * 4
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);   // [N, K]
    const gert::Shape *w_shape = context->GetInputShape(1);   // [M, K]

    const uint32_t dimsX = static_cast<uint32_t>(x_shape->GetStorageShape().GetDimNum());
    const uint32_t dimsW = static_cast<uint32_t>(w_shape->GetStorageShape().GetDimNum());
    if (dimsX < 2 || dimsW < 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(0));
    const uint32_t Kx = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(1));
    const uint32_t M = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(0));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(1));
    if (Kx != Kw) {
        return ge::GRAPH_FAILED;

    const uint32_t K = Kx;

    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // elements

    maxTile = AlignDown(maxTile, ALIGN_ELEMS);

        tileK = (aligned > 0) ? aligned : K;

    if (blockDim < 8u) blockDim = 8u;

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t QNUM_X = 1;
constexpr int32_t QNUM_W = 1;
constexpr int32_t QNUM_TMP = 1;
constexpr int32_t QNUM_SCAL = 1;

class KernelMatmulScaleResidualAddClampLogSumExpMishCustom {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR scaling, GM_ADDR cmin, GM_ADDR cmax,
                                GM_ADDR y,
                                uint32_t N, uint32_t K, uint32_t M, uint32_t tileK)
    {
        N_ = N; K_ = K; M_ = M; tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(M_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, M_);
        sGm_.SetGlobalBuffer((__gm__ float*)scaling, 1);
        loGm_.SetGlobalBuffer((__gm__ float*)cmin, 1);
        hiGm_.SetGlobalBuffer((__gm__ float*)cmax, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_));

        pipe_.InitBuffer(qX_,   QNUM_X,   tileK_ * sizeof(float));
        pipe_.InitBuffer(qW_,   QNUM_W,   tileK_ * sizeof(float));
        pipe_.InitBuffer(qTmp_, QNUM_TMP, tileK_ * sizeof(float));
        pipe_.InitBuffer(qScal_, QNUM_SCAL, 4); // for 1-element local scalars
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || K_ == 0 || M_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Load global scalars once per block
        float sVal = LoadScalar(sGm_, 0);
        float loVal = LoadScalar(loGm_, 0);
        float hiVal = LoadScalar(hiGm_, 0);
        float factor = sVal + 1.0f;

        for (uint32_t row = blockIdx; row < N_; row += blockNum) {
            float m = -__builtin_inff();
            float lse = 0.0f;

            // Iterate over output channels j
            for (uint32_t j = 0; j < M_; ++j) {
                float acc = 0.0f;

                // K-loop in tiles
                for (uint32_t k0 = 0; k0 < K_; k0 += tileK_) {
                    uint32_t len = K_ - k0;
                    if (len > tileK_) len = tileK_;

                    AscendC::LocalTensor<float> xTile = qX_.AllocTensor<float>();
                    AscendC::LocalTensor<float> wTile = qW_.AllocTensor<float>();
                    AscendC::LocalTensor<float> tmp   = qTmp_.AllocTensor<float>();

                    // Copy tiles from GM
                    AscendC::DataCopy(xTile, xGm_[static_cast<uint64_t>(row) * K_ + k0], len);
                    AscendC::DataCopy(wTile, wGm_[static_cast<uint64_t>(j) * K_ + k0],   len);

                    // Elementwise product
                    AscendC::Mul(tmp, xTile, wTile, len);

                    // Reduce sum over tile to scalar (simple loop over local buffer)
                    float part = 0.0f;
                    for (uint32_t t = 0; t < len; ++t) {
                        part += tmp.GetValue(t);
                    }
                    acc += part;

                    qTmp_.FreeTensor(tmp);
                    qW_.FreeTensor(wTile);
                    qX_.FreeTensor(xTile);
                }

                // Add bias
                float bj = LoadScalar(bGm_, j);
                float val = (acc + bj) * factor;

                // Clamp
                if (val < loVal) val = loVal;
                if (val > hiVal) val = hiVal;

                // Streaming log-sum-exp update
                if (val <= m) {
                    lse += expf(val - m);
                } else {
                    lse = lse * expf(m - val) + 1.0f;
                    m = val;
                }
            }

            float yval = m + logf(lse);

            // Mish(y) = y * tanh(log(1 + exp(y)))
            float sp = logf(1.0f + expf(yval));
            float mish = yval * tanhf(sp);
            float outv = yval * mish;

            // write to GM
            AscendC::LocalTensor<float> yLoc = qScal_.AllocTensor<float>();
            yLoc.SetValue(0, outv);
            AscendC::DataCopy(yGm_[row], yLoc, 1);
            qScal_.FreeTensor(yLoc);
        }
    }

private:
    __aicore__ inline float LoadScalar(AscendC::GlobalTensor<float>& gm, uint32_t idx)
    {
        AscendC::LocalTensor<float> loc = qScal_.AllocTensor<float>();
        AscendC::DataCopy(loc, gm[idx], 1);
        float v = loc.GetValue(0);
        qScal_.FreeTensor(loc);
        return v;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  QNUM_X>   qX_;
    AscendC::TQue<AscendC::TPosition::VECIN,  QNUM_W>   qW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM_TMP> qTmp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM_SCAL> qScal_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> sGm_;
    AscendC::GlobalTensor<float> loGm_;
    AscendC::GlobalTensor<float> hiGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t M_{0};
    uint32_t tileK_{0};
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scaling, GM_ADDR clamp_min, GM_ADDR clamp_max,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelMatmulScaleResidualAddClampLogSumExpMishCustom op;
    op.Init(x, weight, bias, scaling, clamp_min, clamp_max, y,
            tilingData.N, tilingData.K, tilingData.M, tilingData.tileK);
    op.Process();
}
