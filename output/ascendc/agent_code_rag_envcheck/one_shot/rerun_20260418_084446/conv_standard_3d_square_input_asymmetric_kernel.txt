kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_ONE = 1;

class KernelConv3dAsym {
public:
    __aicore__ inline KernelConv3dAsym() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t W, uint32_t H, uint32_t D,
                                uint32_t Kw, uint32_t Kh, uint32_t Kd,
                                uint32_t outW, uint32_t outH, uint32_t outD,
                                uint32_t tileCout)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        W_ = W; H_ = H; D_ = D;
        Kw_ = Kw; Kh_ = Kh; Kd_ = Kd;
        outW_ = outW; outH_ = outH; outD_ = outD;
        tileCout_ = tileCout;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cin_ * W_ * H_ * D_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, cout_ * cin_ * Kw_ * Kh_ * Kd_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * cout_ * outW_ * outH_ * outD_);

        // UB buffers:
        // weights tile: tileCout * cin * Kw * Kh * Kd
        // x patch: cin * Kw * Kh * Kd
        // y accum: tileCout
        uint32_t wTileElems = tileCout_ * cin_ * Kw_ * Kh_ * Kd_;
        uint32_t xPatchElems = cin_ * Kw_ * Kh_ * Kd_;
        uint32_t yTileElems = tileCout_;

        pipe_.InitBuffer(wQueue_, BUFFER_ONE, wTileElems * sizeof(float));
        pipe_.InitBuffer(xQueue_, BUFFER_ONE, xPatchElems * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_ONE, yTileElems * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blkIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blkNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        if (cout_ == 0 || outW_ == 0 || outH_ == 0 || outD_ == 0) {
            return;
        }

        const uint32_t startC = blkIdx * tileCout_;
        if (startC >= cout_) {
            return;
        }
        const uint32_t endC = MinUint32(startC + tileCout_, cout_);
        const uint32_t curTileCout = endC - startC;

        // Copy weights tile to UB
        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        const uint32_t wStrideCout = cin_ * Kw_ * Kh_ * Kd_;
        const uint64_t wOffset = (uint64_t)startC * (uint64_t)wStrideCout;
        AscendC::DataCopy(wLocal, wGm_[wOffset], (uint32_t)(curTileCout * wStrideCout));
        wQueue_.EnQue(wLocal);

        // Iterate over N and output spatial positions, compute tileCout outputs per position
        for (uint32_t n = 0; n < n_; ++n) {
            for (uint32_t ow = 0; ow < outW_; ++ow) {
                for (uint32_t oh = 0; oh < outH_; ++oh) {
                    for (uint32_t od = 0; od < outD_; ++od) {
                        ComputePoint(n, ow, oh, od, startC, curTileCout);
                    }
                }
            }
        }

        // free weight local
        AscendC::LocalTensor<float> wLocalFree = wQueue_.DeQue<float>();
        wQueue_.FreeTensor(wLocalFree);
    }

private:
    __aicore__ inline uint32_t MinUint32(uint32_t a, uint32_t b) { return a < b ? a : b; }

    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t c, uint32_t w, uint32_t h, uint32_t d) const
    {
        // Layout [N, C, W, H, D], D is contiguous.
        const uint64_t strideD = 1ull;
        const uint64_t strideH = (uint64_t)D_;
        const uint64_t strideW = (uint64_t)H_ * (uint64_t)D_;
        const uint64_t strideC = (uint64_t)W_ * (uint64_t)H_ * (uint64_t)D_;
        const uint64_t strideN = (uint64_t)cin_ * (uint64_t)W_ * (uint64_t)H_ * (uint64_t)D_;
        return ((uint64_t)n * strideN) + ((uint64_t)c * strideC) + ((uint64_t)w * strideW) + ((uint64_t)h * strideH) + ((uint64_t)d * strideD);
    }

    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t c, uint32_t w, uint32_t h, uint32_t d) const
    {
        // Layout [N, C_out, W_out, H_out, D_out], D_out contiguous.
        const uint64_t strideD = 1ull;
        const uint64_t strideH = (uint64_t)outD_;
        const uint64_t strideW = (uint64_t)outH_ * (uint64_t)outD_;
        const uint64_t strideC = (uint64_t)outW_ * (uint64_t)outH_ * (uint64_t)outD_;
        const uint64_t strideN = (uint64_t)cout_ * (uint64_t)outW_ * (uint64_t)outH_ * (uint64_t)outD_;
        return ((uint64_t)n * strideN) + ((uint64_t)c * strideC) + ((uint64_t)w * strideW) + ((uint64_t)h * strideH) + ((uint64_t)d * strideD);
    }

    __aicore__ inline void ComputePoint(uint32_t n, uint32_t ow, uint32_t oh, uint32_t od,
                                        uint32_t startC, uint32_t curTileCout)
    {
        // Prepare x patch into UB: shape [Cin, Kw, Kh, Kd], Kd contiguous
        AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
        uint32_t idx = 0;
        for (uint32_t c = 0; c < cin_; ++c) {
            for (uint32_t kh = 0; kh < Kh_; ++kh) {
                for (uint32_t kw = 0; kw < Kw_; ++kw) {
                    const uint64_t xOff = XOffset(n, c, ow + kw, oh + kh, od);
                    AscendC::DataCopy(xLocal[idx], xGm_[xOff], Kd_);
                    idx += Kd_;
                }
            }
        }
        xQueue_.EnQue(xLocal);

        // Allocate y accum buffer and zero init
        AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
        AscendC::SetValue(yLocal, 0.0f, curTileCout);

        // Fetch weight tile
        AscendC::LocalTensor<float> wLocal = wQueue_.DeQue<float>();

        // Naive dot product: for each output channel in tile
        const uint32_t K = cin_ * Kw_ * Kh_ * Kd_;
        // Using pointer-like access through unary ops on slices
        for (uint32_t oc = 0; oc < curTileCout; ++oc) {
            float acc = 0.0f;
            // Accumulate over K
            // We'll read xLocal and wLocal as sequences via scalar access
            for (uint32_t k = 0; k < K; ++k) {
                // load x[k]
                AscendC::LocalTensor<float> xElem = xLocal[k];
                AscendC::LocalTensor<float> wElem = wLocal[oc * K + k];
                // Vector ops emulate scalar read: mul-acc across 1 element
                AscendC::Muls(xElem, xElem, 1.0f, 1);
                AscendC::Muls(wElem, wElem, 1.0f, 1);
                // Copy back to temp scalars
                // Note: In practice, use proper element read API. Here we use DataCopy for one element.
                float tmpx = 0.0f, tmpw = 0.0f;
                AscendC::DataCopy(&tmpx, xElem, 1);
                AscendC::DataCopy(&tmpw, wElem, 1);
                acc += tmpx * tmpw;
            }
            // store acc into yLocal[oc]
            AscendC::LocalTensor<float> yElem = yLocal[oc];
            // Copy scalar acc to UB location
            AscendC::DataCopy(yElem, &acc, 1);
        }

        // Write results to GM
        for (uint32_t oc = 0; oc < curTileCout; ++oc) {
            float val = 0.0f;
            AscendC::LocalTensor<float> yElem = yLocal[oc];
            AscendC::DataCopy(&val, yElem, 1);
            const uint64_t yOff = YOffset(n, startC + oc, ow, oh, od);
            AscendC::DataCopy(yGm_[yOff], &val, 1);
        }

        // release locals
        yQueue_.FreeTensor(yLocal);
        wQueue_.EnQue(wLocal);
        xQueue_.FreeTensor(xLocal);
        (void)startC;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_ONE> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_ONE> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_ONE> yQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t W_{0}, H_{0}, D_{0};
    uint32_t Kw_{0}, Kh_{0}, Kd_{0};
    uint32_t outW_{0}, outH_{0}, outD_{0};
    uint32_t tileCout_{1};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);
    KernelConv3dAsym op;
    op.Init(x, weight, y,
            t.n, t.cin, t.cout,
            t.w, t.h, t.d,
            t.kw, t.kh, t.kd,
            t.outw, t.outh, t.outd,
            t.tileCout);
    op.Process();
}
