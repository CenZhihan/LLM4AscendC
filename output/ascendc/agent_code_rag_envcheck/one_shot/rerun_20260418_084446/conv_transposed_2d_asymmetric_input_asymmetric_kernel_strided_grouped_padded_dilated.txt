#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Hin, uint32_t Win,
                                uint32_t Kh, uint32_t Kw,
                                uint32_t Groups, uint32_t CinPerGroup, uint32_t CoPerGroup, uint32_t Cout,
                                uint32_t Hout, uint32_t Wout,
                                uint32_t StrideH, uint32_t StrideW,
                                uint32_t PadH, uint32_t PadW,
                                uint32_t OutPadH, uint32_t OutPadW,
                                uint32_t DilH, uint32_t DilW,
                                uint64_t totalOutElems, uint32_t tileLen)
    {
        N_ = N; Cin_ = Cin; Hin_ = Hin; Win_ = Win;
        Kh_ = Kh; Kw_ = Kw;
        Groups_ = Groups; CinG_ = CinPerGroup; CoG_ = CoPerGroup; Cout_ = Cout;
        Hout_ = Hout; Wout_ = Wout;
        sH_ = StrideH; sW_ = StrideW;
        pH_ = PadH; pW_ = PadW;
        opH_ = OutPadH; opW_ = OutPadW;
        dH_ = DilH; dW_ = DilW;
        totalOut_ = totalOutElems;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * Cin_ * Hin_ * Win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Cin_) * CoG_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * Cout_ * Hout_ * Wout_);

        // UB buffers for scalar-element workflow
        pipe_.InitBuffer(calcQ_, 4, sizeof(float)); // for accum, x, w, tmp
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint64_t strideElems = static_cast<uint64_t>(blockNum) * tileLen_;
        uint64_t base = static_cast<uint64_t>(blockIdx) * tileLen_;

        while (base < totalOut_) {
            uint32_t curLen = tileLen_;
            if (base + curLen > totalOut_) curLen = static_cast<uint32_t>(totalOut_ - base);
            ComputeRange(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeRange(uint64_t base, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            const uint64_t outIdx = base + i;

            // Decode linear index to (n, oc, ho, wo)
            uint64_t t = outIdx;
            const uint32_t wo = static_cast<uint32_t>(t % Wout_); t /= Wout_;
            const uint32_t ho = static_cast<uint32_t>(t % Hout_); t /= Hout_;
            const uint32_t oc = static_cast<uint32_t>(t % Cout_); t /= Cout_;
            const uint32_t n  = static_cast<uint32_t>(t);

            const uint32_t g = oc / CoG_;
            const uint32_t oc_in_g = oc - g * CoG_;

            // UB accum = 0.0
            AscendC::LocalTensor<float> accum = calcQ_.AllocTensor<float>();
            AscendC::Dup(accum, 0.0f, 1);

            // Iterate over contributing inputs
            for (uint32_t kh = 0; kh < Kh_; ++kh) {
                int32_t num_h = static_cast<int32_t>(ho) + static_cast<int32_t>(pH_) - static_cast<int32_t>(kh) * static_cast<int32_t>(dH_);
                if (num_h % static_cast<int32_t>(sH_) != 0) { continue; }
                int32_t hi = num_h / static_cast<int32_t>(sH_);
                if (hi < 0 || hi >= static_cast<int32_t>(Hin_)) { continue; }

                for (uint32_t kw = 0; kw < Kw_; ++kw) {
                    int32_t num_w = static_cast<int32_t>(wo) + static_cast<int32_t>(pW_) - static_cast<int32_t>(kw) * static_cast<int32_t>(dW_);
                    if (num_w % static_cast<int32_t>(sW_) != 0) { continue; }
                    int32_t wi = num_w / static_cast<int32_t>(sW_);
                    if (wi < 0 || wi >= static_cast<int32_t>(Win_)) { continue; }

                    for (uint32_t ic_in_g = 0; ic_in_g < CinG_; ++ic_in_g) {
                        const uint32_t ic = g * CinG_ + ic_in_g;

                        // x index
                        uint64_t x_index = (((static_cast<uint64_t>(n) * Cin_ + ic) * Hin_ + static_cast<uint32_t>(hi)) * Win_ + static_cast<uint32_t>(wi));
                        // w index: [Cin, CoG, Kh, Kw]
                        uint64_t w_index = (((static_cast<uint64_t>(ic) * CoG_ + oc_in_g) * Kh_ + kh) * Kw_ + kw);

                        // Load x and w (1 element)
                        AscendC::LocalTensor<float> xLoc = calcQ_.AllocTensor<float>();
                        AscendC::LocalTensor<float> wLoc = calcQ_.AllocTensor<float>();
                        AscendC::DataCopy(xLoc, xGm_[x_index], 1);
                        AscendC::DataCopy(wLoc, wGm_[w_index], 1);

                        // tmp = x * w
                        AscendC::LocalTensor<float> tmp = calcQ_.AllocTensor<float>();
                        AscendC::Mul(tmp, xLoc, wLoc, 1);
                        // accum += tmp
                        AscendC::Add(accum, accum, tmp, 1);

                        // free temporaries
                        calcQ_.FreeTensor(tmp);
                        calcQ_.FreeTensor(wLoc);
                        calcQ_.FreeTensor(xLoc);
                    }
                }
            }

            // Store to output
            uint64_t y_index = (((static_cast<uint64_t>(n) * Cout_ + oc) * Hout_ + ho) * Wout_ + wo);
            AscendC::DataCopy(yGm_[y_index], accum, 1);
            calcQ_.FreeTensor(accum);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> calcQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Hin_{0}, Win_{0};
    uint32_t Kh_{0}, Kw_{0};
    uint32_t Groups_{0}, CinG_{0}, CoG_{0}, Cout_{0};
    uint32_t Hout_{0}, Wout_{0};
    uint32_t sH_{1}, sW_{1}, pH_{0}, pW_{0}, opH_{0}, opW_{0}, dH_{1}, dW_{1};

    uint64_t totalOut_{0};
    uint32_t tileLen_{1};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dAsymmetricInputAsymmetricKernelStridedGroupedPaddedDilatedCustom ker;
    ker.Init(
        x, weight, y,
        tiling_data.N, tiling_data.Cin, tiling_data.Hin, tiling_data.Win,
        tiling_data.Kh, tiling_data.Kw,
        tiling_data.Groups, tiling_data.CinPerGroup, tiling_data.CoPerGroup, tiling_data.Cout,
