host_operator_src="""

#include "max_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}

} // anonymous

namespace optiling {

// UB model for this kernel:
//   inQueue: 2 * tileInner
//   accBuf:  1 * tileInner
// Total floats = 3 * tileInner => bytes = 12 * tileInner
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative share per core
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor large DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap to limit per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxReductionOverADimensionCustomTilingData tiling;

    const gert::Shape* inShape = context->GetInputShape(0);
    const auto& originShape = inShape->GetOriginShape();
    const std::vector<int64_t> dims = originShape.GetDims();
    const uint32_t rank = static_cast<uint32_t>(dims.size());

    int64_t dimRaw = 0;
    if (const gert::RuntimeAttrs* attrs = context->GetAttrs()) {
        if (attrs->GetSize() >= 1) {
            const int64_t* p = attrs->GetAttrPointer<int64_t>(0);
            if (p != nullptr) dimRaw = *p;
        }
    }
    int64_t axis64 = dimRaw;
    if (rank > 0u) {
        if (axis64 < 0) axis64 += static_cast<int64_t>(rank);
        if (axis64 < 0) axis64 = 0;
        if (axis64 >= static_cast<int64_t>(rank)) axis64 = static_cast<int64_t>(rank) - 1;
    } else {
        axis64 = 0;
    }
    const uint32_t axis = static_cast<uint32_t>(axis64);

    uint64_t outer64 = 1, inner64 = 1, reduce64 = 1;
    for (uint32_t i = 0; i < axis; ++i) outer64 *= static_cast<uint64_t>(dims[i]);
    if (rank > 0) reduce64 = static_cast<uint64_t>(dims[axis]);
    for (uint32_t i = axis + 1; i < rank; ++i) inner64 *= static_cast<uint64_t>(dims[i]);

    const uint32_t outer = static_cast<uint32_t>(outer64);
    const uint32_t reduce = static_cast<uint32_t>(reduce64);
    const uint32_t inner = static_cast<uint32_t>(inner64);

    // tileInner from UB budget: 12 * tileInner bytes
    uint32_t maxTile = UB_BUDGET_BYTES / 12u;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileInner = maxTile;
    if (inner < tileInner) {
        const uint32_t aligned = AlignDown(inner, ALIGN_ELEMS);
        tileInner = (aligned > 0u) ? aligned : inner;
    }
    if (tileInner == 0u) tileInner = 1u; // safety

    const uint32_t tileCnt = CeilDiv(inner, tileInner);
    uint32_t totalTasks = (outer == 0u || inner == 0u) ? 0u : outer * tileCnt;
    if (totalTasks == 0u) totalTasks = 1u;

    // Adaptive blockDim based on tasks
    uint32_t blockDim = totalTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_outer(outer);
    tiling.set_reduce(reduce);
    tiling.set_inner(inner);
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUFFER_NUM = 2;

class KernelMaxReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMaxReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t reduce, uint32_t inner, uint32_t tileInner)
    {
        outer_ = outer;
        reduce_ = reduce;
        inner_ = inner;
        tileInner_ = tileInner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(outer_) * reduce_ * inner_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(outer_) * inner_);

        pipe_.InitBuffer(inQueueX_, IN_BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outer_ == 0 || reduce_ == 0 || inner_ == 0 || tileInner_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tileCnt = (inner_ + tileInner_ - 1) / tileInner_;
        const uint32_t totalTasks = outer_ * tileCnt;

        for (uint32_t task = blockIdx; task < totalTasks; task += blockNum) {
            const uint32_t o = task / tileCnt;
            const uint32_t t = task % tileCnt;
            const uint32_t tileBase = t * tileInner_;
            uint32_t curTile = inner_ - tileBase;
            if (curTile > tileInner_) curTile = tileInner_;

            ReduceTile(o, tileBase, curTile);
        }
    }

private:
    __aicore__ inline void CopyInSlice(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ReduceTile(uint32_t outerIdx, uint32_t innerBase, uint32_t len)
    {
        // Base offset in GM for this outer group
        const uint32_t groupBase = (outerIdx * reduce_) * inner_;

        // Load first slice r=0
        CopyInSlice(groupBase + /*r=0*/ 0u * inner_ + innerBase, len);
        AscendC::LocalTensor<float> first = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> acc = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(acc, first, len);
        inQueueX_.FreeTensor(first);

        // For each remaining r
        for (uint32_t r = 1; r < reduce_; ++r) {
            CopyInSlice(groupBase + r * inner_ + innerBase, len);
            AscendC::LocalTensor<float> cur = inQueueX_.DeQue<float>();
            AscendC::Max(acc, acc, cur, len);
            inQueueX_.FreeTensor(cur);
        }

        // Write out result to Y: offset = outerIdx * inner + innerBase
        const uint32_t yOffset = outerIdx * inner_ + innerBase;
        AscendC::DataCopy(yGm_[yOffset], acc, len);
        tmpQueue_.FreeTensor(acc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_ {0};
    uint32_t reduce_ {0};
    uint32_t inner_ {0};
    uint32_t tileInner_ {0};
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReductionOverADimensionCustom op;
    op.Init(x, y, tiling_data.outer, tiling_data.reduce, tiling_data.inner, tiling_data.tileInner);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t normalize_dim(int64_t dim, int64_t rank) {
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;
    return dim;
}

at::Tensor max_reduction_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    TORCH_CHECK(self.is_contiguous(), "Input tensor must be contiguous");
    const int64_t rank = self.dim();
    TORCH_CHECK(rank >= 1, "Input must have rank >= 1");

    const int64_t axis = normalize_dim(dim, rank);

    std::vector<int64_t> out_sizes;
    out_sizes.reserve(rank > 0 ? (rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i != axis) out_sizes.push_back(self.size(i));
    }

    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnMaxReductionOverADimensionCustom, self, axis, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_reduction_over_a_dimension_custom", &max_reduction_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_reduction_over_a_dimension_custom", &max_reduction_over_a_dimension_impl_npu,
          "Max reduction over a specific dimension (MaxReductionOverADimensionCustom on NPU)");
