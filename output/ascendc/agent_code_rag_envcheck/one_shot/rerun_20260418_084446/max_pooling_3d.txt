namespace {
static inline int64_t ComputeOutDim(int64_t in, int64_t k, int64_t s, int64_t p, bool ceil_mode) {
    const int64_t numer = in + 2 * p - k;
    if (ceil_mode) {
        return static_cast<int64_t>(std::floor((numer + s - 1) / static_cast<double>(s))) + 1;
    } else {
        return static_cast<int64_t>(std::floor(numer / static_cast<double>(s))) + 1;
    }
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *in_shape = context->GetInputShape(0);
    gert::Shape *out_shape = context->GetOutputShape(0);

    int64_t n = 1, c = 1, d = 1, h = 1, w = 1;
    if (in_shape->GetStorageShape().GetDimNum() == 5) {
        n = in_shape->GetStorageShape().GetDim(0);
        c = in_shape->GetStorageShape().GetDim(1);
        d = in_shape->GetStorageShape().GetDim(2);
        h = in_shape->GetStorageShape().GetDim(3);
        w = in_shape->GetStorageShape().GetDim(4);
    } else {
        int64_t total = in_shape->GetStorageShape().GetShapeSize();
        d = total; h = 1; w = 1;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernel_size = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *stride_attr = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *padding_attr = attrs->GetAttrPointer<int64_t>(2);
    const bool *ceil_mode = attrs->GetAttrPointer<bool>(4);

    const int64_t k = *kernel_size;
    const int64_t s = (*stride_attr < 1) ? k : *stride_attr;
    const int64_t p = *padding_attr;

    auto compute_out = [&](int64_t in)->int64_t{
        const int64_t numer = in + 2 * p - k;
        if (*ceil_mode) {
            return static_cast<int64_t>(std::floor((numer + s - 1) / static_cast<double>(s))) + 1;
        } else {
            return static_cast<int64_t>(std::floor(numer / static_cast<double>(s))) + 1;
        }
    };

    const int64_t outD = std::max<int64_t>(compute_out(d), 0);
    const int64_t outH = std::max<int64_t>(compute_out(h), 0);
    const int64_t outW = std::max<int64_t>(compute_out(w), 0);

    gert::Shape sOut({n, c, outD, outH, outW});
    *out_shape = sOut;
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include <cstdint>
#include <limits>

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t sD, uint32_t sH, uint32_t sW,
                                uint32_t pD, uint32_t pH, uint32_t pW,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint32_t totalNC)
    {
        n_ = n; c_ = c;
        inD_ = inD; inH_ = inH; inW_ = inW;
        kD_ = kD; kH_ = kH; kW_ = kW;
        sD_ = sD; sH_ = sH; sW_ = sW;
        pD_ = pD; pH_ = pH; pW_ = pW;
        outD_ = outD; outH_ = outH; outW_ = outW;
        totalNC_ = totalNC;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), static_cast<uint64_t>(n_) * c_ * inD_ * inH_ * inW_);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), static_cast<uint64_t>(n_) * c_ * outD_ * outH_ * outW_);

        // Minimal UB for scalar transfers
        pipe_.InitBuffer(inBuf_, 1, sizeof(float) * 1);
        pipe_.InitBuffer(outBuf_, 1, sizeof(float) * 1);
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || outD_ == 0 || outH_ == 0 || outW_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Flatten NC pairs and distribute across blocks
        for (uint32_t ncBase = blockIdx; ncBase < totalNC_; ncBase += blockNum) {
            const uint32_t nIdx = ncBase / c_;
            const uint32_t cIdx = ncBase % c_;
            ComputeOneNC(nIdx, cIdx);
        }
    }

private:
    __aicore__ inline uint64_t InOffset(uint32_t nIdx, uint32_t cIdx, uint32_t d, uint32_t h, uint32_t w) const
    {
        return (((static_cast<uint64_t>(nIdx) * c_ + cIdx) * inD_ + d) * inH_ + h) * inW_ + w;
    }
    __aicore__ inline uint64_t OutOffset(uint32_t nIdx, uint32_t cIdx, uint32_t d, uint32_t h, uint32_t w) const
    {
        return (((static_cast<uint64_t>(nIdx) * c_ + cIdx) * outD_ + d) * outH_ + h) * outW_ + w;
    }

    __aicore__ inline void ComputeOneNC(uint32_t nIdx, uint32_t cIdx)
    {
        const float negInf = -AscendC::INFINITY;
        for (uint32_t od = 0; od < outD_; ++od) {
            int32_t startD = static_cast<int32_t>(od) * static_cast<int32_t>(sD_) - static_cast<int32_t>(pD_);
            for (uint32_t oh = 0; oh < outH_; ++oh) {
                int32_t startH = static_cast<int32_t>(oh) * static_cast<int32_t>(sH_) - static_cast<int32_t>(pH_);
                for (uint32_t ow = 0; ow < outW_; ++ow) {
                    int32_t startW = static_cast<int32_t>(ow) * static_cast<int32_t>(sW_) - static_cast<int32_t>(pW_);
                    float curMax = negInf;

                    // Iterate over pooling window
                    for (uint32_t kd = 0; kd < kD_; ++kd) {
                        const int32_t id = startD + static_cast<int32_t>(kd);
                        if (id < 0 || id >= static_cast<int32_t>(inD_)) continue;
                        for (uint32_t kh = 0; kh < kH_; ++kh) {
                            const int32_t ih = startH + static_cast<int32_t>(kh);
                            if (ih < 0 || ih >= static_cast<int32_t>(inH_)) continue;
                            for (uint32_t kw = 0; kw < kW_; ++kw) {
                                const int32_t iw = startW + static_cast<int32_t>(kw);
                                if (iw < 0 || iw >= static_cast<int32_t>(inW_)) continue;

                                // Copy one element from GM to UB
                                AscendC::LocalTensor<float> tmpIn = inBuf_.AllocTensor<float>();
                                const uint64_t off = InOffset(nIdx, cIdx, static_cast<uint32_t>(id), static_cast<uint32_t>(ih), static_cast<uint32_t>(iw));
                                AscendC::DataCopy(tmpIn, xGm_[off], 1);
                                inBuf_.EnQue(tmpIn);
                                AscendC::LocalTensor<float> tmpInDe = inBuf_.DeQue<float>();

                                // Read scalar value from UB
                                float v = 0.0f;
                                // API to read scalar may vary; using assumed GetValue
                                v = tmpInDe.GetValue(0);

                                inBuf_.FreeTensor(tmpInDe);

                                // Update max
                                if (v > curMax) curMax = v;
                            }
                        }
                    }

                    // Write scalar to output
                    AscendC::LocalTensor<float> tmpOut = outBuf_.AllocTensor<float>();
                    // API to set scalar may vary; using assumed SetValue
                    tmpOut.SetValue(0, curMax);
                    outBuf_.EnQue(tmpOut);
                    AscendC::LocalTensor<float> tmpOutDe = outBuf_.DeQue<float>();
                    const uint64_t ooff = OutOffset(nIdx, cIdx, od, oh, ow);
                    AscendC::DataCopy(yGm_[ooff], tmpOutDe, 1);
                    outBuf_.FreeTensor(tmpOutDe);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inBuf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outBuf_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{1}, c_{1};
    uint32_t inD_{1}, inH_{1}, inW_{1};
    uint32_t kD_{1}, kH_{1}, kW_{1};
    uint32_t sD_{1}, sH_{1}, sW_{1};
    uint32_t pD_{0}, pH_{0}, pW_{0};
    uint32_t outD_{1}, outH_{1}, outW_{1};
    uint32_t totalNC_{1};
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.inD, tiling_data.inH, tiling_data.inW,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.sD, tiling_data.sH, tiling_data.sW,
            tiling_data.pD, tiling_data.pH, tiling_data.pW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.totalNC);
    op.Process();
}
