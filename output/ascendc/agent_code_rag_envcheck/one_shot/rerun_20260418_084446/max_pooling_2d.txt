namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling2dCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &dims = in_shape->GetOriginShape().GetDims();
    if (dims.size() != 4) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(dims[0]);
    const uint32_t C = static_cast<uint32_t>(dims[1]);
    const uint32_t H = static_cast<uint32_t>(dims[2]);
    const uint32_t W = static_cast<uint32_t>(dims[3]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kPtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *sPtr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *pPtr = attrs->GetAttrPointer<int32_t>(2);

    const uint32_t kH = static_cast<uint32_t>(*kPtr);
    const uint32_t kW = static_cast<uint32_t>(*kPtr);
    const uint32_t sH = static_cast<uint32_t>(*sPtr);
    const uint32_t sW = static_cast<uint32_t>(*sPtr);
    const uint32_t pH = static_cast<uint32_t>(*pPtr);
    const uint32_t pW = static_cast<uint32_t>(*pPtr);

    // Output shape (floor semantics, no ceil_mode)
    const uint32_t outH = (H + 2U * pH >= kH) ? ((H + 2U * pH - kH) / sH + 1U) : 0U;
    const uint32_t outW = (W + 2U * pW >= kW) ? ((W + 2U * pW - kW) / sW + 1U) : 0U;

    const uint32_t totalOut = N * C * outH * outW;

    // UB budgeting: we keep a single output tile and two single-element temp buffers.
    // yLocal: tileLen floats; tmp/accum: negligible.
    // Use conservative UB budget to avoid oversubscription.
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t ALIGN_ELEMS = 2048;
    static constexpr uint32_t HARD_CAP_ELEMS = 32768;

    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOut < tileLen) {
        const uint32_t aligned = AlignDown(totalOut, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOut;
    }
    if (tileLen == 0) tileLen = 1;

    const uint32_t numTiles = CeilDiv(totalOut, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8U) blockDim = 8U;
    if (blockDim > 24U) blockDim = 24U;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_sH(sH);
    tilling.set_sW(sW);
    tiling.set_pH(pH);
    tiling.set_pW(pW);
    tiling.set_totalOut(totalOut);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &dims = x_shape->GetOriginShape().GetDims();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kPtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *sPtr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *pPtr = attrs->GetAttrPointer<int32_t>(2);
    const uint32_t k = static_cast<uint32_t>(*kPtr);
    const uint32_t s = static_cast<uint32_t>(*sPtr);
    const uint32_t p = static_cast<uint32_t>(*pPtr);

    const uint64_t N = dims[0];
    const uint64_t C = dims[1];
    const uint64_t H = dims[2];
    const uint64_t W = dims[3];

    const uint64_t outH = (H + 2U * p >= k) ? ((H + 2U * p - k) / s + 1U) : 0U;
    const uint64_t outW = (W + 2U * p >= k) ? ((W + 2U * p - k) / s + 1U) : 0U;

    y_shape->SetDimNum(4);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, outH);
    y_shape->SetDim(3, outW);

    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t OUT_BUFFER_NUM = 2;

class KernelMaxPooling2dCustom {
public:
    __aicore__ inline KernelMaxPooling2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, const optiling::MaxPooling2dCustomTilingData &t)
    {
        n_ = t.n();
        c_ = t.c();
        h_ = t.h();
        w_ = t.w();
        outH_ = t.outH();
        outW_ = t.outW();
        kH_ = t.kH();
        kW_ = t.kW();
        sH_ = t.sH();
        sW_ = t.sW();
        pH_ = t.pH();
        pW_ = t.pW();
        totalOut_ = t.totalOut();
        tileLen_ = t.tileLen();

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * c_ * h_ * w_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * c_ * outH_ * outW_);

        pipe_.InitBuffer(outQueY_, OUT_BUFFER_NUM, tileLen_ * sizeof(float));
        // two single-element temps for accumulation and incoming value
        pipe_.InitBuffer(tmpQue_, 2, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOut_) return;

        uint32_t curLen = totalOut_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        while (true) {
            // Prefetch next tile boundaries
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalOut_) {
                nextLen = totalOut_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
            }

            ComputeTile(base, curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t base, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueY_.AllocTensor<float>();

        // Prepare per-output-element max computation
        for (uint32_t i = 0; i < len; ++i) {
            const uint32_t linear = base + i;

            // decode to (n, c, oh, ow)
            const uint32_t ow = linear % outW_;
            const uint32_t tmp1 = linear / outW_;
            const uint32_t oh = tmp1 % outH_;
            const uint32_t tmp2 = tmp1 / outH_;
            const uint32_t c = tmp2 % c_;
            const uint32_t n = tmp2 / c_;

            const int32_t hstart = static_cast<int32_t>(oh) * static_cast<int32_t>(sH_) - static_cast<int32_t>(pH_);
            const int32_t wstart = static_cast<int32_t>(ow) * static_cast<int32_t>(sW_) - static_cast<int32_t>(pW_);

            bool found = false;

            AscendC::LocalTensor<float> accum = tmpQue_.AllocTensor<float>();
            AscendC::LocalTensor<float> val = tmpQue_.AllocTensor<float>();

            // Iterate window
            for (uint32_t kh = 0; kh < kH_; ++kh) {
                const int32_t ih = hstart + static_cast<int32_t>(kh);
                if (ih < 0 || ih >= static_cast<int32_t>(h_)) continue;

                for (uint32_t kw = 0; kw < kW_; ++kw) {
                    const int32_t iw = wstart + static_cast<int32_t>(kw);
                    if (iw < 0 || iw >= static_cast<int32_t>(w_)) continue;

                    // load x[n, c, ih, iw] -> val[0]
                    const uint64_t inIdx = (((static_cast<uint64_t>(n) * c_ + c) * h_ + static_cast<uint32_t>(ih)) * w_
                                            + static_cast<uint32_t>(iw));
                    AscendC::DataCopy(val, xGm_[inIdx], 1);

                    if (!found) {
                        AscendC::DataCopy(accum, val, 1);
                        found = true;
                    } else {
                        AscendC::Max(accum, accum, val, 1);
                    }
                }
            }

            // Store result to yLocal[i]
            // By construction, outH/outW ensure there is at least one valid element; found should be true.
            AscendC::DataCopy(yLocal[i], accum, 1);

            tmpQue_.FreeTensor(val);
            tmpQue_.FreeTensor(accum);
        }

        outQueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t kH_{0}, kW_{0};
    uint32_t sH_{0}, sW_{0};
    uint32_t pH_{0}, pW_{0};
    uint32_t totalOut_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void max_pooling_2d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling2dCustom op;
    op.Init(x, y, tiling_data);
    op.Process();
}
