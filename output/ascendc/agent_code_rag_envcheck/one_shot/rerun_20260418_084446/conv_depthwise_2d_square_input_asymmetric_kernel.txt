kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelConvDepthwise2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t width,
                                uint32_t k)
    {
        n_ = n; c_ = c; h_ = h; w_ = width; k_ = k;
        oh_ = h_ - k_ + 1;
        ow_ = w_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * c_ * h_ * w_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(c_ * k_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_ * c_ * oh_ * ow_));

        // UB buffers per core:
        //   accQ_: 1 * W floats
        //   tmpQ_: 1 * W floats
        //   weightQ_: K floats
        pipe_.InitBuffer(accQ_, 1, w_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, w_ * sizeof(float));
        pipe_.InitBuffer(weightQ_, 1, k_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (oh_ == 0 || w_ == 0 || k_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        const uint32_t tasks = n_ * c_;
        if (tasks == 0) return;

        for (uint32_t task = blockIdx; task < tasks; task += blockNum) {
            const uint32_t n = task / c_;
            const uint32_t ch = task % c_;
            ComputeOneNC(n, ch);
        }
    }

private:
    __aicore__ inline void LoadWeights(uint32_t ch, LocalTensor<float> &wLocal)
    {
        const uint32_t wBase = ch * k_;
        DataCopy(wLocal, wGm_[wBase], k_);
    }

    __aicore__ inline void ComputeOneNC(uint32_t n, uint32_t ch)
    {
        // Load weights for this channel
        LocalTensor<float> wLocal = weightQ_.AllocTensor<float>();
        LoadWeights(ch, wLocal);
        weightQ_.EnQue(wLocal); // keep allocated during channel processing

        const uint32_t inBaseNC  = ((n * c_ + ch) * h_) * w_;
        const uint32_t outBaseNC = ((n * c_ + ch) * oh_) * ow_;

        for (uint32_t r = 0; r < oh_; ++r) {
            LocalTensor<float> acc = accQ_.AllocTensor<float>();
            LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();

            // First tap: acc = in[r + 0] * w[0]
            {
                const uint32_t inRowBase = inBaseNC + (r + 0) * w_;
                DataCopy(tmp, xGm_[inRowBase], w_);
                float w0 = wLocal.GetValue(0);
                Muls(acc, tmp, w0, w_);
            }

            // Accumulate remaining taps
            for (uint32_t kh = 1; kh < k_; ++kh) {
                const uint32_t inRowBase = inBaseNC + (r + kh) * w_;
                DataCopy(tmp, xGm_[inRowBase], w_);
                float wk = wLocal.GetValue(kh);
                Muls(tmp, tmp, wk, w_);
                Add(acc, acc, tmp, w_);
            }

            // Write out this output row
            const uint32_t outRowBase = outBaseNC + r * ow_;
            DataCopy(yGm_[outRowBase], acc, w_);

            tmpQ_.FreeTensor(tmp);
            accQ_.FreeTensor(acc);
        }

        weightQ_.DeQue<float>();
        weightQ_.FreeTensor(wLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, 1>  tmpQ_;
    TQue<TPosition::VECOUT, 1> accQ_;
    TQue<TPosition::VECCALC, 1> weightQ_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0}, k_{0};
    uint32_t oh_{0}, ow_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w, tiling_data.k);
    op.Process();
}

at::Tensor conv_dw2d_asym_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    // Output shape: (N, C, H-K+1, W) for depthwise Kx1, stride=1, padding=0, dilation=1
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, \"Expect 4D tensors\");
    TORCH_CHECK(x.size(1) == weight.size(0), \"Channel mismatch between input and weight\");
    TORCH_CHECK(weight.size(3) == 1, \"Kernel width must be 1 (asymmetric Kx1)\");
    TORCH_CHECK(x.size(2) >= weight.size(2), \"Input height must be >= kernel height\");

    auto N = x.size(0);
    auto C = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);
    auto K = weight.size(2);

    at::Tensor y = at::empty({N, C, H - K + 1, W}, x.options());
    EXEC_NPU_CMD(aclnnConvDepthwise2dSquareInputAsymmetricKernelCustom, x, weight, y);
    return y;
}
