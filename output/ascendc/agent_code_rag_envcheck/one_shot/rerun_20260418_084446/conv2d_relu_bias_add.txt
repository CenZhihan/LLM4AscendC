namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

// Conservative tiling parameters
static constexpr uint32_t UB_BUDGET_BYTES = 96 * 1024; // only store small y tiles
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t CalcOutDim(uint32_t in, uint32_t pad, uint32_t dil, uint32_t k, uint32_t stride) {
    const int64_t num = static_cast<int64_t>(in) + 2LL * pad - static_cast<int64_t>(dil) * (static_cast<int64_t>(k) - 1LL) - 1LL;
    return static_cast<uint32_t>(num / static_cast<int64_t>(stride) + 1LL);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dReluBiasAddCustomTilingData tiling;

    // shapes
    const gert::Shape *x_shape = context->GetInputShape(0);    // [N, Cin, H, W]
    const gert::Shape *w_shape = context->GetInputShape(1);    // [Cout, Cin, Kh, Kw]

    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cout= static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Kh  = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw  = static_cast<uint32_t>(w_shape->GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *strideH = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *strideW = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padH    = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *padW    = attrs->GetAttrPointer<int32_t>(3);
    const int32_t *dilH    = attrs->GetAttrPointer<int32_t>(4);
    const int32_t *dilW    = attrs->GetAttrPointer<int32_t>(5);
    const int32_t *hasBias = attrs->GetAttrPointer<int32_t>(6);

    const uint32_t Ho = CalcOutDim(H, *padH, *dilH, Kh, *strideH);
    const uint32_t Wo = CalcOutDim(W, *padW, *dilW, Kw, *strideW);

    const uint64_t totalOut64 = static_cast<uint64_t>(N) * Cout * Ho * Wo;
    const uint32_t totalOut = static_cast<uint32_t>(totalOut64);

    // choose tile length
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOut < tileLen) {
        uint32_t aligned = AlignDown(totalOut, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOut;
    }

    uint32_t numTiles = (tileLen == 0u) ? 1u : CeilDiv(totalOut, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);

    tiling.set_strideW(static_cast<uint32_t>(*strideW));
    tiling.set_padH(static_cast<uint32_t>(*padH));
    tiling.set_padW(static_cast<uint32_t>(*padW));
    tiling.set_dilH(static_cast<uint32_t>(*dilH));
    tiling.set_dilW(static_cast<uint32_t>(*dilW));

    tiling.set_totalOut(totalOut);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(static_cast<uint32_t>(*hasBias));

kernel_src="""
#include "kernel_operator.h"

class KernelConv2dReluBiasAddCustom {
public:
    __aicore__ inline KernelConv2dReluBiasAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR conv_b, GM_ADDR post_b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W,
                                uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilH, uint32_t dilW,
                                uint32_t totalOut, uint32_t tileLength, uint32_t hasBias)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout; H_ = H; W_ = W; Kh_ = Kh; Kw_ = Kw; Ho_ = Ho; Wo_ = Wo;
        strideH_ = strideH; strideW_ = strideW; padH_ = padH; padW_ = padW; dilH_ = dilH; dilW_ = dilW;
        totalOut_ = totalOut; tileLen_ = tileLength; hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint32_t)((uint64_t)N_*Cin_*H_*W_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint32_t)((uint64_t)Cout_*Cin_*Kh_*Kw_));
        bGm_.SetGlobalBuffer((__gm__ float*)conv_b, Cout_);
        pbGm_.SetGlobalBuffer((__gm__ float*)post_b, Cout_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint32_t)((uint64_t)N_*Cout_*Ho_*Wo_));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint64_t strideElems = (uint64_t)blockNum * (uint64_t)tileLen_;
        uint64_t base = (uint64_t)blockIdx * (uint64_t)tileLen_;

        while (base < totalOut_) {
            uint32_t curLen = (uint32_t)((base + tileLen_ <= totalOut_) ? tileLen_ : (totalOut_ - base));
            ComputeTile(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint64_t base, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            const uint64_t idx = base + (uint64_t)i;
            // decode linear index into (n, co, ho, wo)
            uint32_t wo = (uint32_t)(idx % Wo_);
            uint64_t tmp = idx / Wo_;
            uint32_t ho = (uint32_t)(tmp % Ho_);
            tmp /= Ho_;
            uint32_t co = (uint32_t)(tmp % Cout_);
            uint32_t n  = (uint32_t)(tmp / Cout_);

            float acc = 0.0f;

            // Iterate over input channels and kernel window
            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const int32_t in_h = (int32_t)ho * (int32_t)strideH_ - (int32_t)padH_ + (int32_t)kh * (int32_t)dilH_;
                    if ((uint32_t)in_h >= H_) continue;
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const int32_t in_w = (int32_t)wo * (int32_t)strideW_ - (int32_t)padW_ + (int32_t)kw * (int32_t)dilW_;
                        if ((uint32_t)in_w >= W_) continue;

                        const uint64_t x_offset = ((((uint64_t)n * Cin_ + ci) * H_ + (uint32_t)in_h) * W_ + (uint32_t)in_w);
                        const uint64_t w_offset = ((((uint64_t)co * Cin_ + ci) * Kh_ + kh) * Kw_ + kw);

                        float xv = xGm_[x_offset];
                        float wv = wGm_[w_offset];
                        acc += xv * wv;
                    }
                }
            }

            if (hasBias_ != 0u) {
                acc += bGm_[co];
            }
            // ReLU
            if (acc < 0.0f) acc = 0.0f;
            // post bias [Cout,1,1]
            acc += pbGm_[co];

            const uint64_t y_offset = ((((uint64_t)n * Cout_ + co) * Ho_ + ho) * Wo_ + wo);
            yGm_[y_offset] = acc;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> pbGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t strideH_{1}, strideW_{1}, padH_{0}, padW_{0}, dilH_{1}, dilW_{1};
    uint32_t totalOut_{0}, tileLen_{0};
    uint32_t hasBias_{0};
};

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR post_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluBiasAddCustom op;
    op.Init(
        x, weight, conv_bias, post_bias, y,
        tiling_data.N, tiling_data.Cin, tiling_data.Cout,
        tiling_data.H, tiling_data.W,
        tiling_data.Kh, tiling_data.Kw,
        tiling_data.Ho, tiling_data.Wo,
        tiling_data.strideH, tiling_data.strideW,
        tiling_data.padH, tiling_data.padW,
        tiling_data.dilH, tiling_data.dilW,
        tiling_data.totalOut, tiling_data.tileLength, tiling_data.hasBias
    );
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t out_dim(int64_t in, int64_t pad, int64_t dil, int64_t k, int64_t stride) {
    return (in + 2 * pad - dil * (k - 1) - 1) / stride + 1;
}

at::Tensor conv2d_relu_bias_add_impl_npu(const at::Tensor& x,
                                         const at::Tensor& weight,
                                         c10::optional<at::Tensor> conv_bias_opt,
                                         const at::Tensor& post_bias) {
    TORCH_CHECK(x.dim() == 4, "x must be NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be OIHW");
    TORCH_CHECK(post_bias.dim() == 3 || post_bias.dim() == 4, "post_bias must be [Cout,1,1] or [1,Cout,1,1]");

    const int64_t N = x.size(0);
    const int64_t Cin = x.size(1);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);

    const int64_t Cout = weight.size(0);
    const int64_t Kh = weight.size(2);
    const int64_t Kw = weight.size(3);

    const int64_t stride_h = 1, stride_w = 1;
    const int64_t pad_h = 0, pad_w = 0;
    const int64_t dil_h = 1, dil_w = 1;

    const int64_t Ho = out_dim(H, pad_h, dil_h, Kh, stride_h);
    const int64_t Wo = out_dim(W, pad_w, dil_w, Kw, stride_w);

    at::Tensor result = at::empty({N, Cout, Ho, Wo}, x.options());

    bool has_bias = conv_bias_opt.has_value();
    at::Tensor conv_bias_tensor;
    if (has_bias) {
        conv_bias_tensor = *conv_bias_opt;
        TORCH_CHECK(conv_bias_tensor.numel() == Cout, "conv_bias must be [Cout]");
    } else {
        conv_bias_tensor = at::Tensor(); // empty, ignored by kernel
    }

    // Ensure post_bias shape is [Cout,1,1]
    at::Tensor post_bias_bcast = post_bias;
    if (post_bias_bcast.dim() == 4) {
        // expect [1, Cout, 1, 1] or [N, Cout, 1, 1] but we only use channel-wise
        if (post_bias_bcast.size(0) == 1) {
            post_bias_bcast = post_bias_bcast.squeeze(0);
        }
    }
    if (post_bias_bcast.dim() == 1) {
        post_bias_bcast = post_bias_bcast.view({Cout, 1, 1});
    }

    EXEC_NPU_CMD(aclnnConv2dReluBiasAddCustom,
                 x, weight, conv_bias_tensor, post_bias_bcast,
                 (int64_t)stride_h, (int64_t)stride_w,
                 (int64_t)pad_h, (int64_t)pad_w,
                 (int64_t)dil_h, (int64_t)dil_w,
                 (int64_t)(has_bias ? 1 : 0),
                 result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_relu_bias_add_custom", &conv2d_relu_bias_add_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
