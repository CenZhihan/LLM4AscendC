host_operator_src="""

#include "mlp_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MlpCustomTilingData tiling;

    const uint32_t batch = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(0));
    const uint32_t inSize = static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *input_size = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *hidden1    = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *hidden2    = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *output_size= attrs->GetAttrPointer<int32_t>(3);

    // Validate
    const uint32_t h1 = static_cast<uint32_t>(*hidden1);
    const uint32_t h2 = static_cast<uint32_t>(*hidden2);
    const uint32_t outSize = static_cast<uint32_t>(*output_size);

    // UB budget assumptions for simple tiling
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core UB budget
    // Buffers in use for a block of GEMM:
    //  xTile: tileM * tileK
    //  wTile: tileN * tileK
    //  outTile: tileM * tileN
    // Assume 3 buffers + small temps -> bytes = 4 * tileM*tileK + 4*tileN*tileK + 4*tileM*tileN
    // For simplicity, choose fixed aligned tiles within budget.
    uint32_t tileK = 1024;
    uint32_t tileN = 128;
    uint32_t tileM = 8;

    // Adjust tileK to be not greater than inSize/h1/h2/out depending on layer
    auto fit_tile = [](uint32_t val, uint32_t align) -> uint32_t {
        uint32_t t = AlignDown(val, align);
        if (t == 0) t = val;
        return t;
    };
    tileK = fit_tile(tileK, 64u);
    tileN = fit_tile(tileN, 64u);
    tileM = fit_tile(tileM, 8u);

    // compute blockDim across batch tiles
    const uint32_t tilesPerBatch = CeilDiv(batch, tileM);
    uint32_t blockDim = tilesPerBatch;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    // Fill tiling data
    tiling.set_batch(batch);
    tiling.set_inSize(inSize);
    tiling.set_h1(h1);
    tiling.set_h2(h2);
    tiling.set_outSize(outSize);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);
    tiling.set_ws1Elems(static_cast<uint64_t>(batch) * static_cast<uint64_t>(h1));
    tiling.set_ws2Elems(static_cast<uint64_t>(batch) * static_cast<uint64_t>(h2));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(2);
    ws[0] = tiling.ws1Elems() * sizeof(float);
    ws[1] = tiling.ws2Elems() * sizeof(float);
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t INBUF_NUM = 2;
constexpr int32_t OUTBUF_NUM = 2;
constexpr int32_t TMPBUF_NUM = 2;

class KernelMlpCustom {
public:
    __aicore__ inline KernelMlpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wPacked, GM_ADDR bPacked, GM_ADDR y,
                                uint32_t batch, uint32_t inSize, uint32_t h1, uint32_t h2, uint32_t outSize,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                uint64_t ws1Elems, uint64_t ws2Elems,
                                GM_ADDR workspace)
    {
        batch_ = batch;
        inSize_ = inSize;
        h1_ = h1;
        h2_ = h2;
        outSize_ = outSize;

        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(batch_) * inSize_);
        wPackedGm_.SetGlobalBuffer((__gm__ float*)wPacked, static_cast<uint64_t>(inSize_)*h1_ + static_cast<uint64_t>(h1_)*h2_ + static_cast<uint64_t>(h2_)*outSize_);
        bPackedGm_.SetGlobalBuffer((__gm__ float*)bPacked, static_cast<uint64_t>(h1_) + h2_ + outSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(batch_) * outSize_);

        // Workspace: ws1 then ws2
        ws1Gm_.SetGlobalBuffer((__gm__ float*)workspace, ws1Elems);
        ws2Gm_.SetGlobalBuffer((__gm__ float*)((char*)workspace + ws1Elems * sizeof(float)), ws2Elems);

        pipe_.InitBuffer(xQueue_, INBUF_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, INBUF_NUM, tileN_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(oQueue_, OUTBUF_NUM, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMPBUF_NUM, tileN_ * sizeof(float)); // bias/partial
    }

    __aicore__ inline void Process()
    {
        if (batch_ == 0 || inSize_ == 0 || outSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        // Compute per-block rows
        const uint32_t rowStride = blockNum * tileM_;
        uint32_t rowBase = blockIdx * tileM_;
        for (; rowBase < batch_; rowBase += rowStride) {
            const uint32_t mLen = MinUi32(tileM_, batch_ - rowBase);

            // Layer 1: [mLen x inSize_] * [h1_ x inSize_]^T => [mLen x h1_], add b1, ReLU
            ComputeLayer(rowBase, inSize_, h1_, 0u, 0u, /*write to*/ 1);

            // Layer 2: [mLen x h1_] * [h2_ x h1_]^T => [mLen x h2_], add b2, ReLU
            ComputeLayer(rowBase, h1_, h2_, PackedWOffset1(), PackedBOffset1(), /*write to*/ 2);

            // Layer 3: [mLen x h2_] * [outSize_ x h2_]^T => [mLen x outSize_], add b3, no ReLU
            ComputeLayer(rowBase, h2_, outSize_, PackedWOffset2(), PackedBOffset2(), /*write to*/ 3);
        }
    }

private:
    __aicore__ inline uint64_t PackedWOffset1() const { return static_cast<uint64_t>(h1_) * inSize_; }
    __aicore__ inline uint64_t PackedWOffset2() const { return static_cast<uint64_t>(h1_) * inSize_ + static_cast<uint64_t>(h2_) * h1_; }
    __aicore__ inline uint64_t PackedBOffset1() const { return static_cast<uint64_t>(h1_); }
    __aicore__ inline uint64_t PackedBOffset2() const { return static_cast<uint64_t>(h1_) + static_cast<uint64_t>(h2_); }

    __aicore__ inline uint32_t MinUi32(uint32_t a, uint32_t b) { return a < b ? a : b; }

    // layerSelector: 1 -> output to ws1 + ReLU; 2 -> ws2 + ReLU; 3 -> y (no ReLU)
    __aicore__ inline void ComputeLayer(uint32_t rowBase, uint32_t K, uint32_t N,
                                        uint64_t wPackedOffset, uint64_t bPackedOffset,
                                        int layerSelector)
    {
        const uint32_t mLen = MinUi32(tileM_, batch_ - rowBase);
        // Iterate over N in tiles
        for (uint32_t nBase = 0; nBase < N; nBase += tileN_) {
            const uint32_t nLen = MinUi32(tileN_, N - nBase);

            // Initialize outTile with bias
            LocalTensor<float> outTile = oQueue_.AllocTensor<float>();
            // Zero init
            SetValue(outTile, 0.0f, mLen * nLen);

            // Add bias: broadcast bias[nBase:nBase+nLen] across mLen rows
            {
                LocalTensor<float> bLocal = tmpQueue_.AllocTensor<float>();
                // copy bias slice
                DataCopy(bLocal, bPackedGm_[bPackedOffset + nBase], nLen);
                for (uint32_t m = 0; m < mLen; ++m) {
                    DataCopy(outTile[m * nLen], bLocal, nLen);
                }
                tmpQueue_.FreeTensor(bLocal);
            }

            // Accumulate X * W^T across K dimension tiles
            for (uint32_t kBase = 0; kBase < K; kBase += tileK_) {
                const uint32_t kLen = MinUi32(tileK_, K - kBase);

                // Load xTile [mLen x kLen]
                LocalTensor<float> xTile = xQueue_.AllocTensor<float>();
                for (uint32_t m = 0; m < mLen; ++m) {
                    // Source depends on layer
                    if (layerSelector == 1) {
                        // from xGm_ [batch x inSize_]
                        DataCopy(xTile[m * kLen], xGm_[static_cast<uint64_t>(rowBase + m) * inSize_ + kBase], kLen);
                    } else if (layerSelector == 2) {
                        // from ws1Gm_ [batch x h1_]
                        DataCopy(xTile[m * kLen], ws1Gm_[static_cast<uint64_t>(rowBase + m) * h1_ + kBase], kLen);
                    } else {
                        // from ws2Gm_ [batch x h2_]
                        DataCopy(xTile[m * kLen], ws2Gm_[static_cast<uint64_t>(rowBase + m) * h2_ + kBase], kLen);
                    }
                }

                // Load wTile [nLen x kLen] from packed weights at offset
                LocalTensor<float> wTile = wQueue_.AllocTensor<float>();
                for (uint32_t n = 0; n < nLen; ++n) {
                    // weights are stored row-major [N x K]
                    const uint64_t wRowStart = (wPackedOffset + static_cast<uint64_t>(nBase + n) * K + kBase);
                    DataCopy(wTile[n * kLen], wPackedGm_[wRowStart], kLen);
                }

                // Compute outTile += xTile @ wTile^T
                // naive nested loops with vector dot-products
                for (uint32_t m = 0; m < mLen; ++m) {
                    LocalTensor<float> xVec = xTile[m * kLen];
                    for (uint32_t n = 0; n < nLen; ++n) {
                        LocalTensor<float> wVec = wTile[n * kLen];
                        // tmp = xVec * wVec
                        // Accumulate sum into scalar by ReduceSum
                        float sumVal = 0.0f;
                        // Process in chunks to keep temporaries small
                        const uint32_t vStep = 256u;
                        for (uint32_t kk = 0; kk < kLen; kk += vStep) {
                            const uint32_t vlen = (kk + vStep <= kLen) ? vStep : (kLen - kk);
                            LocalTensor<float> tmpMul = tmpQueue_.AllocTensor<float>();
                            Mul(tmpMul, xVec[kk], wVec[kk], vlen);
                            // ReduceSum to scalar
                            float partial = 0.0f;
                            ReduceSum(&partial, tmpMul, vlen);
                            sumVal += partial;
                            tmpQueue_.FreeTensor(tmpMul);
                        }
                        outTile[m * nLen + n] += sumVal;
                    }
                }

                wQueue_.FreeTensor(wTile);
                xQueue_.FreeTensor(xTile);
            } // kBase loop

            // Apply activation if needed and write back
            if (layerSelector == 1) {
                // ReLU and store to ws1
                ReluOrCopyStore(outTile, mLen, nLen, ws1Gm_, rowBase, h1_, nBase, true);
            } else if (layerSelector == 2) {
                ReluOrCopyStore(outTile, mLen, nLen, ws2Gm_, rowBase, h2_, nBase, true);
            } else {
                ReluOrCopyStore(outTile, mLen, nLen, yGm_,  rowBase, outSize_, nBase, false);
            }

            oQueue_.FreeTensor(outTile);
        } // nBase
    }

    template<typename GMType>
    __aicore__ inline void ReluOrCopyStore(LocalTensor<float>& tile, uint32_t mLen, uint32_t nLen,
                                           GMType& dstGm, uint32_t rowBase, uint32_t rowStride,
                                           uint32_t nBase, bool doRelu)
    {
        for (uint32_t m = 0; m < mLen; ++m) {
            if (doRelu) {
                // Apply ReLU in-place on the row
                Maxs(tile[m * nLen], tile[m * nLen], 0.0f, nLen);
            }
            DataCopy(dstGm[static_cast<uint64_t>(rowBase + m) * rowStride + nBase], tile[m * nLen], nLen);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, INBUF_NUM> xQueue_;
    TQue<TPosition::VECIN, INBUF_NUM> wQueue_;
    TQue<TPosition::VECOUT, OUTBUF_NUM> oQueue_;
    TQue<TPosition::VECCALC, TMPBUF_NUM> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wPackedGm_;
    GlobalTensor<float> bPackedGm_;
    GlobalTensor<float> yGm_;
    GlobalTensor<float> ws1Gm_;
    GlobalTensor<float> ws2Gm_;

    uint32_t batch_{0}, inSize_{0}, h1_{0}, h2_{0}, outSize_{0};
    uint32_t tileM_{1}, tileN_{1}, tileK_{1};
};

extern "C" __global__ __aicore__ void mlp_custom(GM_ADDR x, GM_ADDR w_packed, GM_ADDR b_packed, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMlpCustom op;
    op.Init(x, w_packed, b_packed, y,
            tiling_data.batch, tiling_data.inSize, tiling_data.h1, tiling_data.h2, tiling_data.outSize,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.ws1Elems, tiling_data.ws2Elems,
            workspace);
    op.Process();
}
