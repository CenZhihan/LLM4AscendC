host_operator_src="""

#include "matmul_for_lower_triangular_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (per-core):
//   yChunk: 1 * tile
//   bSeg:   1 * tile
//   zeroBuf:1 * tile
// Total floats ~ 3 * tile => bytes = 12 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // reasonable alignment for DMA
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap to control per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForLowerTriangularMatricesCustomTilingData tiling;

    const gert::Shape *ashape = context->GetInputShape(0);
    const gert::Shape *bshape = context->GetInputShape(1);

    // Expect square matrices (N x N)
    const uint64_t N0 = (ashape->GetOriginFormat() == ge::FORMAT_ND && ashape->GetOriginShape().GetDimNum() >= 2)
                        ? static_cast<uint64_t>(ashape->GetOriginShape().GetDim(0))
                        : static_cast<uint64_t>(ashape->GetShapeSize()); // fallback
    const uint64_t N1 = (bshape->GetOriginFormat() == ge::FORMAT_ND && bshape->GetOriginShape().GetDimNum() >= 2)
                        ? static_cast<uint64_t>(bshape->GetOriginShape().GetDim(0))
                        : static_cast<uint64_t>(bshape->GetShapeSize());

    const uint32_t N = static_cast<uint32_t>(N0);

    // tileLen from UB budget: bytes = 12 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (N < tileLen) {
        const uint32_t aligned = AlignDown(N, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : N;
    }
    if (tileLen == 0) tileLen = 1;

    // Adaptive blockDim based on number of rows
    uint32_t blockDim = N;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulForLowerTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForLowerTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileLength)
    {
        N_ = N;
        tileLen_ = tileLength;
        totalLen_ = static_cast<uint64_t>(N_) * static_cast<uint64_t>(N_);

        aGm_.SetGlobalBuffer((__gm__ float*)A, totalLen_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, totalLen_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, totalLen_);

        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(zeroQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(aSegQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a grid-stride set of rows
        for (uint32_t i = blockIdx; i < N_; i += blockNum) {
            // Process lower triangle row i, columns 0..i
            uint32_t jStart = 0;
            while (jStart <= i) {
                uint32_t lenJ = i - jStart + 1;
                if (lenJ > tileLen_) lenJ = tileLen_;

                AscendC::LocalTensor<float> y = yQueue_.AllocTensor<float>();
                AscendC::Duplicate(y, 0.0f, lenJ);

                // Load A row segment A[i, jStart..i] (lenJ elements)
                AscendC::LocalTensor<float> aSeg = aSegQueue_.AllocTensor<float>();
                const uint64_t aOffset = static_cast<uint64_t>(i) * N_ + jStart;
                AscendC::DataCopy(aSeg, aGm_[aOffset], lenJ);

                // Accumulate over k = jStart..i
                for (uint32_t t = 0; t < lenJ; ++t) {
                    const uint32_t k = jStart + t;
                    const uint32_t lenK = t + 1;  // j range within this chunk is jStart..(jStart+t)

                    AscendC::LocalTensor<float> bSeg = bQueue_.AllocTensor<float>();
                    const uint64_t bOffset = static_cast<uint64_t>(k) * N_ + jStart;
                    AscendC::DataCopy(bSeg, bGm_[bOffset], lenK);

                    // Fetch scalar a[i,k] from aSeg and scale bSeg
                    float aVal = aSeg.GetValue(t);
                    AscendC::Muls(bSeg, bSeg, aVal, lenK);
                    AscendC::Add(y, y, bSeg, lenK);

                    bQueue_.FreeTensor(bSeg);
                }

                // Store accumulated chunk to C at row i, columns jStart..jStart+lenJ-1
                const uint64_t cOffset = static_cast<uint64_t>(i) * N_ + jStart;
                AscendC::DataCopy(cGm_[cOffset], y, lenJ);

                aSegQueue_.FreeTensor(aSeg);
                yQueue_.FreeTensor(y);

                jStart += lenJ;
            }

            // Zero the upper triangle part: columns (i+1)..(N_-1)
            if (i + 1 < N_) {
                uint32_t remain = N_ - (i + 1);
                uint32_t col = i + 1;
                while (remain > 0) {
                    uint32_t chunk = (remain > tileLen_) ? tileLen_ : remain;
                    AscendC::LocalTensor<float> z = zeroQueue_.AllocTensor<float>();
                    AscendC::Duplicate(z, 0.0f, chunk);
                    const uint64_t cOff = static_cast<uint64_t>(i) * N_ + col;
                    AscendC::DataCopy(cGm_[cOff], z, chunk);
                    zeroQueue_.FreeTensor(z);
                    remain -= chunk;
                    col += chunk;
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> zeroQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> aSegQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint64_t totalLen_ {0};
    uint32_t N_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForLowerTriangularMatricesCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.tileLength);
    op.Process();
}
