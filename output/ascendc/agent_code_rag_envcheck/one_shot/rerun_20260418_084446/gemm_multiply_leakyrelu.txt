host_operator_src="""

#include "register/op_def_registry.h"
#include "gemm_multiply_leakyrelu_custom_tiling.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// For post-processing (scale + leakyrelu) UB budgeting:
//   inQueueY: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue: 1 * tile
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS     = 2048;
static constexpr uint32_t HARD_CAP_ELEMS  = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmMultiplyLeakyreluCustomTilingData td;

    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);

    // Expect x: [M, K], w: [N, K]
    const uint32_t dims_x = static_cast<uint32_t>(x_shape->GetOriginShape().GetDimNum());
    const uint32_t dims_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDimNum());
    if (dims_x < 2 || dims_w < 2) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Kx = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(x_shape->GetOriginShape().GetDimNum() - 1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(w_shape->GetOriginShape().GetDimNum() - 1));
    if (Kx != Kw) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t totalLen = M * N;

    uint32_t maxTile = UB_BUDGET_BYTES / 20;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLen < tileLen) {
        const uint32_t aligned = AlignDown(totalLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLen;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    td.set_m(M);
    td.set_n(N);
    td.set_k(Kx);
    td.set_totalLength(totalLen);
    td.set_tileLength(tileLen);
    td.set_blockDim(blockDim);

    td.SaveToBuffer(context->GetRawTilingData()->GetData(),
                    context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(td.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const int64_t M = x_shape->GetOriginShape().GetDim(0);
    const int64_t Kx = x_shape->GetOriginShape().GetDim(x_shape->GetOriginShape().GetDimNum() - 1);
    const int64_t N = w_shape->GetOriginShape().GetDim(0);
    const int64_t Kw = w_shape->GetOriginShape().GetDim(w_shape->GetOriginShape().GetDimNum() - 1);

    if (Kx != Kw) {
        return GRAPH_FAILED;
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    gert::Shape origin_y({M, N});
    y_shape->SetOriginShape(origin_y);
    y_shape->SetStorageShape(origin_y);

    return GRAPH_SUCCESS;
}

{

    if (x_dtype != w_dtype) {

    }
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t INOUT_BUFFER_NUM = 2;

class KernelGemmMultiplyLeakyreluCustom {
public:
    __aicore__ inline KernelGemmMultiplyLeakyreluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR multiplier, GM_ADDR negativeSlope,
                                GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t totalLength, uint32_t tileLength)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(M_ * K_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(N_ * K_));
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        mulGm_.SetGlobalBuffer((__gm__ float*)multiplier, 1);
        nsGm_.SetGlobalBuffer((__gm__ float*)negativeSlope, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueY_, INOUT_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, INOUT_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));

        // Load scalar multiplier and negative slope to registers
        AscendC::LocalTensor<float> sLoc = tmpQueue_.AllocTensor<float>();
        AscendC::DataCopy(sLoc, mulGm_[0], 1);
        pipe_.PipeBarrier();
        mul_ = sLoc.GetValue(0);

        AscendC::DataCopy(sLoc, nsGm_[0], 1);
        pipe_.PipeBarrier();
        negSlope_ = sLoc.GetValue(0);
        negCorr_ = negSlope_ - 1.0f;

        tmpQueue_.FreeTensor(sLoc);
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        // Placeholder: GEMM + bias compute stage.
        // In practical implementation, register and run MatMul cube using tiling over (M, N, K)
        // to produce yGm_ = xGm_ @ wGm_^T + bGm_.
        // For example:
        //   REGIST_MATMUL_OBJ(&pipe_, GetSysWorkSpacePtr(), matmulObj_, &tiling_);
        //   matmulObj_.Compute();
        pipe_.PipeBarrier();

        // Post-processing: scale by multiplier, then apply LeakyReLU
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocalIn = inQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> yLocalOut = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // y = y * mul_
        AscendC::Muls(yLocalOut, yLocalIn, mul_, len);

        // y = y + (negativeSlope - 1) * min(y, 0)
        AscendC::Mins(tLocal, yLocalOut, 0.0f, len);
        AscendC::Muls(tLocal, tLocal, negCorr_, len);
        AscendC::Add(yLocalOut, yLocalOut, tLocal, len);

        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue(yLocalOut);
        inQueueY_.FreeTensor(yLocalIn);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INOUT_BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, INOUT_BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> mulGm_;
    AscendC::GlobalTensor<float> nsGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    float mul_{1.0f};
    float negSlope_{0.01f};
    float negCorr_{-0.99f};
};

extern "C" __global__ __aicore__ void gemm_multiply_leakyrelu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b,
    GM_ADDR multiplier, GM_ADDR negative_slope,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmMultiplyLeakyreluCustom op;
    op.Init(x, w, b, multiplier, negative_slope, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}
