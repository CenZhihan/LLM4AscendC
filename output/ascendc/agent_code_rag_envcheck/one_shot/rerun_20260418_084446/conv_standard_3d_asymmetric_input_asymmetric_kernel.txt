#include "kernel_operator.h"

class KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t wi,
                                uint32_t oc, uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint32_t totalOut)
    {
        n_ = n; c_ = c; d_ = d; h_ = h; w_ = wi;
        oc_ = oc; kd_ = kd; kh_ = kh; kw_ = kw;
        outD_ = outD; outH_ = outH; outW_ = outW;
        totalOut_ = totalOut;

        // Set GM tensors
        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_)*c_*d_*h_*w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(oc_)*c_*kd_*kh_*kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_)*oc_*outD_*outH_*outW_);

        // Precompute strides
        sx_n_ = static_cast<uint64_t>(c_)*d_*h_*w_;
        sx_c_ = static_cast<uint64_t>(d_)*h_*w_;
        sx_d_ = static_cast<uint64_t>(h_)*w_;
        sx_h_ = static_cast<uint64_t>(w_);
        sx_w_ = 1ULL;

        sw_oc_ = static_cast<uint64_t>(c_)*kd_*kh_*kw_;
        sw_c_  = static_cast<uint64_t>(kd_)*kh_*kw_;
        sw_kd_ = static_cast<uint64_t>(kh_)*kw_;
        sw_kh_ = static_cast<uint64_t>(kw_);
        sw_kw_ = 1ULL;

        sy_n_  = static_cast<uint64_t>(oc_)*outD_*outH_*outW_;
        sy_oc_ = static_cast<uint64_t>(outD_)*outH_*outW_;
        sy_od_ = static_cast<uint64_t>(outH_)*outW_;
        sy_oh_ = static_cast<uint64_t>(outW_);
        sy_ow_ = 1ULL;

        // Init small calc buffer queue
        pipe_.InitBuffer(calcQueue_, 4, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t outIdx = blockIdx; outIdx < totalOut_; outIdx += blockNum) {
            // Decode output index -> (n, oc, od, oh, ow)
            uint32_t rem = outIdx;

            const uint32_t v0 = static_cast<uint32_t>(sy_n_); // OC*OD*OH*OW
            const uint32_t v1 = static_cast<uint32_t>(sy_oc_);// OD*OH*OW
            const uint32_t v2 = static_cast<uint32_t>(sy_od_);// OH*OW
            const uint32_t v3 = static_cast<uint32_t>(sy_oh_);// OW

            uint32_t n = rem / v0; rem -= n * v0;
            uint32_t oc = rem / v1; rem -= oc * v1;
            uint32_t od = rem / v2; rem -= od * v2;
            uint32_t oh = rem / v3; rem -= oh * v3;
            uint32_t ow = rem;

            // Local accumulator (length 1) initialized to 0
            AscendC::LocalTensor<float> acc = calcQueue_.AllocTensor<float>();
            AscendC::Muls(acc, acc, 0.0f, 1);

            // Accumulate over Cin, KD, KH, KW
            for (uint32_t ic = 0; ic < c_; ++ic) {
                for (uint32_t zd = 0; zd < kd_; ++zd) {
                    for (uint32_t yh = 0; yh < kh_; ++yh) {
                        for (uint32_t xw = 0; xw < kw_; ++xw) {
                            // Compute global indices
                            uint64_t xin = static_cast<uint64_t>(n);
                            uint64_t xic = static_cast<uint64_t>(ic);
                            uint64_t xd  = static_cast<uint64_t>(od + zd);
                            uint64_t xh  = static_cast<uint64_t>(oh + yh);
                            uint64_t xw  = static_cast<uint64_t>(ow + xw);

                            uint64_t xOffset = xin * sx_n_ + xic * sx_c_ + xd * sx_d_ + xh * sx_h_ + xw * sx_w_;

                            uint64_t woc = static_cast<uint64_t>(oc);
                            uint64_t wic = static_cast<uint64_t>(ic);
                            uint64_t wk3 = static_cast<uint64_t>(zd);
                            uint64_t wk2 = static_cast<uint64_t>(yh);
                            uint64_t wk1 = static_cast<uint64_t>(xw);

                            uint64_t wOffset = woc * sw_oc_ + wic * sw_c_ + wk3 * sw_kd_ + wk2 * sw_kh_ + wk1 * sw_kw_;

                            // Load x and w (length 1) into UB and accumulate
                            AscendC::LocalTensor<float> xv = calcQueue_.AllocTensor<float>();
                            AscendC::LocalTensor<float> wv = calcQueue_.AllocTensor<float>();
                            AscendC::DataCopy(xv, xGm_[xOffset], 1);
                            AscendC::DataCopy(wv, wGm_[wOffset], 1);

                            // acc += xv * wv
                            AscendC::Mul(xv, xv, wv, 1);
                            AscendC::Add(acc, acc, xv, 1);

                            calcQueue_.FreeTensor(wv);
                            calcQueue_.FreeTensor(xv);
                        }
                    }
                }
            }

            // Store result to y
            uint64_t yOffset = static_cast<uint64_t>(n) * sy_n_ +
                               static_cast<uint64_t>(oc) * sy_oc_ +
                               static_cast<uint64_t>(od) * sy_od_ +
                               static_cast<uint64_t>(oh) * sy_oh_ +
                               static_cast<uint64_t>(ow) * sy_ow_;

            AscendC::DataCopy(yGm_[yOffset], acc, 1);
            calcQueue_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> calcQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, d_{0}, h_{0}, w_{0};
    uint32_t oc_{0}, kd_{0}, kh_{0}, kw_{0};
    uint32_t outD_{0}, outH_{0}, outW_{0}, totalOut_{0};

    // Strides
    uint64_t sx_n_{0}, sx_c_{0}, sx_d_{0}, sx_h_{0}, sx_w_{0};
    uint64_t sw_oc_{0}, sw_c_{0}, sw_kd_{0}, sw_kh_{0}, sw_kw_{0};
    uint64_t sy_n_{0}, sy_oc_{0}, sy_od_{0}, sy_oh_{0}, sy_ow_{0};
};

extern "C" __global__ __aicore__
void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x,
            weight,
            y,
            tiling_data.n,
            tiling_data.c,
            tiling_data.d,
            tiling_data.h,
            tiling_data.w,
            tiling_data.oc,
            tiling_data.kd,
            tiling_data.kh,
            tiling_data.kw,
            tiling_data.outD,
            tiling_data.outH,
            tiling_data.outW,
            tiling_data.totalOut);
    op.Process();
}
