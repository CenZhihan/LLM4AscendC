host_operator_src="""

#include "matmul_with_transposed_both_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 16;    // prefer 16-element alignment for vector ops
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t AlignUp(uint32_t x, uint32_t a)
{
    return ((x + a - 1u) / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedBothCustomTilingData tiling;

    // Input shapes:
    //   A: (K, M)  -> we will compute using A^T of shape (M, K)
    //   B: (N, K)  -> we will compute using B^T of shape (K, N)
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t aK = a_shape->GetOriginShape().GetDim(0);
    const uint64_t aM = a_shape->GetOriginShape().GetDim(1);
    const uint64_t bN = b_shape->GetOriginShape().GetDim(0);
    const uint64_t bK = b_shape->GetOriginShape().GetDim(1);
    if (aK != bK) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(aM);
    const uint32_t K = static_cast<uint32_t>(aK);
    const uint32_t N = static_cast<uint32_t>(bN);

    // Choose initial tile sizes
    uint32_t tileM = std::min<uint32_t>(AlignDown(128, ALIGN_ELEMS), AlignDown(M > 0 ? M : ALIGN_ELEMS, ALIGN_ELEMS));
    uint32_t tileN = std::min<uint32_t>(AlignDown(128, ALIGN_ELEMS), AlignDown(N > 0 ? N : ALIGN_ELEMS, ALIGN_ELEMS));
    uint32_t tileK = std::min<uint32_t>(AlignDown(256, ALIGN_ELEMS), AlignDown(K > 0 ? K : ALIGN_ELEMS, ALIGN_ELEMS));
    if (tileM == 0) tileM = std::min<uint32_t>(M, ALIGN_ELEMS);
    if (tileN == 0) tileN = std::min<uint32_t>(N, ALIGN_ELEMS);
    if (tileK == 0) tileK = std::min<uint32_t>(K, ALIGN_ELEMS);

    // UB footprint: A(kTile x mTile) + B(nTile x kTile) + C(mTile x nTile) + tmpN(nTile) + tmpMul(nTile)
    auto FitsUB = [&](uint32_t m, uint32_t n, uint32_t k) -> bool {
        uint64_t elems = (uint64_t)k * m + (uint64_t)n * k + (uint64_t)m * n + n + n;
        uint64_t bytes = elems * 4u;
        return bytes <= UB_BUDGET_BYTES && m > 0 && n > 0 && k > 0;
    };

    // Decrease tiles until they fit
    while (!FitsUB(tileM, tileN, tileK)) {
        if (tileK > ALIGN_ELEMS) {
            tileK = AlignDown(tileK / 2u, ALIGN_ELEMS);
            if (tileK == 0) tileK = ALIGN_ELEMS;
        } else if (tileN > ALIGN_ELEMS) {
            tileN = AlignDown(tileN / 2u, ALIGN_ELEMS);
            if (tileN == 0) tileN = ALIGN_ELEMS;
        } else if (tileM > ALIGN_ELEMS) {
            tileM = AlignDown(tileM / 2u, ALIGN_ELEMS);
            if (tileM == 0) tileM = ALIGN_ELEMS;
        } else {
            break;
        }
    }

    // Make sure tile sizes do not exceed dims
    tileM = std::min<uint32_t>(tileM, std::max<uint32_t>(1, AlignDown(M == 0 ? ALIGN_ELEMS : M, ALIGN_ELEMS) ? AlignDown(M, ALIGN_ELEMS) : M));
    tileN = std::min<uint32_t>(tileN, std::max<uint32_t>(1, AlignDown(N == 0 ? ALIGN_ELEMS : N, ALIGN_ELEMS) ? AlignDown(N, ALIGN_ELEMS) : N));
    tileK = std::min<uint32_t>(tileK, std::max<uint32_t>(1, AlignDown(K == 0 ? ALIGN_ELEMS : K, ALIGN_ELEMS) ? AlignDown(K, ALIGN_ELEMS) : K));

    if (!FitsUB(tileM, tileN, tileK)) {
        // Fallback very small tiles
        tileM = std::min<uint32_t>(ALIGN_ELEMS, std::max<uint32_t>(1, M));
        tileN = std::min<uint32_t>(ALIGN_ELEMS, std::max<uint32_t>(1, N));
        tileK = std::min<uint32_t>(ALIGN_ELEMS, std::max<uint32_t>(1, K));
    }

    const uint32_t mTiles = CeilDiv(M, tileM);
    const uint32_t nTiles = CeilDiv(N, tileN);
    uint32_t blockDim = mTiles * nTiles;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim < 1) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_mTiles(mTiles);
    tiling.set_nTiles(nTiles);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // Output Y = (A^T) @ (B^T), where A: (K, M), B: (N, K), so Y: (M, N)
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, a_shape->GetOriginShape().GetDim(1)); // M
    y_shape->SetDim(1, b_shape->GetOriginShape().GetDim(0)); // N
    return GRAPH_SUCCESS;
}

{
    const ge::DataType a_dtype = context->GetInputDataType(0);

    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;

    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelMatmulWithTransposedBothCustom {
public:
    __aicore__ inline KernelMatmulWithTransposedBothCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                uint32_t mTiles, uint32_t nTiles)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN; tileK_ = tileK;
        mTiles_ = mTiles; nTiles_ = nTiles;

        aGm_.SetGlobalBuffer((__gm__ float*)a, (uint64_t)K_ * (uint64_t)M_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)N_ * (uint64_t)K_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * (uint64_t)N_);

        // UB buffers:
        // aTile: tileK * tileM
        // bTile: tileN * tileK
        // cTile: tileM * tileN
        // tmpN: tileN
        // tmpMul: tileN
        pipe_.InitBuffer(aQue_, 1, tileK_ * tileM_ * sizeof(float));
        pipe_.InitBuffer(bQue_, 1, tileN_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(cQue_, 1, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpNQue_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpMulQue_, 1, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = GetBlockNum();
        const uint32_t blockIdx = GetBlockIdx();

        const uint32_t totalTiles = mTiles_ * nTiles_;
        uint32_t tileId = blockIdx;

        while (tileId < totalTiles) {
            const uint32_t mIdx = tileId % mTiles_;
            const uint32_t nIdx = tileId / mTiles_;
            ComputeTile(mIdx, nIdx);
            tileId += blockNum;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t mIdx, uint32_t nIdx)
    {
        const uint32_t mStart = mIdx * tileM_;
        const uint32_t nStart = nIdx * tileN_;
        const uint32_t mLen = (mStart + tileM_ <= M_) ? tileM_ : (M_ - mStart);
        const uint32_t nLen = (nStart + tileN_ <= N_) ? tileN_ : (N_ - nStart);

        // Allocate local tensors
        LocalTensor<float> aLocal = aQue_.AllocTensor<float>();
        LocalTensor<float> bLocal = bQue_.AllocTensor<float>();
        LocalTensor<float> cLocal = cQue_.AllocTensor<float>();
        LocalTensor<float> tmpN = tmpNQue_.AllocTensor<float>();
        LocalTensor<float> tmpMul = tmpMulQue_.AllocTensor<float>();

        // Initialize C tile to zero
        // cLocal layout: row-major [mLen x nLen] placed in a buffer sized [tileM x tileN]
        Duplicate(cLocal, 0.0f, (uint32_t)(tileM_ * tileN_));

        // Loop over K dimension
        for (uint32_t kBase = 0; kBase < K_; kBase += tileK_) {
            const uint32_t kLen = (kBase + tileK_ <= K_) ? tileK_ : (K_ - kBase);

            // Load A block: shape [kLen x mLen] from GM at offsets A[kBase ... kBase+kLen-1, mStart ...]
            for (uint32_t kk = 0; kk < kLen; ++kk) {
                const uint32_t aRowGmOff = (kBase + kk) * M_ + mStart;
                DataCopy(aLocal[kk * tileM_], aGm_[aRowGmOff], mLen);
            }

            // Load B block: shape [nLen x kLen] from GM at offsets B[nStart... , kBase ...]
            for (uint32_t nn = 0; nn < nLen; ++nn) {
                const uint32_t bRowGmOff = (nStart + nn) * K_ + kBase;
                DataCopy(bLocal[nn * tileK_], bGm_[bRowGmOff], kLen);
            }

            // Accumulate C += A(k,m) outer B(n,k) over k
            for (uint32_t kk = 0; kk < kLen; ++kk) {
                // Build tmpN vector of length nLen: tmpN[n] = B[n, kk]
                for (uint32_t nn = 0; nn < nLen; ++nn) {
                    // bLocal row-major [nLen x tileK_]
                    tmpN[nn] = bLocal[nn * tileK_ + kk];
                }

                // For each m row, update C[m, :] += tmpN[:] * A[kk, m]
                for (uint32_t mm = 0; mm < mLen; ++mm) {
                    const float aScalar = aLocal[kk * tileM_ + mm];
                    // tmpMul = tmpN * aScalar
                    Muls(tmpMul, tmpN, aScalar, nLen);
                    // cRow pointer
                    Add(cLocal[mm * tileN_], cLocal[mm * tileN_], tmpMul, nLen);
                }
            }
        }

        // Write C tile back to GM: Y has shape (M, N), row-major
        for (uint32_t mm = 0; mm < mLen; ++mm) {
            const uint32_t yGmOff = (mStart + mm) * N_ + nStart;
            DataCopy(yGm_[yGmOff], cLocal[mm * tileN_], nLen);
        }

        // Free local tensors
        tmpMulQue_.FreeTensor(tmpMul);
        tmpNQue_.FreeTensor(tmpN);
        cQue_.FreeTensor(cLocal);
        bQue_.FreeTensor(bLocal);
        aQue_.FreeTensor(aLocal);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, 1> aQue_;
    TQue<TPosition::VECIN, 1> bQue_;
    TQue<TPosition::VECCALC, 1> cQue_;
    TQue<TPosition::VECCALC, 1> tmpNQue_;
    TQue<TPosition::VECCALC, 1> tmpMulQue_;

    GlobalTensor<float> aGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileM_{0}, tileN_{0}, tileK_{0};
    uint32_t mTiles_{0}, nTiles_{0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR a, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelMatmulWithTransposedBothCustom op;
    op.Init(a, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.mTiles, tiling_data.nTiles);
    op.Process();
}
