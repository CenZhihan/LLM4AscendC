#include "conv_standard_3d_square_input_square_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
}

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core conservative
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 32768; // avoid too large iteration for latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard3dSquareInputSquareKernelCustomTilingData tiling;

    // Input x: [N, Cin, D, H, W]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetStorageShape().GetDimNum() != 5 || w_shape->GetStorageShape().GetDimNum() != 5) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(1));
    const uint32_t D = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(2));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(3));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetStorageShape().GetDim(4));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(0));
    const uint32_t Cin_w = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(1));
    const uint32_t Kd = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(2));
    const uint32_t Kh = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(3));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetStorageShape().GetDim(4));

    if (Cin != Cin_w || !(Kd == Kh && Kh == Kw)) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t K = Kd; // square kernel

    // stride=1, pad=0, dilation=1
    const uint32_t Dout = (D >= K) ? (D - K + 1) : 0;
    const uint32_t Hout = (H >= K) ? (H - K + 1) : 0;
    const uint32_t Wout = (W >= K) ? (W - K + 1) : 0;

    // Tiling across positions (N * Dout * Hout * Wout); process Cout sequentially inside kernel
    const uint64_t positions = static_cast<uint64_t>(N) * Dout * Hout * Wout;

    // UB usage approximately: yTile (T floats) + weight (Cin*K^3 floats) + a few scalars.
    const uint32_t weightElemsPerOc = Cin * K * K * K;
    uint32_t maxTileByUb = 0;
    if (UB_BUDGET_BYTES > weightElemsPerOc * 4U + 1024U) {
        maxTileByUb = (UB_BUDGET_BYTES - weightElemsPerOc * 4U - 1024U) / 4U;
    }
    if (maxTileByUb == 0) maxTileByUb = ALIGN_ELEMS; // fallback
    maxTileByUb = AlignDown(maxTileByUb, ALIGN_ELEMS);
    if (maxTileByUb > HARD_CAP_ELEMS) {
        maxTileByUb = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    }
    uint32_t tileElems = maxTileByUb;
    if (positions < tileElems) {
        uint32_t aligned = AlignDown(static_cast<uint32_t>(positions), ALIGN_ELEMS);
        tileElems = (aligned > 0) ? aligned : static_cast<uint32_t>(positions);
    }
    if (tileElems == 0) tileElems = 1;

    const uint32_t numTiles = (tileElems == 0) ? 1u : CeilDiv(static_cast<uint32_t>(positions), tileElems);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_D(D);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_K(K);
    tiling.set_Dout(Dout);
    tiling.set_Hout(Hout);
    tiling.set_Wout(Wout);
    tiling.set_tileElems(tileElems);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

    const int64_t Kh = w_shape->GetStorageShape().GetDim(3);
    const int64_t Kw = w_shape->GetStorageShape().GetDim(4);

    if (Cin != Cin_w || !(Kd == Kh && Kh == Kw)) {
        return GRAPH_FAILED;

    const int64_t K = Kd;

    const int64_t Hout = H - K + 1;

kernel_src="""

#include "kernel_operator.h"

class KernelConv3DStandard {
public:
    __aicore__ inline KernelConv3DStandard() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t K, uint32_t Dout, uint32_t Hout, uint32_t Wout,
                                uint32_t tileElems)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        D_ = D; H_ = H; W_ = W;
        K_ = K;
        Dout_ = Dout; Hout_ = Hout; Wout_ = Wout;
        tileElems_ = tileElems;

        totalX_ = N_ * Cin_ * D_ * H_ * W_;
        totalW_ = Cout_ * Cin_ * K_ * K_ * K_;
        totalY_ = N_ * Cout_ * Dout_ * Hout_ * Wout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalX_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, totalW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalY_);

        // UB buffers: y tile, weight of one oc, and several 1-element scalars
        const uint32_t weightElemsPerOc = Cin_ * K_ * K_ * K_;
        pipe_.InitBuffer(outQueueY_, 1, tileElems_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, weightElemsPerOc * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 4, sizeof(float)); // x, w, tmp, acc (all 1-element)
    }

    __aicore__ inline void Process()
    {
        if (Dout_ == 0 || Hout_ == 0 || Wout_ == 0) {
            return;
        }
        if (tileElems_ == 0) {
            return;
        }

        const uint32_t P = N_ * Dout_ * Hout_ * Wout_;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileElems_;
        uint32_t base = blockIdx * tileElems_;
        if (base >= P) return;

        while (base < P) {
            uint32_t curLen = P - base;
            if (curLen > tileElems_) curLen = tileElems_;

            // For each output channel, compute curLen positions of its slice
            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                ComputeTileForOc(base, curLen, oc);
            }

            base += strideElems;
        }
    }

private:
    __aicore__ inline uint32_t OffsetX(uint32_t n, uint32_t c, uint32_t z, uint32_t y, uint32_t x) const
    {
        // [N, Cin, D, H, W]
        return (((n * Cin_ + c) * D_ + z) * H_ + y) * W_ + x;
    }

    __aicore__ inline void DecodePos(uint32_t g, uint32_t &n, uint32_t &od, uint32_t &oh, uint32_t &ow) const
    {
        const uint32_t sp = Hout_ * Wout_;
        const uint32_t bsz = Dout_ * sp;

        n = g / bsz;
        uint32_t rem = g % bsz;
        od = rem / sp;
        uint32_t rem2 = rem % sp;
        oh = rem2 / Wout_;
        ow = rem2 % Wout_;
    }

    __aicore__ inline void ComputeTileForOc(uint32_t base, uint32_t len, uint32_t oc)
    {
        // Load weight for this oc: shape [Cin, K, K, K]
        AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
        const uint32_t wPerOc = Cin_ * K_ * K_ * K_;
        const uint32_t ocWOffset = oc * wPerOc;
        AscendC::DataCopy(wLocal, wGm_[ocWOffset], wPerOc);

        // Allocate y tile in UB
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Allocate scalar buffers (1-element UB tensors)
        AscendC::LocalTensor<float> xS = scalarQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> wS = scalarQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp = scalarQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> acc = scalarQueue_.AllocTensor<float>();

        for (uint32_t t = 0; t < len; ++t) {
            uint32_t g = base + t;
            uint32_t n, od, oh, ow;
            DecodePos(g, n, od, oh, ow);

            // acc = 0
            AscendC::Dup(acc, 0.0f, 1);

            // Convolution sum over Cin and K^3
            for (uint32_t ic = 0; ic < Cin_; ++ic) {
                for (uint32_t kz = 0; kz < K_; ++kz) {
                    for (uint32_t ky = 0; ky < K_; ++ky) {
                        for (uint32_t kx = 0; kx < K_; ++kx) {
                            const uint32_t iz = od + kz;
                            const uint32_t iy = oh + ky;
                            const uint32_t ix = ow + kx;

                            // load input scalar
                            const uint32_t xOff = OffsetX(n, ic, iz, iy, ix);
                            AscendC::DataCopy(xS, xGm_[xOff], 1);

                            // load weight scalar from UB
                            const uint32_t wOffLocal = (((ic * K_ + kz) * K_ + ky) * K_ + kx);
                            AscendC::DataCopy(wS, wLocal[wOffLocal], 1);

                            // tmp = xS * wS; acc += tmp
                            AscendC::Mul(tmp, xS, wS, 1);
                            AscendC::Add(acc, acc, tmp, 1);
                        }
                    }
                }
            }

            // store acc to yLocal[t]
            AscendC::DataCopy(yLocal[t], acc, 1);
        }

        // Copy out to GM: oc slice base is oc * (N * Dout * Hout * Wout)
        const uint32_t P = N_ * Dout_ * Hout_ * Wout_;
        const uint32_t yBase = oc * P + base;
        AscendC::DataCopy(yGm_[yBase], yLocal, len);

        // Free buffers
        scalarQueue_.FreeTensor(acc);
        scalarQueue_.FreeTensor(tmp);
        scalarQueue_.FreeTensor(wS);
        scalarQueue_.FreeTensor(xS);
        outQueueY_.FreeTensor(yLocal);
        wQueue_.FreeTensor(wLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t D_{0}, H_{0}, W_{0};
    uint32_t K_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};
    uint32_t tileElems_{0};

    uint32_t totalX_{0}, totalW_{0}, totalY_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3DStandard op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.K, tiling_data.Dout, tiling_data.Hout, tiling_data.Wout,
            tiling_data.tileElems);
    op.Process();
}
