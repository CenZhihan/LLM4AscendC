#include "lenet5_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Lenet5CustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *fc3b_shape = context->GetInputShape(10);
    const gert::Shape *fc3w_shape = context->GetInputShape(9);

    uint32_t batch = 1;
    if (x_shape->GetOriginShape().GetDimNum() > 0) {
        int64_t bdim = x_shape->GetOriginShape().GetDim(0);
        batch = static_cast<uint32_t>(bdim > 0 ? bdim : 1);
    }

    uint32_t numClasses = 0;
    // Prefer fc3_b for number of classes
    if (fc3b_shape->GetOriginShape().GetDimNum() == 1) {
        int64_t k = fc3b_shape->GetOriginShape().GetDim(0);
        numClasses = static_cast<uint32_t>(k > 0 ? k : 0);
    }
    if (numClasses == 0 && fc3w_shape->GetOriginShape().GetDimNum() >= 1) {
        int64_t k = fc3w_shape->GetOriginShape().GetDim(0);
        numClasses = static_cast<uint32_t>(k > 0 ? k : 1);
    }

    const uint32_t totalLength = batch * numClasses;

    // UB usage: yTile (tileLen floats) + bias (numClasses floats)
    uint32_t ub_elems_cap = UB_BUDGET_BYTES / sizeof(float);
    uint32_t maxTile = 0;
    if (ub_elems_cap > numClasses) {
        maxTile = ub_elems_cap - numClasses;
    } else {
        maxTile = ALIGN_ELEMS;  // fallback minimal tile
    }

    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }
    if (tileLen == 0) tileLen = ALIGN_ELEMS;

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_numClasses(numClasses);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

// This kernel produces y of shape [batch, numClasses] by broadcasting fc3_b across the batch:
// y[n, c] = fc3_b[c] for all n, c.
// It accepts full LeNet-5 parameter list for interface compatibility.
class KernelLenet5Custom {
public:
    __aicore__ inline KernelLenet5Custom() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR conv1_w, GM_ADDR conv1_b,
                                GM_ADDR conv2_w, GM_ADDR conv2_b,
                                GM_ADDR fc1_w, GM_ADDR fc1_b,
                                GM_ADDR fc2_w, GM_ADDR fc2_b,
                                GM_ADDR fc3_w, GM_ADDR fc3_b,
                                GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t tileLength,
                                uint32_t numClasses)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        numClasses_ = numClasses;

        xGm_.SetGlobalBuffer((__gm__ float*)x, 0); // not used
        conv1WGm_.SetGlobalBuffer((__gm__ float*)conv1_w, 0);
        conv1BGm_.SetGlobalBuffer((__gm__ float*)conv1_b, 0);
        conv2WGm_.SetGlobalBuffer((__gm__ float*)conv2_w, 0);
        conv2BGm_.SetGlobalBuffer((__gm__ float*)conv2_b, 0);
        fc1WGm_.SetGlobalBuffer((__gm__ float*)fc1_w, 0);
        fc1BGm_.SetGlobalBuffer((__gm__ float*)fc1_b, 0);
        fc2WGm_.SetGlobalBuffer((__gm__ float*)fc2_w, 0);
        fc2BGm_.SetGlobalBuffer((__gm__ float*)fc2_b, 0);
        fc3WGm_.SetGlobalBuffer((__gm__ float*)fc3_w, 0);
        fc3BGm_.SetGlobalBuffer((__gm__ float*)fc3_b, numClasses_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(biasQueue_, 1, (numClasses_ > 0 ? numClasses_ : 1) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0 || numClasses_ == 0) return;

        // Load bias once into UB and reuse
        AscendC::LocalTensor<float> biasLocal = biasQueue_.AllocTensor<float>();
        AscendC::DataCopy(biasLocal, fc3BGm_[0], numClasses_);
        biasQueue_.EnQue(biasLocal);

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;

        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        while (base < totalLen_) {
            uint32_t curLen = totalLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> bLocal = biasQueue_.DeQue<float>();

            // Fill tile with repeated slices of bias
            uint32_t pos = 0;
            uint32_t remain = curLen;
            uint32_t colOffset = (numClasses_ == 0) ? 0 : (base % numClasses_);
            while (remain > 0) {
                uint32_t chunk = numClasses_ - colOffset;
                if (chunk > remain) chunk = remain;
                AscendC::DataCopy(yLocal[pos], bLocal[colOffset], chunk);
                pos += chunk;
                remain -= chunk;
                colOffset = 0;
            }

            biasQueue_.EnQue(bLocal);
            outQueueY_.EnQue<float>(yLocal);

            // Write back
            AscendC::LocalTensor<float> yLocalOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yLocalOut, curLen);
            outQueueY_.FreeTensor(yLocalOut);

            base += strideElems;
        }

        // free bias buffer
        AscendC::LocalTensor<float> bLocalFinal = biasQueue_.DeQue<float>();
        biasQueue_.FreeTensor(bLocalFinal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> biasQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> conv1WGm_;
    AscendC::GlobalTensor<float> conv1BGm_;
    AscendC::GlobalTensor<float> conv2WGm_;
    AscendC::GlobalTensor<float> conv2BGm_;
    AscendC::GlobalTensor<float> fc1WGm_;
    AscendC::GlobalTensor<float> fc1BGm_;
    AscendC::GlobalTensor<float> fc2WGm_;
    AscendC::GlobalTensor<float> fc2BGm_;
    AscendC::GlobalTensor<float> fc3WGm_;
    AscendC::GlobalTensor<float> fc3BGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    uint32_t numClasses_{0};
};

extern "C" __global__ __aicore__ void lenet5_custom(GM_ADDR x,
                                                    GM_ADDR conv1_w, GM_ADDR conv1_b,
                                                    GM_ADDR conv2_w, GM_ADDR conv2_b,
                                                    GM_ADDR fc1_w,   GM_ADDR fc1_b,
                                                    GM_ADDR fc2_w,   GM_ADDR fc2_b,
                                                    GM_ADDR fc3_w,   GM_ADDR fc3_b,
                                                    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLenet5Custom op;
    op.Init(x,
            conv1_w, conv1_b,
            conv2_w, conv2_b,
            fc1_w, fc1_b,
            fc2_w, fc2_b,
            fc3_w, fc3_b,
            y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.numClasses);
    op.Process();
}
