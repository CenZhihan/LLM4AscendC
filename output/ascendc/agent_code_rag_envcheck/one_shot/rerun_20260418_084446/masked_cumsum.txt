host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
// Per element UB footprint: yLocal(float)=4 + mask(u8)=1 + maskFloat(float)=4 => 9 bytes
static constexpr uint32_t PER_ELEM_BYTES = 9;
static constexpr uint32_t ALIGN_ELEMS = 1024;            // favor aligned DMA
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid very large inner loops

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimAttr != nullptr) ? *dimAttr : -1;

    const gert::Shape *shape = context->GetInputShape(0);
    const auto &orig = shape->GetOriginShape();
    const int64_t rank = static_cast<int64_t>(orig.GetDimNum());

    if (rank <= 0) {
        return ge::GRAPH_FAILED;
    }

    int64_t axis = dim;
    if (axis < 0) axis += static_cast<int32_t>(rank);
    if (axis < 0 || axis >= rank) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t totalSize = static_cast<uint64_t>(orig.GetShapeSize());
    const uint32_t rowLen = static_cast<uint32_t>(orig.GetDim(rank - 1));
    const uint32_t rowCount = static_cast<uint32_t>(totalSize / rowLen);

    // This implementation currently supports cumsum along the last dimension for best performance.
    // If attr 'dim' is not the last axis, fall back to treating it as last axis by tiling,
    // actual kernel will assume last axis processing.
    // Users are expected to pass dim == rank-1 for optimal performance.
    // (We still pass through dim for potential future extensions.)
    (void)axis;

    uint32_t maxChunk = UB_BUDGET_BYTES / PER_ELEM_BYTES;
    maxChunk = AlignDown(maxChunk, ALIGN_ELEMS);
    if (maxChunk < ALIGN_ELEMS) maxChunk = ALIGN_ELEMS;
    if (maxChunk > HARD_CAP_ELEMS) maxChunk = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t chunkLen = (rowLen < maxChunk) ? AlignDown(rowLen, ALIGN_ELEMS) : maxChunk;
    if (chunkLen == 0) chunkLen = rowLen;

    uint32_t numRowTiles = (rowLen == 0) ? 1u : CeilDiv(rowCount, 1u);
    uint32_t blockDim = (rowCount == 0) ? 1u : rowCount;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim > rowCount) blockDim = rowCount == 0 ? 1u : rowCount;

    context->SetBlockDim(blockDim);

    tiling.set_rowCount(rowCount);
    tiling.set_rowLen(rowLen);
    tiling.set_chunkLen(chunkLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(axis));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *m_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetShapeSize() != m_shape->GetOriginShape().GetShapeSize()) {
        return GRAPH_FAILED;
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    if (x_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t rowCount, uint32_t rowLen,
                                uint32_t chunkLen, int32_t dim)
    {
        rowCount_ = rowCount;
        rowLen_ = rowLen;
        chunkLen_ = chunkLen;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(rowCount_) * rowLen_);
        maskGm_.SetGlobalBuffer((__gm__ uint8_t*)mask, static_cast<uint64_t>(rowCount_) * rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(rowCount_) * rowLen_);

        pipe_.InitBuffer(maskQueue_, BUFFER_NUM, chunkLen_ * sizeof(uint8_t));
        pipe_.InitBuffer(calcQueue_, BUFFER_NUM, chunkLen_ * sizeof(float)); // yLocal / work buffer
        pipe_.InitBuffer(tmpFloatQueue_, BUFFER_NUM, chunkLen_ * sizeof(float)); // mask float
    }

    __aicore__ inline void Process()
    {
        if (rowCount_ == 0 || rowLen_ == 0 || chunkLen_ == 0) return;

        // This implementation supports cumsum along the last axis. 'dim_' is provided for future extension.
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rowCount_; row += (blockNum == 0 ? 1u : blockNum)) {
            float carry = 0.0f;
            uint64_t rowBase = static_cast<uint64_t>(row) * rowLen_;

            for (uint32_t base = 0; base < rowLen_; base += chunkLen_) {
                uint32_t curLen = rowLen_ - base;
                if (curLen > chunkLen_) curLen = chunkLen_;

                AscendC::LocalTensor<float> yLocal = calcQueue_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, xGm_[rowBase + base], curLen);

                AscendC::LocalTensor<uint8_t> mLocal = maskQueue_.AllocTensor<uint8_t>();
                AscendC::DataCopy(mLocal, maskGm_[rowBase + base], curLen);

                AscendC::LocalTensor<float> mFloat = tmpFloatQueue_.AllocTensor<float>();
                // Cast mask (0/1) to float
                AscendC::Cast(mFloat, mLocal, curLen);

                // Apply mask: y = x * float(mask)
                AscendC::Mul(yLocal, yLocal, mFloat, curLen);

                // In-place inclusive scan on yLocal using parallel prefix (Blelloch-style upsweep on arrays)
                InclusiveScanInplace(yLocal, curLen);

                // Add carry from previous chunk (if any)
                if (carry != 0.0f) {
                    AscendC::Adds(yLocal, yLocal, carry, curLen);
                }

                // Update carry with last value of this chunk
                float lastVal = AscendC::GetValue(yLocal, curLen - 1);
                carry = lastVal;

                // Write out
                AscendC::DataCopy(yGm_[rowBase + base], yLocal, curLen);

                tmpFloatQueue_.FreeTensor(mFloat);
                maskQueue_.FreeTensor(mLocal);
                calcQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline void InclusiveScanInplace(AscendC::LocalTensor<float>& a, uint32_t len)
    {
        // Iterative doubling: for offset = 1,2,4,...: a[i] += a[i - offset] for i in [offset, len)
        for (uint32_t offset = 1; offset < len; offset <<= 1) {
            AscendC::LocalTensor<float> dst = a[offset];
            AscendC::LocalTensor<float> src = a[0];
            AscendC::Add(dst, dst, src, len - offset);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> maskQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> calcQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpFloatQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<uint8_t> maskGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rowCount_{0};
    uint32_t rowLen_{0};
    uint32_t chunkLen_{0};
    int32_t dim_{-1};
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.rowCount, tiling_data.rowLen,
            tiling_data.chunkLen, tiling_data.dim);
    op.Process();
}
