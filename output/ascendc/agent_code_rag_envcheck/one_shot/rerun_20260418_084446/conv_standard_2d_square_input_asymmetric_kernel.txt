kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t CIn, uint32_t H, uint32_t W,
                                uint32_t COut, uint32_t Kh, uint32_t Kw,
                                uint32_t HOut, uint32_t WOut,
                                uint64_t totalOutput)
    {
        N_ = N; CIn_ = CIn; H_ = H; W_ = W;
        COut_ = COut; Kh_ = Kh; Kw_ = Kw;
        HOut_ = HOut; WOut_ = WOut;
        totalOutput_ = totalOutput;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * CIn_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(COut_) * CIn_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * COut_ * HOut_ * WOut_);

        pipe_.InitBuffer(inQueueX_, 1, sizeof(float));      // scalar buffer for input
        pipe_.InitBuffer(inQueueW_, 1, sizeof(float));      // scalar buffer for weight
        pipe_.InitBuffer(outQueueY_, 1, sizeof(float));     // scalar buffer for output
        pipe_.InitBuffer(calcQueue_, 1, sizeof(float));     // scalar buffer for temp product
    }

    __aicore__ inline void Process()
    {
        if (totalOutput_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t idx = blockIdx; idx < totalOutput_; idx += blockNum) {
            // Map flat index to (n, oc, oh, ow)
            uint64_t t = idx;
            const uint64_t hwOutSize = static_cast<uint64_t>(HOut_) * WOut_;
            const uint64_t cHW = static_cast<uint64_t>(COut_) * hwOutSize;

            const uint32_t n  = static_cast<uint32_t>(t / cHW); t = t % cHW;
            const uint32_t oc = static_cast<uint32_t>(t / hwOutSize); t = t % hwOutSize;
            const uint32_t oh = static_cast<uint32_t>(t / WOut_);
            const uint32_t ow = static_cast<uint32_t>(t % WOut_);

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            // Zero initialize the accumulator
            AscendC::Muls(yLocal, yLocal, 0.0f, 1);

            // Convolution accumulation
            for (uint32_t ic = 0; ic < CIn_; ++ic) {
                for (uint32_t ky = 0; ky < Kh_; ++ky) {
                    const uint32_t ih = oh + ky; // no padding, stride=1
                    for (uint32_t kx = 0; kx < Kw_; ++kx) {
                        const uint32_t iw = ow + kx;

                        // Load input scalar x[n, ic, ih, iw]
                        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                        const uint64_t xOff =
                            (((static_cast<uint64_t>(n) * CIn_ + ic) * H_ + ih) * W_ + iw);
                        AscendC::DataCopy(xLocal, xGm_[xOff], 1);
                        inQueueX_.EnQue(xLocal);

                        // Load weight scalar w[oc, ic, ky, kx]
                        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
                        const uint64_t wOff =
                            (((static_cast<uint64_t>(oc) * CIn_ + ic) * Kh_ + ky) * Kw_ + kx);
                        AscendC::DataCopy(wLocal, wGm_[wOff], 1);
                        inQueueW_.EnQue(wLocal);

                        // Compute product and accumulate
                        AscendC::LocalTensor<float> xSc = inQueueX_.DeQue<float>();
                        AscendC::LocalTensor<float> wSc = inQueueW_.DeQue<float>();
                        AscendC::LocalTensor<float> prod = calcQueue_.AllocTensor<float>();
                        AscendC::Mul(prod, xSc, wSc, 1);
                        AscendC::Add(yLocal, yLocal, prod, 1);

                        calcQueue_.FreeTensor(prod);
                        inQueueX_.FreeTensor(xSc);
                        inQueueW_.FreeTensor(wSc);
                    }
                }
            }

            outQueueY_.EnQue(yLocal);
            // Store output scalar y[n, oc, oh, ow]
            AscendC::LocalTensor<float> yCopy = outQueueY_.DeQue<float>();
            const uint64_t yOff =
                (((static_cast<uint64_t>(n) * COut_ + oc) * HOut_ + oh) * WOut_ + ow);
            AscendC::DataCopy(yGm_[yOff], yCopy, 1);
            outQueueY_.FreeTensor(yCopy);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, CIn_{0}, H_{0}, W_{0};
    uint32_t COut_{0}, Kh_{0}, Kw_{0};
    uint32_t HOut_{0}, WOut_{0};
    uint64_t totalOutput_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvStandard2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.CIn, tiling_data.H, tiling_data.W,
            tiling_data.COut, tiling_data.Kh, tiling_data.Kw,
            tiling_data.HOut, tiling_data.WOut,
            tiling_data.totalOutput);
    op.Process();
}
