host_operator_src="""

#include "cumsum_reverse_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
// Total floats = 4 * tile => bytes = 16 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 128 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // DMA-friendly bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumReverseCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const auto& dims = x_shape->GetOriginShape().GetDims();
    const int64_t rank = static_cast<int64_t>(dims.size());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t* dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int64_t axis = static_cast<int64_t>(*dimAttr);
    if (axis < 0) axis += rank;
    if (axis < 0 || axis >= rank) {
        return ge::GRAPH_FAILED;
    }

    uint64_t totalLength = 1;
    for (auto d : dims) totalLength *= static_cast<uint64_t>(d);

    uint32_t axisLen = static_cast<uint32_t>(dims[axis]);
    uint64_t inner64 = 1;
    for (int64_t i = axis + 1; i < rank; ++i) inner64 *= static_cast<uint64_t>(dims[i]);
    uint32_t inner = static_cast<uint32_t>(inner64);

    // Rows is the number of independent lines to scan: product(dims) / axisLen
    uint64_t rows64 = totalLength / axisLen;
    uint32_t rows = static_cast<uint32_t>(rows64);

    // tileLen from UB budget: bytes = 16 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 16;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (axisLen < tileLen) {
        const uint32_t aligned = AlignDown(axisLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : axisLen;
    }

    // Adaptive blockDim based on rows; keep conservative for stability.
    const uint32_t numTilesPerRow = (tileLen == 0) ? 1u : CeilDiv(axisLen, tileLen);
    (void)numTilesPerRow;
    uint32_t blockDim = rows;
    if (blockDim < 8u) blockDim = (rows > 0) ? rows : 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength));
    tiling.set_rows(rows);
    tiling.set_axisLen(axisLen);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_axis(static_cast<int32_t>(axis));
    tiling.set_inner(inner);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}
} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumsumReverseCustom : public OpDef {
public:
    explicit CumsumReverseCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumsumReverseCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t INOUT_BUFFER_NUM = 2;

class KernelCumsumReverseCustom {
public:
    __aicore__ inline KernelCumsumReverseCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t axisLen,
                                uint32_t tileLen, uint32_t inner)
    {
        rows_ = rows;
        axisLen_ = axisLen;
        tileLen_ = tileLen;
        inner_ = inner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(rows_) * axisLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(rows_) * axisLen_);

        pipe_.InitBuffer(inQueueX_, INOUT_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, INOUT_BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // For this kernel, require axis to be the last dimension so lines are contiguous
        if (inner_ != 1 || axisLen_ == 0 || tileLen_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each core processes a subset of rows with grid-stride
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            DoOneRow(row);
        }
    }

private:
    __aicore__ inline void DoOneRow(uint32_t rowIdx)
    {
        const uint64_t rowBase = static_cast<uint64_t>(rowIdx) * axisLen_;
        uint32_t remaining = axisLen_;
        uint64_t pos = axisLen_; // exclusive end
        float carry = 0.0f;

        while (remaining > 0) {
            uint32_t curLen = (remaining > tileLen_) ? tileLen_ : remaining;
            uint64_t chunkStart = pos - curLen;

            // Load current chunk [chunkStart, chunkStart + curLen)
            LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[rowBase + chunkStart], curLen);
            inQueueX_.EnQue(xLocal);

            // Compute suffix scan in-place and add carry
            LocalTensor<float> buf = inQueueX_.DeQue<float>();
            LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

            // yLocal = buf
            AscendC::DataCopy(yLocal, buf, curLen);

            // In-place suffix scan using iterative doubling:
            // for step in {1,2,4,...} do y[i] += y[i+step] for i in [0, len-step)
            for (uint32_t step = 1; step < curLen; step <<= 1) {
                AscendC::Add(yLocal, yLocal, yLocal[step], curLen - step);
            }

            if (carry != 0.0f) {
                AscendC::Adds(yLocal, yLocal, carry, curLen);
            }

            // Update carry with total sum of this chunk (now at yLocal[0])
            float chunkTotal = yLocal.GetValue(0);
            carry = chunkTotal;

            outQueueY_.EnQue<float>(yLocal);
            inQueueX_.FreeTensor(buf);

            // Store current chunk
            LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[rowBase + chunkStart], yOut, curLen);
            outQueueY_.FreeTensor(yOut);

            remaining -= curLen;
            pos -= curLen;
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INOUT_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, INOUT_BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_{0};
    uint32_t axisLen_{0};
    uint32_t tileLen_{0};
    uint32_t inner_{1};
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelCumsumReverseCustom op;
    op.Init(x, y,
            tiling_data.rows, tiling_data.axisLen,
            tiling_data.tileLength, tiling_data.inner);
    op.Process();
}
