static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling1dCustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    // Expect input shape as (N, C, L)
    int64_t N = 1, C = 1, L = 1;
    if (inShape->GetDimNum() == 3) {
        N = inShape->GetDim(0);
        C = inShape->GetDim(1);
        L = inShape->GetDim(2);
    } else {
        // Fallback: flatten leading dims to rows and last dim as length
        int64_t dimNum = inShape->GetDimNum();
        int64_t len = inShape->GetDim(dimNum - 1);
        int64_t rows = 1;
        for (int i = 0; i < dimNum - 1; ++i) rows *= inShape->GetDim(i);
        N = rows;
        C = 1;
        L = len;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    // Read attributes
    const int64_t *kptr = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *sptr = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *pptr = attrs->GetAttrPointer<int64_t>(2);

    int64_t K = (kptr != nullptr) ? *kptr : 1;
    int64_t S = (sptr != nullptr) ? *sptr : 1;
    int64_t P = (pptr != nullptr) ? *pptr : 0;

    // Output length (ceil_mode = false): floor((L + 2P - K) / S) + 1, clamped to >=0
    int64_t numer = (L + 2 * P - K);
    int64_t outLen64 = (numer >= 0) ? (numer / S + 1) : 0;
    if (outLen64 < 0) outLen64 = 0;

    uint32_t totalRows = static_cast<uint32_t>(N * C);
    uint32_t inLen = static_cast<uint32_t>(L);
    uint32_t outLen = static_cast<uint32_t>(outLen64);

    // Configure blockDim based on rows
    uint32_t blockDim = totalRows;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 32u) blockDim = 32u; // conservative

    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_inLen(inLen);
    tiling.set_outLen(outLen);
    tiling.set_kernel(static_cast<uint32_t>(K));
    tiling.set_stride(static_cast<uint32_t>(S));
    tiling.set_padding(static_cast<uint32_t>(P));
    tiling.set_blockDim(blockDim);

kernel_src="""
#include "kernel_operator.h"

class KernelAveragePooling1dCustom {
public:
    __aicore__ inline KernelAveragePooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalRows, uint32_t inLen, uint32_t outLen,
                                uint32_t kernel, uint32_t stride, uint32_t padding)
    {
        totalRows_ = totalRows;
        inLen_ = inLen;
        outLen_ = outLen;
        k_ = kernel;
        s_ = stride;
        p_ = padding;
        denom_ = (k_ > 0) ? (1.0f / static_cast<float>(k_)) : 0.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(totalRows_ * inLen_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(totalRows_ * outLen_));

        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || (outLen_ == 0) || (k_ == 0)) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride loop across rows
        for (uint32_t row = blockIdx; row < totalRows_; row += blockNum) {
            const uint32_t xBase = row * inLen_;
            const uint32_t yBase = row * outLen_;

            for (uint32_t o = 0; o < outLen_; ++o) {
                int64_t inStart = static_cast<int64_t>(o) * static_cast<int64_t>(s_) - static_cast<int64_t>(p_);
                float acc = 0.0f;

                // Sum K elements with implicit zero padding (count_include_pad = true)
                for (uint32_t k = 0; k < k_; ++k) {
                    int64_t idx = inStart + static_cast<int64_t>(k);
                    if (idx >= 0 && idx < static_cast<int64_t>(inLen_)) {
                        // Load single element from GM
                        // Copy one element into local tensor
                        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tLocal, xGm_[xBase + static_cast<uint32_t>(idx)], 1);
                        // Read back from local buffer
                        float val = 0.0f;
                        AscendC::Duplicate(tLocal, 0.0f, 1); // ensure tLocal is valid in VECCALC space
                        // Since direct scalar read from LocalTensor isn't supported,
                        // perform a second copy into another one-element buffer and use a dummy op sequence:
                        // Instead, we can copy GM -> local and then immediately copy local -> GM to y when needed.
                        // Here we simply use DataCopy into tLocal and then reuse it via pipeline.
                        // To accumulate, we need to transfer to scalar; use mov from register by casting pointer.
                        // For portability in AscendC, we utilize AscendC::GetValue to fetch one element if available.
                        // Fallback: do another DataCopy to a second local then use AscendC::Adds on a single element.

                        // Extract element by copying to a second local and using vector op to accumulate:
                        AscendC::LocalTensor<float> sLocal = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(sLocal, tLocal, 1);
                        // Use Adds to move into sLocal (no change), then copy back to tLocal and read via a temporary scalar
                        // Since direct scalar extraction isn't officially provided, we rely on a workaround:
                        // We'll write directly to output element one by one after finishing accumulation.
                        tmpQueue_.FreeTensor(sLocal);
                        tmpQueue_.FreeTensor(tLocal);

                        // Since extracting scalar isn't feasible here, approximate by loading from GM directly again
                        // to a scalar via intrinsic. Use AscendC::LoadData for scalar if available; otherwise,
                        // we approximate by a second DataCopy and assume the compiler optimizes.
                        // For simplicity, perform direct GM access through DataCopy to a local and then to Y.
                        // But accumulation requires scalar; thus, as a pragmatic approach, re-copy to a temp and
                        // use AscendC math ops on single element chain to accumulate in y, which we delay.
                        // To keep the flow valid, we instead accumulate in a scalar using a built-in intrinsic:
                        acc += AscendC::GetValue(xGm_, xBase + static_cast<uint32_t>(idx));
                    }
                }

                float avg = acc * denom_;
                // Write result into GM at yBase + o
                AscendC::LocalTensor<float> outLocal = tmpQueue_.AllocTensor<float>();
                AscendC::Duplicate(outLocal, avg, 1);
                AscendC::DataCopy(yGm_[yBase + o], outLocal, 1);
                tmpQueue_.FreeTensor(outLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalRows_{0};
    uint32_t inLen_{0};
    uint32_t outLen_{0};
    uint32_t k_{0};
    uint32_t s_{1};
    uint32_t p_{0};
    float denom_{0.0f};
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling1dCustom op;
    op.Init(x, y,
            tiling_data.totalRows, tiling_data.inLen, tiling_data.outLen,
            tiling_data.kernel, tiling_data.stride, tiling_data.padding);
    op.Process();
}
