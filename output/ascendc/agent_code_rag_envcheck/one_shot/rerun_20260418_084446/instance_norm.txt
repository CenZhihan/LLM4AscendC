host_operator_src="""

#include "instance_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 1 * tile
//   outQueueY: 1 * tile
//   tmpQueue: 1 * tile
//   reduceQueue: ~1 float
// Total floats ~= 3 * tile + 1
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // align for DMA efficiency
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap to limit per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    InstanceNormCustomTilingData tiling;

    const auto *shape = context->GetInputShape(0);
    const auto &orig = shape->GetOriginShape();

    uint32_t dimNum = static_cast<uint32_t>(orig.GetDimNum());
    uint64_t N = 1, C = 1, H = 1, W = 1;

    if (dimNum >= 1) N = static_cast<uint64_t>(orig.GetDim(0));
    if (dimNum >= 2) C = static_cast<uint64_t>(orig.GetDim(1));
    if (dimNum >= 3) H = static_cast<uint64_t>(orig.GetDim(2));
    if (dimNum >= 4) W = static_cast<uint64_t>(orig.GetDim(3));

    const uint64_t spatialLen64 = H * W;
    const uint64_t groupCount64 = N * C;
    const uint64_t totalLength64 = groupCount64 * spatialLen64;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = 1.0e-5f;
    if (attrs != nullptr) {
        const float *epsPtr = attrs->GetAttrPointer<float>(0);
        if (epsPtr != nullptr) {
            eps = *epsPtr;
        }
    }

    // tileLen from UB budget: ~ (3 * tile) * 4 bytes
    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t spatialLen = static_cast<uint32_t>(spatialLen64);
    if (spatialLen64 > static_cast<uint64_t>(std::numeric_limits<uint32_t>::max())) {
        spatialLen = HARD_CAP_ELEMS; // fallback
    }
    uint32_t tileLen = maxTile;
    if (spatialLen < tileLen) {
        const uint32_t aligned = AlignDown(spatialLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : spatialLen;
    }

    const uint32_t groupCount = static_cast<uint32_t>(groupCount64);
    uint32_t blockDim = groupCount;
    if (blockDim == 0u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLength64));
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_spatialLen(spatialLen);
    tiling.set_groupCount(groupCount);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class InstanceNormCustom : public OpDef {
public:
    explicit InstanceNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("eps").AttrType(OPTIONAL).Float(1.0e-5);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(InstanceNormCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

class KernelInstanceNormCustom {
public:
    __aicore__ inline KernelInstanceNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                uint32_t spatialLen, uint32_t groupCount,
                                float eps)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        spatialLen_ = spatialLen;
        groupCnt_ = groupCount;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(reduceQueue_, 1, sizeof(float)); // scalar buffer for reductions
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || spatialLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t g = blockIdx; g < groupCnt_; g += (blockNum == 0 ? 1u : blockNum)) {
            const uint32_t base = g * spatialLen_;

            float sum = 0.0f;
            float sumsq = 0.0f;

            // First pass: accumulate sum and sum of squares across spatialLen_
            for (uint32_t off = 0; off < spatialLen_; off += tileLen_) {
                uint32_t len = spatialLen_ - off;
                if (len > tileLen_) len = tileLen_;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base + off], len);

                AscendC::LocalTensor<float> sBuf = reduceQueue_.AllocTensor<float>();
                AscendC::ReduceSum(sBuf, xLocal, len);
                float tileSum = sBuf.GetValue(0);
                reduceQueue_.FreeTensor(sBuf);

                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                AscendC::Mul(tmp, xLocal, xLocal, len);

                AscendC::LocalTensor<float> s2Buf = reduceQueue_.AllocTensor<float>();
                AscendC::ReduceSum(s2Buf, tmp, len);
                float tileSumSq = s2Buf.GetValue(0);
                reduceQueue_.FreeTensor(s2Buf);

                tmpQueue_.FreeTensor(tmp);
                inQueueX_.FreeTensor(xLocal);

                sum += tileSum;
                sumsq += tileSumSq;
            }

            const float count = static_cast<float>(spatialLen_);
            const float mean = sum / count;
            const float var = sumsq / count - mean * mean;
            const float invStd = 1.0f / sqrtf(var + eps_);

            // Second pass: normalize and write out
            for (uint32_t off = 0; off < spatialLen_; off += tileLen_) {
                uint32_t len = spatialLen_ - off;
                if (len > tileLen_) len = tileLen_;

                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base + off], len);

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

                AscendC::Adds(yLocal, xLocal, -mean, len);     // y = x - mean
                AscendC::Muls(yLocal, yLocal, invStd, len);    // y = y * invStd

                AscendC::DataCopy(yGm_[base + off], yLocal, len);

                outQueueY_.FreeTensor(yLocal);
                inQueueX_.FreeTensor(xLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> reduceQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t spatialLen_ {0};
    uint32_t groupCnt_ {0};
    float eps_ {1.0e-5f};
};

extern "C" __global__ __aicore__ void instance_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelInstanceNormCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.tileLength,
            tiling_data.spatialLen,
            tiling_data.groupCount,
            tiling_data.eps);
    op.Process();
}
"""
