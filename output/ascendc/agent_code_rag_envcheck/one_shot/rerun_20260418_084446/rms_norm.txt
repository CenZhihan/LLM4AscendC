host_operator_src="""

#include "rms_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
// Buffers: x (1) + y (1) + acc/inv (1) + tmp (1) = 4 * tile
static constexpr uint32_t BYTES_PER_ELEM = 4;
static constexpr uint32_t BUFFERS_PER_TILE = 4;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint64_t ProductExcept(const std::vector<int64_t>& dims, size_t start)
{
    uint64_t p = 1;
    for (size_t i = start; i < dims.size(); ++i) {
        p *= static_cast<uint64_t>(dims[i] < 0 ? 0 : dims[i]);
    }
    return p;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    RmsNormCustomTilingData tiling;

    const gert::Shape* shape = context->GetInputShape(0);
    const gert::ShapeView& origin = shape->GetOriginShape();

    const int32_t dimNum = origin.GetDimNum();
    if (dimNum < 2) {
        return ge::GRAPH_FAILED;
    }

    std::vector<int64_t> dims(dimNum);
    for (int32_t i = 0; i < dimNum; ++i) {
        dims[i] = origin.GetDim(i);
    }

    uint64_t n64 = static_cast<uint64_t>(dims[0]);
    uint64_t c64 = static_cast<uint64_t>(dims[1]);
    uint64_t m64 = (dimNum > 2) ? ProductExcept(dims, 2) : 1ULL;

    if (c64 == 0 || n64 == 0 || m64 == 0) {
        // Degenerate, but keep minimal tiling to avoid div-by-zero
        n64 = n64 == 0 ? 1 : n64;
        c64 = c64 == 0 ? 1 : c64;
        m64 = m64 == 0 ? 1 : m64;
    }

    // Determine tile size along M dimension
    const uint32_t maxTileElems = UB_BUDGET_BYTES / (BUFFERS_PER_TILE * BYTES_PER_ELEM);
    uint32_t tileM = maxTileElems;
    if (tileM > HARD_CAP_ELEMS) tileM = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    if (tileM < ALIGN_ELEMS) tileM = ALIGN_ELEMS;

    const uint32_t m = static_cast<uint32_t>(m64);
    if (m < tileM) {
        const uint32_t aligned = AlignDown(m, ALIGN_ELEMS);
        tileM = (aligned > 0) ? aligned : m;
    }

    const uint32_t n = static_cast<uint32_t>(n64);
    const uint32_t c = static_cast<uint32_t>(c64);
    const uint32_t numTilesM = (tileM == 0) ? 1u : CeilDiv(m, tileM);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsAttr = attrs->GetAttrPointer<float>(0);
    const float eps = (epsAttr != nullptr) ? *epsAttr : 1e-5f;

    // Choose blockDim based on total groups (n * numTilesM)
    uint32_t blockDim = n * numTilesM;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_nSize(n);
    tiling.set_cSize(c);
    tiling.set_mSize(m);
    tiling.set_tileM(tileM);
    tiling.set_numTilesM(numTilesM);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelRmsNormCustom {
public:
    __aicore__ inline KernelRmsNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t nSize, uint32_t cSize, uint32_t mSize,
                                uint32_t tileM, uint32_t numTilesM,
                                float eps)
    {
        n_ = nSize;
        c_ = cSize;
        m_ = mSize;
        tileM_ = tileM;
        numTilesM_ = numTilesM;
        eps_ = eps;
        invC_ = (c_ > 0) ? (1.0f / static_cast<float>(c_)) : 0.0f;
        totalLen_ = n_ * c_ * m_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileM_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileM_ * sizeof(float));
        pipe_.InitBuffer(calcQueue_, 1, tileM_ * sizeof(float));  // acc/inv buffer
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * sizeof(float));   // temp for x^2
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileM_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalGroups = n_ * numTilesM_;
        for (uint32_t g = blockIdx; g < totalGroups; g += blockNum) {
            const uint32_t nIdx = g / numTilesM_;
            const uint32_t tIdx = g % numTilesM_;
            const uint32_t mBase = tIdx * tileM_;
            uint32_t curM = m_ - mBase;
            if (curM > tileM_) curM = tileM_;
            ComputeGroup(nIdx, mBase, curM);
        }
    }

private:
    __aicore__ inline void LoadX(__gm__ float* base, uint32_t offset, AscendC::LocalTensor<float>& xLocal, uint32_t len)
    {
        AscendC::DataCopy(xLocal, base + offset, len);
    }

    __aicore__ inline void StoreY(__gm__ float* base, uint32_t offset, AscendC::LocalTensor<float>& yLocal, uint32_t len)
    {
        AscendC::DataCopy(base + offset, yLocal, len);
    }

    __aicore__ inline void ComputeGroup(uint32_t nIdx, uint32_t mBase, uint32_t lenM)
    {
        // accLocal will hold sum of squares initially, then converted in-place to invRMS
        AscendC::LocalTensor<float> accLocal = calcQueue_.AllocTensor<float>();
        AscendC::Duplicate(accLocal, 0.0f, lenM);

        // First pass: accumulate sum of squares across C
        for (uint32_t cIdx = 0; cIdx < c_; ++cIdx) {
            const uint32_t gmOffset = ((nIdx * c_ + cIdx) * m_) + mBase;

            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            LoadX(xGm_.GetBuffer(), gmOffset, xLocal, lenM);
            inQueueX_.EnQue(xLocal);
            AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();

            AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Mul(tmpLocal, xTile, xTile, lenM);          // x^2
            AscendC::Add(accLocal, accLocal, tmpLocal, lenM);    // accumulate

            tmpQueue_.FreeTensor(tmpLocal);
            inQueueX_.FreeTensor(xTile);
        }

        // accLocal = accLocal * (1/C)
        AscendC::Muls(accLocal, accLocal, invC_, lenM);
        // accLocal = accLocal + eps
        AscendC::Adds(accLocal, accLocal, eps_, lenM);
        // accLocal = rsqrt(accLocal) -> invRMS
        AscendC::Rsqrt(accLocal, accLocal, lenM);

        // Second pass: normalize each channel with invRMS
        for (uint32_t cIdx = 0; cIdx < c_; ++cIdx) {
            const uint32_t gmOffset = ((nIdx * c_ + cIdx) * m_) + mBase;

            // Load X
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            LoadX(xGm_.GetBuffer(), gmOffset, xLocal, lenM);
            inQueueX_.EnQue(xLocal);
            AscendC::LocalTensor<float> xTile = inQueueX_.DeQue<float>();

            // Compute Y = X * invRMS
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Mul(yLocal, xTile, accLocal, lenM);

            // Store Y
            outQueueY_.EnQue(yLocal);
            AscendC::LocalTensor<float> yTile = outQueueY_.DeQue<float>();
            StoreY(yGm_.GetBuffer(), gmOffset, yTile, lenM);

            outQueueY_.FreeTensor(yTile);
            inQueueX_.FreeTensor(xTile);
        }

        calcQueue_.FreeTensor(accLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0};
    uint32_t c_{0};
    uint32_t m_{0};
    uint32_t tileM_{0};
    uint32_t numTilesM_{0};
    uint32_t totalLen_{0};
    float eps_{1e-5f};
    float invC_{0.0f};
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNormCustom op;
    op.Init(x, y,
            tiling_data.nSize, tiling_data.cSize, tiling_data.mSize,
            tiling_data.tileM, tiling_data.numTilesM,
            tiling_data.eps);
    op.Process();
}
"""
