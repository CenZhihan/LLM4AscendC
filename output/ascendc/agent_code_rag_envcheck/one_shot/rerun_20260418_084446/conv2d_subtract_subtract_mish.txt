    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t WCin = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t KH = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t KW = static_cast<uint32_t>(w_shape->GetDim(3));

    if (WCin != Cin) {
        return ge::GRAPH_FAILED;
    }

    // stride=1, pad=0, dilation=1
    if (H < KH || W < KW) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t OH = H - KH + 1;
    const uint32_t OW = W - KW + 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *sub1 = attrs->GetAttrPointer<float>(0);
    const float *sub2 = attrs->GetAttrPointer<float>(1);

    // Determine if bias is provided (3rd input present and sized correctly)
    uint32_t hasBias = 0;
    if (context->GetInputsNum() >= 3) {
        const gert::Shape *b_shape = context->GetInputShape(2);
        if (b_shape != nullptr && b_shape->GetShapeSize() == Cout) {
            hasBias = 1;
        }
    }

    // Blocks: distribute by output spatial pixels
    const uint64_t totalPix = static_cast<uint64_t>(N) * OH * OW;
    uint32_t blockDim = static_cast<uint32_t>(totalPix);
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 32u) blockDim = 32u;  // conservative upper bound

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_KH(KH);
    tiling.set_KW(KW);
    tiling.set_OH(OH);
    tiling.set_OW(OW);
    tiling.set_hasBias(hasBias);
    tiling.set_blockDim(blockDim);
    tiling.set_subtractValue1(sub1 ? *sub1 : 0.0f);
    tiling.set_subtractValue2(sub2 ? *sub2 : 0.0f);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

class KernelConv2dSubtractSubtractMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractSubtractMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR w,
                                GM_ADDR b,
                                GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W,
                                uint32_t KH, uint32_t KW,
                                uint32_t OH, uint32_t OW,
                                uint32_t hasBias,
                                float sub1, float sub2)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; KH_ = KH; KW_ = KW;
        OH_ = OH; OW_ = OW;
        hasBias_ = hasBias;
        sub1_ = sub1; sub2_ = sub2;
        Kdim_ = Cin_ * KH_ * KW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_*Cin_*H_*W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cout_*Cin_*KH_*KW_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_*Cout_*OH_*OW_);

        // UB buffers
        pipe_.InitBuffer(wBufQ_, 1, Kdim_ * sizeof(float));
        pipe_.InitBuffer(patchBufQ_, 1, Kdim_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, 3, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || OH_ == 0 || OW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t totalPix = (uint64_t)N_ * OH_ * OW_;

        // Each block processes a grid-stride sequence of spatial pixels
        for (uint64_t pix = blockIdx; pix < totalPix; pix += blockNum) {
            uint32_t n, oh, ow;
            UnravelPixel(pix, n, oh, ow);

            // For each output channel
            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                // Load weights for this oc
                AscendC::LocalTensor<float> wLocal = wBufQ_.AllocTensor<float>();
                const uint64_t wBase = (uint64_t)oc * Kdim_;
                AscendC::DataCopy(wLocal, wGm_[wBase], Kdim_);
                wBufQ_.EnQue(wLocal);

                // Build input patch vector for this (n, oh, ow)
                AscendC::LocalTensor<float> patchLocal = patchBufQ_.AllocTensor<float>();
                uint32_t off = 0;
                for (uint32_t c = 0; c < Cin_; ++c) {
                    for (uint32_t r = 0; r < KH_; ++r) {
                        const uint64_t xBase = IndexX(n, c, oh + r, ow);
                        AscendC::DataCopy(patchLocal[off], xGm_[xBase], KW_);
                        off += KW_;
                    }
                }
                patchBufQ_.EnQue(patchLocal);

                // Consume buffers to compute dot product
                AscendC::LocalTensor<float> wLocalC = wBufQ_.DeQue<float>();
                AscendC::LocalTensor<float> patchLocalC = patchBufQ_.DeQue<float>();

                float acc = 0.0f;
                for (uint32_t i = 0; i < Kdim_; ++i) {
                    acc += patchLocalC.GetValue(i) * wLocalC.GetValue(i);
                }

                // Add bias if present
                if (hasBias_) {
                    acc += bGm_[oc];
                }

                // Apply subtracts
                acc -= sub1_;
                acc -= sub2_;

                // Mish: y = x * tanh(ln(1 + exp(x)))
                AscendC::LocalTensor<float> valLocal = scalarQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpLocal = scalarQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> actLocal = scalarQ_.AllocTensor<float>();
                valLocal.SetValue(0, acc);
                AscendC::Exp(tmpLocal, valLocal, 1);          // tmp = exp(x)
                AscendC::Adds(tmpLocal, tmpLocal, 1.0f, 1);   // tmp = 1 + exp(x)
                AscendC::Ln(tmpLocal, tmpLocal, 1);           // tmp = ln(1+exp(x))
                AscendC::Tanh(actLocal, tmpLocal, 1);         // act = tanh(tmp)
                AscendC::Mul(actLocal, actLocal, valLocal, 1);// act = x * tanh(...)
                float outv = actLocal.GetValue(0);

                scalarQ_.FreeTensor(actLocal);
                scalarQ_.FreeTensor(tmpLocal);
                scalarQ_.FreeTensor(valLocal);

                // Store result
                const uint64_t yIndex = IndexY(n, oc, oh, ow);
                AscendC::LocalTensor<float> yScalar = scalarQ_.AllocTensor<float>();
                yScalar.SetValue(0, outv);
                AscendC::DataCopy(yGm_[yIndex], yScalar, 1);
                scalarQ_.FreeTensor(yScalar);

                wBufQ_.FreeTensor(wLocalC);
                patchBufQ_.FreeTensor(patchLocalC);
            }
        }
    }

private:
    __aicore__ inline uint64_t IndexX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * Cin_ + c) * H_ + h) * W_ + w;
    }
    __aicore__ inline uint64_t IndexY(uint32_t n, uint32_t oc, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * Cout_ + oc) * OH_ + h) * OW_ + w;
    }
    __aicore__ inline void UnravelPixel(uint64_t pix, uint32_t &n, uint32_t &oh, uint32_t &ow) const
    {
        const uint64_t hw = (uint64_t)OH_ * OW_;
        n = static_cast<uint32_t>(pix / hw);
        const uint64_t rem = pix % hw;
        oh = static_cast<uint32_t>(rem / OW_);
        ow = static_cast<uint32_t>(rem % OW_);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> patchBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, KH_{0}, KW_{0}, OH_{0}, OW_{0};
    uint32_t Kdim_{0};
    uint32_t hasBias_{0};
    float sub1_{0.0f}, sub2_{0.0f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_subtract_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractSubtractMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W,
            tiling_data.KH, tiling_data.KW,
            tiling_data.OH, tiling_data.OW,
            tiling_data.hasBias,
            tiling_data.subtractValue1, tiling_data.subtractValue2);
    op.Process();
}
