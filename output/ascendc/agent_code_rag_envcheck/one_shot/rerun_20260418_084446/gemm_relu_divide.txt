host_operator_src="""

#include "gemm_relu_divide_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

} // anonymous

namespace optiling {

// A conservative UB budget per-core (bytes). We keep well below hardware limit for safety.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// Alignment in elements to favor better DMA efficiency.
static constexpr uint32_t ALIGN_ELEMS = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmReluDivideCustomTilingData tiling;

    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);
    const auto *b_shape = context->GetInputShape(2);
    (void)b_shape; // bias only checked for length consistency by framework

    // Expect: x [M, K], w [N, K], b [N]
    const auto x_origin = x_shape->GetOriginShape();
    const auto w_origin = w_shape->GetOriginShape();

    if (x_origin.GetDimNum() != 2 || w_origin.GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_origin.GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_origin.GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_origin.GetDim(0));
    const uint32_t K_w = static_cast<uint32_t>(w_origin.GetDim(1));
    if (K != K_w) {
        return ge::GRAPH_FAILED;
    }

    // Choose tile sizes based on UB budget:
    // UB holds: X_t (tileM*tileK) + W_t (tileN*tileK) + C (tileM*tileN) + B (tileN)
    // Total floats = tileM*tileK + tileN*tileK + tileM*tileN + tileN
    // Bytes ~ floats * 4
    uint32_t maxFloats = UB_BUDGET_BYTES / 4;
    // Start from pragmatic bases
    uint32_t tileM = 16;
    uint32_t tileN = 64;
    uint32_t tileK = 256;

    // If too large, reduce tileK first, then tileN, then tileM
    auto Fits = [&](uint32_t m, uint32_t n, uint32_t k) -> bool {
        uint64_t need = static_cast<uint64_t>(m) * k + static_cast<uint64_t>(n) * k
                        + static_cast<uint64_t>(m) * n + n + 256; // small safety margin
        return need <= maxFloats;
    };

    // Align to ALIGN_ELEMS for better DMA
    tileM = AlignDown(tileM, ALIGN_ELEMS);
    if (tileM == 0) tileM = ALIGN_ELEMS;
    tileN = AlignDown(tileN, ALIGN_ELEMS);
    if (tileN == 0) tileN = ALIGN_ELEMS;
    tileK = AlignDown(tileK, ALIGN_ELEMS);
    if (tileK == 0) tileK = ALIGN_ELEMS;

    while (!Fits(tileM, tileN, tileK) && tileK > ALIGN_ELEMS) tileK -= ALIGN_ELEMS;
    while (!Fits(tileM, tileN, tileK) && tileN > ALIGN_ELEMS) tileN -= ALIGN_ELEMS;
    while (!Fits(tileM, tileN, tileK) && tileM > ALIGN_ELEMS) tileM -= ALIGN_ELEMS;

    if (!Fits(tileM, tileN, tileK)) {
        // Fallback minimal tiling
        tileM = AlignUp(1u, ALIGN_ELEMS);
        tileN = AlignUp(1u, ALIGN_ELEMS);
        tileK = AlignUp(1u, ALIGN_ELEMS);
    }

    // Make sure tiles are not larger than problem dimensions
    if (tileM > M) tileM = (M >= ALIGN_ELEMS) ? AlignDown(M, ALIGN_ELEMS) : M;
    if (tileN > N) tileN = (N >= ALIGN_ELEMS) ? AlignDown(N, ALIGN_ELEMS) : N;
    if (tileK > K) tileK = (K >= ALIGN_ELEMS) ? AlignDown(K, ALIGN_ELEMS) : K;

    if (tileM == 0) tileM = M > 0 ? M : 1;
    if (tileN == 0) tileN = N > 0 ? N : 1;
    if (tileK == 0) tileK = K > 0 ? K : 1;

    // Blocks: parallelize along M tiles
    uint32_t mTiles = (tileM == 0) ? 1u : CeilDiv(M, tileM);
    uint32_t blockDim = mTiles;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUF_NUM = 1;

class KernelGemmReluDivideCustom {
public:
    __aicore__ inline KernelGemmReluDivideCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN; tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        dGm_.SetGlobalBuffer((__gm__ float*)divisor, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * N_);

        // UB buffers
        pipe_.InitBuffer(xTileQ_, BUF_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(wTileQ_, BUF_NUM, tileN_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(cTileQ_, BUF_NUM, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(bTileQ_, BUF_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, BUF_NUM, sizeof(float));
        pipe_.InitBuffer(tmpVecQ_, BUF_NUM, tileN_ * sizeof(float));  // temp vector for outer-product accum
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) { return; }

        // Load divisor scalar once per block
        float invDiv = 1.0f;
        {
            AscendC::LocalTensor<float> dLocal = scalarQ_.AllocTensor<float>();
            AscendC::DataCopy(dLocal, dGm_[0], 1);
            scalarQ_.EnQue<float>(dLocal);
            AscendC::LocalTensor<float> dLocal2 = scalarQ_.DeQue<float>();
            float dval = *(reinterpret_cast<volatile float*>(dLocal2.GetPhyAddr()));
            invDiv = 1.0f / dval;
            scalarQ_.FreeTensor(dLocal2);
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t mTileStride = blockNum * tileM_;
        uint32_t mBase = blockIdx * tileM_;

        for (; mBase < M_; mBase += mTileStride) {
            const uint32_t curM = (mBase + tileM_ <= M_) ? tileM_ : (M_ - mBase);
            if (curM == 0) break;

            for (uint32_t nBase = 0; nBase < N_; nBase += tileN_) {
                const uint32_t curN = (nBase + tileN_ <= N_) ? tileN_ : (N_ - nBase);

                // Allocate C tile and zero it
                AscendC::LocalTensor<float> cTile = cTileQ_.AllocTensor<float>();
                AscendC::Dup<float>(cTile, 0.0f, curM * curN);

                // K loop
                for (uint32_t kBase = 0; kBase < K_; kBase += tileK_) {
                    const uint32_t curK = (kBase + tileK_ <= K_) ? tileK_ : (K_ - kBase);

                    // Load X tile rows [curM x curK]
                    AscendC::LocalTensor<float> xTile = xTileQ_.AllocTensor<float>();
                    for (uint32_t mi = 0; mi < curM; ++mi) {
                        const uint64_t xOff = static_cast<uint64_t>(mBase + mi) * K_ + kBase;
                        AscendC::DataCopy(xTile[mi * tileK_], xGm_[xOff], curK);
                    }
                    xTileQ_.EnQue<float>(xTile);

                    // Load W tile rows [curN x curK]
                    AscendC::LocalTensor<float> wTile = wTileQ_.AllocTensor<float>();
                    for (uint32_t ni = 0; ni < curN; ++ni) {
                        const uint64_t wOff = static_cast<uint64_t>(nBase + ni) * K_ + kBase;
                        AscendC::DataCopy(wTile[ni * tileK_], wGm_[wOff], curK);
                    }
                    wTileQ_.EnQue<float>(wTile);

                    // Compute C += X * W^T over current K chunk using outer-product style
                    AscendC::LocalTensor<float> xT = xTileQ_.DeQue<float>();
                    AscendC::LocalTensor<float> wT = wTileQ_.DeQue<float>();
                    AscendC::LocalTensor<float> tmpVec = tmpVecQ_.AllocTensor<float>();

                    // Pointers to local buffers (UB)
                    float* cPtr = reinterpret_cast<float*>(cTile.GetPhyAddr());
                    float* xPtr = reinterpret_cast<float*>(xT.GetPhyAddr());
                    float* wPtr = reinterpret_cast<float*>(wT.GetPhyAddr());
                    float* tmpPtr = reinterpret_cast<float*>(tmpVec.GetPhyAddr());

                    // For each kk in curK:
                    for (uint32_t kk = 0; kk < curK; ++kk) {
                        // Build contiguous vector tmpPtr[0:curN] = w[:, kk] for this tile
                        for (uint32_t ni = 0; ni < curN; ++ni) {
                            tmpPtr[ni] = wPtr[ni * tileK_ + kk];
                        }
                        // Accumulate outer-product for each row m: C[m,0:curN] += tmpVec * x[m, kk]
                        for (uint32_t mi = 0; mi < curM; ++mi) {
                            const float aval = xPtr[mi * tileK_ + kk];
                            float* cRow = cPtr + mi * curN;
                            for (uint32_t ni = 0; ni < curN; ++ni) {
                                cRow[ni] += tmpPtr[ni] * aval;
                            }
                        }
                    }

                    tmpVecQ_.FreeTensor(tmpVec);
                    xTileQ_.FreeTensor(xT);
                    wTileQ_.FreeTensor(wT);
                }

                // Load bias for current N tile
                AscendC::LocalTensor<float> bTile = bTileQ_.AllocTensor<float>();
                AscendC::DataCopy(bTile, bGm_[nBase], curN);

                // Apply bias, ReLU, and division by divisor
                float* cPtr2 = reinterpret_cast<float*>(cTile.GetPhyAddr());
                float* bPtr = reinterpret_cast<float*>(bTile.GetPhyAddr());
                for (uint32_t mi = 0; mi < curM; ++mi) {
                    float* cRow = cPtr2 + mi * curN;
                    for (uint32_t ni = 0; ni < curN; ++ni) {
                        float v = cRow[ni] + bPtr[ni];
                        v = v > 0.0f ? v : 0.0f;
                        v = v * invDiv;
                        cRow[ni] = v;
                    }
                }

                // Store Y tile rows
                for (uint32_t mi = 0; mi < curM; ++mi) {
                    const uint64_t yOff = static_cast<uint64_t>(mBase + mi) * N_ + nBase;
                    AscendC::DataCopy(yGm_[yOff], cTile[mi * curN], curN);
                }

                bTileQ_.FreeTensor(bTile);
                cTileQ_.FreeTensor(cTile);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_NUM> xTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_NUM> wTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_NUM> cTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUF_NUM> bTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_NUM> tmpVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUF_NUM> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> dGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileM_{0}, tileN_{0}, tileK_{0};
};

extern "C" __global__ __aicore__ void gemm_relu_divide_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmReluDivideCustom op;
    op.Init(x, w, b, divisor, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
