kernel_src="""

#include "kernel_operator.h"

class KernelConv2dSubtractTanhSubtractAvgPoolCustom {
public:
    __aicore__ inline KernelConv2dSubtractTanhSubtractAvgPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t O, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t PoolK, uint32_t Hp, uint32_t Wp,
                                float sub1, float sub2, uint64_t totalLen)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        O_ = O; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;
        PoolK_ = PoolK; Hp_ = Hp; Wp_ = Wp;
        sub1_ = sub1; sub2_ = sub2; totalLen_ = totalLen;

        const uint64_t xSize = static_cast<uint64_t>(N_) * C_ * H_ * W_;
        const uint64_t wSize = static_cast<uint64_t>(O_) * C_ * Kh_ * Kw_;
        const uint64_t bSize = static_cast<uint64_t>(O_);
        const uint64_t ySize = static_cast<uint64_t>(N_) * O_ * Hp_ * Wp_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        bGm_.SetGlobalBuffer((__gm__ float*)b, bSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(inBuf_, 1, sizeof(float));
        pipe_.InitBuffer(wBuf_, 1, sizeof(float));
        pipe_.InitBuffer(calcBuf_, 3, sizeof(float)); // acc, tmp, out
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint64_t idx = blockIdx; idx < totalLen_; idx += blockNum) {
            ComputeOne(idx);
        }
    }

private:
    __aicore__ inline void ComputeOne(uint64_t flatIdx)
    {
        // Map flatIdx -> (n, oc, py, px)
        uint64_t tmp = flatIdx;
        const uint32_t Wp = Wp_;
        const uint32_t Hp = Hp_;
        const uint32_t O = O_;

        const uint32_t px = (uint32_t)(tmp % Wp); tmp /= Wp;
        const uint32_t py = (uint32_t)(tmp % Hp); tmp /= Hp;
        const uint32_t oc = (uint32_t)(tmp % O);  tmp /= O;
        const uint32_t n  = (uint32_t)tmp;

        AscendC::LocalTensor<float> accPool = calcBuf_.AllocTensor<float>();
        AscendC::Duplicate(accPool, 0.0f, 1);

        // pool accumulation over PoolK x PoolK
        for (uint32_t dy = 0; dy < PoolK_; ++dy) {
            for (uint32_t dx = 0; dx < PoolK_; ++dx) {
                // conv output coordinates
                const uint32_t cy = py * PoolK_ + dy;
                const uint32_t cx = px * PoolK_ + dx;

                // Accumulator for convolution result
                AscendC::LocalTensor<float> accConv = calcBuf_.AllocTensor<float>();

                // Initialize with bias
                // load bias[oc]
                AscendC::LocalTensor<float> biasLocal = calcBuf_.AllocTensor<float>();
                AscendC::DataCopy(biasLocal, bGm_[oc], 1);
                AscendC::DataCopy(accConv, biasLocal, 1);

                // Perform convolution: sum over ci, ky, kx
                for (uint32_t ci = 0; ci < C_; ++ci) {
                    for (uint32_t ky = 0; ky < Kh_; ++ky) {
                        const uint32_t in_y = cy + ky;
                        for (uint32_t kx = 0; kx < Kw_; ++kx) {
                            const uint32_t in_x = cx + kx;

                            const uint64_t xOff =
                                (((uint64_t)n * C_ + ci) * H_ + in_y) * W_ + in_x;
                            const uint64_t wOff =
                                ((((uint64_t)oc * C_ + ci) * Kh_ + ky) * Kw_) + kx;

                            // Load x and w scalars into local buffers
                            AscendC::LocalTensor<float> xLocal = inBuf_.AllocTensor<float>();
                            AscendC::LocalTensor<float> wLocal = wBuf_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[xOff], 1);
                            AscendC::DataCopy(wLocal, wGm_[wOff], 1);

                            // tmp = x * w; accConv += tmp
                            AscendC::LocalTensor<float> tmpMul = calcBuf_.AllocTensor<float>();
                            AscendC::Mul(tmpMul, xLocal, wLocal, 1);
                            AscendC::Add(accConv, accConv, tmpMul, 1);

                            calcBuf_.FreeTensor(tmpMul);
                            wBuf_.FreeTensor(wLocal);
                            inBuf_.FreeTensor(xLocal);
                        }
                    }
                }

                // Nonlinear and subtracts on the conv result (length 1 tensor)
                AscendC::Adds(accConv, accConv, -sub1_, 1);
                AscendC::Tanh(accConv, accConv, 1);
                AscendC::Adds(accConv, accConv, -sub2_, 1);

                // pool accumulate
                AscendC::Add(accPool, accPool, accConv, 1);

                calcBuf_.FreeTensor(biasLocal);
                calcBuf_.FreeTensor(accConv);
            }
        }

        // average pooling: divide by area
        const float invArea = 1.0f / (float)(PoolK_ * PoolK_);
        AscendC::Muls(accPool, accPool, invArea, 1);

        // write out
        const uint64_t yOff = (((uint64_t)n * O_ + oc) * Hp_ + py) * Wp_ + px;
        AscendC::DataCopy(yGm_[yOff], accPool, 1);

        calcBuf_.FreeTensor(accPool);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inBuf_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3> calcBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t O_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t PoolK_{0}, Hp_{0}, Wp_{0};
    float sub1_{0.0f}, sub2_{0.0f};
    uint64_t totalLen_{0};
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractTanhSubtractAvgPoolCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.O, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo,
            tiling_data.PoolK, tiling_data.Hp, tiling_data.Wp,
            tiling_data.subtract1, tiling_data.subtract2, tiling_data.totalLen);
    op.Process();
}
