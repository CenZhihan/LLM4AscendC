host_operator_src="""

#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB usage model (floats):
//   xSeg: tileN
//   wSeg: tileN
//   accK: tileK
// Total floats = (2 * tileN + tileK) => bytes = 4 * (2 * tileN + tileK)
//
// Budget and alignment constraints
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // favor medium DMA bursts
static constexpr uint32_t MAX_TILE_N = 4096;             // practical ceiling
static constexpr uint32_t MAX_TILE_K = 2048;             // practical ceiling
static constexpr uint32_t HARD_CAP_FLOATS = (UB_BUDGET_BYTES / 4);

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;

    // Shapes
    const auto &xShape = context->GetInputShape(0)->GetOriginShape();
    const auto &wShape = context->GetInputShape(1)->GetOriginShape();
    // x: [M, N], w: [K, N], bias: [K]
    const uint32_t M = static_cast<uint32_t>(xShape.GetDim(0));
    const uint32_t N = static_cast<uint32_t>(xShape.GetDim(1));
    const uint32_t K = static_cast<uint32_t>(wShape.GetDim(0));

    // Start from preferred tile sizes
    uint32_t tileN = (N >= MAX_TILE_N) ? MAX_TILE_N : N;
    uint32_t tileK = (K >= MAX_TILE_K) ? MAX_TILE_K : K;

    tileN = AlignDown(tileN, ALIGN_ELEMS);
    if (tileN == 0) tileN = (N == 0) ? 0u : (N < ALIGN_ELEMS ? N : ALIGN_ELEMS);

    // Satisfy UB budget: 2*tileN + tileK <= HARD_CAP_FLOATS
    if (2u * tileN + tileK > HARD_CAP_FLOATS) {
        // Favor keeping tileN reasonably large
        const uint32_t maxTileK = (HARD_CAP_FLOATS > 2u * tileN) ? (HARD_CAP_FLOATS - 2u * tileN) : ALIGN_ELEMS;
        tileK = (tileK > maxTileK) ? maxTileK : tileK;
        if (tileK < ALIGN_ELEMS) tileK = (K < ALIGN_ELEMS ? K : ALIGN_ELEMS);
    }
    tileK = AlignDown(tileK, ALIGN_ELEMS);
    if (tileK == 0) tileK = (K == 0) ? 0u : (K < ALIGN_ELEMS ? K : ALIGN_ELEMS);

    // Recheck UB and if still violated, reduce tileN
    while (2u * tileN + tileK > HARD_CAP_FLOATS && tileN > ALIGN_ELEMS) {
        tileN = AlignDown(tileN / 2u, ALIGN_ELEMS);
        if (tileN == 0) break;
    }
    if (tileN == 0 && N > 0) tileN = (N < ALIGN_ELEMS ? N : ALIGN_ELEMS);

    const uint32_t kTiles = (tileK == 0) ? 1u : CeilDiv(K, tileK);
    const uint32_t totalTasks = (M == 0 || K == 0) ? 1u : (M * kTiles);

    // Choose blockDim conservatively
    uint32_t blockDim = totalTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;

class KernelGemmAddReluCustom {
public:
    __aicore__ inline KernelGemmAddReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN, uint32_t tileK)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileN_ = (tileN == 0 ? 1u : tileN);
        tileK_ = (tileK == 0 ? 1u : tileK);

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * (uint64_t)N_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)K_ * (uint64_t)N_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, K_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * (uint64_t)K_);

        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileN_ * sizeof(float)); // for mul/reduce if needed
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        const uint32_t numKtiles = (K_ + tileK_ - 1u) / tileK_;
        const uint32_t totalTasks = M_ * numKtiles;

        for (uint32_t task = blockIdx; task < totalTasks; task += blockNum) {
            const uint32_t m = task / numKtiles;
            const uint32_t kt = task % numKtiles;
            const uint32_t kBase = kt * tileK_;
            if (m >= M_ || kBase >= K_) continue;

            uint32_t curK = K_ - kBase;
            if (curK > tileK_) curK = tileK_;
            ComputeRowTile(m, kBase, curK);
        }
    }

private:
    // Helper: read one element from LocalTensor (fallback if API supports it)
    __aicore__ inline float LTGet(const LocalTensor<float>& t, uint32_t idx) {
        // Fallback implementation via DataCopy when scalar access is needed.
        // Allocate a 1-length local, copy and read through a temp GM if necessary.
        // For simplicity here, we assume direct accessors exist.
        return t.GetValue(idx);
    }
    __aicore__ inline void LTSet(LocalTensor<float>& t, uint32_t idx, float v) {
        t.SetValue(idx, v);
    }

    __aicore__ inline float DotSeg(const LocalTensor<float>& a, const LocalTensor<float>& b, uint32_t len)
    {
        // Compute sum_i a[i] * b[i]
        // Try vector mul then scalar accumulate with a simple loop via accessors
        float acc = 0.0f;
        for (uint32_t i = 0; i < len; ++i) {
            acc += a.GetValue(i) * b.GetValue(i);
        }
        return acc;
    }

    __aicore__ inline void ComputeRowTile(uint32_t m, uint32_t kBase, uint32_t curK)
    {
        // acc = bias[kBase : kBase+curK]
        LocalTensor<float> acc = accQueue_.AllocTensor<float>();
        DataCopy(acc, bGm_[kBase], curK);

        // Iterate across N in tiles and accumulate
        for (uint32_t nBase = 0; nBase < N_; nBase += tileN_) {
            uint32_t curN = N_ - nBase;
            if (curN > tileN_) curN = tileN_;

            LocalTensor<float> xSeg = xQueue_.AllocTensor<float>();
            DataCopy(xSeg, xGm_[(uint64_t)m * (uint64_t)N_ + nBase], curN);

            for (uint32_t kOff = 0; kOff < curK; ++kOff) {
                LocalTensor<float> wSeg = wQueue_.AllocTensor<float>();
                // w row index = kBase + kOff
                const uint64_t wRowStart = (uint64_t)(kBase + kOff) * (uint64_t)N_ + nBase;
                DataCopy(wSeg, wGm_[wRowStart], curN);

                float partial = DotSeg(xSeg, wSeg, curN);
                float old = acc.GetValue(kOff);
                acc.SetValue(kOff, old + partial);

                wQueue_.FreeTensor(wSeg);
            }

            xQueue_.FreeTensor(xSeg);
        }

        // Apply ReLU: max(acc, 0)
        for (uint32_t kOff = 0; kOff < curK; ++kOff) {
            float v = acc.GetValue(kOff);
            acc.SetValue(kOff, v > 0.0f ? v : 0.0f);
        }

        // Store y row tile
        DataCopy(yGm_[(uint64_t)m * (uint64_t)K_ + kBase], acc, curK);
        accQueue_.FreeTensor(acc);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, BUFFER_NUM> xQueue_;
    TQue<TPosition::VECCALC, BUFFER_NUM> wQueue_;
    TQue<TPosition::VECOUT,  BUFFER_NUM> accQueue_;
    TQue<TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileN_{0};
    uint32_t tileK_{0};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddReluCustom op;
    op.Init(x, w, bias, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileN, tiling_data.tileK);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor gemm_add_relu_impl_npu(const at::Tensor& x,
                                  const at::Tensor& w,
                                  const at::Tensor& bias) {
    // x: [M,N], w: [K,N], bias: [K] -> y: [M,K]
    TORCH_CHECK(x.dtype() == at::kFloat, "x must be float");
    TORCH_CHECK(w.dtype() == at::kFloat, "w must be float");
    TORCH_CHECK(bias.dtype() == at::kFloat, "bias must be float");
    TORCH_CHECK(x.dim() == 2, "x must be 2D");
    TORCH_CHECK(w.dim() == 2, "w must be 2D");
    TORCH_CHECK(bias.dim() == 1, "bias must be 1D");
    TORCH_CHECK(x.size(1) == w.size(1), "x.N must equal w.N");
    TORCH_CHECK(w.size(0) == bias.size(0), "w.K must equal bias.K");

    at::Tensor y = at::empty({x.size(0), w.size(0)}, x.options());
    EXEC_NPU_CMD(aclnnGemmAddReluCustom, x, w, bias, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("gemm_add_relu_custom", &gemm_add_relu_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("gemm_add_relu_custom", &gemm_add_relu_impl_npu, "Gemm + Bias + ReLU (GemmAddReluCustom on NPU)");
