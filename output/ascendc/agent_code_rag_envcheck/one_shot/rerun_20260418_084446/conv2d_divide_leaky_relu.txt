namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dDivideLeakyReluCustomTilingData tiling;

    // x: [N, C_in, H, W]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const uint32_t N  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t kH   = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    const uint32_t kW   = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(3));

    // VALID, stride=1, dilation=1
    const uint32_t oH = (H >= kH) ? (H - kH + 1u) : 0u;
    const uint32_t oW = (W >= kW) ? (W - kW + 1u) : 0u;

    const uint32_t totalOut = N * Cout * oH * oW;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *divisorPtr = attrs->GetAttrPointer<float>(0);          // divisor
    const float *negSlopePtr = attrs->GetAttrPointer<float>(1);         // negative_slope

    float divisor = divisorPtr ? *divisorPtr : 2.0f;
    float negSlope = negSlopePtr ? *negSlopePtr : 0.01f;

    // Determine bias presence
    bool hasBias = false;
    if (context->GetInputNum() >= 3) {
        const gert::Shape *b_shape = context->GetInputShape(2);
        if (b_shape != nullptr && b_shape->GetOriginShape().GetShapeSize() > 0) {
            hasBias = true;
        }
    }

    // Tiling policy: coarse, sequential over output elements.
    // UB isn't aggressively used here; pick a modest tile size for scheduling.
    const uint32_t ALIGN_ELEMS = 256;
    uint32_t tileOut = 2048;
    tileOut = AlignDown(tileOut, ALIGN_ELEMS);
    if (tileOut == 0) tileOut = ALIGN_ELEMS;
    if (tileOut > totalOut && totalOut > 0) tileOut = AlignDown(totalOut, ALIGN_ELEMS) ? AlignDown(totalOut, ALIGN_ELEMS) : totalOut;

    const uint32_t tiles = (tileOut == 0) ? 1u : CeilDiv(totalOut, tileOut);
    uint32_t blockDim = tiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cIn(Cin);
    tiling.set_cOut(Cout);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_kH(kH);
    tiling.set_kW(kW);
    tiling.set_oH(oH);
    tiling.set_oW(oW);
    tiling.set_totalOut(totalOut);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1u : 0u);
    tiling.set_divisor(divisor);
    tiling.set_negativeSlope(negSlope);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

#include "kernel_operator.h"

class KernelConv2dDivideLeakyReluCustom {
public:
    __aicore__ inline KernelConv2dDivideLeakyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t h, uint32_t w,
                                uint32_t kH, uint32_t kW,
                                uint32_t oH, uint32_t oW,
                                uint32_t totalOut, uint32_t tileOut,
                                uint32_t hasBias, float divisor, float negativeSlope)
    {
        x_ = reinterpret_cast<__gm__ float*>(x);
        w_ = reinterpret_cast<__gm__ float*>(weight);
        b_ = reinterpret_cast<__gm__ float*>(bias);
        y_ = reinterpret_cast<__gm__ float*>(y);

        N_ = n; Cin_ = cIn; Cout_ = cOut;
        H_ = h; W_ = w; kH_ = kH; kW_ = kW;
        OH_ = oH; OW_ = oW;
        totalOut_ = totalOut; tileOut_ = tileOut;
        hasBias_ = (hasBias != 0);
        negSlope_ = negativeSlope;
        invDiv_ = (divisor != 0.0f) ? (1.0f / divisor) : 0.5f;
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint64_t strideElems = static_cast<uint64_t>(blockNum) * static_cast<uint64_t>(tileOut_);
        uint64_t base = static_cast<uint64_t>(blockIdx) * static_cast<uint64_t>(tileOut_);
        if (base >= totalOut_) return;

        for (; base < totalOut_; base += strideElems) {
            uint64_t curLen = totalOut_ - base;
            if (curLen > tileOut_) curLen = tileOut_;
            ComputeRange(base, static_cast<uint32_t>(curLen));
        }
    }

private:
    __aicore__ inline void ComputeRange(uint64_t base, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            uint64_t idx = base + static_cast<uint64_t>(i);

            // Unravel idx -> (n, oc, oh, ow)
            uint64_t perN = static_cast<uint64_t>(Cout_) * OH_ * OW_;
            uint32_t n = static_cast<uint32_t>(idx / perN);
            uint64_t rem1 = idx - perN * n;

            uint32_t oc = static_cast<uint32_t>(rem1 / (OH_ * OW_));
            uint64_t rem2 = rem1 - static_cast<uint64_t>(oc) * OH_ * OW_;

            uint32_t oh = static_cast<uint32_t>(rem2 / OW_);
            uint32_t ow = static_cast<uint32_t>(rem2 - static_cast<uint64_t>(oh) * OW_);

            float acc = 0.0f;

            // Convolution accumulate: VALID, stride=1, dilation=1
            for (uint32_t ic = 0; ic < Cin_; ++ic) {
                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    const uint32_t ih = oh + kh;
                    const uint64_t xRowBase = IndexX(n, ic, ih, ow);
                    const uint64_t wRowBase = IndexW(oc, ic, kh, 0);
                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        float xv = x_[xRowBase + kw];
                        float wv = w_[wRowBase + kw];
                        acc += xv * wv;
                    }
                }
            }

            if (hasBias_) {
                acc += b_[oc];
            }

            // Divide by divisor then LeakyReLU
            acc *= invDiv_;
            float yv = (acc >= 0.0f) ? acc : (acc * negSlope_);
            y_[idx] = yv;
        }
    }

    __aicore__ inline uint64_t IndexX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        // NCHW
        return ((((uint64_t)n * Cin_ + c) * H_ + h) * W_ + w);
    }

    __aicore__ inline uint64_t IndexW(uint32_t oc, uint32_t ic, uint32_t kh, uint32_t kw) const
    {
        // [Cout, Cin, kH, kW]
        return ((((uint64_t)oc * Cin_ + ic) * kH_ + kh) * kW_ + kw);
    }

private:
    __gm__ float *x_ {nullptr};
    __gm__ float *w_ {nullptr};
    __gm__ float *b_ {nullptr};
    __gm__ float *y_ {nullptr};

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, kH_{0}, kW_{0};
    uint32_t OH_{0}, OW_{0};
    uint32_t totalOut_{0}, tileOut_{0};
    bool hasBias_{false};
    float negSlope_{0.01f};
    float invDiv_{0.5f};
};

extern "C" __global__ __aicore__ void conv2d_divide_leaky_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dDivideLeakyReluCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.n, tiling_data.cIn, tiling_data.cOut,
            tiling_data.h, tiling_data.w,
            tiling_data.kH, tiling_data.kW,
            tiling_data.oH, tiling_data.oW,
            tiling_data.totalOut, tiling_data.tileOut,
            tiling_data.hasBias, tiling_data.divisor, tiling_data.negativeSlope);
    op.Process();
}
