kernel_src="""
#include "kernel_operator.h"

class KernelConv2dMinAddMultiplyCustom {
public:
    __aicore__ inline KernelConv2dMinAddMultiplyCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR conv_bias, GM_ADDR post_bias, GM_ADDR y,
                                uint32_t n, uint32_t ci, uint32_t hi, uint32_t wi,
                                uint32_t co, uint32_t kh, uint32_t kw,
                                uint32_t ho, uint32_t wo,
                                uint32_t stride_h, uint32_t stride_w,
                                uint32_t pad_h, uint32_t pad_w,
                                uint32_t dilation_h, uint32_t dilation_w,
                                uint32_t groups, uint32_t hasBias,
                                float constantValue, float scalingFactor)
    {
        xGm_ = (__gm__ float*)x;
        wGm_ = (__gm__ float*)w;
        bGm_ = (__gm__ float*)conv_bias;
        pbGm_ = (__gm__ float*)post_bias;
        yGm_ = (__gm__ float*)y;

        n_ = n; ci_ = ci; hi_ = hi; wi_ = wi;
        co_ = co; kh_ = kh; kw_ = kw;
        ho_ = ho; wo_ = wo;
        sH_ = stride_h; sW_ = stride_w;
        pH_ = pad_h; pW_ = pad_w;
        dH_ = dilation_h; dW_ = dilation_w;
        groups_ = groups;
        hasBias_ = hasBias != 0;
        cap_ = constantValue;
        scale_ = scalingFactor;
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || co_ == 0 || ho_ == 0 || wo_ == 0) {
            return;
        }
        // Only groups=1, strides=1, pads=0, dilation=1 are considered
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalSites = n_ * ho_ * wo_;

        for (uint32_t site = blockIdx; site < totalSites; site += blockNum) {
            // decode (n, ho, wo)
            uint32_t tmp = site;
            const uint32_t n = tmp / (ho_ * wo_);
            tmp = tmp % (ho_ * wo_);
            const uint32_t oh = tmp / wo_;
            const uint32_t ow = tmp % wo_;

            for (uint32_t oc = 0; oc < co_; ++oc) {
                float acc = hasBias_ ? bGm_[oc] : 0.0f;

                // Convolution sum
                for (uint32_t ic = 0; ic < ci_; ++ic) {
                    for (uint32_t ky = 0; ky < kh_; ++ky) {
                        for (uint32_t kx = 0; kx < kw_; ++kx) {
                            const int32_t in_h = static_cast<int32_t>(oh) * static_cast<int32_t>(sH_) - static_cast<int32_t>(pH_) + static_cast<int32_t>(ky) * static_cast<int32_t>(dH_);
                            const int32_t in_w = static_cast<int32_t>(ow) * static_cast<int32_t>(sW_) - static_cast<int32_t>(pW_) + static_cast<int32_t>(kx) * static_cast<int32_t>(dW_);
                            if (in_h < 0 || in_w < 0 || in_h >= static_cast<int32_t>(hi_) || in_w >= static_cast<int32_t>(wi_)) {
                                continue;
                            }
                            const uint32_t xOffset =
                                (((n * ci_ + ic) * hi_ + static_cast<uint32_t>(in_h)) * wi_ + static_cast<uint32_t>(in_w));
                            const uint32_t wOffset =
                                (((oc * ci_ + ic) * kh_ + ky) * kw_ + kx);
                            acc += xGm_[xOffset] * wGm_[wOffset];
                        }
                    }
                }

                // post ops: min with cap, add post_bias (broadcast [Co,1,1]), scale
                if (acc > cap_) acc = cap_;
                acc += pbGm_[oc];
                acc *= scale_;

                const uint32_t yOffset =
                    (((n * co_ + oc) * ho_ + oh) * wo_ + ow);
                yGm_[yOffset] = acc;
            }
        }
    }

private:
    __gm__ float* xGm_{nullptr};
    __gm__ float* wGm_{nullptr};
    __gm__ float* bGm_{nullptr};
    __gm__ float* pbGm_{nullptr};
    __gm__ float* yGm_{nullptr};

    uint32_t n_{0}, ci_{0}, hi_{0}, wi_{0};
    uint32_t co_{0}, kh_{0}, kw_{0};
    uint32_t ho_{0}, wo_{0};
    uint32_t sH_{1}, sW_{1};
    uint32_t pH_{0}, pW_{0};
    uint32_t dH_{1}, dW_{1};
    uint32_t groups_{1};
    bool hasBias_{true};
    float cap_{0.0f};
    float scale_{1.0f};
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias,
                                                                      GM_ADDR post_bias, GM_ADDR y,
                                                                      GM_ADDR workspace, GM_ADDR tiling)
{
    if (g_coreType == AscendC::AIV) {
        return;
    }
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinAddMultiplyCustom op;
    op.Init(x, weight, conv_bias, post_bias, y,
            tiling_data.n, tiling_data.ci, tiling_data.hi, tiling_data.wi,
            tiling_data.co, tiling_data.kh, tiling_data.kw,
            tiling_data.ho, tiling_data.wo,
            tiling_data.stride_h, tiling_data.stride_w,
            tiling_data.pad_h, tiling_data.pad_w,
            tiling_data.dilation_h, tiling_data.dilation_w,
            tiling_data.groups, tiling_data.hasBias,
            tiling_data.constantValue, tiling_data.scalingFactor);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor empty_output_like_conv(const at::Tensor& x,
                                                const at::Tensor& w,
                                                int64_t stride_h, int64_t stride_w,
                                                int64_t pad_h, int64_t pad_w,
                                                int64_t dilation_h, int64_t dilation_w)
{
    // NCHW, OIHW
    auto N = x.size(0);
    auto Ci = x.size(1);
    auto Hi = x.size(2);
    auto Wi = x.size(3);

    auto Co = w.size(0);
    auto Kh = w.size(2);
    auto Kw = w.size(3);

    int64_t Ho = (Hi + 2 * pad_h - dilation_h * (Kh - 1) - 1) / stride_h + 1;
    int64_t Wo = (Wi + 2 * pad_w - dilation_w * (Kw - 1) - 1) / stride_w + 1;

    return at::empty({N, Co, Ho, Wo}, x.options());
}

at::Tensor conv2d_min_add_multiply_impl_npu(const at::Tensor& x,
                                            const at::Tensor& weight,
                                            c10::optional<at::Tensor> conv_bias_opt,
                                            const at::Tensor& post_bias,
                                            const at::Tensor& constant_value,
                                            const at::Tensor& scaling_factor)
{
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, "x and weight must be 4D");
    TORCH_CHECK(post_bias.dim() == 3 || post_bias.dim() == 1, "post_bias should be [Co,1,1] or [Co]");
    // Extract scalars
    double cv = constant_value.reshape(-1).item<double>();
    double sf = scaling_factor.reshape(-1).item<double>();

    // Prepare bias
    at::Tensor conv_bias;
    bool has_bias = false;
    if (conv_bias_opt.has_value() && conv_bias_opt.value().defined() && conv_bias_opt.value().numel() > 0) {
        conv_bias = conv_bias_opt.value();
        has_bias = true;
    } else {
        conv_bias = at::empty({0}, x.options());
        has_bias = false;
    }

    // Stride/pad/dilation are fixed to (1,0,1) in this demo
    int64_t stride_h = 1, stride_w = 1, pad_h = 0, pad_w = 0, dilation_h = 1, dilation_w = 1;
    at::Tensor y = empty_output_like_conv(x, weight, stride_h, stride_w, pad_h, pad_w, dilation_h, dilation_w);

    // Launch custom op: attributes are provided explicitly
    EXEC_NPU_CMD(aclnnConv2dMinAddMultiplyCustom,
                 x, weight, conv_bias, post_bias,
                 (int)stride_h, (int)stride_w, (int)pad_h, (int)pad_w,
                 (int)dilation_h, (int)dilation_w, (int)1, // groups=1
                 has_bias, (float)cv, (float)sf,
                 y);

    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_min_add_multiply_custom", &conv2d_min_add_multiply_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
