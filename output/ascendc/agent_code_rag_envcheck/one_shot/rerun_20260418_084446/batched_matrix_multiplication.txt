host_operator_src="""

#include "batched_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    BatchedMatrixMultiplicationCustomTilingData tiling;

    // Try to derive a conservative blockDim from batch size if available.
    uint32_t blockDim = 8;  // safe default
    const gert::Shape *a_shape = context->GetInputShape(0);
    if (a_shape != nullptr) {
        const auto &origin = a_shape->GetOriginShape();
        if (origin.GetDimNum() >= 1) {
            uint64_t batch = static_cast<uint64_t>(origin.GetDim(0));
            if (batch == 0) batch = 1;
            blockDim = static_cast<uint32_t>(batch);
            if (blockDim < 8u) blockDim = 8u;
            if (blockDim > 24u) blockDim = 24u;
        }
    }

    context->SetBlockDim(blockDim);

    // Leave cubeTilingData to be resolved by runtime matmul tiler; store blockDim for kernel reference if needed.
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace required
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return GRAPH_FAILED;
    }
    // A: [B, M, K], B: [B, K, N] -> C: [B, M, N]
    gert::Shape *c_shape = context->GetOutputShape(0);
    std::vector<int64_t> c_dims;
    const auto &ad = a_shape->GetOriginShape();
    const auto &bd = b_shape->GetOriginShape();
    if (ad.GetDimNum() != 3 || bd.GetDimNum() != 3) {
        return GRAPH_FAILED;
    }
    c_dims.push_back(ad.GetDim(0));
    c_dims.push_back(ad.GetDim(1));
    c_dims.push_back(bd.GetDim(2));
    c_shape->SetDimNum(c_dims.size());
    for (size_t i = 0; i < c_dims.size(); ++i) {
        c_shape->SetDim(i, c_dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // A, B: float16; C: float32 (accumulation in fp32)
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class BatchedMatrixMultiplicationCustom : public OpDef {
public:
    explicit BatchedMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(BatchedMatrixMultiplicationCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    // Retrieve tiling data produced by host tiling function or framework tiler.
    GET_TILING_DATA(tilingData, tiling);

    // Define MatMul operand and output types:
    // A: FP16, B: FP16, C: FP32 accumulate
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half,  false, LayoutMode::BSNGD> A_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half,  true,  LayoutMode::BSNGD> B_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD> C_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> BIAS_TYPE;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> batchMatmulKernel;
    AscendC::TPipe pipe;

    // Share full L1/L0C for better perf; no UB usage in this wrapper
    tilingData.cubeTilingData.shareMode   = 0;
    tilingData.cubeTilingData.shareL1Size = FULL_L1_SIZE;
    tilingData.cubeTilingData.shareL0CSize = FULL_L0C_SIZE;
    tilingData.cubeTilingData.shareUbSize  = 0;

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), batchMatmulKernel.matmulObj, &tilingData.cubeTilingData);

    // Batch number from tiling (fallback to 1 if absent)
    int batchNum = tilingData.cubeTilingData.BatchNum > 0 ? tilingData.cubeTilingData.BatchNum : 1;

    // Initialize and execute
    batchMatmulKernel.Init(a, b, nullptr, c, workspace, tilingData.cubeTilingData);
    batchMatmulKernel.Process<false>(&pipe, batchNum, batchNum);
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor bmm_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 3 && B.dim() == 3, "A and B must be 3D tensors: [B, M, K] and [B, K, N]");
    auto B0 = A.size(0);
    TORCH_CHECK(B.size(0) == B0, "Batch dimensions of A and B must match");
    auto M = A.size(1);
    auto K = A.size(2);
    TORCH_CHECK(B.size(1) == K, "Inner dimensions must match for batched matmul");
    auto N = B.size(2);

    // Output is fp32 accumulation; keep API simple and let lower stack handle casts if necessary
    at::Tensor C = at::empty({B0, M, N}, A.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnBatchedMatrixMultiplicationCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("batched_matrix_multiplication_custom", &bmm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("batched_matrix_multiplication_custom", &bmm_impl_npu,
          "Batched MatMul (BatchedMatrixMultiplicationCustom on NPU)");
}
