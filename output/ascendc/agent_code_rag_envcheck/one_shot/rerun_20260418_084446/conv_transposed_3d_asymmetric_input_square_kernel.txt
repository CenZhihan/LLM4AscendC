// UB model for this kernel:
//   xLine: 1 * tileW
//   yLine: 1 * tileW
// Total floats = 2 * tileW => bytes = 8 * tileW
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // align DMA size
static constexpr uint32_t HARD_CAP_ELEMS = 16384;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

    if (xShape->GetOriginShape().GetDimNum() != 5 || wShape->GetOriginShape().GetDimNum() != 5) {
        return ge::GRAPH_FAILED;

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvTransposed3dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t OD, uint32_t OH, uint32_t OW,
                                uint32_t tileW)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        D_ = D; H_ = H; W_ = W;
        kD_ = kD; kH_ = kH; kW_ = kW;
        OD_ = OD; OH_ = OH; OW_ = OW;
        tileW_ = tileW;

        uint64_t xSize = static_cast<uint64_t>(N_) * Cin_ * D_ * H_ * W_;
        uint64_t wSize = static_cast<uint64_t>(Cin_) * Cout_ * kD_ * kH_ * kW_;
        uint64_t ySize = static_cast<uint64_t>(N_) * Cout_ * OD_ * OH_ * OW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileW_ * sizeof(float)); // scratch
        pipe_.InitBuffer(wScalarQueue_, 1, sizeof(float));      // 1 scalar
    }

    __aicore__ inline void Process()
    {
        if (OW_ == 0 || OH_ == 0 || OD_ == 0) return;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalLines = static_cast<uint64_t>(N_) * Cout_ * OD_ * OH_;
        if (blockIdx >= totalLines) return;

        for (uint64_t lineIdx = blockIdx; lineIdx < totalLines; lineIdx += blockNum) {
            uint32_t n, oc, oz, oy;
            DecodeLineIndex(lineIdx, n, oc, oz, oy);

            for (uint32_t base_x = 0; base_x < OW_; base_x += tileW_) {
                uint32_t segLenY = OW_ - base_x;
                if (segLenY > tileW_) segLenY = tileW_;

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Duplicate(yLocal, 0.0f, segLenY);

                // Accumulate contributions
                for (uint32_t ic = 0; ic < Cin_; ++ic) {
                    for (uint32_t kd = 0; kd < kD_; ++kd) {
                        int32_t iz = static_cast<int32_t>(oz) - static_cast<int32_t>(kd);
                        if (iz < 0 || iz >= static_cast<int32_t>(D_)) continue;
                        for (uint32_t kh = 0; kh < kH_; ++kh) {
                            int32_t iy = static_cast<int32_t>(oy) - static_cast<int32_t>(kh);
                            if (iy < 0 || iy >= static_cast<int32_t>(H_)) continue;
                            for (uint32_t kw = 0; kw < kW_; ++kw) {
                                int32_t ix_start = static_cast<int32_t>(base_x) - static_cast<int32_t>(kw);

                                int32_t t0 = 0;
                                if (ix_start < 0) t0 = -ix_start;
                                int32_t t1 = static_cast<int32_t>(segLenY) - 1;
                                int32_t max_t = static_cast<int32_t>(W_) - 1 - ix_start;
                                if (t1 > max_t) t1 = max_t;

                                if (t0 <= t1) {
                                    uint32_t segLen = static_cast<uint32_t>(t1 - t0 + 1);

                                    uint64_t xOff = XOffset(n, ic,
                                                            static_cast<uint32_t>(iz),
                                                            static_cast<uint32_t>(iy),
                                                            static_cast<uint32_t>(ix_start + t0));
                                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                                    AscendC::DataCopy(xLocal, xGm_[xOff], segLen);

                                    // Load weight scalar and compute scaled add
                                    uint64_t wOff = WOffset(ic, oc, kd, kh, kw);
                                    AscendC::LocalTensor<float> wLocal = wScalarQueue_.AllocTensor<float>();
                                    AscendC::DataCopy(wLocal, wGm_[wOff], 1);

                                    // Broadcast weight into tmp buffer and multiply-add
                                    AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                                    AscendC::Duplicate(tmpLocal, wLocal.GetValue(0), segLen); // using local scalar duplication
                                    AscendC::Mul(xLocal, xLocal, tmpLocal, segLen);
                                    AscendC::Add(yLocal[t0], yLocal[t0], xLocal, segLen);

                                    tmpQueue_.FreeTensor(tmpLocal);
                                    wScalarQueue_.FreeTensor(wLocal);
                                    inQueueX_.FreeTensor(xLocal);
                                }
                            }
                        }
                    }
                }

                uint64_t yOff = YOffset(n, oc, oz, oy, base_x);
                AscendC::DataCopy(yGm_[yOff], yLocal, segLenY);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline void DecodeLineIndex(uint64_t idx, uint32_t &n, uint32_t &oc, uint32_t &oz, uint32_t &oy) const
    {
        uint64_t t = idx;
        oy = static_cast<uint32_t>(t % OH_); t /= OH_;
        oz = static_cast<uint32_t>(t % OD_); t /= OD_;
        oc = static_cast<uint32_t>(t % Cout_); t /= Cout_;
        n  = static_cast<uint32_t>(t);
    }

    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t ic, uint32_t iz, uint32_t iy, uint32_t ix) const
    {
        return (((((uint64_t)n * Cin_ + ic) * D_ + iz) * H_ + iy) * W_ + ix);
    }

    __aicore__ inline uint64_t WOffset(uint32_t ic, uint32_t oc, uint32_t kd, uint32_t kh, uint32_t kw) const
    {
        return (((((uint64_t)ic * Cout_ + oc) * kD_ + kd) * kH_ + kh) * kW_ + kw);
    }

    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t oc, uint32_t oz, uint32_t oy, uint32_t ox) const
    {
        return (((((uint64_t)n * Cout_ + oc) * OD_ + oz) * OH_ + oy) * OW_ + ox);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wScalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t D_{0}, H_{0}, W_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};
    uint32_t OD_{0}, OH_{0}, OW_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dAsymmetricInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.OD, tiling_data.OH, tiling_data.OW,
            tiling_data.tileW);
    op.Process();
}
