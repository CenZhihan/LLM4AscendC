kernel_src="""
#include "kernel_operator.h"

class KernelConv2dMinTanhTanhCustom {
public:
    __aicore__ inline KernelConv2dMinTanhTanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t hin, uint32_t win,
                                uint32_t cout, uint32_t kh, uint32_t kw,
                                uint32_t hout, uint32_t wout, uint32_t totalPos,
                                uint32_t hasBias)
    {
        n_ = n; cin_ = cin; hin_ = hin; win_ = win;
        cout_ = cout; kh_ = kh; kw_ = kw;
        hout_ = hout; wout_ = wout; totalPos_ = totalPos;
        hasBias_ = hasBias;

        const uint32_t xSize = n_ * cin_ * hin_ * win_;
        const uint32_t wSize = cout_ * cin_ * kh_ * kw_;
        const uint32_t ySize = n_ * 1u * hout_ * wout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        if (b != 0) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // Only need a few 1-element temporaries; use VECCALC queue.
        pipe_.InitBuffer(calcQ_, 4, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalPos_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t pos = blockIdx; pos < totalPos_; pos += blockNum) {
            // Decode position -> (n, oh, ow)
            const uint32_t posPerN = hout_ * wout_;
            const uint32_t n = pos / posPerN;
            const uint32_t rem = pos % posPerN;
            const uint32_t oh = rem / wout_;
            const uint32_t ow = rem % wout_;

            AscendC::LocalTensor<float> x1 = calcQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> w1 = calcQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> acc = calcQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> minv = calcQ_.AllocTensor<float>();

            // Initialize min accumulator to +inf, acc to 0
            AscendC::Duplicate(minv, 3.402823466e+38f, 1);

            for (uint32_t oc = 0; oc < cout_; ++oc) {
                AscendC::Duplicate(acc, 0.0f, 1);

                // Dot product over K = cin * kh * kw
                for (uint32_t c = 0; c < cin_; ++c) {
                    for (uint32_t r = 0; r < kh_; ++r) {
                        for (uint32_t s = 0; s < kw_; ++s) {
                            const uint32_t ih = oh + r;
                            const uint32_t iw = ow + s;

                            const uint32_t xOff = ((n * cin_ + c) * hin_ + ih) * win_ + iw;
                            const uint32_t wOff = (((oc * cin_ + c) * kh_ + r) * kw_) + s;

                            AscendC::DataCopy(x1, xGm_[xOff], 1);
                            AscendC::DataCopy(w1, wGm_[wOff], 1);
                            AscendC::Mul(x1, x1, w1, 1);
                            AscendC::Add(acc, acc, x1, 1);
                        }
                    }
                }

                if (hasBias_ != 0) {
                    AscendC::DataCopy(w1, bGm_[oc], 1);
                    AscendC::Add(acc, acc, w1, 1);
                }

                // min across output channels
                AscendC::Min(minv, minv, acc, 1);
            }

            // tanh twice
            AscendC::Tanh(minv, minv, 1);
            AscendC::Tanh(minv, minv, 1);

            const uint32_t yOff = ((n * 1u + 0u) * hout_ + oh) * wout_ + ow;
            AscendC::DataCopy(yGm_[yOff], minv, 1);

            // Free local tensors
            calcQ_.FreeTensor(minv);
            calcQ_.FreeTensor(acc);
            calcQ_.FreeTensor(w1);
            calcQ_.FreeTensor(x1);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> calcQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, hin_{0}, win_{0};
    uint32_t cout_{0}, kh_{0}, kw_{0};
    uint32_t hout_{0}, wout_{0}, totalPos_{0};
    uint32_t hasBias_{0};
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias_opt, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanhCustom op;
    op.Init(x, weight, conv_bias_opt, y,
            tiling_data.n, tiling_data.cin, tiling_data.hin, tiling_data.win,
            tiling_data.cout, tiling_data.kh, tiling_data.kw,
            tiling_data.hout, tiling_data.wout, tiling_data.totalPos,
            tiling_data.hasBias);
    op.Process();
}
