    {

        coutPerG_ = coutPerG; groups_ = groups;
        kd_ = kd; kh_ = kh; kw_ = kw;
        sd_ = sd; sh_ = sh; sw_ = sw;
        pd_ = pd; ph_ = ph; pw_ = pw;
        opd_ = opd; oph_ = oph; opw_ = opw;

        dout_ = dout; hout_ = hout; wout_ = wout;

        uint64_t wSize = (uint64_t)cin_ * coutPerG_ * kd_ * kh_ * kw_;
        uint64_t ySize = (uint64_t)n_ * (uint32_t)(coutPerG_ * groups_) * dout_ * hout_ * wout_;

        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(tmpWQ_, 1, sizeof(float));

        const uint32_t totalBlocks = n_ * cout;

        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();
        if (blockIdx >= totalBlocks) return;

        const uint32_t nIdx = blockIdx / cout;
        const uint32_t oc   = blockIdx % cout;
        const uint32_t g    = oc / coutPerG_;
        const uint32_t ocg  = oc % coutPerG_;
        const uint32_t cinPerG = cin_ / groups_;

        const uint64_t x_stride_w = 1ULL;
        const uint64_t x_stride_h = (uint64_t)win_;
        const uint64_t x_stride_d = (uint64_t)hin_ * win_;
        const uint64_t x_stride_c = (uint64_t)din_ * hin_ * win_;
        const uint64_t x_stride_n = (uint64_t)cin_ * x_stride_c;

        const uint64_t y_stride_h = (uint64_t)wout_;
        const uint64_t y_stride_d = (uint64_t)hout_ * wout_;
        const uint64_t y_stride_c = (uint64_t)dout_ * hout_ * wout_;
        const uint64_t y_stride_n = (uint64_t)(cout) * y_stride_c;

        const uint64_t w_k_total = (uint64_t)kd_ * kh_ * kw_;
        const uint64_t w_stride_kw = 1ULL;

                            if ((tH % (int64_t)sh_) != 0) continue;

                            if (ih < 0 || ih >= (int64_t)hin_) continue;

                                if ((tW % (int64_t)sw_) != 0) continue;

                                const uint64_t baseX_spatial =
                                    (uint64_t)id * x_stride_d +
                                    (uint64_t)ih * x_stride_h +
                                    (uint64_t)iw * x_stride_w;
                                const uint64_t baseW_k =
                                    (uint64_t)kd * w_stride_kd +
                                    (uint64_t)kh * w_stride_kh +
                                    (uint64_t)kw * w_stride_kw;

                                for (uint32_t icg = 0; icg < cinPerG; ++icg) {
                                    const uint32_t ic_abs = g * cinPerG + icg;

                                    // X offset
                                    const uint64_t x_off =
                                        (uint64_t)nIdx * x_stride_n +
                                        (uint64_t)ic_abs * x_stride_c +
                                        baseX_spatial;

                                    // W offset
                                    const uint64_t w_off =
                                        (uint64_t)ic_abs * w_stride_cin +
                                        (uint64_t)ocg    * w_stride_ocg +
                                        baseW_k;

                                    AscendC::DataCopy(wLocal1, wGm_[w_off], 1);

                                    AscendC::Add(sumLocal, sumLocal, xLocal1, 1);
                                } // icg
                            } // kw
                        } // kh
                    } // kd

                        (uint64_t)oc   * y_stride_c +
                        (uint64_t)od   * y_stride_d +
                        (uint64_t)oh   * y_stride_h +
                        (uint64_t)ow   * y_stride_w;

                } // ow
            } // oh
        } // od

        tmpWQ_.FreeTensor(wLocal1);
        tmpXQ_.FreeTensor(xLocal1);
        tmpSumQ_.FreeTensor(sumLocal);
    }
