host_operator_src="""

#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
// UB model:
//   inQueue: 2 * tile
//   outQueue: 2 * tile
//   tmpQueue: 1 * tile
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t PER_TILE_BYTES = 20;           // 20 bytes per element as per UB model
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor large DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid extreme per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const auto &ori = shape->GetOriginShape();

    const int32_t rank = static_cast<int32_t>(ori.GetDimNum());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *axisAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t axis = (axisAttr != nullptr) ? *axisAttr : 1;

    // Normalize axis
    if (axis < 0) {
        axis += rank;
    }
    if (axis < 0) axis = 0;
    if (axis >= rank) axis = rank - 1;

    uint64_t totalLength64 = ori.GetShapeSize();
    uint32_t totalLength = static_cast<uint32_t>(totalLength64);

    // Compute reduceLen along axis and outerRows as totalLength / reduceLen
    uint32_t reduceLen = static_cast<uint32_t>(ori.GetDim(axis));
    uint32_t outerRows = (reduceLen == 0) ? 0 : static_cast<uint32_t>(totalLength / reduceLen);

    // Determine tileLen from UB budget: UB_BUDGET_BYTES = PER_TILE_BYTES * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / PER_TILE_BYTES;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (reduceLen < tileLen) {
        const uint32_t aligned = AlignDown(reduceLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : reduceLen;
    }
    if (tileLen == 0) tileLen = reduceLen;

    // Block dim based on number of rows
    uint32_t blockDim = (outerRows == 0) ? 1u : outerRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outerRows(outerRows);
    tiling.set_reduceLen(reduceLen);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_axis(axis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class LogSoftmaxCustom : public OpDef {
public:
    explicit LogSoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("axis").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(LogSoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outerRows, uint32_t reduceLen,
                                uint32_t tileLen)
    {
        outerRows_ = outerRows;
        reduceLen_ = reduceLen;
        tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(outerRows_) * reduceLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(outerRows_) * reduceLen_);

        // Queues: in, out, tmp
        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outerRows_ == 0 || reduceLen_ == 0 || tileLen_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        // Grid-stride rows
        for (uint32_t row = blockIdx; row < outerRows_; row += blockNum) {
            float rowMax = -INFINITY;
            // Pass 1: compute row-wise max
            for (uint32_t base = 0; base < reduceLen_; base += tileLen_) {
                const uint32_t curLen = (base + tileLen_ <= reduceLen_) ? tileLen_ : (reduceLen_ - base);
                CopyIn(row, base, curLen);
                LocalTensor<float> xLocal = inQueue_.DeQue<float>();
                // tileMax = reduce_max(xLocal)
                float tileMax = ReduceMaxLocal(xLocal, curLen);
                if (tileMax > rowMax) rowMax = tileMax;
                inQueue_.FreeTensor(xLocal);
            }

            // Pass 2: compute sumExp = sum(exp(x - rowMax))
            float sumExp = 0.0f;
            for (uint32_t base = 0; base < reduceLen_; base += tileLen_) {
                const uint32_t curLen = (base + tileLen_ <= reduceLen_) ? tileLen_ : (reduceLen_ - base);
                CopyIn(row, base, curLen);
                LocalTensor<float> xLocal = inQueue_.DeQue<float>();
                LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                // tLocal = xLocal - rowMax
                Adds(tLocal, xLocal, -rowMax, curLen);
                // tLocal = exp(tLocal)
                Exp(tLocal, tLocal, curLen);
                // sumExp += reduce_sum(tLocal)
                float tileSum = ReduceSumLocal(tLocal, curLen);
                sumExp += tileSum;
                tmpQueue_.FreeTensor(tLocal);
                inQueue_.FreeTensor(xLocal);
            }
            const float logSumExp = logf(sumExp) + rowMax;

            // Pass 3: write y = x - logSumExp
            for (uint32_t base = 0; base < reduceLen_; base += tileLen_) {
                const uint32_t curLen = (base + tileLen_ <= reduceLen_) ? tileLen_ : (reduceLen_ - base);
                CopyIn(row, base, curLen);
                LocalTensor<float> xLocal = inQueue_.DeQue<float>();
                LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
                Adds(yLocal, xLocal, -logSumExp, curLen);
                outQueue_.EnQue(yLocal);
                inQueue_.FreeTensor(xLocal);
                CopyOut(row, base, curLen);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t row, uint32_t colBase, uint32_t len)
    {
        LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
        const uint64_t gmOffset = static_cast<uint64_t>(row) * reduceLen_ + colBase;
        DataCopy(xLocal, xGm_[gmOffset], len);
        inQueue_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t row, uint32_t colBase, uint32_t len)
    {
        LocalTensor<float> yLocal = outQueue_.DeQue<float>();
        const uint64_t gmOffset = static_cast<uint64_t>(row) * reduceLen_ + colBase;
        DataCopy(yGm_[gmOffset], yLocal, len);
        outQueue_.FreeTensor(yLocal);
    }

    // Helper: reduce max over a local tensor
    __aicore__ inline float ReduceMaxLocal(const LocalTensor<float>& xLocal, uint32_t len)
    {
        // Use vector reduce intrinsic if available; fallback to chunked reduction
        // Here we assume ReduceMax exists; otherwise emulate by iterative reduction using vector Max
        // Stepwise reduction: reduce in chunks into a temp scalar held in register
        float maxVal = -INFINITY;
        // Process in blocks of 256 (or any chunk size) using vector reduce if available
        // Assuming API: float r = ReduceMax(xLocal, len);
        float r = ReduceMax(xLocal, len);
        if (r > maxVal) maxVal = r;
        return maxVal;
    }

    // Helper: reduce sum over a local tensor
    __aicore__ inline float ReduceSumLocal(const LocalTensor<float>& xLocal, uint32_t len)
    {
        // Assuming API: float s = ReduceSum(xLocal, len);
        float s = ReduceSum(xLocal, len);
        return s;
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueue_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueue_;
    TQue<TPosition::VECCALC, 1> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t outerRows_{0};
    uint32_t reduceLen_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y,
            tiling_data.outerRows,
            tiling_data.reduceLen,
            tiling_data.tileLen);
    op.Process();
}
"""
