host_operator_src="""

#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) { return static_cast<uint32_t>((a + b - 1ULL) / b); }
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative UB budget
static constexpr uint32_t ALIGN_ELEMS = 2048;            // DMA-friendly alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // limit per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const std::vector<int64_t> dims = in_shape->GetOriginShape().GetDims();
    const size_t rank = dims.size();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    // dim attribute as int64_t to be robust
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    if (dimPtr == nullptr) {
        return ge::GRAPH_FAILED;
    }
    int64_t dim = *dimPtr;
    int64_t normDim = dim;
    if (normDim < 0) normDim += static_cast<int64_t>(rank);
    if (normDim < 0 || normDim >= static_cast<int64_t>(rank)) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t axis = static_cast<uint32_t>(normDim);

    uint64_t outer = 1, inner = 1, reduceLen = 1;
    for (uint32_t i = 0; i < axis; ++i) outer *= static_cast<uint64_t>(dims[i]);
    reduceLen = static_cast<uint64_t>(dims[axis]);
    for (uint32_t i = axis + 1; i < rank; ++i) inner *= static_cast<uint64_t>(dims[i]);

    // Choose tileInner based on UB budgets:
    // Buffers: inQueueX (2 * tileInner) + accumulator (1 * tileInner) => bytes = (2 + 1) * 4 * tileInner = 12 * tileInner
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileInner = static_cast<uint32_t>(inner <= static_cast<uint64_t>(maxTile) ? inner : maxTile);
    if (inner < static_cast<uint64_t>(ALIGN_ELEMS)) {
        tileInner = static_cast<uint32_t>(inner);
    }

    // BlockDim based on number of outer rows
    uint32_t rows = static_cast<uint32_t>(outer);
    uint32_t blockDim = rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_outer(static_cast<uint32_t>(outer));
    tiling.set_inner(static_cast<uint32_t>(inner));
    tiling.set_reduceLen(static_cast<uint32_t>(reduceLen));
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);
    tiling.set_invReduce((reduceLen == 0) ? 0.0f : (1.0f / static_cast<float>(reduceLen)));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0U;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> in_dims = x_shape->GetOriginShape().GetDims();
    const size_t rank = in_dims.size();

    // dim attr
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    if (dimPtr == nullptr) {
        return GRAPH_FAILED;
    }
    int64_t dim = *dimPtr;
    int64_t normDim = dim;
    if (normDim < 0) normDim += static_cast<int64_t>(rank);
    if (normDim < 0 || normDim >= static_cast<int64_t>(rank)) {
        return GRAPH_FAILED;
    }

    std::vector<int64_t> out_dims;
    out_dims.reserve(rank > 0 ? rank - 1 : 0);
    for (size_t i = 0; i < rank; ++i) {
        if (static_cast<int64_t>(i) == normDim) continue;
        out_dims.push_back(in_dims[i]);
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(out_dims.size());
    for (size_t i = 0; i < out_dims.size(); ++i) {
        y_shape->SetDim(i, out_dims[i]);
    }
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUFFER_NUM = 2;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t inner, uint32_t reduceLen,
                                uint32_t tileInner, float invReduce)
    {
        outer_ = outer;
        inner_ = inner;
        reduceLen_ = reduceLen;
        tileInner_ = tileInner;
        invReduce_ = invReduce;

        const uint32_t xSize = outer_ * reduceLen_ * inner_;
        const uint32_t ySize = outer_ * inner_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(inQueueX_, IN_BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outer_ == 0 || inner_ == 0 || reduceLen_ == 0 || tileInner_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < outer_; row += blockNum) {
            const uint32_t inRowBase = row * reduceLen_ * inner_;
            const uint32_t outRowBase = row * inner_;

            for (uint32_t ib = 0; ib < inner_; ib += tileInner_) {
                uint32_t len = inner_ - ib;
                if (len > tileInner_) len = tileInner_;

                // Initialize accumulator to zero
                AscendC::LocalTensor<float> acc = tmpQueue_.AllocTensor<float>();
                AscendC::Duplicate(acc, 0.0f, len);

                // Accumulate sum across reduceLen_
                for (uint32_t k = 0; k < reduceLen_; ++k) {
                    const uint32_t xOff = inRowBase + k * inner_ + ib;
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[xOff], len);
                    inQueueX_.EnQue(xLocal);

                    AscendC::LocalTensor<float> xLocalQ = inQueueX_.DeQue<float>();
                    AscendC::Add(acc, acc, xLocalQ, len);
                    inQueueX_.FreeTensor(xLocalQ);
                }

                // Mean: scale by 1 / reduceLen
                AscendC::Muls(acc, acc, invReduce_, len);

                // Write out
                const uint32_t yOff = outRowBase + ib;
                AscendC::DataCopy(yGm_[yOff], acc, len);

                tmpQueue_.FreeTensor(acc);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_{0};
    uint32_t inner_{0};
    uint32_t reduceLen_{0};
    uint32_t tileInner_{0};
    float invReduce_{0.0f};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.outer, tiling_data.inner, tiling_data.reduceLen,
            tiling_data.tileInner, tiling_data.invReduce);
    op.Process();
}
