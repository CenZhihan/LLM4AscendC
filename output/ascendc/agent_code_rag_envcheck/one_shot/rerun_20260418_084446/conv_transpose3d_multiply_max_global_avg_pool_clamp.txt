kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom {
public:
    __aicore__ inline KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t hasBias)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        D_ = D; H_ = H; W_ = W;
        kD_ = kD; kH_ = kH; kW_ = kW;
        hasBias_ = hasBias;

        Dout_ = D_ + kD_ - 1;
        Hout_ = H_ + kH_ - 1;
        Wout_ = W_ + kW_ - 1;
        volOut_ = static_cast<uint64_t>(Dout_) * static_cast<uint64_t>(Hout_) * static_cast<uint64_t>(Wout_);

        xLen_ = static_cast<uint64_t>(N_) * Cin_ * D_ * H_ * W_;
        wLen_ = static_cast<uint64_t>(Cin_) * Cout_ * kD_ * kH_ * kW_;
        bLen_ = (hasBias_ != 0) ? Cout_ : 0;
        yLen_ = static_cast<uint64_t>(N_) * Cout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wLen_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, bLen_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, yLen_);

        // Minimal UB buffers (scalar tensors)
        pipe_.InitBuffer(xBuf_, 1, sizeof(float));
        pipe_.InitBuffer(wBuf_, 1, sizeof(float));
        pipe_.InitBuffer(bBuf_, 1, sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, sizeof(float));
        pipe_.InitBuffer(accBuf_, 1, sizeof(float));
        pipe_.InitBuffer(sumBuf_, 1, sizeof(float));
        pipe_.InitBuffer(maxBuf_, 1, sizeof(float));
        pipe_.InitBuffer(prodBuf_, 1, sizeof(float));
        pipe_.InitBuffer(constBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        // Only block 0 works, as tiling set blockDim = 1
        if (GetBlockIdx() != 0) { return; }

        LocalTensor<float> xLocal = xBuf_.AllocTensor<float>();
        LocalTensor<float> wLocal = wBuf_.AllocTensor<float>();
        LocalTensor<float> bLocal = bBuf_.AllocTensor<float>();
        LocalTensor<float> tmpLocal = tmpBuf_.AllocTensor<float>();
        LocalTensor<float> accLocal = accBuf_.AllocTensor<float>();
        LocalTensor<float> sumLocal = sumBuf_.AllocTensor<float>();
        LocalTensor<float> maxLocal = maxBuf_.AllocTensor<float>();
        LocalTensor<float> prodLocal = prodBuf_.AllocTensor<float>();
        LocalTensor<float> constLocal = constBuf_.AllocTensor<float>();

        // Constants
        const float negInf = -3.402823466e+38f;

        for (uint32_t n = 0; n < N_; ++n) {
            for (uint32_t co = 0; co < Cout_; ++co) {
                // sumLocal = 0; maxLocal = -inf
                Dup(sumLocal, 0.0f, 1);
                Dup(maxLocal, negInf, 1);

                // Iterate over output spatial positions (full conv size)
                for (uint32_t oz = 0; oz < Dout_; ++oz) {
                    for (uint32_t oy = 0; oy < Hout_; ++oy) {
                        for (uint32_t ox = 0; ox < Wout_; ++ox) {
                            // accLocal = 0
                            Dup(accLocal, 0.0f, 1);

                            // Accumulate contributions from (ci, kz, ky, kx)
                            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                                for (uint32_t kz = 0; kz < kD_; ++kz) {
                                    int32_t tz = static_cast<int32_t>(oz) - static_cast<int32_t>(kz);
                                    if (tz < 0 || tz >= static_cast<int32_t>(D_)) continue;
                                    for (uint32_t ky = 0; ky < kH_; ++ky) {
                                        int32_t ty = static_cast<int32_t>(oy) - static_cast<int32_t>(ky);
                                        if (ty < 0 || ty >= static_cast<int32_t>(H_)) continue;
                                        for (uint32_t kx = 0; kx < kW_; ++kx) {
                                            int32_t tx = static_cast<int32_t>(ox) - static_cast<int32_t>(kx);
                                            if (tx < 0 || tx >= static_cast<int32_t>(W_)) continue;

                                            // x offset: (((n*Cin + ci)*D + tz)*H + ty)*W + tx
                                            uint64_t xOff = (((static_cast<uint64_t>(n) * Cin_ + ci) * D_ + static_cast<uint32_t>(tz))
                                                             * H_ + static_cast<uint32_t>(ty)) * W_ + static_cast<uint32_t>(tx);

                                            // w offset: ((((ci*Cout + co)*kD + kz)*kH + ky)*kW + kx)
                                            uint64_t wOff = ((((static_cast<uint64_t>(ci) * Cout_ + co) * kD_ + kz)
                                                              * kH_ + ky) * kW_ + kx);

                                            // Load scalars and MAC
                                            DataCopy(xLocal, xGm_[xOff], 1);
                                            DataCopy(wLocal, wGm_[wOff], 1);
                                            Mul(tmpLocal, xLocal, wLocal, 1);
                                            Add(accLocal, accLocal, tmpLocal, 1);
                                        }
                                    }
                                }
                            }

                            // Add bias if present: b[co]
                            if (hasBias_) {
                                DataCopy(bLocal, bGm_[co], 1);
                                Add(accLocal, accLocal, bLocal, 1);
                            }

                            // sumLocal += accLocal
                            Add(sumLocal, sumLocal, accLocal, 1);
                            // maxLocal = max(maxLocal, accLocal)
                            Max(maxLocal, maxLocal, accLocal, 1);
                        }
                    }
                }

                // mean = sum / volOut_
                float invVolHost = 1.0f / static_cast<float>(volOut_);
                Dup(constLocal, invVolHost, 1);
                Mul(prodLocal, sumLocal, constLocal, 1);      // prodLocal holds mean

                // prodLocal = mean * max
                Mul(prodLocal, prodLocal, maxLocal, 1);

                // clamp to [0, 1]
                Dup(constLocal, 0.0f, 1);
                Max(prodLocal, prodLocal, constLocal, 1);
                Dup(constLocal, 1.0f, 1);
                Min(prodLocal, prodLocal, constLocal, 1);

                // Store to y: index = n*Cout + co
                uint64_t yOff = static_cast<uint64_t>(n) * Cout_ + co;
                DataCopy(yGm_[yOff], prodLocal, 1);
            }
        }

        // Free local tensors
        constBuf_.FreeTensor(constLocal);
        prodBuf_.FreeTensor(prodLocal);
        maxBuf_.FreeTensor(maxLocal);
        sumBuf_.FreeTensor(sumLocal);
        accBuf_.FreeTensor(accLocal);
        tmpBuf_.FreeTensor(tmpLocal);
        bBuf_.FreeTensor(bLocal);
        wBuf_.FreeTensor(wLocal);
        xBuf_.FreeTensor(xLocal);
    }

private:
    // Queues and pipeline
    TPipe pipe_;
    TQue<TPosition::VECCALC, 1> xBuf_;
    TQue<TPosition::VECCALC, 1> wBuf_;
    TQue<TPosition::VECCALC, 1> bBuf_;
    TQue<TPosition::VECCALC, 1> tmpBuf_;
    TQue<TPosition::VECCALC, 1> accBuf_;
    TQue<TPosition::VECCALC, 1> sumBuf_;
    TQue<TPosition::VECCALC, 1> maxBuf_;
    TQue<TPosition::VECCALC, 1> prodBuf_;
    TQue<TPosition::VECCALC, 1> constBuf_;

    // Global tensors
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    // Dimensions
    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t D_{0}, H_{0}, W_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};
    uint32_t hasBias_{0};
    uint64_t volOut_{0};

    // Lengths
    uint64_t xLen_{0}, wLen_{0}, bLen_{0}, yLen_{0};
};

extern "C" __global__ __aicore__ void conv_transpose3d_multiply_max_global_avg_pool_clamp_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.hasBias);
    op.Process();
}
