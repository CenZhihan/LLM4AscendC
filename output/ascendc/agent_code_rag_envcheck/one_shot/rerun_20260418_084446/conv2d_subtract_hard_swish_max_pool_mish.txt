kernel_src="""
#include <cmath>
#include "kernel_operator.h"

class KernelConv2dSubtractHardSwishMaxPoolMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractHardSwishMaxPoolMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t Oh, uint32_t Ow,
                                uint32_t Pk, uint32_t Oh2, uint32_t Ow2,
                                uint32_t hasBias, float subtractVal)
    {
        xGm_ = reinterpret_cast<__gm__ float*>(x);
        wGm_ = reinterpret_cast<__gm__ float*>(w);
        bGm_ = reinterpret_cast<__gm__ float*>(b);
        yGm_ = reinterpret_cast<__gm__ float*>(y);

        N_ = N; Cin_ = Cin; H_ = H; W_ = W;
        Cout_ = Cout; Kh_ = Kh; Kw_ = Kw;
        Oh_ = Oh; Ow_ = Ow;
        Pk_ = Pk; Oh2_ = Oh2; Ow2_ = Ow2;
        hasBias_ = (hasBias != 0);
        subtractVal_ = subtractVal;
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cout_ == 0 || Oh2_ == 0 || Ow2_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalPairs = static_cast<uint64_t>(N_) * static_cast<uint64_t>(Cout_);
        for (uint64_t pair = blockIdx; pair < totalPairs; pair += blockNum) {
            const uint32_t n = static_cast<uint32_t>(pair / Cout_);
            const uint32_t oc = static_cast<uint32_t>(pair % Cout_);

            // For each pooled output position
            for (uint32_t ph = 0; ph < Oh2_; ++ph) {
                for (uint32_t pw = 0; pw < Ow2_; ++pw) {
                    float maxVal = -1.0e30f;

                    const uint32_t ohBase = ph * Pk_;
                    const uint32_t owBase = pw * Pk_;
                    const uint32_t rhMax = AscendC::Min(Pk_, Oh_ - ohBase);
                    const uint32_t rwMax = AscendC::Min(Pk_, Ow_ - owBase);

                    for (uint32_t rh = 0; rh < rhMax; ++rh) {
                        for (uint32_t rw = 0; rw < rwMax; ++rw) {
                            const uint32_t oh = ohBase + rh;
                            const uint32_t ow = owBase + rw;

                            // Convolution at (n, oc, oh, ow)
                            float acc = hasBias_ ? bGm_[oc] : 0.0f;

                            // Reduce over Cin, Kh, Kw
                            for (uint32_t c = 0; c < Cin_; ++c) {
                                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                                    const uint32_t in_h = oh + kh;
                                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                        const uint32_t in_w = ow + kw;

                                        const uint64_t xIdx =
                                            (((static_cast<uint64_t>(n) * Cin_ + c) * H_ + in_h) * W_ + in_w);

                                        const uint64_t wIdx =
                                            ((((static_cast<uint64_t>(oc) * Cin_ + c) * Kh_ + kh) * Kw_) + kw);

                                        float xv = xGm_[xIdx];
                                        float wv = wGm_[wIdx];
                                        acc += xv * wv;
                                    }
                                }
                            }

                            // Subtract constant
                            float v = acc - subtractVal_;

                            // HardSwish: v * relu6(v + 3) / 6
                            float t = v + 3.0f;
                            if (t < 0.0f) t = 0.0f;
                            if (t > 6.0f) t = 6.0f;
                            v = v * t * (1.0f / 6.0f);

                            // Max pooling
                            if (v > maxVal) maxVal = v;
                        }
                    }

                    // Mish: x * tanh(softplus(x)), softplus(x)=log(1+exp(x))
                    float sp = logf(1.0f + expf(maxVal));
                    float th = tanhf(sp);
                    float outv = maxVal * th;

                    const uint64_t yIdx =
                        (((static_cast<uint64_t>(n) * Cout_ + oc) * Oh2_ + ph) * Ow2_ + pw);
                    yGm_[yIdx] = outv;
                }
            }
        }
    }

private:
    __gm__ float *xGm_{nullptr};
    __gm__ float *wGm_{nullptr};
    __gm__ float *bGm_{nullptr};
    __gm__ float *yGm_{nullptr};

    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, Kh_{0}, Kw_{0};
    uint32_t Oh_{0}, Ow_{0};
    uint32_t Pk_{0}, Oh2_{0}, Ow2_{0};
    bool hasBias_{false};
    float subtractVal_{0.0f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractHardSwishMaxPoolMishCustom op;
    op.Init(
        x, weight, bias, y,
        tiling_data.N, tiling_data.Cin, tiling_data.H, tiling_data.W,
        tiling_data.Cout, tiling_data.Kh, tiling_data.Kw,
        tiling_data.Oh, tiling_data.Ow,
        tiling_data.Pk, tiling_data.Oh2, tiling_data.Ow2,
        tiling_data.hasBias, tiling_data.subtractVal
    );
    op.Process();
}

#include <torch/library.h>

static inline at::Tensor allocate_output_like_conv_pool(const at::Tensor& x,
                                                        const at::Tensor& w,
                                                        int64_t pool_kernel_size)
{
    const int64_t N = x.size(0);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);

    const int64_t Cout = w.size(0);
    const int64_t Kh = w.size(2);
    const int64_t Kw = w.size(3);

    const int64_t Oh = H - Kh + 1;
    const int64_t Ow = W - Kw + 1;
    const int64_t Oh2 = (Oh > 0 && pool_kernel_size > 0) ? (Oh / pool_kernel_size) : 0;
    const int64_t Ow2 = (Ow > 0 && pool_kernel_size > 0) ? (Ow / pool_kernel_size) : 0;

    auto sizes = at::IntArrayRef({N, Cout, Oh2, Ow2});
    return at::empty(sizes, x.options());
}

    const at::Tensor& x,
    const at::Tensor& weight,

    double subtract_value,
