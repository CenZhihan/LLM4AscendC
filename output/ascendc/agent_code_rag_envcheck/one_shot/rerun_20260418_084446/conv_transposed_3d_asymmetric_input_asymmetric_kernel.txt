kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 1;
constexpr int32_t OUT_BUF_NUM = 1;
constexpr int32_t TMP_BUF_NUM = 2;

class KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t d, uint32_t h, uint32_t wd,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t od, uint32_t oh, uint32_t ow)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        d_ = d; h_ = h; w_ = wd;
        kd_ = kd; kh_ = kh; kw_ = kw;
        od_ = od; oh_ = oh; ow_ = ow;

        rows_ = static_cast<uint64_t>(n_) * cout_ * od_ * oh_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_)*cin_*d_*h_*w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cin_)*cout_*kd_*kh_*kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_)*cout_*od_*oh_*ow_);

        // UB buffers:
        // inQueue: up to W elements at a time for an input width row segment
        // outQueue: OW elements for output row accumulation
        // tmpQueue: auxiliary buffers (broadcasted weights, intermediate)
        pipe_.InitBuffer(inQueue_, IN_BUF_NUM, w_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUT_BUF_NUM, ow_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMP_BUF_NUM, ow_ * sizeof(float)); // large enough for broadcasts
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t row = blockIdx; row < rows_; row += blockNum) {
            // Decode row -> (n, co, oz, oy)
            uint64_t t = row;
            const uint32_t oy = static_cast<uint32_t>(t % oh_); t /= oh_;
            const uint32_t oz = static_cast<uint32_t>(t % od_); t /= od_;
            const uint32_t co = static_cast<uint32_t>(t % cout_); t /= cout_;
            const uint32_t n  = static_cast<uint32_t>(t % n_);

            // UB buffer holding output row y[n,co,oz,oy,0:ow_]
            AscendC::LocalTensor<float> yRow = outQueue_.AllocTensor<float>();
            AscendC::Duplicate(yRow, 0.0f, ow_);

            // Accumulate over cin, kd, kh, kw
            for (uint32_t ci = 0; ci < cin_; ++ci) {
                for (uint32_t dz = 0; dz < kd_; ++dz) {
                    // iz = oz - dz must be in [0, d_-1]
                    int32_t iz = static_cast<int32_t>(oz) - static_cast<int32_t>(dz);
                    if (iz < 0 || iz >= static_cast<int32_t>(d_)) continue;

                    for (uint32_t dy = 0; dy < kh_; ++dy) {
                        int32_t iy = static_cast<int32_t>(oy) - static_cast<int32_t>(dy);
                        if (iy < 0 || iy >= static_cast<int32_t>(h_)) continue;

                        for (uint32_t dx = 0; dx < kw_; ++dx) {
                            // valid ox satisfy 0 <= ix = ox - dx < w_
                            uint32_t st = dx;                     // first ox that has ix=0
                            uint32_t en = dx + w_;               // last ox exclusive that has ix=w_-1
                            if (st >= ow_) continue;
                            if (en > ow_) en = ow_;
                            uint32_t len = en > st ? (en - st) : 0;
                            if (len == 0) continue;

                            // Load input row segment x[n,ci,iz,iy, 0:len]
                            const uint64_t baseX = ((((static_cast<uint64_t>(n) * cin_ + ci) * d_ + static_cast<uint32_t>(iz)) * h_ + static_cast<uint32_t>(iy)) * w_);
                            AscendC::LocalTensor<float> xSeg = inQueue_.AllocTensor<float>();
                            AscendC::DataCopy(xSeg, xGm_[baseX], len);

                            // Load weight scalar w[ci, co, kd-1-dz, kh-1-dy, kw-1-dx]
                            const uint32_t kz = (kd_ - 1u) - dz;
                            const uint32_t ky = (kh_ - 1u) - dy;
                            const uint32_t kx = (kw_ - 1u) - dx;
                            const uint64_t wOff = ((((static_cast<uint64_t>(ci) * cout_ + co) * kd_ + kz) * kh_ + ky) * kw_ + kx);

                            // Broadcast weight scalar into tmp buffer size len
                            AscendC::LocalTensor<float> wVec = tmpQueue_.AllocTensor<float>();
                            // Read the scalar to UB first (1 element)
                            AscendC::LocalTensor<float> wScalarUb = tmpQueue_.AllocTensor<float>();
                            AscendC::DataCopy(wScalarUb, wGm_[wOff], 1);
                            // Duplicate scalar into vector length len
                            // Note: Some backends allow Duplicate from the first element in UB;
                            // here we simply use Duplicate with a runtime scalar fetched via a temporary GM->UB copy.
                            // Assuming Duplicate supports reading from a UB scalar by value
                            // We perform: wVec = broadcast(wScalarUb[0]) of length len
                            // For portability, we copy wScalarUb to GM-like immediate then Duplicate; simplified as below:
                            AscendC::Duplicate(wVec, wScalarUb.GetValue(0), len);

                            // Scale xSeg by weight: xSeg = xSeg * w
                            AscendC::Mul(xSeg, xSeg, wVec, len);

                            // yRow[st:st+len] += xSeg
                            AscendC::LocalTensor<float> ySub = yRow[st];
                            AscendC::Add(ySub, ySub, xSeg, len);

                            // Free temps
                            tmpQueue_.FreeTensor(wScalarUb);
                            tmpQueue_.FreeTensor(wVec);
                            inQueue_.FreeTensor(xSeg);
                        } // dx
                    } // dy
                } // dz
            } // ci

            // Store output row to GM
            const uint64_t baseY = ((((static_cast<uint64_t>(n) * cout_ + co) * od_ + oz) * oh_ + oy) * ow_);
            AscendC::DataCopy(yGm_[baseY], yRow, ow_);
            outQueue_.FreeTensor(yRow);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t d_{0}, h_{0}, w_{0};
    uint32_t kd_{0}, kh_{0}, kw_{0};
    uint32_t od_{0}, oh_{0}, ow_{0};
    uint64_t rows_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling_data, tilingGm);
    KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.od, tiling_data.oh, tiling_data.ow);
    op.Process();
}
