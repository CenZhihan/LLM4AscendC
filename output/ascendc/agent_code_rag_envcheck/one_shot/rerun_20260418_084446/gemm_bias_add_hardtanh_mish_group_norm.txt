#include "kernel_operator.h"
#include <cmath>

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR lin_b, GM_ADDR add_b,
                                GM_ADDR gn_gamma, GM_ADDR gn_beta, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t numGroups, float eps, uint32_t kChunk)
    {
        m_ = m; n_ = n; k_ = k;
        groups_ = numGroups;
        eps_ = eps;
        kChunk_ = (kChunk == 0) ? 256u : kChunk;

        xGm_ = reinterpret_cast<__gm__ float*>(x);
        wGm_ = reinterpret_cast<__gm__ float*>(w);
        linBiasGm_ = reinterpret_cast<__gm__ float*>(lin_b);
        addBiasGm_ = reinterpret_cast<__gm__ float*>(add_b);
        gammaGm_ = reinterpret_cast<__gm__ float*>(gn_gamma);
        betaGm_ = reinterpret_cast<__gm__ float*>(gn_beta);
        yGm_ = reinterpret_cast<__gm__ float*>(y);
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || n_ == 0 || k_ == 0) return;
        if (groups_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // One row per block with grid-stride
        for (uint32_t row = blockIdx; row < m_; row += blockNum) {
            ComputeRow(row);
        }
    }

private:
    __aicore__ inline float hardtanh_scalar(float v)
    {
        if (v < -1.0f) return -1.0f;
        if (v > 1.0f) return 1.0f;
        return v;
    }

    __aicore__ inline float mish_scalar(float v)
    {
        // mish(x) = x * tanh(softplus(x)), softplus = log1p(exp(x))
        float sp = log1pf(expf(v));
        float t = tanhf(sp);
        return v * t;
    }

    __aicore__ inline void ComputeRow(uint32_t row)
    {
        // Group parameters
        const uint32_t groupSize = n_ / groups_;
        if (groupSize == 0) return;

        // For each group within the row
        for (uint32_t g = 0; g < groups_; ++g) {
            const uint32_t colStart = g * groupSize;
            const uint32_t colEnd = colStart + groupSize;

            // First pass: GEMM for each output channel in group, then bias, hardtanh, mish; accumulate stats and stash pre-norm y in output
            float sum = 0.0f;
            float sumsq = 0.0f;

            for (uint32_t nIdx = colStart; nIdx < colEnd; ++nIdx) {
                // Compute dot(x_row, w_row_nIdx)
                float acc = 0.0f;
                // Iterate over k dimension
                for (uint32_t kk = 0; kk < k_; ++kk) {
                    float xv = xGm_[row * k_ + kk];
                    float wv = wGm_[nIdx * k_ + kk]; // weight is [N, K]
                    acc += xv * wv;
                }

                // Add biases
                acc += linBiasGm_[nIdx];
                acc += addBiasGm_[nIdx];

                // Activations
                acc = hardtanh_scalar(acc);
                acc = mish_scalar(acc);

                // Stash for group-norm second pass
                yGm_[row * n_ + nIdx] = acc;

                sum += acc;
                sumsq += acc * acc;
            }

            // Compute group mean and variance
            float mean = sum / static_cast<float>(groupSize);
            float var = sumsq / static_cast<float>(groupSize) - mean * mean;
            if (var < 0.0f) var = 0.0f;
            float invStd = 1.0f / sqrtf(var + eps_);

            // Second pass: normalize and apply affine (gamma/beta)
            for (uint32_t nIdx = colStart; nIdx < colEnd; ++nIdx) {
                float v = yGm_[row * n_ + nIdx];
                float norm = (v - mean) * invStd;
                float out = norm * gammaGm_[nIdx] + betaGm_[nIdx];
                yGm_[row * n_ + nIdx] = out;
            }
        }
    }

private:
    __gm__ float* xGm_ {nullptr};
    __gm__ float* wGm_ {nullptr};
    __gm__ float* linBiasGm_ {nullptr};
    __gm__ float* addBiasGm_ {nullptr};
    __gm__ float* gammaGm_ {nullptr};
    __gm__ float* betaGm_ {nullptr};
    __gm__ float* yGm_ {nullptr};

    uint32_t m_ {0};
    uint32_t n_ {0};
    uint32_t k_ {0};
    uint32_t groups_ {1};
    float eps_ {1e-5f};
    uint32_t kChunk_ {256};
};

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR lin_bias, GM_ADDR bias, GM_ADDR gn_gamma, GM_ADDR gn_beta,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    op.Init(x, weight, lin_bias, bias, gn_gamma, gn_beta, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.numGroups, tiling_data.eps, tiling_data.kChunk);
    op.Process();
}
