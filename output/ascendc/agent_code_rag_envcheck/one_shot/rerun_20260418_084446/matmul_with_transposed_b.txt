#include "matmul_with_transposed_b_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint64_t a, uint32_t b)
{
    return static_cast<uint32_t>((a + b - 1ull) / static_cast<uint64_t>(b));
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Shapes:
    //   A: [M, K]
    //   B: [N, K]  (we multiply A * B^T -> logical B^T is [K, N])
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);
    if (aShape->GetDimNum() != 2 || bShape->GetDimNum() != 2) {
        return ge::GRAPH_PARAM_INVALID;
    }
    const uint64_t M = static_cast<uint64_t>(aShape->GetDim(0));
    const uint64_t K_a = static_cast<uint64_t>(aShape->GetDim(1));
    const uint64_t N = static_cast<uint64_t>(bShape->GetDim(0));
    const uint64_t K_b = static_cast<uint64_t>(bShape->GetDim(1));
    if (K_a != K_b) {
        return ge::GRAPH_PARAM_INVALID;
    }

    // Conservative blockDim choice based on tiles over rows
    // Use one output-row tile per block up to device friendly limits
    uint32_t blockDim = static_cast<uint32_t>(M);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    MatmulWithTransposedBCustomTilingData tiling;
    tiling.set_M(static_cast<uint32_t>(M));
    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_K(static_cast<uint32_t>(K_a));
    tiling.set_blockDim(blockDim);
    tiling.set_isTransA(0u);
    tiling.set_isTransB(1u);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    // A: [M, K], B: [N, K], output C: [M, N]
    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);
    if (aShape->GetDimNum() != 2 || bShape->GetDimNum() != 2) {
        return ge::GRAPH_PARAM_INVALID;
    }
    const int64_t M = aShape->GetDim(0);
    const int64_t K_a = aShape->GetDim(1);
    const int64_t N = bShape->GetDim(0);
    const int64_t K_b = bShape->GetDim(1);
    if (K_a != K_b) {
        return ge::GRAPH_PARAM_INVALID;
    }
    gert::Shape *cShape = context->GetOutputShape(0);
    cShape->SetDimNum(2);
    cShape->SetDim(0, M);
    cShape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_PARAM_INVALID;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {

kernel_src="""
#include "kernel_operator.h"

// This kernel computes C = A * (B^T) where A is [M,K], B is [N,K], C is [M,N].
// It uses the AscendC Matmul high-level API with GM ND tensors and sets isTransB=true
// to avoid explicitly transposing B on GM.

extern "C" __global__ __aicore__ void matmul_with_transposed_b_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                                                      GM_ADDR workspace, GM_ADDR tilingGm)
{
    // Prepare tiling copied from host tiling buffer (opaque to the kernel-side wrapper).
    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    // Define Matmul types: GM ND, float32. B is transposed logically.
    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    // Wrap with a minimal matmul kernel from AscendC samples (MN-DB configuration).
    // The following Matmul Kernel wrapper is available in AscendC examples and
    // drives the Matmul object end-to-end with Init/Process.
    MatmulMN::MatmulKernel<float, float, float, float> matmulKernel;

    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulKernel.matmulObj, &tiling);

    // isTransA=false; isTransB=true to interpret B as [N,K] -> logical [K,N]
    matmulKernel.Init(A, B, nullptr, C, tiling, false, true);
    matmulKernel.Process();
}
"""
