host_operator_src="""

#include "layer_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (per-core):
//   - First pass: 1 * tile for x
//   - Second pass: x + gamma + beta + y + tmp = 5 * tile
// Budget with margin: 5 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LayerNormCustomTilingData tiling;

    const uint32_t totalLength =
        static_cast<uint32_t>(context->GetInputShape(0)->GetOriginShape().GetShapeSize());
    const uint32_t normLength =
        static_cast<uint32_t>(context->GetInputShape(1)->GetOriginShape().GetShapeSize());

    if (normLength == 0) {
        return ge::GRAPH_FAILED;
    }
    if (totalLength % normLength != 0) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t outerRows = totalLength / normLength;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float eps = 1e-5f;
    if (attrs != nullptr && attrs->GetNumArgs() > 0) {
        const float *epsPtr = attrs->GetAttrPointer<float>(0);
        if (epsPtr != nullptr) eps = *epsPtr;
    }

    // tileLen from UB budget: bytes ~ 5 * tile * 4
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // in float elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (normLength < tileLen) {
        const uint32_t aligned = AlignDown(normLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : normLength;
    }

    // One block processes multiple rows in a grid-stride fashion
    uint32_t blockDim = outerRows;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_normLength(normLength);
    tiling.set_outerRows(outerRows);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *g_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);
    // Basic validation: gamma and beta must match normalized shape
    if (g_shape->GetOriginShape().GetShapeSize() != b_shape->GetOriginShape().GetShapeSize()) {
        return GRAPH_FAILED;
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelLayerNormCustom {
public:
    __aicore__ inline KernelLayerNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t totalLength, uint32_t normLength, uint32_t outerRows,
                                uint32_t tileLength, float eps)
    {
        totalLen_ = totalLength;
        normLen_ = normLength;
        outerRows_ = outerRows;
        tileLen_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, normLen_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, normLen_);

        // Queues
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueGamma_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueBeta_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || normLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < outerRows_; row += blockNum) {
            const uint32_t rowBase = row * normLen_;

            // Pass 1: accumulate sum and sumsq across the row
            float sum = 0.0f;
            float sumsq = 0.0f;

            // Prefetch first tile of pass 1
            uint32_t base = rowBase;
            uint32_t remain = normLen_;
            uint32_t cur = (remain > tileLen_) ? tileLen_ : remain;
            if (cur == 0) continue;
            CopyInX(base, cur);

            while (true) {
                uint32_t nextBase = base + cur;
                uint32_t nextLen = 0;
                if (remain > cur) {
                    nextLen = remain - cur;
                    if (nextLen > tileLen_) nextLen = tileLen_;
                    CopyInX(nextBase, nextLen);
                }

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                // partial sums
                float partSum = AscendC::ReduceSum(xLocal, cur);
                // tLocal = x^2
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                AscendC::Mul(tLocal, xLocal, xLocal, cur);
                float partSq = AscendC::ReduceSum(tLocal, cur);
                tmpQueue_.FreeTensor(tLocal);
                inQueueX_.FreeTensor(xLocal);

                sum += partSum;
                sumsq += partSq;

                if (nextLen == 0) break;
                base = nextBase;
                remain -= cur;
                cur = nextLen;
            }

            const float mean = sum / static_cast<float>(normLen_);
            const float var = sumsq / static_cast<float>(normLen_) - mean * mean;
            const float invStd = AscendC::Rsqrt(var + eps_);

            // Prepare scalar vectors
            // We'll duplicate mean and invStd into vectors per tile during pass 2.

            // Pass 2: normalize, scale and shift
            base = rowBase;
            remain = normLen_;
            cur = (remain > tileLen_) ? tileLen_ : remain;
            if (cur == 0) continue;

            // Prefetch first tiles
            CopyInX(base, cur);
            CopyInGamma(base, cur);
            CopyInBeta(base, cur);

            while (true) {
                uint32_t nextBase2 = base + cur;
                uint32_t nextLen2 = 0;
                if (remain > cur) {
                    nextLen2 = remain - cur;
                    if (nextLen2 > tileLen_) nextLen2 = tileLen_;
                    CopyInX(nextBase2, nextLen2);
                    CopyInGamma(nextBase2, nextLen2);
                    CopyInBeta(nextBase2, nextLen2);
                }

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> gLocal = inQueueGamma_.DeQue<float>();
                AscendC::LocalTensor<float> bLocal = inQueueBeta_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> t0Local = tmpQueue_.AllocTensor<float>();

                // t0 = x - mean
                AscendC::Dup(t0Local, mean, cur);
                AscendC::Sub(t0Local, xLocal, t0Local, cur);
                // t0 = t0 * invStd
                AscendC::Muls(t0Local, t0Local, invStd, cur);
                // y = t0 * gamma + beta
                AscendC::Mul(yLocal, t0Local, gLocal, cur);
                AscendC::Add(yLocal, yLocal, bLocal, cur);

                tmpQueue_.FreeTensor(t0Local);
                inQueueX_.FreeTensor(xLocal);
                inQueueGamma_.FreeTensor(gLocal);
                inQueueBeta_.FreeTensor(bLocal);
                outQueueY_.EnQue(yLocal);

                // Write back
                AscendC::LocalTensor<float> yLocal2 = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[base], yLocal2, cur);
                outQueueY_.FreeTensor(yLocal2);

                if (nextLen2 == 0) break;
                base = nextBase2;
                remain -= cur;
                cur = nextLen2;
            }
        }
    }

private:
    __aicore__ inline void CopyInX(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> buf = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(buf, xGm_[gmOffset], len);
        inQueueX_.EnQue(buf);
    }
    __aicore__ inline void CopyInGamma(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> buf = inQueueGamma_.AllocTensor<float>();
        AscendC::DataCopy(buf, gammaGm_[gmOffset], len);
        inQueueGamma_.EnQue(buf);
    }
    __aicore__ inline void CopyInBeta(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> buf = inQueueBeta_.AllocTensor<float>();
        AscendC::DataCopy(buf, betaGm_[gmOffset], len);
        inQueueBeta_.EnQue(buf);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueGamma_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueBeta_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;

    uint32_t totalLen_{0};
    uint32_t normLen_{0};
    uint32_t outerRows_{0};
    uint32_t tileLen_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void layer_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelLayerNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.totalLength,
            tiling_data.normLength,
            tiling_data.outerRows,
            tiling_data.tileLength,
            tiling_data.eps);
    op.Process();
}

import torch

import custom_ops_lib

_FN = 'layer_norm_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, normalized_shape):
        super(ModelNew, self).__init__()
        # Match LayerNorm default params
        self.weight = torch.nn.Parameter(torch.ones(normalized_shape, dtype=torch.float32))
        self.bias = torch.nn.Parameter(torch.zeros(normalized_shape, dtype=torch.float32))
        self.eps = 1e-5

        fn = getattr(custom_ops_lib, _FN)
