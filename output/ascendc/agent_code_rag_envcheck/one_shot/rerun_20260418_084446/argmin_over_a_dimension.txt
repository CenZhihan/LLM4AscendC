host_operator_src="""

#include "argmin_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0U) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1U) / b;
}

} // anonymous namespace

namespace optiling {

// UB model per segment group:
//   valQueue (double buffer): 2 * tileSeg * 4B
//   minVals:                  1 * tileSeg * 4B
//   idxVals:                  1 * tileSeg * 8B
//   mask:                     1 * tileSeg * 1B
//   idxTmp:                   1 * tileSeg * 8B
// Total bytes ~= (8 + 4 + 8 + 1 + 8) * tileSeg = 29 * tileSeg
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t BYTES_PER_SEG   = 29;
static constexpr uint32_t ALIGN_SEGS      = 1024;        // encourage large DMA and vector ops
static constexpr uint32_t MAX_TILE_SEGS   = 8192;        // avoid too-large latency per iteration
static constexpr uint32_t MAX_BLOCKS      = 24;
static constexpr uint32_t MIN_BLOCKS      = 1;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgminOverADimensionCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);

    const gert::Shape* shape = context->GetInputShape(0);
    const gert::Shape& origin = shape->GetOriginShape();
    const int64_t rank = origin.GetDimNum();

    if (rank <= 0) {
        // scalar input is unsupported for argmin
        return ge::GRAPH_FAILED;
    }

    int32_t dim = *dimPtr;
    int32_t axis = dim;
    if (axis < 0) axis += static_cast<int32_t>(rank);
    if (axis < 0 || axis >= rank) {
        return ge::GRAPH_FAILED;
    }

    uint64_t reduceLen = static_cast<uint64_t>(origin.GetDim(axis));
    uint64_t strideK = 1U;
    for (int64_t i = axis + 1; i < rank; ++i) {
        strideK *= static_cast<uint64_t>(origin.GetDim(i));
    }
    uint64_t outerNum = 1U;
    for (int64_t i = 0; i < axis; ++i) {
        outerNum *= static_cast<uint64_t>(origin.GetDim(i));
    }

    // tile segments bounded by UB budget and natural segment width strideK
    uint32_t budgetTile = UB_BUDGET_BYTES / BYTES_PER_SEG;
    if (budgetTile == 0) budgetTile = 1;
    if (budgetTile > MAX_TILE_SEGS) budgetTile = MAX_TILE_SEGS;
    budgetTile = AlignDown(budgetTile, ALIGN_SEGS);
    if (budgetTile == 0) budgetTile = ALIGN_SEGS;

    uint32_t strideK32 = static_cast<uint32_t>(strideK);
    uint32_t tileSeg = budgetTile;
    if (tileSeg > strideK32) tileSeg = (strideK32 == 0 ? 1U : strideK32);
    if (tileSeg == 0) tileSeg = 1;

    uint32_t outerNum32 = static_cast<uint32_t>(outerNum);
    uint32_t blockDim = outerNum32;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_reduceLen(static_cast<uint32_t>(reduceLen));
    tiling.set_strideK(strideK32);
    tiling.set_outerNum(outerNum32);
    tiling.set_tileSeg(tileSeg);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape &origin = x_shape->GetOriginShape();
    int64_t rank = origin.GetDimNum();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dimPtr;
    if (dim < 0) dim += static_cast<int32_t>(rank);
    if (dim < 0 || dim >= rank) return GRAPH_FAILED;

    std::vector<int64_t> out_dims;
    out_dims.reserve(rank > 0 ? static_cast<size_t>(rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == dim) continue;
        out_dims.push_back(origin.GetDim(i));
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    if (out_dims.empty()) {
        // scalar result
        y_shape->SetDimCnt(0);
    } else {
        y_shape->SetDim(out_dims.data(), static_cast<uint32_t>(out_dims.size()));
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // output indices are int64
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float   INF_F32 = 3.402823466e+38F; // FLT_MAX

class KernelArgminOverADimensionCustom {
public:
    __aicore__ inline KernelArgminOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t reduceLen,
                                uint32_t strideK,
                                uint32_t outerNum,
                                uint32_t tileSeg)
    {
        reduceLen_ = reduceLen;
        strideK_   = strideK;
        outerNum_  = outerNum;
        tileSeg_   = tileSeg;

        totalX_ = static_cast<uint64_t>(reduceLen_) * static_cast<uint64_t>(strideK_) * static_cast<uint64_t>(outerNum_);
        totalY_ = static_cast<uint64_t>(strideK_) * static_cast<uint64_t>(outerNum_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(totalX_));
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, static_cast<uint32_t>(totalY_));

        pipe_.InitBuffer(valQueue_, BUFFER_NUM, tileSeg_ * sizeof(float));
        pipe_.InitBuffer(minQueue_, 1, tileSeg_ * sizeof(float));
        pipe_.InitBuffer(maskQueue_, 1, tileSeg_ * sizeof(uint8_t));
        pipe_.InitBuffer(idxQueue_, 1, tileSeg_ * sizeof(int64_t));
        pipe_.InitBuffer(idxTmpQueue_, 1, tileSeg_ * sizeof(int64_t));
    }

    __aicore__ inline void Process()
    {
        if (reduceLen_ == 0 || strideK_ == 0 || outerNum_ == 0 || tileSeg_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes a subset of 'outerNum' (prefix indices).
        for (uint32_t p = blockIdx; p < outerNum_; p += blockNum) {
            const uint64_t baseOuterX = static_cast<uint64_t>(p) * static_cast<uint64_t>(strideK_) * static_cast<uint64_t>(reduceLen_);
            const uint64_t baseOuterY = static_cast<uint64_t>(p) * static_cast<uint64_t>(strideK_);

            for (uint32_t qBase = 0; qBase < strideK_; qBase += tileSeg_) {
                uint32_t curSeg = strideK_ - qBase;
                if (curSeg > tileSeg_) curSeg = tileSeg_;

                // Persistent buffers across reduction steps
                AscendC::LocalTensor<float> minLocal = minQueue_.AllocTensor<float>();
                AscendC::LocalTensor<int64_t> idxLocal = idxQueue_.AllocTensor<int64_t>();
                AscendC::Dup(minLocal, INF_F32, curSeg);
                AscendC::Dup(idxLocal, static_cast<int64_t>(0), curSeg);

                for (uint32_t t = 0; t < reduceLen_; ++t) {
                    AscendC::LocalTensor<float> valLocal = valQueue_.AllocTensor<float>();
                    const uint64_t gmOff = baseOuterX + static_cast<uint64_t>(qBase) + static_cast<uint64_t>(t) * static_cast<uint64_t>(strideK_);
                    AscendC::DataCopy(valLocal, xGm_[gmOff], curSeg);
                    valQueue_.EnQue(valLocal);

                    AscendC::LocalTensor<float> valRead = valQueue_.DeQue<float>();
                    AscendC::LocalTensor<uint8_t> maskLocal = maskQueue_.AllocTensor<uint8_t>();
                    AscendC::Less(maskLocal, valRead, minLocal, curSeg);
                    AscendC::Select(minLocal, minLocal, valRead, maskLocal, curSeg);

                    AscendC::LocalTensor<int64_t> idxTmp = idxTmpQueue_.AllocTensor<int64_t>();
                    AscendC::Dup(idxTmp, static_cast<int64_t>(t), curSeg);
                    AscendC::Select(idxLocal, idxLocal, idxTmp, maskLocal, curSeg);

                    idxTmpQueue_.FreeTensor(idxTmp);
                    maskQueue_.FreeTensor(maskLocal);
                    valQueue_.FreeTensor(valRead);
                }

                // Store indices
                const uint64_t yOff = baseOuterY + static_cast<uint64_t>(qBase);
                AscendC::DataCopy(yGm_[yOff], idxLocal, curSeg);

                idxQueue_.FreeTensor(idxLocal);
                minQueue_.FreeTensor(minLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN,  BUFFER_NUM> valQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>        minQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>        maskQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>        idxQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>        idxTmpQueue_;

    AscendC::GlobalTensor<float>   xGm_;
    AscendC::GlobalTensor<int64_t> yGm_;

    uint32_t reduceLen_{0};
    uint32_t strideK_{0};
    uint32_t outerNum_{0};
    uint32_t tileSeg_{0};

    uint64_t totalX_{0};
    uint64_t totalY_{0};
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgminOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.reduceLen,
            tiling_data.strideK,
            tiling_data.outerNum,
            tiling_data.tileSeg);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t NormalizeDim(int64_t dim, int64_t rank) {
    if (dim < 0) dim += rank;
    return dim;
}

at::Tensor argmin_impl_npu(const at::Tensor& self, int64_t dim) {
    TORCH_CHECK(self.is_npu(), "Input must be on NPU");
    auto sizes = self.sizes();
    int64_t rank = sizes.size();
    TORCH_CHECK(rank > 0, "argmin over scalar tensor is not supported");
    int64_t axis = NormalizeDim(dim, rank);
    TORCH_CHECK(axis >= 0 && axis < rank, "dim out of range");

    std::vector<int64_t> out_sizes;
    out_sizes.reserve(rank - 1);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == axis) continue;
        out_sizes.push_back(sizes[i]);
    }

    at::Tensor result;
    if (out_sizes.empty()) {
        result = at::empty({}, self.options().dtype(at::kLong));
    } else {
        result = at::empty(out_sizes, self.options().dtype(at::kLong));
    }

    EXEC_NPU_CMD(aclnnArgminOverADimensionCustom, self, static_cast<int32_t>(axis), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmin_over_a_dimension_custom", &argmin_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmin_over_a_dimension_custom", &argmin_impl_npu, "Argmin over a specific dimension (ArgminOverADimensionCustom on NPU)");
}
