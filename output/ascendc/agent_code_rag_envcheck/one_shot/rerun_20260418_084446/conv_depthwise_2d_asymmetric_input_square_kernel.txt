kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;
constexpr int32_t TMP_BUF_NUM = 2;

class KernelConvDepthwise2dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t KH, uint32_t KW, uint32_t OH, uint32_t OW)
    {
        n_ = N; c_ = C; h_ = H; w_ = W;
        kh_ = KH; kw_ = KW; oh_ = OH; ow_ = OW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N * C * H * W);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)C * KH * KW);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N * C * OH * OW);

        // Buffers sized to a single output row of length OW
        pipe_.InitBuffer(inSegQ_, IN_BUF_NUM, ow_ * sizeof(float));
        pipe_.InitBuffer(outRowQ_, OUT_BUF_NUM, ow_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, TMP_BUF_NUM, ow_ * sizeof(float));

        // Small buffer for 9 weights
        pipe_.InitBuffer(wLocalQ_, 1, 16 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || oh_ == 0 || ow_ == 0) return;

        if (g_coreType == AscendC::AIV) {
            return;
        }

        const uint32_t totalGroups = n_ * c_;
        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t gc = blockIdx; gc < totalGroups; gc += blockNum) {
            const uint32_t n = gc / c_;
            const uint32_t ch = gc % c_;

            const uint64_t xBase = ((uint64_t)n * c_ + ch) * (uint64_t)h_ * (uint64_t)w_;
            const uint64_t yBase = ((uint64_t)n * c_ + ch) * (uint64_t)oh_ * (uint64_t)ow_;
            const uint64_t wBase = (uint64_t)ch * (uint64_t)kh_ * (uint64_t)kw_;

            // Load 9 weights into local buffer
            LocalTensor<float> wLocal = wLocalQ_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[wBase], kh_ * kw_);
            wLocalQ_.EnQue(wLocal);

            // Extract scalar weights
            wLocal = wLocalQ_.DeQue<float>();
            float w00 = wLocal.GetValue(0);
            float w01 = wLocal.GetValue(1);
            float w02 = wLocal.GetValue(2);
            float w10 = wLocal.GetValue(3);
            float w11 = wLocal.GetValue(4);
            float w12 = wLocal.GetValue(5);
            float w20 = wLocal.GetValue(6);
            float w21 = wLocal.GetValue(7);
            float w22 = wLocal.GetValue(8);
            wLocalQ_.FreeTensor(wLocal);

            for (uint32_t oh = 0; oh < oh_; ++oh) {
                // Allocate output row
                LocalTensor<float> yRow = outRowQ_.AllocTensor<float>();

                // Initialize yRow to 0 by first copying a segment and zeroing it
                // Use a tmp to zero yRow to avoid extra kernel. Copy r0 shift0 then mul by 0.
                {
                    LocalTensor<float> tmp0 = inSegQ_.AllocTensor<float>();
                    AscendC::DataCopy(tmp0, xGm_[xBase + (uint64_t)(oh + 0) * w_ + 0], ow_);
                    AscendC::Muls(yRow, tmp0, 0.0f, ow_);
                    inSegQ_.FreeTensor(tmp0);
                }

                // Accumulate 9 shifted segments
                AccumulateSegment(xBase, oh, 0, w00, yRow);
                AccumulateSegment(xBase, oh, 1, w01, yRow);
                AccumulateSegment(xBase, oh, 2, w02, yRow);

                AccumulateSegment(xBase + (uint64_t)w_, oh, 0, w10, yRow);
                AccumulateSegment(xBase + (uint64_t)w_, oh, 1, w11, yRow);
                AccumulateSegment(xBase + (uint64_t)w_, oh, 2, w12, yRow);

                AccumulateSegment(xBase + (uint64_t)w_ * 2, oh, 0, w20, yRow);
                AccumulateSegment(xBase + (uint64_t)w_ * 2, oh, 1, w21, yRow);
                AccumulateSegment(xBase + (uint64_t)w_ * 2, oh, 2, w22, yRow);

                // Write back y row
                AscendC::DataCopy(yGm_[yBase + (uint64_t)oh * ow_], yRow, ow_);
                outRowQ_.FreeTensor(yRow);
            }
        }
    }

private:
    // baseX: xBase + rowOffsetPrefix where rowOffsetPrefix already includes which of the 3 rows
    // oh: output row index
    // shift: 0,1,2 for kernel column shift
    // scalar: weight scalar
    __aicore__ inline void AccumulateSegment(uint64_t baseXPrefix, uint32_t oh, uint32_t shift,
                                             float scalar, LocalTensor<float>& yRow)
    {
        const uint64_t gmOffset = baseXPrefix + (uint64_t)oh * (uint64_t)w_ + (uint64_t)shift;
        LocalTensor<float> tmp = inSegQ_.AllocTensor<float>();
        AscendC::DataCopy(tmp, xGm_[gmOffset], ow_);
        AscendC::Muls(tmp, tmp, scalar, ow_);
        AscendC::Add(yRow, yRow, tmp, ow_);
        inSegQ_.FreeTensor(tmp);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inSegQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wLocalQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t kh_{0}, kw_{0}, oh_{0}, ow_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling, tilingGm);
    KernelConvDepthwise2dAsymmetricInputSquareKernelCustom op;
    op.Init(x, w, y,
            tiling.nSize, tiling.cSize, tiling.hSize, tiling.wSize,
            tiling.kernelH, tiling.kernelW, tiling.outH, tiling.outW);
    op.Process();
}
