host_operator_src="""

#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"
#include <cstdint>

namespace optiling {

// UB model for this kernel:
//   inQueue: 2 * tileCols
//   outQueue: 2 * tileCols
// Total floats = 4 * tileCols => bytes = 16 * tileCols
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 65536;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &origin = x_shape->GetOriginShape();
    const int32_t rank = static_cast<int32_t>(origin.GetDimNum());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimAttrPtr = attrs->GetAttrPointer<int64_t>(0);
    int32_t axis = (dimAttrPtr != nullptr) ? static_cast<int32_t>(*dimAttrPtr) : 1;

    // Normalize axis
    if (axis < 0) axis += rank;
    if (axis < 0) axis = 0;
    if (axis >= rank) axis = rank - 1;

    // Compute batchSize and rowSize (softmax along 'axis')
    uint64_t totalLength = 1;
    for (int32_t i = 0; i < rank; ++i) {
        totalLength *= static_cast<uint64_t>(origin.GetDim(i));
    }
    uint32_t rowSize = static_cast<uint32_t>(origin.GetDim(axis));
    uint32_t batchSize = static_cast<uint32_t>(totalLength / rowSize);

    // tileCols from UB budget: bytes = 16 * tileCols
    uint32_t maxTile = UB_BUDGET_BYTES / 16;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileCols = maxTile;
    if (rowSize < tileCols) {
        const uint32_t aligned = AlignDown(rowSize, ALIGN_ELEMS);
        tileCols = (aligned > 0) ? aligned : rowSize;
    }

    // Adaptive blockDim based on number of rows; keep conservative for stability.
    uint32_t blockDim = batchSize;
    if (blockDim < 8u) blockDim = (batchSize > 0 ? 1u : 1u); // allow small when few rows
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_batchSize(batchSize);
    tiling.set_rowSize(rowSize);
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(axis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SoftmaxCustom : public OpDef {
public:
    explicit SoftmaxCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SoftmaxCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"
#include <math.h>

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t batchSize, uint32_t rowSize, uint32_t tileCols)
    {
        batchSize_ = batchSize;
        rowSize_ = rowSize;
        tileCols_ = tileCols;
        totalLen_ = batchSize_ * rowSize_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Two in-flight tiles for in/out to enable overlap
        pipe_.InitBuffer(inQueue_, IN_BUF_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUT_BUF_NUM, tileCols_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen == 0 || rowSize_ == 0 || tileCols_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        for (uint32_t row = blockIdx; row < batchSize_; row += blockNum) {
            // Pass 1: compute row max and row sum in a numerically stable streaming manner.
            float m = -INFINITY;
            float s = 0.0f;

            for (uint32_t base = 0; base < rowSize_; base += tileCols_) {
                const uint32_t gmOffset = row * rowSize_ + base;
                uint32_t len = rowSize_ - base;
                if (len > tileCols_) len = tileCols_;

                // Load a tile of the row
                AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
                inQueue_.EnQue(xLocal);

                AscendC::LocalTensor<float> xBuf = inQueue_.DeQue<float>();

                // Compute tile max
                float tileMax = -INFINITY;
                for (uint32_t i = 0; i < len; ++i) {
                    float v = xBuf.GetValue(i);
                    tileMax = (v > tileMax) ? v : tileMax;
                }

                // Update running max and sum
                float new_m = (m > tileMax) ? m : tileMax;
                float scale = (m == -INFINITY && new_m == -INFINITY) ? 0.0f : expf(m - new_m);
                s = s * scale;

                for (uint32_t i = 0; i < len; ++i) {
                    float v = xBuf.GetValue(i);
                    s += expf(v - new_m);
                }
                m = new_m;

                inQueue_.FreeTensor(xBuf);
            }

            // Avoid division by zero; if all -inf, set s to 1
            if (!(s > 0.0f)) s = 1.0f;

            // Pass 2: write normalized outputs
            for (uint32_t base = 0; base < rowSize_; base += tileCols_) {
                const uint32_t gmOffset = row * rowSize_ + base;
                uint32_t len = rowSize_ - base;
                if (len > tileCols_) len = tileCols_;

                AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
                inQueue_.EnQue(xLocal);

                AscendC::LocalTensor<float> xBuf = inQueue_.DeQue<float>();
                AscendC::LocalTensor<float> yBuf = outQueue_.AllocTensor<float>();

                for (uint32_t i = 0; i < len; ++i) {
                    float v = xBuf.GetValue(i);
                    float yv = expf(v - m) / s;
                    yBuf.SetValue(i, yv);
                }

                outQueue_.EnQue(yBuf);
                inQueue_.FreeTensor(xBuf);

                AscendC::LocalTensor<float> yLocal = outQueue_.DeQue<float>();
                AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
                outQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t batchSize_{0};
    uint32_t rowSize_{0};
    uint32_t tileCols_{0};
    uint32_t totalLen_{0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.batchSize, tiling_data.rowSize, tiling_data.tileCols);
    op.Process();
}
"""
