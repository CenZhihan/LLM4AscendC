#include "conv2d_hard_swish_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dHardSwishReluCustomTilingData tiling;

    // Shapes:
    // x: [N, Cin, H, W]
    // weight: [Cout, Cin, K, K]
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    if (xShape == nullptr || wShape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const auto &xDims = xShape->GetOriginShape().GetDims();
    const auto &wDims = wShape->GetOriginShape().GetDims();

    if (xDims.size() != 4 || wDims.size() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(xDims[0]);
    const uint32_t Cin = static_cast<uint32_t>(xDims[1]);
    const uint32_t H   = static_cast<uint32_t>(xDims[2]);
    const uint32_t W   = static_cast<uint32_t>(xDims[3]);

    const uint32_t Cout = static_cast<uint32_t>(wDims[0]);
    // wDims[1] should equal Cin
    const uint32_t K    = static_cast<uint32_t>(wDims[2]); // assume square kernel

    // Stride=1, Padding=0
    if (H < K || W < K) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t outH = H - K + 1;
    const uint32_t outW = W - K + 1;

    const uint32_t totalOut = N * Cout * outH * outW;

    // Bias presence
    bool hasBias = false;
    // Try to detect optional bias input presence
    const gert::Shape *bShape = nullptr;
    // Some runtimes return nullptr for missing optional inputs
    // index 2 corresponds to bias
    bShape = context->GetInputShape(2);
    if (bShape != nullptr) {
        hasBias = true;
    }

    // Choose a reasonable tile length for grid-stride processing
    static constexpr uint32_t ALIGN_ELEMS = 256;
    static constexpr uint32_t HARD_CAP_ELEMS = 65536;

    uint32_t tileLen = (totalOut > HARD_CAP_ELEMS) ? HARD_CAP_ELEMS : totalOut;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = (totalOut >= 1u) ? 1u : 0u;

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOut, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_K(K);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_totalOut(totalOut);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1u : 0u);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    if (xShape == nullptr || wShape == nullptr) {
        return GRAPH_FAILED;
    }
    const auto &xDims = xShape->GetOriginShape().GetDims();
    const auto &wDims = wShape->GetOriginShape().GetDims();
    if (xDims.size() != 4 || wDims.size() != 4) {
        return GRAPH_FAILED;
    }

    const int64_t N   = xDims[0];
    const int64_t H   = xDims[2];
    const int64_t W   = xDims[3];
    const int64_t K   = wDims[2];
    const int64_t Cout = wDims[0];

    const int64_t outH = H - K + 1;
    const int64_t outW = W - K + 1;

    yShape->SetDimNum(4);
    yShape->SetDim(0, N);
    yShape->SetDim(1, Cout);
    yShape->SetDim(2, outH);
    yShape->SetDim(3, outW);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dHardSwishReluCustom {
public:
    __aicore__ inline KernelConv2dHardSwishReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t outH, uint32_t outW,
                                uint32_t totalOut, uint32_t tileLength,
                                uint32_t hasBiasFlag)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout; H_ = H; W_ = W; K_ = K;
        outH_ = outH; outW_ = outW;
        totalOut_ = totalOut; tileLen_ = tileLength;
        hasBias_ = (hasBiasFlag != 0);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(N_) * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(Cout_) * Cin_ * K_ * K_);
        if (b != nullptr) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut_);
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOut_) return;

        while (base < totalOut_) {
            uint32_t remain = totalOut_ - base;
            uint32_t cur = (remain > tileLen_) ? tileLen_ : remain;

            ComputeRange(base, cur);

            base += strideElems;
        }
    }

private:
    __aicore__ inline uint32_t IndexX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((n * Cin_ + c) * H_ + h) * W_) + w;
    }

    __aicore__ inline uint32_t IndexW(uint32_t oc, uint32_t ic, uint32_t kh, uint32_t kw) const
    {
        return (((oc * Cin_ + ic) * K_ + kh) * K_) + kw;
    }

    __aicore__ inline uint32_t IndexY(uint32_t n, uint32_t oc, uint32_t oh, uint32_t ow) const
    {
        return (((n * Cout_ + oc) * outH_ + oh) * outW_) + ow;
    }

    __aicore__ inline void DecodeOutIndex(uint32_t idx,
                                          uint32_t &n, uint32_t &oc,
                                          uint32_t &oh, uint32_t &ow) const
    {
        ow = idx % outW_;
        idx /= outW_;
        oh = idx % outH_;
        idx /= outH_;
        oc = idx % Cout_;
        idx /= Cout_;
        n = idx;
    }

    __aicore__ inline void ComputeRange(uint32_t start, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            uint32_t flat = start + i;

            uint32_t n, oc, oh, ow;
            DecodeOutIndex(flat, n, oc, oh, ow);

            float acc = 0.0f;
            if (hasBias_) {
                acc = bGm_[oc];
            }

            // Convolution sum for this output element
            for (uint32_t ic = 0; ic < Cin_; ++ic) {
                for (uint32_t kh = 0; kh < K_; ++kh) {
                    const uint32_t ih = oh + kh;
                    for (uint32_t kw = 0; kw < K_; ++kw) {
                        const uint32_t iw = ow + kw;
                        const float xv = xGm_[IndexX(n, ic, ih, iw)];
                        const float wv = wGm_[IndexW(oc, ic, kh, kw)];
                        acc = AscendC::FAdd(acc, AscendC::FMul(xv, wv));
                    }
                }
            }

            // Fused HardSwish followed by ReLU:
            // hs = acc * clamp(acc + 3, 0, 6) / 6
            // y = max(hs, 0)
            float t = AscendC::FAdd(acc, 3.0f);
            // clamp t to [0, 6]
            t = (t < 0.0f) ? 0.0f : t;
            t = (t > 6.0f) ? 6.0f : t;
            float hs = AscendC::FMul(acc, t);
            hs = AscendC::FMul(hs, 1.0f / 6.0f);
            float yv = (hs < 0.0f) ? 0.0f : hs;

            yGm_[IndexY(n, oc, oh, ow)] = yv;
        }
    }

private:
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0}, H_{0}, W_{0}, K_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t totalOut_{0};
    uint32_t tileLen_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_hard_swish_relu_custom(GM_ADDR x,
                                                                     GM_ADDR weight,
                                                                     GM_ADDR bias,
                                                                     GM_ADDR y,
                                                                     GM_ADDR workspace,
                                                                     GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dHardSwishReluCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.K,
            tiling_data.outH, tiling_data.outW,
            tiling_data.totalOut, tiling_data.tileLength,
            tiling_data.hasBias);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor alloc_output_like_conv2d(const at::Tensor& x, const at::Tensor& weight) {
    // x: [N, Cin, H, W], weight: [Cout, Cin, K, K]; stride=1, pad=0
    auto sizes_x = x.sizes();
    auto sizes_w = weight.sizes();
    int64_t N = sizes_x[0];
    int64_t H = sizes_x[2];
    int64_t W = sizes_x[3];
    int64_t K = sizes_w[2];
    int64_t Cout = sizes_w[0];
    int64_t outH = H - K + 1;
    int64_t outW = W - K + 1;
    return at::empty({N, Cout, outH, outW}, x.options());
}

at::Tensor conv2d_hard_swish_relu_impl_npu(const at::Tensor& x,
                                           const at::Tensor& weight,
                                           c10::optional<at::Tensor> bias_opt) {
    at::Tensor y = alloc_output_like_conv2d(x, weight);
    if (bias_opt.has_value()) {
        at::Tensor bias = bias_opt.value();
        EXEC_NPU_CMD(aclnnConv2dHardSwishReluCustom, x, weight, bias, y);
    } else {
        EXEC_NPU_CMD(aclnnConv2dHardSwishReluCustom, x, weight, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_hard_swish_relu_custom", &conv2d_hard_swish_relu_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_hard_swish_relu_custom",
