host_operator_src="""

#include "conv2d_scaling_min_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}

// UB budget per core (bytes) used mainly for y-tile buffer + small weight/bias staging
static constexpr uint32_t UB_BUDGET_BYTES = 160 * 1024;
static constexpr uint32_t ALIGN_ELEMS     = 64;
static constexpr uint32_t HARD_CAP_ELEMS  = 16384;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dScalingMinCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);   // [N, Ci, H, W]
    const gert::Shape *w_shape = context->GetInputShape(1);   // [Co, Ci, Kh, Kw]
    if (x_shape->GetOriginShape().GetDimNum() != 4 || w_shape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const int64_t N  = x_shape->GetOriginShape().GetDim(0);
    const int64_t Ci = x_shape->GetOriginShape().GetDim(1);
    const int64_t H  = x_shape->GetOriginShape().GetDim(2);
    const int64_t W  = x_shape->GetOriginShape().GetDim(3);

    const int64_t Co = w_shape->GetOriginShape().GetDim(0);
    const int64_t Ci_w = w_shape->GetOriginShape().GetDim(1);
    const int64_t Kh = w_shape->GetOriginShape().GetDim(2);
    const int64_t Kw = w_shape->GetOriginShape().GetDim(3);

    if (Ci != Ci_w) {
        return ge::GRAPH_FAILED;
    }
    if (Kh <= 0 || Kw <= 0 || H < Kh || W < Kw) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t oh = static_cast<uint32_t>(H - Kh + 1);
    const uint32_t ow = static_cast<uint32_t>(W - Kw + 1);
    const uint32_t totalPos = static_cast<uint32_t>(N) * oh * ow;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scaleValuePtr = attrs->GetAttrPointer<float>(0);
    float scaleValue = (scaleValuePtr != nullptr) ? *scaleValuePtr : 1.0f;

    // y-tile buffer must hold tileLen values
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    // leave headroom for small temporaries; cap to prevent long per-iter latency
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalPos < tileLen) {
        const uint32_t aligned = AlignDown(totalPos, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalPos;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalPos, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_ci(static_cast<uint32_t>(Ci));
    tiling.set_h(static_cast<uint32_t>(H));
    tiling.set_w(static_cast<uint32_t>(W));
    tiling.set_co(static_cast<uint32_t>(Co));
    tiling.set_kh(static_cast<uint32_t>(Kh));
    tiling.set_kw(static_cast<uint32_t>(Kw));
    tiling.set_oh(oh);
    tiling.set_ow(ow);
    tiling.set_totalPos(totalPos);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_scaleValue(scaleValue);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

// Naive reference implementation:
// y[n,1,oh,ow] = min_oc( (sum_{ic,kh,kw} x[n,ic,oh+kh,ow+kw] * w[oc,ic,kh,kw] + bias[oc]) * scale )

class KernelConv2dScalingMinCustom {
public:
    __aicore__ inline KernelConv2dScalingMinCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t ci, uint32_t h, uint32_t wIn,
                                uint32_t co, uint32_t kh, uint32_t kw,
                                uint32_t oh, uint32_t ow,
                                uint32_t totalPos, uint32_t tileLen,
                                float scaleValue)
    {
        n_ = n; ci_ = ci; h_ = h; wIn_ = wIn;
        co_ = co; kh_ = kh; kw_ = kw;
        oh_ = oh; ow_ = ow;
        totalPos_ = totalPos;
        tileLen_ = tileLen;
        scale_ = scaleValue;

        xPtr_ = reinterpret_cast<__gm__ const float*>(x);
        wPtr_ = reinterpret_cast<__gm__ const float*>(w);
        bPtr_ = reinterpret_cast<__gm__ const float*>(b);
        yPtr_ = reinterpret_cast<__gm__ float*>(y);
    }

    __aicore__ inline void Process()
    {
        if (totalPos_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t stridePos = blockNum * tileLen_;
        uint32_t basePos = blockIdx * tileLen_;
        if (basePos >= totalPos_) return;

        while (true) {
            uint32_t curLen = totalPos_ - basePos;
            if (curLen > tileLen_) curLen = tileLen_;
            ComputeTile(basePos, curLen);
            if (basePos + stridePos >= totalPos_) break;
            basePos += stridePos;
        }
    }

private:
    __aicore__ inline uint32_t WeightsPerOC() const { return ci_ * kh_ * kw_; }

    __aicore__ inline void PosToNHW(uint32_t pos, uint32_t &n, uint32_t &oh, uint32_t &ow) const
    {
        const uint32_t hwSize = oh_ * ow_;
        n = pos / hwSize;
        const uint32_t r = pos - n * hwSize;
        oh = r / ow_;
        ow = r - oh * ow_;
    }

    __aicore__ inline void ComputeTile(uint32_t basePos, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            const uint32_t pos = basePos + i;

            uint32_t nIdx, ohIdx, owIdx;
            PosToNHW(pos, nIdx, ohIdx, owIdx);

            float minVal = 0.0f;
            bool first = true;

            const uint32_t wPerOC = WeightsPerOC();

            for (uint32_t oc = 0; oc < co_; ++oc) {
                float acc = 0.0f;

                const uint32_t wBase = oc * wPerOC;

                for (uint32_t ic = 0; ic < ci_; ++ic) {
                    for (uint32_t kh = 0; kh < kh_; ++kh) {
                        const uint32_t ih = ohIdx + kh;
                        const uint32_t wKHBase = wBase + ic * (kh_ * kw_) + kh * kw_;
                        const uint32_t xIHBase = ((nIdx * ci_ + ic) * h_ + ih) * wIn_;
                        for (uint32_t kw = 0; kw < kw_; ++kw) {
                            const uint32_t iw = owIdx + kw;
                            const uint32_t xIndex = xIHBase + iw;
                            const uint32_t wIndex = wKHBase + kw;
                            const float xv = xPtr_[xIndex];
                            const float wv = wPtr_[wIndex];
                            acc += xv * wv;
                        }
                    }
                }

                if (bPtr_ != nullptr) {
                    acc += bPtr_[oc];
                }
                acc *= scale_;

                if (first) {
                    minVal = acc;
                    first = false;
                } else {
                    minVal = (acc < minVal) ? acc : minVal;
                }
            }

            yPtr_[pos] = minVal;
        }
    }

private:
    __gm__ const float* xPtr_ {nullptr};
    __gm__ const float* wPtr_ {nullptr};
    __gm__ const float* bPtr_ {nullptr};
    __gm__ float* yPtr_ {nullptr};

    uint32_t n_ {0}, ci_ {0}, h_ {0}, wIn_ {0};
    uint32_t co_ {0}, kh_ {0}, kw_ {0};
    uint32_t oh_ {0}, ow_ {0};
    uint32_t totalPos_ {0}, tileLen_ {0};
    float scale_ {1.0f};
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dScalingMinCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.n, tiling_data.ci, tiling_data.h, tiling_data.w,
            tiling_data.co, tiling_data.kh, tiling_data.kw,
            tiling_data.oh, tiling_data.ow,
            tiling_data.totalPos, tiling_data.tileLen,
            tiling_data.scaleValue);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor allocate_output_like_conv_reduce(
    const at::Tensor& x, const at::Tensor& weight)
{
    TORCH_CHECK(x.dim() == 4, \"x must be 4D NCHW\");
    TORCH_CHECK(weight.dim() == 4, \"weight must be 4D OIHW\");
    const int64_t N = x.size(0);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);
    const int64_t Kh = weight.size(2);
    const int64_t Kw = weight.size(3);
    TORCH_CHECK(H >= Kh && W >= Kw, \"Invalid kernel for given input\");
    const int64_t Oh = H - Kh + 1;
    const int64_t Ow = W - Kw + 1;
    return at::empty({N, 1, Oh, Ow}, x.options());
}

at::Tensor conv2d_scaling_min_impl_npu(const at::Tensor& x,
                                       const at::Tensor& weight,
                                       const c10::optional<at::Tensor>& bias_opt,
                                       const at::Tensor& scale)
{
    TORCH_CHECK(scale.numel() == 1, \"scale must be scalar tensor\");
    const double s = scale.item<double>();
    at::Tensor y = allocate_output_like_conv_reduce(x, weight);

    if (bias_opt.has_value()) {
        const at::Tensor& b = bias_opt.value();
        EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, b, s, y);
    } else {
        EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, at::Tensor(), s, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv2d_scaling_min_custom\", &conv2d_scaling_min_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"conv2d_scaling_min_custom\", &conv2d_scaling_min_impl_npu,
          \"Conv2D + scale + channelwise min reduction (Conv2dScalingMinCustom on NPU)\");
}
