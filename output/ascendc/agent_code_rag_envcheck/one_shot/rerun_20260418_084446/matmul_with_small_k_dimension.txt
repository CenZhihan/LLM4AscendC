#include "matmul_with_small_k_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (a == 0) ? x : (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return (a == 0) ? x : ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (b == 0) ? 0 : ((a + b - 1u) / b); }
}

namespace optiling {

// UB model for this kernel:
//   aBlock: tileM * K
//   bRow:   tileN            (loaded per k)
//   cTile:  tileM * tileN    (accumulator in UB; row-major)
//   tmp:    tileN
// Total floats = tileM*K + tileN + tileM*tileN + tileN = tileM*K + tileM*tileN + 2*tileN
// We cap UB per-core usage conservatively to 192 KB.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t VEC_ALIGN = 16;   // align elements for better DMA/VEC
static constexpr uint32_t MAX_BLOCKS = 24;  // conservative AICore usage

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithSmallKDimensionCustomTilingData tiling;

    const auto *shapeA = context->GetInputShape(0);
    const auto *shapeB = context->GetInputShape(1);

    // Expect A: [M, K], B: [K, N]
    const uint32_t A_rank = static_cast<uint32_t>(shapeA->GetOriginShape().GetDims());
    const uint32_t B_rank = static_cast<uint32_t>(shapeB->GetOriginShape().GetDims());
    if (A_rank != 2 || B_rank != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(1));
    const uint32_t Kb = static_cast<uint32_t>(shapeB->GetOriginShape().GetDim(0));
    const uint32_t N = static_cast<uint32_t>(shapeB->GetOriginShape().GetDim(1));
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    // Compute tile sizes to fit UB budget
    const uint32_t ubFloats = UB_BUDGET_BYTES / 4; // float32
    // Choose base tileM to be friendly to vectorization and K-loop; prefer 128, fallback to smaller if M small.
    uint32_t tileM = (M >= 128) ? 128u : AlignDown(M, VEC_ALIGN);
    if (tileM == 0) tileM = M;  // when M < VEC_ALIGN

    // Derive tileN from budget: tileM*K + tileM*tileN + 2*tileN <= ubFloats
    // => tileN * (tileM + 2) <= ubFloats - tileM*K
    uint32_t maxTileN = 0;
    if (ubFloats > tileM * K) {
        maxTileN = (ubFloats - tileM * K) / (tileM + 2u);
    }
    // clamp and align
    if (maxTileN == 0) maxTileN = VEC_ALIGN;
    maxTileN = AlignDown(maxTileN, VEC_ALIGN);
    if (maxTileN < VEC_ALIGN) maxTileN = VEC_ALIGN;
    if (maxTileN > N) {
        maxTileN = AlignDown(N, VEC_ALIGN);
        if (maxTileN == 0) maxTileN = N;
    }

    // If M is very small, shrink tileM to match budget and vector alignment
    if (M < tileM) {
        tileM = AlignDown(M, VEC_ALIGN);
        if (tileM == 0) tileM = M;
    }

    // Compute tiles and block dim
    const uint32_t tilesM = CeilDiv(M, tileM);
    const uint32_t tilesN = CeilDiv(N, maxTileN);
    uint32_t totalTiles = tilesM * tilesN;
    if (totalTiles == 0) totalTiles = 1;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(maxTileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulWithSmallKDimensionCustom {
public:
    __aicore__ inline KernelMatmulWithSmallKDimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileM_ = tileM;
        tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(M_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(K_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(M_) * N_);

        // Allocate UB buffers
        // aBlock: tileM_ * K_
        pipe_.InitBuffer(aQue_, BUFFER_NUM, static_cast<uint64_t>(tileM_) * K_ * sizeof(float));
        // bRow: tileN_
        pipe_.InitBuffer(bQue_, BUFFER_NUM, static_cast<uint64_t>(tileN_) * sizeof(float));
        // cTile: tileM_ * tileN_
        pipe_.InitBuffer(cQue_, BUFFER_NUM, static_cast<uint64_t>(tileM_) * tileN_ * sizeof(float));
        // tmp: tileN_
        pipe_.InitBuffer(tmpQue_, BUFFER_NUM, static_cast<uint64_t>(tileN_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t tilesM = CeilDiv(M_, tileM_);
        const uint32_t tilesN = CeilDiv(N_, tileN_);
        const uint32_t totalTiles = tilesM * tilesN;

        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            const uint32_t tm = tileIdx / tilesN;
            const uint32_t tn = tileIdx % tilesN;

            const uint32_t rowStart = tm * tileM_;
            const uint32_t colStart = tn * tileN_;
            const uint32_t curM = MinU32(tileM_, M_ - rowStart);
            const uint32_t curN = MinU32(tileN_, N_ - colStart);

            // Load A block [curM x K_], contiguous
            AscendC::LocalTensor<float> aLocal = aQue_.AllocTensor<float>();
            AscendC::DataCopy(aLocal, aGm_[static_cast<uint64_t>(rowStart) * K_], static_cast<uint64_t>(curM) * K_);
            aQue_.EnQue(aLocal);

            // Prepare C tile buffer
            AscendC::LocalTensor<float> cLocal = cQue_.AllocTensor<float>();
            cQue_.EnQue(cLocal);

            // Temp buffers
            AscendC::LocalTensor<float> tmpLocal = tmpQue_.AllocTensor<float>();
            tmpQue_.EnQue(tmpLocal);

            // Dequeue for compute
            aLocal = aQue_.DeQue<float>();
            cLocal = cQue_.DeQue<float>();
            tmpLocal = tmpQue_.DeQue<float>();

            // Compute: C[row, :] accumulated in cLocal row-major (row length = tileN_)
            for (uint32_t k = 0; k < K_; ++k) {
                AscendC::LocalTensor<float> bLocal = bQue_.AllocTensor<float>();
                // Load one B row segment [curN] starting at (k, colStart)
                AscendC::DataCopy(bLocal, bGm_[static_cast<uint64_t>(k) * N_ + colStart], curN);
                bQue_.EnQue(bLocal);
                bLocal = bQue_.DeQue<float>();

                for (uint32_t i = 0; i < curM; ++i) {
                    const uint32_t aOff = i * K_ + k;
                    float aScalar = aLocal.GetValue(aOff);  // read scalar from UB

                    AscendC::LocalTensor<float> cRow = cLocal[static_cast<uint64_t>(i) * tileN_];
                    if (k == 0) {
                        AscendC::Muls(cRow, bLocal, aScalar, curN);
                    } else {
                        AscendC::Muls(tmpLocal, bLocal, aScalar, curN);
                        AscendC::Add(cRow, cRow, tmpLocal, curN);
                    }
                }

                bQue_.FreeTensor(bLocal);
            }

            // Store C block back to GM by rows
            for (uint32_t i = 0; i < curM; ++i) {
                AscendC::LocalTensor<float> cRow = cLocal[static_cast<uint64_t>(i) * tileN_];
                AscendC::DataCopy(cGm_[static_cast<uint64_t>(rowStart + i) * N_ + colStart], cRow, curN);
            }

            // Free locals
            cQue_.FreeTensor(cLocal);
            tmpQue_.FreeTensor(tmpLocal);
            aQue_.FreeTensor(aLocal);
        }
    }

private:
    __aicore__ static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
    __aicore__ static inline uint32_t MinU32(uint32_t a, uint32_t b) { return (a < b) ? a : b; }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> aQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> cQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileM_{0};
    uint32_t tileN_{0};
};

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithSmallKDimensionCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
