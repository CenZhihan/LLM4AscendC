kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y,
                                uint32_t rows, uint32_t cols, float margin)
    {
        rows_ = rows;
        cols_ = cols;
        margin_ = margin;

        aGm_.SetGlobalBuffer((__gm__ float*)anchor, rows_ * cols_);
        pGm_.SetGlobalBuffer((__gm__ float*)positive, rows_ * cols_);
        nGm_.SetGlobalBuffer((__gm__ float*)negative, rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        // Each iteration processes exactly one row to simplify reduction
        pipe_.InitBuffer(aQueue_, BUFFER_NUM, cols_ * sizeof(float));
        pipe_.InitBuffer(pQueue_, BUFFER_NUM, cols_ * sizeof(float));
        pipe_.InitBuffer(nQueue_, BUFFER_NUM, cols_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, cols_ * sizeof(float));   // for diff buffers
        pipe_.InitBuffer(redQueue_, 2, sizeof(float));           // for scalar reductions
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        // Initialize output to zero once by block 0
        if (blockIdx == 0) {
            LocalTensor<float> initLocal = redQueue_.AllocTensor<float>();
            Duplicate(initLocal, 0.0f, 1);
            DataCopy(yGm_[0], initLocal, 1);
            redQueue_.FreeTensor(initLocal);
        }
        // Ensure all blocks see zero-initialized y
        PipeBarrier();

        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            // Prefetch row
            const uint32_t base = row * cols_;
            CopyInRow(base);

            // Compute distance terms and per-row hinge
            float loss_i = ComputeRow();

            // Accumulate mean via atomic add: loss_i / rows_
            const float add_val = loss_i / static_cast<float>(rows_);
            AtomicAdd(yGm_.GetPhyAddr(0), add_val);
        }
    }

private:
    __aicore__ inline void CopyInRow(uint32_t gmOffset)
    {
        LocalTensor<float> aLocal = aQueue_.AllocTensor<float>();
        LocalTensor<float> pLocal = pQueue_.AllocTensor<float>();
        LocalTensor<float> nLocal = nQueue_.AllocTensor<float>();
        DataCopy(aLocal, aGm_[gmOffset], cols_);
        DataCopy(pLocal, pGm_[gmOffset], cols_);
        DataCopy(nLocal, nGm_[gmOffset], cols_);
        aQueue_.EnQue(aLocal);
        pQueue_.EnQue(pLocal);
        nQueue_.EnQue(nLocal);
    }

    __aicore__ inline float ComputeRow()
    {
        LocalTensor<float> aLocal = aQueue_.DeQue<float>();
        LocalTensor<float> pLocal = pQueue_.DeQue<float>();
        LocalTensor<float> nLocal = nQueue_.DeQue<float>();
        LocalTensor<float> diff1 = tmpQueue_.AllocTensor<float>();
        LocalTensor<float> diff2 = tmpQueue_.AllocTensor<float>();

        // diff1 = (a - p)^2
        Sub(diff1, aLocal, pLocal, cols_);
        Mul(diff1, diff1, diff1, cols_);

        // diff2 = (a - n)^2
        Sub(diff2, aLocal, nLocal, cols_);
        Mul(diff2, diff2, diff2, cols_);

        // Reduce sums to scalars
        LocalTensor<float> sum1 = redQueue_.AllocTensor<float>();
        LocalTensor<float> sum2 = redQueue_.AllocTensor<float>();
        ReduceSum(sum1, diff1, cols_);
        ReduceSum(sum2, diff2, cols_);

        // Take sqrt to get L2 distance
        Sqrt(sum1, sum1, 1);
        Sqrt(sum2, sum2, 1);

        // hinge = max(0, dist_ap - dist_an + margin)
        LocalTensor<float> hingeLocal = redQueue_.AllocTensor<float>();
        Sub(hingeLocal, sum1, sum2, 1);
        Adds(hingeLocal, hingeLocal, margin_, 1);
        Maxs(hingeLocal, hingeLocal, 0.0f, 1);

        // Read back scalar from local buffer
        float hinge_val = 0.0f;
        DataCopy(&hinge_val, hingeLocal, 1);

        // Free queues
        redQueue_.FreeTensor(hingeLocal);
        redQueue_.FreeTensor(sum2);
        redQueue_.FreeTensor(sum1);
        tmpQueue_.FreeTensor(diff2);
        tmpQueue_.FreeTensor(diff1);
        nQueue_.FreeTensor(nLocal);
        pQueue_.FreeTensor(pLocal);
        aQueue_.FreeTensor(aLocal);

        return hinge_val;
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> aQueue_;
    TQue<TPosition::VECIN, BUFFER_NUM> pQueue_;
    TQue<TPosition::VECIN, BUFFER_NUM> nQueue_;
    TQue<TPosition::VECCALC, 2> tmpQueue_;
    TQue<TPosition::VECCALC, 2> redQueue_;

    GlobalTensor<float> aGm_;
    GlobalTensor<float> pGm_;
    GlobalTensor<float> nGm_;
    GlobalTensor<float> yGm_;

    uint32_t rows_{0};
    uint32_t cols_{0};
    float margin_{1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative,
                                                                 GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y, tiling_data.rows, tiling_data.cols, tiling_data.margin);
    op.Process();
}
