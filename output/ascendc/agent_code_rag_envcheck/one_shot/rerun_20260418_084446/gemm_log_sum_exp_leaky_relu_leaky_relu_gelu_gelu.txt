kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;
constexpr float NEG_INF_F = -1.0e30f;
constexpr float NEG_SLOPE = 0.01f;
constexpr float HALF = 0.5f;
constexpr float INV_SQRT2 = 0.70710678118654752440f;

class KernelGemmLseLeakyLeakyGeluGelu {
public:
    __aicore__ inline KernelGemmLseLeakyLeakyGeluGelu() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t m, uint32_t k,
                                uint32_t tileM, uint32_t tileK)
    {
        n_ = n; m_ = m; k_ = k;
        tileM_ = tileM; tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * k_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, m_ * k_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, m_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_);

        // UB buffers
        // xChunk (tileK), wRow (tileK), mulVec (tileK),
        // bTile (tileM), yTile (tileM), tmpTile (tileM) for exp/reduce
        pipe_.InitBuffer(xQueue_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(wRowQueue_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(mulVecQueue_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(bTileQueue_, 1, tileM_ * sizeof(float));
        pipe_.InitBuffer(yTileQueue_, 1, tileM_ * sizeof(float));
        pipe_.InitBuffer(tmpTileQueue_, 1, tileM_ * sizeof(float));
        pipe_.InitBuffer(oneValQueue_, 1, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || m_ == 0 || k_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t row = blockIdx; row < n_; row += blockNum) {
            // Compute logsumexp over M of (x[row] @ w^T + b)
            float runningMax = NEG_INF_F;
            float runningSumExp = 0.0f;

            for (uint32_t jBase = 0; jBase < m_; jBase += tileM_) {
                const uint32_t curM = AscendC::min(tileM_, m_ - jBase);

                AscendC::LocalTensor<float> bTile = bTileQueue_.AllocTensor<float>();
                AscendC::DataCopy(bTile, bGm_[jBase], curM);
                bTileQueue_.EnQue(bTile);

                AscendC::LocalTensor<float> yTile = yTileQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpTile = tmpTileQueue_.AllocTensor<float>();

                // Initialize yTile = bTile
                bTile = bTileQueue_.DeQue<float>();
                AscendC::DataCopy(yTile, bTile, curM);
                bTileQueue_.FreeTensor(bTile);

                // Accumulate GEMV for this tile: yTile[r] += sum_k w[jBase+r,k] * x[row,k]
                for (uint32_t kBase = 0; kBase < k_; kBase += tileK_) {
                    const uint32_t curK = AscendC::min(tileK_, k_ - kBase);

                    AscendC::LocalTensor<float> xChunk = xQueue_.AllocTensor<float>();
                    AscendC::DataCopy(xChunk, xGm_[row * k_ + kBase], curK);
                    xQueue_.EnQue(xChunk);
                    xChunk = xQueue_.DeQue<float>();

                    for (uint32_t r = 0; r < curM; ++r) {
                        // wRow
                        AscendC::LocalTensor<float> wRow = wRowQueue_.AllocTensor<float>();
                        AscendC::DataCopy(wRow, wGm_[(jBase + r) * k_ + kBase], curK);
                        wRowQueue_.EnQue(wRow);

                        wRow = wRowQueue_.DeQue<float>();
                        AscendC::LocalTensor<float> mulVec = mulVecQueue_.AllocTensor<float>();
                        AscendC::Mul(mulVec, wRow, xChunk, curK);

                        // Reduce sum mulVec -> scalar at mulVec[0]
                        AscendC::ReduceSum(mulVec, mulVec, curK);

                        // Add to yTile[r]: yTile[r] += mulVec[0]
                        // Use tmp one-element tensor for broadcasting add
                        AscendC::LocalTensor<float> oneVal = oneValQueue_.AllocTensor<float>();
                        // Copy first element to oneVal[0]
                        AscendC::DataCopy(oneVal, mulVec, 1);

                        // Add scalar into position r
                        // tmpTile[0] = yTile[r]
                        AscendC::DataCopy(tmpTile, yTile[r], 1);
                        AscendC::Add(tmpTile, tmpTile, oneVal, 1);
                        AscendC::DataCopy(yTile[r], tmpTile, 1);

                        oneValQueue_.FreeTensor(oneVal);
                        mulVecQueue_.FreeTensor(mulVec);
                        wRowQueue_.FreeTensor(wRow);
                    }

                    xQueue_.FreeTensor(xChunk);
                }

                // Compute tileMax = max(yTile), then update runningMax and runningSumExp
                // tmpTile used as workspace
                AscendC::ReduceMax(tmpTile, yTile, curM); // tmpTile[0] = max
                // Extract tileMax scalar
                AscendC::LocalTensor<float> oneValA = oneValQueue_.AllocTensor<float>();
                AscendC::DataCopy(oneValA, tmpTile, 1);

                // Read scalar from oneValA[0] into tmpTile[0] then reuse
                // Compute new global max
                // tmpTile[0] holds tileMax, replicate computation on scalar buffers
                // runningMax = max(runningMax, tileMax)
                AscendC::LocalTensor<float> oneValB = oneValQueue_.AllocTensor<float>();
                // Store runningMax in oneValB[0]
                // For simplicity, keep runningMax and runningSumExp as scalars in registers,
                // but update with vector math via single-element tensors.
                // Convert runningMax to tensor
                // Fill oneValB with runningMax by DataCopy from a temp on stack not allowed;
                // Instead, we maintain only tensor math:
                // Compute exp using vector ops; for scalar compares, we choose:
                // We'll pull tileMax from oneValA to host scalar-like var by DataCopy to another 1-element UB and assume direct casting.
                // For device, approximate by:
                // Compare via elementwise Max on 1-length tensors.
                AscendC::Duplicate(oneValB, runningMax, 1);
                AscendC::Max(oneValB, oneValB, oneValA, 1); // oneValB[0] = max(runningMax, tileMax)

                // Now we need newM and adjust sums: newM = oneValB[0]
                float newM;
                {
                    AscendC::LocalTensor<float> tmpOne = tmpTileQueue_.AllocTensor<float>();
                    AscendC::DataCopy(tmpOne, oneValB, 1);
                    // Assume we can read first element into scalar
                    newM = ((volatile float*)(tmpOne.GetAddr()))[0];
                    tmpTileQueue_.FreeTensor(tmpOne);
                }

                // Compute sum(exp(yTile - newM))
                AscendC::Sub(tmpTile, yTile, oneValB, curM);   // tmpTile = yTile - newM
                AscendC::Exp(tmpTile, tmpTile, curM);          // exp(...)
                AscendC::ReduceSum(tmpTile, tmpTile, curM);    // tmpTile[0] = sumExpTile
                AscendC::LocalTensor<float> sumTileVal = oneValQueue_.AllocTensor<float>();
                AscendC::DataCopy(sumTileVal, tmpTile, 1);

                // Compute runningSumExp = runningSumExp * exp(runningMax - newM) + sumTile
                float sumTile, oldM = runningMax;
                {
                    AscendC::LocalTensor<float> tmpOne = tmpTileQueue_.AllocTensor<float>();
                    AscendC::DataCopy(tmpOne, sumTileVal, 1);
                    sumTile = ((volatile float*)(tmpOne.GetAddr()))[0];
                    tmpTileQueue_.FreeTensor(tmpOne);
                }

                float adjust = 0.f;
                {
                    // adjust = exp(oldM - newM)
                    AscendC::LocalTensor<float> adjIn = oneValQueue_.AllocTensor<float>();
                    AscendC::LocalTensor<float> adjOut = oneValQueue_.AllocTensor<float>();
                    float diff = oldM - newM;
                    AscendC::Duplicate(adjIn, diff, 1);
                    AscendC::Exp(adjOut, adjIn, 1);
                    // read back
                    AscendC::LocalTensor<float> tmpOne = tmpTileQueue_.AllocTensor<float>();
                    AscendC::DataCopy(tmpOne, adjOut, 1);
                    adjust = ((volatile float*)(tmpOne.GetAddr()))[0];
                    tmpTileQueue_.FreeTensor(tmpOne);
                    oneValQueue_.FreeTensor(adjIn);
                    oneValQueue_.FreeTensor(adjOut);
                }

                runningSumExp = runningSumExp * adjust + sumTile;
                runningMax = newM;

                oneValQueue_.FreeTensor(oneValA);
                oneValQueue_.FreeTensor(oneValB);
                yTileQueue_.FreeTensor(yTile);
                tmpTileQueue_.FreeTensor(tmpTile);
            }

            // lse = runningMax + log(runningSumExp)
            float lseVal = runningMax;
            {
                AscendC::LocalTensor<float> vIn = oneValQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> vOut = oneValQueue_.AllocTensor<float>();
                AscendC::Duplicate(vIn, runningSumExp, 1);
                AscendC::Log(vOut, vIn, 1);
                AscendC::LocalTensor<float> tmpOne = tmpTileQueue_.AllocTensor<float>();
                AscendC::DataCopy(tmpOne, vOut, 1);
                float logPart = ((volatile float*)(tmpOne.GetAddr()))[0];
                tmpTileQueue_.FreeTensor(tmpOne);
                oneValQueue_.FreeTensor(vIn);
                oneValQueue_.FreeTensor(vOut);
                lseVal += logPart;
            }

            // Two x LeakyReLU with slope 0.01, then two GELU
            // Leaky: y = max(x, 0) + slope * min(x, 0)
            auto leaky_once = [&](float v)->float {
                AscendC::LocalTensor<float> a = oneValQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> b = oneValQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> c = oneValQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> d = oneValQueue_.AllocTensor<float>();
                AscendC::Duplicate(a, v, 1);
                AscendC::Max(b, a, 0.0f, 1);    // b = max(v,0)
                AscendC::Min(c, a, 0.0f, 1);    // c = min(v,0)
                AscendC::Muls(c, c, NEG_SLOPE, 1); // c = slope * min(v,0)
                AscendC::Add(d, b, c, 1);
                AscendC::LocalTensor<float> tmpOne = tmpTileQueue_.AllocTensor<float>();
                AscendC::DataCopy(tmpOne, d, 1);
                float out = ((volatile float*)(tmpOne.GetAddr()))[0];
                tmpTileQueue_.FreeTensor(tmpOne);
                oneValQueue_.FreeTensor(a);
                oneValQueue_.FreeTensor(b);
                oneValQueue_.FreeTensor(c);
                oneValQueue_.FreeTensor(d);
                return out;
            };

            auto gelu_once = [&](float v)->float {
                // GELU(x) = 0.5 * x * (1 + erf(x / sqrt(2)))
                AscendC::LocalTensor<float> a = oneValQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> b = oneValQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> c = oneValQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> d = oneValQueue_.AllocTensor<float>();
                AscendC::Duplicate(a, v, 1);
                AscendC::Muls(b, a, INV_SQRT2, 1);
                AscendC::Erf(c, b, 1);
                AscendC::Adds(c, c, 1.0f, 1);
                AscendC::Mul(d, a, c, 1);
                AscendC::Muls(d, d, HALF, 1);
                AscendC::LocalTensor<float> tmpOne = tmpTileQueue_.AllocTensor<float>();
                AscendC::DataCopy(tmpOne, d, 1);
                float out = ((volatile float*)(tmpOne.GetAddr()))[0];
                tmpTileQueue_.FreeTensor(tmpOne);
                oneValQueue_.FreeTensor(a);
                oneValQueue_.FreeTensor(b);
                oneValQueue_.FreeTensor(c);
                oneValQueue_.FreeTensor(d);
                return out;
            };

            float v = lseVal;
            v = leaky_once(v);
            v = leaky_once(v);
            v = gelu_once(v);
            v = gelu_once(v);

            // Store to y[row]
            AscendC::LocalTensor<float> oneValOut = oneValQueue_.AllocTensor<float>();
            AscendC::Duplicate(oneValOut, v, 1);
            AscendC::DataCopy(yGm_[row], oneValOut, 1);
            oneValQueue_.FreeTensor(oneValOut);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> mulVecQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bTileQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yTileQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpTileQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> oneValQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, m_{0}, k_{0};
    uint32_t tileM_{0}, tileK_{0};
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)

    op.Process();
}
