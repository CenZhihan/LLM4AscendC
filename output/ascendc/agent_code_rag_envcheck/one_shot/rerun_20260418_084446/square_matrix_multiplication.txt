host_operator_src="""

#include "square_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 128 * 1024; // per-core conservative UB share
static constexpr uint32_t ALIGN_ELEMS = 1024;           // prefer large aligned bursts
static constexpr uint32_t HARD_CAP_ELEMS = 16384;       // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SquareMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const uint64_t a0 = a_shape->GetDim(0);
    const uint64_t a1 = a_shape->GetDim(1);
    const uint64_t b0 = b_shape->GetDim(0);
    const uint64_t b1 = b_shape->GetDim(1);
    if (!(a0 == a1 && b0 == b1 && a1 == b0)) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(a0);

    // UB model: we keep one accum tile (y) and one B-row tile in UB:
    // bytes = (2 * tileN) * sizeof(float) = 8 * tileN
    uint32_t maxTileN = UB_BUDGET_BYTES / 8;
    maxTileN = AlignDown(maxTileN, ALIGN_ELEMS);
    if (maxTileN < ALIGN_ELEMS) maxTileN = ALIGN_ELEMS;
    if (maxTileN > HARD_CAP_ELEMS) maxTileN = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileN = maxTileN;
    if (N < tileN) {
        const uint32_t aligned = AlignDown(N, ALIGN_ELEMS);
        tileN = (aligned > 0) ? aligned : N;
    }

    // Block mapping: each core handles a subset of rows with grid-stride
    uint32_t blockDim = CeilDiv(N, 1); // default try N cores then clamp
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *A_shape = context->GetInputShape(0);
    const gert::Shape *B_shape = context->GetInputShape(1);
    if (A_shape->GetDimNum() != 2 || B_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    if (A_shape->GetDim(1) != B_shape->GetDim(0)) {
        return GRAPH_FAILED;
    }
    gert::Shape *C_shape = context->GetOutputShape(0);
    C_shape->SetDimNum(2);
    C_shape->SetDim(0, A_shape->GetDim(0));
    C_shape->SetDim(1, B_shape->GetDim(1));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType A_dtype = context->GetInputDataType(0);
    const ge::DataType B_dtype = context->GetInputDataType(1);
    if (A_dtype != ge::DT_FLOAT || B_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class SquareMatrixMultiplicationCustom : public OpDef {
public:
    explicit SquareMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(SquareMatrixMultiplicationCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelSquareMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelSquareMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileN)
    {
        N_ = N;
        tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(N_) * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(N_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(N_) * N_);

        pipe_.InitBuffer(bRowQueue_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float)); // scalar buffer
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || tileN_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < N_; row += blockNum) {
            for (uint32_t nBase = 0; nBase < N_; nBase += tileN_) {
                const uint32_t lenN = AscendC::min(tileN_, N_ - nBase);

                AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, lenN);

                for (uint32_t k = 0; k < N_; ++k) {
                    // Load B row segment: B[k, nBase : nBase + lenN]
                    AscendC::LocalTensor<float> bLocal = bRowQueue_.AllocTensor<float>();
                    const uint64_t bOffset = static_cast<uint64_t>(k) * N_ + nBase;
                    AscendC::DataCopy(bLocal, bGm_[bOffset], lenN);

                    // Load scalar A[row, k] into tmp buffer
                    AscendC::LocalTensor<float> aScalarLocal = tmpQueue_.AllocTensor<float>();
                    const uint64_t aOffset = static_cast<uint64_t>(row) * N_ + k;
                    AscendC::DataCopy(aScalarLocal, aGm_[aOffset], 1);

                    // Read scalar and accumulate: y += a * b
                    float aVal = aScalarLocal.GetValue(0);

                    AscendC::Muls(bLocal, bLocal, aVal, lenN);
                    AscendC::Add(yLocal, yLocal, bLocal, lenN);

                    tmpQueue_.FreeTensor(aScalarLocal);
                    bRowQueue_.FreeTensor(bLocal);
                }

                // Store result C[row, nBase : nBase + lenN]
                const uint64_t cOffset = static_cast<uint64_t>(row) * N_ + nBase;
                AscendC::DataCopy(cGm_[cOffset], yLocal, lenN);
                yQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSquareMatrixMultiplicationCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.tileN);
    op.Process();
}
"""
