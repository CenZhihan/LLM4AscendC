host_operator_src="""
#include "gemm_sigmoid_scaling_residual_add_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core usable UB, conservative
static constexpr uint32_t ALIGN_ELEMS = 2048;            // align tile length for better DMA
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap per-iteration elements to control latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmSigmoidScalingResidualAddCustomTilingData tiling;

    // Input shapes:
    // x: [M, K]
    // w: [N, K] (linear expects weight transposed internally)
    // b: [N]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    if (x_shape->GetDimNum() != 2 || w_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 1) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t M = static_cast<uint64_t>(x_shape->GetDim(0));
    const uint64_t Kx = static_cast<uint64_t>(x_shape->GetDim(1));
    const uint64_t N = static_cast<uint64_t>(w_shape->GetDim(0));
    const uint64_t Kw = static_cast<uint64_t>(w_shape->GetDim(1));

    if (Kx != Kw || b_shape->GetDim(0) != (int64_t)N) {
        return ge::GRAPH_FAILED;
    }

    // Output y: [M, N]
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, static_cast<int64_t>(M));
    y_shape->SetDim(1, static_cast<int64_t>(N));

    const uint64_t totalLength64 = M * N;
    const uint32_t totalLength = static_cast<uint32_t>(totalLength64);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *scalingAttr = attrs->GetAttrPointer<float>(0);
    float scaling = (scalingAttr != nullptr) ? *scalingAttr : 1.0f;

    // UB model for post-processing (sigmoid and residual):
    // inQueue: 2 * tile
    // outQueue: 2 * tile
    // tmpQueue1: 1 * tile
    // tmpQueue2: 1 * tile
    // Total floats = 6 * tile => bytes = 24 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 24;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_scaling(scaling);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUFFER_NUM = 2;
constexpr int32_t OUT_BUFFER_NUM = 2;
constexpr int32_t TMP_BUFFER_NUM = 1;

class KernelGemmSigmoidScalingResidualAddCustom {
public:
    __aicore__ inline KernelGemmSigmoidScalingResidualAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength,
                                float scaling)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        scaling_ = scaling;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, IN_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, TMP_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, TMP_BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        // prefetch first tile
        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp1 = tmpQueue1_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmp2 = tmpQueue2_.AllocTensor<float>();

        // Compute sigmoid(x) = 1 / (1 + exp(-x))
        AscendC::Muls(tmp1, xLocal, -1.0f, len);      // tmp1 = -x
        AscendC::Exp(tmp1, tmp1, len);                // tmp1 = exp(-x)
        AscendC::Adds(tmp1, tmp1, 1.0f, len);         // tmp1 = exp(-x) + 1
        AscendC::Reciprocal(tmp2, tmp1, len);         // tmp2 = 1 / (exp(-x) + 1) = sigmoid(x)

        // Scale sigmoid and add residual: y = sigmoid(x) * scaling + x
        AscendC::Muls(tmp2, tmp2, scaling_, len);     // tmp2 = sigmoid(x) * s
        AscendC::Add(yLocal, tmp2, xLocal, len);      // y = tmp2 + x

        tmpQueue2_.FreeTensor(tmp2);
        tmpQueue1_.FreeTensor(tmp1);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUFFER_NUM> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUFFER_NUM> tmpQueue2_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    float scaling_{1.0f};
};

extern "C" __global__ __aicore__ void gemm_sigmoid_scaling_residual_add_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    // Note: This kernel performs post-processing on a precomputed linear output (x),
    // applying y = sigmoid(x) * scaling + x. Inputs w and b are not used here.
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSigmoidScalingResidualAddCustom op;
    op.Init(x, y, tiling_data.totalLength, tiling_data.tileLength, tiling_data.scaling);
    op.Process();
}
