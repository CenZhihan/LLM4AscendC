#include "max_pooling_1d_custom_tiling.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

// UB memory budget per core to be conservative
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// Favor burst-friendly tile alignment on output length
static constexpr uint32_t ALIGN_OUT = 128;
// Avoid too large per-iter latency
static constexpr uint32_t HARD_CAP_OUT = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling1dCustomTilingData tiling;

    const gert::Shape &shape = context->GetInputShape(0)->GetOriginShape();
    const uint32_t dimNum = static_cast<uint32_t>(shape.GetDimNum());
    if (dimNum != 3) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(shape.GetDim(0));
    const uint32_t C = static_cast<uint32_t>(shape.GetDim(1));
    const uint32_t L = static_cast<uint32_t>(shape.GetDim(2));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernelSizeAttr = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *strideAttr     = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *paddingAttr    = attrs->GetAttrPointer<int64_t>(2);
    // return_indices ignored on device side for this kernel

    uint32_t ksize  = static_cast<uint32_t>(*kernelSizeAttr);
    uint32_t stride = static_cast<uint32_t>(*strideAttr);
    uint32_t pad    = static_cast<uint32_t>(*paddingAttr);

    if (ksize == 0) { return ge::GRAPH_FAILED; }
    if (stride == 0) { stride = ksize; } // PyTorch default behavior

    const int64_t Lpad = static_cast<int64_t>(L) + 2LL * static_cast<int64_t>(pad);
    if (Lpad < static_cast<int64_t>(ksize)) {
        // still produce one output position as PyTorch does when the window can be placed once
        // but for consistency with standard formula, floor((L+2p - k)/s)+1 can be zero; clamp to at least 1
    }
    uint32_t outL = 0;
    if (Lpad >= static_cast<int64_t>(ksize)) {
        outL = static_cast<uint32_t>(((Lpad - static_cast<int64_t>(ksize)) / static_cast<int64_t>(stride)) + 1LL);
    } else {
        outL = 1u;
    }

    // Determine tileOut respecting UB budget
    // UB usage floats per tile roughly = localInLen + outLen
    // where localInLen = (tileOut - 1)*stride + ksize
    // So floats = (stride + 1) * tileOut + (ksize - stride)
    const uint32_t ubFloats = UB_BUDGET_BYTES / 4;
    uint32_t maxTileOut = 0;
    if ((stride + 1u) > 0) {
        const int64_t numer = static_cast<int64_t>(ubFloats) - static_cast<int64_t>(ksize > stride ? (ksize - stride) : 0u);
        if (numer > 0) {
            maxTileOut = static_cast<uint32_t>(numer / static_cast<int64_t>(stride + 1u));
        } else {
            maxTileOut = ALIGN_OUT;
        }
    } else {
        maxTileOut = ALIGN_OUT;
    }
    if (maxTileOut > HARD_CAP_OUT) maxTileOut = AlignDown(HARD_CAP_OUT, ALIGN_OUT);
    if (maxTileOut < ALIGN_OUT) maxTileOut = ALIGN_OUT;

    uint32_t tileOut = maxTileOut;
    if (outL < tileOut) {
        const uint32_t aligned = AlignDown(outL, ALIGN_OUT);
        tileOut = (aligned > 0) ? aligned : outL;
    }

    const uint32_t MC = N * C;
    uint32_t blockDim = (MC < 24u) ? MC : 24u;
    if (blockDim == 0u) blockDim = 1u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_lIn(L);
    tiling.set_lOut(outL);
    tiling.set_kernelSize(ksize);
    tiling.set_stride(stride);
    tiling.set_padding(pad);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *kernelSizeAttr = attrs->GetAttrPointer<int64_t>(0);
    const int64_t *strideAttr     = attrs->GetAttrPointer<int64_t>(1);
    const int64_t *paddingAttr    = attrs->GetAttrPointer<int64_t>(2);

    int64_t ksize  = *kernelSizeAttr;
    int64_t stride = *strideAttr;
    int64_t pad    = *paddingAttr;
    if (stride == 0) stride = ksize;
    const int64_t L  = x_shape->GetOriginShape().GetDim(2);
    const int64_t Lpad = L + 2 * pad;

    int64_t outL = 1;
    if (Lpad >= ksize) {
        outL = (Lpad - ksize) / stride + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    y_shape->GetStorageShape().SetDim(2, outL);
    y_shape->GetOriginShape().SetDim(2, outL);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t INBUF_NUM = 1;
constexpr int32_t OUTBUF_NUM = 1;

class KernelMaxPooling1dCustom {
public:
    __aicore__ inline KernelMaxPooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t lIn, uint32_t lOut,
                                uint32_t kernelSize, uint32_t stride,
                                uint32_t padding, uint32_t tileOut)
    {
        n_ = n; c_ = c; lIn_ = lIn; lOut_ = lOut;
        ksize_ = kernelSize; stride_ = stride; pad_ = padding;
        tileOut_ = tileOut;

        const uint32_t totalInElems  = n_ * c_ * lIn_;
        const uint32_t totalOutElems = n_ * c_ * lOut_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInElems);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutElems);

        // Max input slice length required per tile
        tileInMaxLen_ = tileOut_ * stride_ + ksize_ - ((stride_ > 0u) ? stride_ : 1u);
        if (tileInMaxLen_ < ksize_) tileInMaxLen_ = ksize_;

        pipe_.InitBuffer(inQueue_, INBUF_NUM, tileInMaxLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUTBUF_NUM, tileOut_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || lIn_ == 0 || lOut_ == 0) return;

        const uint32_t mc = n_ * c_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t ch = blockIdx; ch < mc; ch += blockNum) {
            const uint32_t xChBase = ch * lIn_;
            const uint32_t yChBase = ch * lOut_;

            for (uint32_t outBase = 0; outBase < lOut_; outBase += tileOut_) {
                uint32_t curOut = lOut_ - outBase;
                if (curOut > tileOut_) curOut = tileOut_;

                // Compute required input span for this tile
                const int64_t start0 = static_cast<int64_t>(outBase) * static_cast<int64_t>(stride_) - static_cast<int64_t>(pad_);
                const uint32_t localInLen = (stride_ == 0u) ? (curOut + ksize_ - 1u) : ((curOut - 1u) * stride_ + ksize_);
                const int64_t needStart = start0;
                const int64_t needEnd   = start0 + static_cast<int64_t>(localInLen); // exclusive

                const int64_t inStart = needStart < 0 ? 0 : needStart;
                const int64_t inEnd   = needEnd > static_cast<int64_t>(lIn_) ? static_cast<int64_t>(lIn_) : needEnd;
                const uint32_t validLen = (inEnd > inStart) ? static_cast<uint32_t>(inEnd - inStart) : 0u;
                const uint32_t leftPad  = (inStart > needStart) ? static_cast<uint32_t>(inStart - needStart) : 0u;

                AscendC::LocalTensor<float> inLocal = inQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> yLocal  = outQueue_.AllocTensor<float>();

                // Zero-pad local input buffer over the required localInLen
                AscendC::Duplicate(inLocal, 0.0f, localInLen);

                // Copy valid slice into padded buffer at offset leftPad
                if (validLen > 0) {
                    const uint32_t gmOff = xChBase + static_cast<uint32_t>(inStart);
                    AscendC::DataCopy(inLocal[leftPad], xGm_[gmOff], validLen);
                }

                // Compute max pooling
                // First, initialize yLocal with a very negative number
                const float NEG_INF = -3.402823e38f;
                AscendC::Duplicate(yLocal, NEG_INF, curOut);

                // Sliding-window max via elementwise max over shifted views
                // Windows are contiguous on local buffer when stride == 1; for stride > 1, we still
                // compute full-by-1 intermediate and only write the first curOut elements due to localInLen definition
                // Here we compute exactly curOut positions.
                for (uint32_t k = 0; k < ksize_; ++k) {
                    AscendC::Max(yLocal, yLocal, inLocal[k], curOut);
                }

                // Write back result
                AscendC::DataCopy(yGm_[yChBase + outBase], yLocal, curOut);

                outQueue_.FreeTensor(yLocal);
                inQueue_.FreeTensor(inLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INBUF_NUM>  inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUTBUF_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, lIn_{0}, lOut_{0};
    uint32_t ksize_{1}, stride_{1}, pad_{0};
    uint32_t tileOut_{0}, tileInMaxLen_{0};
};

extern "C" __global__ __aicore__ void max_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling1dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.lIn, tiling_data.lOut,
            tiling_data.kernelSize, tiling_data.stride,
            tiling_data.padding, tiling_data.tileOut);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t calc_out_len_1d(int64_t L, int64_t k, int64_t s, int64_t p) {
    if (s <= 0) s = k;
    int64_t Lpad = L + 2 * p;
    if (Lpad < k) return 1;
    return (Lpad - k) / s + 1;
}

at::Tensor max_pooling_1d_custom_impl_npu(const at::Tensor& self, int64_t kernel_size, c10::optional<int64_t> stride_opt, int64_t padding, bool return_indices) {
    int64_t stride = stride_opt.has_value() ? stride_opt.value() : kernel_size;
    auto sizes = self.sizes();
    TORCH_CHECK(sizes.size() == 3, "Input must be 3D (N, C, L)");
    const int64_t N = sizes[0];
    const int64_t C = sizes[1];
    const int64_t L = sizes[2];
    const int64_t outL = calc_out_len_1d(L, kernel_size, stride, padding);
    at::Tensor result = at::empty({N, C, outL}, self.options());
    // return_indices is ignored in this custom kernel
    EXEC_NPU_CMD(aclnnMaxPooling1dCustom, self, kernel_size, stride, padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_1d_custom", &max_pooling_1d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_1d_custom", &max_pooling_1d_custom_impl_npu, "MaxPooling1D (MaxPooling1dCustom on NPU)");

class ModelNew(torch.nn.Module):
