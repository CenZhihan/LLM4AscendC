namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) { return static_cast<uint32_t>((a + b - 1ULL) / b); }
} // namespace

namespace optiling {

// UB model for this kernel:
//   - Weight buffer: Cin * 3 * 3 floats (per oc task)
//   - xVec: 1 * tileW
//   - tmpVec: 1 * tileW
//   - yVec: 1 * tileW
// Total bytes = 4 * (Cin*9 + 3*tileW)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;              // favor DMA efficiency
static constexpr uint32_t MIN_TILEW = 64;                 // minimal reasonable tile width
static constexpr uint32_t MAX_BLOCKDIM = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dAsymmetricInputSquareKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetDimNum() != 4 || w_shape->GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N    = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H    = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W    = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t KCin = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t KH   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t KW   = static_cast<uint32_t>(w_shape->GetDim(3));

    // Validate conv config: stride=1, pad=0, dilation=1, groups=1, KH=KW=3, KCin == Cin
    if (KH != 3 || KW != 3 || KCin != Cin) {
        return ge::GRAPH_FAILED;
    }
    if (H < 3 || W < 3) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Hout = H - 2;
    const uint32_t Wout = W - 2;

    // Compute tileW from UB budget: 4 * (Cin*9 + 3*tileW) <= UB_BUDGET_BYTES
    uint64_t fixed_weight_bytes = static_cast<uint64_t>(Cin) * 9ULL * 4ULL;
    uint64_t remain = (UB_BUDGET_BYTES > fixed_weight_bytes) ? (UB_BUDGET_BYTES - fixed_weight_bytes) : 0ULL;
    uint32_t tileW = 0;
    if (remain >= 12ULL) {
        tileW = static_cast<uint32_t>(remain / 12ULL);
    }
    if (tileW == 0) tileW = MIN_TILEW;
    tileW = AlignDown(tileW, ALIGN_ELEMS);
    if (tileW < MIN_TILEW) tileW = MIN_TILEW;
    if (tileW > Wout) {
        // Use smaller alignment if Wout is small
        tileW = (Wout >= ALIGN_ELEMS) ? AlignDown(Wout, ALIGN_ELEMS) : Wout;
        if (tileW == 0 && Wout > 0) tileW = Wout;
    }

    // BlockDim selection: use min(max, number of tasks)
    const uint64_t totalTasks64 = static_cast<uint64_t>(N) * static_cast<uint64_t>(Cout) * static_cast<uint64_t>(Hout);
    uint32_t blockDim = static_cast<uint32_t>(totalTasks64);
    if (blockDim > MAX_BLOCKDIM) blockDim = MAX_BLOCKDIM;
    if (blockDim == 0) blockDim = 1;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Hout(Hout);
    tiling.set_Wout(Wout);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvStandard2dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W,
                                uint32_t Hout, uint32_t Wout,
                                uint32_t tileW)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout; H_ = H; W_ = W;
        Hout_ = Hout; Wout_ = Wout; tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cout_ * Cin_ * 3 * 3);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Cout_ * Hout_ * Wout_);

        // Buffers
        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileW_ * sizeof(float));
        // Weight buffer per task: Cin * 3 * 3
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, Cin_ * 9 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || Hout_ == 0 || Wout_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalTasks = (uint64_t)N_ * (uint64_t)Cout_ * (uint64_t)Hout_;
        for (uint64_t task = blockIdx; task < totalTasks; task += blockNum) {
            // Decode (n, oc, h)
            uint64_t t = task;
            const uint32_t h = static_cast<uint32_t>(t % Hout_); t /= Hout_;
            const uint32_t oc = static_cast<uint32_t>(t % Cout_); t /= Cout_;
            const uint32_t n = static_cast<uint32_t>(t % N_);

            // Preload weights for this oc: Cin * 3 * 3 continuous
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            const uint64_t wOffset = ((uint64_t)oc) * (uint64_t)Cin_ * 9ULL;
            AscendC::DataCopy(wLocal, wGm_[wOffset], Cin_ * 9);
            wQueue_.EnQue(wLocal);

            // Iterate width by tiles
            for (uint32_t w0 = 0; w0 < Wout_; w0 += tileW_) {
                uint32_t curW = Wout_ - w0;
                if (curW > tileW_) curW = tileW_;

                AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, curW);

                // Accumulate across Cin and 3x3 kernel
                AscendC::LocalTensor<float> wLocalUse = wQueue_.DeQue<float>();

                for (uint32_t c = 0; c < Cin_; ++c) {
                    // index base in weight local for this input channel
                    const uint32_t wBase = c * 9;

                    // ky, kx loops
                    for (uint32_t ky = 0; ky < 3; ++ky) {
                        for (uint32_t kx = 0; kx < 3; ++kx) {
                            const uint32_t wIdx = wBase + ky * 3 + kx;
                            const float wVal = wLocalUse.GetValue(wIdx);

                            // Load x vector for this (n, c, h+ky, w0+kx : w0+kx+curW)
                            AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                            const uint64_t xOffset =
                                (((uint64_t)n * Cin_ + c) * H_ + (uint64_t)(h + ky)) * W_ + (uint64_t)(w0 + kx);
                            AscendC::DataCopy(xLocal, xGm_[xOffset], curW);

                            // tmp = x * wVal
                            AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                            AscendC::Muls(tmpLocal, xLocal, wVal, curW);
                            // y += tmp
                            AscendC::Add(yLocal, yLocal, tmpLocal, curW);

                            tmpQueue_.FreeTensor(tmpLocal);
                            xQueue_.FreeTensor(xLocal);
                        }
                    }
                }

                // Store y tile
                const uint64_t yOffset =
                    (((uint64_t)n * Cout_ + oc) * Hout_ + (uint64_t)h) * Wout_ + (uint64_t)w0;
                AscendC::DataCopy(yGm_[yOffset], yLocal, curW);
                yQueue_.FreeTensor(yLocal);

                wQueue_.EnQue(wLocalUse); // push back for reuse in next tile
            }

            // Done with this oc's weights
            AscendC::LocalTensor<float> wLocalDone = wQueue_.DeQue<float>();
            wQueue_.FreeTensor(wLocalDone);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0}, H_{0}, W_{0}, Hout_{0}, Wout_{0}, tileW_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dAsymmetricInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W,
            tiling_data.Hout, tiling_data.Wout,
            tiling_data.tileW);
    op.Process();
}
