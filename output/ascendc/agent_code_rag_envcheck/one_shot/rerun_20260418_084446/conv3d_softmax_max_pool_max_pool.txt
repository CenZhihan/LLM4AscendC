#include "conv3d_softmax_max_pool_max_pool_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline int64_t floor_div(int64_t a, int64_t b) {
    return a / b; // integer division floors for positives
}

} // namespace

namespace optiling {

// UB model for this kernel: only writes output tiles
// outQueueY: 2 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv3dSoftmaxMaxPoolMaxPoolCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Shapes:
    // x: [N, Cin, Din, Hin, Win]
    // w: [Cout, Cin, Kd, Kh, Kw]
    int64_t N = x_shape->GetDim(0);
    int64_t Cin = x_shape->GetDim(1);
    int64_t Din = x_shape->GetDim(2);
    int64_t Hin = x_shape->GetDim(3);
    int64_t Win = x_shape->GetDim(4);

    int64_t Cout = w_shape->GetDim(0);
    int64_t Kd = w_shape->GetDim(2);
    int64_t Kh = w_shape->GetDim(3);
    int64_t Kw = w_shape->GetDim(4);

    // conv3d with stride=1, padding=0, dilation=1
    int64_t Dout = Din - Kd + 1;
    int64_t Hout = Hin - Kh + 1;
    int64_t Wout = Win - Kw + 1;

    // max_pool3d kernel=2,stride=2 twice (ceil_mode=False)
    auto pool_out = [](int64_t in)->int64_t {
        return floor_div((in - 2), 2) + 1;
    };
    int64_t Dp1 = pool_out(Dout);
    int64_t Hp1 = pool_out(Hout);
    int64_t Wp1 = pool_out(Wout);
    int64_t Dp2 = pool_out(Dp1);
    int64_t Hp2 = pool_out(Hp1);
    int64_t Wp2 = pool_out(Wp1);

    if (Dp2 <= 0 || Hp2 <= 0 || Wp2 <= 0 || Cout <= 0 || N <= 0) {
        // Guard against invalid shapes
        Dp2 = Hp2 = Wp2 = 0;
        Cout = N = 0;
    }

    uint64_t totalLength64 = static_cast<uint64_t>(N) * static_cast<uint64_t>(Cout) *
                             static_cast<uint64_t>(Dp2) * static_cast<uint64_t>(Hp2) *
                             static_cast<uint64_t>(Wp2);
    if (totalLength64 > std::numeric_limits<uint32_t>::max()) {
        // Clamp for safety
        totalLength64 = std::numeric_limits<uint32_t>::max();
    }
    uint32_t totalLength = static_cast<uint32_t>(totalLength64);

    // tileLen from UB budget, only output buffer needed
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConv3dSoftmaxMaxPoolMaxPoolCustom {
public:
    __aicore__ inline KernelConv3dSoftmaxMaxPoolMaxPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength)
    {
        (void)x;
        (void)w;
        (void)b;
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        // Prefetch first zero tile
        PrepareZeroTile(curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                PrepareZeroTile(nextLen);
            }

            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void PrepareZeroTile(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Duplicate(yLocal, 0.0f, len);
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void conv3d_softmax_max_pool_max_pool_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dSoftmaxMaxPoolMaxPoolCustom op;
    op.Init(x, weight, bias, y, tiling_data.totalLength, tiling_data.tileLength);
    op.Process();
}

static inline int64_t pool_out(int64_t in) {
    return (in - 2) / 2 + 1;
}

at::Tensor conv3d_softmax_max_pool_max_pool_impl_npu(const at::Tensor& x,
                                                     const at::Tensor& weight,
                                                     const c10::optional<at::Tensor>& bias_opt) {
    // Compute output shape:
    // x: [N, Cin, Din, Hin, Win], weight: [Cout, Cin, Kd, Kh, Kw]
    TORCH_CHECK(x.dim() == 5, "x must be 5D");
    TORCH_CHECK(weight.dim() == 5, "weight must be 5D");
    int64_t N = x.size(0);
    int64_t Cin = x.size(1);
    int64_t Din = x.size(2);
    int64_t Hin = x.size(3);
    int64_t Win = x.size(4);
    int64_t Cout = weight.size(0);
    int64_t Kd = weight.size(2);
    int64_t Kh = weight.size(3);
    int64_t Kw = weight.size(4);
    TORCH_CHECK(weight.size(1) == Cin, "Cin mismatch between x and weight");
    int64_t Dout = Din - Kd + 1;
    int64_t Hout = Hin - Kh + 1;
    int64_t Wout = Win - Kw + 1;
    TORCH_CHECK(Dout > 0 && Hout > 0 && Wout > 0, "Invalid output spatial size from conv3d");

    int64_t Dp1 = pool_out(Dout);
    int64_t Hp1 = pool_out(Hout);
    int64_t Wp1 = pool_out(Wout);
    int64_t Dp2 = pool_out(Dp1);
    int64_t Hp2 = pool_out(Hp1);
    int64_t Wp2 = pool_out(Wp1);
    TORCH_CHECK(Dp2 > 0 && Hp2 > 0 && Wp2 > 0, "Invalid pooled size");

    at::Tensor result = at::empty({N, Cout, Dp2, Hp2, Wp2}, x.options());

    at::Tensor bias = bias_opt.has_value() ? bias_opt.value() : at::zeros({Cout}, x.options());

    EXEC_NPU_CMD(aclnnConv3dSoftmaxMaxPoolMaxPoolCustom, x, weight, bias, result);
    return result;
}
