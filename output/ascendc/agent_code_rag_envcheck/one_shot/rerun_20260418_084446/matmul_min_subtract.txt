#include "matmul_min_subtract_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint64_t a, uint64_t b) {
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMinSubtractCustomTilingData tiling;

    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const auto &x_os = x_shape->GetOriginShape();
    const auto &w_os = w_shape->GetOriginShape();
    const int64_t M = x_os.GetDim(0);
    const int64_t K = x_os.GetDim(1);
    const int64_t N = w_os.GetDim(0);
    // w shape should be [N, K]
    if (w_os.GetDim(1) != K || M <= 0 || N <= 0 || K <= 0) {
        return ge::GRAPH_FAILED;
    }

    // Conservative tiling to reduce UB usage and branch complexity
    const uint32_t mTile = 16;   // rows per inner tile
    const uint32_t nTile = 32;   // columns per block tile
    const uint32_t kStep = 128;  // K slice per iterate

    // Blocks cover N dimension tiles
    const uint32_t nTiles = CeilDiv(static_cast<uint64_t>(N), nTile);
    uint32_t blockDim = nTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(static_cast<uint32_t>(M));
    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_K(static_cast<uint32_t>(K));
    tiling.set_mTile(mTile);
    tiling.set_nTile(nTile);
    tiling.set_kStep(kStep);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const auto &x_os = x_shape->GetOriginShape();
    const auto &w_os = w_shape->GetOriginShape();
    if (x_os.GetDim(1) != w_os.GetDim(1)) {
        return GRAPH_FAILED;
    }
    const int64_t M = x_os.GetDim(0);
    const int64_t N = w_os.GetDim(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, M);
    y_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

{
    const ge::DataType dtype = context->GetInputDataType(0);
    // All inputs/outputs must be same float type
    context->SetOutputDataType(0, dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelMatmulMinSubtractCustom {
public:
    __aicore__ inline KernelMatmulMinSubtractCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR c, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t mTile, uint32_t nTile, uint32_t kStep)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        mTile_ = mTile;
        nTile_ = nTile;
        kStep_ = kStep;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * N_);

        // UB buffers
        pipe_.InitBuffer(xRowBufQ_, 1, kStep_ * sizeof(float));
        pipe_.InitBuffer(wRowBufQ_, 1, kStep_ * sizeof(float));
        pipe_.InitBuffer(rowVecQ_, 1, nTile_ * sizeof(float));
        // Load scalar cap (c)
        AscendC::LocalTensor<float> cLocal = rowVecQ_.AllocTensor<float>(); // reuse small alloc
        AscendC::DataCopy(cLocal, cGm_[0], 1);
        cap_ = cLocal.GetValue(0); // assume API exists for scalar read
        rowVecQ_.FreeTensor(cLocal);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint32_t nTileStart = blockIdx;
        const uint32_t nTileStride = blockNum;

        for (uint32_t nTileIdx = nTileStart; nTileIdx * nTile_ < N_; nTileIdx += nTileStride) {
            const uint32_t nBase = nTileIdx * nTile_;
            const uint32_t nLen = AscendC::min(nTile_, N_ - nBase);

            for (uint32_t mBase = 0; mBase < M_; mBase += mTile_) {
                const uint32_t mLen = AscendC::min(mTile_, M_ - mBase);

                // process each row in this m-tile
                for (uint32_t mi = 0; mi < mLen; ++mi) {
                    AscendC::LocalTensor<float> rowVec = rowVecQ_.AllocTensor<float>();
                    // zero-initialize rowVec
                    for (uint32_t j = 0; j < nLen; ++j) {
                        rowVec.SetValue(j, 0.0f);
                    }

                    // compute rowVec[j] = dot(xRow, wRow_j) + b_j
                    for (uint32_t j = 0; j < nLen; ++j) {
                        float acc = 0.0f;
                        for (uint32_t kBase = 0; kBase < K_; kBase += kStep_) {
                            const uint32_t kCur = AscendC::min(kStep_, K_ - kBase);
                            AscendC::LocalTensor<float> xRow = xRowBufQ_.AllocTensor<float>();
                            AscendC::LocalTensor<float> wRow = wRowBufQ_.AllocTensor<float>();

                            // copy x row segment
                            const uint64_t xOff = ((uint64_t)(mBase + mi) * K_) + kBase;
                            AscendC::DataCopy(xRow, xGm_[xOff], kCur);
                            // copy w row segment (w row = column of weight matrix in PyTorch? Here w is [N, K], we need row j)
                            const uint64_t wOff = ((uint64_t)(nBase + j) * K_) + kBase;
                            AscendC::DataCopy(wRow, wGm_[wOff], kCur);

                            // accumulate
                            for (uint32_t t = 0; t < kCur; ++t) {
                                const float xv = xRow.GetValue(t);
                                const float wv = wRow.GetValue(t);
                                acc += xv * wv;
                            }

                            wRowBufQ_.FreeTensor(wRow);
                            xRowBufQ_.FreeTensor(xRow);
                        }

                        // add bias
                        const float bval = bGm_.GetValue(nBase + j); // assume GM scalar read
                        acc += bval;

                        // cap and subtract
                        if (acc > cap_) acc = cap_;
                        acc -= cap_;
                        rowVec.SetValue(j, acc);
                    }

                    // store rowVec to y
                    const uint64_t yOff = ((uint64_t)(mBase + mi) * N_) + nBase;
                    AscendC::DataCopy(yGm_[yOff], rowVec, nLen);

                    rowVecQ_.FreeTensor(rowVec);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  xRowBufQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  wRowBufQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> rowVecQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t mTile_{0}, nTile_{0}, kStep_{0};
    float cap_{0.f};
};

extern "C" __global__ __aicore__ void matmul_min_subtract_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR c, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMinSubtractCustom op;
    op.Init(x, w, b, c, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.mTile, tiling_data.nTile, tiling_data.kStep);
    op.Process();
}
