host_operator_src="""

#include "register/op_def_registry.h"

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = a_shape->GetDim(0);
    const int64_t K1 = a_shape->GetDim(1);
    const int64_t K2 = b_shape->GetDim(0);
    const int64_t N = b_shape->GetDim(1);
    if (K1 != K2) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Inputs in float16, output in float32 accumulate by default.
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class StandardMatrixMultiplicationCustom : public OpDef {
public:
    explicit StandardMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        // Use default cube tiling for AscendC matmul
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(StandardMatrixMultiplicationCustom);
} // namespace ops
"""

kernel_src="""

#include "kernel_operator.h"

extern "C" __global__ __aicore__ void standard_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace,
                                                                            GM_ADDR tilingGm)
{
    if (g_coreType == AscendC::AIV) {
        return;
    }
    // Prepare matmul cube tiling
    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    // Define Matmul types: A and B in GM as FP16, C in GM as FP32 (accumulate in FP32)
    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> mmKernel;
    AscendC::TPipe pipe;
    // Register the Matmul object with system workspace and tiling
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mmKernel.matmulObj, &tiling);

    // Initialize without bias
    mmKernel.Init(a, b, nullptr, c, workspace, tiling);

    // Process with default loop unrolling knobs
    mmKernel.Process<false>(&pipe, 3, 3);
}
"""

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor standard_matrix_multiplication_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Convert inputs to FP16 for cube matmul; output is FP32
    at::Tensor A_half = A.to(at::kHalf);
    at::Tensor B_half = B.to(at::kHalf);
    TORCH_CHECK(A_half.dim() == 2 && B_half.dim() == 2, "Inputs must be 2D matrices.");
    TORCH_CHECK(A_half.size(1) == B_half.size(0), "Incompatible matrix dimensions.");

    at::Tensor C = at::empty({A_half.size(0), B_half.size(1)}, A.options().dtype(at::kFloat));
    EXEC_NPU_CMD(aclnnStandardMatrixMultiplicationCustom, A_half, B_half, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("standard_matrix_multiplication_custom", &standard_matrix_multiplication_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("standard_matrix_multiplication_custom", &standard_matrix_multiplication_impl_npu,
          "Standard matrix multiplication using AscendC cube (StandardMatrixMultiplicationCustom on NPU)");
}
"""
