#include "matmul_scaling_residual_add_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core
static constexpr uint32_t ALIGN_ELEMS = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Shapes:
    // x: [m, k]
    // w: [n, k]
    // b: [n]
    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);
    const auto *b_shape = context->GetInputShape(2);

    if (x_shape->GetOriginShape().GetDimNum() < 2 || w_shape->GetOriginShape().GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t m64 = x_shape->GetOriginShape().GetDim(0);
    const uint64_t kx64 = x_shape->GetOriginShape().GetDim(1);
    const uint64_t n64 = w_shape->GetOriginShape().GetDim(0);
    const uint64_t kw64 = w_shape->GetOriginShape().GetDim(1);
    if (kx64 != kw64) return ge::GRAPH_FAILED;

    const uint32_t m = static_cast<uint32_t>(m64);
    const uint32_t k = static_cast<uint32_t>(kx64);
    const uint32_t n = static_cast<uint32_t>(n64);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    float scaling = 0.0f;
    if (attrs && attrs->GetAttrNum() > 0) {
        const float *s = attrs->GetAttrPointer<float>(0);
        if (s) scaling = *s;
    }

    // Choose conservative tiles that fit UB:
    // UB floats capacity
    const uint32_t ub_floats = UB_BUDGET_BYTES / 4;
    // Default tiles
    uint32_t tileM = 8;
    uint32_t tileN = 128;
    uint32_t tileK = 128;

    // Try to keep xTile + wTile + yTile within UB
    // cost = tileM*tileK + tileN*tileK + tileM*tileN
    auto Fits = [&](uint32_t tm, uint32_t tn, uint32_t tk) -> bool {
        uint64_t cost = static_cast<uint64_t>(tm) * tk +
                        static_cast<uint64_t>(tn) * tk +
                        static_cast<uint64_t>(tm) * tn;
        return cost + 512 < ub_floats; // margin
    };

    // Adjust tiles to problem constraints
    tileM = (m < tileM) ? m : tileM;
    tileN = (n < tileN) ? n : tileN;
    tileK = (k < tileK) ? k : tileK;

    tileM = (tileM == 0) ? 1u : AlignDown(tileM, 1);
    tileN = (tileN == 0) ? 1u : AlignDown(tileN, ALIGN_ELEMS);
    tileK = (tileK == 0) ? 1u : AlignDown(tileK, ALIGN_ELEMS);

    while (!Fits(tileM, tileN, tileK) && tileK > ALIGN_ELEMS) {
        tileK = AlignDown(tileK / 2, ALIGN_ELEMS);
    }
    while (!Fits(tileM, tileN, tileK) && tileN > ALIGN_ELEMS) {
        tileN = AlignDown(tileN / 2, ALIGN_ELEMS);
    }
    while (!Fits(tileM, tileN, tileK) && tileM > 1) {
        tileM = AlignDown(tileM / 2, 1);
    }
    if (!Fits(tileM, tileN, tileK)) {
        // Fallback to minimal tiles
        tileM = (m > 0) ? 1u : 0u;
        tileN = (n > 0) ? AlignDown((n >= ALIGN_ELEMS) ? ALIGN_ELEMS : n, 1) : 0u;
        tileK = (k > 0) ? AlignDown((k >= ALIGN_ELEMS) ? ALIGN_ELEMS : k, 1) : 0u;
    }

    // Parallelize across M tiles
    const uint32_t numMTiles = (tileM == 0) ? 1u : CeilDiv(m, tileM);
    uint32_t blockDim = numMTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    MatmulScalingResidualAddCustomTilingData tiling;
    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);
    tiling.set_scaling(scaling);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulScalingResidualAddCustom {
public:
    __aicore__ inline KernelMatmulScalingResidualAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                float scaling)
    {
        m_ = m; n_ = n; k_ = k;
        tileM_ = (tileM == 0 ? 1u : tileM);
        tileN_ = (tileN == 0 ? 1u : tileN);
        tileK_ = (tileK == 0 ? 1u : tileK);
        scaleCorr_ = 1.0f + scaling;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(m_) * k_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(n_) * k_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, n_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(m_) * n_);

        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, tileN_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileN_ * sizeof(float)); // for bias and temp vectors
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || n_ == 0 || k_ == 0) { return; }

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        // Grid-stride across M tiles
        for (uint32_t mBase = blockIdx * tileM_; mBase < m_; mBase += blockNum * tileM_) {
            const uint32_t mCount = min_u32(tileM_, m_ - mBase);
            // Loop over N tiles serially within a block
            for (uint32_t nBase = 0; nBase < n_; nBase += tileN_) {
                const uint32_t nCount = min_u32(tileN_, n_ - nBase);

                // Allocate Y tile and zero initialize
                LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                Dup(yLocal, 0.0f, mCount * nCount);

                // Iterate over K dimension in chunks
                for (uint32_t kBase = 0; kBase < k_; kBase += tileK_) {
                    const uint32_t kCount = min_u32(tileK_, k_ - kBase);

                    // Load X tile [mCount, kCount] row by row into xLocal
                    LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                    for (uint32_t i = 0; i < mCount; ++i) {
                        DataCopy(xLocal[i * kCount], xGm_[(static_cast<uint64_t>(mBase + i) * k_) + kBase], kCount);
                    }
                    // Load W tile [nCount, kCount] row by row into wLocal
                    LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                    for (uint32_t j = 0; j < nCount; ++j) {
                        DataCopy(wLocal[j * kCount], wGm_[(static_cast<uint64_t>(nBase + j) * k_) + kBase], kCount);
                    }

                    // Accumulate yLocal += X[:, :] @ (W[:, :])^T
                    // y[i, j] += sum_{t=0..kCount-1} x[i, t] * w[j, t]
                    for (uint32_t i = 0; i < mCount; ++i) {
                        // Row pointer into output tile
                        float* yRowPtr = reinterpret_cast<float*>(yLocal.GetAddr()) + static_cast<uint64_t>(i) * nCount;
                        for (uint32_t t = 0; t < kCount; ++t) {
                            // Prepare temp = w[:, t] (length nCount)
                            LocalTensor<float> temp = tmpQueue_.AllocTensor<float>();
                            float* tempPtr = reinterpret_cast<float*>(temp.GetAddr());
                            const float xi = xLocal.GetValue(i * kCount + t);
                            for (uint32_t j = 0; j < nCount; ++j) {
                                tempPtr[j] = wLocal.GetValue(j * kCount + t);
                            }
                            // temp *= xi
                            Muls(temp, temp, xi, nCount);
                            // yRow += temp
                            // Implement yRow += temp via vector Add
                            LocalTensor<float> yRowVec(reinterpret_cast<__ubuf__ float*>(yRowPtr), nCount);
                            Add(yRowVec, yRowVec, temp, nCount);
                            tmpQueue_.FreeTensor(temp);
                        }
                    }

                    wQueue_.FreeTensor(wLocal);
                    xQueue_.FreeTensor(xLocal);
                }

                // Add bias and apply (1 + scaling)
                LocalTensor<float> bias = tmpQueue_.AllocTensor<float>();
                DataCopy(bias, bGm_[nBase], nCount);
                for (uint32_t i = 0; i < mCount; ++i) {
                    LocalTensor<float> yRow = LocalTensor<float>(reinterpret_cast<__ubuf__ float*>(yLocal.GetAddr()) + static_cast<uint64_t>(i) * nCount, nCount);
                    Add(yRow, yRow, bias, nCount);
                    Muls(yRow, yRow, scaleCorr_, nCount);
                }
                tmpQueue_.FreeTensor(bias);

                // Store Y tile back to GM
                for (uint32_t i = 0; i < mCount; ++i) {
                    DataCopy(yGm_[(static_cast<uint64_t>(mBase + i) * n_) + nBase],
                             LocalTensor<float>(reinterpret_cast<__ubuf__ float*>(yLocal.GetAddr()) + static_cast<uint64_t>(i) * nCount, nCount),
                             nCount);
                }

                yQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline uint32_t min_u32(uint32_t a, uint32_t b) { return a < b ? a : b; }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> xQueue_;
    TQue<TPosition::VECIN, BUFFER_NUM> wQueue_;
    TQue<TPosition::VECOUT, BUFFER_NUM> yQueue_;
    TQue<TPosition::VECCALC, 2> tmpQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t m_{0}, n_{0}, k_{0};
    uint32_t tileM_{1}, tileN_{1}, tileK_{1};
    float scaleCorr_{1.0f};
};

extern "C" __global__ __aicore__ void matmul_scaling_residual_add_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScalingResidualAddCustom op;
    op.Init(x, w, b, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.scaling);
    op.Process();
}
