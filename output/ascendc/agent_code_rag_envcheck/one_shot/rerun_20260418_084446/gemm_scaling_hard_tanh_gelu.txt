#include "gemm_scaling_hard_tanh_gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmScalingHardTanhGeluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetOriginShape().GetDimNum() != 2 ||
        w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t K = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));

    // Set conservative tiles to fit in per-core UB ~192KB.
    // A sub: tileM*chunkN, B sub: tileK*chunkN, Acc: tileM*tileK.
    // Choose defaults and clamp to problem size.
    uint32_t tileM = 32;
    uint32_t tileK = 32;
    uint32_t chunkN = 256;

    if (M < tileM) tileM = (M == 0) ? 1u : AlignDown(M, 8u) ? AlignDown(M, 8u) : M;
    if (K < tileK) tileK = (K == 0) ? 1u : AlignDown(K, 8u) ? AlignDown(K, 8u) : K;
    if (N < chunkN) chunkN = (N == 0) ? 1u : AlignDown(N, 16u) ? AlignDown(N, 16u) : N;

    // BlockDim roughly equals number of row tiles, bounded in [8, 24]
    uint32_t numRowTiles = (tileM == 0) ? 1u : CeilDiv(M, tileM);
    uint32_t blockDim = numRowTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_chunkN(chunkN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

using namespace AscendC;

// Simple, conservative fused kernel: Y = GELU(hardtanh((X @ W^T + b) * s))
extern "C" __global__ __aicore__ void gemm_scaling_hard_tanh_gelu_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling, GM_ADDR ht_min, GM_ADDR ht_max,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t M = tiling_data.M;
    const uint32_t N = tiling_data.N;
    const uint32_t K = tiling_data.K;
    const uint32_t tileM = tiling_data.tileM;
    const uint32_t tileK = tiling_data.tileK;

    if (M == 0 || N == 0 || K == 0) {
        return;
    }

    __gm__ float* xGm = reinterpret_cast<__gm__ float*>(x);           // [M, N]
    __gm__ float* wGm = reinterpret_cast<__gm__ float*>(w);           // [K, N]
    __gm__ float* bGm = reinterpret_cast<__gm__ float*>(b);           // [K]
    __gm__ float* sGm = reinterpret_cast<__gm__ float*>(scaling);     // [1]
    __gm__ float* mnGm = reinterpret_cast<__gm__ float*>(ht_min);     // [1]
    __gm__ float* mxGm = reinterpret_cast<__gm__ float*>(ht_max);     // [1]
    __gm__ float* yGm = reinterpret_cast<__gm__ float*>(y);           // [M, K]

    // Load scalar attributes
    float s = 1.0f;
    float htMin = -1e9f;
    float htMax = 1e9f;
    // Direct GM reads for scalars
    s = sGm[0];
    htMin = mnGm[0];
    htMax = mxGm[0];

    const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

    // Grid-stride over row tiles
    const uint32_t rowStride = blockNum * tileM;
    uint32_t rowBase = blockIdx * tileM;

    while (rowBase < M) {
        const uint32_t mEff = (rowBase + tileM <= M) ? tileM : (M - rowBase);

        // Iterate over column tiles
        for (uint32_t colBase = 0; colBase < K; colBase += tileK) {
            const uint32_t kEff = (colBase + tileK <= K) ? tileK : (K - colBase);

            // Compute each output element in the tile
            for (uint32_t mi = 0; mi < mEff; ++mi) {
                const uint32_t xRowOff = (rowBase + mi) * N;

                for (uint32_t kj = 0; kj < kEff; ++kj) {
                    const uint32_t kIdx = colBase + kj;
                    // Accumulate dot(x[i, :], w[kIdx, :])
                    float acc = bGm[kIdx];

                    const uint32_t wRowOff = kIdx * N;
                    // Simple scalar loop over N; may be improved with tiling/vectorization.
                    for (uint32_t n = 0; n < N; ++n) {
                        acc += xGm[xRowOff + n] * wGm[wRowOff + n];
                    }

                    // Scale
                    float v = acc * s;
                    // HardTanh clamp
                    if (v < htMin) v = htMin;
                    if (v > htMax) v = htMax;

                    // Fast GELU approximation using tanh
                    const float c0 = 0.79788456f;    // sqrt(2/pi)
                    const float c1 = 0.044715f;
                    const float v3 = v * v * v;
                    const float u = c0 * (v + c1 * v3);
                    // std::tanhf may be mapped to device math by toolchain
                    const float th = tanhf(u);
                    const float gelu = 0.5f * v * (1.0f + th);

                    yGm[(rowBase + mi) * K + kIdx] = gelu;
                }
            }
        }

        rowBase += rowStride;
    }
}
"""
