host_operator_src="""

#include "cumprod_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // namespace

namespace optiling {

// UB model for this kernel (float32):
//   inQueue: 2 * tileInner
//   outQueue: 2 * tileInner
//   runBuf: 1 * tileInner
// Total floats = 5 * tileInner => bytes = 20 * tileInner
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumprodCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);
    if (dimPtr == nullptr) {
        return ge::GRAPH_PARAM_INVALID;
    }

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape &origin = xShape->GetOriginShape();

    const int32_t rank = static_cast<int32_t>(origin.GetDimNum());
    if (rank <= 0) {
        return ge::GRAPH_PARAM_INVALID;
    }

    int64_t dim = *dimPtr;
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) {
        return ge::GRAPH_PARAM_INVALID;
    }

    uint64_t totalLength64 = static_cast<uint64_t>(origin.GetShapeSize());
    if (totalLength64 == 0) {
        // trivial
        tiling.set_totalLength(0);
        tiling.set_tileInner(0);
        tiling.set_blockDim(1);
        tiling.set_outer(0);
        tiling.set_axis(0);
        tiling.set_inner(0);
        tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                            context->GetRawTilingData()->GetCapacity());
        context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());
        context->SetBlockDim(1);
        size_t *currentWorkspace = context->GetWorkspaceSizes(1);
        currentWorkspace[0] = 0;
        return ge::GRAPH_SUCCESS;
    }

    uint64_t outer64 = 1;
    uint64_t axis64 = 1;
    uint64_t inner64 = 1;

    for (int32_t i = 0; i < dim; ++i) {
        outer64 *= static_cast<uint64_t>(origin.GetDim(i));
    }
    axis64 = static_cast<uint64_t>(origin.GetDim(static_cast<uint32_t>(dim)));
    for (int32_t i = dim + 1; i < rank; ++i) {
        inner64 *= static_cast<uint64_t>(origin.GetDim(i));
    }

    // Clamp to 32-bit for kernel fields
    uint32_t totalLength = static_cast<uint32_t>(totalLength64);
    uint32_t outer = static_cast<uint32_t>(outer64);
    uint32_t axis = static_cast<uint32_t>(axis64);
    uint32_t inner = static_cast<uint32_t>(inner64);

    // tileInner from UB budget: bytes = 20 * tileInner
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileInner = maxTile;
    if (inner < tileInner) {
        const uint32_t aligned = AlignDown(inner, ALIGN_ELEMS);
        tileInner = (aligned > 0) ? aligned : inner;
    }

    if (tileInner == 0) tileInner = 1; // safety

    const uint32_t innerChunks = CeilDiv(inner, tileInner);
    uint32_t totalGroups = (outer == 0) ? 1u : outer;
    totalGroups *= (innerChunks == 0 ? 1u : innerChunks);

    uint32_t blockDim = totalGroups;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim > totalGroups) blockDim = totalGroups;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);
    tiling.set_outer(outer);
    tiling.set_axis(axis);
    tiling.set_inner(inner);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumprodCustom : public OpDef {
public:
    explicit CumprodCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }

OP_ADD(CumprodCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 2;

class KernelCumprodCustom {
public:
    __aicore__ inline KernelCumprodCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t outer, uint32_t axis, uint32_t inner,
                                uint32_t tileInner)
    {
        totalLen_ = totalLength;
        outer_ = outer;
        axis_ = axis;
        inner_ = inner;
        tileInner_ = tileInner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueue_, BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileInner_ * sizeof(float));
        pipe_.InitBuffer(calcQueue_, 1, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || axis_ == 0 || inner_ == 0) {
            // axis_ == 0 is valid (length 0 cannot occur), but if any is zero, nothing to do
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        const uint32_t innerChunks = CeilDiv(inner_, tileInner_);
        const uint32_t totalGroups = outer_ * innerChunks;

        if (blockIdx >= totalGroups) return;

        for (uint32_t g = blockIdx; g < totalGroups; g += blockNum) {
            const uint32_t o = g / innerChunks;
            const uint32_t chunk = g % innerChunks;
            const uint32_t i0 = chunk * tileInner_;
            uint32_t len = inner_ - i0;
            if (len > tileInner_) len = tileInner_;
            if (len == 0) continue;

            const uint64_t base_o = static_cast<uint64_t>(o) * static_cast<uint64_t>(axis_) * static_cast<uint64_t>(inner_);

            // Allocate running buffer
            LocalTensor<float> runBuf = calcQueue_.AllocTensor<float>();

            // a = 0: y = x
            {
                LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

                const uint64_t off = base_o + static_cast<uint64_t>(0) * inner_ + i0;
                DataCopy(xLocal, xGm_[off], len);
                DataCopy(runBuf, xLocal, len);
                DataCopy(yLocal, runBuf, len);
                DataCopy(yGm_[off], yLocal, len);

                outQueue_.FreeTensor(yLocal);
                inQueue_.FreeTensor(xLocal);
            }

            // a >= 1: y = x * runBuf; runBuf = y
            for (uint32_t a = 1; a < axis_; ++a) {
                LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

                const uint64_t off = base_o + static_cast<uint64_t>(a) * inner_ + i0;
                DataCopy(xLocal, xGm_[off], len);

                Mul(yLocal, xLocal, runBuf, len);
                DataCopy(runBuf, yLocal, len);
                DataCopy(yGm_[off], yLocal, len);

                outQueue_.FreeTensor(yLocal);
                inQueue_.FreeTensor(xLocal);
            }

            calcQueue_.FreeTensor(runBuf);
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) const
    {
        return (a + b - 1u) / b;
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM> inQueue_;
    TQue<TPosition::VECOUT, BUFFER_NUM> outQueue_;
    TQue<TPosition::VECCALC, 1> calcQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> yGm_;

    uint32_t totalLen_{0};
    uint32_t outer_{0};
    uint32_t axis_{0};
    uint32_t inner_{0};
    uint32_t tileInner_{0};
};

extern "C" __global__ __aicore__ void cumprod_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumprodCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.outer,
            tiling_data.axis,
            tiling_data.inner,
            tiling_data.tileInner);
    op.Process();
}
