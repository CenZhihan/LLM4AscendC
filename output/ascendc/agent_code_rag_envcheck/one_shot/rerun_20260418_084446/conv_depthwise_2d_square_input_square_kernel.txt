#include "conv_depthwise_2d_square_input_square_kernel_custom_tiling.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}
}

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t MAX_CORES = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dSquareInputSquareKernelCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    if (xShape->GetOriginShape().GetDimNum() != 4 || wShape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(0));
    const uint32_t C = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(1));
    const uint32_t H = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(2));
    const uint32_t W = static_cast<uint32_t>(xShape->GetOriginShape().GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(0));
    const uint32_t CinPerGroup = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(1));
    const uint32_t K = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(2));
    const uint32_t K2 = static_cast<uint32_t>(wShape->GetOriginShape().GetDim(3));

    // Basic validation for depthwise: Cout == C, CinPerGroup == 1, K square
    if (CinPerGroup != 1 || Cout != C || K != K2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t Ho = (H >= K) ? (H - K + 1) : 0;
    const uint32_t Wo = (W >= K) ? (W - K + 1) : 0;

    // UB model: For our simple pipeline we use per-tile buffers:
    //   yVec: tileW
    //   tmpVec: tileW
    //   wLocal: K*K
    // Total floats = 2 * tileW + K*K
    // bytes = 4 * (2 * tileW + K*K)
    uint32_t floatsBudget = UB_BUDGET_BYTES / 4;
    uint32_t maxTileW = 0;
    if (floatsBudget > (K * K)) {
        maxTileW = (floatsBudget - (K * K)) / 2;
    }
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);

    uint32_t tileW = maxTileW;
    if (Wo > 0 && tileW > Wo) {
        const uint32_t aligned = AlignDown(Wo, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : Wo;
    }

    // Grid dimension: parallel over N*C groups
    const uint32_t groups = N * C;
    uint32_t blockDim = groups;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > MAX_CORES) blockDim = MAX_CORES;
    context->SetBlockDim(blockDim);

    tiling.set_nSize(N);
    tiling.set_cSize(C);
    tiling.set_hSize(H);
    tiling.set_wSize(W);
    tiling.set_kSize(K);
    tiling.set_hOut(Ho);
    tiling.set_wOut(Wo);
    tiling.set_usedCoreNum(blockDim);
    tiling.set_tileW(tileW);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t INBUF_NUM = 1;
constexpr int32_t OUTBUF_NUM = 1;
constexpr int32_t TMPBUF_NUM = 1;
constexpr int32_t WBUF_NUM = 1;

class KernelConvDepthwise2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t nSize, uint32_t cSize,
                                uint32_t hSize, uint32_t wSize,
                                uint32_t kSize, uint32_t hOut, uint32_t wOut,
                                uint32_t usedCoreNum, uint32_t tileW)
    {
        nSize_ = nSize;
        cSize_ = cSize;
        hSize_ = hSize;
        wSize_ = wSize;
        kSize_ = kSize;
        hOut_ = hOut;
        wOut_ = wOut;
        usedCoreNum_ = usedCoreNum;
        tileW_ = (tileW == 0) ? wOut_ : tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(nSize_ * cSize_ * hSize_ * wSize_));
        wGm_.SetGlobalBuffer((__gm__ float*)weight, static_cast<uint32_t>(cSize_ * kSize_ * kSize_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(nSize_ * cSize_ * hOut_ * wOut_));

        // UB buffers
        pipe_.InitBuffer(inQueue_, INBUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUTBUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMPBUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, WBUF_NUM, kSize_ * kSize_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (nSize_ == 0 || cSize_ == 0 || hOut_ == 0 || wOut_ == 0 || kSize_ == 0) return;

        const uint32_t groups = nSize_ * cSize_;
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

        // Static partition like DepthToSpace sample (former/tail)
        uint32_t usedCoreNum = usedCoreNum_;
        if (usedCoreNum == 0) usedCoreNum = blockNum;

        const uint32_t formerNumPerCore = (groups + usedCoreNum - 1) / usedCoreNum;
        const uint32_t tailNumPerCore = (groups / usedCoreNum);
        const uint32_t formerCoreNum = (groups % usedCoreNum);

        uint32_t startGroup = 0;
        uint32_t groupCount = 0;
        if (blockIdx < formerCoreNum) {
            startGroup = blockIdx * formerNumPerCore;
            groupCount = formerNumPerCore;
        } else if (tailNumPerCore != 0) {
            startGroup = formerCoreNum * formerNumPerCore + (blockIdx - formerCoreNum) * tailNumPerCore;
            groupCount = tailNumPerCore;
        } else {
            return;
        }
        if (startGroup >= groups) return;
        if (startGroup + groupCount > groups) {
            groupCount = groups - startGroup;
        }

        for (uint32_t gi = 0; gi < groupCount; ++gi) {
            const uint32_t g = startGroup + gi;
            const uint32_t n = g / cSize_;
            const uint32_t c = g % cSize_;

            // Load weight for this channel into UB once
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            const uint32_t wOff = c * (kSize_ * kSize_);
            AscendC::DataCopy(wLocal, wGm_[wOff], kSize_ * kSize_);

            for (uint32_t ho = 0; ho < hOut_; ++ho) {
                for (uint32_t woBase = 0; woBase < wOut_; woBase += tileW_) {
                    const uint32_t curW = (woBase + tileW_ <= wOut_) ? tileW_ : (wOut_ - woBase);

                    // Prepare output accumulator
                    AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
                    AscendC::Duplicate(yLocal, 0.0f, curW);

                    // y += sum_{r,c} weight[r,c] * x[n,c, ho+r, woBase+c : woBase+c+curW]
                    for (uint32_t kr = 0; kr < kSize_; ++kr) {
                        const uint32_t inRowBase = (((n * cSize_ + c) * hSize_) + (ho + kr)) * wSize_;
                        for (uint32_t kc = 0; kc < kSize_; ++kc) {
                            const uint32_t xOff = inRowBase + (woBase + kc);

                            // Load input slice
                            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                            AscendC::DataCopy(tLocal, xGm_[xOff], curW);

                            // Fetch scalar weight from UB
                            const uint32_t wIdx = kr * kSize_ + kc;
                            // Assume LocalTensor supports reading scalar value by GetValue
                            float wval = wLocal.GetValue(wIdx);

                            // Scale and accumulate
                            AscendC::Muls(tLocal, tLocal, wval, curW);
                            AscendC::Add(yLocal, yLocal, tLocal, curW);

                            tmpQueue_.FreeTensor(tLocal);
                        }
                    }

                    // Write back
                    outQueue_.EnQue(yLocal);
                    AscendC::LocalTensor<float> yDeq = outQueue_.DeQue<float>();
                    const uint32_t yOff = (((n * cSize_ + c) * hOut_) + ho) * wOut_ + woBase;
                    AscendC::DataCopy(yGm_[yOff], yDeq, curW);
                    outQueue_.FreeTensor(yDeq);
                }
            }

            wQueue_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INBUF_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUTBUF_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMPBUF_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, WBUF_NUM> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t nSize_{0};
    uint32_t cSize_{0};
    uint32_t hSize_{0};
    uint32_t wSize_{0};
    uint32_t kSize_{0};
    uint32_t hOut_{0};
    uint32_t wOut_{0};
    uint32_t usedCoreNum_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.nSize, tiling_data.cSize,
            tiling_data.hSize, tiling_data.wSize,
            tiling_data.kSize, tiling_data.hOut, tiling_data.wOut,
            tiling_data.usedCoreNum, tiling_data.tileW);
    op.Process();
}
