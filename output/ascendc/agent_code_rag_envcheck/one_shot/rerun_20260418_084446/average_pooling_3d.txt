#include "average_pooling_3d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDivU32(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
static inline int64_t FloorDivNonNeg(int64_t a, int64_t b) {
    return a / b;
}
} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling3dCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &dims = in_shape->GetOriginShape().GetDims();
    if (dims.size() != 5) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = static_cast<uint32_t>(dims[0]);
    const uint32_t C = static_cast<uint32_t>(dims[1]);
    const uint32_t D = static_cast<uint32_t>(dims[2]);
    const uint32_t H = static_cast<uint32_t>(dims[3]);
    const uint32_t W = static_cast<uint32_t>(dims[4]);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *k_ptr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *s_ptr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *p_ptr = attrs->GetAttrPointer<int32_t>(2);

    uint32_t K = static_cast<uint32_t>(*k_ptr);
    uint32_t S = static_cast<uint32_t>(*s_ptr);
    uint32_t P = static_cast<uint32_t>(*p_ptr);

    if (S == 0u) S = K;

    auto safe_out = [](int64_t in, int64_t k, int64_t s, int64_t p)->uint32_t {
        int64_t val = (in + 2 * p - k);
        if (val < 0) val = -1; // so floor division yields 0 when +1
        return static_cast<uint32_t>(FloorDivNonNeg(val, s) + 1);
    };

    const uint32_t outD = safe_out(D, K, S, P);
    const uint32_t outH = safe_out(H, K, S, P);
    const uint32_t outW = safe_out(W, K, S, P);

    const uint64_t totalIn  = static_cast<uint64_t>(N) * C * D * H * W;
    const uint64_t totalOut = static_cast<uint64_t>(N) * C * outD * outH * outW;

    uint32_t blockDim = 16u;
    uint32_t tiles = (totalOut > 0) ? static_cast<uint32_t>(std::min<uint64_t>(totalOut, 65535)) : 1u;
    if (tiles < 8u) blockDim = 8u;
    else if (tiles > 24u) blockDim = 24u;
    else blockDim = tiles;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_d(D);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_outD(outD);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_kernelSize(K);
    tiling.set_stride(S);
    tiling.set_padding(P);
    tiling.set_totalIn(totalIn);
    tiling.set_totalOut(totalOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const auto &dims = x_shape->GetOriginShape().GetDims();
    if (dims.size() != 5) {
        return GRAPH_FAILED;
    }
    const int64_t N = dims[0], C = dims[1], D = dims[2], H = dims[3], W = dims[4];

    const int32_t kernel_size = *context->GetAttrs()->GetAttrPointer<int32_t>(0);
    int32_t stride = *context->GetAttrs()->GetAttrPointer<int32_t>(1);
    const int32_t padding = *context->GetAttrs()->GetAttrPointer<int32_t>(2);
    if (stride == 0) stride = kernel_size;

    auto out_dim = [](int64_t in, int64_t k, int64_t s, int64_t p)->int64_t{
        return (in + 2 * p - k) / s + 1;
    };

    const int64_t outD = out_dim(D, kernel_size, stride, padding);
    const int64_t outH = out_dim(H, kernel_size, stride, padding);
    const int64_t outW = out_dim(W, kernel_size, stride, padding);

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> y_dims = {N, C, outD, outH, outW};
    y_shape->SetDim(y_dims);
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);

}

kernel_src="""
#include "kernel_operator.h"

class KernelAveragePooling3dCustom {
public:
    __aicore__ inline KernelAveragePooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding,
                                uint64_t totalIn, uint64_t totalOut)
    {
        N_ = n; C_ = c; D_ = d; H_ = h; W_ = w;
        oD_ = outD; oH_ = outH; oW_ = outW;
        K_ = kernelSize; S_ = stride; P_ = padding;
        totalIn_ = totalIn; totalOut_ = totalOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalIn_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut_);

        pipe_.InitBuffer(sumBufQ_, 1, sizeof(float));
        pipe_.InitBuffer(tmpValQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t outIdx = blockIdx; outIdx < totalOut_; outIdx += blockNum) {
            uint32_t n, c, od, oh, ow;
            DecodeIndex(outIdx, n, c, od, oh, ow);

            const int32_t baseD = static_cast<int32_t>(od) * static_cast<int32_t>(S_) - static_cast<int32_t>(P_);
            const int32_t baseH = static_cast<int32_t>(oh) * static_cast<int32_t>(S_) - static_cast<int32_t>(P_);
            const int32_t baseW = static_cast<int32_t>(ow) * static_cast<int32_t>(S_) - static_cast<int32_t>(P_);

            AscendC::LocalTensor<float> sumLocal = sumBufQ_.AllocTensor<float>();
            AscendC::Duplicate(sumLocal, 0.0f, 1);

            uint32_t validCount = 0u;

            for (uint32_t kd = 0; kd < K_; ++kd) {
                int32_t id = baseD + static_cast<int32_t>(kd);
                if (id < 0 || id >= static_cast<int32_t>(D_)) continue;
                for (uint32_t kh = 0; kh < K_; ++kh) {
                    int32_t ih = baseH + static_cast<int32_t>(kh);
                    if (ih < 0 || ih >= static_cast<int32_t>(H_)) continue;
                    for (uint32_t kw = 0; kw < K_; ++kw) {
                        int32_t iw = baseW + static_cast<int32_t>(kw);
                        if (iw < 0 || iw >= static_cast<int32_t>(W_)) continue;

                        uint64_t inLinear = IndexIn(n, c, static_cast<uint32_t>(id),
                                                    static_cast<uint32_t>(ih),
                                                    static_cast<uint32_t>(iw));

                        AscendC::LocalTensor<float> valLocal = tmpValQ_.AllocTensor<float>();
                        AscendC::DataCopy(valLocal, xGm_[inLinear], 1);
                        AscendC::Add(sumLocal, sumLocal, valLocal, 1);
                        tmpValQ_.FreeTensor(valLocal);
                        ++validCount;
                    }
                }
            }

            const float invCount = (validCount > 0u) ? (1.0f / static_cast<float>(validCount)) : 0.0f;
            AscendC::Muls(sumLocal, sumLocal, invCount, 1);

            AscendC::DataCopy(yGm_[outIdx], sumLocal, 1);
            sumBufQ_.FreeTensor(sumLocal);
        }
    }

private:
    __aicore__ inline void DecodeIndex(uint64_t idx, uint32_t &n, uint32_t &c, uint32_t &od, uint32_t &oh, uint32_t &ow) const
    {
        uint64_t t = idx;
        ow = static_cast<uint32_t>(t % oW_); t /= oW_;
        oh = static_cast<uint32_t>(t % oH_); t /= oH_;
        od = static_cast<uint32_t>(t % oD_); t /= oD_;
        c  = static_cast<uint32_t>(t % C_);  t /= C_;
        n  = static_cast<uint32_t>(t);
    }

    __aicore__ inline uint64_t IndexIn(uint32_t n, uint32_t c, uint32_t id, uint32_t ih, uint32_t iw) const
    {
        return (((static_cast<uint64_t>(n) * C_ + c) * D_ + id) * H_ + ih) * W_ + iw;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpValQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, D_{0}, H_{0}, W_{0};
    uint32_t oD_{0}, oH_{0}, oW_{0};
    uint32_t K_{1}, S_{1}, P_{0};
    uint64_t totalIn_{0}, totalOut_{0};
};

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling3dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.totalIn, tiling_data.totalOut);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor avg_pool3d_impl_npu(const at::Tensor& self, int64_t kernel_size, int64_t stride, int64_t padding) {
    at::Tensor result = at::empty({0}, self.options());
    if (stride == 0) {
        stride = kernel_size;
    }
    result = at::empty({self.size(0), self.size(1),
                        (self.size(2) + 2 * padding - kernel_size) / stride + 1,
                        (self.size(3) + 2 * padding - kernel_size) / stride + 1,
                        (self.size(4) + 2 * padding - kernel_size) / stride + 1}, self.options());
    EXEC_NPU_CMD(aclnnAveragePooling3dCustom, self, (int)kernel_size, (int)stride, (int)padding, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("average_pooling_3d_custom", &avg_pool3d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("average_pooling_3d_custom", &avg_pool3d_impl_npu, "3D Average Pooling (AveragePooling3dCustom on NPU)");

import torch
import torch_npu
import custom_ops_lib

_FN = 'average_pooling_3d_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, kernel_size: int, stride: int = None, padding: int = 0):
        super(ModelNew, self).__init__()
        self.kernel_size = int(kernel_size)
        self.stride = int(stride) if stride is not None else 0  # 0 means use kernel_size
        self.padding = int(padding)

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        fn = getattr(custom_ops_lib, _FN)
