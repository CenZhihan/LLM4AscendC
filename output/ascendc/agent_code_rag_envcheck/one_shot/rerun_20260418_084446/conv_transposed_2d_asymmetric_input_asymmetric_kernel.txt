static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    using TData = ConvTransposed2dAsymmetricInputAsymmetricKernelCustomTilingData;
    TData tiling;

    // Shapes
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 4 || w_shape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t Hin = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t Win = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));

    const uint32_t Cin_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t Cout  = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    const uint32_t Kh    = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    const uint32_t Kw    = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(3));

    if (Cin_w != Cin) {
        return ge::GRAPH_FAILED;
    }

    // Fixed stride=1, padding=0, dilation=1
    const uint32_t Hout = Hin + Kh - 1;
    const uint32_t Wout = Win + Kw - 1;

    const uint64_t totalOut = static_cast<uint64_t>(N) * Cout * Hout * Wout;

    // UB budget: store only output tile buffer. Conservative per-core UB share.
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t ALIGN_ELEMS = 1024;
    uint32_t maxTile = UB_BUDGET_BYTES / 4; // float
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalOut < tileLen) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(totalOut), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(totalOut);
    }

    const uint32_t numTiles = tileLen == 0 ? 1u : CeilDiv(totalOut, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_hin(Hin);
    tiling.set_win(Win);
    tiling.set_cout(Cout);
    tiling.set_kh(Kh);
    tiling.set_kw(Kw);
    tiling.set_hout(Hout);
    tiling.set_wout(Wout);
    tiling.set_totalOut(totalOut);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed2dAsymInAsymKerCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymInAsymKerCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Hin, uint32_t Win,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t Hout, uint32_t Wout,
                                uint64_t totalOut, uint32_t tileLen)
    {
        N_ = N; Cin_ = Cin; Hin_ = Hin; Win_ = Win;
        Cout_ = Cout; Kh_ = Kh; Kw_ = Kw;
        Hout_ = Hout; Wout_ = Wout;
        totalOut_ = totalOut; tileLen_ = tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * Cin_ * Hin_ * Win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Cin_) * Cout_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * tileLen_;
        uint64_t base = static_cast<uint64_t>(blockIdx) * tileLen_;
        if (base >= totalOut_) return;

        while (true) {
            uint32_t curLen = static_cast<uint32_t>(totalOut_ - base);
            if (curLen > tileLen_) curLen = tileLen_;

            ComputeAndStore(base, curLen);

            if (base + strideElems >= totalOut_) break;
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeAndStore(uint64_t base, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        // Initialize local output tile to zeros
        for (uint32_t i = 0; i < len; ++i) {
            // Assuming SetValue exists for LocalTensor; placeholder for setting element
            yLocal.SetValue(i, 0.0f);
        }

        const uint64_t nStride = static_cast<uint64_t>(Cout_) * Hout_ * Wout_;
        const uint64_t coStride = static_cast<uint64_t>(Hout_) * Wout_;
        const uint32_t s2 = Wout_;

        for (uint32_t i = 0; i < len; ++i) {
            uint64_t g = base + i;
            uint32_t n = static_cast<uint32_t>(g / nStride);
            g -= static_cast<uint64_t>(n) * nStride;
            uint32_t co = static_cast<uint32_t>(g / coStride);
            g -= static_cast<uint64_t>(co) * coStride;
            uint32_t oh = static_cast<uint32_t>(g / s2);
            uint32_t ow = static_cast<uint32_t>(g - static_cast<uint64_t>(oh) * s2);

            float acc = 0.0f;

            // Transposed conv (stride=1, pad=0, dilation=1)
            // ih = oh - (Kh - 1 - kh); iw = ow - (Kw - 1 - kw)
            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const int32_t ih = static_cast<int32_t>(oh) - static_cast<int32_t>(Kh_ - 1 - kh);
                    if (ih < 0 || ih >= static_cast<int32_t>(Hin_)) continue;
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const int32_t iw = static_cast<int32_t>(ow) - static_cast<int32_t>(Kw_ - 1 - kw);
                        if (iw < 0 || iw >= static_cast<int32_t>(Win_)) continue;

                        const uint64_t xOff = (((static_cast<uint64_t>(n) * Cin_ + ci) * Hin_ + static_cast<uint32_t>(ih)) * Win_ + static_cast<uint32_t>(iw));
                        const uint64_t wOff = ((((static_cast<uint64_t>(ci) * Cout_ + co) * Kh_ + kh) * Kw_) + kw);

                        const float xv = xGm_[xOff];
                        const float wv = wGm_[wOff];
                        acc += xv * wv;
                    }
                }
            }
            yLocal.SetValue(i, acc);
        }

        AscendC::DataCopy(yGm_[base], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Hin_{0}, Win_{0};
    uint32_t Cout_{0}, Kh_{0}, Kw_{0};
    uint32_t Hout_{0}, Wout_{0};
    uint64_t totalOut_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dAsymInAsymKerCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.hin, tiling_data.win,
            tiling_data.cout, tiling_data.kh, tiling_data.kw,
            tiling_data.hout, tiling_data.wout,
            tiling_data.totalOut, tiling_data.tileLen);
    op.Process();
}
