host_operator_src="""

#include "conv2d_relu_hard_swish_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous namespace

namespace optiling {

// UB model (per core, per-iteration):
//   accVec: 1 * tileW
//   xVec:   1 * tileW
//   tmpVec: 2 * tileW
// Total floats ~4 * tileW => bytes = 16 * tileW. Keep margin for queues.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 64;
static constexpr uint32_t HARD_CAP_ELEMS = 4096;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dReluHardSwishCustomTilingData tiling;

    // x: [N, Cin, H, W]
    const gert::Shape* x_shape = context->GetInputShape(0);
    // weight: [Cout, Cin, K, K]
    const gert::Shape* w_shape = context->GetInputShape(1);

    const uint32_t N    = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H    = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W    = static_cast<uint32_t>(x_shape->GetDim(3));
    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t K    = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t stride = 1, pad = 0, dil = 1;

    // Compute output sizes for stride=1, pad=0, dil=1
    const uint32_t Hout = (H + 2 * pad - dil * (K - 1) - 1) / stride + 1;
    const uint32_t Wout = (W + 2 * pad - dil * (K - 1) - 1) / stride + 1;

    // tileW from UB budget: bytes ≈ 16 * tileW
    uint32_t maxTile = UB_BUDGET_BYTES / 16;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileW = maxTile;
    if (Wout < tileW) {
        const uint32_t aligned = AlignDown(Wout, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : Wout;
    }

    // Grid strategy: one logical task = (n, oc)
    const uint64_t totalTasks = static_cast<uint64_t>(N) * static_cast<uint64_t>(Cout);
    uint32_t blockDim = static_cast<uint32_t>(totalTasks);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    const bool hasBias = (context->GetInputNum() >= 3) && (context->GetInputShape(2)->GetShapeSize() > 0);

    // Fill tiling data
    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_K(K);
    tiling.set_Hout(Hout);
    tiling.set_Wout(Wout);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1u : 0u);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;

class KernelConv2dReluHardSwishCustom {
public:
    __aicore__ inline KernelConv2dReluHardSwishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t Hout, uint32_t Wout,
                                uint32_t tileW, uint32_t hasBias)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; K_ = K;
        Hout_ = Hout; Wout_ = Wout;
        tileW_ = tileW;
        hasBias_ = (hasBias != 0);

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), N_ * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w), Cout_ * Cin_ * K_ * K_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b), Cout_);
        }
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), N_ * Cout_ * Hout_ * Wout_);

        pipe_.InitBuffer(inQue_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQue_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(calcBuf_, 3 * tileW_ * sizeof(float) + 16);
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cout_ == 0 || Hout_ == 0 || Wout_ == 0 || tileW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        const uint32_t totalTasks = N_ * Cout_; // task = (n, oc)
        for (uint32_t task = blockIdx; task < totalTasks; task += blockNum) {
            const uint32_t n  = task / Cout_;
            const uint32_t oc = task % Cout_;
            ComputeTask(n, oc);
        }
    }

private:
    __aicore__ inline uint64_t XIndex(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((static_cast<uint64_t>(n) * Cin_ + c) * H_ + h) * W_ + w);
    }
    __aicore__ inline uint64_t WIndex(uint32_t oc, uint32_t ic, uint32_t kh, uint32_t kw) const
    {
        return (((static_cast<uint64_t>(oc) * Cin_ + ic) * K_ + kh) * K_ + kw);
    }
    __aicore__ inline uint64_t YIndex(uint32_t n, uint32_t oc, uint32_t h, uint32_t w) const
    {
        return (((static_cast<uint64_t>(n) * Cout_ + oc) * Hout_ + h) * Wout_ + w);
    }

    __aicore__ inline void LoadWeightScalar(uint32_t oc, uint32_t ic, uint32_t kh, uint32_t kw, float &wv)
    {
        const uint64_t wOff = WIndex(oc, ic, kh, kw);
        // Use a 1-element local to fetch scalar
        LocalTensor<float> w1 = calcBuf_.Get<float>(1);
        DataCopy(w1, wGm_[wOff], 1);
        // Assume we can read back first element into scalar
        wv = w1.GetValue(0);
    }

    __aicore__ inline void LoadBiasScalar(uint32_t oc, float &bv)
    {
        if (!hasBias_) { bv = 0.0f; return; }
        LocalTensor<float> b1 = calcBuf_.Get<float>(1);
        DataCopy(b1, bGm_[oc], 1);
        bv = b1.GetValue(0);
    }

    __aicore__ inline void ComputeTask(uint32_t n, uint32_t oc)
    {
        float bias_v = 0.0f;
        LoadBiasScalar(oc, bias_v);

        for (uint32_t oh = 0; oh < Hout_; ++oh) {
            for (uint32_t wBase = 0; wBase < Wout_; wBase += tileW_) {
                uint32_t curW = Wout_ - wBase;
                if (curW > tileW_) curW = tileW_;

                // acc buffer
                LocalTensor<float> acc = outQue_.AllocTensor<float>();
                Duplicate(acc, 0.0f, curW);

                // Accumulate over Cin and KxK
                for (uint32_t ic = 0; ic < Cin_; ++ic) {
                    for (uint32_t kh = 0; kh < K_; ++kh) {
                        const uint32_t ih = oh + kh; // stride=1, pad=0
                        for (uint32_t kw = 0; kw < K_; ++kw) {
                            const uint32_t iwStart = wBase + kw;

                            // Load input vector of length curW
                            LocalTensor<float> xVec = inQue_.AllocTensor<float>();
                            DataCopy(xVec, xGm_[XIndex(n, ic, ih, iwStart)], curW);
                            inQue_.EnQue(xVec);

                            // Multiply by scalar weight and accumulate
                            LocalTensor<float> tmp = calcBuf_.GetWithOffset<float>(curW, 0);
                            float wv = 0.0f;
                            LoadWeightScalar(oc, ic, kh, kw, wv);
                            Muls(tmp, xVec, wv, curW);
                            Add(acc, acc, tmp, curW);

                            LocalTensor<float> xVecDe = inQue_.DeQue<float>();
                            inQue_.FreeTensor(xVecDe);
                        }
                    }
                }

                // Optional bias add
                if (hasBias_) {
                    Adds(acc, acc, bias_v, curW);
                }

                // ReLU: max(acc, 0)
                Maxs(acc, acc, 0.0f, curW);

                // HardSwish: acc * clamp((acc + 3)/6, 0, 1)
                LocalTensor<float> t = calcBuf_.GetWithOffset<float>(curW, tileW_ * sizeof(float));
                Adds(t, acc, 3.0f, curW);
                Muls(t, t, 1.0f / 6.0f, curW);
                Mins(t, t, 1.0f, curW);
                Maxs(t, t, 0.0f, curW);
                Mul(acc, acc, t, curW);

                // Store
                DataCopy(yGm_[YIndex(n, oc, oh, wBase)], acc, curW);
                outQue_.FreeTensor(acc);
            }
        }
    }

private:
    TPipe pipe_;
    TQue<QuePosition::VECIN, BUFFER_NUM>  inQue_;
    TQue<QuePosition::VECOUT, BUFFER_NUM> outQue_;
    TBuf<TPosition::VECCALC> calcBuf_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, K_{0};
    uint32_t Hout_{0}, Wout_{0}, tileW_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias,
                                                                    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConv2dReluHardSwishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.K,
            tiling_data.Hout, tiling_data.Wout,
            tiling_data.tileW, tiling_data.hasBias);
    op.Process();
}
