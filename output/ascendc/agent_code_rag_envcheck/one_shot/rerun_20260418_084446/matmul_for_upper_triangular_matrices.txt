host_operator_src="""
#include "matmul_for_upper_triangular_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForUpperTriangularMatricesCustomTilingData tiling;
    const gert::Shape *a_shape = context->GetInputShape(0);
    if (a_shape == nullptr || a_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const int64_t dim0 = a_shape->GetOriginShape().GetDim(0);
    const int64_t dim1 = a_shape->GetOriginShape().GetDim(1);
    if (dim0 <= 0 || dim1 <= 0 || dim0 != dim1) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(dim0);

    // Simple blockDim heuristic for coverage and parallelism
    uint32_t blockDim = CeilDiv(N, 64u);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return GRAPH_FAILED;
    }
    if (a_shape->GetOriginShape().GetDimNum() != 2 ||
        b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    // Output shape is (N, N)
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *a_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MatmulForUpperTriangularMatricesCustom : public OpDef {
public:
    explicit MatmulForUpperTriangularMatricesCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulForUpperTriangularMatricesCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"

extern "C" __global__ __aicore__ void matmul_for_upper_triangular_matrices_custom(
    GM_ADDR A, GM_ADDR B, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    if (g_coreType == AscendC::AIV) {
        return;
    }

    // For upper-triangular inputs, we can leverage a specialized triangle matmul kernel
    // template as in reference examples and produce an upper-triangular result.
    // The internal matmul engine handles tiling via TCubeTiling.
    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    AscendC::TPipe pipe;

    // Types: using float IO/accumulation for numerical stability in this custom op
    // The bias is not used for pure matmul; pass nullptr.
    MatmulCustom::MatmulTriangleKernel<float, float, float, float> triKernel;
    triKernel.Init(A, B, nullptr, y, tiling);
    triKernel.Process(&pipe);
}
