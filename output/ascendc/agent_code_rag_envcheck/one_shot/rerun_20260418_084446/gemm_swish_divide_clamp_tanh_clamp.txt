#include "gemm_swish_divide_clamp_tanh_clamp_custom_tiling.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core practical UB budget
static constexpr uint32_t ALIGN_ELEMS = 256;            // favor bigger burst length
static constexpr uint32_t MIN_TILE_N = 128;
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmSwishDivideClampTanhClampCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);
    const gert::Shape* b_shape = context->GetInputShape(2);

    // Shapes:
    // x: [M, K]
    // w: [K, N]  (note: host/model passes w.t().contiguous(), so KxN here)
    // b: [N]
    const uint32_t dims_x = static_cast<uint32_t>(x_shape->GetDimNum());
    const uint32_t dims_w = static_cast<uint32_t>(w_shape->GetDimNum());
    const uint32_t dims_b = static_cast<uint32_t>(b_shape->GetDimNum());
    if (dims_x != 2 || dims_w != 2 || dims_b != 1) {
        return ge::GRAPH_FAILED;
    }
    const uint64_t M64 = static_cast<uint64_t>(x_shape->GetDim(0));
    const uint64_t Kx  = static_cast<uint64_t>(x_shape->GetDim(1));
    const uint64_t Kw  = static_cast<uint64_t>(w_shape->GetDim(0));
    const uint64_t N64 = static_cast<uint64_t>(w_shape->GetDim(1));
    if (Kx != Kw || N64 != static_cast<uint64_t>(b_shape->GetDim(0))) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t M = static_cast<uint32_t>(M64);
    const uint32_t K = static_cast<uint32_t>(Kx);
    const uint32_t N = static_cast<uint32_t>(N64);

    // kChunk: small chunk for better locality
    uint32_t kChunk = 64;
    if (kChunk > K) kChunk = (K > 0) ? K : 1u;
    // UB usage estimate per row-tile:
    // acc(Nt) + tmp(Nt) + wtRow(Nt) + bias(Nt) + xChunk(kChunk)
    // bytes = (4*Nt + kChunk) * 4
    const uint32_t maxNtFromUb = (UB_BUDGET_BYTES / 4u > kChunk) ? (UB_BUDGET_BYTES / 4u - kChunk) / 4u : MIN_TILE_N;
    uint32_t tileN = maxNtFromUb;
    if (tileN > N) tileN = N;
    tileN = AlignDown(tileN, ALIGN_ELEMS);
    if (tileN < MIN_TILE_N) {
        tileN = (N < MIN_TILE_N) ? N : MIN_TILE_N;
        tileN = AlignDown(tileN, 16);
        if (tileN == 0) tileN = 16;
    }

    // blockDim based on rows
    uint32_t blockDim = M;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_m(M);
    tiling.set_n(N);
    tiling.set_k(K);
    tiling.set_tileN(tileN);
    tiling.set_kChunk(kChunk);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmSwishDivideClampTanhClampCustom {
public:
    __aicore__ inline KernelGemmSwishDivideClampTanhClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wt, GM_ADDR b, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileN, uint32_t kChunk)
    {
        M_ = m; N_ = n; K_ = k;
        tileN_ = tileN; kChunk_ = kChunk;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wtGm_.SetGlobalBuffer((__gm__ float*)wt, static_cast<uint64_t>(K_) * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        pipe_.InitBuffer(biasQ_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(wtRowQ_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(accQ_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(xChunkQ_, 1, kChunk_ * sizeof(float));
        pipe_.InitBuffer(outQ_, BUFFER_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t mIdx = blockIdx; mIdx < M_; mIdx += blockNum) {
            const uint64_t xRowBase = static_cast<uint64_t>(mIdx) * K_;
            const uint64_t yRowBase = static_cast<uint64_t>(mIdx) * N_;

            for (uint32_t n0 = 0; n0 < N_; n0 += tileN_) {
                const uint32_t lenN = AscendC::min(tileN_, N_ - n0);

                // Load bias tile
                AscendC::LocalTensor<float> biasLocal = biasQ_.AllocTensor<float>();
                AscendC::DataCopy(biasLocal, bGm_[n0], lenN);
                biasQ_.EnQue(biasLocal);

                // Initialize accumulator with bias
                AscendC::LocalTensor<float> accLocal = accQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> biasDeq = biasQ_.DeQue<float>();
                AscendC::DataCopy(accLocal, biasDeq, lenN);
                biasQ_.FreeTensor(biasDeq);

                // K loop
                for (uint32_t k0 = 0; k0 < K_; k0 += kChunk_) {
                    const uint32_t kLen = AscendC::min(kChunk_, K_ - k0);

                    // Load x row chunk
                    AscendC::LocalTensor<float> xChunkLocal = xChunkQ_.AllocTensor<float>();
                    AscendC::DataCopy(xChunkLocal, xGm_[xRowBase + k0], kLen);
                    xChunkQ_.EnQue(xChunkLocal);

                    AscendC::LocalTensor<float> xVals = xChunkQ_.DeQue<float>();
                    // Outer-product accumulation: for each t in chunk
                    for (uint32_t t = 0; t < kLen; ++t) {
                        // scalar a = x[m, k0+t]
                        float a = xVals.GetValue(t);

                        // Load wt row: wt[k0+t, n0:n0+lenN]
                        AscendC::LocalTensor<float> wtRowLocal = wtRowQ_.AllocTensor<float>();
                        const uint64_t wtOff = static_cast<uint64_t>(k0 + t) * N_ + n0;
                        AscendC::DataCopy(wtRowLocal, wtGm_[wtOff], lenN);

                        // tmp = wtRow * a
                        AscendC::LocalTensor<float> tmpLocal = tmpQ_.AllocTensor<float>();
                        AscendC::Muls(tmpLocal, wtRowLocal, a, lenN);

                        // acc += tmp
                        AscendC::Add(accLocal, accLocal, tmpLocal, lenN);

                        tmpQ_.FreeTensor(tmpLocal);
                        wtRowQ_.FreeTensor(wtRowLocal);
                    }
                    xChunkQ_.FreeTensor(xVals);
                }

                // Fused activation chain:
                // swish: acc * sigmoid(acc)
                AscendC::LocalTensor<float> tmp1 = tmpQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmp2 = tmpQ_.AllocTensor<float>();

                AscendC::DataCopy(tmp1, accLocal, lenN);          // tmp1 = acc
                AscendC::Sigmoid(tmp2, tmp1, lenN);               // tmp2 = sigmoid(acc)
                AscendC::Mul(tmp1, tmp1, tmp2, lenN);             // tmp1 = acc * sigmoid(acc)

                // divide by 2
                AscendC::Muls(tmp1, tmp1, 0.5f, lenN);

                // clamp [-1, 1]
                AscendC::Maxs(tmp1, tmp1, -1.0f, lenN);
                AscendC::Mins(tmp1, tmp1, 1.0f, lenN);

                // tanh
                AscendC::Tanh(tmp2, tmp1, lenN);

                // final clamp [-1, 1]
                AscendC::Maxs(tmp2, tmp2, -1.0f, lenN);
                AscendC::Mins(tmp2, tmp2, 1.0f, lenN);

                // Write out
                AscendC::LocalTensor<float> yLocal = outQ_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, tmp2, lenN);
                outQ_.EnQue(yLocal);
                AscendC::LocalTensor<float> yDeq = outQ_.DeQue<float>();
                AscendC::DataCopy(yGm_[yRowBase + n0], yDeq, lenN);
                outQ_.FreeTensor(yDeq);

                tmpQ_.FreeTensor(tmp1);
                tmpQ_.FreeTensor(tmp2);
                accQ_.FreeTensor(accLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> biasQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wtRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> accQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xChunkQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wtGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileN_{0}, kChunk_{0};
};

extern "C" __global__ __aicore__ void gemm_swish_divide_clamp_tanh_clamp_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSwishDivideClampTanhClampCustom op;
    op.Init(x, w, b, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileN, tiling_data.kChunk);
    op.Process();
}
