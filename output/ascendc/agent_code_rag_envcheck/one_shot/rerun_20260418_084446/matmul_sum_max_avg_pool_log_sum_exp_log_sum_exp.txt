namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// UB model for this kernel:
//   redChunk: 1 * tileK
//   wChunk:   1 * tileK
//   xChunk:   1 * tileK
//   tmpVec:   1 * tileK
// Total floats ~ 4 * tileK -> bytes = 16 * tileK
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulSumMaxAvgPoolLogSumExpLogSumExpCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    // b is [O]

    const uint32_t dims_x = static_cast<uint32_t>(x_shape->GetOriginShape().GetDimNum());
    const uint32_t dims_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDimNum());
    if (dims_x < 2 || dims_w < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t B64 = x_shape->GetOriginShape().GetDim(0);
    const uint64_t K64 = x_shape->GetOriginShape().GetDim(1);
    const uint64_t O64 = w_shape->GetOriginShape().GetDim(0);

    const uint32_t B = static_cast<uint32_t>(B64);
    const uint32_t K = static_cast<uint32_t>(K64);
    const uint32_t O = static_cast<uint32_t>(O64);

    // tileK from UB budget: bytes = 16 * tileK
    uint32_t maxTile = UB_BUDGET_BYTES / 16;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileK = maxTile;
    if (K < tileK) {
        const uint32_t aligned = AlignDown(K, ALIGN_ELEMS);
        tileK = (aligned > 0) ? aligned : K;
    }
    if (tileK == 0) tileK = K;

    // Single-core to avoid cross-core reductions
    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_batchSize(B);
    tiling.set_inFeatures(K);
    tiling.set_outFeatures(O);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    currentWorkspace[0] = 0;

kernel_src="""

#include "kernel_operator.h"

class KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom {
public:
    __aicore__ inline KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batchSize, uint32_t inFeatures,
                                uint32_t outFeatures, uint32_t tileK)
    {
        B_ = batchSize;
        K_ = inFeatures;
        O_ = outFeatures;
        tileK_ = tileK == 0 ? inFeatures : tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(B_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(O_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, B_);

        // Buffers:
        // - calcQ_: 3 buffers of tileK for red, x, tmp
        // - scalQ_: 3 buffers of 1 element for partial sums and accumulators
        pipe_.InitBuffer(calcQ_, 3, tileK_ * sizeof(float));
        pipe_.InitBuffer(scalQ_, 3, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = AscendC::GetBlockIdx();
        if (blockIdx != 0) {
            return; // single-core kernel
        }
        if (B_ == 0 || K_ == 0) {
            return;
        }

        // Compute sum(b) into a 1-element tensor sumBLoc
        AscendC::LocalTensor<float> sumBLoc = scalQ_.AllocTensor<float>();
        AscendC::Dup(sumBLoc, 0.0f, 1);

        uint32_t bRemaining = O_;
        uint32_t bOffset = 0;
        while (bRemaining > 0) {
            uint32_t cur = bRemaining;
            if (cur > tileK_) cur = tileK_;
            AscendC::LocalTensor<float> bChunk = calcQ_.AllocTensor<float>();
            AscendC::DataCopy(bChunk, bGm_[bOffset], cur);
            AscendC::LocalTensor<float> tmpScalar = scalQ_.AllocTensor<float>();
            AscendC::ReduceSum(tmpScalar, bChunk, cur);
            AscendC::Add(sumBLoc, sumBLoc, tmpScalar, 1);
            scalQ_.FreeTensor(tmpScalar);
            calcQ_.FreeTensor(bChunk);
            bOffset += cur;
            bRemaining -= cur;
        }

        // Initialize y to zeros
        AscendC::LocalTensor<float> zeroLoc = scalQ_.AllocTensor<float>();
        AscendC::Dup(zeroLoc, 0.0f, 1);
        for (uint32_t i = 0; i < B_; ++i) {
            AscendC::DataCopy(yGm_[i], zeroLoc, 1);
        }
        scalQ_.FreeTensor(zeroLoc);

        // Main accumulation over K in chunks: build redChunk = sum_j w[j, kStart:kEnd]
        for (uint32_t kStart = 0; kStart < K_; kStart += tileK_) {
            uint32_t kLen = K_ - kStart;
            if (kLen > tileK_) kLen = tileK_;

            AscendC::LocalTensor<float> redChunk = calcQ_.AllocTensor<float>();
            AscendC::Dup(redChunk, 0.0f, kLen);

            // Accumulate w rows into redChunk
            for (uint32_t j = 0; j < O_; ++j) {
                AscendC::LocalTensor<float> wChunk = calcQ_.AllocTensor<float>();
                const uint64_t wOff = static_cast<uint64_t>(j) * K_ + kStart;
                AscendC::DataCopy(wChunk, wGm_[wOff], kLen);
                AscendC::Add(redChunk, redChunk, wChunk, kLen);
                calcQ_.FreeTensor(wChunk);
            }

            // For each batch row, compute partial dot and accumulate to y
            for (uint32_t i = 0; i < B_; ++i) {
                AscendC::LocalTensor<float> xChunk = calcQ_.AllocTensor<float>();
                const uint64_t xOff = static_cast<uint64_t>(i) * K_ + kStart;
                AscendC::DataCopy(xChunk, xGm_[xOff], kLen);

                AscendC::LocalTensor<float> tmpVec = calcQ_.AllocTensor<float>();
                AscendC::Mul(tmpVec, xChunk, redChunk, kLen);

                AscendC::LocalTensor<float> pSum = scalQ_.AllocTensor<float>();
                AscendC::ReduceSum(pSum, tmpVec, kLen);

                AscendC::LocalTensor<float> yLoc = scalQ_.AllocTensor<float>();
                AscendC::DataCopy(yLoc, yGm_[i], 1);
                AscendC::Add(yLoc, yLoc, pSum, 1);
                AscendC::DataCopy(yGm_[i], yLoc, 1);

                scalQ_.FreeTensor(yLoc);
                scalQ_.FreeTensor(pSum);
                calcQ_.FreeTensor(tmpVec);
                calcQ_.FreeTensor(xChunk);
            }

            calcQ_.FreeTensor(redChunk);
        }

        // Add sum(b) to all y[i]
        for (uint32_t i = 0; i < B_; ++i) {
            AscendC::LocalTensor<float> yLoc = scalQ_.AllocTensor<float>();
            AscendC::DataCopy(yLoc, yGm_[i], 1);
            AscendC::Add(yLoc, yLoc, sumBLoc, 1);
            AscendC::DataCopy(yGm_[i], yLoc, 1);
            scalQ_.FreeTensor(yLoc);
        }

        scalQ_.FreeTensor(sumBLoc);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3> calcQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3> scalQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t B_{0};
    uint32_t K_{0};
    uint32_t O_{0};
    uint32_t tileK_{0};
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom op;
    op.Init(x, w, b, y,
            tiling_data.batchSize,
            tiling_data.inFeatures,
            tiling_data.outFeatures,
            tiling_data.tileK);
    op.Process();
}
