    const gert::Shape *w_h2o_shape = context->GetInputShape(4);
    const gert::Shape *b_h2o_shape = context->GetInputShape(5);

    if (x_shape->GetOriginShape().GetDimNum() != 3 ||
        h0_shape->GetOriginShape().GetDimNum() != 2 ||
        w_i2h_shape->GetOriginShape().GetDimNum() != 2 ||
        w_h2o_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;

kernel_src="""
#include "kernel_operator.h"

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x,
                                GM_ADDR h0,
                                GM_ADDR w_i2h,
                                GM_ADDR b_i2h,
                                GM_ADDR w_h2o,
                                GM_ADDR b_h2o,
                                GM_ADDR y,
                                uint32_t T, uint32_t B, uint32_t I, uint32_t H, uint32_t O, uint32_t K)
    {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O; K_ = K;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint32_t)(T_ * B_ * I_));
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, (uint32_t)(B_ * H_));
        wI2HGm_.SetGlobalBuffer((__gm__ float*)w_i2h, (uint32_t)(H_ * K_));
        bI2HGm_.SetGlobalBuffer((__gm__ float*)b_i2h, (uint32_t)(H_));
        wH2OGm_.SetGlobalBuffer((__gm__ float*)w_h2o, (uint32_t)(O_ * H_));
        bH2OGm_.SetGlobalBuffer((__gm__ float*)b_h2o, (uint32_t)(O_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint32_t)(T_ * B_ * O_));

        // UB buffers
        const uint32_t maxVec = K_ > H_ ? (K_ > O_ ? K_ : O_) : (H_ > O_ ? H_ : O_);
        pipe_.InitBuffer(hBuf_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(inpBuf_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(wRowKBuf_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(tmpKBuf_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(wRowHBuf_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(tmpHBuf_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(yBuf_, 1, O_ * sizeof(float));
        pipe_.InitBuffer(scalarBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (B_ == 0 || T_ == 0) return;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t b = blockIdx; b < B_; b += blockNum) {
            // h state per batch
            AscendC::LocalTensor<float> hLocal = hBuf_.AllocTensor<float>();
            AscendC::DataCopy(hLocal, h0Gm_[b * H_], H_);

            // iterate timesteps
            for (uint32_t t = 0; t < T_; ++t) {
                // Build input vector [x_t(b), h]
                AscendC::LocalTensor<float> inpLocal = inpBuf_.AllocTensor<float>();
                // Copy x_t(b) [I]
                const uint32_t xOffset = t * (B_ * I_) + b * I_;
                AscendC::DataCopy(inpLocal, xGm_[xOffset], I_);
                // Append h [H]
                AscendC::DataCopy(inpLocal[I_], hLocal, H_);

                // Compute h = tanh(W_i2h * inp + b_i2h)
                for (uint32_t j = 0; j < H_; ++j) {
                    // Load weight row j: [K]
                    AscendC::LocalTensor<float> wRowK = wRowKBuf_.AllocTensor<float>();
                    AscendC::DataCopy(wRowK, wI2HGm_[j * K_], K_);

                    // tmp = wRowK * inpLocal
                    AscendC::LocalTensor<float> tmpK = tmpKBuf_.AllocTensor<float>();
                    AscendC::Mul(tmpK, wRowK, inpLocal, K_);
                    // Reduce to scalar sum in tmpK[0]
                    AscendC::ReduceSum(tmpK, tmpK, K_);

                    // Add bias b_i2h[j]
                    AscendC::LocalTensor<float> bScalar = scalarBuf_.AllocTensor<float>();
                    AscendC::DataCopy(bScalar, bI2HGm_[j], 1);
                    AscendC::Add(tmpK, tmpK, bScalar, 1);

                    // tanh
                    AscendC::Tanh(tmpK, tmpK, 1);

                    // Store into hLocal[j]
                    AscendC::DataCopy(hLocal[j], tmpK, 1);

                    tmpKBuf_.FreeTensor(tmpK);
                    wRowKBuf_.FreeTensor(wRowK);
                    scalarBuf_.FreeTensor(bScalar);
                }

                // y_t(b) = W_h2o * h + b_h2o
                AscendC::LocalTensor<float> yLocal = yBuf_.AllocTensor<float>();
                for (uint32_t o = 0; o < O_; ++o) {
                    AscendC::LocalTensor<float> wRowH = wRowHBuf_.AllocTensor<float>();
                    AscendC::DataCopy(wRowH, wH2OGm_[o * H_], H_);

                    AscendC::LocalTensor<float> tmpH = tmpHBuf_.AllocTensor<float>();
                    AscendC::Mul(tmpH, wRowH, hLocal, H_);
                    AscendC::ReduceSum(tmpH, tmpH, H_);

                    AscendC::LocalTensor<float> bScalar2 = scalarBuf_.AllocTensor<float>();
                    AscendC::DataCopy(bScalar2, bH2OGm_[o], 1);
                    AscendC::Add(tmpH, tmpH, bScalar2, 1);

                    // place scalar into yLocal[o]
                    AscendC::DataCopy(yLocal[o], tmpH, 1);

                    tmpHBuf_.FreeTensor(tmpH);
                    wRowHBuf_.FreeTensor(wRowH);
                    scalarBuf_.FreeTensor(bScalar2);
                }

                // write y at index [t, b, :]
                const uint32_t yOffset = t * (B_ * O_) + b * O_;
                AscendC::DataCopy(yGm_[yOffset], yLocal, O_);
                yBuf_.FreeTensor(yLocal);

                inpBuf_.FreeTensor(inpLocal);
            }
            hBuf_.FreeTensor(hLocal);
        }
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECCALC, 1> hBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> inpBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wRowKBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpKBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wRowHBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpHBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> wI2HGm_;
    AscendC::GlobalTensor<float> bI2HGm_;
    AscendC::GlobalTensor<float> wH2OGm_;
    AscendC::GlobalTensor<float> bH2OGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t T_{0}, B_{0}, I_{0}, H_{0}, O_{0}, K_{0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x,
                                                                GM_ADDR h0,
                                                                GM_ADDR w_i2h,
                                                                GM_ADDR b_i2h,
                                                                GM_ADDR w_h2o,
                                                                GM_ADDR b_h2o,
                                                                GM_ADDR y,
                                                                GM_ADDR workspace,
                                                                GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h, b_i2h, w_h2o, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I, tiling_data.H, tiling_data.O, tiling_data.K);
    op.Process();
}
