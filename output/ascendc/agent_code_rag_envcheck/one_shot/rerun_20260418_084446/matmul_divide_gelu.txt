host_operator_src="""

#include "matmul_divide_gelu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

// UB model for post-op (divide + gelu approx):
// Queues:
//   in: 2 * tile
//   out: 2 * tile
//   tmp1,tmp2,tmp3: 3 * tile
// Total floats: 7 * tile => bytes = 28 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulDivideGeluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    // x: [B, K], w: [N, K] -> y: [B, N]
    const int64_t rank_x = x_shape->GetStorageShape().GetDimNum();
    const int64_t rank_w = w_shape->GetStorageShape().GetDimNum();
    if (rank_x < 2 || rank_w < 2) {
        return ge::GRAPH_FAILED;
    }
    const int64_t B = x_shape->GetStorageShape().GetDim(0);
    const int64_t Kx = x_shape->GetStorageShape().GetDim(rank_x - 1);
    const int64_t N = w_shape->GetStorageShape().GetDim(0);
    const int64_t Kw = w_shape->GetStorageShape().GetDim(rank_w - 1);
    if (Kx != Kw || B <= 0 || N <= 0) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t totalLength = static_cast<uint32_t>(B * N);

    // Compute UB tile size for post-op
    uint32_t maxTile = UB_BUDGET_BYTES / 28; // float elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // matmul workspace if needed is managed internally
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

// Vector post-processing buffer counts
constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulDivideGeluCustom {
public:
    __aicore__ inline KernelMatmulDivideGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y,
                                uint32_t totalLength, uint32_t tileLength)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        // Set GM tensors for post-processing
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);
        divisorGm_ = (__gm__ float*)divisor;

        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue1_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue2_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue3_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void ProcessPost()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        // load divisor scalar
        float d = divisorGm_[0];
        invDiv_ = 1.0f / d;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueueY_.EnQue(yLocal);
    }

    // GELU approximation: 0.5 * x * (1 + tanh(sqrt(2/pi)*(x + 0.044715*x^3)))
    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> yIn = inQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> yOut = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> t1 = tmpQueue1_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2 = tmpQueue2_.AllocTensor<float>();
        AscendC::LocalTensor<float> t3 = tmpQueue3_.AllocTensor<float>();

        // x <- yIn / d
        AscendC::Muls(t1, yIn, invDiv_, len);                  // t1 = x

        // t2 = x^3
        AscendC::Mul(t2, t1, t1, len);                         // t2 = x^2
        AscendC::Mul(t2, t2, t1, len);                         // t2 = x^3

        // t3 = x + 0.044715 * x^3
        AscendC::Muls(t3, t2, 0.044715f, len);
        AscendC::Add(t3, t3, t1, len);

        // t2 = sqrt(2/pi) * t3
        AscendC::Muls(t2, t3, 0.7978845608028654f, len);       // sqrt(2/pi)

        // t3 = tanh(t2)
        AscendC::Tanh(t3, t2, len);

        // yOut = 0.5 * x * (1 + tanh(..))
        AscendC::Adds(t3, t3, 1.0f, len);                      // t3 = 1 + tanh
        AscendC::Mul(t2, t1, t3, len);                         // t2 = x * (1 + tanh)
        AscendC::Muls(yOut, t2, 0.5f, len);                    // yOut = 0.5 * ...

        tmpQueue3_.FreeTensor(t3);
        tmpQueue2_.FreeTensor(t2);
        tmpQueue1_.FreeTensor(t1);
        outQueueY_.EnQue(yOut);
        inQueueY_.FreeTensor(yIn);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue2_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue3_;

    AscendC::GlobalTensor<float> yGm_;
    __gm__ float* divisorGm_{nullptr};

    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    float invDiv_{1.0f};
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor,
                                                                GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    // 1) GEMM + Bias using Matmul kernel
    // Prepare a simple cube tiling environment if needed by registered Matmul object.
    // Define matmul kernel types: A,B inputs in GM, C and Bias in GM as float
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::BSNGD> A_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, true,  LayoutMode::BSNGD> B_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD> C_TYPE;
    typedef AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> BIAS_TYPE;

    BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE> batchMatmulKernel;
    AscendC::TPipe pipe;

    // Use a default tiling copied from GM (framework supplies)
    TCubeTiling cubeTiling;
    CopyTiling(&cubeTiling, tiling);
    cubeTiling.shareMode = 0;
    cubeTiling.shareL1Size = FULL_L1_SIZE;
    cubeTiling.shareL0CSize = FULL_L0C_SIZE;
    cubeTiling.shareUbSize = 0;

    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), batchMatmulKernel.matmulObj, &cubeTiling);
    batchMatmulKernel.Init(x, w, b, y, workspace, cubeTiling);
    // Process single run; batch dims resolved via tiling
    batchMatmulKernel.Process<true>(&pipe, cubeTiling.BatchNum, cubeTiling.BatchNum);

    // 2) Post-processing: divide by scalar and GELU
    GET_TILING_DATA(tilingData, tiling);
    KernelMatmulDivideGeluCustom post;
    post.Init(x, w, b, divisor, y, tilingData.totalLength, tilingData.tileLength);
    post.ProcessPost();
}
