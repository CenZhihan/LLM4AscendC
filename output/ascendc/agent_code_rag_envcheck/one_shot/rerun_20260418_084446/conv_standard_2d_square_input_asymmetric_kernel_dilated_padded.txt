#include "kernel_operator.h"

constexpr int32_t OUT_BUFFER_NUM = 2;

class KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelDilatedPaddedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t CIn, uint32_t H, uint32_t W,
                                uint32_t COut, uint32_t KH, uint32_t KW,
                                uint32_t HOut, uint32_t WOut,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padTop, uint32_t padBottom,
                                uint32_t padLeft, uint32_t padRight,
                                uint32_t dilationH, uint32_t dilationW,
                                uint32_t tilePixels)
    {
        N_ = N; CIn_ = CIn; H_ = H; W_ = W;
        COut_ = COut; KH_ = KH; KW_ = KW;
        HOut_ = HOut; WOut_ = WOut;
        strideH_ = strideH; strideW_ = strideW;
        padTop_ = padTop; padBottom_ = padBottom;
        padLeft_ = padLeft; padRight_ = padRight;
        dilationH_ = dilationH; dilationW_ = dilationW;
        tilePix_ = (tilePixels == 0u) ? 1u : tilePixels;

        // Set global buffers as 1-D arrays
        const uint64_t xElems = (uint64_t)N_ * CIn_ * H_ * W_;
        const uint64_t wElems = (uint64_t)COut_ * CIn_ * KH_ * KW_;
        const uint64_t yElems = (uint64_t)N_ * COut_ * HOut_ * WOut_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xElems);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wElems);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yElems);

        pipe_.InitBuffer(weightQueue_, 1, (uint32_t)(CIn_ * KH_ * KW_) * (uint32_t)sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUFFER_NUM, tilePix_ * (uint32_t)sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, (uint32_t)sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || COut_ == 0 || HOut_ == 0 || WOut_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint32_t combos = N_ * COut_;
        // Each block works on (n, oc) pairs in grid-stride
        for (uint32_t combo = blockIdx; combo < combos; combo += blockNum) {
            const uint32_t n = combo / COut_;
            const uint32_t oc = combo % COut_;
            ComputeOneNOc(n, oc);
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * CIn_ + c) * H_ + h) * W_ + w;
    }
    __aicore__ inline uint64_t OffsetW(uint32_t oc, uint32_t c, uint32_t kh, uint32_t kw) const
    {
        return (((uint64_t)oc * CIn_ + c) * KH_ + kh) * KW_ + kw;
    }
    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t oc, uint32_t oh, uint32_t ow) const
    {
        return (((uint64_t)n * COut_ + oc) * HOut_ + oh) * WOut_ + ow;
    }

    __aicore__ inline void LoadWeightSlice(uint32_t oc)
    {
        const uint32_t slice = CIn_ * KH_ * KW_;
        AscendC::LocalTensor<float> wLocal = weightQueue_.AllocTensor<float>();
        const uint64_t base = (uint64_t)oc * slice;
        AscendC::DataCopy(wLocal, wGm_[base], slice);
        weightQueue_.EnQue(wLocal);
    }

    __aicore__ inline float LoadXScalar(uint64_t gmOffset, AscendC::LocalTensor<float>& tmp)
    {
        // Copy single element from GM to local scalar and return
        AscendC::DataCopy(tmp, xGm_[gmOffset], 1);
        // Direct element access on local tensor (assumed supported by AscendC indexing)
        return tmp.GetValue(0);
    }

    __aicore__ inline void ComputeOneNOc(uint32_t n, uint32_t oc)
    {
        const uint32_t totalPixels = HOut_ * WOut_;
        if (totalPixels == 0) return;

        // Load weight slice for this oc once
        LoadWeightSlice(oc);
        AscendC::LocalTensor<float> wLocal = weightQueue_.DeQue<float>();

        uint32_t processed = 0;
        while (processed < totalPixels) {
            uint32_t cur = totalPixels - processed;
            if (cur > tilePix_) cur = tilePix_;

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Dup(yLocal, 0.0f, cur);

            // Allocate a reusable single-scalar buffer for x loads
            AscendC::LocalTensor<float> tmpScalar = tmpQueue_.AllocTensor<float>();

            for (uint32_t i = 0; i < cur; ++i) {
                const uint32_t outIndex = processed + i;
                const uint32_t oh = outIndex / WOut_;
                const uint32_t ow = outIndex % WOut_;

                float acc = 0.0f;
                // Iterate over CIn, KH, KW
                for (uint32_t c = 0; c < CIn_; ++c) {
                    for (uint32_t kh = 0; kh < KH_; ++kh) {
                        int32_t ih = (int32_t)oh * (int32_t)strideH_ - (int32_t)padTop_ + (int32_t)kh * (int32_t)dilationH_;
                        if (ih < 0 || ih >= (int32_t)H_) continue;
                        for (uint32_t kw = 0; kw < KW_; ++kw) {
                            int32_t iw = (int32_t)ow * (int32_t)strideW_ - (int32_t)padLeft_ + (int32_t)kw * (int32_t)dilationW_;
                            if (iw < 0 || iw >= (int32_t)W_) continue;

                            const uint64_t xOff = OffsetX(n, c, (uint32_t)ih, (uint32_t)iw);
                            const uint64_t wOff = OffsetW(oc, c, kh, kw);
                            // Read x scalar
                            float xv = LoadXScalar(xOff, tmpScalar);
                            // Read weight from local buffer
                            float wv = wLocal.GetValue((uint32_t)wOff - (uint32_t)OffsetW(oc, 0, 0, 0)); // offset within slice
                            acc += xv * wv;
                        }
                    }
                }
                yLocal.SetValue(i, acc);
            }

            tmpQueue_.FreeTensor(tmpScalar);

            // Write out contiguous spatial tile for (n, oc)
            const uint64_t yBase = ((uint64_t)n * COut_ + oc) * (uint64_t)(HOut_ * WOut_) + processed;
            AscendC::DataCopy(yGm_[yBase], yLocal, cur);

            outQueueY_.FreeTensor(yLocal);
            processed += cur;
        }

        weightQueue_.FreeTensor(wLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> weightQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, CIn_{0}, H_{0}, W_{0};
    uint32_t COut_{0}, KH_{0}, KW_{0};
    uint32_t HOut_{0}, WOut_{0};
    uint32_t strideH_{1}, strideW_{1};
    uint32_t padTop_{0}, padBottom_{0}, padLeft_{0}, padRight_{0};
    uint32_t dilationH_{1}, dilationW_{1};
    uint32_t tilePix_{1};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_dilated_padded_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    op.Init(
        x, weight, y,
        tiling_data.N, tiling_data.CIn, tiling_data.H, tiling_data.W,

        tiling_data.HOut, tiling_data.WOut,
        tiling_data.strideH, tiling_data.strideW,

        tiling_data.padLeft, tiling_data.padRight,
        tiling_data.dilationH, tiling_data.dilationW,

    );

}
