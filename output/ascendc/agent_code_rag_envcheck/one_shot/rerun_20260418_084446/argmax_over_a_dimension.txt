static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape &origin = x_shape->GetOriginShape();

    const int32_t rank = static_cast<int32_t>(origin.GetDimNum());
    if (rank <= 0) {
        return GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *attrDim = attrs->GetAttrPointer<int32_t>(0);

    int32_t dim = (attrDim != nullptr) ? *attrDim : 0;
    if (dim < 0) dim += rank;
    if (dim < 0 || dim >= rank) {
        return GRAPH_FAILED;

    // Build output shape = input shape with axis 'dim' removed
    gert::Shape *y_shape = context->GetOutputShape(0);

    out_dims.reserve(static_cast<size_t>(rank - 1));

    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t axisLen, uint32_t inner)
    {
        outer_ = outer;
        axisLen_ = axisLen;
        inner_ = inner;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)outer_ * axisLen_ * inner_);
        yGm_.SetGlobalBuffer((__gm__ int32_t*)y, (uint64_t)outer_ * inner_);

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, inner_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, inner_ * sizeof(int32_t));
        pipe_.InitBuffer(tmpFloatQueue_, 1, inner_ * sizeof(float));     // best values
        pipe_.InitBuffer(tmpIntQueue_, 1, inner_ * sizeof(int32_t));      // best indices
        pipe_.InitBuffer(tmpIntAuxQueue_, 1, inner_ * sizeof(int32_t));   // row index vector
        pipe_.InitBuffer(tmpMaskQueue_, 1, inner_ * sizeof(uint8_t));     // mask
    }

    __aicore__ inline void Process()
    {
        if (outer_ == 0 || inner_ == 0 || axisLen_ == 0) {
            return;
        }

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t o = blockIdx; o < outer_; o += blockNum) {
            const uint64_t baseX = (uint64_t)o * axisLen_ * inner_;
            const uint64_t baseY = (uint64_t)o * inner_;

            // Allocate persistent locals for this slice
            AscendC::LocalTensor<float> bestVal = tmpFloatQueue_.AllocTensor<float>();
            AscendC::LocalTensor<int32_t> bestIdx = tmpIntQueue_.AllocTensor<int32_t>();

            // Initialize with row 0
            LoadRowToQueue(baseX);
            AscendC::LocalTensor<float> row0 = inQueueX_.DeQue<float>();
            AscendC::DataCopy(bestVal, row0, inner_);
            inQueueX_.FreeTensor(row0);

            AscendC::Duplicate(bestIdx, (int32_t)0, inner_);

            // Iterate remaining rows
            for (uint32_t r = 1; r < axisLen_; ++r) {
                const uint64_t rowOff = baseX + (uint64_t)r * inner_;
                LoadRowToQueue(rowOff);
                AscendC::LocalTensor<float> row = inQueueX_.DeQue<float>();

                AscendC::LocalTensor<uint8_t> mask = tmpMaskQueue_.AllocTensor<uint8_t>();
                AscendC::Greater(mask, row, bestVal, inner_);

                // Update best values
                AscendC::Select(bestVal, mask, row, bestVal, inner_);

                // Prepare row index vector and update indices
                AscendC::LocalTensor<int32_t> idxVec = tmpIntAuxQueue_.AllocTensor<int32_t>();
                AscendC::Duplicate(idxVec, (int32_t)r, inner_);
                AscendC::Select(bestIdx, mask, idxVec, bestIdx, inner_);

                // Free temporaries
                tmpIntAuxQueue_.FreeTensor(idxVec);
                tmpMaskQueue_.FreeTensor(mask);
                inQueueX_.FreeTensor(row);
            }

            // Write indices to GM
            AscendC::LocalTensor<int32_t> yLocal = outQueueY_.AllocTensor<int32_t>();
            AscendC::DataCopy(yLocal, bestIdx, inner_);
            outQueueY_.EnQue(yLocal);

            AscendC::LocalTensor<int32_t> yOut = outQueueY_.DeQue<int32_t>();
            AscendC::DataCopy(yGm_[baseY], yOut, inner_);
            outQueueY_.FreeTensor(yOut);

            // Free persistent locals
            tmpIntQueue_.FreeTensor(bestIdx);
            tmpFloatQueue_.FreeTensor(bestVal);
        }
    }

private:
    __aicore__ inline void LoadRowToQueue(uint64_t gmOffset)
    {
        AscendC::LocalTensor<float> rowLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(rowLocal, xGm_[gmOffset], inner_);
        inQueueX_.EnQue(rowLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpFloatQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpIntQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpIntAuxQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMaskQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int32_t> yGm_;

    uint32_t outer_{0};
    uint32_t axisLen_{0};
    uint32_t inner_{0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    op.Init(x, y, tiling_data.outer, tiling_data.axisLen, tiling_data.inner);
    op.Process();
}

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    // Output dtype is int32 in kernel; return as-is for performance.
    // Caller can cast to long if needed.
    std::vector<int64_t> in_sizes = self.sizes().vec();
    if (dim < 0) dim += static_cast<int64_t>(in_sizes.size());
    TORCH_CHECK(dim >= 0 && dim < static_cast<int64_t>(in_sizes.size()), "Invalid dim in argmax custom");

    for (size_t i = 0; i < in_sizes.size(); ++i) {
        if (static_cast<int64_t>(i) == dim) continue;
        out_sizes.push_back(in_sizes[i]);
