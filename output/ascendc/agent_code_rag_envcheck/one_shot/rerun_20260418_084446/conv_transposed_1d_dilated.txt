namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model for this kernel:
//   yLocal: 1 * tile
//   xLocal: 1 * tile (max overlap per k)
//   tmpLocal: 1 * tile (for w*x)
// Total floats ~= 3 * tile => bytes ~= 12 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // DMA-friendly alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed1dDilatedCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t Lin = static_cast<uint32_t>(x_shape->GetDim(2));

    const uint32_t W_Cin = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Cout  = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t K     = static_cast<uint32_t>(w_shape->GetDim(2));

    if (W_Cin != Cin) {
        return ge::GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *strideAttr   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *paddingAttr  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilationAttr = attrs->GetAttrPointer<int32_t>(2);

    const int32_t stride   = (strideAttr   != nullptr) ? *strideAttr   : 1;
    const int32_t padding  = (paddingAttr  != nullptr) ? *paddingAttr  : 0;
    const int32_t dilation = (dilationAttr != nullptr) ? *dilationAttr : 1;

    // Output length for conv_transpose1d (no output_padding):
    // L_out = (L_in - 1) * stride - 2*padding + dilation * (K - 1) + 1
    const uint32_t Lout = static_cast<uint32_t>(
        (static_cast<int64_t>(Lin) - 1) * stride - 2 * padding + dilation * static_cast<int32_t>(K - 1) + 1
    );

    // Tile length from UB budget: bytes ~= 12 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 12;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (Lout < tileLen) {
        const uint32_t aligned = AlignDown(Lout, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : Lout;
    }

    // grid over (N, Cout)
    const uint64_t totalPairs64 = static_cast<uint64_t>(N) * static_cast<uint64_t>(Cout);
    const uint32_t totalPairs = static_cast<uint32_t>(totalPairs64 > 0xFFFFFFFFULL ? 0xFFFFFFFFu : totalPairs64);

    // Choose a conservative block dim
    uint32_t blockDim = (totalPairs == 0u) ? 1u : totalPairs;
    if (blockDim < 8u) blockDim = totalPairs; // small cases: one core per pair if possible
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_lin(Lin);
    tiling.set_lout(Lout);
    tiling.set_kernelSize(K);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_dilation(dilation);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // no workspace
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 3 || w_shape->GetDimNum() != 3) {
        return GRAPH_FAILED;
    }

    const int64_t N   = x_shape->GetDim(0);
    const int64_t Cin = x_shape->GetDim(1);
    const int64_t Lin = x_shape->GetDim(2);

    const int64_t W_Cin = w_shape->GetDim(0);
    const int64_t Cout  = w_shape->GetDim(1);
    const int64_t K     = w_shape->GetDim(2);

    if (W_Cin != Cin) {
        return GRAPH_FAILED;
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *strideAttr   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *paddingAttr  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilationAttr = attrs->GetAttrPointer<int32_t>(2);

    const int32_t stride   = (strideAttr   != nullptr) ? *strideAttr   : 1;
    const int32_t padding  = (paddingAttr  != nullptr) ? *paddingAttr  : 0;
    const int32_t dilation = (dilationAttr != nullptr) ? *dilationAttr : 1;

    const int64_t Lout = (Lin - 1) * stride - 2 * padding + dilation * (K - 1) + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);

}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed1dDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed1dDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t lout,
                                uint32_t kernelSize, uint32_t tileLength,
                                int32_t stride, int32_t padding, int32_t dilation)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        lin_ = lin; lout_ = lout; k_ = kernelSize;
        tileLen_ = tileLength;
        stride_ = stride; padding_ = padding; dilation_ = dilation;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * cin_ * lin_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, cin_ * cout_ * k_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * cout_ * lout_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpScalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || lin_ == 0 || lout_ == 0 || k_ == 0 || tileLen_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalPairs = static_cast<uint64_t>(n_) * static_cast<uint64_t>(cout_);
        if (totalPairs == 0) return;

        // Grid-stride loop over (n, oc) pairs
        for (uint64_t pair = blockIdx; pair < totalPairs; pair += blockNum) {
            const uint32_t nIdx = static_cast<uint32_t>(pair / cout_);
            const uint32_t oc   = static_cast<uint32_t>(pair % cout_);

            const uint64_t xBase = static_cast<uint64_t>(nIdx) * cin_ * lin_;
            const uint64_t yBase = static_cast<uint64_t>(nIdx) * cout_ * lout_ + static_cast<uint64_t>(oc) * lout_;

            // Sweep output length in tiles
            for (uint32_t oBase = 0; oBase < lout_; oBase += tileLen_) {
                uint32_t curLen = lout_ - oBase;
                if (curLen > tileLen_) curLen = tileLen_;

                // yLocal buffer: zero initialize
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, curLen);

                // Accumulate contributions from all (ic, k)
                for (uint32_t ic = 0; ic < cin_; ++ic) {
                    const uint64_t xChanBase = xBase + static_cast<uint64_t>(ic) * lin_;

                    for (uint32_t kk = 0; kk < k_; ++kk) {
                        // shift = -padding + kk * dilation
                        const int32_t shift = -padding_ + static_cast<int32_t>(kk) * dilation_;

                        // For general stride=1 fast path (recommended). For stride>1, we handle mapping exactly.
                        if (stride_ == 1) {
                            // Determine input overlap range mapping into current y tile
                            // in_low in [0, lin_); out_pos = in_pos + shift
                            int32_t in_low  = static_cast<int32_t>(oBase) - shift;
                            int32_t in_high = static_cast<int32_t>(oBase + curLen) - shift;

                            if (in_low < 0) in_low = 0;
                            if (in_high > static_cast<int32_t>(lin_)) in_high = static_cast<int32_t>(lin_);

                            uint32_t run = (in_high > in_low) ? static_cast<uint32_t>(in_high - in_low) : 0u;
                            if (run == 0) continue;

                            // Load x segment
                            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[xChanBase + static_cast<uint64_t>(in_low)], run);

                            // Load weight scalar w(ic, oc, kk)
                            const uint64_t wOff = static_cast<uint64_t>(ic) * cout_ * k_ + static_cast<uint64_t>(oc) * k_ + kk;
                            AscendC::LocalTensor<float> wTmp = tmpScalarQueue_.AllocTensor<float>();
                            AscendC::DataCopy(wTmp, wGm_[wOff], 1);

                            // Read scalar value from local buffer
                            float wVal = wTmp.GetValue(0);

                            tmpScalarQueue_.FreeTensor(wTmp);

                            // tLocal = xLocal * wVal
                            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                            AscendC::Muls(tLocal, xLocal, wVal, run);

                            // yLocal[dest_off : dest_off + run] += tLocal
                            const uint32_t destOff = static_cast<uint32_t>(static_cast<int32_t>(in_low) + shift - static_cast<int32_t>(oBase));
                            AscendC::Add(yLocal[destOff], yLocal[destOff], tLocal, run);

                            tmpQueue_.FreeTensor(tLocal);
                            inQueueX_.FreeTensor(xLocal);
                        } else {
                            // General stride s >= 1
                            // in_pos must satisfy out_pos = in_pos * s + shift
                            // For out tile [oBase, oBase + curLen):
                            // in_pos in [ceil((oBase - shift)/s), floor(((oBase+curLen-1) - shift)/s)]
                            int64_t low_num = static_cast<int64_t>(oBase) - static_cast<int64_t>(shift);
                            int64_t high_num = static_cast<int64_t>(oBase + curLen - 1) - static_cast<int64_t>(shift);

                            int32_t in_low = static_cast<int32_t>((low_num + (stride_ - 1)) / stride_); // ceil_div
                            int32_t in_high = static_cast<int32_t>(high_num / stride_) + 1; // inclusive -> exclusive

                            if (in_low < 0) in_low = 0;
                            if (in_high > static_cast<int32_t>(lin_)) in_high = static_cast<int32_t>(lin_);
                            if (in_high <= in_low) continue;

                            // Load weight scalar
                            const uint64_t wOff = static_cast<uint64_t>(ic) * cout_ * k_ + static_cast<uint64_t>(oc) * k_ + kk;
                            AscendC::LocalTensor<float> wTmp = tmpScalarQueue_.AllocTensor<float>();
                            AscendC::DataCopy(wTmp, wGm_[wOff], 1);
                            float wVal = wTmp.GetValue(0);
                            tmpScalarQueue_.FreeTensor(wTmp);

                            // Process in chunks to stay within xLocal capacity
                            uint32_t pos = static_cast<uint32_t>(in_low);
                            uint32_t end = static_cast<uint32_t>(in_high);
                            while (pos < static_cast<uint32_t>(end)) {
                                uint32_t remaining = end - pos;
                                uint32_t seg = remaining;
                                if (seg > tileLen_) seg = tileLen_;

                                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                                AscendC::DataCopy(xLocal, xGm_[xChanBase + pos], seg);

                                // Multiply by scalar
                                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
                                AscendC::Muls(tLocal, xLocal, wVal, seg);

                                // Scatter into yLocal according to stride
                                // Each in_pos -> out_pos = in_pos * s + shift
                                for (uint32_t i = 0; i < seg; ++i) {
                                    int32_t out_pos = static_cast<int32_t>(pos + i) * stride_ + shift;
                                    uint32_t off = static_cast<uint32_t>(out_pos - static_cast<int32_t>(oBase));
                                    // yLocal[off] += tLocal[i]
                                    float acc = yLocal.GetValue(off);
                                    acc += tLocal.GetValue(i);
                                    yLocal.SetValue(off, acc);
                                }

                                tmpQueue_.FreeTensor(tLocal);
                                inQueueX_.FreeTensor(xLocal);
                                pos += seg;
                            }
                        }
                    } // kk
                } // ic

                // Write back yLocal tile
                AscendC::DataCopy(yGm_[yBase + oBase], yLocal, curLen);
                outQueueY_.FreeTensor(yLocal);
            } // oBase
        } // pair loop
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpScalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t lin_{0}, lout_{0}, k_{0};
    uint32_t tileLen_{0};
    int32_t stride_{1}, padding_{0}, dilation_{1};
};

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1dDilatedCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.lout,
            tiling_data.kernelSize, tiling_data.tileLength,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation);
    op.Process();
}
