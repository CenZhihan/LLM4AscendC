static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core UB budget
static constexpr uint32_t ALIGN_W = 256;                 // W tile alignment for efficient DMA
static constexpr uint32_t MIN_W  = 64;                   // minimum W tile

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dSquareInputSquareKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect NCHW for x and OIHW for weight
    const uint32_t N  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t CIn  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t COut = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Kh   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw   = static_cast<uint32_t>(w_shape->GetDim(3));

    // stride=(1,1), padding=(0,0), dilation=(1,1), groups=1
    const uint32_t OH = (H >= Kh) ? (H - Kh + 1u) : 0u;
    const uint32_t OW = (W >= Kw) ? (W - Kw + 1u) : 0u;

    // Tile parameters
    uint32_t tileCo = (COut < 8u) ? COut : 8u;                // small oc tile for UB reuse
    if (tileCo == 0u) tileCo = 1u;

    // Memory model (floats):
    //   yAcc: tileCo * tileW
    //   tLocal: tileW
    //   zeroVec: tileW
    //   inQueue: tileW
    //   outQueue: tileW
    //   wTile: tileCo * CIn * Kh * Kw
    // Total float elements = tileCo*tileW + 4*tileW + tileCo*CIn*Kh*Kw
    const uint32_t availFloats = UB_BUDGET_BYTES / 4u;
    const uint32_t wTileConst = tileCo * CIn * Kh * Kw;
    uint32_t tileW = 0u;
    if (availFloats > wTileConst) {
        tileW = (availFloats - wTileConst) / (tileCo + 4u);
    }
    // alignment and bounds
    if (tileW > OW) tileW = OW;
    if (tileW > 0) tileW = AlignDown(tileW, ALIGN_W);
    if (tileW < MIN_W) tileW = (OW < MIN_W) ? OW : MIN_W;
    if (tileW == 0) tileW = OW;  // handle OW small

    // Task decomposition: each task = (n, ocTileIdx, oh)
    const uint32_t numOcTiles = CeilDiv(COut, tileCo);
    const uint32_t totalTasks = N * numOcTiles * OH;

    // blockDim selection
    uint32_t blockDim = totalTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_CIn(CIn);
    tiling.set_COut(COut);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);
    tiling.set_OH(OH);
    tiling.set_OW(OW);

    tiling.set_tileCo(tileCo);
    tiling.set_tileW(tileW);
    tiling.set_numOcTiles(numOcTiles);
    tiling.set_totalTasks(totalTasks);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard2dS1P0K3 {
public:
    __aicore__ inline KernelConvStandard2dS1P0K3() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t CIn, uint32_t COut,
                                uint32_t H, uint32_t W,
                                uint32_t Kh, uint32_t Kw,
                                uint32_t OH, uint32_t OW,
                                uint32_t tileCo, uint32_t tileW,
                                uint32_t numOcTiles, uint32_t totalTasks)
    {
        N_ = N; CIn_ = CIn; COut_ = COut;
        H_ = H; W_ = W; Kh_ = Kh; Kw_ = Kw;
        OH_ = OH; OW_ = OW;
        tileCo_ = tileCo; tileW_ = tileW;
        numOcTiles_ = numOcTiles; totalTasks_ = totalTasks;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * CIn_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, COut_ * CIn_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * COut_ * OH_ * OW_);

        // Queues and buffers
        pipe_.InitBuffer(inQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, 1, tileW_ * sizeof(float));

        // calcBuf layout:
        //   [0 ... tileCo_*tileW_-1]         -> yAcc
        //   [tileCo_*tileW_ ... +tileW_-1]   -> tLocal
        //   [tileCo_*tileW_ + tileW_ ... +tileW_-1] -> zeroVec
        //   [ ... weights tile (tileCo_*CIn_*Kh_*Kw_) ]
        const uint32_t yAccElems = tileCo_ * tileW_;
        const uint32_t tElems = tileW_;
        const uint32_t zElems = tileW_;
        const uint32_t wTileElems = tileCo_ * CIn_ * Kh_ * Kw_;
        const uint32_t totalElems = yAccElems + tElems + zElems + wTileElems;
        pipe_.InitBuffer(calcBuf_, 1, totalElems * sizeof(float));
        yAcc_     = calcBuf_.Get<AscendC::LocalTensor<float>>();
        tLocal_   = calcBuf_.GetWithOffset<float>(tElems, yAccElems * sizeof(float));
        zeroVec_  = calcBuf_.GetWithOffset<float>(zElems, (yAccElems + tElems) * sizeof(float));
        wTileBuf_ = calcBuf_.GetWithOffset<float>(wTileElems, (yAccElems + tElems + zElems) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (OH_ == 0 || OW_ == 0 || COut_ == 0 || CIn_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // task: (n, ocTileIdx, oh)
        for (uint32_t task = blockIdx; task < totalTasks_; task += blockNum) {
            uint32_t tmp = task;
            const uint32_t n  = tmp / (numOcTiles_ * OH_);
            tmp -= n * (numOcTiles_ * OH_);
            const uint32_t ocTileIdx = tmp / OH_;
            const uint32_t oh = tmp - ocTileIdx * OH_;

            const uint32_t ocBase = ocTileIdx * tileCo_;
            const uint32_t curCo = (ocBase + tileCo_ <= COut_) ? tileCo_ : (COut_ - ocBase);
            if (curCo == 0) continue;

            // Prefetch weight tile for this oc tile (contiguous in GM)
            const uint32_t wTileElems = curCo * CIn_ * Kh_ * Kw_;
            const uint32_t wGmOffset = ocBase * CIn_ * Kh_ * Kw_;
            AscendC::DataCopy(wTileBuf_.TemplateSlice<float>(0), wGm_[wGmOffset], wTileElems);

            // Process all W tiles on this row
            for (uint32_t owBase = 0; owBase < OW_; owBase += tileW_) {
                const uint32_t curW = (owBase + tileW_ <= OW_) ? tileW_ : (OW_ - owBase);
                if (curW == 0) continue;

                // Create zeroVec by reading the first needed input vector and subtracting itself
                // First combination indices for bootstrap
                uint32_t ic0 = 0, kh0 = 0, kw0 = 0;
                // Load input vector for bootstrap
                {
                    const uint32_t xGmOffset0 = (((n * CIn_ + ic0) * H_ + (oh + kh0)) * W_) + (owBase + kw0);
                    AscendC::LocalTensor<float> inTmp = inQueue_.AllocTensor<float>();
                    AscendC::DataCopy(inTmp, xGm_[xGmOffset0], curW);
                    inQueue_.EnQue(inTmp);
                }
                AscendC::LocalTensor<float> inLocal0 = inQueue_.DeQue<float>();
                AscendC::Sub(zeroVec_, inLocal0, inLocal0, curW);

                // yAcc zero init for each oc in the tile
                for (uint32_t ocOff = 0; ocOff < curCo; ++ocOff) {
                    AscendC::LocalTensor<float> yAccOc = yAcc_.TemplateSlice<float>(ocOff * tileW_);
                    AscendC::Add(yAccOc, zeroVec_, zeroVec_, curW);
                }

                // Accumulate for bootstrap (ic0, kh0, kw0) using inLocal0
                {
                    for (uint32_t ocOff = 0; ocOff < curCo; ++ocOff) {
                        const uint32_t wIdx = (((ocOff * CIn_ + ic0) * Kh_ + kh0) * Kw_) + kw0;
                        // Load scalar weight from local weight tile buffer
                        // Assuming LocalTensor provides access to underlying buffer address
                        float wScalar = *(reinterpret_cast<float*>(wTileBuf_.GetPhyAddr()) + wIdx);
                        AscendC::Muls(tLocal_, inLocal0, wScalar, curW);
                        AscendC::LocalTensor<float> yAccOc = yAcc_.TemplateSlice<float>(ocOff * tileW_);
                        AscendC::Add(yAccOc, yAccOc, tLocal_, curW);
                    }
                }
                inQueue_.FreeTensor(inLocal0);

                // Remaining combinations
                for (uint32_t ic = 0; ic < CIn_; ++ic) {
                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                            if (ic == 0 && kh == 0 && kw == 0) {
                                continue; // already processed
                            }
                            // Load input vector for this combination
                            const uint32_t xGmOffset = (((n * CIn_ + ic) * H_ + (oh + kh)) * W_) + (owBase + kw);
                            AscendC::LocalTensor<float> inTmp = inQueue_.AllocTensor<float>();
                            AscendC::DataCopy(inTmp, xGm_[xGmOffset], curW);
                            inQueue_.EnQue(inTmp);

                            AscendC::LocalTensor<float> inLocal = inQueue_.DeQue<float>();
                            // Accumulate over oc tile
                            for (uint32_t ocOff = 0; ocOff < curCo; ++ocOff) {
                                const uint32_t wIdx = (((ocOff * CIn_ + ic) * Kh_ + kh) * Kw_) + kw;
                                float wScalar = *(reinterpret_cast<float*>(wTileBuf_.GetPhyAddr()) + wIdx);
                                AscendC::Muls(tLocal_, inLocal, wScalar, curW);
                                AscendC::LocalTensor<float> yAccOc = yAcc_.TemplateSlice<float>(ocOff * tileW_);
                                AscendC::Add(yAccOc, yAccOc, tLocal_, curW);
                            }
                            inQueue_.FreeTensor(inLocal);
                        }
                    }
                }

                // Write back y for each oc in the tile
                for (uint32_t ocOff = 0; ocOff < curCo; ++ocOff) {
                    AscendC::LocalTensor<float> yTmp = outQueue_.AllocTensor<float>();
                    AscendC::LocalTensor<float> yAccOc = yAcc_.TemplateSlice<float>(ocOff * tileW_);
                    // Copy yAccOc to yTmp via add with zeroVec
                    AscendC::Add(yTmp, yAccOc, zeroVec_, curW);
                    outQueue_.EnQue(yTmp);

                    const uint32_t oc = ocBase + ocOff;
                    const uint32_t yGmOffset = (((n * COut_ + oc) * OH_ + oh) * OW_) + owBase;
                    AscendC::LocalTensor<float> yLocalOut = outQueue_.DeQue<float>();
                    AscendC::DataCopy(yGm_[yGmOffset], yLocalOut, curW);
                    outQueue_.FreeTensor(yLocalOut);
                }
            } // for owBase
        } // for task
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    // Slices into calcBuf_
    AscendC::LocalTensor<float> yAcc_;
    AscendC::LocalTensor<float> tLocal_;
    AscendC::LocalTensor<float> zeroVec_;
    AscendC::LocalTensor<float> wTileBuf_;

    // shapes
    uint32_t N_, CIn_, COut_;
    uint32_t H_, W_, Kh_, Kw_, OH_, OW_;
    uint32_t tileCo_, tileW_;
    uint32_t numOcTiles_;
    uint32_t totalTasks_;
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling_data, tilingGm);

    KernelConvStandard2dS1P0K3 kernel;
    kernel.Init(
        x, weight, y,
        tiling_data.N, tiling_data.CIn, tiling_data.COut,
        tiling_data.H, tiling_data.W,
        tiling_data.Kh, tiling_data.Kw,
        tiling_data.OH, tiling_data.OW,
        tiling_data.tileCo, tiling_data.tileW,
        tiling_data.numOcTiles, tiling_data.totalTasks
    );
    kernel.Process();
}
