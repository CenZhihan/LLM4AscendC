static inline uint32_t CeilDiv(uint64_t a, uint32_t b)

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMishMishCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const bool hasBias = (context->GetNumInputs() > 2) && (context->GetInputShape(2)->GetOriginShape().GetShapeSize() > 0);

    // Expect 4D NCHW for x, and 4D [Cout, Cin, Kh, Kw] for weight
    const uint64_t N  = static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(0));
    const uint64_t Cin = static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(1));
    const uint64_t H  = static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(2));
    const uint64_t W  = static_cast<uint64_t>(x_shape->GetOriginShape().GetDim(3));

    const uint64_t Cout = static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(0));
    const uint64_t CinW = static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(1));
    const uint64_t Kh   = static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(2));
    const uint64_t Kw   = static_cast<uint64_t>(w_shape->GetOriginShape().GetDim(3));

    // Basic checks (stride=1, padding=0, dilation=1 implied by kernel)
    if (CinW != Cin) {
        return ge::GRAPH_FAILED;
    }
    if (H < Kh || W < Kw) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t Ho = H - Kh + 1;
    const uint64_t Wo = W - Kw + 1;
    const uint64_t totalOut = N * Cout * Ho * Wo;

    // Choose a conservative blockDim
    uint32_t blockDim = static_cast<uint32_t>(totalOut);
    if (blockDim == 0u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_Cin(static_cast<uint32_t>(Cin));
    tiling.set_Cout(static_cast<uint32_t>(Cout));
    tiling.set_H(static_cast<uint32_t>(H));
    tiling.set_W(static_cast<uint32_t>(W));
    tiling.set_Kh(static_cast<uint32_t>(Kh));
    tiling.set_Kw(static_cast<uint32_t>(Kw));
    tiling.set_Ho(static_cast<uint32_t>(Ho));
    tiling.set_Wo(static_cast<uint32_t>(Wo));
    tiling.set_HasBias(hasBias ? 1u : 0u);
    tiling.set_TotalOut(totalOut);
    tiling.set_BlockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dMishMishCustom {
public:
    __aicore__ inline KernelConv2dMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo, uint32_t hasBias, uint64_t totalOut)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo; hasBias_ = (hasBias != 0);
        totalOut_ = totalOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_*Cin_*H_*W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cout_*Cin_*Kh_*Kw_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_*Cout_*Ho_*Wo_);

        // init a small calc buffer for scalar temporaries
        pipe_.InitBuffer(calcQ_, 6, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t idx = blockIdx; idx < totalOut_; idx += blockNum) {
            uint64_t t = idx;
            const uint32_t ow = static_cast<uint32_t>(t % Wo_); t /= Wo_;
            const uint32_t oh = static_cast<uint32_t>(t % Ho_); t /= Ho_;
            const uint32_t co = static_cast<uint32_t>(t % Cout_); t /= Cout_;
            const uint32_t n  = static_cast<uint32_t>(t);

            // Local scalars as 1-length vectors
            AscendC::LocalTensor<float> acc = calcQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> xLoc = calcQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> wLoc = calcQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmp0 = calcQ_.AllocTensor<float>(); // for mul or intermediate
            AscendC::LocalTensor<float> tmp1 = calcQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmp2 = calcQ_.AllocTensor<float>();

            AscendC::Dup(acc, 0.0f, 1);

            // Direct convolution, stride=1, pad=0, dilation=1
            for (uint32_t ic = 0; ic < Cin_; ++ic) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const uint32_t ih = oh + kh;
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const uint32_t iw = ow + kw;

                        const uint64_t xOff =
                            (((uint64_t)n * Cin_ + ic) * H_ + ih) * W_ + iw;
                        const uint64_t wOff =
                            (((uint64_t)co * Cin_ + ic) * Kh_ + kh) * Kw_ + kw;

                        AscendC::DataCopy(xLoc, xGm_[xOff], 1);
                        AscendC::DataCopy(wLoc, wGm_[wOff], 1);
                        AscendC::Mul(tmp0, xLoc, wLoc, 1);
                        AscendC::Add(acc, acc, tmp0, 1);
                    }
                }
            }

            if (hasBias_) {
                AscendC::DataCopy(wLoc, bGm_[co], 1);
                AscendC::Add(acc, acc, wLoc, 1);
            }

            // Mish(x) = x * tanh(ln(1 + exp(x))) applied twice
            // First Mish
            AscendC::Exp(tmp0, acc, 1);                 // tmp0 = exp(acc)
            AscendC::Adds(tmp1, tmp0, 1.0f, 1);         // tmp1 = 1 + exp(acc)
            AscendC::Ln(tmp2, tmp1, 1);                 // tmp2 = ln(1 + exp(acc))
            AscendC::Tanh(tmp1, tmp2, 1);               // tmp1 = tanh(softplus)
            AscendC::Mul(tmp2, acc, tmp1, 1);           // tmp2 = mish(acc)

            // Second Mish
            AscendC::Exp(tmp0, tmp2, 1);                // tmp0 = exp(mish1)
            AscendC::Adds(tmp1, tmp0, 1.0f, 1);         // tmp1 = 1 + exp(mish1)
            AscendC::Ln(tmp0, tmp1, 1);                 // tmp0 = ln(1 + exp(mish1))
            AscendC::Tanh(tmp1, tmp0, 1);               // tmp1 = tanh(softplus(mish1))
            AscendC::Mul(acc, tmp2, tmp1, 1);           // acc = mish(mish(acc))

            const uint64_t yOff =
                (((uint64_t)n * Cout_ + co) * Ho_ + oh) * Wo_ + ow;
            AscendC::DataCopy(yGm_[yOff], acc, 1);

            // Free locals
            calcQ_.FreeTensor(tmp2);
            calcQ_.FreeTensor(tmp1);
            calcQ_.FreeTensor(tmp0);
            calcQ_.FreeTensor(wLoc);
            calcQ_.FreeTensor(xLoc);
            calcQ_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 6> calcQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    bool hasBias_{false};
    uint64_t totalOut_{0};
};

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y,
                                                              GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMishMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo, tiling_data.HasBias, tiling_data.TotalOut);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv2d_mish_mish_impl_npu(const at::Tensor& x,
                                     const at::Tensor& weight,
                                     c10::optional<at::Tensor> bias_opt) {
    at::Tensor result;
    auto out_sizes = x.sizes().vec();
    // Compute output shape: [N, Cout, Ho, Wo]
    int64_t N = x.size(0);
    int64_t Cin = x.size(1);
    int64_t H = x.size(2);
    int64_t W = x.size(3);
    int64_t Cout = weight.size(0);
    int64_t Kh = weight.size(2);
    int64_t Kw = weight.size(3);
    TORCH_CHECK(weight.size(1) == Cin, "Weight Cin mismatch");
    TORCH_CHECK(H >= Kh && W >= Kw, "Input smaller than kernel");
    int64_t Ho = H - Kh + 1;
    int64_t Wo = W - Kw + 1;
    result = at::empty({N, Cout, Ho, Wo}, x.options());

    if (bias_opt.has_value()) {
        auto bias = bias_opt.value();
        EXEC_NPU_CMD(aclnnConv2dMishMishCustom, x, weight, bias, result);
    } else {
        at::Tensor bias; // undefined tensor
        EXEC_NPU_CMD(aclnnConv2dMishMishCustom, x, weight, bias, result);
    }
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_mish_mish_custom", &conv2d_mish_mish_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_mish_mish_custom", &conv2d_mish_mish_impl_npu,
