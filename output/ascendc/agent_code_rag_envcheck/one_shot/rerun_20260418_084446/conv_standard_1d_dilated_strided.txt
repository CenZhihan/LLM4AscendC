#include "conv_standard_1d_dilated_strided_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core conservative budget
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard1dDilatedStridedCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape->GetDimNum() != 3 || w_shape->GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t Lin = static_cast<uint32_t>(x_shape->GetDim(2));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Cin_w= static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t K    = static_cast<uint32_t>(w_shape->GetDim(2));
    if (Cin != Cin_w) return ge::GRAPH_FAILED;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stride   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *padding  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilation = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *groups   = attrs->GetAttrPointer<int32_t>(3);

    if (*groups != 1) {
        // This kernel supports only groups=1
        return ge::GRAPH_FAILED;
    }

    const uint32_t effKw = static_cast<uint32_t>((*dilation) * (K - 1) + 1);
    if (Lin + 2 * (*padding) < effKw) return ge::GRAPH_FAILED;
    const uint32_t Lout = static_cast<uint32_t>((static_cast<int64_t>(Lin) + 2LL * (*padding) - effKw) / (*stride) + 1);

    // UB usage per tile (float32):
    //   y: tileOut
    //   x: tileOut
    //   tmp: tileOut
    //   weights for one Cout: Cin * K
    // Total floats = 3 * tileOut + Cin * K
    const uint32_t ubFloatsBudget = UB_BUDGET_BYTES / 4;
    uint32_t maxTileOut = 0;
    if (ubFloatsBudget > (Cin * K)) {
        maxTileOut = (ubFloatsBudget - (Cin * K)) / 3u;
    } else {
        maxTileOut = ALIGN_ELEMS; // fallback minimal
    }

    maxTileOut = AlignDown(maxTileOut, ALIGN_ELEMS);
    if (maxTileOut < ALIGN_ELEMS) maxTileOut = ALIGN_ELEMS;
    if (maxTileOut > HARD_CAP_ELEMS) maxTileOut = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileOut = maxTileOut;
    if (Lout < tileOut) {
        const uint32_t aligned = AlignDown(Lout, ALIGN_ELEMS);
        tileOut = (aligned > 0) ? aligned : Lout;
    }

    // Blocks map to (n, cout) pairs in a grid-stride way
    const uint32_t pairCount = N * Cout;
    uint32_t blockDim = pairCount;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_Lin(Lin);
    tiling.set_Lout(Lout);
    tiling.set_K(K);
    tiling.set_stride(static_cast<uint32_t>(*stride));
    tiling.set_padding(static_cast<uint32_t>(*padding));
    tiling.set_dilation(static_cast<uint32_t>(*dilation));
    tiling.set_groups(static_cast<uint32_t>(*groups));
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stride   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *padding  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilation = attrs->GetAttrPointer<int32_t>(2);
    const int32_t *groups   = attrs->GetAttrPointer<int32_t>(3);

    if (x_shape->GetDimNum() != 3 || w_shape->GetDimNum() != 3) return GRAPH_FAILED;
    const int64_t N   = x_shape->GetDim(0);
    const int64_t Cin = x_shape->GetDim(1);
    const int64_t Lin = x_shape->GetDim(2);

    const int64_t Cout = w_shape->GetDim(0);
    const int64_t CinW = w_shape->GetDim(1);
    const int64_t K    = w_shape->GetDim(2);
    if (Cin != CinW) return GRAPH_FAILED;
    if (*groups != 1) return GRAPH_FAILED;

    const int64_t effKw = (*dilation) * (K - 1) + 1;
    const int64_t Lout = (Lin + 2 * (*padding) - effKw) / (*stride) + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Lout);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvStandard1dDilatedStridedCustom {
public:
    __aicore__ inline KernelConvStandard1dDilatedStridedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Lin, uint32_t Lout, uint32_t K,
                                uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t groups, uint32_t tileOut)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Lin_ = Lin; Lout_ = Lout; K_ = K;
        stride_ = stride; padding_ = padding; dilation_ = dilation;
        groups_ = groups; tileOut_ = tileOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * Cin_ * Lin_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, Cout_ * Cin_ * K_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * Cout_ * Lout_);

        // UB buffers
        pipe_.InitBuffer(xQueue_, 1, tileOut_ * sizeof(float));   // input slice (per k,cin)
        pipe_.InitBuffer(yQueue_, 1, tileOut_ * sizeof(float));   // output slice
        pipe_.InitBuffer(tmpQueue_, 1, tileOut_ * sizeof(float)); // temp for mul
        pipe_.InitBuffer(wQueue_, 1, (Cin_ * K_) * sizeof(float)); // weights per cout
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || Lout_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t pairCount = N_ * Cout_;
        for (uint32_t pair = blockIdx; pair < pairCount; pair += blockNum) {
            const uint32_t n = pair / Cout_;
            const uint32_t co = pair % Cout_;

            // Load weights for this output channel co into UB once
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            const uint32_t wBase = co * (Cin_ * K_);
            AscendC::DataCopy(wLocal, wGm_[wBase], Cin_ * K_);

            // Compute output along length in tiles
            for (uint32_t p0 = 0; p0 < Lout_; p0 += tileOut_) {
                uint32_t curLen = Lout_ - p0;
                if (curLen > tileOut_) curLen = tileOut_;

                AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

                AscendC::Duplicate(yLocal, 0.0f, curLen);

                // base input offset for this batch
                const uint32_t xBaseN = n * (Cin_ * Lin_);

                // Accumulate over Cin and K
                for (uint32_t ci = 0; ci < Cin_; ++ci) {
                    const uint32_t xBaseNC = xBaseN + ci * Lin_;
                    for (uint32_t k = 0; k < K_; ++k) {
                        // Gather a strided slice: positions p = p0..p0+curLen-1
                        // Input index for each p: inIdx = p * stride_ + k * dilation_ - padding_ (padding=0 typical)
                        const uint32_t firstIn = xBaseNC + (p0 * stride_) + (k * dilation_);
                        // Perform 2D-like copy: nBurst=curLen, burstLen=1, srcStride=stride_-1, dstStride=0
                        // This copies elements: firstIn + i*stride_ into contiguous xLocal[i]
                        AscendC::DataCopy(xLocal, xGm_[firstIn], 1, curLen, (stride_ > 0 ? stride_ - 1 : 0), 0);

                        // Multiply by weight scalar and accumulate
                        const float wVal = wLocal[ci * K_ + k];
                        AscendC::Muls(tLocal, xLocal, wVal, curLen);
                        AscendC::Add(yLocal, yLocal, tLocal, curLen);
                    }
                }

                // Write back
                const uint32_t yBase = n * (Cout_ * Lout_) + co * Lout_ + p0;
                AscendC::DataCopy(yGm_[yBase], yLocal, curLen);

                tmpQueue_.FreeTensor(tLocal);
                xQueue_.FreeTensor(xLocal);
                yQueue_.FreeTensor(yLocal);
            }

            wQueue_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Lin_{0}, Lout_{0}, K_{0};
    uint32_t stride_{1}, padding_{0}, dilation_{1}, groups_{1};
    uint32_t tileOut_{0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dDilatedStridedCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.Lin, tiling_data.Lout, tiling_data.K,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation,
            tiling_data.groups, tiling_data.tileOut);
    op.Process();
}
