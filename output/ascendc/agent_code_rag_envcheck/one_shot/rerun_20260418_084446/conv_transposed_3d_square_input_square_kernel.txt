{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() != 5 || w_shape->GetOriginShape().GetDimNum() != 5) {
        return ge::GRAPH_FAILED;

    const auto &x_dims = x_shape->GetOriginShape().GetDims();
    const auto &w_dims = w_shape->GetOriginShape().GetDims();

    const uint32_t Cin = static_cast<uint32_t>(x_dims[1]);
    const uint32_t inD = static_cast<uint32_t>(x_dims[2]);
    const uint32_t inH = static_cast<uint32_t>(x_dims[3]);
    const uint32_t inW = static_cast<uint32_t>(x_dims[4]);

    const uint32_t W_Cout = static_cast<uint32_t>(w_dims[1]);
    const uint32_t kD = static_cast<uint32_t>(w_dims[2]);
    const uint32_t kH = static_cast<uint32_t>(w_dims[3]);
    const uint32_t kW = static_cast<uint32_t>(w_dims[4]);

        return ge::GRAPH_FAILED;
    }

    // Transposed conv with stride=1, padding=0, dilation=1, output_padding=0
    const uint32_t outD = inD + kD - 1u;
    const uint32_t outH = inH + kH - 1u;
    const uint32_t outW = inW + kW - 1u;

    const uint64_t totalOutLen64 = static_cast<uint64_t>(N) * W_Cout * outD * outH * outW;
    if (totalOutLen64 == 0) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t totalOutLen = static_cast<uint32_t>(totalOutLen64); // Assume fits

    // UB tile for y buffer; additional scratch is tiny (<= few KB), so approximate bytes ~ 4 * tileLen
    uint32_t maxTile = (UB_BUDGET_BYTES / 4);
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalOutLen < tileLen) {
        const uint32_t aligned = AlignDown(totalOutLen, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalOutLen;

    const uint32_t numTiles = CeilDiv(totalOutLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    ConvTransposed3dSquareInputSquareKernelCustomTilingData tiling;
    tiling.set_totalOutLength(totalOutLen);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.set_cin(Cin);
    tiling.set_cout(W_Cout);
    tiling.set_inD(inD);
    tiling.set_inH(inH);
    tiling.set_inW(inW);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t OUT_BUFFER_NUM = 2;

// Naive transposed 3D convolution kernel specialized for stride=1, padding=0, dilation=1.
// Weight layout: [Cin, Cout, kD, kH, kW]
// Input layout:  [N, Cin, D, H, W]
// Output layout: [N, Cout, D + kD - 1, H + kH - 1, W + kW - 1]
class KernelConvTransposed3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalOutLength, uint32_t tileLength,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t inD, uint32_t inH, uint32_t inW,
                                uint32_t kD, uint32_t kH, uint32_t kW)
    {
        totalOutLen_ = totalOutLength;
        tileLen_ = tileLength;

        n_ = n;
        cin_ = cin;
        cout_ = cout;
        inD_ = inD;
        inH_ = inH;
        inW_ = inW;
        kD_ = kD;
        kH_ = kH;
        kW_ = kW;

        outD_ = inD_ + kD_ - 1u;
        outH_ = inH_ + kH_ - 1u;
        outW_ = inW_ + kW_ - 1u;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * cin_ * inD_ * inH_ * inW_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(cin_ * cout_ * kD_ * kH_ * kW_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);

        pipe_.InitBuffer(outQueueY_, OUT_BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpScalarQ_, 2, sizeof(float));              // for scalar x and w
        pipe_.InitBuffer(tmpMulQ_, 1, sizeof(float));                 // for intermediate mul
        // Optional: weights block buffer (not mandatory for scalar path)
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutLen_) return;

        while (true) {
            uint32_t curLen = totalOutLen_ - base;
            if (curLen > tileLen_) curLen = tileLen_;

            ComputeTile(base, curLen);
            StoreTile(base, curLen);

            const uint32_t nextBase = base + strideElems;
            if (nextBase >= totalOutLen_) break;
            base = nextBase;
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t outBase, uint32_t len)
    {
        // Allocate output local buffer
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        // Initialize to zeros
        AscendC::Duplicate(yLocal, 0.0f, len);

        // Scalar temporaries
        AscendC::LocalTensor<float> tmpX = tmpScalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpW = tmpScalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> tmpMul = tmpMulQ_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            const uint64_t gIdx = static_cast<uint64_t>(outBase) + i;

            uint32_t ow = static_cast<uint32_t>(gIdx % outW_);
            uint64_t t1 = gIdx / outW_;
            uint32_t oh = static_cast<uint32_t>(t1 % outH_);
            uint64_t t2 = t1 / outH_;
            uint32_t od = static_cast<uint32_t>(t2 % outD_);
            uint64_t t3 = t2 / outD_;
            uint32_t oc = static_cast<uint32_t>(t3 % cout_);
            uint32_t n = static_cast<uint32_t>(t3 / cout_);

            // Accumulate over input channels and 3x3x3 kernel
            for (uint32_t ic = 0; ic < cin_; ++ic) {
                // Iterate over kernel spatial indices
                for (uint32_t kd = 0; kd < kD_; ++kd) {
                    // input depth index
                    int32_t id = static_cast<int32_t>(od) - static_cast<int32_t>(kd);
                    if (id < 0 || id >= static_cast<int32_t>(inD_)) continue;

                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        int32_t ih = static_cast<int32_t>(oh) - static_cast<int32_t>(kh);
                        if (ih < 0 || ih >= static_cast<int32_t>(inH_)) continue;

                        for (uint32_t kw = 0; kw < kW_; ++kw) {
                            int32_t iw = static_cast<int32_t>(ow) - static_cast<int32_t>(kw);
                            if (iw < 0 || iw >= static_cast<int32_t>(inW_)) continue;

                            const uint64_t xIndex =
                                (((static_cast<uint64_t>(n) * cin_ + ic) * inD_ + static_cast<uint32_t>(id)) * inH_ +
                                 static_cast<uint32_t>(ih)) * inW_ + static_cast<uint32_t>(iw);

                            const uint64_t wIndex =
                                (((static_cast<uint64_t>(ic) * cout_ + oc) * kD_ + kd) * kH_ + kh) * kW_ + kw;

                            // Load scalar x and w from GM and accumulate: y[i] += x * w
                            AscendC::DataCopy(tmpX, xGm_[xIndex], 1);
                            AscendC::DataCopy(tmpW, wGm_[wIndex], 1);

                            AscendC::Mul(tmpMul, tmpX, tmpW, 1);

                            // yLocal[i] += tmpMul[0]
                            AscendC::LocalTensor<float> ySlice = yLocal[i]; // 1-element view
                            AscendC::Add(ySlice, ySlice, tmpMul, 1);
                        }
                    }
                }
            }
        }

        tmpMulQ_.FreeTensor(tmpMul);
        tmpScalarQ_.FreeTensor(tmpW);
        tmpScalarQ_.FreeTensor(tmpX);

        outQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void StoreTile(uint32_t outBase, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[outBase], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> tmpScalarQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpMulQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLen_{0};
    uint32_t tileLen_{0};

    uint32_t n_{0};
    uint32_t cin_{0};
    uint32_t cout_{0};
    uint32_t inD_{0};
    uint32_t inH_{0};
    uint32_t inW_{0};
    uint32_t kD_{0};
    uint32_t kH_{0};
    uint32_t kW_{0};

    uint32_t outD_{0};
    uint32_t outH_{0};
    uint32_t outW_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputSquareKernelCustom op;
    op.Init(
        x, weight, y,
        tiling_data.totalOutLength, tiling_data.tileLength,
        tiling_data.n, tiling_data.cin, tiling_data.cout,
        tiling_data.inD, tiling_data.inH, tiling_data.inW,
        tiling_data.kD, tiling_data.kH, tiling_data.kW
    );
    op.Process();
}
