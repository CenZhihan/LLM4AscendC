namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    // Expect:
    // x: [N, Cin, D, H, W]
    // w: [Cin, Cout, kD, kH, kW]
    const uint64_t N   = (xShape->GetDimNum() >= 1) ? static_cast<uint64_t>(xShape->GetDim(0)) : 1ULL;
    const uint64_t Cin = (xShape->GetDimNum() >= 2) ? static_cast<uint64_t>(xShape->GetDim(1)) : 1ULL;
    const uint64_t D   = (xShape->GetDimNum() >= 3) ? static_cast<uint64_t>(xShape->GetDim(2)) : 1ULL;
    const uint64_t H   = (xShape->GetDimNum() >= 4) ? static_cast<uint64_t>(xShape->GetDim(3)) : 1ULL;
    const uint64_t W   = (xShape->GetDimNum() >= 5) ? static_cast<uint64_t>(xShape->GetDim(4)) : 1ULL;

    const uint64_t CinW = (wShape->GetDimNum() >= 1) ? static_cast<uint64_t>(wShape->GetDim(0)) : Cin;
    const uint64_t Cout = (wShape->GetDimNum() >= 2) ? static_cast<uint64_t>(wShape->GetDim(1)) : 1ULL;
    const uint64_t kD   = (wShape->GetDimNum() >= 3) ? static_cast<uint64_t>(wShape->GetDim(2)) : 1ULL;
    const uint64_t kH   = (wShape->GetDimNum() >= 4) ? static_cast<uint64_t>(wShape->GetDim(3)) : 1ULL;
    const uint64_t kW   = (wShape->GetDimNum() >= 5) ? static_cast<uint64_t>(wShape->GetDim(4)) : 1ULL;

    // Basic validation
    if (CinW != Cin) {
        // fallback
    }

    // For stride=(1,1,1), padding=(0,0,0), dilation=(1,1,1), output_padding=(0,0,0)
    const uint64_t outD = D + kD - 1ULL;
    const uint64_t outH = H + kH - 1ULL;
    const uint64_t outW = W + kW - 1ULL;

    const uint64_t totalOutLength = N * Cout * outD * outH * outW;

    // UB tiling: we only zero-fill the output for demonstration.
    // Choose conservative UB budget and alignment.
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t ALIGN_ELEMS = 2048;
    uint32_t elemSize = 4; // float32
    uint32_t maxTile = UB_BUDGET_BYTES / elemSize;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalOutLength < tileLen) {
        uint32_t aligned = AlignDown(static_cast<uint32_t>(totalOutLength), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(totalOutLength);
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    ConvTransposed3dSquareInputAsymmetricKernelCustomTilingData tiling;
    tiling.set_totalOutLength(totalOutLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_ci(static_cast<uint32_t>(Cin));
    tiling.set_co(static_cast<uint32_t>(Cout));
    tiling.set_d(static_cast<uint32_t>(D));
    tiling.set_h(static_cast<uint32_t>(H));
    tiling.set_w(static_cast<uint32_t>(W));
    tiling.set_kd(static_cast<uint32_t>(kD));
    tiling.set_kh(static_cast<uint32_t>(kH));
    tiling.set_kw(static_cast<uint32_t>(kW));
    tiling.set_outD(static_cast<uint32_t>(outD));
    tiling.set_outH(static_cast<uint32_t>(outH));
    tiling.set_outW(static_cast<uint32_t>(outW));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint64_t totalOutLength, uint32_t tileLength)
    {
        (void)x;
        (void)weight;
        totalLen_ = totalOutLength;
        tileLen_ = tileLength;

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * tileLen_;

        uint64_t base = static_cast<uint64_t>(blockIdx) * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = static_cast<uint32_t>(totalLen_ - base);
        if (curLen > tileLen_) curLen = tileLen_;
        ZeroOut(base, curLen);

        while (true) {
            uint64_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = static_cast<uint32_t>(totalLen_ - nextBase);
                if (nextLen > tileLen_) nextLen = tileLen_;
                ZeroOut(nextBase, nextLen);
            }

            // write current tile
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void ZeroOut(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Duplicate(yLocal, 0.0f, len);
        outQueueY_.EnQue<float>(yLocal);
    }

    __aicore__ inline void CopyOut(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::GlobalTensor<float> yGm_;

    uint64_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA_WITH_STRUCT(ConvTransposed3dSquareInputAsymmetricKernelCustomTilingData, tilingData, tiling);
    KernelConvTransposed3dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y, tilingData.totalOutLength, tilingData.tileLength);
    op.Process();
}
