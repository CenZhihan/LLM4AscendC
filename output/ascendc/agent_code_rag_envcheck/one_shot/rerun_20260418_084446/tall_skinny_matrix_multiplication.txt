host_operator_src="""

#include "tall_skinny_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

// UB usage model (per core, per inner compute):
//  - aRow: K floats
//  - bRow: tileN floats
//  - cRow: tileN floats
//  - tmpRow: tileN floats
// Total floats = K + 3 * tileN
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;     // conservative
static constexpr uint32_t ALIGN_ELEMS = 128;                // vector-friendly alignment
static constexpr uint32_t MIN_TILE_N = 16;
static constexpr uint32_t MAX_TILE_N = 4096;
static constexpr uint32_t DEFAULT_TILE_M = 256;             // rows per tile handled sequentially

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t M64 = a_shape->GetOriginShape().GetDim(0);
    const uint64_t K64 = a_shape->GetOriginShape().GetDim(1);
    const uint64_t BK64 = b_shape->GetOriginShape().GetDim(0);
    const uint64_t N64 = b_shape->GetOriginShape().GetDim(1);
    if (K64 != BK64) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(M64);
    const uint32_t K = static_cast<uint32_t>(K64);
    const uint32_t N = static_cast<uint32_t>(N64);

    // Compute tileN from UB budget:
    // bytes = 4 * (K + 3 * tileN) <= UB_BUDGET_BYTES
    uint32_t maxTileN = 0;
    if (UB_BUDGET_BYTES > 0) {
        const int64_t budgetElems = static_cast<int64_t>(UB_BUDGET_BYTES / 4);
        int64_t tn = (budgetElems - static_cast<int64_t>(K)) / 3;
        if (tn < static_cast<int64_t>(MIN_TILE_N)) tn = MIN_TILE_N;
        if (tn > static_cast<int64_t>(MAX_TILE_N)) tn = MAX_TILE_N;
        maxTileN = static_cast<uint32_t>(tn);
    }
    uint32_t tileN = AlignDown(maxTileN, ALIGN_ELEMS);
    if (tileN < MIN_TILE_N) tileN = MIN_TILE_N;
    if (tileN > N) tileN = (N >= MIN_TILE_N) ? AlignDown(N, MIN_TILE_N) : N;

    // tileM is a scheduling unit, not UB-bound here (we process row-by-row)
    uint32_t tileM = DEFAULT_TILE_M;
    if (tileM > M) tileM = M;

    const uint32_t numRowTiles = CeilDiv(M, tileM);
    const uint32_t numColTiles = CeilDiv(N, tileN);
    uint32_t totalTiles = numRowTiles * numColTiles;
    if (totalTiles == 0) totalTiles = 1;

    // Choose blockDim conservatively
    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    if (a_shape->GetOriginShape().GetDimNum() != 2 || b_shape->GetOriginShape().GetDimNum() != 2) {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t INBUF_A_NUM = 1;
constexpr int32_t INBUF_B_NUM = 2;
constexpr int32_t OUTBUF_NUM = 2;
constexpr int32_t TMPBUF_NUM = 1;

class KernelTallSkinnyMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelTallSkinnyMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t K, uint32_t N,
                                uint32_t tileM, uint32_t tileN)
    {
        M_ = M;
        K_ = K;
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;

        aGm_.SetGlobalBuffer((__gm__ float*)a, (uint64_t)M_ * (uint64_t)K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)K_ * (uint64_t)N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, (uint64_t)M_ * (uint64_t)N_);

        // Buffers: aRow (K), bRow (tileN), cRow (tileN), tmpRow (tileN)
        pipe_.InitBuffer(inQueueA_, INBUF_A_NUM, K_ * sizeof(float));
        pipe_.InitBuffer(inQueueB_, INBUF_B_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(outQueueC_, OUTBUF_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMPBUF_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) {
            return;
        }

        const uint32_t numRowTiles = CeilDiv(M_, tileM_);
        const uint32_t numColTiles = CeilDiv(N_, tileN_);
        const uint32_t totalTiles  = numRowTiles * numColTiles;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t tileIdx = blockIdx; tileIdx < totalTiles; tileIdx += blockNum) {
            const uint32_t rowTile = tileIdx / numColTiles;
            const uint32_t colTile = tileIdx % numColTiles;

            const uint32_t rowBase = rowTile * tileM_;
            const uint32_t colBase = colTile * tileN_;

            uint32_t mLen = M_ - rowBase;
            if (mLen > tileM_) mLen = tileM_;
            uint32_t nLen = N_ - colBase;
            if (nLen > tileN_) nLen = tileN_;

            // Process rows in this tile sequentially
            for (uint32_t r = 0; r < mLen; ++r) {
                const uint32_t aRowGmOff = (rowBase + r) * K_;        // A[row, :]
                const uint32_t cRowGmOff = (rowBase + r) * N_ + colBase; // C[row, colBase:colBase+nLen]

                // Load A row (K elements)
                AscendC::LocalTensor<float> aRowLocal = inQueueA_.AllocTensor<float>();
                AscendC::DataCopy(aRowLocal, aGm_[aRowGmOff], K_);
                inQueueA_.EnQue(aRowLocal);

                // Initialize cRow local
                AscendC::LocalTensor<float> cRowLocal = outQueueC_.AllocTensor<float>();
                AscendC::SetValue(cRowLocal, 0.0f, nLen);

                aRowLocal = inQueueA_.DeQue<float>();

                // Accumulate over K
                for (uint32_t k = 0; k < K_; ++k) {
                    // Load B row segment: B[k, colBase:colBase+nLen]
                    const uint32_t bRowGmOff = k * N_ + colBase;
                    AscendC::LocalTensor<float> bRowLocal = inQueueB_.AllocTensor<float>();
                    AscendC::DataCopy(bRowLocal, bGm_[bRowGmOff], nLen);
                    inQueueB_.EnQue(bRowLocal);

                    bRowLocal = inQueueB_.DeQue<float>();

                    float aVal = aRowLocal.GetValue(k);
                    AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::Muls(tmpLocal, bRowLocal, aVal, nLen);
                    AscendC::Add(cRowLocal, cRowLocal, tmpLocal, nLen);

                    tmpQueue_.FreeTensor(tmpLocal);
                    inQueueB_.FreeTensor(bRowLocal);
                }

                inQueueA_.FreeTensor(aRowLocal);

                // Store cRow into GM
                AscendC::DataCopy(cGm_[cRowGmOff], cRowLocal, nLen);
                outQueueC_.FreeTensor(cRowLocal);
            }
        }
    }

private:
    __aicore__ inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
        return (a + b - 1u) / b;
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INBUF_A_NUM> inQueueA_;
    AscendC::TQue<AscendC::TPosition::VECIN, INBUF_B_NUM> inQueueB_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUTBUF_NUM> outQueueC_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMPBUF_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0};
    uint32_t K_{0};
    uint32_t N_{0};
    uint32_t tileM_{0};
    uint32_t tileN_{0};
};

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTallSkinnyMatrixMultiplicationCustom op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.K, tiling_data.N,
            tiling_data.tileM, tiling_data.tileN);
    op.Process();
}
