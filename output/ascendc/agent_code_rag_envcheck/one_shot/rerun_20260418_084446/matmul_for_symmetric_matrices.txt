host_operator_src="""

#include "matmul_for_symmetric_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForSymmetricMatricesCustomTilingData tiling;

    // Expect both A and B to be square (N x N)
    const auto *shapeA = context->GetInputShape(0);
    const auto *shapeB = context->GetInputShape(1);

    if (shapeA->GetOriginShape().GetDimNum() != 2 ||
        shapeB->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t nA0 = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(0));
    const uint32_t nA1 = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(1));
    const uint32_t nB0 = static_cast<uint32_t>(shapeB->GetOriginShape().GetDim(0));
    const uint32_t nB1 = static_cast<uint32_t>(shapeB->GetOriginShape().GetDim(1));

    if (!(nA0 == nA1 && nB0 == nB1 && nA1 == nB0)) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t N = nA0;

    // Heuristics for tiles, tuned conservatively for UB footprint and DMA efficiency.
    // UB usage per thread (approx):
    //   yLocal: tileN floats
    //   bRow  : tileN floats
    //   aElem : 1 float
    // Total    ~ (2 * tileN + 1) * 4 bytes
    // We choose tileN up to 1024 (aligned to 256) and tileK = 64 (aligned to 16).
    uint32_t tileN = 1024;
    const uint32_t ALIGN_N = 256;
    tileN = AlignDown(tileN, ALIGN_N);
    if (tileN == 0) tileN = ALIGN_N;
    if (N < tileN) {
        uint32_t aligned = AlignDown(N, ALIGN_N);
        tileN = (aligned > 0) ? aligned : N;
    }

    uint32_t tileK = 64;
    const uint32_t ALIGN_K = 16;
    tileK = AlignDown(tileK, ALIGN_K);
    if (tileK == 0) tileK = ALIGN_K;
    if (N < tileK) {
        uint32_t aligned = AlignDown(N, ALIGN_K);
        tileK = (aligned > 0) ? aligned : N;
    }

    // Block Dim: map blocks to rows, each block handles rowsPerBlock rows.
    // Conservative upper bound for stability.
    uint32_t blockDim = 24u;
    if (blockDim > N) blockDim = N;
    if (blockDim < 8u) blockDim = 8u;

    uint32_t rowsPerBlock = CeilDiv(N, blockDim);
    if (rowsPerBlock == 0) rowsPerBlock = 1;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 ||
        b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t nA0 = a_shape->GetOriginShape().GetDim(0);
    const int64_t nA1 = a_shape->GetOriginShape().GetDim(1);
    const int64_t nB0 = b_shape->GetOriginShape().GetDim(0);
    const int64_t nB1 = b_shape->GetOriginShape().GetDim(1);
    if (!(nA0 == nA1 && nB0 == nB1 && nA1 == nB0)) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, nA0);
    c_shape->SetDim(1, nB1);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != ge::DT_FLOAT || b_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelMatmulForSymmetricMatricesCustom {
public:
    __aicore__ inline KernelMatmulForSymmetricMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileN, uint32_t tileK, uint32_t rowsPerBlock)
    {
        N_ = N;
        tileN_ = tileN;
        tileK_ = tileK;
        rowsPerBlock_ = rowsPerBlock;

        const uint32_t total_elems = N_ * N_;
        aGm_.SetGlobalBuffer((__gm__ float*)A, total_elems);
        bGm_.SetGlobalBuffer((__gm__ float*)B, total_elems);
        cGm_.SetGlobalBuffer((__gm__ float*)C, total_elems);

        // Buffers:
        // yLocal: tileN
        // bRow  : tileN
        // aElem : 1
        pipe_.InitBuffer(bufY_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(bufB_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(bufA_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t rowStart = blockIdx * rowsPerBlock_;
        uint32_t rowEnd = rowStart + rowsPerBlock_;
        if (rowStart >= N_) return;
        if (rowEnd > N_) rowEnd = N_;

        for (uint32_t i = rowStart; i < rowEnd; ++i) {
            for (uint32_t j0 = 0; j0 < N_; j0 += tileN_) {
                const uint32_t curN = AscendC::min(tileN_, N_ - j0);

                AscendC::LocalTensor<float> yLocal = bufY_.AllocTensor<float>();
                // Zero yLocal
                AscendC::Muls(yLocal, yLocal, 0.0f, curN);

                // Accumulate over K
                for (uint32_t k0 = 0; k0 < N_; k0 += tileK_) {
                    const uint32_t curK = AscendC::min(tileK_, N_ - k0);

                    for (uint32_t kt = 0; kt < curK; ++kt) {
                        // Load scalar A[i, k]
                        const uint32_t k = k0 + kt;
                        AscendC::LocalTensor<float> aElem = bufA_.AllocTensor<float>();
                        AscendC::DataCopy(aElem, aGm_[i * N_ + k], 1);
                        float aScalar = aElem.GetValue(0);
                        bufA_.FreeTensor(aElem);

                        // Load row segment of B[k, j0 : j0+curN]
                        AscendC::LocalTensor<float> bRow = bufB_.AllocTensor<float>();
                        AscendC::DataCopy(bRow, bGm_[k * N_ + j0], curN);

                        // yLocal += aScalar * bRow
                        AscendC::Muls(bRow, bRow, aScalar, curN);
                        AscendC::Add(yLocal, yLocal, bRow, curN);

                        bufB_.FreeTensor(bRow);
                    }
                }

                // Store C[i, j0 : j0+curN]
                AscendC::DataCopy(cGm_[i * N_ + j0], yLocal, curN);
                bufY_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> bufY_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1>  bufB_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bufA_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t tileN_{0};
    uint32_t tileK_{0};
    uint32_t rowsPerBlock_{1};
};

extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForSymmetricMatricesCustom op;
    op.Init(A, B, C,
            tiling_data.n, tiling_data.tileN, tiling_data.tileK, tiling_data.rowsPerBlock);
    op.Process();
}
