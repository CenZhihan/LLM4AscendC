host_operator_src="""

#include "matmul_with_irregular_shapes_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB budget (conservative)
static constexpr uint32_t ALIGN_N = 256;                 // align N tiles for better DMA
static constexpr uint32_t ALIGN_K = 8;                   // align K tiles for vector ops
static constexpr uint32_t HARD_CAP_N = 4096;             // avoid overly large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithIrregularShapesCustomTilingData tiling;

    const gert::Shape *ashape = context->GetInputShape(0);
    const gert::Shape *bshape = context->GetInputShape(1);

    if (ashape->GetOriginShape().GetDimNum() < 2 || bshape->GetOriginShape().GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(ashape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(ashape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(bshape->GetOriginShape().GetDim(1));

    // Choose a small kTile to control UB usage and latency.
    uint32_t kTile = 16u;
    if (kTile > K) {
        kTile = (K == 0) ? 1u : K;
    }
    // Align down kTile if possible
    kTile = (kTile >= ALIGN_K) ? AlignDown(kTile, ALIGN_K) : kTile;

    // UB usage (floats) for this kernel:
    //   B_tile: double buffer 2 * (kTile * nTile)
    //   c_row: 1 * nTile
    //   a_row: 1 * kTile
    //   tmp:   1 * nTile
    // Total = 2*kTile*nTile + nTile + kTile + nTile = 2*kTile*nTile + 2*nTile + kTile
    const uint32_t budgetF = UB_BUDGET_BYTES / 4u;
    uint32_t nTileMax = 0;
    if (2u * kTile + 2u > 0u) {
        if (budgetF > kTile) {
            nTileMax = (budgetF - kTile) / (2u * kTile + 2u);
        } else {
            nTileMax = ALIGN_N; // fallback minimal
        }
    }
    if (nTileMax < ALIGN_N) nTileMax = ALIGN_N;             // minimal to favor DMA
    if (nTileMax > HARD_CAP_N) nTileMax = HARD_CAP_N;

    // Align nTile
    uint32_t nTile = AlignDown(nTileMax, ALIGN_N);
    if (nTile == 0) nTile = (N < ALIGN_N) ? N : ALIGN_N;
    if (nTile > N) nTile = (N >= ALIGN_N) ? AlignDown(N, ALIGN_N) : N;
    if (nTile == 0) nTile = N;

    // Conservative blockDim based on M
    uint32_t blockDim = M;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileN(nTile);
    tiling.set_tileK(kTile);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM_B = 2;

class KernelMatmulWithIrregularShapesCustom {
public:
    __aicore__ inline KernelMatmulWithIrregularShapesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN, uint32_t tileK)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tN_ = tileN;
        tK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(M_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(K_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(M_) * N_);

        pipe_.InitBuffer(bQueue_, BUFFER_NUM_B, static_cast<uint32_t>(tN_ * tK_ * sizeof(float)));
        pipe_.InitBuffer(cQueue_, 1, static_cast<uint32_t>(tN_ * sizeof(float)));
        pipe_.InitBuffer(aQueue_, 1, static_cast<uint32_t>(tK_ * sizeof(float)));
        pipe_.InitBuffer(tmpQueue_, 1, static_cast<uint32_t>(tN_ * sizeof(float)));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0 || tN_ == 0 || tK_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows to balance irregular M
        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            // For each N-tile (columns)
            for (uint32_t nBase = 0; nBase < N_; nBase += tN_) {
                const uint32_t nLen = (nBase + tN_ <= N_) ? tN_ : (N_ - nBase);

                AscendC::LocalTensor<float> cLocal = cQueue_.AllocTensor<float>();
                AscendC::Dup(cLocal, 0.0f, nLen);

                AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();

                // Iterate K in tiles
                for (uint32_t kBase = 0; kBase < K_; kBase += tK_) {
                    const uint32_t kLen = (kBase + tK_ <= K_) ? tK_ : (K_ - kBase);

                    // Load B tile [kLen, nLen] into UB, row by row
                    AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();
                    for (uint32_t r = 0; r < kLen; ++r) {
                        const uint64_t bOffGm = static_cast<uint64_t>(kBase + r) * N_ + nBase;
                        AscendC::DataCopy(bLocal[r * tN_], bGm_[bOffGm], nLen);
                    }
                    bQueue_.EnQue(bLocal);

                    // Load A row chunk [kLen] into UB
                    AscendC::LocalTensor<float> aLocal = aQueue_.AllocTensor<float>();
                    {
                        const uint64_t aOffGm = static_cast<uint64_t>(m) * K_ + kBase;
                        AscendC::DataCopy(aLocal, aGm_[aOffGm], kLen);
                    }
                    aQueue_.EnQue(aLocal);

                    // DeQue to compute
                    AscendC::LocalTensor<float> bTile = bQueue_.DeQue<float>();
                    AscendC::LocalTensor<float> aChunk = aQueue_.DeQue<float>();

                    // Accumulate cLocal += a[k] * bRow(k,:)
                    for (uint32_t kk = 0; kk < kLen; ++kk) {
                        float aVal = aChunk.GetValue(kk);
                        AscendC::Muls(tmpLocal, bTile[kk * tN_], aVal, nLen);
                        AscendC::Add(cLocal, cLocal, tmpLocal, nLen);
                    }

                    bQueue_.FreeTensor(bTile);
                    aQueue_.FreeTensor(aChunk);
                }

                // Write back cLocal -> C[m, nBase:nBase+nLen]
                const uint64_t cOffGm = static_cast<uint64_t>(m) * N_ + nBase;
                AscendC::DataCopy(cGm_[cOffGm], cLocal, nLen);

                tmpQueue_.FreeTensor(tmpLocal);
                cQueue_.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM_B> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> cQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_ {0};
    uint32_t N_ {0};
    uint32_t K_ {0};
    uint32_t tN_ {0};
    uint32_t tK_ {0};
};

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithIrregularShapesCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
