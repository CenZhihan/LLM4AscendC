host_operator_src="""
#include "matmul_mish_mish_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

// UB model per tile (floats):
//   aRow: kStep
//   wTile: tileN * kStep
//   acc: tileN
//   bias: tileN
//   tmp1: tileN
//   tmp2: tileN
// Total floats ~= kStep * (1 + tileN) + tileN * 4
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ALIGN_ELEMS = 16;              // vector alignment

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMishMishCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    // w: [N, K], b: [N]

    // Choose kStep
    const uint32_t budgetFloats = UB_BUDGET_BYTES / 4;
    uint32_t kStep = 1024u;
    if (kStep > K) kStep = AlignDown(K, ALIGN_ELEMS) > 0 ? AlignDown(K, ALIGN_ELEMS) : K;
    kStep = AlignDown(kStep, ALIGN_ELEMS);
    if (kStep == 0) kStep = (K > 0 ? 1u : 0u);

    // Compute tileN from budget
    uint32_t tileN = 16u;
    if (kStep > 0) {
        const int64_t avail = static_cast<int64_t>(budgetFloats) - static_cast<int64_t>(kStep);
        const int64_t denom = static_cast<int64_t>(kStep) + 4;
        int64_t maxTileN = (denom > 0) ? (avail / denom) : 16;
        if (maxTileN < 1) maxTileN = 1;
        if (maxTileN > 256) maxTileN = 256; // reasonable cap
        tileN = static_cast<uint32_t>(maxTileN);
        tileN = AlignDown(tileN, ALIGN_ELEMS);
        if (tileN == 0) tileN = 16u;
        if (tileN > N) tileN = (N >= ALIGN_ELEMS) ? AlignDown(N, ALIGN_ELEMS) : N;
    }

    uint32_t blockDim = CeilDiv(M, 1u);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileN(tileN);
    tiling.set_kStep(kStep);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *x_shape = context->GetInputShape(0); // [M, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K]
    gert::Shape *y_shape = context->GetOutputShape(0);
    gert::Shape y;
    y.SetDimNum(2);
    y.SetDim(0, x_shape->GetDim(0));
    y.SetDim(1, w_shape->GetDim(0));
    *y_shape = y;
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelMatmulMishMishCustom {
public:
    __aicore__ inline KernelMatmulMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileN, uint32_t kStep)
    {
        M_ = M; N_ = N; K_ = K;
        tileN_ = tileN; kStep_ = kStep;

        xGm_.SetGlobalBuffer((__gm__ float*)x, M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, M_ * N_);

        // Buffers: aRow(kStep), wTile(tileN*kStep), acc(tileN), bias(tileN), tmp1(tileN), tmp2(tileN), out(tileN)
        pipe_.InitBuffer(aRowQ_, 1, kStep_ * sizeof(float));
        pipe_.InitBuffer(wTileQ_, 1, tileN_ * kStep_ * sizeof(float));
        pipe_.InitBuffer(accQ_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(biasQ_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQ1_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQ2_, 1, tileN_ * sizeof(float));
        pipe_.InitBuffer(outQ_, 1, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            for (uint32_t nBase = 0; nBase < N_; nBase += tileN_) {
                const uint32_t nLen = AscendC::min(tileN_, N_ - nBase);

                AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
                AscendC::Dup(acc, 0.0f, nLen);

                // kLoop accumulation
                for (uint32_t kBase = 0; kBase < K_; kBase += kStep_) {
                    const uint32_t kLen = AscendC::min(kStep_, K_ - kBase);

                    AscendC::LocalTensor<float> aRow = aRowQ_.AllocTensor<float>();
                    AscendC::DataCopy(aRow, xGm_[m * K_ + kBase], kLen);

                    AscendC::LocalTensor<float> wTile = wTileQ_.AllocTensor<float>();
                    // Pack weight tile rows into UB
                    for (uint32_t nn = 0; nn < nLen; ++nn) {
                        const uint32_t wRowGmOff = (nBase + nn) * K_ + kBase;
                        AscendC::DataCopy(wTile[ nn * kLen ], wGm_[ wRowGmOff ], kLen);
                    }

                    // Accumulate dot products
                    // acc[nn] += dot(aRow[0:kLen], wTile[nn][0:kLen])
                    for (uint32_t nn = 0; nn < nLen; ++nn) {
                        float partial = 0.0f;
                        for (uint32_t kk = 0; kk < kLen; ++kk) {
                            partial += aRow[kk] * wTile[nn * kLen + kk];
                        }
                        acc[nn] = acc[nn] + partial;
                    }

                    wTileQ_.FreeTensor(wTile);
                    aRowQ_.FreeTensor(aRow);
                }

                // Add bias
                AscendC::LocalTensor<float> bias = biasQ_.AllocTensor<float>();
                AscendC::DataCopy(bias, bGm_[nBase], nLen);
                AscendC::Add(acc, acc, bias, nLen);
                biasQ_.FreeTensor(bias);

                // First Mish: acc = acc * tanh(ln(1 + exp(acc)))
                AscendC::LocalTensor<float> tmp1 = tmpQ1_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmp2 = tmpQ2_.AllocTensor<float>();

                AscendC::Exp(tmp1, acc, nLen);                 // tmp1 = exp(acc)
                AscendC::Adds(tmp1, tmp1, 1.0f, nLen);         // tmp1 = 1 + exp(acc)
                AscendC::Ln(tmp1, tmp1, nLen);                  // tmp1 = ln(1 + exp(acc))
                AscendC::Tanh(tmp2, tmp1, nLen);                // tmp2 = tanh(tmp1)
                AscendC::Mul(acc, acc, tmp2, nLen);             // acc = acc * tmp2

                // Second Mish: out = acc * tanh(ln(1 + exp(acc)))
                AscendC::Exp(tmp1, acc, nLen);
                AscendC::Adds(tmp1, tmp1, 1.0f, nLen);
                AscendC::Ln(tmp1, tmp1, nLen);
                AscendC::Tanh(tmp2, tmp1, nLen);

                AscendC::LocalTensor<float> out = outQ_.AllocTensor<float>();
                AscendC::Mul(out, acc, tmp2, nLen);

                tmpQ2_.FreeTensor(tmp2);
                tmpQ1_.FreeTensor(tmp1);
                accQ_.FreeTensor(acc);

                // Store
                AscendC::DataCopy(yGm_[m * N_ + nBase], out, nLen);
                outQ_.FreeTensor(out);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aRowQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ1_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ2_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileN_{0}, kStep_{0};
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMishMishCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileN, tiling_data.kStep);
    op.Process();
}
