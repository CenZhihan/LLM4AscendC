static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }

} // anonymous namespace

namespace optiling {

// Conservative UB budget per core
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// Defaults that generally fit budget: area = 32*128 = 4096, ocTile = 8
static constexpr uint32_t DEFAULT_TILE_H = 32;
static constexpr uint32_t DEFAULT_TILE_W = 128;
static constexpr uint32_t DEFAULT_OC_TILE = 8;
static constexpr uint32_t MIN_TILE_H = 8;
static constexpr uint32_t MIN_TILE_W = 32;
static constexpr uint32_t MIN_OC_TILE = 1;
static constexpr uint32_t MAX_CORES = 32;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *wdw_shape = context->GetInputShape(1);
    const gert::Shape *wpw_shape = context->GetInputShape(2);

    // Expect NCHW input, depthwise weight [C, 1, 3, 3], pointwise weight [OC, C, 1, 1]
    uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t C = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t H = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    uint32_t W = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));
    uint32_t OC = static_cast<uint32_t>(wpw_shape->GetOriginShape().GetDim(0));

    // initialize tiles
    uint32_t tileH = (H < DEFAULT_TILE_H) ? H : DEFAULT_TILE_H;
    uint32_t tileW = (W < DEFAULT_TILE_W) ? W : DEFAULT_TILE_W;
    uint32_t ocTile = (OC < DEFAULT_OC_TILE) ? OC : DEFAULT_OC_TILE;

    auto fits = [&](uint32_t th, uint32_t tw, uint32_t oct) -> bool {
        // y_acc: oct * th * tw
        // zTile: th * tw
        // xTile: (th + 2) * (tw + 2)
        uint64_t bytes = 0;
        bytes += static_cast<uint64_t>(oct) * th * tw * sizeof(float);
        bytes += static_cast<uint64_t>(th) * tw * sizeof(float);
        bytes += static_cast<uint64_t>(th + 2) * (tw + 2) * sizeof(float);
        // small headroom
        return bytes + 4096 <= UB_BUDGET_BYTES;
    };

    // shrink until fits
    while (!fits(tileH, tileW, ocTile)) {
        if (tileH > MIN_TILE_H) { tileH = (tileH + 1) / 2; continue; }
        if (tileW > MIN_TILE_W) { tileW = (tileW + 1) / 2; continue; }
        if (ocTile > MIN_OC_TILE) { ocTile = (ocTile + 1) / 2; continue; }
        break;
    }
    if (tileH == 0) tileH = 1;
    if (tileW == 0) tileW = 1;
    if (ocTile == 0) ocTile = 1;

    const uint32_t hTiles = CeilDiv(H, tileH);
    const uint32_t wTiles = CeilDiv(W, tileW);
    const uint32_t ocTiles = CeilDiv(OC, ocTile);
    const uint64_t totalTiles = static_cast<uint64_t>(N) * hTiles * wTiles * ocTiles;

    uint32_t blockDim = static_cast<uint32_t>(totalTiles);
    if (blockDim > MAX_CORES) blockDim = MAX_CORES;
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    ConvDepthwiseSeparable2dCustomTilingData tiling;
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_OC(OC);

    tiling.set_tileH(tileH);
    tiling.set_tileW(tileW);
    tiling.set_ocTile(ocTile);

    tiling.set_hTiles(hTiles);
    tiling.set_wTiles(wTiles);
    tiling.set_ocTiles(ocTiles);

    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

using index_t = int64_t;
using scalar_t = float;

struct TileParams {
    uint32_t N, C, H, W, OC;
    uint32_t tileH, tileW, ocTile;
    uint32_t hTiles, wTiles, ocTiles;
    uint32_t blockDim;
};

class KernelConvDepthwiseSeparable2dCustom {
public:
    __aicore__ inline KernelConvDepthwiseSeparable2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w_dw, GM_ADDR w_pw, GM_ADDR y,
                                const TileParams &p)
    {
        params_ = p;
        xPtr_ = reinterpret_cast<__gm__ scalar_t*>(x);
        wdwPtr_ = reinterpret_cast<__gm__ scalar_t*>(w_dw);
        wpwPtr_ = reinterpret_cast<__gm__ scalar_t*>(w_pw);
        yPtr_ = reinterpret_cast<__gm__ scalar_t*>(y);

        // Allocate UB buffers at max tile sizes
        uint32_t area = params_.tileH * params_.tileW;
        uint32_t xTileElems = (params_.tileH + 2) * (params_.tileW + 2);
        uint32_t yAccElems = params_.ocTile * area;

        pipe_.InitBuffer(yAccQ_, 1, yAccElems * sizeof(scalar_t));
        pipe_.InitBuffer(zTileQ_, 1, area * sizeof(scalar_t));
        pipe_.InitBuffer(xTileQ_, 1, xTileElems * sizeof(scalar_t));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalTiles = static_cast<uint64_t>(params_.N) *
                                    params_.hTiles * params_.wTiles * params_.ocTiles;

        for (uint64_t tileIndex = blockIdx; tileIndex < totalTiles; tileIndex += blockNum) {
            uint64_t t = tileIndex;

            uint32_t ocTileIdx = static_cast<uint32_t>(t % params_.ocTiles); t /= params_.ocTiles;
            uint32_t wTileIdx  = static_cast<uint32_t>(t % params_.wTiles);  t /= params_.wTiles;
            uint32_t hTileIdx  = static_cast<uint32_t>(t % params_.hTiles);  t /= params_.hTiles;
            uint32_t nIdx      = static_cast<uint32_t>(t);

            uint32_t hStart = hTileIdx * params_.tileH;
            uint32_t wStart = wTileIdx * params_.tileW;
            uint32_t hLen = params_.tileH;
            uint32_t wLen = params_.tileW;
            if (hStart + hLen > params_.H) hLen = params_.H - hStart;
            if (wStart + wLen > params_.W) wLen = params_.W - wStart;

            uint32_t ocStart = ocTileIdx * params_.ocTile;
            uint32_t ocCount = params_.ocTile;
            if (ocStart + ocCount > params_.OC) ocCount = params_.OC - ocStart;

            ComputeTile(nIdx, hStart, wStart, hLen, wLen, ocStart, ocCount);
        }
    }

private:
    __aicore__ inline index_t XIndex(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((static_cast<index_t>(n) * params_.C + c) * params_.H + h) * params_.W + w);
    }
    __aicore__ inline index_t YIndex(uint32_t n, uint32_t oc, uint32_t h, uint32_t w) const
    {
        return (((static_cast<index_t>(n) * params_.OC + oc) * params_.H + h) * params_.W + w);
    }
    __aicore__ inline index_t WdwIndex(uint32_t c, uint32_t kh, uint32_t kw) const
    {
        // [C, 1, 3, 3]
        return static_cast<index_t>(c) * 9 + kh * 3 + kw;
    }
    __aicore__ inline index_t WpwIndex(uint32_t oc, uint32_t c) const
    {
        // [OC, C, 1, 1]
        return static_cast<index_t>(oc) * params_.C + c;
    }

    __aicore__ inline void ComputeTile(uint32_t nIdx, uint32_t hStart, uint32_t wStart,
                                       uint32_t hLen, uint32_t wLen,
                                       uint32_t ocStart, uint32_t ocCount)
    {
        // Allocate locals
        AscendC::LocalTensor<scalar_t> yAcc = yAccQ_.AllocTensor<scalar_t>();
        AscendC::LocalTensor<scalar_t> zTile = zTileQ_.AllocTensor<scalar_t>();
        AscendC::LocalTensor<scalar_t> xTile = xTileQ_.AllocTensor<scalar_t>();

        const uint32_t tileAreaMax = params_.tileH * params_.tileW;
        const uint32_t xTileStride = params_.tileW + 2; // width of xTile buffer
        const uint32_t xTileHMax = params_.tileH + 2;

        // zero yAcc full buffer
        for (uint32_t i = 0; i < params_.ocTile * tileAreaMax; ++i) {
            yAcc.SetValue(i, 0.0f);
        }

        // per input channel accumulation
        for (uint32_t c = 0; c < params_.C; ++c) {
            scalar_t k[9];
            // load depthwise kernel for channel c
            for (uint32_t kh = 0; kh < 3; ++kh) {
                for (uint32_t kw = 0; kw < 3; ++kw) {
                    k[kh * 3 + kw] = wdwPtr_[WdwIndex(c, kh, kw)];
                }
            }

            // build xTile with padding=1 around the hLen x wLen region
            // xTile layout: (hLen+2) rows by (wLen+2) cols inside a buffer sized for (tileH+2)*(tileW+2)
            for (uint32_t ih = 0; ih < (hLen + 2); ++ih) {
                int64_t gh = static_cast<int64_t>(hStart) + static_cast<int64_t>(ih) - 1;
                for (uint32_t iw = 0; iw < (wLen + 2); ++iw) {
                    int64_t gw = static_cast<int64_t>(wStart) + static_cast<int64_t>(iw) - 1;
                    scalar_t val = 0.0f;
                    if (gh >= 0 && gw >= 0 && gh < static_cast<int64_t>(params_.H) && gw < static_cast<int64_t>(params_.W)) {
                        index_t xoff = XIndex(nIdx, c, static_cast<uint32_t>(gh), static_cast<uint32_t>(gw));
                        val = xPtr_[xoff];
                    }
                    // store at position ih * xTileStride + iw
                    xTile.SetValue(ih * xTileStride + iw, val);
                }
            }

            // compute depthwise conv for this channel into zTile (size hLen * wLen)
            for (uint32_t ih = 0; ih < hLen; ++ih) {
                for (uint32_t iw = 0; iw < wLen; ++iw) {
                    scalar_t acc = 0.0f;
                    // 3x3 kernel
                    // Manually unroll for better ILP
                    scalar_t v00 = xTile.GetValue((ih + 0) * xTileStride + (iw + 0));
                    scalar_t v01 = xTile.GetValue((ih + 0) * xTileStride + (iw + 1));
                    scalar_t v02 = xTile.GetValue((ih + 0) * xTileStride + (iw + 2));
                    scalar_t v10 = xTile.GetValue((ih + 1) * xTileStride + (iw + 0));
                    scalar_t v11 = xTile.GetValue((ih + 1) * xTileStride + (iw + 1));
                    scalar_t v12 = xTile.GetValue((ih + 1) * xTileStride + (iw + 2));
                    scalar_t v20 = xTile.GetValue((ih + 2) * xTileStride + (iw + 0));
                    scalar_t v21 = xTile.GetValue((ih + 2) * xTileStride + (iw + 1));
                    scalar_t v22 = xTile.GetValue((ih + 2) * xTileStride + (iw + 2));

                    acc += v00 * k[0];
                    acc += v01 * k[1];
                    acc += v02 * k[2];
                    acc += v10 * k[3];
                    acc += v11 * k[4];
                    acc += v12 * k[5];
                    acc += v20 * k[6];
                    acc += v21 * k[7];
                    acc += v22 * k[8];

                    zTile.SetValue(ih * params_.tileW + iw, acc);
                }
            }

            // accumulate pointwise: yAcc[ocLocal, pos] += wpw[oc,c]*zTile[pos]
            for (uint32_t ocl = 0; ocl < ocCount; ++ocl) {
                uint32_t oc = ocStart + ocl;
                scalar_t alpha = wpwPtr_[WpwIndex(oc, c)];
                // iterate over active positions
                for (uint32_t ih = 0; ih < hLen; ++ih) {
                    uint32_t rowBase = ih * params_.tileW;
                    for (uint32_t iw = 0; iw < wLen; ++iw) {
                        uint32_t pos = rowBase + iw;
                        scalar_t z = zTile.GetValue(pos);
                        uint32_t yoff = ocl * tileAreaMax + pos;
                        scalar_t oldv = yAcc.GetValue(yoff);
                        yAcc.SetValue(yoff, oldv + alpha * z);
                    }
                }
            }
        }

        // write back yAcc to y in GM
        for (uint32_t ocl = 0; ocl < ocCount; ++ocl) {
            uint32_t oc = ocStart + ocl;
            for (uint32_t ih = 0; ih < hLen; ++ih) {
                for (uint32_t iw = 0; iw < wLen; ++iw) {
                    uint32_t pos = ih * params_.tileW + iw;
                    scalar_t v = yAcc.GetValue(ocl * tileAreaMax + pos);
                    index_t yoff = YIndex(nIdx, oc, hStart + ih, wStart + iw);
                    yPtr_[yoff] = v;
                }
            }
        }

        // free
        xTileQ_.FreeTensor(xTile);
        zTileQ_.FreeTensor(zTile);
        yAccQ_.FreeTensor(yAcc);
    }

private:
    TileParams params_{};

    __gm__ scalar_t *xPtr_{nullptr};
    __gm__ scalar_t *wdwPtr_{nullptr};
    __gm__ scalar_t *wpwPtr_{nullptr};
    __gm__ scalar_t *yPtr_{nullptr};

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yAccQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> zTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xTileQ_;
};

extern "C" __global__ __aicore__ void conv_depthwise_separable_2d_custom(
    GM_ADDR x, GM_ADDR w_depthwise, GM_ADDR w_pointwise, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    TileParams p;
    p.N = tiling_data.N;
    p.C = tiling_data.C;
    p.H = tiling_data.H;
    p.W = tiling_data.W;
    p.OC = tiling_data.OC;

    p.tileH = tiling_data.tileH;
    p.tileW = tiling_data.tileW;
    p.ocTile = tiling_data.ocTile;

    p.hTiles = tiling_data.hTiles;
    p.wTiles = tiling_data.wTiles;
    p.ocTiles = tiling_data.ocTiles;

    p.blockDim = tiling_data.blockDim;

    KernelConvDepthwiseSeparable2dCustom op;
    op.Init(x, w_depthwise, w_pointwise, y, p);
    op.Process();
}
