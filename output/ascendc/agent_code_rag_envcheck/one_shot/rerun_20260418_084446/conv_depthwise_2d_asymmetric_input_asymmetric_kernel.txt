namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model for this kernel (strideW == 1 fast path):
//   yTile: 1 * tileWo
//   rowTile: 1 * tileWo
//   tmpTile: 1 * tileWo
//   weight cache per-channel: kh * kw
// Total floats ~ (3 * tileWo + kh * kw) => bytes ~ 4 * (3 * tileWo + kh * kw)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;             // reasonable DMA alignment
static constexpr uint32_t MIN_TILE = 128;
static constexpr uint32_t MAX_BLOCKDIM = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dAsymmetricInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const uint32_t n  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t c  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t h  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t w  = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t kh = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t kw = static_cast<uint32_t>(w_shape->GetDim(3));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const uint32_t strideH = *attrs->GetAttrPointer<uint32_t>(0);
    const uint32_t strideW = *attrs->GetAttrPointer<uint32_t>(1);
    const uint32_t padH    = *attrs->GetAttrPointer<uint32_t>(2);
    const uint32_t padW    = *attrs->GetAttrPointer<uint32_t>(3);
    const uint32_t dilH    = *attrs->GetAttrPointer<uint32_t>(4);
    const uint32_t dilW    = *attrs->GetAttrPointer<uint32_t>(5);
    const uint32_t groups  = *attrs->GetAttrPointer<uint32_t>(6);

    // Depthwise expectation: groups == c
    // Output spatial
    const uint32_t ho = static_cast<uint32_t>(
        (static_cast<int64_t>(h) + 2 * static_cast<int64_t>(padH) -
         static_cast<int64_t>(dilH) * (static_cast<int64_t>(kh) - 1) - 1) / static_cast<int64_t>(strideH) + 1);
    const uint32_t wo = static_cast<uint32_t>(
        (static_cast<int64_t>(w) + 2 * static_cast<int64_t>(padW) -
         static_cast<int64_t>(dilW) * (static_cast<int64_t>(kw) - 1) - 1) / static_cast<int64_t>(strideW) + 1);

    // Tile along output width wo
    uint32_t tileWo = MIN_TILE;
    {
        // bytes = 4 * (3 * tileWo + kh * kw)
        const uint32_t weightBytes = 4u * kh * kw;
        if (UB_BUDGET_BYTES > weightBytes + 4u * 3u * MIN_TILE) {
            uint32_t maxTile = (UB_BUDGET_BYTES - weightBytes) / (4u * 3u);
            maxTile = AlignDown(maxTile, ALIGN_ELEMS);
            if (maxTile < MIN_TILE) maxTile = MIN_TILE;
            tileWo = maxTile;
        } else {
            tileWo = MIN_TILE;
        }
        if (wo < tileWo) {
            const uint32_t aligned = AlignDown(wo, ALIGN_ELEMS);
            tileWo = (aligned > 0) ? aligned : wo;
        }
        if (tileWo == 0) tileWo = 1;
    }

    // blockDim over n * c
    const uint32_t totalNC = n * c;
    uint32_t blockDim = totalNC;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > MAX_BLOCKDIM) blockDim = MAX_BLOCKDIM;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_c(c);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_kh(kh);
    tiling.set_kw(kw);
    tiling.set_ho(ho);
    tiling.set_wo(wo);

    tiling.set_strideH(strideH);
    tiling.set_strideW(strideW);
    tiling.set_padH(padH);
    tiling.set_padW(padW);
    tiling.set_dilH(dilH);
    tiling.set_dilW(dilW);
    tiling.set_groups(groups);

    tiling.set_tileWo(tileWo);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvDepthwise2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t width,
                                uint32_t kh, uint32_t kw,
                                uint32_t ho, uint32_t wo,
                                uint32_t strideH, uint32_t strideW,
                                uint32_t padH, uint32_t padW,
                                uint32_t dilH, uint32_t dilW,
                                uint32_t groups,
                                uint32_t tileWo)
    {
        n_ = n; c_ = c; h_ = h; w_ = width;
        kh_ = kh; kw_ = kw;
        ho_ = ho; wo_ = wo;
        strideH_ = strideH; strideW_ = strideW;
        padH_ = padH; padW_ = padW;
        dilH_ = dilH; dilW_ = dilW;
        groups_ = groups;
        tileWo_ = tileWo;

        totalX_ = n_ * c_ * h_ * w_;
        totalW_ = c_ * kh_ * kw_;
        totalY_ = n_ * c_ * ho_ * wo_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalX_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, totalW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalY_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileWo_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueRow_, 1, tileWo_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueTmp_, 1, tileWo_ * sizeof(float));
        pipe_.InitBuffer(weightQueue_, 1, kh_ * kw_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || h_ == 0 || w_ == 0 || kh_ == 0 || kw_ == 0 || ho_ == 0 || wo_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalNC = n_ * c_;
        for (uint32_t nc = blockIdx; nc < totalNC; nc += blockNum) {
            const uint32_t nIdx = nc / c_;
            const uint32_t cIdx = nc % c_;

            // load weight for this channel to UB
            AscendC::LocalTensor<float> wLocal = weightQueue_.AllocTensor<float>();
            const uint32_t wBase = cIdx * kh_ * kw_;
            AscendC::DataCopy(wLocal, wGm_[wBase], kh_ * kw_);

            // iterate over output rows
            for (uint32_t hoIdx = 0; hoIdx < ho_; ++hoIdx) {
                const int32_t inRowBase = static_cast<int32_t>(hoIdx) * static_cast<int32_t>(strideH_) - static_cast<int32_t>(padH_);
                // tile across output width
                for (uint32_t baseWo = 0; baseWo < wo_; baseWo += tileWo_) {
                    uint32_t curTile = wo_ - baseWo;
                    if (curTile > tileWo_) curTile = tileWo_;

                    AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                    AscendC::Dup(yLocal, 0.0f, curTile);

                    // fast path: strideW == 1
                    // y_tile = sum_{ky,kx} w[ky,kx] * x_row_segment
                    for (uint32_t ky = 0; ky < kh_; ++ky) {
                        const int32_t hi = inRowBase + static_cast<int32_t>(ky) * static_cast<int32_t>(dilH_);
                        if (hi < 0 || hi >= static_cast<int32_t>(h_)) {
                            continue; // out-of-range row contributes zeros
                        }

                        // base input offset for this (n,c,hi,0)
                        const uint32_t xNcBase = nIdx * c_ * h_ * w_ + cIdx * h_ * w_;
                        const uint32_t xRowBase = xNcBase + static_cast<uint32_t>(hi) * w_;

                        for (uint32_t kx = 0; kx < kw_; ++kx) {
                            // read weight scalar
                            const uint32_t wIdx = ky * kw_ + kx;
                            // assume LocalTensor supports element access
                            float wVal = wLocal.GetValue(wIdx);

                            // row segment covering output [baseWo, baseWo+curTile)
                            // input start column for kx at the beginning of tile when strideW == 1
                            const int32_t inColStart = static_cast<int32_t>(baseWo) * static_cast<int32_t>(strideW_) - static_cast<int32_t>(padW_) + static_cast<int32_t>(kx) * static_cast<int32_t>(dilW_);

                            AscendC::LocalTensor<float> rowLocal = tmpQueueRow_.AllocTensor<float>();
                            AscendC::Dup(rowLocal, 0.0f, curTile);

                            if (strideW_ == 1) {
                                // contiguous copy for valid part
                                int32_t validStart = inColStart < 0 ? 0 : inColStart;
                                int32_t validEnd   = inColStart + static_cast<int32_t>(curTile);
                                if (validEnd > static_cast<int32_t>(w_)) validEnd = static_cast<int32_t>(w_);
                                if (validStart < validEnd) {
                                    const uint32_t copyLen = static_cast<uint32_t>(validEnd - validStart);
                                    const uint32_t dstOff  = static_cast<uint32_t>(validStart - inColStart);
                                    AscendC::DataCopy(rowLocal[dstOff], xGm_[xRowBase + static_cast<uint32_t>(validStart)], copyLen);
                                }

                                AscendC::LocalTensor<float> tmpLocal = tmpQueueTmp_.AllocTensor<float>();
                                AscendC::Muls(tmpLocal, rowLocal, wVal, curTile);
                                AscendC::Add(yLocal, yLocal, tmpLocal, curTile);
                                tmpQueueTmp_.FreeTensor(tmpLocal);
                            } else {
                                // general strideW fallback: per-element accumulation
                                AscendC::LocalTensor<float> tmpLocal = tmpQueueTmp_.AllocTensor<float>();
                                AscendC::Dup(tmpLocal, 0.0f, curTile);
                                for (uint32_t i = 0; i < curTile; ++i) {
                                    const int32_t wi = static_cast<int32_t>(baseWo + i);
                                    const int32_t col = wi * static_cast<int32_t>(strideW_) - static_cast<int32_t>(padW_) + static_cast<int32_t>(kx) * static_cast<int32_t>(dilW_);
                                    float xVal = 0.0f;
                                    if (col >= 0 && col < static_cast<int32_t>(w_)) {
                                        AscendC::LocalTensor<float> scalar = tmpQueueRow_.AllocTensor<float>();
                                        AscendC::DataCopy(scalar, xGm_[xRowBase + static_cast<uint32_t>(col)], 1);
                                        xVal = scalar.GetValue(0);
                                        tmpQueueRow_.FreeTensor(scalar);
                                    }
                                    float acc = yLocal.GetValue(i) + wVal * xVal;
                                    // write back
                                    AscendC::LocalTensor<float> scalarOut = tmpQueueRow_.AllocTensor<float>();
                                    AscendC::Dup(scalarOut, acc, 1);
                                    AscendC::DataCopy(yLocal[i], scalarOut, 1);
                                    tmpQueueRow_.FreeTensor(scalarOut);
                                }
                                tmpQueueTmp_.FreeTensor(tmpLocal);
                            }

                            tmpQueueRow_.FreeTensor(rowLocal);
                        } // kx
                    } // ky

                    // copy out tile
                    const uint32_t yNcBase = nIdx * c_ * ho_ * wo_ + cIdx * ho_ * wo_;
                    const uint32_t yRowBase = yNcBase + hoIdx * wo_;
                    AscendC::DataCopy(yGm_[yRowBase + baseWo], yLocal, curTile);

                    outQueueY_.FreeTensor(yLocal);
                } // baseWo
            } // hoIdx

            weightQueue_.FreeTensor(wLocal);
        } // nc
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueRow_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueTmp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> weightQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t kh_{0}, kw_{0};
    uint32_t ho_{0}, wo_{0};
    uint32_t strideH_{1}, strideW_{1};
    uint32_t padH_{0}, padW_{0};
    uint32_t dilH_{1}, dilW_{1};
    uint32_t groups_{1};
    uint32_t tileWo_{128};

    uint32_t totalX_{0}, totalW_{0}, totalY_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(
        x, weight, y,
        tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
        tiling_data.kh, tiling_data.kw,
        tiling_data.ho, tiling_data.wo,
        tiling_data.strideH, tiling_data.strideW,
        tiling_data.padH, tiling_data.padW,
        tiling_data.dilH, tiling_data.dilW,
        tiling_data.groups,
        tiling_data.tileWo);
    op.Process();
}

static inline int64_t calc_out_dim(int64_t in, int64_t pad, int64_t dil, int64_t k, int64_t stride)
{
    return (in + 2 * pad - dil * (k - 1) - 1) / stride + 1;
}

at::Tensor conv_dw2d_impl_npu(const at::Tensor& x,
                              const at::Tensor& weight,
                              int64_t stride_h, int64_t stride_w,

    TORCH_CHECK(weight.dim() == 4, "weight must be [C, 1, Kh, Kw]");

    const int64_t Kw = weight.size(3);

    const int64_t Wo = calc_out_dim(W, padding_w, dilation_w, Kw, stride_w);
