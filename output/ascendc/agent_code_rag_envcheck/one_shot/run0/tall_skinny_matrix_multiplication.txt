host_tiling_src="""

#include "register/tilingdata_base.h"

namespace optiling {

BEGIN_TILING_DATA_DEF(TallSkinnyMatrixMultiplicationCustomTilingData)
  TILING_DATA_FIELD_DEF(TCubeTiling, cubeTilingData);
  TILING_DATA_FIELD_DEF(uint32_t, M);
  TILING_DATA_FIELD_DEF(uint32_t, N);
  TILING_DATA_FIELD_DEF(uint32_t, K);
  TILING_DATA_FIELD_DEF(uint32_t, blockDim);
END_TILING_DATA_DEF;

REGISTER_TILING_DATA_CLASS(TallSkinnyMatrixMultiplicationCustom, TallSkinnyMatrixMultiplicationCustomTilingData)

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : ((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TallSkinnyMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *ashape = context->GetInputShape(0);
    const gert::Shape *bshape = context->GetInputShape(1);

    // Assume A: [M, K], B: [K, N]
    uint32_t M = static_cast<uint32_t>(ashape->GetDim(0));
    uint32_t K = static_cast<uint32_t>(ashape->GetDim(1));
    uint32_t K2 = static_cast<uint32_t>(bshape->GetDim(0));
    uint32_t N = static_cast<uint32_t>(bshape->GetDim(1));

    // Basic validation
    if (K != K2) {
        return ge::GRAPH_FAILED;
    }

    // Fill minimal cube tiling info; kernel will register Matmul engine with this
    TCubeTiling &cube = tiling.cubeTilingData;
    // Share full on-chip resources by default
    cube.shareMode = 0;
    cube.shareL1Size = FULL_L1_SIZE;
    cube.shareL0CSize = FULL_L0C_SIZE;
    cube.shareUbSize = 0;

    // Dimensions
    cube.M = M;
    cube.N = N;
    cube.Ka = K;

    // Default flags for non-transposed ND matrices
    cube.type = 0;            // NORMAL
    cube.format = 0;          // ND
    cube.leftType = 0;        // Not transposed
    cube.rightType = 1;       // Transposed flag for B if needed; keep ND non-transpose
    cube.leftTranspose = 0;
    cube.rightTranspose = 0;

    // A simple conservative blockDim based on tiles across M
    uint32_t tilesM = (M == 0) ? 1u : CeilDiv(M, 256u);
    uint32_t blockDim = tilesM;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;

    if (a_shape->GetDim(1) != b_shape->GetDim(0)) {

kernel_src="""

#include "kernel_operator.h"

extern "C" __global__ __aicore__ void tall_skinny_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    // Prepare tiling
    GET_TILING_DATA(tilingData, tiling);

    // Define matmul kernel types: A/B: half in GM, C: float in GM
    using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::NORMAL>;
    using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, true,  LayoutMode::NORMAL>;
    using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL>;
    using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

    AscendC::TPipe pipe;

    // Ensure full on-chip resource sharing for matmul engine
    tilingData.cubeTilingData.shareMode = 0;
    tilingData.cubeTilingData.shareL1Size = FULL_L1_SIZE;
    tilingData.cubeTilingData.shareL0CSize = FULL_L0C_SIZE;
    tilingData.cubeTilingData.shareUbSize = 0;

    // Create and register matmul obj
    AscendC::Matmul<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE, CFG_NORM> matmulObj;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulObj, &tilingData.cubeTilingData);

    // Bind GM tensors
    const uint32_t M = tilingData.M;
    const uint32_t N = tilingData.N;
    const uint32_t K = tilingData.K;

    AscendC::GlobalTensor<half> aGlobal;
    AscendC::GlobalTensor<half> bGlobal;
    AscendC::GlobalTensor<float> cGlobal;

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(A), static_cast<uint64_t>(M) * K);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(B), static_cast<uint64_t>(K) * N);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(C), static_cast<uint64_t>(M) * N);

    matmulObj.SetTensorA(aGlobal, false);
    matmulObj.SetTensorB(bGlobal, false);
    // No bias
    matmulObj.IterateAll(cGlobal);
    matmulObj.End();
}
"""

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_contiguous_half_on_device(const at::Tensor& t) {
    at::Tensor x = t;
    if (x.scalar_type() != at::kHalf) {
        x = x.to(at::kHalf);
    }
    if (!x.is_contiguous()) {
        x = x.contiguous();
    }
    return x;
}

at::Tensor tall_skinny_matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Inputs in float16 for better GEMM throughput, output in float32
    at::Tensor Ah = to_contiguous_half_on_device(A);
    at::Tensor Bh = to_contiguous_half_on_device(B);
    // Output shape: [M, N] where A: [M, K], B: [K, N]
    TORCH_CHECK(Ah.dim() == 2 && Bh.dim() == 2, "Inputs must be 2-D tensors");
    TORCH_CHECK(Ah.size(1) == Bh.size(0), "Incompatible matmul dimensions");
    auto M = Ah.size(0);
    auto N = Bh.size(1);
    at::Tensor C = at::empty({M, N}, Ah.options().dtype(at::kFloat));
    EXEC_NPU_CMD(aclnnTallSkinnyMatrixMultiplicationCustom, Ah, Bh, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("tall_skinny_matrix_multiplication_custom", &tall_skinny_matmul_impl_npu,
          "Tall-skinny optimized matmul (TallSkinnyMatrixMultiplicationCustom on NPU)");
}
