    const gert::Shape *conv1_w_shape = context->GetInputShape(1);
    const gert::Shape *conv2_w_shape = context->GetInputShape(3);

    const uint32_t inC= static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t inH= static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));

kernel_src="""
#include "kernel_operator.h"

class KernelLenet5Custom {
public:
    __aicore__ inline KernelLenet5Custom() {}

    __aicore__ inline void Init(
        GM_ADDR x, GM_ADDR conv1_w, GM_ADDR conv1_b, GM_ADDR conv2_w, GM_ADDR conv2_b,
        GM_ADDR fc1_w, GM_ADDR fc1_b, GM_ADDR fc2_w, GM_ADDR fc2_b, GM_ADDR fc3_w, GM_ADDR fc3_b,
        GM_ADDR y,
        uint32_t N, uint32_t inH, uint32_t inW,
        uint32_t conv1OutC, uint32_t conv2OutC,
        uint32_t fc1Out, uint32_t fc2Out, uint32_t fc3Out)
    {
        N_ = N; inH_ = inH; inW_ = inW;
        conv1OutC_ = conv1OutC; conv2OutC_ = conv2OutC;
        fc1Out_ = fc1Out; fc2Out_ = fc2Out; fc3Out_ = fc3Out;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * inH_ * inW_);
        conv1WGm_.SetGlobalBuffer((__gm__ float*)conv1_w, conv1OutC_ * 1 * 5 * 5);
        conv1BGm_.SetGlobalBuffer((__gm__ float*)conv1_b, conv1OutC_);
        conv2WGm_.SetGlobalBuffer((__gm__ float*)conv2_w, conv2OutC_ * conv1OutC_ * 5 * 5);
        conv2BGm_.SetGlobalBuffer((__gm__ float*)conv2_b, conv2OutC_);
        fc1WGm_.SetGlobalBuffer((__gm__ float*)fc1_w, fc1Out_ * 16 * 5 * 5);
        fc1BGm_.SetGlobalBuffer((__gm__ float*)fc1_b, fc1Out_);
        fc2WGm_.SetGlobalBuffer((__gm__ float*)fc2_w, fc2Out_ * fc1Out_);
        fc2BGm_.SetGlobalBuffer((__gm__ float*)fc2_b, fc2Out_);
        fc3WGm_.SetGlobalBuffer((__gm__ float*)fc3_w, fc3Out_ * fc2Out_);
        fc3BGm_.SetGlobalBuffer((__gm__ float*)fc3_b, fc3Out_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * fc3Out_);

        // Derived dims
        conv1H_ = inH_ >= 5 ? (inH_ - 5 + 1) : 0;
        conv1W_ = inW_ >= 5 ? (inW_ - 5 + 1) : 0;
        pool1H_ = conv1H_ / 2;
        pool1W_ = conv1W_ / 2;
        conv2H_ = pool1H_ >= 5 ? (pool1H_ - 5 + 1) : 0;
        conv2W_ = pool1W_ >= 5 ? (pool1W_ - 5 + 1) : 0;
        pool2H_ = conv2H_ / 2;
        pool2W_ = conv2W_ / 2;
        flatLen_ = conv2OutC_ * pool2H_ * pool2W_;

        // UB buffers
        pipe_.InitBuffer(imgQ_, 1, inH_ * inW_ * sizeof(float));
        pipe_.InitBuffer(conv1WQ_, 1, conv1OutC_ * 5 * 5 * sizeof(float));
        pipe_.InitBuffer(conv1BQ_, 1, conv1OutC_ * sizeof(float));
        pipe_.InitBuffer(conv2WQ_, 1, conv2OutC_ * conv1OutC_ * 5 * 5 * sizeof(float));
        pipe_.InitBuffer(conv2BQ_, 1, conv2OutC_ * sizeof(float));

        pipe_.InitBuffer(conv1OutQ_, 1, conv1OutC_ * conv1H_ * conv1W_ * sizeof(float));
        pipe_.InitBuffer(pool1Q_,    1, conv1OutC_ * pool1H_ * pool1W_ * sizeof(float));
        pipe_.InitBuffer(conv2OutQ_, 1, conv2OutC_ * conv2H_ * conv2W_ * sizeof(float));
        pipe_.InitBuffer(pool2Q_,    1, conv2OutC_ * pool2H_ * pool2W_ * sizeof(float));

        pipe_.InitBuffer(fcRowQ_, 1, flatLen_ * sizeof(float));
        pipe_.InitBuffer(fc1OutQ_, 1, fc1Out_ * sizeof(float));
        pipe_.InitBuffer(fc2OutQ_, 1, fc2Out_ * sizeof(float));
        pipe_.InitBuffer(fc1BQ_, 1, fc1Out_ * sizeof(float));
        pipe_.InitBuffer(fc2BQ_, 1, fc2Out_ * sizeof(float));
        pipe_.InitBuffer(fc3BQ_, 1, fc3Out_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || inH_ == 0 || inW_ == 0) return;

        auto conv1W = conv1WQ_.AllocTensor<float>();
        auto conv1B = conv1BQ_.AllocTensor<float>();
        auto conv2W = conv2WQ_.AllocTensor<float>();
        auto conv2B = conv2BQ_.AllocTensor<float>();
        AscendC::DataCopy(conv1W, conv1WGm_[0], conv1OutC_ * 25);
        AscendC::DataCopy(conv1B, conv1BGm_[0], conv1OutC_);
        AscendC::DataCopy(conv2W, conv2WGm_[0], conv2OutC_ * conv1OutC_ * 25);
        AscendC::DataCopy(conv2B, conv2BGm_[0], conv2OutC_);
        conv1WQ_.EnQue(conv1W);
        conv1BQ_.EnQue(conv1B);
        conv2WQ_.EnQue(conv2W);
        conv2BQ_.EnQue(conv2B);

        auto fc1B = fc1BQ_.AllocTensor<float>();
        auto fc2B = fc2BQ_.AllocTensor<float>();
        auto fc3B = fc3BQ_.AllocTensor<float>();
        AscendC::DataCopy(fc1B, fc1BGm_[0], fc1Out_);
        AscendC::DataCopy(fc2B, fc2BGm_[0], fc2Out_);
        AscendC::DataCopy(fc3B, fc3BGm_[0], fc3Out_);
        fc1BQ_.EnQue(fc1B);
        fc2BQ_.EnQue(fc2B);
        fc3BQ_.EnQue(fc3B);

        auto conv1WLT = conv1WQ_.DeQue<float>();
        auto conv1BLT = conv1BQ_.DeQue<float>();
        auto conv2WLT = conv2WQ_.DeQue<float>();
        auto conv2BLT = conv2BQ_.DeQue<float>();
        auto fc1BLT = fc1BQ_.DeQue<float>();
        auto fc2BLT = fc2BQ_.DeQue<float>();
        auto fc3BLT = fc3BQ_.DeQue<float>();

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t n = blockIdx; n < N_; n += blockNum) {
            auto img = imgQ_.AllocTensor<float>();
            AscendC::DataCopy(img, xGm_[n * inH_ * inW_], inH_ * inW_);

            auto conv1Out = conv1OutQ_.AllocTensor<float>();
            // conv1: [1, inH, inW] -> [conv1OutC, conv1H_, conv1W_]
            for (uint32_t oc = 0; oc < conv1OutC_; ++oc) {
                for (uint32_t oy = 0; oy < conv1H_; ++oy) {
                    for (uint32_t ox = 0; ox < conv1W_; ++ox) {
                        float acc = conv1BLT[oc];
                        uint32_t wbase = oc * 25;
                        for (uint32_t ky = 0; ky < 5; ++ky) {
                            for (uint32_t kx = 0; kx < 5; ++kx) {
                                float v = img[(oy + ky) * inW_ + (ox + kx)];
                                float w = conv1WLT[wbase + ky * 5 + kx];
                                acc += v * w;
                            }
                        }
                        if (acc < 0.0f) acc = 0.0f; // ReLU
                        conv1Out[(oc * conv1H_ + oy) * conv1W_ + ox] = acc;
                    }
                }
            }

            auto pool1 = pool1Q_.AllocTensor<float>();
            // max pool 2x2 stride2 on conv1Out
            for (uint32_t oc = 0; oc < conv1OutC_; ++oc) {
                for (uint32_t py = 0; py < pool1H_; ++py) {
                    for (uint32_t px = 0; px < pool1W_; ++px) {
                        uint32_t y0 = py * 2;
                        uint32_t x0 = px * 2;
                        float a = conv1Out[(oc * conv1H_ + y0) * conv1W_ + x0];
                        float b = conv1Out[(oc * conv1H_ + y0) * conv1W_ + (x0 + 1)];
                        float c = conv1Out[(oc * conv1H_ + (y0 + 1)) * conv1W_ + x0];
                        float d = conv1Out[(oc * conv1H_ + (y0 + 1)) * conv1W_ + (x0 + 1)];
                        float m1 = a > b ? a : b;
                        float m2 = c > d ? c : d;
                        float m = m1 > m2 ? m1 : m2;
                        pool1[(oc * pool1H_ + py) * pool1W_ + px] = m;
                    }
                }
            }

            auto conv2Out = conv2OutQ_.AllocTensor<float>();
            // conv2: [conv1OutC, pool1H_, pool1W_] -> [conv2OutC, conv2H_, conv2W_]
            for (uint32_t oc = 0; oc < conv2OutC_; ++oc) {
                for (uint32_t oy = 0; oy < conv2H_; ++oy) {
                    for (uint32_t ox = 0; ox < conv2W_; ++ox) {
                        float acc = conv2BLT[oc];
                        uint32_t wocbase = oc * (conv1OutC_ * 25);
                        for (uint32_t ic = 0; ic < conv1OutC_; ++ic) {
                            uint32_t wbase = wocbase + ic * 25;
                            for (uint32_t ky = 0; ky < 5; ++ky) {
                                for (uint32_t kx = 0; kx < 5; ++kx) {
                                    float v = pool1[(ic * pool1H_ + (oy + ky)) * pool1W_ + (ox + kx)];
                                    float w = conv2WLT[wbase + ky * 5 + kx];
                                    acc += v * w;
                                }
                            }
                        }
                        if (acc < 0.0f) acc = 0.0f; // ReLU
                        conv2Out[(oc * conv2H_ + oy) * conv2W_ + ox] = acc;
                    }
                }
            }

            auto pool2 = pool2Q_.AllocTensor<float>();
            for (uint32_t oc = 0; oc < conv2OutC_; ++oc) {
                for (uint32_t py = 0; py < pool2H_; ++py) {
                    for (uint32_t px = 0; px < pool2W_; ++px) {
                        uint32_t y0 = py * 2;
                        uint32_t x0 = px * 2;
                        float a = conv2Out[(oc * conv2H_ + y0) * conv2W_ + x0];
                        float b = conv2Out[(oc * conv2H_ + y0) * conv2W_ + (x0 + 1)];
                        float c = conv2Out[(oc * conv2H_ + (y0 + 1)) * conv2W_ + x0];
                        float d = conv2Out[(oc * conv2H_ + (y0 + 1)) * conv2W_ + (x0 + 1)];
                        float m1 = a > b ? a : b;
                        float m2 = c > d ? c : d;
                        float m = m1 > m2 ? m1 : m2;
                        pool2[(oc * pool2H_ + py) * pool2W_ + px] = m;
                    }
                }
            }

            // Flatten pool2 into fc input
            auto fc1Out = fc1OutQ_.AllocTensor<float>();
            auto fc2Out = fc2OutQ_.AllocTensor<float>();
            auto fcRow = fcRowQ_.AllocTensor<float>();

            // prepare flattened vector f of size flatLen_
            for (uint32_t i = 0; i < fc1Out_; ++i) { fc1Out[i] = 0.0f; }
            for (uint32_t i = 0; i < fc2Out_; ++i) { fc2Out[i] = 0.0f; }

            // FC1: out = ReLU(W * f + b)
            for (uint32_t o = 0; o < fc1Out_; ++o) {
                AscendC::DataCopy(fcRow, fc1WGm_[o * flatLen_], flatLen_);
                float acc = 0.0f;
                for (uint32_t k = 0; k < flatLen_; ++k) {
                    acc += fcRow[k] * pool2[k];
                }
                acc += fc1BLT[o];
                if (acc < 0.0f) acc = 0.0f;
                fc1Out[o] = acc;
            }

            // FC2: out = ReLU(W * fc1Out + b)
            for (uint32_t o = 0; o < fc2Out_; ++o) {
                AscendC::DataCopy(fcRow, fc2WGm_[o * fc1Out_], fc1Out_);
                float acc = 0.0f;
                for (uint32_t k = 0; k < fc1Out_; ++k) {
                    acc += fcRow[k] * fc1Out[k];
                }
                acc += fc2BLT[o];
                if (acc < 0.0f) acc = 0.0f;
                fc2Out[o] = acc;
            }

            // FC3: logits = W * fc2Out + b (no ReLU)
            for (uint32_t o = 0; o < fc3Out_; ++o) {
                AscendC::DataCopy(fcRow, fc3WGm_[o * fc2Out_], fc2Out_);
                float acc = 0.0f;
                for (uint32_t k = 0; k < fc2Out_; ++k) {
                    acc += fcRow[k] * fc2Out[k];
                }
                acc += fc3BLT[o];
                fcRow[o] = acc; // reuse fcRow buffer to store logits
            }

            AscendC::DataCopy(yGm_[n * fc3Out_], fcRow, fc3Out_);

            // Free temporaries for this sample
            fcRowQ_.FreeTensor(fcRow);
            fc2OutQ_.FreeTensor(fc2Out);
            fc1OutQ_.FreeTensor(fc1Out);
            pool2Q_.FreeTensor(pool2);
            conv2OutQ_.FreeTensor(conv2Out);
            pool1Q_.FreeTensor(pool1);
            conv1OutQ_.FreeTensor(conv1Out);
            imgQ_.FreeTensor(img);
        }
        // Note: persistent weights/biases in UB are intentionally kept until kernel end
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, 1> imgQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> conv1WQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> conv1BQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> conv2WQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> conv2BQ_;

    AscendC::TQue<AscendC::TPosition::VECCALC, 1> conv1OutQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> pool1Q_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> conv2OutQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> pool2Q_;

    AscendC::TQue<AscendC::TPosition::VECCALC, 1> fcRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> fc1OutQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> fc2OutQ_;

    AscendC::TQue<AscendC::TPosition::VECIN, 1> fc1BQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> fc2BQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> fc3BQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> conv1WGm_;
    AscendC::GlobalTensor<float> conv1BGm_;
    AscendC::GlobalTensor<float> conv2WGm_;
    AscendC::GlobalTensor<float> conv2BGm_;
    AscendC::GlobalTensor<float> fc1WGm_;
    AscendC::GlobalTensor<float> fc1BGm_;
    AscendC::GlobalTensor<float> fc2WGm_;
    AscendC::GlobalTensor<float> fc2BGm_;
    AscendC::GlobalTensor<float> fc3WGm_;
    AscendC::GlobalTensor<float> fc3BGm_;
    AscendC::GlobalTensor<float> yGm_;
