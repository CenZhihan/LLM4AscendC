kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM_IN = 2;
constexpr int32_t BUFFER_NUM_TMP = 1;
constexpr int32_t BUFFER_NUM_OUT = 1;

// Naive depthwise 3x3 (stride=1, pad=0, dilation=1) kernel for NCHW float32.
// Each core processes a range of (N * C) depthwise groups.
class KernelConvDepthwise2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t kH, uint32_t kW,
                                uint32_t OH, uint32_t OW,
                                uint32_t stride, uint32_t padding, uint32_t dilation, uint32_t groups)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        kH_ = kH; kW_ = kW; OH_ = OH; OW_ = OW;
        stride_ = stride; padding_ = padding; dilation_ = dilation; groups_ = groups;

        xBatchStride_ = C_ * H_ * W_;
        xChanStride_  = H_ * W_;
        xRowStride_   = W_;

        yBatchStride_ = C_ * OH_ * OW_;
        yChanStride_  = OH_ * OW_;
        yRowStride_   = OW_;

        wChanStride_  = kH_ * kW_; // weight layout: [C, 1, kH, kW]

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), N_ * C_ * H_ * W_);
        wGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w), C_ * kH_ * kW_);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), N_ * C_ * OH_ * OW_);

        // UB buffers sized for one output row (OW elements) and temporaries
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM_IN, OW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM_TMP, OW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM_OUT, OW_ * sizeof(float));

        // Small buffer for 3x3 weight per-channel
        pipe_.InitBuffer(weightQueue_, 1, wChanStride_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || H_ == 0 || W_ == 0 || OH_ == 0 || OW_ == 0) {
            return;
        }

        // Only support the fixed hyper-parameters for this custom kernel
        if (!(kH_ == 3 && kW_ == 3 && stride_ == 1 && padding_ == 0 && dilation_ == 1 && groups_ == C_)) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());
        const uint32_t totalGroups = N_ * C_;

        const uint32_t groupsPerCore = (totalGroups + blockNum - 1u) / blockNum;
        const uint32_t gBegin = blockIdx * groupsPerCore;
        const uint32_t gEnd = Min(totalGroups, gBegin + groupsPerCore);

        for (uint32_t g = gBegin; g < gEnd; ++g) {
            const uint32_t n = g / C_;
            const uint32_t c = g % C_;
            ComputeOneGroup(n, c);
        }
    }

private:
    __aicore__ inline void LoadWeights(uint32_t c,
                                       float &w00, float &w01, float &w02,
                                       float &w10, float &w11, float &w12,
                                       float &w20, float &w21, float &w22)
    {
        const uint32_t wOff = c * wChanStride_;
        LocalTensor<float> wLocal = weightQueue_.AllocTensor<float>();
        DataCopy(wLocal, wGm_[wOff], wChanStride_);
        weightQueue_.EnQue(wLocal);

        // DeQue so we can read
        LocalTensor<float> wLocalRd = weightQueue_.DeQue<float>();
        // Read 9 scalars from local buffer (assumed supported)
        w00 = wLocalRd.GetValue(0);
        w01 = wLocalRd.GetValue(1);
        w02 = wLocalRd.GetValue(2);
        w10 = wLocalRd.GetValue(3);
        w11 = wLocalRd.GetValue(4);
        w12 = wLocalRd.GetValue(5);
        w20 = wLocalRd.GetValue(6);
        w21 = wLocalRd.GetValue(7);
        w22 = wLocalRd.GetValue(8);
        weightQueue_.FreeTensor(wLocalRd);
    }

    __aicore__ inline void ComputeOneGroup(uint32_t n, uint32_t c)
    {
        // Load 3x3 weights for this channel
        float w00, w01, w02, w10, w11, w12, w20, w21, w22;
        LoadWeights(c, w00, w01, w02, w10, w11, w12, w20, w21, w22);

        for (uint32_t ho = 0; ho < OH_; ++ho) {
            // y row buffer
            LocalTensor<float> yRow = outQueueY_.AllocTensor<float>();
            Dup<float>(yRow, 0.0f, OW_);

            // Accumulate across 3x3 kernel
            for (uint32_t kh = 0; kh < 3u; ++kh) {
                const uint32_t inRow = ho + kh; // since stride=1, pad=0
                for (uint32_t kw = 0; kw < 3u; ++kw) {
                    // Load contiguous input segment of length OW_ starting at (n,c,inRow, kw)
                    const uint32_t xBase = n * xBatchStride_ + c * xChanStride_ + inRow * xRowStride_ + kw;
                    LocalTensor<float> xSeg = inQueueX_.AllocTensor<float>();
                    DataCopy(xSeg, xGm_[xBase], OW_);
                    inQueueX_.EnQue(xSeg);

                    // Multiply by scalar weight and accumulate
                    LocalTensor<float> xSegRd = inQueueX_.DeQue<float>();
                    LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

                    const float wscalar = (kh == 0 && kw == 0) ? w00 :
                                          (kh == 0 && kw == 1) ? w01 :
                                          (kh == 0 && kw == 2) ? w02 :
                                          (kh == 1 && kw == 0) ? w10 :
                                          (kh == 1 && kw == 1) ? w11 :
                                          (kh == 1 && kw == 2) ? w12 :
                                          (kh == 2 && kw == 0) ? w20 :
                                          (kh == 2 && kw == 1) ? w21 : w22;

                    Muls(tmp, xSegRd, wscalar, OW_);      // tmp = xSeg * wscalar
                    Add(yRow, yRow, tmp, OW_);            // yRow += tmp

                    tmpQueue_.FreeTensor(tmp);
                    inQueueX_.FreeTensor(xSegRd);
                }
            }

            // Store y row
            const uint32_t yBase = n * yBatchStride_ + c * yChanStride_ + ho * yRowStride_;
            DataCopy(yGm_[yBase], yRow, OW_);
            outQueueY_.FreeTensor(yRow);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, BUFFER_NUM_IN> inQueueX_;
    TQue<TPosition::VECCALC, BUFFER_NUM_TMP> tmpQueue_;
    TQue<TPosition::VECOUT, BUFFER_NUM_OUT> outQueueY_;
    TQue<TPosition::VECIN, 1> weightQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t kH_{0}, kW_{0}, OH_{0}, OW_{0};
    uint32_t stride_{1}, padding_{0}, dilation_{1}, groups_{1};

    uint32_t xBatchStride_{0}, xChanStride_{0}, xRowStride_{0};
    uint32_t yBatchStride_{0}, yChanStride_{0}, yRowStride_{0};
    uint32_t wChanStride_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConvDepthwise2dSquareInputSquareKernelCustom op;
    op.Init(x, w, y,
            tiling_data.nSize, tiling_data.cSize, tiling_data.hSize, tiling_data.wSize,
            tiling_data.kH, tiling_data.kW,
            tiling_data.oH, tiling_data.oW,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation, tiling_data.groups);
    op.Process();
}
