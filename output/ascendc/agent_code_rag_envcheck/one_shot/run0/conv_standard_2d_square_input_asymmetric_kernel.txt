kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin,
                                uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo)
    {
        N_ = N; Cin_ = Cin; H_ = H; W_ = W;
        Cout_ = Cout; Kh_ = Kh; Kw_ = Kw; Ho_ = Ho; Wo_ = Wo;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Cout_) * Cin_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * Cout_ * Ho_ * Wo_);

        // Buffers: xRow (W), yRow (Wo), tmp (Wo), wRow (Kw)
        pipe_.InitBuffer(xRowBuf_, 1, W_ * sizeof(float));
        pipe_.InitBuffer(yRowBuf_, 1, Wo_ * sizeof(float));
        pipe_.InitBuffer(tmpBuf_, 1, Wo_ * sizeof(float));
        pipe_.InitBuffer(wRowBuf_, 1, Kw_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || H_ == 0 || W_ == 0 || Kh_ == 0 || Kw_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalTasks = static_cast<uint64_t>(N_) * static_cast<uint64_t>(Cout_);
        if (totalTasks == 0) return;

        for (uint64_t task = blockIdx; task < totalTasks; task += blockNum) {
            const uint32_t n  = static_cast<uint32_t>(task / Cout_);
            const uint32_t oc = static_cast<uint32_t>(task % Cout_);

            for (uint32_t oh = 0; oh < Ho_; ++oh) {
                AscendC::LocalTensor<float> yRow = yRowBuf_.AllocTensor<float>();
                AscendC::Dup(yRow, 0.0f, Wo_);

                for (uint32_t ic = 0; ic < Cin_; ++ic) {
                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        // Load one weight row (Kw elements) for this (oc, ic, kh)
                        const uint64_t wBase = ((((uint64_t)oc * Cin_ + ic) * Kh_ + kh) * Kw_);
                        AscendC::LocalTensor<float> wRow = wRowBuf_.AllocTensor<float>();
                        AscendC::DataCopy(wRow, wGm_[wBase], Kw_);

                        // Load one input row (W elements) for this (n, ic, oh+kh)
                        const uint32_t ih = oh + kh;
                        const uint64_t xBase = ((((uint64_t)n * Cin_ + ic) * H_ + ih) * W_);
                        AscendC::LocalTensor<float> xRow = xRowBuf_.AllocTensor<float>();
                        AscendC::DataCopy(xRow, xGm_[xBase], W_);

                        // Slide over Kw and accumulate into yRow
                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                            float wVal = AscendC::GetValue(wRow, kw);
                            AscendC::LocalTensor<float> xSlice = xRow[kw];
                            AscendC::LocalTensor<float> tmp = tmpBuf_.AllocTensor<float>();
                            AscendC::Muls(tmp, xSlice, wVal, Wo_);
                            AscendC::Add(yRow, yRow, tmp, Wo_);
                            tmpBuf_.FreeTensor(tmp);
                        }

                        xRowBuf_.FreeTensor(xRow);
                        wRowBuf_.FreeTensor(wRow);
                    }
                }

                // Store y row
                const uint64_t yBase = ((((uint64_t)n * Cout_ + oc) * Ho_ + oh) * Wo_);
                AscendC::DataCopy(yGm_[yBase], yRow, Wo_);
                yRowBuf_.FreeTensor(yRow);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xRowBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yRowBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpBuf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wRowBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, Kh_{0}, Kw_{0}, Ho_{0}, Wo_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.H, tiling_data.W,
            tiling_data.Cout, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv2d_asym_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    // Output shape: [N, Cout, Ho, Wo] with stride=1, pad=0, dilation=1
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, "Expected 4D tensors");
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto H = x.size(2);
    auto W = x.size(3);
    auto Cout = weight.size(0);
    auto CinW = weight.size(1);
    auto Kh = weight.size(2);
    auto Kw = weight.size(3);
    TORCH_CHECK(Cin == CinW, "Input channels must match weight channels");
    TORCH_CHECK(H >= Kh && W >= Kw, "Kernel larger than input");
    auto Ho = H - Kh + 1;
    auto Wo = W - Kw + 1;

    at::Tensor result = at::empty({N, Cout, Ho, Wo}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard2dSquareInputAsymmetricKernelCustom, x, weight, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_standard_2d_square_input_asymmetric_kernel_custom", &conv2d_asym_impl_npu);
}

    m.def("conv_standard_2d_square_input_asymmetric_kernel_custom",
