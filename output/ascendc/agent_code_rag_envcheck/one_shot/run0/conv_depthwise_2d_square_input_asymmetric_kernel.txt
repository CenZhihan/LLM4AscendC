kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvDepthwise2dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvDepthwise2dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t kH, uint32_t kW, uint32_t hOut, uint32_t wOut)
    {
        N_ = n; C_ = c; H_ = h; W_ = w_in;
        kH_ = kH; kW_ = kW; Hout_ = hOut; Wout_ = wOut;

        totalX_ = static_cast<uint32_t>(N_) * C_ * H_ * W_;
        totalW_ = static_cast<uint32_t>(C_) * kH_;
        totalY_ = static_cast<uint32_t>(N_) * C_ * Hout_ * Wout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalX_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, totalW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalY_);

        // Row-sized UB buffers
        rowSizeBytes_ = Wout_ * sizeof(float);
        pipe_.InitBuffer(inRowQ_, BUFFER_NUM, rowSizeBytes_);
        pipe_.InitBuffer(accRowQ_, BUFFER_NUM, rowSizeBytes_);
        // Scalar buffer for weight element load
        pipe_.InitBuffer(scalarQ_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || Hout_ == 0 || Wout_ == 0 || kH_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t ncTotal = N_ * C_;
        for (uint32_t nc = blockIdx; nc < ncTotal; nc += blockNum) {
            const uint32_t n = nc / C_;
            const uint32_t c = nc % C_;

            ComputeOneNC(n, c);
        }
    }

private:
    __aicore__ inline float LoadWeightScalar(uint32_t c, uint32_t r)
    {
        const uint32_t off = c * kH_ + r;
        AscendC::LocalTensor<float> tmp = scalarQ_.AllocTensor<float>();
        AscendC::DataCopy(tmp, wGm_[off], 1);
        // Hypothetical helper to read a single value from UB
        float val = tmp.GetValue(0);
        scalarQ_.FreeTensor(tmp);
        return val;
    }

    __aicore__ inline void ComputeOneNC(uint32_t n, uint32_t c)
    {
        const uint32_t xBase = ((n * C_ + c) * H_) * W_;
        const uint32_t yBase = ((n * C_ + c) * Hout_) * Wout_;

        for (uint32_t hOut = 0; hOut < Hout_; ++hOut) {
            AscendC::LocalTensor<float> acc = accRowQ_.AllocTensor<float>();
            AscendC::Dup(acc, 0.0f, Wout_);

            for (uint32_t r = 0; r < kH_; ++r) {
                const uint32_t xRowOff = xBase + (hOut + r) * W_;
                AscendC::LocalTensor<float> xRow = inRowQ_.AllocTensor<float>();
                AscendC::DataCopy(xRow, xGm_[xRowOff], Wout_);

                const float wv = LoadWeightScalar(c, r);

                // temp = xRow * wv; acc += temp
                AscendC::Muls(xRow, xRow, wv, Wout_);
                AscendC::Add(acc, acc, xRow, Wout_);

                inRowQ_.FreeTensor(xRow);
            }

            // Write accumulated row to GM
            const uint32_t yRowOff = yBase + hOut * Wout_;
            AscendC::DataCopy(yGm_[yRowOff], acc, Wout_);
            accRowQ_.FreeTensor(acc);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inRowQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> accRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t kH_{0}, kW_{0};
    uint32_t Hout_{0}, Wout_{0};
    uint32_t totalX_{0}, totalW_{0}, totalY_{0};
    uint32_t rowSizeBytes_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.kH, tiling_data.kW, tiling_data.hOut, tiling_data.wOut);
    op.Process();
}
