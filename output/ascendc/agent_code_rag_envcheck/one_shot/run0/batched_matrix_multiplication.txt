host_operator_src="""

#include "batched_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// UB model (single-buffered):
//   A: 1 * (tileM * tileK)
//   B: 1 * (tileK * tileN)
//   C: 1 * (tileM * tileN)
// Total floats = tileM*tileN + tileK*(tileM + tileN)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t UB_BUDGET_FLOATS = UB_BUDGET_BYTES / 4;
static constexpr uint32_t BASE_TILE_M = 64;
static constexpr uint32_t BASE_TILE_N = 64;
static constexpr uint32_t MIN_TILE_K = 16;
static constexpr uint32_t K_ALIGN = 16;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    BatchedMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *aShape = context->GetInputShape(0);
    const gert::Shape *bShape = context->GetInputShape(1);

    // Expect A: [B, M, K], B: [B, K, N]
    const uint32_t batch = static_cast<uint32_t>(aShape->GetOriginShape().GetDim(0));
    const uint32_t m = static_cast<uint32_t>(aShape->GetOriginShape().GetDim(1));
    const uint32_t k = static_cast<uint32_t>(aShape->GetOriginShape().GetDim(2));
    const uint32_t n = static_cast<uint32_t>(bShape->GetOriginShape().GetDim(2));

    uint32_t tileM = BASE_TILE_M;
    uint32_t tileN = BASE_TILE_N;
    if (tileM > m) tileM = (m > 0) ? m : 1;
    if (tileN > n) tileN = (n > 0) ? n : 1;

    // Determine tileK from UB budget:
    // UB floats for C:
    uint32_t cFloats = tileM * tileN;
    uint32_t rem = (UB_BUDGET_FLOATS > cFloats) ? (UB_BUDGET_FLOATS - cFloats) : (MIN_TILE_K * (tileM + tileN));
    uint32_t tileK = rem / (tileM + tileN);
    if (tileK < MIN_TILE_K) tileK = MIN_TILE_K;
    if (tileK > k) tileK = k;
    tileK = AlignDown(tileK, K_ALIGN);
    if (tileK == 0) tileK = (k >= K_ALIGN) ? K_ALIGN : k;

    const uint32_t tilesM = CeilDiv(m, tileM);
    const uint32_t tilesN = CeilDiv(n, tileN);
    const uint32_t totalTiles = (tilesM == 0 || tilesN == 0) ? batch : batch * tilesM * tilesN;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_batch(batch);
    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_tilesM(tilesM);
    tiling.set_tilesN(tilesN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    // A: [B, M, K], B: [B, K, N] -> C: [B, M, N]
    std::vector<int64_t> out_dims(3);
    out_dims[0] = a_shape->GetOriginShape().GetDim(0);
    out_dims[1] = a_shape->GetOriginShape().GetDim(1);
    out_dims[2] = b_shape->GetOriginShape().GetDim(2);
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetOriginShape(ge::Shape(out_dims));
    c_shape->SetStorageShape(ge::Shape(out_dims));
    return GRAPH_SUCCESS;
}

{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelBatchedMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelBatchedMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t batch, uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK,
                                uint32_t tilesM, uint32_t tilesN)
    {
        batch_ = batch; m_ = m; n_ = n; k_ = k;
        tileM_ = (tileM == 0) ? 1 : tileM;
        tileN_ = (tileN == 0) ? 1 : tileN;
        tileK_ = (tileK == 0) ? 1 : tileK;
        tilesM_ = (tilesM == 0) ? 1 : tilesM;
        tilesN_ = (tilesN == 0) ? 1 : tilesN;

        const uint32_t aSize = batch_ * m_ * k_;
        const uint32_t bSize = batch_ * k_ * n_;
        const uint32_t cSize = batch_ * m_ * n_;

        aGm_.SetGlobalBuffer((__gm__ float*)a, aSize);
        bGm_.SetGlobalBuffer((__gm__ float*)b, bSize);
        cGm_.SetGlobalBuffer((__gm__ float*)c, cSize);

        pipe_.InitBuffer(aQueue_, BUFFER_NUM, tileM_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, BUFFER_NUM, tileK_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(cQueue_, BUFFER_NUM, tileM_ * tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQueue_, BUFFER_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || n_ == 0 || k_ == 0 || batch_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalTiles = batch_ * tilesM_ * tilesN_;
        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            // Decode tile indices
            const uint32_t bnTiles = tilesM_ * tilesN_;
            const uint32_t bIdx = tileId / bnTiles;
            const uint32_t rem = tileId % bnTiles;
            const uint32_t tm = rem / tilesN_;
            const uint32_t tn = rem % tilesN_;
            const uint32_t mStart = tm * tileM_;
            const uint32_t nStart = tn * tileN_;

            const uint32_t curM = (mStart + tileM_ <= m_) ? tileM_ : (m_ - mStart);
            const uint32_t curN = (nStart + tileN_ <= n_) ? tileN_ : (n_ - nStart);

            AscendC::LocalTensor<float> cLocal = cQueue_.AllocTensor<float>();
            AscendC::Dup(cLocal, 0.0f, tileM_ * tileN_);

            for (uint32_t k0 = 0; k0 < k_; k0 += tileK_) {
                const uint32_t curK = (k0 + tileK_ <= k_) ? tileK_ : (k_ - k0);

                AscendC::LocalTensor<float> aLocal = aQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();

                // Load A tile: rows [mStart, mStart+curM), cols [k0, k0+curK)
                for (uint32_t i = 0; i < curM; ++i) {
                    const uint32_t a_gm_row_off = ((bIdx * m_ + (mStart + i)) * k_) + k0;
                    AscendC::DataCopy(aLocal[i * tileK_], aGm_[a_gm_row_off], curK);
                }

                // Load B tile: rows [k0, k0+curK), cols [nStart, nStart+curN)
                for (uint32_t t = 0; t < curK; ++t) {
                    const uint32_t b_gm_row_off = ((bIdx * k_ + (k0 + t)) * n_) + nStart;
                    AscendC::DataCopy(bLocal[t * tileN_], bGm_[b_gm_row_off], curN);
                }

                // Compute C += A * B (for this K chunk)
                // Row-wise accumulation: for each row i, for each t in K, c_row += a[i,t] * B_row(t,:)
                AscendC::LocalTensor<float> tmpVec = tmpVecQueue_.AllocTensor<float>();
                for (uint32_t i = 0; i < curM; ++i) {
                    for (uint32_t t = 0; t < curK; ++t) {
                        const float alpha = aLocal.GetValue(i * tileK_ + t);
                        // tmpVec = alpha * B_row(t, :)
                        AscendC::Muls(tmpVec, bLocal[t * tileN_], alpha, curN);
                        // c_row += tmpVec
                        AscendC::Add(cLocal[i * tileN_], cLocal[i * tileN_], tmpVec, curN);
                    }
                }
                tmpVecQueue_.FreeTensor(tmpVec);

                bQueue_.FreeTensor(bLocal);
                aQueue_.FreeTensor(aLocal);
            }

            // Store C tile
            for (uint32_t i = 0; i < curM; ++i) {
                const uint32_t c_gm_row_off = ((bIdx * m_ + (mStart + i)) * n_) + nStart;
                AscendC::DataCopy(cGm_[c_gm_row_off], cLocal[i * tileN_], curN);
            }
            cQueue_.FreeTensor(cLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> aQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> cQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpVecQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t batch_ {0};
    uint32_t m_ {0};
    uint32_t n_ {0};
    uint32_t k_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
    uint32_t tileK_ {0};
    uint32_t tilesM_ {0};
    uint32_t tilesN_ {0};
};

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelBatchedMatrixMultiplicationCustom op;
    op.Init(a, b, c,
            tiling_data.batch, tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK,
            tiling_data.tilesM, tiling_data.tilesN);
    op.Process();
}
