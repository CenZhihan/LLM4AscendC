kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;
constexpr int32_t TMP_BUF_NUM = 2;

class KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint32_t tileLenW)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        D_ = D; H_ = H; W_ = W;
        kD_ = kD; kH_ = kH; kW_ = kW;
        outD_ = outD; outH_ = outH; outW_ = outW;
        tileW_ = tileLenW > 0 ? tileLenW : outW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_*Cin_*D_*H_*W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cin_*Cout_*kD_*kH_*kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_*Cout_*outD_*outH_*outW_);

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, TMP_BUF_NUM, (kD_*kH_*kW_ > tileW_ ? kD_*kH_*kW_ : tileW_) * sizeof(float));
    }

    __aicore__ inline void Process(uint64_t numRows)
    {
        if (outW_ == 0 || N_ == 0 || Cin_ == 0 || Cout_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t row = blockIdx; row < numRows; row += blockNum) {
            // Decode (n, co, od, oh) from row index
            uint64_t t = row;
            const uint32_t oh = static_cast<uint32_t>(t % outH_); t /= outH_;
            const uint32_t od = static_cast<uint32_t>(t % outD_); t /= outD_;
            const uint32_t co = static_cast<uint32_t>(t % Cout_); t /= Cout_;
            const uint32_t n  = static_cast<uint32_t>(t % N_);

            // Slide over width dimension in tiles
            for (uint32_t owBase = 0; owBase < outW_; owBase += tileW_) {
                uint32_t curTile = outW_ - owBase;
                if (curTile > tileW_) curTile = tileW_;

                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, curTile);

                // Accumulate over input channels and kernel spatial dims
                for (uint32_t ci = 0; ci < Cin_; ++ci) {
                    // Preload full kernel for (ci, co) into local buffer (kD*kH*kW elements)
                    AscendC::LocalTensor<float> wLocal = tmpQueue_.AllocTensor<float>();
                    const uint64_t wBaseOff = ((((uint64_t)ci) * Cout_) + co) * (uint64_t)(kD_ * kH_ * kW_);
                    AscendC::DataCopy(wLocal, wGm_[wBaseOff], kD_ * kH_ * kW_);

                    for (uint32_t kz = 0; kz < kD_; ++kz) {
                        const int32_t d_in = static_cast<int32_t>(od) - static_cast<int32_t>(kz);
                        if (d_in < 0 || d_in >= static_cast<int32_t>(D_)) continue;
                        for (uint32_t ky = 0; ky < kH_; ++ky) {
                            const int32_t h_in = static_cast<int32_t>(oh) - static_cast<int32_t>(ky);
                            if (h_in < 0 || h_in >= static_cast<int32_t>(H_)) continue;
                            for (uint32_t kx = 0; kx < kW_; ++kx) {
                                // Valid ow where 0 <= ow - kx < W => ow in [kx, kx+W-1]
                                int32_t s = static_cast<int32_t>(owBase);
                                int32_t e = static_cast<int32_t>(owBase + curTile - 1);
                                const int32_t validL = static_cast<int32_t>(kx);
                                const int32_t validR = static_cast<int32_t>(kx) + static_cast<int32_t>(W_) - 1;
                                if (s < validL) s = validL;
                                if (e > validR) e = validR;
                                if (s > e) continue;

                                const uint32_t segLen = static_cast<uint32_t>(e - s + 1);
                                const uint32_t w_in_start = static_cast<uint32_t>(s - static_cast<int32_t>(kx));

                                // Load input slice along width (contiguous)
                                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                                const uint64_t xOff =
                                    (((((uint64_t)n) * Cin_ + ci) * D_ + (uint32_t)d_in) * H_ + (uint32_t)h_in) * W_ + w_in_start;
                                AscendC::DataCopy(xLocal, xGm_[xOff], segLen);

                                // Get weight scalar from local kernel buffer
                                const uint32_t wIdx = (kz * kH_ + ky) * kW_ + kx;
                                const float wVal = wLocal.GetValue(wIdx);

                                // Multiply input by scalar weight
                                AscendC::Muls(xLocal, xLocal, wVal, segLen);

                                // Accumulate into yLocal at offset (s - owBase)
                                const uint32_t yOff = static_cast<uint32_t>(s - static_cast<int32_t>(owBase));
                                AscendC::Add(yLocal[yOff], yLocal[yOff], xLocal, segLen);

                                inQueueX_.FreeTensor(xLocal);
                            }
                        }
                    }
                    tmpQueue_.FreeTensor(wLocal);
                }

                // Store y tile to GM
                const uint64_t yOff =
                    (((((uint64_t)n) * Cout_ + co) * outD_ + od) * outH_ + oh) * outW_ + owBase;
                AscendC::DataCopy(yGm_[yOff], yLocal, curTile);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUF_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t D_{0}, H_{0}, W_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};
    uint32_t outD_{0}, outH_{0}, outW_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.tileLenW);
    op.Process(tiling_data.numRows);
}
