namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
static inline uint32_t FloorPowerOfTwo(uint32_t x)
{
    if (x == 0u) return 0u;
    uint32_t p = 1u;
    while ((p << 1u) && ((p << 1u) <= x)) p <<= 1u;
    return p;
}
} // anonymous namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core conservative
// UB usage per K-tile: xSeg (K) + wSeg (K) + acc (K) + tmp (K) => 4 * K * 4 bytes
static constexpr uint32_t BYTES_PER_ELEM = 4;
static constexpr uint32_t K_BUFFERS = 4;
static constexpr uint32_t MAX_TILEK_CAP = 4096;  // power-of-two, friendly for reduction
static constexpr uint32_t MIN_TILEK = 256;       // ensure reasonable vector length
static constexpr uint32_t DEFAULT_TILEN = 64;    // columns processed per tile

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmSwishDivideClampTanhClampTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    // Expect x: [M, K], w: [N, K], b: [N]
    const int64_t m64 = xShape->GetOriginShape().GetDim(0);
    const int64_t k64 = xShape->GetOriginShape().GetDim(1);
    const int64_t n64 = context->GetInputShape(1)->GetOriginShape().GetDim(0);

    const uint32_t M = static_cast<uint32_t>(m64);
    const uint32_t K = static_cast<uint32_t>(k64);
    const uint32_t N = static_cast<uint32_t>(n64);

    // Choose tileK based on UB budget and caps
    uint32_t maxTileKFromUB = UB_BUDGET_BYTES / (K_BUFFERS * BYTES_PER_ELEM);
    if (maxTileKFromUB > MAX_TILEK_CAP) maxTileKFromUB = MAX_TILEK_CAP;
    if (maxTileKFromUB < MIN_TILEK) maxTileKFromUB = MIN_TILEK;
    // Prefer power-of-two for easy reductions
    uint32_t tileK = FloorPowerOfTwo(maxTileKFromUB);
    if (tileK == 0u) tileK = MIN_TILEK;
    if (tileK > K && K > 0) {
        // still prefer pow2 not exceeding K
        uint32_t p2 = FloorPowerOfTwo(K);
        tileK = (p2 != 0u) ? p2 : K;
    }

    uint32_t tileN = DEFAULT_TILEN;
    if (tileN > N && N > 0) tileN = N;

    // Blocks: distribute rows (M) across blocks
    uint32_t blockDim = (M == 0) ? 1u : M;
    if (blockDim < 8u && blockDim < M) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t QNUM = 1;

class KernelGemmSwishDivideClampTanhClampCustom {
public:
    __aicore__ inline KernelGemmSwishDivideClampTanhClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileK, uint32_t tileN)
    {
        M_ = M; N_ = N; K_ = K;
        tileK_ = tileK; tileN_ = tileN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        pipe_.InitBuffer(xSegQ_, QNUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(wSegQ_, QNUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(accQ_,  QNUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_,  QNUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(yTileQ_, QNUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpTileQ_, QNUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(biasQ_, QNUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < M_; row += blockNum) {
            // Process N in tiles
            for (uint32_t colBase = 0; colBase < N_; colBase += tileN_) {
                uint32_t curN = N_ - colBase;
                if (curN > tileN_) curN = tileN_;

                AscendC::LocalTensor<float> yTile = yTileQ_.AllocTensor<float>();
                // Compute yTile[j] scalars for j in [0, curN)
                for (uint32_t j = 0; j < curN; ++j) {
                    AscendC::LocalTensor<float> acc = accQ_.AllocTensor<float>();
                    AscendC::Dup(acc, 0.0f, tileK_);

                    // Loop over K in segments
                    for (uint32_t kb = 0; kb < K_; kb += tileK_) {
                        const uint32_t curK = (K_ - kb < tileK_) ? (K_ - kb) : tileK_;
                        AscendC::LocalTensor<float> xSeg = xSegQ_.AllocTensor<float>();
                        AscendC::LocalTensor<float> wSeg = wSegQ_.AllocTensor<float>();
                        AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();

                        const uint64_t xOff = static_cast<uint64_t>(row) * K_ + kb;
                        const uint64_t wOff = static_cast<uint64_t>(colBase + j) * K_ + kb;

                        AscendC::DataCopy(xSeg, xGm_[xOff], curK);
                        AscendC::DataCopy(wSeg, wGm_[wOff], curK);

                        AscendC::Mul(tmp, xSeg, wSeg, curK);
                        AscendC::Add(acc, acc, tmp, curK);

                        tmpQ_.FreeTensor(tmp);
                        wSegQ_.FreeTensor(wSeg);
                        xSegQ_.FreeTensor(xSeg);
                    }

                    // Horizontal reduce sum of 'acc' to scalar at acc[0]
                    uint32_t len = tileK_;
                    while (len > 1) {
                        const uint32_t half = len >> 1;
                        if (half > 0) {
                            AscendC::LocalTensor<float> acc0 = acc[0];
                            AscendC::LocalTensor<float> acch = acc[half];
                            AscendC::Add(acc0, acc0, acch, half);
                        }
                        if ((len & 1u) != 0u) {
                            // add the tail element at index len - 1 into acc[0]
                            AscendC::LocalTensor<float> acc0 = acc[0];
                            AscendC::LocalTensor<float> accl = acc[len - 1];
                            AscendC::Add(acc0, acc0, accl, 1);
                        }
                        len = half;
                    }

                    // yTile[j] = acc[0] + b[colBase + j]
                    AscendC::LocalTensor<float> yj = yTile[j];
                    AscendC::LocalTensor<float> acc0 = acc[0];
                    AscendC::DataCopy(yj, acc0, 1);

                    AscendC::LocalTensor<float> bj = biasQ_.AllocTensor<float>();
                    AscendC::DataCopy(bj, bGm_[colBase + j], 1);
                    AscendC::Add(yj, yj, bj, 1);
                    biasQ_.FreeTensor(bj);

                    accQ_.FreeTensor(acc);
                }

                // Apply fused activation chain on yTile[0:curN]
                // Swish: y = y * sigmoid(y)
                AscendC::LocalTensor<float> tmpTile = tmpTileQ_.AllocTensor<float>();
                AscendC::Sigmoid(tmpTile, yTile, curN);
                AscendC::Mul(yTile, yTile, tmpTile, curN);

                // Divide by 2
                AscendC::Muls(yTile, yTile, 0.5f, curN);

                // Clamp to [-1, 1]
                AscendC::Maxs(yTile, yTile, -1.0f, curN);
                AscendC::Mins(yTile, yTile,  1.0f, curN);

                // Tanh
                AscendC::Tanh(yTile, yTile, curN);

                // Clamp again to [-1, 1]
                AscendC::Maxs(yTile, yTile, -1.0f, curN);
                AscendC::Mins(yTile, yTile,  1.0f, curN);

                // Store results
                const uint64_t yOff = static_cast<uint64_t>(row) * N_ + colBase;
                AscendC::DataCopy(yGm_[yOff], yTile, curN);

                tmpTileQ_.FreeTensor(tmpTile);
                yTileQ_.FreeTensor(yTile);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> xSegQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> wSegQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> accQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> yTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> tmpTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, QNUM> biasQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileK_{0}, tileN_{0};
};

extern "C" __global__ __aicore__ void gemm_swish_divide_clamp_tanh_clamp_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSwishDivideClampTanhClampCustom op;
    op.Init(x, w, b, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
