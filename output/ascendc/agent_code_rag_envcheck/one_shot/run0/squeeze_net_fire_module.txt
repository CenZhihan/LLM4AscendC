kernel_src="""

#include "kernel_operator.h"

class KernelSqueezeNetFireModuleCustom {
public:
    __aicore__ inline KernelSqueezeNetFireModuleCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w_sq, GM_ADDR b_sq,
                                GM_ADDR w_e1, GM_ADDR b_e1,
                                GM_ADDR w_e3, GM_ADDR b_e3,
                                GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t Csq, uint32_t E1, uint32_t E3)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        Csq_ = Csq; E1_ = E1; E3_ = E3; Cout_ = E1_ + E3_;

        // Set GM buffers
        uint64_t xSize = (uint64_t)N_ * C_ * H_ * W_;
        uint64_t ySize = (uint64_t)N_ * Cout_ * H_ * W_;
        uint64_t wSqSize = (uint64_t)Csq_ * C_;
        uint64_t bSqSize = (uint64_t)Csq_;
        uint64_t wE1Size = (uint64_t)E1_ * Csq_;
        uint64_t bE1Size = (uint64_t)E1_;
        uint64_t wE3Size = (uint64_t)E3_ * Csq_ * 3 * 3;
        uint64_t bE3Size = (uint64_t)E3_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wSqGm_.SetGlobalBuffer((__gm__ float*)w_sq, wSqSize);
        bSqGm_.SetGlobalBuffer((__gm__ float*)b_sq, bSqSize);
        wE1Gm_.SetGlobalBuffer((__gm__ float*)w_e1, wE1Size);
        bE1Gm_.SetGlobalBuffer((__gm__ float*)b_e1, bE1Size);
        wE3Gm_.SetGlobalBuffer((__gm__ float*)w_e3, wE3Size);
        bE3Gm_.SetGlobalBuffer((__gm__ float*)b_e3, bE3Size);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // UB buffers
        // weights
        pipe_.InitBuffer(wSqQ_, 1, wSqSize * sizeof(float));
        pipe_.InitBuffer(bSqQ_, 1, bSqSize * sizeof(float));
        pipe_.InitBuffer(wE1Q_, 1, wE1Size * sizeof(float));
        pipe_.InitBuffer(bE1Q_, 1, bE1Size * sizeof(float));
        pipe_.InitBuffer(wE3Q_, 1, wE3Size * sizeof(float));
        pipe_.InitBuffer(bE3Q_, 1, bE3Size * sizeof(float));

        // x row (C * W)
        pipe_.InitBuffer(xRowQ_, 1, (uint64_t)C_ * W_ * sizeof(float));
        // s ring (3 rows of Csq * W) - we'll allocate three separate slots
        pipe_.InitBuffer(sRowQ_, 3, (uint64_t)Csq_ * W_ * sizeof(float));
        // out row buffer (W)
        pipe_.InitBuffer(outRowQ_, 1, (uint64_t)W_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || H_ == 0 || W_ == 0) return;

        // Load weights once into UB
        AscendC::LocalTensor<float> wSqLocal = wSqQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> bSqLocal = bSqQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> wE1Local = wE1Q_.AllocTensor<float>();
        AscendC::LocalTensor<float> bE1Local = bE1Q_.AllocTensor<float>();
        AscendC::LocalTensor<float> wE3Local = wE3Q_.AllocTensor<float>();
        AscendC::LocalTensor<float> bE3Local = bE3Q_.AllocTensor<float>();

        AscendC::DataCopy(wSqLocal, wSqGm_, wSqLocal.GetSize());
        AscendC::DataCopy(bSqLocal, bSqGm_, bSqLocal.GetSize());
        AscendC::DataCopy(wE1Local, wE1Gm_, wE1Local.GetSize());
        AscendC::DataCopy(bE1Local, bE1Gm_, bE1Local.GetSize());
        AscendC::DataCopy(wE3Local, wE3Gm_, wE3Local.GetSize());
        AscendC::DataCopy(bE3Local, bE3Gm_, bE3Local.GetSize());

        // Grid-stride over rows
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t totalRows = (uint64_t)N_ * H_;

        for (uint64_t rowIndex = blockIdx; rowIndex < totalRows; rowIndex += blockNum) {
            const uint32_t n = static_cast<uint32_t>(rowIndex / H_);
            const uint32_t h = static_cast<uint32_t>(rowIndex % H_);

            // Prepare s rows for h-1, h, h+1
            AscendC::LocalTensor<float> sTop = sRowQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> sMid = sRowQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> sBot = sRowQ_.AllocTensor<float>();
            ComputeSRow(n, (h == 0) ? -1 : (int32_t)h - 1, wSqLocal, bSqLocal, sTop);
            ComputeSRow(n, (int32_t)h,               wSqLocal, bSqLocal, sMid);
            ComputeSRow(n, (h + 1 >= H_) ? -1 : (int32_t)h + 1, wSqLocal, bSqLocal, sBot);

            // Output buffers
            AscendC::LocalTensor<float> outRow = outRowQ_.AllocTensor<float>();

            // Expand1x1 branch (E1 channels): per output channel, fill row then write
            for (uint32_t oc = 0; oc < E1_; ++oc) {
                // Compute outRow[w] = ReLU(sum_{cs} sMid[cs, w] * wE1[oc, cs] + bE1[oc])
                for (uint32_t w = 0; w < W_; ++w) {
                    float acc = bE1Local.GetValue(oc);
                    for (uint32_t cs = 0; cs < Csq_; ++cs) {
                        float sval = sMid.GetValue(cs * W_ + w);
                        float wv = wE1Local.GetValue(oc * Csq_ + cs);
                        acc += sval * wv;
                    }
                    if (acc < 0.0f) acc = 0.0f;
                    outRow.SetValue(w, acc);
                }
                // Store to GM: y[n, oc, h, :] with channel offset 0..E1_-1
                const uint64_t base = YOffset(n, oc, h, 0);
                AscendC::DataCopy(yGm_[base], outRow, W_);
            }

            // Expand3x3 branch (E3 channels): per oc, compute 3x3 over sTop/sMid/sBot, padding=1
            for (uint32_t oc = 0; oc < E3_; ++oc) {
                for (uint32_t w = 0; w < W_; ++w) {
                    float acc = bE3Local.GetValue(oc);
                    for (uint32_t cs = 0; cs < Csq_; ++cs) {
                        // ky = -1,0,1
                        for (int ky = -1; ky <= 1; ++ky) {
                            AscendC::LocalTensor<float> &rowRef = (ky == -1) ? sTop : ((ky == 0) ? sMid : sBot);
                            for (int kx = -1; kx <= 1; ++kx) {
                                int32_t wx = (int32_t)w + kx;
                                float sval = 0.0f;
                                if (wx >= 0 && wx < (int32_t)W_) {
                                    sval = rowRef.GetValue(cs * W_ + (uint32_t)wx);
                                }
                                uint32_t kyIdx = (uint32_t)(ky + 1);
                                uint32_t kxIdx = (uint32_t)(kx + 1);
                                uint64_t wIdx = (((uint64_t)oc * Csq_ + cs) * 3 + kyIdx) * 3 + kxIdx;
                                float wv = wE3Local.GetValue(wIdx);
                                acc += sval * wv;
                            }
                        }
                    }
                    if (acc < 0.0f) acc = 0.0f;
                    outRow.SetValue(w, acc);
                }
                // Store to GM: y[n, E1_ + oc, h, :]
                const uint64_t base = YOffset(n, E1_ + oc, h, 0);
                AscendC::DataCopy(yGm_[base], outRow, W_);
            }

            outRowQ_.FreeTensor(outRow);
            sRowQ_.FreeTensor(sBot);
            sRowQ_.FreeTensor(sMid);
            sRowQ_.FreeTensor(sTop);
        }

        // Free weight locals
        wSqQ_.FreeTensor(wSqLocal);
        bSqQ_.FreeTensor(bSqLocal);
        wE1Q_.FreeTensor(wE1Local);
        bE1Q_.FreeTensor(bE1Local);
        wE3Q_.FreeTensor(wE3Local);
        bE3Q_.FreeTensor(bE3Local);
    }

private:
    __aicore__ inline void ComputeSRow(uint32_t n, int32_t hRow,
                                       AscendC::LocalTensor<float> &wSqLocal,
                                       AscendC::LocalTensor<float> &bSqLocal,
                                       AscendC::LocalTensor<float> &sOut)
    {
        // If hRow out of range, fill zeros
        if (hRow < 0 || hRow >= (int32_t)H_) {
            for (uint64_t i = 0; i < (uint64_t)Csq_ * W_; ++i) {
                sOut.SetValue(i, 0.0f);
            }
            return;
        }

        // Load x rows for all C channels into UB contiguous buffer: [C, W]
        AscendC::LocalTensor<float> xRows = xRowQ_.AllocTensor<float>();
        for (uint32_t ci = 0; ci < C_; ++ci) {
            const uint64_t base = XOffset(n, ci, (uint32_t)hRow, 0);
            AscendC::DataCopy(xRows[(uint64_t)ci * W_], xGm_[base], W_);
        }

        // For each output squeeze channel and width position, compute 1x1 conv + bias + ReLU
        for (uint32_t oc = 0; oc < Csq_; ++oc) {
            const float b = bSqLocal.GetValue(oc);
            for (uint32_t w = 0; w < W_; ++w) {
                float acc = b;
                for (uint32_t ci = 0; ci < C_; ++ci) {
                    float xv = xRows.GetValue((uint64_t)ci * W_ + w);
                    float wv = wSqLocal.GetValue((uint64_t)oc * C_ + ci);
                    acc += xv * wv;
                }
                if (acc < 0.0f) acc = 0.0f;
                sOut.SetValue((uint64_t)oc * W_ + w, acc);
            }
        }

        xRowQ_.FreeTensor(xRows);
    }

    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * C_ + c) * H_ + h) * W_ + w;
    }

    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * Cout_ + c) * H_ + h) * W_ + w;
    }

private:
    AscendC::TPipe pipe_;
    // Queues
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wSqQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bSqQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wE1Q_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bE1Q_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wE3Q_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bE3Q_;

    AscendC::TQue<AscendC::TPosition::VECIN, 1> xRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3> sRowQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outRowQ_;

    // GM tensors
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wSqGm_;
    AscendC::GlobalTensor<float> bSqGm_;
    AscendC::GlobalTensor<float> wE1Gm_;
    AscendC::GlobalTensor<float> bE1Gm_;
    AscendC::GlobalTensor<float> wE3Gm_;
    AscendC::GlobalTensor<float> bE3Gm_;
    AscendC::GlobalTensor<float> yGm_;

    // dims
    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t Csq_{0}, E1_{0}, E3_{0}, Cout_{0};
};

extern "C" __global__ __aicore__ void squeeze_net_fire_module_custom(GM_ADDR x,
                                                                     GM_ADDR w_squeeze,
                                                                     GM_ADDR b_squeeze,
                                                                     GM_ADDR w_expand1,
                                                                     GM_ADDR b_expand1,
                                                                     GM_ADDR w_expand3,
                                                                     GM_ADDR b_expand3,
                                                                     GM_ADDR y,
                                                                     GM_ADDR workspace,
                                                                     GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSqueezeNetFireModuleCustom op;
    op.Init(x, w_squeeze, b_squeeze, w_expand1, b_expand1, w_expand3, b_expand3, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.csq, tiling_data.e1, tiling_data.e3);
    op.Process();
}
