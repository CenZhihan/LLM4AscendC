host_operator_src="""
#include "matmul_with_small_k_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// We leverage small-K property to cache B tile and stream A rows.
// UB usage (floats): K*tileN + 2*tileN + K  (B_tile + yVec + tmpVec + aRow)
// UB bytes ~= 4 * (K*tileN + 2*tileN + K) <= UB_BUDGET_BYTES
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_TILE_N = 256;     // favor wider DMA bursts
static constexpr uint32_t MIN_TILE_N = 128;
static constexpr uint32_t MAX_BLOCK_DIM = 24;
static constexpr uint32_t MIN_BLOCK_DIM = 8;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithSmallKDimensionCustomTilingData tiling;

    const auto *shapeA = context->GetInputShape(0);
    const auto *shapeB = context->GetInputShape(1);
    if (shapeA->GetDimNum() != 2 || shapeB->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(shapeA->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(shapeA->GetDim(1));
    const uint32_t N = static_cast<uint32_t>(shapeB->GetDim(1));

    // Choose tileN from UB budget
    const uint32_t budgetFloats = UB_BUDGET_BYTES / 4;
    uint32_t tileN = 0;
    if (K + 2 > 0) {
        const int64_t tmp = static_cast<int64_t>(budgetFloats) - static_cast<int64_t>(K);
        int64_t maxTileN = tmp > 0 ? (tmp / static_cast<int64_t>(K + 2)) : 0;
        if (maxTileN < static_cast<int64_t>(MIN_TILE_N)) {
            maxTileN = MIN_TILE_N;
        }
        tileN = static_cast<uint32_t>(maxTileN);
    } else {
        tileN = MIN_TILE_N;
    }

    // Align and clamp
    tileN = AlignDown(tileN, ALIGN_TILE_N);
    if (tileN == 0) tileN = MIN_TILE_N;
    if (tileN > N) {
        // align down but keep >0
        tileN = AlignDown(N, ALIGN_TILE_N);
        if (tileN == 0) tileN = N;
    }

    // tileM does not affect UB footprint in our algorithm; set moderately large for reuse
    uint32_t tileM = 128;
    if (tileM > M) tileM = M;

    const uint32_t numTilesM = (tileM == 0) ? 1u : CeilDiv(M, tileM);
    const uint32_t numTilesN = (tileN == 0) ? 1u : CeilDiv(N, tileN);
    uint32_t totalTiles = numTilesM * numTilesN;

    uint32_t blockDim = totalTiles;
    if (blockDim < MIN_BLOCK_DIM) blockDim = MIN_BLOCK_DIM;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;
    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_numTilesM(numTilesM);
    tiling.set_numTilesN(numTilesN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    // A: [M,K], B: [K,N] => C: [M,N]
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, a_shape->GetDim(0));
    c_shape->SetDim(1, b_shape->GetDim(1));
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Keep float32 throughout
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != ge::DT_FLOAT || b_dtype != ge::DT_FLOAT) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_ONE = 1;

class KernelMatmulWithSmallKDimensionCustom {
public:
    __aicore__ inline KernelMatmulWithSmallKDimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN,
                                uint32_t numTilesM, uint32_t numTilesN)
    {
        M_ = M; N_ = N; K_ = K;
        tileM_ = tileM; tileN_ = tileN;
        numTilesM_ = numTilesM; numTilesN_ = numTilesN;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(M_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(K_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(M_) * N_);

        // UB buffers:
        // - bRow buffer for tileN_ elements, but we will store the entire B tile as K_ rows sequentially
        // - yVec: tileN_
        // - tmpVec: tileN_
        // - aRow: K_
        // B tile buffer size in elements: K_ * tileN_
        pipe_.InitBuffer(bTileQueue_, BUFFER_ONE, static_cast<uint32_t>(K_) * tileN_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_ONE, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_ONE, tileN_ * sizeof(float));
        pipe_.InitBuffer(aRowQueue_, BUFFER_ONE, K_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalTiles = numTilesM_ * numTilesN_;

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t mIdx = tileId / numTilesN_;
            const uint32_t nIdx = tileId % numTilesN_;

            const uint32_t mStart = mIdx * tileM_;
            const uint32_t nStart = nIdx * tileN_;
            const uint32_t mEff = AscendC::min(static_cast<uint32_t>(tileM_), M_ - mStart);
            const uint32_t nEff = AscendC::min(static_cast<uint32_t>(tileN_), N_ - nStart);

            // Cache B tile (K_ rows x nEff cols) in UB once per tile
            AscendC::LocalTensor<float> bTileLocal = bTileQueue_.AllocTensor<float>();
            for (uint32_t kk = 0; kk < K_; ++kk) {
                // Destination slice in UB
                AscendC::LocalTensor<float> bRowLocal = bTileLocal[kk * tileN_];
                AscendC::DataCopy(bRowLocal, bGm_[static_cast<uint64_t>(kk) * N_ + nStart], nEff);
            }

            // For each row of A within this tile
            for (uint32_t i = 0; i < mEff; ++i) {
                // Load A row (length K_)
                AscendC::LocalTensor<float> aRowLocal = aRowQueue_.AllocTensor<float>();
                AscendC::DataCopy(aRowLocal, aGm_[static_cast<uint64_t>(mStart + i) * K_], K_);

                // Prepare y accumulator
                AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, nEff);

                // Accumulate y += a[i,k] * b_row_k
                for (uint32_t kk = 0; kk < K_; ++kk) {
                    AscendC::LocalTensor<float> bRowLocal = bTileLocal[kk * tileN_];

                    // Read scalar a[i,kk] from UB
                    float aScalar = aRowLocal.GetValue(kk);

                    AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                    // tmp = bRow
                    AscendC::DataCopy(tmpLocal, bRowLocal, nEff);
                    // tmp *= aScalar
                    AscendC::Muls(tmpLocal, tmpLocal, aScalar, nEff);
                    // y += tmp
                    AscendC::Add(yLocal, yLocal, tmpLocal, nEff);
                    tmpQueue_.FreeTensor(tmpLocal);
                }

                // Write y row to GM C
                AscendC::DataCopy(cGm_[static_cast<uint64_t>(mStart + i) * N_ + nStart], yLocal, nEff);

                yQueue_.FreeTensor(yLocal);
                aRowQueue_.FreeTensor(aRowLocal);
            }

            bTileQueue_.FreeTensor(bTileLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_ONE> bTileQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_ONE> aRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_ONE> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_ONE> yQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileM_{0}, tileN_{0};
    uint32_t numTilesM_{0}, numTilesN_{0};
};

extern "C" __global__ __aicore__ void matmul_with_small_k_dimension_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithSmallKDimensionCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN,
            tiling_data.numTilesM, tiling_data.numTilesN);
    op.Process();
}
