kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

template <typename T>
__aicore__ inline void gelu_approx(AscendC::LocalTensor<T>& out, const AscendC::LocalTensor<T>& in, uint32_t len)
{
    // GELU approximation: 0.5 * x * (1 + tanh( sqrt(2/pi) * (x + 0.044715 * x^3) ))
    // sqrt(2/pi) ~ 0.7978845608
    const float kAlpha = 0.7978845608f;
    const float kBeta = 0.044715f;

    AscendC::LocalTensor<T> x3 = AscendC::AllocLocalTensor<T>(AscendC::TPosition::VECCALC, len);
    AscendC::LocalTensor<T> t  = AscendC::AllocLocalTensor<T>(AscendC::TPosition::VECCALC, len);
    AscendC::LocalTensor<T> u  = AscendC::AllocLocalTensor<T>(AscendC::TPosition::VECCALC, len);

    // x3 = x^3
    AscendC::Mul(x3, in, in, len);
    AscendC::Mul(x3, x3, in, len);

    // t = x + 0.044715 * x^3
    AscendC::Muls(t, x3, kBeta, len);
    AscendC::Add(t, t, in, len);

    // u = tanh(0.79788456 * t)
    AscendC::Muls(u, t, kAlpha, len);
    AscendC::Tanh(u, u, len);

    // out = 0.5 * x * (1 + u)
    AscendC::Adds(u, u, static_cast<T>(1.0f), len);
    AscendC::Mul(out, in, u, len);
    AscendC::Muls(out, out, static_cast<T>(0.5f), len);

    AscendC::FreeLocalTensor(u);
    AscendC::FreeLocalTensor(t);
    AscendC::FreeLocalTensor(x3);
}

class KernelMatmulDivideGeluCustom {
public:
    __aicore__ inline KernelMatmulDivideGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                GM_ADDR workspace,
                                const TCubeTiling& cubeTiling,
                                uint32_t totalLength, uint32_t tileLength,
                                float divisor)
    {
        totalLen_ = totalLength;
        tileLen_ = tileLength;
        invDiv_ = 1.0f / divisor;

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        // Initialize pipeline queues for post-processing
        pipe_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 2, tileLen_ * sizeof(float));

        // Configure and run Cube MatMul (X * W^T + B) into y (C)
        using A_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD>;
        using B_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::BSNGD>;
        using C_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD>;
        using BIAS_TYPE = AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>;

        batchMatmulKernel_ = AscendC::UniquePtr<BatchMatmulKernel<A_TYPE, B_TYPE, C_TYPE, BIAS_TYPE>>();
        batchMatmulKernel_.Construct();

        tiling_ = cubeTiling;
        // Share all on-chip buffers to matmul
        tiling_.shareMode = 0;
        tiling_.shareL1Size = FULL_L1_SIZE;
        tiling_.shareL0CSize = FULL_L0C_SIZE;
        tiling_.shareUbSize = 0;

        REGIST_MATMUL_OBJ(&pipe_, GetSysWorkSpacePtr(), batchMatmulKernel_->matmulObj, &tiling_);
        batchMatmulKernel_->Init(x, w, b, y, workspace, tiling_);
    }

    __aicore__ inline void Process()
    {
        // Phase 1: GEMM with bias to y
        const int batchNum = tiling_.BatchNum;
        batchMatmulKernel_->Process<true>(&pipe_, batchNum, batchNum);

        // Phase 2: Post-processing y = GELU( (y) / divisor )
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        CopyInY(base, curLen);
        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyInY(nextBase, nextLen);
            }

            Compute(curLen);
            CopyOutY(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyInY(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void Compute(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocalIn  = inQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> yLocalOut = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal    = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> gLocal    = tmpQueue_.AllocTensor<float>();

        // t = y / divisor
        AscendC::Muls(tLocal, yLocalIn, invDiv_, len);

        // g = GELU(t)
        gelu_approx(gLocal, tLocal, len);

        // yOut = g
        AscendC::DataCopy(yLocalOut, gLocal, len);

        tmpQueue_.FreeTensor(gLocal);
        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue(yLocalOut);
        inQueueY_.FreeTensor(yLocalIn);
    }

    __aicore__ inline void CopyOutY(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocalOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocalOut, len);
        outQueueY_.FreeTensor(yLocalOut);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>  inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2>         tmpQueue_;

    AscendC::GlobalTensor<float> yGm_;

    AscendC::UniquePtr<BatchMatmulKernel<
        AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD>,
        AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::BSNGD>,
        AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::BSNGD>,
        AscendC::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>
    >> batchMatmulKernel_;

    TCubeTiling tiling_;
    uint32_t totalLen_{0};
    uint32_t tileLen_{0};
    float invDiv_{1.0f};
};

extern "C" __global__ __aicore__ void matmul_divide_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tilingData, tilingGm);

    KernelMatmulDivideGeluCustom op;
    op.Init(x, w, b, y,
            workspace,
            tilingData.cubeTilingData,
            tilingData.totalLength,
            tilingData.tileLength,
            tilingData.divisor);
    op.Process();
}
