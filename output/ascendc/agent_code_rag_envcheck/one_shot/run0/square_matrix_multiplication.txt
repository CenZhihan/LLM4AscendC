host_operator_src="""

#include "square_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

// A, B, C all float32. UB model for naive blocked GEMM (single-buffer per tile):
//   A_tile: tileM * tileK
//   B_tile: tileK * tileN
//   C_tile: tileM * tileN
// Total floats = tileM*tileK + tileK*tileN + tileM*tileN
// Bytes = 4 * (tileM*tileK + tileK*tileN + tileM*tileN)
// We choose tileM = tileN = tileK = ts for simplicity.
// Then bytes = 12 * ts^2.
// Keep within a conservative UB budget to accommodate queue overheads.
static constexpr uint32_t UB_BUDGET_BYTES = 128 * 1024; // conservative per-core share
static constexpr uint32_t ALIGN_TS = 16;                // favor vector-friendly sizes
static constexpr uint32_t TS_MAX = 128;                 // cap tile size to avoid long inner loops
static constexpr uint32_t TS_MIN = 32;                  // avoid too-small tiles

} // anonymous namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SquareMatrixMultiplicationCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }
    const uint64_t N0 = static_cast<uint64_t>(a_shape->GetDim(0));
    const uint64_t N1 = static_cast<uint64_t>(a_shape->GetDim(1));
    const uint64_t M0 = static_cast<uint64_t>(b_shape->GetDim(0));
    const uint64_t M1 = static_cast<uint64_t>(b_shape->GetDim(1));
    if (N0 != N1 || M0 != M1 || N0 != M0) {
        return ge::GRAPH_FAILED; // only square NxN supported
    }
    const uint32_t N = static_cast<uint32_t>(N0);

    // Compute tile size ts from UB budget: 12 * ts^2 <= UB_BUDGET_BYTES
    uint32_t ts = static_cast<uint32_t>(::floor(::sqrt(static_cast<double>(UB_BUDGET_BYTES) / 12.0)));
    ts = AlignDown(ts, ALIGN_TS);
    if (ts > TS_MAX) ts = TS_MAX;
    if (ts < TS_MIN) ts = TS_MIN;
    if (ts > N) {
        // If N is small, shrink tile size to N aligned down
        const uint32_t aligned = AlignDown(N, ALIGN_TS);
        ts = aligned > 0 ? aligned : N;
    }

    const uint32_t tileM = ts;
    const uint32_t tileN = ts;
    const uint32_t tileK = ts;

    const uint32_t tilesM = CeilDiv(N, tileM);
    const uint32_t tilesN = CeilDiv(N, tileN);
    const uint64_t totalTiles = static_cast<uint64_t>(tilesM) * static_cast<uint64_t>(tilesN);

    // Choose blockDim: spread tiles across cores conservatively
    uint32_t blockDim = 24u; // typical max on 910/910B for AI Core
    if (totalTiles < blockDim) {
        blockDim = static_cast<uint32_t>(totalTiles);
        if (blockDim == 0u) blockDim = 1u;
    }

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace needed
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t QUEUE_DEPTH = 1;

static __aicore__ inline uint32_t CeilDivU32(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

class KernelSquareMatrixMultiplicationCustom {
public:
    __aicore__ inline KernelSquareMatrixMultiplicationCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t N, uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        N_ = N;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, static_cast<uint64_t>(N_) * N_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, static_cast<uint64_t>(N_) * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, static_cast<uint64_t>(N_) * N_);

        // Allocate UB buffers for tiles
        pipe_.InitBuffer(aQueue_, QUEUE_DEPTH, static_cast<uint64_t>(tileM_) * tileK_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, QUEUE_DEPTH, static_cast<uint64_t>(tileK_) * tileN_ * sizeof(float));
        pipe_.InitBuffer(cQueue_, QUEUE_DEPTH, static_cast<uint64_t>(tileM_) * tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;

        const uint32_t tilesM = CeilDivU32(N_, tileM_);
        const uint32_t tilesN = CeilDivU32(N_, tileN_);
        const uint64_t totalTiles = static_cast<uint64_t>(tilesM) * tilesN;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each core processes tile indices in a grid-stride manner
        for (uint64_t t = blockIdx; t < totalTiles; t += blockNum) {
            const uint32_t tileRow = static_cast<uint32_t>(t / tilesN);
            const uint32_t tileCol = static_cast<uint32_t>(t % tilesN);

            const uint32_t mStart = tileRow * tileM_;
            const uint32_t nStart = tileCol * tileN_;
            const uint32_t mLen = (mStart + tileM_ <= N_) ? tileM_ : (N_ - mStart);
            const uint32_t nLen = (nStart + tileN_ <= N_) ? tileN_ : (N_ - nStart);

            // Allocate C local tile and zero-initialize
            AscendC::LocalTensor<float> cLocal = cQueue_.AllocTensor<float>();
            DupZeros(cLocal, static_cast<uint32_t>(tileM_ * tileN_));
            // Only first mLen * nLen used for boundary tiles

            // Accumulate across K dimension
            for (uint32_t kStart = 0; kStart < N_; kStart += tileK_) {
                const uint32_t kLen = (kStart + tileK_ <= N_) ? tileK_ : (N_ - kStart);

                // A sub-tile: shape mLen x kLen, row-major
                AscendC::LocalTensor<float> aLocal = aQueue_.AllocTensor<float>();
                for (uint32_t i = 0; i < mLen; ++i) {
                    const uint32_t aGmOffset = (mStart + i) * N_ + kStart;
                    AscendC::DataCopy(aLocal[i * tileK_], aGm_[aGmOffset], kLen);
                }
                // B sub-tile: shape kLen x nLen, row-major
                AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();
                for (uint32_t k = 0; k < kLen; ++k) {
                    const uint32_t bGmOffset = (kStart + k) * N_ + nStart;
                    AscendC::DataCopy(bLocal[k * tileN_], bGm_[bGmOffset], nLen);
                }

                // Compute: C[m,n] += sum_k A[m,k] * B[k,n]
                // Only operate on mLen x nLen and kLen slices
                for (uint32_t i = 0; i < mLen; ++i) {
                    for (uint32_t k = 0; k < kLen; ++k) {
                        const float aVal = aLocal[i * tileK_ + k];
                        // Fused multiply-add across the N dimension of the tile
                        float* cRowPtr = &cLocal[i * tileN_];
                        const float* bRowPtr = &bLocal[k * tileN_];
                        for (uint32_t j = 0; j < nLen; ++j) {
                            cRowPtr[j] = AscendC::Fma(aVal, bRowPtr[j], cRowPtr[j]);
                        }
                    }
                }

                bQueue_.FreeTensor(bLocal);
                aQueue_.FreeTensor(aLocal);
            }

            // Write C tile back to GM
            for (uint32_t i = 0; i < mLen; ++i) {
                const uint32_t cGmOffset = (mStart + i) * N_ + nStart;
                AscendC::DataCopy(cGm_[cGmOffset], cLocal[i * tileN_], nLen);
            }
            cQueue_.FreeTensor(cLocal);
        }
    }

private:
    // Fill local tensor with zeros for 'size' elements
    __aicore__ inline void DupZeros(AscendC::LocalTensor<float>& t, uint32_t size)
    {
        // Prefer vectorized duplicate if available; fall back to manual set
        AscendC::Duplicate(t, 0.0f, size);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, QUEUE_DEPTH> aQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, QUEUE_DEPTH> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, QUEUE_DEPTH> cQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t N_{0};
    uint32_t tileM_{0};
    uint32_t tileN_{0};
    uint32_t tileK_{0};
};

extern "C" __global__ __aicore__ void square_matrix_multiplication_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSquareMatrixMultiplicationCustom op;
    op.Init(A, B, C, tiling_data.N, tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
"""
