kernel_src="""

#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 1;

class KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom {
public:
    __aicore__ inline KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t M, uint32_t K, uint32_t tileK,
                                float negativeSlope)
    {
        N_ = N;
        M_ = M;
        K_ = K;
        tileK_ = tileK == 0 ? 1 : tileK;
        negSlope_ = negativeSlope;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, M_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_);

        pipe_.InitBuffer(xQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(wQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(outQ_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || M_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < N_; row += blockNum) {
            float maxVal = -3.4028235e38f;

            // Pass 1: compute max across M of (x_row * w_m^T + b_m)
            for (uint32_t m = 0; m < M_; ++m) {
                float acc = 0.0f;
                for (uint32_t ks = 0; ks < K_; ks += tileK_) {
                    uint32_t len = K_ - ks;
                    if (len > tileK_) len = tileK_;
                    AscendC::LocalTensor<float> xLocal = xQ_.AllocTensor<float>();
                    AscendC::LocalTensor<float> wLocal = wQ_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[row * K_ + ks], len);
                    AscendC::DataCopy(wLocal, wGm_[m * K_ + ks], len);
                    // Compute partial dot product on CPU-like loop over local tensor
                    for (uint32_t i = 0; i < len; ++i) {
                        // Assume operator[] indexing is available on LocalTensor
                        acc += xLocal.GetValue(i) * wLocal.GetValue(i);
                    }
                    wQ_.FreeTensor(wLocal);
                    xQ_.FreeTensor(xLocal);
                }
                // add bias b[m]
                {
                    AscendC::LocalTensor<float> bLocal = scalarQ_.AllocTensor<float>();
                    AscendC::DataCopy(bLocal, bGm_[m], 1);
                    acc += bLocal.GetValue(0);
                    scalarQ_.FreeTensor(bLocal);
                }
                if (acc > maxVal) maxVal = acc;
            }

            // Pass 2: compute sum(exp(s - maxVal))
            float sumExp = 0.0f;
            for (uint32_t m = 0; m < M_; ++m) {
                float acc = 0.0f;
                for (uint32_t ks = 0; ks < K_; ks += tileK_) {
                    uint32_t len = K_ - ks;
                    if (len > tileK_) len = tileK_;
                    AscendC::LocalTensor<float> xLocal = xQ_.AllocTensor<float>();
                    AscendC::LocalTensor<float> wLocal = wQ_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[row * K_ + ks], len);
                    AscendC::DataCopy(wLocal, wGm_[m * K_ + ks], len);
                    for (uint32_t i = 0; i < len; ++i) {
                        acc += xLocal.GetValue(i) * wLocal.GetValue(i);
                    }
                    wQ_.FreeTensor(wLocal);
                    xQ_.FreeTensor(xLocal);
                }
                AscendC::LocalTensor<float> bLocal = scalarQ_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[m], 1);
                acc += bLocal.GetValue(0);
                scalarQ_.FreeTensor(bLocal);
                float t = acc - maxVal;
                // use fast exponential for scalar
                sumExp += expf(t);
            }

            float lse = maxVal + logf(sumExp);

            // two LeakyReLU
            lse = (lse >= 0.0f) ? lse : (negSlope_ * lse);
            lse = (lse >= 0.0f) ? lse : (negSlope_ * lse);

            // two GELU (approx via tanh)
            lse = GeluApprox(lse);
            lse = GeluApprox(lse);

            // write result
            AscendC::LocalTensor<float> yLocal = outQ_.AllocTensor<float>();
            yLocal.SetValue(0, lse);
            AscendC::DataCopy(yGm_[row], yLocal, 1);
            outQ_.FreeTensor(yLocal);
        }
    }

private:
    __aicore__ inline float GeluApprox(float x) const
    {
        // 0.5*x*(1 + tanh(√(2/π)*(x + 0.044715x^3)))
        const float kAlpha = 0.7978845608028654f; // sqrt(2/pi)
        const float kCoef = 0.044715f;
        float x3 = x * x * x;
        float inner = kAlpha * (x + kCoef * x3);
        float th = tanhf(inner);
        return 0.5f * x * (1.0f + th);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scalarQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t tileK_ {1};
    float negSlope_ {0.01f};
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom op;
    op.Init(x, w, b, y,
            tiling_data.N, tiling_data.M, tiling_data.K, tiling_data.tileK,
            tiling_data.negativeSlope);
    op.Process();
}
