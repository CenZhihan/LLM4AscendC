namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;
static constexpr uint32_t MAX_PENDING_CAP = 2048;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling1dCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();
    const int64_t n64 = origin.GetDim(0);
    const int64_t c64 = origin.GetDim(1);
    const int64_t l64 = origin.GetDim(2);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernel_size_ptr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stride_ptr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padding_ptr = attrs->GetAttrPointer<int32_t>(2);

    const uint32_t N = static_cast<uint32_t>(std::max<int64_t>(n64, 1));
    const uint32_t C = static_cast<uint32_t>(std::max<int64_t>(c64, 1));
    const uint32_t L = static_cast<uint32_t>(std::max<int64_t>(l64, 0));

    const uint32_t kernel = static_cast<uint32_t>(std::max<int32_t>(*kernel_size_ptr, 1));
    const uint32_t stride = static_cast<uint32_t>(std::max<int32_t>(*stride_ptr, 1));
    const uint32_t padding = static_cast<uint32_t>(std::max<int32_t>(*padding_ptr, 0));

    int64_t numer = static_cast<int64_t>(L) + 2LL * static_cast<int64_t>(padding) - static_cast<int64_t>(kernel);
    int64_t out64 = (numer >= 0) ? (numer / static_cast<int64_t>(stride) + 1) : 0;
    const uint32_t Lout = static_cast<uint32_t>(std::max<int64_t>(out64, 0));

    // UB layout:
    //   inQueueX: 2 * tileLenIn
    //   outQueueY: 1 * outChunk
    //   tmpQueue (pending ring): pendingCap
    // Total floats: 3 * tileLenIn + pendingCap
    uint32_t pendingCap = std::min<uint32_t>(kernel, MAX_PENDING_CAP);
    uint32_t budgetFloats = UB_BUDGET_BYTES / 4;
    uint32_t maxTileIn = 0;
    if (budgetFloats > pendingCap) {
        maxTileIn = (budgetFloats - pendingCap) / 3;
    }
    maxTileIn = AlignDown(std::max<uint32_t>(maxTileIn, ALIGN_ELEMS), ALIGN_ELEMS);
    if (maxTileIn > HARD_CAP_ELEMS) maxTileIn = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    if (maxTileIn == 0) maxTileIn = ALIGN_ELEMS;

    uint32_t tileLenIn = maxTileIn;
    if (L < tileLenIn) {
        uint32_t aligned = AlignDown(L, ALIGN_ELEMS);
        tileLenIn = (aligned > 0) ? aligned : L;
    }
    uint32_t outChunk = tileLenIn;

    const uint32_t numItems = N * C;
    uint32_t blockDim = numItems;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_lenIn(L);
    tiling.set_lenOut(Lout);
    tiling.set_tileLenIn(tileLenIn);
    tiling.set_outChunk(outChunk);
    tiling.set_blockDim(blockDim);
    tiling.set_kernelSize(kernel);
    tiling.set_stride(stride);
    tiling.set_padding(padding);
    tiling.set_pendingCap(pendingCap);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUFFER_NUM = 2;

class KernelAveragePooling1dCustom {
public:
    __aicore__ inline KernelAveragePooling1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t lenIn, uint32_t lenOut,
                                uint32_t tileLenIn, uint32_t outChunk,
                                uint32_t kernelSize, uint32_t stride, uint32_t padding,
                                uint32_t pendingCap)
    {
        n_ = n;
        c_ = c;
        lenIn_ = lenIn;
        lenOut_ = lenOut;
        tileLenIn_ = tileLenIn;
        outChunk_ = outChunk;
        kernelSize_ = kernelSize;
        stride_ = stride;
        padding_ = padding;
        pendingCap_ = pendingCap > 0 ? pendingCap : 1;
        invKernel_ = 1.0f / static_cast<float>(kernelSize_);

        const uint32_t totalIn = n_ * c_ * lenIn_;
        const uint32_t totalOut = n_ * c_ * lenOut_;
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalIn);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut);

        pipe_.InitBuffer(inQueueX_, IN_BUFFER_NUM, tileLenIn_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, outChunk_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, pendingCap_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t numItems = n_ * c_;
        if (numItems == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t item = blockIdx; item < numItems; item += blockNum) {
            ProcessItem(item);
        }
    }

private:
    __aicore__ inline int32_t ClampInt(int32_t v, int32_t lo, int32_t hi)
    {
        return v < lo ? lo : (v > hi ? hi : v);
    }

    __aicore__ inline uint32_t AIdx(uint32_t t)  // clamp to [0, lenIn_]
    {
        int32_t a = static_cast<int32_t>(t * stride_) - static_cast<int32_t>(padding_);
        a = ClampInt(a, 0, static_cast<int32_t>(lenIn_));
        return static_cast<uint32_t>(a);
    }

    __aicore__ inline uint32_t BIdx(uint32_t t)  // clamp to [0, lenIn_]
    {
        int32_t b = static_cast<int32_t>(t * stride_) - static_cast<int32_t>(padding_) + static_cast<int32_t>(kernelSize_);
        b = ClampInt(b, 0, static_cast<int32_t>(lenIn_));
        return static_cast<uint32_t>(b);
    }

    __aicore__ inline void FlushY(AscendC::LocalTensor<float>& yLocal,
                                  uint32_t& yLocalSize,
                                  uint32_t gmYBase,
                                  uint32_t& yWritten)
    {
        if (yLocalSize == 0) return;
        AscendC::DataCopy(yGm_[gmYBase + yWritten], yLocal, yLocalSize);
        yWritten += yLocalSize;
        yLocalSize = 0;
    }

    __aicore__ inline void ProcessItem(uint32_t item)
    {
        const uint32_t gmXBase = item * lenIn_;
        const uint32_t gmYBase = item * lenOut_;

        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> pending = tmpQueue_.AllocTensor<float>();

        uint32_t head = 0;
        uint32_t tail = 0;
        const uint32_t cap = pendingCap_;

        float prefix = 0.0f;
        uint32_t tA = 0;
        uint32_t tB = 0;
        uint32_t yLocalSize = 0;
        uint32_t yWritten = 0;

        // Pre-handle a_idx == 0
        while (tA < lenOut_) {
            const uint32_t aidx = AIdx(tA);
            if (aidx == 0) {
                pending[tail % cap] = prefix; // which is 0 here
                ++tail;
                ++tA;
            } else {
                break;
            }
        }
        // Pre-handle b_idx == 0
        while (tB < lenOut_) {
            const uint32_t bidx = BIdx(tB);
            if (bidx == 0) {
                float sum = prefix - pending[head % cap]; // both 0
                ++head;
                float yv = sum * invKernel_;
                yLocal[yLocalSize++] = yv;
                if (yLocalSize == outChunk_) {
                    FlushY(yLocal, yLocalSize, gmYBase, yWritten);
                }
                ++tB;
            } else {
                break;
            }
        }

        // Iterate over input in tiles
        for (uint32_t base = 0; base < lenIn_;) {
            uint32_t len = lenIn_ - base;
            if (len > tileLenIn_) len = tileLenIn_;

            // Load tile
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[gmXBase + base], len);
            inQueueX_.EnQue(xLocal);
            xLocal = inQueueX_.DeQue<float>();

            for (uint32_t j = 0; j < len; ++j) {
                const uint32_t i = base + j;

                // Push A events at index i (before consuming x[i])
                while (tA < lenOut_) {
                    uint32_t aidx = AIdx(tA);
                    if (aidx == i) {
                        pending[tail % cap] = prefix;
                        ++tail;
                        ++tA;
                    } else {
                        break;
                    }
                }

                // Accumulate prefix with x[i]
                prefix += xLocal[j];

                // Pop B events at index i+1 (after consuming x[i])
                while (tB < lenOut_) {
                    uint32_t bidx = BIdx(tB);
                    if (bidx == (i + 1)) {
                        float sum = prefix - pending[head % cap];
                        ++head;
                        float yv = sum * invKernel_;
                        yLocal[yLocalSize++] = yv;
                        if (yLocalSize == outChunk_) {
                            FlushY(yLocal, yLocalSize, gmYBase, yWritten);
                        }
                        ++tB;
                    } else {
                        break;
                    }
                }
            }

            inQueueX_.FreeTensor(xLocal);
            base += len;
        }

        // Handle a_idx == lenIn_ after finishing scanning
        while (tA < lenOut_) {
            uint32_t aidx = AIdx(tA);
            if (aidx == lenIn_) {
                pending[tail % cap] = prefix; // P(lenIn)
                ++tail;
                ++tA;
            } else {
                break;
            }
        }

        // Finalize remaining outputs where b_idx == lenIn_
        while (tB < lenOut_) {
            uint32_t bidx = BIdx(tB);
            if (bidx == lenIn_) {
                float sum = prefix - pending[head % cap];
                ++head;
                float yv = sum * invKernel_;
                yLocal[yLocalSize++] = yv;
                if (yLocalSize == outChunk_) {
                    FlushY(yLocal, yLocalSize, gmYBase, yWritten);
                }
                ++tB;
            } else {
                break;
            }
        }

        // Flush remaining outputs
        FlushY(yLocal, yLocalSize, gmYBase, yWritten);

        tmpQueue_.FreeTensor(pending);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0};
    uint32_t lenIn_{0}, lenOut_{0};
    uint32_t tileLenIn_{0}, outChunk_{0};
    uint32_t kernelSize_{1}, stride_{1}, padding_{0}, pendingCap_{1};
    float invKernel_{1.0f};
};

extern "C" __global__ __aicore__ void average_pooling_1d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling1dCustom op;
    op.Init(x, y,
            tiling_data.n, tiling_data.c,
            tiling_data.lenIn, tiling_data.lenOut,
            tiling_data.tileLenIn, tiling_data.outChunk,
            tiling_data.kernelSize, tiling_data.stride, tiling_data.padding,
            tiling_data.pendingCap);
    op.Process();
}
