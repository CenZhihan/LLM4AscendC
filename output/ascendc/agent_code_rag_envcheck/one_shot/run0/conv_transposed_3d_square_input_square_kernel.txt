#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelConvTransposed3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t totalOutLen, uint32_t totalInLen, uint32_t totalWLen,
                                uint32_t tileLen,
                                uint32_t N, uint32_t Ci, uint32_t Co,
                                uint32_t D, uint32_t H, uint32_t W,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw)
    {
        totalOutLen_ = totalOutLen;
        totalInLen_  = totalInLen;
        totalWLen_   = totalWLen;
        tileLen_     = tileLen;

        N_ = N; Ci_ = Ci; Co_ = Co;
        D_ = D; H_ = H; W_ = W;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;

        Dout_ = D_ + Kd_ - 1;
        Hout_ = H_ + Kh_ - 1;
        Wout_ = W_ + Kw_ - 1;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalInLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, totalWLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOutLen_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        // Patch buffers sized to Kw (max along x-axis)
        pipe_.InitBuffer(xPatchQueue_, 1, Kw_ * sizeof(float));
        pipe_.InitBuffer(wPatchQueue_, 1, Kw_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalOutLen_) return;

        uint32_t curLen = totalOutLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;

        CopyComputeOut(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalOutLen_) {
                nextLen = totalOutLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyComputeOut(nextBase, nextLen);
            }
            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline uint32_t FlattenX(uint32_t n, uint32_t ci, uint32_t z, uint32_t y, uint32_t x) const
    {
        return (((n * Ci_ + ci) * D_ + z) * H_ + y) * W_ + x;
    }

    __aicore__ inline uint32_t FlattenW(uint32_t ci, uint32_t co, uint32_t kz, uint32_t ky, uint32_t kx) const
    {
        return (((ci * Co_ + co) * Kd_ + kz) * Kh_ + ky) * Kw_ + kx;
    }

    __aicore__ inline uint32_t FlattenY(uint32_t n, uint32_t co, uint32_t z, uint32_t y, uint32_t x) const
    {
        return (((n * Co_ + co) * Dout_ + z) * Hout_ + y) * Wout_ + x;
    }

    __aicore__ inline void DecodeY(uint32_t flat, uint32_t &n, uint32_t &co, uint32_t &oz, uint32_t &oy, uint32_t &ox) const
    {
        ox = flat % Wout_; flat /= Wout_;
        oy = flat % Hout_; flat /= Hout_;
        oz = flat % Dout_; flat /= Dout_;
        co = flat % Co_;   flat /= Co_;
        n  = flat;
    }

    __aicore__ inline void CopyComputeOut(uint32_t gmOffsetOut, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            uint32_t n, co, oz, oy, ox;
            DecodeY(gmOffsetOut + i, n, co, oz, oy, ox);

            float acc = 0.0f;

            // valid kz range
            uint32_t kz_low = (oz >= (Kd_ - 1)) ? (oz - (Kd_ - 1)) : 0;
            uint32_t kz_high = (oz < D_) ? oz : (Kd_ - 1);
            if (kz_high >= Kd_) kz_high = Kd_ - 1;

            // valid ky range
            uint32_t ky_low = (oy >= (Kh_ - 1)) ? (oy - (Kh_ - 1)) : 0;
            uint32_t ky_high = (oy < H_) ? oy : (Kh_ - 1);
            if (ky_high >= Kh_) ky_high = Kh_ - 1;

            // valid kx range (depends only on ox and W/Kw)
            uint32_t kx_low = (ox >= (Kw_ - 1)) ? (ox - (Kw_ - 1)) : 0;
            uint32_t kx_high = (ox < W_) ? ox : (Kw_ - 1);
            if (kx_high >= Kw_) kx_high = Kw_ - 1;

            for (uint32_t ci = 0; ci < Ci_; ++ci) {
                for (uint32_t kz = kz_low; kz <= kz_high; ++kz) {
                    const int32_t iz = static_cast<int32_t>(oz) - static_cast<int32_t>(kz);
                    if (iz < 0 || iz >= static_cast<int32_t>(D_)) continue;

                    for (uint32_t ky = ky_low; ky <= ky_high; ++ky) {
                        const int32_t iy = static_cast<int32_t>(oy) - static_cast<int32_t>(ky);
                        if (iy < 0 || iy >= static_cast<int32_t>(H_)) continue;

                        // valid kx range refined by bounds on ix
                        uint32_t kxL = kx_low;
                        uint32_t kxH = kx_high;
                        // ensure ix in [0, W-1]: ix = ox - kx
                        if (static_cast<int32_t>(ox) - static_cast<int32_t>(kxH) < 0) {
                            const int32_t adj = static_cast<int32_t>(ox);
                            kxH = (adj >= 0) ? static_cast<uint32_t>(adj) : 0;
                        }
                        const int32_t maxKxByRight = static_cast<int32_t>(ox) - static_cast<int32_t>(W_ - 1);
                        if (maxKxByRight > static_cast<int32_t>(kxL)) {
                            kxL = static_cast<uint32_t>(maxKxByRight);
                        }
                        if (kxL > kxH) continue;

                        const uint32_t m = kxH - kxL + 1;
                        const uint32_t ixStart = static_cast<uint32_t>(static_cast<int32_t>(ox) - static_cast<int32_t>(kxH));

                        AscendC::LocalTensor<float> xPatch = xPatchQueue_.AllocTensor<float>();
                        AscendC::LocalTensor<float> wPatch = wPatchQueue_.AllocTensor<float>();

                        // copy contiguous x patch along W axis
                        const uint32_t xOffset = FlattenX(n, ci, static_cast<uint32_t>(iz), static_cast<uint32_t>(iy), ixStart);
                        AscendC::DataCopy(xPatch, xGm_[xOffset], m);

                        // copy contiguous weight segment along kx
                        const uint32_t wOffset = FlattenW(ci, co, kz, ky, kxL);
                        AscendC::DataCopy(wPatch, wGm_[wOffset], m);

                        // accumulate acc += sum_t xPatch[t] * wPatch[m-1-t]  (reverse kx to match ix progression)
                        for (uint32_t t = 0; t < m; ++t) {
                            const float xv = xPatch[t];
                            const float wv = wPatch[m - 1 - t];
                            acc += xv * wv;
                        }

                        xPatchQueue_.FreeTensor(xPatch);
                        wPatchQueue_.FreeTensor(wPatch);
                    }
                }
            }

            yLocal.SetValue(i, acc);
        }

        outQueueY_.EnQue<float>(yLocal);
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffsetOut], yOut, len);
        outQueueY_.FreeTensor(yOut);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xPatchQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wPatchQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalOutLen_ {0};
    uint32_t totalInLen_ {0};
    uint32_t totalWLen_ {0};
    uint32_t tileLen_ {0};

    uint32_t N_ {0}, Ci_ {0}, Co_ {0};
    uint32_t D_ {0}, H_ {0}, W_ {0};
    uint32_t Kd_ {0}, Kh_ {0}, Kw_ {0};
    uint32_t Dout_ {0}, Hout_ {0}, Wout_ {0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.totalOutLength, tiling_data.totalInLength, tiling_data.totalWeightLength,
            tiling_data.tileLength,
            tiling_data.N, tiling_data.Ci, tiling_data.Co,
            tiling_data.D, tiling_data.H, tiling_data.W,
            tiling_data.Kd, tiling_data.Kh, tiling_data.Kw);
    op.Process();
}
