#include "conv_depthwise_separable_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;   // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t BLOCKDIM_MIN = 8;
static constexpr uint32_t BLOCKDIM_MAX = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwiseSeparable2dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *wdw_shape = context->GetInputShape(1);
    const gert::Shape *wpw_shape = context->GetInputShape(2);

    // Expecting NCHW for x, [C_in,1,3,3] for depthwise, [C_out,C_in,1,1] for pointwise
    const uint32_t N = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t CIn = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));
    const uint32_t COut = static_cast<uint32_t>(wpw_shape->GetOriginShape().GetDim(0));

    // Choose ocChunk to balance UB usage
    uint32_t ocChunk = (COut >= 16u) ? 8u : ((COut >= 8u) ? 8u : ((COut >= 4u) ? 4u : 1u));
    if (ocChunk > COut) ocChunk = COut;

    // UB model per tile (H-tile=1 row):
    //   rowBuf: 3 * (tileW + 2)
    //   dwVec: tileW
    //   tmpVec: tileW
    //   outBuf: ocChunk * tileW
    //   wpwBuf: ocChunk * CIn
    // Total floats ~= (3*(tileW+2) + 3*tileW + ocChunk*tileW + ocChunk*CIn)
    //              = ( (6 + ocChunk) * tileW + 6 + ocChunk*CIn )
    // bytes ~= 4 * [ ... ] <= UB_BUDGET_BYTES
    uint32_t maxTileW = 0;
    if (UB_BUDGET_BYTES > 4u * (6u + ocChunk * CIn)) {
        maxTileW = (UB_BUDGET_BYTES / 4u - (6u + ocChunk * CIn)) / (6u + ocChunk);
    }
    if (maxTileW == 0) {
        maxTileW = 256;
    }
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;

    uint32_t tileW = maxTileW;
    if (W < tileW) {
        const uint32_t aligned = AlignDown(W, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : W;
    }
    if (tileW == 0) tileW = std::min<uint32_t>(W, ALIGN_ELEMS);

    const uint32_t tilesPerRow = CeilDiv(W, tileW);
    const uint32_t totalTiles = N * H * tilesPerRow;

    uint32_t blockDim = totalTiles;
    if (blockDim < BLOCKDIM_MIN) blockDim = BLOCKDIM_MIN;
    if (blockDim > BLOCKDIM_MAX) blockDim = BLOCKDIM_MAX;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_CIn(CIn);
    tiling.set_COut(COut);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_tileW(tileW);
    tiling.set_tilesPerRow(tilesPerRow);
    tiling.set_blockDim(blockDim);
    tiling.set_ocChunk(ocChunk);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace_sizes = context->GetWorkspaceSizes(1);
    workspace_sizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvDepthwiseSeparable2dCustom {
public:
    __aicore__ inline KernelConvDepthwiseSeparable2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wdw, GM_ADDR wpw, GM_ADDR y,
                                uint32_t N, uint32_t CIn, uint32_t COut,
                                uint32_t H, uint32_t W,
                                uint32_t tileW, uint32_t tilesPerRow,
                                uint32_t ocChunk)
    {
        N_ = N; CIn_ = CIn; COut_ = COut; H_ = H; W_ = W;
        tileW_ = tileW; tilesPerRow_ = tilesPerRow; ocChunk_ = ocChunk;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint32_t)(N_ * CIn_ * H_ * W_));
        wdwGm_.SetGlobalBuffer((__gm__ float*)wdw, (uint32_t)(CIn_ * 9));
        wpwGm_.SetGlobalBuffer((__gm__ float*)wpw, (uint32_t)(COut_ * CIn_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint32_t)(N_ * COut_ * H_ * W_));

        // Buffers
        pipe_.InitBuffer(rowBufQ_, BUFFER_NUM, (3 * (tileW_ + 2)) * sizeof(float));
        pipe_.InitBuffer(dwVecQ_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQ_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(outVecQ_, BUFFER_NUM, (ocChunk_ * tileW_) * sizeof(float));
        pipe_.InitBuffer(wdwQ_, BUFFER_NUM, 9 * sizeof(float));
        pipe_.InitBuffer(wpwQ_, BUFFER_NUM, (ocChunk_ * CIn_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || CIn_ == 0 || COut_ == 0 || H_ == 0 || W_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalTiles = N_ * H_ * tilesPerRow_;

        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            // Decode tile to (n, h, tileIdx)
            uint32_t rem = t;
            const uint32_t n = rem / (H_ * tilesPerRow_);
            rem = rem % (H_ * tilesPerRow_);
            const uint32_t h = rem / tilesPerRow_;
            const uint32_t tileIdx = rem % tilesPerRow_;

            const uint32_t colStart = tileIdx * tileW_;
            uint32_t curTileW = W_ - colStart;
            if (curTileW > tileW_) curTileW = tileW_;

            // Pre-allocate frequently used locals
            AscendC::LocalTensor<float> rowBuf = rowBufQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> dwVec = dwVecQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpVec = tmpVecQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> wdwLocal = wdwQ_.AllocTensor<float>();

            for (uint32_t oc0 = 0; oc0 < COut_; oc0 += ocChunk_) {
                const uint32_t chunk = (oc0 + ocChunk_ <= COut_) ? ocChunk_ : (COut_ - oc0);

                // Prepare output accumulators
                AscendC::LocalTensor<float> outVec = outVecQ_.AllocTensor<float>();
                AscendC::Duplicate(outVec, 0.0f, chunk * curTileW);

                // Preload pointwise weights (chunk x CIn) into UB
                AscendC::LocalTensor<float> wpwLocal = wpwQ_.AllocTensor<float>();
                for (uint32_t oc = 0; oc < chunk; ++oc) {
                    // Each row is contiguous CIn
                    const uint32_t gmOff = (oc0 + oc) * CIn_;
                    AscendC::DataCopy(wpwLocal[oc * CIn_], wpwGm_[gmOff], CIn_);
                }

                for (uint32_t c = 0; c < CIn_; ++c) {
                    // Load depthwise kernel for channel c (9 floats)
                    AscendC::DataCopy(wdwLocal, wdwGm_[c * 9u], 9u);

                    // Prepare 3-row sliding window buffer with zero-padding on both sides
                    const uint32_t rowSpan = (tileW_ + 2);
                    // Zero rows
                    AscendC::Duplicate(rowBuf[0 * rowSpan], 0.0f, curTileW + 2);
                    AscendC::Duplicate(rowBuf[1 * rowSpan], 0.0f, curTileW + 2);
                    AscendC::Duplicate(rowBuf[2 * rowSpan], 0.0f, curTileW + 2);

                    // Copy valid rows from GM into the center of each row (offset +1)
                    for (int32_t ky = -1; ky <= 1; ++ky) {
                        int32_t r = static_cast<int32_t>(h) + ky;
                        if (r >= 0 && r < static_cast<int32_t>(H_)) {
                            uint32_t gmRowOff = (((n * CIn_ + c) * H_ + static_cast<uint32_t>(r)) * W_) + colStart;
                            AscendC::DataCopy(rowBuf[(ky + 1) * rowSpan + 1], xGm_[gmRowOff], curTileW);
                        }
                    }

                    // Compute depthwise 3x3 conv for this (n,c,h, colStart:colStart+curTileW-1)
                    AscendC::Duplicate(dwVec, 0.0f, curTileW);

                    for (uint32_t dy = 0; dy < 3; ++dy) {
                        for (uint32_t dx = 0; dx < 3; ++dx) {
                            // load shifted window into tmpVec
                            const uint32_t base = dy * rowSpan + dx;
                            AscendC::DataCopy(tmpVec, rowBuf[base], curTileW);
                            // multiply by scalar weight
                            // Get scalar weight from UB
                            float wv = 0.0f;
                            // Assume AscendC provides GetValue to read UB scalar
                            wv = wdwLocal.GetValue(dy * 3u + dx);
                            AscendC::Muls(tmpVec, tmpVec, wv, curTileW);
                            // accumulate
                            AscendC::Add(dwVec, dwVec, tmpVec, curTileW);
                        }
                    }

                    // Accumulate pointwise outputs for oc chunk: out[oc] += w_pw[oc,c] * dwVec
                    for (uint32_t oc = 0; oc < chunk; ++oc) {
                        float pw = wpwLocal.GetValue(oc * CIn_ + c);
                        AscendC::Muls(tmpVec, dwVec, pw, curTileW);
                        AscendC::Add(outVec[oc * curTileW], outVec[oc * curTileW], tmpVec, curTileW);
                    }
                } // end for c

                // Store outputs for this tile and oc chunk
                for (uint32_t oc = 0; oc < chunk; ++oc) {
                    const uint32_t yOff = (((n * COut_ + (oc0 + oc)) * H_ + h) * W_) + colStart;
                    AscendC::DataCopy(yGm_[yOff], outVec[oc * curTileW], curTileW);
                }

                // Free chunk buffers
                wpwQ_.FreeTensor(wpwLocal);
                outVecQ_.FreeTensor(outVec);
            } // end for oc0

            // Free per-tile buffers
            wdwQ_.FreeTensor(wdwLocal);
            tmpVecQ_.FreeTensor(tmpVec);
            dwVecQ_.FreeTensor(dwVec);
            rowBufQ_.FreeTensor(rowBuf);
        } // end for t
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> rowBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> dwVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> outVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> wdwQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> wpwQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wdwGm_;
    AscendC::GlobalTensor<float> wpwGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, CIn_{0}, COut_{0}, H_{0}, W_{0};
    uint32_t tileW_{0}, tilesPerRow_{0}, ocChunk_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_separable_2d_custom(
    GM_ADDR x, GM_ADDR w_depthwise, GM_ADDR w_pointwise, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwiseSeparable2dCustom op;
    op.Init(x, w_depthwise, w_pointwise, y,
            tiling_data.N, tiling_data.CIn, tiling_data.COut,
            tiling_data.H, tiling_data.W,
            tiling_data.tileW, tiling_data.tilesPerRow,
            tiling_data.ocChunk);
    op.Process();
}
