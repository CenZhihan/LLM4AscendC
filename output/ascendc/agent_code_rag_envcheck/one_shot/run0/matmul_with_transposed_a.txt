host_operator_src="""

#include "matmul_with_transposed_a_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core conservative UB budget
static constexpr uint32_t ALIGN_N = 128;                // align tileN for better DMA efficiency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedACustomTilingData tiling;

    // Shapes:
    // Input A is shape (K, M)
    // Input B is shape (K, N)
    const auto *shapeA = context->GetInputShape(0);
    const auto *shapeB = context->GetInputShape(1);

    if (shapeA->GetOriginShape().GetDimNum() != 2 || shapeB->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t K = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(0));
    const uint32_t M = static_cast<uint32_t>(shapeA->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(shapeB->GetOriginShape().GetDim(1));
    // sanity: shapeB->GetOriginShape().GetDim(0) should equal K

    // Simple tiling strategy:
    //   C = A^T * B  => C shape (M, N)
    // UB footprint approximated by:
    //   A_block: tileK * tileM
    //   B_block: tileK * tileN
    //   C_block: tileM * tileN
    //   tmp vector: tileN
    //   Total floats ~ (tileK*tileM + tileK*tileN + tileM*tileN + tileN)
    // Choose fixed tileK and tileM, solve tileN from UB budget.
    const uint32_t tileK = 64; // conservative for bandwidth/utilization
    uint32_t tileM = 4;        // small to keep UB usage under control
    if (tileM > M) tileM = M > 0 ? M : 1;

    const uint32_t ub_floats = UB_BUDGET_BYTES / 4;
    // (ub_floats - tileK*tileM) / (tileK + tileM + 1)
    uint32_t maxTileN = 0;
    if (tileK + tileM + 1 != 0) {
        const int64_t numer = static_cast<int64_t>(ub_floats) - static_cast<int64_t>(tileK) * static_cast<int64_t>(tileM);
        if (numer > 0) {
            maxTileN = static_cast<uint32_t>(numer / static_cast<int64_t>(tileK + tileM + 1));
        }
    }
    if (maxTileN < ALIGN_N) {
        maxTileN = ALIGN_N;
    }
    uint32_t tileN = maxTileN;
    if (tileN > N) {
        tileN = N;
    }
    tileN = AlignDown(tileN, ALIGN_N);
    if (tileN == 0) tileN = (N == 0 ? 0u : 1u);

    // blockDim = number of M tiles, capped to hardware range
    uint32_t mTiles = (tileM == 0) ? 1u : CeilDiv(M, tileM);
    uint32_t blockDim = mTiles;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *shapeA = context->GetInputShape(0); // (K, M)
    const gert::Shape *shapeB = context->GetInputShape(1); // (K, N)
    if (shapeA->GetOriginShape().GetDimNum() != 2 || shapeB->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = shapeA->GetOriginShape().GetDim(1);
    const int64_t N = shapeB->GetOriginShape().GetDim(1);

    gert::Shape *shapeC = context->GetOutputShape(0);

    out.SetDimNum(2);
    out.SetDim(0, M);
    out.SetDim(1, N);
    shapeC->SetOriginShape(out);
    shapeC->SetStorageShape(out);
    return GRAPH_SUCCESS;
}

    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulWithTransposedACustom {
public:
    __aicore__ inline KernelMatmulWithTransposedACustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tM_ = tileM; tN_ = tileN; tK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)A, (uint64_t)K_ * (uint64_t)M_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, (uint64_t)K_ * (uint64_t)N_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, (uint64_t)M_ * (uint64_t)N_);

        // UB buffers
        pipe_.InitBuffer(qABlock_, BUFFER_NUM, tK_ * tM_ * (uint32_t)sizeof(float));
        pipe_.InitBuffer(qBBlock_, BUFFER_NUM, tK_ * tN_ * (uint32_t)sizeof(float));
        pipe_.InitBuffer(qCBlock_, BUFFER_NUM, tM_ * tN_ * (uint32_t)sizeof(float));
        pipe_.InitBuffer(qTmpRow_, BUFFER_NUM, tN_ * (uint32_t)sizeof(float));
        pipe_.InitBuffer(qScalar_, BUFFER_NUM, 1u * (uint32_t)sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint32_t mTiles = (tM_ == 0) ? 1u : ( (M_ + tM_ - 1u) / tM_ );
        // grid-stride over tiles on M dimension
        for (uint32_t tileIndex = blockIdx; tileIndex < mTiles; tileIndex += (blockNum == 0 ? 1u : blockNum)) {
            const uint32_t mBase = tileIndex * tM_;
            if (mBase >= M_) break;
            const uint32_t curM = AscendC::min(tM_, M_ - mBase);

            // loop N tiles
            for (uint32_t nBase = 0; nBase < N_; nBase += tN_) {
                const uint32_t curN = AscendC::min(tN_, N_ - nBase);

                // C tile initialize to zero
                AscendC::LocalTensor<float> cTile = qCBlock_.AllocTensor<float>();
                AscendC::Dup(cTile, 0.0f, curM * curN);

                // accumulate over K
                for (uint32_t kBase = 0; kBase < K_; kBase += tK_) {
                    const uint32_t curK = AscendC::min(tK_, K_ - kBase);

                    // Load A block: shape (curK, curM) from A rows [kBase:kBase+curK), cols [mBase:mBase+curM)
                    AscendC::LocalTensor<float> aBlock = qABlock_.AllocTensor<float>();
                    for (uint32_t pk = 0; pk < curK; ++pk) {
                        const uint64_t aOff = ((uint64_t)(kBase + pk) * (uint64_t)M_) + (uint64_t)mBase;
                        AscendC::DataCopy(aBlock[pk * tM_], aGm_[aOff], curM);
                    }

                    // Load B block: shape (curK, curN) from B rows [kBase:kBase+curK), cols [nBase:nBase+curN)
                    AscendC::LocalTensor<float> bBlock = qBBlock_.AllocTensor<float>();
                    for (uint32_t pk = 0; pk < curK; ++pk) {
                        const uint64_t bOff = ((uint64_t)(kBase + pk) * (uint64_t)N_) + (uint64_t)nBase;
                        AscendC::DataCopy(bBlock[pk * tN_], bGm_[bOff], curN);
                    }

                    // Compute: C_tile(MxN) += sum_{pk=0..curK-1} (A_row_k^T (size M) outer B_row_k (size N))
                    // Naive UB scalar accumulation with 1-length vector ops for correctness.
                    AscendC::LocalTensor<float> tmpRow = qTmpRow_.AllocTensor<float>();
                    AscendC::LocalTensor<float> scalarBuf = qScalar_.AllocTensor<float>();

                    for (uint32_t pk = 0; pk < curK; ++pk) {
                        // for each m position in current tile
                        for (uint32_t im = 0; im < curM; ++im) {
                            // load a scalar a = A[kBase+pk, mBase+im]
                            AscendC::DataCopy(scalarBuf, aBlock[pk * tM_ + im], 1);

                            // For each n in current tile, c[m,n] += a * b[pk, n]
                            // vectorize per row n with temporary row buffer:
                            // tmpRow = bRow * aScalar
                            // But we cannot use dynamic scalar in Muls directly; use elementwise 1-length ops over each n.
                            for (uint32_t jn = 0; jn < curN; ++jn) {
                                // load b scalar
                                AscendC::DataCopy(tmpRow, bBlock[pk * tN_ + jn], 1);
                                // tmpRow = scalarBuf * tmpRow  (size 1)
                                AscendC::Mul(tmpRow, scalarBuf, tmpRow, 1);
                                // cElem += tmpRow
                                AscendC::Add(cTile[im * tN_ + jn], cTile[im * tN_ + jn], tmpRow, 1);
                            }
                        }
                    }

                    qScalar_.FreeTensor(scalarBuf);
                    qTmpRow_.FreeTensor(tmpRow);
                    qBBlock_.FreeTensor(bBlock);
                    qABlock_.FreeTensor(aBlock);
                } // end for kBase

                // Store C tile back
                for (uint32_t im = 0; im < curM; ++im) {
                    const uint64_t cOff = ((uint64_t)(mBase + im) * (uint64_t)N_) + (uint64_t)nBase;
                    AscendC::DataCopy(cGm_[cOff], cTile[im * tN_], curN);
                }
                qCBlock_.FreeTensor(cTile);
            } // end for nBase
        } // end for tileIndex
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qABlock_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qBBlock_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> qCBlock_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> qTmpRow_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> qScalar_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tM_{0}, tN_{0}, tK_{0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_a_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                                                       GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling_data, tilingGm);
    KernelMatmulWithTransposedACustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}

    return C;
