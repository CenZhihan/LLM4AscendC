host_operator_src="""

#include "matmul_for_upper_triangular_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0u) ? 0u : (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForUpperTriangularMatricesCustomTilingData tiling;

    const gert::Shape *shapeA = context->GetInputShape(0);
    const gert::OriginShape &originA = shapeA->GetOriginShape();
    // Expect square matrices (N, N)
    const uint32_t n = static_cast<uint32_t>(originA.GetDim(0));

    // Choose per-row tile length equal to N (fits well in UB for 4096)
    constexpr uint32_t ALIGN_ELEMS = 256;
    constexpr uint32_t HARD_CAP_ELEMS = 8192;
    uint32_t tileLen = n;
    if (tileLen > HARD_CAP_ELEMS) {
        tileLen = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    } else {
        tileLen = AlignDown(tileLen, ALIGN_ELEMS);
        if (tileLen == 0) tileLen = n; // fallback if small
    }

    // BlockDim: assign one row per core with grid-stride
    uint32_t blockDim = CeilDiv(n, 1u);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *A_shape = context->GetInputShape(0);
    const gert::Shape *B_shape = context->GetInputShape(1);
    // Basic sanity: both inputs should be same shape
    if (A_shape->GetOriginShape().GetDimNum() != 2 ||
        B_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    if (A_shape->GetOriginShape().GetDim(0) != B_shape->GetOriginShape().GetDim(0) ||
        A_shape->GetOriginShape().GetDim(1) != B_shape->GetOriginShape().GetDim(1)) {
        return GRAPH_FAILED;
    }
    gert::Shape *C_shape = context->GetOutputShape(0);
    *C_shape = *A_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulForUpperTriangularMatricesCustom {
public:
    __aicore__ inline KernelMatmulForUpperTriangularMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t n, uint32_t tileLength)
    {
        n_ = n;
        tileLen_ = tileLength;

        aGm_.SetGlobalBuffer((__gm__ float*)A, n_ * n_);
        bGm_.SetGlobalBuffer((__gm__ float*)B, n_ * n_);
        cGm_.SetGlobalBuffer((__gm__ float*)C, n_ * n_);

        // Initialize small UB buffers for scalar transfers
        pipe_.InitBuffer(aScalarQueue_, 1, sizeof(float));
        pipe_.InitBuffer(bScalarQueue_, 1, sizeof(float));
        pipe_.InitBuffer(sumScalarQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t i = blockIdx; i < n_; i += blockNum) {
            const uint32_t rowBaseA = i * n_;
            const uint32_t rowBaseC = i * n_;

            // Zero lower triangular part explicitly (j < i)
            for (uint32_t j = 0; j < i; ++j) {
                AscendC::LocalTensor<float> cLocal = sumScalarQueue_.AllocTensor<float>();
                cLocal.SetValue(0, 0.0f);
                AscendC::DataCopy(cGm_[rowBaseC + j], cLocal, 1);
                sumScalarQueue_.FreeTensor(cLocal);
            }

            // Upper triangular: compute C[i, j] = sum_{k=i..j} A[i,k] * B[k,j]
            for (uint32_t j = i; j < n_; ++j) {
                float sum = 0.0f;

                for (uint32_t k = i; k <= j; ++k) {
                    // Load A[i, k]
                    AscendC::LocalTensor<float> aLocal = aScalarQueue_.AllocTensor<float>();
                    AscendC::DataCopy(aLocal, aGm_[rowBaseA + k], 1);
                    const float aVal = aLocal.GetValue(0);
                    aScalarQueue_.FreeTensor(aLocal);

                    // Load B[k, j]
                    const uint32_t rowBaseB = k * n_;
                    AscendC::LocalTensor<float> bLocal = bScalarQueue_.AllocTensor<float>();
                    AscendC::DataCopy(bLocal, bGm_[rowBaseB + j], 1);
                    const float bVal = bLocal.GetValue(0);
                    bScalarQueue_.FreeTensor(bLocal);

                    sum += aVal * bVal;
                }

                // Store C[i, j]
                AscendC::LocalTensor<float> cLocal = sumScalarQueue_.AllocTensor<float>();
                cLocal.SetValue(0, sum);
                AscendC::DataCopy(cGm_[rowBaseC + j], cLocal, 1);
                sumScalarQueue_.FreeTensor(cLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> aScalarQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bScalarQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumScalarQueue_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t n_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void matmul_for_upper_triangular_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForUpperTriangularMatricesCustom op;
    op.Init(A, B, C, tiling_data.n, tiling_data.tileLength);
    op.Process();
}
