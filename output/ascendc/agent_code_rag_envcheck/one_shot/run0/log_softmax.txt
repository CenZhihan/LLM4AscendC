host_operator_src="""
#include <cstdint>
#include <limits>
#include <algorithm>
#include "log_softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue: 1 * tile
//   (small scalar buffers are negligible)
// Total floats ~= 5 * tile => bytes ~= 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS     = 2048;       // DMA-friendly
static constexpr uint32_t HARD_CAP_ELEMS  = 65536;      // latency cap

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    LogSoftmaxCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dimAttr = 1;
    if (attrs && attrs->GetNum() > 0) {
        const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
        if (dimPtr) dimAttr = *dimPtr;
    }

    const gert::Shape *x_shape = context->GetInputShape(0);
    auto origin = x_shape->GetOriginShape();
    const int64_t nd = origin.GetDimNum();

    // Canonicalize axis
    int64_t axis = static_cast<int64_t>(dimAttr);
    if (axis < 0) axis += nd;
    if (axis < 0) axis = 0;
    if (axis >= nd) axis = nd - 1;

    // Collapse shape to (rows, cols) along the specified axis
    uint64_t rows64 = 1;
    uint64_t cols64 = 1;
    for (int64_t i = 0; i < nd; ++i) {
        const uint64_t d = static_cast<uint64_t>(origin.GetDim(i));
        if (i < axis) {
            rows64 *= d;
        } else {
            cols64 *= d;
        }
    }
    if (rows64 == 0 || cols64 == 0) {
        rows64 = 0;
        cols64 = 0;
    }

    const uint32_t rows = static_cast<uint32_t>(rows64);
    const uint32_t cols = static_cast<uint32_t>(cols64);

    // Tile length from UB budget: bytes ~= 20 * tile
    uint32_t maxTile = (UB_BUDGET_BYTES / 20); // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileCols = maxTile;
    if (cols > 0 && cols < tileCols) {
        const uint32_t aligned = AlignDown(cols, ALIGN_ELEMS);
        tileCols = (aligned > 0) ? aligned : cols;
    }

    const uint32_t tilesPerRow = (tileCols == 0) ? 1u : CeilDiv(cols, tileCols);
    // Choose blockDim to partition by rows primarily
    uint32_t blockDim = rows > 0 ? rows : 1u;
    if (blockDim < 8u) blockDim = std::min<uint32_t>(8u, std::max<uint32_t>(1u, blockDim));
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(static_cast<int32_t>(axis));

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace needed
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

}

kernel_src="""
#include "kernel_operator.h"
#include <cmath>

constexpr int32_t BUFFER_NUM = 2;

class KernelLogSoftmaxCustom {
public:
    __aicore__ inline KernelLogSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t tileCols)
    {
        rows_ = rows;
        cols_ = cols;
        tileCols_ = tileCols == 0 ? cols : tileCols;

        const uint64_t totalLen = static_cast<uint64_t>(rows_) * static_cast<uint64_t>(cols_);
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(calcQueue_, 1, tileCols_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0 || tileCols_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            const uint64_t rowOffset = static_cast<uint64_t>(row) * static_cast<uint64_t>(cols_);

            // Pass 1: compute rowMax
            float rowMax = -3.402823466e+38F; // -FLT_MAX
            for (uint32_t c0 = 0; c0 < cols_; c0 += tileCols_) {
                const uint32_t len = AscendC::Min(tileCols_, cols_ - c0);
                CopyIn(rowOffset + c0, len);

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                float tileMax = -3.402823466e+38F;
                AscendC::ReduceMax(tileMax, xLocal, len); // reduce to scalar
                inQueueX_.FreeTensor(xLocal);

                if (tileMax > rowMax) rowMax = tileMax;
            }

            // Pass 2: compute sumExp = sum(exp(x - rowMax))
            float sumExp = 0.0f;
            for (uint32_t c0 = 0; c0 < cols_; c0 += tileCols_) {
                const uint32_t len = AscendC::Min(tileCols_, cols_ - c0);
                CopyIn(rowOffset + c0, len);

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tLocal = calcQueue_.AllocTensor<float>();

                AscendC::Adds(tLocal, xLocal, -rowMax, len); // t = x - rowMax
                AscendC::Exp(tLocal, tLocal, len);            // t = exp(t)

                float partSum = 0.0f;
                AscendC::ReduceSum(partSum, tLocal, len);     // scalar sum of tile

                calcQueue_.FreeTensor(tLocal);
                inQueueX_.FreeTensor(xLocal);

                sumExp += partSum;
            }
            const float logSum = logf(sumExp);

            // Pass 3: write y = x - rowMax - logSum
            for (uint32_t c0 = 0; c0 < cols_; c0 += tileCols_) {
                const uint32_t len = AscendC::Min(tileCols_, cols_ - c0);
                CopyIn(rowOffset + c0, len);

                AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::LocalTensor<float> tLocal = calcQueue_.AllocTensor<float>();

                AscendC::Adds(tLocal, xLocal, -rowMax, len);  // t = x - rowMax
                AscendC::Adds(yLocal, tLocal, -logSum, len);  // y = t - logSum

                calcQueue_.FreeTensor(tLocal);
                outQueueY_.EnQue(yLocal);
                inQueueX_.FreeTensor(xLocal);

                CopyOut(rowOffset + c0, len);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM>   inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM>  outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1>          calcQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_{0};
    uint32_t cols_{0};
    uint32_t tileCols_{0};
};

extern "C" __global__ __aicore__ void log_softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);
    KernelLogSoftmaxCustom op;
    op.Init(x, y, t.rows, t.cols, t.tileCols);
    op.Process();
}
"""
