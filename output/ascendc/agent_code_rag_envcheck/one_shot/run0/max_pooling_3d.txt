static inline int64_t FloorDiv(int64_t a, int64_t b) {
  return a / b;
}

static inline int64_t CeilDiv(int64_t a, int64_t b) {
  return (a + b - 1) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxPooling3dCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    if (in_shape->GetDimNum() != 5) {
        return ge::GRAPH_FAILED;
    }

    const int64_t N = in_shape->GetDim(0);
    const int64_t C = in_shape->GetDim(1);
    const int64_t Di = in_shape->GetDim(2);
    const int64_t Hi = in_shape->GetDim(3);
    const int64_t Wi = in_shape->GetDim(4);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *ksize = attrs->GetListInt(0);  // length 3
    const int64_t *strides = attrs->GetListInt(1); // length 3
    const int64_t *pads = attrs->HasAttr(2) ? attrs->GetListInt(2) : nullptr;
    const bool *ceil_mode_ptr = attrs->HasAttr(3) ? attrs->GetAttrPointer<bool>(3) : nullptr;

    const int64_t kd = ksize[0];
    const int64_t kh = ksize[1];
    const int64_t kw = ksize[2];

    const int64_t sd = strides[0];
    const int64_t sh = strides[1];
    const int64_t sw = strides[2];

    const int64_t pd = (pads != nullptr) ? pads[0] : 0;
    const int64_t ph = (pads != nullptr) ? pads[1] : 0;
    const int64_t pw = (pads != nullptr) ? pads[2] : 0;

    const bool ceil_mode = (ceil_mode_ptr != nullptr) ? *ceil_mode_ptr : false;

    auto out_dim = [&](int64_t in, int64_t k, int64_t s, int64_t p, bool ceilMode) -> int64_t {
        const int64_t num = in + 2 * p - k;
        return (ceilMode ? CeilDiv(num, s) : FloorDiv(num, s)) + 1;
    };

    const int64_t Do = out_dim(Di, kd, sd, pd, ceil_mode);
    const int64_t Ho = out_dim(Hi, kh, sh, ph, ceil_mode);
    const int64_t Wo = out_dim(Wi, kw, sw, pw, ceil_mode);

    // conservative blockDim: parallelize over N*C rows
    uint32_t blockDim = static_cast<uint32_t>(N * C);
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 32u) blockDim = 32u;
    context->SetBlockDim(blockDim);

    tiling.set_n(static_cast<uint32_t>(N));
    tiling.set_c(static_cast<uint32_t>(C));
    tiling.set_di(static_cast<uint32_t>(Di));
    tiling.set_hi(static_cast<uint32_t>(Hi));
    tiling.set_wi(static_cast<uint32_t>(Wi));

    tiling.set_kd(static_cast<uint32_t>(kd));
    tiling.set_kh(static_cast<uint32_t>(kh));
    tiling.set_kw(static_cast<uint32_t>(kw));

    tiling.set_sd(static_cast<uint32_t>(sd));
    tiling.set_sh(static_cast<uint32_t>(sh));
    tiling.set_sw(static_cast<uint32_t>(sw));

    tiling.set_pd(static_cast<uint32_t>(pd));
    tiling.set_ph(static_cast<uint32_t>(ph));
    tiling.set_pw(static_cast<uint32_t>(pw));

    tiling.set_dout(static_cast<uint32_t>(Do));
    tiling.set_hout(static_cast<uint32_t>(Ho));
    tiling.set_wout(static_cast<uint32_t>(Wo));

    tiling.set_ceilMode(ceil_mode ? 1u : 0u);
    tiling.set_blockDim(blockDim);

kernel_src="""
#include "kernel_operator.h"
#include <cfloat>

class KernelMaxPooling3dCustom {
public:
    __aicore__ inline KernelMaxPooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t di, uint32_t hi, uint32_t wi,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t sd, uint32_t sh, uint32_t sw,
                                uint32_t pd, uint32_t ph, uint32_t pw,
                                uint32_t dout, uint32_t hout, uint32_t wout,
                                uint32_t ceilMode)
    {
        xPtr_ = reinterpret_cast<__gm__ float*>(x);
        yPtr_ = reinterpret_cast<__gm__ float*>(y);

        n_ = n; c_ = c; di_ = di; hi_ = hi; wi_ = wi;
        kd_ = kd; kh_ = kh; kw_ = kw;
        sd_ = sd; sh_ = sh; sw_ = sw;
        pd_ = pd; ph_ = ph; pw_ = pw;
        dout_ = dout; hout_ = hout; wout_ = wout;
        ceilMode_ = ceilMode;
    }

    __aicore__ inline void Process()
    {
        const uint32_t ncTotal = n_ * c_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t nc = blockIdx; nc < ncTotal; nc += blockNum) {
            const uint32_t n = nc / c_;
            const uint32_t cc = nc % c_;

            for (uint32_t od = 0; od < dout_; ++od) {
                int64_t dstart = static_cast<int64_t>(od) * sd_ - static_cast<int64_t>(pd_);
                for (uint32_t oh = 0; oh < hout_; ++oh) {
                    int64_t hstart = static_cast<int64_t>(oh) * sh_ - static_cast<int64_t>(ph_);
                    for (uint32_t ow = 0; ow < wout_; ++ow) {
                        int64_t wstart = static_cast<int64_t>(ow) * sw_ - static_cast<int64_t>(pw_);

                        float maxval = -FLT_MAX;

                        // Iterate over pooling window
                        for (uint32_t kd = 0; kd < kd_; ++kd) {
                            int64_t id = dstart + static_cast<int64_t>(kd);
                            if (id < 0 || id >= static_cast<int64_t>(di_)) continue;

                            for (uint32_t kh = 0; kh < kh_; ++kh) {
                                int64_t ih = hstart + static_cast<int64_t>(kh);
                                if (ih < 0 || ih >= static_cast<int64_t>(hi_)) continue;

                                for (uint32_t kw = 0; kw < kw_; ++kw) {
                                    int64_t iw = wstart + static_cast<int64_t>(kw);
                                    if (iw < 0 || iw >= static_cast<int64_t>(wi_)) continue;

                                    uint64_t xOff =
                                        (((static_cast<uint64_t>(n) * c_ + cc) * di_ + static_cast<uint64_t>(id)) * hi_
                                           + static_cast<uint64_t>(ih)) * wi_
                                           + static_cast<uint64_t>(iw);

                                    float v = xPtr_[xOff];
                                    if (v > maxval) maxval = v;
                                }
                            }
                        }

                        uint64_t yOff =
                            (((static_cast<uint64_t>(n) * c_ + cc) * dout_ + od) * hout_ + oh) * wout_ + ow;
                        yPtr_[yOff] = maxval;
                    }
                }
            }
        }
    }

private:
    __gm__ float* xPtr_ {nullptr};
    __gm__ float* yPtr_ {nullptr};

    uint32_t n_ {0}, c_ {0};
    uint32_t di_ {0}, hi_ {0}, wi_ {0};
    uint32_t kd_ {0}, kh_ {0}, kw_ {0};
    uint32_t sd_ {0}, sh_ {0}, sw_ {0};
    uint32_t pd_ {0}, ph_ {0}, pw_ {0};
    uint32_t dout_ {0}, hout_ {0}, wout_ {0};
    uint32_t ceilMode_ {0};
};

extern "C" __global__ __aicore__ void max_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxPooling3dCustom op;
    op.Init(
        x, y,
        tiling_data.n, tiling_data.c,
        tiling_data.di, tiling_data.hi, tiling_data.wi,
        tiling_data.kd, tiling_data.kh, tiling_data.kw,
        tiling_data.sd, tiling_data.sh, tiling_data.sw,
        tiling_data.pd, tiling_data.ph, tiling_data.pw,
        tiling_data.dout, tiling_data.hout, tiling_data.wout,
        tiling_data.ceilMode
    );
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t ceil_div_int64(int64_t a, int64_t b) {
    return (a + b - 1) / b;
}

static inline int64_t out_size_1d(int64_t in, int64_t k, int64_t s, int64_t p, bool ceil_mode) {
    const int64_t num = in + 2 * p - k;
    return (ceil_mode ? ceil_div_int64(num, s) : (num / s)) + 1;
}

at::Tensor max_pooling_3d_custom_impl_npu(const at::Tensor& self,
                                          int64_t kernel_size,
                                          int64_t stride,
                                          int64_t padding,
                                          bool ceil_mode = false) {
    TORCH_CHECK(self.dim() == 5, "Input must be 5D (N, C, D, H, W)");
    std::vector<int64_t> ksize = {kernel_size, kernel_size, kernel_size};
    std::vector<int64_t> strides = {stride, stride, stride};
    std::vector<int64_t> pads = {padding, padding, padding};

    auto sizes = self.sizes();
    const int64_t N = sizes[0], C = sizes[1], D = sizes[2], H = sizes[3], W = sizes[4];
    const int64_t Do = out_size_1d(D, ksize[0], strides[0], pads[0], ceil_mode);
    const int64_t Ho = out_size_1d(H, ksize[1], strides[1], pads[1], ceil_mode);
    const int64_t Wo = out_size_1d(W, ksize[2], strides[2], pads[2], ceil_mode);

    at::Tensor result = at::empty({N, C, Do, Ho, Wo}, self.options());
    EXEC_NPU_CMD(aclnnMaxPooling3dCustom, self, ksize, strides, pads, ceil_mode, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_pooling_3d_custom", &max_pooling_3d_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_pooling_3d_custom", &max_pooling_3d_custom_impl_npu,
          "MaxPooling3D (MaxPooling3dCustom on NPU)",
          py::arg("x"),
          py::arg("kernel_size"),
          py::arg("stride"),
          py::arg("padding"),
          py::arg("ceil_mode") = false);
}
