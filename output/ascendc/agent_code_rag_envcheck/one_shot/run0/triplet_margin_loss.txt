#include "triplet_margin_loss_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    TripletMarginLossCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const auto &ori = shape->GetOriginShape();

    uint32_t rows = 1;
    uint32_t cols = 1;
    if (ori.GetDimNum() == 1) {
        rows = static_cast<uint32_t>(1);
        cols = static_cast<uint32_t>(ori.GetDim(0));
    } else {
        rows = static_cast<uint32_t>(ori.GetDim(0));
        cols = static_cast<uint32_t>(ori.GetDim(1));
    }

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *marginPtr = attrs->GetAttrPointer<float>(0);
    float margin = (marginPtr != nullptr) ? *marginPtr : 1.0f;

    // UB usage per column tile: we hold 4 * tileCols floats (anchor, positive, negative, tmp)
    // bytes = 16 * tileCols
    uint32_t maxTileCols = UB_BUDGET_BYTES / 16;
    if (maxTileCols > HARD_CAP_ELEMS) maxTileCols = HARD_CAP_ELEMS;
    maxTileCols = AlignDown(maxTileCols, ALIGN_ELEMS);
    if (maxTileCols == 0) maxTileCols = ALIGN_ELEMS;

    uint32_t tileCols = maxTileCols;
    if (cols < tileCols) {
        uint32_t aligned = AlignDown(cols, ALIGN_ELEMS);
        tileCols = (aligned > 0) ? aligned : cols;
    }

    // We use a single block to avoid cross-block reduction complexity
    uint32_t blockDim = 1;
    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_tileCols(tileCols);
    tiling.set_blockDim(blockDim);
    tiling.set_margin(margin);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelTripletMarginLossCustom {
public:
    __aicore__ inline KernelTripletMarginLossCustom() {}

    __aicore__ inline void Init(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y,
                                uint32_t rows, uint32_t cols, uint32_t tileCols, float margin)
    {
        rows_ = rows;
        cols_ = cols;
        tileCols_ = tileCols;
        margin_ = margin;

        aGm_.SetGlobalBuffer((__gm__ float*)anchor, rows_ * cols_);
        pGm_.SetGlobalBuffer((__gm__ float*)positive, rows_ * cols_);
        nGm_.SetGlobalBuffer((__gm__ float*)negative, rows_ * cols_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, 1);

        pipe_.InitBuffer(qA_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(qP_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(qN_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(qTmp_, BUFFER_NUM, tileCols_ * sizeof(float));
        pipe_.InitBuffer(qScalar_, 2, sizeof(float)); // for 1-element reductions/accumulators
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) {
            // write zero
            AscendC::LocalTensor<float> outLocal = qScalar_.AllocTensor<float>();
            AscendC::Dup(outLocal, 0.0f, 1);
            AscendC::DataCopy(yGm_[0], outLocal, 1);
            qScalar_.FreeTensor(outLocal);
            return;
        }

        // Accumulator for sum of losses (1 element local tensor)
        AscendC::LocalTensor<float> sumLoss = qScalar_.AllocTensor<float>();
        AscendC::Dup(sumLoss, 0.0f, 1);

        for (uint32_t r = 0; r < rows_; ++r) {
            // per-row accumulators for squared distances
            AscendC::LocalTensor<float> apAcc = qScalar_.AllocTensor<float>();
            AscendC::LocalTensor<float> anAcc = qScalar_.AllocTensor<float>();
            AscendC::Dup(apAcc, 0.0f, 1);
            AscendC::Dup(anAcc, 0.0f, 1);

            for (uint32_t c = 0; c < cols_; c += tileCols_) {
                uint32_t len = cols_ - c;
                if (len > tileCols_) len = tileCols_;

                uint32_t offset = r * cols_ + c;

                AscendC::LocalTensor<float> aLocal = qA_.AllocTensor<float>();
                AscendC::LocalTensor<float> pLocal = qP_.AllocTensor<float>();
                AscendC::LocalTensor<float> nLocal = qN_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpLocal = qTmp_.AllocTensor<float>();

                AscendC::DataCopy(aLocal, aGm_[offset], len);
                AscendC::DataCopy(pLocal, pGm_[offset], len);
                AscendC::DataCopy(nLocal, nGm_[offset], len);

                // tmp = (anchor - positive)
                AscendC::Sub(tmpLocal, aLocal, pLocal, len);
                // tmp = tmp * tmp
                AscendC::Mul(tmpLocal, tmpLocal, tmpLocal, len);

                // reduce sum into scalar
                AscendC::LocalTensor<float> red = qScalar_.AllocTensor<float>();
                AscendC::ReduceSum(red, tmpLocal, len);
                // apAcc += red
                AscendC::Add(apAcc, apAcc, red, 1);
                qScalar_.FreeTensor(red);

                // tmp = (anchor - negative)
                AscendC::Sub(tmpLocal, aLocal, nLocal, len);
                // tmp = tmp * tmp
                AscendC::Mul(tmpLocal, tmpLocal, tmpLocal, len);

                // reduce sum into scalar
                AscendC::LocalTensor<float> red2 = qScalar_.AllocTensor<float>();
                AscendC::ReduceSum(red2, tmpLocal, len);
                // anAcc += red2
                AscendC::Add(anAcc, anAcc, red2, 1);
                qScalar_.FreeTensor(red2);

                qTmp_.FreeTensor(tmpLocal);
                qN_.FreeTensor(nLocal);
                qP_.FreeTensor(pLocal);
                qA_.FreeTensor(aLocal);
            }

            // apAcc = sqrt(apAcc); anAcc = sqrt(anAcc);
            AscendC::Sqrt(apAcc, apAcc, 1);
            AscendC::Sqrt(anAcc, anAcc, 1);

            // apAcc = apAcc - anAcc
            AscendC::Sub(apAcc, apAcc, anAcc, 1);
            // apAcc = apAcc + margin
            AscendC::Adds(apAcc, apAcc, margin_, 1);
            // apAcc = max(apAcc, 0.0)
            AscendC::Maxs(apAcc, apAcc, 0.0f, 1);
            // sumLoss += apAcc
            AscendC::Add(sumLoss, sumLoss, apAcc, 1);

            qScalar_.FreeTensor(anAcc);
            qScalar_.FreeTensor(apAcc);
        }

        // mean = sumLoss / rows_
        AscendC::Divs(sumLoss, sumLoss, static_cast<float>(rows_), 1);

        // write to GM
        AscendC::DataCopy(yGm_[0], sumLoss, 1);
        qScalar_.FreeTensor(sumLoss);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qA_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qP_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> qN_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> qTmp_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 2> qScalar_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> pGm_;
    AscendC::GlobalTensor<float> nGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_{0};
    uint32_t cols_{0};
    uint32_t tileCols_{0};
    float margin_{1.0f};
};

extern "C" __global__ __aicore__ void triplet_margin_loss_custom(GM_ADDR anchor, GM_ADDR positive, GM_ADDR negative, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelTripletMarginLossCustom op;
    op.Init(anchor, positive, negative, y,
            tiling_data.rows, tiling_data.cols, tiling_data.tileCols, tiling_data.margin);
    op.Process();
}
