host_operator_src="""

#include "gemm_relu_divide_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel (tileN = 1):
//   xChunk: 1 * tileK
//   wChunk: 1 * tileK
//   tmpVec: 1 * tileK
//   scalars: small
// Total floats ~= 3 * tileK => bytes ~= 12 * tileK
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 256;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmReluDivideCustomTilingData tiling;

    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);
    const auto *b_shape = context->GetInputShape(2);
    (void)b_shape;

    // Shapes:
    // x: [M, K], w: [N, K], b: [N], divisor: scalar
    const uint32_t rank_x = static_cast<uint32_t>(x_shape->GetOriginShape().GetDimNum());
    const uint32_t rank_w = static_cast<uint32_t>(w_shape->GetOriginShape().GetDimNum());
    if (rank_x < 2 || rank_w < 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    // w second dim should match K

    uint32_t maxTileK = UB_BUDGET_BYTES / 12;  // elements
    maxTileK = AlignDown(maxTileK, ALIGN_ELEMS);
    if (maxTileK < ALIGN_ELEMS) maxTileK = ALIGN_ELEMS;
    if (maxTileK > HARD_CAP_ELEMS) maxTileK = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileK = maxTileK;
    if (K < tileK) {
        const uint32_t aligned = AlignDown(K, ALIGN_ELEMS);
        tileK = (aligned > 0) ? aligned : K;
    }
    if (tileK == 0) tileK = 1;

    // Blocks: spread by M rows
    uint32_t blockDim = M;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    const uint32_t rowsPerBlock = CeilDiv(M, blockDim);

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_rowsPerBlock(rowsPerBlock);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() < 2 || w_shape->GetOriginShape().GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }
    const int64_t M = x_shape->GetOriginShape().GetDim(0);
    const int64_t N = w_shape->GetOriginShape().GetDim(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    gert::Shape out;
    out.SetDimNum(2);
    out.SetDim(0, M);
    out.SetDim(1, N);
    out.SetOriginShape(out);

    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelGemmReluDivideCustom {
public:
    __aicore__ inline KernelGemmReluDivideCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileK, uint32_t rowsPerBlock)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        tileK_ = tileK;
        rowsPerBlock_ = rowsPerBlock;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)N_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        dGm_.SetGlobalBuffer((__gm__ float*)divisor, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)M_ * N_);

        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(scalarQueue_, 4, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t rowStart = blockIdx * rowsPerBlock_;
        uint32_t rowEnd = rowStart + rowsPerBlock_;
        if (rowStart >= M_) return;
        if (rowEnd > M_) rowEnd = M_;

        // Load divisor scalar once per block
        AscendC::LocalTensor<float> dLocal = scalarQueue_.AllocTensor<float>();
        AscendC::DataCopy(dLocal, dGm_[0], 1);
        scalarQueue_.EnQue(dLocal);
        dLocal = scalarQueue_.DeQue<float>();

        for (uint32_t m = rowStart; m < rowEnd; ++m) {
            const uint64_t xRowBase = ((uint64_t)m) * K_;
            const uint64_t yRowBase = ((uint64_t)m) * N_;

            for (uint32_t n = 0; n < N_; ++n) {
                // acc = b[n]
                AscendC::LocalTensor<float> accLocal = scalarQueue_.AllocTensor<float>();
                AscendC::LocalTensor<float> bLocal = scalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[n], 1);
                AscendC::DataCopy(accLocal, bLocal, 1);
                scalarQueue_.FreeTensor(bLocal);

                // Iterate over K in chunks
                for (uint32_t k0 = 0; k0 < K_; k0 += tileK_) {
                    uint32_t len = K_ - k0;
                    if (len > tileK_) len = tileK_;

                    // Load x chunk
                    AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[xRowBase + k0], len);
                    xQueue_.EnQue(xLocal);

                    // Load w chunk
                    AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                    AscendC::DataCopy(wLocal, wGm_[(uint64_t)n * K_ + k0], len);
                    wQueue_.EnQue(wLocal);

                    // Compute partial dot
                    xLocal = xQueue_.DeQue<float>();
                    wLocal = wQueue_.DeQue<float>();

                    AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                    AscendC::Mul(tmpLocal, xLocal, wLocal, len);

                    AscendC::LocalTensor<float> sumLocal = scalarQueue_.AllocTensor<float>();
                    AscendC::ReduceSum(sumLocal, tmpLocal, len);

                    AscendC::Add(accLocal, accLocal, sumLocal, 1);

                    scalarQueue_.FreeTensor(sumLocal);
                    tmpQueue_.FreeTensor(tmpLocal);
                    xQueue_.FreeTensor(xLocal);
                    wQueue_.FreeTensor(wLocal);
                }

                // ReLU and divide by divisor
                AscendC::Maxs(accLocal, accLocal, 0.0f, 1);
                AscendC::Div(accLocal, accLocal, dLocal, 1);

                // Store result y[m, n]
                AscendC::LocalTensor<float> yLocal = scalarQueue_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, accLocal, 1);  // copy acc to yLocal (both 1 elem)
                AscendC::DataCopy(yGm_[yRowBase + n], yLocal, 1);

                scalarQueue_.FreeTensor(yLocal);
                scalarQueue_.FreeTensor(accLocal);
            }
        }

        // free divisor local tensor
        scalarQueue_.FreeTensor(dLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 4> scalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> dGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0};
    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t tileK_{0};
    uint32_t rowsPerBlock_{1};
};

extern "C" __global__ __aicore__ void gemm_relu_divide_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR divisor, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmReluDivideCustom op;
    op.Init(x, w, b, divisor, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileK, tiling_data.rowsPerBlock);
    op.Process();
}
