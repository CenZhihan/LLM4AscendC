static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

    const uint32_t H   = static_cast<uint32_t>(H64);
    const uint32_t D   = static_cast<uint32_t>(D64);

    const uint32_t Kh  = static_cast<uint32_t>(Kh64);
    const uint32_t Kd  = static_cast<uint32_t>(Kd64);

    // No padding, stride=1, dilation=1
    const uint32_t Wout = (W >= Kw) ? (W - Kw + 1u) : 0u;

    // tileD from UB budget
    uint32_t maxTile = UB_BUDGET_BYTES / 8; // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileD = maxTile;
    if (Dout < tileD) {
        const uint32_t aligned = AlignDown(Dout, ALIGN_ELEMS);
        tileD = (aligned > 0) ? aligned : Dout;
    }
    if (tileD == 0u && Dout > 0u) tileD = Dout;

    const uint64_t totalTasks64 = static_cast<uint64_t>(N) * Cout * Wout * Hout;
    uint32_t blockDim = (totalTasks64 == 0) ? 1u : static_cast<uint32_t>(totalTasks64);

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;
constexpr int32_t OUT_BUF_NUM = 2;

class KernelConvStandard3dSquareInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t W, uint32_t H, uint32_t D,
                                uint32_t Kw, uint32_t Kh, uint32_t Kd,
                                uint32_t Wout, uint32_t Hout, uint32_t Dout,
                                uint32_t tileD)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        W_ = W; H_ = H; D_ = D;
        Kw_ = Kw; Kh_ = Kh; Kd_ = Kd;
        Wout_ = Wout; Hout_ = Hout; Dout_ = Dout;
        tileD_ = tileD;

        // Set global buffers
        const uint64_t xSize = (uint64_t)N_ * Cin_ * W_ * H_ * D_;
        const uint64_t wSize = (uint64_t)Cout_ * Cin_ * Kw_ * Kh_ * Kd_;
        const uint64_t ySize = (uint64_t)N_ * Cout_ * Wout_ * Hout_ * Dout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // Init queues
        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileD_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileD_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, sizeof(float)); // for scalar weight scratch if needed
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cout_ == 0 || Wout_ == 0 || Hout_ == 0 || Dout_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // total tasks across (N, Cout, Wout, Hout)
        const uint64_t totalTasks = (uint64_t)N_ * Cout_ * Wout_ * Hout_;

        for (uint64_t task = blockIdx; task < totalTasks; task += blockNum) {
            // map task -> indices (n, oc, w_out, h_out)
            uint64_t t = task;
            const uint64_t per_n = (uint64_t)Cout_ * Wout_ * Hout_;
            const uint32_t n = static_cast<uint32_t>(t / per_n);
            t = t % per_n;
            const uint32_t oc = static_cast<uint32_t>(t / (Wout_ * Hout_));
            t = t % (Wout_ * Hout_);
            const uint32_t w_out = static_cast<uint32_t>(t / Hout_);
            const uint32_t h_out = static_cast<uint32_t>(t % Hout_);

            // stride (N, C, W, H, D) with D fastest
            const uint64_t sD_x = 1ull;
            const uint64_t sH_x = (uint64_t)D_;
            const uint64_t sW_x = (uint64_t)H_ * D_;
            const uint64_t sC_x = (uint64_t)W_ * H_ * D_;
            const uint64_t sN_x = (uint64_t)Cin_ * sC_x;

            // y strides: (N, Cout, Wout, Hout, Dout) with Dout fastest
            const uint64_t sD_y = 1ull;
            const uint64_t sH_y = (uint64_t)Dout_;
            const uint64_t sW_y = (uint64_t)Hout_ * Dout_;
            const uint64_t sC_y = (uint64_t)Wout_ * Hout_ * Dout_;
            const uint64_t sN_y = (uint64_t)Cout_ * sC_y;

            // weight strides: (Cout, Cin, Kw, Kh, Kd) with Kd fastest
            const uint64_t sKz_w = 1ull;
            const uint64_t sKy_w = (uint64_t)Kd_;
            const uint64_t sKx_w = (uint64_t)Kh_ * Kd_;
            const uint64_t sCin_w= (uint64_t)Kw_ * Kh_ * Kd_;
            const uint64_t sCout_w=(uint64_t)Cin_ * sCin_w;

            for (uint32_t baseD = 0; baseD < Dout_; baseD += tileD_) {
                uint32_t len = Dout_ - baseD;
                if (len > tileD_) len = tileD_;

                // y local buffer
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, len);

                // Accumulate over cin and kernel window
                for (uint32_t ci = 0; ci < Cin_; ++ci) {
                    for (uint32_t kz = 0; kz < Kd_; ++kz) {
                        for (uint32_t ky = 0; ky < Kh_; ++ky) {
                            for (uint32_t kx = 0; kx < Kw_; ++kx) {
                                // Load input slice along D (contiguous)
                                const uint64_t d_in = (uint64_t)baseD + kz;
                                const uint64_t w_in = (uint64_t)w_out + kx;
                                const uint64_t h_in = (uint64_t)h_out + ky;

                                const uint64_t xOff = (uint64_t)n * sN_x
                                                     + (uint64_t)ci * sC_x
                                                     + w_in * sW_x
                                                     + h_in * sH_x
                                                     + d_in * sD_x;

                                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                                AscendC::DataCopy(xLocal, xGm_[xOff], len);

                                // Load weight scalar (oc, ci, kx, ky, kz)
                                const uint64_t wOff = (uint64_t)oc * sCout_w
                                                     + (uint64_t)ci * sCin_w
                                                     + (uint64_t)kx * sKx_w
                                                     + (uint64_t)ky * sKy_w
                                                     + (uint64_t)kz * sKz_w;

                                // Fetch weight into a tmp local 1-element buffer, then to scalar
                                AscendC::LocalTensor<float> wLocal = tmpQueue_.AllocTensor<float>();
                                AscendC::DataCopy(wLocal, wGm_[wOff], 1);
                                float wVal = wLocal.GetValue(0); // assumed API to read local scalar
                                tmpQueue_.FreeTensor(wLocal);

                                // y += x * wVal
                                AscendC::Muls(xLocal, xLocal, wVal, len);
                                AscendC::Add(yLocal, yLocal, xLocal, len);

                                inQueueX_.FreeTensor(xLocal);
                            }
                        }
                    }
                }

                // Commit and write out
                outQueueY_.EnQue(yLocal);
                AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();

                const uint64_t yOff = (uint64_t)n * sN_y
                                     + (uint64_t)oc * sC_y
                                     + (uint64_t)w_out * sW_y
                                     + (uint64_t)h_out * sH_y
                                     + (uint64_t)baseD * sD_y;

                AscendC::DataCopy(yGm_[yOff], yOut, len);
                outQueueY_.FreeTensor(yOut);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t W_{0}, H_{0}, D_{0};
    uint32_t Kw_{0}, Kh_{0}, Kd_{0};
    uint32_t Wout_{0}, Hout_{0}, Dout_{0};
    uint32_t tileD_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dSquareInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.W, tiling_data.H, tiling_data.D,
            tiling_data.Kw, tiling_data.Kh, tiling_data.Kd,
            tiling_data.Wout, tiling_data.Hout, tiling_data.Dout,
            tiling_data.tileD);
    op.Process();
}

#include "pytorch_npu_helper.hpp"

at::Tensor conv3d_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    // x: [N,C,W,H,D]; weight: [Cout,Cin,Kw,Kh,Kd]
    // Output: [N,Cout,Wout,Hout,Dout]
    // Allocate output tensor with inferred shape
    TORCH_CHECK(x.dim() == 5 && weight.dim() == 5, "Inputs must be 5D [N,C,W,H,D] and [Cout,Cin,Kw,Kh,Kd]");
    auto N = x.size(0);
    auto Cin = x.size(1);
    auto W = x.size(2);
    auto H = x.size(3);
    auto D = x.size(4);

    auto Cout = weight.size(0);
    auto Kw = weight.size(2);
    auto Kh = weight.size(3);
    auto Kd = weight.size(4);

    TORCH_CHECK(weight.size(1) == Cin, "Weight Cin must match input C");

    auto Wout = W - Kw + 1;
    auto Hout = H - Kh + 1;
    auto Dout = D - Kd + 1;
    TORCH_CHECK(Wout >= 0 && Hout >= 0 && Dout >= 0, "Kernel larger than input in some dimension");

    at::Tensor y = at::empty({N, Cout, Wout, Hout, Dout}, x.options());
    EXEC_NPU_CMD(aclnnConvStandard3dSquareInputAsymmetricKernelCustom, x, weight, y);
    return y;
}

    m.impl("conv_standard_3d_square_input_asymmetric_kernel_custom", &conv3d_impl_npu);
