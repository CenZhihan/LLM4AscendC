namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0u) ? x : ((x / a) * a);
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dReluBiasAddCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    if (xShape == nullptr || wShape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const int64_t N   = xShape->GetDim(0);
    const int64_t Cin = xShape->GetDim(1);
    const int64_t Hin = xShape->GetDim(2);
    const int64_t Win = xShape->GetDim(3);

    const int64_t Cout = wShape->GetDim(0);
    const int64_t Kh   = wShape->GetDim(2);
    const int64_t Kw   = wShape->GetDim(3);

    if (Kh <= 0 || Kw <= 0 || Kh > Hin || Kw > Win) {
        return ge::GRAPH_FAILED;
    }

    const int64_t Hout = Hin - Kh + 1;
    const int64_t Wout = Win - Kw + 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    bool hasConvBias = true;
    if (attrs != nullptr && attrs->GetAttrsNum() > 0) {
        const bool *p = attrs->GetAttrPointer<bool>(0);
        if (p != nullptr) hasConvBias = *p;
    }

    // UB budget estimation:
    // xRow: 1 * tileW
    // yRow: 1 * tileW
    // tmpRow: 1 * tileW
    // Total floats = 3 * tileW -> bytes = 12 * tileW
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
    static constexpr uint32_t ALIGN_ELEMS = 256;
    static constexpr uint32_t HARD_CAP_W = 8192;

    uint32_t maxTileW = UB_BUDGET_BYTES / 12;
    if (maxTileW > HARD_CAP_W) maxTileW = HARD_CAP_W;
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;

    uint32_t tileW = static_cast<uint32_t>(Wout);
    if (tileW > maxTileW) tileW = maxTileW;
    tileW = AlignDown(tileW, ALIGN_ELEMS);
    if (tileW == 0u) tileW = static_cast<uint32_t>(Wout);

    const uint32_t pairs = static_cast<uint32_t>(N * Hout);
    uint32_t blockDim = pairs;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_Cin(static_cast<uint32_t>(Cin));
    tiling.set_Hin(static_cast<uint32_t>(Hin));
    tiling.set_Win(static_cast<uint32_t>(Win));

    tiling.set_Cout(static_cast<uint32_t>(Cout));
    tiling.set_Kh(static_cast<uint32_t>(Kh));
    tiling.set_Kw(static_cast<uint32_t>(Kw));

    tiling.set_Hout(static_cast<uint32_t>(Hout));
    tiling.set_Wout(static_cast<uint32_t>(Wout));
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);
    tiling.set_hasConvBias(hasConvBias);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

using AscendC::DataCopy;

class KernelConv2dReluBiasAddCustom {
public:
    __aicore__ inline KernelConv2dReluBiasAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR convb, GM_ADDR postb, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Hin, uint32_t Win,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t Hout, uint32_t Wout, uint32_t tileW, bool hasConvBias)
    {
        N_ = N; Cin_ = Cin; Hin_ = Hin; Win_ = Win;
        Cout_ = Cout; Kh_ = Kh; Kw_ = Kw;
        Hout_ = Hout; Wout_ = Wout; tileW_ = tileW; hasConvBias_ = hasConvBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Cin_ * Hin_ * Win_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, (uint64_t)Cout_ * Cin_ * Kh_ * Kw_);
        if (hasConvBias_) {
            cbGm_.SetGlobalBuffer((__gm__ float*)convb, Cout_);
        }
        pbGm_.SetGlobalBuffer((__gm__ float*)postb, Cout_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Cout_ * Hout_ * Wout_);

        pipe_.InitBuffer(xRowQ_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(yRowQ_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tileW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (Hout_ == 0 || Wout_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        const uint32_t totalPairs = N_ * Hout_; // each pair is (n, h_out)
        for (uint32_t pair = blockIdx; pair < totalPairs; pair += blockNum) {
            const uint32_t n = pair / Hout_;
            const uint32_t h_out = pair % Hout_;
            // iterate output channels one by one to simplify UB usage
            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                for (uint32_t w0 = 0; w0 < Wout_; w0 += tileW_) {
                    uint32_t curTile = Wout_ - w0;
                    if (curTile > tileW_) curTile = tileW_;

                    AscendC::LocalTensor<float> yLocal = yRowQ_.AllocTensor<float>();
                    AscendC::Duplicate(yLocal, 0.0f, curTile);

                    // Accumulate convolution
                    for (uint32_t ci = 0; ci < Cin_; ++ci) {
                        for (uint32_t kh = 0; kh < Kh_; ++kh) {
                            const uint32_t h_in = h_out + kh;
                            for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                const uint32_t w_in_start = w0 + kw;

                                // Load input row slice x[n, ci, h_in, w_in_start : w_in_start + curTile]
                                AscendC::LocalTensor<float> xLocal = xRowQ_.AllocTensor<float>();
                                const uint64_t x_off = ((((uint64_t)n * Cin_ + ci) * Hin_ + h_in) * Win_ + w_in_start);
                                DataCopy(xLocal, xGm_[x_off], curTile);

                                // Load weight scalar w[oc, ci, kh, kw]
                                const uint64_t w_idx = ((((uint64_t)oc * Cin_ + ci) * Kh_ + kh) * Kw_ + kw);
                                float w_val = wGm_.GetValue(w_idx);

                                // tmp = xLocal * w_val
                                AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                                AscendC::Muls(tmp, xLocal, w_val, curTile);
                                // yLocal += tmp
                                AscendC::Add(yLocal, yLocal, tmp, curTile);

                                tmpQ_.FreeTensor(tmp);
                                xRowQ_.FreeTensor(xLocal);
                            }
                        }
                    }

                    // Add conv_bias if present: yLocal += conv_bias[oc]
                    if (hasConvBias_) {
                        float cb = cbGm_.GetValue(oc);
                        AscendC::Adds(yLocal, yLocal, cb, curTile);
                    }

                    // ReLU: yLocal = max(yLocal, 0)
                    AscendC::Maxs(yLocal, yLocal, 0.0f, curTile);

                    // Add post_bias[oc]
                    float pb = pbGm_.GetValue(oc);
                    AscendC::Adds(yLocal, yLocal, pb, curTile);

                    // Store: y[n, oc, h_out, w0 : w0 + curTile]
                    const uint64_t y_off = ((((uint64_t)n * Cout_ + oc) * Hout_ + h_out) * Wout_ + w0);
                    DataCopy(yGm_[y_off], yLocal, curTile);

                    yRowQ_.FreeTensor(yLocal);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> cbGm_;
    AscendC::GlobalTensor<float> pbGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Hin_{0}, Win_{0};
    uint32_t Cout_{0}, Kh_{0}, Kw_{0};
    uint32_t Hout_{0}, Wout_{0};
    uint32_t tileW_{0};
    bool hasConvBias_{true};
};

extern "C" __global__ __aicore__ void conv2d_relu_bias_add_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR post_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluBiasAddCustom op;
    op.Init(x, weight, conv_bias, post_bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Hin, tiling_data.Win,
            tiling_data.Cout, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Hout, tiling_data.Wout, tiling_data.tileW, tiling_data.hasConvBias);
    op.Process();
}
