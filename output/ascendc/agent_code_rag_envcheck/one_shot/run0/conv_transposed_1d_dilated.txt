#include "conv_transposed_1d_dilated_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
}

namespace optiling {

// Memory model:
//   yLocal:      1 * tile
//   xLocal:      1 * tile
// Total floats = 2 * tile -> bytes = 8 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative
static constexpr uint32_t ALIGN_ELEMS     = 1024;
static constexpr uint32_t HARD_CAP_ELEMS  = 32768;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed1dDilatedCustomTilingData t;
    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);

    // x: (N, Cin, Lin), weight: (Cin, Cout, K)
    const uint64_t n64    = x_shape->GetDim(0);
    const uint64_t cin64  = x_shape->GetDim(1);
    const uint64_t lin64  = x_shape->GetDim(2);
    const uint64_t wcin64 = w_shape->GetDim(0);
    const uint64_t cout64 = w_shape->GetDim(1);
    const uint64_t k64    = w_shape->GetDim(2);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stridePtr   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *paddingPtr  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilationPtr = attrs->GetAttrPointer<int32_t>(2);
    const uint32_t stride   = static_cast<uint32_t>(*stridePtr);
    const uint32_t padding  = static_cast<uint32_t>(*paddingPtr);
    const uint32_t dilation = static_cast<uint32_t>(*dilationPtr);

    // Basic sanity (Cin matches)
    if (cin64 != wcin64) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N   = static_cast<uint32_t>(n64);
    const uint32_t Cin = static_cast<uint32_t>(cin64);
    const uint32_t Cout= static_cast<uint32_t>(cout64);
    const uint32_t Lin = static_cast<uint32_t>(lin64);
    const uint32_t K   = static_cast<uint32_t>(k64);

    // Output length formula:
    // L_out = (L_in - 1) * stride - 2 * padding + dilation * (K - 1) + 1
    const int64_t lout64 = (static_cast<int64_t>(Lin) - 1LL) * static_cast<int64_t>(stride)
                         - 2LL * static_cast<int64_t>(padding)
                         + static_cast<int64_t>(dilation) * static_cast<int64_t>(K - 1U)
                         + 1LL;
    if (lout64 <= 0) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t Lout = static_cast<uint32_t>(lout64);

    // tileLen from UB budget: bytes = 8 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 8;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (Lout < tileLen) {
        const uint32_t aligned = AlignDown(Lout, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : Lout;
    }

    const uint32_t tilesPerPair = (tileLen == 0) ? 1u : CeilDiv(Lout, tileLen);
    // 1D work items: N * Cout * tilesPerPair
    const uint64_t totalWork64 = static_cast<uint64_t>(N) * static_cast<uint64_t>(Cout) * static_cast<uint64_t>(tilesPerPair);
    uint32_t blockDim = (totalWork64 < 8ULL) ? static_cast<uint32_t>(totalWork64) : 24u;
    if (blockDim == 0u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    t.set_n(N);
    t.set_cin(Cin);
    t.set_cout(Cout);
    t.set_lin(Lin);
    t.set_kernelSize(K);
    t.set_stride(stride);
    t.set_padding(padding);
    t.set_dilation(dilation);
    t.set_lout(Lout);
    t.set_tileLength(tileLen);
    t.set_tilesPerPair(tilesPerPair);
    t.set_blockDim(blockDim);

    t.SaveToBuffer(context->GetRawTilingData()->GetData(),
                   context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(t.GetDataSize());

    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // (N,Cin,Lin)
    const gert::Shape *w_shape = context->GetInputShape(1); // (Cin,Cout,K)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *stridePtr   = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *paddingPtr  = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *dilationPtr = attrs->GetAttrPointer<int32_t>(2);

    const int64_t N   = x_shape->GetDim(0);
    const int64_t Cin = x_shape->GetDim(1);
    const int64_t Lin = x_shape->GetDim(2);
    const int64_t Wcin= w_shape->GetDim(0);
    const int64_t Cout= w_shape->GetDim(1);
    const int64_t K   = w_shape->GetDim(2);

    if (Cin != Wcin) {
        return GRAPH_FAILED;
    }

    const int64_t stride   = static_cast<int64_t>(*stridePtr);
    const int64_t padding  = static_cast<int64_t>(*paddingPtr);
    const int64_t dilation = static_cast<int64_t>(*dilationPtr);

    const int64_t Lout = (Lin - 1) * stride - 2 * padding + dilation * (K - 1) + 1;
    if (Lout <= 0) {
        return GRAPH_FAILED;
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Lout);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM  = 2;
constexpr int32_t OUT_BUF_NUM = 1;

class KernelConvTransposed1dDilatedCustom {
public:
    __aicore__ inline KernelConvTransposed1dDilatedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t kernelSize,
                                uint32_t stride, uint32_t padding, uint32_t dilation,
                                uint32_t lout, uint32_t tileLength, uint32_t tilesPerPair)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        lin_ = lin; k_ = kernelSize;
        stride_ = stride; padding_ = padding; dilation_ = dilation;
        lout_ = lout; tileLen_ = tileLength; tilesPerPair_ = tilesPerPair;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * cin_ * lin_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cin_) * cout_ * k_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * cout_ * lout_);

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUF_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || lout_ == 0 || tileLen_ == 0) return;
        // This implementation currently supports stride == 1
        if (stride_ != 1u) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalWork = static_cast<uint64_t>(n_) * cout_ * tilesPerPair_;
        if (totalWork == 0ULL) return;

        for (uint64_t work = blockIdx; work < totalWork; work += blockNum) {
            // Decode (n, co, tileIdx)
            const uint32_t tilesPer = tilesPerPair_;
            const uint64_t ncTiles = static_cast<uint64_t>(cout_) * tilesPer;
            const uint32_t n = static_cast<uint32_t>(work / ncTiles);
            const uint64_t rem = work - static_cast<uint64_t>(n) * ncTiles;
            const uint32_t co = static_cast<uint32_t>(rem / tilesPer);
            const uint32_t tileIdx = static_cast<uint32_t>(rem % tilesPer);

            const uint32_t o0 = tileIdx * tileLen_;
            uint32_t curLen = lout_ - o0;
            if (curLen > tileLen_) curLen = tileLen_;

            // yLocal tile
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::Duplicate(yLocal, 0.0f, tileLen_); // zero init full buffer
            // Only curLen area will be used for this tile

            // Accumulate over ci and kernel taps
            for (uint32_t ci = 0; ci < cin_; ++ci) {
                for (uint32_t kk = 0; kk < k_; ++kk) {
                    // Contribution y[o] += x[i] * w, with i = o + padding - kk*dilation
                    // For valid i in [0, lin_-1], restrict o range:
                    // o in [ -padding + kk*dilation, lin_ - 1 - padding + kk*dilation ]
                    int32_t o_valid_start_global = static_cast<int32_t>(-static_cast<int32_t>(padding_)) + static_cast<int32_t>(kk * dilation_);
                    int32_t o_valid_end_global   = static_cast<int32_t>(lin_) - 1 - static_cast<int32_t>(padding_) + static_cast<int32_t>(kk * dilation_);
                    // Intersect with current tile [o0, o0+curLen)
                    int32_t o_tile_start = static_cast<int32_t>(o0);
                    int32_t o_tile_end   = static_cast<int32_t>(o0 + curLen);
                    int32_t o_valid_start = o_tile_start > o_valid_start_global ? o_tile_start : o_valid_start_global;
                    int32_t o_valid_end   = o_tile_end   < o_valid_end_global + 1 ? o_tile_end : (o_valid_end_global + 1);
                    int32_t lenValid = o_valid_end - o_valid_start;
                    if (lenValid <= 0) {
                        continue;
                    }

                    const uint32_t yLocalOff = static_cast<uint32_t>(o_valid_start - static_cast<int32_t>(o0));
                    const uint32_t iStart = static_cast<uint32_t>(o_valid_start + static_cast<int32_t>(padding_) - static_cast<int32_t>(kk * dilation_));
                    const uint32_t lenU = static_cast<uint32_t>(lenValid);

                    // Load x segment
                    AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                    const uint64_t xBase = (static_cast<uint64_t>(n) * cin_ + ci) * lin_ + iStart;
                    AscendC::DataCopy(xLocal, xGm_[xBase], lenU);

                    // Scale by weight scalar
                    const float wScalar = wGm_[(static_cast<uint64_t>(ci) * cout_ + co) * k_ + kk];
                    AscendC::Muls(xLocal, xLocal, wScalar, lenU);

                    // Accumulate into yLocal slice
                    AscendC::LocalTensor<float> ySlice = yLocal[yLocalOff];
                    AscendC::Add(ySlice, ySlice, xLocal, lenU);

                    inQueueX_.FreeTensor(xLocal);
                }
            }

            // Write back yLocal tile
            const uint64_t yBase = (static_cast<uint64_t>(n) * cout_ + co) * lout_ + o0;
            AscendC::DataCopy(yGm_[yBase], yLocal, curLen);

            outQueueY_.FreeTensor(yLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM>   inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUF_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t lin_{0}, k_{0};
    uint32_t stride_{1}, padding_{0}, dilation_{1};
    uint32_t lout_{0}, tileLen_{0}, tilesPerPair_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_1d_dilated_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed1dDilatedCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.kernelSize,
            tiling_data.stride, tiling_data.padding, tiling_data.dilation,
            tiling_data.lout, tiling_data.tileLength, tiling_data.tilesPerPair);
    op.Process();
}
