    const int64_t sH = *stride_h;
    const int64_t sW = *stride_w;
    const int64_t pH = *pad_h;
    const int64_t pW = *pad_w;
    const int64_t dH = *dil_h;
    const int64_t dW = *dil_w;

    const int64_t oh = (H + 2 * pH >= dH * (KH - 1) + 1)
                       ? ((H + 2 * pH - (dH * (KH - 1) + 1)) / sH + 1)
                       : 0;
    const int64_t ow = (W + 2 * pW >= dW * (KW - 1) + 1)
                       ? ((W + 2 * pW - (dW * (KW - 1) + 1)) / sW + 1)

kernel_src="""
#include "kernel_operator.h"

extern \"C\" __global__ __aicore__
void conv_standard_2d_asymmetric_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    const uint32_t N  = tiling_data.N;
    const uint32_t C  = tiling_data.C;
    const uint32_t H  = tiling_data.H;
    const uint32_t W  = tiling_data.W;
    const uint32_t OC = tiling_data.OC;
    const uint32_t IC = tiling_data.IC;
    const uint32_t KH = tiling_data.KH;
    const uint32_t KW = tiling_data.KW;
    const uint32_t OH = tiling_data.OH;
    const uint32_t OW = tiling_data.OW;
    const uint32_t sH = tiling_data.strideH;
    const uint32_t sW = tiling_data.strideW;
    const int32_t  pH = static_cast<int32_t>(tiling_data.padH);
    const int32_t  pW = static_cast<int32_t>(tiling_data.padW);
    const uint32_t dH = tiling_data.dilH;
    const uint32_t dW = tiling_data.dilW;
    const uint32_t tileOW = (tiling_data.tileOW > 0u) ? tiling_data.tileOW : OW;

    __gm__ float* xPtr = reinterpret_cast<__gm__ float*>(x);
    __gm__ float* wPtr = reinterpret_cast<__gm__ float*>(weight);
    __gm__ float* yPtr = reinterpret_cast<__gm__ float*>(y);

    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

    const uint64_t totalRows = static_cast<uint64_t>(N) * static_cast<uint64_t>(OC) * static_cast<uint64_t>(OH);
    if (totalRows == 0 || OW == 0) {
        return;
    }

    for (uint64_t row = blockIdx; row < totalRows; row += blockNum) {
        uint64_t tmp = row;
        const uint32_t n  = static_cast<uint32_t>(tmp / (static_cast<uint64_t>(OC) * OH));
        tmp = tmp % (static_cast<uint64_t>(OC) * OH);
        const uint32_t oc = static_cast<uint32_t>(tmp / OH);
        const uint32_t oh = static_cast<uint32_t>(tmp % OH);

        for (uint32_t wBase = 0; wBase < OW; wBase += tileOW) {
            const uint32_t currOW = AscendC::min(OW - wBase, tileOW);

            for (uint32_t i = 0; i < currOW; ++i) {
                const uint32_t ow = wBase + i;
                float acc = 0.0f;

                // Compute one output element [n, oc, oh, ow]
                for (uint32_t ic = 0; ic < IC; ++ic) {
                    for (uint32_t kh = 0; kh < KH; ++kh) {
                        const int32_t ih = static_cast<int32_t>(oh) * static_cast<int32_t>(sH)
                                         + static_cast<int32_t>(kh) * static_cast<int32_t>(dH)
                                         - pH;
                        if (ih < 0 || ih >= static_cast<int32_t>(H)) {
                            continue;
                        }
                        for (uint32_t kw = 0; kw < KW; ++kw) {
                            const int32_t iw = static_cast<int32_t>(ow) * static_cast<int32_t>(sW)
                                             + static_cast<int32_t>(kw) * static_cast<int32_t>(dW)
                                             - pW;
                            if (iw < 0 || iw >= static_cast<int32_t>(W)) {
                                continue;
                            }

                            const uint64_t xoff = (((static_cast<uint64_t>(n) * C + ic) * H + static_cast<uint32_t>(ih)) * W
                                                  + static_cast<uint32_t>(iw));
                            const uint64_t woff = ((((static_cast<uint64_t>(oc) * IC) + ic) * KH + kh) * KW + kw);

                            acc += xPtr[xoff] * wPtr[woff];
                        }
                    }
                }

                const uint64_t yoff = (((static_cast<uint64_t>(n) * OC + oc) * OH + oh) * OW + ow);
                yPtr[yoff] = acc;
            }
        }
    }
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline int64_t out_size_1d(int64_t in, int64_t pad, int64_t dil, int64_t k, int64_t stride) {
    if (in + 2 * pad < dil * (k - 1) + 1) return 0;
    return (in + 2 * pad - (dil * (k - 1) + 1)) / stride + 1;
}

at::Tensor conv2d_impl_npu(const at::Tensor& x, const at::Tensor& weight,
                           int64_t stride_h, int64_t stride_w,
                           int64_t pad_h, int64_t pad_w,
                           int64_t dil_h, int64_t dil_w,
                           int64_t groups) {
    TORCH_CHECK(x.dim() == 4, \"x must be NCHW\");
    TORCH_CHECK(weight.dim() == 4, \"weight must be OIHW\");
    TORCH_CHECK(groups == 1, \"groups must be 1 for ConvStandard2dAsymmetricInputSquareKernelCustom\");

    const auto N = x.size(0);
    const auto C = x.size(1);
    const auto H = x.size(2);
    const auto W = x.size(3);

    const auto OC = weight.size(0);
    const auto IC = weight.size(1);
    const auto KH = weight.size(2);
    const auto KW = weight.size(3);

    TORCH_CHECK(C == IC * groups, \"Input channels mismatch with weight and groups\");

    const auto OH = out_size_1d(H, pad_h, dil_h, KH, stride_h);
    const auto OW = out_size_1d(W, pad_w, dil_w, KW, stride_w);

    at::Tensor y = at::empty({N, OC, OH, OW}, x.options());

    EXEC_NPU_CMD(aclnnConvStandard2dAsymmetricInputSquareKernelCustom,
                 x, weight,
                 static_cast<int>(stride_h), static_cast<int>(stride_w),
                 static_cast<int>(pad_h), static_cast<int>(pad_w),
                 static_cast<int>(dil_h), static_cast<int>(dil_w),
                 static_cast<int>(groups),
                 y);
    return y;
}

at::Tensor conv_standard_2d_asymmetric_input_square_kernel_custom_impl(const at::Tensor& x, const at::Tensor& weight) {
    // Defaults: stride=(1,1), padding=(0,0), dilation=(1,1), groups=1
    return conv2d_impl_npu(x, weight, 1, 1, 0, 0, 1, 1, 1);
}
