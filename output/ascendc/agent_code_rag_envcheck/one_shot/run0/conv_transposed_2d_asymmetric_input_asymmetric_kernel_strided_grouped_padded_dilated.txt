#include "kernel_operator.h"

class KernelConvTransposed2dAsymStrGrpPadDil {
public:
    __aicore__ inline KernelConvTransposed2dAsymStrGrpPadDil() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR tiling)
    {
        GET_TILING_DATA(tilingData, tiling);

        N_   = tilingData.N;
        Cin_ = tilingData.Cin;
        H_   = tilingData.H;
        W_   = tilingData.W;

        Cout_ = tilingData.Cout;
        kH_   = tilingData.kH;
        kW_   = tilingData.kW;

        HOut_ = tilingData.HOut;
        WOut_ = tilingData.WOut;

        sH_ = tilingData.strideH;
        sW_ = tilingData.strideW;
        pH_ = tilingData.padH;
        pW_ = tilingData.padW;
        dH_ = tilingData.dilH;
        dW_ = tilingData.dilW;
        oPH_ = tilingData.outPadH;
        oPW_ = tilingData.outPadW;
        groups_ = tilingData.groups;
        coTile_ = tilingData.coTile;

        xGm_ = reinterpret_cast<__gm__ float*>(x);
        wGm_ = reinterpret_cast<__gm__ float*>(w);
        yGm_ = reinterpret_cast<__gm__ float*>(y);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t coStart = blockIdx * coTile_;
        uint32_t coEnd = coStart + coTile_;
        if (coEnd > Cout_) coEnd = Cout_;
        if (coStart >= coEnd) return;

        const uint32_t CoutG = Cout_ / groups_;
        const uint32_t CinG = Cin_ / groups_;

        // Strides in linear memory for NCHW layout
        const uint64_t xStrideN = static_cast<uint64_t>(Cin_) * H_ * W_;
        const uint64_t xStrideC = static_cast<uint64_t>(H_) * W_;
        const uint64_t xStrideH = static_cast<uint64_t>(W_);
        const uint64_t xStrideW = 1;

        const uint64_t yStrideN = static_cast<uint64_t>(Cout_) * HOut_ * WOut_;
        const uint64_t yStrideC = static_cast<uint64_t>(HOut_) * WOut_;
        const uint64_t yStrideH = static_cast<uint64_t>(WOut_);
        const uint64_t yStrideW = 1;

        const uint64_t wStrideCIn   = static_cast<uint64_t>(CoutG) * kH_ * kW_;
        const uint64_t wStrideCOutG = static_cast<uint64_t>(kH_) * kW_;
        const uint64_t wStrideKH    = static_cast<uint64_t>(kW_);
        const uint64_t wStrideKW    = 1;

        for (uint32_t n = 0; n < N_; ++n) {
            for (uint32_t co = coStart; co < coEnd; ++co) {
                const uint32_t g = co / CoutG;       // group index
                const uint32_t coG = co % CoutG;     // output channel in group
                const uint32_t cinStart = g * CinG;
                const uint32_t cinEnd   = cinStart + CinG;

                for (uint32_t ho = 0; ho < HOut_; ++ho) {
                    for (uint32_t wo = 0; wo < WOut_; ++wo) {
                        float acc = 0.0f;

                        for (uint32_t ci = cinStart; ci < cinEnd; ++ci) {
                            for (uint32_t kh = 0; kh < kH_; ++kh) {
                                // Compute the corresponding input h index if valid
                                // rhs_h = ho + padH - kh * dilH
                                int32_t rhs_h = static_cast<int32_t>(ho) + static_cast<int32_t>(pH_) - static_cast<int32_t>(kh * dH_);
                                if (rhs_h < 0) continue;
                                if (rhs_h % static_cast<int32_t>(sH_) != 0) continue;
                                int32_t hi = rhs_h / static_cast<int32_t>(sH_);
                                if (hi < 0 || hi >= static_cast<int32_t>(H_)) continue;

                                for (uint32_t kw = 0; kw < kW_; ++kw) {
                                    int32_t rhs_w = static_cast<int32_t>(wo) + static_cast<int32_t>(pW_) - static_cast<int32_t>(kw * dW_);
                                    if (rhs_w < 0) continue;
                                    if (rhs_w % static_cast<int32_t>(sW_) != 0) continue;
                                    int32_t wi = rhs_w / static_cast<int32_t>(sW_);
                                    if (wi < 0 || wi >= static_cast<int32_t>(W_)) continue;

                                    // Load input and weight and accumulate
                                    const uint64_t xOff = n * xStrideN + ci * xStrideC + static_cast<uint64_t>(hi) * xStrideH + static_cast<uint64_t>(wi) * xStrideW;
                                    const uint64_t wOff = ci * wStrideCIn + coG * wStrideCOutG + kh * wStrideKH + kw * wStrideKW;

                                    acc += xGm_[xOff] * wGm_[wOff];
                                }
                            }
                        }

                        const uint64_t yOff = n * yStrideN + co * yStrideC + static_cast<uint64_t>(ho) * yStrideH + static_cast<uint64_t>(wo) * yStrideW;
                        yGm_[yOff] = acc;
                    }
                }
            }
        }
    }

private:
    __gm__ float *xGm_;
    __gm__ float *wGm_;
    __gm__ float *yGm_;

    uint32_t N_, Cin_, H_, W_;
    uint32_t Cout_, kH_, kW_;
    uint32_t HOut_, WOut_;
    uint32_t sH_, sW_, pH_, pW_, dH_, dW_, oPH_, oPW_;
    uint32_t groups_;
    uint32_t coTile_;
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_strided_grouped_padded_dilated_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelConvTransposed2dAsymStrGrpPadDil op;
    op.Init(x, weight, y, tiling);
    op.Process();
}
