#include "average_pooling_3d_custom_tiling.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    AveragePooling3dCustomTilingData tiling;

    const gert::Shape* in_shape = context->GetInputShape(0);
    const gert::Shape& oshape = in_shape->GetOriginShape();

    uint32_t N = static_cast<uint32_t>(oshape.GetDim(0));
    uint32_t C = static_cast<uint32_t>(oshape.GetDim(1));
    uint32_t Di = static_cast<uint32_t>(oshape.GetDim(2));
    uint32_t Hi = static_cast<uint32_t>(oshape.GetDim(3));
    uint32_t Wi = static_cast<uint32_t>(oshape.GetDim(4));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t k = *attrs->GetAttrPointer<int32_t>(0);
    int32_t s = *attrs->GetAttrPointer<int32_t>(1);
    int32_t p = *attrs->GetAttrPointer<int32_t>(2);
    bool count_include_pad = false;
    if (attrs->GetAttrNum() >= 4) {
        count_include_pad = *attrs->GetAttrPointer<bool>(3);
    }

    if (s <= 0) s = k;

    uint32_t Kd = static_cast<uint32_t>(k);
    uint32_t Kh = static_cast<uint32_t>(k);
    uint32_t Kw = static_cast<uint32_t>(k);

    uint32_t Sd = static_cast<uint32_t>(s);
    uint32_t Sh = static_cast<uint32_t>(s);
    uint32_t Sw = static_cast<uint32_t>(s);

    uint32_t Pd = static_cast<uint32_t>(p);
    uint32_t Ph = static_cast<uint32_t>(p);
    uint32_t Pw = static_cast<uint32_t>(p);

    // Output dims
    uint32_t Do = (Di + 2 * Pd >= Kd) ? static_cast<uint32_t>((Di + 2 * Pd - Kd) / Sd + 1) : 0u;
    uint32_t Ho = (Hi + 2 * Ph >= Kh) ? static_cast<uint32_t>((Hi + 2 * Ph - Kh) / Sh + 1) : 0u;
    uint32_t Wo = (Wi + 2 * Pw >= Kw) ? static_cast<uint32_t>((Wi + 2 * Pw - Kw) / Sw + 1) : 0u;

    uint64_t totalOutElems = static_cast<uint64_t>(N) * C * Do * Ho * Wo;

    // Choose tile length from UB budget: we keep only output tile + tiny tmp row buffer (Kw)
    uint32_t maxTile = UB_BUDGET_BYTES / sizeof(float);
    if (maxTile > HARD_CAP_ELEMS) maxTile = HARD_CAP_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    uint32_t tileLen = maxTile;
    if (totalOutElems < tileLen) {
        uint32_t aligned = AlignDown(static_cast<uint32_t>(totalOutElems), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(totalOutElems);
    }

    uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalOutElems, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_Di(Di);
    tiling.set_Hi(Hi);
    tiling.set_Wi(Wi);

    tiling.set_Do(Do);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);

    tiling.set_Kd(Kd);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);

    tiling.set_Sd(Sd);
    tiling.set_Sh(Sh);
    tiling.set_Sw(Sw);

    tiling.set_Pd(Pd);
    tiling.set_Ph(Ph);
    tiling.set_Pw(Pw);

    tiling.set_CountIncludePad(count_include_pad ? 1u : 0u);

    tiling.set_totalOutElems(totalOutElems);
    tiling.set_spanW(Kw);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelAveragePooling3dCustom {
public:
    __aicore__ inline KernelAveragePooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C,
                                uint32_t Di, uint32_t Hi, uint32_t Wi,
                                uint32_t Do, uint32_t Ho, uint32_t Wo,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t Sd, uint32_t Sh, uint32_t Sw,
                                uint32_t Pd, uint32_t Ph, uint32_t Pw,
                                uint32_t countIncludePad,
                                uint64_t totalOutElems,
                                uint32_t tileLength, uint32_t spanW)
    {
        N_ = N; C_ = C;
        Di_ = Di; Hi_ = Hi; Wi_ = Wi;
        Do_ = Do; Ho_ = Ho; Wo_ = Wo;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        Sd_ = Sd; Sh_ = Sh; Sw_ = Sw;
        Pd_ = Pd; Ph_ = Ph; Pw_ = Pw;
        countIncludePad_ = (countIncludePad != 0);
        totalOut_ = totalOutElems;
        tileLen_ = tileLength;
        spanW_ = (spanW > 0) ? spanW : 1u;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_)*C_*Di_*Hi_*Wi_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalOut_);

        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, spanW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * tileLen_;
        uint64_t base = static_cast<uint64_t>(blockIdx) * tileLen_;
        if (base >= totalOut_) return;

        uint32_t curLen = static_cast<uint32_t>(totalOut_ - base);
        if (curLen > tileLen_) curLen = tileLen_;

        ComputeAndStore(base, curLen);

        while (true) {
            const uint64_t nextBase = base + strideElems;
            if (nextBase >= totalOut_) break;
            uint32_t nextLen = static_cast<uint32_t>(totalOut_ - nextBase);
            if (nextLen > tileLen_) nextLen = tileLen_;
            ComputeAndStore(nextBase, nextLen);
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void ComputeAndStore(uint64_t baseOutIdx, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> rowBuf = tmpQueue_.AllocTensor<float>();

        for (uint32_t i = 0; i < len; ++i) {
            uint64_t outIdx = baseOutIdx + i;

            // Unflatten to (n, c, do, ho, wo)
            uint64_t tmp = outIdx;
            uint32_t wo = static_cast<uint32_t>(tmp % Wo_); tmp /= Wo_;
            uint32_t ho = static_cast<uint32_t>(tmp % Ho_); tmp /= Ho_;
            uint32_t doo = static_cast<uint32_t>(tmp % Do_); tmp /= Do_;
            uint32_t c   = static_cast<uint32_t>(tmp % C_);  tmp /= C_;
            uint32_t n   = static_cast<uint32_t>(tmp);

            // input starting positions with padding
            int32_t dStart = static_cast<int32_t>(doo) * static_cast<int32_t>(Sd_) - static_cast<int32_t>(Pd_);
            int32_t hStart = static_cast<int32_t>(ho)  * static_cast<int32_t>(Sh_) - static_cast<int32_t>(Ph_);
            int32_t wStart = static_cast<int32_t>(wo)  * static_cast<int32_t>(Sw_) - static_cast<int32_t>(Pw_);

            float acc = 0.0f;
            uint32_t validCnt = 0u;

            for (uint32_t kd = 0; kd < Kd_; ++kd) {
                int32_t d = dStart + static_cast<int32_t>(kd);
                if (d < 0 || d >= static_cast<int32_t>(Di_)) {
                    if (countIncludePad_) {
                        // include pad in count
                        validCnt += 0u;
                    }
                    continue;
                }
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    int32_t h = hStart + static_cast<int32_t>(kh);
                    if (h < 0 || h >= static_cast<int32_t>(Hi_)) {
                        if (countIncludePad_) {
                            // include pad in count
                            validCnt += 0u;
                        }
                        continue;
                    }

                    // Determine contiguous w-range within [0, Wi_)
                    int32_t w0 = wStart;
                    int32_t w1 = wStart + static_cast<int32_t>(Kw_);
                    int32_t cw0 = (w0 < 0) ? 0 : w0;
                    int32_t cw1 = (w1 > static_cast<int32_t>(Wi_)) ? static_cast<int32_t>(Wi_) : w1;
                    uint32_t runLen = (cw1 > cw0) ? static_cast<uint32_t>(cw1 - cw0) : 0u;

                    // Base GM offset for this (n,c,d,h) plane
                    uint64_t baseOff = (((static_cast<uint64_t>(n) * C_ + c) * Di_ + static_cast<uint32_t>(d)) * Hi_
                                       + static_cast<uint32_t>(h)) * Wi_;

                    if (runLen > 0) {
                        // copy contiguous segment along W
                        AscendC::DataCopy(rowBuf, xGm_[baseOff + static_cast<uint32_t>(cw0)], runLen);
                        // Accumulate
                        for (uint32_t t = 0; t < runLen; ++t) {
                            float val = rowBuf.GetValue(t);
                            acc += val;
                        }
                        validCnt += runLen;
                    }
                    // If including pad in count
                    if (countIncludePad_) {
                        uint32_t padBefore = (w0 < 0) ? static_cast<uint32_t>(-w0) : 0u;
                        uint32_t padAfter = (w1 > static_cast<int32_t>(Wi_)) ? static_cast<uint32_t>(w1 - static_cast<int32_t>(Wi_)) : 0u;
                        validCnt += padBefore + padAfter;
                    }
                }
                if (countIncludePad_) {
                    // add vertical pads for h outside range: handled implicitly by not adding to acc but
                    // we need to include them in count. Here we already did per-kh inside bounds;
                    // To count pads in H we need to add (total Kh - valid Hs) * Kw for each valid d.
                    // For simplicity, we compute exact count by formula when include pad:
                    // But since default is false, this path is rarely used.
                }
            }

            if (countIncludePad_) {
                // When include pad, denominator is Kd*Kh*Kw always
                validCnt = Kd_ * Kh_ * Kw_;
            }

            float outVal = 0.0f;
            if (validCnt > 0) {
                outVal = acc / static_cast<float>(validCnt);
            }
            yLocal.SetValue(i, outVal);
        }

        outQueueY_.EnQue<float>(yLocal);
        AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[baseOutIdx], yOut, len); // Note: baseOutIdx needs to be passed; workaround below
        outQueueY_.FreeTensor(yOut);

        // Free tmp row buffer
        tmpQueue_.FreeTensor(rowBuf);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0};
    uint32_t Di_{0}, Hi_{0}, Wi_{0};
    uint32_t Do_{0}, Ho_{0}, Wo_{0};
    uint32_t Kd_{1}, Kh_{1}, Kw_{1};
    uint32_t Sd_{1}, Sh_{1}, Sw_{1};
    uint32_t Pd_{0}, Ph_{0}, Pw_{0};
    bool countIncludePad_{false};

    uint64_t totalOut_{0};
    uint32_t tileLen_{0};
    uint32_t spanW_{1};
};

// Wrapper compute with proper copy-out per tile
class KernelAveragePooling3dCustomRunner : public KernelAveragePooling3dCustom {
public:
    __aicore__ inline KernelAveragePooling3dCustomRunner() : KernelAveragePooling3dCustom() {}

    __aicore__ inline void RunTiles()
    {
        if (totalOut_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * tileLen_;
        uint64_t base = static_cast<uint64_t>(blockIdx) * tileLen_;
        if (base >= totalOut_) return;

        do {
            uint32_t curLen = static_cast<uint32_t>(totalOut_ - base);
            if (curLen > tileLen_) curLen = tileLen_;

            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            AscendC::LocalTensor<float> rowBuf = tmpQueue_.AllocTensor<float>();

            for (uint32_t i = 0; i < curLen; ++i) {
                uint64_t outIdx = base + i;

                uint64_t tmp = outIdx;
                uint32_t wo = static_cast<uint32_t>(tmp % Wo_); tmp /= Wo_;
                uint32_t ho = static_cast<uint32_t>(tmp % Ho_); tmp /= Ho_;
                uint32_t doo = static_cast<uint32_t>(tmp % Do_); tmp /= Do_;
                uint32_t c   = static_cast<uint32_t>(tmp % C_);  tmp /= C_;
                uint32_t n   = static_cast<uint32_t>(tmp);

                int32_t dStart = static_cast<int32_t>(doo) * static_cast<int32_t>(Sd_) - static_cast<int32_t>(Pd_);
                int32_t hStart = static_cast<int32_t>(ho)  * static_cast<int32_t>(Sh_) - static_cast<int32_t>(Ph_);
                int32_t wStart = static_cast<int32_t>(wo)  * static_cast<int32_t>(Sw_) - static_cast<int32_t>(Pw_);

                float acc = 0.0f;
                uint32_t validCnt = 0u;

                for (uint32_t kd = 0; kd < Kd_; ++kd) {
                    int32_t d = dStart + static_cast<int32_t>(kd);
                    bool dValid = (d >= 0 && d < static_cast<int32_t>(Di_));
                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        int32_t h = hStart + static_cast<int32_t>(kh);
                        bool hValid = (h >= 0 && h < static_cast<int32_t>(Hi_));

                        int32_t w0 = wStart;
                        int32_t w1 = wStart + static_cast<int32_t>(Kw_);
                        int32_t cw0 = (w0 < 0) ? 0 : w0;
                        int32_t cw1 = (w1 > static_cast<int32_t>(Wi_)) ? static_cast<int32_t>(Wi_) : w1;
                        uint32_t runLen = (cw1 > cw0) ? static_cast<uint32_t>(cw1 - cw0) : 0u;

                        if (dValid && hValid && runLen > 0) {
                            uint64_t baseOff = (((static_cast<uint64_t>(n) * C_ + c) * Di_ + static_cast<uint32_t>(d)) * Hi_
                                               + static_cast<uint32_t>(h)) * Wi_;
                            AscendC::DataCopy(rowBuf, xGm_[baseOff + static_cast<uint32_t>(cw0)], runLen);
                            for (uint32_t t = 0; t < runLen; ++t) {
                                float val = rowBuf.GetValue(t);
                                acc += val;
                            }
                            validCnt += runLen;
                        }

                        if (countIncludePad_) {
                            // Include pad elements in count
                            uint32_t padWBefore = (w0 < 0) ? static_cast<uint32_t>(-w0) : 0u;
                            uint32_t padWAfter = (w1 > static_cast<int32_t>(Wi_)) ? static_cast<uint32_t>(w1 - static_cast<int32_t>(Wi_)) : 0u;
                            if (dValid && hValid) validCnt += padWBefore + padWAfter;
                            else validCnt += Kw_;
                        }
                    }
                    if (countIncludePad_ && !dValid) {
                        validCnt += Kh_ * Kw_;
                    }
                }

                if (!countIncludePad_) {
                    // validCnt already counts only in-bounds samples
                } else {
                    // If include pad, denominator should be full kernel size
                    validCnt = Kd_ * Kh_ * Kw_;
                }

                float outVal = (validCnt > 0) ? (acc / static_cast<float>(validCnt)) : 0.0f;
                yLocal.SetValue(i, outVal);
            }

            outQueueY_.EnQue<float>(yLocal);
            AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[base], yOut, curLen);
            outQueueY_.FreeTensor(yOut);
            tmpQueue_.FreeTensor(rowBuf);

            base += strideElems;
        } while (base < totalOut_);
    }

protected:
    using KernelAveragePooling3dCustom::xGm_;
    using KernelAveragePooling3dCustom::yGm_;
    using KernelAveragePooling3dCustom::outQueueY_;
    using KernelAveragePooling3dCustom::tmpQueue_;
    using KernelAveragePooling3dCustom::N_;
    using KernelAveragePooling3dCustom::C_;
    using KernelAveragePooling3dCustom::Di_;
    using KernelAveragePooling3dCustom::Hi_;
    using KernelAveragePooling3dCustom::Wi_;
    using KernelAveragePooling3dCustom::Do_;
    using KernelAveragePooling3dCustom::Ho_;
    using KernelAveragePooling3dCustom::Wo_;
    using KernelAveragePooling3dCustom::Kd_;
    using KernelAveragePooling3dCustom::Kh_;
    using KernelAveragePooling3dCustom::Kw_;
    using KernelAveragePooling3dCustom::Sd_;
    using KernelAveragePooling3dCustom::Sh_;
    using KernelAveragePooling3dCustom::Sw_;
    using KernelAveragePooling3dCustom::Pd_;
    using KernelAveragePooling3dCustom::Ph_;
    using KernelAveragePooling3dCustom::Pw_;
    using KernelAveragePooling3dCustom::countIncludePad_;
    using KernelAveragePooling3dCustom::totalOut_;
    using KernelAveragePooling3dCustom::tileLen_;
    using KernelAveragePooling3dCustom::spanW_;
};

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling3dCustomRunner op;
