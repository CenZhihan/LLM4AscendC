#include "max_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tileRows
//   maxBuffer: 1 * tileRows
// Total floats = 3 * tileRows; budget below assumes headroom and potential expansions.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaxReductionOverADimensionCustomTilingData tiling;

    const auto *in_shape = context->GetInputShape(0);
    const auto &orig = in_shape->GetOriginShape();
    const uint32_t rank = static_cast<uint32_t>(orig.GetDimNum());

    uint64_t totalLen64 = 1;
    for (uint32_t i = 0; i < rank; ++i) {
        totalLen64 *= static_cast<uint64_t>(orig.GetDim(i));
    }
    const uint32_t totalLength = static_cast<uint32_t>(totalLen64);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t axis = (dimAttr != nullptr) ? *dimAttr : 0;
    if (rank == 0) {
        return ge::GRAPH_FAILED;
    }
    if (axis < 0) {
        axis += static_cast<int32_t>(rank);
    }
    if (axis < 0) axis = 0;
    if (axis >= static_cast<int32_t>(rank)) axis = static_cast<int32_t>(rank) - 1;

    // Compute outer, reduce, inner
    uint32_t outer = 1;
    for (int32_t i = 0; i < axis; ++i) {
        outer *= static_cast<uint32_t>(orig.GetDim(i));
    }
    uint32_t reduce = static_cast<uint32_t>(orig.GetDim(axis));
    uint32_t inner = 1;
    for (uint32_t i = axis + 1; i < rank; ++i) {
        inner *= static_cast<uint32_t>(orig.GetDim(i));
    }

    // tileRows from UB budget. Budget assumes 5 * tileRows floats ~ 20 * tileRows bytes in a more general pipeline.
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileRows = maxTile;
    if (inner < tileRows) {
        uint32_t aligned = AlignDown(inner, ALIGN_ELEMS);
        tileRows = (aligned > 0) ? aligned : inner;
    }
    if (tileRows == 0) tileRows = 1;

    const uint32_t chunksPerOuter = CeilDiv(inner, tileRows);
    uint32_t workItems = outer * chunksPerOuter;

    uint32_t blockDim = workItems;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_outer(outer);
    tiling.set_reduce(reduce);
    tiling.set_inner(inner);
    tiling.set_tileRows(tileRows);
    tiling.set_chunksPerOuter(chunksPerOuter);
    tiling.set_workItems(workItems);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const auto &orig = x_shape->GetOriginShape();

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUF_NUM = 2;

class KernelMaxReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMaxReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t reduce, uint32_t inner,
                                uint32_t tileRows, uint32_t chunksPerOuter, uint32_t workItems)
    {
        outer_ = outer;
        reduce_ = reduce;
        inner_ = inner;
        tileRows_ = tileRows;
        chunksPerOuter_ = chunksPerOuter;
        workItems_ = workItems;

        const uint32_t xTotal = outer_ * reduce_ * inner_;
        const uint32_t yTotal = outer_ * inner_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xTotal);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yTotal);

        pipe_.InitBuffer(inQueueX_, IN_BUF_NUM, tileRows_ * sizeof(float));
        pipe_.InitBuffer(tmpQueueMax_, 1, tileRows_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (reduce_ == 0 || inner_ == 0 || tileRows_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t workIdx = blockIdx; workIdx < workItems_; workIdx += blockNum) {
            const uint32_t outerIndex = workIdx / chunksPerOuter_;
            const uint32_t chunkIndex = workIdx - outerIndex * chunksPerOuter_;
            const uint32_t innerStart = chunkIndex * tileRows_;
            if (innerStart >= inner_) continue;
            uint32_t curRows = inner_ - innerStart;
            if (curRows > tileRows_) curRows = tileRows_;

            // local max buffer
            AscendC::LocalTensor<float> maxLocal = tmpQueueMax_.AllocTensor<float>();

            // Load first slice (k=0) into maxLocal
            {
                const uint32_t k = 0;
                const uint32_t base = outerIndex * (reduce_ * inner_) + k * inner_ + innerStart;
                AscendC::DataCopy(maxLocal, xGm_[base], curRows);
            }

            // Iterate over remaining k to reduce
            for (uint32_t k = 1; k < reduce_; ++k) {
                const uint32_t base = outerIndex * (reduce_ * inner_) + k * inner_ + innerStart;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base], curRows);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xLocalDeq = inQueueX_.DeQue<float>();
                AscendC::Max(maxLocal, maxLocal, xLocalDeq, curRows);
                inQueueX_.FreeTensor(xLocalDeq);
            }

            // Write result to y
            const uint32_t yBase = outerIndex * inner_ + innerStart;
            AscendC::DataCopy(yGm_[yBase], maxLocal, curRows);

            tmpQueueMax_.FreeTensor(maxLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUF_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueueMax_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_ {0};
    uint32_t reduce_ {0};
    uint32_t inner_ {0};
    uint32_t tileRows_ {0};
    uint32_t chunksPerOuter_ {0};
    uint32_t workItems_ {0};
};

extern "C" __global__ __aicore__ void max_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaxReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.outer, tiling_data.reduce, tiling_data.inner,
            tiling_data.tileRows, tiling_data.chunksPerOuter, tiling_data.workItems);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline std::vector<int64_t> compute_reduced_sizes(const at::Tensor& self, int64_t dim) {
    int64_t rank = self.dim();
    int64_t axis = dim;
    if (axis < 0) axis += rank;
    if (axis < 0) axis = 0;
    if (axis >= rank) axis = rank - 1;
    std::vector<int64_t> out_sizes;
    out_sizes.reserve(rank > 0 ? rank - 1 : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == axis) continue;
        out_sizes.push_back(self.size(i));
    }
    return out_sizes;
}

at::Tensor max_reduce_dim_impl_npu(const at::Tensor& self, int64_t dim) {
    auto out_sizes = compute_reduced_sizes(self, dim);
    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnMaxReductionOverADimensionCustom, self, static_cast<int>(dim), result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("max_reduction_over_a_dimension_custom", &max_reduce_dim_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("max_reduction_over_a_dimension_custom", &max_reduce_dim_impl_npu, "Max reduction over a dimension (MaxReductionOverADimensionCustom on NPU)");
