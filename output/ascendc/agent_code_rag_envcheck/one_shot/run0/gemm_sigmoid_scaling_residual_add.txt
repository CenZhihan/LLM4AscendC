#include "gemm_sigmoid_scaling_residual_add_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

// UB model (floats):
//   xSeg: 1 * tileK
//   wSeg: 1 * tileK
//   yAcc: 1 * tileN
//   tmp:  1 * tileN
// Total floats = 2 * tileK + 2 * tileN => bytes = 4 * (2*tileK + 2*tileN)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_K = 256;
static constexpr uint32_t ALIGN_N = 128;
static constexpr uint32_t HARD_CAP_TILEK = 4096;
static constexpr uint32_t HARD_CAP_TILEN = 1024;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmSigmoidScalingResidualAddCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);
    const gert::Shape *s_shape = context->GetInputShape(3);

    // Expect shapes: x [M, K], w [N, K], b [N], scaling [1]
    const uint32_t m = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t k = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t n = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t wk = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));

    if (wk != k) {
        return ge::GRAPH_FAILED;
    }
    if (static_cast<uint32_t>(b_shape->GetOriginShape().GetShapeSize()) != n) {
        return ge::GRAPH_FAILED;
    }
    if (static_cast<uint32_t>(s_shape->GetOriginShape().GetShapeSize()) != 1u) {
        return ge::GRAPH_FAILED;
    }

    // Choose tileK first
    uint32_t tileK = HARD_CAP_TILEK;
    if (tileK > k) tileK = k;
    tileK = AlignDown(tileK, ALIGN_K);
    if (tileK == 0) tileK = AlignDown(k, ALIGN_K);
    if (tileK == 0) tileK = k; // fallback when k < ALIGN_K

    // Compute tileN from UB budget
    const uint32_t budgetFloats = UB_BUDGET_BYTES / 4;
    uint32_t maxTileN = 0;
    if (budgetFloats > (2u * tileK)) {
        maxTileN = (budgetFloats - 2u * tileK) / 2u;
    }
    if (maxTileN == 0) maxTileN = ALIGN_N;
    if (maxTileN > HARD_CAP_TILEN) maxTileN = HARD_CAP_TILEN;
    uint32_t tileN = maxTileN;
    if (tileN > n) tileN = n;
    tileN = AlignDown(tileN, ALIGN_N);
    if (tileN == 0) tileN = (n >= ALIGN_N ? ALIGN_N : n);

    // BlockDim: grid-stride across rows
    uint32_t numRowTiles = (m == 0) ? 1u : m;
    uint32_t blockDim = numRowTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_m(m);
    tiling.set_n(n);
    tiling.set_k(k);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelGemmSigmoidScalingResidualAddCustom {
public:
    __aicore__ inline KernelGemmSigmoidScalingResidualAddCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling, GM_ADDR y,
                                uint32_t m, uint32_t n, uint32_t k,
                                uint32_t tileK, uint32_t tileN)
    {
        m_ = m;
        n_ = n;
        k_ = k;
        tileK_ = tileK;
        tileN_ = tileN;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(m_) * static_cast<uint64_t>(k_));
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(n_) * static_cast<uint64_t>(k_));
        bGm_.SetGlobalBuffer((__gm__ float*)b, n_);
        sGm_.SetGlobalBuffer((__gm__ float*)scaling, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(m_) * static_cast<uint64_t>(n_));

        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (m_ == 0 || n_ == 0 || k_ == 0) return;

        // Load scaling scalar once per core
        float s = 1.0f;
        {
            AscendC::LocalTensor<float> sLocal = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(sLocal, sGm_[0], 1);
            s = sLocal.GetValue(0);
            tmpQueue_.FreeTensor(sLocal);
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < m_; row += blockNum) {
            const uint64_t xRowBase = static_cast<uint64_t>(row) * static_cast<uint64_t>(k_);
            const uint64_t yRowBase = static_cast<uint64_t>(row) * static_cast<uint64_t>(n_);

            for (uint32_t colBase = 0; colBase < n_; colBase += tileN_) {
                uint32_t lenN = n_ - colBase;
                if (lenN > tileN_) lenN = tileN_;

                // yAcc initialized with bias
                AscendC::LocalTensor<float> yAcc = yQueue_.AllocTensor<float>();
                AscendC::DataCopy(yAcc, bGm_[colBase], lenN);

                // Accumulate dot products across K dimension
                for (uint32_t kBase = 0; kBase < k_; kBase += tileK_) {
                    uint32_t lenK = k_ - kBase;
                    if (lenK > tileK_) lenK = tileK_;

                    // Load x segment
                    AscendC::LocalTensor<float> xSeg = xQueue_.AllocTensor<float>();
                    AscendC::DataCopy(xSeg, xGm_[xRowBase + kBase], lenK);

                    // For each output col in tile, load w segment and accumulate
                    for (uint32_t j = 0; j < lenN; ++j) {
                        AscendC::LocalTensor<float> wSeg = wQueue_.AllocTensor<float>();
                        AscendC::DataCopy(wSeg, wGm_[static_cast<uint64_t>(colBase + j) * static_cast<uint64_t>(k_) + kBase], lenK);

                        // Compute partial dot product
                        float acc = 0.0f;
                        for (uint32_t t = 0; t < lenK; ++t) {
                            float xv = xSeg.GetValue(t);
                            float wv = wSeg.GetValue(t);
                            acc += xv * wv;
                        }

                        // yAcc[j] += acc
                        float yprev = yAcc.GetValue(j);
                        yAcc.SetValue(j, yprev + acc);

                        wQueue_.FreeTensor(wSeg);
                    }

                    xQueue_.FreeTensor(xSeg);
                }

                // Fused: sigmoid(yAcc) * s + yAcc
                AscendC::LocalTensor<float> sigLocal = tmpQueue_.AllocTensor<float>();
                AscendC::Sigmoid(sigLocal, yAcc, lenN);      // sigLocal = sigmoid(yAcc)
                AscendC::Muls(sigLocal, sigLocal, s, lenN);  // sigLocal *= s
                AscendC::Add(sigLocal, sigLocal, yAcc, lenN);

                // Write back
                AscendC::DataCopy(yGm_[yRowBase + colBase], sigLocal, lenN);

                tmpQueue_.FreeTensor(sigLocal);
                yQueue_.FreeTensor(yAcc);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> sGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t m_ {0};
    uint32_t n_ {0};
    uint32_t k_ {0};
    uint32_t tileK_ {0};
    uint32_t tileN_ {0};
};

extern "C" __global__ __aicore__ void gemm_sigmoid_scaling_residual_add_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmSigmoidScalingResidualAddCustom op;
    op.Init(x, w, b, scaling, y,
            tiling_data.m, tiling_data.n, tiling_data.k,
            tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
