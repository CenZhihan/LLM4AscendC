#include "register/op_def_registry.h"
#include "matmul_mish_mish_custom_tiling.h"

namespace {
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative UB
static constexpr uint32_t ALIGN_ELEMS = 2048;
static constexpr uint32_t HARD_CAP_ELEMS = 32768;

static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMishMishCustomTilingData tiling;

    // Shapes: x:[M,K], weight:[N,K], bias:[N], y:[M,N]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() < 2 || w_shape->GetOriginShape().GetDimNum() < 2) {
        return ge::GRAPH_FAILED;
    }
    const int64_t M = x_shape->GetOriginShape().GetDim(x_shape->GetOriginShape().GetDimNum() - 2);
    const int64_t Kx = x_shape->GetOriginShape().GetDim(x_shape->GetOriginShape().GetDimNum() - 1);
    const int64_t N  = w_shape->GetOriginShape().GetDim(0);
    const int64_t Kw = w_shape->GetOriginShape().GetDim(1);
    if (Kx != Kw || M <= 0 || N <= 0 || Kx <= 0) {
        return ge::GRAPH_FAILED;
    }

    // Let built-in matmul tiler fill cubeTilingData by framework; here we only decide postprocess tiling.
    const uint32_t totalLength = static_cast<uint32_t>(M * N);

    // UB model for postprocess Mish twice:
    // y_in: 2 * tile, y_out: 2 * tile, tmp1: 1 * tile, tmp2: 1 * tile, tmp3: 1 * tile => ~6 * tile
    // Use 24 bytes/float-window for safety, but stick to 6 * tile floats => bytes = 24 * tile
    uint32_t maxTile = UB_BUDGET_BYTES / 24;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (totalLength < tileLen) {
        const uint32_t aligned = AlignDown(totalLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : totalLength;
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDiv(totalLength, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    // Save tiling data
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace required
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetOriginShape().GetDimNum() < 2 || w_shape->GetOriginShape().GetDimNum() < 2) {
        return GRAPH_FAILED;
    }
    gert::Shape *y_shape = context->GetOutputShape(0);
    auto x_origin = x_shape->GetOriginShape();
    auto w_origin = w_shape->GetOriginShape();
    int64_t M = x_origin.GetDim(x_origin.GetDimNum() - 2);
    int64_t N = w_origin.GetDim(0);
    std::vector<int64_t> out_dims = {M, N};
    y_shape->SetOriginShape(out_dims);
    y_shape->SetStorageShape(out_dims);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    const ge::DataType w_dtype = context->GetInputDataType(1);
    const ge::DataType b_dtype = context->GetInputDataType(2);

        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulMishMishCustom {
public:
    __aicore__ inline KernelMatmulMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace,
                                const TCubeTiling &cubeTiling,
                                uint32_t totalLength, uint32_t tileLength)
    {
        // Matmul init
        REGIST_MATMUL_OBJ(&pipe_, GetSysWorkSpacePtr(), matmul_.matmulObj, &cubeTiling);
        matmul_.Init(x, w, b, y, workspace, cubeTiling);

        // Post-process tiling init
        totalLen_ = totalLength;
        tileLen_ = tileLength;

        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipeVec_.InitBuffer(inQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipeVec_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipeVec_.InitBuffer(tmpQueue_, 3, tileLen_ * sizeof(float)); // for exp/add/log/tanh temp usage
    }

    __aicore__ inline void Process()
    {
        // Phase 1: GEMM with bias
        matmul_.Process<true>(&pipe_);

        // Phase 2: Apply Mish twice on Y in-place
        if (totalLen_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t strideElems = blockNum * tileLen_;
        uint32_t base = blockIdx * tileLen_;
        if (base >= totalLen_) return;

        uint32_t curLen = totalLen_ - base;
        if (curLen > tileLen_) curLen = tileLen_;
        CopyIn(base, curLen);

        while (true) {
            const uint32_t nextBase = base + strideElems;
            uint32_t nextLen = 0;
            if (nextBase < totalLen_) {
                nextLen = totalLen_ - nextBase;
                if (nextLen > tileLen_) nextLen = tileLen_;
                CopyIn(nextBase, nextLen);
            }

            ComputeMishTwice(curLen);
            CopyOut(base, curLen);

            if (nextLen == 0) break;
            base = nextBase;
            curLen = nextLen;
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.AllocTensor<float>();
        AscendC::DataCopy(yLocal, yGm_[gmOffset], len);
        inQueueY_.EnQue(yLocal);
    }

    __aicore__ inline void ComputeMishTwice(uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = inQueueY_.DeQue<float>();
        AscendC::LocalTensor<float> outLocal = outQueueY_.AllocTensor<float>();

        AscendC::LocalTensor<float> t1 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> t2 = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> t3 = tmpQueue_.AllocTensor<float>();

        // First Mish: y1 = y * tanh(ln(1 + exp(y)))
        AscendC::Exp(t1, yLocal, len);                 // t1 = exp(y)
        AscendC::Adds(t2, t1, 1.0f, len);              // t2 = 1 + exp(y)
        AscendC::Ln(t2, t2, len);                      // t2 = ln(1 + exp(y))
        AscendC::Tanh(t3, t2, len);                    // t3 = tanh(softplus(y))
        AscendC::Mul(outLocal, yLocal, t3, len);       // out = y * tanh(softplus(y))

        // Second Mish: y2 = out * tanh(ln(1 + exp(out)))
        AscendC::Exp(t1, outLocal, len);               // t1 = exp(out)
        AscendC::Adds(t2, t1, 1.0f, len);              // t2 = 1 + exp(out)
        AscendC::Ln(t2, t2, len);                      // t2 = ln(1 + exp(out))
        AscendC::Tanh(t3, t2, len);                    // t3 = tanh(softplus(out))
        AscendC::Mul(outLocal, outLocal, t3, len);     // out = out * tanh(softplus(out))

        tmpQueue_.FreeTensor(t3);
        tmpQueue_.FreeTensor(t2);
        tmpQueue_.FreeTensor(t1);

        outQueueY_.EnQue<float>(outLocal);
        inQueueY_.FreeTensor(yLocal);
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> outLocal = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], outLocal, len);
        outQueueY_.FreeTensor(outLocal);
    }

private:
    // Matmul engine and pipe
    MatmulKernel<float, float, float, float> matmul_;
    AscendC::TPipe pipe_;

    // Post-process vector pipeline
    AscendC::TPipe pipeVec_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 3> tmpQueue_;

    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void matmul_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelMatmulMishMishCustom op;
    op.Init(x, weight, bias, y, workspace, tilingData.cubeTilingData, tilingData.totalLength, tilingData.tileLength);
    op.Process();
}
