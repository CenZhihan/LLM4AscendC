static inline uint32_t CeilDiv(uint64_t a, uint32_t b)
{
    return static_cast<uint32_t>((a + b - 1ULL) / b);
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMinAddMultiplyCustomTilingData tiling;

    // Input 0: x [N, Cin, H, W]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape == nullptr || w_shape == nullptr || x_shape->GetOriginShape().GetDimNum() != 4 ||
        w_shape->GetOriginShape().GetDimNum() != 4) {
        return ge::GRAPH_FAILED;
    }

    const auto &x_dims = x_shape->GetOriginShape().GetDims();
    const auto &w_dims = w_shape->GetOriginShape().GetDims();

    const uint32_t N  = static_cast<uint32_t>(x_dims[0]);
    const uint32_t Cin = static_cast<uint32_t>(x_dims[1]);
    const uint32_t H  = static_cast<uint32_t>(x_dims[2]);
    const uint32_t W  = static_cast<uint32_t>(x_dims[3]);

    const uint32_t Cout = static_cast<uint32_t>(w_dims[0]);
    const uint32_t CinW = static_cast<uint32_t>(w_dims[1]);
    const uint32_t Kh = static_cast<uint32_t>(w_dims[2]);
    const uint32_t Kw = static_cast<uint32_t>(w_dims[3]);

    if (CinW != Cin) {
        return ge::GRAPH_FAILED;
    }

    // Fixed stride=1, padding=0, dilation=1
    if (H < Kh || W < Kw) {
        return ge::GRAPH_FAILED;
    }
    const uint32_t Ho = H - Kh + 1;
    const uint32_t Wo = W - Kw + 1;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *constValPtr = attrs->GetAttrPointer<float>(0);
    const float *scaleValPtr = attrs->GetAttrPointer<float>(1);
    const float constantValue = (constValPtr != nullptr) ? *constValPtr : 0.0f;
    const float scalingFactor = (scaleValPtr != nullptr) ? *scaleValPtr : 1.0f;

    bool hasBias = false;
    // Optional bias at input index 2
#ifdef __GERT_TILING_CONTEXT_SUPPORT_OPTIONAL__
    hasBias = !context->IsOptionalInputNull(2);
#else
    // Fallback heuristic
    const gert::Shape *b_shape = context->GetInputShape(2);
    hasBias = (b_shape != nullptr && b_shape->GetOriginShape().GetShapeSize() > 0);
#endif

    // Choose block dim based on total output elements
    const uint64_t totalOut = static_cast<uint64_t>(N) * Cout * Ho * Wo;
    uint32_t blockDim = CeilDiv(totalOut, 16384u);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim > totalOut) blockDim = static_cast<uint32_t>(totalOut > 0 ? totalOut : 1u);
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1u : 0u);
    tiling.set_constantValue(constantValue);
    tiling.set_scalingFactor(scalingFactor);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),

#include "kernel_operator.h"

class KernelConv2dMinAddMultiplyCustom {
public:
    __aicore__ inline KernelConv2dMinAddMultiplyCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR conv_bias, GM_ADDR post_bias, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo, bool hasBias,
                                float constantValue, float scalingFactor)
    {
        xGm_ = (__gm__ const float*)x;
        wGm_ = (__gm__ const float*)w;
        bGm_ = (__gm__ const float*)conv_bias;
        pbGm_ = (__gm__ const float*)post_bias;
        yGm_ = (__gm__ float*)y;

        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;
        hasBias_ = hasBias;
        cval_ = constantValue;
        sfac_ = scalingFactor;
    }

    __aicore__ inline void Process()
    {
        const uint64_t totalOut = static_cast<uint64_t>(N_) * Cout_ * Ho_ * Wo_;
        if (totalOut == 0) {
            return;
        }
        const uint32_t blockNum = AscendC::GetBlockNum();
        const uint32_t blockIdx = AscendC::GetBlockIdx();

        for (uint64_t idx = blockIdx; idx < totalOut; idx += blockNum) {
            // Unravel index: [N, Cout, Ho, Wo]
            uint64_t t = idx;
            const uint32_t WoDim = Wo_;
            const uint32_t HoDim = Ho_;
            const uint32_t CoDim = Cout_;

            const uint32_t co_spatial = static_cast<uint32_t>(t % (static_cast<uint64_t>(HoDim) * WoDim));
            const uint32_t ho = co_spatial / WoDim;
            const uint32_t wo = co_spatial % WoDim;
            t /= (static_cast<uint64_t>(HoDim) * WoDim);
            const uint32_t co = static_cast<uint32_t>(t % CoDim);
            t /= CoDim;
            const uint32_t n = static_cast<uint32_t>(t);

            float acc = 0.0f;
            // Convolution: stride=1, pad=0, dilation=1
            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    const uint32_t ih = ho + kh;
                    const uint32_t baseX1 = (((n * Cin_ + ci) * H_) + ih) * W_;
                    const uint32_t baseW1 = (((co * Cin_ + ci) * Kh_) + kh) * Kw_;
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const uint32_t iw = wo + kw;

                        const uint64_t xIdx = static_cast<uint64_t>(baseX1 + iw);
                        const uint64_t wIdx = static_cast<uint64_t>(baseW1 + kw);

                        float xv = xGm_[xIdx];
                        float wv = wGm_[wIdx];
                        acc = AscendC::Fadd(acc, AscendC::Fmul(xv, wv));
                    }
                }
            }

            if (hasBias_) {
                acc = AscendC::Fadd(acc, bGm_[co]);
            }

            // y = min(acc, cval)
            acc = (acc < cval_) ? acc : cval_;
            // + post_bias[Cout,1,1]
            acc = AscendC::Fadd(acc, pbGm_[co]);
            // * scaling factor
            acc = AscendC::Fmul(acc, sfac_);

            const uint64_t yIdx = ((((uint64_t)n * Cout_) + co) * Ho_ + ho) * Wo_ + wo;
            yGm_[yIdx] = acc;
        }
    }

private:
    __gm__ const float* xGm_ {nullptr};
    __gm__ const float* wGm_ {nullptr};
    __gm__ const float* bGm_ {nullptr};
    __gm__ const float* pbGm_ {nullptr};
    __gm__ float* yGm_ {nullptr};

    uint32_t N_ {0}, Cin_ {0}, Cout_ {0};
    uint32_t H_ {0}, W_ {0}, Kh_ {0}, Kw_ {0};
    uint32_t Ho_ {0}, Wo_ {0};
    bool hasBias_ {false};
    float cval_ {0.0f};
    float sfac_ {1.0f};
};

extern "C" __global__ __aicore__ void conv2d_min_add_multiply_custom(GM_ADDR x,
                                                                      GM_ADDR weight,
                                                                      GM_ADDR conv_bias,
                                                                      GM_ADDR post_bias,
                                                                      GM_ADDR y,
                                                                      GM_ADDR workspace,
                                                                      GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinAddMultiplyCustom op;
    op.Init(x, weight, conv_bias, post_bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo,
            tiling_data.hasBias != 0u,
            tiling_data.constantValue, tiling_data.scalingFactor);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor alloc_output_like_conv2d(const at::Tensor& x,
                                                  const at::Tensor& weight)
{
    // x: [N, Cin, H, W], weight: [Cout, Cin, Kh, Kw], stride=1, pad=0, dil=1
    TORCH_CHECK(x.dim() == 4 && weight.dim() == 4, \"Invalid dims\");
    const int64_t N  = x.size(0);
    const int64_t H  = x.size(2);
    const int64_t W  = x.size(3);
    const int64_t Kh = weight.size(2);
    const int64_t Kw = weight.size(3);
    const int64_t Cout = weight.size(0);
    TORCH_CHECK(H >= Kh && W >= Kw, \"Kernel bigger than input\");
    const int64_t Ho = H - Kh + 1;
    const int64_t Wo = W - Kw + 1;
    return at::empty({N, Cout, Ho, Wo}, x.options());
}

at::Tensor conv2d_min_add_multiply_impl_npu(const at::Tensor& x,
                                            const at::Tensor& weight,
                                            const c10::optional<at::Tensor>& conv_bias_opt,
                                            const at::Tensor& post_bias,
                                            const at::Tensor& constant_value,
                                            const at::Tensor& scaling_factor)
{
    // Convert scalars
    double cv = constant_value.item<double>();
    double sv = scaling_factor.item<double>();

    at::Tensor y = alloc_output_like_conv2d(x, weight);

    if (conv_bias_opt.has_value() && conv_bias_opt.value().defined()) {
        const at::Tensor& conv_bias = conv_bias_opt.value();
        EXEC_NPU_CMD(aclnnConv2dMinAddMultiplyCustom, x, weight, conv_bias, post_bias, cv, sv, y);
    } else {
        at::Tensor empty_bias; // undefined optional
        EXEC_NPU_CMD(aclnnConv2dMinAddMultiplyCustom, x, weight, empty_bias, post_bias, cv, sv, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_min_add_multiply_custom", &conv2d_min_add_multiply_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
