host_operator_src="""
#include "softmax_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0u : (a + b - 1u) / b;
}
} // anonymous

namespace optiling {

// UB model for this kernel (per core):
//   inQueue: 2 * tile
//   outQueue: 2 * tile
//   tmpQueue: 1 * tile
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    SoftmaxCustomTilingData tiling;

    const gert::Shape *inShape = context->GetInputShape(0);
    const auto &dims = inShape->GetOriginShape().GetDims();
    const uint32_t nd = static_cast<uint32_t>(dims.size());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t axis = 1;
    if (attrs != nullptr) {
        if (const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0)) {
            axis = *dimPtr;
        }
    }
    // Normalize negative axis
    if (nd > 0) {
        if (axis < 0) axis += static_cast<int32_t>(nd);
        if (axis < 0) axis = 0;
        if (axis >= static_cast<int32_t>(nd)) axis = static_cast<int32_t>(nd) - 1;
    } else {
        axis = 0;
    }

    uint64_t totalLengthU64 = inShape->GetOriginShape().GetShapeSize();
    uint32_t totalLength = static_cast<uint32_t>(totalLengthU64);

    uint64_t axisSizeU64 = (nd == 0) ? 1ULL : static_cast<uint64_t>(dims[axis]);
    uint32_t rowLength = static_cast<uint32_t>(axisSizeU64);
    uint32_t rows = (rowLength == 0) ? 0u : static_cast<uint32_t>(totalLengthU64 / axisSizeU64);

    // tileLen from UB budget: bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    // tile length is limited by rowLength
    uint32_t tileLen = maxTile;
    if (rowLength < tileLen) {
        const uint32_t aligned = AlignDown(rowLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : rowLength;
    }
    if (tileLen == 0) tileLen = 1;

    // number of rows to schedule as blocks
    uint32_t numTilesPerRow = CeilDiv(rowLength, tileLen);
    uint32_t totalRowTasks = rows; // one task per row
    uint32_t blockDim = totalRowTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_rows(rows);
    tiling.set_rowLength(rowLength);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(axis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""
#include "kernel_operator.h"
#include <cfloat>

constexpr int32_t INBUF_NUM = 2;
constexpr int32_t OUTBUF_NUM = 2;

class KernelSoftmaxCustom {
public:
    __aicore__ inline KernelSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t rowLen, uint32_t tileLen)
    {
        rows_ = rows;
        rowLen_ = rowLen;
        tileLen_ = (tileLen == 0) ? 1u : tileLen;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(rows_) * rowLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(rows_) * rowLen_);

        pipe_.InitBuffer(inQueue_, INBUF_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUTBUF_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || rowLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block works on multiple rows in a grid-stride fashion
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            const uint64_t rowBase = static_cast<uint64_t>(row) * rowLen_;

            // Pass 1: compute row-wise max
            float rowMax = -FLT_MAX;
            for (uint32_t off = 0; off < rowLen_; off += tileLen_) {
                const uint32_t len = (off + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - off);
                CopyIn(rowBase + off, len);
                AscendC::LocalTensor<float> xLocal = inQueue_.DeQue<float>();
                float tileMax = AscendC::ReduceMax(xLocal, len);
                if (tileMax > rowMax) rowMax = tileMax;
                inQueue_.FreeTensor(xLocal);
            }

            // Pass 2: compute sum of exp(x - rowMax)
            float rowSum = 0.0f;
            for (uint32_t off = 0; off < rowLen_; off += tileLen_) {
                const uint32_t len = (off + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - off);
                CopyIn(rowBase + off, len);
                AscendC::LocalTensor<float> xLocal = inQueue_.DeQue<float>();
                AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

                // t = x - rowMax
                AscendC::Adds(tLocal, xLocal, -rowMax, len);
                // t = exp(t)
                AscendC::Exp(tLocal, tLocal, len);

                float part = AscendC::ReduceSum(tLocal, len);
                rowSum += part;

                tmpQueue_.FreeTensor(tLocal);
                inQueue_.FreeTensor(xLocal);
            }
            float invSum = 1.0f / rowSum;

            // Pass 3: compute normalized outputs
            for (uint32_t off = 0; off < rowLen_; off += tileLen_) {
                const uint32_t len = (off + tileLen_ <= rowLen_) ? tileLen_ : (rowLen_ - off);
                CopyIn(rowBase + off, len);

                AscendC::LocalTensor<float> xLocal = inQueue_.DeQue<float>();
                AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

                // y = exp(x - rowMax) * invSum
                AscendC::Adds(yLocal, xLocal, -rowMax, len);
                AscendC::Exp(yLocal, yLocal, len);
                AscendC::Muls(yLocal, yLocal, invSum, len);

                outQueue_.EnQue<float>(yLocal);
                inQueue_.FreeTensor(xLocal);

                CopyOut(rowBase + off, len);
            }
        }
    }

private:
    __aicore__ inline void CopyIn(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        inQueue_.EnQue(xLocal);
    }

    __aicore__ inline void CopyOut(uint64_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INBUF_NUM> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUTBUF_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_ {0};
    uint32_t rowLen_ {0};
    uint32_t tileLen_ {0};
};

extern "C" __global__ __aicore__ void softmax_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelSoftmaxCustom op;
    op.Init(x, y, tiling_data.rows, tiling_data.rowLength, tiling_data.tileLength);
    op.Process();
}
"""
