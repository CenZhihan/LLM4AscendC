static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dReluHardSwishCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);   // NCHW
    const gert::Shape *w_shape = context->GetInputShape(1);   // OIHW

    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H   = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W   = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t K    = static_cast<uint32_t>(w_shape->GetDim(2)); // assume square kernel

    // Hyperparams fixed: stride=1, pad=0, dil=1
    const uint32_t outH = (H >= K) ? (H - K + 1u) : 0u;
    const uint32_t outW = (W >= K) ? (W - K + 1u) : 0u;

    uint32_t tileW = outW;

    uint32_t ocTile = 1;

    // BlockDim based on number of (n, oh) tasks.
    const uint32_t tasks = (outH == 0) ? 1u : (N * outH);
    uint32_t blockDim = tasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    const bool hasBias = (context->GetInputNum() > 2) && (context->GetInputShape(2)->GetShapeSize() > 0);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_K(K);
    tiling.set_outH(outH);
    tiling.set_outW(outW);
    tiling.set_tileW(tileW);
    tiling.set_ocTile(ocTile);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1u : 0u);

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelConv2dReluHardSwishCustom {
public:
    __aicore__ inline KernelConv2dReluHardSwishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t K,
                                uint32_t outH, uint32_t outW,
                                uint32_t tileW, uint32_t ocTile,
                                uint32_t hasBias)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout; H_ = H; W_ = W; K_ = K;
        outH_ = outH; outW_ = outW;
        tileW_ = tileW; ocTile_ = ocTile;
        hasBias_ = hasBias != 0;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_*Cin_*H_*W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, Cout_*Cin_*K_*K_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_*Cout_*outH_*outW_);

        // Buffers
        // weights for ocTile (here ocTile=1 in tiling) => Cin*K*K
        pipe_.InitBuffer(wTileQ_, 1, Cin_ * K_ * K_ * sizeof(float));
        // input row slice length up to tileW_ + (K-1)
        pipe_.InitBuffer(inRowQ_, 1, (tileW_ + (K_ - 1)) * sizeof(float));
        // temp vector for shifted row and output vector
        pipe_.InitBuffer(vecQ_, 2, tileW_ * sizeof(float));
        // tiny queue for single scalar (bias or weight element)
        pipe_.InitBuffer(scalarQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outH_ == 0 || outW_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        const uint32_t tasks = N_ * outH_;
        for (uint32_t t = blockIdx; t < tasks; t += blockNum) {
            const uint32_t n  = t / outH_;
            const uint32_t oh = t - n * outH_;

            for (uint32_t oc = 0; oc < Cout_; oc += ocTile_) {
                const uint32_t curOcTile = ((oc + ocTile_) <= Cout_) ? ocTile_ : (Cout_ - oc);

                // Load weight tile for this oc tile
                // We process oc one by one to reduce UB pressure.
                for (uint32_t occ = 0; occ < curOcTile; ++occ) {
                    const uint32_t ocIdx = oc + occ;
                    LoadWeightTile(ocIdx);

                    // Optional bias
                    float biasVal = 0.0f;
                    if (hasBias_) {
                        biasVal = LoadBias(ocIdx);
                    }

                    // Process entire width in tiles
                    for (uint32_t owStart = 0; owStart < outW_; owStart += tileW_) {
                        const uint32_t curTileW = MinVal(tileW_, outW_ - owStart);

                        // yLocal initialization with bias
                        LocalTensor<float> yLocal = vecQ_.AllocTensor<float>();
                        // zero init then add bias if present
                        Muls(yLocal, yLocal, 0.0f, curTileW);
                        if (hasBias_) {
                            Adds(yLocal, yLocal, biasVal, curTileW);
                        }

                        // Accumulate convolution sum
                        for (uint32_t ic = 0; ic < Cin_; ++ic) {
                            for (uint32_t kh = 0; kh < K_; ++kh) {
                                // Load one input row slice for current (n, ic, ih, owStart..owStart+curTileW+K-2)
                                const uint32_t ih = oh + kh;
                                const uint32_t inRowLen = curTileW + (K_ - 1);
                                CopyInInputRow(n, ic, ih, owStart, inRowLen);

                                LocalTensor<float> inRow = inRowQ_.DeQue<float>();

                                for (uint32_t kw = 0; kw < K_; ++kw) {
                                    // Shifted view from inRow starting at kw, length curTileW
                                    LocalTensor<float> vecShift = vecQ_.AllocTensor<float>();
                                    DataCopy(vecShift, inRow[kw], curTileW);

                                    const uint32_t wIdx = ((ocRelBase_ + 0u) * Cin_ + ic) * (K_ * K_) + (kh * K_ + kw);
                                    // Load scalar weight from loaded wTile (which is arranged as Cin*K*K)
                                    float wScalar = wTileGet(wIdxRel(ic, kh, kw));

                                    // yLocal += wScalar * vecShift
                                    Muls(vecShift, vecShift, wScalar, curTileW);
                                    Add(yLocal, yLocal, vecShift, curTileW);

                                    vecQ_.FreeTensor(vecShift);
                                }

                                inRowQ_.FreeTensor(inRow);
                            }
                        }

                        // Apply ReLU
                        Maxs(yLocal, yLocal, 0.0f, curTileW);
                        // HardSwish: y = y * clamp((y + 3)/6, 0, 1)
                        LocalTensor<float> tLocal = vecQ_.AllocTensor<float>();
                        DataCopy(tLocal, yLocal, curTileW);            // t = y
                        Adds(tLocal, tLocal, 3.0f, curTileW);          // t = y + 3
                        Muls(tLocal, tLocal, 1.0f / 6.0f, curTileW);   // t = (y+3)/6
                        Mins(tLocal, tLocal, 1.0f, curTileW);          // min with 1
                        Maxs(tLocal, tLocal, 0.0f, curTileW);          // max with 0
                        Mul(yLocal, yLocal, tLocal, curTileW);         // y = y * t
                        vecQ_.FreeTensor(tLocal);

                        // Store results to GM
                        CopyOut(n, ocIdx, oh, owStart, yLocal, curTileW);

                        vecQ_.FreeTensor(yLocal);
                    } // owStart
                } // occ
            } // oc
        } // task loop
    }

private:
    __aicore__ inline uint32_t MinVal(uint32_t a, uint32_t b) { return (a < b) ? a : b; }

    __aicore__ inline void LoadWeightTile(uint32_t ocIdx)
    {
        // load weights of shape [Cin, K, K] for a single output channel
        const uint32_t wPerOC = Cin_ * K_ * K_;
        const uint32_t gmOff = ocIdx * wPerOC;
        LocalTensor<float> wLocal = wTileQ_.AllocTensor<float>();
        DataCopy(wLocal, wGm_[gmOff], wPerOC);
        wTileQ_.EnQue(wLocal);
        ocRelBase_ = 0; // relative oc index within tile (single oc tile)
    }

    __aicore__ inline float LoadBias(uint32_t ocIdx)
    {
        LocalTensor<float> bLocal = scalarQ_.AllocTensor<float>();
        DataCopy(bLocal, bGm_[ocIdx], 1);
        // Assume LocalTensor provides GetValue; fallback: copy to temp scalar via vector op not available, so use GetValue.
        float v = bLocal.GetValue(0);
        scalarQ_.FreeTensor(bLocal);
        return v;
    }

    __aicore__ inline float wTileGet(uint32_t relIndex)
    {
        LocalTensor<float> wLocal = wTileQ_.DeQue<float>();
        float v = wLocal.GetValue(relIndex);
        wTileQ_.EnQue(wLocal);
        return v;
    }

    __aicore__ inline uint32_t wIdxRel(uint32_t ic, uint32_t kh, uint32_t kw)
    {
        // index within wTile [Cin, K, K]
        return (ic * (K_ * K_)) + (kh * K_ + kw);
    }

    __aicore__ inline void CopyInInputRow(uint32_t n, uint32_t ic, uint32_t ih, uint32_t owStart, uint32_t len)
    {
        LocalTensor<float> inRow = inRowQ_.AllocTensor<float>();
        const uint32_t gmOff = (((n * Cin_ + ic) * H_ + ih) * W_ + owStart);
        DataCopy(inRow, xGm_[gmOff], len);
        inRowQ_.EnQue(inRow);
    }

    __aicore__ inline void CopyOut(uint32_t n, uint32_t oc, uint32_t oh, uint32_t owStart,
                                   LocalTensor<float>& yLocal, uint32_t len)
    {
        const uint32_t gmOff = (((n * Cout_ + oc) * outH_ + oh) * outW_ + owStart);
        DataCopy(yGm_[gmOff], yLocal, len);
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, 1> wTileQ_;
    TQue<TPosition::VECCALC, 1> inRowQ_;
    TQue<TPosition::VECCALC, 2> vecQ_;
    TQue<TPosition::VECCALC, 1> scalarQ_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0}, H_{0}, W_{0}, K_{0};
    uint32_t outH_{0}, outW_{0};
    uint32_t tileW_{0}, ocTile_{1};
    bool hasBias_{false};
    uint32_t ocRelBase_{0};
};

extern "C" __global__ __aicore__ void conv2d_relu_hard_swish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dReluHardSwishCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.H, tiling_data.W, tiling_data.K,
            tiling_data.outH, tiling_data.outW,
            tiling_data.tileW, tiling_data.ocTile,
            tiling_data.hasBias);
    op.Process();
}
