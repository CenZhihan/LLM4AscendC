#include "conv_standard_2d_square_input_square_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// UB model (float32):
//   outAcc: co_tile * tile_w
//   tmpVec: tile_w
//   rows:   3 * (tile_w + 2)
// Total floats = co_tile*tile_w + 4*tile_w + 6
// Bytes ~= 4 * (co_tile + 4) * tile_w + 24
// We target safe UB usage under ~192 KB.
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS     = 128;
static constexpr uint32_t MAX_BLOCKS      = 32;
static constexpr uint32_t MAX_CO_TILE     = 16; // up to 16 out channels per-core tile
static constexpr uint32_t MIN_TILE_W      = 64;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dSquareInputSquareKernelCustomTilingData tiling;

    // Shapes
    const auto *x_shape = context->GetInputShape(0);
    const auto *w_shape = context->GetInputShape(1);

    // Expect NCHW
    const uint32_t N  = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Ci = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W  = static_cast<uint32_t>(x_shape->GetDim(3));

    // weight [Co, Ci, 3, 3]
    const uint32_t Co = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Kh = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetDim(3));

    // Fixed constraints for this kernel: stride=1,pad=0,dilation=1, Kh=Kw=3
    if (Kh != 3 || Kw != 3) {
        return ge::GRAPH_FAILED;
    }
    if (H < Kh || W < Kw) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t outH = H - Kh + 1;
    const uint32_t outW = W - Kw + 1;

    // Choose co_tile
    uint32_t coTile = Co < 8 ? Co : 8; // default 8, small for stability
    if (coTile > MAX_CO_TILE) coTile = MAX_CO_TILE;
    if (coTile == 0) coTile = 1;

    // Choose tile_w within UB budget:
    // bytes = 4 * (coTile * tileW + tileW + 3*(tileW+2)) ~= 4 * (coTile + 4) * tileW + 24
    uint64_t denom = static_cast<uint64_t>(coTile) + 4u;
    uint64_t maxTileByUb = (UB_BUDGET_BYTES > 64 ? (UB_BUDGET_BYTES - 64) : UB_BUDGET_BYTES) / 4 / denom;
    uint32_t tileW = static_cast<uint32_t>(maxTileByUb);
    if (tileW > outW) tileW = outW;
    if (tileW < MIN_TILE_W) tileW = MIN_TILE_W;
    tileW = AlignDown(tileW, ALIGN_ELEMS);
    if (tileW == 0) tileW = (outW < ALIGN_ELEMS ? outW : ALIGN_ELEMS);

    // Parallel decomposition: tasks = N * outH * ceil_div(Co, coTile)
    const uint32_t coGroups = CeilDiv(Co, coTile);
    const uint32_t tasks = N * outH * coGroups;

    uint32_t blockDim = tasks;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    // Save tiling
    tiling.set_n(N);
    tiling.set_ci(Ci);
    tiling.set_co(Co);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_out_h(outH);
    tiling.set_out_w(outW);
    tiling.set_tile_w(tileW);
    tiling.set_co_tile(coTile);
    tiling.set_block_dim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvStandard2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t ci, uint32_t co,
                                uint32_t h, uint32_t w_in,
                                uint32_t out_h, uint32_t out_w,
                                uint32_t tile_w, uint32_t co_tile)
    {
        N_ = n; Ci_ = ci; Co_ = co; H_ = h; W_ = w_in;
        outH_ = out_h; outW_ = out_w; tileW_ = tile_w; coTile_ = co_tile;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * Ci_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(Co_) * Ci_ * 3 * 3);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(N_) * Co_ * outH_ * outW_);

        // UB buffer: outAcc (coTile * tileW) + tmpVec (tileW) + rows (3*(tileW+2))
        uint32_t floats_outAcc = coTile_ * tileW_;
        uint32_t floats_tmp    = tileW_;
        uint32_t floats_rows   = 3 * (tileW_ + 2);
        ubOutAccBytes_ = floats_outAcc * sizeof(float);
        ubTmpBytes_    = floats_tmp * sizeof(float);
        ubRowsBytes_   = floats_rows * sizeof(float);

        pipe_.InitBuffer(calcBuf_, 1, ubOutAccBytes_ + ubTmpBytes_ + ubRowsBytes_);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blkIdx   = static_cast<uint32_t>(AscendC::GetBlockIdx());

        if (N_ == 0 || Ci_ == 0 || Co_ == 0 || outH_ == 0 || outW_ == 0) return;

        const uint32_t coGroups = (Co_ + coTile_ - 1u) / coTile_;
        const uint32_t tasks = N_ * outH_ * coGroups;

        for (uint32_t task = blkIdx; task < tasks; task += blockNum) {
            // Map task -> (n, oh, coStart)
            uint32_t tmp = task;
            const uint32_t n   = tmp / (outH_ * coGroups);
            tmp               -= n * (outH_ * coGroups);
            const uint32_t oh  = tmp / coGroups;
            const uint32_t grp = tmp - oh * coGroups;
            const uint32_t coStart = grp * coTile_;
            const uint32_t coLen   = (coStart + coTile_ <= Co_) ? coTile_ : (Co_ - coStart);

            ComputeRow(n, oh, coStart, coLen);
        }
    }

private:
    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((static_cast<uint64_t>(n) * Ci_ + c) * H_ + h) * W_ + w);
    }

    __aicore__ inline uint64_t WOffset(uint32_t co, uint32_t ci, uint32_t kh, uint32_t kw) const
    {
        return (((static_cast<uint64_t>(co) * Ci_ + ci) * 3u + kh) * 3u + kw);
    }

    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t co, uint32_t h, uint32_t w) const
    {
        return (((static_cast<uint64_t>(n) * Co_ + co) * outH_ + h) * outW_ + w);
    }

    __aicore__ inline void ComputeRow(uint32_t n, uint32_t oh, uint32_t coStart, uint32_t coLen)
    {
        // Prepare UB regions
        AscendC::LocalTensor<float> outAcc = calcBuf_.Get<float>();
        AscendC::LocalTensor<float> tmpVec = calcBuf_.GetWithOffset<float>(tileW_, ubOutAccBytes_);
        const uint32_t rowsOffsetBytes = ubOutAccBytes_ + ubTmpBytes_;
        AscendC::LocalTensor<float> row0 = calcBuf_.GetWithOffset<float>(tileW_ + 2, rowsOffsetBytes);
        AscendC::LocalTensor<float> row1 = calcBuf_.GetWithOffset<float>(tileW_ + 2, rowsOffsetBytes + (tileW_ + 2) * sizeof(float));
        AscendC::LocalTensor<float> row2 = calcBuf_.GetWithOffset<float>(tileW_ + 2, rowsOffsetBytes + 2 * (tileW_ + 2) * sizeof(float));

        // Loop over width tiles
        for (uint32_t owBase = 0; owBase < outW_; owBase += tileW_) {
            uint32_t curT = outW_ - owBase;
            if (curT > tileW_) curT = tileW_;

            // Initialize outAcc (coLen * curT) to zeros
            for (uint32_t i = 0; i < coLen; ++i) {
                AscendC::LocalTensor<float> ySub = outAcc.GetOffsetTensor(i * tileW_);
                AscendC::Dup(ySub, 0.0f, curT);
            }

            // Reduction over input channels
            for (uint32_t ci = 0; ci < Ci_; ++ci) {
                // Load three rows for this input channel (curT + 2 elements each)
                // row0: input[h = oh + 0][w = owBase .. owBase + curT + 1]
                // row1: input[h = oh + 1][...]
                // row2: input[h = oh + 2][...]
                AscendC::DataCopy(row0, xGm_[XOffset(n, ci, oh + 0, owBase)], curT + 2);
                AscendC::DataCopy(row1, xGm_[XOffset(n, ci, oh + 1, owBase)], curT + 2);
                AscendC::DataCopy(row2, xGm_[XOffset(n, ci, oh + 2, owBase)], curT + 2);

                // For each output channel in this tile
                for (uint32_t k = 0; k < coLen; ++k) {
                    const uint32_t coIdx = coStart + k;

                    // Load 3x3 weights for (coIdx, ci)
                    // w order: [0,0],[0,1],[0,2],[1,0]...[2,2]
                    float wbuf[9];
                    // Fetch from GM (9 floats)
                    // Copy by rows to ensure correctness on all alignments
                    AscendC::DataCopy(reinterpret_cast<AscendC::LocalTensor<float>&>(tmpVec), wGm_[WOffset(coIdx, ci, 0, 0)], 9);
                    // tmpVec[0..8] now holds weights
                    // Extract to scalars
                    for (int t = 0; t < 9; ++t) {
                        wbuf[t] = tmpVec.GetValue(t);
                    }

                    AscendC::LocalTensor<float> ySub = outAcc.GetOffsetTensor(k * tileW_);

                    // Accumulate: for each kernel position, mul row slice by scalar and add
                    // ky=0
                    // kx=0
                    AscendC::DataCopy(tmpVec, row0, curT);
                    AscendC::Muls(tmpVec, tmpVec, wbuf[0], curT);
                    AscendC::Add(ySub, ySub, tmpVec, curT);
                    // kx=1
                    AscendC::DataCopy(tmpVec, row0.GetOffsetTensor(1), curT);
                    AscendC::Muls(tmpVec, tmpVec, wbuf[1], curT);
                    AscendC::Add(ySub, ySub, tmpVec, curT);
                    // kx=2
                    AscendC::DataCopy(tmpVec, row0.GetOffsetTensor(2), curT);
                    AscendC::Muls(tmpVec, tmpVec, wbuf[2], curT);
                    AscendC::Add(ySub, ySub, tmpVec, curT);

                    // ky=1
                    // kx=0
                    AscendC::DataCopy(tmpVec, row1, curT);
                    AscendC::Muls(tmpVec, tmpVec, wbuf[3], curT);
                    AscendC::Add(ySub, ySub, tmpVec, curT);
                    // kx=1
                    AscendC::DataCopy(tmpVec, row1.GetOffsetTensor(1), curT);
                    AscendC::Muls(tmpVec, tmpVec, wbuf[4], curT);
                    AscendC::Add(ySub, ySub, tmpVec, curT);
                    // kx=2
                    AscendC::DataCopy(tmpVec, row1.GetOffsetTensor(2), curT);
                    AscendC::Muls(tmpVec, tmpVec, wbuf[5], curT);
                    AscendC::Add(ySub, ySub, tmpVec, curT);

                    // ky=2
                    // kx=0
                    AscendC::DataCopy(tmpVec, row2, curT);
                    AscendC::Muls(tmpVec, tmpVec, wbuf[6], curT);
                    AscendC::Add(ySub, ySub, tmpVec, curT);
                    // kx=1
                    AscendC::DataCopy(tmpVec, row2.GetOffsetTensor(1), curT);
                    AscendC::Muls(tmpVec, tmpVec, wbuf[7], curT);
                    AscendC::Add(ySub, ySub, tmpVec, curT);
                    // kx=2
                    AscendC::DataCopy(tmpVec, row2.GetOffsetTensor(2), curT);
                    AscendC::Muls(tmpVec, tmpVec, wbuf[8], curT);
                    AscendC::Add(ySub, ySub, tmpVec, curT);
                } // k over coLen
            } // ci over Cin

            // Write back results for this width tile
            for (uint32_t k = 0; k < coLen; ++k) {
                const uint32_t coIdx = coStart + k;
                AscendC::LocalTensor<float> ySub = outAcc.GetOffsetTensor(k * tileW_);
                AscendC::DataCopy(yGm_[YOffset(n, coIdx, oh, owBase)], ySub, curT);
            }
        } // width tiles
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBuf_;
    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Ci_{0}, Co_{0}, H_{0}, W_{0}, outH_{0}, outW_{0}, tileW_{0}, coTile_{0};
    // alias names
    uint32_t N_, Ci_, Co_, H_, W_, outH_, outW_, tileW_, coTile_;

    uint32_t ubOutAccBytes_{0}, ubTmpBytes_{0}, ubRowsBytes_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.ci, tiling_data.co,
            tiling_data.h, tiling_data.w,
            tiling_data.out_h, tiling_data.out_w,
            tiling_data.tile_w, tiling_data.co_tile);
    op.Process();
}
