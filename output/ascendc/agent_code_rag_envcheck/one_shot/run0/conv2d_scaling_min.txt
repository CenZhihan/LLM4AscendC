kernel_src="""

#include "kernel_operator.h"

// Naive reference implementation in AscendC for fused Conv2d (stride=1, pad=0, dilation=1)
// followed by scalar scaling and per-pixel reduce-min along output channel dimension.
// Shapes:
//   x: [N, Cin, H, W]
//   w: [Cout, Cin, Kh, Kw]
//   b: [Cout] (optional)
//   scale: scalar tensor (1 element)
//   y: [N, 1, Ho, Wo], where Ho=H-Kh+1, Wo=W-Kw+1.

class KernelConv2dScalingMinCustom {
public:
    __aicore__ inline KernelConv2dScalingMinCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR s, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t hin, uint32_t win,
                                uint32_t cout, uint32_t kh, uint32_t kw,
                                uint32_t hout, uint32_t wout, bool hasBias)
    {
        n_ = n; cin_ = cin; hin_ = hin; win_ = win;
        cout_ = cout; kh_ = kh; kw_ = kw;
        hout_ = hout; wout_ = wout;
        hasBias_ = hasBias;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * cin_ * hin_ * win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cout_) * cin_ * kh_ * kw_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, cout_);
        } else {
            bGm_.SetGlobalBuffer((__gm__ float*)b, 0);
        }
        sGm_.SetGlobalBuffer((__gm__ float*)s, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * hout_ * wout_);

        // Small VECCALC scratch for scalar-sized local tensors
        pipe_.InitBuffer(calcQ_, 8, 16 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || hout_ == 0 || wout_ == 0) {
            return;
        }

        // Load scale (1 element) into local once
        AscendC::LocalTensor<float> scaleLocal = calcQ_.AllocTensor<float>();
        AscendC::DataCopy(scaleLocal, sGm_[0], 1);

        const uint64_t totalOut = static_cast<uint64_t>(n_) * hout_ * wout_;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t outIdx = blockIdx; outIdx < totalOut; outIdx += blockNum) {
            // Decode (n, oh, ow)
            uint64_t t = outIdx;
            const uint32_t n = static_cast<uint32_t>(t / (static_cast<uint64_t>(hout_) * wout_));
            t = t % (static_cast<uint64_t>(hout_) * wout_);
            const uint32_t oh = static_cast<uint32_t>(t / wout_);
            const uint32_t ow = static_cast<uint32_t>(t % wout_);

            // min accumulator local (1 element)
            AscendC::LocalTensor<float> minLocal = calcQ_.AllocTensor<float>();
            bool initMin = false;

            for (uint32_t oc = 0; oc < cout_; ++oc) {
                // sum accumulator local (1 element), initialize to 0
                AscendC::LocalTensor<float> sumLocal = calcQ_.AllocTensor<float>();
                AscendC::Dup(sumLocal, 0.0f, 1);

                // Accumulate convolution sum over (c, kh, kw)
                for (uint32_t c = 0; c < cin_; ++c) {
                    for (uint32_t r = 0; r < kh_; ++r) {
                        // Base GM offset for x at row (oh + r)
                        const uint64_t xRowBase = (((static_cast<uint64_t>(n) * cin_ + c) * hin_ + (oh + r)) * win_) + ow;
                        for (uint32_t s = 0; s < kw_; ++s) {
                            // Load x[n, c, oh+r, ow+s] -> xLocal
                            AscendC::LocalTensor<float> xLocal = calcQ_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[xRowBase + s], 1);

                            // Load w[oc, c, r, s] -> wLocal
                            const uint64_t wOffset = ((((static_cast<uint64_t>(oc) * cin_ + c) * kh_ + r) * kw_) + s);
                            AscendC::LocalTensor<float> wLocal = calcQ_.AllocTensor<float>();
                            AscendC::DataCopy(wLocal, wGm_[wOffset], 1);

                            // tmp = xLocal * wLocal
                            AscendC::LocalTensor<float> tmpLocal = calcQ_.AllocTensor<float>();
                            AscendC::Mul(tmpLocal, xLocal, wLocal, 1);

                            // sumLocal += tmpLocal
                            AscendC::Add(sumLocal, sumLocal, tmpLocal, 1);

                            calcQ_.FreeTensor(tmpLocal);
                            calcQ_.FreeTensor(wLocal);
                            calcQ_.FreeTensor(xLocal);
                        }
                    }
                }

                // Add bias if present: sumLocal += b[oc]
                if (hasBias_) {
                    AscendC::LocalTensor<float> bLocal = calcQ_.AllocTensor<float>();
                    AscendC::DataCopy(bLocal, bGm_[oc], 1);
                    AscendC::Add(sumLocal, sumLocal, bLocal, 1);
                    calcQ_.FreeTensor(bLocal);
                }

                // Multiply by scale: sumLocal = sumLocal * scaleLocal
                AscendC::Mul(sumLocal, sumLocal, scaleLocal, 1);

                // Update per-pixel min across oc
                if (!initMin) {
                    AscendC::DataCopy(minLocal, sumLocal, 1);
                    initMin = true;
                } else {
                    AscendC::Mins(minLocal, minLocal, sumLocal, 1);
                }

                calcQ_.FreeTensor(sumLocal);
            }

            // Store result y[n, 0, oh, ow]
            const uint64_t yOffset = ((static_cast<uint64_t>(n) * hout_ + oh) * wout_) + ow;
            AscendC::DataCopy(yGm_[yOffset], minLocal, 1);
            calcQ_.FreeTensor(minLocal);
        }

        calcQ_.FreeTensor(scaleLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 8> calcQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> sGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, hin_{0}, win_{0};
    uint32_t cout_{0}, kh_{0}, kw_{0};
    uint32_t hout_{0}, wout_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_scaling_min_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scale, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);

    KernelConv2dScalingMinCustom op;
    op.Init(x, weight, bias, scale, y,
            tiling_data.n, tiling_data.cin, tiling_data.hin, tiling_data.win,
            tiling_data.cout, tiling_data.kh, tiling_data.kw,
            tiling_data.hout, tiling_data.wout,
            tiling_data.hasBias != 0u);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline bool has_bias_tensor(const at::Tensor& t) {
    return t.defined() && t.numel() > 0;
}

at::Tensor conv2d_scaling_min_impl_npu(const at::Tensor& x,
                                       const at::Tensor& weight,
                                       const c10::optional<at::Tensor>& bias_opt,
                                       const at::Tensor& scale) {
    at::Tensor y;
    // Output shape: [N, 1, H-Kh+1, W-Kw+1]
    const auto N = x.size(0);
    const auto H = x.size(2);
    const auto W = x.size(3);
    const auto Kh = weight.size(2);
    const auto Kw = weight.size(3);
    const auto Ho = H - Kh + 1;
    const auto Wo = W - Kw + 1;
    y = at::empty({N, 1, Ho, Wo}, x.options());

    if (bias_opt.has_value() && has_bias_tensor(*bias_opt)) {
        EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, *bias_opt, scale, y);
    } else {
        // Pass an undefined/empty tensor for bias
        at::Tensor empty_bias;
        EXEC_NPU_CMD(aclnnConv2dScalingMinCustom, x, weight, empty_bias, scale, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv2d_scaling_min_custom", &conv2d_scaling_min_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv2d_scaling_min_custom", &conv2d_scaling_min_impl_npu,
          "Conv2D + scaling + min-reduction (Conv2dScalingMinCustom on NPU)");
}
