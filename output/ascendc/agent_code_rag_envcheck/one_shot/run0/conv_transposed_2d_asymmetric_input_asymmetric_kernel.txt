    const uint32_t Ci = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t H  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));

    const uint32_t Ho = H + kH - 1u; // stride=1, pad=0, dilation=1
    const uint32_t Wo = W + kW - 1u;

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t Y_BUFFER_NUM = 1;
constexpr int32_t XROW_BUFFER_NUM = 1;
constexpr int32_t TMP_BUFFER_NUM = 1;
constexpr int32_t WS_BUFFER_NUM = 1;

class KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom {
public:
    __aicore__ inline KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Ci, uint32_t Co,
                                uint32_t H, uint32_t W,
                                uint32_t kH, uint32_t kW,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t tileRows)
    {
        N_ = N; Ci_ = Ci; Co_ = Co;
        H_ = H; W_ = W;
        kH_ = kH; kW_ = kW;
        Ho_ = Ho; Wo_ = Wo;
        tileRows_ = (tileRows == 0u) ? 1u : tileRows;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Ci_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Ci_ * Co_ * kH_ * kW_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Co_ * Ho_ * Wo_);

        // UB buffers:
        // yTile: tileRows * Wo
        // xRow: W
        // tmpRow: W
        const uint32_t yTileElems = tileRows_ * Wo_;
        pipe_.InitBuffer(yTileQueue_, Y_BUFFER_NUM, yTileElems * sizeof(float));
        pipe_.InitBuffer(xRowQueue_, XROW_BUFFER_NUM, W_ * sizeof(float));
        pipe_.InitBuffer(tmpRowQueue_, TMP_BUFFER_NUM, W_ * sizeof(float));
        pipe_.InitBuffer(wScalarQueue_, WS_BUFFER_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Ci_ == 0 || Co_ == 0 || H_ == 0 || W_ == 0) {
            return;
        }

        const uint32_t pairs = N_ * Co_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t p = blockIdx; p < pairs; p += (blockNum == 0 ? 1u : blockNum)) {
            const uint32_t n = p / Co_;
            const uint32_t oc = p % Co_;
            ComputeOnePair(n, oc);
        }
    }

private:
    __aicore__ inline void ComputeOnePair(uint32_t n, uint32_t oc)
    {
        // strides
        const uint64_t sx_n = (uint64_t)Ci_ * H_ * W_;
        const uint64_t sx_c = (uint64_t)H_ * W_;
        const uint64_t sx_h = (uint64_t)W_;

        const uint64_t sw_ci = (uint64_t)Co_ * kH_ * kW_;
        const uint64_t sw_co = (uint64_t)kH_ * kW_;

        const uint64_t sy_n = (uint64_t)Co_ * Ho_ * Wo_;
        const uint64_t sy_c = (uint64_t)Ho_ * Wo_;

        for (uint32_t hStart = 0; hStart < Ho_; hStart += tileRows_) {
            const uint32_t rowsThis = AscendC::min(tileRows_, Ho_ - hStart);
            const uint32_t yTileElems = rowsThis * Wo_;

            AscendC::LocalTensor<float> yTile = yTileQueue_.AllocTensor<float>();
            // zero init
            AscendC::Dup(yTile, 0.0f, yTileElems);

            // Accumulate contributions
            for (uint32_t ci = 0; ci < Ci_; ++ci) {
                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    for (uint32_t r = 0; r < rowsThis; ++r) {
                        int32_t in_h = static_cast<int32_t>(hStart + r) - static_cast<int32_t>(kh);
                        if (in_h < 0 || in_h >= static_cast<int32_t>(H_)) {
                            continue;
                        }

                        // Load x row [W] for this (n, ci, in_h)
                        const uint64_t xoff = (uint64_t)n * sx_n + (uint64_t)ci * sx_c + (uint64_t)in_h * sx_h;
                        AscendC::LocalTensor<float> xRow = xRowQueue_.AllocTensor<float>();
                        AscendC::DataCopy(xRow, xGm_[xoff], W_);

                        // Iterate kernel width
                        for (uint32_t kw = 0; kw < kW_; ++kw) {
                            // Load weight scalar w[ci, oc, kh, kw]
                            const uint64_t woff = (uint64_t)ci * sw_ci + (uint64_t)oc * sw_co + (uint64_t)kh * kW_ + (uint64_t)kw;
                            AscendC::LocalTensor<float> wSc = wScalarQueue_.AllocTensor<float>();
                            AscendC::DataCopy(wSc, wGm_[woff], 1);
                            float wval = wSc.GetValue(0);
                            wScalarQueue_.FreeTensor(wSc);

                            // tmp = xRow * wval
                            AscendC::LocalTensor<float> tmpRow = tmpRowQueue_.AllocTensor<float>();
                            AscendC::Muls(tmpRow, xRow, wval, W_);

                            // yTile segment offset = r * Wo + kw, length W
                            const uint32_t yoff = r * Wo_ + kw;
                            AscendC::Add(yTile[yoff], yTile[yoff], tmpRow, W_);

                            tmpRowQueue_.FreeTensor(tmpRow);
                        }

                        xRowQueue_.FreeTensor(xRow);
                    }
                }
            }

            // Store yTile to GM at [n, oc, hStart:hStart+rowsThis, :]
            const uint64_t yoff = (uint64_t)n * sy_n + (uint64_t)oc * sy_c + (uint64_t)hStart * (uint64_t)Wo_;
            AscendC::DataCopy(yGm_[yoff], yTile, yTileElems);

            yTileQueue_.FreeTensor(yTile);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, Y_BUFFER_NUM> yTileQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, XROW_BUFFER_NUM> xRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, TMP_BUFFER_NUM> tmpRowQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, WS_BUFFER_NUM> wScalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Ci_{0}, Co_{0};
    uint32_t H_{0}, W_{0};
    uint32_t kH_{0}, kW_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t tileRows_{1};
};

extern "C" __global__ __aicore__ void conv_transposed_2d_asymmetric_input_asymmetric_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvTransposed2dAsymmetricInputAsymmetricKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Ci, tiling_data.Co,
            tiling_data.H, tiling_data.W,
            tiling_data.kH, tiling_data.kW,
            tiling_data.Ho, tiling_data.Wo,
            tiling_data.tileRows);
    op.Process();
}
