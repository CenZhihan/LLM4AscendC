host_operator_src="""

#include "conv_standard_1d_dilated_strided_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard1dDilatedStridedCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect x: [N, Cin, Lin], w: [Cout, Cin, K]
    const uint32_t n = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t lin = static_cast<uint32_t>(x_shape->GetDim(2));

    const uint32_t cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t wcin = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t k = static_cast<uint32_t>(w_shape->GetDim(2));

    if (cin != wcin) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t stride = 3u;
    const uint32_t dilation = 4u;

    const uint32_t lout =
        (lin >= (dilation * (k - 1) + 1))
            ? (static_cast<uint32_t>((lin - dilation * (k - 1) - 1) / stride) + 1u)
            : 0u;

    // UB usage: buffers = yTile + xTile + tmpTile + weightTile
    // floatsBudget = UB_BUDGET_BYTES/4
    const uint32_t floatsBudget = UB_BUDGET_BYTES / 4u;
    const uint32_t weightFloats = cin * k;
    uint32_t maxTile = 0;
    if (floatsBudget > weightFloats + 256u) {
        maxTile = (floatsBudget - weightFloats - 256u) / 3u;
    }
    if (maxTile == 0) maxTile = ALIGN_ELEMS;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileOut = maxTile;
    if (lout < tileOut) {
        uint32_t a = AlignDown(lout, ALIGN_ELEMS);
        tileOut = (a > 0) ? a : lout;
    }
    if (tileOut == 0) tileOut = 1;

    // Set blockDim based on N * Cout
    const uint32_t tasks = n * cout;
    uint32_t blockDim = tasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u; // conservative

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_cin(cin);
    tiling.set_cout(cout);
    tiling.set_lin(lin);
    tiling.set_lout(lout);
    tiling.set_k(k);
    tiling.set_stride(stride);
    tiling.set_dilation(dilation);
    tiling.set_tileOut(tileOut);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvStandard1dDilatedStridedCustom {
public:
    __aicore__ inline KernelConvStandard1dDilatedStridedCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t lout,
                                uint32_t k, uint32_t stride, uint32_t dilation,
                                uint32_t tileOut)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        lin_ = lin; lout_ = lout;
        k_ = k; stride_ = stride; dilation_ = dilation;
        tileOut_ = tileOut;

        const uint32_t xTotal = n_ * cin_ * lin_;
        const uint32_t wTotal = cout_ * cin_ * k_;
        const uint32_t yTotal = n_ * cout_ * lout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xTotal);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wTotal);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yTotal);

        // UB buffers
        pipe_.InitBuffer(bufX_, BUFFER_NUM, tileOut_ * sizeof(float));
        pipe_.InitBuffer(bufY_, BUFFER_NUM, tileOut_ * sizeof(float));
        pipe_.InitBuffer(bufT_, BUFFER_NUM, tileOut_ * sizeof(float));
        pipe_.InitBuffer(bufW_, BUFFER_NUM, (cin_ * k_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (lout_ == 0 || cout_ == 0 || cin_ == 0 || n_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t taskId = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalTasks = n_ * cout_;

        // Grid-stride loop over (n, co) tasks
        for (; taskId < totalTasks; taskId += blockNum) {
            const uint32_t nIdx = taskId / cout_;
            const uint32_t co = taskId % cout_;

            // Load weight slice for (co, :, :)
            AscendC::LocalTensor<float> wLocal = bufW_.AllocTensor<float>();
            const uint32_t wOffset = (co * cin_ * k_);
            AscendC::DataCopy(wLocal, wGm_[wOffset], cin_ * k_);
            bufW_.EnQue(wLocal);

            // Iterate output positions in tiles
            for (uint32_t lBase = 0; lBase < lout_; lBase += tileOut_) {
                uint32_t len = lout_ - lBase;
                if (len > tileOut_) len = tileOut_;

                // y accumulator
                AscendC::LocalTensor<float> yLocal = bufY_.AllocTensor<float>();
                AscendC::Dup(yLocal, 0.0f, len);

                // Accumulate over input channels and kernel taps
                AscendC::LocalTensor<float> wL = bufW_.DeQue<float>();
                for (uint32_t ci = 0; ci < cin_; ++ci) {
                    // For each tap k
#pragma unroll
                    for (uint32_t kk = 0; kk < 3; ++kk) { // kernel k is small (expect 3), still robust if k_ != 3
                        if (kk >= k_) break;

                        // Gather strided input along length dimension into UB
                        AscendC::LocalTensor<float> xLocal = bufX_.AllocTensor<float>();
                        const uint32_t xBaseIndex = ((nIdx * cin_ + ci) * lin_) + (lBase * stride_ + kk * dilation_);
                        // 2D copy: nBurst = len, burstLen = 1, srcStride = stride_-1, dstStride = 0
                        AscendC::DataCopy(xLocal, xGm_[xBaseIndex], len, 1, stride_ - 1, 0);

                        // Load scalar weight w[co, ci, kk] from local weight buffer
                        const uint32_t wIdx = ci * k_ + kk;
                        float wScalar = 0.0f;
                        // Read one value from local weight tensor into scalar
                        // Using a temporary 1-length tensor to fetch the scalar to register
                        AscendC::LocalTensor<float> t1 = bufT_.AllocTensor<float>();
                        AscendC::DataCopy(t1, wL[wIdx], 1);
                        // Move to scalar (assume first element)
                        wScalar = AscendC::GetValue(t1, 0); // helper to fetch scalar from local tensor
                        bufT_.FreeTensor(t1);

                        // Multiply-Accumulate: y += x * wScalar
                        AscendC::LocalTensor<float> tLocal = bufT_.AllocTensor<float>();
                        AscendC::Muls(tLocal, xLocal, wScalar, len);
                        AscendC::Add(yLocal, yLocal, tLocal, len);

                        bufT_.FreeTensor(tLocal);
                        bufX_.FreeTensor(xLocal);
                    }
                }
                bufW_.EnQue(wL);

                // Store y tile
                const uint32_t yOffset = ((nIdx * cout_ + co) * lout_) + lBase;
                AscendC::DataCopy(yGm_[yOffset], yLocal, len);

                bufY_.FreeTensor(yLocal);
                // Make sure weight local stays enqueued for reuse across tiles
                wL = bufW_.DeQue<float>();
            }

            // Free weight buffer
            bufW_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bufX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> bufY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bufT_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bufW_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_ {0}, cin_ {0}, cout_ {0};
    uint32_t lin_ {0}, lout_ {0};
    uint32_t k_ {0}, stride_ {0}, dilation_ {0};
    uint32_t tileOut_ {0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_dilated_strided_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dDilatedStridedCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.lout,
            tiling_data.k, tiling_data.stride, tiling_data.dilation,
            tiling_data.tileOut);
    op.Process();
}
