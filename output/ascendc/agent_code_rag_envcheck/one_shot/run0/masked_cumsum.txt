host_operator_src="""

#include "masked_cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   inQueueM: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue: 1 * tile (masked buffer)
// Total floats = 7 * tile => bytes = 28 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MaskedCumsumCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const gert::OriginShape &origin = shape->GetOriginShape();
    const uint32_t totalLength = static_cast<uint32_t>(origin.GetShapeSize());

    const int32_t rank = static_cast<int32_t>(origin.GetDimNum());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (*dimAttr);
    if (dim < 0) dim += rank;
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    uint32_t axisLength = static_cast<uint32_t>(origin.GetDim(dim));
    uint32_t outerCount = 1;
    for (int32_t i = 0; i < dim; ++i) {
        outerCount *= static_cast<uint32_t>(origin.GetDim(i));
    }
    uint32_t innerStride = 1;
    for (int32_t i = dim + 1; i < rank; ++i) {
        innerStride *= static_cast<uint32_t>(origin.GetDim(i));
    }

    // This kernel assumes cumsum axis is the innermost (contiguous) dimension.
    // If not, we still run but tile length will be conservative to limit strided access.
    const bool axisIsLast = (innerStride == 1);
    (void)axisIsLast;

    uint32_t rowCount = (axisLength == 0) ? 0u : (totalLength / axisLength);

    // tileLen from UB budget: bytes = 28 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 28;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (axisLength < tileLen) {
        const uint32_t aligned = AlignDown(axisLength, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : axisLength;
    }

    // Adaptive blockDim based on number of rows; keep conservative for stability.
    uint32_t blockDim = (rowCount == 0) ? 1u : rowCount;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_totalLength(totalLength);
    tiling.set_axisLength(axisLength);
    tiling.set_rowCount(rowCount);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMaskedCumsumCustom {
public:
    __aicore__ inline KernelMaskedCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR mask, GM_ADDR y,
                                uint32_t totalLength, uint32_t axisLength,
                                uint32_t rowCount, uint32_t tileLength, int32_t dim)
    {
        totalLen_ = totalLength;
        axisLen_ = axisLength;
        rowCnt_ = rowCount;
        tileLen_ = tileLength;
        dim_ = dim;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        mGm_.SetGlobalBuffer((__gm__ float*)mask, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(inQueueM_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || axisLen_ == 0 || tileLen_ == 0 || rowCnt_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes rows in a grid-stride fashion.
        for (uint32_t row = blockIdx; row < rowCnt_; row += blockNum) {
            uint32_t baseCol = 0;
            float carry = 0.0f;

            // Prefetch first tile of this row
            uint32_t curLen = (axisLen_ > tileLen_) ? tileLen_ : axisLen_;
            CopyIn(row, baseCol, curLen);

            while (true) {
                // Prefetch next tile early for overlap
                const uint32_t nextBaseCol = baseCol + curLen;
                uint32_t nextLen = 0;
                if (nextBaseCol < axisLen_) {
                    nextLen = axisLen_ - nextBaseCol;
                    if (nextLen > tileLen_) nextLen = tileLen_;
                    CopyIn(row, nextBaseCol, nextLen);
                }

                Compute(curLen, carry);
                CopyOut(row, baseCol, curLen);

                if (nextLen == 0) break;
                baseCol = nextBaseCol;
                curLen = nextLen;
            }
        }
    }

private:
    __aicore__ inline uint32_t RowBaseOffset(uint32_t row) const
    {
        // Assuming last dimension cumsum (contiguous along axis).
        return row * axisLen_;
    }

    __aicore__ inline void CopyIn(uint32_t row, uint32_t col, uint32_t len)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        AscendC::LocalTensor<float> mLocal = inQueueM_.AllocTensor<float>();
        const uint32_t gmOffset = RowBaseOffset(row) + col;
        AscendC::DataCopy(xLocal, xGm_[gmOffset], len);
        AscendC::DataCopy(mLocal, mGm_[gmOffset], len);
        inQueueX_.EnQue(xLocal);
        inQueueM_.EnQue(mLocal);
    }

    __aicore__ inline void Compute(uint32_t len, float &carry)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> mLocal = inQueueM_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();

        // tLocal = xLocal * mLocal (apply mask, mask is float 0/1)
        AscendC::Mul(tLocal, xLocal, mLocal, len);

        // yLocal = prefix sum of tLocal, then add carry to all elements
        // Compute inclusive prefix sum in-place: y[i] = t[i] + y[i-1]
        // Start by copying first element
        float prev = carry;
        for (uint32_t i = 0; i < len; ++i) {
            float v = tLocal.GetValue(i);
            float s = prev + v;
            yLocal.SetValue(i, s);
            prev = s;
        }
        // Update carry for next tile (last element of this tile)
        float newCarry = prev;

        // No need to add carry now; we've already included it in the loop.

        // Free temps and enqueue output
        tmpQueue_.FreeTensor(tLocal);
        outQueueY_.EnQue<float>(yLocal);
        inQueueX_.FreeTensor(xLocal);
        inQueueM_.FreeTensor(mLocal);

        carry = newCarry;
    }

    __aicore__ inline void CopyOut(uint32_t row, uint32_t col, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        const uint32_t gmOffset = RowBaseOffset(row) + col;
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueM_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> mGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t totalLen_ {0};
    uint32_t axisLen_ {0};
    uint32_t rowCnt_ {0};
    uint32_t tileLen_ {0};
    int32_t dim_ { -1 };
};

extern "C" __global__ __aicore__ void masked_cumsum_custom(GM_ADDR x, GM_ADDR mask, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMaskedCumsumCustom op;
    op.Init(x, mask, y,
            tiling_data.totalLength, tiling_data.axisLength,
            tiling_data.rowCount, tiling_data.tileLength, tiling_data.dim);
    op.Process();
}
