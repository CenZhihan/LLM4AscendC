#include "kernel_operator.h"

using namespace AscendC;

class KernelGemmBiasAddHardtanhMishGroupNormCustom {
public:
    __aicore__ inline KernelGemmBiasAddHardtanhMishGroupNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight,
                                GM_ADDR lin_bias, GM_ADDR bias,
                                GM_ADDR gn_gamma, GM_ADDR gn_beta,
                                GM_ADDR y,
                                uint32_t N, uint32_t K, uint32_t C,
                                uint32_t G, uint32_t groupSize,
                                uint32_t kTile,
                                float eps, float minVal, float maxVal)
    {
        N_ = N; K_ = K; C_ = C; G_ = G; groupSize_ = groupSize; kTile_ = kTile;
        eps_ = eps; minVal_ = minVal; maxVal_ = maxVal;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, (uint64_t)C_ * K_);
        lbGm_.SetGlobalBuffer((__gm__ float*)lin_bias, C_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, C_);
        gammaGm_.SetGlobalBuffer((__gm__ float*)gn_gamma, C_);
        betaGm_.SetGlobalBuffer((__gm__ float*)gn_beta, C_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * C_);

        // Queues/buffers
        pipe_.InitBuffer(qX_, 1, kTile_ * sizeof(float));
        pipe_.InitBuffer(qW_, 1, kTile_ * sizeof(float));
        pipe_.InitBuffer(qMul_, 1, kTile_ * sizeof(float));
        pipe_.InitBuffer(qAcc_, 1, groupSize_ * sizeof(float));
        pipe_.InitBuffer(qY_, 1, groupSize_ * sizeof(float));
        pipe_.InitBuffer(qGVec_, 1, groupSize_ * sizeof(float));
        pipe_.InitBuffer(qBVec_, 1, groupSize_ * sizeof(float));
        pipe_.InitBuffer(qTmp_, 1, groupSize_ * sizeof(float));
        pipe_.InitBuffer(qScalar_, 1, sizeof(float)); // for scalar reductions (len=1)
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || K_ == 0 || C_ == 0) return;

        const uint32_t blockNum = (uint32_t)GetBlockNum();
        const uint32_t blockIdx = (uint32_t)GetBlockIdx();

        for (uint32_t n = blockIdx; n < N_; n += blockNum) {
            ProcessRow(n);
        }
    }

private:
    __aicore__ inline void ReduceInPlaceToLen(LocalTensor<float>& v, uint32_t len, uint32_t targetLen = 1)
    {
        uint32_t cur = len;
        while (cur > targetLen) {
            uint32_t half = cur >> 1;
            Add(v, v, v[half], half);
            if ((cur & 1u) != 0u) {
                // add last leftover into v[0]
                Add(v, v, v[cur - 1], 1);
                cur = half + 1;
            } else {
                cur = half;
            }
        }
    }

    __aicore__ inline void BroadcastScalar(LocalTensor<float>& outVec, const LocalTensor<float>& scalarVec, uint32_t len)
    {
        // Assume scalarVec length 1, broadcast to outVec
        // Fill outVec with scalar value by expanding doubling
        // Start: copy first element
        DataCopy(outVec, scalarVec, 1);
        uint32_t filled = 1;
        while (filled < len) {
            uint32_t copyLen = (len - filled < filled) ? (len - filled) : filled;
            DataCopy(outVec[filled], outVec, copyLen);
            filled += copyLen;
        }
    }

    __aicore__ inline void ProcessRow(uint32_t n)
    {
        // Iterate groups
        for (uint32_t g = 0; g < G_; ++g) {
            const uint32_t cStart = g * groupSize_;
            // accumulators per channel in group
            LocalTensor<float> acc = qAcc_.AllocTensor<float>();
            // zero init
            Dup(acc, 0.0f, groupSize_);

            // K-tiled accumulation
            for (uint32_t k0 = 0; k0 < K_; k0 += kTile_) {
                uint32_t lenK = K_ - k0;
                if (lenK > kTile_) lenK = kTile_;

                LocalTensor<float> xChunk = qX_.AllocTensor<float>();
                DataCopy(xChunk, xGm_[n * K_ + k0], lenK);

                for (uint32_t ci = 0; ci < groupSize_; ++ci) {
                    const uint32_t c = cStart + ci;

                    LocalTensor<float> wChunk = qW_.AllocTensor<float>();
                    DataCopy(wChunk, wGm_[c * K_ + k0], lenK);

                    LocalTensor<float> mulVec = qMul_.AllocTensor<float>();
                    Mul(mulVec, xChunk, wChunk, lenK);

                    // reduce mulVec to length 1 at head
                    ReduceInPlaceToLen(mulVec, lenK, 1);
                    // acc[ci] += mulVec[0]
                    Add(acc[ci], acc[ci], mulVec, 1);

                    qMul_.FreeTensor(mulVec);
                    qW_.FreeTensor(wChunk);
                }
                qX_.FreeTensor(xChunk);
            }

            // yVec = acc + lin_bias[group] + bias[group]
            LocalTensor<float> yVec = qY_.AllocTensor<float>();

            LocalTensor<float> lbVec = qTmp_.AllocTensor<float>();
            DataCopy(lbVec, lbGm_[cStart], groupSize_);
            LocalTensor<float> bVec = qBVec_.AllocTensor<float>();
            DataCopy(bVec, bGm_[cStart], groupSize_);

            Add(yVec, acc, lbVec, groupSize_);
            Add(yVec, yVec, bVec, groupSize_);

            // Hardtanh clamp
            Maxs(yVec, yVec, minVal_, groupSize_);
            Mins(yVec, yVec, maxVal_, groupSize_);

            // Mish: y = y * tanh(ln(1 + exp(y)))
            LocalTensor<float> tmp1 = qTmp_.AllocTensor<float>(); // reuse buffer (overwrite lbVec)
            Exp(tmp1, yVec, groupSize_);
            Adds(tmp1, tmp1, 1.0f, groupSize_);
            Ln(tmp1, tmp1, groupSize_);
            Tanh(tmp1, tmp1, groupSize_);
            Mul(yVec, yVec, tmp1, groupSize_);

            // Compute group mean and variance
            // sum = ReduceSum(yVec), sumsq = ReduceSum(yVec*yVec)
            LocalTensor<float> sumBuf = qScalar_.AllocTensor<float>(); // len 1
            LocalTensor<float> work = qTmp_.AllocTensor<float>();
            // Copy yVec to work for reduction
            DataCopy(work, yVec, groupSize_);
            ReduceInPlaceToLen(work, groupSize_, 1);
            DataCopy(sumBuf, work, 1);

            // y2
            Mul(work, yVec, yVec, groupSize_);
            ReduceInPlaceToLen(work, groupSize_, 1);
            LocalTensor<float> sumSqBuf = qScalar_.AllocTensor<float>();
            DataCopy(sumSqBuf, work, 1);

            // meanVec and invStdVec
            LocalTensor<float> meanVec = qTmp_.AllocTensor<float>();
            // mean = sum / groupSize
            // compute scalar mean in sumBuf (1 element) then broadcast
            // scale sumBuf by 1/groupSize
            float invGs = 1.0f / static_cast<float>(groupSize_);
            Muls(sumBuf, sumBuf, invGs, 1);
            BroadcastScalar(meanVec, sumBuf, groupSize_);

            // var = sumSq / groupSize - mean^2
            Muls(sumSqBuf, sumSqBuf, invGs, 1);
            LocalTensor<float> varVec = qTmp_.AllocTensor<float>(); // reuse another buffer (meanVec already used)
            BroadcastScalar(varVec, sumSqBuf, groupSize_);
            // varVec = varVec - meanVec*meanVec
            LocalTensor<float> meanSq = qTmp_.AllocTensor<float>();
            Mul(meanSq, meanVec, meanVec, groupSize_);
            Sub(varVec, varVec, meanSq, groupSize_);

            // invStdVec = 1 / sqrt(var + eps)
            Adds(varVec, varVec, eps_, groupSize_);
            LocalTensor<float> invStdVec = qTmp_.AllocTensor<float>();
            Rsqrt(invStdVec, varVec, groupSize_);

            // norm = (y - mean)*invStd
            Sub(yVec, yVec, meanVec, groupSize_);
            Mul(yVec, yVec, invStdVec, groupSize_);

            // affine: y = y * gamma + beta
            LocalTensor<float> gammaVec = qGVec_.AllocTensor<float>();
            DataCopy(gammaVec, gammaGm_[cStart], groupSize_);
            Mul(yVec, yVec, gammaVec, groupSize_);

            LocalTensor<float> betaVec = qBVec_.AllocTensor<float>(); // bVec was used; reuse or keep separate
            DataCopy(betaVec, betaGm_[cStart], groupSize_);
            Add(yVec, yVec, betaVec, groupSize_);

            // Store to GM
            DataCopy(yGm_[n * C_ + cStart], yVec, groupSize_);

            // Free locals
            qBVec_.FreeTensor(betaVec);
            qGVec_.FreeTensor(gammaVec);
            qTmp_.FreeTensor(invStdVec);
            qTmp_.FreeTensor(meanSq);
            qTmp_.FreeTensor(varVec);
            qTmp_.FreeTensor(meanVec);
            qScalar_.FreeTensor(sumSqBuf);
            qTmp_.FreeTensor(work);
            qScalar_.FreeTensor(sumBuf);
            qTmp_.FreeTensor(tmp1);
            qBVec_.FreeTensor(bVec);
            qTmp_.FreeTensor(lbVec);
            qY_.FreeTensor(yVec);
            qAcc_.FreeTensor(acc);
        }
    }

private:
    TPipe pipe_;

    // Queues
    TQue<TPosition::VECCALC, 1> qX_;
    TQue<TPosition::VECCALC, 1> qW_;
    TQue<TPosition::VECCALC, 1> qMul_;
    TQue<TPosition::VECCALC, 1> qAcc_;
    TQue<TPosition::VECCALC, 1> qY_;
    TQue<TPosition::VECCALC, 1> qGVec_;
    TQue<TPosition::VECCALC, 1> qBVec_;
    TQue<TPosition::VECCALC, 1> qTmp_;
    TQue<TPosition::VECCALC, 1> qScalar_;

    // GM tensors
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> lbGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> gammaGm_;
    GlobalTensor<float> betaGm_;
