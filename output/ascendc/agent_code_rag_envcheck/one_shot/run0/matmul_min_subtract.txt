host_operator_src="""
#include "matmul_min_subtract_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core usable UB for this kernel
static constexpr uint32_t ALIGN_ELEMS_M = 128;          // align tileM for vector efficiency
static constexpr uint32_t DEFAULT_TILEK = 128;          // chunk along K
static constexpr uint32_t MAX_BLOCKS = 24;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulMinSubtractCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0); // [N, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [M, K]

    const uint64_t N64 = (x_shape->GetDimNum() >= 1) ? x_shape->GetDim(0) : 1;
    const uint64_t K64 = (x_shape->GetDimNum() >= 2) ? x_shape->GetDim(1) : 1;
    const uint64_t M64 = (w_shape->GetDimNum() >= 1) ? w_shape->GetDim(0) : 1;

    const uint32_t N = static_cast<uint32_t>(N64);
    const uint32_t M = static_cast<uint32_t>(M64);
    const uint32_t K = static_cast<uint32_t>(K64);

    // Choose tileK fixed and compute tileM based on UB budget:
    // UB consumption ~= 4*(tileM + 2*tileK) bytes per inner loop (yVec + xChunk + wChunk)
    uint32_t tileK = DEFAULT_TILEK;
    if (tileK > K) tileK = AlignDown(K, 16u);
    if (tileK == 0) tileK = (K == 0 ? 1u : K);

    uint32_t maxTileM_elems = 0;
    if (UB_BUDGET_BYTES > (2u * tileK * sizeof(float))) {
        maxTileM_elems = (UB_BUDGET_BYTES - 2u * tileK * sizeof(float)) / sizeof(float);
    }
    uint32_t tileM = maxTileM_elems;
    if (tileM > M) tileM = M;
    tileM = AlignDown(tileM, ALIGN_ELEMS_M);
    if (tileM == 0) tileM = (M < ALIGN_ELEMS_M ? M : ALIGN_ELEMS_M);

    // Blocks: distribute rows (N) over blocks conservatively
    uint32_t blockDim = (N == 0) ? 1u : N;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    const gert::Shape *w_shape = ctx->GetInputShape(1); // [M, K]
    if (x_shape->GetDimNum() < 2 || w_shape->GetDimNum() < 2) {
        return GRAPH_FAILED;
    }
    const int64_t N = x_shape->GetDim(0);
    const int64_t M = w_shape->GetDim(0);
    gert::Shape *y_shape = ctx->GetOutputShape(0);
    y_shape->SetDimNum(2);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, M);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

class KernelMatmulMinSubtractCustom {
public:
    __aicore__ inline KernelMatmulMinSubtractCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR c, GM_ADDR y,
                                uint32_t N, uint32_t M, uint32_t K,
                                uint32_t tileM, uint32_t tileK)
    {
        N_ = N; M_ = M; K_ = K;
        tileM_ = tileM; tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, M_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * M_);

        pipe_.InitBuffer(inQueueX_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, tileM_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || M_ == 0 || K_ == 0) {
            return;
        }

        // Load scalar c once
        float cVal = 0.0f;
        {
            AscendC::LocalTensor<float> cLocal = inQueueX_.AllocTensor<float>();
            AscendC::DataCopy(cLocal, cGm_[0], 1);
            inQueueX_.EnQue(cLocal);
            AscendC::LocalTensor<float> cLocal2 = inQueueX_.DeQue<float>();
            cVal = cLocal2[0];
            inQueueX_.FreeTensor(cLocal2);
        }

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t n = blockIdx; n < N_; n += blockNum) {
            for (uint32_t mStart = 0; mStart < M_; mStart += tileM_) {
                const uint32_t mLen = AscendC::Min(tileM_, M_ - mStart);

                // yLocal initialized with bias b segment
                AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
                AscendC::DataCopy(yLocal, bGm_[mStart], mLen);

                for (uint32_t kStart = 0; kStart < K_; kStart += tileK_) {
                    const uint32_t kLen = AscendC::Min(tileK_, K_ - kStart);

                    // Load x row chunk once for this k-segment
                    AscendC::LocalTensor<float> xChunk = inQueueX_.AllocTensor<float>();
                    const uint64_t xOffset = (uint64_t)n * K_ + kStart;
                    AscendC::DataCopy(xChunk, xGm_[xOffset], kLen);
                    inQueueX_.EnQue(xChunk);
                    AscendC::LocalTensor<float> xBuf = inQueueX_.DeQue<float>();

                    // Iterate each output channel within this tileM block
                    for (uint32_t mOff = 0; mOff < mLen; ++mOff) {
                        // Load corresponding weight chunk for this output channel
                        AscendC::LocalTensor<float> wChunk = inQueueW_.AllocTensor<float>();
                        const uint64_t wOffset = (uint64_t)(mStart + mOff) * K_ + kStart;
                        AscendC::DataCopy(wChunk, wGm_[wOffset], kLen);

                        // Accumulate dot product into yLocal[mOff]
                        float acc = 0.0f;
                        for (uint32_t t = 0; t < kLen; ++t) {
                            acc += xBuf[t] * wChunk[t];
                        }
                        yLocal[mOff] = yLocal[mOff] + acc;

                        inQueueW_.FreeTensor(wChunk);
                    }

                    inQueueX_.FreeTensor(xBuf);
                }

                // Apply clamp with c and subtract c: y = min(y, c) - c
                for (uint32_t i = 0; i < mLen; ++i) {
                    float v = yLocal[i];
                    if (v > cVal) v = cVal;
                    v = v - cVal;
                    yLocal[i] = v;
                }

                // Store y row tile
                const uint64_t yOffset = (uint64_t)n * M_ + mStart;
                AscendC::DataCopy(yGm_[yOffset], yLocal, mLen);
                outQueueY_.FreeTensor(yLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, M_{0}, K_{0};
    uint32_t tileM_{0}, tileK_{0};
};

extern "C" __global__ __aicore__ void matmul_min_subtract_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR c, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulMinSubtractCustom op;
    op.Init(x, w, b, c, y,
            tiling_data.N, tiling_data.M, tiling_data.K,
            tiling_data.tileM, tiling_data.tileK);
    op.Process();
}
