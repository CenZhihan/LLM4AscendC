static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative UB budget
static constexpr uint32_t ALIGN_ELEMS = 256;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dSubtractHardSwishMaxPoolMishCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    // Expect x: [N, Cin, H, W], weight: [Cout, Cin, K, K]
    uint32_t N   = static_cast<uint32_t>(x_shape->GetOriginShape()[0]);
    uint32_t Cin = static_cast<uint32_t>(x_shape->GetOriginShape()[1]);
    uint32_t H   = static_cast<uint32_t>(x_shape->GetOriginShape()[2]);
    uint32_t W   = static_cast<uint32_t>(x_shape->GetOriginShape()[3]);

    uint32_t Cout = static_cast<uint32_t>(w_shape->GetOriginShape()[0]);
    uint32_t K    = static_cast<uint32_t>(w_shape->GetOriginShape()[2]); // assume square kernels

    // Fixed conv params: stride=1, padding=0, dilation=1
    const uint32_t OH = (H >= K) ? (H - K + 1u) : 0u;
    const uint32_t OW = (W >= K) ? (W - K + 1u) : 0u;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *subtractValue = attrs->GetAttrPointer<float>(0);
    const int32_t *poolKernelSize = attrs->GetAttrPointer<int32_t>(1);
    const bool *useBias = attrs->GetAttrPointer<bool>(2);

    const uint32_t pK = static_cast<uint32_t>(*poolKernelSize);
    const uint32_t PH = (pK == 0 || OH == 0) ? 0u : (OH / pK);
    const uint32_t PW = (pK == 0 || OW == 0) ? 0u : (OW / pK);

    // Tile on pooled output linear positions for a fixed (n, oc)
    // UB usage ~ weight(Cin*K*K) + patchRow(K) + results(tileLen) + margins
    const uint32_t weightElems = Cin * K * K;
    uint32_t tileBudgetElems = UB_BUDGET_BYTES / 4; // float32
    uint32_t reserve = weightElems + K + 1024;      // margins and temp
    uint32_t tileLen = (tileBudgetElems > reserve) ? (tileBudgetElems - reserve) : 1u;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0) tileLen = ALIGN_ELEMS;
    const uint64_t phpwTotal = static_cast<uint64_t>(PH) * static_cast<uint64_t>(PW);
    if (phpwTotal > 0 && tileLen > phpwTotal) {
        uint32_t aligned = AlignDown(static_cast<uint32_t>(phpwTotal), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(phpwTotal);
    }

    // Block dim: distribute over N*Cout groups
    const uint32_t groups = (N == 0 || Cout == 0) ? 1u : (N * Cout);
    uint32_t blockDim = groups;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // Save tiling
    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Cout(Cout);
    tiling.set_K(K);
    tiling.set_OH(OH);
    tiling.set_OW(OW);
    tiling.set_PH(PH);
    tiling.set_PW(PW);
    tiling.set_tileLen(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_poolK(pK);

#include "kernel_operator.h"

using namespace AscendC;

class KernelConv2dSubtractHardSwishMaxPoolMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractHardSwishMaxPoolMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                const optiling::Conv2dSubtractHardSwishMaxPoolMishCustomTilingData* tData,
                                TPipe* pipe)
    {
        pipe_ = pipe;
        // Shapes and params
        N_ = tData->N();
        Cin_ = tData->Cin();
        H_ = tData->H();
        W_ = tData->W();
        Cout_ = tData->Cout();
        K_ = tData->K();
        OH_ = tData->OH();
        OW_ = tData->OW();
        PH_ = tData->PH();
        PW_ = tData->PW();
        tileLen_ = tData->tileLen();
        poolK_ = tData->poolK();
        useBias_ = (tData->useBias() != 0);
        subtractValue_ = tData->subtractValue();

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cout_ * Cin_ * K_ * K_);
        if (useBias_) { bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)Cout_); }
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Cout_ * PH_ * PW_);

        // Buffers: weights per oc, patch row of length K, output tile
        const uint32_t weightElems = Cin_ * K_ * K_;
        pipe_->InitBuffer(weightBuf_, 1, weightElems * sizeof(float));
        pipe_->InitBuffer(patchBuf_, 1, K_ * sizeof(float));
        // output tile buffer
        pipe_->InitBuffer(outBuf_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || PH_ == 0 || PW_ == 0) return;

        // Determine group work distribution across blocks: groups = N * Cout
        const uint32_t groups = N_ * Cout_;
        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());

        for (uint32_t g = blockIdx; g < groups; g += blockNum) {
            const uint32_t n = g / Cout_;
            const uint32_t oc = g % Cout_;

            // Load weights for current oc
            LocalTensor<float> wLocal = weightBuf_.AllocTensor<float>();
            const uint64_t wOffset = ((uint64_t)oc) * Cin_ * K_ * K_;
            DataCopy(wLocal, wGm_[wOffset], Cin_ * K_ * K_);

            // Bias value if used
            float biasVal = 0.0f;
            if (useBias_) {
                biasVal = bGm_.GetGM()[oc]; // direct GM pointer access via GetGM(), if supported
            }

            // Output tile buffer
            LocalTensor<float> yTile = outBuf_.AllocTensor<float>();

            const uint64_t phpwTotal = (uint64_t)PH_ * (uint64_t)PW_;
            for (uint64_t base = 0; base < phpwTotal; base += tileLen_) {
                uint32_t curLen = tileLen_;
                if (base + curLen > phpwTotal) {
                    curLen = static_cast<uint32_t>(phpwTotal - base);
                }

                // Compute curLen pooled outputs
                for (uint32_t t = 0; t < curLen; ++t) {
                    uint64_t idx = base + t;
                    uint32_t ph = static_cast<uint32_t>(idx / PW_);
                    uint32_t pw = static_cast<uint32_t>(idx % PW_);
                    const uint32_t oh0 = ph * poolK_;
                    const uint32_t ow0 = pw * poolK_;

                    float poolMax = -3.402823466e+38F; // -FLT_MAX

                    // iterate over pool window
                    for (uint32_t dh = 0; dh < poolK_; ++dh) {
                        for (uint32_t dw = 0; dw < poolK_; ++dw) {
                            const uint32_t oh = oh0 + dh;
                            const uint32_t ow = ow0 + dw;

                            // Naive conv dot for this (n, oc, oh, ow)
                            float acc = 0.0f;
                            // loop c, kh
                            for (uint32_t c = 0; c < Cin_; ++c) {
                                for (uint32_t kh = 0; kh < K_; ++kh) {
                                    // Load input row slice of length K_
                                    LocalTensor<float> xRow = patchBuf_.AllocTensor<float>();
                                    const uint64_t xOffset = ((((uint64_t)n * Cin_ + c) * H_ + (oh + kh)) * W_ + ow);
                                    DataCopy(xRow, xGm_[xOffset], K_);

                                    // weight segment for this c, kh
                                    const uint64_t wSegBase = (((uint64_t)c * K_) + kh) * K_;
                                    // dot product on K_
                                    for (uint32_t kw = 0; kw < K_; ++kw) {
                                        acc += xRow.GetValue(kw) * wLocal.GetValue(wSegBase + kw);
                                    }
                                }
                            }

                            if (useBias_) { acc += biasVal; }
                            // subtract
                            acc -= subtractValue_;
                            // hardswish: acc * clip(acc + 3, 0, 6) / 6
                            float tval = acc + 3.0f;
                            float clipv = tval < 0.0f ? 0.0f : (tval > 6.0f ? 6.0f : tval);
                            float hs = acc * (clipv * (1.0f / 6.0f));
                            // pool max
                            poolMax = (hs > poolMax) ? hs : poolMax;
                        }
                    }

                    // mish: x * tanh(softplus(x)), softplus(x)=ln(1+exp(x))
                    float e = expf(poolMax);
                    float sp = log1pf(e);
                    float th = tanhf(sp);
                    float outv = poolMax * th;

                    yTile.SetValue(t, outv);
                }

                // Store yTile to GM
                const uint64_t yBase = ((((uint64_t)n) * Cout_ + oc) * PH_) * PW_;
                DataCopy(yGm_[yBase + base], yTile, curLen);
            }

            // Free local buffers (optional in this style; queues freed at scope end)
        }
    }

private:
    // Global tensors
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    // Queues/buffers
    TPipe* pipe_{nullptr};
    TQue<TPosition::VECCALC, 1> weightBuf_;
    TQue<TPosition::VECCALC, 1> patchBuf_;
    TQue<TPosition::VECOUT, 1> outBuf_;

    // params
    uint32_t N_{0}, Cin_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, K_{0};
    uint32_t OH_{0}, OW_{0}, PH_{0}, PW_{0};
    uint32_t tileLen_{1};
    uint32_t poolK_{1};
    bool useBias_{false};
    float subtractValue_{0.0f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    TPipe pipe;
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractHardSwishMaxPoolMishCustom op;
    op.Init(x, weight, bias, y, &tiling_data, &pipe);
    op.Process();
}
