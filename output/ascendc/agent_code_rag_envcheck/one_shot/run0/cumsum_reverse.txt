host_operator_src="""

#include "cumsum_reverse_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// UB model for this kernel:
//   inQueue: 1 * tileInner
//   outQueue: 1 * tileInner
// Total floats = 2 * tileInner => bytes = 8 * tileInner
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumReverseCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const auto &origin = shape->GetOriginShape();

    int32_t rank = static_cast<int32_t>(origin.GetDimNum());
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = *dimPtr;
    if (dim < 0) {
        dim += rank;
    }
    if (dim < 0) dim = 0;
    if (dim >= rank) dim = rank - 1;

    uint64_t inner64 = static_cast<uint64_t>(origin.GetDim(dim));
    if (inner64 == 0) inner64 = 1;
    uint64_t total64 = origin.GetShapeSize();
    uint64_t outer64 = (inner64 == 0) ? 0 : (total64 / inner64);

    // tileInner from UB budget: bytes = 8 * tileInner
    uint32_t maxTile = UB_BUDGET_BYTES / 8;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t inner = static_cast<uint32_t>(inner64);
    uint32_t outer = static_cast<uint32_t>(outer64);

    uint32_t tileInner = maxTile;
    if (inner < tileInner) {
        uint32_t aligned = AlignDown(inner, ALIGN_ELEMS);
        tileInner = (aligned > 0) ? aligned : inner;
    }
    if (tileInner == 0) tileInner = 1;

    // Blocks over rows (outer)
    uint32_t blockDim = (outer == 0) ? 1u : outer;
    if (blockDim < 8u) blockDim = (outer >= 8u ? 8u : (outer == 0 ? 1u : outer));
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_outer(outer);
    tiling.set_inner(inner);
    tiling.set_tileInner(tileInner);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumsumReverseCustom : public OpDef {
public:
    explicit CumsumReverseCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(0);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumsumReverseCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUFFERS = 2;
constexpr int32_t OUT_BUFFERS = 2;

class KernelCumsumReverseCustom {
public:
    __aicore__ inline KernelCumsumReverseCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t inner,
                                uint32_t tileInner)
    {
        outer_ = outer;
        inner_ = inner;
        tileInner_ = tileInner;

        total_ = outer_ * inner_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, total_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, total_);

        // UB queues: in/out tiles of size tileInner
        pipe_.InitBuffer(inQueue_, IN_BUFFERS, tileInner_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUT_BUFFERS, tileInner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (total_ == 0 || tileInner_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block processes multiple rows (outer dimension) with grid-stride
        for (uint32_t row = blockIdx; row < outer_; row += blockNum) {
            ProcessRow(row);
        }
    }

private:
    __aicore__ inline void ProcessRow(uint32_t row)
    {
        uint32_t base = row * inner_;
        // Reverse cumsum along last dimension (length = inner_)
        // Process tiles from rightmost to leftmost
        uint32_t remain = inner_;
        float carry = 0.0f;

        while (remain > 0) {
            uint32_t curLen = (remain >= tileInner_) ? tileInner_ : remain;
            uint32_t startPos = remain - curLen;            // start index within the row
            uint32_t gmOffset = base + startPos;

            CopyIn(gmOffset, curLen);
            carry = ComputeTileReverse(curLen, carry);
            CopyOut(gmOffset, curLen);

            remain = startPos;
        }
    }

    __aicore__ inline void CopyIn(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> inLocal = inQueue_.AllocTensor<float>();
        AscendC::DataCopy(inLocal, xGm_[gmOffset], len);
        inQueue_.EnQue<float>(inLocal);
    }

    __aicore__ inline float ComputeTileReverse(uint32_t len, float carryIn)
    {
        AscendC::LocalTensor<float> xLocal = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();

        float acc = carryIn;
        // Compute reverse inclusive cumsum within this tile
        // y[i] = x[i] + sum_{k=i+1..end}(x[k]) + carryIn
        for (int64_t i = static_cast<int64_t>(len) - 1; i >= 0; --i) {
            float v = xLocal.GetValue(i);
            acc += v;
            yLocal.SetValue(i, acc);
        }

        outQueue_.EnQue<float>(yLocal);
        inQueue_.FreeTensor(xLocal);

        return acc; // carry to the next (left) tile
    }

    __aicore__ inline void CopyOut(uint32_t gmOffset, uint32_t len)
    {
        AscendC::LocalTensor<float> yLocal = outQueue_.DeQue<float>();
        AscendC::DataCopy(yGm_[gmOffset], yLocal, len);
        outQueue_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFERS> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFERS> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_{0};
    uint32_t inner_{0};
    uint32_t tileInner_{0};
    uint32_t total_{0};
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverseCustom op;
    op.Init(x, y, tiling_data.outer, tiling_data.inner, tiling_data.tileInner);
    op.Process();
}
"""
