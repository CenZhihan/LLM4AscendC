kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard3dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard3dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t d, uint32_t h, uint32_t wd,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t dout, uint32_t hout, uint32_t wout,
                                uint32_t tileW)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        d_ = d; h_ = h; w_ = wd;
        kd_ = kd; kh_ = kh; kw_ = kw;
        dout_ = dout; hout_ = hout; wout_ = wout;
        tileW_ = (tileW == 0) ? 1u : tileW;

        xTotal_ = static_cast<uint64_t>(n_) * cin_ * d_ * h_ * w_;
        wTotal_ = static_cast<uint64_t>(cout_) * cin_ * kd_ * kh_ * kw_;
        yTotal_ = static_cast<uint64_t>(n_) * cout_ * dout_ * hout_ * wout_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xTotal_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wTotal_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, yTotal_);

        // Buffers: xVec, yVec(accumulator), tmpVec, wScalar(1)
        pipe_.InitBuffer(xVecQ_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(yVecQ_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpVecQ_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(wScalarQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (yTotal_ == 0 || wout_ == 0 || dout_ == 0 || hout_ == 0 || cout_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalLines = static_cast<uint64_t>(n_) * cout_ * dout_ * hout_;

        for (uint64_t line = blockIdx; line < totalLines; line += blockNum) {
            uint64_t tmp = line;
            const uint32_t ho = static_cast<uint32_t>(tmp % hout_); tmp /= hout_;
            const uint32_t do_ = static_cast<uint32_t>(tmp % dout_); tmp /= dout_;
            const uint32_t co = static_cast<uint32_t>(tmp % cout_); tmp /= cout_;
            const uint32_t n  = static_cast<uint32_t>(tmp);

            for (uint32_t wStart = 0; wStart < wout_; wStart += tileW_) {
                const uint32_t len = (wStart + tileW_ <= wout_) ? tileW_ : (wout_ - wStart);

                AscendC::LocalTensor<float> yVec = yVecQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> tmpVec = tmpVecQ_.AllocTensor<float>();
                AscendC::LocalTensor<float> xVec = xVecQ_.AllocTensor<float>();

                bool firstTerm = true;

                for (uint32_t ci = 0; ci < cin_; ++ci) {
                    for (uint32_t kd = 0; kd < kd_; ++kd) {
                        for (uint32_t kh = 0; kh < kh_; ++kh) {
                            for (uint32_t kw = 0; kw < kw_; ++kw) {
                                // Load weight scalar
                                const uint64_t wIndex = ((((static_cast<uint64_t>(co) * cin_ + ci) * kd_ + kd) * kh_ + kh) * kw_ + kw);
                                AscendC::LocalTensor<float> wScalar = wScalarQ_.AllocTensor<float>();
                                AscendC::DataCopy(wScalar, wGm_[wIndex], 1);
                                // Read x vector
                                const uint64_t xIndexBase =
                                    (((((static_cast<uint64_t>(n) * cin_ + ci) * d_ + (do_ + kd)) * h_ + (ho + kh)) * w_) + (wStart + kw));
                                AscendC::DataCopy(xVec, xGm_[xIndexBase], len);

                                // tmpVec = xVec * wScalar
                                float wval = wScalar.GetValue(0);
                                AscendC::Muls(tmpVec, xVec, wval, len);

                                if (firstTerm) {
                                    // yVec = tmpVec
                                    AscendC::DataCopy(yVec, tmpVec, len);
                                    firstTerm = false;
                                } else {
                                    AscendC::Add(yVec, yVec, tmpVec, len);
                                }

                                wScalarQ_.FreeTensor(wScalar);
                            }
                        }
                    }
                }

                // Write back yVec to GM
                const uint64_t yIndexBase =
                    (((((static_cast<uint64_t>(n) * cout_ + co) * dout_ + do_) * hout_ + ho) * wout_) + wStart);
                AscendC::DataCopy(yGm_[yIndexBase], yVec, len);

                xVecQ_.FreeTensor(xVec);
                tmpVecQ_.FreeTensor(tmpVec);
                yVecQ_.FreeTensor(yVec);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wScalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t d_{0}, h_{0}, w_{0};
    uint32_t kd_{0}, kh_{0}, kw_{0};
    uint32_t dout_{0}, hout_{0}, wout_{0};
    uint32_t tileW_{1};

    uint64_t xTotal_{0}, wTotal_{0}, yTotal_{0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard3dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.dout, tiling_data.hout, tiling_data.wout,
            tiling_data.tileW);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor conv3d_out_shape_allocate(const at::Tensor& x, const at::Tensor& weight) {
    TORCH_CHECK(x.dim() == 5, \"x must be 5D NCDHW\");
    TORCH_CHECK(weight.dim() == 5, \"weight must be 5D OCKDKHKW\");
    auto n = x.size(0);
    auto d = x.size(2);
    auto h = x.size(3);
    auto w = x.size(4);
    auto co = weight.size(0);
    auto kd = weight.size(2);
    auto kh = weight.size(3);
    auto kw = weight.size(4);
    auto dout = d - kd + 1;
    auto hout = h - kh + 1;
    auto wout = w - kw + 1;
    TORCH_CHECK(dout > 0 && hout > 0 && wout > 0, \"Invalid shapes for valid 3D conv (stride=1 pad=0)\");
    at::IntArrayRef sizes({n, co, dout, hout, wout});
    return at::empty(sizes, x.options());
}

at::Tensor conv3d_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    at::Tensor result = conv3d_out_shape_allocate(x, weight);
    EXEC_NPU_CMD(aclnnConvStandard3dSquareInputSquareKernelCustom, x, weight, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv_standard_3d_square_input_square_kernel_custom\", &conv3d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
