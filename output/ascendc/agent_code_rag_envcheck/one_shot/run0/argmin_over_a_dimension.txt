host_operator_src="""

#include "argmin_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t CeilDiv(uint64_t a, uint64_t b)
{
    return static_cast<uint32_t>((a + b - 1u) / b);
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgminOverADimensionCustomTilingData tiling;

    const gert::Shape* inShape = context->GetInputShape(0);
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *axisPtr = attrs->GetAttrPointer<int32_t>(0);

    // Get dims
    const int64_t dimCount = inShape->GetOriginShape().GetDimNum();
    if (dimCount <= 0) {
        return ge::GRAPH_FAILED;
    }

    int32_t axis = *axisPtr;
    if (axis < 0) axis += static_cast<int32_t>(dimCount);
    if (axis < 0 || axis >= dimCount) {
        return ge::GRAPH_FAILED;
    }

    uint64_t totalLen = 1;
    uint64_t reduceLen = static_cast<uint64_t>(inShape->GetOriginShape().GetDim(axis));
    uint64_t left = 1;
    uint64_t right = 1;

    for (int64_t i = 0; i < dimCount; ++i) {
        const uint64_t d = static_cast<uint64_t>(inShape->GetOriginShape().GetDim(i));
        totalLen *= d;
        if (i < axis) left *= d;
        else if (i > axis) right *= d;
    }

    const uint64_t segmentCount = left * right;

    // Conservative blockDim selection
    uint32_t blockDim = CeilDiv(segmentCount, 1);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalLength(static_cast<uint32_t>(totalLen));
    tiling.set_reduceLen(static_cast<uint32_t>(reduceLen));
    tiling.set_right(static_cast<uint32_t>(right));
    tiling.set_segmentCount(static_cast<uint32_t>(segmentCount));
    tiling.set_blockDim(blockDim);
    tiling.set_axis(axis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *axisPtr = attrs->GetAttrPointer<int32_t>(0);

    int32_t axis = *axisPtr;
    int64_t dimNum = x_shape->GetOriginShape().GetDimNum();
    if (axis < 0) axis += static_cast<int32_t>(dimNum);

    // Build output dims by removing axis dimension
    std::vector<int64_t> outDims;
    outDims.reserve(dimNum > 0 ? static_cast<size_t>(dimNum - 1) : 0u);
    for (int64_t i = 0; i < dimNum; ++i) {
        if (i == axis) continue;
        outDims.push_back(x_shape->GetOriginShape().GetDim(i));
    }

    gert::Shape outShape;
    outShape.SetDimNum(static_cast<int64_t>(outDims.size()));
    for (size_t i = 0; i < outDims.size(); ++i) {
        outShape.SetDim(static_cast<int64_t>(i), outDims[i]);
    }
    *y_shape = outShape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    context->SetOutputDataType(0, ge::DT_INT64);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class ArgminOverADimensionCustom : public OpDef {
public:
    explicit ArgminOverADimensionCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_INT64})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(REQUIRED).Int();

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(ArgminOverADimensionCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;
static inline __aicore__ float kPosInf() { return 3.402823466e+38F; }

class KernelArgminOverADimensionCustom {
public:
    __aicore__ inline KernelArgminOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t totalLength,
                                uint32_t reduceLen,
                                uint32_t right,
                                uint32_t segmentCount,
                                int32_t axis)
    {
        totalLen_ = totalLength;
        reduceLen_ = reduceLen;
        right_ = right;
        segCount_ = segmentCount;
        axis_ = axis;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ int64_t*)y, segCount_);

        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, sizeof(float)); // scalar slot
    }

    __aicore__ inline void Process()
    {
        if (segCount_ == 0 || reduceLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        uint32_t seg = blockIdx;
        while (seg < segCount_) {
            ComputeOneSegment(seg);
            seg += blockNum;
        }
    }

private:
    __aicore__ inline void ComputeOneSegment(uint32_t seg)
    {
        // seg enumerates over left*right: seg = l * right + r
        const uint32_t l = (right_ == 0) ? 0u : (seg / right_);
        const uint32_t r = (right_ == 0) ? 0u : (seg % right_);
        const uint32_t base = (l * reduceLen_) * right_ + r;

        float minVal = kPosInf();
        int64_t minIdx = 0;

        // Iterate across the reduction dimension with stride = right_
        for (uint32_t j = 0; j < reduceLen_; ++j) {
            const uint32_t idx = base + j * right_;

            // Load one element from GM
            // Using a small UB buffer to ensure compatibility with DMA semantics.
            LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
            DataCopy(tmpLocal, xGm_[idx], 1);
            // Read back to scalar
            float val = tmpLocal.GetValue(0);

            if (val < minVal) {
                minVal = val;
                minIdx = static_cast<int64_t>(j);
            }
            tmpQueue_.FreeTensor(tmpLocal);
        }

        // Write index to output
        yGm_[seg] = minIdx;
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    GlobalTensor<float>   xGm_;
    GlobalTensor<int64_t> yGm_;

    uint32_t totalLen_ {0};
    uint32_t reduceLen_ {0};
    uint32_t right_ {0};
    uint32_t segCount_ {0};
    int32_t  axis_ {0};
};

extern "C" __global__ __aicore__ void argmin_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgminOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.totalLength,
            tiling_data.reduceLen,
            tiling_data.right,
            tiling_data.segmentCount,
            tiling_data.axis);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static std::vector<int64_t> infer_argmin_out_sizes(const at::Tensor& self, int64_t dim) {
    auto sizes = self.sizes().vec();
    int64_t nd = static_cast<int64_t>(sizes.size());
    if (dim < 0) dim += nd;
    TORCH_CHECK(dim >= 0 && dim < nd, \"argmin_over_a_dimension_custom: dim out of range\");
    sizes.erase(sizes.begin() + dim);
    if (sizes.size() == 0) sizes.push_back(1); // handle scalar edge-case
    return sizes;
}

at::Tensor argmin_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    std::vector<int64_t> out_sizes = infer_argmin_out_sizes(self, dim);
    at::Tensor result = at::empty(out_sizes, self.options().dtype(at::kLong));
    EXEC_NPU_CMD(aclnnArgminOverADimensionCustom, self, dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmin_over_a_dimension_custom", &argmin_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmin_over_a_dimension_custom", &argmin_over_a_dimension_impl_npu, "Argmin over a dimension (ArgminOverADimensionCustom on NPU)");
}
