#include "mean_reduction_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <cstdint>
#include <vector>

namespace optiling {

// UB model for this kernel:
//   accQueue: 1 * tile
//   inQueue:  1 * tile
//   outQueue: 1 * tile
// Total floats = 3 * tile => bytes = 12 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // DMA-friendly alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap tile size to avoid long iters

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MeanReductionOverADimensionCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    // dim attribute
    const int64_t *dimPtr = attrs->GetAttrPointer<int64_t>(0);

    const gert::Shape *inShape = context->GetInputShape(0);
    // Get dims of the original shape
    const std::vector<int64_t> dims = inShape->GetOriginShape().GetDims();
    const uint32_t rank = static_cast<uint32_t>(dims.size());

    int64_t axis = *dimPtr;
    int64_t canAxis = axis < 0 ? (axis + static_cast<int64_t>(rank)) : axis;
    if (canAxis < 0 || canAxis >= static_cast<int64_t>(rank)) {
        return ge::GRAPH_FAILED;
    }

    uint64_t outer64 = 1;
    uint64_t inner64 = 1;
    const uint64_t reduceLen64 = static_cast<uint64_t>(dims[static_cast<size_t>(canAxis)]);

    for (int64_t i = 0; i < canAxis; ++i) {
        outer64 *= static_cast<uint64_t>(dims[static_cast<size_t>(i)]);
    }
    for (int64_t i = canAxis + 1; i < static_cast<int64_t>(rank); ++i) {
        inner64 *= static_cast<uint64_t>(dims[static_cast<size_t>(i)]);
    }

    // Bound to uint32_t for kernel params
    const uint32_t outer = static_cast<uint32_t>(outer64);
    const uint32_t inner = static_cast<uint32_t>(inner64);
    const uint32_t reduceLen = static_cast<uint32_t>(reduceLen64);

    // tileLen from UB budget: bytes = 12 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 12;
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (inner < tileLen) {
        const uint32_t aligned = AlignDown(inner, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : inner;
    }
    if (tileLen == 0) tileLen = 1;

    const uint32_t segsPerOuter = CeilDiv(inner, tileLen);
    uint64_t totalWork64 = static_cast<uint64_t>(outer) * static_cast<uint64_t>(segsPerOuter);
    uint32_t blockDim = static_cast<uint32_t>(totalWork64);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_outer(outer);
    tiling.set_inner(inner);
    tiling.set_reduceLen(reduceLen);
    tiling.set_tileLen(tileLen);
    tiling.set_segsPerOuter(segsPerOuter);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    const std::vector<int64_t> dims = x_shape->GetOriginShape().GetDims();
    const uint32_t rank = static_cast<uint32_t>(dims.size());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMeanReductionOverADimensionCustom {
public:
    __aicore__ inline KernelMeanReductionOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t outer, uint32_t inner, uint32_t reduceLen,
                                uint32_t tileLen, uint32_t segsPerOuter)
    {
        outer_ = outer;
        inner_ = inner;
        reduceLen_ = reduceLen;
        tileLen_ = tileLen;
        segsPerOuter_ = segsPerOuter;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(outer_) * reduceLen_ * inner_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(outer_) * inner_);

        pipe_.InitBuffer(loadQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, BUFFER_NUM, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (inner_ == 0 || reduceLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalWork = static_cast<uint64_t>(outer_) * static_cast<uint64_t>(segsPerOuter_);
        for (uint64_t work = blockIdx; work < totalWork; work += blockNum) {
            const uint32_t outerIdx = static_cast<uint32_t>(work / segsPerOuter_);
            const uint32_t segIdx = static_cast<uint32_t>(work % segsPerOuter_);

            const uint32_t innerBase = segIdx * tileLen_;
            if (innerBase >= inner_) continue;
            const uint32_t curLen = AscendC::min(static_cast<uint32_t>(inner_ - innerBase), tileLen_);

            ReduceTile(outerIdx, innerBase, curLen);
        }
    }

private:
    __aicore__ inline void ReduceTile(uint32_t outerIdx, uint32_t innerBase, uint32_t len)
    {
        // Allocate accumulator
        AscendC::LocalTensor<float> accLocal = accQueue_.AllocTensor<float>();

        bool first = true;
        for (uint32_t k = 0; k < reduceLen_; ++k) {
            const uint64_t xOffset = (static_cast<uint64_t>(outerIdx) * reduceLen_ + k) * inner_ + innerBase;

            AscendC::LocalTensor<float> xLocal = loadQueue_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[xOffset], len);

            if (first) {
                AscendC::DataCopy(accLocal, xLocal, len);
                first = false;
            } else {
                AscendC::Add(accLocal, accLocal, xLocal, len);
            }
            loadQueue_.FreeTensor(xLocal);
        }

        // Compute mean
        AscendC::LocalTensor<float> yLocal = outQueue_.AllocTensor<float>();
        const float invLen = 1.0f / static_cast<float>(reduceLen_);
        AscendC::Muls(yLocal, accLocal, invLen, len);

        // Enqueue and copy out
        outQueue_.EnQue(yLocal);
        accQueue_.FreeTensor(accLocal);

        AscendC::LocalTensor<float> yOutLocal = outQueue_.DeQue<float>();
        const uint64_t yOffset = static_cast<uint64_t>(outerIdx) * inner_ + innerBase;
        AscendC::DataCopy(yGm_[yOffset], yOutLocal, len);
        outQueue_.FreeTensor(yOutLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> loadQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outer_ {0};
    uint32_t inner_ {0};
    uint32_t reduceLen_ {0};
    uint32_t tileLen_ {0};
    uint32_t segsPerOuter_ {0};
};

extern "C" __global__ __aicore__ void mean_reduction_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMeanReductionOverADimensionCustom op;
    op.Init(x, y,
            tiling_data.outer, tiling_data.inner, tiling_data.reduceLen,
            tiling_data.tileLen, tiling_data.segsPerOuter);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include <vector>
#include "pytorch_npu_helper.hpp"

static inline int64_t canonical_dim(int64_t dim, int64_t rank) {
    if (dim < 0) dim += rank;
    return dim;
}

at::Tensor mean_reduction_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    const int64_t rank = static_cast<int64_t>(self.dim());
    const int64_t cdim = canonical_dim(dim, rank);

    TORCH_CHECK(cdim >= 0 && cdim < rank, "dim out of range");

    std::vector<int64_t> out_sizes;
    out_sizes.reserve(rank > 0 ? static_cast<size_t>(rank - 1) : 0);
    for (int64_t i = 0; i < rank; ++i) {
        if (i == cdim) continue;
        out_sizes.push_back(self.size(i));
    }

    at::Tensor result = at::empty(out_sizes, self.options());
    EXEC_NPU_CMD(aclnnMeanReductionOverADimensionCustom, self, cdim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("mean_reduction_over_a_dimension_custom", &mean_reduction_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("mean_reduction_over_a_dimension_custom", &mean_reduction_over_a_dimension_impl_npu,
          "Mean reduce over a dimension (MeanReductionOverADimensionCustom on NPU)");
}
