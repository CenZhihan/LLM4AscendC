host_operator_src="""

#include "conv_standard_1d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : ((a + b - 1u) / b);
}
} // namespace

namespace optiling {

// For this straightforward baseline kernel we tile along output length.
// Choose a conservative tile size to limit per-core working set and improve cache locality.
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MAX_TILE_ELEMS = 16384;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard1dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const uint32_t n    = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t cin  = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t lin  = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t ksz  = static_cast<uint32_t>(w_shape->GetDim(2));

    const uint32_t lout = (lin >= ksz) ? (lin - ksz + 1u) : 0u;

    uint32_t tileLen = (lout == 0u) ? 0u : MAX_TILE_ELEMS;
    if (tileLen > lout) tileLen = lout;
    tileLen = AlignDown(tileLen, ALIGN_ELEMS);
    if (tileLen == 0 && lout > 0) tileLen = lout; // small tails

    const uint32_t numTiles = (tileLen == 0u) ? 1u : CeilDiv(lout, tileLen);
    uint32_t blockDim = n * numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_cin(cin);
    tiling.set_cout(cout);
    tiling.set_lin(lin);
    tiling.set_lout(lout);
    tiling.set_ksize(ksz);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape->GetDimNum() != 3 || w_shape->GetDimNum() != 3) {
        return GRAPH_FAILED;
    }
    const int64_t N = x_shape->GetDim(0);
    const int64_t Cin = x_shape->GetDim(1);
    const int64_t Lin = x_shape->GetDim(2);
    const int64_t Cout = w_shape->GetDim(0);
    const int64_t K = w_shape->GetDim(2);

    if (Lin < K) {
        return GRAPH_FAILED;
    }

    gert::Shape *y_shape = context->GetOutputShape(0);
    y_shape->SetDimNum(3);
    y_shape->SetDim(0, N);
    y_shape->SetDim(1, Cout);
    y_shape->SetDim(2, Lin - K + 1);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    const ge::DataType w_dtype = context->GetInputDataType(1);
    if (x_dtype != w_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class ConvStandard1dCustom : public OpDef {
public:
    explicit ConvStandard1dCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("weight")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(ConvStandard1dCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

class KernelConvStandard1dCustom {
public:
    __aicore__ inline KernelConvStandard1dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t lin, uint32_t lout, uint32_t ksize,
                                uint32_t tileLength)
    {
        n_ = n;
        cin_ = cin;
        cout_ = cout;
        lin_ = lin;
        lout_ = lout;
        k_ = ksize;
        tileLen_ = tileLength;

        xPtr_ = reinterpret_cast<__gm__ float*>(x);
        wPtr_ = reinterpret_cast<__gm__ float*>(w);
        yPtr_ = reinterpret_cast<__gm__ float*>(y);
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cin_ == 0 || cout_ == 0 || lin_ == 0 || lout_ == 0 || k_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numTiles = (tileLen_ == 0u) ? 1u : ((lout_ + tileLen_ - 1u) / tileLen_);
        const uint32_t totalSteps = n_ * numTiles;

        for (uint32_t step = blockIdx; step < totalSteps; step += blockNum) {
            const uint32_t n = step / numTiles;
            const uint32_t t = step % numTiles;
            const uint32_t loBase = t * tileLen_;
            uint32_t curLen = lout_ - loBase;
            if (curLen > tileLen_) curLen = tileLen_;
            if (curLen == 0) continue;

            for (uint32_t co = 0; co < cout_; ++co) {
                // For each output channel and for each position in the tile, compute direct conv
                for (uint32_t off = 0; off < curLen; ++off) {
                    const uint32_t l = loBase + off;
                    float acc = 0.0f;
                    // Unroll small K if possible; keep generic loops
                    for (uint32_t ci = 0; ci < cin_; ++ci) {
                        const uint32_t xBase = ((n * cin_ + ci) * lin_) + l;
                        const uint32_t wBase = ((co * cin_ + ci) * k_);
                        for (uint32_t kk = 0; kk < k_; ++kk) {
                            acc += xPtr_[xBase + kk] * wPtr_[wBase + kk];
                        }
                    }
                    const uint32_t yOff = ((n * cout_ + co) * lout_) + l;
                    yPtr_[yOff] = acc;
                }
            }
        }
    }

private:
    __gm__ float* xPtr_{nullptr};
    __gm__ float* wPtr_{nullptr};
    __gm__ float* yPtr_{nullptr};

    uint32_t n_{0};
    uint32_t cin_{0};
    uint32_t cout_{0};
    uint32_t lin_{0};
    uint32_t lout_{0};
    uint32_t k_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv_standard_1d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard1dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.lin, tiling_data.lout, tiling_data.ksize,
            tiling_data.tileLength);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv_standard_1d_impl_npu(const at::Tensor& x, const at::Tensor& weight) {
    TORCH_CHECK(x.dim() == 3, "Input x must be 3D [N, Cin, Lin]");
    TORCH_CHECK(weight.dim() == 3, "Weight must be 3D [Cout, Cin, K]");
    TORCH_CHECK(x.size(1) == weight.size(1), "Cin mismatch between x and weight");
    const int64_t N = x.size(0);
    const int64_t Cin = x.size(1);
    const int64_t Lin = x.size(2);
    const int64_t Cout = weight.size(0);
    const int64_t K = weight.size(2);
    TORCH_CHECK(Lin >= K, "Kernel size must be <= input length for valid conv1d");
    const int64_t Lout = Lin - K + 1;

    auto options = x.options();
    at::Tensor y = at::empty({N, Cout, Lout}, options);
    EXEC_NPU_CMD(aclnnConvStandard1dCustom, x, weight, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv_standard_1d_custom", &conv_standard_1d_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv_standard_1d_custom", &conv_standard_1d_impl_npu, "1D standard convolution (ConvStandard1dCustom on NPU)");
}
