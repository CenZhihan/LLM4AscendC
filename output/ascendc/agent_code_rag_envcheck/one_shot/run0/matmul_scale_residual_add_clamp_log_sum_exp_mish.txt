kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelMatmulScaleResidualAddClampLogSumExpMishCustom {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR scaling, GM_ADDR clampMin, GM_ADDR clampMax,
                                GM_ADDR y,
                                uint32_t M, uint32_t K, uint32_t N, uint32_t tileK)
    {
        M_ = M;
        K_ = K;
        N_ = N;
        tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)M_ * (uint64_t)K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)N_ * (uint64_t)K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);

        scalingGm_.SetGlobalBuffer((__gm__ float*)scaling, 1);
        clampMinGm_.SetGlobalBuffer((__gm__ float*)clampMin, 1);
        clampMaxGm_.SetGlobalBuffer((__gm__ float*)clampMax, 1);

        yGm_.SetGlobalBuffer((__gm__ float*)y, M_);

        // Queues/buffers
        pipe_.InitBuffer(xTileQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(wTileQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, BUFFER_NUM, sizeof(float) * 8); // a few scalars
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        // Load scalar parameters once
        AscendC::LocalTensor<float> sLocal = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> loLocal = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> hiLocal = scalarQ_.AllocTensor<float>();
        AscendC::DataCopy(sLocal, scalingGm_[0], 1);
        AscendC::DataCopy(loLocal, clampMinGm_[0], 1);
        AscendC::DataCopy(hiLocal, clampMaxGm_[0], 1);

        AscendC::LocalTensor<float> oneLocal = scalarQ_.AllocTensor<float>();
        AscendC::Dup(oneLocal, 1.0f, 1);

        AscendC::LocalTensor<float> s1Local = scalarQ_.AllocTensor<float>();
        AscendC::Add(s1Local, sLocal, oneLocal, 1); // s1 = s + 1

        // Grid-stride on rows
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t m = blockIdx; m < M_; m += blockNum) {
            // sumExp scalar = 0
            AscendC::LocalTensor<float> sumExpLocal = scalarQ_.AllocTensor<float>();
            AscendC::Dup(sumExpLocal, 0.0f, 1);

            // Traverse over N output neurons
            for (uint32_t n = 0; n < N_; ++n) {
                // acc scalar = 0
                AscendC::LocalTensor<float> accLocal = scalarQ_.AllocTensor<float>();
                AscendC::Dup(accLocal, 0.0f, 1);

                // Compute dot(x_row[m], w_row[n])
                uint64_t xRowBase = (uint64_t)m * (uint64_t)K_;
                uint64_t wRowBase = (uint64_t)n * (uint64_t)K_;

                uint32_t processed = 0;
                while (processed < K_) {
                    uint32_t len = K_ - processed;
                    if (len > tileK_) len = tileK_;

                    AscendC::LocalTensor<float> xLocal = xTileQ_.AllocTensor<float>();
                    AscendC::DataCopy(xLocal, xGm_[xRowBase + processed], len);

                    AscendC::LocalTensor<float> wLocal = wTileQ_.AllocTensor<float>();
                    AscendC::DataCopy(wLocal, wGm_[wRowBase + processed], len);

                    AscendC::LocalTensor<float> mulLocal = tmpQ_.AllocTensor<float>();
                    AscendC::Mul(mulLocal, xLocal, wLocal, len);

                    // reduce sum into resLocal (scalar)
                    AscendC::LocalTensor<float> resLocal = scalarQ_.AllocTensor<float>();
                    AscendC::ReduceSum(resLocal, mulLocal, len);

                    // accLocal += resLocal
                    AscendC::Add(accLocal, accLocal, resLocal, 1);

                    scalarQ_.FreeTensor(resLocal);
                    tmpQ_.FreeTensor(mulLocal);
                    wTileQ_.FreeTensor(wLocal);
                    xTileQ_.FreeTensor(xLocal);

                    processed += len;
                }

                // accLocal += bias[n]
                AscendC::LocalTensor<float> bLocal = scalarQ_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[n], 1);
                AscendC::Add(accLocal, accLocal, bLocal, 1);

                // val = acc * (s+1)
                AscendC::Mul(accLocal, accLocal, s1Local, 1);

                // clamp(lo, hi)
                AscendC::Max(accLocal, accLocal, loLocal, 1);
                AscendC::Min(accLocal, accLocal, hiLocal, 1);

                // sumExp += exp(val)
                AscendC::LocalTensor<float> eLocal = scalarQ_.AllocTensor<float>();
                AscendC::Exp(eLocal, accLocal, 1);
                AscendC::Add(sumExpLocal, sumExpLocal, eLocal, 1);

                scalarQ_.FreeTensor(eLocal);
                scalarQ_.FreeTensor(bLocal);
                scalarQ_.FreeTensor(accLocal);
            }

            // y = log(sumExp)
            AscendC::LocalTensor<float> yLocal = scalarQ_.AllocTensor<float>();
            AscendC::Log(yLocal, sumExpLocal, 1);

            // mish(y) = y * tanh(log(1 + exp(y)))
            AscendC::LocalTensor<float> tmpExp = scalarQ_.AllocTensor<float>();
            AscendC::Exp(tmpExp, yLocal, 1);
            AscendC::LocalTensor<float> onePlus = scalarQ_.AllocTensor<float>();
            AscendC::Add(onePlus, tmpExp, oneLocal, 1);
            AscendC::LocalTensor<float> softplus = scalarQ_.AllocTensor<float>();
            AscendC::Log(softplus, onePlus, 1);
            AscendC::LocalTensor<float> tnh = scalarQ_.AllocTensor<float>();
            AscendC::Tanh(tnh, softplus, 1);
            AscendC::LocalTensor<float> mishLocal = scalarQ_.AllocTensor<float>();
            AscendC::Mul(mishLocal, yLocal, tnh, 1);

            // out = y * mish(y)
            AscendC::LocalTensor<float> outLocal = scalarQ_.AllocTensor<float>();
            AscendC::Mul(outLocal, yLocal, mishLocal, 1);

            // Write result
            AscendC::DataCopy(yGm_[m], outLocal, 1);

            // Free locals for this row
            scalarQ_.FreeTensor(outLocal);
            scalarQ_.FreeTensor(mishLocal);
            scalarQ_.FreeTensor(tnh);
            scalarQ_.FreeTensor(softplus);
            scalarQ_.FreeTensor(onePlus);
            scalarQ_.FreeTensor(tmpExp);
            scalarQ_.FreeTensor(yLocal);
            scalarQ_.FreeTensor(sumExpLocal);
        }

        // Free shared scalars
        scalarQ_.FreeTensor(s1Local);
        scalarQ_.FreeTensor(oneLocal);
        scalarQ_.FreeTensor(hiLocal);
        scalarQ_.FreeTensor(loLocal);
        scalarQ_.FreeTensor(sLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xTileQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> scalingGm_;
    AscendC::GlobalTensor<float> clampMinGm_;
    AscendC::GlobalTensor<float> clampMaxGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t N_ {0};
    uint32_t tileK_ {0};
};

extern "C" __global__ __aicore__ void matmul_scale_residual_add_clamp_log_sum_exp_mish_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR scaling, GM_ADDR clamp_min, GM_ADDR clamp_max,
    GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulScaleResidualAddClampLogSumExpMishCustom op;
    op.Init(x, weight, bias, scaling, clamp_min, clamp_max, y,
            tiling_data.m, tiling_data.k, tiling_data.n, tiling_data.tileK);
    op.Process();
}
