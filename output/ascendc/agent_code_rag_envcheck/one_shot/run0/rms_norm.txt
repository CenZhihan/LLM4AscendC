host_operator_src="""

#include "rms_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous

namespace optiling {

// UB model for this kernel:
//   xTile: 1 * tileP
//   yTile: 1 * tileP
//   denom: 1 * tileP
//   tmp:   1 * tileP
// Total floats = 4 * tileP => bytes = 16 * tileP
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 16384; // conservative to reduce latency variance

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    RmsNormCustomTilingData tiling;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *numFeaturesPtr = attrs->GetAttrPointer<int64_t>(0);
    const float *epsPtr = attrs->GetAttrPointer<float>(1);

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &origin = in_shape->GetOriginShape();
    int64_t dimNum = 0;
#ifdef __GNUC__
    // best-effort API usage; real API provides these in GE
#endif
    dimNum = origin.GetDimNum();

    uint64_t totalLength = static_cast<uint64_t>(origin.GetShapeSize());
    uint64_t N = 1, C = 1, P = 1;

    if (dimNum >= 1) {
        N = static_cast<uint64_t>(origin.GetDim(0));
    }
    if (dimNum >= 2) {
        C = static_cast<uint64_t>(origin.GetDim(1));
    } else {
        C = static_cast<uint64_t>(*numFeaturesPtr);
    }
    if (dimNum >= 3) {
        P = 1;
        for (int i = 2; i < dimNum; ++i) {
            P *= static_cast<uint64_t>(origin.GetDim(i));
        }
    } else {
        // If input is 2D [N, C], then P=1
        P = (totalLength > 0) ? (totalLength / (N * C)) : 1;
        if (P == 0) P = 1;
    }

    if (N * C * P != totalLength) {
        // fallback to consistent partition if shape API is inconsistent
        uint64_t L = totalLength / C;
        N = 1;
        P = L;
    }

    // Determine tileP based on UB budget (4 vectors of length tileP)
    uint32_t maxTileP = UB_BUDGET_BYTES / 16; // bytes per element group
    if (maxTileP > HARD_CAP_ELEMS) maxTileP = HARD_CAP_ELEMS;
    maxTileP = AlignDown(maxTileP, ALIGN_ELEMS);
    if (maxTileP < ALIGN_ELEMS) maxTileP = ALIGN_ELEMS;

    uint32_t pSize = static_cast<uint32_t>(P);
    uint32_t tileP = maxTileP;
    if (pSize < tileP) {
        uint32_t aligned = AlignDown(pSize, ALIGN_ELEMS);
        tileP = (aligned > 0) ? aligned : pSize;
    }

    const uint32_t tilesPerN = (tileP == 0) ? 1u : CeilDiv(pSize, tileP);
    uint32_t totalTiles = static_cast<uint32_t>(N) * tilesPerN;

    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_C(static_cast<uint32_t>(C));
    tiling.set_P(static_cast<uint32_t>(P));
    tiling.set_tileP(tileP);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(*epsPtr);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class RmsNormCustom : public OpDef {
public:
    explicit RmsNormCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("num_features").AttrType(REQUIRED).Int(1);
        this->Attr("eps").AttrType(OPTIONAL).Float(1e-5);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(RmsNormCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t IN_BUFFER_NUM = 1;
constexpr int32_t OUT_BUFFER_NUM = 1;
constexpr int32_t CALC_BUFFER_NUM = 2; // denom + tmp

class KernelRmsNormCustom {
public:
    __aicore__ inline KernelRmsNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t P,
                                uint32_t tileP, float eps)
    {
        N_ = N;
        C_ = C;
        P_ = P;
        tileP_ = tileP;
        eps_ = eps;

        totalLen_ = static_cast<uint64_t>(N_) * static_cast<uint64_t>(C_) * static_cast<uint64_t>(P_);

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalLen_);

        pipe_.InitBuffer(inQueueX_, IN_BUFFER_NUM, tileP_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, OUT_BUFFER_NUM, tileP_ * sizeof(float));
        pipe_.InitBuffer(calcQueue_, CALC_BUFFER_NUM, tileP_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0 || tileP_ == 0) return;
        const uint32_t tilesPerN = (P_ + tileP_ - 1u) / tileP_;
        const uint32_t totalTiles = N_ * tilesPerN;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t n = tileId / tilesPerN;
            const uint32_t tIdx = tileId % tilesPerN;
            const uint32_t pStart = tIdx * tileP_;
            uint32_t curP = P_ - pStart;
            if (curP > tileP_) curP = tileP_;

            ComputeTile(n, pStart, curP);
        }
    }

private:
    __aicore__ inline void ComputeTile(uint32_t n, uint32_t pStart, uint32_t lenP)
    {
        // denom buffer (accumulator)
        AscendC::LocalTensor<float> denom = calcQueue_.AllocTensor<float>();
        AscendC::Dup(denom, 0.0f, lenP);

        // tmp buffer for x^2
        AscendC::LocalTensor<float> tmp = calcQueue_.AllocTensor<float>();

        // First pass: accumulate sum of squares across C
        for (uint32_t c = 0; c < C_; ++c) {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            const uint64_t base = (((uint64_t)n * (uint64_t)C_ + (uint64_t)c) * (uint64_t)P_) + (uint64_t)pStart;
            AscendC::DataCopy(xLocal, xGm_[base], lenP);
            AscendC::Mul(tmp, xLocal, xLocal, lenP);
            AscendC::Add(denom, denom, tmp, lenP);
            inQueueX_.FreeTensor(xLocal);
        }

        // denom = sqrt( (denom / C) + eps )
        const float invC = 1.0f / static_cast<float>(C_);
        AscendC::Muls(denom, denom, invC, lenP);
        AscendC::Adds(denom, denom, eps_, lenP);
        AscendC::Sqrt(denom, denom, lenP);

        // Second pass: normalize each channel
        for (uint32_t c = 0; c < C_; ++c) {
            AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            const uint64_t base = (((uint64_t)n * (uint64_t)C_ + (uint64_t)c) * (uint64_t)P_) + (uint64_t)pStart;
            AscendC::DataCopy(xLocal, xGm_[base], lenP);
            AscendC::Div(yLocal, xLocal, denom, lenP);
            AscendC::DataCopy(yGm_[base], yLocal, lenP);
            outQueueY_.FreeTensor(yLocal);
            inQueueX_.FreeTensor(xLocal);
        }

        calcQueue_.FreeTensor(tmp);
        calcQueue_.FreeTensor(denom);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, CALC_BUFFER_NUM> calcQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0};
    uint32_t C_{0};
    uint32_t P_{0};
    uint32_t tileP_{0};
    uint64_t totalLen_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void rms_norm_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelRmsNormCustom op;
    op.Init(x, y,
            tiling_data.N, tiling_data.C, tiling_data.P,
            tiling_data.tileP, tiling_data.eps);
    op.Process();
}
