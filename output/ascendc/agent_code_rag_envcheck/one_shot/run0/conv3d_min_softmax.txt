kernel_src="""

#include "kernel_operator.h"
#include <cmath>

class KernelConv3dMinSoftmaxCustom {
public:
    __aicore__ inline KernelConv3dMinSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t d, uint32_t h, uint32_t w_in,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t do_, uint32_t ho, uint32_t wo)
    {
        xPtr_ = reinterpret_cast<__gm__ float*>(x);
        wPtr_ = reinterpret_cast<__gm__ float*>(w);
        bPtr_ = reinterpret_cast<__gm__ float*>(b);
        yPtr_ = reinterpret_cast<__gm__ float*>(y);

        N_ = n; Cin_ = cin; Cout_ = cout;
        D_ = d; H_ = h; W_ = w_in;
        Kd_ = kd; Kh_ = kh; Kw_ = kw;
        Do_ = do_; Ho_ = ho; Wo_ = wo;

        xStrideC_ = D_ * H_ * W_;
        xStrideD_ = H_ * W_;
        xStrideH_ = W_;
        xStrideW_ = 1;

        // y layout: [N, C_out, H_out, W_out]
        yStrideC_ = Ho_ * Wo_;
        yStrideH_ = Wo_;
        yStrideW_ = 1;
    }

    __aicore__ inline void Process()
    {
        // Parallelize across (n, ho, wo) points
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t totalPoints = static_cast<uint64_t>(N_) * static_cast<uint64_t>(Ho_) * static_cast<uint64_t>(Wo_);

        for (uint64_t point = blockIdx; point < totalPoints; point += blockNum) {
            uint64_t tmp = point;
            const uint32_t wo = static_cast<uint32_t>(tmp % Wo_); tmp /= Wo_;
            const uint32_t ho = static_cast<uint32_t>(tmp % Ho_); tmp /= Ho_;
            const uint32_t n  = static_cast<uint32_t>(tmp);

            ComputePoint(n, ho, wo);
        }
    }

private:
    __aicore__ inline void ComputePoint(uint32_t n, uint32_t ho, uint32_t wo)
    {
        // Temporary buffer in registers for per-channel values
        float valsMax[256]; // supports up to 256 channels
        const uint32_t C = Cout_;
        for (uint32_t c = 0; c < C; ++c) {
            valsMax[c] = 1.0e30f; // initialize as +inf for min reduction
        }

        // For each output depth (reduction) position
        for (uint32_t od = 0; od < Do_; ++od) {
            // Compute convolution at (od, ho, wo) for each output channel
            for (uint32_t c = 0; c < C; ++c) {
                float acc = 0.0f;

                // Iterate over input channels and kernel 3x3x3
                for (uint32_t ci = 0; ci < Cin_; ++ci) {
                    for (uint32_t kz = 0; kz < Kd_; ++kz) {
                        const uint32_t id = od + kz;
                        for (uint32_t ky = 0; ky < Kh_; ++ky) {
                            const uint32_t ih = ho + ky;
                            for (uint32_t kx = 0; kx < Kw_; ++kx) {
                                const uint32_t iw = wo + kx;

                                // x index: (((n*Cin + ci)*D + id)*H + ih)*W + iw
                                const uint64_t xIdx =
                                    ((((static_cast<uint64_t>(n) * Cin_ + ci) * D_ + id) * H_ + ih) * W_ + iw);

                                // w index: ((((c*Cin + ci)*Kd + kz)*Kh + ky)*Kw + kx)
                                const uint64_t wIdx =
                                    ((((static_cast<uint64_t>(c) * Cin_ + ci) * Kd_ + kz) * Kh_ + ky) * Kw_ + kx);

                                const float xv = xPtr_[xIdx];
                                const float wv = wPtr_[wIdx];
                                acc += xv * wv;
                            }
                        }
                    }
                }

                // Min reduction along depth
                valsMax[c] = (acc < valsMax[c]) ? acc : valsMax[c];
            }
        }

        // Add bias (independent of depth, after min)
        for (uint32_t c = 0; c < C; ++c) {
            valsMax[c] += bPtr_[c];
        }

        // Softmax across channel dimension for this (n, ho, wo)
        float maxVal = valsMax[0];
        for (uint32_t c = 1; c < C; ++c) {
            if (valsMax[c] > maxVal) maxVal = valsMax[c];
        }

        float sumExp = 0.0f;
        for (uint32_t c = 0; c < C; ++c) {
            valsMax[c] = valsMax[c] - maxVal;
            valsMax[c] = expf(valsMax[c]);
            sumExp += valsMax[c];
        }
        const float invSum = 1.0f / sumExp;
        for (uint32_t c = 0; c < C; ++c) {
            valsMax[c] *= invSum;
        }

        // Store to output: y[n, c, ho, wo]
        const uint64_t baseY = ((static_cast<uint64_t>(n) * Cout_) + 0) * yStrideC_
                             + static_cast<uint64_t>(ho) * yStrideH_
                             + static_cast<uint64_t>(wo) * yStrideW_;
        for (uint32_t c = 0; c < C; ++c) {
            yPtr_[baseY + c * yStrideC_] = valsMax[c];
        }
    }

private:
    __gm__ float* xPtr_;
    __gm__ float* wPtr_;
    __gm__ float* bPtr_;
    __gm__ float* yPtr_;

    uint32_t N_, Cin_, Cout_;
    uint32_t D_, H_, W_;
    uint32_t Kd_, Kh_, Kw_;
    uint32_t Do_, Ho_, Wo_;

    // strides for addressing (not all used in scalar addressing but kept for clarity)
    uint32_t xStrideC_, xStrideD_, xStrideH_, xStrideW_;
    uint32_t yStrideC_, yStrideH_, yStrideW_;
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMinSoftmaxCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.do, tiling_data.ho, tiling_data.wo);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv3d_min_softmax_impl_npu(const at::Tensor& x,
                                       const at::Tensor& weight,
                                       const at::Tensor& bias) {
    // Output tensor shape: [N, Cout, H-Kh+1, W-Kw+1], with min over depth applied
    TORCH_CHECK(x.dim() == 5, "x must be 5D [N, C, D, H, W]");
    TORCH_CHECK(weight.dim() == 5, "weight must be 5D [Cout, Cin, Kd, Kh, Kw]");
    TORCH_CHECK(bias.dim() == 1, "bias must be 1D [Cout]");

    const int64_t N   = x.size(0);
    const int64_t Cin = x.size(1);
    const int64_t D   = x.size(2);
    const int64_t H   = x.size(3);
    const int64_t W   = x.size(4);

    const int64_t Cout = weight.size(0);
    const int64_t Kd   = weight.size(2);
    const int64_t Kh   = weight.size(3);
    const int64_t Kw   = weight.size(4);

    TORCH_CHECK(Cin == weight.size(1), "Cin mismatch between x and weight");
    TORCH_CHECK(bias.size(0) == Cout, "bias size must equal Cout");

    const int64_t Ho = H - Kh + 1;
    const int64_t Wo = W - Kw + 1;
    TORCH_CHECK(D - Kd + 1 > 0 && Ho > 0 && Wo > 0, "Invalid shapes for conv3d with stride=1, padding=0, dilation=1");

    at::Tensor y = at::empty({N, Cout, Ho, Wo}, x.options());

    EXEC_NPU_CMD(aclnnConv3dMinSoftmaxCustom, x, weight, bias, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv3d_min_softmax_custom", &conv3d_min_softmax_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv3d_min_softmax_custom", &conv3d_min_softmax_impl_npu,
