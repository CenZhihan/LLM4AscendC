#include "kernel_operator.h"

class KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom {
public:
    __aicore__ inline KernelConvTranspose3dMultiplyMaxGlobalAvgPoolClampCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c,
                                uint32_t d, uint32_t h, uint32_t w)
    {
        n_ = n; c_ = c; d_ = d; h_ = h; w_ = w;
        totalLen_ = (uint64_t)n_ * c_ * d_ * h_ * w_;
        dhw_ = (uint64_t)d_ * h_ * w_;
        // Input tensor: [N, C, D, H, W] contiguous
        xGm_.SetGlobalBuffer((__gm__ float*)x, totalLen_);
        // Output tensor: [N, C, 1, 1, 1] contiguous (size N*C)
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * c_);

        pipe_.InitBuffer(tmpIn_, 1, sizeof(float));  // 1-element buffer for scalar transfers
        pipe_.InitBuffer(tmpOut_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLen_ == 0) { return; }

        // Single-core reduction: compute global max and per-(N,C) average in one pass.
        float globalMax = -AscendC::INFINITY_FLOAT;

        // Iterate per (n, c) to accumulate sums and track global max
        for (uint32_t n = 0; n < n_; ++n) {
            for (uint32_t c = 0; c < c_; ++c) {
                float sum = 0.0f;
                uint64_t baseNc = ((uint64_t)n * c_ + c) * dhw_;
                for (uint32_t z = 0; z < d_; ++z) {
                    for (uint32_t y = 0; y < h_; ++y) {
                        for (uint32_t x = 0; x < w_; ++x) {
                            uint64_t idx = baseNc + ((uint64_t)z * h_ + y) * w_ + x;
                            float val = LoadScalar(idx);
                            // Update sum and global max
                            sum = sum + val;
                            if (val > globalMax) {
                                globalMax = val;
                            }
                        }
                    }
                }
                // Temporarily store unscaled average into y buffer; will finalize after global max known
                float avg = sum / static_cast<float>(dhw_);
                StoreScalar(((uint64_t)n * c_ + c), avg);
            }
        }

        // Second pass on [N, C] to apply scaling by global max and clamp to [0,1].
        const float clampMin = 0.0f;
        const float clampMax = 1.0f;
        for (uint64_t i = 0; i < (uint64_t)n_ * c_; ++i) {
            float avg = LoadScalarOut(i);
            float scaled = avg * globalMax;
            // clamp to [0, 1]
            if (scaled < clampMin) scaled = clampMin;
            if (scaled > clampMax) scaled = clampMax;
            StoreScalar(i, scaled);
        }
    }

private:
    __aicore__ inline float LoadScalar(uint64_t xIndex)
    {
        AscendC::LocalTensor<float> buf = tmpIn_.AllocTensor<float>();
        AscendC::DataCopy(buf, xGm_[xIndex], 1);
        tmpIn_.EnQue(buf);
        AscendC::LocalTensor<float> got = tmpIn_.DeQue<float>();
        float v = got.GetValue(0);
        tmpIn_.FreeTensor(got);
        return v;
    }

    __aicore__ inline float LoadScalarOut(uint64_t yIndex)
    {
        AscendC::LocalTensor<float> buf = tmpIn_.AllocTensor<float>();
        AscendC::DataCopy(buf, yGm_[yIndex], 1);
        tmpIn_.EnQue(buf);
        AscendC::LocalTensor<float> got = tmpIn_.DeQue<float>();
        float v = got.GetValue(0);
        tmpIn_.FreeTensor(got);
        return v;
    }

    __aicore__ inline void StoreScalar(uint64_t yIndex, float v)
    {
        AscendC::LocalTensor<float> buf = tmpOut_.AllocTensor<float>();
        buf.SetValue(0, v);
        tmpOut_.EnQue(buf);
        AscendC::LocalTensor<float> out = tmpOut_.DeQue<float>();
        AscendC::DataCopy(yGm_[yIndex], out, 1);
        tmpOut_.FreeTensor(out);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> tmpIn_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> tmpOut_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{1}, c_{1}, d_{1}, h_{1}, w_{1};
    uint64_t totalLen_{0};
    uint64_t dhw_{0};
};

}
