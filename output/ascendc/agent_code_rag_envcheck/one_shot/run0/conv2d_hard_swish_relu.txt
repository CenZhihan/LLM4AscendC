kernel_src="""

#include "kernel_operator.h"

class KernelConv2dHardSwishReluCustom {
public:
    __aicore__ inline KernelConv2dHardSwishReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t hIn, uint32_t wIn,
                                uint32_t kH, uint32_t kW,
                                uint32_t hOut, uint32_t wOut)
    {
        xPtr_ = (__gm__ float*)x;
        wPtr_ = (__gm__ float*)w;
        bPtr_ = (__gm__ float*)b;
        yPtr_ = (__gm__ float*)y;

        N_ = n; Cin_ = cIn; Cout_ = cOut;
        Hin_ = hIn; Win_ = wIn;
        kH_ = kH; kW_ = kW;
        Hout_ = hOut; Wout_ = wOut;
        hasBias_ = (b != nullptr);
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || Hout_ == 0 || Wout_ == 0) {
            return;
        }
        const uint32_t totalPos = N_ * Hout_ * Wout_;
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t pos = blockIdx; pos < totalPos; pos += blockNum) {
            // decode n,h,w from linear index over (N, Hout, Wout)
            uint32_t tmp = pos;
            const uint32_t n = tmp / (Hout_ * Wout_);
            tmp -= n * (Hout_ * Wout_);
            const uint32_t h = tmp / Wout_;
            const uint32_t w = tmp - h * Wout_;

            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                float acc = 0.0f;
                if (hasBias_) {
                    acc = bPtr_[oc];
                }
                // convolution inner product
                for (uint32_t ic = 0; ic < Cin_; ++ic) {
                    for (uint32_t kh = 0; kh < kH_; ++kh) {
                        const uint32_t ih = h + kh;
                        const uint32_t baseInHW = ((n * Cin_ + ic) * Hin_ + ih) * Win_;
                        const uint32_t baseW = (((oc * Cin_ + ic) * kH_) + kh) * kW_;
                        for (uint32_t kw = 0; kw < kW_; ++kw) {
                            const uint32_t iw = w + kw;
                            const float xv = xPtr_[baseInHW + iw];
                            const float ww = wPtr_[baseW + kw];
                            acc = AscendC::Fma(xv, ww, acc);  // acc += xv * ww
                        }
                    }
                }

                // HardSwish then ReLU:
                // hs = acc * clamp(acc + 3, 0, 6) / 6; y = max(hs, 0)
                float t = acc + 3.0f;
                if (t < 0.0f) t = 0.0f;
                if (t > 6.0f) t = 6.0f;
                float hs = (acc * t) * (1.0f / 6.0f);
                if (hs < 0.0f) hs = 0.0f;

                const uint32_t outIndex = ((n * Cout_ + oc) * Hout_ + h) * Wout_ + w;
                yPtr_[outIndex] = hs;
            }
        }
    }

private:
    __gm__ float *xPtr_{nullptr};
    __gm__ float *wPtr_{nullptr};
    __gm__ float *bPtr_{nullptr};
    __gm__ float *yPtr_{nullptr};

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Hin_{0}, Win_{0};
    uint32_t kH_{0}, kW_{0};
    uint32_t Hout_{0}, Wout_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_hard_swish_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dHardSwishReluCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.n, tiling_data.cIn, tiling_data.cOut,
            tiling_data.hIn, tiling_data.wIn,
            tiling_data.kH, tiling_data.kW,
            tiling_data.hOut, tiling_data.wOut);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor alloc_output_like_conv2d(const at::Tensor& x, const at::Tensor& w) {
    // x: [N, Cin, H, W], w: [Cout, Cin, kH, kW]
    TORCH_CHECK(x.dim() == 4, \"x must be 4D NCHW\");
    TORCH_CHECK(w.dim() == 4, \"weight must be 4D Cout,Cin,kH,kW\");
    auto N = x.size(0);
    auto H = x.size(2);
    auto W = x.size(3);
    auto Cout = w.size(0);
    auto kH = w.size(2);
    auto kW = w.size(3);
    TORCH_CHECK(H >= kH && W >= kW, \"Input spatial dims must be >= kernel size with padding=0\");
    auto Hout = H - kH + 1;
    auto Wout = W - kW + 1;
    return at::empty({N, Cout, Hout, Wout}, x.options());
}

at::Tensor conv2d_hard_swish_relu_custom_impl_npu(const at::Tensor& x,
                                                  const at::Tensor& weight,
                                                  c10::optional<at::Tensor> bias_opt) {
    at::Tensor y = alloc_output_like_conv2d(x, weight);
    if (bias_opt.has_value()) {
        const at::Tensor& b = bias_opt.value();
        EXEC_NPU_CMD(aclnnConv2dHardSwishReluCustom, x, weight, b, y);
    } else {
        at::Tensor empty_bias; // undefined tensor indicates no bias
        EXEC_NPU_CMD(aclnnConv2dHardSwishReluCustom, x, weight, empty_bias, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv2d_hard_swish_relu_custom\", &conv2d_hard_swish_relu_custom_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"conv2d_hard_swish_relu_custom\", &conv2d_hard_swish_relu_custom_impl_npu,
          \"Conv2D + HardSwish + ReLU fusion (Conv2dHardSwishReluCustom on NPU)\");
}
