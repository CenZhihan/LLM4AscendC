static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForLowerTriangularMatricesCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    if (a_shape->GetOriginShape().GetDimNum() != 2 ||
        b_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t a_m = a_shape->GetOriginShape().GetDim(0);
    const uint64_t a_k = a_shape->GetOriginShape().GetDim(1);
    const uint64_t b_k = b_shape->GetOriginShape().GetDim(0);
    const uint64_t b_n = b_shape->GetOriginShape().GetDim(1);

    if (a_m != a_k || b_k != b_n || a_k != b_k) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(a_m);

    // Choose a conservative blockDim for better occupancy on common 910B configs.
    uint32_t blockDim = (N >= 384) ? 24u : ((N >= 128) ? 16u : ((N >= 32) ? 8u : 1u));
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace_sizes = context->GetWorkspaceSizes(1);
    workspace_sizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape->GetOriginShape().GetDimNum() != 2 ||
        b_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    if (a_shape->GetOriginShape().GetDim(1) != b_shape->GetOriginShape().GetDim(0)) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, a_shape->GetOriginShape().GetDim(0));
    c_shape->SetDim(1, b_shape->GetOriginShape().GetDim(1));
    return GRAPH_SUCCESS;
}

{
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

using half = AscendC::half;

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tilingGm)
{
    if (g_coreType == AscendC::AIV) {
        return;
    }

    // Prepare tiling (Triangular MatMul uses built-in tiling under MatmulCustom)
    TCubeTiling tiling;
    CopyTiling(&tiling, tilingGm);

    AscendC::TPipe pipe;

    // Use triangular matmul helper that computes C = tril(A @ B) for lower-triangular A and B.
    // A and B: half, C: float accumulation/output.
    MatmulCustom::MatmulTriangleKernel<half, half, float, float> triKernel;
    // Bias not used; pass nullptr.
    triKernel.Init(a, b, nullptr, c, tiling);
    triKernel.Process(&pipe);
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static at::Tensor matmul_lower_tri_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Cast to half for compute, accumulate/output as float32
    at::Tensor A_half = A.to(at::kHalf);
    at::Tensor B_half = B.to(at::kHalf);
    // Output as float32 to preserve accuracy
    auto out_sizes = A.sizes().vec();
    at::Tensor C = at::empty(out_sizes, A.options().dtype(at::kFloat));
    EXEC_NPU_CMD(aclnnMatmulForLowerTriangularMatricesCustom, A_half, B_half, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_for_lower_triangular_matrices_custom", &matmul_lower_tri_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_for_lower_triangular_matrices_custom", &matmul_lower_tri_impl_npu,
          "Lower-triangular MatMul (MatmulForLowerTriangularMatricesCustom on NPU)");
}
