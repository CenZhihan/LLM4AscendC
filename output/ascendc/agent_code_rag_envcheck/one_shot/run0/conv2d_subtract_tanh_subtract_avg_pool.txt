#include "kernel_operator.h"

using namespace AscendC;

class KernelConv2dSubtractTanhSubtractAvgPoolCustom {
public:
    __aicore__ inline KernelConv2dSubtractTanhSubtractAvgPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W, uint32_t K, uint32_t Pool,
                                uint32_t HOut, uint32_t WOut,
                                uint32_t HPooled, uint32_t WPooled,
                                uint32_t HasBias, float Sub1, float Sub2)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        H_ = H; W_ = W; K_ = K; Pool_ = Pool;
        HOut_ = HOut; WOut_ = WOut; HPooled_ = HPooled; WPooled_ = WPooled;
        hasBias_ = (HasBias != 0);
        sub1_ = Sub1; sub2_ = Sub2;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * Cin_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, Cout_ * Cin_ * K_ * K_);
        if (b != nullptr) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * Cout_ * HPooled_ * WPooled_);

        pipe_.InitBuffer(wBuf_, 1, Cin_ * K_ * K_ * sizeof(float));
        pipe_.InitBuffer(xRowBuf_, 1, K_ * sizeof(float));
        pipe_.InitBuffer(outScalarBuf_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cin_ == 0 || Cout_ == 0 || HPooled_ == 0 || WPooled_ == 0) {
            return;
        }
        uint32_t blocks = static_cast<uint32_t>(GetBlockNum());
        uint32_t bid = static_cast<uint32_t>(GetBlockIdx());
        const uint32_t totalPairs = N_ * Cout_;

        for (uint32_t pair = bid; pair < totalPairs; pair += blocks) {
            const uint32_t n = pair / Cout_;
            const uint32_t co = pair % Cout_;

            LocalTensor<float> wLocal = wBuf_.AllocTensor<float>();
            // Load weights for output channel co: contiguous block of size Cin*K*K
            const uint32_t wLen = Cin_ * K_ * K_;
            const uint32_t wBase = co * wLen;
            DataCopy(wLocal, wGm_[wBase], wLen);

            float bVal = 0.0f;
            if (hasBias_) {
                // Load bias scalar for channel co
                LocalTensor<float> bLocal = outScalarBuf_.AllocTensor<float>();
                DataCopy(bLocal, bGm_[co], 1);
                bVal = bLocal.GetValue(0);
            }

            // Iterate pooled output
            for (uint32_t ph = 0; ph < HPooled_; ++ph) {
                for (uint32_t pw = 0; pw < WPooled_; ++pw) {
                    float poolAcc = 0.0f;

                    for (uint32_t sh = 0; sh < Pool_; ++sh) {
                        for (uint32_t sw = 0; sw < Pool_; ++sw) {
                            const uint32_t oh = ph * Pool_ + sh;
                            const uint32_t ow = pw * Pool_ + sw;

                            float acc = bVal;

                            // Convolution sum for position (oh, ow)
                            for (uint32_t ci = 0; ci < Cin_; ++ci) {
                                for (uint32_t kh = 0; kh < K_; ++kh) {
                                    // Copy one row of K elements from GM to UB
                                    const uint32_t h_in = oh + kh;
                                    const uint32_t w_in0 = ow;
                                    const uint64_t xBase = IndexX(n, ci, h_in, w_in0);
                                    LocalTensor<float> xRow = xRowBuf_.AllocTensor<float>();
                                    DataCopy(xRow, xGm_[xBase], K_);

                                    const uint32_t wRowBase = (ci * K_ + kh) * K_;
                                    for (uint32_t kw = 0; kw < K_; ++kw) {
                                        const float xv = xRow.GetValue(kw);
                                        const float ww = wLocal.GetValue(wRowBase + kw);
                                        acc += xv * ww;
                                    }
                                }
                            }

                            // Fused elementwise: subtract1 -> tanh -> subtract2
                            acc -= sub1_;
                            acc = fast_tanhf(acc);
                            acc -= sub2_;

                            poolAcc += acc;
                        }
                    }

                    const float norm = 1.0f / static_cast<float>(Pool_ * Pool_);
                    const float yVal = poolAcc * norm;

                    // Store single scalar to GM
                    LocalTensor<float> yLocal = outScalarBuf_.AllocTensor<float>();
                    yLocal.SetValue(0, yVal);
                    const uint64_t yOff = IndexY(n, co, ph, pw);
                    DataCopy(yGm_[yOff], yLocal, 1);
                }
            }

            wBuf_.FreeTensor(wLocal);
        }
    }

private:
    __aicore__ inline uint64_t IndexX(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * Cin_ + c) * H_ + h) * W_ + w;
    }

    __aicore__ inline uint64_t IndexY(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * Cout_ + c) * HPooled_ + h) * WPooled_ + w;
    }

    __aicore__ inline float fast_tanhf(float x) const
    {
        // Rational approximation tanh(x) ~ x*(27 + x^2) / (27 + 9*x^2), clamped to [-1, 1]
        float x2 = x * x;
        float num = x * (27.0f + x2);
        float den = 27.0f + 9.0f * x2;
        float t = num / den;
        if (t > 1.0f) t = 1.0f;
        if (t < -1.0f) t = -1.0f;
        return t;
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECCALC, 1> wBuf_;
    TQue<TPosition::VECCALC, 1> xRowBuf_;
    TQue<TPosition::VECCALC, 1> outScalarBuf_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, K_{0}, Pool_{1};
    uint32_t HOut_{0}, WOut_{0}, HPooled_{0}, WPooled_{0};
    bool hasBias_{false};
    float sub1_{0.0f}, sub2_{0.0f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dSubtractTanhSubtractAvgPoolCustom op;
    op.Init(
        x, weight, conv_bias, y,
        tiling_data.N, tiling_data.Cin, tiling_data.Cout,
        tiling_data.H, tiling_data.W, tiling_data.K, tiling_data.Pool,
        tiling_data.HOut, tiling_data.WOut,
        tiling_data.HPooled, tiling_data.WPooled,
        tiling_data.HasBias, tiling_data.Sub1, tiling_data.Sub2
    );
    op.Process();
}
