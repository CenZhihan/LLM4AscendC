host_operator_src="""

#include "conv_pointwise_2d_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;            // DMA-friendly
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvPointwise2dCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const auto &x_os = x_shape->GetOriginShape();
    const auto &w_os = w_shape->GetOriginShape();

    const uint32_t n   = static_cast<uint32_t>(x_os.GetDim(0));
    const uint32_t cIn = static_cast<uint32_t>(x_os.GetDim(1));
    const uint32_t h   = static_cast<uint32_t>(x_os.GetDim(2));
    const uint32_t w   = static_cast<uint32_t>(x_os.GetDim(3));

    const uint32_t cOut = static_cast<uint32_t>(w_os.GetDim(0));
    const uint32_t kIn  = static_cast<uint32_t>(w_os.GetDim(1));
    // sanity defaults if shapes mismatch
    const uint32_t cinEff = (kIn == cIn) ? cIn : kIn;

    const uint32_t HW = h * w;

    // UB usage: 2*tileP (x vec + y vec) + cIn (weight row) floats => bytes = 4*(2*tileP + cIn)
    uint32_t maxTileP = 0;
    if (UB_BUDGET_BYTES > 4u * cinEff) {
        maxTileP = (UB_BUDGET_BYTES - 4u * cinEff) / 8u; // elements
    }
    maxTileP = AlignDown(maxTileP, ALIGN_ELEMS);
    if (maxTileP < ALIGN_ELEMS) maxTileP = ALIGN_ELEMS;
    if (maxTileP > HARD_CAP_ELEMS) maxTileP = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);
    uint32_t tileP = maxTileP;
    if (HW < tileP) {
        const uint32_t aligned = AlignDown(HW, ALIGN_ELEMS);
        tileP = (aligned > 0) ? aligned : HW;
    }

    // Block mapping: each block processes a subset of output channels.
    uint32_t blockDim = cOut;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(n);
    tiling.set_cIn(cinEff);
    tiling.set_cOut(cOut);
    tiling.set_h(h);
    tiling.set_w(w);
    tiling.set_tileP(tileP);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvPointwise2dCustom {
public:
    __aicore__ inline KernelConvPointwise2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t h, uint32_t wi, uint32_t tileP)
    {
        n_ = n;
        cIn_ = cIn;
        cOut_ = cOut;
        h_ = h;
        w_ = wi;
        HW_ = h_ * w_;
        tileP_ = tileP;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(n_) * cIn_ * HW_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(cOut_) * cIn_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(n_) * cOut_ * HW_);

        pipe_.InitBuffer(xQueue_, BUFFER_NUM, tileP_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, BUFFER_NUM, tileP_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileP_ * sizeof(float));
        pipe_.InitBuffer(wRowQueue_, BUFFER_NUM, cIn_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || cIn_ == 0 || cOut_ == 0 || HW_ == 0 || tileP_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each block handles a grid-stride subset of output channels.
        for (uint32_t oc = blockIdx; oc < cOut_; oc += blockNum) {
            // Load weight row for this output channel once
            AscendC::LocalTensor<float> wLocal = wRowQueue_.AllocTensor<float>();
            AscendC::DataCopy(wLocal, wGm_[static_cast<uint64_t>(oc) * cIn_], cIn_);

            for (uint32_t n = 0; n < n_; ++n) {
                for (uint32_t pBase = 0; pBase < HW_; pBase += tileP_) {
                    uint32_t curP = HW_ - pBase;
                    if (curP > tileP_) curP = tileP_;

                    AscendC::LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                    AscendC::Duplicate(yLocal, 0.0f, curP);

                    for (uint32_t k = 0; k < cIn_; ++k) {
                        // Load a vector of X for this input channel across current spatial tile
                        AscendC::LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                        const uint64_t xOff = static_cast<uint64_t>(n) * cIn_ * HW_
                                            + static_cast<uint64_t>(k) * HW_
                                            + pBase;
                        AscendC::DataCopy(xLocal, xGm_[xOff], curP);

                        // Multiply by corresponding weight scalar and accumulate
                        float wScalar = wLocal.GetValue(k);
                        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                        AscendC::Muls(tmpLocal, xLocal, wScalar, curP);
                        AscendC::Add(yLocal, yLocal, tmpLocal, curP);

                        tmpQueue_.FreeTensor(tmpLocal);
                        xQueue_.FreeTensor(xLocal);
                    }

                    // Store result back to Y
                    const uint64_t yOff = static_cast<uint64_t>(n) * cOut_ * HW_
                                        + static_cast<uint64_t>(oc) * HW_
                                        + pBase;
                    AscendC::DataCopy(yGm_[yOff], yLocal, curP);
                    yQueue_.FreeTensor(yLocal);
                }
            }

            wRowQueue_.FreeTensor(wLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> wRowQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_ {0};
    uint32_t cIn_ {0};
    uint32_t cOut_ {0};
    uint32_t h_ {0};
    uint32_t w_ {0};
    uint32_t HW_ {0};
    uint32_t tileP_ {0};
};

extern "C" __global__ __aicore__ void conv_pointwise_2d_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvPointwise2dCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cIn, tiling_data.cOut,
            tiling_data.h, tiling_data.w, tiling_data.tileP);
    op.Process();
}
