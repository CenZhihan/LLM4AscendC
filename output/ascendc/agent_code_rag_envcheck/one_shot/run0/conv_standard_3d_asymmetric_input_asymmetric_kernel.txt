namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard3dAsymmetricInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const auto &x_origin = x_shape->GetOriginShape();
    const auto &w_origin = w_shape->GetOriginShape();

    const uint32_t N  = static_cast<uint32_t>(x_origin.GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_origin.GetDim(1));
    const uint32_t Di = static_cast<uint32_t>(x_origin.GetDim(2));
    const uint32_t Hi = static_cast<uint32_t>(x_origin.GetDim(3));
    const uint32_t Wi = static_cast<uint32_t>(x_origin.GetDim(4));

    const uint32_t Cout = static_cast<uint32_t>(w_origin.GetDim(0));
    const uint32_t Kd   = static_cast<uint32_t>(w_origin.GetDim(2));
    const uint32_t Kh   = static_cast<uint32_t>(w_origin.GetDim(3));
    const uint32_t Kw   = static_cast<uint32_t>(w_origin.GetDim(4));

    // stride = (1,1,1), padding = (0,0,0), dilation = (1,1,1)
    const uint32_t Do = Di >= Kd ? (Di - Kd + 1u) : 0u;
    const uint32_t Ho = Hi >= Kh ? (Hi - Kh + 1u) : 0u;
    const uint32_t Wo = Wi >= Kw ? (Wi - Kw + 1u) : 0u;

    // UB budgeting: lightweight, as kernel uses direct GM reads for simplicity.
    // Still provide a conservative tile length used for output buffering hints.
    static constexpr uint32_t UB_BUDGET_BYTES = 160 * 1024;
    static constexpr uint32_t ALIGN_ELEMS = 1024;
    const uint32_t kLen = Cin * Kd * Kh * Kw;

    uint32_t maxTile = (UB_BUDGET_BYTES / 4u);
    if (maxTile > 0) {
        // Reserve room conceptually for some scratch space relative to kernel size
        if (maxTile > (kLen * 2u)) {
            maxTile -= (kLen * 2u);
        } else {
            maxTile = ALIGN_ELEMS;
        }
    }
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;

    const uint64_t totalPos = static_cast<uint64_t>(N) * Do * Ho * Wo;
    uint32_t tileLen = maxTile;
    if (totalPos < tileLen) {
        uint32_t aligned = AlignDown(static_cast<uint32_t>(totalPos), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(totalPos);
    }

    uint32_t blockDim = Cout;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_Di(Di);
    tiling.set_Hi(Hi);
    tiling.set_Wi(Wi);
    tiling.set_Do(Do);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_Kd(Kd);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);
    tiling.set_blockDim(blockDim);
    tiling.set_tileLen(tileLen);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelConv3DStd {
public:
    __aicore__ inline KernelConv3DStd() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Di, uint32_t Hi, uint32_t Wi,
                                uint32_t Do, uint32_t Ho, uint32_t Wo,
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                uint32_t /*tileLen*/)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Di_ = Di; Hi_ = Hi; Wi_ = Wi;
        Do_ = Do; Ho_ = Ho; Wo_ = Wo;
        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;

        xPtr_ = (__gm__ float*)x;
        wPtr_ = (__gm__ float*)w;
        yPtr_ = (__gm__ float*)y;
    }

    __aicore__ inline void Process()
    {
        if (Do_ == 0 || Ho_ == 0 || Wo_ == 0) {
            return;
        }

        const uint32_t oc = static_cast<uint32_t>(AscendC::GetBlockIdx());
        if (oc >= Cout_) return;

        for (uint32_t n = 0; n < N_; ++n) {
            for (uint32_t od = 0; od < Do_; ++od) {
                for (uint32_t oh = 0; oh < Ho_; ++oh) {
                    for (uint32_t ow = 0; ow < Wo_; ++ow) {
                        float acc = 0.0f;
                        // Convolution window
                        for (uint32_t c = 0; c < Cin_; ++c) {
                            for (uint32_t kd = 0; kd < Kd_; ++kd) {
                                const uint32_t id = od + kd;  // stride=1, pad=0
                                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                                    const uint32_t ih = oh + kh;
                                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                        const uint32_t iw = ow + kw;

                                        const uint64_t xIndex =
                                            (((uint64_t)n * Cin_ + c) * Di_ + id) * Hi_ * Wi_ +
                                            ((uint64_t)ih * Wi_ + iw);

                                        const uint64_t wIndex =
                                            (((uint64_t)oc * Cin_ + c) * Kd_ + kd) * Kh_ * Kw_ +
                                            ((uint64_t)kh * Kw_ + kw);

                                        acc += xPtr_[xIndex] * wPtr_[wIndex];
                                    }
                                }
                            }
                        }

                        const uint64_t yIndex =
                            (((uint64_t)n * Cout_ + oc) * Do_ + od) * Ho_ * Wo_ +
                            ((uint64_t)oh * Wo_ + ow);

                        yPtr_[yIndex] = acc;
                    }
                }
            }
        }
    }

private:
    __gm__ float* xPtr_ {nullptr};
    __gm__ float* wPtr_ {nullptr};
    __gm__ float* yPtr_ {nullptr};

    uint32_t N_ {0}, Cin_ {0}, Cout_ {0};
    uint32_t Di_ {0}, Hi_ {0}, Wi_ {0};
    uint32_t Do_ {0}, Ho_ {0}, Wo_ {0};
    uint32_t Kd_ {0}, Kh_ {0}, Kw_ {0};
};

extern "C" __global__ __aicore__ void conv_standard_3d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3DStd op;
    op.Init(x, weight, y,
            tiling_data.N, tiling_data.Cin, tiling_data.Cout,
            tiling_data.Di, tiling_data.Hi, tiling_data.Wi,
            tiling_data.Do, tiling_data.Ho, tiling_data.Wo,
            tiling_data.Kd, tiling_data.Kh, tiling_data.Kw,
            tiling_data.tileLen);
    op.Process();
}
