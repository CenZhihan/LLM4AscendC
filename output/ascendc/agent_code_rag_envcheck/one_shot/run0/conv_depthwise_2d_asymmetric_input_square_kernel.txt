kernel_src="""
#include "kernel_operator.h"

class KernelConvDepthwise2dAsymInSq {
public:
    __aicore__ inline KernelConvDepthwise2dAsymInSq() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w_in,
                                uint32_t out_h, uint32_t out_w)
    {
        n_ = n; c_ = c; h_ = h; w_ = w_in; oh_ = out_h; ow_ = out_w;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)n_ * c_ * h_ * w_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)c_ * 1 * 3 * 3);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)n_ * c_ * oh_ * ow_);

        const uint32_t rowBytes = w_ * sizeof(float);
        const uint32_t outRowBytes = ow_ * sizeof(float);

        pipe_.InitBuffer(rowBufQ_, 3, rowBytes);       // three rolling input rows
        pipe_.InitBuffer(outRowQ_, 1, outRowBytes);    // one output row buffer
        pipe_.InitBuffer(tmpQ_, 1, outRowBytes);       // temp vector for shifted columns
        pipe_.InitBuffer(filtQ_, 1, 16 * sizeof(float)); // filter cache (allocate 16 for alignment)
    }

    __aicore__ inline void Process()
    {
        const uint32_t totalNC = n_ * c_;
        if (totalNC == 0 || ow_ == 0 || oh_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        for (uint32_t idx = blockIdx; idx < totalNC; idx += blockNum) {
            const uint32_t n = idx / c_;
            const uint32_t ch = idx % c_;

            const uint64_t in_base_nc  = ((uint64_t)n * c_ + ch) * h_ * w_;
            const uint64_t out_base_nc = ((uint64_t)n * c_ + ch) * oh_ * ow_;
            const uint64_t w_base_c    = (uint64_t)ch * 9U;

            // Allocate local buffers
            AscendC::LocalTensor<float> rowA = rowBufQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> rowB = rowBufQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> rowC = rowBufQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> outRow = outRowQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> tmpVec = tmpQ_.AllocTensor<float>();
            AscendC::LocalTensor<float> filt = filtQ_.AllocTensor<float>();

            // Load filter 3x3 for this channel
            AscendC::DataCopy(filt, wGm_[w_base_c], 9);

            // Preload first two rows
            AscendC::DataCopy(rowA, xGm_[in_base_nc + 0U * w_], w_);
            AscendC::DataCopy(rowB, xGm_[in_base_nc + 1U * w_], w_);

            for (uint32_t oh = 0; oh < oh_; ++oh) {
                // Load next required row
                const uint32_t ih2 = oh + 2U;
                AscendC::DataCopy(rowC, xGm_[in_base_nc + (uint64_t)ih2 * w_], w_);

                // Zero out the output row
                AscendC::Duplicate(outRow, 0.0f, ow_);

                // Accumulate over 3x3 neighborhood
                // r = 0
                {
                    float k00 = filt.GetValue(0);
                    float k01 = filt.GetValue(1);
                    float k02 = filt.GetValue(2);

                    AscendC::DataCopy(tmpVec, rowA[0], ow_);
                    AscendC::Muls(tmpVec, tmpVec, k00, ow_);
                    AscendC::Add(outRow, outRow, tmpVec, ow_);

                    AscendC::DataCopy(tmpVec, rowA[1], ow_);
                    AscendC::Muls(tmpVec, tmpVec, k01, ow_);
                    AscendC::Add(outRow, outRow, tmpVec, ow_);

                    AscendC::DataCopy(tmpVec, rowA[2], ow_);
                    AscendC::Muls(tmpVec, tmpVec, k02, ow_);
                    AscendC::Add(outRow, outRow, tmpVec, ow_);
                }

                // r = 1
                {
                    float k10 = filt.GetValue(3);
                    float k11 = filt.GetValue(4);
                    float k12 = filt.GetValue(5);

                    AscendC::DataCopy(tmpVec, rowB[0], ow_);
                    AscendC::Muls(tmpVec, tmpVec, k10, ow_);
                    AscendC::Add(outRow, outRow, tmpVec, ow_);

                    AscendC::DataCopy(tmpVec, rowB[1], ow_);
                    AscendC::Muls(tmpVec, tmpVec, k11, ow_);
                    AscendC::Add(outRow, outRow, tmpVec, ow_);

                    AscendC::DataCopy(tmpVec, rowB[2], ow_);
                    AscendC::Muls(tmpVec, tmpVec, k12, ow_);
                    AscendC::Add(outRow, outRow, tmpVec, ow_);
                }

                // r = 2
                {
                    float k20 = filt.GetValue(6);
                    float k21 = filt.GetValue(7);
                    float k22 = filt.GetValue(8);

                    AscendC::DataCopy(tmpVec, rowC[0], ow_);
                    AscendC::Muls(tmpVec, tmpVec, k20, ow_);
                    AscendC::Add(outRow, outRow, tmpVec, ow_);

                    AscendC::DataCopy(tmpVec, rowC[1], ow_);
                    AscendC::Muls(tmpVec, tmpVec, k21, ow_);
                    AscendC::Add(outRow, outRow, tmpVec, ow_);

                    AscendC::DataCopy(tmpVec, rowC[2], ow_);
                    AscendC::Muls(tmpVec, tmpVec, k22, ow_);
                    AscendC::Add(outRow, outRow, tmpVec, ow_);
                }

                // Store output row
                AscendC::DataCopy(yGm_[out_base_nc + (uint64_t)oh * ow_], outRow, ow_);

                // Roll the window: A <- B, B <- C, C reused
                AscendC::LocalTensor<float> t = rowA;
                rowA = rowB;
                rowB = rowC;
                rowC = t;
            }

            // Free buffers
            filtQ_.FreeTensor(filt);
            tmpQ_.FreeTensor(tmpVec);
            outRowQ_.FreeTensor(outRow);
            rowBufQ_.FreeTensor(rowA);
            rowBufQ_.FreeTensor(rowB);
            rowBufQ_.FreeTensor(rowC);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 3> rowBufQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> filtQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0}, oh_{0}, ow_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2dAsymInSq op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.out_h, tiling_data.out_w);
    op.Process();
}
