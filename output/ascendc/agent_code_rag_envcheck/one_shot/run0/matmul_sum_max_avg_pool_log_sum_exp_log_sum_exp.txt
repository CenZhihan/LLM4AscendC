    const uint32_t inFeatures = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));

    //   tUb:    TK floats
    //   wRowUb: TK floats

    //   mulUb:  TK floats
    //   accUb:  TR floats
    //   bUb:    JO floats
    // Total bytes ~= 4 * TK * 4 + TR * 4 + JO * 4
    // Choose TK, TR, JO so that total < UB_BUDGET_BYTES conservatively.

    if (tileK > inFeatures) tileK = inFeatures;
    tileK = AlignDown(tileK, ALIGN_K);
    if (tileK == 0) tileK = AlignUp(inFeatures, ALIGN_K); // handle small K

    if (tileRows > batch) tileRows = batch == 0 ? 1u : batch;

    // UB bytes estimate:
    // mainVecs = 4 * tileK * 4; acc = tileRows * 4; bUb = tileOut * 4
    uint64_t estBytes = static_cast<uint64_t>(4ull * tileK * 4ull + tileRows * 4ull + tileOut * 4ull);

#include "kernel_operator.h"

constexpr int32_t BUF_NUM = 2;

class KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom {
public:
    __aicore__ inline KernelMatmulSumMaxAvgPoolLogSumExpLogSumExpCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t batch, uint32_t inFeatures, uint32_t outFeatures,
                                uint32_t tileK, uint32_t tileRows, uint32_t tileOut)
    {
        N_ = batch;
        K_ = inFeatures;
        O_ = outFeatures;
        TK_ = tileK == 0 ? inFeatures : tileK;
        TR_ = tileRows == 0 ? 1u : tileRows;
        TO_ = tileOut == 0 ? 16u : tileOut;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(N_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(O_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_);

        pipe_.InitBuffer(tQue_, 1, TK_ * sizeof(float));
        pipe_.InitBuffer(wRowQue_, 1, TK_ * sizeof(float));
        pipe_.InitBuffer(xQue_, 1, TK_ * sizeof(float));
        pipe_.InitBuffer(mulQue_, 1, TK_ * sizeof(float));
        pipe_.InitBuffer(accQue_, 1, TR_ * sizeof(float));
        pipe_.InitBuffer(bQue_, 1, TO_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || K_ == 0 || O_ == 0) { return; }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t rowStride = blockNum * TR_;
        uint32_t baseRow = blockIdx * TR_;
        if (baseRow >= N_) return;

        while (baseRow < N_) {
            uint32_t curRows = N_ - baseRow;
            if (curRows > TR_) curRows = TR_;

            AscendC::LocalTensor<float> accLocal = accQue_.AllocTensor<float>();
            AscendC::Dup(accLocal, 0.0f, curRows);

            for (uint32_t j0 = 0; j0 < O_; j0 += TO_) {
                uint32_t joLen = O_ - j0;
                if (joLen > TO_) joLen = TO_;

                // Load b tile
                AscendC::LocalTensor<float> bLocal = bQue_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[j0], joLen);

                for (uint32_t k0 = 0; k0 < K_; k0 += TK_) {
                    uint32_t tkLen = K_ - k0;
                    if (tkLen > TK_) tkLen = TK_;

                    // tUb = sum_{j in tile} w[j, k0:k0+tkLen]
                    AscendC::LocalTensor<float> tLocal = tQue_.AllocTensor<float>();
                    AscendC::Dup(tLocal, 0.0f, tkLen);

                    for (uint32_t joff = 0; joff < joLen; ++joff) {
                        AscendC::LocalTensor<float> wRowLocal = wRowQue_.AllocTensor<float>();
                        AscendC::DataCopy(wRowLocal, wGm_[ (j0 + joff) * K_ + k0 ], tkLen);
                        AscendC::Add(tLocal, tLocal, wRowLocal, tkLen);
                        wRowQue_.FreeTensor(wRowLocal);
                    }

                    for (uint32_t r = 0; r < curRows; ++r) {
                        AscendC::LocalTensor<float> xLocal = xQue_.AllocTensor<float>();
                        AscendC::DataCopy(xLocal, xGm_[ (baseRow + r) * K_ + k0 ], tkLen);

                        AscendC::LocalTensor<float> mulLocal = mulQue_.AllocTensor<float>();
                        AscendC::Mul(mulLocal, xLocal, tLocal, tkLen);

                        ReduceToScalarInPlace(mulLocal, tkLen);

                        AscendC::LocalTensor<float> accElem = accLocal[r];
                        AscendC::Add(accElem, accElem, mulLocal, 1);

                        mulQue_.FreeTensor(mulLocal);
                        xQue_.FreeTensor(xLocal);
                    }

                    tQue_.FreeTensor(tLocal);
                }

                // Reduce bLocal to scalar bSumTile and add to each row accumulator
                ReduceToScalarInPlace(bLocal, joLen);
                for (uint32_t r = 0; r < curRows; ++r) {
                    AscendC::LocalTensor<float> accElem = accLocal[r];
                    AscendC::Add(accElem, accElem, bLocal, 1);
                }
                bQue_.FreeTensor(bLocal);
            }

            // Write out curRows results
            AscendC::DataCopy(yGm_[baseRow], accLocal, curRows);
            accQue_.FreeTensor(accLocal);

            baseRow += rowStride;
        }
    }

private:
    __aicore__ inline void ReduceToScalarInPlace(AscendC::LocalTensor<float>& vec, uint32_t len)
    {
        if (len == 0) return;
        uint32_t cur = len;
        while (cur > 1) {
            uint32_t half = cur >> 1;
            uint32_t pair = half << 1;
            if (half > 0) {
                AscendC::LocalTensor<float> dst = vec;
                AscendC::LocalTensor<float> src1 = vec;
                AscendC::LocalTensor<float> src2 = vec[half];
                AscendC::Add(dst, src1, src2, half);
            }
            if (cur > pair) {
                // handle tail element
                AscendC::LocalTensor<float> dst = vec;
                AscendC::LocalTensor<float> src1 = vec;
                AscendC::LocalTensor<float> src2 = vec[pair];
                AscendC::Add(dst, src1, src2, 1);
                cur = half + 1;
            } else {
                cur = half;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wRowQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> mulQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> accQue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0};
    uint32_t K_{0};
    uint32_t O_{0};
    uint32_t TK_{0};
    uint32_t TR_{0};
    uint32_t TO_{0};
};

extern "C" __global__ __aicore__ void matmul_sum_max_avg_pool_log_sum_exp_log_sum_exp_custom(
