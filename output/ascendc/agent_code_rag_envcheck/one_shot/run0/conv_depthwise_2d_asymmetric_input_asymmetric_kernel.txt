static inline int64_t ComputeOutDim(int64_t in, int64_t pad, int64_t dil, int64_t k, int64_t stride)
{
    // floor((in + 2*pad - dil*(k-1) - 1)/stride + 1)
    const int64_t eff = dil * (k - 1) + 1;
    return (in + 2 * pad - eff) / stride + 1;
}
}

    const int32_t *dil_w    = attrs->GetAttrPointer<int32_t>(5);

    const int64_t N = xdims[0];
    const int64_t C = xdims[1];
    const int64_t H = xdims[2];
    const int64_t W = xdims[3];

    const int64_t KH = wdims[2];
    const int64_t KW = wdims[3];

    const int64_t sH = (stride_h != nullptr) ? *stride_h : 1;
    const int64_t sW = (stride_w != nullptr) ? *stride_w : 1;
    const int64_t pH = (pad_h    != nullptr) ? *pad_h    : 0;
    const int64_t pW = (pad_w    != nullptr) ? *pad_w    : 0;
    const int64_t dH = (dil_h    != nullptr) ? *dil_h    : 1;
    const int64_t dW = (dil_w    != nullptr) ? *dil_w    : 1;

    auto outH = (H + 2 * pH - dH * (KH - 1) - 1) / sH + 1;
    auto outW = (W + 2 * pW - dW * (KW - 1) - 1) / sW + 1;

    y_shape->SetDim(0, N);
    y_shape->SetDim(1, C);
    y_shape->SetDim(2, outH);
    y_shape->SetDim(3, outW);
    return GRAPH_SUCCESS;

#include "kernel_operator.h"

class KernelConvDepthwise2DDepthwise {
public:
    __aicore__ inline KernelConvDepthwise2DDepthwise() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t KH, uint32_t KW, uint32_t OH, uint32_t OW,
                                int32_t strideH, int32_t strideW,
                                int32_t padH, int32_t padW,
                                int32_t dilationH, int32_t dilationW)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        KH_ = KH; KW_ = KW; OH_ = OH; OW_ = OW;
        strideH_ = strideH; strideW_ = strideW;
        padH_ = padH; padW_ = padW;
        dilationH_ = dilationH; dilationW_ = dilationW;

        xPtr_ = reinterpret_cast<__gm__ float*>(x);
        wPtr_ = reinterpret_cast<__gm__ float*>(w);
        yPtr_ = reinterpret_cast<__gm__ float*>(y);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t groups = N_ * C_;
        for (uint32_t g = blockIdx; g < groups; g += blockNum) {
            const uint32_t n = g / C_;
            const uint32_t c = g % C_;

            const uint64_t xBase = ((uint64_t)n * C_ + c) * (uint64_t)H_ * (uint64_t)W_;
            const uint64_t wBase = ((uint64_t)c) * (uint64_t)KH_ * (uint64_t)KW_;
            const uint64_t yBase = ((uint64_t)n * C_ + c) * (uint64_t)OH_ * (uint64_t)OW_;

            for (uint32_t oh = 0; oh < OH_; ++oh) {
                const int32_t hStart = static_cast<int32_t>(oh) * strideH_ - padH_;
                for (uint32_t ow = 0; ow < OW_; ++ow) {
                    const int32_t wStart = static_cast<int32_t>(ow) * strideW_ - padW_;
                    float acc = 0.0f;

                    for (uint32_t r = 0; r < KH_; ++r) {
                        const int32_t hIn = hStart + static_cast<int32_t>(r) * dilationH_;
                        if (hIn < 0 || hIn >= static_cast<int32_t>(H_)) {
                            continue;
                        }
                        const uint64_t xRowBase = xBase + static_cast<uint64_t>(hIn) * (uint64_t)W_;
                        const uint64_t wRowBase = wBase + (uint64_t)r * (uint64_t)KW_;

                        for (uint32_t s = 0; s < KW_; ++s) {
                            const int32_t wIn = wStart + static_cast<int32_t>(s) * dilationW_;
                            if (wIn < 0 || wIn >= static_cast<int32_t>(W_)) {
                                continue;
                            }
                            const float wv = wPtr_[wRowBase + s];
                            const float xv = xPtr_[xRowBase + static_cast<uint64_t>(wIn)];
                            acc += wv * xv;
                        }
                    }

                    yPtr_[yBase + (uint64_t)oh * (uint64_t)OW_ + ow] = acc;
                }
            }
        }
    }

private:
    __gm__ float* xPtr_ {nullptr};
    __gm__ float* wPtr_ {nullptr};
    __gm__ float* yPtr_ {nullptr};

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t KH_{0}, KW_{0}, OH_{0}, OW_{0};
    int32_t strideH_{1}, strideW_{1}, padH_{0}, padW_{0}, dilationH_{1}, dilationW_{1};
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvDepthwise2DDepthwise op;
    op.Init(
        x, weight, y,
        tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
        tiling_data.KH, tiling_data.KW, tiling_data.OH, tiling_data.OW,
        tiling_data.strideH, tiling_data.strideW,
        tiling_data.padH, tiling_data.padW,
        tiling_data.dilationH, tiling_data.dilationW
    );
    op.Process();
}
