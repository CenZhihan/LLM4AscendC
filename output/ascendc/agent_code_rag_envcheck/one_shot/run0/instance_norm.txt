#include "instance_norm_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   inQueueX: 2 * tile
//   outQueueY: 2 * tile
//   tmpQueue: 1 * tile
// Total floats = 5 * tile => bytes = 20 * tile
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 2048;            // favor larger DMA bursts
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // avoid too-large per-iter latency

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    InstanceNormCustomTilingData tiling;

    const gert::Shape *in_shape = context->GetInputShape(0);
    const auto &dims = in_shape->GetOriginShape().GetDims();
    if (dims.size() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = static_cast<uint32_t>(dims[0]);
    const uint32_t C = static_cast<uint32_t>(dims[1]);
    const uint32_t H = static_cast<uint32_t>(dims[2]);
    const uint32_t W = static_cast<uint32_t>(dims[3]);
    const uint32_t groupHW = H * W;
    const uint32_t totalGroups = N * C;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *epsPtr = attrs->GetAttrPointer<float>(0);
    const float eps = (epsPtr != nullptr) ? *epsPtr : 1e-5f;

    // tileLen from UB budget: bytes = 20 * tileLen
    uint32_t maxTile = UB_BUDGET_BYTES / 20;  // elements
    maxTile = AlignDown(maxTile, ALIGN_ELEMS);
    if (maxTile < ALIGN_ELEMS) maxTile = ALIGN_ELEMS;
    if (maxTile > HARD_CAP_ELEMS) maxTile = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileLen = maxTile;
    if (groupHW < tileLen) {
        const uint32_t aligned = AlignDown(groupHW, ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : groupHW;
    }
    if (tileLen == 0) tileLen = 1;

    // Adaptive blockDim based on number of (N*C) groups.
    uint32_t blockDim = totalGroups;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);
    tiling.set_eps(eps);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 2;

class KernelInstanceNormCustom {
public:
    __aicore__ inline KernelInstanceNormCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t tileLength, float eps)
    {
        n_ = n;
        c_ = c;
        h_ = h;
        w_ = w;
        hw_ = h_ * w_;
        groups_ = n_ * c_;
        tileLen_ = tileLength;
        eps_ = eps;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(n_ * c_ * h_ * w_));
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(n_ * c_ * h_ * w_));
        gammaGm_.SetGlobalBuffer((__gm__ float*)gamma, c_);
        betaGm_.SetGlobalBuffer((__gm__ float*)beta, c_);

        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileLen_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileLen_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (groups_ == 0 || hw_ == 0 || tileLen_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t g = blockIdx; g < groups_; g += blockNum) {
            const uint32_t nIdx = g / c_;
            const uint32_t cIdx = g % c_;
            const uint64_t base = static_cast<uint64_t>(g) * static_cast<uint64_t>(hw_);
            // Pass 1: compute sum and sum of squares for this (n,c) group
            float sum = 0.0f;
            float sum2 = 0.0f;
            uint32_t processed = 0;
            while (processed < hw_) {
                uint32_t len = hw_ - processed;
                if (len > tileLen_) len = tileLen_;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base + processed], len);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xL = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();

                // tmp = x
                AscendC::DataCopy(tmp, xL, len);
                AscendC::PipeBarrier<PIPE_V>();
                AscendC::ReduceSum(tmp, tmp, tmp, len);
                AscendC::SetFlag<HardEvent::V_S>(EVENT_ID0);
                AscendC::WaitFlag<HardEvent::V_S>(EVENT_ID0);
                float tileSum = tmp.GetValue(0);
                AscendC::SetFlag<HardEvent::S_V>(EVENT_ID0);
                AscendC::WaitFlag<HardEvent::S_V>(EVENT_ID0);

                // tmp = x * x
                AscendC::Mul(tmp, xL, xL, len);
                AscendC::PipeBarrier<PIPE_V>();
                AscendC::ReduceSum(tmp, tmp, tmp, len);
                AscendC::SetFlag<HardEvent::V_S>(EVENT_ID1);
                AscendC::WaitFlag<HardEvent::V_S>(EVENT_ID1);
                float tileSum2 = tmp.GetValue(0);
                AscendC::SetFlag<HardEvent::S_V>(EVENT_ID1);
                AscendC::WaitFlag<HardEvent::S_V>(EVENT_ID1);

                sum += tileSum;
                sum2 += tileSum2;

                tmpQueue_.FreeTensor(tmp);
                inQueueX_.FreeTensor(xL);

                processed += len;
            }

            const float M = static_cast<float>(hw_);
            const float mean = sum / M;
            const float var = AscendC::fmax((sum2 / M) - mean * mean, 0.0f);
            const float invStd = 1.0f / AscendC::sqrt(var + eps_);

            // Load per-channel gamma and beta scalars
            float gammaVal = 1.0f;
            float betaVal = 0.0f;
            {
                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                AscendC::DataCopy(tmp, gammaGm_[cIdx], 1);
                AscendC::PipeBarrier<PIPE_V>();
                AscendC::SetFlag<HardEvent::V_S>(EVENT_ID2);
                AscendC::WaitFlag<HardEvent::V_S>(EVENT_ID2);
                gammaVal = tmp.GetValue(0);
                AscendC::SetFlag<HardEvent::S_V>(EVENT_ID2);
                AscendC::WaitFlag<HardEvent::S_V>(EVENT_ID2);
                AscendC::DataCopy(tmp, betaGm_[cIdx], 1);
                AscendC::PipeBarrier<PIPE_V>();
                AscendC::SetFlag<HardEvent::V_S>(EVENT_ID3);
                AscendC::WaitFlag<HardEvent::V_S>(EVENT_ID3);
                betaVal = tmp.GetValue(0);
                AscendC::SetFlag<HardEvent::S_V>(EVENT_ID3);
                AscendC::WaitFlag<HardEvent::S_V>(EVENT_ID3);
                tmpQueue_.FreeTensor(tmp);
            }

            // Pass 2: normalize and apply affine for this group
            processed = 0;
            while (processed < hw_) {
                uint32_t len = hw_ - processed;
                if (len > tileLen_) len = tileLen_;
                AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                AscendC::DataCopy(xLocal, xGm_[base + processed], len);
                inQueueX_.EnQue(xLocal);

                AscendC::LocalTensor<float> xL = inQueueX_.DeQue<float>();
                AscendC::LocalTensor<float> yL = outQueueY_.AllocTensor<float>();

                // y = (x - mean) * invStd
                AscendC::Adds(yL, xL, -mean, len);
                AscendC::PipeBarrier<PIPE_V>();
                AscendC::Muls(yL, yL, invStd, len);
                AscendC::PipeBarrier<PIPE_V>();

                // y = y * gamma + beta
                AscendC::Muls(yL, yL, gammaVal, len);
                AscendC::PipeBarrier<PIPE_V>();
                AscendC::Adds(yL, yL, betaVal, len);
                AscendC::PipeBarrier<PIPE_V>();

                outQueueY_.EnQue<float>(yL);
                inQueueX_.FreeTensor(xL);

                AscendC::LocalTensor<float> yOut = outQueueY_.DeQue<float>();
                AscendC::DataCopy(yGm_[base + processed], yOut, len);
                outQueueY_.FreeTensor(yOut);

                processed += len;
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;
    AscendC::GlobalTensor<float> gammaGm_;
    AscendC::GlobalTensor<float> betaGm_;

    uint32_t n_{0};
    uint32_t c_{0};
    uint32_t h_{0};
    uint32_t w_{0};
    uint32_t hw_{0};
    uint32_t groups_{0};
    uint32_t tileLen_{0};
    float eps_{1e-5f};
};

extern "C" __global__ __aicore__ void instance_norm_custom(GM_ADDR x, GM_ADDR gamma, GM_ADDR beta, GM_ADDR y,
                                                            GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelInstanceNormCustom op;
    op.Init(x, gamma, beta, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.tileLength, tiling_data.eps);
    op.Process();
}
