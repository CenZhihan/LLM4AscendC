kernel_src="""
#include "kernel_operator.h"

extern \"C\" __global__ __aicore__ void matmul_scaling_residual_add_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR scaling,
                                                                           GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    // Tiling: rows/cols/features and blockDim
    GET_TILING_DATA(tiling_data, tiling);
    const uint32_t N = tiling_data.nRows;
    const uint32_t K = tiling_data.inFeat;
    const uint32_t M = tiling_data.outFeat;

    // Global buffers
    __gm__ half*  xGm = reinterpret_cast<__gm__ half*>(x);        // [N, K]
    __gm__ half*  wGm = reinterpret_cast<__gm__ half*>(w);        // [M, K]
    __gm__ float* bGm = reinterpret_cast<__gm__ float*>(b);       // [M]
    __gm__ float* sGm = reinterpret_cast<__gm__ float*>(scaling); // [1]
    __gm__ float* yGm = reinterpret_cast<__gm__ float*>(y);       // [N, M]

    // Read scaling scalar and compute effective factor for y*s + y = y*(1+s)
    float s = 0.0f;
    if (sGm != nullptr) {
        s = sGm[0];
    }
    const float factor = 1.0f + s;

    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());

    // Row-parallel grid stride
    for (uint32_t n = blockIdx; n < N; n += (blockNum == 0 ? 1U : blockNum)) {
        const uint64_t xBase = static_cast<uint64_t>(n) * static_cast<uint64_t>(K);
        const uint64_t yBase = static_cast<uint64_t>(n) * static_cast<uint64_t>(M);

        for (uint32_t m = 0; m < M; ++m) {
            // Accumulator initialized with bias
            float acc = bGm[m];
            const uint64_t wBase = static_cast<uint64_t>(m) * static_cast<uint64_t>(K);

            // Dot product: acc += dot(x[n, :], w[m, :])
            for (uint32_t k = 0; k < K; ++k) {
                float xv = static_cast<float>(xGm[xBase + k]);
                float wv = static_cast<float>(wGm[wBase + k]);
                acc += xv * wv;
            }

            // Apply y = y * s + y = y * (1 + s)
            acc *= factor;

            yGm[yBase + m] = acc;
        }
    }
}
"""

#include <torch/library.h>

static inline at::Tensor to_contig_dtype_npu(const at::Tensor& t, at::ScalarType dt) {
    at::Tensor r = t.to(dt);
    if (!r.is_contiguous()) r = r.contiguous();
    return r;
}

at::Tensor matmul_scaling_residual_add_impl_npu(const at::Tensor& x,
                                                const at::Tensor& w,
                                                const at::Tensor& b,
                                                const at::Tensor& scaling) {
    // Cast inputs to expected dtypes and contiguous on NPU
    at::Tensor x_cast = to_contig_dtype_npu(x, at::kHalf);
    at::Tensor w_cast = to_contig_dtype_npu(w, at::kHalf);
    at::Tensor b_cast = to_contig_dtype_npu(b, at::kFloat);
    at::Tensor s_cast = to_contig_dtype_npu(scaling, at::kFloat);

    // Output tensor [N, M], float
    auto y_sizes = std::vector<int64_t>{x_cast.size(0), w_cast.size(0)};
    at::Tensor y = at::empty(y_sizes, x_cast.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnMatmulScalingResidualAddCustom, x_cast, w_cast, b_cast, s_cast, y);
    return y;
}
