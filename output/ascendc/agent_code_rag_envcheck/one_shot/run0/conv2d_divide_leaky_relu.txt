#include "conv2d_divide_leaky_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// UB budget and alignment for one-OC row-conv implementation:
// Buffers per-core in UB:
//   yRow: 1 * tileW
//   xVec: 1 * tileW
//   tmp:  1 * tileW
//   weightK: K = C * Kh * Kw
//   bias: 1
// Total floats = (3 * tileW + K + 1) => bytes = 4 * (3 * tileW + K + 1)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 16;             // favor vector alignment
static constexpr uint32_t MAX_TILE_W = 8192;            // cap to avoid long iter latency

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dDivideLeakyReluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0); // NCHW
    const gert::Shape *w_shape = context->GetInputShape(1); // Cout,C,Kh,Kw

    const uint32_t N = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t C = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t H = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t W = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Cw   = static_cast<uint32_t>(w_shape->GetDim(1));
    const uint32_t Kh   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw   = static_cast<uint32_t>(w_shape->GetDim(3));

    if (C != Cw) {
        return ge::GRAPH_FAILED;
    }

    // stride=1, padding=0, dilation=1
    const uint32_t Ho = (H >= Kh) ? (H - Kh + 1) : 0;
    const uint32_t Wo = (W >= Kw) ? (W - Kw + 1) : 0;

    const uint32_t K = C * Kh * Kw;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *divisor = attrs->GetAttrPointer<float>(0);
    const float *negativeSlope = attrs->GetAttrPointer<float>(1);

    float invDiv = (divisor && *divisor != 0.0f) ? (1.0f / *divisor) : 0.5f;

    // Optional bias presence detection (best-effort)
    bool hasBias = false;
#if defined(HAS_GET_INPUT_TENSOR_NUM)
    hasBias = (context->GetInputTensorNum() > 2);
#else
    if (context->GetInputShapeSize() > 2) {
        const gert::Shape *b_shape = context->GetInputShape(2);
        hasBias = (b_shape != nullptr && b_shape->GetOriginShape().GetShapeSize() > 0);
    }
#endif

    // Compute tileW from UB budget
    uint32_t maxTileW = 0;
    if (K + 1u < (UB_BUDGET_BYTES / 4u)) {
        const uint32_t rem = (UB_BUDGET_BYTES / 4u) - (K + 1u);
        maxTileW = rem / 3u;
    } else {
        maxTileW = ALIGN_ELEMS;
    }

    if (maxTileW > MAX_TILE_W) maxTileW = MAX_TILE_W;
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;

    uint32_t tileW = maxTileW;
    if (Wo > 0 && tileW > Wo) {
        tileW = AlignDown(Wo, ALIGN_ELEMS);
        if (tileW == 0) tileW = Wo;
    }

    // Blocks across N * Ho (each block handles row lines in grid-stride)
    uint32_t totalRows = (Ho == 0) ? 1u : (N * Ho);
    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    // Save tiling data
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Cout(Cout);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1u : 0u);
    tiling.set_invDivisor(invDiv);
    tiling.set_negativeSlope(*negativeSlope);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dDivideLeakyReluCustom {
public:
    __aicore__ inline KernelConv2dDivideLeakyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t tileW, uint32_t hasBias,
                                float invDivisor, float negativeSlope)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        Cout_ = Cout; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;
        tileW_ = tileW;
        hasBias_ = hasBias != 0;
        invDiv_ = invDivisor;
        negSlope_ = negativeSlope;
        negCorr_ = negativeSlope - 1.0f;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * C_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, Cout_ * C_ * Kh_ * Kw_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * Cout_ * Ho_ * Wo_);

        // Buffers: xVec (tileW), yRow (tileW), tmp (tileW), weightK (K), bias (1)
        const uint32_t K = C_ * Kh_ * Kw_;
        pipe_.InitBuffer(xQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, 1, K * sizeof(float));
        pipe_.InitBuffer(bQueue_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (Ho_ == 0 || Wo_ == 0 || Cout_ == 0 || C_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalRows = N_ * Ho_;
        if (totalRows == 0) return;

        for (uint32_t row = blockIdx; row < totalRows; row += blockNum) {
            const uint32_t n = row / Ho_;
            const uint32_t ho = row % Ho_;

            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                // Load weight vector K for this output channel
                AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                const uint32_t K = C_ * Kh_ * Kw_;
                const uint32_t wOffset = oc * K;
                AscendC::DataCopy(wLocal, wGm_[wOffset], K);

                // Load bias if present
                float biasVal = 0.0f;
                if (hasBias_) {
                    AscendC::LocalTensor<float> bLocal = bQueue_.AllocTensor<float>();
                    AscendC::DataCopy(bLocal, bGm_[oc], 1);
                    biasVal = bLocal.GetValue(0);
                    bQueue_.FreeTensor(bLocal);
                }

                for (uint32_t wStart = 0; wStart < Wo_; wStart += tileW_) {
                    uint32_t curW = Wo_ - wStart;
                    if (curW > tileW_) curW = tileW_;

                    // yRow init with bias (or zero)
                    AscendC::LocalTensor<float> yRow = yQueue_.AllocTensor<float>();
                    AscendC::Dup(yRow, biasVal, curW);

                    // Accumulate convolution
                    uint32_t kIndex = 0;
                    for (uint32_t ic = 0; ic < C_; ++ic) {
                        for (uint32_t kh = 0; kh < Kh_; ++kh) {
                            const uint32_t inH = ho + kh;
                            for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                const uint32_t inWStart = wStart + kw;

                                // Load input vector for this (ic,kh,kw)
                                AscendC::LocalTensor<float> xVec = xQueue_.AllocTensor<float>();
                                const uint64_t xOffset = (((uint64_t)n * C_ + ic) * H_ + inH) * W_ + inWStart;
                                AscendC::DataCopy(xVec, xGm_[xOffset], curW);

                                // Multiply-accumulate: yRow += xVec * wScalar
                                const float wScalar = wLocal.GetValue(kIndex);
                                AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                                AscendC::Muls(tmp, xVec, wScalar, curW);
                                AscendC::Add(yRow, yRow, tmp, curW);
                                tmpQueue_.FreeTensor(tmp);

                                xQueue_.FreeTensor(xVec);
                                ++kIndex;
                            }
                        }
                    }

                    // Post-ops: divide by divisor (scale by invDiv), then LeakyReLU
                    AscendC::Muls(yRow, yRow, invDiv_, curW);
                    AscendC::LocalTensor<float> t = tmpQueue_.AllocTensor<float>();
                    AscendC::Mins(t, yRow, 0.0f, curW);
                    const float corr = negCorr_;
                    AscendC::Muls(t, t, corr, curW);
                    AscendC::Add(yRow, yRow, t, curW);
                    tmpQueue_.FreeTensor(t);

                    // Store to GM
                    const uint64_t yOffset = (((uint64_t)n * Cout_ + oc) * Ho_ + ho) * Wo_ + wStart;
                    AscendC::DataCopy(yGm_[yOffset], yRow, curW);
                    yQueue_.FreeTensor(yRow);
                }
                wQueue_.FreeTensor(wLocal);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> yQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> bQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t tileW_{0};
    bool hasBias_{false};
    float invDiv_{0.5f};
    float negSlope_{0.01f};
    float negCorr_{-0.99f};
};

extern "C" __global__ __aicore__ void conv2d_divide_leaky_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dDivideLeakyReluCustom op;
    op.Init(x, weight, bias, y,
            tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
            tiling_data.Cout, tiling_data.Kh, tiling_data.Kw,
            tiling_data.Ho, tiling_data.Wo,
            tiling_data.tileW, tiling_data.hasBias,
            tiling_data.invDivisor, tiling_data.negativeSlope);
    op.Process();
}
