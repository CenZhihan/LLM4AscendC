#include "gemm_add_relu_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : ((x / a) * a);
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : ((a + b - 1u) / b);
}

// UB model for this kernel:
//   - xRow (K elements)
//   - yAcc (tileM)
//   - wRow (tileM)
//   - tmpVec (tileM)
//   - biasTile (tileM)
// Total bytes ~= 4*(tileM)*4 + K*4 = (16*tileM + 4*K)
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // per-core conservative budget
static constexpr uint32_t ALIGN_M_ELEMS = 256;
static constexpr uint32_t HARD_CAP_M_ELEMS = 65536;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    GemmAddReluCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    const gert::Shape *b_shape = context->GetInputShape(2);

    // Expect shapes:
    //   x: [N, K]
    //   w: [K, M]   (pre-transposed in python binding for contiguous access)
    //   bias: [M]
    const auto &x_origin = x_shape->GetOriginShape();
    const auto &w_origin = w_shape->GetOriginShape();
    const auto &b_origin = b_shape->GetOriginShape();

    const uint32_t N = static_cast<uint32_t>(x_origin.GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_origin.GetDim(1));
    const uint32_t KM = static_cast<uint32_t>(w_origin.GetShapeSize());
    uint32_t wK = (w_origin.GetDimNum() >= 1) ? static_cast<uint32_t>(w_origin.GetDim(0)) : 0u;
    uint32_t wM = (w_origin.GetDimNum() >= 2) ? static_cast<uint32_t>(w_origin.GetDim(1)) : 0u;
    // Fallback if API doesn't provide dims directly
    if (wK == 0u || wM == 0u) {
        wK = K;
        wM = (K == 0u) ? 0u : (KM / K);
    }
    const uint32_t M = wM;

    // UB estimation
    uint32_t reservedXBytes = K * 4u;
    uint32_t bytesForVecs = (UB_BUDGET_BYTES > reservedXBytes) ? (UB_BUDGET_BYTES - reservedXBytes) : 0u;
    // We need yAcc + wRow + tmpVec + biasTile = 4 * tileM * 4 bytes
    uint32_t maxTileM = (bytesForVecs / 16u);

    if (maxTileM == 0) {
        maxTileM = ALIGN_M_ELEMS;
    }
    maxTileM = AlignDown(maxTileM, ALIGN_M_ELEMS);
    if (maxTileM < ALIGN_M_ELEMS) maxTileM = ALIGN_M_ELEMS;
    if (maxTileM > HARD_CAP_M_ELEMS) maxTileM = AlignDown(HARD_CAP_M_ELEMS, ALIGN_M_ELEMS);

    uint32_t tileM = maxTileM;
    if (M < tileM) {
        const uint32_t aligned = AlignDown(M, ALIGN_M_ELEMS);
        tileM = (aligned > 0) ? aligned : M;
    }
    if (tileM == 0) tileM = 1;

    const uint32_t numMTiles = (tileM == 0u) ? 1u : CeilDiv(M, tileM);
    const uint32_t totalTiles = (N == 0u) ? 0u : (N * numMTiles);
    uint32_t blockDim = totalTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_M(M);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t INBUF_NUM = 2;

class KernelGemmAddReluCustom {
public:
    __aicore__ inline KernelGemmAddReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y,
                                uint32_t N, uint32_t M, uint32_t K, uint32_t tileM)
    {
        N_ = N;
        M_ = M;
        K_ = K;
        tileM_ = tileM;

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint32_t>(N_) * K_);
        // w is expected to be [K, M] row-major contiguous
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint32_t>(K_) * M_);
        bGm_.SetGlobalBuffer((__gm__ float*)bias, M_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint32_t>(N_) * M_);

        // Buffers:
        // - yAcc tile (tileM)
        // - wRow tile (tileM)
        // - tmpVec (tileM)
        // - bias tile (tileM)
        // - xRow (K)
        pipe_.InitBuffer(outQueueY_, 1, tileM_ * sizeof(float));
        pipe_.InitBuffer(inQueueW_, INBUF_NUM, tileM_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tileM_ * sizeof(float));
        pipe_.InitBuffer(biasQueue_, 1, tileM_ * sizeof(float));
        pipe_.InitBuffer(xRowQueue_, 1, K_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || M_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t numMTiles = (tileM_ == 0u) ? 1u : ( (M_ + tileM_ - 1u) / tileM_ );
        const uint32_t totalTiles = (N_ == 0u) ? 0u : (N_ * numMTiles);
        if (totalTiles == 0u) return;

        // Grid-stride over (n, mTile) tiles
        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t n = tileId / numMTiles;
            const uint32_t mTileIdx = tileId % numMTiles;
            const uint32_t mStart = mTileIdx * tileM_;
            uint32_t curM = M_ - mStart;
            if (curM > tileM_) curM = tileM_;
            ComputeTile(n, mStart, curM);
        }
    }

private:
    __aicore__ inline void LoadXRow(uint32_t n)
    {
        // Copy x[n, 0:K] to local
        AscendC::LocalTensor<float> xRow = xRowQueue_.AllocTensor<float>();
        const uint32_t offset = n * K_;
        AscendC::DataCopy(xRow, xGm_[offset], K_);
        xRowQueue_.EnQue<float>(xRow);
    }

    __aicore__ inline float ReadXVal(uint32_t k)
    {
        // Helper to read the k-th element from xRow in queue
        AscendC::LocalTensor<float> xRow = xRowQueue_.DeQue<float>();
        // NOTE: Local element access; pseudo-API assuming direct indexing
        float xVal = xRow.GetValue(k);
        xRowQueue_.EnQue<float>(xRow);
        return xVal;
    }

    __aicore__ inline void ComputeTile(uint32_t n, uint32_t mStart, uint32_t curM)
    {
        // Prepare accumulators
        AscendC::LocalTensor<float> yAcc = outQueueY_.AllocTensor<float>();
        AscendC::Dup(yAcc, 0.0f, curM);

        // Load x row once
        LoadXRow(n);

        // Accumulate yAcc += x[n,k] * w[k, mStart:mStart+curM]
        for (uint32_t k = 0; k < K_; ++k) {
            // Load w row segment for this k (contiguous in w[K,M])
            AscendC::LocalTensor<float> wRow = inQueueW_.AllocTensor<float>();
            const uint32_t wOff = k * M_ + mStart;
            AscendC::DataCopy(wRow, wGm_[wOff], curM);
            inQueueW_.EnQue<float>(wRow);

            // Get scalar xVal = x[n,k]
            float xVal = ReadXVal(k);

            // tmpVec = wRow * xVal
            AscendC::LocalTensor<float> tmpVec = tmpQueue_.AllocTensor<float>();
            AscendC::Muls(tmpVec, wRow, xVal, curM);

            // yAcc += tmpVec
            AscendC::Add(yAcc, yAcc, tmpVec, curM);

            tmpQueue_.FreeTensor(tmpVec);
            inQueueW_.DeQue<float>(); // retrieve wRow for freeing
            inQueueW_.FreeTensor(wRow);
        }

        // Add bias
        AscendC::LocalTensor<float> bTile = biasQueue_.AllocTensor<float>();
        AscendC::DataCopy(bTile, bGm_[mStart], curM);
        AscendC::Add(yAcc, yAcc, bTile, curM);

        // ReLU
        AscendC::Maxs(yAcc, yAcc, 0.0f, curM);

        // Store
        const uint32_t yOff = n * M_ + mStart;
        AscendC::DataCopy(yGm_[yOff], yAcc, curM);

        biasQueue_.FreeTensor(bTile);
        outQueueY_.FreeTensor(yAcc);

        // Release xRow
        AscendC::LocalTensor<float> xRow = xRowQueue_.DeQue<float>();
        xRowQueue_.FreeTensor(xRow);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECIN, INBUF_NUM> inQueueW_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> biasQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> xRowQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_ {0};
    uint32_t M_ {0};
    uint32_t K_ {0};
    uint32_t tileM_ {0};
};

extern "C" __global__ __aicore__ void gemm_add_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelGemmAddReluCustom op;
    op.Init(x, w, bias, y,
            tiling_data.N, tiling_data.M, tiling_data.K, tiling_data.tileM);
    op.Process();
}
