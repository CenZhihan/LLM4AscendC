{
    const gert::Shape *x_shape = context->GetInputShape(0);

    const int64_t Din = x_shape->GetDim(2);
    const int64_t Hin = x_shape->GetDim(3);
    const int64_t Win = x_shape->GetDim(4);

    const int64_t W_Cin     = w_shape->GetDim(0);
    const int64_t W_CoutG   = w_shape->GetDim(1);
    const int64_t Kd        = w_shape->GetDim(2);
    const int64_t Kh        = w_shape->GetDim(3);
    const int64_t Kw        = w_shape->GetDim(4);
    if (W_Cin != Cin) return GRAPH_FAILED;

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int64_t *strides_ptr = attrs->GetListInt(0);
    const int64_t *pads_ptr    = attrs->GetListInt(1);
    const int64_t *outpads_ptr = attrs->GetListInt(2);
    const int64_t  groups      = attrs->GetAttrInt(3);
    (void)strides_ptr; (void)pads_ptr; (void)outpads_ptr;

    const int64_t strideD = strides_ptr ? strides_ptr[0] : 2;
    const int64_t strideH = strides_ptr ? strides_ptr[1] : 2;
    const int64_t strideW = strides_ptr ? strides_ptr[2] : 2;
    const int64_t padD    = pads_ptr ? pads_ptr[0] : 1;
    const int64_t padH    = pads_ptr ? pads_ptr[1] : 2;
    const int64_t padW    = pads_ptr ? pads_ptr[2] : 3;

    const int64_t outPadH = outpads_ptr ? outpads_ptr[1] : 1;
    const int64_t outPadW = outpads_ptr ? outpads_ptr[2] : 1;

    const int64_t Cout  = W_CoutG * G;
    const int64_t Dout  = (Din - 1) * strideD - 2 * padD + (Kd - 1) + outPadD + 1;
    const int64_t Hout  = (Hin - 1) * strideH - 2 * padH + (Kh - 1) + outPadH + 1;

    std::vector<int64_t> y_dims = {N, Cout, Dout, Hout, Wout};
    y_shape->SetDimNum(5);
    for (size_t i=0;i<y_dims.size();++i) y_shape->SetDim(i, y_dims[i]);

    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

class KernelConvTransposed3dAIKG {
public:
    __aicore__ inline KernelConvTransposed3dAIKG() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, uint32_t totalOutElems, uint32_t tileOutElems,
                                // x dims
                                uint32_t N, uint32_t Cin, uint32_t Din, uint32_t Hin, uint32_t Win,
                                // y dims
                                uint32_t Cout, uint32_t Dout, uint32_t Hout, uint32_t Wout,
                                // strides
                                uint32_t xStrideN, uint32_t xStrideC, uint32_t xStrideD, uint32_t xStrideH,
                                uint32_t yStrideN, uint32_t yStrideC, uint32_t yStrideD, uint32_t yStrideH,
                                // kernel dims
                                uint32_t Kd, uint32_t Kh, uint32_t Kw,
                                // weight strides
                                uint32_t wStrideCin, uint32_t wStrideCoutG, uint32_t wStrideKd, uint32_t wStrideKh,
                                // attrs
                                uint32_t strideD, uint32_t strideH, uint32_t strideW,
                                uint32_t padD, uint32_t padH, uint32_t padW,
                                uint32_t outPadD, uint32_t outPadH, uint32_t outPadW,
                                uint32_t groups)
    {
        (void)outPadD; (void)outPadH; (void)outPadW; // mapping doesn't directly use output_padding
        totalOutElems_ = totalOutElems;
        tileOutElems_  = tileOutElems;

        N_ = N; Cin_ = Cin; Din_ = Din; Hin_ = Hin; Win_ = Win;
        Cout_ = Cout; Dout_ = Dout; Hout_ = Hout; Wout_ = Wout;

        xStrideN_ = xStrideN; xStrideC_ = xStrideC; xStrideD_ = xStrideD; xStrideH_ = xStrideH;
        yStrideN_ = yStrideN; yStrideC_ = yStrideC; yStrideD_ = yStrideD; yStrideH_ = yStrideH;

        Kd_ = Kd; Kh_ = Kh; Kw_ = Kw;
        wStrideCin_ = wStrideCin; wStrideCoutG_ = wStrideCoutG; wStrideKd_ = wStrideKd; wStrideKh_ = wStrideKh;

        strideD_ = strideD; strideH_ = strideH; strideW_ = strideW;
        padD_ = padD; padH_ = padH; padW_ = padW;

        groups_ = groups;
        ocPerGroup_ = Cout_ / groups_;
        cinPerGroup_ = Cin_ / groups_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * Cin_ * Din_ * Hin_ * Win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cin_ * ocPerGroup_ * Kd_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Cout_ * Dout_ * Hout_ * Wout_);

        pipe_.InitBuffer(yBufQ_, 2, tileOutElems_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutElems_ == 0 || tileOutElems_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(GetBlockIdx());
        const uint32_t strideElems = blockNum * tileOutElems_;

        uint32_t base = blockIdx * tileOutElems_;
        if (base >= totalOutElems_) return;

        while (base < totalOutElems_) {
            uint32_t cur = totalOutElems_ - base;
            if (cur > tileOutElems_) cur = tileOutElems_;

            LocalTensor<float> yLocal = yBufQ_.AllocTensor<float>();
            // compute y[base : base+cur)
            ComputeTile(base, yLocal, cur);
            // write back
            DataCopy(yGm_[base], yLocal, cur);
            yBufQ_.FreeTensor(yLocal);

            base += strideElems;
        }
    }

private:
    __aicore__ inline void UnravelOutIndex(uint32_t idx, uint32_t &n, uint32_t &oc, uint32_t &od, uint32_t &oh, uint32_t &ow)
    {
        uint32_t t = idx;
        ow = t % Wout_; t /= Wout_;
        oh = t % Hout_; t /= Hout_;
        od = t % Dout_; t /= Dout_;
        oc = t % Cout_; t /= Cout_;
        n  = t;
    }

    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t ic, uint32_t id, uint32_t ih, uint32_t iw) const
    {
        return (uint64_t)n * xStrideN_ + (uint64_t)ic * xStrideC_ +
               (uint64_t)id * xStrideD_ + (uint64_t)ih * xStrideH_ + (uint64_t)iw;
    }

    __aicore__ inline uint64_t WOffset(uint32_t ic, uint32_t ocg, uint32_t kd, uint32_t kh, uint32_t kw) const
    {
        return (uint64_t)ic * wStrideCin_ + (uint64_t)ocg * wStrideCoutG_ +
               (uint64_t)kd * wStrideKd_ + (uint64_t)kh * wStrideKh_ + (uint64_t)kw;
    }

    __aicore__ inline void ComputeTile(uint32_t base, LocalTensor<float> &yLocal, uint32_t len)
    {
        // initialize to 0
        if (len > 0) {
            // simple zero fill
            for (uint32_t i = 0; i < len; ++i) {
                yLocal.SetValue(i, 0.0f);
            }
        }

        for (uint32_t i = 0; i < len; ++i) {
            uint32_t gIndex = base + i;

            uint32_t n, oc, od, oh, ow;
            UnravelOutIndex(gIndex, n, oc, od, oh, ow);

            const uint32_t g    = oc / ocPerGroup_;
            const uint32_t ocg  = oc - g * ocPerGroup_;
            const uint32_t cinStart = g * cinPerGroup_;
            const uint32_t cinEnd   = cinStart + cinPerGroup_;

            float acc = 0.0f;

            // iterate over kernel positions and contributing input positions
            for (uint32_t kd = 0; kd < Kd_; ++kd) {
                int32_t id_num = (int32_t)od + (int32_t)padD_ - (int32_t)kd;
                if (id_num < 0) continue;
                if ((uint32_t)id_num > (uint32_t)0x7fffffff) continue; // guard
                if (id_num % (int32_t)strideD_ != 0) continue;
                uint32_t id = (uint32_t)(id_num / (int32_t)strideD_);
                if (id >= Din_) continue;

                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    int32_t ih_num = (int32_t)oh + (int32_t)padH_ - (int32_t)kh;
                    if (ih_num < 0) continue;
                    if (ih_num % (int32_t)strideH_ != 0) continue;
                    uint32_t ih = (uint32_t)(ih_num / (int32_t)strideH_);
                    if (ih >= Hin_) continue;

                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        int32_t iw_num = (int32_t)ow + (int32_t)padW_ - (int32_t)kw;
                        if (iw_num < 0) continue;
                        if (iw_num % (int32_t)strideW_ != 0) continue;
                        uint32_t iw = (uint32_t)(iw_num / (int32_t)strideW_);
                        if (iw >= Win_) continue;

                        // sum over input channels in this group
                        for (uint32_t ic = cinStart; ic < cinEnd; ++ic) {
                            float xval, wval;
                            // load x[n, ic, id, ih, iw]
                            DataCopy(&xval, xGm_.GetAddr(XOffset(n, ic, id, ih, iw)), 1);
                            // load w[ic, ocg, kd, kh, kw]
                            DataCopy(&wval, wGm_.GetAddr(WOffset(ic, ocg, kd, kh, kw)), 1);
                            acc += xval * wval;
                        }
                    }
                }
            }
            yLocal.SetValue(i, acc);
        }
    }
