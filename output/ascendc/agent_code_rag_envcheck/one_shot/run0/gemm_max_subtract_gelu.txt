kernel_src="""
#include "kernel_operator.h"
#include "tiling/gemm_max_subtract_gelu_custom_tiling.h"

class KernelGemmMaxSubtractGeluCustom {
public:
    __aicore__ inline KernelGemmMaxSubtractGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K, uint32_t maxDim)
    {
        xGm_ = reinterpret_cast<__gm__ float*>(x);
        wGm_ = reinterpret_cast<__gm__ float*>(w);
        bGm_ = reinterpret_cast<__gm__ float*>(b);
        yGm_ = reinterpret_cast<__gm__ float*>(y);

        M_ = M; N_ = N; K_ = K; maxDim_ = maxDim;
    }

    __aicore__ inline float GeluScalar(float x) {
        // tanh-based approximate GELU
        const float k0 = 0.7978845608028654f; // sqrt(2/pi)
        const float k1 = 0.044715f;
        float x3 = x * x * x;
        float u = k0 * (x + k1 * x3);
        float t = AscendC::tanh(u); // scalar tanh from AscendC
        return 0.5f * x * (1.0f + t);
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        if (maxDim_ == 1u) {
            // Reduce over dim=1 (feature dim N) per row, keepdim=True -> [M, 1]
            // Parallelize over rows.
            for (uint32_t m = blockIdx; m < M_; m += blockNum) {
                float maxVal = -__FLT_MAX__;
                const uint32_t xBase = m * K_;
                for (uint32_t n = 0; n < N_; ++n) {
                    float acc = bGm_[n];
                    const uint32_t wBase = n * K_;
                    // Dot product of x[m,:] and w[n,:]
                    for (uint32_t k = 0; k < K_; ++k) {
                        acc += xGm_[xBase + k] * wGm_[wBase + k];
                    }
                    if (acc > maxVal) maxVal = acc;
                }
                // mean over dim=1 for [*,1] equals itself; center -> zero
                float centered = maxVal - maxVal;
                float g = GeluScalar(centered);
                yGm_[m] = g;
            }
        } else {
            // Reduce over dim=0 (batch dim M), keepdim=True -> [1, N]
            // For correctness of subsequent mean (over dim=1 -> across N), use block 0 only.
            if (blockIdx != 0u) return;

            // Phase 1: compute max over M for each n and accumulate sum for mean
            double sum = 0.0;
            for (uint32_t n = 0; n < N_; ++n) {
                float maxVal = -__FLT_MAX__;
                const uint32_t wBase = n * K_;
                for (uint32_t m = 0; m < M_; ++m) {
                    float acc = bGm_[n];
                    const uint32_t xBase = m * K_;
                    for (uint32_t k = 0; k < K_; ++k) {
                        acc += xGm_[xBase + k] * wGm_[wBase + k];
                    }
                    if (acc > maxVal) maxVal = acc;
                }
                yGm_[n] = maxVal;  // temporarily store raw maxima
                sum += static_cast<double>(maxVal);
            }
            float mean = (N_ > 0u) ? static_cast<float>(sum / static_cast<double>(N_)) : 0.0f;

            // Phase 2: center and apply GELU
            for (uint32_t n = 0; n < N_; ++n) {
                float centered = yGm_[n] - mean;
                float g = GeluScalar(centered);
                yGm_[n] = g;
            }
        }
    }

private:
    __gm__ float* xGm_ {nullptr};
    __gm__ float* wGm_ {nullptr};
    __gm__ float* bGm_ {nullptr};
    __gm__ float* yGm_ {nullptr};

    uint32_t M_{0}, N_{0}, K_{0}, maxDim_{1};
};

extern "C" __global__ __aicore__ void gemm_max_subtract_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelGemmMaxSubtractGeluCustom op;
    op.Init(x, w, b, y, tilingData.M, tilingData.N, tilingData.K, tilingData.maxDim);
    op.Process();
}
