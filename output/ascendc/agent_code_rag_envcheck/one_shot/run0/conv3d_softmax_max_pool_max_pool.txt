kernel_src="""
#include "kernel_operator.h"

class KernelConv3dSoftmaxMaxPoolMaxPoolCustom {
public:
    __aicore__ inline KernelConv3dSoftmaxMaxPoolMaxPoolCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t Di, uint32_t Hi, uint32_t Wi,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t Do, uint32_t Ho, uint32_t Wo,
                                uint32_t D1, uint32_t H1, uint32_t W1,
                                uint32_t D2, uint32_t H2, uint32_t W2,
                                uint64_t ws0Bytes, uint64_t ws1Bytes, uint64_t ws1OffsetBytes)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout;
        Di_ = Di; Hi_ = Hi; Wi_ = Wi;
        kD_ = kD; kH_ = kH; kW_ = kW;
        Do_ = Do; Ho_ = Ho; Wo_ = Wo;
        D1_ = D1; H1_ = H1; W1_ = W1;
        D2_ = D2; H2_ = H2; W2_ = W2;

        xGm_ = reinterpret_cast<__gm__ float*>(x);
        wGm_ = reinterpret_cast<__gm__ float*>(w);
        bGm_ = reinterpret_cast<__gm__ float*>(b);
        yGm_ = reinterpret_cast<__gm__ float*>(y);

        ws0_ = reinterpret_cast<__gm__ float*>(workspace);
        ws1_ = reinterpret_cast<__gm__ float*>(reinterpret_cast<char*>(workspace) + ws1OffsetBytes);

        // Precompute strides for NCDHW and for channels-last buffers
        strideX_W_ = 1u;
        strideX_H_ = Wi_;
        strideX_D_ = Hi_ * Wi_;
        strideX_C_ = Di_ * Hi_ * Wi_;
        strideX_N_ = Cin_ * Di_ * Hi_ * Wi_;

        // channels-last: [N, D, H, W, C]
        strideWS_C_ = 1u;
        strideWS_W_ = Cout_;
        strideWS_H_ = Wo_ * Cout_;
        strideWS_D_ = Ho_ * Wo_ * Cout_;
        strideWS_N_ = Do_ * Ho_ * Wo_ * Cout_;

        // channels-last first pool
        strideWS1_C_ = 1u;
        strideWS1_W_ = Cout_;
        strideWS1_H_ = W1_ * Cout_;
        strideWS1_D_ = H1_ * W1_ * Cout_;
        strideWS1_N_ = D1_ * H1_ * W1_ * Cout_;

        // strides for output y in NCDHW
        strideY_W_ = 1u;
        strideY_H_ = W2_;
        strideY_D_ = H2_ * W2_;
        strideY_C_ = D2_ * H2_ * W2_;
        strideY_N_ = Cout_ * D2_ * H2_ * W2_;
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || Cout_ == 0) return;

        uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride across batch dimension
        for (uint32_t n = blockIdx; n < N_; n += blockNum) {
            ConvAndSoftmaxToWS0(n);
            MaxPool3D(ws0_, Do_, Ho_, Wo_, ws1_, D1_, H1_, W1_);
            MaxPool3D(ws1_, D1_, H1_, W1_, nullptr, D2_, H2_, W2_); // directly write to y
        }
    }

private:
    __aicore__ inline uint64_t XIndex(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        return static_cast<uint64_t>(n) * strideX_N_ +
               static_cast<uint64_t>(c) * strideX_C_ +
               static_cast<uint64_t>(d) * strideX_D_ +
               static_cast<uint64_t>(h) * strideX_H_ +
               static_cast<uint64_t>(w) * strideX_W_;
    }

    __aicore__ inline uint64_t WIndex(uint32_t co, uint32_t ci, uint32_t kd, uint32_t kh, uint32_t kw) const
    {
        // [Cout, Cin, kD, kH, kW], contiguous in last dims
        return (((static_cast<uint64_t>(co) * Cin_ + ci) * kD_ + kd) * kH_ + kh) * kW_ + kw;
    }

    __aicore__ inline uint64_t WS0Index(uint32_t n, uint32_t d, uint32_t h, uint32_t w, uint32_t c) const
    {
        return static_cast<uint64_t>(n) * strideWS_N_ +
               static_cast<uint64_t>(d) * strideWS_D_ +
               static_cast<uint64_t>(h) * strideWS_H_ +
               static_cast<uint64_t>(w) * strideWS_W_ +
               static_cast<uint64_t>(c) * strideWS_C_;
    }

    __aicore__ inline uint64_t WS1Index(uint32_t n, uint32_t d, uint32_t h, uint32_t w, uint32_t c) const
    {
        return static_cast<uint64_t>(n) * strideWS1_N_ +
               static_cast<uint64_t>(d) * strideWS1_D_ +
               static_cast<uint64_t>(h) * strideWS1_H_ +
               static_cast<uint64_t>(w) * strideWS1_W_ +
               static_cast<uint64_t>(c) * strideWS1_C_;
    }

    __aicore__ inline uint64_t YIndex(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        return static_cast<uint64_t>(n) * strideY_N_ +
               static_cast<uint64_t>(c) * strideY_C_ +
               static_cast<uint64_t>(d) * strideY_D_ +
               static_cast<uint64_t>(h) * strideY_H_ +
               static_cast<uint64_t>(w) * strideY_W_;
    }

    __aicore__ inline void ConvAndSoftmaxToWS0(uint32_t n)
    {
        // Local buffer for channels vector
        for (uint32_t od = 0; od < Do_; ++od) {
            const uint32_t inDBase = od;
            for (uint32_t oh = 0; oh < Ho_; ++oh) {
                const uint32_t inHBase = oh;
                for (uint32_t ow = 0; ow < Wo_; ++ow) {
                    const uint32_t inWBase = ow;

                    // Compute convolution for all output channels at this voxel
                    // Store results temporarily in registers/stack
                    // Note: small Cout expected; otherwise consider tiling
                    float maxv = -3.402823466e+38F;
                    // First pass: compute raw conv outputs, track max
                    for (uint32_t co = 0; co < Cout_; ++co) {
                        float acc = bGm_[co];
                        for (uint32_t ci = 0; ci < Cin_; ++ci) {
                            for (uint32_t kd = 0; kd < kD_; ++kd) {
                                const uint32_t id = inDBase + kd;
                                for (uint32_t kh = 0; kh < kH_; ++kh) {
                                    const uint32_t ih = inHBase + kh;
                                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                                        const uint32_t iw = inWBase + kw;
                                        const uint64_t xIdx = XIndex(n, ci, id, ih, iw);
                                        const uint64_t wIdx = WIndex(co, ci, kd, kh, kw);
                                        acc += xGm_[xIdx] * wGm_[wIdx];
                                    }
                                }
                            }
                        }
                        // Temporarily write raw conv to ws0 slot, will normalize in-place
                        const uint64_t baseIdx = WS0Index(n, od, oh, ow, co);
                        ws0_[baseIdx] = acc;
                        if (acc > maxv) maxv = acc;
                    }

                    // Second pass: softmax across channels, in-place on ws0
                    float sumExp = 0.f;
                    for (uint32_t co = 0; co < Cout_; ++co) {
                        const uint64_t idx = WS0Index(n, od, oh, ow, co);
                        const float shifted = ws0_[idx] - maxv;
                        // Use device-side expf
                        float e = AscendC::Exp(shifted);
                        ws0_[idx] = e;
                        sumExp += e;
                    }
                    const float invSum = 1.0f / sumExp;
                    for (uint32_t co = 0; co < Cout_; ++co) {
                        const uint64_t idx = WS0Index(n, od, oh, ow, co);
                        ws0_[idx] = ws0_[idx] * invSum;
                    }
                }
            }
        }
    }

    __aicore__ inline void MaxPool3D(__gm__ float *src, uint32_t Di, uint32_t Hi, uint32_t Wi,
                                     __gm__ float *dst, uint32_t Do, uint32_t Ho, uint32_t Wo)
    {
        // If dst is nullptr, write to final y in NCDHW layout after second pooling
        const bool finalWrite = (dst == nullptr);
        for (uint32_t n = 0; n < N_; ++n) {
            for (uint32_t od = 0; od < Do; ++od) {
                const uint32_t id0 = od * 2;
                for (uint32_t oh = 0; oh < Ho; ++oh) {
                    const uint32_t ih0 = oh * 2;
                    for (uint32_t ow = 0; ow < Wo; ++ow) {
                        const uint32_t iw0 = ow * 2;
                        for (uint32_t co = 0; co < Cout_; ++co) {
                            // Max over 2x2x2 window from src laid out channels-last [N, Di, Hi, Wi, C]
                            float m = -3.402823466e+38F;
                            // 8 points
                            for (uint32_t rd = 0; rd < 2; ++rd) {
                                const uint32_t id = id0 + rd;
                                if (id >= Di) continue;
                                for (uint32_t rh = 0; rh < 2; ++rh) {
                                    const uint32_t ih = ih0 + rh;
                                    if (ih >= Hi) continue;
                                    // two elements along W are contiguous in Wi dimension
                                    const uint32_t iwA = iw0 + 0;
                                    const uint32_t iwB = iw0 + 1;
                                    if (iwA < Wi) {
                                        const uint64_t idxA = WSIndexGeneric(n, id, ih, iwA, co, Di, Hi, Wi);
                                        const float vA = src[idxA];
                                        if (vA > m) m = vA;
                                    }
                                    if (iwB < Wi) {
                                        const uint64_t idxB = WSIndexGeneric(n, id, ih, iwB, co, Di, Hi, Wi);
                                        const float vB = src[idxB];
                                        if (vB > m) m = vB;
                                    }
                                }
                            }

                            if (!finalWrite) {
                                const uint64_t outIdx = WSIndexGeneric(n, od, oh, ow, co, Do, Ho, Wo, /*channelsLast*/ true);
                                dst[outIdx] = m;
                            } else {
                                const uint64_t yIdx = YIndex(n, co, od, oh, ow);
                                yGm_[yIdx] = m;
                            }
                        }
                    }
                }
            }
        }
    }

    __aicore__ inline uint64_t WSIndexGeneric(uint32_t n, uint32_t d, uint32_t h, uint32_t w, uint32_t c,
                                              uint32_t D, uint32_t H, uint32_t W, bool channelsLast = true) const
    {
        // Always channels-last for our workspace buffers
        // idx = (((n*D + d)*H + h)*W + w)*C + c
        const uint64_t C = Cout_;
        (void)channelsLast;
        return ((((static_cast<uint64_t>(n) * D + d) * H + h) * W + w) * C + c);
    }

private:
    // GM pointers
    __gm__ float *xGm_ {nullptr};
    __gm__ float *wGm_ {nullptr};
    __gm__ float *bGm_ {nullptr};
    __gm__ float *yGm_ {nullptr};
    __gm__ float *ws0_ {nullptr};
    __gm__ float *ws1_ {nullptr};

    // Shapes
    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Di_{0}, Hi_{0}, Wi_{0};
    uint32_t kD_{0}, kH_{0}, kW_{0};
    uint32_t Do_{0}, Ho_{0}, Wo_{0};
    uint32_t D1_{0}, H1_{0}, W1_{0};
    uint32_t D2_{0}, H2_{0}, W2_{0};

    // Strides
    uint32_t strideX_W_{0}, strideX_H_{0}, strideX_D_{0}, strideX_C_{0}, strideX_N_{0};
    uint32_t strideWS_C_{0}, strideWS_W_{0}, strideWS_H_{0}, strideWS_D_{0}, strideWS_N_{0};
    uint32_t strideWS1_C_{0}, strideWS1_W_{0}, strideWS1_H_{0}, strideWS1_D_{0}, strideWS1_N_{0};
    uint32_t strideY_W_{0}, strideY_H_{0}, strideY_D_{0}, strideY_C_{0}, strideY_N_{0};
};
