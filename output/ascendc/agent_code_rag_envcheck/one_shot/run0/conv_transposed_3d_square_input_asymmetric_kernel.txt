namespace {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDivU32(uint64_t a, uint32_t b)
{
    return static_cast<uint32_t>((a + b - 1) / b);
}

} // namespace

namespace optiling {

// We'll process outputs in 1-D tiles of elements and compute each element independently.
// UB usage per core is tiny (a few scalars), so tileLength mainly controls load balance.
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t HARD_CAP_ELEMS = 16384;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed3dSquareInputAsymmetricKernelCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    const uint32_t N  = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t Ci = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t Di = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    const uint32_t Hi = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(3));
    const uint32_t Wi = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(4));

    const uint32_t WiCin = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0)); // in_channels
    const uint32_t Co    = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1)); // out_channels
    const uint32_t Kd    = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(2));
    const uint32_t Kh    = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(3));
    const uint32_t Kw    = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(4));

    if (WiCin != Ci) {
        return ge::GRAPH_FAILED;
    }

    // Transposed conv3d with stride=1, padding=0, dilation=1 -> full convolution
    const uint32_t Do = Di + Kd - 1;
    const uint32_t Ho = Hi + Kh - 1;
    const uint32_t Wo = Wi + Kw - 1;

    const uint64_t totalOutLen = static_cast<uint64_t>(N) * Co * Do * Ho * Wo;

    // Choose tile length primarily for scheduling (we compute element-by-element).
    uint32_t tileLen = HARD_CAP_ELEMS;
    if (totalOutLen < tileLen) {
        const uint32_t aligned = AlignDown(static_cast<uint32_t>(totalOutLen), ALIGN_ELEMS);
        tileLen = (aligned > 0) ? aligned : static_cast<uint32_t>(totalOutLen);
    }

    const uint32_t numTiles = (tileLen == 0) ? 1u : CeilDivU32(totalOutLen, tileLen);
    uint32_t blockDim = numTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cin(Ci);
    tiling.set_cout(Co);
    tiling.set_di(Di);
    tiling.set_hi(Hi);
    tiling.set_wi(Wi);
    tiling.set_kd(Kd);
    tiling.set_kh(Kh);
    tiling.set_kw(Kw);
    tiling.set_do_(Do);
    tiling.set_ho_(Ho);
    tiling.set_wo_(Wo);
    tiling.set_totalOutLen(totalOutLen);
    tiling.set_tileLength(tileLen);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *wk = context->GetWorkspaceSizes(1);
    wk[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvTransposed3dSquareInputAsymm {
public:
    __aicore__ inline KernelConvTransposed3dSquareInputAsymm() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t di, uint32_t hi, uint32_t wi,
                                uint32_t kd, uint32_t kh, uint32_t kw,
                                uint32_t do_, uint32_t ho_, uint32_t wo_,
                                uint64_t totalOutLen, uint32_t tileLength)
    {
        n_ = n; cin_ = cin; cout_ = cout;
        di_ = di; hi_ = hi; wi_ = wi;
        kd_ = kd; kh_ = kh; kw_ = kw;
        doo_ = do_; hoo_ = ho_; woo_ = wo_;
        totalOutLen_ = totalOutLen;
        tileLen_ = tileLength;

        const uint64_t xSize = static_cast<uint64_t>(n_) * cin_ * di_ * hi_ * wi_;
        const uint64_t wSize = static_cast<uint64_t>(cin_) * cout_ * kd_ * kh_ * kw_;
        const uint64_t ySize = totalOutLen_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(xBufQ_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(wBufQ_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(tmpBufQ_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(yBufQ_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalOutLen_ == 0 || tileLen_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * tileLen_;
        uint64_t base = static_cast<uint64_t>(blockIdx) * tileLen_;

        while (base < totalOutLen_) {
            uint32_t curLen = static_cast<uint32_t>(totalOutLen_ - base);
            if (curLen > tileLen_) curLen = tileLen_;
            ComputeRange(base, curLen);
            base += strideElems;
        }
    }

private:
    __aicore__ inline void ComputeRange(uint64_t base, uint32_t len)
    {
        // Output strides for unravel
        const uint64_t s1 = static_cast<uint64_t>(doo_) * hoo_ * woo_;
        const uint64_t s2 = static_cast<uint64_t>(hoo_) * woo_;
        const uint64_t s3 = static_cast<uint64_t>(woo_);

        for (uint32_t j = 0; j < len; ++j) {
            const uint64_t outIdx = base + j;

            uint64_t t = outIdx;
            const uint32_t n  = static_cast<uint32_t>(t / (static_cast<uint64_t>(cout_) * s1));
            t -= static_cast<uint64_t>(n) * cout_ * s1;
            const uint32_t oc = static_cast<uint32_t>(t / s1);
            t -= static_cast<uint64_t>(oc) * s1;
            const uint32_t od = static_cast<uint32_t>(t / s2);
            t -= static_cast<uint64_t>(od) * s2;
            const uint32_t oh = static_cast<uint32_t>(t / s3);
            const uint32_t ow = static_cast<uint32_t>(t % s3);

            bool hasInit = false;
            AscendC::LocalTensor<float> yLocal;

            for (uint32_t ic = 0; ic < cin_; ++ic) {
                for (uint32_t kd = 0; kd < kd_; ++kd) {
                    // input depth index
                    int32_t id = static_cast<int32_t>(od) - static_cast<int32_t>(kd);
                    if (id < 0 || id >= static_cast<int32_t>(di_)) continue;

                    for (uint32_t kh = 0; kh < kh_; ++kh) {
                        int32_t ih = static_cast<int32_t>(oh) - static_cast<int32_t>(kh);
                        if (ih < 0 || ih >= static_cast<int32_t>(hi_)) continue;

                        for (uint32_t kw = 0; kw < kw_; ++kw) {
                            int32_t iw = static_cast<int32_t>(ow) - static_cast<int32_t>(kw);
                            if (iw < 0 || iw >= static_cast<int32_t>(wi_)) continue;

                            // Load x[n, ic, id, ih, iw] -> xLocal[0]
                            AscendC::LocalTensor<float> xLocal = xBufQ_.AllocTensor<float>();
                            const uint64_t xOff =
                                ((((static_cast<uint64_t>(n) * cin_ + ic) * di_ + static_cast<uint32_t>(id)) * hi_
                                   + static_cast<uint32_t>(ih)) * wi_ + static_cast<uint32_t>(iw));
                            AscendC::DataCopy(xLocal, xGm_[xOff], 1);

                            // Load w[ic, oc, kd, kh, kw] -> wLocal[0]
                            AscendC::LocalTensor<float> wLocal = wBufQ_.AllocTensor<float>();
                            const uint64_t wOff =
                                ((((static_cast<uint64_t>(ic) * cout_ + oc) * kd_ + kd) * kh_ + kh) * kw_ + kw);
                            AscendC::DataCopy(wLocal, wGm_[wOff], 1);

                            // temp = xLocal * wLocal
                            AscendC::LocalTensor<float> tmpLocal = tmpBufQ_.AllocTensor<float>();
                            AscendC::Mul(tmpLocal, xLocal, wLocal, 1);

                            if (!hasInit) {
                                yLocal = yBufQ_.AllocTensor<float>();
                                AscendC::DataCopy(yLocal, tmpLocal, 1); // initialize sum
                                hasInit = true;
                            } else {
                                AscendC::Add(yLocal, yLocal, tmpLocal, 1);
                            }

                            tmpBufQ_.FreeTensor(tmpLocal);
                            wBufQ_.FreeTensor(wLocal);
                            xBufQ_.FreeTensor(xLocal);
                        }
                    }
                }
            }

            // Store y if it was initialized (it always should be for valid shapes)
            if (hasInit) {
                const uint64_t yOff = outIdx;
                AscendC::DataCopy(yGm_[yOff], yLocal, 1);
                yBufQ_.FreeTensor(yLocal);
            } else {
                // Fallback: write zero if no contributions (should not occur)
                AscendC::LocalTensor<float> yZero = yBufQ_.AllocTensor<float>();
                // Create zero by subtracting same buffer after initialization
                AscendC::LocalTensor<float> ztmp = tmpBufQ_.AllocTensor<float>();
                // Load a dummy weight and x, then compute zero by mul and sub
                AscendC::Mul(ztmp, ztmp, ztmp, 1); // ztmp becomes 0 if initially undefined behavior avoided by zero-length? ensure zero via DataCopy from y itself if needed
                AscendC::DataCopy(yZero, ztmp, 1);
                const uint64_t yOff = outIdx;
                AscendC::DataCopy(yGm_[yOff], yZero, 1);
                tmpBufQ_.FreeTensor(ztmp);
                yBufQ_.FreeTensor(yZero);
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> xBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> wBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpBufQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yBufQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cin_{0}, cout_{0};
    uint32_t di_{0}, hi_{0}, wi_{0};
    uint32_t kd_{0}, kh_{0}, kw_{0};
    uint32_t doo_{0}, hoo_{0}, woo_{0};
    uint64_t totalOutLen_{0};
    uint32_t tileLen_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_square_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    GET_TILING_DATA(tiling_data, tilingGm);

    KernelConvTransposed3dSquareInputAsymm op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.di, tiling_data.hi, tiling_data.wi,
            tiling_data.kd, tiling_data.kh, tiling_data.kw,
            tiling_data.do_, tiling_data.ho_, tiling_data.wo_,
            tiling_data.totalOutLen, tiling_data.tileLength);
    op.Process();
}
