namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMishMishCustomTilingData tiling;

    // Input x: [N, C, H, W]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);

    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const auto &x_dims = x_shape->GetOriginShape().GetDims();
    const auto &w_dims = w_shape->GetOriginShape().GetDims();

    if (x_dims.size() != 4 || w_dims.size() != 4) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N  = static_cast<uint32_t>(x_dims[0]);
    const uint32_t C  = static_cast<uint32_t>(x_dims[1]);
    const uint32_t H  = static_cast<uint32_t>(x_dims[2]);
    const uint32_t W  = static_cast<uint32_t>(x_dims[3]);

    const uint32_t Cout = static_cast<uint32_t>(w_dims[0]);
    const uint32_t Cin  = static_cast<uint32_t>(w_dims[1]);
    const uint32_t Kh   = static_cast<uint32_t>(w_dims[2]);
    const uint32_t Kw   = static_cast<uint32_t>(w_dims[3]);

    if (Cin != C) {
        return ge::GRAPH_FAILED;
    }

    // stride=1, padding=0, dilation=1 (as per model)
    const uint32_t Oh = (H >= Kh) ? (H - Kh + 1u) : 0u;
    const uint32_t Ow = (W >= Kw) ? (W - Kw + 1u) : 0u;

    // Optional bias presence
    bool hasBias = false;
    if (context->GetInputsNum() >= 3) {
        const gert::Shape *b_shape = context->GetInputShape(2);
        if (b_shape != nullptr && b_shape->GetOriginShape().GetShapeSize() > 0) {
            hasBias = true;
        }
    }

    // Block dim heuristic: one block per (N*Oh*Ow) up to a cap
    uint64_t spatialSites = static_cast<uint64_t>(N) * static_cast<uint64_t>(Oh) * static_cast<uint64_t>(Ow);
    uint32_t blockDim = static_cast<uint32_t>(spatialSites);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_c(C);
    tiling.set_h(H);
    tiling.set_w(W);
    tiling.set_cout(Cout);
    tiling.set_kh(Kh);
    tiling.set_kw(Kw);
    tiling.set_oh(Oh);
    tiling.set_ow(Ow);
    tiling.set_hasBias(hasBias ? 1u : 0u);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include <cmath>

class KernelConv2dMishMishCustom {
public:
    __aicore__ inline KernelConv2dMishMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t wIn,
                                uint32_t cout, uint32_t kh, uint32_t kw,
                                uint32_t oh, uint32_t ow, uint32_t hasBias)
    {
        x_ = reinterpret_cast<__gm__ float*>(x);
        w_ = reinterpret_cast<__gm__ float*>(w);
        b_ = reinterpret_cast<__gm__ float*>(b);
        y_ = reinterpret_cast<__gm__ float*>(y);

        N_ = n; C_ = c; H_ = h; W_ = wIn;
        Cout_ = cout; Kh_ = kh; Kw_ = kw;
        Oh_ = oh; Ow_ = ow;
        hasBias_ = hasBias != 0u;
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0 || C_ == 0 || H_ == 0 || W_ == 0 || Cout_ == 0 || Kh_ == 0 || Kw_ == 0) {
            return;
        }
        if (Oh_ == 0 || Ow_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t totalSites = static_cast<uint64_t>(N_) * static_cast<uint64_t>(Oh_) * static_cast<uint64_t>(Ow_);
        if (blockIdx >= totalSites) return;

        for (uint64_t site = blockIdx; site < totalSites; site += blockNum) {
            // Decode (n, oh, ow) from site index
            const uint32_t sitesPerN = Oh_ * Ow_;
            const uint32_t n = static_cast<uint32_t>(site / sitesPerN);
            const uint32_t rem = static_cast<uint32_t>(site % sitesPerN);
            const uint32_t oh = rem / Ow_;
            const uint32_t ow = rem % Ow_;

            // For each output channel
            for (uint32_t oc = 0; oc < Cout_; ++oc) {
                float acc = hasBias_ ? b_[oc] : 0.0f;

                // Convolution: stride=1, pad=0, dilation=1
                for (uint32_t ic = 0; ic < C_; ++ic) {
                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                        const uint32_t ih = oh + kh;
                        const uint32_t baseX_h = ((n * C_ + ic) * H_ + ih) * W_;
                        const uint32_t baseW_h = (((oc * C_ + ic) * Kh_ + kh) * Kw_);
                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                            const uint32_t iw = ow + kw;
                            const float xv = x_[baseX_h + iw];
                            const float wv = w_[baseW_h + kw];
                            acc = acc + xv * wv;
                        }
                    }
                }

                // y = mish(mish(acc)); mish(x) = x * tanh(softplus(x)), softplus(x)=ln(1+exp(x))
                float y1 = mish(acc);
                float y2 = mish(y1);

                const uint32_t yIndex = (((n * Cout_ + oc) * Oh_ + oh) * Ow_) + ow;
                y_[yIndex] = y2;
            }
        }
    }

private:
    __aicore__ inline float mish(float x) const
    {
        // numerically stable softplus and tanh
        float sp;
        if (x > 20.0f) {
            sp = x; // softplus ~ x for large positive
        } else if (x < -20.0f) {
            sp = std::exp(x); // softplus ~ exp(x) for large negative
        } else {
            sp = std::log1p(std::exp(x));
        }
        float t = std::tanh(sp);
        return x * t;
    }

private:
    __gm__ float* x_ {nullptr};
    __gm__ float* w_ {nullptr};
    __gm__ float* b_ {nullptr};
    __gm__ float* y_ {nullptr};

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t Cout_{0}, Kh_{0}, Kw_{0};
    uint32_t Oh_{0}, Ow_{0};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_mish_mish_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMishMishCustom op;
    op.Init(x, weight, conv_bias, y,
            tiling_data.n, tiling_data.c, tiling_data.h, tiling_data.w,
            tiling_data.cout, tiling_data.kh, tiling_data.kw,
            tiling_data.oh, tiling_data.ow, tiling_data.hasBias);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor conv2d_mish_mish_impl_npu(const at::Tensor& x,
                                     const at::Tensor& weight,
                                     c10::optional<at::Tensor> conv_bias_opt) {
    // Output shape inference on host for allocation
    TORCH_CHECK(x.dim() == 4, \"x must be 4D NCHW\");
    TORCH_CHECK(weight.dim() == 4, \"weight must be 4D OIHW\");
    TORCH_CHECK(x.size(1) == weight.size(1), \"in_channels mismatch\");

    const int64_t N = x.size(0);
    const int64_t C = x.size(1);
    const int64_t H = x.size(2);
    const int64_t W = x.size(3);
    const int64_t Cout = weight.size(0);
    const int64_t Kh = weight.size(2);
    const int64_t Kw = weight.size(3);
    TORCH_CHECK(H >= Kh && W >= Kw, \"kernel larger than input without padding is invalid\");

    const int64_t Oh = H - Kh + 1;
    const int64_t Ow = W - Kw + 1;

    at::Tensor y = at::empty({N, Cout, Oh, Ow}, x.options());

    if (conv_bias_opt.has_value()) {
        EXEC_NPU_CMD(aclnnConv2dMishMishCustom, x, weight, conv_bias_opt.value(), y);
    } else {
        at::Tensor empty_bias; // undefined tensor to indicate no bias
        EXEC_NPU_CMD(aclnnConv2dMishMishCustom, x, weight, empty_bias, y);
    }
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"conv2d_mish_mish_custom\", &conv2d_mish_mish_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"conv2d_mish_mish_custom\", &conv2d_mish_mish_impl_npu, \"Conv2d + Mish + Mish fused (Conv2dMishMishCustom on NPU)\");
}
