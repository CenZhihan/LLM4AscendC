#include "vanilla_rnn_hidden_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
}

// UB planning:
// Buffers per-core (floats):
//   x_row: I
//   h_prev: H
//   pre_tile: tileH
//   tmp_tile: tileH
//   w_tile: tileH
//   y_tile: tileO
//   wout_tile: tileO
// Total floats = I + H + (3 * tileH) + (2 * tileO)
// Budget bytes = 192 KB conservative => 192*1024 / 4 floats = 49152 floats.
static constexpr uint32_t UB_BUDGET_FLOATS = (192u * 1024u) / 4u;
static constexpr uint32_t MIN_TILE = 16;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    VanillaRnnHiddenCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);   // [T,B,I]
    const gert::Shape *h0_shape = context->GetInputShape(1);  // [B,H]
    const gert::Shape *w1t_shape = context->GetInputShape(2); // [K,H]
    const gert::Shape *b1_shape = context->GetInputShape(3);  // [H]
    const gert::Shape *w2t_shape = context->GetInputShape(4); // [H,O]
    const gert::Shape *b2_shape = context->GetInputShape(5);  // [O]

    uint32_t T = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    uint32_t B = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    uint32_t I = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(2));
    uint32_t H = static_cast<uint32_t>(b1_shape->GetOriginShape().GetDim(0));
    uint32_t O = static_cast<uint32_t>(b2_shape->GetOriginShape().GetDim(0));

    // initial guesses
    uint32_t tileH = (H < 1024u) ? H : 1024u;
    uint32_t tileO = (O < 1024u) ? O : 1024u;
    tileH = (tileH < MIN_TILE) ? H : AlignDown(tileH, MIN_TILE);
    tileO = (tileO < MIN_TILE) ? O : AlignDown(tileO, MIN_TILE);

    // Fit buffers within budget
    auto fits = [&](uint32_t tH, uint32_t tO)->bool {
        uint64_t need = static_cast<uint64_t>(I) + static_cast<uint64_t>(H) +
                        static_cast<uint64_t>(3u * tH) + static_cast<uint64_t>(2u * tO);
        return need <= UB_BUDGET_FLOATS;
    };

    if (!fits(tileH, tileO)) {
        // Try reducing the larger contributor iteratively
        for (int iter = 0; iter < 64 && !fits(tileH, tileO); ++iter) {
            if (3u * tileH >= 2u * tileO && tileH > MIN_TILE) {
                tileH = AlignDown(tileH / 2u, MIN_TILE);
                if (tileH == 0) tileH = MIN_TILE;
                if (tileH > H) tileH = AlignDown(H, MIN_TILE);
            } else if (tileO > MIN_TILE) {
                tileO = AlignDown(tileO / 2u, MIN_TILE);
                if (tileO == 0) tileO = MIN_TILE;
                if (tileO > O) tileO = AlignDown(O, MIN_TILE);
            } else {
                break;
            }
        }
    }
    if (tileH == 0) tileH = (H == 0) ? 0 : ((H < MIN_TILE) ? H : AlignDown(H, MIN_TILE));
    if (tileO == 0) tileO = (O == 0) ? 0 : ((O < MIN_TILE) ? O : AlignDown(O, MIN_TILE));

    // block dim across batches primarily
    uint32_t blockDim = B;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_T(T);
    tiling.set_B(B);
    tiling.set_I(I);
    tiling.set_H(H);
    tiling.set_O(O);
    tiling.set_tileH((tileH == 0) ? H : tileH);
    tiling.set_tileO((tileO == 0) ? O : tileO);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

    // x: [T,B,I] => y: [T,B,O]

kernel_src="""

#include "kernel_operator.h"

class KernelVanillaRnnHiddenCustom {
public:
    __aicore__ inline KernelVanillaRnnHiddenCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR h0,
                                GM_ADDR w1t, GM_ADDR b1,
                                GM_ADDR w2t, GM_ADDR b2,
                                GM_ADDR y,
                                uint32_t T, uint32_t B, uint32_t I, uint32_t H, uint32_t O,
                                uint32_t tileH, uint32_t tileO)
    {
        T_ = T; B_ = B; I_ = I; H_ = H; O_ = O;
        tileH_ = (tileH == 0) ? H : tileH;
        tileO_ = (tileO == 0) ? O : tileO;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)T_ * B_ * I_);
        h0Gm_.SetGlobalBuffer((__gm__ float*)h0, (uint64_t)B_ * H_);
        w1tGm_.SetGlobalBuffer((__gm__ float*)w1t, (uint64_t)(I_ + H_) * H_);
        b1Gm_.SetGlobalBuffer((__gm__ float*)b1, H_);
        w2tGm_.SetGlobalBuffer((__gm__ float*)w2t, (uint64_t)H_ * O_);
        b2Gm_.SetGlobalBuffer((__gm__ float*)b2, O_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)T_ * B_ * O_);

        // Queues/buffers
        pipe_.InitBuffer(xRowQ_, 1, I_ * sizeof(float));
        pipe_.InitBuffer(hPrevQ_, 1, H_ * sizeof(float));
        pipe_.InitBuffer(preQ_, 1, tileH_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tileH_ * sizeof(float));
        pipe_.InitBuffer(wTileQ_, 1, tileH_ * sizeof(float));
        pipe_.InitBuffer(yTileQ_, 1, tileO_ * sizeof(float));
        pipe_.InitBuffer(wOutTileQ_, 1, tileO_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (T_ == 0 || B_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t b = blockIdx; b < B_; b += blockNum) {
            ProcessOneBatch(b);
        }
    }

private:
    __aicore__ inline float LoadLocalScalar(const AscendC::LocalTensor<float>& vec, uint32_t idx) {
        // Access one element. Provided for readability; implementation relies on DSL support.
        return vec.GetValue(idx);
    }

    __aicore__ inline void ProcessOneBatch(uint32_t bIdx)
    {
        // h_prev buffer persistent across time steps
        AscendC::LocalTensor<float> hPrev = hPrevQ_.AllocTensor<float>();
        AscendC::DataCopy(hPrev, h0Gm_[bIdx * H_], H_);

        for (uint32_t t = 0; t < T_; ++t) {
            // Load x[t, b, :]
            AscendC::LocalTensor<float> xRow = xRowQ_.AllocTensor<float>();
            uint64_t x_off = ((uint64_t)t * B_ + bIdx) * I_;
            AscendC::DataCopy(xRow, xGm_[x_off], I_);

            // For better locality, accumulate y per o-tile while iterating over h tiles.
            for (uint32_t oStart = 0; oStart < O_; oStart += tileO_) {
                const uint32_t oLen = AscendC::min((uint32_t)tileO_, O_ - oStart);

                // y_tile = b2[oStart:oStart+oLen]
                AscendC::LocalTensor<float> yTile = yTileQ_.AllocTensor<float>();
                AscendC::DataCopy(yTile, b2Gm_[oStart], oLen);

                // Iterate H in tiles: compute h_tile and accumulate into yTile
                for (uint32_t hStart = 0; hStart < H_; hStart += tileH_) {
                    const uint32_t hLen = AscendC::min((uint32_t)tileH_, H_ - hStart);

                    // pre_tile = b1[hStart:hStart+hLen]
                    AscendC::LocalTensor<float> pre = preQ_.AllocTensor<float>();
                    AscendC::DataCopy(pre, b1Gm_[hStart], hLen);

                    AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                    AscendC::LocalTensor<float> wTile = wTileQ_.AllocTensor<float>();

                    // Contribution from x: for k in [0, I)
                    for (uint32_t k = 0; k < I_; ++k) {
                        // s = xRow[k]
                        float s = LoadLocalScalar(xRow, k);
                        // wTile = w1t[k, hStart:hStart+hLen]
                        uint64_t w1_row_off = ((uint64_t)k) * H_ + hStart;
                        AscendC::DataCopy(wTile, w1tGm_[w1_row_off], hLen);
                        // tmp = wTile * s
                        AscendC::Muls(tmp, wTile, s, hLen);
                        // pre += tmp
                        AscendC::Add(pre, pre, tmp, hLen);
                    }

                    // Contribution from hPrev: for k in [0, H)
                    for (uint32_t k = 0; k < H_; ++k) {
                        float s = LoadLocalScalar(hPrev, k);
                        uint64_t w1_row_off = ((uint64_t)I_ + k) * H_ + hStart;
                        AscendC::DataCopy(wTile, w1tGm_[w1_row_off], hLen);
                        AscendC::Muls(tmp, wTile, s, hLen);
                        AscendC::Add(pre, pre, tmp, hLen);
                    }

                    // h_tile = tanh(pre)
                    AscendC::Tanh(pre, pre, hLen);

                    // Write back h_tile to hPrev for next time
                    AscendC::DataCopy(hPrev[hStart], pre, hLen);

                    // y accumulation: yTile += sum_j pre[j] * w2t[hStart+j, oStart:oStart+oLen]
                    AscendC::LocalTensor<float> wOutTile = wOutTileQ_.AllocTensor<float>();
                    for (uint32_t j = 0; j < hLen; ++j) {
                        float s = LoadLocalScalar(pre, j);
                        uint64_t w2_row_off = ((uint64_t)hStart + j) * O_ + oStart;
                        AscendC::DataCopy(wOutTile, w2tGm_[w2_row_off], oLen);
                        AscendC::Muls(wOutTile, wOutTile, s, oLen);
                        AscendC::Add(yTile, yTile, wOutTile, oLen);
                    }

                    wOutTileQ_.FreeTensor(wOutTile);
                    wTileQ_.FreeTensor(wTile);
                    tmpQ_.FreeTensor(tmp);
                    preQ_.FreeTensor(pre);
                }

                // Store y_tile to y[t, b, oStart:oStart+oLen]
                uint64_t y_off = ((uint64_t)t * B_ + bIdx) * O_ + oStart;
                AscendC::DataCopy(yGm_[y_off], yTile, oLen);
                yTileQ_.FreeTensor(yTile);
            }

            xRowQ_.FreeTensor(xRow);
        }

        hPrevQ_.FreeTensor(hPrev);
    }

private:
    AscendC::TPipe pipe_;

    AscendC::TQue<AscendC::TPosition::VECIN, 1>   xRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> hPrevQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> preQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> yTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> wOutTileQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> h0Gm_;
    AscendC::GlobalTensor<float> w1tGm_;
    AscendC::GlobalTensor<float> b1Gm_;
    AscendC::GlobalTensor<float> w2tGm_;
    AscendC::GlobalTensor<float> b2Gm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t T_{0}, B_{0}, I_{0}, H_{0}, O_{0};
    uint32_t tileH_{0}, tileO_{0};
};

extern "C" __global__ __aicore__ void vanilla_rnn_hidden_custom(GM_ADDR x, GM_ADDR h0,
                                                                GM_ADDR w_i2h_t, GM_ADDR b_i2h,
                                                                GM_ADDR w_h2o_t, GM_ADDR b_h2o,
                                                                GM_ADDR y,
                                                                GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelVanillaRnnHiddenCustom op;
    op.Init(x, h0, w_i2h_t, b_i2h, w_h2o_t, b_h2o, y,
            tiling_data.T, tiling_data.B, tiling_data.I, tiling_data.H, tiling_data.O,
            tiling_data.tileH, tiling_data.tileO);
    op.Process();
}

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_contig_fp32(const at::Tensor& t) {
    at::Tensor r = t;
    if (r.scalar_type() != at::kFloat) r = r.to(at::kFloat);
    if (!r.is_contiguous()) r = r.contiguous();
    return r;
}

at::Tensor vanilla_rnn_hidden_impl_npu(const at::Tensor& x,
                                       const at::Tensor& h0,
                                       const at::Tensor& w_i2h,
                                       const at::Tensor& b_i2h,
                                       const at::Tensor& w_h2o,
                                       const at::Tensor& b_h2o) {
    TORCH_CHECK(x.dim() == 3, "x must be [T,B,I]");
    TORCH_CHECK(h0.dim() == 2, "h0 must be [B,H]");
    TORCH_CHECK(w_i2h.dim() == 2, "w_i2h must be [H,K]");
    TORCH_CHECK(b_i2h.dim() == 1, "b_i2h must be [H]");
    TORCH_CHECK(w_h2o.dim() == 2, "w_h2o must be [O,H]");
    TORCH_CHECK(b_h2o.dim() == 1, "b_h2o must be [O]");

    auto x_ = to_contig_fp32(x);
    auto h0_ = to_contig_fp32(h0);
    auto w_i2h_ = to_contig_fp32(w_i2h);
    auto b_i2h_ = to_contig_fp32(b_i2h);
    auto w_h2o_ = to_contig_fp32(w_h2o);
    auto b_h2o_ = to_contig_fp32(b_h2o);

    // Transpose weights for kernel-friendly layout
    at::Tensor w_i2h_t = w_i2h_.transpose(0, 1).contiguous(); // [K,H]
    at::Tensor w_h2o_t = w_h2o_.transpose(0, 1).contiguous(); // [H,O]

    // Prepare output
    auto sizes = x_.sizes().vec();
    sizes[2] = b_h2o_.size(0); // O
    at::Tensor y = at::empty(sizes, x_.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnVanillaRnnHiddenCustom, x_, h0_, w_i2h_t, b_i2h_, w_h2o_t, b_h2o_, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("vanilla_rnn_hidden_custom", &vanilla_rnn_hidden_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("vanilla_rnn_hidden_custom", &vanilla_rnn_hidden_impl_npu,
          "Vanilla RNN hidden+projection fused op (VanillaRnnHiddenCustom on NPU)");
}
