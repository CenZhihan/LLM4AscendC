#include "matmul_subtract_multiply_relu_custom_tiling.h"

namespace optiling {

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

// UB budget and simple tiling strategy:
// UB needs per K-chunk:
//   xChunk: 1 * tileK
//   wChunk: tileN * tileK
//   prod:   1 * tileK
//   scalar buffers: a few 1-element buffers
// floats per step ~ (tileN + 2) * tileK + small
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
static constexpr uint32_t ALIGN_K = 256;
static constexpr uint32_t MAX_TILE_N = 1; // keep 1 for simplicity and robustness

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0); // [M, K]
    const gert::Shape *w_shape = context->GetInputShape(1); // [N, K] (F.linear weight layout)
    const gert::Shape *b_shape = context->GetInputShape(2); // [N]
    (void)b_shape;

    if (x_shape->GetOriginShape().GetDimNum() != 2 || w_shape->GetOriginShape().GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(x_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(0));
    const uint32_t Kw = static_cast<uint32_t>(w_shape->GetOriginShape().GetDim(1));
    if (Kw != K) {
        return ge::GRAPH_FAILED;
    }

    uint32_t tileN = MAX_TILE_N;
    // compute tileK from UB budget: (tileN + 2) * tileK * 4 < budget
    const uint32_t perElemBytes = 4u;
    uint32_t maxTileK = UB_BUDGET_BYTES / perElemBytes;
    if (maxTileK > 0) {
        maxTileK = maxTileK / (tileN + 2u);
    }
    if (maxTileK == 0) maxTileK = ALIGN_K;
    uint32_t tileK = AlignDown(maxTileK, ALIGN_K);
    if (tileK == 0) tileK = (K >= ALIGN_K ? ALIGN_K : K);
    if (tileK > K) tileK = AlignDown(K, ALIGN_K) > 0 ? AlignDown(K, ALIGN_K) : K;

    // blockDim - one block processes stride over rows
    uint32_t blockDim = (M < 1u) ? 1u : M;
    if (blockDim > 24u) blockDim = 24u;
    if (blockDim == 0u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    MatmulSubtractMultiplyReluCustomTilingData tiling;
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileK(tileK);
    tiling.set_tileN(tileN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

// Fused kernel: y = ReLU(((x @ w^T) + b) - sub) * mul
// Shapes:
//   x: [M, K] (row-major)
//   w: [N, K] (row-major; F.linear weight layout)
//   b: [N]
//   sub: [1] (scalar tensor)
//   mul: [1] (scalar tensor)
//   y: [M, N]
//
// Tiling:
//   - Iterate rows with grid-stride across blocks
//   - Process one output column at a time (tileN=1) for simplicity
//   - Iterate K in chunks of tileK, reduce by sum into a 1-element accumulator

class KernelMatmulSubtractMultiplyReluCustom {
public:
    __aicore__ inline KernelMatmulSubtractMultiplyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR sub, GM_ADDR mul, GM_ADDR y,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileK, uint32_t tileN)
    {
        M_ = M; N_ = N; K_ = K;
        tileK_ = tileK; tileN_ = (tileN == 0 ? 1u : tileN);

        xGm_.SetGlobalBuffer((__gm__ float*)x, static_cast<uint64_t>(M_) * K_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, static_cast<uint64_t>(N_) * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, N_);
        subGm_.SetGlobalBuffer((__gm__ float*)sub, 1);
        mulGm_.SetGlobalBuffer((__gm__ float*)mul, 1);
        yGm_.SetGlobalBuffer((__gm__ float*)y, static_cast<uint64_t>(M_) * N_);

        // Queues and buffers
        pipe_.InitBuffer(xVecQ_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(wVecQ_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(prodVecQ_, 1, tileK_ * sizeof(float));
        pipe_.InitBuffer(scalarQ_, 4, sizeof(float)); // accum, sum, sub, mul
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Load scalar sub and mul into UB once per block
        AscendC::LocalTensor<float> subLocal = scalarQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> mulLocal = scalarQ_.AllocTensor<float>();
        AscendC::DataCopy(subLocal, subGm_[0], 1);
        AscendC::DataCopy(mulLocal, mulGm_[0], 1);

        for (uint32_t row = blockIdx; row < M_; row += blockNum) {
            // Each row independently
            for (uint32_t col = 0; col < N_; ++col) {
                // Initialize accumulator to 0
                AscendC::LocalTensor<float> acc = scalarQ_.AllocTensor<float>();
                AscendC::Duplicate(acc, 0.0f, 1);

                // Accumulate over K in chunks
                for (uint32_t k0 = 0; k0 < K_; k0 += tileK_) {
                    uint32_t len = K_ - k0;
                    if (len > tileK_) len = tileK_;

                    // Load x row chunk and w row chunk
                    AscendC::LocalTensor<float> xChunk = xVecQ_.AllocTensor<float>();
                    AscendC::LocalTensor<float> wChunk = wVecQ_.AllocTensor<float>();
                    AscendC::DataCopy(xChunk, xGm_[row * K_ + k0], len);
                    AscendC::DataCopy(wChunk, wGm_[col * K_ + k0], len);

                    // Multiply and reduce sum to scalar
                    AscendC::LocalTensor<float> prod = prodVecQ_.AllocTensor<float>();
                    AscendC::Mul(prod, xChunk, wChunk, len);

                    AscendC::LocalTensor<float> sumBuf = scalarQ_.AllocTensor<float>();
                    AscendC::ReduceSum(sumBuf, prod, len); // sumBuf[0] = sum(prod)

                    // acc += sumBuf
                    AscendC::Add(acc, acc, sumBuf, 1);

                    // Free temporaries
                    scalarQ_.FreeTensor(sumBuf);
                    prodVecQ_.FreeTensor(prod);
                    wVecQ_.FreeTensor(wChunk);
                    xVecQ_.FreeTensor(xChunk);
                }

                // Add bias
                AscendC::LocalTensor<float> biasLocal = scalarQ_.AllocTensor<float>();
                AscendC::DataCopy(biasLocal, bGm_[col], 1);
                AscendC::Add(acc, acc, biasLocal, 1);

                // Subtract scalar 'sub' and multiply by 'mul'
                AscendC::Sub(acc, acc, subLocal, 1);
                AscendC::Mul(acc, acc, mulLocal, 1);

                // ReLU
                AscendC::Maxs(acc, acc, 0.0f, 1);

                // Store to Y
                AscendC::DataCopy(yGm_[row * N_ + col], acc, 1);

                // Free locals
                scalarQ_.FreeTensor(biasLocal);
                scalarQ_.FreeTensor(acc);
            }
        }

        scalarQ_.FreeTensor(mulLocal);
        scalarQ_.FreeTensor(subLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> xVecQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> prodVecQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 4> scalarQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> subGm_;
    AscendC::GlobalTensor<float> mulGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tileK_{0}, tileN_{1};
};

extern "C" __global__ __aicore__ void matmul_subtract_multiply_relu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR sub, GM_ADDR mul, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulSubtractMultiplyReluCustom op;
    op.Init(x, w, b, sub, mul, y,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileK, tiling_data.tileN);
    op.Process();
}
