#include "conv_standard_2d_asymmetric_input_asymmetric_kernel_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core UB budget (conservative)
static constexpr uint32_t ALIGN_WO = 8;                   // favor DMA friendly width tiles
static constexpr uint32_t MAX_BLOCK_DIM = 24;            // typical aicore upper bound

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    using TD = ConvStandard2dAsymmetricInputAsymmetricKernelCustomTilingData;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);

    const int64_t N64  = xShape->GetOriginShape().GetDim(0);
    const int64_t C64  = xShape->GetOriginShape().GetDim(1);
    const int64_t H64  = xShape->GetOriginShape().GetDim(2);
    const int64_t W64  = xShape->GetOriginShape().GetDim(3);

    const int64_t Co64 = wShape->GetOriginShape().GetDim(0);
    const int64_t Cw64 = wShape->GetOriginShape().GetDim(1);
    const int64_t Kh64 = wShape->GetOriginShape().GetDim(2);
    const int64_t Kw64 = wShape->GetOriginShape().GetDim(3);

    // basic checks (groups=1, dilation=1, stride=1, padding=0 are assumed)
    if (C64 != Cw64) { return ge::GRAPH_FAILED; }
    if (Kh64 <= 0 || Kw64 <= 0) { return ge::GRAPH_FAILED; }
    if (H64 < Kh64 || W64 < Kw64) { return ge::GRAPH_FAILED; }

    const uint32_t N  = static_cast<uint32_t>(N64);
    const uint32_t C  = static_cast<uint32_t>(C64);
    const uint32_t H  = static_cast<uint32_t>(H64);
    const uint32_t W  = static_cast<uint32_t>(W64);
    const uint32_t Co = static_cast<uint32_t>(Co64);
    const uint32_t Kh = static_cast<uint32_t>(Kh64);
    const uint32_t Kw = static_cast<uint32_t>(Kw64);

    const uint32_t Ho = static_cast<uint32_t>(H - Kh + 1);
    const uint32_t Wo = static_cast<uint32_t>(W - Kw + 1);

    // UB budget in float elements
    const uint32_t budgetFloats = UB_BUDGET_BYTES / 4;

    // choose tileCo adaptively so that weight tile fits
    uint32_t tileCo = 8;
    if (tileCo > Co) tileCo = Co == 0 ? 1u : Co;
    auto weightElemsFor = [&](uint32_t tCo) { return tCo * C * Kh * Kw; };

    while (tileCo > 1 && weightElemsFor(tileCo) > budgetFloats / 2) {
        tileCo >>= 1;
    }
    if (weightElemsFor(tileCo) > budgetFloats - 1024u) {
        // still too large; fall back to minimum
        tileCo = 1;
    }
    if (tileCo == 0) tileCo = 1;

    // compute max tileWo from remaining UB
    // total = weights + inputPatch + outputTile
    // inputPatch per segment = C * Kh * (tileWo + Kw - 1)
    // outputTile = tileCo * tileWo
    uint32_t maxTileWo = 1;
    {
        const uint32_t usedByWeights = weightElemsFor(tileCo);
        const uint32_t guard = 1024; // leave headroom
        if (usedByWeights + guard < budgetFloats) {
            uint32_t rem = budgetFloats - usedByWeights - guard;
            if (C > 0 && Kh > 0) {
                // rem >= (C*Kh) * (tileWo + Kw - 1) + (tileCo * tileWo)
                // approximate: (C*Kh + tileCo) * tileWo <= rem - (C*Kh)*(Kw - 1)
                uint64_t denom = static_cast<uint64_t>(C) * Kh + tileCo;
                uint64_t numer = (rem > C * Kh * (Kw - 1)) ? (rem - C * Kh * (Kw - 1)) : 0u;
                if (denom > 0) {
                    maxTileWo = static_cast<uint32_t>(numer / denom);
                }
            }
        }
        if (maxTileWo < 1) maxTileWo = 1;
        if (maxTileWo > Wo) maxTileWo = Wo;
        maxTileWo = AlignDown(maxTileWo, ALIGN_WO);
        if (maxTileWo == 0) maxTileWo = (Wo >= ALIGN_WO) ? ALIGN_WO : Wo;
    }

    // block mapping: parallelize over output-channel tiles
    const uint32_t coTiles = CeilDiv(Co, tileCo);
    uint32_t blockDim = coTiles;
    if (blockDim > MAX_BLOCK_DIM) blockDim = MAX_BLOCK_DIM;
    if (blockDim < 1u) blockDim = 1u;

    context->SetBlockDim(blockDim);

    TD tiling;
    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_Co(Co);
    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_tileCo(tileCo);
    tiling.set_tileWo(maxTileWo);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

}

kernel_src="""
#include "kernel_operator.h"

class KernelConv2dDirect {
public:
    __aicore__ inline KernelConv2dDirect() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR weight, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t Co, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t tileCo, uint32_t tileWo)
    {
        N_ = N; C_ = C; H_ = H; W_ = W;
        Co_ = Co; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;
        tileCo_ = tileCo; tileWo_ = tileWo;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_ * C_ * H_ * W_);
        wGm_.SetGlobalBuffer((__gm__ float*)weight, (uint64_t)Co_ * C_ * Kh_ * Kw_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_ * Co_ * Ho_ * Wo_);

        // UB buffer planning
        uint32_t maxPatchW = tileWo_ + Kw_ - 1;
        uint64_t weightElems = (uint64_t)tileCo_ * C_ * Kh_ * Kw_;
        uint64_t inputElems  = (uint64_t)C_ * Kh_ * maxPatchW;
        uint64_t outputElems = (uint64_t)tileCo_ * tileWo_;

        uint64_t totalElems = weightElems + inputElems + outputElems;
        uint64_t totalBytes = totalElems * sizeof(float);
        pipe_.InitBuffer(calcBuf_, totalBytes);

        // layout offsets
        offWeight_ = 0;
        offInput_  = (uint64_t)weightElems * sizeof(float);
        offOutput_ = offInput_ + (uint64_t)inputElems * sizeof(float);
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t coTiles = (Co_ + tileCo_ - 1u) / tileCo_;

        for (uint32_t coTile = blockIdx; coTile < coTiles; coTile += blockNum) {
            const uint32_t coStart = coTile * tileCo_;
            const uint32_t curCo   = (coStart + tileCo_ <= Co_) ? tileCo_ : (Co_ - coStart);

            // weight tile copy: contiguous across curCo
            {
                AscendC::LocalTensor<float> wLocal = calcBuf_.GetWithOffset<float>(tileCo_ * C_ * Kh_ * Kw_, offWeight_);
                const uint64_t wOffGm = (uint64_t)coStart * C_ * Kh_ * Kw_;
                const uint64_t len    = (uint64_t)curCo * C_ * Kh_ * Kw_;
                AscendC::DataCopy(wLocal, wGm_[wOffGm], len);
            }

            for (uint32_t n = 0; n < N_; ++n) {
                for (uint32_t ho = 0; ho < Ho_; ++ho) {
                    for (uint32_t w0 = 0; w0 < Wo_; w0 += tileWo_) {
                        const uint32_t curWo = (w0 + tileWo_ <= Wo_) ? tileWo_ : (Wo_ - w0);
                        const uint32_t patchW = curWo + Kw_ - 1;

                        AscendC::LocalTensor<float> inLocal  = calcBuf_.GetWithOffset<float>(C_ * Kh_ * (tileWo_ + Kw_ - 1), offInput_);
                        AscendC::LocalTensor<float> wLocal   = calcBuf_.GetWithOffset<float>(tileCo_ * C_ * Kh_ * Kw_, offWeight_);
                        AscendC::LocalTensor<float> outLocal = calcBuf_.GetWithOffset<float>(tileCo_ * tileWo_, offOutput_);

                        // zero out output tile
                        for (uint32_t i = 0; i < curWo * curCo; ++i) {
                            outLocal[i] = 0.0f;
                        }

                        // load input patch: for each ci and kh row, contiguous copy of 'patchW'
                        for (uint32_t ci = 0; ci < C_; ++ci) {
                            for (uint32_t kh = 0; kh < Kh_; ++kh) {
                                const uint64_t gmOff = (((uint64_t)n * C_ + ci) * H_ + (ho + kh)) * W_ + w0;
                                const uint32_t locOff = (ci * Kh_ + kh) * (tileWo_ + Kw_ - 1); // max stride
                                AscendC::DataCopy(inLocal[locOff], xGm_[gmOff], patchW);
                            }
                        }

                        // compute outLocal
                        for (uint32_t coRel = 0; coRel < curCo; ++coRel) {
                            for (uint32_t wRel = 0; wRel < curWo; ++wRel) {
                                float acc = 0.0f;
                                // sum over ci, kh, kw
                                for (uint32_t ci = 0; ci < C_; ++ci) {
                                    for (uint32_t kh = 0; kh < Kh_; ++kh) {
                                        const uint32_t inRowBase = (ci * Kh_ + kh) * (tileWo_ + Kw_ - 1);
                                        const uint32_t inBase    = inRowBase + wRel; // add kw later
                                        const uint32_t wRowBase  = ((coRel * C_ + ci) * Kh_ + kh) * Kw_;
                                        for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                            const float xv = inLocal[inBase + kw];
                                            const float ww = wLocal[wRowBase + kw];
                                            acc += xv * ww;
                                        }
                                    }
                                }
                                outLocal[coRel * tileWo_ + wRel] = acc;
                            }
                        }

                        // store to GM: each output channel is contiguous across width
                        for (uint32_t coRel = 0; coRel < curCo; ++coRel) {
                            const uint32_t co = coStart + coRel;
                            const uint64_t yOff = (((uint64_t)n * Co_ + co) * Ho_ + ho) * Wo_ + w0;
                            const uint32_t locOff = coRel * tileWo_;
                            AscendC::DataCopy(yGm_[yOff], outLocal[locOff], curWo);
                        }
                    } // w0
                } // ho
            } // n
        } // coTile
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TBuf<AscendC::TPosition::VECCALC> calcBuf_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0};
    uint32_t Co_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};
    uint32_t tileCo_{0}, tileWo_{0};

    uint64_t offWeight_{0};
    uint64_t offInput_{0};
    uint64_t offOutput_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_asymmetric_input_asymmetric_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dDirect op;
    op.Init(
        x, weight, y,
        tiling_data.N, tiling_data.C, tiling_data.H, tiling_data.W,
        tiling_data.Co, tiling_data.Kh, tiling_data.Kw,
        tiling_data.Ho, tiling_data.Wo,
        tiling_data.tileCo, tiling_data.tileWo
    );
    op.Process();
}
