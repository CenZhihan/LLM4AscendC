#include "conv2d_min_tanh_tanh_custom_tiling.h"

namespace {

static inline uint32_t CeilDiv(uint64_t a, uint32_t b)
{
    return static_cast<uint32_t>((a + b - 1) / b);
}

} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv2dMinTanhTanhCustomTilingData tiling;

    const gert::Shape *x_shape = context->GetInputShape(0);  // [N, Cin, Hin, Win]
    const gert::Shape *w_shape = context->GetInputShape(1);  // [Cout, Cin, Kh, Kw]

    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t N  = x_shape->GetDim(0);
    const uint64_t Cin = x_shape->GetDim(1);
    const uint64_t Hin = x_shape->GetDim(2);
    const uint64_t Win = x_shape->GetDim(3);

    const uint64_t Cout = w_shape->GetDim(0);
    const uint64_t Cin_w = w_shape->GetDim(1);
    const uint64_t Kh = w_shape->GetDim(2);
    const uint64_t Kw = w_shape->GetDim(3);

    if (Cin != Cin_w) {
        return ge::GRAPH_FAILED;
    }

    if (Hin < Kh || Win < Kw) {
        return ge::GRAPH_FAILED;
    }

    const uint64_t Ho = Hin - Kh + 1;
    const uint64_t Wo = Win - Kw + 1;

    const uint64_t totalPixels64 = N * Ho * Wo;

    // Optional bias
    bool hasBias = false;
    if (context->GetInputNum() > 2) {
        const gert::Shape *b_shape = context->GetInputShape(2);
        if (b_shape != nullptr && b_shape->GetShapeSize() > 0) {
            hasBias = true;
        }
    }

    // Extremely conservative tile setting: process one output pixel per iteration.
    const uint32_t tilePixels = 1;
    uint32_t blockDim = CeilDiv(totalPixels64, tilePixels);
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 32u) blockDim = 32u;
    context->SetBlockDim(blockDim);

    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_Cin(static_cast<uint32_t>(Cin));
    tiling.set_Hin(static_cast<uint32_t>(Hin));
    tiling.set_Win(static_cast<uint32_t>(Win));
    tiling.set_Cout(static_cast<uint32_t>(Cout));
    tiling.set_Kh(static_cast<uint32_t>(Kh));
    tiling.set_Kw(static_cast<uint32_t>(Kw));
    tiling.set_Ho(static_cast<uint32_t>(Ho));
    tiling.set_Wo(static_cast<uint32_t>(Wo));
    tiling.set_K(static_cast<uint32_t>(Cin * Kh * Kw));
    tiling.set_totalPixels(static_cast<uint32_t>(totalPixels64));
    tiling.set_tilePixels(tilePixels);
    tiling.set_blockDim(blockDim);
    tiling.set_hasBias(hasBias ? 1u : 0u);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

    const int64_t Cin = x_shape->GetDim(1);

    const int64_t Cout = w_shape->GetDim(0);
    const int64_t CinW = w_shape->GetDim(1);
    const int64_t Kh = w_shape->GetDim(2);

kernel_src="""

#include "kernel_operator.h"

class KernelConv2dMinTanhTanhCustom {
public:
    __aicore__ inline KernelConv2dMinTanhTanhCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Hin, uint32_t Win,
                                uint32_t Cout, uint32_t Kh, uint32_t Kw,
                                uint32_t Ho, uint32_t Wo,
                                uint32_t totalPixels, uint32_t tilePixels, uint32_t hasBias)
    {
        N_ = N; Cin_ = Cin; Hin_ = Hin; Win_ = Win;
        Cout_ = Cout; Kh_ = Kh; Kw_ = Kw;
        Ho_ = Ho; Wo_ = Wo;
        totalPixels_ = totalPixels;
        tilePixels_ = tilePixels;
        hasBias_ = hasBias != 0;

        const uint32_t xSize = N_ * Cin_ * Hin_ * Win_;
        const uint32_t wSize = Cout_ * Cin_ * Kh_ * Kw_;
        const uint32_t ySize = N_ * Ho_ * Wo_; // channel=1

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(calcQ_, 6, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalPixels_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t stridePixels = blockNum * tilePixels_;

        uint32_t base = blockIdx * tilePixels_;
        while (base < totalPixels_) {
            const uint32_t len = ((base + tilePixels_) <= totalPixels_) ? tilePixels_ : (totalPixels_ - base);
            for (uint32_t i = 0; i < len; ++i) {
                ComputeOnePixel(base + i);
            }
            base += stridePixels;
        }
    }

private:
    __aicore__ inline void ComputeOnePixel(uint32_t pixIdx)
    {
        // Map flat pixel index to (n, ho, wo)
        const uint32_t hw = Ho_ * Wo_;
        const uint32_t n  = pixIdx / hw;
        const uint32_t rem = pixIdx % hw;
        const uint32_t ho = rem / Wo_;
        const uint32_t wo = rem % Wo_;

        // min accumulator (1-element local tensor)
        AscendC::LocalTensor<float> minLocal = calcQ_.AllocTensor<float>();
        bool minInit = false;

        for (uint32_t co = 0; co < Cout_; ++co) {
            // acc for this output channel
            AscendC::LocalTensor<float> acc = calcQ_.AllocTensor<float>();
            if (hasBias_) {
                AscendC::DataCopy(acc, bGm_[co], 1);
            } else {
                AscendC::Dup(acc, 0.0f, 1);
            }

            // Iterate K = Cin * Kh * Kw
            for (uint32_t c = 0; c < Cin_; ++c) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        const uint32_t hIn = ho + kh;
                        const uint32_t wIn = wo + kw;

                        // x offset: ((n*Cin + c)*Hin + hIn)*Win + wIn
                        const uint32_t xoff = ((n * Cin_ + c) * Hin_ + hIn) * Win_ + wIn;
                        // w offset: (((co*Cin + c)*Kh + kh)*Kw + kw)
                        const uint32_t woff = (((co * Cin_ + c) * Kh_ + kh) * Kw_ + kw);

                        AscendC::LocalTensor<float> xScal = calcQ_.AllocTensor<float>();
                        AscendC::LocalTensor<float> wScal = calcQ_.AllocTensor<float>();
                        AscendC::LocalTensor<float> tmp = calcQ_.AllocTensor<float>();

                        AscendC::DataCopy(xScal, xGm_[xoff], 1);
                        AscendC::DataCopy(wScal, wGm_[woff], 1);
                        AscendC::Mul(tmp, xScal, wScal, 1);
                        AscendC::Add(acc, acc, tmp, 1);

                        calcQ_.FreeTensor(tmp);
                        calcQ_.FreeTensor(wScal);
                        calcQ_.FreeTensor(xScal);
                    }
                }
            }

            if (!minInit) {
                AscendC::DataCopy(minLocal, acc, 1);
                minInit = true;
            } else {
                // minLocal = min(minLocal, acc)
                AscendC::Min(minLocal, minLocal, acc, 1);
            }

            calcQ_.FreeTensor(acc);
        }

        // Apply tanh twice
        AscendC::Tanh(minLocal, minLocal, 1);
        AscendC::Tanh(minLocal, minLocal, 1);

        // Store result to y: ((n*1 + 0)*Ho + ho)*Wo + wo = n*Ho*Wo + ho*Wo + wo
        const uint32_t yoff = n * (Ho_ * Wo_) + ho * Wo_ + wo;
        AscendC::DataCopy(yGm_[yoff], minLocal, 1);

        calcQ_.FreeTensor(minLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 6> calcQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Hin_{0}, Win_{0};
    uint32_t Cout_{0}, Kh_{0}, Kw_{0};
    uint32_t Ho_{0}, Wo_{0};

    uint32_t totalPixels_{0};
    uint32_t tilePixels_{1};
    bool hasBias_{false};
};

extern "C" __global__ __aicore__ void conv2d_min_tanh_tanh_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv2dMinTanhTanhCustom op;
    op.Init(
        x, weight, bias, y,
        tiling_data.N, tiling_data.Cin, tiling_data.Hin, tiling_data.Win,
        tiling_data.Cout, tiling_data.Kh, tiling_data.Kw,
        tiling_data.Ho, tiling_data.Wo,
        tiling_data.totalPixels, tiling_data.tilePixels, tiling_data.hasBias
    );
    op.Process();
}

#include <torch/library.h>

at::Tensor conv2d_min_tanh_tanh_impl_npu(const at::Tensor& x,
                                         const at::Tensor& weight,
                                         c10::optional<at::Tensor> bias_opt) {
    TORCH_CHECK(x.dim() == 4, "x must be 4D NCHW");
    TORCH_CHECK(weight.dim() == 4, "weight must be 4D OIHW");
    TORCH_CHECK(x.scalar_type() == at::kFloat, "Only float32 supported");
    TORCH_CHECK(weight.scalar_type() == at::kFloat, "Only float32 supported");

    const int64_t N = x.size(0);
    const int64_t Cin = x.size(1);
    const int64_t Hin = x.size(2);
    const int64_t Win = x.size(3);

    const int64_t Cout = weight.size(0);
    const int64_t CinW = weight.size(1);
    const int64_t Kh = weight.size(2);
    const int64_t Kw = weight.size(3);

    TORCH_CHECK(Cin == CinW, "Input channels mismatch with weight");

    TORCH_CHECK(Hin >= Kh && Win >= Kw, "Kernel bigger than input");
    const int64_t Ho = Hin - Kh + 1;
    const int64_t Wo = Win - Kw + 1;

    at::Tensor result = at::empty({N, 1, Ho, Wo}, x.options());

    if (bias_opt.has_value() && bias_opt.value().defined()) {
        auto bias = bias_opt.value();
        TORCH_CHECK(bias.dim() == 1 && bias.size(0) == Cout, "bias must be 1D with size Cout");
        EXEC_NPU_CMD(aclnnConv2dMinTanhTanhCustom, x, weight, bias, result);
    } else {
        EXEC_NPU_CMD(aclnnConv2dMinTanhTanhCustom, x, weight, result);
    }
    return result;
}

    m.impl("conv2d_min_tanh_tanh_custom", &conv2d_min_tanh_tanh_impl_npu);
