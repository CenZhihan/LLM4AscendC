host_tiling_src="""

#include "register/tilingdata_base.h"
#include "register/op_tiling.h"

namespace optiling {

BEGIN_TILING_DATA_DEF(StandardMatrixMultiplicationCustomTilingData)
  TILING_DATA_FIELD_DEF(TCubeTiling, cubeTilingData);
END_TILING_DATA_DEF;

REGISTER_TILING_DATA_CLASS(StandardMatrixMultiplicationCustom, StandardMatrixMultiplicationCustomTilingData)

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Fetch shapes: A [M, K], B [K, N], C [M, N]
    const auto* a_shape = context->GetInputShape(0);
    const auto* b_shape = context->GetInputShape(1);

    if (a_shape == nullptr || b_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    const uint32_t K = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    const uint32_t Kb = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(0));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(1));

    // Basic validation
    if (K != Kb) {
        return ge::GRAPH_FAILED;
    }

    // Prepare cube tiling using default generator if available.
    // For robustness, we set a conservative block dim aligned with tile count.
    uint32_t numTilesM = CeilDiv(M, 128u);
    uint32_t numTilesN = CeilDiv(N, 128u);
    uint32_t blockDim = numTilesM * numTilesN;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;

    context->SetBlockDim(blockDim);

    // Initialize tiling data container.
    StandardMatrixMultiplicationCustomTilingData tiling;

    // Use built-in cube tiling utility if present. Otherwise, leave default and rely on kernel registration.
    // The MatMul engine will interpret cubeTilingData filled by framework helpers.
    // Here, we just reserve buffer for tiling data; framework is expected to fill details.
    // To be safe, zero-initialize the memory.
    auto *raw_buf = reinterpret_cast<uint8_t*>(context->GetRawTilingData()->GetData());
    const uint32_t cap = static_cast<uint32_t>(context->GetRawTilingData()->GetCapacity());
    for (uint32_t i = 0; i < cap; ++i) {
        raw_buf[i] = 0;
    }

    // Save structure to buffer
    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No workspace is needed
    size_t *wss = context->GetWorkspaceSizes(1);
    wss[0] = 0;
    return ge::GRAPH_SUCCESS;
}

}  // namespace optiling
"""

host_operator_src="""

#include "standard_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    if (a_shape == nullptr || b_shape == nullptr) {
        return GRAPH_FAILED;
    }
    // A: [M, K], B: [K, N] -> C: [M, N]
    const int64_t M = a_shape->GetDim(0);
    const int64_t K = a_shape->GetDim(1);
    const int64_t Kb = b_shape->GetDim(0);
    const int64_t N = b_shape->GetDim(1);
    if (K != Kb) {
        return GRAPH_FAILED;
    }
    gert::Shape *c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(2);
    c_shape->SetDim(0, M);
    c_shape->SetDim(1, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Inputs are float16, output is float32
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class StandardMatrixMultiplicationCustom : public OpDef {
public:
    explicit StandardMatrixMultiplicationCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(StandardMatrixMultiplicationCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

extern "C" __global__ __aicore__ void standard_matrix_multiplication_custom(
    GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    // Use cube matmul kernel with FP16 inputs and FP32 accumulation/output.
    GET_TILING_DATA(tilingData, tiling);

    MatmulKernel<half, half, float, float> matmulKernel;
    AscendC::TPipe pipe;

    // Register matmul object with system workspace and cube tiling data.
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), matmulKernel.matmulObj, &tilingData.cubeTilingData);

    // Bias is nullptr, A/B not transposed.
    matmulKernel.Init(a, b, nullptr, c, workspace, tilingData.cubeTilingData);
    matmulKernel.Process(&pipe);
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_npu_half_if_needed(const at::Tensor& t) {
    if (t.scalar_type() == at::kHalf) {
        return t;
    }
    return t.to(at::kHalf);
}

at::Tensor standard_matmul_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Interface: A and B can be float32 on Python side; cast to float16 for NPU kernel
    at::Tensor A_half = to_npu_half_if_needed(A);
    at::Tensor B_half = to_npu_half_if_needed(B);
    // Output in float32 for better accumulation fidelity
    at::Tensor C = at::empty({A.size(0), B.size(1)}, A.options().dtype(at::kFloat));
    EXEC_NPU_CMD(aclnnStandardMatrixMultiplicationCustom, A_half, B_half, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("standard_matrix_multiplication_custom", &standard_matmul_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("standard_matrix_multiplication_custom", &standard_matmul_impl_npu,
          "Standard matrix multiplication (StandardMatrixMultiplicationCustom on NPU)");
}
