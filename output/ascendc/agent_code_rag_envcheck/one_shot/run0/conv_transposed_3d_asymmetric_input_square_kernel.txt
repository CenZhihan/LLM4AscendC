kernel_src="""
#include "kernel_operator.h"

struct ConvT3DParams {
    uint32_t n, ci, co;
    uint32_t di, hi, wi;
    uint32_t kd, kh, kw;
    uint32_t dOut, hOut, wOut;
    uint64_t totalLen;
    uint32_t tileLen;
};

class KernelConvTransposed3D {
public:
    __aicore__ inline KernelConvTransposed3D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                const ConvT3DParams& p)
    {
        p_ = p;

        xGm_.SetGlobalBuffer((__gm__ float*)x,
                             static_cast<uint64_t>(p_.n) * p_.ci * p_.di * p_.hi * p_.wi);
        wGm_.SetGlobalBuffer((__gm__ float*)w,
                             static_cast<uint64_t>(p_.ci) * p_.co * p_.kd * p_.kh * p_.kw);
        yGm_.SetGlobalBuffer((__gm__ float*)y,
                             static_cast<uint64_t>(p_.n) * p_.co * p_.dOut * p_.hOut * p_.wOut);

        // Minimal queues for scalar pipeline
        pipe_.InitBuffer(inQ_, 1, sizeof(float));
        pipe_.InitBuffer(wQ_, 1, sizeof(float));
        pipe_.InitBuffer(calcQ_, 1, sizeof(float));
        pipe_.InitBuffer(outQ_, 1, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (p_.totalLen == 0 || p_.tileLen == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint64_t strideElems = static_cast<uint64_t>(blockNum) * p_.tileLen;
        uint64_t base = static_cast<uint64_t>(blockIdx) * p_.tileLen;

        while (base < p_.totalLen) {
            uint32_t cur = static_cast<uint32_t>(p_.totalLen - base);
            if (cur > p_.tileLen) cur = p_.tileLen;
            ComputeTile(base, cur);
            base += strideElems;
        }
    }

private:
    __aicore__ inline uint64_t OffsetX(uint32_t n, uint32_t c, uint32_t z, uint32_t y, uint32_t x) const
    {
        return (((static_cast<uint64_t>(n) * p_.ci + c) * p_.di + z) * p_.hi + y) * p_.wi + x;
    }
    __aicore__ inline uint64_t OffsetW(uint32_t ci, uint32_t co, uint32_t kd, uint32_t kh, uint32_t kw) const
    {
        return ((((static_cast<uint64_t>(ci) * p_.co) + co) * p_.kd + kd) * p_.kh + kh) * p_.kw + kw;
    }
    __aicore__ inline uint64_t OffsetY(uint32_t n, uint32_t c, uint32_t z, uint32_t y, uint32_t x) const
    {
        return (((static_cast<uint64_t>(n) * p_.co + c) * p_.dOut + z) * p_.hOut + y) * p_.wOut + x;
    }

    __aicore__ inline void DecodeOutIndex(uint64_t idx, uint32_t &n, uint32_t &co, uint32_t &oz,
                                          uint32_t &oy, uint32_t &ox) const
    {
        uint64_t t = idx;
        ox = static_cast<uint32_t>(t % p_.wOut); t /= p_.wOut;
        oy = static_cast<uint32_t>(t % p_.hOut); t /= p_.hOut;
        oz = static_cast<uint32_t>(t % p_.dOut); t /= p_.dOut;
        co = static_cast<uint32_t>(t % p_.co);   t /= p_.co;
        n  = static_cast<uint32_t>(t);
    }

    __aicore__ inline void ComputeTile(uint64_t base, uint32_t len)
    {
        for (uint32_t i = 0; i < len; ++i) {
            const uint64_t outIndex = base + i;
            uint32_t n, co, oz, oy, ox;
            DecodeOutIndex(outIndex, n, co, oz, oy, ox);

            // accumulator in UB (length 1)
            AscendC::LocalTensor<float> acc = calcQ_.AllocTensor<float>();
            AscendC::Dup(acc, 0.0f, 1);

            // Iterate over input channels and kernel
            for (uint32_t ci = 0; ci < p_.ci; ++ci) {
                for (uint32_t kd = 0; kd < p_.kd; ++kd) {
                    // iz = oz - (kd_out) where kd_out = (K-1 - kd)
                    const int32_t iz = static_cast<int32_t>(oz) - static_cast<int32_t>(p_.kd - 1 - kd);
                    if (iz < 0 || iz >= static_cast<int32_t>(p_.di)) continue;
                    for (uint32_t kh = 0; kh < p_.kh; ++kh) {
                        const int32_t iy = static_cast<int32_t>(oy) - static_cast<int32_t>(p_.kh - 1 - kh);
                        if (iy < 0 || iy >= static_cast<int32_t>(p_.hi)) continue;
                        for (uint32_t kw = 0; kw < p_.kw; ++kw) {
                            const int32_t ix = static_cast<int32_t>(ox) - static_cast<int32_t>(p_.kw - 1 - kw);
                            if (ix < 0 || ix >= static_cast<int32_t>(p_.wi)) continue;

                            const uint64_t xOff = OffsetX(n, ci, static_cast<uint32_t>(iz),
                                                          static_cast<uint32_t>(iy), static_cast<uint32_t>(ix));
                            const uint64_t wOff = OffsetW(ci, co, kd, kh, kw);

                            AscendC::LocalTensor<float> xLocal = inQ_.AllocTensor<float>();
                            AscendC::DataCopy(xLocal, xGm_[xOff], 1);
                            inQ_.EnQue(xLocal);
                            AscendC::LocalTensor<float> xVal = inQ_.DeQue<float>();

                            AscendC::LocalTensor<float> wLocal = wQ_.AllocTensor<float>();
                            AscendC::DataCopy(wLocal, wGm_[wOff], 1);
                            wQ_.EnQue(wLocal);
                            AscendC::LocalTensor<float> wVal = wQ_.DeQue<float>();

                            AscendC::LocalTensor<float> mulLocal = calcQ_.AllocTensor<float>();
                            AscendC::Mul(mulLocal, xVal, wVal, 1);
                            AscendC::Add(acc, acc, mulLocal, 1);

                            calcQ_.FreeTensor(mulLocal);
                            wQ_.FreeTensor(wVal);
                            inQ_.FreeTensor(xVal);
                        }
                    }
                }
            }

            // Write result
            const uint64_t yOff = OffsetY(n, co, oz, oy, ox);
            AscendC::DataCopy(yGm_[yOff], acc, 1);
            calcQ_.FreeTensor(acc);
        }
    }

private:
    ConvT3DParams p_{};

    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> wQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQ_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tilingGm)
{
    if (g_coreType == AscendC::AIV) {
        return;
    }
    GET_TILING_DATA(tilingData, tilingGm);

    ConvT3DParams p;
    p.n   = tilingData.n;
    p.ci  = tilingData.ci;
    p.co  = tilingData.co;
    p.di  = tilingData.di;
    p.hi  = tilingData.hi;
    p.wi  = tilingData.wi;
    p.kd  = tilingData.kd;
    p.kh  = tilingData.kh;
    p.kw  = tilingData.kw;
    p.dOut = tilingData.dOut;
    p.hOut = tilingData.hOut;
    p.wOut = tilingData.wOut;
    p.totalLen = tilingData.totalLength;
    p.tileLen  = tilingData.tileLength;

    KernelConvTransposed3D op;
    op.Init(x, weight, y, p);
    op.Process();
}
