#include "matmul_with_irregular_shapes_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a) { return (x / a) * a; }
static inline uint32_t AlignUp(uint32_t x, uint32_t a) { return ((x + a - 1u) / a) * a; }
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) { return (a + b - 1u) / b; }
}

// Conservative UB budget for a single core
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;
// Reserve some bytes to account for queues and temps
static constexpr uint32_t UB_SAFE_BYTES = 180 * 1024;

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    const gert::Shape* a_shape = context->GetInputShape(0);
    const gert::Shape* b_shape = context->GetInputShape(1);

    if (a_shape->GetDimNum() != 2 || b_shape->GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t M = static_cast<uint32_t>(a_shape->GetDim(0));
    const uint32_t K = static_cast<uint32_t>(a_shape->GetDim(1));
    const uint32_t K2 = static_cast<uint32_t>(b_shape->GetDim(0));
    const uint32_t N = static_cast<uint32_t>(b_shape->GetDim(1));

    if (K != K2) {
        return ge::GRAPH_FAILED;
    }

    // Initial candidate tiles
    uint32_t tileK = 128u;
    uint32_t tileN = (N >= 128u) ? 128u : AlignDown(N, 16u);
    if (tileN == 0u) tileN = N;  // small N

    // Determine maximum tileM under UB constraints: bytes = 4*(tileM*tileK + tileK*tileN + tileM*tileN)
    // We try decreasing tileN if needed to make tileM >= 1.
    uint32_t tileM = 64u;
    const uint32_t maxBytes = UB_SAFE_BYTES;

    auto fits = [&](uint32_t m, uint32_t n, uint32_t k) -> bool {
        uint64_t need = 4ull * (static_cast<uint64_t>(m) * k + static_cast<uint64_t>(k) * n + static_cast<uint64_t>(m) * n);
        // add temp buffer of n floats
        need += 4ull * n;
        return need <= maxBytes;
    };

    if (!fits(tileM, tileN, tileK)) {
        // Compute a feasible tileM given tileN and tileK
        while (tileN >= 16u) {
            // Solve m <= (maxBytes/4 - k*n) / (k + n)
            int64_t num = static_cast<int64_t>(maxBytes / 4u) - static_cast<int64_t>(tileK) * static_cast<int64_t>(tileN);
            int64_t den = static_cast<int64_t>(tileK) + static_cast<int64_t>(tileN);
            int64_t m_est = (den > 0) ? (num / den) : 0;
            if (m_est < 1) {
                tileN = AlignDown(tileN / 2u, 16u);
                if (tileN == 0u) tileN = 1u;
                continue;
            }
            tileM = static_cast<uint32_t>(m_est);
            tileM = AlignDown(tileM, 16u);
            if (tileM == 0u) tileM = 1u;
            if (fits(tileM, tileN, tileK)) break;
            tileN = AlignDown(tileN / 2u, 16u);
            if (tileN == 0u) tileN = 1u;
        }
        if (!fits(tileM, tileN, tileK)) {
            // Fall back to minimal tiles
            tileM = 8u;
            tileN = (tileN > 0u) ? tileN : 8u;
            while (!fits(tileM, tileN, tileK) && tileN > 1u) {
                tileN = AlignDown(tileN / 2u, 1u);
            }
        }
    }

    if (tileM > M) tileM = M;
    if (tileN > N) tileN = N;
    if (tileK > K) tileK = K;
    if (tileM == 0u) tileM = 1u;
    if (tileN == 0u) tileN = 1u;
    if (tileK == 0u) tileK = 1u;

    // Blocks: distribute M tiles across cores
    const uint32_t numMTiles = CeilDiv(M, tileM);
    uint32_t blockDim = numMTiles;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    MatmulWithIrregularShapesCustomTilingData tiling;
    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""

#include "kernel_operator.h"

// We implement a tiled GEMM: C[M,N] = A[M,K] * B[K,N] for float32.
// Tiling strategy: Block along M across cores. Within each core,
// iterate over N tiles, and accumulate across K tiles.

class KernelMatmulWithIrregularShapesCustom {
public:
    __aicore__ inline KernelMatmulWithIrregularShapesCustom() {}

    __aicore__ inline void Init(GM_ADDR a, GM_ADDR b, GM_ADDR c,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        M_ = M; N_ = N; K_ = K;
        tM_ = tileM; tN_ = tileN; tK_ = tileK;

        aGm_.SetGlobalBuffer((__gm__ float*)a, (uint64_t)M_ * K_);
        bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)K_ * N_);
        cGm_.SetGlobalBuffer((__gm__ float*)c, (uint64_t)M_ * N_);

        // Init queues with maximum tile sizes
        pipe_.InitBuffer(aInQ_, 1, tM_ * tK_ * sizeof(float));
        pipe_.InitBuffer(bInQ_, 1, tK_ * tN_ * sizeof(float));
        pipe_.InitBuffer(cOutQ_, 1, tM_ * tN_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, 1, tN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || N_ == 0 || K_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride across tiles of M
        uint32_t mTileIdx = blockIdx;
        while (true) {
            uint32_t mBase = mTileIdx * tM_;
            if (mBase >= M_) break;
            uint32_t mChunk = M_ - mBase;
            if (mChunk > tM_) mChunk = tM_;

            // Iterate over N tiles
            for (uint32_t nBase = 0; nBase < N_; nBase += tN_) {
                uint32_t nChunk = N_ - nBase;
                if (nChunk > tN_) nChunk = tN_;

                // Allocate C panel in UB
                AscendC::LocalTensor<float> cLocal = cOutQ_.AllocTensor<float>();
                // We will construct cLocal row by row during accumulation.

                bool firstKTile = true;

                // Accumulate across K tiles
                for (uint32_t kBase = 0; kBase < K_; kBase += tK_) {
                    uint32_t kChunk = K_ - kBase;
                    if (kChunk > tK_) kChunk = tK_;

                    // Copy A panel [mChunk x kChunk] into UB
                    AscendC::LocalTensor<float> aLocal = aInQ_.AllocTensor<float>();
                    {
                        // Copy each row segment
                        for (uint32_t mi = 0; mi < mChunk; ++mi) {
                            uint64_t a_gm_off = static_cast<uint64_t>(mBase + mi) * K_ + kBase;
                            uint32_t a_loc_off = mi * kChunk;
                            AscendC::DataCopy(aLocal[a_loc_off], aGm_[a_gm_off], kChunk);
                        }
                    }
                    aInQ_.EnQue(aLocal);

                    // Copy B panel [kChunk x nChunk] into UB
                    AscendC::LocalTensor<float> bLocal = bInQ_.AllocTensor<float>();
                    {
                        for (uint32_t kk = 0; kk < kChunk; ++kk) {
                            uint64_t b_gm_off = static_cast<uint64_t>(kBase + kk) * N_ + nBase;
                            uint32_t b_loc_off = kk * nChunk;
                            AscendC::DataCopy(bLocal[b_loc_off], bGm_[b_gm_off], nChunk);
                        }
                    }
                    bInQ_.EnQue(bLocal);

                    // Dequeue to compute
                    AscendC::LocalTensor<float> aPanel = aInQ_.DeQue<float>();
                    AscendC::LocalTensor<float> bPanel = bInQ_.DeQue<float>();

                    // Compute C += A_panel * B_panel
                    for (uint32_t mi = 0; mi < mChunk; ++mi) {
                        // c_row is at offset mi * nChunk
                        uint32_t c_off = mi * nChunk;
                        // Iterate k contribution
                        for (uint32_t kk = 0; kk < kChunk; ++kk) {
                            float a_scalar = aPanel[mi * kChunk + kk];
                            AscendC::LocalTensor<float> b_row = bPanel[kk * nChunk];

                            if (firstKTile && kk == 0) {
                                // c_row = a_scalar * b_row
                                AscendC::Muls(cLocal[c_off], b_row, a_scalar, nChunk);
                            } else {
                                AscendC::LocalTensor<float> tmp = tmpQ_.AllocTensor<float>();
                                AscendC::Muls(tmp, b_row, a_scalar, nChunk);
                                AscendC::Add(cLocal[c_off], cLocal[c_off], tmp, nChunk);
                                tmpQ_.FreeTensor(tmp);
                            }
                        }
                    }

                    firstKTile = false;

                    // Free panels
                    aInQ_.FreeTensor(aPanel);
                    bInQ_.FreeTensor(bPanel);
                }

                // Enqueue and then copy out the C panel
                cOutQ_.EnQue(cLocal);
                CopyOutC(mBase, nBase, mChunk, nChunk);
                // cLocal consumed and freed inside CopyOutC
            }

            mTileIdx += blockNum;
        }
    }

private:
    __aicore__ inline void CopyOutC(uint32_t mBase, uint32_t nBase,
                                    uint32_t mChunk, uint32_t nChunk)
    {
        AscendC::LocalTensor<float> cLocal = cOutQ_.DeQue<float>();
        // Write each row back to GM
        for (uint32_t mi = 0; mi < mChunk; ++mi) {
            uint64_t c_gm_off = static_cast<uint64_t>(mBase + mi) * N_ + nBase;
            uint32_t c_loc_off = mi * nChunk;
            AscendC::DataCopy(cGm_[c_gm_off], cLocal[c_loc_off], nChunk);
        }
        cOutQ_.FreeTensor(cLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> aInQ_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> bInQ_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> cOutQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQ_;

    AscendC::GlobalTensor<float> aGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> cGm_;

    uint32_t M_{0}, N_{0}, K_{0};
    uint32_t tM_{0}, tN_{0}, tK_{0};
};

extern "C" __global__ __aicore__ void matmul_with_irregular_shapes_custom(GM_ADDR a, GM_ADDR b, GM_ADDR c, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithIrregularShapesCustom op;
    op.Init(a, b, c,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.tileM, tiling_data.tileN, tiling_data.tileK);
    op.Process();
}
