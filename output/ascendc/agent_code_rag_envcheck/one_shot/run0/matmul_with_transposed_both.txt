#include "matmul_with_transposed_both_custom_tiling.h"

namespace optiling {

// UB budget and alignment choices
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // 192KB per-core conservative
static constexpr uint32_t ALIGN_ELEMS = 16;             // prefer multiples of 16
static constexpr uint32_t MAX_BLOCKS = 24;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulWithTransposedBothCustomTilingData tiling;

    // A shape: (K, M), B shape: (N, K); C shape: (M, N)
    const auto *A_shape = context->GetInputShape(0);
    const auto *B_shape = context->GetInputShape(1);

    const uint32_t K = static_cast<uint32_t>(A_shape->GetOriginShape().GetDim(0));
    const uint32_t M = static_cast<uint32_t>(A_shape->GetOriginShape().GetDim(1));
    const uint32_t N = static_cast<uint32_t>(B_shape->GetOriginShape().GetDim(0));

    // Choose base tile sizes
    uint32_t Tm = 64;
    uint32_t Tn = 64;

    // Compute Tk from UB budget: floats needed ~= Tk*(Tm+Tn) + Tm*Tn
    const uint32_t budgetFloats = UB_BUDGET_BYTES / 4;
    uint32_t Tk = 256;
    {
        uint32_t overhead = Tm * Tn + 1024; // some margin
        if (budgetFloats > overhead + (Tm + Tn)) {
            uint32_t maxTk = (budgetFloats - overhead) / (Tm + Tn);
            if (maxTk == 0) maxTk = ALIGN_ELEMS;
            Tk = (maxTk > 1024) ? 1024 : maxTk; // clamp
            Tk = AlignDown(Tk, ALIGN_ELEMS);
            if (Tk == 0) Tk = ALIGN_ELEMS;
        }
        if (Tk > K) {
            Tk = AlignDown(K, ALIGN_ELEMS);
            if (Tk == 0) Tk = K;
        }
    }

    // Adjust Tm/Tn if extremely small tensors
    if (M < Tm) {
        Tm = (M >= ALIGN_ELEMS) ? AlignDown(M, ALIGN_ELEMS) : M;
    }
    if (N < Tn) {
        Tn = (N >= ALIGN_ELEMS) ? AlignDown(N, ALIGN_ELEMS) : N;
    }

    const uint32_t tilesM = (Tm == 0) ? 0u : CeilDiv(M, Tm);
    const uint32_t tilesN = (Tn == 0) ? 0u : CeilDiv(N, Tn);
    uint32_t totalTiles = tilesM * tilesN;
    if (totalTiles == 0) totalTiles = 1;

    uint32_t blockDim = totalTiles;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_M(M);
    tiling.set_N(N);
    tiling.set_K(K);
    tiling.set_Tm(Tm);
    tiling.set_Tn(Tn);
    tiling.set_Tk(Tk);
    tiling.set_tilesM(tilesM);
    tiling.set_tilesN(tilesN);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulWithTransposedBothCustom {
public:
    __aicore__ inline KernelMatmulWithTransposedBothCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t M, uint32_t N, uint32_t K,
                                uint32_t Tm, uint32_t Tn, uint32_t Tk,
                                uint32_t tilesM, uint32_t tilesN)
    {
        M_ = M;
        N_ = N;
        K_ = K;
        Tm_ = (Tm == 0) ? 1u : Tm;
        Tn_ = (Tn == 0) ? 1u : Tn;
        Tk_ = (Tk == 0) ? 1u : Tk;
        tilesM_ = (tilesM == 0) ? ((M_ + Tm_ - 1u) / Tm_) : tilesM;
        tilesN_ = (tilesN == 0) ? ((N_ + Tn_ - 1u) / Tn_) : tilesN;

        A_gm_.SetGlobalBuffer((__gm__ float*)A, K_ * M_);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, N_ * K_);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, M_ * N_);

        // Buffers:
        // - A_sub: Tk * Tm
        // - B_sub: Tn * Tk
        // - C_tile: Tm * Tn
        // - tmpVec: max(Tm, Tn)
        pipe_.InitBuffer(aQueue_, BUFFER_NUM, (size_t)Tm_ * Tk_ * sizeof(float));
        pipe_.InitBuffer(bQueue_, BUFFER_NUM, (size_t)Tn_ * Tk_ * sizeof(float));
        pipe_.InitBuffer(cQueue_, BUFFER_NUM, (size_t)Tm_ * Tn_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, (size_t)((Tm_ > Tn_) ? Tm_ : Tn_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t totalTiles = tilesM_ * tilesN_;
        if (totalTiles == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t tileId = blockIdx; tileId < totalTiles; tileId += blockNum) {
            const uint32_t bm = tileId / tilesN_;
            const uint32_t bn = tileId % tilesN_;

            const uint32_t m0 = bm * Tm_;
            const uint32_t n0 = bn * Tn_;
            const uint32_t curTm = (m0 + Tm_ <= M_) ? Tm_ : (M_ - m0);
            const uint32_t curTn = (n0 + Tn_ <= N_) ? Tn_ : (N_ - n0);
            if (curTm == 0 || curTn == 0) continue;

            // Allocate C tile and zero-init
            AscendC::LocalTensor<float> cTile = cQueue_.AllocTensor<float>();
            AscendC::Duplicate(cTile, 0.0f, (uint32_t)(Tm_ * Tn_)); // zero entire tile

            // Iterate over K in chunks of Tk_
            for (uint32_t k0 = 0; k0 < K_; k0 += Tk_) {
                const uint32_t curTk = ((k0 + Tk_) <= K_) ? Tk_ : (K_ - k0);
                if (curTk == 0) break;

                // Load A_sub: shape (curTk x curTm) from GM A (K x M)
                AscendC::LocalTensor<float> aSub = aQueue_.AllocTensor<float>();
                for (uint32_t rk = 0; rk < curTk; ++rk) {
                    const uint32_t gmOffset = (k0 + rk) * M_ + m0;
                    AscendC::DataCopy(aSub[rk * Tm_], A_gm_[gmOffset], curTm);
                }
                // Load B_sub: shape (curTn x curTk) from GM B (N x K)
                AscendC::LocalTensor<float> bSub = bQueue_.AllocTensor<float>();
                for (uint32_t rn = 0; rn < curTn; ++rn) {
                    const uint32_t gmOffset = (n0 + rn) * K_ + k0;
                    AscendC::DataCopy(bSub[rn * Tk_], B_gm_[gmOffset], curTk);
                }

                // Compute accumulate: C_tile (Tm x Tn) += A_sub^T (Tm x curTk) * B_sub^T (curTk x Tn)
                // We implement:
                // for kk in [0,curTk):
                //   aRow = A_sub[kk, 0:curTm]
                //   for j in [0,curTn):
                //     b_scalar = B_sub[j, kk]
                //     C_row_mj[0:curTm] += aRow * b_scalar
                for (uint32_t kk = 0; kk < curTk; ++kk) {
                    AscendC::LocalTensor<float> aRow = aSub[kk * Tm_]; // length >= curTm

                    for (uint32_t j = 0; j < curTn; ++j) {
                        // Read scalar bSub[j, kk]
                        // Assuming LocalTensor supports .GetValue(index) to fetch scalar
                        float b_scalar = AscendC::GetValue(bSub, j * Tk_ + kk);
                        if (b_scalar == 0.0f) {
                            continue;
                        }
                        // temp = aRow[0:curTm] * b_scalar
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::Muls(tmp, aRow, b_scalar, curTm);

                        // cTile row m: offset m * Tn_ + j; But we want to add tmp (len curTm) into cTile rows' j-th column
                        // For better vectorization, store cTile as row-major (Tm x Tn), and update row by row
                        // tmp contains contribution for column j across all rows m
                        for (uint32_t m = 0; m < curTm; ++m) {
                            // cTile[m, j] += tmp[m]
                            float prev = AscendC::GetValue(cTile, m * Tn_ + j);
                            float addv = AscendC::GetValue(tmp, m);
                            AscendC::SetValue(cTile, m * Tn_ + j, prev + addv);
                        }

                        tmpQueue_.FreeTensor(tmp);
                    }
                }

                bQueue_.FreeTensor(bSub);
                aQueue_.FreeTensor(aSub);
            }

            // Store C_tile (row-major Tm x Tn) to GM C (M x N), only valid curTm x curTn
            for (uint32_t m = 0; m < curTm; ++m) {
                const uint32_t gmOffset = (m0 + m) * N_ + n0;
                AscendC::DataCopy(C_gm_[gmOffset], cTile[m * Tn_], curTn);
            }

            cQueue_.FreeTensor(cTile);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> aQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> bQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> cQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t M_ {0}, N_ {0}, K_ {0};
    uint32_t Tm_ {0}, Tn_ {0}, Tk_ {0};
    uint32_t tilesM_ {0}, tilesN_ {0};
};

extern "C" __global__ __aicore__ void matmul_with_transposed_both_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulWithTransposedBothCustom op;
    op.Init(A, B, C,
            tiling_data.M, tiling_data.N, tiling_data.K,
            tiling_data.Tm, tiling_data.Tn, tiling_data.Tk,
            tiling_data.tilesM, tiling_data.tilesN);
    op.Process();
}
