#include "matmul_for_symmetric_matrices_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

// UB model for this kernel:
//   aRow: 1 * tileK
//   bTile: tileN * tileK
//   yRow: 1 * tileN
//   bColVec: 1 * tileN
// Total floats = tileK + tileN*tileK + 2*tileN = tileK*(tileN + 1) + 2*tileN
static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_M = 32;
static constexpr uint32_t ALIGN_N = 32;
static constexpr uint32_t ALIGN_K = 64;
static constexpr uint32_t MIN_M = 32;
static constexpr uint32_t MIN_N = 32;
static constexpr uint32_t MIN_K = 64;
static constexpr uint32_t MAX_BLOCKS = 24;
static constexpr uint32_t MIN_BLOCKS = 8;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    MatmulForSymmetricMatricesCustomTilingData tiling;

    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    // Expect square matrices: (N, N)
    const uint32_t nA0 = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(0));
    const uint32_t nA1 = static_cast<uint32_t>(a_shape->GetOriginShape().GetDim(1));
    const uint32_t nB0 = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(0));
    const uint32_t nB1 = static_cast<uint32_t>(b_shape->GetOriginShape().GetDim(1));

    if (nA0 != nA1 || nB0 != nB1 || nA0 != nB0) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N = nA0;

    // Choose tileN first: favor 64 if possible
    uint32_t tileN = (N >= 64u) ? 64u : AlignDown(N, ALIGN_N);
    if (tileN < MIN_N) tileN = MIN_N;
    tileN = AlignDown(tileN, ALIGN_N);
    if (tileN == 0) tileN = MIN_N;

    // Choose tileM similarly
    uint32_t tileM = (N >= 64u) ? 64u : AlignDown(N, ALIGN_M);
    if (tileM < MIN_M) tileM = MIN_M;
    tileM = AlignDown(tileM, ALIGN_M);
    if (tileM == 0) tileM = MIN_M;

    // UB budget to derive tileK
    const uint32_t ubFloats = UB_BUDGET_BYTES / 4u;
    // tileK*(tileN + 1) + 2*tileN <= ubFloats
    uint32_t tileK = (ubFloats > 2u * tileN) ? (ubFloats - 2u * tileN) / (tileN + 1u) : MIN_K;
    tileK = AlignDown(tileK, ALIGN_K);
    if (tileK < MIN_K) tileK = MIN_K;

    // Cap tileK to N
    if (tileK > N) tileK = AlignDown(N, ALIGN_K);
    if (tileK == 0) tileK = N;

    // Adaptive blockDim based on number of tiles in the (M,N) grid
    const uint32_t tilesM = CeilDiv(N, tileM);
    const uint32_t tilesN = CeilDiv(N, tileN);
    const uint32_t totalTiles = tilesM * tilesN;

    uint32_t blockDim = totalTiles;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;

    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_tileM(tileM);
    tiling.set_tileN(tileN);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *A_shape = context->GetInputShape(0);
    const gert::Shape *B_shape = context->GetInputShape(1);
    // For matmul of NxN by NxN => NxN
    if (A_shape->GetOriginShape().GetDimNum() != 2 ||
        B_shape->GetOriginShape().GetDimNum() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t nA0 = A_shape->GetOriginShape().GetDim(0);
    const int64_t nA1 = A_shape->GetOriginShape().GetDim(1);
    const int64_t nB0 = B_shape->GetOriginShape().GetDim(0);
    const int64_t nB1 = B_shape->GetOriginShape().GetDim(1);
    if (nA0 != nA1 || nB0 != nB1 || nA0 != nB0) {
        return GRAPH_FAILED;
    }
    gert::Shape *C_shape = context->GetOutputShape(0);
    C_shape->SetDimNum(2);
    C_shape->SetDim(0, nA0);
    C_shape->SetDim(1, nB1);
    return GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelMatmulForSymmetricMatricesCustom {
public:
    __aicore__ inline KernelMatmulForSymmetricMatricesCustom() {}

    __aicore__ inline void Init(GM_ADDR A, GM_ADDR B, GM_ADDR C,
                                uint32_t n, uint32_t tileM, uint32_t tileN, uint32_t tileK)
    {
        N_ = n;
        tileM_ = tileM;
        tileN_ = tileN;
        tileK_ = tileK;

        const uint32_t totalElems = N_ * N_;
        A_gm_.SetGlobalBuffer((__gm__ float*)A, totalElems);
        B_gm_.SetGlobalBuffer((__gm__ float*)B, totalElems);
        C_gm_.SetGlobalBuffer((__gm__ float*)C, totalElems);

        // Local buffers
        pipe_.InitBuffer(aRowQ_, BUFFER_NUM, tileK_ * sizeof(float));
        pipe_.InitBuffer(bTileQ_, BUFFER_NUM, tileN_ * tileK_ * sizeof(float));
        pipe_.InitBuffer(yRowQ_, BUFFER_NUM, tileN_ * sizeof(float));
        pipe_.InitBuffer(colQ_, BUFFER_NUM, tileN_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (N_ == 0) return;

        const uint32_t tilesM = (tileM_ == 0) ? 1u : (N_ + tileM_ - 1u) / tileM_;
        const uint32_t tilesN = (tileN_ == 0) ? 1u : (N_ + tileN_ - 1u) / tileN_;
        const uint32_t totalTiles = tilesM * tilesN;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t t = blockIdx; t < totalTiles; t += blockNum) {
            const uint32_t iTile = t / tilesN;
            const uint32_t jTile = t % tilesN;

            const uint32_t iBase = iTile * tileM_;
            const uint32_t jBase = jTile * tileN_;
            const uint32_t curM = AscendC::min(tileM_, N_ - iBase);
            const uint32_t curN = AscendC::min(tileN_, N_ - jBase);

            ComputeTile(iBase, jBase, curM, curN);
        }
    }

private:
    __aicore__ inline void LoadBTile(uint32_t jBase, uint32_t kStart, uint32_t curN, uint32_t curK,
                                     AscendC::LocalTensor<float>& bTile)
    {
        // Load B rows [jBase, jBase+curN) over k range [kStart, kStart+curK)
        for (uint32_t j = 0; j < curN; ++j) {
            const uint32_t bRowGmOffset = (jBase + j) * N_ + kStart;
            const uint32_t bRowLocalOffset = j * tileK_;
            AscendC::DataCopy(bTile[bRowLocalOffset], B_gm_[bRowGmOffset], curK);
        }
    }

    __aicore__ inline void LoadARow(uint32_t iRow, uint32_t kStart, uint32_t curK,
                                    AscendC::LocalTensor<float>& aRow)
    {
        const uint32_t aRowGmOffset = iRow * N_ + kStart;
        AscendC::DataCopy(aRow, A_gm_[aRowGmOffset], curK);
    }

    __aicore__ inline void ComputeTile(uint32_t iBase, uint32_t jBase, uint32_t curM, uint32_t curN)
    {
        // Local tensors
        AscendC::LocalTensor<float> aRow = aRowQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> bTile = bTileQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> yRow = yRowQ_.AllocTensor<float>();
        AscendC::LocalTensor<float> bCol = colQ_.AllocTensor<float>();

        for (uint32_t i = 0; i < curM; ++i) {
            // Initialize output accumulator row to zeros
            AscendC::Dup(yRow, 0.0f, curN);

            // Iterate K dimension in blocks
            for (uint32_t kStart = 0; kStart < N_; kStart += tileK_) {
                const uint32_t curK = AscendC::min(tileK_, N_ - kStart);

                // Load A row segment and B tile rows segment
                LoadARow(iBase + i, kStart, curK, aRow);
                LoadBTile(jBase, kStart, curN, curK, bTile);

                // Accumulate: yRow += aRow[k] * B[j,k] for all j in tileN
                for (uint32_t k = 0; k < curK; ++k) {
                    // Gather k-th elements across the selected rows of B into contiguous vector
                    // bCol[j] = B[jBase + j, kStart + k] == B[jBase + j, kStart + k]
                    for (uint32_t j = 0; j < curN; ++j) {
                        bCol[j] = bTile[j * tileK_ + k];
                    }
                    // Scale by A[i, k] and add
                    const float aVal = aRow[k];
                    AscendC::Muls(bCol, bCol, aVal, curN);
                    AscendC::Add(yRow, yRow, bCol, curN);
                }
            }

            // Write back the accumulated row to C
            const uint32_t cRowGmOffset = (iBase + i) * N_ + jBase;
            AscendC::DataCopy(C_gm_[cRowGmOffset], yRow, curN);
        }

        // Free locals
        colQ_.FreeTensor(bCol);
        yRowQ_.FreeTensor(yRow);
        bTileQ_.FreeTensor(bTile);
        aRowQ_.FreeTensor(aRow);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> aRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> bTileQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> yRowQ_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> colQ_;

    AscendC::GlobalTensor<float> A_gm_;
    AscendC::GlobalTensor<float> B_gm_;
    AscendC::GlobalTensor<float> C_gm_;

    uint32_t N_ {0};
    uint32_t tileM_ {0};
    uint32_t tileN_ {0};
    uint32_t tileK_ {0};
};

extern "C" __global__ __aicore__ void matmul_for_symmetric_matrices_custom(GM_ADDR A, GM_ADDR B, GM_ADDR C, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelMatmulForSymmetricMatricesCustom op;
    op.Init(A, B, C,
            tiling_data.n,
            tiling_data.tileM,
            tiling_data.tileN,
            tiling_data.tileK);
    op.Process();
}
