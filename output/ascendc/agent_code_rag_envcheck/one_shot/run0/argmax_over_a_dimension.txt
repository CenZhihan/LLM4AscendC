host_operator_src="""

#include "argmax_over_a_dimension_custom_tiling.h"
#include "register/op_def_registry.h"

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // conservative per-core share
static constexpr uint32_t ALIGN_ELEMS = 1024;
static constexpr uint32_t MAX_CHUNK = 65536;

static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ArgmaxOverADimensionCustomTilingData tiling;

    const gert::Shape *shape = context->GetInputShape(0);
    const auto &origin = shape->GetOriginShape();
    const std::vector<int64_t> dims = origin.GetDims();
    const uint32_t rank = static_cast<uint32_t>(dims.size());

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *dimAttr = attrs->GetAttrPointer<int32_t>(0);
    int32_t dim = (dimAttr != nullptr) ? *dimAttr : -1;

    int32_t axis = dim;
    if (axis < 0) axis += static_cast<int32_t>(rank);
    if (axis < 0 || axis >= static_cast<int32_t>(rank)) {
        // Fallback: reduce last axis
        axis = static_cast<int32_t>(rank) - 1;
    }

    // We design the kernel to reduce along the last axis. The frontend should transpose
    // when axis != last. Here, we compute rows (product of dims except last) and cols (last dim).
    uint64_t totalSize = 1;
    for (auto d : dims) totalSize *= static_cast<uint64_t>(d);
    const uint32_t cols = (rank > 0) ? static_cast<uint32_t>(dims.back()) : 1u;
    const uint32_t rows = (cols == 0) ? 0u : static_cast<uint32_t>(totalSize / cols);

    // Choose chunkCols to fit into UB budget. One float buffer of chunkCols is enough.
    uint32_t maxChunk = UB_BUDGET_BYTES / sizeof(float);
    maxChunk = AlignDown(maxChunk, ALIGN_ELEMS);
    if (maxChunk < ALIGN_ELEMS) maxChunk = ALIGN_ELEMS;
    if (maxChunk > MAX_CHUNK) maxChunk = AlignDown(MAX_CHUNK, ALIGN_ELEMS);
    uint32_t chunkCols = (cols < maxChunk) ? AlignDown(cols, ALIGN_ELEMS) : maxChunk;
    if (chunkCols == 0) chunkCols = cols;

    uint32_t blockDim = rows;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;

    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_chunkCols(chunkCols);
    tiling.set_blockDim(blockDim);
    tiling.set_dim(dim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);

    const auto &origin = x_shape->GetOriginShape();
    std::vector<int64_t> dims = origin.GetDims();

    // Attribute dim (optional)
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = -1;
    if (attrs != nullptr && attrs->GetAttrNum() > 0) {
        const int32_t *p = attrs->GetAttrPointer<int32_t>(0);
        if (p) dim = *p;
    }

    int32_t axis = dim;
    int32_t rank = static_cast<int32_t>(dims.size());
    if (axis < 0) axis += rank;
    if (axis < 0 || axis >= rank) axis = rank - 1;

    // Output dims = input dims with the reduction axis removed
    if (rank > 0) {
        dims.erase(dims.begin() + axis);
    }
    y_shape->SetDimNum(dims.size());
    for (size_t i = 0; i < dims.size(); ++i) {
        y_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Output is index type int32
    context->SetOutputDataType(0, ge::DT_INT32);
    return GRAPH_SUCCESS;
}
} // namespace ge

kernel_src="""

#include "kernel_operator.h"
#include <cfloat>

constexpr int32_t BUFFER_NUM = 1;

class KernelArgmaxOverADimensionCustom {
public:
    __aicore__ inline KernelArgmaxOverADimensionCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t chunkCols)
    {
        rows_ = rows;
        cols_ = cols;
        chunkCols_ = chunkCols;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)rows_ * (uint64_t)cols_);
        yGm_.SetGlobalBuffer((__gm__ int32_t*)y, rows_);

        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, chunkCols_ * sizeof(float));
        outPipe_.InitBuffer(outIdxQueue_, BUFFER_NUM, 1 * sizeof(int32_t));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) return;

        const uint32_t blockNum = (uint32_t)AscendC::GetBlockNum();
        const uint32_t blockIdx = (uint32_t)AscendC::GetBlockIdx();

        // Grid-stride over rows
        for (uint32_t row = blockIdx; row < rows_; row += blockNum) {
            ComputeRow(row);
        }
    }

private:
    __aicore__ inline void ComputeRow(uint32_t row)
    {
        float bestVal = -FLT_MAX;
        int32_t bestIdx = 0;

        uint32_t processed = 0;
        const uint64_t baseOffset = (uint64_t)row * (uint64_t)cols_;

        while (processed < cols_) {
            const uint32_t cur = (cols_ - processed > chunkCols_) ? chunkCols_ : (cols_ - processed);

            AscendC::LocalTensor<float> xLocal = tmpQueue_.AllocTensor<float>();
            AscendC::DataCopy(xLocal, xGm_[baseOffset + processed], cur);

            // Scalar scan inside the chunk to update argmax
            for (uint32_t i = 0; i < cur; ++i) {
                // Hypothetical API to read scalar from local tensor; for illustration.
                float v = xLocal.GetValue(i);
                if (v > bestVal) {
                    bestVal = v;
                    bestIdx = (int32_t)(processed + i);
                }
            }

            tmpQueue_.FreeTensor(xLocal);
            processed += cur;
        }

        // Write out single index for this row
        AscendC::LocalTensor<int32_t> outLocal = outIdxQueue_.AllocTensor<int32_t>();
        outLocal.SetValue(0, bestIdx);
        outIdxQueue_.EnQue(outLocal);

        AscendC::LocalTensor<int32_t> outLocalDeq = outIdxQueue_.DeQue<int32_t>();
        AscendC::DataCopy(yGm_[row], outLocalDeq, 1);
        outIdxQueue_.FreeTensor(outLocalDeq);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TPipe outPipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outIdxQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<int32_t> yGm_;

    uint32_t rows_ {0};
    uint32_t cols_ {0};
    uint32_t chunkCols_ {0};
};

extern "C" __global__ __aicore__ void argmax_over_a_dimension_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelArgmaxOverADimensionCustom op;
    // We expect axis to be last after frontend permutation; cols is the last-dimension length.
    op.Init(x, y, tiling_data.rows, tiling_data.cols, tiling_data.chunkCols);
    op.Process();
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor argmax_over_a_dimension_impl_npu(const at::Tensor& self, int64_t dim) {
    // Reduce along last dim in device kernel; frontend should transpose when needed.
    int64_t rank = self.dim();
    int64_t reduce_axis = dim;
    if (reduce_axis < 0) reduce_axis += rank;

    // Output shape: input with the reduction axis removed
    std::vector<int64_t> out_sizes = self.sizes().vec();
    if (reduce_axis >= 0 && reduce_axis < (int64_t)out_sizes.size()) {
        out_sizes.erase(out_sizes.begin() + reduce_axis);
    }

    // Indices type: int32 for this custom kernel
    auto result = at::empty(out_sizes, self.options().dtype(at::kInt));

    EXEC_NPU_CMD(aclnnArgmaxOverADimensionCustom, self, (int)dim, result);
    return result;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("argmax_over_a_dimension_custom", &argmax_over_a_dimension_impl_npu,
          "Argmax over a dimension (ArgmaxOverADimensionCustom on NPU)");
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = 'argmax_over_a_dimension_custom'

class ModelNew(torch.nn.Module):
    def __init__(self, dim: int):
        super(ModelNew, self).__init__()
        self.dim = dim

    def forward(self, x: torch.Tensor) -> torch.Tensor:
        # Kernel reduces along the last dimension. If dim is not last, transpose to last.
        dim = self.dim
        if dim < 0:
            dim = x.dim() + dim
        if dim != x.dim() - 1:
            x = x.transpose(dim, x.dim() - 1).contiguous()
            dim = -1  # reduce last
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, dim)
'''
