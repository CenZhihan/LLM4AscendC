kernel_src="""
#include "kernel_operator.h"

// NOTE:
// This kernel computes y = GELU(GELU(LeakyReLU(LeakyReLU(LogSumExp(x, dim=1, keepdim=True))))) per batch row.
// The inputs w and b are present to match the PyBind contract but are not used here.
// For numerical stability, LogSumExp is computed via max-trick within UB.

constexpr int32_t IN_Q_NUM = 1;
constexpr int32_t OUT_Q_NUM = 1;

class KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom {
public:
    __aicore__ inline KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, uint32_t B, uint32_t N)
    {
        (void)w; (void)b; // Unused by this kernel

        B_ = B;
        N_ = N;

        xGm_.SetGlobalBuffer((__gm__ float*)x, B_ * N_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, B_); // output [B, 1] flattened

        // UB buffers:
        // - inQueueRow: 1 row of length N
        // - tmpQueue:   1 row of length N (for shifted-exp)
        // - outQueue:   1 scalar per row
        pipe_.InitBuffer(inQueueRow_, IN_Q_NUM, N_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, N_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, OUT_Q_NUM, 1 * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (B_ == 0 || N_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t row = blockIdx; row < B_; row += blockNum) {
            // Copy one row from GM
            CopyInRow(row);

            // Compute logsumexp on the row -> scalar
            ComputeRow();

            // Apply activations chain on scalar and write out
            PostActAndCopyOut(row);
        }
    }

private:
    __aicore__ inline void CopyInRow(uint32_t row)
    {
        AscendC::LocalTensor<float> rowLocal = inQueueRow_.AllocTensor<float>();
        AscendC::DataCopy(rowLocal, xGm_[row * N_], N_);
        inQueueRow_.EnQue(rowLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<float> rowLocal = inQueueRow_.DeQue<float>();
        AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> scalarLocal = outQueue_.AllocTensor<float>(); // will hold lse

        // 1) maxVal = reduce_max(rowLocal)
        float maxVal = -__FLT_MAX__;
        // Manually scan UB row to find max
        // Assuming element-wise access is supported via pointer-like iteration
        for (uint32_t i = 0; i < N_; ++i) {
            // Create a length-1 subview and load scalar
            AscendC::LocalTensor<float> elem = rowLocal[i];
            float v;
            AscendC::DataCopy(&v, elem, 1);
            if (v > maxVal) maxVal = v;
        }

        // 2) tmpLocal = exp(rowLocal - maxVal)
        AscendC::Adds(tmpLocal, rowLocal, -maxVal, N_);
        AscendC::Exp(tmpLocal, tmpLocal, N_);

        // 3) sumExp = reduce_sum(tmpLocal)
        float sumExp = 0.0f;
        for (uint32_t i = 0; i < N_; ++i) {
            AscendC::LocalTensor<float> elem = tmpLocal[i];
            float v;
            AscendC::DataCopy(&v, elem, 1);
            sumExp += v;
        }

        // 4) lse = log(sumExp) + maxVal
        float lse = AscendC::Log(sumExp) + maxVal;

        // write lse to scalarLocal[0]
        AscendC::DataCopy(scalarLocal, &lse, 1);

        // release
        tmpQueue_.FreeTensor(tmpLocal);
        inQueueRow_.FreeTensor(rowLocal);
        outQueue_.EnQue<float>(scalarLocal);
    }

    __aicore__ inline void PostActAndCopyOut(uint32_t row)
    {
        AscendC::LocalTensor<float> scalarLocal = outQueue_.DeQue<float>();
        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>(); // reuse for small ops
        AscendC::LocalTensor<float> outLocal = outQueue_.AllocTensor<float>();

        // Two LeakyReLU with negative_slope=0.01 on a scalar (len=1)
        // y = x + (neg - 1) * min(x, 0)
        const float neg = 0.01f;
        const float negCorr = neg - 1.0f;

        AscendC::Mins(tmp, scalarLocal, 0.0f, 1);
        AscendC::Muls(tmp, tmp, negCorr, 1);
        AscendC::Add(outLocal, scalarLocal, tmp, 1);

        AscendC::Mins(tmp, outLocal, 0.0f, 1);
        AscendC::Muls(tmp, tmp, negCorr, 1);
        AscendC::Add(outLocal, outLocal, tmp, 1);

        // Two GELU on a scalar
        AscendC::Gelu(outLocal, outLocal, 1);
        AscendC::Gelu(outLocal, outLocal, 1);

        // Store to GM at y[row]
        AscendC::DataCopy(yGm_[row], outLocal, 1);

        // Free locals
        tmpQueue_.FreeTensor(tmp);
        outQueue_.FreeTensor(outLocal);
        outQueue_.FreeTensor(scalarLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, IN_Q_NUM> inQueueRow_;
    AscendC::TQue<AscendC::TPosition::VECOUT, OUT_Q_NUM> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t B_{0};
    uint32_t N_{0};
};

extern "C" __global__ __aicore__ void gemm_log_sum_exp_leaky_relu_leaky_relu_gelu_gelu_custom(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    (void)workspace;
    GET_TILING_DATA(tilingData, tiling);
    KernelGemmLogSumExpLeakyReluLeakyReluGeluGeluCustom op;
    op.Init(x, w, b, y, tilingData.batch, tilingData.feature);
    op.Process();
}
