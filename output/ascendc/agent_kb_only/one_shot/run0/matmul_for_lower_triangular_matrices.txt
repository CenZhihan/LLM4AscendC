host_operator_src="""
#include "register/op_def_registry.h"
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"   // Matmul tiling API headers

namespace optiling {

static inline uint32_t GetSquareNFromShape(const gert::Shape* shape)
{
    // Expect 2D square (N, N). Fallback to product if not strictly 2D.
    if (shape->GetDimNum() == 2) {
        return static_cast<uint32_t>(shape->GetDim(0));
    }
    return static_cast<uint32_t>(shape->GetShapeSize());
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());

    // Read input shapes
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);

    const uint32_t N = GetSquareNFromShape(a_shape);

    // Configure multi-core matmul tiling for ND layout, A/B: float16, C: float
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);
    tiling.SetDim(1); // let tiler decide actual core usage
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBias(false);
    tiling.SetShape(N, N, N);
    tiling.SetSingleShape(N, N, N);
    tiling.SetOrgShape(N, N, N);
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // blockDim suggestion from tiler
    uint32_t coreNum = tiling.GetCoreNum();
    if (coreNum == 0) coreNum = 1;
    context->SetBlockDim(coreNum);

    // Save TCubeTiling to raw tiling buffer
    auto *rawBuf = context->GetRawTilingData();
    if (rawBuf == nullptr) {
        return ge::GRAPH_FAILED;
    }
    size_t cap = rawBuf->GetCapacity();
    size_t bytes = sizeof(optiling::TCubeTiling);
    if (cap < bytes) {
        return ge::GRAPH_FAILED;
    }
    // Copy as plain bytes
    ::memcpy(rawBuf->GetData(), &tilingData, bytes);
    rawBuf->SetDataSize(bytes);

    // No extra workspace requested on host
    size_t *workspaceSizes = context->GetWorkspaceSizes(1);
    workspaceSizes[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *a_shape = context->GetInputShape(0);
    const gert::Shape *b_shape = context->GetInputShape(1);
    // Output shape equals matmul(A,B) -> (N,N)
    gert::Shape *c_shape = context->GetOutputShape(0);
    // Assume square input; otherwise, multiply dims
    if (a_shape->GetDimNum() == 2 && b_shape->GetDimNum() == 2) {
        c_shape->SetDimNum(2);
        c_shape->SetDim(0, a_shape->GetDim(0));
        c_shape->SetDim(1, b_shape->GetDim(1));
    } else {
        *c_shape = *a_shape;
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // Inputs: float16, Output: float
    context->SetOutputDataType(0, ge::DT_FLOAT);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class MatmulForLowerTriangularMatricesCustom : public OpDef {
public:
    explicit MatmulForLowerTriangularMatricesCustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT16})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("C")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulForLowerTriangularMatricesCustom);
} // namespace ops

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using AscendC::GlobalTensor;

extern "C" __global__ __aicore__ void matmul_for_lower_triangular_matrices_custom(
    GM_ADDR AGM, GM_ADDR BGM, GM_ADDR CGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // Matmul type definitions: A/B are half ND, C is float ND, Normal layout
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::Normal> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::Normal> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::Normal> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    set_atomic_none();

    // Load TCubeTiling from GM
    optiling::TCubeTiling tiling;
    auto tilingGMU32 = reinterpret_cast<__gm__ const uint32_t*>(tilingGM);
    uint32_t *tilingU32 = reinterpret_cast<uint32_t*>(&tiling);
    for (uint32_t i = 0; i < sizeof(optiling::TCubeTiling) / sizeof(uint32_t); ++i) {
        tilingU32[i] = tilingGMU32[i];
    }

    // Bind GM tensors without explicit size (overload supported)
    GlobalTensor<half> aGlobal;
    GlobalTensor<half> bGlobal;
    GlobalTensor<float> cGlobal;
    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(AGM));
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(BGM));
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(CGM));

    // Create Matmul instance and register with TPipe
    matmul::Matmul<aType, bType, cType, biasType> mm;
    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);

    mm.Init(&tiling);
    // Set tensors and workspace
    mm.SetTensorA(aGlobal[0], false);
    mm.SetTensorB(bGlobal[0], false);
    mm.SetWorkspace(workspaceGM, 0);
    // Single-matmul execution in NORMAL layout: 1 batch, batchA=1, batchB=1
    mm.IterateNBatch(1, 1, 1, false);
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_half_if_needed(const at::Tensor& t) {
    if (t.scalar_type() == at::kHalf) return t;
    return t.to(at::kHalf);
}

at::Tensor matmul_lower_tri_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // Inputs are lower triangular; kernel performs matmul and returns result.
    // We cast A,B to half to match kernel schema, output is float32.
    at::Tensor A_half = to_half_if_needed(A);
    at::Tensor B_half = to_half_if_needed(B);

    // Allocate float32 output
    auto out_sizes = at::IntArrayRef({A.size(-2), B.size(-1)});
    at::Tensor C = at::empty(out_sizes, A.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnMatmulForLowerTriangularMatricesCustom, A_half, B_half, C);

    // The product of two lower triangular matrices is lower triangular by definition.
    // If strict masking is desired, uncomment the following:
    // C = at::tril(C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_for_lower_triangular_matrices_custom", &matmul_lower_tri_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_for_lower_triangular_matrices_custom", &matmul_lower_tri_impl_npu,
          "Matmul for lower-triangular matrices (MatmulForLowerTriangularMatricesCustom on NPU)");
}
