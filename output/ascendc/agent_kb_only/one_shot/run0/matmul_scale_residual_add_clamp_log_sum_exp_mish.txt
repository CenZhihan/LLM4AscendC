kernel_src="""
#include "kernel_operator.h"

// We'll implement a GEMM over rows with K-tiling and per-column accumulation.
// Then apply y = clamp((1+s) * y, lo, hi).
// Then compute z = logsumexp(y, dim=1, keepdim=True) using SoftmaxFlashV2 primitives.
// Finally output: z * mish(z), where mish(z) = z * tanh(ln(1 + exp(z))).
// Tiling data provides M, K, N and kTile for UB.

using namespace AscendC;

constexpr int32_t BUFFER_NUM = 1;

namespace MatmulScaleResAddClampLSEMishCustom {

struct TilingData {
    uint32_t M;
    uint32_t K;
    uint32_t N;
    uint32_t kTile;
    uint32_t blockDim;
};

class KernelMatmulScaleResidualAddClampLogSumExpMishCustom {
public:
    __aicore__ inline KernelMatmulScaleResidualAddClampLogSumExpMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b,
                                GM_ADDR scaling, GM_ADDR cmin, GM_ADDR cmax,
                                GM_ADDR y, const TilingData& t)
    {
        M_ = t.M;
        K_ = t.K;
        N_ = t.N;
        kTile_ = (t.kTile == 0) ? K_ : t.kTile;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), static_cast<uint32_t>(M_) * K_);
        wGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(w), static_cast<uint32_t>(N_) * K_);
        bGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(b), N_);
        sGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(scaling), 1);
        loGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cmin), 1);
        hiGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(cmax), 1);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), static_cast<uint32_t>(M_) * 1);

        // Queues/buffers
        // For per-segment vectors on K: xSeg, wSeg, mulTmp, redWork
        pipe_.InitBuffer(xSegQ_, BUFFER_NUM, kTile_ * sizeof(float));
        pipe_.InitBuffer(wSegQ_, BUFFER_NUM, kTile_ * sizeof(float));
        pipe_.InitBuffer(tmpQ_, BUFFER_NUM, kTile_ * sizeof(float));
        pipe_.InitBuffer(redWorkQ_, BUFFER_NUM, kTile_ * sizeof(float)); // work buffer for ReduceSum

        // Buffers for row-wise post ops
        pipe_.InitBuffer(rowQ_, 1, N_ * sizeof(float));        // full row buffer for LSE
        pipe_.InitBuffer(auxQ_, 1, N_ * sizeof(float));        // exp dst or aux
        pipe_.InitBuffer(blockQ_, 1, 32u);                     // minimal scalar buffer
    }

    __aicore__ inline void Process()
    {
        if (M_ == 0 || K_ == 0 || N_ == 0) return;

        // Load scalars s, lo, hi
        float scaleHost = 0.f, loHost = 0.f, hiHost = 0.f;
        {
            LocalTensor<float> oneS = blockQ_.AllocTensor<float>();
            DataCopy(oneS, sGm_, 1);
            PipeBarrier<PIPE_V>();
            scaleHost = oneS.GetValue(0);
            blockQ_.FreeTensor(oneS);
        }
        {
            LocalTensor<float> oneLo = blockQ_.AllocTensor<float>();
            DataCopy(oneLo, loGm_, 1);
            PipeBarrier<PIPE_V>();
            loHost = oneLo.GetValue(0);
            blockQ_.FreeTensor(oneLo);
        }
        {
            LocalTensor<float> oneHi = blockQ_.AllocTensor<float>();
            DataCopy(oneHi, hiGm_, 1);
            PipeBarrier<PIPE_V>();
            hiHost = oneHi.GetValue(0);
            blockQ_.FreeTensor(oneHi);
        }
        const float sAll = scaleHost + 1.0f;

        const uint32_t blkNum = static_cast<uint32_t>(GetBlockNum());
        const uint32_t blkIdx = static_cast<uint32_t>(GetBlockIdx());

        for (uint32_t m = blkIdx; m < M_; m += blkNum) {
            // Compute row m: y_row[n] = dot(x[m,:], w[n,:]) + b[n]
            // We'll iterate columns sequentially for simplicity.
            for (uint32_t n = 0; n < N_; ++n) {
                float acc = 0.0f;
                for (uint32_t kb = 0; kb < K_; kb += kTile_) {
                    uint32_t cur = K_ - kb;
                    if (cur > kTile_) cur = kTile_;

                    // xSeg <- x[m, kb:kb+cur]
                    LocalTensor<float> xSeg = xSegQ_.AllocTensor<float>();
                    DataCopy(xSeg, xGm_[m * K_ + kb], cur);
                    // wSeg <- w[n, kb:kb+cur]
                    LocalTensor<float> wSeg = wSegQ_.AllocTensor<float>();
                    DataCopy(wSeg, wGm_[n * K_ + kb], cur);
                    // mulTmp = xSeg * wSeg
                    LocalTensor<float> mulTmp = tmpQ_.AllocTensor<float>();
                    Mul(mulTmp, xSeg, wSeg, cur);

                    // Reduce sum over mulTmp
                    LocalTensor<float> redWork = redWorkQ_.AllocTensor<float>();
                    ReduceSum(mulTmp, mulTmp, redWork, cur);
                    PipeBarrier<PIPE_V>();
                    // Retrieve accumulation result
                    float partial = GetAccVal<float>();
                    acc += partial;

                    // free
                    redWorkQ_.FreeTensor(redWork);
                    tmpQ_.FreeTensor(mulTmp);
                    wSegQ_.FreeTensor(wSeg);
                    xSegQ_.FreeTensor(xSeg);
                }
                // add bias
                LocalTensor<float> btmp = blockQ_.AllocTensor<float>();
                DataCopy(btmp, bGm_[n], 1);
                PipeBarrier<PIPE_V>();
                acc += btmp.GetValue(0);
                blockQ_.FreeTensor(btmp);

                // fused scale-residual: y = y * s + y = y * (1 + s)
                acc *= sAll;

                // clamp
                if (acc < loHost) acc = loHost;
                if (acc > hiHost) acc = hiHost;

                // Store into row buffer (UB)
                if (n == 0) {
                    LocalTensor<float> row = rowQ_.AllocTensor<float>();
                    // initialize row with zeros to avoid UB usage faults
                    Muls(row, row, 0.0f, N_);
                    row.SetValue(0, acc);
                    rowQ_.EnQue(row);
                } else {
                    LocalTensor<float> row = rowQ_.DeQue<float>();
                    row.SetValue(n, acc);
                    rowQ_.EnQue(row);
                }
            }

            // Copy UB row to itself for LSE input
            LocalTensor<float> row = rowQ_.DeQue<float>();
            // Compute per-row logsumexp via SoftmaxFlashV2 primitives
            // Allocate buffers for outputs
            LocalTensor<float> dstTmp = auxQ_.AllocTensor<float>();      // same length as row
            LocalTensor<float> expSum = auxQ_.AllocTensor<float>();      // broadcast block
            LocalTensor<float> maxVal = auxQ_.AllocTensor<float>();      // broadcast block
            LocalTensor<uint8_t> tmpSpace = auxQ_.ReinterpretCast<uint8_t>(); // reuse for tmp

            // Build shape info and tiling
            SoftMaxShapeInfo shape;
            shape.srcM = 1;
            shape.srcK = N_;
            shape.oriSrcM = 1;
            shape.oriSrcK = N_;
            // Reserve local workspace for tiling-dependent temp space
            const uint32_t ty1 = sizeof(float);
            const uint32_t ty2 = sizeof(float);
            // use kernel tiling func to get tiling
            auto sftTiling = SoftMaxFlashV2TilingFunc(shape, ty1, ty2, 0, false, false, false, false);

            // Call SoftmaxFlashV2 (normal mode, not update)
            SoftmaxFlashV2<float, false, false>(dstTmp, expSum, maxVal, row, dstTmp, expSum, maxVal, tmpSpace, sftTiling, {});

            // expSum/maxVal are block-broadcasted. Take the first element.
            float mVal = maxVal.GetValue(0);
            float sVal = expSum.GetValue(0);

            // z = mVal + ln(sVal)
            float z = 0.0f;
            {
                LocalTensor<float> scalIn = blockQ_.AllocTensor<float>();
                LocalTensor<float> scalOut = blockQ_.AllocTensor<float>();
                scalIn.SetValue(0, sVal);
                Ln(scalOut, scalIn, 1);
                PipeBarrier<PIPE_V>();
                float lnSum = scalOut.GetValue(0);
                blockQ_.FreeTensor(scalOut);
                blockQ_.FreeTensor(scalIn);
                z = mVal + lnSum;
            }

            // mish(z) = z * tanh(ln(1 + exp(z)))
            float mishZ = 0.0f;
            {
                LocalTensor<float> tin = blockQ_.AllocTensor<float>();
                LocalTensor<float> tout = blockQ_.AllocTensor<float>();
                LocalTensor<float> ttmp = blockQ_.AllocTensor<float>();

                tin.SetValue(0, z);
                // e = exp(z)
                Exp(tout, tin, 1);
                float e = tout.GetValue(0);
                // sp = ln(1 + e)
                ttmp.SetValue(0, e + 1.0f);
                Ln(tout, ttmp, 1);
                float sp = tout.GetValue(0);
                // t = tanh(sp)
                tin.SetValue(0, sp);
                Tanh(tout, tin, 1);
                float t = tout.GetValue(0);
                mishZ = z * t; // z * tanh(softplus(z))

                blockQ_.FreeTensor(ttmp);
                blockQ_.FreeTensor(tout);
                blockQ_.FreeTensor(tin);
            }

            float outVal = z * mishZ;

            // Store to GM output y[m, 0]
            LocalTensor<float> outScal = blockQ_.AllocTensor<float>();
            outScal.SetValue(0, outVal);
            DataCopy(yGm_[m * 1 + 0], outScal, 1);
            blockQ_.FreeTensor(outScal);

            // free row/aux buffers
            auxQ_.FreeTensor(maxVal);
            auxQ_.FreeTensor(expSum);
            auxQ_.FreeTensor(dstTmp);
            rowQ_.FreeTensor(row);
        }
    }

private:
    // Global tensors
    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> bGm_;
    GlobalTensor<float> sGm_;
    GlobalTensor<float> loGm_;
    GlobalTensor<float> hiGm_;
    GlobalTensor<float> yGm_;

    // Queues
    TPipe pipe_;
    TQue<QuePosition::VECIN, BUFFER_NUM> xSegQ_;
    TQue<QuePosition::VECIN, BUFFER_NUM> wSegQ_;
    TQue<QuePosition::VECCALC, BUFFER_NUM> tmpQ_;
    TQue<QuePosition::VECCALC, BUFFER_NUM> redWorkQ_;
    TQue<QuePosition::VECOUT, 1> rowQ_;
    TQue<QuePosition::VECCALC, 1> auxQ_;
    TQue<QuePosition::VECCALC, 1> blockQ_;

    // Shape parameters
    uint32_t M_{0}, K_{0}, N_{0}, kTile_{0};
};

} // namespace MatmulScaleResAddClampLSEMishCustom
