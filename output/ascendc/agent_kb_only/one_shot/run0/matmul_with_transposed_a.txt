host_tiling_src="""
#pragma once
#include <cstring>
#include "register/op_def_registry.h"
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"

namespace optiling {

static inline uint64_t GetShapeElemNum(const gert::Shape *shape)
{
    return static_cast<uint64_t>(shape->GetOriginShape().GetShapeSize());
}

static ge::graphStatus TilingFunc(gert::TilingContext* context)
{
    // Read input shapes
    const gert::Shape* a_shape = context->GetInputShape(0);  // expected [K, M]
    const gert::Shape* b_shape = context->GetInputShape(1);  // expected [K, N]

    const auto& a_dims = a_shape->GetOriginShape().GetDims();
    const auto& b_dims = b_shape->GetOriginShape().GetDims();
    if (a_dims.size() != 2 || b_dims.size() != 2) {
        return ge::GRAPH_FAILED;
    }

    // Effective matmul after A^T: (M, K) x (K, N) -> (M, N)
    const uint32_t M = static_cast<uint32_t>(a_dims[1]);
    const uint32_t K = static_cast<uint32_t>(a_dims[0]);
    const uint32_t N = static_cast<uint32_t>(b_dims[1]);

    // Build tiling using MultiCoreMatmulTiling
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling mmTiling(ascendcPlatform);
    mmTiling.SetDim(1); // let tiler decide core usage

    // A is transposed logically, B is not.
    mmTiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                      matmul_tiling::DataType::DT_FLOAT, /*isTrans*/ true);
    mmTiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                      matmul_tiling::DataType::DT_FLOAT, /*isTrans*/ false);
    mmTiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                      matmul_tiling::DataType::DT_FLOAT);
    mmTiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                         matmul_tiling::DataType::DT_FLOAT);
    mmTiling.EnableBias(false);

    mmTiling.SetShape(M, N, K);
    mmTiling.SetSingleShape(M, N, K);
    mmTiling.SetOrgShape(M, N, K);
    mmTiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = mmTiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Set block dim decided by tiler
    context->SetBlockDim(mmTiling.GetCoreNum());

    // Save TCubeTiling to buffer
    auto* raw = context->GetRawTilingData();
    const size_t sz = sizeof(optiling::TCubeTiling);
    if (raw->GetCapacity() < sz) {
        return ge::GRAPH_FAILED;
    }
    std::memcpy(raw->GetData(), &tilingData, sz);
    raw->SetDataSize(sz);

    // No extra workspace required by host
    size_t* workspaces = context->GetWorkspaceSizes(1);
    workspaces[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling
"""

host_operator_src="""
#include "matmul_with_transposed_a_custom_tiling.h"
#include "register/op_def_registry.h"

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext* context)
{
    const gert::Shape* a_shape = context->GetInputShape(0); // [K, M]
    const gert::Shape* b_shape = context->GetInputShape(1); // [K, N]
    const auto& ad = a_shape->GetOriginShape().GetDims();
    const auto& bd = b_shape->GetOriginShape().GetDims();
    if (ad.size() != 2 || bd.size() != 2) {
        return GRAPH_FAILED;
    }
    const int64_t M = ad[1];
    const int64_t N = bd[1];
    gert::Shape* y_shape = context->GetOutputShape(0);

    gert::Shape origin;
    origin.SetDimNum(2);
    int64_t dims[2] = {M, N};
    origin.SetDims(dims, 2);
    y_shape->SetOriginShape(origin);
    y_shape->SetStorageShape(origin);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext* context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

} // namespace ge

namespace ops {

class MatmulWithTransposedACustom : public OpDef {
public:
    explicit MatmulWithTransposedACustom(const char* name) : OpDef(name)
    {
        this->Input("A")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Input("B")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("Y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);

        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(MatmulWithTransposedACustom);

} // namespace ops

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = float;
using B_T = float;
using C_T = float;
using BiasT = float;

extern "C" __global__ __aicore__ void matmul_with_transposed_a_custom(GM_ADDR aGM, GM_ADDR bGM, GM_ADDR yGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // Define MatmulType: NORMAL layout. A uses transpose feature.
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, A_T, true,  LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, B_T, false, LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

    set_atomic_none();

    // Load TCubeTiling from GM
    TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < static_cast<int>(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }

    // Initialize GM tensors
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;

    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * sizeof(A_T);
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * sizeof(B_T);
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * sizeof(C_T);

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(aGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(bGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(yGM), sizeC);

    tiling.shareMode = 0;
    tiling.shareL1Size = 512 * 1024;
    tiling.shareL0CSize = 128 * 1024;
    tiling.shareUbSize = 0;

    int offset_a = 0, offset_b = 0, offset_c = 0;

    AscendC::GlobalTensor<A_T> gm_a;
    gm_a.SetGlobalBuffer(const_cast<__gm__ A_T*>(aGlobal[offset_a].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreK);
    AscendC::GlobalTensor<B_T> gm_b;
    gm_b.SetGlobalBuffer(const_cast<__gm__ B_T*>(bGlobal[offset_b].GetPhyAddr()),
                         tiling.singleCoreK * tiling.singleCoreN);
    AscendC::GlobalTensor<C_T> gm_c;
    gm_c.SetGlobalBuffer(const_cast<__gm__ C_T*>(cGlobal[offset_c].GetPhyAddr()),
                         tiling.singleCoreM * tiling.singleCoreN);

    // Create Matmul instance with NORMAL config
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;

    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);

    mm1.Init(&tiling);

    // Set matrices: A is transposed logically; B is not transposed.
    mm1.SetTensorA(gm_a, /*isTransposeAIn*/ 1);
    mm1.SetTensorB(gm_b, /*isTransposeBIn*/ 0);
    mm1.SetWorkspace(workspaceGM, 0);

    // Single matmul as one-batch iterate
    mm1.IterateBatch(gm_c, /*batchA*/ 1, /*batchB*/ 1, /*enSequentialWrite*/ false);
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor matmul_with_transposed_a_custom_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    // A: [K, M], B: [K, N] -> Y: [M, N]
    TORCH_CHECK(A.dim() == 2 && B.dim() == 2, \"Inputs must be 2D\");
    TORCH_CHECK(A.size(0) == B.size(0), \"K dimension must match between A and B\");
    at::Tensor Y = at::empty({A.size(1), B.size(1)}, A.options());
    EXEC_NPU_CMD(aclnnMatmulWithTransposedACustom, A, B, Y);
    return Y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"matmul_with_transposed_a_custom\", &matmul_with_transposed_a_custom_impl_npu);

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
