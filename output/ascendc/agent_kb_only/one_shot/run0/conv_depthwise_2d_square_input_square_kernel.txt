static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwise2dSquareInputSquareKernelCustomTilingData tiling;

    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    // Expect x: [N, C, H, W], weight: [C, 1, KH, KW]
    const uint32_t N = static_cast<uint32_t>(xShape->GetDim(0));
    const uint32_t C = static_cast<uint32_t>(xShape->GetDim(1));
    const uint32_t H = static_cast<uint32_t>(xShape->GetDim(2));
    const uint32_t W = static_cast<uint32_t>(xShape->GetDim(3));
    const uint32_t KH = static_cast<uint32_t>(wShape->GetDim(2));
    const uint32_t KW = static_cast<uint32_t>(wShape->GetDim(3));

    if (KH != 3 || KW != 3) {
        // Fallback safe-guard: only 3x3 supported by this custom kernel
        return ge::GRAPH_FAILED;
    }
    const uint32_t Ho = H - KH + 1; // no padding, stride=1
    const uint32_t Wo = W - KW + 1;

    // UB budget planning:
    // We'll compute row-by-row. Per row buffers:
    //  - temp: Wo floats
    //  - acc:  Wo floats
    // We process tileRows rows per iteration to improve locality for weight reuse.
    // UB ~ 2 * Wo * 4 bytes * 1 = ~ 2KB for Wo=510; plenty small.
    // Choose tileRows to amortize scheduling overhead; keep modest (e.g., 32).
    uint32_t tileRows = 32;
    if (tileRows > Ho) tileRows = Ho;

    // Blocks: distribute across N*C channel-batch pairs
    const uint32_t totalTasks = N * C;
    uint32_t blockDim = totalTasks;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 64u) blockDim = 64u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_C(C);
    tiling.set_H(H);
    tiling.set_W(W);
    tiling.set_KH(KH);
    tiling.set_KW(KW);
    tiling.set_Ho(Ho);
    tiling.set_Wo(Wo);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0; // no extra workspace
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvDW2D {
public:
    __aicore__ inline KernelConvDW2D() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t N, uint32_t C, uint32_t H, uint32_t W,
                                uint32_t KH, uint32_t KW, uint32_t Ho, uint32_t Wo,
                                uint32_t tileRows)
    {
        N_ = N; C_ = C; H_ = H; W_ = W; KH_ = KH; KW_ = KW; Ho_ = Ho; Wo_ = Wo;
        tileRows_ = tileRows;

        const uint32_t xSize = N_ * C_ * H_ * W_;
        const uint32_t wSize = C_ * KH_ * KW_;
        const uint32_t ySize = N_ * C_ * Ho_ * Wo_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        // Per-row temp and accumulator buffers
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, Wo_ * sizeof(float));
        pipe_.InitBuffer(accQueue_, BUFFER_NUM, Wo_ * sizeof(float));
        // Small buffer to bring one channel's 3x3 kernel
        pipe_.InitBuffer(wQueue_, BUFFER_NUM, KH_ * KW_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint32_t totalTasks = N_ * C_;
        for (uint32_t t = blockIdx; t < totalTasks; t += blockNum) {
            const uint32_t n = t / C_;
            const uint32_t c = t % C_;

            // Load weights for this channel
            AscendC::LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
            const uint32_t wBase = c * (KH_ * KW_);
            AscendC::DataCopy(wLocal, wGm_[wBase], KH_ * KW_);

            // Read 9 weight scalars from local buffer
            float w00 = wLocal.GetValue(0);
            float w01 = wLocal.GetValue(1);
            float w02 = wLocal.GetValue(2);
            float w10 = wLocal.GetValue(3);
            float w11 = wLocal.GetValue(4);
            float w12 = wLocal.GetValue(5);
            float w20 = wLocal.GetValue(6);
            float w21 = wLocal.GetValue(7);
            float w22 = wLocal.GetValue(8);
            wQueue_.FreeTensor(wLocal);

            const uint32_t xBase = ((n * C_ + c) * H_ * W_);
            const uint32_t yBase = ((n * C_ + c) * Ho_ * Wo_);

            for (uint32_t rowStart = 0; rowStart < Ho_; rowStart += tileRows_) {
                uint32_t rowsThis = tileRows_;
                if (rowStart + rowsThis > Ho_) rowsThis = Ho_ - rowStart;

                // Compute rows one by one to keep UB usage small
                for (uint32_t r = 0; r < rowsThis; ++r) {
                    // acc buffer
                    AscendC::LocalTensor<float> accLocal = accQueue_.AllocTensor<float>();
                    bool first = true;

                    // 3x3 taps
                    // Each tempLocal loads a contiguous slice of length Wo_ from GM
                    // Offsets: ((rowStart + r + dy) * W_ + dx)
                    {
                        // dy=0, dx=0
                        AscendC::LocalTensor<float> tmp0 = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp0, xGm_[xBase + (rowStart + r + 0) * W_ + 0], Wo_);
                        AscendC::Muls(tmp0, tmp0, w00, Wo_);
                        AscendC::DataCopy(accLocal, tmp0, Wo_);
                        tmpQueue_.FreeTensor(tmp0);
                        first = false;
                    }
                    {
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, xGm_[xBase + (rowStart + r + 0) * W_ + 1], Wo_);
                        AscendC::Muls(tmp, tmp, w01, Wo_);
                        AscendC::Add(accLocal, accLocal, tmp, Wo_);
                        tmpQueue_.FreeTensor(tmp);
                    }
                    {
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, xGm_[xBase + (rowStart + r + 0) * W_ + 2], Wo_);
                        AscendC::Muls(tmp, tmp, w02, Wo_);
                        AscendC::Add(accLocal, accLocal, tmp, Wo_);
                        tmpQueue_.FreeTensor(tmp);
                    }
                    {
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, xGm_[xBase + (rowStart + r + 1) * W_ + 0], Wo_);
                        AscendC::Muls(tmp, tmp, w10, Wo_);
                        AscendC::Add(accLocal, accLocal, tmp, Wo_);
                        tmpQueue_.FreeTensor(tmp);
                    }
                    {
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, xGm_[xBase + (rowStart + r + 1) * W_ + 1], Wo_);
                        AscendC::Muls(tmp, tmp, w11, Wo_);
                        AscendC::Add(accLocal, accLocal, tmp, Wo_);
                        tmpQueue_.FreeTensor(tmp);
                    }
                    {
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, xGm_[xBase + (rowStart + r + 1) * W_ + 2], Wo_);
                        AscendC::Muls(tmp, tmp, w12, Wo_);
                        AscendC::Add(accLocal, accLocal, tmp, Wo_);
                        tmpQueue_.FreeTensor(tmp);
                    }
                    {
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, xGm_[xBase + (rowStart + r + 2) * W_ + 0], Wo_);
                        AscendC::Muls(tmp, tmp, w20, Wo_);
                        AscendC::Add(accLocal, accLocal, tmp, Wo_);
                        tmpQueue_.FreeTensor(tmp);
                    }
                    {
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, xGm_[xBase + (rowStart + r + 2) * W_ + 1], Wo_);
                        AscendC::Muls(tmp, tmp, w21, Wo_);
                        AscendC::Add(accLocal, accLocal, tmp, Wo_);
                        tmpQueue_.FreeTensor(tmp);
                    }
                    {
                        AscendC::LocalTensor<float> tmp = tmpQueue_.AllocTensor<float>();
                        AscendC::DataCopy(tmp, xGm_[xBase + (rowStart + r + 2) * W_ + 2], Wo_);
                        AscendC::Muls(tmp, tmp, w22, Wo_);
                        AscendC::Add(accLocal, accLocal, tmp, Wo_);
                        tmpQueue_.FreeTensor(tmp);
                    }

                    // Write out one row
                    AscendC::DataCopy(yGm_[yBase + (rowStart + r) * Wo_], accLocal, Wo_);
                    accQueue_.FreeTensor(accLocal);
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> accQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, C_{0}, H_{0}, W_{0}, KH_{0}, KW_{0}, Ho_{0}, Wo_{0}, tileRows_{0};
    // alias to avoid confusion
    uint32_t N_, C_, H_, W_, KH_, KW_, Ho_, Wo_, tileRows_;
};

extern "C" __global__ __aicore__ void conv_depthwise_2d_square_input_square_kernel_custom(
    GM_ADDR x, GM_ADDR w, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConvDW2D op;
    op.Init(x, w, y,
            tilingData.N, tilingData.C, tilingData.H, tilingData.W,
            tilingData.KH, tilingData.KW, tilingData.Ho, tilingData.Wo,
            tilingData.tileRows);
    op.Process();
}
