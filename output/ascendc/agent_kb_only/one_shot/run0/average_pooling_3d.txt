namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const std::vector<int64_t> &dims = x_shape->GetOriginShape().GetDims();

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const int32_t *kernelPtr = attrs->GetAttrPointer<int32_t>(0);
    const int32_t *stridePtr = attrs->GetAttrPointer<int32_t>(1);
    const int32_t *padPtr    = attrs->GetAttrPointer<int32_t>(2);

    int64_t k = static_cast<int64_t>(*kernelPtr);
    int64_t s = (stridePtr && *stridePtr > 0) ? static_cast<int64_t>(*stridePtr) : k;
    int64_t p = padPtr ? static_cast<int64_t>(*padPtr) : 0;

    int64_t N=1,C=1,D=1,H=1,W=1;
    if (dims.size() == 5) {
        N = dims[0]; C = dims[1]; D = dims[2]; H = dims[3]; W = dims[4];
    } else if (dims.size() == 3) {
        D = dims[0]; H = dims[1]; W = dims[2];
    }

    auto outD = (D + 2 * p - k) / s + 1;
    auto outH = (H + 2 * p - k) / s + 1;
    auto outW = (W + 2 * p - k) / s + 1;

    gert::Shape *y_shape = context->GetOutputShape(0);
    std::vector<int64_t> outDims = {N, C, outD, outH, outW};
    y_shape->SetDimNum(outDims.size());
    for (size_t i = 0; i < outDims.size(); ++i) y_shape->SetDim(i, outDims[i]);
    return GRAPH_SUCCESS;
}

    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;

kernel_src="""
#include "kernel_operator.h"

class KernelAveragePooling3dCustom {
public:
    __aicore__ inline KernelAveragePooling3dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, GM_ADDR /*workspace*/,
                                uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w,
                                uint32_t outD, uint32_t outH, uint32_t outW,
                                uint32_t kD, uint32_t kH, uint32_t kW,
                                uint32_t sD, uint32_t sH, uint32_t sW,
                                uint32_t pD, uint32_t pH, uint32_t pW,
                                uint8_t countIncludePad)
    {
        n_ = n; c_ = c; d_ = d; h_ = h; w_ = w;
        outD_ = outD; outH_ = outH; outW_ = outW;
        kD_ = kD; kH_ = kH; kW_ = kW;
        sD_ = sD; sH_ = sH; sW_ = sW;
        pD_ = pD; pH_ = pH; pW_ = pW;
        countIncludePad_ = (countIncludePad != 0);

        inSize_  = static_cast<uint64_t>(n_) * c_ * d_ * h_ * w_;
        outSize_ = static_cast<uint64_t>(n_) * c_ * outD_ * outH_ * outW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, inSize_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, outSize_);

        // Two small VECCALC queues for scalar ops
        pipe_.InitBuffer(valQue_, 1, sizeof(float)); // for individual element
        pipe_.InitBuffer(sumQue_, 1, sizeof(float)); // for running sum
    }

    __aicore__ inline void Process()
    {
        if (outSize_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint64_t outIndex = blockIdx; outIndex < outSize_; outIndex += blockNum) {
            // Decompose outIndex to (n, c, od, oh, ow)
            uint64_t tmp = outIndex;
            uint32_t ow = tmp % outW_; tmp /= outW_;
            uint32_t oh = tmp % outH_; tmp /= outH_;
            uint32_t od = tmp % outD_; tmp /= outD_;
            uint32_t cc = tmp % c_;    tmp /= c_;
            uint32_t nn = tmp; // remaining

            // Compute pooling window start indices (with padding)
            int64_t dStart = static_cast<int64_t>(od) * sD_ - static_cast<int64_t>(pD_);
            int64_t hStart = static_cast<int64_t>(oh) * sH_ - static_cast<int64_t>(pH_);
            int64_t wStart = static_cast<int64_t>(ow) * sW_ - static_cast<int64_t>(pW_);

            uint32_t validCount = 0;
            bool hasInit = false;

            AscendC::LocalTensor<float> sumLocal = sumQue_.AllocTensor<float>();
            // Traverse kernel window
            for (uint32_t kd = 0; kd < kD_; ++kd) {
                int64_t id = dStart + static_cast<int64_t>(kd);
                if (id < 0 || id >= static_cast<int64_t>(d_)) continue;
                for (uint32_t kh = 0; kh < kH_; ++kh) {
                    int64_t ih = hStart + static_cast<int64_t>(kh);
                    if (ih < 0 || ih >= static_cast<int64_t>(h_)) continue;
                    for (uint32_t kw = 0; kw < kW_; ++kw) {
                        int64_t iw = wStart + static_cast<int64_t>(kw);
                        if (iw < 0 || iw >= static_cast<int64_t>(w_)) continue;

                        uint64_t inOffset = ((((static_cast<uint64_t>(nn) * c_ + cc) * d_ +
                                                static_cast<uint32_t>(id)) * h_ +
                                               static_cast<uint32_t>(ih)) * w_ +
                                              static_cast<uint32_t>(iw));
                        AscendC::LocalTensor<float> valLocal = valQue_.AllocTensor<float>();
                        AscendC::DataCopy(valLocal, xGm_[inOffset], 1);

                        if (!hasInit) {
                            AscendC::DataCopy(sumLocal, valLocal, 1);
                            hasInit = true;
                        } else {
                            AscendC::Add(sumLocal, sumLocal, valLocal, 1);
                        }
                        valQue_.FreeTensor(valLocal);
                        ++validCount;
                    }
                }
            }

            uint32_t divisor = countIncludePad_ ? (kD_ * kH_ * kW_) : validCount;
            if (!hasInit) {
                // No valid element in the window: output 0
                // Build zero from any element times 0
                AscendC::LocalTensor<float> valLocal = valQue_.AllocTensor<float>();
                AscendC::DataCopy(valLocal, xGm_[0], 1);
                AscendC::Muls(sumLocal, valLocal, 0.0f, 1);
                valQue_.FreeTensor(valLocal);
            } else {
                if (divisor > 0) {
                    float scale = 1.0f / static_cast<float>(divisor);
                    AscendC::Muls(sumLocal, sumLocal, scale, 1);
                } else {
                    // divisor == 0, set to 0
                    AscendC::LocalTensor<float> valLocal = valQue_.AllocTensor<float>();
                    AscendC::DataCopy(valLocal, xGm_[0], 1);
                    AscendC::Muls(sumLocal, valLocal, 0.0f, 1);
                    valQue_.FreeTensor(valLocal);
                }
            }

            AscendC::DataCopy(yGm_[outIndex], sumLocal, 1);
            sumQue_.FreeTensor(sumLocal);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> valQue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> sumQue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{1}, c_{1}, d_{1}, h_{1}, w_{1};
    uint32_t outD_{1}, outH_{1}, outW_{1};
    uint32_t kD_{1}, kH_{1}, kW_{1};
    uint32_t sD_{1}, sH_{1}, sW_{1};
    uint32_t pD_{0}, pH_{0}, pW_{0};
    bool countIncludePad_{false};

    uint64_t inSize_{0}, outSize_{0};
};

extern "C" __global__ __aicore__ void average_pooling_3d_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelAveragePooling3dCustom op;
    op.Init(x, y, workspace,
            tiling_data.n, tiling_data.c, tiling_data.d, tiling_data.h, tiling_data.w,
            tiling_data.outD, tiling_data.outH, tiling_data.outW,
            tiling_data.kD, tiling_data.kH, tiling_data.kW,
            tiling_data.sD, tiling_data.sH, tiling_data.sW,
            tiling_data.pD, tiling_data.pH, tiling_data.pW,
            tiling_data.countIncludePad);
    op.Process();
}
