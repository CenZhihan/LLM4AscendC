kernel_src="""
#include "kernel_operator.h"

class KernelConv2dSubtractHardSwishMaxPoolMishCustom {
public:
    __aicore__ inline KernelConv2dSubtractHardSwishMaxPoolMishCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t n, uint32_t c, uint32_t h, uint32_t w,
                                uint32_t hOut, uint32_t wOut,
                                uint32_t poolKernel, float subtractValue,
                                uint32_t tmpBufferSize)
    {
        n_ = n; c_ = c; h_ = h; w_ = w;
        hOut_ = hOut; wOut_ = wOut;
        k_ = poolKernel;
        subtract_ = subtractValue;

        xGm_.SetGlobalBuffer((__gm__ float*)x, n_ * c_ * h_ * w_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, n_ * c_ * hOut_ * wOut_);

        // We will use single-element locals repeatedly.
        pipe_.InitBuffer(inQueue_, 1, sizeof(float));
        pipe_.InitBuffer(auxQueue_, 1, sizeof(float));
        pipe_.InitBuffer(outQueue_, 1, sizeof(float));
        pipe_.InitBuffer(tmpQueue_, 1, tmpBufferSize);
    }

    __aicore__ inline void Process()
    {
        if (n_ == 0 || c_ == 0 || hOut_ == 0 || wOut_ == 0) { return; }
        // Only support k_ >= 1, implemented for k_ == 2 typical case.
        const uint32_t stride = k_;
        for (uint32_t ni = 0; ni < n_; ++ni) {
            for (uint32_t ci = 0; ci < c_; ++ci) {
                for (uint32_t ho = 0; ho < hOut_; ++ho) {
                    const uint32_t hi0 = ho * stride;
                    for (uint32_t wo = 0; wo < wOut_; ++wo) {
                        const uint32_t wi0 = wo * stride;

                        // Compute pooled value: max over k_ x k_ window after subtract + hardswish.
                        bool first = true;
                        AscendC::LocalTensor<float> pooledLocal = outQueue_.AllocTensor<float>();

                        for (uint32_t kh = 0; kh < k_; ++kh) {
                            uint32_t hi = hi0 + kh;
                            if (hi >= h_) continue;
                            for (uint32_t kw = 0; kw < k_; ++kw) {
                                uint32_t wi = wi0 + kw;
                                if (wi >= w_) continue;

                                // Load one element
                                AscendC::LocalTensor<float> xLocal = inQueue_.AllocTensor<float>();
                                uint32_t xIndex = ((((ni * c_ + ci) * h_ + hi) * w_) + wi);
                                AscendC::DataCopy(xLocal, xGm_[xIndex], 1);

                                // x = x - subtract_
                                AscendC::Adds(xLocal, xLocal, -subtract_, 1);

                                // HardSwish: x * clamp(x+3, 0, 6) / 6
                                AscendC::LocalTensor<float> tLocal = auxQueue_.AllocTensor<float>();
                                AscendC::Adds(tLocal, xLocal, 3.0f, 1);
                                AscendC::Maxs(tLocal, tLocal, 0.0f, 1);
                                AscendC::Mins(tLocal, tLocal, 6.0f, 1);
                                AscendC::Muls(tLocal, tLocal, (1.0f/6.0f), 1);
                                AscendC::Mul(xLocal, xLocal, tLocal, 1); // xLocal now hardswish(x - subtract)

                                // Update max into pooledLocal
                                if (first) {
                                    // Initialize pooledLocal with current value
                                    AscendC::DataCopy(pooledLocal, xLocal, 1);
                                    first = false;
                                } else {
                                    AscendC::Max(pooledLocal, pooledLocal, xLocal, 1);
                                }

                                auxQueue_.FreeTensor(tLocal);
                                inQueue_.FreeTensor(xLocal);
                            }
                        }

                        // Mish: x * tanh(ln(1 + exp(x)))
                        // tmp1 = exp(pooled)
                        AscendC::LocalTensor<float> tmp1 = inQueue_.AllocTensor<float>();
                        AscendC::Exp(tmp1, pooledLocal, 1);
                        AscendC::Adds(tmp1, tmp1, 1.0f, 1);
                        AscendC::LocalTensor<float> softplus = auxQueue_.AllocTensor<float>();
                        AscendC::Ln(softplus, tmp1, 1);

                        // tanh(softplus) -> use Tanh high-level API with shared tmp buffer
                        AscendC::LocalTensor<uint8_t> sharedTmpBuffer = tmpQueue_.AllocTensor<uint8_t>();
                        AscendC::LocalTensor<float> tanhVal = inQueue_.AllocTensor<float>();
                        AscendC::Tanh<float, false>(tanhVal, softplus, sharedTmpBuffer, 1);
                        tmpQueue_.FreeTensor(sharedTmpBuffer);

                        // y = pooled * tanhVal
                        AscendC::Mul(pooledLocal, pooledLocal, tanhVal, 1);

                        // Store to output
                        uint32_t yIndex = ((((ni * c_ + ci) * hOut_ + ho) * wOut_) + wo);
                        AscendC::DataCopy(yGm_[yIndex], pooledLocal, 1);

                        inQueue_.FreeTensor(tmp1);
                        auxQueue_.FreeTensor(softplus);
                        inQueue_.FreeTensor(tanhVal);
                        outQueue_.FreeTensor(pooledLocal);
                    }
                }
            }
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, 1> inQueue_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> auxQueue_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, 1> outQueue_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, 1> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, c_{0}, h_{0}, w_{0};
    uint32_t hOut_{0}, wOut_{0};
    uint32_t k_{2};
    float subtract_{0.0f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_hard_swish_max_pool_mish_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv2dSubtractHardSwishMaxPoolMishCustom op;
    op.Init(x, y,
            tilingData.n, tilingData.c, tilingData.h, tilingData.w,
            tilingData.hOut, tilingData.wOut,
            tilingData.poolKernel, tilingData.subtractValue,
            tilingData.tmpBufferSize);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}

model_src='''
import torch
import torch_npu
import custom_ops_lib

_FN = "conv2d_subtract_hard_swish_max_pool_mish_custom"

class ModelNew(torch.nn.Module):
    def __init__(self, in_channels, out_channels, kernel_size, subtract_value, pool_kernel_size):
        super(ModelNew, self).__init__()
        self.subtract_value = float(subtract_value)
        self.pool_kernel_size = int(pool_kernel_size)
        # conv parameters are provided dynamically in forward (weight, bias)

    def forward(self, x: torch.Tensor, weight: torch.Tensor, conv_bias_opt: torch.Tensor):
        # Keep conv2d on framework side as per original semantics
        conv_b = conv_bias_opt if conv_bias_opt is not None else None
        x = torch.nn.functional.conv2d(x, weight, conv_b, stride=1, padding=0, dilation=1)
        # Fused custom kernel performs: subtract -> HardSwish -> MaxPool2d -> Mish
        fn = getattr(custom_ops_lib, _FN)
        return fn(x, self.subtract_value, self.pool_kernel_size)
