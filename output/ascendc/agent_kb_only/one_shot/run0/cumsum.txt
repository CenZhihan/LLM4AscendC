host_operator_src="""

#include "cumsum_custom_tiling.h"
#include "register/op_def_registry.h"

namespace {

static inline uint32_t AlignUp(uint32_t x, uint32_t a)
{
    return (x + a - 1u) / a * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // anonymous namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    CumsumCustomTilingData tiling;

    // Only ND 2D tensors are supported.
    const gert::Shape* in_shape = context->GetInputShape(0);
    const auto& origin = in_shape->GetOriginShape();

    if (origin.GetDimNum() != 2) {
        return ge::GRAPH_FAILED;
    }

    uint32_t rows = static_cast<uint32_t>(origin.GetDim(0));
    uint32_t cols = static_cast<uint32_t>(origin.GetDim(1));

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = 1;
    if (attrs && attrs->GetAttrNum() > 0) {
        const int32_t *dimPtr = attrs->GetAttrPointer<int32_t>(0);
        if (dimPtr) dim = *dimPtr;
    }
    uint32_t isLastAxis = (dim == 1) ? 1u : 0u;

    // Compute inner with 32-byte alignment: inner*sizeof(T) must be multiple of 32
    // For float32, sizeof(T)=4 => inner multiple of 8 elements.
    const uint32_t typeSize = 4;
    const uint32_t inner = AlignUp(cols * typeSize, 32) / typeSize;

    // UB budget and tiling for per-tile rows
    // UB usage per tile (bytes): src(tileRows*inner) + dst(tileRows*inner) + lastRow(inner)
    // = (2*tileRows*inner + inner) * sizeof(T)
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
    uint32_t maxElems = UB_BUDGET_BYTES / typeSize;
    uint32_t tileRows = 1;
    if (inner > 0 && maxElems > inner) {
        tileRows = (maxElems - inner) / (2 * inner);
        if (tileRows == 0) tileRows = 1;
    }

    if (tileRows > rows) tileRows = rows;

    const uint32_t numTiles = CeilDiv(rows, tileRows);
    uint32_t blockDim = numTiles;
    if (blockDim < 1u) blockDim = 1u;
    if (blockDim > 32u) blockDim = 32u; // cap active cores

    context->SetBlockDim(blockDim);

    tiling.set_rows(rows);
    tiling.set_cols(cols);
    tiling.set_inner(inner);
    tiling.set_tileRows(tileRows);
    tiling.set_blockDim(blockDim);
    tiling.set_isLastAxis(isLastAxis);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    gert::Shape *y_shape = context->GetOutputShape(0);
    *y_shape = *x_shape;
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType x_dtype = context->GetInputDataType(0);
    context->SetOutputDataType(0, x_dtype);
    return GRAPH_SUCCESS;
}
} // namespace ge

namespace ops {
class CumsumCustom : public OpDef {
public:
    explicit CumsumCustom(const char* name) : OpDef(name)
    {
        this->Input("x")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Output("y")
            .ParamType(REQUIRED)
            .DataType({ge::DT_FLOAT})
            .Format({ge::FORMAT_ND})
            .UnknownShapeFormat({ge::FORMAT_ND});

        this->Attr("dim").AttrType(OPTIONAL).Int(1);

        this->SetInferShape(ge::InferShape).SetInferDataType(ge::InferDataType);
        this->AICore().SetTiling(optiling::TilingFunc);
        this->AICore().AddConfig("ascend910b");
    }
};

OP_ADD(CumsumCustom);
} // namespace ops

kernel_src="""

#include "kernel_operator.h"

class KernelCumsumCustom {
public:
    __aicore__ inline KernelCumsumCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y,
                                uint32_t rows, uint32_t cols,
                                uint32_t inner, uint32_t tileRows,
                                uint32_t isLastAxis)
    {
        rows_ = rows;
        cols_ = cols;
        inner_ = inner;
        tileRows_ = tileRows;
        isLastAxis_ = isLastAxis;

        totalElems_ = rows_ * cols_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, totalElems_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, totalElems_);

        // Buffers in UB
        // srcLocal: tileRows * inner
        // dstLocal: tileRows * inner
        // lastRowLocal: inner
        pipe_.InitBuffer(inQueue_, 1, tileRows_ * inner_ * sizeof(float));
        pipe_.InitBuffer(outQueue_, 1, tileRows_ * inner_ * sizeof(float));
        pipe_.InitBuffer(lastRowQueue_, 1, inner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (rows_ == 0 || cols_ == 0) return;

        const uint32_t tilesPerCoreStride = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t coreIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
        const uint32_t totalTiles = (rows_ + tileRows_ - 1u) / tileRows_;

        for (uint32_t tileIdx = coreIdx; tileIdx < totalTiles; tileIdx += tilesPerCoreStride) {
            uint32_t startRow = tileIdx * tileRows_;
            if (startRow >= rows_) break;
            uint32_t curRows = rows_ - startRow;
            if (curRows > tileRows_) curRows = tileRows_;

            CopyIn(startRow, curRows);
            Compute(curRows);
            CopyOut(startRow, curRows);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t startRow, uint32_t curRows)
    {
        AscendC::LocalTensor<float> srcLocal = inQueue_.AllocTensor<float>();
        // Zero-initialize to ensure padded tail does not affect results.
        AscendC::Duplicate<float>(srcLocal, 0.0f, curRows * inner_);

        // Copy each row: first 'cols_' elements filled from GM, padding remains zero.
        for (uint32_t r = 0; r < curRows; ++r) {
            const uint32_t gmOffset = (startRow + r) * cols_;
            const uint32_t lcOffset = r * inner_;
            AscendC::DataCopy(srcLocal[lcOffset], xGm_[gmOffset], cols_);
        }

        inQueue_.EnQue(srcLocal);
    }

    __aicore__ inline void Compute(uint32_t curRows)
    {
        AscendC::LocalTensor<float> srcLocal = inQueue_.DeQue<float>();
        AscendC::LocalTensor<float> dstLocal = outQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> lastRowLocal = lastRowQueue_.AllocTensor<float>();

        const AscendC::CumSumInfo info{curRows, inner_};

        if (isLastAxis_) {
            static constexpr AscendC::CumSumConfig cfg{true, false, false};
            AscendC::CumSum<float, cfg>(dstLocal, lastRowLocal, srcLocal, info);
        } else {
            static constexpr AscendC::CumSumConfig cfg{false, false, false};
            AscendC::CumSum<float, cfg>(dstLocal, lastRowLocal, srcLocal, info);
        }

        lastRowQueue_.FreeTensor(lastRowLocal);
        outQueue_.EnQue<float>(dstLocal);
        inQueue_.FreeTensor(srcLocal);
    }

    __aicore__ inline void CopyOut(uint32_t startRow, uint32_t curRows)
    {
        AscendC::LocalTensor<float> dstLocal = outQueue_.DeQue<float>();

        // Copy back only the actual 'cols_' elements per row, ignoring padded tail.
        for (uint32_t r = 0; r < curRows; ++r) {
            const uint32_t gmOffset = (startRow + r) * cols_;
            const uint32_t lcOffset = r * inner_;
            AscendC::DataCopy(yGm_[gmOffset], dstLocal[lcOffset], cols_);
        }

        outQueue_.FreeTensor(dstLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> lastRowQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t rows_{0};
    uint32_t cols_{0};
    uint32_t inner_{0};
    uint32_t tileRows_{1};
    uint32_t isLastAxis_{1};
    uint32_t totalElems_{0};
};

extern "C" __global__ __aicore__ void cumsum_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumCustom op;
    op.Init(x, y,
            tiling_data.rows, tiling_data.cols,
            tiling_data.inner, tiling_data.tileRows,
            tiling_data.isLastAxis);
    op.Process();
}
"""
