    const auto &shape2d = storage->GetStorageShape();

    uint32_t inner  = shape2d.GetDim(1);

    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    int32_t dim = 1;

        if (pdim != nullptr) dim = *pdim;

    uint32_t blockDim = outter;
    if (blockDim == 0) blockDim = 1;
    if (blockDim > 32u) blockDim = 32u;

    tiling.set_inner(inner);

kernel_src="""
#include "kernel_operator.h"

class KernelCumsumReverseCustom {
public:
    __aicore__ inline KernelCumsumReverseCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR y, uint32_t outter, uint32_t inner, int32_t dim)
    {
        outter_ = outter;
        inner_ = inner;
        dim_ = dim;

        xGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(x), static_cast<uint64_t>(outter_) * inner_);
        yGm_.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(y), static_cast<uint64_t>(outter_) * inner_);

        // Buffers: x row, calc row (reuse for prefix/delta/result), y row, lastRow (for CumSum API requirement)
        pipe_.InitBuffer(inQueueX_, 1, inner_ * sizeof(float));
        pipe_.InitBuffer(calcQueue_, 1, inner_ * sizeof(float));
        pipe_.InitBuffer(outQueueY_, 1, inner_ * sizeof(float));
        pipe_.InitBuffer(lastRowQueue_, 1, inner_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (outter_ == 0 || inner_ == 0) return;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Process per row; reverse-cumsum along last axis (dim == 1)
        for (uint32_t row = blockIdx; row < outter_; row += blockNum) {
            CopyInRow(row);
            ComputeRow();
            CopyOutRow(row);
        }
    }

private:
    __aicore__ inline void CopyInRow(uint32_t row)
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
        const uint64_t gmOffset = static_cast<uint64_t>(row) * inner_;
        AscendC::DataCopy(xLocal, xGm_[gmOffset], inner_);
        inQueueX_.EnQue(xLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        AscendC::LocalTensor<float> xLocal = inQueueX_.DeQue<float>();
        AscendC::LocalTensor<float> calcLocal = calcQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::LocalTensor<float> lastRowLocal = lastRowQueue_.AllocTensor<float>();

        // Step 1: prefix sum s = cumsum(x) along last axis
        static constexpr AscendC::CumSumConfig cfg{true, false, false};
        const AscendC::CumSumInfo infoRow{1, inner_};
        AscendC::CumSum<float, cfg>(calcLocal, lastRowLocal, xLocal, infoRow);

        // Step 2: prepare delta vector d
        // d[0] = T (total sum) = s[last]; d[i] = -x[i-1] for i in [1, inner_-1]
        // Copy T into yLocal[0] temporarily
        AscendC::DataCopy(yLocal[0], calcLocal[inner_ - 1], 1);
        // Copy x[0..inner_-2] -> calc[1..inner_-1]
        if (inner_ > 1) {
            AscendC::DataCopy(calcLocal[1], xLocal[0], inner_ - 1);
            AscendC::Muls(calcLocal[1], calcLocal[1], -1.0f, inner_ - 1);
        }
        // Set calc[0] = T
        AscendC::DataCopy(calcLocal[0], yLocal[0], 1);

        // Step 3: r = cumsum(d) along last axis, store directly to yLocal
        AscendC::CumSum<float, cfg>(yLocal, lastRowLocal, calcLocal, infoRow);

        lastRowQueue_.FreeTensor(lastRowLocal);
        calcQueue_.FreeTensor(calcLocal);
        outQueueY_.EnQue(yLocal);
        inQueueX_.FreeTensor(xLocal);
    }

    __aicore__ inline void CopyOutRow(uint32_t row)
    {
        AscendC::LocalTensor<float> yLocal = outQueueY_.DeQue<float>();
        const uint64_t gmOffset = static_cast<uint64_t>(row) * inner_;
        AscendC::DataCopy(yGm_[gmOffset], yLocal, inner_);
        outQueueY_.FreeTensor(yLocal);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, 1> inQueueX_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> calcQueue_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECOUT, 1> lastRowQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t outter_{0};
    uint32_t inner_{0};
    int32_t dim_{1};
};

extern "C" __global__ __aicore__ void cumsum_reverse_custom(GM_ADDR x, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelCumsumReverseCustom op;
    op.Init(x, y, tiling_data.outter, tiling_data.inner, tiling_data.dim);
    op.Process();
}
