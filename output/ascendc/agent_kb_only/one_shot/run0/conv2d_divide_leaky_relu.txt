kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConv2dDivideLeakyReluCustom {
public:
    __aicore__ inline KernelConv2dDivideLeakyReluCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y,
                                uint32_t n, uint32_t cIn, uint32_t cOut,
                                uint32_t h, uint32_t wIn, uint32_t k,
                                uint32_t outH, uint32_t outW,
                                uint32_t hasBias,
                                float divisor, float negativeSlope)
    {
        n_ = n; cIn_ = cIn; cOut_ = cOut; h_ = h; w_ = wIn; k_ = k;
        outH_ = outH; outW_ = outW;
        hasBias_ = (hasBias != 0);
        invDiv_ = (divisor != 0.0f) ? (1.0f / divisor) : 0.5f; // expected 0.5 for divisor 2.0
        negCorr_ = negativeSlope - 1.0f;

        xLen_ = static_cast<uint64_t>(n_) * cIn_ * h_ * w_;
        wLen_ = static_cast<uint64_t>(cOut_) * cIn_ * k_ * k_;
        bLen_ = static_cast<uint64_t>(cOut_);
        yLen_ = static_cast<uint64_t>(n_) * cOut_ * outH_ * outW_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xLen_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wLen_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, bLen_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, yLen_);

        // Queues: scalar DMA buffers and temporary buffers
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(inQueueW_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(inQueueB_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (yLen_ == 0) { return; }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        const uint64_t total = static_cast<uint64_t>(n_) * cOut_ * outH_ * outW_;

        for (uint64_t idx = blockIdx; idx < total; idx += blockNum) {
            uint32_t oc = static_cast<uint32_t>(idx % cOut_);
            uint64_t t = idx / cOut_;
            uint32_t ow = static_cast<uint32_t>(t % outW_);
            t /= outW_;
            uint32_t oh = static_cast<uint32_t>(t % outH_);
            uint32_t n = static_cast<uint32_t>(t / outH_);

            // Allocate local for accumulation (length 1)
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            yLocal.SetValue(0, 0.0f);

            // Accumulate convolution sum
            for (uint32_t ci = 0; ci < cIn_; ++ci) {
                for (uint32_t kh = 0; kh < k_; ++kh) {
                    const uint32_t in_h = oh + kh;
                    for (uint32_t kw = 0; kw < k_; ++kw) {
                        const uint32_t in_w = ow + kw;
                        // Input offset: (((n * C_in + ci) * H + in_h) * W + in_w)
                        uint64_t inOff = (((static_cast<uint64_t>(n) * cIn_ + ci) * h_ + in_h) * w_ + in_w);
                        // Weight offset: (((oc * C_in + ci) * K + kh) * K + kw)
                        uint64_t wOff = (((static_cast<uint64_t>(oc) * cIn_ + ci) * k_ + kh) * k_ + kw);

                        AscendC::LocalTensor<float> xLocal = inQueueX_.AllocTensor<float>();
                        AscendC::LocalTensor<float> wLocal = inQueueW_.AllocTensor<float>();
                        AscendC::LocalTensor<float> tmpMul = tmpQueue_.AllocTensor<float>();

                        AscendC::DataCopy(xLocal, xGm_[inOff], 1);
                        AscendC::DataCopy(wLocal, wGm_[wOff], 1);
                        AscendC::Mul(tmpMul, xLocal, wLocal, 1);
                        AscendC::Add(yLocal, yLocal, tmpMul, 1);

                        tmpQueue_.FreeTensor(tmpMul);
                        inQueueW_.FreeTensor(wLocal);
                        inQueueX_.FreeTensor(xLocal);
                    }
                }
            }

            // Add bias if present
            if (hasBias_) {
                AscendC::LocalTensor<float> bLocal = inQueueB_.AllocTensor<float>();
                AscendC::DataCopy(bLocal, bGm_[oc], 1);
                AscendC::Add(yLocal, yLocal, bLocal, 1);
                inQueueB_.FreeTensor(bLocal);
            }

            // Divide by divisor (multiply by invDiv_)
            AscendC::Muls(yLocal, yLocal, invDiv_, 1);

            // LeakyReLU: y = y + (negativeSlope - 1) * min(y, 0)
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>();
            AscendC::Mins(tLocal, yLocal, 0.0f, 1);
            AscendC::Muls(tLocal, tLocal, negCorr_, 1);
            AscendC::Add(yLocal, yLocal, tLocal, 1);
            tmpQueue_.FreeTensor(tLocal);

            // Write back
            uint64_t yOff = (((static_cast<uint64_t>(n) * cOut_ + oc) * outH_ + oh) * outW_ + ow);
            outQueueY_.EnQue<float>(yLocal);
            AscendC::LocalTensor<float> yDeq = outQueueY_.DeQue<float>();
            AscendC::DataCopy(yGm_[yOff], yDeq, 1);
            outQueueY_.FreeTensor(yDeq);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueW_;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueB_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> bGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t n_{0}, cIn_{0}, cOut_{0}, h_{0}, w_{0}, k_{0};
    uint32_t outH_{0}, outW_{0};
    bool hasBias_{false};
    float invDiv_{0.5f};
    float negCorr_{-0.99f};

    uint64_t xLen_{0}, wLen_{0}, bLen_{0}, yLen_{0};
};

extern "C" __global__ __aicore__ void conv2d_divide_leaky_relu_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tilingData, tiling);
    KernelConv2dDivideLeakyReluCustom op;
    op.Init(x, weight, bias, y,
            tilingData.n, tilingData.cIn, tilingData.cOut,
            tilingData.h, tilingData.w, tilingData.k,
            tilingData.outH, tilingData.outW,
            tilingData.hasBias,
            tilingData.divisor, tilingData.negativeSlope);
    if (TILING_KEY_IS(1)) {
        op.Process();
    }
}
