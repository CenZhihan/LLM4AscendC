namespace {
static inline uint32_t CeilDiv(uint32_t a, uint32_t b) {
    return (a + b - 1u) / b;
}
static inline uint32_t AlignDown(uint32_t x, uint32_t a) {
    return (x / a) * a;
}
}

namespace optiling {

// Simple UB budget model
static constexpr uint32_t UB_BUDGET_BYTES = 128 * 1024; // per-core conservative budget
static constexpr uint32_t MAX_BLOCKS = 24;
static constexpr uint32_t MIN_BLOCKS = 8;

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvDepthwiseSeparable2dCustomTilingData tiling;

    // Shapes: x [N, Cin, H, W], w_depthwise [Cin, 1, 3, 3], w_pointwise [Cout, Cin, 1, 1]
    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* wdw_shape = context->GetInputShape(1);
    const gert::Shape* wpw_shape = context->GetInputShape(2);

    // Access origin dims (NCHW)
    const int64_t N = x_shape->GetOriginShape().GetDim(0);
    const int64_t Cin = x_shape->GetOriginShape().GetDim(1);
    const int64_t H = x_shape->GetOriginShape().GetDim(2);
    const int64_t W = x_shape->GetOriginShape().GetDim(3);
    const int64_t Cout = wpw_shape->GetOriginShape().GetDim(0);

    // Choose tile sizes heuristically
    uint32_t tileH = 8;   // rows per tile
    uint32_t tileK = 16;  // output channels per tile

    // Try to keep y tile+buffers within UB budget
    // Rough bytes: XBuf ~ (tileH+2)*W*4 + DLocal ~ tileH*W*4 + tLocal ~ tileH*W*4 + yLocal ~ tileH*W*tileK*4 + WpwTile ~ tileK*Cin*4
    // Solve for feasible tileK given W and tileH
    uint32_t W_u = static_cast<uint32_t>(W);
    uint32_t Cin_u = static_cast<uint32_t>(Cin);
    uint32_t baseBytes = (tileH + 2) * W_u * 4 + tileH * W_u * 4 + tileH * W_u * 4;
    uint32_t maxK_by_ub = 1;
    if (W_u > 0) {
        if (UB_BUDGET_BYTES > baseBytes) {
            maxK_by_ub = (UB_BUDGET_BYTES - baseBytes) / (tileH * W_u * 4 + Cin_u * 4);
            if (maxK_by_ub == 0) maxK_by_ub = 1;
            if (maxK_by_ub > 64) maxK_by_ub = 64;
        }
    }
    if (tileK > maxK_by_ub) tileK = maxK_by_ub;

    if (tileK == 0) tileK = 8;

    // Boundaries
    if (tileH < 4) tileH = 4;
    if (tileH > static_cast<uint32_t>(H)) tileH = static_cast<uint32_t>(H);
    if (tileK > static_cast<uint32_t>(Cout)) tileK = static_cast<uint32_t>(Cout);

    // Blocks: parallelize over (N * ceil(H/tileH) * ceil(Cout/tileK))
    const uint32_t tilesH = CeilDiv(static_cast<uint32_t>(H), tileH);
    const uint32_t tilesK = CeilDiv(static_cast<uint32_t>(Cout), tileK);
    uint32_t blockDim = static_cast<uint32_t>(N) * tilesH * tilesK;
    if (blockDim < MIN_BLOCKS) blockDim = MIN_BLOCKS;
    if (blockDim > MAX_BLOCKS) blockDim = MAX_BLOCKS;
    context->SetBlockDim(blockDim);

    tiling.set_N(static_cast<uint32_t>(N));
    tiling.set_Cin(static_cast<uint32_t>(Cin));
    tiling.set_Cout(static_cast<uint32_t>(Cout));
    tiling.set_H(static_cast<uint32_t>(H));
    tiling.set_W(static_cast<uint32_t>(W));
    tiling.set_tileH(tileH);
    tiling.set_tileK(tileK);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(), context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvDepthwiseSeparable2dCustom {
public:
    __aicore__ inline KernelConvDepthwiseSeparable2dCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR wdw, GM_ADDR wpw, GM_ADDR y,
                                uint32_t N, uint32_t Cin, uint32_t Cout,
                                uint32_t H, uint32_t W,
                                uint32_t tileH, uint32_t tileK)
    {
        N_ = N; Cin_ = Cin; Cout_ = Cout; H_ = H; W_ = W;
        tileH_ = tileH; tileK_ = tileK;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * Cin_ * H_ * W_);
        wdwGm_.SetGlobalBuffer((__gm__ float*)wdw, Cin_ * 3 * 3);
        wpwGm_.SetGlobalBuffer((__gm__ float*)wpw, Cout_ * Cin_);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * Cout_ * H_ * W_);

        // Buffers:
        // XBuf: (tileH+2) * W
        // DLocal: tileH * W
        // tLocal: tileH * W
        // yLocal: tileH * W * tileK
        // wPwTile: tileK * Cin
        pipe_.InitBuffer(inQueueX_, BUFFER_NUM, (tileH_ + 2) * W_ * sizeof(float));
        pipe_.InitBuffer(inQueueWd_, BUFFER_NUM, 9 * sizeof(float));
        pipe_.InitBuffer(inQueueWp_, BUFFER_NUM, tileK_ * Cin_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileH_ * W_ * sizeof(float) * 2);
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileH_ * W_ * tileK_ * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        const uint32_t tilesH = (H_ + tileH_ - 1) / tileH_;
        const uint32_t tilesK = (Cout_ + tileK_ - 1) / tileK_;
        const uint32_t totalWork = N_ * tilesH * tilesK;

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        for (uint32_t work = blockIdx; work < totalWork; work += blockNum) {
            // decode work id -> (n, ht, kt)
            uint32_t tmp = work;
            const uint32_t kt = tmp % tilesK; tmp /= tilesK;
            const uint32_t ht = tmp % tilesH; tmp /= tilesH;
            const uint32_t n = tmp;

            const uint32_t h0 = ht * tileH_;
            const uint32_t actualH = (h0 + tileH_ <= H_) ? tileH_ : (H_ - h0);
            const uint32_t k0 = kt * tileK_;
            const uint32_t actualK = (k0 + tileK_ <= Cout_) ? tileK_ : (Cout_ - k0);
            if (actualH == 0 || actualK == 0) continue;

            // Load pointwise weight tile [actualK, Cin]
            AscendC::LocalTensor<float> wPwLocal = inQueueWp_.AllocTensor<float>();
            const uint32_t wpwOffset = (k0 * Cin_);
            AscendC::DataCopy(wPwLocal, wpwGm_[wpwOffset], actualK * Cin_);
            inQueueWp_.EnQue(wPwLocal);

            // Allocate yLocal and zero-init per output channel block
            AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
            const uint32_t tileHW = actualH * W_;
            for (uint32_t kk = 0; kk < actualK; ++kk) {
                AscendC::LocalTensor<float> ySub = yLocal[kk * tileH_ * W_];
                AscendC::Muls(ySub, ySub, 0.0f, tileHW);
            }

            // Temporary buffers for DLocal and tLocal
            AscendC::LocalTensor<float> dLocal = tmpQueue_.AllocTensor<float>(); // size tileH_*W_
            dLocal.SetSize(tileH_ * W_);
            AscendC::LocalTensor<float> tLocal = tmpQueue_.AllocTensor<float>(); // size tileH_*W_
            tLocal.SetSize(tileH_ * W_);

            // For each input channel (depthwise)
            for (uint32_t c = 0; c < Cin_; ++c) {
                // Load depthwise kernel 3x3 for channel c
                AscendC::LocalTensor<float> wDwLocal = inQueueWd_.AllocTensor<float>();
                AscendC::DataCopy(wDwLocal, wdwGm_[c * 9], 9);
                inQueueWd_.EnQue(wDwLocal);

                // Build XBuf for (actualH+2) rows
                AscendC::LocalTensor<float> xBuf = inQueueX_.AllocTensor<float>();
                const uint32_t rowsToLoad = actualH + 2;
                for (uint32_t r = 0; r < rowsToLoad; ++r) {
                    int32_t gr = static_cast<int32_t>(h0) + static_cast<int32_t>(r) - 1;
                    AscendC::LocalTensor<float> rowLocal = xBuf[r * W_];
                    if (gr < 0 || gr >= static_cast<int32_t>(H_)) {
                        // zero row
                        AscendC::Muls(rowLocal, rowLocal, 0.0f, W_);
                    } else {
                        const uint32_t xOffset = (((n * Cin_ + c) * H_) + static_cast<uint32_t>(gr)) * W_;
                        AscendC::DataCopy(rowLocal, xGm_[xOffset], W_);
                    }
                }
                inQueueX_.EnQue(xBuf);

                // Depthwise conv 3x3 with padding=1, stride=1 for tile rows
                {
                    AscendC::LocalTensor<float> wdw = inQueueWd_.DeQue<float>();
                    AscendC::LocalTensor<float> xPatch = inQueueX_.DeQue<float>();
                    for (uint32_t rh = 0; rh < actualH; ++rh) {
                        for (uint32_t col = 0; col < W_; ++col) {
                            float acc = 0.0f;
                            // 3x3 kernel
                            for (int ky = 0; ky < 3; ++ky) {
                                const uint32_t inRow = rh + ky; // already includes top halo
                                for (int kx = 0; kx < 3; ++kx) {
                                    int inCol = static_cast<int>(col) + kx - 1;
                                    if (inCol >= 0 && inCol < static_cast<int>(W_)) {
                                        float val = xPatch.GetValue(inRow * W_ + static_cast<uint32_t>(inCol));
                                        float wv = wdw.GetValue(ky * 3 + kx);
                                        acc += val * wv;
                                    }
                                }
                            }
                            dLocal.SetValue(rh * W_ + col, acc);
                        }
                    }
                    inQueueWd_.FreeTensor(wdw);
                    inQueueX_.FreeTensor(xPatch);
                }

                // Accumulate into yLocal over output channel tile using pointwise weights
                AscendC::LocalTensor<float> wPwT = inQueueWp_.DeQue<float>();
                for (uint32_t kk = 0; kk < actualK; ++kk) {
                    float wval = wPwT.GetValue(kk * Cin_ + c);
                    AscendC::LocalTensor<float> ySub = yLocal[kk * tileH_ * W_];
                    AscendC::Muls(tLocal, dLocal, wval, tileHW);
                    AscendC::Add(ySub, ySub, tLocal, tileHW);
                }
                inQueueWp_.EnQue(wPwT); // keep in queue for next c
            }

            // Done with accumulations, write back yLocal tile for each output channel in this tile
            for (uint32_t kk = 0; kk < actualK; ++kk) {
                AscendC::LocalTensor<float> ySub = yLocal[kk * tileH_ * W_];
                const uint32_t outCh = k0 + kk;
                const uint32_t yBase = ((n * Cout_ + outCh) * H_ + h0) * W_;
                AscendC::DataCopy(yGm_[yBase], ySub, tileHW);
            }

            // free locals
            tmpQueue_.FreeTensor(tLocal);
            tmpQueue_.FreeTensor(dLocal);
            outQueueY_.FreeTensor(yLocal);
            AscendC::LocalTensor<float> wPwFin = inQueueWp_.DeQue<float>();
            inQueueWp_.FreeTensor(wPwFin);
        }
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueX_;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueWd_;
    AscendC::TQue<AscendC::QuePosition::VECIN, BUFFER_NUM> inQueueWp_;
    AscendC::TQue<AscendC::QuePosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueueY_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wdwGm_;
    AscendC::GlobalTensor<float> wpwGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0}, H_{0}, W_{0};
    uint32_t tileH_{0}, tileK_{0};
};

extern "C" __global__ __aicore__ void conv_depthwise_separable_2d_custom(GM_ADDR x, GM_ADDR w_depthwise, GM_ADDR w_pointwise, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(t, tiling);
    KernelConvDepthwiseSeparable2dCustom op;
    op.Init(x, w_depthwise, w_pointwise, y,
            t.N, t.Cin, t.Cout, t.H, t.W, t.tileH, t.tileK);
    op.Process();
}
