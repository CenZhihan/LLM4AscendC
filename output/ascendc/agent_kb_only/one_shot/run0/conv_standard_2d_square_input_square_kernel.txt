namespace {
static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (x / a) * a;
}
static inline uint32_t AlignUp(uint32_t x, uint32_t a)
{
    return ((x + a - 1u) / a) * a;
}
static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}
}

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvStandard2dSquareInputSquareKernelCustomTilingData tiling;

    const gert::Shape* x_shape = context->GetInputShape(0);
    const gert::Shape* w_shape = context->GetInputShape(1);

    // Expect ND: [N, Cin, Hin, Win] and [Cout, Cin, Kh, Kw]
    const uint32_t N   = static_cast<uint32_t>(x_shape->GetDim(0));
    const uint32_t Cin = static_cast<uint32_t>(x_shape->GetDim(1));
    const uint32_t Hin = static_cast<uint32_t>(x_shape->GetDim(2));
    const uint32_t Win = static_cast<uint32_t>(x_shape->GetDim(3));

    const uint32_t Cout = static_cast<uint32_t>(w_shape->GetDim(0));
    const uint32_t Kh   = static_cast<uint32_t>(w_shape->GetDim(2));
    const uint32_t Kw   = static_cast<uint32_t>(w_shape->GetDim(3));

    // Stride=1, padding=0, dilation=1
    const uint32_t Hout = Hin - Kh + 1;
    const uint32_t Wout = Win - Kw + 1;

    // UB budget and tile along Wout
    static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024; // conservative per-core share
    static constexpr uint32_t ALIGN_ELEMS = 128;

    // We use three local vectors of length ~tileW (yLocal, tmpLocal, rowLocal-(+2)).
    // Rough bytes ~= 4 * (3*tileW + 2)
    uint32_t maxTileW = (UB_BUDGET_BYTES / 4 > 2) ? ((UB_BUDGET_BYTES / 4) - 2) / 3 : 64u;
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);

    uint32_t tileW = (Wout < maxTileW) ? AlignDown(Wout, ALIGN_ELEMS) : maxTileW;
    if (tileW == 0) tileW = (Wout > 0) ? Wout : ALIGN_ELEMS;

    // blockDim based on cout
    uint32_t blockDim = Cout;
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_n(N);
    tiling.set_cin(Cin);
    tiling.set_cout(Cout);
    tiling.set_hin(Hin);
    tiling.set_win(Win);
    tiling.set_hout(Hout);
    tiling.set_wout(Wout);
    tiling.set_tileW(tileW);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;
    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"

constexpr int32_t BUFFER_NUM = 1;

class KernelConvStandard2dSquareInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvStandard2dSquareInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y,
                                uint32_t n, uint32_t cin, uint32_t cout,
                                uint32_t hin, uint32_t win,
                                uint32_t hout, uint32_t wout,
                                uint32_t tileW)
    {
        N_ = n; Cin_ = cin; Cout_ = cout;
        Hin_ = hin; Win_ = win; Hout_ = hout; Wout_ = wout;
        tileW_ = tileW;

        xGm_.SetGlobalBuffer((__gm__ float*)x, N_ * Cin_ * Hin_ * Win_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, Cout_ * Cin_ * 3 * 3);
        yGm_.SetGlobalBuffer((__gm__ float*)y, N_ * Cout_ * Hout_ * Wout_);

        pipe_.InitBuffer(inQueueRow_, BUFFER_NUM, (tileW_ + 2) * sizeof(float));
        pipe_.InitBuffer(outQueueY_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(tmpQueue_, BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(wScalarQueue_, BUFFER_NUM, sizeof(float)); // for 1-element scalar load
    }

    __aicore__ inline void Process()
    {
        if (Hout_ == 0 || Wout_ == 0 || Cout_ == 0 || Cin_ == 0 || N_ == 0) {
            return;
        }

        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Each core processes a subset of output channels
        for (uint32_t n = 0; n < N_; ++n) {
            for (uint32_t co = blockIdx; co < Cout_; co += blockNum) {
                for (uint32_t oh = 0; oh < Hout_; ++oh) {
                    for (uint32_t ow0 = 0; ow0 < Wout_; ow0 += tileW_) {
                        const uint32_t L = (ow0 + tileW_ <= Wout_) ? tileW_ : (Wout_ - ow0);
                        ComputeTile(n, co, oh, ow0, L);
                    }
                }
            }
        }
    }

private:
    __aicore__ inline float LoadWeightScalar(uint32_t co, uint32_t ci, uint32_t ky, uint32_t kx)
    {
        const uint32_t wOffset = (((co * Cin_ + ci) * 3 + ky) * 3 + kx);
        AscendC::LocalTensor<float> wLoc = wScalarQueue_.AllocTensor<float>();
        AscendC::DataCopy(wLoc, wGm_[wOffset], 1);
        wScalarQueue_.EnQue(wLoc);
        AscendC::LocalTensor<float> wBuf = wScalarQueue_.DeQue<float>();
        // Read the scalar from local tensor (first element)
        float wVal = wBuf.GetValue(0);
        wScalarQueue_.FreeTensor(wBuf);
        return wVal;
    }

    __aicore__ inline void ComputeTile(uint32_t n, uint32_t co, uint32_t oh, uint32_t ow0, uint32_t L)
    {
        // Allocate output local tile and zero it
        AscendC::LocalTensor<float> yLocal = outQueueY_.AllocTensor<float>();
        AscendC::Duplicate(yLocal, 0.0f, L);

        // Accumulate over Cin and 3x3 kernel
        for (uint32_t ci = 0; ci < Cin_; ++ci) {
            for (uint32_t ky = 0; ky < 3; ++ky) {
                // Load input row segment of length L + 2 (for 3x3 kernel over width)
                const uint32_t in_y = oh + ky;
                const uint32_t in_x0 = ow0;
                const uint32_t xRowOffset = ((((n * Cin_ + ci) * Hin_ + in_y) * Win_) + in_x0);
                AscendC::LocalTensor<float> rowLocal = inQueueRow_.AllocTensor<float>();
                AscendC::DataCopy(rowLocal, xGm_[xRowOffset], L + 2);
                inQueueRow_.EnQue(rowLocal);

                AscendC::LocalTensor<float> inRow = inQueueRow_.DeQue<float>();

                // For kx = 0..2, multiply inRow[kx : kx+L-1] by weight and add to yLocal
                for (uint32_t kx = 0; kx < 3; ++kx) {
                    const float wVal = LoadWeightScalar(co, ci, ky, kx);
                    AscendC::LocalTensor<float> tmpLocal = tmpQueue_.AllocTensor<float>();
                    // Multiply vector slice by scalar weight
                    AscendC::Muls(tmpLocal, inRow[kx], wVal, L);
                    // Accumulate to output
                    AscendC::Add(yLocal, yLocal, tmpLocal, L);
                    tmpQueue_.FreeTensor(tmpLocal);
                }

                inQueueRow_.FreeTensor(inRow);
            }
        }

        // Store output tile
        outQueueY_.EnQue<float>(yLocal);
        const uint32_t yOffset = ((((n * Cout_ + co) * Hout_ + oh) * Wout_) + ow0);
        AscendC::LocalTensor<float> yBuf = outQueueY_.DeQue<float>();
        AscendC::DataCopy(yGm_[yOffset], yBuf, L);
        outQueueY_.FreeTensor(yBuf);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQueueRow_;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQueueY_;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQueue_;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> wScalarQueue_;

    AscendC::GlobalTensor<float> xGm_;
    AscendC::GlobalTensor<float> wGm_;
    AscendC::GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t Hin_{0}, Win_{0}, Hout_{0}, Wout_{0};
    uint32_t tileW_{0};
};

extern "C" __global__ __aicore__ void conv_standard_2d_square_input_square_kernel_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConvStandard2dSquareInputSquareKernelCustom op;
    op.Init(x, weight, y,
            tiling_data.n, tiling_data.cin, tiling_data.cout,
            tiling_data.hin, tiling_data.win,
            tiling_data.hout, tiling_data.wout,
            tiling_data.tileW);
    op.Process();
}
