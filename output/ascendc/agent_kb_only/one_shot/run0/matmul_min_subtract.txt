namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Input shapes:
    // x: [M, K]
    // w: [N, K]  (will be used as transposed in kernel)
    // b: [N]
    // c: scalar
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    const gert::Shape *bShape = context->GetInputShape(2);

    const int32_t xDims = xShape->GetOriginShape().GetDimNum();
    const int32_t wDims = wShape->GetOriginShape().GetDimNum();

    if (xDims < 2 || wDims < 2) {
        return ge::GRAPH_FAILED;
    }

    const int64_t M = xShape->GetOriginShape().GetDim(xDims - 2);
    const int64_t Kx = xShape->GetOriginShape().GetDim(xDims - 1);
    const int64_t N = wShape->GetOriginShape().GetDim(wDims - 2);
    const int64_t Kw = wShape->GetOriginShape().GetDim(wDims - 1);

    if (Kx != Kw) {
        return ge::GRAPH_FAILED;
    }

    // Infer output shape [M, N]
    gert::Shape *yShape = context->GetOutputShape(0);
    std::vector<int64_t> outDims;
    for (int i = 0; i < xDims - 2; ++i) {
        outDims.push_back(xShape->GetOriginShape().GetDim(i)); // keep leading dims if any
    }
    outDims.push_back(M);
    outDims.push_back(N);
    yShape->SetDimNum(static_cast<int>(outDims.size()));
    for (size_t i = 0; i < outDims.size(); ++i) {
        yShape->SetDim(i, outDims[i]);
    }

    // Matmul single-core tiling
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling mt(ascendcPlatform);

    // A: half ND, B: half ND (we will set transpose flag in kernel), C: float ND, Bias: float
    mt.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    mt.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT16);
    mt.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    mt.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    mt.SetShape(static_cast<int64_t>(M), static_cast<int64_t>(N), static_cast<int64_t>(Kx));
    mt.SetOrgShape(static_cast<int64_t>(M), static_cast<int64_t>(N), static_cast<int64_t>(Kx));
    mt.SetBias(true);
    mt.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling cubeTiling {};
    int ret = mt.GetTiling(cubeTiling);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Choose a conservative blockDim (single-core default); runtime grid-stride in kernel for post ops
    context->SetBlockDim(1);

    // Save raw TCubeTiling bytes to tiling buffer
    auto *raw = context->GetRawTilingData();
    const size_t bytes = sizeof(optiling::TCubeTiling);
    if (raw->GetCapacity() < bytes) {
        return ge::GRAPH_FAILED;
    }
    std::memcpy(raw->GetData(), &cubeTiling, bytes);
    raw->SetDataSize(bytes);

    // Workspace for matmul; set to 0 for default
    size_t *wk = context->GetWorkspaceSizes(1);
    wk[0] = 0;

    return ge::GRAPH_SUCCESS;
}

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using AscendC::GlobalTensor;

constexpr int32_t VEC_BUFFER_NUM = 2;

extern "C" __global__ __aicore__ void matmul_min_subtract_custom(
    GM_ADDR xGM, GM_ADDR wGM, GM_ADDR bGM, GM_ADDR cGM, GM_ADDR yGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // 1) Load TCubeTiling from GM (raw bytes from host tiling)
    TCubeTiling tiling;
    {
        auto tgm = reinterpret_cast<__gm__ uint32_t*>(tilingGM);
        auto tl = reinterpret_cast<uint32_t*>(&tiling);
        for (int i = 0; i < static_cast<int>(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tgm, ++tl) {
            *tl = *tgm;
        }
    }

    // 2) Initialize GM tensors for Matmul
    // Shapes: A(M,K) = x, B(N,K) = w (used with transpose), Bias(N), C(M,N) = y
    GlobalTensor<half> aGlobal;
    GlobalTensor<half> bGlobal;
    GlobalTensor<float> cGlobal;
    GlobalTensor<float> biasGlobal;

    // Compute total element sizes based on tiling info
    // For NORMAL layout, we can derive from M, N, K stored in tiling.singleM/N/K
    int32_t M = static_cast<int32_t>(tiling.singleM);
    int32_t N = static_cast<int32_t>(tiling.singleN);
    int32_t K = static_cast<int32_t>(tiling.singleK);

    int32_t sizeA = M * K;
    int32_t sizeB = N * K;
    int32_t sizeBias = N;
    int32_t sizeC = M * N;

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(xGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ half*>(wGM), sizeB);
    biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bGM), sizeBias);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(yGM), sizeC);

    // 3) Run Matmul (x @ w^T + b)
    matmul::Matmul<
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, false, LayoutMode::Normal>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, half, true,  LayoutMode::Normal>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::Normal>,
        matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float>> mm;

    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
    mm.Init(&tiling);
    mm.SetTensorA(aGlobal[0], false);
    mm.SetTensorB(bGlobal[0], true);
    mm.SetWorkspace(workspaceGM, 0);
    mm.SetBias(biasGlobal[0]);
    mm.SetTensorC(cGlobal[0]);   // write result into yGM
    mm.IterateNBatch(1, 1, 1, false);

    // 4) Post-processing: y = min(y, cap) - cap
    // Read scalar c
    float capVal = *reinterpret_cast<__gm__ float*>(cGM);

    const uint32_t totalLen = static_cast<uint32_t>(sizeC);
    if (totalLen == 0) {
        return;
    }

    // Choose a reasonable vector tile
    uint32_t tileLen = 16384; // elements
    if (tileLen > totalLen) tileLen = totalLen;

    // Queues for vector processing
    AscendC::TQue<AscendC::TPosition::VECIN, VEC_BUFFER_NUM> inQueueY;
    AscendC::TQue<AscendC::TPosition::VECOUT, VEC_BUFFER_NUM> outQueueY;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> tmpQueue;

    pipe.InitBuffer(inQueueY, VEC_BUFFER_NUM, tileLen * sizeof(float));
    pipe.InitBuffer(outQueueY, VEC_BUFFER_NUM, tileLen * sizeof(float));
    pipe.InitBuffer(tmpQueue, 1, tileLen * sizeof(float));

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
    const uint32_t strideElems = blockNum * tileLen;

    uint32_t base = blockIdx * tileLen;
    if (base >= totalLen) {
        return;
    }

    auto CopyIn = [&](uint32_t off, uint32_t len) {
        auto yLocal = inQueueY.AllocTensor<float>();
        AscendC::DataCopy(yLocal, cGlobal[off], len);
        inQueueY.EnQue(yLocal);
    };

    auto Compute = [&](uint32_t len) {
        auto yLocal = inQueueY.DeQue<float>();
        auto tmpLocal = tmpQueue.AllocTensor<float>();
        auto outLocal = outQueueY.AllocTensor<float>();

        AscendC::Mins(tmpLocal, yLocal, capVal, len);     // tmp = min(y, cap)
        AscendC::Adds(outLocal, tmpLocal, -capVal, len);  // out = tmp - cap

        tmpQueue.FreeTensor(tmpLocal);
        outQueueY.EnQue(outLocal);
        inQueueY.FreeTensor(yLocal);
    };

    auto CopyOut = [&](uint32_t off, uint32_t len) {
        auto outLocal = outQueueY.DeQue<float>();
        AscendC::DataCopy(cGlobal[off], outLocal, len);
        outQueueY.FreeTensor(outLocal);
    };

    // Prefetch first tile
    uint32_t curLen = totalLen - base;
    if (curLen > tileLen) curLen = tileLen;
    CopyIn(base, curLen);

    while (true) {
        const uint32_t nextBase = base + strideElems;
        uint32_t nextLen = 0;
        if (nextBase < totalLen) {
            nextLen = totalLen - nextBase;
            if (nextLen > tileLen) nextLen = tileLen;
            CopyIn(nextBase, nextLen);
        }

        Compute(curLen);
        CopyOut(base, curLen);

        if (nextLen == 0) break;
        base = nextBase;
        curLen = nextLen;
    }
}

python_bind_src="""

#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_contiguous_dtype(const at::Tensor& t, at::ScalarType dt) {
    if (t.scalar_type() == dt && t.is_contiguous()) return t;
    return t.to(dt).contiguous();
}

at::Tensor matmul_min_subtract_impl_npu(const at::Tensor& x,
                                        const at::Tensor& w,
                                        const at::Tensor& b,
                                        const at::Tensor& c)
{
    // Contract: x:[M,K] half, w:[N,K] half, b:[N] float, c:[1] float -> y:[M,N] float
    at::Tensor x_half = to_contiguous_dtype(x, at::kHalf);
    at::Tensor w_half = to_contiguous_dtype(w, at::kHalf);
    at::Tensor b_f32  = to_contiguous_dtype(b, at::kFloat);
    at::Tensor c_f32  = to_contiguous_dtype(c, at::kFloat);

    // Infer output shape
    TORCH_CHECK(x_half.dim() >= 2 && w_half.dim() >= 2, "x and w must be 2D+");
    auto M = x_half.size(-2);
    auto K = x_half.size(-1);
    TORCH_CHECK(w_half.size(-1) == K, "K mismatch between x and w");
    auto N = w_half.size(-2);
    std::vector<int64_t> out_sizes;
    for (int i = 0; i < x_half.dim() - 2; ++i) out_sizes.push_back(x_half.size(i));
    out_sizes.push_back(M);
    out_sizes.push_back(N);

    at::Tensor y = at::empty(out_sizes, x_half.options().dtype(at::kFloat));

    EXEC_NPU_CMD(aclnnMatmulMinSubtractCustom, x_half, w_half, b_f32, c_f32, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_min_subtract_custom", &matmul_min_subtract_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_min_subtract_custom", &matmul_min_subtract_impl_npu,
          "Fused Matmul + Min + Subtract (MatmulMinSubtractCustom on NPU)");
}
