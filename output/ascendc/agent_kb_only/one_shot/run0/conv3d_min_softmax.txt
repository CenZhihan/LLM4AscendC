#include "conv3d_min_softmax_custom_tiling.h"

namespace {

static inline uint64_t MulSafe(uint64_t a, uint64_t b) {
    if (a == 0 || b == 0) return 0;
    return a * b;
}

static inline uint32_t ClampU32(uint64_t v) {
    return v > 0xFFFFFFFFull ? 0xFFFFFFFFu : static_cast<uint32_t>(v);
}

} // anonymous namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    Conv3dMinSoftmaxCustomTilingData tiling;

    // Input is NHWC (prepared in python binding), shape rank could be 4 or unknown.
    const gert::Shape *inShape = context->GetInputShape(0);
    const ge::Shape &orig = inShape->GetOriginShape();

    uint64_t N = 1, H = 1, W = 1, C = 1;
    if (orig.GetDimNum() >= 4) {
        N = static_cast<uint64_t>(orig.GetDim(0));
        H = static_cast<uint64_t>(orig.GetDim(1));
        W = static_cast<uint64_t>(orig.GetDim(2));
        C = static_cast<uint64_t>(orig.GetDim(3));
    } else {
        // Fallback to flat 2D: [m, n]
        uint64_t total = static_cast<uint64_t>(inShape->GetShapeSize());
        C = (orig.GetDimNum() >= 1) ? static_cast<uint64_t>(orig.GetDim(orig.GetDimNum() - 1)) : total;
        if (C == 0) C = 1;
        N = 1;
        H = 1;
        W = total / C;
        if (W == 0) W = 1;
    }

    const uint64_t totalRows64 = MulSafe(N, MulSafe(H, W));
    const uint32_t totalRows = ClampU32(totalRows64);
    const uint32_t cols = ClampU32(C);

    // Set a conservative blockDim to distribute rows to cores.
    uint32_t blockDim = totalRows;
    if (blockDim < 8u) blockDim = (totalRows == 0 ? 1u : totalRows);
    if (blockDim > 24u) blockDim = 24u;
    context->SetBlockDim(blockDim);

    tiling.set_totalRows(totalRows);
    tiling.set_cols(cols);
    tiling.set_blockDim(blockDim);

    tiling.SaveToBuffer(context->GetRawTilingData()->GetData(),
                        context->GetRawTilingData()->GetCapacity());
    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    // No GM workspace required.
    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;
    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

kernel_src="""
#include "kernel_operator.h"

constexpr int32_t INOUT_BUF_NUM = 2;

class KernelConv3dMinSoftmaxCustom {
public:
    __aicore__ inline KernelConv3dMinSoftmaxCustom() {}

    __aicore__ inline void Init(GM_ADDR in, GM_ADDR out, uint32_t totalRows, uint32_t cols)
    {
        totalRows_ = totalRows;
        cols_ = cols;

        inGm_.SetGlobalBuffer((__gm__ half*)in, static_cast<uint64_t>(totalRows_) * cols_);
        outGm_.SetGlobalBuffer((__gm__ half*)out, static_cast<uint64_t>(totalRows_) * cols_);

        // Per-row streaming UB plan:
        // - inQueueHalf/outQueueHalf: 1 row (cols_) each, double buffered.
        // - rowFloat: 1 row (cols_) as float compute buffer.
        // - workQueue: workspace for ReduceMax/ReduceSum.
        // - scalarQueue: store single-block accumulators (max/sum).
        pipe_.InitBuffer(inQueueHalf_, INOUT_BUF_NUM, cols_ * sizeof(half));
        pipe_.InitBuffer(outQueueHalf_, INOUT_BUF_NUM, cols_ * sizeof(half));
        pipe_.InitBuffer(rowFloatQueue_, 1, cols_ * sizeof(float));
        pipe_.InitBuffer(workQueue_, 1, 2048);     // conservative workspace for reductions
        pipe_.InitBuffer(scalarQueue_, 1, 32 * 3); // a few 32B blocks for max/sum/broadcast
    }

    __aicore__ inline void Process()
    {
        if (totalRows_ == 0 || cols_ == 0) {
            return;
        }
        const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
        const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());

        // Grid-stride over rows
        for (uint32_t row = blockIdx; row < totalRows_; row += (blockNum == 0 ? 1u : blockNum)) {
            CopyIn(row);
            ComputeRow();
            CopyOut(row);
        }
    }

private:
    __aicore__ inline void CopyIn(uint32_t rowIdx)
    {
        AscendC::LocalTensor<half> inLocal = inQueueHalf_.AllocTensor<half>();
        AscendC::DataCopy(inLocal, inGm_[static_cast<uint64_t>(rowIdx) * cols_], cols_);
        inQueueHalf_.EnQue(inLocal);
    }

    __aicore__ inline void ComputeRow()
    {
        // Deque 1 row of half -> cast to float for reduction/exp
        AscendC::LocalTensor<half> inHalf = inQueueHalf_.DeQue<half>();
        AscendC::LocalTensor<float> rowFloat = rowFloatQueue_.AllocTensor<float>();
        AscendC::Cast(rowFloat, inHalf, cols_);

        // Workspace and scalar buffers
        AscendC::LocalTensor<float> workLocal = workQueue_.AllocTensor<float>();
        AscendC::LocalTensor<float> maxBuf = scalarQueue_.AllocTensor<float>();   // 32 bytes
        AscendC::LocalTensor<float> sumBuf = scalarQueue_.AllocTensor<float>();   // another 32 bytes
        AscendC::LocalTensor<float> bcast = scalarQueue_.AllocTensor<float>();    // temp for Duplicate

        // max
        AscendC::ReduceMax<float>(maxBuf, rowFloat, workLocal, cols_, false);

        // x = x - max
        AscendC::Duplicate(bcast, maxBuf, cols_); // broadcast maxBuf[0] to length cols_
        AscendC::Sub(rowFloat, rowFloat, bcast, cols_);

        // exp(x)
        AscendC::Exp(rowFloat, rowFloat, cols_);

        // sum of exp
        AscendC::ReduceSum<float>(sumBuf, rowFloat, workLocal, cols_);

        // normalize: x = x / sum
        AscendC::Duplicate(bcast, sumBuf, cols_);
        AscendC::Div(rowFloat, rowFloat, bcast, cols_);

        // cast back to half and enqueue
        AscendC::LocalTensor<half> outHalf = outQueueHalf_.AllocTensor<half>();
        AscendC::Cast(outHalf, rowFloat, cols_);

        rowFloatQueue_.FreeTensor(rowFloat);
        workQueue_.FreeTensor(workLocal);
        scalarQueue_.FreeTensor(maxBuf);
        scalarQueue_.FreeTensor(sumBuf);
        scalarQueue_.FreeTensor(bcast);

        outQueueHalf_.EnQue(outHalf);
        inQueueHalf_.FreeTensor(inHalf);
    }

    __aicore__ inline void CopyOut(uint32_t rowIdx)
    {
        AscendC::LocalTensor<half> outHalf = outQueueHalf_.DeQue<half>();
        AscendC::DataCopy(outGm_[static_cast<uint64_t>(rowIdx) * cols_], outHalf, cols_);
        outQueueHalf_.FreeTensor(outHalf);
    }

private:
    AscendC::TPipe pipe_;
    AscendC::TQue<AscendC::TPosition::VECIN, INOUT_BUF_NUM> inQueueHalf_;
    AscendC::TQue<AscendC::TPosition::VECOUT, INOUT_BUF_NUM> outQueueHalf_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> rowFloatQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> workQueue_;
    AscendC::TQue<AscendC::TPosition::VECCALC, 1> scalarQueue_;

    AscendC::GlobalTensor<half> inGm_;
    AscendC::GlobalTensor<half> outGm_;

    uint32_t totalRows_ {0};
    uint32_t cols_ {0};
};

extern "C" __global__ __aicore__ void conv3d_min_softmax_custom(GM_ADDR t, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    KernelConv3dMinSoftmaxCustom op;
    op.Init(t, y, tiling_data.totalRows, tiling_data.cols);
    op.Process();
}

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

static inline at::Tensor to_contiguous_nhwc_fp16(const at::Tensor& x_nchw)
{
    // x_nchw: [N, C, H, W] -> [N, H, W, C] half, contiguous
    at::Tensor x_nhwc = x_nchw.permute({0, 2, 3, 1}).contiguous();
    return x_nhwc.to(at::kHalf);
}

static inline at::Tensor from_contiguous_nhwc_fp16_to_nchw_fp32(const at::Tensor& x_nhwc_half)
{
    // x_nhwc_half: [N, H, W, C] half -> [N, C, H, W] float
    at::Tensor x_nchw_half = x_nhwc_half.permute({0, 3, 1, 2}).contiguous();
    return x_nchw_half.to(at::kFloat);
}

at::Tensor conv3d_min_softmax_impl_npu(const at::Tensor& x,
                                       const at::Tensor& weight,
                                       const c10::optional<at::Tensor>& bias_opt)
{
    at::Tensor bias = bias_opt.has_value() ? *bias_opt : at::Tensor();

    // 1) conv3d (uses built-in NPU op)
    at::Tensor y = at::convolution(x, weight, bias, /*stride*/{1,1,1},
                                   /*padding*/{0,0,0}, /*dilation*/{1,1,1},
                                   /*transposed*/false, /*output_padding*/{0,0,0},
                                   /*groups*/1);

    // 2) min over D (dim=2 of NCDHW) -> NCHW
    at::Tensor y_min = at::amin(y, {2});

    // 3) NHWC half for row-wise softmax along last axis (C)
    at::Tensor y_nhwc_half = to_contiguous_nhwc_fp16(y_min);
    at::Tensor out_nhwc_half = at::empty_like(y_nhwc_half, at::device(y_nhwc_half.device()).dtype(at::kHalf));

    // 4) Launch custom AscendC kernel for softmax over last axis
    EXEC_NPU_CMD(aclnnConv3dMinSoftmaxCustom, y_nhwc_half, out_nhwc_half);

    // 5) Back to NCHW float
    at::Tensor result = from_contiguous_nhwc_fp16_to_nchw_fp32(out_nhwc_half);
    return result;
}

TORCH_LIBRARY(myops, m) {
    m.def("conv3d_min_softmax_custom(Tensor x, Tensor weight, Tensor? bias=None) -> Tensor");
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("conv3d_min_softmax_custom", &conv3d_min_softmax_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("conv3d_min_softmax_custom", &conv3d_min_softmax_impl_npu, "Conv3d-Min-Softmax fused (custom AscendC kernel for softmax)");
}
