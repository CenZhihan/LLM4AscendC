kernel_src="""
#include "kernel_operator.h"

using namespace AscendC;

class KernelConv2dSubTanhPool {
public:
    __aicore__ inline KernelConv2dSubTanhPool() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR b, GM_ADDR y, GM_ADDR tilingGm)
    {
        GET_TILING_DATA(tiling_data, tilingGm);

        N_ = tiling_data.N;
        Cin_ = tiling_data.Cin;
        Cout_ = tiling_data.Cout;
        H_ = tiling_data.H;
        W_ = tiling_data.W;
        Kh_ = tiling_data.Kh;
        Kw_ = tiling_data.Kw;
        Oh_ = tiling_data.Oh;
        Ow_ = tiling_data.Ow;
        Ph_ = tiling_data.Ph;
        Pw_ = tiling_data.Pw;
        ksp_ = tiling_data.ksp;
        tileCo_ = tiling_data.tileCo;
        patchH_ = tiling_data.patchH;
        patchW_ = tiling_data.patchW;
        hasBias_ = tiling_data.hasBias != 0;
        sub1_ = tiling_data.subtract1;
        sub2_ = tiling_data.subtract2;

        xGm_.SetGlobalBuffer((__gm__ float*)x, (uint64_t)N_*Cin_*H_*W_);
        wGm_.SetGlobalBuffer((__gm__ float*)w, (uint64_t)Cout_*Cin_*Kh_*Kw_);
        if (hasBias_) {
            bGm_.SetGlobalBuffer((__gm__ float*)b, (uint64_t)Cout_);
        }
        yGm_.SetGlobalBuffer((__gm__ float*)y, (uint64_t)N_*Cout_*Ph_*Pw_);

        // UB buffers
        uint32_t wTileElems = tileCo_ * Cin_ * Kh_ * Kw_;
        uint32_t xPatchElems = Cin_ * patchH_ * patchW_;
        uint32_t vecElems = tileCo_;

        pipe_.InitBuffer(wQueue_, 1, (uint64_t)wTileElems * sizeof(float));
        pipe_.InitBuffer(xPatchQueue_, 1, (uint64_t)xPatchElems * sizeof(float));
        pipe_.InitBuffer(vecAQueue_, 1, (uint64_t)vecElems * sizeof(float)); // convVec
        pipe_.InitBuffer(vecBQueue_, 1, (uint64_t)vecElems * sizeof(float)); // tmpVec
        pipe_.InitBuffer(vecCQueue_, 1, (uint64_t)vecElems * sizeof(float)); // tanhVec
        pipe_.InitBuffer(accQueue_, 1, (uint64_t)vecElems * sizeof(float));  // accVec
        pipe_.InitBuffer(biasQueue_, 1, (uint64_t)vecElems * sizeof(float)); // biasVec
    }

    __aicore__ inline void Process()
    {
        if (Cout_ == 0 || Ph_ == 0 || Pw_ == 0) return;

        uint32_t blockNum = (uint32_t)GetBlockNum();
        uint32_t blockIdx = (uint32_t)GetBlockIdx();

        // co-tile grid-stride
        const uint32_t coTiles = (Cout_ + tileCo_ - 1) / tileCo_;
        for (uint32_t coTileIdx = blockIdx; coTileIdx < coTiles; coTileIdx += blockNum) {
            uint32_t coStart = coTileIdx * tileCo_;
            uint32_t curCo = Cout_ - coStart;
            if (curCo > tileCo_) curCo = tileCo_;

            // Load weight tile [curCo, Cin, Kh, Kw] contiguous
            LocalTensor<float> wUb = wQueue_.AllocTensor<float>();
            uint32_t wTileElems = curCo * Cin_ * Kh_ * Kw_;
            uint64_t wBase = (uint64_t)coStart * Cin_ * Kh_ * Kw_;
            DataCopy(wUb, wGm_[wBase], wTileElems);
            wQueue_.EnQue(wUb);

            // Load bias tile if present
            LocalTensor<float> bUb;
            if (hasBias_) {
                bUb = biasQueue_.AllocTensor<float>();
                DataCopy(bUb, bGm_[coStart], curCo);
                biasQueue_.EnQue(bUb);
            }

            wUb = wQueue_.DeQue<float>();
            if (hasBias_) bUb = biasQueue_.DeQue<float>();

            // Iterate over pooled positions q = n*Ph*Pw + ph*Pw + pw
            const uint64_t totalQ = (uint64_t)N_ * Ph_ * Pw_;
            for (uint64_t q = 0; q < totalQ; ++q) {
                uint32_t n, ph, pw;
                DecomposeQ(q, n, ph, pw);
                // Copy input patch of size [Cin, patchH_, patchW_] starting at (h0=ph*ksp_, w0=pw*ksp_)
                uint32_t h0 = ph * ksp_;
                uint32_t w0 = pw * ksp_;
                LocalTensor<float> xUb = xPatchQueue_.AllocTensor<float>();
                LoadInputPatch(xUb, n, h0, w0);
                xPatchQueue_.EnQue(xUb);

                xUb = xPatchQueue_.DeQue<float>();

                // accVec = 0
                LocalTensor<float> accVec = accQueue_.AllocTensor<float>();
                Duplicate(accVec, 0.0f, (int32_t)curCo);

                // For each position in pooling window
                for (uint32_t pi = 0; pi < ksp_; ++pi) {
                    for (uint32_t pj = 0; pj < ksp_; ++pj) {
                        // convVec = conv(x patch at offset (pi,pj), wUb)
                        LocalTensor<float> convVec = vecAQueue_.AllocTensor<float>();
                        Duplicate(convVec, 0.0f, (int32_t)curCo);
                        ComputeConvVec(convVec, wUb, xUb, pi, pj, curCo);

                        // add bias if present: convVec += bUb
                        if (hasBias_) {
                            Add(convVec, convVec, bUb, (int32_t)curCo);
                        }

                        // tmpVec = convVec - sub1_
                        LocalTensor<float> tmpVec = vecBQueue_.AllocTensor<float>();
                        // tmpVec <- scalar sub1
                        Duplicate(tmpVec, sub1_, (int32_t)curCo);
                        Sub(tmpVec, convVec, tmpVec, (int32_t)curCo);

                        // tanhVec = tanh(tmpVec)
                        LocalTensor<float> tanhVec = vecCQueue_.AllocTensor<float>();
                        Tanh(tanhVec, tmpVec, (uint32_t)curCo);

                        // tmpVec = tanhVec - sub2_
                        Duplicate(tmpVec, sub2_, (int32_t)curCo);
                        Sub(tmpVec, tanhVec, tmpVec, (int32_t)curCo);

                        // accVec += tmpVec
                        Add(accVec, accVec, tmpVec, (int32_t)curCo);

                        vecCQueue_.FreeTensor(tanhVec);
                        vecBQueue_.FreeTensor(tmpVec);
                        vecAQueue_.FreeTensor(convVec);
                    }
                }

                // accVec *= (1.0f / (ksp_*ksp_))
                float invArea = 1.0f / (float)(ksp_ * ksp_);
                Muls(accVec, accVec, invArea, (int32_t)curCo);

                // Store to GM: y[n, coStart:coStart+curCo, ph, pw]
                for (uint32_t cL = 0; cL < curCo; ++cL) {
                    uint64_t yOff = YOffset(n, coStart + cL, ph, pw);
                    DataCopy(yGm_[yOff], accVec[cL], 1);
                }

                accQueue_.FreeTensor(accVec);
                xPatchQueue_.FreeTensor(xUb);
            }

            if (hasBias_) biasQueue_.FreeTensor(bUb);
            wQueue_.FreeTensor(wUb);
        }
    }

private:
    __aicore__ inline uint64_t XOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * Cin_ + c) * H_ + h) * W_ + w;
    }
    __aicore__ inline uint64_t WOffset(uint32_t co, uint32_t c, uint32_t kh, uint32_t kw) const
    {
        return (((uint64_t)co * Cin_ + c) * Kh_ + kh) * Kw_ + kw;
    }
    __aicore__ inline uint64_t YOffset(uint32_t n, uint32_t c, uint32_t h, uint32_t w) const
    {
        return (((uint64_t)n * Cout_ + c) * Ph_ + h) * Pw_ + w;
    }

    __aicore__ inline void DecomposeQ(uint64_t q, uint32_t &n, uint32_t &ph, uint32_t &pw) const
    {
        uint64_t perN = (uint64_t)Ph_ * Pw_;
        n = (uint32_t)(q / perN);
        uint64_t rem = q % perN;
        ph = (uint32_t)(rem / Pw_);
        pw = (uint32_t)(rem % Pw_);
    }

    // Load input patch for a given (n, h0, w0) with size [Cin, patchH_, patchW_] into xUb
    __aicore__ inline void LoadInputPatch(LocalTensor<float> &xUb, uint32_t n, uint32_t h0, uint32_t w0)
    {
        // For each channel and row, copy patchW_ floats
        for (uint32_t c = 0; c < Cin_; ++c) {
            for (uint32_t rh = 0; rh < patchH_; ++rh) {
                uint32_t gh = h0 + rh;
                uint64_t gmOff = XOffset(n, c, gh, w0);
                uint64_t ubOff = ((uint64_t)c * patchH_ + rh) * patchW_;
                DataCopy(xUb[ubOff], xGm_[gmOff], patchW_);
            }
        }
    }

    // convVec[co] = sum_{c,kh,kw} w[co,c,kh,kw] * xUb[c, pi+kh, pj+kw]
    __aicore__ inline void ComputeConvVec(LocalTensor<float> &convVec,
                                          LocalTensor<float> &wUb,
                                          LocalTensor<float> &xUb,
                                          uint32_t pi, uint32_t pj, uint32_t curCo)
    {
        uint32_t CKKSIZE = Cin_ * Kh_ * Kw_;
        for (uint32_t coL = 0; coL < curCo; ++coL) {
            float sum = 0.0f;
            uint64_t wBase = (uint64_t)coL * CKKSIZE;
            for (uint32_t c = 0; c < Cin_; ++c) {
                for (uint32_t kh = 0; kh < Kh_; ++kh) {
                    for (uint32_t kw = 0; kw < Kw_; ++kw) {
                        uint64_t wIdx = wBase + ((uint64_t)c * Kh_ + kh) * Kw_ + kw;
                        uint64_t xIdx = ((uint64_t)c * patchH_ + (pi + kh)) * patchW_ + (pj + kw);
                        float wv = wUb.GetValue(wIdx);
                        float xv = xUb.GetValue(xIdx);
                        sum += wv * xv;
                    }
                }
            }
            convVec.SetValue(coL, sum);
        }
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN, 1> wQueue_;
    TQue<TPosition::VECIN, 1> xPatchQueue_;
    TQue<TPosition::VECCALC, 1> vecAQueue_;
    TQue<TPosition::VECCALC, 1> vecBQueue_;
    TQue<TPosition::VECCALC, 1> vecCQueue_;
    TQue<TPosition::VECCALC, 1> accQueue_;
    TQue<TPosition::VECIN, 1> biasQueue_;

    GlobalTensor<float> xGm_, wGm_, bGm_, yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t H_{0}, W_{0}, Kh_{0}, Kw_{0};
    uint32_t Oh_{0}, Ow_{0}, Ph_{0}, Pw_{0}, ksp_{1};
    uint32_t tileCo_{1}, patchH_{0}, patchW_{0};
    bool hasBias_{false};
    float sub1_{0.0f}, sub2_{0.0f};
};

extern "C" __global__ __aicore__ void conv2d_subtract_tanh_subtract_avg_pool_custom(GM_ADDR x, GM_ADDR weight, GM_ADDR conv_bias, GM_ADDR y, GM_ADDR workspace, GM_ADDR tiling)
{
    KernelConv2dSubTanhPool op;
    op.Init(x, weight, conv_bias, y, tiling);
    op.Process();
}
