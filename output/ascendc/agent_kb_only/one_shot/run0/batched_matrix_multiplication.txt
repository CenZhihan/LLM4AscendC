host_operator_src="""

#include "batched_matrix_multiplication_custom_tiling.h"
#include "register/op_def_registry.h"
#include <algorithm>
#include <cstdint>
#include <cstring>

// Matmul tiling headers
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_tiling.h"

namespace {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

} // namespace

namespace optiling {

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Get input shapes
    const gert::Shape* a_shape = context->GetInputShape(0); // [B, M, K]
    const gert::Shape* b_shape = context->GetInputShape(1); // [B, K, N]
    if (a_shape->GetDimNum() != 3 || b_shape->GetDimNum() != 3) {
        return ge::GRAPH_FAILED;
    }

    const int64_t B = a_shape->GetDim(0);
    const int64_t M = a_shape->GetDim(1);
    const int64_t K = a_shape->GetDim(2);
    const int64_t B2 = b_shape->GetDim(0);
    const int64_t K2 = b_shape->GetDim(1);
    const int64_t N = b_shape->GetDim(2);
    if (B <= 0 || M <= 0 || K <= 0 || N <= 0 || B != B2 || K != K2) {
        return ge::GRAPH_FAILED;
    }

    // Build Matmul tiling using NORMAL(BMNK) layout
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MultiCoreMatmulTiling tiling(ascendcPlatform);

    const int32_t m = static_cast<int32_t>(M);
    const int32_t n = static_cast<int32_t>(N);
    const int32_t k = static_cast<int32_t>(K);
    const int32_t batch = static_cast<int32_t>(B);

    tiling.SetDim(1);
    tiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                    matmul_tiling::DataType::DT_FLOAT);
    tiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND,
                       matmul_tiling::DataType::DT_FLOAT);
    tiling.SetShape(m, n, k);
    tiling.SetOrgShape(m, n, k);
    tiling.SetBias(false);
    tiling.SetBufferSpace(-1, -1, -1);

    // NORMAL layout batches
    tiling.SetBatchInfoForNormal(batch, batch, m, n, k);
    tiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling tilingData;
    int ret = tiling.GetTiling(tilingData);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // Determine blockDim: use a conservative cap based on number of batches
    uint32_t blockDim = static_cast<uint32_t>(std::min<int64_t>(B, 24));
    if (blockDim == 0) blockDim = 1;
    context->SetBlockDim(blockDim);

    // Pack raw tiling bytes: [TCubeTiling | uint32_t B | uint32_t M | uint32_t N | uint32_t K]
    const uint32_t extraU32Count = 4;
    const uint32_t rawSize = sizeof(optiling::TCubeTiling) + extraU32Count * sizeof(uint32_t);
    void* rawBuf = context->GetRawTilingData()->GetData();
    if (context->GetRawTilingData()->GetCapacity() < rawSize) {
        return ge::GRAPH_FAILED;
    }

    std::memcpy(rawBuf, &tilingData, sizeof(optiling::TCubeTiling));
    uint32_t* meta = reinterpret_cast<uint32_t*>(reinterpret_cast<uint8_t*>(rawBuf) + sizeof(optiling::TCubeTiling));
    meta[0] = static_cast<uint32_t>(B);
    meta[1] = static_cast<uint32_t>(M);
    meta[2] = static_cast<uint32_t>(N);
    meta[3] = static_cast<uint32_t>(K);

    context->GetRawTilingData()->SetDataSize(rawSize);

    // No workspace needed
    size_t *currentWorkspace = context->GetWorkspaceSizes(1);
    currentWorkspace[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {

static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape* a_shape = context->GetInputShape(0); // [B, M, K]
    const gert::Shape* b_shape = context->GetInputShape(1); // [B, K, N]
    if (a_shape->GetDimNum() != 3 || b_shape->GetDimNum() != 3) {
        return GRAPH_FAILED;
    }
    const int64_t B = a_shape->GetDim(0);
    const int64_t M = a_shape->GetDim(1);
    const int64_t N = b_shape->GetDim(2);

    gert::Shape* c_shape = context->GetOutputShape(0);
    c_shape->SetDimNum(3);
    c_shape->SetDim(0, B);
    c_shape->SetDim(1, M);
    c_shape->SetDim(2, N);
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    const ge::DataType a_dtype = context->GetInputDataType(0);
    const ge::DataType b_dtype = context->GetInputDataType(1);
    if (a_dtype != b_dtype) {
        return GRAPH_FAILED;
    }
    context->SetOutputDataType(0, a_dtype);
    return GRAPH_SUCCESS;
}

} // namespace ge

kernel_src="""

#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using A_T = float;
using B_T = float;
using C_T = float;

extern "C" __global__ __aicore__ void batched_matrix_multiplication_custom(GM_ADDR aGM, GM_ADDR bGM, GM_ADDR cGM, GM_ADDR tilingGM, GM_ADDR workspaceGM)
{
    // Define Matmul types for NORMAL layout (BMNK), no transpose
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, B_T, false, LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, C_T> biasType; // unused

    // Load TCubeTiling from GM (raw tiling bytes packed by host)
    optiling::TCubeTiling tiling;
    auto gm_u32 = reinterpret_cast<__gm__ const uint32_t*>(tilingGM);
    uint32_t* local_u32 = reinterpret_cast<uint32_t*>(&tiling);
    for (uint32_t i = 0; i < sizeof(optiling::TCubeTiling) / sizeof(uint32_t); ++i) {
        local_u32[i] = gm_u32[i];
    }

    // Read appended metadata [B, M, N, K] right after TCubeTiling
    const __gm__ uint32_t* meta = gm_u32 + static_cast<uint32_t>(sizeof(optiling::TCubeTiling) / sizeof(uint32_t));
    const uint32_t B = meta[0];
    const uint32_t M = meta[1];
    const uint32_t N = meta[2];
    const uint32_t K = meta[3];

    // Set global tensors
    AscendC::GlobalTensor<A_T> aGlobal;
    AscendC::GlobalTensor<B_T> bGlobal;
    AscendC::GlobalTensor<C_T> cGlobal;

    const uint64_t aElems = static_cast<uint64_t>(B) * M * K;
    const uint64_t bElems = static_cast<uint64_t>(B) * K * N;
    const uint64_t cElems = static_cast<uint64_t>(B) * M * N;

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ A_T*>(aGM), aElems);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ B_T*>(bGM), bElems);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ C_T*>(cGM), cElems);

    // Create Matmul instance
    matmul::Matmul<aType, bType, cType, biasType> mm;
    AscendC::TPipe pipe;
    g_cubeTPipePtr = &pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
    mm.Init(&tiling);

    // Bind matrices
    mm.SetTensorA(aGlobal);
    mm.SetTensorB(bGlobal);
    // No bias
    mm.SetWorkspace(workspaceGM, 0);

    // Compute all data on a single core (the tiling covers multi-core and batching)
    mm.IterateAll(cGlobal);
    mm.End();
}

#include <torch/library.h>

at::Tensor bmm_impl_npu(const at::Tensor& A, const at::Tensor& B) {
    TORCH_CHECK(A.dim() == 3 && B.dim() == 3, \"Inputs must be 3D tensors [B, M, K] and [B, K, N]\");
    TORCH_CHECK(A.size(0) == B.size(0) && A.size(2) == B.size(1), \"Batch and K dimensions must match\");

    auto M = A.size(1);
    auto N = B.size(2);
    at::Tensor C = at::empty({Bsz, M, N}, A.options());
    EXEC_NPU_CMD(aclnnBatchedMatrixMultiplicationCustom, A, B, C);
    return C;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl(\"batched_matrix_multiplication_custom\", &bmm_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def(\"batched_matrix_multiplication_custom\", &bmm_impl_npu, \"Batched MatMul (BatchedMatrixMultiplicationCustom on NPU)\");
