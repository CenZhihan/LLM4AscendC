kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using AscendC::GlobalTensor;
using AscendC::LocalTensor;

typedef half A_T;
typedef half B_T;
typedef float C_T;
typedef float BiasT;

struct TailInfo {
    uint32_t N;
    uint32_t K;
    uint32_t C;
    uint32_t G;
    float eps;
    float htMin;
    float htMax;
};

constexpr int32_t TILE_C = 2048;
constexpr int32_t BUFFER_NUM = 2;

template<typename T>
__aicore__ inline void ClampVec(LocalTensor<T>& x, float minv, float maxv, uint32_t len)
{
    AscendC::Maxs(x, x, static_cast<T>(minv), len);
    AscendC::Mins(x, x, static_cast<T>(maxv), len);
}

template<typename T>
__aicore__ inline void Mish(LocalTensor<T>& x, LocalTensor<T>& tmp, uint32_t len)
{
    // tmp = ln(1 + exp(x))
    AscendC::Exp(tmp, x, len);
    AscendC::Adds(tmp, tmp, static_cast<T>(1.0f), len);
    AscendC::Ln(tmp, tmp, len);
    // tmp = tanh(tmp)
    AscendC::Tanh(tmp, tmp, len);
    // x = x * tmp
    AscendC::Mul(x, x, tmp, len);
}

extern "C" __global__ __aicore__ void gemm_bias_add_hardtanh_mish_group_norm_custom(
    GM_ADDR xGM, GM_ADDR wGM, GM_ADDR linBiasGM, GM_ADDR addBiasGM,
    GM_ADDR gammaGM, GM_ADDR betaGM, GM_ADDR yGM, GM_ADDR tilingGM, GM_ADDR workspaceGM)
{
    // Load TCubeTiling from GM
    optiling::TCubeTiling cubeTiling;
    {
        auto gmPtr = (__gm__ uint32_t*)tilingGM;
        auto locPtr = (uint32_t*)&cubeTiling;
        for (int i = 0; i < (int)(sizeof(optiling::TCubeTiling) / sizeof(uint32_t)); ++i, ++gmPtr, ++locPtr) {
            *locPtr = *gmPtr;
        }
    }
    // Load tail info
    TailInfo tailInfo;
    {
        const uint8_t *base = reinterpret_cast<const uint8_t*>(tilingGM) + sizeof(optiling::TCubeTiling);
        const uint32_t *uTail = reinterpret_cast<const uint32_t*>(base);
        tailInfo.N = uTail[0];
        tailInfo.K = uTail[1];
        tailInfo.C = uTail[2];
        tailInfo.G = uTail[3];
        const float *fTail = reinterpret_cast<const float*>(uTail + 4);
        tailInfo.eps = fTail[0];
        tailInfo.htMin = fTail[1];
        tailInfo.htMax = fTail[2];
    }

    const uint32_t N = tailInfo.N;
    const uint32_t K = tailInfo.K;
    const uint32_t C = tailInfo.C;
    const uint32_t G = tailInfo.G;
    const uint32_t groupSize = C / G;

    // Set GM tensors
    GlobalTensor<A_T> aGlobal;
    GlobalTensor<B_T> bGlobal;
    GlobalTensor<C_T> cGlobal;
    GlobalTensor<BiasT> linBiasGlobal;
    GlobalTensor<BiasT> addBiasGlobal;
    GlobalTensor<BiasT> gammaGlobal;
    GlobalTensor<BiasT> betaGlobal;

    aGlobal.SetGlobalBuffer((__gm__ A_T*)xGM, N * K);
    bGlobal.SetGlobalBuffer((__gm__ B_T*)wGM, C * K); // weight [C, K], transpose in Matmul
    cGlobal.SetGlobalBuffer((__gm__ C_T*)yGM, N * C);
    linBiasGlobal.SetGlobalBuffer((__gm__ BiasT*)linBiasGM, C);
    addBiasGlobal.SetGlobalBuffer((__gm__ BiasT*)addBiasGM, C);
    gammaGlobal.SetGlobalBuffer((__gm__ BiasT*)gammaGM, C);
    betaGlobal.SetGlobalBuffer((__gm__ BiasT*)betaGM, C);

    // 1) GEMM: y = x @ weight^T + lin_bias
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, A_T, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, B_T, true,  LayoutMode::NORMAL> bType; // transpose weight
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, C_T, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType <AscendC::TPosition::GM, CubeFormat::ND, BiasT> biasType;

    matmul::Matmul<aType, bType, cType, biasType> mm;
    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm);
    mm.Init(&cubeTiling);
    mm.SetTensorA(aGlobal, false);
    mm.SetTensorB(bGlobal, true);
    mm.SetWorkspace(workspaceGM, 0);
    mm.SetBias(linBiasGlobal);
    // Single-batch NORMAL
    mm.IterateBatch(cGlobal, 1, 1, false);

    // Common queues for element-wise passes
    AscendC::TPipe vecPipe;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQ;
    AscendC::TQue<AscendC::TPosition::VECIN, BUFFER_NUM> inQ2;
    AscendC::TQue<AscendC::TPosition::VECCALC, BUFFER_NUM> tmpQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, BUFFER_NUM> outQ;
    vecPipe.InitBuffer(inQ, BUFFER_NUM, TILE_C * sizeof(C_T));
    vecPipe.InitBuffer(inQ2, BUFFER_NUM, TILE_C * sizeof(C_T));
    vecPipe.InitBuffer(tmpQ, BUFFER_NUM, TILE_C * sizeof(C_T));
    vecPipe.InitBuffer(outQ, BUFFER_NUM, TILE_C * sizeof(C_T));

    const uint32_t blockNum = static_cast<uint32_t>(AscendC::GetBlockNum());
    const uint32_t blockIdx = static_cast<uint32_t>(AscendC::GetBlockIdx());
    if (blockNum == 0u) { return; }

    // 2) BiasAdd + HardTanh + Mish, in-place on y
    for (uint32_t n = blockIdx; n < N; n += blockNum) {
        uint32_t base = n * C;
        for (uint32_t c0 = 0; c0 < C; c0 += TILE_C) {
            uint32_t len = AscendC::min((uint32_t)TILE_C, C - c0);

            LocalTensor<C_T> yLoc = inQ.AllocTensor<C_T>();
            LocalTensor<C_T> bLoc = inQ2.AllocTensor<C_T>();
            LocalTensor<C_T> tLoc = tmpQ.AllocTensor<C_T>();

            AscendC::DataCopy(yLoc, cGlobal[base + c0], len);
            AscendC::DataCopy(bLoc, addBiasGlobal[c0], len);

            AscendC::Add(yLoc, yLoc, bLoc, len);                   // y += bias
            ClampVec<C_T>(yLoc, tailInfo.htMin, tailInfo.htMax, len); // HardTanh
            Mish<C_T>(yLoc, tLoc, len);                             // Mish

            outQ.EnQue(yLoc);
            inQ2.FreeTensor(bLoc);
            tmpQ.FreeTensor(tLoc);

            LocalTensor<C_T> yOut = outQ.DeQue<C_T>();
            AscendC::DataCopy(cGlobal[base + c0], yOut, len);
            outQ.FreeTensor(yOut);
            inQ.FreeTensor(yLoc);
        }
    }

    // 3) GroupNorm: per (n, group), normalize y and then apply gamma/beta
    for (uint32_t n = blockIdx; n < N; n += blockNum) {
        const uint32_t rowBase = n * C;
        for (uint32_t g = 0; g < G; ++g) {
            const uint32_t cStart = g * groupSize;

            // Load group slice
            LocalTensor<C_T> yLoc = inQ.AllocTensor<C_T>();
            AscendC::DataCopy(yLoc, cGlobal[rowBase + cStart], groupSize);

            // Compute sum and sumsq
            LocalTensor<C_T> sqLoc = inQ2.AllocTensor<C_T>();
            AscendC::Mul(sqLoc, yLoc, yLoc, groupSize);

            LocalTensor<C_T> sumLoc = tmpQ.AllocTensor<C_T>();
            LocalTensor<C_T> sumsqLoc = outQ.AllocTensor<C_T>();
            // ReduceSum into first element (assumes API stores result to head of dst tensor)
            AscendC::ReduceSum(sumLoc, yLoc, groupSize);
            AscendC::ReduceSum(sumsqLoc, sqLoc, groupSize);

            __ub__ C_T* sumUb = sumLoc.GetPhyAddr();
            __ub__ C_T* sumsqUb = sumsqLoc.GetPhyAddr();
            float sumVal = static_cast<float>(sumUb[0]);
            float sumsqVal = static_cast<float>(sumsqUb[0]);

            tmpQ.FreeTensor(sumLoc);
            outQ.FreeTensor(sumsqLoc);
            inQ2.FreeTensor(sqLoc);

            const float gs = static_cast<float>(groupSize);
            const float mean = sumVal / gs;
            float var = sumsqVal / gs - mean * mean;
            if (var < 0.f) var = 0.f; // numerical guard
            const float invStd = 1.0f / AscendC::sqrt(var + tailInfo.eps);

            // Normalize: y = (y - mean) * invStd
            AscendC::Adds(yLoc, yLoc, -mean, groupSize);
            AscendC::Muls(yLoc, yLoc, invStd, groupSize);

            // Affine: y = y * gamma + beta
            LocalTensor<C_T> gammaLoc = inQ2.AllocTensor<C_T>();
            LocalTensor<C_T> betaLoc = tmpQ.AllocTensor<C_T>();
            AscendC::DataCopy(gammaLoc, gammaGlobal[cStart], groupSize);
            AscendC::DataCopy(betaLoc,  betaGlobal[cStart],  groupSize);

            AscendC::Mul(yLoc, yLoc, gammaLoc, groupSize);
            AscendC::Add(yLoc, yLoc, betaLoc, groupSize);

            // Store back
            AscendC::DataCopy(cGlobal[rowBase + cStart], yLoc, groupSize);

            inQ2.FreeTensor(gammaLoc);
            tmpQ.FreeTensor(betaLoc);
            inQ.FreeTensor(yLoc);
        }
    }
}
