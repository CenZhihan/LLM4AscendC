static inline uint32_t AlignDown(uint32_t x, uint32_t a)
{
    return (a == 0) ? x : (x / a) * a;
}

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (b == 0) ? 0 : ((a + b - 1u) / b);
}

} // namespace

namespace optiling {

static constexpr uint32_t UB_BUDGET_BYTES = 192 * 1024;  // per-core conservative UB usage
static constexpr uint32_t ALIGN_ELEMS = 1024;            // DMA-friendly alignment
static constexpr uint32_t HARD_CAP_ELEMS = 32768;        // cap tile width

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    ConvTransposed3dAsymmetricInputSquareKernelCustomTilingData tiling;

    // Shapes:
    // x: [N, Cin, D, H, W]
    // weight: [Cin, Cout, Kd, Kh, Kw] (PyTorch conv_transpose3d layout)
    const gert::Shape *xShape = context->GetInputShape(0);
    const gert::Shape *wShape = context->GetInputShape(1);
    if (xShape->GetDimNum() != 5 || wShape->GetDimNum() != 5) {
        return ge::GRAPH_FAILED;
    }

    const uint32_t N    = static_cast<uint32_t>(xShape->GetDim(0));
    const uint32_t Cin  = static_cast<uint32_t>(xShape->GetDim(1));
    const uint32_t D    = static_cast<uint32_t>(xShape->GetDim(2));
    const uint32_t H    = static_cast<uint32_t>(xShape->GetDim(3));
    const uint32_t W    = static_cast<uint32_t>(xShape->GetDim(4));

    const uint32_t WinCin = static_cast<uint32_t>(wShape->GetDim(0));
    const uint32_t Cout   = static_cast<uint32_t>(wShape->GetDim(1));
    const uint32_t Kd     = static_cast<uint32_t>(wShape->GetDim(2));
    const uint32_t Kh     = static_cast<uint32_t>(wShape->GetDim(3));
    const uint32_t Kw     = static_cast<uint32_t>(wShape->GetDim(4));

    if (WinCin != Cin) {
        return ge::GRAPH_FAILED;
    }

    // This custom kernel specializes to stride=1, pad=0, dilation=1, groups=1
    const uint32_t Dout = D + Kd - 1;
    const uint32_t Hout = H + Kh - 1;
    const uint32_t Wout = W + Kw - 1;

    // UB model: yLine: 1 * tileW, xLine: 1 * tileW => 2 * tileW floats
    // Add small temp for weights cache if needed; still 2 * tileW dominates.
    uint32_t maxTileW = (UB_BUDGET_BYTES / sizeof(float)) / 2;
    maxTileW = AlignDown(maxTileW, ALIGN_ELEMS);
    if (maxTileW < ALIGN_ELEMS) maxTileW = ALIGN_ELEMS;
    if (maxTileW > HARD_CAP_ELEMS) maxTileW = AlignDown(HARD_CAP_ELEMS, ALIGN_ELEMS);

    uint32_t tileW = maxTileW;
    if (Wout < tileW) {
        const uint32_t aligned = AlignDown(Wout, ALIGN_ELEMS);
        tileW = (aligned > 0) ? aligned : Wout;
    }

    const uint64_t totalLines = static_cast<uint64_t>(N) *
                                static_cast<uint64_t>(Cout) *
                                static_cast<uint64_t>(Dout) *
                                static_cast<uint64_t>(Hout);

    // Conservative blockDim: one line per block, bounded
    uint32_t blockDim = static_cast<uint32_t>(totalLines);
    if (blockDim < 8u) blockDim = 8u;
    if (blockDim > 32u) blockDim = 32u;
    context->SetBlockDim(blockDim);

    tiling.set_N(N);
    tiling.set_Cin(Cin);
    tiling.set_Cout(Cout);
    tiling.set_D(D);
    tiling.set_H(H);
    tiling.set_W(W);

    tiling.set_Kh(Kh);
    tiling.set_Kw(Kw);

    tiling.set_Wout(Wout);

    tiling.set_blockDim(blockDim);

    context->GetRawTilingData()->SetDataSize(tiling.GetDataSize());

    size_t *workspace = context->GetWorkspaceSizes(1);
    workspace[0] = 0;

kernel_src="""

#include "kernel_operator.h"

using namespace AscendC;

constexpr int32_t X_BUFFER_NUM = 1;
constexpr int32_t Y_BUFFER_NUM = 1;
constexpr int32_t W_BUFFER_NUM = 1;

struct CT3DTile {
    uint32_t N, Cin, Cout;
    uint32_t D, H, W;
    uint32_t Kd, Kh, Kw;
    uint32_t Dout, Hout, Wout;
    uint32_t tileW;
    uint64_t totalLines;
};

class KernelConvTransposed3dAsymmetricInputSquareKernelCustom {
public:
    __aicore__ inline KernelConvTransposed3dAsymmetricInputSquareKernelCustom() {}

    __aicore__ inline void Init(GM_ADDR x, GM_ADDR w, GM_ADDR y, const CT3DTile &t)
    {
        N_ = t.N; Cin_ = t.Cin; Cout_ = t.Cout;
        D_ = t.D; H_ = t.H; W_ = t.W;
        Kd_ = t.Kd; Kh_ = t.Kh; Kw_ = t.Kw;
        Dout_ = t.Dout; Hout_ = t.Hout; Wout_ = t.Wout;
        tileW_ = t.tileW;
        totalLines_ = t.totalLines;

        const uint64_t xSize = (uint64_t)N_ * Cin_ * D_ * H_ * W_;
        const uint64_t ySize = (uint64_t)N_ * Cout_ * Dout_ * Hout_ * Wout_;
        const uint64_t wSize = (uint64_t)Cin_ * Cout_ * Kd_ * Kh_ * Kw_;

        xGm_.SetGlobalBuffer((__gm__ float*)x, xSize);
        wGm_.SetGlobalBuffer((__gm__ float*)w, wSize);
        yGm_.SetGlobalBuffer((__gm__ float*)y, ySize);

        pipe_.InitBuffer(xQueue_, X_BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(yQueue_, Y_BUFFER_NUM, tileW_ * sizeof(float));
        pipe_.InitBuffer(wQueue_, W_BUFFER_NUM, (uint32_t)(Kd_*Kh_*Kw_) * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        if (totalLines_ == 0 || tileW_ == 0) return;

        const uint32_t blockNum = (uint32_t)GetBlockNum();
        const uint32_t blockIdx = (uint32_t)GetBlockIdx();

        for (uint64_t lineIdx = blockIdx; lineIdx < totalLines_; lineIdx += blockNum) {
            // Decode (n, co, do, ho) from lineIdx
            uint64_t tmp = lineIdx;
            const uint32_t ho = (uint32_t)(tmp % Hout_); tmp /= Hout_;
            const uint32_t do_ = (uint32_t)(tmp % Dout_); tmp /= Dout_;
            const uint32_t co = (uint32_t)(tmp % Cout_); tmp /= Cout_;
            const uint32_t n  = (uint32_t)(tmp % N_);

            for (uint32_t w0 = 0; w0 < Wout_; w0 += tileW_) {
                uint32_t curLen = Wout_ - w0;
                if (curLen > tileW_) curLen = tileW_;

                // yLocal: accumulation buffer for this line segment
                LocalTensor<float> yLocal = yQueue_.AllocTensor<float>();
                Duplicate(yLocal, 0.0f, (int32_t)curLen);

                // For each input channel, load its 3x3x3 weights for (cin,co) pair
                for (uint32_t ci = 0; ci < Cin_; ++ci) {
                    // Load 27 weights into local buffer
                    LocalTensor<float> wLocal = wQueue_.AllocTensor<float>();
                    const uint64_t wBase = (((uint64_t)ci * Cout_ + co) * Kd_ * Kh_ * Kw_);
                    DataCopy(wLocal, wGm_[wBase], (uint32_t)(Kd_ * Kh_ * Kw_));

                    // Iterate over kernel positions
                    for (uint32_t kd = 0; kd < Kd_; ++kd) {
                        const int32_t xd = (int32_t)do_ - (int32_t)kd;
                        if (xd < 0 || xd >= (int32_t)D_) continue;
                        for (uint32_t kh = 0; kh < Kh_; ++kh) {
                            const int32_t xh = (int32_t)ho - (int32_t)kh;
                            if (xh < 0 || xh >= (int32_t)H_) continue;
                            for (uint32_t kw = 0; kw < Kw_; ++kw) {
                                // Find intersection on W axis for this tile segment
                                // valid w satisfy xw = w - kw in [0, W-1] => w in [kw, kw+W-1]
                                const uint32_t segStart = (w0 > kw) ? w0 : kw;
                                const uint32_t segEnd = ((w0 + curLen) < (kw + W_)) ? (w0 + curLen) : (kw + W_);
                                if (segStart >= segEnd) continue;
                                const uint32_t segLen = segEnd - segStart;

                                // Load x slice of length segLen
                                LocalTensor<float> xLocal = xQueue_.AllocTensor<float>();
                                const uint64_t xBase = indexX(n, ci, (uint32_t)xd, (uint32_t)xh, (uint32_t)(segStart - kw));
                                DataCopy(xLocal, xGm_[xBase], segLen);

                                // Multiply by scalar weight and accumulate into yLocal at offset (segStart - w0)
                                // Get weight scalar from wLocal
                                const uint32_t wFlat = (kd * Kh_ + kh) * Kw_ + kw;
                                // Note: The following GetValue is a typical helper to read a local scalar.
                                // If unavailable, it can be replaced by an alternative intrinsic.
                                float wScalar = wLocal.GetValue(wFlat);

                                Muls(xLocal, xLocal, wScalar, segLen);
                                // yLocal[dstOffset: dstOffset+segLen) += xLocal
                                const uint32_t dstOffset = segStart - w0;
                                Add(yLocal[dstOffset], yLocal[dstOffset], xLocal, segLen);

                                xQueue_.FreeTensor(xLocal);
                            }
                        }
                    }
                    wQueue_.FreeTensor(wLocal);
                }

                // Write back yLocal -> y GM at (n, co, do, ho, w0: w0+curLen)
                const uint64_t yBase = indexY(n, co, do_, ho, w0);
                DataCopy(yGm_[yBase], yLocal, curLen);
                yQueue_.FreeTensor(yLocal);
            }
        }
    }

private:
    __aicore__ inline uint64_t indexX(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        // [N, Cin, D, H, W]
        return (((uint64_t)n * Cin_ + c) * D_ + d) * H_ * W_ + (uint64_t)h * W_ + w;
    }

    __aicore__ inline uint64_t indexY(uint32_t n, uint32_t c, uint32_t d, uint32_t h, uint32_t w) const
    {
        // [N, Cout, Dout, Hout, Wout]
        return (((uint64_t)n * Cout_ + c) * Dout_ + d) * Hout_ * Wout_ + (uint64_t)h * Wout_ + w;
    }

private:
    TPipe pipe_;
    TQue<TPosition::VECIN,  X_BUFFER_NUM> xQueue_;
    TQue<TPosition::VECOUT, Y_BUFFER_NUM> yQueue_;
    TQue<TPosition::VECCALC, W_BUFFER_NUM> wQueue_;

    GlobalTensor<float> xGm_;
    GlobalTensor<float> wGm_;
    GlobalTensor<float> yGm_;

    uint32_t N_{0}, Cin_{0}, Cout_{0};
    uint32_t D_{0}, H_{0}, W_{0};
    uint32_t Kd_{0}, Kh_{0}, Kw_{0};
    uint32_t Dout_{0}, Hout_{0}, Wout_{0};
    uint32_t tileW_{0};
    uint64_t totalLines_{0};
};

extern "C" __global__ __aicore__ void conv_transposed_3d_asymmetric_input_square_kernel_custom(GM_ADDR x,
                                                                                               GM_ADDR weight,
                                                                                               GM_ADDR y,
                                                                                               GM_ADDR workspace,
                                                                                               GM_ADDR tiling)
{
    GET_TILING_DATA(tiling_data, tiling);
    CT3DTile t;
    t.N        = tiling_data.N;
    t.Cin      = tiling_data.Cin;
    t.Cout     = tiling_data.Cout;
    t.D        = tiling_data.D;
    t.H        = tiling_data.H;
    t.W        = tiling_data.W;
    t.Kd       = tiling_data.Kd;
    t.Kh       = tiling_data.Kh;
    t.Kw       = tiling_data.Kw;
    t.Dout     = tiling_data.Dout;
    t.Hout     = tiling_data.Hout;
    t.Wout     = tiling_data.Wout;
    t.tileW    = tiling_data.tileW;
    t.totalLines = tiling_data.totalLines;

    KernelConvTransposed3dAsymmetricInputSquareKernelCustom op;
    op.Init(x, weight, y, t);
    op.Process();
}
