host_tiling_src="""
#include <cstring>
#include "register/tilingdata_base.h"
#include "graph/tensor.h"
#include "graph/utils/type_utils.h"
#include "tiling/platform/platform_ascendc.h"
#include "tiling/matmul_api_tiling.h"

namespace optiling {

static inline uint32_t CeilDiv(uint32_t a, uint32_t b)
{
    return (a + b - 1u) / b;
}

static ge::graphStatus TilingFunc(gert::TilingContext *context)
{
    // Shapes:
    // x: [M, K]
    // w: [N, K] (will be transposed in kernel)
    // b: [N]
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) {
        return ge::GRAPH_FAILED;
    }

    uint64_t M = 1;
    for (size_t i = 0; i + 1 < x_shape->GetDimNum(); ++i) {
        M *= static_cast<uint64_t>(x_shape->GetDim(i));
    }
    const uint64_t K = static_cast<uint64_t>(x_shape->GetDim(x_shape->GetDimNum() - 1));
    const uint64_t N = static_cast<uint64_t>(w_shape->GetDim(0)); // out_features

    // Read attrs
    const gert::RuntimeAttrs *attrs = context->GetAttrs();
    const float *subtractValue = attrs->GetAttrPointer<float>(0);
    const float *multiplyValue = attrs->GetAttrPointer<float>(1);
    float sv = (subtractValue != nullptr) ? *subtractValue : 0.0f;
    float mv = (multiplyValue != nullptr) ? *multiplyValue : 1.0f;

    // Build tiling for single-core Matmul (NORMAL layout)
    auto ascendcPlatform = platform_ascendc::PlatformAscendC(context->GetPlatformInfo());
    matmul_tiling::MatmulApiTiling mmTiling(ascendcPlatform);
    mmTiling.SetAType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    mmTiling.SetBType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    mmTiling.SetCType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    mmTiling.SetBiasType(matmul_tiling::TPosition::GM, matmul_tiling::CubeFormat::ND, matmul_tiling::DataType::DT_FLOAT);
    mmTiling.SetShape(static_cast<uint32_t>(M), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    mmTiling.SetOrgShape(static_cast<uint32_t>(M), static_cast<uint32_t>(N), static_cast<uint32_t>(K));
    mmTiling.SetBias(true);
    mmTiling.SetBufferSpace(-1, -1, -1);

    optiling::TCubeTiling cubeTiling;
    int ret = mmTiling.GetTiling(cubeTiling);
    if (ret != 0) {
        return ge::GRAPH_FAILED;
    }

    // BlockDim: single-core execution for simplicity/fusion
    context->SetBlockDim(1);

    // Save tiling: [TCubeTiling | float subtractValue | float multiplyValue]
    auto *raw = context->GetRawTilingData();
    const size_t totalSize = sizeof(optiling::TCubeTiling) + sizeof(float) * 2;
    if (raw->GetCapacity() < totalSize) {
        return ge::GRAPH_FAILED;
    }
    uint8_t *buf = reinterpret_cast<uint8_t *>(raw->GetData());
    std::memcpy(buf, &cubeTiling, sizeof(cubeTiling));
    std::memcpy(buf + sizeof(cubeTiling), &sv, sizeof(float));
    std::memcpy(buf + sizeof(cubeTiling) + sizeof(float), &mv, sizeof(float));
    raw->SetDataSize(totalSize);

    // No workspace needed by this kernel
    size_t *ws = context->GetWorkspaceSizes(1);
    ws[0] = 0;

    return ge::GRAPH_SUCCESS;
}

} // namespace optiling

namespace ge {
static ge::graphStatus InferShape(gert::InferShapeContext *context)
{
    const gert::Shape *x_shape = context->GetInputShape(0);
    const gert::Shape *w_shape = context->GetInputShape(1);
    if (x_shape == nullptr || w_shape == nullptr) return GRAPH_FAILED;

    gert::Shape *y_shape = context->GetOutputShape(0);
    // Output: same as x with last dim replaced by w[0]
    std::vector<int64_t> dims;
    dims.reserve(x_shape->GetDimNum());
    for (size_t i = 0; i + 1 < x_shape->GetDimNum(); ++i) {
        dims.push_back(x_shape->GetDim(i));
    }
    dims.push_back(w_shape->GetDim(0));
    y_shape->SetDimNum(dims.size());
    for (size_t i = 0; i < dims.size(); ++i) {
        y_shape->SetDim(i, dims[i]);
    }
    return GRAPH_SUCCESS;
}

static ge::graphStatus InferDataType(gert::InferDataTypeContext *context)
{
    // All tensors are float
    context->SetOutputDataType(0, context->GetInputDataType(0));
    return GRAPH_SUCCESS;
}

kernel_src="""
#include "kernel_operator.h"
#include "lib/matmul_intf.h"

using AscendC::GlobalTensor;
using AscendC::LocalTensor;

extern "C" __global__ __aicore__ void matmul_subtract_multiply_relu_custom(
    GM_ADDR xGM, GM_ADDR wGM, GM_ADDR bGM, GM_ADDR yGM, GM_ADDR workspaceGM, GM_ADDR tilingGM)
{
    // Define Matmul types: NORMAL layout, B transposed (weight is [N, K])
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> aType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, true,  LayoutMode::NORMAL> bType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float, false, LayoutMode::NORMAL> cType;
    typedef matmul::MatmulType<AscendC::TPosition::GM, CubeFormat::ND, float> biasType;

    set_atomic_none();

    // Load TCubeTiling from tilingGM
    TCubeTiling tiling;
    auto tempTilingGM = (__gm__ uint32_t*)tilingGM;
    auto tempTiling = (uint32_t*)&tiling;
    for (int i = 0; i < static_cast<int>(sizeof(TCubeTiling) / sizeof(uint32_t)); ++i, ++tempTilingGM, ++tempTiling) {
        *tempTiling = *tempTilingGM;
    }
    // Read fused scalar attrs appended after TCubeTiling
    auto scalarPtr = (__gm__ float*)tempTilingGM; // points past TCubeTiling
    float subtractValue = *scalarPtr;
    float multiplyValue = *(scalarPtr + 1);

    // Initialize GM tensors
    AscendC::GlobalTensor<float> aGlobal;
    AscendC::GlobalTensor<float> bGlobal;
    AscendC::GlobalTensor<float> cGlobal;
    AscendC::GlobalTensor<float> biasGlobal;

    int32_t sizeA = tiling.ALayoutInfoB * tiling.singleCoreM * tiling.singleCoreK * static_cast<int32_t>(sizeof(float));
    int32_t sizeB = tiling.BLayoutInfoB * tiling.singleCoreK * tiling.singleCoreN * static_cast<int32_t>(sizeof(float));
    int32_t sizeC = tiling.CLayoutInfoB * tiling.singleCoreM * tiling.singleCoreN * static_cast<int32_t>(sizeof(float));
    int32_t sizeBias = tiling.CLayoutInfoB * tiling.singleCoreN * static_cast<int32_t>(sizeof(float));

    aGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(xGM), sizeA);
    bGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(wGM), sizeB);
    cGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(yGM), sizeC);
    biasGlobal.SetGlobalBuffer(reinterpret_cast<__gm__ float*>(bGM), sizeBias);

    // Create and init Matmul instance
    constexpr static MatmulConfig MM_CFG = GetNormalConfig(false, false, false, BatchMode::BATCH_LESS_THAN_L1);
    matmul::Matmul<aType, bType, cType, biasType, MM_CFG> mm1;
    AscendC::TPipe pipe;
    REGIST_MATMUL_OBJ(&pipe, GetSysWorkSpacePtr(), mm1);
    mm1.Init(&tiling);
    mm1.SetTensorA(aGlobal, 0);
    mm1.SetTensorB(bGlobal, 1);
    mm1.SetWorkspace(workspaceGM, 0);
    if (tiling.isBias) {
        mm1.SetBias(biasGlobal);
    }

    // Execute Matmul (single-batch NORMAL scenario)
    mm1.IterateBatch(cGlobal, 1, 1, false);

    // Post-ops: y = ReLU( (y - subtractValue) * multiplyValue )
    // Process in tiles to fit UB
    constexpr uint32_t TILE_ELEMS = 8192; // conservative UB usage
    AscendC::TPipe vpipe;
    AscendC::TQue<AscendC::TPosition::VECIN, 2> inQ;
    AscendC::TQue<AscendC::TPosition::VECOUT, 2> outQ;
    vpipe.InitBuffer(inQ, 2, TILE_ELEMS * sizeof(float));
    vpipe.InitBuffer(outQ, 2, TILE_ELEMS * sizeof(float));

    const uint32_t totalElems = static_cast<uint32_t>(sizeC / sizeof(float));
    for (uint32_t base = 0; base < totalElems; base += TILE_ELEMS) {
        uint32_t len = ((base + TILE_ELEMS) <= totalElems) ? TILE_ELEMS : (totalElems - base);

        LocalTensor<float> xLocal = inQ.AllocTensor<float>();
        AscendC::DataCopy(xLocal, cGlobal[base], len);

        LocalTensor<float> yLocal = outQ.AllocTensor<float>();

        // yLocal = xLocal - subtractValue
        AscendC::Adds(yLocal, xLocal, -subtractValue, len);
        // yLocal = yLocal * multiplyValue
        AscendC::Muls(yLocal, yLocal, multiplyValue, len);
        // yLocal = max(yLocal, 0.0f)
        AscendC::Maxs(yLocal, yLocal, 0.0f, len);

        outQ.EnQue<float>(yLocal);
        inQ.EnQue<float>(xLocal);

        LocalTensor<float> yOut = outQ.DeQue<float>();
        AscendC::DataCopy(cGlobal[base], yOut, len);
        outQ.FreeTensor(yOut);

        LocalTensor<float> xIn = inQ.DeQue<float>();
        inQ.FreeTensor(xIn);
    }
}
"""

python_bind_src="""
#include <torch/library.h>
#include <torch/extension.h>
#include "pytorch_npu_helper.hpp"

at::Tensor matmul_sub_mul_relu_impl_npu(const at::Tensor& x,
                                        const at::Tensor& w,
                                        const at::Tensor& b,
                                        const at::Tensor& sub,
                                        const at::Tensor& mul) {
    // Extract scalar attrs
    double subtract_value = sub.item<double>();
    double multiply_value = mul.item<double>();
    // Output
    // y shape: x[..., -1] replaced by w.size(0)
    auto out_sizes = x.sizes().vec();
    if (!out_sizes.empty()) {
        out_sizes.back() = w.size(0);
    }
    at::Tensor y = at::empty(out_sizes, x.options());
    // Call custom operator
    EXEC_NPU_CMD(aclnnMatmulSubtractMultiplyReluCustom, x, w, b, subtract_value, multiply_value, y);
    return y;
}

TORCH_LIBRARY_IMPL(myops, PrivateUse1, m) {
    m.impl("matmul_subtract_multiply_relu_custom", &matmul_sub_mul_relu_impl_npu);
}

PYBIND11_MODULE(TORCH_EXTENSION_NAME, m) {
    m.def("matmul_subtract_multiply_relu_custom", &matmul_sub_mul_relu_impl_npu,
          "Fused Linear - Subtract - Multiply - ReLU (MatmulSubtractMultiplyReluCustom on NPU)");
}
